CLASS Solution  
    FUNCTION findStrobogrammatic(n)  
        FUNCTION dfs(u)  
            IF u EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF u EQUALS one  
                RETURN list containing string zero string one string eight  
            END IF  
            SET ans TO empty list  
            FOR each element v IN result of CALLING dfs with argument u MINUS two  
                FOR each pair of characters l and r IN list containing string one one string eight eight string six nine string nine six  
                    APPEND concatenation of character l with string v with character r TO list ans  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND concatenation of string zero with string v with string zero TO list ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN result of CALLING dfs with argument n  
    END FUNCTION  
END CLASS
