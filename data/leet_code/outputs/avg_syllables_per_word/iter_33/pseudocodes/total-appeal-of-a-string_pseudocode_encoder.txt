CLASS Solution  
    FUNCTION appealSum WITH PARAMETER s  
        SET last TO a list of twenty six elements each SET TO negative one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each pair of i and char FROM the enumeration of s  
            SET index TO the numerical code of char MINUS the numerical code of the lowercase letter a  
            SET current_appeal TO current_appeal PLUS i MINUS element at position index of last  
            SET total_appeal TO total_appeal PLUS current_appeal  
            SET element at position index of last TO i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
