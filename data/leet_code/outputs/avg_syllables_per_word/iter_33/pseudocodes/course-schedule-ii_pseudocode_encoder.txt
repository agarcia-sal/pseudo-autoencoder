CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO an empty mapping from each course from zero TO numCourses MINUS one TO an empty list  
        SET in_degrees TO a list of zeros with length numCourses  
        
        FOR each pair dest AND src IN prerequisites  
            APPEND dest TO the list at key src IN adj_list  
            INCREMENT the element at position dest IN in_degrees BY one  
        END FOR  
        
        SET queue TO a new double ended queue with each course from zero TO numCourses MINUS one WHERE the element at position course IN in_degrees EQUALS zero  
        SET topological_order TO an empty list  
        
        WHILE queue is not empty  
            SET current TO the element removed from the left end of queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN the list at key current IN adj_list  
                DECREMENT the element at position neighbor IN in_degrees BY one  
                IF the element at position neighbor IN in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the length of topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
