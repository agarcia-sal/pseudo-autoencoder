CLASS Solution  
    FUNCTION minBuildTime(blocks AS list OF integers split AS integer)  
        TRANSFORM blocks INTO a structure where the smallest element can be efficiently extracted repeatedly  
        WHILE the count of elements in blocks GREATER THAN one  
            EXTRACT the smallest element from blocks and ASSIGN it to first  
            EXTRACT the next smallest element from blocks and ASSIGN it to second  
            SET combined_time TO split PLUS the value of second  
            INSERT combined_time INTO blocks maintaining the efficient smallest element extraction structure  
        END WHILE  
        RETURN the element at position one of blocks  
    END FUNCTION  
END CLASS
