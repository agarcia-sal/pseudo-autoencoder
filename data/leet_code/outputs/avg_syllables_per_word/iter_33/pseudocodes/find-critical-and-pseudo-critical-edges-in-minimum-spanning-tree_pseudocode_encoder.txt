CLASS UnionFind  
    FUNCTION __init__(self n)  
        SET self.parent TO list of integers from zero TO n MINUS one  
        SET self.rank TO list of integer one repeated n times  
    END FUNCTION  

    FUNCTION find(self u)  
        IF element at position u of self.parent NOT EQUALS u THEN  
            SET element at position u of self.parent TO self.find(element at position u of self.parent)  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self u v)  
        SET rootU TO self.find(u)  
        SET rootV TO self.find(v)  

        IF rootU NOT EQUALS rootV THEN  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank THEN  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank THEN  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self n edges)  
        FOR index i AND element edge IN elements of edges WITH index  
            APPEND i TO edge  
        END FOR  

        SORT edges BY weight which is element at position three of each edge  

        FUNCTION mst(exclude=None include=None)  
            SET uf TO new UnionFind with parameter n  
            SET weight TO zero  

            IF include IS NOT None THEN  
                CALL uf.union with parameters element at position zero of include AND element at position one of include  
                INCREMENT weight BY element at position two of include  
            END IF  

            FOR each element edge IN edges  
                IF element at position three of edge EQUALS exclude THEN  
                    CONTINUE TO next iteration  
                END IF  

                IF uf.union(element at position zero of edge element at position one of edge) THEN  
                    INCREMENT weight BY element at position two of edge  
                END IF  
            END FOR  

            SET root TO uf.find(zero)  
            FOR index i FROM zero TO n MINUS one  
                IF uf.find(i) NOT EQUALS root THEN  
                    RETURN infinity  
                END IF  
            END FOR  

            RETURN weight  
        END FUNCTION  

        SET mst_weight TO result of calling mst with no parameters  
        SET critical TO empty list  
        SET pseudo_critical TO empty list  

        FOR each element edge IN edges  
            IF result of calling mst with exclude parameter as element at position three of edge GREATER THAN mst_weight THEN  
                APPEND element at position three of edge TO critical  
            ELSE IF result of calling mst with include parameter as tuple of element zero edge element one edge element two edge EQUALS mst_weight THEN  
                APPEND element at position three of edge TO pseudo_critical  
            END IF  
        END FOR  

        RETURN list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
