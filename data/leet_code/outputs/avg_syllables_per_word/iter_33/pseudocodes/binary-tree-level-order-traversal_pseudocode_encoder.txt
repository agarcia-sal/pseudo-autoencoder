CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO A new TreeNode INITIALIZED WITH the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element REMOVED FROM the FRONT OF queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.left TO A new TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.right TO A new TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS TO q.val THEN  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH p.left AND q.left AND is_same_tree WITH p.right AND q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root IS none THEN  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a double-ended queue INITIALIZED WITH root  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR each element FROM one TO level_size  
                SET node TO the element REMOVED FROM the FRONT OF queue  
                APPEND node.val TO current_level  
                IF node.left IS not none THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS not none THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
