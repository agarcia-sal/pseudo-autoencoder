CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO an empty set  
        FOR each mine IN mines  
            ADD the tuple formed by the elements of mine TO mines_set  
        END FOR  
        
        SET dp TO a list containing n elements each of which is a list containing n elements each of which is a list of four zeros representing directions right down left up  
        
        SET directions TO a list containing the pairs representing zero one one zero zero minus one minus one zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair consisting of i and j is NOT IN mines_set THEN  
                    IF j GREATER THAN zero THEN  
                        SET element at position zero of element at position j of element at position i of dp TO the value of element at position zero of element at position j MINUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i GREATER THAN zero THEN  
                        SET element at position one of element at position j of element at position i of dp TO the value of element at position one of element at position j of element at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the pair consisting of i and j is NOT IN mines_set THEN  
                    IF j LESS THAN n MINUS one THEN  
                        SET element at position two of element at position j of element at position i of dp TO the value of element at position two of element at position j PLUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i LESS THAN n MINUS one THEN  
                        SET element at position three of element at position j of element at position i of dp TO the value of element at position three of element at position j of element at position i PLUS one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum value among the elements of element at position j of element at position i of dp  
                    IF plus_sign_order GREATER THAN result THEN  
                        SET result TO plus_sign_order  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
