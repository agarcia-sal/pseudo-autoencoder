CLASS Solution  
    FUNCTION repeatedStringMatch(a, b)  
        SET minimum repetitions TO the ceiling of the length of b DIVIDED BY the length of a  
        IF the string b is contained within the concatenation of the string a repeated minimum repetitions times  
            RETURN minimum repetitions  
        END IF  
        IF the string b is contained within the concatenation of the string a repeated the sum of minimum repetitions PLUS one times  
            RETURN the sum of minimum repetitions PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
