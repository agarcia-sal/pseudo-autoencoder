CLASS Solution  
    FUNCTION countPairs(nums k)  
        SET gcd_count TO empty map with default integer zero  
        
        FOR each num IN nums  
            SET current_gcd TO greatest common divisor of num AND k  
            INCREMENT value at key current_gcd in gcd_count BY one  
        END FOR  
        
        SET total_pairs TO zero  
        SET gcd_values TO list of keys in gcd_count  
        SET n TO length of gcd_values  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET gcd1 TO element at position i of gcd_values  
                SET gcd2 TO element at position j of gcd_values  
                SET product TO gcd1 MULTIPLIED BY gcd2  
                SET remainder TO product MODULO k  
                IF remainder EQUALS zero  
                    IF i EQUALS j  
                        SET count TO value at key gcd1 in gcd_count  
                        SET pairs_for_this TO count MULTIPLIED BY count MINUS one DIVIDED BY two  
                        INCREMENT total_pairs BY pairs_for_this  
                    ELSE  
                        SET count1 TO value at key gcd1 in gcd_count  
                        SET count2 TO value at key gcd2 in gcd_count  
                        INCREMENT total_pairs BY count1 MULTIPLIED BY count2  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_pairs  
    END FUNCTION  
END CLASS
