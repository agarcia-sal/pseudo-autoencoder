CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO empty list  
        
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack is greater than zero AND asteroid is less than zero AND element at position the LENGTH OF stack MINUS one of stack is greater than zero  
                IF element at position the LENGTH OF stack MINUS one of stack is less than negative asteroid  
                    REMOVE element at position the LENGTH OF stack MINUS one of stack FROM stack  
                    CONTINUE to the next iteration of the WHILE loop  
                ELSE IF element at position the LENGTH OF stack MINUS one of stack EQUALS negative asteroid  
                    REMOVE element at position the LENGTH OF stack MINUS one of stack FROM stack  
                END IF  
                BREAK out of the WHILE loop  
            END WHILE  
            IF the WHILE loop did not BREAK  
                APPEND asteroid TO stack  
            END IF  
        END FOR  
        
        RETURN stack  
    END FUNCTION  
END CLASS
