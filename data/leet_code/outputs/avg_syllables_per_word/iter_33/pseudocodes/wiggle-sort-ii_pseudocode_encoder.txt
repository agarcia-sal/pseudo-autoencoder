CLASS Solution  
    FUNCTION wiggleSort(nums)  
        CALL sort nums in ascending order  
        SET sorted_nums TO a new list containing each element of nums in order  
        SET n TO the number of elements in nums  
        SET left TO the integer division of (n MINUS one) BY two  
        SET right TO n MINUS one  
        FOR index FROM zero TO n MINUS one  
            IF the remainder of index DIVIDED BY two EQUALS zero  
                SET element at position index of nums TO element at position left of sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET element at position index of nums TO element at position right of sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
