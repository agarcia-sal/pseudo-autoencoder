CLASS Solution  
    FUNCTION findPairs(nums k)  
        IF k LESS THAN zero  
            RETURN zero  
        END IF  
        SET num_counts TO empty dictionary  
        FOR each num IN nums  
            SET current_count TO value at key num in num_counts OR zero if key does not exist  
            SET value at key num in num_counts TO current_count PLUS one  
        END FOR  
        SET unique_pairs TO zero  
        FOR each num IN keys of num_counts  
            IF k EQUALS zero  
                IF value at key num in num_counts GREATER THAN one  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF key equal to num PLUS k EXISTS in num_counts  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
