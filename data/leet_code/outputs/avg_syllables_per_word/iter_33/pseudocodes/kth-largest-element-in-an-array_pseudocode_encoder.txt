CLASS Solution  
    FUNCTION findKthLargest(nums k)  
        SET min_heap TO an empty list  
        FOR each number IN nums  
            PUSH the number INTO the min_heap using heap insertion  
            IF the LENGTH OF min_heap GREATER THAN k  
                REMOVE the smallest element FROM min_heap using heap removal  
            END IF  
        END FOR  
        RETURN the element at position one of min_heap  
    END FUNCTION  
END CLASS
