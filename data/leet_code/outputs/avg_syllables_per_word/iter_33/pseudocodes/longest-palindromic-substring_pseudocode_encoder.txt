CLASS Solution  
    FUNCTION longestPalindrome WITH parameter s  
        SET n TO the LENGTH OF s  
        SET f TO a two dimensional list with n rows and n columns where each element is TRUE  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM the value of n MINUS two DOWN TO zero  
            FOR j FROM the value of i PLUS one TO the value of n MINUS one  
                SET the element at position i and j of f TO FALSE  
                IF the element at position i of s EQUALS the element at position j of s  
                    SET the element at position i and j of f TO the element at position i PLUS one and j MINUS one of f  
                    IF the element at position i and j of f AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the substring of s from position k TO position k PLUS mx MINUS one  
    END FUNCTION  
END CLASS
