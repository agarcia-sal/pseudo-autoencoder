CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the length of values AND the element at position i of values is not nothing  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values is not nothing  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN True  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node EQUALS nothing  
                RETURN zero  
            END IF  
            SET current_sum TO the val attribute of node PLUS subtree_sum(the left attribute of node) PLUS subtree_sum(the right attribute of node)  
            INCREMENT the element at key current_sum in count BY one  
            RETURN current_sum  
        END FUNCTION  
        
        SET count TO an empty mapping with integer default values  
        CALL subtree_sum with root  
        SET max_freq TO the maximum value among all values in count or zero if count is empty  
        SET result TO an empty list  
        FOR each key and frequency IN count  
            IF frequency EQUALS max_freq  
                APPEND key TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
