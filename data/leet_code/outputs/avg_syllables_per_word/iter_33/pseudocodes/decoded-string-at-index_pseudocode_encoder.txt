CLASS Solution  
    FUNCTION decodeAtIndex(s string_parameter k integer_parameter) RETURNS string  
        SET length TO zero  
        FOR each character IN s  
            IF character IS DIGIT  
                SET length TO length MULTIPLIED BY the INTEGER VALUE OF character  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each character IN the REVERSE ORDER OF s  
            SET k TO k MODULO length  
            IF k EQUALS zero AND character IS ALPHABETIC LETTER  
                RETURN character  
            END IF  
            IF character IS DIGIT  
                SET length TO length DIVIDED BY the INTEGER VALUE OF character  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
