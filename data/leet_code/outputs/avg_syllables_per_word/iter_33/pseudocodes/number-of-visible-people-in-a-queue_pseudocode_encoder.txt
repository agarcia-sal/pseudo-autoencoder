CLASS Solution  
    FUNCTION canSeePersonsCount WITH parameter heights  
        SET n TO the LENGTH OF heights  
        SET answer TO list of zeroes of LENGTH n  
        SET stack TO empty list  
        FOR index i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position element at the LAST POSITION OF stack of heights IS LESS THAN the element at position i of heights  
                SET popped_index TO the element REMOVED FROM the LAST POSITION OF stack  
                INCREMENT the element at position popped_index of answer BY one  
            END WHILE  
            IF the LENGTH OF stack IS GREATER THAN zero  
                INCREMENT the element at position the LAST POSITION OF stack of answer BY one  
            END IF  
            APPEND i TO stack  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
