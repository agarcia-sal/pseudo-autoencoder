CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a new mapping with default value as empty list  
        FOR each pair of departure location and arrival location IN the tickets sorted in descending lexical order  
            APPEND the arrival location TO the list associated with the departure location in graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit(airport)  
            WHILE the list associated with airport in graph is not empty  
                CALL visit with the last element removed from the list associated with airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit with the string that is the abbreviation for John F Kennedy airport  
        RETURN the reverse order of itinerary  
    END FUNCTION  
END CLASS
