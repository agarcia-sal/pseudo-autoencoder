CLASS Solution  
    FUNCTION rangeBitwiseAnd(left INTEGER PARAMETER right INTEGER PARAMETER) RETURNS INTEGER  
        SET shift TO zero  
        WHILE left LESS THAN right  
            SET left TO the result of left DIVIDED BY two discarding remainder  
            SET right TO the result of right DIVIDED BY two discarding remainder  
            INCREMENT shift BY one  
        END WHILE  
        SET result TO left MULTIPLIED BY two RAISED TO THE POWER OF shift  
        RETURN result  
    END FUNCTION  
END CLASS
