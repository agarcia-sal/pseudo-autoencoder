CLASS Solution  
    FUNCTION shortestDistance(maze, start, destination)  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero of maze  
        SET directions TO the LIST OF pairs each consisting of an integer one and integer zero, integer minus one and integer zero, integer zero and integer one, integer zero and integer minus one  
        SET start TO the TUPLE formed from start  
        SET destination TO the TUPLE formed from destination  
        SET visited TO an EMPTY set  
        SET pq TO a LIST containing a SINGLE element which is a PAIR consisting of integer zero and start  
        
        WHILE pq is NOT empty  
            SET dist TO the first element of the result of removing the smallest element from pq heap  
            SET x TO the first element of the second element of the same removed pair  
            SET y TO the second element of the second element of the same removed pair  
            
            IF x EQUALS the first element of destination AND y EQUALS the second element of destination  
                RETURN dist  
            END IF  
            
            IF the PAIR consisting of x and y IS IN visited  
                CONTINUE to the next iteration of the while loop  
            END IF  
            
            ADD the PAIR consisting of x and y TO visited  
            
            FOR each pair dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the element at position new_x PLUS dx of maze AND position new_y PLUS dy IS EQUALS zero  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF the PAIR consisting of new_x and new_y IS NOT IN visited  
                    ADD to pq heap the PAIR consisting of dist PLUS count and the PAIR consisting of new_x and new_y  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN integer minus one  
    END FUNCTION  
END CLASS
