CLASS Solution  
    FUNCTION isNStraightHand(hand groupSize)  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN False  
        END IF  

        SET count TO a collection of frequencies of elements in hand  
        SET unique_cards TO the list of distinct elements in count sorted in ascending order  

        FOR each card IN unique_cards  
            IF the value associated with card in count GREATER THAN zero THEN  
                SET needed TO the value associated with card in count  
                FOR index FROM zero TO groupSize MINUS one  
                    IF the value associated with card PLUS index in count LESS THAN needed THEN  
                        RETURN False  
                    END IF  
                    DECREMENT the value associated with card PLUS index in count BY needed  
                END FOR  
            END IF  
        END FOR  

        RETURN True  
    END FUNCTION  
END CLASS
