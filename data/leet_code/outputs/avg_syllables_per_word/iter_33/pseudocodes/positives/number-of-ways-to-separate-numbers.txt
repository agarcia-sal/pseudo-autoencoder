CLASS Solution  
    FUNCTION numberOfCombinations(num)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET n TO the LENGTH OF num  
        SET memo TO an empty dictionary  
        
        FUNCTION dp(index, prev_len)  
            IF index EQUALS n THEN  
                RETURN one  
            END IF  
            IF the tuple of index AND prev_len EXISTS AS a key IN memo THEN  
                RETURN the value in memo ASSOCIATED WITH the tuple of index AND prev_len  
            END IF  
            
            SET count TO zero  
            FOR length FROM one TO the LENGTH OF num MINUS index PLUS one  
                IF the element at position index of num EQUALS the character zero THEN  
                    BREAK the FOR loop  
                END IF  
                
                SET current_num TO the substring from position index TO position index PLUS length MINUS one of num  
                
                IF prev_len EQUALS zero OR current_num IS GREATER THAN OR EQUAL TO the substring from position index MINUS prev_len TO position index MINUS one of num THEN  
                    SET count TO the SUM of count AND the result of dp AT index PLUS length AND length MODULO MOD  
                END IF  
            END FOR  
            
            ASSIGN the value of count TO memo AT key tuple of index AND prev_len  
            RETURN count  
        END FUNCTION  
        
        RETURN dp AT zero AND zero  
    END FUNCTION  
END CLASS
