CLASS Solution  
    FUNCTION uniquePathsWithObstacles(obstacleGrid)  
        IF obstacleGrid EQUALS empty OR element at position zero at position zero of obstacleGrid EQUALS one  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF obstacleGrid  
        SET n TO the LENGTH OF element at position zero of obstacleGrid  
        SET dp TO a new two dimensional list with m rows each containing n zeros  
        SET element at position zero at position zero of dp TO one  
        FOR j FROM one TO n MINUS one  
            IF element at position zero at position j of obstacleGrid EQUALS zero  
                SET element at position zero at position j of dp TO element at position zero at position j MINUS one of dp  
            END IF  
        END FOR  
        FOR i FROM one TO m MINUS one  
            IF element at position i at position zero of obstacleGrid EQUALS zero  
                SET element at position i at position zero of dp TO element at position i MINUS one at position zero of dp  
            END IF  
        END FOR  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF element at position i at position j of obstacleGrid EQUALS zero  
                    SET element at position i at position j of dp TO element at position i MINUS one at position j of dp PLUS element at position i at position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one at position n MINUS one of dp  
    END FUNCTION  
END CLASS
