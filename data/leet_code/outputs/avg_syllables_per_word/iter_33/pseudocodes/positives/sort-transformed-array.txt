CLASS Solution  
    FUNCTION sortTransformedArray(nums WITH TYPE List of integer, a WITH TYPE integer, b WITH TYPE integer, c WITH TYPE integer) RETURNS List of integer  
        FUNCTION quadratic(x)  
            RETURN a MULTIPLIED BY x MULTIPLIED BY x PLUS b MULTIPLIED BY x PLUS c  
        END FUNCTION  
        
        SET n TO the LENGTH OF nums  
        SET result TO a new list of integer zeros with LENGTH n  
        SET left TO zero  
        SET right TO n MINUS one  
        
        IF a GREATER THAN OR EQUAL TO zero  
            SET index TO n MINUS one  
            WHILE left LESS THAN OR EQUAL TO right  
                IF quadratic of element at position left of nums GREATER THAN quadratic of element at position right of nums  
                    SET element at position index of result TO quadratic of element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET element at position index of result TO quadratic of element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                DECREMENT index BY one  
            END WHILE  
        ELSE  
            SET index TO zero  
            WHILE left LESS THAN OR EQUAL TO right  
                IF quadratic of element at position left of nums LESS THAN quadratic of element at position right of nums  
                    SET element at position index of result TO quadratic of element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET element at position index of result TO quadratic of element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                INCREMENT index BY one  
            END WHILE  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
