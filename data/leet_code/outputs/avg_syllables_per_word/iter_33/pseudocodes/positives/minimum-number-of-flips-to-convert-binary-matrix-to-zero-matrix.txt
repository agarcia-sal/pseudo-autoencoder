CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix x y)  
            SET directions TO list of pairs negative one zero PLUS one zero ZERO negative one ZERO one ZERO zero  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n  
                    SET element at position nx of matrix element at position ny TO one MINUS element at position nx of matrix element at position ny  
                END IF  
            END FOR  
        END FUNCTION  

        FUNCTION matrix_to_tuple(matrix)  
            SET result TO empty list  
            FOR each row IN matrix  
                APPEND tuple converted from row TO result  
            END FOR  
            RETURN tuple converted from result  
        END FUNCTION  

        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET target TO tuple converted from list built by repeating tuple converted from list built by repeating zero for n times for m times  
        SET initial_state TO matrix_to_tuple of mat  
        SET queue TO list containing pair initial_state and zero  
        SET visited TO empty set  
        ADD initial_state TO visited  

        WHILE queue is not empty  
            REMOVE element at position zero FROM queue AND ASSIGN to current_state and steps  
            IF current_state EQUALS target  
                RETURN steps  
            END IF  

            SET current_matrix TO list built by converting each row in current_state to list  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO deep copy of current_matrix  
                    CALL flip ON next_matrix with parameters i and j  
                    SET next_state TO matrix_to_tuple of next_matrix  
                    IF next_state NOT IN visited  
                        ADD next_state TO visited  
                        APPEND pair next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
