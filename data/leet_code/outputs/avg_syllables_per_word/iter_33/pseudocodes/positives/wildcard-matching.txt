CLASS Solution  
    FUNCTION isMatch(s parameter s TO string, p parameter p TO string) RETURNS boolean  
        SET dp TO list of lists of boolean with the number of elements equal to the length of s PLUS one, each element is a list of boolean with the number of elements equal to the length of p PLUS one, all elements initialized to False  
        
        SET element at position zero of element at position zero of dp TO True  
        
        FOR j FROM one TO the length of p  
            IF element at position j MINUS one of p EQUALS the character represented by asterisk  
                SET element at position zero of element at position j of dp TO element at position zero of element at position j MINUS one of dp  
            END IF  
        END FOR  
        
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF element at position j MINUS one of p EQUALS the character represented by asterisk  
                    SET element at position i of element at position j of dp TO element at position i MINUS one of element at position j of dp OR element at position i of element at position j MINUS one of dp  
                ELSE IF element at position j MINUS one of p EQUALS the character represented by question mark OR element at position i MINUS one of s EQUALS element at position j MINUS one of p  
                    SET element at position i of element at position j of dp TO element at position i MINUS one of element at position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position the length of s of element at position the length of p of dp  
    END FUNCTION  
END CLASS
