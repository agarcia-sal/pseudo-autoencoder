CLASS Solution  
    FUNCTION maxSum(nums1 nums1 LIST OF INTEGER nums2 nums2 LIST OF INTEGER) RETURNS INTEGER  
        SET i TO zero  
        SET j TO zero  
        SET sum1 TO zero  
        SET sum2 TO zero  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS one  
        
        WHILE i LESS THAN THE LENGTH OF nums1 AND j LESS THAN THE LENGTH OF nums2  
            IF element at position i of nums1 LESS THAN element at position j of nums2  
                INCREMENT sum1 BY element at position i of nums1  
                INCREMENT i BY one  
            ELSE IF element at position i of nums1 GREATER THAN element at position j of nums2  
                INCREMENT sum2 BY element at position j of nums2  
                INCREMENT j BY one  
            ELSE  
                SET sum1 TO the GREATER VALUE BETWEEN sum1 AND sum2 PLUS element at position i of nums1  
                SET sum2 TO sum1  
                INCREMENT i BY one  
                INCREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i LESS THAN THE LENGTH OF nums1  
            INCREMENT sum1 BY element at position i of nums1  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE j LESS THAN THE LENGTH OF nums2  
            INCREMENT sum2 BY element at position j of nums2  
            INCREMENT j BY one  
        END WHILE  
        
        RETURN the GREATER VALUE BETWEEN sum1 AND sum2 MODULO MOD  
    END FUNCTION  
END CLASS
