CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN list containing zero  
        END IF  
        SET prev_gray TO CALL grayCode WITH argument n MINUS one ON self  
        SET mask TO one MULTIPLIED BY two RAISED TO THE POWER OF n MINUS one  
        SET current_gray TO prev_gray APPENDED WITH list built by FOR each num IN reversed prev_gray CREATE mask OR num  
        RETURN current_gray  
    END FUNCTION  
END CLASS
