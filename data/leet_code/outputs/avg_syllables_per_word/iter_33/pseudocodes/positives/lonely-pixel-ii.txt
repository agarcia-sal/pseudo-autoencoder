CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture EQUALS empty OR element at position zero of picture EQUALS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO a list of zeros with LENGTH m  
        SET col_count TO a list of zeros with LENGTH n  
        SET row_patterns TO a default dictionary of integers  
        FOR r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of elements of element at position r of picture as strings joined in sequence  
            INCREMENT the count of pattern in row_patterns BY one  
            FOR c FROM zero TO n MINUS one  
                IF the element at position c of element at position r of picture EQUALS black pixel  
                    INCREMENT element at position r of row_count BY one  
                    INCREMENT element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        SET lonely_pixels TO zero  
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF the element at position c of element at position r of picture EQUALS black pixel  
                    SET pattern TO the concatenation of elements of element at position r of picture as strings joined in sequence  
                    IF element at position r of row_count EQUALS target AND element at position c of col_count EQUALS target AND the count of pattern in row_patterns EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
