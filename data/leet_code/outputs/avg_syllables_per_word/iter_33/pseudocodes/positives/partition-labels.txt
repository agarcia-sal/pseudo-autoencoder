CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO empty mapping from character to position  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position index of s  
            SET last_occurrence at key character TO index  
        END FOR  
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position index of s  
            IF end LESS THAN last_occurrence at key character THEN  
                SET end TO last_occurrence at key character  
            END IF  
            IF index EQUALS end THEN  
                APPEND end MINUS start PLUS one TO partition_sizes  
                SET start TO index PLUS one  
            END IF  
        END FOR  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
