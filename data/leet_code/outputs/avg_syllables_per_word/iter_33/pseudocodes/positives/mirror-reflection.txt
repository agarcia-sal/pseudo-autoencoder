CLASS Solution  
    FUNCTION mirrorReflection(p, q)  
        DEFINE lcm AS the least common multiple of p AND q  
        SET vertical_reflections TO lcm DIVIDED BY p  
        SET horizontal_reflections TO lcm DIVIDED BY q  
        IF the remainder when vertical_reflections is divided by two EQUALS zero  
            RETURN zero  
        ELSE IF the remainder when horizontal_reflections is divided by two EQUALS one  
            RETURN one  
        ELSE  
            RETURN two  
        END IF  
    END FUNCTION  
END CLASS
