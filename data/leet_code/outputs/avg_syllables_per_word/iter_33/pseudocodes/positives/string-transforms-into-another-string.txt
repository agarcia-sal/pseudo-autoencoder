CLASS Solution  
    FUNCTION canConvert(str1 str2)  
        IF str1 EQUALS str2  
            RETURN True  
        END IF  
        SET conversion_map TO empty mapping  
        FOR each pair of char1 AND char2 IN corresponding elements of str1 AND str2  
            IF char1 IN keys of conversion_map  
                IF value at key char1 of conversion_map NOT EQUALS char2  
                    RETURN False  
                END IF  
            ELSE  
                SET value at key char1 of conversion_map TO char2  
            END IF  
        END FOR  
        IF the LENGTH OF the set of all unique characters in str2 EQUALS twenty six  
            RETURN False  
        END IF  
        RETURN True  
    END FUNCTION  
END CLASS
