CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO a sorted list of the first elements of each pair in flowers  
        SET end_times TO a sorted list of the second elements of each pair in flowers  
        SET result TO an empty list  
        FOR each person IN people  
            SET started TO the count of elements in start_times that are less than OR EQUAL TO person  
            SET ended TO the count of elements in end_times that are strictly LESS THAN person  
            SET bloom_count TO started MINUS ended  
            APPEND bloom_count TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
