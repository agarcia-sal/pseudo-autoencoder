CLASS Solution  
    FUNCTION characterReplacement(s parameter, k parameter)  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        CREATE count AS a mapping from character to integer with default zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the value at key element at position right of s in count BY one  
            IF the value at key element at position right of s in count GREATER THAN max_count  
                SET max_count TO the value at key element at position right of s in count  
            END IF  
            
            IF the LENGTH OF the window defined by right MINUS left PLUS one MINUS max_count GREATER THAN k  
                DECREMENT the value at key element at position left of s in count BY one  
                INCREMENT left BY one  
            END IF  
            
            IF the LENGTH OF the window defined by right MINUS left PLUS one GREATER THAN max_length  
                SET max_length TO the LENGTH OF the window defined by right MINUS left PLUS one  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
