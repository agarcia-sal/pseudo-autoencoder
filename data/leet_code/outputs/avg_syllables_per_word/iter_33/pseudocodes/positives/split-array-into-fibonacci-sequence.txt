CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR each end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF the element at position start of num EQUALS character zero AND end GREATER THAN start PLUS one  
                    CONTINUE to the next iteration  
                END IF  
                SET next_num TO the integer value of the substring from position start TO position end MINUS one of num  
                IF next_num GREATER THAN OR EQUAL TO two RAISED TO THE POWER OF thirty ONE  
                    BREAK the loop  
                END IF  
                IF the LENGTH OF path LESS THAN two OR next_num EQUALS the element at position the LENGTH OF path MINUS one of path PLUS the element at position the LENGTH OF path MINUS two of path  
                    SET result TO the invocation of backtrack with arguments end and the concatenation of path and next_num  
                    IF result IS NOT EMPTY  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN an empty list  
        END FUNCTION  
        
        RETURN the invocation of backtrack with arguments zero and an empty list  
    END FUNCTION  
END CLASS
