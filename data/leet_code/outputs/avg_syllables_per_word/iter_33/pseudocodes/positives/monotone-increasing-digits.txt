CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO list of each character in the string representation of n  
        SET length TO the length of digits  
        SET marker TO length  
        FOR i FROM length MINUS one TO one stepping backwards by one  
            IF the element at position i of digits LESS THAN the element at position i MINUS one of digits  
                SET element at position i MINUS one of digits TO the string representation of the integer value of element at position i MINUS one of digits MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one  
            SET element at position i of digits TO the character representing the digit nine  
        END FOR  
        SET result_string TO concatenation of all elements in digits  
        RETURN the integer value of result_string  
    END FUNCTION  
END CLASS
