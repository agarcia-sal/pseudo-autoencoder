CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO list of zeros with length batchSize  
        FOR each group IN groups  
            INCREMENT element at position of the remainder of group DIVIDED BY batchSize of remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder, variadic remainder_count)  
            SET remainder_count TO a list converted from remainder_count tuple  
            SET max_happy TO zero  
            
            FOR index FROM zero TO batchSize MINUS one  
                IF element at position index of remainder_count EQUALS zero  
                    CONTINUE to next iteration  
                END IF  
                
                DECREMENT element at position index of remainder_count BY one  
                SET new_remainder TO the remainder of remainder MINUS index DIVIDED BY batchSize using modulo operation  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS CALL dp WITH new_remainder AND the unpacked remainder_count  
                ELSE  
                    SET happy_groups TO CALL dp WITH new_remainder AND the unpacked remainder_count  
                END IF  
                
                IF happy_groups GREATER THAN max_happy  
                    SET max_happy TO happy_groups  
                END IF  
                
                INCREMENT element at position index of remainder_count BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN CALL dp WITH zero AND the unpacked remainder_count  
    END FUNCTION  
END CLASS
