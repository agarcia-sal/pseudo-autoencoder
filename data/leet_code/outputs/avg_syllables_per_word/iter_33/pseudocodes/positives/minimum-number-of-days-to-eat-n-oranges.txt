CLASS Solution  
    FUNCTION minDays(n)  
        FUNCTION dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            SET option_one TO the remainder of remaining DIVIDED BY two PLUS dp(the quotient of remaining DIVIDED BY two)  
            SET option_two TO the remainder of remaining DIVIDED BY three PLUS dp(the quotient of remaining DIVIDED BY three)  
            SET minimum_option TO the smaller value BETWEEN option_one AND option_two  
            RETURN one PLUS minimum_option  
        END FUNCTION  
        RETURN dp(n)  
    END FUNCTION  
END CLASS
