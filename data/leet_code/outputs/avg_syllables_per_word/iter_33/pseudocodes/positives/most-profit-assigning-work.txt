CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO list combining elements at corresponding positions of difficulty and profit into pairs  
        SORT jobs by the first element of each pair in ascending order  
        SORT worker in ascending order  
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the length of jobs  
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND the first element of the pair at position job_index of jobs LESS THAN OR EQUAL TO ability  
                IF the second element of the pair at position job_index of jobs GREATER THAN best_profit  
                    SET best_profit TO the second element of the pair at position job_index of jobs  
                END IF  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        RETURN max_profit  
    END FUNCTION  
END CLASS
