CLASS Solution  
    FUNCTION maxSlidingWindow(nums k)  
        IF nums EQUALS empty OR k EQUALS zero  
            RETURN empty list  
        END IF  

        SET deq TO empty double ended queue  
        SET result TO empty list  

        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT empty AND the element at position zero of deq LESS THAN i MINUS k PLUS one  
                REMOVE the element at position zero from deq  
            END IF  

            WHILE deq IS NOT empty AND the element at position element at position the last position of deq of nums LESS THAN element at position i of nums  
                REMOVE the last element from deq  
            END WHILE  

            APPEND i TO deq  

            IF i GREATER THAN OR EQUAL TO k MINUS one  
                APPEND the element at position element at position zero of deq of nums TO result  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
