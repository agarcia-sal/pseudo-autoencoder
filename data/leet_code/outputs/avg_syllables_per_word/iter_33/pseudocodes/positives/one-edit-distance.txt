CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF calling isOneEditDistance WITH t AND s  
        END IF  
        
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        
        IF diff GREATER THAN one  
            RETURN False  
        END IF  
        
        IF diff EQUALS zero  
            SET count TO zero  
            FOR each pair of characters c1 AND c2 FROM s AND t IN ORDER  
                IF c1 NOT EQUALS c2  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            RETURN whether count EQUALS one  
        ELSE  
            FOR index FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position index of s NOT EQUALS the element at position index of t  
                    RETURN whether the substring from position index TO the END of s EQUALS the substring from position index PLUS one TO the END of t  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
