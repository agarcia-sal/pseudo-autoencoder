CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUALS plus SIGN OR token EQUALS minus SIGN OR token EQUALS multiplication SIGN OR token EQUALS division SIGN  
                SET b TO element at last position of stack  
                REMOVE element at last position FROM stack  
                SET a TO element at last position of stack  
                REMOVE element at last position FROM stack  
                IF token EQUALS plus SIGN  
                    APPEND the sum of a PLUS b TO stack  
                ELSE IF token EQUALS minus SIGN  
                    APPEND the difference of a MINUS b TO stack  
                ELSE IF token EQUALS multiplication SIGN  
                    APPEND the product of a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS division SIGN  
                    APPEND the integer part of a DIVIDED BY b TO stack  
                END IF  
            ELSE  
                APPEND the integer value of token TO stack  
            END IF  
        END FOR  
        RETURN element at first position of stack  
    END FUNCTION  
END CLASS
