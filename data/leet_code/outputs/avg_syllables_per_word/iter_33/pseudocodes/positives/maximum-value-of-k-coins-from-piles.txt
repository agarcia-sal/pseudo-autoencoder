CLASS Solution  
    FUNCTION maxValueOfCoins(piles k)  
        SET n TO the LENGTH OF piles  
        SET dp TO a two dimensional list WITH n PLUS one ROWS AND k PLUS one COLUMNS INITIALIZED TO zero  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one of piles  
                SET current_value TO zero  
                FOR l FROM zero TO the MINIMUM OF j AND the LENGTH OF current_pile  
                    IF l GREATER THAN zero  
                        SET current_value TO current_value PLUS element at position l MINUS one of current_pile  
                    END IF  
                    SET dp element at position i element at position j TO the GREATER VALUE BETWEEN dp element at position i element at position j AND dp element at position i MINUS one element at position j MINUS l PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position n element at position k  
    END FUNCTION  
END CLASS
