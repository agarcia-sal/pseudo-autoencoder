CLASS Solution  
    FUNCTION loudAndRich WITH PARAMETERS richer quiet  
        SET graph TO a new empty mapping from element to list  
        FOR each pair a b IN richer  
            APPEND element a TO the list at key b of graph  
        END FOR  
        SET n TO the LENGTH OF quiet  
        SET answer TO a list of length n with each element set TO negative one  
        FUNCTION dfs WITH PARAMETER person  
            IF the element at position person of answer EQUALS negative one  
                SET the element at position person of answer TO person  
                FOR each neighbor IN the list at key person of graph  
                    SET candidate TO the result of calling dfs WITH argument neighbor  
                    IF the element at position candidate of quiet LESS THAN the element at position the element at position person of answer of quiet  
                        SET the element at position person of answer TO candidate  
                    END IF  
                END FOR  
            END IF  
            RETURN the element at position person of answer  
        END FUNCTION  
        FOR i FROM zero TO n MINUS one  
            CALL dfs WITH argument i  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
