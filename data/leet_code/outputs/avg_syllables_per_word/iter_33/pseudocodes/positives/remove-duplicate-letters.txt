CLASS Solution  
    FUNCTION removeDuplicateLetters(s)  
        SET last_occurrence TO an empty dictionary  
        FOR each index idx FROM zero TO the LENGTH OF s MINUS one  
            SET character char TO element at position idx of s  
            SET the value at key char of last_occurrence TO idx  
        END FOR  
        
        SET stack TO an empty list  
        SET in_stack TO an empty set  
        
        FOR each index idx FROM zero TO the LENGTH OF s MINUS one  
            SET character char TO element at position idx of s  
            
            IF char IS IN in_stack  
                CONTINUE TO the next iteration of the loop  
            END IF  
            
            WHILE the LENGTH OF stack IS GREATER THAN zero AND char IS LESS THAN element at the last position of stack AND idx IS LESS THAN the value at key element at the last position of stack of last_occurrence  
                SET removed_char TO the last element removed FROM stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            
            APPEND char TO stack  
            ADD char TO in_stack  
        END FOR  
        
        SET result TO the concatenation of all elements in stack in order  
        RETURN result  
    END FUNCTION  
END CLASS
