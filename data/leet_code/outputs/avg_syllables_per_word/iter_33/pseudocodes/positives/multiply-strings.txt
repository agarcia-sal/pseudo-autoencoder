CLASS Solution  
    FUNCTION multiply(num1  
    num2)  
        IF num1 EQUALS the string zero OR num2 EQUALS the string zero  
            RETURN the string zero  
        END IF  
        SET result TO a list of zero repeated the LENGTH OF num1 PLUS the LENGTH OF num2 times  
        SET num1 TO the characters of num1 in reverse order  
        SET num2 TO the characters of num2 in reverse order  
        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the numeric code of the element at position i of num1 MINUS the numeric code of the character zero MULTIPLIED BY the numeric code of the element at position j of num2 MINUS the numeric code of the character zero  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS the element at position p1 of result  
                SET the element at position p1 of result TO the remainder of sum_ DIVIDED BY ten  
                INCREMENT the element at position p2 of result BY the quotient of sum_ DIVIDED BY ten  
            END FOR  
        END FOR  
        SET result_str TO the concatenation of the string representations of the elements of result in reverse order  
        RETURN result_str with all leading occurrences of the character zero removed  
    END FUNCTION  
END CLASS
