CLASS Solution  
    FUNCTION checkSubarraySum(nums, k)  
        SET remainder_map TO a dictionary mapping zero TO negative one  
        SET current_sum TO zero  
        FOR each index i WITH element num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF k NOT EQUALS zero  
                SET current_sum TO the remainder of current_sum DIVIDED BY k  
            END IF  
            IF current_sum IS A KEY IN remainder_map  
                IF the difference BETWEEN i AND the value associated WITH current_sum IN remainder_map IS GREATER THAN one  
                    RETURN True  
                END IF  
            ELSE  
                ASSIGN the value i TO the key current_sum IN remainder_map  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
