CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO ten POWERED BY nine PLUS one  
        SET n TO the LENGTH OF s  
        SET dp TO a two-dimensional list of SIZE n PLUS one BY n PLUS one FILLED WITH zero  
        
        FOR j FROM zero TO n  
            SET element at position j of element at position zero of dp TO one  
        END FOR  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            
            IF element at position i MINUS one of s EQUALS the character I  
                FOR j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position j of element at position i MINUS one of dp MODULO MOD  
                    SET element at position j of element at position i of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR j FROM n MINUS i DOWN TO zero BY MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position j PLUS one of element at position i MINUS one of dp MODULO MOD  
                    SET element at position j of element at position i of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO zero  
        FOR each value IN element at position n of dp  
            SET result TO result PLUS value  
        END FOR  
        RETURN result MODULO MOD  
    END FUNCTION  
END CLASS
