CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num EQUALS one  
            RETURN one  
        END IF  
        
        SET factors TO empty list  
        
        FOR digit FROM nine TO two DECREMENTING BY one  
            WHILE remainder of num DIVIDED BY digit EQUALS zero  
                APPEND digit TO factors  
                SET num TO integer division of num BY digit  
            END WHILE  
        END FOR  
        
        IF num NOT EQUALS one  
            RETURN zero  
        END IF  
        
        SORT factors in ascending order  
        
        SET result TO number formed by concatenating each element in factors as string in order  
        
        IF result GREATER THAN two raised to the power of thirty one MINUS one  
            RETURN zero  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
