CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one  
        END IF  
        
        SET n TO n PLUS twenty four DIVIDED BY twenty five rounded down to the nearest integer greater than or equal to the result  
        
        SET memo TO an empty collection mapping pairs to probabilities  
        
        FUNCTION dp(a b)  
            IF the pair of a and b EXISTS AS A KEY IN memo  
                RETURN the value FROM memo FOR the pair of a and b  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN zero point five  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one  
            END IF  
            
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero  
            END IF  
            
            SET prob TO zero point two five MULTIPLIED BY the sum of dp of a MINUS four b, dp of a MINUS three b MINUS one, dp of a MINUS two b MINUS two, and dp of a MINUS one b MINUS three  
            
            SET the value in memo FOR the pair of a and b TO prob  
            RETURN prob  
        END FUNCTION  
        
        RETURN dp WITH parameters n and n  
    END FUNCTION  
END CLASS
