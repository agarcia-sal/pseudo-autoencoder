CLASS Solution  
    FUNCTION stoneGameVIII(stones)  
        SET prefix_sum TO a list of zeros of the LENGTH OF stones  
        SET element at position one of prefix_sum TO element at position one of stones  
        FOR index FROM two TO the LENGTH OF stones  
            SET element at position index of prefix_sum TO element at position index MINUS one of prefix_sum PLUS element at position index of stones  
        END FOR  
        SET dp TO element at the last position of prefix_sum  
        FOR index FROM the LENGTH OF stones MINUS two DOWN TO two BY negative one  
            IF dp GREATER THAN prefix_sum at position index MINUS dp  
                CONTINUE TO the next iteration  
            ELSE  
                SET dp TO prefix_sum at position index MINUS dp  
            END IF  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
