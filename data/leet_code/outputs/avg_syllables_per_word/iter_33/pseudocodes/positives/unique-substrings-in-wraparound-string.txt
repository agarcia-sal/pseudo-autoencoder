CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO an empty dictionary mapping each character from a to z TO zero  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND the character code of element at position index of s MINUS the character code of element at position index MINUS one of s EQUALS one OR element at position index of s EQUALS character a AND element at position index MINUS one of s EQUALS character z  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET existing_value TO the value in max_length for the character at position index of s  
            SET max_length for the character at position index of s TO the GREATER OF existing_value AND current_length  
        END FOR  
        SET total_unique_substrings TO zero  
        FOR each value IN the values of max_length  
            INCREMENT total_unique_substrings BY value  
        END FOR  
        RETURN total_unique_substrings  
    END FUNCTION  
END CLASS
