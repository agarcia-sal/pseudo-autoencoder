CLASS Solution  
    FUNCTION findMaxValueOfEquation(points, k)  
        SET max_value TO negative infinity  
        ASSIGN a new empty deque TO deque_  
        
        FOR each pair xj and yj IN points  
            WHILE deque_ is not empty AND the value of xj MINUS the second element of the element at position zero of deque_ is GREATER THAN k  
                REMOVE the element at position zero FROM deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO the greater value BETWEEN max_value AND the sum of yj PLUS xj PLUS the first element of the element at position zero of deque_  
            END IF  
            
            WHILE deque_ is not empty AND the difference between yj MINUS xj is GREATER THAN OR EQUAL TO the first element of the element at the last position of deque_  
                REMOVE the element at the last position FROM deque_  
            END WHILE  
            
            APPEND the pair of the difference between yj MINUS xj AND the value xj TO deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
