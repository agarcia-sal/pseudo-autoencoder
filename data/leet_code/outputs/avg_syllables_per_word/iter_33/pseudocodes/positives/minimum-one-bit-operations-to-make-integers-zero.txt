CLASS Solution  
    FUNCTION minimumOneBitOperations(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE two RAISED TO THE POWER OF k PLUS one LESS THAN OR EQUALS n  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO two RAISED TO THE POWER OF k  
        RETURN two RAISED TO THE POWER OF k PLUS one MINUS one MINUS self CALL minimumOneBitOperations WITH n EXCLUSIVE OR most_significant_bit  
    END FUNCTION  
END CLASS
