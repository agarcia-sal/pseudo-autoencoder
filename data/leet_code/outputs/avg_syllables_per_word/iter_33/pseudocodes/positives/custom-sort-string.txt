CLASS Solution  
    FUNCTION customSortString(order  
                                s)  
        SET d TO empty dictionary  
        FOR each index IN range from zero TO the length of order MINUS one  
            ASSIGN element at position index of order TO current_character  
            SET current_character in dictionary d TO index  
        END FOR  
        SET sorted_list TO the result of sorting string s WHERE the key FOR each character x IS the value FROM dictionary d AT key x IF it exists OTHERWISE zero  
        SET result_string TO the concatenation of all elements in sorted_list in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
