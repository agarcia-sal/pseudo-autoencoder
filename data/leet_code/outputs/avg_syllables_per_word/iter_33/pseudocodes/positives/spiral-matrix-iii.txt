CLASS Solution  
    FUNCTION spiralMatrixIII(rows  
                             cols  
                             rStart  
                             cStart) RETURNS list of list of integers  
        SET directions TO list of pairs of integers representing north-east-south-west unit moves one step at a time starting from east direction including east south west north  
        SET direction_index TO zero representing starting facing east  
        SET steps TO zero representing number of steps to move in current direction  
        SET increment TO zero representing counter to track when to increase steps  
        SET visited TO empty list representing visited coordinates  
        SET x TO rStart  
        SET y TO cStart  
        
        WHILE the LENGTH OF visited is LESS THAN rows MULTIPLIED BY cols  
            IF increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR each count FROM one TO steps  
                IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN rows AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN cols  
                    APPEND list containing x and y TO visited  
                END IF  
                SET dx TO element at position direction_index of directions first element  
                SET dy TO element at position direction_index of directions second element  
                INCREMENT x BY dx  
                INCREMENT y BY dy  
            END FOR  
            SET direction_index TO the remainder of direction_index PLUS one DIVIDED BY four  
            INCREMENT increment BY one  
        END WHILE  
        
        RETURN visited  
    END FUNCTION  
END CLASS
