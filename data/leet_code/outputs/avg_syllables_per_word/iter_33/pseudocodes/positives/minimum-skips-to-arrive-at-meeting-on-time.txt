CLASS Solution  
    FUNCTION minSkips(dist speed hoursBefore)  
        SET n TO the LENGTH OF dist  
        SET dp TO a two dimensional list with n rows and n PLUS one columns where each element is INFINITY  
        SET total_dist TO the SUM of all elements IN dist  
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore THEN  
            RETURN the NEGATIVE of one  
        END IF  
        
        SET element at position zero and position zero of dp TO element at position zero of dist  
        FOR j FROM one TO n MINUS one  
            SET element at position zero and position j of dp TO element at position zero of dist  
        END FOR  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j LESS THAN i THEN  
                    SET element at position i and position j of dp TO the CEILING of element at position i MINUS one and position j of dp DIVIDED BY speed MULTIPLIED BY speed PLUS element at position i of dist  
                END IF  
                IF j GREATER THAN zero THEN  
                    SET element at position i and position j of dp TO the MINIMUM of element at position i and position j of dp AND element at position i MINUS one and position j MINUS one of dp PLUS element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            IF element at position n MINUS one and position j of dp LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN the NEGATIVE of one  
    END FUNCTION  
END CLASS
