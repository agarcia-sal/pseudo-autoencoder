CLASS Solution  
    FUNCTION canVisitAllRooms WITH PARAMETER rooms  
        SET visited TO empty collection representing a set  
        SET stack TO collection containing the element zero  
        
        WHILE the LENGTH OF stack IS GREATER THAN zero  
            SET current_room TO the element REMOVED FROM the END OF stack  
            ADD current_room TO the collection visited  
            FOR each key IN the element at position current_room of rooms  
                IF key IS NOT IN visited  
                    ADD key TO the END OF stack  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF visited EQUALS the LENGTH OF rooms  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
