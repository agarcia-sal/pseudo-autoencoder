CLASS Solution  
    FUNCTION smallestDistancePair(nums k)  
        SORT the list nums in ascending order  
        SET n TO the LENGTH OF the list nums  
        
        FUNCTION count(mid)  
            SET count TO zero  
            SET left TO zero  
            FOR each right FROM zero TO n MINUS one  
                WHILE the element at position right of the list nums MINUS the element at position left of the list nums GREATER THAN mid  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT count BY right MINUS left  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the element at the last position of the list nums MINUS the element at the first position of the list nums  
        WHILE left LESS THAN right  
            SET mid TO the sum of left AND right DIVIDED BY two with integer division  
            IF the result of count with mid AS argument LESS THAN k  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
