CLASS Solution  
    FUNCTION canFinish(numCourses numCourses PARAMETERS, prerequisites prerequisites PARAMETERS)  
        ASSIGN reference TO a new empty mapping from item to list named graph  
        SET indegree TO a list of zeros of length numCourses  
        
        FOR each pair of course and prereq IN prerequisites  
            APPEND course TO the list at key prereq in graph  
            INCREMENT the element at position course of indegree BY one  
        END FOR  
        
        SET queue TO a new double-ended queue containing each course in the range from zero TO numCourses MINUS one with the condition that the element at position course of indegree EQUALS zero  
        
        WHILE queue is not empty  
            REMOVE and ASSIGN the element from the front of queue TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN the list at key current_course in graph  
                DECREMENT the element at position next_course of indegree BY one  
                IF the element at position next_course of indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the condition that numCourses EQUALS zero  
    END FUNCTION  
END CLASS
