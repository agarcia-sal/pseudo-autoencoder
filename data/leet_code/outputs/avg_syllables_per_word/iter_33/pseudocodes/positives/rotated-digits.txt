CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO collection of characters two five six nine  
        SET same_digits TO collection of characters zero one eight  
        
        FUNCTION is_good_number(num)  
            SET num_str TO string representation of num  
            IF all digit IN union of change_digits and same_digits for each digit IN num_str THEN  
                IF any digit IN change_digits for each digit IN num_str THEN  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR i FROM one TO n DO  
            IF is_good_number(i) EQUALS true THEN  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
