CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO splitting path by the character slash  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component EQUALS an empty string OR component EQUALS a single period  
                CONTINUE to the next iteration  
            ELSE IF component EQUALS two periods  
                IF stack is not empty  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                ADD component TO the end of stack  
            END IF  
        END FOR  
        SET simplified_path TO concatenation of the character slash AND concatenation of all elements in stack separated by the character slash  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
