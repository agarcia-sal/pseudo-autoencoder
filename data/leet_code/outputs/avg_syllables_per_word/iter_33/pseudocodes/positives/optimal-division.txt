CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the string representation of the element at position one of nums  
        END IF  
        
        IF the LENGTH OF nums EQUALS two  
            RETURN the string representation of the element at position one of nums  
            CONCATENATED WITH the division word  
            CONCATENATED WITH the string representation of the element at position two of nums  
        END IF  
        
        RETURN the string representation of the element at position one of nums  
        CONCATENATED WITH the division word  
        CONCATENATED WITH an opening parenthesis word  
        CONCATENATED WITH the string join of the division word between the string representations of elements from position two TO the length of nums of nums  
        CONCATENATED WITH a closing parenthesis word  
    END FUNCTION  
END CLASS
