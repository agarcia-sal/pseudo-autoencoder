CLASS Solution  
    FUNCTION minDistance(houses LIST OF INTEGERS k INTEGER)  
        SORT houses IN ASCENDING ORDER  
        SET n TO the LENGTH OF houses  
        
        INITIALIZE cost AS a TWO DIMENSIONAL LIST OF SIZE n BY n WITH all ELEMENTS SET TO zero  
        FOR i FROM n MINUS one DOWNTO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost AT position i and position j TO cost AT position i PLUS one and position j MINUS one PLUS the ELEMENT AT position j OF houses MINUS the ELEMENT AT position i OF houses  
            END FOR  
        END FOR  
        
        INITIALIZE dp AS a TWO DIMENSIONAL LIST OF SIZE n PLUS one BY k PLUS one WITH all ELEMENTS SET TO positive infinity  
        SET dp AT position zero and position zero TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET dp AT position i and position j TO THE MINIMUM OF dp AT position i and position j AND dp AT position p and position j MINUS one PLUS cost AT position p and position i MINUS one  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp AT position n and position k  
    END FUNCTION  
END CLASS
