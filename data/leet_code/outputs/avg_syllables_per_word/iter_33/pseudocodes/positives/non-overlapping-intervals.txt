CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF the LENGTH OF intervals EQUALS zero  
            RETURN zero  
        END IF  
        SORT intervals BY the element at position two of each interval  
        SET end TO the element at position two of the first interval in intervals  
        SET count TO zero  
        FOR i FROM two TO the LENGTH OF intervals  
            IF the element at position one of the interval at position i GREATER THAN OR EQUAL TO end  
                SET end TO the element at position two of the interval at position i  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
