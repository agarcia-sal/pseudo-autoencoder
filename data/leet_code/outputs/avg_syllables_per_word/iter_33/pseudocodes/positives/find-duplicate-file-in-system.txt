CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO an empty map from content TO list of file paths  
        FOR each path IN paths  
            SET parts TO the list of elements obtained by splitting path at each space  
            SET directory TO element at position one of parts  
            FOR each file_info IN the elements of parts starting from position two to the last element  
                SET file_name TO the substring of file_info from the start TO the position before the first opening parenthesis  
                SET content TO the substring of file_info from the position after the first opening parenthesis TO the position before the last character  
                APPEND the concatenation of directory PLUS the forward slash PLUS file_name TO the list associated with content in content_to_paths  
            END FOR  
        END FOR  
        SET result TO an empty list  
        FOR each group_of_paths IN the values of content_to_paths  
            IF the size of group_of_paths GREATER THAN one  
                APPEND group_of_paths TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
