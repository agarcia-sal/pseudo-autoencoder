CLASS Solution  
    FUNCTION countPalindromicSubsequences(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        
        SET next_occurrence TO a two dimensional list with four rows and n columns each initialized to negative one  
        SET prev_occurrence TO a two dimensional list with four rows and n columns each initialized to negative one  
        
        FOR c FROM zero TO three  
            SET char TO the character that is the letter a PLUS c positions forward in the alphabet  
            SET last TO negative one  
            FOR i FROM zero TO n MINUS one  
                IF element at position i of s EQUALS char  
                    SET last TO i  
                END IF  
                SET element at position c and i of prev_occurrence TO last  
            END FOR  
            
            SET last TO negative one  
            FOR i FROM n MINUS one DOWN TO zero  
                IF element at position i of s EQUALS char  
                    SET last TO i  
                END IF  
                SET element at position c and i of next_occurrence TO last  
            END FOR  
        END FOR  
        
        SET dp TO a two dimensional list with n rows and n columns each initialized to zero  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i and i of dp TO one  
        END FOR  
        
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                FOR c FROM zero TO three  
                    SET char TO the character that is the letter a PLUS c positions forward in the alphabet  
                    SET left TO element at position c and i of next_occurrence  
                    SET right TO element at position c and j of prev_occurrence  
                    
                    IF left GREATER THAN j OR right LESS THAN i  
                        CONTINUE TO the next iteration of the loop  
                    ELSE IF left EQUALS right  
                        SET element at position i and j of dp TO the element at position i and j of dp PLUS one MODULO MOD  
                    ELSE  
                        SET element at position i and j of dp TO the element at position i and j of dp PLUS the element at position left PLUS one and right MINUS one of dp PLUS two MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero and n MINUS one of dp  
    END FUNCTION  
END CLASS
