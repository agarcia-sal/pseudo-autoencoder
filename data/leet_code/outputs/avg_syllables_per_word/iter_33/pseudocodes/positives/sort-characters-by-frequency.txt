CLASS Solution  
    FUNCTION frequencySort(s)  
        CALL function Counter with s TO frequency  
        CALL function sorted with keys of frequency using ordering by frequency of element at position current character in frequency from greatest to least TO sorted_characters  
        SET result TO empty string  
        FOR each character IN sorted_characters  
            APPEND character REPEATED frequency of character times TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
