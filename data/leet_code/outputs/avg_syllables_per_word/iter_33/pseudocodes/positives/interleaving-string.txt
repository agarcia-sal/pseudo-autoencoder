CLASS Solution  
    FUNCTION isInterleave(s1 s1 AS string s2 s2 AS string s3 s3 AS string) RETURNS boolean  
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 NOT EQUALS the LENGTH OF s3  
            RETURN false  
        END IF  
        SET dp TO a list OF boolean values all set TO false with the LENGTH OF s2 PLUS one  
        SET the element at position zero of dp TO true  
        FOR i FROM zero TO the LENGTH OF s1  
            FOR j FROM zero TO the LENGTH OF s2  
                IF i GREATER THAN zero  
                    SET the element at position j of dp TO the element at position j of dp AND the character at position i MINUS one IN s1 EQUALS the character at position i PLUS j MINUS one IN s3  
                END IF  
                IF j GREATER THAN zero  
                    SET the element at position j of dp TO the element at position j of dp OR the element at position j MINUS one of dp AND the character at position j MINUS one IN s2 EQUALS the character at position i PLUS j MINUS one IN s3  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position the LENGTH OF s2 of dp  
    END FUNCTION  
END CLASS
