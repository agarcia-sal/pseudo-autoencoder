CLASS Solution  
    FUNCTION numRabbits(answers)  
        SET answer_counts TO the count of each distinct element IN answers  
        SET total_rabbits TO zero  
        FOR each pair of answer AND count IN the pairs of answer_counts  
            SET group_size TO answer PLUS one  
            SET groups TO the result of count PLUS answer DIVIDED BY group_size rounded down to the nearest whole number  
            INCREMENT total_rabbits BY groups MULTIPLIED BY group_size  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
