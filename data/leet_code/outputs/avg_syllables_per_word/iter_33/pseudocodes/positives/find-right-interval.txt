CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO list of pairs of element at position zero of interval and index for each index and interval in enumeration of intervals  
        SORT start_times by first element of each pair in ascending order  
        SET result TO empty list  
        FOR each interval IN intervals  
            SET end_time TO element at position one of interval  
            SET idx TO the position returned by binary search for the place where a pair starting with end_time can be inserted in start_times to maintain sorted order  
            IF idx LESS THAN the LENGTH OF start_times THEN  
                APPEND element at position one of element at position idx of start_times TO result  
            ELSE  
                APPEND the integer negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
