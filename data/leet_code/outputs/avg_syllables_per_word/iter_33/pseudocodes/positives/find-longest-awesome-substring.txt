CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a map from zero to negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        FOR each index i AND character char IN s with enumeration  
            SET digit TO the integer value of character char  
            SET prefix_xor TO prefix_xor XOR a value with one shifted left by digit times  
            IF prefix_xor IS a key in prefix_map THEN  
                SET max_length TO the larger of max_length AND the difference between i AND the element at position prefix_xor of prefix_map  
            END IF  
            FOR each index j FROM zero TO nine  
                SET flipped TO prefix_xor XOR a value with one shifted left by j times  
                IF flipped IS a key in prefix_map THEN  
                    SET max_length TO the larger of max_length AND the difference between i AND the element at position flipped of prefix_map  
                END IF  
            END FOR  
            IF prefix_xor IS NOT a key in prefix_map THEN  
                SET the element at position prefix_xor of prefix_map TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
