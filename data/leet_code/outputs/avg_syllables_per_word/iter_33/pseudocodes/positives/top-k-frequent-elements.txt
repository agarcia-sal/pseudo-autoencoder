CLASS Solution  
    FUNCTION topKFrequent(nums k) RETURNS list of integers  
        SET frequency_map TO a mapping from integer to integer with default zero  
        FOR each num IN nums  
            INCREMENT value at key num in frequency_map BY one  
        END FOR  

        SET max_frequency TO the greatest value among all values in frequency_map  
        SET buckets TO a list of empty lists with the length of max_frequency PLUS one  

        FOR each num freq IN all key value pairs in frequency_map  
            APPEND num TO the element at position freq of buckets  
        END FOR  

        SET result TO an empty list  
        FOR freq FROM max_frequency DOWN TO one  
            IF the length of the element at position freq of buckets IS GREATER THAN zero  
                APPEND all elements of the element at position freq of buckets TO result  
                IF the length of result IS GREATER THAN OR EQUAL TO k  
                    RETURN the first k elements of result  
                END IF  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
