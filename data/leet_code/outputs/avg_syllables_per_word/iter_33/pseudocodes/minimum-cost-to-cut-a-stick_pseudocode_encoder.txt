CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO the sorted result of appending zero and n to cuts  
        
        FUNCTION dp(left, right)  
            IF the result of right MINUS left is LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO a value representing infinity  
            FOR i FROM the result of left PLUS one TO the result of right MINUS one  
                SET cost TO the element at position right of cuts MINUS the element at position left of cuts PLUS dp(left, i) PLUS dp(i, right)  
                IF cost IS LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp(zero, the result of the length of cuts MINUS one)  
    END FUNCTION  
END CLASS
