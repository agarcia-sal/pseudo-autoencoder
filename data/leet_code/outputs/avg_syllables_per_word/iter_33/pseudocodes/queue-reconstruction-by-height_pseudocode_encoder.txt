CLASS Solution  
    FUNCTION reconstructQueue(people)  
        SORT people BY the NEGATION OF element at position zero of person AND THEN BY element at position one of person IN ASCENDING ORDER  
        SET queue TO an empty list  
        FOR each person IN people  
            INSERT person INTO queue AT the position indicated BY element at position one of person  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
