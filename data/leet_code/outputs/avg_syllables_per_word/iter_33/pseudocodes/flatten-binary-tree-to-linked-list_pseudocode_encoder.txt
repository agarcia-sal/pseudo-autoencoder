CLASS TreeNode  
    FUNCTION __init__ with parameters val SET TO zero left SET TO none right SET TO none  
        SET the value property TO val  
        SET the left child property TO left  
        SET the right child property TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the values collection does not contain any elements  
        RETURN none  
    END IF  
    ASSIGN reference TO a new TreeNode object initialized with the first element of values  
    SET counter i TO one  
    CREATE an empty double-ended queue named queue  
    APPEND the root reference TO the queue  
    WHILE the queue contains any elements  
        REMOVE and ASSIGN the first element of queue TO node  
        IF the counter i is less than the length of values AND the element at position i of values is not none  
            SET the left child property of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the left child property of node TO the queue  
        END IF  
        INCREMENT i BY one  
        IF the counter i is less than the length of values AND the element at position i of values is not none  
            SET the right child property of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the right child property of node TO the queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN the root reference  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p does not exist AND q does not exist  
        RETURN true  
    ELSE IF p does not exist OR q does not exist  
        RETURN false  
    ELSE IF the value property of p NOT EQUALS the value property of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of the result of is_same_tree with parameters the left child property of p and the left child property of q AND the result of is_same_tree with parameters the right child property of p and the right child property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten with parameter root  
        IF root does not exist  
            RETURN  
        END IF  
        CREATE a list named stack containing the root element  
        WHILE the stack contains any elements  
            REMOVE and ASSIGN the last element of stack TO node  
            IF the right child property of node exists  
                APPEND the right child property of node TO stack  
            END IF  
            IF the left child property of node exists  
                APPEND the left child property of node TO stack  
            END IF  
            SET the left child property of node TO none  
            IF the stack contains any elements  
                SET the right child property of node TO the last element of stack  
            ELSE  
                SET the right child property of node TO none  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
