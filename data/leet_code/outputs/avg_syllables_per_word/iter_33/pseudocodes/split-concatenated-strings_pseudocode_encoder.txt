CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the reverse of the element at position index of strs  
            IF reversed_str GREATER THAN the element at position index of strs  
                SET the element at position index of strs TO reversed_str  
            END IF  
        END FOR  
  
        SET doubled_strs TO the concatenation of strs WITH strs  
  
        SET max_string TO an empty string  
  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN the collection consisting of False AND True  
                IF is_reversed EQUALS True  
                    SET current TO the reverse of the element at position index of strs  
                ELSE  
                    SET current TO the element at position index of strs  
                END IF  
  
                SET remaining TO the concatenation of the elements from position index PLUS one TO position index PLUS the LENGTH OF strs MINUS one of doubled_strs  
  
                FOR cut_position FROM zero TO the LENGTH OF current  
                    SET candidate TO the concatenation of the substring from cut_position TO the end of current WITH remaining WITH the substring from the start TO cut_position of current  
  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
  
        RETURN max_string  
    END FUNCTION  
END CLASS
