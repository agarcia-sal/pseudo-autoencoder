CLASS Solution  
    FUNCTION longestSubsequenceRepeatedK WITH PARAMETERS s AND k  
        IMPORT the data structure Counter from collections  
  
        SET freq TO Counter of sequence s  
  
        SET valid_chars TO an empty list  
        FOR each character AND count IN the items of freq  
            IF count GREATER THAN OR EQUAL TO k  
                APPEND character TO valid_chars  
            END IF  
        END FOR  
  
        SORT valid_chars in descending lexicographical order  
  
        FUNCTION is_valid_subsequence WITH PARAMETER candidate  
            SET count TO zero  
            SET index TO zero  
            FOR each character IN sequence s  
                IF character EQUALS the element at position index of candidate  
                    INCREMENT index BY one  
                    IF index EQUALS the LENGTH OF candidate  
                        INCREMENT count BY one  
                        SET index TO zero  
                        IF count EQUALS k  
                            RETURN True  
                        END IF  
                    END IF  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
  
        SET max_length TO the LENGTH OF valid_chars DIVIDED BY k  
  
        SET best_candidate TO an empty string  
  
        FUNCTION backtrack WITH PARAMETER candidate  
            IF the LENGTH OF candidate GREATER THAN the LENGTH OF best_candidate  
                SET best_candidate TO candidate  
            END IF  
            FOR each character IN valid_chars  
                SET new_candidate TO candidate PLUS character  
                IF is_valid_subsequence WITH new_candidate RETURNS True  
                    CALL backtrack WITH new_candidate  
                END IF  
            END FOR  
        END FUNCTION  
  
        CALL backtrack WITH an empty string  
        RETURN best_candidate  
    END FUNCTION  
END CLASS
