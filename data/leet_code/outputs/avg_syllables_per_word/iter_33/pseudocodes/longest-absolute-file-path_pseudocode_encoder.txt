CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO list of elements obtained by splitting input at each occurrence of newline character  
        SET stack TO empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO the number of tab characters contained in line  
            SET name TO the substring of line starting from position depth to the end  
            
            WHILE the LENGTH OF stack IS GREATER THAN depth  
                REMOVE the last element FROM stack  
            END WHILE  
            
            IF the LENGTH OF stack IS GREATER THAN zero  
                SET current_length TO the last element of stack PLUS the LENGTH OF name PLUS one  
            ELSE  
                SET current_length TO the LENGTH OF name  
            END IF  
            
            IF name CONTAINS the character period  
                IF current_length IS GREATER THAN max_length  
                    SET max_length TO current_length  
                END IF  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
