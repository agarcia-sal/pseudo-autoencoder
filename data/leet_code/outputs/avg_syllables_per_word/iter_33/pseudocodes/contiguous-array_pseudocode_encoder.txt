CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary with element zero mapped to element minus one  
        SET max_length TO zero  
        SET count TO zero  
        FOR each index i AND element num IN the enumeration of nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            IF count IS A KEY IN prefix_sum_index  
                SET max_length TO the GREATER OF max_length AND the result of i MINUS the element at key count IN prefix_sum_index  
            ELSE  
                SET the element at key count IN prefix_sum_index TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
