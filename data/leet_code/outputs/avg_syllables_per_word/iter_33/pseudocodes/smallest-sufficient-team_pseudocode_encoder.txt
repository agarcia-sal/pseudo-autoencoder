CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        SET skill_to_index TO empty dictionary  
        SET index_counter TO zero  
        FOR each skill IN req_skills  
            ASSIGN skill_to_index at skill TO index_counter  
            INCREMENT index_counter BY one  
        END FOR  
        SET total_skills TO the LENGTH OF req_skills  
        SET total_people TO the LENGTH OF people  
        SET people_skill_masks TO empty list of size total_people filled with zeroes  
        FOR i FROM zero TO total_people MINUS one  
            SET skill_list TO element at position i of people  
            SET skill_mask TO zero  
            FOR each skill IN skill_list  
                SET skill_mask TO skill_mask OR one shifted LEFT BY skill_to_index at skill  
            END FOR  
            SET element at position i OF people_skill_masks TO skill_mask  
        END FOR  
        SET infinite_value TO infinity  
        SET dp_min_team_size TO list of size two raised to total_skills filled with infinite_value  
        SET dp_last_person_index TO list of size two raised to total_skills filled with zero  
        SET dp_previous_skill_set TO list of size two raised to total_skills filled with zero  
        SET element at position zero OF dp_min_team_size TO zero  
        FOR current_skill_set FROM zero TO two raised to total_skills MINUS one  
            IF element at position current_skill_set OF dp_min_team_size EQUALS infinite_value  
                CONTINUE TO next iteration  
            END IF  
            FOR person_index FROM zero TO total_people MINUS one  
                SET combined_skill_set TO current_skill_set OR element at position person_index OF people_skill_masks  
                IF element at position current_skill_set OF dp_min_team_size PLUS one LESS THAN element at position combined_skill_set OF dp_min_team_size  
                    SET element at position combined_skill_set OF dp_min_team_size TO element at position current_skill_set OF dp_min_team_size PLUS one  
                    SET element at position combined_skill_set OF dp_last_person_index TO person_index  
                    SET element at position combined_skill_set OF dp_previous_skill_set TO current_skill_set  
                END IF  
            END FOR  
        END FOR  
        SET final_skill_set TO two raised to total_skills MINUS one  
        SET answer_team TO empty list  
        WHILE final_skill_set NOT EQUALS zero  
            APPEND element at position final_skill_set OF dp_last_person_index TO answer_team  
            SET final_skill_set TO element at position final_skill_set OF dp_previous_skill_set  
        END WHILE  
        RETURN answer_team  
    END FUNCTION  
END CLASS
