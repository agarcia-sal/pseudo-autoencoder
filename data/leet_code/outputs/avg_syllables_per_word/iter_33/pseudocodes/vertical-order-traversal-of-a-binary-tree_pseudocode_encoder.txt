CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO A TreeNode OBJECT INITIALIZED WITH the element at position one of values  
    SET i TO one  
    INITIALIZE queue AS an empty double ended queue  
    APPEND root TO queue  
    WHILE queue HOLDS elements  
        SET node TO the element REMOVED FROM the LEFT SIDE of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.left TO A TreeNode OBJECT INITIALIZED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.right TO A TreeNode OBJECT INITIALIZED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalTraversal(self, root)  
        IF root EQUALS None  
            RETURN an empty list  
        END IF  
        INITIALIZE node_dict AS a dictionary defaulting to empty list  
        SET min_col TO zero  
        SET max_col TO zero  

        FUNCTION dfs(node, row, col)  
            IF node EQUALS None  
                RETURN  
            END IF  
            APPEND the pair consisting of row and node.val TO the list at key col in node_dict  
            IF col LESS THAN min_col  
                SET min_col TO col  
            END IF  
            IF col GREATER THAN max_col  
                SET max_col TO col  
            END IF  
            CALL dfs ON node.left WITH row PLUS one AND col MINUS one  
            CALL dfs ON node.right WITH row PLUS one AND col PLUS one  
        END FUNCTION  

        CALL dfs ON root WITH zero FOR row AND zero FOR col  

        INITIALIZE result AS an empty list  
        FOR col FROM min_col TO max_col  
            SORT the list at key col in node_dict BY the first element of each pair THEN BY the second element of each pair IN ascending order  
            EXTRACT the second elements from each pair in the sorted list  
            APPEND this extracted list TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
