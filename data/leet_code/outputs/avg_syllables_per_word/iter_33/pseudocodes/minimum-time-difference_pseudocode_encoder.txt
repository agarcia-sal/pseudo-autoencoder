CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET split_result TO list of elements obtained by splitting time at character colon  
            SET hours TO element at position zero of split_result converted to integer  
            SET minutes TO element at position one of split_result converted to integer  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
        
        SET minutes_list TO list of elements obtained by applying to_minutes to each element in timePoints  
        SORT minutes_list IN ascending order  
        
        SET min_diff TO infinity  
        FOR i FROM one TO LENGTH OF minutes_list MINUS one  
            SET current_difference TO element at position i of minutes_list MINUS element at position i MINUS one of minutes_list  
            IF current_difference LESS THAN min_diff  
                SET min_diff TO current_difference  
            END IF  
        END FOR  
        
        SET circular_difference TO one thousand four hundred forty MINUS element at last position of minutes_list PLUS element at position zero of minutes_list  
        IF circular_difference LESS THAN min_diff  
            SET min_diff TO circular_difference  
        END IF  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
