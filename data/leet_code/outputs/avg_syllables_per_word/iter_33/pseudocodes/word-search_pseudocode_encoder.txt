CLASS Solution  
    FUNCTION exist(board word)  
        IF board EQUALS empty  
            RETURN False  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        SET visited TO an empty list  
        FOR i FROM zero TO m MINUS one  
            SET row TO an empty list  
            FOR j FROM zero TO n MINUS one  
                APPEND False TO row  
            END FOR  
            APPEND row TO visited  
        END FOR  

        FUNCTION dfs(x y index)  
            IF index EQUALS the LENGTH OF word  
                RETURN True  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of visited AND element at position y OF element at position x OF visited EQUALS True OR element at position x OF board AND element at position y OF element at position x OF board NOT EQUALS element at position index OF word  
                RETURN False  
            END IF  

            SET element at position x OF visited AND element at position y OF element at position x OF visited TO True  
            FOR each pair dx dy IN the list of pairs consisting of negative one and zero PLUS one and zero PLUS zero and negative one PLUS zero and one  
                IF dfs(x PLUS dx y PLUS dy index PLUS one)  
                    RETURN True  
                END IF  
            END FOR  
            SET element at position x OF visited AND element at position y OF element at position x OF visited TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs(i j zero)  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
