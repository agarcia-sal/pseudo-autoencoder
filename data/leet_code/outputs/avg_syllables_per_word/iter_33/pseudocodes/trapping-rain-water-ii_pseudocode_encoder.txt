CLASS Solution  
    FUNCTION trapRainWater(heightMap)  
        IF heightMap EQUALS empty OR element at position zero of heightMap EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF heightMap  
        SET n TO the LENGTH OF element at position zero of heightMap  
        SET visited TO a two dimensional list of Boolean FALSE values with m rows and n columns  
        SET min_heap TO an empty list  
        
        FOR i FROM zero TO m MINUS one  
            PUSH to min_heap a tuple consisting of element at position i of heightMap at position zero , i , zero  
            PUSH to min_heap a tuple consisting of element at position i of heightMap at position n MINUS one , i , n MINUS one  
            SET element at position i of visited at position zero TO TRUE  
            SET element at position i of visited at position n MINUS one TO TRUE  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            PUSH to min_heap a tuple consisting of element at position zero of heightMap at position j , zero , j  
            PUSH to min_heap a tuple consisting of element at position m MINUS one of heightMap at position j , m MINUS one , j  
            SET element at position zero of visited at position j TO TRUE  
            SET element at position m MINUS one of visited at position j TO TRUE  
        END FOR  
        
        SET directions TO a list of four tuples as follows  
            first tuple is minus one and zero  
            second tuple is one and zero  
            third tuple is zero and minus one  
            fourth tuple is zero and one  
        
        SET water_trapped TO zero  
        
        WHILE min_heap is not empty  
            POP from min_heap to get three values height x y  
            
            FOR each tuple dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx of visited at position ny EQUALS FALSE  
                    SET element at position nx of visited at position ny TO TRUE  
                    INCREMENT water_trapped BY the result of MAXIMUM BETWEEN zero AND height MINUS element at position nx of heightMap at position ny  
                    PUSH to min_heap a tuple consisting of the MAXIMUM BETWEEN height AND element at position nx of heightMap at position ny , nx , ny  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN water_trapped  
    END FUNCTION  
END CLASS
