CLASS Solution  
    FUNCTION wordSquares(words)  
        SET prefix_to_words TO empty mapping from text TO list of text  
        FOR each word IN words  
            FOR index FROM zero TO length of word MINUS one  
                SET prefix TO substring from position zero TO position index MINUS one of word  
                APPEND word TO list at prefix_to_words at prefix  
            END FOR  
        END FOR  
        
        SET results TO empty list  
        
        FUNCTION backtrack(square)  
            IF length of square EQUALS length of element at position zero of words THEN  
                APPEND square TO results  
                RETURN  
            END IF  
            
            SET prefix TO empty text  
            FOR each word_element IN square  
                SET prefix TO prefix PLUS element at position length of square of word_element  
            END FOR  
            
            FOR each candidate IN list at prefix_to_words at prefix  
                CALL backtrack with argument square PLUS list containing candidate  
            END FOR  
        END FUNCTION  
        
        FOR each word IN words  
            CALL backtrack with argument list containing word  
        END FOR  
        
        RETURN results  
    END FUNCTION  
END CLASS
