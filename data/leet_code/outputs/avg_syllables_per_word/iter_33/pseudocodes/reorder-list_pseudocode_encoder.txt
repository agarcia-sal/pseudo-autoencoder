CLASS ListNode  
    FUNCTION __init__ with parameters val DEFAULTING TO zero AND next DEFAULTING TO none  
        SET the attribute val TO val  
        SET the attribute next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the value of values EQUALS none OR the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET head TO a new ListNode initialized with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements starting from position two TO the end of values  
        SET node TO a new ListNode initialized with val  
        SET the attribute next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none THEN  
        RETURN False  
    END IF  
    RETURN the logical AND of the condition that the attribute val of p1 EQUALS the attribute val of p2 AND the result of calling is_same_list with parameters the attribute next of p1 AND the attribute next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList with parameters self AND head  
        IF head EQUALS none OR the attribute next of head EQUALS none THEN  
            RETURN  
        END IF  

        SET slow TO head  
        SET fast TO head  
        WHILE fast NOT EQUALS none AND the attribute next of fast NOT EQUALS none  
            SET slow TO the attribute next of slow  
            SET fast TO the attribute next of the attribute next of fast  
        END WHILE  

        SET prev TO none  
        SET curr TO slow  
        WHILE curr NOT EQUALS none  
            SET next_temp TO the attribute next of curr  
            SET the attribute next of curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  

        SET first TO head  
        SET second TO prev  
        WHILE the attribute next of second NOT EQUALS none  
            SET temp1 TO the attribute next of first  
            SET temp2 TO the attribute next of second  
            SET the attribute next of first TO second  
            SET the attribute next of second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
