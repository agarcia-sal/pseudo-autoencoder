CLASS Solution  
    FUNCTION maxDistance(arrays)  
        SET min_val TO element at position zero of element at position zero of arrays  
        SET max_val TO element at position zero of element at position equal to the length of element at position zero of arrays MINUS one of arrays  
        SET max_distance TO zero  
        FOR i FROM one TO the length of arrays MINUS one  
            SET current_array TO element at position i of arrays  
            SET current_min TO element at position zero of current_array  
            SET current_max TO element at position zero of current_array PLUS the length of current_array MINUS one  
            SET max_distance TO the greatest value among max_distance AND the absolute value of current_max MINUS min_val AND the absolute value of max_val MINUS current_min  
            IF min_val GREATER THAN current_min  
                SET min_val TO current_min  
            END IF  
            IF max_val LESS THAN current_max  
                SET max_val TO current_max  
            END IF  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
