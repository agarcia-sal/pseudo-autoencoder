CLASS Solution  
    FUNCTION leastBricks(wall)  
        SET gap_counts TO a new dictionary with default value zero for any key  
        FOR each row IN wall  
            SET total TO zero  
            FOR each brick IN the elements of row excluding the last element  
                SET total TO total PLUS the current brick  
                INCREMENT the value in gap_counts corresponding to key total BY one  
            END FOR  
        END FOR  
        SET max_gaps TO the maximum value among all values in gap_counts or zero if gap_counts is empty  
        RETURN the length of wall MINUS max_gaps  
    END FUNCTION  
END CLASS
