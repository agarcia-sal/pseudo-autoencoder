CLASS Solution  
    FUNCTION fractionToDecimal(numerator integer, denominator integer) RETURNS string  
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero  
            SET sign TO the string consisting of a single minus character  
        ELSE  
            SET sign TO the empty string  
        END IF  
        
        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  
        
        SET quotient TO the result of integer division of numerator BY denominator  
        SET remainder TO the remainder of numerator DIVIDED BY denominator  
        SET integer_part TO the string representation of quotient  
        
        IF remainder EQUALS zero  
            RETURN the concatenation of sign AND integer_part  
        END IF  
        
        SET seen_remainders TO an empty dictionary mapping remainders TO positions  
        SET fractional_part TO an empty list  
        
        WHILE remainder NOT EQUALS zero  
            IF remainder IS in the keys of seen_remainders  
                SET start TO the value associated to remainder in seen_remainders  
                SET non_repeating TO the concatenation of the elements from position zero UP TO position start EXCLUSIVE of fractional_part  
                SET repeating TO the concatenation of the elements FROM position start TO the end of fractional_part  
                RETURN the concatenation of sign, integer_part, a single dot character, non_repeating, a left parenthesis character, repeating, AND a right parenthesis character  
            END IF  
            
            ASSIGN the current length of fractional_part AS the value at remainder key in seen_remainders  
            
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO the result of integer division of remainder BY denominator  
            SET remainder TO the remainder of remainder DIVIDED BY denominator  
            APPEND the string representation of quotient TO fractional_part  
        END WHILE  
        
        RETURN the concatenation of sign, integer_part, a single dot character, AND the concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
