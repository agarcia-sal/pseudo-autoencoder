CLASS Solution  
    FUNCTION shortestDistance(grid)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN negative one  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        SET total_distance TO a list consisting of m elements each being a list consisting of n zeroes  
        SET reach_count TO a list consisting of m elements each being a list consisting of n zeroes  
        SET building_count TO zero  
        
        SET directions TO a list consisting of four elements each being a pair representing one of the following directions: one step down with zero steps sideways, one step up with zero steps sideways, zero steps down with one step rightwards, zero steps down with one step leftwards  
        
        FUNCTION bfs(start_x, start_y)  
            SET queue TO a deque initialized with a pair of start_x and start_y  
            SET visited TO a list consisting of m elements each being a list consisting of n elements each set to false  
            SET distance TO zero  
            SET reachable TO zero  
            
            WHILE the LENGTH OF queue is greater than zero  
                SET size TO the LENGTH OF queue  
                INCREMENT distance BY one  
                
                FOR each iteration number from zero TO size MINUS one  
                    REMOVE the element from the left of queue and ASSIGN its first component TO x and second component TO y  
                    
                    FOR each pair dx, dy IN directions  
                        SET nx TO x PLUS dx  
                        SET ny TO y PLUS dy  
                        
                        IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND element at position nx of visited at position ny EQUALS false AND element at position nx of grid at position ny EQUALS zero  
                            SET element at position nx of visited at position ny TO true  
                            APPEND the pair of nx and ny TO queue  
                            INCREMENT element at position nx of total_distance at position ny BY distance  
                            INCREMENT element at position nx of reach_count at position ny BY one  
                            INCREMENT reachable BY one  
                        END IF  
                    END FOR  
                END FOR  
            END WHILE  
            
            RETURN reachable  
        END FUNCTION  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    INCREMENT building_count BY one  
                    IF bfs(i, j) IS LESS THAN building_count MINUS one  
                        RETURN negative one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET min_distance TO a very large number representing infinity  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of grid at position j EQUALS zero AND element at position i of reach_count at position j EQUALS building_count  
                    SET min_distance TO the smaller value BETWEEN min_distance AND element at position i of total_distance at position j  
                END IF  
            END FOR  
        END FOR  
        
        IF min_distance IS EQUAL TO a very large number representing infinity  
            RETURN negative one  
        ELSE  
            RETURN min_distance  
        END IF  
    END FUNCTION  
END CLASS
