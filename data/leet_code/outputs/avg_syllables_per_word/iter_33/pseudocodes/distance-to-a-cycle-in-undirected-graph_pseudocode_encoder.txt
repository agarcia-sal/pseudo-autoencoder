CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO an empty default collection with list as default type  
        FOR each pair u AND v IN edges  
            APPEND v TO the list at key u IN graph  
            APPEND u TO the list at key v IN graph  
        END FOR  
        
        FUNCTION dfs(node, parent, path, visited)  
            IF node IS IN visited THEN  
                SET cycle_start TO the position of node IN path  
                RETURN the sublist of path FROM position cycle_start TO the end  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN the list at key node OF graph  
                IF neighbor IS NOT EQUAL TO parent THEN  
                    SET cycle TO dfs(neighbor, node, path, visited)  
                    IF cycle IS NOT null THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE the last element FROM path  
            RETURN null  
        END FUNCTION  
        
        SET visited TO an empty set  
        SET cycle TO null  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                SET cycle TO dfs(node, negative one, an empty list, visited)  
                IF cycle IS NOT null THEN  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET distance TO a list of zeros with length n  
        SET queue TO a dequeue initialized with all nodes in cycle  
        SET visited TO a set initialized with all nodes in cycle  
        
        SET level TO zero  
        WHILE queue IS NOT empty  
            FOR index FROM zero TO the length of queue MINUS one  
                SET node TO the element removed from the front of queue  
                SET the element at position node OF distance TO level  
                FOR each neighbor IN the list at key node OF graph  
                    IF neighbor IS NOT IN visited THEN  
                        ADD neighbor TO visited  
                        ADD neighbor TO the end of queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  
        
        RETURN distance  
    END FUNCTION  
END CLASS
