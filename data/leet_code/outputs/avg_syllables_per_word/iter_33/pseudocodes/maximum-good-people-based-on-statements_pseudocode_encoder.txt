CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO the NUMBER ONE LEFT SHIFTED MINUS one BY n TIMES  
            SET good TO a list of n elements each being false  
            SET bad TO a list of n elements each being false  
            FOR j FROM zero TO n MINUS one  
                IF the result of i BITWISE AND the NUMBER ONE LEFT SHIFTED BY j TIMES IS NOT EQUAL TO zero  
                    SET element at position j of good TO true  
                ELSE  
                    SET element at position j of bad TO true  
                END IF  
            END FOR  
            SET valid TO true  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of good IS true  
                    FOR k FROM zero TO n MINUS one  
                        IF (element at position k of element at position j of statements EQUALS zero AND element at position k of good IS true) OR (element at position k of element at position j of statements EQUALS one AND element at position k of bad IS true)  
                            SET valid TO false  
                            BREAK from the inner loop  
                        END IF  
                    END FOR  
                    IF valid IS false  
                        BREAK from the outer loop  
                    END IF  
                END IF  
            END FOR  
            IF valid IS true  
                SET count_good TO zero  
                FOR each value IN good  
                    IF value IS true  
                        INCREMENT count_good BY one  
                    END IF  
                END FOR  
                IF count_good GREATER THAN max_good  
                    SET max_good TO count_good  
                END IF  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
