CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF the string s EQUALS empty string  
                RETURN empty string  
            END IF  
            SET canonical_form TO an empty list  
            FOR each character c IN the string s  
                SET difference TO the ordinal value of the character c MINUS the ordinal value of the element at position one of the string s MODULO twenty six  
                APPEND the string representation of the difference TO canonical_form  
            END FOR  
            RETURN the concatenation of the elements in canonical_form SEPARATED BY a comma  
        END FUNCTION  
        
        SET groups TO a new dictionary with default value as an empty list  
        
        FOR each string s IN the list strings  
            SET canonical_form TO the result of calling get_canonical_form with the string s  
            APPEND the string s TO the value in groups for the key canonical_form  
        END FOR  
        
        RETURN the list of all values contained in groups  
    END FUNCTION  
END CLASS
