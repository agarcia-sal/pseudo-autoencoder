CLASS Solution  
    FUNCTION poorPigs WITH parameters buckets minutesToDie minutesToTest  
        SET states TO the INTEGER DIVISION of minutesToTest BY minutesToDie PLUS one  
        SET logarithm_of_buckets TO the LOGARITHM of buckets  
        SET logarithm_of_states TO the LOGARITHM of states  
        SET pigs TO the SMALLEST INTEGER GREATER THAN OR EQUAL TO the DIVISION of logarithm_of_buckets BY logarithm_of_states  
        RETURN pigs  
    END FUNCTION  
END CLASS
