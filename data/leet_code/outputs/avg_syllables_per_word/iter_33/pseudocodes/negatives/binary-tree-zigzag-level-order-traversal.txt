CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    ASSIGN root TO A NEW INSTANCE OF TreeNode WITH the element at position one of values AS val  
    SET i TO one  
    ASSIGN queue TO AN EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        REMOVE the FIRST element FROM queue AND ASSIGN it TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            ASSIGN node.left TO A NEW INSTANCE OF TreeNode WITH the element at position i of values AS val  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            ASSIGN node.right TO A NEW INSTANCE OF TreeNode WITH the element at position i of values AS val  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the val ATTRIBUTE OF p NOT EQUALS the val ATTRIBUTE OF q  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH p.left AND q.left AND AND the RESULT OF is_same_tree WITH p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self, root)  
        IF root IS none  
            RETURN AN EMPTY LIST  
        END IF  
        ASSIGN result TO AN EMPTY LIST  
        ASSIGN queue TO A DOUBLE ENDED QUEUE CONTAINING root  
        SET left_to_right TO true  
        WHILE the LENGTH OF queue GREATER THAN zero  
            ASSIGN level_size TO the LENGTH OF queue  
            ASSIGN current_level TO AN EMPTY DOUBLE ENDED QUEUE  
            FOR counter FROM one TO level_size  
                REMOVE the FIRST element FROM queue AND ASSIGN it TO node  
                IF left_to_right IS true  
                    APPEND the val ATTRIBUTE OF node TO current_level  
                ELSE  
                    INSERT the val ATTRIBUTE OF node AT THE BEGINNING OF current_level  
                END IF  
                IF the left ATTRIBUTE OF node IS NOT none  
                    APPEND the left ATTRIBUTE OF node TO queue  
                END IF  
                IF the right ATTRIBUTE OF node IS NOT none  
                    APPEND the right ATTRIBUTE OF node TO queue  
                END IF  
            END FOR  
            APPEND A LIST CONVERTED FROM current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
