CLASS Solution  
    FUNCTION numberOfWays WITH PARAMETER corridor  
        SET MOD TO ten RAISED TO the power OF nine PLUS one  
        SET seats TO empty list  
        FOR each position_and_character IN corridor WITH index position  
            IF character AT position EQUALS seat character  
                APPEND position TO seats  
            END IF  
        END FOR  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM three TO the LENGTH OF seats WITH STEP two  
            SET difference TO the element AT position index OF seats MINUS the element AT position index MINUS one OF seats  
            SET ways TO ways MULTIPLIED BY difference MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
