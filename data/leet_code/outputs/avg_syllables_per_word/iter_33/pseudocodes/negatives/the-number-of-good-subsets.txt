CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of elements in nums  
        SET dp TO a list of zeroes with length two EXPONENTIATED BY ten  
        SET element at position zero of dp TO one  
        
        FOR each num IN the keys of count  
            IF num EQUALS one  
                SET element at position zero of dp TO the element at position zero of dp MULTIPLIED BY two EXPONENTIATED BY the element at position one in count  
                CONTINUE to the next iteration  
            END IF  
            IF num NOT IN prime_factors  
                CONTINUE to the next iteration  
            END IF  
            
            SET factors TO the value in prime_factors associated with num  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask PLUS two EXPONENTIATED BY the quantity of prime MINUS one  
            END FOR  
            
            FOR m FROM two EXPONENTIATED BY ten MINUS one DOWNTO zero  
                IF the bitwise AND of m AND mask EQUALS zero  
                    SET element at position bitwise OR of m AND mask of dp TO the element at position bitwise OR of m AND mask of dp PLUS the element at position m of dp MULTIPLIED BY the element in count associated with num  
                    SET element at position bitwise OR of m AND mask of dp TO element at position bitwise OR of m AND mask of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the summation of elements of dp from position one to the end  
        SET result TO result MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
