CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with value element at position one of values  
    SET i TO one plus one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element obtained by removing from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET node.left TO a new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET node.right TO a new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value  
        RETURN True  
    ELSE IF p IS no value OR q IS no value  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION recoverTree(self, root)  
        SET first TO no value  
        SET second TO no value  
        SET prev TO no value  
        SET curr TO root  
        WHILE curr IS NOT no value  
            IF curr.left IS no value  
                IF prev IS NOT no value AND prev.val GREATER THAN curr.val  
                    IF first IS no value  
                        SET first TO prev  
                    END IF  
                    SET second TO curr  
                END IF  
                SET prev TO curr  
                SET curr TO curr.right  
            ELSE  
                SET pred TO curr.left  
                WHILE pred.right IS NOT no value AND pred.right NOT EQUALS curr  
                    SET pred TO pred.right  
                END WHILE  
                IF pred.right IS no value  
                    SET pred.right TO curr  
                    SET curr TO curr.left  
                ELSE  
                    SET pred.right TO no value  
                    IF prev IS NOT no value AND prev.val GREATER THAN curr.val  
                        IF first IS no value  
                            SET first TO prev  
                        END IF  
                        SET second TO curr  
                    END IF  
                    SET prev TO curr  
                    SET curr TO curr.right  
                END IF  
            END IF  
        END WHILE  
        IF first IS NOT no value AND second IS NOT no value  
            SET temporary TO first.val  
            SET first.val TO second.val  
            SET second.val TO temporary  
        END IF  
    END FUNCTION  
END CLASS
