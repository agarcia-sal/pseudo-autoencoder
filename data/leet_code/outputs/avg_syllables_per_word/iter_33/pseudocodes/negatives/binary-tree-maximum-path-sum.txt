CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values as val  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.left TO a new TreeNode with the element at position i of values as val  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.right TO a new TreeNode with the element at position i of values as val  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p.left and q.left AS arguments AND the result of is_same_tree with p.right and q.right AS arguments  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum(self, root)  
        FUNCTION helper(node)  
            IF node IS None THEN  
                RETURN negative infinity, negative infinity  
            END IF  
            SET left_max, left_gain TO the result of helper with node.left AS argument  
            SET right_max, right_gain TO the result of helper with node.right AS argument  
            SET current_max TO the val attribute of node PLUS the greater of left_gain AND zero PLUS the greater of right_gain AND zero  
            SET current_gain TO the val attribute of node PLUS the greatest of left_gain right_gain AND zero  
            SET global_max TO the greatest of left_max right_max AND current_max  
            RETURN global_max, current_gain  
        END FUNCTION  
        RETURN the first element of the result of helper with root AS argument  
    END FUNCTION  
END CLASS
