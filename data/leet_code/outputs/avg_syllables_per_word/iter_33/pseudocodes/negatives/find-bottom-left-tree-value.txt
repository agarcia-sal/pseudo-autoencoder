CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the collection values DOES NOT EXIST OR the LENGTH OF values EQUALS zero  
        RETURN nothing  
    END IF  
    SET root TO a new instance of TreeNode WITH the element at position one of values AS val  
    SET i TO one  
    CREATE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        REMOVE the first element FROM queue AND ASSIGN it TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values EXISTS AND the element at position i of values IS NOT nothing  
            SET node.left TO a new instance of TreeNode WITH the element at position i of values AS val  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values EXISTS AND the element at position i of values IS NOT nothing  
            SET node.right TO a new instance of TreeNode WITH the element at position i of values AS val  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p DOES NOT EXIST AND q DOES NOT EXIST  
        RETURN True  
    ELSE IF p DOES NOT EXIST OR q DOES NOT EXIST  
        RETURN False  
    ELSE IF the value attribute of p IS NOT EQUAL TO the value attribute of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self, root)  
        CREATE a double ended queue named queue WITH root AS its only element  
        SET leftmost_value TO nothing  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one  
                REMOVE the first element FROM queue AND ASSIGN it TO node  
                IF i EQUALS zero  
                    SET leftmost_value TO the value attribute of node  
                END IF  
                IF the left attribute of node EXISTS  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node EXISTS  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
