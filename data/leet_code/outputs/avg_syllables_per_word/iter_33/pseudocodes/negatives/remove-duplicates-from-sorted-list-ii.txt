CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO NEW ListNode with element at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO the end of values  
        SET node TO NEW ListNode with val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the val of p1 EQUALS the val of p2 AND the result of is_same_list with the next of p1 and the next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates(self head)  
        SET dummy TO NEW ListNode with zero and head  
        SET prev TO dummy  
        WHILE head IS NOT None  
            IF the next of head IS NOT None AND the val of head EQUALS the val of the next of head  
                WHILE the next of head IS NOT None AND the val of head EQUALS the val of the next of head  
                    SET head TO the next of head  
                END WHILE  
                SET the next of prev TO the next of head  
            ELSE  
                SET prev TO the next of prev  
            END IF  
            SET head TO the next of head  
        END WHILE  
        RETURN the next of dummy  
    END FUNCTION  
END CLASS
