CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values  
    SET i TO one  
    CREATE an empty queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the first element removed from the queue  
        IF i LESS THAN the length of values AND the element at position i of values NOT EQUALS none  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values NOT EQUALS none  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of is_same_tree with the left child of p and the left child of q AND is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION printTree(self, root)  
        FUNCTION getHeight(node)  
            IF node EQUALS none  
                RETURN zero  
            END IF  
            RETURN one PLUS the maximum of getHeight with the left child of node AND getHeight with the right child of node  
        END FUNCTION  
        SET height TO getHeight of root  
        SET width TO two RAISED TO THE POWER OF height MINUS one  
        SET res TO a list containing height elements EACH being a list containing width elements EACH being an empty string  
        FUNCTION placeNode(node, row, left, right)  
            IF node EQUALS none  
                RETURN  
            END IF  
            SET mid TO the integer division of the sum of left and right BY two  
            SET the element at position mid of the list at position row of res TO the string representation of the value of node  
            CALL placeNode with the left child of node, row PLUS one, left, mid  
            CALL placeNode with the right child of node, row PLUS one, mid, right  
        END FUNCTION  
        CALL placeNode with root, zero, zero, width  
        RETURN res  
    END FUNCTION  
END CLASS
