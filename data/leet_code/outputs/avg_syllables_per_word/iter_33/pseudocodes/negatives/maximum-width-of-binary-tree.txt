CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with value element at position one of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element removed from queue  
        IF i LESS THAN the length of values AND element at position i of values is not none  
            SET node.left TO a new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values is not none  
            SET node.right TO a new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree(self, root)  
        IF root is none  
            RETURN zero  
        END IF  
        SET queue TO a deque containing the tuple of root and zero  
        SET max_width TO zero  
        WHILE queue is not empty  
            SET level_length TO the length of queue  
            SET first_index TO the second element of the first tuple in queue  
            FOR counter FROM one TO level_length  
                SET node, index TO the first tuple removed from queue  
                IF counter EQUALS level_length  
                    IF index MINUS first_index PLUS one GREATER THAN max_width  
                        SET max_width TO index MINUS first_index PLUS one  
                    END IF  
                END IF  
                IF node.left is not none  
                    APPEND the tuple of node.left and two MULTIPLIED BY index TO queue  
                END IF  
                IF node.right is not none  
                    APPEND the tuple of node.right and two MULTIPLIED BY index PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
