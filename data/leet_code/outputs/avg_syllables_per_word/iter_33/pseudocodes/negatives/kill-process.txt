CLASS Solution  
    FUNCTION killProcess(pid AS list of integers ppid AS list of integers kill AS integer) RETURNS list of integers  
        SET tree TO a mapping from integer keys to list of integers with default empty list  
        FOR each pair of child and parent IN the parallel elements of pid and ppid  
            APPEND child TO the list corresponding to parent in tree  
        END FOR  
        SET queue TO a list containing only the integer kill  
        SET killed_processes TO an empty list  
        WHILE the list queue is not empty  
            SET current TO the element at the first position of queue  
            REMOVE the element at the first position from queue  
            APPEND current TO killed_processes  
            FOR each child IN the list corresponding to current key in tree  
                APPEND child TO queue  
            END FOR  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
