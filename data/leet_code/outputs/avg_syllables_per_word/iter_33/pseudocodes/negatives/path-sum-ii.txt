CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF the list values is empty THEN  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    INITIALIZE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE the length of queue is greater than zero DO  
        REMOVE the element at the front of queue and assign it TO node  
        IF i is less than the length of values AND the element at position i of values is not none THEN  
            SET the left attribute of node TO a new TreeNode with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not none THEN  
            SET the right attribute of node TO a new TreeNode with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p is none AND q is none THEN  
        RETURN true  
    ELSE IF p is none OR q is none THEN  
        RETURN false  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION dfs(node, current_sum, path)  
            IF node is none THEN  
                RETURN  
            END IF  
            APPEND the value attribute of node TO path  
            INCREMENT current_sum BY the value attribute of node  
            IF the left attribute of node is none AND the right attribute of node is none AND current_sum EQUALS targetSum THEN  
                APPEND a copy of path TO result  
            END IF  
            CALL dfs WITH the left attribute of node, current_sum, and path  
            CALL dfs WITH the right attribute of node, current_sum, and path  
            REMOVE the last element FROM path  
        END FUNCTION  
        SET result TO an empty list  
        CALL dfs WITH root, zero, and an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
