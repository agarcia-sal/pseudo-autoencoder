CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value the element at position one of values and left None and right None  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode with value the element at position i of values and left None and right None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode with value the element at position i of values and left None and right None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters the left attribute of p and the left attribute of q AND the result of is_same_tree with parameters the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom(self, root)  
        IF root IS None  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a double-ended queue initialized with root  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR each element in the range from one TO level_size inclusive  
                SET node TO the element removed from the left side of queue  
                APPEND the val attribute of node TO current_level  
                IF the left attribute of node IS NOT None  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node IS NOT None  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result in reverse order  
    END FUNCTION  
END CLASS
