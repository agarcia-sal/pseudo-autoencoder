CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the end of values  
        SET node TO a new ListNode with val  
        SET the next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND is_same_list of the next of p1 AND the next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne(self, head)  
        FUNCTION reverseList(node)  
            SET prev TO None  
            WHILE node IS NOT None  
                SET next_node TO the next of node  
                SET the next of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO reverseList of head  
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current IS NOT None AND carry EQUALS one  
            SET the value of current TO the value of current PLUS carry  
            SET carry TO the value of current DIVIDED BY ten  
            SET the value of current TO the value of current MODULO ten  
            
            IF the next of current EQUALS None AND carry EQUALS one  
                SET the next of current TO a new ListNode with carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO the next of current  
        END WHILE  
        
        RETURN reverseList of reversed_head  
    END FUNCTION  
END CLASS
