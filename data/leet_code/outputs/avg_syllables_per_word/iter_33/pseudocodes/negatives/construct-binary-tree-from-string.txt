CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with value being element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values from position two TO the last element  
        SET node TO new ListNode with value val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the comparison of p1.val EQUALS p2.val AND is_same_list with parameters p1.next AND p2.next  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value being element at position one of values  
    SET i TO one  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the first element removed from queue  
        IF i LESS THAN the length of values AND the element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters p.left AND q.left AND is_same_tree with parameters p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO the length of s  
                RETURN None AND index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the length of s AND (the element at position index of s IS a digit OR the element at position index of s EQUALS the minus character)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO new TreeNode with value the integer parsed from the substring of s from position start TO position index MINUS one  
            IF index LESS THAN the length of s AND the element at position index of s EQUALS the opening parenthesis character  
                SET node.left AND index TO the values returned by parse_tree with parameter index PLUS one  
            END IF  
            IF index LESS THAN the length of s AND the element at position index of s EQUALS the opening parenthesis character  
                SET node.right AND index TO the values returned by parse_tree with parameter index PLUS one  
            END IF  
            IF index LESS THAN the length of s AND the element at position index of s EQUALS the closing parenthesis character  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND _ TO parse_tree with parameter zero  
        RETURN root  
    END FUNCTION  
END CLASS
