CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        
        SET dp TO a list of m PLUS one elements  
        FOR index_i FROM zero TO m  
            SET row TO a list of n PLUS one elements each initialized TO negative infinity  
            ASSIGN row TO the element at position index_i of dp  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO element at position i MINUS one of nums1 MULTIPLIED BY element at position j MINUS one of nums2  
                
                SET max_value TO the greatest value among current_product  
                AND element at position i MINUS one of dp at position j  
                AND element at position i of dp at position j MINUS one  
                AND element at position i MINUS one of dp at position j MINUS one PLUS current_product  
                
                SET element at position i of dp at position j TO max_value  
            END FOR  
        END FOR  
        
        RETURN element at position m of dp at position n  
    END FUNCTION  
END CLASS
