CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO the SORTED LIST of PAIRS of WAGE element DIVIDED BY QUALITY element AND QUALITY element FOR each WAGE element AND QUALITY element FROM ZIP of wage AND quality  
        SET max_heap TO an EMPTY LIST  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each ratio AND q IN workers  
            PUSH the NEGATIVE of q INTO max_heap  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap GREATER THAN k THEN  
                INCREMENT total_quality BY the ELEMENT REMOVED FROM max_heap VIA HEAP POP  
            END IF  
            IF the LENGTH OF max_heap EQUALS k THEN  
                SET min_cost TO the MINIMUM OF min_cost AND total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
