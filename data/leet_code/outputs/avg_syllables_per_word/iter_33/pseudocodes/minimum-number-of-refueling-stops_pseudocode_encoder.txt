CLASS Solution  
    FUNCTION minRefuelStops(target startFuel stations)  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a list containing target AND zero TO stations  

        FOR each element containing position AND fuel IN stations  
            WHILE the LENGTH OF max_heap is greater than zero AND current_fuel is LESS THAN position  
                SET current_fuel TO current_fuel MINUS the removal of the smallest element from max_heap  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel is LESS THAN position  
                RETURN the negative one  
            END IF  
            ADD the negative of fuel TO max_heap  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
