CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET max_so_far TO an empty list  
        SET current_max TO element at position one of arr  
        FOR each num IN arr  
            IF current_max GREATER THAN OR EQUAL TO num  
                SET current_max TO current_max  
            ELSE  
                SET current_max TO num  
            END IF  
            APPEND current_max TO max_so_far  
        END FOR  
        SET sorted_arr TO arr sorted in ascending order  
        SET chunks TO zero  
        SET max_chunk TO zero  
        FOR i FROM one TO the LENGTH OF arr  
            IF max_chunk GREATER THAN OR EQUAL TO element at position i of arr  
                SET max_chunk TO max_chunk  
            ELSE  
                SET max_chunk TO element at position i of arr  
            END IF  
            IF max_chunk EQUALS element at position i of sorted_arr  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        RETURN chunks  
    END FUNCTION  
END CLASS
