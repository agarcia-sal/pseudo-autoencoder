CLASS Solution  
    FUNCTION isRectangleCover(rectangles)  
        SET X1 TO positive infinity  
        SET Y1 TO positive infinity  
        SET X2 TO negative infinity  
        SET Y2 TO negative infinity  
        
        SET points TO an empty set  
        SET area TO zero  
        
        FOR each rectangle WITH elements x1 y1 x2 y2 IN rectangles  
            IF X1 GREATER THAN x1 THEN  
                SET X1 TO x1  
            END IF  
            IF Y1 GREATER THAN y1 THEN  
                SET Y1 TO y1  
            END IF  
            IF X2 LESS THAN x2 THEN  
                SET X2 TO x2  
            END IF  
            IF Y2 LESS THAN y2 THEN  
                SET Y2 TO y2  
            END IF  
            
            INCREMENT area BY the result of the subtraction of x1 FROM x2 MULTIPLIED BY the result of the subtraction of y1 FROM y2  
            
            FOR each point p IN the list containing tuple of x1 y1 tuple of x1 y2 tuple of x2 y1 and tuple of x2 y2  
                IF point p IS IN points THEN  
                    REMOVE point p FROM points  
                ELSE  
                    ADD point p TO points  
                END IF  
            END FOR  
        END FOR  
        
        IF area NOT EQUALS the result of the subtraction of X1 FROM X2 MULTIPLIED BY the result of the subtraction of Y1 FROM Y2 THEN  
            RETURN false  
        END IF  
        
        IF points NOT EQUALS the set containing tuples of X1 Y1 X1 Y2 X2 Y1 and X2 Y2 THEN  
            RETURN false  
        END IF  
        
        RETURN true  
    END FUNCTION  
END CLASS
