CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO one billion PLUS seven  
        SET dp TO list of three elements each set to zero  
        FOR each num IN nums  
            IF num EQUALS zero  
                SET element at position zero of dp TO element at position zero of dp MULTIPLIED BY two PLUS one MODULO MOD  
            ELSE IF num EQUALS one  
                SET element at position one of dp TO element at position one of dp MULTIPLIED BY two PLUS element at position zero of dp MODULO MOD  
            ELSE IF num EQUALS two  
                SET element at position two of dp TO element at position two of dp MULTIPLIED BY two PLUS element at position one of dp MODULO MOD  
            END IF  
        END FOR  
        RETURN element at position two of dp  
    END FUNCTION  
END CLASS
