CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value being element at position zero of values  
    SET i TO one  
    SET queue TO new deque collection  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None THEN  
            SET node.left TO new TreeNode with value being element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None THEN  
            SET node.right TO new TreeNode with value being element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters p.left and q.left AND is_same_tree with parameters p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION helper(node, current_sum, prefix_sums)  
            IF node EQUALS None THEN  
                RETURN zero  
            END IF  
            SET current_sum TO current_sum PLUS node.val  
            SET paths_to_here TO the value associated with key current_sum MINUS targetSum in prefix_sums IF EXISTS OTHERWISE zero  
            SET prefix_sums at key current_sum TO the value associated with key current_sum in prefix_sums IF EXISTS OTHERWISE zero PLUS one  
            SET paths_in_left TO the result of calling helper with parameters node.left, current_sum, prefix_sums  
            SET paths_in_right TO the result of calling helper with parameters node.right, current_sum, prefix_sums  
            DECREMENT the value associated with key current_sum in prefix_sums BY one  
            RETURN paths_to_here PLUS paths_in_left PLUS paths_in_right  
        END FUNCTION  
        SET prefix_sums TO a new mapping with key zero associated with value one  
        RETURN the result of calling helper with parameters root, zero, prefix_sums  
    END FUNCTION  
END CLASS
