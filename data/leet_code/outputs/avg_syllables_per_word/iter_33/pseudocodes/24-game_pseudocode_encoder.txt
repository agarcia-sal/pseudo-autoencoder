CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the ELEMENT at position zero of nums MINUS twenty four LESS THAN one divided BY ten to the power of six  
            END IF  
            
            FOR i FROM zero TO the LENGTH OF nums MINUS one  
                FOR j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO an EMPTY list  
                        FOR k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND the ELEMENT at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR each operation IN the list of division operation and multiplication operation and addition operation and subtraction operation  
                            IF operation IS addition operation OR operation IS multiplication operation THEN  
                                IF j GREATER THAN i THEN  
                                    CONTINUE to next operation  
                                END IF  
                            END IF  
                            
                            IF operation IS NOT division operation OR the ELEMENT at position j of nums NOT EQUALS zero THEN  
                                APPEND the RESULT of APPLYING operation TO the ELEMENT at position i of nums AND the ELEMENT at position j of nums TO new_nums  
                                IF solve(new_nums) RETURNS true THEN  
                                    RETURN true  
                                END IF  
                                REMOVE the LAST element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
