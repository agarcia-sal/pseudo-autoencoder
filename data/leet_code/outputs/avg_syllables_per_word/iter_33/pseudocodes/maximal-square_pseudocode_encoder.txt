CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET dp TO a collection of zeroes with m plus one rows each containing n plus one zeroes  
        SET max_side TO zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i minus one of matrix AND element at position j minus one of that element EQUALS the character one  
                    SET dp at position i and j TO the minimum value among dp at position i minus one and j AND dp at position i and j minus one AND dp at position i minus one and j minus one PLUS one  
                    SET max_side TO the maximum value between max_side AND dp at position i and j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
