CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val NEXT  
        SET the property val TO val  
        SET the property next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF the list values EQUALS empty  
        RETURN nothing  
    END IF  
    SET head TO a new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position two TO the end  
        SET node TO a new ListNode with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 EQUALS nothing AND p2 EQUALS nothing  
        RETURN true  
    END IF  
    IF p1 EQUALS nothing OR p2 EQUALS nothing  
        RETURN false  
    END IF  
    RETURN the val property of p1 EQUALS the val property of p2 AND is_same_list WITH PARAMETERS the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers WITH PARAMETERS l1 l2  
        FUNCTION pushIntoStack WITH PARAMETER node  
            SET stack TO an empty list  
            WHILE node is not nothing  
                APPEND the val property of node TO stack  
                SET node TO the next property of node  
            END WHILE  
            RETURN stack  
        END FUNCTION  

        SET stack1 TO pushIntoStack WITH PARAMETER l1  
        SET stack2 TO pushIntoStack WITH PARAMETER l2  

        SET carry TO zero  
        SET result TO nothing  

        WHILE the LENGTH OF stack1 IS GREATER THAN zero OR the LENGTH OF stack2 IS GREATER THAN zero OR carry IS GREATER THAN zero  
            IF the LENGTH OF stack1 IS GREATER THAN zero  
                SET val1 TO the last element of stack1 and REMOVE that element FROM stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF the LENGTH OF stack2 IS GREATER THAN zero  
                SET val2 TO the last element of stack2 and REMOVE that element FROM stack2  
            ELSE  
                SET val2 TO zero  
            END IF  

            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  

            SET new_node TO a new ListNode with digit  
            SET the next property of new_node TO result  
            SET result TO new_node  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
