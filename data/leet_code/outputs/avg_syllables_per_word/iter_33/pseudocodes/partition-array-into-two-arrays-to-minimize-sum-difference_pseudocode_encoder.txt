CLASS Solution
    FUNCTION minimumDifference(nums)
        SET n TO the LENGTH OF nums DIVIDED BY two
        SET total_sum TO the SUM OF all elements IN nums
        SET target TO total_sum DIVIDED BY two
        SET min_diff TO positive infinity
        
        SET left_half TO the elements from position one TO position n of nums
        SET right_half TO the elements from position n PLUS one TO the end of nums
        
        SET left_sums TO a list of empty dictionaries with n PLUS one elements
        SET right_sums TO a list of empty dictionaries with n PLUS one elements
        
        FOR i FROM zero TO n
            FOR each combination combo of i elements selected FROM left_half
                SET subset_sum TO the SUM OF elements IN combo
                IF subset_sum NOT IN left_sums at position i
                    SET left_sums at position i with key subset_sum TO zero
                END IF
                INCREMENT the value at left_sums at position i with key subset_sum BY one
            END FOR
            
            FOR each combination combo of i elements selected FROM right_half
                SET subset_sum TO the SUM OF elements IN combo
                IF subset_sum NOT IN right_sums at position i
                    SET right_sums at position i with key subset_sum TO zero
                END IF
                INCREMENT the value at right_sums at position i with key subset_sum BY one
            END FOR
        END FOR
        
        FOR i FROM zero TO n
            SET left_values TO the sorted list of keys FROM left_sums at position i
            SET right_values TO the sorted list of keys FROM right_sums at position n MINUS i
            
            SET j TO zero
            SET k TO the LENGTH OF right_values MINUS one
            WHILE j LESS THAN the LENGTH OF left_values AND k GREATER THAN OR EQUAL TO zero
                SET current_sum TO the element at position j of left_values PLUS the element at position k of right_values
                SET candidate_diff TO total_sum MINUS two MULTIPLIED BY current_sum
                SET candidate_diff TO the absolute value of candidate_diff
                IF candidate_diff LESS THAN min_diff
                    SET min_diff TO candidate_diff
                END IF
                
                IF current_sum LESS THAN target
                    INCREMENT j BY one
                ELSE IF current_sum GREATER THAN target
                    DECREMENT k BY one
                ELSE
                    RETURN zero
                END IF
            END WHILE
        END FOR
        
        RETURN min_diff
    END FUNCTION
END CLASS
