CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        CREATE a dictionary cost to digit mapping each element in cost to the string representation of its one based index  
        SET dp TO a list starting with an empty string followed by target number of elements each set to a string representing negative one  
        FOR t FROM one TO target  
            FOR each cost value c and corresponding digit string d in the dictionary cost to digit  
                IF t is greater than or equal to c AND element at position t MINUS c of dp does NOT EQUAL the string representing negative one  
                    SET candidate TO concatenation of d followed by element at position t MINUS c of dp  
                    IF element at position t of dp EQUALS the string representing negative one OR LENGTH OF candidate greater than LENGTH OF element at position t of dp OR (LENGTH OF candidate EQUALS LENGTH OF element at position t of dp AND candidate lexicographically greater than element at position t of dp)  
                        SET element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        IF element at position target of dp NOT EQUALS the string representing negative one  
            RETURN element at position target of dp  
        ELSE  
            RETURN the string representing zero  
        END IF  
    END FUNCTION  
END CLASS
