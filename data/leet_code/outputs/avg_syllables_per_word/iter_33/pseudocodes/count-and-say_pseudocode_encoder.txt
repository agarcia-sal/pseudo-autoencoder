CLASS Solution  
    FUNCTION countAndSay WITH PARAMETER n  
        IF n EQUALS one  
            RETURN the character one expressed as string  
        END IF  
        SET current_sequence TO the character one expressed as string  
        FOR each integer underscore FROM one TO n MINUS one  
            SET next_sequence TO the empty string  
            SET count TO one  
            SET previous_char TO the element at position zero of current_sequence  
            FOR each integer i FROM one TO the length of current_sequence MINUS one  
                IF the element at position i of current_sequence EQUALS previous_char  
                    INCREMENT count BY one  
                ELSE  
                    APPEND the string representation of count TO next_sequence  
                    APPEND previous_char TO next_sequence  
                    SET previous_char TO the element at position i of current_sequence  
                    SET count TO one  
                END IF  
            END FOR  
            APPEND the string representation of count TO next_sequence  
            APPEND previous_char TO next_sequence  
            SET current_sequence TO next_sequence  
        END FOR  
        RETURN current_sequence  
    END FUNCTION  
END CLASS
