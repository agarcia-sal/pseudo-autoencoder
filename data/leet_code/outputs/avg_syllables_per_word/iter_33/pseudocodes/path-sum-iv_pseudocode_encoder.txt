CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty collection of key value pairs  
        FOR each num IN nums  
            SET depth TO the integer division of num BY one hundred  
            SET pos TO the integer division of num BY ten MODULO ten  
            SET value TO num MODULO ten  
            ASSIGN value TO the element at key composed of depth and pos IN tree  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the element at key composed of depth and pos IN tree  
            IF current_value EQUALS nothing  
                RETURN zero  
            END IF  
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child TO a pair composed of depth PLUS one and pos MULTIPLIED BY two MINUS one  
            SET right_child TO a pair composed of depth PLUS one and pos MULTIPLIED BY two  
            IF the element at key left_child IN tree EQUALS nothing AND the element at key right_child IN tree EQUALS nothing  
                RETURN new_path_sum  
            END IF  
            SET left_sum TO the result of calling dfs WITH the first element of left_child the second element of left_child and new_path_sum  
            SET right_sum TO the result of calling dfs WITH the first element of right_child the second element of right_child and new_path_sum  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN the result of calling dfs WITH one one and zero  
    END FUNCTION  
END CLASS
