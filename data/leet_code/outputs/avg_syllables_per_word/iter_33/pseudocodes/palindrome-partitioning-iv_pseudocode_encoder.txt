CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS reversed sub  
        END FUNCTION  
        
        SET n TO the LENGTH OF s  
        
        SET is_pal TO a two dimensional list of Boolean FALSE values with dimensions n BY n  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF element at position i of s EQUALS element at position j of s AND (j MINUS i LESS THAN OR EQUAL TO one OR element at position i PLUS one and position j MINUS one of is_pal IS TRUE)  
                    SET element at position i and position j of is_pal TO TRUE  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM one TO n MINUS two  
            IF element at position zero and position i MINUS one of is_pal IS TRUE  
                FOR j FROM i TO n MINUS two  
                    IF element at position i and position j of is_pal IS TRUE AND element at position j PLUS one and position n MINUS one of is_pal IS TRUE  
                        RETURN TRUE  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN FALSE  
    END FUNCTION  
END CLASS
