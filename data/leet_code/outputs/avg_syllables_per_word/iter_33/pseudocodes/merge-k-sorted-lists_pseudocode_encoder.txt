CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS nothing THEN  
        RETURN nothing  
    END IF  
    SET head TO A new ListNode with value EQUALS element at position one of values  
    SET p TO head  
    FOR each val IN all elements of values starting from position two to the end  
        SET node TO A new ListNode with value EQUALS val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS nothing AND p2 EQUALS nothing THEN  
        RETURN true  
    END IF  
    IF p1 EQUALS nothing OR p2 EQUALS nothing THEN  
        RETURN false  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND RESULT OF is_same_list with parameters the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists(self lists)  
        SET min_heap TO an empty list  
        FOR each index i AND list lst IN lists with enumeration  
            IF lst IS NOT nothing THEN  
                ADD the tuple consisting of the value of lst AND i AND lst TO min_heap using heap push operation  
            END IF  
        END FOR  
        SET dummy TO A new ListNode  
        SET current TO dummy  
        WHILE min_heap IS NOT empty DO  
            REMOVE the smallest tuple of value val index i AND node FROM min_heap using heap pop operation  
            SET the next element of current TO node  
            SET current TO the next element of current  
            IF the next element of node IS NOT nothing THEN  
                ADD the tuple consisting of the value of the next element of node AND i AND the next element of node TO min_heap using heap push operation  
            END IF  
        END WHILE  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
