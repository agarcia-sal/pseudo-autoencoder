CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator_of_t TO iterator of t  
            RETURN whether for every character_in_s the character_in_s is found in iterator_of_t in order  
        END FUNCTION  
        
        SORT strs by the negative length of each element PLUS lexicographical order of each element  
        
        FOR each index_i AND word IN enumeration of strs  
            SET unique TO true  
            FOR each index_j AND other_word IN enumeration of strs  
                IF index_i NOT EQUALS index_j AND is_subsequence(word, other_word) THEN  
                    SET unique TO false  
                    BREAK inner loop  
                END IF  
            END FOR  
            IF unique THEN  
                RETURN the length of word  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
