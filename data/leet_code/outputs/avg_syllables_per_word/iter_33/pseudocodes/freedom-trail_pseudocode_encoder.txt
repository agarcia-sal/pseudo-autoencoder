CLASS Solution  
    FUNCTION findRotateSteps(ring, key)  
        ASSIGN reference TO a new dictionary that assigns an empty list for each new key in char_positions  
        FOR each index i AND character char IN ring  
            APPEND i TO the list at key char in char_positions  
        END FOR  
        SET n TO the LENGTH OF ring  
        
        FUNCTION dp(i, prev_pos)  
            IF i EQUALS the LENGTH OF key THEN  
                RETURN zero  
            END IF  
            
            SET min_steps TO infinity  
            FOR each position pos IN the list at key at position i in key in char_positions  
                SET clockwise TO the remainder of the subtraction of prev_pos FROM pos DIVIDED BY n  
                SET anticlockwise TO the remainder of the subtraction of pos FROM prev_pos DIVIDED BY n  
                SET steps TO the minimum of clockwise AND anticlockwise PLUS one PLUS the result of dp with parameters i PLUS one AND pos  
                SET min_steps TO the minimum of min_steps AND steps  
            END FOR  
            
            RETURN min_steps  
        END FUNCTION  
        
        RETURN dp with parameters zero AND zero  
    END FUNCTION  
END CLASS
