CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR i FROM zero TO eight  
                IF element at position i of row at position row of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            FOR i FROM zero TO eight  
                IF element at position col of row at position i of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY integer division of row BY three  
            SET start_col TO three MULTIPLIED BY integer division of col BY three  
            FOR i FROM zero TO two  
                FOR j FROM zero TO two  
                    IF element at position start_col PLUS j of row at position start_row PLUS i of board EQUALS num  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
        FUNCTION solve(board)  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF element at position col of row at position row of board EQUALS the single character dot  
                        FOR num IN the string representations of numbers from one TO nine  
                            IF is_valid(board, row, col, num)  
                                SET element at position col of row at position row of board TO num  
                                IF solve(board) HOLDS True  
                                    RETURN True  
                                END IF  
                                SET element at position col of row at position row of board TO the single character dot  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
        CALL solve(board)  
    END FUNCTION  
END CLASS
