CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        
        SET last TO a list of negative one REPEATED twenty six TIMES  
        SET last2 TO a list of negative one REPEATED twenty six TIMES  
        
        SET result TO zero  
        
        FOR each index i AND character char IN s  
            SET idx TO the position of char in the uppercase alphabet starting FROM zero  
            
            SET contribution TO the difference of i MINUS the element at position idx of last  
            MULTIPLIED BY the difference of the element at position idx of last MINUS the element at position idx of last2  
            SET result TO the sum of result PLUS contribution  
            SET result TO the remainder of result DIVIDED BY MOD  
            
            SET the element at position idx of last2 TO the element at position idx of last  
            SET the element at position idx of last TO i  
        END FOR  
        
        FOR each index i FROM zero TO twenty five  
            SET contribution TO the difference of n MINUS the element at position i of last  
            MULTIPLIED BY the difference of the element at position i of last MINUS the element at position i of last2  
            SET result TO the sum of result PLUS contribution  
            SET result TO the remainder of result DIVIDED BY MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
