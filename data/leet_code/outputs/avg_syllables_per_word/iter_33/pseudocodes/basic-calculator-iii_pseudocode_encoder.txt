CLASS Solution  
    FUNCTION calculate WITH PARAMETER s  
        FUNCTION helper WITH PARAMETER s  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO the plus character  
            WHILE the LENGTH OF s GREATER THAN zero holds  
                SET char TO the element at position zero of s  
                REMOVE the element at position zero FROM s  
                IF char IS a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                IF char EQUALS open parenthesis character  
                    SET num TO the result of calling helper WITH PARAMETER s  
                END IF  
                IF char IS one of the characters plus OR minus OR multiplication OR division OR open parenthesis OR close parenthesis OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS the plus character  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS the minus character  
                        APPEND the negation of num TO stack  
                    ELSE IF sign EQUALS the multiplication character  
                        SET the last element of stack TO the last element of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS the division character  
                        SET the last element of stack TO the integer value of the last element of stack DIVIDED BY num with truncation towards zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS close parenthesis character  
                    BREAK the while loop  
                END IF  
            END WHILE  
            RETURN the sum of all elements in stack  
        END FUNCTION  
        SET s TO the list of all characters in s  
        RETURN the result of calling helper WITH PARAMETER s  
    END FUNCTION  
END CLASS
