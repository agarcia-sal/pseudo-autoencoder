CLASS Solution  
    FUNCTION maximumScore(nums k)  
        SET left TO k  
        SET right TO k  
        SET current_min TO element at position k of nums  
        SET max_score TO current_min  
        WHILE left GREATER THAN zero OR right LESS THAN the LENGTH OF nums MINUS one  
            IF left EQUALS zero  
                INCREMENT right BY one  
            ELSE IF right EQUALS the LENGTH OF nums MINUS one  
                DECREMENT left BY one  
            ELSE IF element at position left MINUS one of nums GREATER THAN OR EQUAL TO element at position right PLUS one of nums  
                DECREMENT left BY one  
            ELSE  
                INCREMENT right BY one  
            END IF  
            SET current_min TO the MINIMUM OF current_min AND element at position left of nums AND element at position right of nums  
            SET score TO current_min MULTIPLIED BY the difference of right MINUS left PLUS one  
            SET max_score TO the MAXIMUM OF max_score AND score  
        END WHILE  
        RETURN max_score  
    END FUNCTION  
END CLASS
