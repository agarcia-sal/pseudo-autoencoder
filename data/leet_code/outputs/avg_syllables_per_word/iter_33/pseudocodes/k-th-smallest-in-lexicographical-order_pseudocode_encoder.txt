CLASS Solution  
    FUNCTION findKthNumber(n n INTEGER, k k INTEGER) RETURNS INTEGER  
        FUNCTION count_steps(prefix prefix INTEGER, n n INTEGER) RETURNS INTEGER  
            SET next_prefix TO prefix PLUS one  
            SET steps TO zero  
            WHILE prefix LESS THAN OR EQUAL TO n  
                SET steps TO steps PLUS the MINIMUM OF the value of n MINUS prefix PLUS one AND next_prefix MINUS prefix  
                SET prefix TO prefix MULTIPLIED BY ten  
                SET next_prefix TO next_prefix MULTIPLIED BY ten  
            END WHILE  
            RETURN steps  
        END FUNCTION  
  
        SET current TO one  
        SET k TO k MINUS one  
        WHILE k GREATER THAN zero  
            SET steps TO count_steps WITH arguments current AND n  
            IF steps LESS THAN OR EQUAL TO k  
                SET current TO current PLUS one  
                SET k TO k MINUS steps  
            ELSE  
                SET current TO current MULTIPLIED BY ten  
                SET k TO k MINUS one  
            END IF  
        END WHILE  
        RETURN current  
    END FUNCTION  
END CLASS
