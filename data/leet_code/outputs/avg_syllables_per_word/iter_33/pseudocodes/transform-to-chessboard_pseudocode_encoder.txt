CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF element at position i of board AND element at position j of element at position i of board NOT EQUALS element at position zero of element at position i of board OR element at position j of element at position zero of board NOT EQUALS element at position zero of element at position zero of board  
                    RETURN negative one  
                END IF  
            END FOR  
            
            INCREMENT rowSum BY element at position zero of element at position i of board  
            INCREMENT colSum BY element at position i of element at position zero of board  
            
            IF element at position zero of element at position i of board EQUALS the remainder when i DIVIDED BY two  
                INCREMENT rowSwap BY one  
            END IF  
            IF element at position i of element at position zero of board EQUALS the remainder when i DIVIDED BY two  
                INCREMENT colSwap BY one  
            END IF  
        END FOR  
        
        IF the integer division of N by two GREATER THAN rowSum OR rowSum GREATER THAN the integer division of N PLUS one by two  
            RETURN negative one  
        END IF  
        IF the integer division of N by two GREATER THAN colSum OR colSum GREATER THAN the integer division of N PLUS one by two  
            RETURN negative one  
        END IF  
        
        IF the remainder when N DIVIDED BY two EQUALS one  
            IF the remainder when rowSwap DIVIDED BY two EQUALS one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder when colSwap DIVIDED BY two EQUALS one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the minimum of rowSwap AND N MINUS rowSwap  
            SET colSwap TO the minimum of colSwap AND N MINUS colSwap  
        END IF  
        
        SET result TO the integer division of the sum of rowSwap AND colSwap by two  
        RETURN result  
    END FUNCTION  
END CLASS
