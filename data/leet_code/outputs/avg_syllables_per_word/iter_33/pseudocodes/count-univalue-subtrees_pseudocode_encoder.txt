CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the value attribute TO val  
        SET the left child attribute TO left  
        SET the right child attribute TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values  
    IF the list values EQUALS empty  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values  
    SET i TO one  
    CREATE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        REMOVE and ASSIGN node TO the element removed from the left end of queue  
        IF i LESS THAN the length of values AND the element at position i of values IS NOT no value  
            SET the left child attribute of node TO a new TreeNode with the element at position i of values  
            APPEND the left child attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values IS NOT no value  
            SET the right child attribute of node TO a new TreeNode with the element at position i of values  
            APPEND the right child attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS no value AND q EQUALS no value  
        RETURN True  
    ELSE IF p EQUALS no value OR q EQUALS no value  
        RETURN False  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters the left child attribute of p and the left child attribute of q AND the result of is_same_tree with parameters the right child attribute of p and the right child attribute of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION countUnivalSubtrees with parameters root  
        FUNCTION helper with parameters node  
            IF node EQUALS no value  
                RETURN True AND zero  
            END IF  
            SET is_left_unival AND left_count TO the result of helper with the left child attribute of node  
            SET is_right_unival AND right_count TO the result of helper with the right child attribute of node  
            IF is_left_unival AND is_right_unival  
                IF the left child attribute of node EQUALS no value OR the value attribute of the left child attribute of node EQUALS the value attribute of node  
                    AND the right child attribute of node EQUALS no value OR the value attribute of the right child attribute of node EQUALS the value attribute of node  
                    RETURN True AND left_count PLUS right_count PLUS one  
                END IF  
            END IF  
            RETURN False AND left_count PLUS right_count  
        END FUNCTION  
        SET discard AND count TO the result of helper with root  
        RETURN count  
    END FUNCTION  
END CLASS
