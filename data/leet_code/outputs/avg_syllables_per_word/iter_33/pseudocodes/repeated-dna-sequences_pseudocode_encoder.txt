CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO an empty dictionary with default value zero  
        FOR index FROM zero TO the LENGTH OF s MINUS ten PLUS one  
            SET sequence TO the substring from position index TO position index PLUS ten MINUS one of s  
            INCREMENT the value of sequence_count at sequence BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each sequence_key IN the keys of sequence_count  
            IF the value of sequence_count at sequence_key GREATER THAN one  
                APPEND sequence_key TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
