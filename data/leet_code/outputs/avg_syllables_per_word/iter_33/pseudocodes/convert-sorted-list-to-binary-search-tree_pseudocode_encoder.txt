CLASS ListNode  
    FUNCTION __init__ (val EQUALS zero NEXT EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node (values)  
    IF values IS NOT PRESENT  
        RETURN none  
    END IF  
    SET head TO NEW ListNode WITH element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position two TO the last position  
        SET node TO NEW ListNode WITH val  
        SET the next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list (p1 p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS NOT PRESENT OR p2 IS NOT PRESENT  
        RETURN false  
    END IF  
    RETURN the val of p1 EQUALS the val of p2 AND is_same_list WITH the next of p1 AND the next of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ (val EQUALS zero LEFT EQUALS none RIGHT EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node (values)  
    IF values IS NOT PRESENT  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN OR EQUAL TO the LENGTH OF values AND element at position i of values IS NOT none  
            SET the left of node TO NEW TreeNode WITH element at position i of values  
            APPEND the left of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN OR EQUAL TO the LENGTH OF values AND element at position i of values IS NOT none  
            SET the right of node TO NEW TreeNode WITH element at position i of values  
            APPEND the right of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree (p q)  
    IF p IS NOT PRESENT AND q IS NOT PRESENT  
        RETURN true  
    ELSE IF p IS NOT PRESENT OR q IS NOT PRESENT  
        RETURN false  
    ELSE IF the val of p NOT EQUALS the val of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH the left of p AND the left of q AND is_same_tree WITH the right of p AND the right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST (head)  
        IF head IS NOT PRESENT  
            RETURN none  
        END IF  

        FUNCTION findMiddle (left right)  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUALS right AND the next of fast NOT EQUALS right  
                SET slow TO the next of slow  
                SET fast TO the next of the next of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  

        FUNCTION convertListToBST (left right)  
            IF left EQUALS right  
                RETURN none  
            END IF  
            SET mid TO findMiddle WITH left AND right  
            SET node TO NEW TreeNode WITH the val of mid  
            SET the left of node TO convertListToBST WITH left AND mid  
            SET the right of node TO convertListToBST WITH the next of mid AND right  
            RETURN node  
        END FUNCTION  

        RETURN convertListToBST WITH head AND none  
    END FUNCTION  
END CLASS
