CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO two dimensional list of False with number of rows equal to the length of s plus one and number of columns equal to the length of p plus one  
        SET element at position zero zero of dp TO True  
        FOR j FROM one TO the length of p inclusive  
            IF element at position j minus one of p EQUALS the character representing asterisk  
                SET element at position zero j of dp TO element at position zero j minus two of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the length of s inclusive  
            FOR j FROM one TO the length of p inclusive  
                IF element at position j minus one of p EQUALS element at position i minus one of s OR element at position j minus one of p EQUALS the character representing dot  
                    SET element at position i j of dp TO element at position i minus one j minus one of dp  
                ELSE IF element at position j minus one of p EQUALS the character representing asterisk  
                    SET element at position i j of dp TO element at position i j minus two of dp  
                    IF element at position j minus two of p EQUALS element at position i minus one of s OR element at position j minus two of p EQUALS the character representing dot  
                        SET element at position i j of dp TO element at position i j of dp OR element at position i minus one j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the length of s the length of p of dp  
    END FUNCTION  
END CLASS
