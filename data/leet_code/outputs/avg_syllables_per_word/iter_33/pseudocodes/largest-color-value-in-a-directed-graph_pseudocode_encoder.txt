CLASS Solution  
    FUNCTION largestPathValue(colors, edges)  
        SET n TO the LENGTH OF colors  
        SET graph TO an empty mapping from keys to empty lists  
        SET in_degree TO a list of zeros of LENGTH n  
        
        FOR each pair u AND v IN edges  
            APPEND v TO the list at key u in graph  
            INCREMENT the element at position v of in_degree BY one  
        END FOR  
        
        SET queue TO a double-ended collection initialized with each index i FROM zero TO n MINUS one where the element at position i of in_degree EQUALS zero  
        SET dp TO a list of lists each containing zeros of LENGTH twenty-six repeated n times  
        SET visited TO zero  
        SET max_color_value TO zero  
        
        WHILE queue is not empty  
            REMOVE the first element from queue AND ASSIGN it TO node  
            INCREMENT the element at position the alphabetical index of the character at position node in colors of dp at position node BY one  
            SET max_color_value TO the larger value BETWEEN max_color_value AND the element at position the alphabetical index of the character at position node in colors of dp at position node  
            INCREMENT visited BY one  
            
            FOR each neighbor IN the list at key node in graph  
                FOR c FROM zero TO twenty-five  
                    SET the element at position c of dp at position neighbor TO the larger value BETWEEN the element at position c of dp at position neighbor AND the element at position c of dp at position node  
                END FOR  
                DECREMENT the element at position neighbor of in_degree BY one  
                IF the element at position neighbor of in_degree EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF visited EQUALS n  
            RETURN max_color_value  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
