CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        SET dp TO a list of dictionaries INITIALIZED with zero for each index FROM zero TO n MINUS one  
        SET total_count TO zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                SET difference_to_d TO the element at position i of nums MINUS the element at position j of nums  
                IF the difference_to_d EXISTS as a key in the dictionary at position j of dp  
                    INCREMENT the value at key difference_to_d in the dictionary at position i of dp BY the sum of the value at key difference_to_d in the dictionary at position j of dp PLUS one  
                    INCREMENT total_count BY the value at key difference_to_d in the dictionary at position j of dp  
                ELSE  
                    INCREMENT the value at key difference_to_d in the dictionary at position i of dp BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN total_count  
    END FUNCTION  
END CLASS
