CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO frequency count of each character in s  
        SET max_heap TO empty list  
        FOR each character and frequency IN count  
            APPEND pair of frequency MINUS frequency and character TO max_heap  
        END FOR  
        TRANSFORM max_heap into a heap structure based on the first element of each pair  
        
        SET prev_char TO no value  
        SET prev_freq TO zero  
        SET result TO empty list  
        
        WHILE the max_heap is not empty OR prev_freq LESS THAN zero holds  
            IF the max_heap is empty AND prev_freq LESS THAN zero holds  
                RETURN empty string  
            END IF  
            
            SET freq and char TO element extracted from the top of max_heap heap  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq LESS THAN zero holds  
                INSERT pair of prev_freq and prev_char INTO max_heap heap  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
