CLASS Solution  
    FUNCTION numSubmatrixSumTarget(matrix, target)  
        FUNCTION subarraySum(nums, target)  
            SET count TO zero  
            ASSIGN reference TO a new mapping from integers to integers called seen_sums  
            SET cumulative_sum TO zero  
            FOR each num IN nums  
                SET cumulative_sum TO cumulative_sum PLUS num  
                IF cumulative_sum EQUALS target THEN  
                    INCREMENT count BY one  
                END IF  
                IF cumulative_sum MINUS target EXISTS AS A KEY IN seen_sums THEN  
                    INCREMENT count BY the value associated with the key cumulative_sum MINUS target in seen_sums  
                END IF  
                INCREMENT the value associated with the key cumulative_sum in seen_sums BY one  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET rows TO the LENGTH OF matrix  
        SET cols TO the LENGTH OF element at position zero of matrix  
        SET total_count TO zero  
        
        FOR top FROM zero TO rows MINUS one  
            ASSIGN reference TO a new list called current_cols containing cols number of zeros  
            FOR bottom FROM top TO rows MINUS one  
                FOR col FROM zero TO cols MINUS one  
                    SET element at position col of current_cols TO element at position col of current_cols PLUS element at position col of element at position bottom of matrix  
                END FOR  
                SET total_count TO total_count PLUS subarraySum(current_cols, target)  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
