CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s EQUALS empty string OR element at position one of s EQUALS character zero  
            RETURN zero  
        END IF  

        SET dp TO list of zeros with length EQUALS the LENGTH OF s PLUS one  
        SET element at position one of dp TO one  
        SET element at position two of dp TO one  

        FOR i FROM three TO the LENGTH OF s PLUS one  
            IF element at position i MINUS one of s NOT EQUALS character zero  
                SET element at position i of dp TO element at position i of dp PLUS element at position i MINUS one of dp  
            END IF  

            SET two_digit_string TO substring from position i MINUS two TO position i of s  
            SET two_digit TO the integer value of two_digit_string  
            IF two_digit GREATER THAN OR EQUAL TO ten AND two_digit LESS THAN OR EQUAL TO twenty six  
                SET element at position i of dp TO element at position i of dp PLUS element at position i MINUS two of dp  
            END IF  
        END FOR  

        RETURN element at position the LENGTH OF dp of dp  
    END FUNCTION  
END CLASS
