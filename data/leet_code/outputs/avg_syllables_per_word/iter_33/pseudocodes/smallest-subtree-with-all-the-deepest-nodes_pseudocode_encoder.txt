CLASS TreeNode  
    FUNCTION __init__ WITH parameters self val left right  
        SET the value property of self TO val  
        SET the left property of self TO left  
        SET the right property of self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH parameter values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the first element of values as val  
    SET i TO one  
    CREATE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the length of values AND the element at position i of values is not none  
            SET the left property of node TO a new TreeNode with the element at position i of values as val  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values is not none  
            SET the right property of node TO a new TreeNode with the element at position i of values as val  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH parameters p q  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the value property of p NOT EQUALS the value property of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with the left property of p and the left property of q AND the result of is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION subtreeWithAllDeepest WITH parameters self root  
        FUNCTION dfs WITH parameter node  
            IF node EQUALS none  
                RETURN node AND zero  
            END IF  
            SET left_node AND left_depth TO the result of dfs with the left property of node  
            SET right_node AND right_depth TO the result of dfs with the right property of node  
            IF left_depth GREATER THAN right_depth  
                RETURN left_node AND left_depth PLUS one  
            ELSE IF right_depth GREATER THAN left_depth  
                RETURN right_node AND right_depth PLUS one  
            ELSE  
                RETURN node AND left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN the first element of the result of dfs with root  
    END FUNCTION  
END CLASS
