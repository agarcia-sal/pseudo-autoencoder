CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty sequence  
        SET d TO empty sequence  
        
        FOR each char IN s  
            IF character char is a digit  
                SET d TO concatenation of d and char  
            ELSE IF character char is an alphabetic letter  
                SET a TO concatenation of a and char  
            ELSE IF character char equals the opening bracket character  
                APPEND the tuple of a and the conversion of d to integer TO stack  
                SET a TO empty sequence  
                SET d TO empty sequence  
            ELSE IF character char equals the closing bracket character  
                ASSIGN p and n TO the tuple obtained by removing and returning the last element of stack  
                SET a TO concatenation of p and the sequence a repeated n times  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
