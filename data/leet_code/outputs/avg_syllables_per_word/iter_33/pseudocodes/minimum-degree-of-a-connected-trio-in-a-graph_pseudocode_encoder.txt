CLASS Solution  
    FUNCTION minTrioDegree(n parameter, edges parameter)  
        SET graph TO empty mapping from integer to empty set for each integer FROM one TO n parameter  
        FOR each pair u parameter, v parameter IN edges parameter  
            ADD v parameter TO the set corresponding to u parameter in graph  
            ADD u parameter TO the set corresponding to v parameter in graph  
        END FOR  
        SET min_degree TO infinite value  
        SET trio_found TO false  
        FOR i parameter FROM one TO n parameter  
            FOR j parameter FROM i parameter PLUS one TO n parameter  
                FOR k parameter FROM j parameter PLUS one TO n parameter  
                    IF element j parameter IS IN the set corresponding to i parameter in graph AND element k parameter IS IN the set corresponding to i parameter in graph AND element k parameter IS IN the set corresponding to j parameter in graph  
                        SET trio_found TO true  
                        SET trio_degree TO the LENGTH OF the set corresponding to i parameter in graph PLUS the LENGTH OF the set corresponding to j parameter in graph PLUS the LENGTH OF the set corresponding to k parameter in graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF trio_found EQUALS true  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
