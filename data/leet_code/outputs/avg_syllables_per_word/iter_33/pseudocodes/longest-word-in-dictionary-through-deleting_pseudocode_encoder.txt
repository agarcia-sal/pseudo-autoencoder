CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator_to_s TO an iterator OVER s  
            FOR each character IN word  
                IF character DOES NOT APPEAR IN iterator_to_s AFTER its current position  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SORT dictionary BY descending LENGTH OF each element AND ascending LEXICOGRAPHICAL ORDER OF each element  
        
        FOR each word IN dictionary  
            IF is_subsequence(word, s) EQUALS true  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN an empty string  
    END FUNCTION  
END CLASS
