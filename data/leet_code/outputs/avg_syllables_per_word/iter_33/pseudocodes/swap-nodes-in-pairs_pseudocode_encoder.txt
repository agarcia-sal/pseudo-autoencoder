CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the collection values IS empty  
        RETURN absence of any value  
    END IF  
    SET head TO a new ListNode with the element at position one of values  
    SET pointer p TO head  
    FOR each val IN the elements of values starting from position two TO the end  
        SET node TO a new ListNode with val  
        SET the next attribute of pointer p TO node  
        SET pointer p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 IS absence of any value AND p2 IS absence of any value  
        RETURN truth value true  
    END IF  
    IF p1 IS absence of any value OR p2 IS absence of any value  
        RETURN truth value false  
    END IF  
    RETURN the logical AND of the equality of the value attribute of p1 WITH the value attribute of p2 AND the recursive call of is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs with parameter head  
        SET dummy TO a new ListNode with zero as the value  
        SET the next attribute of dummy TO head  
        SET current TO dummy  
        
        WHILE both the next attribute of current IS NOT absence of any value AND the next attribute of the next attribute of current IS NOT absence of any value  
            SET first TO the next attribute of current  
            SET second TO the next attribute of the next attribute of current  
            
            SET the next attribute of current TO second  
            SET the next attribute of first TO the next attribute of second  
            SET the next attribute of second TO first  
            
            SET current TO the next attribute of the next attribute of current  
        END WHILE  
        
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
