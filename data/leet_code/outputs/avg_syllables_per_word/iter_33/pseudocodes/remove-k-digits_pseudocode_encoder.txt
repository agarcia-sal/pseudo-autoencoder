CLASS Solution  
    FUNCTION removeKdigits(num k)  
        SET stack TO an empty list  
        FOR each digit IN the sequence of characters in num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND element at the last position of stack GREATER THAN digit  
                REMOVE the element at the last position FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        IF k EQUALS zero  
            SET final_stack TO stack  
        ELSE  
            SET final_stack TO the sequence of elements in stack from the first position TO the length of stack MINUS k  
        END IF  
        SET joined_result TO the concatenation of all elements in final_stack  
        SET result TO the substring of joined_result starting from the first position WHERE the character IS NOT EQUAL TO the character zero TO the end of joined_result  
        IF result IS NOT empty  
            RETURN result  
        ELSE  
            RETURN the character zero  
        END IF  
    END FUNCTION  
END CLASS
