CLASS Solution  
    FUNCTION compareVersion(version1 version1 STRING, version2 version2 STRING) RETURNS INTEGER  
        SET v1_revisions TO the list of substrings of version1 separated by the character representing a dot  
        SET v2_revisions TO the list of substrings of version2 separated by the character representing a dot  
        SET max_length TO the greater of the length of v1_revisions AND the length of v2_revisions  
        FOR index i FROM zero TO max_length MINUS one  
            IF index i LESS THAN the length of v1_revisions  
                SET v1 TO the integer value of the element at position i of v1_revisions  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF index i LESS THAN the length of v2_revisions  
                SET v2 TO the integer value of the element at position i of v2_revisions  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 LESS THAN v2  
                RETURN negative one  
            ELSE IF v1 GREATER THAN v2  
                RETURN positive one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
