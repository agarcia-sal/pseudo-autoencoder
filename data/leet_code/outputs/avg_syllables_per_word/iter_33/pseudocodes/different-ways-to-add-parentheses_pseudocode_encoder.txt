CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS the plus symbol  
                        APPEND the sum of l PLUS r TO results  
                    ELSE IF operator EQUALS the minus symbol  
                        APPEND the difference of l MINUS r TO results  
                    ELSE IF operator EQUALS the multiplication symbol  
                        APPEND the product of l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
  
        FUNCTION helper(sub_expr)  
            IF sub_expr CONSISTS OF digits ONLY  
                RETURN list containing the integer value of sub_expr  
            END IF  
  
            SET results TO empty list  
            FOR index FROM zero TO the length of sub_expr MINUS one  
                IF the element at position index of sub_expr EQUALS the plus symbol OR EQUALS the minus symbol OR EQUALS the multiplication symbol  
                    SET left_results TO the result of calling helper on the substring from position zero TO position index MINUS one of sub_expr  
                    SET right_results TO the result of calling helper on the substring from position index PLUS one TO the end of sub_expr  
                    EXTEND results BY the result of calling compute with left_results AND right_results AND the element at position index of sub_expr  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  
  
        RETURN the result of calling helper with expression  
    END FUNCTION  
END CLASS
