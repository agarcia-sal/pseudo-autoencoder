CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with value as element at position one of values  
    SET i TO one  
    CREATE an empty double-ended queue named queue  
    APPEND root TO queue  
    WHILE there exists any element in queue  
        REMOVE and ASSIGN node TO the element from the front of queue  
        IF i is LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET left child of node TO a new TreeNode with value as element at position i of values  
            APPEND left child of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET right child of node TO a new TreeNode with value as element at position i of values  
            APPEND right child of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value  
        RETURN True  
    ELSE IF p IS no value OR q IS no value  
        RETURN False  
    ELSE IF value of p IS NOT EQUALS value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters left child of p AND left child of q AND is_same_tree with parameters right child of p AND right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self, root, k)  
        IF left child of root IS no value AND right child of root IS no value  
            RETURN value of root  
        END IF  
        CREATE an empty default dictionary named graph with lists as default values  
        FUNCTION dfs(node, parent)  
            IF parent IS NOT no value  
                APPEND value of parent TO the list in graph under key of value of node  
                APPEND value of node TO the list in graph under key of value of parent  
            END IF  
            IF left child of node IS NOT no value  
                CALL dfs with parameters left child of node AND node  
            END IF  
            IF right child of node IS NOT no value  
                CALL dfs with parameters right child of node AND node  
            END IF  
        END FUNCTION  
        CALL dfs with parameters root AND no value  
        CREATE a double-ended queue named queue containing element k  
        CREATE an empty set named visited  
        WHILE there exists any element in queue  
            REMOVE and ASSIGN current TO the element from the front of queue  
            ADD current TO visited  
            IF the LENGTH OF the list in graph under key current IS EQUALS one AND current IS NOT EQUALS value of root  
                RETURN current  
            END IF  
            FOR each neighbor IN the list in graph under key current  
                IF neighbor IS NOT IN visited  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
