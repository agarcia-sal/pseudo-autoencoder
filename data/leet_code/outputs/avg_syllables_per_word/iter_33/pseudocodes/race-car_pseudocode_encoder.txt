CLASS Solution  
    FUNCTION racecar(target)  
        ASSIGN reference TO an empty queue with initial element as a tuple of position zero speed one steps zero  
        ASSIGN reference TO an empty set containing the tuple of position zero and speed one as visited states  
        WHILE the queue is not empty  
            REMOVE the first element from the queue and ASSIGN position speed and steps from that element  
            IF position EQUALS target  
                RETURN steps  
            END IF  
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple of next_position and next_speed is NOT IN visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of next_position and next_speed TO visited  
                APPEND the tuple of next_position next_speed steps PLUS one TO the queue  
            END IF  
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple of position and next_speed is NOT IN visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of position and next_speed TO visited  
                APPEND the tuple of position next_speed steps PLUS one TO the queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
