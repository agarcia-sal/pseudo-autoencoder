CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO a list containing numbers from zero TO n MINUS one  
        SET self.size TO a list containing one repeated n times  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u THEN  
            SET element at position u of self.parent TO self.find(element at position u of self.parent)  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET root_u TO self.find(u)  
        SET root_v TO self.find(v)  
        IF root_u NOT EQUALS root_v THEN  
            IF element at position root_u of self.size GREATER THAN element at position root_v of self.size THEN  
                SET element at position root_v of self.parent TO root_u  
                INCREMENT element at position root_u of self.size BY element at position root_v of self.size  
            ELSE  
                SET element at position root_u of self.parent TO root_v  
                INCREMENT element at position root_v of self.size BY element at position root_u of self.size  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION friendRequests(self, n, restrictions, requests)  
        ASSIGN reference TO object UnionFind CREATED WITH parameter n TO variable uf  
        SET result TO an empty list  
        SET restriction_map TO an empty dictionary  

        FOR each pair of values x and y IN restrictions  
            IF x NOT IN keys of restriction_map THEN  
                SET element at key x of restriction_map TO an empty set  
            END IF  
            IF y NOT IN keys of restriction_map THEN  
                SET element at key y of restriction_map TO an empty set  
            END IF  
            ADD y TO the set at key x of restriction_map  
            ADD x TO the set at key y of restriction_map  
        END FOR  

        FUNCTION can_union(u, v)  
            SET root_u TO uf.find(u)  
            SET root_v TO uf.find(v)  
            FOR each person IN the set at key root_u of restriction_map OR empty set if absent  
                IF uf.find(person) EQUALS root_v THEN  
                    RETURN False  
                END IF  
            END FOR  
            FOR each person IN the set at key root_v of restriction_map OR empty set if absent  
                IF uf.find(person) EQUALS root_u THEN  
                    RETURN False  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  

        FOR each pair of values u and v IN requests  
            IF uf.find(u) EQUALS uf.find(v) THEN  
                APPEND True TO result  
            ELSE IF can_union(u, v) THEN  
                CALL uf.union(u, v)  
                APPEND True TO result  
            ELSE  
                APPEND False TO result  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
