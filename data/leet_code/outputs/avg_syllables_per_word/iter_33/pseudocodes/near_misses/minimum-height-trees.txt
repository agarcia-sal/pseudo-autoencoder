CLASS Solution  
    FUNCTION findMinHeightTrees(n, edges)  
        IF n EQUALS one  
            RETURN a list containing the element zero  
        END IF  
        ASSIGN graph TO an empty map with default value as an empty list  
        ASSIGN degree TO a list of zeros with length equal to n  
        FOR each pair of elements u and v IN edges  
            APPEND element v TO the list at key u in graph  
            APPEND element u TO the list at key v in graph  
            INCREMENT element at position u of degree BY one  
            INCREMENT element at position v of degree BY one  
        END FOR  
        ASSIGN leaves TO an empty double ended queue  
        FOR each index i FROM zero TO n MINUS one  
            IF element at position i of degree EQUALS one  
                APPEND element i TO leaves  
            END IF  
        END FOR  
        SET remaining_nodes TO n  
        WHILE remaining_nodes GREATER THAN two  
            ASSIGN leaves_count TO the length of leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR counter FROM one TO leaves_count  
                REMOVE element from the left of leaves and ASSIGN TO leaf  
                FOR each neighbor IN the list at key leaf in graph  
                    DECREMENT element at position neighbor of degree BY one  
                    IF element at position neighbor of degree EQUALS one  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN a list created from the elements in leaves  
    END FUNCTION  
END CLASS
