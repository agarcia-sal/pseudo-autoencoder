CLASS Solution  
    FUNCTION getModifiedArray(length updates)  
        SET diff TO list of zeros with size length PLUS one  
        FOR each element startIdx endIdx inc IN updates  
            SET element at position startIdx of diff TO element at position startIdx of diff PLUS inc  
            IF endIdx PLUS one LESS THAN length THEN  
                SET element at position endIdx PLUS one of diff TO element at position endIdx PLUS one of diff MINUS inc  
            END IF  
        END FOR  
        SET arr TO list of zeros with size length  
        SET element at position zero of arr TO element at position zero of diff  
        FOR i FROM one TO length MINUS one  
            SET element at position i of arr TO element at position i MINUS one of arr PLUS element at position i of diff  
        END FOR  
        RETURN arr  
    END FUNCTION  
END CLASS
