CLASS Solution  
    FUNCTION smallestGoodBase(n)  
        SET n TO the integer value of n  
        SET max_length TO the integer value of the base two logarithm of n INCREMENTED BY one  
        FOR m FROM max_length DOWN TO two  
            SET k TO the integer part of n RAISED TO THE power of one DIVIDED BY the difference of m MINUS one  
            IF k LESS THAN two  
                CONTINUE to the next iteration of the loop  
            END IF  
            SET numerator TO k RAISED TO THE power of m MINUS one  
            SET denominator TO k MINUS one  
            SET num TO the quotient of numerator DIVIDED BY denominator  
            IF num EQUALS n  
                RETURN the string representation of k  
            END IF  
        END FOR  
        RETURN the string representation of the difference of n MINUS one  
    END FUNCTION  
END CLASS
