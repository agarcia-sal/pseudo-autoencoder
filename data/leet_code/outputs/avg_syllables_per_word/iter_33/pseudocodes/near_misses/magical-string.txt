CLASS Solution  
    FUNCTION magicalString(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        IF n LESS THAN OR EQUAL TO three  
            RETURN one  
        END IF  
        
        SET s TO list consisting of one followed by two followed by two  
        SET count TO one  
        SET index TO two  
        
        WHILE the LENGTH OF s LESS THAN n  
            SET next_num TO three MINUS the element at position negative one of s  
            APPEND next_num repeated the number of times equal to the element at position index of s TO s  
            IF next_num EQUALS one  
                INCREMENT count BY the element at position index of s  
            END IF  
            INCREMENT index BY one  
        END WHILE  
        
        IF the LENGTH OF s GREATER THAN n  
            DECREMENT count BY the element at position negative one of s MULTIPLIED BY the result of the LENGTH OF s MINUS n  
        END IF  
        
        RETURN count  
    END FUNCTION  
END CLASS
