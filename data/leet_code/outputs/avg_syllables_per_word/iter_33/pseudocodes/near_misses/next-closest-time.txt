CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters in time with all colons removed  
        SET hour AND minute TO the conversion of the parts before and after the colon in time to integers  
        
        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each character h1 IN digits  
                FOR each character h2 IN digits  
                    FOR each character m1 IN digits  
                        FOR each character m2 IN digits  
                            SET hour_candidate TO the integer value of the concatenation of h1 and h2  
                            SET minute_candidate TO the integer value of the concatenation of m1 and m2  
                            IF hour_candidate LESS THAN twenty-four AND minute_candidate LESS THAN sixty  
                                APPEND the pair of hour_candidate and minute_candidate TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN times sorted in increasing order  
        END FUNCTION  
        
        SET all_times TO the result of generate_times applied to digits  
        SET current_time_index TO the position of the pair consisting of hour and minute in all_times  
        
        IF current_time_index PLUS one LESS THAN the length of all_times  
            SET next_hour AND next_minute TO the pair at position current_time_index PLUS one in all_times  
        ELSE  
            SET next_hour AND next_minute TO the first pair in all_times  
        END IF  
        
        RETURN the string formed by next_hour formatted as two digits followed by a colon followed by next_minute formatted as two digits  
    END FUNCTION  
END CLASS
