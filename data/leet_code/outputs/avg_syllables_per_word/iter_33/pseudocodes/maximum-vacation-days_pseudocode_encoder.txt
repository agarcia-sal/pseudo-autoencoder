CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the element at position zero of days  
        
        SET dp TO a list of negative one repeated n times  
        SET the element at position zero of dp TO zero  
        
        SET graph TO an empty dictionary mapping each integer i from zero TO n MINUS one TO an empty list  
        FOR each integer i FROM zero TO n MINUS one  
            FOR each integer j FROM zero TO n MINUS one  
                IF the element at position i of flights EQUALS one AT the element at position j THEN  
                    APPEND j TO the list mapped by i in graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR each integer week FROM zero TO k MINUS one  
            SET new_dp TO a list of negative one repeated n times  
            FOR each integer i FROM zero TO n MINUS one  
                IF the element at position i of dp EQUALS negative one THEN  
                    CONTINUE TO the next iteration  
                END IF  
                SET the element at position i of new_dp TO the maximum VALUE BETWEEN the element at position i of new_dp AND the element at position i of dp PLUS the element at position i of days PLUS the element at position week of days  
                FOR each integer j IN the list mapped by i in graph  
                    SET the element at position j of new_dp TO the maximum VALUE BETWEEN the element at position j of new_dp AND the element at position i of dp PLUS the element at position j of days PLUS the element at position week of days  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the maximum VALUE IN dp  
    END FUNCTION  
END CLASS
