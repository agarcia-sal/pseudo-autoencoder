CLASS Solution  
    FUNCTION wordBreak(s string, wordDict list of string) RETURNS boolean  
        SET wordSet TO CONVERT wordDict TO a set of strings  
        SET dp TO a list of boolean values where each value is False WITH the LENGTH OF s PLUS one  
        SET element at position zero of dp TO True  
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM zero TO i MINUS one  
                IF element at position j of dp EQUALS True AND the substring of s from position j PLUS one TO position i IS IN wordSet  
                    SET element at position i of dp TO True  
                    BREAK the inner loop  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the LENGTH OF s of dp  
    END FUNCTION  
END CLASS
