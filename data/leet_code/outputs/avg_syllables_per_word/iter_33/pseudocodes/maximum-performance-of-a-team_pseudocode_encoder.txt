CLASS Solution  
    FUNCTION maxPerformance(n parameter, speed parameter, efficiency parameter, k parameter)  
        SET engineers TO list of pairs each consisting of element at position index of efficiency parameter AND element at position index of speed parameter FOR each index IN range from zero TO n parameter MINUS one  
        SORT engineers IN ORDER OF elements at position zero in each pair from HIGHEST TO LOWEST  
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair consisting of eff variable and spd variable IN engineers  
            ADD spd variable TO speed_heap USING heap push operation  
            INCREMENT speed_sum BY spd variable  
            IF the LENGTH OF speed_heap IS GREATER THAN k parameter  
                SET smallest_speed TO the element REMOVED FROM speed_heap USING heap pop operation  
                DECREMENT speed_sum BY smallest_speed  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff variable  
            IF current_performance IS GREATER THAN max_performance  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        RETURN max_performance MODULO one billion PLUS seven  
    END FUNCTION  
END CLASS
