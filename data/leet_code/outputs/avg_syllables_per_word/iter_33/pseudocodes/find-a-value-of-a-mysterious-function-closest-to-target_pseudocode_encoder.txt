CLASS Solution  
    FUNCTION closestToTarget(arr PARAMETERS arr PARAMETER target)  
        SET seen TO an empty set of unique values  
        SET min_diff TO infinity value  
        FOR each num IN arr  
            SET new_seen TO an empty set  
            FOR each x IN seen  
                ADD the bitwise AND of num AND x TO new_seen  
            END FOR  
            SET seen TO the union of new_seen AND the set containing num  
            FOR each val IN seen  
                SET diff TO the absolute value of val MINUS target  
                IF diff LESS THAN min_diff THEN  
                    SET min_diff TO diff  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
