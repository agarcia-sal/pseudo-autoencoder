CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO empty list  
        FOR each number IN target  
            APPEND negative of number TO max_heap  
        END FOR  
        CALL heapify WITH max_heap  
        
        SET total_sum TO zero  
        FOR each number IN target  
            INCREMENT total_sum BY number  
        END FOR  
        
        WHILE element at position zero of max_heap LESS THAN negative of one holds  
            SET largest TO negative of result of heappop called with max_heap  
            SET rest_sum TO total_sum MINUS largest  
            
            IF largest LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero THEN  
                RETURN False  
            END IF  
            
            SET new_value TO largest MODULO rest_sum  
            IF new_value EQUALS zero AND rest_sum NOT EQUAL TO one THEN  
                RETURN False  
            END IF  
            
            SET total_sum TO rest_sum PLUS new_value  
            CALL heappush WITH max_heap AND negative of new_value  
        END WHILE  
        
        RETURN True  
    END FUNCTION  
END CLASS
