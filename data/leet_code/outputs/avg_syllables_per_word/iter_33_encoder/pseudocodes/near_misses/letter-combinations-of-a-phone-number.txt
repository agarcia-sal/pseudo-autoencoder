CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string THEN  
            RETURN empty list  
        END IF  
        
        SET phone_map TO mapping of digit two TO letters a b c AND digit three TO letters d e f AND digit four TO letters g h i AND digit five TO letters j k l AND digit six TO letters m n o AND digit seven TO letters p q r s AND digit eight TO letters t u v AND digit nine TO letters w x y z  
        
        FUNCTION backtrack(combination, next_digits)  
            IF the LENGTH OF next_digits EQUALS zero THEN  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN letters mapped by the element at position zero of next_digits in phone_map  
                    CALL backtrack WITH the concatenation of combination AND letter AND the substring of next_digits from position one TO the end  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO empty list  
        CALL backtrack WITH empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
