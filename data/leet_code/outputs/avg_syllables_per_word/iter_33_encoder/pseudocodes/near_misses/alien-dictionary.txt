CLASS Solution  
    FUNCTION alienOrder(words)  
        SET graph TO an empty mapping of character to set of characters  
        SET indegree TO an empty mapping of character to integer  
        SET all_chars TO the set of all characters occurring in the concatenation of all elements in words  
        FOR index FROM zero TO the length of words MINUS one MINUS one  
            SET word1 TO the element at position index of words  
            SET word2 TO the element at position index PLUS one of words  
            SET min_length TO the lesser value between the length of word1 and the length of word2  
            SET found TO false  
            FOR position FROM zero TO min_length MINUS one  
                IF the character at position of word1 NOT EQUALS the character at position of word2  
                    IF the character at position of word2 IS NOT in the set associated with the character at position of word1 in graph  
                        ADD the character at position of word2 TO the set associated with the character at position of word1 in graph  
                        INCREMENT the integer value associated with the character at position of word2 in indegree BY one  
                    END IF  
                    SET found TO true  
                    BREAK out of the inner loop  
                END IF  
            END FOR  
            IF found EQUALS false AND the length of word1 GREATER THAN the length of word2  
                RETURN an empty string  
            END IF  
        END FOR  
        SET queue TO a double-ended queue initialized with all characters from all_chars for which the integer value in indegree is equal to zero or the character is not in indegree  
        SET result TO an empty list  
        WHILE the queue is not empty  
            REMOVE the element from the left end of queue and ASSIGN it to char  
            APPEND char TO result  
            FOR each neighbor IN the set associated with char in graph  
                DECREMENT the integer value associated with neighbor in indegree BY one  
                IF the integer value associated with neighbor in indegree EQUALS zero  
                    APPEND neighbor TO the right end of queue  
                END IF  
            END FOR  
        END WHILE  
        IF the length of result NOT EQUALS the length of all_chars  
            RETURN an empty string  
        END IF  
        RETURN the concatenation of all elements of result as a single string  
    END FUNCTION  
END CLASS
