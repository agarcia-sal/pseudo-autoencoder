CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        SET subset_sums TO a list OF zero VALUES OF SIZE two RAISED TO THE POWER OF n  
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO two RAISED TO THE POWER OF i MINUS one  
                SET element at position mask OR with two RAISED TO THE POWER OF i OF subset_sums TO element at position mask OF subset_sums PLUS element at position i OF jobs  
            END FOR  
        END FOR  
        
        FUNCTION dp(k, mask)  
            IF k EQUALS one THEN  
                RETURN element at position mask OF subset_sums  
            END IF  
            SET res TO positive infinity  
            SET submask TO mask  
            WHILE submask GREATER THAN zero  
                SET temp_result_one TO dp(k MINUS one, mask XOR submask)  
                SET temp_result_two TO element at position submask OF subset_sums  
                SET candidate TO the GREATER OF temp_result_one AND temp_result_two  
                IF candidate LESS THAN res THEN  
                    SET res TO candidate  
                END IF  
                SET submask TO submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        RETURN dp(k, two RAISED TO THE POWER OF n MINUS one)  
    END FUNCTION  
END CLASS
