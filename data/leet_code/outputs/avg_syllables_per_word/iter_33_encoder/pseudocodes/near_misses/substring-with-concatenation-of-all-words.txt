CLASS Solution  
    FUNCTION findSubstring(s parameter TO string, words parameter TO list of strings) RETURNS list of integers  
        IF the LENGTH OF s EQUALS zero OR the LENGTH OF words EQUALS zero OR the LENGTH OF the element at position zero of words EQUALS zero  
            RETURN an empty list  
        END IF  
        
        SET word_length TO the LENGTH OF the element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the frequency COUNTER of words  
        SET result TO an empty list  
        
        FOR index i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO an empty frequency COUNTER  
            FOR index j FROM i TO i PLUS total_length MINUS one IN STEPS OF word_length  
                SET word TO the substring from position j TO position j PLUS word_length MINUS one of s  
                INCREMENT the count of word IN seen BY one  
                IF the count of word IN seen IS GREATER THAN the count of word IN word_count  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF the inner loop did not break  
                APPEND i TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
