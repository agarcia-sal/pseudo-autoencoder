CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO an empty list containing n empty lists  
        FOR each x y time IN edges  
            APPEND the pair of y AND time TO element at position x of graph  
            APPEND the pair of x AND time TO element at position y of graph  
        END FOR  
        
        SET pq TO a list containing a tuple of element at position zero of passingFees AND zero AND zero  
        SET visited_time TO a list of length n with each element set to infinity  
        SET element at position zero of visited_time TO zero  
        
        WHILE pq is not empty  
            REMOVE from pq the element with the smallest first element and ASSIGN current_cost current_city current_time FROM that element  
            
            IF current_city EQUALS n MINUS one  
                RETURN current_cost  
            END IF  
            
            FOR each neighbor travel_time IN element at position current_city of graph  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS element at position neighbor of passingFees  
                
                IF new_time LESS THAN OR EQUAL TO maxTime AND (new_time LESS THAN element at position neighbor of visited_time OR new_cost LESS THAN current_cost)  
                    SET element at position neighbor of visited_time TO new_time  
                    ADD the tuple of new_cost neighbor new_time TO pq maintaining its priority order by first element  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
