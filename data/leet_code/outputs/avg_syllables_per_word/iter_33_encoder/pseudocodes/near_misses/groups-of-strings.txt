CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO a list containing the sequence of numbers from zero to size MINUS one  
        SET self.rank TO a list containing size number of zeros  
        SET self.size TO a list containing size number of ones  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF the element at position u of self.parent NOT EQUALS u THEN  
            SET the element at position u of self.parent TO the result of calling self.find with argument the element at position u of self.parent  
        END IF  
        RETURN the element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET rootU TO the result of calling self.find with argument u  
        SET rootV TO the result of calling self.find with argument v  
        IF rootU NOT EQUALS rootV THEN  
            IF the element at position rootU of self.rank GREATER THAN the element at position rootV of self.rank THEN  
                SET the element at position rootV of self.parent TO rootU  
                INCREMENT the element at position rootU of self.size BY the element at position rootV of self.size  
            ELSE IF the element at position rootU of self.rank LESS THAN the element at position rootV of self.rank THEN  
                SET the element at position rootU of self.parent TO rootV  
                INCREMENT the element at position rootV of self.size BY the element at position rootU of self.size  
            ELSE  
                SET the element at position rootV of self.parent TO rootU  
                INCREMENT the element at position rootU of self.size BY the element at position rootV of self.size  
                INCREMENT the element at position rootU of self.rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION groupStrings(self, words)  
        FUNCTION bitmask(word)  
            SET mask TO zero  
            FOR each char IN word  
                SET mask TO mask OR the value one MULTIPLIED BY two RAISED TO the power of the numerical code of char MINUS the numerical code of the lowercase letter a  
            END FOR  
            RETURN mask  
        END FUNCTION  

        SET word_map TO a new mapping with default value empty list  
        SET n TO the length of words  
        SET uf TO a new instance of UnionFind with argument n  

        FOR each index i and word IN words  
            SET mask TO the result of calling bitmask with argument word  
            APPEND i TO the list at key mask of word_map  
        END FOR  

        FOR each index i and word IN words  
            SET mask TO the result of calling bitmask with argument word  
            FOR each number j FROM zero TO twenty five  
                SET new_mask TO the value of mask XOR one MULTIPLIED BY two RAISED TO the power of j  
                IF new_mask is a key in word_map THEN  
                    FOR each element k IN the list at key new_mask of word_map  
                        CALL uf.union with arguments i and k  
                    END FOR  
                END IF  

                IF the value of mask AND one MULTIPLIED BY two RAISED TO the power of j EQUALS zero THEN  
                    SET new_mask TO the value of mask OR one MULTIPLIED BY two RAISED TO the power of j  
                    IF new_mask is a key in word_map THEN  
                        FOR each element k IN the list at key new_mask of word_map  
                            CALL uf.union with arguments i and k  
                        END FOR  
                    END IF  
                END IF  

                IF the value of mask AND one MULTIPLIED BY two RAISED TO the power of j NOT EQUALS zero THEN  
                    FOR each number k FROM zero TO twenty five  
                        IF the value of mask AND one MULTIPLIED BY two RAISED TO the power of k EQUALS zero THEN  
                            SET new_mask TO the value of mask XOR one MULTIPLIED BY two RAISED TO the power of j OR one MULTIPLIED BY two RAISED TO the power of k  
                            IF new_mask is a key in word_map THEN  
                                FOR each element l IN the list at key new_mask of word_map  
                                    CALL uf.union with arguments i and l  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        SET root_sizes TO a new mapping with default value zero  
        FOR each number i FROM zero TO n MINUS one  
            SET root TO the result of calling uf.find with argument i  
            INCREMENT the value at key root of root_sizes BY one  
        END FOR  

        SET max_group_size TO the maximum value among the values of root_sizes  
        SET num_groups TO the number of keys in root_sizes  

        RETURN a list containing num_groups followed by max_group_size  
    END FUNCTION  
END CLASS
