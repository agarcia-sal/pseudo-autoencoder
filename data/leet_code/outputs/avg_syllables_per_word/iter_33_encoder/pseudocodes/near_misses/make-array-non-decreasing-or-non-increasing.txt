CLASS Solution  
    FUNCTION convertArray(nums)  
        FUNCTION min_operations_to_non_decreasing(nums)  
            SET max_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF max_heap IS NOT empty AND the NEGATION of the element at position zero of max_heap IS GREATER THAN num  
                    INCREMENT operations BY the NEGATION of the value obtained by REMOVING the element at position zero of max_heap MINUS num  
                    ADD the NEGATION of num TO max_heap  
                END IF  
                ADD the NEGATION of num TO max_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
        
        FUNCTION min_operations_to_non_increasing(nums)  
            SET min_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF min_heap IS NOT empty AND the element at position zero of min_heap IS LESS THAN num  
                    INCREMENT operations BY num MINUS the value obtained by REMOVING the element at position zero of min_heap  
                    ADD num TO min_heap  
                END IF  
                ADD num TO min_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
        
        RETURN the MINIMUM OF the value returned by min_operations_to_non_decreasing with argument nums AND the value returned by min_operations_to_non_increasing with argument nums  
    END FUNCTION  
END CLASS
