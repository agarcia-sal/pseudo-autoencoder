CLASS Solution  
    FUNCTION minTaps(n, ranges)  
        SET max_range TO a list of zeros with the length equal to the value of n PLUS one  
        
        FOR index i FROM zero TO the value of n  
            IF the element at position i of ranges GREATER THAN zero  
                SET left TO the greater value between zero AND the result of index i MINUS the element at position i of ranges  
                SET right TO the lesser value between the value of n AND the result of index i PLUS the element at position i of ranges  
                IF the element at position left of max_range LESS THAN the value of right  
                    SET the element at position left of max_range TO the value of right  
                END IF  
            END IF  
        END FOR  
        
        SET taps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        
        FOR index i FROM zero TO the value of n  
            IF index i GREATER THAN farthest  
                RETURN negative one  
            END IF  
            IF index i GREATER THAN current_end  
                INCREMENT taps BY one  
                SET current_end TO farthest  
            END IF  
            IF farthest LESS THAN the element at position i of max_range  
                SET farthest TO the element at position i of max_range  
            END IF  
        END FOR  
        
        RETURN taps  
    END FUNCTION  
END CLASS
