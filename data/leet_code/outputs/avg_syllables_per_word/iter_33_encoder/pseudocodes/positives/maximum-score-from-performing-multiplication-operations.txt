CLASS Solution  
    FUNCTION maximumScore(nums, multipliers)  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zeroes WITH LENGTH m PLUS one  
        FOR i FROM m MINUS one TO zero DECREMENTING BY one  
            SET new_dp TO a list of zeroes WITH LENGTH m PLUS one  
            FOR left FROM zero TO i INCLUSIVE  
                SET right TO n MINUS one MINUS (i MINUS left)  
                SET mul TO the element at position i of multipliers  
                SET value_from_left TO mul MULTIPLIED BY the element at position left of nums PLUS the element at position left PLUS one of dp  
                SET value_from_right TO mul MULTIPLIED BY the element at position right of nums PLUS the element at position left of dp  
                SET the element at position left of new_dp TO the maximum BETWEEN value_from_left AND value_from_right  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN the element at position zero of dp  
    END FUNCTION  
END CLASS
