CLASS Solution  
    FUNCTION orderlyQueue(s, k)  
        IF k GREATER THAN one  
            RETURN the concatenation of all characters in the collection obtained by arranging all characters of input string in ascending lexicographical order  
        ELSE  
            SET n TO the length of input string s  
            SET smallest TO s  
            FOR i FROM one TO n MINUS one  
                SET rotated TO the concatenation of substring from position i TO the end of string s PLUS substring from position zero TO position i MINUS one of string s  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
