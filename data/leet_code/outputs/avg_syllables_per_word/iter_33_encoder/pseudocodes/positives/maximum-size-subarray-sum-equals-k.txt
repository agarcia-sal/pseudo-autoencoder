CLASS Solution  
    FUNCTION maxSubArrayLen(nums, k)  
        SET cumulative_sum_index TO a dictionary mapping zero TO negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        FOR each index i AND element num IN nums with enumeration  
            SET cumulative_sum TO cumulative_sum PLUS num  
            IF cumulative_sum MINUS k IS a key in cumulative_sum_index  
                SET max_length TO the greater value BETWEEN max_length AND i MINUS the value at key cumulative_sum MINUS k in cumulative_sum_index  
            END IF  
            IF cumulative_sum IS NOT a key in cumulative_sum_index  
                SET the value at key cumulative_sum in cumulative_sum_index TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
