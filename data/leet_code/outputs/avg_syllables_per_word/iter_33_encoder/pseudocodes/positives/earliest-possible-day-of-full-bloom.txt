CLASS Solution  
    FUNCTION earliestFullBloom(plantTime growTime)  
        SET tasks TO list of pairs each containing element at position i of plantTime AND element at position i of growTime FOR index i FROM zero TO length of plantTime MINUS one  
        SORT tasks BY second element of each pair IN descending order  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair IN tasks  
            DECOMPOSE pair INTO plant AND grow  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF bloom_day GREATER THAN earliest_bloom  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
