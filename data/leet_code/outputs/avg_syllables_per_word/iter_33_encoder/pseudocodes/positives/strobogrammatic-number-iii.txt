CLASS Solution  
    FUNCTION strobogrammaticInRange(low, high)  
        FUNCTION generate_strobogrammatic(n, final_length)  
            IF n EQUALS zero  
                RETURN a list containing an empty string  
            END IF  
            IF n EQUALS one  
                RETURN a list containing the strings zero one and eight  
            END IF  
            SET middles TO the result of generate_strobogrammatic of n MINUS two and final_length  
            SET result TO an empty list  
            FOR each middle IN middles  
                IF n NOT EQUALS final_length  
                    APPEND the string zero concatenated with middle concatenated with zero TO result  
                END IF  
                APPEND the string one concatenated with middle concatenated with one TO result  
                APPEND the string six concatenated with middle concatenated with nine TO result  
                APPEND the string eight concatenated with middle concatenated with eight TO result  
                APPEND the string nine concatenated with middle concatenated with six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        FUNCTION count_strobogrammatic(low, high)  
            SET count TO zero  
            FOR length FROM the length of low TO the length of high  
                FOR each num IN generate_strobogrammatic of length and length  
                    IF (length EQUALS the length of low AND num LESS THAN low) OR (length EQUALS the length of high AND num GREATER THAN high)  
                        CONTINUE to the next iteration  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN count_strobogrammatic of low and high  
    END FUNCTION  
END CLASS
