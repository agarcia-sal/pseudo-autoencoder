CLASS Solution  
    FUNCTION generateParenthesis(n)  
        FUNCTION backtrack(s as string empty string left as integer zero right as integer zero)  
            IF length of s EQUALS two MULTIPLIED BY n THEN  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n THEN  
                CALL backtrack WITH s CONCATENATED WITH opening parenthesis left INCREMENTED BY one right  
            END IF  
            IF right LESS THAN left THEN  
                CALL backtrack WITH s CONCATENATED WITH closing parenthesis left right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH default parameters  
        RETURN result  
    END FUNCTION  
END CLASS
