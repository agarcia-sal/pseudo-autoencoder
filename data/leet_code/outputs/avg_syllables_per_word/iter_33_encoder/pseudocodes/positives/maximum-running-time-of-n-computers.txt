CLASS Solution  
    FUNCTION maxRunTime WITH parameters n batteries  
        FUNCTION canRunFor WITH parameter time  
            SET total TO zero  
            FOR each battery IN batteries  
                INCREMENT total BY the lesser VALUE BETWEEN battery AND time  
            END FOR  
            RETURN total GREATER THAN OR EQUAL TO time MULTIPLIED BY n  
        END FUNCTION  
        SET left TO zero  
        SET right TO the SUM OF batteries DIVIDED BY n  
        WHILE left LESS THAN right  
            SET mid TO the INTEGER RESULT OF left PLUS right PLUS one DIVIDED BY two  
            IF canRunFor WITH mid IS true  
                SET left TO mid  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
