CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the STRING REPRESENTATION OF count TO path  
                END IF  
                APPEND the CONCATENATION OF all elements IN path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH the value of index PLUS one the list formed by the elements of path followed by the string representation of count followed by the element at position index of word and zero AS arguments  
                ELSE  
                    CALL backtrack WITH the value of index PLUS one the list formed by the elements of path followed by the element at position index of word and zero AS arguments  
                END IF  
                CALL backtrack WITH the value of index PLUS one path and the value of count PLUS one AS arguments  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH zero an empty list and zero AS arguments  
        RETURN result  
    END FUNCTION  
END CLASS
