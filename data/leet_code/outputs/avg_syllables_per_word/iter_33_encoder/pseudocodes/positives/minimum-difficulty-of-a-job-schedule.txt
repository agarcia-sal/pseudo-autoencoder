CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO the LENGTH OF jobDifficulty  
        IF n LESS THAN d THEN  
            RETURN the NEGATIVE of one  
        END IF  
        
        FUNCTION dp(i, d)  
            IF d EQUALS one THEN  
                RETURN the MAXIMUM value IN the SUBLIST OF jobDifficulty STARTING AT position i TO the END  
            END IF  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR j FROM i TO n MINUS d  
                SET current_max TO the GREATER BETWEEN current_max AND element at position j OF jobDifficulty  
                SET candidate TO current_max PLUS dp(j PLUS one, d MINUS one)  
                SET min_difficulty TO the SMALLER BETWEEN min_difficulty AND candidate  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN dp(zero, d)  
    END FUNCTION  
END CLASS
