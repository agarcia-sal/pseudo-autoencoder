CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO the count of each element in nums  
        SET unique_nums TO the sorted list of distinct elements in count  
        SET dp TO a list of zeros with the length equal to the length of unique_nums PLUS one  
        SET element at position one of dp TO the element at position zero of unique_nums MULTIPLIED BY the count of the element at position zero of unique_nums  
        
        FOR index i FROM two TO the length of unique_nums PLUS one  
            SET num TO the element at position i MINUS one of unique_nums  
            
            IF num EQUALS the element at position i MINUS two of unique_nums PLUS one  
                SET element at position i of dp TO the maximum of element at position i MINUS one of dp AND the element at position i MINUS two of dp PLUS num MULTIPLIED BY the count of num  
            ELSE  
                SET element at position i of dp TO the element at position i MINUS one of dp PLUS num MULTIPLIED BY the count of num  
            END IF  
        END FOR  
        
        RETURN the element at the last position of dp  
    END FUNCTION  
END CLASS
