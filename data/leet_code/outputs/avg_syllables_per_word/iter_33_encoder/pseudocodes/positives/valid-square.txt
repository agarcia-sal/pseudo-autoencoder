CLASS Solution  
    FUNCTION validSquare(p1 p2 p3 p4)  
        FUNCTION squared_distance(point1 point2)  
            SET difference_in_x TO element at position zero of point1 MINUS element at position zero of point2  
            SET difference_in_y TO element at position one of point1 MINUS element at position one of point2  
            SET squared_difference_x TO difference_in_x MULTIPLIED BY difference_in_x  
            SET squared_difference_y TO difference_in_y MULTIPLIED BY difference_in_y  
            SET total_squared_distance TO squared_difference_x PLUS squared_difference_y  
            RETURN total_squared_distance  
        END FUNCTION  
      
        SET points TO list consisting of p1 p2 p3 p4  
        SET distances TO empty set  
  
        FOR i FROM zero TO three  
            FOR j FROM i PLUS one TO three  
                SET dist TO squared_distance element at position i of points element at position j of points  
                IF dist EQUALS zero  
                    RETURN false  
                END IF  
                ADD dist TO distances  
            END FOR  
        END FOR  
  
        IF the size of distances EQUALS two  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
