CLASS Solution  
    FUNCTION isTransformable WITH PARAMETERS s t  
        CREATE a mapping from digits to empty lists called pos  
        FOR each index i and character c IN the sequence s  
            ADD the index i TO the heap list corresponding to the digit represented by the character c in pos  
        END FOR  
        FOR each index i and character c IN the sequence t  
            SET digit TO the digit represented by the character c  
            IF the list corresponding to digit in pos IS empty THEN  
                RETURN false  
            END IF  
            FOR each number smaller FROM zero TO digit MINUS one  
                IF the list corresponding to smaller in pos IS NOT empty AND the smallest element in that list IS LESS THAN the smallest element in the list for digit in pos THEN  
                    RETURN false  
                END IF  
            END FOR  
            REMOVE the smallest element FROM the list corresponding to digit in pos  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
