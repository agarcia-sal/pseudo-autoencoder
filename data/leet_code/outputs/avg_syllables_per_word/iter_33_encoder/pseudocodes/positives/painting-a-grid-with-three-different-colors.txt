CLASS Solution  
    FUNCTION colorTheGrid WITH PARAMETERS m n  
        SET MOD TO one billion PLUS seven  
        
        FUNCTION generate_valid_columns WITH PARAMETER prev_col  
            SET colors TO list containing one two three  
            SET valid_cols TO empty list  
            
            FUNCTION backtrack WITH PARAMETERS col row  
                IF row EQUALS m THEN  
                    APPEND the tuple of col TO valid_cols  
                    RETURN  
                END IF  
                FOR each color IN colors  
                    IF col IS empty OR element at position the last index of col NOT EQUALS color THEN  
                        IF prev_col IS NOT none AND element at position row of prev_col EQUALS color THEN  
                            CONTINUE TO next iteration  
                        END IF  
                        APPEND color TO col  
                        CALL backtrack WITH col AND row PLUS one  
                        REMOVE the last element FROM col  
                    END IF  
                END FOR  
            END FUNCTION  
            
            CALL backtrack WITH empty list AND zero  
            RETURN valid_cols  
        END FUNCTION  
        
        SET all_valid_cols TO CALL generate_valid_columns WITH argument none  
        SET num_valid_cols TO the LENGTH OF all_valid_cols  
        
        SET compatible_cols TO empty dictionary WITH keys from zero TO num_valid_cols MINUS one each mapped to an empty list  
        FOR i FROM zero TO num_valid_cols MINUS one  
            FOR j FROM zero TO num_valid_cols MINUS one  
                SET all_positions_different TO true  
                FOR k FROM zero TO m MINUS one  
                    IF element at position k of element at position i of all_valid_cols EQUALS element at position k of element at position j of all_valid_cols THEN  
                        SET all_positions_different TO false  
                        BREAK OUT OF inner loop  
                    END IF  
                END FOR  
                IF all_positions_different THEN  
                    APPEND j TO compatible_cols AT key i  
                END IF  
            END FOR  
        END FOR  
        
        SET dp TO list containing num_valid_cols elements each set to one  
        
        FOR each ignored_variable FROM one TO n MINUS one  
            SET next_dp TO list containing num_valid_cols elements each set to zero  
            FOR i FROM zero TO num_valid_cols MINUS one  
                FOR each j IN compatible_cols AT key i  
                    SET next_dp AT position j TO next_dp AT position j PLUS dp AT position i  
                    SET next_dp AT position j TO next_dp AT position j MODULO MOD  
                END FOR  
            END FOR  
            SET dp TO next_dp  
        END FOR  
        
        SET total_ways TO zero  
        FOR each count IN dp  
            INCREMENT total_ways BY count  
        END FOR  
        RETURN total_ways MODULO MOD  
    END FUNCTION  
END CLASS
