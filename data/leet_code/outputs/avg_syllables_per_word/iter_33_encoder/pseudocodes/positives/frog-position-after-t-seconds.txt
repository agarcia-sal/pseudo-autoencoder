CLASS Solution  
    FUNCTION frogPosition(n parameter, edges parameter, t parameter, target parameter)  
        IF n parameter EQUALS one  
            RETURN the number one point zero  
        END IF  
        SET graph TO a new mapping with default value as empty list  
        FOR each pair u parameter and v parameter IN edges parameter  
            APPEND v parameter TO element at position u parameter of graph  
            APPEND u parameter TO element at position v parameter of graph  
        END FOR  
        SET queue TO a new double ended queue containing a tuple of the number one the number one point zero and the number zero  
        SET visited TO a new set containing the number one  
        WHILE queue is not empty  
            REMOVE and ASSIGN first element of queue TO current_node parameter current_probability parameter current_time parameter  
            IF current_time parameter EQUALS t parameter OR current_node parameter NOT EQUALS one AND THE LENGTH OF element at position current_node parameter of graph EQUALS one  
                IF current_node parameter EQUALS target parameter  
                    RETURN current_probability parameter  
                ELSE  
                    CONTINUE to the next iteration of the WHILE loop  
                END IF  
            END IF  
            SET possible_moves TO THE LENGTH OF element at position current_node parameter of graph MINUS one IF current_node parameter NOT EQUALS one ELSE zero  
            FOR each neighbor parameter IN element at position current_node parameter of graph  
                IF neighbor parameter NOT IN visited  
                    ADD neighbor parameter TO visited  
                    APPEND a tuple of neighbor parameter current_probability parameter DIVIDED BY possible_moves current_time parameter PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN the number zero point zero  
    END FUNCTION  
END CLASS
