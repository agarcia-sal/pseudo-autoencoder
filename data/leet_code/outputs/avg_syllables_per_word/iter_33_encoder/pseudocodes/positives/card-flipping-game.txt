CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO empty set  
        FOR each element x AND each element y in fronts AND backs in parallel  
            IF x EQUALS y  
                ADD x TO same  
            END IF  
        END FOR  
        SET all_numbers TO the set of all elements in fronts and all elements in backs combined  
        SET good_numbers TO all_numbers MINUS same  
        IF good_numbers is not empty  
            SET minimum_good_number TO the smallest element in good_numbers  
            RETURN minimum_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
