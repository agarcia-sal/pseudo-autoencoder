CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the set of tuples formed by converting each element of points into a tuple  
        SET min_x TO the minimum value of the first element of each tuple in unique_points  
        SET max_x TO the maximum value of the first element of each tuple in unique_points  
        SET line_of_reflection TO min_x PLUS max_x DIVIDED BY two  
        FOR each tuple of x and y IN unique_points  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the tuple of reflected_x and y IS NOT an element of unique_points  
                RETURN False  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
