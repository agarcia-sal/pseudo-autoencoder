CLASS Solution  
    FUNCTION numBusesToDestination(routes routes AS List OF List OF Integer, source source AS Integer, target target AS Integer) RETURNS Integer  
        IF source EQUALS target  
            RETURN zero  
        END IF  

        SET stop_to_buses TO an empty mapping from stops TO lists OF buses  
        FOR each bus WITH index bus_index AND route route_list IN routes  
            FOR each stop IN route_list  
                APPEND bus_index TO the list of buses at stop in stop_to_buses  
            END FOR  
        END FOR  

        SET bus_queue TO an empty double-ended queue  
        SET visited_buses TO an empty set  
        SET visited_stops TO an empty set  

        FOR each bus IN the list of buses at source in stop_to_buses  
            APPEND the pair consisting of bus and one TO bus_queue  
            ADD bus TO visited_buses  
            ADD source TO visited_stops  
        END FOR  

        WHILE bus_queue is not empty  
            REMOVE the first element from bus_queue and ASSIGN the bus part TO current_bus and the bus count part TO bus_count  

            FOR each stop IN the route at position current_bus of routes  
                IF stop EQUALS target  
                    RETURN bus_count  
                END IF  

                IF stop is not an element of visited_stops  
                    ADD stop TO visited_stops  
                    FOR each next_bus IN the list of buses at stop in stop_to_buses  
                        IF next_bus is not an element of visited_buses  
                            ADD next_bus TO visited_buses  
                            APPEND the pair consisting of next_bus and bus_count PLUS one TO bus_queue  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
