CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO the LENGTH OF text  
        CREATE an empty SET called substrings  
        FOR each length FROM one TO n DIVIDED BY two INCLUSIVE  
            FOR each start FROM zero TO n MINUS two MULTIPLIED BY length INCLUSIVE  
                SET a TO the SUBSTRING from position start TO position start PLUS length MINUS one OF text  
                SET b TO the SUBSTRING from position start PLUS length TO position start PLUS two MULTIPLIED BY length MINUS one OF text  
                IF a EQUALS b  
                    ADD concatenation of a AND b TO the SET substrings  
                END IF  
            END FOR  
        END FOR  
        SET result TO the SIZE OF substrings  
        RETURN result  
    END FUNCTION  
END CLASS
