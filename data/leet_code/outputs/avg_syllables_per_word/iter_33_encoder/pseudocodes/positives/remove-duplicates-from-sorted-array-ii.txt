CLASS Solution  
    FUNCTION removeDuplicates(nums)  
        IF nums EQUALS empty list THEN  
            RETURN zero  
        END IF  
        SET write_index TO zero  
        FOR i FROM zero TO the smaller value BETWEEN two AND the length of nums MINUS one  
            SET element at position write_index of nums TO element at position i of nums  
            INCREMENT write_index BY one  
        END FOR  
        FOR i FROM two TO the length of nums MINUS one  
            IF element at position i of nums NOT EQUALS element at position write_index MINUS two of nums THEN  
                SET element at position write_index of nums TO element at position i of nums  
                INCREMENT write_index BY one  
            END IF  
        END FOR  
        RETURN write_index  
    END FUNCTION  
END CLASS
