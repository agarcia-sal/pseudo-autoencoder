CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        IF the LENGTH OF nums LESS THAN three THEN  
            RETURN zero  
        END IF  
        
        SET total_slices TO zero  
        SET current_length TO two  
        
        FOR i FROM three TO the LENGTH OF nums  
            IF the element at position i of nums MINUS the element at position i MINUS one of nums EQUALS the element at position i MINUS one of nums MINUS the element at position i MINUS two of nums THEN  
                INCREMENT current_length BY one  
                INCREMENT total_slices BY current_length MINUS two  
            ELSE  
                SET current_length TO two  
            END IF  
        END FOR  
        
        RETURN total_slices  
    END FUNCTION  
END CLASS
