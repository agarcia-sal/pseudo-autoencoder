CLASS Solution  
    FUNCTION leastOpsExpressTarget(x target)  
        FUNCTION dp(target)  
            IF target EQUALS zero  
                RETURN negative one  
            END IF  
            IF target LESS THAN x  
                RETURN the MINIMUM BETWEEN the value of two MULTIPLIED BY target MINUS one AND the value of two MULTIPLIED BY the difference BETWEEN x AND target  
            END IF  
            SET power TO one  
            WHILE the value of x RAISED TO THE power IS LESS THAN OR EQUAL TO target  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET res TO the result of dp of the difference BETWEEN target AND the value of x RAISED TO THE power PLUS the value of power  
            IF the value of x RAISED TO THE sum of power AND one MINUS target IS LESS THAN target  
                SET res TO the MINIMUM BETWEEN res AND the result of dp of the difference BETWEEN the value of x RAISED TO THE sum of power AND one AND target PLUS the value of power PLUS one  
            END IF  
            RETURN res  
        END FUNCTION  
        RETURN dp of target  
    END FUNCTION  
END CLASS
