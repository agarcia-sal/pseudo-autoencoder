CLASS Solution  
    FUNCTION subarraySum(nums k)  
        SET cumulative_sum_freq TO a dictionary with the element zero mapped to the element one  
        SET current_sum TO zero  
        SET count TO zero  
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF current_sum MINUS k EXISTS AS A KEY IN cumulative_sum_freq  
                SET count TO count PLUS the value mapped to the key current_sum MINUS k in cumulative_sum_freq  
            END IF  
            IF current_sum EXISTS AS A KEY IN cumulative_sum_freq  
                SET the value mapped to the key current_sum in cumulative_sum_freq TO the value mapped to the key current_sum in cumulative_sum_freq PLUS one  
            ELSE  
                SET the value mapped to the key current_sum in cumulative_sum_freq TO one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
