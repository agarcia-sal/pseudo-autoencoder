CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            SET first_difference_horizontal TO element at position zero of a MINUS element at position zero of o  
            SET first_difference_vertical TO element at position one of a MINUS element at position one of o  
            SET second_difference_horizontal TO element at position zero of b MINUS element at position zero of o  
            SET second_difference_vertical TO element at position one of b MINUS element at position one of o  
            SET cross_product_value TO first_difference_horizontal MULTIPLIED BY second_difference_vertical MINUS first_difference_vertical MULTIPLIED BY second_difference_horizontal  
            RETURN cross_product_value  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        
        IF n LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  
        
        SET orientation TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET o TO element at position i of points  
            SET a TO element at position (i PLUS one) MODULO n of points  
            SET b TO element at position (i PLUS two) MODULO n of points  
            SET cross TO cross_product(o, a, b)  
            
            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO minus one  
                    END IF  
                ELSE IF (orientation EQUALS one AND cross LESS THAN zero) OR (orientation EQUALS minus one AND cross GREATER THAN zero)  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
