CLASS Solution  
    FUNCTION pourWater(heights the list of integers volume the integer k the integer) RETURNS the list of integers  
        SET n TO the LENGTH OF heights  
        
        FOR each iteration FROM one TO volume  
            SET i TO k  
            WHILE i GREATER THAN one AND element at position i MINUS one of heights LESS THAN OR EQUAL TO element at position i of heights HOLDS  
                DECREMENT i BY one  
            END WHILE  
            WHILE i LESS THAN k AND element at position i of heights EQUALS element at position i PLUS one of heights HOLDS  
                INCREMENT i BY one  
            END WHILE  
            IF i LESS THAN k THEN  
                SET element at position i of heights TO element at position i of heights PLUS one  
                CONTINUE  
            END IF  
            
            SET i TO k  
            WHILE i LESS THAN n MINUS one AND element at position i PLUS one of heights LESS THAN OR EQUAL TO element at position i of heights HOLDS  
                INCREMENT i BY one  
            END WHILE  
            WHILE i GREATER THAN k AND element at position i of heights EQUALS element at position i MINUS one of heights HOLDS  
                DECREMENT i BY one  
            END WHILE  
            IF i GREATER THAN k THEN  
                SET element at position i of heights TO element at position i of heights PLUS one  
                CONTINUE  
            END IF  
            
            SET element at position k of heights TO element at position k of heights PLUS one  
        END FOR  
        
        RETURN heights  
    END FUNCTION  
END CLASS
