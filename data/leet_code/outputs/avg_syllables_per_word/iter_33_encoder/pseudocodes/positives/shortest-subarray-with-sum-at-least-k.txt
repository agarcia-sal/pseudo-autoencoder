CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO a list containing a single element zero  
        FOR each num IN nums  
            APPEND element at the last position of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        CREATE an empty double ended queue named dq  
        SET min_length TO infinity  
        FOR each index i and current_sum in the enumeration of prefix_sum  
            WHILE dq is not empty AND current_sum MINUS the element at the position at the first element of dq in prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET min_length TO the smaller of min_length AND i MINUS the element removed from the first position of dq  
            END WHILE  
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO the element at the position at the last element of dq in prefix_sum  
                REMOVE the element from the last position of dq  
            END WHILE  
            APPEND i TO dq  
        END FOR  
        IF min_length IS NOT EQUAL TO infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
