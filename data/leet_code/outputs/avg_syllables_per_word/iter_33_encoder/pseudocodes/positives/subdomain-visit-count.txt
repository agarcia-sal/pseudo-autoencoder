CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO an empty mapping from string to integer with default zero  
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain by whitespace into count_string and domain_string  
            CONVERT count_string TO integer and ASSIGN TO count  
            SPLIT domain_string by the dot character INTO a list named subdomains  
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one  
                JOIN the elements of subdomains FROM position index TO the last element USING the dot character INTO subdomain  
                INCREMENT the value at domain_count for key subdomain BY count  
            END FOR  
        END FOR  
        SET result_list TO an empty list  
        FOR each key_value_pair IN the items of domain_count  
            EXTRACT domain and count from key_value_pair  
            CREATE a string consisting of the textual representation of count, a space, and domain  
            APPEND this string TO result_list  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
