CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        SET q TO an empty double ended queue initialized with a list containing the element zero  
        SET ans TO an empty list  
        WHILE q is not empty  
            SET path TO the element removed from the front of q  
            SET u TO the element at the last position of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the loop  
            END IF  
            FOR each v IN the element at position u of graph  
                SET new_path TO path with element v appended at the end  
                APPEND new_path TO q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
