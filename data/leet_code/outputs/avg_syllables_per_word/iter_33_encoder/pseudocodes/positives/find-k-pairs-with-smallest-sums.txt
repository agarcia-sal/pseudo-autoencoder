CLASS Solution  
    FUNCTION kSmallestPairs(nums1 nums1 AS List OF INTEGER nums2 nums2 AS List OF INTEGER k k AS INTEGER) RETURNS List OF List OF INTEGER  
        IF nums1 EQUALS empty OR nums2 EQUALS empty OR k LESS THAN OR EQUAL TO zero THEN  
            RETURN empty list  
        END IF  
        SET min_heap TO empty list  
        FOR i FROM zero TO the MINIMUM OF k AND the LENGTH OF nums1 MINUS one  
            PUSH onto min_heap the tuple of the element at position i of nums1 PLUS the element at position zero of nums2 the index i and zero  
        END FOR  
        SET result TO empty list  
        WHILE the LENGTH OF min_heap GREATER THAN zero AND the LENGTH OF result LESS THAN k  
            SET tuple TO the result of popping the smallest element from min_heap  
            SET i TO the second element of tuple  
            SET j TO the third element of tuple  
            APPEND the list containing the element at position i of nums1 and the element at position j of nums2 TO result  
            IF j PLUS one LESS THAN the LENGTH OF nums2 THEN  
                PUSH onto min_heap the tuple of the element at position i of nums1 PLUS the element at position j PLUS one of nums2 the index i and j PLUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
