CLASS Solution  
    FUNCTION myPow(x FLOAT, n INTEGER) RETURNS FLOAT  
        FUNCTION qpow(a FLOAT, n INTEGER) RETURNS FLOAT  
            SET ans TO one  
            WHILE n GREATER THAN zero  
                IF the remainder of n DIVIDED BY two EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two discarding remainder  
            END WHILE  
            RETURN ans  
        END FUNCTION  
  
        IF n GREATER THAN OR EQUAL TO zero  
            RETURN qpow with arguments x and n  
        ELSE  
            RETURN one DIVIDED BY qpow with arguments x and the negative of n  
        END IF  
    END FUNCTION  
END CLASS
