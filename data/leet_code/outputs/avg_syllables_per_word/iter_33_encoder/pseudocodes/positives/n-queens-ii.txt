CLASS Solution  
    FUNCTION totalNQueens with parameter n  
        FUNCTION is_not_under_attack with parameters row and col  
            RETURN NOT the sum of element at position col of cols PLUS element at position difference of row MINUS col of hills PLUS element at position sum of row PLUS col of dales EQUALS zero  
        END FUNCTION  
        
        FUNCTION place_queen with parameters row and col  
            SET element at position col of cols TO one  
            SET element at position difference of row MINUS col of hills TO one  
            SET element at position sum of row PLUS col of dales TO one  
        END FUNCTION  
        
        FUNCTION remove_queen with parameters row and col  
            SET element at position col of cols TO zero  
            SET element at position difference of row MINUS col of hills TO zero  
            SET element at position sum of row PLUS col of dales TO zero  
        END FUNCTION  
        
        FUNCTION backtrack with parameters row and count  
            FOR col FROM zero TO n MINUS one  
                IF is_not_under_attack with arguments row and col EQUALS true  
                    CALL place_queen with arguments row and col  
                    IF sum of row PLUS one EQUALS n  
                        INCREMENT count BY one  
                    ELSE  
                        SET count TO CALL backtrack with arguments sum of row PLUS one and count  
                    END IF  
                    CALL remove_queen with arguments row and col  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET cols TO a list of zero repeated n times  
        SET hills TO a list of zero repeated the quantity of two MULTIPLIED BY n MINUS one times  
        SET dales TO a list of zero repeated the quantity of two MULTIPLIED BY n MINUS one times  
        
        RETURN the result of CALL backtrack with no arguments  
    END FUNCTION  
END CLASS
