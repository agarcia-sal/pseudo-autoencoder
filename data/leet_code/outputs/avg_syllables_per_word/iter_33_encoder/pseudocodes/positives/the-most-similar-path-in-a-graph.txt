CLASS Solution  
    FUNCTION mostSimilar(n targetPath roads names)  
        SET graph TO a new mapping with default empty list  
        FOR each u v pair IN roads  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
        SET m TO the LENGTH OF targetPath  
        FUNCTION dp(i j)  
            IF i EQUALS zero THEN  
                IF element at position j of names NOT EQUALS element at position i of targetPath THEN  
                    RETURN one AND list containing j  
                ELSE  
                    RETURN zero AND list containing j  
                END IF  
            END IF  
            SET min_cost TO positive infinity  
            SET min_path TO empty list  
            FOR each k IN element at position j of graph  
                SET cost_path_pair TO dp(i MINUS one j)  
                SET cost TO first element of cost_path_pair  
                SET path TO second element of cost_path_pair  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                    SET min_path TO path  
                END IF  
            END FOR  
            IF element at position j of names NOT EQUALS element at position i of targetPath THEN  
                SET current_cost TO min_cost PLUS one  
            ELSE  
                SET current_cost TO min_cost  
            END IF  
            RETURN current_cost AND concatenation of min_path AND list containing j  
        END FUNCTION  
        SET min_cost TO positive infinity  
        SET min_path TO empty list  
        FOR each j FROM zero TO n MINUS one  
            SET cost_path_pair TO dp(m MINUS one j)  
            SET cost TO first element of cost_path_pair  
            SET path TO second element of cost_path_pair  
            IF cost LESS THAN min_cost THEN  
                SET min_cost TO cost  
                SET min_path TO path  
            END IF  
        END FOR  
        RETURN min_path  
    END FUNCTION  
END CLASS
