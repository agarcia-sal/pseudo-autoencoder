CLASS Solution  
    FUNCTION carFleet(target position speed)  
        SET pairs TO list of elements each is position element paired with speed element  
        SORT pairs in order from largest to smallest based on the position element of each pair  
        SET time_to_target TO empty list  
        FOR each pair IN pairs  
            SET p TO position element of pair  
            SET s TO speed element of pair  
            SET time TO the difference between target and p DIVIDED BY s  
            APPEND time TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
