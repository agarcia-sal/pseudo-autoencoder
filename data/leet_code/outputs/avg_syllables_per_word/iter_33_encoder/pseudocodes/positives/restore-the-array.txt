CLASS Solution  
    FUNCTION numberOfArrays(s, k)  
        SET MOD to one billion PLUS seven  
        SET n to the LENGTH OF s  
          
        INITIALIZE dp AS a list of zero values of size n PLUS one  
        SET element at position zero of dp TO one  
          
        FOR each i FROM one TO n inclusive  
            FOR each j FROM zero TO i MINUS one inclusive  
                IF element at position j of s NOT EQUALS character zero AND  
                   the integer value represented by the substring of s from position j TO position i MINUS one IS GREATER THAN OR EQUAL TO one AND IS LESS THAN OR EQUAL TO k THEN  
                    SET element at position i of dp TO the sum of element at position i of dp AND element at position j of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
          
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
