CLASS Solution  
    FUNCTION rectangleArea(rectangles)  
        SET MOD TO ten TO THE POWER OF nine PLUS one  
        SET y_coords TO an empty set  
        FOR each rectangle IN rectangles  
            SET x1 TO element at position zero of rectangle  
            SET y1 TO element at position one of rectangle  
            SET x2 TO element at position two of rectangle  
            SET y2 TO element at position three of rectangle  
            ADD y1 TO y_coords  
            ADD y2 TO y_coords  
        END FOR  
        
        SET y_coords TO the sorted list of elements of y_coords  
        SET y_index TO an empty dictionary  
        FOR each index AND y IN the enumeration of y_coords  
            ASSIGN y_index at key y TO index  
        END FOR  
        
        SET count TO a list of zeros with the length equal to the length of y_coords  
        SET events TO an empty list  
        FOR each rectangle IN rectangles  
            SET x1 TO element at position zero of rectangle  
            SET y1 TO element at position one of rectangle  
            SET x2 TO element at position two of rectangle  
            SET y2 TO element at position three of rectangle  
            APPEND a tuple containing x1 PLUS one PLUS y1 PLUS y2 TO events  
            APPEND a tuple containing x2 MINUS one PLUS y1 PLUS y2 TO events  
        END FOR  
        
        SORT events by the first element of each tuple in ascending order  
        
        SET prev_x TO zero  
        SET area TO zero  
        
        FOR each event IN events  
            SET x TO element at position zero of event  
            SET eventType TO element at position one of event  
            SET y1 TO element at position two of event  
            SET y2 TO element at position three of event  
            
            SET current_length TO zero  
            FOR index FROM one TO the length of count MINUS one  
                IF element at position index MINUS one of count GREATER THAN zero THEN  
                    INCREMENT current_length BY the difference between element at position index of y_coords AND element at position index MINUS one of y_coords  
                END IF  
            END FOR  
            
            INCREMENT area BY current_length MULTIPLIED BY the difference between x AND prev_x  
            SET area TO area MODULO MOD  
            
            IF eventType EQUALS one THEN  
                FOR index FROM element at position y_index with key y1 TO element at position y_index with key y2 MINUS one  
                    INCREMENT element at position index of count BY one  
                END FOR  
            ELSE  
                FOR index FROM element at position y_index with key y1 TO element at position y_index with key y2 MINUS one  
                    DECREMENT element at position index of count BY one  
                END FOR  
            END IF  
            
            SET prev_x TO x  
        END FOR  
        
        RETURN area  
    END FUNCTION  
END CLASS
