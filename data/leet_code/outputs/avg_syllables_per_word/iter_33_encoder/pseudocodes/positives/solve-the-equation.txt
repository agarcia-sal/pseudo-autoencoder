CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO the result of replacing each plus sign in side with a special separator followed by plus sign THEN replacing each minus sign in side with a special separator followed by minus sign THEN splitting the resulting string by the special separator  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token is empty string  
                    CONTINUE to next iteration  
                END IF  
                IF token contains the character x  
                    IF token EQUALS the character x OR token EQUALS plus sign followed by character x  
                        INCREMENT x_count BY one  
                    ELSE IF token EQUALS minus sign followed by character x  
                        DECREMENT x_count BY one  
                    ELSE  
                        SET integer_representation TO the integer value of token after removing the character x  
                        INCREMENT x_count BY integer_representation  
                    END IF  
                ELSE  
                    SET integer_value TO the integer value of token  
                    INCREMENT num_sum BY integer_value  
                END IF  
            END FOR  
            RETURN x_count AND num_sum  
        END FUNCTION  

        SET left_side AND right_side TO the two parts of equation separated by equal sign  
        SET left_x AND left_num TO the values returned by parse_side with left_side as argument  
        SET right_x AND right_num TO the values returned by parse_side with right_side as argument  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN the phrase Infinite solutions  
            ELSE  
                RETURN the phrase No solution  
            END IF  
        ELSE  
            SET quotient TO total_num DIVIDED BY total_x using integer division  
            RETURN the string concatenation of character x followed by equal sign followed by quotient  
        END IF  
    END FUNCTION  
END CLASS
