CLASS Solution  
    FUNCTION maximumNumberOfOnes(width height sideLength maxOnes)  
        SET freq TO an empty list of zeros with length sideLength MULTIPLIED BY sideLength  
        FOR variable i FROM zero TO width MINUS one  
            FOR variable j FROM zero TO height MINUS one  
                SET position TO the remainder of variable i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS the remainder of variable j DIVIDED BY sideLength  
                INCREMENT element at position position of freq BY one  
            END FOR  
        END FOR  
        SORT freq IN descending order  
        SET total_sum TO zero  
        FOR index FROM zero TO maxOnes MINUS one  
            INCREMENT total_sum BY element at position index of freq  
        END FOR  
        RETURN total_sum  
    END FUNCTION  
END CLASS
