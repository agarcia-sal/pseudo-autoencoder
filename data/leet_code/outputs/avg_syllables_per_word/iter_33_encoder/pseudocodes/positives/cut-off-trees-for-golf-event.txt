CLASS Solution  
    FUNCTION cutOffTree(forest)  
        IF forest EQUALS empty OR element at position zero of forest EQUALS empty  
            RETURN negative one  
        END IF  

        SET trees TO an empty list  
        FOR index i FROM zero TO the LENGTH OF forest MINUS one  
            FOR index j FROM zero TO the LENGTH OF element at position i of forest MINUS one  
                IF element at position i of forest at position j GREATER THAN one  
                    APPEND a tuple of element at position i of forest at position j AND i AND j TO trees  
                END IF  
            END FOR  
        END FOR  
        SORT trees IN ascending order BY the first element of each tuple  

        FUNCTION bfs(start, end)  
            IF start EQUALS end  
                RETURN zero  
            END IF  
            SET m TO the LENGTH OF forest  
            SET n TO the LENGTH OF element at position zero of forest  
            SET queue TO a deque initialized with start  
            SET visited TO a set initialized with start  
            SET steps TO zero  

            WHILE queue is not empty  
                SET current_queue_length TO the LENGTH OF queue  
                FOR index from zero TO current_queue_length MINUS one  
                    REMOVE the first element FROM queue AND ASSIGN it TO x AND y  
                    FOR each pair dx AND dy IN the list of pairs minus one zero AND plus one zero AND zero minus one AND zero plus one  
                        SET nx TO x PLUS dx  
                        SET ny TO y PLUS dy  
                        IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n  
                            AND the pair nx AND ny IS NOT IN visited  
                            AND element at position nx of forest at position ny NOT EQUALS zero  
                                IF the pair nx AND ny EQUALS end  
                                    RETURN steps PLUS one  
                                END IF  
                                ADD the pair nx AND ny TO visited  
                                APPEND the pair nx AND ny TO queue  
                            END IF  
                        END IF  
                    END FOR  
                END FOR  
                INCREMENT steps BY one  
            END WHILE  

            RETURN negative one  
        END FUNCTION  

        SET x TO zero  
        SET y TO zero  
        SET total_steps TO zero  

        FOR each tuple height AND tx AND ty IN trees  
            SET steps TO bfs with parameters the pair x AND y AND the pair tx AND ty  
            IF steps EQUALS negative one  
                RETURN negative one  
            END IF  
            INCREMENT total_steps BY steps  
            SET x TO tx  
            SET y TO ty  
        END FOR  

        RETURN total_steps  
    END FUNCTION  
END CLASS
