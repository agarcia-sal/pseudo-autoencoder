CLASS Solution  
    FUNCTION updateBoard(board  click)  
        FUNCTION count_adjacent_mines(r  c)  
            SET count TO zero  
            FOR i FROM r MINUS one TO r PLUS one  
                FOR j FROM c MINUS one TO c PLUS one  
                    IF i GREATER THAN OR EQUAL TO zero AND i LESS THAN the LENGTH OF board AND j GREATER THAN OR EQUAL TO zero AND j LESS THAN the LENGTH OF element at position zero of board AND element at position i of board AT position j EQUALS the character MINE  
                        INCREMENT count BY one  
                    END IF  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        FUNCTION dfs(r  c)  
            IF NOT (r GREATER THAN OR EQUAL TO zero AND r LESS THAN the LENGTH OF board AND c GREATER THAN OR EQUAL TO zero AND c LESS THAN the LENGTH OF element at position zero of board)  
                RETURN  
            END IF  
            IF element at position r of board AT position c NOT EQUALS the character EMPTY  
                RETURN  
            END IF  
            
            SET mine_count TO count_adjacent_mines(r  c)  
            IF mine_count GREATER THAN zero  
                SET element at position r of board AT position c TO the STRING REPRESENTATION OF mine_count  
            ELSE  
                SET element at position r of board AT position c TO the character BLANK  
                FOR i FROM r MINUS one TO r PLUS one  
                    FOR j FROM c MINUS one TO c PLUS one  
                        CALL dfs(i  j)  
                    END FOR  
                END FOR  
            END IF  
        END FUNCTION  
        
        DESTRUCTURE click INTO row  col  
        IF element at position row of board AT position col EQUALS the character MINE  
            SET element at position row of board AT position col TO the character EXPLODED  
            RETURN board  
        END IF  
        
        CALL dfs(row  col)  
        RETURN board  
    END FUNCTION  
END CLASS
