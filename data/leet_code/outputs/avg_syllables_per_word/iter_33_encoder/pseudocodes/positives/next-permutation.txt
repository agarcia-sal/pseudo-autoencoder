CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET i TO the LENGTH OF nums MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND element at position i of nums GREATER THAN OR EQUAL TO element at position i PLUS one of nums  
            DECREMENT i BY one  
        END WHILE  
        
        IF i GREATER THAN OR EQUAL TO zero  
            SET j TO the LENGTH OF nums MINUS one  
            WHILE element at position j of nums LESS THAN OR EQUAL TO element at position i of nums  
                DECREMENT j BY one  
            END WHILE  
            
            CALL swap_elements_at_positions WITH nums AND positions i AND j  
        END IF  
        
        CALL reverse_subsequence_starting_at_position WITH nums AND position i PLUS one  
    END FUNCTION  

    FUNCTION swap_elements_at_positions(nums, position_one, position_two)  
        SET temporary_value TO element at position position_one of nums  
        SET element at position position_one of nums TO element at position position_two of nums  
        SET element at position position_two of nums TO temporary_value  
    END FUNCTION  

    FUNCTION reverse_subsequence_starting_at_position(nums, start_position)  
        SET end_position TO the LENGTH OF nums MINUS one  
        WHILE start_position LESS THAN end_position  
            CALL swap_elements_at_positions WITH nums AND positions start_position AND end_position  
            INCREMENT start_position BY one  
            DECREMENT end_position BY one  
        END WHILE  
    END FUNCTION  
END CLASS
