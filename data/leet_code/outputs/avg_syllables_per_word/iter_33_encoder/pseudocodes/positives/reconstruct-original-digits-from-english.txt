CLASS Solution  
    FUNCTION originalDigits(s)  
        SET count TO mapping of character frequencies in s  
        SET digits TO list of zero FOR each position FROM zero TO nine  
        SET digit at position zero TO the frequency of character z in count  
        SET digit at position two TO the frequency of character w in count  
        SET digit at position four TO the frequency of character u in count  
        SET digit at position six TO the frequency of character x in count  
        SET digit at position eight TO the frequency of character g in count  
        SET digit at position three TO the frequency of character h in count MINUS digit at position eight  
        SET digit at position five TO the frequency of character f in count MINUS digit at position four  
        SET digit at position seven TO the frequency of character s in count MINUS digit at position six  
        SET digit at position nine TO the frequency of character i in count MINUS digit at position five MINUS digit at position six MINUS digit at position eight  
        SET digit at position one TO the frequency of character o in count MINUS digit at position zero MINUS digit at position two MINUS digit at position four  
        SET result TO empty list  
        FOR index FROM zero TO nine  
            FOR count_index FROM one TO digit at position index  
                APPEND index converted to string TO result  
            END FOR  
        END FOR  
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
