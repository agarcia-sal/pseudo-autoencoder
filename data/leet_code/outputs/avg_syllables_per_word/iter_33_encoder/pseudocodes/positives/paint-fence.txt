CLASS Solution  
    FUNCTION numWays(n parameter n INTEGER k parameter k INTEGER) RETURNS INTEGER  
        IF n EQUALS zero THEN  
            RETURN zero  
        END IF  
        IF n EQUALS one THEN  
            RETURN k  
        END IF  
        IF n EQUALS two THEN  
            RETURN k MULTIPLIED BY k  
        END IF  
        
        SET same TO k  
        SET diff TO k MULTIPLIED BY k MINUS one  
        SET total TO same PLUS diff  
        
        FOR i FROM three TO n  
            SET new_same TO diff  
            SET new_diff TO total MULTIPLIED BY k MINUS one  
            SET same TO new_same  
            SET diff TO new_diff  
            SET total TO same PLUS diff  
        END FOR  
        
        RETURN total  
    END FUNCTION  
END CLASS
