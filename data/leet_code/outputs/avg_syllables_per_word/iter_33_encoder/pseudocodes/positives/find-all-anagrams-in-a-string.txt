CLASS Solution  
    FUNCTION findAnagrams(s parameter s AS string, p parameter p AS string) RETURNS list of integers  
        SET length of s TO len_s  
        SET length of p TO len_p  
        IF len_p GREATER THAN len_s THEN  
            RETURN an empty list  
        END IF  
        SET p_count TO the frequency counter of each character IN string p  
        SET s_count TO the frequency counter of each character IN the substring from position zero TO position len_p MINUS one of string s  
        SET anagrams TO an empty list  
        IF the frequency counter s_count EQUALS the frequency counter p_count THEN  
            APPEND zero TO anagrams  
        END IF  
        FOR i FROM len_p TO len_s MINUS one  
            INCREMENT by one the count of the character at position i of string s IN s_count  
            DECREMENT by one the count of the character at position i MINUS len_p of string s IN s_count  
            IF the count of the character at position i MINUS len_p of string s IN s_count EQUALS zero THEN  
                REMOVE the character at position i MINUS len_p of string s FROM s_count  
            END IF  
            IF the frequency counter s_count EQUALS the frequency counter p_count THEN  
                APPEND the value of i MINUS len_p PLUS one TO anagrams  
            END IF  
        END FOR  
        RETURN anagrams  
    END FUNCTION  
END CLASS
