CLASS Solution  
    FUNCTION minimumFinishTime(tires number list of list of numbers changeTime number numLaps number) RETURNS number  
        SET min_times TO list of length fifteen filled with positive infinity  
        FOR each pair f r IN tires  
            SET total_time TO zero  
            FOR index i FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER OF i  
                IF lap_time GREATER THAN changeTime PLUS f THEN  
                    BREAK  
                END IF  
                INCREMENT total_time BY lap_time  
                IF min_times at position i GREATER THAN total_time THEN  
                    SET min_times at position i TO total_time  
                END IF  
            END FOR  
        END FOR  
        SET dp TO list of length numLaps PLUS one filled with positive infinity  
        SET dp at position zero TO zero  
        FOR index i FROM one TO numLaps  
            FOR index j FROM zero TO the minimum between i MINUS one AND fourteen  
                SET potential_time TO dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
                IF dp at position i GREATER THAN potential_time THEN  
                    SET dp at position i TO potential_time  
                END IF  
            END FOR  
        END FOR  
        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
