CLASS Solution  
    FUNCTION kthGrammar(n n INTEGER, k k INTEGER) RETURNS INTEGER  
        IF n EQUALS one THEN  
            RETURN zero  
        END IF  
        SET length_of_previous_row TO two MULTIPLIED BY two RAISED TO THE POWER OF n MINUS two MINUS two  
        IF k LESS THAN OR EQUAL TO length_of_previous_row THEN  
            RETURN CALL kthGrammar WITH n MINUS one AND k  
        ELSE  
            RETURN one MINUS CALL kthGrammar WITH n MINUS one AND k MINUS length_of_previous_row  
        END IF  
    END FUNCTION  
END CLASS
