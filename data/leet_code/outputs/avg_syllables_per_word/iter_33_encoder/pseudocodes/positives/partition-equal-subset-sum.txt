CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO the sum of elements in nums  
        IF the remainder of total_sum DIVIDED BY two NOT EQUALS zero  
            RETURN false  
        END IF  
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of false VALUES with the LENGTH OF target PLUS one  
        SET the element at position zero of dp TO true  
        FOR each element num IN nums  
            FOR index i FROM target DOWN TO num  
                SET the element at position i of dp TO the logical OR of the element at position i of dp AND the element at position i MINUS num of dp  
            END FOR  
        END FOR  
        RETURN the element at position target of dp  
    END FUNCTION  
END CLASS
