CLASS Solution  
    FUNCTION scoreOfStudents(s, answers)  
        FUNCTION evaluate_expression(expression)  
            SET tokens TO empty list  
            SET num TO zero  
            FOR each character IN expression  
                IF character is a digit THEN  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of character  
                ELSE  
                    APPEND num TO tokens  
                    APPEND character TO tokens  
                    SET num TO zero  
                END IF  
            END FOR  
            APPEND num TO tokens  
            
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens DO  
                IF element at position i of tokens EQUALS the character representing multiplication THEN  
                    SET element at position i MINUS one of tokens TO element at position i MINUS one of tokens MULTIPLIED BY element at position i PLUS one of tokens  
                    REMOVE elements at positions from i TO i PLUS two MINUS one FROM tokens  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
            
            SET result TO element at position zero of tokens  
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens DO  
                IF element at position i of tokens EQUALS the character representing addition THEN  
                    SET result TO result PLUS element at position i PLUS one of tokens  
                END IF  
                INCREMENT i BY two  
            END WHILE  
            
            RETURN result  
        END FUNCTION  
        
        FUNCTION get_possible_answers(expression)  
            IF expression is composed only of digits THEN  
                RETURN a set containing the integer value of expression  
            END IF  
            
            SET results TO empty set  
            FOR index i FROM one TO the LENGTH OF expression MINUS one MINUS one STEP two DO  
                SET left_results TO get_possible_answers(substring from position zero TO position i MINUS one of expression)  
                SET right_results TO get_possible_answers(substring from position i PLUS one TO the end of expression)  
                FOR each left_element IN left_results DO  
                    FOR each right_element IN right_results DO  
                        IF element at position i of expression EQUALS the character representing addition THEN  
                            SET result TO left_element PLUS right_element  
                        ELSE  
                            SET result TO left_element MULTIPLIED BY right_element  
                        END IF  
                        IF result LESS THAN OR EQUAL TO one thousand THEN  
                            ADD result TO results  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        SET correct_answer TO evaluate_expression(s)  
        SET possible_answers TO get_possible_answers(s)  
        
        SET points TO zero  
        FOR each answer IN answers DO  
            IF answer EQUALS correct_answer THEN  
                INCREMENT points BY five  
            ELSE IF answer IS IN possible_answers THEN  
                INCREMENT points BY two  
            END IF  
        END FOR  
        
        RETURN points  
    END FUNCTION  
END CLASS
