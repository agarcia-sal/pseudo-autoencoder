CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones EQUALS an empty list OR the LENGTH OF stones LESS THAN two THEN  
            RETURN False  
        END IF  
        
        SET stone_set TO a new empty set  
        FOR each stone IN stones  
            ADD stone TO stone_set  
        END FOR  
        
        SET memo TO an empty dictionary  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the last element of stones THEN  
                RETURN True  
            END IF  
            
            IF the tuple of position and jump IS IN memo THEN  
                RETURN the value from memo with key tuple of position and jump  
            END IF  
            
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump GREATER THAN zero AND the sum of position PLUS next_jump IS IN stone_set THEN  
                    IF can_jump_to(the sum of position PLUS next_jump, next_jump) THEN  
                        SET the value in memo with key tuple of position and jump TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the value in memo with key tuple of position and jump TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(the first element of stones, zero)  
    END FUNCTION  
END CLASS
