CLASS Solution  
    FUNCTION getXORSum(arr1 arr1_type AS List OF integers arr2 arr2_type AS List OF integers) RETURNS integer  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        FOR each num IN arr1  
            SET xor_sum_arr1 TO xor_sum_arr1 XOR num  
        END FOR  
        FOR each num IN arr2  
            SET xor_sum_arr2 TO xor_sum_arr2 XOR num  
        END FOR  
        RETURN xor_sum_arr1 AND xor_sum_arr2  
    END FUNCTION  
END CLASS
