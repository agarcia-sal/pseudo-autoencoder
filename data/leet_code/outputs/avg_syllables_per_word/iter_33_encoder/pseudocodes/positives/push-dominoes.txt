CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a list of zeroes with the LENGTH OF n  

        SET force TO zero  
        FOR i FROM zero TO n MINUS one  
            IF element at position i of dominoes EQUALS the character representing rightward push  
                SET force TO n  
            ELSE IF element at position i of dominoes EQUALS the character representing leftward push  
                SET force TO zero  
            ELSE  
                IF force MINUS one GREATER THAN zero  
                    SET force TO force MINUS one  
                ELSE  
                    SET force TO zero  
                END IF  
            END IF  
            INCREMENT element at position i of forces BY force  
        END FOR  

        SET force TO zero  
        FOR i FROM n MINUS one TO zero DECREMENT BY one  
            IF element at position i of dominoes EQUALS the character representing leftward push  
                SET force TO n  
            ELSE IF element at position i of dominoes EQUALS the character representing rightward push  
                SET force TO zero  
            ELSE  
                IF force MINUS one GREATER THAN zero  
                    SET force TO force MINUS one  
                ELSE  
                    SET force TO zero  
                END IF  
            END IF  
            DECREMENT element at position i of forces BY force  
        END FOR  

        SET result TO an empty list  
        FOR each force_value IN forces  
            IF force_value GREATER THAN zero  
                APPEND the character representing rightward push TO result  
            ELSE IF force_value LESS THAN zero  
                APPEND the character representing leftward push TO result  
            ELSE  
                APPEND the character representing no push TO result  
            END IF  
        END FOR  

        RETURN the concatenation of all elements in result as a string  
    END FUNCTION  
END CLASS
