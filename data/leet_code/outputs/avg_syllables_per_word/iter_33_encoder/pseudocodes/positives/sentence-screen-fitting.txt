CLASS Solution  
    FUNCTION wordsTyping(sentence SENTENCE LIST OF STRINGS rows INTEGER cols INTEGER) RETURNS INTEGER  
        SET s TO the concatenation of all elements of sentence with a single space inserted between each followed by a single trailing space  
        SET total_length TO the length of s  
        SET position TO zero  
        FOR each iteration IN the range from one TO rows inclusive  
            INCREMENT position BY cols  
            IF the element at position modulo total_length of s EQUALS a single space  
                INCREMENT position BY one  
            ELSE  
                WHILE position is greater than zero AND the element at position minus one modulo total_length of s NOT EQUALS a single space  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        RETURN the integer division of position BY total_length  
    END FUNCTION  
END CLASS
