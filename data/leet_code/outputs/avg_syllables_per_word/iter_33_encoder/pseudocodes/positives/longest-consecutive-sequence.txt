CLASS Solution  
    FUNCTION longestConsecutive(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN zero  
        END IF  
        SET num_set TO the SET OF nums  
        SET longest_streak TO zero  
        FOR each num IN num_set  
            IF the ELEMENT AT POSITION the VALUE OF num MINUS one OF num_set DOES NOT EXIST THEN  
                SET current_num TO num  
                SET current_streak TO one  
                WHILE the ELEMENT AT POSITION the VALUE OF current_num PLUS one OF num_set EXISTS DO  
                    INCREMENT current_num BY one  
                    INCREMENT current_streak BY one  
                END WHILE  
                SET longest_streak TO the GREATER OF longest_streak AND current_streak  
            END IF  
        END FOR  
        RETURN longest_streak  
    END FUNCTION  
END CLASS
