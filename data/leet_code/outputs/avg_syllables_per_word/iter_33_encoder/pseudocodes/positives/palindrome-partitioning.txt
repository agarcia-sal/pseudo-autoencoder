CLASS Solution  
    FUNCTION partition(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS the reverse of sub  
        END FUNCTION  
        
        FUNCTION backtrack(start, path)  
            IF start EQUALS the length of s THEN  
                APPEND path TO result  
                RETURN  
            END IF  
            FOR end FROM start PLUS one TO the length of s PLUS one  
                IF is_palindrome(the substring from position start TO position end MINUS one of s) THEN  
                    CALL backtrack WITH end AND the concatenation of path AND the substring from position start TO position end MINUS one of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL backtrack WITH zero AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
