CLASS Solution  
    FUNCTION rotate(matrix)  
        SET n TO the LENGTH OF matrix  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET temporary_value TO the element at position j of element at position i of matrix  
                SET the element at position j of element at position i of matrix TO the element at position i of element at position j of matrix  
                SET the element at position i of element at position j of matrix TO temporary_value  
            END FOR  
        END FOR  
        FOR i FROM zero TO n MINUS one  
            CALL reverse_row WITH the element at position i of matrix AS parameter  
        END FOR  
    END FUNCTION  
  
    FUNCTION reverse_row(row)  
        SET left_index TO zero  
        SET right_index TO the LENGTH OF row MINUS one  
        WHILE left_index LESS THAN right_index  
            SET temporary_value TO the element at position right_index of row  
            SET the element at position right_index of row TO the element at position left_index of row  
            SET the element at position left_index of row TO temporary_value  
            INCREMENT left_index BY one  
            DECREMENT right_index BY one  
        END WHILE  
    END FUNCTION  
END CLASS
