CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO an empty collection with keys as characters and values as positions  
        SET max_length TO zero  
        SET start TO zero  

        FOR each position end and character char IN s  
            IF char IS IN char_index_map AND the value at char in char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO the value at char in char_index_map PLUS one  
            END IF  
            SET the value at char in char_index_map TO end  
            SET max_length TO the greater value BETWEEN max_length AND end MINUS start PLUS one  
        END FOR  

        RETURN max_length  
    END FUNCTION  
END CLASS
