CLASS Solution  
    FUNCTION networkDelayTime(times PARAMETER, n PARAMETER, k PARAMETER)  
        SET graph TO an empty mapping from integers to lists  
        FOR index FROM one TO n  
            SET element at position index of graph TO an empty list  
        END FOR  
        FOR each triple u v w IN times  
            APPEND the pair of v AND w TO the list assigned to element at position u of graph  
        END FOR  
        SET dist TO an empty mapping from integers to values  
        FOR index FROM one TO n  
            SET element at position index of dist TO positive infinity  
        END FOR  
        SET element at position k of dist TO zero  
        SET min_heap TO a list containing a pair of zero AND k  
        WHILE the length of min_heap is greater than zero  
            REMOVE the element with the smallest first component FROM min_heap AND ASSIGN those two components TO current_dist AND node  
            IF current_dist GREATER THAN element at position node of dist  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            FOR each pair neighbor AND weight IN element at position node of graph  
                SET distance TO current_dist PLUS weight  
                IF distance LESS THAN element at position neighbor of dist  
                    SET element at position neighbor of dist TO distance  
                    ADD the pair of distance AND neighbor TO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        SET max_time TO the maximum value among all values in dist  
        IF max_time NOT EQUALS positive infinity  
            RETURN max_time  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
