CLASS Solution  
    FUNCTION exclusiveTime WITH PARAMETERS n logs  
        SET exclusive_time TO a list of zero values of length n  
        SET stack TO an empty list  
        FOR each log IN logs  
            SET function_id action timestamp TO the result of splitting log by the word colon  
            CONVERT function_id TO an integer  
            CONVERT timestamp TO an integer  
            IF action EQUALS the word start  
                IF stack is not empty  
                    SET prev_function_id prev_timestamp TO the last element in stack  
                    INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND a pair consisting of function_id and timestamp TO stack  
            ELSE  
                SET prev_function_id prev_timestamp TO the result of removing the last element from stack  
                INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp PLUS one  
                IF stack is not empty  
                    SET last_function_id last_timestamp TO the last element in stack  
                    REPLACE the last element in stack WITH a pair consisting of last_function_id and timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
