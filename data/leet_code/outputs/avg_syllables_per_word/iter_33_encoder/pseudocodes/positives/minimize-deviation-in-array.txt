CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO positive infinity  
        FOR each num IN nums  
            IF remainder of num DIVIDED BY two EQUALS one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            IF min_value GREATER THAN num  
                SET min_value TO num  
            END IF  
            PUSH negative num INTO max_heap as a heap element  
        END FOR  
        
        SET min_deviation TO positive infinity  
        
        WHILE remainder of element at position zero of max_heap DIVIDED BY two EQUALS zero  
            SET max_value TO negative value popped from max_heap as a heap element  
            IF min_deviation GREATER THAN difference between max_value MINUS min_value  
                SET min_deviation TO difference between max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY two using integer division  
            IF min_value GREATER THAN max_value  
                SET min_value TO max_value  
            END IF  
            PUSH negative max_value INTO max_heap as a heap element  
        END WHILE  
        
        IF min_deviation GREATER THAN difference between negative element at position zero of max_heap MINUS min_value  
            SET min_deviation TO difference between negative element at position zero of max_heap MINUS min_value  
        END IF  
        
        RETURN min_deviation  
    END FUNCTION  
END CLASS
