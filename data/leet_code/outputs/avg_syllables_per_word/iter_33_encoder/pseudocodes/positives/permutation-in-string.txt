CLASS Solution  
    FUNCTION checkInclusion(s1  s2)  
        SET length of s1 TO the LENGTH OF s1  
        SET length of s2 TO the LENGTH OF s2  
        IF length of s1 GREATER THAN length of s2  
            RETURN False  
        END IF  
        SET count of characters in s1 TO the frequency count of each character in s1  
        SET count of characters in the first window of s2 TO the frequency count of each character in the substring from position one TO position length of s1 of s2  
        IF count of characters in s1 EQUALS count of characters in the first window of s2  
            RETURN True  
        END IF  
        FOR index FROM position length of s1 PLUS one TO position length of s2  
            INCREMENT the frequency count of the character at position index of s2 BY one in the current window count  
            DECREMENT the frequency count of the character at the position index MINUS length of s1 in s2 BY one in the current window count  
            IF the frequency count of the character at the position index MINUS length of s1 in s2 EQUALS zero in the current window count  
                REMOVE the character at the position index MINUS length of s1 in s2 FROM the current window count  
            END IF  
            IF count of characters in s1 EQUALS the current window count  
                RETURN True  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
