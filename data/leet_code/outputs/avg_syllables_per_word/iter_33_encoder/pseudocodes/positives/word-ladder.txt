CLASS Solution  
    FUNCTION ladderLength(beginWord endWord wordList)  
        IF endWord NOT IN the collection wordList  
            RETURN zero  
        END IF  
        
        SET word_dict TO an empty default dictionary of lists  
        SET word_length TO the LENGTH OF the sequence beginWord  
        
        FOR each word IN the collection wordList  
            FOR index FROM zero TO word_length MINUS one  
                SET intermediate_key TO the concatenation of substring from position zero TO position index MINUS one of word PLUS a placeholder PLUS substring from position index PLUS one TO the end of word  
                APPEND word TO the list associated with intermediate_key in word_dict  
            END FOR  
        END FOR  
        
        SET queue TO a new empty double ended queue with the initial element being a pair consisting of beginWord and one  
        SET visited TO a new set containing beginWord  
        
        WHILE the queue is not empty  
            REMOVE the first element from queue and unpack it into current_word and current_length  
            
            FOR index FROM zero TO word_length MINUS one  
                SET intermediate_word TO the concatenation of substring from position zero TO position index MINUS one of current_word PLUS a placeholder PLUS substring from position index PLUS one TO the end of current_word  
                FOR each word IN the list associated with intermediate_word in word_dict  
                    IF word EQUALS endWord  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited  
                        ADD word TO visited  
                        ADD the pair consisting of word and current_length PLUS one TO the end of queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN zero  
    END FUNCTION  
END CLASS
