CLASS TreeNode  
    FUNCTION __init__(self, val SET TO zero, left SET TO none, right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the collection values EQUALS empty THEN  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node.left TO a new instance of TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node.right TO a new instance of TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the left child of p and the left child of q AND the result of calling is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST(self, root)  
        SET self.max_sum TO zero  
        
        FUNCTION helper(node)  
            IF node EQUALS none THEN  
                RETURN True AND zero AND positive infinity AND negative infinity  
            END IF  
            SET left_is_bst, left_sum, left_min, left_max TO the result of calling helper with the left child of node  
            SET right_is_bst, right_sum, right_min, right_max TO the result of calling helper with the right child of node  
            IF left_is_bst AND right_is_bst AND left_max LESS THAN the value of node AND the value of node LESS THAN right_min THEN  
                SET current_sum TO left_sum PLUS right_sum PLUS the value of node  
                SET current_min TO the smaller value BETWEEN the value of node AND left_min  
                SET current_max TO the larger value BETWEEN the value of node AND right_max  
                SET self.max_sum TO the larger value BETWEEN self.max_sum AND current_sum  
                RETURN True AND current_sum AND current_min AND current_max  
            END IF  
            RETURN False AND zero AND positive infinity AND negative infinity  
        END FUNCTION  
        
        CALL helper with root  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
