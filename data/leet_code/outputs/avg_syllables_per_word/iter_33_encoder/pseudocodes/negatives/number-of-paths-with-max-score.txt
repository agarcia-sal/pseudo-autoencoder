CLASS Solution  
    FUNCTION pathsWithMaxScore(board)  
        SET MOD TO ten POWERED BY nine PLUS one  
        SET n TO the LENGTH OF board  
        SET dp TO a three dimensional list of size n by n where each element is a list containing negative infinity and zero  
        SET the element at position n MINUS one and n MINUS one of dp TO a list containing zero and one  
        FOR i FROM n MINUS one DOWN TO zero INCLUSIVE  
            FOR j FROM n MINUS one DOWN TO zero INCLUSIVE  
                IF the element at position i of board at position j of board EQUALS the character representing a barrier OR the element at position i of board at position j of board EQUALS the character representing the start point  
                    CONTINUE to the next iteration of the inner loop  
                END IF  
                IF the element at position i of board at position j of board EQUALS the character representing the end point  
                    SET value TO zero  
                ELSE  
                    SET value TO the integer representation of the element at position i of board at position j of board  
                END IF  
                FOR each pair consisting of x and y IN the list containing the pairs one and zero, zero and one, one and one  
                    SET ni TO i PLUS x  
                    SET nj TO j PLUS y  
                    IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n  
                        IF the first element of dp at position ni and nj IS GREATER THAN the first element of dp at position i and j  
                            SET dp at position i and j TO a list containing the first element of dp at position ni and nj PLUS value AND the second element of dp at position ni and nj  
                        ELSE IF the first element of dp at position ni and nj EQUALS the first element of dp at position i and j  
                            SET the second element of dp at position i and j TO the sum of the second element of dp at position i and j AND the second element of dp at position ni and nj MODULO MOD  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF the first element of dp at position zero and zero EQUALS negative infinity  
            RETURN a list containing zero AND zero  
        END IF  
        RETURN a list containing the first element of dp at position zero and zero AND the second element of dp at position zero and zero  
    END FUNCTION  
END CLASS
