CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values EQUALS nothing  
        RETURN nothing  
    END IF  
    SET root TO new instance of TreeNode WITH PARAMETER element at position one of values  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO remove and return the first element of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing  
            SET node dot left TO new instance of TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing  
            SET node dot right TO new instance of TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN True  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN False  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN RESULT OF is_same_tree WITH PARAMETERS p dot left q dot left AND RESULT OF is_same_tree WITH PARAMETERS p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION convertBST WITH PARAMETERS self root  
        SET self dot sum TO zero  
        FUNCTION traverse WITH PARAMETER node  
            IF node EQUALS nothing  
                RETURN  
            END IF  
            CALL traverse WITH PARAMETER node dot right  
            INCREMENT self dot sum BY node dot val  
            SET node dot val TO self dot sum  
            CALL traverse WITH PARAMETER node dot left  
        END FUNCTION  
        CALL traverse WITH PARAMETER root  
        RETURN root  
    END FUNCTION  
END CLASS
