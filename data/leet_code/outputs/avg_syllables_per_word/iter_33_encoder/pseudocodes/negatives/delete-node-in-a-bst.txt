CLASS TreeNode  
    FUNCTION __init__ with parameters self val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with element at position one of values  
    SET i TO one  
    INITIALIZE queue AS an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element removed from the left of queue  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p.left and q.left AND the result of is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode with parameters self root key  
        IF root EQUALS None  
            RETURN None  
        END IF  
        IF root.val GREATER THAN key  
            SET root.left TO the result of calling deleteNode on self with root.left and key  
            RETURN root  
        END IF  
        IF root.val LESS THAN key  
            SET root.right TO the result of calling deleteNode on self with root.right and key  
            RETURN root  
        END IF  
        IF root.left EQUALS None  
            RETURN root.right  
        END IF  
        IF root.right EQUALS None  
            RETURN root.left  
        END IF  
        SET node TO root.right  
        WHILE node.left NOT EQUAL TO None  
            SET node TO node.left  
        END WHILE  
        SET node.left TO root.left  
        SET root TO root.right  
        RETURN root  
    END FUNCTION  
END CLASS
