CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value at position one of values AND left None AND right None  
    SET i TO one  
    INITIALIZE queue AS new deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE first element FROM queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with value at position i of values AND left None AND right None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with value at position i of values AND left None AND right None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left attribute of p, left attribute of q) AND is_same_tree(right attribute of p, right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree(self, root)  
        FUNCTION subtree_sum(node)  
            IF node EQUALS None  
                RETURN zero  
            END IF  
            RETURN value at attribute val of node PLUS subtree_sum(left attribute of node) PLUS subtree_sum(right attribute of node)  
        END FUNCTION  
        
        SET total_sum TO subtree_sum(root)  
        
        IF total_sum MODULO two NOT EQUALS zero  
            RETURN False  
        END IF  
        
        SET target TO total_sum DIVIDED BY two  
        SET found TO list containing a single element False  
        
        FUNCTION can_split(node)  
            IF node EQUALS None  
                RETURN zero  
            END IF  
            SET current_sum TO value at attribute val of node PLUS can_split(left attribute of node) PLUS can_split(right attribute of node)  
            IF current_sum EQUALS target AND node NOT EQUALS root  
                SET element at position zero of found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  
        
        CALL can_split(root)  
        RETURN element at position zero of found  
    END FUNCTION  
END CLASS
