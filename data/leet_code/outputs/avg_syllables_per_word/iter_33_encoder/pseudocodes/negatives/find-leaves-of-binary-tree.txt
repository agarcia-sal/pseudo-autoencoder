CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS an empty list  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the first element REMOVED from queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT NONE  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT NONE  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p IS NONE AND q IS NONE  
        RETURN true  
    ELSE IF p IS NONE OR q IS NONE  
        RETURN false  
    ELSE IF the val attribute of p IS NOT EQUALS TO the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments the left attribute of p AND the left attribute of q AND the result of is_same_tree with arguments the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves with parameters root  
        FUNCTION collectLeaves with parameter node  
            IF node IS NONE  
                RETURN minus one  
            END IF  
            SET left_height TO the result of collectLeaves with argument the left attribute of node  
            SET right_height TO the result of collectLeaves with argument the right attribute of node  
            SET current_height TO the maximum of left_height AND right_height PLUS one  
            IF current_height IS GREATER THAN OR EQUAL TO the LENGTH OF result  
                APPEND an empty list TO result  
            END IF  
            APPEND the val attribute of node TO the element at position current_height of result  
            RETURN current_height  
        END FUNCTION  
        SET result TO an empty list  
        CALL collectLeaves with argument root  
        RETURN result  
    END FUNCTION  
END CLASS
