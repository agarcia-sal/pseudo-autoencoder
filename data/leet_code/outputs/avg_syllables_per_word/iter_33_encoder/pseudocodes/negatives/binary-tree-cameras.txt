CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with value equal to the element at position one of values and left and right equal to no value  
    SET i TO one  
    CREATE an empty double ended queue named queue  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not no value THEN  
            SET node.left TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not no value THEN  
            SET node.right TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p is no value AND q is no value THEN  
        RETURN true  
    ELSE IF p is no value OR q is no value THEN  
        RETURN false  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q THEN  
        RETURN false  
    ELSE  
        RETURN the logical AND of is_same_tree called with the left attribute of p and the left attribute of q AND is_same_tree called with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover(self, root)  
        FUNCTION dfs(node)  
            IF node is no value THEN  
                RETURN zero as no camera status, zero as monitored status, and infinite value as camera status  
            END IF  
            SET left_no_camera, left_monitored, left_camera TO the three values returned by dfs called with the left attribute of node  
            SET right_no_camera, right_monitored, right_camera TO the three values returned by dfs called with the right attribute of node  
            SET no_camera TO the sum of left_monitored PLUS right_monitored  
            SET monitored TO the minimum among the following three values:  
                left_camera PLUS the minimum of right_monitored AND right_camera  
                right_camera PLUS the minimum of left_monitored AND left_camera  
                left_camera PLUS right_camera  
            SET camera TO one PLUS the minimum among left_no_camera, left_monitored, left_camera PLUS the minimum among right_no_camera, right_monitored, right_camera  
            RETURN no_camera, monitored, camera  
        END FUNCTION  
        SET root_no_camera, root_monitored, root_camera TO the three values returned by dfs called with root  
        RETURN the minimum of root_monitored AND root_camera  
    END FUNCTION  
END CLASS
