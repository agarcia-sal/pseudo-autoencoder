CLASS Solution  
    FUNCTION calculate WITH parameter s  
        FUNCTION helper WITH parameter s  
            SET stack TO empty list  
            SET sign TO plus  
            SET num TO zero  
            
            WHILE the LENGTH OF s GREATER THAN zero holds  
                SET char TO element at position zero of s  
                REMOVE element at position zero FROM s  
                
                IF char represents a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value represented by char  
                END IF  
                
                IF char EQUALS open parenthesis  
                    SET num TO the result of calling helper WITH argument s  
                END IF  
                
                IF the NEGATION of char represents a digit AND the NEGATION of char represents whitespace OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS plus  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus  
                        APPEND the negative value of num TO stack  
                    ELSE IF sign EQUALS multiplication symbol  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS division symbol  
                        SET the element at the last position of stack TO the integer result of the element at the last position of stack DIVIDED BY num  
                    END IF  
                    
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                
                IF char EQUALS close parenthesis  
                    BREAK the WHILE loop  
                END IF  
            END WHILE  
            
            RETURN the sum of all elements in stack  
        END FUNCTION  
        
        RETURN the result of calling helper WITH argument the list of characters in s  
    END FUNCTION  
END CLASS
