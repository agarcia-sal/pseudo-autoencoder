CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO a new empty set  
        SET min_heap TO a list containing a tuple of the element at position zero zero of grid the number zero the number zero  
        ADD the tuple of the number zero the number zero TO visited  
        SET directions TO a list containing tuples of the number zero and the number one the number one and the number zero the number zero and the negative number one the negative number one and the number zero  
        WHILE min_heap is not empty  
            REMOVE and RETURN the smallest element from min_heap and ASSIGN to variables t i j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            FOR each tuple di dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND the tuple ni nj NOT IN visited  
                    ADD the tuple ni nj TO visited  
                    ADD to min_heap a tuple of the GREATER OF t AND the element at position ni nj of grid the value ni the value nj  
                END IF  
            END FOR  
        END WHILE  
        RETURN the negative number one  
    END FUNCTION  
END CLASS
