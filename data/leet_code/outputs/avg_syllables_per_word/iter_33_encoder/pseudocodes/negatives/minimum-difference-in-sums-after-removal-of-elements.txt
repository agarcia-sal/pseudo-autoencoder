CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
        
        SET min_heap TO empty list  
        SET min_sums TO a list OF zeros WITH LENGTH equal TO two MULTIPLIED BY n PLUS one  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            PUSH negative of element at position i of nums ONTO min_heap  
            IF the LENGTH OF min_heap GREATER THAN n THEN  
                REMOVE the smallest element FROM min_heap  
            END IF  
            IF the LENGTH OF min_heap EQUALS n THEN  
                SET element at position i of min_sums TO negative of the SUM OF elements IN min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO empty list  
        SET max_sums TO a list OF zeros WITH LENGTH equal TO two MULTIPLIED BY n PLUS one  
        
        FOR i FROM the value of three MULTIPLIED BY n MINUS one DOWN TO n  
            PUSH element at position i of nums ONTO max_heap  
            IF the LENGTH OF max_heap GREATER THAN n THEN  
                REMOVE the smallest element FROM max_heap  
            END IF  
            IF the LENGTH OF max_heap EQUALS n THEN  
                SET element at position i of max_sums TO the SUM OF elements IN max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO positive infinity  
        
        FOR i FROM n MINUS one TO two MULTIPLIED BY n MINUS one  
            SET min_diff TO the smaller value BETWEEN min_diff AND element at position i of min_sums MINUS element at position i PLUS one of max_sums  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
