CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO new TreeNode with the element at position one of values  
    SET i TO one  
    INITIALIZE queue as an empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.left TO new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.right TO new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p.left AND q.left AND the result of is_same_tree with p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, inorder, postorder)  
        IF the LENGTH OF inorder EQUALS zero OR the LENGTH OF postorder EQUALS zero  
            RETURN none  
        END IF  
        REMOVE the last element from postorder AND ASSIGN it TO root_val  
        SET root TO new TreeNode with root_val  
        SET root_index TO the position of root_val in inorder  
        SET root.right TO the result of calling buildTree with the slice of inorder from root_index PLUS one TO the end AND postorder  
        SET root.left TO the result of calling buildTree with the slice of inorder from the start TO root_index MINUS one AND postorder  
        RETURN root  
    END FUNCTION  
END CLASS
