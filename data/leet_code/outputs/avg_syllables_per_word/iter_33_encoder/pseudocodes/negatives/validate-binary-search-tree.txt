CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO A new TreeNode with the element at position one of values as val and left as None and right as None  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element taken from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET node.left TO A new TreeNode with the element at position i of values as val and left as None and right as None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET node.right TO A new TreeNode with the element at position i of values as val and left as None and right as None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree applied to the left attribute of p and the left attribute of q AND the result of is_same_tree applied to the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low, high)  
            IF node EQUALS None THEN  
                RETURN True  
            END IF  
            IF the val attribute of node LESS THAN OR EQUAL TO low OR the val attribute of node GREATER THAN OR EQUAL TO high THEN  
                RETURN False  
            END IF  
            RETURN the result of validate applied to the left attribute of node low and the val attribute of node AND the result of validate applied to the right attribute of node the val attribute of node and high  
        END FUNCTION  
        RETURN validate applied to root the smallest possible number and the largest possible number  
    END FUNCTION  
END CLASS
