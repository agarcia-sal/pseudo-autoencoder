CLASS Solution  
    FUNCTION jobScheduling(startTime, endTime, profit)  
        SET jobs TO the collection of tuples combining each element at the same position of startTime endTime profit sorted by the second element of each tuple in ascending order  
        SET end_times TO the collection of the second element of each tuple in jobs in order  
        SET dp TO a list of zeros with the length being the length of jobs PLUS one  
        FOR i FROM one TO the length of jobs  
            SET j TO the position returned by the binary search for the position to insert the first element of the tuple at position i MINUS one of jobs into end_times using right insertion method  
            IF the element at position i MINUS one of dp is GREATER THAN OR EQUAL TO the element at position j of dp PLUS the third element of the tuple at position i MINUS one of jobs THEN  
                SET the element at position i of dp TO the element at position i MINUS one of dp  
            ELSE  
                SET the element at position i of dp TO the element at position j of dp PLUS the third element of the tuple at position i MINUS one of jobs  
            END IF  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
