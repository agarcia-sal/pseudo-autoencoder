CLASS Solution  
    FUNCTION numberOfCombinations(num)  
        SET MOD TO ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten PLUS seven  
        SET n TO the LENGTH OF num  
        SET memo TO an EMPTY DICTIONARY  
        
        FUNCTION dp(index, prev_len)  
            IF index EQUALS n THEN  
                RETURN one  
            END IF  
            IF the PAIR OF index AND prev_len IS IN memo THEN  
                RETURN the VALUE ASSOCIATED WITH the PAIR OF index AND prev_len IN memo  
            END IF  
            
            SET count TO zero  
            FOR length FROM one TO n MINUS index PLUS one  
                IF the ELEMENT at POSITION index OF num EQUALS the CHARACTER ZERO THEN  
                    BREAK OUT OF the LOOP  
                END IF  
                
                SET current_num TO the SUBSTRING FROM POSITION index TO POSITION index PLUS length MINUS one OF num  
                
                IF prev_len EQUALS zero OR current_num GREATER THAN OR EQUAL TO the SUBSTRING FROM POSITION index MINUS prev_len TO POSITION index MINUS one OF num THEN  
                    SET count TO count PLUS THE RESULT OF FUNCTION dp WITH ARGUMENTS index PLUS length AND length  
                    SET count TO count MODULO MOD  
                END IF  
            END FOR  
            
            SET the VALUE ASSOCIATED WITH the PAIR OF index AND prev_len IN memo TO count  
            RETURN count  
        END FUNCTION  
        
        RETURN the RESULT OF FUNCTION dp WITH ARGUMENTS zero AND zero  
    END FUNCTION  
END CLASS
