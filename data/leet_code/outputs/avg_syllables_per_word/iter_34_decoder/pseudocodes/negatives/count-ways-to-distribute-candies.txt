CLASS Solution  
    FUNCTION waysToDistribute(n parameter n INTEGER, k parameter k INTEGER) RETURNS INTEGER  
        SET MOD TO ten POWER nine PLUS one  
        SET dp TO a new two dimensional list with n PLUS one rows and k PLUS one columns each initialized to zero  
        SET element at position zero of row zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET element at position j of row i of dp TO the remainder after dividing the sum of the product of j AND the element at position j of row i MINUS one of dp AND the element at position j MINUS one of row i MINUS one of dp BY MOD  
            END FOR  
        END FOR  
        RETURN the element at position k of row n of dp  
    END FUNCTION  
END CLASS
