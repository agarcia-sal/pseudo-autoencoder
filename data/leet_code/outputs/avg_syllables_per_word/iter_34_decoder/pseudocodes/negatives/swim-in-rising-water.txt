CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an empty set  
        SET min_heap TO a list containing a tuple of the element at position zero of row zero of grid FOLLOWED BY zero FOLLOWED BY zero  
        ADD a tuple of zero FOLLOWED BY zero TO visited  
        SET directions TO a list containing the tuples of zero FOLLOWED BY one FOLLOWED BY one FOLLOWED BY zero FOLLOWED BY zero FOLLOWED BY MINUS one FOLLOWED BY MINUS one FOLLOWED BY zero  
        
        WHILE min_heap is not empty  
            REMOVE the element with the smallest first value of min_heap AND ASSIGN IT TO t AND i AND j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            
            FOR each di AND dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND the tuple of ni FOLLOWED BY nj IS NOT IN visited  
                    ADD the tuple of ni FOLLOWED BY nj TO visited  
                    ADD to min_heap the tuple of the GREATER VALUE BETWEEN t AND the element at position ni of row nj of grid FOLLOWED BY ni FOLLOWED BY nj  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN MINUS one  
    END FUNCTION  
END CLASS
