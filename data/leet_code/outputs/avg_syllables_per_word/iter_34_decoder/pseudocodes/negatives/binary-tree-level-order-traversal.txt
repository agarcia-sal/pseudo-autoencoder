CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with value equal to the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater than zero  
        SET node TO the first element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.left TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.right TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value THEN  
        RETURN true  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN false  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q THEN  
        RETURN false  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the left attribute of p and the left attribute of q AND the result of calling is_same_tree with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root IS no value THEN  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a double-ended queue initialized with root as its only element  
        WHILE the LENGTH OF queue is greater than zero  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR each index FROM zero TO level_size MINUS one  
                SET node TO the first element removed from the left side of queue  
                APPEND the value attribute of node TO current_level  
                IF the left attribute of node IS NOT no value THEN  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node IS NOT no value THEN  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
