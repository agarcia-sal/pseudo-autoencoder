CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET n TO the LENGTH OF s  
        SET last TO a list of negative one repeated twenty six times  
        SET last2 TO a list of negative one repeated twenty six times  
        SET result TO zero  
        FOR each position i and character char IN s with enumeration  
            SET idx TO the character code number of char MINUS the character code number of uppercase A  
            SET contribution TO (position i MINUS element at position idx of last) MULTIPLIED BY (element at position idx of last MINUS element at position idx of last2)  
            SET result TO result PLUS contribution  
            SET result TO result MODULO MOD  
            SET element at position idx of last2 TO element at position idx of last  
            SET element at position idx of last TO position i  
        END FOR  
        FOR each index i FROM zero TO twenty five  
            SET contribution TO (n MINUS element at position i of last) MULTIPLIED BY (element at position i of last MINUS element at position i of last2)  
            SET result TO result PLUS contribution  
            SET result TO result MODULO MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
