CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the val property of p NOT EQUALS the val property of q  
        RETURN false  
    ELSE  
        RETURN the Boolean AND of the result of is_same_tree invoked with the left property of p and the left property of q AND the result of is_same_tree invoked with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow(self, root, val, depth)  
        IF depth EQUALS one  
            RETURN a new TreeNode with val as the val argument AND root as the left argument AND none as the right argument  
        END IF  
        SET queue TO a double ended queue containing a single pair consisting of root AND one  
        WHILE queue is not empty  
            SET node AND level TO the pair removed from the left end of queue  
            IF level EQUALS depth MINUS one  
                SET node.left TO a new TreeNode with val as the val argument AND the original left property of node as the left argument AND none as the right argument  
                SET node.right TO a new TreeNode with val as the val argument AND none as the left argument AND the original right property of node as the right argument  
            ELSE IF level LESS THAN depth MINUS one  
                IF the left property of node is not none  
                    APPEND a pair consisting of the left property of node AND level PLUS one TO queue  
                END IF  
                IF the right property of node is not none  
                    APPEND a pair consisting of the right property of node AND level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
