CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        INITIALIZE skill_to_index_map AS empty mapping  
        SET skill_index TO zero  
        FOR each skill IN req_skills  
            ASSIGN skill_index TO the current skill in skill_to_index_map  
            INCREMENT skill_index BY one  
        END FOR  
        SET total_skills TO the LENGTH OF req_skills  
        SET total_people TO the LENGTH OF people  
        INITIALIZE skill_bit_mask_list AS list of zeros with LENGTH total_people  
        FOR person_index FROM zero TO total_people MINUS one  
            FOR each skill IN the list of skills for the person at person_index IN people  
                SET the element at person_index in skill_bit_mask_list TO the element at person_index in skill_bit_mask_list OR a bit shifted by the skill index from skill_to_index_map for the skill  
            END FOR  
        END FOR  
        INITIALIZE f_list AS list of infinite values with LENGTH TWO POWER total_skills  
        INITIALIZE g_list AS list of zeros with LENGTH TWO POWER total_skills  
        INITIALIZE h_list AS list of zeros with LENGTH TWO POWER total_skills  
        SET the element at position zero in f_list TO zero  
        FOR skill_state FROM zero TO TWO POWER total_skills MINUS one  
            IF the element at position skill_state in f_list EQUALS infinite  
                CONTINUE to next iteration  
            END IF  
            FOR person_index FROM zero TO total_people MINUS one  
                SET combined_state TO skill_state OR the element at person_index in skill_bit_mask_list  
                IF the element at skill_state in f_list PLUS one LESS THAN the element at combined_state in f_list  
                    SET the element at combined_state in f_list TO the element at skill_state in f_list PLUS one  
                    SET the element at combined_state in g_list TO person_index  
                    SET the element at combined_state in h_list TO skill_state  
                END IF  
            END FOR  
        END FOR  
        SET current_state TO TWO POWER total_skills MINUS one  
        INITIALIZE answer_list AS empty list  
        WHILE current_state NOT EQUAL TO zero  
            APPEND the element at current_state in g_list TO answer_list  
            SET current_state TO the element at current_state in h_list  
        END WHILE  
        RETURN answer_list  
    END FUNCTION  
END CLASS
