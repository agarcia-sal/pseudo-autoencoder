CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET property val TO parameter val  
        SET property next TO parameter next  
    END FUNCTION  
END CLASS

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with element at position one of values  
    SET p TO head  
    FOR each val IN elements of values FROM position two TO the end  
        SET node TO new ListNode with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the val property of p1 EQUALS the val property of p2 AND is_same_list with parameters the next property of p1 AND the next property of p2  
END FUNCTION

CLASS TreeNode  
    FUNCTION __init__ with parameters val AND left AND right  
        SET property val TO parameter val  
        SET property left TO parameter left  
        SET property right TO parameter right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET the left property of node TO new TreeNode with element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET the right property of node TO new TreeNode with element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree with parameters p AND q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the val property of p NOT EQUALS the val property of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters the left property of p AND the left property of q AND is_same_tree with parameters the right property of p AND the right property of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION sortedListToBST with parameter head  
        IF head EQUALS None  
            RETURN None  
        END IF  

        FUNCTION findMiddle with parameters left AND right  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUALS right AND the next property of fast NOT EQUALS right  
                SET slow TO the next property of slow  
                SET fast TO the next property of the next property of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  

        FUNCTION convertListToBST with parameters left AND right  
            IF left EQUALS right  
                RETURN None  
            END IF  
            SET mid TO findMiddle with parameters left AND right  
            SET node TO new TreeNode with the val property of mid  
            SET the left property of node TO convertListToBST with parameters left AND mid  
            SET the right property of node TO convertListToBST with parameters the next property of mid AND right  
            RETURN node  
        END FUNCTION  

        RETURN convertListToBST with parameters head AND None  
    END FUNCTION  
END CLASS
