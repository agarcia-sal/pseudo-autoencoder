CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the collection values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO a new TreeNode CREATED WITH the element at position one of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element REMOVED FROM the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode CREATED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode CREATED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH the left attribute of p AND the left attribute of q  
            AND the result of is_same_tree WITH the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, inorder, postorder)  
        IF inorder IS EMPTY OR postorder IS EMPTY  
            RETURN None  
        END IF  
        SET root_val TO the element REMOVED FROM the end of postorder  
        SET root TO a new TreeNode CREATED WITH root_val  
        SET root_index TO the position of root_val IN inorder  
        SET root.right TO the result of calling buildTree ON the sublist of inorder FROM position root_index PLUS one TO the end  
            AND the current postorder  
        SET root.left TO the result of calling buildTree ON the sublist of inorder FROM the start TO position root_index MINUS one  
            AND the current postorder  
        RETURN root  
    END FUNCTION  
END CLASS
