CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY THEN  
        RETURN None  
    END IF  
    SET head TO a new ListNode constructed with element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values from position two TO the last position  
        SET node TO a new ListNode constructed with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 IS None AND p2 IS None THEN  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None THEN  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND is_same_list(the next element of p1 the next element of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self head)  
        IF head IS None OR the next element of head IS None THEN  
            RETURN head  
        END IF  
        SET dummy TO a new ListNode constructed with the value negative five thousand and one  
        SET the next element of dummy TO head  
        SET current TO the next element of head  
        SET the next element of head TO None  
        WHILE current IS NOT None  
            SET next_node TO the next element of current  
            SET prev TO dummy  
            WHILE the next element of prev IS NOT None AND the value of the next element of prev IS LESS THAN the value of current  
                SET prev TO the next element of prev  
            END WHILE  
            SET the next element of current TO the next element of prev  
            SET the next element of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
