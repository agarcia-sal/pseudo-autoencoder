CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the val attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values starting from the position one to the end  
        SET node TO a new ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the condition that the val attribute of p1 EQUALS the val attribute of p2 AND the recursive call to is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ with parameters val AND left AND right  
        SET the val attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    CALL the append method of queue with root  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the result of calling the method that removes and returns the element from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values DOES NOT EQUAL None  
            SET the left attribute of node TO a new TreeNode with the element at position i of values  
            CALL the append method of queue with the left attribute of node  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values DOES NOT EQUAL None  
            SET the right attribute of node TO a new TreeNode with the element at position i of values  
            CALL the append method of queue with the right attribute of node  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p AND q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of the logical AND of the recursive call to is_same_tree with the left attribute of p AND the left attribute of q AND the recursive call to is_same_tree with the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree with parameter s  
        FUNCTION parse_tree with parameter index  
            IF index GREATER THAN OR EQUAL TO the LENGTH OF s  
                RETURN None AND index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the LENGTH OF s AND the element at position index of s IS A DIGIT OR the element at position index of s EQUALS the character indicating negative sign  
                INCREMENT index BY one  
            END WHILE  
            SET node TO a new TreeNode created with the integer value represented by the substring of s from position start TO position index MINUS one  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character indicating an open parenthesis  
                SET the left attribute of node AND index TO the result of calling parse_tree with index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character indicating an open parenthesis  
                SET the right attribute of node AND index TO the result of calling parse_tree with index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character indicating a close parenthesis  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND discard TO the result of calling parse_tree with zero  
        RETURN root  
    END FUNCTION  
END CLASS
