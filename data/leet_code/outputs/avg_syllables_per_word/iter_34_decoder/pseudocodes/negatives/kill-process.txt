CLASS Solution  
    FUNCTION killProcess(pid parameter as list of integers ppid parameter as list of integers kill parameter as integer) RETURNS list of integers  
        SET tree TO a mapping from integer to list of integers with default empty list  
        FOR each pair child and parent IN the combination of pid and ppid  
            APPEND child TO the list at key parent in tree  
        END FOR  
        SET queue TO a list containing kill  
        SET killed_processes TO an empty list  
        WHILE the LENGTH OF queue is GREATER THAN zero  
            REMOVE the element at position zero from queue and ASSIGN it TO current  
            APPEND current TO killed_processes  
            APPEND all elements from the list at key current in tree TO queue  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
