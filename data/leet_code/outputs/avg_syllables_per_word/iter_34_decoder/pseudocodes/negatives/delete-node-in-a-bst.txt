CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the value property TO val  
        SET the left property TO left  
        SET the right property TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the list values does not have any elements  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode created with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue has elements  
        SET node TO the first element removed from the left of queue  
        IF i is less than the length of values AND the element at position i of values is not no value  
            SET the left property of node TO a new TreeNode created with the element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not no value  
            SET the right property of node TO a new TreeNode created with the element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p does not exist AND q does not exist  
        RETURN true  
    END IF  
    IF p does not exist OR q does not exist  
        RETURN false  
    END IF  
    IF the value property of p NOT EQUALS the value property of q  
        RETURN false  
    END IF  
    RETURN the result of is_same_tree called with the left property of p and the left property of q AND the result of is_same_tree called with the right property of p and the right property of q  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode with parameters root key  
        IF root does not exist  
            RETURN no value  
        END IF  
        IF the value property of root GREATER THAN key  
            SET the left property of root TO the result of calling deleteNode with the left property of root and key  
            RETURN root  
        END IF  
        IF the value property of root LESS THAN key  
            SET the right property of root TO the result of calling deleteNode with the right property of root and key  
            RETURN root  
        END IF  
        IF the left property of root does not exist  
            RETURN the right property of root  
        END IF  
        IF the right property of root does not exist  
            RETURN the left property of root  
        END IF  
        SET node TO the right property of root  
        WHILE the left property of node exists  
            SET node TO the left property of node  
        END WHILE  
        SET the left property of node TO the left property of root  
        SET root TO the right property of root  
        RETURN root  
    END FUNCTION  
END CLASS
