CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of each element in nums  
        SET dp TO a list of zeroes with length equal to two raised to the power of ten  
        SET the element at position zero of dp TO one  
        
        FOR each num IN the keys of count  
            IF num EQUALS one  
                SET the element at position zero of dp TO the element at position zero of dp MULTIPLIED BY two raised to the power of the element at position one in count taken as an integer  
                CONTINUE TO the next iteration  
            END IF  
            
            IF num NOT EQUALS any key in the prime factors mapping  
                CONTINUE TO the next iteration  
            END IF  
            
            SET factors TO the list of prime factors associated with num in the prime factors mapping  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask OR two raised to the power of prime MINUS one  
            END FOR  
            
            FOR m FROM two raised to the power of ten MINUS one DOWN TO zero  
                IF the bitwise AND of m and mask EQUALS zero  
                    SET the element at position the bitwise OR of m and mask of dp TO the sum of the current element at that position of dp AND the product of the element at position m of dp MULTIPLIED BY the frequency count of num TAKEN MODULO the given modulus  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of elements in dp from position one onwards TAKEN MODULO the given modulus  
        RETURN result  
    END FUNCTION  
END CLASS
