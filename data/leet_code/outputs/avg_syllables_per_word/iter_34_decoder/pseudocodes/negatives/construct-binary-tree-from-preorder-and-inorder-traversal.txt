CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values and no left and right  
    SET i TO one  
    CREATE empty queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET node.left TO new TreeNode with value equal to element at position i of values and no left and right  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET node.right TO new TreeNode with value equal to element at position i of values and no left and right  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value  
        RETURN true  
    ELSE IF p IS no value OR q IS no value  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with p.left and q.left AND the result of is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF the LENGTH OF preorder EQUALS zero OR the LENGTH OF inorder EQUALS zero  
            RETURN no value  
        END IF  
        SET root_val TO element at position one of preorder  
        SET root TO new TreeNode with value equal to root_val and no left and right  
        SET root_index_inorder TO the position of root_val in inorder  
        SET root.left TO the result of calling buildTree of self with the sublist of preorder from position two TO the position root_index_inorder PLUS one and the sublist of inorder from position one TO the position root_index_inorder  
        SET root.right TO the result of calling buildTree of self with the sublist of preorder from the position root_index_inorder PLUS two TO the end and the sublist of inorder from the position root_index_inorder PLUS two TO the end  
        RETURN root  
    END FUNCTION  
END CLASS
