CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value equal to the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater THAN zero  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.left TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.right TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum(self, root)  
        FUNCTION helper(node)  
            IF node EQUALS None  
                RETURN negative infinity, negative infinity  
            END IF  
            SET left_max, left_gain TO helper(the left child of node)  
            SET right_max, right_gain TO helper(the right child of node)  
            SET current_max TO the value of node PLUS the maximum of left_gain AND zero PLUS the maximum of right_gain AND zero  
            SET current_gain TO the value of node PLUS the maximum of left_gain, right_gain, AND zero  
            SET global_max TO the maximum of left_max, right_max, AND current_max  
            RETURN global_max, current_gain  
        END FUNCTION  
        RETURN the first element of helper(root)  
    END FUNCTION  
END CLASS
