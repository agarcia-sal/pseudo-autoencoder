CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode WITH value ELEMENT at position zero OF values AND left None AND right None  
    SET i TO one  
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND ELEMENT at position i OF values IS NOT None  
            SET node.left TO A NEW TreeNode WITH value ELEMENT at position i OF values AND left None AND right None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND ELEMENT at position i OF values IS NOT None  
            SET node.right TO A NEW TreeNode WITH value ELEMENT at position i OF values AND left None AND right None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self, root)  
        IF root IS None  
            RETURN AN EMPTY LIST  
        END IF  
        SET result TO AN EMPTY LIST  
        SET queue TO A DOUBLE ENDED QUEUE INITIALIZED WITH root  
        SET left_to_right TO True  
        WHILE queue IS NOT EMPTY  
            SET level_size TO THE LENGTH OF queue  
            SET current_level TO AN EMPTY DOUBLE ENDED QUEUE  
            FOR iterator FROM zero TO level_size MINUS one  
                SET node TO THE FIRST ELEMENT REMOVED FROM queue  
                IF left_to_right IS True  
                    APPEND node.val TO current_level  
                ELSE  
                    INSERT node.val AT THE BEGINNING OF current_level  
                END IF  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND A LIST CONVERTED FROM current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
