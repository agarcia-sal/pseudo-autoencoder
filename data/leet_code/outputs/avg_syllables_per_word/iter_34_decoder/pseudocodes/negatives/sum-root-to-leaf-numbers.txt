CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the values collection is empty  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode initialized with element at position one of values  
    SET i TO one  
    CREATE an empty double ended queue called queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i is less than the length of values AND element at position i of values is not nothing  
            SET the left attribute of node TO a new TreeNode initialized with element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND element at position i of values is not nothing  
            SET the right attribute of node TO a new TreeNode initialized with element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p is nothing AND q is nothing  
        RETURN True  
    ELSE IF p is nothing OR q is nothing  
        RETURN False  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q  
        RETURN False  
    ELSE  
        RETURN the logical AND of is_same_tree with the left attribute of p and the left attribute of q AND is_same_tree with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers with parameters root  
        FUNCTION dfs with parameters node current_number  
            IF node is nothing  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS the value attribute of node  
            IF the left attribute of node is nothing AND the right attribute of node is nothing  
                RETURN current_number  
            END IF  
            SET left_sum TO dfs with the left attribute of node and current_number  
            SET right_sum TO dfs with the right attribute of node and current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN dfs with root and zero  
    END FUNCTION  
END CLASS
