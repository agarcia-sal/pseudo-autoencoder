CLASS Solution  
    FUNCTION countOrders WITH PARAMETER n  
        SET MOD TO the number one followed by nine zeros PLUS the number seven  
        SET count TO the number one  
        FOR i FROM the number two TO n  
            SET count TO count MULTIPLIED BY i MULTIPLIED BY the number two MULTIPLIED BY i MINUS the number one  
            SET count TO count MODULO MOD  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
