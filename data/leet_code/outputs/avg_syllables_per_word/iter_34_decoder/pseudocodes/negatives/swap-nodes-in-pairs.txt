CLASS ListNode  
    FUNCTION __init__(self val SET TO zero next SET TO None)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with value as element at position one of values  
    SET p TO head  
    FOR each val IN elements of values starting from position two to the end  
        SET node TO new ListNode with value as val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value of element p1 dot val EQUALS the value of element p2 dot val AND the result of is_same_list with arguments p1 dot next AND p2 dot next  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs(self head)  
        SET dummy TO new ListNode with value SET TO zero  
        SET dummy dot next TO head  
        SET current TO dummy  
        
        WHILE current dot next NOT EQUALS None AND current dot next dot next NOT EQUALS None  
            SET first TO current dot next  
            SET second TO current dot next dot next  
            
            SET current dot next TO second  
            SET first dot next TO second dot next  
            SET second dot next TO first  
            
            SET current TO current dot next dot next  
        END WHILE  
        
        RETURN dummy dot next  
    END FUNCTION  
END CLASS
