CLASS Solution  
    FUNCTION numberOfWays WITH PARAMETER corridor  
        SET MOD TO one billion PLUS seven  
        SET seats TO empty list  
        FOR each index AND element IN corridor  
            IF element EQUALS seat symbol  
                APPEND index TO seats  
            END IF  
        END FOR  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM three TO the LENGTH OF seats IN STEPS OF two  
            SET difference TO element at position index of seats MINUS element at position index MINUS one of seats  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
