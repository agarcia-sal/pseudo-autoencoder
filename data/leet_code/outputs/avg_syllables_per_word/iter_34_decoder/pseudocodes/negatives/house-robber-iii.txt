CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty list THEN  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode with value element at position one of values WITH left set to nothing AND right set to nothing  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing THEN  
            SET node.left TO a new TreeNode with value element at position i of values WITH left set to nothing AND right set to nothing  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing THEN  
            SET node.right TO a new TreeNode with value element at position i of values WITH left set to nothing AND right set to nothing  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS nothing AND q EQUALS nothing THEN  
        RETURN True  
    ELSE IF p EQUALS nothing OR q EQUALS nothing THEN  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rob(self, root)  
        FUNCTION rob_from_node(node)  
            IF node EQUALS nothing THEN  
                RETURN zero, zero  
            END IF  
            SET left_robbed, left_not_robbed TO rob_from_node(the left child of node)  
            SET right_robbed, right_not_robbed TO rob_from_node(the right child of node)  
            SET rob_this_node TO the value of node PLUS left_not_robbed PLUS right_not_robbed  
            SET not_rob_this_node TO the maximum of left_robbed AND left_not_robbed PLUS the maximum of right_robbed AND right_not_robbed  
            RETURN rob_this_node, not_rob_this_node  
        END FUNCTION  
        RETURN the maximum of rob_from_node(root)  
    END FUNCTION  
END CLASS
