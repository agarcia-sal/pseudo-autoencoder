CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET transformed_expression TO expression WITH every occurrence of MINUS REPLACED BY PLUS MINUS  
        SET list_of_fractions TO the result of SPLITTING transformed_expression BY PLUS  
        SET sum_of_fractions TO the result of SUMMING a fraction object CREATED FROM each item IN list_of_fractions WHEN the item IS NOT empty  
        SET numerator_of_result TO the numerator PART of sum_of_fractions  
        SET denominator_of_result TO the denominator PART of sum_of_fractions  
        SET final_result TO the concatenation OF numerator_of_result AS string PLUS a forward slash PLUS denominator_of_result AS string  
        RETURN final_result  
    END FUNCTION  
END CLASS
