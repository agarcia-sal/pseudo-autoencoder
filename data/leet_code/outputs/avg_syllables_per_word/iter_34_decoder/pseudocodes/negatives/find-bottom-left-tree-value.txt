CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the input list values IS EMPTY  
        RETURN null  
    END IF  
    SET root TO a new TreeNode with value equal to element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT null  
            SET node.left TO a new TreeNode with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT null  
            SET node.right TO a new TreeNode with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS null AND q IS null  
        RETURN True  
    ELSE IF p IS null OR q IS null  
        RETURN False  
    ELSE IF the value of p DOES NOT EQUAL the value of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with p.left and q.left AND the result of calling is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self, root)  
        SET queue TO a double ended queue containing root  
        SET leftmost_value TO null  
        WHILE queue IS NOT EMPTY  
            SET level_size TO the LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one  
                SET node TO the element removed from the front of queue  
                IF i EQUALS zero  
                    SET leftmost_value TO the value of node  
                END IF  
                IF the left child of node EXISTS  
                    APPEND the left child of node TO queue  
                END IF  
                IF the right child of node EXISTS  
                    APPEND the right child of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
