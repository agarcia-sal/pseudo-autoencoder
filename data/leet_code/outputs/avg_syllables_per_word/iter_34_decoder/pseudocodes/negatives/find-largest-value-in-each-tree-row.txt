CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO a new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO a new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments p.left and q.left AND the result of is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION largestValues(self, root)  
        IF root IS none  
            RETURN an empty list  
        END IF  
        SET queue TO a new double ended queue containing root  
        SET largest_values TO an empty list  
        WHILE queue is not empty  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR index FROM zero TO level_size MINUS one  
                SET node TO the first element removed from the left side of queue  
                IF max_value LESS THAN node.val  
                    SET max_value TO node.val  
                END IF  
                IF node.left IS NOT none  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT none  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
