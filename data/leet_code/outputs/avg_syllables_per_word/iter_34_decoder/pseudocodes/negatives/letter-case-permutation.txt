CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO the LENGTH OF t  
                APPEND concatenation of elements of t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH argument i PLUS one  
            IF element at position i of t IS an alphabetic character  
                SET element at position i of t TO the character obtained by toggling case of element at position i of t  
                CALL dfs WITH argument i PLUS one  
            END IF  
        END FUNCTION  
        SET t TO the list of characters of s  
        SET ans TO an empty list  
        CALL dfs WITH argument zero  
        RETURN ans  
    END FUNCTION  
END CLASS
