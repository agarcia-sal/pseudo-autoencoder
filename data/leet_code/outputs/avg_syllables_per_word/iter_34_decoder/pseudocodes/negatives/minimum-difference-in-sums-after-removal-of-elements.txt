CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
        
        SET min_heap TO an empty list  
        SET min_sums TO a list of zeros with the LENGTH OF two MULTIPLIED BY n PLUS one  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            PUSH the NEGATIVE of element at position i of nums INTO min_heap  
            IF the LENGTH OF min_heap GREATER THAN n THEN  
                REMOVE the smallest element FROM min_heap  
            END IF  
            IF the LENGTH OF min_heap EQUALS n THEN  
                SET element at position i of min_sums TO the NEGATIVE of the SUM of elements in min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO an empty list  
        SET max_sums TO a list of zeros with the LENGTH OF two MULTIPLIED BY n PLUS one  
        
        FOR i FROM three MULTIPLIED BY n MINUS one DOWN TO n  
            PUSH element at position i of nums INTO max_heap  
            IF the LENGTH OF max_heap GREATER THAN n THEN  
                REMOVE the smallest element FROM max_heap  
            END IF  
            IF the LENGTH OF max_heap EQUALS n THEN  
                SET element at position i of max_sums TO the SUM of elements in max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO infinity  
        FOR i FROM n MINUS one TO two MULTIPLIED BY n MINUS one  
            SET min_diff TO the MINIMUM of min_diff AND element at position i of min_sums MINUS element at position i PLUS one of max_sums  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
