CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO A NEW INSTANCE OF TreeNode USING element at position one of values  
    SET i TO one  
    SET queue TO AN EMPTY DEQUE  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO A NEW INSTANCE OF TreeNode USING element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO A NEW INSTANCE OF TreeNode USING element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree OF p.left AND q.left AND is_same_tree OF p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION printTree(self, root)  
        FUNCTION getHeight(node)  
            IF node IS none  
                RETURN zero  
            END IF  
            RETURN one PLUS THE GREATER VALUE BETWEEN getHeight OF node.left AND getHeight OF node.right  
        END FUNCTION  

        SET height TO getHeight OF root  
        SET width TO two RAISED TO THE POWER OF height MINUS one  
        SET res TO A LIST OF height ELEMENTS WHERE EACH ELEMENT IS A LIST OF width ELEMENTS CONTAINING EMPTY STRINGS  

        FUNCTION placeNode(node, row, left, right)  
            IF node IS none  
                RETURN  
            END IF  
            SET mid TO THE INTEGER VALUE OF left PLUS right DIVIDED BY two  
            SET element at position mid of element at position row of res TO THE STRING REPRESENTATION OF node.val  
            CALL placeNode WITH node.left, row PLUS one, left, mid  
            CALL placeNode WITH node.right, row PLUS one, mid, right  
        END FUNCTION  

        CALL placeNode WITH root, zero, zero, width  
        RETURN res  
    END FUNCTION  
END CLASS
