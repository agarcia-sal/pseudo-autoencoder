CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values as value and nothing as left and right  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS nothing  
            SET node.left TO a new TreeNode with the element at position i of values as value and nothing as left and right  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS nothing  
            SET node.right TO a new TreeNode with the element at position i of values as value and nothing as left and right  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN true  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN false  
    ELSE IF the value property of p NOT EQUALS the value property of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree called with the left property of p and the left property of q AND the result of is_same_tree called with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView(self, root)  
        IF root EQUALS nothing  
            RETURN an empty list  
        END IF  
        SET right_side_view TO an empty list  
        SET queue TO a double ended queue initialized with root as the single element  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET level_length TO the LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO the element removed from the front of queue  
                IF i EQUALS level_length MINUS one  
                    APPEND the value property of node TO right_side_view  
                END IF  
                IF the left property of node NOT EQUALS nothing  
                    APPEND the left property of node TO queue  
                END IF  
                IF the right property of node NOT EQUALS nothing  
                    APPEND the right property of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
