CLASS Solution  
    FUNCTION profitableSchemes(n profit_minimum group profit)  
        SET constant_mod TO ten POWERED BY nine PLUS one  
        SET dp TO a list containing n PLUS one elements where each element IS a list containing profit_minimum PLUS one zeros  
        FOR index_i FROM zero TO n  
            SET element at position zero of element at position index_i of dp TO one  
        END FOR  
        
        FOR each element_profit AND element_group IN profit AND group RESPECTIVELY  
            FOR index_i FROM n DOWN TO element_group  
                FOR index_j FROM profit_minimum DOWN TO zero  
                    SET value_previous TO element at position index_i MINUS element_group of dp  
                    SET j_minus_p TO index_j MINUS element_profit  
                    IF j_minus_p LESS THAN zero  
                        SET j_minus_p TO zero  
                    END IF  
                    SET value_to_add TO element at position j_minus_p of value_previous  
                    SET element at position index_j of element at position index_i of dp TO element at position index_j of element at position index_i of dp PLUS value_to_add  
                    SET element at position index_j of element at position index_i of dp TO element at position index_j of element at position index_i of dp MODULO constant_mod  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position profit_minimum of element at position n of dp  
    END FUNCTION  
END CLASS
