CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET dictionary d TO empty mapping  
        FOR index i FROM zero TO the LENGTH OF order MINUS one  
            SET element at position i of order TO character c  
            SET element at position c of dictionary d TO i  
        END FOR  
        SET list_of_characters TO the characters of s sorted BY for each character x the value obtained by GETTING FROM dictionary d the value at character x IF IT EXISTS OTHERWISE zero  
        SET result_string TO the concatenation of characters in list_of_characters in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
