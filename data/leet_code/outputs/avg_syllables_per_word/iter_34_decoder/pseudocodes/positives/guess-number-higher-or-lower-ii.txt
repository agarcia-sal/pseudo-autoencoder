CLASS Solution  
    FUNCTION getMoneyAmount(n)  
        FUNCTION dp(left, right)  
            IF left GREATER THAN OR EQUAL TO right THEN  
                RETURN zero  
            END IF  
            SET min_cost TO positive infinity  
            FOR each pivot FROM left TO right  
                SET left_cost TO dp(left, pivot MINUS one)  
                SET right_cost TO dp(pivot PLUS one, right)  
                SET max_cost TO left_cost IF left_cost GREATER THAN OR EQUAL TO right_cost OTHERWISE right_cost  
                SET cost TO pivot PLUS max_cost  
                SET min_cost TO cost IF cost LESS THAN min_cost OTHERWISE min_cost  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN dp(one, n)  
    END FUNCTION  
END CLASS
