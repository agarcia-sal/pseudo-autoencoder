CLASS Solution  
    FUNCTION minInsertions with parameter s  
        FUNCTION longest_palindromic_subsequence with parameter s  
            SET n TO the LENGTH OF s  
            CREATE dp AS two dimensional array of size n by n FILLED WITH zero  
            FOR i FROM zero TO n MINUS one  
                SET element at position i of element at position i of dp TO one  
            END FOR  
            FOR cl FROM two TO n  
                FOR i FROM zero TO n MINUS cl  
                    SET j TO i PLUS cl MINUS one  
                    IF element at position i of s EQUALS element at position j of s AND cl EQUALS two THEN  
                        SET element at position i of element at position j of dp TO two  
                    ELSE IF element at position i of s EQUALS element at position j of s THEN  
                        SET element at position i of element at position j of dp TO element at position i PLUS one of element at position j MINUS one of dp PLUS two  
                    ELSE  
                        SET element at position i of element at position j of dp TO the GREATER OF element at position i of element at position j MINUS one of dp AND element at position i PLUS one of element at position j of dp  
                    END IF  
                END FOR  
            END FOR  
            RETURN element at position zero of element at position n MINUS one of dp  
        END FUNCTION  
        SET result TO the LENGTH OF s MINUS longest_palindromic_subsequence with argument s  
        RETURN result  
    END FUNCTION  
END CLASS
