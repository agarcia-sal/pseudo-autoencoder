CLASS Solution  
    FUNCTION longestPalindromeSubseq(s)  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of size n BY n with all values SET TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AT element at position i of dp TO one  
        END FOR  
        
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of s EQUALS element at position j of s  
                    SET element at position i AT element at position j of dp TO element at position i PLUS one AT element at position j MINUS one of dp PLUS two  
                ELSE  
                    IF element at position i PLUS one AT element at position j of dp GREATER THAN OR EQUAL TO element at position i AT element at position j MINUS one of dp  
                        SET element at position i AT element at position j of dp TO element at position i PLUS one AT element at position j of dp  
                    ELSE  
                        SET element at position i AT element at position j of dp TO element at position i AT element at position j MINUS one of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero AT element at position n MINUS one of dp  
    END FUNCTION  
END CLASS
