CLASS Solution  
    FUNCTION checkInclusion(s1 parameter, s2 parameter)  
        SET length of s1 TO length of s1  
        SET length of s2 TO length of s2  
        
        IF length of s1 GREATER THAN length of s2 THEN  
            RETURN false  
        END IF  
        
        SET count of characters in s1 TO the frequency count of each character in s1  
        SET count of characters in the first window of s2 TO the frequency count of each character in the substring from position one TO position length of s1 of s2  
        
        IF count of characters in s1 EQUALS count of characters in the first window of s2 THEN  
            RETURN true  
        END IF  
        
        FOR variable i FROM length of s1 PLUS one TO length of s2 DO  
            INCREMENT the count of the character at position i of s2 BY one in the count of characters in the current window  
            DECREMENT the count of the character at position i MINUS length of s1 of s2 BY one in the count of characters in the current window  
            
            IF the count of the character at position i MINUS length of s1 of s2 IN the count of characters in the current window EQUALS zero THEN  
                REMOVE the character at position i MINUS length of s1 of s2 FROM the count of characters in the current window  
            END IF  
            
            IF count of characters in s1 EQUALS count of characters in the current window THEN  
                RETURN true  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
