CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a LIST of size n FILLED WITH zeroes  

        SET force TO zero  
        FOR i FROM zero TO n MINUS one  
            IF element at position i of dominoes EQUALS uppercase letter R  
                SET force TO n  
            ELSE IF element at position i of dominoes EQUALS uppercase letter L  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER value BETWEEN force MINUS one AND zero  
            END IF  
            INCREMENT element at position i of forces BY force  
        END FOR  

        SET force TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            IF element at position i of dominoes EQUALS uppercase letter L  
                SET force TO n  
            ELSE IF element at position i of dominoes EQUALS uppercase letter R  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER value BETWEEN force MINUS one AND zero  
            END IF  
            DECREMENT element at position i of forces BY force  
        END FOR  

        SET result TO an EMPTY LIST  
        FOR each f IN forces  
            IF f GREATER THAN zero  
                APPEND uppercase letter R TO result  
            ELSE IF f LESS THAN zero  
                APPEND uppercase letter L TO result  
            ELSE  
                APPEND period character TO result  
            END IF  
        END FOR  

        RETURN the CONCATENATION of all elements in result AS a STRING  
    END FUNCTION  
END CLASS
