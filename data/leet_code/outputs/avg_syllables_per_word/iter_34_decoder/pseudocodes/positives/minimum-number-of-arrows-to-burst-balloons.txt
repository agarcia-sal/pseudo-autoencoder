CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        SORT points BASED ON the end coordinate of each balloon  
        SET arrows TO zero  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF the starting coordinate of the current balloon GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO the ending coordinate of the current balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
