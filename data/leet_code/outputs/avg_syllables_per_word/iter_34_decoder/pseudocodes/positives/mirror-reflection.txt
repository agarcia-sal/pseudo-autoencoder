CLASS Solution  
    FUNCTION mirrorReflection(p INTEGER PARAMETER q INTEGER PARAMETER) RETURNS INTEGER  
        DEFINE FUNCTION gcd(a INTEGER PARAMETER b INTEGER PARAMETER) RETURNS INTEGER  
            WHILE b NOT EQUALS zero  
                SET temporary_variable TO b  
                SET b TO a MODULO b  
                SET a TO temporary_variable  
            END WHILE  
            RETURN a  
        END FUNCTION  

        SET lcm TO p MULTIPLIED BY q DIVIDED BY the RESULT OF gcd WITH PARAMETERS p AND q  
        SET vertical_reflections TO lcm DIVIDED BY p  
        SET horizontal_reflections TO lcm DIVIDED BY q  

        IF vertical_reflections MODULO two EQUALS zero  
            RETURN zero  
        ELSE IF horizontal_reflections MODULO two NOT EQUALS zero  
            RETURN one  
        ELSE  
            RETURN two  
        END IF  
    END FUNCTION  
END CLASS
