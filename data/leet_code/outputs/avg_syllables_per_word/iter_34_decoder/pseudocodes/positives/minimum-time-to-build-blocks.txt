CLASS Solution  
    FUNCTION minBuildTime WITH parameters blocks AND split  
        CALL heapify TO transform blocks INTO a heap structure  
        WHILE the LENGTH OF blocks IS GREATER THAN one  
            SET first TO the element removed from blocks with the smallest value  
            SET second TO the element removed from blocks with the smallest value  
            SET combined_time TO split PLUS second  
            INSERT combined_time INTO blocks maintaining heap property  
        END WHILE  
        RETURN the element at position zero of blocks  
    END FUNCTION  
END CLASS
