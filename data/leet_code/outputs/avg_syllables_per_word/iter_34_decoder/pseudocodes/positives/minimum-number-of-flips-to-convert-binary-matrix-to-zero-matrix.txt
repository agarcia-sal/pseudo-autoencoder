CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix x y)  
            SET directions TO list of pairs of minus one and zero PLUS one and zero PLUS zero and minus one PLUS zero and one PLUS zero and zero  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n  
                    SET element at position nx of matrix at position ny TO one MINUS element at position nx of matrix at position ny  
                END IF  
            END FOR  
        END FUNCTION  

        FUNCTION matrix_to_tuple(matrix)  
            SET result TO empty tuple  
            FOR each row IN matrix  
                APPEND tuple of row TO result  
            END FOR  
            RETURN result  
        END FUNCTION  

        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET target TO tuple of tuples of zero repeated n times repeated m times  
        SET initial_state TO matrix_to_tuple applied to mat  
        SET queue TO list containing pair initial_state and zero  
        SET visited TO empty set  
        ADD initial_state TO visited  

        WHILE the LENGTH OF queue GREATER THAN zero  
            SET current_state TO element at position zero of element at position zero of queue  
            SET steps TO element at position one of element at position zero of queue  
            REMOVE element at position zero FROM queue  
            IF current_state EQUALS target  
                RETURN steps  
            END IF  

            SET current_matrix TO list of lists of each row in current_state  

            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO deep copy of current_matrix  
                    CALL flip with next_matrix i j  
                    SET next_state TO matrix_to_tuple applied to next_matrix  
                    IF next_state NOT IN visited  
                        ADD next_state TO visited  
                        APPEND pair next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN minus one  
    END FUNCTION  
END CLASS
