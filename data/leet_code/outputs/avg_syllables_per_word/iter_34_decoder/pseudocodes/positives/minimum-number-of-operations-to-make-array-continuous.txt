CLASS Solution  
    FUNCTION minOperations(nums)  
        SET n TO the LENGTH OF nums  
        SET unique_nums TO the LIST OF ELEMENTS IN nums WITH DUPLICATES REMOVED AND SORTED IN ASCENDING ORDER  
        SET min_operations TO n  
        SET j TO zero  
        FOR i FROM zero TO the LENGTH OF unique_nums MINUS one  
            WHILE j IS LESS THAN the LENGTH OF unique_nums AND the ELEMENT AT POSITION j OF unique_nums IS LESS THAN the ELEMENT AT POSITION i OF unique_nums PLUS n  
                INCREMENT j BY one  
            END WHILE  
            SET min_operations TO THE LESSER OF min_operations AND n MINUS THE DIFFERENCE BETWEEN j AND i  
        END FOR  
        RETURN min_operations  
    END FUNCTION  
END CLASS
