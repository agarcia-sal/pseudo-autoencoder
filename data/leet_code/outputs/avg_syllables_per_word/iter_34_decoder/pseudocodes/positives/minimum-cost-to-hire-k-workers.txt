CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO list of pairs of wage element DIVIDED BY quality element AND quality element FOR each element FROM wage and quality collections combined in corresponding order  
        SORT workers IN ascending order ACCORDING TO the first element of each pair  
        SET max_heap TO empty collection  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each pair ratio quality_value IN workers  
            PUSH negative quality_value TO max_heap  
            INCREMENT total_quality BY quality_value  
            IF the LENGTH OF max_heap GREATER THAN k THEN  
                REMOVE element FROM max_heap WHICH IS the smallest element  
                INCREMENT total_quality BY the removed element  
            END IF  
            IF the LENGTH OF max_heap EQUALS k THEN  
                SET current_cost TO total_quality MULTIPLIED BY ratio  
                IF current_cost LESS THAN min_cost THEN  
                    SET min_cost TO current_cost  
                END IF  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
