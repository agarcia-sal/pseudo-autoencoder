CLASS Solution  
    FUNCTION champagneTower(poured, query_row, query_glass)  
        SET current_row TO a list containing poured as the only element  
        FOR row FROM one TO query_row  
            SET next_row TO a list of zeros with length equal to row PLUS one  
            FOR i FROM zero TO the LENGTH OF current_row MINUS one  
                IF element at position i of current_row GREATER THAN one  
                    SET excess TO element at position i of current_row MINUS one DIVIDED BY two  
                    INCREMENT element at position i of next_row BY excess  
                    INCREMENT element at position i PLUS one of next_row BY excess  
                END IF  
            END FOR  
            ASSIGN reference TO object next_row INTO current_row  
        END FOR  
        IF element at position query_glass of current_row LESS THAN OR EQUAL TO one  
            RETURN element at position query_glass of current_row  
        ELSE  
            RETURN one  
        END IF  
    END FUNCTION  
END CLASS
