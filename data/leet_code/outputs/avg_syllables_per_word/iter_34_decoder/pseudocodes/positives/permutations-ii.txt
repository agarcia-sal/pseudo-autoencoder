CLASS Solution  
    FUNCTION permuteUnique(nums)  
        FUNCTION backtrack(path, used, res)  
            IF the LENGTH OF path EQUALS the LENGTH OF nums  
                APPEND a COPY OF path TO res  
                RETURN  
            END IF  
            FOR index i FROM zero TO the LENGTH OF nums MINUS one  
                IF the element at position i of used IS true OR (i GREATER THAN zero AND the element at position i of nums EQUALS the element at position i MINUS one of nums AND the element at position i MINUS one of used IS false)  
                    CONTINUE to the next iteration  
                END IF  
                SET the element at position i of used TO true  
                APPEND the element at position i of nums TO path  
                CALL backtrack WITH arguments path used res  
                REMOVE the last element FROM path  
                SET the element at position i of used TO false  
            END FOR  
        END FUNCTION  
        SORT nums IN ascending order  
        SET res TO empty list  
        SET used TO a list of false values with the LENGTH OF nums  
        CALL backtrack WITH arguments empty list used res  
        RETURN res  
    END FUNCTION  
END CLASS
