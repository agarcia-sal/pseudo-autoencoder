CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF the element at position i of board AND the element at position j of element at position i of board NOT EQUALS the element at position i of element at position zero of board AND the element at position j of element at position zero of board NOT EQUALS the element at position j of element at position zero of board AND the element at position zero of element at position zero of board  
                    RETURN minus one  
                END IF  
            END FOR  
            
            INCREMENT rowSum BY the element at position zero of element at position i of board  
            INCREMENT colSum BY the element at position i of element at position zero of board  
            
            IF the element at position zero of element at position i of board EQUALS the remainder of i DIVIDED BY two  
                INCREMENT rowSwap BY one  
            END IF  
            
            IF the element at position i of element at position zero of board EQUALS the remainder of i DIVIDED BY two  
                INCREMENT colSwap BY one  
            END IF  
        END FOR  
        
        IF the integer division of N BY two GREATER THAN rowSum OR rowSum GREATER THAN the integer division of N PLUS one BY two  
            RETURN minus one  
        END IF  
        
        IF the integer division of N BY two GREATER THAN colSum OR colSum GREATER THAN the integer division of N PLUS one BY two  
            RETURN minus one  
        END IF  
        
        IF the remainder of N DIVIDED BY two NOT EQUALS zero  
            IF the remainder of rowSwap DIVIDED BY two NOT EQUALS zero  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            
            IF the remainder of colSwap DIVIDED BY two NOT EQUALS zero  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the MINIMUM OF rowSwap AND N MINUS rowSwap  
            SET colSwap TO the MINIMUM OF colSwap AND N MINUS colSwap  
        END IF  
        
        RETURN the integer division of the SUM OF rowSwap AND colSwap BY two  
    END FUNCTION  
END CLASS
