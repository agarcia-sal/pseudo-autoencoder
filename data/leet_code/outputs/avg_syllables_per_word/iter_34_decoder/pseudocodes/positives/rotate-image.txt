CLASS Solution  
    FUNCTION rotate(matrix)  
        SET n TO the LENGTH OF matrix  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET temporary_value TO the element at position j OF the element at position i OF matrix  
                SET the element at position j OF the element at position i OF matrix TO the element at position i OF the element at position j OF matrix  
                SET the element at position i OF the element at position j OF matrix TO temporary_value  
            END FOR  
        END FOR  
        FOR i FROM zero TO n MINUS one  
            CALL reverse_row ON the element at position i OF matrix  
        END FOR  
    END FUNCTION  
  
    FUNCTION reverse_row(row)  
        SET start_index TO zero  
        SET end_index TO the LENGTH OF row MINUS one  
        WHILE start_index LESS THAN end_index  
            SET temporary_value TO the element at position start_index OF row  
            SET the element at position start_index OF row TO the element at position end_index OF row  
            SET the element at position end_index OF row TO temporary_value  
            INCREMENT start_index BY one  
            DECREMENT end_index BY one  
        END WHILE  
    END FUNCTION  
END CLASS
