CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO the list of each character obtained by converting n to a string  
        SET length TO the number of elements in digits  
        SET i TO length MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND element at position i of digits GREATER THAN OR EQUAL TO element at position i PLUS one of digits  
            DECREMENT i BY one  
        END WHILE  
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
        SET j TO length MINUS one  
        WHILE element at position j of digits LESS THAN OR EQUAL TO element at position i of digits  
            DECREMENT j BY one  
        END WHILE  
        SET temporary TO element at position i of digits  
        SET element at position i of digits TO element at position j of digits  
        SET element at position j of digits TO temporary  
        SET digits FROM position i PLUS one TO the end TO the reverse of the elements from position i PLUS one TO the end of digits  
        SET result TO the integer obtained by concatenating all elements of digits in order  
        IF result GREATER THAN two TO THE POWER OF thirty ONE MINUS one  
            RETURN negative one  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
