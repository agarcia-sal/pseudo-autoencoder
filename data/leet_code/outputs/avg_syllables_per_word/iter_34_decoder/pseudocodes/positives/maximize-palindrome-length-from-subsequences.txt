CLASS Solution  
    FUNCTION longestPalindrome(word1, word2)  
        SET s TO the concatenation of word1 AND word2  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of size n BY n FILLED WITH zeros  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AND i OF dp TO one  
        END FOR  
        
        SET max_len TO zero  
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i OF s EQUALS element at position j OF s  
                    SET element at position i AND j OF dp TO element at position i PLUS one AND j MINUS one OF dp PLUS two  
                    IF i LESS THAN the LENGTH OF word1 AND j GREATER THAN OR EQUAL TO the LENGTH OF word1  
                        SET max_len TO the GREATER OF max_len AND element at position i AND j OF dp  
                    END IF  
                ELSE  
                    SET element at position i AND j OF dp TO the GREATER OF element at position i PLUS one AND j OF dp AND element at position i AND j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_len  
    END FUNCTION  
END CLASS
