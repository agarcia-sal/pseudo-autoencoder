CLASS Solution  
    FUNCTION merge(intervals)  
        SORT intervals BY the element at position zero of each interval  
        SET merged TO empty list  
        FOR each interval IN intervals  
            IF merged EQUALS empty list OR the element at position one of the last element in merged IS LESS THAN the element at position zero of interval  
                APPEND interval TO merged  
            ELSE  
                SET the element at position one of the last element in merged TO the GREATER THAN OR EQUAL TO of the element at position one of the last element in merged AND the element at position one of interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
