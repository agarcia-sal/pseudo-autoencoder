CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid EQUALS empty OR the element at position zero of grid EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero of grid  
        
        CREATE two dimensional list dp WITH zero VALUES WITH m ROWS AND n COLUMNS  
        
        SET the element at position zero zero of dp TO the element at position zero zero of grid  
        
        FOR j FROM one TO n MINUS one  
            SET the element at position zero j of dp TO the element at position zero j MINUS one of dp PLUS the element at position zero j of grid  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            SET the element at position i zero of dp TO the element at position i MINUS one zero of dp PLUS the element at position i zero of grid  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET the element at position i j of dp TO the minimum BETWEEN the element at position i MINUS one j of dp AND the element at position i j MINUS one of dp PLUS the element at position i j of grid  
            END FOR  
        END FOR  
        
        RETURN the element at position m MINUS one n MINUS one of dp  
    END FUNCTION  
END CLASS
