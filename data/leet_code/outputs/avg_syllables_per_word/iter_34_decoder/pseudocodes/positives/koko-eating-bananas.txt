CLASS Solution  
    FUNCTION minEatingSpeed WITH PARAMETERS piles AND h  
        FUNCTION canFinish WITH PARAMETER k  
            SET hours TO zero  
            FOR each pile IN piles  
                INCREMENT hours BY the smallest integer greater than or equal to the result of pile DIVIDED BY k  
            END FOR  
            RETURN whether hours is LESS THAN OR EQUAL TO h  
        END FUNCTION  

        SET left TO one  
        SET right TO the greatest element in piles  

        WHILE left is LESS THAN OR EQUAL TO right  
            SET mid TO the integer result of left PLUS right DIVIDED BY two  
            IF canFinish WITH ARGUMENT mid RETURNS true  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  

        RETURN left  
    END FUNCTION  
END CLASS
