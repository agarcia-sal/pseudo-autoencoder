CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuples where each tuple is each row in board  
        SET target TO tuple of tuples representing one two three and four five zero  
        
        SET neighbors TO dictionary mapping zero position zero TO positions one and three  
            position one TO positions zero two and four  
            position two TO positions one and five  
            position three TO positions zero and four  
            position four TO positions one three and five  
            position five TO positions two and four  
        
        FOR variable i FROM zero TO one  
            FOR variable j FROM zero TO two  
                IF element at position j of element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        
        SET queue TO double ended queue containing tuple of start zero_pos and zero  
        SET visited TO set containing start  
        
        WHILE queue is not empty  
            REMOVE and SET state zero_pos moves TO the leftmost element of queue  
            
            IF state EQUALS target  
                RETURN moves  
            END IF  
            
            FOR each neighbor IN elements mapped by zero_pos in neighbors  
                SET new_state TO list of lists where each element is list version of each element in state  
                SET i TO zero_pos DIVIDED BY three  
                SET j TO zero_pos MODULO three  
                SET ni TO neighbor DIVIDED BY three  
                SET nj TO neighbor MODULO three  
                
                SET temporary TO element at position j of element at position i of new_state  
                SET element at position j of element at position i of new_state TO element at position nj of element at position ni of new_state  
                SET element at position nj of element at position ni of new_state TO temporary  
                
                SET new_tuple_state TO tuple of tuples where each tuple is each row in new_state  
                
                IF new_tuple_state is not an element of visited  
                    ADD new_tuple_state TO visited  
                    APPEND tuple of new_tuple_state neighbor moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the negative of one  
    END FUNCTION  
END CLASS
