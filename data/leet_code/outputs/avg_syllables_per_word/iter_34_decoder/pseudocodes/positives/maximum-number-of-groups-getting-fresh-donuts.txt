CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  

        SET remainder_count TO list of zero repeated batchSize times  

        FOR each group IN groups  
            SET remainder_index TO the element at position group MINUS (the MULTIPLIED BY floor division of group by batchSize) MULTIPLIED BY batchSize  
            INCREMENT element at position remainder_index of remainder_count BY one  
        END FOR  

        FUNCTION dp(remainder, variable number of remainder_count_elements)  
            SET remainder_count TO the list of remainder_count_elements  
            SET max_happy TO zero  

            FOR index FROM zero TO batchSize MINUS one  
                IF element at position index of remainder_count EQUALS zero  
                    CONTINUE to next iteration  
                END IF  

                DECREMENT element at position index of remainder_count BY one  
                SET new_remainder TO remainder MINUS index PLUS batchSize MULTIPLIED BY two (to ensure positive) MINUS ( (remainder MINUS index) DIVIDED BY batchSize MULTIPLIED BY batchSize )  

                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS dp(new_remainder, all elements of remainder_count)  
                ELSE  
                    SET happy_groups TO dp(new_remainder, all elements of remainder_count)  
                END IF  

                IF happy_groups GREATER THAN max_happy  
                    SET max_happy TO happy_groups  
                END IF  

                INCREMENT element at position index of remainder_count BY one  
            END FOR  

            RETURN max_happy  
        END FUNCTION  

        RETURN dp(zero, all elements of remainder_count)  
    END FUNCTION  
END CLASS
