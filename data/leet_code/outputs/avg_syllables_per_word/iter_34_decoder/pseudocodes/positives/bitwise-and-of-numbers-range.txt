CLASS Solution  
    FUNCTION rangeBitwiseAnd(left PARAMETER, right PARAMETER)  
        SET shift TO zero  
        WHILE left LESS THAN right  
            SET left TO the integer division of left BY two  
            SET right TO the integer division of right BY two  
            INCREMENT shift BY one  
        END WHILE  
        SET result TO the multiplication of left BY two RAISED TO THE power of shift  
        RETURN result  
    END FUNCTION  
END CLASS
