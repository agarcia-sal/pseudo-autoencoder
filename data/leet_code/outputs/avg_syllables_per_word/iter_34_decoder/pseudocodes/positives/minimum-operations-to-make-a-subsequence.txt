CLASS Solution  
    FUNCTION minOperations(target, arr)  
        SET index_map TO an empty collection mapping values to indices  
        FOR each index FROM the first position TO the last position IN target  
            SET value TO element at position index of target  
            SET index_map at key value TO index  
        END FOR  
  
        SET transformed_arr TO an empty list  
        FOR each value IN arr  
            IF value IS IN index_map  
                APPEND index_map at key value TO transformed_arr  
            END IF  
        END FOR  
  
        SET lis TO an empty list  
        FOR each num IN transformed_arr  
            SET pos TO the position to insert num into lis keeping lis sorted in ascending order using binary search  
            IF pos EQUALS the LENGTH OF lis  
                APPEND num TO lis  
            ELSE  
                SET element at position pos of lis TO num  
            END IF  
        END FOR  
  
        SET lcs_length TO the LENGTH OF lis  
        RETURN the LENGTH OF target MINUS lcs_length  
    END FUNCTION  
END CLASS
