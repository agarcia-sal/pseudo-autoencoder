CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a new list containing the element negative one  
        SET max_length TO zero  
        FOR each index i AND each character char IN the enumeration of s  
            IF character char EQUALS the opening parenthesis character  
                APPEND index i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF the stack is empty  
                    APPEND index i TO stack  
                ELSE  
                    SET current_length TO index i MINUS the last element of stack  
                    IF current_length GREATER THAN max_length  
                        SET max_length TO current_length  
                    END IF  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
