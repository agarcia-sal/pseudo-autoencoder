CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET number_of_rows_in_mat1 TO the LENGTH OF mat1  
        SET number_of_columns_in_mat1 TO the LENGTH OF element at position zero of mat1  
        SET number_of_rows_in_mat2 TO the LENGTH OF mat2  
        SET number_of_columns_in_mat2 TO the LENGTH OF element at position zero of mat2  
        SET result TO an empty list  
        FOR index_i FROM zero TO number_of_rows_in_mat1 MINUS one  
            SET row TO an empty list  
            FOR index_j FROM zero TO number_of_columns_in_mat2 MINUS one  
                APPEND zero TO row  
            END FOR  
            APPEND row TO result  
        END FOR  
        FOR index_i FROM zero TO number_of_rows_in_mat1 MINUS one  
            FOR index_j FROM zero TO number_of_columns_in_mat2 MINUS one  
                FOR index_p FROM zero TO number_of_columns_in_mat1 MINUS one  
                    SET element_at_position_index_i_of_result_at_position_index_j TO element at position index_i of result at position index_j PLUS element at position index_i of mat1 at position index_p MULTIPLIED BY element at position index_p of mat2 at position index_j  
                    SET element at position index_i of result at position index_j TO element_at_position_index_i_of_result_at_position_index_j  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
