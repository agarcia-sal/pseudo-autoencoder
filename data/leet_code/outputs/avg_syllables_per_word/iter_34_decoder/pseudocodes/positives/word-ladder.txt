CLASS Solution  
    FUNCTION ladderLength(beginWord endWord wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN zero  
        END IF  
        
        SET word_dict TO empty dictionary where each key maps to an empty list  
        SET word_length TO the LENGTH OF beginWord  
        
        FOR each word IN wordList  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_key TO concatenation of substring from position zero TO position i MINUS one of word PLUS the character symbol called star PLUS substring from position i PLUS one TO position word_length MINUS one of word  
                APPEND word TO the list in word_dict associated with intermediate_key  
            END FOR  
        END FOR  
        
        SET queue TO an empty double ended queue initialized with tuple containing beginWord and one  
        SET visited TO a set initialized with beginWord  
        
        WHILE queue is not empty  
            REMOVE and ASSIGN the first tuple from queue TO current_word and current_length  
            
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_word TO concatenation of substring from position zero TO position i MINUS one of current_word PLUS the character symbol called star PLUS substring from position i PLUS one TO position word_length MINUS one of current_word  
                
                FOR each word IN the list from word_dict associated with intermediate_word  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND tuple containing word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN zero  
    END FUNCTION  
END CLASS
