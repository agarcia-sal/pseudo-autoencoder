CLASS Solution  
    FUNCTION numRabbits WITH parameter answers  
        SET answer_counts TO the count of occurrences of each element IN answers  
        SET total_rabbits TO zero  
        FOR each answer AND count pair IN the items of answer_counts  
            SET group_size TO answer PLUS one  
            SET groups TO the result of dividing count PLUS answer BY group_size  
            INCREMENT total_rabbits BY groups MULTIPLIED BY group_size  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
