CLASS Solution  
    FUNCTION exist(board, word)  
        IF board EQUALS an empty list  
            RETURN false  
        END IF  
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        CREATE visited AS a list containing m elements EACH of which is a list containing n elements EACH of which is false  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS the LENGTH OF word  
                RETURN true  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of visited WITHIN which element at position y IS true OR element at position x of board WITHIN which element at position y NOT EQUALS element at position index of word  
                RETURN false  
            END IF  
            SET element at position x of visited WITHIN which element at position y TO true  
            FOR each pair dx dy IN the list containing pairs of minus one zero and one zero and zero minus one and zero one  
                IF dfs(x PLUS dx, y PLUS dy, index PLUS one)  
                    RETURN true  
                END IF  
            END FOR  
            SET element at position x of visited WITHIN which element at position y TO false  
            RETURN false  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs(i, j, zero)  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
