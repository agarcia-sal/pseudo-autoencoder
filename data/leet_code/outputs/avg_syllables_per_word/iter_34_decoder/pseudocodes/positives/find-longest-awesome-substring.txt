CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO mapping from zero TO negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        FOR each position i AND character char IN s  
            SET digit TO integer value of character char  
            SET prefix_xor TO prefix_xor EXCLUSIVE OR bit value one LEFT SHIFTED BY digit  
            IF prefix_xor EXISTS as a key IN prefix_map  
                SET possible_length TO position i MINUS value of prefix_xor key IN prefix_map  
                IF possible_length GREATER THAN max_length  
                    SET max_length TO possible_length  
                END IF  
            END IF  
            FOR each bit_index j FROM zero TO nine  
                SET flipped TO prefix_xor EXCLUSIVE OR bit value one LEFT SHIFTED BY bit_index j  
                IF flipped EXISTS as a key IN prefix_map  
                    SET possible_length TO position i MINUS value of flipped key IN prefix_map  
                    IF possible_length GREATER THAN max_length  
                        SET max_length TO possible_length  
                    END IF  
                END IF  
            END FOR  
            IF prefix_xor DOES NOT EXIST as a key IN prefix_map  
                SET value of prefix_xor key IN prefix_map TO position i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
