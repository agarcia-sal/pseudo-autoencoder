CLASS Solution  
    FUNCTION makesquare(matchsticks)  
        SET total_length TO the sum of all elements in matchsticks  
        IF the remainder of total_length DIVIDED BY four NOT EQUALS zero THEN  
            RETURN false  
        END IF  
        SET side_length TO total_length DIVIDED BY four  
        SORT matchsticks in descending order  
        
        FUNCTION can_form_sides(sides, index)  
            IF index EQUALS the LENGTH OF matchsticks THEN  
                FOR each side IN sides  
                    IF side NOT EQUALS side_length THEN  
                        RETURN false  
                    END IF  
                END FOR  
                RETURN true  
            END IF  
            
            FOR i FROM zero TO three  
                IF the element at position i of sides PLUS the element at position index of matchsticks LESS THAN OR EQUAL TO side_length THEN  
                    INCREMENT the element at position i of sides BY the element at position index of matchsticks  
                    IF can_form_sides(sides, index PLUS one) THEN  
                        RETURN true  
                    END IF  
                    DECREMENT the element at position i of sides BY the element at position index of matchsticks  
                END IF  
                IF the element at position i of sides EQUALS zero THEN  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN can_form_sides(a list of four zeros, zero)  
    END FUNCTION  
END CLASS
