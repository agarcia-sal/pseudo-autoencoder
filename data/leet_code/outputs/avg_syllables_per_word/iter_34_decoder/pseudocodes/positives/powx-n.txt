CLASS Solution  
    FUNCTION myPow(x, n)  
        FUNCTION qpow(a, n)  
            SET ans TO one  
            WHILE n GREATER THAN zero  
                IF the bitwise AND of n AND one EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO the integer division of n BY two discarding remainder  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n GREATER THAN OR EQUAL TO zero  
            RETURN qpow with parameters x and n  
        ELSE  
            RETURN one DIVIDED BY qpow with parameters x and the negation of n  
        END IF  
    END FUNCTION  
END CLASS
