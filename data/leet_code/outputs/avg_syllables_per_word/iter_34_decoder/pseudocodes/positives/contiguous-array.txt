CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary with key zero and value negative one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each index i AND element num IN nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF prefix_sum_index CONTAINS key count  
                SET max_length TO the GREATER OF max_length AND index i MINUS element at key count of prefix_sum_index  
            ELSE  
                SET element at key count of prefix_sum_index TO index i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
