CLASS Solution  
    FUNCTION findLongestChain(pairs)  
        SORT pairs IN ORDER OF the second element of each pair FROM LOWEST TO HIGHEST  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each pair IN pairs  
            SET start TO the first element of pair  
            SET end TO the second element of pair  
            IF start IS GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
