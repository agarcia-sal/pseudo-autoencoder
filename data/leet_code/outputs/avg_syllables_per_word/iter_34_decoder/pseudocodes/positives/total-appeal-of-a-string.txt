CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO a list of length twenty six with each element set to minus one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each position i AND character char IN the enumeration of s  
            SET index TO the numeric code of character char MINUS the numeric code of character a  
            INCREMENT current_appeal BY position i MINUS element at position index of last  
            INCREMENT total_appeal BY current_appeal  
            SET element at position index of last TO position i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
