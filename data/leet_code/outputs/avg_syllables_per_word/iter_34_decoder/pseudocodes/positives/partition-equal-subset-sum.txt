CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total sum TO the sum of all elements in nums  
        IF the remainder of total sum DIVIDED BY two NOT EQUALS zero  
            RETURN false  
        END IF  
        SET target TO total sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of boolean false repeated for each number FROM zero TO target INCLUSIVE  
        SET element at position zero of dp TO true  
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET element at position i of dp TO element at position i of dp OR element at position i MINUS num of dp  
            END FOR  
        END FOR  
        RETURN element at position target of dp  
    END FUNCTION  
END CLASS
