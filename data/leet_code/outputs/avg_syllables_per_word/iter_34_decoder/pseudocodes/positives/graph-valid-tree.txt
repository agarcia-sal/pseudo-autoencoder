CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF the LENGTH OF edges NOT EQUALS the VALUE OF n MINUS one THEN  
            RETURN false  
        END IF  
        
        SET adj_list TO a MAPPING FROM each integer FROM zero TO n MINUS one TO an EMPTY LIST  
        FOR each pair a AND b IN edges  
            APPEND b TO the LIST at POSITION a OF adj_list  
            APPEND a TO the LIST at POSITION b OF adj_list  
        END FOR  
        
        SET visited TO an EMPTY SET  
        
        FUNCTION dfs(node)  
            IF node IS AN ELEMENT OF visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the LIST at POSITION node OF adj_list  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        
        CALL dfs WITH zero  
        
        RETURN the LENGTH OF visited EQUALS n  
    END FUNCTION  
END CLASS
