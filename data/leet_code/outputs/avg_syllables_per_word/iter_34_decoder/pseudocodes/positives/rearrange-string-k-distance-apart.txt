CLASS Solution  
    FUNCTION rearrangeString(s string k integer) RETURNS string  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO a mapping of characters TO their frequency in s  
        
        SET max_heap TO a list of pairs formed by the negative frequency and the character FOR each character AND its count IN freq  
        ORGANIZE max_heap INTO a heap structure based on the first element of each pair  
        
        SET wait_queue TO an empty list  
        
        SET result TO an empty list  
        
        WHILE max_heap is not empty  
            REMOVE the pair with the smallest first element from max_heap AND ASSIGN its components TO count AND char  
            APPEND char TO result  
            
            APPEND a pair consisting of count INCREMENTED BY one AND char TO wait_queue  
            
            IF the LENGTH OF wait_queue EQUALS k  
                REMOVE the first pair from wait_queue AND ASSIGN its components TO count AND char  
                IF count LESS THAN zero  
                    ADD the pair count AND char BACK into max_heap maintaining heap property  
                END IF  
            END IF  
        END WHILE  
        
        IF the LENGTH OF result DOES NOT EQUAL the LENGTH OF s  
            RETURN an empty string  
        END IF  
        
        RETURN the concatenation of all characters in result  
    END FUNCTION  
END CLASS
