CLASS Solution  
    FUNCTION getXORSum(arr1 arr1 IS A LIST OF INTEGER arr2 arr2 IS A LIST OF INTEGER) RETURNS INTEGER  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        FOR each num IN arr1  
            SET xor_sum_arr1 TO the exclusive or of xor_sum_arr1 AND num  
        END FOR  
        FOR each num IN arr2  
            SET xor_sum_arr2 TO the exclusive or of xor_sum_arr2 AND num  
        END FOR  
        RETURN the bitwise and of xor_sum_arr1 AND xor_sum_arr2  
    END FUNCTION  
END CLASS
