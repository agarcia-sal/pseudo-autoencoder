CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the SET OF each element CONVERTED TO tuple IN points  
        SET min_x TO the MINIMUM OF element at position zero OF each element IN unique_points  
        SET max_x TO the MAXIMUM OF element at position zero OF each element IN unique_points  
        SET line_of_reflection TO min_x PLUS max_x DIVIDED BY two  
        FOR each element IN unique_points  
            SET x TO element at position zero OF element  
            SET y TO element at position one OF element  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the PAIR composed of reflected_x AND y NOT IN unique_points  
                RETURN False  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
