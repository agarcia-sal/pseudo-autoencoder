CLASS Solution  
    FUNCTION findRadius(houses, heaters)  
        SORT houses IN ascending order  
        SORT heaters IN ascending order  
        
        SET min_radius TO zero  
        SET j TO zero  
        
        FOR each house IN houses  
            WHILE j LESS THAN length of heaters MINUS one AND absolute value of element at position j of heaters MINUS house GREATER THAN OR EQUAL TO absolute value of element at position j PLUS one of heaters MINUS house  
                INCREMENT j BY one  
            END WHILE  
            
            SET min_radius TO maximum of min_radius AND absolute value of element at position j of heaters MINUS house  
        END FOR  
        
        RETURN min_radius  
    END FUNCTION  
END CLASS
