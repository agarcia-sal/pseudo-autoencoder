CLASS Solution  
    FUNCTION minimumMoves(grid)  
        SET n TO the LENGTH OF grid  
        SET queue TO a new empty double ended queue with the single element consisting of the values zero zero zero one zero  
        SET visited TO a new empty set containing the single tuple of values zero zero zero one  
        
        WHILE the queue is not empty  
            REMOVE and RETURN the first element of queue and ASSIGN the resulting five values to r1 c1 r2 c2 steps  
            
            IF the value of r1 EQUALS the value of n MINUS one AND the value of c1 EQUALS the value of n MINUS two AND the value of r2 EQUALS the value of n MINUS one AND the value of c2 EQUALS the value of n MINUS one THEN  
                RETURN steps  
            END IF  
            
            IF the value of c2 PLUS one LESS THAN n AND the element at position r1 of grid at position c1 PLUS one EQUALS zero AND the element at position r2 of grid at position c2 PLUS one EQUALS zero AND the tuple of values r1 c1 PLUS one r2 c2 PLUS one NOT IN visited THEN  
                ADD the tuple of values r1 c1 PLUS one r2 c2 PLUS one TO visited  
                APPEND the tuple of values r1 c1 PLUS one r2 c2 PLUS one steps PLUS one TO queue  
            END IF  
            
            IF the value of r2 PLUS one LESS THAN n AND the element at position r1 PLUS one of grid at position c1 EQUALS zero AND the element at position r2 PLUS one of grid at position c2 EQUALS zero AND the tuple of values r1 PLUS one c1 r2 PLUS one c2 NOT IN visited THEN  
                ADD the tuple of values r1 PLUS one c1 r2 PLUS one c2 TO visited  
                APPEND the tuple of values r1 PLUS one c1 r2 PLUS one c2 steps PLUS one TO queue  
            END IF  
            
            IF the value of r1 EQUALS the value of r2 AND the value of r1 PLUS one LESS THAN n AND the element at position r1 PLUS one of grid at position c1 EQUALS zero AND the element at position r1 PLUS one of grid at position c2 EQUALS zero AND the tuple of values r1 c1 r1 PLUS one c1 NOT IN visited THEN  
                ADD the tuple of values r1 c1 r1 PLUS one c1 TO visited  
                APPEND the tuple of values r1 c1 r1 PLUS one c1 steps PLUS one TO queue  
            END IF  
            
            IF the value of c1 EQUALS the value of c2 AND the value of c1 PLUS one LESS THAN n AND the element at position r1 of grid at position c1 PLUS one EQUALS zero AND the element at position r2 of grid at position c2 PLUS one EQUALS zero AND the tuple of values r1 c1 r1 c1 PLUS one NOT IN visited THEN  
                ADD the tuple of values r1 c1 r1 c1 PLUS one TO visited  
                APPEND the tuple of values r1 c1 r1 c1 PLUS one steps PLUS one TO queue  
            END IF  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
