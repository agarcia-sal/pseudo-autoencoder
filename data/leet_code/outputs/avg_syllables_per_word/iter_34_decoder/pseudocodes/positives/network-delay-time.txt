CLASS Solution  
    FUNCTION networkDelayTime(times AS List of List of Integer, n AS Integer, k AS Integer) RETURNS Integer  
        SET graph TO empty dictionary  
        FOR index FROM one TO n  
            SET element at position index of graph TO empty list  
        END FOR  
        
        FOR each triple consisting of u v w IN times  
            APPEND pair consisting of v and w TO element at position u of graph  
        END FOR  
        
        SET dist TO empty dictionary  
        FOR index FROM one TO n  
            SET element at position index of dist TO infinity  
        END FOR  
        SET element at position k of dist TO zero  
        
        SET min_heap TO list containing pair consisting of zero and k  
        
        WHILE min_heap is not empty  
            SET current_dist and node TO the pair popped from min_heap according to minimum first element  
            
            IF current_dist IS GREATER THAN element at position node of dist  
                CONTINUE to next iteration  
            END IF  
            
            FOR each pair consisting of neighbor and weight IN element at position node of graph  
                SET distance TO current_dist PLUS weight  
                
                IF distance IS LESS THAN element at position neighbor of dist  
                    SET element at position neighbor of dist TO distance  
                    PUSH pair consisting of distance and neighbor INTO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        SET max_time TO the maximum value among all values in dist  
        
        IF max_time IS NOT EQUAL TO infinity  
            RETURN max_time  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
