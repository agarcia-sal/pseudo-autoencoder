CLASS Solution  
    FUNCTION confusingNumberII(n)  
        SET rotate_map TO mapping of digit zero TO zero digit one TO one digit six TO nine digit eight TO eight digit nine TO six  
        SET valid_digits TO list of digit zero digit one digit six digit eight digit nine  
        
        FUNCTION is_confusing(num)  
            SET original_num TO num  
            SET rotated_num TO zero  
            WHILE num GREATER THAN zero holds  
                SET digit TO remainder of num DIVIDED BY ten  
                SET rotated_num TO rotated_num MULTIPLIED BY ten PLUS value at key digit of rotate_map  
                SET num TO quotient of num DIVIDED BY ten  
            END WHILE  
            RETURN whether rotated_num is NOT EQUALS TO original_num  
        END FUNCTION  
        
        FUNCTION count_confusing_numbers(limit current)  
            IF current GREATER THAN limit THEN  
                RETURN zero  
            END IF  
            SET count TO one IF current IS NOT EQUALS TO zero AND is_confusing(current) IS true OTHERWISE zero  
            FOR each digit IN valid_digits  
                SET new_number TO current MULTIPLIED BY ten PLUS digit  
                IF new_number IS EQUALS TO zero THEN  
                    CONTINUE TO next digit  
                END IF  
                INCREMENT count BY count_confusing_numbers(limit new_number)  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN count_confusing_numbers(n zero)  
    END FUNCTION  
END CLASS
