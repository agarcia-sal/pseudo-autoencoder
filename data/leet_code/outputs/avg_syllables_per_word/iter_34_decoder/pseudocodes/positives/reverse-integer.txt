CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN zero  
            SET sign TO negative one  
        ELSE  
            SET sign TO one  
        END IF  
        SET absolute_value_of_x TO the absolute value of x  
        SET string_representation TO the string form of absolute_value_of_x  
        SET reversed_string TO the characters of string_representation in reverse order  
        SET reversed_x TO the integer form of reversed_string  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative two RAISED TO THE THIRTY-FIRST POWER OR result GREATER THAN positive two RAISED TO THE THIRTY-FIRST POWER MINUS one  
            RETURN zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
