CLASS Solution  
    FUNCTION subarraySum(nums k)  
        SET cumulative_sum_freq TO a dictionary with key zero mapped to value one  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF there EXISTS a key EQUALS current_sum MINUS k IN cumulative_sum_freq  
                INCREMENT count BY the value of cumulative_sum_freq at key current_sum MINUS k  
            END IF  
            IF there EXISTS a key EQUALS current_sum IN cumulative_sum_freq  
                INCREMENT the value of cumulative_sum_freq at key current_sum BY one  
            ELSE  
                SET the value of cumulative_sum_freq at key current_sum TO one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
