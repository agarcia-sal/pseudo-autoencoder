CLASS Solution  
    FUNCTION numDupDigitsAtMostN(n)  
        FUNCTION count_unique_digits(x)  
            SET s TO the string representation of x  
            SET length TO the length of s  
            SET count TO zero  
            FOR i FROM one TO length MINUS one  
                INCREMENT count BY nine MULTIPLIED BY perm of nine MINUS one PLUS one MINUS i  
            END FOR  
            SET seen TO an empty set  
            SET i TO zero  
            WHILE i LESS THAN length  
                SET digit TO the element at position i of s converted to integer  
                SET start_value TO zero  
                IF i EQUALS zero  
                    SET start_value TO one  
                END IF  
                FOR d FROM start_value TO digit MINUS one  
                    IF d NOT IN seen  
                        INCREMENT count BY perm of nine MINUS i AND length MINUS i MINUS one  
                    END IF  
                END FOR  
                IF digit IN seen  
                    BREAK the loop  
                END IF  
                ADD digit TO seen  
                INCREMENT i BY one  
            END WHILE  
            IF i EQUALS length  
                INCREMENT count BY one  
            END IF  
            RETURN count  
        END FUNCTION  
        RETURN n MINUS count_unique_digits of n  
    END FUNCTION  

    FUNCTION perm(m, n)  
        IF n EQUALS zero  
            RETURN one  
        END IF  
        SET result TO one  
        FOR i FROM zero TO n MINUS one  
            SET result TO result MULTIPLIED BY m MINUS i  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
