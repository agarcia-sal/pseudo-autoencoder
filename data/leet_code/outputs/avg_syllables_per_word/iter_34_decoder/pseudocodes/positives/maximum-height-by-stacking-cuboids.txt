CLASS Solution  
    FUNCTION maxHeight(cuboids)  
        FOR each cuboid IN cuboids  
            SORT the dimensions of cuboid in non-decreasing order  
        END FOR  
        SORT cuboids in lexicographical order based on their dimensions  
        SET n TO the LENGTH OF cuboids  
        CREATE a list named dp with n elements all initialized to zero  
        FOR i FROM zero TO n MINUS one  
            SET element at position i of dp TO the height dimension of cuboid at position i  
            FOR j FROM zero TO i MINUS one  
                IF the first dimension of cuboid at position j LESS THAN OR EQUAL TO the first dimension of cuboid at position i AND  
                   the second dimension of cuboid at position j LESS THAN OR EQUAL TO the second dimension of cuboid at position i AND  
                   the third dimension of cuboid at position j LESS THAN OR EQUAL TO the third dimension of cuboid at position i THEN  
                    SET element at position i of dp TO the maximum of element at position i of dp AND element at position j of dp PLUS the height dimension of cuboid at position i  
                END IF  
            END FOR  
        END FOR  
        RETURN the maximum element in dp  
    END FUNCTION  
END CLASS
