CLASS Solution
    FUNCTION exclusiveTime(n parameter, logs parameter)
        SET exclusive_time TO an empty list
        FOR index FROM zero TO n MINUS one
            APPEND zero TO exclusive_time
        END FOR
        SET stack TO an empty list
        
        FOR each log IN logs
            SPLIT log INTO function_id string AND action string AND timestamp string USING delimiter colon
            SET function_id TO the integer conversion of function_id string
            SET timestamp TO the integer conversion of timestamp string
            
            IF action EQUALS start
                IF the LENGTH OF stack IS GREATER THAN zero
                    SET prev_function_id TO the function identifier AT the last element of stack
                    SET prev_timestamp TO the timestamp AT the last element of stack
                    INCREMENT exclusive_time at prev_function_id BY timestamp MINUS prev_timestamp
                END IF
                APPEND the pair consisting of function_id AND timestamp TO stack
            ELSE
                SET prev_function_id TO the function identifier AT the last element of stack
                SET prev_timestamp TO the timestamp AT the last element of stack
                REMOVE the last element FROM stack
                INCREMENT exclusive_time at prev_function_id BY timestamp MINUS prev_timestamp PLUS one
                IF the LENGTH OF stack IS GREATER THAN zero
                    SET the timestamp element of the last element of stack TO timestamp PLUS one
                END IF
            END IF
        END FOR
        
        RETURN exclusive_time
    END FUNCTION
END CLASS
