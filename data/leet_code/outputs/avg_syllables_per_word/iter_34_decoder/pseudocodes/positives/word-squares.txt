CLASS Solution  
    FUNCTION wordSquares(words)  
        SET prefix_to_words TO an empty mapping from string to list of strings  
        FOR each word IN words  
            FOR index FROM zero TO the LENGTH OF word MINUS one  
                SET prefix TO the substring of word from position zero TO position index MINUS one  
                APPEND word TO the list at prefix_to_words corresponding to prefix  
            END FOR  
        END FOR  
        
        SET results TO an empty list  
        
        FUNCTION backtrack(square)  
            IF the LENGTH OF square EQUALS the LENGTH OF the first element of words THEN  
                APPEND square TO results  
                RETURN  
            END IF  
            
            SET prefix TO an empty string  
            FOR each word_element IN square  
                APPEND the element at position the LENGTH OF square of word_element TO prefix  
            END FOR  
            
            FOR each candidate IN the list at prefix_to_words corresponding to prefix  
                CALL backtrack WITH the argument of the concatenation of square with candidate as a new list element  
            END FOR  
        END FUNCTION  
        
        FOR each word IN words  
            CALL backtrack WITH a list containing only word  
        END FOR  
        
        RETURN results  
    END FUNCTION  
END CLASS
