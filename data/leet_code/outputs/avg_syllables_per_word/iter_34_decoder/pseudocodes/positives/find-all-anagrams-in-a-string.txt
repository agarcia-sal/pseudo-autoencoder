CLASS Solution  
    FUNCTION findAnagrams WITH PARAMETERS s p  
        SET length_of_s TO the LENGTH OF s  
        SET length_of_p TO the LENGTH OF p  
        IF length_of_p GREATER THAN length_of_s  
            RETURN an EMPTY list  
        END IF  
        SET p_count TO the COUNTER OF all elements IN p  
        SET s_count TO the COUNTER OF the substring of s FROM position zero TO position length_of_p MINUS one  
        SET anagrams TO an EMPTY list  
        IF s_count EQUALS p_count  
            APPEND zero TO anagrams  
        END IF  
        FOR index FROM length_of_p TO length_of_s MINUS one  
            INCREMENT the COUNT OF element at position index of s IN s_count BY one  
            DECREMENT the COUNT OF element at position index MINUS length_of_p of s IN s_count BY one  
            IF the COUNT OF element at position index MINUS length_of_p of s IN s_count EQUALS zero  
                REMOVE element at position index MINUS length_of_p of s FROM s_count  
            END IF  
            IF s_count EQUALS p_count  
                APPEND index MINUS length_of_p PLUS one TO anagrams  
            END IF  
        END FOR  
        RETURN anagrams  
    END FUNCTION  
END CLASS
