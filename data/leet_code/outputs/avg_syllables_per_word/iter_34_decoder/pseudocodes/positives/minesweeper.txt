CLASS Solution  
    FUNCTION updateBoard(board, click)  
        FUNCTION count_adjacent_mines(row, column)  
            SET mine_count TO zero  
            FOR each index_i FROM row MINUS one TO row PLUS one  
                FOR each index_j FROM column MINUS one TO column PLUS one  
                    IF index_i IS GREATER THAN OR EQUAL TO zero AND index_i IS LESS THAN the LENGTH OF board AND index_j IS GREATER THAN OR EQUAL TO zero AND index_j IS LESS THAN the LENGTH OF element at position zero of board AND element at position index_i and index_j of board EQUALS a mine designation  
                        INCREMENT mine_count BY one  
                    END IF  
                END FOR  
            END FOR  
            RETURN mine_count  
        END FUNCTION  
      
        FUNCTION dfs(row, column)  
            IF NOT (row IS GREATER THAN OR EQUAL TO zero AND row IS LESS THAN the LENGTH OF board AND column IS GREATER THAN OR EQUAL TO zero AND column IS LESS THAN the LENGTH OF element at position zero of board)  
                RETURN  
            END IF  
            IF element at position row and column of board NOT EQUALS unexplored designation  
                RETURN  
            END IF  
          
            SET adjacent_mines TO count_adjacent_mines(row, column)  
            IF adjacent_mines IS GREATER THAN zero  
                SET element at position row and column of board TO the string representation of adjacent_mines  
            ELSE  
                SET element at position row and column of board TO blank designation  
                FOR each index_i FROM row MINUS one TO row PLUS one  
                    FOR each index_j FROM column MINUS one TO column PLUS one  
                        CALL dfs with arguments index_i and index_j  
                    END FOR  
                END FOR  
            END IF  
        END FUNCTION  
      
        SET row TO the element at position zero of click  
        SET col TO the element at position one of click  
        IF element at position row and col of board EQUALS mine designation  
            SET element at position row and col of board TO exploded designation  
            RETURN board  
        END IF  
      
        CALL dfs with arguments row and col  
        RETURN board  
    END FUNCTION  
END CLASS
