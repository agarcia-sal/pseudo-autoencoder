CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET list_of_reversed_words TO the sorted list of each word REVERSED in words  
        SET total_length TO zero  
        FOR index FROM zero TO the LENGTH OF list_of_reversed_words MINUS one  
            IF index PLUS one EQUALS the LENGTH OF list_of_reversed_words OR NOT the element at position index PLUS one of list_of_reversed_words STARTS WITH the element at position index of list_of_reversed_words  
                INCREMENT total_length BY the LENGTH OF the element at position index of list_of_reversed_words PLUS one  
            END IF  
        END FOR  
        RETURN total_length  
    END FUNCTION  
END CLASS
