CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO the smallest possible number  
        SET stack TO an empty list  
        FOR each num IN the elements of nums in reverse order  
            IF num LESS THAN second  
                RETURN True  
            END IF  
            WHILE the stack is not empty AND element at the last position of stack LESS THAN num  
                SET second TO the element removed from the last position of stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
