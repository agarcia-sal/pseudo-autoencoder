CLASS Solution  
    FUNCTION minInterval(intervals, queries)  
        CALL intervals SORT BY the first element of interval in ascending order  
        SET sorted_queries TO an empty list  
        
        FOR each element index and query_value IN queries WITH index  
            APPEND pair of query_value AND index TO sorted_queries  
        END FOR  
        CALL sorted_queries SORT BY the first element of pair in ascending order  
        
        SET min_heap TO an empty list  
        SET result TO a list of negative one with length EQUAL TO the LENGTH OF queries  
        
        SET interval_index TO zero  
        
        FOR each pair query and original_index IN sorted_queries  
            WHILE interval_index LESS THAN the LENGTH OF intervals AND  
                  element at position zero of element at position interval_index of intervals LESS THAN OR EQUAL TO query  
                SET start TO element at position zero of element at position interval_index of intervals  
                SET end TO element at position one of element at position interval_index of intervals  
                SET interval_size TO end MINUS start PLUS one  
                CALL push interval_size AND end ONTO min_heap  
                INCREMENT interval_index BY one  
            END WHILE  
            
            WHILE the LENGTH OF min_heap GREATER THAN zero AND  
                  element at position one of element at position zero of min_heap LESS THAN query  
                CALL pop FROM min_heap  
            END WHILE  
            
            IF the LENGTH OF min_heap GREATER THAN zero  
                SET element at position original_index OF result TO element at position zero of element at position zero of min_heap  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
