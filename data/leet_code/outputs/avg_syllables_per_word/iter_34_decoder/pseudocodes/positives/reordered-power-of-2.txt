CLASS Solution  
    FUNCTION reorderedPowerOf2(n)  
        SET power_of_two_sorted_digits TO empty collection of unique tuples  
        SET power_of_two TO one  
        WHILE power_of_two LESS THAN OR EQUAL TO ten RAISED TO THE POWER OF nine holds  
            SET sorted_digits TO the tuple of the digits of power_of_two arranged in ascending order  
            ADD sorted_digits TO power_of_two_sorted_digits  
            SET power_of_two TO power_of_two MULTIPLIED BY two  
        END WHILE  
        SET sorted_n_digits TO the tuple of the digits of n arranged in ascending order  
        RETURN whether sorted_n_digits is contained IN power_of_two_sorted_digits  
    END FUNCTION  
END CLASS
