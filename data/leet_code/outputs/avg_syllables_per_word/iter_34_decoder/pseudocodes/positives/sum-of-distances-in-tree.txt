CLASS Solution  
    FUNCTION sumOfDistancesInTree(n, edges)  
        CREATE a new graph as a default dictionary with lists as default values  
        FOR each pair of nodes a and b IN edges  
            APPEND node b TO the adjacency list of node a in graph  
            APPEND node a TO the adjacency list of node b in graph  
        END FOR  

        CREATE a list named count with size n and INITIALIZE every element TO one  
        CREATE a list named dist with size n and INITIALIZE every element TO zero  

        FUNCTION dfs1(u, p)  
            FOR each node v IN the adjacency list of node u in graph  
                IF node v NOT EQUALS node p THEN  
                    CALL dfs1 with parameters v and u  
                    INCREMENT count at position u BY count at position v  
                    INCREMENT dist at position u BY dist at position v PLUS count at position v  
                END IF  
            END FOR  
        END FUNCTION  

        CALL dfs1 with parameters zero and negative one  

        FUNCTION dfs2(u, p)  
            FOR each node v IN the adjacency list of node u in graph  
                IF node v NOT EQUALS node p THEN  
                    SET dist at position v TO dist at position u MINUS count at position v PLUS n MINUS count at position v  
                    CALL dfs2 with parameters v and u  
                END IF  
            END FOR  
        END FUNCTION  

        CALL dfs2 with parameters zero and negative one  

        RETURN dist  
    END FUNCTION  
END CLASS
