CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
  
        CREATE a two dimensional list dp with m PLUS one rows each containing n PLUS one zeros  
  
        FOR i FROM zero TO m  
            SET element at position zero of dp at position i TO i  
        END FOR  
  
        FOR j FROM zero TO n  
            SET element at position j of dp at position zero TO j  
        END FOR  
  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of word1 EQUALS element at position j MINUS one of word2  
                    SET element at position j of dp at position i TO element at position j MINUS one of dp at position i MINUS one  
                ELSE  
                    SET element at position j of dp at position i TO one PLUS the MINIMUM OF element at position j of dp at position i MINUS one  
                                                                 AND element at position j MINUS one of dp at position i  
                                                                 AND element at position j MINUS one of dp at position i MINUS one  
                END IF  
            END FOR  
        END FOR  
  
        RETURN element at position n of dp at position m  
    END FUNCTION  
END CLASS
