CLASS Solution  
    FUNCTION shortestDistance(maze, start, destination)  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero of maze  
        SET directions TO the LIST containing the PAIRS of values one and zero MINUS one and zero zero and one zero and MINUS one  
        SET start TO the TUPLE of start  
        SET destination TO the TUPLE of destination  
        SET visited TO an EMPTY SET  
        SET pq TO the LIST containing the PAIR of zero and start  

        WHILE pq is NOT EMPTY  
            SET dist, position TO the RESULT OF REMOVING the SMALLEST ELEMENT FROM pq  
            SET x TO the ELEMENT at position zero of position  
            SET y TO the ELEMENT at position one of position  
            
            IF x EQUALS the ELEMENT at position zero of destination AND y EQUALS the ELEMENT at position one of destination  
                RETURN dist  
            END IF  
            
            IF position IS IN visited  
                CONTINUE TO the NEXT ITERATION  
            END IF  
            
            ADD position TO visited  
            
            FOR each dx, dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the ELEMENT at position new_x PLUS dx of maze AT position new_y PLUS dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF the TUPLE of new_x and new_y IS NOT IN visited  
                    INSERT the PAIR of dist PLUS count and the TUPLE of new_x and new_y INTO pq maintaining ORDER BY the FIRST ELEMENT OF THE PAIR being the DISTANCE  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the NEGATIVE OF one  
    END FUNCTION  
END CLASS
