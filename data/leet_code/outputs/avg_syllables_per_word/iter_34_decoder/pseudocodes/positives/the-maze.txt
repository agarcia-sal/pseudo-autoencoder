CLASS Solution  
    FUNCTION hasPath maze start destination  
        SET directions TO the list containing the pairs of zero and one one and zero zero and negative one negative one and zero  
        SET m TO the length of maze  
        SET n TO the length of the element at position zero of maze  
        SET visited TO an empty set  
        
        FUNCTION dfs x y  
            IF the pair of x and y is a member of visited THEN  
                RETURN False  
            END IF  
            IF the list containing x and y EQUALS destination THEN  
                RETURN True  
            END IF  
            ADD the pair of x and y TO visited  
            
            FOR each pair dx dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx PLUS dx is greater than or EQUAL TO zero AND nx PLUS dx is LESS THAN m AND ny PLUS dy is greater than or EQUAL TO zero AND ny PLUS dy is LESS THAN n AND the element at position nx PLUS dx of maze at position ny PLUS dy EQUALS zero  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF dfs nx ny THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN dfs the element at position zero of start the element at position one of start  
    END FUNCTION  
END CLASS
