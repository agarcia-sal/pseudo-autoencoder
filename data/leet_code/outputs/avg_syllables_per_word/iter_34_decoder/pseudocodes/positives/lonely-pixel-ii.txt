CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture EQUALS nothing OR element at position zero of picture EQUALS nothing  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO list of zeros with length m  
        SET col_count TO list of zeros with length n  
        SET row_patterns TO a new mapping from string to integer with default value zero  
        
        FOR r FROM zero TO m MINUS one  
            SET pattern TO concatenation of all elements of element at position r of picture as a single string  
            INCREMENT the value in row_patterns at key pattern BY one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS the character representing a black pixel  
                    INCREMENT element at position r of row_count BY one  
                    INCREMENT element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS the character representing a black pixel  
                    SET pattern TO concatenation of all elements of element at position r of picture as a single string  
                    IF element at position r of row_count EQUALS target AND element at position c of col_count EQUALS target AND the value in row_patterns at key pattern EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
