CLASS Solution  
    FUNCTION minPatches(nums n)  
        SET missing TO the smallest positive number that cannot yet be formed from the numbers considered  
        SET patches TO the count of numbers added to enable forming all numbers up to n  
        SET i TO the current index in nums under consideration  
        
        WHILE missing LESS THAN OR EQUAL TO n  
            IF i LESS THAN the LENGTH OF nums AND element at position i of nums LESS THAN OR EQUAL TO missing  
                SET missing TO missing PLUS element at position i of nums  
                INCREMENT i BY one  
            ELSE  
                SET missing TO missing PLUS missing  
                INCREMENT patches BY one  
            END IF  
        END WHILE  
        
        RETURN patches  
    END FUNCTION  
END CLASS
