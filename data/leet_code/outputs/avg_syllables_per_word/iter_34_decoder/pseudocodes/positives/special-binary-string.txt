CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each index i AND character char IN the enumeration of s  
                IF character char EQUALS the character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND the concatenation of the character one PLUS the result of calling dfs on the substring of s from position start incremented by one TO position i PLUS the character zero TO the list specials  
                    SET start TO index i incremented by one  
                END IF  
            END FOR  
            SORT the list specials in reverse lexicographical order  
            RETURN the concatenation of all elements in specials  
        END FUNCTION  
        RETURN the result of calling dfs on s  
    END FUNCTION  
END CLASS
