CLASS Solution  
    FUNCTION kEmptySlots(bulbs PARAMETERS bulbs AS list of integers k AS integer)  
        SET bloom_days TO empty list  
        FOR day FROM one TO the LENGTH OF bulbs  
            SET position TO element at position day MINUS one of bulbs  
            SET i TO the insertion position of position in bloom_days to maintain sorted order  
            IF i GREATER THAN zero AND position MINUS element at position i MINUS one of bloom_days MINUS one EQUALS k  
                RETURN day  
            END IF  
            IF i LESS THAN the LENGTH OF bloom_days AND element at position i of bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position INTO bloom_days AT position i to maintain sorted order  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
