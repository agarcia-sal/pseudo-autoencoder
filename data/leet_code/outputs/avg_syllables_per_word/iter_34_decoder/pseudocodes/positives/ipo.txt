CLASS Solution  
    FUNCTION findMaximizedCapital(k INTEGER, w INTEGER, profits LIST OF INTEGER, capital LIST OF INTEGER) RETURNS INTEGER  
        SET projects TO the LIST of PAIRS formed by pairing each element of capital with the corresponding element of profits  
        SORT projects IN ascending order BY the first element of each PAIR representing capital  
        
        SET max_heap TO an empty LIST  
        
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        
        FOR each iteration FROM one TO k  
            WHILE i LESS THAN n AND the first element of the PAIR at position i of projects LESS THAN OR EQUAL TO w  
                ADD the NEGATIVE of the second element of the PAIR at position i of projects TO max_heap using heap insertion  
                INCREMENT i BY one  
            END WHILE  
            
            IF max_heap IS empty  
                EXIT the FOR loop  
            END IF  
            
            SET w TO w PLUS the NEGATIVE of the value obtained by removing the smallest element from max_heap using heap removal  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
