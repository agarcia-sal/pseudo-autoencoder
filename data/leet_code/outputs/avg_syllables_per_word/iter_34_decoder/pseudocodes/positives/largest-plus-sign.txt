CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO an empty set  
        FOR each mine IN mines  
            ADD the tuple consisting of the elements of mine TO mines_set  
        END FOR  
  
        SET dp TO a two dimensional list with n rows and n columns where each element is a list of four zeros  
  
        SET directions TO the list containing the pairs zero PLUS one, one PLUS zero, zero MINUS one, and minus one PLUS zero  
  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair consisting of i and j IS NOT IN mines_set THEN  
                    IF j IS GREATER THAN zero THEN  
                        SET element at position zero of element at position j MINUS one of element at position i of dp TO element at position zero of element at position j MINUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i IS GREATER THAN zero THEN  
                        SET element at position one of element at position j of element at position i MINUS one of dp TO element at position one of element at position j of element at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
  
        SET result TO zero  
  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the pair consisting of i and j IS NOT IN mines_set THEN  
                    IF j IS LESS THAN n MINUS one THEN  
                        SET element at position two of element at position j PLUS one of element at position i of dp TO element at position two of element at position j PLUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i IS LESS THAN n MINUS one THEN  
                        SET element at position three of element at position j of element at position i PLUS one of dp TO element at position three of element at position j of element at position i PLUS one of dp PLUS one  
                    END IF  
  
                    SET plus_sign_order TO the minimum value among the elements of element at position j of element at position i of dp  
  
                    IF plus_sign_order IS GREATER THAN result THEN  
                        SET result TO plus_sign_order  
                    END IF  
                END IF  
            END FOR  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
