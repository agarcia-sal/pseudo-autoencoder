CLASS Solution  
    FUNCTION minRefuelStops(target startFuel stations)  
        SET max_heap TO an empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a list containing target and zero TO stations  

        FOR each element IN stations  
            SET position TO element at position zero of element  
            SET fuel TO element at position one of element  

            WHILE the LENGTH OF max_heap is greater than zero AND current_fuel LESS THAN position  
                SET largest_available_fuel TO the element REMOVED FROM max_heap with the greatest value  
                DECREMENT current_fuel BY negative largest_available_fuel  
                INCREMENT stops BY one  
            END WHILE  

            IF current_fuel LESS THAN position  
                RETURN negative one  
            END IF  

            ADD negative fuel TO max_heap WITH heap insertion behavior  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
