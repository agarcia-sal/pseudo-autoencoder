CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF the LENGTH OF remaining EQUALS zero  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR i FROM zero TO THE LENGTH OF remaining MINUS one  
                SET new_path TO path PLUS element at position i of remaining  
                SET new_remaining TO concatenation of elements from position zero TO position i MINUS one of remaining PLUS elements from position i PLUS one TO THE END of remaining  
                CALL backtrack(new_path, new_remaining)  
            END FOR  
        END FUNCTION  

        SET permutations TO an empty list  
        CALL backtrack(empty list, nums)  
        RETURN permutations  
    END FUNCTION  
END CLASS
