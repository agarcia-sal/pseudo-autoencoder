CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO a mapping of numbers to their frequencies in nums  
        SET end_with TO a mapping of numbers to the counts of subsequences ending with that number  
        
        FOR each num IN nums  
            IF the frequency count of num in num_count EQUALS zero  
                CONTINUE to the next iteration  
            END IF  
            
            IF the count of subsequences ending with the number which is one less than num in end_with IS GREATER THAN zero  
                DECREMENT the count of subsequences ending with the number which is one less than num in end_with BY one  
                INCREMENT the count of subsequences ending with num in end_with BY one  
            ELSE IF the frequency count of the number which is one greater than num in num_count IS GREATER THAN zero AND the frequency count of the number which is two greater than num in num_count IS GREATER THAN zero  
                DECREMENT the frequency count of the number which is one greater than num in num_count BY one  
                DECREMENT the frequency count of the number which is two greater than num in num_count BY one  
                INCREMENT the count of subsequences ending with the number which is two greater than num in end_with BY one  
            ELSE  
                RETURN false  
            END IF  
            
            DECREMENT the frequency count of num in num_count BY one  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
