CLASS Solution  
    FUNCTION maxSubArrayLen(nums k)  
        SET cumulative sum index TO a dictionary with key zero assigned to value negative one  
        SET cumulative sum TO zero  
        SET max length TO zero  
        FOR each index i AND each number num IN nums  
            SET cumulative sum TO cumulative sum PLUS num  
            IF cumulative sum MINUS k EXISTS AS A KEY IN cumulative sum index  
                SET candidate length TO i MINUS the element at key cumulative sum MINUS k of cumulative sum index  
                IF candidate length GREATER THAN max length  
                    SET max length TO candidate length  
                END IF  
            END IF  
            IF cumulative sum DOES NOT EXIST AS A KEY IN cumulative sum index  
                ASSIGN the element at key cumulative sum of cumulative sum index TO i  
            END IF  
        END FOR  
        RETURN max length  
    END FUNCTION  
END CLASS
