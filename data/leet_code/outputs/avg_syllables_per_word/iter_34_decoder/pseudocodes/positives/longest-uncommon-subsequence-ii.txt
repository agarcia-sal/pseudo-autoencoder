CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator_reference TO the iterator OVER t  
            RETURN the logical AND of for each character in s TEST character IS IN iterator_reference  
        END FUNCTION  
        
        SORT strs BY length in descending order AND lexicographically ascending order  
        
        FOR index i FROM zero TO the LENGTH OF strs MINUS one  
            SET word TO element at position i of strs  
            SET unique TO true  
            
            FOR index j FROM zero TO the LENGTH OF strs MINUS one  
                SET other TO element at position j of strs  
                IF i NOT EQUALS j AND is_subsequence of word IN other HOLDS true  
                    SET unique TO false  
                    BREAK the inner loop  
                END IF  
            END FOR  
            
            IF unique IS true  
                RETURN the LENGTH OF word  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
