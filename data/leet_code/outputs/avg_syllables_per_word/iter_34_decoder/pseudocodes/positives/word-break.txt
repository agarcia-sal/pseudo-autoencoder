CLASS Solution  
    FUNCTION wordBreak(s string, wordDict list of string) RETURNS boolean  
        SET wordSet TO set of elements in wordDict  
        SET dp TO list of boolean values of length the LENGTH OF s PLUS one  
        SET element at position zero of dp TO true  
        
        FOR index i FROM one TO the LENGTH OF s PLUS one  
            FOR index j FROM zero TO i MINUS one  
                IF element at position j of dp EQUALS true AND the substring from position j TO position i MINUS one of s IS IN wordSet  
                    SET element at position i of dp TO true  
                    BREAK inner loop  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position the LENGTH OF s of dp  
    END FUNCTION  
END CLASS
