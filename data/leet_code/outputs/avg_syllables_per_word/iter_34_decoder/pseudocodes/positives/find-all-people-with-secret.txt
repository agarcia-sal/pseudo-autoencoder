CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO a set containing zero AND firstPerson  
        SORT meetings BY the third element of each meeting representing the time  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO the third element of the meeting at position i  
            SET graph TO a default dictionary mapping each key to an empty list  
            WHILE i LESS THAN the LENGTH OF meetings AND the third element of the meeting at position i EQUALS current_time  
                SET x TO the first element of the meeting at position i  
                SET y TO the second element of the meeting at position i  
                APPEND y TO the list in graph at key x  
                APPEND x TO the list in graph at key y  
                INCREMENT i BY one  
            END WHILE  
            SET queue TO a double-ended queue initialized with each person in know_secret that is a key in graph  
            SET visited TO an empty set  
            WHILE the queue is not empty  
                REMOVE the element from the front of queue AND ASSIGN it TO person  
                IF person IS IN visited  
                    CONTINUE to the next iteration of the WHILE loop  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list in graph at key person  
                    IF neighbor IS NOT IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN the list converted from know_secret  
    END FUNCTION  
END CLASS
