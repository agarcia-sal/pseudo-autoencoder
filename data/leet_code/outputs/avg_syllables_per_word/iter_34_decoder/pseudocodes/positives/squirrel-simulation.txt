CLASS Solution  
    FUNCTION minDistance(height PARAMETER height INTEGER, width PARAMETER width INTEGER, tree PARAMETER tree LIST OF INTEGER, squirrel PARAMETER squirrel LIST OF INTEGER, nuts PARAMETER nuts LIST OF LIST OF INTEGER) RETURNS INTEGER  
        FUNCTION distance(p1 PARAMETER p1 LIST OF INTEGER, p2 PARAMETER p2 LIST OF INTEGER) RETURNS INTEGER  
            SET horizontal_difference TO element at position zero of p1 MINUS element at position zero of p2  
            IF horizontal_difference LESS THAN zero  
                SET horizontal_difference TO horizontal_difference MULTIPLIED BY negative one  
            END IF  
            SET vertical_difference TO element at position one of p1 MINUS element at position one of p2  
            IF vertical_difference LESS THAN zero  
                SET vertical_difference TO vertical_difference MULTIPLIED BY negative one  
            END IF  
            RETURN horizontal_difference PLUS vertical_difference  
        END FUNCTION  
        
        SET total_distance TO zero  
        SET max_save TO negative infinity  
        
        FOR each nut IN nuts  
            INCREMENT total_distance BY two MULTIPLIED BY distance(nut, tree)  
            SET first_nut_distance TO distance(squirrel, nut) PLUS distance(nut, tree)  
            SET later_nut_distance TO two MULTIPLIED BY distance(nut, tree)  
            SET save TO later_nut_distance MINUS first_nut_distance  
            IF save GREATER THAN max_save  
                SET max_save TO save  
            END IF  
        END FOR  
        
        RETURN total_distance MINUS max_save  
    END FUNCTION  
END CLASS
