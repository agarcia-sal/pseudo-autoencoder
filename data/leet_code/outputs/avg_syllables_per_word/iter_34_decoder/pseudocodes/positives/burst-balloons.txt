CLASS Solution  
    FUNCTION maxCoins(nums)  
        SET nums TO a new list made by placing one at the beginning FOLLOWED BY all elements of nums FOLLOWED BY one at the end  
        SET n TO the LENGTH OF nums  
        SET dp TO a two dimensional list of zeros with dimensions n BY n  
        FOR length FROM two TO n MINUS one  
            FOR i FROM zero TO n MINUS length MINUS one  
                SET j TO i PLUS length  
                FOR k FROM i PLUS one TO j MINUS one  
                    SET coins TO element at position i of nums MULTIPLIED BY element at position k of nums MULTIPLIED BY element at position j of nums PLUS element at position i THEN k of dp PLUS element at position k THEN j of dp  
                    IF element at position i THEN j of dp LESS THAN coins  
                        SET element at position i THEN j of dp TO coins  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position zero THEN n MINUS one of dp  
    END FUNCTION  
END CLASS
