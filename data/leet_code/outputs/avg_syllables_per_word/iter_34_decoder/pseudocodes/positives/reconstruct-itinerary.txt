CLASS Solution  
    FUNCTION findItinerary WITH parameter tickets  
        SET graph TO a new mapping that produces an empty list for any new key  
        FOR each pair of departure location named frm AND arrival location named to IN the tickets sorted in descending order  
            APPEND to TO the list at key frm in graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit WITH parameter airport  
            WHILE the list at key airport in graph holds any elements  
                CALL visit WITH argument the element removed from the end of the list at key airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit WITH argument the string representing the letters J F K  
        RETURN the list itinerary in reversed order  
    END FUNCTION  
END CLASS
