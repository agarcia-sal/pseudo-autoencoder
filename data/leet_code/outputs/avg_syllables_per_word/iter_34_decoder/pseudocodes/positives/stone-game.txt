CLASS Solution  
    FUNCTION stoneGame(piles)  
        FUNCTION dfs(i, j)  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            SET left_choice TO element at position i of piles MINUS dfs(i PLUS one, j)  
            SET right_choice TO element at position j of piles MINUS dfs(i, j MINUS one)  
            SET result TO the GREATER OF left_choice AND right_choice  
            RETURN result  
        END FUNCTION  
        SET game_result TO dfs(zero, the LENGTH OF piles MINUS one) GREATER THAN zero  
        RETURN game_result  
    END FUNCTION  
END CLASS
