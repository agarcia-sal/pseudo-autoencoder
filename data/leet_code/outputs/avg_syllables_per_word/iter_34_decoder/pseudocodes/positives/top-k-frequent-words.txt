CLASS Solution  
    FUNCTION topKFrequent(words AS List of strings, k AS integer) RETURNS List of strings  
        SET count TO the result of counting the frequency of each word in words  
        SET heap TO an empty list  
        FOR each word AND corresponding frequency IN count items  
            APPEND a pair of negative frequency and word TO heap  
        END FOR  
        TRANSFORM heap into a heap structure  
        SET result TO an empty list  
        FOR repetition FROM one TO k  
            REMOVE the smallest element from heap as a pair  
            APPEND the word part of this pair TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
