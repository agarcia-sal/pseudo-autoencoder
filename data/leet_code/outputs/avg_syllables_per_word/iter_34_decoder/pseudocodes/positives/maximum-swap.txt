CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO list of string characters representing num  
        SET last TO empty dictionary  
        FOR each index i AND digit character d IN digits  
            SET integer_digit TO integer value of d  
            SET last at key integer_digit TO i  
        END FOR  
        FOR each index i AND digit character d IN digits  
            SET current_digit TO integer value of d  
            FOR digit d2 FROM nine DOWN TO current_digit PLUS one  
                IF last at key d2 EXISTS AND last at key d2 GREATER THAN i THEN  
                    SET temporary TO element at position i of digits  
                    SET element at position i of digits TO element at position last at key d2 of digits  
                    SET element at position last at key d2 of digits TO temporary  
                    SET result_string TO concatenation of all elements in digits as string  
                    SET result_number TO integer value of result_string  
                    RETURN result_number  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
