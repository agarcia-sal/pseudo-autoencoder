CLASS Solution  
    FUNCTION findPaths WITH PARAMETERS m n maxMove startRow startColumn  
        SET MOD TO the integer one followed by nine zeros PLUS seven  
        SET directions TO the collection of four pairs representing zero and one zero and minus one one and zero and minus one and zero  
        INITIALIZE dp AS a three dimensional collection WITH zero VALUES with outer dimension of size m inner dimension of size n and innermost dimension of size maxMove PLUS one  
        FOR k FROM one TO maxMove  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    FOR each pair di dj IN directions  
                        SET ni TO i PLUS di  
                        SET nj TO j PLUS dj  
                        IF ni LESS THAN zero OR ni GREATER THAN OR EQUAL TO m OR nj LESS THAN zero OR nj GREATER THAN OR EQUAL TO n  
                            SET dp element at position i THEN j THEN k TO dp element at position i THEN j THEN k PLUS one MODULO MOD  
                        ELSE  
                            SET dp element at position i THEN j THEN k TO dp element at position i THEN j THEN k PLUS dp element at position ni THEN nj THEN k MINUS one MODULO MOD  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position startRow THEN startColumn THEN maxMove  
    END FUNCTION  
END CLASS
