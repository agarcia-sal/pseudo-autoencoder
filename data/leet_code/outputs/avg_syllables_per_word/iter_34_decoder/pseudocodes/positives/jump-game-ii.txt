CLASS Solution  
    FUNCTION jump(nums)  
        IF the LENGTH OF nums LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        SET jumps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        FOR i FROM zero TO the LENGTH OF nums MINUS one MINUS one  
            SET farthest TO the GREATER THAN value BETWEEN farthest AND the SUM of i PLUS the element at position i of nums  
            IF i EQUALS current_end  
                INCREMENT jumps BY one  
                SET current_end TO farthest  
                IF current_end GREATER THAN OR EQUAL TO the LENGTH OF nums MINUS one  
                    BREAK the FOR loop  
                END IF  
            END IF  
        END FOR  
        RETURN jumps  
    END FUNCTION  
END CLASS
