CLASS Solution  
    FUNCTION sequenceReconstruction(nums, sequences)  
        SET n TO the LENGTH OF nums  
        SET graph TO an empty default dictionary with list as default value  
        SET indegree TO a list of zeros with the LENGTH OF n PLUS one  
        
        FOR each sequence IN sequences  
            FOR i FROM zero TO the LENGTH OF sequence MINUS one MINUS one  
                SET u TO element at position i of sequence  
                SET v TO element at position i PLUS one of sequence  
                APPEND v TO the list at key u in graph  
                INCREMENT indegree at position v BY one  
            END FOR  
        END FOR  
        
        SET queue TO a deque containing each integer i FROM one TO n WHERE indegree at position i EQUALS zero  
        
        IF the LENGTH OF queue NOT EQUALS one  
            RETURN false  
        END IF  
        
        SET index TO zero  
        WHILE queue is not empty  
            IF the LENGTH OF queue NOT EQUALS one  
                RETURN false  
            END IF  
            
            SET node TO the first element removed FROM the left of queue  
            IF node NOT EQUALS element at position index of nums  
                RETURN false  
            END IF  
            INCREMENT index BY one  
            
            FOR each neighbor IN the list at key node in graph  
                DECREMENT indegree at position neighbor BY one  
                IF indegree at position neighbor EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN index EQUALS n  
    END FUNCTION  
END CLASS
