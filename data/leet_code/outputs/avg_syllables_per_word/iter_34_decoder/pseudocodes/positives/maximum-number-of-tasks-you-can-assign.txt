CLASS Solution  
    FUNCTION maxTaskAssign(tasks LIST OF integer, workers LIST OF integer, pills integer, strength integer) RETURNS integer  
        CALL sort ON tasks  
        CALL sort ON workers  
        
        FUNCTION canAssignTasks(k integer) RETURNS boolean  
            SET available_workers TO sublist from position LENGTH OF workers MINUS k TO end OF workers  
            SET pills_used TO zero  
            
            FOR i FROM k MINUS one DOWN TO zero  
                SET task TO element at position i OF tasks  
                SET worker_idx TO the position of the smallest element IN available_workers GREATER THAN OR EQUAL TO task  
                
                IF worker_idx IS LESS THAN LENGTH OF available_workers  
                    REMOVE element at position worker_idx FROM available_workers  
                ELSE IF pills_used IS LESS THAN pills  
                    SET worker_idx TO the position of the smallest element IN available_workers GREATER THAN OR EQUAL TO task MINUS strength  
                    
                    IF worker_idx IS LESS THAN LENGTH OF available_workers  
                        REMOVE element at position worker_idx FROM available_workers  
                        INCREMENT pills_used BY one  
                    ELSE  
                        RETURN false  
                    END IF  
                ELSE  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the minimum of LENGTH OF tasks AND LENGTH OF workers PLUS one  
        
        WHILE left IS LESS THAN right  
            SET mid TO the integer division of left PLUS right BY two  
            
            IF canAssignTasks(mid) IS true  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left MINUS one  
    END FUNCTION  
END CLASS
