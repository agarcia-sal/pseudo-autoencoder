CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO an empty collection mapping each course FROM zero TO numCourses MINUS one TO an empty list  
        SET in_degrees TO a list of zeros WITH length numCourses  
        
        FOR each pair of destination_course AND source_course IN prerequisites  
            APPEND destination_course TO the list at key source_course in adj_list  
            INCREMENT in_degrees element at position destination_course BY one  
        END FOR  
        
        SET queue TO a collection of courses FROM zero TO numCourses MINUS one WHERE the in_degrees element at position course EQUALS zero  
        SET topological_order TO an empty list  
        
        WHILE queue is not empty  
            SET current TO the first element removed FROM queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN the list at key current IN adj_list  
                DECREMENT in_degrees element at position neighbor BY one  
                IF in_degrees element at position neighbor EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the length of topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
