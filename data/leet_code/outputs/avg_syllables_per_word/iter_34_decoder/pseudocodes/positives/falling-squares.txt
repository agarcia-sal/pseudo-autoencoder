CLASS Solution  
    FUNCTION fallingSquares(positions)  
        SET max_heights TO an empty list  
        SET intervals TO an empty list  
        SET current_max_height TO zero  
        FOR each element in positions  
            SET left TO the first value of element  
            SET side_length TO the second value of element  
            SET right TO left PLUS side_length MINUS one  
            SET height TO side_length  
            FOR each interval_element IN intervals  
                SET interval_left TO the first value of interval_element  
                SET interval_right TO the second value of interval_element  
                SET interval_height TO the third value of interval_element  
                IF left LESS THAN OR EQUAL TO interval_right AND right GREATER THAN OR EQUAL TO interval_left THEN  
                    IF height LESS THAN side_length PLUS interval_height THEN  
                        SET height TO side_length PLUS interval_height  
                    END IF  
                END IF  
            END FOR  
            APPEND the tuple consisting of left right and height TO intervals  
            IF current_max_height LESS THAN height THEN  
                SET current_max_height TO height  
            END IF  
            APPEND current_max_height TO max_heights  
        END FOR  
        RETURN max_heights  
    END FUNCTION  
END CLASS
