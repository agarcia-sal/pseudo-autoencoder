CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the string formed by characters of element at position index of strs in reverse order  
            IF reversed_str IS GREATER THAN the element at position index of strs  
                SET element at position index of strs TO reversed_str  
            END IF  
        END FOR  
        
        SET doubled_strs TO the concatenation of strs WITH strs  
        
        SET max_string TO empty string  
        
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN the list containing FALSE and TRUE  
                IF is_reversed IS TRUE  
                    SET current TO the string formed by characters of element at position index of strs in reverse order  
                ELSE  
                    SET current TO the element at position index of strs  
                END IF  
                
                SET remaining TO the concatenation of elements from position index plus one TO position index plus the LENGTH OF strs IN doubled_strs  
                
                FOR cut_position FROM zero TO the LENGTH OF current  
                    SET candidate TO the concatenation of substring from position cut_position TO the end of current WITH remaining WITH substring from position zero TO position cut_position of current  
                    IF candidate IS GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
