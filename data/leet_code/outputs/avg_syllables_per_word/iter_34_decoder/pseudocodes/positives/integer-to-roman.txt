CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO list of integer one thousand PLUS nine hundred PLUS five hundred PLUS four hundred PLUS one hundred PLUS ninety PLUS fifty PLUS forty PLUS ten PLUS nine PLUS five PLUS four PLUS one  
        SET syms TO list of string M PLUS CM PLUS D PLUS CD PLUS C PLUS XC PLUS L PLUS XL PLUS X PLUS IX PLUS V PLUS IV PLUS I  
        SET roman_numeral TO an empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero  
            SET times TO num DIVIDED BY element at position i of val  
            FOR each iteration FROM one TO times  
                APPEND element at position i of syms TO roman_numeral  
                DECREMENT num BY element at position i of val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
