CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO positive infinity  
        
        FOR each num IN nums  
            IF element at position one of num MODULO two EQUALS one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            IF min_value GREATER THAN num  
                SET min_value TO num  
            END IF  
            APPEND negative num TO max_heap maintaining heap property  
        END FOR  
        
        SET min_deviation TO positive infinity  
        
        WHILE element at position one of element at position one of max_heap MODULO two EQUALS zero  
            SET max_value TO negative of top element removed from max_heap maintaining heap property  
            IF min_deviation GREATER THAN max_value MINUS min_value  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY two  
            IF min_value GREATER THAN max_value  
                SET min_value TO max_value  
            END IF  
            APPEND negative max_value TO max_heap maintaining heap property  
        END WHILE  
        
        IF min_deviation GREATER THAN negative of top element at max_heap MINUS min_value  
            SET min_deviation TO negative of top element at max_heap MINUS min_value  
        END IF  
        
        RETURN min_deviation  
    END FUNCTION  
END CLASS
