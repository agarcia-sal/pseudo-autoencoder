CLASS Solution  
    FUNCTION minimumOneBitOperations WITH PARAMETER n  
        IF n EQUALS zero THEN  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE two RAISED TO THE power OF k PLUS one THAT IS LESS THAN OR EQUAL TO n HOLDS  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO two RAISED TO THE power OF k  
        RETURN two RAISED TO THE power OF k PLUS one MINUS one MINUS minimumOneBitOperations OF n EXCLUSIVE OR most_significant_bit  
    END FUNCTION  
END CLASS
