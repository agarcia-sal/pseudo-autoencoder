CLASS Solution  
    FUNCTION lenLongestFibSubseq(arr)  
        SET index TO empty mapping  
        FOR each position i FROM zero TO the length of arr MINUS one  
            ASSIGN reference TO i FOR element at position i of arr IN index  
        END FOR  
        SET dp TO empty mapping  
        SET max_length TO zero  
        
        FOR each position k FROM zero TO the length of arr MINUS one  
            FOR each position j FROM zero TO k MINUS one  
                SET difference TO element at position k of arr MINUS element at position j of arr  
                SET i TO the value in index for difference OR minus one if not present  
                IF i GREATER THAN OR EQUAL TO zero AND i LESS THAN j THEN  
                    SET previous_length TO the value in dp for the pair of i and j OR two if not present  
                    SET dp value for the pair of j and k TO previous_length PLUS one  
                    IF dp value for the pair of j and k GREATER THAN max_length THEN  
                        SET max_length TO dp value for the pair of j and k  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF max_length GREATER THAN OR EQUAL TO three THEN  
            RETURN max_length  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
