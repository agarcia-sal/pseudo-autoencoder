CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL sort arr2 in ascending order  
        SET dp TO a dictionary with key negative one and value zero  
        
        FOR each num IN arr1  
            SET new_dp TO an empty dictionary  
            
            FOR each last_num and ops IN items of dp  
                IF num GREATER THAN last_num  
                    SET the value at key num in new_dp TO the minimum value between the current value at key num in new_dp or infinity AND ops  
                END IF  
                
                SET idx TO the position to insert last_num in arr2 to keep arr2 sorted on the right side  
                IF idx LESS THAN the length of arr2  
                    SET the value at key element at position idx of arr2 in new_dp TO the minimum value between the current value at that key in new_dp or infinity AND ops PLUS one  
                END IF  
            END FOR  
            
            IF new_dp is empty  
                RETURN negative one  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the minimum value among all values in dp  
    END FUNCTION  
END CLASS
