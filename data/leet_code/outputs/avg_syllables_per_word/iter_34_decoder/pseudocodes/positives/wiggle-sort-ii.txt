CLASS Solution  
    FUNCTION wiggleSort(nums)  
        CALL sort_nums_in_natural_order(nums)  
        SET sorted_nums TO copy_of_all_elements_in nums  
        SET n TO length_of nums  
        SET left TO integer_division_of the difference of n AND one BY two  
        SET right TO difference_of n AND one  
        FOR i FROM zero TO difference_of n AND one  
            IF remainder_of i DIVIDED BY two EQUALS zero  
                SET element at position i of nums TO element at position left of sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET element at position i of nums TO element at position right of sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
