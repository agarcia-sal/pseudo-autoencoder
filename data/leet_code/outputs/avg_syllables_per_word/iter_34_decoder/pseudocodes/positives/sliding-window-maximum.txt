CLASS Solution  
    FUNCTION maxSlidingWindow(nums k)  
        IF nums EQUALS empty OR k EQUALS zero  
            RETURN empty list  
        END IF  
        ASSIGN reference TO an empty deque named deq  
        ASSIGN reference TO an empty list named result  
        FOR index i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq is not empty AND element at position zero of deq LESS THAN i MINUS k PLUS one  
                REMOVE element at position zero FROM deq  
            END IF  
            WHILE deq is not empty AND element of nums at element at last position of deq LESS THAN element of nums at i  
                REMOVE element at last position FROM deq  
            END WHILE  
            APPEND i TO deq  
            IF i GREATER THAN OR EQUAL TO k MINUS one  
                APPEND element of nums at element at position zero of deq TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
