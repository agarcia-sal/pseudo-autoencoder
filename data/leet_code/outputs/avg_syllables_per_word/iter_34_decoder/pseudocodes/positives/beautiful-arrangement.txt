CLASS Solution  
    FUNCTION countArrangement(n)  
        FUNCTION backtrack(used, index)  
            IF index EQUALS n PLUS one  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR i FROM one TO n  
                IF element at position i of used EQUALS False AND (i MODULO index EQUALS zero OR index MODULO i EQUALS zero)  
                    SET element at position i of used TO True  
                    INCREMENT count BY backtrack(used, index PLUS one)  
                    SET element at position i of used TO False  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET used TO list of False values repeated n PLUS one times  
        RETURN backtrack(used, one)  
    END FUNCTION  
END CLASS
