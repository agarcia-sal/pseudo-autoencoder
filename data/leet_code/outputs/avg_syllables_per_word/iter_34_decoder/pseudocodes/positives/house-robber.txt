CLASS Solution  
    FUNCTION rob(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position zero of nums  
        END IF  
        SET dp TO a list of zeros with the LENGTH OF nums  
        SET the element at position zero of dp TO the element at position zero of nums  
        IF the element at position zero of nums GREATER THAN OR EQUAL TO the element at position one of nums  
            SET the element at position one of dp TO the element at position zero of nums  
        ELSE  
            SET the element at position one of dp TO the element at position one of nums  
        END IF  
        FOR index i FROM two TO the LENGTH OF nums MINUS one  
            IF the element at position i MINUS one of dp GREATER THAN OR EQUAL TO the element at position i MINUS two of dp PLUS the element at position i of nums  
                SET the element at position i of dp TO the element at position i MINUS one of dp  
            ELSE  
                SET the element at position i of dp TO the element at position i MINUS two of dp PLUS the element at position i of nums  
            END IF  
        END FOR  
        RETURN the element at the last position of dp  
    END FUNCTION  
END CLASS
