CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        SET q TO an empty double ended queue containing a list with element zero  
        SET ans TO an empty list  
        WHILE the LENGTH OF q is GREATER THAN zero  
            SET path TO the element removed from the left end of q  
            SET u TO the element at position the LENGTH OF path MINUS one of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the loop  
            END IF  
            FOR each v IN the element at position u of graph  
                APPEND a new list consisting of the elements of path FOLLOWED BY v TO q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
