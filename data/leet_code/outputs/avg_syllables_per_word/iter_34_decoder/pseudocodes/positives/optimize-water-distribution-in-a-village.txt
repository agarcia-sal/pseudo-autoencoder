CLASS Solution  
    FUNCTION minCostToSupplyWater(n, wells, pipes)  
        FOR index FROM zero TO n MINUS one  
            APPEND list containing zero AND index PLUS one AND element at position index of wells TO pipes  
        END FOR  
        SORT pipes BASED ON the third element of each item IN ascending order  
        SET parent TO list of integers FROM zero TO n  
        FUNCTION find(x)  
            IF element at position x of parent NOT EQUALS x  
                SET element at position x of parent TO result of find WITH argument element at position x of parent  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        FUNCTION union(x, y)  
            SET rootX TO result of find WITH argument x  
            SET rootY TO result of find WITH argument y  
            IF rootX NOT EQUALS rootY  
                SET element at position rootX of parent TO rootY  
                RETURN True  
            END IF  
            RETURN False  
        END FUNCTION  
        SET min_cost TO zero  
        FOR each element IN pipes  
            LET house1 BE first element of current element  
            LET house2 BE second element of current element  
            LET cost BE third element of current element  
            IF union WITH arguments house1 AND house2 RETURNS True  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
