CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty collection of key value pairs  
        FOR each element num IN nums  
            SET depth TO the value of num DIVIDED BY one hundred rounded down to the nearest whole number  
            SET pos TO the result of num DIVIDED BY ten rounded down to the nearest whole number MODULO ten  
            SET value TO the remainder of num DIVIDED BY ten  
            ASSIGN reference TO value in tree with key composed of depth and pos  
        END FOR  
  
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the value in tree at key composed of depth and pos  
            IF current_value EQUALS to none  
                RETURN zero  
            END IF  
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child TO a pair composed of depth PLUS one and pos MULTIPLIED BY two MINUS one  
            SET right_child TO a pair composed of depth PLUS one and pos MULTIPLIED BY two  
            IF left_child NOT IN tree AND right_child NOT IN tree  
                RETURN new_path_sum  
            END IF  
            SET left_sum TO the result of dfs called with the first element of left_child, the second element of left_child, and new_path_sum  
            SET right_sum TO the result of dfs called with the first element of right_child, the second element of right_child, and new_path_sum  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
  
        RETURN the result of dfs called with one for depth, one for pos, and zero for path_sum  
    END FUNCTION  
END CLASS
