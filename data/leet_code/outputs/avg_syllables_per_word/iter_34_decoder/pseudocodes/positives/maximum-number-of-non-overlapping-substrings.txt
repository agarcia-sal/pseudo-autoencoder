CLASS Solution  
    FUNCTION maxNumOfSubstrings(s)  
        SET first TO empty dictionary  
        SET last TO empty dictionary  
        FOR each index AND character IN enumeration of s  
            IF character NOT IN first  
                SET first AT character TO index  
            END IF  
            SET last AT character TO index  
        END FOR  
        
        SET intervals TO empty list  
        FOR each character IN keys of first  
            SET start TO first AT character  
            SET end TO last AT character  
            SET i TO start  
            WHILE i LESS THAN OR EQUAL TO end  
                IF first AT element at position i of s LESS THAN start  
                    BREAK the loop  
                END IF  
                SET end TO the GREATER THAN OF end AND last AT element at position i of s  
                INCREMENT i BY one  
            END WHILE  
            IF loop ended WITHOUT break  
                APPEND pair of start AND end TO intervals  
            END IF  
        END FOR  
        
        SORT intervals BY the second element of each pair in ascending order  
        
        SET result TO empty list  
        SET prev_end TO negative one  
        FOR each pair start AND end IN intervals  
            IF start GREATER THAN prev_end  
                APPEND the substring of s from position start TO position end to result  
                SET prev_end TO end  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
