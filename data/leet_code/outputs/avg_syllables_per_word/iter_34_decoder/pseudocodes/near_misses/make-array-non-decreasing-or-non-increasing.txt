CLASS Solution  
    FUNCTION convertArray(nums)  
        FUNCTION min_operations_to_non_decreasing(nums)  
            SET max_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF max_heap is not empty AND the negative of the element at position zero of max_heap GREATER THAN num  
                    INCREMENT operations BY the negative of the element extracted from max_heap MINUS num  
                    INSERT the negative of num INTO max_heap  
                END IF  
                INSERT the negative of num INTO max_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
        
        FUNCTION min_operations_to_non_increasing(nums)  
            SET min_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF min_heap is not empty AND the element at position zero of min_heap LESS THAN num  
                    INCREMENT operations BY num MINUS the element extracted from min_heap  
                    INSERT num INTO min_heap  
                END IF  
                INSERT num INTO min_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
        
        RETURN the smaller value BETWEEN min_operations_to_non_decreasing(nums) AND min_operations_to_non_increasing(nums)  
    END FUNCTION  
END CLASS
