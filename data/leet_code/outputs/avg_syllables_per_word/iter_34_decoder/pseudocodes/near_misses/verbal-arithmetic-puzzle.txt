CLASS Solution  
    FUNCTION isSolvable(words, result)  
        SET unique_chars TO the set of all characters found by concatenating all words in words plus result  
        IF the LENGTH OF unique_chars GREATER THAN ten  
            RETURN false  
        END IF  
        
        SET non_zero_chars TO the set of first characters of each word in words WHERE the LENGTH OF word GREATER THAN one UNION the first character of result IF the LENGTH OF result GREATER THAN one OTHERWISE an empty string  
        
        SET char_to_digit TO an empty mapping from characters to digits  
        SET digit_to_char TO an empty mapping from digits to characters  
        
        FUNCTION can_assign(char, digit)  
            RETURN char NOT IN char_to_digit AND (digit NOT IN digit_to_char OR digit_to_char[digit] EQUALS char)  
        END FUNCTION  
        
        FUNCTION assign(char, digit)  
            SET char_to_digit at key char TO digit  
            SET digit_to_char at key digit TO char  
        END FUNCTION  
        
        FUNCTION unassign(char, digit)  
            REMOVE the key char FROM char_to_digit  
            REMOVE the key digit FROM digit_to_char  
        END FUNCTION  
        
        FUNCTION word_value(word)  
            SET value_sum TO zero  
            FOR each position i and character char IN the word starting from the last character TO the first character WITH enumeration  
                INCREMENT value_sum BY char_to_digit at key char MULTIPLIED BY ten RAISED TO the power of i  
            END FOR  
            RETURN value_sum  
        END FUNCTION  
        
        FUNCTION backtrack(index, col, sum_col)  
            IF col GREATER THAN OR EQUAL TO the maximum LENGTH OF any word in words AND col GREATER THAN OR EQUAL TO the LENGTH OF result  
                RETURN sum_col EQUALS zero  
            END IF  
            
            IF index EQUALS the LENGTH OF words  
                IF col LESS THAN the LENGTH OF result  
                    SET digit_char TO the character at position LENGTH OF result MINUS one MINUS col of result  
                    IF digit_char IN char_to_digit  
                        IF char_to_digit at key digit_char EQUALS sum_col MODULO ten  
                            RETURN backtrack ZERO col PLUS one sum_col INTEGER DIVIDED BY ten  
                        ELSE  
                            RETURN false  
                        END IF  
                    ELSE IF can_assign(digit_char sum_col MODULO ten) AND (sum_col MODULO ten NOT EQUALS zero OR digit_char NOT IN non_zero_chars)  
                        assign digit_char sum_col MODULO ten  
                        IF backtrack ZERO col PLUS one sum_col INTEGER DIVIDED BY ten  
                            RETURN true  
                        END IF  
                        unassign digit_char sum_col MODULO ten  
                    END IF  
                ELSE  
                    RETURN sum_col EQUALS zero  
                END IF  
                RETURN false  
            END IF  
            
            SET current_word TO element at position index of words  
            IF col GREATER THAN OR EQUAL TO the LENGTH OF current_word  
                RETURN backtrack index PLUS one col sum_col  
            END IF  
            
            SET current_char TO the character at position LENGTH OF current_word MINUS one MINUS col of current_word  
            IF current_char IN char_to_digit  
                RETURN backtrack index PLUS one col sum_col PLUS char_to_digit at key current_char  
            END IF  
            
            FOR digit FROM zero TO nine  
                IF can_assign current_char digit AND (digit NOT EQUALS zero OR current_char NOT IN non_zero_chars)  
                    assign current_char digit  
                    IF backtrack index PLUS one col sum_col PLUS digit  
                        RETURN true  
                    END IF  
                    unassign current_char digit  
                END IF  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        RETURN backtrack ZERO ZERO ZERO  
    END FUNCTION  
END CLASS
