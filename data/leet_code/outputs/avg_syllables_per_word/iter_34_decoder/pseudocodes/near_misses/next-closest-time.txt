CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the unique characters of time with all colon characters removed  
        SET hour TO the integer value obtained by converting the substring consisting of the first two characters of time  
        SET minute TO the integer value obtained by converting the substring consisting of the last two characters of time  

        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each first hour digit IN digits  
                FOR each second hour digit IN digits  
                    FOR each first minute digit IN digits  
                        FOR each second minute digit IN digits  
                            SET hour_value TO the integer value obtained by concatenating first hour digit and second hour digit  
                            SET minute_value TO the integer value obtained by concatenating first minute digit and second minute digit  
                            IF hour_value LESS THAN twenty-four AND minute_value LESS THAN sixty  
                                APPEND the tuple consisting of hour_value and minute_value TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN times sorted in ascending order  
        END FUNCTION  

        SET all_times TO the result of generate_times invoked with digits  
        SET current_time_index TO the position of the tuple consisting of hour and minute in all_times  
        IF current_time_index PLUS one LESS THAN the length of all_times  
            SET next_hour TO the first element of the tuple at position current_time_index PLUS one of all_times  
            SET next_minute TO the second element of the tuple at position current_time_index PLUS one of all_times  
        ELSE  
            SET next_hour TO the first element of the tuple at position zero of all_times  
            SET next_minute TO the second element of the tuple at position zero of all_times  
        END IF  

        RETURN the string composed of next_hour expressed as two digit number followed by a colon followed by next_minute expressed as two digit number  
    END FUNCTION  
END CLASS
