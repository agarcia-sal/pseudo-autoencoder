CLASS Solution  
    FUNCTION smallestMissingValueSubtree(parents, nums)  
        SET n TO the LENGTH OF parents  
        INITIALIZE tree AS a mapping from each node to a list of its child nodes  
        
        FOR index FROM one TO n MINUS one  
            APPEND index TO the list of children for the parent at position index in parents  
        END FOR  
        
        SET node_with_one TO negative one  
        FOR index FROM zero TO n MINUS one  
            IF the element at position index of nums EQUALS one THEN  
                SET node_with_one TO index  
                BREAK the loop  
            END IF  
        END FOR  
        
        IF node_with_one EQUALS negative one THEN  
            RETURN a list of one repeated n times  
        END IF  
        
        INITIALIZE result AS a list of one repeated n times  
        INITIALIZE visited AS an empty set  
        
        FUNCTION dfs(node)  
            ADD node TO visited  
            INITIALIZE genetic_values AS a set containing the element at position node of nums  
            FOR each child IN the list of children for node in tree  
                IF child NOT IN visited THEN  
                    SET updated_values TO the result of dfs called with child  
                    ADD all elements of updated_values TO genetic_values  
                END IF  
            END FOR  
            
            SET smallest_missing TO one  
            WHILE smallest_missing IN genetic_values DO  
                INCREMENT smallest_missing BY one  
            END WHILE  
            
            SET the element at position node of result TO smallest_missing  
            RETURN genetic_values  
        END FUNCTION  
        
        INITIALIZE path AS an empty list  
        SET current TO node_with_one  
        WHILE current NOT EQUALS negative one  
            APPEND current TO path  
            SET current TO the element at position current of parents  
        END WHILE  
        
        INITIALIZE path_set AS a set containing all elements of path  
        
        FOR each node IN the reverse order of path  
            IF node IN visited THEN  
                CONTINUE to the next iteration  
            END IF  
            CALL dfs(node)  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
