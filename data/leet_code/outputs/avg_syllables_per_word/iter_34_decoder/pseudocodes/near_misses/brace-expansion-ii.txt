CLASS Solution  
    FUNCTION braceExpansionII(expression)  
        FUNCTION parse_expression(index)  
            SET current_set TO an empty set  
            SET current_product TO a list containing an empty string  
            
            WHILE index LESS THAN the LENGTH OF expression  
                SET char TO element at position index of expression  
                
                IF char EQUALS the symbol representing opening curly brace  
                    SET inner_set AND index TO the result of parse_expression with index PLUS one  
                    SET new_product TO an empty list  
                    FOR each string_a IN current_product  
                        FOR each string_b IN inner_set  
                            APPEND concatenation of string_a AND string_b TO new_product  
                        END FOR  
                    END FOR  
                    SET current_product TO new_product  
                
                ELSE IF char EQUALS the symbol representing closing curly brace  
                    FOR each p IN current_product  
                        ADD the concatenation of characters in p TO current_set  
                    END FOR  
                    RETURN current_set AND index  
                
                ELSE IF char EQUALS the symbol representing comma  
                    FOR each p IN current_product  
                        ADD the concatenation of characters in p TO current_set  
                    END FOR  
                    SET current_product TO a list containing an empty string  
                
                ELSE  # char is a lowercase letter  
                    SET new_product TO an empty list  
                    FOR each string_a IN current_product  
                        APPEND concatenation of string_a AND char TO new_product  
                    END FOR  
                    SET current_product TO new_product  
                
                INCREMENT index BY one  
            END WHILE  
            
            FOR each p IN current_product  
                ADD the concatenation of characters in p TO current_set  
            END FOR  
            
            RETURN current_set AND index  
        END FUNCTION  
        
        SET result_set AND a throwaway variable TO parse_expression with zero  
        RETURN sorted list of elements in result_set  
    END FUNCTION  
END CLASS
