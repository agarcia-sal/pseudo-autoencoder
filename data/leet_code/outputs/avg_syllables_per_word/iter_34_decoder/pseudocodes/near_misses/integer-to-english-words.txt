CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  
        
        SET below_twenty TO list of words empty string One Two Three Four Five Six Seven Eight Nine Ten Eleven Twelve Thirteen Fourteen Fifteen Sixteen Seventeen Eighteen Nineteen  
        SET tens TO list of words empty string empty string Twenty Thirty Forty Fifty Sixty Seventy Eighty Ninety  
        SET thousands TO list of words empty string Thousand Million Billion  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            END IF  
            ELSE IF n LESS THAN twenty  
                RETURN element at position n of below_twenty concatenated with a space  
            END IF  
            ELSE IF n LESS THAN one hundred  
                RETURN element at position n DIVIDED BY ten of tens concatenated with a space concatenated with result of calling helper with n MODULO ten  
            END IF  
            ELSE  
                RETURN element at position n DIVIDED BY one hundred of below_twenty concatenated with the word Hundred concatenated with a space concatenated with result of calling helper with n MODULO one hundred  
            END ELSE  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero  
            IF num MODULO one thousand NOT EQUALS zero  
                SET result TO result of calling helper with num MODULO one thousand concatenated with element at position i of thousands concatenated with a space concatenated with result  
            END IF  
            SET num TO num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
