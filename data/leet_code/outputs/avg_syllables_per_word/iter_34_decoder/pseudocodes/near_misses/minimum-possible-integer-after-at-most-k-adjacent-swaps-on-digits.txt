CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO list of individual characters in num  
        SET n TO the LENGTH OF num  
        
        WHILE k GREATER THAN zero holds  
            SET swapped TO false  
            FOR i FROM zero TO the LENGTH OF num MINUS one MINUS one  
                IF element at position i of num GREATER THAN element at position i PLUS one of num THEN  
                    SET min_digit TO the character representing nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO the SMALLER OF i PLUS k PLUS one AND n MINUS one  
                        IF element at position j of num LESS THAN min_digit THEN  
                            SET min_digit TO element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET a new list TO element at position min_index of num APPENDED TO the sublist of num FROM position i TO position min_index MINUS one  
                    REPLACE the sublist of num FROM position i TO position min_index WITH the new list  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO true  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            
            IF swapped EQUALS false THEN  
                BREAK the WHILE loop  
            END IF  
        END WHILE  
        
        SET result TO concatenation of all elements in num  
        RETURN result  
    END FUNCTION  
END CLASS
