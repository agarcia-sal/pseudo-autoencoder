CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        SET subset_sums TO a list OF zeroes with the LENGTH OF two RAISED TO the power of n  
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO two RAISED TO the power of i MINUS one  
                SET element at position mask OR two RAISED TO the power of i OF subset_sums TO element at position mask OF subset_sums PLUS element at position i OF jobs  
            END FOR  
        END FOR  

        FUNCTION dp(k_param, mask_param)  
            IF k_param EQUALS one  
                RETURN element at position mask_param OF subset_sums  
            END IF  
            SET res TO infinity  
            SET submask TO mask_param  
            WHILE submask IS NOT zero  
                SET res TO the MINIMUM OF res AND the MAXIMUM OF dp with k_param MINUS one and mask_param XOR submask AND element at position submask OF subset_sums  
                SET submask TO submask MINUS one AND mask_param  
            END WHILE  
            RETURN res  
        END FUNCTION  

        RETURN dp with parameters k AND two RAISED TO the power of n MINUS one  
    END FUNCTION  
END CLASS
