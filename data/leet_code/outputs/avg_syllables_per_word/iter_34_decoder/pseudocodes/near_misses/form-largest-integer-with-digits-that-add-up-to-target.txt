CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        SET cost_to_digit TO an empty dictionary  
        FOR each index i AND each element c IN the enumeration of cost  
            SET the element at key c of cost_to_digit TO the string representation of the value of index i PLUS one  
        END FOR  
          
        SET dp TO a list consisting of an empty string FOLLOWED BY target number of elements each set TO a string representing negative one  
          
        FOR each integer t FROM one TO target  
            FOR each key c AND value d IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND the element at position t MINUS c of dp NOT EQUALS a string representing negative one THEN  
                    SET candidate TO the concatenation of d PLUS the element at position t MINUS c of dp  
                    IF the element at position t of dp EQUALS a string representing negative one OR the LENGTH OF candidate GREATER THAN the LENGTH OF the element at position t of dp OR both the LENGTH OF candidate EQUALS the LENGTH OF the element at position t of dp AND candidate GREATER THAN the element at position t of dp THEN  
                        SET the element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
          
        IF the element at position target of dp NOT EQUALS a string representing negative one THEN  
            RETURN the element at position target of dp  
        ELSE  
            RETURN the string zero  
        END IF  
    END FUNCTION  
END CLASS
