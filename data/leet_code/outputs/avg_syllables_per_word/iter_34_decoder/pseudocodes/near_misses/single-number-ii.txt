CLASS Solution  
    FUNCTION singleNumber WITH PARAMETER nums  
        SET ones TO zero  
        SET twos TO zero  
        FOR each num IN nums  
            SET ones TO bitwise XOR of ones AND num AND bitwise NOT of twos  
            SET twos TO bitwise XOR of twos AND num AND bitwise NOT of ones  
        END FOR  
        RETURN ones  
    END FUNCTION  
END CLASS
