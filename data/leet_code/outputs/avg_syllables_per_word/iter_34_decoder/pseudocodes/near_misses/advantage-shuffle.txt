CLASS Solution  
    FUNCTION advantageCount(nums1, nums2)  
        CALL sort nums1 in ascending order  
        SET nums2_with_indices TO a list of pairs where each pair contains the element of nums2 and its original position in nums2  
        CALL sort nums2_with_indices in descending order by the element value  
        SET result TO a list of zeros with the same length as nums1  
        SET left TO zero  
        SET right TO the length of nums1 MINUS one  
        FOR each pair (value, index) IN nums2_with_indices  
            IF element at position right of nums1 GREATER THAN value  
                SET element at position index of result TO element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index of result TO element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
