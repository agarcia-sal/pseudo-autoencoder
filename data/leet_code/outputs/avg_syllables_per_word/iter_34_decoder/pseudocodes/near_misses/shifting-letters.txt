CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO zero  
        FOR i FROM the LENGTH OF shifts MINUS one TO zero DECREMENT BY one  
            INCREMENT total_shifts BY element at position i of shifts  
            SET element at position i of shifts TO total_shifts  
        END FOR  
        SET result TO an empty list  
        FOR i FROM zero TO the LENGTH OF s MINUS one  
            SET numerical_value_of_character TO the numerical position of character at position i of s MINUS the numerical position of character a  
            SET shifted_value TO numerical_value_of_character PLUS element at position i of shifts  
            SET wrapped_value TO shifted_value MODULO twenty six  
            SET new_character TO the character corresponding to the numerical position of character a PLUS wrapped_value  
            APPEND new_character TO result  
        END FOR  
        SET output_string TO the concatenation of all elements in result  
        RETURN output_string  
    END FUNCTION  
END CLASS
