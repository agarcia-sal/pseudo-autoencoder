CLASS Solution  
    FUNCTION crackSafe(n k)  
        SET visited TO empty collection of unique elements  
        SET sequence TO empty list  

        FUNCTION dfs(node)  
            FOR each i FROM zero TO k MINUS one  
                SET edge TO concatenation of node AND string representation of i  
                IF edge is NOT IN visited  
                    ADD edge TO visited  
                    CALL dfs with argument substring of edge from position one TO the end  
                    APPEND string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  

        SET start_node TO string composed of character zero REPEATED n MINUS one times  
        CALL dfs with argument start_node  

        RETURN concatenation of all elements of sequence FOLLOWED BY start_node  
    END FUNCTION  
END CLASS
