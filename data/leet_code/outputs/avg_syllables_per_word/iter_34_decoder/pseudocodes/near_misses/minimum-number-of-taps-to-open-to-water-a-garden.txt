CLASS Solution  
    FUNCTION minTaps WITH parameters n AND ranges  
        SET max_range TO a list of zeros with length equal to n PLUS one  
        FOR index i FROM zero TO n  
            IF element at position i of ranges GREATER THAN zero  
                SET left TO the greater value between zero AND index i MINUS element at position i of ranges  
                SET right TO the lesser value between n AND index i PLUS element at position i of ranges  
                SET element at position left of max_range TO the greater value between element at position left of max_range AND right  
            END IF  
        END FOR  
        SET taps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        FOR index i FROM zero TO n  
            IF index i GREATER THAN farthest  
                RETURN negative one  
            END IF  
            IF index i GREATER THAN current_end  
                INCREMENT taps BY one  
                SET current_end TO farthest  
            END IF  
            SET farthest TO the greater value between farthest AND element at position i of max_range  
        END FOR  
        RETURN taps  
    END FUNCTION  
END CLASS
