CLASS Solution  
    FUNCTION maxSumOfThreeSubarrays(nums PARAMETER, k PARAMETER)  
        SET n TO the LENGTH OF nums  
        SET sums TO a LIST of zeros with LENGTH equal TO n MINUS k PLUS one  
        SET current_sum TO the SUM of the ELEMENTS from POSITION one TO POSITION k OF nums  
        SET the ELEMENT at POSITION one OF sums TO current_sum  
        
        FOR i FROM two TO n MINUS k PLUS one  
            SET current_sum TO current_sum PLUS ELEMENT at POSITION i PLUS k MINUS one OF nums MINUS ELEMENT at POSITION i MINUS one OF nums  
            SET the ELEMENT at POSITION i OF sums TO current_sum  
        END FOR  
        
        SET left TO a LIST of zeros with LENGTH equal TO n MINUS k PLUS one  
        SET right TO a LIST with ALL ELEMENTS equal TO n MINUS k with LENGTH equal TO n MINUS k PLUS one  
        
        SET max_sum TO zero  
        FOR i FROM one TO n MINUS k PLUS one  
            IF ELEMENT at POSITION i OF sums GREATER THAN ELEMENT at POSITION max_sum OF sums  
                SET max_sum TO i  
            END IF  
            SET the ELEMENT at POSITION i OF left TO max_sum  
        END FOR  
        
        SET max_sum TO n MINUS k  
        FOR i FROM n MINUS k PLUS one DOWN TO one  
            IF ELEMENT at POSITION i OF sums GREATER THAN OR EQUAL TO ELEMENT at POSITION max_sum OF sums  
                SET max_sum TO i  
            END IF  
            SET the ELEMENT at POSITION i OF right TO max_sum  
        END FOR  
        
        SET max_total TO zero  
        SET result TO a LIST of three zeros  
        FOR j FROM k PLUS one TO n MINUS two TIMES k PLUS one  
            SET i TO ELEMENT at POSITION j MINUS k OF left  
            SET l TO ELEMENT at POSITION j PLUS k OF right  
            SET total TO ELEMENT at POSITION i OF sums PLUS ELEMENT at POSITION j OF sums PLUS ELEMENT at POSITION l OF sums  
            IF total GREATER THAN max_total  
                SET max_total TO total  
                SET result TO a LIST containing i THEN j THEN l  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
