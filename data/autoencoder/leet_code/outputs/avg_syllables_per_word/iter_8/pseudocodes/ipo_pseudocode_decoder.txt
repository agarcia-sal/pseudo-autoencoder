CLASS Solution  
    FUNCTION findMaximizedCapital(k, w, profits, capital)  
        SET projects TO list of pairs of capital and profits zipped together  
        SORT projects by capital in ascending order  
        
        SET max_heap TO empty list  
        SET i TO 0  
        SET n TO length of projects  
        
        FOR each iteration FROM 1 TO k  
            WHILE i LESS THAN n AND capital of projects at i LESS THAN OR EQUAL TO w  
                ADD negative profit of projects at i TO max_heap  
                INCREMENT i BY 1  
            END WHILE  
            
            IF max_heap IS empty  
                BREAK the loop  
            END IF  
            
            SET w TO w PLUS negative of the top element removed from max_heap  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
