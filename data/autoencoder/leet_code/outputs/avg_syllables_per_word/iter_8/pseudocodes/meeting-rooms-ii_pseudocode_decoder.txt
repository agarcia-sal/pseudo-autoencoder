CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals IS EMPTY  
            RETURN 0  
        END IF  

        SORT intervals BY start time  

        SET min_heap TO empty list  

        FOR each meeting IN intervals  
            IF min_heap IS NOT EMPTY AND meeting start time IS GREATER THAN OR EQUAL TO earliest end time in min_heap  
                REMOVE earliest end time FROM min_heap  
            END IF  

            ADD current meeting end time TO min_heap  
        END FOR  

        RETURN size OF min_heap  
    END FUNCTION  
END CLASS
