CLASS Solution  
    FUNCTION networkDelayTime(times, n, k)  
        SET graph TO dictionary with keys from 1 to n, each mapped to empty list  
        FOR each u, v, w IN times  
            APPEND pair of (v, w) TO graph at key u  
        SET dist TO dictionary with keys from 1 to n, each mapped to infinity  
        SET dist at key k TO 0  
        SET min_heap TO list containing pair of (0, k)  
        WHILE min_heap is not empty  
            SET current_dist, node TO remove smallest item from min_heap  
            IF current_dist GREATER THAN dist at node  
                CONTINUE  
            FOR each neighbor, weight IN graph at node  
                SET distance TO current_dist PLUS weight  
                IF distance LESS THAN dist at neighbor  
                    SET dist at neighbor TO distance  
                    ADD pair of (distance, neighbor) TO min_heap  
        SET max_time TO maximum value in dist values  
        IF max_time EQUALS infinity  
            RETURN -1  
        ELSE  
            RETURN max_time  
    END FUNCTION  
END CLASS
