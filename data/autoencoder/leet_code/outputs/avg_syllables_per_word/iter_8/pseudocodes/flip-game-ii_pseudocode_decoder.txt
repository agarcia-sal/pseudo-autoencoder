CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF state IS IN memo THEN  
                RETURN memo[state]  
            
            FOR i FROM 0 TO LENGTH OF state MINUS 2  
                IF SUBSTRING OF state FROM i TO i PLUS 2 EQUALS "++" THEN  
                    SET new_state TO CONCATENATION OF SUBSTRING OF state FROM START TO i, "--", AND SUBSTRING OF state FROM i PLUS 2 TO END  
                    IF NOT can_win_helper(new_state) THEN  
                        SET memo[state] TO True  
                        RETURN True  
            
            SET memo[state] TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_win_helper(currentState)  
    END FUNCTION  
END CLASS
