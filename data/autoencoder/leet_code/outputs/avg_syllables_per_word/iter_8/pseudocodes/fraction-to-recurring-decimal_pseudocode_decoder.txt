CLASS Solution  
    FUNCTION fractionToDecimal(numerator, denominator)  
        IF numerator LESS THAN 0 XOR denominator LESS THAN 0 THEN  
            SET sign TO "-"  
        ELSE  
            SET sign TO empty string  
        END IF  
        
        SET numerator TO absolute value of numerator  
        SET denominator TO absolute value of denominator  
        
        SET quotient AND remainder TO result of dividing numerator BY denominator with remainder  
        SET integer_part TO string representation of quotient  
        
        IF remainder EQUALS 0 THEN  
            RETURN concatenation of sign AND integer_part  
        END IF  
        
        SET seen_remainders TO empty dictionary  
        SET fractional_part TO empty list  
        
        WHILE remainder NOT EQUAL TO 0 DO  
            IF remainder IS IN seen_remainders THEN  
                SET start TO seen_remainders[remainder]  
                SET non_repeating TO concatenation of elements in fractional_part UP TO start  
                SET repeating TO concatenation of elements in fractional_part FROM start TO end  
                RETURN concatenation of sign, integer_part, ".", non_repeating, "(", repeating, ")"  
            END IF  
            
            SET seen_remainders[remainder] TO length of fractional_part  
            
            SET remainder TO remainder MULTIPLIED BY 10  
            SET quotient AND remainder TO result of dividing remainder BY denominator with remainder  
            APPEND string representation of quotient TO fractional_part  
        END WHILE  
        
        RETURN concatenation of sign, integer_part, ".", concatenation of elements in fractional_part  
    END FUNCTION  
END CLASS
