CLASS Solution  
    FUNCTION smallestSufficientTeam(req_skills, people)  
        SET d TO mapping each skill in req_skills TO its index  
        SET m TO length of req_skills  
        SET n TO length of people  
        SET p TO list of zeros with length n  
        FOR i FROM 0 TO n MINUS 1  
            FOR each skill s IN people at index i  
                SET p at i TO p at i BITWISE OR with one shifted LEFT BY d at s  
        SET f TO list of inf with size two raised TO m  
        SET g TO list of zeros with size two raised TO m  
        SET h TO list of zeros with size two raised TO m  
        SET f at 0 TO 0  
        FOR i FROM 0 TO two raised TO m MINUS 1  
            IF f at i EQUALS inf  
                CONTINUE to next iteration  
            FOR j FROM 0 TO n MINUS 1  
                IF f at i PLUS 1 LESS THAN f at i BITWISE OR p at j  
                    SET f at i BITWISE OR p at j TO f at i PLUS 1  
                    SET g at i BITWISE OR p at j TO j  
                    SET h at i BITWISE OR p at j TO i  
        SET i TO two raised TO m MINUS 1  
        SET ans TO empty list  
        WHILE i NOT EQUAL TO 0  
            APPEND g at i TO ans  
            SET i TO h at i  
        RETURN ans  
    END FUNCTION  
END CLASS
