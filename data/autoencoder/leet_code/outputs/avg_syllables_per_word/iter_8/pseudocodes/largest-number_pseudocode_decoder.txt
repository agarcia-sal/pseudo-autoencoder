CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO list of string representations of each number in nums  
        
        FUNCTION compare(x, y)  
            IF concatenation of x and y IS GREATER THAN concatenation of y and x  
                RETURN -1  
            ELSE IF concatenation of x and y IS LESS THAN concatenation of y and x  
                RETURN 1  
            ELSE  
                RETURN 0  
        END FUNCTION  
        
        SORT nums_str USING compare as comparator  
        
        IF first element of nums_str IS EQUAL TO "0"  
            RETURN "0"  
        
        RETURN concatenation of all elements in nums_str  
    END FUNCTION  
END CLASS
