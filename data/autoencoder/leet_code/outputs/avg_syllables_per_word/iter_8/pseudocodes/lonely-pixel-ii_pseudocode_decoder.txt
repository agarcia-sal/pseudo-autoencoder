CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture IS empty OR first element of picture IS empty  
            RETURN 0  
          
        SET m TO LENGTH OF picture  
        SET n TO LENGTH OF first element of picture  
        SET row_count TO list of zeros of length m  
        SET col_count TO list of zeros of length n  
        SET row_patterns TO empty mapping from pattern to integer  
          
        FOR each r FROM 0 TO m MINUS 1  
            SET pattern TO concatenation of elements in picture at row r  
            INCREMENT value in row_patterns for pattern BY 1  
            FOR each c FROM 0 TO n MINUS 1  
                IF element at picture[r][c] IS equal to 'B'  
                    INCREMENT row_count at r BY 1  
                    INCREMENT col_count at c BY 1  
          
        SET lonely_pixels TO 0  
          
        FOR each r FROM 0 TO m MINUS 1  
            FOR each c FROM 0 TO n MINUS 1  
                IF element at picture[r][c] IS equal to 'B'  
                    SET pattern TO concatenation of elements in picture at row r  
                    IF row_count at r IS equal to target AND col_count at c IS equal to target AND row_patterns at pattern IS equal to target  
                        INCREMENT lonely_pixels BY 1  
          
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
