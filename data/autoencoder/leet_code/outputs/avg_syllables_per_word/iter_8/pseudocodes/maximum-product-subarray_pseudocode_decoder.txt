CLASS Solution  
    FUNCTION maxProduct(nums)  
        IF nums IS empty  
            RETURN 0  
        END IF  
        
        SET max_product TO first element of nums  
        SET min_product TO first element of nums  
        SET result TO first element of nums  
        
        FOR each num IN nums starting from second element  
            IF num LESS THAN 0  
                SWAP max_product AND min_product  
            END IF  
            
            SET max_product TO the greater value between num AND max_product multiplied BY num  
            SET min_product TO the lesser value between num AND min_product multiplied BY num  
            SET result TO the greater value between result AND max_product  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
