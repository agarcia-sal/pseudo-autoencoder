CLASS Solution  
    FUNCTION consecutiveNumbersSum(n)  
        SET count TO 0  
        FOR k FROM 1 TO n  
            IF remainder of (2 multiplied by n) DIVIDED BY k EQUALS 0 THEN  
                SET m TO ( (2 multiplied by n DIVIDED BY k) MINUS k PLUS 1 ) DIVIDED BY 2  
                IF m GREATER THAN 0 AND m EQUALS integer part of m THEN  
                    INCREMENT count BY 1  
                END IF  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
