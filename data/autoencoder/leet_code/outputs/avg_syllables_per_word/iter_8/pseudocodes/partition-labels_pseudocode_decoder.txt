CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO DICTIONARY mapping each character IN s TO its last index  
        SET start TO 0  
        SET end TO 0  
        SET partition_sizes TO EMPTY LIST  
        FOR each index i AND character char IN s  
            SET end TO the GREATER OF end AND last_occurrence at char  
            IF i EQUALS end  
                APPEND end MINUS start PLUS 1 TO partition_sizes  
                SET start TO i PLUS 1  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
