CLASS Solution  
    FUNCTION recoverArray(nums)  
        SORT nums  
        SET n TO LENGTH OF nums  
        SET first TO nums AT INDEX 0  
        
        FOR i FROM 1 TO n MINUS 1  
            SET k TO (nums AT INDEX i MINUS first) DIVIDED BY 2  
            IF k LESS THAN OR EQUAL TO 0 OR first PLUS 2 TIMES k NOT EQUAL TO nums AT INDEX i  
                CONTINUE TO NEXT i  
            
            SET count TO COUNTER OF nums  
            SET arr TO EMPTY LIST  
            
            FOR each x IN nums  
                IF count AT x EQUAL TO 0  
                    CONTINUE TO NEXT x  
                IF count AT (x PLUS 2 TIMES k) EQUAL TO 0  
                    BREAK FROM LOOP  
                APPEND (x PLUS k) TO arr  
                DECREMENT count AT x BY 1  
                DECREMENT count AT (x PLUS 2 TIMES k) BY 1  
            
            IF LENGTH OF arr EQUAL TO n DIVIDED BY 2  
                RETURN arr  
        
        RETURN EMPTY LIST  
END CLASS
