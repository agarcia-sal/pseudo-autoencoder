CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO 0  
        SET cows TO 0  
        SET secret_count TO new dictionary with default integer 0  
        SET guess_count TO new dictionary with default integer 0  

        FOR each pair s, g IN secret and guess zipped together  
            IF s EQUALS g THEN  
                INCREMENT bulls BY 1  
            ELSE  
                INCREMENT secret_count at key s BY 1  
                INCREMENT guess_count at key g BY 1  
        
        FOR each digit IN secret_count keys  
            INCREMENT cows BY minimum of secret_count at digit AND guess_count at digit  

        RETURN string formed by bulls concatenated with "A" concatenated with cows concatenated with "B"  
    END FUNCTION  
END CLASS
