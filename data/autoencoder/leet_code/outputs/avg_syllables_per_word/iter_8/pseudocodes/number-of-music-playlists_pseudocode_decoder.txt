CLASS Solution  
    FUNCTION numMusicPlaylists(n, goal, k)  
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7  
        INITIALIZE dp AS A LIST OF LISTS WITH SIZE (goal PLUS 1) BY (n PLUS 1) FILLED WITH 0  
        SET dp AT 0 AT 0 TO 1  
        
        FOR i FROM 1 TO goal  
            FOR j FROM 1 TO n  
                SET dp AT i AT j TO dp AT (i MINUS 1) AT (j MINUS 1) MULTIPLIED BY (n MINUS (j MINUS 1))  
                SET dp AT i AT j TO dp AT i AT j MODULO MOD  
                
                IF j GREATER THAN k THEN  
                    INCREMENT dp AT i AT j BY dp AT (i MINUS 1) AT j MULTIPLIED BY (j MINUS k)  
                    SET dp AT i AT j TO dp AT i AT j MODULO MOD  
        RETURN dp AT goal AT n  
    END FUNCTION  
END CLASS
