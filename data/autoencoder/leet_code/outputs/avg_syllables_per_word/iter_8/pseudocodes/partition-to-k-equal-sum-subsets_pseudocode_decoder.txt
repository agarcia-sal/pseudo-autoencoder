CLASS Solution  
    FUNCTION canPartitionKSubsets(nums, k)  
        SET total_sum TO sum of all elements in nums  
        IF total_sum modulo k IS NOT EQUAL TO 0  
            RETURN False  
        SET target TO total_sum divided by k  
        SORT nums in descending order  
        
        FUNCTION can_partition(index, k, current_sum, used)  
            IF k IS EQUAL TO 0  
                RETURN True  
            IF current_sum IS EQUAL TO target  
                RETURN can_partition(0, k minus 1, 0, used)  
            
            FOR i FROM index TO length of nums minus 1  
                IF used at position i IS False AND current_sum plus nums at i IS LESS THAN OR EQUAL TO target  
                    SET used at position i TO True  
                    IF can_partition(i plus 1, k, current_sum plus nums at i, used)  
                        RETURN True  
                    SET used at position i TO False  
            RETURN False  
        END FUNCTION  
        
        SET used TO list of False with length equal to length of nums  
        RETURN can_partition(0, k, 0, used)  
    END FUNCTION  
END CLASS
