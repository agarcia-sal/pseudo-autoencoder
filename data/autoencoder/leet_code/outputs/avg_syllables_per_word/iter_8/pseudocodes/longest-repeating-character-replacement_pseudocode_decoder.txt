CLASS Solution  
    FUNCTION characterReplacement(s, k)  
        SET max_length TO 0  
        SET max_count TO 0  
        SET left TO 0  
        SET count TO a mapping with default integer value  
  
        FOR right FROM 0 TO LENGTH OF s MINUS 1  
            INCREMENT count AT s[right] BY 1  
            SET max_count TO the greater value BETWEEN max_count AND count AT s[right]  
  
            IF the value OF (right MINUS left PLUS 1) MINUS max_count IS GREATER THAN k  
                DECREMENT count AT s[left] BY 1  
                INCREMENT left BY 1  
            END IF  
  
            SET max_length TO the greater value BETWEEN max_length AND (right MINUS left PLUS 1)  
        END FOR  
  
        RETURN max_length  
    END FUNCTION  
END CLASS
