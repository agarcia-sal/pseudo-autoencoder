CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO 0  
            SET start TO 0  
            SET specials TO empty list  
            
            FOR each index i, character char IN s  
                IF char EQUALS '1' THEN  
                    INCREMENT count BY 1  
                ELSE  
                    DECREMENT count BY 1  
                END IF  
                
                IF count EQUALS 0 THEN  
                    APPEND concatenation of '1', dfs(substring of s from start plus 1 TO i minus 1), and '0' TO specials  
                    SET start TO i plus 1  
                END IF  
            END FOR  
            
            SORT specials IN reverse lexicographical order  
            RETURN concatenation of all elements in specials  
        END FUNCTION  
        
        RETURN dfs(s)  
    END FUNCTION  
END CLASS
