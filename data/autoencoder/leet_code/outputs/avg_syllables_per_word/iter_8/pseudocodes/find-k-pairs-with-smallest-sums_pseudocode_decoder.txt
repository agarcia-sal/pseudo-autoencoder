CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k IS LESS THAN OR EQUAL TO 0 THEN  
            RETURN EMPTY LIST  
        END IF  

        SET min_heap TO EMPTY LIST  
        FOR i FROM 0 TO MINIMUM OF k AND LENGTH OF nums1 MINUS 1  
            PUSH INTO min_heap THE TUPLE OF SUM OF nums1 AT i AND nums2 AT 0, i, AND 0  
        END FOR  

        SET result TO EMPTY LIST  
        WHILE min_heap IS NOT EMPTY AND LENGTH OF result IS LESS THAN k  
            POP THE SMALLEST ELEMENT FROM min_heap INTO _, i, j  
            APPEND LIST OF nums1 AT i AND nums2 AT j TO result  
            IF j PLUS 1 IS LESS THAN LENGTH OF nums2 THEN  
                PUSH INTO min_heap THE TUPLE OF SUM OF nums1 AT i AND nums2 AT j PLUS 1, i, AND j PLUS 1  
            END IF  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
