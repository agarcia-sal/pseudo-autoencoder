CLASS Solution  
    FUNCTION palindromePartition(s, k)  
        FUNCTION min_changes_to_palindrome(sub)  
            SET i TO 0  
            SET j TO LENGTH OF sub MINUS 1  
            SET changes TO 0  
            WHILE i LESS THAN j  
                IF CHARACTER AT sub WITH INDEX i NOT EQUAL TO CHARACTER AT sub WITH INDEX j  
                    INCREMENT changes BY 1  
                INCREMENT i BY 1  
                DECREMENT j BY 1  
            END WHILE  
            RETURN changes  
        END FUNCTION

        SET n TO LENGTH OF s  
        INITIALIZE dp AS A TWO-DIMENSIONAL LIST WITH n PLUS 1 ROWS AND k PLUS 1 COLUMNS FILLED WITH INFINITY  
        SET dp AT INDEX 0 AND 0 TO 0  

        FOR i FROM 1 TO n INCLUSIVE  
            FOR j FROM 1 TO MINIMUM OF k AND i INCLUSIVE  
                FOR start FROM 0 TO i MINUS 1  
                    SET changes TO CALL min_changes_to_palindrome WITH ARGUMENT SUBSTRING OF s FROM start TO i  
                    SET dp AT i AND j TO THE MINIMUM OF dp AT i AND j AND dp AT start AND j MINUS 1 PLUS changes  
                END FOR  
            END FOR  
        END FOR  

        RETURN dp AT n AND k  
    END FUNCTION  
END CLASS
