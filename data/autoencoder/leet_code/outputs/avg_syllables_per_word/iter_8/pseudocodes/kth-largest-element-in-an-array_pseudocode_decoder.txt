CLASS Solution  
    FUNCTION findKthLargest(nums, k)  
        SET min_heap TO empty list  
        
        FOR each num IN nums  
            ADD num TO min_heap USING heap push  
            IF LENGTH OF min_heap GREATER THAN k  
                REMOVE smallest element FROM min_heap USING heap pop  
        
        RETURN smallest element FROM min_heap  
    END FUNCTION  
END CLASS
