CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO NEW TreeNode WITH values AT index 0  
    SET i TO 1  
    SET queue TO NEW empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO queue REMOVE FIRST ELEMENT  
        IF i IS LESS THAN LENGTH OF values AND values AT index i IS NOT None  
            SET node.left TO NEW TreeNode WITH values AT index i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values AT index i IS NOT None  
            SET node.right TO NEW TreeNode WITH values AT index i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder(self, root)  
        IF root IS None  
            RETURN empty list  
        SET column_table TO empty dictionary WITH default empty list  
        SET queue TO deque CONTAINING tuple (root, 0)  
        SET min_column TO 0  
        SET max_column TO 0  
        WHILE queue IS NOT empty  
            SET node, column_index TO queue REMOVE FIRST ELEMENT  
            IF node IS NOT None  
                APPEND node.val TO column_table AT key column_index  
                SET min_column TO MINIMUM OF min_column AND column_index  
                SET max_column TO MAXIMUM OF max_column AND column_index  
                APPEND tuple (node.left, column_index MINUS 1) TO queue  
                APPEND tuple (node.right, column_index PLUS 1) TO queue  
        SET result TO empty list  
        FOR x FROM min_column TO max_column INCLUSIVE  
            APPEND column_table AT key x TO result  
        RETURN result  
    END FUNCTION  
END CLASS
