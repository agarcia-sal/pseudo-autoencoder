CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 1  
            PUSH tuple of (element at index 0 of nums at i, i, 0) INTO min_heap  
            IF element at index 0 of nums at i GREATER THAN max_value THEN  
                SET max_value TO element at index 0 of nums at i  
        
        SET smallest_range TO infinity  
        SET result TO list containing -100000 and 100000  
        
        WHILE min_heap IS NOT empty  
            POP smallest tuple FROM min_heap INTO min_value, list_index, element_index  
            
            IF max_value MINUS min_value LESS THAN smallest_range THEN  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO list containing min_value and max_value  
            
            IF element_index PLUS 1 LESS THAN LENGTH OF nums at list_index THEN  
                SET next_value TO element at index (element_index PLUS 1) of nums at list_index  
                PUSH tuple of (next_value, list_index, element_index PLUS 1) INTO min_heap  
                IF next_value GREATER THAN max_value THEN  
                    SET max_value TO next_value  
            ELSE  
                BREAK  
        
        RETURN result  
    END FUNCTION  
END CLASS
