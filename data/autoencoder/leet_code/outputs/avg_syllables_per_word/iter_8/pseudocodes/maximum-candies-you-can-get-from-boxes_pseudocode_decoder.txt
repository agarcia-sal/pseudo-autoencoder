CLASS Solution  
    FUNCTION maxCandies(status, candies, keys, containedBoxes, initialBoxes)  
        SET boxes_we_have TO set containing elements of initialBoxes  
        SET keys_we_have TO empty set  
        SET total_candies TO 0  
        
        SET progress_made TO true  
        
        WHILE progress_made  
            SET progress_made TO false  
            FOR each box IN list of boxes_we_have  
                IF status of box EQUALS 1 OR box IN keys_we_have  
                    INCREMENT total_candies BY candies of box  
                    ADD all elements of keys of box TO keys_we_have  
                    ADD all elements of containedBoxes of box TO boxes_we_have  
                    REMOVE box FROM boxes_we_have  
                    SET progress_made TO true  
        RETURN total_candies  
    END FUNCTION  
END CLASS
