CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1, str2)  
        SET m TO LENGTH OF str1  
        SET n TO LENGTH OF str2  
        
        INITIALIZE dp TABLE WITH DIMENSIONS (m PLUS 1) BY (n PLUS 1) FILLED WITH 0  
        
        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF CHARACTER AT POSITION i MINUS 1 IN str1 EQUAL TO CHARACTER AT POSITION j MINUS 1 IN str2  
                    SET dp AT POSITION i, j TO dp AT POSITION i MINUS 1, j MINUS 1 PLUS 1  
                ELSE  
                    SET dp AT POSITION i, j TO MAXIMUM OF dp AT POSITION i MINUS 1, j AND dp AT POSITION i, j MINUS 1  
        
        SET i TO m  
        SET j TO n  
        INITIALIZE scs AS EMPTY LIST  
        
        WHILE i GREATER THAN 0 AND j GREATER THAN 0  
            IF CHARACTER AT POSITION i MINUS 1 IN str1 EQUAL TO CHARACTER AT POSITION j MINUS 1 IN str2  
                APPEND CHARACTER AT POSITION i MINUS 1 IN str1 TO scs  
                DECREMENT i BY 1  
                DECREMENT j BY 1  
            ELSE IF dp AT POSITION i MINUS 1, j GREATER THAN dp AT POSITION i, j MINUS 1  
                APPEND CHARACTER AT POSITION i MINUS 1 IN str1 TO scs  
                DECREMENT i BY 1  
            ELSE  
                APPEND CHARACTER AT POSITION j MINUS 1 IN str2 TO scs  
                DECREMENT j BY 1  
        
        WHILE i GREATER THAN 0  
            APPEND CHARACTER AT POSITION i MINUS 1 IN str1 TO scs  
            DECREMENT i BY 1  
        
        WHILE j GREATER THAN 0  
            APPEND CHARACTER AT POSITION j MINUS 1 IN str2 TO scs  
            DECREMENT j BY 1  
        
        REVERSE scs  
        
        RETURN CONCATENATION OF ELEMENTS IN scs  
    END FUNCTION  
END CLASS
