CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO list of characters representing n  
        SET length TO count of digits  
        SET marker TO length  
        FOR i FROM length MINUS 1 DOWNTO 1  
            IF digits at position i IS LESS THAN digits at position i MINUS 1  
                SET digits at position i MINUS 1 TO string of (integer value of digits at position i MINUS 1 MINUS 1)  
                SET marker TO i  
        FOR i FROM marker TO length MINUS 1  
            SET digits at position i TO '9'  
        RETURN integer formed by concatenating digits  
    END FUNCTION  
END CLASS
