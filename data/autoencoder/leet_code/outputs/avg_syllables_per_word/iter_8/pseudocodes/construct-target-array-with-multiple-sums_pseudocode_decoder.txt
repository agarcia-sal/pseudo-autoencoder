CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO list of negative values from target  
        CONVERT max_heap TO heap structure  
        
        SET total_sum TO sum of target  
        
        WHILE first element of max_heap IS LESS THAN negative one  
            SET largest TO negative value of removing top element from max_heap  
            SET rest_sum TO total_sum MINUS largest  
            
            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum IS EQUAL TO zero  
                RETURN False  
            
            SET new_value TO largest MODULO rest_sum  
            IF new_value IS EQUAL TO zero AND rest_sum IS NOT EQUAL TO one  
                RETURN False  
            
            SET total_sum TO rest_sum PLUS new_value  
            ADD negative new_value TO max_heap  
        
        RETURN True  
    END FUNCTION  
END CLASS
