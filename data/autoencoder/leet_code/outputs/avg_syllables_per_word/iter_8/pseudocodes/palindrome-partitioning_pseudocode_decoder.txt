CLASS Solution  
    FUNCTION partition(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS reverse of sub  
        END FUNCTION  
        
        FUNCTION backtrack(start, path)  
            IF start EQUALS length of s THEN  
                APPEND path TO result  
                RETURN  
            END IF  
            FOR end FROM start PLUS 1 TO length of s PLUS 1  
                IF is_palindrome(substring of s FROM start TO end) THEN  
                    CALL backtrack WITH end AND path APPENDED BY substring of s FROM start TO end  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL backtrack WITH 0 AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
