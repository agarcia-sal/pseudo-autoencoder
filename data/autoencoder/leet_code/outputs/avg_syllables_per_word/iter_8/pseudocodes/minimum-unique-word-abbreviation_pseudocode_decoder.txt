CLASS Solution  
    FUNCTION minAbbreviation(target, dictionary)  
        FUNCTION is_valid_abbr(s, abbr)  
            SET i TO 0  
            SET j TO 0  
            WHILE i LESS THAN LENGTH OF s AND j LESS THAN LENGTH OF abbr  
                IF abbr[j] IS A DIGIT THEN  
                    IF abbr[j] EQUALS '0' THEN  
                        RETURN False  
                    SET shift TO 0  
                    WHILE j LESS THAN LENGTH OF abbr AND abbr[j] IS A DIGIT  
                        SET shift TO shift MULTIPLIED BY 10 PLUS numeric value of abbr[j]  
                        INCREMENT j BY 1  
                    INCREMENT i BY shift  
                ELSE  
                    IF s[i] NOT EQUAL TO abbr[j] THEN  
                        RETURN False  
                    INCREMENT i BY 1  
                    INCREMENT j BY 1  
            RETURN i EQUALS LENGTH OF s AND j EQUALS LENGTH OF abbr  
        END FUNCTION  

        FUNCTION generate_abbrs(word, length, abbr, index, skip_count)  
            IF index EQUALS length THEN  
                IF skip_count GREATER THAN 0 THEN  
                    APPEND string representation of skip_count TO abbr  
                APPEND abbr TO abbrs  
                RETURN  
            CALL generate_abbrs WITH (word, length, abbr, index INCREMENTED BY 1, skip_count INCREMENTED BY 1)  
            IF skip_count GREATER THAN 0 THEN  
                APPEND string representation of skip_count TO abbr  
            CALL generate_abbrs WITH (word, length, abbr APPENDED WITH word[index], index INCREMENTED BY 1, 0)  
        END FUNCTION  

        FUNCTION find_unique_abbr()  
            FOR length FROM 1 TO LENGTH OF target  
                CLEAR abbrs  
                CALL generate_abbrs WITH (target, LENGTH OF target, empty string, 0, 0)  
                SET min_abbr TO None  
                SET min_length TO infinity  
                FOR EACH abbr IN abbrs  
                    IF FOR ALL word IN filtered_dict THE RESULT OF is_valid_abbr(word, abbr) IS False THEN  
                        IF LENGTH OF abbr LESS THAN min_length THEN  
                            SET min_length TO LENGTH OF abbr  
                            SET min_abbr TO abbr  
                IF min_abbr IS NOT None THEN  
                    RETURN min_abbr  
        END FUNCTION  

        SET filtered_dict TO list of words IN dictionary WHERE LENGTH OF word EQUALS LENGTH OF target  
        INITIALIZE abbrs AS EMPTY LIST  
        RETURN find_unique_abbr()  
    END FUNCTION  
END CLASS
