CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        IF LENGTH OF nums LESS THAN 3 THEN  
            RETURN 0  
        END IF  
        
        SET total_slices TO 0  
        SET current_length TO 2  
        
        FOR i FROM 2 TO LENGTH OF nums MINUS 1  
            IF nums AT i MINUS nums AT i MINUS 1 EQUALS nums AT i MINUS 1 MINUS nums AT i MINUS 2 THEN  
                INCREMENT current_length BY 1  
                INCREMENT total_slices BY current_length MINUS 2  
            ELSE  
                SET current_length TO 2  
            END IF  
        END FOR  
        
        RETURN total_slices  
    END FUNCTION  
END CLASS
