CLASS Solution  
    FUNCTION rearrangeSticks(n, k)  
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7  
        INITIALIZE dp AS A LIST OF LISTS WITH DIMENSIONS (n PLUS 1) BY (k PLUS 1) FILLED WITH ZEROS  
        SET dp AT 0 INDEX 0 TO 1  
        FOR i FROM 1 TO n INCLUSIVE  
            FOR j FROM 1 TO k INCLUSIVE  
                SET dp AT i INDEX j TO (dp AT (i MINUS 1) INDEX (j MINUS 1) PLUS (i MINUS 1) MULTIPLIED BY dp AT (i MINUS 1) INDEX j) MODULO MOD  
        RETURN dp AT n INDEX k  
    END FUNCTION  
END CLASS
