CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS plus sign  
                        APPEND l plus r TO results  
                    ELSE IF operator EQUALS minus sign  
                        APPEND l minus r TO results  
                    ELSE IF operator EQUALS multiplication sign  
                        APPEND l multiplied BY r TO results  
            RETURN results  
        END FUNCTION  
        
        FUNCTION helper(sub_expr)  
            IF sub_expr IS composed only of digits  
                RETURN list containing integer value of sub_expr  
            
            SET results TO empty list  
            FOR each index i FROM zero TO length of sub_expr minus one  
                IF character at sub_expr position i IS one of plus sign, minus sign, multiplication sign  
                    SET left_results TO helper of substring of sub_expr from start TO i exclusive  
                    SET right_results TO helper of substring of sub_expr from i plus one TO end  
                    EXTEND results BY compute with left_results, right_results AND character at sub_expr position i  
            RETURN results  
        END FUNCTION  
        
        RETURN helper with expression  
    END FUNCTION  
END CLASS
