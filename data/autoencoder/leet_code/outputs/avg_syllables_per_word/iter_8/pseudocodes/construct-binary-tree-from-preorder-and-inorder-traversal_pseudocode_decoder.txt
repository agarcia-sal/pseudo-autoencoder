CLASS TreeNode  
    FUNCTION __init__(self, val SET TO 0, left SET TO None, right SET TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO TreeNode(values at position 0)  
    SET i TO 1  
    INITIALIZE queue AS empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element removed from queue  
        IF i IS LESS THAN length of values AND values at position i IS NOT None  
            SET node.left TO TreeNode(values at position i)  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN length of values AND values at position i IS NOT None  
            SET node.right TO TreeNode(values at position i)  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF preorder IS empty OR inorder IS empty  
            RETURN None  
        SET root_val TO preorder at position 0  
        SET root TO TreeNode(root_val)  
        SET root_index_inorder TO position of root_val IN inorder  
        SET root.left TO self.buildTree(preorder elements from position 1 TO root_index_inorder inclusive, inorder elements BEFORE root_index_inorder)  
        SET root.right TO self.buildTree(preorder elements AFTER root_index_inorder, inorder elements AFTER root_index_inorder)  
        RETURN root  
    END FUNCTION  
END CLASS
