CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        SET graph TO a default dictionary mapping to empty list  
        SET indegree TO list of zeros with length numCourses  
        
        FOR each pair course, prereq IN prerequisites  
            APPEND course TO graph at key prereq  
            INCREMENT indegree at index course BY 1  
        
        SET queue TO a deque containing all courses where indegree is equal to 0  
        
        WHILE queue is not empty  
            SET current_course TO the first element removed from queue  
            DECREMENT numCourses BY 1  
            
            FOR each next_course IN graph at key current_course  
                DECREMENT indegree at index next_course BY 1  
                IF indegree at index next_course is equal to 0  
                    APPEND next_course TO queue  
        
        RETURN whether numCourses is equal to 0  
    END FUNCTION  
END CLASS
