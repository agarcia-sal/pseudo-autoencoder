CLASS Solution  
    FUNCTION maxJumps(arr, d)  
        FUNCTION dp(i) WITH CACHE  
            SET max_jumps TO 1  
            FOR x FROM 1 TO d  
                IF i PLUS x IS LESS THAN LENGTH OF arr THEN  
                    IF arr AT (i PLUS x) IS GREATER THAN OR EQUAL TO arr AT i THEN  
                        BREAK LOOP  
                    SET max_jumps TO THE GREATER OF max_jumps AND (1 PLUS dp(i PLUS x))  
            FOR x FROM 1 TO d  
                IF i MINUS x IS GREATER THAN OR EQUAL TO 0 THEN  
                    IF arr AT (i MINUS x) IS GREATER THAN OR EQUAL TO arr AT i THEN  
                        BREAK LOOP  
                    SET max_jumps TO THE GREATER OF max_jumps AND (1 PLUS dp(i MINUS x))  
            RETURN max_jumps  
        RETURN THE GREATER VALUE FROM APPLYING dp FUNCTION TO EACH INDEX i IN RANGE LENGTH OF arr  
    END FUNCTION  
END CLASS
