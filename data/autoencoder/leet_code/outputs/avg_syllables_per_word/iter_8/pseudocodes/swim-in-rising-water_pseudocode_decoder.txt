CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO LENGTH OF grid  
        SET visited TO empty set  
        SET min_heap TO list containing tuple of (grid at row 0 column 0, 0, 0)  
        ADD tuple (0, 0) TO visited  
        
        SET directions TO list of tuples (0, 1), (1, 0), (0, -1), (-1, 0)  
        
        WHILE min_heap is not empty  
            REMOVE and RETURN smallest element FROM min_heap INTO t, i, j  
            IF i EQUALS n MINUS 1 AND j EQUALS n MINUS 1 THEN  
                RETURN t  
            END IF  
            
            FOR each di, dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO 0 AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO 0 AND nj LESS THAN n AND tuple (ni, nj) NOT IN visited THEN  
                    ADD tuple (ni, nj) TO visited  
                    INSERT tuple (maximum of t AND grid at ni nj, ni, nj) INTO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN -1  
    END FUNCTION  
END CLASS
