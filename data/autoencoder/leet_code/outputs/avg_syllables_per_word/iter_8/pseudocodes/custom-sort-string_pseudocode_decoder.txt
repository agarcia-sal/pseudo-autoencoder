CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO empty dictionary  
        FOR each character c AND its index i IN order  
            SET d AT c TO i  
        SET sorted_list TO s sorted by key WHERE key IS value from d AT character OR 0 if not found  
        SET result TO concatenation of characters in sorted_list  
        RETURN result  
    END FUNCTION  
END CLASS
