CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS LENGTH OF nums THEN  
                IF current_sum EQUALS target THEN  
                    RETURN 1  
                ELSE  
                    RETURN 0  
            END IF  
            IF memo CONTAINS KEY (index, current_sum) THEN  
                RETURN memo VALUE FOR (index, current_sum)  
            END IF  
            SET positive TO dfs(index INCREMENTED BY 1, current_sum INCREMENTED BY nums AT index, memo)  
            SET negative TO dfs(index INCREMENTED BY 1, current_sum DECREMENTED BY nums AT index, memo)  
            SET memo VALUE FOR (index, current_sum) TO positive PLUS negative  
            RETURN memo VALUE FOR (index, current_sum)  
        END FUNCTION  
        RETURN dfs(0, 0, EMPTY MAP)  
    END FUNCTION  
END CLASS
