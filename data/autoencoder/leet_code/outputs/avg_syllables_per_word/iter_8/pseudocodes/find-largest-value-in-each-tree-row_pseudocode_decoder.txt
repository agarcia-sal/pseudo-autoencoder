CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO TreeNode(values at position 0)  
    SET i TO 1  
    INITIALIZE queue AS deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO queue REMOVE first element  
        IF i LESS THAN LENGTH OF values AND values at position i IS NOT None  
            SET node.left TO TreeNode(values at position i)  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values at position i IS NOT None  
            SET node.right TO TreeNode(values at position i)  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues(self, root)  
        IF root IS None  
            RETURN empty list  
        INITIALIZE queue AS deque containing root  
        INITIALIZE largest_values AS empty list  
        WHILE queue IS NOT empty  
            SET level_size TO LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR each IN RANGE level_size  
                SET node TO queue REMOVE first element  
                IF node.val GREATER THAN max_value  
                    SET max_value TO node.val  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
            APPEND max_value TO largest_values  
        RETURN largest_values  
    END FUNCTION  
END CLASS
