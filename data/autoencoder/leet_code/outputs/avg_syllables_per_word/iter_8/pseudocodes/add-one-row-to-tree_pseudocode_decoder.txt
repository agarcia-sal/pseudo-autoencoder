CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode with value values at position zero  
    SET i TO 1  
    SET queue TO empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from front of queue  
        IF i LESS THAN length of values AND values at position i IS NOT None  
            SET node.left TO new TreeNode with value values at position i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values at position i IS NOT None  
            SET node.right TO new TreeNode with value values at position i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow(self, root, val, depth)  
        IF depth IS EQUAL TO 1  
            RETURN new TreeNode with value val AND left child root  
        SET queue TO empty queue  
        APPEND tuple containing root AND 1 TO queue  
        WHILE queue IS NOT empty  
            SET node AND level TO element removed from front of queue  
            IF level IS EQUAL TO depth MINUS 1  
                SET node.left TO new TreeNode with value val AND left child node.left  
                SET node.right TO new TreeNode with value val AND right child node.right  
            ELSE IF level LESS THAN depth MINUS 1  
                IF node.left EXISTS  
                    APPEND tuple containing node.left AND level PLUS 1 TO queue  
                IF node.right EXISTS  
                    APPEND tuple containing node.right AND level PLUS 1 TO queue  
        RETURN root  
    END FUNCTION  
END CLASS
