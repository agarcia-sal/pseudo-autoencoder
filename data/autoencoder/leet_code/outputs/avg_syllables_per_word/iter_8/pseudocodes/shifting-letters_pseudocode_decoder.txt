CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO 0  
        FOR i FROM LENGTH OF shifts MINUS 1 DOWNTO 0  
            INCREMENT total_shifts BY shifts AT i  
            SET shifts AT i TO total_shifts  
        END FOR  

        SET result TO empty list  
        FOR i FROM 0 TO LENGTH OF s MINUS 1  
            SET original_position TO numeric position of character s AT i RELATIVE TO 'a'  
            SET shifted_position TO original_position PLUS shifts AT i  
            SET wrapped_position TO shifted_position MODULO 26  
            SET new_char TO character REPRESENTED BY wrapped_position OFFSET FROM 'a'  
            APPEND new_char TO result  
        END FOR  

        RETURN concatenation of all characters IN result  
    END FUNCTION  
END CLASS
