CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO dictionary with key 0 and value 0  

        FOR each rod IN rods  
            SET new_dp TO copy of dp  
            FOR each key d AND value y IN dp  
                SET greater_key TO d PLUS rod  
                SET new_dp at greater_key TO maximum of new_dp at greater_key OR 0 AND y  
                
                SET difference_key TO absolute value of d MINUS rod  
                SET new_dp at difference_key TO maximum of new_dp at difference_key OR 0 AND y PLUS minimum of d AND rod  
            END FOR  
            SET dp TO new_dp  
        END FOR  

        RETURN dp at key 0  
    END FUNCTION  
END CLASS
