CLASS Solution  
    FUNCTION integerBreak(n)  
        SET dp TO list of zeros with length n plus one  
        SET dp at index 1 TO 1  
  
        FOR i FROM 2 TO n  
            SET max_product TO 0  
            FOR j FROM 1 TO i minus 1  
                SET max_product TO the greatest among max_product, j multiplied by dp at index i minus j, and j multiplied by i minus j  
            SET dp at index i TO max_product  
  
        RETURN dp at index n  
    END FUNCTION  
END CLASS
