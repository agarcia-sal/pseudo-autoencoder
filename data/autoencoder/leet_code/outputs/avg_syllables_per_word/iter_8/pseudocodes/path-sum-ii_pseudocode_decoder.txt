CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode with value values at index 0  
    SET i TO 1  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO queue removed from front  
        IF i LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.left TO new TreeNode with value values at index i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.right TO new TreeNode with value values at index i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION dfs(node, current_sum, path)  
            IF node IS None  
                RETURN  
            APPEND node.val TO path  
            SET current_sum TO current_sum PLUS node.val  
            IF node.left IS None AND node.right IS None AND current_sum EQUALS targetSum  
                APPEND copy of path TO result  
            CALL dfs(node.left, current_sum, path)  
            CALL dfs(node.right, current_sum, path)  
            REMOVE last element FROM path  
        END FUNCTION  
        SET result TO empty list  
        CALL dfs(root, 0, empty list)  
        RETURN result  
    END FUNCTION  
END CLASS
