CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
  
        FOR each char IN s  
            IF char is a digit THEN  
                APPEND char TO d  
            ELSE IF char is an alphabetic character THEN  
                APPEND char TO a  
            ELSE IF char equals '[' THEN  
                APPEND tuple of (a, d converted to integer) TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char equals ']' THEN  
                SET p, n TO last element popped from stack  
                SET a TO p concatenated WITH a repeated n times  
        RETURN a  
    END FUNCTION  
END CLASS
