CLASS Solution  
    FUNCTION bestRotation(nums)  
        SET n TO LENGTH OF nums  
        SET delta TO LIST OF n ZEROS  
        
        FOR each i, num IN ENUMERATE nums  
            IF num LESS THAN OR EQUAL TO i  
                INCREMENT delta AT POSITION 0 BY 1  
                IF i MINUS num PLUS 1 LESS THAN n  
                    DECREMENT delta AT POSITION i MINUS num PLUS 1 BY 1  
            END IF  
            
            IF i PLUS 1 LESS THAN n  
                INCREMENT delta AT POSITION i PLUS 1 BY 1  
                IF i PLUS n MINUS num PLUS 1 LESS THAN n  
                    DECREMENT delta AT POSITION i PLUS n MINUS num PLUS 1 BY 1  
                END IF  
            END IF  
        END FOR  
        
        SET score TO 0  
        SET max_score TO 0  
        SET best_k TO 0  
        
        FOR k FROM 0 TO n MINUS 1  
            INCREMENT score BY delta AT POSITION k  
            IF score GREATER THAN max_score  
                SET max_score TO score  
                SET best_k TO k  
            END IF  
        END FOR  
        
        RETURN best_k  
    END FUNCTION  
END CLASS
