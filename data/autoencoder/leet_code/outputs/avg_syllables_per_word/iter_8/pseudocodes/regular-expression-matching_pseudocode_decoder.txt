CLASS Solution  
    FUNCTION isMatch(s, p)  
        INITIALIZE dp AS two-dimensional list of False with dimensions LENGTH OF s plus 1 BY LENGTH OF p plus 1  
        SET dp at row 0 column 0 TO True  

        FOR j FROM 1 TO LENGTH OF p  
            IF character at position j minus 1 in p EQUALS '*'  
                SET dp at row 0 column j TO value at dp row 0 column j minus 2  
        
        FOR i FROM 1 TO LENGTH OF s  
            FOR j FROM 1 TO LENGTH OF p  
                IF character at position j minus 1 in p EQUALS character at position i minus 1 in s OR character at position j minus 1 in p EQUALS '.'  
                    SET dp at row i column j TO value at dp row i minus 1 column j minus 1  
                ELSE IF character at position j minus 1 in p EQUALS '*'  
                    SET dp at row i column j TO value at dp row i column j minus 2  
                    IF character at position j minus 2 in p EQUALS character at position i minus 1 in s OR character at position j minus 2 in p EQUALS '.'  
                        SET dp at row i column j TO logical OR of dp at row i column j AND dp at row i minus 1 column j  
        
        RETURN value at dp row LENGTH OF s column LENGTH OF p  
    END FUNCTION  
END CLASS
