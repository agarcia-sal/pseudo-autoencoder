CLASS Solution  
    FUNCTION reverseWords(s)  
        REVERSE s ENTIRELY  
        SET start TO 0  
        FOR end FROM 0 TO LENGTH OF s MINUS 1  
            IF s AT end EQUALS space character  
                REVERSE s FROM start TO end MINUS 1  
                SET start TO end PLUS 1  
        REVERSE s FROM start TO END  
    END FUNCTION  
END CLASS
