CLASS Solution  
    FUNCTION findMin(nums)  
        SET left TO 0  
        SET right TO LENGTH OF nums MINUS 1  
        
        IF element at position left in nums IS LESS THAN element at position right in nums  
            RETURN element at position left in nums  
        
        WHILE left IS LESS THAN right  
            SET mid TO left PLUS right DIVIDED BY 2  
            
            IF element at position mid in nums IS GREATER THAN element at position right in nums  
                SET left TO mid PLUS 1  
            ELSE  
                SET right TO mid  
        
        RETURN element at position left in nums  
    END FUNCTION  
END CLASS
