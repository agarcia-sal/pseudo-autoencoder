CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums IS EMPTY  
            RETURN 0  
        END IF  

        SET dp TO empty list  

        FOR each num IN nums  
            SET idx TO insertion position of num in dp using binary search for left insertion point  

            IF idx IS EQUAL TO length of dp  
                APPEND num TO dp  
            ELSE  
                SET element at idx in dp TO num  
            END IF  
        END FOR  

        RETURN length of dp  
    END FUNCTION  
END CLASS
