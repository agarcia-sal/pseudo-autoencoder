CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO dictionary with keys as letters 'a' to 'z' and values initialized to 0  
        SET current_length TO 0  
        
        FOR i FROM 0 TO length of s MINUS 1  
            IF i GREATER THAN 0 AND (ASCII value of s at i MINUS ASCII value of s at i MINUS 1 EQUALS 1 OR s at i EQUALS 'a' AND s at i MINUS 1 EQUALS 'z') THEN  
                INCREMENT current_length BY 1  
            ELSE  
                SET current_length TO 1  
            END IF  
            
            SET max_length at s at i TO maximum of max_length at s at i AND current_length  
        END FOR  
        
        RETURN sum of all values in max_length  
    END FUNCTION  
END CLASS
