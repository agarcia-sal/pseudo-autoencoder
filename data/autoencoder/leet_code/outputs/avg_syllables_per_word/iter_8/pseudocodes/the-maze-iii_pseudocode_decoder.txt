CLASS Solution  
    FUNCTION findShortestWay(maze, ball, hole)  
        SET directions TO list of (0, 1, 'r'), (1, 0, 'd'), (0, -1, 'l'), (-1, 0, 'u')  
        SET m TO LENGTH OF maze  
        SET n TO LENGTH OF maze AT 0  
        SET start TO CONVERT ball TO tuple  
        SET hole TO CONVERT hole TO tuple  
        SET pq TO list containing tuple of 0, empty string, and start  
        SET visited TO empty set  
        
        WHILE pq is not empty  
            REMOVE and SET dist, path, (x, y) TO element from pq with smallest dist  
            
            IF (x, y) EQUALS hole  
                RETURN path  
            
            IF (x, y) IS IN visited  
                CONTINUE  
            
            ADD (x, y) TO visited  
            
            FOR each dx, dy, direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO 0  
                
                WHILE new_x plus dx IS GREATER THAN OR EQUAL TO 0 AND LESS THAN m AND new_y plus dy IS GREATER THAN OR EQUAL TO 0 AND LESS THAN n AND maze AT (new_x plus dx) AND (new_y plus dy) EQUALS 0  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY 1  
                    IF (new_x, new_y) EQUALS hole  
                        BREAK  
                
                IF (new_x, new_y) IS NOT IN visited  
                    ADD tuple of dist plus count, path concatenated with direction, and (new_x, new_y) TO pq  
        
        RETURN string "impossible"  
    END FUNCTION  
END CLASS
