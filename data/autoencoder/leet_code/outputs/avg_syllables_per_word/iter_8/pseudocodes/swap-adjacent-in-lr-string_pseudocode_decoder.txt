CLASS Solution  
    FUNCTION canTransform(start, end)  
        SET start_filtered TO list of pairs (character, index) FROM start WHERE character is not 'X'  
        SET end_filtered TO list of pairs (character, index) FROM end WHERE character is not 'X'  
        
        IF LENGTH OF start_filtered NOT EQUAL TO LENGTH OF end_filtered THEN  
            RETURN False  
        END IF  
        
        FOR each pair (c1, i1), (c2, i2) IN corresponding elements of start_filtered AND end_filtered  
            IF c1 NOT EQUAL TO c2 THEN  
                RETURN False  
            END IF  
            IF c1 EQUALS 'L' AND i1 LESS THAN i2 THEN  
                RETURN False  
            END IF  
            IF c1 EQUALS 'R' AND i1 GREATER THAN i2 THEN  
                RETURN False  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
