CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO sorted list of pairs (wage divided by quality, quality) for each pair in wage and quality  
        SET max_heap TO empty list  
        SET total_quality TO 0  
        SET min_cost TO infinity  
        
        FOR each ratio, q IN workers  
            ADD negative q TO max_heap  
            INCREMENT total_quality BY q  
            
            IF LENGTH OF max_heap GREATER THAN k  
                REMOVE largest quality FROM max_heap AND ADD it TO total_quality  
            END IF  
            
            IF LENGTH OF max_heap EQUAL TO k  
                SET min_cost TO minimum of min_cost AND total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
