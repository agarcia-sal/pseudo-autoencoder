CLASS Solution  
    FUNCTION minCost(costs)  
        IF costs IS empty  
            RETURN 0  
        SET prev_red TO costs AT index 0 AT index 0  
        SET prev_blue TO costs AT index 0 AT index 1  
        SET prev_green TO costs AT index 0 AT index 2  
        FOR i FROM 1 TO LENGTH OF costs MINUS 1  
            SET current_red TO costs AT index i AT index 0 PLUS MINIMUM OF prev_blue AND prev_green  
            SET current_blue TO costs AT index i AT index 1 PLUS MINIMUM OF prev_red AND prev_green  
            SET current_green TO costs AT index i AT index 2 PLUS MINIMUM OF prev_red AND prev_blue  
            SET prev_red TO current_red  
            SET prev_blue TO current_blue  
            SET prev_green TO current_green  
        RETURN MINIMUM OF prev_red, prev_blue, prev_green  
    END FUNCTION  
END CLASS
