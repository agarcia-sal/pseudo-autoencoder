CLASS Solution  
    FUNCTION numberOfBoomerangs(points)  
        SET ans TO 0  
        FOR each p1 IN points  
            SET cnt TO a new Counter  
            FOR each p2 IN points  
                SET dx TO p1 at index 0 MINUS p2 at index 0  
                SET dy TO p1 at index 1 MINUS p2 at index 1  
                SET d TO dx multiplied BY dx PLUS dy multiplied BY dy  
                INCREMENT ans BY cnt VALUE AT d  
                INCREMENT cnt VALUE AT d BY 1  
        SET ans TO ans multiplied BY 2  
        RETURN ans  
    END FUNCTION  
END CLASS
