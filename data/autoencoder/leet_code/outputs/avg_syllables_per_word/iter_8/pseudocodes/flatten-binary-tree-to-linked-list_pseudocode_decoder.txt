CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO new TreeNode with values[0]  
    SET i TO 1  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO queue removing the leftmost element  
        IF i IS LESS THAN LENGTH OF values AND values[i] IS NOT None  
            SET node.left TO new TreeNode with values[i]  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values[i] IS NOT None  
            SET node.right TO new TreeNode with values[i]  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root IS None  
            RETURN  
        SET stack TO a list containing root  
        WHILE stack IS NOT EMPTY  
            SET node TO stack removing the last element  
            IF node.right IS NOT None  
                APPEND node.right TO stack  
            IF node.left IS NOT None  
                APPEND node.left TO stack  
            SET node.left TO None  
            IF stack IS NOT EMPTY  
                SET node.right TO stack last element  
            ELSE  
                SET node.right TO None  
    END FUNCTION  
END CLASS
