CLASS ListNode  
    FUNCTION __init__(val = 0, next = None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    SET head TO new ListNode WITH val AS first element of values  
    SET p TO head  
    FOR each val IN values starting from second element  
        SET node TO new ListNode WITH val AS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1, l2)  
        SET dummy_head TO new ListNode WITH val 0  
        SET current TO dummy_head  
        SET carry TO 0  
        
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry IS NOT 0  
            IF l1 IS NOT None  
                SET val1 TO l1.val  
            ELSE  
                SET val1 TO 0  
            END IF  
            
            IF l2 IS NOT None  
                SET val2 TO l2.val  
            ELSE  
                SET val2 TO 0  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY 10  
            SET current.next TO new ListNode WITH val AS total MODULO 10  
            SET current TO current.next  
            
            IF l1 IS NOT None  
                SET l1 TO l1.next  
            END IF  
            
            IF l2 IS NOT None  
                SET l2 TO l2.next  
            END IF  
        END WHILE  
        
        RETURN dummy_head.next  
    END FUNCTION  
END CLASS
