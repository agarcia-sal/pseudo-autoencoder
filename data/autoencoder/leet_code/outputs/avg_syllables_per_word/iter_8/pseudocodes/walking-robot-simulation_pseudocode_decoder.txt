CLASS Solution  
    FUNCTION robotSim(commands, obstacles)  
        SET directions TO list of (0, 1), (1, 0), (0, -1), (-1, 0)  
        SET x TO 0  
        SET y TO 0  
        SET dir_index TO 0  
        SET obstacles_set TO set of tuples created from obstacles  
        SET max_distance TO 0  
        
        FOR each command IN commands  
            IF command EQUALS -2  
                SET dir_index TO dir_index MINUS 1  
                IF dir_index LESS THAN 0  
                    SET dir_index TO dir_index PLUS 4  
            ELSE IF command EQUALS -1  
                SET dir_index TO dir_index PLUS 1  
                SET dir_index TO dir_index MODULO 4  
            ELSE  
                SET dx, dy TO directions AT INDEX dir_index  
                FOR counter FROM 1 TO command  
                    SET next_x TO x PLUS dx  
                    SET next_y TO y PLUS dy  
                    IF tuple of next_x, next_y IS IN obstacles_set  
                        BREAK loop  
                    SET x TO next_x  
                    SET y TO next_y  
                SET current_distance TO x MULTIPLIED BY x PLUS y MULTIPLIED BY y  
                IF current_distance GREATER THAN max_distance  
                    SET max_distance TO current_distance  
        RETURN max_distance  
    END FUNCTION  
END CLASS
