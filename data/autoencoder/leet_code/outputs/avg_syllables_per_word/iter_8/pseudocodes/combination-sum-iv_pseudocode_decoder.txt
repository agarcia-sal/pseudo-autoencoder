CLASS Solution  
    FUNCTION combinationSum4(nums, target)  
        CREATE list dp WITH size target PLUS 1 FILLED WITH 0  
        SET dp at index 0 TO 1  

        FOR amount FROM 1 TO target INCLUSIVE  
            FOR each num IN nums  
                IF amount GREATER THAN OR EQUAL TO num THEN  
                    INCREMENT dp at index amount BY dp at index amount MINUS num  
                END IF  
            END FOR  
        END FOR  

        RETURN dp at index target  
    END FUNCTION  
END CLASS
