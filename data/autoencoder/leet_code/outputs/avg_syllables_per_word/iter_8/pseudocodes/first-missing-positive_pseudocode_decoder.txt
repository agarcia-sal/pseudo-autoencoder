CLASS Solution  
    FUNCTION firstMissingPositive(nums)  
        SET n TO LENGTH OF nums  
          
        FOR i FROM 0 TO n MINUS 1  
            WHILE nums AT i IS GREATER THAN OR EQUAL TO 1 AND nums AT i IS LESS THAN OR EQUAL TO n AND nums AT (nums AT i MINUS 1) IS NOT EQUAL TO nums AT i  
                SET correct_index TO nums AT i MINUS 1  
                SWAP nums AT i WITH nums AT correct_index  
          
        FOR i FROM 0 TO n MINUS 1  
            IF nums AT i IS NOT EQUAL TO i PLUS 1  
                RETURN i PLUS 1  
          
        RETURN n PLUS 1  
    END FUNCTION  
END CLASS
