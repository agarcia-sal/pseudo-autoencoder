CLASS Solution  
    FUNCTION countArrangement(n)  
        FUNCTION backtrack(used, index)  
            IF index EQUALS n PLUS 1 THEN  
                RETURN 1  
            END IF  
            SET count TO 0  
            FOR i FROM 1 TO n DO  
                IF used AT i IS FALSE AND (i MODULO index EQUALS 0 OR index MODULO i EQUALS 0) THEN  
                    SET used AT i TO TRUE  
                    INCREMENT count BY backtrack(used, index PLUS 1)  
                    SET used AT i TO FALSE  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET used TO LIST OF FALSE VALUES WITH LENGTH n PLUS 1  
        RETURN backtrack(used, 1)  
    END FUNCTION  
END CLASS
