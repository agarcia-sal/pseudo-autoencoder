CLASS Solution  
    FUNCTION merge(intervals)  
        SORT intervals BY their first element  
        SET merged TO empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR last element's second value in merged IS LESS THAN interval's first value  
                APPEND interval TO merged  
            ELSE  
                SET last element's second value in merged TO MAXIMUM of last element's second value and interval's second value  
        RETURN merged  
    END FUNCTION  
END CLASS
