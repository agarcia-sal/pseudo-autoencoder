CLASS Solution  
    FUNCTION checkInclusion(s1, s2)  
        IMPORT Counter FROM collections  
        
        SET len_s1 TO LENGTH OF s1  
        SET len_s2 TO LENGTH OF s2  
        
        IF len_s1 GREATER THAN len_s2 THEN  
            RETURN False  
        
        SET count_s1 TO Counter OF s1  
        SET count_s2 TO Counter OF SUBSTRING OF s2 FROM 0 TO len_s1  
        
        IF count_s1 EQUALS count_s2 THEN  
            RETURN True  
        
        FOR i FROM len_s1 TO len_s2 MINUS 1 DO  
            INCREMENT count_s2 VALUE FOR s2 AT INDEX i BY 1  
            
            DECREMENT count_s2 VALUE FOR s2 AT INDEX (i MINUS len_s1) BY 1  
            
            IF count_s2 VALUE FOR s2 AT INDEX (i MINUS len_s1) EQUALS 0 THEN  
                REMOVE s2 AT INDEX (i MINUS len_s1) FROM count_s2  
            
            IF count_s1 EQUALS count_s2 THEN  
                RETURN True  
        
        RETURN False  
    END FUNCTION  
END CLASS
