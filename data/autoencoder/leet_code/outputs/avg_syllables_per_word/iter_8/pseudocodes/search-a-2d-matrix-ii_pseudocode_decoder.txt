CLASS Solution  
    FUNCTION searchMatrix(matrix, target)  
        IF matrix IS empty OR first element of matrix IS empty  
            RETURN False  
        END IF  
        
        SET row TO 0  
        SET col TO length of first element of matrix MINUS 1  
        
        WHILE row IS LESS THAN length of matrix AND col IS GREATER THAN OR EQUAL TO 0  
            IF element at matrix[row][col] IS EQUAL TO target  
                RETURN True  
            ELSE IF element at matrix[row][col] IS GREATER THAN target  
                DECREMENT col BY 1  
            ELSE  
                INCREMENT row BY 1  
            END IF  
        END WHILE  
        
        RETURN False  
    END FUNCTION  
END CLASS
