CLASS Solution  
    FUNCTION sortTransformedArray(nums, a, b, c)  
        FUNCTION quadratic(x)  
            RETURN a MULTIPLIED BY x MULTIPLIED BY x PLUS b MULTIPLIED BY x PLUS c  
        END FUNCTION  
  
        SET n TO LENGTH OF nums  
        SET result TO LIST OF SIZE n FILLED WITH ZEROES  
        SET left TO 0  
        SET right TO n MINUS 1  
  
        IF a GREATER THAN OR EQUAL TO 0 THEN  
            SET index TO n MINUS 1  
            WHILE left LESS THAN OR EQUAL TO right  
                IF quadratic(nums AT left) GREATER THAN quadratic(nums AT right) THEN  
                    SET result AT index TO quadratic(nums AT left)  
                    INCREMENT left BY 1  
                ELSE  
                    SET result AT index TO quadratic(nums AT right)  
                    DECREMENT right BY 1  
                END IF  
                DECREMENT index BY 1  
            END WHILE  
        ELSE  
            SET index TO 0  
            WHILE left LESS THAN OR EQUAL TO right  
                IF quadratic(nums AT left) LESS THAN quadratic(nums AT right) THEN  
                    SET result AT index TO quadratic(nums AT left)  
                    INCREMENT left BY 1  
                ELSE  
                    SET result AT index TO quadratic(nums AT right)  
                    DECREMENT right BY 1  
                END IF  
                INCREMENT index BY 1  
            END WHILE  
        END IF  
  
        RETURN result  
    END FUNCTION  
END CLASS
