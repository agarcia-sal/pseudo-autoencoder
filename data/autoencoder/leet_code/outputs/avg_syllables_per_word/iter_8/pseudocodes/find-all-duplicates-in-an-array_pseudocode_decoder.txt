CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR i FROM 0 TO LENGTH OF nums MINUS 1  
            WHILE VALUE AT nums[i] IS NOT EQUAL TO VALUE AT nums[VALUE AT nums[i] MINUS 1]  
                SWAP VALUE AT nums[VALUE AT nums[i] MINUS 1] WITH VALUE AT nums[i]  
        SET result_list TO EMPTY LIST  
        FOR EACH index, value IN ENUMERATE nums  
            IF value IS NOT EQUAL TO index PLUS 1  
                APPEND value TO result_list  
        RETURN result_list  
    END FUNCTION  
END CLASS
