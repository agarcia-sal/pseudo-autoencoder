CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1, val2)  
            RETURN (val1 GREATER THAN 0 AND val2 GREATER THAN 0) OR (val1 LESS THAN 0 AND val2 LESS THAN 0)  
        END FUNCTION  
        
        SET n TO LENGTH OF nums  
        
        FOR i FROM 0 TO n MINUS 1  
            SET slow TO i  
            SET fast TO i  
            
            IF nums AT i EQUALS 0 THEN  
                CONTINUE TO NEXT ITERATION  
            END IF  
            
            WHILE TRUE  
                SET slow TO (slow PLUS nums AT slow) MODULO n  
                IF slow LESS THAN 0 THEN  
                    INCREMENT slow BY n  
                END IF  
                
                SET fast TO (fast PLUS nums AT fast) MODULO n  
                IF fast LESS THAN 0 THEN  
                    INCREMENT fast BY n  
                END IF  
                
                SET fast TO (fast PLUS nums AT fast) MODULO n  
                IF fast LESS THAN 0 THEN  
                    INCREMENT fast BY n  
                END IF  
                
                IF NOT sameDirection(nums AT slow, nums AT i) OR NOT sameDirection(nums AT fast, nums AT i) THEN  
                    BREAK  
                END IF  
                
                IF slow EQUALS fast THEN  
                    IF slow EQUALS ((slow PLUS nums AT slow) MODULO n) THEN  
                        BREAK  
                    END IF  
                    RETURN TRUE  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN FALSE  
    END FUNCTION  
END CLASS
