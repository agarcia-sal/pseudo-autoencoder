CLASS Solution  
    FUNCTION maxSumOfThreeSubarrays(nums, k)  
        SET n TO LENGTH OF nums  
        SET sums TO LIST OF ZEROS OF LENGTH n MINUS k PLUS 1  
        SET current_sum TO SUM OF FIRST k ELEMENTS OF nums  
        SET sums AT POSITION 0 TO current_sum  
        
        FOR i FROM 1 TO n MINUS k  
            SET current_sum TO current_sum PLUS nums AT POSITION i PLUS k MINUS 1 MINUS nums AT POSITION i MINUS 1  
            SET sums AT POSITION i TO current_sum  
        
        SET left TO LIST OF ZEROS OF LENGTH n MINUS k PLUS 1  
        SET right TO LIST FILLED WITH n MINUS k OF LENGTH n MINUS k PLUS 1  
        
        SET max_sum TO 0  
        FOR i FROM 0 TO n MINUS k  
            IF sums AT POSITION i GREATER THAN sums AT POSITION max_sum  
                SET max_sum TO i  
            SET left AT POSITION i TO max_sum  
        
        SET max_sum TO n MINUS k  
        FOR i FROM n MINUS k DOWNTO 0  
            IF sums AT POSITION i GREATER THAN OR EQUAL TO sums AT POSITION max_sum  
                SET max_sum TO i  
            SET right AT POSITION i TO max_sum  
        
        SET max_total TO 0  
        SET result TO LIST OF ZEROS OF LENGTH 3  
        FOR j FROM k TO n MINUS 2 TIMES k  
            SET i TO left AT POSITION j MINUS k  
            SET l TO right AT POSITION j PLUS k  
            SET total TO sums AT POSITION i PLUS sums AT POSITION j PLUS sums AT POSITION l  
            IF total GREATER THAN max_total  
                SET max_total TO total  
                SET result TO LIST OF i, j, l  
        
        RETURN result  
    END FUNCTION  
END CLASS
