CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO sorted list of all start times from flowers  
        SET end_times TO sorted list of all end times from flowers  
        SET result TO empty list  
        FOR each person IN people  
            SET started TO count of start_times elements less than or equal to person using binary search  
            SET ended TO count of end_times elements strictly less than person using binary search  
            APPEND started minus ended TO result  
        RETURN result  
    END FUNCTION  
END CLASS
