CLASS Solution
    FUNCTION splitArray(nums, k)
        FUNCTION canSplit(max_sum)
            SET current_sum TO 0
            SET subarrays TO 1
            FOR each num IN nums
                IF current_sum PLUS num GREATER THAN max_sum
                    INCREMENT subarrays BY 1
                    SET current_sum TO num
                ELSE
                    INCREMENT current_sum BY num
            RETURN subarrays LESS THAN OR EQUAL TO k
        END FUNCTION

        SET left TO maximum value IN nums
        SET right TO sum OF nums
        WHILE left LESS THAN right
            SET mid TO left PLUS right DIVIDED BY 2
            IF canSplit(mid)
                SET right TO mid
            ELSE
                SET left TO mid PLUS 1
        RETURN left
    END FUNCTION
END CLASS
