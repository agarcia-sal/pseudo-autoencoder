CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO frequency count of characters in s  
        SET max_heap TO list of pairs with negative frequency and character from count  
        INITIALIZE max_heap as a heap structure  
        
        SET prev_char TO None  
        SET prev_freq TO 0  
        SET result TO empty list  
        
        WHILE max_heap is not empty OR prev_freq is less than 0  
            IF max_heap is empty AND prev_freq is less than 0  
                RETURN empty string  
            
            REMOVE top element from max_heap and SET freq and char accordingly  
            APPEND char TO result  
            
            INCREMENT freq BY 1  // since freq is negative and one instance is used  
            
            IF prev_freq is less than 0  
                ADD (prev_freq, prev_char) TO max_heap  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        
        RETURN concatenation of characters in result  
    END FUNCTION  
END CLASS
