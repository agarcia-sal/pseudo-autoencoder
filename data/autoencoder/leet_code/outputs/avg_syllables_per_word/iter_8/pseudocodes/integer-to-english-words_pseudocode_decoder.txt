CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS 0  
            RETURN "Zero"  
        END IF  
        
        SET below_twenty TO list of words for numbers from zero to nineteen  
        SET tens TO list of words for multiples of ten from twenty to ninety  
        SET thousands TO list of ["", "Thousand", "Million", "Billion"]  
        
        FUNCTION helper(n)  
            IF n EQUALS 0  
                RETURN empty string  
            ELSE IF n LESS THAN 20  
                RETURN below_twenty AT n CONCATENATED WITH space  
            ELSE IF n LESS THAN 100  
                RETURN tens AT n DIVIDED BY 10 CONCATENATED WITH space CONCATENATED WITH helper(n MODULO 10)  
            ELSE  
                RETURN below_twenty AT n DIVIDED BY 100 CONCATENATED WITH " Hundred " CONCATENATED WITH helper(n MODULO 100)  
            END IF  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO 0  
        
        WHILE num GREATER THAN 0  
            IF num MODULO 1000 NOT EQUAL TO 0  
                SET result TO helper(num MODULO 1000) CONCATENATED WITH thousands AT i CONCATENATED WITH space CONCATENATED WITH result  
            END IF  
            SET num TO num DIVIDED BY 1000  
            INCREMENT i BY 1  
        END WHILE  
        
        RETURN result WITHOUT trailing spaces  
    END FUNCTION  
END CLASS
