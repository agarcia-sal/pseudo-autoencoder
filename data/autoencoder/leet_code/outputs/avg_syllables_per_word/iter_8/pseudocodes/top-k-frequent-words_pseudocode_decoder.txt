CLASS Solution  
    FUNCTION topKFrequent(words, k)  
        SET count TO frequency mapping of words  
        SET heap TO list of pairs with negative frequency and corresponding word from count  
        TRANSFORM heap INTO a heap structure  
        SET result TO empty list  
        FOR each number FROM 1 TO k  
            REMOVE the smallest element from heap and APPEND its word part TO result  
        RETURN result  
    END FUNCTION  
END CLASS
