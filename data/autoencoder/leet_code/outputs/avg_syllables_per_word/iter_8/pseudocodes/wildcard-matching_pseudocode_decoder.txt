CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO a two-dimensional list of False with dimensions (LENGTH OF s PLUS 1) BY (LENGTH OF p PLUS 1)  
        SET dp AT 0 AND 0 TO True  

        FOR j FROM 1 TO LENGTH OF p  
            IF character AT (j MINUS 1) IN p EQUALS '*' THEN  
                SET dp AT 0 AND j TO dp AT 0 AND (j MINUS 1)  
            END IF  
        END FOR  

        FOR i FROM 1 TO LENGTH OF s  
            FOR j FROM 1 TO LENGTH OF p  
                IF character AT (j MINUS 1) IN p EQUALS '*' THEN  
                    SET dp AT i AND j TO (dp AT (i MINUS 1) AND j) OR (dp AT i AND (j MINUS 1))  
                ELSE IF character AT (j MINUS 1) IN p EQUALS '?' OR character AT (i MINUS 1) IN s EQUALS character AT (j MINUS 1) IN p THEN  
                    SET dp AT i AND j TO dp AT (i MINUS 1) AND (j MINUS 1)  
                END IF  
            END FOR  
        END FOR  

        RETURN dp AT LENGTH OF s AND LENGTH OF p  
    END FUNCTION  
END CLASS
