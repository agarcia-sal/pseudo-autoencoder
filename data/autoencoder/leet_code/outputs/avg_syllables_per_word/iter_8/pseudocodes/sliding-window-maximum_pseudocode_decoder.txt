CLASS Solution  
    FUNCTION maxSlidingWindow(nums, k)  
        IF nums is empty OR k EQUALS 0 THEN  
            RETURN empty list  
        END IF  
        
        INITIALIZE deq AS empty deque  
        INITIALIZE result AS empty list  
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 1 DO  
            IF deq is not empty AND FIRST ELEMENT OF deq IS LESS THAN i MINUS k PLUS 1 THEN  
                REMOVE FIRST ELEMENT FROM deq  
            END IF  
            
            WHILE deq is not empty AND ELEMENT AT LAST INDEX OF deq IN nums IS LESS THAN ELEMENT AT i IN nums DO  
                REMOVE LAST ELEMENT FROM deq  
            END WHILE  
            
            ADD i TO deq  
            
            IF i IS GREATER THAN OR EQUAL TO k MINUS 1 THEN  
                APPEND ELEMENT AT FIRST INDEX OF deq IN nums TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
