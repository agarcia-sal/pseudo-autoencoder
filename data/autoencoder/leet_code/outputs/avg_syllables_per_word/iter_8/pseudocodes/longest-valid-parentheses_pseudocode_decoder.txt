CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO list containing -1  
        SET max_length TO 0  

        FOR each index i and character char IN s  
            IF char EQUALS '(' THEN  
                APPEND i TO stack  
            ELSE  
                REMOVE last element FROM stack  
                IF stack IS empty THEN  
                    APPEND i TO stack  
                ELSE  
                    SET current_length TO i MINUS last element IN stack  
                    IF current_length GREATER THAN max_length THEN  
                        SET max_length TO current_length  
        RETURN max_length  
    END FUNCTION  
END CLASS
