CLASS Solution  
    FUNCTION numSquares(n)  
        SET dp TO a list of size n plus 1 with all values set to infinity  
        SET dp at index 0 TO 0  
        SET squares TO an empty list  
        SET i TO 1  
        WHILE i multiplied by i is less than or equal to n  
            APPEND i multiplied by i TO squares  
            INCREMENT i BY 1  
        FOR i FROM 1 TO n inclusive  
            FOR each square IN squares  
                IF i is less than square  
                    BREAK the inner loop  
                SET dp at index i TO the minimum of dp at index i and dp at index i minus square plus 1  
        RETURN dp at index n  
    END FUNCTION  
END CLASS
