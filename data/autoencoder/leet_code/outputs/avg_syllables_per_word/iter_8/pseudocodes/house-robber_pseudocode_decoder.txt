CLASS Solution  
    FUNCTION rob(nums)  
        IF nums IS empty  
            RETURN 0  
        END IF  
        IF LENGTH OF nums IS EQUAL TO 1  
            RETURN first element of nums  
        END IF  
          
        SET dp TO a list of zeros with length equal to LENGTH OF nums  
        SET first element of dp TO first element of nums  
        SET second element of dp TO the greater value between first element and second element of nums  
          
        FOR i FROM 2 TO LENGTH OF nums MINUS 1  
            SET dp at position i TO the greater value between dp at position i minus 1 AND dp at position i minus 2 PLUS nums at position i  
        END FOR  
          
        RETURN last element of dp  
    END FUNCTION  
END CLASS
