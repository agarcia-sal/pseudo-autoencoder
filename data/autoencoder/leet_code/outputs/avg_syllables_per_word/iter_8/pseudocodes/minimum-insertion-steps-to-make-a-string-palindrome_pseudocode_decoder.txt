CLASS Solution  
    FUNCTION minInsertions(s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO LENGTH OF s  
            CREATE dp AS 2D array of size n by n initialized TO 0  
            FOR i FROM 0 TO n MINUS 1  
                SET dp at i,i TO 1  
            FOR cl FROM 2 TO n  
                FOR i FROM 0 TO n MINUS cl  
                    SET j TO i PLUS cl MINUS 1  
                    IF character at s[i] EQUALS character at s[j] AND cl EQUALS 2  
                        SET dp at i,j TO 2  
                    ELSE IF character at s[i] EQUALS character at s[j]  
                        SET dp at i,j TO dp at i plus 1, j minus 1 PLUS 2  
                    ELSE  
                        SET dp at i,j TO MAXIMUM of dp at i,j minus 1 AND dp at i plus 1,j  
            RETURN dp at 0, n minus 1  
        END FUNCTION  
        RETURN LENGTH OF s MINUS longest_palindromic_subsequence(s)  
    END FUNCTION  
END CLASS
