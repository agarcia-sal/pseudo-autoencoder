CLASS Solution  
    FUNCTION getCollisionTimes(cars)  
        SET n TO LENGTH OF cars  
        SET result TO list of -1.0 repeated n times  
        SET stack TO empty list  

        FOR i FROM n MINUS 1 DOWNTO 0  
            SET pos, speed TO cars at index i  

            WHILE stack IS NOT empty  
                SET next_index TO last element in stack  
                SET next_pos, next_speed TO cars at index next_index  

                IF speed LESS THAN OR EQUAL TO next_speed  
                    REMOVE last element from stack  
                    CONTINUE to next iteration of WHILE  
                END IF  

                SET collision_time TO (next_pos MINUS pos) DIVIDED BY (speed MINUS next_speed)  

                IF result at next_index GREATER THAN 0 AND collision_time GREATER THAN OR EQUAL TO result at next_index  
                    REMOVE last element from stack  
                    CONTINUE to next iteration of WHILE  
                END IF  

                SET result at i TO collision_time  
                BREAK out of WHILE  
            END WHILE  

            APPEND i TO stack  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
