CLASS Solution  
    FUNCTION minDays(n)  
        FUNCTION dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO 1 THEN  
                RETURN remaining  
            END IF  
            SET option_one TO remaining MODULO 2 PLUS dp(remaining DIVIDED BY 2)  
            SET option_two TO remaining MODULO 3 PLUS dp(remaining DIVIDED BY 3)  
            RETURN 1 PLUS MINIMUM OF option_one AND option_two  
        END FUNCTION  
        RETURN dp(n)  
    END FUNCTION  
END CLASS
