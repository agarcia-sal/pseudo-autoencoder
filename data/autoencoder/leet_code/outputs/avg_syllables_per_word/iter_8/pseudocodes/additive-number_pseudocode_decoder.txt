CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS LENGTH OF num THEN  
                RETURN True  
            END IF  
            
            SET expected TO string representation OF (first PLUS second)  
            SET length TO LENGTH OF expected  
            
            IF start PLUS length GREATER THAN LENGTH OF num OR substring OF num FROM start TO start PLUS length NOT EQUAL TO expected THEN  
                RETURN False  
            END IF  
            
            RETURN is_valid_sequence(start PLUS length, second, integer value OF expected)  
        END FUNCTION  
        
        SET n TO LENGTH OF num  
        FOR i FROM 1 TO n DIVIDED BY 2 INCLUSIVE  
            FOR j FROM i PLUS 1 TO n EXCLUSIVE  
                SET first TO substring OF num FROM START TO i  
                SET second TO substring OF num FROM i TO j  
                
                IF (LENGTH OF first GREATER THAN 1 AND first STARTS WITH '0') OR (LENGTH OF second GREATER THAN 1 AND second STARTS WITH '0') THEN  
                    CONTINUE TO NEXT ITERATION  
                END IF  
                
                IF is_valid_sequence(j, integer value OF first, integer value OF second) THEN  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
