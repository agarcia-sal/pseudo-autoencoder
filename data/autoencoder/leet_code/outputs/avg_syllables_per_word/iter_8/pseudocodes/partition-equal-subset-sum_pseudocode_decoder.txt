CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO sum of nums  
        IF total_sum modulo 2 is not equal to 0 THEN  
            RETURN False  
        END IF  
        SET target TO total_sum divided by 2  
        SET n TO length of nums  
        SET dp TO list of False values with length target plus 1  
        SET dp at index 0 TO True  
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET dp at index i TO dp at index i OR dp at index i minus num  
            END FOR  
        END FOR  
        RETURN dp at index target  
    END FUNCTION  
END CLASS
