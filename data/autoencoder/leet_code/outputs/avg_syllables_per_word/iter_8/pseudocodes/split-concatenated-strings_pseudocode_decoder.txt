CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR i FROM 0 TO LENGTH OF strs MINUS 1  
            SET reversed_str TO reversal of strs at index i  
            IF reversed_str IS lexicographically GREATER THAN strs at index i  
                SET strs at index i TO reversed_str  
        
        SET doubled_strs TO concatenation of strs WITH strs  
        SET max_string TO empty string  
        
        FOR i FROM 0 TO LENGTH OF strs MINUS 1  
            FOR is_reversed IN [False, True]  
                IF is_reversed IS True  
                    SET current TO reversal of strs at index i  
                ELSE  
                    SET current TO strs at index i  
                
                SET remaining TO concatenation of elements in doubled_strs FROM index i plus 1 UP TO index i plus LENGTH OF strs MINUS 1  
                
                FOR j FROM 0 TO LENGTH OF current  
                    SET candidate TO concatenation of substring of current FROM index j TO end, remaining, and substring of current FROM start TO index j minus 1  
                    IF candidate IS lexicographically GREATER THAN max_string  
                        SET max_string TO candidate  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
