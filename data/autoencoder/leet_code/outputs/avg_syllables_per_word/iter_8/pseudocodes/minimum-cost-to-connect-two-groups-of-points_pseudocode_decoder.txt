CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO LENGTH OF cost  
        SET size2 TO LENGTH OF cost AT INDEX 0  
        SET dp TO LIST OF LIST WITH (size1 + 1) ELEMENTS EACH CONTAINING (2 RAISED TO size2) ELEMENTS ALL SET TO infinity  
        SET dp AT INDEX 0 AT INDEX 0 TO 0  
        
        FOR i FROM 0 TO size1 MINUS 1  
            FOR mask FROM 0 TO (2 RAISED TO size2) MINUS 1  
                FOR j FROM 0 TO size2 MINUS 1  
                    SET dp AT INDEX i PLUS 1 AT INDEX (mask OR (1 LEFT SHIFTED BY j)) TO MINIMUM OF dp AT INDEX i PLUS 1 AT INDEX (mask OR (1 LEFT SHIFTED BY j)) AND dp AT INDEX i AT INDEX mask PLUS cost AT INDEX i AT INDEX j  
                FOR j FROM 0 TO size2 MINUS 1  
                    IF mask AND (1 LEFT SHIFTED BY j) IS NOT ZERO  
                        SET dp AT INDEX i PLUS 1 AT INDEX mask TO MINIMUM OF dp AT INDEX i PLUS 1 AT INDEX mask AND dp AT INDEX i PLUS 1 AT INDEX (mask XOR (1 LEFT SHIFTED BY j)) PLUS cost AT INDEX i AT INDEX j  
        
        RETURN dp AT INDEX size1 AT INDEX (2 RAISED TO size2) MINUS 1  
    END FUNCTION  
END CLASS
