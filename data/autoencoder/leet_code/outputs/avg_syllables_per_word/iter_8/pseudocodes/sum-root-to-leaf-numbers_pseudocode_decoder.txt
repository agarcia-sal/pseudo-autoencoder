CLASS TreeNode  
    FUNCTION __init__(self, val SET TO 0, left SET TO None, right SET TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values is empty  
        RETURN None  
    SET root TO new TreeNode with val SET TO first element of values  
    SET i TO 1  
    SET queue TO new empty queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from front of queue  
        IF i is less than length of values AND values at i is not None  
            SET node.left TO new TreeNode with val SET TO values at i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i is less than length of values AND values at i is not None  
            SET node.right TO new TreeNode with val SET TO values at i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p is None AND q is None  
        RETURN True  
    ELSE IF p is None OR q is None  
        RETURN False  
    ELSE IF p.val is not equal to q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers(self, root)  
        FUNCTION dfs(node, current_number)  
            IF node is None  
                RETURN 0  
            SET current_number TO current_number multiplied BY 10 PLUS node.val  
            IF node.left is None AND node.right is None  
                RETURN current_number  
            SET left_sum TO dfs(node.left, current_number)  
            SET right_sum TO dfs(node.right, current_number)  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN dfs(root, 0)  
    END FUNCTION  
END CLASS
