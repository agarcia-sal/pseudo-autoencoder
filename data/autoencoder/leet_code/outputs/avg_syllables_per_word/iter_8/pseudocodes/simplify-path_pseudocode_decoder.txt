CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO result of splitting path by '/'  
        SET stack TO empty list  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS '.'  
                CONTINUE to next component  
            ELSE IF component EQUALS '..'  
                IF stack is not empty  
                    REMOVE last element from stack  
            ELSE  
                APPEND component TO stack  
        SET simplified_path TO string formed by concatenating '/' WITH elements of stack joined by '/'  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
