CLASS Solution  
    FUNCTION smallestGoodBase(n)  
        SET n TO integer conversion of n  
        SET max_length TO integer part of binary logarithm of n PLUS 1  
        
        FOR m FROM max_length DOWN TO 2 STEP DOWN BY 1  
            SET k TO integer part of n RAISED TO THE POWER of (1 DIVIDED BY (m MINUS 1))  
            IF k LESS THAN 2 THEN  
                CONTINUE TO NEXT ITERATION  
            END IF  
            
            SET num TO integer division of (k RAISED TO THE POWER m MINUS 1) BY (k MINUS 1)  
            
            IF num EQUALS n THEN  
                RETURN string representation of k  
            END IF  
        END FOR  
        
        RETURN string representation of (n MINUS 1)  
    END FUNCTION  
END CLASS
