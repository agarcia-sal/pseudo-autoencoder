CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MOD TO 10 RAISED TO THE POWER OF 9 PLUS 7  
        SET end_with_0 TO 0  
        SET end_with_1 TO 0  
        SET has_zero TO WHETHER binary CONTAINS '0'  
        FOR each bit IN binary  
            IF bit EQUALS '1'  
                SET end_with_1 TO remainder of (end_with_0 PLUS end_with_1 PLUS 1) DIVIDED BY MOD  
            ELSE  
                SET end_with_0 TO remainder of (end_with_0 PLUS end_with_1) DIVIDED BY MOD  
        RETURN remainder of (end_with_0 PLUS end_with_1 PLUS has_zero) DIVIDED BY MOD  
    END FUNCTION  
END CLASS
