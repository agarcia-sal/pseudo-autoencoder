CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuples created from each row in board  
        SET target TO tuple of tuples (1, 2, 3) and (4, 5, 0)  
        
        SET neighbors TO dictionary with keys and sets:  
            0 maps to set containing 1 and 3  
            1 maps to set containing 0, 2, and 4  
            2 maps to set containing 1 and 5  
            3 maps to set containing 0 and 4  
            4 maps to set containing 1, 3, and 5  
            5 maps to set containing 2 and 4  
        
        FOR i FROM 0 TO 1  
            FOR j FROM 0 TO 2  
                IF start at position i and j EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY 3 PLUS j  
        
        SET queue TO a deque initialized with tuple containing start, zero_pos, and 0  
        SET visited TO a set containing start  
        
        WHILE queue IS NOT empty  
            REMOVE the leftmost element from queue and unpack to state, zero_pos, and moves  
            
            IF state EQUALS target  
                RETURN moves  
            
            FOR each neighbor IN neighbors at zero_pos  
                SET new_state TO list of lists converted from state  
                SET i TO zero_pos DIVIDED BY 3  
                SET j TO zero_pos MODULO 3  
                SET ni TO neighbor DIVIDED BY 3  
                SET nj TO neighbor MODULO 3  
                
                SWAP elements in new_state at positions (i, j) and (ni, nj)  
                SET new_tuple_state TO tuple of tuples created from each row in new_state  
                
                IF new_tuple_state NOT IN visited  
                    ADD new_tuple_state TO visited  
                    APPEND tuple containing new_tuple_state, neighbor, and moves PLUS 1 TO queue  
        
        RETURN negative 1  
    END FUNCTION  
END CLASS
