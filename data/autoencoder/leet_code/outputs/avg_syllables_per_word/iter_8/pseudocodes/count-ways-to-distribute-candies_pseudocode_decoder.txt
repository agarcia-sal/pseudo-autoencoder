CLASS Solution  
    FUNCTION waysToDistribute(n, k)  
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7  
        INITIALIZE dp AS A 2-D LIST WITH DIMENSIONS (n PLUS 1) BY (k PLUS 1) FILLED WITH ZEROES  
        SET dp AT 0 AND 0 TO 1  
        FOR i FROM 1 TO n INCLUSIVE  
            FOR j FROM 1 TO k INCLUSIVE  
                SET dp AT i AND j TO (j MULTIPLIED BY dp AT i MINUS 1 AND j PLUS dp AT i MINUS 1 AND j MINUS 1) MODULO MOD  
        RETURN dp AT n AND k  
    END FUNCTION  
END CLASS
