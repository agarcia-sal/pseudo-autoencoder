CLASS Solution  
    FUNCTION canVisitAllRooms(rooms)  
        SET visited TO empty set  
        SET stack TO list containing 0  
        
        WHILE stack is not empty  
            SET current_room TO last element removed from stack  
            ADD current_room TO visited  
            FOR each key IN rooms at current_room  
                IF key is not in visited  
                    APPEND key TO stack  
        RETURN size of visited EQUALS size of rooms  
    END FUNCTION  
END CLASS
