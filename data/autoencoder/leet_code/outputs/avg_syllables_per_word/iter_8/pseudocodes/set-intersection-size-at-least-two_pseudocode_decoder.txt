CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals BY end ASCENDING AND start DESCENDING  
        SET first TO intervals AT 0 AT 1 MINUS 1  
        SET second TO intervals AT 0 AT 1  
        SET count TO 2  
        FOR each start, end IN intervals FROM SECOND ELEMENT TO END  
            IF start GREATER THAN second  
                SET first TO end MINUS 1  
                SET second TO end  
                INCREMENT count BY 2  
            ELSE IF start GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY 1  
        RETURN count  
    END FUNCTION  
END CLASS
