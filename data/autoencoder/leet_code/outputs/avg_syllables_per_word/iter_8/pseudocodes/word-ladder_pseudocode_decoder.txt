CLASS Solution  
    FUNCTION ladderLength(beginWord, endWord, wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN 0  
        END IF  
  
        SET word_dict TO a dictionary with default value as empty list  
        SET word_length TO LENGTH OF beginWord  
  
        FOR each word IN wordList  
            FOR i FROM 0 TO word_length MINUS 1  
                SET intermediate_word TO CONCATENATION OF substring of word FROM start TO i, "*" , substring of word FROM i+1 TO end  
                APPEND word TO word_dict AT intermediate_word  
            END FOR  
        END FOR  
  
        SET queue TO a deque containing a tuple of (beginWord, 1)  
        SET visited TO a set containing beginWord  
  
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue INTO current_word, current_length  
  
            FOR i FROM 0 TO word_length MINUS 1  
                SET intermediate_word TO CONCATENATION OF substring of current_word FROM start TO i, "*" , substring of current_word FROM i+1 TO end  
  
                FOR each word IN word_dict AT intermediate_word  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS 1  
                    END IF  
  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND (word, current_length PLUS 1) TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
  
        RETURN 0  
    END FUNCTION  
END CLASS
