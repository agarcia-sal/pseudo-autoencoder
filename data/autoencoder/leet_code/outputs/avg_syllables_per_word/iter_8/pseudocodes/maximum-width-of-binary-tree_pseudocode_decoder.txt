CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO TreeNode with value values AT position 0  
    SET i TO 1  
    SET queue TO empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i IS LESS THAN LENGTH OF values AND values AT position i IS NOT None  
            SET node.left TO TreeNode with value values AT position i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values AT position i IS NOT None  
            SET node.right TO TreeNode with value values AT position i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree(self, root)  
        IF root IS None  
            RETURN 0  
        END IF  
        SET queue TO double-ended queue containing tuple of root and 0  
        SET max_width TO 0  
        WHILE queue IS NOT EMPTY  
            SET level_length TO LENGTH OF queue  
            SET first_index TO second item OF FIRST element IN queue  
            FOR each number FROM 0 TO level_length MINUS 1  
                SET node, index TO REMOVE first element FROM queue  
                IF number IS EQUAL TO level_length MINUS 1  
                    SET max_width TO GREATER OF max_width AND index MINUS first_index PLUS 1  
                END IF  
                IF node.left EXISTS  
                    APPEND tuple of node.left and index TIMES 2 TO queue  
                END IF  
                IF node.right EXISTS  
                    APPEND tuple of node.right and index TIMES 2 PLUS 1 TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
