CLASS Solution  
    FUNCTION isInterleave(s1, s2, s3)  
        IF LENGTH OF s1 PLUS LENGTH OF s2 IS NOT EQUAL TO LENGTH OF s3  
            RETURN False  
        END IF  

        SET dp TO LIST OF False REPEATED LENGTH OF s2 PLUS ONE TIMES  
        SET dp AT INDEX 0 TO True  

        FOR i FROM 0 TO LENGTH OF s1 INCLUSIVE  
            FOR j FROM 0 TO LENGTH OF s2 INCLUSIVE  
                IF i IS GREATER THAN 0  
                    SET dp AT INDEX j TO dp AT INDEX j AND CHARACTER AT s1 INDEX i MINUS 1 EQUALS CHARACTER AT s3 INDEX i PLUS j MINUS 1  
                END IF  
                IF j IS GREATER THAN 0  
                    SET dp AT INDEX j TO dp AT INDEX j OR (dp AT INDEX j MINUS 1 AND CHARACTER AT s2 INDEX j MINUS 1 EQUALS CHARACTER AT s3 INDEX i PLUS j MINUS 1)  
                END IF  
            END FOR  
        END FOR  

        RETURN dp AT INDEX LENGTH OF s2  
    END FUNCTION  
END CLASS
