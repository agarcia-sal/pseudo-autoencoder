CLASS Solution  
    FUNCTION flipLights(n, presses)  
        SET n TO the minimum of n and 6  
          
        IF presses EQUALS 0 THEN  
            RETURN 1  
          
        IF n EQUALS 1 THEN  
            RETURN 2  
          
        IF n EQUALS 2 THEN  
            IF presses EQUALS 1 THEN  
                RETURN 3  
            ELSE  
                RETURN 4  
          
        IF presses EQUALS 1 THEN  
            RETURN 4  
          
        IF presses EQUALS 2 THEN  
            RETURN 7  
          
        RETURN 8  
    END FUNCTION  
END CLASS
