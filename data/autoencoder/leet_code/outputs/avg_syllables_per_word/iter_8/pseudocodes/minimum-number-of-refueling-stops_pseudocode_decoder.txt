CLASS Solution  
    FUNCTION minRefuelStops(target, startFuel, stations)  
        INITIALIZE max_heap AS empty list  
        SET current_fuel TO startFuel  
        SET stops TO 0  
        APPEND list of target and 0 TO stations  

        FOR each position and fuel IN stations  
            WHILE max_heap is not empty AND current_fuel is less than position  
                INCREMENT current_fuel BY removal of largest fuel amount from max_heap  
                INCREMENT stops BY 1  
            IF current_fuel is less than position  
                RETURN -1  
            ADD negative of fuel TO max_heap to maintain max-heap behavior  
        RETURN stops  
    END FUNCTION  
END CLASS
