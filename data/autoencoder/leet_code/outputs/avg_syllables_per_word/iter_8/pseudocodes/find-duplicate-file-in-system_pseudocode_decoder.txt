CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO empty mapping from content TO list of paths  
        FOR each path IN paths  
            SET parts TO split path by whitespace  
            SET directory TO first element of parts  
            FOR each file_info IN parts starting from second element  
                SET file_name and content TO split file_info at first occurrence of '('  
                REMOVE the last character from content  
                APPEND concatenation of directory, "/", and file_name TO content_to_paths at key content  
        SET result TO empty list  
        FOR each path_list IN content_to_paths values  
            IF length of path_list GREATER THAN 1  
                APPEND path_list TO result  
        RETURN result  
    END FUNCTION  
END CLASS
