CLASS Solution  
    FUNCTION calculateMinimumHP(dungeon)  
        SET m TO LENGTH OF dungeon  
        SET n TO LENGTH OF dungeon AT POSITION 0  
        CREATE dp AS TWO-DIMENSIONAL LIST WITH DIMENSIONS m PLUS 1 BY n PLUS 1 FILLED WITH INFINITY  
        SET dp AT POSITION m AND n MINUS 1 TO 1  
        SET dp AT POSITION m MINUS 1 AND n TO 1  
        
        FOR i FROM m MINUS 1 DOWNTO 0  
            FOR j FROM n MINUS 1 DOWNTO 0  
                SET min_health TO MINIMUM OF dp AT POSITION i PLUS 1 AND j AND dp AT POSITION i AND j PLUS 1 MINUS dungeon AT POSITION i AND j  
                SET dp AT POSITION i AND j TO GREATER OF min_health AND 1  
        
        RETURN dp AT POSITION 0 AND 0  
    END FUNCTION  
END CLASS
