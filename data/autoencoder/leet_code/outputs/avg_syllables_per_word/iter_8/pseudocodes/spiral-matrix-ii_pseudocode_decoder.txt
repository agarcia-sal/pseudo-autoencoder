CLASS Solution  
    FUNCTION generateMatrix(n)  
        SET matrix TO a two-dimensional list of size n by n filled with zeros  
        SET top TO 0  
        SET bottom TO n MINUS 1  
        SET left TO 0  
        SET right TO n MINUS 1  
        SET num TO 1  
        
        WHILE top IS LESS THAN OR EQUAL TO bottom AND left IS LESS THAN OR EQUAL TO right  
            FOR i FROM left TO right INCLUSIVE  
                SET matrix AT position top, i TO num  
                INCREMENT num BY 1  
            SET top TO top PLUS 1  
            
            FOR i FROM top TO bottom INCLUSIVE  
                SET matrix AT position i, right TO num  
                INCREMENT num BY 1  
            SET right TO right MINUS 1  
            
            IF top IS LESS THAN OR EQUAL TO bottom  
                FOR i FROM right DOWN TO left INCLUSIVE  
                    SET matrix AT position bottom, i TO num  
                    INCREMENT num BY 1  
                SET bottom TO bottom MINUS 1  
            
            IF left IS LESS THAN OR EQUAL TO right  
                FOR i FROM bottom DOWN TO top INCLUSIVE  
                    SET matrix AT position i, left TO num  
                    INCREMENT num BY 1  
                SET left TO left PLUS 1  
        
        RETURN matrix  
    END FUNCTION  
END CLASS
