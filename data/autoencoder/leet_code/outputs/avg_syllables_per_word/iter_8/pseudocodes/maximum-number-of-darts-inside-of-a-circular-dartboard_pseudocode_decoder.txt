CLASS Solution  
    FUNCTION numPoints(darts, r)  
        FUNCTION distance(p1, p2)  
            RETURN square root of ((p1 at index 0 MINUS p2 at index 0) squared PLUS (p1 at index 1 MINUS p2 at index 1) squared)  
        END FUNCTION  
        
        FUNCTION circle_center(p1, p2, r)  
            SET d TO distance between p1 and p2  
            IF d GREATER THAN 2 MULTIPLIED BY r  
                RETURN None  
            SET a TO square root of (r squared MINUS (d DIVIDED BY 2) squared)  
            SET h TO list of [(p1 at index 0 PLUS p2 at index 0) DIVIDED BY 2, (p1 at index 1 PLUS p2 at index 1) DIVIDED BY 2]  
            SET dx TO (p2 at index 1 MINUS p1 at index 1) MULTIPLIED BY a DIVIDED BY d  
            SET dy TO (p2 at index 0 MINUS p1 at index 0) MULTIPLIED BY a DIVIDED BY d  
            RETURN list of [(h at index 0 MINUS dx, h at index 1 PLUS dy), (h at index 0 PLUS dx, h at index 1 MINUS dy)]  
        END FUNCTION  
        
        FUNCTION is_inside(circle, point, r)  
            RETURN distance between circle and point LESS THAN OR EQUAL TO r  
        END FUNCTION  
        
        SET n TO length of darts  
        IF n EQUAL TO 1  
            RETURN 1  
        
        SET max_count TO 1  
        
        FOR i FROM 0 TO n MINUS 1  
            FOR j FROM i PLUS 1 TO n MINUS 1  
                SET centers TO circle_center of darts at i, darts at j, r  
                IF centers IS NOT None  
                    FOR each center IN centers  
                        SET count TO 0  
                        FOR each dart IN darts  
                            IF is_inside of center, dart, r IS True  
                                INCREMENT count BY 1  
                        IF count GREATER THAN max_count  
                            SET max_count TO count  
        
        RETURN max_count  
    END FUNCTION  
END CLASS
