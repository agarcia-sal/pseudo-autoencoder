CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO set of characters in time without colon  
        SET hour, minute TO integer values from splitting time by colon  
        
        FUNCTION generate_times(digits)  
            SET times TO empty list  
            FOR each h1 IN digits  
                FOR each h2 IN digits  
                    FOR each m1 IN digits  
                        FOR each m2 IN digits  
                            SET hour_candidate TO integer value of concatenation of h1 and h2  
                            SET minute_candidate TO integer value of concatenation of m1 and m2  
                            IF hour_candidate LESS THAN 24 AND minute_candidate LESS THAN 60  
                                APPEND tuple of (hour_candidate, minute_candidate) TO times  
            RETURN sorted times  
        END FUNCTION  
        
        SET all_times TO generate_times(digits)  
        SET current_time_index TO index of tuple (hour, minute) IN all_times  
        
        IF current_time_index PLUS 1 LESS THAN length of all_times  
            SET next_hour, next_minute TO element at current_time_index PLUS 1 IN all_times  
        ELSE  
            SET next_hour, next_minute TO first element IN all_times  
        
        RETURN formatted string of next_hour and next_minute as "HH:MM" with leading zeros  
    END FUNCTION  
END CLASS
