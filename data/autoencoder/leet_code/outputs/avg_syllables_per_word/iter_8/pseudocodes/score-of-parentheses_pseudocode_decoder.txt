CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO empty list  
        SET score TO 0  
        FOR each char IN s  
            IF char IS equal to '(' THEN  
                APPEND score TO stack  
                SET score TO 0  
            ELSE  
                SET score TO value popped FROM stack PLUS (1 IF score EQUALS 0 ELSE 2 MULTIPLIED BY score)  
        RETURN score  
    END FUNCTION  
END CLASS
