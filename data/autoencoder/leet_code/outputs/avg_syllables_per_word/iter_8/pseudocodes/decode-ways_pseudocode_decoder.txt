CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s IS EMPTY OR first character of s IS EQUAL TO '0'  
            RETURN 0  
        END IF  

        SET dp TO list of zeros WITH LENGTH of s PLUS 1  
        SET dp at index 0 TO 1  
        SET dp at index 1 TO 1  

        FOR i FROM 2 TO LENGTH of s PLUS 1  
            IF character at index i MINUS 1 of s IS NOT EQUAL TO '0'  
                INCREMENT dp at index i BY dp at index i MINUS 1  
            END IF  

            SET two_digit TO integer value of substring of s FROM index i MINUS 2 TO i  
            IF two_digit IS GREATER THAN OR EQUAL TO 10 AND two_digit IS LESS THAN OR EQUAL TO 26  
                INCREMENT dp at index i BY dp at index i MINUS 2  
            END IF  
        END FOR  

        RETURN last element of dp  
    END FUNCTION  
END CLASS
