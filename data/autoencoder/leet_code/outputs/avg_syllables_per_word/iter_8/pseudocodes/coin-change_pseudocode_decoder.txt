CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        SET dp TO list of size amount plus 1 with each element set TO amount plus 1  
        SET dp at index 0 TO 0  
        FOR a FROM 1 TO amount INCLUSIVE  
            FOR each coin IN coins  
                IF a minus coin IS GREATER THAN OR EQUAL TO 0 THEN  
                    SET dp at index a TO minimum of dp at index a AND dp at index a minus coin plus 1  
        IF dp at index amount IS EQUAL TO amount plus 1 THEN  
            RETURN -1  
        ELSE  
            RETURN dp at index amount  
    END FUNCTION  
END CLASS
