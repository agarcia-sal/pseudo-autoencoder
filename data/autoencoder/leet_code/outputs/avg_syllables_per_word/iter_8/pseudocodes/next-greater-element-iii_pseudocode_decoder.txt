CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO list of characters FROM string representation of n  
        SET length TO size of digits  
        
        SET i TO length MINUS 2  
        WHILE i GREATER THAN OR EQUAL TO 0 AND digits AT i GREATER THAN OR EQUAL TO digits AT i PLUS 1  
            DECREMENT i BY 1  
        
        IF i EQUALS -1  
            RETURN -1  
        
        SET j TO length MINUS 1  
        WHILE digits AT j LESS THAN OR EQUAL TO digits AT i  
            DECREMENT j BY 1  
        
        SWAP digits AT i WITH digits AT j  
        
        SET digits TO concatenation of digits UP TO i PLUS 1 AND reverse of digits FROM i PLUS 1 TO END  
        
        SET result TO integer converted FROM concatenation of digits  
        
        IF result GREATER THAN (2 RAISED TO 31) MINUS 1  
            RETURN -1  
        
        RETURN result  
    END FUNCTION  
END CLASS
