CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO first element of nums  
        SET current_sum TO first element of nums  
        FOR each num IN nums starting from second element  
            SET current_sum TO greater value between num AND current_sum PLUS num  
            SET max_sum TO greater value between max_sum AND current_sum  
        RETURN max_sum  
    END FUNCTION  
END CLASS
