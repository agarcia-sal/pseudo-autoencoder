CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SPLIT time INTO hours AND minutes BY ':'  
            SET hours TO integer value of hours  
            SET minutes TO integer value of minutes  
            RETURN hours MULTIPLIED BY 60 PLUS minutes  
        END FUNCTION  

        SET minutes_list TO list obtained by applying to_minutes FUNCTION to each element of timePoints  
        SORT minutes_list in ascending order  
        SET min_diff TO a very large number representing infinity  

        FOR i FROM 1 TO LENGTH OF minutes_list MINUS 1  
            SET current_diff TO minutes_list AT i MINUS minutes_list AT i MINUS 1  
            IF current_diff IS LESS THAN min_diff  
                SET min_diff TO current_diff  
            END IF  
        END FOR  

        SET circular_diff TO (1440 MINUS minutes_list AT LAST ELEMENT) PLUS minutes_list AT FIRST ELEMENT  
        IF circular_diff IS LESS THAN min_diff  
            SET min_diff TO circular_diff  
        END IF  

        RETURN min_diff  
    END FUNCTION  
END CLASS
