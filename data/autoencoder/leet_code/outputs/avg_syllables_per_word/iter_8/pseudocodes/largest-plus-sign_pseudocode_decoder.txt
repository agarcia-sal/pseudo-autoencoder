CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO set of tuples converted from each element in mines  
        SET dp TO two-dimensional list of size n by n where each element is a list of four zeros  
        SET directions TO list containing (0, 1), (1, 0), (0, -1), (-1, 0)  
        
        FOR i FROM 0 TO n MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF tuple of (i, j) NOT IN mines_set THEN  
                    IF j GREATER THAN 0 THEN  
                        SET dp at i, j, index 0 TO dp at i, j MINUS 1, index 0 PLUS 1  
                    IF i GREATER THAN 0 THEN  
                        SET dp at i, j, index 1 TO dp at i MINUS 1, j, index 1 PLUS 1  
        
        SET result TO 0  
        FOR i FROM n MINUS 1 DOWN TO 0  
            FOR j FROM n MINUS 1 DOWN TO 0  
                IF tuple of (i, j) NOT IN mines_set THEN  
                    IF j LESS THAN n MINUS 1 THEN  
                        SET dp at i, j, index 2 TO dp at i, j PLUS 1, index 2 PLUS 1  
                    IF i LESS THAN n MINUS 1 THEN  
                        SET dp at i, j, index 3 TO dp at i PLUS 1, j, index 3 PLUS 1  
                    SET plus_sign_order TO minimum value IN dp at i, j  
                    IF plus_sign_order GREATER THAN result THEN  
                        SET result TO plus_sign_order  
        
        RETURN result  
    END FUNCTION  
END CLASS
