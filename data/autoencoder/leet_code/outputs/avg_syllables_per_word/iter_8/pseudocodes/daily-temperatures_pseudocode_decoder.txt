CLASS Solution  
    FUNCTION dailyTemperatures(temperatures)  
        SET n TO LENGTH OF temperatures  
        SET answer TO list of zeros with length n  
        SET stack TO empty list  
        
        FOR i FROM 0 TO n MINUS 1  
            WHILE stack is not empty AND temperatures at i GREATER THAN temperatures at index at top of stack  
                SET prev_index TO REMOVE last element from stack  
                SET answer at prev_index TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
