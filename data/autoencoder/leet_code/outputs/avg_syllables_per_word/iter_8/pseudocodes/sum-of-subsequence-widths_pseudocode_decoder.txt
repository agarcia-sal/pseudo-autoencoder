CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MOD TO 1000000007  
        SORT nums IN ASCENDING ORDER  
        SET n TO LENGTH OF nums  
        SET total_sum TO 0  
        INITIALIZE power AS LIST OF SIZE n WITH ALL ELEMENTS SET TO 1  
        
        FOR i FROM 1 TO n MINUS 1  
            SET power AT i TO power AT i MINUS 1 MULTIPLIED BY 2 MODULO MOD  
        
        FOR i FROM 0 TO n MINUS 1  
            SET contribution TO nums AT i MULTIPLIED BY (power AT i MINUS power AT n MINUS i MINUS 1)  
            SET total_sum TO total_sum PLUS contribution MODULO MOD  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
