CLASS Solution  
    FUNCTION kthSmallestPath(destination, k)  
        SET row TO first element of destination  
        SET col TO second element of destination  
        INITIALIZE path AS empty list  
        SET total_moves TO col PLUS row  
        FOR i FROM 0 TO total_moves MINUS 1  
            IF col GREATER THAN 0 THEN  
                SET paths_with_H TO number of combinations choosing col MINUS 1 FROM row PLUS col MINUS 1  
                IF k LESS THAN OR EQUAL TO paths_with_H THEN  
                    APPEND 'H' TO path  
                    DECREMENT col BY 1  
                ELSE  
                    APPEND 'V' TO path  
                    DECREMENT row BY 1  
                    DECREMENT k BY paths_with_H  
            ELSE  
                APPEND 'V' TO path  
                DECREMENT row BY 1  
        RETURN concatenation of elements in path  
    END FUNCTION  
END CLASS
