CLASS Solution  
    FUNCTION twoSum(numbers, target)  
        SET left TO 0  
        SET right TO LENGTH OF numbers MINUS 1  
        
        WHILE left LESS THAN right  
            SET current_sum TO VALUE AT numbers[left] PLUS VALUE AT numbers[right]  
            IF current_sum EQUALS target  
                RETURN LIST CONTAINING left PLUS 1 AND right PLUS 1  
            ELSE IF current_sum LESS THAN target  
                INCREMENT left BY 1  
            ELSE  
                DECREMENT right BY 1  
        END WHILE  
    END FUNCTION  
END CLASS
