CLASS Solution  
    FUNCTION numMatchingSubseq(s, words)  
        SET waiting TO a dictionary mapping each unique character in s TO an empty deque  
        FOR each word IN words  
            SET it TO an iterator of word  
            SET first_letter TO the next element of it  
            IF first_letter IS IN waiting  
                APPEND it TO waiting at first_letter  
        SET match_count TO 0  
        FOR each character c IN s  
            IF c IS IN waiting  
                SET current_iterators TO waiting at c  
                SET waiting at c TO an empty deque  
                FOR each it IN current_iterators  
                    SET next_char TO the next element of it OR none if exhausted  
                    IF next_char IS none  
                        INCREMENT match_count BY 1  
                    ELSE  
                        IF next_char IS IN waiting  
                            APPEND it TO waiting at next_char  
        RETURN match_count  
    END FUNCTION  
END CLASS
