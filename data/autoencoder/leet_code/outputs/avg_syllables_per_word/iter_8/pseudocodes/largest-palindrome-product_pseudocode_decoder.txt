CLASS Solution  
    FUNCTION largestPalindrome(n)  
        IF n EQUALS 1 THEN  
            RETURN 9  
        END IF  

        SET upper_limit TO 10 RAISED TO n MINUS 1  
        SET lower_limit TO 10 RAISED TO (n MINUS 1)  

        SET max_palindrome TO 0  

        FOR i FROM upper_limit DOWN TO lower_limit DO  
            FOR j FROM i DOWN TO lower_limit DO  
                SET product TO i MULTIPLIED BY j  
                IF product LESS THAN OR EQUAL TO max_palindrome THEN  
                    BREAK  
                END IF  
                SET str_product TO STRING REPRESENTATION OF product  
                IF str_product EQUALS REVERSE OF str_product THEN  
                    SET max_palindrome TO product  
                END IF  
            END FOR  
        END FOR  

        RETURN max_palindrome MODULO 1337  
    END FUNCTION  
END CLASS
