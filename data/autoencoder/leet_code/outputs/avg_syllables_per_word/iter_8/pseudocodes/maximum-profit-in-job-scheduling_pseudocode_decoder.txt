CLASS Solution  
    FUNCTION jobScheduling(startTime, endTime, profit)  
        SET jobs TO list of tuples combining startTime, endTime, and profit SORTED BY job end time  
        SET end_times TO list of end times extracted from jobs  
        SET dp TO list of zeros with length of number of jobs PLUS one  
        FOR i FROM 1 TO length of jobs  
            SET j TO position found by binary searching end_times FOR jobs at index i MINUS one start time  
            SET dp at index i TO maximum of dp at index i MINUS one AND dp at index j PLUS profit of job at index i MINUS one  
        RETURN last element of dp  
    END FUNCTION  
END CLASS
