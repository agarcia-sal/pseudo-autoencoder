CLASS Solution  
    FUNCTION combinationSum2(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS 0 THEN  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN 0 THEN  
                RETURN  
            END IF  
            FOR i FROM start TO LENGTH OF candidates MINUS 1 DO  
                IF i GREATER THAN start AND candidates AT i EQUALS candidates AT i MINUS 1 THEN  
                    CONTINUE  
                END IF  
                CALL backtrack WITH (i PLUS 1, target MINUS candidates AT i, path APPENDED WITH candidates AT i)  
            END FOR  
        END FUNCTION  

        SORT candidates  
        SET result TO EMPTY LIST  
        CALL backtrack WITH (0, target, EMPTY LIST)  
        RETURN result  
    END FUNCTION  
END CLASS
