CLASS Solution  
    FUNCTION profitableSchemes(n, minProfit, group, profit)  
        SET MOD TO 10 raised to the power of 9 plus 7  
        INITIALIZE dp AS a two-dimensional list with dimensions (n plus 1) by (minProfit plus 1) filled with zeroes  
        FOR i FROM 0 TO n INCLUSIVE  
            SET dp at i and 0 TO 1  
        FOR each p, g IN pairs of profit and group  
            FOR i FROM n DOWNTO g  
                FOR j FROM minProfit DOWNTO 0  
                    SET dp at i and j TO dp at i and j plus dp at (i minus g) and (maximum of 0 and j minus p)  
                    SET dp at i and j TO dp at i and j MODULO MOD  
        RETURN dp at n and minProfit  
    END FUNCTION  
END CLASS
