CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO set containing 0 and firstPerson  
        SORT meetings BY the third element of each meeting (time)  
        SET i TO 0  
        WHILE i LESS THAN length of meetings  
            SET current_time TO the third element of meetings at index i  
            SET graph TO a new mapping from person to list of connected persons  
            WHILE i LESS THAN length of meetings AND the third element of meetings at index i EQUALS current_time  
                SET x, y, _ TO the meeting at index i  
                APPEND y TO graph at key x  
                APPEND x TO graph at key y  
                INCREMENT i BY 1  
            END WHILE  
            SET queue TO a queue initialized with persons from know_secret that appear in graph  
            SET visited TO an empty set  
            WHILE queue IS NOT empty  
                REMOVE front person FROM queue AND SET TO person  
                IF person IS IN visited  
                    CONTINUE to next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN graph at key person  
                    IF neighbor NOT IN visited  
                        ADD neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN list converted from know_secret  
    END FUNCTION  
END CLASS
