CLASS Solution  
    FUNCTION maximumScore(nums, multipliers)  
        SET m TO LENGTH OF multipliers  
        SET n TO LENGTH OF nums  
        SET dp TO LIST OF ZEROS WITH LENGTH m PLUS 1  
        FOR i FROM m MINUS 1 DOWN TO 0  
            SET new_dp TO LIST OF ZEROS WITH LENGTH m PLUS 1  
            FOR left FROM 0 TO i  
                SET right TO n MINUS 1 MINUS (i MINUS left)  
                SET mul TO multipliers AT INDEX i  
                SET new_dp AT INDEX left TO MAXIMUM OF (mul TIMES nums AT INDEX left PLUS dp AT INDEX left PLUS 1) AND (mul TIMES nums AT INDEX right PLUS dp AT INDEX left)  
            SET dp TO new_dp  
        RETURN dp AT INDEX 0  
    END FUNCTION  
END CLASS
