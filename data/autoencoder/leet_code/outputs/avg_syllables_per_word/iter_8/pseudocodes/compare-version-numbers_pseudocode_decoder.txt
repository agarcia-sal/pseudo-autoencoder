CLASS Solution  
    FUNCTION compareVersion(version1, version2)  
        SET v1_revisions TO SPLIT version1 BY '.'  
        SET v2_revisions TO SPLIT version2 BY '.'  
        
        SET max_length TO MAXIMUM OF LENGTH OF v1_revisions AND LENGTH OF v2_revisions  
        
        FOR i FROM 0 TO max_length MINUS 1  
            IF i LESS THAN LENGTH OF v1_revisions THEN  
                SET v1 TO CONVERT v1_revisions AT INDEX i TO INTEGER  
            ELSE  
                SET v1 TO 0  
            IF i LESS THAN LENGTH OF v2_revisions THEN  
                SET v2 TO CONVERT v2_revisions AT INDEX i TO INTEGER  
            ELSE  
                SET v2 TO 0  
            
            IF v1 LESS THAN v2 THEN  
                RETURN -1  
            ELSE IF v1 GREATER THAN v2 THEN  
                RETURN 1  
        
        RETURN 0  
    END FUNCTION  
END CLASS
