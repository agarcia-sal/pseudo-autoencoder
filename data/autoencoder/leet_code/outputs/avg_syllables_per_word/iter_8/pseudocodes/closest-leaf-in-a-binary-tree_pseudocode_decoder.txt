CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO TreeNode WITH val AS values AT index 0  
    SET i TO 1  
    INITIALIZE queue AS empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE first node FROM queue AND SET TO node  
        IF i IS LESS THAN length OF values AND values AT index i IS NOT None  
            SET node.left TO TreeNode WITH val AS values AT index i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN length OF values AND values AT index i IS NOT None  
            SET node.right TO TreeNode WITH val AS values AT index i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self, root, k)  
        IF root.left IS None AND root.right IS None  
            RETURN root.val  

        INITIALIZE graph AS default dictionary WITH list AS default factory  

        FUNCTION dfs(node, parent = None)  
            IF parent IS NOT None  
                APPEND parent.val TO graph AT key node.val  
                APPEND node.val TO graph AT key parent.val  
            IF node.left IS NOT None  
                CALL dfs(node.left, node)  
            IF node.right IS NOT None  
                CALL dfs(node.right, node)  
        END FUNCTION  

        CALL dfs(root)  

        INITIALIZE queue AS deque WITH element k  
        INITIALIZE visited AS empty set  

        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND SET TO current  
            ADD current TO visited  

            IF length OF graph AT key current IS EQUAL TO 1 AND current IS NOT EQUAL TO root.val  
                RETURN current  

            FOR each neighbor IN graph AT key current  
                IF neighbor IS NOT IN visited  
                    APPEND neighbor TO queue  
        END WHILE  

        RETURN -1  
    END FUNCTION  
END CLASS
