CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO LENGTH OF mat1  
        SET k TO LENGTH OF mat1 AT 0  
        SET k TO LENGTH OF mat2  
        SET n TO LENGTH OF mat2 AT 0  
        
        SET result TO new matrix of zeros with m rows and n columns  
        
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                FOR p FROM 0 TO k MINUS 1  
                    INCREMENT result AT i AND j BY mat1 AT i AND p MULTIPLIED BY mat2 AT p AND j  
        RETURN result  
    END FUNCTION  
END CLASS
