CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO new TreeNode with val AS values at index zero  
    SET i TO 1  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i IS LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.left TO new TreeNode with val AS values at index i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.right TO new TreeNode with val AS values at index i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(self, root, k)  
        SET stack TO empty list  
        SET current TO root  
        SET count TO 0  
        WHILE stack IS NOT EMPTY OR current IS NOT None  
            WHILE current IS NOT None  
                APPEND current TO stack  
                SET current TO current.left  
            SET current TO REMOVE last element FROM stack  
            INCREMENT count BY 1  
            IF count IS EQUAL TO k  
                RETURN current.val  
            SET current TO current.right  
    END FUNCTION  
END CLASS
