CLASS Solution
    FUNCTION alienOrder(words)
        SET graph TO mapping with default value as empty set
        SET indegree TO mapping with default value as zero

        SET all_chars TO set of all characters found by joining all words

        FOR i FROM zero TO LENGTH OF words MINUS one
            SET word1 TO words at position i
            SET word2 TO words at position i plus one
            SET min_length TO minimum of LENGTH OF word1 and LENGTH OF word2
            SET found TO false
            FOR j FROM zero TO min_length MINUS one
                IF character at position j in word1 IS NOT EQUAL TO character at position j in word2
                    IF character at position j in word2 IS NOT IN graph at character at position j in word1
                        ADD character at position j in word2 TO graph at character at position j in word1
                        INCREMENT indegree at character at position j in word2 BY one
                    SET found TO true
                    BREAK inner loop
            IF found IS false AND LENGTH OF word1 IS GREATER THAN LENGTH OF word2
                RETURN empty string

        SET queue TO collection of characters in all_chars WHERE indegree of character IS EQUAL TO zero
        SET result TO empty list

        WHILE queue IS NOT empty
            SET char TO first element removed from queue
            APPEND char TO result
            FOR each neighbor IN graph at char
                DECREMENT indegree at neighbor BY one
                IF indegree at neighbor IS EQUAL TO zero
                    ADD neighbor TO queue

        IF LENGTH OF result IS NOT EQUAL TO LENGTH OF all_chars
            RETURN empty string

        RETURN concatenation of all characters in result
    END FUNCTION
END CLASS
