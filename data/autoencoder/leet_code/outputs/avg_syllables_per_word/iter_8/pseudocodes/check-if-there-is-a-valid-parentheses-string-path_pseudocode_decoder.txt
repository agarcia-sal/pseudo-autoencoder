CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF grid AT index 0  
        
        IF (m PLUS n MINUS 1) MODULO 2 IS NOT EQUAL TO 0  
            RETURN False  
        
        INITIALIZE dp AS a three-dimensional list of Booleans with dimensions (m PLUS 1) BY (n PLUS 1) BY (m PLUS n PLUS 1), ALL SET TO False  
        
        SET dp AT indices 0, 1, 0 TO True  
        
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                FOR k FROM 0 TO m PLUS n  
                    IF dp AT indices i, j PLUS 1, k IS True OR dp AT indices i PLUS 1, j, k IS True  
                        IF grid AT i, j IS '(' AND k PLUS 1 LESS THAN m PLUS n PLUS 1  
                            SET dp AT indices i PLUS 1, j PLUS 1, k PLUS 1 TO True  
                        ELSE IF grid AT i, j IS ')' AND k MINUS 1 GREATER THAN OR EQUAL TO 0  
                            SET dp AT indices i PLUS 1, j PLUS 1, k MINUS 1 TO True  
        
        RETURN dp AT indices m, n, 0  
    END FUNCTION  
END CLASS
