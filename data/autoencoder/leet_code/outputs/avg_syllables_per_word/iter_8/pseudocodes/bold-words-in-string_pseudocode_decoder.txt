CLASS Solution  
    FUNCTION boldWords(words, s)  
        SET bold TO list of False repeated LENGTH OF s times  
        
        FOR each word IN words  
            SET start TO POSITION OF word IN s  
            WHILE start IS NOT EQUAL TO NEGATIVE ONE  
                FOR i FROM start TO start PLUS LENGTH OF word MINUS ONE  
                    SET bold AT i TO True  
                SET start TO POSITION OF word IN s STARTING FROM start PLUS ONE  
        
        SET result TO empty list  
        SET i TO 0  
        WHILE i IS LESS THAN LENGTH OF s  
            IF bold AT i IS True  
                APPEND "<b>" TO result  
                WHILE i IS LESS THAN LENGTH OF s AND bold AT i IS True  
                    APPEND s AT i TO result  
                    INCREMENT i BY 1  
                APPEND "</b>" TO result  
            ELSE  
                APPEND s AT i TO result  
                INCREMENT i BY 1  
        
        RETURN CONCATENATION OF elements in result  
    END FUNCTION  
END CLASS
