CLASS Solution  
    FUNCTION kMirror(k, n)  
        FUNCTION is_palindrome(s)  
            RETURN s IS EQUAL TO s REVERSED  
        END FUNCTION  
  
        FUNCTION to_base_k(num, k)  
            IF num IS EQUAL TO 0 THEN  
                RETURN "0"  
            END IF  
            SET digits TO empty list  
            WHILE num IS GREATER THAN 0  
                APPEND remainder of num DIVIDED BY k TO digits  
                SET num TO num DIVIDED BY k WITHOUT remainder  
            END WHILE  
            RETURN concatenation of digits reversed AS strings  
        END FUNCTION  
  
        FUNCTION generate_palindromes(length)  
            IF length IS EQUAL TO 1 THEN  
                FOR i FROM 1 TO 9  
                    YIELD string representation of i  
                END FOR  
            ELSE  
                SET start TO 10 RAISED TO THE POWER OF integer division of (length MINUS 1) BY 2  
                SET end TO 10 RAISED TO THE POWER OF integer division of (length PLUS 1) BY 2  
                FOR half FROM start TO end MINUS 1  
                    SET half_str TO string representation of half  
                    IF length MODULO 2 IS EQUAL TO 0 THEN  
                        SET full TO half_str CONCATENATED WITH reverse of half_str  
                    ELSE  
                        SET full TO half_str CONCATENATED WITH reverse of half_str EXCLUDING last character  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
  
        SET k_mirror_numbers TO empty list  
        SET length TO 1  
        WHILE size of k_mirror_numbers IS LESS THAN n  
            FOR pal IN generate_palindromes(length)  
                SET num TO integer conversion of pal  
                IF is_palindrome(to_base_k(num, k)) THEN  
                    APPEND num TO k_mirror_numbers  
                    IF size of k_mirror_numbers IS EQUAL TO n THEN  
                        BREAK loop  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY 1  
        END WHILE  
  
        RETURN sum of k_mirror_numbers  
    END FUNCTION  
END CLASS
