CLASS Solution  
    FUNCTION maxKilledEnemies(grid)  
        IF grid IS empty OR grid AT 0 IS empty THEN  
            RETURN 0  
        END IF  
        
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF grid AT 0  
        SET max_kills TO 0  
        
        SET row_kills TO 0  
        SET col_kills TO LIST OF n ZEROS  
        
        FOR i FROM 0 TO m MINUS 1 DO  
            FOR j FROM 0 TO n MINUS 1 DO  
                IF j EQUALS 0 OR grid AT i AT (j MINUS 1) EQUALS 'W' THEN  
                    SET row_kills TO 0  
                    FOR k FROM j TO n MINUS 1 DO  
                        IF grid AT i AT k EQUALS 'E' THEN  
                            INCREMENT row_kills BY 1  
                        ELSE IF grid AT i AT k EQUALS 'W' THEN  
                            BREAK  
                        END IF  
                    END FOR  
                END IF  
                
                IF i EQUALS 0 OR grid AT (i MINUS 1) AT j EQUALS 'W' THEN  
                    SET col_kills AT j TO 0  
                    FOR k FROM i TO m MINUS 1 DO  
                        IF grid AT k AT j EQUALS 'E' THEN  
                            INCREMENT col_kills AT j BY 1  
                        ELSE IF grid AT k AT j EQUALS 'W' THEN  
                            BREAK  
                        END IF  
                    END FOR  
                END IF  
                
                IF grid AT i AT j EQUALS '0' THEN  
                    SET max_kills TO MAXIMUM OF max_kills AND (row_kills PLUS col_kills AT j)  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_kills  
    END FUNCTION  
END CLASS
