CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO list of strings from 1 TO n inclusive  
        WHILE n GREATER THAN 1  
            FOR i FROM 0 TO n DIVIDED BY 2 MINUS 1  
                SET teams at position i TO string concatenation of "(" concatenated with teams at i concatenated with "," concatenated with teams at n MINUS 1 MINUS i concatenated with ")"  
            SET n TO n DIVIDED BY 2  
        RETURN teams at position 0  
    END FUNCTION  
END CLASS
