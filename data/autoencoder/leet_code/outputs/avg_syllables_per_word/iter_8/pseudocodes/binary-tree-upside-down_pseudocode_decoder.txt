CLASS TreeNode  
    FUNCTION __init__(self, val SET TO 0, left SET TO None, right SET TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode(values AT index 0)  
    SET i TO 1  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO dequeue the first element from queue  
        IF i LESS THAN length of values AND values AT index i IS NOT None  
            SET node.left TO new TreeNode(values AT index i)  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values AT index i IS NOT None  
            SET node.right TO new TreeNode(values AT index i)  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree(self, root)  
        IF root IS None OR root.left IS None  
            RETURN root  
        SET new_root TO self.upsideDownBinaryTree(root.left)  
        SET root.left.left TO root.right  
        SET root.left.right TO root  
        SET root.left TO None  
        SET root.right TO None  
        RETURN new_root  
    END FUNCTION  
END CLASS
