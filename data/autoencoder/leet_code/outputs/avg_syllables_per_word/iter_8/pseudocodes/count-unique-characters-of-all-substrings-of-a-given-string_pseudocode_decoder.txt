CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO 10 RAISED TO THE POWER OF 9 PLUS 7  
        SET n TO LENGTH OF s  
        SET last TO LIST OF SIZE 26 FILLED WITH -1  
        SET last2 TO LIST OF SIZE 26 FILLED WITH -1  
        SET result TO 0  
        
        FOR EACH index, char IN ENUMERATION OF s  
            SET idx TO ORDINAL VALUE OF char MINUS ORDINAL VALUE OF 'A'  
            SET result TO result PLUS (index MINUS last AT idx) MULTIPLIED BY (last AT idx MINUS last2 AT idx) MODULO MOD  
            SET last2 AT idx TO last AT idx  
            SET last AT idx TO index  
        END FOR  
        
        FOR i FROM 0 TO 25  
            SET result TO result PLUS (n MINUS last AT i) MULTIPLIED BY (last AT i MINUS last2 AT i) MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
