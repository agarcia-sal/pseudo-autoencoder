CLASS Solution  
    FUNCTION addBoldTag(s, words)  
        IF words IS EMPTY  
            RETURN s  
        END IF  
        
        SET n TO LENGTH OF s  
        SET mask TO LIST OF Boolean False WITH LENGTH n  
        
        FOR each word IN words  
            SET start TO 0  
            WHILE start IS LESS THAN OR EQUAL TO n MINUS LENGTH OF word  
                SET start TO POSITION OF word IN s STARTING FROM start  
                IF start IS EQUAL TO -1  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS LENGTH OF word MINUS 1  
                    SET mask AT i TO True  
                END FOR  
                INCREMENT start BY 1  
            END WHILE  
        END FOR  
        
        SET result TO EMPTY LIST  
        SET i TO 0  
        WHILE i IS LESS THAN n  
            IF mask AT i IS True  
                APPEND "<b>" TO result  
                WHILE i IS LESS THAN n AND mask AT i IS True  
                    APPEND s AT i TO result  
                    INCREMENT i BY 1  
                END WHILE  
                APPEND "</b>" TO result  
            ELSE  
                APPEND s AT i TO result  
                INCREMENT i BY 1  
            END IF  
        END WHILE  
        
        RETURN CONCATENATION OF result  
    END FUNCTION  
END CLASS
