CLASS Solution  
    FUNCTION canCompleteCircuit(gas, cost)  
        IF sum of gas IS LESS THAN sum of cost  
            RETURN negative one  
        SET start TO zero  
        SET total_tank TO zero  
        FOR each index i FROM zero TO length of gas MINUS one  
            INCREMENT total_tank BY gas at i MINUS cost at i  
            IF total_tank IS LESS THAN zero  
                SET start TO i PLUS one  
                SET total_tank TO zero  
        RETURN start  
    END FUNCTION  
END CLASS
