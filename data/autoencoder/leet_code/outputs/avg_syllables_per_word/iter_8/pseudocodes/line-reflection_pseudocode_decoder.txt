CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO set of tuples converted from points  
        SET min_x TO minimum x-coordinate among unique_points  
        SET max_x TO maximum x-coordinate among unique_points  
        SET line_of_reflection TO sum of min_x and max_x DIVIDED BY 2  
        FOR each point (x, y) IN unique_points  
            SET reflected_x TO two times line_of_reflection MINUS x  
            IF tuple of (reflected_x, y) NOT IN unique_points  
                RETURN False  
        RETURN True  
    END FUNCTION  
END CLASS
