CLASS Solution  
    FUNCTION findIntegers(n)  
        SET bin_rep TO binary representation of n as string without leading indicator  
        SET length TO length of bin_rep  
        
        INITIALIZE dp AS list of zeros with size length plus 1  
        SET dp at index 0 TO 1  
        SET dp at index 1 TO 2  
        
        FOR i FROM 2 TO length INCLUSIVE  
            SET dp at index i TO dp at index i minus 1 PLUS dp at index i minus 2  
        
        SET result TO 0  
        SET prev_bit TO 0  
        
        FOR i FROM 0 TO length MINUS 1  
            IF character at index i in bin_rep EQUALS '1' THEN  
                INCREMENT result BY dp at index length MINUS i MINUS 1  
                IF prev_bit EQUALS 1 THEN  
                    DECREMENT result BY 1  
                    BREAK loop  
            END IF  
            SET prev_bit TO 1 IF character at index i in bin_rep EQUALS '1' ELSE 0  
        END FOR  
        
        RETURN result PLUS 1  
    END FUNCTION  
END CLASS
