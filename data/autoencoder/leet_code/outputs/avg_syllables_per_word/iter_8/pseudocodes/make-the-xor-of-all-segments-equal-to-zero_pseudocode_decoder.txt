CLASS Solution  
    FUNCTION minChanges(nums, k)  
        DEFINE groups AS DEFAULT DICTIONARY OF LIST  
        FOR i FROM 0 TO LENGTH OF nums MINUS 1  
            APPEND nums AT i TO groups AT i MODULO k  
        DEFINE freq AS LIST OF DEFAULT DICTIONARY OF INTEGER WITH SIZE k  
        DEFINE min_size AS LIST OF INFINITY WITH SIZE k  
        FOR i FROM 0 TO k MINUS 1  
            FOR each num IN groups AT i  
                INCREMENT freq AT i AT num BY 1  
            SET min_size AT i TO LENGTH OF groups AT i  
        DEFINE dp AS LIST OF INFINITY WITH SIZE TWO RAISED TO 10  
        SET dp AT 0 TO 0  
        FOR i FROM 0 TO k MINUS 1  
            DEFINE new_dp AS LIST OF INFINITY WITH SIZE TWO RAISED TO 10  
            SET min_dp TO MINIMUM VALUE IN dp  
            FOR j FROM 0 TO TWO RAISED TO 10 MINUS 1  
                SET new_dp AT j TO min_dp PLUS min_size AT i  
                FOR each num, count IN freq AT i  
                    SET new_dp AT j TO MINIMUM OF new_dp AT j AND dp AT (j XOR num) PLUS min_size AT i MINUS count  
            SET dp TO new_dp  
        RETURN dp AT 0  
    END FUNCTION  
END CLASS
