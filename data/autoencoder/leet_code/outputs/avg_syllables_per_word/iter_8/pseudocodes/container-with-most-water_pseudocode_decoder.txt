CLASS Solution  
    FUNCTION maxArea(height)  
        SET left TO 0  
        SET right TO LENGTH OF height MINUS 1  
        SET max_area TO 0  
        
        WHILE left LESS THAN right  
            SET current_area TO MINIMUM OF height AT left AND height AT right MULTIPLIED BY right MINUS left  
            SET max_area TO MAXIMUM OF max_area AND current_area  
            
            IF height AT left LESS THAN height AT right  
                INCREMENT left BY 1  
            ELSE  
                DECREMENT right BY 1  
        RETURN max_area  
    END FUNCTION  
END CLASS
