CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF maxChoosableInteger MULTIPLIED BY (maxChoosableInteger PLUS 1) DIVIDED BY 2 IS LESS THAN desiredTotal THEN  
            RETURN False  
        END IF  

        IF desiredTotal IS LESS THAN OR EQUAL TO 0 THEN  
            RETURN True  
        END IF  

        SET memo TO empty dictionary  

        FUNCTION can_win(current_total, used_numbers)  
            IF (current_total, used_numbers) EXISTS IN memo THEN  
                RETURN memo VALUE FOR (current_total, used_numbers)  
            END IF  

            FOR number FROM 1 TO maxChoosableInteger DO  
                IF number IS NOT IN used_numbers THEN  
                    IF current_total PLUS number IS GREATER THAN OR EQUAL TO desiredTotal OR  
                       NOT can_win(current_total PLUS number, used_numbers INCLUDING number) THEN  
                        SET memo VALUE FOR (current_total, used_numbers) TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  

            SET memo VALUE FOR (current_total, used_numbers) TO False  
            RETURN False  
        END FUNCTION  

        RETURN can_win(0, 0)  
    END FUNCTION  
END CLASS
