CLASS Solution  
    FUNCTION crackSafe(n, k)  
        SET visited TO empty set  
        SET sequence TO empty list  

        FUNCTION dfs(node)  
            FOR i FROM 0 TO k MINUS 1  
                SET edge TO node CONCATENATED WITH string representation of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH parameter edge WITHOUT its first character  
                    APPEND string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  

        SET start_node TO string of '0' REPEATED n MINUS 1 times  
        CALL dfs WITH parameter start_node  

        SET result TO concatenation of all elements in sequence  
        RETURN result CONCATENATED WITH start_node  
    END FUNCTION  
END CLASS
