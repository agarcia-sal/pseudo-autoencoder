CLASS Solution  
    FUNCTION numWays(steps, arrLen)  
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7  
        SET max_index TO MINIMUM OF steps DIVIDED BY 2 AND arrLen MINUS 1  
        
        SET dp TO LIST OF ZEROS WITH LENGTH max_index PLUS 1  
        SET dp AT POSITION 0 TO 1  
        
        FOR each _ IN RANGE OF steps  
            SET prev_dp TO COPY OF dp  
            FOR i FROM 0 TO max_index  
                SET dp AT POSITION i TO prev_dp AT POSITION i  
                IF i GREATER THAN 0 THEN  
                    SET dp AT POSITION i TO dp AT POSITION i PLUS prev_dp AT POSITION i MINUS 1  
                    SET dp AT POSITION i TO dp AT POSITION i MODULO MOD  
                END IF  
                IF i LESS THAN max_index THEN  
                    SET dp AT POSITION i TO dp AT POSITION i PLUS prev_dp AT POSITION i PLUS 1  
                    SET dp AT POSITION i TO dp AT POSITION i MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp AT POSITION 0  
    END FUNCTION  
END CLASS
