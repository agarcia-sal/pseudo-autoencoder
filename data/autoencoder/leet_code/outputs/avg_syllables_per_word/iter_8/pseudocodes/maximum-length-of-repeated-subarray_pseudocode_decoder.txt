CLASS Solution  
    FUNCTION findLength(nums1, nums2)  
        SET dp TO a 2D array of zeros with dimensions (LENGTH OF nums1 PLUS 1) BY (LENGTH OF nums2 PLUS 1)  
        SET max_length TO 0  
        FOR i FROM 1 TO LENGTH OF nums1  
            FOR j FROM 1 TO LENGTH OF nums2  
                IF element AT position (i MINUS 1) IN nums1 EQUALS element AT position (j MINUS 1) IN nums2 THEN  
                    SET dp AT position [i][j] TO dp AT position [i MINUS 1][j MINUS 1] PLUS 1  
                    IF dp AT position [i][j] GREATER THAN max_length THEN  
                        SET max_length TO dp AT position [i][j]  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
