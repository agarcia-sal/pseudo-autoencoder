CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO 10 raised to the power 9 plus 7  
        INITIALIZE dp AS list of zeros with length LENGTH OF s plus 1  
        SET dp at index 0 TO 1  
  
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS '*' THEN  
                RETURN 9  
            ELSE IF char EQUALS '0' THEN  
                RETURN 0  
            ELSE  
                RETURN 1  
            END IF  
        END FUNCTION  
  
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS '*' AND char2 EQUALS '*' THEN  
                RETURN 15  
            ELSE IF char1 EQUALS '*' THEN  
                IF char2 LESS THAN OR EQUAL TO '6' THEN  
                    RETURN 2  
                ELSE  
                    RETURN 1  
                END IF  
            ELSE IF char2 EQUALS '*' THEN  
                IF char1 EQUALS '1' THEN  
                    RETURN 9  
                ELSE IF char1 EQUALS '2' THEN  
                    RETURN 6  
                ELSE  
                    RETURN 0  
                END IF  
            ELSE  
                SET num TO integer value formed by concatenating char1 and char2  
                IF num GREATER THAN OR EQUAL TO 10 AND num LESS THAN OR EQUAL TO 26 THEN  
                    RETURN 1  
                ELSE  
                    RETURN 0  
                END IF  
            END IF  
        END FUNCTION  
  
        FOR i FROM 1 TO LENGTH OF s DO  
            SET dp at index i TO dp at index i PLUS dp at index i MINUS 1 TIMES ways_to_decode_single(s at index i MINUS 1) MODULO MOD  
            IF i GREATER THAN 1 THEN  
                SET dp at index i TO dp at index i PLUS dp at index i MINUS 2 TIMES ways_to_decode_pair(s at index i MINUS 2, s at index i MINUS 1) MODULO MOD  
            END IF  
        END FOR  
  
        RETURN dp at last index  
    END FUNCTION  
END CLASS
