CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET num TO 0  
            SET sign TO '+'  
            WHILE LENGTH OF s GREATER THAN 0  
                SET char TO REMOVE FIRST ELEMENT FROM s  
                IF char IS DIGIT  
                    SET num TO num MULTIPLIED BY 10 PLUS numeric value of char  
                END IF  
                IF char EQUALS '('  
                    SET num TO helper(s)  
                END IF  
                IF char IN '+-*/()' OR LENGTH OF s EQUALS 0  
                    IF sign EQUALS '+'  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS '-'  
                        APPEND negative num TO stack  
                    ELSE IF sign EQUALS '*'  
                        SET last element of stack TO last element of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS '/'  
                        SET last element of stack TO integer part of last element of stack DIVIDED BY num WITH TRUNCATION TOWARDS ZERO  
                    END IF  
                    SET sign TO char  
                    SET num TO 0  
                END IF  
                IF char EQUALS ')'  
                    EXIT LOOP  
                END IF  
            END WHILE  
            RETURN sum of all elements in stack  
        END FUNCTION  
        
        SET s TO list of characters from s  
        RETURN helper(s)  
    END FUNCTION  
END CLASS
