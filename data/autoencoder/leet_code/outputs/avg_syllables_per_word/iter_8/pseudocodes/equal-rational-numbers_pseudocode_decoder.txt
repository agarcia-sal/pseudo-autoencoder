CLASS Solution
    FUNCTION isRationalEqual(s, t)
        SET ratios TO list containing 1, 1 DIVIDED BY 9, 1 DIVIDED BY 99, 1 DIVIDED BY 999, 1 DIVIDED BY 9999

        FUNCTION valueOf(s)
            IF s DOES NOT CONTAIN '(' THEN
                RETURN conversion of s TO float
            END IF

            SET leftParenIndex TO position of '(' in s
            SET rightParenIndex TO position of ')' in s
            SET dotIndex TO position of '.' in s

            SET integerAndNonRepeating TO conversion of substring of s from start TO leftParenIndex TO float
            SET nonRepeatingLength TO leftParenIndex MINUS dotIndex MINUS 1

            SET repeating TO integer value of substring of s from leftParenIndex PLUS 1 TO rightParenIndex
            SET repeatingLength TO rightParenIndex MINUS leftParenIndex MINUS 1

            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY 0.1 TO THE POWER OF nonRepeatingLength MULTIPLIED BY ratios AT repeatingLength
        END FUNCTION

        RETURN absolute value of (valueOf(s) MINUS valueOf(t)) LESS THAN 0.000000001
    END FUNCTION
END CLASS
