CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH val AS values AT POSITION 0  
    SET i TO 1  
    SET queue TO NEW empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i IS LESS THAN LENGTH OF values AND values AT POSITION i IS NOT None  
            SET node.left TO NEW TreeNode WITH val AS values AT POSITION i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values AT POSITION i IS NOT None  
            SET node.right TO NEW TreeNode WITH val AS values AT POSITION i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS None  
                RETURN 0  
            RETURN node.val PLUS subtree_sum(node.left) PLUS subtree_sum(node.right)  
        END FUNCTION  

        SET total_sum TO subtree_sum(root)  

        IF total_sum MODULO 2 IS NOT 0  
            RETURN False  

        SET target TO total_sum DIVIDED BY 2  
        SET found TO LIST CONTAINING False  

        FUNCTION can_split(node)  
            IF node IS None  
                RETURN 0  
            SET current_sum TO node.val PLUS can_split(node.left) PLUS can_split(node.right)  
            IF current_sum IS EQUAL TO target AND node IS NOT root  
                SET found AT POSITION 0 TO True  
            RETURN current_sum  
        END FUNCTION  

        CALL can_split(root)  
        RETURN found AT POSITION 0  
    END FUNCTION  
END CLASS
