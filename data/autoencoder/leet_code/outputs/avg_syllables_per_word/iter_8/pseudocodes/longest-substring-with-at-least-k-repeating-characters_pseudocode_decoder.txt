CLASS Solution  
    FUNCTION longestSubstring(s, k)  
        IF LENGTH OF s LESS THAN k THEN  
            RETURN 0  
        END IF  
        
        SET count TO frequency count of characters in s  
        
        FOR each char IN count  
            IF count of char LESS THAN k THEN  
                RETURN maximum value AMONG longestSubstring(substring, k) FOR each substring IN s SPLIT BY char  
            END IF  
        END FOR  
        
        RETURN LENGTH OF s  
    END FUNCTION  
END CLASS
