CLASS Solution  
    FUNCTION originalDigits(s)  
        SET count TO frequency count of each character in s  
        SET digits TO list of size 10 initialized with zeros  
        
        SET digits at index 0 TO count of character 'z'  
        SET digits at index 2 TO count of character 'w'  
        SET digits at index 4 TO count of character 'u'  
        SET digits at index 6 TO count of character 'x'  
        SET digits at index 8 TO count of character 'g'  
        
        SET digits at index 3 TO count of character 'h' MINUS digits at index 8  
        SET digits at index 5 TO count of character 'f' MINUS digits at index 4  
        SET digits at index 7 TO count of character 's' MINUS digits at index 6  
        SET digits at index 9 TO count of character 'i' MINUS digits at index 5 MINUS digits at index 6 MINUS digits at index 8  
        SET digits at index 1 TO count of character 'o' MINUS digits at index 0 MINUS digits at index 2 MINUS digits at index 4  
        
        SET result TO empty list  
        FOR i FROM 0 TO 9  
            APPEND string of digit i repeated digits at index i times TO result  
        
        RETURN concatenation of all strings in result  
    END FUNCTION  
END CLASS
