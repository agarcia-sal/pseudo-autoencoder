CLASS Solution  
    FUNCTION findKthNumber(n, k)  
        FUNCTION count_steps(prefix, n)  
            SET next_prefix TO prefix plus 1  
            SET steps TO 0  
            WHILE prefix less than or equal to n  
                INCREMENT steps BY minimum of (n minus prefix plus 1) and (next_prefix minus prefix)  
                SET prefix TO prefix multiplied by 10  
                SET next_prefix TO next_prefix multiplied by 10  
            END WHILE  
            RETURN steps  
        END FUNCTION  

        SET current TO 1  
        DECREMENT k BY 1  
        WHILE k greater than 0  
            SET steps TO count_steps(current, n)  
            IF steps less than or equal to k THEN  
                INCREMENT current BY 1  
                DECREMENT k BY steps  
            ELSE  
                SET current TO current multiplied by 10  
                DECREMENT k BY 1  
            END IF  
        END WHILE  
        RETURN current  
    END FUNCTION  
END CLASS
