CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of integers from 0 TO n  
        SET self.rank TO list of zeros with length n plus 1  
        SET self.count TO n  
    END FUNCTION  
      
    FUNCTION find(self, u)  
        IF self.parent AT u IS NOT EQUAL TO u THEN  
            SET self.parent AT u TO self.find(self.parent AT u)  
        END IF  
        RETURN self.parent AT u  
    END FUNCTION  
      
    FUNCTION union(self, u, v)  
        SET rootU TO self.find(u)  
        SET rootV TO self.find(v)  
        IF rootU IS NOT EQUAL TO rootV THEN  
            IF self.rank AT rootU IS GREATER THAN self.rank AT rootV THEN  
                SET self.parent AT rootV TO rootU  
            ELSE IF self.rank AT rootU IS LESS THAN self.rank AT rootV THEN  
                SET self.parent AT rootU TO rootV  
            ELSE  
                SET self.parent AT rootV TO rootU  
                INCREMENT self.rank AT rootU BY 1  
            END IF  
            DECREMENT self.count BY 1  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION maxNumEdgesToRemove(self, n, edges)  
        SET uf_common TO new UnionFind with parameter n  
        SET uf_alice TO new UnionFind with parameter n  
        SET uf_bob TO new UnionFind with parameter n  
          
        SET used_edges TO 0  
          
        FOR each t, u, v IN edges  
            IF t IS EQUAL TO 3 THEN  
                IF uf_common.union(u, v) THEN  
                    CALL uf_alice.union(u, v)  
                    CALL uf_bob.union(u, v)  
                    INCREMENT used_edges BY 1  
                END IF  
            END IF  
        END FOR  
          
        FOR each t, u, v IN edges  
            IF t IS EQUAL TO 1 AND uf_alice.union(u, v) THEN  
                INCREMENT used_edges BY 1  
            ELSE IF t IS EQUAL TO 2 AND uf_bob.union(u, v) THEN  
                INCREMENT used_edges BY 1  
            END IF  
        END FOR  
          
        IF uf_alice.count IS EQUAL TO 1 AND uf_bob.count IS EQUAL TO 1 THEN  
            RETURN LENGTH OF edges MINUS used_edges  
        ELSE  
            RETURN -1  
        END IF  
    END FUNCTION  
END CLASS
