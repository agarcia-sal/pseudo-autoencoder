CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO set of elements x where each x and y are taken from fronts and backs respectively and x is equal to y  
        SET all_numbers TO set of all elements from fronts combined with backs  
        SET good_numbers TO all_numbers excluding elements in same  
        IF good_numbers is not empty  
            RETURN minimum element in good_numbers  
        ELSE  
            RETURN 0  
        END IF  
    END FUNCTION  
END CLASS
