CLASS Solution  
    FUNCTION maximumScore(nums, k)  
        SET left TO k  
        SET right TO k  
        SET current_min TO nums AT k  
        SET max_score TO current_min  
        
        WHILE left GREATER THAN 0 OR right LESS THAN LENGTH OF nums MINUS 1  
            IF left EQUALS 0  
                INCREMENT right BY 1  
            ELSE IF right EQUALS LENGTH OF nums MINUS 1  
                DECREMENT left BY 1  
            ELSE IF nums AT left MINUS 1 GREATER THAN OR EQUAL TO nums AT right PLUS 1  
                DECREMENT left BY 1  
            ELSE  
                INCREMENT right BY 1  
            
            SET current_min TO MINIMUM OF current_min, nums AT left, nums AT right  
            SET score TO current_min MULTIPLIED BY (right MINUS left PLUS 1)  
            SET max_score TO MAXIMUM OF max_score, score  
        
        RETURN max_score  
    END FUNCTION  
END CLASS
