CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO list of size 15 filled with infinity  
        FOR each pair f, r IN tires  
            SET total_time TO 0  
            FOR i FROM 0 TO 14  
                SET lap_time TO f MULTIPLIED BY r TO THE POWER OF i  
                IF lap_time GREATER THAN changeTime PLUS f  
                    BREAK the inner loop  
                INCREMENT total_time BY lap_time  
                SET min_times at index i TO the minimum of min_times at index i and total_time  
        SET dp TO list of size numLaps PLUS 1 filled with infinity  
        SET dp at index 0 TO 0  
        FOR i FROM 1 TO numLaps  
            FOR j FROM 0 TO the minimum of i MINUS 1 and 14  
                SET dp at index i TO the minimum of dp at index i and dp at index i MINUS j MINUS 1 PLUS min_times at index j PLUS changeTime  
        RETURN dp at index numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
