CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x IS LESS THAN 0 OR x IS GREATER THAN OR EQUAL TO LENGTH OF grid OR y IS LESS THAN 0 OR y IS GREATER THAN OR EQUAL TO LENGTH OF grid AT 0 OR (x, y) IS IN visited OR grid AT x AT y IS EQUAL TO 0  
                RETURN  
            ADD (x, y) TO visited  
            FOR each (dx, dy) IN [(-1, 0), (1, 0), (0, -1), (0, 1)]  
                CALL dfs WITH (x PLUS dx, y PLUS dy, visited)  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO empty set  
            SET land_cells TO list of (i, j) FOR i FROM 0 TO LENGTH OF grid MINUS 1 AND j FROM 0 TO LENGTH OF grid AT 0 MINUS 1 WHERE grid AT i AT j IS EQUAL TO 1  
            IF land_cells IS EMPTY  
                RETURN True  
            CALL dfs WITH (land_cells AT 0 AT 0, land_cells AT 0 AT 1, visited)  
            RETURN LENGTH OF visited IS EQUAL TO LENGTH OF land_cells  
        END FUNCTION  

        IF is_connected() IS False  
            RETURN 0  

        FOR i FROM 0 TO LENGTH OF grid MINUS 1  
            FOR j FROM 0 TO LENGTH OF grid AT 0 MINUS 1  
                IF grid AT i AT j IS EQUAL TO 1  
                    SET grid AT i AT j TO 0  
                    IF is_connected() IS False  
                        RETURN 1  
                    SET grid AT i AT j TO 1  
        RETURN 2  
    END FUNCTION  
END CLASS
