CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO LENGTH OF arr  
        SET min_heap TO empty collection representing a min-priority queue  
        
        FOR i FROM 0 TO n MINUS 2  
            INSERT fraction formed by arr at i DIVIDED BY arr at n MINUS 1 WITH indices i and n MINUS 1 INTO min_heap  
        
        FOR count FROM 1 TO k MINUS 1  
            EXTRACT smallest fraction WITH indices i and j FROM min_heap  
            IF j MINUS 1 GREATER THAN i THEN  
                INSERT fraction formed by arr at i DIVIDED BY arr at j MINUS 1 WITH indices i and j MINUS 1 INTO min_heap  
        
        EXTRACT smallest fraction WITH indices i and j FROM min_heap  
        RETURN list containing arr at i and arr at j  
    END FUNCTION  
END CLASS
