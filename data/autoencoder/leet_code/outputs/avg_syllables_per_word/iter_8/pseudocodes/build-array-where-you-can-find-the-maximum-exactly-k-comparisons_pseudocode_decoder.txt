CLASS Solution  
    FUNCTION numOfArrays(n, m, k)  
        SET MOD TO 10 power 9 plus 7  
        
        INITIALIZE dp AS a three-dimensional list of size (n plus 1) by (m plus 1) by (k plus 1) FILLED WITH zero  
        
        FOR j FROM 1 TO m INCLUSIVE  
            SET dp at 1, j, 1 TO 1  
        
        FOR i FROM 2 TO n INCLUSIVE  
            FOR j FROM 1 TO m INCLUSIVE  
                FOR p FROM 1 TO k INCLUSIVE  
                    FOR x FROM 1 TO j minus 1 INCLUSIVE  
                        SET dp at i, j, p TO dp at i, j, p PLUS dp at i minus 1, x, p minus 1  
                        SET dp at i, j, p TO dp at i, j, p MODULO MOD  
                    
                    SET dp at i, j, p TO dp at i, j, p PLUS dp at i minus 1, j, p MULTIPLIED BY j  
                    SET dp at i, j, p TO dp at i, j, p MODULO MOD  
        
        SET result TO 0  
        FOR j FROM 1 TO m INCLUSIVE  
            SET result TO result PLUS dp at n, j, k  
            SET result TO result MODULO MOD  
        
        RETURN result  
    END FUNCTION  
END CLASS
