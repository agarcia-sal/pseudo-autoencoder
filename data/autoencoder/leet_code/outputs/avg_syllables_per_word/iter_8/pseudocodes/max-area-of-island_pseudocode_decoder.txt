CLASS Solution  
    FUNCTION maxAreaOfIsland(grid)  
        IF grid IS EMPTY  
            RETURN 0  
        END IF  
        
        SET max_area TO 0  
        SET rows TO LENGTH OF grid  
        SET cols TO LENGTH OF FIRST ELEMENT IN grid  
        
        FUNCTION dfs(r, c)  
            IF r LESS THAN 0 OR r GREATER THAN OR EQUAL TO rows OR c LESS THAN 0 OR c GREATER THAN OR EQUAL TO cols OR grid AT POSITION r,c EQUALS 0  
                RETURN 0  
            END IF  
            SET grid AT POSITION r,c TO 0  
            SET area TO 1  
            INCREMENT area BY dfs(r PLUS 1, c)  
            INCREMENT area BY dfs(r MINUS 1, c)  
            INCREMENT area BY dfs(r, c PLUS 1)  
            INCREMENT area BY dfs(r, c MINUS 1)  
            RETURN area  
        END FUNCTION  
        
        FOR r FROM 0 TO rows MINUS 1  
            FOR c FROM 0 TO cols MINUS 1  
                IF grid AT POSITION r,c EQUALS 1  
                    SET max_area TO GREATER OF max_area AND dfs(r, c)  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
