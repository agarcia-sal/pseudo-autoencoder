CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        INITIALIZE dp AS 2D array WITH dimensions (m plus 1) by (n plus 1) FILLED WITH zeros  
        
        FOR each string s IN strs  
            SET count_0 TO number of zeros IN s  
            SET count_1 TO number of ones IN s  
            
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp[i][j] TO the greater value BETWEEN dp[i][j] AND dp[i minus count_0][j minus count_1] plus 1  
        
        RETURN dp[m][n]  
    END FUNCTION  
END CLASS
