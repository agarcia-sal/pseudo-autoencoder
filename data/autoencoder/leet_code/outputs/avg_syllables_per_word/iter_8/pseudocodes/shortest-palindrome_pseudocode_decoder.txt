CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s IS empty  
            RETURN s  
        END IF  
          
        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS reverse_of check_str  
        END FUNCTION  
          
        FOR i FROM length of s DOWNTO 0  
            IF is_palindrome(substring of s FROM start TO i)  
                SET to_add TO reverse_of substring of s FROM i TO end  
                RETURN concatenation of to_add AND s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
