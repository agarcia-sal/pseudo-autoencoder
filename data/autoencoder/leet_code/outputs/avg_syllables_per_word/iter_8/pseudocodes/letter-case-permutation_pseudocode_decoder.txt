CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO LENGTH OF t THEN  
                APPEND concatenation OF t TO ans  
                RETURN  
            END IF  
            CALL dfs(i INCREMENTED BY 1)  
            IF t AT i IS AN ALPHABET LETTER THEN  
                TOGGLE CASE OF t AT i  
                CALL dfs(i INCREMENTED BY 1)  
            END IF  
        END FUNCTION  
      
        SET t TO list OF CHARACTERS FROM s  
        SET ans TO EMPTY LIST  
        CALL dfs WITH 0  
        RETURN ans  
    END FUNCTION  
END CLASS
