CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH val AS values AT INDEX zero AND left AS None AND right AS None  
    SET i TO one  
    INITIALIZE queue AS new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO queue REMOVE FIRST ELEMENT  
        IF i IS LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.left TO NEW TreeNode WITH val AS values AT INDEX i AND left AS None AND right AS None  
            APPEND node.left TO queue  
        INCREMENT i BY one  
        IF i IS LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.right TO NEW TreeNode WITH val AS values AT INDEX i AND left AS None AND right AS None  
            APPEND node.right TO queue  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str(self, root)  
        FUNCTION helper(node)  
            IF node IS None  
                RETURN empty string  
            SET result TO string representation OF node.val  
            IF node.left IS NOT None OR node.right IS NOT None  
                SET result TO result CONCATENATED WITH "(" CONCATENATED WITH helper(node.left) CONCATENATED WITH ")"  
            IF node.right IS NOT None  
                SET result TO result CONCATENATED WITH "(" CONCATENATED WITH helper(node.right) CONCATENATED WITH ")"  
            RETURN result  
        END FUNCTION  
        RETURN helper(root)  
    END FUNCTION  
END CLASS
