CLASS Solution  
    FUNCTION searchRange(nums, target)  
        FUNCTION find_left(nums, target)  
            SET left TO 0  
            SET right TO LENGTH OF nums MINUS 1  
            WHILE left IS LESS THAN OR EQUAL TO right  
                SET mid TO (left PLUS right) DIVIDED BY 2  
                IF nums AT mid IS LESS THAN target  
                    SET left TO mid PLUS 1  
                ELSE  
                    SET right TO mid MINUS 1  
            RETURN left  
        END FUNCTION  

        FUNCTION find_right(nums, target)  
            SET left TO 0  
            SET right TO LENGTH OF nums MINUS 1  
            WHILE left IS LESS THAN OR EQUAL TO right  
                SET mid TO (left PLUS right) DIVIDED BY 2  
                IF nums AT mid IS LESS THAN OR EQUAL TO target  
                    SET left TO mid PLUS 1  
                ELSE  
                    SET right TO mid MINUS 1  
            RETURN right  
        END FUNCTION  

        SET left_index TO find_left(nums, target)  
        SET right_index TO find_right(nums, target)  

        IF left_index IS LESS THAN OR EQUAL TO right_index AND right_index IS GREATER THAN OR EQUAL TO 0 AND right_index IS LESS THAN LENGTH OF nums AND nums AT left_index IS EQUAL TO target  
            RETURN LIST CONTAINING left_index AND right_index  
        ELSE  
            RETURN LIST CONTAINING MINUS 1 AND MINUS 1  
    END FUNCTION  
END CLASS
