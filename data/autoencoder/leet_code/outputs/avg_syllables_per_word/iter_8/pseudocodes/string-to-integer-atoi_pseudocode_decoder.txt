CLASS Solution  
    FUNCTION myAtoi(s)  
        SET INT_MAX TO 2 RAISED TO THE POWER OF 31 MINUS 1  
        SET INT_MIN TO NEGATIVE 2 RAISED TO THE POWER OF 31  

        SET result TO 0  
        SET index TO 0  
        SET n TO LENGTH OF s  
        SET sign TO 1  

        WHILE index LESS THAN n AND CHARACTER AT s AT index EQUALS SPACE  
            INCREMENT index BY 1  

        IF index LESS THAN n AND (CHARACTER AT s AT index EQUALS PLUS OR CHARACTER AT s AT index EQUALS MINUS)  
            IF CHARACTER AT s AT index EQUALS MINUS  
                SET sign TO NEGATIVE 1  
            ELSE  
                SET sign TO 1  
            INCREMENT index BY 1  

        WHILE index LESS THAN n AND CHARACTER AT s AT index IS A DIGIT  
            SET digit TO NUMERIC VALUE OF CHARACTER AT s AT index  
            IF result GREATER THAN (INT_MAX MINUS digit) DIVIDED BY 10  
                IF sign EQUALS 1  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
            SET result TO result MULTIPLIED BY 10 PLUS digit  
            INCREMENT index BY 1  

        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
