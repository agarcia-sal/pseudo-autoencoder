CLASS Solution  
    FUNCTION totalHammingDistance(nums)  
        SET total_distance TO 0  
        FOR i FROM 0 TO 31  
            SET count_ones TO 0  
            FOR each num IN nums  
                IF the i-th bit of num IS 1  
                    INCREMENT count_ones BY 1  
            SET count_zeros TO LENGTH OF nums MINUS count_ones  
            INCREMENT total_distance BY count_ones MULTIPLIED BY count_zeros  
        RETURN total_distance  
    END FUNCTION  
END CLASS
