CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL arr2.sort()  
        
        SET dp TO dictionary with key -1 and value 0  
        
        FOR each num IN arr1  
            SET new_dp TO empty dictionary  
            FOR each last_num, ops IN dp  
                IF num GREATER THAN last_num THEN  
                    SET new_dp[num] TO minimum of new_dp.get(num, infinity) AND ops  
                END IF  
                
                SET idx TO result of binary search for position to insert last_num in arr2 (right side)  
                IF idx LESS THAN length of arr2 THEN  
                    SET new_dp[arr2 at idx] TO minimum of new_dp.get(arr2 at idx, infinity) AND ops PLUS 1  
                END IF  
            END FOR  
            
            IF new_dp IS empty THEN  
                RETURN -1  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN minimum value in dp  
    END FUNCTION  
END CLASS
