CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH val values AT POSITION 0  
    SET i TO 1  
    SET queue TO NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT POSITION i IS NOT None  
            SET node.left TO NEW TreeNode WITH val values AT POSITION i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT POSITION i IS NOT None  
            SET node.right TO NEW TreeNode WITH val values AT POSITION i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom(self, root)  
        IF root IS None  
            RETURN EMPTY LIST  
        END IF  
        SET result TO EMPTY LIST  
        SET queue TO NEW deque CONTAINING root  
        WHILE queue IS NOT EMPTY  
            SET level_size TO LENGTH OF queue  
            SET current_level TO EMPTY LIST  
            FOR EACH _ IN RANGE FROM 0 TO level_size MINUS 1  
                SET node TO REMOVE FIRST ELEMENT FROM queue  
                APPEND node.val TO current_level  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result IN REVERSED ORDER  
    END FUNCTION  
END CLASS
