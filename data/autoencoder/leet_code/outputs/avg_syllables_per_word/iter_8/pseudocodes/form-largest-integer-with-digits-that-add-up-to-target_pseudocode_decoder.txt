CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        SET cost_to_digit TO DICTIONARY mapping each cost value c TO string of its digit index plus one  
        SET dp TO LIST with first element empty string and following target elements as string "-1"  
        FOR t FROM 1 TO target INCLUSIVE  
            FOR each c, d IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND dp at position t MINUS c NOT EQUAL TO "-1"  
                    SET candidate TO concatenation of d and dp at position t MINUS c  
                    IF dp at position t EQUAL TO "-1" OR length of candidate GREATER THAN length of dp at position t OR  
                       (length of candidate EQUAL TO length of dp at position t AND candidate GREATER THAN dp at position t)  
                        SET dp at position t TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        IF dp at position target NOT EQUAL TO "-1"  
            RETURN dp at position target  
        ELSE  
            RETURN "0"  
        END IF  
    END FUNCTION  
END CLASS
