CLASS Solution  
    FUNCTION countPrimes(n)  
        IF n LESS THAN OR EQUAL TO 2 THEN  
            RETURN 0  
        END IF  
        
        SET is_prime TO list of True repeated n times  
        SET is_prime at index 0 TO False  
        SET is_prime at index 1 TO False  
        
        FOR start FROM 2 TO integer part of square root of n PLUS 1  
            IF is_prime at index start IS True THEN  
                FOR multiple FROM start times start TO n MINUS 1 STEP start  
                    SET is_prime at index multiple TO False  
                END FOR  
            END IF  
        END FOR  
        
        RETURN count of True values in is_prime  
    END FUNCTION  
END CLASS
