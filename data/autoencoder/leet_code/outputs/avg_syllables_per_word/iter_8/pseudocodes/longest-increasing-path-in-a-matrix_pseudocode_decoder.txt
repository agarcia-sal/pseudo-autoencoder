CLASS Solution  
    FUNCTION longestIncreasingPath(matrix)  
        IF matrix IS EMPTY OR matrix AT 0 IS EMPTY THEN  
            RETURN 0  
        END IF  
        
        SET m TO LENGTH OF matrix  
        SET n TO LENGTH OF matrix AT 0  
        SET directions TO LIST OF (1, 0), (-1, 0), (0, 1), (0, -1)  
        SET dp TO 2D LIST OF SIZE m BY n WITH ALL VALUES AS -1  
        
        FUNCTION dfs(x, y)  
            IF dp AT x AT y IS NOT EQUAL TO -1 THEN  
                RETURN dp AT x AT y  
            END IF  
            
            SET max_length TO 1  
            FOR EACH (dx, dy) IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO 0 AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO 0 AND ny IS LESS THAN n AND matrix AT nx AT ny IS GREATER THAN matrix AT x AT y THEN  
                    SET max_length TO GREATER OF max_length AND 1 PLUS dfs(nx, ny)  
                END IF  
            END FOR  
            
            SET dp AT x AT y TO max_length  
            RETURN max_length  
        END FUNCTION  
        
        SET longest_path TO 0  
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                SET longest_path TO GREATER OF longest_path AND dfs(i, j)  
            END FOR  
        END FOR  
        
        RETURN longest_path  
    END FUNCTION  
END CLASS
