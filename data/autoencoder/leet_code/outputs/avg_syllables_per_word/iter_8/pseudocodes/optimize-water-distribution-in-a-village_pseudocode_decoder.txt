CLASS Solution  
    FUNCTION minCostToSupplyWater(n, wells, pipes)  
        FOR i FROM 0 TO n MINUS 1  
            APPEND list containing 0, i PLUS 1, and wells at i TO pipes  
        
        SORT pipes BY third element in each sublist ASCENDING  
        
        SET parent TO list of integers from 0 TO n  
        
        FUNCTION find(x)  
            IF parent at x IS NOT EQUAL TO x  
                SET parent at x TO find(parent at x)  
            END IF  
            RETURN parent at x  
        END FUNCTION  
        
        FUNCTION union(x, y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX IS NOT EQUAL TO rootY  
                SET parent at rootX TO rootY  
                RETURN TRUE  
            END IF  
            RETURN FALSE  
        END FUNCTION  
        
        SET min_cost TO 0  
        
        FOR each house1, house2, cost IN pipes  
            IF union(house1, house2) IS TRUE  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
