CLASS Solution  
    FUNCTION numberOfArrays(s, k)  
        SET MOD TO 10 RAISED TO THE POWER 9 PLUS 7  
        SET n TO LENGTH OF s  
        
        INITIALIZE dp AS LIST OF SIZE n PLUS 1 WITH ALL ELEMENTS SET TO 0  
        SET dp AT INDEX 0 TO 1  
        
        FOR i FROM 1 TO n INCLUSIVE  
            FOR j FROM 0 TO i MINUS 1  
                IF CHARACTER AT s AT j IS NOT ZERO AND INTEGER VALUE OF SUBSTRING s FROM j TO i IS GREATER THAN OR EQUAL TO 1 AND LESS THAN OR EQUAL TO k  
                    SET dp AT INDEX i TO SUM OF dp AT INDEX i AND dp AT INDEX j MODULO MOD  
        
        RETURN dp AT INDEX n  
    END FUNCTION  
END CLASS
