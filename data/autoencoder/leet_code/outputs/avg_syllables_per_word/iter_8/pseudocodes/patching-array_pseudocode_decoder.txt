CLASS Solution  
    FUNCTION minPatches(nums, n)  
        SET missing TO 1  
        SET patches TO 0  
        SET i TO 0  
        
        WHILE missing IS LESS THAN OR EQUAL TO n  
            IF i IS LESS THAN LENGTH OF nums AND nums AT i IS LESS THAN OR EQUAL TO missing  
                INCREMENT missing BY nums AT i  
                INCREMENT i BY 1  
            ELSE  
                INCREMENT missing BY missing  
                INCREMENT patches BY 1  
            END IF  
        END WHILE  
        
        RETURN patches  
    END FUNCTION  
END CLASS
