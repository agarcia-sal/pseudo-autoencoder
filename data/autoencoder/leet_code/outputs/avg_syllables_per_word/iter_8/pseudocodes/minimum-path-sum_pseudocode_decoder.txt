CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid IS EMPTY OR grid AT INDEX 0 IS EMPTY  
            RETURN 0  
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF grid AT INDEX 0  
        CREATE dp AS 2D list OF SIZE m BY n FILLED WITH 0  
        SET dp AT INDEX 0 AT INDEX 0 TO grid AT INDEX 0 AT INDEX 0  
        FOR j FROM 1 TO n MINUS 1  
            SET dp AT INDEX 0 AT INDEX j TO dp AT INDEX 0 AT INDEX j MINUS 1 PLUS grid AT INDEX 0 AT INDEX j  
        FOR i FROM 1 TO m MINUS 1  
            SET dp AT INDEX i AT INDEX 0 TO dp AT INDEX i MINUS 1 AT INDEX 0 PLUS grid AT INDEX i AT INDEX 0  
        FOR i FROM 1 TO m MINUS 1  
            FOR j FROM 1 TO n MINUS 1  
                SET dp AT INDEX i AT INDEX j TO MINIMUM OF dp AT INDEX i MINUS 1 AT INDEX j AND dp AT INDEX i AT INDEX j MINUS 1 PLUS grid AT INDEX i AT INDEX j  
        RETURN dp AT INDEX m MINUS 1 AT INDEX n MINUS 1  
    END FUNCTION  
END CLASS
