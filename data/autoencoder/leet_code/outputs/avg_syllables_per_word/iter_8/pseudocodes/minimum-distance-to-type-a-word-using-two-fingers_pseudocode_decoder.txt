CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 IS None  
                RETURN 0  
            SET x1, y1 TO DIVISION AND MODULO OF ASCII VALUE OF c1 MINUS ASCII VALUE OF 'A' BY 6  
            SET x2, y2 TO DIVISION AND MODULO OF ASCII VALUE OF c2 MINUS ASCII VALUE OF 'A' BY 6  
            RETURN ABSOLUTE VALUE OF x1 MINUS x2 PLUS ABSOLUTE VALUE OF y1 MINUS y2  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2)  
            IF i EQUALS LENGTH OF word  
                RETURN 0  
            SET c TO CHARACTER AT POSITION i IN word  
            SET d1 TO distance(f1, c) PLUS dp(i INCREMENTED BY 1, c, f2)  
            SET d2 TO distance(f2, c) PLUS dp(i INCREMENTED BY 1, f1, c)  
            RETURN MINIMUM OF d1 AND d2  
        END FUNCTION  
        
        RETURN dp(0, None, None)  
    END FUNCTION  
END CLASS
