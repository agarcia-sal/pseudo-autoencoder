CLASS Solution  
    FUNCTION findClosestElements(arr, k, x)  
        SET left TO position where x would be inserted in arr to keep it sorted MINUS 1  
        SET right TO left PLUS 1  
  
        WHILE difference between right and left MINUS 1 IS LESS THAN k  
            IF left EQUALS -1 OR (right IS LESS THAN length of arr AND absolute value of arr[right] MINUS x IS LESS THAN absolute value of arr[left] MINUS x)  
                INCREMENT right BY 1  
            ELSE  
                DECREMENT left BY 1  
        RETURN sublist of arr FROM left PLUS 1 TO right  
    END FUNCTION  
END CLASS
