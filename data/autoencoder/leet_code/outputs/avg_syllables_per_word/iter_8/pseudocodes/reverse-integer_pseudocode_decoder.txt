CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN 0 THEN  
            SET sign TO -1  
        ELSE  
            SET sign TO 1  
        END IF  
        
        SET reversed_x TO integer value of the reverse of the string representation of the absolute value of x  
        
        SET result TO sign MULTIPLIED BY reversed_x  
        
        IF result LESS THAN negative of 2 raised to 31 OR result GREATER THAN 2 raised to 31 MINUS 1 THEN  
            RETURN 0  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
