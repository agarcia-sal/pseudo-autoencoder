CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO dictionary mapping each course from 0 TO numCourses MINUS 1 TO empty list  
        SET in_degrees TO list of zeros with length numCourses  
        
        FOR each pair dest, src IN prerequisites  
            APPEND dest TO adj_list AT src  
            INCREMENT in_degrees AT dest BY 1  
        
        SET queue TO a new deque containing courses from 0 TO numCourses MINUS 1 WHERE in_degrees AT course EQUALS 0  
        SET topological_order TO empty list  
        
        WHILE queue IS NOT empty  
            SET current TO remove and return leftmost element of queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN adj_list AT current  
                DECREMENT in_degrees AT neighbor BY 1  
                IF in_degrees AT neighbor EQUALS 0 THEN  
                    APPEND neighbor TO queue  
        
        IF LENGTH OF topological_order EQUALS numCourses THEN  
            RETURN topological_order  
        ELSE  
            RETURN empty list  
    END FUNCTION  
END CLASS
