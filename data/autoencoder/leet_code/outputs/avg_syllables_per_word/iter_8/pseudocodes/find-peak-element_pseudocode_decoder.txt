CLASS Solution  
    FUNCTION findPeakElement(nums)  
        SET left TO 0  
        SET right TO LENGTH OF nums MINUS 1  
        WHILE left IS LESS THAN right  
            SET mid TO (left PLUS right) DIVIDED BY 2  
            IF nums AT mid IS GREATER THAN nums AT mid PLUS 1  
                SET right TO mid  
            ELSE  
                SET left TO mid PLUS 1  
        RETURN left  
    END FUNCTION  
END CLASS
