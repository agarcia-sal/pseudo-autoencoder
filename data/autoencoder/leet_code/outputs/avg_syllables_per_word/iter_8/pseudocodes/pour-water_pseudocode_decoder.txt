CLASS Solution  
    FUNCTION pourWater(heights, volume, k)  
        SET n TO LENGTH OF heights  
        
        FOR each unit FROM 1 TO volume  
            SET i TO k  
            WHILE i GREATER THAN 0 AND heights AT (i MINUS 1) LESS THAN OR EQUAL TO heights AT i  
                DECREMENT i BY 1  
            WHILE i LESS THAN k AND heights AT i EQUAL TO heights AT (i PLUS 1)  
                INCREMENT i BY 1  
            IF i LESS THAN k  
                INCREMENT heights AT i BY 1  
                CONTINUE to next iteration  
            
            SET i TO k  
            WHILE i LESS THAN n MINUS 1 AND heights AT (i PLUS 1) LESS THAN OR EQUAL TO heights AT i  
                INCREMENT i BY 1  
            WHILE i GREATER THAN k AND heights AT i EQUAL TO heights AT (i MINUS 1)  
                DECREMENT i BY 1  
            IF i GREATER THAN k  
                INCREMENT heights AT i BY 1  
                CONTINUE to next iteration  
            
            INCREMENT heights AT k BY 1  
        
        RETURN heights  
    END FUNCTION  
END CLASS
