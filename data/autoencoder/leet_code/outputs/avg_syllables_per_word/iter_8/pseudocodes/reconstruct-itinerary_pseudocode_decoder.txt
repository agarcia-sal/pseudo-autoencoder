CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a map with default empty list  
        FOR each pair (frm, to) IN tickets sorted in reverse order  
            APPEND to TO graph at key frm  
        SET itinerary TO empty list  
        FUNCTION visit(airport)  
            WHILE graph at airport is not empty  
                CALL visit WITH argument graph at airport removing last element  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        CALL visit WITH argument "JFK"  
        RETURN itinerary reversed  
    END FUNCTION  
END CLASS
