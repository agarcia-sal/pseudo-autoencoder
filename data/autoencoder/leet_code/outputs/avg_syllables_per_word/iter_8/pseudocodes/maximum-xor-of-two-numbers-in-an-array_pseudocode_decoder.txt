CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO 0  
        FOR i FROM 31 DOWNTO 0  
            SET max_xor TO max_xor MULTIPLIED BY 2  
            SET current_candidate TO max_xor INCREMENTED BY 1  
            SET prefixes TO empty set  
            FOR each num IN nums  
                SET prefix TO num DIVIDED BY 2 TO THE POWER OF i  
                ADD prefix TO prefixes  
            FOR each prefix IN prefixes  
                SET needed TO current_candidate XOR prefix  
                IF needed IS IN prefixes  
                    SET max_xor TO current_candidate  
                    BREAK  
        RETURN max_xor  
    END FUNCTION  
END CLASS
