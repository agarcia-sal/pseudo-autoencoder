CLASS Solution  
    FUNCTION reorderedPowerOf2(n)  
        SET power_of_2_sorted_digits TO empty set  
        SET power_of_2 TO 1  
        WHILE power_of_2 LESS THAN OR EQUAL TO 10 RAISED TO 9  
            APPEND sorted digits of power_of_2 AS tuple TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY 2  
        SET sorted_n_digits TO sorted digits of n AS tuple  
        IF sorted_n_digits IS IN power_of_2_sorted_digits  
            RETURN true  
        ELSE  
            RETURN false  
    END FUNCTION  
END CLASS
