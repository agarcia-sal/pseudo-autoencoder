CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO LENGTH OF nums1  
        SET dp TO list of size (2 RAISED TO n) with all elements SET TO infinity  
        SET dp AT index 0 TO 0  
        
        FOR mask FROM 0 TO (2 RAISED TO n) MINUS 1  
            SET bit_count TO NUMBER OF ONES IN BINARY REPRESENTATION OF mask  
            FOR j FROM 0 TO n MINUS 1  
                IF mask CONTAINS BIT AT POSITION j  
                    SET previous_mask TO mask WITHOUT BIT j  
                    SET candidate_value TO dp AT previous_mask PLUS XOR OF nums1 AT (bit_count MINUS 1) AND nums2 AT j  
                    SET dp AT mask TO MINIMUM OF dp AT mask AND candidate_value  
        RETURN dp AT (2 RAISED TO n MINUS 1)  
    END FUNCTION  
END CLASS
