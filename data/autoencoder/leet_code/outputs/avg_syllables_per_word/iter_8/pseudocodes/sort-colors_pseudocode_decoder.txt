CLASS Solution  
    FUNCTION sortColors(nums)  
        SET low TO 0  
        SET mid TO 0  
        SET high TO LENGTH OF nums MINUS 1  
        
        WHILE mid LESS THAN OR EQUAL TO high  
            IF nums AT mid EQUAL TO 0  
                SWAP nums AT low WITH nums AT mid  
                INCREMENT low BY 1  
                INCREMENT mid BY 1  
            ELSE IF nums AT mid EQUAL TO 1  
                INCREMENT mid BY 1  
            ELSE  
                SWAP nums AT mid WITH nums AT high  
                DECREMENT high BY 1  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
