CLASS Solution  
    FUNCTION maxIncreaseKeepingSkyline(grid)  
        SET n TO LENGTH OF grid  
        SET row_maxes TO LIST OF MAXIMUM VALUE IN EACH row OF grid  
        SET col_maxes TO LIST OF MAXIMUM VALUE IN EACH column OF grid  
        
        SET total_increase TO 0  
        FOR each i FROM 0 TO n MINUS 1  
            FOR each j FROM 0 TO n MINUS 1  
                SET max_height TO MINIMUM OF row_maxes AT i AND col_maxes AT j  
                SET height_difference TO max_height MINUS grid AT i AND j  
                IF height_difference GREATER THAN 0  
                    INCREMENT total_increase BY height_difference  
        RETURN total_increase  
    END FUNCTION  
END CLASS
