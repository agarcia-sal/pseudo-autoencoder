CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO set containing '2', '5', '6', '9'  
        SET same_digits TO set containing '0', '1', '8'  

        FUNCTION is_good_number(num)  
            SET num_str TO string representation of num  
            IF all digits in num_str are in union of change_digits and same_digits THEN  
                IF any digit in num_str is in change_digits THEN  
                    RETURN True  
                ELSE  
                    RETURN False  
            ELSE  
                RETURN False  
        END FUNCTION  

        SET good_count TO 0  
        FOR each i FROM 1 TO n INCLUSIVE  
            IF is_good_number(i) THEN  
                INCREMENT good_count BY 1  
        RETURN good_count  
    END FUNCTION  
END CLASS
