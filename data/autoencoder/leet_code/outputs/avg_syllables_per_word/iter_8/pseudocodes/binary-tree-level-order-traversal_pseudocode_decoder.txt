CLASS TreeNode  
    FUNCTION __init__(self, val SET TO 0, left SET TO None, right SET TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO new TreeNode WITH val SET TO values[0]  
    SET i TO 1  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None  
            SET node.left TO new TreeNode WITH val SET TO values[i]  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None  
            SET node.right TO new TreeNode WITH val SET TO values[i]  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root IS None  
            RETURN empty list  
        SET result TO empty list  
        SET queue TO deque CONTAINING root  
        WHILE queue IS NOT EMPTY  
            SET level_size TO LENGTH OF queue  
            SET current_level TO empty list  
            FOR each _ FROM 1 TO level_size  
                SET node TO REMOVE FIRST ELEMENT FROM queue  
                APPEND node.val TO current_level  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
