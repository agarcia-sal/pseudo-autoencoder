CLASS Solution  
    FUNCTION findMaxAverage(nums, k)  
        FUNCTION canFindAverage(mid)  
            SET prefix_sums TO list of zeros with length LENGTH OF nums PLUS 1  
            SET min_prefix_sum TO list of zeros with length LENGTH OF nums PLUS 1  
            
            FOR i FROM 0 TO LENGTH OF nums MINUS 1  
                SET prefix_sums AT i PLUS 1 TO prefix_sums AT i PLUS nums AT i MINUS mid  
                
                IF i GREATER THAN OR EQUAL TO k  
                    SET min_prefix_sum AT i PLUS 1 TO MINIMUM OF min_prefix_sum AT i AND prefix_sums AT i PLUS 1 MINUS k  
                ELSE  
                    SET min_prefix_sum AT i PLUS 1 TO min_prefix_sum AT i  
                
                IF i GREATER THAN OR EQUAL TO k MINUS 1 AND prefix_sums AT i PLUS 1 MINUS min_prefix_sum AT i PLUS 1 GREATER THAN OR EQUAL TO 0  
                    RETURN True  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET low TO MINIMUM OF nums  
        SET high TO MAXIMUM OF nums  
        
        WHILE high MINUS low GREATER THAN 0.00001  
            SET mid TO low PLUS high DIVIDED BY 2  
            IF canFindAverage(mid)  
                SET low TO mid  
            ELSE  
                SET high TO mid  
        END WHILE  
        
        RETURN low  
    END FUNCTION  
END CLASS
