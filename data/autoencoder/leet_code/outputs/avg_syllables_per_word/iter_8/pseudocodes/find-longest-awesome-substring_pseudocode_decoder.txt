CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a mapping with key 0 and value -1  
        SET prefix_xor TO 0  
        SET max_length TO 0  
        
        FOR each index i and character char IN s  
            SET digit TO integer value of char  
            SET prefix_xor TO prefix_xor toggled at bit digit  
            
            IF prefix_xor exists in prefix_map THEN  
                SET max_length TO the maximum of max_length and (i minus prefix_map at prefix_xor)  
            END IF  
            
            FOR j FROM 0 TO 9  
                SET flipped TO prefix_xor toggled at bit j  
                IF flipped exists in prefix_map THEN  
                    SET max_length TO the maximum of max_length and (i minus prefix_map at flipped)  
                END IF  
            END FOR  
            
            IF prefix_xor not exists in prefix_map THEN  
                SET prefix_map at prefix_xor TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
