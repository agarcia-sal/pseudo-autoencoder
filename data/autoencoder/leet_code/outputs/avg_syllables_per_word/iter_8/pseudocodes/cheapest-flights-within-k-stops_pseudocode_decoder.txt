CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO an empty mapping from each integer in range 0 to n minus 1 TO an empty list  
        
        FOR each flight IN flights  
            SET u TO first element of flight  
            SET v TO second element of flight  
            SET w TO third element of flight  
            APPEND pair of v and w TO graph at key u  
        
        SET min_heap TO a list containing tuple of 0, minus 1, and src  
        SET min_cost TO a mapping with key src and value minus 1  
        
        WHILE min_heap is not empty  
            REMOVE the element with the smallest cost from min_heap and unpack to cost, stops, city  
            
            IF city is equal to dst  
                RETURN cost  
            
            IF stops is greater than or equal to k  
                CONTINUE to next iteration  
            
            FOR each pair neighbor and price in graph at key city  
                SET next_cost TO cost plus price  
                SET next_stops TO stops plus 1  
                
                IF neighbor not in min_cost OR next_stops is less than min_cost at neighbor OR next_cost is less than the first element of the first tuple in min_heap  
                    SET min_cost at neighbor TO next_stops  
                    ADD tuple of next_cost, next_stops, neighbor TO min_heap  
        
        RETURN minus 1  
    END FUNCTION  
END CLASS
