CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO LENGTH OF strs  
        SET n TO LENGTH OF strs AT INDEX 0  
        SET dp TO LIST OF n ELEMENTS EACH SET TO 1  
        FOR i FROM 1 TO n MINUS 1  
            FOR j FROM 0 TO i MINUS 1  
                IF FOR ALL k FROM 0 TO m MINUS 1 IT IS TRUE THAT strs AT k AND j IS LESS THAN OR EQUAL TO strs AT k AND i  
                    SET dp AT i TO THE GREATER OF dp AT i AND dp AT j PLUS 1  
        SET longest_subsequence TO MAXIMUM VALUE IN dp  
        RETURN n MINUS longest_subsequence  
    END FUNCTION  
END CLASS
