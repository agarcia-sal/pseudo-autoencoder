CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO LENGTH OF frag  
            FOR d FROM 1 TO n INCLUSIVE  
                SET left TO FIRST d CHARACTERS OF frag  
                SET right TO CHARACTERS OF frag FROM POSITION d TO END  
                IF (left DOES NOT START WITH '0' OR left IS EQUAL TO '0') AND (right DOES NOT END WITH '0')  
                    YIELD CONCATENATION OF left AND ('.' IF d IS NOT EQUAL TO n ELSE EMPTY) AND right  
                END IF  
            END FOR  
        END FUNCTION  
      
        SET s TO SUBSTRING OF s FROM SECOND CHARACTER TO SECOND LAST CHARACTER  
        RETURN LIST OF STRINGS FORMATTED AS '(' + a + ', ' + b + ')' FOR EACH i FROM 1 TO LENGTH OF s MINUS 1 INCLUSIVE, AND FOR EACH a, b IN PRODUCT OF make(s FROM START TO i) AND make(s FROM i TO END)  
    END FUNCTION  
END CLASS
