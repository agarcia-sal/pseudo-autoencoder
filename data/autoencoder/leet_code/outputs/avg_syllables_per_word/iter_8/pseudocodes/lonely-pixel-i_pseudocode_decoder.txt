CLASS Solution  
    FUNCTION findLonelyPixel(picture)  
        IF picture IS EMPTY OR picture AT INDEX 0 IS EMPTY  
            RETURN 0  
        END IF  
  
        SET m TO LENGTH OF picture  
        SET n TO LENGTH OF picture AT INDEX 0  
        SET row_count TO list of zeros WITH LENGTH m  
        SET col_count TO list of zeros WITH LENGTH n  
  
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF picture AT INDEX i AT INDEX j EQUALS 'B'  
                    INCREMENT row_count AT INDEX i BY 1  
                    INCREMENT col_count AT INDEX j BY 1  
                END IF  
            END FOR  
        END FOR  
  
        SET lonely_pixel_count TO 0  
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF picture AT INDEX i AT INDEX j EQUALS 'B' AND row_count AT INDEX i EQUALS 1 AND col_count AT INDEX j EQUALS 1  
                    INCREMENT lonely_pixel_count BY 1  
                END IF  
            END FOR  
        END FOR  
  
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
