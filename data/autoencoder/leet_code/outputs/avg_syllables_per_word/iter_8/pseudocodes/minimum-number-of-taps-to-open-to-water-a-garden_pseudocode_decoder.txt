CLASS Solution  
    FUNCTION minTaps(n, ranges)  
        SET max_range TO a list of zeros with length n plus one  
        
        FOR i FROM 0 TO n INCLUSIVE  
            IF ranges at i GREATER THAN 0 THEN  
                SET left TO the greater of 0 and i MINUS ranges at i  
                SET right TO the lesser of n and i PLUS ranges at i  
                SET max_range at left TO the greater of max_range at left and right  
        
        SET taps TO 0  
        SET current_end TO 0  
        SET farthest TO 0  
        
        FOR i FROM 0 TO n INCLUSIVE  
            IF i GREATER THAN farthest THEN  
                RETURN negative 1  
            IF i GREATER THAN current_end THEN  
                INCREMENT taps BY 1  
                SET current_end TO farthest  
            SET farthest TO the greater of farthest and max_range at i  
        
        RETURN taps  
    END FUNCTION  
END CLASS
