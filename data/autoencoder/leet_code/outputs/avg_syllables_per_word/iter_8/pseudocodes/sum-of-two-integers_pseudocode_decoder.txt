CLASS Solution  
    FUNCTION getSum(a, b)  
        SET MAX TO maximum 32-bit integer value  
        SET mask TO mask for 32-bit values  
        
        WHILE b IS NOT EQUAL TO 0  
            SET carry TO bitwise AND of a and b shifted left by 1  
            SET a TO bitwise XOR of a and b AND mask  
            SET b TO carry AND mask  
        
        IF a IS LESS THAN OR EQUAL TO MAX  
            RETURN a  
        ELSE  
            RETURN bitwise complement of bitwise XOR of a and mask  
    END FUNCTION  
END CLASS
