CLASS Solution  
    FUNCTION stoneGameVIII(stones)  
        INITIALIZE prefix_sum AS list of zeros with length of stones  
        SET first element of prefix_sum TO first element of stones  
        FOR i FROM 1 TO length of stones MINUS 1  
            SET prefix_sum at i TO prefix_sum at i MINUS 1 PLUS stones at i  
        SET dp TO last element of prefix_sum  
        FOR i FROM length of stones MINUS 2 DOWN TO 1  
            SET dp TO maximum of dp AND prefix_sum at i MINUS dp  
        RETURN dp  
    END FUNCTION  
END CLASS
