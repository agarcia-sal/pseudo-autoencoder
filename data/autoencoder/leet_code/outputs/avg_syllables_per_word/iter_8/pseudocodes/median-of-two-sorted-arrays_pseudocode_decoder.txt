CLASS Solution  
    FUNCTION findMedianSortedArrays(nums1, nums2)  
        IF LENGTH OF nums1 IS GREATER THAN LENGTH OF nums2 THEN  
            SET temp TO nums1  
            SET nums1 TO nums2  
            SET nums2 TO temp  
        END IF  
        
        SET x TO LENGTH OF nums1  
        SET y TO LENGTH OF nums2  
        SET low TO 0  
        SET high TO x  
        
        WHILE low IS LESS THAN OR EQUAL TO high DO  
            SET partitionX TO INTEGER DIVISION OF (low PLUS high) BY 2  
            SET partitionY TO INTEGER DIVISION OF (x PLUS y PLUS 1) BY 2 MINUS partitionX  
            
            IF partitionX IS EQUAL TO 0 THEN  
                SET maxLeftX TO NEGATIVE_INFINITY  
            ELSE  
                SET maxLeftX TO ELEMENT AT (partitionX MINUS 1) OF nums1  
            END IF  
            
            IF partitionX IS EQUAL TO x THEN  
                SET minRightX TO POSITIVE_INFINITY  
            ELSE  
                SET minRightX TO ELEMENT AT partitionX OF nums1  
            END IF  
            
            IF partitionY IS EQUAL TO 0 THEN  
                SET maxLeftY TO NEGATIVE_INFINITY  
            ELSE  
                SET maxLeftY TO ELEMENT AT (partitionY MINUS 1) OF nums2  
            END IF  
            
            IF partitionY IS EQUAL TO y THEN  
                SET minRightY TO POSITIVE_INFINITY  
            ELSE  
                SET minRightY TO ELEMENT AT partitionY OF nums2  
            END IF  
            
            IF maxLeftX IS LESS THAN OR EQUAL TO minRightY AND maxLeftY IS LESS THAN OR EQUAL TO minRightX THEN  
                IF (x PLUS y) MODULO 2 IS EQUAL TO 0 THEN  
                    RETURN AVERAGE OF MAXIMUM OF maxLeftX AND maxLeftY AND MINIMUM OF minRightX AND minRightY  
                ELSE  
                    RETURN MAXIMUM OF maxLeftX AND maxLeftY  
                END IF  
            ELSE IF maxLeftX IS GREATER THAN minRightY THEN  
                SET high TO partitionX MINUS 1  
            ELSE  
                SET low TO partitionX PLUS 1  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
