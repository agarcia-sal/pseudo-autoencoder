CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits IS empty  
            RETURN empty list  
        END IF  
        
        SET phone_map TO mapping of digits to corresponding letters  
        
        FUNCTION backtrack(combination, next_digits)  
            IF LENGTH OF next_digits IS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN phone_map AT next_digits FIRST element  
                    CALL backtrack WITH combination concatenated with letter AND next_digits excluding first element  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO empty list  
        CALL backtrack WITH empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
