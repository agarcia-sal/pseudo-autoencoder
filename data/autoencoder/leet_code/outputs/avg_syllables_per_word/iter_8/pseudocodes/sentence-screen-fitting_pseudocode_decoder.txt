CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO concatenation of all words in sentence separated by single spaces followed by a single space  
        SET total_length TO length of s  
        SET position TO 0  
        
        FOR index FROM 1 TO rows  
            SET position TO position plus cols  
            
            IF character in s at position modulo total_length IS a space  
                SET position TO position plus 1  
            ELSE  
                WHILE position GREATER THAN 0 AND character in s at (position minus 1) modulo total_length IS NOT a space  
                    SET position TO position minus 1  
        
        RETURN position DIVIDED BY total_length  
    END FUNCTION  
END CLASS
