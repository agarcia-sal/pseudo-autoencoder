CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF LENGTH OF nums EQUALS 1 THEN  
                RETURN absolute value of nums AT index 0 MINUS 24 IS LESS THAN 0.000001  
            END IF  
            
            FOR i FROM 0 TO LENGTH OF nums MINUS 1  
                FOR j FROM 0 TO LENGTH OF nums MINUS 1  
                    IF i IS NOT EQUAL TO j THEN  
                        SET new_nums TO list of nums AT k FOR k FROM 0 TO LENGTH OF nums MINUS 1 WHERE k IS NOT i AND k IS NOT j  
                        
                        FOR op IN list of operations truediv, mul, add, sub  
                            IF (op IS add OR op IS mul) AND j IS GREATER THAN i THEN  
                                CONTINUE TO NEXT op  
                            END IF  
                            
                            IF op IS NOT truediv OR nums AT j IS NOT EQUAL TO 0 THEN  
                                APPEND result of op applied to nums AT i AND nums AT j TO new_nums  
                                IF solve(new_nums) IS TRUE THEN  
                                    RETURN True  
                                END IF  
                                REMOVE last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        IMPORT truediv, mul, add, sub FROM operator  
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
