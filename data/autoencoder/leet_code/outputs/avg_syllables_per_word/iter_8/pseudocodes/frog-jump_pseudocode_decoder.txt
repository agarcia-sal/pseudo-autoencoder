CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS empty OR LENGTH OF stones IS LESS THAN 2  
            RETURN False  
        
        SET stone_set TO a set containing all elements in stones  
        SET memo TO an empty map  
        
        FUNCTION can_jump_to(position, jump)  
            IF position IS EQUAL TO the last element in stones  
                RETURN True  
            
            IF memo CONTAINS the key (position, jump)  
                RETURN the value associated with (position, jump) in memo  
            
            FOR each next_jump FROM (jump MINUS 1) TO (jump PLUS 1)  
                IF next_jump IS GREATER THAN 0 AND position PLUS next_jump IS IN stone_set  
                    IF can_jump_to(position PLUS next_jump, next_jump) RETURNS True  
                        SET memo WITH key (position, jump) TO True  
                        RETURN True  
            
            SET memo WITH key (position, jump) TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(the first element in stones, 0)  
    END FUNCTION  
END CLASS
