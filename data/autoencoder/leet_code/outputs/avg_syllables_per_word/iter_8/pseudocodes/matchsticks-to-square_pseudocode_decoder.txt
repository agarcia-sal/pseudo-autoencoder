CLASS Solution
    FUNCTION makesquare(matchsticks)
        SET total_length TO sum of matchsticks
        IF total_length modulo 4 is not equal to 0
            RETURN False
        SET side_length TO total_length divided by 4
        SORT matchsticks in descending order

        FUNCTION can_form_sides(sides, index)
            IF index is equal to length of matchsticks
                RETURN True if all sides are equal to side_length ELSE False

            FOR i FROM 0 TO 3
                IF sides at i plus matchsticks at index is less than or equal to side_length
                    INCREMENT sides at i BY matchsticks at index
                    IF can_form_sides(sides, index plus 1) is True
                        RETURN True
                    DECREMENT sides at i BY matchsticks at index
                IF sides at i is equal to 0
                    BREAK
            RETURN False
        END FUNCTION

        RETURN can_form_sides(list of four zeros, 0)
    END FUNCTION
END CLASS
