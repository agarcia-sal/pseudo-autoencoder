CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO empty list  
        
        FOR each asteroid IN asteroids  
            WHILE stack is not empty AND asteroid is less than 0 AND last element in stack is greater than 0  
                IF last element in stack is less than negative asteroid  
                    REMOVE last element from stack  
                    CONTINUE  
                ELSE IF last element in stack equals negative asteroid  
                    REMOVE last element from stack  
                END IF  
                BREAK  
            END WHILE  
            ELSE  
                APPEND asteroid TO stack  
            END ELSE  
        END FOR  
        
        RETURN stack  
    END FUNCTION  
END CLASS
