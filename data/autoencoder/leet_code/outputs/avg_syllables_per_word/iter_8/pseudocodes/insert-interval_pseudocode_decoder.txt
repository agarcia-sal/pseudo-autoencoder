CLASS Solution  
    FUNCTION insert(intervals, newInterval)  
        IF intervals IS EMPTY  
            RETURN list containing newInterval  
        END IF  
  
        SET merged TO empty list  
        SET i TO 0  
        SET n TO length of intervals  
  
        WHILE i LESS THAN n AND intervals at position i's end LESS THAN newInterval's start  
            APPEND intervals at position i TO merged  
            INCREMENT i BY 1  
        END WHILE  
  
        WHILE i LESS THAN n AND intervals at position i's start LESS THAN OR EQUAL TO newInterval's end  
            SET newInterval's start TO minimum of newInterval's start AND intervals at position i's start  
            SET newInterval's end TO maximum of newInterval's end AND intervals at position i's end  
            INCREMENT i BY 1  
        END WHILE  
  
        APPEND newInterval TO merged  
  
        WHILE i LESS THAN n  
            APPEND intervals at position i TO merged  
            INCREMENT i BY 1  
        END WHILE  
  
        RETURN merged  
    END FUNCTION  
END CLASS
