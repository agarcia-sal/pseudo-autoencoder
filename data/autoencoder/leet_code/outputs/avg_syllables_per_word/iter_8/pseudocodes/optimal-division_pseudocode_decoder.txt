CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF LENGTH OF nums IS EQUAL TO 1 THEN  
            RETURN string representation OF the first element IN nums  
        ELSE IF LENGTH OF nums IS EQUAL TO 2 THEN  
            RETURN concatenation OF string representation OF first element IN nums, " divided by ", and string representation OF second element IN nums  
        ELSE  
            RETURN concatenation OF string representation OF first element IN nums, " divided by ", "(", and concatenation OF string representations OF elements FROM second TO last IN nums separated BY " divided by ", and ")"  
        END IF  
    END FUNCTION  
END CLASS
