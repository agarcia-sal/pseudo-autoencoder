CLASS Solution  
    FUNCTION myPow(x, n)  
        FUNCTION qpow(a, n)  
            SET ans TO 1  
            WHILE n NOT EQUAL TO 0  
                IF n AND 1 IS TRUTHY  
                    SET ans TO ans MULTIPLIED BY a  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY 2 USING FLOOR DIVISION  
            RETURN ans  
        END FUNCTION  

        IF n GREATER THAN OR EQUAL TO 0  
            RETURN qpow(x, n)  
        ELSE  
            RETURN 1 DIVIDED BY qpow(x, NEGATION OF n)  
    END FUNCTION  
END CLASS
