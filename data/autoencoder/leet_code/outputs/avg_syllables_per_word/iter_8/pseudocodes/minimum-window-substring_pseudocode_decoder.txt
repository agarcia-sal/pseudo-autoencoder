CLASS Solution  
    FUNCTION minWindow(s, t)  
        SET t_count TO frequency count of characters in t  
        SET required TO number of unique characters in t_count  
        
        SET left TO 0  
        SET right TO 0  
        SET formed TO 0  
        SET window_counts TO an empty frequency count  
        
        SET min_length TO infinity  
        SET min_start TO 0  
        
        WHILE right LESS THAN length of s  
            ADD character s[right] TO window_counts  
            
            IF s[right] is in t_count AND window_counts[s[right]] EQUALS t_count[s[right]] THEN  
                INCREMENT formed BY 1  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO s[left]  
                
                IF (right MINUS left PLUS 1) LESS THAN min_length THEN  
                    SET min_length TO (right MINUS left PLUS 1)  
                    SET min_start TO left  
                END IF  
                
                DECREMENT window_counts[char] BY 1  
                IF char is in t_count AND window_counts[char] LESS THAN t_count[char] THEN  
                    DECREMENT formed BY 1  
                END IF  
                
                INCREMENT left BY 1  
            END WHILE  
            
            INCREMENT right BY 1  
        END WHILE  
        
        IF min_length EQUALS infinity THEN  
            RETURN empty string  
        ELSE  
            RETURN substring of s FROM min_start TO min_start PLUS min_length  
        END IF  
    END FUNCTION  
END CLASS
