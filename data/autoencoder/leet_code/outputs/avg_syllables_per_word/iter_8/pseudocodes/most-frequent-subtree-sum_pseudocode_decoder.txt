CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO TreeNode WITH val values AT INDEX 0  
    SET i TO 1  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i IS LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.left TO TreeNode WITH val values AT INDEX i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.right TO TreeNode WITH val values AT INDEX i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS None  
                RETURN 0  
            SET current_sum TO node.val PLUS subtree_sum(node.left) PLUS subtree_sum(node.right)  
            INCREMENT count AT KEY current_sum BY 1  
            RETURN current_sum  
        END FUNCTION  

        SET count TO empty default dictionary with integer default  
        CALL subtree_sum WITH root  
        SET max_freq TO MAXIMUM value IN count OR 0 IF count IS EMPTY  
        SET result TO EMPTY LIST  
        FOR each key, freq IN count  
            IF freq IS EQUAL TO max_freq  
                APPEND key TO result  
        RETURN result  
    END FUNCTION  
END CLASS
