CLASS Solution  
    FUNCTION hasPath(maze, start, destination)  
        SET directions TO list of (0, 1), (1, 0), (0, -1), (-1, 0)  
        SET m TO LENGTH OF maze  
        SET n TO LENGTH OF maze AT index 0  
        SET visited TO empty set  
        
        FUNCTION dfs(x, y)  
            IF (x, y) IS IN visited THEN  
                RETURN False  
            IF list of x, y IS EQUAL TO destination THEN  
                RETURN True  
            ADD (x, y) TO visited  
            
            FOR each dx, dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx plus dx IS GREATER THAN OR EQUAL TO 0 AND nx plus dx IS LESS THAN m AND ny plus dy IS GREATER THAN OR EQUAL TO 0 AND ny plus dy IS LESS THAN n AND maze AT nx plus dx AT ny plus dy IS EQUAL TO 0  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                IF dfs(nx, ny) RETURNS True THEN  
                    RETURN True  
            RETURN False  
        END FUNCTION  
        
        RETURN dfs(start AT index 0, start AT index 1)  
    END FUNCTION  
END CLASS
