CLASS Solution  
    FUNCTION topKFrequent(nums, k)  
        SET frequency_map TO a map with default integer zero  
        FOR each num IN nums  
            INCREMENT frequency_map at key num BY 1  
        SET max_frequency TO the maximum value in frequency_map  
        SET buckets TO a list of empty lists with length max_frequency PLUS 1  
        FOR each num, freq IN frequency_map items  
            APPEND num TO buckets at index freq  
        SET result TO an empty list  
        FOR freq FROM max_frequency DOWNTO 1  
            IF the length of buckets at index freq IS GREATER THAN 0  
                APPEND all elements of buckets at index freq TO result  
                IF the length of result IS GREATER THAN OR EQUAL TO k  
                    RETURN the first k elements of result  
        RETURN result  
END CLASS
