CLASS Solution  
    FUNCTION numTrees(n)  
        SET dp TO a list of zeros with length n plus 1  
        SET dp at index 0 TO 1  
        SET dp at index 1 TO 1  
        FOR i FROM 2 TO n inclusive  
            FOR j FROM 1 TO i inclusive  
                SET dp at index i TO dp at index i PLUS dp at index (j MINUS 1) MULTIPLIED BY dp at index (i MINUS j)  
        RETURN dp at index n  
    END FUNCTION  
END CLASS
