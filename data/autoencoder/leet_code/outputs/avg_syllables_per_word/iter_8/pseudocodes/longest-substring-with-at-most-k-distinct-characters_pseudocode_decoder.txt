CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s, k)  
        IF k EQUALS 0  
            RETURN 0  
        END IF  
        
        SET char_map TO empty mapping  
        SET left TO 0  
        SET max_length TO 0  
        
        FOR right FROM 0 TO length of s MINUS 1  
            SET char_map AT s AT right TO right  
            
            IF size of char_map GREATER THAN k  
                SET leftmost_char TO key in char_map WITH minimum value  
                SET left TO char_map AT leftmost_char PLUS 1  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET max_length TO the greater value BETWEEN max_length AND right MINUS left PLUS 1  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
