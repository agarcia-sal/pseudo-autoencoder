CLASS TreeNode  
    FUNCTION __init__(val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH val VALUES AT index 0  
    SET i TO 1  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.left TO new TreeNode WITH val values AT i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.right TO new TreeNode WITH val values AT i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION convertBST(root)  
        SET self.sum TO 0  
        
        FUNCTION traverse(node)  
            IF node IS None  
                RETURN  
            END IF  
            CALL traverse WITH node.right  
            INCREMENT self.sum BY node.val  
            SET node.val TO self.sum  
            CALL traverse WITH node.left  
        END FUNCTION  
        
        CALL traverse WITH root  
        RETURN root  
    END FUNCTION  
END CLASS
