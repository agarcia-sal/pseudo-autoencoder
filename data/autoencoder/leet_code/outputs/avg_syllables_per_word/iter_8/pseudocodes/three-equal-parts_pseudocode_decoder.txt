CLASS Solution  
    FUNCTION threeEqualParts(arr)  
        SET total_ones TO sum of all elements in arr  
        
        IF total_ones EQUALS 0  
            RETURN list containing 0 and 2  
        
        IF total_ones MODULO 3 IS NOT 0  
            RETURN list containing -1 and -1  
        
        SET part_ones TO total_ones DIVIDED BY 3  
        
        SET first_part_index TO 0  
        SET count_ones TO 0  
        FOR i FROM 0 TO length of arr MINUS 1  
            IF arr at position i EQUALS 1  
                INCREMENT count_ones BY 1  
                IF count_ones EQUALS 1  
                    SET first_part_index TO i  
                IF count_ones EQUALS part_ones  
                    EXIT LOOP  
        
        SET second_part_index TO 0  
        SET count_ones TO 0  
        FOR i FROM 0 TO length of arr MINUS 1  
            IF arr at position i EQUALS 1  
                INCREMENT count_ones BY 1  
                IF count_ones EQUALS part_ones PLUS 1  
                    SET second_part_index TO i  
                    EXIT LOOP  
        
        SET third_part_index TO 0  
        SET count_ones TO 0  
        FOR i FROM 0 TO length of arr MINUS 1  
            IF arr at position i EQUALS 1  
                INCREMENT count_ones BY 1  
                IF count_ones EQUALS (2 TIMES part_ones) PLUS 1  
                    SET third_part_index TO i  
                    EXIT LOOP  
        
        SET n TO length of arr MINUS third_part_index  
        
        IF first_part_index PLUS n IS LESS THAN OR EQUAL TO length of arr AND  
           second_part_index PLUS n IS LESS THAN OR EQUAL TO length of arr AND  
           slice of arr from first_part_index TO first_part_index PLUS n EQUALS slice of arr from second_part_index TO second_part_index PLUS n AND  
           slice of arr from second_part_index TO second_part_index PLUS n EQUALS slice of arr from third_part_index TO end  
            RETURN list containing first_part_index PLUS n MINUS 1 AND second_part_index PLUS n  
        
        RETURN list containing -1 AND -1  
    END FUNCTION  
END CLASS
