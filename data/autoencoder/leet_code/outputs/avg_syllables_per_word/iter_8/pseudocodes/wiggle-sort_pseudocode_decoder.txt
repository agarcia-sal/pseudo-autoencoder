CLASS Solution  
    FUNCTION wiggleSort(nums)  
        SET less TO true  
        FOR i FROM 0 TO LENGTH OF nums MINUS 1  
            IF less  
                IF element AT nums[i] GREATER THAN element AT nums[i plus 1]  
                    SWAP elements AT nums[i] AND nums[i plus 1]  
            ELSE  
                IF element AT nums[i] LESS THAN element AT nums[i plus 1]  
                    SWAP elements AT nums[i] AND nums[i plus 1]  
            SET less TO NOT less  
        END FOR  
    END FUNCTION  
END CLASS
