CLASS Solution  
    FUNCTION checkPossibility(nums)  
        SET modified TO false  
        
        FOR i FROM 1 TO LENGTH OF nums MINUS 1  
            IF nums AT i LESS THAN nums AT i MINUS 1 THEN  
                IF modified IS true THEN  
                    RETURN false  
                SET modified TO true  
                
                IF i LESS THAN 2 OR nums AT i GREATER THAN OR EQUAL TO nums AT i MINUS 2 THEN  
                    SET nums AT i MINUS 1 TO nums AT i  
                ELSE  
                    SET nums AT i TO nums AT i MINUS 1  
        RETURN true  
    END FUNCTION  
END CLASS
