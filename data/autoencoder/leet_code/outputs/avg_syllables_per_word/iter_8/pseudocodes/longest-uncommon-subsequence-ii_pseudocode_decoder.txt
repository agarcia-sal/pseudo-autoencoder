CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET it TO iterator over t  
            RETURN for all characters c in s, c appears in it in order  
        END FUNCTION  
        
        SORT strs by descending length and lexicographically ascending order  
        
        FOR each index i and word IN strs  
            SET unique TO true  
            FOR each index j and other IN strs  
                IF i is not equal to j AND is_subsequence(word, other) IS true  
                    SET unique TO false  
                    BREAK inner loop  
            IF unique IS true  
                RETURN length of word  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
