CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO LENGTH OF word1  
        SET n TO LENGTH OF word2  

        INITIALIZE dp AS 2D ARRAY OF SIZE (m PLUS 1) BY (n PLUS 1) FILLED WITH 0  

        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF CHARACTER AT POSITION i MINUS 1 IN word1 EQUALS CHARACTER AT POSITION j MINUS 1 IN word2  
                    SET dp AT i AND j TO dp AT i MINUS 1 AND j MINUS 1 PLUS 1  
                ELSE  
                    SET dp AT i AND j TO GREATER OF dp AT i MINUS 1 AND j AND dp AT i AND j MINUS 1  
        
        SET lcs_length TO dp AT m AND n  

        RETURN (m MINUS lcs_length) PLUS (n MINUS lcs_length)  
    END FUNCTION  
END CLASS
