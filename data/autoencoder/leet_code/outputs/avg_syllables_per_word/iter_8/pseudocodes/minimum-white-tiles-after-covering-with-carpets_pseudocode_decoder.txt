CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO LENGTH OF floor  

        INITIALIZE dp AS TWO-DIMENSIONAL LIST OF SIZE (n PLUS 1) BY (numCarpets PLUS 1) WITH ALL ZEROS  

        FOR i FROM 1 TO n INCLUSIVE  
            SET dp[i][0] TO dp[i MINUS 1][0] PLUS 1 IF floor AT POSITION (i MINUS 1) IS '1' ELSE 0  
        END FOR  

        FOR j FROM 1 TO numCarpets INCLUSIVE  
            FOR i FROM 1 TO n INCLUSIVE  
                SET dp[i][j] TO dp[i MINUS 1][j] PLUS 1 IF floor AT POSITION (i MINUS 1) IS '1' ELSE 0  
                SET dp[i][j] TO THE MINIMUM OF dp[i][j] AND dp[MAXIMUM OF 0 AND (i MINUS carpetLen)][j MINUS 1]  
            END FOR  
        END FOR  

        RETURN dp[n][numCarpets]  
    END FUNCTION  
END CLASS
