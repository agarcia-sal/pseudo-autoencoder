CLASS Solution  
    FUNCTION minOperations(nums)  
        SET n TO LENGTH OF nums  
        SET unique_nums TO SORTED UNIQUE ELEMENTS OF nums  
        
        SET min_operations TO n  
        SET j TO 0  
        FOR i FROM 0 TO LENGTH OF unique_nums MINUS 1  
            WHILE j LESS THAN LENGTH OF unique_nums AND unique_nums[j] LESS THAN unique_nums[i] PLUS n  
                INCREMENT j BY 1  
            SET min_operations TO MINIMUM OF min_operations AND n MINUS (j MINUS i)  
        RETURN min_operations  
    END FUNCTION  
END CLASS
