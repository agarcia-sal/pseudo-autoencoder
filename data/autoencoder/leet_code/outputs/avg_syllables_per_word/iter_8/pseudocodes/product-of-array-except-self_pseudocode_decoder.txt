CLASS Solution  
    FUNCTION productExceptSelf(nums)  
        SET n TO LENGTH OF nums  
        SET answer TO list of 1 repeated n times  
        
        SET left_product TO 1  
        FOR i FROM 0 TO n MINUS 1  
            SET answer at i TO left_product  
            MULTIPLY left_product BY nums at i  
        
        SET right_product TO 1  
        FOR i FROM n MINUS 1 DOWNTO 0  
            MULTIPLY answer at i BY right_product  
            MULTIPLY right_product BY nums at i  
        
        RETURN answer  
    END FUNCTION  
END CLASS
