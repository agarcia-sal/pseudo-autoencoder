CLASS Solution  
    FUNCTION isTransformable(s, t)  
        INITIALIZE pos AS mapping from digit TO min-heap of positions  
        FOR each index i AND character c IN s  
            ADD i TO heap pos AT key CONVERT c TO digit  
        FOR each index i AND character c IN t  
            SET digit TO CONVERT c TO digit  
            IF heap pos AT digit IS empty THEN  
                RETURN False  
            FOR each smaller_digit FROM 0 TO digit MINUS 1  
                IF heap pos AT smaller_digit IS NOT empty AND smallest element IN heap pos AT smaller_digit IS LESS THAN smallest element IN heap pos AT digit THEN  
                    RETURN False  
            REMOVE smallest element FROM heap pos AT digit  
        RETURN True  
    END FUNCTION  
END CLASS
