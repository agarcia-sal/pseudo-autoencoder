CLASS Solution  
    FUNCTION boxDelivering(boxes, portsCount, maxBoxes, maxWeight)  
        SET n TO LENGTH OF boxes  
        SET dp TO list of size n plus 1 FILLED WITH infinity  
        SET dp at index 0 TO 0  
        
        SET left TO 0  
        SET current_weight TO 0  
        SET extra_trips TO 0  
        
        FOR right FROM 0 TO n MINUS 1  
            INCREMENT current_weight BY boxes at index right position 1  
            IF right GREATER THAN 0 AND boxes at index right position 0 NOT EQUAL TO boxes at index right MINUS 1 position 0  
                INCREMENT extra_trips BY 1  
            
            WHILE left LESS THAN OR EQUAL TO right AND (right MINUS left PLUS 1 GREATER THAN maxBoxes OR current_weight GREATER THAN maxWeight)  
                DECREMENT current_weight BY boxes at index left position 1  
                IF left GREATER THAN 0 AND boxes at index left position 0 NOT EQUAL TO boxes at index left MINUS 1 position 0  
                    DECREMENT extra_trips BY 1  
                INCREMENT left BY 1  
            
            SET dp at index right PLUS 1 TO dp at index left PLUS 2 PLUS extra_trips  
        
        RETURN dp at index n  
    END FUNCTION  
END CLASS
