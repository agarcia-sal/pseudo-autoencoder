CLASS Solution  
    FUNCTION primePalindrome(n)  
        FUNCTION is_prime(num)  
            IF num LESS THAN 2  
                RETURN False  
            IF num EQUALS 2 OR num EQUALS 3  
                RETURN True  
            IF num MODULO 2 EQUALS 0 OR num MODULO 3 EQUALS 0  
                RETURN False  
            SET i TO 5  
            WHILE i TIMES i LESS THAN OR EQUAL TO num  
                IF num MODULO i EQUALS 0 OR num MODULO (i PLUS 2) EQUALS 0  
                    RETURN False  
                INCREMENT i BY 6  
            RETURN True  
        END FUNCTION  

        FUNCTION is_palindrome(num)  
            RETURN string representation of num EQUALS reverse of string representation of num  
        END FUNCTION  

        IF n LESS THAN OR EQUAL TO 2  
            RETURN 2  

        IF n GREATER THAN OR EQUAL TO 10 AND n LESS THAN OR EQUAL TO 11  
            RETURN 11  

        WHILE True  
            IF n MODULO 2 EQUALS 0  
                INCREMENT n BY 1  
                CONTINUE  
            IF is_palindrome(n) AND is_prime(n)  
                RETURN n  
            IF LENGTH OF string representation of n MODULO 2 EQUALS 0  
                SET n TO 10 RAISED TO LENGTH OF string representation of n PLUS 1  
            ELSE  
                INCREMENT n BY 2  
    END FUNCTION  
END CLASS
