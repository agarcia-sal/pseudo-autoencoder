CLASS Solution  
    FUNCTION subarraySum(nums, k)  
        SET cumulative_sum_freq TO dictionary with key 0 mapped to value 1  
        SET current_sum TO 0  
        SET count TO 0  
        
        FOR each num IN nums  
            INCREMENT current_sum BY num  
            IF current_sum MINUS k EXISTS AS KEY IN cumulative_sum_freq  
                INCREMENT count BY cumulative_sum_freq VALUE AT KEY current_sum MINUS k  
            END IF  
            IF current_sum EXISTS AS KEY IN cumulative_sum_freq  
                INCREMENT cumulative_sum_freq VALUE AT KEY current_sum BY 1  
            ELSE  
                SET cumulative_sum_freq VALUE AT KEY current_sum TO 1  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
