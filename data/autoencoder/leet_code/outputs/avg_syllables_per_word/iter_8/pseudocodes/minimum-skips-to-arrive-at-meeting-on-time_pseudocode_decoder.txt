CLASS Solution  
    FUNCTION minSkips(dist, speed, hoursBefore)  
        SET n TO LENGTH OF dist  
        SET dp TO two-dimensional list of size n by (n plus 1) filled with infinity  
        SET total_dist TO SUM OF dist  
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore THEN  
            RETURN -1  
        
        SET dp at row 0, column 0 TO dist at index 0  
        FOR j FROM 1 TO n minus 1 DO  
            SET dp at row 0, column j TO dist at index 0  
        
        FOR i FROM 1 TO n minus 1 DO  
            FOR j FROM 0 TO i DO  
                IF j LESS THAN i THEN  
                    SET dp at row i, column j TO CEILING OF (dp at row i minus 1, column j DIVIDED BY speed) MULTIPLIED BY speed PLUS dist at index i  
                IF j GREATER THAN 0 THEN  
                    SET dp at row i, column j TO MINIMUM OF dp at row i, column j AND dp at row i minus 1, column j minus 1 PLUS dist at index i  
        
        FOR j FROM 0 TO n minus 1 DO  
            IF dp at row n minus 1, column j LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
        
        RETURN -1  
    END FUNCTION  
END CLASS
