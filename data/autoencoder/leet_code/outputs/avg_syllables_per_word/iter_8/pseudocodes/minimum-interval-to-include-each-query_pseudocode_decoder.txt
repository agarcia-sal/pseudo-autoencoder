CLASS Solution  
    FUNCTION minInterval(intervals, queries)  
        SORT intervals BY their start time  
        SET sorted_queries TO list of pairs (query, original_index) sorted by query  
          
        SET min_heap TO empty heap  
        SET result TO list of length equal to queries filled with -1  
          
        SET interval_index TO 0  
        FOR each query, original_index IN sorted_queries  
            WHILE interval_index LESS THAN length of intervals AND intervals at interval_index start LESS THAN OR EQUAL TO query  
                SET start, end TO interval at interval_index  
                SET interval_size TO end MINUS start PLUS 1  
                ADD (interval_size, end) TO min_heap  
                INCREMENT interval_index BY 1  
            END WHILE  
              
            WHILE min_heap IS NOT empty AND smallest end in min_heap LESS THAN query  
                REMOVE smallest element from min_heap  
            END WHILE  
              
            IF min_heap IS NOT empty  
                SET result at original_index TO smallest interval_size in min_heap  
            END IF  
        END FOR  
          
        RETURN result  
    END FUNCTION  
END CLASS
