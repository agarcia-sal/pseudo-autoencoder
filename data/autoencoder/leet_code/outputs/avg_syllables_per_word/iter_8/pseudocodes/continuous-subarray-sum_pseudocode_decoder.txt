CLASS Solution  
    FUNCTION checkSubarraySum(nums, k)  
        SET remainder_map TO a dictionary with key 0 and value negative one  
        SET current_sum TO zero  
        
        FOR each index i and element num IN nums  
            INCREMENT current_sum BY num  
            
            IF k is not equal to zero  
                SET current_sum TO current_sum modulo k  
            
            IF current_sum exists in remainder_map  
                IF i minus value at remainder_map with key current_sum is greater than one  
                    RETURN True  
            ELSE  
                SET remainder_map at key current_sum TO i  
        
        RETURN False  
    END FUNCTION  
END CLASS
