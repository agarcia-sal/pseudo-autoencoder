CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO set converted from bank  
        SET q TO deque containing tuple of start and 0  
        SET mp TO dictionary mapping 'A' TO "TCG", 'T' TO "ACG", 'C' TO "ATG", 'G' TO "ATC"  
        WHILE q is not empty  
            SET t, step TO first element removed from q  
            IF t EQUALS end  
                RETURN step  
            FOR i, v IN enumeration of characters in t  
                FOR j IN characters mapped from v in mp  
                    SET next TO concatenation of substring of t before i, j, and substring of t after i  
                    IF next IS IN s  
                        APPEND tuple of next and step plus 1 TO q  
                        REMOVE next FROM s  
        RETURN negative 1  
    END FUNCTION  
END CLASS
