CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO sorted list of pairs (interval start, original index) from intervals  
        SET result TO empty list  
        FOR each interval IN intervals  
            SET end TO interval end value  
            SET idx TO binary search position of first start time GREATER THAN OR EQUAL TO end IN start_times  
            IF idx IS LESS THAN length of start_times THEN  
                APPEND index at start_times[idx] TO result  
            ELSE  
                APPEND -1 TO result  
        RETURN result  
    END FUNCTION  
END CLASS
