CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS 0 THEN  
            RETURN list containing 0  
        END IF  
        
        SET prev_gray TO CALL grayCode WITH n MINUS 1  
        
        SET mask TO 1 SHIFTED LEFT BY n MINUS 1  
        SET reflected_list TO REVERSED prev_gray  
        SET appended_list TO EMPTY LIST  
        FOR each num IN reflected_list  
            APPEND mask OR num TO appended_list  
        END FOR  
        
        SET current_gray TO prev_gray CONCATENATED WITH appended_list  
        
        RETURN current_gray  
    END FUNCTION  
END CLASS
