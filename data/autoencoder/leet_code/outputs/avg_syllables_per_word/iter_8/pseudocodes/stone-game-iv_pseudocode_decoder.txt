CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO a list of False repeated (n plus 1) times  
        FOR i FROM 1 TO n  
            SET j TO 1  
            WHILE j times j LESS THAN OR EQUAL TO i  
                IF dp at (i minus j times j) IS False  
                    SET dp at i TO True  
                    BREAK loop  
                END IF  
                INCREMENT j BY 1  
            END WHILE  
        END FOR  
        RETURN dp at n  
    END FUNCTION  
END CLASS
