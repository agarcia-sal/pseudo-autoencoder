CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF LENGTH OF s GREATER THAN LENGTH OF t THEN  
            RETURN isOneEditDistance(t, s)  
        END IF  
        
        SET diff TO LENGTH OF t MINUS LENGTH OF s  
        
        IF diff GREATER THAN 1 THEN  
            RETURN False  
        END IF  
        
        IF diff EQUALS 0 THEN  
            SET count TO 0  
            FOR each pair of characters c1, c2 IN s AND t  
                IF c1 NOT EQUAL TO c2 THEN  
                    INCREMENT count BY 1  
                END IF  
            END FOR  
            RETURN count EQUALS 1  
        ELSE  
            FOR each index i FROM 0 TO LENGTH OF s MINUS 1  
                IF CHARACTER AT s AT i NOT EQUAL TO CHARACTER AT t AT i THEN  
                    RETURN SUBSTRING OF s FROM i TO END EQUALS SUBSTRING OF t FROM i PLUS 1 TO END  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
