CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO UNIQUE ELEMENTS OF wordDict  
        INITIALIZE dp AS LIST OF LENGTH LENGTH OF s PLUS ONE WITH ALL VALUES False  
        SET dp AT INDEX 0 TO True  
        
        FOR i FROM 1 TO LENGTH OF s  
            FOR j FROM 0 TO i MINUS 1  
                IF dp AT INDEX j IS True AND SUBSTRING OF s FROM j TO i IS IN wordSet  
                    SET dp AT INDEX i TO True  
                    EXIT INNER LOOP  
        
        RETURN dp AT INDEX LENGTH OF s  
    END FUNCTION  
END CLASS
