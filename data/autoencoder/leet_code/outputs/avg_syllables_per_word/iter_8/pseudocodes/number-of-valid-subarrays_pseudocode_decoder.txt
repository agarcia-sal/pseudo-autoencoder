CLASS Solution  
    FUNCTION validSubarrays(nums)  
        SET stack TO empty list  
        SET count TO 0  
        
        FOR each num IN nums  
            WHILE stack is not empty AND last element in stack GREATER THAN num  
                REMOVE last element FROM stack  
            APPEND num TO stack  
            INCREMENT count BY LENGTH OF stack  
        RETURN count  
    END FUNCTION  
END CLASS
