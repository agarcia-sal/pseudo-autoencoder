CLASS Solution  
    FUNCTION minMovesToMakePalindrome(s)  
        SET s TO list of characters from s  
        SET n TO length of s  
        SET moves TO 0  
        
        WHILE length of s GREATER THAN 1  
            SET matched TO false  
            FOR i FROM length of s MINUS 1 DOWNTO 1  
                IF character at s[i] EQUALS character at s[0]  
                    FOR j FROM i TO length of s MINUS 2  
                        SWAP character at s[j] WITH character at s[j PLUS 1]  
                        INCREMENT moves BY 1  
                    REMOVE first character from s  
                    REMOVE last character from s  
                    SET matched TO true  
                    BREAK FOR  
            IF matched IS false  
                REMOVE first character from s  
        
        RETURN moves  
    END FUNCTION  
END CLASS
