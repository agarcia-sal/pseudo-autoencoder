CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF grid AT INDEX 0  
        SET directions TO list of coordinate pairs representing right, down, left, and up movements  
        SET min_heap TO list containing a tuple of zero cost and starting coordinates (0, 0, 0)  
        SET visited TO set containing starting coordinates (0, 0)  
        
        WHILE min_heap is not empty  
            EXTRACT the element with the smallest cost from min_heap INTO cost, x, y  
            
            IF x EQUALS m MINUS 1 AND y EQUALS n MINUS 1  
                RETURN cost  
            
            FOR each dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO 0 AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO 0 AND ny LESS THAN n AND (nx, ny) NOT IN visited  
                    ADD (nx, ny) TO visited  
                    SET new_cost TO cost PLUS grid AT nx AND ny  
                    ADD (new_cost, nx, ny) TO min_heap  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
