CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode with values at index 0  
    SET i TO 1  
    SET queue TO empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element removed from queue  
        IF i LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.left TO new TreeNode with values at index i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.right TO new TreeNode with values at index i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST(self, root, val)  
        FUNCTION dfs(root)  
            IF root IS None  
                RETURN new TreeNode with val  
            IF root.val LESS THAN val  
                SET root.right TO dfs(root.right)  
            ELSE  
                SET root.left TO dfs(root.left)  
            RETURN root  
        END FUNCTION  
        RETURN dfs(root)  
    END FUNCTION  
END CLASS
