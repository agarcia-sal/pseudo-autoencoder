CLASS Solution  
    FUNCTION maxA(n)  
        IF n IS LESS THAN OR EQUAL TO 3 THEN  
            RETURN n  
        END IF  
        
        SET dp TO list of zeros with length n PLUS 1  
        
        FOR i FROM 1 TO 3 DO  
            SET dp at i TO i  
        END FOR  
        
        FOR i FROM 4 TO n DO  
            FOR j FROM i MINUS 3 DOWN TO 1 DO  
                SET candidate TO dp at j MULTIPLIED BY (i MINUS j MINUS 1)  
                IF candidate IS GREATER THAN dp at i THEN  
                    SET dp at i TO candidate  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at n  
    END FUNCTION  
END CLASS
