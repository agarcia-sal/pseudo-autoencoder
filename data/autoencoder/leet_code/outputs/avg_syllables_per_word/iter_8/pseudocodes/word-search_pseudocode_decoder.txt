CLASS Solution  
    FUNCTION exist(board, word)  
        IF board IS empty THEN  
            RETURN False  
        SET m TO LENGTH OF board  
        SET n TO LENGTH OF board AT 0  
        SET visited TO matrix of False with dimensions m BY n  

        FUNCTION dfs(x, y, index)  
            IF index IS EQUAL TO LENGTH OF word THEN  
                RETURN True  
            IF x IS LESS THAN 0 OR x IS GREATER THAN OR EQUAL TO m OR y IS LESS THAN 0 OR y IS GREATER THAN OR EQUAL TO n OR visited at x and y IS True OR board at x and y IS NOT EQUAL TO word at index THEN  
                RETURN False  
            SET visited at x and y TO True  
            FOR each pair dx, dy IN list of (-1, 0), (1, 0), (0, -1), (0, 1) DO  
                IF dfs(x plus dx, y plus dy, index plus 1) RETURNS True THEN  
                    RETURN True  
            SET visited at x and y TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM 0 TO m MINUS 1 DO  
            FOR j FROM 0 TO n MINUS 1 DO  
                IF dfs(i, j, 0) RETURNS True THEN  
                    RETURN True  
        RETURN False  
    END FUNCTION  
END CLASS
