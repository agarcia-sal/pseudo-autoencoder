CLASS Solution  
    FUNCTION fallingSquares(positions)  
        SET max_heights TO empty list  
        SET intervals TO empty list  
        SET current_max_height TO 0  

        FOR each left, side_length IN positions  
            SET right TO left PLUS side_length MINUS 1  
            SET height TO side_length  

            FOR each interval_left, interval_right, interval_height IN intervals  
                IF left LESS THAN OR EQUAL TO interval_right AND right GREATER THAN OR EQUAL TO interval_left  
                    IF side_length PLUS interval_height GREATER THAN height  
                        SET height TO side_length PLUS interval_height  
                    END IF  
                END IF  
            END FOR  

            APPEND tuple of left, right, height TO intervals  
            IF height GREATER THAN current_max_height  
                SET current_max_height TO height  
            END IF  
            APPEND current_max_height TO max_heights  
        END FOR  

        RETURN max_heights  
    END FUNCTION  
END CLASS
