CLASS Solution  
    FUNCTION numberOfWays(corridor)  
        SET MOD TO 10 raised to power 9 plus 7  
        SET seats TO list of indices i for each character x in corridor WHERE x equals 'S'  
        
        IF length of seats MODULO 2 NOT EQUAL TO 0 OR length of seats LESS THAN 2 THEN  
            RETURN 0  
        END IF  
        
        SET ways TO 1  
        FOR i FROM 2 TO length of seats STEP 2 DO  
            SET difference TO seats at position i MINUS seats at position i minus 1  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways MODULO MOD  
        END FOR  
        
        RETURN ways  
    END FUNCTION  
END CLASS
