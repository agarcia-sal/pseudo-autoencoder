CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR i FROM 0 TO 8  
                IF board[row][i] EQUALS num  
                    RETURN False  
            FOR i FROM 0 TO 8  
                IF board[i][col] EQUALS num  
                    RETURN False  
            SET start_row TO 3 MULTIPLIED BY (row DIVIDED BY 3)  
            SET start_col TO 3 MULTIPLIED BY (col DIVIDED BY 3)  
            FOR i FROM 0 TO 2  
                FOR j FROM 0 TO 2  
                    IF board[start_row PLUS i][start_col PLUS j] EQUALS num  
                        RETURN False  
            RETURN True  
        END FUNCTION  
  
        FUNCTION solve(board)  
            FOR row FROM 0 TO 8  
                FOR col FROM 0 TO 8  
                    IF board[row][col] EQUALS '.'  
                        FOR num IN STRING REPRESENTATIONS OF NUMBERS FROM 1 TO 9  
                            IF is_valid(board, row, col, num)  
                                SET board[row][col] TO num  
                                IF solve(board)  
                                    RETURN True  
                                SET board[row][col] TO '.'  
                        RETURN False  
            RETURN True  
        END FUNCTION  
  
        CALL solve(board)  
    END FUNCTION  
END CLASS
