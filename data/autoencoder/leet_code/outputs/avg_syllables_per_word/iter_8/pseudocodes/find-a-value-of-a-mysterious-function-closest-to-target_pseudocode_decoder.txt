CLASS Solution  
    FUNCTION closestToTarget(arr, target)  
        SET seen TO empty set  
        SET min_diff TO infinity  
        FOR each num IN arr  
            SET seen TO set containing bitwise AND of num and each element IN seen UNION set containing num  
            FOR each val IN seen  
                SET min_diff TO smaller value BETWEEN min_diff AND absolute value OF val MINUS target  
        RETURN min_diff  
    END FUNCTION  
END CLASS
