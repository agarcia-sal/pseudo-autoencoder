CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF LENGTH OF t EQUALS LENGTH OF s THEN  
                APPEND t TO ans  
                RETURN  
            FOR each character c AND value v IN cnt  
                IF v GREATER THAN 1 THEN  
                    DECREMENT cnt[c] BY 2  
                    CALL dfs WITH (c CONCATENATED WITH t CONCATENATED WITH c)  
                    INCREMENT cnt[c] BY 2  
        END FUNCTION  

        SET cnt TO frequency count of characters in s  
        SET mid TO empty string  
        FOR each character c AND value v IN cnt  
            IF v MODULO 2 EQUALS 1 THEN  
                IF mid IS NOT empty THEN  
                    RETURN empty list  
                SET mid TO c  
                DECREMENT cnt[c] BY 1  
        SET ans TO empty list  
        CALL dfs WITH mid  
        RETURN ans  
END CLASS
