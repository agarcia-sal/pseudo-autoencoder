CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        SET graph TO dictionary mapping each integer from 1 to n inclusive TO an empty set  
        
        FOR each pair (u, v) IN edges  
            ADD v TO the set mapped by u in graph  
            ADD u TO the set mapped by v in graph  
        
        SET min_degree TO infinity  
        SET trio_found TO false  
        
        FOR i FROM 1 TO n inclusive  
            FOR j FROM i plus 1 TO n inclusive  
                FOR k FROM j plus 1 TO n inclusive  
                    IF j IS IN graph mapped by i AND k IS IN graph mapped by i AND k IS IN graph mapped by j  
                        SET trio_found TO true  
                        SET trio_degree TO length of graph mapped by i PLUS length of graph mapped by j PLUS length of graph mapped by k MINUS 6  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
        
        IF trio_found IS true  
            RETURN min_degree  
        ELSE  
            RETURN negative 1  
    END FUNCTION  
END CLASS
