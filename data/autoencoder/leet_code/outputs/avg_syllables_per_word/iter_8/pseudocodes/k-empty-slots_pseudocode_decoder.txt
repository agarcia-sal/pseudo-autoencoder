CLASS Solution  
    FUNCTION kEmptySlots(bulbs, k)  
        SET bloom_days TO empty list  
        
        FOR each day and position IN bulbs with day starting at 1  
            SET i TO position in bloom_days where position can be inserted to keep list sorted  
            
            IF i GREATER THAN 0 AND position MINUS bloom_days at index (i MINUS 1) MINUS 1 EQUALS k  
                RETURN day  
            END IF  
            
            IF i LESS THAN LENGTH OF bloom_days AND bloom_days at index i MINUS position MINUS 1 EQUALS k  
                RETURN day  
            END IF  
            
            INSERT position INTO bloom_days AT index i  
        END FOR  
        
        RETURN negative 1  
    END FUNCTION  
END CLASS
