CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO list containing 0  
        FOR each num IN nums  
            APPEND last element of prefix_sum PLUS num TO prefix_sum  
          
        SET dq TO an empty deque  
        SET min_length TO infinite value  
          
        FOR each index i AND current_sum IN enumeration of prefix_sum  
            WHILE dq is not empty AND current_sum MINUS prefix_sum at front index of dq IS GREATER THAN OR EQUAL TO k  
                SET min_length TO minimum of min_length AND difference between i AND front index removed from dq  
              
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO prefix_sum at last index of dq  
                REMOVE last index from dq  
              
            APPEND i TO dq  
          
        IF min_length IS NOT infinite value  
            RETURN min_length  
        ELSE  
            RETURN negative one  
    END FUNCTION  
END CLASS
