CLASS Solution  
    FUNCTION longestObstacleCourseAtEachPosition(obstacles)  
        SET n TO LENGTH OF obstacles  
        SET ans TO empty list  
        SET increasing_subseq TO empty list  
        
        FOR each obstacle IN obstacles  
            SET idx TO position to insert obstacle in increasing_subseq preserving order such that obstacle is placed after equal elements  
            
            IF idx IS EQUAL TO LENGTH OF increasing_subseq  
                APPEND obstacle TO increasing_subseq  
            ELSE  
                SET element at idx of increasing_subseq TO obstacle  
            END IF  
            
            APPEND idx PLUS 1 TO ans  
        END FOR  
        
        RETURN ans  
    END FUNCTION  
END CLASS
