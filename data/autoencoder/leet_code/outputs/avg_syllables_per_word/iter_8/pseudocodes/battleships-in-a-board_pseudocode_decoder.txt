CLASS Solution  
    FUNCTION countBattleships(board)  
        SET count TO 0  
        FOR i FROM 0 TO LENGTH OF board MINUS 1  
            FOR j FROM 0 TO LENGTH OF board AT 0 MINUS 1  
                IF board AT i AT j EQUALS 'X' THEN  
                    IF (i EQUALS 0 OR board AT i MINUS 1 AT j EQUALS '.') AND (j EQUALS 0 OR board AT i AT j MINUS 1 EQUALS '.') THEN  
                        INCREMENT count BY 1  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
