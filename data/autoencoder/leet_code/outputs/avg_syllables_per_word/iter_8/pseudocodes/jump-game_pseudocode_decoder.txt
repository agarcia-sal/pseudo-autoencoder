CLASS Solution  
    FUNCTION canJump(nums)  
        SET farthest TO 0  
        FOR i FROM 0 TO LENGTH OF nums MINUS 1  
            IF i GREATER THAN farthest  
                RETURN False  
            SET farthest TO MAXIMUM OF farthest AND i PLUS nums AT i  
        RETURN farthest GREATER THAN OR EQUAL TO LENGTH OF nums MINUS 1  
    END FUNCTION  
END CLASS
