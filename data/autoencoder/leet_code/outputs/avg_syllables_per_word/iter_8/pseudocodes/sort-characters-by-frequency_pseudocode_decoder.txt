CLASS Solution  
    FUNCTION frequencySort(s)  
        SET frequency TO COUNT frequency OF each character IN s  
        SET sorted_characters TO LIST OF characters IN frequency SORTED BY frequency OF character IN DESCENDING ORDER  
        SET result TO EMPTY STRING  
        FOR each char IN sorted_characters  
            APPEND char REPEATED frequency OF char TIMES TO result  
        RETURN result  
    END FUNCTION  
END CLASS
