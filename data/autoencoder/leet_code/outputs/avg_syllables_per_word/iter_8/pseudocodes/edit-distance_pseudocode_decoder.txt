CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO LENGTH OF word1  
        SET n TO LENGTH OF word2  

        SET dp TO new two-dimensional list of size (m plus 1) by (n plus 1) with all elements initialized to 0  

        FOR i FROM 0 TO m  
            SET dp at position i, 0 TO i  
        FOR j FROM 0 TO n  
            SET dp at position 0, j TO j  

        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF character at position i minus 1 in word1 EQUALS character at position j minus 1 in word2  
                    SET dp at position i, j TO dp at position i minus 1, j minus 1  
                ELSE  
                    SET dp at position i, j TO 1 plus MINIMUM OF dp at position i minus 1, j, dp at position i, j minus 1, dp at position i minus 1, j minus 1  
        RETURN dp at position m, n  
    END FUNCTION  
END CLASS
