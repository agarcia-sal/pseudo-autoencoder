CLASS Solution  
    FUNCTION nthSuperUglyNumber(n, primes)  
        SET ugly_numbers TO list containing 1  
        SET heap TO list of pairs (prime, index) for each prime in primes with its index  
        TRANSFORM heap INTO a min-heap  
        SET indices TO list of zeros with length equal to length of primes  
        
        WHILE LENGTH OF ugly_numbers LESS THAN n  
            REMOVE smallest element FROM heap INTO next_ugly AND prime_index  
            
            IF next_ugly IS NOT EQUAL TO last element in ugly_numbers  
                APPEND next_ugly TO ugly_numbers  
            END IF  
            
            INCREMENT indices at prime_index BY 1  
            SET next_multiple TO primes at prime_index MULTIPLIED BY ugly_numbers at indices at prime_index  
            ADD (next_multiple, prime_index) TO heap  
        END WHILE  
        
        RETURN last element in ugly_numbers  
    END FUNCTION  
END CLASS
