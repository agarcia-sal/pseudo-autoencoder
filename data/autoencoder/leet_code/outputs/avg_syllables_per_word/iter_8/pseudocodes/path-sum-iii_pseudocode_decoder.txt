CLASS TreeNode  
    FUNCTION __init__(val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode with val as first element of values  
    SET i TO 1  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.left TO new TreeNode with val as values at i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.right TO new TreeNode with val as values at i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(root, targetSum)  
        FUNCTION helper(node, current_sum, prefix_sums)  
            IF node IS None  
                RETURN 0  
            SET current_sum TO current_sum PLUS node.val  
            SET paths_to_here TO prefix_sums GET VALUE FOR (current_sum MINUS targetSum) OR 0  
            SET prefix_sums[current_sum] TO prefix_sums GET VALUE FOR current_sum OR 0 PLUS 1  
            SET paths_in_left TO helper(node.left, current_sum, prefix_sums)  
            SET paths_in_right TO helper(node.right, current_sum, prefix_sums)  
            DECREMENT prefix_sums[current_sum] BY 1  
            RETURN paths_to_here PLUS paths_in_left PLUS paths_in_right  
        END FUNCTION  
        SET prefix_sums TO DICTIONARY WITH KEY 0 AND VALUE 1  
        RETURN helper(root, 0, prefix_sums)  
    END FUNCTION  
END CLASS
