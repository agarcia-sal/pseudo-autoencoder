CLASS Solution  
    FUNCTION splitArray(nums)  
        SET n TO LENGTH OF nums  
        IF n LESS THAN 7  
            RETURN False  
        END IF  
        
        SET prefix_sum TO LIST OF SIZE n WITH ALL ELEMENTS 0  
        SET prefix_sum AT 0 TO nums AT 0  
        FOR i FROM 1 TO n MINUS 1  
            SET prefix_sum AT i TO prefix_sum AT i MINUS 1 PLUS nums AT i  
        END FOR  
        
        FOR j FROM 3 TO n MINUS 4  
            SET seen TO EMPTY SET  
            FOR i FROM 1 TO j MINUS 2  
                IF prefix_sum AT i MINUS 1 EQUALS prefix_sum AT j MINUS 1 MINUS prefix_sum AT i  
                    ADD prefix_sum AT i MINUS 1 TO seen  
                END IF  
            END FOR  
            
            FOR k FROM j PLUS 2 TO n MINUS 2  
                IF prefix_sum AT n MINUS 1 MINUS prefix_sum AT k EQUALS prefix_sum AT k MINUS 1 MINUS prefix_sum AT j AND prefix_sum AT k MINUS 1 MINUS prefix_sum AT j IN seen  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
