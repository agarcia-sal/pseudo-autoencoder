CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET max_so_far TO empty list  
        SET current_max TO first element of arr  
        FOR each num IN arr  
            SET current_max TO greater of current_max and num  
            APPEND current_max TO max_so_far  
        SET sorted_arr TO sorted version of arr  
        SET chunks TO 0  
        SET max_chunk TO 0  
        FOR index FROM 0 TO length of arr MINUS 1  
            SET max_chunk TO greater of max_chunk and element at arr index  
            IF max_chunk EQUALS element at sorted_arr index THEN  
                INCREMENT chunks BY 1  
        RETURN chunks  
    END FUNCTION  
END CLASS
