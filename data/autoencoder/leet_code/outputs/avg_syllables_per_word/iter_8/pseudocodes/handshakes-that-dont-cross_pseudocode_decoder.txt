CLASS Solution  
    FUNCTION numberOfWays(numPeople)  
        SET MOD TO 10 RAISED TO THE POWER 9 PLUS 7  
        INITIALIZE dp AS LIST OF ZEROS WITH LENGTH numPeople PLUS 1  
        SET dp AT INDEX 0 TO 1  
        SET dp AT INDEX 2 TO 1  
        FOR i FROM 4 TO numPeople IN STEPS OF 2  
            FOR j FROM 0 TO i MINUS 1 IN STEPS OF 2  
                SET dp AT INDEX i TO dp AT INDEX i PLUS dp AT INDEX j MULTIPLIED BY dp AT INDEX i MINUS 2 MINUS j MODULO MOD  
        RETURN dp AT INDEX numPeople  
    END FUNCTION  
END CLASS
