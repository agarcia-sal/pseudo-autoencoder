CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO a default dictionary with list as default value  
        FOR each string s IN strs  
            SET sorted_str TO the characters of s arranged in sorted order joined as a string  
            APPEND s TO the list in anagrams corresponding to key sorted_str  
        RETURN a list containing all the values from anagrams  
    END FUNCTION  
END CLASS
