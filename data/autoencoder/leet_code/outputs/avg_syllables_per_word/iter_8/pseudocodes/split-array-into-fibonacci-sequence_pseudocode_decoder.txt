CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS LENGTH OF num AND LENGTH OF path IS GREATER THAN OR EQUAL TO 3  
                RETURN path  
            END IF  
            FOR end FROM start PLUS 1 TO LENGTH OF num PLUS 1  
                IF character AT num AT start EQUALS '0' AND end IS GREATER THAN start PLUS 1  
                    CONTINUE TO NEXT ITERATION  
                END IF  
                SET next_num TO integer VALUE OF substring OF num FROM start TO end  
                IF next_num IS GREATER THAN OR EQUAL TO 2 RAISED TO 31  
                    BREAK LOOP  
                END IF  
                IF LENGTH OF path IS LESS THAN 2 OR next_num EQUALS SUM OF LAST TWO ELEMENTS OF path  
                    SET result TO backtrack(end, path CONCATENATED WITH next_num)  
                    IF result IS NOT EMPTY  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN empty list  
        END FUNCTION  
        RETURN backtrack(0, empty list)  
    END FUNCTION  
END CLASS
