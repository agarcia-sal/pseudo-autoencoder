CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO list of length 26 with all values set to -1  
        SET total_appeal TO 0  
        SET current_appeal TO 0  
        
        FOR each index i and character char IN s  
            SET index TO position of char in alphabet starting from 0  
            INCREMENT current_appeal BY i MINUS last AT position index  
            INCREMENT total_appeal BY current_appeal  
            SET last AT position index TO i  
        
        RETURN total_appeal  
    END FUNCTION  
END CLASS
