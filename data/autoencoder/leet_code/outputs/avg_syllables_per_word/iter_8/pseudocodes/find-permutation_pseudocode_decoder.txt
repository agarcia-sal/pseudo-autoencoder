CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO list of integers from 1 TO length of s PLUS 1  
        SET start TO 0  
        FOR end FROM 0 TO length of s  
            IF end EQUALS length of s OR character at position end in s EQUALS 'I'  
                REVERSE the segment of result FROM start TO end INCLUSIVE  
                SET start TO end PLUS 1  
        RETURN result  
    END FUNCTION  
END CLASS
