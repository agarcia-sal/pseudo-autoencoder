CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO infinity  
        FOR each num IN nums  
            IF num MODULO 2 EQUALS 1  
                SET num TO num MULTIPLIED BY 2  
            END IF  
            SET min_value TO the smaller of min_value AND num  
            ADD negative of num TO max_heap as a heap element  
        END FOR  
        
        SET min_deviation TO infinity  
        
        WHILE the top element of max_heap MODULO 2 EQUALS 0  
            SET max_value TO negative of removing the top element from max_heap  
            SET min_deviation TO the smaller of min_deviation AND (max_value MINUS min_value)  
            SET max_value TO max_value DIVIDED BY 2 (integer division)  
            SET min_value TO the smaller of min_value AND max_value  
            ADD negative of max_value TO max_heap as a heap element  
        END WHILE  
        
        SET min_deviation TO the smaller of min_deviation AND (negative of top element of max_heap MINUS min_value)  
        
        RETURN min_deviation  
    END FUNCTION  
END CLASS
