CLASS Solution  
    FUNCTION countOrders(n)  
        SET MOD TO 10 raised to the power 9 plus 7  
        SET count TO 1  
        FOR i FROM 2 TO n INCLUSIVE  
            SET count TO count multiplied BY i multiplied BY (2 multiplied BY i minus 1) MODULO MOD  
        RETURN count  
    END FUNCTION  
END CLASS
