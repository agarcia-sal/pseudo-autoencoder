CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO 1  
            RETURN s  
        END IF  
        
        SET freq TO frequency count of characters in s  
        
        SET max_heap TO list of pairs with negative frequency and character from freq  
        TRANSFORM max_heap INTO a heap structure  
        
        SET wait_queue TO empty list  
        SET result TO empty list  
        
        WHILE max_heap is not empty  
            REMOVE the element with highest frequency from max_heap and SET TO count, char  
            APPEND char TO result  
            
            APPEND pair of count incremented by one and char TO wait_queue  
            
            IF length of wait_queue EQUALS k  
                REMOVE the first element from wait_queue and SET TO count, char  
                IF count LESS THAN zero  
                    ADD pair count, char TO max_heap  
                END IF  
            END IF  
        END WHILE  
        
        IF length of result NOT EQUAL TO length of s  
            RETURN empty string  
        END IF  
        
        RETURN concatenation of characters in result  
    END FUNCTION  
END CLASS
