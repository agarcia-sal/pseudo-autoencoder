CLASS Solution  
    FUNCTION isValidPalindrome(s, k)  
        FUNCTION longest_palindromic_subseq(s)  
            SET n TO LENGTH OF s  
            INITIALIZE dp AS TWO-DIMENSIONAL LIST OF SIZE n BY n WITH ZEROS  
            
            FOR i FROM 0 TO n MINUS 1  
                SET dp AT i AND i TO 1  
            
            FOR length FROM 2 TO n  
                FOR i FROM 0 TO n MINUS length  
                    SET j TO i PLUS length MINUS 1  
                    IF CHARACTER AT s AT i EQUALS CHARACTER AT s AT j THEN  
                        SET dp AT i AND j TO dp AT i PLUS 1 AND j MINUS 1 PLUS 2  
                    ELSE  
                        SET dp AT i AND j TO MAXIMUM OF dp AT i PLUS 1 AND j AND dp AT i AND j MINUS 1  
            
            RETURN dp AT 0 AND n MINUS 1  
        
        SET longest_palindrome_length TO CALL longest_palindromic_subseq WITH s  
        RETURN LENGTH OF s MINUS longest_palindrome_length LESS THAN OR EQUAL TO k  
    END FUNCTION  
END CLASS
