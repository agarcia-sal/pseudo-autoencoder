CLASS Solution  
    FUNCTION new21Game(n, k, maxPts)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k THEN  
                RETURN 1 IF i LESS THAN OR EQUAL TO n OTHERWISE 0  
            END IF  
            IF i EQUAL TO k MINUS 1 THEN  
                RETURN MINIMUM OF (n MINUS k PLUS 1) AND maxPts DIVIDED BY maxPts  
            END IF  
            RETURN dfs(i PLUS 1) PLUS (dfs(i PLUS 1) MINUS dfs(i PLUS maxPts PLUS 1)) DIVIDED BY maxPts  
        END FUNCTION  
        RETURN dfs(0)  
END CLASS
