CLASS Solution  
    FUNCTION visiblePoints(points, angle, location)  
        SET angles TO empty list  
        SET extra_points TO 0  
        
        FOR each (x, y) IN points  
            IF x EQUALS location at index 0 AND y EQUALS location at index 1 THEN  
                INCREMENT extra_points BY 1  
                CONTINUE to next iteration  
            END IF  
            SET angle_rad TO arctangent of (y MINUS location at index 1, x MINUS location at index 0)  
            SET angle_deg TO convert angle_rad TO degrees  
            APPEND angle_deg TO angles  
        END FOR  
        
        SORT angles IN ascending order  
        
        SET angles TO concatenate angles WITH each angle plus 360  
        
        SET max_visible TO 0  
        SET left TO 0  
        
        FOR right FROM 0 TO length of angles MINUS 1  
            WHILE difference BETWEEN angles at right AND angles at left IS GREATER THAN angle  
                INCREMENT left BY 1  
            END WHILE  
            SET max_visible TO maximum of max_visible AND (right MINUS left PLUS 1)  
        END FOR  
        
        RETURN max_visible PLUS extra_points  
    END FUNCTION  
END CLASS
