CLASS Solution  
    FUNCTION orderlyQueue(s, k)  
        IF k GREATER THAN 1 THEN  
            RETURN concatenation of sorting characters of s  
        ELSE  
            SET n TO length of s  
            SET smallest TO s  
            FOR i FROM 1 TO n MINUS 1  
                SET rotated TO concatenation of substring of s from i to end AND substring of s from start to i  
                IF rotated LESS THAN smallest THEN  
                    SET smallest TO rotated  
            RETURN smallest  
    END FUNCTION  
END CLASS
