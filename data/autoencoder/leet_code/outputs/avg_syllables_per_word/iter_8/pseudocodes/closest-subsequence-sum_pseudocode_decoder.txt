CLASS Solution  
    FUNCTION minAbsDifference(nums, goal)  
        SET n TO LENGTH OF nums  
        SET mid TO n DIVIDED BY 2  
        
        SET left_sums TO LIST CONTAINING 0  
        FOR each num IN nums UP TO mid  
            EXTEND left_sums BY NEW LIST OF EACH s PLUS num FOR s IN left_sums  
        
        SET right_sums TO LIST CONTAINING 0  
        FOR each num IN nums FROM mid TO END  
            EXTEND right_sums BY NEW LIST OF EACH s PLUS num FOR s IN right_sums  
        
        SORT right_sums  
        
        SET min_diff TO infinity  
        
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO POSITION TO INSERT target IN right_sums USING BINARY SEARCH  
            
            IF pos IS LESS THAN LENGTH OF right_sums THEN  
                SET min_diff TO SMALLER OF min_diff AND ABSOLUTE VALUE OF left_sum PLUS right_sums AT pos MINUS goal  
            END IF  
            
            IF pos IS GREATER THAN 0 THEN  
                SET min_diff TO SMALLER OF min_diff AND ABSOLUTE VALUE OF left_sum PLUS right_sums AT pos MINUS 1 MINUS goal  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
