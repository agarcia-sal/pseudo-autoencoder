CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS reverse of sub  
        END FUNCTION  
        
        SET n TO length of s  
        
        SET is_pal TO two-dimensional list of size n by n FILLED WITH False  
        
        FOR i FROM n MINUS 1 DOWNTO 0  
            FOR j FROM i TO n MINUS 1  
                IF character at s index i EQUALS character at s index j AND  
                   (j MINUS i LESS THAN OR EQUAL TO 1 OR is_pal at [i PLUS 1][j MINUS 1] IS True)  
                    SET is_pal at [i][j] TO True  
        
        FOR i FROM 1 TO n MINUS 2  
            IF is_pal at [0][i MINUS 1] IS True  
                FOR j FROM i TO n MINUS 2  
                    IF is_pal at [i][j] IS True AND is_pal at [j PLUS 1][n MINUS 1] IS True  
                        RETURN True  
        RETURN False  
    END FUNCTION  
END CLASS
