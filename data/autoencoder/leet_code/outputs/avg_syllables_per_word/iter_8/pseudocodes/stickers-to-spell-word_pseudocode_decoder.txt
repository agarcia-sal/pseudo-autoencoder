CLASS Solution  
    FUNCTION minStickers(stickers, target)  
        SET stickers TO list of frequency maps for each sticker in stickers  
        SORT stickers by descending order of their highest character frequency  
        SET queue TO new queue containing tuple of target and zero  
        SET memo TO map with target as key and zero as value  
        
        WHILE queue is not empty  
            SET current_target, used_stickers TO dequeue from queue  
            
            FOR each sticker IN stickers  
                IF first character of current_target is NOT in sticker THEN  
                    CONTINUE to next sticker  
                
                SET remaining TO frequency map of current_target  
                SUBTRACT frequencies of sticker from remaining  
                SET new_target TO concatenation of characters in remaining sorted  
                
                IF new_target is empty THEN  
                    RETURN used_stickers plus one  
                
                IF new_target is NOT in memo THEN  
                    SET memo entry for new_target TO used_stickers plus one  
                    ENQUEUE tuple of new_target and used_stickers plus one into queue  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
