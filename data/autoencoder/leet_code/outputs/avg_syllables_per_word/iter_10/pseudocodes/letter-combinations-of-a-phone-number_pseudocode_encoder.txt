CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string  
            RETURN empty list  
        END IF  
        
        SET phone_map TO a mapping from digits TO their corresponding letters as follows  
            digit two maps TO letters a b c  
            digit three maps TO letters d e f  
            digit four maps TO letters g h i  
            digit five maps TO letters j k l  
            digit six maps TO letters m n o  
            digit seven maps TO letters p q r s  
            digit eight maps TO letters t u v  
            digit nine maps TO letters w x y z  
        
        FUNCTION backtrack(combination, next_digits)  
            IF the length OF next_digits EQUALS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the letters mapped BY phone_map at the first element OF next_digits  
                    CALL backtrack WITH combination concatenated WITH letter, AND next_digits starting from the second element  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO empty list  
        CALL backtrack WITH empty string, AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
