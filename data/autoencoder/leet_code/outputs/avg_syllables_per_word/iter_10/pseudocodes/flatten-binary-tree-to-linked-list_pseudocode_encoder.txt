CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET val TO parameter val  
        SET left TO parameter left  
        SET right TO parameter right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS nothing  
        RETURN nothing  
    END IF  
    SET root TO new TreeNode CREATED WITH first element of values  
    SET i TO one  
    INITIALIZE an empty queue named queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing  
            SET node.left TO new TreeNode CREATED WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing  
            SET node.right TO new TreeNode CREATED WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN true  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN false  
    ELSE IF value at p NOT EQUALS value at q  
        RETURN false  
    ELSE  
        RETURN is_same_tree with arguments p.left and q.left AND is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten with parameter root  
        IF root EQUALS nothing  
            RETURN  
        END IF  
        INITIALIZE a list named stack CONTAINING root  
        WHILE stack IS NOT empty  
            SET node TO last element REMOVED FROM stack  
            IF node.right NOT EQUALS nothing  
                APPEND node.right TO stack  
            END IF  
            IF node.left NOT EQUALS nothing  
                APPEND node.left TO stack  
            END IF  
            SET node.left TO nothing  
            IF stack IS NOT empty  
                SET node.right TO last element of stack  
            ELSE  
                SET node.right TO nothing  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
