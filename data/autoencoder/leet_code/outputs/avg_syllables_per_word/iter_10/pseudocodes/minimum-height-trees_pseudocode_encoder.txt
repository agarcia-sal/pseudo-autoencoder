CLASS Solution  
    FUNCTION findMinHeightTrees(n, edges)  
        IF n EQUALS one  
            RETURN a list containing zero  
        END IF  
        
        SET graph TO a new default dictionary with list as default value  
        SET degree TO a list of zero repeated n times  
        
        FOR each pair u, v IN edges  
            APPEND v TO the list at key u in graph  
            APPEND u TO the list at key v in graph  
            INCREMENT the element at position u in degree BY one  
            INCREMENT the element at position v in degree BY one  
        END FOR  
        
        SET leaves TO a new deque containing all indices i from zero TO n MINUS one WHERE the element at position i in degree EQUALS one  
        
        SET remaining_nodes TO n  
        WHILE remaining_nodes GREATER THAN two  
            SET leaves_count TO the length of leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR index FROM one TO leaves_count  
                SET leaf TO the first element removed from leaves  
                FOR each neighbor IN the list at key leaf in graph  
                    DECREMENT the element at position neighbor in degree BY one  
                    IF the element at position neighbor in degree EQUALS one  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN the list converted from leaves  
    END FUNCTION  
END CLASS
