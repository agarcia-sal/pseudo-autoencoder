CLASS Solution  
    FUNCTION countSubgraphsForEachDiameter WITH PARAMETERS n, edges  
        SET graph TO a mapping from each integer element FROM one TO n INCLUDING one MAPPED TO an empty list  
        FOR each pair u, v IN edges  
            APPEND v TO the list mapped by u in graph  
            APPEND u TO the list mapped by v in graph  
        END FOR  
        
        FUNCTION bfs WITH PARAMETER subtree  
            SET start TO an arbitrary element taken from subtree  
            SET queue TO a list initialized with start as first element  
            SET visited TO a set containing start  
            SET distance TO zero  
            SET farthest_node TO start  
            
            WHILE queue is not empty  
                REMOVE first element FROM queue AND SET TO node  
                FOR each neighbor IN the list mapped by node in graph  
                    IF neighbor is an element of subtree AND neighbor is not an element of visited  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                        SET farthest_node TO neighbor  
                    END IF  
                END FOR  
            END WHILE  
            
            SET queue TO a list initialized with farthest_node as first element  
            SET visited TO a set containing farthest_node  
            SET max_distance TO zero  
            
            WHILE queue is not empty  
                SET size TO the LENGTH OF queue  
                FOR index FROM one TO size  
                    REMOVE first element FROM queue AND SET TO node  
                    FOR each neighbor IN the list mapped by node in graph  
                        IF neighbor is an element of subtree AND neighbor is not an element of visited  
                            ADD neighbor TO visited  
                            APPEND neighbor TO queue  
                        END IF  
                    END FOR  
                END FOR  
                IF queue is not empty  
                    INCREMENT max_distance BY one  
                END IF  
            END WHILE  
            
            RETURN max_distance  
        END FUNCTION  
        
        SET result TO a list of zero repeated n MINUS one times  
        
        FOR size FROM two TO n  
            FOR each subset IN all combinations of integers FROM one TO n WITH length size  
                SET visited TO an empty set  
                SET stack TO a list initialized with first element of subset  
                WHILE stack is not empty  
                    REMOVE last element FROM stack AND SET TO node  
                    IF node is not an element of visited  
                        ADD node TO visited  
                        FOR each neighbor IN the list mapped by node in graph  
                            IF neighbor is an element of subset AND neighbor is not an element of visited  
                                APPEND neighbor TO stack  
                            END IF  
                        END FOR  
                    END IF  
                END WHILE  
                
                IF visited EQUALS the set converted from subset  
                    SET diameter TO bfs WITH subtree converted from subset to set  
                    IF diameter GREATER THAN zero  
                        INCREMENT element at position diameter MINUS one of result BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
