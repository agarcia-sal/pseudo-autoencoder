CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an EMPTY set  
        SET min_heap TO a list containing the tuple of element at position zero zero of grid, zero, zero  
        ADD the tuple of zero and zero TO visited  
        
        SET directions TO a list containing the tuples zero one, one zero, zero MINUS one, MINUS one zero  
        
        WHILE min_heap IS NOT empty  
            REMOVE and RETURN the smallest element FROM min_heap AS t, i, j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            
            FOR each di, dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND the tuple ni nj IS NOT IN visited  
                    ADD the tuple ni nj TO visited  
                    PUSH the tuple of the maximum OF t AND element at position ni nj of grid, ni, nj INTO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN MINUS one  
    END FUNCTION  
END CLASS
