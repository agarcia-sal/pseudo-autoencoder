CLASS Solution  
    FUNCTION removeDuplicateLetters(s)  
        SET last_occurrence TO a dictionary that maps each character IN s TO its last position index IN s  
        SET stack TO an empty list  
        SET in_stack TO an empty set  
        
        FOR each index_and_character AT position idx AND character char IN s  
            IF char IS IN in_stack  
                CONTINUE to next iteration  
            END IF  
            
            WHILE the LENGTH OF stack IS GREATER THAN zero AND character char IS LESS THAN element at the last position OF stack AND idx IS LESS THAN the last_occurrence OF element at the last position OF stack  
                SET removed_char TO the removal of the last element FROM stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            
            APPEND char TO stack  
            ADD char TO in_stack  
        END FOR  
        
        SET result TO the concatenation of elements IN stack  
        RETURN result  
    END FUNCTION  
END CLASS
