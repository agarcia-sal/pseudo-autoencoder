CLASS Solution  
    FUNCTION partition(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS sub IN REVERSE ORDER  
        END FUNCTION  

        FUNCTION backtrack(start, path)  
            IF start EQUALS LENGTH OF s  
                APPEND path TO result  
                RETURN  
            END IF  
            FOR end FROM start PLUS one TO LENGTH OF s  
                IF is_palindrome(ELEMENTS OF s FROM start TO end MINUS one)  
                    CALL backtrack WITH end AND path APPENDED WITH ELEMENTS OF s FROM start TO end MINUS one  
                END IF  
            END FOR  
        END FUNCTION  

        SET result TO empty list  
        CALL backtrack WITH zero AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
