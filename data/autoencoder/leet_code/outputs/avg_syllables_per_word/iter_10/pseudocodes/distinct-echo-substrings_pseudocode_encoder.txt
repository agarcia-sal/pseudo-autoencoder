CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO the LENGTH OF text  
        SET substrings TO an empty set  
        FOR length FROM one TO n DIVIDED BY two PLUS one MINUS one  
            FOR start FROM zero TO n MINUS two MULTIPLIED BY length PLUS one MINUS one  
                SET a TO substring of text from position start TO position start PLUS length MINUS one  
                SET b TO substring of text from position start PLUS length TO position start PLUS two MULTIPLIED BY length MINUS one  
                IF a EQUALS b  
                    ADD concatenation of a AND b TO substrings  
                END IF  
            END FOR  
        END FOR  
        RETURN the LENGTH OF substrings  
    END FUNCTION  
END CLASS
