CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the TUPLE of x and y IS IN visited OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            ADD the TUPLE of x and y TO visited  
            FOR each the TUPLE of dx and dy IN the LIST of the TUPLE negative one and zero, the TUPLE one and zero, the TUPLE zero and negative one, the TUPLE zero and one  
                CALL dfs WITH PARAMETERS x PLUS dx, y PLUS dy, visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an EMPTY SET  
            SET land_cells TO an EMPTY LIST  
            FOR index i FROM zero TO the LENGTH OF grid MINUS one  
                FOR index j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                    IF element at position i of grid at position j EQUALS one  
                        APPEND the TUPLE of i and j TO land_cells  
                    END IF  
                END FOR  
            END FOR  
            IF the LENGTH OF land_cells EQUALS zero  
                RETURN true  
            END IF  
            CALL dfs WITH PARAMETERS element at position zero of land_cells AT position zero, element at position zero of land_cells AT position one, visited  
            RETURN the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF NOT is_connected()  
            RETURN zero  
        END IF  

        FOR index i FROM zero TO the LENGTH OF grid MINUS one  
            FOR index j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    SET element at position i of grid at position j TO zero  
                    IF NOT is_connected()  
                        RETURN one  
                    END IF  
                    SET element at position i of grid at position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
