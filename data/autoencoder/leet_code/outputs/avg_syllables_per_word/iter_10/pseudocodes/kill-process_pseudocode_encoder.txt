CLASS Solution  
    FUNCTION killProcess(pid, ppid, kill)  
        SET tree TO a new dictionary with default empty list for missing keys  
        FOR each child AND parent IN pairs from pid AND ppid  
            APPEND child TO the list at key parent IN tree  
        END FOR  
        SET queue TO a new list containing kill  
        SET killed_processes TO an empty list  
        WHILE the LENGTH OF queue IS greater than zero  
            SET current TO the first element of queue  
            REMOVE first element FROM queue  
            APPEND current TO killed_processes  
            FOR each child IN the list at key current IN tree  
                APPEND child TO queue  
            END FOR  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
