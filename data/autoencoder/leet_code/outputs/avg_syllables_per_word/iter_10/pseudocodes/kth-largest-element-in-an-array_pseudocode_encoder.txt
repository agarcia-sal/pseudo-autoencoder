CLASS Solution  
    FUNCTION findKthLargest(nums, k)  
        SET min_heap TO an empty collection representing a heap  
        FOR each num IN nums  
            ADD num TO min_heap maintaining heap order  
            IF the LENGTH OF min_heap GREATER THAN k  
                REMOVE the smallest element FROM min_heap maintaining heap order  
            END IF  
        END FOR  
        RETURN the smallest element at position zero OF min_heap  
    END FUNCTION  
END CLASS
