CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN list containing zero  
        END IF  
        SET prev_gray TO CALL grayCode WITH argument n MINUS one  
        SET mask TO one MULTIPLIED BY two RAISED TO THE POWER OF n MINUS one  
        SET current_gray TO prev_gray CONCATENATED WITH list made by FOR each num IN reversed prev_gray APPEND mask OR num TO this list  
        RETURN current_gray  
    END FUNCTION  
END CLASS
