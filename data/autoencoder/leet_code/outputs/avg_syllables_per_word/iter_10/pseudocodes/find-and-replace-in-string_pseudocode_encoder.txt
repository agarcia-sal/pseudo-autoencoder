CLASS Solution  
    FUNCTION findReplaceString(s, indices, sources, targets)  
        SET replacements TO a list of tuples pairing each element from indices, sources, and targets  
        SORT replacements BY the first element of each tuple in ascending order  
        
        SET result TO an empty list  
        SET prev_end TO zero  
        
        FOR each tuple of index, source, target IN replacements  
            APPEND the substring of s from prev_end UP TO index TO result  
            
            IF the substring of s from index UP TO index PLUS the LENGTH OF source EQUALS source  
                APPEND target TO result  
            ELSE  
                APPEND the substring of s from index UP TO index PLUS the LENGTH OF source TO result  
            END IF  
            
            SET prev_end TO index PLUS the LENGTH OF source  
        END FOR  
        
        APPEND the substring of s from prev_end TO the END TO result  
        
        SET joined_result TO the concatenation of all elements in result  
        RETURN joined_result  
    END FUNCTION  
END CLASS
