CLASS Solution  
    FUNCTION numOfArrays(n, m, k)  
        SET MOD TO one billion PLUS seven  
        
        CALL initialize_dp WITH n PLUS one, m PLUS one, k PLUS one RETURNING dp  
        
        FOR each j FROM one TO m  
            SET element at position one of dp at position j at position one TO one  
        END FOR  
        
        FOR each i FROM two TO n  
            FOR each j FROM one TO m  
                FOR each p FROM one TO k  
                    
                    FOR each x FROM one TO j MINUS one  
                        SET element at position i of dp at position j at position p TO element at position i of dp at position j at position p PLUS element at position i MINUS one of dp at position x at position p MINUS one  
                        SET element at position i of dp at position j at position p TO element at position i of dp at position j at position p MODULO MOD  
                    END FOR  
                    
                    SET element at position i of dp at position j at position p TO element at position i of dp at position j at position p PLUS element at position i MINUS one of dp at position j at position p MULTIPLIED BY j  
                    SET element at position i of dp at position j at position p TO element at position i of dp at position j at position p MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR each j FROM one TO m  
            SET result TO result PLUS element at position n of dp at position j at position k  
            SET result TO result MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
    
    FUNCTION initialize_dp(length_n, length_m, length_k)  
        SET dp TO empty list  
        FOR each i FROM zero TO length_n MINUS one  
            SET list_m TO empty list  
            FOR each j FROM zero TO length_m MINUS one  
                SET list_k TO empty list  
                FOR each p FROM zero TO length_k MINUS one  
                    APPEND zero TO list_k  
                END FOR  
                APPEND list_k TO list_m  
            END FOR  
            APPEND list_m TO dp  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
