CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the TUPLE of index AND current_sum IS IN memo  
                RETURN the VALUE OF memo AT the TUPLE of index AND current_sum  
            END IF  
            SET positive TO dfs WITH index PLUS one, current_sum PLUS the ELEMENT at position index OF nums, AND memo  
            SET negative TO dfs WITH index PLUS one, current_sum MINUS the ELEMENT at position index OF nums, AND memo  
            SET memo AT the TUPLE of index AND current_sum TO positive PLUS negative  
            RETURN the VALUE OF memo AT the TUPLE of index AND current_sum  
        END FUNCTION  
        RETURN dfs WITH zero, zero, AND an EMPTY COLLECTION  
    END FUNCTION  
END CLASS
