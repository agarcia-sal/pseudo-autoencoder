CLASS Solution  
    FUNCTION numMagicSquaresInside(grid)  
        FUNCTION is_magic(i, j)  
            IF element at position i and j of grid NOT EQUALS five  
                RETURN False  
            END IF  
            SET numbers TO empty set  
            FOR x FROM i MINUS one TO i PLUS one  
                FOR y FROM j MINUS one TO j PLUS one  
                    IF element at position x and y of grid LESS THAN one OR element at position x and y of grid GREATER THAN nine OR element at position x and y of grid IN numbers  
                        RETURN False  
                    END IF  
                    ADD element at position x and y of grid TO numbers  
                END FOR  
            END FOR  
            FOR start FROM i MINUS one TO i PLUS one  
                IF sum of elements at positions start and j MINUS one PLUS start and j PLUS start and j PLUS one of grid NOT EQUALS fifteen OR sum of elements at positions i MINUS one and start PLUS i and start PLUS i PLUS one and start of grid NOT EQUALS fifteen  
                    RETURN False  
                END IF  
            END FOR  
            IF sum of elements at positions i MINUS one and j MINUS one PLUS i and j PLUS i PLUS one and j PLUS one of grid NOT EQUALS fifteen OR sum of elements at positions i MINUS one and j PLUS one PLUS i and j PLUS i PLUS one and j MINUS one of grid NOT EQUALS fifteen  
                RETURN False  
            END IF  
            RETURN True  
        END FUNCTION  
        SET count TO zero  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF first element of grid  
        FOR i FROM one TO rows MINUS two  
            FOR j FROM one TO cols MINUS two  
                IF is_magic(i, j)  
                    INCREMENT count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
