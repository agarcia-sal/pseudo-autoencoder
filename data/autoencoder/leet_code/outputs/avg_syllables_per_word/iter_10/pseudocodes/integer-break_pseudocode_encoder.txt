CLASS Solution  
    FUNCTION integerBreak(n)  
        SET dp TO list of zero repeated n PLUS one times  
        SET element at position one of dp TO one  
        FOR i FROM two TO n  
            SET max_product TO zero  
            FOR j FROM one TO i MINUS one  
                SET product_one TO max_product  
                SET product_two TO j MULTIPLIED BY element at position i MINUS j of dp  
                SET product_three TO j MULTIPLIED BY i MINUS j  
                SET max_product TO the maximum of product_one, product_two, and product_three  
            END FOR  
            SET element at position i of dp TO max_product  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
