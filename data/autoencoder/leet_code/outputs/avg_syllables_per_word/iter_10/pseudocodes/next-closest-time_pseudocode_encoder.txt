CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of all characters in time with the colon character removed  
        SET hour AND minute TO the integer values of the two parts resulting from splitting time by the colon character  
        
        FUNCTION generate_times(digits)  
            SET times TO empty list  
            FOR each h1 IN digits  
                FOR each h2 IN digits  
                    FOR each m1 IN digits  
                        FOR each m2 IN digits  
                            SET hour TO the integer value of the concatenation of h1 and h2  
                            SET minute TO the integer value of the concatenation of m1 and m2  
                            IF hour LESS THAN twenty four AND minute LESS THAN sixty  
                                APPEND the tuple of hour and minute TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the sorted form of times  
        END FUNCTION  
        
        SET all_times TO the result of calling generate_times with digits  
        SET current_time_index TO the position of the tuple of hour and minute in all_times  
        
        IF current_time_index PLUS one LESS THAN the length of all_times  
            SET next_hour AND next_minute TO the elements at the position current_time_index PLUS one of all_times  
        ELSE  
            SET next_hour AND next_minute TO the elements at the position zero of all_times  
        END IF  
        
        RETURN the string formed by zero-padding next_hour to two digits, followed by a colon character, followed by zero-padding next_minute to two digits  
    END FUNCTION  
END CLASS
