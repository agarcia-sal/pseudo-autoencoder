CLASS Solution  
    FUNCTION maxHeight(cuboids)  
        FOR each cuboid IN cuboids  
            SORT the dimensions of cuboid in ascending order  
        END FOR  
        SORT cuboids lexicographically  
        SET n TO the LENGTH OF cuboids  
        SET dp TO a list of zeroes of length n  
        FOR index i FROM zero TO n MINUS one  
            SET dp at position i TO the element at position two of cuboid at position i  
            FOR index j FROM zero TO i MINUS one  
                IF the element at position zero of cuboid at position j LESS THAN OR EQUAL TO the element at position zero of cuboid at position i AND  
                   the element at position one of cuboid at position j LESS THAN OR EQUAL TO the element at position one of cuboid at position i AND  
                   the element at position two of cuboid at position j LESS THAN OR EQUAL TO the element at position two of cuboid at position i  
                THEN  
                    SET dp at position i TO the GREATER BETWEEN dp at position i AND dp at position j PLUS the element at position two of cuboid at position i  
                END IF  
            END FOR  
        END FOR  
        RETURN the MAXIMUM VALUE FROM dp  
    END FUNCTION  
END CLASS
