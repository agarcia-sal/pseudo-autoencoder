CLASS Solution  
    FUNCTION maxPerformance(n, speed, efficiency, k)  
        SET engineers TO a list of pairs of elements from efficiency and speed at the same positions  
        SORT engineers in order of first element from highest to lowest  
        SET speed_heap TO an empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair consisting of eff and spd IN engineers  
            ADD spd TO speed_heap using a method to maintain smallest element order  
            INCREMENT speed_sum BY spd  
            IF the LENGTH OF speed_heap is greater than k  
                REMOVE the smallest element FROM speed_heap and DECREMENT speed_sum BY that element  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance is greater than max_performance  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        RETURN max_performance modulo one billion plus seven  
    END FUNCTION  
END CLASS
