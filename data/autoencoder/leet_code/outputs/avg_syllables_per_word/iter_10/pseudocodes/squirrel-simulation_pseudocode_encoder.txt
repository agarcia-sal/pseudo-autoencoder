CLASS Solution  
    FUNCTION minDistance(height, width, tree, squirrel, nuts)  
        FUNCTION distance(p1, p2)  
            RETURN absolute value of element at position zero of p1 MINUS element at position zero of p2 PLUS absolute value of element at position one of p1 MINUS element at position one of p2  
        END FUNCTION  
        
        SET total_distance TO zero  
        SET max_save TO negative infinity  
        
        FOR each nut IN nuts  
            SET total_distance TO total_distance PLUS two MULTIPLIED BY the result of calling distance with arguments nut and tree  
            SET first_nut_distance TO the result of calling distance with arguments squirrel and nut PLUS the result of calling distance with arguments nut and tree  
            SET later_nut_distance TO two MULTIPLIED BY the result of calling distance with arguments nut and tree  
            SET save TO later_nut_distance MINUS first_nut_distance  
            IF save GREATER THAN max_save  
                SET max_save TO save  
            END IF  
        END FOR  
        
        RETURN total_distance MINUS max_save  
    END FUNCTION  
END CLASS
