CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET it TO an iterator over t  
            RETURN the result of checking for each character c in s WHETHER c EXISTS IN it in sequence  
        END FUNCTION  

        SORT strs BY length of element MINUS ONE MULTIPLIED BY length PLUS element lexicographically ascending  

        FOR each index i AND element word IN strs  
            SET unique TO true  
            FOR each index j AND element other IN strs  
                IF i NOT EQUALS j AND is_subsequence(word, other) EQUALS true  
                    SET unique TO false  
                    BREAK inner loop  
                END IF  
            END FOR  
            IF unique EQUALS true  
                RETURN the length OF word  
            END IF  
        END FOR  

        RETURN negative one  
    END FUNCTION  
END CLASS
