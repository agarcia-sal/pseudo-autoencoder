CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        INITIALIZE dp AS a two-dimensional list with n PLUS one rows and n PLUS one columns FILLED WITH zero  
        
        FOR each j FROM zero TO n  
            SET element at position j of row zero of dp TO one  
        END FOR  
        
        FOR each i FROM one TO n  
            SET prefix_sum TO zero  
            IF element at position i MINUS one of s EQUALS the character I  
                FOR each j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position j of row i MINUS one of dp MODULO MOD  
                    SET element at position j of row i of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR each j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO prefix_sum PLUS element at position j PLUS one of row i MINUS one of dp MODULO MOD  
                    SET element at position j of row i of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO the SUM OF elements of row n of dp MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
