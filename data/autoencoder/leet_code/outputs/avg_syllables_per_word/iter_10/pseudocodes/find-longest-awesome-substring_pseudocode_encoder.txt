CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a new empty map with key zero mapped TO minus one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each position i AND character char IN s  
            CONVERT char TO its integer value AND STORE IN digit  
            SET prefix_xor TO prefix_xor EXCLUSIVE OR WITH one LEFT SHIFTED BY digit  
            
            IF prefix_xor EXISTS AS A KEY IN prefix_map  
                SET max_length TO the GREATER OF max_length AND i MINUS the value of prefix_map at prefix_xor  
            END IF  
            
            FOR each number j FROM zero TO nine  
                SET flipped TO prefix_xor EXCLUSIVE OR WITH one LEFT SHIFTED BY j  
                IF flipped EXISTS AS A KEY IN prefix_map  
                    SET max_length TO the GREATER OF max_length AND i MINUS the value of prefix_map at flipped  
                END IF  
            END FOR  
            
            IF prefix_xor DOES NOT EXIST AS A KEY IN prefix_map  
                SET the value of prefix_map at prefix_xor TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
