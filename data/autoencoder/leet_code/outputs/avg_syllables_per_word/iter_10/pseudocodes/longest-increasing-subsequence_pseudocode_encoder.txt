CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums EQUALS empty  
            RETURN zero  
        END IF  
        
        SET dp TO empty list  
        
        FOR each num IN nums  
            SET idx TO the position found by binary searching dp to insert num where elements are not less than num  
            
            IF idx EQUALS the LENGTH OF dp  
                APPEND num TO dp  
            ELSE  
                SET element at position idx of dp TO num  
            END IF  
        END FOR  
        
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
