CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO the frequency count of digits  
        SET total_sum TO the sum of digits  
        SET remainder TO total_sum MODULO three  
        IF remainder EQUALS one  
            FOR each d IN the list containing one, four, seven  
                IF count of d GREATER THAN zero  
                    DECREMENT count of d BY one  
                    BREAK out of the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each d IN the list containing two, five, eight  
                    IF count of d GREATER THAN one  
                        DECREMENT count of d BY two  
                        BREAK out of the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
        IF remainder EQUALS two  
            FOR each d IN the list containing two, five, eight  
                IF count of d GREATER THAN zero  
                    DECREMENT count of d BY one  
                    BREAK out of the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each d IN the list containing one, four, seven  
                    IF count of d GREATER THAN one  
                        DECREMENT count of d BY two  
                        BREAK out of the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
        SET result TO empty list  
        FOR d FROM nine TO zero DECREMENTED BY one  
            APPEND the string form of d repeated count of d times TO result  
        END FOR  
        SET final_number TO concatenation of all elements in result  
        IF final_number is not empty AND the first character of final_number EQUALS zero  
            RETURN zero as string  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
