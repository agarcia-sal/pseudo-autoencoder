CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        INITIALIZE radiant AS an empty double-ended queue  
        INITIALIZE dire AS an empty double-ended queue  
        
        FOR each index i AND character s IN senate  
            IF s EQUALS character R  
                APPEND i TO radiant  
            ELSE  
                APPEND i TO dire  
            END IF  
        END FOR  
        
        WHILE radiant IS NOT empty AND dire IS NOT empty  
            SET r_index TO the first element REMOVED FROM radiant  
            SET d_index TO the first element REMOVED FROM dire  
            
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO dire  
            END IF  
        END WHILE  
        
        IF radiant IS NOT empty  
            RETURN the string Radiant  
        ELSE  
            RETURN the string Dire  
        END IF  
    END FUNCTION  
END CLASS
