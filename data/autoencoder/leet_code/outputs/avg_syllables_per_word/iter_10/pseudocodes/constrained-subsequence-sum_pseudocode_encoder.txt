CLASS Solution  
    FUNCTION constrainedSubsetSum(nums, k)  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zeroes with LENGTH n  
        SET element at position zero of dp TO element at position zero of nums  
        SET max_sum TO element at position zero of nums  
        SET q TO a new deque containing the single element zero  
        FOR i FROM one TO n MINUS one  
            IF element at position at front of q in dp IS GREATER THAN zero  
                SET dp at position i TO element at position i of nums PLUS element at position at front of q in dp  
            ELSE  
                SET dp at position i TO element at position i of nums PLUS zero  
            END IF  
            IF max_sum IS LESS THAN dp at position i  
                SET max_sum TO dp at position i  
            END IF  
            WHILE q IS NOT empty AND dp at position i IS GREATER THAN OR EQUAL TO dp at position at last element of q  
                REMOVE last element FROM q  
            END WHILE  
            APPEND i TO q  
            IF element at position at front of q EQUALS i MINUS k  
                REMOVE first element FROM q  
            END IF  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
