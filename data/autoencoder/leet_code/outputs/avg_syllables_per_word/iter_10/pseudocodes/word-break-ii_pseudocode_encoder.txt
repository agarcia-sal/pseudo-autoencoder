CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO an empty set containing all elements of wordDict  
        SET memo TO an empty dictionary  
        
        FUNCTION backtrack(start)  
            IF start IS IN memo  
                RETURN the value at position start of memo  
            END IF  
            IF start EQUALS the length of s  
                RETURN a list containing the empty string  
            END IF  
            
            SET res TO an empty list  
            FOR end FROM start PLUS one TO the length of s PLUS one  
                SET word TO the substring of s from position start TO position end MINUS one  
                IF word IS IN wordSet  
                    FOR each sentence IN backtrack(end)  
                        IF sentence IS empty  
                            APPEND word TO res  
                        ELSE  
                            APPEND the concatenation of word, a space, and sentence TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            
            SET the value at position start of memo TO res  
            RETURN res  
        END FUNCTION  
        
        RETURN backtrack(zero)  
    END FUNCTION  
END CLASS
