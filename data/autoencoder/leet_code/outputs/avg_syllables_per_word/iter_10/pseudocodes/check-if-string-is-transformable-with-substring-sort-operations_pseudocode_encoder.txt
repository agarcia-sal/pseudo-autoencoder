CLASS Solution  
    FUNCTION isTransformable(s, t)  
        IMPORT defaultdict FROM collections  
        IMPORT heappop, heappush FROM heapq  
        
        SET pos TO a new defaultdict mapping to empty lists  
        
        FOR each position i AND character c IN s  
            SET digit TO integer value of character c  
            HEAPPUSH position i INTO pos at key digit  
        END FOR  
        
        FOR each position i AND character c IN t  
            SET digit TO integer value of character c  
            IF pos at key digit IS empty  
                RETURN false  
            END IF  
            FOR each smaller FROM zero TO digit MINUS one  
                IF pos at key smaller IS NOT empty AND element at position zero of pos at key smaller IS LESS THAN element at position zero of pos at key digit  
                    RETURN false  
                END IF  
            END FOR  
            HEAPPOP the element at key digit in pos  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
