CLASS Solution  
    FUNCTION shortestDistance(maze, start, destination)  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero of maze  
        SET directions TO the COLLECTION containing the PAIRS one and zero, MINUS one and zero, zero and one, zero and MINUS one  
        SET start TO the TUPLE of start  
        SET destination TO the TUPLE of destination  
        SET visited TO an EMPTY SET  
        SET pq TO a COLLECTION containing the PAIR zero and start  

        WHILE pq IS NOT empty  
            SET dist and position TO the ELEMENT REMOVED FROM pq WITH THE SMALLEST FIRST VALUE  
            SET x TO the FIRST element of position  
            SET y TO the SECOND element of position  

            IF x EQUALS the FIRST element of destination AND y EQUALS the SECOND element of destination  
                RETURN dist  
            END IF  

            IF position IS IN visited  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  

            ADD position TO visited  

            FOR each dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the ELEMENT at position new_x PLUS dx of maze at position new_y PLUS dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  

                IF the PAIR new_x and new_y IS NOT IN visited  
                    ADD the PAIR dist PLUS count and the PAIR new_x and new_y TO pq WITH PRIORITY BASED ON THE FIRST VALUE  
                END IF  
            END FOR  
        END WHILE  

        RETURN minus one  
    END FUNCTION  
END CLASS
