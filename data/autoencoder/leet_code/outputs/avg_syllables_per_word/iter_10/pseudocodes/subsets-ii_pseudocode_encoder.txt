CLASS Solution  
    FUNCTION subsetsWithDup(nums)  
        CALL sort ON nums  
        SET result TO empty list  
        SET subset TO empty list  
        
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO the LENGTH OF nums  
                APPEND a copy of subset TO result  
                RETURN  
            END IF  
            
            APPEND the element at position i of nums TO subset  
            CALL dfs WITH i PLUS one  
            
            REMOVE last element FROM subset  
            WHILE i PLUS one LESS THAN the LENGTH OF nums AND element at position i of nums EQUALS element at position i PLUS one of nums  
                INCREMENT i BY one  
            END WHILE  
            CALL dfs WITH i PLUS one  
        END FUNCTION  
        
        CALL dfs WITH zero  
        RETURN result  
    END FUNCTION  
END CLASS
