CLASS Solution  
    FUNCTION minWindow(s, t)  
        IMPORT Counter FROM collections  
        SET t_count TO Counter OF t  
        SET required TO the LENGTH OF t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO Counter  
        SET min_length TO positive infinity  
        SET min_start TO zero  
        
        WHILE right LESS THAN the LENGTH OF s  
            INCREMENT the COUNT of character at position right of s IN window_counts BY one  
            
            IF character at position right of s IS IN t_count AND the COUNT of character at position right of s IN window_counts EQUALS the COUNT of character at position right of s IN t_count  
                INCREMENT formed BY one  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO character at position left of s  
                
                IF right MINUS left PLUS one LESS THAN min_length  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                
                DECREMENT the COUNT of char IN window_counts BY one  
                IF char IS IN t_count AND the COUNT of char IN window_counts LESS THAN the COUNT of char IN t_count  
                    DECREMENT formed BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT right BY one  
        END WHILE  
        
        IF min_length EQUALS positive infinity  
            RETURN empty string  
        ELSE  
            RETURN the SUBSTRING of s FROM min_start TO min_start PLUS min_length  
        END IF  
    END FUNCTION  
END CLASS
