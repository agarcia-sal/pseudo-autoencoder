CLASS Solution  
    FUNCTION minDistance(houses, k)  
        CALL houses SORT  
        SET n TO the LENGTH OF houses  
        
        CALL cost INITIALIZE_AS two dimensional list OF SIZE n BY n WITH zero  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost AT position i AND position j TO cost AT position i PLUS one AND position j MINUS one PLUS element at position j OF houses MINUS element at position i OF houses  
            END FOR  
        END FOR  
        
        CALL dp INITIALIZE_AS two dimensional list OF SIZE n PLUS one BY k PLUS one WITH infinity  
        SET dp AT position zero AND position zero TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET dp AT position i AND position j TO the MINIMUM OF dp AT position i AND position j AND dp AT position p AND position j MINUS one PLUS cost AT position p AND position i MINUS one  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp AT position n AND position k  
    END FUNCTION  
END CLASS
