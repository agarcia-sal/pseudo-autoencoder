CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a list of LENGTH n FILLED WITH negative one  
        SET stack TO an empty list  
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO i MODULO n  
            WHILE stack IS NOT empty AND the element at position last element of stack in nums IS LESS THAN the element at position index in nums  
                SET popped_index TO the REMOVED last element FROM stack  
                SET the element at position popped_index in result TO the element at position index in nums  
            END WHILE  
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
