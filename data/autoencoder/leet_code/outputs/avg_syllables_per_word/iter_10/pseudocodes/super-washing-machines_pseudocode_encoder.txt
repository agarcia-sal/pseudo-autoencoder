CLASS Solution  
    FUNCTION findMinMoves(machines)  
        SET total_dresses TO the SUM OF machines  
        SET n TO the LENGTH OF machines  
        
        IF total_dresses MODULO n NOT EQUALS zero  
            RETURN negative one  
        END IF  
        
        SET target TO total_dresses DIVIDED BY n  
        SET left_sum TO zero  
        SET max_moves TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET left_excess TO left_sum MINUS target MULTIPLIED BY i  
            SET right_excess TO total_dresses MINUS left_sum MINUS the element at position i of machines MINUS target MULTIPLIED BY (n MINUS i MINUS one)  
            
            SET current_moves TO the MAXIMUM OF absolute value of left_excess, absolute value of right_excess, the element at position i of machines MINUS target  
            SET max_moves TO the MAXIMUM OF max_moves AND current_moves  
            
            INCREMENT left_sum BY the element at position i of machines  
        END FOR  
        
        RETURN max_moves  
    END FUNCTION  
END CLASS
