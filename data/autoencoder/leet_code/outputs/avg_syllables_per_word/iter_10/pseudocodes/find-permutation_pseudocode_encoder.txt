CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO list of numbers from one TO the LENGTH OF s PLUS one  
        SET start TO zero  
        FOR end FROM zero TO the LENGTH OF s  
            IF end EQUALS the LENGTH OF s OR the element at position end of s EQUALS the letter I  
                REVERSE the segment of result FROM position start TO position end PLUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
