CLASS Solution  
    FUNCTION nearestPalindromic(n)  
        SET length TO the LENGTH OF n  
        IF length EQUALS one  
            RETURN the STRING REPRESENTATION OF the INTEGER REPRESENTATION OF n MINUS one  
        END IF  
        SET candidates TO an EMPTY SET  
        ADD TEN RAISED TO THE POWER OF length PLUS one TO candidates  
        ADD TEN RAISED TO THE POWER OF length MINUS one MINUS one TO candidates  
        SET prefix TO the INTEGER REPRESENTATION OF the SUBSTRING OF n FROM position zero TO the INTEGER RESULT OF the INTEGER DIVISION OF length PLUS one BY two  
        FOR i FROM negative one TO two MINUS one  
            SET new_prefix TO the STRING REPRESENTATION OF prefix PLUS i  
            IF length MODULO two EQUALS zero  
                SET candidate TO new_prefix CONCATENATED WITH the REVERSE OF new_prefix  
            ELSE  
                SET candidate TO new_prefix CONCATENATED WITH the REVERSE OF new_prefix EXCLUDING the LAST CHARACTER  
            END IF  
            ADD the INTEGER REPRESENTATION OF candidate TO candidates  
        END FOR  
        SET original_number TO the INTEGER REPRESENTATION OF n  
        REMOVE original_number FROM candidates IF PRESENT  
        SET closest TO the ELEMENT OF candidates WITH the MINIMUM VALUE OF the TUPLE OF the ABSOLUTE VALUE OF the ELEMENT MINUS original_number AND the ELEMENT ITSELF  
        RETURN the STRING REPRESENTATION OF closest  
    END FUNCTION  
END CLASS
