CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        
        CALL initializeDPTable WITH m PLUS one AND n PLUS one RETURNING dp  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO element at position i MINUS one of nums1 MULTIPLIED BY element at position j MINUS one of nums2  
                
                SET dp at position i and j TO the maximum of  
                    current_product,  
                    element at position i MINUS one and j of dp,  
                    element at position i and j MINUS one of dp,  
                    element at position i MINUS one and j MINUS one of dp PLUS current_product  
            END FOR  
        END FOR  
        
        RETURN element at position m and n of dp  
    END FUNCTION  
  
    FUNCTION initializeDPTable(rows, columns)  
        SET table TO empty list  
        FOR each index FROM zero TO rows MINUS one  
            SET row TO empty list  
            FOR each inner_index FROM zero TO columns MINUS one  
                APPEND negative infinity TO row  
            END FOR  
            APPEND row TO table  
        END FOR  
        RETURN table  
    END FUNCTION  
END CLASS
