CLASS Solution  
    FUNCTION findShortestWay(maze, ball, hole)  
        SET directions TO list of tuples: zero and one and character r, one and zero and character d, zero and negative one and character l, negative one and zero and character u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the first element of maze  
        SET start TO tuple conversion of ball  
        SET hole TO tuple conversion of hole  
        SET pq TO list containing a tuple of zero, empty string, and start  
        SET visited TO empty set  
        
        WHILE pq IS NOT empty  
            SET dist, path, (x, y) TO the result of removing the smallest element from pq  
            
            IF x EQUALS the first element of hole AND y EQUALS the second element of hole  
                RETURN path  
            END IF  
            
            IF (x, y) IS IN visited  
                CONTINUE to next iteration of the WHILE loop  
            END IF  
            
            ADD (x, y) TO visited  
            
            FOR each dx, dy, direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the element at position new_x PLUS dx of maze at position new_y PLUS dy EQUALS zero  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY one  
                    IF new_x EQUALS the first element of hole AND new_y EQUALS the second element of hole  
                        BREAK the WHILE loop  
                    END IF  
                END WHILE  
                
                IF (new_x, new_y) IS NOT IN visited  
                    ADD tuple of dist PLUS count, concatenation of path and direction, and tuple of new_x and new_y TO pq maintaining smallest element removal property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string "impossible"  
    END FUNCTION  
END CLASS
