CLASS Solution  
    FUNCTION countPrimes WITH PARAMETER n  
        IF n LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
        SET is_prime TO a list of TRUE values with the length of n  
        SET the element at position zero of is_prime TO FALSE  
        SET the element at position one of is_prime TO FALSE  
        FOR start FROM two TO the INTEGER PART OF the square root of n PLUS one  
            IF the element at position start of is_prime EQUALS TRUE  
                FOR multiple FROM start MULTIPLIED BY start TO n MINUS one STEP start  
                    SET the element at position multiple of is_prime TO FALSE  
                END FOR  
            END IF  
        END FOR  
        SET prime_count TO zero  
        FOR each element IN is_prime  
            IF element EQUALS TRUE  
                INCREMENT prime_count BY one  
            END IF  
        END FOR  
        RETURN prime_count  
    END FUNCTION  
END CLASS
