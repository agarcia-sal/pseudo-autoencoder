CLASS Solution  
    FUNCTION countArrangement(n)  
        FUNCTION backtrack(used, index)  
            IF index EQUALS n PLUS one  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR i FROM one TO n  
                IF NOT the element at position i of used AND (i modulo index EQUALS zero OR index modulo i EQUALS zero)  
                    SET the element at position i of used TO True  
                    INCREMENT count BY backtrack(used, index PLUS one)  
                    SET the element at position i of used TO False  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET used TO a list of False of length n PLUS one  
        RETURN backtrack(used, one)  
    END FUNCTION  
END CLASS
