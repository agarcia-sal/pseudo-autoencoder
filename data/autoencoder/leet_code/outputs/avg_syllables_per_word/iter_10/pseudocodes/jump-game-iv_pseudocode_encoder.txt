CLASS Solution
    FUNCTION minJumps(arr)
        SET n TO the LENGTH OF arr
        IF n EQUALS one
            RETURN zero
        END IF
        
        SET value_indices TO an empty default dictionary mapping to list
        FOR each i AND value IN enumeration of arr
            APPEND i TO the list at key value IN value_indices
        END FOR
        
        SET queue TO a new deque containing the tuple of zero AND zero
        SET visited TO a new set containing zero
        
        WHILE queue IS NOT empty
            SET current_index AND steps TO the first element REMOVED FROM queue
            
            IF current_index EQUALS n MINUS one
                RETURN steps
            END IF
            
            SET next_indices TO a new list containing current_index PLUS one, current_index MINUS one
            APPEND all elements from the list at key arr[current_index] IN value_indices TO next_indices
            
            FOR each next_index IN next_indices
                IF next_index IS GREATER THAN OR EQUAL TO zero AND next_index IS LESS THAN n AND next_index NOT IN visited
                    ADD next_index TO visited
                    APPEND the tuple of next_index AND steps PLUS one TO queue
                END IF
            END FOR
            
            SET the list at key arr[current_index] IN value_indices TO an empty list
        END WHILE
        
        RETURN negative one
    END FUNCTION
END CLASS
