CLASS ListNode  
    FUNCTION __init__ with parameters val EQUALS zero, next EQUALS None  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode constructed with the first element of values  
    SET p TO head  
    FOR each val IN values starting from the second element to the end  
        SET node TO new ListNode constructed with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1, p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the condition that element val of p1 EQUALS element val of p2 AND the result of is_same_list called with parameters element next of p1, element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists with parameter lists  
        SET min_heap TO empty list  
        FOR each index i AND element lst IN lists  
            IF lst IS NOT None  
                CALL heap push operation with parameters min_heap AND a tuple of element val of lst, i, AND lst  
            END IF  
        END FOR  
        SET dummy TO new ListNode constructed with default parameters  
        SET current TO dummy  
        WHILE min_heap IS NOT empty  
            CALL heap pop operation on min_heap AND UNPACK the result INTO val, i, node  
            SET element next of current TO node  
            SET current TO element next of current  
            IF element next of node IS NOT None  
                CALL heap push operation with parameters min_heap AND a tuple of element val of element next of node, i, AND element next of node  
            END IF  
        END WHILE  
        RETURN element next of dummy  
    END FUNCTION  
END CLASS
