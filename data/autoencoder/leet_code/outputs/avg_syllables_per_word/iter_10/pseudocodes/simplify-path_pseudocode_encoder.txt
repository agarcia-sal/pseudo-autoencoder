CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the RESULT OF splitting path by the character slash  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component EQUALS an empty string OR component EQUALS a period  
                CONTINUE to the next iteration  
            ELSE IF component EQUALS two periods  
                IF the LENGTH OF stack GREATER THAN zero  
                    REMOVE last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO the concatenation of a slash and the joining of elements in stack separated by a slash  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
