CLASS Solution  
    FUNCTION minimumTotal(triangle)  
        FOR row FROM the LENGTH OF triangle MINUS two DOWN TO zero  
            FOR col FROM zero TO the LENGTH OF the element at position row of triangle MINUS one  
                SET element at position col of the element at position row of triangle TO the element at position col of the element at position row of triangle PLUS the smaller of the element at position col of the element at position row PLUS one of triangle AND the element at position col PLUS one of the element at position row PLUS one of triangle  
            END FOR  
        END FOR  
        RETURN the element at position zero of the element at position zero of triangle  
    END FUNCTION  
END CLASS
