CLASS Solution  
    FUNCTION minStickers(stickers, target)  
        SET stickers TO List formed by CONVERTING each sticker IN stickers TO a frequency map  
        SORT stickers BY the negative of the maximum character frequency in each sticker  
        
        SET queue TO a double-ended queue initialized with a pair of target string AND zero  
        SET memo TO a mapping from target string TO zero  
        
        WHILE queue is not empty  
            REMOVE first element FROM queue AND DESTRUCTURE it TO current_target AND used_stickers  
            
            FOR each sticker IN stickers  
                IF first character of current_target IS NOT IN sticker THEN  
                    CONTINUE TO next sticker  
                END IF  
                
                SET remaining TO frequency map of current_target  
                SUBTRACT frequencies of sticker FROM remaining  
                
                SET new_target TO string formed by joining sorted elements from remaining  
                
                IF new_target IS empty string THEN  
                    RETURN used_stickers PLUS one  
                END IF  
                
                IF new_target IS NOT IN memo THEN  
                    SET memo new_target TO used_stickers PLUS one  
                    APPEND pair of new_target AND used_stickers PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
