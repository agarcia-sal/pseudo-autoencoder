CLASS Solution  
    FUNCTION canDivideIntoSubsequences(nums, k)  
        SET freq TO frequency count of elements in nums  
        SET max_freq TO the maximum value in the collection of frequencies in freq  
        SET length_divided_by_k TO the length of nums DIVIDED BY k  
        IF length_divided_by_k GREATER THAN OR EQUAL TO max_freq  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
