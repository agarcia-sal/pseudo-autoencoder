CLASS TreeNode  
    FUNCTION __init__ with parameters val EQUALS zero, left EQUALS none, right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO create new TreeNode with parameter value at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO remove first element FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO create new TreeNode with parameter value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO create new TreeNode with parameter value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF value at p DOES NOT EQUAL value at q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(value at left of p, value at left of q) AND is_same_tree(value at right of p, value at right of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree with parameters self, root  
        FUNCTION subtree_sum with parameter node  
            IF node IS none  
                RETURN zero  
            END IF  
            RETURN value at node PLUS subtree_sum(value at left of node) PLUS subtree_sum(value at right of node)  
        END FUNCTION  

        SET total_sum TO subtree_sum(root)  

        IF total_sum MODULO two DOES NOT EQUAL zero  
            RETURN False  
        END IF  

        SET target TO total_sum DIVIDED BY two  
        SET found TO a list containing False  

        FUNCTION can_split with parameter node  
            IF node IS none  
                RETURN zero  
            END IF  
            SET current_sum TO value at node PLUS can_split(value at left of node) PLUS can_split(value at right of node)  
            IF current_sum EQUALS target AND node DOES NOT EQUAL root  
                SET element at position zero of found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  

        CALL can_split with parameter root  
        RETURN element at position zero of found  
    END FUNCTION  
END CLASS
