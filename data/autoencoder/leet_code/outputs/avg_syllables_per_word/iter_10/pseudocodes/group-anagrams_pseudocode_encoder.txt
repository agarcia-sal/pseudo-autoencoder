CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        CREATE anagrams AS a new default dictionary with list as default value  
        FOR each s IN strs  
            SET sorted_str TO the concatenation of characters obtained by sorting characters of s in ascending order  
            APPEND s TO the list in anagrams at key sorted_str  
        END FOR  
        SET result TO the list of values from anagrams  
        RETURN result  
    END FUNCTION  
END CLASS
