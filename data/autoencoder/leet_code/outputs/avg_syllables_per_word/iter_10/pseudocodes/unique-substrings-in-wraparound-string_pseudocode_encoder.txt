CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a dictionary with keys as characters from 'a' to 'z' and values set to zero  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND (the ordinal value of character at position index of s MINUS the ordinal value of character at position index MINUS one of s EQUALS one OR character at position index of s EQUALS 'a' AND character at position index MINUS one of s EQUALS 'z')  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET value AT key character at position index of s IN max_length TO the greater value BETWEEN value AT key character at position index of s IN max_length AND current_length  
        END FOR  
        SET total_sum TO the sum OF all values IN max_length  
        RETURN total_sum  
    END FUNCTION  
END CLASS
