CLASS Solution
    FUNCTION reverseWords(s)
        CALL reverse the entire list s
        SET start TO zero
        FOR end FROM zero TO the LENGTH OF s MINUS one
            IF element at position end of s EQUALS a space character
                CALL reverse the sublist of s from start TO end MINUS one
                SET start TO end PLUS one
            END IF
        END FOR
        CALL reverse the sublist of s from start TO the last element
    END FUNCTION
END CLASS
