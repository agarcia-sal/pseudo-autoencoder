CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO an empty list  
        FOR index FROM zero TO n MINUS one  
            APPEND an empty list TO graph  
        END FOR  
        FOR each edge IN edges  
            SET x TO the first element of edge  
            SET y TO the second element of edge  
            SET time TO the third element of edge  
            APPEND a tuple of y and time TO the element at position x of graph  
            APPEND a tuple of x and time TO the element at position y of graph  
        END FOR  
        SET pq TO a list containing a tuple of the element at position zero of passingFees, zero, zero  
        SET visited_time TO a list of infinity repeated n times  
        SET the element at position zero of visited_time TO zero  
        WHILE pq is not empty  
            REMOVE first element FROM pq AS current_cost, current_city, current_time  
            IF current_city EQUALS n MINUS one  
                RETURN current_cost  
            END IF  
            FOR each neighbor, travel_time IN the element at position current_city of graph  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS the element at position neighbor of passingFees  
                IF new_time LESS THAN OR EQUAL TO maxTime AND (new_time LESS THAN the element at position neighbor of visited_time OR new_cost LESS THAN current_cost)  
                    SET the element at position neighbor of visited_time TO new_time  
                    APPEND the tuple of new_cost, neighbor, new_time TO pq  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
