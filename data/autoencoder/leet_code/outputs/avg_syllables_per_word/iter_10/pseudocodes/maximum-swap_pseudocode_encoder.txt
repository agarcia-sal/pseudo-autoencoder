CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO list of string representations of num  
        SET last TO empty dictionary  
        FOR index FROM zero TO length of digits MINUS one  
            SET element at position index of digits TO digit_char  
            SET last at integer value of digit_char TO index  
        END FOR  
        FOR index FROM zero TO length of digits MINUS one  
            SET element at position index of digits TO digit_char  
            FOR d2 FROM nine DOWN TO integer value of digit_char PLUS one  
                IF last at d2 EXISTS AND last at d2 GREATER THAN index  
                    SET temp TO element at position index of digits  
                    SET element at position index of digits TO element at position last at d2 of digits  
                    SET element at position last at d2 of digits TO temp  
                    SET result_string TO concatenation of all elements of digits  
                    SET result_number TO integer value of result_string  
                    RETURN result_number  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
