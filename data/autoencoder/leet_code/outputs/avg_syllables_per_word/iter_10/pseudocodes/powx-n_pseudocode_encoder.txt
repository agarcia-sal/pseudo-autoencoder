CLASS Solution  
    FUNCTION myPow(x, n)  
        FUNCTION qpow(a, n)  
            SET ans TO one  
            WHILE n NOT EQUALS zero  
                IF n BITWISE AND one NOT EQUALS zero  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two USING INTEGER DIVISION  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n GREATER THAN OR EQUAL TO zero  
            RETURN qpow(x, n)  
        ELSE  
            RETURN one DIVIDED BY qpow(x, n MULTIPLIED BY minus one)  
        END IF  
    END FUNCTION  
END CLASS
