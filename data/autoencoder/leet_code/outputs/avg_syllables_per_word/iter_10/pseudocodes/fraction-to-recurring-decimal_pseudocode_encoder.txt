CLASS Solution  
    FUNCTION fractionToDecimal(numerator, denominator)  
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero  
            SET sign TO minus  
        ELSE  
            SET sign TO empty string  
        END IF  
  
        SET numerator TO absolute value of numerator  
        SET denominator TO absolute value of denominator  
  
        SET quotient TO numerator DIVIDED BY denominator  
        SET remainder TO numerator MODULO denominator  
        SET integer_part TO string representation of quotient  
  
        IF remainder EQUALS zero  
            RETURN concatenation of sign and integer_part  
        END IF  
  
        SET seen_remainders TO empty dictionary  
        SET fractional_part TO empty list  
  
        WHILE remainder NOT EQUALS zero  
            IF remainder IS IN seen_remainders  
                SET start TO value in seen_remainders for remainder  
                SET non_repeating TO concatenation of elements in fractional_part up to start position  
                SET repeating TO concatenation of elements in fractional_part from start position onward  
                RETURN concatenation of sign, integer_part, period, non_repeating, open parenthesis, repeating, close parenthesis  
            END IF  
  
            SET value in seen_remainders for remainder TO length of fractional_part  
  
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO remainder DIVIDED BY denominator  
            SET remainder TO remainder MODULO denominator  
            APPEND string representation of quotient TO fractional_part  
        END WHILE  
  
        RETURN concatenation of sign, integer_part, period, concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
