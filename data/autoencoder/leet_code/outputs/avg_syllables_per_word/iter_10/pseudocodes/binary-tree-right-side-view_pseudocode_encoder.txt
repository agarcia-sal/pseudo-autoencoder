CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET val TO parameter val  
        SET left TO parameter left  
        SET right TO parameter right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO remove first element FROM queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF val of p NOT EQUALS val of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters val left of p, val left of q AND is_same_tree with parameters val right of p, val right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView with parameter root  
        IF root EQUALS None  
            RETURN empty list  
        END IF  
        SET right_side_view TO empty list  
        SET queue TO a double-ended queue initialized with root  
        WHILE queue IS NOT empty  
            SET level_length TO length of queue  
            FOR index FROM zero TO level_length MINUS one  
                SET node TO remove first element FROM queue  
                IF index EQUALS level_length MINUS one  
                    APPEND val of node TO right_side_view  
                END IF  
                IF left of node NOT EQUALS None  
                    APPEND left of node TO queue  
                END IF  
                IF right of node NOT EQUALS None  
                    APPEND right of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
