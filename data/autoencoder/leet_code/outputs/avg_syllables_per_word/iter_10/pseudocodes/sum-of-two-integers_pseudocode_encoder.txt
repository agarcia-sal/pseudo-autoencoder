CLASS Solution  
    FUNCTION getSum(a, b)  
        SET MAX TO the hexadecimal value seven FFFFFF F  
        SET mask TO the hexadecimal value FFFFFFFF  
        WHILE b NOT EQUALS zero  
            SET carry TO the result of a AND b MULTIPLIED BY two  
            SET a TO the result of a XOR b AND mask  
            SET b TO the result of carry AND mask  
        END WHILE  
        IF a LESS THAN OR EQUAL TO MAX  
            RETURN a  
        ELSE  
            RETURN the bitwise negation of the result of a XOR mask  
        END IF  
    END FUNCTION  
END CLASS
