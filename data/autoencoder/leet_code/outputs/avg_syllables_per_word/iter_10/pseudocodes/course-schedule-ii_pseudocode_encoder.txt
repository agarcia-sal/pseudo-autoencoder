CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO the result of CALLING createAdjacencyList WITH numCourses  
        SET in_degrees TO the result of CALLING createInDegreesList WITH numCourses  
        
        FOR each pair dest, src IN prerequisites  
            APPEND dest TO the list at position src IN adj_list  
            INCREMENT the element at position dest IN in_degrees BY one  
        END FOR  
        
        SET queue TO the result of CALLING createQueue WITH numCourses AND in_degrees  
        SET topological_order TO an empty list  
        
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET current TO the result of REMOVING the first element FROM queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN the list at position current IN adj_list  
                DECREMENT the element at position neighbor IN in_degrees BY one  
                IF the element at position neighbor IN in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
    
    FUNCTION createAdjacencyList(numCourses)  
        SET adjacency_list TO an empty map  
        FOR index FROM zero TO numCourses MINUS one  
            SET the element at position index IN adjacency_list TO an empty list  
        END FOR  
        RETURN adjacency_list  
    END FUNCTION  
    
    FUNCTION createInDegreesList(numCourses)  
        SET in_degrees_list TO a list of zeros with length numCourses  
        RETURN in_degrees_list  
    END FUNCTION  
    
    FUNCTION createQueue(numCourses, in_degrees)  
        SET new_queue TO an empty list  
        FOR index FROM zero TO numCourses MINUS one  
            IF the element at position index IN in_degrees EQUALS zero  
                APPEND index TO new_queue  
            END IF  
        END FOR  
        RETURN new_queue  
    END FUNCTION  
END CLASS
