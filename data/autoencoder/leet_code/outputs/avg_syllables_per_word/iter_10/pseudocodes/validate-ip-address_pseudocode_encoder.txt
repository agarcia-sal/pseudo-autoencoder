CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS ONLY OF DIGITS  
                SET num TO CONVERT block TO INTEGER  
                IF num LESS THAN OR EQUAL TO two hundred fifty five AND CONVERT num TO STRING EQUALS block  
                    RETURN True  
                ELSE  
                    RETURN False  
                END IF  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF LENGTH OF block GREATER THAN OR EQUAL TO one AND LENGTH OF block LESS THAN OR EQUAL TO four  
                TRY  
                    CONVERT block TO INTEGER USING BASE sixteen  
                    RETURN True  
                CATCH ValueError  
                    RETURN False  
                END TRY  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO SPLIT queryIP BY the character DOT  
        SET parts_ipv6 TO SPLIT queryIP BY the character COLON  

        IF LENGTH OF parts_ipv4 EQUALS four AND FOR each part IN parts_ipv4 IS_VALID_IPV4_BLOCK(part) IS True FOR ALL parts  
            RETURN the string "IPv4"  
        ELSE IF LENGTH OF parts_ipv6 EQUALS eight AND FOR each part IN parts_ipv6 IS_VALID_IPV6_BLOCK(part) IS True FOR ALL parts  
            RETURN the string "IPv6"  
        ELSE  
            RETURN the string "Neither"  
        END IF  
    END FUNCTION  
END CLASS
