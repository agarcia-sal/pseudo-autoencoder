CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO set containing zero and firstPerson  
        SORT meetings BY element at position two of each meeting  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO element at position two of meeting at position i of meetings  
            SET graph TO a new dictionary with default value as list  
            WHILE i LESS THAN the LENGTH OF meetings AND element at position two of meeting at position i of meetings EQUALS current_time  
                SET x TO element at position zero of meeting at position i of meetings  
                SET y TO element at position one of meeting at position i of meetings  
                APPEND y TO list at key x IN graph  
                APPEND x TO list at key y IN graph  
                INCREMENT i BY one  
            END WHILE  
            SET queue TO a new deque containing each person IN know_secret WHERE person IS IN keys of graph  
            SET visited TO empty set  
            WHILE LENGTH OF queue GREATER THAN zero  
                REMOVE first element FROM queue AND SET to person  
                IF person IN visited  
                    CONTINUE to next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN list at key person IN graph  
                    IF neighbor NOT IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN list created from know_secret  
    END FUNCTION  
END CLASS
