CLASS Solution  
    FUNCTION numMatchingSubseq(s, words)  
        SET waiting TO an empty mapping from characters to empty double-ended queues for each unique character in s  
        FOR each word IN words  
            SET it TO an iterator over word  
            SET first_letter TO the next element of it  
            IF first_letter IS IN waiting  
                APPEND it TO the double-ended queue mapped by first_letter in waiting  
            END IF  
        END FOR  
        
        SET match_count TO zero  
        
        FOR each character c IN s  
            IF c IS IN waiting  
                SET current_iterators TO the double-ended queue mapped by c in waiting  
                SET the double-ended queue mapped by c in waiting TO an empty double-ended queue  
                FOR each iterator it IN current_iterators  
                    SET next_char TO the next element of it OR none if exhausted  
                    IF next_char IS none  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char IS IN waiting  
                            APPEND it TO the double-ended queue mapped by next_char in waiting  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN match_count  
    END FUNCTION  
END CLASS
