CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulting to zero, left defaulting to none, right defaulting to none  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with parameter element at position zero of values  
    SET i TO one  
    SET queue TO new double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode with parameter element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode with parameter element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with parameters the left attribute of p, the left attribute of q AND the result of is_same_tree with parameters the right attribute of p, the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue with parameter root  
        SET queue TO new double-ended queue containing root  
        SET leftmost_value TO none  
        WHILE queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one  
                SET node TO REMOVE first element FROM queue  
                IF i EQUALS zero  
                    SET leftmost_value TO the val attribute of node  
                END IF  
                IF the left attribute of node IS NOT none  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node IS NOT none  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
