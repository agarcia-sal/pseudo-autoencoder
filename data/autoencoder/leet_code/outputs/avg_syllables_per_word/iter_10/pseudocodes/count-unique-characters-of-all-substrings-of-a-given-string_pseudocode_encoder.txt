CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET last TO a list of twenty six elements each SET TO minus one  
        SET last2 TO a list of twenty six elements each SET TO minus one  
        SET result TO zero  
        FOR each index i AND character char IN s  
            SET idx TO the UNICODE of char MINUS the UNICODE of capital A  
            SET contribution_first TO i MINUS the element at position idx of last  
            SET contribution_second TO the element at position idx of last MINUS the element at position idx of last2  
            SET result TO result PLUS contribution_first MULTIPLIED BY contribution_second  
            SET result TO result MODULO MOD  
            SET the element at position idx of last2 TO the element at position idx of last  
            SET the element at position idx of last TO i  
        END FOR  
        FOR index i FROM zero TO twenty five  
            SET contribution_first TO n MINUS the element at position i of last  
            SET contribution_second TO the element at position i of last MINUS the element at position i of last2  
            SET result TO result PLUS contribution_first MULTIPLIED BY contribution_second  
            SET result TO result MODULO MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
