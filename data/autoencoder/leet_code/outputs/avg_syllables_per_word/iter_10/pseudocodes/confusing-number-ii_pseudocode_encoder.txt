CLASS Solution  
    FUNCTION confusingNumberII(n)  
        SET rotate_map TO mapping zero maps to zero, one maps to one, six maps to nine, eight maps to eight, nine maps to six  
        SET valid_digits TO list of zero, one, six, eight, nine  
        
        FUNCTION is_confusing(num)  
            SET original_num TO num  
            SET rotated_num TO zero  
            WHILE num GREATER THAN zero  
                SET digit TO remainder of num DIVIDED BY ten  
                SET rotated_num TO rotated_num MULTIPLIED BY ten PLUS the value mapped from digit in rotate_map  
                SET num TO integer division of num BY ten  
            END WHILE  
            IF rotated_num NOT EQUALS original_num  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        FUNCTION count_confusing_numbers(limit, current)  
            IF current GREATER THAN limit  
                RETURN zero  
            END IF  
            IF current NOT EQUALS zero AND is_confusing(current) IS true  
                SET count TO one  
            ELSE  
                SET count TO zero  
            END IF  
            FOR each digit IN valid_digits  
                SET new_number TO current MULTIPLIED BY ten PLUS digit  
                IF new_number EQUALS zero  
                    CONTINUE TO next digit  
                END IF  
                SET count TO count PLUS count_confusing_numbers(limit, new_number)  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN count_confusing_numbers(n, zero)  
    END FUNCTION  
END CLASS
