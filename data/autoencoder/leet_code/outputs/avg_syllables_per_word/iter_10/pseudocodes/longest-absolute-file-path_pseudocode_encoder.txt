CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO the result of splitting input BY newline  
        SET stack TO empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO the number of tab characters in line  
            SET name TO the substring of line starting at position depth plus one  
            
            WHILE the LENGTH OF stack GREATER THAN depth  
                REMOVE last element FROM stack  
            END WHILE  
            
            IF stack IS NOT empty  
                SET current_length TO the last element in stack PLUS the LENGTH OF name PLUS one  
            ELSE  
                SET current_length TO the LENGTH OF name  
            END IF  
            
            IF name CONTAINS the character period  
                SET max_length TO the greater value BETWEEN max_length AND current_length  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
