CLASS Solution  
    FUNCTION minMoves(nums, k)  
        SET positions TO empty list  
        FOR each index i AND element num IN enumeration of nums  
            IF num EQUALS one  
                APPEND i TO positions  
            END IF  
        END FOR  

        FUNCTION calculate_cost(start, end)  
            SET mid TO start PLUS end DIVIDED BY two  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR index i FROM start TO end MINUS one  
                SET cost TO cost PLUS absolute value of element at position i of positions MINUS (median MINUS (mid MINUS i))  
            END FOR  
            RETURN cost  
        END FUNCTION  

        SET min_cost TO infinity  
        FOR index i FROM zero TO LENGTH OF positions MINUS k  
            SET min_cost TO minimum of min_cost AND calculate_cost(i, i PLUS k)  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
