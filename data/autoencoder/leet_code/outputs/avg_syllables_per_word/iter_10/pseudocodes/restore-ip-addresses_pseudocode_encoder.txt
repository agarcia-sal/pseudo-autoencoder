CLASS Solution  
    FUNCTION restoreIpAddresses with parameter s  
        FUNCTION is_valid with parameter segment  
            IF the LENGTH OF segment EQUALS one OR (the element at position zero of segment NOT EQUALS zero AND the INTEGER VALUE OF segment LESS THAN OR EQUAL TO two hundred and fifty-five)  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        FUNCTION backtrack with parameters start and path  
            IF the LENGTH OF path EQUALS four  
                IF start EQUALS the LENGTH OF s  
                    APPEND the STRING JOINED BY period FROM path TO result  
                END IF  
                RETURN  
            END IF  
            FOR length FROM one TO three  
                IF start PLUS length LESS THAN OR EQUAL TO the LENGTH OF s  
                    SET segment TO the SUBSTRING OF s FROM start UP TO start PLUS length  
                    IF is_valid with segment EQUALS true  
                        CALL backtrack with start PLUS length and path PLUS a LIST CONTAINING segment  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  

        SET result TO an empty list  
        CALL backtrack with zero and an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
