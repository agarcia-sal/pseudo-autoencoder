CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        FOR each char IN s  
            IF char IS digit  
                SET d TO d PLUS char  
            ELSE IF char IS alphabet letter  
                SET a TO a PLUS char  
            ELSE IF char IS character equal to left bracket  
                APPEND tuple of a AND integer value of d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char IS character equal to right bracket  
                SET p AND n TO last element popped FROM stack  
                SET a TO p PLUS a REPEATED multiplied BY n  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
