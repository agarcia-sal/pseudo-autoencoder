CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO TreeNode with value equal to element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE first element FROM queue AND SET TO node  
        IF i LESS THAN length of values AND element at position i of values IS NOT None  
            SET node.left TO TreeNode with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values IS NOT None  
            SET node.right TO TreeNode with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF preorder IS empty OR inorder IS empty  
            RETURN None  
        END IF  
        SET root_val TO element at position zero of preorder  
        SET root TO TreeNode with value root_val  
        SET root_index_inorder TO position of root_val in inorder  
        SET root.left TO self.buildTree(sublist of preorder from position one TO root_index_inorder PLUS one, sublist of inorder from position zero TO root_index_inorder)  
        SET root.right TO self.buildTree(sublist of preorder from position root_index_inorder PLUS one TO end, sublist of inorder from position root_index_inorder PLUS one TO end)  
        RETURN root  
    END FUNCTION  
END CLASS
