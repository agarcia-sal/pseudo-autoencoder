CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO split the string num by the plus sign  
            SET real TO convert to integer the element at position zero of parts  
            SET imaginary TO convert to integer the element at position one of parts excluding the last character  
            RETURN real, imaginary  
        END FUNCTION  
  
        SET real1, imag1 TO the result of calling parse_complex with num1  
        SET real2, imag2 TO the result of calling parse_complex with num2  
  
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
  
        RETURN the string concatenation of real_part, plus sign, imaginary_part, and the character i  
    END FUNCTION  
END CLASS
