CLASS Solution  
    FUNCTION spiralMatrixIII(rows, cols, rStart, cStart)  
        SET directions TO a list containing the pairs zero and one, one and zero, zero and negative one, negative one and zero  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO an empty list  
        SET x TO rStart  
        SET y TO cStart  
        
        WHILE the length of visited is less than rows MULTIPLIED BY cols  
            IF increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR each number FROM one TO steps  
                IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN rows AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN cols  
                    APPEND a list containing x and y TO visited  
                END IF  
                SET dx TO element at position direction_index of directions at position zero  
                SET dy TO element at position direction_index of directions at position one  
                SET x TO x PLUS dx  
                SET y TO y PLUS dy  
            END FOR  
            SET direction_index TO (direction_index PLUS one) MODULO four  
            INCREMENT increment BY one  
        END WHILE  
        
        RETURN visited  
    END FUNCTION  
END CLASS
