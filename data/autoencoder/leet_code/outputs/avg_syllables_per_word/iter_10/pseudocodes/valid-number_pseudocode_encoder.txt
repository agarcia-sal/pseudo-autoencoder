CLASS Solution  
    FUNCTION isNumber(s)  
        SET number_regex TO the string starting with caret plus or minus zero or one time followed by open parenthesis open parenthesis digit one or more times followed by open parenthesis dot digit zero or more times followed by close parenthesis zero or one time close parenthesis or open parenthesis dot digit one or more times close parenthesis close parenthesis followed by open parenthesis e plus or minus zero or one time digit one or more times close parenthesis zero or one time followed by dollar sign  
        IF the full match of number_regex with s is not equal to none  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
