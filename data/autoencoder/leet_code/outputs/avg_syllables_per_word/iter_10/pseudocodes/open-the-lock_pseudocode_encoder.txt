CLASS Solution  
    FUNCTION openLock(deadends, target)  
        SET deadends_set TO a set containing all elements of deadends  
        
        IF string "0000" IS IN deadends_set  
            RETURN negative one  
        END IF  
        
        IF target EQUALS string "0000"  
            RETURN zero  
        END IF  
        
        SET queue TO a deque initialized with a tuple of string "0000" and zero  
        SET visited TO a set containing string "0000"  
        
        FUNCTION neighbors(state)  
            SET result TO an empty list  
            FOR index FROM zero TO three  
                SET digit TO the integer value of character at position index of state  
                FOR move IN the list containing negative one and one  
                    SET new_digit TO the sum of digit and move modulo ten  
                    SET new_state TO the concatenation of the substring of state from start to position index exclusive, the string representation of new_digit, and the substring of state from position index plus one to end  
                    APPEND new_state TO result  
                END FOR  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        WHILE the length of queue IS GREATER THAN zero  
            REMOVE the first element FROM queue AND SET current_state and steps TO the two elements of that removed tuple  
            
            FOR each neighbor IN neighbors(current_state)  
                IF neighbor NOT IN visited AND neighbor NOT IN deadends_set  
                    IF neighbor EQUALS target  
                        RETURN steps PLUS one  
                    END IF  
                    ADD neighbor TO visited  
                    APPEND a tuple of neighbor and steps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
