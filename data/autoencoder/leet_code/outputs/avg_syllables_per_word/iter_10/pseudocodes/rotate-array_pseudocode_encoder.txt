CLASS Solution  
    FUNCTION rotate(nums, k)  
        SET n TO the LENGTH OF nums  
        SET k TO k MODULO n  
        
        FUNCTION reverse(start, end)  
            WHILE start LESS THAN end  
                SET temp TO element at position start of nums  
                SET element at position start of nums TO element at position end of nums  
                SET element at position end of nums TO temp  
                INCREMENT start BY one  
                DECREMENT end BY one  
            END WHILE  
        END FUNCTION  
        
        CALL reverse with start as zero and end as n MINUS one  
        CALL reverse with start as zero and end as k MINUS one  
        CALL reverse with start as k and end as n MINUS one  
    END FUNCTION  
END CLASS
