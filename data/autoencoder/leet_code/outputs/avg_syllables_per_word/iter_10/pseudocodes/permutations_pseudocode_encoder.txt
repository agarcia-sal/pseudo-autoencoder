CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF remaining EQUALS empty  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR i FROM zero TO the LENGTH OF remaining MINUS one  
                SET new_path TO path PLUS element at position i of remaining  
                SET new_remaining TO elements from start TO i MINUS one OF remaining PLUS elements from i PLUS one TO end OF remaining  
                CALL backtrack WITH new_path, new_remaining  
            END FOR  
        END FUNCTION  
        SET permutations TO empty list  
        CALL backtrack WITH empty list, nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
