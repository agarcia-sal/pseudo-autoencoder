CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val equal to element at position zero of values  
    SET i TO one  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element removed FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode with val equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode with val equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters p.left, q.left AND is_same_tree with parameters p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum with parameters root, targetSum  
        FUNCTION helper with parameters node, current_sum, prefix_sums  
            IF node IS None  
                RETURN zero  
            END IF  
            INCREMENT current_sum BY node.val  
            SET paths_to_here TO value at key (current_sum MINUS targetSum) in prefix_sums OR zero if key is not present  
            SET value at key current_sum in prefix_sums TO value at key current_sum in prefix_sums OR zero PLUS one  
            SET paths_in_left TO helper with parameters node.left, current_sum, prefix_sums  
            SET paths_in_right TO helper with parameters node.right, current_sum, prefix_sums  
            DECREMENT value at key current_sum in prefix_sums BY one  
            RETURN paths_to_here PLUS paths_in_left PLUS paths_in_right  
        END FUNCTION  
        SET prefix_sums TO dictionary with zero mapped to one  
        RETURN helper with parameters root, zero, prefix_sums  
    END FUNCTION  
END CLASS
