CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO the result of sorting the list obtained by pairing each element FROM wage DIVIDED BY the corresponding element FROM quality WITH the corresponding element FROM quality  
        SET max_heap TO empty list  
        SET total_quality TO zero  
        SET min_cost TO infinity  
        FOR each ratio, q IN workers  
            CALL heappush WITH max_heap AND negative q  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap GREATER THAN k  
                INCREMENT total_quality BY the result of heappop FROM max_heap  
            END IF  
            IF the LENGTH OF max_heap EQUALS k  
                SET min_cost TO the smaller value BETWEEN min_cost AND total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
