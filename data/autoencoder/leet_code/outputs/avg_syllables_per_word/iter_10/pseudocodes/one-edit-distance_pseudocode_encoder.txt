CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF calling isOneEditDistance WITH t AND s  
        END IF  
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        IF diff GREATER THAN one  
            RETURN False  
        END IF  
        IF diff EQUALS zero  
            SET count TO zero  
            FOR each index FROM zero TO the LENGTH OF s MINUS one  
                IF the ELEMENT at position index of s NOT EQUALS the ELEMENT at position index of t  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            RETURN count EQUALS one  
        ELSE  
            FOR each index FROM zero TO the LENGTH OF s MINUS one  
                IF the ELEMENT at position index of s NOT EQUALS the ELEMENT at position index of t  
                    RETURN the SUBSTRING of s FROM position index TO the END EQUALS the SUBSTRING of t FROM position index PLUS one TO the END  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
