CLASS Solution  
    FUNCTION findMaximizedCapital(k, w, profits, capital)  
        SET projects TO the LIST OF PAIRS OF corresponding elements FROM capital AND profits  
        SORT projects BY the first element of each pair in ascending order  
  
        SET max_heap TO the empty list  
  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
  
        FOR each iteration FROM one TO k  
            WHILE i LESS THAN n AND element at position i of projects FIRST element LESS THAN OR EQUAL TO w  
                INSERT the NEGATIVE of element at position i of projects SECOND element INTO max_heap USING heap PUSH operation  
                INCREMENT i BY one  
            END WHILE  
  
            IF max_heap IS empty  
                BREAK the loop  
            END IF  
  
            SET w TO w PLUS the NEGATIVE of the element REMOVED FROM max_heap USING heap POP operation  
        END FOR  
  
        RETURN w  
    END FUNCTION  
END CLASS
