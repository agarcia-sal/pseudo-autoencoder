CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num  
                IF the element at position start of num EQUALS zero AND end GREATER THAN start PLUS one  
                    CONTINUE to the next iteration  
                END IF  
                SET next_num TO the integer value represented by the substring of num from position start TO end MINUS one  
                IF next_num GREATER THAN OR EQUAL TO two RAISED TO THE power OF thirty ONE  
                    BREAK the loop  
                END IF  
                IF the LENGTH OF path LESS THAN two OR next_num EQUALS the sum OF the element at the LAST position OF path AND the element at the SECOND LAST position OF path  
                    SET result TO backtrack(end, path APPENDED WITH next_num)  
                    IF result NOT EQUALS an empty list  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN an empty list  
        END FUNCTION  
        RETURN backtrack(zero, an empty list)  
    END FUNCTION  
END CLASS
