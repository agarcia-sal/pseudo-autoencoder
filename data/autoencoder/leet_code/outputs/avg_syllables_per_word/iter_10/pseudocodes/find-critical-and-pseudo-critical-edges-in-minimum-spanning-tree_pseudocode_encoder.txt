CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of integers from zero TO n MINUS one  
        SET self.rank TO list of integer one repeated n times  
    END FUNCTION  
  
    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u THEN  
            SET element at position u of self.parent TO result of self.find with parameter element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  
  
    FUNCTION union(self, u, v)  
        SET rootU TO result of self.find with parameter u  
        SET rootV TO result of self.find with parameter v  
        IF rootU NOT EQUALS rootV THEN  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank THEN  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank THEN  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)  
        FOR each index i AND element edge IN edges with indexes  
            APPEND i TO edge  
        END FOR  
  
        CALL edges.sort USING key as function that returns element at position two of input  
  
        FUNCTION mst(exclude, include)  
            SET uf TO new instance of UnionFind with parameter n  
            SET weight TO zero  
            IF include IS NOT None THEN  
                CALL uf.union WITH parameters element at position zero of include AND element at position one of include  
                INCREMENT weight BY element at position two of include  
            END IF  
  
            FOR each element u, v, w, idx IN edges  
                IF idx EQUALS exclude THEN  
                    CONTINUE to next iteration  
                END IF  
                IF CALL uf.union WITH parameters u AND v RETURNS True THEN  
                    INCREMENT weight BY w  
                END IF  
            END FOR  
  
            SET root TO result of uf.find with parameter zero  
            FOR index i FROM zero TO n MINUS one  
                IF result of uf.find with parameter i NOT EQUALS root THEN  
                    RETURN infinity  
                END IF  
            END FOR  
  
            RETURN weight  
        END FUNCTION  
  
        SET mst_weight TO result of mst with no parameters  
        SET critical TO empty list  
        SET pseudo_critical TO empty list  
  
        FOR each element u, v, w, idx IN edges  
            IF result of mst with parameter exclude equal to idx GREATER THAN mst_weight THEN  
                APPEND idx TO critical  
            ELSE IF result of mst with parameter include equal to tuple of u, v, w EQUALS mst_weight THEN  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR  
  
        RETURN list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
