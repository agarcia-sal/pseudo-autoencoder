CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO result of calling defaultdict with int as argument  
        FOR each i IN nums  
            FOR each j IN nums  
                SET current_and TO bitwise AND of element i and element j  
                INCREMENT value at key current_and in pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each k IN nums  
            FOR each pair_and key AND count value IN pair_and_count  
                IF bitwise AND of pair_and and k EQUALS zero  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
