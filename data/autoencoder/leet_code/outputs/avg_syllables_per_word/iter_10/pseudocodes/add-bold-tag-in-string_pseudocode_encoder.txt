CLASS Solution  
    FUNCTION addBoldTag(s, words)  
        IF words EQUALS empty  
            RETURN s  
        END IF  

        SET n TO the LENGTH OF s  
        SET mask TO list of FALSE repeated n times  

        FOR each word IN words  
            SET start TO zero  
            WHILE start LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the POSITION OF word IN s BEGINNING AT start  
                IF start EQUALS negative one  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET element at position i OF mask TO TRUE  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  

        SET result TO empty list  
        SET i TO zero  
        WHILE i LESS THAN n  
            IF element at position i OF mask EQUALS TRUE  
                APPEND the string "<b>" TO result  
                WHILE i LESS THAN n AND element at position i OF mask EQUALS TRUE  
                    APPEND element at position i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string "</b>" TO result  
            ELSE  
                APPEND element at position i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  

        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
