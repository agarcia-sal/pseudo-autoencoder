CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  

        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF the element at position i of board AND the element at position zero of board AND the element at position j of the first element of board AND the element at position zero of the first element of board combined with EXCLUSIVE OR operations IS NOT EQUAL TO zero  
                    RETURN minus one  
                END IF  
            END FOR  

            INCREMENT rowSum BY the element at position i of board at position zero  
            INCREMENT colSum BY the element at position zero of board at position i  
            INCREMENT rowSwap BY the VALUE OF WHETHER the element at position i of board at position zero IS EQUAL TO i MODULO two  
            INCREMENT colSwap BY the VALUE OF WHETHER the element at position zero of board at position i IS EQUAL TO i MODULO two  
        END FOR  

        IF integer DIVISION of N BY two IS GREATER THAN rowSum OR rowSum IS GREATER THAN integer DIVISION of the SUM of N AND one BY two  
            RETURN minus one  
        END IF  

        IF integer DIVISION of N BY two IS GREATER THAN colSum OR colSum IS GREATER THAN integer DIVISION of the SUM of N AND one BY two  
            RETURN minus one  
        END IF  

        IF N MODULO two IS EQUAL TO one  
            IF rowSwap MODULO two IS EQUAL TO one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  

            IF colSwap MODULO two IS EQUAL TO one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the MINIMUM OF rowSwap AND N MINUS rowSwap  
            SET colSwap TO the MINIMUM OF colSwap AND N MINUS colSwap  
        END IF  

        SET result TO integer DIVISION of the SUM of rowSwap AND colSwap BY two  
        RETURN result  
    END FUNCTION  
END CLASS
