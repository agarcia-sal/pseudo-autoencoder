CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO empty list  
        FOR each building IN buildings  
            SET left TO first element of building  
            SET right TO second element of building  
            SET height TO third element of building  
            APPEND tuple of left, negative height, right TO events  
            APPEND tuple of right, zero, zero TO events  
        END FOR  
      
        SORT events BY first element then second element then third element ascending  
      
        SET result TO list containing list of zero and zero  
        SET max_heap TO list containing tuple of zero and positive infinity  
      
        FOR each event IN events  
            SET x TO first element of event  
            SET negH TO second element of event  
            SET r TO third element of event  
          
            WHILE first element of max_heap has second element LESS THAN OR EQUAL TO x  
                REMOVE first element FROM max_heap using heap pop  
            END WHILE  
          
            IF negH NOT EQUALS zero  
                ADD tuple of negH and r TO max_heap using heap push  
            END IF  
          
            IF second element of last element of result NOT EQUALS negative of first element of first element of max_heap  
                APPEND list of x and negative of first element of first element of max_heap TO result  
            END IF  
        END FOR  
      
        IF first element of result EQUALS list of zero and zero  
            REMOVE first element FROM result  
        END IF  
      
        RETURN result  
    END FUNCTION  
END CLASS
