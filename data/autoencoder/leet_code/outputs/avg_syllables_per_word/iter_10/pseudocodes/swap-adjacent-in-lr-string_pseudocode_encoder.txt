CLASS Solution  
    FUNCTION canTransform(start, end)  
        SET start_filtered TO an empty list  
        FOR index FROM zero TO the LENGTH OF start MINUS one  
            SET c TO element at position index of start  
            IF c NOT EQUALS character 'X'  
                APPEND tuple of c and index TO start_filtered  
            END IF  
        END FOR  
        
        SET end_filtered TO an empty list  
        FOR index FROM zero TO the LENGTH OF end MINUS one  
            SET c TO element at position index of end  
            IF c NOT EQUALS character 'X'  
                APPEND tuple of c and index TO end_filtered  
            END IF  
        END FOR  
        
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        
        FOR position FROM zero TO the LENGTH OF start_filtered MINUS one  
            SET tuple_start TO element at position position of start_filtered  
            SET tuple_end TO element at position position of end_filtered  
            SET c1 TO first element of tuple_start  
            SET i1 TO second element of tuple_start  
            SET c2 TO first element of tuple_end  
            SET i2 TO second element of tuple_end  
            
            IF c1 NOT EQUALS c2  
                RETURN false  
            END IF  
            
            IF c1 EQUALS character 'L' AND i1 LESS THAN i2  
                RETURN false  
            END IF  
            
            IF c1 EQUALS character 'R' AND i1 GREATER THAN i2  
                RETURN false  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
