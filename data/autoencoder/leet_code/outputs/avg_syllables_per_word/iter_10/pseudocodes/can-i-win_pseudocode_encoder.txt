CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF maxChoosableInteger MULTIPLIED BY (maxChoosableInteger PLUS one) DIVIDED BY two LESS THAN desiredTotal  
            RETURN False  
        END IF  
        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN True  
        END IF  
        SET memo TO an empty dictionary  
        FUNCTION can_win(current_total, used_numbers)  
            IF the tuple of current_total AND used_numbers EXISTS AS A KEY IN memo  
                RETURN the value at memo FOR the tuple of current_total AND used_numbers  
            END IF  
            FOR number FROM one TO maxChoosableInteger  
                IF NOT the bitwise AND of used_numbers AND one LEFT SHIFTED BY number IS NOT zero  
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win(current_total PLUS number, the bitwise OR of used_numbers AND one LEFT SHIFTED BY number)  
                        SET memo AT tuple of current_total AND used_numbers TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            SET memo AT tuple of current_total AND used_numbers TO False  
            RETURN False  
        END FUNCTION  
        RETURN can_win(zero, zero)  
    END FUNCTION  
END CLASS
