CLASS Solution  
    FUNCTION findPairs(nums, k)  
        IF k LESS THAN zero  
            RETURN zero  
        END IF  
  
        SET num_counts TO empty dictionary  
        FOR each num IN nums  
            IF num NOT IN num_counts  
                SET num_counts[num] TO zero  
            END IF  
            INCREMENT num_counts[num] BY one  
        END FOR  
  
        SET unique_pairs TO zero  
        FOR each num IN num_counts  
            IF k EQUALS zero  
                IF num_counts[num] GREATER THAN one  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF num PLUS k IN num_counts  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
  
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
