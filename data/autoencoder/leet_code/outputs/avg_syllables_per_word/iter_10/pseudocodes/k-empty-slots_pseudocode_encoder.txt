CLASS Solution  
    FUNCTION kEmptySlots(bulbs, k)  
        SET bloom_days TO an empty list  
        FOR each day and position IN the enumeration of bulbs starting from one  
            SET i TO the position where position can be inserted in bloom_days maintaining sorted order  
            IF i GREATER THAN zero AND position MINUS the element at position i MINUS one element in bloom_days EQUALS k  
                RETURN day  
            END IF  
            IF i LESS THAN the LENGTH OF bloom_days AND the element at position i IN bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position INTO bloom_days AT position i  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
