CLASS Solution  
    FUNCTION minSubArrayLen(target, nums)  
        SET left TO zero  
        SET current_sum TO zero  
        SET min_length TO infinity  
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            SET current_sum TO current_sum PLUS element at position right of nums  
            WHILE current_sum GREATER THAN OR EQUAL TO target  
                SET min_length TO the MINIMUM OF min_length AND right MINUS left PLUS one  
                SET current_sum TO current_sum MINUS element at position left of nums  
                INCREMENT left BY one  
            END WHILE  
        END FOR  
        IF min_length NOT EQUALS infinity  
            RETURN min_length  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
