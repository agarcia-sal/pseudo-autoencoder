CLASS Solution
    FUNCTION customSortString(order, s)
        SET d TO an empty dictionary
        SET i TO zero
        FOR each character c IN order
            SET the element at key c in d TO i
            INCREMENT i BY one
        END FOR
        SET sorted_characters TO s sorted by the value obtained from d for each character x, using zero if x is not a key in d
        SET result_string TO the concatenation of all characters in sorted_characters in order
        RETURN result_string
    END FUNCTION
END CLASS
