CLASS Solution  
    FUNCTION checkInclusion(s1, s2)  
        SET len_s1 TO the LENGTH OF s1  
        SET len_s2 TO the LENGTH OF s2  
        IF len_s1 GREATER THAN len_s2  
            RETURN FALSE  
        END IF  
        SET count_s1 TO a frequency counter of characters in s1  
        SET count_s2 TO a frequency counter of characters in the first len_s1 characters of s2  
        IF count_s1 EQUALS count_s2  
            RETURN TRUE  
        END IF  
        FOR index FROM len_s1 TO len_s2 MINUS one  
            INCREMENT count of character at position index of s2 BY one in count_s2  
            DECREMENT count of character at position index MINUS len_s1 of s2 BY one in count_s2  
            IF count of character at position index MINUS len_s1 of s2 EQUALS zero in count_s2  
                REMOVE that character FROM count_s2  
            END IF  
            IF count_s1 EQUALS count_s2  
                RETURN TRUE  
            END IF  
        END FOR  
        RETURN FALSE  
    END FUNCTION  
END CLASS
