CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO an empty set  
        FOR each mine IN mines  
            APPEND a tuple containing the elements of mine TO mines_set  
        END FOR  
        
        SET dp TO a two-dimensional list of size n by n where each element is a list of four zeros  
        
        SET directions TO a list containing the tuples zero and one, one and zero, zero and negative one, and negative one and zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the tuple of i and j NOT IN mines_set THEN  
                    IF j GREATER THAN zero THEN  
                        SET value at position zero of list at position j MINUS one of list at position i in dp TO value at position zero of list at position j MINUS one of list at position i in dp PLUS one  
                        SET value at position zero of list at position j of list at position i in dp TO the previous value just set  
                    END IF  
                    IF i GREATER THAN zero THEN  
                        SET value at position one of list at position j of list at position i MINUS one in dp TO value at position one of list at position j of list at position i MINUS one in dp PLUS one  
                        SET value at position one of list at position j of list at position i in dp TO the previous value just set  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the tuple of i and j NOT IN mines_set THEN  
                    IF j LESS THAN n MINUS one THEN  
                        SET value at position two of list at position j PLUS one of list at position i in dp TO value at position two of list at position j PLUS one of list at position i in dp PLUS one  
                        SET value at position two of list at position j of list at position i in dp TO the previous value just set  
                    END IF  
                    IF i LESS THAN n MINUS one THEN  
                        SET value at position three of list at position j of list at position i PLUS one in dp TO value at position three of list at position j of list at position i PLUS one in dp PLUS one  
                        SET value at position three of list at position j of list at position i in dp TO the previous value just set  
                    END IF  
                    
                    SET plus_sign_order TO the minimum of the list at position j of list at position i in dp  
                    IF plus_sign_order GREATER THAN result THEN  
                        SET result TO plus_sign_order  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
