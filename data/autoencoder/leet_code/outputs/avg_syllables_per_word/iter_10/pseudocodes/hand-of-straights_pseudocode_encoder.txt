CLASS Solution  
    FUNCTION isNStraightHand(hand, groupSize)  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero  
            RETURN False  
        END IF  
        
        SET count TO a new count of frequencies of elements in hand  
        SET unique_cards TO the sorted list of keys in count  
        
        FOR each card IN unique_cards  
            IF the value of count at card GREATER THAN zero  
                SET needed TO the value of count at card  
                FOR index FROM zero TO groupSize MINUS one  
                    IF the value of count at the sum of card PLUS index LESS THAN needed  
                        RETURN False  
                    END IF  
                    DECREMENT the value of count at the sum of card PLUS index BY needed  
                END FOR  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
