CLASS Solution  
    FUNCTION kSimilarity(s1, s2)  
        FUNCTION neighbors(s)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND the element at position i of s EQUALS the element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF the element at position j of s EQUALS the element at position i of s2 AND the element at position j of s NOT EQUALS the element at position j of s2  
                    YIELD the concatenation of: the substring of s from start TO position i MINUS one, the element at position j of s, the substring of s from position i PLUS one TO position j MINUS one, the element at position i of s, the substring of s from position j PLUS one TO end  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET queue TO a new double-ended queue initialized with the tuple of s1 and zero  
        SET visited TO a new set containing s1  
        
        WHILE the queue IS NOT empty  
            SET s and swaps TO the first element removed from the left side of the queue  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors(s)  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND the tuple of neighbor and swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
