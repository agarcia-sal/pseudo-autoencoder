CLASS Solution  
    FUNCTION isSelfCrossing(distance)  
        SET n TO the LENGTH OF distance  
        IF n LESS THAN 4  
            RETURN False  
        END IF  
        
        FOR index FROM 3 TO n MINUS 1  
            IF the element at position index of distance GREATER THAN OR EQUAL TO the element at position index MINUS 2 of distance AND the element at position index MINUS 1 of distance LESS THAN OR EQUAL TO the element at position index MINUS 3 of distance  
                RETURN True  
            END IF  
            
            IF index GREATER THAN OR EQUAL TO 4 AND the element at position index MINUS 1 of distance EQUALS the element at position index MINUS 3 of distance AND the sum of the element at position index of distance PLUS the element at position index MINUS 4 of distance GREATER THAN OR EQUAL TO the element at position index MINUS 2 of distance  
                RETURN True  
            END IF  
            
            IF index GREATER THAN OR EQUAL TO 5 AND the element at position index MINUS 2 of distance GREATER THAN OR EQUAL TO the element at position index MINUS 4 of distance AND the element at position index MINUS 3 of distance GREATER THAN OR EQUAL TO the element at position index MINUS 1 of distance AND the sum of the element at position index MINUS 1 of distance PLUS the element at position index MINUS 5 of distance GREATER THAN OR EQUAL TO the element at position index MINUS 3 of distance AND the sum of the element at position index of distance PLUS the element at position index MINUS 4 of distance GREATER THAN OR EQUAL TO the element at position index MINUS 2 of distance  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
