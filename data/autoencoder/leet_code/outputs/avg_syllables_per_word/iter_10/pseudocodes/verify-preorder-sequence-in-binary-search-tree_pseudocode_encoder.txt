CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO empty list  
        SET lower_bound TO negative infinity  
        FOR each value IN preorder  
            IF value LESS THAN lower_bound  
                RETURN False  
            END IF  
            WHILE stack NOT EMPTY AND value GREATER THAN last element of stack  
                SET lower_bound TO last element of stack  
                REMOVE last element FROM stack  
            END WHILE  
            APPEND value TO stack  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
