CLASS Solution  
    FUNCTION minOperations(target, arr)  
        SET index_map TO empty mapping  
        FOR each pair of index, value IN enumeration of target  
            SET index_map at key value TO index  
        END FOR  
        
        SET transformed_arr TO empty list  
        FOR each value IN arr  
            IF value EXISTS IN index_map  
                APPEND index_map at key value TO transformed_arr  
            END IF  
        END FOR  
        
        SET lis TO empty list  
        FOR each num IN transformed_arr  
            SET pos TO position of first element GREATER THAN OR EQUAL TO num IN lis  
            IF pos EQUALS length of lis  
                APPEND num TO lis  
            ELSE  
                SET element at position pos OF lis TO num  
            END IF  
        END FOR  
        
        SET lcs_length TO length of lis  
        
        RETURN length of target MINUS lcs_length  
    END FUNCTION  
END CLASS
