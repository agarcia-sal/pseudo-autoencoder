CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS reverse of check_str  
        END FUNCTION  
        
        FOR i FROM the LENGTH OF s DOWN TO zero  
            IF is_palindrome(substring of s from start TO position i)  
                SET to_add TO reverse of substring of s from position i TO end  
                RETURN concatenation of to_add FOLLOWED BY s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
