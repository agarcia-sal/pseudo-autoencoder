CLASS Solution  
    FUNCTION computeArea(ax1, ay1, ax2, ay2, bx1, by1, bx2, by2)  
        SET area1 TO the difference between ax2 and ax1 MULTIPLIED BY the difference between ay2 and ay1  
        SET area2 TO the difference between bx2 and bx1 MULTIPLIED BY the difference between by2 and by1  
        
        IF ax2 LESS THAN bx2 THEN  
            SET min_ax2_bx2 TO ax2  
        ELSE  
            SET min_ax2_bx2 TO bx2  
        END IF  
        
        IF ax1 GREATER THAN bx1 THEN  
            SET max_ax1_bx1 TO ax1  
        ELSE  
            SET max_ax1_bx1 TO bx1  
        END IF  
        
        SET overlap_width TO the maximum between zero and the difference between min_ax2_bx2 and max_ax1_bx1  
        
        IF ay2 LESS THAN by2 THEN  
            SET min_ay2_by2 TO ay2  
        ELSE  
            SET min_ay2_by2 TO by2  
        END IF  
        
        IF ay1 GREATER THAN by1 THEN  
            SET max_ay1_by1 TO ay1  
        ELSE  
            SET max_ay1_by1 TO by1  
        END IF  
        
        SET overlap_height TO the maximum between zero and the difference between min_ay2_by2 and max_ay1_by1  
        
        SET overlap_area TO overlap_width MULTIPLIED BY overlap_height  
        
        SET total_area TO area1 PLUS area2 MINUS overlap_area  
        
        RETURN total_area  
    END FUNCTION  
END CLASS
