CLASS Solution  
    FUNCTION mostSimilar(n, roads, names, targetPath)  
        SET graph TO call create_defaultdict_of_lists()  
        FOR each pair u, v IN roads  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
        
        SET m TO the LENGTH OF targetPath  
        
        FUNCTION dp(i, j)  
            IF i EQUALS zero  
                IF element at position j of names NOT EQUALS element at position i of targetPath  
                    RETURN one, list containing j  
                ELSE  
                    RETURN zero, list containing j  
                END IF  
            END IF  
            
            SET min_cost TO positive infinity  
            SET min_path TO empty list  
            FOR each k IN element at position j of graph  
                SET cost, path TO dp(i MINUS one, k)  
                IF cost LESS THAN min_cost  
                    SET min_cost TO cost  
                    SET min_path TO path  
                END IF  
            END FOR  
            
            SET current_cost TO min_cost PLUS (one IF element at position j of names NOT EQUALS element at position i of targetPath ELSE zero)  
            RETURN current_cost, concatenation of min_path with list containing j  
        END FUNCTION  
        
        SET min_cost TO positive infinity  
        SET min_path TO empty list  
        FOR j FROM zero TO n MINUS one  
            SET cost, path TO dp(m MINUS one, j)  
            IF cost LESS THAN min_cost  
                SET min_cost TO cost  
                SET min_path TO path  
            END IF  
        END FOR  
        
        RETURN min_path  
    END FUNCTION  
END CLASS
