CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with value at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode with value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode with value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree for left child of p AND left child of q AND is_same_tree for right child of p AND right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self, root, k)  
        IF root.left IS none AND root.right IS none  
            RETURN root.val  
        END IF  
        
        SET graph TO an empty default dictionary of lists  
        
        FUNCTION dfs(node, parent EQUALS none)  
            IF parent IS NOT none  
                APPEND parent.val TO the list at key node.val in graph  
                APPEND node.val TO the list at key parent.val in graph  
            END IF  
            IF node.left IS NOT none  
                CALL dfs with node.left and node as arguments  
            END IF  
            IF node.right IS NOT none  
                CALL dfs with node.right and node as arguments  
            END IF  
        END FUNCTION  
        
        CALL dfs with root as argument  
        
        SET queue TO a double-ended queue containing k  
        SET visited TO an empty set  
        
        WHILE queue IS NOT empty  
            SET current TO the first element REMOVED FROM queue  
            ADD current TO visited  
            
            IF the LENGTH OF the list at key current in graph EQUALS one AND current NOT EQUALS root.val  
                RETURN current  
            END IF  
            
            FOR each neighbor IN the list at key current in graph  
                IF neighbor NOT IN visited  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
