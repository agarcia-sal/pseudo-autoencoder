CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a list with size TWO RAISED TO THE POWER OF n filled with infinity  
        SET the element at position zero of dp TO zero  
        
        FOR mask FROM zero TO TWO RAISED TO THE POWER OF n MINUS one  
            SET bit_count TO the NUMBER OF ONES IN THE BINARY REPRESENTATION OF mask  
            FOR j FROM zero TO n MINUS one  
                IF the bit at position j IN mask IS SET  
                    SET dp at position mask TO the MINIMUM OF dp at position mask AND dp at position (mask WITH bit j CLEARED) PLUS the BITWISE XOR of element at position bit_count MINUS one in nums1 AND element at position j in nums2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at position TWO RAISED TO THE POWER OF n MINUS one  
    END FUNCTION  
END CLASS
