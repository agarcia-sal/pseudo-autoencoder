CLASS Solution  
    FUNCTION kthSmallestProduct(nums1, nums2, k)  
        FUNCTION count(p)  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each x IN nums1  
                IF x GREATER THAN zero  
                    INCREMENT cnt BY the POSITION TO INSERT p DIVIDED BY x TO THE RIGHT IN nums2  
                ELSE IF x LESS THAN zero  
                    INCREMENT cnt BY n MINUS the POSITION TO INSERT p DIVIDED BY x TO THE LEFT IN nums2  
                ELSE  
                    INCREMENT cnt BY n MULTIPLIED BY one IF p GREATER THAN OR EQUAL TO zero ELSE zero  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  

        SET mx TO the MAXIMUM OF THE ABSOLUTE VALUE OF the FIRST ELEMENT OF nums1 AND THE ABSOLUTE VALUE OF the LAST ELEMENT OF nums1  
        MULTIPLIED BY the MAXIMUM OF THE ABSOLUTE VALUE OF the FIRST ELEMENT OF nums2 AND THE ABSOLUTE VALUE OF the LAST ELEMENT OF nums2  
        RETURN the POSITION TO INSERT k TO THE LEFT IN THE RANGE FROM NEGATIVE mx TO mx PLUS one USING count AS THE KEY MINUS mx  
    END FUNCTION  
END CLASS
