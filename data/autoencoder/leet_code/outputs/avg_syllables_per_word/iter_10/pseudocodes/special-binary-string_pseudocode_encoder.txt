CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each index i AND character char IN s  
                IF char EQUALS character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND concatenation of character one, dfs(substring of s from position start PLUS one TO position i MINUS one), and character zero TO specials  
                    SET start TO i PLUS one  
                END IF  
            END FOR  
            SORT specials IN descending lexicographical order  
            RETURN concatenation of all elements in specials  
        END FUNCTION  
        RETURN dfs(s)  
    END FUNCTION  
END CLASS
