CLASS Solution  
    FUNCTION minChanges(nums, k)  
        SET groups TO result of defaultdict of list  
        FOR i FROM zero TO LENGTH OF nums MINUS one  
            APPEND element at position i of nums TO element at position i MODULO k of groups  
        END FOR  

        SET freq TO list containing k elements each as a defaultdict of integer  
        SET min_size TO list containing k elements each as infinity  
        FOR i FROM zero TO k MINUS one  
            FOR each num IN element at position i of groups  
                INCREMENT element at key num in element at position i of freq BY one  
            END FOR  
            SET element at position i of min_size TO LENGTH OF element at position i of groups  
        END FOR  

        SET dp TO list of length two POWER ten with each element as infinity  
        SET element at position zero of dp TO zero  

        FOR i FROM zero TO k MINUS one  
            SET new_dp TO list of length two POWER ten with each element as infinity  
            SET min_dp TO minimum value IN dp  
            FOR j FROM zero TO two POWER ten MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position i of min_size  
                FOR each num, count IN element at position i of freq  
                    SET element at position j of new_dp TO MINIMUM OF element at position j of new_dp AND element at position j XOR num of dp PLUS element at position i of min_size MINUS count  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  

        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
