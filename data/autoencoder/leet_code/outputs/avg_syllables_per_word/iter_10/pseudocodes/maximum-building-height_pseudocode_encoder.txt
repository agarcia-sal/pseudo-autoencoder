CLASS Solution  
    FUNCTION maxBuilding(n, restrictions)  
        APPEND list containing one and zero TO restrictions  
        SORT restrictions  
        
        FOR i FROM one TO the LENGTH OF restrictions MINUS one  
            SET idx TO element at position i at position zero of restrictions  
            SET height TO element at position i at position one of restrictions  
            SET prev_idx TO element at position i MINUS one at position zero of restrictions  
            SET prev_height TO element at position i MINUS one at position one of restrictions  
            SET element at position i at position one of restrictions TO the MINIMUM OF height AND prev_height PLUS idx MINUS prev_idx  
        END FOR  
        
        FOR i FROM the LENGTH OF restrictions MINUS two DOWN TO zero  
            SET idx TO element at position i at position zero of restrictions  
            SET height TO element at position i at position one of restrictions  
            SET next_idx TO element at position i PLUS one at position zero of restrictions  
            SET next_height TO element at position i PLUS one at position one of restrictions  
            SET element at position i at position one of restrictions TO the MINIMUM OF height AND next_height PLUS next_idx MINUS idx  
        END FOR  
        
        SET max_height TO zero  
        SET prev_idx TO element at position zero at position zero of restrictions  
        SET prev_height TO element at position zero at position one of restrictions  
        
        FOR i FROM one TO the LENGTH OF restrictions MINUS one  
            SET idx TO element at position i at position zero of restrictions  
            SET height TO element at position i at position one of restrictions  
            SET max_height TO the MAXIMUM OF max_height AND the INTEGER DIVIDED VALUE OF idx MINUS prev_idx PLUS prev_height PLUS height BY two  
            SET prev_idx TO idx  
            SET prev_height TO height  
        END FOR  
        
        SET max_height TO the MAXIMUM OF max_height AND prev_height PLUS n MINUS prev_idx  
        
        RETURN max_height  
    END FUNCTION  
END CLASS
