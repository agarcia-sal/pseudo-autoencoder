CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            
            SET expected TO the STRING REPRESENTATION OF first PLUS second  
            SET length TO the LENGTH OF expected  
            
            IF start PLUS length GREATER THAN the LENGTH OF num OR the SUBSTRING of num FROM start TO start PLUS length NOT EQUALS expected  
                RETURN False  
            END IF  
            
            RETURN is_valid_sequence(start PLUS length, second, the INTEGER REPRESENTATION OF expected)  
        END FUNCTION  
        
        SET n TO the LENGTH OF num  
        FOR i FROM one TO n DIVIDED BY two PLUS one  
            FOR j FROM i PLUS one TO n MINUS one PLUS one  
                SET first TO the SUBSTRING of num FROM zero TO i  
                SET second TO the SUBSTRING of num FROM i TO j  
                
                IF the LENGTH OF first GREATER THAN one AND the CHARACTER AT POSITION zero OF first EQUALS zero OR the LENGTH OF second GREATER THAN one AND the CHARACTER AT POSITION zero OF second EQUALS zero  
                    CONTINUE TO next iteration  
                END IF  
                
                IF is_valid_sequence(j, the INTEGER REPRESENTATION OF first, the INTEGER REPRESENTATION OF second)  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
