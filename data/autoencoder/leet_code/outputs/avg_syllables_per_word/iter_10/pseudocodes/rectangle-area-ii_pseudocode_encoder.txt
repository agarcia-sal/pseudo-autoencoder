CLASS Solution  
    FUNCTION rectangleArea(rectangles)  
        SET MOD TO one billion PLUS seven  
        
        SET y_coords TO empty set  
        FOR each x1, y1, x2, y2 IN rectangles  
            ADD y1 TO y_coords  
            ADD y2 TO y_coords  
        END FOR  
        
        SET y_coords TO the sorted list of elements in y_coords  
        SET y_index TO an empty mapping from y to index  
        FOR index FROM zero TO the LENGTH OF y_coords MINUS one  
            SET element at position index of y_coords TO y  
            SET y_index at key y TO index  
        END FOR  
        
        SET count TO a list of zeros of length the LENGTH OF y_coords  
        
        SET events TO empty list  
        FOR each x1, y1, x2, y2 IN rectangles  
            APPEND tuple of x1, one, y1, y2 TO events  
            APPEND tuple of x2, minus one, y1, y2 TO events  
        END FOR  
        
        SORT events by first element x-coordinate in ascending order  
        
        SET prev_x TO zero  
        SET area TO zero  
        
        FOR each x, eventType, y1, y2 IN events  
            SET current_length TO zero  
            FOR i FROM one TO the LENGTH OF count MINUS one  
                IF the element at position i minus one of count IS GREATER THAN zero  
                    INCREMENT current_length BY the element at position i of y_coords MINUS the element at position i minus one of y_coords  
                END IF  
            END FOR  
            
            INCREMENT area BY current_length MULTIPLIED BY (x MINUS prev_x)  
            SET area TO area MODULO MOD  
            
            IF eventType EQUALS one  
                FOR i FROM y_index at key y1 TO y_index at key y2 MINUS one  
                    INCREMENT element at position i of count BY one  
                END FOR  
            ELSE  
                FOR i FROM y_index at key y1 TO y_index at key y2 MINUS one  
                    DECREMENT element at position i of count BY one  
                END FOR  
            END IF  
            
            SET prev_x TO x  
        END FOR  
        
        RETURN area  
    END FUNCTION  
END CLASS
