CLASS Solution  
    FUNCTION validSubarrays(nums)  
        SET stack TO empty list  
        SET count TO zero  
        FOR each num IN nums  
            WHILE the LENGTH OF stack is greater than zero AND the element at the last position of stack is GREATER THAN num  
                REMOVE last element FROM stack  
            END WHILE  
            APPEND num TO stack  
            INCREMENT count BY the LENGTH OF stack  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
