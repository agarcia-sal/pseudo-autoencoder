CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO a list of string representations of numbers from one TO n  
        WHILE n GREATER THAN one  
            FOR i FROM zero TO n divided by two MINUS one  
                SET element at position i of teams TO opening parenthesis CONCATENATED WITH element at position i of teams CONCATENATED WITH comma CONCATENATED WITH element at position n MINUS one MINUS i of teams CONCATENATED WITH closing parenthesis  
            END FOR  
            SET n TO n divided by two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
