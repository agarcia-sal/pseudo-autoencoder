CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MOD TO one billion PLUS seven  
        SET end_with_0 TO zero  
        SET end_with_1 TO zero  
        SET has_zero TO the BOOLEAN VALUE of whether zero IS IN binary  
        FOR each bit IN binary  
            IF bit EQUALS character one  
                SET end_with_1 TO remainder OF (end_with_0 PLUS end_with_1 PLUS one) DIVIDED BY MOD  
            ELSE  
                SET end_with_0 TO remainder OF (end_with_0 PLUS end_with_1) DIVIDED BY MOD  
            END IF  
        END FOR  
        RETURN remainder OF (end_with_0 PLUS end_with_1 PLUS the numeric VALUE OF has_zero) DIVIDED BY MOD  
    END FUNCTION  
END CLASS
