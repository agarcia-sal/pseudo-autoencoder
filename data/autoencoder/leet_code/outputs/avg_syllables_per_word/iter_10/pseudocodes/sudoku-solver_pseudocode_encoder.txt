CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR i FROM zero TO eight  
                IF element at position i of row of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            FOR i FROM zero TO eight  
                IF element at position col of row i of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY row DIVIDED BY three  
            SET start_col TO three MULTIPLIED BY col DIVIDED BY three  
            FOR i FROM zero TO two  
                FOR j FROM zero TO two  
                    IF element at position start_col PLUS j of row start_row PLUS i of board EQUALS num  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  

        FUNCTION solve(board)  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF element at position col of row of board EQUALS decimal point  
                        FOR num IN the string representations of numbers from one TO nine  
                            IF is_valid(board, row, col, num)  
                                SET element at position col of row of board TO num  
                                IF solve(board)  
                                    RETURN True  
                                END IF  
                                SET element at position col of row of board TO decimal point  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  

        CALL solve(board)  
    END FUNCTION  
END CLASS
