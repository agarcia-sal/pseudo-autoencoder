CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO the LENGTH OF floor  
        INITIALIZE dp AS a two-dimensional list with dimensions n PLUS one by numCarpets PLUS one filled with zeroes  
        FOR i FROM one TO n  
            SET dp at position i and zero TO dp at position i minus one and zero PLUS the expression that element at position i minus one of floor EQUALS one  
        END FOR  
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp at position i and j TO dp at position i minus one and j PLUS the expression that element at position i minus one of floor EQUALS one  
                SET dp at position i and j TO the smaller value BETWEEN dp at position i and j AND dp at position result of maximum BETWEEN zero AND i minus carpetLen and j minus one  
            END FOR  
        END FOR  
        RETURN dp at position n and numCarpets  
    END FUNCTION  
END CLASS
