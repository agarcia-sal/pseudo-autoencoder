CLASS Solution  
    FUNCTION minInterval(intervals, queries)  
        SORT intervals BY the start time of each interval  
        SET sorted_queries TO the sorted list of pairs of query and original index for each index and query in queries  
        SET min_heap TO an empty list  
        SET result TO a list of negative one repeated LENGTH OF queries times  
        SET interval_index TO zero  
        FOR each query and original_index IN sorted_queries  
            WHILE interval_index less than LENGTH OF intervals AND element at position interval_index of intervals start LESS THAN OR EQUAL TO query  
                SET start TO element at position interval_index and position zero of intervals  
                SET end TO element at position interval_index and position one of intervals  
                SET interval_size TO end MINUS start PLUS one  
                ADD the pair of interval_size and end TO min_heap with heap property maintained  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE min_heap is not empty AND the second element of the first pair in min_heap LESS THAN query  
                REMOVE the first element FROM min_heap with heap property maintained  
            END WHILE  
            IF min_heap is not empty  
                SET element at position original_index of result TO the first element of the first pair in min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
