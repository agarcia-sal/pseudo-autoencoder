CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num EQUALS one  
            RETURN one  
        END IF  

        SET factors TO an empty list  
        FOR digit FROM nine DOWNTO two  
            WHILE num MODULO digit EQUALS zero  
                APPEND digit TO factors  
                SET num TO num DIVIDED BY digit  
            END WHILE  
        END FOR  

        IF num NOT EQUALS one  
            RETURN zero  
        END IF  

        SORT factors IN ascending order  

        SET result_string TO concatenate all elements of factors AS strings in order  
        SET result TO convert result_string TO integer  

        IF result GREATER THAN two POWER thirty ONE MINUS one  
            RETURN zero  
        END IF  

        RETURN result  
    END FUNCTION  
END CLASS
