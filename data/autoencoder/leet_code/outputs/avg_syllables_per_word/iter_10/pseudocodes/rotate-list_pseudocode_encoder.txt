CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO a new ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN values starting from position one TO the last element  
        SET node TO a new ListNode initialized with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the element val of p1 EQUALS the element val of p2 AND is_same_list(the element next of p1, the element next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight(self, head, k)  
        IF head EQUALS None OR the element next of head EQUALS None OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE the element next of tail IS NOT None  
            SET tail TO the element next of tail  
            INCREMENT length BY one  
        END WHILE  
        SET the element next of tail TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR index FROM zero TO steps_to_new_tail MINUS one  
            SET new_tail TO the element next of new_tail  
        END FOR  
        SET new_head TO the element next of new_tail  
        SET the element next of new_tail TO None  
        RETURN new_head  
    END FUNCTION  
END CLASS
