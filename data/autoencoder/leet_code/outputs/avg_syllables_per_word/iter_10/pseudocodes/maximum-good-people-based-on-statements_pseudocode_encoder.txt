CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            SET good TO a list of n elements all set TO False  
            SET bad TO a list of n elements all set TO False  
            FOR j FROM zero TO n MINUS one  
                IF the result of bitwise AND between i and two RAISED TO THE POWER OF j IS NOT EQUAL TO zero  
                    SET element at position j of good TO True  
                ELSE  
                    SET element at position j of bad TO True  
                END IF  
            END FOR  
            SET valid TO True  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of good IS True  
                    FOR k FROM zero TO n MINUS one  
                        IF (element at position j of statements at position k EQUALS zero AND element at position k of good IS True) OR (element at position j of statements at position k EQUALS one AND element at position k of bad IS True)  
                            SET valid TO False  
                            BREAK  
                        END IF  
                    END FOR  
                    IF valid IS False  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            IF valid IS True  
                SET max_good TO the GREATER OF max_good AND the SUM OF elements of good THAT ARE True  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
