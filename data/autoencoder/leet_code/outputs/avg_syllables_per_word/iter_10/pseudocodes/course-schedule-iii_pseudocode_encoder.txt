CLASS Solution  
    FUNCTION scheduleCourse(courses)  
        SORT courses BY the value of element at position one of each course  
        SET max_heap TO empty list  
        SET current_end_day TO zero  
        FOR each course IN courses  
            SET duration TO element at position zero of course  
            SET last_day TO element at position one of course  
            IF current_end_day PLUS duration LESS THAN OR EQUAL TO last_day  
                ADD negative duration TO max_heap preserving max-heap property  
                INCREMENT current_end_day BY duration  
            ELSE IF max_heap IS NOT empty AND NEGATIVE of element at position zero of max_heap GREATER THAN duration  
                INCREMENT current_end_day BY duration PLUS REMOVE the element with maximum value FROM max_heap  
                ADD negative duration TO max_heap preserving max-heap property  
            END IF  
        END FOR  
        RETURN the LENGTH OF max_heap  
    END FUNCTION  
END CLASS
