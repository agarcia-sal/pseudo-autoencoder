CLASS Solution  
    FUNCTION sortTransformedArray(nums, a, b, c)  
        FUNCTION quadratic(x)  
            RETURN a MULTIPLIED BY x MULTIPLIED BY x PLUS b MULTIPLIED BY x PLUS c  
        END FUNCTION  
        
        SET n TO LENGTH OF nums  
        SET result TO list of zero SET n times  
        SET left TO zero  
        SET right TO n MINUS one  
        
        IF a GREATER THAN OR EQUAL TO zero  
            SET index TO n MINUS one  
            WHILE left LESS THAN OR EQUAL TO right  
                IF quadratic(element at position left of nums) GREATER THAN quadratic(element at position right of nums)  
                    SET element at position index of result TO quadratic(element at position left of nums)  
                    INCREMENT left BY one  
                ELSE  
                    SET element at position index of result TO quadratic(element at position right of nums)  
                    DECREMENT right BY one  
                END IF  
                DECREMENT index BY one  
            END WHILE  
        ELSE  
            SET index TO zero  
            WHILE left LESS THAN OR EQUAL TO right  
                IF quadratic(element at position left of nums) LESS THAN quadratic(element at position right of nums)  
                    SET element at position index of result TO quadratic(element at position left of nums)  
                    INCREMENT left BY one  
                ELSE  
                    SET element at position index of result TO quadratic(element at position right of nums)  
                    DECREMENT right BY one  
                END IF  
                INCREMENT index BY one  
            END WHILE  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
