CLASS Solution  
    FUNCTION largestRectangleArea(heights)  
        SET stack TO a list containing minus one  
        SET max_area TO zero  
        APPEND zero TO heights  
        FOR each index i AND element h IN enumeration of heights  
            WHILE the last element of stack NOT EQUALS minus one AND h LESS THAN the element at position last element of stack of heights  
                SET height TO the element at position stack removed from the end of heights  
                SET width TO i MINUS the last element of stack MINUS one  
                SET max_area TO the greater of max_area AND height MULTIPLIED BY width  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        RETURN max_area  
    END FUNCTION  
END CLASS
