CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the string at position index of strs REVERSED  
            IF reversed_str GREATER THAN the string at position index of strs  
                SET the string at position index of strs TO reversed_str  
            END IF  
        END FOR  
        
        SET doubled_strs TO strs CONCATENATED WITH strs  
        
        SET max_string TO empty string  
        
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN the collection containing False and True  
                IF is_reversed EQUALS True  
                    SET current TO the string at position index of strs REVERSED  
                ELSE  
                    SET current TO the string at position index of strs  
                END IF  
                
                SET remaining TO the concatenation of strings from position index plus one TO position index plus the LENGTH OF strs MINUS one of doubled_strs  
                
                FOR j FROM zero TO the LENGTH OF current  
                    SET candidate TO the substring of current from position j to end CONCATENATED WITH remaining CONCATENATED WITH the substring of current from start TO position j MINUS one  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
