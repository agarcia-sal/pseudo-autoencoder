CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO empty list  
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack is greater than zero AND asteroid LESS THAN zero AND element at position the LAST of stack GREATER THAN zero  
                IF element at position the LAST of stack LESS THAN the NEGATION OF asteroid  
                    REMOVE last element FROM stack  
                    CONTINUE to next iteration of the WHILE loop  
                ELSE IF element at position the LAST of stack EQUALS the NEGATION OF asteroid  
                    REMOVE last element FROM stack  
                END IF  
                BREAK out of the WHILE loop  
            END WHILE  
            IF the WHILE loop did not BREAK  
                APPEND asteroid TO stack  
            END IF  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
