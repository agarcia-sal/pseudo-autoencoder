CLASS Solution  
    FUNCTION areConnected(n, threshold, queries)  
        IF threshold EQUALS zero  
            RETURN a list of True repeated the LENGTH OF queries  
        END IF  
        
        SET parent TO a list of integers from zero TO n  
        
        FUNCTION find(x)  
            IF the element at position x of parent NOT EQUALS x  
                SET the element at position x of parent TO the result of find of the element at position x of parent  
            END IF  
            RETURN the element at position x of parent  
        END FUNCTION  
        
        FUNCTION union(x, y)  
            SET rootX TO find of x  
            SET rootY TO find of y  
            IF rootX NOT EQUALS rootY  
                SET the element at position rootY of parent TO rootX  
            END IF  
        END FUNCTION  
        
        FOR i FROM threshold PLUS one TO n  
            FOR j FROM two MULTIPLIED BY i TO n STEP i  
                CALL union with i and j  
            END FOR  
        END FOR  
        
        SET result TO an empty list  
        FOR each pair (a, b) IN queries  
            APPEND the result of equality test between find of a AND find of b TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
