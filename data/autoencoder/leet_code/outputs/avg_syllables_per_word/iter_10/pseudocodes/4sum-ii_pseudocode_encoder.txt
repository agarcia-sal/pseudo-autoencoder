CLASS Solution  
    FUNCTION fourSumCount(nums1, nums2, nums3, nums4)  
        SET sum_ab TO empty map  
        FOR each element a IN nums1  
            FOR each element b IN nums2  
                SET current_sum TO a PLUS b  
                IF current_sum EXISTS IN sum_ab  
                    INCREMENT value at current_sum IN sum_ab BY one  
                ELSE  
                    SET value at current_sum IN sum_ab TO one  
                END IF  
            END FOR  
        END FOR  
        SET count TO zero  
        FOR each element c IN nums3  
            FOR each element d IN nums4  
                SET target TO NEGATION OF (c PLUS d)  
                IF target EXISTS IN sum_ab  
                    INCREMENT count BY value at target IN sum_ab  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
