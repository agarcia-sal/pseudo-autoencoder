CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO list of integers from zero TO size MINUS one  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u THEN  
            SET element at position u of self.parent TO self.find(element at position u of self.parent)  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET root_u TO self.find(u)  
        SET root_v TO self.find(v)  
        IF root_u NOT EQUALS root_v THEN  
            SET element at position root_u of self.parent TO root_v  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION matrixRankTransform(self, matrix)  
        SET m TO length of matrix  
        SET n TO length of element at position zero of matrix  
        SET value_to_indices TO a new dictionary that returns empty list on missing keys  

        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                APPEND tuple of r and c TO value_to_indices value at element at position r of matrix at position c  
            END FOR  
        END FOR  

        SET row_rank TO list of zero repeated m times  
        SET col_rank TO list of zero repeated n times  
        SET result TO list containing m elements each of which is list of zero repeated n times  

        FOR each value IN keys of value_to_indices sorted in ascending order  
            SET uf TO new UnionFind with parameter m PLUS n  
            SET rank TO a new dictionary that returns zero on missing keys  

            FOR each tuple r and c IN value_to_indices value  
                CALL uf.union with parameters r and c PLUS m  
            END FOR  

            FOR each tuple r and c IN value_to_indices value  
                SET root TO uf.find(r)  
                SET element at key root of rank TO maximum of element at key root of rank AND maximum of element at position r of row_rank AND element at position c of col_rank PLUS one  
            END FOR  

            FOR each tuple r and c IN value_to_indices value  
                SET element at position r of result at position c TO element at key root of rank where root IS uf.find(r)  
                SET element at position r of row_rank TO element at key root of rank where root IS uf.find(r)  
                SET element at position c of col_rank TO element at key root of rank where root IS uf.find(r)  
            END FOR  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
