CLASS Solution  
    FUNCTION findLonelyPixel(picture)  
        IF picture EQUALS empty OR element at position zero of picture EQUALS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO a list of zero repeated m times  
        SET col_count TO a list of zero repeated n times  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF element at position i of picture at position j EQUALS character B  
                    INCREMENT element at position i of row_count BY one  
                    INCREMENT element at position j of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        SET lonely_pixel_count TO zero  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF element at position i of picture at position j EQUALS character B AND element at position i of row_count EQUALS one AND element at position j of col_count EQUALS one  
                    INCREMENT lonely_pixel_count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
