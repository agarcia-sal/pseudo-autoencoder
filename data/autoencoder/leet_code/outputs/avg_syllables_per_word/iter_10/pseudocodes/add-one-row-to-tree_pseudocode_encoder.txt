CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val equal to element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element removed FROM queue  
        IF i LESS THAN length of values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode with val equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode with val equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow(self, root, val, depth)  
        IF depth EQUALS one  
            RETURN a new TreeNode with val equal to val AND left equal to root  
        END IF  
        SET queue TO a new double-ended queue containing a tuple of root AND one  
        WHILE queue IS NOT empty  
            SET node, level TO first element removed FROM queue  
            IF level EQUALS depth MINUS one  
                SET node.left TO a new TreeNode with val equal to val AND left equal to node.left  
                SET node.right TO a new TreeNode with val equal to val AND right equal to node.right  
            ELSE IF level LESS THAN depth MINUS one  
                IF node.left IS NOT none  
                    APPEND tuple of node.left AND level PLUS one TO queue  
                END IF  
                IF node.right IS NOT none  
                    APPEND tuple of node.right AND level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
