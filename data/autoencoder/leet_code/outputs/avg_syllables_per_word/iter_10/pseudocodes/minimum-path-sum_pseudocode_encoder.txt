CLASS Solution
    FUNCTION minPathSum(grid)
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty
            RETURN zero
        END IF
        
        SET m TO the LENGTH OF grid
        SET n TO the LENGTH OF element at position zero of grid
        
        CALL create_2D_list WITH parameters m AND n RETURNING dp
        
        SET element at position zero zero of dp TO element at position zero zero of grid
        
        FOR j FROM one TO n MINUS one
            SET element at position zero j of dp TO element at position zero j MINUS one of dp PLUS element at position zero j of grid
        END FOR
        
        FOR i FROM one TO m MINUS one
            SET element at position i zero of dp TO element at position i MINUS one zero of dp PLUS element at position i zero of grid
        END FOR
        
        FOR i FROM one TO m MINUS one
            FOR j FROM one TO n MINUS one
                SET element at position i j of dp TO the MINIMUM OF element at position i MINUS one j of dp AND element at position i j MINUS one of dp PLUS element at position i j of grid
            END FOR
        END FOR
        
        RETURN element at position m MINUS one n MINUS one of dp
    END FUNCTION
    
    FUNCTION create_2D_list(m, n)
        INITIALIZE empty list result
        FOR i FROM zero TO m MINUS one
            INITIALIZE empty list row
            FOR j FROM zero TO n MINUS one
                APPEND zero TO row
            END FOR
            APPEND row TO result
        END FOR
        RETURN result
    END FUNCTION
END CLASS
