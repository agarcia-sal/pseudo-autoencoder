CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO list of characters in the string representation of n  
        SET length TO the LENGTH OF digits  
        
        SET i TO length MINUS two  
        WHILE i IS GREATER THAN OR EQUAL TO zero AND element at position i of digits IS GREATER THAN OR EQUAL TO element at position i PLUS one of digits  
            DECREMENT i BY one  
        END WHILE  
        
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
        
        SET j TO length MINUS one  
        WHILE element at position j of digits IS LESS THAN OR EQUAL TO element at position i of digits  
            DECREMENT j BY one  
        END WHILE  
        
        SET temporary variable TO element at position i of digits  
        SET element at position i of digits TO element at position j of digits  
        SET element at position j of digits TO temporary variable  
        
        SET suffix TO the sublist of digits from position i PLUS one TO the end of digits  
        SET reversed suffix TO the reverse of suffix  
        SET digits TO the concatenation of the sublist of digits from start TO position i PLUS one AND reversed suffix  
        
        SET result TO integer conversion of the concatenation of all elements of digits  
        
        IF result IS GREATER THAN two TO THE POWER OF thirty-one MINUS one  
            RETURN negative one  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
