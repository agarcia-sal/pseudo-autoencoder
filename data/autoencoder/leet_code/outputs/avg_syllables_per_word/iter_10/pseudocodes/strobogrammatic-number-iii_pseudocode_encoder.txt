CLASS Solution  
    FUNCTION strobogrammaticInRange(low, high)  
        FUNCTION generate_strobogrammatic(n, final_length)  
            IF n EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF n EQUALS one  
                RETURN list containing string zero, string one, and string eight  
            END IF  
            SET middles TO generate_strobogrammatic(n MINUS two, final_length)  
            SET result TO empty list  
            FOR each middle IN middles  
                IF n NOT EQUALS final_length  
                    APPEND string zero CONCATENATED WITH middle CONCATENATED WITH string zero TO result  
                END IF  
                APPEND string one CONCATENATED WITH middle CONCATENATED WITH string one TO result  
                APPEND string six CONCATENATED WITH middle CONCATENATED WITH string nine TO result  
                APPEND string eight CONCATENATED WITH middle CONCATENATED WITH string eight TO result  
                APPEND string nine CONCATENATED WITH middle CONCATENATED WITH string six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        FUNCTION count_strobogrammatic(low, high)  
            SET count TO zero  
            FOR length FROM LENGTH OF low TO LENGTH OF high  
                FOR each num IN generate_strobogrammatic(length, length)  
                    IF (length EQUALS LENGTH OF low AND num LESS THAN low) OR (length EQUALS LENGTH OF high AND num GREATER THAN high)  
                        CONTINUE to next iteration  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN count_strobogrammatic(low, high)  
    END FUNCTION  
END CLASS
