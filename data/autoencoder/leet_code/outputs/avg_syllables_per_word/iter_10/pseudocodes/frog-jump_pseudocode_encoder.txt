CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones EQUALS empty OR the LENGTH OF stones LESS THAN two  
            RETURN False  
        END IF  
        
        SET stone_set TO the SET OF stones  
        SET memo TO an empty dictionary  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the last element of stones  
                RETURN True  
            END IF  
            
            IF the TUPLE OF position AND jump EXISTS AS a KEY IN memo  
                RETURN the VALUE ASSOCIATED WITH the TUPLE OF position AND jump IN memo  
            END IF  
            
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump GREATER THAN zero AND position PLUS next_jump EXISTS IN stone_set  
                    IF can_jump_to(position PLUS next_jump, next_jump)  
                        SET the VALUE ASSOCIATED WITH the TUPLE OF position AND jump IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the VALUE ASSOCIATED WITH the TUPLE OF position AND jump IN memo TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(the first element of stones, zero)  
    END FUNCTION  
END CLASS
