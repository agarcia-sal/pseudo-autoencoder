CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value at position zero of values  
    SET i TO one  
    SET queue TO empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE first element FROM queue and SET to node  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree with parameters p, q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF val of p NOT EQUALS val of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters left of p, left of q AND is_same_tree with parameters right of p, right of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION tree2str with parameter root  
        FUNCTION helper with parameter node  
            IF node EQUALS None  
                RETURN empty string  
            END IF  
            SET result TO string representation of value of node  
            IF left of node EXISTS OR right of node EXISTS  
                SET result TO result PLUS open parenthesis PLUS helper with parameter left of node PLUS close parenthesis  
            END IF  
            IF right of node EXISTS  
                SET result TO result PLUS open parenthesis PLUS helper with parameter right of node PLUS close parenthesis  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN helper with parameter root  
    END FUNCTION  
END CLASS
