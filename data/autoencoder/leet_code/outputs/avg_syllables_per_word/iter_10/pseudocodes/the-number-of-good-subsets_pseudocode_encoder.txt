CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO a frequency count of each number in nums  
        SET dp TO a list of zeros with length equal to two raised to the power of ten  
        SET the element at position zero of dp TO one  
        
        FOR each num IN the keys of count  
            IF num EQUALS one  
                SET the element at position zero of dp TO the element at position zero of dp MULTIPLIED BY two raised to the power of the count of one in count  
                SET the element at position zero of dp TO the element at position zero of dp MODULO MOD  
                CONTINUE to next iteration  
            END IF  
            IF num NOT IN prime_factors  
                CONTINUE to next iteration  
            END IF  
            
            SET factors TO the list of prime factors for num from prime_factors  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask BITWISE OR with one shifted left by prime MINUS one positions  
            END FOR  
            
            FOR m FROM two raised to the power of ten MINUS one DOWN TO zero  
                IF bitwise AND of m and mask EQUALS zero  
                    SET the element at position bitwise OR of m and mask of dp TO the element at that position PLUS the element at position m of dp MULTIPLIED BY the count of num in count  
                    SET the element at position bitwise OR of m and mask of dp TO the element at that position MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of elements of dp from position one to the end  
        SET result TO result MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
