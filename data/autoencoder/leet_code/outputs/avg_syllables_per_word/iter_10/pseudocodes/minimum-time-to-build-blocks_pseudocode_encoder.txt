CLASS Solution  
    FUNCTION minBuildTime(blocks, split)  
        CALL heapify blocks TO arrange blocks as a min-heap  
        WHILE the LENGTH OF blocks GREATER THAN one  
            SET first TO the result of removing the smallest element FROM blocks  
            SET second TO the result of removing the smallest element FROM blocks  
            SET combined_time TO split PLUS second  
            ADD combined_time TO blocks maintaining min-heap property  
        END WHILE  
        RETURN element at position zero OF blocks  
    END FUNCTION  
END CLASS
