CLASS Solution  
    FUNCTION uniquePaths(m, n)  
        SET dp TO result of CALL createTwoDimensionalArrayWithValue(m, n, one)  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET value_at_position_i_j OF dp TO value_at_position_i_minus_one_j OF dp PLUS value_at_position_i_j_minus_one OF dp  
            END FOR  
        END FOR  
        RETURN value_at_position_m_minus_one_n_minus_one OF dp  
    END FUNCTION  
  
    FUNCTION createTwoDimensionalArrayWithValue(row_count, column_count, value)  
        SET new_array TO empty list  
        FOR i FROM zero TO row_count MINUS one  
            SET new_row TO empty list  
            FOR j FROM zero TO column_count MINUS one  
                APPEND value TO new_row  
            END FOR  
            APPEND new_row TO new_array  
        END FOR  
        RETURN new_array  
    END FUNCTION  
END CLASS
