CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        CALL initialize_2d_boolean_list WITH n AND SET TO f  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET element at position i,j of f TO False  
                IF character at position i of s EQUALS character at position j of s  
                    SET element at position i,j of f TO element at position i PLUS one,j MINUS one of f  
                    IF element at position i,j of f IS True AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN substring of s FROM position k TO position k PLUS mx  
    END FUNCTION  

    FUNCTION initialize_2d_boolean_list(n)  
        CREATE a two dimensional list f OF size n BY n  
        FOR each index i FROM zero TO n MINUS one  
            FOR each index j FROM zero TO n MINUS one  
                SET element at position i,j of f TO True  
            END FOR  
        END FOR  
        RETURN f  
    END FUNCTION  
END CLASS
