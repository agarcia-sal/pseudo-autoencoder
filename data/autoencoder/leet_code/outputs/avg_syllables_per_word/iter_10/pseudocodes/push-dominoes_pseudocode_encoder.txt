CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a list of size n FILLED WITH zero  
        
        SET force TO zero  
        FOR i FROM zero TO n MINUS one  
            IF element at position i of dominoes EQUALS the character R  
                SET force TO n  
            ELSE IF element at position i of dominoes EQUALS the character L  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER OF force MINUS one AND zero  
            END IF  
            INCREMENT element at position i OF forces BY force  
        END FOR  
        
        SET force TO zero  
        FOR i FROM n MINUS one TO zero STEP minus one  
            IF element at position i of dominoes EQUALS the character L  
                SET force TO n  
            ELSE IF element at position i of dominoes EQUALS the character R  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER OF force MINUS one AND zero  
            END IF  
            DECREMENT element at position i OF forces BY force  
        END FOR  
        
        SET result TO an empty list  
        FOR each f IN forces  
            IF f GREATER THAN zero  
                APPEND the character R TO result  
            ELSE IF f LESS THAN zero  
                APPEND the character L TO result  
            ELSE  
                APPEND the character dot TO result  
            END IF  
        END FOR  
        
        RETURN the concatenation of all elements IN result  
    END FUNCTION  
END CLASS
