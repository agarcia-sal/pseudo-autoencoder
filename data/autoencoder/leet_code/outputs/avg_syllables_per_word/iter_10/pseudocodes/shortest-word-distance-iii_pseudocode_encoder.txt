CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict, word1, word2)  
        SET min_distance TO infinity  
        SET prev_word1 TO minus one  
        SET prev_word2 TO minus one  
        SET same_word TO word1 EQUALS word2  
        
        FOR each index i FROM zero TO the LENGTH OF wordsDict MINUS one  
            SET word TO element at position i of wordsDict  
            
            IF word EQUALS word1  
                IF same_word  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO i  
            END IF  
            
            IF word EQUALS word2  
                IF NOT same_word  
                    SET prev_word2 TO i  
                END IF  
            END IF  
            
            IF prev_word1 NOT EQUALS minus one AND prev_word2 NOT EQUALS minus one  
                SET min_distance TO the MINIMUM OF min_distance AND the ABSOLUTE VALUE OF prev_word1 MINUS prev_word2  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
