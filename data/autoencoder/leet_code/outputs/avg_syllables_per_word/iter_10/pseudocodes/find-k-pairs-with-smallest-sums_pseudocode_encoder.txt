CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF nums1 EQUALS empty OR nums2 EQUALS empty OR k LESS THAN OR EQUAL TO zero  
            RETURN empty list  
        END IF  
        
        SET min_heap TO empty list  
        FOR index FROM zero TO the MINIMUM OF k MINUS one AND the LENGTH OF nums1 MINUS one  
            PUSH a tuple of the SUM OF element at position index of nums1 AND the element at position zero of nums2, index, zero INTO min_heap  
        END FOR  
        
        SET result TO empty list  
        WHILE min_heap IS NOT empty AND the LENGTH OF result LESS THAN k  
            POP the smallest tuple FROM min_heap AND UNPACK INTO _, i, j  
            APPEND a list of element at position i of nums1 AND element at position j of nums2 TO result  
            IF j PLUS one LESS THAN the LENGTH OF nums2  
                PUSH a tuple of the SUM OF element at position i of nums1 AND element at position j PLUS one of nums2, i, j PLUS one INTO min_heap  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
