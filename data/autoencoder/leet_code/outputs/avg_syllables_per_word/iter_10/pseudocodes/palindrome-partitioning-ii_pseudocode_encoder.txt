CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  

        CALL initialize_is_palindrome_table WITH n  
        FOR i FROM zero TO n MINUS one  
            SET element at position i of is_palindrome at position i TO True  
        END FOR  

        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET element at position start of is_palindrome at position end TO s at position start EQUALS s at position end  
                ELSE  
                    SET element at position start of is_palindrome at position end TO s at position start EQUALS s at position end AND element at position start PLUS one of is_palindrome at position end MINUS one  
                END IF  
            END FOR  
        END FOR  

        CALL initialize_min_cuts_list WITH n  
        FOR i FROM zero TO n MINUS one  
            IF element at position zero of is_palindrome at position i  
                SET element at position i of min_cuts TO zero  
            ELSE  
                SET element at position i of min_cuts TO positive infinity  
                FOR j FROM one TO i  
                    IF element at position j of is_palindrome at position i AND element at position j MINUS one of min_cuts PLUS one LESS THAN element at position i of min_cuts  
                        SET element at position i of min_cuts TO element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  

        RETURN element at position n MINUS one of min_cuts  
    END FUNCTION  

    FUNCTION initialize_is_palindrome_table(n)  
        SET is_palindrome TO a two-dimensional list with dimensions n BY n FILLED WITH False  
    END FUNCTION  

    FUNCTION initialize_min_cuts_list(n)  
        SET min_cuts TO a list with length n FILLED WITH zero  
    END FUNCTION  
END CLASS
