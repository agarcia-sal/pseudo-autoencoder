CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  
        
        SET below_twenty TO the list of words corresponding to numbers from zero up to nineteen including empty string at position zero  
        SET tens TO the list of words for multiples of ten from zero to ninety including empty strings at first two positions  
        SET thousands TO the list of scale words for thousands (empty string, Thousand, Million, Billion)  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            END IF  
            ELSE IF n LESS THAN twenty  
                RETURN the element at position n of below_twenty concatenated with a space  
            END ELSE IF  
            ELSE IF n LESS THAN one hundred  
                RETURN the element at position n DIVIDED BY ten of tens concatenated with a space concatenated with the result of helper called with n MODULO ten  
            END ELSE IF  
            ELSE  
                RETURN the element at position n DIVIDED BY one hundred of below_twenty concatenated with the word Hundred and a space concatenated with the result of helper called with n MODULO one hundred  
            END ELSE  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero  
            IF num MODULO one thousand NOT EQUALS zero  
                SET result TO the result of helper called with num MODULO one thousand concatenated with the element at position i of thousands concatenated with a space and then concatenated with result  
            END IF  
            SET num TO num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN the result string with any surrounding spaces removed  
    END FUNCTION  
END CLASS
