CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s, k)  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
        
        SET char_map TO empty map  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO LENGTH OF s MINUS one  
            SET character_at_right TO element at position right of s  
            SET element at key character_at_right of char_map TO right  
            
            IF LENGTH OF char_map GREATER THAN k  
                SET leftmost_char TO element in char_map WITH MINIMUM value among char_map values  
                SET left TO value at key leftmost_char of char_map PLUS one  
                REMOVE element at key leftmost_char FROM char_map  
            END IF  
            
            SET current_length TO right MINUS left PLUS one  
            IF current_length GREATER THAN max_length  
                SET max_length TO current_length  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
