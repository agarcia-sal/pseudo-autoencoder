CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value at position zero of values  
    SET i TO one  
    SET queue TO a new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE first element FROM queue AND SET TO node  
        IF i LESS THAN the LENGTH OF values AND value at position i of values IS NOT None  
            SET node.left TO new TreeNode with value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND value at position i of values IS NOT None  
            SET node.right TO new TreeNode with value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree with arguments p.left, q.left AND is_same_tree with arguments p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest with parameters root, k  
        SET stack TO empty list  
        SET current TO root  
        SET count TO zero  
        WHILE stack IS NOT empty OR current IS NOT None  
            WHILE current IS NOT None  
                APPEND current TO stack  
                SET current TO current.left  
            END WHILE  
            REMOVE last element FROM stack AND SET TO current  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN current.val  
            END IF  
            SET current TO current.right  
        END WHILE  
    END FUNCTION  
END CLASS
