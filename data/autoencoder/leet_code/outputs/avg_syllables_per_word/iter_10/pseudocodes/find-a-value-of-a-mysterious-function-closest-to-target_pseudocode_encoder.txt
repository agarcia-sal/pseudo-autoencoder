CLASS Solution  
    FUNCTION closestToTarget(arr, target)  
        SET seen TO an empty set  
        SET min_diff TO infinity  
        FOR each num IN arr  
            SET new_seen TO an empty set  
            FOR each x IN seen  
                APPEND bitwise AND of num AND x TO new_seen  
            END FOR  
            SET seen TO the union of new_seen AND a set containing num  
            FOR each val IN seen  
                SET current_diff TO the absolute value of val MINUS target  
                IF current_diff LESS THAN min_diff  
                    SET min_diff TO current_diff  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
