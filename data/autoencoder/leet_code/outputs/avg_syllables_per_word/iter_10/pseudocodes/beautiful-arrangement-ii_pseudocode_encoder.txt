CLASS Solution  
    FUNCTION constructArray(n, k)  
        SET result TO list of integers from one TO n MINUS k MINUS one  
        SET left TO n MINUS k  
        SET right TO n  
        WHILE left LESS THAN OR EQUAL TO right  
            IF left EQUALS right  
                APPEND left TO result  
            ELSE  
                APPEND left TO result  
                APPEND right TO result  
            END IF  
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
