CLASS ListNode  
    FUNCTION __init__(val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with val EQUALS element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values STARTING FROM position one  
        SET node TO new ListNode with val EQUALS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN (value at p1.val EQUALS value at p2.val) AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(head)  
        IF head IS None OR head.next IS None  
            RETURN head  
        END IF  
        SET dummy TO new ListNode with val EQUALS negative five thousand and one  
        SET dummy.next TO head  
        SET current TO head.next  
        SET head.next TO None  
        WHILE current IS NOT None  
            SET next_node TO current.next  
            SET prev TO dummy  
            WHILE prev.next IS NOT None AND value at prev.next.val LESS THAN value at current.val  
                SET prev TO prev.next  
            END WHILE  
            SET current.next TO prev.next  
            SET prev.next TO current  
            SET current TO next_node  
        END WHILE  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
