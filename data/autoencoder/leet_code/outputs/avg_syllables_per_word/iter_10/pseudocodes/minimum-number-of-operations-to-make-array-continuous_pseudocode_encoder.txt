CLASS Solution  
    FUNCTION minOperations(nums)  
        SET n TO the LENGTH OF nums  
        SET unique_nums TO the sorted list of the unique elements of nums  
        SET min_operations TO n  
        SET j TO zero  
        FOR i FROM zero TO the LENGTH OF unique_nums MINUS one  
            WHILE j LESS THAN the LENGTH OF unique_nums AND element at position j of unique_nums LESS THAN element at position i of unique_nums PLUS n  
                INCREMENT j BY one  
            END WHILE  
            SET min_operations TO the smaller of min_operations AND n MINUS (j MINUS i)  
        END FOR  
        RETURN min_operations  
    END FUNCTION  
END CLASS
