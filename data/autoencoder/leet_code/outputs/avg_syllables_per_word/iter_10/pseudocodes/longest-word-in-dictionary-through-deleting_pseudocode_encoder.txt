CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET it TO iterator of s  
            FOR each character IN word  
                IF character NOT IN it  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SORT dictionary BY length of element at position zero DECREASING AND BY element at position zero IN INCREASING LEXICOGRAPHICAL ORDER  
        
        FOR each word IN dictionary  
            IF is_subsequence(word, s) EQUALS true  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN empty string  
    END FUNCTION  
END CLASS
