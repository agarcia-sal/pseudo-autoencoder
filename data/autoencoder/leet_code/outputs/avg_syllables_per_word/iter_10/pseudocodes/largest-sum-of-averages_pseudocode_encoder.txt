CLASS Solution  
    FUNCTION largestSumOfAverages(nums, k)  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of zeroes of LENGTH n PLUS one  
        FOR index FROM zero TO n MINUS one  
            SET element at position index PLUS one of prefix_sum TO element at position index of prefix_sum PLUS element at position index of nums  
        END FOR  
        
        FUNCTION average(i, j)  
            SET total TO element at position j PLUS one of prefix_sum MINUS element at position i of prefix_sum  
            SET count TO j MINUS i PLUS one  
            RETURN total DIVIDED BY count  
        END FUNCTION  
        
        SET dp TO a list containing n PLUS one elements EACH being a list of zeroes of LENGTH k PLUS one  
        
        FOR index FROM zero TO n MINUS one  
            SET element at position index at position one of dp TO call FUNCTION average WITH parameters index AND n MINUS one  
        END FOR  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    SET candidate TO element at position j at position partitions MINUS one of dp PLUS call FUNCTION average WITH parameters i AND j MINUS one  
                    IF candidate GREATER THAN element at position i at position partitions of dp  
                        SET element at position i at position partitions of dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero at position k of dp  
    END FUNCTION  
END CLASS
