CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO an empty dictionary  
        FOR each index i FROM zero TO LENGTH OF s MINUS one  
            SET character char TO element at position i of s  
            SET last_occurrence at key char TO i  
        END FOR  
        
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO an empty list  
        
        FOR each index i FROM zero TO LENGTH OF s MINUS one  
            SET character char TO element at position i of s  
            SET end TO the greater value BETWEEN end AND value at key char in last_occurrence  
            
            IF i EQUALS end THEN  
                APPEND end MINUS start PLUS one TO partition_sizes  
                SET start TO i PLUS one  
            END IF  
        END FOR  
        
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
