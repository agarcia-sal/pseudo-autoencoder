CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulted to zero, left defaulted to none, right defaulted to none  
        SET val attribute TO val  
        SET left attribute TO left  
        SET right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at position zero of values  
    SET i TO one  
    SET queue TO a new deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed FROM queue  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.left TO new TreeNode with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.right TO new TreeNode with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree with p.left and q.left AND is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree with parameter root  
        IF root EQUALS none  
            RETURN none  
        END IF  
        SET root.left TO call pruneTree with root.left  
        SET root.right TO call pruneTree with root.right  
        IF root.val EQUALS zero AND root.left EQUALS none AND root.right EQUALS none  
            RETURN none  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
