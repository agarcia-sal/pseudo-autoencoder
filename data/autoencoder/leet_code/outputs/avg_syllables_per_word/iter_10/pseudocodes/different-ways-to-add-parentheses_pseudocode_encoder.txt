CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS character PLUS  
                        APPEND l PLUS r TO results  
                    ELSE IF operator EQUALS character MINUS  
                        APPEND l MINUS r TO results  
                    ELSE IF operator EQUALS character MULTIPLY  
                        APPEND l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        FUNCTION helper(sub_expr)  
            IF sub_expr IS composed only of digits  
                RETURN list containing the integer value of sub_expr  
            END IF  
            
            SET results TO empty list  
            FOR index FROM zero TO the LENGTH OF sub_expr MINUS one  
                IF the element at position index of sub_expr IS one of the characters PLUS, MINUS, OR MULTIPLY  
                    SET left_results TO helper(substring of sub_expr from start TO index MINUS one)  
                    SET right_results TO helper(substring of sub_expr from index PLUS one TO end)  
                    EXTEND results BY the result of compute(left_results, right_results, element at position index of sub_expr)  
                END IF  
            END FOR  
            
            RETURN results  
        END FUNCTION  
        
        RETURN helper(expression)  
    END FUNCTION  
END CLASS
