CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO the FRAGMENT from the START TO position d EXCLUSIVE  
                SET right TO the FRAGMENT from position d TO the END  
                IF (left DOES NOT START WITH zero OR left EQUALS zero) AND (right DOES NOT END WITH zero)  
                    YIELD left PLUS IF d NOT EQUALS n THEN '.' ELSE empty STRING PLUS right  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO the SUBSTRING OF s FROM position one TO the SECOND TO LAST position  
        RETURN the LIST formed BY  
            FOR i FROM one TO the LENGTH OF s MINUS one  
                FOR each a, b IN the CARTESIAN PRODUCT OF make(THE SUBSTRING OF s FROM START TO position i EXCLUSIVE) AND make(THE SUBSTRING OF s FROM position i TO END)  
                    YIELD OPEN PARENTHESIS PLUS a PLUS COMMA PLUS SPACE PLUS b PLUS CLOSE PARENTHESIS  
                END FOR  
            END FOR  
    END FUNCTION  
END CLASS
