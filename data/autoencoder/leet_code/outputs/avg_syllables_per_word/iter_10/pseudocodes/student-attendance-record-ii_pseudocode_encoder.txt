CLASS Solution  
    FUNCTION checkRecord(n)  
        SET MOD TO one billion PLUS seven  
        
        CALL initialize_dp WITH n RETURNING dp  
        
        SET element at position zero of dp at position zero at position zero TO one  
        
        FOR i FROM one TO n  
            FOR j FROM zero TO one  
                FOR k FROM zero TO two  
                    SET element at position zero of dp at position i at position j TO element at position zero of dp at position i at position j PLUS element at position k of dp at position i MINUS one at position j  
                    SET element at position zero of dp at position i at position j TO element at position zero of dp at position i at position j MODULO MOD  
                    
                    IF j EQUALS zero  
                        SET element at position zero of dp at position i at position one TO element at position zero of dp at position i at position one PLUS element at position k of dp at position i MINUS one at position zero  
                        SET element at position zero of dp at position i at position one TO element at position zero of dp at position i at position one MODULO MOD  
                    END IF  
                    
                    IF k LESS THAN two  
                        SET element at position k PLUS one of dp at position i at position j TO element at position k PLUS one of dp at position i at position j PLUS element at position k of dp at position i MINUS one at position j  
                        SET element at position k PLUS one of dp at position i at position j TO element at position k PLUS one of dp at position i at position j MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR j FROM zero TO one  
            FOR k FROM zero TO two  
                SET result TO result PLUS element at position k of dp at position n at position j  
                SET result TO result MODULO MOD  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
    
    
    FUNCTION initialize_dp(n)  
        CREATE dp AS list  
        FOR i FROM zero TO n  
            CREATE level_one AS list  
            FOR j FROM zero TO one  
                CREATE level_two AS list  
                FOR k FROM zero TO two  
                    APPEND zero TO level_two  
                END FOR  
                APPEND level_two TO level_one  
            END FOR  
            APPEND level_one TO dp  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
