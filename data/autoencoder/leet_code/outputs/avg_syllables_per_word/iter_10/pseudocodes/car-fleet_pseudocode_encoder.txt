CLASS Solution  
    FUNCTION carFleet(target, position, speed)  
        SET list_of_pairs TO the result of sorting the pairs of elements at matching positions in position and speed in descending order by the first element of the pair  
        SET time_to_target TO an empty list  
        FOR each pair consisting of p and s IN list_of_pairs  
            SET time_to_reach TO target MINUS p DIVIDED BY s  
            APPEND time_to_reach TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
