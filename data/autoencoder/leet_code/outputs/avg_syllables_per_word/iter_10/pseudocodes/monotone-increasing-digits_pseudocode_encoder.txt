CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO list of characters of string representation of n  
        SET length TO the LENGTH OF digits  
        SET marker TO length  
        FOR index FROM length MINUS ONE DOWN TO one  
            IF digit at position index of digits LESS THAN digit at position index MINUS one of digits  
                SET digit at position index MINUS one of digits TO string of integer value of digit at position index MINUS one of digits MINUS one  
                SET marker TO index  
            END IF  
        END FOR  
        FOR index FROM marker TO length MINUS one  
            SET digit at position index of digits TO character nine  
        END FOR  
        RETURN integer value of concatenation of all elements in digits  
    END FUNCTION  
END CLASS
