CLASS Solution  
    FUNCTION orderlyQueue(s, k)  
        IF k GREATER THAN 1  
            SET sorted_characters TO characters of s arranged in ascending order  
            SET result_string TO concatenation of sorted_characters in their order  
            RETURN result_string  
        ELSE  
            SET n TO the LENGTH OF s  
            SET smallest TO s  
            FOR index FROM 1 TO n MINUS 1  
                SET rotated TO concatenation of substring of s from position index TO end AND substring of s from position zero TO position index MINUS one  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
