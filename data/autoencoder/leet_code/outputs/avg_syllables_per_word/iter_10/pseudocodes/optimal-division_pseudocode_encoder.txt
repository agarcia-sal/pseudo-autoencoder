CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the STRING REPRESENTATION OF the element at position zero of nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the STRING REPRESENTATION OF the element at position zero of nums CONCATENATED WITH the CHARACTER FORWARD SLASH CONCATENATED WITH the STRING REPRESENTATION OF the element at position one of nums  
        END IF  
        RETURN the STRING REPRESENTATION OF the element at position zero of nums CONCATENATED WITH the CHARACTER FORWARD SLASH CONCATENATED WITH the CHARACTER LEFT PARENTHESIS CONCATENATED WITH the STRING REPRESENTATION OF the element at position one of nums  
            CONCATENATED WITH the RESULT OF FOR each element STARTING FROM position two TO the LAST ELEMENT OF nums CONCATENATED WITH the CHARACTER FORWARD SLASH CONCATENATED WITH the STRING REPRESENTATION OF that element  
            CONCATENATED WITH the CHARACTER RIGHT PARENTHESIS  
    END FUNCTION  
END CLASS
