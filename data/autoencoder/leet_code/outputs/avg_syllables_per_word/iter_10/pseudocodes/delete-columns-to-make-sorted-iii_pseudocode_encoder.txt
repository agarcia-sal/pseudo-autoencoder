CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero of strs  
        SET dp TO a list of n elements each SET TO one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF for each k FROM zero TO m MINUS one the element at position j of element at position k of strs LESS THAN OR EQUAL TO the element at position i of element at position k of strs  
                    SET dp at position i TO the maximum OF dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        SET longest_non_decreasing_subsequence TO the maximum element of dp  
        SET result TO n MINUS longest_non_decreasing_subsequence  
        RETURN result  
    END FUNCTION  
END CLASS
