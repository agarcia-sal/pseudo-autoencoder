CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with value at position zero of values  
    SET p TO head  
    FOR each val IN values starting from position one TO the end  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the value at p1 EQUALS the value at p2 AND is_same_list with parameters the next of p1 AND the next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween(self, head, left, right)  
        IF head IS none OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO new ListNode with zero  
        SET dummy.next TO head  
        SET prev TO dummy  
        FOR index FROM one TO left MINUS one  
            SET prev TO prev.next  
        END FOR  
        SET current TO prev.next  
        FOR index FROM one TO right MINUS left  
            SET next_node TO current.next  
            SET current.next TO next_node.next  
            SET next_node.next TO prev.next  
            SET prev.next TO next_node  
        END FOR  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
