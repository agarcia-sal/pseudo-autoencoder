CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        SET graph TO an empty mapping from number TO set  
        FOR index FROM one TO n  
            SET element at position index of graph TO an empty set  
        END FOR  
        
        FOR each pair u and v IN edges  
            ADD v TO the set associated with u in graph  
            ADD u TO the set associated with v in graph  
        END FOR  
        
        SET min_degree TO infinity  
        SET trio_found TO false  
        
        FOR i FROM one TO n  
            FOR j FROM i PLUS one TO n  
                FOR k FROM j PLUS one TO n  
                    IF j IS AN ELEMENT OF the set associated with i in graph AND k IS AN ELEMENT OF the set associated with i in graph AND k IS AN ELEMENT OF the set associated with j in graph  
                        SET trio_found TO true  
                        SET trio_degree TO the LENGTH OF the set associated with i in graph PLUS the LENGTH OF the set associated with j in graph PLUS the LENGTH OF the set associated with k in graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        IF trio_found  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
