CLASS Solution  
    FUNCTION catMouseGame(graph)  
        SET DRAW TO zero  
        SET MOUSE_WIN TO one  
        SET CAT_WIN TO two  
        SET n TO the LENGTH OF graph  
        
        FUNCTION dfs(turn, mouse_pos, cat_pos)  
            IF turn EQUALS two MULTIPLIED BY n  
                RETURN DRAW  
            END IF  
            IF mouse_pos EQUALS zero  
                RETURN MOUSE_WIN  
            END IF  
            IF mouse_pos EQUALS cat_pos  
                RETURN CAT_WIN  
            END IF  
            
            IF turn MODULO two EQUALS zero  
                SET current_player TO the STRING mouse  
            ELSE  
                SET current_player TO the STRING cat  
            END IF  
            
            IF current_player EQUALS the STRING mouse  
                SET result TO CAT_WIN  
                FOR each next_pos IN the ELEMENTS AT POSITION mouse_pos OF graph  
                    SET next_result TO dfs(turn PLUS one, next_pos, cat_pos)  
                    IF next_result EQUALS MOUSE_WIN  
                        RETURN MOUSE_WIN  
                    ELSE IF next_result EQUALS DRAW  
                        SET result TO DRAW  
                    END IF  
                END FOR  
                RETURN result  
            ELSE  
                SET result TO MOUSE_WIN  
                FOR each next_pos IN the ELEMENTS AT POSITION cat_pos OF graph  
                    IF next_pos EQUALS zero  
                        CONTINUE TO next iteration  
                    END IF  
                    SET next_result TO dfs(turn PLUS one, mouse_pos, next_pos)  
                    IF next_result EQUALS CAT_WIN  
                        RETURN CAT_WIN  
                    ELSE IF next_result EQUALS DRAW  
                        SET result TO DRAW  
                    END IF  
                END FOR  
                RETURN result  
            END IF  
        END FUNCTION  
        
        RETURN dfs(zero, one, two)  
    END FUNCTION  
END CLASS
