CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO a new mapping with default empty list  
        FOR each pair u, v IN edges  
            APPEND v TO the list at key u in graph  
            APPEND u TO the list at key v in graph  
        END FOR  
  
        FUNCTION dfs(node, parent, path, visited)  
            IF node IS IN visited  
                SET cycle_start TO the position of node in path  
                RETURN the sublist of path FROM cycle_start TO the end  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN the list at key node in graph  
                IF neighbor IS NOT EQUAL TO parent  
                    SET cycle TO dfs(neighbor, node, path, visited)  
                    IF cycle IS NOT NONE  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE last element FROM path  
            RETURN NONE  
        END FUNCTION  
  
        SET visited TO an empty set  
        SET cycle TO NONE  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited  
                SET cycle TO dfs(node, MINUS one, an empty list, visited)  
                IF cycle IS NOT NONE  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
  
        SET distance TO a list of zeros of length n  
        SET queue TO a new double-ended queue initialized with elements of cycle  
        SET visited TO a set initialized with elements of cycle  
  
        SET level TO zero  
        WHILE queue IS NOT empty  
            SET length TO the number of elements in queue  
            FOR index FROM one TO length  
                REMOVE first element FROM queue AND SET to node  
                SET distance at position node TO level  
                FOR each neighbor IN the list at key node in graph  
                    IF neighbor IS NOT IN visited  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  
  
        RETURN distance  
    END FUNCTION  
END CLASS
