CLASS Solution  
    FUNCTION isInterleave(s1, s2, s3)  
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 DOES NOT EQUAL the LENGTH OF s3  
            RETURN False  
        END IF  

        SET dp TO a list of False repeated the LENGTH OF s2 PLUS one times  
        SET the element at position zero of dp TO True  

        FOR i FROM zero TO the LENGTH OF s1  
            FOR j FROM zero TO the LENGTH OF s2  
                IF i IS GREATER THAN zero  
                    SET the element at position j of dp TO the element at position j of dp AND the element at position i MINUS one of s1 EQUALS the element at position i PLUS j MINUS one of s3  
                END IF  
                IF j IS GREATER THAN zero  
                    SET the element at position j of dp TO the element at position j of dp OR (the element at position j MINUS one of dp AND the element at position j MINUS one of s2 EQUALS the element at position i PLUS j MINUS one of s3)  
                END IF  
            END FOR  
        END FOR  

        RETURN the element at position the LENGTH OF s2 of dp  
    END FUNCTION  
END CLASS
