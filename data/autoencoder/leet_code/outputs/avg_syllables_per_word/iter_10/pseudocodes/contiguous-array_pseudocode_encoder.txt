CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO dictionary with key zero and value negative one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each index i AND element num IN nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count IS IN prefix_sum_index  
                SET max_length TO maximum of max_length AND i MINUS value at key count IN prefix_sum_index  
            ELSE  
                SET value at key count IN prefix_sum_index TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
