CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO the sorted list of pairs where each pair is the start element of an interval and its index in intervals  
        SET result TO an empty list  
        FOR each pair of ignore_value and end_element IN intervals  
            SET idx TO the position found by locating the leftmost place to insert a pair starting with end_element in start_times  
            IF idx IS LESS THAN the LENGTH OF start_times  
                APPEND the second element of the pair at position idx in start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
