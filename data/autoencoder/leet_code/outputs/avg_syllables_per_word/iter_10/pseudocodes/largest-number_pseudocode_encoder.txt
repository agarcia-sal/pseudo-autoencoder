CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO a new empty list  
        FOR each number IN nums  
            APPEND the string representation of number TO nums_str  
        END FOR  
        
        FUNCTION compare(x, y)  
            IF the concatenation of x FOLLOWED BY y IS GREATER THAN the concatenation of y FOLLOWED BY x  
                RETURN minus one  
            ELSE IF the concatenation of x FOLLOWED BY y IS LESS THAN the concatenation of y FOLLOWED BY x  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str USING compare as the comparator  
        
        IF the element at position zero of nums_str EQUALS the string zero  
            RETURN the string zero  
        END IF  
        
        SET result TO the concatenation of all elements of nums_str in order  
        RETURN result  
    END FUNCTION  
END CLASS
