CLASS ListNode  
    FUNCTION __init__ with parameters val, next  
        SET val TO val  
        SET next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN values starting from position one to the end  
        SET node TO a new ListNode initialized with val  
        SET the next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1, p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the result of (the val of p1 EQUALS the val of p2) AND is_same_list(the next of p1, the next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers with parameters l1, l2  
        SET dummy_head TO a new ListNode initialized with zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry IS NOT zero  
            IF l1 IS NOT None  
                SET val1 TO the val of l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            
            IF l2 IS NOT None  
                SET val2 TO the val of l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET the next of current TO a new ListNode initialized with total MODULO ten  
            SET current TO the next of current  
            
            IF l1 IS NOT None  
                SET l1 TO the next of l1  
            END IF  
            
            IF l2 IS NOT None  
                SET l2 TO the next of l2  
            END IF  
        END WHILE  
        
        RETURN the next of dummy_head  
    END FUNCTION  
END CLASS
