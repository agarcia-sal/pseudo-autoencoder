CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero, left EQUALS none, right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode WITH val EQUALS element at position zero of values  
    SET i TO one  
    SET queue TO a new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode WITH val EQUALS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode WITH val EQUALS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH ARGUMENTS p.left, q.left AND is_same_tree WITH ARGUMENTS p.right, q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION buildTree WITH PARAMETERS self, inorder, postorder  
        IF inorder IS empty OR postorder IS empty  
            RETURN none  
        END IF  
        SET root_val TO last element removed from postorder  
        SET root TO new TreeNode WITH val EQUALS root_val  
        SET root_index TO the POSITION OF root_val IN inorder  
        SET root.right TO CALL self.buildTree WITH ARGUMENTS the elements from position root_index PLUS one TO end of inorder, postorder  
        SET root.left TO CALL self.buildTree WITH ARGUMENTS the elements from start TO position root_index MINUS one of inorder, postorder  
        RETURN root  
    END FUNCTION  
END CLASS
