CLASS Solution  
    FUNCTION multiply(num1, num2)  
        IF num1 EQUALS "0" OR num2 EQUALS "0"  
            RETURN "0"  
        END IF  

        SET result TO a list of zeros with the length EQUAL TO the LENGTH OF num1 PLUS the LENGTH OF num2  

        SET num1 TO the reversal of num1  
        SET num2 TO the reversal of num2  

        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the numerical value of character at position i of num1 MINUS the numerical value of character "0" MULTIPLIED BY the numerical value of character at position j of num2 MINUS the numerical value of character "0"  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS element at position p1 of result  

                SET element at position p1 of result TO sum_ MODULO ten  
                INCREMENT element at position p2 of result BY sum_ DIVIDED BY ten  
            END FOR  
        END FOR  

        SET result_str TO the concatenation of string representations of elements in the reversal of result  
        RETURN result_str with all leading "0" characters removed  
    END FUNCTION  
END CLASS
