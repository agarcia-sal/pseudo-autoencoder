CLASS Solution  
    FUNCTION maxEnvelopes(envelopes)  
        SORT envelopes BY element at position zero IN ascending order AND element at position one IN descending order  
        SET dp TO empty list  
        FOR each element consisting of underscore and h IN envelopes  
            SET idx TO the position found by binary search for h IN dp that preserves sorted order  
            IF idx EQUALS the LENGTH OF dp  
                APPEND h TO dp  
            ELSE  
                SET element at position idx of dp TO h  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
