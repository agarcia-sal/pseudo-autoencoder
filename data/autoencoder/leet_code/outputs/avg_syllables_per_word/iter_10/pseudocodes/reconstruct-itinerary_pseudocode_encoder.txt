CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a new dictionary with default value as empty list  
        FOR each pair (frm, to) IN the reverse order of the sorted tickets  
            APPEND to TO the list at key frm in graph  
        END FOR  
        SET itinerary TO an empty list  
        FUNCTION visit(airport)  
            WHILE the list at key airport in graph is not empty  
                CALL visit WITH the last element removed FROM the list at key airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        CALL visit WITH the string "JFK"  
        RETURN the reverse of itinerary  
    END FUNCTION  
END CLASS
