CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO plus symbol  
            WHILE the LENGTH OF s IS GREATER THAN zero  
                SET char TO the first element of s REMOVED FROM s  
                IF char IS a digit character  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                IF char EQUALS opening parenthesis  
                    SET num TO the RESULT OF calling helper WITH s  
                END IF  
                IF char EQUALS plus symbol OR char EQUALS minus symbol OR char EQUALS multiplication symbol OR char EQUALS division symbol OR char EQUALS opening parenthesis OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS plus symbol  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus symbol  
                        APPEND the negation of num TO stack  
                    ELSE IF sign EQUALS multiplication symbol  
                        SET the last element of stack TO the last element of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS division symbol  
                        SET the last element of stack TO the integer RESULT OF the last element of stack DIVIDED BY num WITH truncation towards zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS closing parenthesis  
                    BREAK the WHILE loop  
                END IF  
            END WHILE  
            RETURN the SUM of elements in stack  
        END FUNCTION  
        SET s TO the list of characters of s  
        RETURN the RESULT OF calling helper WITH s  
    END FUNCTION  
END CLASS
