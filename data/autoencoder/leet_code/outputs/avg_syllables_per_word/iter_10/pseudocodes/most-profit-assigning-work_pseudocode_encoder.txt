CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO empty list  
        FOR index FROM zero TO the LENGTH OF difficulty MINUS one  
            APPEND element at position index of difficulty AND element at position index of profit AS a pair TO jobs  
        END FOR  
        SORT jobs BY the first element of each pair (difficulty) IN ascending order  
        SORT worker IN ascending order  
        
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the LENGTH OF jobs  
        
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND the first element of element at position job_index of jobs LESS THAN OR EQUAL TO ability  
                IF the second element of element at position job_index of jobs GREATER THAN best_profit  
                    SET best_profit TO the second element of element at position job_index of jobs  
                END IF  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
