CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        SET char_count TO result of calling an auxiliary FUNCTION defaultdict of integer counts  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the count at element at position right of s IN char_count BY one  
            
            WHILE the NUMBER OF keys IN char_count GREATER THAN two  
                DECREMENT the count at element at position left of s IN char_count BY one  
                IF the count at element at position left of s IN char_count EQUALS zero  
                    REMOVE the element at position left of s FROM char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_length TO the GREATER OF max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
