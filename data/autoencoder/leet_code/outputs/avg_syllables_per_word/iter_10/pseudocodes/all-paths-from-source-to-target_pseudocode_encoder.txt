CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        SET q TO an empty queue initialized with a list containing zero  
        SET ans TO an empty list  
        WHILE q is not empty  
            SET path TO the first element removed FROM q  
            SET u TO the last element of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            FOR each v IN the element at position u of graph  
                APPEND a new list formed by concatenating path and v TO q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
