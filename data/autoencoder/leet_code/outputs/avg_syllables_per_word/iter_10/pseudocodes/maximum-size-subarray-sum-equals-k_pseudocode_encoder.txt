CLASS Solution  
    FUNCTION maxSubArrayLen(nums, k)  
        SET cumulative_sum_index TO a dictionary with key zero mapped to value minus one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        
        FOR index, num IN enumeration of nums  
            SET cumulative_sum TO cumulative_sum PLUS num  
            
            IF cumulative_sum MINUS k IS A KEY IN cumulative_sum_index  
                SET max_length TO the GREATER OF max_length AND index MINUS the VALUE AT key cumulative_sum MINUS k in cumulative_sum_index  
            END IF  
            
            IF cumulative_sum IS NOT A KEY IN cumulative_sum_index  
                SET the VALUE AT key cumulative_sum in cumulative_sum_index TO index  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
