CLASS Solution  
    FUNCTION numWays(steps, arrLen)  
        SET MODULO TO one billion PLUS seven  
        SET maximum index TO the MINIMUM OF steps DIVIDED BY two AND arrLen MINUS one  
        CALL initialize_dp_list WITH maximum index PLUS one RETURNING dp  
        SET element at position zero of dp TO one  
        
        FOR each step FROM one TO steps  
            CALL copy_list WITH dp RETURNING previous_dp  
            FOR each index FROM zero TO maximum index  
                SET element at position index of dp TO element at position index of previous_dp  
                IF index GREATER THAN zero  
                    SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS one of previous_dp  
                    SET element at position index of dp TO element at position index of dp MODULO MODULO  
                END IF  
                IF index LESS THAN maximum index  
                    SET element at position index of dp TO element at position index of dp PLUS element at position index PLUS one of previous_dp  
                    SET element at position index of dp TO element at position index of dp MODULO MODULO  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS  

FUNCTION initialize_dp_list(size)  
    CREATE list dp WITH size elements ALL SET TO zero  
    RETURN dp  
END FUNCTION  

FUNCTION copy_list(original_list)  
    CREATE new list copied_list AND COPY all elements FROM original_list  
    RETURN copied_list  
END FUNCTION
