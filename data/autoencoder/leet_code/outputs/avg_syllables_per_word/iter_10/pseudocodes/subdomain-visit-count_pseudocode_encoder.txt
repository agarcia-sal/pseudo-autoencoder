CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO the result of calling defaultdict with parameter integer  
        
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain INTO count_string AND domain_string by the first space  
            SET count TO count_string converted to integer  
            SPLIT domain_string INTO subdomains by the period character  
            
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one  
                JOIN the elements of subdomains from position index TO the end using the period character INTO subdomain  
                INCREMENT the value at key subdomain in domain_count BY count  
            END FOR  
        END FOR  
        
        RETURN a list constructed by FOR each domain and count IN the items of domain_count, creating a string of count FOLLOWED BY space FOLLOWED BY domain  
    END FUNCTION  
END CLASS
