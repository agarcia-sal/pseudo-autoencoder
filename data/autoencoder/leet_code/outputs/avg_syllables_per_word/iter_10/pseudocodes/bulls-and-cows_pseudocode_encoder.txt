CLASS Solution
    FUNCTION getHint(secret, guess)
        SET bulls TO zero
        SET cows TO zero

        CALL initialize_defaultdict_secret_count
        CALL initialize_defaultdict_guess_count

        FOR each pair s and g IN parallel elements of secret and guess
            IF s EQUALS g
                INCREMENT bulls BY one
            ELSE
                INCREMENT value at key s IN secret_count BY one
                INCREMENT value at key g IN guess_count BY one
            END IF
        END FOR

        FOR each digit IN keys of secret_count
            INCREMENT cows BY smaller value BETWEEN value at key digit IN secret_count AND value at key digit IN guess_count
        END FOR

        RETURN concatenation of bulls followed by letter A followed by cows followed by letter B
    END FUNCTION
END CLASS
