CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL nums SORT  
        SET n TO LENGTH OF nums  
        SET first TO element at position zero of nums  
        FOR i FROM one TO n MINUS one  
            SET k TO element at position i of nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at position i of nums  
                CONTINUE  
            END IF  
            SET count TO CALL Counter WITH nums  
            SET arr TO empty list  
            FOR each x IN nums  
                IF element x of count EQUALS zero  
                    CONTINUE  
                END IF  
                IF element x PLUS two MULTIPLIED BY k of count EQUALS zero  
                    BREAK  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT element x of count BY one  
                DECREMENT element x PLUS two MULTIPLIED BY k of count BY one  
            END FOR  
            IF LENGTH OF arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN empty list  
    END FUNCTION  
END CLASS
