CLASS Solution  
    FUNCTION numberOfWays(numPeople)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a list of zeros with LENGTH equal to numPeople PLUS one  
        SET element at position zero of dp TO one  
        SET element at position two of dp TO one  
        FOR index i FROM four TO numPeople IN STEPS OF two  
            FOR index j FROM zero TO i MINUS one IN STEPS OF two  
                SET element at position i of dp TO the sum of element at position i of dp PLUS the product of element at position j of dp AND element at position i MINUS two MINUS j of dp  
                SET element at position i of dp TO element at position i of dp MODULO MOD  
            END FOR  
        END FOR  
        RETURN element at position numPeople of dp  
    END FUNCTION  
END CLASS
