CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty  
            RETURN zero  
        END IF  
        SORT intervals BY start time of each meeting  
        SET min_heap TO empty list  
        FOR each meeting IN intervals  
            IF min_heap NOT EQUALS empty AND start time of meeting GREATER THAN OR EQUAL TO first element of min_heap  
                REMOVE first element FROM min_heap  
            END IF  
            APPEND end time of meeting TO min_heap MAINTAINING min_heap AS MINIMUM HEAP  
        END FOR  
        RETURN LENGTH OF min_heap  
    END FUNCTION  
END CLASS
