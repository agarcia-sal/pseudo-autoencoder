CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value as element at position zero of values  
    SET i TO one  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED from queue  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.left TO new TreeNode with value as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.right TO new TreeNode with value as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self, root)  
        IF root IS none THEN  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO new double-ended queue containing root  
        SET left_to_right TO true  
        WHILE queue IS NOT empty  
            SET level_size TO LENGTH OF queue  
            SET current_level TO new empty double-ended queue  
            FOR each index FROM zero TO level_size MINUS one  
                SET node TO first element REMOVED from queue  
                IF left_to_right IS true THEN  
                    APPEND node.val TO current_level  
                ELSE  
                    INSERT node.val AT FRONT OF current_level  
                END IF  
                IF node.left IS NOT none THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT none THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND list converted from current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
