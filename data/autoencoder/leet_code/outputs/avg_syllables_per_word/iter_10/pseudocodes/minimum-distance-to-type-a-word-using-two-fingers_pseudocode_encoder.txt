CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS None  
                RETURN zero  
            END IF  
            SET x1 AND y1 TO the QUOTIENT AND REMAINDER OF the ORDINAL VALUE OF c1 MINUS the ORDINAL VALUE OF character A DIVIDED BY six  
            SET x2 AND y2 TO the QUOTIENT AND REMAINDER OF the ORDINAL VALUE OF c2 MINUS the ORDINAL VALUE OF character A DIVIDED BY six  
            RETURN the ABSOLUTE VALUE OF x1 MINUS x2 PLUS the ABSOLUTE VALUE OF y1 MINUS y2  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the LENGTH OF word  
                RETURN zero  
            END IF  
            SET c TO the ELEMENT AT position i OF word  
            SET d1 TO distance(f1, c) PLUS dp(i PLUS one, c, f2)  
            SET d2 TO distance(f2, c) PLUS dp(i PLUS one, f1, c)  
            RETURN the MINIMUM OF d1 AND d2  
        END FUNCTION  
        
        RETURN dp(zero, None, None)  
    END FUNCTION  
END CLASS
