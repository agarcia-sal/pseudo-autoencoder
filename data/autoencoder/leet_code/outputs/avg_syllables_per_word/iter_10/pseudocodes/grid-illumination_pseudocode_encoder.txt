CLASS Solution  
    FUNCTION gridIllumination(n, lamps, queries)  
        INITIALIZE row_count TO a mapping from integer TO integer with default zero  
        INITIALIZE col_count TO a mapping from integer TO integer with default zero  
        INITIALIZE diag_count TO a mapping from integer TO integer with default zero  
        INITIALIZE anti_diag_count TO a mapping from integer TO integer with default zero  
        
        INITIALIZE lamp_positions TO an empty set  
        
        FOR each pair r, c IN lamps  
            IF pair r, c NOT IN lamp_positions  
                ADD pair r, c TO lamp_positions  
                INCREMENT value at key r IN row_count BY one  
                INCREMENT value at key c IN col_count BY one  
                INCREMENT value at key r MINUS c IN diag_count BY one  
                INCREMENT value at key r PLUS c IN anti_diag_count BY one  
            END IF  
        END FOR  
        
        SET directions TO the list of pairs negative one MINUS one, negative one zero, negative one one, zero MINUS one, zero zero, zero one, one MINUS one, one zero, one one  
        
        INITIALIZE result TO an empty list  
        
        FOR each pair r, c IN queries  
            IF value at key r IN row_count GREATER THAN zero OR value at key c IN col_count GREATER THAN zero OR value at key r MINUS c IN diag_count GREATER THAN zero OR value at key r PLUS c IN anti_diag_count GREATER THAN zero  
                APPEND one TO result  
            ELSE  
                APPEND zero TO result  
            END IF  
            
            FOR each pair dr, dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF pair nr, nc IN lamp_positions  
                    REMOVE pair nr, nc FROM lamp_positions  
                    DECREMENT value at key nr IN row_count BY one  
                    DECREMENT value at key nc IN col_count BY one  
                    DECREMENT value at key nr MINUS nc IN diag_count BY one  
                    DECREMENT value at key nr PLUS nc IN anti_diag_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
