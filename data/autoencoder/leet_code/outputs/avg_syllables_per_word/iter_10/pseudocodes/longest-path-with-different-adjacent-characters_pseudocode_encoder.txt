CLASS Solution  
    FUNCTION longestPath(parent, s)  
        FROM collections IMPORT defaultdict  
        SET tree TO defaultdict OF list  
        FOR each index i FROM zero TO LENGTH OF parent MINUS one  
            SET p TO element at position i of parent  
            IF p NOT EQUALS negative one  
                APPEND i TO element at position p of tree  
            END IF  
        END FOR  
        SET self dot result TO one  
        
        FUNCTION dfs(node)  
            SET max1 TO zero  
            SET max2 TO zero  
            FOR each child IN element at position node of tree  
                SET child_length TO dfs(child)  
                IF character at position child of s NOT EQUALS character at position node of s  
                    IF child_length GREATER THAN max1  
                        SET max2 TO max1  
                        SET max1 TO child_length  
                    ELSE IF child_length GREATER THAN max2  
                        SET max2 TO child_length  
                    END IF  
                END IF  
            END FOR  
            SET self dot result TO maximum OF self dot result AND max1 PLUS max2 PLUS one  
            RETURN max1 PLUS one  
        END FUNCTION  
        
        CALL dfs WITH zero  
        RETURN self dot result  
    END FUNCTION  
END CLASS
