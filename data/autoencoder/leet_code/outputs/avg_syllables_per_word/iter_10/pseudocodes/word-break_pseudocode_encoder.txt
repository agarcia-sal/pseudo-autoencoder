CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO the set of all elements in wordDict  
        SET dp TO a list of False values of length the LENGTH OF s PLUS one  
        SET the element at position zero of dp TO True  
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM zero TO i MINUS one  
                IF the element at position j of dp EQUALS True AND the substring of s FROM position j TO position i IS IN wordSet  
                    SET the element at position i of dp TO True  
                    BREAK the inner loop  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position the LENGTH OF s of dp  
    END FUNCTION  
END CLASS
