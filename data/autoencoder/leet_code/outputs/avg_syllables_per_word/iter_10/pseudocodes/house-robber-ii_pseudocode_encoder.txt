CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF houses IS empty  
                RETURN zero  
            END IF  
            IF the LENGTH OF houses EQUALS one  
                RETURN the element at position zero of houses  
            END IF  
            SET dp TO a list of zeros with the LENGTH OF houses  
            SET the element at position zero of dp TO the element at position zero of houses  
            SET the element at position one of dp TO the GREATER OF the element at position zero of houses AND the element at position one of houses  
            FOR index FROM two TO the LENGTH OF houses MINUS one  
                SET the element at position index of dp TO the GREATER OF the element at position index MINUS one of dp AND the SUM OF the element at position index MINUS two of dp AND the element at position index of houses  
            END FOR  
            RETURN the last element of dp  
        END FUNCTION  
        IF nums IS empty  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position zero of nums  
        END IF  
        RETURN the GREATER OF rob_linear(all elements of nums EXCLUDING the last one) AND rob_linear(all elements of nums EXCLUDING the first one)  
    END FUNCTION  
END CLASS
