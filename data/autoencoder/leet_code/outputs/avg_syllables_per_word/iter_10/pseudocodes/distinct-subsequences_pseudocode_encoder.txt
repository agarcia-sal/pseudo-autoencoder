CLASS Solution  
    FUNCTION numDistinct(s, t)  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
        SET dp TO a two dimensional list with m PLUS one rows and n PLUS one columns all initialized to zero  
        FOR i FROM zero TO m  
            SET element at position i and zero of dp TO one  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of s EQUALS element at position j MINUS one of t  
                    SET element at position i and j of dp TO element at position i MINUS one and j MINUS one of dp PLUS element at position i MINUS one and j of dp  
                ELSE  
                    SET element at position i and j of dp TO element at position i MINUS one and j of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position m and n of dp  
    END FUNCTION  
END CLASS
