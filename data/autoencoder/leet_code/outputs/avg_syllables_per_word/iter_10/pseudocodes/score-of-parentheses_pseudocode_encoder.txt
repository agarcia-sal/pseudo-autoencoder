CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO empty list  
        SET score TO zero  
        FOR each char IN s  
            IF char EQUALS open parenthesis  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET popped_score TO the last element REMOVED FROM stack  
                IF score EQUALS zero  
                    SET score TO popped_score PLUS one  
                ELSE  
                    SET score TO popped_score PLUS two MULTIPLIED BY score  
                END IF  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
