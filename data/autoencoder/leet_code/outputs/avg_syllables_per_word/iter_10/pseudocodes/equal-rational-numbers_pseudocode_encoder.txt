CLASS Solution  
    FUNCTION isRationalEqual(s, t)  
        SET ratios TO list containing one, one DIVIDED BY nine, one DIVIDED BY ninety-nine, one DIVIDED BY nine hundred ninety-nine, one DIVIDED BY nine thousand nine hundred ninety-nine  

        FUNCTION valueOf(s)  
            IF position of open parenthesis IN s EQUALS minus one  
                RETURN s CONVERTED TO float  
            END IF  

            SET leftParenIndex TO position of open parenthesis IN s  
            SET rightParenIndex TO position of close parenthesis IN s  
            SET dotIndex TO position of decimal point IN s  

            SET integerAndNonRepeating TO s SLICED FROM start TO leftParenIndex CONVERTED TO float  
            SET nonRepeatingLength TO leftParenIndex MINUS dotIndex MINUS one  

            SET repeating TO s SLICED FROM position one after leftParenIndex TO rightParenIndex CONVERTED TO integer  
            SET repeatingLength TO rightParenIndex MINUS leftParenIndex MINUS one  

            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY zero point one RAISED TO nonRepeatingLength MULTIPLIED BY element at position repeatingLength of ratios  
        END FUNCTION  

        RETURN absolute value of valueOf(s) MINUS valueOf(t) LESS THAN one times ten RAISED TO minus nine  
    END FUNCTION  
END CLASS
