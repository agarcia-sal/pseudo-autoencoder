CLASS Solution  
    FUNCTION makesquare(matchsticks)  
        SET total_length TO zero  
        FOR each element IN matchsticks  
            INCREMENT total_length BY element  
        END FOR  
        
        IF total_length MODULO four NOT EQUALS zero  
            RETURN False  
        END IF  
        
        SET side_length TO total_length DIVIDED BY four  
        
        CALL sort matchsticks IN descending ORDER  
        
        FUNCTION can_form_sides(sides, index)  
            IF index EQUALS the LENGTH OF matchsticks  
                FOR each side IN sides  
                    IF side NOT EQUALS side_length  
                        RETURN False  
                    END IF  
                END FOR  
                RETURN True  
            END IF  
            
            FOR i FROM zero TO three  
                IF element at position i of sides PLUS element at position index of matchsticks LESS THAN OR EQUAL TO side_length  
                    SET element at position i of sides TO element at position i of sides PLUS element at position index of matchsticks  
                    IF can_form_sides(sides, index PLUS one)  
                        RETURN True  
                    END IF  
                    SET element at position i of sides TO element at position i of sides MINUS element at position index of matchsticks  
                END IF  
                
                IF element at position i of sides EQUALS zero  
                    BREAK  
                END IF  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        RETURN can_form_sides(list consisting of four zeros, zero)  
    END FUNCTION  
END CLASS
