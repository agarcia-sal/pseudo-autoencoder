CLASS Solution
    FUNCTION frogPosition(n, edges, t, target)
        IF n EQUALS one
            RETURN one point zero
        END IF

        SET graph TO a default mapping from keys to empty lists

        FOR each pair u and v IN edges
            APPEND v TO the list at key u in graph
            APPEND u TO the list at key v in graph
        END FOR

        SET queue TO a double-ended queue containing the tuple of one, one point zero, and zero
        SET visited TO a set containing one

        WHILE queue is not empty
            REMOVE first element FROM queue and UNPACK into current_node, current_probability, current_time

            IF current_time EQUALS t OR (current_node NOT EQUALS one AND the LENGTH OF the list at key current_node in graph EQUALS one)
                IF current_node EQUALS target
                    RETURN current_probability
                ELSE
                    CONTINUE to the next iteration of the WHILE loop
                END IF
            END IF

            SET possible_moves TO the LENGTH OF the list at key current_node in graph MINUS (one IF current_node NOT EQUALS one ELSE zero)

            FOR each neighbor IN the list at key current_node in graph
                IF neighbor NOT IN visited
                    ADD neighbor TO visited
                    APPEND the tuple of neighbor, current_probability DIVIDED BY possible_moves, and current_time PLUS one TO queue
                END IF
            END FOR
        END WHILE

        RETURN zero point zero
    END FUNCTION
END CLASS
