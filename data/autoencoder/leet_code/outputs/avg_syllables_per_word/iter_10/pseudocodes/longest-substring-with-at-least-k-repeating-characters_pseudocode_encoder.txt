CLASS Solution  
    FUNCTION longestSubstring(s, k)  
        IF length of s LESS THAN k  
            RETURN zero  
        END IF  
        SET count TO the result of calling Counter on s  
        FOR each char IN count  
            IF the element at char of count LESS THAN k  
                RETURN the maximum value among  
                    FOR each substring IN the result of splitting s by char  
                        CALL longestSubstring with substring and k  
                    END FOR  
            END IF  
        END FOR  
        RETURN length of s  
    END FUNCTION  
END CLASS
