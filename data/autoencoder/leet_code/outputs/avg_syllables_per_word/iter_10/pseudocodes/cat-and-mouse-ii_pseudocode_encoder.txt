CLASS Solution  
    FUNCTION canMouseWin(grid, catJump, mouseJump)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET cat_start TO zero  
        SET mouse_start TO zero  
        SET food TO zero  
        SET dirs TO a sequence of values negative one, zero, one, zero, negative one  
        SET g_mouse TO an empty list for each element FROM zero TO m MULTIPLIED BY n MINUS one  
        SET g_cat TO an empty list for each element FROM zero TO m MULTIPLIED BY n MINUS one  
        
        FOR each index i AND element row IN grid  
            FOR each index j AND element c IN row  
                IF c EQUALS the symbol for wall  
                    CONTINUE to next iteration  
                END IF  
                SET v TO i MULTIPLIED BY n PLUS j  
                IF c EQUALS the symbol for cat  
                    SET cat_start TO v  
                ELSE IF c EQUALS the symbol for mouse  
                    SET mouse_start TO v  
                ELSE IF c EQUALS the symbol for food  
                    SET food TO v  
                END IF  
                FOR each pair (a, b) FROM consecutive elements OF dirs  
                    FOR k FROM zero TO mouseJump  
                        SET x TO i PLUS k MULTIPLIED BY a  
                        SET y TO j PLUS k MULTIPLIED BY b  
                        IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of grid AT position y EQUALS the symbol for wall  
                            BREAK the loop  
                        END IF  
                        APPEND x MULTIPLIED BY n PLUS y TO element at position v OF g_mouse  
                    END FOR  
                    FOR k FROM zero TO catJump  
                        SET x TO i PLUS k MULTIPLIED BY a  
                        SET y TO j PLUS k MULTIPLIED BY b  
                        IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of grid AT position y EQUALS the symbol for wall  
                            BREAK the loop  
                        END IF  
                        APPEND x MULTIPLIED BY n PLUS y TO element at position v OF g_cat  
                    END FOR  
                END FOR  
            END FOR  
        END FOR  
        RETURN the result of calling FUNCTION calc WITH g_mouse, g_cat, mouse_start, cat_start, AND food EQUALS one  
    END FUNCTION  

    FUNCTION calc(g_mouse, g_cat, mouse_start, cat_start, hole)  
        FUNCTION get_prev_states(state)  
            SET m TO element at position zero of state  
            SET c TO element at position one of state  
            SET t TO element at position two of state  
            SET pt TO t XOR one  
            SET pre TO an empty list  
            IF pt EQUALS one  
                FOR each pc IN element at position c OF g_cat  
                    IF element at position m OF ans AT position pc AT position one EQUALS zero  
                        APPEND the tuple of m, pc, AND pt TO pre  
                    END IF  
                END FOR  
            ELSE  
                FOR each pm IN element at position m OF g_mouse  
                    IF element at position pm OF ans AT position c AT position zero EQUALS zero  
                        APPEND the tuple of pm, c, AND zero TO pre  
                    END IF  
                END FOR  
            END IF  
            RETURN pre  
        END FUNCTION  

        SET n TO the LENGTH OF g_mouse  
        SET degree TO a three-dimensional list with dimensions n BY n BY two, each initialized to zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET the element at position zero of element at position j of element at position i of degree TO the LENGTH OF element at position i OF g_mouse  
                SET the element at position one of element at position j of element at position i of degree TO the LENGTH OF element at position j OF g_cat  
            END FOR  
        END FOR  

        SET ans TO a three-dimensional list with dimensions n BY n BY two, each initialized to zero  
        SET q TO an empty double-ended queue  
        FOR i FROM zero TO n MINUS one  
            SET element at position hole OF ans AT position i AT position one TO one  
            SET element at position i OF ans AT position hole AT position zero TO two  
            SET element at position i OF ans AT position i AT position one TO two  
            SET element at position i OF ans AT position i AT position zero TO two  
            APPEND the tuple of hole, i, AND one TO q  
            APPEND the tuple of i, hole, AND zero TO q  
            APPEND the tuple of i, i, AND zero TO q  
            APPEND the tuple of i, i, AND one TO q  
        END FOR  
        WHILE q IS NOT empty  
            SET state TO the element removed from the left of q  
            SET t TO element at position of element at position zero of state, element at position one of state, AND element at position two of state OF ans  
            FOR each prev_state IN the result of calling get_prev_states WITH state  
                SET pm TO element at position zero of prev_state  
                SET pc TO element at position one of prev_state  
                SET pt TO element at position two of prev_state  
                IF pt EQUALS t MINUS one  
                    SET element at position pm OF ans AT position pc AT position pt TO t  
                    APPEND prev_state TO q  
                ELSE  
                    DECREMENT element at position pm OF degree AT position pc AT position pt BY one  
                    IF element at position pm OF degree AT position pc AT position pt EQUALS zero  
                        SET element at position pm OF ans AT position pc AT position pt TO t  
                        APPEND prev_state TO q  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN element at position mouse_start OF ans AT position cat_start AT position zero  
    END FUNCTION  
END CLASS
