CLASS Solution  
    FUNCTION originalDigits(s)  
        SET count TO the frequency count of characters in s  
        SET digits TO a list of ten zeros  
        
        SET element at position zero of digits TO the element at key 'z' of count  
        SET element at position two of digits TO the element at key 'w' of count  
        SET element at position four of digits TO the element at key 'u' of count  
        SET element at position six of digits TO the element at key 'x' of count  
        SET element at position eight of digits TO the element at key 'g' of count  
        
        SET element at position three of digits TO the element at key 'h' of count MINUS the element at position eight of digits  
        SET element at position five of digits TO the element at key 'f' of count MINUS the element at position four of digits  
        SET element at position seven of digits TO the element at key 's' of count MINUS the element at position six of digits  
        SET element at position nine of digits TO the element at key 'i' of count MINUS the element at position five of digits MINUS the element at position six of digits MINUS the element at position eight of digits  
        SET element at position one of digits TO the element at key 'o' of count MINUS the element at position zero of digits MINUS the element at position two of digits MINUS the element at position four of digits  
        
        SET result TO an empty list  
        FOR each index i FROM zero TO nine  
            FOR each count FROM one TO the element at position i of digits  
                APPEND the string representation of i TO result  
            END FOR  
        END FOR  
        
        RETURN the concatenation of all elements of result  
    END FUNCTION  
END CLASS
