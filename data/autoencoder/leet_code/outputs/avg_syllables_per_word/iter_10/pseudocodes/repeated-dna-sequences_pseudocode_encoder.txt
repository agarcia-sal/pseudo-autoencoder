CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO a new dictionary with default integer values  
        FOR index FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO substring of s from position index TO position index PLUS ten MINUS one  
            INCREMENT the count of sequence IN sequence_count BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each sequence, count IN sequence_count  
            IF count GREATER THAN one  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
