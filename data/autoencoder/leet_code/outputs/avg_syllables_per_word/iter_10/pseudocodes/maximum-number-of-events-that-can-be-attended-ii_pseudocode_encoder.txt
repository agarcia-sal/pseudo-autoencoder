CLASS Solution  
    FUNCTION maxValue(events, k)  
        CALL events TO BE SORTED BY the end day of each event  
        SET n TO the LENGTH OF events  
        CREATE dp AS a list of lists with n PLUS one elements each containing k PLUS one zeros  
        FOR i FROM one TO n  
            SET start TO the first element of event at position i MINUS one of events  
            SET end TO the second element of event at position i MINUS one of events  
            SET value TO the third element of event at position i MINUS one of events  
            SET j TO the position found by binary searching for start in events based on end days using bisect left  
            FOR l FROM one TO k  
                SET dp at position i and l TO the GREATER OF dp at position i MINUS one and l AND dp at position j and l MINUS one PLUS value  
            END FOR  
        END FOR  
        RETURN dp at position n and k  
    END FUNCTION  
END CLASS
