CLASS Solution  
    FUNCTION networkDelayTime(times, n, k)  
        SET graph TO an empty dictionary mapping each integer from one TO n TO an empty list  
        
        FOR each element u, v, w IN times  
            APPEND a pair of v AND w TO the list at key u IN graph  
        END FOR  
        
        SET dist TO a dictionary mapping each integer from one TO n TO positive infinity  
        SET the value of dist at key k TO zero  
        
        SET min_heap TO a list containing a pair of zero AND k  
        
        WHILE min_heap is not empty  
            REMOVE and SET current_dist AND node TO the element with the smallest first value from min_heap  
            
            IF current_dist GREATER THAN the value of dist at key node  
                CONTINUE the next iteration of the WHILE loop  
            END IF  
            
            FOR each element neighbor, weight IN the list at key node IN graph  
                SET distance TO current_dist PLUS weight  
                
                IF distance LESS THAN the value of dist at key neighbor  
                    SET the value of dist at key neighbor TO distance  
                    ADD the pair of distance AND neighbor TO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        SET max_time TO the maximum value among all values in dist  
        
        IF max_time NOT EQUALS positive infinity  
            RETURN max_time  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
