CLASS Solution  
    FUNCTION findMaxAverage(nums, k)  
        FUNCTION canFindAverage(mid)  
            SET prefix_sums TO a list of zeros with LENGTH EQUAL TO the LENGTH OF nums PLUS one  
            SET min_prefix_sum TO a list of zeros with LENGTH EQUAL TO the LENGTH OF nums PLUS one  
            FOR i FROM zero TO the LENGTH OF nums MINUS one  
                SET element at position INCREMENTED BY one of prefix_sums TO element at position i of prefix_sums PLUS element at position i of nums MINUS mid  
                IF i GREATER THAN OR EQUAL TO k  
                    SET element at position INCREMENTED BY one of min_prefix_sum TO the MINIMUM OF element at position i of min_prefix_sum AND element at position i PLUS one MINUS k of prefix_sums  
                ELSE  
                    SET element at position INCREMENTED BY one of min_prefix_sum TO element at position i of min_prefix_sum  
                END IF  
                IF i GREATER THAN OR EQUAL TO k MINUS one AND element at position INCREMENTED BY one of prefix_sums MINUS element at position INCREMENTED BY one of min_prefix_sum GREATER THAN OR EQUAL TO zero  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        SET low TO the MINIMUM ELEMENT of nums  
        SET high TO the MAXIMUM ELEMENT of nums  
        WHILE high MINUS low GREATER THAN one TIMES ten TO THE POWER OF MINUS five  
            SET mid TO low PLUS high DIVIDED BY two  
            IF canFindAverage(mid)  
                SET low TO mid  
            ELSE  
                SET high TO mid  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
