CLASS Solution  
    FUNCTION largestDivisibleSubset(nums)  
        SORT nums INTO ascending order  
        SET n TO the LENGTH OF nums  
        SET f TO a list of size n WITH all elements SET TO one  
        SET k TO zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF element at position i of nums DIVIDED BY element at position j of nums HAS zero remainder THEN  
                    SET f at position i TO the GREATER OF f at position i AND f at position j PLUS one  
                END IF  
            END FOR  
            IF f at position k LESS THAN f at position i THEN  
                SET k TO i  
            END IF  
        END FOR  
        SET m TO f at position k  
        SET i TO k  
        SET ans TO empty list  
        WHILE m GREATER THAN zero  
            IF element at position k of nums DIVIDED BY element at position i of nums HAS zero remainder AND f at position i EQUALS m THEN  
                APPEND element at position i of nums TO ans  
                SET k TO i  
                DECREMENT m BY one  
            END IF  
            DECREMENT i BY one  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
