CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val equal to element at position zero of values  
    SET i TO one  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE first element FROM queue AND SET TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUAL TO None  
            SET node.left TO new TreeNode with val equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUAL TO None  
            SET node.right TO new TreeNode with val equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder(self, root)  
        IF root EQUALS None  
            RETURN empty list  
        END IF  
        SET column_table TO new default dictionary with list as default value  
        SET queue TO new empty double-ended queue  
        APPEND tuple of root and zero TO queue  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND UNPACK TO node AND column_index  
            IF node NOT EQUALS None  
                APPEND node.val TO list at key column_index in column_table  
                IF column_index LESS THAN min_column  
                    SET min_column TO column_index  
                END IF  
                IF column_index GREATER THAN max_column  
                    SET max_column TO column_index  
                END IF  
                APPEND tuple of node.left AND column_index MINUS one TO queue  
                APPEND tuple of node.right AND column_index PLUS one TO queue  
            END IF  
        END WHILE  
        SET result TO empty list  
        FOR x FROM min_column TO max_column  
            APPEND list at key x in column_table TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
