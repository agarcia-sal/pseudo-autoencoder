CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MOD TO one billion PLUS seven  
        SORT nums IN ascending order  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        SET power TO a list of LENGTH n FILLED WITH one  
        
        FOR index FROM one TO n MINUS one  
            SET power at position index TO power at position index MINUS one MULTIPLIED BY two MODULO MOD  
        END FOR  
        
        FOR index FROM zero TO n MINUS one  
            SET total_sum TO total_sum PLUS element at position index of nums MULTIPLIED BY (power at position index MINUS power at position n MINUS index MINUS one) MODULO MOD  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
