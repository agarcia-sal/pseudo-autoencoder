CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO empty list  
        FOR each num IN target  
            APPEND negative of num TO max_heap  
        END FOR  
        CALL heapify FUNCTION on max_heap  
          
        SET total_sum TO zero  
        FOR each element IN target  
            INCREMENT total_sum BY element  
        END FOR  
          
        WHILE element at position zero of max_heap LESS THAN negative one  
            SET largest TO negative of result of popping element from max_heap  
            SET rest_sum TO total_sum MINUS largest  
              
            IF largest LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero  
                RETURN False  
            END IF  
              
            SET new_value TO largest MODULO rest_sum  
            IF new_value EQUALS zero AND rest_sum NOT EQUALS one  
                RETURN False  
            END IF  
              
            SET total_sum TO rest_sum PLUS new_value  
            PUSH negative of new_value INTO max_heap  
        END WHILE  
          
        RETURN True  
    END FUNCTION  
END CLASS
