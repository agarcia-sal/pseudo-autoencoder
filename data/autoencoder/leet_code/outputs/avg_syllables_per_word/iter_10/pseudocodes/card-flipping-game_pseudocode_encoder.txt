CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO empty set  
        FOR each pair of elements x and y FROM fronts AND backs  
            IF x EQUALS y  
                ADD x TO same  
            END IF  
        END FOR  
        SET all_numbers TO set of elements FROM fronts AND backs combined  
        SET good_numbers TO all_numbers MINUS same  
        IF good_numbers IS NOT empty  
            SET result TO minimum element IN good_numbers  
            RETURN result  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
