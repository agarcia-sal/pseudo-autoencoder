CLASS Solution  
    FUNCTION findLongestChain(pairs)  
        CALL auxiliary sort_pairs_by_second_element WITH pairs  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each pair IN pairs  
            SET start TO element at position zero of pair  
            SET end TO element at position one of pair  
            IF start GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
  
    FUNCTION sort_pairs_by_second_element WITH pairs  
        REARRANGE pairs IN ORDER BASED ON the second element of each pair in ascending order  
    END FUNCTION  
END CLASS
