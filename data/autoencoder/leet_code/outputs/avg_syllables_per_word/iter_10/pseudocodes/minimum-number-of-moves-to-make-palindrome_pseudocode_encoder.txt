CLASS Solution  
    FUNCTION minMovesToMakePalindrome(s)  
        SET s TO list converted from s  
        SET n TO the LENGTH OF s  
        SET moves TO zero  
          
        WHILE the LENGTH OF s GREATER THAN one  
            SET matched TO false  
            FOR i FROM the LENGTH OF s MINUS one DOWN TO one  
                IF the element at position i of s EQUALS the element at position zero of s  
                    FOR j FROM i TO the LENGTH OF s MINUS two  
                        SWAP the elements at positions j and j PLUS one of s  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE first element FROM s  
                    REMOVE last element FROM s  
                    SET matched TO true  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            IF matched EQUALS false  
                REMOVE first element FROM s  
            END IF  
        END WHILE  
        RETURN moves  
    END FUNCTION  
END CLASS
