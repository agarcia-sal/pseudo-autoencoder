CLASS Solution  
    FUNCTION generateParenthesis(n)  
        FUNCTION backtrack(s, left, right)  
            IF LENGTH OF s EQUALS 2 MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s APPENDED BY open parenthesis, left INCREMENTED BY one, right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s APPENDED BY close parenthesis, left, right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
