CLASS Solution  
    FUNCTION minimumOneBitOperations(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE one LEFT SHIFTED BY k PLUS one LESS THAN OR EQUAL TO n  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO one LEFT SHIFTED BY k  
        RETURN one LEFT SHIFTED BY k PLUS one MINUS one MINUS minimumOneBitOperations(n XOR most_significant_bit)  
    END FUNCTION  
END CLASS
