CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO an empty dictionary mapping to empty lists  
        FOR each path IN paths  
            SET parts TO the fragmentation of path by spaces  
            SET directory TO the first element of parts  
            FOR each file_info IN elements of parts starting from the second element  
                SET file_name TO the segment of file_info before the first occurrence of opening parenthesis  
                SET content TO the segment of file_info after the first occurrence of opening parenthesis excluding the last character  
                APPEND the concatenation of directory, slash, and file_name TO the list at content_to_paths at key content  
            END FOR  
        END FOR  
        RETURN the list collected by selecting each list of paths from values of content_to_paths where the length of that list is greater than one  
    END FUNCTION  
END CLASS
