CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of integers from zero TO n inclusive  
        SET self.rank TO list of zeros with length n PLUS one  
        SET self.count TO n  
    END FUNCTION  
  
    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u THEN  
            SET element at position u of self.parent TO self.find(element at position u of self.parent)  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  
  
    FUNCTION union(self, u, v)  
        SET rootU TO self.find(u)  
        SET rootV TO self.find(v)  
        IF rootU NOT EQUALS rootV THEN  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank THEN  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank THEN  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            DECREMENT self.count BY one  
            RETURN true  
        END IF  
        RETURN false  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION maxNumEdgesToRemove(self, n, edges)  
        SET uf_common TO new UnionFind with parameter n  
        SET uf_alice TO new UnionFind with parameter n  
        SET uf_bob TO new UnionFind with parameter n  
  
        SET used_edges TO zero  
  
        FOR each element consisting of t, u, v IN edges  
            IF t EQUALS three THEN  
                IF uf_common.union(u, v) THEN  
                    CALL uf_alice.union with parameters u, v  
                    CALL uf_bob.union with parameters u, v  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  
  
        FOR each element consisting of t, u, v IN edges  
            IF t EQUALS one AND uf_alice.union(u, v) THEN  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND uf_bob.union(u, v) THEN  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  
  
        IF uf_alice.count EQUALS one AND uf_bob.count EQUALS one THEN  
            RETURN length of edges MINUS used_edges  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
