CLASS Solution  
    FUNCTION maxProduct(words)  
        SET word_sets TO empty list  
        FOR each word IN words  
            APPEND set of characters in word TO word_sets  
        END FOR  
        
        SET max_product TO zero  
        
        FOR i FROM zero TO LENGTH OF words MINUS one  
            FOR j FROM i PLUS one TO LENGTH OF words MINUS one  
                IF NOT the intersection of element at position i of word_sets AND element at position j of word_sets IS NOT empty  
                    SET product TO the LENGTH OF element at position i of words MULTIPLIED BY the LENGTH OF element at position j of words  
                    IF product GREATER THAN max_product  
                        SET max_product TO product  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_product  
    END FUNCTION  
END CLASS
