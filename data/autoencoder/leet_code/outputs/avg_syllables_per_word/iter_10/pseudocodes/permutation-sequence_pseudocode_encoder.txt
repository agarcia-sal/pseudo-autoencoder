CLASS Solution  
    FUNCTION getPermutation(n, k)  
        SET numbers TO the list of numbers FROM one TO n  
        DECREMENT k BY one  
        SET result TO empty list  
        FOR i FROM n TO one STEP minus one  
            SET fact TO factorial of i MINUS one  
            SET index TO k DIVIDED BY fact  
            APPEND the element at position index of numbers converted TO string TO result  
            REMOVE the element at position index FROM numbers  
            SET k TO k MODULO fact  
        END FOR  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
