CLASS Solution  
    FUNCTION minInsertions(s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO the LENGTH OF s  
            CALL create_two_dimensional_array_of_zeros WITH n AND STORE AS dp  
            
            FOR i FROM zero TO n MINUS one  
                SET element at position i position i of dp TO one  
            END FOR  
            
            FOR cl FROM two TO n  
                FOR i FROM zero TO n MINUS cl  
                    SET j TO i PLUS cl MINUS one  
                    IF element at position i of s EQUALS element at position j of s AND cl EQUALS two  
                        SET element at position i position j of dp TO two  
                    ELSE IF element at position i of s EQUALS element at position j of s  
                        SET element at position i position j of dp TO element at position i PLUS one position j MINUS one of dp PLUS two  
                    ELSE  
                        SET element at position i position j of dp TO the GREATER OF element at position i position j MINUS one of dp AND element at position i PLUS one position j of dp  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN element at position zero position n MINUS one of dp  
        END FUNCTION  
        
        RETURN the LENGTH OF s MINUS longest_palindromic_subsequence(s)  
    END FUNCTION  
END CLASS  

FUNCTION create_two_dimensional_array_of_zeros(n)  
    SET new_array TO empty list  
    FOR index FROM zero TO n MINUS one  
        CREATE list_of_zeros WITH n elements all SET TO zero  
        APPEND list_of_zeros TO new_array  
    END FOR  
    RETURN new_array  
END FUNCTION
