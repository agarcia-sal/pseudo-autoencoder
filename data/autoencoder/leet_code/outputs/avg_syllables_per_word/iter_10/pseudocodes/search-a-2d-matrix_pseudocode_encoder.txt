CLASS Solution  
    FUNCTION searchMatrix(matrix, target)  
        IF matrix IS empty OR element at position zero of matrix IS empty  
            RETURN False  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER DIVISION of left PLUS right BY two  
            SET mid_value TO element at position INTEGER DIVISION of mid BY n of matrix AT the REMAINDER of mid DIVIDED BY n  
            IF mid_value EQUALS target  
                RETURN True  
            ELSE IF mid_value LESS THAN target  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
