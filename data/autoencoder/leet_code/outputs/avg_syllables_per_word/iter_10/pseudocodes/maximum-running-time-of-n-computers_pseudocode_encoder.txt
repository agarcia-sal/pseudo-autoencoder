CLASS Solution  
    FUNCTION maxRunTime(n, batteries)  
        FUNCTION canRunFor(time)  
            SET total TO zero  
            FOR each battery IN batteries  
                IF battery LESS THAN time  
                    INCREMENT total BY battery  
                ELSE  
                    INCREMENT total BY time  
                END IF  
            END FOR  
            IF total GREATER THAN OR EQUAL TO time MULTIPLIED BY n  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET left TO zero  
        SET right TO the SUM OF batteries DIVIDED BY n  
        WHILE left LESS THAN right  
            SET mid TO left PLUS right PLUS one DIVIDED BY two  
            IF canRunFor(mid)  
                SET left TO mid  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
