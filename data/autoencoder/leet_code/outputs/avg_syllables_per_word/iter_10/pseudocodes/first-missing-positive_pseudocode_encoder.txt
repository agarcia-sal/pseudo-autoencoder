CLASS Solution  
    FUNCTION firstMissingPositive(nums)  
        SET n TO the LENGTH OF nums  
        FOR index FROM zero TO n MINUS one  
            WHILE element at position index of nums GREATER THAN OR EQUAL TO one AND element at position index of nums LESS THAN OR EQUAL TO n AND element at position element at position index of nums MINUS one of nums NOT EQUALS element at position index of nums  
                SET correct_index TO element at position index of nums MINUS one  
                SET temp TO element at position index of nums  
                SET element at position index of nums TO element at position correct_index of nums  
                SET element at position correct_index of nums TO temp  
            END WHILE  
        END FOR  
        FOR index FROM zero TO n MINUS one  
            IF element at position index of nums NOT EQUALS index PLUS one  
                RETURN index PLUS one  
            END IF  
        END FOR  
        RETURN n PLUS one  
    END FUNCTION  
END CLASS
