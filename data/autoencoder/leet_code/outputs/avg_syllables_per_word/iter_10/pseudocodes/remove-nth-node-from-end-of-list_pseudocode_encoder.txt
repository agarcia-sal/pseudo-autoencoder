CLASS ListNode  
    FUNCTION __init__(val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with val equal to element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values from position one to the end  
        SET node TO new ListNode with val equal to val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN (element val of p1 EQUALS element val of p2) AND is_same_list(element next of p1, element next of p2)  
END FUNCTION

CLASS Solution  
    FUNCTION removeNthFromEnd(head, n)  
        SET dummy TO new ListNode with val equal to zero  
        SET dummy.next TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one  
            SET first TO element next of first  
        END FOR  
        WHILE first IS NOT none  
            SET first TO element next of first  
            SET second TO element next of second  
        END WHILE  
        SET second.next TO element next of element next of second  
        RETURN element next of dummy  
    END FUNCTION  
END CLASS
