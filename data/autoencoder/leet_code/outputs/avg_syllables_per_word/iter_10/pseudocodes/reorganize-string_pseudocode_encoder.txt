CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO Counter OF s  
        SET max_heap TO empty list  
        FOR each character AND frequency IN count items  
            APPEND tuple of negative frequency AND character TO max_heap  
        END FOR  
        CALL heapify FUNCTION ON max_heap  
        
        SET prev_char TO None  
        SET prev_freq TO zero  
        SET result TO empty list  
        
        WHILE max_heap IS NOT empty OR prev_freq LESS THAN zero  
            IF max_heap IS empty AND prev_freq LESS THAN zero  
                RETURN empty string  
            END IF  
            
            SET freq AND char TO result OF heappop ON max_heap  
            APPEND char TO result  
            
            SET freq TO freq PLUS one  
            
            IF prev_freq LESS THAN zero  
                CALL heappush ON max_heap WITH tuple prev_freq AND prev_char  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN concatenation OF all elements IN result  
    END FUNCTION  
END CLASS
