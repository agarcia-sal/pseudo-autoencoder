CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a new list containing minus one  
        SET max_length TO zero  
        FOR each index i AND character char IN s  
            IF char EQUALS open parenthesis  
                APPEND i TO stack  
            ELSE  
                REMOVE last element FROM stack  
                IF stack IS empty  
                    APPEND i TO stack  
                ELSE  
                    SET difference TO i MINUS element at last position of stack  
                    IF difference GREATER THAN max_length  
                        SET max_length TO difference  
                    END IF  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
