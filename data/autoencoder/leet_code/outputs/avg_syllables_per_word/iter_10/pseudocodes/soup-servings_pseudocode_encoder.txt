CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN 4800  
            RETURN one point zero  
        END IF  
        SET n TO the quantity n PLUS twenty four DIVIDED BY twenty five  
        SET memo TO an empty COLLECTION  
        FUNCTION dp(a, b)  
            IF the PAIR of a and b IS IN memo  
                RETURN the VALUE of memo AT the PAIR of a and b  
            END IF  
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN zero point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one point zero  
            END IF  
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero point zero  
            END IF  
            SET prob TO zero point two five MULTIPLIED BY the SUM of  
                dp OF a MINUS four AND b, PLUS  
                dp OF a MINUS three AND b MINUS one, PLUS  
                dp OF a MINUS two AND b MINUS two, PLUS  
                dp OF a MINUS one AND b MINUS three  
            SET memo AT the PAIR of a and b TO prob  
            RETURN prob  
        END FUNCTION  
        RETURN dp OF n AND n  
    END FUNCTION  
END CLASS
