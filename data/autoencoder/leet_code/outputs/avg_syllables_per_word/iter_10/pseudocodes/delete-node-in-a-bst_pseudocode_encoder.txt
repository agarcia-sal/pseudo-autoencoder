CLASS TreeNode  
    FUNCTION __init__ with parameters val EQUALS zero, left EQUALS none, right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  
  
FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode initialized with element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO remove first element FROM queue  
        IF i LESS THAN length of values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  
  
FUNCTION is_same_tree with parameters p and q  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree with arguments p.left and q.left AND is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  
  
CLASS Solution  
    FUNCTION deleteNode with parameters self, root, key  
        IF root EQUALS none  
            RETURN none  
        END IF  
        IF root.val GREATER THAN key  
            SET root.left TO self.deleteNode with arguments root.left and key  
            RETURN root  
        END IF  
        IF root.val LESS THAN key  
            SET root.right TO self.deleteNode with arguments root.right and key  
            RETURN root  
        END IF  
        IF root.left EQUALS none  
            RETURN root.right  
        END IF  
        IF root.right EQUALS none  
            RETURN root.left  
        END IF  
        SET node TO root.right  
        WHILE node.left IS NOT none  
            SET node TO node.left  
        END WHILE  
        SET node.left TO root.left  
        SET root TO root.right  
        RETURN root  
    END FUNCTION  
END CLASS
