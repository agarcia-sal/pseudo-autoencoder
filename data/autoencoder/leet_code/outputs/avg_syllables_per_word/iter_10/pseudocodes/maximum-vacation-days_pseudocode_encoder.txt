CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the element at position zero of days  
          
        SET dp TO a list of n elements each set TO negative one  
        SET the element at position zero of dp TO zero  
          
        SET graph TO an empty mapping from each integer FROM zero TO n MINUS one TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i of flights AT position j EQUALS one  
                    APPEND j TO the list at key i of graph  
                END IF  
            END FOR  
        END FOR  
          
        FOR week FROM zero TO k MINUS one  
            SET new_dp TO a list of n elements each set TO negative one  
            FOR i FROM zero TO n MINUS one  
                IF the element at position i of dp EQUALS negative one  
                    CONTINUE to next iteration  
                END IF  
                IF the element at position i of new_dp LESS THAN the element at position i of dp PLUS the element at position i of days AT position week  
                    SET the element at position i of new_dp TO the element at position i of dp PLUS the element at position i of days AT position week  
                END IF  
                FOR each j IN the list at key i of graph  
                    IF the element at position j of new_dp LESS THAN the element at position i of dp PLUS the element at position j of days AT position week  
                        SET the element at position j of new_dp TO the element at position i of dp PLUS the element at position j of days AT position week  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
          
        SET max_value TO negative infinity  
        FOR each value IN dp  
            IF value GREATER THAN max_value  
                SET max_value TO value  
            END IF  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
