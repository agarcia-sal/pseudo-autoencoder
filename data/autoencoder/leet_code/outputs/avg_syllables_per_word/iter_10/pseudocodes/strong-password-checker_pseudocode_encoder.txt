CLASS Solution  
    FUNCTION strongPasswordChecker(password)  
        SET n TO the LENGTH OF password  
        
        SET has_lower TO false  
        SET has_upper TO false  
        SET has_digit TO false  
        FOR each character c IN password  
            IF c IS A LOWERCASE LETTER  
                SET has_lower TO true  
            END IF  
            IF c IS AN UPPERCASE LETTER  
                SET has_upper TO true  
            END IF  
            IF c IS A DIGIT  
                SET has_digit TO true  
            END IF  
        END FOR  
        
        SET missing_types TO three MINUS the SUM OF has_lower, has_upper, and has_digit  
        
        SET replace TO zero  
        SET one TO zero  
        SET two TO zero  
        SET i TO two  
        WHILE i IS LESS THAN n  
            IF character at position i of password EQUALS character at position i MINUS one of password AND character at position i OF password EQUALS character at position i MINUS two of password  
                SET length TO two  
                WHILE i IS LESS THAN n AND character at position i of password EQUALS character at position i MINUS one of password  
                    INCREMENT length BY one  
                    INCREMENT i BY one  
                END WHILE  
                INCREMENT replace BY length DIVIDED BY three  
                IF length MODULO three EQUALS zero  
                    INCREMENT one BY one  
                ELSE IF length MODULO three EQUALS one  
                    INCREMENT two BY one  
                END IF  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        IF n IS LESS THAN six  
            RETURN the GREATER OF missing_types AND six MINUS n  
        ELSE IF n IS LESS THAN OR EQUAL TO twenty  
            RETURN the GREATER OF missing_types AND replace  
        ELSE  
            SET delete TO n MINUS twenty  
            
            SET replace TO replace MINUS the MINIMUM OF delete AND one  
            SET replace TO replace MINUS the INTEGER DIVISION BY two OF the MINIMUM OF the GREATER OF delete MINUS one AND zero, AND two MULTIPLIED BY two  
            SET replace TO replace MINUS the INTEGER DIVISION BY three OF the GREATER OF delete MINUS one MINUS two MULTIPLIED BY two AND zero  
            
            RETURN delete PLUS the GREATER OF missing_types AND replace  
        END IF  
    END FUNCTION  
END CLASS
