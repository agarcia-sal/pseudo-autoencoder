CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of infinity repeated fifteen times  
        FOR each pair of f and r IN tires  
            SET total_time TO zero  
            FOR i FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER OF i  
                IF lap_time GREATER THAN changeTime PLUS f  
                    BREAK OUT OF the inner loop  
                END IF  
                INCREMENT total_time BY lap_time  
                SET min_times at position i TO the lesser OF min_times at position i AND total_time  
            END FOR  
        END FOR  
        
        SET dp TO a list of infinity repeated numLaps PLUS one times  
        SET dp at position zero TO zero  
        
        FOR i FROM one TO numLaps  
            FOR j FROM zero TO the lesser OF i MINUS one AND fourteen  
                SET dp at position i TO the lesser OF dp at position i AND dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
            END FOR  
        END FOR  
        
        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
