CLASS Solution  
    FUNCTION kMirror(k, n)  
        FUNCTION is_palindrome(s)  
            RETURN s EQUALS reversed s  
        END FUNCTION  
        
        FUNCTION to_base_k(num, k)  
            IF num EQUALS zero  
                RETURN string zero  
            END IF  
            SET digits TO empty list  
            WHILE num GREATER THAN zero  
                APPEND remainder of num DIVIDED BY k TO digits  
                SET num TO quotient of num DIVIDED BY k  
            END WHILE  
            RETURN concatenation of each element converted to string in reverse order of digits  
        END FUNCTION  
        
        FUNCTION generate_palindromes(length)  
            IF length EQUALS one  
                FOR i FROM one TO nine  
                    YIELD string conversion of i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO the INTEGER division of (length MINUS one) BY two  
                SET end TO ten RAISED TO the INTEGER division of (length PLUS one) BY two  
                FOR half FROM start TO end MINUS one  
                    SET half_str TO string conversion of half  
                    IF length MODULO two EQUALS zero  
                        SET full TO half_str CONCATENATED WITH reverse of half_str  
                    ELSE  
                        SET full TO half_str CONCATENATED WITH reverse of half_str omitting last character  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO empty list  
        SET length TO one  
        WHILE length OF k_mirror_numbers LESS THAN n  
            FOR pal IN generate_palindromes(length)  
                SET num TO integer conversion of pal  
                IF is_palindrome(to_base_k(num, k))  
                    APPEND num TO k_mirror_numbers  
                    IF length OF k_mirror_numbers EQUALS n  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN sum of elements in k_mirror_numbers  
    END FUNCTION  
END CLASS
