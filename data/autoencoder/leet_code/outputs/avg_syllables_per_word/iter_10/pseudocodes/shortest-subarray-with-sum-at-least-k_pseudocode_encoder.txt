CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO a list containing zero  
        FOR each num IN nums  
            APPEND the element at position length of prefix_sum MINUS one of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        
        SET dq TO an empty double-ended queue  
        SET min_length TO positive infinity  
        
        FOR each index i and current_sum IN prefix_sum with index  
            WHILE dq is not empty AND current_sum MINUS the element at position first element of dq of prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET min_length TO the minimum of min_length AND i MINUS the first element removed FROM dq  
            END WHILE  
            
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO the element at position last element of dq of prefix_sum  
                REMOVE last element FROM dq  
            END WHILE  
            
            APPEND i TO dq  
        END FOR  
        
        IF min_length IS NOT EQUAL TO positive infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
