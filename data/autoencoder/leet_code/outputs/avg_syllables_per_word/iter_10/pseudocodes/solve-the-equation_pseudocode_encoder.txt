CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO side replacing each plus character WITH concatenation of hash character and plus character then replacing each minus character WITH concatenation of hash character and minus character then splitting by hash character  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token EQUALS empty string  
                    CONTINUE to next iteration  
                END IF  
                IF token CONTAINS character x  
                    IF token EQUALS character x OR token EQUALS concatenation of plus character and character x  
                        INCREMENT x_count BY one  
                    ELSE IF token EQUALS concatenation of minus character and character x  
                        DECREMENT x_count BY one  
                    ELSE  
                        INCREMENT x_count BY integer conversion of token with character x removed  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY integer conversion of token  
                END IF  
            END FOR  
            RETURN x_count, num_sum  
        END FUNCTION  
        
        SET left_side, right_side TO result of splitting equation by equal character  
        SET left_x, left_num TO result of calling parse_side WITH left_side  
        SET right_x, right_num TO result of calling parse_side WITH right_side  
        
        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  
        
        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN string Infinite solutions  
            ELSE  
                RETURN string No solution  
            END IF  
        ELSE  
            RETURN concatenation of character x, equal character, integer division of total_num BY total_x  
        END IF  
    END FUNCTION  
END CLASS
