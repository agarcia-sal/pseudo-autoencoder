CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one TO zero STEP minus one  
            SET total_shifts TO total_shifts PLUS element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        
        SET result TO empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET new_character_code TO the remainder of the sum of the difference between the code of character at position index of s and the code of character a PLUS element at position index of shifts DIVIDED BY twenty six  
            SET new_character_code TO new_character_code PLUS the code of character a  
            SET new_character TO the character corresponding to new_character_code  
            APPEND new_character TO result  
        END FOR  
        
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
