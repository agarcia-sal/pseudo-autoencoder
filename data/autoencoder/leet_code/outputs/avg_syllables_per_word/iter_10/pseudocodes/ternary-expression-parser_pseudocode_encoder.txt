CLASS Solution
    FUNCTION parseTernary(expression)
        FUNCTION evaluate(index)
            IF index EQUALS the LENGTH OF expression MINUS one OR character at position index PLUS one of expression EQUALS colon
                RETURN character at position index of expression, index PLUS two
            END IF
            SET true_part, next_index TO evaluate(index PLUS two)
            SET false_part, final_index TO evaluate(next_index)
            IF character at position index of expression EQUALS letter T
                RETURN true_part, final_index
            ELSE
                RETURN false_part, final_index
            END IF
        END FUNCTION
        SET result, underscore TO evaluate(zero)
        RETURN result
    END FUNCTION
END CLASS
