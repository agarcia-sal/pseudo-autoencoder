CLASS Solution  
    FUNCTION nthMagicalNumber(n, a, b)  
        SET MOD TO one billion PLUS seven  
        
        FUNCTION gcd(x, y)  
            WHILE y IS NOT zero  
                SET temp_x TO y  
                SET temp_y TO x MINUS (x DIVIDED BY y MULTIPLIED BY y)  
                SET x TO temp_x  
                SET y TO temp_y  
            END WHILE  
            RETURN x  
        END FUNCTION  
        
        FUNCTION lcm(x, y)  
            SET gcd_value TO CALL gcd WITH x, y  
            SET result TO x MULTIPLIED BY y DIVIDED BY gcd_value  
            RETURN result  
        END FUNCTION  
        
        SET L TO CALL lcm WITH a, b  
        SET left TO one  
        SET right TO n MULTIPLIED BY the smaller OF a AND b  
        
        WHILE left IS LESS THAN right  
            SET mid TO the integer division OF left PLUS right BY two  
            SET count TO mid DIVIDED BY a PLUS mid DIVIDED BY b MINUS mid DIVIDED BY L  
            IF count IS LESS THAN n  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left MODULO MOD  
    END FUNCTION  
END CLASS
