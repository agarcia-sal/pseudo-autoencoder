CLASS ListNode  
    FUNCTION __init__ with parameters val EQUALS zero, next EQUALS none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO a new ListNode with parameter the element at position zero of values  
    SET p TO head  
    FOR each val IN values starting from position one TO the last position  
        SET node TO a new ListNode with parameter val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1, p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the element val of p1 EQUALS the element val of p2 AND is_same_list with parameters the element next of p1, the element next of p2  
END FUNCTION

        FUNCTION reverseList with parameter node  
            SET prev TO none  
            WHILE node IS truthy  
                SET next_node TO the element next of node  
                SET the element next of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO reverseList with parameter head  
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current IS truthy AND carry EQUALS one  
            INCREMENT the element val of current BY carry  
            SET carry TO the element val of current DIVIDED BY ten  
            SET the element val of current TO the element val of current MODULO ten  
            
            IF the element next of current IS none AND carry EQUALS one  
                SET the element next of current TO a new ListNode with parameter carry  
                SET carry TO zero

            SET prev TO current

        END WHILE  
        
        RETURN reverseList with parameter reversed_head
