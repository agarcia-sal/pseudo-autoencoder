CLASS Solution  
    FUNCTION replaceNonCoprimes WITH PARAMETER nums  
        SET stack TO empty list  
        FOR each num IN nums  
            WHILE the LENGTH OF stack IS GREATER THAN zero  
                SET g TO gcd of element at position LENGTH OF stack MINUS one of stack AND num  
                IF g EQUALS one  
                    BREAK out of WHILE loop  
                END IF  
                SET num TO element at position LENGTH OF stack MINUS one of stack MULTIPLIED BY num DIVIDED BY g  
                REMOVE last element FROM stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
