CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture EQUALS empty OR element at position zero of picture EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO a list of zero REPEATED m TIMES  
        SET col_count TO a list of zero REPEATED n TIMES  
        CALL defaultdict WITH int AND SET TO row_patterns  
        
        FOR r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of all elements at positions of picture at r  
            INCREMENT the value for key pattern in row_patterns BY one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of picture at r EQUALS character B  
                    INCREMENT element at position r of row_count BY one  
                    INCREMENT element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of picture at r EQUALS character B  
                    SET pattern TO the concatenation of all elements at positions of picture at r  
                    IF element at position r of row_count EQUALS target AND element at position c of col_count EQUALS target AND the value for key pattern in row_patterns EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
