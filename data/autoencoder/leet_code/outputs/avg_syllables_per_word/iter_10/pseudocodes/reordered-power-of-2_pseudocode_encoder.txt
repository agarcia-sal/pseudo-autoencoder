CLASS Solution  
    FUNCTION reorderedPowerOf2(n)  
        SET power_of_2_sorted_digits TO an empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 LESS THAN OR EQUAL TO ten raised TO the power of nine  
            COMPUTE the sorted digits of the string representation of power_of_2  
            ADD this sorted digits tuple TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  
        COMPUTE the sorted digits of the string representation of n  
        RETURN whether sorted digits of n ARE IN power_of_2_sorted_digits  
    END FUNCTION  
END CLASS
