CLASS Solution  
    FUNCTION checkSubarraySum(nums, k)  
        SET remainder_map TO a dictionary with key zero MAPPED TO value minus one  
        SET current_sum TO zero  
        
        FOR each index i AND element num IN nums  
            INCREMENT current_sum BY num  
            
            IF k NOT EQUALS zero  
                SET current_sum TO current_sum MODULO k  
            END IF  
            
            IF current_sum EXISTS AS A KEY IN remainder_map  
                IF i MINUS the VALUE AT key current_sum IN remainder_map GREATER THAN one  
                    RETURN True  
                END IF  
            ELSE  
                SET the VALUE AT key current_sum IN remainder_map TO i  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
