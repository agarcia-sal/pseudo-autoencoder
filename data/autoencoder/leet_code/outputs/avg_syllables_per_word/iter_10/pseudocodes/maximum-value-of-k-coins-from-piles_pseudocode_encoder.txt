CLASS Solution  
    FUNCTION maxValueOfCoins(piles, k)  
        SET n TO the LENGTH OF piles  
        SET dp TO a list with n PLUS one elements each being a list of k PLUS one zeros  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one of piles  
                SET current_value TO zero  
                FOR l FROM zero TO MINIMUM OF j AND the LENGTH OF current_pile  
                    IF l GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one of current_pile  
                    END IF  
                    SET dp_element_at_i_j TO element at position i and j of dp  
                    SET candidate_value TO element at position i MINUS one and j MINUS l of dp PLUS current_value  
                    IF candidate_value GREATER THAN dp_element_at_i_j  
                        SET element at position i and j of dp TO candidate_value  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position n and k of dp  
    END FUNCTION  
END CLASS
