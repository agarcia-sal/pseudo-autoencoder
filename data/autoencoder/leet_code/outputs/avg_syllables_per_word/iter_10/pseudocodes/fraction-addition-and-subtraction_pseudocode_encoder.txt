CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET list_of_fractions TO the result of REPLACING each MINUS SIGN in expression WITH PLUS SIGN FOLLOWED BY MINUS SIGN THEN SPLITTING BY PLUS SIGN  
        SET result TO zero fraction  
        FOR each fraction_string IN list_of_fractions  
            IF fraction_string IS NOT empty  
                ADD fraction represented by fraction_string TO result  
            END IF  
        END FOR  
        RETURN the NUMERATOR of result CONCATENATED WITH forward slash CONCATENATED WITH the DENOMINATOR of result  
    END FUNCTION  
END CLASS
