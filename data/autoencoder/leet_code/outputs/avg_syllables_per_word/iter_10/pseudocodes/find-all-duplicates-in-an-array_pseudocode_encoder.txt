CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            WHILE the element at position i of nums NOT EQUALS the element at position the element at position i of nums MINUS one of nums  
                SET temporary_value TO the element at position i of nums  
                SET the element at position i of nums TO the element at position temporary_value MINUS one of nums  
                SET the element at position temporary_value MINUS one of nums TO temporary_value  
            END WHILE  
        END FOR  
        SET duplicates_list TO empty list  
        FOR each index i AND value v FROM enumerating nums  
            IF v NOT EQUALS i PLUS one  
                APPEND v TO duplicates_list  
            END IF  
        END FOR  
        RETURN duplicates_list  
    END FUNCTION  
END CLASS
