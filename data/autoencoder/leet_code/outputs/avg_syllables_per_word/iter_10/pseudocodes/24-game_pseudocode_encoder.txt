CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF element at position zero of nums MINUS twenty FOUR LESS THAN one TIMES ten TO THE POWER OF MINUS six  
            END IF  
            
            FOR i FROM zero TO the LENGTH OF nums MINUS one  
                FOR j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO an empty list  
                        FOR k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR each op IN the list consisting of true division, multiplication, addition, subtraction  
                            IF (op EQUALS addition OR op EQUALS multiplication) AND j GREATER THAN i  
                                CONTINUE with the next iteration  
                            END IF  
                            
                            IF op NOT EQUAL TO true division OR element at position j of nums NOT EQUAL TO zero  
                                APPEND the result of applying op TO element at position i of nums AND element at position j of nums TO new_nums  
                                IF solve(new_nums) RETURNS True  
                                    RETURN True  
                                END IF  
                                REMOVE last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
