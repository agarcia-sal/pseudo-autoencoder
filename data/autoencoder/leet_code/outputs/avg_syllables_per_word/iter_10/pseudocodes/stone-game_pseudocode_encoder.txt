CLASS Solution  
    FUNCTION stoneGame WITH parameter piles  
        FUNCTION dfs WITH parameters i, j  
            IF i GREATER THAN j  
                RETURN zero  
            END IF  
            SET left_choice TO element at position i of piles MINUS dfs WITH parameters i PLUS one, j  
            SET right_choice TO element at position j of piles MINUS dfs WITH parameters i, j MINUS one  
            SET result TO the GREATER OF left_choice AND right_choice  
            RETURN result  
        END FUNCTION  
        SET final_result TO dfs WITH parameters zero, the LENGTH OF piles MINUS one  
        RETURN final_result GREATER THAN zero  
    END FUNCTION  
END CLASS
