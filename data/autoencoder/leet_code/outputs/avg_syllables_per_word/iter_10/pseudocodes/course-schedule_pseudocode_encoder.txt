CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        SET graph TO a mapping from keys to empty lists  
        SET indegree TO a list of zeros with length numCourses  
        
        FOR each pair course and prereq IN prerequisites  
            APPEND course TO the list at key prereq in graph  
            INCREMENT element at position course of indegree BY one  
        END FOR  
        
        SET queue TO a double-ended queue containing each course FROM zero TO numCourses MINUS one WHERE element at position course of indegree EQUALS zero  
        
        WHILE queue is not empty  
            REMOVE first element FROM queue AND SET TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN the list at key current_course in graph  
                DECREMENT element at position next_course of indegree BY one  
                IF element at position next_course of indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN numCourses EQUALS zero  
    END FUNCTION  
END CLASS
