CLASS Solution  
    FUNCTION decodeAtIndex(s, k)  
        SET length TO zero  
        
        FOR each char IN s  
            IF char IS a digit  
                SET length TO length MULTIPLIED BY integer value of char  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        
        FOR each char IN reverse order of s  
            SET k TO k MODULO length  
            
            IF k EQUALS zero AND char IS a letter  
                RETURN char  
            END IF  
            
            IF char IS a digit  
                SET length TO length DIVIDED BY integer value of char  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
