CLASS Solution  
    FUNCTION new21Game(n, k, maxPts)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k  
                IF i LESS THAN OR EQUAL TO n  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF i EQUALS k MINUS one  
                SET result TO minimum of (n MINUS k PLUS one) AND maxPts DIVIDED BY maxPts  
                RETURN result  
            END IF  
            SET first_dfs TO dfs(i PLUS one)  
            SET second_dfs TO dfs(i PLUS maxPts PLUS one)  
            SET result TO first_dfs PLUS (first_dfs MINUS second_dfs) DIVIDED BY maxPts  
            RETURN result  
        END FUNCTION  
        RETURN dfs(zero)  
    END FUNCTION  
END CLASS
