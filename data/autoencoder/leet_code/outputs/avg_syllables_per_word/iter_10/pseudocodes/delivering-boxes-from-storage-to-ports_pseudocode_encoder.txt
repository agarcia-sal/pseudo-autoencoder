CLASS Solution  
    FUNCTION boxDelivering(boxes, portsCount, maxBoxes, maxWeight)  
        SET n TO the LENGTH OF boxes  
        SET dp TO a list of size n PLUS one FILLED WITH infinite value  
        SET the element at position zero of dp TO zero  
        
        SET left TO zero  
        SET current_weight TO zero  
        SET extra_trips TO zero  
        
        FOR right FROM zero TO n MINUS one  
            INCREMENT current_weight BY the second element of the element at position right of boxes  
            IF right GREATER THAN zero AND the first element of the element at position right of boxes NOT EQUALS the first element of the element at position right MINUS one of boxes  
                INCREMENT extra_trips BY one  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND (right MINUS left PLUS one GREATER THAN maxBoxes OR current_weight GREATER THAN maxWeight)  
                DECREMENT current_weight BY the second element of the element at position left of boxes  
                IF left GREATER THAN zero AND the first element of the element at position left of boxes NOT EQUALS the first element of the element at position left MINUS one of boxes  
                    DECREMENT extra_trips BY one  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET the element at position right PLUS one of dp TO the element at position left of dp PLUS two PLUS extra_trips  
        END FOR  
        
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
