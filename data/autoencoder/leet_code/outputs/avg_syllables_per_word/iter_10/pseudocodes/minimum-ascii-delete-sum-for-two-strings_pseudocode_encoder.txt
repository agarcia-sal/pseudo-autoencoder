CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        SET dp TO a two dimensional list of size m PLUS one by n PLUS one filled with zeros  
        FOR j FROM one TO n  
            SET element at position zero comma j of dp TO element at position zero comma j MINUS one of dp PLUS the ASCII VALUE OF character at position j MINUS one of s2  
        END FOR  
        FOR i FROM one TO m  
            SET element at position i comma zero of dp TO element at position i MINUS one comma zero of dp PLUS the ASCII VALUE OF character at position i MINUS one of s1  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one of s1 EQUALS character at position j MINUS one of s2  
                    SET element at position i comma j of dp TO element at position i MINUS one comma j MINUS one of dp  
                ELSE  
                    SET element at position i comma j of dp TO the MINIMUM OF element at position i MINUS one comma j of dp PLUS the ASCII VALUE OF character at position i MINUS one of s1 AND element at position i comma j MINUS one of dp PLUS the ASCII VALUE OF character at position j MINUS one of s2  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position m comma n of dp  
    END FUNCTION  
END CLASS
