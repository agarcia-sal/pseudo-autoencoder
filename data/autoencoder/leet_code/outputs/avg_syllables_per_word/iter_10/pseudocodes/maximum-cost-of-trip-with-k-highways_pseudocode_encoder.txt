CLASS Solution  
    FUNCTION maximumCost(n, highways, k)  
        DEFINE FUNCTION dfs(city, visited, cost, highways_used)  
            IF highways_used EQUALS k  
                SET max_cost TO the GREATER OF max_cost AND cost  
                RETURN  
            END IF  
            FOR each element consisting of neighbor and toll IN the list at key city OF graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    CALL dfs WITH neighbor, visited, cost PLUS toll, highways_used PLUS one  
                    REMOVE neighbor FROM visited  
                END IF  
            END FOR  
        END FUNCTION  
        
        INITIALIZE graph TO an empty mapping with default value as empty list  
        FOR each element consisting of city1, city2, and toll IN highways  
            APPEND the pair of city2 and toll TO the list at key city1 OF graph  
            APPEND the pair of city1 and toll TO the list at key city2 OF graph  
        END FOR  
        
        SET max_cost TO minus one  
        FOR start_city FROM zero TO n MINUS one  
            INITIALIZE visited TO a set containing start_city  
            CALL dfs WITH start_city, visited, zero, zero  
        END FOR  
        RETURN max_cost  
    END FUNCTION  
END CLASS
