CLASS Trie  
    FUNCTION __init__()  
        SET self.children TO a list of twenty-six None elements  
        SET self.is_end TO False  
    END FUNCTION  
  
    FUNCTION insert(w)  
        SET node TO self  
        FOR each character c IN w  
            SET idx TO the numeric code of character c MINUS the numeric code of character 'a'  
            IF the element at position idx of node.children EQUALS None THEN  
                SET the element at position idx of node.children TO a new Trie instance  
            END IF  
            SET node TO the element at position idx of node.children  
        END FOR  
        SET node.is_end TO True  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION findAllConcatenatedWordsInADict(words)  
        FUNCTION dfs(w, start)  
            IF start EQUALS the LENGTH OF w THEN  
                RETURN True  
            END IF  
            SET node TO trie  
            FOR index i FROM start TO the LENGTH OF w MINUS one  
                SET idx TO the numeric code of character at position i of w MINUS the numeric code of character 'a'  
                IF the element at position idx of node.children EQUALS None THEN  
                    RETURN False  
                END IF  
                SET node TO the element at position idx of node.children  
                IF node.is_end AND dfs(w, i PLUS one) THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
  
        SET trie TO a new Trie instance  
        SET ans TO an empty list  
        SORT words by the LENGTH OF each word in ascending order  
        FOR each word w IN words  
            IF w IS NOT the empty string THEN  
                IF dfs(w, zero) THEN  
                    APPEND w TO ans  
                ELSE  
                    trie.insert(w)  
                END IF  
            END IF  
        END FOR  
        RETURN ans  
    END FUNCTION  
END CLASS
