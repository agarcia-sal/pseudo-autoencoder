CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO the LENGTH OF arr  
        SET min_heap TO empty list  
        FOR index i FROM zero TO n MINUS one MINUS one  
            ADD the tuple of the element at position i of arr DIVIDED BY the element at position n MINUS one of arr, i, and n MINUS one TO min_heap using heap insertion  
        END FOR  
        
        FOR count FROM zero TO k MINUS one MINUS one  
            REMOVE the smallest tuple FROM min_heap and ASSIGN the values TO ignored_variable, i, j  
            IF j MINUS one IS GREATER THAN i  
                ADD the tuple of the element at position i of arr DIVIDED BY the element at position j MINUS one of arr, i, and j MINUS one TO min_heap using heap insertion  
            END IF  
        END FOR  
        
        REMOVE the smallest tuple FROM min_heap and ASSIGN the values TO ignored_variable, i, j  
        RETURN list containing the element at position i of arr and the element at position j of arr  
    END FUNCTION  
END CLASS
