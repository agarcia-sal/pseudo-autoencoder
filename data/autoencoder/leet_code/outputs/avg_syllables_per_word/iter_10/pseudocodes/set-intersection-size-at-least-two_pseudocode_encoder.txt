CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals BY the second element of interval AND BY negative first element of interval  
        SET first TO the second element of the first interval MINUS one  
        SET second TO the second element of the first interval  
        SET count TO two  
        FOR each start, end IN intervals from the second interval to the last interval  
            IF start GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
