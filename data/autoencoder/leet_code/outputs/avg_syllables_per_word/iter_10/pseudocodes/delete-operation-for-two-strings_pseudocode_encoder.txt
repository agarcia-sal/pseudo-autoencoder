CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
  
        SET dp TO the RESULT OF create_2D_array WITH parameters m PLUS one AND n PLUS one  
  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2  
                    SET the element at position i, j of dp TO the element at position i MINUS one, j MINUS one of dp PLUS one  
                ELSE  
                    SET the element at position i, j of dp TO the GREATER OF the element at position i MINUS one, j of dp AND the element at position i, j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
  
        SET lcs_length TO the element at position m, n of dp  
  
        RETURN m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  
  
    FUNCTION create_2D_array(rows, columns)  
        CREATE a list named result  
        FOR each index FROM zero TO rows MINUS one  
            CREATE a list named row with columns number of zeros  
            APPEND row TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
