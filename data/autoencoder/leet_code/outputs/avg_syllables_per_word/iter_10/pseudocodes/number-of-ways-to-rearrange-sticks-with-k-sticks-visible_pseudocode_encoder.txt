CLASS Solution  
    FUNCTION rearrangeSticks(n, k)  
        SET MOD TO ten RAISED TO the power of nine PLUS one  
        CALL initialize_dp WITH n PLUS one AND k PLUS one RETURNING dp  
        SET element at position zero of dp to position zero TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET dp_i_j_visible TO element at position i MINUS one of dp AT position j MINUS one  
                SET dp_i_j_not_visible TO i MINUS one MULTIPLIED BY element at position i MINUS one of dp AT position j  
                SET element at position i of dp AT position j TO dp_i_j_visible PLUS dp_i_j_not_visible  
                SET element at position i of dp AT position j TO element at position i of dp AT position j MODULO MOD  
            END FOR  
        END FOR  
        RETURN element at position n of dp AT position k  
    END FUNCTION

    FUNCTION initialize_dp(rows, columns)  
        SET dp TO empty list  
        FOR each index FROM zero TO rows MINUS one  
            SET row TO empty list  
            FOR each inner_index FROM zero TO columns MINUS one  
                APPEND zero TO row  
            END FOR  
            APPEND row TO dp  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
