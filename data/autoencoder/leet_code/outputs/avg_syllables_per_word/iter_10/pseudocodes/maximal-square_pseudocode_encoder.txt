CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET dp TO the RESULT OF calling helper FUNCTION initializeMatrix WITH parameters m PLUS one AND n PLUS one  
        SET max_side TO zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of matrix AND position j MINUS one EQUALS character ONE  
                    SET dp at position i AND j TO the MINIMUM OF dp at position i MINUS one AND j, dp at position i AND j MINUS one, AND dp at position i MINUS one AND j MINUS one PLUS one  
                    SET max_side TO the GREATER VALUE OF max_side AND dp at position i AND j  
                END IF  
            END FOR  
        END FOR  
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS  

FUNCTION initializeMatrix(rows, columns)  
    RETURN a two-dimensional list of zeros with rows number of rows AND columns number of columns  
END FUNCTION
