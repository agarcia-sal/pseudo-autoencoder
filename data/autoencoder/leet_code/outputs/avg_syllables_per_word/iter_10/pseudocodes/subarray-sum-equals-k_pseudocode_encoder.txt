CLASS Solution  
    FUNCTION subarraySum(nums, k)  
        SET cumulative_sum_freq TO dictionary with key zero SET TO one  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            
            IF cumulative_sum_freq CONTAINS key current_sum MINUS k  
                INCREMENT count BY the value of cumulative_sum_freq at key current_sum MINUS k  
            END IF  
            
            IF cumulative_sum_freq CONTAINS key current_sum  
                INCREMENT the value of cumulative_sum_freq at key current_sum BY one  
            ELSE  
                SET the value of cumulative_sum_freq at key current_sum TO one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
