CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND value v IN cnt  
                IF v GREATER THAN 1  
                    DECREMENT cnt at character c BY two  
                    CALL dfs WITH the CONCATENATION of character c PLUS t PLUS character c  
                    INCREMENT cnt at character c BY two  
                END IF  
            END FOR  
        END FUNCTION  

        SET cnt TO the Counter of s  
        SET mid TO the empty string  
        FOR each character c AND value v IN cnt  
            IF v MODULO two EQUALS one  
                IF mid IS NOT the empty string  
                    RETURN the empty list  
                END IF  
                SET mid TO character c  
                DECREMENT cnt at character c BY one  
            END IF  
        END FOR  
        SET ans TO the empty list  
        CALL dfs WITH mid  
        RETURN ans  
    END FUNCTION  
END CLASS
