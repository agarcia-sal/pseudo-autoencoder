CLASS Solution  
    FUNCTION minimumWeight(n, edges, src1, src2, dest)  
        FUNCTION dijkstra(graph, start)  
            SET distances TO a list of size n with all elements set TO infinity  
            SET element at position start of distances TO zero  
            SET heap TO a list containing a tuple of zero and start  
            
            WHILE heap is not empty  
                SET current_distance, current_node TO the result of removing the smallest element from heap  
                
                IF current_distance GREATER THAN element at position current_node of distances  
                    CONTINUE to the next iteration  
                END IF  
                
                FOR each neighbor, weight IN element at position current_node of graph  
                    SET distance TO current_distance PLUS weight  
                    
                    IF distance LESS THAN element at position neighbor of distances  
                        SET element at position neighbor of distances TO distance  
                        ADD the tuple of distance and neighbor TO heap  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN distances  
        END FUNCTION  
        
        SET graph TO an empty list repeated n times  
        SET reverse_graph TO an empty list repeated n times  
        
        FOR each u, v, w IN edges  
            APPEND the tuple of v and w TO element at position u of graph  
            APPEND the tuple of u and w TO element at position v of reverse_graph  
        END FOR  
        
        SET dist_from_src1 TO the result of dijkstra called with graph and src1  
        SET dist_from_src2 TO the result of dijkstra called with graph and src2  
        SET dist_to_dest TO the result of dijkstra called with reverse_graph and dest  
        
        SET min_weight TO infinity  
        
        FOR index FROM zero TO n MINUS one  
            IF element at position index of dist_from_src1 NOT EQUALS infinity AND  
               element at position index of dist_from_src2 NOT EQUALS infinity AND  
               element at position index of dist_to_dest NOT EQUALS infinity  
                SET sum_distances TO element at position index of dist_from_src1 PLUS element at position index of dist_from_src2 PLUS element at position index of dist_to_dest  
                IF sum_distances LESS THAN min_weight  
                    SET min_weight TO sum_distances  
                END IF  
            END IF  
        END FOR  
        
        IF min_weight NOT EQUALS infinity  
            RETURN min_weight  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
