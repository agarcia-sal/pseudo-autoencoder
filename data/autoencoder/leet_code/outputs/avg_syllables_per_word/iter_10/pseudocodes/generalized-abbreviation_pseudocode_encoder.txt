CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the STRING representation OF count TO path  
                END IF  
                APPEND the CONCATENATION OF elements OF path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH index PLUS one, path CONCATENATED WITH the STRING representation OF count AND the element at position index OF word, ZERO  
                ELSE  
                    CALL backtrack WITH index PLUS one, path CONCATENATED WITH the element at position index OF word, ZERO  
                END IF  
                CALL backtrack WITH index PLUS one, path, count PLUS one  
            END IF  
        END FUNCTION  

        SET result TO an empty list  
        CALL backtrack WITH zero, an empty list, zero  
        RETURN result  
    END FUNCTION  
END CLASS
