CLASS Solution  
    FUNCTION isValidPalindrome(s, k)  
        FUNCTION longest_palindromic_subseq(s)  
            SET n TO the LENGTH OF s  
            CALL initialize_dp_table WITH n, RETURNING dp  
            FOR index i FROM zero TO n MINUS one  
                SET element at position i and i of dp TO one  
            END FOR  
            FOR length FROM two TO n  
                FOR index i FROM zero TO n MINUS length  
                    SET j TO i PLUS length MINUS one  
                    IF character at position i of s EQUALS character at position j of s  
                        SET element at position i and j of dp TO element at position i PLUS one and j MINUS one of dp PLUS two  
                    ELSE  
                        SET element at position i and j of dp TO the GREATER OF element at position i PLUS one and j of dp AND element at position i and j MINUS one of dp  
                    END IF  
                END FOR  
            END FOR  
            RETURN element at position zero and n MINUS one of dp  
        END FUNCTION  
        
        SET longest_palindrome_length TO longest_palindromic_subseq(s)  
        RETURN the LENGTH OF s MINUS longest_palindrome_length LESS THAN OR EQUAL TO k  
    END FUNCTION  
  
    FUNCTION initialize_dp_table(n)  
        SET dp TO an empty list  
        FOR index FROM zero TO n MINUS one  
            APPEND a list of zeros of length n TO dp  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
