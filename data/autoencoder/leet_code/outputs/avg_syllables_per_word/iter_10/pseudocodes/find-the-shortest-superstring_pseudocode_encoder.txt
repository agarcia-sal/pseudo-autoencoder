CLASS Solution  
    FUNCTION shortestSuperstring(words)  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap(i, j)  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i of words AND the LENGTH OF element at position j of words DOWN TO one  
                IF the SUBSTRING of element at position i of words from the position LENGTH OF element at position i of words MINUS k PLUS one TO the END EQUALS the SUBSTRING of element at position j of words from the START TO position k THEN  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp(mask, i)  
            IF mask EQUALS two RAISED TO THE POWER OF n MINUS one THEN  
                RETURN zero AND the EMPTY STRING  
            END IF  
            
            SET min_len TO infinity  
            SET best_path TO the EMPTY STRING  
            FOR j FROM zero TO n MINUS one  
                IF bitwise AND of mask AND two RAISED TO THE POWER OF j EQUALS zero THEN  
                    SET length AND path TO dp(bitwise OR of mask AND two RAISED TO THE POWER OF j, j)  
                    SET length TO length PLUS the LENGTH OF element at position j of words MINUS overlap(i, j)  
                    IF length LESS THAN min_len THEN  
                        SET min_len TO length  
                        SET best_path TO the SUBSTRING of element at position j of words from position overlap(i, j) TO THE END CONCATENATED WITH path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO infinity  
        SET shortest_path TO the EMPTY STRING  
        FOR i FROM zero TO n MINUS one  
            SET length AND path TO dp(two RAISED TO THE POWER OF i, i)  
            SET length TO length PLUS the LENGTH OF element at position i of words  
            IF length LESS THAN min_len THEN  
                SET min_len TO length  
                SET shortest_path TO element at position i of words CONCATENATED WITH path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
