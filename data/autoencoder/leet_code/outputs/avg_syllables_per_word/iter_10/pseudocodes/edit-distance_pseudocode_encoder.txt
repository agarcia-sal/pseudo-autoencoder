CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        CALL initializeTwoDimensionalList WITH parameters m PLUS one AND n PLUS one AND SET TO dp  
        
        FOR i FROM zero TO m  
            SET element at position zero of element at position i of dp TO i  
        END FOR  
        FOR j FROM zero TO n  
            SET element at position j of element at position zero of dp TO j  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of word1 EQUALS element at position j MINUS one of word2  
                    SET element at position j of element at position i of dp TO element at position j MINUS one of element at position i MINUS one of dp  
                ELSE  
                    SET temp_min TO the MINIMUM OF element at position j of element at position i MINUS one of dp, element at position j MINUS one of element at position i of dp, element at position j MINUS one of element at position i MINUS one of dp  
                    SET element at position j of element at position i of dp TO one PLUS temp_min  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of element at position m of dp  
    END FUNCTION  
END CLASS  

FUNCTION initializeTwoDimensionalList(rows, columns)  
    CREATE an empty list called result  
    FOR each number FROM zero TO rows MINUS one  
        CREATE a list of zeros of length columns  
        APPEND this list TO result  
    END FOR  
    RETURN result  
END FUNCTION
