CLASS Solution  
    FUNCTION ladderLength(beginWord, endWord, wordList)  
        IF endWord NOT IN wordList  
            RETURN zero  
        END IF  

        SET word_dict TO a new dictionary with default value as empty list  
        SET word_length TO the LENGTH OF beginWord  

        FOR each word IN wordList  
            FOR index FROM zero TO word_length MINUS one  
                SET intermediate_key TO concatenation of substring of word from start TO index MINUS one PLUS asterisk PLUS substring of word from index PLUS one TO end  
                APPEND word TO list at key intermediate_key in word_dict  
            END FOR  
        END FOR  

        SET queue TO a new deque containing tuple of beginWord and one  
        SET visited TO a new set containing beginWord  

        WHILE queue is NOT empty  
            REMOVE first element FROM queue INTO current_word AND current_length  

            FOR index FROM zero TO word_length MINUS one  
                SET intermediate_word TO concatenation of substring of current_word from start TO index MINUS one PLUS asterisk PLUS substring of current_word from index PLUS one TO end  

                FOR each word IN list at key intermediate_word in word_dict  
                    IF word EQUALS endWord  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited  
                        ADD word TO visited  
                        APPEND tuple of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN zero  
    END FUNCTION  
END CLASS
