CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO the concatenation of all words in sentence separated by a single space FOLLOWED BY a single space  
        SET total_length TO the LENGTH OF s  
        SET position TO zero  
        
        FOR each _ FROM one TO rows  
            INCREMENT position BY cols  
            
            IF the character in s at position MODULO total_length EQUALS a space  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the character in s at (position MINUS one) MODULO total_length NOT EQUALS a space  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        
        RETURN position DIVIDED BY total_length  
    END FUNCTION  
END CLASS
