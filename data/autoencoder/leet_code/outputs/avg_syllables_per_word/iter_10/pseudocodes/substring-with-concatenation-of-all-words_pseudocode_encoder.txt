CLASS Solution  
    FUNCTION findSubstring(s, words)  
        IF s EQUALS empty OR words EQUALS empty OR the element at position zero of words EQUALS empty  
            RETURN empty list  
        END IF  
        
        SET word_length TO the LENGTH OF the element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the result of Counter applied to words  
        SET result TO empty list  
        
        FOR index i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO an empty Counter  
            FOR index j FROM i TO i PLUS total_length, with step size word_length  
                SET word TO the substring of s from position j TO position j PLUS word_length  
                INCREMENT the count of word in seen BY one  
                IF the count of word in seen GREATER THAN the count of word in word_count  
                    BREAK  
                END IF  
            END FOR  
            IF the inner loop was not broken  
                APPEND i TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
