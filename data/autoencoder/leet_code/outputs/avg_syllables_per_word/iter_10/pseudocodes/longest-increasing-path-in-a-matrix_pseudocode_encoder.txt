CLASS Solution  
    FUNCTION longestIncreasingPath(matrix)  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET directions TO a list containing the elements ONE AND ZERO, MINUS ONE AND ZERO, ZERO AND ONE, ZERO AND MINUS ONE  
        SET dp TO the result of calling the FUNCTION create_2d_list WITH parameters m AND n AND initial value MINUS one  
        
        FUNCTION dfs(x, y)  
            IF element at position x of dp at position y NOT EQUALS MINUS one  
                RETURN element at position x of dp at position y  
            END IF  
            
            SET max_length TO one  
            FOR each dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx of matrix at position ny GREATER THAN element at position x of matrix at position y  
                    SET max_length TO the GREATER OF max_length AND one PLUS dfs(nx, ny)  
                END IF  
            END FOR  
            
            SET element at position x of dp at position y TO max_length  
            RETURN max_length  
        END FUNCTION  
        
        SET longest_path TO zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET longest_path TO the GREATER OF longest_path AND dfs(i, j)  
            END FOR  
        END FOR  
        
        RETURN longest_path  
    END FUNCTION  
  
    FUNCTION create_2d_list(m, n, initial_value)  
        SET result TO an empty list  
        FOR i FROM zero TO m MINUS one  
            SET row TO an empty list  
            FOR j FROM zero TO n MINUS one  
                APPEND initial_value TO row  
            END FOR  
            APPEND row TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
