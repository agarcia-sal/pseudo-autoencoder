CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        SET dp TO a two-dimensional list with dimensions m PLUS one by n PLUS one filled with zero  
        FOR each s IN strs  
            SET count_0 TO the number of times character zero APPEARS IN s  
            SET count_1 TO the number of times character one APPEARS IN s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp element at position i and j TO the maximum OF dp element at position i and j AND dp element at position i MINUS count_0 and j MINUS count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position m and n  
    END FUNCTION  
END CLASS
