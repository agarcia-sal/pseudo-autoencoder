CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO zero  
        FOR each number IN nums  
            INCREMENT total_sum BY number  
        END FOR  
 
        IF total_sum MODULO two NOT EQUALS zero  
            RETURN false  
        END IF  
 
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
 
        CALL initialize_dp_array(target) RETURNING dp  
 
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET dp element at position i TO dp element at position i OR dp element at position i MINUS num  
            END FOR  
        END FOR  
 
        RETURN dp element at position target  
    END FUNCTION  
 
    FUNCTION initialize_dp_array(target)  
        CREATE a list dp WITH target PLUS one elements ALL SET TO false  
        SET dp element at position zero TO true  
        RETURN dp  
    END FUNCTION  
END CLASS
