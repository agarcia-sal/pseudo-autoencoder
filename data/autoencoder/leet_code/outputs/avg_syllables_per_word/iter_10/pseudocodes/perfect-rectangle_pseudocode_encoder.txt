CLASS Solution  
    FUNCTION isRectangleCover(rectangles)  
        SET X1 TO positive infinity  
        SET Y1 TO positive infinity  
        SET X2 TO negative infinity  
        SET Y2 TO negative infinity  
        
        SET points TO an empty set  
        SET area TO zero  
        
        FOR each rectangle IN rectangles  
            DESTRUCTURE rectangle INTO x1, y1, x2, y2  
            
            SET X1 TO the smaller OF X1 AND x1  
            SET Y1 TO the smaller OF Y1 AND y1  
            SET X2 TO the larger OF X2 AND x2  
            SET Y2 TO the larger OF Y2 AND y2  
            
            INCREMENT area BY (x2 MINUS x1) MULTIPLIED BY (y2 MINUS y1)  
            
            FOR each point p IN list of points consisting of (x1, y1), (x1, y2), (x2, y1), and (x2, y2)  
                IF p IS IN points  
                    REMOVE p FROM points  
                ELSE  
                    ADD p TO points  
                END IF  
            END FOR  
        END FOR  
        
        IF area IS NOT EQUALS TO (X2 MINUS X1) MULTIPLIED BY (Y2 MINUS Y1)  
            RETURN False  
        END IF  
        
        IF points IS NOT EQUALS TO set consisting of (X1, Y1), (X1, Y2), (X2, Y1), and (X2, Y2)  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
