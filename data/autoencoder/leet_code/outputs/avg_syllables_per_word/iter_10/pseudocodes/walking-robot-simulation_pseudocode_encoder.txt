CLASS Solution  
    FUNCTION robotSim(commands, obstacles)  
        SET directions TO list containing tuple zero PLUS one, tuple one PLUS zero, tuple zero MINUS one, tuple MINUS one PLUS zero  
        SET x TO zero  
        SET y TO zero  
        SET dir_index TO zero  
        SET obstacles_set TO set of tuples converted from each element in obstacles  
        SET max_distance TO zero  
        FOR each command IN commands  
            IF command EQUALS MINUS two  
                SET dir_index TO dir_index MINUS one MODULO four  
            ELSE IF command EQUALS MINUS one  
                SET dir_index TO dir_index PLUS one MODULO four  
            ELSE  
                SET dx TO element at position dir_index FIRST element of directions  
                SET dy TO element at position dir_index SECOND element of directions  
                FOR step FROM one TO command  
                    SET next_x TO x PLUS dx  
                    SET next_y TO y PLUS dy  
                    IF tuple of next_x AND next_y IS IN obstacles_set  
                        BREAK  
                    END IF  
                    SET x TO next_x  
                    SET y TO next_y  
                END FOR  
                SET current_distance TO x MULTIPLIED BY x PLUS y MULTIPLIED BY y  
                IF current_distance GREATER THAN max_distance  
                    SET max_distance TO current_distance  
                END IF  
            END IF  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
