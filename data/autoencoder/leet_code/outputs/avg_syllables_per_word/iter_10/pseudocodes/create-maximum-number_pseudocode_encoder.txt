CLASS Solution  
    FUNCTION maxNumber(nums1, nums2, k)  
        FUNCTION maxSingleArray(nums, k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO empty list  
            FOR each num IN nums  
                WHILE drop IS GREATER THAN zero AND stack IS NOT empty AND the element at the last position of stack IS LESS THAN num  
                    REMOVE last element FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the first k elements OF stack  
        END FUNCTION  
        
        FUNCTION merge(arr1, arr2)  
            SET result TO empty list  
            FOR each element FROM one TO the SUM OF the LENGTH OF arr1 AND the LENGTH OF arr2  
                SET chosen_list TO the list that IS GREATER BETWEEN arr1 AND arr2 when compared lexicographically  
                REMOVE first element FROM chosen_list AND APPEND it TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        SET max_number TO empty list  
        FOR i FROM zero TO k  
            IF i IS LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i IS LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO merge(maxSingleArray(nums1, i), maxSingleArray(nums2, k MINUS i))  
                SET max_number TO the greater list BETWEEN max_number AND candidate when compared lexicographically  
            END IF  
        END FOR  
        
        RETURN max_number  
    END FUNCTION  
END CLASS
