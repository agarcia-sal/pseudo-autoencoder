CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF state EXISTS IN memo  
                RETURN the element associated with state IN memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring from element at position index TO element at position index PLUS one OF state EQUALS two PLUS signs  
                    SET new_state TO the concatenation of substring from start TO element before position index OF state PLUS two minus signs PLUS substring from element at position index PLUS two TO end OF state  
                    IF NOT can_win_helper(new_state)  
                        SET the element associated with state IN memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the element associated with state IN memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN can_win_helper(currentState)  
    END FUNCTION  
END CLASS
