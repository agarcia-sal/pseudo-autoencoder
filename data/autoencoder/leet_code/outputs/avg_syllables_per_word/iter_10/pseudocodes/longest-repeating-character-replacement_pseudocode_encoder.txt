CLASS Solution  
    FUNCTION characterReplacement(s, k)  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        SET count TO a new collection mapping characters to zero counts  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT count of element at position right of s BY one  
            IF max_count LESS THAN count of element at position right of s THEN  
                SET max_count TO count of element at position right of s  
            END IF  
            
            IF the LENGTH OF the window from left TO right MINUS max_count GREATER THAN k THEN  
                DECREMENT count of element at position left of s BY one  
                INCREMENT left BY one  
            END IF  
            
            IF max_length LESS THAN the LENGTH OF the window from left TO right THEN  
                SET max_length TO the LENGTH OF the window from left TO right  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
