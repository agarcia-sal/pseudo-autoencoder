CLASS Solution
    FUNCTION smallestGoodBase(n)
        SET n TO integer conversion of n
        SET max_length TO integer conversion of logarithm base two of n PLUS one
        FOR m FROM max_length DOWN TO two
            SET k TO integer conversion of n RAISED TO THE POWER of one DIVIDED BY m MINUS one
            IF k LESS THAN two
                CONTINUE to next iteration
            END IF
            SET num TO integer division of k RAISED TO THE POWER m MINUS one BY k MINUS one
            IF num EQUALS n
                RETURN string conversion of k
            END IF
        END FOR
        RETURN string conversion of n MINUS one
    END FUNCTION
END CLASS
