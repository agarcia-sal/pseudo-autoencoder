CLASS ListNode  
    FUNCTION __init__ with parameters val, next  
        SET val TO val  
        SET next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO a new ListNode with value at position zero of values  
    SET p TO head  
    FOR each val IN values starting from position one to the end  
        SET node TO a new ListNode with val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1, p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN result of comparison if value at p1 EQUALS value at p2 AND is_same_list with parameters next of p1, next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList with parameter head  
        IF head IS None OR next of head IS None OR next of next of head IS None  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO next of head  
        SET even_head TO even  
        WHILE even IS NOT None AND next of even IS NOT None  
            SET next of odd TO next of even  
            SET odd TO next of odd  
            SET next of even TO next of odd  
            SET even TO next of even  
        END WHILE  
        SET next of odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
