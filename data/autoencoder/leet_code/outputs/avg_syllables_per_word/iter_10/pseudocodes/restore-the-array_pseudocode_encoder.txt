CLASS Solution  
    FUNCTION numberOfArrays(s, k)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a list of zero repeated n PLUS one times  
        SET the element at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO i MINUS one  
                IF the character at position j of s NOT EQUALS zero AND the integer representation of the substring from position j TO position i MINUS one OF s IS GREATER THAN OR EQUAL TO one AND IS LESS THAN OR EQUAL TO k  
                    SET the element at position i of dp TO the sum of the element at position i of dp AND the element at position j of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
