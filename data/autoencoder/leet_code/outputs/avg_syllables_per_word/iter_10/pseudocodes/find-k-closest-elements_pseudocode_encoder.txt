CLASS Solution  
    FUNCTION findClosestElements(arr, k, x)  
        SET left TO bisect the left position of x in arr MINUS one  
        SET right TO left PLUS one  
        
        WHILE right MINUS left MINUS one LESS THAN k  
            IF left EQUALS minus one OR (right LESS THAN the LENGTH OF arr AND the absolute value of the element at position right of arr MINUS x LESS THAN the absolute value of the element at position left of arr MINUS x)  
                INCREMENT right BY one  
            ELSE  
                DECREMENT left BY one  
            END IF  
        END WHILE  
        
        RETURN the segment of arr from position left PLUS one TO position right  
    END FUNCTION  
END CLASS
