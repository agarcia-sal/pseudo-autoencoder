CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO zero  
        FOR i FROM thirty-one TO zero DECREMENT BY one  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            SET prefixes TO an empty set  
            FOR each num IN nums  
                SET prefix TO num DIVIDED BY two RAISED TO THE POWER OF i  
                ADD prefix TO prefixes  
            END FOR  
            FOR each prefix IN prefixes  
                IF the result of current_candidate EXCLUSIVE OR prefix EXISTS IN prefixes  
                    SET max_xor TO current_candidate  
                    BREAK the current loop  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
