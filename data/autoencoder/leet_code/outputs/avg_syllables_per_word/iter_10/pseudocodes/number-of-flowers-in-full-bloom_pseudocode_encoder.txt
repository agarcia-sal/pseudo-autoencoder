CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO the sorted list of elements s for each pair s and e in flowers  
        SET end_times TO the sorted list of elements e for each pair s and e in flowers  
        SET result TO empty list  
        FOR each person IN people  
            SET started TO the position found by bisecting to the right in start_times with person  
            SET ended TO the position found by bisecting to the left in end_times with person  
            APPEND started MINUS ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
