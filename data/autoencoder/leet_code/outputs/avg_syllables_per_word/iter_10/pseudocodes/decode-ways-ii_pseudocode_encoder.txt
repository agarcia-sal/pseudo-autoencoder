CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a list of zeros with the length equal to the LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  

        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS the character star  
                RETURN nine  
            ELSE IF char EQUALS the character zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION

        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS the character star AND char2 EQUALS the character star  
                RETURN fifteen  
            ELSE IF char1 EQUALS the character star  
                IF char2 LESS THAN OR EQUAL TO the character six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS the character star  
                IF char1 EQUALS the character one  
                    RETURN nine  
                ELSE IF char1 EQUALS the character two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer value of char1 appended by char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION

        FOR each index FROM one TO the LENGTH OF s  
            SET dp at position index TO the remainder after division by MOD of dp at position index PLUS dp at position index MINUS one MULTIPLIED BY ways_to_decode_single of the element at position index MINUS one of s  
            IF index GREATER THAN one  
                SET dp at position index TO the remainder after division by MOD of dp at position index PLUS dp at position index MINUS two MULTIPLIED BY ways_to_decode_pair of the element at position index MINUS two of s AND the element at position index MINUS one of s  
            END IF  
        END FOR  

        RETURN the element at the last position of dp  
    END FUNCTION  
END CLASS
