CLASS Solution  
    FUNCTION maxRotateFunction(nums)  
        SET n TO the LENGTH OF nums  
        IF n EQUALS one  
            RETURN zero  
        END IF  
          
        SET total_sum TO the SUM OF all elements IN nums  
        SET current_sum TO the SUM OF the ELEMENT AT position i OF nums MULTIPLIED BY i FOR each i AND num IN enumeration OF nums  
          
        SET max_value TO current_sum  
          
        FOR k FROM one TO n MINUS one  
            SET current_sum TO current_sum PLUS total_sum MINUS n MULTIPLIED BY ELEMENT AT position n MINUS k OF nums  
            IF max_value LESS THAN current_sum  
                SET max_value TO current_sum  
            END IF  
        END FOR  
          
        RETURN max_value  
    END FUNCTION  
END CLASS
