CLASS Solution  
    FUNCTION canConvert(str1, str2)  
        IF str1 EQUALS str2  
            RETURN True  
        END IF  

        SET conversion_map TO empty mapping  

        FOR each char1, char2 IN corresponding elements OF str1 AND str2  
            IF char1 IN conversion_map  
                IF value corresponding TO char1 IN conversion_map NOT EQUALS char2  
                    RETURN False  
                END IF  
            ELSE  
                SET conversion_map AT char1 TO char2  
            END IF  
        END FOR  

        IF number OF unique characters IN str2 EQUALS twenty six  
            RETURN False  
        END IF  

        RETURN True  
    END FUNCTION  
END CLASS
