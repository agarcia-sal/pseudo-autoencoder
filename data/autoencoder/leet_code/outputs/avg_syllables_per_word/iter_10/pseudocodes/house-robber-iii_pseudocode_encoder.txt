CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero, left EQUALS None, right EQUALS None  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETERS values  
    IF values EQUALS empty OR values EQUALS None  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH val EQUALS element at position zero of values  
    SET i TO one  
    INITIALIZE queue AS empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE first element FROM queue AND SET TO node  
        IF i LESS THAN length of values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode WITH val EQUALS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode WITH val EQUALS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left, q.left AND is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION rob WITH PARAMETERS root  
        FUNCTION rob_from_node WITH PARAMETERS node  
            IF node EQUALS None  
                RETURN zero, zero  
            END IF  
            SET left_robbed, left_not_robbed TO rob_from_node WITH PARAMETERS node.left  
            SET right_robbed, right_not_robbed TO rob_from_node WITH PARAMETERS node.right  
            SET rob_this_node TO node.val PLUS left_not_robbed PLUS right_not_robbed  
            SET not_rob_this_node TO maximum OF left_robbed AND left_not_robbed PLUS maximum OF right_robbed AND right_not_robbed  
            RETURN rob_this_node, not_rob_this_node  
        END FUNCTION  
        RETURN maximum OF rob_from_node WITH PARAMETERS root  
    END FUNCTION  
END CLASS
