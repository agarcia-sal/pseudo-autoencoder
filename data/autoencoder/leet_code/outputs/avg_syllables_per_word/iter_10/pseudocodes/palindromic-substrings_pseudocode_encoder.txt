CLASS Solution  
    FUNCTION countSubstrings(s)  
        FUNCTION expand_around_center(left, right)  
            SET count TO zero  
            WHILE left GREATER THAN OR EQUAL TO zero AND right LESS THAN length of s AND element at position left of s EQUALS element at position right of s  
                INCREMENT count BY one  
                DECREMENT left BY one  
                INCREMENT right BY one  
            END WHILE  
            RETURN count  
        END FUNCTION  
        
        SET total_palindromes TO zero  
        FOR i FROM zero TO length of s MINUS one  
            INCREMENT total_palindromes BY expand_around_center(i, i)  
            INCREMENT total_palindromes BY expand_around_center(i, i PLUS one)  
        END FOR  
        RETURN total_palindromes  
    END FUNCTION  
END CLASS
