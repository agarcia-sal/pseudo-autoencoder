CLASS Solution  
    FUNCTION repeatedStringMatch(a, b)  
        SET min_repeats TO the NEGATION of the NEGATION of the LENGTH OF b DIVIDED BY the LENGTH OF a  
        IF b IS a SUBSTRING OF a REPEATED min_repeats TIMES  
            RETURN min_repeats  
        END IF  
        IF b IS a SUBSTRING OF a REPEATED the SUM of min_repeats AND one TIMES  
            RETURN the SUM of min_repeats AND one  
        END IF  
        RETURN the NEGATION of one  
    END FUNCTION  
END CLASS
