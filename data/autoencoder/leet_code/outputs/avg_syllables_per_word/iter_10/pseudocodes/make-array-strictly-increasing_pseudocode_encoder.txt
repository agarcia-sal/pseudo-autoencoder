CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL arr2.sort TO SORT arr2 IN ASCENDING ORDER  
        SET dp TO a DICTIONARY WITH key OF NEGATIVE ONE AND value OF ZERO  
        
        FOR each num IN arr1  
            SET new_dp TO an EMPTY DICTIONARY  
            
            FOR each last_num AND ops IN dp.items  
                IF num GREATER THAN last_num  
                    SET current_min TO value OF new_dp AT key num OR INFINITY IF NOT PRESENT  
                    SET new_dp AT key num TO the MINIMUM OF current_min AND ops  
                END IF  
                
                SET idx TO the POSITION FOR BISECT RIGHT OF last_num IN arr2  
                IF idx LESS THAN the LENGTH OF arr2  
                    SET replacement_num TO element AT position idx OF arr2  
                    SET current_min_replacement TO value OF new_dp AT key replacement_num OR INFINITY IF NOT PRESENT  
                    SET new_dp AT key replacement_num TO the MINIMUM OF current_min_replacement AND ops PLUS ONE  
                END IF  
            END FOR  
            
            IF the LENGTH OF new_dp EQUALS ZERO  
                RETURN NEGATIVE ONE  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the MINIMUM OF all VALUES IN dp  
    END FUNCTION  
END CLASS
