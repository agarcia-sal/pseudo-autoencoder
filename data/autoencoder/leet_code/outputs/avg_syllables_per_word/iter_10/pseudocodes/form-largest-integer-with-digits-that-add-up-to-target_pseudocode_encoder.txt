CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        CREATE an empty dictionary named cost_to_digit  
        FOR each index i FROM zero TO the LENGTH OF cost MINUS one  
            SET element c TO the element at position i of cost  
            SET element d TO the STRING representation of i PLUS one  
            SET the value at key c in cost_to_digit TO d  
        END FOR  
        
        CREATE a list named dp WITH the empty string as first element FOLLOWED BY target TIMES the string representing minus one  
        
        FOR each integer t FROM one TO target  
            FOR each key c and value d IN cost_to_digit  
                IF t IS GREATER THAN OR EQUAL TO c AND the element at position t MINUS c of dp IS NOT the string representing minus one  
                    SET candidate TO the string d CONCATENATED WITH the element at position t MINUS c of dp  
                    IF the element at position t of dp IS the string representing minus one OR the LENGTH OF candidate IS GREATER THAN the LENGTH OF the element at position t of dp OR the LENGTH OF candidate IS EQUAL TO the LENGTH OF the element at position t of dp AND candidate IS GREATER THAN the element at position t of dp  
                        SET the element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF the element at position target of dp IS NOT the string representing minus one  
            RETURN the element at position target of dp  
        ELSE  
            RETURN the string zero  
        END IF  
    END FUNCTION  
END CLASS
