CLASS Solution  
    FUNCTION primePalindrome(n)  
        FUNCTION is_prime(num)  
            IF num LESS THAN 2  
                RETURN False  
            END IF  
            IF num EQUALS 2 OR num EQUALS 3  
                RETURN True  
            END IF  
            IF num MODULO 2 EQUALS zero OR num MODULO 3 EQUALS zero  
                RETURN False  
            END IF  
            SET i TO five  
            WHILE i MULTIPLIED BY i LESS THAN OR EQUAL TO num  
                IF num MODULO i EQUALS zero OR num MODULO (i PLUS two) EQUALS zero  
                    RETURN False  
                END IF  
                INCREMENT i BY six  
            END WHILE  
            RETURN True  
        END FUNCTION  

        FUNCTION is_palindrome(num)  
            RETURN string representation of num EQUALS reverse of string representation of num  
        END FUNCTION  

        IF n LESS THAN OR EQUAL TO two  
            RETURN two  
        END IF  

        IF n GREATER THAN OR EQUAL TO ten AND n LESS THAN OR EQUAL TO eleven  
            RETURN eleven  
        END IF  

        WHILE True  
            IF n MODULO two EQUALS zero  
                INCREMENT n BY one  
                CONTINUE to next iteration of WHILE  
            END IF  
            IF is_palindrome(n) AND is_prime(n)  
                RETURN n  
            END IF  
            IF length of string representation of n MODULO two EQUALS zero  
                SET n TO ten RAISED TO THE POWER OF length of string representation of n PLUS one  
            ELSE  
                INCREMENT n BY two  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
