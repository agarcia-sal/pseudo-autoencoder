CLASS Solution  
    FUNCTION longestPalindrome(word1, word2)  
        SET s TO concatenation of word1 and word2  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of size n BY n FILLED WITH zeros  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AND i of dp TO one  
        END FOR  
        
        SET max_len TO zero  
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of s EQUALS element at position j of s  
                    SET element at position i AND j of dp TO element at position i PLUS one AND j MINUS one of dp PLUS two  
                    IF i LESS THAN LENGTH OF word1 AND j GREATER THAN OR EQUAL TO LENGTH OF word1  
                        SET max_len TO the GREATER VALUE BETWEEN max_len AND element at position i AND j of dp  
                    END IF  
                ELSE  
                    SET element at position i AND j of dp TO the GREATER VALUE BETWEEN element at position i PLUS one AND j of dp AND element at position i AND j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_len  
    END FUNCTION  
END CLASS
