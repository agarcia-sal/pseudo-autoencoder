CLASS Solution  
    FUNCTION waysToDistribute(n, k)  
        SET MOD TO one billion PLUS seven  
        INITIALIZE dp AS a two-dimensional list with (n PLUS one) rows and (k PLUS one) columns with all elements SET TO zero  
        SET element at position zero of row zero in dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET element at position j of row i in dp TO the REMAINDER AFTER DIVIDING the SUM of j MULTIPLIED BY element at position j of row (i MINUS one) in dp AND element at position (j MINUS one) of row (i MINUS one) in dp BY MOD  
            END FOR  
        END FOR  
        RETURN element at position k of row n in dp  
    END FUNCTION  
END CLASS
