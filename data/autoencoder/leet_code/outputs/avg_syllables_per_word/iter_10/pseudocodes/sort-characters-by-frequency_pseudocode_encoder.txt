CLASS Solution  
    FUNCTION frequencySort(s)  
        IMPORT Counter FROM collections  
        SET frequency TO Counter OF s  
        SET sorted_characters TO the list OF keys OF frequency SORTED BY the frequency AT character IN descending order  
        SET result TO empty string  
        FOR each character IN sorted_characters  
            APPEND character REPEATED frequency AT character TIMES TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
