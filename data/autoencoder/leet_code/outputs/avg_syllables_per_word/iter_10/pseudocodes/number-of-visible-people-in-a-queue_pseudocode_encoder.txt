CLASS Solution  
    FUNCTION canSeePersonsCount(heights)  
        SET n TO the LENGTH OF heights  
        SET answer TO a list of zeros with LENGTH n  
        SET stack TO an empty list  
        
        FOR i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack is GREATER THAN zero AND the element at position last element of stack of heights is LESS THAN the element at position i of heights  
                SET top_index TO the last element of stack  
                REMOVE last element FROM stack  
                INCREMENT the element at position top_index of answer BY one  
            END WHILE  
            IF the LENGTH OF stack is GREATER THAN zero  
                INCREMENT the element at position the last element of stack of answer BY one  
            END IF  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
