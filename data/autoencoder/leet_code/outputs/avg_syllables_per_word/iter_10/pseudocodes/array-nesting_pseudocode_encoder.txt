CLASS Solution  
    FUNCTION arrayNesting(nums)  
        SET visited TO list of False repeated the LENGTH OF nums times  
        SET max_length TO zero  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF element at position i of visited EQUALS False  
                SET current_length TO zero  
                SET k TO i  
                WHILE element at position k of visited EQUALS False  
                    SET element at position k of visited TO True  
                    SET k TO element at position k of nums  
                    INCREMENT current_length BY one  
                END WHILE  
                SET max_length TO the GREATER THAN OF max_length AND current_length  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
