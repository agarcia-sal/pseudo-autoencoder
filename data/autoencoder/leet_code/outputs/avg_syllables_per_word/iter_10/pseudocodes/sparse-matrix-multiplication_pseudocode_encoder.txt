CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF the element at position zero of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF the element at position zero of mat2  
        SET result TO the creation of a list containing m elements EACH of which is a list containing n zeroes  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR p FROM zero TO k MINUS one  
                    SET the element at position i then j of result TO the element at position i then j of result PLUS the element at position i then p of mat1 MULTIPLIED BY the element at position p then j of mat2  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
