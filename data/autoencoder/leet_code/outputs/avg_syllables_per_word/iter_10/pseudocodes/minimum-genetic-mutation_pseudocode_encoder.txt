CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO a set containing all elements of bank  
        SET q TO a deque containing the tuple of start and zero  
        SET mp TO a map with key 'A' mapping to string 'TCG', key 'T' mapping to string 'ACG', key 'C' mapping to string 'ATG', key 'G' mapping to string 'ATC'  
        WHILE the length of q is greater than zero  
            SET t, step TO the leftmost element removed FROM q  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR i FROM zero TO the length of t MINUS one  
                SET v TO the element at position i of t  
                FOR each j IN the string mapped by v in mp  
                    SET next TO the concatenation of substring of t from start to position i MINUS one, followed by j, followed by substring of t from position i PLUS one to end  
                    IF next IS IN s  
                        APPEND the tuple of next and step PLUS one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
