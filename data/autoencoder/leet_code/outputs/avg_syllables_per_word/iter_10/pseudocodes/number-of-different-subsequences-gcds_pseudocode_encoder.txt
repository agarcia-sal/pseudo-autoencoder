CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET max_num TO the maximum element in nums  
        SET num_set TO the set formed from elements of nums  
        SET count TO zero  
        
        FOR g FROM one TO max_num  
            SET gcd_val TO zero  
            FOR m FROM g TO max_num IN STEPS OF g  
                IF m IS an element of num_set  
                    SET gcd_val TO the greatest common divisor of gcd_val AND m  
                    IF gcd_val EQUALS g  
                        INCREMENT count BY one  
                        BREAK the FOR loop over m  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
