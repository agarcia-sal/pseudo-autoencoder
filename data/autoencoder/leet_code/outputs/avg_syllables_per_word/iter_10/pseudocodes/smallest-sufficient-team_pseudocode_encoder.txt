CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        SET d TO an empty mapping from skill to index  
        FOR each index i FROM zero TO the LENGTH OF req_skills MINUS one  
            SET skill s TO element at position i of req_skills  
            SET element at position s of d TO i  
        END FOR  
        SET m TO the LENGTH OF req_skills  
        SET n TO the LENGTH OF people  
        SET p TO a list containing n zeros  
        FOR each index i FROM zero TO n MINUS one  
            SET ss TO element at position i of people  
            FOR each skill s IN ss  
                SET element at position i of p TO element at position i of p BITWISE OR WITH one LEFT SHIFTED BY element at position s of d  
            END FOR  
        END FOR  
        SET f TO a list containing two TO THE POWER OF m copies of infinity  
        SET g TO a list containing two TO THE POWER OF m zeros  
        SET h TO a list containing two TO THE POWER OF m zeros  
        SET element at position zero of f TO zero  
        FOR each integer i FROM zero TO two TO THE POWER OF m MINUS one  
            IF element at position i of f EQUALS infinity THEN  
                CONTINUE to the next iteration of the loop  
            END IF  
            FOR each integer j FROM zero TO n MINUS one  
                SET combined TO bitwise OR of i and element at position j of p  
                IF element at position i of f PLUS one LESS THAN element at position combined of f THEN  
                    SET element at position combined of f TO element at position i of f PLUS one  
                    SET element at position combined of g TO j  
                    SET element at position combined of h TO i  
                END IF  
            END FOR  
        END FOR  
        SET i TO two TO THE POWER OF m MINUS one  
        SET ans TO an empty list  
        WHILE i NOT EQUALS zero  
            APPEND element at position i of g TO ans  
            SET i TO element at position i of h  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
