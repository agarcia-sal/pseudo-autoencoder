CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF the LENGTH OF edges NOT EQUALS n MINUS one  
            RETURN False  
        END IF  
        
        FUNCTION initializeAdjacencyList(n)  
            SET adjacency_list TO empty dictionary  
            FOR index FROM zero TO n MINUS one  
                SET element at position index of adjacency_list TO empty list  
            END FOR  
            RETURN adjacency_list  
        END FUNCTION  
        
        SET adj_list TO initializeAdjacencyList(n)  
        FOR each pair a, b IN edges  
            APPEND b TO element at position a of adj_list  
            APPEND a TO element at position b of adj_list  
        END FOR  
        
        SET visited TO empty set  
        
        FUNCTION dfs(node)  
            IF node IN visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN element at position node of adj_list  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        
        CALL dfs WITH zero  
        RETURN the LENGTH OF visited EQUALS n  
    END FUNCTION  
END CLASS
