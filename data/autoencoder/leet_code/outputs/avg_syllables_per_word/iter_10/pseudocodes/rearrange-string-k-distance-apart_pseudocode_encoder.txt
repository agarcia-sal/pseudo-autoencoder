CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO a frequency count of each character in s  
        
        SET max_heap TO an empty list  
        FOR each character char AND its count count IN freq items  
            APPEND a pair of negative count AND char TO max_heap  
        END FOR  
        TRANSFORM max_heap INTO a heap structure  
        
        SET wait_queue TO an empty list  
        
        SET result TO an empty list  
        
        WHILE max_heap is not empty  
            REMOVE and obtain the pair count AND char with highest priority FROM max_heap  
            APPEND char TO result  
            
            APPEND a pair of count PLUS one AND char TO wait_queue  
            
            IF the LENGTH OF wait_queue EQUALS k  
                REMOVE the first element FROM wait_queue AND obtain count AND char  
                IF count LESS THAN zero  
                    ADD the pair count AND char BACK INTO max_heap  
                END IF  
            END IF  
        END WHILE  
        
        IF the LENGTH OF result NOT EQUALS the LENGTH OF s  
            RETURN an empty string  
        END IF  
        
        RETURN the concatenation of all characters in result  
    END FUNCTION  
END CLASS
