CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with val AS element at position zero of values  
    SET p TO head  
    FOR each val IN elements from position one to the end of values  
        SET node TO new ListNode with val AS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN (element val of p1 EQUALS element val of p2) AND is_same_list(element next of p1, element next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList(self, head)  
        IF head EQUALS None OR element next of head EQUALS None  
            RETURN  
        END IF  

        SET slow TO head  
        SET fast TO head  
        WHILE fast NOT EQUALS None AND element next of fast NOT EQUALS None  
            SET slow TO element next of slow  
            SET fast TO element next of element next of fast  
        END WHILE  

        SET prev TO None  
        SET curr TO slow  
        WHILE curr NOT EQUALS None  
            SET next_temp TO element next of curr  
            SET element next of curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  

        SET first TO head  
        SET second TO prev  
        WHILE element next of second NOT EQUALS None  
            SET temp1 TO element next of first  
            SET temp2 TO element next of second  
            SET element next of first TO second  
            SET element next of second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
