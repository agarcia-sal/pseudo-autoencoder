CLASS Solution
    FUNCTION topKFrequent(words, k)
        SET count TO the result of counting the frequency of each element in words
        SET heap TO new empty list
        FOR each element word AND element freq IN count items
            APPEND a tuple consisting of negative freq AND word TO heap
        END FOR
        APPLY heapify operation TO heap
        SET result TO new empty list
        FOR index FROM zero TO k MINUS one
            SET top_element TO the result of popping the smallest element from heap
            APPEND element at position one of top_element TO result
        END FOR
        RETURN result
    END FUNCTION
END CLASS
