CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO the list of characters of num  
        SET n TO the LENGTH OF num  
        
        WHILE k GREATER THAN zero  
            SET swapped TO False  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF element at position i of num GREATER THAN element at position i PLUS one of num  
                    SET min_digit TO character nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO the MINIMUM OF i PLUS k PLUS one AND n MINUS one  
                        IF element at position j of num LESS THAN min_digit  
                            SET min_digit TO element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET num from position i TO min_index INCLUSIVE TO a list with the element at position min_index of num FOLLOWED BY the elements from position i TO min_index MINUS one of num  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO True  
                    BREAK  
                END IF  
            END FOR  
            
            IF swapped EQUALS False  
                BREAK  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements in num  
    END FUNCTION  
END CLASS
