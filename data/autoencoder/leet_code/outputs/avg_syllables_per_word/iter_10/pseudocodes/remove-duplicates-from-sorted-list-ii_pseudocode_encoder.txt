CLASS ListNode  
    FUNCTION __init__ with parameters val, next  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with val equal to element at position zero of values  
    SET p TO head  
    FOR each val IN values starting from position one  
        SET node TO new ListNode with val equal to val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1, p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN p1.val EQUALS p2.val AND is_same_list with parameters p1.next, p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates with parameters self, head  
        SET dummy TO new ListNode with val equal to zero, next equal to head  
        SET prev TO dummy  
        WHILE head IS NOT None  
            IF head.next IS NOT None AND head.val EQUALS head.next.val  
                WHILE head.next IS NOT None AND head.val EQUALS head.next.val  
                    SET head TO head.next  
                END WHILE  
                SET prev.next TO head.next  
            ELSE  
                SET prev TO prev.next  
            END IF  
            SET head TO head.next  
        END WHILE  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
