CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero, left EQUALS none, right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH value at position zero of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN the LENGTH OF values AND value at position i of values IS NOT none  
            SET node.left TO NEW TreeNode WITH value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND value at position i of values IS NOT none  
            SET node.right TO NEW TreeNode WITH value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF value at p NOT EQUALS value at q  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left, q.left AND is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover WITH PARAMETER root  
        FUNCTION dfs WITH PARAMETER node  
            IF node EQUALS none  
                RETURN zero, zero, infinite  
            END IF  
            SET left_no_camera, left_monitored, left_camera TO dfs WITH PARAMETER node.left  
            SET right_no_camera, right_monitored, right_camera TO dfs WITH PARAMETER node.right  
            SET no_camera TO left_monitored PLUS right_monitored  
            SET monitored TO THE MINIMUM OF (left_camera PLUS THE MINIMUM OF right_monitored AND right_camera), (right_camera PLUS THE MINIMUM OF left_monitored AND left_camera), (left_camera PLUS right_camera)  
            SET camera TO one PLUS THE MINIMUM OF left_no_camera, left_monitored, left_camera PLUS THE MINIMUM OF right_no_camera, right_monitored, right_camera  
            RETURN no_camera, monitored, camera  
        END FUNCTION  
        SET root_no_camera, root_monitored, root_camera TO dfs WITH PARAMETER root  
        RETURN THE MINIMUM OF root_monitored AND root_camera  
    END FUNCTION  
END CLASS
