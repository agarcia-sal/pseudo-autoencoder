CLASS Solution  
    FUNCTION maxSlidingWindow(nums, k)  
        IF nums EQUALS empty OR k EQUALS zero  
            RETURN empty list  
        END IF  
        SET deq TO empty deque  
        SET result TO empty list  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF the LENGTH OF deq IS GREATER THAN zero AND the element at position zero of deq LESS THAN i MINUS k PLUS one  
                REMOVE first element FROM deq  
            END IF  
            WHILE the LENGTH OF deq IS GREATER THAN zero AND the element at position element at position last of deq of nums LESS THAN element at position i of nums  
                REMOVE last element FROM deq  
            END WHILE  
            APPEND i TO deq  
            IF i GREATER THAN OR EQUAL TO k MINUS one  
                APPEND element at position element at position zero of deq of nums TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
