CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val as element at position zero of values  
    SET i TO one  
    SET queue TO empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS not empty  
        SET node TO first element FROM queue and REMOVE it  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode with val as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode with val as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves(self, root)  
        FUNCTION collectLeaves(node)  
            IF node IS None  
                RETURN negative one  
            END IF  
            SET left_height TO collectLeaves(node.left)  
            SET right_height TO collectLeaves(node.right)  
            SET current_height TO greater of left_height AND right_height PLUS one  
            IF current_height GREATER THAN OR EQUAL TO LENGTH OF result  
                APPEND empty list TO result  
            END IF  
            APPEND node.val TO element at position current_height of result  
            RETURN current_height  
        END FUNCTION  
        SET result TO empty list  
        CALL collectLeaves WITH root  
        RETURN result  
    END FUNCTION  
END CLASS
