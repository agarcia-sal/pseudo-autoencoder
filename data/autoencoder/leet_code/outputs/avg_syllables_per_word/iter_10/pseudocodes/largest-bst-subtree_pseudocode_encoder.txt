CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val set to element at position zero of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN length of values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode with val set to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode with val set to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p.val NOT EQUALS value of q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree(self, root)  
        FUNCTION dfs(node)  
            IF node IS none  
                RETURN infinity, negative infinity, zero  
            END IF  
            SET left_min, left_max, left_size TO dfs(left child of node)  
            SET right_min, right_max, right_size TO dfs(right child of node)  
            IF left_max LESS THAN value of node.val AND value of node.val LESS THAN right_min  
                RETURN minimum of value of node.val AND left_min, maximum of value of node.val AND right_max, left_size PLUS right_size PLUS one  
            END IF  
            RETURN negative infinity, infinity, maximum of left_size AND right_size  
        END FUNCTION  
        RETURN third element of dfs(root)  
    END FUNCTION  
END CLASS
