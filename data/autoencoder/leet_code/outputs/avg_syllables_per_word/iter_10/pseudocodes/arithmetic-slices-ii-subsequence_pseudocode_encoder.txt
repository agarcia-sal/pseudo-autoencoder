CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN 3  
            RETURN zero  
        END IF  
        
        SET dp TO a new list  
        FOR index FROM zero TO n MINUS one  
            APPEND a new default dictionary of integer TO dp  
        END FOR  
        
        SET total_count TO zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                SET d TO element at position i of nums MINUS element at position j of nums  
                IF d EXISTS AS key IN element at position j of dp  
                    INCREMENT element at position i of dp at key d BY element at position j of dp at key d PLUS one  
                    INCREMENT total_count BY element at position j of dp at key d  
                ELSE  
                    INCREMENT element at position i of dp at key d BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
