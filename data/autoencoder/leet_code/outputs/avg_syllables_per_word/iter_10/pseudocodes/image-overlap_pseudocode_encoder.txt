CLASS Solution  
    FUNCTION largestOverlap(img1, img2)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of img1 EQUALS one  
                    APPEND the pair of i and j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of img2 EQUALS one  
                    APPEND the pair of i and j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a mapping from pairs to integers with default zero  
        FOR each pair x1, y1 IN points1  
            FOR each pair x2, y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value at key pair dx, dy IN overlap_count BY one  
            END FOR  
        END FOR  
        RETURN the maximum value among the values of overlap_count with default zero  
    END FUNCTION  
END CLASS
