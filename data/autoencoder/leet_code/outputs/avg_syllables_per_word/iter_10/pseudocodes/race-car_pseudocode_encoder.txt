CLASS Solution  
    FUNCTION racecar(target)  
        SET queue TO a new deque containing a tuple of zero, one, zero  
        SET visited TO a new set containing a tuple of zero, one  
        
        WHILE the length of queue is greater than zero  
            SET position, speed, steps TO the first element removed from the left of queue  
            
            IF position EQUALS target  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple of next_position, next_speed is NOT IN visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of next_position, next_speed TO visited  
                APPEND the tuple of next_position, next_speed, steps PLUS one TO queue  
            END IF  
            
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple of position, next_speed is NOT IN visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of position, next_speed TO visited  
                APPEND the tuple of position, next_speed, steps PLUS one TO queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
