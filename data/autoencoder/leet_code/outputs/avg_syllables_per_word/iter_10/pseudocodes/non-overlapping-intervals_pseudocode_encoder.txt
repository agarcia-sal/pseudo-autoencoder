CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals EQUALS empty  
            RETURN zero  
        END IF  
        SORT intervals BY the second element of each interval in ascending order  
        SET end TO the second element of the first interval in intervals  
        SET count TO zero  
        FOR index FROM one TO the LENGTH OF intervals MINUS one  
            IF the first element of the interval at position index IN intervals GREATER THAN OR EQUAL TO end  
                SET end TO the second element of the interval at position index IN intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
