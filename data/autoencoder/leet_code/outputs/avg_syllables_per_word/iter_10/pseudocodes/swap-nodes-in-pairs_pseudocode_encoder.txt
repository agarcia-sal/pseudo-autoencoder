CLASS ListNode  
    FUNCTION __init__(val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with val equal to element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values from position one to the end  
        SET node TO new ListNode with val equal to val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the result of (element val of p1 EQUALS element val of p2) AND call is_same_list with parameters element next of p1 AND element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs(head)  
        SET dummy TO new ListNode with val equal to zero  
        SET dummy.next TO head  
        SET current TO dummy  
        
        WHILE element next of current NOT EQUALS none AND element next of element next of current NOT EQUALS none  
            SET first TO element next of current  
            SET second TO element next of element next of current  
            
            SET element next of current TO second  
            SET element next of first TO element next of second  
            SET element next of second TO first  
            
            SET current TO element next of element next of current  
        END WHILE  
        
        RETURN element next of dummy  
    END FUNCTION  
END CLASS
