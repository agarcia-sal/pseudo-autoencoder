CLASS Solution  
    FUNCTION minSkips(dist, speed, hoursBefore)  
        SET n TO the LENGTH OF dist  
        
        DEFINE dp AS a list of lists WHERE each inner list CONTAINS n PLUS one elements SET TO infinity  
        FOR index FROM zero TO n MINUS one  
            SET the element at position index of dp TO a new list CONTAINING n PLUS one elements SET TO infinity  
        END FOR  
        
        SET total_dist TO the SUM OF all elements IN dist  
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore  
            RETURN negative one  
        END IF  
        
        SET the element at position zero position zero of dp TO the element at position zero of dist  
        
        FOR j FROM one TO n MINUS one  
            SET the element at position zero position j of dp TO the element at position zero of dist  
        END FOR  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j LESS THAN i  
                    SET temp TO the CEILING OF the element at position i MINUS one position j of dp DIVIDED BY speed  
                    MULTIPLY temp BY speed  
                    SET the element at position i position j of dp TO temp PLUS the element at position i of dist  
                END IF  
                
                IF j GREATER THAN zero  
                    SET the element at position i position j of dp TO the MINIMUM OF the current value at dp position i position j AND the element at position i MINUS one position j MINUS one of dp PLUS the element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            IF the element at position n MINUS one position j of dp LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
