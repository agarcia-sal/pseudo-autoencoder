CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH first element of values  
    SET i TO one  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element REMOVED from queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF val of p NOT EQUALS val of q  
        RETURN False  
    ELSE  
        RETURN result of is_same_tree called with p.left and q.left AND result of is_same_tree called with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION countUnivalSubtrees with parameter root  
        FUNCTION helper with parameter node  
            IF node EQUALS None  
                RETURN True, zero  
            END IF  
            SET is_left_unival, left_count TO result of helper called with node.left  
            SET is_right_unival, right_count TO result of helper called with node.right  
            IF is_left_unival AND is_right_unival  
                IF (node.left EQUALS None OR val of node.left EQUALS val of node) AND (node.right EQUALS None OR val of node.right EQUALS val of node)  
                    RETURN True, left_count PLUS right_count PLUS one  
                END IF  
            END IF  
            RETURN False, left_count PLUS right_count  
        END FUNCTION  
        SET _, count TO result of helper called with root  
        RETURN count  
    END FUNCTION  
END CLASS
