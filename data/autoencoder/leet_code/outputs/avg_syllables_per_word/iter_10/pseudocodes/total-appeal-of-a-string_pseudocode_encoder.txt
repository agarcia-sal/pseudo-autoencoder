CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO a list containing twenty-six elements each set TO minus one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        
        FOR each position i AND character char IN s  
            SET index TO the numerical code of char MINUS the numerical code of character a  
            INCREMENT current_appeal BY position i MINUS the element at position index of last  
            INCREMENT total_appeal BY current_appeal  
            SET the element at position index of last TO position i  
        END FOR  
        
        RETURN total_appeal  
    END FUNCTION  
END CLASS
