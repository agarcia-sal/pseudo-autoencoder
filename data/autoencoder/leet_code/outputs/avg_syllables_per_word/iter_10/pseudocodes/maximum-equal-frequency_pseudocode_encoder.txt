CLASS Solution  
    FUNCTION maxEqualFreq(nums)  
        SET count TO an empty Counter collection  
        SET freq TO an empty Counter collection  
        SET max_len TO zero  
        
        FOR each index i AND number num IN enumeration of nums  
            INCREMENT the element num IN count BY one  
            
            IF the element num IN count IS GREATER THAN one  
                DECREMENT the element at position count of num MINUS one IN freq BY one  
                IF the element at position count of num MINUS one IN freq EQUALS zero  
                    REMOVE the element at position count of num MINUS one FROM freq  
                END IF  
            END IF  
            
            INCREMENT the element at position count of num IN freq BY one  
            
            IF the LENGTH OF freq EQUALS one AND (the element one IN freq EXISTS OR the first value of freq EQUALS one)  
                SET max_len TO i PLUS one  
            ELSE IF the LENGTH OF freq EQUALS two AND ((the element one IN freq EXISTS AND the element one OF freq EQUALS one) OR (the MAXIMUM key of freq MINUS the MINIMUM key of freq EQUALS one AND the element at MAXIMUM key of freq EQUALS one))  
                SET max_len TO i PLUS one  
            END IF  
        END FOR  
        
        RETURN max_len  
    END FUNCTION  
END CLASS
