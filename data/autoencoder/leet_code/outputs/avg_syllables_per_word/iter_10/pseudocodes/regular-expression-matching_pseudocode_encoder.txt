CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO a two-dimensional list of False with dimensions the LENGTH OF s plus one BY the LENGTH OF p plus one  
        
        SET the element at position zero zero of dp TO True  
        
        FOR j FROM one TO the LENGTH OF p  
            IF the character at position j minus one of p EQUALS the character star  
                SET the element at position zero j of dp TO the element at position zero j minus two of dp  
            END IF  
        END FOR  
        
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM one TO the LENGTH OF p  
                IF the character at position j minus one of p EQUALS the character at position i minus one of s OR the character at position j minus one of p EQUALS the character dot  
                    SET the element at position i j of dp TO the element at position i minus one j minus one of dp  
                ELSE IF the character at position j minus one of p EQUALS the character star  
                    SET the element at position i j of dp TO the element at position i j minus two of dp  
                    IF the character at position j minus two of p EQUALS the character at position i minus one of s OR the character at position j minus two of p EQUALS the character dot  
                        SET the element at position i j of dp TO the element at position i j of dp OR the element at position i minus one j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position the LENGTH OF s the LENGTH OF p of dp  
    END FUNCTION  
END CLASS
