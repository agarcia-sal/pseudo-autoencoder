CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i GREATER THAN j  
                RETURN zero  
            END IF  
            IF the key formed by i and j EXISTS IN memo  
                RETURN the value at memo for key i and j  
            END IF  
            SET result TO the value returned by dp with parameters i, j MINUS one, and memo PLUS one  
            FOR each k FROM i TO j MINUS one  
                IF the character at position k of s EQUALS the character at position j of s  
                    SET possible_result TO the value returned by dp with parameters i, k MINUS one, and memo PLUS the value returned by dp with parameters k, j MINUS one, and memo  
                    IF possible_result LESS THAN result  
                        SET result TO possible_result  
                    END IF  
                END IF  
            END FOR  
            SET memo keyed by i and j TO result  
            RETURN result  
        END FUNCTION  
        RETURN the value returned by dp with parameters zero, the length of s MINUS one, and an empty dictionary  
    END FUNCTION  
END CLASS
