CLASS Solution  
    FUNCTION leastInterval(tasks, n)  
        SET task_counts TO the frequency count of each element IN tasks  
        SET max_heap TO empty list  
        FOR each count IN the values of task_counts  
            APPEND negative count TO max_heap  
        END FOR  
        TRANSFORM max_heap INTO a heap structure  
          
        SET time TO zero  
        SET queue TO empty list of pairs consisting of time plus n and task count  
          
        WHILE max_heap IS NOT empty OR queue IS NOT empty  
            WHILE queue IS NOT empty AND the first element at position zero of queue at position zero EQUALS time  
                REMOVE first element FROM queue AND SET the removed pair TO time_value AND count  
                IF count IS less than zero  
                    PUSH count INTO max_heap  
                END IF  
            END WHILE  
              
            IF max_heap IS NOT empty  
                REMOVE the smallest element FROM max_heap AND SET it TO count  
                INCREMENT count BY one  
                IF count IS less than zero  
                    APPEND a pair consisting of time PLUS n PLUS one AND count TO queue  
                END IF  
            END IF  
              
            INCREMENT time BY one  
        END WHILE  
          
        RETURN time  
    END FUNCTION  
END CLASS
