CLASS Solution  
    FUNCTION minPushBox(grid)  
        SET directions TO the list of tuples of MINUS one and zero, one and zero, zero and MINUS one, zero and one  
        
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS character S  
                    SET player_start TO the tuple of i and j  
                ELSE IF element at position i of grid at position j EQUALS character B  
                    SET box_start TO the tuple of i and j  
                ELSE IF element at position i of grid at position j EQUALS character T  
                    SET target TO the tuple of i and j  
                END IF  
            END FOR  
        END FOR  
        
        SET queue TO a new double-ended queue with the single element tuple of player_start, box_start, and zero  
        SET visited TO an empty set  
        ADD the tuple of player_start and box_start TO visited  
        
        WHILE queue is not empty  
            REMOVE the first element FROM queue and SET to player_pos, box_pos, push_count  
            
            IF box_pos EQUALS target  
                RETURN push_count  
            END IF  
            
            FOR each tuple dx, dy IN directions  
                SET new_player_pos TO the tuple of the first element of player_pos PLUS dx and the second element of player_pos PLUS dy  
                
                IF CALLING is_valid_position WITH new_player_pos and grid RETURNS true  
                    IF new_player_pos EQUALS box_pos  
                        SET new_box_pos TO the tuple of the first element of box_pos PLUS dx and the second element of box_pos PLUS dy  
                        
                        IF CALLING is_valid_position WITH new_box_pos and grid RETURNS true AND the tuple of new_player_pos and new_box_pos NOT IN visited  
                            ADD the tuple of new_player_pos and new_box_pos TO visited  
                            APPEND the tuple of new_player_pos, new_box_pos, and push_count PLUS one TO queue  
                        END IF  
                    ELSE  
                        IF the tuple of new_player_pos and box_pos NOT IN visited  
                            ADD the tuple of new_player_pos and box_pos TO visited  
                            APPEND the tuple of new_player_pos, box_pos, and push_count TO queue  
                        END IF  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN MINUS one  
    END FUNCTION  
  
    FUNCTION is_valid_position(position, grid)  
        SET x TO the first element of position  
        SET y TO the second element of position  
        RETURN x GREATER THAN OR EQUAL TO zero AND x LESS THAN the LENGTH OF grid AND y GREATER THAN OR EQUAL TO zero AND y LESS THAN the LENGTH OF element at position zero of grid AND element at position x of grid at position y NOT EQUALS character hashtag  
    END FUNCTION  
END CLASS
