CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s EQUALS empty string  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET character_difference TO ordinal value of c MINUS ordinal value of first character of s  
                SET adjusted_difference TO character_difference MODULO twenty six  
                APPEND string representation of adjusted_difference TO canonical_form  
            END FOR  
            SET joined_form TO concatenation of elements in canonical_form separated by comma  
            RETURN joined_form  
        END FUNCTION  
  
        SET groups TO new empty dictionary with default list values  
  
        FOR each string s IN strings  
            SET canonical_form TO get_canonical_form(s)  
            APPEND s TO groups at key canonical_form  
        END FOR  
  
        RETURN list of values of groups  
    END FUNCTION  
END CLASS
