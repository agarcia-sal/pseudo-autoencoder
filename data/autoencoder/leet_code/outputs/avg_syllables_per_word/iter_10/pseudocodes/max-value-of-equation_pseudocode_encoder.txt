CLASS Solution  
    FUNCTION findMaxValueOfEquation(points, k)  
        SET max_value TO negative infinity  
        SET deque_ TO an empty double ended queue  
        
        FOR each pair of xj and yj IN points  
            WHILE deque_ is not empty AND xj MINUS the second element at position zero of deque_ GREATER THAN k  
                REMOVE first element FROM deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO the greater value BETWEEN max_value AND yj PLUS xj PLUS the first element at position zero of deque_  
            END IF  
            
            WHILE deque_ is not empty AND yj MINUS xj GREATER THAN OR EQUAL TO the first element at the last position of deque_  
                REMOVE last element FROM deque_  
            END WHILE  
            
            APPEND the pair of yj MINUS xj AND xj TO deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
