CLASS Solution  
    FUNCTION merge(intervals)  
        SORT intervals IN ASCENDING ORDER BASED ON element at position zero of each interval  
        SET merged TO empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR element at position one of last element of merged IS LESS THAN element at position zero of interval  
                APPEND interval TO merged  
            ELSE  
                SET element at position one of last element of merged TO the GREATER OF element at position one of last element of merged AND element at position one of interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
