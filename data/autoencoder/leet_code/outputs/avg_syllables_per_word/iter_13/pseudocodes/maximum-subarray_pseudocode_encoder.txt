CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO element at position zero of nums  
        SET current_sum TO element at position zero of nums  
        FOR each num IN the elements of nums starting from position one to the end  
            SET current_sum TO THE MAXIMUM OF num AND current_sum PLUS num  
            SET max_sum TO THE MAXIMUM OF max_sum AND current_sum  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
