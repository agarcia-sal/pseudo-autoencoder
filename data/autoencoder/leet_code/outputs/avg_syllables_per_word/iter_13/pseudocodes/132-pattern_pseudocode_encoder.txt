CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        FOR each num IN the elements of nums in reverse order  
            IF num IS LESS THAN second  
                RETURN True  
            END IF  
            WHILE stack IS NOT empty AND element at position the last index of stack IS LESS THAN num  
                SET second TO element removed from the top of stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
