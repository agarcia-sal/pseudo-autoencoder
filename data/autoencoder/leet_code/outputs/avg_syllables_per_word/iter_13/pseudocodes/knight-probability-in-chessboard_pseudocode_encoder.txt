CLASS Solution  
    FUNCTION knightProbability(n, k, row, column)  
        SET moves TO the list containing pairs of minus two and minus one plus minus one and plus two plus one and minus two plus one and plus two plus two and minus one plus two and plus one  
        SET dp TO a two dimensional list with n rows and n columns each element initialized TO zero  
        SET element at position row and column of dp TO one  
        FOR each iteration from zero TO k minus one  
            SET new_dp TO a two dimensional list with n rows and n columns each element initialized TO zero  
            FOR r FROM zero TO n minus one  
                FOR c FROM zero TO n minus one  
                    IF element at position r and c of dp EQUALS zero THEN  
                        CONTINUE to next iteration of c loop  
                    END IF  
                    FOR each pair dr and dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n THEN  
                            SET element at position nr and nc of new_dp TO element at position nr and nc of new_dp PLUS element at position r and c of dp DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        SET total_probability TO zero  
        FOR each row IN dp  
            FOR each probability IN row  
                INCREMENT total_probability BY probability  
            END FOR  
        END FOR  
        RETURN total_probability  
    END FUNCTION  
END CLASS
