CLASS Solution  
    FUNCTION minInteger(num AS string, k AS integer) RETURNS string  
        SET num TO list containing each character of num  
        SET n TO the LENGTH OF num  
        
        WHILE k GREATER THAN zero HOLDS  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF element at position i of num GREATER THAN element at position i PLUS one of num THEN  
                    SET min_digit TO character nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO the LESSER OF i PLUS k PLUS one AND n  
                        IF element at position j of num LESS THAN min_digit THEN  
                            SET min_digit TO element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET sublist TO list containing element at position min_index of num FOLLOWED BY elements from position i TO min_index MINUS one of num  
                    ASSIGN elements from position i TO min_index of num TO sublist  
                    SET k TO k MINUS (min_index MINUS i)  
                    SET swapped TO true  
                    BREAK  
                END IF  
            END FOR  
            
            IF swapped IS false THEN  
                BREAK  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements of num into a string  
    END FUNCTION  
END CLASS
