CLASS Solution  
    FUNCTION maxJumps(arr AS List OF integers, d AS integer) RETURNS integer  
        FUNCTION dp(i AS integer) RETURNS integer  
            SET max_jumps TO one  
            FOR each x FROM one TO d  
                IF sum of i and x IS LESS THAN the LENGTH OF arr THEN  
                    IF element at position sum of i and x of arr IS GREATER THAN OR EQUAL TO element at position i of arr THEN  
                        BREAK  
                    END IF  
                    SET max_jumps TO the GREATER OF max_jumps AND one PLUS dp WITH argument sum of i and x  
                END IF  
            END FOR  
            FOR each x FROM one TO d  
                IF difference of i and x IS GREATER THAN OR EQUAL TO zero THEN  
                    IF element at position difference of i and x of arr IS GREATER THAN OR EQUAL TO element at position i of arr THEN  
                        BREAK  
                    END IF  
                    SET max_jumps TO the GREATER OF max_jumps AND one PLUS dp WITH argument difference of i and x  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        RETURN the GREATER OF dp WITH each i IN range from zero TO one LESS THAN the LENGTH OF arr  
    END FUNCTION  
END CLASS
