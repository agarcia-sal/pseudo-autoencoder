CLASS Solution  
    FUNCTION numberOfArrays(s AS STRING, k AS INTEGER) RETURNS INTEGER  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        INITIALIZE dp AS list OF zero WITH size n PLUS one  
        SET element at position zero of dp TO one  
        FOR each integer i FROM one TO n  
            FOR each integer j FROM zero TO i MINUS one  
                IF element at position j of s DOES NOT EQUAL character zero AND  
                   the INTEGER VALUE OF substring from position j TO i MINUS one of s IS GREATER THAN OR EQUAL TO one AND LESS THAN OR EQUAL TO k  
                    SET element at position i of dp TO element at position i of dp PLUS element at position j of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
