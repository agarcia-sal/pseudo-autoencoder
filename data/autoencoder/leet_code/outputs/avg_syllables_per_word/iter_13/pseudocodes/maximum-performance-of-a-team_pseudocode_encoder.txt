CLASS Solution  
    FUNCTION maxPerformance(n, speed, efficiency, k)  
        SET engineers TO the LIST OF PAIRS WHERE each PAIR IS an element from efficiency AND the corresponding element from speed  
        SORT engineers IN DESCENDING ORDER BASED ON the FIRST element OF each PAIR which is efficiency  
        SET speed_heap TO an empty LIST  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each PAIR IN engineers  
            ASSIGN eff TO the FIRST element OF PAIR  
            ASSIGN spd TO the SECOND element OF PAIR  
            ADD spd TO speed_heap USING a METHOD THAT MAINTAINS A MINIMUM HEAP PROPERTY  
            INCREMENT speed_sum BY spd  
            IF the LENGTH OF speed_heap IS GREATER THAN k THEN  
                REMOVE the SMALLEST element FROM speed_heap  
                DECREMENT speed_sum BY the REMOVED element  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            SET max_performance TO the GREATER value BETWEEN max_performance AND current_performance  
        END FOR  
        RETURN the REMAINDER OF max_performance DIVIDED BY THE INTEGER value of ten RAISED TO THE POWER OF nine PLUS seven  
    END FUNCTION  
END CLASS
