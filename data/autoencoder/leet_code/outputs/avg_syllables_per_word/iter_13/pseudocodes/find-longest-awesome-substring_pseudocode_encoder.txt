CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a new mapping with key zero assigned to value minus one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET char TO the element at position index of s  
            SET digit TO the integer value corresponding to char  
            SET prefix_xor TO prefix_xor EXCLUSIVE OR with the value of one MULTIPLIED BY two RAISED TO THE POWER OF digit  
            
            IF prefix_xor EXISTS AS A KEY IN prefix_map  
                SET max_length TO THE GREATER OF max_length AND index MINUS the VALUE AT prefix_xor IN prefix_map  
            END IF  
            
            FOR bit_position FROM zero TO nine  
                SET flipped TO prefix_xor EXCLUSIVE OR with the value of one MULTIPLIED BY two RAISED TO THE POWER OF bit_position  
                IF flipped EXISTS AS A KEY IN prefix_map  
                    SET max_length TO THE GREATER OF max_length AND index MINUS the VALUE AT flipped IN prefix_map  
                END IF  
            END FOR  
            
            IF prefix_xor DOES NOT EXIST AS A KEY IN prefix_map  
                ASSIGN prefix_map AT KEY prefix_xor TO index  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
