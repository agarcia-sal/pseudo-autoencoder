CLASS Solution  
    FUNCTION integerReplacement(n)  
        SET count TO zero  
        WHILE the value of n IS NOT EQUAL TO one  
            IF the remainder when n is divided BY two EQUALS zero  
                SET n TO the integer division of n BY two  
            ELSE  
                IF n IS EQUAL TO three  
                    DECREMENT n BY one  
                ELSE IF the remainder when the sum of n AND one is divided BY four EQUALS zero  
                    INCREMENT n BY one  
                ELSE  
                    DECREMENT n BY one  
                END IF  
            END IF  
            INCREMENT count BY one  
        END WHILE  
        RETURN count  
    END FUNCTION  
END CLASS
