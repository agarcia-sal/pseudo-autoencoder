CLASS Solution  
    FUNCTION countComponents WITH PARAMETERS n AND edges  
        INITIALIZE graph AS an empty mapping from each integer FROM zero TO n MINUS one TO an empty list  
        FOR each pair a AND b IN edges  
            APPEND b TO the list associated WITH key a IN graph  
            APPEND a TO the list associated WITH key b IN graph  
        END FOR  
        
        INITIALIZE visited AS an empty set  
        
        FUNCTION dfs WITH PARAMETER node  
            IF node IS IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list associated WITH key node IN graph  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        
        SET components TO zero  
        FOR each node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                CALL dfs WITH node  
                INCREMENT components BY one  
            END IF  
        END FOR  
        
        RETURN components  
    END FUNCTION  
END CLASS
