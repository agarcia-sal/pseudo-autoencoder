CLASS Solution  
    FUNCTION maxSumOfThreeSubarrays(nums AS List OF integers, k AS integer) RETURNS List OF integers  
        SET n TO the LENGTH OF nums  
        SET sums TO a new list of size n MINUS k PLUS one FILLED WITH zero  
        SET current_sum TO the SUM OF the ELEMENTS from position zero TO k MINUS one of nums  
        SET the element at position zero of sums TO current_sum  
        
        FOR index i FROM one TO n MINUS k  
            SET current_sum TO current_sum PLUS the element at position i PLUS k MINUS one of nums MINUS the element at position i MINUS one of nums  
            SET the element at position i of sums TO current_sum  
        END FOR  
        
        SET left TO a new list of size n MINUS k PLUS one FILLED WITH zero  
        SET right TO a new list of size n MINUS k PLUS one WHERE every element IS n MINUS k  
        
        SET max_sum TO zero  
        FOR index i FROM zero TO n MINUS k  
            IF the element at position i of sums IS GREATER THAN the element at position max_sum of sums  
                SET max_sum TO i  
            END IF  
            SET the element at position i of left TO max_sum  
        END FOR  
        
        SET max_sum TO n MINUS k  
        FOR index i FROM n MINUS k DOWN TO zero  
            IF the element at position i of sums IS GREATER THAN OR EQUAL TO the element at position max_sum of sums  
                SET max_sum TO i  
            END IF  
            SET the element at position i of right TO max_sum  
        END FOR  
        
        SET max_total TO zero  
        SET result TO a new list of three elements all zero  
        FOR index j FROM k TO n MINUS two TIMES k  
            SET i TO the element at position j MINUS k of left  
            SET l TO the element at position j PLUS k of right  
            SET total TO the element at position i of sums PLUS the element at position j of sums PLUS the element at position l of sums  
            IF total IS GREATER THAN max_total  
                SET max_total TO total  
                SET result TO a new list with elements i, j, l  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
