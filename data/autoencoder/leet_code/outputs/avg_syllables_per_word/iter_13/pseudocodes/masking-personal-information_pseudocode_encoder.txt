CLASS Solution  
    FUNCTION maskPII WITH PARAMETER s AS string  
        IF string contains the character at sign  
            SET local TO the substring of s before the character at sign  
            SET domain TO the substring of s after the character at sign  
            SET masked_local TO the first character of local CONCATENATED WITH five repetitions of the character asterisk CONCATENATED WITH the last character of local  
            RETURN the lowercase form of masked_local CONCATENATED WITH the character at sign CONCATENATED WITH the lowercase form of domain  
        ELSE  
            SET digits TO the result of joining all characters in s that are digits  
            SET local_number TO the concatenation of three characters asterisk CONCATENATED WITH hyphen CONCATENATED WITH three characters asterisk CONCATENATED WITH hyphen CONCATENATED WITH the last four characters of digits  
            IF the length of digits EQUALS ten  
                RETURN local_number  
            ELSE  
                SET country_code_length TO the length of digits MINUS ten  
                SET country_code TO the character plus SIGN CONCATENATED WITH country_code_length repetitions of the character asterisk CONCATENATED WITH hyphen  
                RETURN country_code CONCATENATED WITH local_number  
            END IF  
        END IF  
    END FUNCTION  
END CLASS
