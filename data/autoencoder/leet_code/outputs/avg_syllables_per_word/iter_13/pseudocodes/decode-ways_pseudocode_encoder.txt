CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s IS empty OR element at position zero of s EQUALS character zero  
            RETURN zero  
        END IF  
        SET dp TO list of zeros with length EQUAL TO the LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO one  
        FOR index i FROM two TO the LENGTH OF s PLUS one  
            IF element at position i MINUS one of s NOT EQUALS character zero  
                INCREMENT element at position i of dp BY element at position i MINUS one of dp  
            END IF  
            SET two_digit TO the integer value of substring from position i MINUS two TO i MINUS one of s  
            IF two_digit GREATER THAN OR EQUAL TO ten AND two_digit LESS THAN OR EQUAL TO twenty six  
                INCREMENT element at position i of dp BY element at position i MINUS two of dp  
            END IF  
        END FOR  
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
