CLASS Solution  
    FUNCTION convert(s STRING parameter, numRows INTEGER parameter) RETURNS STRING  
        IF numRows EQUALS one THEN  
            RETURN s  
        END IF  
        
        SET rows TO a list of empty strings with the LENGTH OF numRows  
        SET current_row TO zero  
        SET going_down TO FALSE  
        
        FOR each character IN s  
            SET the element at position current_row of rows TO the element at position current_row of rows CONCATENATED WITH character  
            IF current_row EQUALS zero OR current_row EQUALS numRows MINUS one THEN  
                SET going_down TO NOT going_down  
            END IF  
            IF going_down IS TRUE THEN  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
        
        SET result TO the concatenation of all elements in rows in order  
        RETURN result  
    END FUNCTION  
END CLASS
