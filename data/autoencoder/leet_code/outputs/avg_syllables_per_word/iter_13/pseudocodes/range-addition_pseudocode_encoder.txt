CLASS Solution  
    FUNCTION getModifiedArray LENGTH updates  
        SET diff TO new list of zero with the LENGTH EQUAL TO length PLUS one  
        FOR each element IN updates  
            ASSIGN startIdx TO element at position zero of element  
            ASSIGN endIdx TO element at position one of element  
            ASSIGN inc TO element at position two of element  
            INCREMENT element at position startIdx of diff BY inc  
            IF endIdx PLUS one IS LESS THAN length THEN  
                DECREMENT element at position endIdx PLUS one of diff BY inc  
            END IF  
        END FOR  
        SET arr TO new list of zero with the LENGTH EQUAL TO length  
        SET element at position zero of arr TO element at position zero of diff  
        FOR index FROM one TO length MINUS one  
            SET element at position index of arr TO element at position index MINUS one of arr PLUS element at position index of diff  
        END FOR  
        RETURN arr  
    END FUNCTION  
END CLASS
