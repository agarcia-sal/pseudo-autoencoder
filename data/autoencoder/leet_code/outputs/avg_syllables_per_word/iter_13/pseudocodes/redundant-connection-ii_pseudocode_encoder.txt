CLASS Solution  
    FUNCTION findRedundantDirectedConnection(edges)  
        FUNCTION find(parent, i)  
            IF element at position i of parent NOT EQUALS i THEN  
                SET element at position i of parent TO the result of find with parent and element at position i of parent  
            END IF  
            RETURN element at position i of parent  
        END FUNCTION  
        
        FUNCTION union(parent, rank, x, y)  
            SET rootX TO the result of find with parent and x  
            SET rootY TO the result of find with parent and y  
            IF rootX NOT EQUALS rootY THEN  
                IF element at position rootX of rank GREATER THAN element at position rootY of rank THEN  
                    SET element at position rootY of parent TO rootX  
                ELSE IF element at position rootX of rank LESS THAN element at position rootY of rank THEN  
                    SET element at position rootX of parent TO rootY  
                ELSE  
                    SET element at position rootY of parent TO rootX  
                    INCREMENT element at position rootX of rank BY one  
                END IF  
                RETURN True  
            END IF  
            RETURN False  
        END FUNCTION  
        
        SET n TO the LENGTH OF edges  
        SET parent TO a list representing the sequence of integers from zero TO n inclusive  
        SET rank TO a list filled with zero repeated n plus one times  
        
        SET candidate1 TO None  
        SET candidate2 TO None  
        SET last TO None  
        
        FOR each pair u and v IN edges  
            IF element at position v of parent NOT EQUALS v THEN  
                SET candidate1 TO a pair consisting of element at position v of parent and v  
                SET candidate2 TO the pair u and v  
            ELSE  
                IF the result of union with parent rank u and v EQUALS False THEN  
                    SET last TO the pair u and v  
                END IF  
            END IF  
        END FOR  
        
        IF candidate1 EQUALS None THEN  
            RETURN last  
        END IF  
        
        IF last NOT EQUALS None THEN  
            RETURN candidate1  
        END IF  
        
        RETURN candidate2  
    END FUNCTION  
END CLASS
