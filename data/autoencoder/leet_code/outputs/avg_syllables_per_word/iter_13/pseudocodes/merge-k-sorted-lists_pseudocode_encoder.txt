CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS None  
        SET the val attribute of self TO val  
        SET the next attribute of self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new instance of ListNode initialized with the element at POSITION zero of values  
    SET p TO head  
    FOR each val IN the elements of values from POSITION one TO the end  
        SET node TO a new instance of ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the result of the logical AND between the equality of the val attribute of p1 AND the val attribute of p2 AND the call of is_same_list WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists WITH PARAMETERS self lists  
        SET min_heap TO an empty list  
        FOR each index i AND element lst IN the enumeration of lists  
            IF lst IS NOT None  
                PUSH a tuple consisting of the val attribute of lst the index i and lst INTO min_heap using a min-heap insertion procedure  
            END IF  
        END FOR  
        SET dummy TO a new instance of ListNode initialized with default values  
        SET current TO dummy  
        WHILE min_heap IS NOT empty  
            POP the tuple with the smallest first element FROM min_heap using a min-heap extraction procedure AND UNPACK it INTO val i node  
            SET the next attribute of current TO node  
            SET current TO the next attribute of current  
            IF the next attribute of node IS NOT None  
                PUSH a tuple consisting of the val attribute of the next attribute of node the index i and the next attribute of node INTO min_heap using a min-heap insertion procedure  
            END IF  
        END WHILE  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
