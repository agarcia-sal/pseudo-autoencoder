CLASS Solution  
    FUNCTION wordsTyping(sentence AS List OF strings, rows AS integer, cols AS integer) RETURNS integer  
        SET s TO the result of joining all elements of sentence with a single space CONCATENATED WITH a single space  
        SET total_length TO the LENGTH OF s  
        SET position TO zero  
        FOR each number FROM one TO rows  
            INCREMENT position BY cols  
            IF the element at position MODULO total_length of s EQUALS a single space  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the element at the position MINUS one MODULO total_length of s NOT EQUALS a single space  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        RETURN position DIVIDED BY total_length  
    END FUNCTION  
END CLASS
