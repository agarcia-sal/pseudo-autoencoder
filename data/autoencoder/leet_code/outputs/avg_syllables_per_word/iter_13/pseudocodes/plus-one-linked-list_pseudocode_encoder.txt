CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS none  
        SET self DOT val TO val  
        SET self DOT next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode WITH value element at position zero of values  
    SET p TO head  
    FOR each val IN the elements starting from position one TO the end of values  
        SET node TO new ListNode WITH value val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND is_same_list WITH PARAMETERS the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne WITH PARAMETERS self head  
        FUNCTION reverseList WITH PARAMETERS node  
            SET prev TO none  
            WHILE node IS NOT none  
                SET next_node TO the next attribute of node  
                SET the next attribute of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO reverseList WITH PARAMETERS head  
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current IS NOT none AND carry IS NOT zero  
            SET the val attribute of current TO the val attribute of current PLUS carry  
            SET carry TO the val attribute of current DIVIDED BY ten  
            SET the val attribute of current TO the val attribute of current MODULO ten  
            
            IF the next attribute of current IS none AND carry IS NOT zero  
                SET the next attribute of current TO new ListNode WITH value carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO the next attribute of current  
        END WHILE  
        
        RETURN reverseList WITH PARAMETERS reversed_head  
    END FUNCTION  
END CLASS
