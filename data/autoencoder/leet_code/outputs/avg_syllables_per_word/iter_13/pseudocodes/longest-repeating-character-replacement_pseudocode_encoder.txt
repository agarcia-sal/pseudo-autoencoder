CLASS Solution  
    FUNCTION characterReplacement(s, k)  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        SET count TO a mapping of characters to integer starting at zero for any unseen character  
        FOR each right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the element at position right of string s IN count BY one  
            SET max_count TO the GREATER OF max_count AND the element at position right of string s IN count  
            IF the LENGTH OF the current window from left TO right MINUS max_count IS GREATER THAN k THEN  
                DECREMENT the element at position left of string s IN count BY one  
                INCREMENT left BY one  
            END IF  
            SET max_length TO the GREATER OF max_length AND the LENGTH OF the current window from left TO right  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
