CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix, x, y)  
            SET directions TO a list containing the pairs minus one zero and one zero zero minus one zero one and zero zero  
            FOR each dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET the element at position nx of matrix at position ny TO one MINUS the element at position nx of matrix at position ny  
                END IF  
            END FOR  
        END FUNCTION  

        FUNCTION matrix_to_tuple(matrix)  
            RETURN a tuple converted from each row converted to a tuple from matrix  
        END FUNCTION  

        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF the element at position zero of mat  
        SET target TO a tuple converted from a repeated tuple of zero each repeated n times repeated m times  
        SET initial_state TO the result of matrix_to_tuple applied to mat  
        SET queue TO a list containing the pair of initial_state and zero  
        SET visited TO an empty set  
        ADD initial_state TO visited  

        WHILE queue IS NOT empty  
            REMOVE the element at position zero from queue and SET current_state and steps TO the unpacked result  
            IF current_state EQUALS target  
                RETURN steps  
            END IF  

            SET current_matrix TO a list converted from each row converted to a list from current_state  

            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO a deep copy of current_matrix  
                    CALL flip with arguments next_matrix i and j  
                    SET next_state TO the result of matrix_to_tuple applied to next_matrix  
                    IF next_state IS NOT IN visited  
                        ADD next_state TO visited  
                        APPEND the pair of next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
