CLASS Solution  
    FUNCTION networkDelayTime(times AS List of List of integer comma n AS integer comma k AS integer) RETURNS integer  
        SET graph TO an empty mapping from each integer from one TO n TO an empty list  
        FOR each triple u comma v comma w IN times  
            APPEND a pair of v comma w TO element at position u of graph  
        END FOR  
        SET dist TO a mapping from each integer from one TO n TO positive infinity  
        SET element at position k of dist TO zero  
        SET min_heap TO a list containing a pair of zero comma k  
        WHILE min_heap is not empty  
            REMOVE and RETURN the smallest element from min_heap as current_dist comma node  
            IF current_dist GREATER THAN element at position node of dist  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            FOR each pair neighbor comma weight IN element at position node of graph  
                SET distance TO current_dist PLUS weight  
                IF distance LESS THAN element at position neighbor of dist  
                    SET element at position neighbor of dist TO distance  
                    ADD the pair distance comma neighbor INTO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        SET max_time TO the maximum value among all values in dist  
        IF max_time NOT EQUALS positive infinity  
            RETURN max_time  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
