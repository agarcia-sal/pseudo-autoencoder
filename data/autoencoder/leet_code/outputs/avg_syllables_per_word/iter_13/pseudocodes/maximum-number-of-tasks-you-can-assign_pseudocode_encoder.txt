CLASS Solution  
    FUNCTION maxTaskAssign(tasks, workers, pills, strength)  
        CALL sort ON tasks  
        CALL sort ON workers  
        
        FUNCTION canAssignTasks(k)  
            SET available_workers TO the LIST containing the last k elements of workers  
            SET pills_used TO zero  
            
            FOR i FROM k MINUS one DOWN TO zero  
                SET task TO element at position i of tasks  
                SET worker_idx TO the POSITION to INSERT task IN available_workers TO KEEP available_workers SORTED  
                IF worker_idx LESS THAN the LENGTH OF available_workers  
                    REMOVE the element at position worker_idx FROM available_workers  
                ELSE IF pills_used LESS THAN pills  
                    SET worker_idx TO the POSITION to INSERT task MINUS strength IN available_workers TO KEEP available_workers SORTED  
                    IF worker_idx LESS THAN the LENGTH OF available_workers  
                        REMOVE the element at position worker_idx FROM available_workers  
                        INCREMENT pills_used BY one  
                    ELSE  
                        RETURN False  
                    END IF  
                ELSE  
                    RETURN False  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the MINIMUM OF the LENGTH OF tasks AND the LENGTH OF workers PLUS one  
        WHILE left LESS THAN right  
            SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
            IF canAssignTasks(mid) EQUALS True  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN left MINUS one  
    END FUNCTION  
END CLASS
