CLASS TreeNode  
    FUNCTION __init__ WITH parameters val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH parameter values of type list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS greater than zero  
        SET node TO an element removed from the front of queue  
        IF i IS less than the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO a new TreeNode constructed with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS less than the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO a new TreeNode constructed with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH parameters p and q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value of p IS NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        SET left_trees_same TO the result of is_same_tree with parameters p dot left and q dot left  
        SET right_trees_same TO the result of is_same_tree with parameters p dot right and q dot right  
        RETURN left_trees_same AND right_trees_same  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten WITH parameters self and root of optional TreeNode type  
        IF root IS none  
            RETURN  
        END IF  
        SET stack TO a list containing root  
        WHILE the LENGTH OF stack IS greater than zero  
            SET node TO the element removed from the top of stack  
            IF node dot right IS NOT none  
                APPEND node dot right TO stack  
            END IF  
            IF node dot left IS NOT none  
                APPEND node dot left TO stack  
            END IF  
            SET node dot left TO none  
            IF the LENGTH OF stack IS greater than zero  
                SET node dot right TO the element at position negative one of stack  
            ELSE  
                SET node dot right TO none  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
