CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero LEFT EQUAL TO none RIGHT EQUAL TO none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values DOES NOT EXIST  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY DOUBLE-ENDED QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p DOES NOT EXIST AND q DOES NOT EXIST  
        RETURN True  
    ELSE IF p DOES NOT EXIST OR q DOES NOT EXIST  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN THE LOGICAL AND OF is_same_tree WITH PARAMETERS p.left AND q.left AND is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION convertBST WITH PARAMETER root AS OPTIONAL TreeNode  
        SET self.sum TO zero  

        FUNCTION traverse WITH PARAMETER node  
            IF node DOES NOT EXIST  
                RETURN  
            END IF  
            CALL traverse WITH PARAMETER node.right  
            INCREMENT self.sum BY node.val  
            SET node.val TO self.sum  
            CALL traverse WITH PARAMETER node.left  
        END FUNCTION  

        CALL traverse WITH PARAMETER root  
        RETURN root  
    END FUNCTION  
END CLASS
