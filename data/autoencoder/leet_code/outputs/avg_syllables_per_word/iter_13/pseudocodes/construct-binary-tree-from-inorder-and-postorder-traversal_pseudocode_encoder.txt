CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero left EQUALS none right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO A NEW double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element at the front of queue REMOVED  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS TO q.val  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS p.left q.left AND the RESULT OF is_same_tree WITH PARAMETERS p.right q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree WITH PARAMETERS self inorder postorder  
        IF the LENGTH OF inorder EQUALS zero OR the LENGTH OF postorder EQUALS zero  
            RETURN none  
        END IF  
        SET root_val TO the element REMOVED FROM THE END of postorder  
        SET root TO A NEW TreeNode INITIALIZED WITH root_val  
        SET root_index TO the POSITION OF root_val IN inorder  
        SET root.right TO the RESULT OF self.buildTree WITH PARAMETERS the sublist of inorder FROM position root_index PLUS one TO THE END AND postorder  
        SET root.left TO the RESULT OF self.buildTree WITH PARAMETERS the sublist of inorder FROM position zero TO position root_index MINUS one INCLUDING root_index EXCLUDED AND postorder  
        RETURN root  
    END FUNCTION  
END CLASS
