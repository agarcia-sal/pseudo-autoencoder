CLASS Solution  
    FUNCTION isScramble WITH PARAMETERS s1 AND s2  
        IF the string s1 IS EQUALS TO the string s2  
            RETURN True  
        END IF  
        IF the characters of string s1 SORTED ARE NOT EQUALS TO the characters of string s2 SORTED  
            RETURN False  
        END IF  
        SET n TO the LENGTH OF string s1  
        FOR index i FROM one TO n MINUS one  
            IF the INVOCATION OF isScramble WITH the substring of s1 FROM position one TO i AND the substring of s2 FROM position one TO i RETURNS True AND  
               the INVOCATION OF isScramble WITH the substring of s1 FROM position i PLUS one TO the END AND the substring of s2 FROM position i PLUS one TO the END RETURNS True  
                RETURN True  
            END IF  
            IF the INVOCATION OF isScramble WITH the substring of s1 FROM position one TO i AND the substring of s2 FROM position n MINUS i PLUS one TO the END RETURNS True AND  
               the INVOCATION OF isScramble WITH the substring of s1 FROM position i PLUS one TO the END AND the substring of s2 FROM position one TO n MINUS i RETURNS True  
                RETURN True  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
