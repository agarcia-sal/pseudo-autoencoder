CLASS Solution  
    FUNCTION minDistance(height, width, tree, squirrel, nuts)  
        FUNCTION distance(p1, p2)  
            RETURN THE ABSOLUTE VALUE OF THE ELEMENT at position zero OF p1 MINUS THE ELEMENT at position zero OF p2 PLUS THE ABSOLUTE VALUE OF THE ELEMENT at position one OF p1 MINUS THE ELEMENT at position one OF p2  
        END FUNCTION  
        
        SET total_distance TO zero  
        SET max_save TO NEGATIVE INFINITY  
        
        FOR each nut IN nuts  
            INCREMENT total_distance BY two MULTIPLIED BY distance OF nut AND tree  
            SET first_nut_distance TO distance OF squirrel AND nut PLUS distance OF nut AND tree  
            SET later_nut_distance TO two MULTIPLIED BY distance OF nut AND tree  
            SET save TO later_nut_distance MINUS first_nut_distance  
            SET max_save TO THE GREATER VALUE OF max_save AND save  
        END FOR  
        
        RETURN total_distance MINUS max_save  
    END FUNCTION  
END CLASS
