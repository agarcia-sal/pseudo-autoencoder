CLASS Solution  
    FUNCTION canVisitAllRooms with parameter rooms  
        SET visited TO empty set  
        SET stack TO list containing zero  
        WHILE stack is not empty  
            SET current_room TO element removed from the end of stack  
            ADD current_room TO visited  
            FOR each key IN element at position current_room of rooms  
                IF key IS NOT IN visited  
                    APPEND key TO stack  
                END IF  
            END FOR  
        END WHILE  
        RETURN the LENGTH OF visited EQUALS the LENGTH OF rooms  
    END FUNCTION  
END CLASS
