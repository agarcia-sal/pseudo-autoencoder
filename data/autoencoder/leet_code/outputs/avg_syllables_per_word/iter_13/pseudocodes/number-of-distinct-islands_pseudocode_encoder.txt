CLASS Solution  
    FUNCTION numDistinctIslands(grid)  
        FUNCTION dfs(i, j, direction)  
            IF i IS GREATER THAN OR EQUAL TO zero AND i IS LESS THAN the LENGTH OF grid AND j IS GREATER THAN OR EQUAL TO zero AND j IS LESS THAN the LENGTH OF element at position zero of grid AND element at position i at position j of grid EQUALS one  
                SET element at position i at position j of grid TO zero  
                APPEND direction TO path  
                CALL dfs WITH arguments i PLUS one, j, direction LETTER d  
                CALL dfs WITH arguments i MINUS one, j, direction LETTER u  
                CALL dfs WITH arguments i, j PLUS one, direction LETTER r  
                CALL dfs WITH arguments i, j MINUS one, direction LETTER l  
                APPEND direction LETTER b TO path  
            END IF  
            RETURN path  
        END FUNCTION  
        
        SET unique_islands TO an empty set  
        
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i at position j of grid EQUALS one  
                    SET path TO an empty list  
                    ADD the result of calling dfs WITH arguments i, j, direction LETTER o CONVERTED to a tuple TO unique_islands  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the LENGTH OF unique_islands  
    END FUNCTION  
END CLASS
