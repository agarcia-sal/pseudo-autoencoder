CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO an empty mapping of each course from zero TO numCourses MINUS one paired with an empty list  
        SET in_degrees TO a list of zeros with length equal to numCourses  
        
        FOR each pair consisting of destination and source IN prerequisites  
            APPEND destination TO the list at key source in adj_list  
            INCREMENT the element at position destination in in_degrees BY one  
        END FOR  
        
        SET queue TO a double ended queue containing each course FROM zero TO numCourses MINUS one WHERE the element at position course in in_degrees EQUALS zero  
        SET topological_order TO an empty list  
        
        WHILE queue has elements  
            SET current TO the element removed from the left of queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN the list at key current in adj_list  
                DECREMENT the element at position neighbor in in_degrees BY one  
                IF the element at position neighbor in in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
