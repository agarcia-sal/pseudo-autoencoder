CLASS Solution  
    FUNCTION kSimilarity(s1, s2)  
        IMPORT deque FROM collections  
        
        FUNCTION neighbors(s)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND element at position i of s EQUALS element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF element at position j of s EQUALS element at position i of s2 AND element at position j of s IS NOT EQUAL TO element at position j of s2  
                    YIELD a new string formed by concatenating substring of s from position zero TO i MINUS one PLUS element at position j of s PLUS substring of s from position i PLUS one TO j MINUS one PLUS element at position i of s PLUS substring of s from position j PLUS one TO the end  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET queue TO a new deque containing a pair of s1 and zero  
        SET visited TO a new set containing s1  
        
        WHILE queue IS NOT EMPTY  
            ASSIGN s and swaps TO the first element removed from queue  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN the result of calling neighbors with s  
                IF neighbor IS NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND a pair of neighbor and swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
