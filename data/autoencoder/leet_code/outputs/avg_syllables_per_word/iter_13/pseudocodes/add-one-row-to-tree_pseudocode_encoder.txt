CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the values collection IS empty  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO a new double-ended queue  
    APPEND root TO queue  
    WHILE the queue IS not empty  
        SET node TO the element removed from the front of queue  
        IF i IS less than the LENGTH OF values collection AND the element at position i of values IS not nothing  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS less than the LENGTH OF values collection AND the element at position i of values IS not nothing  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS nothing AND q IS nothing  
        RETURN true  
    ELSE IF p IS nothing OR q IS nothing  
        RETURN false  
    ELSE IF the value property of p IS NOT EQUALS the value property of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with the left property of p AND the left property of q  
        AND the result of is_same_tree with the right property of p AND the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow(self, root, val, depth)  
        IF depth EQUALS one  
            RETURN a new TreeNode initialized with val, left assigned to root  
        END IF  
        SET queue TO a new double-ended queue containing the tuple of root AND one  
        WHILE the queue IS not empty  
            SET node, level TO the tuple removed from the front of queue  
            IF level EQUALS depth MINUS one  
                SET node.left TO a new TreeNode initialized with val, left assigned to the left property of node  
                SET node.right TO a new TreeNode initialized with val, right assigned to the right property of node  
            ELSE IF level IS less than depth MINUS one  
                IF the left property of node IS not nothing  
                    APPEND the tuple of node.left AND level PLUS one TO queue  
                END IF  
                IF the right property of node IS not nothing  
                    APPEND the tuple of node.right AND level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
