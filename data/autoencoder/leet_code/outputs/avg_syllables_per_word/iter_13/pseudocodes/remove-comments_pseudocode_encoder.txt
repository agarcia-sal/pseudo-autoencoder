CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO False  
        SET result TO empty list  
        SET current_line TO empty list  
        FOR each line IN source  
            SET i TO zero  
            WHILE i IS LESS THAN the LENGTH OF line  
                IF in_block_comment IS True  
                    IF substring from position i TO position i PLUS one of line EQUALS the string denoting asterisk followed by forward slash  
                        SET in_block_comment TO False  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF substring from position i TO position i PLUS one of line EQUALS the string denoting two forward slashes  
                        BREAK the WHILE loop  
                    ELSE IF substring from position i TO position i PLUS one of line EQUALS the string denoting forward slash followed by asterisk  
                        SET in_block_comment TO True  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND element at position i of line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  
            IF current_line IS NOT empty AND in_block_comment IS False  
                APPEND the concatenation of all elements in current_line TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
