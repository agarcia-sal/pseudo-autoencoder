CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO a new dictionary with default value zero for any key  
        FOR each index i FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO the substring from position i TO i PLUS ten MINUS one of s  
            INCREMENT the count at key sequence in sequence_count BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each sequence and count IN sequence_count entries  
            IF count GREATER THAN one THEN  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
