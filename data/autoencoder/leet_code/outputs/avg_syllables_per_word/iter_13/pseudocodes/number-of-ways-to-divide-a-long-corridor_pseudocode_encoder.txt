CLASS Solution  
    FUNCTION numberOfWays(corridor)  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven  
        SET seats TO empty list  
        FOR each position AND each character IN corridor  
            IF character EQUALS character S  
                APPEND position TO seats  
            END IF  
        END FOR  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM two TO the LENGTH OF seats MINUS one STEP two  
            SET difference TO element at position index of seats MINUS element at position index MINUS one of seats  
            SET ways TO ways MULTIPLIED BY difference MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
