CLASS Solution  
    FUNCTION maxCandies WITH PARAMETERS status AS List of integers candies AS List of integers keys AS List of List of integers containedBoxes AS List of List of integers initialBoxes AS List of integers  
        SET boxes_we_have TO the set of elements in initialBoxes  
        SET keys_we_have TO an empty set  
        SET total_candies TO zero  
        SET progress_made TO true  
        WHILE progress_made holds  
            SET progress_made TO false  
            FOR each box IN the list formed from the elements in boxes_we_have  
                IF the element at position box of status EQUALS one OR box IS an element of keys_we_have  
                    INCREMENT total_candies BY the element at position box of candies  
                    ADD all elements from the list at position box of keys TO keys_we_have  
                    ADD all elements from the list at position box of containedBoxes TO boxes_we_have  
                    REMOVE box FROM boxes_we_have  
                    SET progress_made TO true  
                END IF  
            END FOR  
        END WHILE  
        RETURN total_candies  
    END FUNCTION  
END CLASS
