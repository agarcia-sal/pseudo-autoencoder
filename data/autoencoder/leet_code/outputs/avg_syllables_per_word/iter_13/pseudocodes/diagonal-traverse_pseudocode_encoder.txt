CLASS Solution  
    FUNCTION findDiagonalOrder(mat)  
        IF mat IS EMPTY OR element at position zero of mat IS EMPTY  
            RETURN an empty list  
        END IF  
        
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET result TO an empty list  
        SET r TO zero  
        SET c TO zero  
        SET direction TO one  
        
        FOR each iteration FROM zero TO the MULTIPLICATION OF m AND n MINUS one  
            APPEND element at position r of mat and position c of that element TO result  
            IF direction EQUALS one  
                SET new_r TO r MINUS one  
                SET new_c TO c PLUS one  
            ELSE  
                SET new_r TO r PLUS one  
                SET new_c TO c MINUS one  
            END IF  
            
            IF new_r IS LESS THAN zero OR new_r EQUALS m OR new_c IS LESS THAN zero OR new_c EQUALS n  
                IF direction EQUALS one  
                    IF new_c EQUALS n  
                        SET new_r TO r PLUS one  
                        SET new_c TO c  
                    ELSE  
                        SET new_r TO zero  
                        SET new_c TO c PLUS one  
                    END IF  
                ELSE  
                    IF new_r EQUALS m  
                        SET new_r TO r  
                        SET new_c TO c PLUS one  
                    ELSE  
                        SET new_r TO r PLUS one  
                        SET new_c TO zero  
                    END IF  
                END IF  
                SET direction TO the NEGATION OF direction  
            END IF  
            
            SET r TO new_r  
            SET c TO new_c  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
