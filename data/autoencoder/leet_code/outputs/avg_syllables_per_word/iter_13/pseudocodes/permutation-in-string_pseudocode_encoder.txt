CLASS Solution  
    FUNCTION checkInclusion(s1 AS string, s2 AS string) RETURNS boolean  
        FROM the collection library ASSIGN Counter TO the frequency counting structure
        
        SET length of s1 TO len_s1  
        SET length of s2 TO len_s2  
        
        IF len_s1 IS GREATER THAN len_s2  
            RETURN false  
        END IF  
        
        SET count_s1 TO the frequency count of each character in s1 using Counter  
        SET count_s2 TO the frequency count of each character in the substring from position zero TO the position len_s1 MINUS one of s2 using Counter  
        
        IF count_s1 EQUALS count_s2  
            RETURN true  
        END IF  
        
        FOR each index i FROM len_s1 TO len_s2 MINUS one  
            INCREMENT the frequency count of the character at position i of s2 in count_s2 BY one  
            
            DECREMENT the frequency count of the character at position i MINUS len_s1 of s2 in count_s2 BY one  
            
            IF the frequency count of the character at position i MINUS len_s1 of s2 in count_s2 EQUALS zero  
                REMOVE the character at position i MINUS len_s1 of s2 FROM count_s2  
            END IF  
            
            IF count_s1 EQUALS count_s2  
                RETURN true  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
