CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF houses IS empty  
                RETURN zero  
            END IF  
            IF the LENGTH OF houses EQUALS one  
                RETURN element at position zero of houses  
            END IF  
            SET dp TO a list of zero elements with the LENGTH OF houses  
            SET element at position zero of dp TO element at position zero of houses  
            SET element at position one of dp TO the GREATER OF element at position zero of houses AND element at position one of houses  
            FOR index FROM two TO the LENGTH OF houses MINUS one  
                SET element at position index of dp TO the GREATER OF element at position index MINUS one of dp AND the SUM OF element at position index MINUS two of dp AND element at position index of houses  
            END FOR  
            RETURN the element at the LAST POSITION of dp  
        END FUNCTION  
        IF nums IS empty  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN element at position zero of nums  
        END IF  
        RETURN the GREATER OF rob_linear with the SUBLIST of nums FROM position zero TO the POSITION BEFORE the LAST AND rob_linear with the SUBLIST of nums FROM position one TO the LAST POSITION  
    END FUNCTION  
END CLASS
