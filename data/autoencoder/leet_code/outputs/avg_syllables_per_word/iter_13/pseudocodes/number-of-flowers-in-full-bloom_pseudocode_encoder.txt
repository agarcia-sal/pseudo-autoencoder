CLASS Solution  
    FUNCTION fullBloomFlowers(flowers AS List OF List OF Integer, people AS List OF Integer) RETURNS List OF Integer  
        SET start_times TO sorted list of the first element of each pair in flowers  
        SET end_times TO sorted list of the second element of each pair in flowers  
        SET result TO empty list  
        FOR each person IN people  
            SET started TO the position to insert person in start_times such that all elements to the left are less than or equal to person  
            SET ended TO the position to insert person in end_times such that all elements to the left are strictly less than person  
            APPEND the difference between started and ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
