CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET list_of_fractions TO expression REPLACED all occurrences of minus SIGN WITH plus SIGN FOLLOWED BY minus SIGN THEN SPLIT by plus SIGN  
        SET result_fraction TO zero as Fraction TYPE  
        FOR each fraction_string IN list_of_fractions  
            IF fraction_string IS NOT empty string  
                ADD Fraction REPRESENTED BY fraction_string TO result_fraction  
            END IF  
        END FOR  
        RETURN string REPRESENTING numerator OF result_fraction FOLLOWED BY slash FOLLOWED BY denominator OF result_fraction  
    END FUNCTION  
END CLASS
