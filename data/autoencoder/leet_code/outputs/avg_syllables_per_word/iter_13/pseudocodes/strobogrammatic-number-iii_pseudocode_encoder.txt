CLASS Solution  
    FUNCTION strobogrammaticInRange(low STRING, high STRING) RETURNS INTEGER  
        FUNCTION generate_strobogrammatic(n INTEGER, final_length INTEGER) RETURNS LIST OF STRING  
            IF n EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF n EQUALS one  
                RETURN list containing string zero string one string eight  
            END IF  
            SET middles TO generate_strobogrammatic(n MINUS two, final_length)  
            SET result TO empty list  
            FOR each middle IN middles  
                IF n NOT EQUAL TO final_length  
                    APPEND string zero concatenated with middle concatenated with string zero TO result  
                END IF  
                APPEND string one concatenated with middle concatenated with string one TO result  
                APPEND string six concatenated with middle concatenated with string nine TO result  
                APPEND string eight concatenated with middle concatenated with string eight TO result  
                APPEND string nine concatenated with middle concatenated with string six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        FUNCTION count_strobogrammatic(low STRING, high STRING) RETURNS INTEGER  
            SET count TO zero  
            FOR length FROM the LENGTH OF low TO the LENGTH OF high  
                FOR each num IN generate_strobogrammatic(length, length)  
                    IF length EQUALS the LENGTH OF low AND num LESS THAN low OR length EQUALS the LENGTH OF high AND num GREATER THAN high  
                        CONTINUE to next iteration  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN count_strobogrammatic(low, high)  
    END FUNCTION  
END CLASS
