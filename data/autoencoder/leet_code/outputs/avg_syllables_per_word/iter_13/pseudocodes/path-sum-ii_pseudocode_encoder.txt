CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH ARGUMENT element at position zero of values  
    SET i TO one  
    SET queue TO NEW double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH ARGUMENT element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH ARGUMENT element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val IS NOT EQUALS TO q dot val  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH ARGUMENTS p dot left q dot left AND is_same_tree WITH ARGUMENTS p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum WITH PARAMETERS self root targetSum  
        FUNCTION dfs WITH PARAMETERS node current_sum path  
            IF node IS none  
                RETURN  
            END IF  
            APPEND node dot val TO path  
            SET current_sum TO current_sum PLUS node dot val  
            IF node dot left IS none AND node dot right IS none AND current_sum EQUALS targetSum  
                APPEND A copy OF path TO result  
            END IF  
            CALL dfs WITH ARGUMENTS node dot left current_sum path  
            CALL dfs WITH ARGUMENTS node dot right current_sum path  
            REMOVE last element FROM path  
        END FUNCTION  
        SET result TO empty list  
        CALL dfs WITH ARGUMENTS root zero empty list  
        RETURN result  
    END FUNCTION  
END CLASS
