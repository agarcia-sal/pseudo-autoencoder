CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO an empty list  
        FOR each character IN expression  
            IF character EQUALS any of the characters t or f or exclamation mark or ampersand or vertical bar  
                APPEND character TO stk  
            ELSE IF character EQUALS closing parenthesis  
                SET t TO zero  
                SET f TO zero  
                WHILE the last element of stk EQUALS any of the characters t or f  
                    INCREMENT t BY one if the last element of stk EQUALS t OTHERWISE increment zero  
                    INCREMENT f BY one if the last element of stk EQUALS f OTHERWISE increment zero  
                    REMOVE the last element FROM stk  
                END WHILE  
                SET op TO the last element popped FROM stk  
                IF op EQUALS exclamation mark  
                    SET c TO t if f EQUALS zero OTHERWISE f  
                ELSE IF op EQUALS ampersand  
                    SET c TO f if f GREATER THAN zero OTHERWISE t  
                ELSE IF op EQUALS vertical bar  
                    SET c TO t if t GREATER THAN zero OTHERWISE f  
                END IF  
                APPEND c TO stk  
            END IF  
        END FOR  
        RETURN the first element of stk EQUALS t  
    END FUNCTION  
END CLASS
