CLASS Solution  
    FUNCTION findNumberOfLIS(nums)  
        IF nums EQUALS empty  
            RETURN zero  
        END IF  

        SET n TO the LENGTH OF nums  
        SET lengths TO a new list of size n FILLED WITH one  
        SET counts TO a new list of size n FILLED WITH one  

        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO i MINUS one  
                IF the element at position i of nums IS GREATER THAN the element at position j of nums  
                    IF the element at position j of lengths PLUS one IS GREATER THAN the element at position i of lengths  
                        SET the element at position i of lengths TO the element at position j of lengths PLUS one  
                        SET the element at position i of counts TO the element at position j of counts  
                    ELSE IF the element at position j of lengths PLUS one EQUALS the element at position i of lengths  
                        INCREMENT the element at position i of counts BY the element at position j of counts  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        SET longest TO the MAXIMUM VALUE in lengths  
        SET total_count TO zero  
        FOR index i FROM zero TO n MINUS one  
            IF the element at position i of lengths EQUALS longest  
                INCREMENT total_count BY the element at position i of counts  
            END IF  
        END FOR  

        RETURN total_count  
    END FUNCTION  
END CLASS
