CLASS Solution  
    FUNCTION mincostToHireWorkers WITH PARAMETERS quality AS List OF integers wage AS List OF integers k AS integer RETURNS float  
        SET workers TO the LIST OF tuples CONSISTING OF EACH wage VALUE DIVIDED BY corresponding quality VALUE AND corresponding quality VALUE FOR EACH pair OF wage AND quality TAKEN TOGETHER  
        SORT workers BY the FIRST ELEMENT OF each tuple IN ascending order  
        
        SET max_heap TO an empty list TO REPRESENT a max heap by storing NEGATIVE OF quality VALUES  
        SET total_quality TO zero  
        SET min_cost TO infinity  
        
        FOR each pair OF ratio AND q IN workers  
            PUSH NEGATIVE OF q INTO max_heap  
            INCREMENT total_quality BY q  
            
            IF the LENGTH OF max_heap IS GREATER THAN k THEN  
                REMOVE the smallest element FROM max_heap AND ADD it TO total_quality  
            END IF  
            
            IF the LENGTH OF max_heap IS EQUAL TO k THEN  
                SET min_cost TO the LOWER VALUE BETWEEN min_cost AND total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
