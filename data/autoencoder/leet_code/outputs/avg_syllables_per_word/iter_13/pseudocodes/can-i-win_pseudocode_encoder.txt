CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the product of maxChoosableInteger AND the sum of maxChoosableInteger PLUS one DIVIDED BY two is LESS THAN desiredTotal  
            RETURN false  
        END IF  
        IF desiredTotal IS LESS THAN OR EQUAL TO zero  
            RETURN true  
        END IF  
        SET memo TO an empty associative collection  
        FUNCTION can_win(current_total, used_numbers)  
            IF the pair of current_total AND used_numbers IS IN memo  
                RETURN the value associated with the pair of current_total AND used_numbers in memo  
            END IF  
            FOR each number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers AND two RAISED TO THE POWER OF number IS EQUAL TO zero  
                    IF the sum of current_total AND number IS GREATER THAN OR EQUAL TO desiredTotal OR calling can_win with the sum of current_total AND number AND the bitwise OR of used_numbers AND two RAISED TO THE POWER OF number RETURNS false  
                        SET the value in memo associated with the pair of current_total AND used_numbers TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            SET the value in memo associated with the pair of current_total AND used_numbers TO false  
            RETURN false  
        END FUNCTION  
        RETURN the result of calling can_win with zero AND zero  
    END FUNCTION  
END CLASS
