CLASS Solution  
    FUNCTION solveSudoku WITH parameters board  
        FUNCTION is_valid WITH parameters board row col num  
            FOR index FROM zero TO eight  
                IF element at position index of element at position row of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            FOR index FROM zero TO eight  
                IF element at position row of element at position col of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY the INTEGER DIVISION OF row BY three  
            SET start_col TO three MULTIPLIED BY the INTEGER DIVISION OF col BY three  
            FOR i FROM zero TO two  
                FOR j FROM zero TO two  
                    IF element at position start_row PLUS i of element at position start_col PLUS j of board EQUALS num  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        FUNCTION solve WITH parameter board  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF element at position row of element at position col of board EQUALS the character representing empty cell  
                        FOR num AS each string number FROM one TO nine  
                            IF is_valid WITH arguments board row col num RETURNS True  
                                SET element at position row of element at position col of board TO num  
                                IF solve WITH argument board RETURNS True  
                                    RETURN True  
                                END IF  
                                SET element at position row of element at position col of board TO the character representing empty cell  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        CALL solve WITH argument board  
    END FUNCTION  
END CLASS
