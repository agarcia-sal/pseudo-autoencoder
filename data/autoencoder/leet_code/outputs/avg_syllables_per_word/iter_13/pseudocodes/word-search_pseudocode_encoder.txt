CLASS Solution  
    FUNCTION exist(board, word)  
        IF board EQUALS empty  
            RETURN False  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        SET visited TO a two dimensional list of Boolean False values with m rows and n columns  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS the LENGTH OF word  
                RETURN True  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x THEN element at position y of visited EQUALS True OR element at position x THEN element at position y of board NOT EQUALS element at position index of word  
                RETURN False  
            END IF  

            SET element at position x THEN element at position y of visited TO True  
            FOR each pair dx AND dy IN a list of four pairs consisting of minus one and zero plus one and zero zero and minus one zero and plus one  
                IF dfs x PLUS dx y PLUS dy index PLUS one RETURNS True  
                    RETURN True  
                END IF  
            END FOR  
            SET element at position x THEN element at position y of visited TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs i j zero RETURNS True  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
