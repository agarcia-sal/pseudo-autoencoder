CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1, val2)  
            RETURN (val1 GREATER THAN zero AND val2 GREATER THAN zero) OR (val1 LESS THAN zero AND val2 LESS THAN zero)  
        END FUNCTION  
      
        SET n TO the LENGTH OF nums  
      
        FOR i FROM zero TO n MINUS one  
            SET slow TO i  
            SET fast TO i  
          
            IF element at position i of nums EQUALS zero  
                CONTINUE TO next iteration of the FOR loop  
            END IF  
          
            WHILE true holds  
                SET slow TO the remainder after division of the SUM of slow PLUS element at position slow of nums BY n  
                IF slow LESS THAN zero  
                    INCREMENT slow BY n  
                END IF  
              
                SET fast TO the remainder after division of the SUM of fast PLUS element at position fast of nums BY n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  
                SET fast TO the remainder after division of the SUM of fast PLUS element at position fast of nums BY n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  
              
                IF NOT sameDirection(element at position slow of nums, element at position i of nums) OR NOT sameDirection(element at position fast of nums, element at position i of nums)  
                    BREAK the WHILE loop  
                END IF  
              
                IF slow EQUALS fast  
                    IF slow EQUALS the remainder after division of the SUM of slow PLUS element at position slow of nums BY n  
                        BREAK the WHILE loop  
                    END IF  
                    RETURN true  
                END IF  
            END WHILE  
        END FOR  
      
        RETURN false  
    END FUNCTION  
END CLASS
