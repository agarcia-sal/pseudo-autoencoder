CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH value element at position zero of values  
    SET i TO one  
    SET queue TO A NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO A NEW TreeNode WITH value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO A NEW TreeNode WITH value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH arguments p.left AND q.left AND the result of is_same_tree WITH arguments p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom(self, root)  
        IF root IS none  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO A NEW deque INITIALIZED WITH root  
        WHILE queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR count FROM one TO level_size  
                SET node TO element removed from the left end of queue  
                APPEND node.val TO current_level  
                IF node.left IS NOT none  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT none  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result IN reverse order  
    END FUNCTION  
END CLASS
