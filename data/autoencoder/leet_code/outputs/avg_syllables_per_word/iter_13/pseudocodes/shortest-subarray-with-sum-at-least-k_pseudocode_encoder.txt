CLASS Solution  
    FUNCTION shortestSubarray(nums AS List of integer, k AS integer) RETURNS integer  
        SET prefix_sum TO a list containing zero  
        FOR each number IN nums  
            SET last_element TO element at the last position of prefix_sum  
            SET new_sum TO last_element PLUS number  
            APPEND new_sum TO prefix_sum  
        END FOR  
        ASSIGN dq TO an empty double ended queue  
        SET min_length TO infinity  
        FOR each index and current_sum IN enumeration of prefix_sum  
            WHILE dq is not empty AND current_sum MINUS element at position element at position zero of dq of prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET candidate_length TO index MINUS result of removing and returning element at position zero of dq  
                IF candidate_length IS LESS THAN min_length  
                    SET min_length TO candidate_length  
                END IF  
            END WHILE  
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO element at position element at the last position of dq of prefix_sum  
                REMOVE element at the last position of dq  
            END WHILE  
            APPEND index TO dq  
        END FOR  
        IF min_length IS EQUAL TO infinity  
            RETURN negative one  
        ELSE  
            RETURN min_length  
        END IF  
    END FUNCTION  
END CLASS
