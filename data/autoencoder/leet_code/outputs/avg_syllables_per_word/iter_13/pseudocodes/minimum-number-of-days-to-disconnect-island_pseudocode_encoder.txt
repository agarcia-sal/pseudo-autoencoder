CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the pair of x AND y IS IN visited OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            ADD the pair of x AND y TO visited  
            FOR each pair of dx AND dy IN the list that contains minus one ZERO one ZERO ZERO one  
                CALL dfs with parameter x PLUS dx AND y PLUS dy AND visited  
            END FOR  
        END FUNCTION  
        
        FUNCTION is_connected()  
            SET visited TO an empty set  
            SET land_cells TO the list of pairs of i AND j FOR each i FROM zero TO the LENGTH OF grid MINUS one AND for each j FROM zero TO the LENGTH OF element at position zero of grid MINUS one WHERE element at position i of grid at position j EQUALS one  
            IF land_cells IS empty  
                RETURN True  
            END IF  
            CALL dfs with the first element at position zero of land_cells AS x AND the second element at position one of land_cells AS y AND visited  
            RETURN the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  
        
        IF the call to is_connected() RETURNS False  
            RETURN zero  
        END IF  
        
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    SET element at position i of grid at position j TO zero  
                    IF the call to is_connected() RETURNS False  
                        RETURN one  
                    END IF  
                    SET element at position i of grid at position j TO one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN two  
    END FUNCTION  
END CLASS
