FUNCTION list_node(values)
    IF values EQUALS empty
        RETURN none
    END IF
    SET head TO NEW ListNode WITH val EQUAL TO element at position one of values
    SET p TO head
    FOR each val IN elements of values from position two TO the end
        SET node TO NEW ListNode WITH val EQUAL TO val
        SET next of p TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 EQUALS none AND p2 EQUALS none
        RETURN true
    END IF
    IF p1 EQUALS none OR p2 EQUALS none
        RETURN false
    END IF
    RETURN the value at p1 EQUALS the value at p2 AND is_same_list(the next of p1, the next of p2)
END FUNCTION

CLASS ListNode
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)
        SET the value at self TO val
        SET the next of self TO next
    END FUNCTION
END CLASS

CLASS Solution
    FUNCTION addTwoNumbers(self, l1, l2)
        SET dummy_head TO NEW ListNode WITH val EQUAL TO zero
        SET current TO dummy_head
        SET carry TO zero
        WHILE l1 IS NOT none OR l2 IS NOT none OR carry IS NOT zero
            IF l1 IS NOT none
                SET val1 TO the value at l1
            ELSE
                SET val1 TO zero
            END IF
            IF l2 IS NOT none
                SET val2 TO the value at l2
            ELSE
                SET val2 TO zero
            END IF
            SET total TO val1 PLUS val2 PLUS carry
            SET carry TO total DIVIDED BY ten
            SET next of current TO NEW ListNode WITH val EQUAL TO total MODULO ten
            SET current TO next of current
            IF l1 IS NOT none
                SET l1 TO the next of l1
            END IF
            IF l2 IS NOT none
                SET l2 TO the next of l2
            END IF
        END WHILE
        RETURN the next of dummy_head
    END FUNCTION
END CLASS
