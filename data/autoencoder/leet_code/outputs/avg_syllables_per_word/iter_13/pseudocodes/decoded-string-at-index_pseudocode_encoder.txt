CLASS Solution  
    FUNCTION decodeAtIndex(s, k)  
        SET length TO zero  
        FOR each char IN s  
            IF char IS a digit  
                SET digit_value TO the integer value of char  
                SET length TO length MULTIPLIED BY digit_value  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each char IN the reverse order of s  
            SET k TO the remainder of k DIVIDED BY length  
            IF k EQUALS zero AND char IS a letter  
                RETURN char  
            END IF  
            IF char IS a digit  
                SET digit_value TO the integer value of char  
                SET length TO length DIVIDED BY digit_value  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
