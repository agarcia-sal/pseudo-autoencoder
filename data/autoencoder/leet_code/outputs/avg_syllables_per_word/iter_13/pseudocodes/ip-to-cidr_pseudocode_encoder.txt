CLASS Solution  
    FUNCTION ipToCIDR(ip PARAMETER, n PARAMETER) RETURNS list of strings  
        FUNCTION ip_to_int(ip PARAMETER) RETURNS integer  
            SET parts TO list of integers converted from splitting ip at each dot  
            SET res TO zero  
            FOR each part IN parts  
                SET res TO res MULTIPLIED BY two HUNDRED FIFTY SIX PLUS part  
            END FOR  
            RETURN res  
        END FUNCTION  

        FUNCTION int_to_ip(num PARAMETER) RETURNS string  
            RETURN concatenation of strings representing the element at position index of the sequence obtained by dividing num DIVIDED BY two HUNDRED FIFTY SIX RAISED TO power index MODULO two HUNDRED FIFTY SIX where index decrements from three TO zero separated by dot  
        END FUNCTION  

        FUNCTION largest_cidr_size(start PARAMETER, n PARAMETER) RETURNS integer  
            IF bitwise AND of start AND one EQUALS one  
                RETURN one  
            END IF  
            SET x TO one  
            WHILE x LESS THAN OR EQUAL TO n AND bitwise AND of start AND x MINUS one EQUALS zero  
                SET x TO x MULTIPLIED BY two  
            END WHILE  
            RETURN x DIVIDED BY two  
        END FUNCTION  

        SET result TO empty list  
        SET start_ip TO ip_to_int(ip)  

        WHILE n GREATER THAN zero  
            SET size TO largest_cidr_size(start_ip, n)  
            APPEND concatenation of int_to_ip(start_ip) FOLLOWED BY forward slash FOLLOWED BY thirty TWO MINUS bit length of size PLUS one TO result  
            SET start_ip TO start_ip PLUS size  
            SET n TO n MINUS size  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
