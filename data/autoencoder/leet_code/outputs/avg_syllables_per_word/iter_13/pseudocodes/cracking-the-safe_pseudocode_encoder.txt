CLASS Solution  
    FUNCTION crackSafe WITH PARAMETERS n AND k  
        SET visited TO an empty set of unique sequences  
        SET sequence TO an empty list to store the Eulerian path elements  
        
        FUNCTION dfs WITH PARAMETER node  
            FOR each integer i FROM zero TO k MINUS one  
                SET edge TO the concatenation of node WITH the string form of i  
                IF edge NOT IN visited  
                    ADD edge TO visited  
                    CALL dfs WITH the substring of edge FROM position one TO the end  
                    APPEND the string form of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO the string consisting of the character zero REPEATED n MINUS one times  
        CALL dfs WITH start_node  
        
        SET result_sequence TO the concatenation of all elements in sequence IN REVERSE ORDER FOLLOWED BY start_node  
        RETURN result_sequence  
    END FUNCTION  
END CLASS
