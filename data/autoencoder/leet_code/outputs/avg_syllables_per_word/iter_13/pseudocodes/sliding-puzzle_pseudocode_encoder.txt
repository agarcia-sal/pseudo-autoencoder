CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO a tuple of tuples where each tuple corresponds to each row of board  
        SET target TO a tuple containing tuple one two three and tuple four five zero  
        SET neighbors TO a mapping where zero maps to a set containing one and three where one maps to a set containing zero two and four where two maps to a set containing one and five where three maps to a set containing zero and four where four maps to a set containing one three and five where five maps to a set containing two and four  
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position i of start at position j EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a deque initialized with a tuple containing start zero_pos and zero  
        SET visited TO a set initialized with start  
        WHILE queue is not empty  
            REMOVE the element from the left of queue AND ASSIGN TO state zero_pos moves  
            IF state EQUALS target  
                RETURN moves  
            END IF  
            FOR each neighbor IN the set mapped from zero_pos in neighbors  
                SET new_state TO a list where each element is a list converted from each tuple in state  
                SET i TO zero_pos DIVIDED BY three discarding remainder  
                SET j TO zero_pos MODULO three  
                SET ni TO neighbor DIVIDED BY three discarding remainder  
                SET nj TO neighbor MODULO three  
                SWAP the element at position i of new_state at position j WITH the element at position ni of new_state at position nj  
                SET new_tuple_state TO a tuple of tuples where each tuple corresponds to each row of new_state  
                IF new_tuple_state NOT IN visited  
                    ADD new_tuple_state TO visited  
                    APPEND a tuple containing new_tuple_state neighbor moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
