CLASS Solution  
    FUNCTION kthSmallest(matrix AS List of List of integer, k AS integer) RETURNS integer  
        FUNCTION countLessEqual(mid AS integer) RETURNS integer  
            SET count TO zero  
            SET row TO the LENGTH OF matrix MINUS one  
            SET col TO zero  
            WHILE row IS GREATER THAN OR EQUAL TO zero AND col IS LESS THAN THE LENGTH OF matrix  
                IF the element at position row of matrix AT position col OF matrix IS LESS THAN OR EQUAL TO mid  
                    INCREMENT count BY row PLUS one  
                    INCREMENT col BY one  
                ELSE  
                    DECREMENT row BY one  
                END IF  
            END WHILE  
            RETURN count  
        END FUNCTION  
        
        SET n TO the LENGTH OF matrix  
        SET low TO the element at position zero of matrix AT position zero OF matrix  
        SET high TO the element at position n MINUS one of matrix AT position n MINUS one OF matrix  
        
        WHILE low IS LESS THAN high  
            SET mid TO low PLUS high DIVIDED BY two USING INTEGER DIVISION  
            IF countLessEqual(mid) IS LESS THAN k  
                SET low TO mid PLUS one  
            ELSE  
                SET high TO mid  
            END IF  
        END WHILE  
        
        RETURN low  
    END FUNCTION  
END CLASS
