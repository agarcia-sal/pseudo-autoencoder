CLASS Solution  
    FUNCTION maxNumOfSubstrings(s)  
        SET first TO an empty mapping from characters to positions  
        SET last TO an empty mapping from characters to positions  
        SET index TO zero  
        FOR each character c IN s with index index  
            IF character c is not a key in first  
                SET first at key c TO index  
            END IF  
            SET last at key c TO index  
            INCREMENT index BY one  
        END FOR  
        
        SET intervals TO an empty list  
        FOR each character c IN keys of first  
            SET start TO value of first at key c  
            SET end TO value of last at key c  
            SET i TO start  
            WHILE i is less than or equal to end  
                IF value of first at key of character at position i of s is less than start  
                    BREAK the loop  
                END IF  
                SET end TO the greater value between end and value of last at key of character at position i of s  
                INCREMENT i BY one  
            END WHILE  
            IF loop did not break prematurely  
                APPEND the pair consisting of start and end TO intervals  
            END IF  
        END FOR  
        
        SORT intervals in ascending order according to the second element of each pair  
        
        SET result TO an empty list  
        SET prev_end TO negative one  
        FOR each pair start and end IN intervals  
            IF start is greater than prev_end  
                APPEND the substring from position start to position end of s TO result  
                SET prev_end TO end  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
