CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO the result of REPLACING every plus sign in side WITH the character number sign plus sign AND THEN REPLACING every minus sign in side WITH the character number sign minus sign AND THEN SPLITTING the resulting string BY the character number sign  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF the token IS empty  
                    CONTINUE to the next token  
                END IF  
                IF the token CONTAINS the character x  
                    IF the token IS equal TO the character x OR the token IS equal TO the plus sign character x  
                        INCREMENT x_count BY one  
                    ELSE IF the token IS equal TO the minus sign character x  
                        DECREMENT x_count BY one  
                    ELSE  
                        INCREMENT x_count BY the integer value CONVERTED from the token AFTER REMOVING the character x  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY the integer value CONVERTED from the token  
                END IF  
            END FOR  
            RETURN x_count AND num_sum  
        END FUNCTION

        SPLIT equation BY the equal sign INTO left_side AND right_side  
        SET left_x AND left_num TO the result of CALLING parse_side WITH left_side  
        SET right_x AND right_num TO the result of CALLING parse_side WITH right_side  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN Infinite solutions  
            ELSE  
                RETURN No solution  
            END IF  
        ELSE  
            RETURN the concatenation of the character x equal sign WITH the result of total_num DIVIDED BY total_x USING integer division  
        END IF  
    END FUNCTION  
END CLASS
