CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO the frequency count of elements in nums  
        SET end_with TO an empty frequency count of subsequence end numbers  
        
        FOR each num IN nums  
            IF element at position num of num_count EQUALS zero  
                CONTINUE to the next iteration  
            END IF  
            
            IF element at position numeric value of num MINUS one of end_with GREATER THAN zero  
                DECREMENT element at position numeric value of num MINUS one of end_with BY one  
                INCREMENT element at position num of end_with BY one  
            ELSE IF element at position numeric value of num PLUS one of num_count GREATER THAN zero AND element at position numeric value of num PLUS two of num_count GREATER THAN zero  
                DECREMENT element at position numeric value of num PLUS one of num_count BY one  
                DECREMENT element at position numeric value of num PLUS two of num_count BY one  
                INCREMENT element at position numeric value of num PLUS two of end_with BY one  
            ELSE  
                RETURN False  
            END IF  
            
            DECREMENT element at position num of num_count BY one  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
