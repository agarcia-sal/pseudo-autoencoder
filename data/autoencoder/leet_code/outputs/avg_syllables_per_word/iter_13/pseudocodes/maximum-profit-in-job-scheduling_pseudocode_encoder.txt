CLASS Solution  
    FUNCTION jobScheduling(startTime LIST OF INTEGER endTime LIST OF INTEGER profit LIST OF INTEGER) RETURNS INTEGER  
        SET jobs TO the list of tuples each containing corresponding elements from startTime endTime profit SORTED BY the second element of each tuple which is the end time  
        SET end_times TO an empty list  
        FOR each job IN jobs  
            APPEND the second element of job which is the end time TO end_times  
        END FOR  
        SET dp TO a list of zeros with LENGTH equal to the LENGTH OF jobs PLUS one  
        FOR index FROM one TO the LENGTH OF jobs  
            SET current_job_start TO the first element of the element at position index MINUS one of jobs which is the start time of the current job  
            SET j TO the POSITION FOUND BY performing a binary search for the rightmost insertion point of current_job_start IN end_times  
            SET dp at position index TO the MAXIMUM OF dp at position index MINUS one and dp at position j PLUS the third element of the element at position index MINUS one of jobs which is the profit of the current job  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
