CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO a list containing the characters representing the number n converted to text  
        SET length TO the NUMBER OF elements in digits  
        SET marker TO length  
        FOR index FROM length MINUS one TO one IN DECREMENTS OF one  
            IF the element at position index of digits is LESS THAN the element at position index MINUS one of digits  
                SET the element at position index MINUS one of digits TO the text representation of the NUMBER obtained by converting the element at position index MINUS one of digits to a number MINUS one  
                SET marker TO index  
            END IF  
        END FOR  
        FOR index FROM marker TO length MINUS one  
            SET the element at position index of digits TO the text representing the digit nine  
        END FOR  
        RETURN the NUMBER obtained by concatenating all elements of digits as text  
    END FUNCTION  
END CLASS
