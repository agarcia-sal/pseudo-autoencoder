CLASS Solution  
    FUNCTION rotate(matrix)  
        SET n TO the LENGTH OF matrix  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET temp TO the element at position j of the element at position i of matrix  
                SET the element at position j of the element at position i of matrix TO the element at position i of the element at position j of matrix  
                SET the element at position i of the element at position j of matrix TO temp  
            END FOR  
        END FOR  
        FOR i FROM zero TO n MINUS one  
            SET the element at position i of matrix TO the REVERSAL OF the element at position i of matrix  
        END FOR  
    END FUNCTION  
END CLASS
