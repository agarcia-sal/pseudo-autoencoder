CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one DOWN TO zero  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET alphabetical_position TO the alphabetical position of character at position index of s MINUS the alphabetical position of character a PLUS element at position index of shifts  
            SET wrapped_position TO alphabetical_position MODULO twenty six  
            SET new_character TO the character corresponding to wrapped_position PLUS the alphabetical position of character a  
            APPEND new_character TO result  
        END FOR  
        RETURN the concatenation of all characters in result  
    END FUNCTION  
END CLASS
