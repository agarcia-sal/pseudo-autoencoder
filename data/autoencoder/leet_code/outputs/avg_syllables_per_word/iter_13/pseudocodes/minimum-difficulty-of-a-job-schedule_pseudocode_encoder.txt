CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty list OF integer, d integer)  
        SET n TO the LENGTH OF jobDifficulty  
        
        IF n LESS THAN d THEN  
            RETURN negative one  
        END IF  
        
        FUNCTION dp(i integer, d integer)  
            IF d EQUALS one THEN  
                RETURN the MAXIMUM value AMONG the elements FROM position i TO the END OF jobDifficulty  
            END IF  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR j FROM i TO n MINUS d  
                SET current_max TO the GREATER OF current_max AND element at position j of jobDifficulty  
                SET min_difficulty TO the LESSER OF min_difficulty AND current_max PLUS CALL dp WITH arguments j PLUS one AND d MINUS one  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN CALL dp WITH arguments zero AND d  
    END FUNCTION  
END CLASS
