CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO an empty dictionary that returns an empty list for missing keys  
        FOR each path IN paths  
            SET parts TO the list obtained by splitting path by space  
            SET directory TO the element at position zero of parts  
            FOR each file_info IN elements of parts from position one TO the end  
                SET file_name TO the substring of file_info from the start TO the position before the first opening parenthesis  
                SET content TO the substring of file_info from the position after the first opening parenthesis TO the position before the last character of file_info  
                APPEND the concatenation of directory PLUS a slash PLUS file_name TO the list at key content in content_to_paths  
            END FOR  
        END FOR  
        SET result TO an empty list  
        FOR each paths_list IN the values of content_to_paths  
            IF the length of paths_list IS GREATER THAN one  
                APPEND paths_list TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
