CLASS Solution  
    FUNCTION kMirror(k, n)  
        FUNCTION is_palindrome(s)  
            RETURN s EQUALS the REVERSE OF s  
        END FUNCTION  
        
        FUNCTION to_base_k(num, base)  
            IF num EQUALS zero  
                RETURN the STRING REPRESENTATION OF zero  
            END IF  
            SET digits TO an EMPTY LIST  
            WHILE num GREATER THAN zero  
                APPEND the REMAINDER OF num DIVIDED BY base TO digits  
                SET num TO the INTEGER DIVISION OF num BY base  
            END WHILE  
            RETURN the CONCATENATION OF the STRING REPRESENTATIONS OF the ELEMENTS IN digits IN REVERSE ORDER  
        END FUNCTION  
        
        FUNCTION generate_palindromes(length)  
            IF length EQUALS one  
                FOR each integer i FROM one TO nine  
                    YIELD the STRING REPRESENTATION OF i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO THE POWER OF the INTEGER DIVISION OF length MINUS one BY two  
                SET end TO ten RAISED TO THE POWER OF the INTEGER DIVISION OF length PLUS one BY two  
                FOR each integer half FROM start TO end MINUS one  
                    SET half_str TO the STRING REPRESENTATION OF half  
                    IF length MODULO two EQUALS zero  
                        SET full TO half_str CONCATENATED WITH the REVERSE OF half_str  
                    ELSE  
                        SET full TO half_str CONCATENATED WITH the REVERSE OF the SUBSTRING OF half_str FROM the SECOND LAST POSITION TO the FIRST POSITION IN REVERSE ORDER  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO an EMPTY LIST  
        SET length TO one  
        WHILE the LENGTH OF k_mirror_numbers LESS THAN n  
            FOR each palindrome pal IN generate_palindromes(length)  
                SET num TO the INTEGER CONVERSION OF pal  
                IF is_palindrome(to_base_k(num, k))  
                    APPEND num TO k_mirror_numbers  
                    IF the LENGTH OF k_mirror_numbers EQUALS n  
                        BREAK the FOR LOOP  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN the SUM OF all ELEMENTS IN k_mirror_numbers  
    END FUNCTION  
END CLASS
