CLASS Solution  
    FUNCTION findRightInterval WITH PARAMETERS intervals  
        SET start_times TO sorted list of pairs of element at position zero of each interval AND the index of that interval in intervals  
        SET result TO empty list  
        FOR each pair underscore AND end_element IN intervals  
            SET idx TO position found by binary search for the smallest element in start_times where the first element of the pair is greater than or equal to end_element  
            IF idx is less than the length of start_times  
                APPEND element at position one of element at position idx of start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
