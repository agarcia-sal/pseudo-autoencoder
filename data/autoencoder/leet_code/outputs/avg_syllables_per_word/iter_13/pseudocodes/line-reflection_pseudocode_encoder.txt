CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO a collection of unique point pairs converted from points  
        SET minimum_x_coordinate TO the smallest first element of each point in unique_points  
        SET maximum_x_coordinate TO the largest first element of each point in unique_points  
        SET line_of_reflection TO the sum of minimum_x_coordinate PLUS maximum_x_coordinate DIVIDED BY two  
        FOR each pair of values x_coordinate and y_coordinate IN unique_points  
            SET reflected_x_coordinate TO two MULTIPLIED BY line_of_reflection MINUS x_coordinate  
            IF the pair consisting of reflected_x_coordinate and y_coordinate IS NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
