CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero next EQUAL TO none  
        SET self val TO val  
        SET self next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values of type list  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO new ListNode initialized WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values from position one TO the end  
        SET node TO new ListNode initialized WITH val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the comparison WHETHER the val attribute of p1 EQUALS the val attribute of p2 AND the result of is_same_list WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween WITH PARAMETERS self head of optional ListNode left of type integer right of type integer  
        IF head IS none OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO new ListNode initialized WITH zero  
        SET the next element of dummy TO head  
        SET prev TO dummy  
        FOR index FROM one TO left MINUS one  
            SET prev TO the next element of prev  
        END FOR  
        SET current TO the next element of prev  
        FOR index FROM one TO right MINUS left  
            SET next_node TO the next element of current  
            SET the next element of current TO the next element of next_node  
            SET the next element of next_node TO the next element of prev  
            SET the next element of prev TO next_node  
        END FOR  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
