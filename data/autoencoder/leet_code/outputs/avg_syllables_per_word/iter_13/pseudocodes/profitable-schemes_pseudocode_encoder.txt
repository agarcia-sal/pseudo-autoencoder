CLASS Solution  
    FUNCTION profitableSchemes WITH PARAMETERS n INTEGER minProfit INTEGER group LIST OF INTEGER profit LIST OF INTEGER  
        SET MODULO_VALUE TO one billion PLUS seven  
        INITIALIZE dp AS a two dimensional list of zeros with dimensions n PLUS one BY minProfit PLUS one  
        FOR each index i FROM zero TO n  
            SET element at position zero of dp at position i TO one  
        END FOR  
        FOR each pair p profit AND g group zipped together  
            FOR each index i FROM n DOWN TO g  
                FOR each index j FROM minProfit DOWN TO zero  
                    SET element at position j of dp at position i TO element at position j of dp at position i PLUS element at position maximum between zero AND j MINUS p of dp at position i MINUS g  
                    SET element at position j of dp at position i TO element at position j of dp at position i MODULO MODULO_VALUE  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position minProfit of dp at position n  
    END FUNCTION  
END CLASS
