CLASS Solution  
    FUNCTION majorityElement(nums)  
        SET n1 TO zero  
        SET n2 TO zero  
        SET m1 TO zero  
        SET m2 TO one  
        FOR each m IN nums  
            IF m EQUALS m1  
                INCREMENT n1 BY one  
            ELSE IF m EQUALS m2  
                INCREMENT n2 BY one  
            ELSE IF n1 EQUALS zero  
                SET m1 TO m  
                SET n1 TO one  
            ELSE IF n2 EQUALS zero  
                SET m2 TO m  
                SET n2 TO one  
            ELSE  
                DECREMENT n1 BY one  
                DECREMENT n2 BY one  
            END IF  
        END FOR  
        RETURN the LIST OF all m FROM the LIST CONTAINING m1 AND m2 WHERE THE COUNT OF m IN nums IS GREATER THAN the LENGTH OF nums DIVIDED BY three  
    END FUNCTION  
END CLASS
