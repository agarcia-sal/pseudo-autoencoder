CLASS Solution  
    FUNCTION minSkips(dist AS List OF integer, speed AS integer, hoursBefore AS integer) RETURNS integer  
        SET n TO the LENGTH OF dist  
        SET dp TO a new two dimensional list with n rows and n plus one columns filled with positive infinity  
        SET total_dist TO the SUM OF all elements in dist  
        
        IF total_dist DIVIDED BY speed IS GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  
        
        SET element at position zero and zero in dp TO element at position zero of dist  
        FOR j FROM one TO n minus one  
            SET element at position zero and j in dp TO element at position zero of dist  
        END FOR  
        
        FOR i FROM one TO n minus one  
            FOR j FROM zero TO i  
                IF j IS LESS THAN i THEN  
                    SET element at position i and j in dp TO the CEILING OF the element at position i minus one and j in dp DIVIDED BY speed MULTIPLIED BY speed PLUS element at position i in dist  
                END IF  
                IF j IS GREATER THAN zero THEN  
                    SET element at position i and j in dp TO the MINIMUM OF element at position i and j in dp AND element at position i minus one and j minus one in dp PLUS element at position i in dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n minus one  
            IF element at position n minus one and j in dp IS LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
