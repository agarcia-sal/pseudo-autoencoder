CLASS Solution  
    FUNCTION getPermutation(n, k)  
        SET numbers TO list of integers from one TO n  
        SET k TO k MINUS one  
        SET result TO empty list  
        FOR i FROM n TO one DECREMENT BY one  
            SET fact TO factorial of i MINUS one  
            SET index TO integer division of k BY fact  
            APPEND string representation of element at position index of numbers TO result  
            REMOVE element at position index FROM numbers  
            SET k TO remainder of k DIVIDED BY fact  
        END FOR  
        SET joined_result TO concatenation of all elements of result into a single string  
        RETURN joined_result  
    END FUNCTION  
END CLASS
