CLASS Solution  
    FUNCTION frogPosition(n, edges, t, target)  
        IF n EQUALS one  
            RETURN one point zero  
        END IF  
        
        SET graph TO result of calling function defaultdict with list as parameter  
        FOR each pair of u AND v IN edges  
            APPEND v TO list at key u of graph  
            APPEND u TO list at key v of graph  
        END FOR  
        
        SET queue TO result of calling function deque with a list containing a tuple of one, one point zero, and zero as element  
        SET visited TO a set containing one as element  
        
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            REMOVE the element at the front of queue AND ASSIGN the first value TO current_node  
            ASSIGN the second value TO current_probability  
            ASSIGN the third value TO current_time  
            
            IF current_time EQUALS t OR current_node NOT EQUALS one AND the LENGTH OF list at key current_node of graph EQUALS one  
                IF current_node EQUALS target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE to the next iteration of WHILE loop  
                END IF  
            END IF  
            
            SET possible_moves TO the LENGTH OF list at key current_node of graph MINUS one IF current_node NOT EQUALS one OTHERWISE zero  
            
            FOR each neighbor IN list at key current_node of graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND a tuple of neighbor, current_probability DIVIDED BY possible_moves, AND current_time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN zero point zero  
    END FUNCTION  
END CLASS
