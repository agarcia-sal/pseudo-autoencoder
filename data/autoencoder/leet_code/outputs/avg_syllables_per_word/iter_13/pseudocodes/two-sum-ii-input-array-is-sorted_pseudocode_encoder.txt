CLASS Solution  
    FUNCTION twoSum(numbers, target)  
        SET left TO zero  
        SET right TO the LENGTH OF numbers MINUS one  
        WHILE left IS LESS THAN right  
            SET current_sum TO element at position left of numbers PLUS element at position right of numbers  
            IF current_sum EQUALS target  
                RETURN value of left PLUS one and value of right PLUS one as a list  
            ELSE IF current_sum IS LESS THAN target  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
