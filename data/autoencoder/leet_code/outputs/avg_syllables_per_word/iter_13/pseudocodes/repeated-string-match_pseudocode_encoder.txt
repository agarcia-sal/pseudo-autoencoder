CLASS Solution  
    FUNCTION repeatedStringMatch(a, b)  
        SET minimum repetitions TO the negative of the integer division of the length of string b BY the length of string a with the operation treated as ceiling  
        IF string b IS A SUBSTRING OF string a REPEATED minimum repetitions TIMES  
            RETURN minimum repetitions  
        END IF  
        IF string b IS A SUBSTRING OF string a REPEATED minimum repetitions PLUS one TIME  
            RETURN minimum repetitions PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
