CLASS Solution  
    FUNCTION exclusiveTime(n, logs)  
        SET exclusive_time TO list of zero values of length n  
        SET stack TO empty list  
        
        FOR each log IN logs  
            SPLIT log INTO function_id part action part timestamp by colon  
            CONVERT function_id part TO integer and SET TO function_id  
            CONVERT timestamp part TO integer and SET TO timestamp  
            
            IF action EQUALS the word start  
                IF stack is not empty  
                    SET prev_function_id and prev_timestamp TO the element at last position of stack  
                    INCREMENT element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND the pair of function_id and timestamp TO stack  
            ELSE  
                SET prev_function_id and prev_timestamp TO element popped from the end of stack  
                INCREMENT element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp PLUS one  
                IF stack is not empty  
                    SET element at last position of stack TO pair of the first element at last position of stack and timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
