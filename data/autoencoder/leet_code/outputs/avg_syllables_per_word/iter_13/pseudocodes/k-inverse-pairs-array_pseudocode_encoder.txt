CLASS Solution  
    FUNCTION kInversePairs(n AS integer, k AS integer) RETURNS integer  
        SET MODULO TO one billion PLUS seven  
        SET dp TO a two dimensional list of zeros with the count of rows equal to n PLUS one and the count of columns equal to k PLUS one  
        SET element at position zero of the first list in dp TO one  
        
        FOR each index i FROM one TO n  
            SET prefix_sum TO zero  
            FOR each index j FROM zero TO k  
                SET prefix_sum TO the remainder of the sum of prefix_sum PLUS element at position j of the list at position i MINUS one of dp DIVIDED BY MODULO  
                IF j IS GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO the remainder of prefix_sum MINUS element at position j MINUS i of the list at position i MINUS one of dp PLUS MODULO DIVIDED BY MODULO  
                END IF  
                SET element at position j of the list at position i of dp TO prefix_sum  
            END FOR  
        END FOR  
        
        RETURN element at position k of the list at position n of dp  
    END FUNCTION  
END CLASS
