CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                RETURN one IF current_sum EQUALS target ELSE zero  
            END IF  
            IF the PAIR OF index AND current_sum EXISTS IN memo  
                RETURN the VALUE ASSOCIATED WITH the PAIR OF index AND current_sum IN memo  
            END IF  
            SET positive TO the RESULT OF CALLING dfs WITH index INCREMENTED BY one AND current_sum INCREMENTED BY element at position index of nums AND memo  
            SET negative TO the RESULT OF CALLING dfs WITH index INCREMENTED BY one AND current_sum DECREMENTED BY element at position index of nums AND memo  
            SET the VALUE ASSOCIATED WITH the PAIR OF index AND current_sum IN memo TO positive PLUS negative  
            RETURN the VALUE ASSOCIATED WITH the PAIR OF index AND current_sum IN memo  
        END FUNCTION  
        RETURN the RESULT OF CALLING dfs WITH zero FOR index AND zero FOR current_sum AND an EMPTY DICTIONARY FOR memo  
    END FUNCTION  
END CLASS
