CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1 STRING, str2 STRING) RETURNS STRING  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        
        SET dp TO a two dimensional list with m PLUS one ROWS and n PLUS one COLUMNS initialized TO zero
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2  
                    SET element at position i and j of dp TO element at position i MINUS one and j MINUS one of dp PLUS one  
                ELSE  
                    SET element at position i and j of dp TO the GREATER OF element at position i MINUS one and j of dp AND element at position i and j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        SET i TO m  
        SET j TO n  
        SET scs TO an empty list  
        
        WHILE i IS GREATER THAN zero AND j IS GREATER THAN zero  
            IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF the element at position i MINUS one and j of dp IS GREATER THAN the element at position i and j MINUS one of dp  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND the element at position j MINUS one of str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i IS GREATER THAN zero  
            APPEND the element at position i MINUS one of str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        
        WHILE j IS GREATER THAN zero  
            APPEND the element at position j MINUS one of str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        
        REVERSE the order of elements in scs  
        
        SET result_string TO the concatenation of all elements in scs  
        RETURN result_string  
    END FUNCTION  
END CLASS
