CLASS Solution  
    FUNCTION minmaxGasDist WITH PARAMETERS stations AND k  
        FUNCTION possible WITH PARAMETER max_dist  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                SET distance_difference TO element at position index of stations MINUS element at position index MINUS one of stations  
                SET division_result TO distance_difference DIVIDED BY max_dist  
                SET integer_result TO the INTEGER PART of division_result  
                INCREMENT additional_stations BY integer_result  
            END FOR  
            IF additional_stations IS LESS THAN OR EQUAL TO k THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE the VALUE OF right MINUS the VALUE OF left IS GREATER THAN zero POINT zero zero zero zero zero one  
            SET mid TO the VALUE OF left PLUS the VALUE OF right DIVIDED BY two  
            IF the RESULT OF calling possible WITH mid IS true THEN  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
