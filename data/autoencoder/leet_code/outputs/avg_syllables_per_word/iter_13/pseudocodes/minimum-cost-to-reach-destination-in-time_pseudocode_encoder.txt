CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO a new list containing n empty lists  
        FOR each x y time IN edges  
            APPEND the pair consisting of y and time TO the element at position x of graph  
            APPEND the pair consisting of x and time TO the element at position y of graph  
        END FOR  
        SET pq TO a new list containing the triple consisting of the element at position zero of passingFees, zero, and zero  
        SET visited_time TO a new list containing n elements each being positive infinity  
        SET the element at position zero of visited_time TO zero  
        WHILE pq is not empty  
            REMOVE the element with smallest current_cost from pq and ASSIGN its components TO current_cost current_city current_time  
            IF current_city EQUALS n MINUS one THEN  
                RETURN current_cost  
            END IF  
            FOR each neighbor travel_time IN the element at position current_city of graph  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS the element at position neighbor of passingFees  
                IF new_time IS LESS THAN OR EQUAL TO maxTime AND (new_time IS LESS THAN the element at position neighbor of visited_time OR new_cost IS LESS THAN current_cost) THEN  
                    SET the element at position neighbor of visited_time TO new_time  
                    ADD the triple consisting of new_cost neighbor new_time TO pq maintaining its priority order by current_cost  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
