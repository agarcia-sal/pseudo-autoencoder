CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero left EQUALS none right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with parameter element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS not empty  
        SET node TO element removed from the left side of queue  
        IF i IS less than the length of values AND element at position i of values IS NOT none THEN  
            SET node.left TO new TreeNode with parameter element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS less than the length of values AND element at position i of values IS NOT none THEN  
            SET node.right TO new TreeNode with parameter element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF the value of p IS NOT EQUAL TO the value of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left q.left AND is_same_tree WITH PARAMETERS p.right q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum WITH PARAMETERS root AS optional TreeNode  
        FUNCTION helper WITH PARAMETER node  
            IF node IS none THEN  
                RETURN negative infinity AND negative infinity  
            END IF  
            SET left_max AND left_gain TO helper WITH PARAMETER node.left  
            SET right_max AND right_gain TO helper WITH PARAMETER node.right  
            SET current_max TO the value of node PLUS the GREATER BETWEEN left_gain AND zero PLUS the GREATER BETWEEN right_gain AND zero  
            SET current_gain TO the value of node PLUS the GREATER AMONG left_gain AND right_gain AND zero  
            SET global_max TO the GREATER AMONG left_max AND right_max AND current_max  
            RETURN global_max AND current_gain  
        END FUNCTION  
        RETURN the first element of helper WITH PARAMETER root  
    END FUNCTION  
END CLASS
