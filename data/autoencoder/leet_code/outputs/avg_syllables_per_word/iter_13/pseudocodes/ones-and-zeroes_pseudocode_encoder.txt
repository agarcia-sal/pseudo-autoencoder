CLASS Solution  
    FUNCTION findMaxForm(strs LIST OF STRING, m INTEGER, n INTEGER) RETURNS INTEGER  
        SET dp TO two dimensional list of dimensions m PLUS one BY n PLUS one FILLED WITH zero  
        FOR each s IN strs  
            SET count_0 TO number of zeros IN s  
            SET count_1 TO number of ones IN s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET value_at_i_j TO element at position i of dp AT position j  
                    SET value_at_i_minus_count0_j_minus_count1 TO element at position i MINUS count_0 of dp AT position j MINUS count_1  
                    SET element at position i of dp AT position j TO the GREATER OF value_at_i_j AND value_at_i_minus_count0_j_minus_count1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position m of dp AT position n  
    END FUNCTION  
END CLASS
