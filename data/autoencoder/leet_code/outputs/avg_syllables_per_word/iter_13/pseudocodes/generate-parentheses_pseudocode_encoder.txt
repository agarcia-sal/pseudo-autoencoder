CLASS Solution  
    FUNCTION generateParenthesis WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETER s DEFAULT empty string WITH PARAMETER left DEFAULT zero WITH PARAMETER right DEFAULT zero  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s CONCATENATED WITH opening parenthesis WITH left INCREMENTED BY one WITH right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s CONCATENATED WITH closing parenthesis WITH left WITH right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH default parameters  
        RETURN result  
    END FUNCTION  
END CLASS
