CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO empty mapping  
        
        FUNCTION can_win_helper(state)  
            IF the element at position state of memo EXISTS  
                RETURN the element at position state of memo  
            END IF  
            
            FOR each index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring from position index TO index PLUS one of state EQUALS two plus signs  
                    SET new_state TO the substring from position zero TO index of state CONCATENATED WITH two minus signs CONCATENATED WITH the substring from position index PLUS two TO the LENGTH OF state MINUS one of state  
                    IF it IS NOT the case that can_win_helper APPLIED TO new_state HOLDS  
                        SET the element at position state of memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the element at position state of memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN can_win_helper APPLIED TO currentState  
    END FUNCTION  
END CLASS
