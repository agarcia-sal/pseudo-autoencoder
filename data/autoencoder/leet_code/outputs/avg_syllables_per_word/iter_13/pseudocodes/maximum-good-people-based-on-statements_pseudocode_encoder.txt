CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        
        FOR each integer i FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            SET good TO a list OF boolean False of size n  
            SET bad TO a list OF boolean False of size n  
            
            FOR each integer j FROM zero TO n MINUS one  
                IF the BITWISE AND of i AND two RAISED TO THE POWER OF j IS NOT EQUAL TO zero THEN  
                    SET the element at position j of good TO True  
                ELSE  
                    SET the element at position j of bad TO True  
                END IF  
            END FOR  
            
            SET valid TO True  
            FOR each integer j FROM zero TO n MINUS one  
                IF the element at position j of good IS True THEN  
                    FOR each integer k FROM zero TO n MINUS one  
                        IF the element at position j of the element at position k of statements EQUALS zero AND the element at position k of good IS True OR the element at position j of the element at position k of statements EQUALS one AND the element at position k of bad IS True THEN  
                            SET valid TO False  
                            BREAK from the inner loop  
                        END IF  
                    END FOR  
                    IF valid IS False THEN  
                        BREAK from the outer loop  
                    END IF  
                END IF  
            END FOR  
            
            IF valid IS True THEN  
                SET max_good TO the GREATER OF max_good AND the SUM OF True elements IN good  
            END IF  
        END FOR  
        
        RETURN max_good  
    END FUNCTION  
END CLASS
