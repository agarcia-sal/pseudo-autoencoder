CLASS Solution  
    FUNCTION leastOpsExpressTarget(x integer, target integer) RETURNS integer  
        DEFINE FUNCTION dp(target_value integer) RETURNS integer  
            IF target_value EQUALS zero THEN  
                RETURN negative one  
            END IF  
            IF target_value LESS THAN x THEN  
                SET first_option TO two MULTIPLIED BY target_value MINUS one  
                SET second_option TO two MULTIPLIED BY x MINUS target_value  
                RETURN the MINIMUM OF first_option AND second_option  
            END IF  
            SET power TO one  
            WHILE x RAISED TO THE POWER OF power LESS THAN OR EQUAL TO target_value HOLDS  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET result TO dp OF target_value MINUS x RAISED TO THE POWER OF power PLUS power  
            IF x RAISED TO THE POWER OF power PLUS one MINUS target_value LESS THAN target_value THEN  
                SET alternative TO dp OF x RAISED TO THE POWER OF power PLUS one MINUS target_value PLUS power PLUS one  
                SET result TO the MINIMUM OF result AND alternative  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN dp OF target  
    END FUNCTION  
END CLASS
