CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of integers FROM zero TO n MINUS one  
        SET self.rank TO list containing the integer one repeated n times  
    END FUNCTION  
  
    FUNCTION find(self, u)  
        IF element at position u of self.parent DOES NOT EQUAL u  
            SET element at position u of self.parent TO the result of calling self.find with the element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  
  
    FUNCTION union(self, u, v)  
        SET rootU TO the result of calling self.find with u  
        SET rootV TO the result of calling self.find with v  
        
        IF rootU DOES NOT EQUAL rootV  
            IF element at position rootU of self.rank IS GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank IS LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            RETURN boolean true  
        END IF  
        RETURN boolean false  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)  
        FOR each index i AND element edge IN edges with index  
            APPEND i TO edge  
        END FOR  
  
        SORT edges IN ascending order BASED ON element at position two of each edge  
  
        FUNCTION mst(exclude, include)  
            ASSIGN pointer uf TO a new UnionFind object with parameter n  
            SET weight TO zero  
            
            IF include IS NOT equal TO none  
                CALL uf.union WITH first element of include AND second element of include  
                INCREMENT weight BY third element of include  
            END IF  
            
            FOR each u, v, w, idx IN edges  
                IF idx EQUALS exclude  
                    CONTINUE to next iteration  
                END IF  
                IF calling uf.union with u AND v RETURNS true  
                    INCREMENT weight BY w  
                END IF  
            END FOR  
            
            SET root TO the result of calling uf.find with zero  
            FOR each integer i FROM zero TO n MINUS one  
                IF the result of calling uf.find with i DOES NOT EQUAL root  
                    RETURN positive infinity  
                END IF  
            END FOR  
            
            RETURN weight  
        END FUNCTION  
  
        SET mst_weight TO the result of calling mst without arguments  
        
        SET critical TO an empty list  
        SET pseudo_critical TO an empty list  
        
        FOR each u, v, w, idx IN edges  
            IF the result of calling mst with exclude equal to idx IS GREATER THAN mst_weight  
                APPEND idx TO critical  
            ELSE IF the result of calling mst with include equal to the tuple u, v, w IS EQUAL TO mst_weight  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR  
        
        RETURN list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
