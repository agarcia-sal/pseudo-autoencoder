CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val EQUALS element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from the left side of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode with val EQUALS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode with val EQUALS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(element at attribute left of p, element at attribute left of q) AND is_same_tree(element at attribute right of p, element at attribute right of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION boundaryOfBinaryTree(self, root)  
        IF root IS None  
            RETURN an empty list  
        END IF  
        FUNCTION is_leaf(node)  
            RETURN node.left IS None AND node.right IS None  
        END FUNCTION  
        SET left_boundary TO an empty list  
        SET current TO element at attribute left of root  
        WHILE current IS NOT None  
            IF NOT is_leaf(current)  
                APPEND element at attribute val of current TO left_boundary  
            END IF  
            IF element at attribute left of current IS NOT None  
                SET current TO element at attribute left of current  
            ELSE  
                SET current TO element at attribute right of current  
            END IF  
        END WHILE  
        SET right_boundary TO an empty list  
        SET current TO element at attribute right of root  
        WHILE current IS NOT None  
            IF NOT is_leaf(current)  
                APPEND element at attribute val of current TO right_boundary  
            END IF  
            IF element at attribute right of current IS NOT None  
                SET current TO element at attribute right of current  
            ELSE  
                SET current TO element at attribute left of current  
            END IF  
        END WHILE  
        SET leaves TO an empty list  
        FUNCTION collect_leaves(node)  
            IF node IS None  
                RETURN  
            END IF  
            IF is_leaf(node)  
                APPEND element at attribute val of node TO leaves  
            END IF  
            CALL collect_leaves with argument element at attribute left of node  
            CALL collect_leaves with argument element at attribute right of node  
        END FUNCTION  
        CALL collect_leaves with argument root  
        IF is_leaf(root)  
            SET leaves TO an empty list  
        END IF  
        RETURN list consisting of element at attribute val of root FOLLOWED BY all elements of left_boundary FOLLOWED BY all elements of leaves FOLLOWED BY all elements of right_boundary IN REVERSE ORDER  
    END FUNCTION  
END CLASS
