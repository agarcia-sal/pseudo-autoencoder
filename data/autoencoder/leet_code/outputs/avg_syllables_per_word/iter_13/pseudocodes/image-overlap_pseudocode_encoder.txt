CLASS Solution  
    FUNCTION largestOverlap(img1, img2)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF element at position j of element at position i of img1 EQUALS one  
                    APPEND the PAIR OF i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF element at position j of element at position i of img2 EQUALS one  
                    APPEND the PAIR OF i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        
        SET overlap_count TO a default dictionary with integer default zero  
        
        FOR each PAIR x1 AND y1 IN points1  
            FOR each PAIR x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value associated with the PAIR OF dx AND dy IN overlap_count BY one  
            END FOR  
        END FOR  
        
        SET maximum_overlap TO zero  
        IF overlap_count has any values  
            SET maximum_overlap TO the maximum value among all values in overlap_count  
        END IF  
        RETURN maximum_overlap  
    END FUNCTION  
END CLASS
