CLASS Solution  
    FUNCTION canConvert(str1 parameter representing string, str2 parameter representing string) RETURNS boolean  
        IF str1 EQUALS str2 THEN  
            RETURN True  
        END IF  
        
        SET conversion_map TO empty mapping from characters to characters  
        
        FOR each character named char1 AND each character named char2 TAKEN TOGETHER FROM str1 AND str2  
            IF the mapping conversion_map CONTAINS key char1 THEN  
                IF the value at key char1 in conversion_map DOES NOT EQUAL char2 THEN  
                    RETURN False  
                END IF  
            ELSE  
                SET the value at key char1 in conversion_map TO char2  
            END IF  
        END FOR  
        
        IF the count of unique characters in str2 EQUALS twenty six THEN  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
