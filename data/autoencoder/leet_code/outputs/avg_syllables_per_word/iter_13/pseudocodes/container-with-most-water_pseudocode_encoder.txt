CLASS Solution  
    FUNCTION maxArea WITH parameter height  
        SET left TO zero  
        SET right TO the LENGTH OF height MINUS one  
        SET max_area TO zero  
        
        WHILE left IS LESS THAN right  
            SET current_area TO the SMALLEST OF element at position left of height AND element at position right of height MULTIPLIED BY right MINUS left  
            SET max_area TO the LARGEST OF max_area AND current_area  
            
            IF element at position left of height IS LESS THAN element at position right of height  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
