CLASS Solution  
    FUNCTION superpalindromesInRange(left STRING, right STRING) RETURNS INTEGER  
        FUNCTION is_palindrome(num) RETURNS BOOLEAN  
            RETURN string conversion of num EQUALS string conversion of num REVERSED  
        END FUNCTION  
        
        FUNCTION generate_palindromes(limit) RETURNS LIST OF INTEGER  
            SET palindromes TO empty list  
            FOR i FROM one TO one hundred thousand MINUS one  
                SET s TO string conversion of i  
                APPEND integer conversion of concatenation of s WITH substring of s FROM second LAST TO first IN REVERSE ORDER TO palindromes  
                APPEND integer conversion of concatenation of s WITH s REVERSED TO palindromes  
            END FOR  
            RETURN list of elements p FROM palindromes WHERE p IS LESS THAN OR EQUAL TO limit  
        END FUNCTION  
        
        SET left TO integer conversion of left  
        SET right TO integer conversion of right  
        SET limit TO integer conversion of right RAISED TO THE POWER OF zero point five PLUS one  
        SET palindromes TO generate_palindromes(limit)  
        
        SET count TO zero  
        FOR each p IN palindromes  
            SET square TO p MULTIPLIED BY p  
            IF left IS LESS THAN OR EQUAL TO square AND square IS LESS THAN OR EQUAL TO right AND is_palindrome(square)  
                INCREMENT count BY one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
