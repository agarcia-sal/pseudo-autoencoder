CLASS Solution  
    FUNCTION findLUSlength WITH parameter strs AS list of strings  
        FUNCTION is_subsequence WITH parameters s AND t  
            SET iterator_of_t TO the iterator OVER t  
            RETURN whether for every character in s the character APPEARS IN iterator_of_t IN ORDER  
        END FUNCTION  

        SORT strs BY descending LENGTH OF each string AND THEN BY alphabetical order  

        FOR each index i ALONG WITH word IN strs  
            SET unique TO true  
            FOR each index j ALONG WITH other IN strs  
                IF i IS NOT EQUAL TO j AND is_subsequence WITH s AS word AND t AS other RETURNS true  
                    SET unique TO false  
                    BREAK inner loop  
                END IF  
            END FOR  
            IF unique IS true  
                RETURN LENGTH OF word  
            END IF  
        END FOR  

        RETURN negative one  
    END FUNCTION  
END CLASS
