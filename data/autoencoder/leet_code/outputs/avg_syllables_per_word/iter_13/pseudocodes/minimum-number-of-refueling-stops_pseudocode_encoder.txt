CLASS Solution  
    FUNCTION minRefuelStops WITH PARAMETERS target startFuel stations  
        CREATE an empty list named max_heap to represent the max-heap of fuel amounts from passed stations  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a list consisting of target and zero TO stations to act as the last station with zero fuel  
        
        FOR each position and fuel IN stations  
            WHILE max_heap is not empty AND current_fuel is LESS THAN position  
                REMOVE the largest fuel amount FROM max_heap and SET this value TO largest_fuel (note the original heap stores negative for max behavior)  
                INCREMENT current_fuel BY the negation of largest_fuel  
                INCREMENT stops BY one  
            END WHILE  
            
            IF current_fuel is LESS THAN position  
                RETURN negative one  
            END IF  
            
            PUSH negative fuel INTO max_heap to maintain max-heap property  
        END FOR  
        
        RETURN stops  
    END FUNCTION  
END CLASS
