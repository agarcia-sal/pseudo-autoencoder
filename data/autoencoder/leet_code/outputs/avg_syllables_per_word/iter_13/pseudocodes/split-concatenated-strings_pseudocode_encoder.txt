CLASS Solution  
    FUNCTION splitLoopedString WITH PARAMETER strs AS list of strings  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_string TO element at position index of strs REVERSED  
            IF reversed_string GREATER THAN element at position index of strs  
                SET element at position index of strs TO reversed_string  
            END IF  
        END FOR  
        
        SET doubled_strs TO concatenation of strs WITH strs  
        
        SET max_string TO empty string  
        
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each boolean is_reversed IN collection consisting of False AND True  
                IF is_reversed EQUALS True  
                    SET current TO element at position index of strs REVERSED  
                ELSE  
                    SET current TO element at position index of strs  
                END IF  
                
                SET remaining TO concatenation of elements from position index PLUS one TO index PLUS the LENGTH OF strs MINUS one of doubled_strs  
                
                FOR cut_position FROM zero TO the LENGTH OF current  
                    SET candidate TO concatenation of substring from cut_position TO the end of current WITH remaining WITH substring from the start TO cut_position OF current  
                    
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
