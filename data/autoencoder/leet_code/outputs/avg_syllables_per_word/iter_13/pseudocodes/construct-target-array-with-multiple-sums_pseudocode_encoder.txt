CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO empty list  
        FOR each num IN target  
            APPEND negative of num TO max_heap  
        END FOR  
        CALL heapify FUNCTION OF heapq MODULE ON max_heap  

        SET total_sum TO zero  
        FOR each num IN target  
            INCREMENT total_sum BY num  
        END FOR  

        WHILE element at position zero of max_heap IS LESS THAN negative one  
            SET largest TO negative of result of popping the smallest element from max_heap using heapq module  
            SET rest_sum TO total_sum MINUS largest  

            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum IS EQUAL TO zero  
                RETURN False  
            END IF  

            SET new_value TO largest MODULO rest_sum  
            IF new_value IS EQUAL TO zero AND rest_sum IS NOT EQUAL TO one  
                RETURN False  
            END IF  

            SET total_sum TO rest_sum PLUS new_value  
            PUSH negative of new_value INTO max_heap using heapq module  
        END WHILE  

        RETURN True  
    END FUNCTION  
END CLASS
