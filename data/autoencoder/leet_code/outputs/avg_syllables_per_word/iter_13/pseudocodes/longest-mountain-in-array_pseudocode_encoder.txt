CLASS Solution  
    FUNCTION longestMountain(arr)  
        SET n TO the LENGTH OF arr  
        SET longest TO zero  
        
        SET i TO one  
        WHILE i IS LESS THAN n MINUS one  
            IF the element at position i MINUS one of arr IS LESS THAN the element at position i of arr AND the element at position i of arr IS GREATER THAN the element at position i PLUS one of arr  
                SET left TO i MINUS one  
                SET right TO i PLUS one  
                
                WHILE left IS GREATER THAN zero AND the element at position left MINUS one of arr IS LESS THAN the element at position left of arr  
                    DECREMENT left BY one  
                END WHILE  
                
                WHILE right IS LESS THAN n MINUS one AND the element at position right of arr IS GREATER THAN the element at position right PLUS one of arr  
                    INCREMENT right BY one  
                END WHILE  
                
                SET longest TO the GREATER VALUE BETWEEN longest AND right MINUS left PLUS one  
                
                SET i TO right  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN longest  
    END FUNCTION  
END CLASS
