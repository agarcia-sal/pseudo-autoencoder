CLASS Solution  
    FUNCTION numPoints(darts, r)  
        FUNCTION distance(p1, p2)  
            RETURN the square root of the sum of the square of the difference between element at position zero of p1 and element at position zero of p2 AND the square of the difference between element at position one of p1 and element at position one of p2  
        END FUNCTION  
  
        FUNCTION circle_center(p1, p2, r)  
            SET d TO distance with arguments p1 AND p2  
            IF d GREATER THAN two TIMES r  
                RETURN None  
            END IF  
            SET a TO the square root of r MULTIPLIED BY r MINUS d DIVIDED BY two MULTIPLIED BY d DIVIDED BY two  
            SET h TO a list containing the element which is the sum of element at position zero of p1 AND element at position zero of p2 DIVIDED BY two AND the element which is the sum of element at position one of p1 AND element at position one of p2 DIVIDED BY two  
            SET dx TO the product of the difference between element at position one of p2 AND element at position one of p1 AND a DIVIDED BY d  
            SET dy TO the product of the difference between element at position zero of p2 AND element at position zero of p1 AND a DIVIDED BY d  
            RETURN a list containing a tuple of h at position zero MINUS dx AND h at position one PLUS dy AND a tuple of h at position zero PLUS dx AND h at position one MINUS dy  
        END FUNCTION  
  
        FUNCTION is_inside(circle, point, r)  
            RETURN distance with arguments circle AND point LESS THAN OR EQUAL TO r  
        END FUNCTION  
  
        SET n TO the length of darts  
        IF n EQUALS one  
            RETURN one  
        END IF  
  
        SET max_count TO one  
  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET centers TO circle_center with arguments element at position i of darts AND element at position j of darts AND r  
                IF centers IS NOT None  
                    FOR each center IN centers  
                        SET count TO zero  
                        FOR each dart IN darts  
                            IF is_inside with arguments center AND dart AND r  
                                INCREMENT count BY one  
                            END IF  
                        END FOR  
                        IF count GREATER THAN max_count  
                            SET max_count TO count  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
  
        RETURN max_count  
    END FUNCTION  
END CLASS
