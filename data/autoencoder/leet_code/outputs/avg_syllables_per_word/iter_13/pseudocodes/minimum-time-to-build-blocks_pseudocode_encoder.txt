CLASS Solution  
    FUNCTION minBuildTime(blocks OF TYPE List OF integers, split OF TYPE integer) RETURNS integer  
        APPLY heapify TO transform blocks INTO a min heap data structure  
        WHILE the LENGTH OF blocks IS GREATER THAN one  
            REMOVE the element with the smallest value FROM blocks AND ASSIGN TO first  
            REMOVE the element with the smallest value FROM blocks AGAIN AND ASSIGN TO second  
            SET combined_time TO split PLUS the value of second  
            INSERT combined_time INTO blocks maintaining heap property  
        END WHILE  
        RETURN the element at position zero of blocks as the minimum build time  
    END FUNCTION  
END CLASS
