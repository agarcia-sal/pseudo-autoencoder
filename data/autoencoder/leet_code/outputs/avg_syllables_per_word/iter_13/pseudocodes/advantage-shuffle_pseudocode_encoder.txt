CLASS Solution  
    FUNCTION advantageCount(nums1 AS List OF integers, nums2 AS List OF integers) RETURNS List OF integers  
        CALL sort TO nums1  
        SET nums2_with_indices TO the result of sorting the list consisting of pairs each containing value and original position of the element at that position of nums2 ordered from highest value to lowest value  
        SET result TO a new list of zeros with the length equal to the length of nums1  
        SET left TO zero  
        SET right TO the length of nums1 MINUS one  
        FOR each pair of value and index IN nums2_with_indices  
            IF the element at position right of nums1 is GREATER THAN value THEN  
                SET the element at position index of result TO the element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET the element at position index of result TO the element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
