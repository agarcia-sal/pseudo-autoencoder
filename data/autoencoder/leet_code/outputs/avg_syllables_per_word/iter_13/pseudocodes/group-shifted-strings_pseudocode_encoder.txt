CLASS Solution  
    FUNCTION groupStrings WITH strings  
        FUNCTION get_canonical_form WITH s  
            IF s EQUALS an empty string  
                RETURN an empty string  
            END IF  
            SET canonical_form TO an empty list  
            FOR each character c IN s  
                SET difference TO the Unicode code point of character c MINUS the Unicode code point of element at position zero of s  
                SET wrapped_difference TO difference MODULO twenty six  
                APPEND the string representation of wrapped_difference TO canonical_form  
            END FOR  
            SET joined_canonical_form TO the concatenation of elements of canonical_form SEPARATED BY a comma  
            RETURN joined_canonical_form  
        END FUNCTION  
        
        SET groups TO a new mapping from keys to a list of values  
        
        FOR each string s IN strings  
            SET canonical_form TO the result of calling get_canonical_form WITH s  
            APPEND s TO the list at key canonical_form in groups  
        END FOR  
        
        RETURN the list of all values in groups  
    END FUNCTION  
END CLASS
