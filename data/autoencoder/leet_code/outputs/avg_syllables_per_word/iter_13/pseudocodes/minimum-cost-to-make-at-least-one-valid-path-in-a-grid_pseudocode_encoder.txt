CLASS Solution  
    FUNCTION minCost(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO a list containing four elements: an ordered pair of zero and one, an ordered pair of zero and NEGATIVE one, an ordered pair of one and zero, an ordered pair of NEGATIVE one and zero  
        SET costs TO a two dimensional list of size m by n where each element is assigned to infinity  
        SET element at position zero of element at position zero of costs TO zero  
        
        SET pq TO a list containing a single tuple of zero zero zero representing the cost and coordinates  
        SET visited TO an empty set  
        
        WHILE pq is not empty  
            EXTRACT from pq the element with the smallest cost and ASSIGN to cost i j  
            
            IF the ordered pair of i and j is contained in visited  
                CONTINUE to next iteration  
            END IF  
            ADD the ordered pair of i and j to visited  
            
            IF i is EQUAL TO m MINUS one AND j is EQUAL TO n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each index k and ordered pair di dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni is GREATER THAN OR EQUAL TO zero AND ni is LESS THAN m AND nj is GREATER THAN OR EQUAL TO zero AND nj is LESS THAN n  
                    SET new_cost TO cost PLUS one IF k PLUS one is NOT EQUAL TO element at position i of grid at position j OTHERWISE zero  
                    IF new_cost is LESS THAN element at position ni of costs at position nj  
                        SET element at position ni of costs at position nj TO new_cost  
                        ADD the tuple new_cost ni nj TO pq maintaining pq as a priority queue based on cost  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
