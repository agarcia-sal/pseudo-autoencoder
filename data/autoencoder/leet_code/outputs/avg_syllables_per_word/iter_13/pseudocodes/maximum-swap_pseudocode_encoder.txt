CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO the list of characters formed by converting num to string  
        SET last TO an empty dictionary  
        FOR each index i AND digit d IN digits  
            SET the element at position integer value of d of last TO i  
        END FOR  
        FOR each index i AND digit d IN digits  
            FOR each digit candidate FROM nine DOWN TO the integer value of d PLUS one  
                IF the value at position candidate in last EXISTS AND IS GREATER THAN i  
                    SET temporary variable TO element at position i of digits  
                    SET element at position i of digits TO element at position value at position candidate of last of digits  
                    SET element at position value at position candidate of last of digits TO temporary variable  
                    RETURN the integer formed by concatenating all elements of digits  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
