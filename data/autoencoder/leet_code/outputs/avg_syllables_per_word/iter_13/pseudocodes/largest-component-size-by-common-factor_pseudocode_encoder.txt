CLASS UnionFind  
    FUNCTION __init__(size)  
        SET self.parent TO a list of integers from zero TO size MINUS one  
        SET self.rank TO a list of ones with length size  
        SET self.size TO a list of ones with length size  
    END FUNCTION  

    FUNCTION find(u)  
        IF the element at position u of self.parent NOT EQUALS u THEN  
            SET the element at position u of self.parent TO the result of calling find with argument the element at position u of self.parent  
        END IF  
        RETURN the element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(u v)  
        SET rootU TO the result of calling find with argument u  
        SET rootV TO the result of calling find with argument v  
        IF rootU NOT EQUALS rootV THEN  
            IF the element at position rootU of self.rank GREATER THAN the element at position rootV of self.rank THEN  
                SET the element at position rootV of self.parent TO rootU  
                INCREMENT the element at position rootU of self.size BY the element at position rootV of self.size  
            ELSE IF the element at position rootU of self.rank LESS THAN the element at position rootV of self.rank THEN  
                SET the element at position rootU of self.parent TO rootV  
                INCREMENT the element at position rootV of self.size BY the element at position rootU of self.size  
            ELSE  
                SET the element at position rootV of self.parent TO rootU  
                INCREMENT the element at position rootU of self.rank BY one  
                INCREMENT the element at position rootU of self.size BY the element at position rootV of self.size  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION largestComponentSize(nums)  
        SET max_num TO the maximum value in nums  
        SET uf TO a new UnionFind object initialized with max_num PLUS one  
        SET prime_to_index TO an empty dictionary  
        FOR each num IN nums  
            FOR each prime IN the result of calling prime_factors with argument num  
                IF prime EXISTS IN prime_to_index THEN  
                    CALL union on uf with arguments the element at key prime of prime_to_index AND num  
                ELSE  
                    SET the element at key prime of prime_to_index TO num  
                END IF  
            END FOR  
        END FOR  
        SET max_component_size TO zero  
        FOR each num IN nums  
            SET root TO the result of calling find on uf with argument num  
            IF max_component_size LESS THAN the element at position root of uf.size THEN  
                SET max_component_size TO the element at position root of uf.size  
            END IF  
        END FOR  
        RETURN max_component_size  
    END FUNCTION  

    FUNCTION prime_factors(n)  
        SET factors TO an empty set  
        WHILE n MODULO two EQUALS zero DO  
            ADD two TO factors  
            SET n TO n DIVIDED BY two  
        END WHILE  
        FOR i FROM three TO the integer part of the square root of n PLUS one STEP two  
            WHILE n MODULO i EQUALS zero DO  
                ADD i TO factors  
                SET n TO n DIVIDED BY i  
            END WHILE  
        END FOR  
        IF n GREATER THAN two THEN  
            ADD n TO factors  
        END IF  
        RETURN factors  
    END FUNCTION  
END CLASS
