CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix is empty OR element at position zero of matrix is empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET dp TO a two dimensional list of zeros with m PLUS one rows and n PLUS one columns  
        SET max_side TO zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of matrix element at position j MINUS one EQUALS the string one  
                    SET dp element at position i and j TO the MINIMUM OF dp element at position i MINUS one and j AND dp element at position i and j MINUS one AND dp element at position i MINUS one and j MINUS one PLUS one  
                    SET max_side TO the GREATER OF max_side AND dp element at position i and j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
