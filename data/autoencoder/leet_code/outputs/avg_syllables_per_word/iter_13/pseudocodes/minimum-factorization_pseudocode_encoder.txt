CLASS Solution  
    FUNCTION smallestFactorization WITH PARAMETER num  
        IF num EQUALS one  
            RETURN one  
        END IF  

        SET factors TO empty list  
        FOR digit FROM nine TO two DECREMENTING BY one  
            WHILE the remainder of num DIVIDED BY digit EQUALS zero  
                APPEND digit TO factors  
                SET num TO the quotient of num DIVIDED BY digit  
            END WHILE  
        END FOR  

        IF num NOT EQUALS one  
            RETURN zero  
        END IF  

        SORT factors IN ASCENDING ORDER  
        SET result TO the integer formed by concatenating the string representations of elements in factors IN ORDER  

        IF result GREATER THAN two RAISED TO THE THIRTY FIRST MINUS one  
            RETURN zero  
        END IF  

        RETURN result  
    END FUNCTION  
END CLASS
