CLASS Solution  
    FUNCTION leastInterval(tasks LIST OF STRING, n INTEGER) RETURNS INTEGER  
        SET task_counts TO the count of each element in tasks  
        SET max_heap TO an empty list  
        FOR each count IN the values of task_counts  
            APPEND the negative value of count TO max_heap  
        END FOR  
        APPLY heapify operation to max_heap to organize it as a min heap representing max heap behavior  
        
        SET time TO zero  
        SET queue TO an empty list representing a collection of tuples each containing a time and a task count  
        
        WHILE max_heap is not empty OR queue is not empty  
            WHILE queue is not empty AND the time part of the first element of queue EQUALS time  
                REMOVE the first element from queue and DESTRUCTURE it into unused variable and count  
                IF count IS LESS THAN zero  
                    PUSH count INTO max_heap using heap push operation  
                END IF  
            END WHILE  
            
            IF max_heap is not empty  
                REMOVE the smallest element from max_heap using heap pop and ASSIGN it to count  
                INCREMENT count BY one  
                IF count IS LESS THAN zero  
                    APPEND a tuple containing time PLUS n PLUS one and count TO queue  
                END IF  
            END IF  
            
            INCREMENT time BY one  
        END WHILE  
        
        RETURN time  
    END FUNCTION  
END CLASS
