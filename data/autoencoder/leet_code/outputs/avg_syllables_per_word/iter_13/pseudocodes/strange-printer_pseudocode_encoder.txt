CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i IS GREATER THAN j  
                RETURN zero  
            END IF  
            IF tuple of i and j IS IN memo  
                RETURN the element associated with tuple of i and j in memo  
            END IF  
            SET result TO the result of dp WITH parameters i AND j MINUS one AND memo PLUS one  
            FOR each k FROM i TO j MINUS one  
                IF the element at position k of s EQUALS the element at position j of s  
                    SET result TO the minimum of result AND the sum of dp WITH parameters i AND k MINUS one AND memo AND dp WITH parameters k AND j MINUS one AND memo  
                END IF  
            END FOR  
            SET the element associated with tuple of i and j in memo TO result  
            RETURN result  
        END FUNCTION  
        RETURN the result of dp WITH parameters zero AND the length of s MINUS one AND an empty dictionary  
    END FUNCTION  
END CLASS
