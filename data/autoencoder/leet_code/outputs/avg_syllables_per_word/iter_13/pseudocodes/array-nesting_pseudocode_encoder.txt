CLASS Solution  
    FUNCTION arrayNesting(nums)  
        SET visited TO a list of boolean false values with the length equal to the LENGTH OF nums  
        SET max_length TO zero  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF element at position index of visited IS EQUAL TO boolean false  
                SET current_length TO zero  
                SET k TO index  
                WHILE element at position k of visited IS EQUAL TO boolean false  
                    SET element at position k of visited TO boolean true  
                    SET k TO element at position k of nums  
                    INCREMENT current_length BY one  
                END WHILE  
                IF max_length IS LESS THAN current_length  
                    SET max_length TO current_length  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
