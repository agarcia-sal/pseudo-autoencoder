CLASS Solution  
    FUNCTION minDistance(houses AS List OF Integer, k AS Integer) RETURNS Integer  
        CALL the method sort OF houses  
        SET n TO the length OF houses  
        
        SET cost TO result of FUNCTION initialize_two_dimensional_list WITH parameters n AND n AND initial_value zero  
        
        FOR i FROM n MINUS one DOWNTO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost element at row i AND column j TO cost element at row i PLUS one AND column j MINUS one PLUS houses element at position j MINUS houses element at position i  
            END FOR  
        END FOR  
        
        SET dp TO result of FUNCTION initialize_two_dimensional_list WITH parameters n PLUS one AND k PLUS one AND initial_value positive infinity  
        SET dp element at row zero AND column zero TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET potential_value TO dp element at row p AND column j MINUS one PLUS cost element at row p AND column i MINUS one  
                    IF dp element at row i AND column j GREATER THAN potential_value THEN  
                        SET dp element at row i AND column j TO potential_value  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp element at row n AND column k  
    END FUNCTION  
    
    FUNCTION initialize_two_dimensional_list(rows AS Integer, columns AS Integer, initial_value AS any) RETURNS two dimensional List  
        SET result TO empty list  
        FOR each row_index FROM zero TO rows MINUS one  
            SET current_row TO empty list  
            FOR each column_index FROM zero TO columns MINUS one  
                APPEND initial_value TO current_row  
            END FOR  
            APPEND current_row TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
