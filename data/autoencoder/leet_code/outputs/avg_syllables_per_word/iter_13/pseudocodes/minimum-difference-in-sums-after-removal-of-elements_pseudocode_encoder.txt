CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
        
        SET min_heap TO an empty list  
        SET min_sums TO a list of zeros with LENGTH equal to two MULTIPLIED BY n PLUS one  
        
        FOR each index i FROM zero TO two MULTIPLIED BY n MINUS one  
            PUSH the negative of the element at position i of nums INTO min_heap AS A HEAP  
            IF the LENGTH OF min_heap IS GREATER THAN n  
                POP the smallest element FROM min_heap AS A HEAP  
            END IF  
            IF the LENGTH OF min_heap IS EQUAL TO n  
                SET the element at position i of min_sums TO the negative of the SUM OF elements in min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO an empty list  
        SET max_sums TO a list of zeros with LENGTH equal to two MULTIPLIED BY n PLUS one  
        
        FOR each index i FROM three MULTIPLIED BY n MINUS one DOWN TO n  
            PUSH the element at position i of nums INTO max_heap AS A HEAP  
            IF the LENGTH OF max_heap IS GREATER THAN n  
                POP the smallest element FROM max_heap AS A HEAP  
            END IF  
            IF the LENGTH OF max_heap IS EQUAL TO n  
                SET the element at position i of max_sums TO the SUM OF elements in max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO infinity  
        FOR each index i FROM n MINUS one TO two MULTIPLIED BY n MINUS one  
            SET min_diff TO the MINIMUM OF min_diff AND the element at position i of min_sums MINUS the element at position i PLUS one of max_sums  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
