CLASS Solution  
    FUNCTION getSum(a, b)  
        SET MAX TO the maximum thirty two bits integer  
        SET mask TO the mask representing thirty two bits all set to one  
        WHILE b is not equal to zero  
            SET carry TO the bitwise AND of a AND b shifted LEFT BY one position  
            SET a TO the bitwise XOR of a AND b AND masked by mask  
            SET b TO carry masked by mask  
        END WHILE  
        IF a is less than or equal to MAX  
            RETURN a  
        ELSE  
            RETURN the bitwise negation of the bitwise XOR of a AND mask  
        END IF  
    END FUNCTION  
END CLASS
