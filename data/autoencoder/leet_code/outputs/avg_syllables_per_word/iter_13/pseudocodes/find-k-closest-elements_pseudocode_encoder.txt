CLASS Solution  
    FUNCTION findClosestElements(arr LIST OF INTEGER comma k INTEGER comma x INTEGER) RETURNS LIST OF INTEGER  
        SET left TO the insertion position at the left side of x in arr MINUS one  
        SET right TO left PLUS one  
        WHILE the value of right MINUS left MINUS one IS LESS THAN k  
            IF left EQUALS minus one OR the value of right IS LESS THAN the LENGTH OF arr AND the absolute value of element at position right of arr MINUS x IS LESS THAN the absolute value of element at position left of arr MINUS x  
                INCREMENT right BY one  
            ELSE  
                DECREMENT left BY one  
            END IF  
        END WHILE  
        RETURN elements from position left PLUS one TO position right MINUS one of arr  
    END FUNCTION  
END CLASS
