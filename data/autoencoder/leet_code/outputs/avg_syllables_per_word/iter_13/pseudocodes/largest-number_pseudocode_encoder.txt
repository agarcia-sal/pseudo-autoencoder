CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO CONVERT each element OF nums TO string USING MAP FUNCTION  
        
        FUNCTION compare(x, y)  
            IF concatenation of x FOLLOWED BY y IS GREATER THAN concatenation of y FOLLOWED BY x  
                RETURN negative one  
            ELSE IF concatenation of x FOLLOWED BY y IS LESS THAN concatenation of y FOLLOWED BY x  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str USING custom comparison FUNCTION compare  
        
        IF element at position zero of nums_str EQUALS string zero  
            RETURN string zero  
        END IF  
        
        RETURN concatenation of all elements OF nums_str IN ORDER  
    END FUNCTION  
END CLASS
