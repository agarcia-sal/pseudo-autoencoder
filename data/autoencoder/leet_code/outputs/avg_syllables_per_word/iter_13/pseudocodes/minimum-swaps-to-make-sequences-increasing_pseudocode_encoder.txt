CLASS Solution  
    FUNCTION minSwap(nums1, nums2)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a LIST OF n ELEMENTS WHERE each ELEMENT IS A LIST OF TWO ZEROS  
        SET element at position zero of dp TO a LIST OF zero AND one RESPECTIVELY  
        FOR index i FROM one TO n MINUS one  
            SET condition_one TO the ELEMENT at position i of nums1 IS GREATER THAN the ELEMENT at position i MINUS one of nums1 AND the ELEMENT at position i of nums2 IS GREATER THAN the ELEMENT at position i MINUS one of nums2  
            SET condition_two TO the ELEMENT at position i of nums1 IS GREATER THAN the ELEMENT at position i MINUS one of nums2 AND the ELEMENT at position i of nums2 IS GREATER THAN the ELEMENT at position i MINUS one of nums1  
            IF condition_one HOLDS  
                IF condition_two HOLDS  
                    SET element at position zero of dp at position i TO the MINIMUM of the ELEMENT at position zero of dp at position i MINUS one AND the ELEMENT at position one of dp at position i MINUS one  
                    SET element at position one of dp at position i TO the ELEMENT at position zero of dp at position i PLUS one  
                ELSE  
                    SET element at position zero of dp at position i TO the ELEMENT at position zero of dp at position i MINUS one  
                    SET element at position one of dp at position i TO the ELEMENT at position one of dp at position i MINUS one PLUS one  
                END IF  
            ELSE  
                IF condition_two HOLDS  
                    SET element at position zero of dp at position i TO the ELEMENT at position one of dp at position i MINUS one  
                    SET element at position one of dp at position i TO the ELEMENT at position zero of dp at position i MINUS one PLUS one  
                ELSE  
                    SET element at position zero of dp at position i TO the ELEMENT at position one of dp at position i MINUS one  
                    SET element at position one of dp at position i TO the ELEMENT at position zero of dp at position i MINUS one PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN the MINIMUM of the ELEMENT at position zero of dp at position n MINUS one AND the ELEMENT at position one of dp at position n MINUS one  
    END FUNCTION  
END CLASS
