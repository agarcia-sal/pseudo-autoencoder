CLASS Solution  
    FUNCTION wiggleSort(nums)  
        CALL sort THE list nums IN ascending order  
        SET sorted_nums TO A copy OF the list nums  
        SET n TO the LENGTH OF the list nums  
        SET left TO the INTEGER division OF the difference BETWEEN n AND one BY two  
        SET right TO the difference BETWEEN n AND one  
        FOR index FROM zero TO the difference BETWEEN n AND one  
            IF the remainder OF dividing index BY two EQUALS zero  
                SET element at position index of list nums TO element at position left of list sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET element at position index of list nums TO element at position right of list sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
