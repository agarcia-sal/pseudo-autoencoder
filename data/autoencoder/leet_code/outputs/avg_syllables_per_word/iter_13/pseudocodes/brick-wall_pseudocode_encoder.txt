CLASS Solution  
    FUNCTION leastBricks WITH PARAMETERS wall  
        SET gap_counts TO an empty default dictionary with default integer values  
        
        FOR each row IN wall  
            SET total TO zero  
            FOR each brick in row from the first element TO the element before the last element  
                SET total TO total PLUS brick  
                INCREMENT the value at position total in gap_counts BY one  
            END FOR  
        END FOR  
        
        SET max_gaps TO the maximum value among all values in gap_counts OR zero if gap_counts is empty  
        
        RETURN the length of wall MINUS max_gaps  
    END FUNCTION  
END CLASS
