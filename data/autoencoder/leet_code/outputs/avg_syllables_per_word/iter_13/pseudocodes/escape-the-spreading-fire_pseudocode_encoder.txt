CLASS Solution  
    FUNCTION maximumMinutes(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO the LIST containing tuples ONE and ZERO MINUS ONE and ZERO ZERO and ONE ZERO and MINUS ONE  
        
        FUNCTION can_escape(wait)  
            SET fire_times TO a two dimensional list of size m by n FILLED WITH infinity  
            SET queue TO an empty double ended queue  
            
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    IF element at position i of grid EQUALS one  
                        APPEND the tuple i j and zero TO queue  
                        SET element at position i then j of fire_times TO zero  
                    END IF  
                END FOR  
            END FOR  
            
            WHILE queue IS NOT EMPTY  
                REMOVE the first element FROM queue AND ASSIGN TO x y and t  
                FOR each dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND element at position nx then ny of grid EQUALS zero AND element at position nx then ny of fire_times IS infinity  
                        SET element at position nx then ny of fire_times TO t PLUS one  
                        APPEND the tuple nx ny and t PLUS one TO queue  
                    END IF  
                END FOR  
            END WHILE  
            
            SET queue TO a double ended queue containing the tuple zero zero and wait  
            SET visited TO a set containing the tuple zero zero  
            
            WHILE queue IS NOT EMPTY  
                REMOVE the first element FROM queue AND ASSIGN TO x y and t  
                FOR each dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND the tuple nx ny IS NOT IN visited AND element at position nx then ny of grid EQUALS zero  
                        IF nx EQUALS m MINUS one AND ny EQUALS n MINUS one  
                            IF element at position nx then ny of fire_times IS GREATER THAN OR EQUAL TO t PLUS one  
                                RETURN True  
                            ELSE  
                                RETURN False  
                            END IF  
                        END IF  
                        IF element at position nx then ny of fire_times IS GREATER THAN t PLUS one  
                            ADD the tuple nx ny TO visited  
                            APPEND the tuple nx ny and t PLUS one TO queue  
                        END IF  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN False  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO m MULTIPLIED BY n  
        SET result TO negative one  
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER DIVISION OF the sum of left and right BY two  
            IF can_escape(mid) EQUALS True  
                SET result TO mid  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        
        IF result IS NOT EQUAL TO m MULTIPLIED BY n  
            RETURN result  
        ELSE  
            RETURN ten RAISED TO THE POWER OF nine  
        END IF  
    END FUNCTION  
END CLASS
