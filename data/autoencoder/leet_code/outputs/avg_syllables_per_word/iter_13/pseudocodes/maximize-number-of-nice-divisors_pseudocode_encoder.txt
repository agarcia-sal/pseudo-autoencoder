CLASS Solution  
    FUNCTION maxNiceDivisors(primeFactors)  
        SET MOD TO ten raised to the power of nine PLUS seven  
        
        IF primeFactors LESS THAN OR EQUAL TO three  
            RETURN primeFactors  
        END IF  
        
        SET num_threes TO the integer division of primeFactors BY three  
        SET remainder TO the modulo of primeFactors BY three  
        
        IF remainder EQUALS zero  
            RETURN three raised to the power of num_threes modulo MOD  
        ELSE IF remainder EQUALS one  
            RETURN the product of three raised to the power of num_threes MINUS one modulo MOD AND four modulo MOD  
        ELSE  
            RETURN the product of three raised to the power of num_threes modulo MOD AND two modulo MOD  
        END IF  
    END FUNCTION  
END CLASS
