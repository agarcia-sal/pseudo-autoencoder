CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF the CONDITION values IS empty HOLDS  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETER element at position zero of values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty HOLDS  
        REMOVE the first element FROM queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none HOLDS  
            SET node dot left TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none HOLDS  
            SET node dot right TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none HOLDS  
        RETURN True  
    ELSE IF p IS none OR q IS none HOLDS  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS q dot val HOLDS  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS p dot left q dot left AND the RESULT OF is_same_tree WITH PARAMETERS p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST WITH PARAMETERS self root AS Optional TreeNode  
        SET self dot max_sum TO zero  
        
        FUNCTION helper WITH PARAMETER node  
            IF node IS none HOLDS  
                RETURN True zero positive infinity negative infinity  
            END IF  
            
            SET left_is_bst left_sum left_min left_max TO the RESULT OF helper WITH PARAMETER node dot left  
            SET right_is_bst right_sum right_min right_max TO the RESULT OF helper WITH PARAMETER node dot right  
            
            IF left_is_bst IS True AND right_is_bst IS True AND left_max IS LESS THAN node dot val AND node dot val IS LESS THAN right_min HOLDS  
                SET current_sum TO left_sum PLUS right_sum PLUS node dot val  
                SET current_min TO the SMALLER OF node dot val AND left_min  
                SET current_max TO the GREATER OF node dot val AND right_max  
                SET self dot max_sum TO the GREATER OF self dot max_sum AND current_sum  
                RETURN True current_sum current_min current_max  
            END IF  
            
            RETURN False zero positive infinity negative infinity  
        END FUNCTION  
        
        CALL helper WITH PARAMETER root  
        RETURN self dot max_sum  
    END FUNCTION  
END CLASS
