CLASS Solution  
    FUNCTION trapRainWater(heightMap)  
        IF heightMap IS EMPTY OR element at position zero of heightMap IS EMPTY  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF heightMap  
        SET n TO the LENGTH OF element at position zero of heightMap  
        SET visited TO two dimensional list of False values with m rows and n columns  
        SET min_heap TO an empty list  
        
        FOR index i FROM zero TO m MINUS one  
            PUSH tuple consisting of element at position i of heightMap at position zero, i, zero INTO min_heap using heap push operation  
            PUSH tuple consisting of element at position i of heightMap at position n MINUS one, i, n MINUS one INTO min_heap using heap push operation  
            SET element at position i of visited at position zero TO True  
            SET element at position i of visited at position n MINUS one TO True  
        END FOR  
        
        FOR index j FROM zero TO n MINUS one  
            PUSH tuple consisting of element at position zero of heightMap at position j, zero, j INTO min_heap using heap push operation  
            PUSH tuple consisting of element at position m MINUS one of heightMap at position j, m MINUS one, j INTO min_heap using heap push operation  
            SET element at position zero of visited at position j TO True  
            SET element at position m MINUS one of visited at position j TO True  
        END FOR  
        
        SET directions TO list of tuples negative one zero, one zero, zero negative one, zero one  
        SET water_trapped TO zero  
        
        WHILE min_heap IS NOT EMPTY  
            POP the smallest element FROM min_heap USING heap pop operation AND ASSIGN TO height, x, y  
            
            FOR each tuple dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND element at position nx of visited at position ny IS False  
                    SET element at position nx of visited at position ny TO True  
                    SET difference TO height MINUS element at position nx of heightMap at position ny  
                    IF difference IS GREATER THAN zero  
                        INCREMENT water_trapped BY difference  
                    END IF  
                    SET max_height TO height IF height IS GREATER THAN element at position nx of heightMap at position ny ELSE element at position nx of heightMap at position ny  
                    PUSH tuple consisting of max_height, nx, ny INTO min_heap USING heap push operation  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN water_trapped  
    END FUNCTION  
END CLASS
