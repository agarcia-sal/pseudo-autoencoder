CLASS Solution  
    FUNCTION longestSubstring(s, k)  
        IF the LENGTH OF s LESS THAN k  
            RETURN zero  
        END IF  
        SET count TO A COLLECTION THAT COUNTS THE FREQUENCY OF EACH CHARACTER IN s  
        FOR each character IN count  
            IF the FREQUENCY OF character IN count LESS THAN k  
                RETURN THE MAXIMUM VALUE OF THE RESULT OF CALLING longestSubstring WITH each substring OBTAINED BY SPLITTING s BY character AND k  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
