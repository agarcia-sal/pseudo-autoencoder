CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MODULO TO one BILLION PLUS seven  
        SET end_with_zero TO zero  
        SET end_with_one TO zero  
        SET has_zero TO WHETHER the character zero APPEARS IN the string binary  
        FOR each bit IN binary  
            IF bit EQUALS character one  
                SET end_with_one TO the SUM OF end_with_zero PLUS end_with_one PLUS one MODULO MODULO  
            ELSE  
                SET end_with_zero TO the SUM OF end_with_zero PLUS end_with_one MODULO MODULO  
            END IF  
        END FOR  
        RETURN the SUM OF end_with_zero PLUS end_with_one PLUS has_zero MODULO MODULO  
    END FUNCTION  
END CLASS
