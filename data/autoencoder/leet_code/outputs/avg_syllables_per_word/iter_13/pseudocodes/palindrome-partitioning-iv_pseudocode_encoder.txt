CLASS Solution  
    FUNCTION checkPartitioning WITH PARAMETER s  
        FUNCTION is_palindrome WITH PARAMETER sub  
            IF sub EQUALS sub REVERSED  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET n TO the LENGTH OF s  
        SET is_pal TO two dimensional array of size n BY n FILLED WITH false  

        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF the element at position i of s EQUALS the element at position j of s  
                   AND (the difference of j MINUS i IS LESS THAN OR EQUAL TO one OR the element at position i PLUS one AND position j MINUS one of is_pal IS true)  
                    SET the element at position i AND j of is_pal TO true  
                END IF  
            END FOR  
        END FOR  

        FOR i FROM one TO n MINUS two  
            IF the element at position zero AND i MINUS one of is_pal IS true  
                FOR j FROM i TO n MINUS two  
                    IF the element at position i AND j of is_pal IS true  
                       AND the element at position j PLUS one AND n MINUS one of is_pal IS true  
                        RETURN true  
                    END IF  
                END FOR  
            END IF  
        END FOR  

        RETURN false  
    END FUNCTION  
END CLASS
