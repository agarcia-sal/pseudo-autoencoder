CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL sort the list nums in ascending order  
        SET n TO the LENGTH OF nums  
        SET first TO the element at position zero of nums  
        FOR i FROM one TO n MINUS one  
            SET k TO the element at position i of nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS the element at position i of nums  
                CONTINUE to the next iteration of the FOR loop  
            END IF  
            SET count TO a counter of the elements in nums capturing their frequencies  
            SET arr TO an empty list  
            FOR each x IN nums  
                IF the frequency of x in count EQUALS zero  
                    CONTINUE to the next iteration of the FOR loop  
                END IF  
                IF the frequency of element equal to x PLUS two MULTIPLIED BY k in count EQUALS zero  
                    BREAK out of the FOR loop  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT the frequency of x in count BY one  
                DECREMENT the frequency of element equal to x PLUS two MULTIPLIED BY k in count BY one  
            END FOR  
            IF the LENGTH OF arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an empty list  
    END FUNCTION  
END CLASS
