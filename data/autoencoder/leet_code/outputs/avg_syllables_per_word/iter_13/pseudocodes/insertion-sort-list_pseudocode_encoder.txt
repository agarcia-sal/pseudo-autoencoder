CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULT zero AND next DEFAULT none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the BOOLEAN VALUE OF values EQUALS false  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH element at position zero of values  
    SET p TO head  
    FOR each val IN the SUBLIST of values FROM position one TO THE END  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET the element at position next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN BOOLEAN TRUE  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN BOOLEAN FALSE  
    END IF  
    RETURN BOOLEAN EXPRESSION where the element val of p1 EQUALS the element val of p2 AND the RESULT OF is_same_list WITH PARAMETERS the element next of p1 AND the element next of p2 HOLDS TRUE  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList WITH PARAMETERS head  
        IF head EQUALS none OR the element next of head EQUALS none  
            RETURN head  
        END IF  
        SET dummy TO A NEW ListNode INITIALIZED WITH negative five thousand AND ONE  
        SET the element next of dummy TO head  
        SET current TO the element next of head  
        SET the element next of head TO none  

        WHILE current HOLDS TRUE  
            SET next_node TO the element next of current  
            SET prev TO dummy  
            WHILE the element next of prev HOLDS TRUE AND the element val of the element next of prev IS LESS THAN the element val of current  
                SET prev TO the element next of prev  
            END WHILE  
            SET the element next of current TO the element next of prev  
            SET the element next of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the element next of dummy  
    END FUNCTION  
END CLASS
