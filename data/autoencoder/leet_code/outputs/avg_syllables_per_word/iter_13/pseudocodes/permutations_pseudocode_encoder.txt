CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF the LENGTH OF remaining EQUALS zero  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR index FROM one TO the LENGTH OF remaining  
                SET new_path TO path PLUS the element at position index of remaining  
                SET new_remaining TO the SUBLIST of remaining FROM position one TO the position immediately BEFORE index PLUS the SUBLIST of remaining FROM the position immediately AFTER index TO the END  
                CALL backtrack WITH new_path AND new_remaining  
            END FOR  
        END FUNCTION  
        SET permutations TO an empty list  
        CALL backtrack WITH an empty list AND nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
