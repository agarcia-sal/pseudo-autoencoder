CLASS Solution  
    FUNCTION flipLights WITH PARAMETERS n, presses  
        SET n TO n MINUS THE VALUE OF n MINUS six OR n WHICHEVER IS LESSER  
        IF presses EQUALS zero  
            RETURN one  
        END IF  
        IF n EQUALS one  
            RETURN two  
        END IF  
        IF n EQUALS two  
            IF presses EQUALS one  
                RETURN three  
            ELSE  
                RETURN four  
            END IF  
        END IF  
        IF presses EQUALS one  
            RETURN four  
        END IF  
        IF presses EQUALS two  
            RETURN seven  
        END IF  
        RETURN eight  
    END FUNCTION  
END CLASS
