CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO result of splitting num AT the character plus  
            SET real TO the integer conversion OF element at position zero OF parts  
            SET imaginary TO the integer conversion OF the substring of element at position one OF parts FROM position zero TO the LENGTH OF element at position one OF parts MINUS one  
            RETURN real AND imaginary  
        END FUNCTION  
        
        SET real1 AND imag1 TO the result OF parse_complex CALLED WITH num1  
        SET real2 AND imag2 TO the result OF parse_complex CALLED WITH num2  
        
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        
        RETURN the concatenation OF real_part FOLLOWED BY the character plus FOLLOWED BY imaginary_part FOLLOWED BY the character i  
    END FUNCTION  
END CLASS
