CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            SET expected TO the STRING REPRESENTATION OF first PLUS second  
            SET length TO the LENGTH OF expected  
            IF start PLUS length GREATER THAN the LENGTH OF num OR the SUBSTRING FROM start TO start PLUS length OF num DOES NOT EQUAL expected  
                RETURN False  
            END IF  
            RETURN the RESULT OF is_valid_sequence WITH start SET TO start PLUS length AND first SET TO second AND second SET TO the INTEGER VALUE OF expected  
        END FUNCTION  
        
        SET n TO the LENGTH OF num  
        FOR i FROM one TO the INTEGER RESULT OF n DIVIDED BY two PLUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET first TO the SUBSTRING FROM position zero TO i OF num  
                SET second TO the SUBSTRING FROM position i TO j OF num  
                IF the LENGTH OF first GREATER THAN one AND the ELEMENT at POSITION zero OF first EQUALS character zero OR the LENGTH OF second GREATER THAN one AND the ELEMENT at POSITION zero OF second EQUALS character zero  
                    CONTINUE to the next iteration  
                END IF  
                IF the RESULT OF is_valid_sequence WITH start SET TO j AND first SET TO the INTEGER VALUE OF first AND second SET TO the INTEGER VALUE OF second  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
