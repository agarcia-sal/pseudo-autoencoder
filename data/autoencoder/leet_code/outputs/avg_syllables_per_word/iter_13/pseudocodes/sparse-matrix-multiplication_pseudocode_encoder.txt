CLASS Solution  
    FUNCTION multiply(mat1 AS List OF List OF integer, mat2 AS List OF List OF integer) RETURNS List OF List OF integer  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF element at position zero of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF element at position zero of mat2  
        SET result TO a list containing m elements each being a list containing n elements each initialized to zero  
        FOR each i FROM zero TO m MINUS one  
            FOR each j FROM zero TO n MINUS one  
                FOR each p FROM zero TO k MINUS one  
                    INCREMENT element at position j of element at position i of result BY element at position p of element at position i of mat1 MULTIPLIED BY element at position j of element at position p of mat2  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
