CLASS Solution  
    FUNCTION largestSumOfAverages(nums, k)  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a LIST OF ZEROES of LENGTH n PLUS one  
        FOR each index i FROM zero TO n MINUS one  
            SET element at position i PLUS one OF prefix_sum TO element at position i OF prefix_sum PLUS element at position i OF nums  
        END FOR  
        
        FUNCTION average(i, j)  
            SET numerator TO element at position j PLUS one OF prefix_sum MINUS element at position i OF prefix_sum  
            SET denominator TO j MINUS i PLUS one  
            RETURN numerator DIVIDED BY denominator  
        END FUNCTION  
        
        SET dp TO a TWO DIMENSIONAL LIST OF ZEROES with n PLUS one ROWS AND k PLUS one COLUMNS  
        
        FOR each index i FROM zero TO n MINUS one  
            SET element at position i AND one OF dp TO average(i, n MINUS one)  
        END FOR  
        
        FOR each index i FROM n MINUS one DOWN TO zero  
            FOR each partitions FROM two TO k  
                FOR each index j FROM i PLUS one TO n MINUS one  
                    SET candidate_value TO element at position j AND partitions MINUS one OF dp PLUS average(i, j MINUS one)  
                    IF candidate_value GREATER THAN element at position i AND partitions OF dp THEN  
                        SET element at position i AND partitions OF dp TO candidate_value  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero AND k OF dp  
    END FUNCTION  
END CLASS
