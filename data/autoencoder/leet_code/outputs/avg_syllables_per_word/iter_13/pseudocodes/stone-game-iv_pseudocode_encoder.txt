CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO list of False values with length n PLUS one  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE the square of j is LESS THAN OR EQUAL TO i  
                IF the element at position i MINUS the square of j of dp EQUALS False  
                    SET the element at position i of dp TO True  
                    BREAK out of the WHILE loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN the element at position n of dp  
    END FUNCTION  
END CLASS
