CLASS Solution  
    FUNCTION uniquePaths WITH PARAMETERS m AND n  
        SET dp TO a two dimensional array of size m by n where each element is set to one  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET element at position i and position j of dp TO element at position i MINUS one and position j of dp PLUS element at position i and position j MINUS one of dp  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one and position n MINUS one of dp  
    END FUNCTION  
END CLASS
