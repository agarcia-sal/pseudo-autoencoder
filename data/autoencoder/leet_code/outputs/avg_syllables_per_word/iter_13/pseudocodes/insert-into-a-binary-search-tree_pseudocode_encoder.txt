CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero left EQUAL TO none right EQUAL TO none  
        SET the value property TO val  
        SET the left property TO left  
        SET the right property TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF the values list IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO A NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left property of node TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right property of node TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value property of p IS NOT EQUAL TO the value property of q  
        RETURN false  
    ELSE  
        RETURN THE RESULT OF is_same_tree WITH PARAMETERS the left property of p AND the left property of q  
            AND THE RESULT OF is_same_tree WITH PARAMETERS the right property of p AND the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST WITH PARAMETERS root AND val  
        FUNCTION dfs WITH PARAMETER root  
            IF root IS none  
                RETURN A NEW TreeNode INITIALIZED WITH val  
            END IF  
            IF the value property of root IS LESS THAN val  
                SET the right property of root TO THE RESULT OF dfs WITH THE right property of root AS ARGUMENT  
            ELSE  
                SET the left property of root TO THE RESULT OF dfs WITH THE left property of root AS ARGUMENT  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN THE RESULT OF dfs WITH root AS ARGUMENT  
    END FUNCTION  
END CLASS
