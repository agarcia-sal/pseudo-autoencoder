CLASS Solution  
    FUNCTION maxSlidingWindow(nums LIST OF INTEGER comma k INTEGER) RETURNS LIST OF INTEGER  
        IF nums IS EMPTY OR k EQUALS zero THEN  
            RETURN empty list  
        END IF  
        SET deq TO new empty double ended queue  
        SET result TO new empty list  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT EMPTY AND element at position zero of deq IS LESS THAN i MINUS k PLUS one THEN  
                REMOVE the element from the front of deq  
            END IF  
            WHILE deq IS NOT EMPTY AND element at position element at position minus one of deq of nums IS LESS THAN element at position i of nums DO  
                REMOVE the element from the end of deq  
            END WHILE  
            APPEND i TO deq  
            IF i IS GREATER THAN OR EQUAL TO k MINUS one THEN  
                APPEND element at position element at position zero of deq of nums TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
