CLASS Solution  
    FUNCTION generatePalindromes WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER t  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND corresponding count v IN the ITEMS OF cnt  
                IF v IS GREATER THAN one  
                    DECREMENT the COUNT OF c IN cnt BY two  
                    CALL dfs WITH the STRING FORMED BY CONCATENATING c PLUS t PLUS c  
                    INCREMENT the COUNT OF c IN cnt BY two  
                END IF  
            END FOR  
        END FUNCTION  

        ASSIGN cnt TO THE RESULT OF CALLING Counter WITH s  
        SET mid TO the EMPTY STRING  
        FOR each character c AND corresponding count v IN the ITEMS OF cnt  
            IF v MODULO two EQUALS one  
                IF mid IS NOT the EMPTY STRING  
                    RETURN THE EMPTY LIST  
                END IF  
                SET mid TO c  
                DECREMENT the COUNT OF c IN cnt BY one  
            END IF  
        END FOR  
        SET ans TO the EMPTY LIST  
        CALL dfs WITH mid  
        RETURN ans  
    END FUNCTION  
END CLASS
