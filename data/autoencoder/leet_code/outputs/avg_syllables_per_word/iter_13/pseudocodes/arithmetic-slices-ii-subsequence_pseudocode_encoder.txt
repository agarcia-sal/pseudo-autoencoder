CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        
        SET dp TO a list of length n where each element is a mapping from integer differences to integer counts initialized to zero  
        SET total_count TO zero  
        
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO i MINUS one  
                SET d TO element at position i of nums MINUS element at position j of nums  
                
                IF d EXISTS AS A KEY IN the mapping at position j of dp  
                    INCREMENT the count associated with key d at position i of dp BY the count associated with key d at position j of dp PLUS one  
                    INCREMENT total_count BY the count associated with key d at position j of dp  
                ELSE  
                    INCREMENT the count associated with key d at position i of dp BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
