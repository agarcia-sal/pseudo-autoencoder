CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the SET of all characters in time with the colon character REMOVED  
        SET hour TO the INTEGER VALUE of the substring from position zero TO one of time  
        SET minute TO the INTEGER VALUE of the substring from position three TO four of time  
        
        FUNCTION generate_times(digits)  
            SET times TO an empty LIST  
            FOR each first_digit_of_hour IN digits  
                FOR each second_digit_of_hour IN digits  
                    FOR each first_digit_of_minute IN digits  
                        FOR each second_digit_of_minute IN digits  
                            SET hour_value TO the INTEGER VALUE of the concatenation of first_digit_of_hour WITH second_digit_of_hour  
                            SET minute_value TO the INTEGER VALUE of the concatenation of first_digit_of_minute WITH second_digit_of_minute  
                            IF hour_value IS LESS THAN twenty FOUR AND minute_value IS LESS THAN sixty  
                                APPEND the PAIR of hour_value AND minute_value TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN times SORTED IN ASCENDING ORDER  
        END FUNCTION  
        
        SET all_times TO the RESULT OF generate_times WITH digits AS ARGUMENT  
        SET current_time_index TO the POSITION of the PAIR of hour AND minute IN all_times  
        
        IF current_time_index PLUS one IS LESS THAN the LENGTH OF all_times  
            SET next_hour TO the hour PART OF the element at position current_time_index PLUS one OF all_times  
            SET next_minute TO the minute PART OF the element at position current_time_index PLUS one OF all_times  
        ELSE  
            SET next_hour TO the hour PART OF the FIRST ELEMENT OF all_times  
            SET next_minute TO the minute PART OF the FIRST ELEMENT OF all_times  
        END IF  
        
        RETURN next_hour FORMATTED AS two digits FOLLOWED BY a colon FOLLOWED BY next_minute FORMATTED AS two digits  
    END FUNCTION  
END CLASS
