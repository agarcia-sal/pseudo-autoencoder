CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        FOR index FROM zero TO LENGTH OF nums MINUS one  
            CALL heap push ON min_heap WITH tuple containing element at position zero of list at position index of nums AND index AND zero  
            SET max_value TO GREATER THAN VALUE BETWEEN max_value AND element at position zero of list at position index of nums  
        END FOR  
        SET smallest_range TO infinity  
        SET result TO list containing negative one hundred thousand AND one hundred thousand  
        WHILE min_heap is not empty  
            CALL heap pop ON min_heap AND DECOMPOSE INTO min_value AND list_index AND element_index  
            IF max_value MINUS min_value LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO list containing min_value AND max_value  
            END IF  
            IF element_index PLUS one LESS THAN LENGTH OF list at position list_index of nums  
                SET next_value TO element at position element_index PLUS one of list at position list_index of nums  
                CALL heap push ON min_heap WITH tuple containing next_value AND list_index AND element_index PLUS one  
                SET max_value TO GREATER THAN VALUE BETWEEN max_value AND next_value  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
