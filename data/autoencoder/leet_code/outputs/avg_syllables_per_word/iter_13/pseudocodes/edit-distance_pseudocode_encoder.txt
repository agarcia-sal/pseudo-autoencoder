CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        SET dp TO a two dimensional list with m PLUS one rows and n PLUS one columns ALL initialized TO zero  
  
        FOR i FROM zero TO m  
            SET element at position zero of the list at position i of dp TO i  
        END FOR  
  
        FOR j FROM zero TO n  
            SET element at position j of the list at position zero of dp TO j  
        END FOR  
  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2  
                    SET element at position j of the list at position i of dp TO the element at position j MINUS one of the list at position i MINUS one of dp  
                ELSE  
                    SET element at position j of the list at position i of dp TO one PLUS the minimum value among the element at position j of the list at position i MINUS one of dp, the element at position j MINUS one of the list at position i of dp, and the element at position j MINUS one of the list at position i MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
  
        RETURN the element at position n of the list at position m of dp  
    END FUNCTION  
END CLASS
