CLASS Solution  
    FUNCTION numWays(n INTEGER, k INTEGER) RETURNS INTEGER  
        IF n EQUALS zero THEN  
            RETURN zero  
        END IF  
        IF n EQUALS one THEN  
            RETURN k  
        END IF  
        IF n EQUALS two THEN  
            RETURN k MULTIPLIED BY k  
        END IF  
        SET same TO k  
        SET diff TO k MULTIPLIED BY the difference of k MINUS one  
        SET total TO the sum of same AND diff  
        FOR index FROM three TO n  
            SET new_same TO diff  
            SET new_diff TO total MULTIPLIED BY the difference of k MINUS one  
            SET same TO new_same  
            SET diff TO new_diff  
            SET total TO the sum of same AND diff  
        END FOR  
        RETURN total  
    END FUNCTION  
END CLASS
