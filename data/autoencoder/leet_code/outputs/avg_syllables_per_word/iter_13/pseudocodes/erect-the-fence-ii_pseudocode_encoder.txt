CLASS Solution  
    FUNCTION outerTrees(trees)  
        FUNCTION dist_sq(p1, p2)  
            RETURN the SQUARE OF the DIFFERENCE BETWEEN element at position zero of p1 AND element at position zero of p2 PLUS the SQUARE OF the DIFFERENCE BETWEEN element at position one of p1 AND element at position one of p2  
        END FUNCTION  

        FUNCTION circumcenter(p1, p2, p3)  
            SET ax TO element at position zero of p1  
            SET ay TO element at position one of p1  
            SET bx TO element at position zero of p2  
            SET by TO element at position one of p2  
            SET cx TO element at position zero of p3  
            SET cy TO element at position one of p3  
            SET d TO two MULTIPLIED BY the SUM of ax MULTIPLIED BY the DIFFERENCE of by MINUS cy PLUS bx MULTIPLIED BY the DIFFERENCE of cy MINUS ay PLUS cx MULTIPLIED BY the DIFFERENCE of ay MINUS by  
            SET ux TO the SUM of ax SQUARED PLUS ay SQUARED MULTIPLIED BY the DIFFERENCE of by MINUS cy PLUS bx SQUARED PLUS by SQUARED MULTIPLIED BY the DIFFERENCE of cy MINUS ay PLUS cx SQUARED PLUS cy SQUARED MULTIPLIED BY the DIFFERENCE of ay MINUS by DIVIDED BY d  
            SET uy TO the SUM of ax SQUARED PLUS ay SQUARED MULTIPLIED BY the DIFFERENCE of cx MINUS bx PLUS bx SQUARED PLUS by SQUARED MULTIPLIED BY the DIFFERENCE of ax MINUS cx PLUS cx SQUARED PLUS cy SQUARED MULTIPLIED BY the DIFFERENCE of bx MINUS ax DIVIDED BY d  
            RETURN a list containing ux and uy, and the SQUARE ROOT of the SUM of the SQUARE OF the DIFFERENCE of ux MINUS ax PLUS the SQUARE OF the DIFFERENCE of uy MINUS ay  
        END FUNCTION  

        FUNCTION welzl(points, boundary, n)  
            IF n EQUALS zero OR the LENGTH OF boundary EQUALS three  
                IF the LENGTH OF boundary EQUALS zero  
                    RETURN a list containing zero and zero, and zero  
                ELSE IF the LENGTH OF boundary EQUALS one  
                    RETURN the single element of boundary, and zero  
                ELSE IF the LENGTH OF boundary EQUALS two  
                    SET center TO a list containing the MEAN of element at position zero of the first element of boundary AND element at position zero of the second element of boundary AND the MEAN of element at position one of the first element of boundary AND element at position one of the second element of boundary  
                    SET radius TO the SQUARE ROOT of the DISTANCE SQUARED BETWEEN the first and second element of boundary DIVIDED BY two  
                    RETURN center, radius  
                ELSE  
                    RETURN the result of circumcenter with the three elements of boundary as arguments  
                END IF  
            END IF  
            
            SET p TO the element at position n MINUS one of points  
            SET center, radius TO the result of welzl with points, boundary, and n MINUS one  
            IF the DISTANCE SQUARED BETWEEN center AND p LESS THAN OR EQUAL TO radius MULTIPLIED BY radius  
                RETURN center, radius  
            END IF  
            
            RETURN the result of welzl with points, the concatenation of boundary AND a list containing p, and n MINUS one  
        END FUNCTION  

        APPLY random shuffle TO trees  
        SET center, radius TO the result of welzl with trees, empty list, and the LENGTH OF trees  
        RETURN a list containing element at position zero of center, element at position one of center, radius  
    END FUNCTION  
END CLASS
