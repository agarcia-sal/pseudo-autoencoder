CLASS Solution  
    FUNCTION multiply(num1 AS string, num2 AS string) RETURNS string  
        IF num1 EQUALS character zero OR num2 EQUALS character zero  
            RETURN character zero  
        END IF  
        SET result TO list OF zeroes WITH length EQUAL TO the LENGTH OF num1 PLUS the LENGTH OF num2  
        SET num1 TO reversed sequence of characters of num1  
        SET num2 TO reversed sequence of characters of num2  
        FOR index i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR index j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the integer VALUE of element at position i of num1 MINUS integer VALUE of character zero MULTIPLIED BY the integer VALUE of element at position j of num2 MINUS integer VALUE of character zero  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS element at position p1 of result  
                SET element at position p1 of result TO sum_ MODULO ten  
                INCREMENT element at position p2 of result BY sum_ DIVIDED BY ten  
            END FOR  
        END FOR  
        SET result_str TO concatenation of string representations of elements of reversed result  
        RETURN result_str WITH all leading characters equal to character zero removed  
    END FUNCTION  
END CLASS
