CLASS Solution  
    FUNCTION wordSquares(words)  
        FROM collections IMPORT defaultdict  
        SET prefix_to_words TO a new default dictionary with list as default value  
        FOR each word IN words  
            FOR each index FROM zero TO the LENGTH OF word MINUS one  
                SET prefix TO substring from position zero TO index MINUS one of word  
                APPEND word TO the list at prefix in prefix_to_words  
            END FOR  
        END FOR  

        SET results TO an empty list  

        FUNCTION backtrack(square)  
            IF the LENGTH OF square EQUALS the LENGTH OF the element at position zero of words THEN  
                APPEND square TO results  
                RETURN  
            END IF  
            SET prefix TO the concatenation of the element at position equal to the LENGTH OF square of each word IN square  
            FOR each candidate IN the list at prefix in prefix_to_words  
                CALL backtrack with the argument equal to the list consisting of the elements of square followed by candidate  
            END FOR  
        END FUNCTION  

        FOR each word IN words  
            CALL backtrack with the argument equal to the list containing word  
        END FOR  

        RETURN results  
    END FUNCTION  
END CLASS
