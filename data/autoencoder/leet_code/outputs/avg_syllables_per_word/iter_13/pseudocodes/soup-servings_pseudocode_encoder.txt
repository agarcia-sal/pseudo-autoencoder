CLASS Solution  
    FUNCTION soupServings(n)  
        IF n IS GREATER THAN four thousand eight hundred  
            RETURN one point zero  
        END IF  
      
        SET n TO the result of n PLUS twenty four DIVIDED BY twenty five rounded down to the nearest integer greater than or equal to the division result  
      
        SET memo TO an empty collection for storing computed results  
      
        FUNCTION dp(a, b)  
            IF the pair a and b EXISTS AS a key IN memo  
                RETURN the value CORRESPONDING TO the key of pair a and b IN memo  
            END IF  
          
            IF a IS LESS THAN OR EQUAL TO zero AND b IS LESS THAN OR EQUAL TO zero  
                RETURN zero point five  
            END IF  
          
            IF a IS LESS THAN OR EQUAL TO zero  
                RETURN one point zero  
            END IF  
          
            IF b IS LESS THAN OR EQUAL TO zero  
                RETURN zero point zero  
            END IF  
          
            SET prob TO zero point two five MULTIPLIED BY the sum of the following:  
                the result of dp applied to a MINUS four AND b  
                the result of dp applied to a MINUS three AND b MINUS one  
                the result of dp applied to a MINUS two AND b MINUS two  
                the result of dp applied to a MINUS one AND b MINUS three  
          
            SET the value associated with key pair a and b IN memo TO prob  
            RETURN prob  
        END FUNCTION  
      
        RETURN the result of dp applied to n AND n  
    END FUNCTION  
END CLASS
