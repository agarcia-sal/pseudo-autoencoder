CLASS Solution  
    FUNCTION minChanges(nums AS List OF integers, k AS integer)  
        SET groups TO a new mapping from integer TO list of integers with default empty list  
        FOR index FROM zero TO LENGTH OF nums MINUS one  
            APPEND element at position index of nums TO element at position remainder of index DIVIDED BY k in groups  
        END FOR  

        SET freq TO a new list containing k mappings from integer TO integer with default zero  
        SET min_size TO a new list containing k elements all set to infinity  
        FOR index FROM zero TO k MINUS one  
            FOR each number IN element at position index of groups  
                INCREMENT element at position number of element at position index of freq BY one  
            END FOR  
            SET element at position index of min_size TO the LENGTH OF element at position index of groups  
        END FOR  

        SET dp TO a new list containing one thousand twenty four elements all set to infinity  
        SET element at position zero of dp TO zero  

        FOR index FROM zero TO k MINUS one  
            SET new_dp TO a new list containing one thousand twenty four elements all set to infinity  
            SET min_dp TO the minimum value in dp  
            FOR j FROM zero TO one thousand twenty three  
                SET element at position j of new_dp TO min_dp PLUS element at position index of min_size  
                FOR each pair of number and count IN elements of element at position index of freq  
                    SET candidate_value TO element at position bitwise exclusive or of j and number of dp PLUS element at position index of min_size MINUS count  
                    IF candidate_value LESS THAN element at position j of new_dp  
                        SET element at position j of new_dp TO candidate_value  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  

        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
