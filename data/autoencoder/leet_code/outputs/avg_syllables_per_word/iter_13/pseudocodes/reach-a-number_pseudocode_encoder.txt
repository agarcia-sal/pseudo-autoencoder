CLASS Solution  
    FUNCTION reachNumber WITH PARAMETER target  
        SET target TO the ABSOLUTE VALUE OF target  
        SET k TO zero  
        SET sum_steps TO zero  
        WHILE sum_steps IS LESS THAN target  
            INCREMENT k BY one  
            INCREMENT sum_steps BY k  
        END WHILE  
        IF the DIFFERENCE BETWEEN sum_steps AND target MODULO two EQUALS zero  
            RETURN k  
        ELSE  
            IF k MODULO two EQUALS zero  
                RETURN k PLUS one  
            ELSE  
                RETURN k PLUS two  
            END IF  
        END IF  
    END FUNCTION  
END CLASS
