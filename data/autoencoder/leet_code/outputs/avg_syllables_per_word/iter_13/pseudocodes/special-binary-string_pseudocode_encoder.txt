CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            
            FOR each index AND character IN the enumeration of s  
                IF character EQUALS digit one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                
                IF count EQUALS zero  
                    APPEND character digit one CONCATENATED WITH dfs of substring from position start PLUS one TO index CONCATENATED WITH character digit zero TO specials  
                    SET start TO index PLUS one  
                END IF  
            END FOR  
            
            SORT specials IN descending lexicographical order  
            SET result TO the concatenation of all elements in specials  
            RETURN result  
        END FUNCTION  
        
        RETURN dfs of s  
    END FUNCTION  
END CLASS
