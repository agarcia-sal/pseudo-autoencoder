CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO a list of zeros with length batchSize  
        FOR each group IN groups  
            INCREMENT element at position the remainder of group DIVIDED BY batchSize of remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder, variable number of remainder_count parameters)  
            SET remainder_count TO a list CONSTRUCTED FROM the variable remainder_count parameters  
            SET max_happy TO zero  
            
            FOR each index i FROM zero TO batchSize MINUS one  
                IF element at position i of remainder_count EQUALS zero  
                    CONTINUE to next iteration  
                END IF  
                
                DECREMENT element at position i of remainder_count BY one  
                SET new_remainder TO the remainder of remainder MINUS i DIVIDED BY batchSize  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS the RESULT OF CALLING dp WITH new_remainder AND the elements of remainder_count AS parameters  
                ELSE  
                    SET happy_groups TO the RESULT OF CALLING dp WITH new_remainder AND the elements of remainder_count AS parameters  
                END IF  
                
                IF happy_groups GREATER THAN max_happy  
                    SET max_happy TO happy_groups  
                END IF  
                
                INCREMENT element at position i of remainder_count BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN the RESULT OF CALLING dp WITH zero AND the elements of remainder_count AS parameters  
    END FUNCTION  
END CLASS
