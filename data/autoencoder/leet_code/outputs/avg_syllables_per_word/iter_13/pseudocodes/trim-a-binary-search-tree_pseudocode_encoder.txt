CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH parameter val EQUALS element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        REMOVE the first element FROM queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO A NEW TreeNode WITH parameter val EQUALS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO A NEW TreeNode WITH parameter val EQUALS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST(self, root, low, high)  
        IF root IS none  
            RETURN none  
        END IF  
        IF root.val IS LESS THAN low  
            RETURN self.trimBST(root.right, low, high)  
        END IF  
        IF root.val IS GREATER THAN high  
            RETURN self.trimBST(root.left, low, high)  
        END IF  
        SET root.left TO self.trimBST(root.left, low, high)  
        SET root.right TO self.trimBST(root.right, low, high)  
        RETURN root  
    END FUNCTION  
END CLASS
