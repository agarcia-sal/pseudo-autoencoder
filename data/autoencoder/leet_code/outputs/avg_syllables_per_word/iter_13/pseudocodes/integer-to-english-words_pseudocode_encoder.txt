CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  
        
        SET below_twenty TO the list of words for numbers zero to nineteen starting with empty string for zero  
        SET tens TO the list of words for ten multiples starting with empty strings for positions zero and one  
        SET thousands TO the list containing empty string followed by Thousand Million and Billion  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            ELSE IF n LESS THAN twenty  
                RETURN the element at position n of below_twenty concatenated with a single space  
            ELSE IF n LESS THAN one hundred  
                RETURN the element at position of the result of n DIVIDED BY ten of tens concatenated with a single space concatenated with the result of helper called with the remainder of n DIVIDED BY ten  
            ELSE  
                RETURN the element at position of the result of n DIVIDED BY one hundred of below_twenty concatenated with the word Hundred concatenated with a single space concatenated with the result of helper called with the remainder of n DIVIDED BY one hundred  
            END IF  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero  
            IF the remainder of num DIVIDED BY one thousand NOT EQUALS zero  
                SET result TO the concatenation of helper called with the remainder of num DIVIDED BY one thousand concatenated with the element at position i of thousands concatenated with a single space concatenated with result  
            END IF  
            SET num TO the result of integer division of num BY one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
