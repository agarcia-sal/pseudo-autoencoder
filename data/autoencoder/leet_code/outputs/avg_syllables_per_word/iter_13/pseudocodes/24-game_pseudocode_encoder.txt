CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF element at position zero of nums MINUS twenty four IS LESS THAN one TIMES TEN TO THE POWER OF MINUS six  
            END IF  
            
            FOR index i FROM zero TO the LENGTH OF nums MINUS one  
                FOR index j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i IS NOT EQUAL TO j  
                        SET new_nums TO all elements at positions k of nums WHERE k IS NOT EQUAL TO i AND k IS NOT EQUAL TO j  
                        
                        FOR each op IN the collection of true division MULTIPLICATION ADDITION SUBTRACTION operations  
                            IF (op IS ADDITION OR op IS MULTIPLICATION) AND j IS GREATER THAN i  
                                CONTINUE TO the next iteration to avoid duplicate calculations  
                            END IF  
                            
                            IF op IS NOT true division OR element at position j of nums IS NOT EQUAL TO zero  
                                APPEND the result of operation op APPLIED TO element at position i of nums AND element at position j of nums TO new_nums  
                                IF solve(new_nums) RETURNS true  
                                    RETURN true  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        FROM operator IMPORT true division MULTIPLICATION ADDITION SUBTRACTION  
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
