CLASS ListNode  
    FUNCTION __init__(val ASSIGNED TO zero, next ASSIGNED TO none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO a new ListNode constructed with the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values from position one TO the end  
        SET node TO a new ListNode constructed with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the equality test whether the val attribute of p1 EQUALS the val attribute of p2 AND a recursive call to is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(val ASSIGNED TO zero, left ASSIGNED TO none, right ASSIGNED TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS greater than zero  
        SET node TO the element REMOVED FROM THE FRONT of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right attribute of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of a recursive call to is_same_tree with the left attribute of p AND the left attribute of q AND a recursive call to is_same_tree with the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO the LENGTH OF s  
                RETURN none AND index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the LENGTH OF s AND (the element at position index of s IS a digit OR the element at position index of s EQUALS a minus sign)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO a new TreeNode constructed with the integer conversion of the substring from position start TO the position one LESS THAN index of s  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS an opening parenthesis  
                SET the left attribute of node AND index TO the result of parse_tree with the argument one GREATER THAN index  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS an opening parenthesis  
                SET the right attribute of node AND index TO the result of parse_tree with the argument one GREATER THAN index  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS a closing parenthesis  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND discard_value TO the result of parse_tree with zero as the argument  
        RETURN root  
    END FUNCTION  
END CLASS
