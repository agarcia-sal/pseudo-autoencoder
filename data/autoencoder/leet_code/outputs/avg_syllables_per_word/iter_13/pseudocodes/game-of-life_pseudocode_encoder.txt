CLASS Solution  
    FUNCTION gameOfLife(board)  
        IF board IS empty OR element at position zero of board IS empty  
            RETURN  
        END IF  
        
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        
        SET directions TO the LIST OF pairs consisting of MINUS one and MINUS one MINUS one and zero MINUS one and one zero and MINUS one zero and one one and MINUS one one and zero one and one  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET live_neighbors TO zero  
                FOR each pair dx and dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN m AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND (element at position ni of board AND element at position nj of that IS EQUAL TO one OR element at position ni of board AND element at position nj of that IS EQUAL TO three)  
                        INCREMENT live_neighbors BY one  
                    END IF  
                END FOR  
                
                IF element at position i of board AND element at position j of that IS EQUAL TO one  
                    IF live_neighbors IS LESS THAN two OR live_neighbors IS GREATER THAN three  
                        SET element at position i of board AND element at position j of that TO three  
                    END IF  
                ELSE  
                    IF live_neighbors IS EQUAL TO three  
                        SET element at position i of board AND element at position j of that TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board AND element at position j of that IS EQUAL TO two  
                    SET element at position i of board AND element at position j of that TO one  
                ELSE IF element at position i of board AND element at position j of that IS EQUAL TO three  
                    SET element at position i of board AND element at position j of that TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
