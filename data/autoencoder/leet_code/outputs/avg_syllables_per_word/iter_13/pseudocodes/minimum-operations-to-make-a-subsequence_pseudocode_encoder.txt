CLASS Solution  
    FUNCTION minOperations(target AS List of integers arr AS List of integers) RETURNS integer  
        SET index_map TO an empty mapping from integer TO integer  
        FOR each index FROM zero TO the LENGTH OF target MINUS one  
            SET value TO element at position index of target  
            ASSIGN index_map at key value TO index  
        END FOR  
        
        SET transformed_arr TO an empty list of integers  
        FOR each value IN arr  
            IF index_map contains key value THEN  
                APPEND index_map at key value TO transformed_arr  
            END IF  
        END FOR  
        
        SET lis TO an empty list of integers  
        FOR each num IN transformed_arr  
            SET pos TO the position where num WOULD BE INSERTED TO KEEP lis IN NON-DECREASING ORDER USING BINARY SEARCH  
            IF pos EQUALS the LENGTH OF lis THEN  
                APPEND num TO lis  
            ELSE  
                SET element at position pos of lis TO num  
            END IF  
        END FOR  
        
        SET lcs_length TO the LENGTH OF lis  
        RETURN the LENGTH OF target MINUS lcs_length  
    END FUNCTION  
END CLASS
