CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO the frequency count of digits in the input list digits  
        SET total_sum TO the sum of all elements in digits  
        SET remainder TO the remainder when total_sum is divided by three  
        
        IF remainder EQUALS one  
            FOR each digit d IN the list containing one four seven  
                IF the frequency of digit d in count is greater than zero  
                    DECREMENT the frequency of digit d in count BY one  
                    EXIT the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit d IN the list containing two five eight  
                    IF the frequency of digit d in count is greater than one  
                        DECREMENT the frequency of digit d in count BY two  
                        EXIT the loop  
                    END IF  
                END FOR  
        END IF  
        
        IF remainder EQUALS two  
            FOR each digit d IN the list containing two five eight  
                IF the frequency of digit d in count is greater than zero  
                    DECREMENT the frequency of digit d in count BY one  
                    EXIT the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit d IN the list containing one four seven  
                    IF the frequency of digit d in count is greater than one  
                        DECREMENT the frequency of digit d in count BY two  
                        EXIT the loop  
                    END IF  
                END FOR  
        END IF  
        
        SET result TO an empty list  
        FOR digit d FROM nine DOWNTO zero  
            APPEND the string representation of digit d repeated the frequency of digit d in count times TO result  
        END FOR  
        
        SET final_number TO the concatenation of all elements in result  
        
        IF final_number is not empty AND the first character of final_number EQUALS zero  
            RETURN the string zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
