CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct WITH parameters s AND k  
        IF k EQUALS zero THEN  
            RETURN zero  
        END IF  
        SET char_map TO empty mapping  
        SET left TO zero  
        SET max_length TO zero  
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            SET the element at position right of s AS key IN char_map TO right  
            IF the NUMBER OF KEYS IN char_map GREATER THAN k THEN  
                SET leftmost_char TO the key IN char_map WITH the MINIMUM value AMONG ALL VALUES  
                SET left TO the value at key leftmost_char IN char_map PLUS one  
                REMOVE the element WITH key leftmost_char FROM char_map  
            END IF  
            SET max_length TO the GREATER VALUE BETWEEN max_length AND right MINUS left PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
