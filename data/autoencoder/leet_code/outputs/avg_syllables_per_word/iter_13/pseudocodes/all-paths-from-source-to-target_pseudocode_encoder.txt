CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        ASSIGN pointer q TO a new deque initialized with a list containing zero  
        SET ans TO an empty list  
        WHILE the deque q is NOT EMPTY  
            SET path TO the first element REMOVED FROM the deque q  
            SET u TO the element at the last position of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE TO the next iteration of the WHILE loop  
            END IF  
            FOR each element v IN the element at position u of graph  
                APPEND a new list formed by concatenating path followed by v TO the deque q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
