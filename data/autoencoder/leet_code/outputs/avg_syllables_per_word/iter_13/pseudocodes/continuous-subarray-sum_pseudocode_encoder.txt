CLASS Solution  
    FUNCTION checkSubarraySum(nums AS list of integer, k AS integer) RETURNS boolean  
        INITIALIZE dictionary remainder_map WITH key zero AND value negative one  
        SET current_sum TO zero  
        FOR each index i AND element num IN enumeration of nums  
            INCREMENT current_sum BY num  
            IF k IS NOT EQUAL TO zero THEN  
                SET current_sum TO current_sum MODULO k  
            END IF  
            IF current_sum IS A KEY IN remainder_map THEN  
                IF index i MINUS value AT key current_sum IN remainder_map IS GREATER THAN one THEN  
                    RETURN true  
                END IF  
            ELSE  
                ASSIGN key current_sum IN remainder_map TO index i  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
