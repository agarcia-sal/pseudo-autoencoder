CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS ONLY OF DIGITS THEN  
                SET num TO CONVERT block TO INTEGER  
                IF num IS LESS THAN OR EQUAL TO two hundred fifty five AND CONVERT num TO STRING EQUALS block THEN  
                    RETURN TRUE  
                ELSE  
                    RETURN FALSE  
                END IF  
            ELSE  
                RETURN FALSE  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF the LENGTH OF block IS GREATER THAN OR EQUAL TO one AND THE LENGTH OF block IS LESS THAN OR EQUAL TO four THEN  
                TRY  
                    CONVERT block TO INTEGER INTERPRETED AS BASE sixteen  
                    RETURN TRUE  
                EXCEPT WHEN CONVERSION FAILS  
                    RETURN FALSE  
                END TRY  
            ELSE  
                RETURN FALSE  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO the RESULT OF SPLITTING queryIP BY the CHARACTER dot  
        SET parts_ipv6 TO the RESULT OF SPLITTING queryIP BY the CHARACTER colon  

        IF the LENGTH OF parts_ipv4 EQUALS four AND FOR each part IN parts_ipv4 THE CALL TO is_valid_ipv4_block WITH part RETURNS TRUE THEN  
            RETURN the STRING IPv four  
        ELSE IF the LENGTH OF parts_ipv6 EQUALS eight AND FOR each part IN parts_ipv6 THE CALL TO is_valid_ipv6_block WITH part RETURNS TRUE THEN  
            RETURN the STRING IPv six  
        ELSE  
            RETURN the STRING Neither  
        END IF  
    END FUNCTION  
END CLASS
