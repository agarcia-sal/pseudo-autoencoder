CLASS Solution  
    FUNCTION maxAreaOfIsland(grid)  
        IF grid IS EQUAL TO empty  
            RETURN zero  
        END IF  
        
        SET max_area TO zero  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF the element at position zero of grid  
        
        FUNCTION dfs(r, c)  
            IF r IS LESS THAN zero OR r IS GREATER THAN OR EQUAL TO rows OR c IS LESS THAN zero OR c IS GREATER THAN OR EQUAL TO cols OR the element at position r of grid at position c IS EQUAL TO zero  
                RETURN zero  
            END IF  
            SET the element at position r of grid at position c TO zero  
            SET area TO one  
            INCREMENT area BY dfs(r PLUS one, c)  
            INCREMENT area BY dfs(r MINUS one, c)  
            INCREMENT area BY dfs(r, c PLUS one)  
            INCREMENT area BY dfs(r, c MINUS one)  
            RETURN area  
        END FUNCTION  
        
        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF the element at position r of grid at position c IS EQUAL TO one  
                    SET max_area TO THE GREATER BETWEEN max_area AND dfs(r, c)  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
