CLASS Solution  
    FUNCTION numMusicPlaylists(n, goal, k)  
        SET MODULO TO one billion PLUS seven  
        INITIALIZE dp AS a two dimensional list with goal PLUS one rows and n PLUS one columns filled with zero  
        SET element at position zero of dp at position zero TO one  
        FOR index_i FROM one TO goal  
            FOR index_j FROM one TO n  
                SET element at position index_i of dp at position index_j TO element at position index_i MINUS one of dp at position index_j MINUS one MULTIPLIED BY n MINUS index_j MINUS one PLUS one  
                SET element at position index_i of dp at position index_j TO element at position index_i of dp at position index_j MODULO MODULO  
                IF index_j GREATER THAN k  
                    INCREMENT element at position index_i of dp at position index_j BY element at position index_i MINUS one of dp at position index_j MULTIPLIED BY index_j MINUS k  
                    SET element at position index_i of dp at position index_j TO element at position index_i of dp at position index_j MODULO MODULO  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position goal of dp at position n  
    END FUNCTION  
END CLASS
