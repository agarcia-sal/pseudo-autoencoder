CLASS Solution  
    FUNCTION numRescueBoats(people list of integers, limit integer) RETURNS integer  
        CALL FUNCTION sort ON people list to order elements from smallest to largest  
        SET left TO zero  
        SET right TO the LENGTH OF people MINUS one  
        SET boats TO zero  
        WHILE left IS LESS THAN OR EQUAL TO right  
            IF element at position left of people PLUS element at position right of people IS LESS THAN OR EQUAL TO limit  
                INCREMENT left BY one  
            END IF  
            DECREMENT right BY one  
            INCREMENT boats BY one  
        END WHILE  
        RETURN boats  
    END FUNCTION  
END CLASS
