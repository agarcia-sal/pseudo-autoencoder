CLASS Solution  
    FUNCTION uniquePathsWithObstacles(obstacleGrid)  
        IF obstacleGrid DOES NOT EXIST OR element at position zero zero of obstacleGrid EQUALS one  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF obstacleGrid  
        SET n TO the LENGTH OF element at position zero of obstacleGrid  
        SET dp TO the two dimensional list OF zero WITH m ROWS AND n COLUMNS  
        SET element at position zero zero of dp TO one  
        FOR each column_index FROM one TO n MINUS one  
            IF element at position zero column_index of obstacleGrid EQUALS zero  
                SET element at position zero column_index of dp TO element at position zero column_index MINUS one of dp  
            END IF  
        END FOR  
        FOR each row_index FROM one TO m MINUS one  
            IF element at position row_index zero of obstacleGrid EQUALS zero  
                SET element at position row_index zero of dp TO element at position row_index MINUS one zero of dp  
            END IF  
        END FOR  
        FOR each row_index FROM one TO m MINUS one  
            FOR each column_index FROM one TO n MINUS one  
                IF element at position row_index column_index of obstacleGrid EQUALS zero  
                    SET element at position row_index column_index of dp TO element at position row_index MINUS one column_index of dp PLUS element at position row_index column_index MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one n MINUS one of dp  
    END FUNCTION  
END CLASS
