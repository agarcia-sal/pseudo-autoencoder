CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        SET tasks TO a list of pairs each consisting of an element at position index of plantTime together with an element at position index of growTime for each valid index  
        SORT tasks by the second element of each pair in descending order  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair consisting of plant and grow IN tasks  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF bloom_day IS GREATER THAN earliest_bloom  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
