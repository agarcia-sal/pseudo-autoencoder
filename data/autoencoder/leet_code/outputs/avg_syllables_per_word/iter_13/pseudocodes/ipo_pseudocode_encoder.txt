CLASS Solution  
    FUNCTION findMaximizedCapital(k INTEGER, w INTEGER, profits LIST OF INTEGER, capital LIST OF INTEGER) RETURNS INTEGER  
        SET projects TO the LIST OF TUPLES CREATED BY PAIRING each element of capital WITH the corresponding element of profits  
        SORT projects IN ASCENDING ORDER BASED ON the first element of each tuple representing capital required  
        
        SET max_heap TO an EMPTY LIST  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        
        FOR each repetition FROM one TO k  
            WHILE i LESS THAN n AND the first element at position i of projects LESS THAN OR EQUAL TO w  
                ADD the NEGATIVE of the second element at position i of projects TO max_heap USING the heap PUSH operation  
                INCREMENT i BY one  
            END WHILE  
            
            IF max_heap IS EMPTY  
                BREAK the loop  
            END IF  
            
            SET w TO w PLUS the NEGATIVE of the element REMOVED FROM max_heap USING the heap POP operation  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
