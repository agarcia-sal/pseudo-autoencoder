CLASS Solution  
    FUNCTION minEatingSpeed WITH PARAMETERS piles AND h  
        FUNCTION canFinish WITH PARAMETER k  
            SET hours TO zero  
            FOR each pile IN piles  
                INCREMENT hours BY the CEILING OF pile DIVIDED BY k  
            END FOR  
            IF hours IS LESS THAN OR EQUAL TO h  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET left TO one  
        SET right TO the MAXIMUM VALUE OF piles  

        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER RESULT OF left PLUS right DIVIDED BY two  
            IF canFinish WITH PARAMETER mid RETURNS true  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  

        RETURN left  
    END FUNCTION  
END CLASS
