CLASS Solution  
    FUNCTION maxVacationDays(flights AS List OF List OF integer, days AS List OF List OF integer) RETURNS integer  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF element at position zero of days  
          
        CREATE dp AS list OF integers WITH size n WHERE each element IS negative one  
        SET element at position zero of dp TO zero  
  
        CREATE graph AS mapping FROM integer TO list OF integers WITH keys FROM zero TO n MINUS one WHERE each value IS empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of element at position i of flights EQUALS one  
                    APPEND j TO element at position i of graph  
                END IF  
            END FOR  
        END FOR  
  
        FOR week FROM zero TO k MINUS one  
            CREATE new_dp AS list OF integers WITH size n WHERE each element IS negative one  
            FOR i FROM zero TO n MINUS one  
                IF element at position i of dp EQUALS negative one  
                    CONTINUE TO next i  
                END IF  
                SET element at position i of new_dp TO the maximum OF element at position i of new_dp AND element at position i of dp PLUS element at position week of element at position i of days  
                FOR each city j IN element at position i of graph  
                    SET element at position j of new_dp TO the maximum OF element at position j of new_dp AND element at position i of dp PLUS element at position week of element at position j of days  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
  
        RETURN the maximum element IN dp  
    END FUNCTION  
END CLASS
