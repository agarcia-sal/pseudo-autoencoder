CLASS Solution  
    FUNCTION getHint(secret AS string, guess AS string) RETURNS string  
        IMPORT defaultdict FROM collections  
        SET bulls TO zero  
        SET cows TO zero  
        SET secret_count TO a new defaultdict of integer  
        SET guess_count TO a new defaultdict of integer  
        FOR each pair of characters s AND g IN secret AND guess simultaneously  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value at key s IN secret_count BY one  
                INCREMENT the value at key g IN guess_count BY one  
            END IF  
        END FOR  
        FOR each digit IN keys of secret_count  
            INCREMENT cows BY the minimum of the value at key digit IN secret_count AND the value at key digit IN guess_count  
        END FOR  
        RETURN a string composed of bulls FOLLOWED BY uppercase letter A FOLLOWED BY cows FOLLOWED BY uppercase letter B  
    END FUNCTION  
END CLASS
