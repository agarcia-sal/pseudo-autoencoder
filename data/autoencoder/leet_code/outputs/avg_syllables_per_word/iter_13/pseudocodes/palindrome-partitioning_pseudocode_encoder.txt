CLASS Solution  
    FUNCTION partition WITH PARAMETERS s  
        FUNCTION is_palindrome WITH PARAMETER sub  
            RETURN sub EQUALS THE REVERSED SEQUENCE OF sub  
        END FUNCTION  
        
        FUNCTION backtrack WITH PARAMETERS start AND path  
            IF start EQUALS THE LENGTH OF s THEN  
                APPEND path TO result  
                RETURN  
            END IF  
            FOR end FROM start PLUS one TO THE LENGTH OF s PLUS one  
                IF is_palindrome OF THE SUBSTRING OF s FROM POSITION start TO end THEN  
                    CALL backtrack WITH PARAMETERS end AND THE CONCATENATION OF path AND THE SUBSTRING OF s FROM POSITION start TO end  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL backtrack WITH PARAMETERS zero AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
