CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero COMMA left EQUALS none COMMA right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH value AT position zero of values  
    SET i TO one  
    SET queue TO NEW deque  
    CALL APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element REMOVED FROM FRONT OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH value at position i of values  
            CALL APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH value at position i of values  
            CALL APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p COMMA q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS TO q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH p dot left AND q dot left AND is_same_tree WITH p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rob(root)  
        FUNCTION rob_from_node(node)  
            IF node IS none  
                RETURN zero COMMA zero  
            END IF  
            SET left_robbed COMMA left_not_robbed TO rob_from_node WITH node dot left  
            SET right_robbed COMMA right_not_robbed TO rob_from_node WITH node dot right  
            SET rob_this_node TO node dot val PLUS left_not_robbed PLUS right_not_robbed  
            SET not_rob_this_node TO the MAXIMUM OF left_robbed AND left_not_robbed PLUS the MAXIMUM OF right_robbed AND right_not_robbed  
            RETURN rob_this_node COMMA not_rob_this_node  
        END FUNCTION  
        RETURN the MAXIMUM OF rob_from_node WITH root  
    END FUNCTION  
END CLASS
