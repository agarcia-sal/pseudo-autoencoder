CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero COMMA left EQUALS None COMMA right EQUALS None  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO A NEW EMPTY double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p COMMA q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS TO q.val  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS p.left COMMA q.left AND the RESULT OF is_same_tree WITH PARAMETERS p.right COMMA q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue WITH PARAMETERS self COMMA root  
        SET queue TO A NEW double ended queue INITIALIZED WITH root AS THE ONLY ELEMENT  
        SET leftmost_value TO None  
        WHILE queue IS NOT EMPTY  
            SET level_size TO the LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one  
                SET node TO THE FIRST ELEMENT REMOVED FROM queue  
                IF i EQUALS zero  
                    SET leftmost_value TO node.val  
                END IF  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
