CLASS Solution  
    FUNCTION checkValidString WITH parameter s  
        SET low TO zero  
        SET high TO zero  
        FOR each character IN s  
            IF character EQUALS left parenthesis  
                INCREMENT low BY one  
                INCREMENT high BY one  
            ELSE IF character EQUALS right parenthesis  
                DECREMENT low BY one  
                DECREMENT high BY one  
            ELSE  
                DECREMENT low BY one  
                INCREMENT high BY one  
            END IF  
            IF high LESS THAN zero  
                RETURN false  
            END IF  
            SET low TO the GREATER OF low AND zero  
        END FOR  
        RETURN whether low EQUALS zero  
    END FUNCTION  
END CLASS
