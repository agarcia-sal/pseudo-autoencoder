CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero left EQUALS None right EQUALS None  
        SET the val ATTRIBUTE OF self TO val  
        SET the left ATTRIBUTE OF self TO left  
        SET the right ATTRIBUTE OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF the list values IS empty  
        RETURN None  
    END IF  
    SET root TO A NEW INSTANCE OF TreeNode WITH THE element at position zero OF values AS val  
    SET i TO one  
    SET queue TO A NEW DEQUE DATA STRUCTURE  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE AND RETURN the FIRST ELEMENT FROM queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT None  
            SET the left ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode WITH THE element at position i OF values AS val  
            APPEND the left ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT None  
            SET the right ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode WITH THE element at position i OF values AS val  
            APPEND the right ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUALS TO the val ATTRIBUTE OF q  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS the left ATTRIBUTE OF p AND the left ATTRIBUTE OF q AND AND is_same_tree WITH PARAMETERS the right ATTRIBUTE OF p AND the right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode WITH PARAMETERS self root AS Optional TreeNode key AS integer RETURNING Optional TreeNode  
        IF root IS None  
            RETURN None  
        END IF  
        IF the val ATTRIBUTE OF root IS GREATER THAN key  
            SET the left ATTRIBUTE OF root TO self.deleteNode WITH PARAMETERS the left ATTRIBUTE OF root AND key  
            RETURN root  
        END IF  
        IF the val ATTRIBUTE OF root IS LESS THAN key  
            SET the right ATTRIBUTE OF root TO self.deleteNode WITH PARAMETERS the right ATTRIBUTE OF root AND key  
            RETURN root  
        END IF  
        IF the left ATTRIBUTE OF root IS None  
            RETURN the right ATTRIBUTE OF root  
        END IF  
        IF the right ATTRIBUTE OF root IS None  
            RETURN the left ATTRIBUTE OF root  
        END IF  
        SET node TO the right ATTRIBUTE OF root  
        WHILE the left ATTRIBUTE OF node IS NOT None  
            SET node TO the left ATTRIBUTE OF node  
        END WHILE  
        SET the left ATTRIBUTE OF node TO the left ATTRIBUTE OF root  
        SET root TO the right ATTRIBUTE OF root  
        RETURN root  
    END FUNCTION  
END CLASS
