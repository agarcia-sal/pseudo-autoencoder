CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1 AS List of Integer, arr2 AS List of Integer) RETURNS Integer  
        SORT arr2 IN ASCENDING ORDER  
        SET dp TO A DICTIONARY WITH KEY negative one AND VALUE zero  
        
        FOR each num IN arr1  
            SET new_dp TO AN EMPTY DICTIONARY  
            FOR each pair OF last_num AND ops IN dp  
                IF num IS GREATER THAN last_num  
                    SET the value at key num IN new_dp TO THE MINIMUM OF the CURRENT VALUE AT key num IN new_dp OR infinity AND ops  
                END IF  
                
                SET idx TO THE POSITION WHERE last_num COULD BE INSERTED TO THE RIGHT IN arr2 TO MAINTAIN ASCENDING ORDER USING BINARY SEARCH  
                IF idx IS LESS THAN THE LENGTH OF arr2  
                    SET the value at key element at position idx of arr2 IN new_dp TO THE MINIMUM OF the CURRENT VALUE AT THAT KEY IN new_dp OR infinity AND ops PLUS one  
                END IF  
            END FOR  
            
            IF new_dp IS EMPTY  
                RETURN negative one  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN THE MINIMUM VALUE AMONG ALL VALUES OF dp  
    END FUNCTION  
END CLASS
