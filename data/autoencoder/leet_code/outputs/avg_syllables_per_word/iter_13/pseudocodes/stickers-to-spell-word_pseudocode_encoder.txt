CLASS Solution  
    FUNCTION minStickers(stickers AS List OF strings, target AS string) RETURNS integer  
        SET stickers TO a list of frequency count collections constructed from each element in stickers  
        SORT stickers such that the element with the maximum frequency count of any character is first in descending order  
        SET queue TO a double ended queue initialized with a pair consisting of the target string and zero used stickers  
        SET memo TO a dictionary associating the target string with zero  
  
        WHILE the queue is not empty  
            REMOVE and retrieve the first pair from the queue assigning to current_target and used_stickers  
  
            FOR each sticker IN stickers  
                IF the element at position one of current_target is NOT present in sticker  
                    CONTINUE to next sticker  
                END IF  
  
                SET remaining TO a frequency count collection constructed from current_target  
                SUBTRACT the counts of sticker from remaining  
                SET new_target TO the string constructed by sorting the elements represented in remaining  
  
                IF new_target is empty  
                    RETURN used_stickers PLUS one  
                END IF  
  
                IF new_target is NOT present in memo  
                    ASSIGN to memo at new_target the value used_stickers PLUS one  
                    APPEND the pair consisting of new_target and used_stickers PLUS one to the queue  
                END IF  
            END FOR  
        END WHILE  
  
        RETURN negative one  
    END FUNCTION  
END CLASS
