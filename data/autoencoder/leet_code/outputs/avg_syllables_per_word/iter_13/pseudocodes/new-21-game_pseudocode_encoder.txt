CLASS Solution  
    FUNCTION new21Game WITH PARAMETERS n INTEGER k INTEGER maxPts INTEGER  
        FUNCTION dfs WITH PARAMETER i INTEGER  
            IF i GREATER THAN OR EQUAL TO k  
                RETURN one IF i LESS THAN OR EQUAL TO n OTHERWISE zero  
            END IF  
            IF i EQUALS k MINUS one  
                RETURN the SMALLER OF n MINUS k PLUS one AND maxPts DIVIDED BY maxPts  
            END IF  
            RETURN dfs WITH i PLUS one PLUS the DIFFERENCE OF dfs WITH i PLUS one AND dfs WITH i PLUS maxPts PLUS one DIVIDED BY maxPts  
        END FUNCTION  
        RETURN dfs WITH zero  
END CLASS
