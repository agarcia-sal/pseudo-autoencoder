CLASS Solution  
    FUNCTION racecar(target)  
        SET queue TO a new deque initialized with a triple of zero position one speed and zero steps  
        SET visited TO a new set initialized with a pair of zero position and one speed  
        WHILE queue has elements  
            REMOVE the element at the front of queue and UNPACK into position speed and steps  
            IF position EQUALS target  
                RETURN steps  
            END IF  
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the pair of next_position and next_speed is NOT IN visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                ADD the pair of next_position and next_speed TO visited  
                APPEND the triple of next_position next_speed and steps PLUS one TO queue  
            END IF  
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the pair of position and next_speed is NOT IN visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                ADD the pair of position and next_speed TO visited  
                APPEND the triple of position next_speed and steps PLUS one TO queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
