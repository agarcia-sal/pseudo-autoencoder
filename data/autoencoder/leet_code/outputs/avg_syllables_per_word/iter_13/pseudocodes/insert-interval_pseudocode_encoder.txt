CLASS Solution  
    FUNCTION insert(intervals AS List OF List OF integers newInterval AS List OF integers) RETURNS List OF List OF integers  
        IF intervals EQUALS empty THEN  
            RETURN a list containing newInterval  
        END IF  
        
        SET merged TO empty list  
        SET i TO zero  
        SET n TO the LENGTH OF intervals  
        
        WHILE i LESS THAN n AND element at position i of intervals at position one LESS THAN the element at position zero of newInterval holds  
            APPEND element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE i LESS THAN n AND element at position i of intervals at position zero LESS THAN OR EQUAL TO the element at position one of newInterval holds  
            SET element at position zero of newInterval TO the smaller value BETWEEN element at position zero of newInterval AND element at position zero of intervals at position i  
            SET element at position one of newInterval TO the larger value BETWEEN element at position one of newInterval AND element at position one of intervals at position i  
            INCREMENT i BY one  
        END WHILE  
        
        APPEND newInterval TO merged  
        
        WHILE i LESS THAN n  
            APPEND element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN merged  
    END FUNCTION  
END CLASS
