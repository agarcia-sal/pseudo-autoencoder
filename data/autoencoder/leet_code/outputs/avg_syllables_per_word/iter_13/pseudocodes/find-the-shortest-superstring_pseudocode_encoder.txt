CLASS Solution  
    FUNCTION shortestSuperstring(words)  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap(i, j)  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i of words AND the LENGTH OF element at position j of words DOWN TO one  
                IF the SUBSTRING from position LENGTH OF element at position i of words MINUS k PLUS one TO the LENGTH OF element at position i of words of element at position i of words EQUALS the SUBSTRING from position one TO k of element at position j of words  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp(mask, i)  
            IF mask EQUALS one SHIFTED LEFT BY n MINUS one  
                RETURN zero AND empty string  
            END IF  
            
            SET min_len TO positive infinity  
            SET best_path TO empty string  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND of mask AND one SHIFTED LEFT BY j EQUALS zero  
                    SET length AND path TO the RESULT OF dp with the BITWISE OR of mask AND one SHIFTED LEFT BY j AND parameter j  
                    SET length TO length PLUS the LENGTH OF element at position j of words MINUS the RESULT OF overlap with parameters i AND j  
                    IF length LESS THAN min_len  
                        SET min_len TO length  
                        SET best_path TO the SUBSTRING from position overlap of i AND j PLUS one TO the LENGTH OF element at position j of words of element at position j of words CONCATENATED WITH path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO positive infinity  
        SET shortest_path TO empty string  
        FOR i FROM zero TO n MINUS one  
            SET length AND path TO the RESULT OF dp with one SHIFTED LEFT BY i AND parameter i  
            SET length TO length PLUS the LENGTH OF element at position i of words  
            IF length LESS THAN min_len  
                SET min_len TO length  
                SET shortest_path TO element at position i of words CONCATENATED WITH path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
