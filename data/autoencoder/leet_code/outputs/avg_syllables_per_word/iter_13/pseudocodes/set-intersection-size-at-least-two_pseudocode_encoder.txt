CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        CALL sort intervals BY the element at position one of each interval FOLLOWED BY the NEGATION of the element at position zero of each interval  
        SET first TO the element at position one of the interval at position zero MINUS one  
        SET second TO the element at position one of the interval at position zero  
        SET count TO two  
        FOR each interval with start as the element at position zero AND end as the element at position one IN the collection of intervals starting from position one TO the end  
            IF start IS GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
