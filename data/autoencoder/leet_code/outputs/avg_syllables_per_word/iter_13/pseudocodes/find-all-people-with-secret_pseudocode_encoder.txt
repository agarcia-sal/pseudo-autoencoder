CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO a set containing zero AND firstPerson  
        SORT meetings BY the value at position three of each meeting in ascending order  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF meetings  
            SET current_time TO the value at position three of the meeting at position i in meetings  
            SET graph TO a new mapping from person TO a list of connected persons  
            WHILE i IS LESS THAN the LENGTH OF meetings AND the value at position three of the meeting at position i EQUALS current_time  
                SET x TO the value at position one of the meeting at position i in meetings  
                SET y TO the value at position two of the meeting at position i in meetings  
                APPEND y TO the list associated with x in graph  
                APPEND x TO the list associated with y in graph  
                INCREMENT i BY one  
            END WHILE  
            SET queue TO a double ended queue containing each person from know_secret THAT is a key in graph  
            SET visited TO an empty set  
            WHILE queue is not empty  
                REMOVE the person from the front of queue  
                IF person is in visited  
                    CONTINUE to the next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list associated with person in graph  
                    IF neighbor is not in visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN a list containing all elements of know_secret  
    END FUNCTION  
END CLASS
