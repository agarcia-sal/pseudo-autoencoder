CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven  
        SET dp TO a list of zeros of LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS star  
                RETURN nine  
            ELSE IF char EQUALS zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS star AND char2 EQUALS star  
                RETURN fifteen  
            ELSE IF char1 EQUALS star  
                IF char2 LESS THAN OR EQUAL TO six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS star  
                IF char1 EQUALS one  
                    RETURN nine  
                ELSE IF char1 EQUALS two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO integer CONVERSION OF char1 CONCATENATED WITH char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
        
        FOR index FROM one TO LENGTH OF s  
            SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS one of dp MULTIPLIED BY ways_to_decode_single(element at position index MINUS one of s)  
            SET element at position index of dp TO element at position index of dp MODULO MOD  
            
            IF index GREATER THAN one  
                SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS two of dp MULTIPLIED BY ways_to_decode_pair(element at position index MINUS two of s, element at position index MINUS one of s)  
                SET element at position index of dp TO element at position index of dp MODULO MOD  
            END IF  
        END FOR  
        
        RETURN element at last position of dp  
    END FUNCTION  
END CLASS
