CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val OPTIONAL defaulting TO zero, left OPTIONAL defaulting TO none, right OPTIONAL defaulting TO none  
        SET the attribute val OF self TO val  
        SET the attribute left OF self TO left  
        SET the attribute right OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF the values list IS empty  
        RETURN none  
    END IF  
    SET root TO A new instance OF TreeNode initialized WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE the element at the front OF queue AND SET TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left attribute OF node TO A new instance OF TreeNode initialized WITH the element at position i of values  
            APPEND the left attribute OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right attribute OF node TO A new instance OF TreeNode initialized WITH the element at position i of values  
            APPEND the right attribute OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the val attribute OF p IS NOT EQUAL TO the val attribute OF q  
        RETURN False  
    ELSE  
        RETURN the result OF calling is_same_tree WITH the left attribute OF p AND the left attribute OF q AND the result OF calling is_same_tree WITH the right attribute OF p AND the right attribute OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder WITH PARAMETERS self AND root OPTIONAL OF TYPE TreeNode  
        IF root IS none  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a double ended queue initialized WITH root  
        WHILE queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR index FROM zero TO level_size MINUS one  
                REMOVE the element at the front OF queue AND SET TO node  
                APPEND the val attribute OF node TO current_level  
                IF the left attribute OF node IS NOT none  
                    APPEND the left attribute OF node TO queue  
                END IF  
                IF the right attribute OF node IS NOT none  
                    APPEND the right attribute OF node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
