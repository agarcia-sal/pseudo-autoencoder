CLASS Solution  
    FUNCTION findMinHeightTrees(n, edges)  
        IF n EQUALS one  
            RETURN list containing zero  
        END IF  
        
        SET graph TO dictionary mapping each integer key TO list of integers  
        SET degree TO list of n zeros  
        
        FOR each pair of integers u and v IN edges  
            APPEND v TO list at key u IN graph  
            APPEND u TO list at key v IN graph  
            INCREMENT element at position u IN degree BY one  
            INCREMENT element at position v IN degree BY one  
        END FOR  
        
        SET leaves TO double ended queue initialized with all integers i FROM zero TO n MINUS one WHERE element at position i IN degree EQUALS one  
        
        SET remaining_nodes TO n  
        WHILE remaining_nodes GREATER THAN two  
            SET leaves_count TO the LENGTH OF leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR each iteration FROM one TO leaves_count  
                REMOVE element from the left of leaves AND ASSIGN TO leaf  
                FOR each neighbor IN list at key leaf IN graph  
                    DECREMENT element at position neighbor IN degree BY one  
                    IF element at position neighbor IN degree EQUALS one  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN list converted from leaves  
    END FUNCTION  
END CLASS
