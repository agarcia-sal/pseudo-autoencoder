CLASS Solution  
    FUNCTION minimumWhiteTiles(floor NUM OF CHARACTERS, numCarpets NUMERIC, carpetLen NUMERIC) RETURNS NUMERIC  
        SET n TO the LENGTH OF floor  
        SET dp TO a two dimensional list OF zero with n PLUS one ROWS and numCarpets PLUS one COLUMNS  
        FOR i FROM one TO n  
            SET dp at position i at position zero TO the value at position i MINUS one at position zero PLUS the result of checking IF the element at position i MINUS one of floor EQUALS the character one THEN one OTHERWISE zero  
        END FOR  
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp at position i at position j TO the value at position i MINUS one at position j PLUS the result of checking IF the element at position i MINUS one of floor EQUALS the character one THEN one OTHERWISE zero  
                SET dp at position i at position j TO the MINIMUM of the current dp value at position i at position j AND the value at position of the MAXIMUM of zero AND i MINUS carpetLen at position j MINUS one  
            END FOR  
        END FOR  
        RETURN the value at position n at position numCarpets of dp  
    END FUNCTION  
END CLASS
