CLASS Solution  
    FUNCTION maxNumber(nums1, nums2, k)  
        FUNCTION maxSingleArray(nums, k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO empty list  
            FOR each number IN nums  
                WHILE drop greater than zero AND stack not empty AND element at last position of stack LESS THAN number  
                    REMOVE element from last position of stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND number TO stack  
            END FOR  
            RETURN the SUBLIST of stack FROM position zero TO position k minus one inclusive  
        END FUNCTION  
  
        FUNCTION merge(arr1, arr2)  
            SET result TO empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR position FROM zero TO total_length minus one inclusive  
                IF arr1 GREATER THAN arr2 in LEXICOGRAPHICAL ORDER  
                    APPEND the element at position zero of arr1 TO result  
                    REMOVE the element at position zero from arr1  
                ELSE  
                    APPEND the element at position zero of arr2 TO result  
                    REMOVE the element at position zero from arr2  
                END IF  
            END FOR  
            RETURN result  
        END FUNCTION  
  
        SET max_number TO empty list  
        FOR i FROM zero TO k inclusive  
            IF i LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO merge THE RESULT OF maxSingleArray WITH nums1 AND i AND THE RESULT OF maxSingleArray WITH nums2 AND k MINUS i  
                IF candidate GREATER THAN max_number in LEXICOGRAPHICAL ORDER  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
