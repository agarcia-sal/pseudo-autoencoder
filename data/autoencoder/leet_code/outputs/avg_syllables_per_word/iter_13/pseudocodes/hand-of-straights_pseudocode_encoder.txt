CLASS Solution  
    FUNCTION isNStraightHand(hand AS List OF integers, groupSize AS integer) AS boolean  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN false  
        END IF  
        
        SET count TO a mapping FROM each unique card IN hand TO its frequency USING a counting collection  
        SET unique_cards TO the collection OF unique cards in the keys of count SORTED in ascending order  
        
        FOR each card IN unique_cards  
            IF the frequency of card IN count GREATER THAN zero THEN  
                SET needed TO the frequency of card IN count  
                FOR index FROM zero TO groupSize MINUS one  
                    IF the frequency of the element at position equal to card PLUS index IN count LESS THAN needed THEN  
                        RETURN false  
                    END IF  
                    SET the frequency of the element at position equal to card PLUS index IN count TO the frequency MINUS needed  
                END FOR  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
