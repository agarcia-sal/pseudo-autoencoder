CLASS Solution  
    FUNCTION lengthOfLongestSubstring WITH parameter s  
        SET char_index_map TO empty mapping from characters to indices  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each position end AND character char IN enumeration of the string s  
            IF the character char IS PRESENT IN char_index_map AND the value at key char in char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO the value at key char in char_index_map PLUS one  
            END IF  
            SET the value at key char in char_index_map TO end  
            SET max_length TO the GREATER OF max_length AND end MINUS start PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
