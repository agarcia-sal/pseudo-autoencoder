CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO INSTANTIATION OF TreeNode WITH PARAMETER element at position zero of values  
    SET i TO one  
    SET queue TO a new DEQUE collection  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO INSTANTIATION OF TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO INSTANTIATION OF TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS TO q dot val  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS p dot left AND q dot left AND the RESULT OF is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree WITH PARAMETERS self nums  
        IF the LENGTH OF nums EQUALS zero  
            RETURN none  
        END IF  
        SET max_val TO the MAXIMUM VALUE IN nums  
        SET max_index TO the POSITION OF max_val IN nums  
        SET root TO INSTANTIATION OF TreeNode WITH PARAMETER max_val  
        SET root dot left TO the RESULT OF self dot constructMaximumBinaryTree WITH PARAMETER the SUBLIST OF nums FROM position zero TO position max_index MINUS one  
        SET root dot right TO the RESULT OF self dot constructMaximumBinaryTree WITH PARAMETER the SUBLIST OF nums FROM position max_index PLUS one TO the END OF nums  
        RETURN root  
    END FUNCTION  
END CLASS
