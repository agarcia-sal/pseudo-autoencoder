CLASS Solution  
    FUNCTION minSubArrayLen(target INTEGER, nums LIST OF INTEGER) RETURNS INTEGER  
        SET left TO zero  
        SET current_sum TO zero  
        SET min_length TO positive infinity  
        
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            SET current_sum TO current_sum PLUS the element at position right of nums  
            
            WHILE current_sum GREATER THAN OR EQUAL TO target  
                SET min_length TO the MINIMUM of min_length AND the result of right MINUS left PLUS one  
                SET current_sum TO current_sum MINUS the element at position left of nums  
                INCREMENT left BY one  
            END WHILE  
        END FOR  
        
        IF min_length NOT EQUALS positive infinity  
            RETURN min_length  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
