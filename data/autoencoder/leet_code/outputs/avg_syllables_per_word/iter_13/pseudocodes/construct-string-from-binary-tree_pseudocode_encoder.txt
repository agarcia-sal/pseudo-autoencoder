CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero, left EQUALS none, right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH INITIAL VALUE element at position zero of values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element popped from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH INITIAL VALUE element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH INITIAL VALUE element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val IS NOT EQUAL TO q dot val  
        RETURN false  
    ELSE  
        RETURN logical AND of is_same_tree WITH PARAMETERS p dot left, q dot left AND is_same_tree WITH PARAMETERS p dot right, q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str WITH PARAMETERS root AS optional TreeNode  
        FUNCTION helper WITH PARAMETER node  
            IF node IS none  
                RETURN empty string  
            END IF  
            SET result TO string conversion of node dot val  
            IF node dot left IS NOT none OR (node dot left IS none AND node dot right IS NOT none)  
                SET result TO result CONCATENATED WITH opening parenthesis CONCATENATED WITH helper WITH PARAMETER node dot left CONCATENATED WITH closing parenthesis  
            END IF  
            IF node dot right IS NOT none  
                SET result TO result CONCATENATED WITH opening parenthesis CONCATENATED WITH helper WITH PARAMETER node dot right CONCATENATED WITH closing parenthesis  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN helper WITH PARAMETER root  
    END FUNCTION  
END CLASS
