CLASS Solution  
    FUNCTION minInsertions(s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO the LENGTH OF s  
            SET dp TO a two dimensional array of size n BY n initialized with zero  
            
            FOR each index i FROM zero TO n MINUS one  
                SET the element at position i at position i of dp TO one  
            END FOR  
            
            FOR each substring_length FROM two TO n  
                FOR each start_index FROM zero TO n MINUS substring_length  
                    SET end_index TO start_index PLUS substring_length MINUS one  
                    IF the character at position start_index of s EQUALS the character at position end_index of s AND substring_length EQUALS two  
                        SET the element at position start_index at position end_index of dp TO two  
                    ELSE IF the character at position start_index of s EQUALS the character at position end_index of s  
                        SET the element at position start_index at position end_index of dp TO the element at position start_index PLUS one at position end_index MINUS one of dp PLUS two  
                    ELSE  
                        SET the element at position start_index at position end_index of dp TO the GREATER VALUE BETWEEN the element at position start_index at position end_index MINUS one of dp AND the element at position start_index PLUS one at position end_index of dp  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN the element at position zero at position n MINUS one of dp  
        END FUNCTION  
        
        RETURN the LENGTH OF s MINUS longest_palindromic_subsequence(s)  
    END FUNCTION  
END CLASS
