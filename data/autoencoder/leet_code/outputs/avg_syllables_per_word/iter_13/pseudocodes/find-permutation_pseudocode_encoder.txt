CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO list of integers starting from one INCREMENTED BY one UP TO THE LENGTH OF s PLUS one  
        SET start TO zero  
        FOR end FROM zero TO THE LENGTH OF s  
            IF end EQUALS THE LENGTH OF s OR element at position end of s EQUALS THE CHARACTER I  
                REVERSE the segment of result FROM position start TO end PLUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
