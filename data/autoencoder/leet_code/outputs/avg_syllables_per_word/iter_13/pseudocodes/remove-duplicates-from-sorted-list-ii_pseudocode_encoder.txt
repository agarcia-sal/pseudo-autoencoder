CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the val PROPERTY OF self TO val  
        SET the next PROPERTY OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO the last position  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET the next PROPERTY OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the val PROPERTY OF p1 EQUALS the val PROPERTY OF p2 AND is_same_list WITH PARAMETERS the next PROPERTY OF p1 AND the next PROPERTY OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates WITH PARAMETERS head  
        SET dummy TO A NEW ListNode INITIALIZED WITH zero AND head  
        SET prev TO dummy  
        WHILE head IS NOT None  
            IF the next PROPERTY OF head IS NOT None AND the val PROPERTY OF head EQUALS the val PROPERTY OF the next PROPERTY OF head  
                WHILE the next PROPERTY OF head IS NOT None AND the val PROPERTY OF head EQUALS the val PROPERTY OF the next PROPERTY OF head  
                    SET head TO the next PROPERTY OF head  
                END WHILE  
                SET the next PROPERTY OF prev TO the next PROPERTY OF head  
            ELSE  
                SET prev TO the next PROPERTY OF prev  
            END IF  
            SET head TO the next PROPERTY OF head  
        END WHILE  
        RETURN the next PROPERTY OF dummy  
    END FUNCTION  
END CLASS
