CLASS Solution  
    FUNCTION findLonelyPixel(picture)  
        IF picture EQUALS nothing OR element at position zero of picture EQUALS nothing  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO a list of zero values with the LENGTH OF m  
        SET col_count TO a list of zero values with the LENGTH OF n  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF element at position i of picture EQUALS nothing THEN CONTINUE  
                IF element at position j of element at position i of picture EQUALS the character B  
                    INCREMENT element at position i of row_count BY one  
                    INCREMENT element at position j of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        SET lonely_pixel_count TO zero  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF element at position j of element at position i of picture EQUALS the character B AND element at position i of row_count EQUALS one AND element at position j of col_count EQUALS one  
                    INCREMENT lonely_pixel_count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
