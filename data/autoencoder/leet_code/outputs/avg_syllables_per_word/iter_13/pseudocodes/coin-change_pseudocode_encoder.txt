CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        SET dp TO a list of length amount plus one with each element set to amount plus one  
        SET element at position zero of dp TO zero  
        FOR each integer a FROM one TO amount  
            FOR each coin IN coins  
                IF the value a MINUS coin is greater than or equal to zero  
                    SET element at position a of dp TO the lesser value BETWEEN element at position a of dp AND element at position a MINUS coin of dp PLUS one  
                END IF  
            END FOR  
        END FOR  
        IF element at position amount of dp IS NOT EQUAL TO amount plus one  
            RETURN element at position amount of dp  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
