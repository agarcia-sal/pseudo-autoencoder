CLASS Solution  
    FUNCTION computeArea(ax1 ax1 INTEGER ay1 ay1 INTEGER ax2 ax2 INTEGER ay2 ay2 INTEGER bx1 bx1 INTEGER by1 by1 INTEGER bx2 bx2 INTEGER by2 by2 INTEGER) RETURNS INTEGER  
        SET area1 TO ax2 MINUS ax1 MULTIPLIED BY ay2 MINUS ay1  
        SET area2 TO bx2 MINUS bx1 MULTIPLIED BY by2 MINUS by1  
        SET overlap_width TO THE MAXIMUM OF zero AND THE MINIMUM OF ax2 AND bx2 MINUS THE MAXIMUM OF ax1 AND bx1  
        SET overlap_height TO THE MAXIMUM OF zero AND THE MINIMUM OF ay2 AND by2 MINUS THE MAXIMUM OF ay1 AND by1  
        SET overlap_area TO overlap_width MULTIPLIED BY overlap_height  
        SET total_area TO area1 PLUS area2 MINUS overlap_area  
        RETURN total_area  
    END FUNCTION  
END CLASS
