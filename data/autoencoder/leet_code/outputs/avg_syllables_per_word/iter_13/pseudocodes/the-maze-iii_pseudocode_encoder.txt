    FUNCTION findShortestWay WITH PARAMETERS maze AS LIST OF LIST OF INTEGER, ball AS LIST OF INTEGER, hole AS LIST OF INTEGER  
        SET directions TO LIST OF FOUR ELEMENTS WHERE EACH ELEMENT IS A TRIPLE CONTAINING integer zero AND integer one AND character r or d or l or u REPRESENTING MOVEMENT TO THE RIGHT DOWN LEFT AND UP RESPECTIVELY  
        SET m TO THE LENGTH OF maze  
        SET n TO THE LENGTH OF element at position zero of maze  
        SET start TO A TUPLE CONTAINING THE ELEMENTS OF ball  
        SET hole TO A TUPLE CONTAINING THE ELEMENTS OF hole  
        SET pq TO A LIST CONTAINING A TRIPLE OF ZERO DISTANCE AN EMPTY STRING PATH AND start POSITION  
        SET visited TO AN EMPTY SET  
        
        WHILE pq IS NOT EMPTY  
            REMOVE AND OBTAIN THE ELEMENT WITH THE SMALLEST DISTANCE AND ASSOCIATED path AND POSITION x AND y FROM pq  
            
            IF TUPLE OF x AND y EQUALS hole  
                RETURN path  
            END IF  
            
            IF TUPLE OF x AND y IS IN visited  
                CONTINUE TO THE NEXT ITERATION OF THE WHILE LOOP  
            END IF  
            
            ADD TUPLE OF x AND y TO visited  
            
            FOR EACH dx dy direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO ZERO  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO ZERO AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO ZERO AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of element at position new_y PLUS dy of maze EQUALS ZERO  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY ONE  
                    IF TUPLE OF new_x AND new_y EQUALS hole  
                        BREAK OUT OF THE WHILE LOOP  
                    END IF  
                END WHILE  
                
                IF TUPLE OF new_x AND new_y IS NOT IN visited  
                    ADD A TRIPLE OF dist PLUS count AND path CONCATENATED WITH direction AND TUPLE OF new_x AND new_y TO pq USING HEAP PUSH OPERATION  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN THE STRING impossible  
    END FUNCTION  
END CLASS
