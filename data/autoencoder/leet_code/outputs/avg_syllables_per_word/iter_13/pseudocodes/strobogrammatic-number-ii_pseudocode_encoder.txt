CLASS Solution  
    FUNCTION findStrobogrammatic WITH PARAMETER n  
        FUNCTION dfs WITH PARAMETER u  
            IF u EQUALS zero  
                RETURN list containing an empty string  
            END IF  
            IF u EQUALS one  
                RETURN list containing string zero string one string eight  
            END IF  
            SET ans TO empty list  
            FOR each v IN the RESULT OF THE CALL TO dfs WITH PARAMETER u MINUS two  
                FOR each pair l r IN the set of string pairs one one string eight eight string six nine string nine six  
                    APPEND the concatenation of l followed by v followed by r TO ans  
                END FOR  
                IF u NOT EQUAL TO n  
                    APPEND the concatenation of string zero followed by v followed by string zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN RESULT OF THE CALL TO dfs WITH PARAMETER n  
    END FUNCTION  
END CLASS
