CLASS Solution  
    FUNCTION totalNQueens WITH PARAMETER n  
        FUNCTION is_not_under_attack WITH PARAMETERS row AND col  
            RETURN NOT the SUM of element at position col of collection cols PLUS element at position row MINUS col of collection hills PLUS element at position row PLUS col of collection dales IS GREATER THAN ZERO  
        END FUNCTION  
        FUNCTION place_queen WITH PARAMETERS row AND col  
            SET element at position col of collection cols TO one  
            SET element at position row MINUS col of collection hills TO one  
            SET element at position row PLUS col of collection dales TO one  
        END FUNCTION  
        FUNCTION remove_queen WITH PARAMETERS row AND col  
            SET element at position col of collection cols TO zero  
            SET element at position row MINUS col of collection hills TO zero  
            SET element at position row PLUS col of collection dales TO zero  
        END FUNCTION  
        FUNCTION backtrack WITH PARAMETERS row EQUAL TO zero AND count EQUAL TO zero  
            FOR each col FROM zero TO n MINUS one  
                IF is_not_under_attack WITH PARAMETERS row AND col EQUALS true  
                    CALL place_queen WITH PARAMETERS row AND col  
                    IF row PLUS one EQUALS n  
                        INCREMENT count BY one  
                    ELSE  
                        SET count TO backtrack WITH PARAMETERS row PLUS one AND count  
                    END IF  
                    CALL remove_queen WITH PARAMETERS row AND col  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET cols TO a list of zeros of LENGTH n  
        SET hills TO a list of zeros of LENGTH two MULTIPLIED BY n MINUS one  
        SET dales TO a list of zeros of LENGTH two MULTIPLIED BY n MINUS one  
        RETURN backtrack WITH no PARAMETERS  
    END FUNCTION  
END CLASS
