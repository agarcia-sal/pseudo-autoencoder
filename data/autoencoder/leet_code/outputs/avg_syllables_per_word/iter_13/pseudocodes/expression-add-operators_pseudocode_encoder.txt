CLASS Solution  
    FUNCTION addOperators(num, target) RETURNS list of string  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO substring from position index TO position i MINUS one of num  
                IF the LENGTH OF operand GREATER THAN one AND the element at position zero of operand EQUALS zero  
                    CONTINUE TO the next iteration  
                END IF  
                
                IF index EQUALS zero  
                    CALL dfs WITH parameters i, operand, the INTEGER VALUE OF operand, the INTEGER VALUE OF operand  
                ELSE  
                    CALL dfs WITH parameters i, the concatenation of path PLUS the character PLUS the operand, the value PLUS the INTEGER VALUE OF operand, the INTEGER VALUE OF operand  
                    CALL dfs WITH parameters i, the concatenation of path PLUS the character MINUS the operand, the value MINUS the INTEGER VALUE OF operand, the NEGATIVE INTEGER VALUE OF operand  
                    CALL dfs WITH parameters i, the concatenation of path PLUS the character MULTIPLY the operand, the value MINUS prev PLUS prev MULTIPLIED BY the INTEGER VALUE OF operand, prev MULTIPLIED BY the INTEGER VALUE OF operand  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL dfs WITH parameters zero, empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
