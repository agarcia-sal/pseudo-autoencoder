CLASS Solution  
    FUNCTION spiralMatrixIII(rows, cols, rStart, cStart)  
        SET directions TO list of pairs zero and one zero and one zero and minus one minus one and zero representing east south west and north respectively  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO empty list  
        SET x TO rStart  
        SET y TO cStart  
        
        WHILE the LENGTH OF visited IS LESS THAN rows MULTIPLIED BY cols holds  
            IF increment MODULO two EQUALS zero THEN  
                INCREMENT steps BY one  
            END IF  
            FOR each count FROM one TO steps  
                IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN rows AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN cols THEN  
                    APPEND list of x and y TO visited  
                END IF  
                SET dx TO element at position direction_index of directions AT first position  
                SET dy TO element at position direction_index of directions AT second position  
                INCREMENT x BY dx  
                INCREMENT y BY dy  
            END FOR  
            SET direction_index TO the REMAINDER OF direction_index PLUS one DIVIDED BY four  
            INCREMENT increment BY one  
        END WHILE  
        
        RETURN visited  
    END FUNCTION  
END CLASS
