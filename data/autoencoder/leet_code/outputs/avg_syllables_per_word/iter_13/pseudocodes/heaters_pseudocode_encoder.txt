CLASS Solution  
    FUNCTION findRadius(houses AS List of integers, heaters AS List of integers) RETURNS integer  
        SORT houses IN ascending order  
        SORT heaters IN ascending order  
        
        SET min_radius TO zero  
        SET j TO zero  
        
        FOR each house IN houses  
            WHILE j IS LESS THAN the LENGTH OF heaters MINUS one AND the ABSOLUTE VALUE OF the value at position j of heaters MINUS house IS GREATER THAN OR EQUAL TO the ABSOLUTE VALUE OF the value at position j PLUS one of heaters MINUS house  
                INCREMENT j BY one  
            END WHILE  
            
            SET min_radius TO the GREATER OF min_radius AND the ABSOLUTE VALUE OF the value at position j of heaters MINUS house  
        END FOR  
        
        RETURN min_radius  
    END FUNCTION  
END CLASS
