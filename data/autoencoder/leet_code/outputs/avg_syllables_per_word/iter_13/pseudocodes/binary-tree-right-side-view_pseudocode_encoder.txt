CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero, left EQUAL TO none, right EQUAL TO none  
        SET self DOT val TO val  
        SET self DOT left TO left  
        SET self DOT right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH VALUE the element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO the first element REMOVED from queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node DOT left TO NEW TreeNode WITH VALUE the element at position i of values  
            APPEND node DOT left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node DOT right TO NEW TreeNode WITH VALUE the element at position i of values  
            APPEND node DOT right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p DOT val NOT EQUALS q DOT val  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH p DOT left AND q DOT left AND is_same_tree WITH p DOT right AND q DOT right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView WITH PARAMETERS self, root  
        IF root EQUALS none  
            RETURN an empty list  
        END IF  
        SET right_side_view TO an empty list  
        SET queue TO a new double ended queue containing root as the only element  
        WHILE the queue IS NOT empty  
            SET level_length TO the LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO the first element REMOVED from queue  
                IF i EQUALS level_length MINUS one  
                    APPEND node DOT val TO right_side_view  
                END IF  
                IF node DOT left IS NOT none  
                    APPEND node DOT left TO queue  
                END IF  
                IF node DOT right IS NOT none  
                    APPEND node DOT right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
