CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of integers FROM zero TO n inclusive  
        SET self.rank TO list of zeros with length n plus one  
        SET self.count TO n  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u of self.parent IS NOT EQUAL TO u  
            SET element at position u of self.parent TO self.find(element at position u of self.parent)  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET rootU TO self.find(u)  
        SET rootV TO self.find(v)  
        IF rootU IS NOT EQUAL TO rootV  
            IF element at position rootU of self.rank IS GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank IS LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            DECREMENT self.count BY one  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION maxNumEdgesToRemove(self, n, edges)  
        ASSIGN pointer/reference TO new UnionFind with parameter n AS uf_common  
        ASSIGN pointer/reference TO new UnionFind with parameter n AS uf_alice  
        ASSIGN pointer/reference TO new UnionFind with parameter n AS uf_bob  

        SET used_edges TO zero  

        FOR each element consisting of t u and v IN edges  
            IF t EQUALS three  
                IF uf_common.union(u, v) IS True  
                    uf_alice.union(u, v)  
                    uf_bob.union(u, v)  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  

        FOR each element consisting of t u and v IN edges  
            IF t EQUALS one AND uf_alice.union(u, v) IS True  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND uf_bob.union(u, v) IS True  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  

        IF uf_alice.count EQUALS one AND uf_bob.count EQUALS one  
            RETURN the LENGTH OF edges MINUS used_edges  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
