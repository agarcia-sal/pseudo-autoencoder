CLASS Solution  
    FUNCTION getMoneyAmount WITH PARAMETERS n  
        DEFINE FUNCTION dp WITH PARAMETERS left AND right  
            IF left GREATER THAN OR EQUAL TO right THEN  
                RETURN zero  
            END IF  
            SET min_cost TO positive infinity  
            FOR pivot FROM left TO right  
                SET left_cost TO CALL dp WITH left AND pivot MINUS one  
                SET right_cost TO CALL dp WITH pivot PLUS one AND right  
                SET max_sub_cost TO THE GREATER OF left_cost AND right_cost  
                SET cost TO pivot PLUS max_sub_cost  
                SET min_cost TO THE LESSER OF min_cost AND cost  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN CALL dp WITH one AND n  
END CLASS
