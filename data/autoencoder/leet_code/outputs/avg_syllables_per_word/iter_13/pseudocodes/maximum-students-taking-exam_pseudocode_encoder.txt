CLASS Solution  
    FUNCTION maxStudents WITH parameter seats AS List of List of strings  
        SET m TO the LENGTH OF seats  
        SET n TO the LENGTH OF element at position zero of seats  
        
        SET broken TO a list of zeroes with the LENGTH OF m  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF element at position j of element at position i of seats EQUALS the character hash symbol  
                    SET element at position i of broken TO element at position i of broken BITWISE OR ONE LEFT SHIFTED BY j  
                END IF  
            END FOR  
        END FOR  
        
        FUNCTION is_valid WITH parameter arrangement  
            RETURN arrangement BITWISE AND arrangement RIGHT SHIFTED BY one EQUALS zero  
        END FUNCTION  
        
        SET dp TO a list of empty dictionaries with the LENGTH OF m  
        
        FOR arrangement FROM zero TO one LEFT SHIFTED BY n MINUS one  
            IF is_valid WITH arrangement HOLDS AND arrangement BITWISE AND element at position zero of broken EQUALS zero  
                SET value at key arrangement of element at position zero of dp TO the COUNT OF BITS EQUAL TO ONE IN arrangement  
            END IF  
        END FOR  
        
        FOR index i FROM one TO m MINUS one  
            FOR arrangement FROM zero TO one LEFT SHIFTED BY n MINUS one  
                IF is_valid WITH arrangement HOLDS AND arrangement BITWISE AND element at position i of broken EQUALS zero  
                    SET max_students TO zero  
                    FOR each prev_arrangement IN keys of element at position i MINUS one of dp  
                        IF arrangement BITWISE AND prev_arrangement RIGHT SHIFTED BY one EQUALS zero AND arrangement BITWISE AND prev_arrangement LEFT SHIFTED BY one EQUALS zero  
                            IF value at key prev_arrangement of element at position i MINUS one of dp GREATER THAN max_students  
                                SET max_students TO value at key prev_arrangement of element at position i MINUS one of dp  
                            END IF  
                        END IF  
                    END FOR  
                    SET value at key arrangement of element at position i of dp TO max_students PLUS the COUNT OF BITS EQUAL TO ONE IN arrangement  
                END IF  
            END FOR  
        END FOR  
        
        IF element at the last position of dp IS NOT empty  
            RETURN the maximum VALUE in the dictionary at the last position of dp  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
