CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO zero  
        FOR bit_position FROM thirty one TO zero DECREMENT BY one  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            INITIALIZE prefixes AS an empty set  
            FOR each number IN nums  
                SET prefix TO integer division of number BY two RAISED TO THE POWER OF bit_position  
                ADD prefix TO prefixes  
            END FOR  
            FOR each prefix_element IN prefixes  
                IF the bitwise exclusive OR of current_candidate AND prefix_element IS AN ELEMENT IN prefixes  
                    SET max_xor TO current_candidate  
                    BREAK out of the loop  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
