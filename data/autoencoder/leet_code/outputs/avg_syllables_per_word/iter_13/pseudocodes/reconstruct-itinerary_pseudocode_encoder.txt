CLASS Solution  
    FUNCTION findItinerary WITH parameter tickets  
        SET graph TO a new default mapping from keys TO empty lists  
        FOR each pair of departure airport named frm AND arrival airport named to IN the reversed list of tickets sorted in ascending alphabetical order  
            APPEND the arrival airport to the list at key departure airport in the graph  
        END FOR  
        SET itinerary TO an empty list  
        FUNCTION visit WITH parameter airport  
            WHILE there exists an element in the list at key airport in the graph  
                CALL visit WITH argument the last element removed from the list at key airport in the graph  
            END WHILE  
            APPEND the airport parameter TO itinerary  
        END FUNCTION  
        CALL visit WITH argument the string representing the airport code J F K  
        RETURN the reversed order of the list itinerary  
    END FUNCTION  
END CLASS
