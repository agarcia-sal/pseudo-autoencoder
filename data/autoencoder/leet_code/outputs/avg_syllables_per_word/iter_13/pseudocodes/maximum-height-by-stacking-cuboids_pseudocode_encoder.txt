CLASS Solution  
    FUNCTION maxHeight WITH PARAMETERS cuboids  
        FOR each cuboid IN cuboids  
            SORT dimensions of cuboid IN ascending order  
        END FOR  
        SORT cuboids IN lexicographical ascending order  
        SET n TO the LENGTH OF cuboids  
        INITIALIZE dp AS a list of zeros WITH size n  
        FOR i FROM zero TO n MINUS one  
            SET element at position i of dp TO element at position three of cuboid at position i  
            FOR j FROM zero TO i MINUS one  
                IF element at position one of cuboid at position j LESS THAN OR EQUAL TO element at position one of cuboid at position i  
                AND element at position two of cuboid at position j LESS THAN OR EQUAL TO element at position two of cuboid at position i  
                AND element at position three of cuboid at position j LESS THAN OR EQUAL TO element at position three of cuboid at position i  
                    SET element at position i of dp TO the GREATER OF element at position i of dp AND element at position j of dp PLUS element at position three of cuboid at position i  
                END IF  
            END FOR  
        END FOR  
        RETURN the MAXIMUM value contained IN dp  
    END FUNCTION  
END CLASS
