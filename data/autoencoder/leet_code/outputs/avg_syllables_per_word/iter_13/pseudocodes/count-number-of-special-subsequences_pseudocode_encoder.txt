CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO one billion PLUS seven  
        SET dp TO list containing zero at first position zero at second position zero at third position  
        FOR each num IN nums  
            IF num EQUALS zero  
                SET element at position zero of dp TO two MULTIPLIED BY element at position zero of dp PLUS one MODULO MOD  
            ELSE IF num EQUALS one  
                SET element at position one of dp TO two MULTIPLIED BY element at position one of dp PLUS element at position zero of dp MODULO MOD  
            ELSE IF num EQUALS two  
                SET element at position two of dp TO two MULTIPLIED BY element at position two of dp PLUS element at position one of dp MODULO MOD  
            END IF  
        END FOR  
        RETURN element at position two of dp  
    END FUNCTION  
END CLASS
