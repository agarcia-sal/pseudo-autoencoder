CLASS Solution  
    FUNCTION minimumDistance WITH PARAMETER word  
        FUNCTION distance WITH PARAMETERS c1 AND c2  
            IF c1 EQUALS None  
                RETURN zero  
            END IF  
            SET x1 TO THE RESULT OF DIVIDING the NUMERIC VALUE OF c1 MINUS THE NUMERIC VALUE OF uppercase A BY six AND TAKING THE INTEGER QUOTIENT  
            SET y1 TO THE RESULT OF DIVIDING the NUMERIC VALUE OF c1 MINUS THE NUMERIC VALUE OF uppercase A BY six AND TAKING THE REMAINDER  
            SET x2 TO THE RESULT OF DIVIDING the NUMERIC VALUE OF c2 MINUS THE NUMERIC VALUE OF uppercase A BY six AND TAKING THE INTEGER QUOTIENT  
            SET y2 TO THE RESULT OF DIVIDING the NUMERIC VALUE OF c2 MINUS THE NUMERIC VALUE OF uppercase A BY six AND TAKING THE REMAINDER  
            RETURN THE ABSOLUTE VALUE OF x1 MINUS x2 PLUS THE ABSOLUTE VALUE OF y1 MINUS y2  
        END FUNCTION  
        
        FUNCTION dp WITH PARAMETERS i AND f1 AND f2  
            IF i EQUALS THE LENGTH OF word  
                RETURN zero  
            END IF  
            SET c TO THE ELEMENT AT POSITION i OF word  
            SET d1 TO THE RESULT OF distance WITH ARGUMENTS f1 AND c PLUS THE RESULT OF dp WITH ARGUMENTS i PLUS one AND c AND f2  
            SET d2 TO THE RESULT OF distance WITH ARGUMENTS f2 AND c PLUS THE RESULT OF dp WITH ARGUMENTS i PLUS one AND f1 AND c  
            RETURN THE MINIMUM OF d1 AND d2  
        END FUNCTION  
        
        RETURN THE RESULT OF dp WITH ARGUMENTS zero AND None AND None  
    END FUNCTION  
END CLASS
