CLASS Solution  
    FUNCTION countSubgraphsForEachDiameter WITH PARAMETERS n AND edges  
        SET graph TO A MAPPING FROM each INTEGER i FROM ONE TO n TO an EMPTY LIST  
        FOR each pair u AND v IN edges  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
        
        FUNCTION bfs WITH PARAMETER subtree  
            SET start TO an ARBITRARY element FROM subtree  
            SET queue TO a NEW double ended QUEUE initialized WITH start  
            SET visited TO a NEW SET containing start  
            SET distance TO zero  
            SET farthest_node TO start  
            
            WHILE queue IS NOT empty  
                REMOVE the FIRST element FROM queue AND ASSIGN TO node  
                FOR each neighbor IN element at position node of graph  
                    IF neighbor IS an ELEMENT OF subtree AND neighbor IS NOT IN visited  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                        SET farthest_node TO neighbor  
                    END IF  
                END FOR  
            END WHILE  
            
            SET queue TO a NEW double ended QUEUE initialized WITH farthest_node  
            SET visited TO a NEW SET containing farthest_node  
            SET max_distance TO zero  
            
            WHILE queue IS NOT empty  
                SET size TO the LENGTH OF queue  
                FOR each index FROM one TO size  
                    REMOVE the FIRST element FROM queue AND ASSIGN TO node  
                    FOR each neighbor IN element at position node of graph  
                        IF neighbor IS an ELEMENT OF subtree AND neighbor IS NOT IN visited  
                            ADD neighbor TO visited  
                            APPEND neighbor TO queue  
                        END IF  
                    END FOR  
                END FOR  
                IF queue IS NOT empty  
                    INCREMENT max_distance BY one  
                END IF  
            END WHILE  
            
            RETURN max_distance  
        END FUNCTION  
        
        SET result TO A LIST OF zeros WITH LENGTH n MINUS one  
        
        FOR each size FROM two TO n  
            FOR each subset IN combinations OF INTEGERS FROM one TO n TAKEN size AT A TIME  
                SET visited TO an EMPTY SET  
                SET stack TO a LIST containing the FIRST element of subset  
                WHILE stack IS NOT empty  
                    REMOVE the LAST element FROM stack AND ASSIGN TO node  
                    IF node IS NOT IN visited  
                        ADD node TO visited  
                        FOR each neighbor IN element at position node of graph  
                            IF neighbor IS an ELEMENT OF subset AND neighbor IS NOT IN visited  
                                APPEND neighbor TO stack  
                            END IF  
                        END FOR  
                    END IF  
                END WHILE  
                
                IF visited EQUALS the SET converted FROM subset  
                    SET diameter TO bfs OF the SET converted FROM subset  
                    IF diameter IS GREATER THAN zero  
                        INCREMENT element at position diameter MINUS one OF result BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
