CLASS Solution  
    FUNCTION topKFrequent(words AS List OF strings, k AS integer) RETURNS List OF strings  
        SET count TO the frequency mapping of each word IN words using a counting method  
        SET heap TO a new list created by pairing each word and its frequency from count WHERE each pair consists of the negative frequency and the word itself  
        APPLY the heap construction procedure TO heap to organize it into a min-heap  
        SET result TO an empty list  
        FOR index FROM one TO k  
            SET extracted_element TO the element obtained by removing the root element FROM heap using the heap extraction procedure  
            APPEND the word part of extracted_element TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
