CLASS Solution  
    FUNCTION optimalDivision WITH parameter nums  
        IF the LENGTH OF nums EQUALS one  
            RETURN the STRING REPRESENTATION OF the element at position zero of nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the STRING REPRESENTATION OF the element at position zero of nums CONCATENATED WITH the CHARACTER FOR FORWARD SLASH CONCATENATED WITH the STRING REPRESENTATION OF the element at position one of nums  
        END IF  
        RETURN the STRING REPRESENTATION OF the element at position zero of nums CONCATENATED WITH the CHARACTER FOR FORWARD SLASH CONCATENATED WITH an OPENING PARENTHESIS CONCATENATED WITH the STRING FORWARD SLASH JOINING of the STRING REPRESENTATIONS OF elements from position one TO the END OF nums CONCATENATED WITH a CLOSING PARENTHESIS  
    END FUNCTION  
END CLASS
