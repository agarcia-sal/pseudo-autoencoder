CLASS Solution  
    FUNCTION rearrangeSticks(n integer, k integer) RETURNS integer  
        SET MOD TO TEN RAISED TO THE POWER OF NINE PLUS SEVEN  
        
        CREATE dp AS a two dimensional list containing zero for each position for index i FROM zero TO n INCLUSIVE and for index j FROM zero TO k INCLUSIVE  
        
        SET element at position zero of element at position zero of dp TO one  
        
        FOR i FROM one TO n INCLUSIVE  
            FOR j FROM one TO k INCLUSIVE  
                SET element at position j of element at position i of dp TO the REMAINDER AFTER DIVISION OF the SUM OF element at position j MINUS one of element at position i MINUS one of dp PLUS the PRODUCT OF i MINUS one AND element at position j of element at position i MINUS one of dp BY MOD  
            END FOR  
        END FOR  
        
        RETURN element at position k of element at position n of dp  
    END FUNCTION  
END CLASS
