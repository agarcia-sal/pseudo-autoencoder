CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF element at position zero of cost  
        INITIALIZE dp AS a two dimensional list with size1 PLUS one rows and two TO THE POWER OF size2 columns FILLED WITH infinity value  
        SET element at position zero at zero of dp TO zero  
        FOR each index i FROM zero TO size1 MINUS one  
            FOR each integer mask FROM zero TO two TO THE POWER OF size2 MINUS one  
                FOR each index j FROM zero TO size2 MINUS one  
                    SET combined_mask TO the result of mask OR the result of one SHIFTED LEFT BY j positions  
                    SET element at position i PLUS one at combined_mask of dp TO the MINIMUM OF element at position i PLUS one at combined_mask of dp AND element at position i at mask of dp PLUS element at position i at j of cost  
                END FOR  
                FOR each index j FROM zero TO size2 MINUS one  
                    IF the BITWISE AND of mask AND one SHIFTED LEFT BY j positions IS NOT EQUAL TO zero THEN  
                        SET toggled_mask TO the result of mask XOR one SHIFTED LEFT BY j positions  
                        SET element at position i PLUS one at mask of dp TO the MINIMUM OF element at position i PLUS one at mask of dp AND element at position i PLUS one at toggled_mask of dp PLUS element at position i at j of cost  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position size1 at the result of two TO THE POWER OF size2 MINUS one of dp  
    END FUNCTION  
END CLASS
