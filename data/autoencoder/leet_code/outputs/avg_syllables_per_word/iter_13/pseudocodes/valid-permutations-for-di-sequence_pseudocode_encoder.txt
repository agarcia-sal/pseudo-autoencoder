CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of size n PLUS one by n PLUS one FILLED WITH zero  
        FOR each j FROM zero TO n  
            SET element at position zero at position j of dp TO one  
        END FOR  
        FOR each i FROM one TO n  
            SET prefix_sum TO zero  
            IF the element at position i MINUS one of s EQUALS the character I  
                FOR each j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO the sum of prefix_sum AND element at position i MINUS one at position j of dp MODULO MOD  
                    SET element at position i at position j of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR each j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO the sum of prefix_sum AND element at position i MINUS one at position j PLUS one of dp MODULO MOD  
                    SET element at position i at position j of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        SET result TO the sum of all elements of element at position n of dp MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
