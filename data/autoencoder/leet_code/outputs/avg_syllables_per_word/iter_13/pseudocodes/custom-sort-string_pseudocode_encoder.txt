CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO an empty dictionary mapping characters TO integer positions representing the custom order  
        FOR each pair of index and character IN order paired with their positions starting from zero  
            SET the dictionary value at character TO index  
        END FOR  
        SET sorted_characters TO the collection of characters from s sorted according to the following rule  
            FOR each character x IN s  
                OBTAIN the value from the dictionary for character x if it exists otherwise zero  
            END FOR  
        JOIN all characters in sorted_characters into a single string  
        RETURN the resulting joined string  
    END FUNCTION  
END CLASS
