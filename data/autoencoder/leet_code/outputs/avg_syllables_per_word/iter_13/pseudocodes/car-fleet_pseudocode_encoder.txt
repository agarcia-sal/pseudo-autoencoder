CLASS Solution  
    FUNCTION carFleet WITH PARAMETERS target position speed  
        SET pairs TO the PAIRING OF elements at positions in position WITH elements at positions in speed  
        SET sorted_pairs TO pairs SORTED IN REVERSE ORDER BY the FIRST element of each pair  
        SET time_to_target TO an EMPTY LIST  
        FOR each pair IN sorted_pairs  
            SET position_element TO the FIRST element of pair  
            SET speed_element TO the SECOND element of pair  
            SET time_element TO the DIFFERENCE BETWEEN target AND position_element DIVIDED BY speed_element  
            APPEND time_element TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time_element IN time_to_target  
            IF time_element IS GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time_element  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
