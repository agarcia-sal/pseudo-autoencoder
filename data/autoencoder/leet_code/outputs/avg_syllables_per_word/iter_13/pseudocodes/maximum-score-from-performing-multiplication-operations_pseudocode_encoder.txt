CLASS Solution  
    FUNCTION maximumScore(nums AS List OF integer, multipliers AS List OF integer) RETURNS integer  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        
        SET dp TO a list of zeros with LENGTH OF m PLUS one  
        
        FOR i FROM m MINUS one DOWN TO zero  
            SET new_dp TO a list of zeros with LENGTH OF m PLUS one  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS the DIFFERENCE OF i MINUS left  
                SET mul TO the element at position i of multipliers  
                SET value_from_left TO mul MULTIPLIED BY the element at position left of nums PLUS the element at position left PLUS one of dp  
                SET value_from_right TO mul MULTIPLIED BY the element at position right of nums PLUS the element at position left of dp  
                SET the element at position left of new_dp TO the GREATER OF value_from_left AND value_from_right  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the element at position zero of dp  
    END FUNCTION  
END CLASS
