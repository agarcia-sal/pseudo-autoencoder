CLASS Solution  
    FUNCTION kthSmallestPath(destination WITH TYPE List OF integer, k WITH TYPE integer) RETURNS string  
        SET row TO element at position zero of destination  
        SET col TO element at position one of destination  
        SET path TO an empty list  
        SET total_moves TO the sum of col PLUS row  
        FOR index FROM zero TO total_moves MINUS one  
            IF col GREATER THAN zero  
                SET paths_with_H TO the number of combinations of row PLUS col MINUS one CHOOSE col MINUS one  
                IF k LESS THAN OR EQUAL TO paths_with_H  
                    APPEND the character H TO path  
                    DECREMENT col BY one  
                ELSE  
                    APPEND the character V TO path  
                    DECREMENT row BY one  
                    DECREMENT k BY paths_with_H  
                END IF  
            ELSE  
                APPEND the character V TO path  
                DECREMENT row BY one  
            END IF  
        END FOR  
        RETURN the concatenation of all elements of path into a single string  
    END FUNCTION  
END CLASS
