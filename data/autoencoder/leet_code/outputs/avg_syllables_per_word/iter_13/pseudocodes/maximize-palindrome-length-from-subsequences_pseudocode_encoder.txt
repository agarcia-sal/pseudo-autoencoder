CLASS Solution  
    FUNCTION longestPalindrome WITH PARAMETERS word1 word2  
        SET s TO concatenation of word1 with word2  
        SET n TO the LENGTH OF s  
        SET dp TO two dimensional list of zeros with size n by n  
        FOR each index i FROM zero TO n MINUS one  
            SET element at position i at position i of dp TO one  
        END FOR  
        SET max_len TO zero  
        FOR each length FROM two TO n  
            FOR each index i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of s EQUALS element at position j of s THEN  
                    SET element at position i at position j of dp TO element at position i PLUS one at position j MINUS one of dp PLUS two  
                    IF i IS LESS THAN the LENGTH OF word1 AND j IS GREATER THAN OR EQUAL TO the LENGTH OF word1 THEN  
                        SET max_len TO the GREATER VALUE BETWEEN max_len AND element at position i at position j of dp  
                    END IF  
                ELSE  
                    SET element at position i at position j of dp TO the GREATER VALUE BETWEEN element at position i PLUS one at position j of dp AND element at position i at position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN max_len  
    END FUNCTION  
END CLASS
