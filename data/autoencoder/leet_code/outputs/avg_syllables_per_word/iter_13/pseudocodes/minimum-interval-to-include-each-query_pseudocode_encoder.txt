CLASS Solution  
    FUNCTION minInterval(intervals AS List OF List OF Integer, queries AS List OF Integer) RETURNS List OF Integer  
        SORT intervals BY the element at position zero of each interval IN ascending order  
        SET sorted_queries TO the list of pairs each containing query and its original position in queries sorted BY the query IN ascending order  
        SET min_heap TO an empty list to represent a heap storing pairs of interval size and interval end  
        SET result TO a list of the LENGTH OF queries with all elements set TO negative one  
        SET interval_index TO zero  
        FOR each pair query AND original_index IN sorted_queries  
            WHILE interval_index IS LESS THAN the LENGTH OF intervals AND the element at position zero of the interval at position interval_index of intervals IS LESS THAN OR EQUAL TO query  
                SET start TO the element at position zero of the interval at position interval_index of intervals  
                SET end TO the element at position one of the interval at position interval_index of intervals  
                SET interval_size TO end MINUS start PLUS one  
                ADD the pair interval_size and end TO min_heap maintaining the heap property  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE min_heap IS NOT EMPTY AND the second element of the first pair in min_heap IS LESS THAN query  
                REMOVE the pair with the smallest first element FROM min_heap maintaining the heap property  
            END WHILE  
            IF min_heap IS NOT EMPTY  
                SET the element at position original_index of result TO the first element of the first pair in min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
