CLASS Solution  
    FUNCTION killProcess(pid AS List OF integers, ppid AS List OF integers, kill AS integer) RETURNS List OF integers  
        SET tree TO an empty mapping from integer TO list of integers with default empty list  
        FOR each child AND parent IN corresponding positions of pid AND ppid  
            APPEND child TO element at key parent of tree  
        END FOR  
        
        SET queue TO a list containing kill  
        SET killed_processes TO an empty list  
        
        WHILE the LENGTH OF queue is greater than zero  
            SET current TO element at position zero of queue  
            REMOVE element at position zero FROM queue  
            APPEND current TO killed_processes  
            EXTEND queue BY element at key current of tree  
        END WHILE  
        
        RETURN killed_processes  
    END FUNCTION  
END CLASS
