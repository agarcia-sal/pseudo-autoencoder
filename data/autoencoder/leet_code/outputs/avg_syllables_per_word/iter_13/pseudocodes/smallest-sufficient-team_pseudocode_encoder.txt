CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        CREATE dictionary d TO MAP each skill IN req_skills TO its position index  
        SET m TO the LENGTH OF req_skills  
        SET n TO the LENGTH OF people  
        INITIALIZE list p WITH n ZERO elements  
        FOR each index i AND skill list ss IN people  
            FOR each skill s IN ss  
                SET element at position i of p TO the bitwise OR of element at position i of p AND a value OF one shifted LEFT BY the position index of skill s IN dictionary d  
            END FOR  
        END FOR  
        INITIALIZE list f WITH a size OF two RAISED TO the power OF m, FILLED WITH infinity values  
        INITIALIZE list g WITH a size OF two RAISED TO the power OF m, FILLED WITH zero values  
        INITIALIZE list h WITH a size OF two RAISED TO the power OF m, FILLED WITH zero values  
        SET element at position zero of f TO zero  
        FOR integer i FROM zero TO one LESS THAN two RAISED TO the power OF m  
            IF element at position i of f EQUALS infinity  
                CONTINUE TO next iteration  
            END IF  
            FOR integer j FROM zero TO one LESS THAN n  
                IF element at position i of f PLUS one IS LESS THAN element at position bitwise OR of i AND element at position j of p of f  
                    SET element at position bitwise OR of i AND element at position j of p of f TO element at position i of f PLUS one  
                    SET element at position bitwise OR of i AND element at position j of p of g TO j  
                    SET element at position bitwise OR of i AND element at position j of p of h TO i  
                END IF  
            END FOR  
        END FOR  
        SET i TO two RAISED TO the power OF m MINUS one  
        INITIALIZE empty list ans  
        WHILE i IS NOT zero  
            APPEND element at position i of g TO ans  
            SET i TO element at position i of h  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
