CLASS Solution  
    FUNCTION boldWords(words AS list of strings, s AS string) RETURNS string  
        SET bold TO new list of boolean values of LENGTH OF s all set to false  
        FOR each word IN words  
            SET start TO the POSITION OF word IN s  
            WHILE start is NOT EQUAL TO negative one  
                FOR index FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET element at position index of bold TO true  
                END FOR  
                SET start TO the POSITION OF word IN s STARTING FROM start PLUS one  
            END WHILE  
        END FOR  
        SET result TO new empty list  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF s  
            IF element at position i of bold IS true  
                APPEND the string indicating start of bold tag TO result  
                WHILE i LESS THAN the LENGTH OF s AND element at position i of bold IS true  
                    APPEND element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string indicating end of bold tag TO result  
            ELSE  
                APPEND element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
