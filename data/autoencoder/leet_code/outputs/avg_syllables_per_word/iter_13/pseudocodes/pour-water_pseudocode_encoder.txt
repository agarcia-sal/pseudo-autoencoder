CLASS Solution  
    FUNCTION pourWater(heights AS List OF integers volume AS integer k AS integer) RETURNS List OF integers  
        SET n TO the LENGTH OF heights  
        
        FOR each iteration FROM one TO volume  
            SET i TO k  
            WHILE i IS GREATER THAN zero AND the element at position i MINUS one of heights IS LESS THAN OR EQUAL TO the element at position i of heights  
                DECREMENT i BY one  
            END WHILE  
            
            WHILE i IS LESS THAN k AND the element at position i of heights IS EQUAL TO the element at position i PLUS one of heights  
                INCREMENT i BY one  
            END WHILE  
            
            IF i IS LESS THAN k  
                SET the element at position i of heights TO the element at position i of heights PLUS one  
                CONTINUE TO the next iteration OF the FOR loop  
            END IF  
            
            SET i TO k  
            WHILE i IS LESS THAN n MINUS one AND the element at position i PLUS one of heights IS LESS THAN OR EQUAL TO the element at position i of heights  
                INCREMENT i BY one  
            END WHILE  
            
            WHILE i IS GREATER THAN k AND the element at position i of heights IS EQUAL TO the element at position i MINUS one of heights  
                DECREMENT i BY one  
            END WHILE  
            
            IF i IS GREATER THAN k  
                SET the element at position i of heights TO the element at position i of heights PLUS one  
                CONTINUE TO the next iteration OF the FOR loop  
            END IF  
            
            SET the element at position k of heights TO the element at position k of heights PLUS one  
        END FOR  
        
        RETURN heights  
    END FUNCTION  
END CLASS
