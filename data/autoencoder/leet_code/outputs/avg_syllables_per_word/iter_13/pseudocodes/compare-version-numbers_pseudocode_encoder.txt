CLASS Solution  
    FUNCTION compareVersion with parameters version1 and version2  
        SET v1_revisions TO result of splitting version1 by the character dot  
        SET v2_revisions TO result of splitting version2 by the character dot  
        SET max_length TO the GREATER OF the LENGTH OF v1_revisions AND the LENGTH OF v2_revisions  
        FOR index FROM zero TO max_length MINUS one  
            IF index LESS THAN the LENGTH OF v1_revisions  
                SET v1 TO the INTEGER VALUE OF the element at position index of v1_revisions  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF index LESS THAN the LENGTH OF v2_revisions  
                SET v2 TO the INTEGER VALUE OF the element at position index of v2_revisions  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 LESS THAN v2  
                RETURN negative one  
            ELSE IF v1 GREATER THAN v2  
                RETURN one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
