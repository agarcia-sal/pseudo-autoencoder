CLASS Solution  
    FUNCTION mirrorReflection WITH PARAMETERS p q  
        SET lcm TO the LEAST COMMON MULTIPLE OF p AND q  
        SET vertical_reflections TO lcm DIVIDED BY p  
        SET horizontal_reflections TO lcm DIVIDED BY q  
        IF vertical_reflections MODULO two EQUALS zero  
            RETURN zero  
        ELSE IF horizontal_reflections MODULO two EQUALS one  
            RETURN one  
        ELSE  
            RETURN two  
        END IF  
    END FUNCTION  
END CLASS
