CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of each number in nums  
        INITIALIZE dp as a list of zeroes with size two raised to the power ten  
        SET the element at position zero of dp TO one to represent one way to have a product of one by choosing nothing  
        
        FOR each num IN the keys of count  
            IF num EQUALS one  
                SET the element at position zero of dp TO the element at position zero of dp MULTIPLIED BY two raised to the power of the element at position num of count  
                SET the element at position zero of dp TO the remainder of the preceding value when divided by MODULUS  
                CONTINUE to the next iteration  
            END IF  
            
            IF num NOT IN the keys of prime_factors  
                CONTINUE to the next iteration  
            END IF  
            
            SET factors TO the value associated with num in prime_factors  
            SET mask TO zero  
            
            FOR each prime IN factors  
                SET mask TO the bitwise OR of mask and two raised to the power of prime MINUS one  
            END FOR  
            
            FOR m FROM the value of two raised to the power ten MINUS one DOWN TO zero  
                IF the bitwise AND of m and mask EQUALS zero  
                    SET the element at position m OR mask of dp TO the remainder of the sum of the element at position m OR mask of dp and the product of the element at position m of dp and the element at position num of count when divided by MODULUS  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the remainder of the sum of the elements of dp starting from position one to the end when divided by MODULUS  
        RETURN result  
    END FUNCTION  
END CLASS
