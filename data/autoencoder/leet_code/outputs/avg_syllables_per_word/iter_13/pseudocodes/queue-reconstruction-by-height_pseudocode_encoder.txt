CLASS Solution  
    FUNCTION reconstructQueue(people)  
        CALL sort ON people USING a key which FOR each element x RETURNS the NEGATION OF element at position zero of x AND element at position one of x  
        SET queue TO an empty list  
        FOR each person IN people  
            INSERT person INTO queue AT the position given by element at position one of person  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
