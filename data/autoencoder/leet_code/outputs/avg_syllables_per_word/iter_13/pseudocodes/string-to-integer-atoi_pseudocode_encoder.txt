CLASS Solution  
    FUNCTION myAtoi(s)  
        SET INT_MAX TO two POWERED BY thirty ONE MINUS one  
        SET INT_MIN TO negative two POWERED BY thirty ONE  
        SET result TO zero  
        SET index TO zero  
        SET n TO the LENGTH OF s  
        SET sign TO one  
        WHILE index IS LESS THAN n AND element at position index of s EQUALS a single space character  
            INCREMENT index BY one  
        END WHILE  
        IF index IS LESS THAN n AND (element at position index of s EQUALS plus SIGN OR element at position index of s EQUALS minus SIGN)  
            IF element at position index of s EQUALS minus SIGN  
                SET sign TO negative one  
            ELSE  
                SET sign TO one  
            END IF  
            INCREMENT index BY one  
        END IF  
        WHILE index IS LESS THAN n AND element at position index of s IS A DIGIT  
            SET digit TO the INTEGER VALUE OF element at position index of s  
            IF result IS GREATER THAN INT_MAX MINUS digit DIVIDED BY ten  
                IF sign EQUALS one  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
                END IF  
            END IF  
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        END WHILE  
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
