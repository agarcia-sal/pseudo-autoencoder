CLASS Solution  
    FUNCTION removeBoxes(boxes)  
        FUNCTION dp(l, r, k, memo)  
            IF l GREATER THAN r  
                RETURN zero  
            END IF  
            IF the TUPLE of l r k IS IN memo  
                RETURN the ELEMENT CORRESPONDING TO the TUPLE of l r k IN memo  
            END IF  
            WHILE r GREATER THAN l AND the ELEMENT at position r of boxes EQUALS the ELEMENT at position r MINUS one of boxes  
                DECREMENT r BY one  
                INCREMENT k BY one  
            END WHILE  
            SET res TO the RESULT OF dp WITH PARAMETERS l r MINUS one zero memo PLUS the PRODUCT OF the QUANTITY k PLUS one AND ITSELF  
            FOR i FROM l TO r MINUS one  
                IF the ELEMENT at position i of boxes EQUALS the ELEMENT at position r of boxes  
                    SET res TO the MAXIMUM VALUE OF res AND the SUM OF dp WITH l i k PLUS one memo AND dp WITH i PLUS one r MINUS one zero memo  
                END IF  
            END FOR  
            SET the ELEMENT CORRESPONDING TO the TUPLE of l r k IN memo TO res  
            RETURN res  
        END FUNCTION  
        RETURN dp WITH PARAMETERS zero LENGTH OF boxes MINUS one zero an EMPTY DICTIONARY  
    END FUNCTION  
END CLASS
