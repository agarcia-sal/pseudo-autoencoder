CLASS Solution  
    FUNCTION topKFrequent(nums AS list of integers, k AS integer) RETURNS list of integers  
        SET frequency_map TO a new default dictionary mapping integers to integers  
        FOR each num IN nums  
            INCREMENT the value at key num in frequency_map BY one  
        END FOR  
        SET max_frequency TO the maximum value among all values in frequency_map  
        SET buckets TO a new list containing empty lists, with the count equal to max_frequency PLUS one  
        FOR each key num and value freq IN the pairs of frequency_map  
            APPEND num TO the element at position freq of buckets  
        END FOR  
        SET result TO an empty list  
        FOR freq FROM max_frequency DOWN TO one DECREMENTING BY one  
            IF the LENGTH OF the element at position freq of buckets IS GREATER THAN zero  
                APPEND all elements from the element at position freq of buckets TO result  
                IF the LENGTH OF result IS GREATER THAN OR EQUAL TO k  
                    RETURN the sublist from position one TO position k of result  
                END IF  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
