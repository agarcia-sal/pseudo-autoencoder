CLASS Solution  
    FUNCTION isBipartite(graph)  
        SET n TO the LENGTH OF graph  
        SET color TO a list of size n filled with zero  
        
        FUNCTION dfs(node, c)  
            SET element at position node of color TO c  
            FOR each neighbor IN element at position node of graph  
                IF element at position neighbor of color EQUALS c THEN  
                    RETURN false  
                END IF  
                IF element at position neighbor of color EQUALS zero THEN  
                    IF the result of calling dfs with arguments neighbor and NEGATIVE c IS false THEN  
                        RETURN false  
                    END IF  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        FOR index i FROM zero TO n MINUS one  
            IF element at position i of color EQUALS zero THEN  
                IF the result of calling dfs with arguments i and one IS false THEN  
                    RETURN false  
                END IF  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
