CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the result of splitting path by slash character  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS single period  
                CONTINUE to next iteration  
            ELSE IF component EQUALS two periods  
                IF the LENGTH OF stack IS GREATER THAN zero  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO a slash character CONCATENATED WITH the JOIN of stack elements separated by a slash character  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
