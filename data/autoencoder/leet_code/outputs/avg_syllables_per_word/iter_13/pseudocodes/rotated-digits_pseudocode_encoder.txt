CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO the set consisting of the characters two five six nine  
        SET same_digits TO the set consisting of the characters zero one eight  
        
        FUNCTION is_good_number(num)  
            SET num_str TO the string representation of num  
            IF all digits in num_str are elements of the union of change_digits and same_digits holds  
                IF any digit in num_str is an element of change_digits holds  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO the sum of is_good_number(i) for each integer i from one TO n inclusive  
        RETURN good_count  
    END FUNCTION  
END CLASS
