CLASS Solution  
    FUNCTION maxValue(events WITH TYPE List OF List OF Integer, k WITH TYPE Integer) RETURNS Integer  
        PERFORM a sort on events BY the value of the element at position one of each event  
        SET n TO the LENGTH OF events  
        
        SET dp TO a two dimensional list with dimensions n PLUS one and k PLUS one FILLED WITH zeroes representing maximum values attainable  
        
        FOR i FROM one TO n  
            SET start TO the element at position zero of the event at position i MINUS one of events  
            SET end TO the element at position one of the event at position i MINUS one of events  
            SET value TO the element at position two of the event at position i MINUS one of events  
            
            SET j TO the position found by performing a binary search LEFT BISECTION on events using the value of start AGAINST the end day OF events  
            
            FOR l FROM one TO k  
                SET dp at position i and l TO the GREATER OF the value at dp position i MINUS one and l AND the value at dp position j and l MINUS one PLUS value  
            END FOR  
        END FOR  
        
        RETURN the value at dp position n and k  
    END FUNCTION  
END CLASS
