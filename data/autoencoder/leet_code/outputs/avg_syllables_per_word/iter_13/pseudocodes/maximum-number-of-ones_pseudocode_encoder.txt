CLASS Solution  
    FUNCTION maximumNumberOfOnes(width height sideLength maxOnes)  
        SET freq TO a list of zeros with length sideLength MULTIPLIED BY sideLength  
        FOR each index i FROM zero TO width MINUS one  
            FOR each index j FROM zero TO height MINUS one  
                SET position TO remainder of i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS remainder of j DIVIDED BY sideLength  
                INCREMENT element at position of freq BY one  
            END FOR  
        END FOR  
        SORT freq IN descending order  
        SET result TO the sum of the first maxOnes elements of freq  
        RETURN result  
    END FUNCTION  
END CLASS
