CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO the mapping of each number in nums to its frequency  
        SET unique_nums TO the list of unique numbers from count sorted in ascending order  
        SET dp TO list of zeros with length equal to the length of unique_nums plus one  
        SET the element at position one of dp TO the element at position zero of unique_nums MULTIPLIED BY the count of that element in count  
        FOR index FROM two TO the length of unique_nums plus one  
            SET num TO the element at position index MINUS one of unique_nums  
            IF num EQUALS the element at position index MINUS two of unique_nums PLUS one  
                SET the element at position index of dp TO the maximum value BETWEEN the element at position index MINUS one of dp AND the element at position index MINUS two of dp PLUS num MULTIPLIED BY the count of num in count  
            ELSE  
                SET the element at position index of dp TO the element at position index MINUS one of dp PLUS num MULTIPLIED BY the count of num in count  
            END IF  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
