CLASS Solution  
    FUNCTION fractionToDecimal(numerator INTEGER, denominator INTEGER) RETURNS STRING  
        IF numerator LESS THAN zero EXCLUSIVE OR denominator LESS THAN zero IS true AND numerator LESS THAN zero IS NOT equal TO denominator LESS THAN zero  
            SET sign TO minus sign  
        ELSE  
            SET sign TO empty string  
        END IF  
        SET numerator TO absolute value of numerator  
        SET denominator TO absolute value of denominator  
        SET quotient TO the result of integer division of numerator BY denominator  
        SET remainder TO the remainder of numerator DIVIDED BY denominator  
        SET integer_part TO the string representation of quotient  
        IF remainder EQUALS zero  
            RETURN concatenation of sign WITH integer_part  
        END IF  
        INITIALIZE empty mapping named seen_remainders  
        INITIALIZE empty list named fractional_part  
        WHILE remainder IS NOT equal TO zero  
            IF remainder EXISTS IN mapping seen_remainders  
                SET start TO the value associated with remainder in seen_remainders  
                SET non_repeating TO concatenation of elements of fractional_part FROM position zero TO start minus one  
                SET repeating TO concatenation of elements of fractional_part FROM position start TO the last position  
                RETURN concatenation of sign WITH integer_part WITH decimal point WITH non_repeating WITH opening parenthesis WITH repeating WITH closing parenthesis  
            END IF  
            STORE the current LENGTH OF fractional_part WITH key remainder IN mapping seen_remainders  
            MULTIPLY remainder BY ten  
            SET quotient TO the result of integer division of remainder BY denominator  
            SET remainder TO the remainder of remainder DIVIDED BY denominator  
            APPEND the string representation of quotient TO fractional_part  
        END WHILE  
        RETURN concatenation of sign WITH integer_part WITH decimal point WITH concatenation of all elements IN fractional_part  
    END FUNCTION  
END CLASS
