CLASS Solution  
    FUNCTION getFactors WITH parameter n  
        FUNCTION backtrack WITH parameters start AND target  
            FOR each i FROM start TO the INTEGER PART OF the SQUARE ROOT OF target PLUS one  
                IF the REMAINDER of target DIVIDED BY i EQUALS zero  
                    APPEND the concatenation of path AND the LIST consisting of i AND target DIVIDED BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH i AND target DIVIDED BY i  
                    REMOVE the last element FROM path  
                END IF  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        SET path TO an empty list  
        CALL backtrack WITH two AND n  
        RETURN result  
    END FUNCTION  
END CLASS
