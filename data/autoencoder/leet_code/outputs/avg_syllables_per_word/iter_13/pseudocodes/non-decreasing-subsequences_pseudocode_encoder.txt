CLASS Solution  
    FUNCTION findSubsequences(nums)  
        FUNCTION dfs(u, last, t)  
            IF u EQUALS the LENGTH OF nums  
                IF the LENGTH OF t GREATER THAN one  
                    APPEND a copy of t TO ans  
                END IF  
                RETURN  
            END IF  
            IF element at position u of nums GREATER THAN OR EQUAL TO last  
                APPEND element at position u of nums TO t  
                CALL dfs WITH u INCREMENTED BY one AND last SET TO element at position u of nums AND t AS CURRENT LIST  
                REMOVE last element FROM t  
            END IF  
            IF element at position u of nums NOT EQUALS last  
                CALL dfs WITH u INCREMENTED BY one AND last UNCHANGED AND t AS CURRENT LIST  
            END IF  
        END FUNCTION  
        SET ans TO empty list  
        CALL dfs WITH zero FOR u AND minus one thousand FOR last AND empty list FOR t  
        RETURN ans  
    END FUNCTION  
END CLASS
