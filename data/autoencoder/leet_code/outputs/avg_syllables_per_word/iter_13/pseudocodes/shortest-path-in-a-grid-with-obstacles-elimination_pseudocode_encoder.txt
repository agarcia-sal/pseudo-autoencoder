CLASS Solution  
    FUNCTION shortestPath(grid, k)  
        IF grid IS NOT PRESENT OR element at position zero of grid IS NOT PRESENT  
            RETURN negative one  
        END IF  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        IF k IS GREATER THAN OR EQUAL TO m PLUS n MINUS two  
            RETURN m PLUS n MINUS two  
        END IF  
        SET directions TO list containing tuple zero and one, tuple one and zero, tuple zero and negative one, tuple negative one and zero  
        SET queue TO new deque initialized with tuple zero zero k zero  
        SET visited TO new set initialized with tuple zero zero k  
        WHILE queue IS NOT EMPTY  
            ASSIGN x y remaining_k steps TO REMOVE the FIRST element FROM queue  
            IF x EQUALS m MINUS one AND y EQUALS n MINUS one  
                RETURN steps  
            END IF  
            FOR each tuple dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET new_k TO remaining_k MINUS element at position nx of grid at position ny  
                    IF new_k IS GREATER THAN OR EQUAL TO zero AND tuple nx ny new_k IS NOT PRESENT IN visited  
                        ADD tuple nx ny new_k TO visited  
                        APPEND tuple nx ny new_k steps PLUS one TO queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
