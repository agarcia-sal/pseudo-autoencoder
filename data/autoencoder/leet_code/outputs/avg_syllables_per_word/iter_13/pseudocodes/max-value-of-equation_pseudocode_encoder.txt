CLASS Solution  
    FUNCTION findMaxValueOfEquation(points WITH TYPE List of List of integer, k WITH TYPE integer) RETURNS integer  
        SET max_value TO negative infinity  
        SET deque_ TO an empty double ended queue  
        
        FOR each pair consisting of xj and yj IN points  
            WHILE deque_ is not empty AND xj MINUS element at position zero and position one of deque_ is greater than k  
                REMOVE element at position zero FROM deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO the greater of max_value AND yj PLUS xj PLUS element at position zero and position zero of deque_  
            END IF  
            
            WHILE deque_ is not empty AND yj MINUS xj is greater than or equal to element at last position and position zero of deque_  
                REMOVE element at last position FROM deque_  
            END WHILE  
            
            APPEND the pair consisting of yj MINUS xj AND xj TO deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
