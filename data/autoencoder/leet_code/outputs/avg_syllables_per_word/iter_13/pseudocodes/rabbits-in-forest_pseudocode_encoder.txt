CLASS Solution  
    FUNCTION numRabbits WITH parameter answers AS list of integers  
        SET answer_counts TO the count of each unique element in answers  
        SET total_rabbits TO zero  
        FOR each answer and corresponding count IN answer_counts entries  
            SET group_size TO answer PLUS one  
            SET groups TO the result of integer division of count PLUS answer BY group_size  
            INCREMENT total_rabbits BY groups MULTIPLIED BY group_size  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
