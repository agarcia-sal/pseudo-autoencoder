CLASS Solution  
    FUNCTION numTilings(n)  
        SET MOD TO one billion PLUS seven  
        
        CREATE list dp OF zeroes WITH length n PLUS one  
        CREATE list dp1 OF zeroes WITH length n PLUS one  
        
        SET element at position zero of dp TO one  
        SET element at position one of dp TO one  
        
        IF n IS GREATER THAN OR EQUAL TO two  
            SET element at position two of dp TO two  
            SET element at position two of dp1 TO one  
        END IF  
        
        FOR i FROM three TO n  
            SET element at position i of dp TO the remainder after division of the sum of element at position i MINUS one of dp PLUS element at position i MINUS two of dp PLUS two MULTIPLIED BY element at position i MINUS one of dp1 BY MOD  
            SET element at position i of dp1 TO the remainder after division of the sum of element at position i MINUS two of dp PLUS element at position i MINUS one of dp1 BY MOD  
        END FOR  
        
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
