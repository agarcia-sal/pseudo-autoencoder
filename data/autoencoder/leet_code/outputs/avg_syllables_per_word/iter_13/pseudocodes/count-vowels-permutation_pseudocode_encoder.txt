CLASS Solution  
    FUNCTION countVowelPermutation WITH PARAMETER n  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven  
        SET a TO one  
        SET e TO one  
        SET i TO one  
        SET o TO one  
        SET u TO one  
        FOR each number FROM two TO n  
            SET a_next TO the SUM OF e AND i AND u  
            SET e_next TO the SUM OF a AND i  
            SET i_next TO the SUM OF e AND o  
            SET o_next TO i  
            SET u_next TO the SUM OF i AND o  
            SET a TO the REMAINDER AFTER DIVIDING a_next BY MOD  
            SET e TO the REMAINDER AFTER DIVIDING e_next BY MOD  
            SET i TO the REMAINDER AFTER DIVIDING i_next BY MOD  
            SET o TO the REMAINDER AFTER DIVIDING o_next BY MOD  
            SET u TO the REMAINDER AFTER DIVIDING u_next BY MOD  
        END FOR  
        SET total_to_return TO the REMAINDER AFTER DIVIDING the SUM OF a AND e AND i AND o AND u BY MOD  
        RETURN total_to_return  
    END FUNCTION  
END CLASS
