CLASS Solution  
    FUNCTION shortestPalindrome WITH PARAMETER s  
        IF s EQUALS empty string  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome WITH PARAMETER check_str  
            RETURN check_str EQUALS the reverse of check_str  
        END FUNCTION  
        
        FOR index FROM the LENGTH OF s DOWN TO zero  
            IF is_palindrome WITH ARGUMENT the substring from position zero TO index MINUS one of s  
                SET to_add TO the reverse of the substring from position index TO the end of s  
                RETURN the concatenation of to_add AND s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
