CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO the LIST CREATED BY PAIRING each ELEMENT OF difficulty WITH the CORRESPONDING ELEMENT OF profit  
        SORT jobs BY DIFFICULTY IN ASCENDING ORDER  
        SORT worker IN ASCENDING ORDER  
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the LENGTH OF jobs  
        FOR each ability IN worker  
            WHILE job_index IS LESS THAN n AND the DIFFICULTY of the ELEMENT at POSITION job_index OF jobs IS LESS THAN OR EQUAL TO ability  
                SET best_profit TO the GREATER VALUE BETWEEN best_profit AND the PROFIT of the ELEMENT at POSITION job_index OF jobs  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        RETURN max_profit  
    END FUNCTION  
END CLASS
