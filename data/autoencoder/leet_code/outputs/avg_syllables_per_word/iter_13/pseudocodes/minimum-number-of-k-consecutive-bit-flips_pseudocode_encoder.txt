CLASS Solution  
    FUNCTION minKBitFlips(nums AS List OF integer, k AS integer) RETURNS integer  
        SET n TO the LENGTH OF nums  
        SET flip_count TO zero  
        INITIALIZE flipped TO a list of zeros with the LENGTH OF n PLUS one  
        SET flip_state TO zero  
        
        FOR index FROM zero TO n MINUS one  
            SET flip_state TO the EXCLUSIVE OR of flip_state AND the element at position index of flipped  
            
            IF the element at position index of nums EQUALS flip_state THEN  
                IF index PLUS k GREATER THAN n THEN  
                    RETURN negative one  
                END IF  
                SET flip_state TO the EXCLUSIVE OR of flip_state AND one  
                SET the element at position index PLUS k of flipped TO the EXCLUSIVE OR of the element at position index PLUS k of flipped AND one  
                INCREMENT flip_count BY one  
            END IF  
        END FOR  
        
        RETURN flip_count  
    END FUNCTION  
END CLASS
