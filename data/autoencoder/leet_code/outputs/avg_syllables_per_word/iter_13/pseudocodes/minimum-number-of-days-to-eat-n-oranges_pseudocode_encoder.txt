CLASS Solution  
    FUNCTION minDays WITH parameter n  
        FUNCTION dp WITH parameter remaining  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            RETURN one PLUS THE MINIMUM OF the remainder of remaining DIVIDED BY two PLUS dp of remaining DIVIDED BY two AND the remainder of remaining DIVIDED BY three PLUS dp of remaining DIVIDED BY three  
        END FUNCTION  
        RETURN dp of n  
END CLASS
