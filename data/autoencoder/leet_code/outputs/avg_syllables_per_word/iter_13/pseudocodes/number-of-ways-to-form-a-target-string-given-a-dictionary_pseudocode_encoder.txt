CLASS Solution  
    FUNCTION numWays(words AS List of strings, target AS string) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF the first element of words  
        SET m TO the LENGTH OF target  
        SET freq TO a list containing n elements where each element IS a default dictionary mapping characters TO zero  
        FOR each word IN words  
            FOR each character WITH index j IN word  
                INCREMENT the element at key character of the dictionary at position j of freq BY one  
            END FOR  
        END FOR  
        SET dp TO a two dimensional list with m plus one rows AND n plus one columns each initialized TO zero  
        FOR j FROM zero TO n  
            SET the element at row zero column j of dp TO one  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET the element at row i column j of dp TO the value at row i column j minus one of dp  
                IF the frequency of the character at position i minus one of target in the dictionary at position j minus one of freq IS GREATER THAN zero  
                    INCREMENT the element at row i column j of dp BY the product of the value at row i minus one column j minus one of dp AND the frequency just tested  
                    SET the element at row i column j of dp TO the remainder of the element at row i column j of dp DIVIDED BY MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at row m column n of dp  
    END FUNCTION  
END CLASS
