CLASS Solution  
    FUNCTION findIntegers WITH parameter n AS integer  
        SET bin_rep TO the binary representation of the number n starting from position two TO the end of that representation  
        SET length TO the LENGTH OF bin_rep  
        
        DEFINE dp AS a list with length incremented by one elements all set to zero  
        SET the element at position zero of dp TO one  
        SET the element at position one of dp TO two  
        
        FOR index FROM two TO length inclusive  
            SET the element at position index of dp TO the sum of the element at the position index minus one of dp AND the element at the position index minus two of dp  
        END FOR  
        
        SET result TO zero  
        SET prev_bit TO zero  
        
        FOR index FROM zero TO length minus one  
            IF the element at position index of bin_rep EQUALS one as a character  
                INCREMENT result BY the element at position length minus index minus one of dp  
                IF prev_bit EQUALS one  
                    DECREMENT result BY one  
                    BREAK out of the loop  
                END IF  
            END IF  
            SET prev_bit TO one IF the element at position index of bin_rep EQUALS one as a character OTHERWISE zero  
        END FOR  
        
        RETURN the sum of result AND one  
    END FUNCTION  
END CLASS
