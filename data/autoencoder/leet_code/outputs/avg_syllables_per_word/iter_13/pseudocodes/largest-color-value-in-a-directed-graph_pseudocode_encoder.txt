CLASS Solution  
    FUNCTION largestPathValue(colors, edges)  
        SET n TO the LENGTH OF colors  
        SET graph TO a mapping from each key TO an empty list by default  
        SET in_degree TO a list of zero elements of size n  
        
        FOR each pair u and v IN edges  
            APPEND v TO the list associated with u in graph  
            INCREMENT the element at position v of in_degree BY one  
        END FOR  
        
        SET queue TO a double ended queue containing each index i from zero TO n MINUS one for which the element at position i of in_degree EQUALS zero  
        SET dp TO a list containing n lists each of length twenty six filled with zero  
        SET visited TO zero  
        SET max_color_value TO zero  
        
        WHILE queue IS NOT empty  
            REMOVE the first element from queue and ASSIGN it TO node  
            INCREMENT the element at the position determined by the character code of the character at position node of colors MINUS the character code of lowercase a in the list at position node of dp BY one  
            IF max_color_value IS LESS THAN the element at the position determined by the character code of the character at position node of colors MINUS the character code of lowercase a in the list at position node of dp THEN  
                SET max_color_value TO that element  
            END IF  
            INCREMENT visited BY one  
            
            FOR each neighbor IN the list associated with node in graph  
                FOR c FROM zero TO twenty five  
                    SET the element at position c in the list at position neighbor of dp TO the greater value BETWEEN the current element at position c of that list AND the element at position c in the list at position node of dp  
                END FOR  
                DECREMENT the element at position neighbor of in_degree BY one  
                IF the element at position neighbor of in_degree EQUALS zero THEN  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF visited EQUALS n THEN  
            RETURN max_color_value  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
