CLASS Solution  
    FUNCTION minimumDeviation WITH parameter nums as List of Integer  
        SET max_heap TO empty list  
        SET min_value TO positive infinity  
        FOR each num IN nums  
            IF the remainder of num DIVIDED BY two EQUALS one THEN  
                SET num TO num MULTIPLIED BY two  
            END IF  
            IF min_value GREATER THAN num THEN  
                SET min_value TO num  
            END IF  
            PUSH the negative of num INTO max_heap using heap operation  
        END FOR  
        
        SET min_deviation TO positive infinity  
        
        WHILE the element at position zero of max_heap modulo two EQUALS zero  
            REMOVE the top element from max_heap using heap operation and negate it ASSIGN TO max_value  
            IF min_deviation GREATER THAN max_value MINUS min_value THEN  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY two ignoring remainder  
            IF min_value GREATER THAN max_value THEN  
                SET min_value TO max_value  
            END IF  
            PUSH the negative of max_value INTO max_heap using heap operation  
        END WHILE  
        
        IF min_deviation GREATER THAN the negative of the element at position zero of max_heap MINUS min_value THEN  
            SET min_deviation TO the negative of the element at position zero of max_heap MINUS min_value  
        END IF  
        
        RETURN min_deviation  
    END FUNCTION  
END CLASS
