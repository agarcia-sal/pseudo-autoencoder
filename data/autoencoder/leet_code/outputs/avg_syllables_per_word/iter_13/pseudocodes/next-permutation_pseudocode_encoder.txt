CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET i TO the LENGTH OF nums MINUS two  
        WHILE i IS GREATER THAN OR EQUAL TO zero AND element at position i of nums IS GREATER THAN OR EQUAL TO element at position i PLUS one of nums  
            DECREMENT i BY one  
        END WHILE  
        IF i IS GREATER THAN OR EQUAL TO zero  
            SET j TO the LENGTH OF nums MINUS one  
            WHILE element at position j of nums IS LESS THAN OR EQUAL TO element at position i of nums  
                DECREMENT j BY one  
            END WHILE  
            SET temporary TO element at position i of nums  
            SET element at position i of nums TO element at position j of nums  
            SET element at position j of nums TO temporary  
        END IF  
        SET left TO i PLUS one  
        SET right TO the LENGTH OF nums MINUS one  
        WHILE left IS LESS THAN right  
            SET temporary TO element at position left of nums  
            SET element at position left of nums TO element at position right of nums  
            SET element at position right of nums TO temporary  
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
    END FUNCTION  
END CLASS
