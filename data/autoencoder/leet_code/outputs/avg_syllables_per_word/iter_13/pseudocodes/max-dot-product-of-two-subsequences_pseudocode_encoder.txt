CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        
        SET dp TO two dimensional list OF size m PLUS one BY n PLUS one  
        FOR each index i FROM zero TO m  
            FOR each index j FROM zero TO n  
                SET element at position j of element at position i of dp TO the NEGATIVE OF INFINITY CONSTANT  
            END FOR  
        END FOR  
        
        FOR each index i FROM one TO m  
            FOR each index j FROM one TO n  
                SET current_product TO element at position i MINUS one of nums1 MULTIPLIED BY element at position j MINUS one of nums2  
                SET element at position j of element at position i of dp TO THE MAXIMUM OF current_product AND element at position j of element at position i MINUS one of dp AND element at position j MINUS one of element at position i of dp AND element at position j MINUS one of element at position i MINUS one of dp PLUS current_product  
            END FOR  
        END FOR  
        
        RETURN element at position n of element at position m of dp  
    END FUNCTION  
END CLASS
