CLASS Solution  
    FUNCTION minMutation(start string, end string, bank list of strings) RETURNS integer  
        SET s TO the SET constructed FROM bank  
        SET q TO an empty double ended queue  
        APPEND the PAIR of start AND zero TO q  
        SET mp TO a MAPPING from character A TO the string consisting of characters T C G  
        SET mp AT character T TO the string consisting of characters A C G  
        SET mp AT character C TO the string consisting of characters A T G  
        SET mp AT character G TO the string consisting of characters A T C  
        WHILE the LENGTH OF q is greater than zero  
            REMOVE the FIRST element FROM q and ASSIGN the first part of the PAIR TO t and the second part to step  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR index i FROM zero TO the LENGTH OF t MINUS one  
                SET character v TO the element at position i of string t  
                FOR each character j IN the string mp AT key v  
                    SET next TO the concatenation of the substring FROM start position zero TO position i MINUS one of string t  
                    FOLLOWED BY character j  
                    FOLLOWED BY the substring FROM position i PLUS one TO the last position of string t  
                    IF next IS AN ELEMENT OF s  
                        APPEND the PAIR of next AND step PLUS one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
