CLASS Solution  
    FUNCTION minimumXORSum(nums1 AS List OF integer, nums2 AS List OF integer)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a list of INFINITY repeated TWO TO THE POWER OF n times  
        SET element at position zero of dp TO zero  
        
        FOR each mask FROM zero TO TWO TO THE POWER OF n MINUS one  
            SET bit_count TO the NUMBER OF ONES IN THE BINARY REPRESENTATION OF mask  
            FOR each j FROM zero TO n MINUS one  
                IF the BITWISE AND of mask AND TWO TO THE POWER OF j EQUALS TRUE  
                    SET dp AT position mask TO the MINIMUM OF dp AT position mask AND dp AT position BITWISE XOR OF mask AND TWO TO THE POWER OF j PLUS the BITWISE XOR of element at position bit_count MINUS one of nums1 AND element at position j of nums2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position BITWISE XOR OF TWO TO THE POWER OF n MINUS one of dp  
    END FUNCTION  
END CLASS
