CLASS Solution  
    FUNCTION restoreIpAddresses(s)  
        FUNCTION is_valid(segment)  
            IF the LENGTH OF segment EQUALS one OR (the element at position zero of segment NOT EQUALS zero AND the INTEGER VALUE OF segment LESS THAN OR EQUAL TO two hundred fifty five)  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        FUNCTION backtrack(start, path)  
            IF the LENGTH OF path EQUALS four  
                IF start EQUALS the LENGTH OF s  
                    APPEND the CONCATENATION OF the ELEMENTS OF path SEPARATED BY the DOT TO result  
                END IF  
                RETURN  
            END IF  
            FOR length FROM one TO three  
                IF start PLUS length LESS THAN OR EQUAL TO the LENGTH OF s  
                    SET segment TO the substring from position start TO start PLUS length MINUS one of string s  
                    IF is_valid(segment)  
                        CALL backtrack WITH start PLUS length AND path PLUS list containing segment  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  

        SET result TO empty list  
        CALL backtrack WITH zero AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
