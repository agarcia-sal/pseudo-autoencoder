CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary mapping zero TO minus one  
        SET max_length TO zero  
        SET count TO zero  
        FOR each index and num IN enumeration of nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            IF count EXISTS AS A KEY IN prefix_sum_index  
                SET max_length TO the greater value BETWEEN max_length AND index MINUS the element at key count OF prefix_sum_index  
            ELSE  
                SET the element at key count OF prefix_sum_index TO index  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
