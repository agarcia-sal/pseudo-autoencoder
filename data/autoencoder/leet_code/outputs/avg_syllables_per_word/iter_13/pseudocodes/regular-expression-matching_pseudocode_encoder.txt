CLASS Solution  
    FUNCTION isMatch(s, p)  
        CREATE two dimensional list dp with dimensions as the LENGTH OF s PLUS one BY the LENGTH OF p PLUS one FILLED WITH false values  
        SET element at position zero zero of dp TO true  
        FOR index j FROM one TO the LENGTH OF p  
            IF element at position j MINUS one of p EQUALS to the character asterisk  
                SET element at position zero j of dp TO element at position zero j MINUS two of dp  
            END IF  
        END FOR  
        FOR index i FROM one TO the LENGTH OF s  
            FOR index j FROM one TO the LENGTH OF p  
                IF element at position j MINUS one of p EQUALS to element at position i MINUS one of s OR element at position j MINUS one of p EQUALS to character dot  
                    SET element at position i j of dp TO element at position i MINUS one j MINUS one of dp  
                ELSE IF element at position j MINUS one of p EQUALS to character asterisk  
                    SET element at position i j of dp TO element at position i j MINUS two of dp  
                    IF element at position j MINUS two of p EQUALS to element at position i MINUS one of s OR element at position j MINUS two of p EQUALS to character dot  
                        SET element at position i j of dp TO element at position i j of dp OR element at position i MINUS one j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the LENGTH OF s the LENGTH OF p of dp  
    END FUNCTION  
END CLASS
