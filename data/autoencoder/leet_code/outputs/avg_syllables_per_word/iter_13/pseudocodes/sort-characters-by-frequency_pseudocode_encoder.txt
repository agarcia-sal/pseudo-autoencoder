CLASS Solution  
    FUNCTION frequencySort WITH parameter s  
        SET frequency TO CALL Counter WITH s  
        SET sorted_characters TO SORT THE COLLECTION OF KEYS OF frequency BY THE FREQUENCY OF THE CHARACTER IN frequency IN DESCENDING ORDER  
        SET result TO AN EMPTY STRING  
        FOR each character IN sorted_characters  
            APPEND character REPEATED THE FREQUENCY OF character IN frequency TIMES TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
