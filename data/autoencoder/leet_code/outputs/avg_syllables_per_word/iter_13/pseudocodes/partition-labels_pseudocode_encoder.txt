CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO a new empty dictionary  
        FOR each index i FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position i of s  
            SET the value at key character in last_occurrence TO index i  
        END FOR  
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO a new empty list  
        FOR each index i FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position i of s  
            IF the value at key character in last_occurrence IS GREATER THAN end  
                SET end TO the value at key character in last_occurrence  
            END IF  
            IF index i EQUALS end  
                SET partition_size TO end MINUS start PLUS one  
                APPEND partition_size TO partition_sizes  
                SET start TO index i PLUS one  
            END IF  
        END FOR  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
