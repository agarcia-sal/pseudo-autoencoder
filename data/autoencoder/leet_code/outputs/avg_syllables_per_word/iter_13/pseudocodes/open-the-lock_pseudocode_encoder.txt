CLASS Solution  
    FUNCTION openLock(deadends, target)  
        SET deadends_set TO a set constructed from deadends  
        
        IF the string zero zero zero zero IS an element of deadends_set  
            RETURN the integer negative one  
        END IF  
        
        IF target EQUALS the string zero zero zero zero  
            RETURN the integer zero  
        END IF  
        
        SET queue TO a double ended queue containing a tuple of the string zero zero zero zero and the integer zero  
        SET visited TO a set containing the string zero zero zero zero  
        
        FUNCTION neighbors(state)  
            SET result TO an empty list  
            FOR index FROM zero TO three  
                SET digit TO the integer converted from the element at position index of state  
                FOR move IN the list containing negative one and positive one  
                    SET new_digit TO the remainder of the sum of digit and move divided by ten  
                    SET new_state TO the concatenation of the substring from position zero TO index minus one of state, the string representation of new_digit, and the substring from position index plus one TO the end of state  
                    APPEND new_state TO result  
                END FOR  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        WHILE the length of queue IS greater than zero  
            REMOVE the element at the front of queue AND ASSIGN it TO current_state and steps  
            
            FOR each neighbor IN neighbors(current_state)  
                IF neighbor IS NOT an element of visited AND neighbor IS NOT an element of deadends_set  
                    IF neighbor EQUALS target  
                        RETURN the sum of steps and one  
                    END IF  
                    ADD neighbor TO visited  
                    APPEND a tuple of neighbor and the sum of steps and one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the integer negative one  
    END FUNCTION  
END CLASS
