CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each element l IN left  
                FOR each element r IN right  
                    IF operator EQUALS the plus symbol  
                        APPEND the sum of l AND r TO results  
                    ELSE IF operator EQUALS the minus symbol  
                        APPEND the difference of l MINUS r TO results  
                    ELSE IF operator EQUALS the multiplication symbol  
                        APPEND the product of l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  

        FUNCTION helper(sub_expr)  
            IF sub_expr CONSISTS ONLY OF DIGITS  
                RETURN a list converted from sub_expr containing integer values  
            END IF  

            SET results TO empty list  
            FOR each index i FROM zero TO the LENGTH OF sub_expr MINUS one  
                IF the element at position i of sub_expr IS ONE OF the plus symbol OR the minus symbol OR the multiplication symbol  
                    SET left_results TO the return value of helper with the substring from position zero TO the position one LESS THAN i of sub_expr  
                    SET right_results TO the return value of helper with the substring from position one GREATER THAN i TO the end of sub_expr  
                    APPEND all elements from the return value of compute with left_results AND right_results AND the element at position i of sub_expr TO results  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  

        RETURN the return value of helper with expression  
    END FUNCTION  
END CLASS
