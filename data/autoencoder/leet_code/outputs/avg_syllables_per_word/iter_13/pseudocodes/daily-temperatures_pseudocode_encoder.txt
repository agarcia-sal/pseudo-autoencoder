CLASS Solution  
    FUNCTION dailyTemperatures WITH parameter temperatures AS list of integers  
        SET n TO the LENGTH OF temperatures  
        INITIALIZE answer AS a list OF zeroes WITH the LENGTH OF n  
        INITIALIZE stack AS an empty list  
        FOR index i FROM zero TO n MINUS one  
            WHILE the stack is NOT empty AND the element at position i of temperatures IS GREATER THAN the element at the position at the last element of stack of temperatures  
                REMOVE the last element FROM stack AND ASSIGN it TO prev_index  
                SET the element at position prev_index of answer TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
