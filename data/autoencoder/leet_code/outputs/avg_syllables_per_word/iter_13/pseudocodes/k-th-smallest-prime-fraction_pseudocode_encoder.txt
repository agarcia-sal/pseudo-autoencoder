CLASS Solution  
    FUNCTION kthSmallestPrimeFraction WITH PARAMETERS arr AND k  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  
        FOR each index i FROM zero TO n MINUS one EXCLUSIVE  
            ASSIGN fraction TO element at position i of arr DIVIDED BY element at position n MINUS one of arr  
            PUSH onto min_heap the tuple consisting of fraction AND i AND n MINUS one  
        END FOR  
        FOR each iteration FROM zero TO k MINUS one EXCLUSIVE  
            POP the smallest tuple FROM min_heap INTO variables discarded_fraction AND i AND j  
            IF j MINUS one IS GREATER THAN i THEN  
                ASSIGN new_fraction TO element at position i of arr DIVIDED BY element at position j MINUS one of arr  
                PUSH onto min_heap the tuple consisting of new_fraction AND i AND j MINUS one  
            END IF  
        END FOR  
        POP the smallest tuple FROM min_heap INTO variables discarded_fraction AND i AND j  
        RETURN a list consisting of element at position i of arr AND element at position j of arr  
    END FUNCTION  
END CLASS
