CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator_over_s TO an iterator over s  
            RETURN the result of testing that for every character IN word the character IS IN iterator_over_s in sequence  
        END FUNCTION  

        SORT dictionary BY the LENGTH OF each element in descending order AND BY lexicographical order in ascending order  

        FOR each word IN dictionary  
            IF the result of is_subsequence with parameters word and s IS TRUE  
                RETURN word  
            END IF  
        END FOR  

        RETURN an empty string  
    END FUNCTION  
END CLASS
