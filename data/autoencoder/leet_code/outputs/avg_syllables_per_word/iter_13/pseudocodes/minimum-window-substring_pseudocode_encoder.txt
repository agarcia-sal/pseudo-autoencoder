CLASS Solution  
    FUNCTION minWindow(s STRING, t STRING) RETURNS STRING  
        SET t_count TO a frequency counter of each character in t  
        SET required TO the number of unique characters in t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero representing the number of unique characters from t satisfied in the current window  
        SET window_counts TO an empty frequency counter  
        SET min_length TO positive infinity  
        SET min_start TO zero  
        
        WHILE the value of right is LESS THAN the LENGTH OF s  
            INCREMENT the count of the character at position right of s in window_counts BY one  
            
            IF the character at position right of s is present IN t_count  
            AND the count of the character at position right of s in window_counts EQUALS the count in t_count  
                INCREMENT formed BY one  
            END IF  
            
            WHILE left is LESS THAN OR EQUAL TO right  
            AND formed EQUALS required  
                SET char TO the character at position left of s  
                
                IF the difference between right and left PLUS one is LESS THAN min_length  
                    SET min_length TO the difference between right and left PLUS one  
                    SET min_start TO left  
                END IF  
                
                DECREMENT the count of char in window_counts BY one  
                IF char is present IN t_count  
                AND the count of char in window_counts is LESS THAN the count in t_count  
                    DECREMENT formed BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT right BY one  
        END WHILE  
        
        IF min_length IS EQUAL TO positive infinity  
            RETURN an empty string  
        ELSE  
            RETURN the substring of s starting at position min_start TO position min_start PLUS min_length MINUS one  
        END IF  
    END FUNCTION  
END CLASS
