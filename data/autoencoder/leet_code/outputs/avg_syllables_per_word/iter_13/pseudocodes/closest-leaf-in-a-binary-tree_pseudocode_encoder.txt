CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new instance of TreeNode with parameter value at position zero of values  
    SET i TO one  
    SET queue TO new empty deque collection  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO new instance of TreeNode with parameter value at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO new instance of TreeNode with parameter value at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val IS NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with parameters p dot left and q dot left AND the result of is_same_tree with parameters p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(root, k)  
        IF root dot left IS none AND root dot right IS none  
            RETURN root dot val  
        END IF  
        SET graph TO new default dictionary of list  
        
        FUNCTION dfs(node, parent)  
            IF parent IS NOT none  
                APPEND parent dot val TO the list associated with key node dot val IN graph  
                APPEND node dot val TO the list associated with key parent dot val IN graph  
            END IF  
            IF node dot left IS NOT none  
                CALL dfs WITH parameters node dot left AND node  
            END IF  
            IF node dot right IS NOT none  
                CALL dfs WITH parameters node dot right AND node  
            END IF  
        END FUNCTION  
        
        CALL dfs WITH parameters root AND none  
        
        SET queue TO new deque initialized with element k  
        SET visited TO new empty set  
        
        WHILE queue IS NOT empty  
            SET current TO the element removed from the left of queue  
            ADD current TO visited  
            
            IF the LENGTH OF the list associated with key current IN graph IS EQUALS one AND current IS NOT EQUALS root dot val  
                RETURN current  
            END IF  
            
            FOR each neighbor IN the list associated with key current IN graph  
                IF neighbor IS NOT in visited  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
