CLASS Solution  
    FUNCTION reorderedPowerOf2 WITH PARAMETER n AS integer  
        SET power_of_2_sorted_digits TO an empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 IS LESS THAN OR EQUAL TO one billion  
            ADD the tuple of the digits of power_of_2 SORTED TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  
        SET sorted_n_digits TO the tuple of the digits of n SORTED  
        RETURN whether sorted_n_digits IS IN power_of_2_sorted_digits  
    END FUNCTION  
END CLASS
