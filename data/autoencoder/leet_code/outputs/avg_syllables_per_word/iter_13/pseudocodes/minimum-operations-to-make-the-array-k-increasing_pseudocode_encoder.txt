CLASS Solution  
    FUNCTION kIncreasing(arr AS List OF integer, k AS integer) RETURNS integer  
        FUNCTION longest_increasing_subsequence(subarr AS List OF integer) RETURNS integer  
            SET lis TO empty list  
            FOR each number IN subarr  
                IF lis IS empty OR number IS GREATER THAN OR EQUAL TO element at position LENGTH OF lis MINUS one OF lis  
                    APPEND number TO lis  
                ELSE  
                    SET pos TO the position where number CAN BE INSERTED TO MAINTAIN sorted order using bisect right method on lis  
                    SET element at position pos OF lis TO number  
                END IF  
            END FOR  
            RETURN LENGTH OF lis  
        END FUNCTION  
        
        SET total_operations TO zero  
        SET n TO LENGTH OF arr  
        
        FOR start FROM zero TO k MINUS one  
            SET subarr TO empty list  
            FOR i FROM start TO n MINUS one STEP k  
                APPEND element at position i OF arr TO subarr  
            END FOR  
            SET lis_length TO longest_increasing_subsequence(subarr)  
            INCREMENT total_operations BY LENGTH OF subarr MINUS lis_length  
        END FOR  
        
        RETURN total_operations  
    END FUNCTION  
END CLASS
