CLASS Solution  
    FUNCTION kEmptySlots(bulbs AS List OF integers, k AS integer) RETURNS integer  
        SET bloom_days TO empty list  
        FOR each day and position IN enumeration of bulbs starting from one  
            SET insertion_index TO the position where position would be inserted in bloom_days to keep it sorted  
            IF insertion_index is greater than zero AND position MINUS element at position insertion_index MINUS one of bloom_days EQUALS k  
                RETURN day  
            END IF  
            IF insertion_index is less than the LENGTH OF bloom_days AND element at position insertion_index of bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position INTO bloom_days at insertion_index to keep bloom_days sorted  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
