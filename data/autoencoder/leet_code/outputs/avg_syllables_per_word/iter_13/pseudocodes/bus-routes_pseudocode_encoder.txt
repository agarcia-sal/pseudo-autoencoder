CLASS Solution  
    FUNCTION numBusesToDestination(routes AS List OF List OF Integer, source AS Integer, target AS Integer) RETURNS Integer  
        IF source EQUALS target THEN  
            RETURN zero  
        END IF  

        SET stop_to_buses TO a new default dictionary with lists as default value  

        FOR each bus WITH index bus_number AND route IN routes  
            FOR each stop IN route  
                APPEND bus_number TO the list corresponding to stop IN stop_to_buses  
            END FOR  
        END FOR  

        SET bus_queue TO a new double-ended queue  
        SET visited_buses TO a new empty set  
        SET visited_stops TO a new empty set  

        FOR each bus IN the list corresponding to source IN stop_to_buses  
            APPEND the tuple containing bus AND one TO bus_queue  
            ADD bus TO visited_buses  
            ADD source TO visited_stops  
        END FOR  

        WHILE bus_queue is not empty  
            REMOVE the leftmost element from bus_queue AND SET it TO current_bus AND bus_count  

            FOR each stop IN the route corresponding to current_bus in routes  
                IF stop EQUALS target THEN  
                    RETURN bus_count  
                END IF  

                IF stop is not contained in visited_stops THEN  
                    ADD stop TO visited_stops  
                    FOR each next_bus IN the list corresponding to stop IN stop_to_buses  
                        IF next_bus is not contained in visited_buses THEN  
                            ADD next_bus TO visited_buses  
                            APPEND the tuple containing next_bus AND bus_count PLUS one TO bus_queue  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
