CLASS Solution  
    FUNCTION isTransformable(s, t)  
        ASSIGN pointer TO dictionary with default value an empty list named pos  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            ASSIGN variable c TO element at position index of s  
            CALL heappush with arguments element at position index of pos at key CONVERT c TO integer and index  
        END FOR  
        FOR index FROM zero TO the LENGTH OF t MINUS one  
            ASSIGN variable c TO element at position index of t  
            SET digit TO CONVERT c TO integer  
            IF list at key digit in pos is empty  
                RETURN False  
            END IF  
            FOR smaller FROM zero TO digit MINUS one  
                IF list at key smaller in pos is not empty AND first element at position zero of list at key smaller in pos IS LESS THAN first element at position zero of list at key digit in pos  
                    RETURN False  
                END IF  
            END FOR  
            CALL heappop with argument list at key digit in pos  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
