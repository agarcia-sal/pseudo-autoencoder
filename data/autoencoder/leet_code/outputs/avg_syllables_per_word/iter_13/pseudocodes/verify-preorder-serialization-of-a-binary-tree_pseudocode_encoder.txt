CLASS Solution  
    FUNCTION isValidSerialization(preorder)  
        SET nodes TO the result of splitting preorder by the delimiter comma  
        SET slots TO one  
        FOR each node IN nodes  
            IF slots EQUALS zero  
                RETURN False  
            END IF  
            DECREMENT slots BY one  
            IF node NOT EQUALS the character hash  
                INCREMENT slots BY two  
            END IF  
        END FOR  
        RETURN whether slots EQUALS zero  
    END FUNCTION  
END CLASS
