    FUNCTION pacificAtlantic WITH parameter heights  
        FUNCTION bfs WITH parameters q AND vis  
            WHILE q IS NOT EMPTY  
                FOR each element FROM zero TO THE LENGTH OF q MINUS one  
                    SET i AND j TO THE FIRST AND SECOND ELEMENTS OF THE ITEM REMOVED FROM THE FRONT OF q  
                    FOR each pair a AND b IN the COLLECTION of ROW ZERO COLUMN MINUS one ROW ZERO COLUMN ONE ROW ONE COLUMN ZERO AND ROW MINUS ONE COLUMN ZERO  
                        SET x TO i PLUS a  
                        SET y TO j PLUS b  
                        IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN m AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN n AND THE TUPLE x AND y IS NOT IN vis AND THE ELEMENT AT POSITION x OF heights AND POSITION y OF THAT ROW IS GREATER THAN OR EQUAL TO THE ELEMENT AT i AND j OF heights THEN  
                            ADD THE TUPLE x AND y TO vis  
                            APPEND THE TUPLE x AND y TO THE END OF q  
                        END IF  
                    END FOR  
                END FOR  
            END WHILE  
        END FUNCTION  

        SET m TO THE LENGTH OF heights  
        SET n TO THE LENGTH OF THE ELEMENT AT POSITION zero OF heights  
        SET vis1 TO AN EMPTY SET  
        SET vis2 TO AN EMPTY SET  
        SET q1 TO AN EMPTY DOUBLE ENDED QUEUE  
        SET q2 TO AN EMPTY DOUBLE ENDED QUEUE  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF i EQUALS zero OR j EQUALS zero THEN  
                    ADD THE TUPLE i AND j TO vis1  
                    APPEND THE TUPLE i AND j TO THE END OF q1  
                END IF  
                IF i EQUALS m MINUS one OR j EQUALS n MINUS one THEN  
                    ADD THE TUPLE i AND j TO vis2  
                    APPEND THE TUPLE i AND j TO THE END OF q2  
                END IF  
            END FOR  
        END FOR  
        CALL bfs WITH q1 AND vis1  
        CALL bfs WITH q2 AND vis2  
        SET result TO AN EMPTY LIST  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF THE TUPLE i AND j IS IN vis1 AND THE TUPLE i AND j IS IN vis2 THEN  
                    APPEND THE TUPLE i AND j TO result  
                END IF  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
