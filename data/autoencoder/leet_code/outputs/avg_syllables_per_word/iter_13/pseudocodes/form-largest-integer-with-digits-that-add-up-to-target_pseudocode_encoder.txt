CLASS Solution  
    FUNCTION largestNumber(cost AS List OF integers, target AS integer) RETURNS string  
        SET cost_to_digit TO an empty mapping from integer TO string  
        FOR each index AND each cost_value IN cost WITH their positions  
            SET digit_string TO string representation of index PLUS one  
            ASSIGN cost_value TO digit_string IN cost_to_digit  
        END FOR  
        
        SET dp TO a list with an empty string FOLLOWED BY target copies OF the string minus one  
        
        FOR t FROM one TO target  
            FOR each cost_value AND digit_string IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO cost_value AND element at position t MINUS cost_value of dp IS NOT the string minus one  
                    SET candidate TO concatenation of digit_string FOLLOWED BY element at position t MINUS cost_value of dp  
                    IF element at position t of dp IS the string minus one OR LENGTH OF candidate GREATER THAN LENGTH OF element at position t of dp OR LENGTH OF candidate EQUALS LENGTH OF element at position t of dp AND candidate LEXICOGRAPHICALLY GREATER THAN element at position t of dp  
                        SET element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF element at position target of dp IS NOT the string minus one  
            RETURN element at position target of dp  
        ELSE  
            RETURN the string zero  
        END IF  
    END FUNCTION  
END CLASS
