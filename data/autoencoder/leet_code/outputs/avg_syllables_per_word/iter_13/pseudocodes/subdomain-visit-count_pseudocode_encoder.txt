CLASS Solution  
    FUNCTION subdomainVisits WITH PARAMETER cpdomains  
        SET domain_count TO a new default mapping from string to integer  
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain INTO count and domain by whitespace  
            SET count TO the integer value of count  
            SPLIT domain INTO subdomains by the period word  
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one  
                SET subdomain TO the concatenation of the elements from position index TO the end of subdomains JOINED BY the period word  
                INCREMENT the value at key subdomain IN domain_count BY count  
            END FOR  
        END FOR  
        RETURN a list formed by for each domain and count pair IN the items of domain_count CREATE a string with count AS prefix FOLLOWED BY a whitespace FOLLOWED BY domain  
    END FUNCTION  
END CLASS
