CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a dictionary mapping each character from alphabet letters a to z TO zero  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND the NUMERIC UNICODE VALUE OF element at position index of s MINUS the NUMERIC UNICODE VALUE OF element at position index MINUS one of s EQUALS one OR element at position index of s EQUALS letter a AND element at position index MINUS one of s EQUALS letter z  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET max_length at key element at position index of s TO the GREATER OF max_length at key element at position index of s AND current_length  
        END FOR  
        RETURN the SUM OF all values in max_length  
    END FUNCTION  
END CLASS
