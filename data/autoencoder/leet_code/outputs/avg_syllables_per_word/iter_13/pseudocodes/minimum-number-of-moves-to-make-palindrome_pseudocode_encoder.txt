CLASS Solution  
    FUNCTION minMovesToMakePalindrome WITH PARAMETER s of type string  
        SET s TO list converted from s  
        SET n TO the LENGTH OF s  
        SET moves TO zero  
        
        WHILE the LENGTH OF s IS GREATER THAN one  
            FOR i FROM the LENGTH OF s MINUS one DOWN TO one DECREMENTING BY one  
                IF the element at position i of s EQUALS the element at position zero of s  
                    FOR j FROM i TO the LENGTH OF s MINUS two  
                        SET temporary TO the element at position j PLUS one of s  
                        SET element at position j PLUS one of s TO the element at position j of s  
                        SET element at position j of s TO temporary  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE element at position zero FROM s  
                    REMOVE element at the last position FROM s  
                    BREAK out of the FOR loop  
                END IF  
            END FOR  
            IF no BREAK occurred in the FOR loop  
                REMOVE element at position zero FROM s  
            END IF  
        END WHILE  
        
        RETURN moves  
    END FUNCTION  
END CLASS
