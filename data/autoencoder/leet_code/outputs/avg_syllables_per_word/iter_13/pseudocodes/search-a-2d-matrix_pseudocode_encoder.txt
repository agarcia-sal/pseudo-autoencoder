CLASS Solution  
    FUNCTION searchMatrix(matrix WITH TYPE LIST OF LIST OF INTEGERS, target WITH TYPE INTEGER) RETURNS BOOLEAN  
        IF matrix IS EMPTY OR element at position zero of matrix IS EMPTY  
            RETURN false  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER QUOTIENT OF left PLUS right DIVIDED BY two  
            SET mid_value TO element at position integer quotient of mid DIVIDED BY n of matrix AT position remainder of mid DIVIDED BY n  
            IF mid_value EQUALS target  
                RETURN true  
            ELSE IF mid_value IS LESS THAN target  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN false  
    END FUNCTION  
END CLASS
