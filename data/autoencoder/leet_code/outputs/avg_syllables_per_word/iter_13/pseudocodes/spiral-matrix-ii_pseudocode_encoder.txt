CLASS Solution  
    FUNCTION generateMatrix(n)  
        SET matrix TO a two dimensional list with n rows each containing n zeros  
        SET top TO zero  
        SET bottom TO n MINUS one  
        SET left TO zero  
        SET right TO n MINUS one  
        SET num TO one  
        WHILE top IS LESS THAN OR EQUAL TO bottom AND left IS LESS THAN OR EQUAL TO right  
            FOR each i FROM left TO right  
                SET element at position top of matrix at position i TO num  
                INCREMENT num BY one  
            END FOR  
            INCREMENT top BY one  
            FOR each i FROM top TO bottom  
                SET element at position i of matrix at position right TO num  
                INCREMENT num BY one  
            END FOR  
            DECREMENT right BY one  
            IF top IS LESS THAN OR EQUAL TO bottom  
                FOR each i FROM right DOWN TO left  
                    SET element at position bottom of matrix at position i TO num  
                    INCREMENT num BY one  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            IF left IS LESS THAN OR EQUAL TO right  
                FOR each i FROM bottom DOWN TO top  
                    SET element at position i of matrix at position left TO num  
                    INCREMENT num BY one  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        RETURN matrix  
    END FUNCTION  
END CLASS
