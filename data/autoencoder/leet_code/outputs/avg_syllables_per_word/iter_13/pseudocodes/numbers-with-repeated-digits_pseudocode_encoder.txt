CLASS Solution  
    FUNCTION numDupDigitsAtMostN(n)  
        FUNCTION count_unique_digits(x)  
            SET s TO THE STRING REPRESENTATION OF x  
            SET length TO THE LENGTH OF s  
            SET count TO THE SUM OVER i FROM ONE TO length MINUS ONE OF nine MULTIPLIED BY perm WITH PARAMETERS nine MINUS i PLUS ONE AND i MINUS one  
            SET seen TO THE EMPTY SET  
            FOR each index i AND character digit IN s WITH INDEX  
                FOR d FROM IF i EQUALS zero THEN one ELSE zero TO THE INTEGER VALUE OF digit MINUS one  
                    IF d NOT IN seen  
                        INCREMENT count BY perm WITH PARAMETERS nine MINUS i AND length MINUS i MINUS one  
                    END IF  
                END FOR  
                IF THE INTEGER VALUE OF digit IN seen  
                    BREAK THE LOOP  
                END IF  
                ADD THE INTEGER VALUE OF digit TO seen  
            END FOR ELSE  
                INCREMENT count BY one  
            END ELSE  
            RETURN count  
        END FUNCTION  
        RETURN n MINUS count_unique_digits WITH PARAMETER n  
    END FUNCTION  

    FUNCTION perm(m, n)  
        IF n EQUALS zero  
            RETURN one  
        END IF  
        SET result TO one  
        FOR i FROM zero TO n MINUS one  
            SET result TO result MULTIPLIED BY m MINUS i  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
