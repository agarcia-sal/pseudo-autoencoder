CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH parameter element at position zero of values  
    SET i TO one  
    SET queue TO NEW deque collection  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE element at position zero FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH parameter element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH parameter element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val IS NOT EQUALS TO q dot val  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH parameters p dot left AND q dot left AND is_same_tree WITH parameters p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree(self, root)  
        IF root IS none OR root dot left IS none  
            RETURN root  
        END IF  
        SET new_root TO self dot upsideDownBinaryTree WITH parameter root dot left  
        SET root dot left dot left TO root dot right  
        SET root dot left dot right TO root  
        SET root dot left TO none  
        SET root dot right TO none  
        RETURN new_root  
    END FUNCTION  
END CLASS
