CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO a dictionary with key zero and value zero  
        
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each key d and value y IN dp  
                SET key_added_to_taller TO d PLUS rod  
                IF new_dp contains key key_added_to_taller THEN  
                    SET value_at_key TO the value at key key_added_to_taller of new_dp  
                    SET max_value TO the greater value BETWEEN value_at_key AND y  
                    SET the value at key key_added_to_taller of new_dp TO max_value  
                ELSE  
                    SET the value at key key_added_to_taller of new_dp TO y  
                END IF  
                
                SET difference_after_subtraction TO the absolute value of d MINUS rod  
                SET min_between_d_and_rod TO the smaller value BETWEEN d AND rod  
                SET new_value TO y PLUS min_between_d_and_rod  
                IF new_dp contains key difference_after_subtraction THEN  
                    SET value_at_difference TO the value at key difference_after_subtraction of new_dp  
                    SET max_value_to_set TO the greater value BETWEEN value_at_difference AND new_value  
                    SET the value at key difference_after_subtraction of new_dp TO max_value_to_set  
                ELSE  
                    SET the value at key difference_after_subtraction of new_dp TO new_value  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the value at key zero of dp  
    END FUNCTION  
END CLASS
