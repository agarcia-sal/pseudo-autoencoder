CLASS Solution  
    FUNCTION kSmallestPairs(nums1 OF LIST OF INTEGER nums2 OF LIST OF INTEGER k OF INTEGER) RETURNS LIST OF LIST OF INTEGER  
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k IS LESS THAN OR EQUAL TO zero THEN  
            RETURN EMPTY LIST  
        END IF  
        
        SET min_heap TO EMPTY LIST  
        FOR i FROM zero TO THE MINIMUM OF k AND THE LENGTH OF nums1 MINUS one  
            CALL THE HEAP PUSH FUNCTION TO ADD A TUPLE CONSISTING OF THE SUM OF element at position i of nums1 AND element at position zero of nums2, i, AND zero TO min_heap  
        END FOR  
        
        SET result TO EMPTY LIST  
        WHILE min_heap IS NOT EMPTY AND THE LENGTH OF result IS LESS THAN k  
            CALL THE HEAP POP FUNCTION TO REMOVE AND RETURN THE SMALLEST TUPLE FROM min_heap AND ASSIGN IT TO underscore IGNORE THE FIRST ELEMENT, i, AND j  
            APPEND A LIST CONSISTING OF element at position i of nums1 AND element at position j of nums2 TO result  
            IF j PLUS one IS LESS THAN THE LENGTH OF nums2 THEN  
                CALL THE HEAP PUSH FUNCTION TO ADD A TUPLE CONSISTING OF THE SUM OF element at position i of nums1 AND element at position j PLUS one of nums2, i, AND j PLUS one TO min_heap  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
