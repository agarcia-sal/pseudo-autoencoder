CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO NEW ListNode WITH value element at position zero of values  
    SET p TO head  
    FOR each val IN elements from position one TO the last of values  
        SET node TO NEW ListNode WITH value val  
        SET pointer next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the value of element p1 EQUALS the value of element p2 AND is_same_list of pointer next of p1 AND pointer next of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH value element at position zero of values  
    SET i TO one  
    SET queue TO NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET pointer left of node TO NEW TreeNode WITH value element at position i of values  
            APPEND pointer left of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET pointer right of node TO NEW TreeNode WITH value element at position i of values  
            APPEND pointer right of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value of element p IS NOT EQUAL TO the value of element q  
        RETURN false  
    ELSE  
        RETURN is_same_tree of pointer left of p AND pointer left of q AND is_same_tree of pointer right of p AND pointer right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST(self, head)  
        IF head IS none  
            RETURN none  
        END IF  

        FUNCTION findMiddle(left, right)  
            SET slow TO left  
            SET fast TO left  
            WHILE fast IS NOT EQUAL TO right AND pointer next of fast IS NOT EQUAL TO right  
                SET slow TO pointer next of slow  
                SET fast TO pointer next of pointer next of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  

        FUNCTION convertListToBST(left, right)  
            IF left IS EQUAL TO right  
                RETURN none  
            END IF  
            SET mid TO findMiddle(left, right)  
            SET node TO NEW TreeNode WITH value the value of element mid  
            SET pointer left of node TO convertListToBST(left, mid)  
            SET pointer right of node TO convertListToBST(pointer next of mid, right)  
            RETURN node  
        END FUNCTION  

        RETURN convertListToBST(head, none)  
    END FUNCTION  
END CLASS
