CLASS Solution  
    FUNCTION subsets WITH nums AS list of integer  
        IF the LENGTH OF nums EQUALS zero  
            RETURN a list containing an empty list  
        END IF  
        SET subsets_without_first TO the RESULT OF calling subsets WITH the sublist of nums from position one TO the END  
        SET subsets_with_first TO an empty list  
        FOR each subset IN subsets_without_first  
            SET new_subset TO a list consisting of the element at position zero of nums FOLLOWED BY the elements of subset  
            APPEND new_subset TO subsets_with_first  
        END FOR  
        RETURN the concatenation of subsets_without_first AND subsets_with_first  
    END FUNCTION  
END CLASS
