CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of fifteen elements each initialized to infinity  
        FOR each pair of fixed_time AND rate IN tires  
            SET total_time TO zero  
            FOR each index FROM zero TO fourteen  
                SET lap_time TO fixed_time MULTIPLIED BY rate RAISED TO THE POWER OF index  
                IF lap_time IS GREATER THAN changeTime PLUS fixed_time  
                    EXIT THE LOOP  
                END IF  
                INCREMENT total_time BY lap_time  
                SET min_times at position index TO the LOWER OF min_times at position index AND total_time  
            END FOR  
        END FOR  
        
        SET dp TO a list of numLaps PLUS one elements each initialized to infinity  
        SET dp at position zero TO zero  
        
        FOR each i FROM one TO numLaps  
            FOR each j FROM zero TO the LOWER OF i MINUS one AND fourteen  
                SET current_value TO dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
                SET dp at position i TO the LOWER OF dp at position i AND current_value  
            END FOR  
        END FOR  
        
        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
