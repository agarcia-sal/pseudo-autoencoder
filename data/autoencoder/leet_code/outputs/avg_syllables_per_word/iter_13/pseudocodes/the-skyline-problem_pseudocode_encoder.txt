CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO empty list  
        FOR each building IN buildings  
            SET left TO element at position zero of building  
            SET right TO element at position one of building  
            SET height TO element at position two of building  
            APPEND a tuple composed of left AND negative height AND right TO events  
            APPEND a tuple composed of right AND zero AND zero TO events  
        END FOR  
        SORT events BY first element then by second element then by third element in ascending order  
        SET result TO list containing one list of zero AND zero  
        SET max_heap TO list containing one tuple of zero AND positive infinity  
        FOR each event IN events  
            SET x TO element at position zero of event  
            SET negH TO element at position one of event  
            SET r TO element at position two of event  
            WHILE the second element of the first tuple in max_heap IS LESS THAN OR EQUAL TO x  
                REMOVE the first tuple FROM max_heap using heap pop operation  
            END WHILE  
            IF negH IS NOT EQUAL TO zero  
                INSERT the tuple composed of negH AND r INTO max_heap using heap push operation  
            END IF  
            IF the second element of the last list in result IS NOT EQUAL TO negative of the first element of the first tuple in max_heap  
                APPEND list composed of x AND negative of the first element of the first tuple in max_heap TO result  
            END IF  
        END FOR  
        IF the first list in result IS EQUAL TO list of zero AND zero  
            REMOVE the first list FROM result  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
