CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val DEFAULT zero left DEFAULT none right DEFAULT none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF the collection values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE the collection queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO a new TreeNode initialized with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO a new TreeNode initialized with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUAL TO q dot val  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree WITH PARAMETERS p dot left q dot left AND the result of calling is_same_tree WITH PARAMETERS p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION printTree WITH PARAMETERS self root  
        FUNCTION getHeight WITH PARAMETERS node  
            IF node IS none  
                RETURN zero  
            END IF  
            RETURN one PLUS the maximum value BETWEEN the result of calling getHeight WITH PARAMETERS node dot left AND the result of calling getHeight WITH PARAMETERS node dot right  
        END FUNCTION  
        
        SET height TO the result of calling getHeight WITH PARAMETERS root  
        SET width TO two RAISED TO THE POWER OF height MINUS one  
        SET res TO a new list WITH height times containing a list WITH width times of empty string elements  
        
        FUNCTION placeNode WITH PARAMETERS node row left right  
            IF node IS none  
                RETURN  
            END IF  
            SET mid TO the integer division of the sum of left AND right  
            SET the element at position mid of the list at position row of res TO the string representation of node dot val  
            CALL placeNode WITH PARAMETERS node dot left row PLUS one left mid  
            CALL placeNode WITH PARAMETERS node dot right row PLUS one mid right  
        END FUNCTION  
        
        CALL placeNode WITH PARAMETERS root zero zero width  
        RETURN res  
    END FUNCTION  
END CLASS
