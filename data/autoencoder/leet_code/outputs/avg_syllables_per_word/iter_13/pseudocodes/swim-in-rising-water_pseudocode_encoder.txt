CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an empty SET  
        SET min_heap TO a list containing a tuple with the element at position zero zero of grid followed BY zero FOLLOWED BY zero  
        ADD the tuple of zero AND zero TO visited  
        SET directions TO a list containing tuples zero one AND one zero AND zero MINUS one AND MINUS one zero  
        
        WHILE min_heap IS NOT EMPTY  
            REMOVE the SMALLEST element FROM min_heap AND ASSIGN TO t AND i AND j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            
            FOR each di AND dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND the tuple of ni AND nj IS NOT IN visited  
                    ADD the tuple of ni AND nj TO visited  
                    ADD a tuple containing the GREATER VALUE BETWEEN t AND the element at position ni nj of grid FOLLOWED BY ni FOLLOWED BY nj TO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
