CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS an empty string THEN  
            RETURN an empty list  
        END IF  
        
        SET phone_map TO a mapping from character digit TWO TO string containing A B C letters with characters  
        SET phone_map for digit THREE TO string containing D E F letters with characters  
        SET phone_map for digit FOUR TO string containing G H I letters with characters  
        SET phone_map for digit FIVE TO string containing J K L letters with characters  
        SET phone_map for digit SIX TO string containing M N O letters with characters  
        SET phone_map for digit SEVEN TO string containing P Q R S letters with characters  
        SET phone_map for digit EIGHT TO string containing T U V letters with characters  
        SET phone_map for digit NINE TO string containing W X Y Z letters with characters  
        
        FUNCTION backtrack(combination, next_digits)  
            IF the LENGTH OF next_digits EQUALS zero THEN  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the string that maps digit at position zero of next_digits in phone_map  
                    CALL backtrack with combination concatenated with letter and substring from position one TO end of next_digits  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO an empty list  
        CALL backtrack with an empty string and digits  
        RETURN output  
    END FUNCTION  
END CLASS
