CLASS Solution  
    FUNCTION movesToStamp(stamp, target) RETURNS List of Integer  
        SET m TO the LENGTH OF stamp  
        SET n TO the LENGTH OF target  
        SET indeg TO a list of the value m repeated for the count of n MINUS m PLUS one  
        SET q TO an empty deque  
        SET g TO a list of empty lists repeated for the count of n  
        FOR index i FROM zero TO n MINUS m  
            FOR each pair of index j AND character c IN the enumeration of stamp  
                IF the character at position i PLUS j of target EQUALS c THEN  
                    DECREMENT the element at position i of indeg BY one  
                    IF the element at position i of indeg EQUALS zero THEN  
                        APPEND i TO q  
                    END IF  
                ELSE  
                    APPEND i TO the list at position i PLUS j of g  
                END IF  
            END FOR  
        END FOR  
        SET ans TO an empty list  
        SET vis TO a list of the value False repeated for the count of n  
        WHILE q is not empty  
            REMOVE and SET i TO the first element of q  
            APPEND i TO ans  
            FOR index j FROM zero TO m MINUS one  
                IF the element at position i PLUS j of vis IS False THEN  
                    SET the element at position i PLUS j of vis TO True  
                    FOR each element k IN the list at position i PLUS j of g  
                        DECREMENT the element at position k of indeg BY one  
                        IF the element at position k of indeg EQUALS zero THEN  
                            APPEND k TO q  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  
        IF all elements of vis ARE True THEN  
            RETURN the reverse of ans  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
