CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the val property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO a new ListNode initialized with element at position zero of values  
    SET p TO head  
    FOR each val IN elements starting from position one to the end of values  
        SET node TO a new ListNode initialized with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN comparison of the val property of p1 EQUALS the val property of p2 AND the result of is_same_list called with the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList WITH PARAMETERS head  
        IF head IS None OR the next property of head IS None OR the next property of the next property of head IS None  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO the next property of head  
        SET even_head TO even  
        WHILE even IS NOT None AND the next property of even IS NOT None  
            SET the next property of odd TO the next property of even  
            SET odd TO the next property of odd  
            SET the next property of even TO the next property of odd  
            SET even TO the next property of even  
        END WHILE  
        SET the next property of odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
