CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO two dimensional list of boolean false with the size of the length of s plus one rows and the length of p plus one columns  
        SET element at position zero zero of dp TO true  
        FOR j FROM one TO the length of p  
            IF character at position j minus one of p EQUALS star  
                SET element at position zero j of dp TO element at position zero j minus one of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF character at position j minus one of p EQUALS star  
                    SET element at position i j of dp TO element at position i minus one j of dp OR element at position i j minus one of dp  
                ELSE IF character at position j minus one of p EQUALS question mark OR character at position i minus one of s EQUALS character at position j minus one of p  
                    SET element at position i j of dp TO element at position i minus one j minus one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the length of s the length of p of dp  
    END FUNCTION  
END CLASS
