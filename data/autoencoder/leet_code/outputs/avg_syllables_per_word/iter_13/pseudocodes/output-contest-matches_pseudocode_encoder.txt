CLASS Solution  
    FUNCTION findContestMatch WITH PARAMETER n  
        SET teams TO list OF string REPRESENTATIONS of numbers FROM one TO n INCLUSIVE  
        WHILE n IS GREATER THAN one  
            FOR i FROM zero TO n DIVIDED BY two MINUS one  
                SET element at position i of teams TO concatenation of open parenthesis followed BY element at position i of teams FOLLOWED BY comma FOLLOWED BY element at position n MINUS one MINUS i of teams FOLLOWED BY close parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
