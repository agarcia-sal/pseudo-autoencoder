CLASS Solution  
    FUNCTION numberOfCombinations(num)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF num  
        SET memo TO an empty dictionary  
        
        FUNCTION dp(index, prev_len)  
            IF index EQUALS n THEN  
                RETURN one  
            END IF  
            
            IF the TUPLE of index AND prev_len EXISTS AS A KEY IN memo THEN  
                RETURN the VALUE ASSOCIATED WITH the TUPLE of index AND prev_len IN memo  
            END IF  
            
            SET count TO zero  
            FOR length FROM one TO n MINUS index INCLUSIVE  
                IF the CHARACTER at position index OF num EQUALS zero THEN  
                    BREAK out of the FOR loop  
                END IF  
                
                SET current_num TO the SUBSTRING FROM position index TO position index PLUS length MINUS one OF num  
                
                IF prev_len EQUALS zero OR current_num IS GREATER THAN OR EQUAL TO the SUBSTRING FROM position index MINUS prev_len TO position index MINUS one OF num THEN  
                    SET count TO the REMAINDER OF count PLUS dp OF index PLUS length AND length DIVIDED BY MODULO MOD  
                END IF  
            END FOR  
            
            ASSIGN the VALUE count TO the KEY TUPLE of index AND prev_len IN memo  
            RETURN count  
        END FUNCTION  
        
        RETURN dp OF zero AND zero  
    END FUNCTION  
END CLASS
