CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY two  
        SET total_sum TO the SUM OF nums  
        SET target TO total_sum DIVIDED BY two  
        SET min_diff TO positive infinity  
        
        SET left_half TO the SUBLIST of nums from position zero TO one LESS THAN n  
        SET right_half TO the SUBLIST of nums from position n TO the END of nums  
        
        SET left_sums TO a LIST of EMPTY MAPS with the COUNT OF n PLUS one  
        SET right_sums TO a LIST of EMPTY MAPS with the COUNT OF n PLUS one  
        
        FOR i FROM zero TO n INCLUSIVE  
            FOR each combo IN all combinations of left_half TAKEN i AT A TIME  
                SET subset_sum TO the SUM OF combo  
                IF subset_sum NOT IN left_sums AT position i  
                    ASSIGN zero TO left_sums AT position i AT key subset_sum  
                END IF  
                INCREMENT left_sums AT position i AT key subset_sum BY one  
            END FOR  
            
            FOR each combo IN all combinations of right_half TAKEN i AT A TIME  
                SET subset_sum TO the SUM OF combo  
                IF subset_sum NOT IN right_sums AT position i  
                    ASSIGN zero TO right_sums AT position i AT key subset_sum  
                END IF  
                INCREMENT right_sums AT position i AT key subset_sum BY one  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO n INCLUSIVE  
            SET left_values TO the SORTED LIST OF KEYS from left_sums AT position i  
            SET right_values TO the SORTED LIST OF KEYS from right_sums AT position n MINUS i  
            
            SET j TO zero  
            SET k TO the LENGTH OF right_values MINUS one  
            
            WHILE j LESS THAN the LENGTH OF left_values AND k GREATER THAN OR EQUAL TO zero  
                SET current_sum TO the element at position j of left_values PLUS the element at position k of right_values  
                SET candidate_diff TO total_sum MINUS two TIMES current_sum  
                IF candidate_diff IS LESS THAN zero  
                    SET candidate_diff TO the NEGATION OF candidate_diff  
                END IF  
                SET min_diff TO the SMALLER OF min_diff AND candidate_diff  
                
                IF current_sum LESS THAN target  
                    INCREMENT j BY one  
                ELSE IF current_sum GREATER THAN target  
                    DECREMENT k BY one  
                ELSE  
                    RETURN zero  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
