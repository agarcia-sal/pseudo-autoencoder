CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        SORT points BY the second element at each position of points IN ASCENDING ORDER  
        SET arrows TO zero  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF the first element at position zero of balloon IS GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO the second element at position one of balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
