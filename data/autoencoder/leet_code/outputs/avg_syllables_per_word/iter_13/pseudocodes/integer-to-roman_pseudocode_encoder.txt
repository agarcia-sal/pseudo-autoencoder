CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO a list of one thousand nine hundred, five hundred four hundred, one hundred ninety, fifty forty, ten nine, five four, one  
        SET syms TO a list of roman numeral symbol em, ce em, dee, ce dee, ce, ex ce, el el, ex el, ex, ex eye ex, vee, eye vee, eye  
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num IS GREATER THAN zero  
            SET how_many_times TO the integer division of num BY element at position i OF val  
            FOR each iteration FROM one TO how_many_times  
                APPEND element at position i OF syms TO roman_numeral  
                DECREMENT num BY element at position i OF val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
