CLASS Solution  
    FUNCTION minJumps WITH PARAMETER arr  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO a mapping from values TO empty lists  
        FOR each index i FROM zero TO n MINUS one  
            SET value TO element at position i of arr  
            APPEND i TO the list mapped by value in value_indices  
        END FOR  
        
        SET queue TO a double ended queue initialized WITH a pair of zero and zero representing current index and current steps  
        SET visited TO a set containing zero  
        
        WHILE queue is not empty  
            REMOVE the leftmost element from queue and ASSIGN TO current_index and steps  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO a list containing current_index PLUS one, current_index MINUS one, and all indices mapped by element at position current_index of arr in value_indices  
            
            FOR each next_index IN next_indices  
                IF next_index IS GREATER THAN OR EQUAL TO zero AND next_index IS LESS THAN n AND next_index IS NOT IN visited  
                    ADD next_index TO visited  
                    APPEND a pair of next_index and steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list mapped by element at position current_index of arr in value_indices TO empty list  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
