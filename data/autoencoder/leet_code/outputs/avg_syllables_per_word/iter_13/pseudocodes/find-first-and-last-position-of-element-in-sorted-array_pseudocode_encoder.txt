CLASS Solution  
    FUNCTION searchRange(nums AS List OF integers, target AS integer) AS List OF integers  
        FUNCTION find_left(nums AS List OF integers, target AS integer) AS integer  
            SET left TO zero  
            SET right TO the LENGTH OF nums MINUS one  
            WHILE left IS LESS THAN OR EQUAL TO right  
                SET mid TO the INTEGER PART OF the SUM of left PLUS right DIVIDED BY two  
                IF the element at position mid of nums IS LESS THAN target  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN left  
        END FUNCTION  

        FUNCTION find_right(nums AS List OF integers, target AS integer) AS integer  
            SET left TO zero  
            SET right TO the LENGTH OF nums MINUS one  
            WHILE left IS LESS THAN OR EQUAL TO right  
                SET mid TO the INTEGER PART OF the SUM of left PLUS right DIVIDED BY two  
                IF the element at position mid of nums IS LESS THAN OR EQUAL TO target  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN right  
        END FUNCTION  

        SET left_index TO find_left WITH parameters nums AND target  
        SET right_index TO find_right WITH parameters nums AND target  

        IF left_index IS LESS THAN OR EQUAL TO right_index AND zero IS LESS THAN OR EQUAL TO right_index AND right_index IS LESS THAN the LENGTH OF nums AND the element at position left_index of nums IS EQUALS TO target  
            RETURN a LIST CONTAINING left_index AND right_index  
        ELSE  
            RETURN a LIST CONTAINING MINUS one AND MINUS one  
        END IF  
    END FUNCTION  
END CLASS
