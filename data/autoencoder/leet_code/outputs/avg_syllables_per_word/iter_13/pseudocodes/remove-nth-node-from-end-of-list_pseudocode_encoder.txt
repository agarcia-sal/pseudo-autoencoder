FUNCTION list_node(values)  
    IF the LIST values IS EMPTY  
        RETURN nothing  
    END IF  
    SET head TO NEW ListNode with parameter val EQUALS element at position zero of values  
    SET p TO head  
    FOR each val IN values STARTING FROM position one TO the END  
        SET node TO NEW ListNode with parameter val EQUALS val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS nothing AND p2 IS nothing  
        RETURN True  
    END IF  
    IF p1 IS nothing OR p2 IS nothing  
        RETURN False  
    END IF  
    RETURN the condition where the val property of p1 EQUALS the val property of p2 AND the result of is_same_list called with the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self, val SET TO zero, next SET TO nothing  
        SET the val property of self TO val  
        SET the next property of self TO next  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION removeNthFromEnd WITH PARAMETERS self, head, n  
        SET dummy TO NEW ListNode with parameter val EQUALS zero  
        SET the next property of dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR index FROM zero TO n PLUS one  
            SET first TO the next property of first  
        END FOR  
        WHILE first IS NOT nothing  
            SET first TO the next property of first  
            SET second TO the next property of second  
        END WHILE  
        SET the next property of second TO the next property of the next property of second  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
