CLASS Solution  
    FUNCTION numDistinct(s, t)  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
        SET dp TO a two dimensional list with m PLUS one rows and n PLUS one columns filled with zero  
        FOR each i FROM zero TO m  
            SET element at position zero of the list at position i of dp TO one  
        END FOR  
        FOR each i FROM one TO m  
            FOR each j FROM one TO n  
                IF element at position i MINUS one of s EQUALS element at position j MINUS one of t  
                    SET element at position j of the list at position i of dp TO element at position j MINUS one of the list at position i MINUS one of dp PLUS element at position j of the list at position i MINUS one of dp  
                ELSE  
                    SET element at position j of the list at position i of dp TO element at position j of the list at position i MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of the list at position m of dp  
    END FUNCTION  
END CLASS
