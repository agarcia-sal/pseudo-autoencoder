CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty collection  
            RETURN zero  
        END IF  
        SORT intervals BY the element at position zero of each meeting  
        SET min_heap TO empty collection to function as minimum priority queue  
        FOR each meeting IN intervals  
            IF min_heap is not empty AND the element at position zero of meeting IS GREATER THAN OR EQUAL TO the element at position zero of min_heap  
                REMOVE the smallest element FROM min_heap  
            END IF  
            ADD the element at position one of meeting TO min_heap maintaining its ordering as minimum priority queue  
        END FOR  
        RETURN the LENGTH OF min_heap  
    END FUNCTION  
END CLASS
