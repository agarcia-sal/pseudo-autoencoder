CLASS Solution  
    FUNCTION findBlackPixel(picture AS List OF List OF string, target AS integer) RETURNS integer  
        IF picture IS EMPTY OR the LENGTH OF picture AT position zero IS ZERO  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF picture AT position zero  
        SET row_count TO a list of zeroes with LENGTH m  
        SET col_count TO a list of zeroes with LENGTH n  
        SET row_patterns TO a new mapping from string to integer with default value zero  
        
        FOR each r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of all elements of picture AT position r interpreted as a single string  
            INCREMENT row_patterns AT key pattern BY one  
            FOR each c FROM zero TO n MINUS one  
                IF element at position c of picture at position r EQUALS letter B  
                    INCREMENT element at position r of row_count BY one  
                    INCREMENT element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR each r FROM zero TO m MINUS one  
            FOR each c FROM zero TO n MINUS one  
                IF element at position c of picture at position r EQUALS letter B  
                    SET pattern TO the concatenation of all elements of picture AT position r interpreted as a single string  
                    IF element at position r of row_count EQUALS target AND element at position c of col_count EQUALS target AND row_patterns AT key pattern EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
