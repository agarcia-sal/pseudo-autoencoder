CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN a list consisting of zero  
        END IF  
        SET prev_gray TO the result of calling grayCode with parameter n MINUS one on this instance  
        SET mask TO one SHIFTED LEFT BY n MINUS one positions  
        SET reversed_prev_gray TO the reversed sequence of prev_gray  
        SET extended_part TO an empty list  
        FOR each number IN reversed_prev_gray  
            SET combined_number TO mask OR number  
            APPEND combined_number TO extended_part  
        END FOR  
        SET current_gray TO the concatenation of prev_gray FOLLOWED BY extended_part  
        RETURN current_gray  
    END FUNCTION  
END CLASS
