CLASS Solution  
    FUNCTION triangleNumber(nums)  
        CALL sort ON nums  
        SET count TO zero  
        SET n TO the LENGTH OF nums  
        
        FOR k FROM n MINUS one TO two IN STEPS OF MINUS one  
            SET i TO zero  
            SET j TO k MINUS one  
            WHILE i LESS THAN j  
                IF the element at position i of nums PLUS the element at position j of nums GREATER THAN the element at position k of nums  
                    INCREMENT count BY j MINUS i  
                    DECREMENT j BY one  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
