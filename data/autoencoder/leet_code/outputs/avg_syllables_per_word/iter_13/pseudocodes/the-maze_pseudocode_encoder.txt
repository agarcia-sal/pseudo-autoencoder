CLASS Solution  
    FUNCTION hasPath WITH PARAMETERS maze AS list of list of integers comma start AS list of integers comma destination AS list of integers  
        SET directions TO list of tuples representing zero and one comma one and zero comma zero and negative one comma negative one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET visited TO an empty set  
        
        FUNCTION dfs WITH PARAMETERS x comma y  
            IF tuple containing x and y IS IN visited THEN  
                RETURN false  
            END IF  
            IF list containing x and y EQUALS destination THEN  
                RETURN true  
            END IF  
            ADD the tuple containing x and y TO visited  
            
            FOR each tuple containing dx and dy IN directions  
                SET nx TO x  
                SET ny TO y  
                
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND element at position nx PLUS dx of maze AND element at position ny PLUS dy IS EQUAL TO zero  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                
                IF dfs WITH nx comma ny RETURNS true THEN  
                    RETURN true  
                END IF  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        RETURN dfs WITH element at position zero of start comma element at position one of start  
    END FUNCTION  
END CLASS
