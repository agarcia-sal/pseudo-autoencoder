CLASS Solution  
    FUNCTION minArea(image, x, y)  
        IF image IS empty OR the element at position zero of image IS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF image  
        SET n TO the LENGTH OF element at position zero of image  
        
        FUNCTION hasBlackPixelInRow(row)  
            FOR each pixel IN element at position row of image  
                IF pixel EQUALS the character one  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION hasBlackPixelInColumn(col)  
            FOR row FROM zero TO m MINUS one  
                IF element at position col of element at position row of image EQUALS the character one  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION findTop()  
            SET low TO zero  
            SET high TO x  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF low PLUS high BY two  
                IF hasBlackPixelInRow(mid)  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findBottom()  
            SET low TO x  
            SET high TO m MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF low PLUS high PLUS one BY two  
                IF hasBlackPixelInRow(mid)  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findLeft()  
            SET low TO zero  
            SET high TO y  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF low PLUS high BY two  
                IF hasBlackPixelInColumn(mid)  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findRight()  
            SET low TO y  
            SET high TO n MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF low PLUS high PLUS one BY two  
                IF hasBlackPixelInColumn(mid)  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        SET top TO the RESULT OF findTop  
        SET bottom TO the RESULT OF findBottom  
        SET left TO the RESULT OF findLeft  
        SET right TO the RESULT OF findRight  
        
        SET height TO bottom MINUS top PLUS one  
        SET width TO right MINUS left PLUS one  
        SET area TO height MULTIPLIED BY width  
        
        RETURN area  
    END FUNCTION  
END CLASS
