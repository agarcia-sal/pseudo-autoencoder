CLASS Solution  
    FUNCTION rob(nums)  
        IF nums IS empty  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN element at position zero of nums  
        END IF  
        SET dp TO a list of zeros with the LENGTH OF nums  
        SET element at position zero of dp TO element at position zero of nums  
        SET element at position one of dp TO the GREATER VALUE BETWEEN element at position zero of nums AND element at position one of nums  
        FOR index FROM two TO the LENGTH OF nums MINUS one  
            SET element at position index of dp TO the GREATER VALUE BETWEEN element at position index MINUS one of dp AND the SUM OF element at position index MINUS two of dp AND element at position index of nums  
        END FOR  
        RETURN element at the LAST POSITION OF dp  
    END FUNCTION  
END CLASS
