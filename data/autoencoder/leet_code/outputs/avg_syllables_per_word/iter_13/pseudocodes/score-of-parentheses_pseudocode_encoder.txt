CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO empty list  
        SET score TO zero  
        FOR each character IN s  
            IF character EQUALS opening parenthesis  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET popped_value TO the last element removed FROM stack  
                IF score EQUALS zero  
                    SET score TO popped_value PLUS one  
                ELSE  
                    SET score TO popped_value PLUS two MULTIPLIED BY score  
                END IF  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
