CLASS Solution  
    FUNCTION maximumCost(n, highways, k)  
        FROM collections IMPORT defaultdict  
        SET graph TO defaultdict WITH default value as empty list  
        FOR each city1, city2, toll IN highways  
            APPEND tuple consisting of city2 AND toll TO element at position city1 of graph  
            APPEND tuple consisting of city1 AND toll TO element at position city2 of graph  
        END FOR  
        
        FUNCTION dfs(city, visited, cost, highways_used)  
            NONLOCAL max_cost  
            
            IF highways_used EQUALS k THEN  
                SET max_cost TO THE GREATER OF max_cost AND cost  
                RETURN  
            END IF  
            
            FOR each neighbor, toll IN element at position city of graph  
                IF neighbor NOT IN visited THEN  
                    APPEND neighbor TO visited  
                    CALL dfs WITH arguments neighbor, visited, cost PLUS toll, highways_used PLUS one  
                    REMOVE neighbor FROM visited  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET max_cost TO negative one  
        FOR start_city FROM zero TO n MINUS one  
            SET visited TO set containing start_city  
            CALL dfs WITH arguments start_city, visited, zero, zero  
        END FOR  
        
        RETURN max_cost  
    END FUNCTION  
END CLASS
