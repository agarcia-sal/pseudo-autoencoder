CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO an empty list  
        FOR each asteroid IN asteroids  
            WHILE the stack is not empty AND the asteroid is less than zero AND the element at the last position of the stack is greater than zero  
                IF the element at the last position of the stack is less than the negation of the asteroid  
                    REMOVE the element at the last position FROM the stack  
                    CONTINUE to the next iteration of the WHILE loop  
                ELSE IF the element at the last position of the stack is equal to the negation of the asteroid  
                    REMOVE the element at the last position FROM the stack  
                END IF  
                BREAK the WHILE loop  
            END WHILE  
            IF the WHILE loop did not break  
                APPEND the asteroid TO the stack  
            END IF  
        END FOR  
        RETURN the stack  
    END FUNCTION  
END CLASS
