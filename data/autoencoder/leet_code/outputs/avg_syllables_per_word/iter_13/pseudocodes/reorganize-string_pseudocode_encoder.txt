CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO the frequency COUNTER of characters IN s  
        SET max_heap TO the list of pairs consisting of NEGATIVE frequency AND corresponding character FOR each character AND frequency IN count  
        CALL auxiliary FUNCTION heapify TO transform max_heap INTO a heap structure  
        
        SET previous_character TO none  
        SET previous_frequency TO zero  
        SET result TO an empty list  
        
        WHILE max_heap IS NOT empty OR previous_frequency IS LESS THAN zero  
            IF max_heap IS empty AND previous_frequency IS LESS THAN zero  
                RETURN empty string  
            END IF  
            
            REMOVE the element with the highest priority FROM max_heap ASSIGN TO current_frequency AND current_character  
            APPEND current_character TO result  
            
            INCREMENT current_frequency BY one  
            
            IF previous_frequency IS LESS THAN zero  
                INSERT the pair of previous_frequency AND previous_character INTO max_heap  
            END IF  
            
            SET previous_character TO current_character  
            SET previous_frequency TO current_frequency  
        END WHILE  
        
        RETURN the concatenation of all elements IN result AS a single string  
    END FUNCTION  
END CLASS
