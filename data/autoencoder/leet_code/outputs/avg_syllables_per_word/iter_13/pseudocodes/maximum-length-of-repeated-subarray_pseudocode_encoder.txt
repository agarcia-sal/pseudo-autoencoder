CLASS Solution  
    FUNCTION findLength(nums1 list OF integers, nums2 list OF integers) RETURNS integer  
        SET dp TO auxiliary FUNCTION initialize_two_dimensional_list WITH arguments the LENGTH OF nums1 PLUS one AND the LENGTH OF nums2 PLUS one  
        SET max_length TO zero  
        FOR i FROM one TO THE LENGTH OF nums1  
            FOR j FROM one TO THE LENGTH OF nums2  
                IF element at position i MINUS one of nums1 EQUALS element at position j MINUS one of nums2  
                    SET element at position i and j of dp TO element at position i MINUS one and j MINUS one of dp PLUS one  
                    IF element at position i and j of dp GREATER THAN max_length  
                        SET max_length TO element at position i and j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  

    FUNCTION initialize_two_dimensional_list(rows integer, columns integer) RETURNS list  
        SET new_list TO empty list  
        FOR each count FROM one TO rows  
            SET inner_list TO empty list  
            FOR each inner_count FROM one TO columns  
                APPEND zero TO inner_list  
            END FOR  
            APPEND inner_list TO new_list  
        END FOR  
        RETURN new_list  
    END FUNCTION  
END CLASS
