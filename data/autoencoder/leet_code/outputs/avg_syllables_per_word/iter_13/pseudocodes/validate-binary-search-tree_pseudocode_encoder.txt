CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero, left EQUAL TO none, right EQUAL TO none  
        SET the attribute val OF this INSTANCE TO val  
        SET the attribute left OF this INSTANCE TO left  
        SET the attribute right OF this INSTANCE TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new INSTANCE OF TreeNode INITIALIZED WITH the element at position zero OF values  
    SET i TO one  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element REMOVED FROM the LEFT side OF queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none  
            SET node.left TO a new INSTANCE OF TreeNode INITIALIZED WITH the element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none  
            SET node.right TO a new INSTANCE OF TreeNode INITIALIZED WITH the element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUAL TO the val ATTRIBUTE OF q  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS the left ATTRIBUTE OF p AND the left ATTRIBUTE OF q AND the RESULT OF is_same_tree WITH PARAMETERS the right ATTRIBUTE OF p AND the right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST WITH PARAMETERS self AND root AS Optional TreeNode  
        FUNCTION validate WITH PARAMETERS node, low EQUAL TO negative infinity, high EQUAL TO positive infinity  
            IF node IS none  
                RETURN True  
            END IF  
            IF the val ATTRIBUTE OF node IS LESS THAN OR EQUAL TO low OR the val ATTRIBUTE OF node IS GREATER THAN OR EQUAL TO high  
                RETURN False  
            END IF  
            RETURN the RESULT OF validate WITH PARAMETERS the left ATTRIBUTE OF node, low AND the val ATTRIBUTE OF node AND the RESULT OF validate WITH PARAMETERS the right ATTRIBUTE OF node, the val ATTRIBUTE OF node, high  
        END FUNCTION  
        RETURN the RESULT OF validate WITH PARAMETER root  
    END FUNCTION  
END CLASS
