CLASS Solution  
    FUNCTION smallestGoodBase(n)  
        SET n TO the integer conversion OF n  
        SET max_length TO the integer conversion OF the base two logarithm OF n PLUS one  
        FOR m FROM max_length DOWN TO two  
            SET k TO the integer conversion OF n raised TO the power of the reciprocal OF m MINUS one  
            IF k LESS THAN two  
                CONTINUE to the next iteration  
            END IF  
            SET num TO the integer division OF k raised TO the power of m MINUS one BY k MINUS one  
            IF num EQUALS n  
                RETURN the string conversion OF k  
            END IF  
        END FOR  
        RETURN the string conversion OF n MINUS one  
    END FUNCTION  
END CLASS
