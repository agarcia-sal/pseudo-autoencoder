CLASS Solution  
    FUNCTION appealSum WITH parameter s  
        SET last TO a list of negative one REPEATED twenty six times  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each position i AND character char IN enumeration of s  
            SET index TO the Unicode code point number of char MINUS the Unicode code point number of the lowercase letter a  
            INCREMENT current_appeal BY position i MINUS element at position index of last  
            INCREMENT total_appeal BY current_appeal  
            SET element at position index of last TO position i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
