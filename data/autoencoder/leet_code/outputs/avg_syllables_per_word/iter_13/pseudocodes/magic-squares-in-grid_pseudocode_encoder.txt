CLASS Solution  
    FUNCTION numMagicSquaresInside(grid)  
        FUNCTION is_magic(i, j)  
            IF element at position i and j of grid NOT EQUALS five THEN  
                RETURN false  
            END IF  
            SET numbers TO empty set  
            FOR x FROM i MINUS one TO i PLUS one  
                FOR y FROM j MINUS one TO j PLUS one  
                    IF element at position x and y of grid LESS THAN one OR element at position x and y of grid GREATER THAN nine OR element at position x and y of grid IS IN numbers THEN  
                        RETURN false  
                    END IF  
                    APPEND element at position x and y of grid TO numbers  
                END FOR  
            END FOR  
            FOR start FROM i MINUS one TO i PLUS one  
                IF sum of elements at positions start and j MINUS one AND start and j AND start and j PLUS one of grid NOT EQUALS fifteen OR sum of elements at positions i MINUS one and start AND i and start AND i PLUS one and start of grid NOT EQUALS fifteen THEN  
                    RETURN false  
                END IF  
            END FOR  
            IF sum of elements at positions i MINUS one and j MINUS one AND i and j AND i PLUS one and j PLUS one of grid NOT EQUALS fifteen OR sum of elements at positions i MINUS one and j PLUS one AND i and j AND i PLUS one and j MINUS one of grid NOT EQUALS fifteen THEN  
                RETURN false  
            END IF  
            RETURN true  
        END FUNCTION  
        SET count TO zero  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        FOR i FROM one TO rows MINUS one  
            FOR j FROM one TO cols MINUS one  
                IF is_magic(i, j) THEN  
                    INCREMENT count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
