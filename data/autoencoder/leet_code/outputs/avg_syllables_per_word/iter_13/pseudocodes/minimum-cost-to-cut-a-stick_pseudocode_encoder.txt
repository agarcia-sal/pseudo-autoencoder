CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO sorted list resulting from combining cuts WITH zero AND n  
        
        FUNCTION dp(left, right)  
            IF the value of right MINUS the value of left IS LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO infinite value  
            FOR each integer i FROM the value of left PLUS one TO the value of right MINUS one  
                SET cost TO the element at position right of cuts MINUS the element at position left of cuts PLUS the result of dp WITH parameters left AND i PLUS the result of dp WITH parameters i AND right  
                IF cost IS LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN the result of dp WITH parameters zero AND the value of the LENGTH OF cuts MINUS one  
    END FUNCTION  
END CLASS
