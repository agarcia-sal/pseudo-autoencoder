CLASS Solution  
    FUNCTION validTree WITH PARAMETERS n AND edges  
        IF the LENGTH OF edges NOT EQUALS the SUBTRACTION OF one FROM n  
            RETURN false  
        END IF  
        SET adj_list TO an EMPTY MAPPING OF each INTEGER FROM zero TO THE SUBTRACTION OF one FROM n MAPPED TO an EMPTY LIST  
        FOR each pair node_a AND node_b IN edges  
            APPEND node_b TO the LIST at key node_a IN adj_list  
            APPEND node_a TO the LIST at key node_b IN adj_list  
        END FOR  
        SET visited TO an EMPTY SET  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS AN ELEMENT OF visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the LIST at key node IN adj_list  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        CALL dfs WITH zero  
        RETURN the EXPRESSION THAT the LENGTH OF visited EQUALS n  
    END FUNCTION  
END CLASS
