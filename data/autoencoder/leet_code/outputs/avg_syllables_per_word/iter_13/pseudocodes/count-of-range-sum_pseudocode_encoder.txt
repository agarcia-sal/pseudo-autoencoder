CLASS Solution  
    FUNCTION countRangeSum(nums AS List OF integers, lower AS integer, upper AS integer) RETURNS integer  
        SET prefix_sums TO list containing the single element zero  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            
            SET left TO current_sum MINUS upper  
            SET right TO current_sum MINUS lower  
            
            SET count TO count PLUS the result of the difference BETWEEN the number of elements in prefix_sums less than or equal to right AND the number of elements in prefix_sums less than left
            
            INSERT current_sum INTO prefix_sums maintaining sorted order  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
