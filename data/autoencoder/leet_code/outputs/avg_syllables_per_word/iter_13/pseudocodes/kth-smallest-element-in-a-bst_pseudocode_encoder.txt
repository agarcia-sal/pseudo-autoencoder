CLASS TreeNode  
    FUNCTION __init__ with parameters val default zero left default none right default none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values list  
    IF the LENGTH OF values list EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode object initialized with element at position zero of values list  
    SET i TO one  
    SET queue TO a new deque object  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values list AND element at position i of values list IS NOT none  
            SET node dot left TO a new TreeNode object initialized with element at position i of values list  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values list AND element at position i of values list IS NOT none  
            SET node dot right TO a new TreeNode object initialized with element at position i of values list  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree with parameters p q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN is_same_tree with arguments p dot left q dot left AND is_same_tree with arguments p dot right q dot right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION kthSmallest with parameters self root k  
        SET stack TO an empty list  
        SET current TO root  
        SET count TO zero  
        WHILE stack is not empty OR current IS NOT none  
            WHILE current IS NOT none  
                APPEND current TO stack  
                SET current TO current dot left  
            END WHILE  
            SET current TO the last element removed from stack  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN current dot val  
            END IF  
            SET current TO current dot right  
        END WHILE  
    END FUNCTION  
END CLASS
