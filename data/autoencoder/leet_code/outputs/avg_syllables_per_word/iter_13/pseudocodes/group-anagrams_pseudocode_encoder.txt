CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO a default dictionary mapping keys to empty lists  
        FOR each string s IN strs  
            SET sorted_string TO the concatenation of characters obtained by arranging the characters of s IN ascending order  
            APPEND s TO the list at key sorted_string in anagrams  
        END FOR  
        SET result TO the list of all values contained in anagrams  
        RETURN result  
    END FUNCTION  
END CLASS
