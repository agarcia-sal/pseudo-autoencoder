CLASS Solution  
    FUNCTION maximumScore(scores, edges)  
        SET graph TO a new default dictionary with list as default value  
        FOR each pair u and v IN edges  
            APPEND the pair of the element at position v of scores AND v TO the element at position u of graph  
            APPEND the pair of the element at position u of scores AND u TO the element at position v of graph  
        END FOR  
        FOR each node IN graph  
            SET the element at position node of graph TO the list of the three largest elements from the element at position node of graph based on score value  
        END FOR  
        SET max_score TO negative one  
        FOR each pair u and v IN edges  
            FOR each pair score1 and x IN the element at position u of graph  
                FOR each pair score2 and y IN the element at position v of graph  
                    IF the number of distinct elements in the set consisting of x AND y AND u AND v EQUALS four  
                        SET max_score TO the maximum of max_score AND the sum of score1 PLUS score2 PLUS the element at position u of scores PLUS the element at position v of scores  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN max_score  
    END FUNCTION  
END CLASS
