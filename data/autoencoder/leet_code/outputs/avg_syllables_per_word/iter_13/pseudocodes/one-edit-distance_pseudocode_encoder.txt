CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF isOneEditDistance WITH t AND s AS PARAMETERS  
        END IF  
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        IF diff GREATER THAN one  
            RETURN false  
        END IF  
        IF diff EQUALS zero  
            RETURN the RESULT OF the SUM OF each comparison WHETHER character from s at position index NOT EQUALS character from t at position index FOR each pair OF characters at corresponding positions in s and t IS EQUAL TO one  
        ELSE  
            FOR index FROM zero TO the LENGTH OF s MINUS one  
                IF character from s at position index NOT EQUALS character from t at position index  
                    RETURN WHETHER the substring from position index TO the END of s IS EQUAL TO the substring from position index PLUS one TO the END of t  
                END IF  
            END FOR  
            RETURN true  
        END IF  
    END FUNCTION  
END CLASS
