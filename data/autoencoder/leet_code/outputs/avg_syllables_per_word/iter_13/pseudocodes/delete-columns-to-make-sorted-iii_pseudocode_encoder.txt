CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO LENGTH OF strs  
        SET n TO LENGTH OF element at position zero of strs  
        SET dp TO list of length n where each element IS one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF for all k FROM zero TO m MINUS one the element at position j of element at position k of strs IS LESS THAN OR EQUAL TO the element at position i of element at position k of strs  
                    SET dp at position i TO the MAXIMUM OF dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        SET longest_non_decreasing_subsequence_length TO the MAXIMUM value in dp  
        RETURN n MINUS longest_non_decreasing_subsequence_length  
    END FUNCTION  
END CLASS
