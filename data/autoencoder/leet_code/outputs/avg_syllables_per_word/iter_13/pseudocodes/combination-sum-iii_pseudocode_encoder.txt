CLASS Solution  
    FUNCTION combinationSum3 WITH PARAMETERS k AND n  
        FUNCTION backtrack WITH PARAMETERS start AND path AND target AND k  
            IF target EQUALS zero AND k EQUALS zero  
                APPEND A COPY OF path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero OR k LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO nine  
                APPEND i TO path  
                CALL backtrack WITH i PLUS one AND path AND target MINUS i AND k MINUS one  
                REMOVE last element FROM path  
            END FOR  
        END FUNCTION  
  
        SET result TO empty list  
        CALL backtrack WITH one AND empty list AND n AND k  
        RETURN result  
    END FUNCTION  
END CLASS
