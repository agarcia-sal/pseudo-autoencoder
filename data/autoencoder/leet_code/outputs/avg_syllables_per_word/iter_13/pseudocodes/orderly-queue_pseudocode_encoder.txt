CLASS Solution  
    FUNCTION orderlyQueue(s AS string, k AS integer) RETURNS string  
        IF k GREATER THAN one  
            RETURN string GENERATED BY JOINING the SORTED characters of s  
        ELSE  
            SET n TO the LENGTH OF s  
            SET smallest TO s  
            FOR index FROM one TO n MINUS one  
                SET rotated TO the SUBSTRING from position index TO the END of s CONCATENATED WITH the SUBSTRING from position zero TO position index MINUS one of s  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
