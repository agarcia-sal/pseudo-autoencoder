CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            RETURN the SUM OF the COUNT of elements IN nums WHERE the element is LESS THAN OR EQUAL TO x GREATER THAN x  
        END FUNCTION  
        RETURN the RESULT OF performing a BINARY SEARCH FOR the first position FROM zero TO the LENGTH OF nums WHERE f RETURNS true  
    END FUNCTION  
END CLASS
