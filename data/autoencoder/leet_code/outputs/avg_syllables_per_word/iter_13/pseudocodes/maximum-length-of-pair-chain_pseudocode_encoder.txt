CLASS Solution  
    FUNCTION findLongestChain(pairs)  
        CALL sort pairs BY the element at position two of each pair in ascending order  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each pair IN pairs  
            DESTRUCTURE pair INTO start and end  
            IF start IS GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
