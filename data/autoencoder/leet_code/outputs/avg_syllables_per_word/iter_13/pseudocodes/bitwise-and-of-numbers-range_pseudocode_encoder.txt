CLASS Solution  
    FUNCTION rangeBitwiseAnd(left AS integer, right AS integer) RETURNS integer  
        SET shift TO zero  
        WHILE left IS LESS THAN right  
            SET left TO left DIVIDED BY two TAKING THE INTEGER PART  
            SET right TO right DIVIDED BY two TAKING THE INTEGER PART  
            INCREMENT shift BY one  
        END WHILE  
        RETURN left MULTIPLIED BY two RAISED TO THE POWER OF shift  
    END FUNCTION  
END CLASS
