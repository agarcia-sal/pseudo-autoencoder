CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR index i FROM zero TO N MINUS one  
            FOR index j FROM zero TO N MINUS one  
                IF element at position i of board AND element at position j of board EXCLUSIVE OR element at position i of board AND element at position zero of board EXCLUSIVE OR element at position zero of board AND element at position j of board EXCLUSIVE OR element at position zero of board AND element at position zero of board IS TRUE  
                    RETURN negative one  
                END IF  
            END FOR  
            SET rowSum TO rowSum PLUS element at position i of board AND element at position zero of board  
            SET colSum TO colSum PLUS element at position zero of board AND element at position i of board  
            SET rowSwap TO rowSwap PLUS one IF element at position i of board AND element at position zero of board EQUALS remainder of i DIVIDED BY two ELSE rowSwap PLUS zero  
            SET colSwap TO colSwap PLUS one IF element at position zero of board AND element at position i of board EQUALS remainder of i DIVIDED BY two ELSE colSwap PLUS zero  
        END FOR  
        IF integer DIVISION of N BY two IS GREATER THAN rowSum OR rowSum IS GREATER THAN integer DIVISION of the SUM of N AND one BY two  
            RETURN negative one  
        END IF  
        IF integer DIVISION of N BY two IS GREATER THAN colSum OR colSum IS GREATER THAN integer DIVISION of the SUM of N AND one BY two  
            RETURN negative one  
        END IF  
        IF remainder of N DIVIDED BY two IS EQUAL TO one  
            IF remainder of rowSwap DIVIDED BY two IS EQUAL TO one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF remainder of colSwap DIVIDED BY two IS EQUAL TO one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the MINIMUM VALUE BETWEEN rowSwap AND N MINUS rowSwap  
            SET colSwap TO the MINIMUM VALUE BETWEEN colSwap AND N MINUS colSwap  
        END IF  
        RETURN integer DIVISION of the SUM of rowSwap AND colSwap BY two  
    END FUNCTION  
END CLASS
