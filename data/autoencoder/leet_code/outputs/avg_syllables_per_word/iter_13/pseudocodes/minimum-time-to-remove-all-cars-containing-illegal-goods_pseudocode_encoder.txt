CLASS Solution  
    FUNCTION minimumTime WITH PARAMETERS s  
        SET n TO the LENGTH OF s  
        SET left_costs TO a list of zeroes WITH length n  
        SET right_costs TO a list of zeroes WITH length n  

        IF the element at position zero of s EQUALS the character one  
            SET the element at position zero of left_costs TO one  
        ELSE  
            SET the element at position zero of left_costs TO zero  
        END IF  

        FOR index i FROM one TO n MINUS one  
            IF the element at position i of s EQUALS the character zero  
                SET the element at position i of left_costs TO the element at position i MINUS one of left_costs  
            ELSE  
                SET the element at position i of left_costs TO the MINIMUM BETWEEN i PLUS one AND the element at position i MINUS one of left_costs PLUS two  
            END IF  
        END FOR  

        IF the element at position n MINUS one of s EQUALS the character one  
            SET the element at position n MINUS one of right_costs TO one  
        ELSE  
            SET the element at position n MINUS one of right_costs TO zero  
        END IF  

        FOR index i FROM n MINUS two DOWN TO zero STEP minus one  
            IF the element at position i of s EQUALS the character zero  
                SET the element at position i of right_costs TO the element at position i PLUS one of right_costs  
            ELSE  
                SET the element at position i of right_costs TO the MINIMUM BETWEEN n MINUS i AND the element at position i PLUS one of right_costs PLUS two  
            END IF  
        END FOR  

        SET min_time TO the MINIMUM BETWEEN the element at position n MINUS one of left_costs AND the element at position zero of right_costs  

        FOR index i FROM zero TO n MINUS two  
            SET min_time TO the MINIMUM BETWEEN min_time AND the element at position i of left_costs PLUS the element at position i PLUS one of right_costs  
        END FOR  

        RETURN min_time  
    END FUNCTION  
END CLASS
