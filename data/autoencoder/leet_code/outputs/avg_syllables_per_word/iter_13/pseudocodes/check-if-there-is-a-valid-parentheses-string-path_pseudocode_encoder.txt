CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        IF the TOTAL OF m PLUS n MINUS one MODULO two NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        CALL auxiliary FUNCTION initialize_dp WITH parameters m PLUS one AND n PLUS one AND m PLUS n PLUS one RETURNING dp  
        SET the element at position zero ONE ONE zero of dp TO True  
        FOR each i FROM zero TO m MINUS one  
            FOR each j FROM zero TO n MINUS one  
                FOR each k FROM zero TO m PLUS n DO NOT EXCEED  
                    IF the element at position i j PLUS one k of dp OR the element at position i PLUS one j k of dp IS True THEN  
                        IF the element at position i j of grid EQUALS opening parenthesis AND k PLUS one LESS THAN m PLUS n PLUS one THEN  
                            SET the element at position i PLUS one j PLUS one k PLUS one of dp TO True  
                        ELSE IF the element at position i j of grid EQUALS closing parenthesis AND k MINUS one GREATER THAN OR EQUAL TO zero THEN  
                            SET the element at position i PLUS one j PLUS one k MINUS one of dp TO True  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN the element at position m n zero of dp  
    END FUNCTION  

    FUNCTION initialize_dp(rows, columns, depth)  
        RETURN a three dimensional structure of rows by columns by depth filled with False  
    END FUNCTION  
END CLASS
