CLASS Solution  
    FUNCTION wordBreak WITH PARAMETERS s AND wordDict  
        SET wordSet TO the SET of elements IN wordDict  
        SET dp TO a LIST OF boolean FALSE values WITH the LENGTH equal TO the LENGTH OF s PLUS one  
        SET element at position zero OF dp TO boolean TRUE  
        FOR index i FROM one TO the LENGTH OF s PLUS one  
            FOR index j FROM zero TO index i MINUS one  
                IF element at position j OF dp EQUALS boolean TRUE AND substring from position j TO index i MINUS one OF s IS AN ELEMENT OF wordSet  
                    SET element at position i OF dp TO boolean TRUE  
                    BREAK the inner loop  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the LENGTH OF s OF dp  
    END FUNCTION  
END CLASS
