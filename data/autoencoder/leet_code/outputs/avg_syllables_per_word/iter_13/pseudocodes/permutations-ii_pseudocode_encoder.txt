CLASS Solution  
    FUNCTION permuteUnique(nums)  
        FUNCTION backtrack(path, used, res)  
            IF the LENGTH OF path EQUALS the LENGTH OF nums  
                APPEND a copy of path TO res  
                RETURN  
            END IF  
            FOR each index FROM zero TO the LENGTH OF nums MINUS one  
                IF the element at position index of used IS TRUE OR  
                   index GREATER THAN zero AND the element at position index of nums EQUALS the element at position index MINUS one of nums AND the element at position index MINUS one of used IS FALSE  
                    CONTINUE to the next iteration  
                END IF  
                SET the element at position index of used TO TRUE  
                APPEND the element at position index of nums TO path  
                CALL backtrack WITH path, used, res  
                REMOVE the last element FROM path  
                SET the element at position index of used TO FALSE  
            END FOR  
        END FUNCTION  
        SORT nums  
        SET res TO an empty list  
        SET used TO a list of FALSE repeated the LENGTH OF nums times  
        CALL backtrack WITH an empty list, used, res  
        RETURN res  
    END FUNCTION  
END CLASS
