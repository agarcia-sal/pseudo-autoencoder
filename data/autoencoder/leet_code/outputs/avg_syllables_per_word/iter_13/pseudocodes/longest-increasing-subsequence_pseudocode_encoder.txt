CLASS Solution  
    FUNCTION lengthOfLIS WITH PARAMETER nums  
        IF nums EQUALS an empty list  
            RETURN zero  
        END IF  
        SET dp TO an empty list  
        FOR each num IN nums  
            SET idx TO the result of performing binary search to find the leftmost position where num can be inserted in dp to maintain sorted order  
            IF idx EQUALS the length of dp  
                APPEND num TO dp  
            ELSE  
                SET the element at position idx of dp TO num  
            END IF  
        END FOR  
        RETURN the length of dp  
    END FUNCTION  
END CLASS
