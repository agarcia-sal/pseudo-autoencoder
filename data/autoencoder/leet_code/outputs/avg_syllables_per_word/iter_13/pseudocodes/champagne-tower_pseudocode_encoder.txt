CLASS Solution  
    FUNCTION champagneTower(poured integer, query_row integer, query_glass integer) RETURNS float  
        SET current_row TO list containing poured  
        FOR each row FROM one TO query_row  
            SET next_row TO list of zeros with LENGTH EQUAL TO row PLUS one  
            FOR each index FROM zero TO LENGTH OF current_row MINUS one  
                IF element at position index of current_row GREATER THAN one  
                    SET excess TO element at position index of current_row MINUS one DIVIDED BY two  
                    SET element at position index of next_row TO element at position index of next_row PLUS excess  
                    SET element at position index PLUS one of next_row TO element at position index PLUS one of next_row PLUS excess  
                END IF  
            END FOR  
            SET current_row TO next_row  
        END FOR  
        IF element at position query_glass of current_row LESS THAN one  
            RETURN element at position query_glass of current_row  
        ELSE  
            RETURN one  
        END IF  
    END FUNCTION  
END CLASS
