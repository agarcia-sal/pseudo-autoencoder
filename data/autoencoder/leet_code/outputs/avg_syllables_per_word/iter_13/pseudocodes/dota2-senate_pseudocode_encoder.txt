CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        CREATE queue radiant TO HOLD integers  
        CREATE queue dire TO HOLD integers  
        FOR each position index AND senator s IN senate  
            IF senator s EQUALS the character representing Radiant  
                APPEND position index TO radiant  
            ELSE  
                APPEND position index TO dire  
            END IF  
        END FOR  
        WHILE the queue radiant is not empty AND the queue dire is not empty  
            SET r_index TO the element REMOVED FROM the FRONT OF queue radiant  
            SET d_index TO the element REMOVED FROM the FRONT OF queue dire  
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the LENGTH OF senate TO queue radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO queue dire  
            END IF  
        END WHILE  
        IF the queue radiant is not empty  
            RETURN the string indicating Radiant  
        ELSE  
            RETURN the string indicating Dire  
        END IF  
    END FUNCTION  
END CLASS
