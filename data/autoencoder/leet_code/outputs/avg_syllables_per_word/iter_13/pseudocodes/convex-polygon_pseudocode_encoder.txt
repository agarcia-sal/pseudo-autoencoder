CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            RETURN the value obtained by the quantity of the element at position zero of a MINUS the element at position zero of o MULTIPLIED BY the quantity of the element at position one of b MINUS the element at position one of o MINUS the quantity of the element at position one of a MINUS the element at position one of o MULTIPLIED BY the quantity of the element at position zero of b MINUS the element at position zero of o  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN true  
        END IF  
        
        SET orientation TO zero  
        
        FOR index i FROM zero TO n MINUS one  
            SET o TO element at position i of points  
            SET a TO element at position the remainder of the quantity of i PLUS one DIVIDED BY n of points  
            SET b TO element at position the remainder of the quantity of i PLUS two DIVIDED BY n of points  
            SET cross TO cross_product with o, a, b  
            
            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS negative one AND cross GREATER THAN zero  
                    RETURN false  
                END IF  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
