CLASS Solution  
    FUNCTION myPow(x AS float, n AS integer) RETURNS float  
        FUNCTION qpow(a AS float, n AS integer) RETURNS float  
            SET ans TO one  
            WHILE n IS NOT EQUAL TO zero  
                IF the least significant bit of n IS equal to one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two ignoring remainder  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n IS GREATER THAN OR EQUAL TO zero  
            RETURN the RESULT OF qpow WITH PARAMETERS x AND n  
        ELSE  
            RETURN one DIVIDED BY the RESULT OF qpow WITH PARAMETERS x AND the NEGATION OF n  
        END IF  
    END FUNCTION  
END CLASS
