CLASS Solution  
    FUNCTION maxDistance(arrays)  
        SET min_val TO the element at position zero of the element at position zero of arrays  
        SET max_val TO the element at the last position of the element at position zero of arrays  
        SET max_distance TO zero  
        FOR index i FROM one TO the LENGTH OF arrays MINUS one  
            SET current_array TO the element at position i of arrays  
            SET current_min TO the element at position zero of current_array  
            SET current_max TO the element at the last position of current_array  
            SET max_distance TO the maximum OF max_distance AND the absolute VALUE OF current_max MINUS min_val AND the absolute VALUE OF max_val MINUS current_min  
            SET min_val TO the minimum OF min_val AND current_min  
            SET max_val TO the maximum OF max_val AND current_max  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
