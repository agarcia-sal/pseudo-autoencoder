CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO a set containing tuples converted from each element of mines  
        SET dp TO a two dimensional grid of size n by n where each element is a list of four zeros representing directions right down left up  
        SET directions TO the list of pairs zero one one zero zero minus one minus one zero representing right down left up directions  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair i j is not contained in mines_set THEN  
                    IF j IS GREATER THAN zero THEN  
                        SET the element at position zero of the list at position j of the list at position i of dp TO the value of the element at position zero of the list at position j minus one of the list at position i of dp PLUS one  
                    END IF  
                    IF i IS GREATER THAN zero THEN  
                        SET the element at position one of the list at position j of the list at position i of dp TO the value of the element at position one of the list at position j of the list at position i minus one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR i FROM n MINUS one TO zero STEP minus one  
            FOR j FROM n MINUS one TO zero STEP minus one  
                IF the pair i j is not contained in mines_set THEN  
                    IF j IS LESS THAN n MINUS one THEN  
                        SET the element at position two of the list at position j of the list at position i of dp TO the value of the element at position two of the list at position j plus one of the list at position i of dp PLUS one  
                    END IF  
                    IF i IS LESS THAN n MINUS one THEN  
                        SET the element at position three of the list at position j of the list at position i of dp TO the value of the element at position three of the list at position j of the list at position i plus one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum of the list at position j of the list at position i of dp  
                    SET result TO the maximum of result AND plus_sign_order  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
