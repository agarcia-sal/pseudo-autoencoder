CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero left EQUAL TO none right EQUAL TO none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode initialized WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left child of node TO a new instance of TreeNode initialized WITH the element at position i of values  
            APPEND the left child of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right child of node TO a new instance of TreeNode initialized WITH the element at position i of values  
            APPEND the right child of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the value of p IS NOT EQUAL TO the value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS the left child of p AND the left child of q AND the result of is_same_tree WITH PARAMETERS the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues WITH PARAMETERS self root  
        IF root IS none  
            RETURN an empty list  
        END IF  
        SET queue TO a double-ended queue initialized WITH root as its only element  
        SET largest_values TO an empty list  
        WHILE queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR each number FROM zero TO level_size MINUS one  
                SET node TO the element removed from the front of queue  
                IF max_value IS LESS THAN the value of node  
                    SET max_value TO the value of node  
                END IF  
                IF the left child of node IS NOT none  
                    APPEND the left child of node TO queue  
                END IF  
                IF the right child of node IS NOT none  
                    APPEND the right child of node TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
