CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO an empty set  
        FOR each position FROM zero TO the LENGTH OF fronts MINUS one  
            SET x TO element at position position of fronts  
            SET y TO element at position position of backs  
            IF x EQUALS y  
                ADD x TO the set same  
            END IF  
        END FOR  
        SET all_numbers TO a set containing all elements from fronts AND all elements from backs  
        SET good_numbers TO the set difference of all_numbers MINUS same  
        IF the set good_numbers is not empty  
            SET minimum_good_number TO the smallest element in good_numbers  
            RETURN minimum_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
