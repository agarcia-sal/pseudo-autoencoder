CLASS Solution  
    FUNCTION reachableNodes(edges OF LIST OF LIST OF INTEGER, maxMoves OF INTEGER, n OF INTEGER) RETURNS INTEGER  
        SET graph TO an empty mapping from integer to empty list  
        FOR index FROM zero TO n MINUS one  
            SET element at position index of graph TO an empty list  
        END FOR  
        
        FOR each element of edges AS u COMMA v COMMA cnt  
            APPEND a tuple of v and cnt TO element at position u of graph  
            APPEND a tuple of u and cnt TO element at position v of graph  
        END FOR  
        
        SET distances TO a list of length n WHERE each element IS positive infinity  
        SET element at position zero of distances TO zero  
        SET priority_queue TO a list containing a tuple of zero and zero  
        
        WHILE priority_queue IS NOT empty  
            REMOVE and RETURN the tuple with the smallest first element FROM priority_queue AS dist AND node  
            IF dist IS GREATER THAN element at position node of distances  
                CONTINUE to the next iteration  
            END IF  
            FOR each tuple of neighbor and cnt IN element at position node of graph  
                SET new_dist TO dist PLUS cnt PLUS one  
                IF new_dist IS LESS THAN element at position neighbor of distances  
                    SET element at position neighbor of distances TO new_dist  
                    ADD the tuple of new_dist and neighbor TO priority_queue maintaining heap order  
                END IF  
            END FOR  
        END WHILE  
        
        SET reachable_nodes TO zero  
        
        FOR each element d IN distances  
            IF d IS LESS THAN OR EQUAL TO maxMoves  
                INCREMENT reachable_nodes BY one  
            END IF  
        END FOR  
        
        SET used_edges TO an empty mapping  
        
        FOR each element of edges AS u COMMA v COMMA cnt  
            SET moves_u TO the maximum of zero AND maxMoves MINUS element at position u of distances  
            SET moves_v TO the maximum of zero AND maxMoves MINUS element at position v of distances  
            SET reachable_new_nodes TO the minimum of cnt AND moves_u PLUS moves_v  
            INCREMENT reachable_nodes BY reachable_new_nodes  
            SET element with key of the tuple u and v in used_edges TO reachable_new_nodes  
        END FOR  
        
        RETURN reachable_nodes  
    END FUNCTION  
END CLASS
