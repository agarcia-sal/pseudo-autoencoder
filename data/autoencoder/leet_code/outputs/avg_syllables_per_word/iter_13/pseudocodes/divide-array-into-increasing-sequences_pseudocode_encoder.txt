CLASS Solution  
    FUNCTION canDivideIntoSubsequences(nums LIST OF INTEGER, k INTEGER) RETURNS BOOLEAN  
        SET freq TO a mapping of elements to their frequency counts within nums  
        SET max_freq TO the greatest frequency value within the values of freq  
        SET length_divided_by_k TO the LENGTH OF nums DIVIDED BY k  
        IF length_divided_by_k IS GREATER THAN OR EQUAL TO max_freq  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
