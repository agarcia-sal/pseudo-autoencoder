CLASS Solution  
    FUNCTION numOfWays WITH PARAMETER n  
        SET MODULO_VALUE TO one billion PLUS seven  
        SET count_a TO six  
        SET count_b TO six  
        FOR each integer FROM two TO n  
            SET new_count_a TO the sum of three MULTIPLIED BY count_a AND two MULTIPLIED BY count_b  
            SET new_count_a TO new_count_a MODULO MODULO_VALUE  
            SET new_count_b TO the sum of two MULTIPLIED BY count_a AND two MULTIPLIED BY count_b  
            SET new_count_b TO new_count_b MODULO MODULO_VALUE  
            SET count_a TO new_count_a  
            SET count_b TO new_count_b  
        END FOR  
        SET result TO the sum of count_a AND count_b  
        SET result TO result MODULO MODULO_VALUE  
        RETURN result  
    END FUNCTION  
END CLASS
