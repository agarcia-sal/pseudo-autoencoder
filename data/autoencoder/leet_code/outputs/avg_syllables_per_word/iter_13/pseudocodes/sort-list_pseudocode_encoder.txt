CLASS ListNode  
    FUNCTION __init__(val EQUALS zero COMMA next EQUALS none)  
        SET the val PROPERTY OF self TO val  
        SET the next PROPERTY OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode WITH THE ELEMENT at position zero OF values  
    SET p TO head  
    FOR each val IN values FROM position one TO THE END  
        SET node TO A NEW ListNode WITH val  
        SET the next PROPERTY OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 COMMA p2)  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN THE RESULT OF THE LOGICAL AND BETWEEN THE CONDITION THAT the val PROPERTY OF p1 EQUALS THE val PROPERTY OF p2 AND THE RECURSIVE CALL TO is_same_list WITH THE next PROPERTY OF p1 AND THE next PROPERTY OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList(head)  
        IF head IS none OR the next PROPERTY OF head IS none  
            RETURN head  
        END IF  
        
        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO the next PROPERTY OF head  
            WHILE fast IS NOT none AND the next PROPERTY OF fast IS NOT none  
                SET slow TO the next PROPERTY OF slow  
                SET fast TO the next PROPERTY OF the next PROPERTY OF fast  
            END WHILE  
            SET mid TO the next PROPERTY OF slow  
            SET the next PROPERTY OF slow TO none  
            RETURN head COMMA mid  
        END FUNCTION  
        
        FUNCTION merge(l1 COMMA l2)  
            SET dummy TO A NEW ListNode  
            SET current TO dummy  
            WHILE l1 IS NOT none AND l2 IS NOT none  
                IF the val PROPERTY OF l1 IS LESS THAN the val PROPERTY OF l2  
                    SET the next PROPERTY OF current TO l1  
                    SET l1 TO the next PROPERTY OF l1  
                ELSE  
                    SET the next PROPERTY OF current TO l2  
                    SET l2 TO the next PROPERTY OF l2  
                END IF  
                SET current TO the next PROPERTY OF current  
            END WHILE  
            IF l1 IS NOT none  
                SET the next PROPERTY OF current TO l1  
            ELSE  
                SET the next PROPERTY OF current TO l2  
            END IF  
            RETURN the next PROPERTY OF dummy  
        END FUNCTION  
        
        SET left AND right TO THE RESULT OF split WITH head  
        SET left TO THE RESULT OF THE RECURSIVE CALL TO sortList WITH left  
        SET right TO THE RESULT OF THE RECURSIVE CALL TO sortList WITH right  
        
        RETURN THE RESULT OF merge WITH left AND right  
    END FUNCTION  
END CLASS
