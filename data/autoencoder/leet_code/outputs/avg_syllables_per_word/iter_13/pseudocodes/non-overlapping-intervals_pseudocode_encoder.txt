CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals EQUALS empty  
            RETURN zero  
        END IF  
        SORT intervals BY the end element at position one of each interval  
        SET end TO the end element at position one of the interval at position zero of intervals  
        SET count TO zero  
        FOR index FROM one TO the LENGTH OF intervals MINUS one  
            IF the start element at position zero of the interval at position index of intervals GREATER THAN OR EQUAL TO end  
                SET end TO the end element at position one of the interval at position index of intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
