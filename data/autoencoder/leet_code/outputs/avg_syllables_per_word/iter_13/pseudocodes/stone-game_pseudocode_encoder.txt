CLASS Solution  
    FUNCTION stoneGame WITH parameter piles  
        FUNCTION dfs WITH parameters i AND j  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            SET score_from_left TO element at position i of piles MINUS dfs CALLED WITH i INCREMENTED BY one AND j  
            SET score_from_right TO element at position j of piles MINUS dfs CALLED WITH i AND j DECREMENTED BY one  
            RETURN the GREATER OF score_from_left AND score_from_right  
        END FUNCTION  
        SET result TO dfs CALLED WITH zero AND the LENGTH OF piles MINUS one  
        RETURN result GREATER THAN zero  
    END FUNCTION  
END CLASS
