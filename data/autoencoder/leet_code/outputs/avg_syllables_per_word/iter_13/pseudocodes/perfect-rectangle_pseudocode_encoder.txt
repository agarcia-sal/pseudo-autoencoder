CLASS Solution  
    FUNCTION isRectangleCover(rectangles)  
        SET X1 TO positive infinity  
        SET Y1 TO positive infinity  
        SET X2 TO negative infinity  
        SET Y2 TO negative infinity  
        
        SET points TO an empty set  
        SET area TO zero  
        
        FOR each rectangle consisting of x1 y1 x2 y2 IN rectangles  
            SET X1 TO the minimum of X1 AND x1  
            SET Y1 TO the minimum of Y1 AND y1  
            SET X2 TO the maximum of X2 AND x2  
            SET Y2 TO the maximum of Y2 AND y2  
            
            INCREMENT area BY the result of subtracting x1 FROM x2 MULTIPLIED BY the result of subtracting y1 FROM y2  
            
            FOR each point p IN a list consisting of the tuple pairs of x1 y1 then x1 y2 then x2 y1 then x2 y2  
                IF p IS PRESENT IN points  
                    REMOVE element p FROM points  
                ELSE  
                    APPEND element p TO points  
                END IF  
            END FOR  
        END FOR  
        
        IF area IS NOT EQUAL TO the product of subtracting X1 FROM X2 AND subtracting Y1 FROM Y2  
            RETURN False  
        END IF  
        
        IF points IS NOT EQUAL TO the set consisting of tuple pairs X1 Y1 AND X1 Y2 AND X2 Y1 AND X2 Y2  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
