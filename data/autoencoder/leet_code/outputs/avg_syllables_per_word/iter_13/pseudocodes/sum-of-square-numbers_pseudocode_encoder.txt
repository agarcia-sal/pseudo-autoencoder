CLASS Solution  
    FUNCTION judgeSquareSum(c)  
        SET left TO zero  
        SET right TO the INTEGER PART OF the SQUARE ROOT OF c  
        WHILE left is LESS THAN OR EQUAL TO right holds  
            SET current_sum TO left MULTIPLIED BY left PLUS right MULTIPLIED BY right  
            IF current_sum EQUALS c THEN  
                RETURN true  
            ELSE IF current_sum is LESS THAN c THEN  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        RETURN false  
    END FUNCTION  
END CLASS
