CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS empty OR the LENGTH OF stones IS LESS THAN two  
            RETURN False  
        END IF  
        
        SET stone_set TO a set containing all elements from stones  
        SET memo TO an empty dictionary  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the element at the last position of stones  
                RETURN True  
            END IF  
            
            IF the pair position AND jump EXISTS as a key in memo  
                RETURN the value stored in memo for the key pair position AND jump  
            END IF  
            
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump IS GREATER THAN zero AND the sum of position AND next_jump EXISTS in stone_set  
                    IF can_jump_to the sum of position AND next_jump AND next_jump RETURNS True  
                        SET memo at key pair position AND jump TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET memo at key pair position AND jump TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to the first element of stones AND zero  
    END FUNCTION  
END CLASS
