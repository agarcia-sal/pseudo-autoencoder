CLASS Solution  
    FUNCTION threeEqualParts(arr)  
        SET total_ones TO the SUM of all elements IN arr  
        
        IF total_ones EQUALS zero  
            RETURN a list containing zero and two  
        END IF  
        
        IF the REMAINDER when total_ones is DIVIDED BY three DOES NOT EQUAL zero  
            RETURN a list containing negative one and negative one  
        END IF  
        
        SET part_ones TO total_ones DIVIDED BY three  
        
        SET first_part_index TO zero  
        SET count_ones TO zero  
        FOR i FROM zero TO the LENGTH OF arr MINUS one  
            IF the element at position i of arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS one  
                    SET first_part_index TO i  
                END IF  
                IF count_ones EQUALS part_ones  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET second_part_index TO zero  
        SET count_ones TO zero  
        FOR i FROM zero TO the LENGTH OF arr MINUS one  
            IF the element at position i of arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS part_ones PLUS one  
                    SET second_part_index TO i  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET third_part_index TO zero  
        SET count_ones TO zero  
        FOR i FROM zero TO the LENGTH OF arr MINUS one  
            IF the element at position i of arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS two MULTIPLIED BY part_ones PLUS one  
                    SET third_part_index TO i  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET n TO the LENGTH OF arr MINUS third_part_index  
        IF first_part_index PLUS n IS LESS THAN OR EQUAL TO the LENGTH OF arr AND  
           second_part_index PLUS n IS LESS THAN OR EQUAL TO the LENGTH OF arr AND  
           the sublist from first_part_index TO first_part_index PLUS n MINUS one of arr EQUALS the sublist from second_part_index TO second_part_index PLUS n MINUS one of arr AND  
           the sublist from first_part_index TO first_part_index PLUS n MINUS one of arr EQUALS the sublist from third_part_index TO the END of arr  
            RETURN a list containing the value of first_part_index PLUS n MINUS one and the value of second_part_index PLUS n  
        END IF  
        
        RETURN a list containing negative one and negative one  
    END FUNCTION  
END CLASS
