CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        CREATE a mapping graph from default empty list for each key  
        CREATE a list indegree with numCourses elements all set to zero  
        
        FOR each pair course and prereq IN prerequisites  
            APPEND course TO element at position prereq of graph  
            INCREMENT element at position course of indegree BY one  
        END FOR  
        
        CREATE a queue containing each course from zero TO numCourses MINUS one WHERE element at position course of indegree EQUALS zero  
        
        WHILE the queue is not empty  
            REMOVE the element from the front of the queue and ASSIGN it TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN element at position current_course of graph  
                DECREMENT element at position next_course of indegree BY one  
                IF element at position next_course of indegree EQUALS zero  
                    APPEND next_course TO the queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN whether numCourses EQUALS zero  
    END FUNCTION  
END CLASS
