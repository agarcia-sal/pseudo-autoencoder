CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO a new empty mapping from each integer FROM zero TO n MINUS one TO empty list  
        FOR each triple u v w IN flights  
            APPEND pair of v and w TO element at position u of graph  
        END FOR  
  
        SET min_heap TO a list containing a triple of zero cost and negative one stops and src city  
        SET min_cost TO a new mapping with key src assigned to negative one  
  
        WHILE min_heap is NOT empty  
            REMOVE and OBTAIN the triple cost stops city FROM min_heap by the operation that removes the smallest element according to the first component  
  
            IF city EQUALS dst  
                RETURN cost  
            END IF  
  
            IF stops IS GREATER THAN OR EQUAL TO k  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
  
            FOR each pair neighbor price IN element at position city of graph  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  
  
                IF neighbor is NOT a key in min_cost OR next_stops IS LESS THAN the element at key neighbor in min_cost OR next_cost IS LESS THAN the first component of the smallest triple in min_heap  
                    SET the element at key neighbor in min_cost TO next_stops  
                    ADD the triple next_cost next_stops neighbor TO min_heap by the operation that adds an element maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
  
        RETURN negative one  
    END FUNCTION  
END CLASS
