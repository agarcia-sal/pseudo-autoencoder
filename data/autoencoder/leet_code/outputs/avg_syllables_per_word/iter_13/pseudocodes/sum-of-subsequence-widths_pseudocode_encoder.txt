CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MODULO TO ten RAISED TO THE POWER OF nine PLUS seven  
        SORT the elements IN nums IN ASCENDING ORDER  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        CREATE a list named power WITH n ELEMENTS ALL SET TO one  
        
        FOR index FROM one TO n MINUS one  
            SET element at position index of power TO the ELEMENT at position index MINUS one of power MULTIPLIED BY two THEN TAKE THE REMAINDER AFTER DIVISION BY MODULO  
        END FOR  
        
        FOR index FROM zero TO n MINUS one  
            SET total_sum TO total_sum PLUS the ELEMENT at position index of nums MULTIPLIED BY the ELEMENT at position index of power MINUS the ELEMENT at position n MINUS index MINUS one of power THEN TAKE THE REMAINDER AFTER DIVISION BY MODULO  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
