CLASS Solution  
    FUNCTION longestWord(words)  
        CALL method TO SORT words BY length AND THEN BY lexicographical order  
        CREATE empty SET valid_words  
        SET longest_word TO empty string  
        FOR each word IN words  
            IF length of word EQUALS one OR substring from position zero TO length of word MINUS one OF word IS IN valid_words  
                ADD word TO valid_words  
                IF length of word GREATER THAN length of longest_word  
                    SET longest_word TO word  
                END IF  
            END IF  
        END FOR  
        RETURN longest_word  
    END FUNCTION  
END CLASS
