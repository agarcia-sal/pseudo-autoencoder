CLASS Solution  
    FUNCTION canDistribute(nums AS list of integers, quantity AS list of integers) RETURNS boolean  
        SET count TO list of the frequency values of each unique element in nums  
        SORT quantity IN DESCENDING ORDER  
        FUNCTION can_satisfy(index AS integer) RETURNS boolean  
            IF index EQUALS the LENGTH OF quantity THEN  
                RETURN true  
            END IF  
            FOR each position i FROM zero TO the LENGTH OF count MINUS one  
                IF the element at position i of count IS GREATER THAN OR EQUAL TO the element at position index of quantity THEN  
                    DECREMENT the element at position i of count BY the element at position index of quantity  
                    IF calling can_satisfy with parameter index PLUS one RETURNS true THEN  
                        RETURN true  
                    END IF  
                    INCREMENT the element at position i of count BY the element at position index of quantity  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        RETURN the result of calling can_satisfy with parameter zero  
    END FUNCTION  
END CLASS
