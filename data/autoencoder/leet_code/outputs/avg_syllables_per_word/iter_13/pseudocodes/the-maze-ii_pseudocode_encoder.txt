CLASS Solution  
    FUNCTION shortestDistance WITH PARAMETERS maze AS list of list of integer AND start AS list of integer AND destination AS list of integer  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO a list containing the pairs one and zero MINUS one and zero AND zero and one AND zero and MINUS one  
        SET start TO the tuple representation of start  
        SET destination TO the tuple representation of destination  
        SET visited TO an empty set  
        SET priority queue pq TO a list containing a pair of zero and start  

        WHILE the priority queue pq is not empty  
            REMOVE and get the pair with the smallest first element from pq ASSIGN distance to dist AND the pair elements to x and y  
            
            IF the pair x and y EQUALS destination  
                RETURN dist  
            END IF  
            
            IF the pair x and y IS IN visited  
                CONTINUE with the next iteration of the WHILE loop  
            END IF  
            
            ADD the pair x and y TO visited  
            
            FOR each pair dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the element at position new_x PLUS dx of the maze at position new_y PLUS dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF the pair new_x and new_y IS NOT IN visited  
                    ADD a pair of dist PLUS count AND a pair of new_x and new_y TO the priority queue pq  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
