CLASS Solution  
    FUNCTION ladderLength(beginWord, endWord, wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN zero  
        END IF  
        
        SET word_dict TO an empty dictionary mapping keys TO lists of words  
        SET word_length TO the LENGTH OF beginWord  
        
        FOR each word IN wordList  
            FOR each index FROM zero TO word_length MINUS one  
                SET intermediate_key TO the concatenation of the substring from position zero TO index MINUS one of word PLUS the character Asterisk PLUS the substring from position index PLUS one TO the end of word  
                APPEND word TO the list corresponding TO intermediate_key in word_dict  
            END FOR  
        END FOR  
        
        SET queue TO an empty double-ended queue containing the tuple of beginWord AND the integer one  
        SET visited TO a set containing beginWord  
        
        WHILE the queue is not empty  
            REMOVE the element at the front of the queue AND ASSIGN it TO current_word AND current_length  
            
            FOR each index FROM zero TO word_length MINUS one  
                SET intermediate_word TO the concatenation of the substring from position zero TO index MINUS one of current_word PLUS the character Asterisk PLUS the substring from position index PLUS one TO the end of current_word  
                FOR each word IN the list corresponding TO intermediate_word in word_dict  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND the tuple of word AND the integer current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN zero  
    END FUNCTION  
END CLASS
