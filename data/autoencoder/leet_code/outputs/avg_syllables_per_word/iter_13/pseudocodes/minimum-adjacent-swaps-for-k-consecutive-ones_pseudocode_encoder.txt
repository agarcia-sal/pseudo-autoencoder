CLASS Solution  
    FUNCTION minMoves(nums AS List OF int, k AS int) RETURNS int  
        SET positions TO an empty list  
        FOR each index AND number IN the enumeration of nums  
            IF number EQUALS one  
                APPEND index TO positions  
            END IF  
        END FOR  
        
        FUNCTION calculate_cost(start AS int, end AS int) RETURNS int  
            SET mid TO start PLUS end DIVIDED BY two  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR index FROM start TO end MINUS one  
                INCREMENT cost BY the absolute value of the subtraction of element at position index of positions AND the subtraction of median AND the subtraction of mid AND index  
            END FOR  
            RETURN cost  
        END FUNCTION  
        
        SET min_cost TO infinity  
        FOR i FROM zero TO the LENGTH OF positions MINUS k  
            SET min_cost TO the minimum of min_cost AND the result of calculate_cost with arguments i AND i PLUS k  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
