CLASS Solution  
    FUNCTION canPartitionKSubsets(nums, k)  
        SET total_sum TO the SUM OF elements IN nums  
        IF the REMAINDER of total_sum DIVIDED BY k IS NOT EQUAL TO zero  
            RETURN False  
        END IF  
        SET target TO total_sum DIVIDED BY k  
        SORT nums IN descending order  
        
        FUNCTION can_partition(index, k, current_sum, used)  
            IF k EQUALS zero  
                RETURN True  
            END IF  
            IF current_sum EQUALS target  
                RETURN the RESULT OF can_partition WITH parameters zero, k MINUS one, zero, used  
            END IF  
            
            FOR i FROM index TO the LENGTH OF nums MINUS one  
                IF element at position i of used IS False AND the SUM OF current_sum PLUS element at position i of nums IS LESS THAN OR EQUAL TO target  
                    SET element at position i of used TO True  
                    IF the RESULT OF can_partition WITH parameters i PLUS one, k, current_sum PLUS element at position i of nums, used IS True  
                        RETURN True  
                    END IF  
                    SET element at position i of used TO False  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET used TO a list OF False values WITH the LENGTH OF nums  
        RETURN the RESULT OF can_partition WITH parameters zero, k, zero, used  
    END FUNCTION  
END CLASS
