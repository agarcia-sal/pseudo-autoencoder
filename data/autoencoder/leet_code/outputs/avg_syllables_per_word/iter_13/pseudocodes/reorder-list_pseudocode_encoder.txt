CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val ASSIGNED ZERO AND next ASSIGNED None  
        SET attribute val OF self TO val  
        SET attribute next OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO NEW ListNode WITH element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO end  
        SET node TO NEW ListNode WITH val  
        SET attribute next OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN attribute val OF p1 EQUALS attribute val OF p2 AND FUNCTION CALL is_same_list WITH PARAMETERS attribute next OF p1 AND attribute next OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList WITH PARAMETERS head AS Optional ListNode  
        IF head EQUALS None OR attribute next OF head EQUALS None  
            RETURN  
        END IF  
        SET slow TO head  
        SET fast TO head  
        WHILE fast IS NOT None AND attribute next OF fast IS NOT None  
            SET slow TO attribute next OF slow  
            SET fast TO attribute next OF attribute next OF fast  
        END WHILE  
        SET prev TO None  
        SET curr TO slow  
        WHILE curr IS NOT None  
            SET next_temp TO attribute next OF curr  
            SET attribute next OF curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  
        SET first TO head  
        SET second TO prev  
        WHILE attribute next OF second IS NOT None  
            SET temp1 TO attribute next OF first  
            SET temp2 TO attribute next OF second  
            SET attribute next OF first TO second  
            SET attribute next OF second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
