CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET last TO a list containing negative one repeated twenty six times  
        SET last2 TO a list containing negative one repeated twenty six times  
        SET result TO zero  
        FOR each pair of index i AND character char IN enumeration of s  
            SET idx TO the ASCII code of char MINUS the ASCII code of capital A  
            SET result TO the REMAINDER after division by MOD of result PLUS the difference between i AND element at position idx of last MULTIPLIED BY the difference between element at position idx of last AND element at position idx of last2  
            SET element at position idx of last2 TO element at position idx of last  
            SET element at position idx of last TO i  
        END FOR  
        FOR each index i FROM zero TO twenty five  
            SET result TO the REMAINDER after division by MOD of result PLUS the difference between n AND element at position i of last MULTIPLIED BY the difference between element at position i of last AND element at position i of last2  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
