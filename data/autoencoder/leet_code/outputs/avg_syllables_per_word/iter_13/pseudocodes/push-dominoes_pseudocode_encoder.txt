CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a LIST of zero values WITH LENGTH n  
  
        SET force TO zero  
        FOR each index i FROM zero TO n MINUS one  
            IF the element at position i of dominoes EQUALS character R  
                SET force TO n  
            ELSE IF the element at position i of dominoes EQUALS character L  
                SET force TO zero  
            ELSE  
                SET force TO the MAXIMUM VALUE BETWEEN force MINUS one AND zero  
            END IF  
            INCREMENT the element at position i of forces BY force  
        END FOR  
  
        SET force TO zero  
        FOR each index i FROM n MINUS one DOWN TO zero  
            IF the element at position i of dominoes EQUALS character L  
                SET force TO n  
            ELSE IF the element at position i of dominoes EQUALS character R  
                SET force TO zero  
            ELSE  
                SET force TO the MAXIMUM VALUE BETWEEN force MINUS one AND zero  
            END IF  
            DECREMENT the element at position i of forces BY force  
        END FOR  
  
        SET result TO an empty LIST  
        FOR each force value f IN forces  
            IF f GREATER THAN zero  
                APPEND character R TO result  
            ELSE IF f LESS THAN zero  
                APPEND character L TO result  
            ELSE  
                APPEND character dot TO result  
            END IF  
        END FOR  
  
        RETURN the CONCATENATION of all elements in result  
    END FUNCTION  
END CLASS
