CLASS Solution  
    FUNCTION maxA(n)  
        IF n LESS THAN OR EQUAL TO three THEN  
            RETURN n  
        END IF  
        SET dp TO list OF zeroes OF LENGTH n PLUS one  
        FOR i FROM one TO three  
            SET element at position i of dp TO i  
        END FOR  
        FOR i FROM four TO n  
            FOR j FROM i MINUS three DOWN TO one  
                SET element at position i of dp TO the GREATER OF the element at position i of dp AND the element at position j of dp MULTIPLIED BY (i MINUS j MINUS one)  
            END FOR  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
