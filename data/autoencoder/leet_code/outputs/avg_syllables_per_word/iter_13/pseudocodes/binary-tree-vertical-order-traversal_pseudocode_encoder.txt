CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero left EQUALS None right EQUALS None  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values list  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH THE ELEMENT at position zero OF values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element REMOVED FROM THE FRONT of queue  
        IF i LESS THAN the LENGTH OF values AND the ELEMENT at position i of values IS NOT None THEN  
            SET node.left TO NEW TreeNode WITH THE ELEMENT at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the ELEMENT at position i of values IS NOT None THEN  
            SET node.right TO NEW TreeNode WITH THE ELEMENT at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUALS TO the val ATTRIBUTE OF q THEN  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS THE left ATTRIBUTE OF p AND THE left ATTRIBUTE OF q AND the RESULT OF is_same_tree WITH PARAMETERS THE right ATTRIBUTE OF p AND THE right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder WITH PARAMETERS self root OPTIONAL OF TYPE TreeNode  
        IF root IS None THEN  
            RETURN an empty list  
        END IF  
        SET column_table TO a new default dictionary WITH list AS DEFAULT FACTORY  
        SET queue TO a new double ended queue INITIALIZED WITH A tuple OF root AND zero  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE queue IS NOT empty  
            SET node AND column_index TO the elements REMOVED FROM THE FRONT OF queue  
            IF node IS NOT None THEN  
                APPEND the val ATTRIBUTE OF node TO the ELEMENT at key column_index OF column_table  
                IF column_index LESS THAN min_column THEN  
                    SET min_column TO column_index  
                END IF  
                IF column_index GREATER THAN max_column THEN  
                    SET max_column TO column_index  
                END IF  
                APPEND A tuple OF the left ATTRIBUTE OF node AND column_index MINUS one TO queue  
                APPEND A tuple OF the right ATTRIBUTE OF node AND column_index PLUS one TO queue  
            END IF  
        END WHILE  
        RETURN A list CONSTRUCTED BY THE ELEMENT at key x OF column_table FOR EACH x FROM min_column TO max_column INCLUSIVE  
    END FUNCTION  
END CLASS
