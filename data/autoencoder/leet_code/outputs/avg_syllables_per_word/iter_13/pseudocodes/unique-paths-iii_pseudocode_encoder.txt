CLASS Solution  
    FUNCTION uniquePathsIII(grid)  
        FUNCTION dfs(x, y, steps)  
            IF element at position x of grid EQUALS 2 AND element at position y of element at position x of grid EQUALS 2 THEN  
                RETURN steps EQUALS zero  
            END IF  
            SET original TO element at position x of element at position y of grid  
            SET element at position x of element at position y of grid TO negative two  
            SET count TO zero  
            FOR each pair dx AND dy IN list containing pair negative one AND zero, pair one AND zero, pair zero AND negative one, pair zero AND one  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx of element at position ny of grid GREATER THAN OR EQUAL TO zero THEN  
                    INCREMENT count BY the result of dfs WITH nx, ny, steps MINUS one AS parameters  
                END IF  
            END FOR  
            SET element at position x of element at position y of grid TO original  
            RETURN count  
        END FUNCTION  

        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET start_x TO zero  
        SET start_y TO zero  
        SET empty_squares TO one  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of element at position j of grid EQUALS one THEN  
                    SET start_x TO i  
                    SET start_y TO j  
                ELSE IF element at position i of element at position j of grid EQUALS zero THEN  
                    INCREMENT empty_squares BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN the result of dfs WITH start_x, start_y, empty_squares AS parameters  
    END FUNCTION  
END CLASS
