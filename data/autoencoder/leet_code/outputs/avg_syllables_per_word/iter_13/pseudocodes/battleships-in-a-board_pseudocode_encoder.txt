CLASS Solution  
    FUNCTION countBattleships(board)  
        SET count TO zero  
        FOR i FROM zero TO the LENGTH OF board MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of board MINUS one  
                IF the element at position i of board and position j of that element EQUALS character X THEN  
                    IF the value of i EQUALS zero OR the element at position i MINUS one of board and position j of that element EQUALS character representing a single dot AND the value of j EQUALS zero OR the element at position i of board and position j MINUS one of that element EQUALS character representing a single dot THEN  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
