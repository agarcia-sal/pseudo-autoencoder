CLASS Solution  
    FUNCTION decodeString WITH PARAMETER s  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each character IN s  
            IF character IS a digit  
                SET d TO d PLUS character  
            ELSE IF character IS a letter of the alphabet  
                SET a TO a PLUS character  
            ELSE IF character IS equal to the opening bracket symbol  
                APPEND the tuple containing a AND the integer conversion of d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF character IS equal to the closing bracket symbol  
                SET p AND n TO the last element pop from stack  
                SET a TO p PLUS a REPEATED n times  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
