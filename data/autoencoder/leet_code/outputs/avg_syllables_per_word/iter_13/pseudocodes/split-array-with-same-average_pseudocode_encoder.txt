CLASS Solution  
    FUNCTION splitArraySameAverage(nums)  
        SET n TO the LENGTH OF nums  
        SET total_sum TO the SUM OF all elements IN nums  
        
        IF n EQUALS one  
            RETURN False  
        END IF  
        
        SET dp TO a list OF empty sets WITH length equal TO integer DIVISION of n BY two PLUS one  
        ADD zero TO the set AT position zero OF dp  
        
        FOR each num IN nums  
            FOR size FROM integer DIVISION of n BY two DOWN TO one  
                FOR each s IN the set AT position size MINUS one OF dp  
                    SET new_sum TO s PLUS num  
                    IF new_sum MULTIPLIED BY n EQUALS size MULTIPLIED BY total_sum  
                        RETURN True  
                    END IF  
                    ADD new_sum TO the set AT position size OF dp  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
