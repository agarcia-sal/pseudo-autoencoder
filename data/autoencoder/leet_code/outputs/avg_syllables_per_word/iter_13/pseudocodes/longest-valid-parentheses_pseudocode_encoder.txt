CLASS Solution  
    FUNCTION longestValidParentheses WITH PARAMETER s  
        SET stack TO a new list initialized with minus one  
        SET max_length TO zero  
        FOR each index i AND each character char IN the enumeration of s  
            IF char EQUALS open parenthesis  
                APPEND i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF the LENGTH OF stack EQUALS zero  
                    APPEND i TO stack  
                ELSE  
                    SET max_length TO the GREATER BETWEEN max_length AND i MINUS the last element of stack  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
