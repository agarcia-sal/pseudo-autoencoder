CLASS Solution  
    FUNCTION killProcess(pid AS List OF Integer, ppid AS List OF Integer, kill AS Integer) RETURNS List OF Integer  
        CALL build_default_dict_list TO ASSIGN tree  
        
        FOR each pair child AND parent IN zip of pid AND ppid  
            APPEND child TO element at position parent OF tree  
        END FOR  
        
        SET queue TO list containing kill  
        SET killed_processes TO empty list  
        
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            REMOVE element at position zero FROM queue AND ASSIGN TO current  
            APPEND current TO killed_processes  
            EXTEND queue BY elements AT position current OF tree  
        END WHILE  
        
        RETURN killed_processes  
    END FUNCTION  
END CLASS
