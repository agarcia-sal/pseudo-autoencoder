CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome(check_str)  
            IF check_str EQUALS the reverse of check_str  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        FOR i FROM the LENGTH OF s DOWN TO zero  
            IF CALL is_palindrome WITH PARAMETERS the substring from position zero TO i OF s HOLDS true  
                SET to_add TO the reverse of the substring from position i TO the end OF s  
                RETURN the concatenation of to_add AND s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
