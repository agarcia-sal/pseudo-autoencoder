CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN zero  
            SET sign TO negative one  
        ELSE  
            SET sign TO one  
        END IF  
        SET string_representation TO the absolute value of x converted TO string  
        SET reversed_string TO the characters of string_representation in reverse order  
        SET reversed_x TO reversed_string converted TO integer  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative two RAISED TO the power of thirty one OR result GREATER THAN two RAISED TO the power of thirty one MINUS one  
            RETURN zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
