CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO substring from position one TO d OF frag  
                SET right TO substring from position d PLUS one TO n OF frag  
                IF (left DOES NOT START WITH character zero OR left EQUALS character zero) AND (right DOES NOT END WITH character zero)  
                    RETURN left CONCATENATED WITH (a decimal point IF d DOES NOT EQUAL n OTHERWISE an empty string) CONCATENATED WITH right  
                END IF  
            END FOR  
        END FUNCTION  
  
        SET s TO substring from position two TO the LENGTH OF s MINUS one OF s  
        SET result TO an empty list  
        FOR i FROM one TO the LENGTH OF s MINUS one  
            FOR each pair a, b IN the CARTESIAN PRODUCT OF make(substring from position one TO i OF s) AND make(substring from position i PLUS one TO the LENGTH OF s OF s)  
                APPEND an opening parenthesis CONCATENATED WITH a CONCATENATED WITH a comma and space CONCATENATED WITH b CONCATENATED WITH a closing parenthesis TO result  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
