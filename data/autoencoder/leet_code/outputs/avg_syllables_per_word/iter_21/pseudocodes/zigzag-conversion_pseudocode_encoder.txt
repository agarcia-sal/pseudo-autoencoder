CLASS Solution  
    FUNCTION convert(s, numRows)  
        IF numRows EQUALS one  
            RETURN s  
        END IF  
        CALL create_empty_string_list WITH PARAMETERS numRows AND SET RESULT TO rows  
        SET current_row TO zero  
        SET going_down TO false  
        FOR each char IN s  
            SET string at position current_row OF rows TO string at position current_row OF rows PLUS char  
            IF current_row EQUALS zero OR current_row EQUALS numRows MINUS one  
                SET going_down TO NOT going_down  
            END IF  
            IF going_down  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
        CALL join_strings_in_list WITH PARAMETERS rows AND RETURN RESULT  
    END FUNCTION  

FUNCTION create_empty_string_list(numRows)  
    SET result TO an empty list  
    FOR index FROM zero TO numRows MINUS one  
        APPEND empty string TO result  
    END FOR  
    RETURN result  
END FUNCTION  

FUNCTION join_strings_in_list(string_list)  
    SET joined_string TO empty string  
    FOR each item IN string_list  
        SET joined_string TO joined_string PLUS item  
    END FOR  
    RETURN joined_string  
END FUNCTION  
END CLASS
