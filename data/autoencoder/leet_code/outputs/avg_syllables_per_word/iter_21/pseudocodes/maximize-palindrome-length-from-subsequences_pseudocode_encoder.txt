CLASS Solution
    FUNCTION longestPalindrome(word1, word2)
        SET s TO the concatenation of word1 and word2
        SET n TO the LENGTH OF s
        SET dp TO a two-dimensional collection of size n BY n FILLED WITH zeroes
        
        FOR i FROM zero TO n MINUS one
            SET the element at position i AT position i OF dp TO one
        END FOR
        
        SET max_len TO zero
        FOR length FROM two TO n
            FOR i FROM zero TO n MINUS length
                SET j TO i PLUS length MINUS one
                IF the character at position i OF s EQUALS the character at position j OF s
                    SET the element at position i AT position j OF dp TO the element at position i PLUS one AT position j MINUS one OF dp PLUS two
                    IF i LESS THAN the LENGTH OF word1 AND j GREATER THAN OR EQUAL TO the LENGTH OF word1
                        SET max_len TO the GREATER OF max_len AND the element at position i AT position j OF dp
                    END IF
                ELSE
                    SET the element at position i AT position j OF dp TO the GREATER OF the element at position i PLUS one AT position j OF dp AND the element at position i AT position j MINUS one OF dp
                END IF
            END FOR
        END FOR
        
        RETURN max_len
    END FUNCTION
END CLASS
