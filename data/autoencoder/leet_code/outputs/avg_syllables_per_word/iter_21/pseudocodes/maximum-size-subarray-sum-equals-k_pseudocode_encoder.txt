CLASS Solution  
    FUNCTION maxSubArrayLen(nums, k)  
        SET cumulative_sum_index TO a dictionary WITH key zero AND value negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        
        FOR each index i AND element num IN nums  
            INCREMENT cumulative_sum BY num  
            
            IF cumulative_sum MINUS k EXISTS AS A KEY IN cumulative_sum_index THEN  
                SET max_length TO THE GREATER OF max_length AND i MINUS THE VALUE AT KEY cumulative_sum MINUS k IN cumulative_sum_index  
            END IF  
            
            IF cumulative_sum DOES NOT EXIST AS A KEY IN cumulative_sum_index THEN  
                SET THE VALUE AT KEY cumulative_sum IN cumulative_sum_index TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
