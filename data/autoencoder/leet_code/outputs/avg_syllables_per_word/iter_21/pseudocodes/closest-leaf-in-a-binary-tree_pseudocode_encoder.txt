CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty THEN  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH val EQUALS element at position zero OF values  
    SET i TO one  
    SET queue TO NEW empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element REMOVED FROM FRONT OF queue  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None THEN  
            SET node.left TO NEW TreeNode WITH val EQUALS element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None THEN  
            SET node.right TO NEW TreeNode WITH val EQUALS element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self, root, k)  
        IF root.left IS None AND root.right IS None THEN  
            RETURN root.val  
        END IF  
        SET graph TO NEW empty mapping OF lists WITH default type as list  
        
        FUNCTION dfs(node, parent EQUALS None)  
            IF parent IS NOT None THEN  
                APPEND parent.val TO element at key node.val OF graph  
                APPEND node.val TO element at key parent.val OF graph  
            END IF  
            IF node.left IS NOT None THEN  
                CALL dfs WITH PARAMETERS node.left, node  
            END IF  
            IF node.right IS NOT None THEN  
                CALL dfs WITH PARAMETERS node.right, node  
            END IF  
        END FUNCTION  
        
        CALL dfs WITH PARAMETERS root  
        
        SET queue TO NEW empty deque CONTAINING element k  
        SET visited TO NEW empty set  
        
        WHILE queue IS NOT empty  
            SET current TO element REMOVED FROM FRONT OF queue  
            ADD current TO visited  
            
            IF LENGTH OF element at key current OF graph EQUALS one AND current NOT EQUALS root.val THEN  
                RETURN current  
            END IF  
            
            FOR each neighbor IN element at key current OF graph  
                IF neighbor NOT IN visited THEN  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN MINUS one  
    END FUNCTION  
END CLASS
