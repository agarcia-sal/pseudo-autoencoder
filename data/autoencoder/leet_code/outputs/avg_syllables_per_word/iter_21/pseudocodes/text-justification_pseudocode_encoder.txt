CLASS Solution  
    FUNCTION fullJustify(words, maxWidth)  
        FUNCTION justify_line(line, num_of_letters, is_last_line)  
            IF the LENGTH OF line EQUALS one OR is_last_line IS TRUE  
                RETURN the CONCATENATION OF the ELEMENTS IN line JOINED BY a SINGLE SPACE FOLLOWED BY SPACES REPEATED the DIFFERENCE BETWEEN maxWidth AND num_of_letters AND the LENGTH OF line MINUS one TIMES  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps  
                SET extra_spaces TO total_spaces MODULO num_of_gaps  
                FOR index FROM zero TO num_of_gaps MINUS one  
                    IF index LESS THAN extra_spaces  
                        APPEND SPACES REPEATED spaces_between_words PLUS one TO the ELEMENT at position index OF line  
                    ELSE  
                        APPEND SPACES REPEATED spaces_between_words TO the ELEMENT at position index OF line  
                    END IF  
                END FOR  
                RETURN the CONCATENATION OF the ELEMENTS IN line WITH NO SEPARATORS  
            END IF  
        END FUNCTION  
        
        SET result TO an EMPTY LIST  
        SET line TO an EMPTY LIST  
        SET num_of_letters TO zero  
        
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth  
                APPEND CALL TO justify_line WITH PARAMETERS line, num_of_letters, FALSE TO result  
                SET line TO an EMPTY LIST  
                SET num_of_letters TO zero  
            END IF  
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
        
        APPEND CALL TO justify_line WITH PARAMETERS line, num_of_letters, TRUE TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
