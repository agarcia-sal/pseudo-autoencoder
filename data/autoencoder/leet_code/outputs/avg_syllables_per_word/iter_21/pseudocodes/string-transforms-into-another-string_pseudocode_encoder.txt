CLASS Solution  
    FUNCTION canConvert(str1, str2)  
        IF str1 EQUALS str2 THEN  
            RETURN True  
        END IF  

        SET conversion_map TO an empty mapping from characters TO characters  

        FOR each pair of characters char1 AND char2 IN simultaneous traversal of str1 AND str2  
            IF char1 IS A KEY IN conversion_map THEN  
                IF the value mapped from char1 IN conversion_map DOES NOT EQUAL char2 THEN  
                    RETURN False  
                END IF  
            ELSE  
                SET the mapping from char1 TO char2 IN conversion_map  
            END IF  
        END FOR  

        SET unique_characters_in_str2 TO the set of all distinct characters IN str2  
        IF the count OF unique_characters_in_str2 EQUALS twenty six THEN  
            RETURN False  
        END IF  

        RETURN True  
    END FUNCTION  
END CLASS
