CLASS Solution  
    FUNCTION rangeBitwiseAnd(left INTEGER, right INTEGER) RETURNS INTEGER  
        SET shift TO zero  
        WHILE left IS LESS THAN right  
            SET left TO left DIVIDED BY two DISCARDING FRACTIONAL PART  
            SET right TO right DIVIDED BY two DISCARDING FRACTIONAL PART  
            INCREMENT shift BY one  
        END WHILE  
        RETURN left MULTIPLIED BY two RAISED TO THE POWER OF shift  
    END FUNCTION  
END CLASS
