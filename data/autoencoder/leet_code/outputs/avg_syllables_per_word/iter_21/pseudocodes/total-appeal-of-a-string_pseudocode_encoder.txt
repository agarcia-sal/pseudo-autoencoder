CLASS Solution  
    FUNCTION appealSum(s)  
        CALL initialize_last_with_negative_one FOR integer_twenty_six  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        
        FOR each position_index AND character IN enumeration OF s  
            CALL convert_character_to_index WITH parameter character AND ASSIGN RESULT TO index  
            INCREMENT current_appeal BY position_index MINUS element at position index OF last  
            INCREMENT total_appeal BY current_appeal  
            SET element at position index OF last TO position_index  
        END FOR  
        
        RETURN total_appeal  
    END FUNCTION  

    FUNCTION initialize_last_with_negative_one FOR count  
        CREATE list_of_length count  
        FOR each position FROM zero TO count MINUS one  
            SET element at position OF list TO minus one  
        END FOR  
        RETURN list  
    END FUNCTION  

    FUNCTION convert_character_to_index(character)  
        SET index_of_character TO the Unicode code point of character  
        SET index_of_a TO the Unicode code point of lowercase character a  
        SET index TO index_of_character MINUS index_of_a  
        RETURN index  
    END FUNCTION  
END CLASS
