CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF the LENGTH OF remaining EQUALS zero  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR i FROM zero TO the LENGTH OF remaining MINUS one  
                SET new_path TO the CONCATENATION OF path AND element at position i OF remaining  
                SET new_remaining TO the SUBLIST of remaining FROM position zero TO i MINUS one CONCATENATED WITH the SUBLIST of remaining FROM position i PLUS one TO the END  
                CALL backtrack WITH PARAMETERS new_path AND new_remaining  
            END FOR  
        END FUNCTION  
        SET permutations TO an empty list  
        CALL backtrack WITH PARAMETERS an empty list AND nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
