CLASS Solution  
    FUNCTION minTaps(n, ranges)  
        CALL initialize_max_range_array WITH PARAMETERS n ASSIGN TO max_range  
        
        FOR position FROM zero TO n  
            IF element at position position OF ranges GREATER THAN zero  
                CALL compute_left_boundary WITH PARAMETERS position AND element at position position OF ranges ASSIGN TO left  
                CALL compute_right_boundary WITH PARAMETERS n AND position AND element at position position OF ranges ASSIGN TO right  
                SET element at position left OF max_range TO the GREATER OF element at position left OF max_range AND right  
            END IF  
        END FOR  
        
        SET taps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        
        FOR position FROM zero TO n  
            IF position GREATER THAN farthest  
                RETURN negative one  
            END IF  
            IF position GREATER THAN current_end  
                INCREMENT taps BY one  
                SET current_end TO farthest  
            END IF  
            SET farthest TO the GREATER OF farthest AND element at position position OF max_range  
        END FOR  
        
        RETURN taps  
    END FUNCTION
    
    FUNCTION initialize_max_range_array WITH PARAMETERS n  
        RETURN a list of zeros OF length n PLUS one  
    END FUNCTION
    
    FUNCTION compute_left_boundary WITH PARAMETERS position AND range_value  
        IF position MINUS range_value LESS THAN zero  
            RETURN zero  
        ELSE  
            RETURN position MINUS range_value  
        END IF  
    END FUNCTION
    
    FUNCTION compute_right_boundary WITH PARAMETERS n AND position AND range_value  
        IF position PLUS range_value GREATER THAN n  
            RETURN n  
        ELSE  
            RETURN position PLUS range_value  
        END IF  
    END FUNCTION  
END CLASS
