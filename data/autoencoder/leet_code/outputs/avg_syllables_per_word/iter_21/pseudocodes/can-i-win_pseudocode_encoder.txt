CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the result of maxChoosableInteger MULTIPLIED BY the sum of maxChoosableInteger PLUS one DIVIDED BY two IS LESS THAN desiredTotal  
            RETURN False  
        END IF  

        IF desiredTotal IS LESS THAN OR EQUAL TO zero  
            RETURN True  
        END IF  

        SET memo TO an empty dictionary  

        FUNCTION can_win(current_total, used_numbers)  
            IF the pair of current_total AND used_numbers EXISTS AS a key IN memo  
                RETURN the value ASSOCIATED WITH the key of current_total AND used_numbers IN memo  
            END IF  

            FOR number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers AND the number ONE SHIFTED LEFT BY number EQUALS zero  
                    IF current_total PLUS number IS GREATER THAN OR EQUAL TO desiredTotal OR NOT CALL can_win WITH PARAMETERS current_total PLUS number AND the bitwise OR of used_numbers AND the number ONE SHIFTED LEFT BY number  
                        SET the value ASSOCIATED WITH the key of current_total AND used_numbers IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  

            SET the value ASSOCIATED WITH the key of current_total AND used_numbers IN memo TO False  
            RETURN False  
        END FUNCTION  

        RETURN CALL can_win WITH PARAMETERS zero AND zero  
    END FUNCTION  
END CLASS
