CLASS Solution  
    FUNCTION countOrders(n)  
        SET MOD TO one billion PLUS seven  
        SET count TO one  
        FOR i FROM two TO n  
            SET count TO count MULTIPLIED BY i MULTIPLIED BY two MULTIPLIED BY i MINUS one  
            SET count TO count MODULO MOD  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
