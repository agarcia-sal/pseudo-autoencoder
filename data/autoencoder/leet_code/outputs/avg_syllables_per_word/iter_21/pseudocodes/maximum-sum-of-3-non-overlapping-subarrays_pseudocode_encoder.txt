CLASS Solution  
    FUNCTION maxSumOfThreeSubarrays(nums, k)  
        SET n TO the LENGTH OF nums  
        SET sums TO a list of zeros with the LENGTH OF n MINUS k PLUS one  
        SET current_sum TO the SUM of elements from position zero TO k MINUS one OF nums  
        SET the element at position zero OF sums TO current_sum  
        
        FOR i FROM one TO n MINUS k  
            INCREMENT current_sum BY the element at position i PLUS k MINUS one OF nums MINUS the element at position i MINUS one OF nums  
            SET the element at position i OF sums TO current_sum  
        END FOR  
        
        SET left TO a list of zeros with the LENGTH OF n MINUS k PLUS one  
        SET right TO a list where each element is n MINUS k with the LENGTH OF n MINUS k PLUS one  
        
        SET max_sum TO zero  
        FOR i FROM zero TO n MINUS k  
            IF the element at position i OF sums IS GREATER THAN the element at position max_sum OF sums  
                SET max_sum TO i  
            END IF  
            SET the element at position i OF left TO max_sum  
        END FOR  
        
        SET max_sum TO n MINUS k  
        FOR i FROM n MINUS k DOWN TO zero  
            IF the element at position i OF sums IS GREATER THAN OR EQUAL TO the element at position max_sum OF sums  
                SET max_sum TO i  
            END IF  
            SET the element at position i OF right TO max_sum  
        END FOR  
        
        SET max_total TO zero  
        SET result TO a list of three zeros  
        FOR j FROM k TO n MINUS two TIMES k  
            SET i TO the element at position j MINUS k OF left  
            SET l TO the element at position j PLUS k OF right  
            SET total TO the element at position i OF sums PLUS the element at position j OF sums PLUS the element at position l OF sums  
            IF total IS GREATER THAN max_total  
                SET max_total TO total  
                SET result TO a list containing i, j, and l  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
