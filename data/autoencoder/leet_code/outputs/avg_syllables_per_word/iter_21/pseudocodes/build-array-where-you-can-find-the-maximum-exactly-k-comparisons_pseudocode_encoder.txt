CLASS Solution  
    FUNCTION numOfArrays(n, m, k)  
        SET MOD TO one multiplied by ten raised to the power of nine PLUS seven  
        
        FUNCTION initialize_dp()  
            CREATE three dimensional list dp WITH dimensions n PLUS one BY m PLUS one BY k PLUS one FILLED with zero  
            RETURN dp  
        END FUNCTION
        
        SET dp TO CALL initialize_dp  
        
        FOR each integer j FROM one TO m  
            SET element at position one, j, and one OF dp TO one  
        END FOR  
        
        FOR each integer i FROM two TO n  
            FOR each integer j FROM one TO m  
                FOR each integer p FROM one TO k  
                    FOR each integer x FROM one TO j MINUS one  
                        SET element at position i, j, p OF dp TO element at position i, j, p OF dp PLUS element at position i MINUS one, x, p MINUS one OF dp  
                        SET element at position i, j, p OF dp TO element at position i, j, p OF dp MODULO MOD  
                    END FOR  
                    SET element at position i, j, p OF dp TO element at position i, j, p OF dp PLUS element at position i MINUS one, j, p OF dp MULTIPLIED BY j  
                    SET element at position i, j, p OF dp TO element at position i, j, p OF dp MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        
        FOR each integer j FROM one TO m  
            SET result TO result PLUS element at position n, j, k OF dp  
            SET result TO result MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
