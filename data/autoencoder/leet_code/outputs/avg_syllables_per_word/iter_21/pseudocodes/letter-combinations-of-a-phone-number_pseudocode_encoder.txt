CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string THEN  
            RETURN empty list  
        END IF  
        
        SET phone_map TO a mapping from digits TO corresponding letter strings where digit two maps TO the string of letters a b c the digit three maps TO the string d e f the digit four maps TO the string g h i the digit five maps TO the string j k l the digit six maps TO the string m n o the digit seven maps TO the string p q r s the digit eight maps TO the string t u v and the digit nine maps TO the string w x y z  
        
        FUNCTION backtrack(combination next_digits)  
            IF the length of next_digits EQUALS zero THEN  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the string at position zero OF next_digits in phone_map  
                    CALL backtrack WITH PARAMETERS combination PLUS letter and the substring from position one TO the end OF next_digits  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO empty list  
        CALL backtrack WITH PARAMETERS empty string and digits  
        RETURN output  
    END FUNCTION  
END CLASS
