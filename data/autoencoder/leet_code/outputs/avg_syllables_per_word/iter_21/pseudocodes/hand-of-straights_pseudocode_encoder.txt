CLASS Solution  
    FUNCTION isNStraightHand(hand, groupSize)  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        
        CALL count TO THE FREQUENCY COUNT OF elements IN hand  
        CALL unique_cards TO THE SORTED LIST OF KEYS IN count  
        
        FOR each card IN unique_cards  
            IF the VALUE at key card IN count IS GREATER THAN zero THEN  
                SET needed TO the VALUE at key card IN count  
                FOR index FROM zero TO groupSize MINUS one  
                    IF the VALUE at key card PLUS index IN count IS LESS THAN needed THEN  
                        RETURN False  
                    END IF  
                    DECREMENT the VALUE at key card PLUS index IN count BY needed  
                END FOR  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
