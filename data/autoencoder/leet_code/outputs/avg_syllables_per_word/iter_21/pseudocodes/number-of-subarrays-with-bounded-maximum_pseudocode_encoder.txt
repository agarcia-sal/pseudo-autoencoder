CLASS Solution  
    FUNCTION numSubarrayBoundedMax(nums, left, right)  
        FUNCTION count(max_value)  
            SET count TO zero  
            SET current_length TO zero  
            FOR each num IN nums  
                IF num LESS THAN OR EQUAL TO max_value  
                    INCREMENT current_length BY one  
                    INCREMENT count BY current_length  
                ELSE  
                    SET current_length TO zero  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
      
        SET result_to_return TO the VALUE RETURNED BY CALL count WITH PARAMETERS right MINUS the VALUE RETURNED BY CALL count WITH PARAMETERS left MINUS one  
        RETURN result_to_return  
    END FUNCTION  
END CLASS
