CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET it TO an iterator OVER t  
            RETURN WHETHER all characters c IN s ARE PRESENT IN SEQUENTIAL ORDER USING it  
        END FUNCTION  
        
        SORT strs BY length IN DESCENDING ORDER AND THEN BY lexicographical ORDER  
        
        FOR each index i AND element word IN strs  
            SET unique TO true  
            FOR each index j AND element other IN strs  
                IF i NOT EQUALS j AND CALL is_subsequence WITH PARAMETERS word AND other HOLDS TRUE  
                    SET unique TO false  
                    BREAK THE INNER LOOP  
                END IF  
            END FOR  
            
            IF unique HOLDS TRUE  
                RETURN THE LENGTH OF word  
            END IF  
        END FOR  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
