CLASS Solution  
    FUNCTION addOperators(num, target) RETURNS List of strings  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO substring from position index TO position i MINUS one OF num  
                IF the LENGTH OF operand GREATER THAN one AND the element at position zero OF operand EQUALS zero  
                    CONTINUE TO next iteration  
                END IF  
                
                IF index EQUALS zero  
                    CALL dfs WITH PARAMETERS i, operand, the CONVERSION OF operand TO integer, the CONVERSION OF operand TO integer  
                ELSE  
                    CALL dfs WITH PARAMETERS i, the CONCATENATION OF path, the PLUS SIGN, and operand, the SUM OF value AND the CONVERSION OF operand TO integer, the CONVERSION OF operand TO integer  
                    CALL dfs WITH PARAMETERS i, the CONCATENATION OF path, the MINUS SIGN, and operand, the DIFFERENCE OF value AND the CONVERSION OF operand TO integer, the NEGATION OF the CONVERSION OF operand TO integer  
                    CALL dfs WITH PARAMETERS i, the CONCATENATION OF path, the MULTIPLICATION SIGN, and operand, the DIFFERENCE OF value AND prev PLUS the PRODUCT OF prev AND the CONVERSION OF operand TO integer, the PRODUCT OF prev AND the CONVERSION OF operand TO integer  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL dfs WITH PARAMETERS zero, an empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
