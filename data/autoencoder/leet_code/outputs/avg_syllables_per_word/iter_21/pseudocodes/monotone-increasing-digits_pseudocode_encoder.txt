CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO CONVERT number n TO a list of its digit characters  
        SET length TO the LENGTH OF digits  
        SET marker TO length  
        FOR i FROM length MINUS one DOWN TO one STEP minus one  
            IF the element at position i OF digits IS LESS THAN the element at position i MINUS one OF digits  
                SET the element at position i MINUS one OF digits TO the string representation of the integer value of the element at position i MINUS one OF digits MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one  
            SET the element at position i OF digits TO the character nine  
        END FOR  
        SET result_string TO CONCATENATE all elements OF digits INTO a single string  
        SET result_number TO CONVERT result_string TO an integer  
        RETURN result_number  
    END FUNCTION  
END CLASS
