CLASS Solution  
    FUNCTION maximizeXor(nums, queries)  
        CALL nums SORT IN ASCENDING ORDER  
        
        SET sorted_queries TO CALL sorted WITH PARAMETERS enumerate OF queries AND a KEY that TAKES x AND RETURNS the element at position one OF element at position one OF x  
        
        SET result TO A LIST OF the LENGTH OF queries FILLED WITH the VALUE MINUS ONE  
        
        SET trie TO an EMPTY MAPPING  
        
        SET num_index TO zero  
        
        FOR each tuple query_index AND pair xi mi IN sorted_queries  
            WHILE num_index IS LESS THAN the LENGTH OF nums AND the ELEMENT at position num_index OF nums IS LESS THAN OR EQUAL TO mi  
                SET num TO the ELEMENT at position num_index OF nums  
                SET node TO trie  
                FOR i FROM thirty ONE DOWN TO zero STEP MINUS ONE  
                    SET bit TO THE RESULT OF SHIFTING num RIGHT BY i BITS AND TAKING THE ELEMENT AT POSITION zero OF THE BINARY AND OPERATION WITH one  
                    IF bit IS NOT A KEY IN node  
                        SET the ELEMENT at key bit OF node TO an EMPTY MAPPING  
                    END IF  
                    SET node TO the ELEMENT at key bit OF node  
                END FOR  
                INCREMENT num_index BY one  
            END WHILE  
            
            IF trie IS EMPTY  
                CONTINUE TO THE NEXT ITERATION OF THE FOR LOOP  
            END IF  
            
            SET node TO trie  
            SET max_xor TO zero  
            FOR i FROM thirty ONE DOWN TO zero STEP MINUS ONE  
                SET bit TO THE RESULT OF SHIFTING xi RIGHT BY i BITS AND TAKING THE ELEMENT AT POSITION zero OF THE BINARY AND OPERATION WITH one  
                SET toggle_bit TO one MINUS bit  
                IF toggle_bit IS A KEY IN node  
                    SET max_xor TO max_xor OR THE RESULT OF SHIFTING one LEFT BY i BITS  
                    SET node TO the ELEMENT at key toggle_bit OF node  
                ELSE  
                    SET node TO the ELEMENT at key bit OF node  
                END IF  
            END FOR  
            
            SET the ELEMENT at position query_index OF result TO max_xor  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
