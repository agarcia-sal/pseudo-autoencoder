CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET max_so_far TO an empty list  
        SET current_max TO element at position zero OF arr  
        FOR each num IN arr  
            SET current_max TO the greater value BETWEEN current_max AND num  
            APPEND current_max TO max_so_far  
        END FOR  
        
        SET sorted_arr TO CALL sorted WITH PARAMETERS arr  
        SET chunks TO zero  
        SET max_chunk TO zero  
        
        FOR i FROM zero TO the LENGTH OF arr MINUS one  
            SET max_chunk TO the greater value BETWEEN max_chunk AND element at position i OF arr  
            IF max_chunk EQUALS element at position i OF sorted_arr  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        
        RETURN chunks  
    END FUNCTION  
END CLASS
