CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO an empty list  
        SET max_value TO negative infinity  
        
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            CALL heap push WITH PARAMETERS min_heap AND a tuple of element at position zero OF list at position index OF nums, index, and zero  
            IF max_value LESS THAN element at position zero OF list at position index OF nums THEN  
                SET max_value TO element at position zero OF list at position index OF nums  
            END IF  
        END FOR  
        
        SET smallest_range TO positive infinity  
        SET result TO a list with elements negative one hundred thousand and one hundred thousand  
        
        WHILE min_heap HOLDS  
            CALL heap pop WITH PARAMETERS min_heap AND ASSIGN the result TO min_value, list_index, element_index  
            
            IF max_value MINUS min_value LESS THAN smallest_range THEN  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO a list with elements min_value and max_value  
            END IF  
            
            IF element_index PLUS one LESS THAN the LENGTH OF list at position list_index OF nums THEN  
                SET next_value TO element at position element_index PLUS one OF list at position list_index OF nums  
                CALL heap push WITH PARAMETERS min_heap AND a tuple of next_value, list_index, element_index PLUS one  
                IF max_value LESS THAN next_value THEN  
                    SET max_value TO next_value  
                END IF  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
