CLASS Solution  
    FUNCTION frequencySort(s)  
        CALL import_Counter  
        SET frequency TO CALL Counter WITH PARAMETERS s  
        SET sorted_characters TO CALL sort_keys_of_frequency_by_descending_frequency WITH PARAMETERS frequency  
        SET result TO CALL build_result_string_with_repeated_characters WITH PARAMETERS sorted_characters AND frequency  
        RETURN result  
    END FUNCTION  

    FUNCTION import_Counter  
        PERFORM import OF Counter FROM collections  
    END FUNCTION  

    FUNCTION Counter(s)  
        RETURN a mapping of each character in s TO its total number of occurrences  
    END FUNCTION  

    FUNCTION sort_keys_of_frequency_by_descending_frequency(frequency)  
        SET keys_list TO the list of all keys IN frequency  
        SORT keys_list IN order that characters WITH HIGHER frequency APPEAR BEFORE characters WITH LOWER frequency  
        RETURN keys_list  
    END FUNCTION  

    FUNCTION build_result_string_with_repeated_characters(sorted_characters, frequency)  
        SET result_string TO empty string  
        FOR each character IN sorted_characters  
            SET repeated_characters TO character REPEATED frequency at position character TIMES  
            APPEND repeated_characters TO result_string  
        END FOR  
        RETURN result_string  
    END FUNCTION  
END CLASS
