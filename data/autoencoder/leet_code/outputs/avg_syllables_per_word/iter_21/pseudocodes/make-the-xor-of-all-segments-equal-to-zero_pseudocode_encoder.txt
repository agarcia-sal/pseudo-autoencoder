CLASS Solution  
    FUNCTION minChanges(nums, k)  
        SET groups TO call create_defaultdict_of_list  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            CALL append element at position index OF nums TO element at position index MODULO k OF groups  
        END FOR  
          
        SET freq TO call create_list_of_defaultdict_of_int WITH PARAMETER k  
        SET min_size TO list_of_infinity_of_length k  
        FOR index FROM zero TO k MINUS one  
            FOR each number IN element at position index OF groups  
                INCREMENT element at position number OF element at position index OF freq BY one  
            END FOR  
            SET element at position index OF min_size TO the LENGTH OF element at position index OF groups  
        END FOR  
          
        SET dp TO list_of_infinity_of_length two RAISED TO THE POWER OF ten  
        SET element at position zero OF dp TO zero  
          
        FOR index FROM zero TO k MINUS one  
            SET new_dp TO list_of_infinity_of_length two RAISED TO THE POWER OF ten  
            SET min_dp TO minimum value IN dp  
            FOR number FROM zero TO two RAISED TO THE POWER OF ten MINUS one  
                SET element at position number OF new_dp TO min_dp PLUS element at position index OF min_size  
                FOR each number_key AND count IN element at position index OF freq  
                    SET candidate TO element at position number XOR number_key OF dp PLUS element at position index OF min_size MINUS count  
                    IF candidate LESS THAN element at position number OF new_dp  
                        SET element at position number OF new_dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
          
        RETURN element at position zero OF dp  
    END FUNCTION  
END CLASS
