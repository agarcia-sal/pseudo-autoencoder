CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO CALL ListNode WITH PARAMETERS values AT position zero  
    SET p TO head  
    FOR each val IN the SUBLIST OF values FROM position one TO the END  
        SET node TO CALL ListNode WITH PARAMETERS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the VALUE AT p1.val EQUALS the VALUE AT p2.val AND CALL is_same_list WITH PARAMETERS p1.next, p2.next  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO CALL TreeNode WITH PARAMETERS values AT position zero  
    SET i TO one  
    SET queue TO CALL deque WITH NO PARAMETERS  
    CALL the METHOD append ON queue WITH PARAMETERS root  
    WHILE queue IS NOT EMPTY  
        SET node TO CALL the METHOD popleft ON queue WITH NO PARAMETERS  
        IF i LESS THAN the LENGTH OF values AND the ELEMENT AT position i OF values IS NOT None  
            SET node.left TO CALL TreeNode WITH PARAMETERS the ELEMENT AT position i OF values  
            CALL the METHOD append ON queue WITH PARAMETERS node.left  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the ELEMENT AT position i OF values IS NOT None  
            SET node.right TO CALL TreeNode WITH PARAMETERS the ELEMENT AT position i OF values  
            CALL the METHOD append ON queue WITH PARAMETERS node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    END IF  
    IF p IS None OR q IS None  
        RETURN False  
    END IF  
    IF the VALUE AT p.val NOT EQUALS the VALUE AT q.val  
        RETURN False  
    END IF  
    RETURN CALL is_same_tree WITH PARAMETERS p.left, q.left AND CALL is_same_tree WITH PARAMETERS p.right, q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO the LENGTH OF s  
                RETURN None, index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the LENGTH OF s AND the ELEMENT AT position index OF s IS A DIGIT OR the ELEMENT AT position index OF s EQUALS the CHARACTER minus  
                INCREMENT index BY one  
            END WHILE  
            SET node TO CALL TreeNode WITH PARAMETERS the INTEGER CONVERSION OF the SUBSTRING FROM position start TO index OF s  
            IF index LESS THAN the LENGTH OF s AND the ELEMENT AT position index OF s EQUALS the CHARACTER left parenthesis  
                SET node.left, index TO CALL parse_tree WITH PARAMETERS index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the ELEMENT AT position index OF s EQUALS the CHARACTER left parenthesis  
                SET node.right, index TO CALL parse_tree WITH PARAMETERS index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the ELEMENT AT position index OF s EQUALS the CHARACTER right parenthesis  
                INCREMENT index BY one  
            END IF  
            RETURN node, index  
        END FUNCTION  
        SET root, _ TO CALL parse_tree WITH PARAMETERS zero  
        RETURN root  
    END FUNCTION  
END CLASS
