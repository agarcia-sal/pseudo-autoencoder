CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO two-dimensional list of FALSE values with the number of rows EQUAL TO the length of s PLUS one and the number of columns EQUAL TO the length of p PLUS one  
        SET the element at row zero column zero OF dp TO TRUE  
          
        FOR j FROM one TO the length of p  
            IF the element at position j MINUS one OF p EQUALS the character STAR  
                SET the element at row zero column j OF dp TO the element at row zero column j MINUS two OF dp  
            END IF  
        END FOR  
          
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF the element at position j MINUS one OF p EQUALS the element at position i MINUS one OF s OR the element at position j MINUS one OF p EQUALS the character DOT  
                    SET the element at row i column j OF dp TO the element at row i MINUS one column j MINUS one OF dp  
                ELSE IF the element at position j MINUS one OF p EQUALS the character STAR  
                    SET the element at row i column j OF dp TO the element at row i column j MINUS two OF dp  
                    IF the element at position j MINUS two OF p EQUALS the element at position i MINUS one OF s OR the element at position j MINUS two OF p EQUALS the character DOT  
                        SET the element at row i column j OF dp TO the element at row i column j OF dp OR the element at row i MINUS one column j OF dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
          
        RETURN the element at row the length of s column the length of p OF dp  
    END FUNCTION  
END CLASS
