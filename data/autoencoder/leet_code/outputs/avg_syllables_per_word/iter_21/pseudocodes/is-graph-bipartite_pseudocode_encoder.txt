CLASS Solution  
    FUNCTION isBipartite(graph)  
        SET n TO the LENGTH OF graph  
        SET color TO a LIST of zeroes WITH LENGTH n
        
        FUNCTION dfs(node, c)  
            SET the element at position node OF color TO c  
            FOR each neighbor IN the element at position node OF graph  
                IF the element at position neighbor OF color EQUALS c THEN  
                    RETURN False  
                END IF  
                IF the element at position neighbor OF color EQUALS zero THEN  
                    IF NOT CALL dfs WITH PARAMETERS neighbor AND negative c THEN  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        FOR i FROM zero TO n MINUS one  
            IF the element at position i OF color EQUALS zero THEN  
                IF NOT CALL dfs WITH PARAMETERS i AND one THEN  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
