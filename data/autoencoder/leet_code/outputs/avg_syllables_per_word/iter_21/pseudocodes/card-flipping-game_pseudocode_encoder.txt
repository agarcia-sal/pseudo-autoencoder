CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO an empty set  
        FOR each pair consisting of element front_element at position index FROM fronts AND element back_element at position index FROM backs  
            IF front_element EQUALS back_element  
                ADD front_element TO same  
            END IF  
        END FOR  
        SET all_numbers TO the set containing all elements FROM fronts AND all elements FROM backs  
        SET good_numbers TO the set all_numbers MINUS the set same  
        IF the set good_numbers IS NOT empty  
            SET minimum_good_number TO the minimum element IN good_numbers  
            RETURN minimum_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
