CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL nums.sort WITH PARAMETERS  
        SET n TO the LENGTH OF nums  
        SET first TO element at position zero OF nums  
        FOR i FROM one TO n MINUS one  
            SET k TO element at position i OF nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at position i OF nums  
                CONTINUE TO next iteration  
            END IF  
            SET count TO CALL Counter WITH PARAMETERS nums  
            SET arr TO an empty list  
            FOR each x IN nums  
                IF value at key x IN count EQUALS zero  
                    CONTINUE TO next iteration  
                END IF  
                IF value at key x PLUS two MULTIPLIED BY k IN count EQUALS zero  
                    BREAK the loop  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT the value at key x IN count BY one  
                DECREMENT the value at key x PLUS two MULTIPLIED BY k IN count BY one  
            END FOR  
            IF the LENGTH OF arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an empty list  
    END FUNCTION  
END CLASS
