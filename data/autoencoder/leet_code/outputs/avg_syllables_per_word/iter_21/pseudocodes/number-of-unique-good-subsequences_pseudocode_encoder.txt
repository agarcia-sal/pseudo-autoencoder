CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MOD TO one billion PLUS seven  
        SET end_with_0 TO zero  
        SET end_with_1 TO zero  
        SET has_zero TO a BOOLEAN value indicating WHETHER the character zero IS IN the string binary  
        FOR each bit IN the string binary  
            IF the bit EQUALS the character one THEN  
                SET end_with_1 TO the sum of end_with_0 PLUS end_with_1 PLUS one MODULO MOD  
            ELSE  
                SET end_with_0 TO the sum of end_with_0 PLUS end_with_1 MODULO MOD  
            END IF  
        END FOR  
        RETURN the sum of end_with_0 PLUS end_with_1 PLUS has_zero MODULO MOD  
    END FUNCTION  
END CLASS
