CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS None THEN  
                RETURN zero  
            END IF  
            SET x1 TO the QUOTIENT OF the ORDINAL VALUE OF c1 MINUS the ORDINAL VALUE OF character A DIVIDED BY six  
            SET y1 TO the REMAINDER OF the ORDINAL VALUE OF c1 MINUS the ORDINAL VALUE OF character A DIVIDED BY six  
            SET x2 TO the QUOTIENT OF the ORDINAL VALUE OF c2 MINUS the ORDINAL VALUE OF character A DIVIDED BY six  
            SET y2 TO the REMAINDER OF the ORDINAL VALUE OF c2 MINUS the ORDINAL VALUE OF character A DIVIDED BY six  
            RETURN the ABSOLUTE VALUE OF x1 MINUS x2 PLUS the ABSOLUTE VALUE OF y1 MINUS y2  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the LENGTH OF word THEN  
                RETURN zero  
            END IF  
            SET c TO the element at position i OF word  
            SET d1 TO the CALL TO distance WITH PARAMETERS f1 AND c PLUS the CALL TO dp WITH PARAMETERS i PLUS one, c, AND f2  
            SET d2 TO the CALL TO distance WITH PARAMETERS f2 AND c PLUS the CALL TO dp WITH PARAMETERS i PLUS one, f1, AND c  
            RETURN the MINIMUM OF d1 AND d2  
        END FUNCTION  
        
        RETURN the CALL TO dp WITH PARAMETERS zero, None, AND None  
    END FUNCTION  
END CLASS
