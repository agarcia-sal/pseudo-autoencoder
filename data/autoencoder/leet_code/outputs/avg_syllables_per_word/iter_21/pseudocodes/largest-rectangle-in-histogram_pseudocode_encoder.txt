CLASS Solution  
    FUNCTION largestRectangleArea(heights)  
        SET stack TO a list containing the single element negative one  
        SET max_area TO zero  
        APPEND zero TO heights  
        
        FOR each index i AND each element h IN heights  
            WHILE the last element in stack NOT EQUALS negative one AND h LESS THAN the element at position the last element in stack OF heights HOLDS  
                SET height TO the element at position the last element removed FROM stack OF heights  
                SET width TO i MINUS the element at position the last element in stack MINUS one  
                SET max_area TO the maximum VALUE BETWEEN max_area AND height MULTIPLIED BY width  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
