CLASS Solution  
    FUNCTION compareVersion(version1, version2)  
        CALL version1 SPLIT BY period AND ASSIGN RESULT TO v1_revisions  
        CALL version2 SPLIT BY period AND ASSIGN RESULT TO v2_revisions  
        
        SET max_length TO the GREATER OF the LENGTH OF v1_revisions AND the LENGTH OF v2_revisions  
        
        FOR index FROM zero TO max_length MINUS one  
            IF index IS LESS THAN the LENGTH OF v1_revisions  
                CALL the element at position index OF v1_revisions CONVERT TO integer AND ASSIGN TO v1  
            ELSE  
                SET v1 TO zero  
            END IF  
            
            IF index IS LESS THAN the LENGTH OF v2_revisions  
                CALL the element at position index OF v2_revisions CONVERT TO integer AND ASSIGN TO v2  
            ELSE  
                SET v2 TO zero  
            END IF  
            
            IF v1 IS LESS THAN v2  
                RETURN negative one  
            ELSE IF v1 IS GREATER THAN v2  
                RETURN positive one  
            END IF  
        END FOR  
        
        RETURN zero  
    END FUNCTION  
END CLASS
