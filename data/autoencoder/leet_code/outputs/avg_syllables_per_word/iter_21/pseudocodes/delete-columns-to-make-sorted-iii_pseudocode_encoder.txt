CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero OF strs  
        SET dp TO a list of n elements each initialized TO one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF FOR every index k FROM zero TO m MINUS one the element at position j OF element at position k OF strs IS LESS THAN OR EQUAL TO the element at position i OF element at position k OF strs HOLDS  
                    IF the element at position i OF dp IS LESS THAN the element at position j OF dp PLUS one  
                        SET the element at position i OF dp TO the element at position j OF dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        SET longest_subsequence TO the MAXIMUM element IN dp  
        SET result TO n MINUS longest_subsequence  
        RETURN result  
    END FUNCTION  
END CLASS
