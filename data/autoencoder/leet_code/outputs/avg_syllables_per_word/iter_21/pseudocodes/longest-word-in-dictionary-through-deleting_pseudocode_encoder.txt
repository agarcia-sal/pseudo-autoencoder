CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET it TO an iterator over string s  
            FOR each character IN word  
                IF character NOT IN the remaining elements of iterator it  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        CALL a sorting method on dictionary TO ORDER BY negative length of element FOLLOWED BY element lexicographically  
        
        FOR each word IN dictionary  
            IF CALL is_subsequence WITH PARAMETERS word AND s RETURNS true  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN an empty string  
    END FUNCTION  
END CLASS
