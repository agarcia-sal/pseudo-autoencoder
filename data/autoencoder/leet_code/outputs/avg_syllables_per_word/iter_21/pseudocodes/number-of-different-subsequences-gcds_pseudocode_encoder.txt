CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET max_num TO the maximum element IN nums  
        SET num_set TO the set OF all elements IN nums  
        SET count TO zero  
        
        FOR each integer g FROM one TO max_num INCLUSIVE  
            SET gcd_val TO zero  
            FOR each integer m FROM g TO max_num INCLUSIVE WITH STEP SIZE g  
                IF m IS AN ELEMENT IN num_set  
                    SET gcd_val TO the greatest common divisor OF gcd_val AND m  
                    IF gcd_val EQUALS g  
                        INCREMENT count BY one  
                        BREAK the inner loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
