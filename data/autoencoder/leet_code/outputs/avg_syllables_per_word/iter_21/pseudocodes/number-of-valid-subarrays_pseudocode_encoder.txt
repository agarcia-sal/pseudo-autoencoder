CLASS Solution  
    FUNCTION validSubarrays(nums)  
        SET stack TO an empty list  
        SET count TO zero  
        
        FOR each num IN nums  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND element at position the LENGTH OF stack MINUS one OF stack IS GREATER THAN num  
                REMOVE the element at position the LENGTH OF stack MINUS one FROM stack  
            END WHILE  
            APPEND num TO stack  
            INCREMENT count BY the LENGTH OF stack  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
