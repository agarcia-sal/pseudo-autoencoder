CLASS Solution  
    FUNCTION findKthLargest(nums, k)  
        SET min_heap TO an empty list  
        FOR each num IN nums  
            CALL heap_push WITH PARAMETERS min_heap and num  
            IF the LENGTH OF min_heap GREATER THAN k  
                CALL heap_pop WITH PARAMETERS min_heap  
            END IF  
        END FOR  
        RETURN the element at position zero OF min_heap  
    END FUNCTION  
END CLASS
