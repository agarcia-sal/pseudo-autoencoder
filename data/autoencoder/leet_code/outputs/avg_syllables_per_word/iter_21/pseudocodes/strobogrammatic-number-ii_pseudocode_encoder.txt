CLASS Solution  
    FUNCTION findStrobogrammatic(n)  
        FUNCTION dfs(u)  
            IF u EQUALS zero  
                RETURN list CONTAINING empty string  
            END IF  
            IF u EQUALS one  
                RETURN list CONTAINING string zero, string one, AND string eight  
            END IF  
            SET ans TO empty list  
            FOR each v IN CALL dfs WITH PARAMETER u MINUS two  
                FOR each pair IN list OF strings eleven, eighty eight, sixty nine, ninety six  
                    APPEND concatenation OF first character OF pair PLUS v PLUS second character OF pair TO ans  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND concatenation OF string zero PLUS v PLUS string zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN CALL dfs WITH PARAMETER n  
    END FUNCTION  
END CLASS
