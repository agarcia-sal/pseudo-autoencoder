CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS NONE OR the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode OBJECT INITIALIZED WITH the element at position zero OF values  
    SET p TO head  
    FOR each val IN the sublist OF values STARTING FROM position one TO the end  
        SET node TO A NEW ListNode OBJECT INITIALIZED WITH val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS NONE AND p2 IS NONE THEN  
        RETURN true  
    END IF  
    IF p1 IS NONE OR p2 IS NONE THEN  
        RETURN false  
    END IF  
    RETURN (p1 dot val EQUALS p2 dot val) AND CALL is_same_list WITH PARAMETERS p1 dot next, p2 dot next  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight(self, head, k)  
        IF head IS NONE OR head dot next IS NONE OR k EQUALS zero THEN  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE tail dot next IS NOT NONE  
            SET tail TO tail dot next  
            INCREMENT length BY one  
        END WHILE  
        SET tail dot next TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR index FROM zero TO steps_to_new_tail  
            SET new_tail TO new_tail dot next  
        END FOR  
        SET new_head TO new_tail dot next  
        SET new_tail dot next TO none  
        RETURN new_head  
    END FUNCTION  
END CLASS
