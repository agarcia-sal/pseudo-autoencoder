CLASS Solution  
    FUNCTION numOfWays(n)  
        SET MODULO TO one billion PLUS seven  
        SET a TO six  
        SET b TO six  
        FOR each integer index FROM two TO n  
            SET new_a TO the sum of three times a PLUS two times b  
            SET new_b TO the sum of two times a PLUS two times b  
            SET a TO the remainder of new_a DIVIDED BY MODULO  
            SET b TO the remainder of new_b DIVIDED BY MODULO  
        END FOR  
        SET result TO the remainder of the sum of a PLUS b DIVIDED BY MODULO  
        RETURN result  
    END FUNCTION  
END CLASS
