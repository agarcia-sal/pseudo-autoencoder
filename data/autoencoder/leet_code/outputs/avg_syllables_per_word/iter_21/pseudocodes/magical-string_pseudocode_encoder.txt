CLASS Solution  
    FUNCTION magicalString(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        IF n LESS THAN OR EQUAL TO three  
            RETURN one  
        END IF  

        SET s TO a list containing the elements one, two, two  
        SET count TO one  
        SET index TO two  

        WHILE the LENGTH OF s LESS THAN n  
            SET next_num TO three MINUS the last element at the end OF s  
            EXTEND s BY a list containing next_num REPEATED the element at position index OF s TIMES  
            IF next_num EQUALS one  
                INCREMENT count BY the element at position index OF s  
            END IF  
            INCREMENT index BY one  
        END WHILE  

        IF the LENGTH OF s GREATER THAN n  
            DECREMENT count BY the last element at the end OF s MULTIPLIED BY the difference BETWEEN the LENGTH OF s AND n  
        END IF  

        RETURN count  
    END FUNCTION  
END CLASS
