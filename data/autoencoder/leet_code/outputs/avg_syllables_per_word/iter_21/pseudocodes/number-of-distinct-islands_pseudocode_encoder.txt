CLASS Solution  
    FUNCTION numDistinctIslands(grid)  
        FUNCTION dfs(i, j, direction)  
            IF i IS GREATER THAN OR EQUAL TO zero AND i IS LESS THAN the LENGTH OF grid AND j IS GREATER THAN OR EQUAL TO zero AND j IS LESS THAN the LENGTH OF element at position zero OF grid AND element at position i OF grid at position j EQUALS one  
                SET element at position i OF grid at position j TO zero  
                APPEND direction TO path  
                CALL dfs WITH PARAMETERS i PLUS one, j, character d  
                CALL dfs WITH PARAMETERS i MINUS one, j, character u  
                CALL dfs WITH PARAMETERS i, j PLUS one, character r  
                CALL dfs WITH PARAMETERS i, j MINUS one, character l  
                APPEND character b TO path  
            END IF  
            RETURN path  
        END FUNCTION  
        
        SET unique_islands TO an empty set  
        
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero OF grid MINUS one  
                IF element at position i OF grid at position j EQUALS one  
                    SET path TO an empty list  
                    ADD the result of CALLING dfs WITH PARAMETERS i, j, character o CONVERTED TO an immutable tuple TO unique_islands  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the LENGTH OF unique_islands  
    END FUNCTION  
END CLASS
