CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN list_containing_zero  
        END IF  
        SET prev_gray TO CALL grayCode WITH PARAMETERS n MINUS one  
        SET mask TO one LEFT SHIFTED BY n MINUS one  
        SET current_gray TO prev_gray PLUS list_comprised_of mask OR number FOR each number IN the REVERSED prev_gray  
        RETURN current_gray  
    END FUNCTION  
END CLASS
