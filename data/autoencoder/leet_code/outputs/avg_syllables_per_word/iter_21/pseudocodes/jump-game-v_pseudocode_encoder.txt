CLASS Solution  
    FUNCTION maxJumps(arr, d)  
        FUNCTION dp(i)  
            SET max_jumps TO one  
            FOR x FROM one TO d  
                IF i PLUS x LESS THAN THE LENGTH OF arr THEN  
                    IF element at position i PLUS x OF arr GREATER THAN OR EQUAL TO element at position i OF arr THEN  
                        BREAK  
                    END IF  
                    SET max_jumps TO the GREATER VALUE BETWEEN max_jumps AND one PLUS CALL dp WITH PARAMETERS i PLUS x  
                END IF  
            END FOR  
            FOR x FROM one TO d  
                IF i MINUS x GREATER THAN OR EQUAL TO zero THEN  
                    IF element at position i MINUS x OF arr GREATER THAN OR EQUAL TO element at position i OF arr THEN  
                        BREAK  
                    END IF  
                    SET max_jumps TO the GREATER VALUE BETWEEN max_jumps AND one PLUS CALL dp WITH PARAMETERS i MINUS x  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        SET result TO the GREATEST VALUE FROM CALL dp WITH PARAMETERS i FOR i FROM zero TO THE LENGTH OF arr MINUS one  
        RETURN result  
    END FUNCTION  
END CLASS
