CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO CALL split_path WITH PARAMETERS path  
        SET stack TO CALL create_empty_list WITH NO PARAMETERS  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS single period symbol  
                CONTINUE TO the NEXT iteration  
            ELSE IF component EQUALS double period symbol  
                IF the LENGTH OF stack GREATER THAN zero  
                    CALL remove_last_element FROM stack  
                END IF  
            ELSE  
                CALL append_element COMPONENT TO stack  
            END IF  
        END FOR  
        SET simplified_path TO the concatenation of a forward slash symbol AND the concatenation of stack elements joined BY a forward slash symbol  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
