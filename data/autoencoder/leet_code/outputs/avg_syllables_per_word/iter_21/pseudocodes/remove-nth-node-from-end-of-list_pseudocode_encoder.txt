CLASS ListNode  
    FUNCTION __init__(val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode CREATED WITH THE ELEMENT at position zero OF values  
    SET p TO head  
    FOR each val IN values STARTING FROM position one TO THE END  
        SET node TO A NEW ListNode CREATED WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN (THE VALUE p1.val EQUALS THE VALUE p2.val) AND CALL is_same_list WITH PARAMETERS p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd(head, n)  
        SET dummy TO A NEW ListNode CREATED WITH zero  
        SET dummy.next TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one  
            SET first TO first.next  
        END FOR  
        WHILE first IS NOT none  
            SET first TO first.next  
            SET second TO second.next  
        END WHILE  
        SET second.next TO THE ELEMENT at position next OF THE ELEMENT at position next OF second  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
