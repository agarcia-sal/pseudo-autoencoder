CLASS Solution  
    FUNCTION kthSmallestProduct(nums1, nums2, k)  
        FUNCTION count(p)  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each x IN nums1  
                IF x GREATER THAN zero  
                    INCREMENT cnt BY the POSITION TO THE RIGHT where p DIVIDED BY x WOULD BE INSERTED IN nums2 TO KEEP ORDER  
                ELSE IF x LESS THAN zero  
                    INCREMENT cnt BY n MINUS the POSITION TO THE LEFT where p DIVIDED BY x WOULD BE INSERTED IN nums2 TO KEEP ORDER  
                ELSE  
                    INCREMENT cnt BY n MULTIPLIED BY the BOOLEAN VALUE OF p GREATER THAN OR EQUAL TO zero  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  

        SET mx TO the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums1 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums1 MULTIPLIED BY the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums2 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums2  
        RETURN the POSITION TO THE LEFT where k WOULD BE INSERTED IN the RANGE FROM negative mx TO mx INCLUSIVE USING count AS THE KEY MINUS mx  
    END FUNCTION  
END CLASS
