CLASS Solution  
    FUNCTION findMaximizedCapital(k, w, profits, capital)  
        SET projects TO the LIST OF PAIRS FOR each element at position index OF capital AND element at position index OF profits  
        SORT projects BY the first element OF each pair in ascending order  
        
        SET max_heap TO an EMPTY list  
        
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        
        FOR iteration FROM one TO k  
            WHILE i LESS THAN n AND the first element at position i OF projects LESS THAN OR EQUAL TO w  
                CALL heap push operation TO ADD the NEGATIVE of the second element at position i OF projects TO max_heap  
                INCREMENT i BY one  
            END WHILE  
            
            IF max_heap IS EMPTY  
                BREAK the loop  
            END IF  
            
            SET w TO w PLUS the NEGATIVE of the result of the heap pop operation FROM max_heap  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
