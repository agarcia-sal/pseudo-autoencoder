CLASS Solution  
    FUNCTION numberOfArrays(s, k)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        CALL create_list_of_zeros WITH PARAMETERS n PLUS one AND ASSIGN TO dp  
        SET element at position zero OF dp TO one  
        
        FOR i FROM one TO n  
            FOR j FROM zero TO i MINUS one  
                IF element at position j OF s DOES NOT EQUAL zero AND  
                   CONVERT substring from position j TO i MINUS one OF s TO integer IS GREATER THAN OR EQUAL TO one AND  
                   CONVERT substring from position j TO i MINUS one OF s TO integer IS LESS THAN OR EQUAL TO k THEN  
                    SET element at position i OF dp TO (element at position i OF dp PLUS element at position j OF dp) MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n OF dp  
    END FUNCTION  
END CLASS
