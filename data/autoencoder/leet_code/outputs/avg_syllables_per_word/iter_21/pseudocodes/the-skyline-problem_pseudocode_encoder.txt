CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO empty list  
        FOR each element consisting of left right height IN buildings  
            APPEND a tuple consisting of left NEGATIVE height right TO events  
            APPEND a tuple consisting of right zero zero TO events  
        END FOR  
        CALL the sort method on events  
        SET result TO a list consisting of a list containing zero zero  
        SET max_heap TO a list consisting of a tuple containing zero infinity  
        FOR each element consisting of x negH r IN events  
            WHILE the second element at position zero OF max_heap IS LESS THAN OR EQUAL TO x  
                CALL the heap pop operation ON max_heap  
            END WHILE  
            IF negH IS NOT EQUAL TO zero  
                CALL the heap push operation ON max_heap WITH PARAMETERS negH r  
            END IF  
            IF the second element of the last element of result IS NOT EQUAL TO NEGATIVE the first element at position zero OF max_heap  
                APPEND a list consisting of x NEGATIVE the first element at position zero OF max_heap TO result  
            END IF  
        END FOR  
        IF the first element of result IS EQUAL TO a list consisting of zero zero  
            REMOVE the first element FROM result  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
