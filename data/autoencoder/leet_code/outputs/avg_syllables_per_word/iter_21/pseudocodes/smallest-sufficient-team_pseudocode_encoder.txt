CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        SET d TO CALL build_skill_index_map WITH PARAMETERS req_skills  
        SET m TO the LENGTH OF req_skills  
        SET n TO the LENGTH OF people  
        SET p TO a list of size n FILLED WITH zero  
        FOR each index i FROM zero TO n MINUS one  
            SET ss TO element at position i OF people  
            FOR each skill s IN ss  
                SET p at position i TO the BITWISE OR OF p at position i AND ONE LEFT SHIFTED BY value at key s OF d  
            END FOR  
        END FOR  
        SET f TO a list of size TWO RAISED TO THE POWER OF m FILLED WITH infinity  
        SET g TO a list of size TWO RAISED TO THE POWER OF m FILLED WITH zero  
        SET h TO a list of size TWO RAISED TO THE POWER OF m FILLED WITH zero  
        SET f at position zero TO zero  
        FOR each integer i FROM zero TO TWO RAISED TO THE POWER OF m MINUS one  
            IF f at position i EQUALS infinity  
                CONTINUE TO the next iteration  
            END IF  
            FOR each integer j FROM zero TO n MINUS one  
                IF f at position i PLUS one IS LESS THAN f at position BITWISE OR OF i AND p at position j  
                    SET f at position BITWISE OR OF i AND p at position j TO f at position i PLUS one  
                    SET g at position BITWISE OR OF i AND p at position j TO j  
                    SET h at position BITWISE OR OF i AND p at position j TO i  
                END IF  
            END FOR  
        END FOR  
        SET i TO TWO RAISED TO THE POWER OF m MINUS one  
        SET ans TO an empty list  
        WHILE i IS NOT zero  
            APPEND element at position i OF g TO ans  
            SET i TO element at position i OF h  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
