CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
        
        FOR each num IN nums  
            SET depth TO num DIVIDED BY one hundred  
            SET pos TO the result of num DIVIDED BY ten THEN MODULO by ten  
            SET value TO num MODULO by ten  
            SET the element at key composed of depth AND pos OF tree TO value  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the element at key composed of depth AND pos OF tree OR None IF NOT PRESENT  
            IF current_value EQUALS None  
                RETURN zero  
            END IF  
            
            SET new_path_sum TO path_sum PLUS current_value  
            
            SET left_child TO a tuple of depth PLUS one AND two MULTIPLIED BY pos MINUS one  
            SET right_child TO a tuple of depth PLUS one AND two MULTIPLIED BY pos  
            
            IF left_child NOT IN tree AND right_child NOT IN tree  
                RETURN new_path_sum  
            END IF  
            
            SET left_sum TO CALL dfs WITH PARAMETERS element at position zero OF left_child, element at position one OF left_child, new_path_sum  
            SET right_sum TO CALL dfs WITH PARAMETERS element at position zero OF right_child, element at position one OF right_child, new_path_sum  
            
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN CALL dfs WITH PARAMETERS one, one, zero  
    END FUNCTION  
END CLASS
