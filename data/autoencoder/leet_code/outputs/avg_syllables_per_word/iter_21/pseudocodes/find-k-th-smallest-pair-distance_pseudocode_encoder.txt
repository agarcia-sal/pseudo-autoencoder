CLASS Solution  
    FUNCTION smallestDistancePair(nums, k)  
        CALL nums.sort WITH PARAMETERS  
        SET n TO the LENGTH OF nums  
        
        FUNCTION count(mid)  
            SET count TO zero  
            SET left TO zero  
            FOR right FROM zero TO n MINUS one  
                WHILE the VALUE AT POSITION right OF nums MINUS the VALUE AT POSITION left OF nums GREATER THAN mid HOLDS  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT count BY right MINUS left  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the VALUE AT POSITION n MINUS one OF nums MINUS the VALUE AT POSITION zero OF nums  
        WHILE left LESS THAN right HOLDS  
            SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
            IF CALL count WITH PARAMETER mid LESS THAN k THEN  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
