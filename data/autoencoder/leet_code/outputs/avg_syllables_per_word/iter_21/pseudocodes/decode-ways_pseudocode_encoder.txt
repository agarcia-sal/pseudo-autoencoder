CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s IS empty OR the element at position zero OF s EQUALS character zero  
            RETURN zero  
        END IF  
        
        CALL a FUNCTION TO INITIALIZE dp AS a list OF zeros WITH the LENGTH OF s PLUS one  
        SET the element at position zero OF dp TO one  
        SET the element at position one OF dp TO one  
        
        FOR index FROM two TO the LENGTH OF s  
            IF the element at position index MINUS one OF s NOT EQUALS character zero  
                INCREMENT the element at position index OF dp BY the element at position index MINUS one OF dp  
            END IF  
            
            CALL a FUNCTION TO CONVERT the substring from position index MINUS two TO index OF s TO an integer AND ASSIGN to two_digit  
            IF two_digit GREATER THAN OR EQUAL TO ten AND two_digit LESS THAN OR EQUAL TO twenty six  
                INCREMENT the element at position index OF dp BY the element at position index MINUS two OF dp  
            END IF  
        END FOR  
        
        RETURN the element at the last position OF dp  
    END FUNCTION  
END CLASS
