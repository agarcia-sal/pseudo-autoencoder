CLASS Solution  
    FUNCTION maxProduct(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN zero  
        END IF  
        SET max_product TO element at position zero OF nums  
        SET min_product TO element at position zero OF nums  
        SET result TO element at position zero OF nums  
        FOR each num IN the SUBSEQUENCE of nums FROM position one TO the END  
            IF num IS LESS THAN zero THEN  
                SET temporary_variable TO max_product  
                SET max_product TO min_product  
                SET min_product TO temporary_variable  
            END IF  
            IF num IS GREATER THAN max_product MULTIPLIED BY num THEN  
                SET max_product TO num  
            ELSE  
                SET max_product TO max_product MULTIPLIED BY num  
            END IF  
            IF num IS LESS THAN min_product MULTIPLIED BY num THEN  
                SET min_product TO num  
            ELSE  
                SET min_product TO min_product MULTIPLIED BY num  
            END IF  
            IF result IS LESS THAN max_product THEN  
                SET result TO max_product  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
