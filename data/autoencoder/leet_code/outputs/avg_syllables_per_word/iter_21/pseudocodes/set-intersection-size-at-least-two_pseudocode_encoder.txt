CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        CALL intervals SORTED BY the second element OF each sublist IN ASCENDING ORDER AND THEN BY the negation OF the first element OF each sublist IN DESCENDING ORDER  
  
        SET first TO the second element of the element at position zero OF intervals MINUS one  
        SET second TO the second element of the element at position zero OF intervals  
        SET count TO two  
  
        FOR each element after the element at position zero IN intervals  
            SET start TO the first element OF the current element  
            SET end TO the second element OF the current element  
  
            IF start GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
  
        RETURN count  
    END FUNCTION  
END CLASS
