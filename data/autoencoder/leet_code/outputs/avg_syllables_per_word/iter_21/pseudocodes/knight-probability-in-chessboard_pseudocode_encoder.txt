CLASS Solution  
    FUNCTION knightProbability(n, k, row, column)  
        SET moves TO list OF pairs: minus two PLUS minus one, minus two PLUS one, minus one PLUS minus two, minus one PLUS two, one PLUS minus two, one PLUS two, two PLUS minus one, two PLUS one  
        
        SET dp TO two dimensional list OF size n BY n INITIALIZED WITH zeroes  
        SET element at position row OF dp AT column TO one  
        
        FOR each move_count FROM one TO k  
            SET new_dp TO two dimensional list OF size n BY n INITIALIZED WITH zeroes  
            
            FOR each r FROM zero TO n MINUS one  
                FOR each c FROM zero TO n MINUS one  
                    IF element at position r OF dp AT c EQUALS zero  
                        CONTINUE TO next iteration  
                    END IF  
                    FOR each pair dr PLUS dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n  
                            INCREMENT element at position nr OF new_dp AT nc BY element at position r OF dp AT c DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            
            SET dp TO new_dp  
        END FOR  
        
        SET total_probability TO zero  
        FOR each row_list IN dp  
            FOR each probability_value IN row_list  
                INCREMENT total_probability BY probability_value  
            END FOR  
        END FOR  
        
        RETURN total_probability  
    END FUNCTION  
END CLASS
