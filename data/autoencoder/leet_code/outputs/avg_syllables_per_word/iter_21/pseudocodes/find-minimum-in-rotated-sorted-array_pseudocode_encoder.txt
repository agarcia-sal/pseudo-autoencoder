CLASS Solution  
    FUNCTION findMin(nums)  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        IF element at position left OF nums IS LESS THAN element at position right OF nums  
            RETURN element at position left OF nums  
        END IF  
        WHILE left IS LESS THAN right  
            SET mid TO the INTEGER DIVISION OF the SUM OF left AND right BY two  
            IF element at position mid OF nums IS GREATER THAN element at position right OF nums  
                SET left TO mid INCREMENTED BY one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN element at position left OF nums  
    END FUNCTION  
END CLASS
