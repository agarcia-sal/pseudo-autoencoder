CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            CALL split_time WITH PARAMETERS time AND colon TO OBTAIN hours_string AND minutes_string  
            CALL convert_string_to_integer WITH PARAMETERS hours_string TO OBTAIN hours  
            CALL convert_string_to_integer WITH PARAMETERS minutes_string TO OBTAIN minutes  
            SET total_minutes TO hours MULTIPLIED BY sixty PLUS minutes  
            RETURN total_minutes  
        END FUNCTION  

        CALL apply_function_to_each_element WITH PARAMETERS to_minutes AND timePoints TO OBTAIN minutes_list  
        CALL sort_list WITH PARAMETERS minutes_list  
        SET min_diff TO infinity_value  

        FOR index FROM one TO LENGTH OF minutes_list MINUS one  
            SET current_difference TO element at position index OF minutes_list MINUS element at position index MINUS one OF minutes_list  
            IF current_difference LESS THAN min_diff  
                SET min_diff TO current_difference  
            END IF  
        END FOR  

        SET circular_difference TO (one thousand four hundred forty MINUS element at last position OF minutes_list) PLUS element at first position OF minutes_list  
        IF circular_difference LESS THAN min_diff  
            SET min_diff TO circular_difference  
        END IF  

        RETURN min_diff  
    END FUNCTION  
END CLASS
