CLASS Solution  
    FUNCTION countBattleships(board)  
        SET count TO zero  
        FOR i FROM zero TO the LENGTH OF board MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero OF board MINUS one  
                IF element at position i OF board EQUALS element at position j OF element at position i OF board EQUALS the character X THEN  
                    IF (i EQUALS zero OR element at position i MINUS one OF board AT position j EQUALS the character representing a dot) AND (j EQUALS zero OR element at position i OF board AT position j MINUS one EQUALS the character representing a dot) THEN  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
