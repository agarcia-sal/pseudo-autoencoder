CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        CALL zip WITH PARAMETERS plantTime AND growTime AND ASSIGN reference TO paired_tasks  
        CALL sorted WITH PARAMETERS paired_tasks AND A FUNCTION THAT TAKES x AND RETURNS NEGATIVE THE element at position one OF x AND ASSIGN reference TO tasks  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair IN tasks  
            SET plant TO element at position zero OF pair  
            SET grow TO element at position one OF pair  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF bloom_day GREATER THAN earliest_bloom THEN  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
