CLASS Solution  
    FUNCTION poorPigs(buckets, minutesToDie, minutesToTest)  
        SET states TO the INTEGER DIVISION OF minutesToTest BY minutesToDie PLUS one  
        SET logarithm_of_buckets TO the MATHEMATICAL LOGARITHM OF buckets  
        SET logarithm_of_states TO the MATHEMATICAL LOGARITHM OF states  
        SET pigs TO the SMALLEST INTEGER GREATER THAN OR EQUAL TO the RESULT OF dividing logarithm_of_buckets BY logarithm_of_states  
        RETURN pigs  
    END FUNCTION  
END CLASS
