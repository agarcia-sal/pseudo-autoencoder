CLASS Solution  
    FUNCTION maxPoints(points)  
        FUNCTION slope(p1, p2)  
            IF element at position zero OF p1 EQUALS element at position zero OF p2  
                RETURN positive infinity  
            END IF  
            SET difference_of_y TO element at position one OF p1 MINUS element at position one OF p2  
            SET difference_of_x TO element at position zero OF p1 MINUS element at position zero OF p2  
            SET result TO difference_of_y DIVIDED BY difference_of_x  
            RETURN result  
        END FUNCTION  
  
        SET max_points TO zero  
  
        FOR i FROM zero TO LENGTH OF points MINUS one  
            SET slopes TO a new mapping with default integer values  
            FOR j FROM zero TO LENGTH OF points MINUS one  
                IF i NOT EQUALS j  
                    SET s TO CALL slope WITH PARAMETERS element at position i OF points AND element at position j OF points  
                    INCREMENT the value associated to s IN slopes BY one  
                    IF the value associated to s IN slopes GREATER THAN max_points  
                        SET max_points TO the value associated to s IN slopes  
                    END IF  
                END IF  
            END FOR  
        END FOR  
  
        IF LENGTH OF points GREATER THAN zero  
            RETURN max_points PLUS one  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
