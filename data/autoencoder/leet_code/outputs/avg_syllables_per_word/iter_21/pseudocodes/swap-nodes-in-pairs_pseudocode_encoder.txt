CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode WITH val EQUALS element at position zero OF values  
    SET p TO head  
    FOR each val IN elements from position one TO the end OF values  
        SET node TO A NEW ListNode WITH val EQUALS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the value at p1 IS EQUALS TO the value at p2 AND CALL is_same_list WITH PARAMETERS the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs(self, head)  
        SET dummy TO A NEW ListNode WITH val EQUALS zero  
        SET dummy.next TO head  
        SET current TO dummy  
        
        WHILE the next element of current IS NOT none AND the next element of the next element of current IS NOT none HOLDS  
            SET first TO the next element of current  
            SET second TO the next element of the next element of current  
            
            SET the next element of current TO second  
            SET the next element of first TO the next element of second  
            SET the next element of second TO first  
            
            SET current TO the next element of the next element of current  
        END WHILE  
        
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
