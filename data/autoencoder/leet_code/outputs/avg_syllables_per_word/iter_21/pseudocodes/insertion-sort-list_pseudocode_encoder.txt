CLASS ListNode  
    FUNCTION __init__(val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO CALL ListNode WITH PARAMETERS element at position zero OF values  
    SET p TO head  
    FOR each val IN elements from position one TO the end OF values  
        SET node TO CALL ListNode WITH PARAMETERS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN (p1.val EQUALS p2.val) AND CALL is_same_list WITH PARAMETERS p1.next, p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(head)  
        IF head IS None OR head.next IS None  
            RETURN head  
        END IF  
        
        SET dummy TO CALL ListNode WITH PARAMETERS minus five thousand and one  
        SET dummy.next TO head  
        SET current TO head.next  
        SET head.next TO None  
        
        WHILE current IS NOT None  
            SET next_node TO current.next  
            
            SET prev TO dummy  
            WHILE prev.next IS NOT None AND (element at position val OF prev.next) LESS THAN (element at position val OF current)  
                SET prev TO prev.next  
            END WHILE  
            
            SET current.next TO prev.next  
            SET prev.next TO current  
            
            SET current TO next_node  
        END WHILE  
        
        RETURN dummy.next  
    END FUNCTION  
END CLASS
