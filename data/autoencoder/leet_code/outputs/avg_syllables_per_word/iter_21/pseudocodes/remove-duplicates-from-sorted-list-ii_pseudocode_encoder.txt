CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node WITH PARAMETERS values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH element at position zero OF values  
    SET p TO head  
    FOR each val IN elements from position one TO THE END OF values  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    IF p1.val EQUALS p2.val AND CALL is_same_list WITH PARAMETERS p1.next AND p2.next  
        RETURN true  
    ELSE  
        RETURN false  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION deleteDuplicates WITH PARAMETERS head  
        SET dummy TO A NEW ListNode INITIALIZED WITH zero AND head  
        SET prev TO dummy  
        WHILE head HOLDS  
            IF head.next HOLDS AND head.val EQUALS element at position zero OF A LIST CONTAINING head.next.val  
                WHILE head.next HOLDS AND head.val EQUALS element at position zero OF A LIST CONTAINING head.next.val  
                    SET head TO head.next  
                END WHILE  
                SET prev.next TO head.next  
            ELSE  
                SET prev TO prev.next  
            END IF  
            SET head TO head.next  
        END WHILE  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
