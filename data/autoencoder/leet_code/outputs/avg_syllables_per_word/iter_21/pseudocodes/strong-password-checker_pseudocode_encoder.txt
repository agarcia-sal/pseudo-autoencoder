CLASS Solution  
    FUNCTION strongPasswordChecker(password)  
        SET n TO the LENGTH OF password  
        
        SET has_lower TO FALSE  
        SET has_upper TO FALSE  
        SET has_digit TO FALSE  
        
        FOR each character IN password  
            IF character IS A LOWERCASE LETTER  
                SET has_lower TO TRUE  
            END IF  
            IF character IS AN UPPERCASE LETTER  
                SET has_upper TO TRUE  
            END IF  
            IF character IS A DIGIT  
                SET has_digit TO TRUE  
            END IF  
        END FOR  
        
        SET missing_types TO THREE MINUS the SUM OF has_lower PLUS has_upper PLUS has_digit  
        
        SET replace TO ZERO  
        SET one TO ZERO  
        SET two TO ZERO  
        SET i TO TWO  
        WHILE i IS LESS THAN n  
            IF the character at position i OF password EQUALS the character at position i MINUS ONE OF password AND the character at position i OF password EQUALS the character at position i MINUS TWO OF password  
                SET length TO TWO  
                WHILE i IS LESS THAN n AND the character at position i OF password EQUALS the character at position i MINUS ONE OF password  
                    INCREMENT length BY ONE  
                    INCREMENT i BY ONE  
                END WHILE  
                INCREMENT replace BY the INTEGER DIVISION OF length BY THREE  
                IF length MODULO THREE EQUALS ZERO  
                    INCREMENT one BY ONE  
                ELSE IF length MODULO THREE EQUALS ONE  
                    INCREMENT two BY ONE  
                END IF  
            ELSE  
                INCREMENT i BY ONE  
            END IF  
        END WHILE  
        
        IF n IS LESS THAN SIX  
            RETURN the MAXIMUM OF missing_types AND SIX MINUS n  
        ELSE IF n IS LESS THAN OR EQUAL TO TWENTY  
            RETURN the MAXIMUM OF missing_types AND replace  
        ELSE  
            SET delete TO n MINUS TWENTY  
            
            SET replace TO replace MINUS the MINIMUM OF delete AND one  
            SET replace TO replace MINUS the INTEGER DIVISION BY TWO OF the MINIMUM OF THE MAXIMUM OF delete MINUS one AND ZERO AND two MULTIPLIED BY TWO  
            SET replace TO replace MINUS the INTEGER DIVISION BY THREE OF the MAXIMUM OF delete MINUS one MINUS two MULTIPLIED BY TWO AND ZERO  
            
            RETURN delete PLUS the MAXIMUM OF missing_types AND replace  
        END IF  
    END FUNCTION  
END CLASS
