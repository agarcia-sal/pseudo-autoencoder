CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        CALL sorted WITH PARAMETERS a list constructed by the following step:  
            FOR each element interval WITH index i IN intervals  
                CREATE a tuple consisting of element at position zero OF interval AND index i  
            END FOR  
        ASSIGN the resulting sorted list TO start_times  
  
        SET result TO an empty list  
  
        FOR each element ignored WITH index end IN intervals  
            CALL binary search left insertion point bisect left ON start_times WITH PARAMETERS a tuple consisting of end at position zero  
            ASSIGN the returned index TO idx  
  
            IF idx IS LESS THAN the LENGTH OF start_times  
                APPEND element at position one OF element at position idx OF start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
