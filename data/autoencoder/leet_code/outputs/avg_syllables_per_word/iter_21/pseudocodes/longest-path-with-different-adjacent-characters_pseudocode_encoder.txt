CLASS Solution  
    FUNCTION longestPath(parent, s)  
        CALL import_defaultdict WITHOUT PARAMETERS  
        SET tree TO CALL create_defaultdict_of_lists WITHOUT PARAMETERS  
        
        FOR index FROM zero TO the LENGTH OF parent MINUS one  
            SET p TO element at position index OF parent  
            IF p NOT EQUALS minus one  
                CALL append index TO the list at key p IN tree  
            END IF  
        END FOR  
        
        SET self.result TO one  
        
        FUNCTION dfs(node)  
            SET max1 TO zero  
            SET max2 TO zero  
            
            FOR each child IN the list at key node IN tree  
                SET child_length TO CALL dfs WITH PARAMETERS child  
                IF character at position child OF s NOT EQUALS character at position node OF s  
                    IF child_length GREATER THAN max1  
                        SET max2 TO max1  
                        SET max1 TO child_length  
                    ELSE IF child_length GREATER THAN max2  
                        SET max2 TO child_length  
                    END IF  
                END IF  
            END FOR  
            
            SET self.result TO the GREATER OF self.result AND max1 PLUS max2 PLUS one  
            RETURN max1 PLUS one  
        END FUNCTION  
        
        CALL dfs WITH PARAMETERS zero  
        RETURN self.result  
    END FUNCTION  
END CLASS
