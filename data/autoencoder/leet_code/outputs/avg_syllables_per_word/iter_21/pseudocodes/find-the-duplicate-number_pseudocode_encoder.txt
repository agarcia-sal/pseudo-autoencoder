CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count TO zero  
            FOR each v IN nums  
                IF v LESS THAN OR EQUAL TO x  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            RETURN count GREATER THAN x  
        END FUNCTION  
        RETURN CALL bisect_left WITH PARAMETERS range FROM zero TO the LENGTH OF nums AND value True AND key f  
    END FUNCTION  
END CLASS
