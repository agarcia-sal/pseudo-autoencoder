CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO CALL TreeNode WITH PARAMETERS element at position zero OF values  
    SET i TO one  
    SET queue TO CALL deque WITH PARAMETERS no parameters  
    CALL APPEND ELEMENT root TO queue  
    WHILE queue HOLDS elements  
        SET node TO CALL popleft ON queue  
        IF i LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            CALL APPEND ELEMENT node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            CALL APPEND ELEMENT node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left AND q.left AND CALL is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION convertBST(self, root)  
        SET self.sum TO zero  
        FUNCTION traverse(node)  
            IF node IS None  
                RETURN  
            END IF  
            CALL traverse WITH PARAMETERS node.right  
            INCREMENT self.sum BY node.val  
            SET node.val TO self.sum  
            CALL traverse WITH PARAMETERS node.left  
        END FUNCTION  
        CALL traverse WITH PARAMETERS root  
        RETURN root  
    END FUNCTION  
END CLASS
