CLASS Solution  
    FUNCTION findContestMatch(n)  
        CALL initialize_teams WITH PARAMETERS n AND STORE RESULT IN teams  
        WHILE n IS GREATER THAN one  
            CALL pair_teams WITH PARAMETERS teams AND n AND STORE RESULT IN teams  
            SET n TO n DIVIDED BY two  
        END WHILE  
        RETURN element at position zero OF teams  
    END FUNCTION  

    FUNCTION initialize_teams(n)  
        CREATE empty list ASSIGNED TO teams  
        FOR index FROM one TO n  
            APPEND string REPRESENTATION OF index TO teams  
        END FOR  
        RETURN teams  
    END FUNCTION  

    FUNCTION pair_teams(teams, n)  
        CREATE empty list ASSIGNED TO new_teams  
        FOR index FROM zero TO n DIVIDED BY two MINUS one  
            SET paired_team TO STRING CONCATENATION OF LEFT PARENTHESIS FOLLOWED BY element at position index OF teams FOLLOWED BY COMMA FOLLOWED BY element at position n MINUS one MINUS index OF teams FOLLOWED BY RIGHT PARENTHESIS  
            APPEND paired_team TO new_teams  
        END FOR  
        RETURN new_teams  
    END FUNCTION  
END CLASS
