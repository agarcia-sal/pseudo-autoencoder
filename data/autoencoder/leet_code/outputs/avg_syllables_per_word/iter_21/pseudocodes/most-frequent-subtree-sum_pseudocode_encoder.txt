CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH val EQUALS element at position zero OF values  
    SET i TO one  
    CALL queue TO NEW double-ended queue  
    CALL append root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE the element from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO NEW TreeNode WITH val EQUALS element at position i OF values  
            CALL append node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO NEW TreeNode WITH val EQUALS element at position i OF values  
            CALL append node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left AND q.left AND is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(root)  
        FUNCTION subtree_sum(node)  
            IF node IS None  
                RETURN zero  
            END IF  
            SET current_sum TO node.val PLUS subtree_sum WITH PARAMETER node.left PLUS subtree_sum WITH PARAMETER node.right  
            INCREMENT count AT current_sum BY one  
            RETURN current_sum  
        END FUNCTION  
        
        SET count TO a new empty mapping WITH integer default values  
        CALL subtree_sum WITH PARAMETER root  
        
        SET max_freq TO the maximum value AMONG all values in count WITH default zero  
        
        SET result TO an empty list  
        FOR each key, freq IN all items OF count  
            IF freq EQUALS max_freq  
                APPEND key TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
