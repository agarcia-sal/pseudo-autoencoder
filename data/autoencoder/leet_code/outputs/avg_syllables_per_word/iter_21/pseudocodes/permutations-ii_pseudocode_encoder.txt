CLASS Solution  
    FUNCTION permuteUnique(nums)  
        FUNCTION backtrack(path, used, res)  
            IF the LENGTH OF path EQUALS the LENGTH OF nums THEN  
                APPEND a COPY OF path TO res  
                RETURN  
            END IF  
            FOR i FROM zero TO the LENGTH OF nums MINUS one DO  
                IF element at position i OF used IS True OR  
                   (i GREATER THAN zero AND element at position i OF nums EQUALS element at position i MINUS one OF nums AND element at position i MINUS one OF used IS False) THEN  
                    CONTINUE TO the NEXT iteration  
                END IF  
                SET element at position i OF used TO True  
                APPEND element at position i OF nums TO path  
                CALL backtrack WITH PARAMETERS path, used, res  
                REMOVE the last element FROM path  
                SET element at position i OF used TO False  
            END FOR  
        END FUNCTION  
        CALL the sort METHOD ON nums TO ARRANGE elements in NON-DECREASING ORDER  
        SET res TO an empty list  
        SET used TO a list OF False WITH the LENGTH OF nums  
        CALL backtrack WITH PARAMETERS an empty list, used, res  
        RETURN res  
    END FUNCTION  
END CLASS
