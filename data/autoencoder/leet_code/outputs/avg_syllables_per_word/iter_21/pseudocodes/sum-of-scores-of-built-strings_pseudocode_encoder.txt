CLASS Solution  
    FUNCTION sumScores(s)  
        SET n TO the LENGTH OF s  
        SET lcp TO a LIST OF zero REPEATED n TIMES  

        SET l TO zero  
        SET r TO zero  
        FOR index i FROM one TO n MINUS one  
            IF i LESS THAN OR EQUAL TO r  
                SET lcp AT position i TO the MINIMUM BETWEEN r MINUS i PLUS one AND lcp AT position i MINUS l  
            END IF  

            WHILE i PLUS lcp AT position i LESS THAN n AND the CHARACTER AT position lcp AT position i OF s EQUALS the CHARACTER AT position i PLUS lcp AT position i OF s  
                INCREMENT lcp AT position i BY one  
            END WHILE  

            IF i PLUS lcp AT position i MINUS one GREATER THAN r  
                SET l TO i  
                SET r TO i PLUS lcp AT position i MINUS one  
            END IF  
        END FOR  

        SET total_sum TO the SUM OF lcp PLUS n  
        RETURN total_sum  
    END FUNCTION  
END CLASS
