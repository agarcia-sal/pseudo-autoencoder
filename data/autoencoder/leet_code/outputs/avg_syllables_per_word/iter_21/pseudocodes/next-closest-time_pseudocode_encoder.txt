CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters FROM time WITH all occurrences of colon REMOVED  
        SET hour AND minute TO the result of converting each element of the list obtained by splitting time by colon TO integer values  
        
        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each character h1 IN digits  
                FOR each character h2 IN digits  
                    FOR each character m1 IN digits  
                        FOR each character m2 IN digits  
                            SET hour_value TO the integer value obtained by concatenating h1 AND h2  
                            SET minute_value TO the integer value obtained by concatenating m1 AND m2  
                            IF hour_value LESS THAN twenty four AND minute_value LESS THAN sixty  
                                APPEND the tuple containing hour_value AND minute_value TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN times SORTED in ascending order  
        END FUNCTION  
        
        SET all_times TO the result of CALLING generate_times WITH digits AS parameter  
        SET current_time_index TO the position of the tuple containing hour AND minute IN all_times  
        
        IF current_time_index PLUS one LESS THAN the LENGTH OF all_times  
            SET next_hour AND next_minute TO the element at position current_time_index PLUS one OF all_times  
        ELSE  
            SET next_hour AND next_minute TO the element at position zero OF all_times  
        END IF  
        
        RETURN the string formed by formatting next_hour AS two digit zero padded, A colon character, AND next_minute AS two digit zero padded  
    END FUNCTION  
END CLASS
