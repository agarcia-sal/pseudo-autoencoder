CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO an empty collection mapping characters to indices  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each index end AND corresponding character char IN s  
            IF char EXISTS IN char_index_map AND the value at char_index_map for char IS GREATER THAN OR EQUAL TO start  
                SET start TO the value at char_index_map for char PLUS one  
            END IF  
            SET the value at char_index_map for char TO end  
            SET max_length TO the GREATER OF max_length AND end MINUS start PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
