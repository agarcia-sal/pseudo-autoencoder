CLASS Solution  
    FUNCTION findKthNumber(m, n, k)  
        FUNCTION enough(x)  
            SET count TO zero  
            FOR i FROM one TO m  
                INCREMENT count BY the smaller VALUE BETWEEN the INTEGER DIVISION OF x BY i AND n  
            END FOR  
            IF count IS GREATER THAN OR EQUAL TO k  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET low TO one  
        SET high TO m MULTIPLIED BY n  
        WHILE low IS LESS THAN high  
            SET mid TO the INTEGER DIVISION OF the SUM of low AND high BY two  
            IF enough(mid) IS true  
                SET high TO mid  
            ELSE  
                SET low TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
