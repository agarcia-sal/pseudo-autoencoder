CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO CALL TreeNode WITH PARAMETERS element at position zero OF values  
    SET i TO one  
    SET queue TO CALL deque WITH PARAMETERS empty  
    CALL queue APPEND WITH PARAMETERS root  
    WHILE queue HOLDS elements  
        SET node TO CALL queue Popleft  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.left  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left, q.left AND CALL is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(self, root, k)  
        SET stack TO empty list  
        SET current TO root  
        SET count TO zero  
        WHILE stack IS NOT empty OR current IS NOT none  
            WHILE current IS NOT none  
                CALL stack APPEND WITH PARAMETERS current  
                SET current TO current.left  
            END WHILE  
            SET current TO CALL stack POP  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN current.val  
            END IF  
            SET current TO current.right  
        END WHILE  
    END FUNCTION  
END CLASS
