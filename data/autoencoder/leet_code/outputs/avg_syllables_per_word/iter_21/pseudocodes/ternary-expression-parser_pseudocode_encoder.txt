CLASS Solution  
    FUNCTION parseTernary(expression)  
        FUNCTION evaluate(index)  
            IF index EQUALS the LENGTH OF expression MINUS one OR the character at position index PLUS one OF expression EQUALS the character colon  
                RETURN the character at position index OF expression AND index PLUS two  
            END IF  
          
            CALL evaluate WITH PARAMETERS index PLUS two AND ASSIGN RESULTS TO true_part AND next_index  
            CALL evaluate WITH PARAMETERS next_index AND ASSIGN RESULTS TO false_part AND final_index  
          
            IF the character at position index OF expression EQUALS the character T  
                RETURN true_part AND final_index  
            ELSE  
                RETURN false_part AND final_index  
            END IF  
        END FUNCTION  
      
        CALL evaluate WITH PARAMETERS zero AND ASSIGN RESULTS TO result AND an unused_variable  
        RETURN result  
    END FUNCTION  
END CLASS
