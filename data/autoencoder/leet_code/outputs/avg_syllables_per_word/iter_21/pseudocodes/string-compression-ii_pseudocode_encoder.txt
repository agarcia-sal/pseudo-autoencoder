CLASS Solution  
    FUNCTION getLengthOfOptimalCompression(s, k)  
        FUNCTION dp(i, k_parameter, prev_char, prev_count)  
            IF k_parameter LESS THAN zero  
                RETURN positive infinity  
            END IF  
            IF i EQUALS the LENGTH OF s  
                RETURN zero  
            END IF  
            IF character at position i OF s EQUALS prev_char  
                IF prev_count EQUALS one OR prev_count EQUALS nine OR prev_count EQUALS ninety nine  
                    SET increment TO one  
                ELSE  
                    SET increment TO zero  
                END IF  
                RETURN increment PLUS CALL dp WITH PARAMETERS i PLUS one, k_parameter, prev_char, prev_count PLUS one  
            ELSE  
                SET keep TO one PLUS CALL dp WITH PARAMETERS i PLUS one, k_parameter, character at position i OF s, one  
                SET delete TO CALL dp WITH PARAMETERS i PLUS one, k_parameter MINUS one, prev_char, prev_count  
                RETURN the MINIMUM OF keep AND delete  
            END IF  
        END FUNCTION  
        RETURN CALL dp WITH PARAMETERS zero, k, empty string, zero  
    END FUNCTION  
END CLASS
