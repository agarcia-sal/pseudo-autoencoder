CLASS Solution  
    FUNCTION replaceWords(dictionary, sentence)  
        CALL dictionary.sort WITH PARAMETERS key EQUALS length  
        SET words TO sentence SPLIT INTO WORDS  
        FUNCTION replace(word)  
            FOR each root IN dictionary  
                IF word STARTS WITH root THEN  
                    RETURN root  
                END IF  
            END FOR  
            RETURN word  
        END FUNCTION  
        SET replaced_words TO EMPTY LIST  
        FOR each word IN words  
            APPEND CALL replace WITH PARAMETERS word TO replaced_words  
        END FOR  
        SET result_sentence TO JOIN replaced_words WITH SINGLE SPACE CHARACTER  
        RETURN result_sentence  
    END FUNCTION  
END CLASS
