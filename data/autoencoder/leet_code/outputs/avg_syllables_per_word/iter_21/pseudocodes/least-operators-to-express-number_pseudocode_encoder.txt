CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        FUNCTION dp(target)  
            IF target EQUALS zero  
                RETURN negative one  
            END IF  
            IF target LESS THAN x  
                RETURN the minimum OF the expression two MULTIPLIED BY target MINUS one AND two MULTIPLIED BY x MINUS target  
            END IF  
              
            SET power TO one  
            WHILE x RAISED TO THE power IS LESS THAN OR EQUAL TO target HOLDS  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
              
            SET res TO CALL dp WITH PARAMETERS target MINUS x RAISED TO THE power PLUS power  
              
            IF x RAISED TO THE power PLUS one MINUS target IS LESS THAN target  
                SET res TO the minimum OF res AND CALL dp WITH PARAMETERS x RAISED TO THE power PLUS one MINUS target PLUS power PLUS one  
            END IF  
              
            RETURN res  
        END FUNCTION  
          
        RETURN CALL dp WITH PARAMETERS target  
    END FUNCTION  
END CLASS
