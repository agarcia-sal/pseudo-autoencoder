CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO TreeNode WITH PARAMETERS value at position zero OF values  
    SET i TO one  
    SET queue TO a new deque collection  
    APPEND root TO queue  
    WHILE queue HOLDS elements  
        REMOVE leftmost element FROM queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO TreeNode WITH PARAMETERS element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO TreeNode WITH PARAMETERS element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF value property of p NOT EQUALS value property of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree WITH PARAMETERS left child of p AND left child of q AND the result of calling is_same_tree WITH PARAMETERS right child of p AND right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST(root, val)  
        FUNCTION dfs(root)  
            IF root IS None  
                RETURN TreeNode WITH PARAMETERS val  
            END IF  
            IF value property of root LESS THAN val  
                SET right child property of root TO the result of calling dfs WITH PARAMETERS right child property of root  
            ELSE  
                SET left child property of root TO the result of calling dfs WITH PARAMETERS left child property of root  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN the result of calling dfs WITH PARAMETERS root  
    END FUNCTION  
END CLASS
