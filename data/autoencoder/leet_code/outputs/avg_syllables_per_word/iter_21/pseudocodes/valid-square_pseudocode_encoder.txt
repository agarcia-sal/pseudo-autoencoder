CLASS Solution  
    FUNCTION validSquare(p1, p2, p3, p4)  
        FUNCTION squared_distance(p1, p2)  
            SET difference_in_first_coordinate TO element at position zero OF p1 MINUS element at position zero OF p2  
            SET difference_in_second_coordinate TO element at position one OF p1 MINUS element at position one OF p2  
            SET squared_distance_value TO difference_in_first_coordinate MULTIPLIED BY difference_in_first_coordinate PLUS difference_in_second_coordinate MULTIPLIED BY difference_in_second_coordinate  
            RETURN squared_distance_value  
        END FUNCTION  
        
        SET points TO list containing p1, p2, p3, p4  
        SET distances TO empty set  
        
        FOR i FROM zero TO three  
            FOR j FROM i PLUS one TO three  
                SET dist TO CALL squared_distance WITH PARAMETERS element at position i OF points AND element at position j OF points  
                IF dist EQUALS zero THEN  
                    RETURN False  
                END IF  
                ADD dist TO distances  
            END FOR  
        END FOR  
        
        IF the SIZE OF distances EQUALS two THEN  
            RETURN True  
        ELSE  
            RETURN False  
        END IF  
    END FUNCTION  
END CLASS
