CLASS Solution  
    FUNCTION numDupDigitsAtMostN(n)  
        FUNCTION count_unique_digits(x)  
            SET s TO the string representation of x  
            SET length TO the length of s  
            SET count TO zero  
            FOR i FROM one TO length MINUS one  
                SET count TO count PLUS nine MULTIPLIED BY CALL perm WITH PARAMETERS nine COMMA i MINUS one  
            END FOR  
            SET seen TO an empty set  
            FOR i FROM zero TO length MINUS one  
                SET digit TO the integer value of element at position i OF s  
                IF i EQUALS zero  
                    SET start_digit TO one  
                ELSE  
                    SET start_digit TO zero  
                END IF  
                FOR d FROM start_digit TO digit MINUS one  
                    IF d IS NOT IN seen  
                        SET count TO count PLUS CALL perm WITH PARAMETERS nine MINUS i COMMA length MINUS i MINUS one  
                    END IF  
                END FOR  
                IF digit IS IN seen  
                    BREAK the loop  
                END IF  
                ADD digit TO seen  
            END FOR  
            IF the loop over i COMPLETED WITHOUT BREAKING  
                INCREMENT count BY one  
            END IF  
            RETURN count  
        END FUNCTION  
        SET result TO n MINUS CALL count_unique_digits WITH PARAMETER n  
        RETURN result  
    END FUNCTION  

    FUNCTION perm(m, n)  
        IF n EQUALS zero  
            RETURN one  
        END IF  
        SET result TO one  
        FOR i FROM zero TO n MINUS one  
            SET result TO result MULTIPLIED BY m MINUS i  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
