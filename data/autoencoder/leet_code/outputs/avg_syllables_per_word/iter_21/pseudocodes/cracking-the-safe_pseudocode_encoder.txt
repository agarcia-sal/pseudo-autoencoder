CLASS Solution  
    FUNCTION crackSafe(n, k)  
        SET visited TO an empty set  
        SET sequence TO an empty list  
        
        FUNCTION dfs(node)  
            FOR i FROM zero TO k MINUS one  
                SET edge TO the string node PLUS the string representation of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH PARAMETERS the substring from position one TO end OF edge  
                    APPEND the string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO the string consisting of character zero REPEATED n MINUS one TIMES  
        CALL dfs WITH PARAMETERS start_node  
        
        SET result TO the concatenation of the reversed sequence FOLLOWED BY start_node  
        RETURN result  
    END FUNCTION  
END CLASS
