CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO the result of calling method defaultdict WITH PARAMETERS integer  
        FOR each element i IN nums  
            FOR each element j IN nums  
                SET current_key TO the result of bitwise AND operation BETWEEN i AND j  
                INCREMENT the value at current_key IN pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each element k IN nums  
            FOR each key-value pair pair_and and count IN pair_and_count  
                IF the result of bitwise AND operation BETWEEN pair_and AND k EQUALS zero  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
