CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF state IS a key in memo  
                RETURN the value associated with state in memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring from position index TO index PLUS one OF state EQUALS two plus signs  
                    SET new_state TO the concatenation of the substring from start TO index OF state, two minus signs, and the substring from position index PLUS two TO end OF state  
                    IF NOT calling can_win_helper WITH PARAMETERS new_state  
                        SET the value associated with state in memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value associated with state in memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN calling can_win_helper WITH PARAMETERS currentState  
    END FUNCTION  
END CLASS
