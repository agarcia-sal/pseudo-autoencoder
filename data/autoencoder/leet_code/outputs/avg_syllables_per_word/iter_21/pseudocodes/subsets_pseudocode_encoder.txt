CLASS Solution  
    FUNCTION subsets(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN a list CONTAINING an empty list  
        END IF  
        SET subsets_without_first TO CALL subsets WITH PARAMETERS the list of elements from position one TO the end OF nums  
        SET subsets_with_first TO an empty list  
        FOR each subset IN subsets_without_first  
            SET new_subset TO a list CONTAINING the element at position zero OF nums FOLLOWED BY elements OF subset  
            APPEND new_subset TO subsets_with_first  
        END FOR  
        SET all_subsets TO the concatenation OF subsets_without_first AND subsets_with_first  
        RETURN all_subsets  
    END FUNCTION  
END CLASS
