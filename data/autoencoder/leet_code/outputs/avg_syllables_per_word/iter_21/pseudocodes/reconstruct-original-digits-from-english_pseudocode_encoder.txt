CLASS Solution  
    FUNCTION originalDigits(s)  
        CALL Counter WITH PARAMETERS s AND ASSIGN reference TO count  
        
        INITIALIZE digits AS a list of ten zeros  
        
        SET element at position zero OF digits TO element at key character z OF count  
        SET element at position two OF digits TO element at key character w OF count  
        SET element at position four OF digits TO element at key character u OF count  
        SET element at position six OF digits TO element at key character x OF count  
        SET element at position eight OF digits TO element at key character g OF count  
        
        SET element at position three OF digits TO element at key character h OF count MINUS element at position eight OF digits  
        SET element at position five OF digits TO element at key character f OF count MINUS element at position four OF digits  
        SET element at position seven OF digits TO element at key character s OF count MINUS element at position six OF digits  
        SET element at position nine OF digits TO element at key character i OF count MINUS element at position five OF digits MINUS element at position six OF digits MINUS element at position eight OF digits  
        SET element at position one OF digits TO element at key character o OF count MINUS element at position zero OF digits MINUS element at position two OF digits MINUS element at position four OF digits  
        
        INITIALIZE result AS an empty list  
        FOR index FROM zero TO nine  
            SET repeated_digit_string TO string representation of index REPEATED element at position index OF digits TIMES  
            APPEND repeated_digit_string TO result  
        END FOR  
        
        RETURN concatenation of all elements IN result  
    END FUNCTION  
END CLASS
