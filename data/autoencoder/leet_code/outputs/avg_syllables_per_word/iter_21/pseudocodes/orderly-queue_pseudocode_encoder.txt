CLASS Solution  
    FUNCTION orderlyQueue(s, k)  
        IF k GREATER THAN one  
            RETURN the RESULT OF CALLING a FUNCTION THAT sorts the characters of s IN ascending order AND THEN concatenates them into a single string  
        ELSE  
            SET n TO the LENGTH OF s  
            SET smallest TO s  
            FOR i FROM one TO n MINUS one  
                SET rotated TO the CONCATENATION OF the SUBSTRING FROM position i TO the END OF s AND the SUBSTRING FROM position zero TO i MINUS one OF s  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
