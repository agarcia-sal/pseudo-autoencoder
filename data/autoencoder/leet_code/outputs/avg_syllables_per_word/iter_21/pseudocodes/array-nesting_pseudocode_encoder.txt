CLASS Solution  
    FUNCTION arrayNesting(nums)  
        SET visited TO a list of boolean false values with the length EQUAL TO the LENGTH OF nums  
        SET max_length TO zero  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF NOT the element at position index OF visited EQUALS true  
                SET current_length TO zero  
                SET k TO index  
                WHILE NOT the element at position k OF visited EQUALS true  
                    SET the element at position k OF visited TO true  
                    SET k TO the element at position k OF nums  
                    INCREMENT current_length BY one  
                END WHILE  
                IF max_length LESS THAN current_length  
                    SET max_length TO current_length  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
