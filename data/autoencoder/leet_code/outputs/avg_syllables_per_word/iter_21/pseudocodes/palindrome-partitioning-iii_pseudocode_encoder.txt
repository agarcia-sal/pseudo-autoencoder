CLASS Solution  
    FUNCTION palindromePartition(s, k)  
        FUNCTION min_changes_to_palindrome(sub)  
            SET i TO zero  
            SET j TO the LENGTH OF sub MINUS one  
            SET changes TO zero  
            WHILE i LESS THAN j HOLDS  
                IF element at position i OF sub NOT EQUALS element at position j OF sub  
                    INCREMENT changes BY one  
                END IF  
                INCREMENT i BY one  
                DECREMENT j BY one  
            END WHILE  
            RETURN changes  
        END FUNCTION  

        SET n TO the LENGTH OF s  

        FUNCTION initialize_dp(n, k)  
            CREATE a two-dimensional list dp WITH n PLUS one ROWS AND k PLUS one COLUMNS  
            SET each element OF dp TO positive infinity  
            RETURN dp  
        END FUNCTION  

        SET dp TO CALL initialize_dp WITH PARAMETERS n AND k  
        SET element at position zero AND position zero OF dp TO zero  

        FOR i FROM one TO n  
            FOR j FROM one TO the MINIMUM OF k AND i  
                FOR start FROM zero TO i MINUS one  
                    SET changes TO CALL min_changes_to_palindrome WITH PARAMETER the substring of s FROM start TO i  
                    SET dp element at position i AND j TO the MINIMUM OF dp element at position i AND j AND the SUM OF dp element at position start AND j MINUS one AND changes  
                END FOR  
            END FOR  
        END FOR  

        RETURN dp element at position n AND k  
    END FUNCTION  
END CLASS
