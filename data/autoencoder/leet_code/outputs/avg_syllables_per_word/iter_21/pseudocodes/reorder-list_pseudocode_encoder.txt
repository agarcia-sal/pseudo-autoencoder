CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO CALL ListNode WITH PARAMETERS element at position zero OF values  
    SET p TO head  
    FOR each val IN elements from position one TO end OF values  
        SET node TO CALL ListNode WITH PARAMETERS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN (value of p1.val EQUALS value of p2.val) AND CALL is_same_list WITH PARAMETERS p1.next, p2.next  
END FUNCTION

CLASS Solution  
    FUNCTION reorderList(self, head)  
        IF head EQUALS None OR head.next EQUALS None  
            RETURN  
        END IF  

        SET slow TO head  
        SET fast TO head  
        WHILE fast NOT EQUAL TO None AND fast.next NOT EQUAL TO None  
            SET slow TO slow.next  
            SET fast TO fast.next.next  
        END WHILE  

        SET prev TO None  
        SET curr TO slow  
        WHILE curr NOT EQUAL TO None  
            SET next_temp TO curr.next  
            SET curr.next TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  

        SET first TO head  
        SET second TO prev  
        WHILE second.next NOT EQUAL TO None  
            SET temp1 TO first.next  
            SET temp2 TO second.next  
            SET first.next TO second  
            SET second.next TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
