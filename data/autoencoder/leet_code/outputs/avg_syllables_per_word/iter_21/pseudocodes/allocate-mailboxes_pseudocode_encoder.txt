CLASS Solution  
    FUNCTION minDistance(houses, k)  
        CALL houses SORT IN ASCENDING ORDER  
        SET n TO the LENGTH OF houses  
        
        CALL cost TO THE RESULT OF calling construct_cost_matrix WITH PARAMETERS houses AND n  
        
        CALL dp TO THE RESULT OF calling construct_dp_table WITH PARAMETERS n AND k  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET potential_distance TO the VALUE AT position p OF dp AND position j MINUS one OF dp PLUS the VALUE AT position p OF cost AND position i MINUS one OF cost  
                    IF potential_distance LESS THAN the VALUE AT position i OF dp AND position j OF dp THEN  
                        SET the VALUE AT position i OF dp AND position j OF dp TO potential_distance  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN the VALUE AT position n OF dp AND position k OF dp  
    END FUNCTION  
      
    FUNCTION construct_cost_matrix(houses, n)  
        CALL cost TO AN n BY n TWO-DIMENSIONAL LIST FILLED WITH ZERO  
        FOR i FROM n MINUS one DOWNTO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost AT position i AND position j TO the VALUE AT position i PLUS one AND position j MINUS one OF cost PLUS the VALUE AT position j OF houses MINUS the VALUE AT position i OF houses  
            END FOR  
        END FOR  
        RETURN cost  
    END FUNCTION  
      
    FUNCTION construct_dp_table(n, k)  
        CALL dp TO AN n PLUS one BY k PLUS one TWO-DIMENSIONAL LIST FILLED WITH POSITIVE INFINITY  
        SET the VALUE AT position zero OF dp AND position zero OF dp TO zero  
        RETURN dp  
    END FUNCTION  
END CLASS
