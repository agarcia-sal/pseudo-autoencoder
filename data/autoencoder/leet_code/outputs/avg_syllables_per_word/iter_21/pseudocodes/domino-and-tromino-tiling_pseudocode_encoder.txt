CLASS Solution  
    FUNCTION numTilings(n)  
        SET MOD TO one billion PLUS seven  
        
        SET dp TO a list OF zeros WITH length n PLUS one  
        SET dp1 TO a list OF zeros WITH length n PLUS one  
        
        SET element at position zero OF dp TO one  
        SET element at position one OF dp TO one  
        
        IF n IS GREATER THAN OR EQUAL TO two THEN  
            SET element at position two OF dp TO two  
            SET element at position two OF dp1 TO one  
        END IF  
        
        FOR index FROM three TO n DO  
            SET element at position index OF dp TO the remainder WHEN the sum OF element at position index MINUS one OF dp PLUS element at position index MINUS two OF dp PLUS two MULTIPLIED BY element at position index MINUS one OF dp1 IS DIVIDED BY MOD  
            SET element at position index OF dp1 TO the remainder WHEN the sum OF element at position index MINUS two OF dp PLUS element at position index MINUS one OF dp1 IS DIVIDED BY MOD  
        END FOR  
        
        RETURN element at position n OF dp  
    END FUNCTION  
END CLASS
