CLASS Solution  
    FUNCTION hasPath(maze, start, destination)  
        SET directions TO list containing tuple zero and one tuple one and zero tuple zero and minus one tuple minus one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero OF maze  
        SET visited TO empty set  
        
        FUNCTION dfs(x, y)  
            IF tuple x and y IS IN visited THEN  
                RETURN False  
            END IF  
            IF list containing x and y EQUALS destination THEN  
                RETURN True  
            END IF  
            ADD tuple x and y TO visited  
            
            FOR each tuple dx, dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND element at position nx PLUS dx OF maze AT position ny PLUS dy EQUALS zero DO  
                    SET nx TO nx PLUS dx  
                    SET ny TO ny PLUS dy  
                END WHILE  
                IF CALL dfs WITH PARAMETERS nx, ny THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN CALL dfs WITH PARAMETERS element at position zero OF start, element at position one OF start  
    END FUNCTION  
END CLASS
