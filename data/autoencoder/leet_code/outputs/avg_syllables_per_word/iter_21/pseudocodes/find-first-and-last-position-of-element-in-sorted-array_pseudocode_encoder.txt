CLASS Solution  
    FUNCTION searchRange(nums, target)  
        FUNCTION find_left(nums, target)  
            SET left TO zero  
            SET right TO the LENGTH OF nums MINUS one  
            WHILE left LESS THAN OR EQUAL TO right HOLDS  
                SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
                IF element at position mid OF nums LESS THAN target THEN  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN left  
        END FUNCTION  

        FUNCTION find_right(nums, target)  
            SET left TO zero  
            SET right TO the LENGTH OF nums MINUS one  
            WHILE left LESS THAN OR EQUAL TO right HOLDS  
                SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
                IF element at position mid OF nums LESS THAN OR EQUAL TO target THEN  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN right  
        END FUNCTION  

        SET left_index TO CALL find_left WITH PARAMETERS nums AND target  
        SET right_index TO CALL find_right WITH PARAMETERS nums AND target  

        IF left_index LESS THAN OR EQUAL TO right_index AND zero LESS THAN OR EQUAL TO right_index AND right_index LESS THAN the LENGTH OF nums AND element at position left_index OF nums EQUALS target THEN  
            RETURN list CONTAINING left_index AND right_index  
        ELSE  
            RETURN list CONTAINING minus one AND minus one  
        END IF  
    END FUNCTION  
END CLASS
