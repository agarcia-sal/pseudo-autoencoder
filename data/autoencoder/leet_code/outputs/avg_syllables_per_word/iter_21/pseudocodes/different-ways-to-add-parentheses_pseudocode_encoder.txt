CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO an empty list  
            FOR each element l IN left  
                FOR each element r IN right  
                    IF operator EQUALS the plus symbol  
                        APPEND the summation of l PLUS r TO results  
                    ELSE IF operator EQUALS the minus symbol  
                        APPEND the subtraction of l MINUS r TO results  
                    ELSE IF operator EQUALS the multiplication symbol  
                        APPEND the multiplication of l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        FUNCTION helper(sub_expr)  
            IF sub_expr CONSISTS ONLY OF DIGITS  
                RETURN a list containing the integer conversion of sub_expr  
            END IF  
            
            SET results TO an empty list  
            FOR index FROM zero TO the LENGTH OF sub_expr MINUS one  
                SET character TO the element at position index OF sub_expr  
                IF character IS EQUAL TO the plus symbol OR character IS EQUAL TO the minus symbol OR character IS EQUAL TO the multiplication symbol  
                    SET left_results TO CALL helper WITH PARAMETERS the substring from position zero TO index MINUS one OF sub_expr  
                    SET right_results TO CALL helper WITH PARAMETERS the substring from position index PLUS one TO the end OF sub_expr  
                    CALL compute WITH PARAMETERS left_results COMMA right_results COMMA character AND EXTEND results BY THE RETURNED LIST  
                END IF  
            END FOR  
            
            RETURN results  
        END FUNCTION  
        
        RETURN CALL helper WITH PARAMETERS expression  
    END FUNCTION  
END CLASS
