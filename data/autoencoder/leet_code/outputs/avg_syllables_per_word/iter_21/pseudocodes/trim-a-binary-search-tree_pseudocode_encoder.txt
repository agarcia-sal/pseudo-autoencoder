CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH val EQUALS element at position zero OF values  
    SET i TO one  
    SET queue TO NEW empty double-ended queue  
    CALL APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO NEW TreeNode WITH val EQUALS element at position i OF values  
            CALL APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO NEW TreeNode WITH val EQUALS element at position i OF values  
            CALL APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left, q.left AND CALL is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST(self, root, low, high)  
        IF root IS None  
            RETURN None  
        END IF  
        IF root.val IS LESS THAN low  
            RETURN CALL trimBST WITH PARAMETERS root.right, low, high  
        END IF  
        IF root.val IS GREATER THAN high  
            RETURN CALL trimBST WITH PARAMETERS root.left, low, high  
        END IF  
        SET root.left TO CALL trimBST WITH PARAMETERS root.left, low, high  
        SET root.right TO CALL trimBST WITH PARAMETERS root.right, low, high  
        RETURN root  
    END FUNCTION  
END CLASS
