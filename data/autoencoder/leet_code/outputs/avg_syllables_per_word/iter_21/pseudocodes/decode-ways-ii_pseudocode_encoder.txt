CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven  
        
        SET dp TO LIST OF ZERO WITH SIZE EQUAL TO LENGTH OF s PLUS one  
        SET element at position zero OF dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS the CHARACTER STAR THEN  
                RETURN nine  
            ELSE IF char EQUALS the CHARACTER ZERO THEN  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS the CHARACTER STAR AND char2 EQUALS the CHARACTER STAR THEN  
                RETURN fifteen  
            ELSE IF char1 EQUALS the CHARACTER STAR THEN  
                IF char2 LESS THAN OR EQUAL TO the CHARACTER six THEN  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS the CHARACTER STAR THEN  
                IF char1 EQUALS the CHARACTER one THEN  
                    RETURN nine  
                ELSE IF char1 EQUALS the CHARACTER two THEN  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO integer CONVERTED FROM concatenation OF char1 AND char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six THEN  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
        
        FOR i FROM one TO LENGTH OF s  
            SET element at position i OF dp TO  
                (element at position i OF dp PLUS element at position i MINUS one OF dp MULTIPLIED BY CALL ways_to_decode_single WITH PARAMETERS element at position i MINUS one OF s) MODULO MOD  
            
            IF i GREATER THAN one THEN  
                SET element at position i OF dp TO  
                    (element at position i OF dp PLUS element at position i MINUS two OF dp MULTIPLIED BY CALL ways_to_decode_pair WITH PARAMETERS element at position i MINUS two OF s AND element at position i MINUS one OF s) MODULO MOD  
            END IF  
        END FOR  
        
        RETURN element at position LENGTH OF dp MINUS one OF dp  
    END FUNCTION  
END CLASS
