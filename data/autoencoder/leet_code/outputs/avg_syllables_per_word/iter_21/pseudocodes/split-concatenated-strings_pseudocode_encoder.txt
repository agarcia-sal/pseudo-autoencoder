CLASS Solution
    FUNCTION splitLoopedString(strs)
        FOR index FROM zero TO the LENGTH OF strs MINUS one
            SET reversed_str TO the REVERSE OF element at position index OF strs
            IF reversed_str IS GREATER THAN element at position index OF strs
                SET element at position index OF strs TO reversed_str
            END IF
        END FOR

        SET doubled_strs TO the CONCATENATION OF strs FOLLOWED BY strs

        SET max_string TO the empty string

        FOR index FROM zero TO the LENGTH OF strs MINUS one
            FOR each is_reversed IN the collection containing False AND True
                IF is_reversed IS True
                    SET current TO the REVERSE OF element at position index OF strs
                ELSE
                    SET current TO element at position index OF strs
                END IF

                SET remaining TO the CONCATENATION OF elements from position index PLUS one TO index PLUS the LENGTH OF strs MINUS one OF doubled_strs

                FOR j FROM zero TO the LENGTH OF current
                    SET candidate TO the CONCATENATION OF substring from position j TO the END OF current FOLLOWED BY remaining FOLLOWED BY substring from position zero TO j MINUS one OF current
                    IF candidate IS GREATER THAN max_string
                        SET max_string TO candidate
                    END IF
                END FOR
            END FOR
        END FOR

        RETURN max_string
    END FUNCTION
END CLASS
