CLASS Solution  
    FUNCTION colorTheGrid(m, n)  
        SET MOD TO one billion PLUS seven  
        
        FUNCTION generate_valid_columns(prev_col)  
            SET colors TO list containing one, two, three  
            SET valid_cols TO empty list  
            
            FUNCTION backtrack(col, row)  
                IF row EQUALS m THEN  
                    APPEND tuple converted from col TO valid_cols  
                    RETURN  
                END IF  
                FOR each color IN colors  
                    IF col IS empty OR element at position negative one OF col NOT EQUALS color THEN  
                        IF prev_col IS NOT null AND element at position row OF prev_col EQUALS color THEN  
                            CONTINUE TO next iteration  
                        END IF  
                        APPEND color TO col  
                        CALL backtrack WITH PARAMETERS col AND row PLUS one  
                        REMOVE last element FROM col  
                    END IF  
                END FOR  
            END FUNCTION  
            
            CALL backtrack WITH PARAMETERS empty list AND zero  
            RETURN valid_cols  
        END FUNCTION  
        
        SET all_valid_cols TO CALL generate_valid_columns WITH PARAMETERS null  
        SET num_valid_cols TO the LENGTH OF all_valid_cols  
        
        SET compatible_cols TO dictionary mapping each integer i FROM zero TO num_valid_cols MINUS one TO empty list  
        FOR each integer i FROM zero TO num_valid_cols MINUS one  
            FOR each integer j FROM zero TO num_valid_cols MINUS one  
                SET compatible TO true  
                FOR each integer k FROM zero TO m MINUS one  
                    IF element at position k OF all_valid_cols at position i EQUALS element at position k OF all_valid_cols at position j THEN  
                        SET compatible TO false  
                        BREAK out of the inner loop  
                    END IF  
                END FOR  
                IF compatible THEN  
                    APPEND j TO the list at key i IN compatible_cols  
                END IF  
            END FOR  
        END FOR  
        
        SET dp TO list containing num_valid_cols times the integer one  
        
        FOR each integer _ FROM one TO n MINUS one  
            SET next_dp TO list containing num_valid_cols times the integer zero  
            FOR each integer i FROM zero TO num_valid_cols MINUS one  
                FOR each integer j IN the list at key i IN compatible_cols  
                    SET next_dp at position j TO (element at position j OF next_dp PLUS element at position i OF dp) MODULO MOD  
                END FOR  
            END FOR  
            SET dp TO next_dp  
        END FOR  
        
        SET total_ways TO zero  
        FOR each integer count IN dp  
            INCREMENT total_ways BY count  
        END FOR  
        RETURN total_ways MODULO MOD  
    END FUNCTION  
END CLASS
