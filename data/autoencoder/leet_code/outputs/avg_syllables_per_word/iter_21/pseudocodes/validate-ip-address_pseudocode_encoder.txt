CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS ONLY OF DIGITS  
                SET num TO conversion of block TO integer  
                IF num LESS THAN OR EQUAL TO two hundred fifty five AND conversion of num TO string EQUALS block  
                    RETURN True  
                ELSE  
                    RETURN False  
                END IF  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF length of block IS GREATER THAN OR EQUAL TO one AND length of block IS LESS THAN OR EQUAL TO four  
                TRY  
                    CONVERT block FROM hexadecimal string TO integer  
                    RETURN True  
                CATCH exception OF type ValueError  
                    RETURN False  
                END TRY  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO list of substrings FROM queryIP SPLIT BY the character dot  
        SET parts_ipv6 TO list of substrings FROM queryIP SPLIT BY the character colon  

        IF length of parts_ipv4 EQUALS four AND FOR each part IN parts_ipv4 CALL is_valid_ipv4_block WITH PARAMETERS part IS True FOR ALL  
            RETURN string IPv4  
        ELSE IF length of parts_ipv6 EQUALS eight AND FOR each part IN parts_ipv6 CALL is_valid_ipv6_block WITH PARAMETERS part IS True FOR ALL  
            RETURN string IPv6  
        ELSE  
            RETURN string Neither  
        END IF  
    END FUNCTION  
END CLASS
