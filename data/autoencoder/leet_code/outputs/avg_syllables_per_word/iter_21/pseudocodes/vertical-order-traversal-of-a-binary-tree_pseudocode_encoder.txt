CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH val EQUALS element at position zero OF values  
    SET i TO one  
    SET queue TO NEW empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE element from the front OF queue  
        IF i IS LESS THAN length of values AND element at position i OF values IS NOT None  
            SET node.left TO NEW TreeNode WITH val EQUALS element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN length of values AND element at position i OF values IS NOT None  
            SET node.right TO NEW TreeNode WITH val EQUALS element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION verticalTraversal(self, root)  
        IF root IS None  
            RETURN empty list  
        END IF  
        SET node_dict TO NEW default dictionary with list as default value  
        SET min_col TO zero  
        SET max_col TO zero  

        FUNCTION dfs(node, row, col)  
            IF node IS None  
                RETURN  
            END IF  
            APPEND tuple of row AND node.val TO node_dict at key col  
            SET min_col TO THE MINIMUM OF min_col AND col  
            SET max_col TO THE MAXIMUM OF max_col AND col  
            CALL dfs WITH PARAMETERS node.left, row PLUS one, col MINUS one  
            CALL dfs WITH PARAMETERS node.right, row PLUS one, col PLUS one  
        END FUNCTION  

        CALL dfs WITH PARAMETERS root, zero, zero  

        SET result TO empty list  
        FOR col FROM min_col TO max_col  
            SET sorted_list TO list of values extracted from sorting node_dict at key col BY row value THEN node value  
            APPEND sorted_list TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
