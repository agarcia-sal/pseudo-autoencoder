CLASS Solution  
    FUNCTION networkDelayTime(times, n, k)  
        SET graph TO AN EMPTY MAPPING FROM integers TO EMPTY LISTS  
        FOR each integer i FROM one TO n  
            SET graph AT position i TO AN EMPTY LIST  
        END FOR  
        FOR each element u, v, w IN times  
            APPEND A PAIR OF v AND w TO the LIST AT graph AT position u  
        END FOR  
  
        SET dist TO AN EMPTY MAPPING FROM integers TO INFINITY  
        FOR each integer i FROM one TO n  
            SET dist AT position i TO POSITIVE INFINITY  
        END FOR  
        SET dist AT position k TO zero  
  
        SET min_heap TO A LIST CONTAINING A PAIR OF zero AND k  
  
        WHILE the LENGTH OF min_heap IS GREATER THAN zero HOLDS  
            CALL heapq POP FROM min_heap AND ASSIGN THE RESULT TO current_dist AND node  
  
            IF current_dist IS GREATER THAN dist AT position node  
                CONTINUE TO NEXT ITERATION OF THE LOOP  
            END IF  
  
            FOR each pair neighbor AND weight IN graph AT position node  
                SET distance TO current_dist PLUS weight  
  
                IF distance IS LESS THAN dist AT position neighbor  
                    SET dist AT position neighbor TO distance  
                    CALL heapq PUSH TO min_heap WITH PARAMETERS distance AND neighbor  
                END IF  
            END FOR  
        END WHILE  
  
        SET max_time TO THE MAXIMUM VALUE AMONG ALL VALUES IN dist  
  
        IF max_time IS NOT EQUAL TO POSITIVE INFINITY  
            RETURN max_time  
        ELSE  
            RETURN MINUS one  
        END IF  
    END FUNCTION  
END CLASS
