CLASS Solution  
    FUNCTION numDistinct(s, t)  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
        
        CALL dp TO THE RESULT OF create_2D_array WITH PARAMETERS m PLUS one AND n PLUS one  
        
        FOR i FROM zero TO m  
            SET element at position zero OF element at position i OF dp TO one  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one OF s EQUALS element at position j MINUS one OF t THEN  
                    SET element at position j OF element at position i OF dp TO  
                        element at position j MINUS one OF element at position i MINUS one OF dp PLUS element at position j OF element at position i MINUS one OF dp  
                ELSE  
                    SET element at position j OF element at position i OF dp TO element at position j OF element at position i MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n OF element at position m OF dp  
    END FUNCTION  
      
    FUNCTION create_2D_array(rows, columns)  
        SET result TO a new list  
        FOR i FROM zero TO rows MINUS one  
            SET new_row TO a new list  
            FOR j FROM zero TO columns MINUS one  
                APPEND zero TO new_row  
            END FOR  
            APPEND new_row TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
