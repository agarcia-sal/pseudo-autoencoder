CLASS Solution  
    FUNCTION maxNumber(nums1, nums2, k)  
        FUNCTION maxSingleArray(nums, k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop GREATER THAN zero AND stack IS NOT empty AND element at the last position OF stack LESS THAN num  
                    REMOVE element at the last position FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the SUBLIST of stack FROM position zero TO position k  
        END FUNCTION  

        FUNCTION merge(arr1, arr2)  
            SET result TO an empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR index FROM zero TO total_length MINUS one  
                IF arr1 GREATER THAN arr2  
                    APPEND element at position zero OF arr1 TO result  
                    REMOVE element at position zero FROM arr1  
                ELSE  
                    APPEND element at position zero OF arr2 TO result  
                    REMOVE element at position zero FROM arr2  
                END IF  
            END FOR  
            RETURN result  
        END FUNCTION  

        SET max_number TO an empty list  
        FOR i FROM zero TO k  
            IF i LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO CALL merge WITH PARAMETERS CALL maxSingleArray WITH PARAMETERS nums1 AND i, CALL maxSingleArray WITH PARAMETERS nums2 AND k MINUS i  
                IF candidate GREATER THAN max_number  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
