CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO a two dimensional list OF size n BY n WITH all VALUES SET TO true  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET the element at position i AND j OF f TO false  
                IF the element at position i OF s EQUALS the element at position j OF s  
                    SET the element at position i AND j OF f TO the element at position i PLUS one AND j MINUS one OF f  
                    IF the element at position i AND j OF f IS true AND mx IS LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the substring from position k TO k PLUS mx MINUS one OF s  
    END FUNCTION  
END CLASS
