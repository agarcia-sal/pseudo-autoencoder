CLASS Solution  
    FUNCTION checkSubarraySum(nums, k)  
        SET remainder_map TO a dictionary with key zero and value minus one  
        SET current_sum TO zero  
        FOR each index i AND element num IN nums  
            INCREMENT current_sum BY num  
            IF k NOT EQUALS zero THEN  
                SET current_sum TO the remainder of current_sum DIVIDED BY k  
            END IF  
            IF current_sum EXISTS AS a key IN remainder_map THEN  
                IF the difference BETWEEN i AND the element at key current_sum OF remainder_map IS GREATER THAN one THEN  
                    RETURN True  
                END IF  
            ELSE  
                ASSIGN the element at key current_sum OF remainder_map TO i  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
