CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO CALL ListNode WITH PARAMETERS element at position zero OF values  
    SET p TO head  
    FOR each val IN elements from position one TO the end OF values  
        SET node TO CALL ListNode WITH PARAMETERS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN (element val OF p1 EQUALS element val OF p2) AND CALL is_same_list WITH PARAMETERS element next OF p1, element next OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList(self, head)  
        IF head EQUALS None OR element next OF head EQUALS None OR element next OF element next OF head EQUALS None  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO element next OF head  
        SET even_head TO even  
        WHILE even IS NOT None AND element next OF even IS NOT None  
            SET odd.next TO element next OF even  
            SET odd TO element next OF odd  
            SET even.next TO element next OF odd  
            SET even TO element next OF even  
        END WHILE  
        SET odd.next TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
