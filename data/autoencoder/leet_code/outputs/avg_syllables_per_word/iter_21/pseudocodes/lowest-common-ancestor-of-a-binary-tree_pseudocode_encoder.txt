CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH val EQUALS element at position zero OF values  
    SET i TO one  
    SET queue TO a new double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left side OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO NEW TreeNode WITH val EQUALS element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO NEW TreeNode WITH val EQUALS element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN the result of CALL is_same_tree WITH PARAMETERS p.left AND q.left AND  
        CALL is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION lowestCommonAncestor(self, root, p, q)  
        IF root IS none OR root IS EQUAL TO p OR root IS EQUAL TO q  
            RETURN root  
        END IF  
        SET left TO CALL self.lowestCommonAncestor WITH PARAMETERS root.left, p, q  
        SET right TO CALL self.lowestCommonAncestor WITH PARAMETERS root.right, p, q  
        IF left IS NOT none AND right IS NOT none  
            RETURN root  
        END IF  
        IF left IS NOT none  
            RETURN left  
        ELSE  
            RETURN right  
        END IF  
    END FUNCTION  
END CLASS
