CLASS Solution  
    FUNCTION merge(intervals)  
        CALL intervals.sort WITH PARAMETERS key EQUALS function THAT RETURNS element at position zero OF parameter x  
        SET merged TO an empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR element at position one OF the last element OF merged IS LESS THAN element at position zero OF interval  
                APPEND interval TO merged  
            ELSE  
                SET element at position one OF the last element OF merged TO the maximum value BETWEEN element at position one OF the last element OF merged AND element at position one OF interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
