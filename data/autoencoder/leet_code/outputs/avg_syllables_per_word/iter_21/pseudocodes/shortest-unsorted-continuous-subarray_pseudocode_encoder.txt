CLASS Solution  
    FUNCTION findUnsortedSubarray(nums)  
        CALL sorted WITH PARAMETERS nums AND ASSIGN RESULT TO sorted_nums  
        SET start TO negative one  
        SET end TO negative two  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF element at position index OF nums NOT EQUALS element at position index OF sorted_nums  
                IF start EQUALS negative one  
                    SET start TO index  
                END IF  
                SET end TO index  
            END IF  
        END FOR  
        SET subarray_length TO end MINUS start PLUS one  
        RETURN subarray_length  
    END FUNCTION  
END CLASS
