CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO CALL TreeNode WITH PARAMETERS element at position zero OF values  
    SET i TO one  
    INITIALIZE queue AS A NEW EMPTY DEQUE  
    CALL queue APPEND WITH PARAMETERS root  
    WHILE queue IS NOT EMPTY  
        SET node TO CALL queue POP LEFT WITH NO PARAMETERS  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.left  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p IS NOT EQUALS value of q  
        RETURN false  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS value of p.left AND value of q.left AND CALL is_same_tree WITH PARAMETERS value of p.right AND value of q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers(self, root)  
        FUNCTION dfs(node, current_number)  
            IF node IS none  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS value of node  
            IF node.left IS none AND node.right IS none  
                RETURN current_number  
            END IF  
            SET left_sum TO CALL dfs WITH PARAMETERS node.left AND current_number  
            SET right_sum TO CALL dfs WITH PARAMETERS node.right AND current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN CALL dfs WITH PARAMETERS root AND zero  
    END FUNCTION  
END CLASS
