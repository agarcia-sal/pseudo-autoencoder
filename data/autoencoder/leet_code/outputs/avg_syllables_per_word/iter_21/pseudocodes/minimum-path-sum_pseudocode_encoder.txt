CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid EQUALS empty OR the element at position zero OF grid EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero OF grid  
        
        CALL create_2D_list WITH PARAMETERS m AND n  
        ASSIGN reference TO dp  
        
        SET element at position zero ZERO OF dp TO the element at position zero ZERO OF grid  
        
        FOR j FROM one TO n MINUS one  
            SET element at position zero j OF dp TO the element at position zero j MINUS one OF dp PLUS the element at position zero j OF grid  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            SET element at position i zero OF dp TO the element at position i MINUS one zero OF dp PLUS the element at position i zero OF grid  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET element at position i j OF dp TO the minimum BETWEEN the element at position i MINUS one j OF dp AND the element at position i j MINUS one OF dp PLUS the element at position i j OF grid  
            END FOR  
        END FOR  
        
        RETURN the element at position m MINUS one n MINUS one OF dp  
    END FUNCTION  
    
    FUNCTION create_2D_list(rows, columns)  
        CREATE a list named result  
        FOR row_index FROM zero TO rows MINUS one  
            CREATE a list named new_row  
            FOR column_index FROM zero TO columns MINUS one  
                APPEND zero TO new_row  
            END FOR  
            APPEND new_row TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
