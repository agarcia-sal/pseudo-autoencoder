CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path IS GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF the element at position start OF num EQUALS zero AND end IS GREATER THAN start PLUS one  
                    CONTINUE TO the NEXT ITERATION OF the FOR LOOP  
                END IF  
                SET next_num TO the INTEGER CONVERSION OF the substring from position start TO end MINUS one OF num  
                IF next_num IS GREATER THAN OR EQUAL TO two RAISED TO THE POWER OF thirty ONE  
                    BREAK OUT OF the FOR LOOP  
                END IF  
                IF the LENGTH OF path IS LESS THAN two OR next_num EQUALS the element at last position OF path PLUS the element at second LAST position OF path  
                    SET result TO CALL backtrack WITH PARAMETERS end AND path WITH next_num APPENDED  
                    IF result IS NOT EMPTY  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN an EMPTY LIST  
        END FUNCTION  
        RETURN CALL backtrack WITH PARAMETERS zero AND an EMPTY LIST  
    END FUNCTION  
END CLASS
