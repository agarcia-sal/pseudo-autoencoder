CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR i FROM zero TO eight  
                IF element at position row AND i OF board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            FOR i FROM zero TO eight  
                IF element at position i AND col OF board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY integer DIVISION OF row BY three  
            SET start_col TO three MULTIPLIED BY integer DIVISION OF col BY three  
            FOR i FROM zero TO two  
                FOR j FROM zero TO two  
                    IF element at position start_row PLUS i AND start_col PLUS j OF board EQUALS num  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
  
        FUNCTION solve(board)  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF element at position row AND col OF board EQUALS the character dot  
                        FOR num IN the result of converting numbers from one TO nine TO strings  
                            IF is_valid(board, row, col, num) HOLDS True  
                                SET element at position row AND col OF board TO num  
                                IF solve(board) HOLDS True  
                                    RETURN True  
                                END IF  
                                SET element at position row AND col OF board TO the character dot  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
  
        CALL solve WITH PARAMETERS board  
    END FUNCTION  
END CLASS
