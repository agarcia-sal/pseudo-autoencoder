CLASS Solution  
    FUNCTION decodeAtIndex(s STRING, k INTEGER) RETURNS STRING  
        SET length TO zero  
        
        FOR each char IN s  
            IF char IS A DIGIT  
                SET length TO length MULTIPLIED BY the INTEGER VALUE OF char  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        
        FOR each char IN the REVERSE ORDER OF s  
            SET k TO k MODULO length  
            
            IF k EQUALS zero AND char IS A LETTER OF THE ALPHABET  
                RETURN char  
            END IF  
            
            IF char IS A DIGIT  
                SET length TO length DIVIDED BY the INTEGER VALUE OF char  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
