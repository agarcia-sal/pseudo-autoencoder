CLASS Solution  
    FUNCTION minSubArrayLen(target, nums)  
        SET left TO zero  
        SET current_sum TO zero  
        SET min_length TO positive infinity  
          
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            INCREMENT current_sum BY the element at position right OF nums  
            
            WHILE current_sum GREATER THAN OR EQUAL TO target HOLDS  
                IF min_length GREATER THAN right MINUS left PLUS one THEN  
                    SET min_length TO right MINUS left PLUS one  
                END IF  
                DECREMENT current_sum BY the element at position left OF nums  
                INCREMENT left BY one  
            END WHILE  
        
        IF min_length EQUALS positive infinity THEN  
            RETURN zero  
        ELSE  
            RETURN min_length  
        END IF  
    END FUNCTION  
END CLASS
