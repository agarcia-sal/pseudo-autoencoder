CLASS Solution  
    FUNCTION maxPerformance(n, speed, efficiency, k)  
        SET engineers TO CALL zip WITH PARAMETERS efficiency AND speed CONVERTED TO LIST  
        CALL sort ON engineers WITH PARAMETERS reverse EQUALS true  
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        
        FOR each pair OF eff AND spd IN engineers  
            CALL heap push ON speed_heap WITH PARAMETER spd  
            INCREMENT speed_sum BY spd  
            
            IF the LENGTH OF speed_heap GREATER THAN k  
                SET removed_speed TO CALL heap pop ON speed_heap  
                DECREMENT speed_sum BY removed_speed  
            END IF  
            
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance GREATER THAN max_performance  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        
        SET modulus_base TO ten RAISED TO THE POWER OF nine PLUS one  
        SET max_performance TO max_performance MODULO modulus_base  
        RETURN max_performance  
    END FUNCTION  
END CLASS
