CLASS Solution  
    FUNCTION multiply(num1, num2)  
        IF num1 EQUALS the string zero OR num2 EQUALS the string zero  
            RETURN the string zero  
        END IF  
  
        SET result TO a list of zeros with a length EQUAL TO the sum of the LENGTH OF num1 AND the LENGTH OF num2  
  
        SET num1 TO the reverse of num1  
        SET num2 TO the reverse of num2  
  
        FOR index i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR index j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the MULTIPLICATION of the numeric value of the element at position i OF num1 WITH the numeric value of the element at position j OF num2  
                SET p1 TO the SUM of i AND j  
                SET p2 TO the SUM of i AND j AND one  
                SET sum_ TO the SUM of mul AND the element at position p1 OF result  
  
                SET the element at position p1 OF result TO sum_ MODULO ten  
                INCREMENT the element at position p2 OF result BY sum_ DIVIDED BY ten  
            END FOR  
        END FOR  
  
        SET reversed_result_str TO the concatenation of the string representations of the elements of result in reverse order  
        SET stripped_result_str TO reversed_result_str with all leading zeros REMOVED  
        RETURN stripped_result_str  
    END FUNCTION  
END CLASS
