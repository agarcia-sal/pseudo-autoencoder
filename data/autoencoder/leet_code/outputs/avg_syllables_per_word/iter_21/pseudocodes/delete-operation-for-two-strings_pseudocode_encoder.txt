CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        CALL createTwoDimensionalArray WITH PARAMETERS m PLUS one, n PLUS one  
        ASSIGN reference TO dp  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one OF word1 EQUALS the element at position j MINUS one OF word2  
                    SET the element at position i OF dp AT position j TO the element at position i MINUS one OF dp AT position j MINUS one PLUS one  
                ELSE  
                    SET the element at position i OF dp AT position j TO the maximum VALUE BETWEEN the element at position i MINUS one OF dp AT position j AND the element at position i OF dp AT position j MINUS one  
                END IF  
            END FOR  
        END FOR  
        
        SET lcs_length TO the element at position m OF dp AT position n  
        
        SET minimum_deletions TO m MINUS lcs_length PLUS n MINUS lcs_length  
        
        RETURN minimum_deletions  
    END FUNCTION  
END CLASS  

FUNCTION createTwoDimensionalArray(rows, columns)  
    SET result TO an empty list  
    FOR index FROM zero TO rows MINUS one  
        CALL createListOfZeros WITH PARAMETER columns  
        APPEND the created list TO result  
    END FOR  
    RETURN result  
END FUNCTION

FUNCTION createListOfZeros(length)  
    SET list_of_zeros TO an empty list  
    FOR index FROM zero TO length MINUS one  
        APPEND zero TO list_of_zeros  
    END FOR  
    RETURN list_of_zeros  
END FUNCTION
