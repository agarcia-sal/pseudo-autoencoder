CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO the result of splitting num by the plus symbol  
            SET real TO the integer conversion of the element at position zero OF parts  
            SET imaginary_string TO the element at position one OF parts  
            SET imaginary TO the integer conversion of the substring from position zero TO the length of imaginary_string MINUS one OF imaginary_string  
            RETURN real AND imaginary  
        END FUNCTION  

        SET real1 AND imag1 TO CALL parse_complex WITH PARAMETERS num1  
        SET real2 AND imag2 TO CALL parse_complex WITH PARAMETERS num2  

        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  

        RETURN a string concatenation of real_part, plus symbol, imaginary_part, and the letter i  
    END FUNCTION  
END CLASS
