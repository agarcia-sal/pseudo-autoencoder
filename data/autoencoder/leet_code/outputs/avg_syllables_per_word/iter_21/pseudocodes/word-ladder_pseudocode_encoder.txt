CLASS Solution  
    FUNCTION ladderLength(beginWord, endWord, wordList)  
        IF endWord NOT IN wordList  
            RETURN zero  
        END IF  
        
        CALL word_dict TO CREATE A DEFAULT DICTIONARY THAT MAPS TO A LIST  
        SET word_length TO THE LENGTH OF beginWord  
        
        FOR each word IN wordList  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_key TO CONCATENATION OF substring from position zero TO i MINUS one OF word PLUS THE CHARACTER ASTERISK PLUS substring from position i PLUS one TO word_length MINUS one OF word  
                APPEND word TO THE LIST AT intermediate_key IN word_dict  
            END FOR  
        END FOR  
        
        CALL queue TO CREATE A DEQUE INITIALIZED WITH beginWord AND one AS A PAIR  
        CALL visited TO CREATE A SET INITIALIZED WITH beginWord  
        
        WHILE queue IS NOT EMPTY  
            REMOVE FIRST ELEMENT FROM queue AND ASSIGN TO current_word AND current_length  
            
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_word TO CONCATENATION OF substring from position zero TO i MINUS one OF current_word PLUS THE CHARACTER ASTERISK PLUS substring from position i PLUS one TO word_length MINUS one OF current_word  
                
                FOR each word IN THE LIST AT intermediate_word IN word_dict  
                    IF word EQUALS endWord  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited  
                        ADD word TO visited  
                        APPEND word AND current_length PLUS one AS A PAIR TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN zero  
    END FUNCTION  
END CLASS
