CLASS Solution  
    FUNCTION splitArraySameAverage(nums)  
        SET n TO the LENGTH OF nums  
        SET total_sum TO the SUM OF nums  
        
        IF n EQUALS one  
            RETURN False  
        END IF  
        
        SET dp TO a list CONTAINING n DIVIDED BY two INTEGERS PLUS one SETS  
        ADD zero TO the SET at position zero IN dp  
        
        FOR each num IN nums  
            FOR size FROM n DIVIDED BY two TO one DECREMENTING BY one  
                FOR each s IN the SET at position size MINUS one IN dp  
                    SET new_sum TO s PLUS num  
                    IF new_sum MULTIPLIED BY n EQUALS size MULTIPLIED BY total_sum  
                        RETURN True  
                    END IF  
                    ADD new_sum TO the SET at position size IN dp  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
