CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        
        SET last TO a list of twenty-six elements EACH SET TO minus one  
        SET last2 TO a list of twenty-six elements EACH SET TO minus one  
        
        SET result TO zero  
        
        FOR each index i AND character char IN the enumeration of s  
            SET idx TO the Unicode code point of char MINUS the Unicode code point of uppercase letter A  
            
            SET contribution TO (i MINUS the element at position idx OF last) MULTIPLIED BY (the element at position idx OF last MINUS the element at position idx OF last2)  
            SET result TO the remainder after division of (result PLUS contribution) BY MOD  
            
            SET the element at position idx OF last2 TO the element at position idx OF last  
            SET the element at position idx OF last TO i  
        END FOR  
        
        FOR index i FROM zero TO twenty-five  
            SET contribution TO (n MINUS the element at position i OF last) MULTIPLIED BY (the element at position i OF last MINUS the element at position i OF last2)  
            SET result TO the remainder after division of (result PLUS contribution) BY MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
