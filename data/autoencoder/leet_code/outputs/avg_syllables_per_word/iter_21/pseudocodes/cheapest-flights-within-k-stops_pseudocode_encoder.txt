CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO a new empty dictionary  
        FOR index FROM zero TO n MINUS one  
            SET the element at position index OF graph TO an empty list  
        END FOR  
        FOR each element consisting of u, v, w IN flights  
            APPEND the pair consisting of v and w TO the list AT key u OF graph  
        END FOR  
      
        SET min_heap TO a list containing the triple consisting of zero, negative one, and src  
        SET min_cost TO a dictionary with key src assigned to value negative one  
      
        WHILE min_heap HOLDS  
            CALL function to remove and RETURN the smallest element FROM min_heap AND ASSIGN it TO cost, stops, city  
          
            IF city EQUALS dst THEN  
                RETURN cost  
            END IF  
          
            IF stops IS GREATER THAN OR EQUAL TO k THEN  
                CONTINUE TO the next iteration OF the WHILE loop  
            END IF  
          
            FOR each element consisting of neighbor and price IN the list AT key city OF graph  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  
              
                IF neighbor IS NOT PRESENT AS A KEY IN min_cost OR next_stops IS LESS THAN the element AT key neighbor OF min_cost OR next_cost IS LESS THAN the first element of the first tuple IN min_heap THEN  
                    SET the element AT key neighbor OF min_cost TO next_stops  
                    CALL function to insert the triple consisting of next_cost, next_stops, neighbor INTO min_heap  
                END IF  
            END FOR  
        END WHILE  
      
        RETURN negative one  
    END FUNCTION  
END CLASS
