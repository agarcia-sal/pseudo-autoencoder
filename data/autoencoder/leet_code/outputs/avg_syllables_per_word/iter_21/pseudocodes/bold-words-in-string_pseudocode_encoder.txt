CLASS Solution  
    FUNCTION boldWords(words, s)  
        SET bold TO a list of Boolean False values of the same LENGTH AS s  
        FOR each word IN words  
            SET start TO the POSITION OF word IN s  
            WHILE start IS NOT EQUAL TO negative one  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET element at position i OF bold TO True  
                END FOR  
                SET start TO the POSITION OF word IN s STARTING FROM start PLUS one  
            END WHILE  
        END FOR  
        SET result TO an empty list  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF s  
            IF element at position i OF bold IS True  
                APPEND the string consisting of the opening bold tag TO result  
                WHILE i IS LESS THAN the LENGTH OF s AND element at position i OF bold IS True  
                    APPEND the CHARACTER at position i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string consisting of the closing bold tag TO result  
            ELSE  
                APPEND the CHARACTER at position i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the CONCATENATION OF all elements IN result AS a single string  
    END FUNCTION  
END CLASS
