CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s, k)  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
        
        SET char_map TO empty dictionary  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            SET element at position right OF s AS key IN char_map WITH VALUE right  
            
            IF the LENGTH OF char_map GREATER THAN k  
                SET leftmost_char TO the key IN char_map WITH THE MINIMUM corresponding VALUE  
                SET left TO the VALUE ASSOCIATED WITH leftmost_char IN char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET current_length TO right MINUS left PLUS one  
            IF current_length GREATER THAN max_length  
                SET max_length TO current_length  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
