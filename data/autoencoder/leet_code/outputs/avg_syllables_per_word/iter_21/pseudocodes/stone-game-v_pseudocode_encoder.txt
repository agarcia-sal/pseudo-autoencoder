CLASS Solution  
    FUNCTION stoneGameV(stoneValue)  
        FUNCTION get_range_sum(left, right)  
            RETURN element at position right PLUS one OF prefix_sum MINUS element at position left OF prefix_sum  
        END FUNCTION  
  
        FUNCTION dp(left, right)  
            IF left EQUALS right THEN  
                RETURN zero  
            END IF  
            SET max_score TO zero  
            FOR i FROM left TO right MINUS one  
                SET left_sum TO CALL get_range_sum WITH PARAMETERS left AND i  
                SET right_sum TO CALL get_range_sum WITH PARAMETERS i PLUS one AND right  
                IF left_sum LESS THAN right_sum THEN  
                    IF max_score LESS THAN left_sum PLUS CALL dp WITH PARAMETERS left AND i THEN  
                        SET max_score TO left_sum PLUS CALL dp WITH PARAMETERS left AND i  
                    END IF  
                ELSE IF left_sum GREATER THAN right_sum THEN  
                    IF max_score LESS THAN right_sum PLUS CALL dp WITH PARAMETERS i PLUS one AND right THEN  
                        SET max_score TO right_sum PLUS CALL dp WITH PARAMETERS i PLUS one AND right  
                    END IF  
                ELSE  
                    IF max_score LESS THAN left_sum PLUS CALL dp WITH PARAMETERS left AND i THEN  
                        SET max_score TO left_sum PLUS CALL dp WITH PARAMETERS left AND i  
                    END IF  
                    IF max_score LESS THAN right_sum PLUS CALL dp WITH PARAMETERS i PLUS one AND right THEN  
                        SET max_score TO right_sum PLUS CALL dp WITH PARAMETERS i PLUS one AND right  
                    END IF  
                END IF  
            END FOR  
            RETURN max_score  
        END FUNCTION  
  
        SET prefix_sum TO a list OF zeros WITH the LENGTH OF stoneValue PLUS one  
        FOR i FROM zero TO the LENGTH OF stoneValue MINUS one  
            SET element at position i PLUS one OF prefix_sum TO element at position i OF prefix_sum PLUS element at position i OF stoneValue  
        END FOR  
  
        RETURN CALL dp WITH PARAMETERS zero AND the LENGTH OF stoneValue MINUS one  
    END FUNCTION  
END CLASS
