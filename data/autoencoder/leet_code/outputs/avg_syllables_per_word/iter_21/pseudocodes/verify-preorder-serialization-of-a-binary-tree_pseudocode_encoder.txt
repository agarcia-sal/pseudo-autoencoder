CLASS Solution  
    FUNCTION isValidSerialization(preorder)  
        SET nodes TO CALL split_string_by_comma WITH PARAMETERS preorder  
        SET slots TO one  
        FOR each node IN nodes  
            IF slots EQUALS zero  
                RETURN false  
            END IF  
            DECREMENT slots BY one  
            IF node NOT EQUALS the character hash symbol  
                INCREMENT slots BY two  
            END IF  
        END FOR  
        RETURN slots EQUALS zero  
    END FUNCTION  
END CLASS
