CLASS Solution  
    FUNCTION waysToDistribute(n, k)  
        SET MOD TO one billion PLUS seven  
        CALL initialize_dp_array WITH PARAMETERS n, k  
        SET element at position zero AT position zero OF dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET product TO j MULTIPLIED BY element at position i MINUS one AT position j OF dp  
                SET sum_value TO product PLUS element at position i MINUS one AT position j MINUS one OF dp  
                SET element at position i AT position j OF dp TO sum_value MODULO MOD  
            END FOR  
        END FOR  
        RETURN element at position n AT position k OF dp  
    END FUNCTION  
  
    FUNCTION initialize_dp_array(n, k)  
        CREATE dp AS a two-dimensional list WITH n PLUS one rows AND k PLUS one columns FILLED WITH zero  
        RETURN dp  
    END FUNCTION  
END CLASS
