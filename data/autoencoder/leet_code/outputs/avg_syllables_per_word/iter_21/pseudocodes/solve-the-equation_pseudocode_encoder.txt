CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO THE RESULT OF REPLACING PLUS SYMBOL WITH HASH PLUS SYMBOL IN side AND THEN REPLACING MINUS SYMBOL WITH HASH MINUS SYMBOL IN THAT RESULT AND THEN SPLITTING THAT RESULT BY HASH SYMBOL  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token EQUALS empty string  
                    CONTINUE TO NEXT token  
                END IF  
                IF token CONTAINS character x  
                    IF token EQUALS character x OR token EQUALS PLUS SYMBOL FOLLOWED BY character x  
                        INCREMENT x_count BY one  
                    ELSE IF token EQUALS MINUS SYMBOL FOLLOWED BY character x  
                        DECREMENT x_count BY one  
                    ELSE  
                        SET x_count TO x_count PLUS THE INTEGER VALUE OF THE RESULT OF REMOVING character x FROM token  
                    END IF  
                ELSE  
                    SET num_sum TO num_sum PLUS THE INTEGER VALUE OF token  
                END IF  
            END FOR  
            RETURN x_count AND num_sum  
        END FUNCTION  
  
        SET left_side AND right_side TO THE RESULT OF SPLITTING equation BY EQUALS SYMBOL  
        SET left_x AND left_num TO CALL parse_side WITH PARAMETER left_side  
        SET right_x AND right_num TO CALL parse_side WITH PARAMETER right_side  
  
        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  
  
        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN THE STRING Infinite solutions  
            ELSE  
                RETURN THE STRING No solution  
            END IF  
        ELSE  
            RETURN THE STRING x EQUALS THE RESULT OF total_num DIVIDED BY total_x USING INTEGER DIVISION  
        END IF  
    END FUNCTION  
END CLASS
