CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO sorted New List CONSISTING OF elements OF cuts PLUS zero AND n  
        
        FUNCTION dp(left, right)  
            IF right MINUS left LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO positive infinity  
            FOR i FROM left PLUS one TO right MINUS one  
                SET cost TO element at position right OF cuts MINUS element at position left OF cuts PLUS CALL dp WITH PARAMETERS left AND i PLUS CALL dp WITH PARAMETERS i AND right  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN CALL dp WITH PARAMETERS zero AND the LENGTH OF cuts MINUS one  
    END FUNCTION  
END CLASS
