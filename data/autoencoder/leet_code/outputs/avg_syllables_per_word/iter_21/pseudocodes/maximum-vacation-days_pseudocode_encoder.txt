CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the element at position zero OF days  
          
        SET dp TO a list of n elements each SET TO negative one  
        SET the element at position zero OF dp TO zero  
          
        SET graph TO an empty dictionary with keys from zero TO n MINUS one each mapping TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i OF flights AT position j EQUALS one  
                    APPEND j TO the list mapped by key i IN graph  
                END IF  
            END FOR  
        END FOR  
          
        FOR week FROM zero TO k MINUS one  
            SET new_dp TO a list of n elements each SET TO negative one  
            FOR i FROM zero TO n MINUS one  
                IF the element at position i OF dp EQUALS negative one  
                    CONTINUE to next iteration of the FOR loop  
                END IF  
                SET the element at position i OF new_dp TO the greater VALUE BETWEEN the element at position i OF new_dp AND the element at position i OF dp PLUS the element at position i OF days AT position week  
                FOR each j IN the list mapped by key i IN graph  
                    SET the element at position j OF new_dp TO the greater VALUE BETWEEN the element at position j OF new_dp AND the element at position i OF dp PLUS the element at position j OF days AT position week  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
          
        SET max_value TO the greatest VALUE IN dp  
        RETURN max_value  
    END FUNCTION  
END CLASS
