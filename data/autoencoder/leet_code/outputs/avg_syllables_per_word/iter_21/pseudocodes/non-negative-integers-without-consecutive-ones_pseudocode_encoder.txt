CLASS Solution  
    FUNCTION findIntegers(n)  
        CALL bin_representation WITH PARAMETERS n TO OBTAIN binary representation of n as string WITHOUT the leading prefix  
        SET length TO the LENGTH OF bin_representation  
        
        INITIALIZE dp AS a list of zeros with a size of length PLUS one  
        SET element at position zero OF dp TO one  
        SET element at position one OF dp TO two  
        
        FOR index FROM two TO length INCLUSIVE  
            SET element at position index OF dp TO the SUM of elements at positions index MINUS one AND index MINUS two OF dp  
        END FOR  
        
        SET result TO zero  
        SET prev_bit TO zero  
        
        FOR index FROM zero TO length MINUS one  
            IF the element at position index OF bin_representation EQUALS the character representing one  
                INCREMENT result BY element at position length MINUS index MINUS one OF dp  
                IF prev_bit EQUALS one  
                    DECREMENT result BY one  
                    BREAK the loop  
                END IF  
            END IF  
            SET prev_bit TO one IF the element at position index OF bin_representation EQUALS the character representing one ELSE zero  
        END FOR  
        
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
