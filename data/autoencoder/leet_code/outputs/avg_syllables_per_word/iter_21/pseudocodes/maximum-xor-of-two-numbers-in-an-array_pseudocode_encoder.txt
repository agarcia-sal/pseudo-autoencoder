CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO zero  
        FOR i FROM thirty-one TO zero DECREMENT BY one  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor INCREMENTED BY one  
            CALL initializePrefixes WITH PARAMETERS nums AND i  
            SET prefixes TO the result OF initializePrefixes  
            FOR each prefix IN prefixes  
                CALL checkPrefixes WITH PARAMETERS current_candidate AND prefix AND prefixes  
                IF the result OF checkPrefixes IS true  
                    SET max_xor TO current_candidate  
                    BREAK the loop  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
  
    FUNCTION initializePrefixes(nums, i)  
        CREATE empty set prefixes  
        FOR each num IN nums  
            SET prefix TO the integer division of num BY two RAISED TO THE POWER OF i  
            ADD prefix TO prefixes  
        END FOR  
        RETURN prefixes  
    END FUNCTION  
  
    FUNCTION checkPrefixes(current_candidate, prefix, prefixes)  
        SET xor_value TO the bitwise exclusive OR of current_candidate AND prefix  
        IF xor_value IS IN prefixes  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
