CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF the LENGTH OF nums1 EQUALS zero OR the LENGTH OF nums2 EQUALS zero OR k LESS THAN OR EQUAL TO zero  
            RETURN an empty list  
        END IF  

        SET min_heap TO an empty list  

        FOR index i FROM zero TO the MINIMUM OF k AND the LENGTH OF nums1 MINUS one  
            CALL the heap push operation TO ADD the tuple of the sum of element at position i OF nums1 AND element at position zero OF nums2, the index i, AND zero INTO min_heap  
        END FOR  

        SET result TO an empty list  

        WHILE min_heap IS NOT empty AND the LENGTH OF result IS LESS THAN k  
            CALL the heap pop operation TO REMOVE the smallest element FROM min_heap AND ASSIGN the second element OF the tuple TO index i AND the third element OF the tuple TO index j  
            APPEND the list containing element at position i OF nums1 AND element at position j OF nums2 TO result  
            IF j PLUS one IS LESS THAN the LENGTH OF nums2  
                CALL the heap push operation TO ADD the tuple of the sum of element at position i OF nums1 AND element at position j PLUS one OF nums2, the index i, AND j PLUS one INTO min_heap  
            END IF  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
