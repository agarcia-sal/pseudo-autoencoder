CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        CALL defaultdict WITH PARAMETERS integer FUNCTION AND ASSIGN reference TO char_count  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the COUNT OF the CHARACTER at position right OF s IN char_count BY one  
            
            WHILE the NUMBER OF KEYS IN char_count GREATER THAN two HOLDS  
                DECREMENT the COUNT OF the CHARACTER at position left OF s IN char_count BY one  
                IF the COUNT OF the CHARACTER at position left OF s IN char_count EQUALS zero THEN  
                    REMOVE the CHARACTER at position left OF s FROM char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_length TO the GREATER OF max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
