CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a new default mapping from keys to empty lists  
        FOR each pair of departure AIRPORT and destination AIRPORT IN the list of tickets sorted in ascending order and then reversed  
            APPEND the destination AIRPORT TO the list associated with the departure AIRPORT in graph  
        END FOR  
        SET itinerary TO an empty list  
        FUNCTION visit(airport)  
            WHILE the list associated with airport in graph IS NOT empty  
                CALL visit WITH PARAMETERS the element removed from the end of the list associated with airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        CALL visit WITH PARAMETERS the string representing the airport JFK  
        RETURN the reverse of itinerary  
    END FUNCTION  
END CLASS
