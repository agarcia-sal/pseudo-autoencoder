CLASS Solution  
    FUNCTION minMoves(nums, k)  
        SET positions TO CALL extract_positions WITH PARAMETERS nums  
        
        FUNCTION calculate_cost(start, end)  
            SET mid TO start PLUS end DIVIDED BY two  
            SET median TO element at position mid OF positions  
            SET cost TO zero  
            FOR index FROM start TO end MINUS one  
                SET offset TO mid MINUS index  
                SET target_position TO median MINUS offset  
                SET distance TO absolute VALUE OF element at position index OF positions MINUS target_position  
                INCREMENT cost BY distance  
            END FOR  
            RETURN cost  
        END FUNCTION  
        
        SET min_cost TO positive infinity  
        FOR index FROM zero TO the LENGTH OF positions MINUS k  
            SET current_cost TO CALL calculate_cost WITH PARAMETERS index AND index PLUS k  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
