CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO an empty list  
      
        FOR each token IN tokens  
            IF token EQUALS one of the characters PLUS MINUS MULTIPLY DIVIDE  
                SET b TO the top element FROM stack BY removing it  
                SET a TO the top element FROM stack BY removing it  
                IF token EQUALS PLUS  
                    APPEND a PLUS b TO stack  
                ELSE IF token EQUALS MINUS  
                    APPEND a MINUS b TO stack  
                ELSE IF token EQUALS MULTIPLY  
                    APPEND a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS DIVIDE  
                    APPEND integer division result of a DIVIDED BY b that truncates towards zero TO stack  
                END IF  
            ELSE  
                APPEND integer conversion of token TO stack  
            END IF  
        END FOR  
      
        RETURN element at position zero OF stack  
    END FUNCTION  
END CLASS
