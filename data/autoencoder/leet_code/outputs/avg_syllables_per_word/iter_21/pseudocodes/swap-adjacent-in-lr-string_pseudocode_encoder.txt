CLASS Solution  
    FUNCTION canTransform(start, end)  
        SET start_filtered TO CALL list_of_character_and_index_pairs_excluding_character WITH PARAMETERS start AND character_to_exclude EQUALS character X  
        SET end_filtered TO CALL list_of_character_and_index_pairs_excluding_character WITH PARAMETERS end AND character_to_exclude EQUALS character X  
        
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        
        FOR each pair_of_character_and_index FROM start_filtered AND each_pair_of_character_and_index FROM end_filtered  
            SET character_one TO the character FROM the first pair_of_character_and_index  
            SET index_one TO the index FROM the first pair_of_character_and_index  
            SET character_two TO the character FROM the second pair_of_character_and_index  
            SET index_two TO the index FROM the second pair_of_character_and_index  
            
            IF character_one NOT EQUALS character_two  
                RETURN false  
            END IF  
            
            IF character_one EQUALS character L AND index_one LESS THAN index_two  
                RETURN false  
            END IF  
            
            IF character_one EQUALS character R AND index_one GREATER THAN index_two  
                RETURN false  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
    
    FUNCTION list_of_character_and_index_pairs_excluding_character(input_string, character_to_exclude)  
        SET result_list TO an empty list  
        FOR index FROM zero TO the LENGTH OF input_string MINUS one  
            SET current_character TO the character at position index OF input_string  
            IF current_character NOT EQUALS character_to_exclude  
                APPEND the pair of current_character AND index TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
