CLASS Solution  
    FUNCTION getPermutation(n, k)  
        CALL import math WITH PARAMETERS  
        SET numbers TO a list of integers from one TO n  
        DECREMENT k BY one  
        SET result TO an empty list  
        FOR i FROM n TO one DECREMENTED BY one  
            SET fact TO the factorial of i MINUS one  
            SET index TO the integer division of k BY fact  
            APPEND the string conversion of the element at position index OF numbers TO result  
            REMOVE the element at position index FROM numbers  
            SET k TO the remainder of k DIVIDED BY fact  
        END FOR  
        RETURN the concatenation of all elements OF result  
    END FUNCTION  
END CLASS
