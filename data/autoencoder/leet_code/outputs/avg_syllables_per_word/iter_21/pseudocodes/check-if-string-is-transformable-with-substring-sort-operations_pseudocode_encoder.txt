CLASS Solution  
    FUNCTION isTransformable(s, t)  
        CALL import_defaultdict FROM collections  
        CALL import_heappop FROM heapq  
        CALL import_heappush FROM heapq  
        
        CALL SET pos TO CALL defaultdict WITH PARAMETERS list
        
        FOR each index_and_character AS i AND c IN ENUMERATE s  
            CALL SET digit_value TO THE integer representation OF character c  
            CALL heappush WITH PARAMETERS pos AT position digit_value AND i  
        END FOR  
        
        FOR each index_and_character AS i AND c IN ENUMERATE t  
            CALL SET digit TO THE integer representation OF character c  
            IF pos AT position digit IS empty  
                RETURN false  
            END IF  
            FOR each smaller_digit FROM zero TO digit MINUS one  
                IF pos AT position smaller_digit IS NOT empty AND the element at position zero OF pos AT position smaller_digit IS LESS THAN the element at position zero OF pos AT position digit  
                    RETURN false  
                END IF  
            END FOR  
            CALL heappop WITH PARAMETERS pos AT position digit  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
