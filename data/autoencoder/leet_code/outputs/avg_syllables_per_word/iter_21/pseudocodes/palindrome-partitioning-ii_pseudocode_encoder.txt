CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
      
        CALL FUNCTION create_palindrome_table WITH PARAMETERS s AND n  
        ASSIGN reference TO is_palindrome OBJECT RETURNED FROM create_palindrome_table  
      
        CALL FUNCTION create_min_cuts_array WITH PARAMETERS is_palindrome AND n  
        ASSIGN reference TO min_cuts OBJECT RETURNED FROM create_min_cuts_array  
      
        RETURN element at position n MINUS one OF min_cuts  
    END FUNCTION  
   
    FUNCTION create_palindrome_table(s, n)  
        SET is_palindrome TO two-DIMENSIONAL LIST OF size n BY n FILLED WITH false  
        FOR index i FROM zero TO n MINUS one  
            SET element at position i AT position i OF is_palindrome TO true  
        END FOR  
      
        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET element at position start AT position end OF is_palindrome TO element at position start OF s EQUALS element at position end OF s  
                ELSE  
                    SET element at position start AT position end OF is_palindrome TO (element at position start OF s EQUALS element at position end OF s) AND element at position start PLUS one AT position end MINUS one OF is_palindrome  
                END IF  
            END FOR  
        END FOR  
      
        RETURN is_palindrome  
    END FUNCTION  
   
    FUNCTION create_min_cuts_array(is_palindrome, n)  
        SET min_cuts TO LIST OF size n FILLED WITH zero  
        FOR index i FROM zero TO n MINUS one  
            IF element at position zero AT position i OF is_palindrome  
                SET element at position i OF min_cuts TO zero  
            ELSE  
                SET element at position i OF min_cuts TO INFINITY  
                FOR index j FROM one TO i  
                    IF element at position j AT position i OF is_palindrome AND element at position j MINUS one OF min_cuts PLUS one LESS THAN element at position i OF min_cuts  
                        SET element at position i OF min_cuts TO element at position j MINUS one OF min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
      
        RETURN min_cuts  
    END FUNCTION  
END CLASS
