CLASS Solution  
    FUNCTION subarraysWithKDistinct(nums, k)  
        FUNCTION atMostKDistinct(t)  
            SET count TO NEW defaultdict OF integer  
            SET left TO zero  
            SET result TO zero  
            FOR right FROM zero TO the LENGTH OF nums MINUS one  
                INCREMENT the element at position right OF nums IN count BY one  
                WHILE the NUMBER OF KEYS IN count IS GREATER THAN t  
                    DECREMENT the element at position left OF nums IN count BY one  
                    IF the element at position left OF nums IN count EQUALS zero  
                        REMOVE the element at position left OF nums FROM count  
                    END IF  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT result BY right MINUS left PLUS one  
            END FOR  
            RETURN result  
        END FUNCTION  
        RETURN CALL atMostKDistinct WITH PARAMETERS k MINUS CALL atMostKDistinct WITH PARAMETERS k MINUS one  
    END FUNCTION  
END CLASS
