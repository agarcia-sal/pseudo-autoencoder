CLASS Solution  
    FUNCTION canDivideIntoSubsequences(nums, k)  
        CALL Counter WITH PARAMETERS nums AND ASSIGN reference TO freq  
        SET max_freq TO the maximum value OF freq values  
        SET division_result TO the length OF nums DIVIDED BY k  
        IF division_result IS GREATER THAN OR EQUAL TO max_freq  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
