CLASS Solution
    FUNCTION addBoldTag(s, words)
        IF the LENGTH OF words EQUALS zero
            RETURN s
        END IF

        SET n TO the LENGTH OF s
        SET mask TO a list of False values of LENGTH n

        FOR each word IN words
            SET start TO zero
            WHILE start IS LESS THAN OR EQUAL TO n MINUS the LENGTH OF word
                SET start TO the POSITION OF word starting FROM start IN s
                IF start EQUALS negative one
                    BREAK
                END IF
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one
                    SET the element at position i OF mask TO True
                END FOR
                INCREMENT start BY one
            END WHILE
        END FOR

        SET result TO an empty list
        SET i TO zero
        WHILE i IS LESS THAN n
            IF the element at position i OF mask IS True
                APPEND the string "OPEN BOLD TAG" TO result
                WHILE i IS LESS THAN n AND the element at position i OF mask IS True
                    APPEND the character at position i OF s TO result
                    INCREMENT i BY one
                END WHILE
                APPEND the string "CLOSE BOLD TAG" TO result
            ELSE
                APPEND the character at position i OF s TO result
                INCREMENT i BY one
            END IF
        END WHILE

        RETURN the concatenation of all elements OF result
    END FUNCTION
END CLASS
