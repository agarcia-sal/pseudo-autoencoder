CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS NOT True  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETERS element at position zero OF values  
    SET i TO one  
    SET queue TO CALL deque WITH PARAMETERS no arguments  
    CALL append WITH PARAMETERS root ON queue  
    WHILE queue HOLDS True  
        SET node TO CALL popleft WITH PARAMETERS no arguments ON queue  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO NEW TreeNode WITH PARAMETERS element at position i OF values  
            CALL append WITH PARAMETERS node.left ON queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO NEW TreeNode WITH PARAMETERS element at position i OF values  
            CALL append WITH PARAMETERS node.right ON queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS NOT True AND q EQUALS NOT True  
        RETURN True  
    ELSE IF p EQUALS NOT True OR q EQUALS NOT True  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left, q.left AND CALL is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover(self, root)  
        FUNCTION dfs(node)  
            IF node EQUALS NOT True  
                RETURN zero, zero, infinity  
            END IF  
            SET left_no_camera, left_monitored, left_camera TO CALL dfs WITH PARAMETERS node.left  
            SET right_no_camera, right_monitored, right_camera TO CALL dfs WITH PARAMETERS node.right  
            SET no_camera TO left_monitored PLUS right_monitored  
            SET monitored TO THE MINIMUM OF left_camera PLUS THE MINIMUM OF right_monitored, right_camera  
                                        AND right_camera PLUS THE MINIMUM OF left_monitored, left_camera  
                                        AND left_camera PLUS right_camera  
            SET camera TO one PLUS THE MINIMUM OF left_no_camera, left_monitored, left_camera PLUS THE MINIMUM OF right_no_camera, right_monitored, right_camera  
            RETURN no_camera, monitored, camera  
        END FUNCTION  
        SET root_no_camera, root_monitored, root_camera TO CALL dfs WITH PARAMETERS root  
        RETURN THE MINIMUM OF root_monitored, root_camera  
    END FUNCTION  
END CLASS
