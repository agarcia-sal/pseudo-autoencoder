CLASS Solution  
    FUNCTION advantageCount(nums1, nums2)  
        CALL sort ON nums1 TO ARRANGE ELEMENTS IN ASCENDING ORDER  
        
        SET nums2_with_indices TO CALL sorted WITH PARAMETERS A LIST OF PAIRS OF EACH VALUE AND ORIGINAL POSITION FROM nums2, WITH ORDER FROM HIGHEST TO LOWEST VALUE  
        
        SET result TO A LIST OF ZERO ELEMENTS WITH THE SAME LENGTH AS nums1  
        
        SET left TO zero  
        SET right TO THE LENGTH OF nums1 MINUS one  
        
        FOR each pair OF value AND index IN nums2_with_indices  
            IF element AT POSITION right OF nums1 IS GREATER THAN value  
                SET element AT POSITION index OF result TO element AT POSITION right OF nums1  
                DECREMENT right BY one  
            ELSE  
                SET element AT POSITION index OF result TO element AT POSITION left OF nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
