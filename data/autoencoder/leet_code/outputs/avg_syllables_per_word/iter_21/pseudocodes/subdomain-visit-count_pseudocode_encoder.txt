CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO a new default dictionary with integer default value  
        
        FOR each cpdomain IN cpdomains  
            CALL split function WITH PARAMETERS cpdomain AND a space TO OBTAIN count AND domain  
            SET count TO the integer conversion OF count  
            CALL split function WITH PARAMETERS domain AND a period TO OBTAIN subdomains  
            
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one  
                SET subdomain TO the concatenation of elements FROM position index TO the END OF subdomains JOINED BY a period  
                INCREMENT the value AT key subdomain IN domain_count BY count  
            END FOR  
        END FOR  
        
        SET result_list TO an empty list  
        FOR each domain AND count IN the items OF domain_count  
            APPEND the concatenation OF count AS STRING, a space, AND domain TO result_list  
        END FOR  
        
        RETURN result_list  
    END FUNCTION  
END CLASS
