CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO CALL Counter WITH PARAMETERS nums  
        SET end_with TO CALL Counter WITH PARAMETERS empty collection  
        
        FOR each num IN nums  
            IF element at position num OF num_count EQUALS zero  
                CONTINUE  
            END IF  
            
            IF element at position num MINUS one OF end_with GREATER THAN zero  
                DECREMENT element at position num MINUS one OF end_with BY one  
                INCREMENT element at position num OF end_with BY one  
            ELSE IF element at position num PLUS one OF num_count GREATER THAN zero AND element at position num PLUS two OF num_count GREATER THAN zero  
                DECREMENT element at position num PLUS one OF num_count BY one  
                DECREMENT element at position num PLUS two OF num_count BY one  
                INCREMENT element at position num PLUS two OF end_with BY one  
            ELSE  
                RETURN False  
            END IF  
            
            DECREMENT element at position num OF num_count BY one  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
