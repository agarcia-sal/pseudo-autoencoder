CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an empty set  
        SET min_heap TO a list containing a tuple of element at position zero zero OF grid, zero, zero  
        ADD the tuple of zero and zero TO visited  
        SET directions TO a list containing tuples zero one, one zero, zero minus one, minus one zero  
        
        WHILE min_heap IS NOT empty  
            CALL heapq function heappop WITH PARAMETERS min_heap AND ASSIGN RESULTS TO t i j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            
            FOR each tuple di dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND the tuple of ni nj NOT IN visited  
                    ADD the tuple of ni nj TO visited  
                    CALL heapq function heappush WITH PARAMETERS min_heap AND a tuple of the GREATER OF t AND the element at position ni nj OF grid, ni, nj  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
