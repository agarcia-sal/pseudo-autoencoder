CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO list containing one thousand, nine hundred, five hundred, four hundred, one hundred, ninety, fifty, forty, ten, nine, five, four, one  
        SET syms TO list containing M, CM, D, CD, C, XC, L, XL, X, IX, V, IV, I  
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero HOLDS  
            SET count TO num DIVIDED BY element at position i OF val  
            FOR each iteration FROM one TO count  
                APPEND element at position i OF syms TO roman_numeral  
                DECREMENT num BY element at position i OF val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
