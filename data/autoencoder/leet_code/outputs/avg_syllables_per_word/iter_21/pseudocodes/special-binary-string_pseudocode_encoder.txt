CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            
            FOR each index i AND character char IN s  
                IF char EQUALS one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                
                IF count EQUALS zero  
                    APPEND concatenation of one PLUS CALL dfs WITH PARAMETERS substring from position start PLUS one TO position i MINUS one PLUS zero TO specials  
                    SET start TO i PLUS one  
                END IF  
            END FOR  
            
            CALL sort ON specials IN reverse lexicographical order  
            SET result TO concatenation of all elements in specials  
            RETURN result  
        END FUNCTION  
        
        RETURN CALL dfs WITH PARAMETERS s  
    END FUNCTION  
END CLASS
