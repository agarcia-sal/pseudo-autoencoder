CLASS Solution  
    FUNCTION racecar(target)  
        CALL initialize_queue WITH PARAMETERS initial_position ZERO initial_speed ONE initial_steps ZERO  
        SET queue TO the initialized queue  
        CALL initialize_visited_set WITH PARAMETERS initial_position ZERO initial_speed ONE  
        SET visited TO the initialized set  
        
        WHILE the queue is NOT empty  
            CALL remove_first_element FROM queue  
            SET position TO the first element of the removed tuple  
            SET speed TO the second element of the removed tuple  
            SET steps TO the third element of the removed tuple  
            
            IF position EQUALS target  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple of next_position and next_speed is NOT IN visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                CALL add tuple of next_position and next_speed TO visited  
                CALL append tuple of next_position next_speed and steps PLUS one TO queue  
            END IF  
            
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple of position and next_speed is NOT IN visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                CALL add tuple of position and next_speed TO visited  
                CALL append tuple of position next_speed and steps PLUS one TO queue  
            END IF  
            
        END WHILE  
    END FUNCTION  
END CLASS
