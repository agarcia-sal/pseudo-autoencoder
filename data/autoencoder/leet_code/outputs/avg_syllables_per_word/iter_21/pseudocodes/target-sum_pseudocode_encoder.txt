CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums THEN  
                IF current_sum EQUALS target THEN  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the TUPLE of index AND current_sum IS IN memo THEN  
                RETURN the ELEMENT at the TUPLE of index AND current_sum OF memo  
            END IF  
            SET positive TO CALL dfs WITH PARAMETERS index PLUS one, current_sum PLUS the ELEMENT at position index OF nums, memo  
            SET negative TO CALL dfs WITH PARAMETERS index PLUS one, current_sum MINUS the ELEMENT at position index OF nums, memo  
            SET the ELEMENT at the TUPLE of index AND current_sum OF memo TO positive PLUS negative  
            RETURN the ELEMENT at the TUPLE of index AND current_sum OF memo  
        END FUNCTION  
        RETURN CALL dfs WITH PARAMETERS zero, zero, an EMPTY DICTIONARY  
    END FUNCTION  
END CLASS
