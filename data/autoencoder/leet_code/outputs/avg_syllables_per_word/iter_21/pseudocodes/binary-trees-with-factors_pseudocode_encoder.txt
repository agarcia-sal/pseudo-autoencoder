CLASS Solution
    FUNCTION numFactoredBinaryTrees(arr)
        SET MOD TO one billion PLUS seven
        CALL arr.sort
        SET dp TO CALL create_dictionary_with_keys_from_arr_and_values_one
        FOR index i FROM zero TO the LENGTH OF arr MINUS one
            SET num TO element at position i OF arr
            FOR index j FROM zero TO i MINUS one
                IF the remainder of num DIVIDED BY element at position j OF arr EQUALS zero THEN
                    SET right_child TO num DIVIDED BY element at position j OF arr
                    IF right_child IS A KEY IN dp THEN
                        SET dp[num] TO the REMAINDER OF dp[num] PLUS dp[element at position j OF arr] MULTIPLIED BY dp[right_child] DIVIDED BY MOD
                    END IF
                END IF
            END FOR
        END FOR
        SET total_trees TO the SUM OF all values IN dp
        SET result TO the REMAINDER OF total_trees DIVIDED BY MOD
        RETURN result
    END FUNCTION
END CLASS
