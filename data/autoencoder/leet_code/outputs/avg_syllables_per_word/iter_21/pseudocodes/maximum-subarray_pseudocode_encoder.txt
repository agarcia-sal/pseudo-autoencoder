CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO element at position zero OF nums  
        SET current_sum TO element at position zero OF nums  
        FOR each num IN the subsection of nums starting from position one TO the end  
            IF num IS GREATER THAN current_sum PLUS num  
                SET current_sum TO num  
            ELSE  
                SET current_sum TO current_sum PLUS num  
            END IF  
            IF max_sum IS LESS THAN current_sum  
                SET max_sum TO current_sum  
            END IF  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
