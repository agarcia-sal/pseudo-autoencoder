CLASS Solution  
    FUNCTION getSum(a integer, b integer) RETURNS integer  
        SET MAX TO the maximum thirty-two bits integer value  
        SET mask TO the mask for thirty-two bits  

        WHILE b NOT EQUALS zero HOLDS  
            SET carry TO the result of bitwise AND of a AND b SHIFTED LEFT BY one position  
            SET a TO the result of bitwise XOR of a AND b ANDed WITH mask  
            SET b TO the result of bitwise AND of carry AND mask  
        END WHILE  

        IF a IS LESS THAN OR EQUAL TO MAX THEN  
            RETURN a  
        ELSE  
            RETURN the bitwise complement of the result of bitwise XOR of a AND mask  
        END IF  
    END FUNCTION  
END CLASS
