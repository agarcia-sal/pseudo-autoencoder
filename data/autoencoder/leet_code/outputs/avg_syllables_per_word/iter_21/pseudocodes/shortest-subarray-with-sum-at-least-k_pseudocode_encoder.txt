CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO a list containing a single element zero  
        FOR each num IN nums  
            SET last_element TO element at the last position OF prefix_sum  
            APPEND last_element PLUS num TO prefix_sum  
        END FOR  
        
        ASSIGN reference TO an empty double-ended queue TO dq  
        SET min_length TO infinity  
        
        FOR each index i AND element current_sum IN prefix_sum WITH enumeration  
            WHILE dq is not empty AND current_sum MINUS element at position at the front of dq IN prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET candidate_length TO i MINUS element removed from the front of dq  
                SET min_length TO the smaller BETWEEN min_length AND candidate_length  
            END WHILE  
            
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO element at position at the last of dq IN prefix_sum  
                REMOVE element from the back of dq  
            END WHILE  
            
            APPEND i TO the back of dq  
        END FOR  
        
        IF min_length IS NOT EQUAL TO infinity  
            RETURN min_length  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
