CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS not PRESENT  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETERS values AT position zero  
    SET i TO one  
    CALL deque FUNCTION TO CREATE queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO NEW TreeNode WITH PARAMETERS element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO NEW TreeNode WITH PARAMETERS element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left AND q.left AND CALL is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST(self, root)  
        SET self.max_sum TO zero  
        
        FUNCTION helper(node)  
            IF node IS none  
                RETURN true, zero, positive infinity, negative infinity  
            END IF  
            
            CALL helper WITH PARAMETERS node.left AND ASSIGN RESULTS TO left_is_bst, left_sum, left_min, left_max  
            CALL helper WITH PARAMETERS node.right AND ASSIGN RESULTS TO right_is_bst, right_sum, right_min, right_max  
            
            IF left_is_bst IS true AND right_is_bst IS true AND left_max LESS THAN node.val AND node.val LESS THAN right_min  
                SET current_sum TO left_sum PLUS right_sum PLUS node.val  
                SET current_min TO THE SMALLER OF node.val AND left_min  
                SET current_max TO THE LARGER OF node.val AND right_max  
                SET self.max_sum TO THE GREATER OF self.max_sum AND current_sum  
                RETURN true, current_sum, current_min, current_max  
            END IF  
            
            RETURN false, zero, positive infinity, negative infinity  
        END FUNCTION  
        
        CALL helper WITH PARAMETERS root  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
