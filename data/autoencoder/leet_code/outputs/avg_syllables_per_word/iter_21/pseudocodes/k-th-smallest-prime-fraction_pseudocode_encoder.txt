CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an EMPTY list  
        FOR index i FROM zero TO n MINUS one MINUS one  
            CALL heappush ON min_heap WITH PARAMETERS the RATIO of element at position i OF arr DIVIDED BY element at position n MINUS one OF arr, i, n MINUS one  
        END FOR  
        FOR count FROM zero TO k MINUS one MINUS one  
            CALL heappop ON min_heap AND ASSIGN the RESULT TO variables discard, i, j  
            IF j MINUS one IS GREATER THAN i  
                CALL heappush ON min_heap WITH PARAMETERS the RATIO of element at position i OF arr DIVIDED BY element at position j MINUS one OF arr, i, j MINUS one  
            END IF  
        END FOR  
        CALL heappop ON min_heap AND ASSIGN the RESULT TO variables discard, i, j  
        RETURN a list CONTAINING element at position i OF arr AND element at position j OF arr  
    END FUNCTION  
END CLASS
