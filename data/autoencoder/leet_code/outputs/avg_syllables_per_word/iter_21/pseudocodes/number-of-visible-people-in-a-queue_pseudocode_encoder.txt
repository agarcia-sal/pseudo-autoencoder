CLASS Solution  
    FUNCTION canSeePersonsCount(heights)  
        SET n TO the LENGTH OF heights  
        SET answer TO a list of zeros WITH LENGTH n  
        SET stack TO an empty list  
        FOR i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position the LAST ELEMENT OF stack OF heights IS LESS THAN the element at position i OF heights HOLDS  
                SET popped_index TO the REMOVED LAST ELEMENT OF stack  
                INCREMENT the element at position popped_index OF answer BY one  
            END WHILE  
            IF the LENGTH OF stack IS GREATER THAN zero  
                INCREMENT the element at position the LAST ELEMENT OF stack OF answer BY one  
            END IF  
            APPEND i TO stack  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
