CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO a new empty mapping from each course in the range from zero TO numCourses MINUS one TO an empty list  
        SET in_degrees TO a new list of size numCourses with all elements set TO zero  
        
        FOR each pair OF dest and src IN prerequisites  
            APPEND dest TO the list at key src IN adj_list  
            INCREMENT element at position dest OF in_degrees BY one  
        END FOR  
        
        SET queue TO a new empty double-ended queue initialized with all courses from zero TO numCourses MINUS one WHERE the element at that course's position IN in_degrees EQUALS zero  
        SET topological_order TO a new empty list  
        
        WHILE queue IS NOT empty  
            SET current TO the element removed from the left of queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN the list at current key IN adj_list  
                DECREMENT element at position neighbor OF in_degrees BY one  
                IF element at position neighbor OF in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the length of topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN a new empty list  
        END IF  
    END FUNCTION  
END CLASS
