CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO CALL convert_list_to_set WITH PARAMETERS wordDict  
        SET dp TO CALL initialize_boolean_list WITH PARAMETERS LENGTH OF s PLUS one  
        SET element at position zero OF dp TO True  
        FOR index i FROM one TO LENGTH OF s PLUS one  
            FOR index j FROM zero TO i MINUS one  
                IF element at position j OF dp EQUALS True AND substring from position j TO i MINUS one OF s IS IN wordSet  
                    SET element at position i OF dp TO True  
                    BREAK inner FOR loop  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position LENGTH OF s OF dp  
    END FUNCTION  

    FUNCTION convert_list_to_set(wordDict)  
        RETURN a set containing all elements FROM wordDict  
    END FUNCTION  

    FUNCTION initialize_boolean_list(size)  
        CREATE list_of_booleans OF length size WITH all elements SET TO False  
        RETURN list_of_booleans  
    END FUNCTION  
END CLASS
