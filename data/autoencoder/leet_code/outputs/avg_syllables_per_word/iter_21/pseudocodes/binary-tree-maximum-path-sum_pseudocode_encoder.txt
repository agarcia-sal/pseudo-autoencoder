CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH val EQUALS element at position zero OF values  
    SET i TO one  
    SET queue TO NEW deque  
    CALL queue APPEND WITH PARAMETERS root  
    WHILE queue IS NOT empty  
        SET node TO CALL queue popleft WITH NO PARAMETERS  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO NEW TreeNode WITH val EQUALS element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.left  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO NEW TreeNode WITH val EQUALS element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left, q.left AND CALL is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum(root)  
        FUNCTION helper(node)  
            IF node IS None  
                RETURN negative infinity, negative infinity  
            END IF  
            SET left_max, left_gain TO CALL helper WITH PARAMETERS node.left  
            SET right_max, right_gain TO CALL helper WITH PARAMETERS node.right  
            SET current_max TO node.val PLUS the GREATER OF left_gain AND zero PLUS the GREATER OF right_gain AND zero  
            SET current_gain TO node.val PLUS the GREATER OF left_gain, right_gain, AND zero  
            SET global_max TO the GREATER OF left_max, right_max, AND current_max  
            RETURN global_max, current_gain  
        END FUNCTION  
        RETURN the FIRST element OF CALL helper WITH PARAMETERS root  
    END FUNCTION  
END CLASS
