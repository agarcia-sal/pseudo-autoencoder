CLASS Solution  
    FUNCTION minimumMountainRemovals(nums)  
        SET n TO the LENGTH OF nums  
        
        SET lis TO a list of size n FILLED WITH the value one  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF element at position i OF nums IS GREATER THAN element at position j OF nums  
                    SET lis at position i TO the GREATER OF the value at position i OF lis AND the value at position j OF lis PLUS one  
                END IF  
            END FOR  
        END FOR  
        
        SET lds TO a list of size n FILLED WITH the value one  
        FOR i FROM n MINUS one DOWNTO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                IF element at position i OF nums IS GREATER THAN element at position j OF nums  
                    SET lds at position i TO the GREATER OF the value at position i OF lds AND the value at position j OF lds PLUS one  
                END IF  
            END FOR  
        END FOR  
        
        SET max_mountain_length TO zero  
        FOR i FROM one TO n MINUS two  
            IF the value at position i OF lis IS GREATER THAN one AND the value at position i OF lds IS GREATER THAN one  
                SET max_mountain_length TO the GREATER OF max_mountain_length AND the value at position i OF lis PLUS the value at position i OF lds MINUS one  
            END IF  
        END FOR  
        
        RETURN n MINUS max_mountain_length  
    END FUNCTION  
END CLASS
