CLASS Solution  
    FUNCTION minimumTime(s)  
        SET n TO the LENGTH OF s  
        SET left_costs TO a LIST OF zero WITH LENGTH n  
        SET right_costs TO a LIST OF zero WITH LENGTH n  
        
        IF the element at position zero OF s EQUALS the CHARACTER one  
            SET the element at position zero OF left_costs TO one  
        ELSE  
            SET the element at position zero OF left_costs TO zero  
        END IF  
        
        FOR i FROM one TO n MINUS one  
            IF the element at position i OF s EQUALS the CHARACTER zero  
                SET the element at position i OF left_costs TO the element at position i MINUS one OF left_costs  
            ELSE  
                SET the element at position i OF left_costs TO the MINIMUM OF i PLUS one AND the element at position i MINUS one OF left_costs PLUS two  
            END IF  
        END FOR  
        
        IF the element at position n MINUS one OF s EQUALS the CHARACTER one  
            SET the element at position n MINUS one OF right_costs TO one  
        ELSE  
            SET the element at position n MINUS one OF right_costs TO zero  
        END IF  
        
        FOR i FROM n MINUS two DOWN TO zero STEP minus one  
            IF the element at position i OF s EQUALS the CHARACTER zero  
                SET the element at position i OF right_costs TO the element at position i PLUS one OF right_costs  
            ELSE  
                SET the element at position i OF right_costs TO the MINIMUM OF n MINUS i AND the element at position i PLUS one OF right_costs PLUS two  
            END IF  
        END FOR  
        
        SET min_time TO the MINIMUM OF the element at position n MINUS one OF left_costs AND the element at position zero OF right_costs  
        
        FOR i FROM zero TO n MINUS two  
            SET min_time TO the MINIMUM OF min_time AND the element at position i OF left_costs PLUS the element at position i PLUS one OF right_costs  
        END FOR  
        
        RETURN min_time  
    END FUNCTION  
END CLASS
