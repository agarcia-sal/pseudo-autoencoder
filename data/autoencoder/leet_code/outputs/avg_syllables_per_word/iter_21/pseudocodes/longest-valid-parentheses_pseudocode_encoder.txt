CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a new list containing negative one as the single element  
        SET max_length TO zero  
        FOR each index i AND character char IN s  
            IF character char EQUALS open parenthesis  
                APPEND index i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF stack IS empty  
                    APPEND index i TO stack  
                ELSE  
                    SET current_length TO index i MINUS the last element in stack  
                    SET max_length TO the GREATER OF max_length AND current_length  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
