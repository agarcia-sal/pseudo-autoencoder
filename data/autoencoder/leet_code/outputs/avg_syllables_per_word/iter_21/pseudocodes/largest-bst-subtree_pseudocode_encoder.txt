CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode initialized WITH the element at position zero OF values  
    SET i TO one  
    SET queue TO a new double-ended queue  
    CALL append WITH PARAMETERS root ON queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front OF queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT None  
            SET node.left TO new TreeNode initialized WITH the element at position i OF values  
            CALL append WITH PARAMETERS node.left ON queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT None  
            SET node.right TO new TreeNode initialized WITH the element at position i OF values  
            CALL append WITH PARAMETERS node.right ON queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF value at p IS NOT EQUAL TO value at q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree WITH PARAMETERS value at p.left AND value at q.left AND the result of calling is_same_tree WITH PARAMETERS value at p.right AND value at q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree(self, root)  
        
        FUNCTION dfs(node)  
            IF node IS None  
                RETURN positive infinity, negative infinity, zero  
            END IF  
            
            SET left_min, left_max, left_size TO the result of calling dfs WITH PARAMETERS value at node.left  
            SET right_min, right_max, right_size TO the result of calling dfs WITH PARAMETERS value at node.right  
            
            IF left_max LESS THAN value at node AND value at node LESS THAN right_min  
                RETURN the smaller of value at node AND left_min, the greater of value at node AND right_max, left_size PLUS right_size PLUS one  
            END IF  
            
            RETURN negative infinity, positive infinity, the greater of left_size AND right_size  
        END FUNCTION  
        
        RETURN the third element of the tuple returned by dfs called WITH PARAMETERS root  
    END FUNCTION  
END CLASS
