CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO CALL sorted WITH PARAMETERS a collection of tuples each containing wage element DIVIDED BY quality element AND quality element FROM simultaneous iteration over wage AND quality  
        SET max_heap TO an empty list  
        SET total_quality TO zero  
        SET min_cost TO a floating point representation of positive infinity  
        FOR each ratio AND q IN workers  
            CALL heappush WITH PARAMETERS max_heap AND negative q  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap IS GREATER THAN k  
                INCREMENT total_quality BY CALL heappop WITH PARAMETERS max_heap  
            END IF  
            IF the LENGTH OF max_heap IS EQUAL TO k  
                SET min_cost TO the LESSER OF min_cost AND total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
