CLASS Solution  
    FUNCTION findLength(nums1, nums2)  
        SET dp TO the result of CREATE_TWO_DIMENSIONAL_LIST with PARAMETERS the LENGTH OF nums1 PLUS one AND the LENGTH OF nums2 PLUS one  
        SET max_length TO zero  
        FOR i FROM one TO the LENGTH OF nums1  
            FOR j FROM one TO the LENGTH OF nums2  
                IF element at position i MINUS one OF nums1 EQUALS element at position j MINUS one OF nums2  
                    SET element at position i AND j OF dp TO element at position i MINUS one AND j MINUS one OF dp PLUS one  
                    IF element at position i AND j OF dp GREATER THAN max_length  
                        SET max_length TO element at position i AND j OF dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS  

FUNCTION CREATE_TWO_DIMENSIONAL_LIST(row_count, column_count)  
    SET result TO empty list  
    FOR each number FROM one TO row_count  
        SET row TO empty list  
        FOR each number FROM one TO column_count  
            APPEND zero TO row  
        END FOR  
        APPEND row TO result  
    END FOR  
    RETURN result  
END FUNCTION
