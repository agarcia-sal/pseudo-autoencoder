CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO CALL split_string_by_newline WITH PARAMETERS input  
        SET stack TO empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO CALL count_tab_characters WITH PARAMETERS line  
            SET name TO substring from position depth TO the end OF line  
            
            WHILE the LENGTH OF stack GREATER THAN depth HOLDS  
                CALL remove_last_element FROM stack  
            END WHILE  
            
            IF the LENGTH OF stack GREATER THAN zero THEN  
                SET current_length TO the element at the last position OF stack PLUS the LENGTH OF name PLUS one  
            ELSE  
                SET current_length TO the LENGTH OF name  
            END IF  
            
            IF the character period IS PRESENT IN name THEN  
                SET max_length TO the GREATER VALUE BETWEEN max_length AND current_length  
            ELSE  
                CALL append_element current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
