CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        CALL defaultdict WITH PARAMETERS list AND ASSIGN reference TO anagrams  
        FOR each element s IN strs  
            CALL sorted WITH PARAMETERS s AND ASSIGN reference TO sorted_characters  
            CALL join WITH PARAMETERS sorted_characters AND ASSIGN reference TO sorted_str  
            CALL append WITH PARAMETERS s ON the list AT key sorted_str OF anagrams  
        END FOR  
        CALL values ON anagrams AND CONVERT result TO list AND RETURN the result  
    END FUNCTION  
END CLASS
