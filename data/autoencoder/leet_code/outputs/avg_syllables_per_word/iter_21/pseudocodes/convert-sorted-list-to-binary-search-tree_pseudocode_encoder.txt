CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET head TO NEW ListNode WITH val AS element at position zero OF values  
    SET p TO head  
    FOR each val IN elements from position one TO the end OF values  
        SET node TO NEW ListNode WITH val AS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN (element val OF p1 EQUALS element val OF p2) AND CALL is_same_list WITH PARAMETERS element next OF p1 AND element next OF p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH val AS element at position zero OF values  
    SET i TO one  
    SET queue TO NEW deque  
    CALL queue.append WITH PARAMETERS root  
    WHILE queue IS NOT empty  
        SET node TO CALL queue.popleft WITH NO PARAMETERS  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values NOT EQUALS None THEN  
            SET node.left TO NEW TreeNode WITH val AS element at position i OF values  
            CALL queue.append WITH PARAMETERS node.left  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values NOT EQUALS None THEN  
            SET node.right TO NEW TreeNode WITH val AS element at position i OF values  
            CALL queue.append WITH PARAMETERS node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF element val OF p NOT EQUALS element val OF q THEN  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS element left OF p AND element left OF q AND CALL is_same_tree WITH PARAMETERS element right OF p AND element right OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST(self, head)  
        IF head EQUALS None THEN  
            RETURN None  
        END IF  
      
        FUNCTION findMiddle(left, right)  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUALS right AND element next OF fast NOT EQUALS right  
                SET slow TO element next OF slow  
                SET fast TO element next OF element next OF fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  
      
        FUNCTION convertListToBST(left, right)  
            IF left EQUALS right THEN  
                RETURN None  
            END IF  
          
            SET mid TO CALL findMiddle WITH PARAMETERS left AND right  
          
            SET node TO NEW TreeNode WITH val AS element val OF mid  
            SET node.left TO CALL convertListToBST WITH PARAMETERS left AND mid  
            SET node.right TO CALL convertListToBST WITH PARAMETERS element next OF mid AND right  
            RETURN node  
        END FUNCTION  
      
        RETURN CALL convertListToBST WITH PARAMETERS head AND None  
    END FUNCTION  
END CLASS
