CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode WITH val EQUALS element at position zero OF values  
    SET p TO head  
    FOR each val IN elements FROM position one TO THE END OF values  
        SET node TO A NEW ListNode WITH val EQUALS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN p1.val EQUALS p2.val AND CALL is_same_list WITH PARAMETERS p1.next AND p2.next  
END FUNCTION

CLASS Solution  
    FUNCTION sortList(self, head)  
        IF head EQUALS None OR head.next EQUALS None  
            RETURN head  
        END IF  

        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO element at position one OF head.next IF head.next EXISTS  
            WHILE fast IS NOT None AND fast.next IS NOT None  
                SET slow TO slow.next  
                SET fast TO element at position two OF fast.next.next IF fast.next.next EXISTS  
            END WHILE  
            SET mid TO slow.next  
            SET slow.next TO None  
            RETURN head, mid  
        END FUNCTION  

        FUNCTION merge(l1, l2)  
            SET dummy TO A NEW ListNode  
            SET current TO dummy  
            WHILE l1 IS NOT None AND l2 IS NOT None  
                IF l1.val LESS THAN l2.val  
                    SET current.next TO l1  
                    SET l1 TO l1.next  
                ELSE  
                    SET current.next TO l2  
                    SET l2 TO l2.next  
                END IF  
                SET current TO current.next  
            END WHILE  
            SET current.next TO l1 IF l1 IS NOT None ELSE l2  
            RETURN dummy.next  
        END FUNCTION  

        SET left, right TO CALL split WITH PARAMETER head  
        SET left TO CALL self.sortList WITH PARAMETER left  
        SET right TO CALL self.sortList WITH PARAMETER right  
        RETURN CALL merge WITH PARAMETERS left AND right  
    END FUNCTION  
END CLASS
