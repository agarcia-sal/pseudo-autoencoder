CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values HOLDS no ELEMENTS THEN  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETERS the element at position zero OF values  
    SET i TO one  
    SET queue TO NEW double-ended queue  
    APPEND root TO queue  
    WHILE queue HOLDS ELEMENTS  
        SET node TO the FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none THEN  
            SET node.left TO NEW TreeNode WITH PARAMETERS the element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none THEN  
            SET node.right TO NEW TreeNode WITH PARAMETERS the element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val THEN  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left AND q.left AND CALL is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView(self, root)  
        IF root IS none THEN  
            RETURN an EMPTY list  
        END IF  
        SET right_side_view TO an EMPTY list  
        SET queue TO NEW double-ended queue INITIALIZED WITH ELEMENT root  
        WHILE queue HOLDS ELEMENTS  
            SET level_length TO the LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO the FIRST ELEMENT REMOVED FROM queue  
                IF i EQUALS level_length MINUS one THEN  
                    APPEND node.val TO right_side_view  
                END IF  
                IF node.left IS NOT none THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT none THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
