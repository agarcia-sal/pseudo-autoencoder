CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero OF values  
    SET i TO one  
    SET queue TO A NEW deque  
    CALL queue APPEND WITH PARAMETERS root  
    WHILE queue IS NOT empty  
        SET node TO THE FIRST element REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO A NEW TreeNode INITIALIZED WITH element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.left  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO A NEW TreeNode INITIALIZED WITH element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left, q.left AND CALL is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION largestValues(self, root)  
        IF root IS none  
            RETURN AN EMPTY list  
        END IF  
        SET queue TO A NEW deque INITIALIZED WITH root AS THE ONLY ELEMENT  
        SET largest_values TO AN EMPTY list  
        WHILE queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO NEGATIVE infinity  
            FOR index FROM zero TO level_size MINUS one  
                SET node TO THE FIRST element REMOVED FROM queue  
                IF node.val IS GREATER THAN max_value  
                    SET max_value TO node.val  
                END IF  
                IF node.left IS NOT none  
                    CALL queue APPEND WITH PARAMETERS node.left  
                END IF  
                IF node.right IS NOT none  
                    CALL queue APPEND WITH PARAMETERS node.right  
                END IF  
            END FOR  
            CALL largest_values APPEND WITH PARAMETERS max_value  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
