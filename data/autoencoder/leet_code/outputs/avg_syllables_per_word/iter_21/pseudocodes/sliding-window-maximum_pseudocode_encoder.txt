CLASS Solution  
    FUNCTION maxSlidingWindow(nums, k)  
        IF nums IS empty OR k EQUALS zero  
            RETURN an empty list  
        END IF  
        SET deq TO an empty double-ended queue  
        SET result TO an empty list  
        FOR index i FROM zero TO the length of nums MINUS one  
            IF deq IS NOT empty AND the element at position zero OF deq IS LESS THAN i MINUS k PLUS one  
                REMOVE the element from the front of deq  
            END IF  
            WHILE deq IS NOT empty AND the element at the position at the end OF deq in nums IS LESS THAN the element at position i OF nums  
                REMOVE the element from the end of deq  
            END WHILE  
            APPEND i TO the end of deq  
            IF i IS GREATER THAN OR EQUAL TO k MINUS one  
                APPEND the element at the position at the front OF deq in nums TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
