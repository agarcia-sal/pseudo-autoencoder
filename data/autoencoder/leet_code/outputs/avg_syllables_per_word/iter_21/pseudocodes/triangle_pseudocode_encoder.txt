CLASS Solution  
    FUNCTION minimumTotal(triangle)  
        FOR row FROM the LENGTH OF triangle MINUS two DOWN TO zero  
            FOR col FROM zero TO the LENGTH OF element at position row OF triangle MINUS one  
                SET minimum_of_adjacent TO the smaller VALUE BETWEEN element at position col OF element at position row PLUS one OF triangle AND element at position col PLUS one OF element at position row PLUS one OF triangle  
                INCREMENT element at position col OF element at position row OF triangle BY minimum_of_adjacent  
            END FOR  
        END FOR  
        RETURN element at position zero OF element at position zero OF triangle  
    END FUNCTION  
END CLASS
