CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a new list of LENGTH n FILLED WITH negative one  
        SET stack TO a new empty list  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO i MODULO n  
            WHILE stack IS NOT empty AND element at position the LAST element OF stack OF nums IS LESS THAN element at position index OF nums  
                SET popped_index TO REMOVE the LAST element FROM stack  
                SET element at position popped_index OF result TO element at position index OF nums  
            END WHILE  
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
