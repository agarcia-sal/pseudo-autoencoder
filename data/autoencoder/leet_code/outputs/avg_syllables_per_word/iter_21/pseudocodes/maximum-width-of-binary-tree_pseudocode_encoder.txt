CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO CALL TreeNode WITH PARAMETERS element at position zero OF values  
    SET i TO one  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left AND q.left AND CALL is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree(self, root)  
        IF root IS None  
            RETURN zero  
        END IF  
        SET queue TO new deque CONTAINING tuple of root AND zero  
        SET max_width TO zero  
        WHILE queue IS NOT empty  
            SET level_length TO the LENGTH OF queue  
            SET first_index TO the second element OF the first element OF queue  
            FOR each index FROM zero TO level_length MINUS one  
                SET node, index_value TO REMOVE first element FROM queue  
                IF index IS EQUAL TO level_length MINUS one  
                    IF max_width IS LESS THAN index_value MINUS first_index PLUS one  
                        SET max_width TO index_value MINUS first_index PLUS one  
                    END IF  
                END IF  
                IF node.left IS NOT None  
                    APPEND tuple of node.left AND index_value MULTIPLIED BY two TO queue  
                END IF  
                IF node.right IS NOT None  
                    APPEND tuple of node.right AND index_value MULTIPLIED BY two PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
