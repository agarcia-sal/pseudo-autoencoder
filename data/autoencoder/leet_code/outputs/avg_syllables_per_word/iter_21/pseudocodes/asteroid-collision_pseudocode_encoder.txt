CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO an empty list  
        
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND asteroid IS LESS THAN zero AND the element at position the LENGTH OF stack MINUS one OF stack IS GREATER THAN zero HOLDS  
                IF the element at position the LENGTH OF stack MINUS one OF stack IS LESS THAN the NEGATION OF asteroid THEN  
                    REMOVE the element at position the LENGTH OF stack MINUS one OF stack FROM stack  
                    CONTINUE to the next iteration of the WHILE loop  
                ELSE IF the element at position the LENGTH OF stack MINUS one OF stack EQUALS the NEGATION OF asteroid THEN  
                    REMOVE the element at position the LENGTH OF stack MINUS one OF stack FROM stack  
                END IF  
                BREAK from the WHILE loop  
            END WHILE  
            IF the WHILE loop did not break THEN  
                APPEND asteroid TO stack  
            END IF  
        END FOR  
        
        RETURN stack  
    END FUNCTION  
END CLASS
