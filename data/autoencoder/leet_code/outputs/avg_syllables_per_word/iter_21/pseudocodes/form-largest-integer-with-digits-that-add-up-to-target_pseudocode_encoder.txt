CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        SET cost_to_digit TO result of CALLING conversion FROM cost  
        SET dp TO result of CALLING initialize_dp_array WITH PARAMETERS target  
        FOR t FROM one TO target  
            FOR each c, d IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND element at position t MINUS c OF dp NOT EQUALS string representing negative one THEN  
                    SET candidate TO concatenation of d FOLLOWED BY element at position t MINUS c OF dp  
                    IF element at position t OF dp EQUALS string representing negative one OR length of candidate GREATER THAN length of element at position t OF dp OR (length of candidate EQUALS length of element at position t OF dp AND candidate GREATER THAN element at position t OF dp) THEN  
                        SET element at position t OF dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        IF element at position target OF dp NOT EQUALS string representing negative one THEN  
            RETURN element at position target OF dp  
        ELSE  
            RETURN string representing zero  
        END IF  
    END FUNCTION  

    FUNCTION conversion(cost)  
        CREATE empty dictionary called result  
        FOR index FROM zero TO length of cost MINUS one  
            SET key TO element at position index OF cost  
            SET value TO string of index PLUS one  
            ASSIGN key TO value IN result  
        END FOR  
        RETURN result  
    END FUNCTION  

    FUNCTION initialize_dp_array(target)  
        CREATE list called dp_array WITH element string empty FOLLOWED BY target elements each being string representing negative one  
        RETURN dp_array  
    END FUNCTION  
END CLASS
