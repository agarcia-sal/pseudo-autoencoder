CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s IS GREATER THAN the LENGTH OF t  
            RETURN CALL isOneEditDistance WITH PARAMETERS t AND s  
        END IF  
        
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        
        IF diff IS GREATER THAN one  
            RETURN False  
        END IF  
        
        IF diff EQUALS zero  
            SET difference_count TO zero  
            FOR each pair OF characters c1 AND c2 IN corresponding positions OF s AND t  
                IF c1 IS NOT EQUAL TO c2  
                    INCREMENT difference_count BY one  
                END IF  
            END FOR  
            RETURN whether difference_count EQUALS one  
        ELSE  
            FOR index FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position index OF s IS NOT EQUAL TO the element at position index OF t  
                    RETURN whether the substring from position index TO the END OF s IS EQUAL TO the substring from position index PLUS one TO the END OF t  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
