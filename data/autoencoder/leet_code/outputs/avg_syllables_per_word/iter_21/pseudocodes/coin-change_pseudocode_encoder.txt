CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        SET dp TO the result of calling create_list_with_size AND PARAMETERS amount PLUS one AND initial value amount PLUS one  
        SET element at position zero OF dp TO zero  
        FOR a FROM one TO amount  
            FOR each coin IN coins  
                IF a MINUS coin IS GREATER THAN OR EQUAL TO zero  
                    SET dp at position a TO the minimum OF dp at position a AND dp at position a MINUS coin PLUS one  
                END IF  
            END FOR  
        END FOR  
        IF dp at position amount IS NOT EQUAL TO amount PLUS one  
            RETURN dp at position amount  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
