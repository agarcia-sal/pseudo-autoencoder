CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO CALL Counter WITH PARAMETERS digits  
        SET total_sum TO the SUM OF elements IN digits  
        SET remainder TO total_sum MODULO three  
  
        IF remainder EQUALS one  
            FOR each digit IN the list containing one FOUR and SEVEN  
                IF the element at position digit OF count IS GREATER THAN zero  
                    DECREMENT the element at position digit OF count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            IF the loop was not broken  
                FOR each digit IN the list containing two FIVE and EIGHT  
                    IF the element at position digit OF count IS GREATER THAN one  
                        DECREMENT the element at position digit OF count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
  
        IF remainder EQUALS two  
            FOR each digit IN the list containing two FIVE and EIGHT  
                IF the element at position digit OF count IS GREATER THAN zero  
                    DECREMENT the element at position digit OF count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            IF the loop was not broken  
                FOR each digit IN the list containing one FOUR and SEVEN  
                    IF the element at position digit OF count IS GREATER THAN one  
                        DECREMENT the element at position digit OF count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END IF  
        END IF  
  
        SET result TO an empty list  
        FOR digit FROM nine DOWN TO zero  
            APPEND the string representation of digit REPEATED the element at position digit OF count TIMES TO result  
        END FOR  
  
        SET final_number TO the concatenation of all elements IN result  
  
        IF final_number IS empty OR the element at position zero OF final_number EQUALS zero  
            RETURN the string zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
