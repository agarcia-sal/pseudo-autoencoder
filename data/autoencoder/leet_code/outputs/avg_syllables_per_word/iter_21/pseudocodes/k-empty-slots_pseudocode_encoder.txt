CLASS Solution  
    FUNCTION kEmptySlots(bulbs, k)  
        SET bloom_days TO an empty list  
        
        FOR day FROM one TO the LENGTH OF bulbs  
            SET position TO element at position day MINUS one OF bulbs  
            
            CALL bisect_left WITH PARAMETERS bloom_days AND position AND ASSIGN result TO i  
            
            IF i GREATER THAN zero AND position MINUS element at position i MINUS one OF bloom_days MINUS one EQUALS k  
                RETURN day  
            END IF  
            
            IF i LESS THAN the LENGTH OF bloom_days AND element at position i OF bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            
            INSERT position AT position i INTO bloom_days  
        END FOR  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
