CLASS Solution  
    FUNCTION findMaxAverage(nums, k)  
        FUNCTION canFindAverage(mid)  
            SET prefix_sums TO a list of zeros WITH length EQUAL TO the length of nums PLUS one  
            SET min_prefix_sum TO a list of zeros WITH length EQUAL TO the length of nums PLUS one  
            
            FOR index FROM zero TO the length of nums MINUS one  
                SET element at position index PLUS one OF prefix_sums TO element at position index OF prefix_sums PLUS element at position index OF nums MINUS mid  
                
                IF index GREATER THAN OR EQUAL TO k  
                    SET element at position index PLUS one OF min_prefix_sum TO the minimum VALUE BETWEEN element at position index OF min_prefix_sum AND element at position index PLUS one MINUS k OF prefix_sums  
                ELSE  
                    SET element at position index PLUS one OF min_prefix_sum TO element at position index OF min_prefix_sum  
                END IF  
                
                IF index GREATER THAN OR EQUAL TO k MINUS one AND element at position index PLUS one OF prefix_sums MINUS element at position index PLUS one OF min_prefix_sum GREATER THAN OR EQUAL TO zero  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET low TO the minimum VALUE IN nums  
        SET high TO the maximum VALUE IN nums  
        
        WHILE high MINUS low GREATER THAN 0 POINT 00001  
            SET mid TO low PLUS high DIVIDED BY two  
            IF CALL canFindAverage WITH PARAMETERS mid IS True  
                SET low TO mid  
            ELSE  
                SET high TO mid  
            END IF  
        END WHILE  
        
        RETURN low  
    END FUNCTION  
END CLASS
