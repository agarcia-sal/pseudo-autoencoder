CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO the LENGTH OF text  
        SET substrings TO an empty SET  

        FOR length FROM one TO the INTEGER DIVISION OF n BY two PLUS one  
            FOR start FROM zero TO n MINUS two TIMES length PLUS one  
                SET a TO the SUBSTRING from position start TO start PLUS length MINUS one OF text  
                SET b TO the SUBSTRING from position start PLUS length TO start PLUS two TIMES length MINUS one OF text  
                IF a EQUALS b  
                    ADD the CONCATENATION of a AND b TO substrings  
                END IF  
            END FOR  
        END FOR  

        RETURN the LENGTH OF substrings  
    END FUNCTION  
END CLASS
