CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND its count v IN cnt  
                IF v GREATER THAN 1  
                    DECREMENT cnt AT key c BY two  
                    CALL dfs WITH PARAMETERS c PLUS t PLUS c  
                    INCREMENT cnt AT key c BY two  
                END IF  
            END FOR  
        END FUNCTION  

        SET cnt TO the Counter object representing the count of each character IN s  
        SET mid TO empty string  
        FOR each character c AND its count v IN cnt  
            IF v MODULO two EQUALS one  
                IF mid IS NOT empty string  
                    RETURN empty list  
                END IF  
                SET mid TO c  
                DECREMENT cnt AT key c BY one  
            END IF  
        END FOR  
        SET ans TO empty list  
        CALL dfs WITH PARAMETERS mid  
        RETURN ans  
    END FUNCTION  
END CLASS
