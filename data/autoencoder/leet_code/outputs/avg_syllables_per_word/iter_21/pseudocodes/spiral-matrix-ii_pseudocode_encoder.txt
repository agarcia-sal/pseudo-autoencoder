CLASS Solution  
    FUNCTION generateMatrix(n)  
        CALL initialize_matrix WITH PARAMETERS n  
        SET matrix TO the RESULT OF initialize_matrix  
        
        SET top TO zero  
        SET bottom TO n MINUS one  
        SET left TO zero  
        SET right TO n MINUS one  
        
        SET num TO one  
        
        WHILE top LESS THAN OR EQUAL TO bottom AND left LESS THAN OR EQUAL TO right HOLDS  
            FOR i FROM left TO right  
                SET element at row top AND column i OF matrix TO num  
                INCREMENT num BY one  
            END FOR  
            INCREMENT top BY one  
            
            FOR i FROM top TO bottom  
                SET element at row i AND column right OF matrix TO num  
                INCREMENT num BY one  
            END FOR  
            DECREMENT right BY one  
            
            IF top LESS THAN OR EQUAL TO bottom THEN  
                FOR i FROM right DOWN TO left  
                    SET element at row bottom AND column i OF matrix TO num  
                    INCREMENT num BY one  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            
            IF left LESS THAN OR EQUAL TO right THEN  
                FOR i FROM bottom DOWN TO top  
                    SET element at row i AND column left OF matrix TO num  
                    INCREMENT num BY one  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        
        RETURN matrix  
    END FUNCTION  
END CLASS
