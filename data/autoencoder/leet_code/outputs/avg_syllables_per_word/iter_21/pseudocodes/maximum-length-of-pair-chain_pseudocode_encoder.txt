CLASS Solution  
    FUNCTION findLongestChain(pairs)  
        CALL pairs.sort WITH PARAMETERS key FUNCTION(element)  
            RETURN element at position one OF element  
        END FUNCTION  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each element IN pairs  
            SET start TO element at position zero OF element  
            SET end TO element at position one OF element  
            IF start GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
