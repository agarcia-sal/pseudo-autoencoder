CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO list obtained by converting n to string and then to list of characters  
        SET length TO the LENGTH OF digits  
          
        SET i TO length MINUS two  
        WHILE i IS GREATER THAN OR EQUAL TO zero AND element at position i OF digits IS GREATER THAN OR EQUAL TO element at position i PLUS one OF digits  
            DECREMENT i BY one  
        END WHILE  
          
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
          
        SET j TO length MINUS one  
        WHILE element at position j OF digits IS LESS THAN OR EQUAL TO element at position i OF digits  
            DECREMENT j BY one  
        END WHILE  
          
        SET temporary TO element at position i OF digits  
        SET element at position i OF digits TO element at position j OF digits  
        SET element at position j OF digits TO temporary  
          
        SET digits AFTER position i TO the REVERSE OF the subsequence from position i PLUS one TO the end OF digits  
          
        SET result TO integer converted from concatenating all elements OF digits  
          
        IF result IS GREATER THAN two RAISED TO THE THIRTY FIRST MINUS one  
            RETURN negative one  
        END IF  
          
        RETURN result  
    END FUNCTION  
END CLASS
