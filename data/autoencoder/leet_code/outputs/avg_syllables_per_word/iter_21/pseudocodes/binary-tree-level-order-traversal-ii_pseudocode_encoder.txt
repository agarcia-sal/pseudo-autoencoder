CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO CALL TreeNode WITH PARAMETERS element at position zero OF values  
    SET i TO one  
    SET queue TO CALL deque WITHOUT PARAMETERS  
    CALL queue.append WITH PARAMETERS root  
    WHILE queue IS NOT empty  
        SET node TO CALL queue.popleft WITHOUT PARAMETERS  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            CALL queue.append WITH PARAMETERS node.left  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            CALL queue.append WITH PARAMETERS node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left, q.left AND CALL is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom(self, root)  
        IF root IS None  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO CALL deque WITH PARAMETERS list CONTAINING root  
        WHILE queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO empty list  
            FOR each index FROM zero TO level_size MINUS one  
                SET node TO CALL queue.popleft WITHOUT PARAMETERS  
                APPEND node.val TO current_level  
                IF node.left IS NOT None  
                    CALL queue.append WITH PARAMETERS node.left  
                END IF  
                IF node.right IS NOT None  
                    CALL queue.append WITH PARAMETERS node.right  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN the REVERSED result  
    END FUNCTION  
END CLASS
