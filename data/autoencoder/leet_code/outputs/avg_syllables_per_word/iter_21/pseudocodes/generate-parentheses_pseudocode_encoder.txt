CLASS Solution  
    FUNCTION generateParenthesis(n)  
        FUNCTION backtrack(s WITH DEFAULT empty string, left WITH DEFAULT zero, right WITH DEFAULT zero)  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH PARAMETERS s CONCATENATED WITH an opening parenthesis CHARACTER, left INCREMENTED BY one, right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH PARAMETERS s CONCATENATED WITH a closing parenthesis CHARACTER, left, right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITHOUT PARAMETERS  
        RETURN result  
    END FUNCTION  
END CLASS
