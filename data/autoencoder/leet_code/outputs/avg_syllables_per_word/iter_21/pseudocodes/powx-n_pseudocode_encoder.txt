CLASS Solution  
    FUNCTION myPow(x, n)  
        FUNCTION qpow(a, n)  
            SET ans TO one  
            WHILE n GREATER THAN zero HOLDS  
                IF n MODULO two EQUALS one THEN  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two USING INTEGER FLOOR DIVISION  
            END WHILE  
            RETURN ans  
        END FUNCTION  

        IF n GREATER THAN OR EQUAL TO zero THEN  
            RETURN CALL qpow WITH PARAMETERS x AND n  
        ELSE  
            RETURN one DIVIDED BY CALL qpow WITH PARAMETERS x AND n MULTIPLIED BY negative one  
        END IF  
    END FUNCTION  
END CLASS
