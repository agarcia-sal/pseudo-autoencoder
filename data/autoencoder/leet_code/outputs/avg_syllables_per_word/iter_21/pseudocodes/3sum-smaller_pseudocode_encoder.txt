CLASS Solution  
    FUNCTION threeSumSmaller(nums, target)  
        CALL sort ON nums WITH PARAMETERS  
        SET count TO zero  
        SET n TO the LENGTH OF nums  
        
        FOR i FROM zero TO n MINUS two MINUS one  
            SET left TO i PLUS one  
            SET right TO n MINUS one  
            WHILE left LESS THAN right  
                SET current_sum TO element at position i OF nums PLUS element at position left OF nums PLUS element at position right OF nums  
                IF current_sum LESS THAN target  
                    INCREMENT count BY right MINUS left  
                    INCREMENT left BY one  
                ELSE  
                    DECREMENT right BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
