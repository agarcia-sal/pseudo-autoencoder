CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a new mapping FROM zero TO negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each index i AND element char IN s  
            CALL digit TO the integer conversion OF char  
            SET prefix_xor TO prefix_xor XOR one LEFT SHIFTED BY digit  
            
            IF prefix_xor IS a key IN prefix_map  
                SET max_length TO the greater value BETWEEN max_length AND index i MINUS the value AT prefix_xor IN prefix_map  
            END IF  
            
            FOR each integer j FROM zero TO nine  
                SET flipped TO prefix_xor XOR one LEFT SHIFTED BY j  
                IF flipped IS a key IN prefix_map  
                    SET max_length TO the greater value BETWEEN max_length AND index i MINUS the value AT flipped IN prefix_map  
                END IF  
            END FOR  
            
            IF prefix_xor IS NOT a key IN prefix_map  
                SET the value AT prefix_xor IN prefix_map TO index i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
