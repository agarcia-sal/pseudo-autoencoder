CLASS Solution  
    FUNCTION integerBreak(n)  
        SET dp TO a list of zeros with length equal TO n PLUS one  
        SET element at position one OF dp TO one  
        FOR i FROM two TO n  
            SET max_product TO zero  
            FOR j FROM one TO i MINUS one  
                SET potential_product_one TO j MULTIPLIED BY element at position i MINUS j OF dp  
                SET potential_product_two TO j MULTIPLIED BY i MINUS j  
                IF potential_product_one GREATER THAN max_product  
                    SET max_product TO potential_product_one  
                END IF  
                IF potential_product_two GREATER THAN max_product  
                    SET max_product TO potential_product_two  
                END IF  
            END FOR  
            SET element at position i OF dp TO max_product  
        END FOR  
        RETURN element at position n OF dp  
    END FUNCTION  
END CLASS
