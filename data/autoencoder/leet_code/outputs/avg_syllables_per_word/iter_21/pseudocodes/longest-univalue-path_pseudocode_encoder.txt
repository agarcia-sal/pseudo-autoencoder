CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETER element at position zero OF values  
    SET i TO one  
    SET queue TO NEW empty deque  
    CALL queue APPEND WITH PARAMETERS root  
    WHILE queue IS NOT empty  
        SET node TO CALL queue POP left  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO NEW TreeNode WITH PARAMETER element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.left  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO NEW TreeNode WITH PARAMETER element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left AND q.left AND CALL is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestUnivaluePath(root)  
        FUNCTION dfs(node)  
            IF node IS None  
                RETURN zero, zero  
            END IF  
            SET left_length, left_value TO CALL dfs WITH PARAMETERS node.left  
            SET right_length, right_value TO CALL dfs WITH PARAMETERS node.right  
            SET left_arrow TO zero  
            SET right_arrow TO zero  
            IF node.left IS NOT None AND node.left.val EQUALS node.val  
                SET left_arrow TO left_length PLUS one  
            END IF  
            IF node.right IS NOT None AND node.right.val EQUALS node.val  
                SET right_arrow TO right_length PLUS one  
            END IF  
            SET self.max_length TO THE maximum OF self.max_length AND left_arrow PLUS right_arrow  
            RETURN THE maximum OF left_arrow AND right_arrow, node.val  
        END FUNCTION  
        SET self.max_length TO zero  
        CALL dfs WITH PARAMETERS root  
        RETURN self.max_length  
    END FUNCTION  
END CLASS
