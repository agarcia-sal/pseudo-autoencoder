CLASS Solution  
    FUNCTION largestSumOfAverages(nums, k)  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list OF zeros WITH LENGTH n PLUS one  
        FOR i FROM zero TO n MINUS one  
            SET element at position i PLUS one OF prefix_sum TO element at position i OF prefix_sum PLUS element at position i OF nums  
        END FOR  
        
        FUNCTION average(i, j)  
            RETURN the value OF element at position j PLUS one OF prefix_sum MINUS element at position i OF prefix_sum DIVIDED BY j MINUS i PLUS one  
        END FUNCTION  
        
        SET dp TO a two-dimensional list OF zeros WITH dimensions n PLUS one BY k PLUS one  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AND one OF dp TO CALL average WITH PARAMETERS i AND n MINUS one  
        END FOR  
        
        FOR i FROM n MINUS one DOWNTO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    SET element at position i AND partitions OF dp TO the GREATER OF element at position i AND partitions OF dp AND element at position j AND partitions MINUS one OF dp PLUS CALL average WITH PARAMETERS i AND j MINUS one  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero AND k OF dp  
    END FUNCTION  
END CLASS
