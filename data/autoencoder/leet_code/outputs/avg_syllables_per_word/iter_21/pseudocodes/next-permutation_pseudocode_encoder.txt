CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET i TO the LENGTH OF nums MINUS two  
        WHILE i IS GREATER THAN OR EQUAL TO zero AND element at position i OF nums IS GREATER THAN OR EQUAL TO element at position i PLUS one OF nums  
            DECREMENT i BY one  
        END WHILE  
        IF i IS GREATER THAN OR EQUAL TO zero  
            SET j TO the LENGTH OF nums MINUS one  
            WHILE element at position j OF nums IS LESS THAN OR EQUAL TO element at position i OF nums  
                DECREMENT j BY one  
            END WHILE  
            SET temporary_variable TO element at position i OF nums  
            SET element at position i OF nums TO element at position j OF nums  
            SET element at position j OF nums TO temporary_variable  
        END IF  
        SET sublist TO the SUBLIST of nums FROM position i PLUS one TO the LAST POSITION  
        SET reversed_sublist TO the SUBLIST in REVERSED ORDER OF sublist  
        FOR index FROM zero TO the LENGTH OF reversed_sublist MINUS one  
            SET element at position i PLUS one PLUS index OF nums TO element at position index OF reversed_sublist  
        END FOR  
    END FUNCTION  
END CLASS
