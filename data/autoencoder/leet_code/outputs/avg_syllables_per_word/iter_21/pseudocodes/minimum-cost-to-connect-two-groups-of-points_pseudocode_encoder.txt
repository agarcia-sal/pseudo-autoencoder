CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF the element at position zero OF cost  
        CALL create_two_dimensional_list WITH PARAMETERS size1 PLUS one AND the value TWO RAISED TO THE POWER OF size2 ASSIGNING TO dp  
        SET the element at position zero AT dp AND position zero OF dp TO zero  
        
        FOR i FROM zero TO size1 MINUS one  
            FOR mask FROM zero TO TWO RAISED TO THE POWER OF size2 MINUS one  
                FOR j FROM zero TO size2 MINUS one  
                    CALL minimum BETWEEN the element at position i PLUS one AT dp AND the element at position mask OR THE VALUE TWO RAISED TO THE POWER OF j AND the sum OF the element at position i AT dp AND position mask AND the element at position i AT cost AND position j ASSIGNING TO the element at position i PLUS one AT dp AND position mask OR THE VALUE TWO RAISED TO THE POWER OF j  
                END FOR  
                FOR j FROM zero TO size2 MINUS one  
                    IF mask AND THE VALUE TWO RAISED TO THE POWER OF j IS NOT EQUAL TO zero  
                        CALL minimum BETWEEN the element at position i PLUS one AT dp AND position mask AND the sum OF the element at position i PLUS one AT dp AND position mask XOR THE VALUE TWO RAISED TO THE POWER OF j AND the element at position i AT cost AND position j ASSIGNING TO the element at position i PLUS one AT dp AND position mask  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN the element at position size1 AT dp AND position TWO RAISED TO THE POWER OF size2 MINUS one  
    END FUNCTION  

    FUNCTION create_two_dimensional_list(rows, columns)  
        SET result_list TO an empty list  
        FOR each number FROM zero TO rows MINUS one  
            SET inner_list TO a new list  
            FOR each number FROM zero TO columns MINUS one  
                APPEND positive infinity TO inner_list  
            END FOR  
            APPEND inner_list TO result_list  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
