CLASS Solution  
    FUNCTION numDistinctIslands2(grid)  
        FUNCTION dfs(x, y, island)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero OF grid OR element at position x OF grid at position y EQUALS zero  
                RETURN  
            END IF  
            SET element at position x OF grid at position y TO zero  
            APPEND tuple of x AND y TO island  
            CALL dfs WITH PARAMETERS x PLUS one, y, island  
            CALL dfs WITH PARAMETERS x MINUS one, y, island  
            CALL dfs WITH PARAMETERS x, y PLUS one, island  
            CALL dfs WITH PARAMETERS x, y MINUS one, island  
        END FUNCTION  

        FUNCTION normalize(island)  
            SET shapes TO empty list  
            FOR each pair dx AND dy IN list of tuples containing one AND one, one AND minus one, minus one AND one, minus one AND minus one  
                SET new_island TO empty list  
                FOR each pair x AND y IN island  
                    APPEND tuple of x MULTIPLIED BY dx AND y MULTIPLIED BY dy TO new_island  
                END FOR  
                SORT new_island in ascending order  
                SET min_x TO minimum x value among all pairs in new_island  
                SET min_y TO minimum y value among all pairs in new_island  
                APPEND tuple of each pair where x IS replaced by x MINUS min_x AND y IS replaced by y MINUS min_y FOR each pair x AND y IN new_island TO shapes  
            END FOR  
            FOR index FROM zero TO LENGTH OF shapes MINUS one  
                APPEND tuple of each pair where first element IS y AND second element IS negative x FOR each pair x AND y IN element at position index OF shapes TO shapes  
            END FOR  
            RETURN minimum element in shapes according to default ordering  
        END FUNCTION  

        SET islands TO empty set  
        FOR i FROM zero TO LENGTH OF grid MINUS one  
            FOR j FROM zero TO LENGTH OF element at position zero OF grid MINUS one  
                IF element at position i OF grid at position j EQUALS one  
                    SET island TO empty list  
                    CALL dfs WITH PARAMETERS i, j, island  
                    ADD result of CALLING normalize WITH PARAMETER island TO islands  
                END IF  
            END FOR  
        END FOR  
        RETURN the LENGTH OF islands  
    END FUNCTION  
END CLASS
