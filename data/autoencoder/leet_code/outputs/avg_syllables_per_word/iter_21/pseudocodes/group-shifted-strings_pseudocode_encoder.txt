CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF the LENGTH OF s EQUALS zero  
                RETURN empty string  
            END IF  
            SET canonical_form TO an empty list  
            FOR each character c IN s  
                SET difference TO the Unicode code point of character c MINUS the Unicode code point of the element at position zero OF s  
                SET modulo_difference TO difference MODULO twenty six  
                APPEND the string representation of modulo_difference TO canonical_form  
            END FOR  
            SET joined_canonical_form TO the concatenation of elements in canonical_form JOINED BY a comma  
            RETURN joined_canonical_form  
        END FUNCTION  
        SET groups TO a new dictionary defaulting to empty list  
        FOR each string s IN strings  
            SET canonical_form TO CALL get_canonical_form WITH PARAMETERS s  
            APPEND s TO the list at key canonical_form IN groups  
        END FOR  
        SET result TO the list of all values IN groups  
        RETURN result  
    END FUNCTION  
END CLASS
