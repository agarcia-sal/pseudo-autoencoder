CLASS Solution  
    FUNCTION primePalindrome(n)  
        FUNCTION is_prime(num)  
            IF num LESS THAN 2  
                RETURN False  
            END IF  
            IF num EQUALS 2 OR num EQUALS 3  
                RETURN True  
            END IF  
            IF num MODULO 2 EQUALS 0 OR num MODULO 3 EQUALS 0  
                RETURN False  
            END IF  
            SET i TO 5  
            WHILE i MULTIPLIED BY i LESS THAN OR EQUAL TO num  
                IF num MODULO i EQUALS 0 OR num MODULO (i PLUS 2) EQUALS 0  
                    RETURN False  
                END IF  
                INCREMENT i BY 6  
            END WHILE  
            RETURN True  
        END FUNCTION  

        FUNCTION is_palindrome(num)  
            RETURN the string representation of num EQUALS the reversed string representation of num  
        END FUNCTION  

        IF n LESS THAN OR EQUAL TO 2  
            RETURN 2  
        END IF  

        IF n GREATER THAN OR EQUAL TO 10 AND n LESS THAN OR EQUAL TO 11  
            RETURN 11  
        END IF  

        WHILE True  
            IF n MODULO 2 EQUALS 0  
                INCREMENT n BY 1  
                CONTINUE TO next iteration of WHILE  
            END IF  
            IF is_palindrome(n) AND is_prime(n)  
                RETURN n  
            END IF  
            IF the length OF the string representation of n MODULO 2 EQUALS 0  
                SET n TO 10 RAISED TO THE POWER OF the length OF the string representation of n PLUS 1  
            ELSE  
                INCREMENT n BY 2  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
