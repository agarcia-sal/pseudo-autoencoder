CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        
        WHILE n GREATER THAN length MULTIPLIED BY count HOLDS  
            DECREMENT n BY length MULTIPLIED BY count  
            INCREMENT length BY one  
            SET count TO count MULTIPLIED BY ten  
            SET start TO start MULTIPLIED BY ten  
        END WHILE  
        
        SET start TO start PLUS (n MINUS one) DIVIDED BY length  
        SET digit_string TO CONVERT start TO string  
        SET digit_index TO (n MINUS one) MODULO length  
        SET result_digit TO CONVERT element at position digit_index OF digit_string TO integer  
        RETURN result_digit  
    END FUNCTION  
END CLASS
