CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL arr2.sort() TO SORT arr2 IN ASCENDING ORDER  
        
        SET dp TO A DICTIONARY WITH KEY SET TO MINUS ONE AND VALUE SET TO ZERO  
        
        FOR each num IN arr1  
            SET new_dp TO AN EMPTY DICTIONARY  
            FOR each last_num AND ops IN dp.items()  
                IF num GREATER THAN last_num THEN  
                    SET current_minimum TO THE VALUE ASSOCIATED WITH num IN new_dp OR INFINITY IF NOT PRESENT  
                    SET new_dp[num] TO THE MINIMUM OF current_minimum AND ops  
                END IF  
                
                SET idx TO THE POSITION TO INSERT last_num IN arr2 TO KEEP IT SORTED ON THE RIGHT SIDE USING BISECT RIGHT  
                IF idx LESS THAN THE LENGTH OF arr2 THEN  
                    SET replacement_num TO THE ELEMENT AT POSITION idx OF arr2  
                    SET current_minimum_replacement TO THE VALUE ASSOCIATED WITH replacement_num IN new_dp OR INFINITY IF NOT PRESENT  
                    SET new_dp[replacement_num] TO THE MINIMUM OF current_minimum_replacement AND ops PLUS ONE  
                END IF  
            END FOR  
            
            IF new_dp IS EMPTY THEN  
                RETURN MINUS ONE  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN THE MINIMUM VALUE AMONG ALL VALUES IN dp  
    END FUNCTION  
END CLASS
