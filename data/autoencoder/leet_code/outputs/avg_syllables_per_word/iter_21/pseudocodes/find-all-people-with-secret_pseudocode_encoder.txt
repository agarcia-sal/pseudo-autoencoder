CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO a set containing zero AND firstPerson  
        CALL sort ON meetings WITH PARAMETERS that ELEMENTS are ordered BY the element at position two OF each meeting  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF meetings  
            SET current_time TO the element at position two OF the element at position i OF meetings  
            SET graph TO a new mapping WHERE each key maps TO an empty list  
            WHILE i IS LESS THAN the LENGTH OF meetings AND the element at position two OF the element at position i OF meetings EQUALS current_time  
                SET x TO the element at position zero OF the element at position i OF meetings  
                SET y TO the element at position one OF the element at position i OF meetings  
                APPEND y TO the list at key x OF graph  
                APPEND x TO the list at key y OF graph  
                INCREMENT i BY one  
            END WHILE  
            SET queue TO a new double-ended queue CONTAINING each person IN know_secret THAT is a key IN graph  
            SET visited TO an empty set  
            WHILE queue IS NOT empty  
                REMOVE and RETURN the element from the left of queue AND ASSIGN it TO person  
                IF person IS in visited  
                    CONTINUE to the next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list at key person OF graph  
                    IF neighbor IS NOT in visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN a list CONTAINING all elements of know_secret  
    END FUNCTION  
END CLASS
