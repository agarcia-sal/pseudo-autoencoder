CLASS Solution  
    FUNCTION longestPalindromeSubseq(s)  
        SET n TO the LENGTH OF s  
        CALL create2DArray WITH PARAMETERS n AND n TO CREATE dp  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i THEN position i OF dp TO one  
        END FOR  
        
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i OF s EQUALS element at position j OF s THEN  
                    SET element at position i PLUS one THEN position j MINUS one OF dp PLUS two TO element at position i THEN position j OF dp  
                ELSE  
                    SET the MAXIMUM OF element at position i PLUS one THEN position j OF dp AND element at position i THEN position j MINUS one OF dp TO element at position i THEN position j OF dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero THEN position n MINUS one OF dp  
    END FUNCTION  
END CLASS
