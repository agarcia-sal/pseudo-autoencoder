CLASS Solution  
    FUNCTION maximumScore(nums, k)  
        SET left TO k  
        SET right TO k  
        SET current_min TO element at position k OF nums  
        SET max_score TO current_min  
        
        WHILE left GREATER THAN zero OR right LESS THAN LENGTH OF nums MINUS one HOLDS  
            IF left EQUALS zero THEN  
                INCREMENT right BY one  
            ELSE IF right EQUALS LENGTH OF nums MINUS one THEN  
                DECREMENT left BY one  
            ELSE IF element at position left MINUS one OF nums GREATER THAN OR EQUAL TO element at position right PLUS one OF nums THEN  
                DECREMENT left BY one  
            ELSE  
                INCREMENT right BY one  
            END IF  
            
            SET current_min TO the smaller value BETWEEN current_min AND element at position left OF nums AND element at position right OF nums  
            SET score TO current_min MULTIPLIED BY the difference BETWEEN right AND left PLUS one  
            SET max_score TO the larger value BETWEEN max_score AND score  
        END WHILE  
        
        RETURN max_score  
    END FUNCTION  
END CLASS
