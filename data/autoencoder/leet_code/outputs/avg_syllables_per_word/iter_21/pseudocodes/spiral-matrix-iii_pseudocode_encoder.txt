CLASS Solution  
    FUNCTION spiralMatrixIII(rows, cols, rStart, cStart)  
        SET directions TO list containing tuple zero AND one, tuple one AND zero, tuple zero AND negative one, tuple negative one AND zero  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO empty list  
        SET x TO rStart  
        SET y TO cStart  
        
        WHILE the LENGTH OF visited IS LESS THAN rows MULTIPLIED BY cols  
            IF the value of increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            
            FOR each iteration FROM one TO steps  
                IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN rows AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN cols  
                    APPEND the list containing x AND y TO visited  
                END IF  
                
                SET dx TO element at position direction_index OF directions AT position zero  
                SET dy TO element at position direction_index OF directions AT position one  
                SET x TO x PLUS dx  
                SET y TO y PLUS dy  
            END FOR  
            
            SET direction_index TO the remainder of direction_index PLUS one DIVIDED BY four  
            INCREMENT increment BY one  
        END WHILE  
        
        RETURN visited  
    END FUNCTION  
END CLASS
