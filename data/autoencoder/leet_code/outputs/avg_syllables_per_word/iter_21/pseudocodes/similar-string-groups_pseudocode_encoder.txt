CLASS Solution  
    FUNCTION numSimilarGroups(strs)  
        FUNCTION are_similar(s1, s2)  
            SET diff TO an empty list  
            FOR index FROM zero TO the LENGTH OF s1 MINUS one  
                IF element at position index OF s1 NOT EQUALS element at position index OF s2  
                    APPEND index TO diff  
                END IF  
            END FOR  
            RETURN the LENGTH OF diff EQUALS zero OR (the LENGTH OF diff EQUALS two AND element at position element at position zero OF diff OF s1 EQUALS element at position element at position one OF diff OF s2 AND element at position element at position one OF diff OF s1 EQUALS element at position element at position zero OF diff OF s2)  
        END FUNCTION  
        
        FUNCTION dfs(node, visited)  
            CALL visited ADD node  
            FOR each neighbor IN the element at position node OF graph  
                IF neighbor NOT IN visited  
                    CALL dfs WITH PARAMETERS neighbor AND visited  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET n TO the LENGTH OF strs  
        SET graph TO an empty dictionary WITH keys AS integers FROM zero TO n MINUS one AND values AS empty lists  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                IF CALL are_similar WITH PARAMETERS element at position i OF strs AND element at position j OF strs RETURNS true  
                    APPEND j TO the element at position i OF graph  
                    APPEND i TO the element at position j OF graph  
                END IF  
            END FOR  
        END FOR  
        
        SET visited TO an empty set  
        SET num_groups TO zero  
        
        FOR i FROM zero TO n MINUS one  
            IF i NOT IN visited  
                CALL dfs WITH PARAMETERS i AND visited  
                INCREMENT num_groups BY one  
            END IF  
        END FOR  
        
        RETURN num_groups  
    END FUNCTION  
END CLASS
