CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO the LIST OF characters FROM num  
        SET n TO the LENGTH OF num  
          
        WHILE k GREATER THAN zero HOLDS  
            SET swapped TO False  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF character at position i OF num GREATER THAN character at position i PLUS one OF num THEN  
                    SET min_digit TO character nine  
                    SET min_index TO minus one  
                    FOR j FROM i TO THE MINIMUM OF i PLUS k PLUS one AND n MINUS one  
                        IF character at position j OF num LESS THAN min_digit THEN  
                            SET min_digit TO character at position j OF num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    SET num FROM position i TO min_index PLUS one TO A NEW LIST CONSISTING OF character at position min_index OF num FOLLOWED BY the SUBLIST FROM i TO min_index MINUS one OF num  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO True  
                    BREAK  
                END IF  
            END FOR  
            IF NOT swapped THEN  
                BREAK  
            END IF  
        END WHILE  
        RETURN THE STRING FORMED BY JOINING ALL CHARACTERS OF num  
    END FUNCTION  
END CLASS
