CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode WITH val EQUALS element at position zero OF values  
    SET i TO one  
    CALL queue TO A NEW double-ended queue  
    CALL append root TO queue  
    WHILE queue IS NOT empty  
        CALL popleft element FROM queue INTO node  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO A NEW TreeNode WITH val EQUALS element at position i OF values  
            CALL append node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO A NEW TreeNode WITH val EQUALS element at position i OF values  
            CALL append node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left AND q.left AND CALL is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root IS None  
            RETURN  
        END IF  
        SET stack TO A NEW list WITH element root  
        WHILE stack IS NOT empty  
            CALL pop element FROM stack INTO node  
            IF node.right IS NOT None  
                CALL append node.right TO stack  
            END IF  
            IF node.left IS NOT None  
                CALL append node.left TO stack  
            END IF  
            SET node.left TO None  
            IF stack IS NOT empty  
                SET node.right TO element at last position OF stack  
            ELSE  
                SET node.right TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
