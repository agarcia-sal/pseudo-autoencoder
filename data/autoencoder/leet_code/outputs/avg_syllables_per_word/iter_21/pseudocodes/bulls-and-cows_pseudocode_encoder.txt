CLASS Solution  
    FUNCTION getHint(secret, guess)  
        CALL import_defaultdict FROM collections  
        SET bulls TO zero  
        SET cows TO zero  
        SET secret_count TO CALL defaultdict WITH PARAMETERS integer  
        SET guess_count TO CALL defaultdict WITH PARAMETERS integer  
        FOR each pair_of_characters s AND g IN secret AND guess  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value at key s IN secret_count BY one  
                INCREMENT the value at key g IN guess_count BY one  
            END IF  
        END FOR  
        FOR each digit IN keys OF secret_count  
            INCREMENT cows BY the MINIMUM OF the value at key digit IN secret_count AND the value at key digit IN guess_count  
        END FOR  
        SET result TO CONCATENATION OF bulls AS STRING, the character A, cows AS STRING, and the character B  
        RETURN result  
    END FUNCTION  
END CLASS
