CLASS Solution  
    FUNCTION topKFrequent(words, k)  
        CALL Counter FUNCTION WITH PARAMETERS words AND ASSIGN RESULT TO count  
        
        FUNCTION build_heap FROM count  
            CREATE empty list NAMED heap  
            FOR each word AND frequency IN count  
                APPEND tuple CONSISTING OF negative frequency FOLLOWED BY word TO heap  
            END FOR  
            CALL heapify FUNCTION WITH PARAMETERS heap  
            RETURN heap  
        END FUNCTION  
        
        CALL build_heap FUNCTION WITH PARAMETERS count AND ASSIGN RESULT TO heap  
        
        CREATE empty list NAMED result  
        FOR index FROM one TO k  
            CALL heappop FUNCTION WITH PARAMETERS heap AND ASSIGN RESULT TO popped_element  
            SET word TO element at position two OF popped_element  
            APPEND word TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
