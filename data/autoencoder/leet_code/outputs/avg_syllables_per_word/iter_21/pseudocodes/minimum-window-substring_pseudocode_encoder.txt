CLASS Solution  
    FUNCTION minWindow(s, t)  
        CALL import collections WITH PARAMETERS  
        SET t_count TO CALL Counter WITH PARAMETERS t  
        SET required TO the LENGTH OF t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO CALL Counter WITH PARAMETERS  
        SET min_length TO positive infinity  
        SET min_start TO zero  
        
        WHILE right LESS THAN the LENGTH OF s  
            INCREMENT the element at position right OF s IN window_counts BY one  
            
            IF the element at position right OF s IS IN t_count AND the element at position right OF s IN window_counts EQUALS the element at position right OF s IN t_count  
                INCREMENT formed BY one  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO the element at position left OF s  
                
                IF right MINUS left PLUS one LESS THAN min_length  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                
                DECREMENT the count of char IN window_counts BY one  
                IF char IS IN t_count AND the count of char IN window_counts LESS THAN the count of char IN t_count  
                    DECREMENT formed BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT right BY one  
        END WHILE  
        
        IF min_length EQUALS positive infinity  
            RETURN an empty string  
        ELSE  
            RETURN the substring from position min_start TO min_start PLUS min_length OF s  
        END IF  
    END FUNCTION  
END CLASS
