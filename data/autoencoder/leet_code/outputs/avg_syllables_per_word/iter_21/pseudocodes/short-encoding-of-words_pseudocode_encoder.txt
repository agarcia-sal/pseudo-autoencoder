CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET reversed_words TO CALL sorted WITH PARAMETERS FOR each word IN words CALLING reverse_string WITH PARAMETERS word  
        SET total_length TO zero  
        FOR index FROM zero TO the LENGTH OF reversed_words MINUS one  
            IF index PLUS one EQUALS the LENGTH OF reversed_words OR NOT reversed_words AT POSITION index PLUS one STARTS WITH reversed_words AT POSITION index  
                INCREMENT total_length BY the LENGTH OF reversed_words AT POSITION index PLUS one  
            END IF  
        END FOR  
        RETURN total_length  
    END FUNCTION  
  
    FUNCTION reverse_string(word)  
        SET reversed_word TO empty string  
        FOR position FROM the LENGTH OF word MINUS one DOWN TO zero  
            APPEND character at position OF word TO reversed_word  
        END FOR  
        RETURN reversed_word  
    END FUNCTION  
END CLASS
