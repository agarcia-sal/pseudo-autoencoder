CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        CALL createTwoDimensionalArray WITH PARAMETERS m PLUS one AND n PLUS one AND zero  
        ASSIGN reference TO dp  

        FOR each s IN strs  
            CALL countCharacterOccurrences WITH PARAMETERS s AND zeroCharacter  
            ASSIGN reference TO count_0  
            CALL countCharacterOccurrences WITH PARAMETERS s AND oneCharacter  
            ASSIGN reference TO count_1  

            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    CALL getMaximum OF dp AT position i AND j AND dp AT position i MINUS count_0 AND j MINUS count_1 PLUS one  
                    SET dp element at position i AND j TO the CALLED maximum  
                END FOR  
            END FOR  
        END FOR  

        RETURN element at position m AND n OF dp  
    END FUNCTION  
END CLASS
