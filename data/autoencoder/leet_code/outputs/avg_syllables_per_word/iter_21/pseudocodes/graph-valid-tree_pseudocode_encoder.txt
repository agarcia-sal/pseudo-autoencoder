CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF the LENGTH OF edges NOT EQUALS n MINUS one THEN  
            RETURN false  
        END IF  
        
        SET adj_list TO a mapping FROM each integer i FROM zero TO n MINUS one TO an empty list  
        FOR each pair a, b IN edges  
            APPEND b TO the list AT key a OF adj_list  
            APPEND a TO the list AT key b OF adj_list  
        END FOR  
        
        SET visited TO an empty set  
        
        FUNCTION dfs(node)  
            IF node IS IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list AT key node OF adj_list  
                CALL dfs WITH PARAMETERS neighbor  
            END FOR  
        END FUNCTION  
        
        CALL dfs WITH PARAMETERS zero  
        
        RETURN the LENGTH OF visited EQUALS n  
    END FUNCTION  
END CLASS
