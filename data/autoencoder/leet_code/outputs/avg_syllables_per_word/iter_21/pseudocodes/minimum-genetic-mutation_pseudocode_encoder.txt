CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO the SET created from bank  
        SET q TO a new DEQUE containing the TUPLE of start and zero  
        SET mp TO a MAPPING where key A maps to the STRING containing characters T C G  
                              key T maps to the STRING containing characters A C G  
                              key C maps to the STRING containing characters A T G  
                              key G maps to the STRING containing characters A T C  
        WHILE the LENGTH OF q IS GREATER THAN zero  
            SET t and step TO the result of REMOVING the FIRST element FROM q  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR each index i and character v IN the STRING t  
                FOR each character j IN the STRING mp AT key v  
                    SET next TO the CONCATENATION of the substring from POSITION zero TO i MINUS one OF t  
                                  PLUS the character j  
                                  PLUS the substring from POSITION i PLUS one TO the END OF t  
                    IF next IS IN s  
                        APPEND the TUPLE of next and step PLUS one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
