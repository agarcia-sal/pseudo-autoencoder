CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        CALL create_two_dimensional_list WITH PARAMETERS m PLUS one AND n PLUS one AND zero  
        ASSIGN reference TO dp  
        
        FOR i FROM zero TO m  
            SET the element at position zero OF the element at position i OF dp TO i  
        END FOR  
        
        FOR j FROM zero TO n  
            SET the element at position j OF the element at position zero OF dp TO j  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one OF word1 EQUALS the element at position j MINUS one OF word2 THEN  
                    SET the element at position j OF the element at position i OF dp TO the element at position j MINUS one OF the element at position i MINUS one OF dp  
                ELSE  
                    SET deletion_cost TO the element at position j OF the element at position i MINUS one OF dp  
                    SET insertion_cost TO the element at position j MINUS one OF the element at position i OF dp  
                    SET replacement_cost TO the element at position j MINUS one OF the element at position i MINUS one OF dp  
                    SET the element at position j OF the element at position i OF dp TO one PLUS the MINIMUM OF deletion_cost AND insertion_cost AND replacement_cost  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position n OF the element at position m OF dp  
    END FUNCTION  
END CLASS
