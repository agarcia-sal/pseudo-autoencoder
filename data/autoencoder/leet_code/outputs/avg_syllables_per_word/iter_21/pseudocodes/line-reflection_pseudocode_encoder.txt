CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO CALL convert_list_of_lists_to_set_of_tuples WITH PARAMETERS points  
        SET min_x TO CALL find_minimum_x_coordinate WITH PARAMETERS unique_points  
        SET max_x TO CALL find_maximum_x_coordinate WITH PARAMETERS unique_points  
        SET line_of_reflection TO (min_x PLUS max_x) DIVIDED BY two  
        FOR each point IN unique_points  
            SET x TO element at position zero OF point  
            SET y TO element at position one OF point  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF tuple consisting of reflected_x AND y NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  

    FUNCTION convert_list_of_lists_to_set_of_tuples(points)  
        SET temp_set TO empty set  
        FOR each element IN points  
            CONVERT element TO tuple  
            APPEND the converted tuple TO temp_set  
        END FOR  
        RETURN temp_set  
    END FUNCTION  

    FUNCTION find_minimum_x_coordinate(unique_points)  
        SET minimum_value TO positive infinity  
        FOR each point IN unique_points  
            SET current_x TO element at position zero OF point  
            IF current_x LESS THAN minimum_value  
                SET minimum_value TO current_x  
            END IF  
        END FOR  
        RETURN minimum_value  
    END FUNCTION  

    FUNCTION find_maximum_x_coordinate(unique_points)  
        SET maximum_value TO negative infinity  
        FOR each point IN unique_points  
            SET current_x TO element at position zero OF point  
            IF current_x GREATER THAN maximum_value  
                SET maximum_value TO current_x  
            END IF  
        END FOR  
        RETURN maximum_value  
    END FUNCTION  
END CLASS
