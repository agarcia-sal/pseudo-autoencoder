CLASS Solution  
    FUNCTION increasingTriplet(nums)  
        SET first TO positive infinity  
        SET second TO positive infinity  
        
        FOR each num IN nums  
            IF num LESS THAN OR EQUAL TO first  
                SET first TO num  
            ELSE IF num LESS THAN OR EQUAL TO second  
                SET second TO num  
            ELSE  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
