CLASS Solution  
    FUNCTION medianSlidingWindow(nums, k)  
        SET lo TO empty list  
        SET hi TO empty list  

        FUNCTION balance_heaps()  
            IF the LENGTH OF lo IS GREATER THAN the LENGTH OF hi PLUS one  
                CALL heappush WITH PARAMETERS hi AND NEGATIVE OF heappop CALLED WITH lo  
            ELSE IF the LENGTH OF hi IS GREATER THAN the LENGTH OF lo  
                CALL heappush WITH PARAMETERS lo AND NEGATIVE OF heappop CALLED WITH hi  
            END IF  
        END FUNCTION  

        FUNCTION add_num(num)  
            IF lo IS empty OR num IS LESS THAN OR EQUAL TO NEGATIVE OF the element at position zero OF lo  
                CALL heappush WITH PARAMETERS lo AND NEGATIVE OF num  
            ELSE  
                CALL heappush WITH PARAMETERS hi AND num  
            END IF  
            CALL balance_heaps  
        END FUNCTION  

        FUNCTION remove_num(num)  
            IF num IS LESS THAN OR EQUAL TO NEGATIVE OF the element at position zero OF lo  
                REMOVE NEGATIVE OF num FROM lo  
                IF lo IS NOT empty  
                    CALL heapify WITH PARAMETER lo  
                END IF  
            ELSE  
                REMOVE num FROM hi  
                IF hi IS NOT empty  
                    CALL heapify WITH PARAMETER hi  
                END IF  
            END IF  
            CALL balance_heaps  
        END FUNCTION  

        FUNCTION get_median()  
            IF the LENGTH OF lo IS GREATER THAN the LENGTH OF hi  
                RETURN NEGATIVE OF the element at position zero OF lo  
            ELSE  
                RETURN NEGATIVE OF the element at position zero OF lo PLUS the element at position zero OF hi DIVIDED BY two  
            END IF  
        END FUNCTION  

        FOR index FROM zero TO k MINUS one  
            CALL add_num WITH PARAMETER the element at position index OF nums  
        END FOR  

        SET medians TO list CONTAINING get_median  
        
        FOR index FROM k TO the LENGTH OF nums MINUS one  
            CALL add_num WITH PARAMETER the element at position index OF nums  
            CALL remove_num WITH PARAMETER the element at position index MINUS k OF nums  
            APPEND get_median TO medians  
        END FOR  

        RETURN medians  
    END FUNCTION  
END CLASS
