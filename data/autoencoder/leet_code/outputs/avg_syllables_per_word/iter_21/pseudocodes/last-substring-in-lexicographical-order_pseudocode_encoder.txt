CLASS Solution  
    FUNCTION lastSubstring(s)  
        SET n TO the LENGTH OF s  
        SET max_char TO the MAXIMUM CHARACTER IN s  
        SET max_indices TO an EMPTY LIST  
        FOR i FROM zero TO n MINUS one  
            IF the CHARACTER at position i OF s EQUALS max_char  
                APPEND i TO max_indices  
            END IF  
        END FOR  
        
        SET max_substring TO an EMPTY STRING  
        FOR each start IN max_indices  
            SET current_substring TO the SUBSTRING from position start TO the END OF s  
            IF current_substring GREATER THAN max_substring  
                SET max_substring TO current_substring  
            END IF  
        END FOR  
        
        RETURN max_substring  
    END FUNCTION  
END CLASS
