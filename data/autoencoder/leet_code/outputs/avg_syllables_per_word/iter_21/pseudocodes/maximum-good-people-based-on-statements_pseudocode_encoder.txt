CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR integer i FROM zero TO the VALUE OF one LEFT-SHIFTED BY n MINUS one  
            SET good TO a LIST OF BOOLEAN FALSE VALUES WITH LENGTH n  
            SET bad TO a LIST OF BOOLEAN FALSE VALUES WITH LENGTH n  
            FOR integer j FROM zero TO n MINUS one  
                IF the RESULT OF BITWISE-AND BETWEEN i AND the VALUE OF one LEFT-SHIFTED BY j IS NOT EQUAL TO zero  
                    SET the ELEMENT at position j OF good TO TRUE  
                ELSE  
                    SET the ELEMENT at position j OF bad TO TRUE  
                END IF  
            END FOR  
            SET valid TO TRUE  
            FOR integer j FROM zero TO n MINUS one  
                IF the ELEMENT at position j OF good IS TRUE  
                    FOR integer k FROM zero TO n MINUS one  
                        IF the ELEMENT at position k OF good IS TRUE AND the ELEMENT at position k OF the ELEMENT at position j OF statements EQUALS zero  
                            SET valid TO FALSE  
                            EXIT THE INNER FOR LOOP  
                        ELSE IF the ELEMENT at position k OF bad IS TRUE AND the ELEMENT at position k OF the ELEMENT at position j OF statements EQUALS one  
                            SET valid TO FALSE  
                            EXIT THE INNER FOR LOOP  
                        END IF  
                    END FOR  
                    IF valid IS FALSE  
                        EXIT THE OUTER FOR LOOP  
                    END IF  
                END IF  
            END FOR  
            IF valid IS TRUE  
                SET max_good TO the GREATER OF max_good AND the SUM OF TRUE VALUES IN good  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
