CLASS ListNode  
    FUNCTION __init__(val EQUALS zero, next EQUALS None)  
        SET this.val TO val  
        SET this.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS an empty list THEN  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH element at position zero OF values  
    SET p TO head  
    FOR each val IN the sublist of values FROM position one TO the end  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the comparison of p1.val EQUALS p2.val AND CALL is_same_list WITH PARAMETERS p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1, l2)  
        FUNCTION pushIntoStack(node)  
            SET stack TO an empty list  
            WHILE node HOLDS  
                APPEND node.val TO stack  
                SET node TO node.next  
            END WHILE  
            RETURN stack  
        END FUNCTION  
      
        SET stack1 TO CALL pushIntoStack WITH PARAMETER l1  
        SET stack2 TO CALL pushIntoStack WITH PARAMETER l2  
      
        SET carry TO zero  
        SET result TO None  
      
        WHILE stack1 IS NOT empty OR stack2 IS NOT empty OR carry IS NOT zero  
            IF stack1 IS NOT empty THEN  
                SET val1 TO REMOVE the last element FROM stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF stack2 IS NOT empty THEN  
                SET val2 TO REMOVE the last element FROM stack2  
            ELSE  
                SET val2 TO zero  
            END IF  
          
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  
          
            SET new_node TO A NEW ListNode INITIALIZED WITH digit  
            SET new_node.next TO result  
            SET result TO new_node  
        END WHILE  
      
        RETURN result  
    END FUNCTION  
END CLASS
