CLASS Solution  
    FUNCTION countPrimes(n)  
        IF n LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
        
        CALL initialize_prime_list WITH PARAMETERS n ASSIGN TO is_prime  
        CALL mark_zero_and_one_as_not_prime WITH PARAMETERS is_prime  
        
        SET limit TO the INTEGER PART OF the SQUARE ROOT OF n PLUS one  
        FOR start FROM two TO limit MINUS one  
            IF element at position start OF is_prime EQUALS true  
                FOR multiple FROM start MULTIPLIED BY start TO n MINUS one IN STEPS OF start  
                    SET element at position multiple OF is_prime TO false  
                END FOR  
            END IF  
        END FOR  
        
        SET prime_count TO the CALL sum_elements_of_list WITH PARAMETERS is_prime  
        RETURN prime_count  
    END FUNCTION  

    FUNCTION initialize_prime_list(n)  
        CREATE a list of length n WITH all elements SET TO true  
        RETURN the created list  
    END FUNCTION  

    FUNCTION mark_zero_and_one_as_not_prime(is_prime)  
        SET element at position zero OF is_prime TO false  
        SET element at position one OF is_prime TO false  
    END FUNCTION  

    FUNCTION sum_elements_of_list(boolean_list)  
        SET counter TO zero  
        FOR each element IN boolean_list  
            IF element EQUALS true  
                INCREMENT counter BY one  
            END IF  
        END FOR  
        RETURN counter  
    END FUNCTION  
END CLASS
