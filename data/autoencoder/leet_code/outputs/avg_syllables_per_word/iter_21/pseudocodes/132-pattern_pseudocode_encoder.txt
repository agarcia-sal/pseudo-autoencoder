CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        
        FOR each num IN reversed order OF nums  
            IF num IS LESS THAN second  
                RETURN True  
            END IF  
            WHILE stack IS NOT empty AND element at last position OF stack IS LESS THAN num  
                SET second TO element removed from last position OF stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
