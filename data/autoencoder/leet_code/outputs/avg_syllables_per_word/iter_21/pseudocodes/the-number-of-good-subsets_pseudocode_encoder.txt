CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO CALL Counter WITH PARAMETERS nums  
        SET dp TO a list of zeroes with the length of two raised to the power of ten  
        SET element at position zero OF dp TO one  
        
        FOR each num IN keys of count  
            IF num EQUALS one  
                SET element at position zero OF dp TO element at position zero OF dp MULTIPLIED BY two raised to the power of element at position one OF count  
                SET element at position zero OF dp TO the remainder of element at position zero OF dp DIVIDED BY MODULO  
                CONTINUE  
            END IF  
            IF num NOT IN prime_factors  
                CONTINUE  
            END IF  
            
            SET factors TO element at position num OF prime_factors  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask PLUS two raised to the power of prime MINUS one  
            END FOR  
            
            FOR m FROM two raised to the power of ten MINUS one DOWN TO zero  
                IF the bitwise AND of m AND mask EQUALS zero  
                    SET element at position the bitwise OR of m AND mask OF dp TO the remainder of element at position the bitwise OR of m AND mask OF dp PLUS element at position m OF dp MULTIPLIED BY element at position num OF count DIVIDED BY MODULO  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the remainder of the sum of elements from position one TO the end OF dp DIVIDED BY MODULO  
        RETURN result  
    END FUNCTION  
END CLASS
