CLASS Solution  
    FUNCTION isRationalEqual(s, t)  
        SET ratios TO list containing one, one DIVIDED BY nine, one DIVIDED BY ninety nine, one DIVIDED BY nine hundred ninety nine, one DIVIDED BY nine thousand nine hundred ninety nine  
        
        FUNCTION valueOf(s)  
            IF the position of opening parenthesis in s EQUALS negative one  
                RETURN the floating point conversion of s  
            END IF  
            
            SET leftParenIndex TO the position of opening parenthesis in s  
            SET rightParenIndex TO the position of closing parenthesis in s  
            SET dotIndex TO the position of decimal point in s  
            
            SET integerAndNonRepeating TO the floating point conversion of the substring from the start TO position leftParenIndex MINUS one OF s  
            SET nonRepeatingLength TO leftParenIndex MINUS dotIndex MINUS one  
            
            SET repeating TO the integer conversion of the substring from position leftParenIndex PLUS one TO position rightParenIndex MINUS one OF s  
            SET repeatingLength TO rightParenIndex MINUS leftParenIndex MINUS one  
            
            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY zero point one RAISED TO THE POWER OF nonRepeatingLength MULTIPLIED BY the element at position repeatingLength OF ratios  
        END FUNCTION  
        
        RETURN the absolute value of valueOf of s MINUS valueOf of t LESS THAN one E TO THE POWER OF MINUS nine  
    END FUNCTION  
END CLASS
