CLASS Solution  
    FUNCTION findClosestElements(arr, k, x)  
        SET left TO the index where x would be inserted to keep arr sorted MINUS one  
        SET right TO left PLUS one  
        
        WHILE right MINUS left MINUS one LESS THAN k HOLDS  
            IF left EQUALS minus one OR (right LESS THAN the LENGTH OF arr AND the absolute value of the element at position right OF arr MINUS x LESS THAN the absolute value of the element at position left OF arr MINUS x) THEN  
                INCREMENT right BY one  
            ELSE  
                DECREMENT left BY one  
            END IF  
        END WHILE  
        
        RETURN the sublist of arr from position left PLUS one TO position right MINUS one INCLUSIVE  
    END FUNCTION  
END CLASS
