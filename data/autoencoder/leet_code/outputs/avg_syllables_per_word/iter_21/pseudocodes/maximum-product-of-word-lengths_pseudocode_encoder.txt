CLASS Solution  
    FUNCTION maxProduct(words)  
        CALL word_sets TO CALL convert_words_to_character_sets WITH PARAMETERS words  
        SET max_product TO zero  
        FOR i FROM zero TO the LENGTH OF words MINUS one  
            FOR j FROM i PLUS one TO the LENGTH OF words MINUS one  
                IF the INTERSECTION of element at position i OF word_sets AND element at position j OF word_sets IS empty  
                    SET product TO the LENGTH OF element at position i OF words MULTIPLIED BY the LENGTH OF element at position j OF words  
                    IF product GREATER THAN max_product  
                        SET max_product TO product  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_product  
    END FUNCTION  

    FUNCTION convert_words_to_character_sets(words)  
        SET result TO empty list  
        FOR each word IN words  
            CONVERT word TO a collection of unique characters AS character_set  
            APPEND character_set TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
