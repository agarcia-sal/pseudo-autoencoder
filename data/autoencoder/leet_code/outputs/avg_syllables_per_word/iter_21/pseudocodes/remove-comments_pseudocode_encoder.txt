CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO False  
        SET result TO empty list  
        SET current_line TO empty list  

        FOR each line IN source  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF line  
                IF in_block_comment  
                    IF substring from position i TO i PLUS one OF line EQUALS the string indicating end of block comment  
                        SET in_block_comment TO False  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF substring from position i TO i PLUS one OF line EQUALS the string indicating line comment start  
                        BREAK the while loop  
                    ELSE IF substring from position i TO i PLUS one OF line EQUALS the string indicating block comment start  
                        SET in_block_comment TO True  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND character at position i OF line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  

            IF current_line is not empty AND in_block_comment IS False  
                APPEND concatenation of all elements in current_line TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
