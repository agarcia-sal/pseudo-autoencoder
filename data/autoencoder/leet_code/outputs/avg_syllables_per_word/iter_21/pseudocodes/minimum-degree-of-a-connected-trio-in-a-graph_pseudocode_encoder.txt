CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        CALL build_graph WITH PARAMETERS n AND edges AND ASSIGN RESULT TO graph  
        SET min_degree TO positive infinity  
        SET trio_found TO false  
        FOR i FROM one TO n  
            FOR j FROM i PLUS one TO n  
                FOR k FROM j PLUS one TO n  
                    IF element j IS IN the set at key i OF graph AND element k IS IN the set at key i OF graph AND element k IS IN the set at key j OF graph THEN  
                        SET trio_found TO true  
                        CALL calculate_trio_degree WITH PARAMETERS graph i j k AND ASSIGN RESULT TO trio_degree  
                        SET min_degree TO the lesser VALUE BETWEEN min_degree AND trio_degree  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF trio_found IS true THEN  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  

    FUNCTION build_graph(n, edges)  
        SET graph TO an empty dictionary  
        FOR i FROM one TO n  
            SET the value at key i OF graph TO an empty set  
        END FOR  
        FOR each pair u v IN edges  
            ADD v TO the set at key u OF graph  
            ADD u TO the set at key v OF graph  
        END FOR  
        RETURN graph  
    END FUNCTION  

    FUNCTION calculate_trio_degree(graph, i, j, k)  
        SET degree_i TO the LENGTH OF the set at key i OF graph  
        SET degree_j TO the LENGTH OF the set at key j OF graph  
        SET degree_k TO the LENGTH OF the set at key k OF graph  
        SET total_degree TO degree_i PLUS degree_j PLUS degree_k MINUS six  
        RETURN total_degree  
    END FUNCTION  
END CLASS
