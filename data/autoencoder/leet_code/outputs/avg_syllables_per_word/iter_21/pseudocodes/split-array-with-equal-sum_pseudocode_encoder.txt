CLASS Solution  
    FUNCTION splitArray(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN seven  
            RETURN False  
        END IF  
        
        CALL prefix_sum TO CALL computePrefixSum WITH PARAMETERS nums AND n  
        
        FOR j FROM three TO n MINUS three MINUS one  
            CALL seen TO INITIALIZE an EMPTY SET  
            FOR i FROM one TO j MINUS one MINUS one  
                IF element at position i MINUS one OF prefix_sum EQUALS element at position j MINUS one OF prefix_sum MINUS element at position i OF prefix_sum  
                    ADD element at position i MINUS one OF prefix_sum TO seen  
                END IF  
            END FOR  
            FOR k FROM j PLUS two TO n MINUS one MINUS one  
                IF element at position n MINUS one OF prefix_sum MINUS element at position k OF prefix_sum EQUALS element at position k MINUS one OF prefix_sum MINUS element at position j OF prefix_sum  
                    AND element at position k MINUS one OF prefix_sum MINUS element at position j OF prefix_sum IS IN seen  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
    
    FUNCTION computePrefixSum(nums, n)  
        SET prefix_sum TO a list of zeros WITH LENGTH n  
        SET element at position zero OF prefix_sum TO element at position zero OF nums  
        FOR i FROM one TO n MINUS one  
            SET element at position i OF prefix_sum TO element at position i MINUS one OF prefix_sum PLUS element at position i OF nums  
        END FOR  
        RETURN prefix_sum  
    END FUNCTION  
END CLASS
