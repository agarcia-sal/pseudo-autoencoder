CLASS Solution  
    FUNCTION possibleToStamp(grid, stampHeight, stampWidth)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero OF grid  
        
        FUNCTION prefix_sum(matrix)  
            SET p TO a LIST of LISTS of zero WITH m PLUS one ROWS AND n PLUS one COLUMNS  
            FOR r FROM one TO m  
                FOR c FROM one TO n  
                    SET element at position c OF row r OF p TO element at position c MINUS one OF row r MINUS one OF matrix PLUS element at position c OF row r MINUS one OF p PLUS element at position c MINUS one OF row r OF p MINUS element at position c MINUS one OF row r MINUS one OF p  
                END FOR  
            END FOR  
            RETURN p  
        END FUNCTION  
        
        FUNCTION submatrix_sum(p, r1, c1, r2, c2)  
            RETURN element at position c2 PLUS one OF row r2 PLUS one OF p MINUS element at position c1 OF row r2 PLUS one OF p MINUS element at position c2 PLUS one OF row r1 OF p PLUS element at position c1 OF row r1 OF p  
        END FUNCTION  
        
        SET p_grid TO CALL prefix_sum WITH PARAMETERS grid  
        SET p_stamps TO a LIST of LISTS of zero WITH m PLUS one ROWS AND n PLUS one COLUMNS  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c OF row r OF grid EQUALS zero THEN  
                    IF r PLUS stampHeight LESS THAN OR EQUAL TO m AND c PLUS stampWidth LESS THAN OR EQUAL TO n THEN  
                        IF CALL submatrix_sum WITH PARAMETERS p_grid, r, c, r PLUS stampHeight MINUS one, c PLUS stampWidth MINUS one EQUALS zero THEN  
                            INCREMENT element at position c OF row r OF p_stamps BY one  
                        END IF  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET p_stamps TO CALL prefix_sum WITH PARAMETERS p_stamps  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c OF row r OF grid EQUALS zero THEN  
                    IF CALL submatrix_sum WITH PARAMETERS p_stamps, MAXIMUM OF zero AND r MINUS stampHeight PLUS one, MAXIMUM OF zero AND c MINUS stampWidth PLUS one, r, c EQUALS zero THEN  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
