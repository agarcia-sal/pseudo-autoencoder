CLASS Solution  
    FUNCTION maxTotalFruits(fruits, startPos, k)  
        SET max_fruits TO zero  
        SET total_fruits TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO the LENGTH OF fruits MINUS one  
            SET position TO element at position zero OF element at position right OF fruits  
            SET amount TO element at position one OF element at position right OF fruits  
            INCREMENT total_fruits BY amount  
            
            WHILE left IS LESS THAN OR EQUAL TO right AND NOT  
                (startPos MINUS k IS LESS THAN OR EQUAL TO element at position zero OF element at position left OF fruits AND  
                 element at position zero OF element at position left OF fruits IS LESS THAN OR EQUAL TO startPos PLUS k AND  
                 startPos MINUS k IS LESS THAN OR EQUAL TO element at position zero OF element at position right OF fruits AND  
                 element at position zero OF element at position right OF fruits IS LESS THAN OR EQUAL TO startPos PLUS k AND  
                 THE MINIMUM OF THE ABSOLUTE VALUE OF (element at position zero OF element at position right OF fruits MINUS startPos) AND THE ABSOLUTE VALUE OF (element at position zero OF element at position left OF fruits MINUS startPos) PLUS  
                 (element at position zero OF element at position right OF fruits MINUS element at position zero OF element at position left OF fruits) IS LESS THAN OR EQUAL TO k)  
                DECREMENT total_fruits BY element at position one OF element at position left OF fruits  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_fruits TO THE GREATER OF max_fruits AND total_fruits  
        END FOR  
        
        RETURN max_fruits  
    END FUNCTION  
END CLASS
