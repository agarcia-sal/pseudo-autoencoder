CLASS Solution  
    FUNCTION minSwapsCouples(row)  
        CALL position TO result of initialising empty dictionary  
        FOR each pair of index i AND person person IN row starting from zero IN STEPS OF two  
            CALL position ASSIGN person TO i  
        END FOR  
        SET swaps TO zero  
        FOR index i FROM zero TO the LENGTH OF row MINUS one IN STEPS OF two  
            SET correct_pair TO element at position i OF row EXCLUSIVE OR WITH one  
            IF element at position i PLUS one OF row NOT EQUALS correct_pair  
                SET swap_idx TO element at position correct_pair OF position  
                CALL temporary_variable TO element at position i PLUS one OF row  
                SET element at position i PLUS one OF row TO element at position swap_idx OF row  
                SET element at position swap_idx OF row TO temporary_variable  
                SET element at position element at position swap_idx OF row OF position TO swap_idx  
                SET element at position element at position i PLUS one OF row OF position TO i PLUS one  
                INCREMENT swaps BY one  
            END IF  
        END FOR  
        RETURN swaps  
    END FUNCTION  
END CLASS
