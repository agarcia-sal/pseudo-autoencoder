CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            WHILE the element at position i OF nums NOT EQUALS the element at position the element at position i OF nums MINUS one OF nums HOLDS  
                SET temporary_value TO the element at position i OF nums  
                SET the element at position i OF nums TO the element at position temporary_value MINUS one OF nums  
                SET the element at position temporary_value MINUS one OF nums TO temporary_value  
            END WHILE  
        END FOR  
        SET result_list TO an empty list  
        FOR each index i AND corresponding value v IN nums WITH their indices  
            IF v NOT EQUALS i PLUS one THEN  
                APPEND v TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
