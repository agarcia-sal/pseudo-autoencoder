CLASS Solution  
    FUNCTION minCost(houses, cost, m, n, target)  
        FUNCTION dp(i, prev_color, groups)  
            IF i EQUALS m  
                IF groups EQUALS target  
                    RETURN zero  
                ELSE  
                    RETURN infinity AS FLOAT  
                END IF  
            END IF  
            
            IF element at position i OF houses NOT EQUALS zero  
                SET new_groups TO groups PLUS one IF element at position i OF houses NOT EQUALS prev_color OTHERWISE groups  
                RETURN CALL dp WITH PARAMETERS i PLUS one, element at position i OF houses, new_groups  
            END IF  
            
            SET min_cost TO infinity AS FLOAT  
            FOR color FROM one TO n  
                SET new_groups TO groups PLUS one IF color NOT EQUALS prev_color OTHERWISE groups  
                SET current_cost TO element at position i AND color MINUS one OF cost PLUS CALL dp WITH PARAMETERS i PLUS one, color, new_groups  
                IF current_cost LESS THAN min_cost  
                    SET min_cost TO current_cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        SET result TO CALL dp WITH PARAMETERS zero, zero, zero  
        IF result NOT EQUALS infinity AS FLOAT  
            RETURN result  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
