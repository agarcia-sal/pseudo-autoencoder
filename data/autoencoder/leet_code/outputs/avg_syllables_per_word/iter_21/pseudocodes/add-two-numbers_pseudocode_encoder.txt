CLASS ListNode  
    FUNCTION __init__(val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO A new ListNode WITH val EQUALS element at position zero OF values  
    SET p TO head  
    FOR each val IN elements from position one TO the last position OF values  
        SET node TO A new ListNode WITH val EQUALS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN (element val OF p1 EQUALS element val OF p2) AND CALL is_same_list WITH PARAMETERS element next OF p1, element next OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1, l2)  
        SET dummy_head TO A new ListNode WITH val EQUALS zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry IS NOT zero HOLDS  
            IF l1 IS NOT None  
                SET val1 TO element val OF l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT None  
                SET val2 TO element val OF l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET current.next TO A new ListNode WITH val EQUALS total MODULO ten  
            SET current TO element next OF current  
            
            IF l1 IS NOT None  
                SET l1 TO element next OF l1  
            END IF  
            IF l2 IS NOT None  
                SET l2 TO element next OF l2  
            END IF  
        END WHILE  
        
        RETURN element next OF dummy_head  
    END FUNCTION  
END CLASS
