CLASS Solution  
    FUNCTION maxHeight(cuboids)  
        FOR each cuboid IN cuboids  
            CALL sort ON cuboid TO ARRANGE its dimensions IN non-decreasing order  
        END FOR  

        CALL sort ON cuboids TO ARRANGE all cuboids IN lexicographical order  

        SET n TO the LENGTH OF cuboids  
        SET dp TO a list OF size n WHERE each element IS initialized TO zero  

        FOR i FROM zero TO n MINUS one  
            SET dp at position i TO the element at position two OF cuboid at position i  

            FOR j FROM zero TO i MINUS one  
                IF the element at position zero OF cuboid at position j IS LESS THAN OR EQUAL TO the element at position zero OF cuboid at position i  
                AND the element at position one OF cuboid at position j IS LESS THAN OR EQUAL TO the element at position one OF cuboid at position i  
                AND the element at position two OF cuboid at position j IS LESS THAN OR EQUAL TO the element at position two OF cuboid at position i  
                THEN  
                    SET dp at position i TO the maximum BETWEEN dp at position i AND dp at position j PLUS the element at position two OF cuboid at position i  
                END IF  
            END FOR  
        END FOR  

        RETURN the maximum VALUE IN dp  
    END FUNCTION  
END CLASS
