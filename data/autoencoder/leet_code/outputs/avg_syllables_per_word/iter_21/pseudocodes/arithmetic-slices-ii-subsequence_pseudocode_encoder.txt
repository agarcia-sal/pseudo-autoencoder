CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN 3  
            RETURN zero  
        END IF  
        
        SET dp TO a new list CONTAINING n elements EACH BEING a new default dictionary WITH integer default  
        SET total_count TO zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                SET d TO element at position i OF nums MINUS element at position j OF nums  
                
                IF dp at position j CONTAINS key d  
                    INCREMENT dp at position i AND key d BY dp at position j AND key d PLUS one  
                    INCREMENT total_count BY dp at position j AND key d  
                ELSE  
                    INCREMENT dp at position i AND key d BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
