CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a DICTIONARY mapping each character FROM a TO z TO zero  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND the UNICODE code point OF the character at position index OF s MINUS the UNICODE code point OF the character at position index MINUS one OF s EQUALS one OR the character at position index OF s EQUALS a AND the character at position index MINUS one OF s EQUALS z  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET max_length AT the character at position index OF s TO the GREATER VALUE BETWEEN max_length AT the character at position index OF s AND current_length  
        END FOR  
        SET total_unique_substrings TO the SUM OF all values IN max_length  
        RETURN total_unique_substrings  
    END FUNCTION  
END CLASS
