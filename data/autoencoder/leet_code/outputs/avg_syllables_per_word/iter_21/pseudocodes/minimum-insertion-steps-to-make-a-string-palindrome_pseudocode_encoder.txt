CLASS Solution  
    FUNCTION minInsertions(s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO the LENGTH OF s  
            SET dp TO a TWO-DIMENSIONAL ARRAY of size n BY n FILLED WITH zero  
            FOR i FROM zero TO n MINUS one  
                SET the element at position i AND position i OF dp TO one  
            END FOR  
            FOR cl FROM two TO n  
                FOR i FROM zero TO n MINUS cl  
                    SET j TO i PLUS cl MINUS one  
                    IF the character at position i OF s EQUALS the character at position j OF s AND cl EQUALS two  
                        SET the element at position i AND position j OF dp TO two  
                    ELSE IF the character at position i OF s EQUALS the character at position j OF s  
                        SET the element at position i AND position j OF dp TO the element at position i PLUS one AND position j MINUS one OF dp PLUS two  
                    ELSE  
                        SET the element at position i AND position j OF dp TO the maximum OF the element at position i AND position j MINUS one OF dp AND the element at position i PLUS one AND position j OF dp  
                    END IF  
                END FOR  
            END FOR  
            RETURN the element at position zero AND position n MINUS one OF dp  
        END FUNCTION  
        RETURN the LENGTH OF s MINUS CALL longest_palindromic_subsequence WITH PARAMETERS s  
    END FUNCTION  
END CLASS
