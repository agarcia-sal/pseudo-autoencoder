CLASS Solution  
    FUNCTION numTrees(n)  
        CREATE a list named dp WITH n PLUS one elements INITIALIZED TO zero  
        SET element at position zero OF dp TO one  
        SET element at position one OF dp TO one  
        FOR i FROM two TO n  
            FOR j FROM one TO i  
                INCREMENT element at position i OF dp BY element at position j MINUS one OF dp MULTIPLIED BY element at position i MINUS j OF dp  
            END FOR  
        END FOR  
        RETURN element at position n OF dp  
    END FUNCTION  
END CLASS
