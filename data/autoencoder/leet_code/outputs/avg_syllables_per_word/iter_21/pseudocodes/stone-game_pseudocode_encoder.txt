CLASS Solution  
    FUNCTION stoneGame(piles)  
        FUNCTION dfs(i, j)  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            SET left_choice TO element at position i OF piles MINUS CALL dfs WITH PARAMETERS i PLUS one AND j  
            SET right_choice TO element at position j OF piles MINUS CALL dfs WITH PARAMETERS i AND j MINUS one  
            SET result TO the GREATER VALUE BETWEEN left_choice AND right_choice  
            RETURN result  
        END FUNCTION  
        SET total_piles_length TO the LENGTH OF piles MINUS one  
        SET final_result TO CALL dfs WITH PARAMETERS zero AND total_piles_length  
        RETURN final_result GREATER THAN zero  
    END FUNCTION  
END CLASS
