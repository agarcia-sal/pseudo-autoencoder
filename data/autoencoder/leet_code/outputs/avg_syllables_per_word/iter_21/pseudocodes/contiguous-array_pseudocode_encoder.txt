CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary with key zero assigned value minus one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each index i AND element num IN nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count EXISTS AS A KEY IN prefix_sum_index  
                SET max_length TO the GREATER OF max_length AND the VALUE of i MINUS the VALUE associated with count IN prefix_sum_index  
            ELSE  
                ASSIGN prefix_sum_index AT KEY count TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
