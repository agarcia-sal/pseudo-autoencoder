CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF element at position zero OF mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF element at position zero OF mat2  
        
        CALL initialize_result_matrix WITH PARAMETERS m AND n  
        SET result TO the RETURN VALUE OF initialize_result_matrix  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR p FROM zero TO k MINUS one  
                    SET element at position j OF element at position i OF result TO element at position j OF element at position i OF result PLUS element at position p OF element at position i OF mat1 MULTIPLIED BY element at position j OF element at position p OF mat2  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
    
    FUNCTION initialize_result_matrix(m, n)  
        SET matrix TO an empty list  
        FOR index_i FROM zero TO m MINUS one  
            SET row TO an empty list  
            FOR index_j FROM zero TO n MINUS one  
                APPEND zero TO row  
            END FOR  
            APPEND row TO matrix  
        END FOR  
        RETURN matrix  
    END FUNCTION  
END CLASS
