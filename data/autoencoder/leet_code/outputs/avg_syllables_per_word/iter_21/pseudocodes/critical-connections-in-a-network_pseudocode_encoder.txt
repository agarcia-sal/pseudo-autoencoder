CLASS Solution  
    FUNCTION criticalConnections(n, connections)  
        FUNCTION dfs(node, parent, disc, low, time, graph, critical)  
            SET element at position node OF disc TO element at position node OF time at position zero  
            SET element at position node OF low TO element at position node OF time at position zero  
            INCREMENT element at position zero OF time BY one  
              
            FOR each neighbor IN element at position node OF graph  
                IF element at position neighbor OF disc EQUALS minus one THEN  
                    CALL dfs WITH PARAMETERS neighbor, node, disc, low, time, graph, critical  
                    SET element at position node OF low TO the MINIMUM OF element at position node OF low AND element at position neighbor OF low  
                    IF element at position neighbor OF low GREATER THAN element at position node OF disc THEN  
                        APPEND list containing node AND neighbor TO critical  
                    END IF  
                ELSE IF neighbor NOT EQUALS parent THEN  
                    SET element at position node OF low TO the MINIMUM OF element at position node OF low AND element at position neighbor OF disc  
                END IF  
            END FOR  
        END FUNCTION  
      
        SET graph TO result of FUNCTION buildGraph WITH PARAMETERS n, connections  
        SET disc TO result of FUNCTION initializeList WITH PARAMETERS n, minus one  
        SET low TO result of FUNCTION initializeList WITH PARAMETERS n, minus one  
        SET time TO list containing zero  
        SET critical TO empty list  
      
        FOR node FROM zero TO n MINUS one  
            IF element at position node OF disc EQUALS minus one THEN  
                CALL dfs WITH PARAMETERS node, minus one, disc, low, time, graph, critical  
            END IF  
        END FOR  
      
        RETURN critical  
    END FUNCTION  
  
    FUNCTION buildGraph(n, connections)  
        SET graph TO empty list  
        FOR index FROM zero TO n MINUS one  
            APPEND empty list TO graph  
        END FOR  
        FOR each pair u AND v IN connections  
            APPEND v TO element at position u OF graph  
            APPEND u TO element at position v OF graph  
        END FOR  
        RETURN graph  
    END FUNCTION  
  
    FUNCTION initializeList(size, initial_value)  
        SET new_list TO empty list  
        FOR index FROM zero TO size MINUS one  
            APPEND initial_value TO new_list  
        END FOR  
        RETURN new_list  
    END FUNCTION  
END CLASS
