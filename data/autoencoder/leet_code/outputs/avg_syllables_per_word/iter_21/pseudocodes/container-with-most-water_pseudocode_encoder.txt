CLASS Solution  
    FUNCTION maxArea(height)  
        SET left TO zero  
        SET right TO the LENGTH OF height MINUS one  
        SET max_area TO zero  
        
        WHILE left LESS THAN right HOLDS  
            SET smaller_height TO height AT POSITION left IF height AT POSITION left LESS THAN height AT POSITION right OTHERWISE height AT POSITION right  
            SET current_area TO smaller_height MULTIPLIED BY (right MINUS left)  
            SET max_area TO the GREATER VALUE BETWEEN max_area AND current_area  
            
            IF height AT POSITION left LESS THAN height AT POSITION right  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
