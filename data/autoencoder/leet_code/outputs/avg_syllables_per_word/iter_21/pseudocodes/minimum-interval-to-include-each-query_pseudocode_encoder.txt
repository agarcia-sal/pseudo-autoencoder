CLASS Solution  
    FUNCTION minInterval(intervals, queries)  
        CALL intervals.sort WITH PARAMETERS  
        SET sorted_queries TO a list of pairs each composed of element q AND its index i FROM queries in sorted order BY q  

        SET min_heap TO an empty list  
        SET result TO a list composed of negative one repeated the LENGTH OF queries times  

        SET interval_index TO zero  
        FOR each pair composed of query AND original_index IN sorted_queries  
            WHILE interval_index LESS THAN the LENGTH OF intervals AND element at position interval_index OF intervals AT position zero LESS THAN OR EQUAL TO query  
                SET start TO element at position interval_index OF intervals AT position zero  
                SET end TO element at position interval_index OF intervals AT position one  
                SET interval_size TO end MINUS start PLUS one  
                CALL heap push operation ON min_heap WITH PARAMETERS the pair composed of interval_size AND end  
                INCREMENT interval_index BY one  
            END WHILE  

            WHILE the LENGTH OF min_heap GREATER THAN zero AND the element at position zero OF element at position zero OF min_heap LESS THAN query  
                CALL heap pop operation ON min_heap  
            END WHILE  

            IF the LENGTH OF min_heap GREATER THAN zero  
                SET element at position original_index OF result TO element at position zero OF element at position zero OF min_heap  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
