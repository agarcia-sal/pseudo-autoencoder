CLASS Solution  
    FUNCTION minKBitFlips(nums, k)  
        SET n TO the LENGTH OF nums  
        SET flip_count TO zero  
        SET flipped TO a list OF zeros WITH LENGTH n PLUS one  
        SET flip_state TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET flip_state TO the result OF flip_state EXCLUSIVE OR element at position i OF flipped  
            
            IF element at position i OF nums EQUALS flip_state THEN  
                IF i PLUS k GREATER THAN n THEN  
                    RETURN negative one  
                END IF  
                
                SET flip_state TO the result OF flip_state EXCLUSIVE OR one  
                SET element at position i PLUS k OF flipped TO the result OF element at position i PLUS k OF flipped EXCLUSIVE OR one  
                INCREMENT flip_count BY one  
            END IF  
        END FOR  
        
        RETURN flip_count  
    END FUNCTION  
END CLASS
