CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO the LENGTH OF floor  
        
        CALL constructor_dp WITH PARAMETERS n AND numCarpets  
        ASSIGN reference TO dp  
        
        FOR i FROM one TO n  
            SET dp at position i and zero TO dp at position i minus one and zero PLUS the RESULT OF check_tile_at_position i minus one OF floor  
        END FOR  
        
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET option_without_carpet TO dp at position i minus one and j PLUS the RESULT OF check_tile_at_position i minus one OF floor  
                SET option_with_carpet TO dp at position MAXIMUM OF zero AND i minus carpetLen and j minus one  
                SET dp at position i and j TO the MINIMUM OF option_without_carpet AND option_with_carpet  
            END FOR  
        END FOR  
        
        RETURN dp at position n and numCarpets  
    END FUNCTION  
    
    FUNCTION constructor_dp(n, numCarpets)  
        CREATE a two-dimensional list with n plus one rows AND numCarpets plus one columns FILLED WITH zeros  
        RETURN the created two-dimensional list  
    END FUNCTION  
    
    FUNCTION check_tile_at_position(index, floor)  
        IF the character at position index OF floor EQUALS one  
            RETURN one  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
