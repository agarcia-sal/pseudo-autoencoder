CLASS Solution  
    FUNCTION reorderedPowerOf2(n)  
        SET power_of_2_sorted_digits TO an empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 IS LESS THAN OR EQUAL TO one billion  
            CALL conversion of power_of_2 TO string  
            CALL sorting of the string converted from power_of_2 INTO a sequence of characters  
            CALL conversion of the sorted sequence INTO a tuple named sorted_digits  
            ADD sorted_digits TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  
        CALL conversion of n TO string  
        CALL sorting of the string converted from n INTO a sequence of characters  
        CALL conversion of the sorted sequence INTO a tuple named sorted_n_digits  
        IF sorted_n_digits IS CONTAINED IN power_of_2_sorted_digits  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
