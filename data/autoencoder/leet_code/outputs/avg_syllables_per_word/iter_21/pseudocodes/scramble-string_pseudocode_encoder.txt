CLASS Solution  
    FUNCTION isScramble(s1, s2)  
        IF the string s1 EQUALS the string s2  
            RETURN True  
        END IF  
  
        IF the characters of string s1 SORTED ARE NOT EQUAL TO the characters of string s2 SORTED  
            RETURN False  
        END IF  
  
        SET n TO the LENGTH OF string s1  
  
        FOR i FROM one TO n MINUS one  
            IF CALL isScramble WITH PARAMETERS the substring from position zero TO i MINUS one OF string s1 AND the substring from position zero TO i MINUS one OF string s2  
                AND CALL isScramble WITH PARAMETERS the substring from position i TO the END OF string s1 AND the substring from position i TO the END OF string s2  
                RETURN True  
            END IF  
  
            IF CALL isScramble WITH PARAMETERS the substring from position zero TO i MINUS one OF string s1 AND the substring from position n MINUS i TO the END OF string s2  
                AND CALL isScramble WITH PARAMETERS the substring from position i TO the END OF string s1 AND the substring from position zero TO n MINUS i MINUS one OF string s2  
                RETURN True  
            END IF  
        END FOR  
  
        RETURN False  
    END FUNCTION  
END CLASS
