CLASS Solution  
    FUNCTION wiggleMaxLength(nums)  
        IF the LENGTH OF nums LESS THAN 2  
            RETURN the LENGTH OF nums  
        END IF  
        
        SET up TO one  
        SET down TO one  
        
        FOR index FROM one TO the LENGTH OF nums MINUS one  
            IF the element at position index OF nums GREATER THAN the element at position index MINUS one OF nums  
                SET up TO down PLUS one  
            ELSE IF the element at position index OF nums LESS THAN the element at position index MINUS one OF nums  
                SET down TO up PLUS one  
            END IF  
        END FOR  
        
        RETURN the GREATER VALUE OF up AND down  
    END FUNCTION  
END CLASS
