CLASS Solution  
    FUNCTION triangleNumber(nums)  
        CALL nums SORT  
        SET count TO zero  
        SET n TO the LENGTH OF nums  
        
        FOR k FROM n MINUS one TO two STEP MINUS one  
            SET i TO zero  
            SET j TO k MINUS one  
            WHILE i LESS THAN j HOLDS  
                IF element at position i OF nums PLUS element at position j OF nums GREATER THAN element at position k OF nums  
                    INCREMENT count BY j MINUS i  
                    DECREMENT j BY one  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
