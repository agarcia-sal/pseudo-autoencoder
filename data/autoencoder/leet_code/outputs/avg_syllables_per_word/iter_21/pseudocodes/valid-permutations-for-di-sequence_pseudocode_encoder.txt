CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of zeroes with size n PLUS one BY n PLUS one  
        
        FOR each j FROM zero TO n  
            SET element at position j OF element at position zero OF dp TO one  
        END FOR  
        
        FOR each i FROM one TO n  
            SET prefix_sum TO zero  
            
            IF element at position i MINUS one OF s EQUALS the character I  
                FOR each j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position j OF element at position i MINUS one OF dp  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                    SET element at position j OF element at position i OF dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR each j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO prefix_sum PLUS element at position j PLUS one OF element at position i MINUS one OF dp  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                    SET element at position j OF element at position i OF dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO zero  
        FOR each value IN element at position n OF dp  
            SET result TO result PLUS value  
        END FOR  
        SET result TO result MODULO MOD  
        
        RETURN result  
    END FUNCTION  
END CLASS
