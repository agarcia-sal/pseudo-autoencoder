CLASS Solution  
    FUNCTION boxDelivering(boxes WITH TYPE List OF List OF Integer, portsCount WITH TYPE Integer, maxBoxes WITH TYPE Integer, maxWeight WITH TYPE Integer)  
        SET n TO the LENGTH OF boxes  
        SET dp TO a list of size n PLUS one FILLED WITH positive infinity  
        SET the element at position zero OF dp TO zero  
        
        SET left TO zero  
        SET current_weight TO zero  
        SET extra_trips TO zero  
        
        FOR right FROM zero TO n MINUS one  
            INCREMENT current_weight BY the element at position one OF the element at position right OF boxes  
            IF right GREATER THAN zero AND the element at position zero OF the element at position right OF boxes NOT EQUALS the element at position zero OF the element at position right MINUS one OF boxes THEN  
                INCREMENT extra_trips BY one  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND (right MINUS left PLUS one GREATER THAN maxBoxes OR current_weight GREATER THAN maxWeight) HOLDS  
                DECREMENT current_weight BY the element at position one OF the element at position left OF boxes  
                IF left GREATER THAN zero AND the element at position zero OF the element at position left OF boxes NOT EQUALS the element at position zero OF the element at position left MINUS one OF boxes THEN  
                    DECREMENT extra_trips BY one  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET the element at position right PLUS one OF dp TO the element at position left OF dp PLUS two PLUS extra_trips  
        END FOR  
        
        RETURN the element at position n OF dp  
    END FUNCTION  
END CLASS
