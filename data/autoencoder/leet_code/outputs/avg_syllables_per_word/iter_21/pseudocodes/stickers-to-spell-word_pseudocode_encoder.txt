CLASS Solution  
    FUNCTION minStickers(stickers, target)  
        SET stickers TO THE RESULT OF CALLING Counter ON each sticker IN stickers TO GET their CHARACTER FREQUENCY COUNTERS  
        SORT stickers BY THE NEGATION OF THE MAXIMUM VALUE AMONG THE FREQUENCIES OF EACH sticker TO PRIORITIZE stickers WITH THE MOST FREQUENT CHARACTER  
        SET queue TO A NEW DEQUE CONTAINING A TUPLE OF target AND ZERO  
        SET memo TO A DICTIONARY WITH target AS KEY AND ZERO AS VALUE  
        WHILE queue IS NOT EMPTY  
            REMOVE AND RETURN THE LEFTMOST ELEMENT OF queue INTO current_target AND used_stickers  
            FOR each sticker IN stickers  
                IF THE FIRST CHARACTER OF current_target IS NOT PRESENT IN sticker  
                    CONTINUE TO THE NEXT sticker  
                END IF  
                SET remaining TO A NEW Counter CONSTRUCTED FROM current_target  
                SUBTRACT THE COUNTER sticker FROM remaining  
                SET new_target TO A STRING FORMED BY JOINING THE SORTED LIST OF ELEMENTS FROM remaining  
                IF new_target IS AN EMPTY STRING  
                    RETURN used_stickers PLUS ONE  
                END IF  
                IF new_target IS NOT A KEY IN memo  
                    SET memo[new_target] TO used_stickers PLUS ONE  
                    APPEND A TUPLE OF new_target AND used_stickers PLUS ONE TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN MINUS ONE  
    END FUNCTION  
END CLASS
