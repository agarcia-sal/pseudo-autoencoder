CLASS Solution  
    FUNCTION checkInclusion(s1, s2)  
        IMPORT Counter FROM collections  
        
        SET len_s1 TO the LENGTH OF s1  
        SET len_s2 TO the LENGTH OF s2  
        
        IF len_s1 GREATER THAN len_s2  
            RETURN False  
        END IF  
        
        SET count_s1 TO CALL Counter WITH PARAMETERS s1  
        SET count_s2 TO CALL Counter WITH PARAMETERS substring from position zero TO len_s1 MINUS one OF s2  
        
        IF count_s1 EQUALS count_s2  
            RETURN True  
        END IF  
        
        FOR index FROM len_s1 TO len_s2 MINUS one  
            INCREMENT the count of the element at position index OF s2 IN count_s2 BY one  
            DECREMENT the count of the element at position index MINUS len_s1 OF s2 IN count_s2 BY one  
            
            IF the count of the element at position index MINUS len_s1 OF s2 IN count_s2 EQUALS zero  
                REMOVE the element at position index MINUS len_s1 OF s2 FROM count_s2  
            END IF  
            
            IF count_s1 EQUALS count_s2  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
