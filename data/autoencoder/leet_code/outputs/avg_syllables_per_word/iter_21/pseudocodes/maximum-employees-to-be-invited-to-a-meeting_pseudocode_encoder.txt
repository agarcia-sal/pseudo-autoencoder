CLASS Solution  
    FUNCTION maximumInvitations(favorite)  
        SET n TO the LENGTH OF favorite  
        SET graph TO the DEFAULT dictionary with LIST as default value  
        SET in_degree TO a LIST of zero with length n  
        
        FOR index FROM zero TO n MINUS one  
            APPEND index TO the LIST at key element at position index OF favorite IN graph  
            INCREMENT element at position index OF favorite IN in_degree BY one  
        END FOR  
        
        SET mutual_chains TO zero  
        SET visited TO a LIST of FALSE with length n  
        
        FOR index FROM zero TO n MINUS one  
            IF element at position element at position index OF favorite OF favorite EQUALS index AND index LESS THAN element at position index OF favorite  
                SET a TO index  
                SET b TO element at position index OF favorite  
                SET chain_a TO CALL find_chain_length WITH PARAMETERS a graph visited  
                SET chain_b TO CALL find_chain_length WITH PARAMETERS b graph visited  
                INCREMENT mutual_chains BY chain_a PLUS chain_b  
            END IF  
        END FOR  
        
        SET longest_cycle TO zero  
        SET visited TO a LIST of FALSE with length n  
        
        FOR index FROM zero TO n MINUS one  
            IF element at position index OF visited IS FALSE  
                SET cycle_length TO CALL find_cycle_length WITH PARAMETERS index graph visited favorite  
                SET longest_cycle TO the GREATER OF longest_cycle AND cycle_length  
            END IF  
        END FOR  
        
        RETURN the GREATER OF mutual_chains AND longest_cycle  
    END FUNCTION  
    
    FUNCTION find_chain_length(start graph visited)  
        SET length TO zero  
        SET queue TO a DEQUE containing start  
        SET element at position start OF visited TO TRUE  
        
        WHILE queue IS NOT empty  
            SET current TO the FIRST element REMOVED FROM queue  
            FOR each neighbor IN the LIST at key current IN graph  
                IF element at position neighbor OF visited IS FALSE  
                    SET element at position neighbor OF visited TO TRUE  
                    APPEND neighbor TO queue  
                    INCREMENT length BY one  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN length  
    END FUNCTION  
    
    FUNCTION find_cycle_length(start graph visited favorite)  
        IF element at position start OF visited IS TRUE  
            RETURN zero  
        END IF  
        
        SET stack TO an EMPTY LIST  
        SET current TO start  
        
        WHILE element at position current OF visited IS FALSE  
            SET element at position current OF visited TO TRUE  
            APPEND current TO stack  
            SET current TO element at position current OF favorite  
        END WHILE  
        
        IF current IS NOT an ELEMENT IN stack  
            RETURN zero  
        END IF  
        
        SET cycle_start_index TO the POSITION OF current IN stack  
        SET cycle_length TO the LENGTH OF stack MINUS cycle_start_index  
        
        RETURN cycle_length  
    END FUNCTION  
END CLASS
