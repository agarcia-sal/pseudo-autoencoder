CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO the LENGTH OF t  
                APPEND the STRING CREATED BY JOINING all elements OF t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH PARAMETERS i PLUS one  
            IF the element at position i OF t IS an alphabetic character  
                SET the element at position i OF t TO the character OBTAINED BY applying BITWISE XOR of the ASCII code of the element at position i OF t WITH thirty-two  
                CALL dfs WITH PARAMETERS i PLUS one  
            END IF  
        END FUNCTION  

        SET t TO the list CREATED FROM s  
        SET ans TO an empty list  
        CALL dfs WITH PARAMETERS zero  
        RETURN ans  
    END FUNCTION  
END CLASS
