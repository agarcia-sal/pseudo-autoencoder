CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        CALL defaultdict WITH PARAMETERS integer TO CREATE sequence_count  
        FOR index FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO the substring from position index TO index PLUS ten MINUS one OF s  
            INCREMENT the value at key sequence OF sequence_count BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each key_value_pair IN the items OF sequence_count  
            SET seq TO the key OF key_value_pair  
            SET count TO the value OF key_value_pair  
            IF count GREATER THAN one THEN  
                APPEND seq TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
