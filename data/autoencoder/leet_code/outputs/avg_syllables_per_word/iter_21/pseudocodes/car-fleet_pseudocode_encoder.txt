CLASS Solution  
    FUNCTION carFleet(target, position, speed)  
        CALL sorted_zip_list TO CALL sorted WITH PARAMETERS CALL zip WITH PARAMETERS position AND speed AND reverse SET TO true  
        CALL time_to_target TO CALL list_comprehension FOR each pair p AND s IN sorted_zip_list TO SET target MINUS p DIVIDED BY s  
  
        SET fleets TO zero  
        SET current_max_time TO zero  
  
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
  
        RETURN fleets  
    END FUNCTION  
END CLASS
