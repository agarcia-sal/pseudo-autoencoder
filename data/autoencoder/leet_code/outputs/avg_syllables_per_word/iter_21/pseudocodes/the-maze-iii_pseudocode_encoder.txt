CLASS Solution  
    FUNCTION findShortestWay(maze, ball, hole)  
        SET directions TO a list containing tuples of zero AND one AND letter r, one AND zero AND letter d, zero AND negative one AND letter l, negative one AND zero AND letter u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero OF maze  
        SET start TO the tuple conversion of ball  
        SET hole TO the tuple conversion of hole  
        SET pq TO a list containing the tuple of zero, empty string, AND start  
        SET visited TO an empty set  
        
        WHILE pq HOLDS  
            CALL heappop ON pq AND ASSIGN the results TO dist, path, AND a tuple containing x AND y  
            
            IF the tuple containing x AND y EQUALS hole  
                RETURN path  
            END IF  
            
            IF the tuple containing x AND y IS IN visited  
                CONTINUE to the next iteration  
            END IF  
            
            ADD the tuple containing x AND y TO visited  
            
            FOR each tuple containing dx, dy, AND direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the element at position new_x PLUS dx OF maze AT position new_y PLUS dy EQUALS zero  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY one  
                    IF the tuple containing new_x AND new_y EQUALS hole  
                        BREAK from the loop  
                    END IF  
                END WHILE  
                
                IF the tuple containing new_x AND new_y IS NOT IN visited  
                    CALL heappush ON pq WITH PARAMETERS dist PLUS count, path CONCATENATED WITH direction, AND the tuple containing new_x AND new_y  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
