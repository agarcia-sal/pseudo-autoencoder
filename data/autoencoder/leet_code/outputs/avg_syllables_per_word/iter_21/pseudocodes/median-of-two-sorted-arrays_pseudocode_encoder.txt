CLASS Solution  
    FUNCTION findMedianSortedArrays(nums1, nums2)  
        IF the LENGTH OF nums1 GREATER THAN the LENGTH OF nums2  
            SET temporary_variable TO nums1  
            SET nums1 TO nums2  
            SET nums2 TO temporary_variable  
        END IF  
        
        SET x TO the LENGTH OF nums1  
        SET y TO the LENGTH OF nums2  
        SET low TO zero  
        SET high TO x  
        
        WHILE low LESS THAN OR EQUAL TO high HOLDS  
            SET partitionX TO the INTEGER DIVISION OF the SUM OF low AND high BY two  
            SET partitionY TO the INTEGER DIVISION OF the SUM OF x AND y PLUS one BY two MINUS partitionX  
            
            IF partitionX EQUALS zero  
                SET maxLeftX TO negative infinity  
            ELSE  
                SET maxLeftX TO element at position partitionX MINUS one OF nums1  
            END IF  
            
            IF partitionX EQUALS x  
                SET minRightX TO positive infinity  
            ELSE  
                SET minRightX TO element at position partitionX OF nums1  
            END IF  
            
            IF partitionY EQUALS zero  
                SET maxLeftY TO negative infinity  
            ELSE  
                SET maxLeftY TO element at position partitionY MINUS one OF nums2  
            END IF  
            
            IF partitionY EQUALS y  
                SET minRightY TO positive infinity  
            ELSE  
                SET minRightY TO element at position partitionY OF nums2  
            END IF  
            
            IF maxLeftX LESS THAN OR EQUAL TO minRightY AND maxLeftY LESS THAN OR EQUAL TO minRightX  
                IF the REMAINDER OF the SUM OF x AND y DIVIDED BY two EQUALS zero  
                    SET maximum_left TO maxLeftX  
                    IF maxLeftY GREATER THAN maximum_left  
                        SET maximum_left TO maxLeftY  
                    END IF  
                    SET minimum_right TO minRightX  
                    IF minRightY LESS THAN minimum_right  
                        SET minimum_right TO minRightY  
                    END IF  
                    SET median TO the SUM OF maximum_left AND minimum_right DIVIDED BY two  
                    RETURN median  
                ELSE  
                    IF maxLeftX GREATER THAN maxLeftY  
                        RETURN maxLeftX  
                    ELSE  
                        RETURN maxLeftY  
                    END IF  
                END IF  
            ELSE IF maxLeftX GREATER THAN minRightY  
                DECREMENT high BY one PLUS partitionX  
            ELSE  
                INCREMENT low BY one PLUS partitionX  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
