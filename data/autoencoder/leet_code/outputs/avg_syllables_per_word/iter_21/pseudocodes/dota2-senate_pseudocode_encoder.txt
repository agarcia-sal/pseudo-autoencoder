CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        CALL initialize radiant AS empty double-ended queue  
        CALL initialize dire AS empty double-ended queue  
        
        FOR each position index AND senator_character IN senate  
            IF senator_character EQUALS the character representing Radiant PARTY THEN  
                APPEND position index TO radiant  
            ELSE  
                APPEND position index TO dire  
            END IF  
        END FOR  
        
        WHILE radiant IS NOT empty AND dire IS NOT empty  
            SET r_index TO the element REMOVED FROM the front OF radiant  
            SET d_index TO the element REMOVED FROM the front OF dire  
            
            IF r_index LESS THAN d_index THEN  
                APPEND r_index PLUS the LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO dire  
            END IF  
        END WHILE  
        
        IF radiant IS NOT empty THEN  
            RETURN the string indicating Radiant PARTY victory  
        ELSE  
            RETURN the string indicating Dire PARTY victory  
        END IF  
    END FUNCTION  
END CLASS
