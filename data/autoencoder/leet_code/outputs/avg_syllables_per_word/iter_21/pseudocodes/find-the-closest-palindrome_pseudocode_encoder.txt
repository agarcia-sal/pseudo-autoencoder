CLASS Solution  
    FUNCTION nearestPalindromic(n)  
        SET length TO the LENGTH OF n  
        IF length EQUALS one  
            SET original_number TO the INTEGER VALUE OF n  
            SET result TO original_number MINUS one  
            RETURN the STRING REPRESENTATION OF result  
        END IF  
        
        SET candidates TO an EMPTY SET  
        
        ADD TEN RAISED TO THE POWER OF length PLUS one TO candidates  
        ADD TEN RAISED TO THE POWER OF length MINUS one MINUS one TO candidates  
        
        SET prefix TO the INTEGER VALUE OF the SUBSTRING from position zero TO the INTEGER VALUE OF the QUOTIENT OF length PLUS one DIVIDED BY two OF n  
        
        FOR i FROM negative one TO positive one  
            SET new_prefix TO the STRING REPRESENTATION OF prefix PLUS i  
            
            IF length MODULO two EQUALS zero  
                SET candidate TO new_prefix CONCATENATED WITH the REVERSE OF new_prefix  
            ELSE  
                SET candidate TO new_prefix CONCATENATED WITH the SUBSTRING from the POSITION OF the LENGTH OF new_prefix MINUS two DOWN TO zero IN REVERSE ORDER OF new_prefix  
            END IF  
            
            ADD the INTEGER VALUE OF candidate TO candidates  
        END FOR  
        
        SET original_number TO the INTEGER VALUE OF n  
        REMOVE original_number FROM candidates IF IT EXISTS  
        
        SET closest TO the ELEMENT OF candidates WITH THE MINIMUM VALUE ACCORDING TO THE TUPLE OF THE ABSOLUTE VALUE OF element MINUS original_number AND element ITSELF  
        
        RETURN the STRING REPRESENTATION OF closest  
    END FUNCTION  
END CLASS
