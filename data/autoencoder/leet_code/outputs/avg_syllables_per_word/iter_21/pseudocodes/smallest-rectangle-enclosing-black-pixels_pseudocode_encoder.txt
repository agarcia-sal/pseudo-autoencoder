CLASS Solution  
    FUNCTION minArea(image, x, y)  
        IF image DOES NOT EXIST OR the LENGTH OF image EQUALS zero OR the LENGTH OF element at position zero OF image EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF image  
        SET n TO the LENGTH OF element at position zero OF image  
        
        FUNCTION hasBlackPixelInRow(row)  
            FOR each pixel IN element at position row OF image  
                IF pixel EQUALS the character one THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION hasBlackPixelInColumn(col)  
            FOR row FROM zero TO m MINUS one  
                IF element at position col OF element at position row OF image EQUALS the character one THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION findTop()  
            SET low TO zero  
            SET high TO x  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the SUM of low AND high BY two  
                IF CALL hasBlackPixelInRow WITH PARAMETERS mid IS True THEN  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findBottom()  
            SET low TO x  
            SET high TO m MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the SUM of low, high, AND one BY two  
                IF CALL hasBlackPixelInRow WITH PARAMETERS mid IS True THEN  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findLeft()  
            SET low TO zero  
            SET high TO y  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the SUM of low AND high BY two  
                IF CALL hasBlackPixelInColumn WITH PARAMETERS mid IS True THEN  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findRight()  
            SET low TO y  
            SET high TO n MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF the SUM of low, high, AND one BY two  
                IF CALL hasBlackPixelInColumn WITH PARAMETERS mid IS True THEN  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        SET top TO CALL findTop WITH NO PARAMETERS  
        SET bottom TO CALL findBottom WITH NO PARAMETERS  
        SET left TO CALL findLeft WITH NO PARAMETERS  
        SET right TO CALL findRight WITH NO PARAMETERS  
        
        RETURN the PRODUCT of the SUM of bottom MINUS top PLUS one AND the SUM of right MINUS left PLUS one  
    END FUNCTION  
END CLASS
