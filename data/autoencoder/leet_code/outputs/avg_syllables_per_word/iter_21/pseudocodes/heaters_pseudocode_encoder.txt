CLASS Solution  
    FUNCTION findRadius(houses, heaters)  
        CALL the METHOD sort ON houses  
        CALL the METHOD sort ON heaters  
          
        SET min_radius TO zero  
        SET j TO zero  
          
        FOR each house IN houses  
            WHILE j IS LESS THAN the LENGTH OF heaters MINUS one AND the ABSOLUTE VALUE OF the RESULT OF heaters AT POSITION j MINUS house IS GREATER THAN OR EQUAL TO the ABSOLUTE VALUE OF the RESULT OF heaters AT POSITION j PLUS one MINUS house HOLDS  
                INCREMENT j BY one  
            END WHILE  
              
            SET min_radius TO the GREATER VALUE BETWEEN min_radius AND the ABSOLUTE VALUE OF the RESULT OF heaters AT POSITION j MINUS house  
        END FOR  
          
        RETURN min_radius  
    END FUNCTION  
END CLASS
