CLASS Solution  
    FUNCTION maxRunTime(n, batteries)  
        FUNCTION canRunFor(time)  
            SET total TO zero  
            FOR each battery IN batteries  
                INCREMENT total BY the smaller VALUE BETWEEN battery AND time  
            END FOR  
            RETURN total GREATER THAN OR EQUAL TO time MULTIPLIED BY n  
        END FUNCTION  

        SET left TO zero  
        SET right TO the sum of all elements IN batteries DIVIDED BY n  
        WHILE left LESS THAN right HOLDS  
            SET mid TO the integer result of left PLUS right PLUS one DIVIDED BY two  
            IF canRunFor(mid) HOLDS  
                SET left TO mid  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
