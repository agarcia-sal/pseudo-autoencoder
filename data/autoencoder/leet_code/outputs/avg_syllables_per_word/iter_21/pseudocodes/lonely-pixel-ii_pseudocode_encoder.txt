CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture EQUALS nothing OR element at position zero OF picture EQUALS nothing  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero OF picture  
        SET row_count TO a list of zeros with the LENGTH OF m  
        SET col_count TO a list of zeros with the LENGTH OF n  
        CALL defaultdict WITH PARAMETERS integer AND ASSIGN reference TO row_patterns  
        
        FOR r FROM zero TO m MINUS one  
            SET pattern TO the CONCATENATION OF all elements IN element at position r OF picture  
            INCREMENT the value at key pattern IN row_patterns BY one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c OF element at position r OF picture EQUALS the character B  
                    INCREMENT element at position r OF row_count BY one  
                    INCREMENT element at position c OF col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c OF element at position r OF picture EQUALS the character B  
                    SET pattern TO the CONCATENATION OF all elements IN element at position r OF picture  
                    IF element at position r OF row_count EQUALS target AND element at position c OF col_count EQUALS target AND the value at key pattern IN row_patterns EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
