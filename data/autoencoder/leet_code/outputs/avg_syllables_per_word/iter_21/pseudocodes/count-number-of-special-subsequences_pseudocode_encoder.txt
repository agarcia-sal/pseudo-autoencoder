CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a list containing zero at position zero, zero at position one, zero at position two  
        
        FOR each num IN nums  
            IF num EQUALS zero  
                SET element at position zero OF dp TO the REST OF (two MULTIPLIED BY the element at position zero OF dp PLUS one) DIVIDED BY MODULO MOD  
            ELSE IF num EQUALS one  
                SET element at position one OF dp TO the REST OF (two MULTIPLIED BY the element at position one OF dp PLUS the element at position zero OF dp) DIVIDED BY MODULO MOD  
            ELSE IF num EQUALS two  
                SET element at position two OF dp TO the REST OF (two MULTIPLIED BY the element at position two OF dp PLUS the element at position one OF dp) DIVIDED BY MODULO MOD  
            END IF  
        END FOR  
        
        RETURN element at position two OF dp  
    END FUNCTION  
END CLASS
