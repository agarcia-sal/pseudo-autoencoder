CLASS Solution  
    FUNCTION rotate(matrix)  
        SET n TO the LENGTH OF matrix  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET temporary_variable TO the element at position j of the element at position i OF matrix  
                SET the element at position j of the element at position i OF matrix TO the element at position i of the element at position j OF matrix  
                SET the element at position i of the element at position j OF matrix TO temporary_variable  
            END FOR  
        END FOR  
        FOR i FROM zero TO n MINUS one  
            SET left_index TO zero  
            SET right_index TO n MINUS one  
            WHILE left_index IS LESS THAN right_index HOLDS  
                SET temporary_variable TO the element at position right_index OF the element at position i OF matrix  
                SET the element at position right_index OF the element at position i OF matrix TO the element at position left_index OF the element at position i OF matrix  
                SET the element at position left_index OF the element at position i OF matrix TO temporary_variable  
                INCREMENT left_index BY one  
                DECREMENT right_index BY one  
            END WHILE  
        END FOR  
    END FUNCTION  
END CLASS
