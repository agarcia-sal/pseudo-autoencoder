CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO CALL ListNode WITH PARAMETERS element at position zero OF values  
    SET p TO head  
    FOR each val IN elements FROM position one TO end OF values  
        SET node TO CALL ListNode WITH PARAMETERS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    IF value val OF p1 EQUALS value val OF p2 AND CALL is_same_list WITH PARAMETERS next OF p1 AND next OF p2  
        RETURN True  
    ELSE  
        RETURN False  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION partition(self, head, x)  
        SET less_head TO CALL ListNode WITH PARAMETERS zero  
        SET greater_head TO CALL ListNode WITH PARAMETERS zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current IS NOT None  
            IF value val OF current IS LESS THAN x  
                SET next OF less TO current  
                SET less TO next OF less  
            ELSE  
                SET next OF greater TO current  
                SET greater TO next OF greater  
            END IF  
            SET current TO next OF current  
        END WHILE  
        SET next OF less TO next OF greater_head  
        SET next OF greater TO None  
        RETURN next OF less_head  
    END FUNCTION  
END CLASS
