CLASS Solution  
    FUNCTION reverseWords(s)  
        CALL s REVERSE  
        SET start TO zero  
        FOR end FROM zero TO the LENGTH OF s MINUS one  
            IF element at position end OF s EQUALS a space character  
                CALL the segment of s FROM position start TO position end MINUS one REVERSE AND ASSIGN TO the same segment of s  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        CALL the segment of s FROM position start TO the END REVERSE AND ASSIGN TO the same segment of s  
    END FUNCTION  
END CLASS
