CLASS Solution  
    FUNCTION getFactors(n)  
        FUNCTION backtrack(start, target)  
            FOR i FROM start TO THE INTEGER PART OF THE SQUARE ROOT OF target PLUS one  
                IF target MODULO i EQUALS zero  
                    APPEND the elements of path FOLLOWED BY i AND target DIVIDED BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH PARAMETERS i AND target DIVIDED BY i  
                    REMOVE the last element FROM path  
                END IF  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        SET path TO an empty list  
        CALL backtrack WITH PARAMETERS two AND n  
        RETURN result  
    END FUNCTION  
END CLASS
