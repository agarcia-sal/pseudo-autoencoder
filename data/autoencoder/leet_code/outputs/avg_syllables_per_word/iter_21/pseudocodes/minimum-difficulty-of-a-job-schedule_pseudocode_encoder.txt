CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO the LENGTH OF jobDifficulty  
        
        IF n LESS THAN d  
            RETURN negative one  
        END IF  
        
        FUNCTION dp(i, d)  
            IF d EQUALS one  
                RETURN the MAXIMUM VALUE IN the SUBSEQUENCE of jobDifficulty FROM position i TO the END  
            END IF  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            
            FOR j FROM i TO n MINUS d  
                SET current_max TO the MAXIMUM VALUE BETWEEN current_max AND the ELEMENT at position j OF jobDifficulty  
                SET min_difficulty TO the MINIMUM VALUE BETWEEN min_difficulty AND current_max PLUS CALL dp WITH PARAMETERS j PLUS one AND d MINUS one  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN CALL dp WITH PARAMETERS zero AND d  
    END FUNCTION  
END CLASS
