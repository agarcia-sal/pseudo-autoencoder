CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a new list of zeroes with LENGTH n  

        SET force TO zero  
        FOR i FROM zero TO n MINUS one  
            IF the element at position i OF dominoes EQUALS the character representing pushing to the right  
                SET force TO n  
            ELSE IF the element at position i OF dominoes EQUALS the character representing pushing to the left  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER OF force MINUS one AND zero  
            END IF  
            INCREMENT the element at position i OF forces BY force  
        END FOR  

        SET force TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            IF the element at position i OF dominoes EQUALS the character representing pushing to the left  
                SET force TO n  
            ELSE IF the element at position i OF dominoes EQUALS the character representing pushing to the right  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER OF force MINUS one AND zero  
            END IF  
            DECREMENT the element at position i OF forces BY force  
        END FOR  

        SET result TO an empty list  
        FOR each f IN forces  
            IF f GREATER THAN zero  
                APPEND the character representing pushing to the right TO result  
            ELSE IF f LESS THAN zero  
                APPEND the character representing pushing to the left TO result  
            ELSE  
                APPEND the character representing no push TO result  
            END IF  
        END FOR  

        RETURN the concatenation of all elements IN result  
    END FUNCTION  
END CLASS
