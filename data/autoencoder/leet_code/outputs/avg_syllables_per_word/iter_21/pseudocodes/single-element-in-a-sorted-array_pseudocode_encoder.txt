CLASS Solution  
    FUNCTION singleNonDuplicate(nums)  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        
        WHILE left LESS THAN right HOLDS  
            SET mid TO left PLUS the RESULT OF right MINUS left DIVIDED BY two  
            
            IF mid MODULO two EQUALS one THEN  
                DECREMENT mid BY one  
            END IF  
            
            IF element at position mid OF nums EQUALS element at position mid PLUS one OF nums THEN  
                SET left TO mid PLUS two  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN element at position left OF nums  
    END FUNCTION  
END CLASS
