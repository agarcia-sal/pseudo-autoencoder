CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        
        CALL FUNCTION initialize_subset_sums WITH PARAMETERS n AND jobs AND STORE RESULT IN subset_sums
        
        FUNCTION initialize_subset_sums(n, jobs)  
            SET subset_sums TO a list of zero VALUES WITH LENGTH EQUAL TO two RAISED TO THE POWER OF n  
            FOR i FROM zero TO n MINUS one  
                FOR mask FROM zero TO two RAISED TO THE POWER OF i MINUS one  
                    SET element at position mask OR two RAISED TO THE POWER OF i OF subset_sums TO element at position mask OF subset_sums PLUS element at position i OF jobs  
                END FOR  
            END FOR  
            RETURN subset_sums  
        END FUNCTION
        
        FUNCTION dp(k, mask)  
            IF k EQUALS one  
                RETURN element at position mask OF subset_sums  
            END IF  
            SET res TO positive infinity  
            SET submask TO mask  
            WHILE submask DOES NOT EQUAL zero  
                SET candidate TO maximum of CALL dp WITH PARAMETERS k MINUS one AND mask XOR submask AND element at position submask OF subset_sums  
                SET res TO minimum of res AND candidate  
                SET submask TO submask MINUS one BITWISE AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION
        
        RETURN CALL dp WITH PARAMETERS k AND two RAISED TO THE POWER OF n MINUS one  
    END FUNCTION  
END CLASS
