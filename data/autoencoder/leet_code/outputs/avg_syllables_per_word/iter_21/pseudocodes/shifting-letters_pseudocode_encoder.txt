CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one DOWN TO zero  
            INCREMENT total_shifts BY element at position index OF shifts  
            SET element at position index OF shifts TO total_shifts  
        END FOR  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET original_character_code TO the numerical Unicode code of element at position index OF s  
            SET base_character_code TO the numerical Unicode code of the lowercase letter a  
            SET shifted_value TO original_character_code MINUS base_character_code PLUS element at position index OF shifts  
            SET wrapped_value TO the remainder after dividing shifted_value BY twenty six  
            SET new_character_code TO wrapped_value PLUS base_character_code  
            SET new_character TO the character represented by new_character_code  
            APPEND new_character TO result  
        END FOR  
        RETURN the concatenation of all elements OF result as a single string  
    END FUNCTION  
END CLASS
