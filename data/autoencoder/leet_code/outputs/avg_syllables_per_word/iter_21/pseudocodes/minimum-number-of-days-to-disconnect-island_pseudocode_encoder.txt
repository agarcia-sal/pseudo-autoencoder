CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero OF grid OR the TUPLE of x AND y IS IN visited OR element at position x OF grid AT position y EQUALS zero  
                RETURN  
            END IF  
            CALL ADD the TUPLE of x AND y TO visited  
            FOR each PAIR dx AND dy IN the LIST containing the TUPLES of MINUS one AND zero, one AND zero, zero AND MINUS one, AND zero AND one  
                CALL dfs WITH PARAMETERS x PLUS dx, y PLUS dy, visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO empty SET  
            SET land_cells TO the LIST of the TUPLES of i AND j FOR each i FROM zero TO the LENGTH OF grid MINUS one AND FOR each j FROM zero TO the LENGTH OF element at position zero OF grid MINUS one WHERE element at position i OF grid AT position j EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero  
                RETURN True  
            END IF  
            CALL dfs WITH PARAMETERS the element at position zero OF element at position zero OF land_cells, the element at position one OF element at position zero OF land_cells, visited  
            RETURN whether the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF NOT the CALL TO is_connected()  
            RETURN zero  
        END IF  

        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero OF grid MINUS one  
                IF element at position i OF grid AT position j EQUALS one  
                    SET element at position i OF grid AT position j TO zero  
                    IF NOT the CALL TO is_connected()  
                        RETURN one  
                    END IF  
                    SET element at position i OF grid AT position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
