CLASS Solution  
    FUNCTION minSkips(dist, speed, hoursBefore)  
        SET n TO the LENGTH OF dist  
        CALL dp TO the FUNCTION create_two_dimensional_list WITH PARAMETERS n AND n PLUS one AND math.inf  
        SET total_dist TO the SUM OF all elements IN dist  
        
        IF total_dist DIVIDED BY speed IS GREATER THAN hoursBefore  
            RETURN negative one  
        END IF  
        
        SET the element at position zero zero OF dp TO the element at position zero OF dist  
        FOR j FROM one TO n MINUS one  
            SET the element at position zero j OF dp TO the element at position zero OF dist  
        END FOR  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j IS LESS THAN i  
                    SET the element at position i j OF dp TO the VALUE OF math.ceil OF the element at position i MINUS one j OF dp DIVIDED BY speed MULTIPLIED BY speed PLUS the element at position i OF dist  
                END IF  
                IF j IS GREATER THAN zero  
                    SET the element at position i j OF dp TO the MINIMUM OF the current element at position i j OF dp AND the element at position i MINUS one j MINUS one OF dp PLUS the element at position i OF dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            IF the element at position n MINUS one j OF dp IS LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
