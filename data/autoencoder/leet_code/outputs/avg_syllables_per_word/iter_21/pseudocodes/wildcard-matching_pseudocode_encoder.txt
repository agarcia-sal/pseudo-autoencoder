CLASS Solution  
    FUNCTION isMatch(s, p)  
        CALL initialize_dp_table WITH PARAMETERS s, p AND ASSIGN TO dp  
        
        SET element at position zero OF dp at position zero TO True  
        
        FOR j FROM one TO the LENGTH OF p  
            IF element at position j MINUS one OF p EQUALS the character Asterisk  
                SET element at position zero OF dp at position j TO element at position zero OF dp at position j MINUS one  
            END IF  
        END FOR  
        
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM one TO the LENGTH OF p  
                IF element at position j MINUS one OF p EQUALS the character Asterisk  
                    SET element at position i OF dp at position j TO element at position i MINUS one OF dp at position j OR element at position i OF dp at position j MINUS one  
                ELSE IF element at position j MINUS one OF p EQUALS the character Question Mark OR element at position i MINUS one OF s EQUALS element at position j MINUS one OF p  
                    SET element at position i OF dp at position j TO element at position i MINUS one OF dp at position j MINUS one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position the LENGTH OF s OF dp at position the LENGTH OF p  
    END FUNCTION  
END CLASS
