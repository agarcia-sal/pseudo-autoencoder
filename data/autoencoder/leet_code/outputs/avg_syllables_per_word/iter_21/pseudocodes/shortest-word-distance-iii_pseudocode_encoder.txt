CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict, word1, word2)  
        SET min_distance TO positive infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO the condition that word1 EQUALS word2  
        
        FOR each index i AND element word IN wordsDict  
            IF word EQUALS word1  
                IF same_word HOLDS  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO i  
            END IF  
            
            IF word EQUALS word2  
                IF NOT same_word HOLDS  
                    SET prev_word2 TO i  
                END IF  
            END IF  
            
            IF prev_word1 NOT EQUALS negative one AND prev_word2 NOT EQUALS negative one  
                SET min_distance TO the smaller VALUE BETWEEN min_distance AND the ABSOLUTE VALUE OF prev_word1 MINUS prev_word2  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
