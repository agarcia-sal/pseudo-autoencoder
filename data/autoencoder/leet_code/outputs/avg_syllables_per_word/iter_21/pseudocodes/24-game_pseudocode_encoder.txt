CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the difference BETWEEN the element at position zero OF nums AND twenty four IS LESS THAN one MULTIPLIED BY ten TO THE POWER OF negative six  
            END IF  
            
            FOR i FROM zero TO the LENGTH OF nums MINUS one  
                FOR j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        CALL create_new_nums WITH PARAMETERS nums AND i AND j  
                        SET new_nums TO THE RETURNED RESULT  
                        
                        FOR each operation IN the list consisting of true division multiplication addition subtraction  
                            IF (operation IS addition OR operation IS multiplication) AND j GREATER THAN i  
                                CONTINUE TO the next iteration  
                            END IF  
                            
                            IF operation IS NOT true division OR the element at position j OF nums NOT EQUALS zero  
                                CALL apply_operation WITH PARAMETERS operation AND the elements at positions i AND j OF nums  
                                APPEND the result TO new_nums  
                                
                                IF solve(new_nums) HOLDS  
                                    RETURN True  
                                END IF  
                                
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        FUNCTION create_new_nums(nums, i, j)  
            SET result TO an empty list  
            FOR k FROM zero TO the LENGTH OF nums MINUS one  
                IF k NOT EQUALS i AND k NOT EQUALS j  
                    APPEND the element at position k OF nums TO result  
                END IF  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        FUNCTION apply_operation(operation, value1, value2)  
            IF operation IS true division  
                RETURN value1 DIVIDED BY value2  
            ELSE IF operation IS multiplication  
                RETURN value1 MULTIPLIED BY value2  
            ELSE IF operation IS addition  
                RETURN value1 PLUS value2  
            ELSE IF operation IS subtraction  
                RETURN value1 MINUS value2  
            END IF  
        END FUNCTION  
        
        CALL solve WITH PARAMETERS cards  
        RETURN the result of solve  
    END FUNCTION  
END CLASS
