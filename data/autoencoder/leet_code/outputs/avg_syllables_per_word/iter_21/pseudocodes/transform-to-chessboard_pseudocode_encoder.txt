CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF the XOR of element at position i and j OF board AND element at position i and zero OF board AND element at position zero and j OF board AND element at position zero and zero OF board IS NOT EQUAL TO zero  
                    RETURN negative one  
                END IF  
            END FOR  
            
            INCREMENT rowSum BY element at position i and zero OF board  
            INCREMENT colSum BY element at position zero and i OF board  
            INCREMENT rowSwap BY one IF element at position i and zero OF board EQUALS the remainder of i DIVIDED BY two ELSE zero  
            INCREMENT colSwap BY one IF element at position zero and i OF board EQUALS the remainder of i DIVIDED BY two ELSE zero  
        END FOR  
        
        IF the INTEGER DIVISION of N BY two IS GREATER THAN rowSum OR rowSum IS GREATER THAN the INTEGER DIVISION of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the INTEGER DIVISION of N BY two IS GREATER THAN colSum OR colSum IS GREATER THAN the INTEGER DIVISION of N PLUS one BY two  
            RETURN negative one  
        END IF  
        
        IF the remainder of N DIVIDED BY two IS NOT EQUAL TO zero  
            IF the remainder of rowSwap DIVIDED BY two IS NOT EQUAL TO zero  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of colSwap DIVIDED BY two IS NOT EQUAL TO zero  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the MINIMUM BETWEEN rowSwap AND N MINUS rowSwap  
            SET colSwap TO the MINIMUM BETWEEN colSwap AND N MINUS colSwap  
        END IF  
        
        SET result TO the INTEGER DIVISION of the SUM of rowSwap AND colSwap BY two  
        RETURN result  
    END FUNCTION  
END CLASS
