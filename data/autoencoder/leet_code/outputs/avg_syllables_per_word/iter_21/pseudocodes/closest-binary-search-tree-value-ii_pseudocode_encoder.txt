CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY THEN  
        RETURN None  
    END IF  
    SET root TO CALL TreeNode WITH PARAMETERS val EQUALS element at position zero OF values  
    SET i TO one  
    SET queue TO a new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None THEN  
            SET node.left TO CALL TreeNode WITH PARAMETERS val EQUALS element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None THEN  
            SET node.right TO CALL TreeNode WITH PARAMETERS val EQUALS element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val THEN  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left, q.left AND CALL is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues(root, target, k)  
        FUNCTION inorder_traversal(node)  
            IF node IS None THEN  
                RETURN  
            END IF  
            CALL inorder_traversal WITH PARAMETERS node.left  
            IF the LENGTH OF result IS LESS THAN k THEN  
                APPEND node.val TO result  
            ELSE  
                IF the absolute value OF node.val MINUS target IS LESS THAN the absolute value OF the element at position zero OF result MINUS target THEN  
                    REMOVE the element from the front OF result  
                    APPEND node.val TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal WITH PARAMETERS node.right  
        END FUNCTION  
      
        SET result TO a new empty double-ended queue  
        CALL inorder_traversal WITH PARAMETERS root  
        RETURN a new list created from result  
    END FUNCTION  
END CLASS
