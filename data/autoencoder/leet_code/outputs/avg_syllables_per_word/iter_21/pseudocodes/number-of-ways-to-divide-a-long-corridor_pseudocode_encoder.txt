CLASS Solution  
    FUNCTION numberOfWays(corridor)  
        SET MOD TO one billion PLUS seven  
        SET seats TO the list of positions i FOR each character x AND index i IN corridor WHERE character x EQUALS seat indicator S  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM two TO the LENGTH OF seats MINUS one STEP two  
            SET difference TO the element at position index OF seats MINUS the element at position index MINUS one OF seats  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
