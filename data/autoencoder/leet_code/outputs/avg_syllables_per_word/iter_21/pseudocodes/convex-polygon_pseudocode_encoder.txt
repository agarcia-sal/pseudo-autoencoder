CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            RETURN the DIFFERENCE BETWEEN the PRODUCT OF THE DIFFERENCE OF the element at position zero OF a AND the element at position zero OF o AND the DIFFERENCE OF the element at position one OF b AND the element at position one OF o AND THE PRODUCT OF THE DIFFERENCE OF the element at position one OF a AND the element at position one OF o AND THE DIFFERENCE OF the element at position zero OF b AND the element at position zero OF o  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  
        
        SET orientation TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET o TO element at position i OF points  
            SET a TO element at position THE REMAINDER OF i PLUS one DIVIDED BY n OF points  
            SET b TO element at position THE REMAINDER OF i PLUS two DIVIDED BY n OF points  
            SET cross TO CALL cross_product WITH PARAMETERS o a b  
            
            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS negative one AND cross GREATER THAN zero  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
