CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    ASSIGN root TO NEW TreeNode WITH PARAMETERS element at position zero OF values  
    SET i TO one  
    SET queue TO a new deque collection  
    CALL queue append WITH PARAMETERS root  
    WHILE queue HOLDS  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO NEW TreeNode WITH PARAMETERS element at position i OF values  
            CALL queue append WITH PARAMETERS node.left  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO NEW TreeNode WITH PARAMETERS element at position i OF values  
            CALL queue append WITH PARAMETERS node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left, q.left AND CALL is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive(self, root)  
        IF root IS none  
            RETURN zero  
        END IF  
  
        FUNCTION dfs(node, parent, current_length)  
            IF node IS none  
                RETURN current_length  
            END IF  
  
            IF parent IS NOT none AND node.val EQUALS parent PLUS one  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
  
            SET left_length TO CALL dfs WITH PARAMETERS node.left, node.val, current_length  
            SET right_length TO CALL dfs WITH PARAMETERS node.right, node.val, current_length  
  
            RETURN the maximum value AMONG current_length, left_length, right_length  
        END FUNCTION  
  
        RETURN CALL dfs WITH PARAMETERS root, none, zero  
    END FUNCTION  
END CLASS
