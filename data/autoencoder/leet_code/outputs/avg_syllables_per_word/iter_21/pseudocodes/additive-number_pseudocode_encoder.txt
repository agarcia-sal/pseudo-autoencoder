CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            
            SET expected TO the STRING REPRESENTATION OF first PLUS second  
            SET length TO the LENGTH OF expected  
            
            IF start PLUS length GREATER THAN the LENGTH OF num OR the SUBSTRING OF num FROM position start TO start PLUS length NOT EQUALS expected  
                RETURN False  
            END IF  
            
            RETURN CALL is_valid_sequence WITH PARAMETERS start PLUS length, second, the INTEGER REPRESENTATION OF expected  
        END FUNCTION  
        
        SET n TO the LENGTH OF num  
        FOR i FROM one TO n DIVIDED BY two PLUS one  
            FOR j FROM i PLUS one TO n  
                SET first TO the SUBSTRING OF num FROM position zero TO i  
                SET second TO the SUBSTRING OF num FROM position i TO j  
                
                IF the LENGTH OF first GREATER THAN one AND the CHARACTER at position zero OF first EQUALS zero OR the LENGTH OF second GREATER THAN one AND the CHARACTER at position zero OF second EQUALS zero  
                    CONTINUE TO next iteration of inner FOR  
                END IF  
                
                IF CALL is_valid_sequence WITH PARAMETERS j, the INTEGER REPRESENTATION OF first, the INTEGER REPRESENTATION OF second  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
