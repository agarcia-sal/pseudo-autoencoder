CLASS Solution  
    FUNCTION threeEqualParts(arr)  
        SET total_ones TO the SUM OF all elements IN arr  
        
        IF total_ones EQUALS zero  
            RETURN a list CONTAINING zero AND two  
        END IF  
        
        IF total_ones MODULO three NOT EQUALS zero  
            RETURN a list CONTAINING negative one AND negative one  
        END IF  
        
        SET part_ones TO total_ones DIVIDED BY three  
        
        SET first_part_index TO zero  
        SET count_ones TO zero  
        FOR each index i FROM zero TO the LENGTH OF arr MINUS one  
            IF element at position i OF arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS one  
                    SET first_part_index TO i  
                END IF  
                IF count_ones EQUALS part_ones  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET second_part_index TO zero  
        SET count_ones TO zero  
        FOR each index i FROM zero TO the LENGTH OF arr MINUS one  
            IF element at position i OF arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS part_ones PLUS one  
                    SET second_part_index TO i  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET third_part_index TO zero  
        SET count_ones TO zero  
        FOR each index i FROM zero TO the LENGTH OF arr MINUS one  
            IF element at position i OF arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS two MULTIPLIED BY part_ones PLUS one  
                    SET third_part_index TO i  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET n TO the LENGTH OF arr MINUS third_part_index  
        
        IF first_part_index PLUS n LESS THAN OR EQUAL TO the LENGTH OF arr  
             AND second_part_index PLUS n LESS THAN OR EQUAL TO the LENGTH OF arr  
             AND the sublist from first_part_index TO first_part_index PLUS n MINUS one OF arr EQUALS the sublist from second_part_index TO second_part_index PLUS n MINUS one OF arr  
             AND the sublist from second_part_index TO second_part_index PLUS n MINUS one OF arr EQUALS the sublist from third_part_index TO the END OF arr  
            RETURN a list CONTAINING first_part_index PLUS n MINUS one AND second_part_index PLUS n  
        END IF  
        
        RETURN a list CONTAINING negative one AND negative one  
    END FUNCTION  
END CLASS
