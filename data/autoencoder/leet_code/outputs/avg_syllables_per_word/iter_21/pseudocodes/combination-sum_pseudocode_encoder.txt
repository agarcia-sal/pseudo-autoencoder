CLASS Solution  
    FUNCTION combinationSum(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                CALL backtrack WITH PARAMETERS i, target MINUS element at position i OF candidates, path PLUS element at position i OF candidates AS A NEW LIST  
            END FOR  
        END FUNCTION  

        SET result TO an empty list  
        CALL backtrack WITH PARAMETERS zero, target, an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
