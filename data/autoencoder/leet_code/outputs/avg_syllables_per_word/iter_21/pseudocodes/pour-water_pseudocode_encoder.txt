CLASS Solution  
    FUNCTION pourWater(heights, volume, k)  
        SET n TO the LENGTH OF heights  
        
        FOR each unnamed_element FROM one TO volume  
            SET i TO k  
            WHILE i IS GREATER THAN zero AND the element at position i MINUS one OF heights IS LESS THAN OR EQUAL TO the element at position i OF heights  
                DECREMENT i BY one  
            END WHILE  
            
            WHILE i IS LESS THAN k AND the element at position i OF heights EQUALS the element at position i PLUS one OF heights  
                INCREMENT i BY one  
            END WHILE  
            
            IF i IS LESS THAN k  
                INCREMENT the element at position i OF heights BY one  
                CONTINUE TO the next unnamed_element OF the FOR loop  
            END IF  
            
            SET i TO k  
            WHILE i IS LESS THAN n MINUS one AND the element at position i PLUS one OF heights IS LESS THAN OR EQUAL TO the element at position i OF heights  
                INCREMENT i BY one  
            END WHILE  
            
            WHILE i IS GREATER THAN k AND the element at position i OF heights EQUALS the element at position i MINUS one OF heights  
                DECREMENT i BY one  
            END WHILE  
            
            IF i IS GREATER THAN k  
                INCREMENT the element at position i OF heights BY one  
                CONTINUE TO the next unnamed_element OF the FOR loop  
            END IF  
            
            INCREMENT the element at position k OF heights BY one  
        END FOR  
        
        RETURN heights  
    END FUNCTION  
END CLASS
