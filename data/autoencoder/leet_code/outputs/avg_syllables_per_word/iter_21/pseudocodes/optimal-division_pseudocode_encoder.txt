CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the STRING REPRESENTATION OF element at position zero OF nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the STRING REPRESENTATION OF element at position zero OF nums FOLLOWED BY THE WORD division FOLLOWED BY the STRING REPRESENTATION OF element at position one OF nums  
        END IF  
        RETURN the STRING REPRESENTATION OF element at position zero OF nums FOLLOWED BY THE WORD division FOLLOWED BY an OPENING PARENTHESIS FOLLOWED BY the STRING REPRESENTATION OF element at position one OF nums  
            FOLLOWED BY THE WORD division FOLLOWED BY the STRING REPRESENTATION OF element at position two OF nums  
            FOLLOWED BY THE WORD division FOLLOWED BY the STRING REPRESENTATION OF each subsequent element FROM position three TO the END OF nums JOINED BY THE WORD division  
            FOLLOWED BY a CLOSING PARENTHESIS  
    END FUNCTION  
END CLASS
