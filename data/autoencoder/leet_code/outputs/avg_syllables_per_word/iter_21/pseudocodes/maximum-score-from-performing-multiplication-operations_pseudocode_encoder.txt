CLASS Solution  
    FUNCTION maximumScore(nums, multipliers)  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        
        SET dp TO a list OF zeros WITH LENGTH m PLUS one  
        
        FOR i FROM m MINUS one TO zero INCLUSIVE IN DECREMENT BY one  
            SET new_dp TO a list OF zeros WITH LENGTH m PLUS one  
            FOR left FROM zero TO i INCLUSIVE  
                SET right TO n MINUS one MINUS (i MINUS left)  
                SET mul TO the element at position i OF multipliers  
                IF mul MULTIPLIED BY the element at position left OF nums PLUS the element at position left PLUS one OF dp IS GREATER THAN mul MULTIPLIED BY the element at position right OF nums PLUS the element at position left OF dp  
                    SET the element at position left OF new_dp TO mul MULTIPLIED BY the element at position left OF nums PLUS the element at position left PLUS one OF dp  
                ELSE  
                    SET the element at position left OF new_dp TO mul MULTIPLIED BY the element at position right OF nums PLUS the element at position left OF dp  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the element at position zero OF dp  
    END FUNCTION  
END CLASS
