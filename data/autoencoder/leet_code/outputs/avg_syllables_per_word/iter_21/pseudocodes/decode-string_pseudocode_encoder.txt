CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO an empty list  
        SET a TO an empty string  
        SET d TO an empty string  
        
        FOR each char IN s  
            IF char is a digit character  
                SET d TO d concatenated with char  
            ELSE IF char is an alphabetic character  
                SET a TO a concatenated with char  
            ELSE IF char EQUALS the left bracket character  
                CALL the append operation ON stack WITH PARAMETERS the tuple of a and the conversion of d TO an integer  
                SET a TO an empty string  
                SET d TO an empty string  
            ELSE IF char EQUALS the right bracket character  
                SET the tuple p and n TO the result of popping the last element FROM stack  
                SET a TO p concatenated with a repeated n TIMES  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
