CLASS Solution  
    FUNCTION minmaxGasDist(stations, k)  
        FUNCTION possible(max_dist)  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                SET distance_difference TO element at position index OF stations MINUS element at position index MINUS one OF stations  
                SET count TO distance_difference DIVIDED BY max_dist CONVERTED TO INTEGER  
                INCREMENT additional_stations BY count  
            END FOR  
            RETURN additional_stations LESS THAN OR EQUAL TO k  
        END FUNCTION  

        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE right MINUS left GREATER THAN zero POINT zero zero zero zero zero one  
            SET mid TO left PLUS right DIVIDED BY two  
            IF possible CALLED WITH PARAMETERS mid  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
