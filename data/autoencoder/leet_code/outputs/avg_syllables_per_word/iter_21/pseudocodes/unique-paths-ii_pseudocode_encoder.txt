CLASS Solution  
    FUNCTION uniquePathsWithObstacles(obstacleGrid)  
        IF obstacleGrid EQUALS empty OR element at position zero zero OF obstacleGrid EQUALS one  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF obstacleGrid  
        SET n TO the LENGTH OF element at position zero OF obstacleGrid  
        SET dp TO a two-dimensional list with m rows and n columns all SET TO zero  
        
        SET element at position zero zero OF dp TO one  
        
        FOR j FROM one TO n MINUS one  
            IF element at position zero j OF obstacleGrid EQUALS zero  
                SET element at position zero j OF dp TO element at position zero j MINUS one OF dp  
            END IF  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            IF element at position i zero OF obstacleGrid EQUALS zero  
                SET element at position i zero OF dp TO element at position i MINUS one zero OF dp  
            END IF  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF element at position i j OF obstacleGrid EQUALS zero  
                    SET element at position i j OF dp TO element at position i MINUS one j OF dp PLUS element at position i j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position m MINUS one n MINUS one OF dp  
    END FUNCTION  
END CLASS
