CLASS Solution  
    FUNCTION findSubsequences(nums)  
        FUNCTION dfs(u, last, t)  
            IF u EQUALS the LENGTH OF nums THEN  
                IF the LENGTH OF t GREATER THAN one THEN  
                    APPEND a COPY OF t TO ans  
                END IF  
                RETURN  
            END IF  
            IF element at position u OF nums GREATER THAN OR EQUAL TO last THEN  
                APPEND element at position u OF nums TO t  
                CALL dfs WITH PARAMETERS u PLUS one, element at position u OF nums, t  
                REMOVE the last element FROM t  
            END IF  
            IF element at position u OF nums NOT EQUALS last THEN  
                CALL dfs WITH PARAMETERS u PLUS one, last, t  
            END IF  
        END FUNCTION  
        SET ans TO an empty list  
        CALL dfs WITH PARAMETERS zero, negative one thousand, an empty list  
        RETURN ans  
    END FUNCTION  
END CLASS
