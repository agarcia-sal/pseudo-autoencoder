CLASS Solution
    FUNCTION canVisitAllRooms(rooms)
        SET visited TO empty collection of unique elements
        SET stack TO collection containing zero as the only element

        WHILE the collection stack IS NOT empty
            SET current_room TO the element removed from the end of collection stack
            ADD current_room TO collection visited

            FOR each key IN element at position current_room OF collection rooms
                IF key IS NOT IN collection visited
                    ADD key TO collection stack
                END IF
            END FOR
        END WHILE

        SET total_visited TO the number of elements in collection visited
        SET total_rooms TO the number of elements in collection rooms

        IF total_visited EQUALS total_rooms
            RETURN True
        ELSE
            RETURN False
        END IF
    END FUNCTION
END CLASS
