CLASS Solution  
    FUNCTION visiblePoints(points, angle, location)  
        SET angles TO an empty list  
        SET extra_points TO zero  
        
        FOR each element comprising x and y IN points  
            IF x EQUALS the element at position zero OF location AND y EQUALS the element at position one OF location  
                INCREMENT extra_points BY one  
                CONTINUE to the next iteration  
            END IF  
            SET angle_rad TO the arctangent of (y MINUS the element at position one OF location) DIVIDED BY (x MINUS the element at position zero OF location)  
            SET angle_deg TO the conversion of angle_rad FROM radians TO degrees  
            APPEND angle_deg TO angles  
        END FOR  
        
        SORT angles IN ascending order  
        
        SET angles TO angles concatenated WITH each element in angles INCREMENTED BY three hundred sixty  
        
        SET max_visible TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO the LENGTH OF angles MINUS one  
            WHILE the element at position right OF angles MINUS the element at position left OF angles GREATER THAN angle  
                INCREMENT left BY one  
            END WHILE  
            SET max_visible TO the maximum value BETWEEN max_visible AND (right MINUS left PLUS one)  
        END FOR  
        
        RETURN max_visible PLUS extra_points  
    END FUNCTION  
END CLASS
