CLASS Solution  
    FUNCTION removeKdigits(num, k)  
        SET stack TO empty list  
        
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND element at last position OF stack GREATER THAN digit  
                CALL stack to remove the last element  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        
        IF k EQUALS zero  
            SET final_stack TO stack  
        ELSE  
            SET final_stack TO sublist of stack from first element TO element at position LENGTH OF stack MINUS k EXCLUSIVE  
        END IF  
        
        SET result TO removal of leading zeros FROM concatenation of all elements IN final_stack  
        
        IF result IS NOT empty  
            RETURN result  
        ELSE  
            RETURN zero as string  
        END IF  
    END FUNCTION  
END CLASS
