CLASS Solution  
    FUNCTION maxSum(nums1, nums2)  
        SET i TO zero  
        SET j TO zero  
        SET sum1 TO zero  
        SET sum2 TO zero  
        SET MOD TO one billion PLUS seven  
        
        WHILE i IS LESS THAN the LENGTH OF nums1 AND j IS LESS THAN the LENGTH OF nums2  
            IF element at position i OF nums1 IS LESS THAN element at position j OF nums2  
                INCREMENT sum1 BY element at position i OF nums1  
                INCREMENT i BY one  
            ELSE IF element at position i OF nums1 IS GREATER THAN element at position j OF nums2  
                INCREMENT sum2 BY element at position j OF nums2  
                INCREMENT j BY one  
            ELSE  
                SET sum1 TO the GREATER OF sum1 AND sum2 PLUS element at position i OF nums1  
                SET sum2 TO sum1  
                INCREMENT i BY one  
                INCREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i IS LESS THAN the LENGTH OF nums1  
            INCREMENT sum1 BY element at position i OF nums1  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE j IS LESS THAN the LENGTH OF nums2  
            INCREMENT sum2 BY element at position j OF nums2  
            INCREMENT j BY one  
        END WHILE  
        
        SET result TO the GREATER OF sum1 AND sum2  
        SET result TO result MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
