CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  
        
        SET below_twenty TO list containing empty string and words for numbers One through Nineteen in order  
        SET tens TO list containing empty strings and words for multiples of Ten from Twenty through Ninety in order  
        SET thousands TO list containing empty string and words Thousand, Million, Billion in order  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            END IF  
            IF n LESS THAN twenty  
                RETURN element at position n OF below_twenty concatenated WITH space  
            END IF  
            IF n LESS THAN one hundred  
                RETURN element at position n DIVIDED BY ten OF tens concatenated WITH space concatenated WITH CALL helper WITH parameter n MODULO ten  
            END IF  
            RETURN element at position n DIVIDED BY one hundred OF below_twenty concatenated WITH word Hundred concatenated WITH space concatenated WITH CALL helper WITH parameter n MODULO one hundred  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero HOLDS  
            IF num MODULO one thousand NOT EQUALS zero  
                SET result TO CALL helper WITH parameter num MODULO one thousand concatenated WITH element at position i OF thousands concatenated WITH space concatenated WITH result  
            END IF  
            SET num TO num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
