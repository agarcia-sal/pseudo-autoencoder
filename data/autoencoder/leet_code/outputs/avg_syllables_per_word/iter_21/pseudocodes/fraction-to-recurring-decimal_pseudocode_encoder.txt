CLASS Solution  
    FUNCTION fractionToDecimal(numerator, denominator)  
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero  
            SET sign TO the string SYMBOL DASH  
        ELSE  
            SET sign TO the empty string  
        END IF  

        SET numerator TO the absolute value OF numerator  
        SET denominator TO the absolute value OF denominator  

        CALL the FUNCTION divmod WITH PARAMETERS numerator AND denominator AND ASSIGN the first RETURN VALUE TO quotient AND the second RETURN VALUE TO remainder  
        SET integer_part TO the string representation OF quotient  

        IF remainder EQUALS zero  
            RETURN the concatenation OF sign AND integer_part  
        END IF  

        SET seen_remainders TO an empty dictionary  
        SET fractional_part TO an empty list  

        WHILE remainder NOT EQUALS zero HOLDS  
            IF remainder IS A KEY IN seen_remainders  
                SET start TO the value ASSOCIATED WITH the key remainder IN seen_remainders  
                SET non_repeating TO the concatenation OF elements FROM position zero TO position start MINUS one OF fractional_part  
                SET repeating TO the concatenation OF elements FROM position start TO the end OF fractional_part  
                RETURN the concatenation OF sign AND integer_part AND the string SYMBOL DOT AND non_repeating AND the string SYMBOL OPEN PARENTHESIS AND repeating AND the string SYMBOL CLOSE PARENTHESIS  
            END IF  

            SET the key remainder IN seen_remainders TO the LENGTH OF fractional_part  

            SET remainder TO remainder MULTIPLIED BY ten  
            CALL the FUNCTION divmod WITH PARAMETERS remainder AND denominator AND ASSIGN the first RETURN VALUE TO quotient AND the second RETURN VALUE TO remainder  
            APPEND the string representation OF quotient TO fractional_part  
        END WHILE  

        RETURN the concatenation OF sign AND integer_part AND the string SYMBOL DOT AND the concatenation OF all elements OF fractional_part  
    END FUNCTION  
END CLASS
