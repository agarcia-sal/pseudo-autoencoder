CLASS Solution  
    FUNCTION reconstructQueue(people)  
        CALL people.sort WITH PARAMETERS using a key that orders each element first BY negation OF element at position zero THEN BY element at position one IN ascending order  
        SET queue TO an empty list  
        FOR each person IN people  
            INSERT person INTO queue AT the position indicated BY element at position one OF person  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
