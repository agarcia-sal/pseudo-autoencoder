CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS empty OR the LENGTH OF stones IS LESS THAN two  
            RETURN False  
        END IF  
        
        SET stone_set TO a set CONTAINING all elements OF stones  
        SET memo TO an empty dictionary  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the last element OF stones  
                RETURN True  
            END IF  
            
            IF the tuple OF position AND jump IS a key IN memo  
                RETURN the value ASSOCIATED WITH the tuple OF position AND jump IN memo  
            END IF  
            
            FOR each next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump IS GREATER THAN zero AND position PLUS next_jump IS AN element IN stone_set  
                    IF CALL can_jump_to WITH PARAMETERS position PLUS next_jump AND next_jump RETURNS True  
                        SET memo AT the key OF tuple OF position AND jump TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET memo AT the key OF tuple OF position AND jump TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN CALL can_jump_to WITH PARAMETERS the element at position zero OF stones AND zero  
    END FUNCTION  
END CLASS
