CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO the LENGTH OF nums1  
        CALL a FUNCTION to INITIALIZE dp AS a LIST OF size TWO RAISED TO THE POWER n FILLED WITH positive infinity  
        SET element at position zero OF dp TO zero  
        
        FOR each mask FROM zero TO TWO RAISED TO THE POWER n MINUS one  
            SET bit_count TO the NUMBER OF ONES IN THE BINARY REPRESENTATION OF mask  
            
            FOR each j FROM zero TO n MINUS one  
                IF the BITWISE AND OF mask AND TWO RAISED TO THE POWER j IS NOT EQUAL TO zero  
                    SET dp at position mask TO THE MINIMUM VALUE BETWEEN dp at position mask AND dp at the POSITION OF BITWISE XOR OF mask AND TWO RAISED TO THE POWER j PLUS THE BITWISE XOR OF element at position bit_count MINUS one OF nums1 AND element at position j OF nums2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at POSITION TWO RAISED TO THE POWER n MINUS one  
    END FUNCTION  
END CLASS
