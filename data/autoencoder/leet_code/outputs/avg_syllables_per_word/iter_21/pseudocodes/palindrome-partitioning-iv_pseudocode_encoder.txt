CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS substring from position LENGTH OF sub MINUS one TO zero OF sub  
        END FUNCTION  
        
        SET n TO LENGTH OF s  
        
        FUNCTION initialize_palindrome_matrix(size)  
            SET matrix TO empty list  
            FOR index FROM zero TO size MINUS one  
                SET row TO list OF size elements all SET TO False  
                APPEND row TO matrix  
            END FOR  
            RETURN matrix  
        END FUNCTION  
        
        SET is_pal TO CALL initialize_palindrome_matrix WITH PARAMETERS n  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF character at position i OF s EQUALS character at position j OF s AND (j MINUS i LESS THAN OR EQUAL TO one OR element at position i PLUS one AND j MINUS one OF is_pal HOLDS True)  
                    SET element at position i AND j OF is_pal TO True  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM one TO n MINUS two  
            IF element at position zero AND i MINUS one OF is_pal HOLDS True  
                FOR j FROM i TO n MINUS two  
                    IF element at position i AND j OF is_pal HOLDS True AND element at position j PLUS one AND n MINUS one OF is_pal HOLDS True  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
