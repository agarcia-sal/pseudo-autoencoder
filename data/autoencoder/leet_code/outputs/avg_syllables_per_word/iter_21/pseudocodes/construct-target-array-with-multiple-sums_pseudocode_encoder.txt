CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO an empty list  
        FOR each num IN target  
            APPEND negative num TO max_heap  
        END FOR  
        CALL heapify ON max_heap  
        
        SET total_sum TO the total sum OF all elements IN target  
        
        WHILE element at position zero OF max_heap IS LESS THAN negative one HOLDS  
            SET largest TO negative CALL heappop ON max_heap  
            SET rest_sum TO total_sum MINUS largest  
            
            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero THEN  
                RETURN False  
            END IF  
            
            SET new_value TO largest MODULO rest_sum  
            
            IF new_value EQUALS zero AND rest_sum IS NOT EQUAL TO one THEN  
                RETURN False  
            END IF  
            
            SET total_sum TO rest_sum PLUS new_value  
            CALL heappush ON max_heap WITH PARAMETERS negative new_value  
        END WHILE  
        
        RETURN True  
    END FUNCTION  
END CLASS
