CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty  
            RETURN zero  
        END IF  

        CALL intervals SORT WITH KEY element at position zero OF element  

        SET min_heap TO empty list  

        FOR each meeting IN intervals  
            IF min_heap IS NOT empty AND element at position zero OF meeting GREATER THAN OR EQUAL TO element at position zero OF min_heap  
                CALL remove smallest element FROM min_heap  
            END IF  

            CALL add element meeting at position one TO min_heap  
        END FOR  

        SET number_of_rooms TO the LENGTH OF min_heap  
        RETURN number_of_rooms  
    END FUNCTION  
END CLASS
