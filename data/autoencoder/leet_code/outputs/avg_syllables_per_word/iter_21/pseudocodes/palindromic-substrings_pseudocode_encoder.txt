CLASS Solution  
    FUNCTION countSubstrings(s)  
        FUNCTION expand_around_center(left, right)  
            SET count TO zero  
            WHILE left GREATER THAN OR EQUAL TO zero AND right LESS THAN the LENGTH OF s AND element at position left OF s EQUALS element at position right OF s HOLDS  
                INCREMENT count BY one  
                DECREMENT left BY one  
                INCREMENT right BY one  
            END WHILE  
            RETURN count  
        END FUNCTION  
        
        SET total_palindromes TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET total_palindromes TO total_palindromes PLUS CALL expand_around_center WITH PARAMETERS index AND index  
            SET total_palindromes TO total_palindromes PLUS CALL expand_around_center WITH PARAMETERS index AND index PLUS one  
        END FOR  
        RETURN total_palindromes  
    END FUNCTION  
END CLASS
