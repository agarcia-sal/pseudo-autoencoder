CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        CALL sort_points_based_on_end_coordinate WITH PARAMETERS points  
        SET arrows TO zero  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF element at position zero OF balloon IS GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO element at position one OF balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  

    FUNCTION sort_points_based_on_end_coordinate(points)  
        SORT the list points BY the element at position one OF each sublist IN ascending order  
    END FUNCTION  
END CLASS
