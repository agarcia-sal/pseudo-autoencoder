CLASS Solution  
    FUNCTION minRefuelStops(target, startFuel, stations)  
        SET max_heap TO an empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        CALL APPEND a list CONTAINING position target AND fuel zero TO the list stations  

        FOR each pair position AND fuel IN stations  
            WHILE max_heap IS NOT empty AND current_fuel IS LESS THAN position  
                SET largest_fuel TO the result of CALL popping the smallest element FROM max_heap USING a min-heap removal  
                DECREMENT current_fuel BY largest_fuel MULTIPLIED BY minus one  
                INCREMENT stops BY one  
            END WHILE  

            IF current_fuel IS LESS THAN position  
                RETURN negative one  
            END IF  

            CALL pushing a value fuel MULTIPLIED BY minus one INTO the min-heap max_heap  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
