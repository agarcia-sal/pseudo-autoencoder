CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO CALL get_last_occurrence_dictionary WITH PARAMETERS s  
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO an empty list  
        FOR each position i AND character char IN s with indices  
            SET end TO the GREATER OF end AND the element at position char OF last_occurrence  
            IF i EQUALS end THEN  
                APPEND end MINUS start PLUS one TO partition_sizes  
                SET start TO i PLUS one  
            END IF  
        END FOR  
        RETURN partition_sizes  
    END FUNCTION  

    FUNCTION get_last_occurrence_dictionary(s)  
        SET result TO an empty dictionary  
        FOR each position idx AND character char IN s with indices  
            SET element at position char OF result TO idx  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
