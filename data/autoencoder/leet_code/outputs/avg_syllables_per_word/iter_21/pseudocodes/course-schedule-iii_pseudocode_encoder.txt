CLASS Solution  
    FUNCTION scheduleCourse(courses)  
        CALL courses SORT BY the second element at each position IN ascending order  
        SET max_heap TO an empty list  
        SET current_end_day TO zero  
        FOR each pair duration AND last_day IN courses  
            IF current_end_day PLUS duration LESS THAN OR EQUAL TO last_day  
                CALL function to add negative duration TO max_heap  
                INCREMENT current_end_day BY duration  
            ELSE IF max_heap IS NOT empty AND negative value of the first element at position zero OF max_heap GREATER THAN duration  
                INCREMENT current_end_day BY duration PLUS the result of removing the first element FROM max_heap  
                CALL function to add negative duration TO max_heap  
            END IF  
        END FOR  
        SET number_of_courses TO the LENGTH OF max_heap  
        RETURN number_of_courses  
    END FUNCTION  
END CLASS
