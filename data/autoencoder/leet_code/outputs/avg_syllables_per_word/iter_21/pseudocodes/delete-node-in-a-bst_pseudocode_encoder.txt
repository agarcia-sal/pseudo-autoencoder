CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO CALL TreeNode WITH PARAMETERS element at position zero OF values  
    SET i TO one  
    SET queue TO CALL deque WITHOUT PARAMETERS  
    CALL queue APPEND WITH PARAMETERS root  
    WHILE queue IS NOT empty  
        SET node TO CALL queue POP LEFT WITHOUT PARAMETERS  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.left  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO CALL TreeNode WITH PARAMETERS element at position i OF values  
            CALL queue APPEND WITH PARAMETERS node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH PARAMETERS p.left, q.left AND CALL is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode(root, key)  
        IF root IS None  
            RETURN None  
        END IF  
        IF root.val IS GREATER THAN key  
            SET root.left TO CALL self.deleteNode WITH PARAMETERS root.left, key  
            RETURN root  
        END IF  
        IF root.val IS LESS THAN key  
            SET root.right TO CALL self.deleteNode WITH PARAMETERS root.right, key  
            RETURN root  
        END IF  
        IF root.left IS None  
            RETURN root.right  
        END IF  
        IF root.right IS None  
            RETURN root.left  
        END IF  
        SET node TO root.right  
        WHILE node.left IS NOT None  
            SET node TO node.left  
        END WHILE  
        SET node.left TO root.left  
        SET root TO root.right  
        RETURN root  
    END FUNCTION  
END CLASS
