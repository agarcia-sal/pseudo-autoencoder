CLASS Solution  
    FUNCTION findPairs(nums, k)  
        IF k LESS THAN zero  
            RETURN zero  
        END IF  
  
        SET num_counts TO an empty dictionary  
        FOR each num IN nums  
            IF num IS A KEY IN num_counts  
                INCREMENT value associated WITH num IN num_counts BY one  
            ELSE  
                SET value associated WITH num IN num_counts TO one  
            END IF  
        END FOR  
  
        SET unique_pairs TO zero  
        FOR each num IN keys OF num_counts  
            IF k EQUALS zero  
                IF value associated WITH num IN num_counts GREATER THAN one  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF num PLUS k IS A KEY IN num_counts  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
  
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
