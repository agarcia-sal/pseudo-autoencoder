CLASS Solution  
    FUNCTION twoSum(numbers, target)  
        SET left TO zero  
        SET right TO the LENGTH OF numbers MINUS one  
        WHILE left LESS THAN right HOLDS  
            SET current_sum TO element at position left OF numbers PLUS element at position right OF numbers  
            IF current_sum EQUALS target THEN  
                RETURN list containing left PLUS one AND right PLUS one  
            ELSE IF current_sum LESS THAN target THEN  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
