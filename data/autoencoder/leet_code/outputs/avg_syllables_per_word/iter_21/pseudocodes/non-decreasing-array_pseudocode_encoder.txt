CLASS Solution  
    FUNCTION checkPossibility(nums)  
        SET modified TO false  
        FOR index FROM one TO THE LENGTH OF nums MINUS one  
            IF the element at position index OF nums IS LESS THAN the element at position index MINUS one OF nums  
                IF modified IS true  
                    RETURN false  
                END IF  
                SET modified TO true  
                IF index IS LESS THAN two OR the element at position index OF nums IS GREATER THAN OR EQUAL TO the element at position index MINUS two OF nums  
                    SET the element at position index MINUS one OF nums TO the element at position index OF nums  
                ELSE  
                    SET the element at position index OF nums TO the element at position index MINUS one OF nums  
                END IF  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
