CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO the SUM OF all elements IN nums  
        
        IF total_sum MODULO two NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        
        CALL initialize_dp WITH PARAMETERS target TO OBTAIN dp  
        
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET dp at position i TO dp at position i OR dp at position i MINUS num  
            END FOR  
        END FOR  
        
        RETURN dp at position target  
    END FUNCTION
    
    FUNCTION initialize_dp(target)  
        CREATE list dp WITH target PLUS one elements ALL SET TO False  
        SET dp at position zero TO True  
        RETURN dp  
    END FUNCTION  
END CLASS
