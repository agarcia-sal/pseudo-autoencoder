CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO the result of calling list ON the range FROM zero TO size MINUS one INCLUSIVE  
        SET self.rank TO a list CONSISTING OF the number one repeated size TIMES  
        SET self.size TO a list CONSISTING OF the number one repeated size TIMES  
    END FUNCTION  
      
    FUNCTION find(self, u)  
        IF the element at position u OF self.parent NOT EQUALS u THEN  
            SET the element at position u OF self.parent TO the result of CALLING self.find WITH PARAMETERS the element at position u OF self.parent  
        END IF  
        RETURN the element at position u OF self.parent  
    END FUNCTION  
      
    FUNCTION union(self, u, v)  
        SET rootU TO the result of CALLING self.find WITH PARAMETERS u  
        SET rootV TO the result of CALLING self.find WITH PARAMETERS v  
        IF rootU NOT EQUALS rootV THEN  
            IF the element at position rootU OF self.rank GREATER THAN the element at position rootV OF self.rank THEN  
                SET the element at position rootV OF self.parent TO rootU  
                INCREMENT the element at position rootU OF self.size BY the element at position rootV OF self.size  
            ELSE IF the element at position rootU OF self.rank LESS THAN the element at position rootV OF self.rank THEN  
                SET the element at position rootU OF self.parent TO rootV  
                INCREMENT the element at position rootV OF self.size BY the element at position rootU OF self.size  
            ELSE  
                SET the element at position rootV OF self.parent TO rootU  
                INCREMENT the element at position rootU OF self.rank BY one  
                INCREMENT the element at position rootU OF self.size BY the element at position rootV OF self.size  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION largestComponentSize(self, nums)  
        SET max_num TO the maximum element OF nums  
        SET uf TO a new instance of UnionFind WITH PARAMETERS max_num PLUS one  
        SET prime_to_index TO an empty mapping  
        FOR each num IN nums  
            FOR each prime IN the result of CALLING self.prime_factors WITH PARAMETERS num  
                IF prime IS IN prime_to_index THEN  
                    CALL uf.union WITH PARAMETERS the value mapped by prime in prime_to_index AND num  
                ELSE  
                    SET the value mapped by prime in prime_to_index TO num  
                END IF  
            END FOR  
        END FOR  
        SET max_component_size TO zero  
        FOR each num IN nums  
            SET root TO the result of CALLING uf.find WITH PARAMETERS num  
            IF the element at position root OF uf.size GREATER THAN max_component_size THEN  
                SET max_component_size TO the element at position root OF uf.size  
            END IF  
        END FOR  
        RETURN max_component_size  
    END FUNCTION  
      
    FUNCTION prime_factors(self, n)  
        SET factors TO an empty set  
        WHILE n MODULO two EQUALS zero  
            ADD two TO factors  
            SET n TO n DIVIDED BY two  
        END WHILE  
        FOR i FROM three TO the integer part of the square root of n PLUS one IN STEPS OF two  
            WHILE n MODULO i EQUALS zero  
                ADD i TO factors  
                SET n TO n DIVIDED BY i  
            END WHILE  
        END FOR  
        IF n GREATER THAN two THEN  
            ADD n TO factors  
        END IF  
        RETURN factors  
    END FUNCTION  
END CLASS
