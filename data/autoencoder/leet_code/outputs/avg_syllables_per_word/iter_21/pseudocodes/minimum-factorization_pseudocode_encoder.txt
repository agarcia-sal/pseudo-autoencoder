CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num EQUALS one  
            RETURN one  
        END IF  
        
        SET factors TO an empty list  
        FOR digit FROM nine DOWN TO two  
            WHILE num MODULO digit EQUALS zero HOLDS  
                APPEND digit TO factors  
                SET num TO num DIVIDED BY digit  
            END WHILE  
        END FOR  
        
        IF num NOT EQUALS one  
            RETURN zero  
        END IF  
        
        SORT factors IN ascending order  
        SET string_result TO the concatenation OF string representations OF each element IN factors  
        SET result TO the integer representation OF string_result  
        
        IF result GREATER THAN two POWER thirty ONE MINUS one  
            RETURN zero  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
