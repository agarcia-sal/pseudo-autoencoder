CLASS Solution  
    FUNCTION maxProfit(prices, fee)  
        SET cash TO zero  
        SET hold TO negative infinity  
        FOR each price IN prices  
            SET cash TO the GREATER OF cash AND the SUM OF hold PLUS price MINUS fee  
            SET hold TO the GREATER OF hold AND the DIFFERENCE OF cash MINUS price  
        END FOR  
        RETURN cash  
    END FUNCTION  
END CLASS
