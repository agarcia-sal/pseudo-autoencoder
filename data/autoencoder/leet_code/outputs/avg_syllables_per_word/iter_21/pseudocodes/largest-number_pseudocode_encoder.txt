CLASS Solution  
    FUNCTION largestNumber(nums)  
        CALL map_function TO CONVERT each element IN nums TO string AND ASSIGN RESULT TO nums_str  
        
        FUNCTION compare(x, y)  
            SET concatenated_xy TO concatenation OF x FOLLOWED BY y  
            SET concatenated_yx TO concatenation OF y FOLLOWED BY x  
            IF concatenated_xy GREATER THAN concatenated_yx  
                RETURN negative one  
            ELSE IF concatenated_xy LESS THAN concatenated_yx  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        CALL sort METHOD ON nums_str WITH KEY DEFINED BY COMPARISON FUNCTION compare  
        
        IF element at position zero OF nums_str EQUALS string zero  
            RETURN string zero  
        END IF  
        
        SET result TO concatenation OF all elements IN nums_str IN ORDER  
        RETURN result  
    END FUNCTION  
END CLASS
