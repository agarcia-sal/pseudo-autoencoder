CLASS Solution  
    FUNCTION rearrangeSticks(n, k)  
        SET MOD TO one billion PLUS seven  
        CALL initialize_dp_table WITH PARAMETERS n, k AND ASSIGN TO dp  
        SET element at position zero OF dp AT position zero TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET visible_case TO element at position j MINUS one OF dp AT position i MINUS one  
                SET not_visible_case TO i MINUS one MULTIPLIED BY element at position j OF dp AT position i MINUS one  
                SET element at position j OF dp AT position i TO visible_case PLUS not_visible_case MODULO MOD  
            END FOR  
        END FOR  
        RETURN element at position k OF dp AT position n  
    END FUNCTION  

    FUNCTION initialize_dp_table(n, k)  
        CREATE empty list dp  
        FOR index_i FROM zero TO n  
            CREATE list row OF size k PLUS one FILLED WITH zero  
            APPEND row TO dp  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
