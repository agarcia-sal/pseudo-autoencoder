CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO CALL zip WITH PARAMETERS difficulty AND profit CONVERTED TO list  
        CALL sort ON jobs  
        CALL sort ON worker  
        
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the LENGTH OF jobs  
        
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND element at position job_index OF jobs AT position zero LESS THAN OR EQUAL TO ability  
                IF element at position job_index OF jobs AT position one GREATER THAN best_profit  
                    SET best_profit TO element at position job_index OF jobs AT position one  
                END IF  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
