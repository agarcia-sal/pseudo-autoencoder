CLASS Solution  
    FUNCTION minDays(n)  
        FUNCTION dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            SET option_one TO the remainder when remaining IS DIVIDED BY two PLUS CALL dp WITH PARAMETERS the integer division of remaining BY two  
            SET option_two TO the remainder when remaining IS DIVIDED BY three PLUS CALL dp WITH PARAMETERS the integer division of remaining BY three  
            SET result TO one PLUS the lesser value BETWEEN option_one AND option_two  
            RETURN result  
        END FUNCTION  
        RETURN CALL dp WITH PARAMETERS n  
    END FUNCTION  
END CLASS
