CLASS Solution  
    FUNCTION exist(board, word)  
        IF board EQUALS an empty value  
            RETURN False  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF the element at position zero OF board  
        SET visited TO a new list containing m elements EACH being a list containing n elements EACH set TO False  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS the LENGTH OF word  
                RETURN True  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR the element at position x OF visited at position y IS True OR the element at position x OF board at position y NOT EQUALS the element at position index OF word  
                RETURN False  
            END IF  

            SET the element at position x OF visited at position y TO True  
            FOR each pair dx, dy IN the list containing the pairs minus one and zero, one and zero, zero and minus one, zero and one  
                IF CALL dfs WITH PARAMETERS x PLUS dx, y PLUS dy, index PLUS one RETURNS True  
                    RETURN True  
                END IF  
            END FOR  
            SET the element at position x OF visited at position y TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF CALL dfs WITH PARAMETERS i, j, zero RETURNS True  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
