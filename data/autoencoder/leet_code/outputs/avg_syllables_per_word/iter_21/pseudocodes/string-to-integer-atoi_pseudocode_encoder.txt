CLASS Solution  
    FUNCTION myAtoi(s)  
        SET INT_MAX TO two RAISED TO THE POWER OF thirty ONE MINUS one  
        SET INT_MIN TO NEGATIVE two RAISED TO THE POWER OF thirty ONE  
        SET result TO zero  
        SET index TO zero  
        SET n TO the LENGTH OF s  
        SET sign TO one  
        
        WHILE index LESS THAN n AND the element at position index OF s EQUALS a space  
            INCREMENT index BY one  
        END WHILE  
        
        IF index LESS THAN n AND (the element at position index OF s EQUALS the plus SIGN OR the element at position index OF s EQUALS the minus SIGN)  
            IF the element at position index OF s EQUALS the minus SIGN  
                SET sign TO NEGATIVE one  
            ELSE  
                SET sign TO one  
            END IF  
            INCREMENT index BY one  
        END IF  
        
        WHILE index LESS THAN n AND the element at position index OF s IS A DIGIT  
            SET digit TO the INTEGER VALUE OF the element at position index OF s  
            IF result GREATER THAN (INT_MAX MINUS digit) DIVIDED BY ten  
                IF sign EQUALS one  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
                END IF  
            END IF  
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        END WHILE  
        
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
