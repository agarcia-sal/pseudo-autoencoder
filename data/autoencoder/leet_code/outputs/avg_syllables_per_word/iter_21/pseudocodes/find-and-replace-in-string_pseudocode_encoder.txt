CLASS Solution  
    FUNCTION findReplaceString(s, indices, sources, targets)  
        CALL zip FUNCTION WITH PARAMETERS indices, sources, targets AND ASSIGN RESULT TO list_of_tuples  
        CALL sorted FUNCTION WITH PARAMETERS list_of_tuples AND ASSIGN RESULT TO replacements  
        SET result TO empty list  
        SET prev_end TO zero  
        FOR each element IN replacements  
            DESTRUCTURE element INTO index, source, target  
            APPEND substring from position prev_end TO index OF s TO result  
            IF substring from position index TO index PLUS the LENGTH OF source OF s EQUALS source  
                APPEND target TO result  
            ELSE  
                APPEND substring from position index TO index PLUS the LENGTH OF source OF s TO result  
            END IF  
            SET prev_end TO index PLUS the LENGTH OF source  
        END FOR  
        APPEND substring from position prev_end TO the LENGTH OF s OF s TO result  
        CALL the JOIN OPERATION ON result AND RETURN THE RESULT  
    END FUNCTION  
END CLASS
