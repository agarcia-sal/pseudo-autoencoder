CLASS Solution  
    FUNCTION characterReplacement(s, k)  
        CALL defaultdict WITH PARAMETERS int AND ASSIGN reference TO count  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the element at position right OF s IN count BY one  
            IF max_count LESS THAN the element at position right OF s IN count  
                SET max_count TO the element at position right OF s IN count  
            END IF  
            
            IF the EXPRESSION right MINUS left PLUS one MINUS max_count GREATER THAN k  
                DECREMENT the element at position left OF s IN count BY one  
                INCREMENT left BY one  
            END IF  
            
            IF max_length LESS THAN right MINUS left PLUS one  
                SET max_length TO right MINUS left PLUS one  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
