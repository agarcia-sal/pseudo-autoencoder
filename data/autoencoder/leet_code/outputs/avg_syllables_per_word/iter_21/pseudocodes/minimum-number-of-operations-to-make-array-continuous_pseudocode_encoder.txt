CLASS Solution  
    FUNCTION minOperations(nums)  
        SET n TO the LENGTH OF nums  
        SET unique_nums TO the result of CONVERTING nums TO a set AND THEN SORTING the set INTO a list  
        
        SET min_operations TO n  
        SET j TO zero  
        FOR each i FROM zero TO the LENGTH OF unique_nums MINUS one  
            WHILE j LESS THAN the LENGTH OF unique_nums AND the element at position j OF unique_nums LESS THAN the element at position i OF unique_nums PLUS n  
                INCREMENT j BY one  
            END WHILE  
            SET min_operations TO the MINIMUM OF min_operations AND n MINUS the DIFFERENCE BETWEEN j AND i  
        END FOR  
        
        RETURN min_operations  
    END FUNCTION  
END CLASS
