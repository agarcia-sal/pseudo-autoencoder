CLASS Solution  
    FUNCTION longestSubstring(s, k)  
        IF the LENGTH OF s LESS THAN k  
            RETURN zero  
        END IF  
        
        CALL Counter WITH PARAMETERS s AND ASSIGN reference TO count  
        
        FOR each char IN count  
            IF element at position char OF count LESS THAN k  
                RETURN the MAXIMUM OF  
                    CALL longestSubstring WITH PARAMETERS substring AND k  
                        FOR each substring IN the RESULT OF SPLITTING s BY char  
            END IF  
        END FOR  
        
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
