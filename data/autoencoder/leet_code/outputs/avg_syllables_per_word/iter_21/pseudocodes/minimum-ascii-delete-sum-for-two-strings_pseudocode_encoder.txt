CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
          
        CALL createTwoDimensionalArray WITH PARAMETERS m PLUS one AND n PLUS one  
        ASSIGN reference TO dp  
          
        FOR j FROM one TO n  
            SET element at position j OF row at position zero OF dp TO element at position j MINUS one OF row at position zero OF dp PLUS the ASCII code of character at position j MINUS one OF s2  
        END FOR  
          
        FOR i FROM one TO m  
            SET element at position zero OF row at position i OF dp TO element at position zero OF row at position i MINUS one OF dp PLUS the ASCII code of character at position i MINUS one OF s1  
        END FOR  
          
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one OF s1 EQUALS character at position j MINUS one OF s2  
                    SET element at position j OF row at position i OF dp TO element at position j MINUS one OF row at position i MINUS one OF dp  
                ELSE  
                    SET first_candidate TO element at position j OF row at position i MINUS one OF dp PLUS the ASCII code of character at position i MINUS one OF s1  
                    SET second_candidate TO element at position j MINUS one OF row at position i OF dp PLUS the ASCII code of character at position j MINUS one OF s2  
                    SET element at position j OF row at position i OF dp TO the MINIMUM OF first_candidate AND second_candidate  
                END IF  
            END FOR  
        END FOR  
          
        RETURN element at position n OF row at position m OF dp  
    END FUNCTION  
END CLASS
