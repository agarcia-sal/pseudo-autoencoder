CLASS Solution
    FUNCTION isInterleave(s1, s2, s3)
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 NOT EQUALS the LENGTH OF s3
            RETURN false
        END IF

        SET dp TO a list of boolean false values of size the LENGTH OF s2 PLUS one
        SET element at position zero OF dp TO true

        FOR index i FROM zero TO the LENGTH OF s1 INCLUSIVE
            FOR index j FROM zero TO the LENGTH OF s2 INCLUSIVE
                IF index i GREATER THAN zero
                    SET element at position j OF dp TO element at position j OF dp AND character at position i MINUS one OF s1 EQUALS character at position i PLUS j MINUS one OF s3
                END IF
                IF index j GREATER THAN zero
                    SET element at position j OF dp TO element at position j OF dp OR (element at position j MINUS one OF dp AND character at position j MINUS one OF s2 EQUALS character at position i PLUS j MINUS one OF s3)
                END IF
            END FOR
        END FOR

        RETURN element at position the LENGTH OF s2 OF dp
    END FUNCTION
END CLASS
