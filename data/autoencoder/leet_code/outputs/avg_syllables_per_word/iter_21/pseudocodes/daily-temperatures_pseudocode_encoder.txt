CLASS Solution  
    FUNCTION dailyTemperatures(temperatures)  
        SET n TO the LENGTH OF temperatures  
        SET answer TO a LIST OF zero ELEMENTS OF LENGTH n  
        SET stack TO an EMPTY LIST  
        
        FOR i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position i OF temperatures IS GREATER THAN the element at position element at the LAST POSITION OF stack OF temperatures  
                SET prev_index TO the LAST ELEMENT OF stack  
                REMOVE the LAST ELEMENT FROM stack  
                SET the element at position prev_index OF answer TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
