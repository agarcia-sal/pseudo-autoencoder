CLASS Solution  
    FUNCTION shortestSuperstring(words)  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap(i, j)  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i OF words AND the LENGTH OF element at position j OF words DOWN TO one BY DECREMENTING BY one  
                IF the substring from the POSITION of LENGTH MINUS k PLUS one TO THE END OF element at position i OF words EQUALS the substring from POSITION one TO k OF element at position j OF words  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp(mask, i)  
            IF mask EQUALS the VALUE OBTAINED BY SHIFTING one LEFT BY n MINUS one  
                RETURN zero AND an EMPTY STRING  
            END IF  
            
            SET min_len TO infinity  
            SET best_path TO an EMPTY STRING  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND OF mask AND the VALUE OBTAINED BY SHIFTING one LEFT BY j EQUALS zero  
                    CALL dp WITH PARAMETERS the BITWISE OR OF mask AND the VALUE OBTAINED BY SHIFTING one LEFT BY j AND j  
                    ASSIGN THE RETURNED VALUES TO length AND path  
                    SET length TO length PLUS the LENGTH OF element at position j OF words MINUS overlap(i, j)  
                    IF length LESS THAN min_len  
                        SET min_len TO length  
                        SET best_path TO the SUBSTRING from POSITION overlap(i, j) PLUS one TO THE END OF element at position j OF words CONCATENATED WITH path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO infinity  
        SET shortest_path TO an EMPTY STRING  
        FOR i FROM zero TO n MINUS one  
            CALL dp WITH PARAMETERS the VALUE OBTAINED BY SHIFTING one LEFT BY i AND i  
            ASSIGN THE RETURNED VALUES TO length AND path  
            SET length TO length PLUS the LENGTH OF element at position i OF words  
            IF length LESS THAN min_len  
                SET min_len TO length  
                SET shortest_path TO element at position i OF words CONCATENATED WITH path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
