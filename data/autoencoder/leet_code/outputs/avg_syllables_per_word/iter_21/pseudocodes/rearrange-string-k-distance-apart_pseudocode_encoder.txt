CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO CALL Counter WITH PARAMETERS s  
        
        SET max_heap TO EMPTY LIST  
        FOR each char AND count IN freq ITEMS  
            APPEND NEGATIVE count AND char AS A TUPLE TO max_heap  
        END FOR  
        CALL heapq.heapify WITH PARAMETERS max_heap  
        
        SET wait_queue TO EMPTY LIST  
        
        SET result TO EMPTY LIST  
        
        WHILE max_heap IS NOT EMPTY  
            CALL heapq.heappop WITH PARAMETERS max_heap AND ASSIGN TO count AND char  
            APPEND char TO result  
            
            APPEND count PLUS one AND char AS A TUPLE TO wait_queue  
            
            IF THE LENGTH OF wait_queue EQUALS k  
                REMOVE THE ELEMENT AT POSITION zero FROM wait_queue AND ASSIGN TO count AND char  
                IF count LESS THAN zero  
                    CALL heapq.heappush WITH PARAMETERS max_heap AND THE TUPLE count AND char  
                END IF  
            END IF  
        END WHILE  
        
        IF THE LENGTH OF result NOT EQUALS THE LENGTH OF s  
            RETURN AN EMPTY STRING  
        END IF  
        
        RETURN THE CONCATENATION OF ALL ELEMENTS IN result  
    END FUNCTION  
END CLASS
