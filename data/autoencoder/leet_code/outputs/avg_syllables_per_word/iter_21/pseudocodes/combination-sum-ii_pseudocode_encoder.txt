CLASS Solution  
    FUNCTION combinationSum2(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                IF i GREATER THAN start AND element at position i OF candidates EQUALS element at position i MINUS one OF candidates  
                    CONTINUE  
                END IF  
                CALL backtrack WITH PARAMETERS i PLUS one, target MINUS element at position i OF candidates, path PLUS list containing element at position i OF candidates  
            END FOR  
        END FUNCTION  
        CALL sort ON candidates  
        SET result TO empty list  
        CALL backtrack WITH PARAMETERS zero, target, empty list  
        RETURN result  
    END FUNCTION  
END CLASS
