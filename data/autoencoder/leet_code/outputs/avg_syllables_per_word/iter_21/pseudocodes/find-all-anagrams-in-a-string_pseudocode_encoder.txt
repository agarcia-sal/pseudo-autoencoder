CLASS Solution  
    FUNCTION findAnagrams(s, p)  
        SET len_s TO the LENGTH OF s  
        SET len_p TO the LENGTH OF p  
        
        IF len_p GREATER THAN len_s  
            RETURN an EMPTY list  
        END IF  
        
        CALL p_count TO Counter CREATED FROM p  
        CALL s_count TO Counter CREATED FROM the SUBSTRING FROM position zero TO len_p OF s  
        
        SET anagrams TO an EMPTY list  
        
        IF s_count EQUALS p_count  
            APPEND zero TO anagrams  
        END IF  
        
        FOR i FROM len_p TO len_s MINUS one  
            INCREMENT the COUNT OF the ELEMENT at position i OF s IN s_count BY one  
            DECREMENT the COUNT OF the ELEMENT at position i MINUS len_p OF s IN s_count BY one  
            
            IF the COUNT OF the ELEMENT at position i MINUS len_p OF s IN s_count EQUALS zero  
                REMOVE the ELEMENT at position i MINUS len_p OF s FROM s_count  
            END IF  
            
            IF s_count EQUALS p_count  
                APPEND i MINUS len_p PLUS one TO anagrams  
            END IF  
        END FOR  
        
        RETURN anagrams  
    END FUNCTION  
END CLASS
