CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO a set containing characters two, five, six, nine  
        SET same_digits TO a set containing characters zero, one, eight  
        
        FUNCTION is_good_number(num)  
            SET num_str TO the string representation of num  
            IF every digit IN num_str IS IN the union of change_digits AND same_digits  
                IF any digit IN num_str IS IN change_digits  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR each integer i FROM one TO n  
            IF is_good_number(i) HOLDS  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
