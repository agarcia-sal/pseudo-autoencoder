CLASS Solution  
    FUNCTION atMostNGivenDigitSet(digits, n)  
        FUNCTION count_numbers_with_length(length)  
            RETURN the LENGTH OF digits MULTIPLIED BY ITSELF length TIMES  
        END FUNCTION  
        
        FUNCTION count_numbers_up_to_n(n)  
            SET str_n TO the STRING REPRESENTATION OF n  
            SET length TO the LENGTH OF str_n  
            SET count TO zero  
            
            FOR l FROM one TO length MINUS one  
                INCREMENT count BY CALL count_numbers_with_length WITH PARAMETERS l  
            END FOR  
            
            FOR i FROM zero TO length MINUS one  
                SET current_digit TO the INTEGER VALUE OF the CHARACTER at position i OF str_n  
                FOR each digit IN digits  
                    SET digit_value TO the INTEGER VALUE OF digit  
                    IF digit_value LESS THAN current_digit  
                        INCREMENT count BY CALL count_numbers_with_length WITH PARAMETERS length MINUS i MINUS one  
                    ELSE IF digit_value EQUALS current_digit  
                        IF i EQUALS length MINUS one  
                            INCREMENT count BY one  
                        END IF  
                    ELSE  
                        EXIT the FOR loop over digits  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN count  
        END FUNCTION  
        
        RETURN CALL count_numbers_up_to_n WITH PARAMETERS n  
    END FUNCTION  
END CLASS
