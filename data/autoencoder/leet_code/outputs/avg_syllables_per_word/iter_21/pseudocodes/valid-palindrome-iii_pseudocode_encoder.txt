CLASS Solution  
    FUNCTION isValidPalindrome(s, k)  
        FUNCTION longest_palindromic_subseq(s)  
            SET n TO the LENGTH OF s  
            SET dp TO a two-dimensional list OF size n BY n FILLED WITH zero  
            
            FOR i FROM zero TO n MINUS one  
                SET element at position i AND i OF dp TO one  
            END FOR  
            
            FOR length FROM two TO n  
                FOR i FROM zero TO n MINUS length  
                    SET j TO i PLUS length MINUS one  
                    IF element at position i OF s EQUALS element at position j OF s THEN  
                        SET element at position i AND j OF dp TO element at position i PLUS one AND j MINUS one OF dp PLUS two  
                    ELSE  
                        SET element at position i AND j OF dp TO the MAXIMUM OF element at position i PLUS one AND j OF dp AND element at position i AND j MINUS one OF dp  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN element at position zero AND n MINUS one OF dp  
        END FUNCTION  
        
        SET longest_palindrome_length TO CALL longest_palindromic_subseq WITH PARAMETERS s  
        RETURN the LENGTH OF s MINUS longest_palindrome_length LESS THAN OR EQUAL TO k  
    END FUNCTION  
END CLASS
