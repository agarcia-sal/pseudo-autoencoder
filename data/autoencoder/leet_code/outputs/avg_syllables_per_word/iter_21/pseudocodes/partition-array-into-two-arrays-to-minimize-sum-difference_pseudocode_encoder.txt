CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY two  
        SET total_sum TO the SUM OF all elements IN nums  
        SET target TO total_sum DIVIDED BY two  
        SET min_diff TO infinity  
        
        SET left_half TO the SUBLIST of nums FROM position zero TO position n MINUS one  
        SET right_half TO the SUBLIST of nums FROM position n TO the END OF nums  
        
        SET left_sums TO a LIST OF empty DICTIONARIES with a SIZE OF n PLUS one  
        SET right_sums TO a LIST OF empty DICTIONARIES with a SIZE OF n PLUS one  
        
        FOR i FROM zero TO n  
            FOR each combination OF i elements IN left_half  
                SET subset_sum TO the SUM OF elements IN the combination  
                IF subset_sum NOT IN left_sums AT position i  
                    SET left_sums AT position i WITH key subset_sum TO zero  
                END IF  
                INCREMENT the VALUE at left_sums AT position i WITH key subset_sum BY one  
            END FOR  
            
            FOR each combination OF i elements IN right_half  
                SET subset_sum TO the SUM OF elements IN the combination  
                IF subset_sum NOT IN right_sums AT position i  
                    SET right_sums AT position i WITH key subset_sum TO zero  
                END IF  
                INCREMENT the VALUE at right_sums AT position i WITH key subset_sum BY one  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO n  
            SET left_values TO the SORTED LIST OF KEYS IN left_sums AT position i  
            SET right_values TO the SORTED LIST OF KEYS IN right_sums AT position n MINUS i  
            
            SET j TO zero  
            SET k TO the LENGTH OF right_values MINUS one  
            WHILE j LESS THAN the LENGTH OF left_values AND k GREATER THAN OR EQUAL TO zero  
                SET current_sum TO the VALUE at left_values AT position j PLUS the VALUE at right_values AT position k  
                IF min_diff GREATER THAN the ABSOLUTE VALUE OF total_sum MINUS two MULTIPLIED BY current_sum  
                    SET min_diff TO the ABSOLUTE VALUE OF total_sum MINUS two MULTIPLIED BY current_sum  
                END IF  
                
                IF current_sum LESS THAN target  
                    INCREMENT j BY one  
                ELSE IF current_sum GREATER THAN target  
                    DECREMENT k BY one  
                ELSE  
                    RETURN zero  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
