CLASS Solution  
    FUNCTION new21Game(n, k, maxPts)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k THEN  
                RETURN one WHEN i LESS THAN OR EQUAL TO n OTHERWISE zero  
            END IF  
            IF i EQUALS k MINUS one THEN  
                SET minimum_value TO the smaller value BETWEEN n MINUS k PLUS one AND maxPts  
                RETURN minimum_value DIVIDED BY maxPts  
            END IF  
            SET first_call TO CALL dfs WITH PARAMETERS i PLUS one  
            SET second_call TO CALL dfs WITH PARAMETERS i PLUS maxPts PLUS one  
            SET difference TO first_call MINUS second_call  
            RETURN first_call PLUS difference DIVIDED BY maxPts  
        END FUNCTION  
        RETURN CALL dfs WITH PARAMETERS zero  
    END FUNCTION  
END CLASS
