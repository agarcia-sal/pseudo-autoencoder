CLASS Solution  
    FUNCTION longestMountain(arr)  
        SET n TO the LENGTH OF arr  
        SET longest TO zero  
        
        SET i TO one  
        WHILE i IS LESS THAN n MINUS one  
            IF element at position i MINUS one OF arr IS LESS THAN element at position i OF arr AND element at position i OF arr IS GREATER THAN element at position i PLUS one OF arr  
                SET left TO i MINUS one  
                SET right TO i PLUS one  
                
                WHILE left IS GREATER THAN zero AND element at position left MINUS one OF arr IS LESS THAN element at position left OF arr  
                    DECREMENT left BY one  
                END WHILE  
                
                WHILE right IS LESS THAN n MINUS one AND element at position right OF arr IS GREATER THAN element at position right PLUS one OF arr  
                    INCREMENT right BY one  
                END WHILE  
                
                IF longest IS LESS THAN right MINUS left PLUS one  
                    SET longest TO right MINUS left PLUS one  
                END IF  
                
                SET i TO right  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN longest  
    END FUNCTION  
END CLASS
