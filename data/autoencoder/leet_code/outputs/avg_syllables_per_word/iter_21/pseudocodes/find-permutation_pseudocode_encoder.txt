CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO CALL list_with_consecutive_integers FROM one TO the LENGTH OF s PLUS one  
        SET start TO zero  
        FOR end FROM zero TO the LENGTH OF s  
            IF end EQUALS the LENGTH OF s OR the element at position end OF s EQUALS the character I  
                CALL reverse_sublist ON result FROM position start TO position end PLUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
