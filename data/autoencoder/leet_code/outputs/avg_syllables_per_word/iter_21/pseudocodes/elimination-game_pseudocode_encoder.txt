CLASS Solution  
    FUNCTION lastRemaining(n)  
        SET remaining TO n  
        SET step TO one  
        SET left_to_right TO true  
        SET head TO one  
        
        WHILE remaining GREATER THAN one HOLDS  
            IF left_to_right OR remaining MODULO two EQUALS one THEN  
                INCREMENT head BY step  
            END IF  
            MULTIPLY step BY two AND ASSIGN TO step  
            DIVIDE remaining BY two AND ASSIGN TO remaining  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        
        RETURN head  
    END FUNCTION  
END CLASS
