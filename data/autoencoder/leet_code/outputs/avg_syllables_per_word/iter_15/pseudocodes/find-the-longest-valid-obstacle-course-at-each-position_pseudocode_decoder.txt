CLASS Solution  
    FUNCTION longestObstacleCourseAtEachPosition WITH PARAMETER obstacles AS list of integers  
        SET n TO the LENGTH OF obstacles  
        SET ans TO an empty list  
        SET increasing_subseq TO an empty list  
        
        FOR each obstacle IN obstacles  
            SET idx TO the position AFTER the last position WHERE element is LESS THAN OR EQUAL TO obstacle IN increasing_subseq using binary search  
            
            IF idx EQUALS the LENGTH OF increasing_subseq  
                APPEND obstacle TO increasing_subseq  
            ELSE  
                SET the element at position idx of increasing_subseq TO obstacle  
            END IF  
            
            APPEND idx PLUS one TO ans  
        END FOR  
        
        RETURN ans  
    END FUNCTION  
END CLASS
