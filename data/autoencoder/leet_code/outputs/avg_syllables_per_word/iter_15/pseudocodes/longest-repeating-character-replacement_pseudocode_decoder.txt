CLASS Solution  
    FUNCTION characterReplacement(s k)  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        ASSIGN count TO a mapping with default zero for any character  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the element at position right of s in count BY one  
            SET max_count TO the GREATER OF max_count AND the element at position right of s in count  
            
            IF the LENGTH OF the substring from position left TO right of s MINUS max_count IS GREATER THAN k  
                DECREMENT the element at position left of s in count BY one  
                INCREMENT left BY one  
            END IF  
            
            SET current_window_length TO right MINUS left PLUS one  
            SET max_length TO the GREATER OF max_length AND current_window_length  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
