CLASS Solution  
    FUNCTION splitArraySameAverage(nums)  
        SET n TO the LENGTH OF nums  
        SET total_sum TO the SUM OF all elements IN nums  
        
        IF n EQUALS one  
            RETURN false  
        END IF  
        
        SET dp TO a LIST containing SETS with the NUMBER OF ELEMENTS EQUAL TO the INTEGER DIVISION OF n BY two PLUS one  
        ADD zero TO the SET at position zero OF dp  
        
        FOR each num IN nums  
            FOR size FROM the INTEGER DIVISION OF n BY two DOWN TO one  
                FOR each s IN the SET at position size MINUS one OF dp  
                    SET new_sum TO s PLUS num  
                    IF new_sum MULTIPLIED BY n EQUALS size MULTIPLIED BY total_sum  
                        RETURN true  
                    END IF  
                    ADD new_sum TO the SET at position size OF dp  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
