CLASS Solution  
    FUNCTION visiblePoints(points AS list of list of integer, angle AS integer, location AS list of integer) RETURNS integer  
        SET angles TO empty list  
        SET extra_points TO zero  
        
        FOR each pair x AND y IN points  
            IF x EQUALS element at position zero of location AND y EQUALS element at position one of location  
                INCREMENT extra_points BY one  
                CONTINUE to next iteration of loop  
            END IF  
            SET angle_rad TO arctangent of the difference of element at position one of y MINUS element at position one of location AND the difference of element at position zero of x MINUS element at position zero of location  
            SET angle_deg TO degrees conversion of angle_rad  
            APPEND angle_deg TO angles  
        END FOR  
        
        SORT angles in ascending order  
        
        SET angles TO concatenation of angles AND a list constructed by adding three hundred sixty to each element in angles  
        
        SET max_visible TO zero  
        SET left TO zero  
        
        FOR each integer right FROM zero TO the length of angles MINUS one  
            WHILE the difference of element at position right of angles MINUS element at position left of angles IS GREATER THAN angle  
                INCREMENT left BY one  
            END WHILE  
            SET max_visible TO the greater of max_visible AND the difference of right MINUS left PLUS one  
        END FOR  
        
        RETURN the sum of max_visible AND extra_points  
    END FUNCTION  
END CLASS
