CLASS Solution  
    FUNCTION numDupDigitsAtMostN(n)  
        FUNCTION count_unique_digits(x)  
            SET s TO the string representation of x  
            SET length TO the LENGTH OF s  
            SET count TO zero  
            FOR each integer i FROM one TO length MINUS one  
                INCREMENT count BY nine MULTIPLIED BY perm of nine MINUS one AND i MINUS one  
            END FOR  
            SET seen TO an empty set of digits  
            FOR each index i AND corresponding digit character digit IN s  
                FOR each integer d FROM zero TO the integer value of digit MINUS one  
                    IF i EQUALS zero THEN  
                        SET d TO start from one  
                    END IF  
                    IF d IS NOT IN seen THEN  
                        INCREMENT count BY perm of nine MINUS i AND length MINUS i MINUS one  
                    END IF  
                END FOR  
                IF the integer value of digit IS IN seen THEN  
                    BREAK the loop  
                END IF  
                ADD the integer value of digit TO seen  
            END FOR  
            IF the loop over s DID NOT BREAK THEN  
                INCREMENT count BY one  
            END IF  
            RETURN count  
        END FUNCTION  

        RETURN n MINUS count_unique_digits of n  
    END FUNCTION  

    FUNCTION perm(m, n)  
        IF n EQUALS zero THEN  
            RETURN one  
        END IF  
        SET result TO one  
        FOR each integer i FROM zero TO n MINUS one  
            SET result TO result MULTIPLIED BY m MINUS i  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
