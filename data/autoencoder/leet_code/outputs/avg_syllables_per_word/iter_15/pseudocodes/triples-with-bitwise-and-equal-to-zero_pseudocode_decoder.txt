CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO a new dictionary with default integer zero  
        FOR each element i IN nums  
            FOR each element j IN nums  
                INCREMENT the value at key representing the bitwise AND of i AND j in pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each element k IN nums  
            FOR each key pair_and AND corresponding value count IN pair_and_count  
                IF the bitwise AND of pair_and AND k EQUALS zero THEN  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
