CLASS Solution  
    FUNCTION isValidSerialization(preorder)  
        SET nodes TO the result of splitting the string preorder by the separator represented by a comma  
        SET slots TO one  
        FOR each node IN nodes  
            IF slots EQUALS zero  
                RETURN false  
            END IF  
            DECREMENT slots BY one  
            IF node IS NOT EQUAL TO the character representing a hash  
                INCREMENT slots BY two  
            END IF  
        END FOR  
        RETURN the result of testing whether slots EQUALS zero  
    END FUNCTION  
END CLASS
