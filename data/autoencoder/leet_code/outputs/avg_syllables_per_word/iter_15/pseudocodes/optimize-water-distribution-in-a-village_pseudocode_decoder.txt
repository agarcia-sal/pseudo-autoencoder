CLASS Solution  
    FUNCTION minCostToSupplyWater(n, wells, pipes)  
        FOR index FROM zero TO n MINUS one  
            APPEND list containing zero AND index PLUS one AND element at position index of wells TO pipes  
        END FOR  
        SORT pipes BY the element at position three of each item in ascending order  
        SET parent TO list of integers FROM zero TO n inclusive  
        FUNCTION find(x)  
            IF element at position x of parent IS NOT EQUAL TO x  
                SET element at position x of parent TO find(element at position x of parent)  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        FUNCTION union(x, y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX IS NOT EQUAL TO rootY  
                SET element at position rootX of parent TO rootY  
                RETURN true  
            END IF  
            RETURN false  
        END FUNCTION  
        SET min_cost TO zero  
        FOR each house1 house2 cost IN pipes  
            IF union(house1, house2) RETURNS true  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
