CLASS Solution  
    FUNCTION kEmptySlots(bulbs AS List OF Integer, k AS Integer)  
        SET bloom_days TO an empty list  
        FOR each day AND position FROM one TO the LENGTH OF bulbs WITH INDEXING STARTING AT one IN bulbs  
            SET i TO the position at which position SHOULD BE INSERTED TO KEEP bloom_days IN sorted order BASED ON bisect left of bloom_days WITH position  
            IF i IS GREATER THAN zero AND position MINUS element at position i MINUS one of bloom_days MINUS one EQUALS k  
                RETURN day  
            END IF  
            IF i IS LESS THAN the LENGTH OF bloom_days AND element at position i of bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position INTO bloom_days AT position i  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
