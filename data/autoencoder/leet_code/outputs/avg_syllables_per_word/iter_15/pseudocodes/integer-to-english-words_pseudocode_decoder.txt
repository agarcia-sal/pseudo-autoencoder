CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  
        
        SET below_twenty TO a list of words representing numbers from zero up to nineteen in order with empty string for zero position  
        SET tens TO a list of words representing multiples of ten from zero up to ninety with empty strings for first two positions  
        SET thousands TO a list of scale words for thousands, millions, and billions with empty string for units position  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN an empty string  
            ELSE IF n LESS THAN twenty  
                RETURN the element at position n of below_twenty concatenated with a space  
            ELSE IF n LESS THAN one hundred  
                RETURN the element at position the integer division of n by ten of tens concatenated with a space concatenated with the result of calling helper with the remainder of n divided by ten  
            ELSE  
                RETURN the element at position the integer division of n by one hundred of below_twenty concatenated with the word Hundred concatenated with a space concatenated with the result of calling helper with the remainder of n divided by one hundred  
            END IF  
        END FUNCTION  
        
        SET result TO an empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero  
            IF the remainder of num divided by one thousand NOT EQUALS zero  
                SET result TO the result of calling helper with the remainder of num divided by one thousand concatenated with the element at position i of thousands concatenated with a space concatenated with result  
            END IF  
            SET num TO the integer division of num by one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
