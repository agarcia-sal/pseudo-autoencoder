CLASS Solution  
    FUNCTION findReplaceString(s string, indices list of integers, sources list of strings, targets list of strings) RETURNS string  
        SET replacements TO list of tuples of elements from indices sources AND targets sorted by the first element of each tuple representing index  
        SET result TO empty list  
        SET prev_end TO zero  
        FOR each tuple consisting of index source AND target IN replacements  
            APPEND substring from position prev_end TO position index of string s TO result  
            IF substring from position index TO position index PLUS the LENGTH OF source of string s EQUALS source THEN  
                APPEND target TO result  
            ELSE  
                APPEND substring from position index TO position index PLUS the LENGTH OF source of string s TO result  
            END IF  
            SET prev_end TO index PLUS the LENGTH OF source  
        END FOR  
        APPEND substring from position prev_end TO the end of string s TO result  
        RETURN the concatenation of all elements in list result  
    END FUNCTION  
END CLASS
