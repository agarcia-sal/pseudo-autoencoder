CLASS Solution  
    FUNCTION primePalindrome WITH PARAMETER n  
        FUNCTION is_prime WITH PARAMETER num  
            IF num LESS THAN two  
                RETURN false  
            END IF  
            IF num EQUALS two OR num EQUALS three  
                RETURN true  
            END IF  
            IF num MODULO two EQUALS zero OR num MODULO three EQUALS zero  
                RETURN false  
            END IF  
            SET i TO five  
            WHILE i MULTIPLIED BY i LESS THAN OR EQUAL TO num  
                IF num MODULO i EQUALS zero OR num MODULO i PLUS two EQUALS zero  
                    RETURN false  
                END IF  
                INCREMENT i BY six  
            END WHILE  
            RETURN true  
        END FUNCTION  

        FUNCTION is_palindrome WITH PARAMETER num  
            RETURN the string representation of num EQUALS the reverse of the string representation of num  
        END FUNCTION  

        IF n LESS THAN OR EQUAL TO two  
            RETURN two  
        END IF  

        IF n GREATER THAN OR EQUAL TO ten AND n LESS THAN OR EQUAL TO eleven  
            RETURN eleven  
        END IF  

        WHILE true  
            IF n MODULO two EQUALS zero  
                INCREMENT n BY one  
                CONTINUE TO the next iteration of the WHILE loop  
            END IF  
            IF is_palindrome OF n AND is_prime OF n  
                RETURN n  
            END IF  
            IF the LENGTH OF the string representation of n MODULO two EQUALS zero  
                SET n TO ten RAISED TO the power of the LENGTH OF the string representation of n PLUS one  
            ELSE  
                INCREMENT n BY two  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
