CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of math infinity repeated fifteen times  
        FOR each first_element f and second_element r IN tires  
            SET total_time TO zero  
            FOR index i FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER OF i  
                IF lap_time IS GREATER THAN changeTime PLUS f  
                    BREAK the current loop  
                END IF  
                INCREMENT total_time BY lap_time  
                SET min_times element at position i TO the lesser value between min_times element at position i AND total_time  
            END FOR  
        END FOR  
  
        SET dp TO a list of math infinity repeated (numLaps PLUS one) times  
        SET dp element at position zero TO zero  
  
        FOR index i FROM one TO numLaps  
            FOR index j FROM zero TO the lesser value between i MINUS one AND fourteen  
                SET dp element at position i TO the lesser value between dp element at position i AND dp element at position i MINUS j MINUS one PLUS min_times element at position j PLUS changeTime  
            END FOR  
        END FOR  
  
        RETURN dp element at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
