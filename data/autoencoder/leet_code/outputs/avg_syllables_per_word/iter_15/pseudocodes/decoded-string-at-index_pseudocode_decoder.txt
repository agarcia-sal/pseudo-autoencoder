CLASS Solution  
    FUNCTION decodeAtIndex(s, k)  
        SET length TO zero  
        FOR each char IN s  
            IF char IS A DIGIT  
                SET length TO length MULTIPLIED BY the NUMERIC VALUE OF char  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each char IN THE REVERSE ORDER OF s  
            SET k TO the REMAINDER OF k DIVIDED BY length  
            IF k EQUALS zero AND char IS AN ALPHABETICAL CHARACTER  
                RETURN char  
            END IF  
            IF char IS A DIGIT  
                SET length TO length DIVIDED BY the NUMERIC VALUE OF char  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
