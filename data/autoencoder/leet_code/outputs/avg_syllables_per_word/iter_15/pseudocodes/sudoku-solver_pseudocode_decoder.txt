CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR each index FROM zero TO eight  
                IF element at position col of the row at position row of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            FOR each index FROM zero TO eight  
                IF element at position col of the row at position index of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY the integer division RESULT OF row DIVIDED BY three  
            SET start_col TO three MULTIPLIED BY the integer division RESULT OF col DIVIDED BY three  
            FOR each i FROM zero TO two  
                FOR each j FROM zero TO two  
                    IF element at position start_col PLUS j of the row at position start_row PLUS i of board EQUALS num  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
        FUNCTION solve(board)  
            FOR each row FROM zero TO eight  
                FOR each col FROM zero TO eight  
                    IF element at position col of the row at position row of board EQUALS the character dot  
                        FOR each num IN the string representations of numbers from one TO nine  
                            IF is_valid(board, row, col, num)  
                                SET element at position col of the row at position row of board TO num  
                                IF solve(board)  
                                    RETURN True  
                                END IF  
                                SET element at position col of the row at position row of board TO the character dot  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
        CALL solve(board)  
    END FUNCTION  
END CLASS
