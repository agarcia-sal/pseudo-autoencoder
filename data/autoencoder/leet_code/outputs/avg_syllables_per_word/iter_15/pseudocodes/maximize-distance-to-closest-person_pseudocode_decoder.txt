CLASS Solution  
    FUNCTION maxDistToClosest(seats)  
        SET max_distance TO zero  
        SET start TO zero  
        SET end TO zero  
        SET n TO the LENGTH OF seats  
        
        WHILE start IS LESS THAN n AND element at position start of seats EQUALS zero  
            INCREMENT start BY one  
        END WHILE  
        
        SET end TO n MINUS one  
        WHILE end IS GREATER THAN OR EQUAL TO zero AND element at position end of seats EQUALS zero  
            DECREMENT end BY one  
        END WHILE  
        
        SET max_distance TO the GREATER OF max_distance AND start  
        
        SET max_distance TO the GREATER OF max_distance AND n MINUS end MINUS one  
        
        SET current_distance TO zero  
        FOR i FROM start TO end  
            IF element at position i of seats EQUALS zero  
                INCREMENT current_distance BY one  
                SET max_distance TO the GREATER OF max_distance AND the INTEGER DIVISION OF the SUM OF current_distance AND one BY two  
            ELSE  
                SET current_distance TO zero  
            END IF  
        END FOR  
        
        RETURN max_distance  
    END FUNCTION  
END CLASS
