CLASS Solution  
    FUNCTION mincostToHireWorkers(quality AS List of integers, wage AS List of integers, k AS integer) RETURNS floating point number  
        SET workers TO THE RESULT OF SORTING the COLLECTION CREATED BY FOR each pair OF value w FROM wage AND value q FROM quality PAIRING the DIVISION of w BY q AND q  
        SET max_heap TO empty list  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each pair OF ratio AND q IN workers  
            APPLY heap push TO max_heap WITH NEGATION OF q AS VALUE  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap IS GREATER THAN k  
                SET removed_quality TO APPLY heap pop TO max_heap  
                INCREMENT total_quality BY removed_quality  
            END IF  
            IF the LENGTH OF max_heap IS EQUAL TO k  
                SET current_cost TO total_quality MULTIPLIED BY ratio  
                IF current_cost IS LESS THAN min_cost  
                    SET min_cost TO current_cost  
                END IF  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
