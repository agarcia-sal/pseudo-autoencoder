CLASS Solution  
    FUNCTION maxProduct(words)  
        SET word_sets TO a new list  
        FOR each word IN words  
            APPEND the set of unique characters of word TO word_sets  
        END FOR  
        
        SET max_product TO zero  
        
        FOR index i FROM zero TO the LENGTH OF words MINUS one  
            FOR index j FROM i PLUS one TO the LENGTH OF words MINUS one  
                IF the set at position i of word_sets INTERSECTED WITH the set at position j of word_sets IS EMPTY  
                    SET product TO the LENGTH OF the word at position i MULTIPLIED BY the LENGTH OF the word at position j  
                    IF product GREATER THAN max_product  
                        SET max_product TO product  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_product  
    END FUNCTION  
END CLASS
