CLASS Solution  
    FUNCTION nextGreaterElement WITH parameter n of type integer  
        SET digits TO the list of characters representing the string form of n  
        SET length TO the count of elements in digits  
        SET i TO the value of length MINUS two  
        WHILE i is greater than or equal to zero AND the element at position i of digits is GREATER THAN OR EQUAL TO the element at position i PLUS one of digits  
            DECREMENT i BY one  
        END WHILE  
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
        SET j TO the value of length MINUS one  
        WHILE the element at position j of digits is LESS THAN OR EQUAL TO the element at position i of digits  
            DECREMENT j BY one  
        END WHILE  
        SET temporary TO the element at position i of digits  
        SET the element at position i of digits TO the element at position j of digits  
        SET the element at position j of digits TO temporary  
        SET digits TO the concatenation of the elements from position zero TO position i of digits WITH the elements from position i PLUS one TO the last position of digits in reverse order  
        SET result TO the integer value represented by joining all elements of digits into a string  
        IF result IS GREATER THAN two raised to the power of thirty ONE MINUS one  
            RETURN negative one  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
