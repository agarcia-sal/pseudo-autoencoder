CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one point zero  
        END IF  
        
        SET n TO the integer result of n PLUS twenty four DIVIDED BY twenty five  
        
        SET memo TO an empty dictionary  
        
        FUNCTION dp(a, b)  
            IF the pair of a AND b EXISTS IN memo  
                RETURN the element at the pair of a AND b IN memo  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one point zero  
            END IF  
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero point zero  
            END IF  
            
            SET prob TO point two five MULTIPLIED BY the sum of  
                dp the argument of a MINUS four AND b PLUS  
                dp the argument of a MINUS three AND b MINUS one PLUS  
                dp the argument of a MINUS two AND b MINUS two PLUS  
                dp the argument of a MINUS one AND b MINUS three  
            
            ASSIGN memo at the pair of a AND b TO prob  
            RETURN prob  
        END FUNCTION  
        
        RETURN dp the argument of n AND n  
    END FUNCTION  
END CLASS
