CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO an empty mapping from content to list of paths  
        FOR each path IN paths  
            SET parts TO the list obtained by splitting path by blank space  
            SET directory TO the first element of parts  
            FOR each file_info IN the elements of parts starting from the second element  
                SET file_name TO the substring of file_info before the first occurrence of opening parenthesis  
                SET content TO the substring of file_info from the position after the first opening parenthesis TO the position before the last character  
                APPEND the concatenation of directory plus forward slash plus file_name TO the list associated with content in content_to_paths  
            END FOR  
        END FOR  
        RETURN the list of all lists of paths in content_to_paths where the number of elements in the list is greater than one  
    END FUNCTION  
END CLASS
