CLASS Solution  
    FUNCTION movesToStamp(stamp, target)  
        SET m TO the LENGTH OF stamp  
        SET n TO the LENGTH OF target  
        SET indeg TO a list of m REPEATED n MINUS m PLUS one TIMES  
        SET q TO an empty double ended queue  
        SET g TO a list containing empty lists, repeated n times  
        FOR each index i FROM zero TO n MINUS m inclusive  
            FOR each index j AND character c IN positions and elements of stamp  
                IF the character at position i PLUS j of target EQUALS c  
                    DECREMENT the element at position i of indeg BY one  
                    IF the element at position i of indeg EQUALS zero  
                        APPEND i TO q  
                    END IF  
                ELSE  
                    APPEND i TO the list at position i PLUS j of g  
                END IF  
            END FOR  
        END FOR  
        SET ans TO an empty list  
        SET vis TO a list of False values repeated n times  
        WHILE q is not empty  
            SET i TO the element removed from the left of q  
            APPEND i TO ans  
            FOR each index j FROM zero TO m MINUS one inclusive  
                IF the element at position i PLUS j of vis IS False  
                    SET the element at position i PLUS j of vis TO True  
                    FOR each element k IN the list at position i PLUS j of g  
                        DECREMENT the element at position k of indeg BY one  
                        IF the element at position k of indeg EQUALS zero  
                            APPEND k TO q  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  
        IF all elements of vis ARE True  
            RETURN the list ans in reverse order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
