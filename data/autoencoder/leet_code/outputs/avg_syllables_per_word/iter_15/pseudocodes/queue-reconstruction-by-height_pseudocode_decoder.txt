CLASS Solution  
    FUNCTION reconstructQueue WITH parameters people  
        CALL sort ON people USING a key that orders elements FIRST BY negative of element at position zero THEN BY element at position one in ascending order  
        SET queue TO an empty list  
        FOR each person IN people  
            INSERT person INTO queue AT position equal to element at position one of person  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
