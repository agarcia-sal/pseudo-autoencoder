CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO an empty list  
        SET score TO zero  
        FOR each character IN s  
            IF character EQUALS the opening parenthesis  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET score TO the element popped from stack PLUS if score EQUALS zero THEN one ELSE two MULTIPLIED BY score  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
