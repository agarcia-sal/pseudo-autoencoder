CLASS Solution  
    FUNCTION exclusiveTime(n, logs)  
        SET exclusive_time TO a list of zero repeated n times  
        SET stack TO an empty list  
        FOR each log IN logs  
            SPLIT log INTO function_id action timestamp by colon separator  
            CONVERT function_id TO integer  
            CONVERT timestamp TO integer  
            IF action EQUALS the word start  
                IF stack is not empty  
                    SET prev_function_id TO the first element of the last tuple in stack  
                    SET prev_timestamp TO the second element of the last tuple in stack  
                    INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND the tuple consisting of function_id and timestamp TO stack  
            ELSE  
                REMOVE the last tuple from stack AND ASSIGN the first element to prev_function_id AND the second element to prev_timestamp  
                INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp PLUS one  
                IF stack is not empty  
                    SET the last tuple of stack TO a tuple consisting of the first element of the last tuple of stack AND timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
