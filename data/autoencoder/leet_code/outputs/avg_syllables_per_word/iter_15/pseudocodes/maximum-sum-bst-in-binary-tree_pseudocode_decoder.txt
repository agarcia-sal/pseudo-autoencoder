CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH parameter value at position zero of values  
    SET i TO one  
    ASSIGN queue TO A NEW double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        ASSIGN node TO THE element FROM THE LEFT END OF queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO A NEW TreeNode WITH parameter value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO A NEW TreeNode WITH parameter value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF value of p IS NOT EQUALS TO value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION maxSumBST(self, root)  
        SET self.max_sum TO zero  

        FUNCTION helper(node)  
            IF node IS none  
                RETURN True, zero, positive infinity, negative infinity  
            END IF  

            SET left_is_bst, left_sum, left_min, left_max TO helper(left child of node)  
            SET right_is_bst, right_sum, right_min, right_max TO helper(right child of node)  

            IF left_is_bst IS True AND right_is_bst IS True AND left_max IS LESS THAN value of node AND value of node IS LESS THAN right_min  
                SET current_sum TO left_sum PLUS right_sum PLUS value of node  
                SET current_min TO THE MINIMUM OF value of node AND left_min  
                SET current_max TO THE MAXIMUM OF value of node AND right_max  
                SET self.max_sum TO THE MAXIMUM OF self.max_sum AND current_sum  
                RETURN True, current_sum, current_min, current_max  
            END IF  

            RETURN False, zero, positive infinity, negative infinity  
        END FUNCTION  

        CALL helper WITH root  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
