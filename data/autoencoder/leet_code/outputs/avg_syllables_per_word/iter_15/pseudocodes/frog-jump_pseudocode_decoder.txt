CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS EMPTY OR the LENGTH OF stones IS LESS THAN two  
            RETURN False  
        END IF  
        
        SET stone_set TO the SET CREATED FROM stones  
        
        SET memo TO an EMPTY DICTIONARY  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the element at the LAST POSITION OF stones  
                RETURN True  
            END IF  
            
            IF the TUPLE OF position AND jump IS A KEY IN memo  
                RETURN the VALUE ASSOCIATED WITH the TUPLE OF position AND jump IN memo  
            END IF  
            
            FOR each next_jump FROM one LESS THAN jump TO one MORE THAN jump  
                IF next_jump IS GREATER THAN zero AND the SUM OF position AND next_jump IS AN ELEMENT OF stone_set  
                    IF can_jump_to(the SUM OF position AND next_jump, next_jump) HOLDS TRUE  
                        SET the VALUE ASSOCIATED WITH the TUPLE OF position AND jump IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the VALUE ASSOCIATED WITH the TUPLE OF position AND jump IN memo TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(the ELEMENT at the FIRST POSITION OF stones, zero)  
    END FUNCTION  
END CLASS
