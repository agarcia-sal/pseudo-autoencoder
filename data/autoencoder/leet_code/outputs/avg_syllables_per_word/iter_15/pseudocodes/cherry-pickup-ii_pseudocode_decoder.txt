CLASS Solution  
    FUNCTION cherryPickup(grid)  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF the element at position zero of grid  
        
        FUNCTION dp(r, c1, c2)  
            IF c1 LESS THAN zero OR c1 GREATER THAN OR EQUAL TO cols OR c2 LESS THAN zero OR c2 GREATER THAN OR EQUAL TO cols THEN  
                RETURN zero  
            END IF  
            
            IF c1 EQUALS c2 THEN  
                SET cherries TO the element at position r of grid AT position c1  
            ELSE  
                SET cherries TO the element at position r of grid AT position c1 PLUS the element at position r of grid AT position c2  
            END IF  
            
            IF r EQUALS rows MINUS one THEN  
                RETURN cherries  
            END IF  
            
            SET max_cherries TO zero  
            FOR each dc1 IN the list of values negative one zero and one  
                FOR each dc2 IN the list of values negative one zero and one  
                    SET candidate TO dp of r PLUS one COMMA c1 PLUS dc1 COMMA c2 PLUS dc2  
                    IF candidate GREATER THAN max_cherries THEN  
                        SET max_cherries TO candidate  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN cherries PLUS max_cherries  
        END FUNCTION  
        
        RETURN dp of zero COMMA zero COMMA cols MINUS one  
    END FUNCTION  
END CLASS
