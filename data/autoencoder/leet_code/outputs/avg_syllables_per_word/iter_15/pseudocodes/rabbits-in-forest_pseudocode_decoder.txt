CLASS Solution  
    FUNCTION numRabbits WITH parameter answers AS List OF INTEGER  
        SET answer_counts TO COUNTER constructed FROM answers  
        SET total_rabbits TO zero  
        FOR each answer AND count IN the ITEMS OF answer_counts  
            SET group_size TO answer PLUS one  
            SET groups TO the INTEGER DIVISION OF count PLUS answer BY group_size  
            INCREMENT total_rabbits BY groups MULTIPLIED BY group_size  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
