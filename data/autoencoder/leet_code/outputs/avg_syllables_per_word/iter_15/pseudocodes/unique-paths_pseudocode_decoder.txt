CLASS Solution  
    FUNCTION uniquePaths WITH PARAMETERS m AND n  
        SET dp TO a two dimensional array WITH m rows AND n columns WHERE each element IS one  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET element at position i of dp AND position j TO element at position i MINUS one of dp AND position j PLUS element at position i of dp AND position j MINUS one  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one of dp AND position n MINUS one  
    END FUNCTION  
END CLASS
