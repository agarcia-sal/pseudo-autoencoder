CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET dictionary d TO an empty mapping from characters TO integers  
        SET index i TO zero  
        FOR each character c IN order  
            SET dictionary d at character c TO index i  
            INCREMENT index i BY one  
        END FOR  
        SET list_of_characters TO the characters in string s sorted in ascending order according to the value obtained by retrieving the value of character x from dictionary d if it exists otherwise zero  
        SET result_string TO the concatenation of elements in list_of_characters in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
