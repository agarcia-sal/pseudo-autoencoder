CLASS Solution  
    FUNCTION findKthNumber(m, n, k)  
        FUNCTION enough(x)  
            SET count TO zero  
            FOR i FROM one TO m  
                SET minimum_value TO the smaller value BETWEEN x DIVIDED BY i AND n  
                INCREMENT count BY minimum_value  
            END FOR  
            RETURN whether count IS GREATER THAN OR EQUAL TO k  
        END FUNCTION  
        
        SET low TO one  
        SET high TO m MULTIPLIED BY n  
        WHILE low IS LESS THAN high  
            SET mid TO the integer division result of low PLUS high DIVIDED BY two  
            IF enough(mid) IS true  
                SET high TO mid  
            ELSE  
                SET low TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
