CLASS Solution  
    FUNCTION letterCasePermutation WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER i  
            IF i IS GREATER THAN OR EQUAL TO THE LENGTH OF t  
                APPEND the concatenation of all elements IN t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH i PLUS one  
            IF the element at position i of t IS A LETTER OF THE ALPHABET  
                SET the element at position i of t TO the character obtained by applying bitwise exclusive or between the character code of the element at position i of t AND thirty two  
                CALL dfs WITH i PLUS one  
            END IF  
        END FUNCTION  
        SET t TO a list OF characters FROM s  
        SET ans TO an empty list  
        CALL dfs WITH zero  
        RETURN ans  
    END FUNCTION  
END CLASS
