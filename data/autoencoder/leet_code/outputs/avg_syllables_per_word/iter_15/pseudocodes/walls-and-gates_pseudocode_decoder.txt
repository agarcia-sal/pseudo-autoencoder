CLASS Solution  
    FUNCTION wallsAndGates(rooms)  
        IF rooms IS empty OR element at position zero of rooms IS empty  
            RETURN  
        END IF  
        SET rows TO the LENGTH OF rooms  
        SET cols TO the LENGTH OF element at position zero of rooms  
        CREATE an empty double ended queue named queue  
        FOR each row index r FROM zero TO rows MINUS one  
            FOR each column index c FROM zero TO cols MINUS one  
                IF element at position r then c of rooms EQUALS zero  
                    APPEND the tuple of r and c TO queue  
                END IF  
            END FOR  
        END FOR  
        SET directions TO the list containing tuples of zero and one THEN one and zero THEN zero and negative one THEN negative one and zero  
        WHILE queue IS NOT empty  
            REMOVE and ASSIGN the leftmost element of queue TO r and c  
            FOR each pair dr and dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr IS GREATER THAN OR EQUAL TO zero AND nr IS LESS THAN rows AND nc IS GREATER THAN OR EQUAL TO zero AND nc IS LESS THAN cols AND element at position nr then nc of rooms EQUALS two billion one hundred forty seven million four hundred eighty three thousand six hundred forty seven  
                    SET element at position nr then nc of rooms TO element at position r then c of rooms PLUS one  
                    APPEND the tuple of nr and nc TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
