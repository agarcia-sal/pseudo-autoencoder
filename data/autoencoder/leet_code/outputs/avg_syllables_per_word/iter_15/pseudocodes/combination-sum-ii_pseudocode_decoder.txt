CLASS Solution  
    FUNCTION combinationSum2(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO LENGTH OF candidates MINUS one  
                IF i GREATER THAN start AND element at position i of candidates EQUALS element at position i MINUS one of candidates  
                    CONTINUE  
                END IF  
                CALL backtrack WITH arguments i PLUS one AND target MINUS element at position i of candidates AND path CONCATENATED WITH element at position i of candidates  
            END FOR  
        END FUNCTION  
        CALL sort ON candidates  
        SET result TO empty list  
        CALL backtrack WITH arguments zero AND target AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
