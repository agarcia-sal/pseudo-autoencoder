CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty OR values IS none  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH parameter value at position zero of values  
    SET i TO one  
    SET queue TO NEW empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH parameter value at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH parameter value at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH parameters p dot left AND q dot left AND is_same_tree WITH parameters p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves(self, root)  
        FUNCTION collectLeaves(node)  
            IF node IS none  
                RETURN negative one  
            END IF  

            SET left_height TO collectLeaves WITH parameter node dot left  
            SET right_height TO collectLeaves WITH parameter node dot right  

            SET current_height TO the GREATER OF left_height AND right_height PLUS one  

            IF current_height IS GREATER THAN OR EQUAL TO the LENGTH OF result  
                APPEND an empty list TO result  
            END IF  

            APPEND node dot val TO the element at position current_height of result  

            RETURN current_height  
        END FUNCTION  

        SET result TO an empty list  
        CALL collectLeaves WITH parameter root  
        RETURN result  
    END FUNCTION  
END CLASS
