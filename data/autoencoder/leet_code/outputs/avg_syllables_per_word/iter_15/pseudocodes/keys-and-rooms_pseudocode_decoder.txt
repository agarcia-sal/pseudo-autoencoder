CLASS Solution  
    FUNCTION canVisitAllRooms WITH PARAMETERS rooms  
        SET visited TO empty collection of unique elements  
        SET stack TO collection containing zero  
        WHILE the LENGTH OF stack IS GREATER THAN zero  
            SET current_room TO element at last position of stack  
            REMOVE element at last position of stack  
            ADD current_room TO visited  
            FOR each key IN element at position current_room of rooms  
                IF key IS NOT IN visited  
                    APPEND key TO stack  
                END IF  
            END FOR  
        END WHILE  
        IF the LENGTH OF visited EQUALS the LENGTH OF rooms  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
