CLASS Solution  
    FUNCTION distinctEchoSubstrings WITH PARAMETER text  
        SET n TO the LENGTH OF text  
        CREATE substrings AS an empty set  
        FOR length FROM one TO n DIVIDED BY two PLUS one  
            FOR start FROM zero TO n MINUS two MULTIPLIED BY length PLUS one  
                SET a TO the substring from position start TO start PLUS length MINUS one of text  
                SET b TO the substring from position start PLUS length TO start PLUS two MULTIPLIED BY length MINUS one of text  
                IF a EQUALS b  
                    ADD the CONCATENATION of a AND b TO substrings  
                END IF  
            END FOR  
        END FOR  
        RETURN the SIZE OF substrings  
    END FUNCTION  
END CLASS
