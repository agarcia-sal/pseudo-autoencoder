CLASS Solution  
    FUNCTION searchMatrix(matrix parameter NAMED matrix TAKING a list of lists of integers AND target parameter NAMED target TAKING an integer) RETURNS boolean  
        IF matrix IS empty OR element at position zero of matrix IS empty  
            RETURN boolean false  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER DIVISION OF the SUM OF left AND right BY two  
            SET mid_value TO element at position the INTEGER DIVISION OF mid BY n of matrix AT position the MODULO OF mid BY n  
            IF mid_value EQUALS target  
                RETURN boolean true  
            ELSE IF mid_value IS LESS THAN target  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN boolean false  
    END FUNCTION  
END CLASS
