CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A new ListNode CREATED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the end  
        SET node TO A new ListNode CREATED WITH val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN THE TEST WHETHER the val attribute of p1 EQUALS the val attribute of p2 AND is_same_list CALLED WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A new TreeNode CREATED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO AN empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS not empty  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left attribute of node TO A new TreeNode CREATED WITH the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right attribute of node TO A new TreeNode CREATED WITH the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the val attribute of p IS NOT EQUALS TO the val attribute of q  
        RETURN false  
    ELSE  
        RETURN THE TEST WHETHER is_same_tree CALLED WITH the left attribute of p AND the left attribute of q AND is_same_tree CALLED WITH the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree WITH PARAMETERS self s  
        FUNCTION parse_tree WITH PARAMETER index  
            IF index IS GREATER THAN OR EQUAL TO THE LENGTH OF s  
                RETURN none AND index  
            END IF  
            SET start TO index  
            WHILE index IS LESS THAN THE LENGTH OF s AND (the element at position index of s IS A DIGIT OR the element at position index of s IS EQUALS TO the minus character)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO A new TreeNode CREATED WITH THE INTEGER VALUE PARSED FROM THE SUBSTRING FROM position start TO index OF s  
            IF index IS LESS THAN THE LENGTH OF s AND the element at position index of s IS EQUALS TO the opening parenthesis character  
                SET node.left AND index TO THE RESULT OF parse_tree CALLED WITH index PLUS one  
            END IF  
            IF index IS LESS THAN THE LENGTH OF s AND the element at position index of s IS EQUALS TO the opening parenthesis character  
                SET node.right AND index TO THE RESULT OF parse_tree CALLED WITH index PLUS one  
            END IF  
            IF index IS LESS THAN THE LENGTH OF s AND the element at position index of s IS EQUALS TO the closing parenthesis character  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND _ TO THE RESULT OF parse_tree CALLED WITH zero  
        RETURN root  
    END FUNCTION  
END CLASS
