CLASS Solution  
    FUNCTION pyramidTransition WITH PARAMETERS bottom AND allowed  
        SET allowed_map TO a mapping WHERE each key maps TO a mapping WHERE each key maps TO a list of possible top blocks  
        FOR each rule IN allowed  
            SET left TO the first character of rule  
            SET right TO the second character of rule  
            SET top TO the third character of rule  
            APPEND top TO the list at position right in the mapping at position left in allowed_map  
        END FOR  
        
        FUNCTION can_build_pyramid WITH PARAMETERS current_bottom AND current_top  
            IF the LENGTH OF current_bottom EQUALS one  
                RETURN true  
            END IF  
            IF the LENGTH OF current_top EQUALS the LENGTH OF current_bottom MINUS one  
                RETURN the result of calling can_build_pyramid WITH current_top and an empty list  
            END IF  
            
            SET left TO the element at position the LENGTH OF current_top of current_bottom  
            SET right TO the element at position the LENGTH OF current_top PLUS one of current_bottom  
            IF allowed_map HAS a key left AND the mapping at position left HAS a key right  
                FOR each top IN the list at position right in allowed_map at position left  
                    IF calling can_build_pyramid WITH current_bottom and current_top concatenated WITH top RETURNS true  
                        RETURN true  
                    END IF  
                END FOR  
            END IF  
            
            RETURN false  
        END FUNCTION  
        
        RETURN the result of calling can_build_pyramid WITH bottom and an empty list  
    END FUNCTION  
END CLASS
