CLASS Solution  
    FUNCTION longestValidParentheses with parameter s  
        SET stack TO a new list with the single element negative one  
        SET max_length TO zero  
        FOR each index i and character char IN the enumeration of s  
            IF character char EQUALS opening parenthesis  
                APPEND index i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF stack IS empty  
                    APPEND index i TO stack  
                ELSE  
                    SET max_length TO the greater value between max_length AND index i MINUS the last element of stack  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
