CLASS Solution  
    FUNCTION reorganizeString WITH PARAMETER s  
        SET count TO the frequency count of each character in s  
        SET max_heap TO an empty list  
        FOR each character and frequency IN count  
            APPEND the tuple of negative frequency and character TO max_heap  
        END FOR  
        TRANSFORM max_heap into a heap structure  
        
        SET prev_char TO none  
        SET prev_freq TO zero  
        SET result TO an empty list  
        
        WHILE max_heap is not empty OR prev_freq is less than zero  
            IF max_heap is empty AND prev_freq is less than zero  
                RETURN an empty string  
            END IF  
            
            REMOVE and ASSIGN the tuple of freq and char FROM the top element of max_heap  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq is less than zero  
                ADD the tuple of prev_freq and prev_char INTO max_heap maintaining heap property  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
