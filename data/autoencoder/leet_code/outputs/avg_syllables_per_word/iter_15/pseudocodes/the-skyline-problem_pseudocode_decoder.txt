CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO an empty list  
        FOR each building IN buildings  
            SET left TO the first element of building  
            SET right TO the second element of building  
            SET height TO the third element of building  
            APPEND a tuple consisting of left position, negative height, and right position TO events  
            APPEND a tuple consisting of right position, zero, and zero TO events  
        END FOR  
        
        SORT events in ascending order first by the first element then by the second element then by the third element  
        
        SET result TO a list containing a single element which is a list of zero and zero  
        SET max_heap TO a list containing a single tuple of zero and positive infinity  
        
        FOR each event IN events  
            SET x TO the first element of event  
            SET negH TO the second element of event  
            SET r TO the third element of event  
            
            WHILE the second element of the first tuple in max_heap is less than or equal to x holds  
                REMOVE the first element from max_heap following heap property  
            END WHILE  
            
            IF negH is not equal to zero THEN  
                ADD the tuple consisting of negH and r TO max_heap maintaining heap property  
            END IF  
            
            IF the second element of the last element in result is not equal to the negative of the first element of the first tuple in max_heap THEN  
                APPEND a list consisting of x and the negative of the first element of the first tuple in max_heap TO result  
            END IF  
        END FOR  
        
        IF the first element of result is a list of zero and zero THEN  
            REMOVE the first element from result  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
