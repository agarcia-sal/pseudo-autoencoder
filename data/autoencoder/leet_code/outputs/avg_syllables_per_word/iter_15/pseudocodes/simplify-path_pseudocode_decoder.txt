CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the result of SPLITTING path BY the character representing forward slash  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component EQUALS an empty string OR component EQUALS a single period  
                CONTINUE to the next iteration  
            ELSE IF component EQUALS two periods  
                IF the LENGTH OF stack IS GREATER THAN zero  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO the character representing forward slash CONCATENATED WITH the joining of elements IN stack WITH the character representing forward slash AS separator  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
