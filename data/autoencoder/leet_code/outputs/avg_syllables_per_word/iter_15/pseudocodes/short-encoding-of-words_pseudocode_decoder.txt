CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET reversed words TO words TRANSFORMED BY reversing each word AND sorted in ascending order  
        SET total length TO zero  
        FOR index FROM zero TO the LENGTH OF reversed words MINUS one  
            IF index PLUS one EQUALS the LENGTH OF reversed words OR the element at position index PLUS one of reversed words DOES NOT START WITH the element at position index of reversed words  
                INCREMENT total length BY the LENGTH OF the element at position index of reversed words PLUS one  
            END IF  
        END FOR  
        RETURN total length  
    END FUNCTION  
END CLASS
