CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO the LENGTH OF floor  
        CREATE a two dimensional list dp with dimensions n PLUS one BY numCarpets PLUS one FILLED WITH zero  
        FOR i FROM one TO n  
            SET dp AT position i AT zero carpets TO dp AT position i MINUS one AT zero carpets PLUS one IF the element at position i MINUS one of floor EQUALS one OTHERWISE zero  
        END FOR  
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp AT position i AT j carpets TO dp AT position i MINUS one AT j carpets PLUS one IF the element at position i MINUS one of floor EQUALS one OTHERWISE zero  
                SET dp AT position i AT j carpets TO THE MINIMUM OF dp AT position i AT j carpets AND dp AT position THE GREATER VALUE BETWEEN zero AND i MINUS carpetLen AT j MINUS one carpets  
            END FOR  
        END FOR  
        RETURN dp AT position n AT numCarpets  
    END FUNCTION  
END CLASS
