CLASS Solution  
    FUNCTION minArea(image, x, y)  
        IF image IS EMPTY OR the LENGTH OF element at position zero of image IS ZERO  
            RETURN zero  
        END IF  

        SET m TO the LENGTH OF image  
        SET n TO the LENGTH OF element at position zero of image  

        FUNCTION hasBlackPixelInRow(row)  
            FOR each pixel IN element at position row of image  
                IF pixel EQUALS one  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  

        FUNCTION hasBlackPixelInColumn(col)  
            FOR each row FROM zero TO m MINUS one  
                IF element at position col of element at position row of image EQUALS one  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  

        FUNCTION findTop()  
            SET low TO zero  
            SET high TO x  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF low PLUS high BY two  
                IF hasBlackPixelInRow(mid)  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  

        FUNCTION findBottom()  
            SET low TO x  
            SET high TO m MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF low PLUS high PLUS one BY two  
                IF hasBlackPixelInRow(mid)  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  

        FUNCTION findLeft()  
            SET low TO zero  
            SET high TO y  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF low PLUS high BY two  
                IF hasBlackPixelInColumn(mid)  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  

        FUNCTION findRight()  
            SET low TO y  
            SET high TO n MINUS one  
            WHILE low LESS THAN high  
                SET mid TO the INTEGER DIVISION OF low PLUS high PLUS one BY two  
                IF hasBlackPixelInColumn(mid)  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  

        SET top TO findTop()  
        SET bottom TO findBottom()  
        SET left TO findLeft()  
        SET right TO findRight()  

        RETURN the VALUE OF bottom MINUS top PLUS one MULTIPLIED BY right MINUS left PLUS one  
    END FUNCTION  
END CLASS
