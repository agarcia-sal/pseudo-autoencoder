CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO the SET containing all elements of bank  
        SET q TO a double ended queue initially containing the PAIR of start and zero  
        SET mp TO a MAPPING from character A TO the string TCG AND from character T TO the string ACG AND from character C TO the string ATG AND from character G TO the string ATC  
        WHILE the queue q is not empty  
            REMOVE the first element from the queue q and ASSIGN its first component TO t AND its second component TO step  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR index i FROM zero TO the LENGTH OF t MINUS one  
                SET v TO the element at position i of string t  
                FOR each character j IN the string mp[v]  
                    SET next TO the concatenation of the substring from position zero TO i MINUS one of string t PLUS character j PLUS the substring from position i PLUS one TO the last position of string t  
                    IF next IS AN ELEMENT OF the set s  
                        APPEND the PAIR of next and step PLUS one TO the queue q  
                        REMOVE next FROM the set s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
