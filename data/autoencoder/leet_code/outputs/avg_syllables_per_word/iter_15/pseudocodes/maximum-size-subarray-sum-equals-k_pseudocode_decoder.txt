CLASS Solution  
    FUNCTION maxSubArrayLen(nums list OF integer, k integer) RETURNS integer  
        SET cumulative_sum_index TO a dictionary with zero mapped TO negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        
        FOR each index i WITH value num FROM enumeration of nums  
            SET cumulative_sum TO cumulative_sum PLUS num  
            
            IF cumulative_sum MINUS k EXISTS AS A KEY IN cumulative_sum_index  
                SET candidate_length TO i MINUS the element at key cumulative_sum MINUS k OF cumulative_sum_index  
                IF candidate_length GREATER THAN max_length  
                    SET max_length TO candidate_length  
                END IF  
            END IF  
            
            IF cumulative_sum NOT EXISTS AS A KEY IN cumulative_sum_index  
                ASSIGN key cumulative_sum OF cumulative_sum_index TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
