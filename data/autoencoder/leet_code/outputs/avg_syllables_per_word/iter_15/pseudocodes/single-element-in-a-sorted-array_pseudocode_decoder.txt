CLASS Solution  
    FUNCTION singleNonDuplicate(nums)  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        WHILE left IS LESS THAN right  
            SET mid TO left PLUS the DIFFERENCE BETWEEN right AND left DIVIDED BY two USING INTEGER DIVISION  
            IF mid MODULO two IS EQUALS one  
                DECREMENT mid BY one  
            END IF  
            IF element at position mid of nums IS EQUALS TO element at position mid PLUS one of nums  
                SET left TO mid PLUS two  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN element at position left of nums  
    END FUNCTION  
END CLASS
