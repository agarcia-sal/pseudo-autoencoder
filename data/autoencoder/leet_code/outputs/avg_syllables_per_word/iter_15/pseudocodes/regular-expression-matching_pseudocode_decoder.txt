CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO two dimensional list with the LENGTH OF s PLUS one rows and the LENGTH OF p PLUS one columns filled with false values  
        SET the element at position zero zero of dp TO true  
        FOR each integer j FROM one TO the LENGTH OF p  
            IF element at position j MINUS one of p EQUALS the character representing asterisk  
                SET the element at position zero j of dp TO the element at position zero j MINUS two of dp  
            END IF  
        END FOR  
        FOR each integer i FROM one TO the LENGTH OF s  
            FOR each integer j FROM one TO the LENGTH OF p  
                IF element at position j MINUS one of p EQUALS element at position i MINUS one of s OR element at position j MINUS one of p EQUALS the character representing dot  
                    SET the element at position i j of dp TO the element at position i MINUS one j MINUS one of dp  
                ELSE IF element at position j MINUS one of p EQUALS the character representing asterisk  
                    SET the element at position i j of dp TO the element at position i j MINUS two of dp  
                    IF element at position j MINUS two of p EQUALS element at position i MINUS one of s OR element at position j MINUS two of p EQUALS the character representing dot  
                        SET the element at position i j of dp TO the element at position i j of dp OR the element at position i MINUS one j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position the LENGTH OF s the LENGTH OF p of dp  
    END FUNCTION  
END CLASS
