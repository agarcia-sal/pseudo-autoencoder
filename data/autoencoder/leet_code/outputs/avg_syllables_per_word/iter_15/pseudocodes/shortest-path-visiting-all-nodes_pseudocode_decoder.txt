CLASS Solution  
    FUNCTION shortestPathLength(graph)  
        IF graph EQUALS empty  
            RETURN zero  
        END IF  

        SET n TO the LENGTH OF graph  
        SET queue TO a double ended queue CONTAINING tuples OF each node AND one SHIFTED LEFT BY the node FROM zero TO n MINUS one  
        SET visited TO a set CONTAINING tuples OF each node AND one SHIFTED LEFT BY the node FROM zero TO n MINUS one  

        SET steps TO zero  
        SET all_nodes_visited TO one SHIFTED LEFT BY n MINUS one  

        WHILE queue is not empty  
            SET number_of_elements TO the LENGTH OF queue  
            FOR index FROM zero TO number_of_elements MINUS one  
                REMOVE the leftmost element FROM queue AND ASSIGN IT TO current_node AND visited_nodes  

                IF visited_nodes EQUALS all_nodes_visited  
                    RETURN steps  
                END IF  

                FOR each neighbor IN element at position current_node of graph  
                    SET next_visited_nodes TO visited_nodes OR one SHIFTED LEFT BY neighbor  
                    SET next_state TO tuple OF neighbor AND next_visited_nodes  

                    IF next_state NOT IN visited  
                        ADD next_state TO visited  
                        APPEND next_state TO queue  
                    END IF  
                END FOR  
            END FOR  

            INCREMENT steps BY one  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
