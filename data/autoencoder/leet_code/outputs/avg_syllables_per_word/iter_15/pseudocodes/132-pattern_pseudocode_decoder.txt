CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO an empty list  
        FOR each number IN the reverse order of nums  
            IF number IS LESS THAN second  
                RETURN true  
            END IF  
            WHILE stack IS NOT empty AND the element at the last position of stack IS LESS THAN number  
                SET second TO the element popped from the end of stack  
            END WHILE  
            APPEND number TO stack  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
