CLASS Solution  
    FUNCTION minimumTime(s)  
        SET n TO the LENGTH OF s  
        SET left_costs TO a list of zero values of LENGTH n  
        SET right_costs TO a list of zero values of LENGTH n  

        IF the element at position zero of s EQUALS the character one  
            SET the element at position zero of left_costs TO one  
        ELSE  
            SET the element at position zero of left_costs TO zero  
        END IF  

        FOR index i FROM one TO n MINUS one  
            IF the element at position i of s EQUALS the character zero  
                SET the element at position i of left_costs TO the element at position i MINUS one of left_costs  
            ELSE  
                SET the element at position i of left_costs TO the smaller VALUE BETWEEN i PLUS one AND the element at position i MINUS one of left_costs PLUS two  
            END IF  
        END FOR  

        IF the element at the last position of s EQUALS the character one  
            SET the element at the last position of right_costs TO one  
        ELSE  
            SET the element at the last position of right_costs TO zero  
        END IF  

        FOR index i FROM n MINUS two DOWN TO zero  
            IF the element at position i of s EQUALS the character zero  
                SET the element at position i of right_costs TO the element at position i PLUS one of right_costs  
            ELSE  
                SET the element at position i of right_costs TO the smaller VALUE BETWEEN n MINUS i AND the element at position i PLUS one of right_costs PLUS two  
            END IF  
        END FOR  

        SET min_time TO the smaller VALUE BETWEEN the element at the last position of left_costs AND the element at position zero of right_costs  

        FOR index i FROM zero TO n MINUS two  
            SET min_time TO the smaller VALUE BETWEEN min_time AND the sum of the element at position i of left_costs PLUS the element at position i PLUS one of right_costs  
        END FOR  

        RETURN min_time  
    END FUNCTION  
END CLASS
