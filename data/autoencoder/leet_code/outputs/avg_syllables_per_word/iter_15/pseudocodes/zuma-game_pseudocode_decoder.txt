CLASS Solution  
    FUNCTION findMinStep(board hand)  
        FUNCTION clean(board)  
            SET stack TO empty list  
            FOR each ball IN board  
                IF stack IS NOT empty AND element at position zero of element at position LENGTH OF stack MINUS one of stack IS NOT EQUALS TO ball AND second element at position LENGTH OF stack MINUS one of stack IS GREATER THAN OR EQUAL TO three  
                    REMOVE element at position LENGTH OF stack MINUS one FROM stack  
                END IF  
                IF stack IS empty OR element at position zero of element at position LENGTH OF stack MINUS one of stack IS NOT EQUALS TO ball  
                    APPEND list containing ball AND one TO stack  
                ELSE  
                    INCREMENT second element at position LENGTH OF stack MINUS one of stack BY one  
                END IF  
            END FOR  
            IF stack IS NOT empty AND second element at position LENGTH OF stack MINUS one of stack IS GREATER THAN OR EQUAL TO three  
                REMOVE element at position LENGTH OF stack MINUS one FROM stack  
            END IF  
            RETURN concatenation of each ball REPEATED count times FOR each ball AND count IN stack  
        END FUNCTION  
        
        FUNCTION dfs(board hand_counter)  
            IF board IS empty  
                RETURN zero  
            END IF  
            SET min_steps TO positive infinity  
            SET i TO zero  
            WHILE i IS LESS THAN length of board  
                SET j TO i PLUS one  
                WHILE j IS LESS THAN length of board AND element at position i of board EQUALS element at position j of board  
                    INCREMENT j BY one  
                END WHILE  
                SET increment TO three MINUS (j MINUS i)  
                IF value at key element at position i of board IN hand_counter IS GREATER THAN OR EQUAL TO increment  
                    SET new_hand_counter TO a copy of hand_counter  
                    DECREMENT value at key element at position i of board IN new_hand_counter BY increment  
                    SET new_board TO result of calling clean WITH concatenation of substring from start TO i MINUS one of board WITH substring from j TO end of board  
                    SET steps TO result of calling dfs WITH new_board AND new_hand_counter  
                    IF steps IS GREATER THAN OR EQUAL TO zero  
                        SET min_steps TO minimum OF min_steps AND steps PLUS increment  
                    END IF  
                END IF  
                SET i TO j  
            END WHILE  
            IF min_steps IS EQUAL TO positive infinity  
                RETURN negative one  
            ELSE  
                RETURN min_steps  
            END IF  
        END FUNCTION  
        
        SET hand_counter TO counter initialized with hand  
        RETURN result of calling dfs WITH board AND hand_counter  
    END FUNCTION  
END CLASS
