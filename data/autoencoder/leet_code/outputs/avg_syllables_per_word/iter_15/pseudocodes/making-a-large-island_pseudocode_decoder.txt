CLASS Solution  
    FUNCTION largestIsland(grid)  
        IF grid EQUALS no value OR element at position zero of grid EQUALS no value  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF grid  
        SET directions TO COLLECTION OF position zero ZERO AND position one ONE AND position zero ONE AND position one ZERO AND position zero ZERO AND position one MINUS ONE AND position one MINUS ONE AND position zero ZERO  

        FUNCTION is_valid(x, y)  
            RETURN x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN n AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN n  
        END FUNCTION  

        FUNCTION dfs(x, y, index)  
            SET area TO one  
            SET element at position x AND position y of grid TO index  
            FOR each pair dx AND dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF is_valid(nx, ny) AND element at position nx AND position ny of grid EQUALS one  
                    INCREMENT area BY dfs(nx, ny, index)  
                END IF  
            END FOR  
            RETURN area  
        END FUNCTION  

        SET island_areas TO empty dictionary  
        SET index TO two  

        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i AND position j of grid EQUALS one  
                    SET value at key index of island_areas TO dfs(i, j, index)  
                    INCREMENT index BY one  
                END IF  
            END FOR  
        END FOR  

        IF island_areas EQUALS no value  
            RETURN one  
        END IF  

        SET max_island_size TO the MAXIMUM VALUE in island_areas  

        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i AND position j of grid EQUALS zero  
                    SET neighbors TO empty collection  
                    FOR each pair dx AND dy IN directions  
                        SET nx TO i PLUS dx  
                        SET ny TO j PLUS dy  
                        IF is_valid(nx, ny) AND element at position nx AND position ny of grid IS GREATER THAN one  
                            ADD element at position nx AND position ny of grid TO neighbors  
                        END IF  
                    END FOR  
                    SET current_island_size TO one PLUS the SUM OF the values at keys each element IN neighbors OF island_areas  
                    SET max_island_size TO the MAXIMUM OF max_island_size AND current_island_size  
                END IF  
            END FOR  
        END FOR  

        RETURN max_island_size  
    END FUNCTION  
END CLASS
