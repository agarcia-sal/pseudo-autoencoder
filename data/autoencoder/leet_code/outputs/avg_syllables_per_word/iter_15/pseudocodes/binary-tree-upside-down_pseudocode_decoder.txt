CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS None right EQUALS None  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values list  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode constructed WITH element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is GREATER THAN zero  
        REMOVE and RETURN the element from the front of queue ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO a new TreeNode constructed WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO a new TreeNode constructed WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value property of p IS NOT EQUALS the value property of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS the left property of p AND the left property of q AND the result of is_same_tree WITH PARAMETERS the right property of p AND the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree WITH PARAMETERS self root optional TreeNode  
        IF root IS None OR the left property of root IS None  
            RETURN root  
        END IF  
        SET new_root TO the result of calling upsideDownBinaryTree ON self WITH PARAMETERS the left property of root  
        SET the left property of the left property of root TO the right property of root  
        SET the right property of the left property of root TO root  
        SET the left property of root TO None  
        SET the right property of root TO None  
        RETURN new_root  
    END FUNCTION  
END CLASS
