CLASS Solution  
    FUNCTION fullBloomFlowers WITH parameters flowers AND people  
        SET start_times TO the SORTED LIST of the first element of each pair in flowers  
        SET end_times TO the SORTED LIST of the second element of each pair in flowers  
        SET result TO an empty LIST  
        FOR each person IN people  
            SET started TO the position after the last element in start_times that is LESS THAN OR EQUAL TO person  
            SET ended TO the position of the first element in end_times that is GREATER THAN OR EQUAL TO person  
            APPEND the difference between started and ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
