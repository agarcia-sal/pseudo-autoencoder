CLASS Solution  
    FUNCTION maximumScore(scores, edges)  
        SET graph TO a new mapping with default values as empty list  
        FOR each pair u and v IN edges  
            APPEND the tuple consisting of element at position v of scores and v TO the element at position u of graph  
            APPEND the tuple consisting of element at position u of scores and u TO the element at position v of graph  
        END FOR  
        
        FOR each node IN the keys of graph  
            SET the element at position node of graph TO the list of three largest elements of the element at position node of graph  
        END FOR  
        
        SET max_score TO negative one  
        
        FOR each pair u and v IN edges  
            FOR each pair score1 and x IN the element at position u of graph  
                FOR each pair score2 and y IN the element at position v of graph  
                    IF the number of unique elements among x y u and v EQUALS four  
                        SET max_score TO the maximum between max_score and the sum of score1 score2 element at position u of scores and element at position v of scores  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_score  
    END FUNCTION  
END CLASS
