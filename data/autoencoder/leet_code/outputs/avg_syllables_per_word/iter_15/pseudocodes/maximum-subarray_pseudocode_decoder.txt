CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO element at position zero of nums  
        SET current_sum TO element at position zero of nums  
        FOR each num IN elements of nums from position one TO the last position  
            SET current_sum TO the greater value BETWEEN num AND current_sum PLUS num  
            SET max_sum TO the greater value BETWEEN max_sum AND current_sum  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
