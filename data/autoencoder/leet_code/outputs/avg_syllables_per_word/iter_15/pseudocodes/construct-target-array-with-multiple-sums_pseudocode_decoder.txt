CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO an empty list  
        FOR each number IN target  
            APPEND the negative of number TO max_heap  
        END FOR  
        CONVERT max_heap INTO a heap structure  
        SET total_sum TO the sum of all elements in target  
        WHILE the element at position zero of max_heap is LESS THAN negative one  
            SET largest TO the negative of the element removed from max_heap that is the smallest element  
            SET rest_sum TO total_sum MINUS largest  
            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum IS EQUAL TO zero  
                RETURN False  
            END IF  
            SET new_value TO largest MODULO rest_sum  
            IF new_value IS EQUAL TO zero AND rest_sum IS NOT EQUAL TO one  
                RETURN False  
            END IF  
            SET total_sum TO rest_sum PLUS new_value  
            ADD the negative of new_value TO max_heap maintaining heap properties  
        END WHILE  
        RETURN True  
    END FUNCTION  
END CLASS
