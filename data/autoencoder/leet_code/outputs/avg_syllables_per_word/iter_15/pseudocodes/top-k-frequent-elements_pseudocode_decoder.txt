CLASS Solution  
    FUNCTION topKFrequent(nums AS List OF Integer, k AS Integer) AS List OF Integer  
        SET frequency_map TO a new mapping that returns zero for missing keys  
        FOR each num IN nums  
            INCREMENT the value at key num IN frequency_map BY one  
        END FOR  
        SET max_frequency TO the greatest value among all values in frequency_map  
        SET buckets TO a new list of empty lists with count max_frequency PLUS one  
        FOR each key num and value freq IN frequency_map  
            APPEND num TO the list at position freq of buckets  
        END FOR  
        SET result TO a new empty list  
        FOR freq FROM max_frequency DOWN TO one  
            IF the length of the list at position freq of buckets is greater than zero  
                EXTEND result BY all elements in the list at position freq of buckets  
                IF the length of result is greater than or equals to k  
                    RETURN the first k elements of result  
                END IF  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
