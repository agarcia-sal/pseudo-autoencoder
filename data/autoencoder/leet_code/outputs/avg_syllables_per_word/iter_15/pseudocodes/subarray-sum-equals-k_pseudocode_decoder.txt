CLASS Solution  
    FUNCTION subarraySum(nums k)  
        SET cumulative_sum_freq TO a dictionary with key zero MAPPED TO value one  
        SET current_sum TO zero  
        SET count TO zero  
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF current_sum MINUS k IS A KEY IN cumulative_sum_freq  
                INCREMENT count BY the value at key current_sum MINUS k in cumulative_sum_freq  
            END IF  
            IF current_sum IS A KEY IN cumulative_sum_freq  
                INCREMENT the value at key current_sum in cumulative_sum_freq BY one  
            ELSE  
                SET the value at key current_sum in cumulative_sum_freq TO one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
