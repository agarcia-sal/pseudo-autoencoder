CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the string at position index of strs REVERSED  
            IF reversed_str IS GREATER THAN the string at position index of strs  
                SET the string at position index of strs TO reversed_str  
            END IF  
        END FOR  
        
        SET doubled_strs TO the concatenation of strs WITH strs  
        
        SET max_string TO the empty string  
        
        FOR start_index FROM zero TO the LENGTH OF strs MINUS one  
            FOR is_reversed IN the list containing false and true  
                IF is_reversed IS true  
                    SET current TO the string at position start_index of strs REVERSED  
                ELSE  
                    SET current TO the string at position start_index of strs  
                END IF  
                
                SET remaining TO the concatenation of the strings from position start_index PLUS one TO start_index PLUS the LENGTH OF strs MINUS one in doubled_strs  
                
                FOR cut_position FROM zero TO the LENGTH OF current  
                    SET candidate TO the substring from position cut_position TO the end of current CONCATENATED WITH remaining CONCATENATED WITH the substring from position zero TO cut_position MINUS one of current  
                    
                    IF candidate IS GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
