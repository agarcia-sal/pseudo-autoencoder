CLASS Solution  
    FUNCTION minEatingSpeed WITH PARAMETERS piles AND h  
        FUNCTION canFinish WITH PARAMETER k  
            SET hours TO zero  
            FOR each pile IN piles  
                INCREMENT hours BY the CEILING OF pile DIVIDED BY k  
            END FOR  
            RETURN hours LESS THAN OR EQUAL TO h  
        END FUNCTION  
      
        SET left TO one  
        SET right TO the MAXIMUM ELEMENT OF piles  
      
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER RESULT OF left PLUS right DIVIDED BY two  
            IF canFinish WITH ARGUMENT mid RETURNS true  
                DECREMENT right BY one FROM mid  
            ELSE  
                INCREMENT left BY one FROM mid  
            END IF  
        END WHILE  
      
        RETURN left  
    END FUNCTION  
END CLASS
