CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF the LENGTH OF nums LESS THAN two  
            RETURN zero  
        END IF  
      
        FUNCTION countingSort(arr, exp)  
            SET n TO the LENGTH OF arr  
            SET output TO a list of zeros of LENGTH n  
            SET count TO a list of ten zeros  
          
            FOR each index FROM zero TO n MINUS one  
                SET digit_index TO the integer DIVISION OF the element at position index of arr BY exp  
                INCREMENT element at position the remainder of digit_index DIVIDED BY ten of count BY one  
            END FOR  
          
            FOR i FROM one TO nine  
                INCREMENT element at position i of count BY element at position i MINUS one of count  
            END FOR  
          
            SET i TO n MINUS one  
            WHILE i GREATER THAN OR EQUAL TO zero  
                SET digit_index TO the integer DIVISION OF the element at position i of arr BY exp  
                ASSIGN element at position element at position the remainder of digit_index DIVIDED BY ten of count MINUS one of output TO element at position i of arr  
                DECREMENT element at position the remainder of digit_index DIVIDED BY ten of count BY one  
                DECREMENT i BY one  
            END WHILE  
          
            FOR i FROM zero TO n MINUS one  
                SET element at position i of arr TO element at position i of output  
            END FOR  
        END FUNCTION  
      
        FUNCTION radixSort(arr)  
            SET max1 TO the MAXIMUM of arr  
            SET exp TO one  
            WHILE the integer DIVISION of max1 BY exp GREATER THAN zero  
                CALL countingSort WITH arr AND exp  
                SET exp TO exp MULTIPLIED BY ten  
            END WHILE  
        END FUNCTION  
      
        CALL radixSort WITH nums  
      
        SET max_diff TO zero  
        FOR i FROM one TO the LENGTH OF nums MINUS one  
            SET current_diff TO the element at position i of nums MINUS the element at position i MINUS one of nums  
            IF current_diff GREATER THAN max_diff  
                SET max_diff TO current_diff  
            END IF  
        END FOR  
      
        RETURN max_diff  
    END FUNCTION  
END CLASS
