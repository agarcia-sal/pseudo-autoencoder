CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the val ATTRIBUTE OF this OBJECT TO val  
        SET the next ATTRIBUTE OF this OBJECT TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO a NEW ListNode INITIALIZED WITH the element at position zero OF values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the last position  
        SET node TO a NEW ListNode INITIALIZED WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND is_same_list CALLED WITH the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd WITH PARAMETERS head AND n  
        SET dummy TO a NEW ListNode INITIALIZED WITH zero  
        SET the next ATTRIBUTE OF dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n INCLUSIVE  
            SET first TO the next ATTRIBUTE OF first  
        END FOR  
        WHILE first NOT EQUALS none  
            SET first TO the next ATTRIBUTE OF first  
            SET second TO the next ATTRIBUTE OF second  
        END WHILE  
        SET the next ATTRIBUTE OF second TO the next ATTRIBUTE OF the next ATTRIBUTE OF second  
        RETURN the next ATTRIBUTE OF dummy  
    END FUNCTION  
END CLASS
