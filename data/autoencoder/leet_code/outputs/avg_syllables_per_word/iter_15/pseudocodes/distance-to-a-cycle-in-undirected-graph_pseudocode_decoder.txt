CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO a default dictionary that returns an empty list  
        FOR each u comma v IN edges  
            APPEND v TO the element at position u of graph  
            APPEND u TO the element at position v of graph  
        END FOR  
        
        FUNCTION dfs(node, parent, path, visited)  
            IF node IS IN visited THEN  
                SET cycle_start TO the position of node in path  
                RETURN the sublist of path from position cycle_start TO the end  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN the element at position node of graph  
                IF neighbor IS NOT EQUAL TO parent THEN  
                    SET cycle TO the call of dfs with neighbor comma node comma path comma visited  
                    IF cycle IS NOT NIL THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE the last element FROM path  
            RETURN NIL  
        END FUNCTION  
        
        SET visited TO an empty set  
        SET cycle TO NIL  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                SET cycle TO the call of dfs with node comma negative one comma empty list comma visited  
                IF cycle IS NOT NIL THEN  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET distance TO a list of zeros with length n  
        SET queue TO a deque initialized with cycle  
        SET visited TO a set initialized with cycle  
        
        SET level TO zero  
        WHILE queue is not empty  
            FOR each iteration FROM one TO the length of queue  
                REMOVE the first element FROM queue and ASSIGN TO node  
                SET the element at position node of distance TO level  
                FOR each neighbor IN the element at position node of graph  
                    IF neighbor IS NOT IN visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  
        
        RETURN distance  
    END FUNCTION  
END CLASS
