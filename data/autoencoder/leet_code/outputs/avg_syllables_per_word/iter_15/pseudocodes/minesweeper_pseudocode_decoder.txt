CLASS Solution  
    FUNCTION updateBoard(board AS list of lists of strings, click AS list of integers) AS list of lists of strings  
        FUNCTION count_adjacent_mines(r AS integer, c AS integer) AS integer  
            SET count TO zero  
            FOR i FROM r MINUS one TO r PLUS one  
                FOR j FROM c MINUS one TO c PLUS one  
                    IF i IS GREATER THAN OR EQUAL TO zero AND i IS LESS THAN the LENGTH OF board AND j IS GREATER THAN OR EQUAL TO zero AND j IS LESS THAN the LENGTH OF element at position zero OF board AND element at position i of board at position j EQUALS the character representing a mine  
                        INCREMENT count BY one  
                    END IF  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        FUNCTION dfs(r AS integer, c AS integer)  
            IF r IS LESS THAN zero OR r IS GREATER THAN OR EQUAL TO the LENGTH OF board OR c IS LESS THAN zero OR c IS GREATER THAN OR EQUAL TO the LENGTH OF element at position zero OF board  
                RETURN  
            END IF  
            IF element at position r of board at position c IS NOT the character representing an unrevealed empty square  
                RETURN  
            END IF  
            
            SET mine_count TO count_adjacent_mines(r, c)  
            IF mine_count IS GREATER THAN zero  
                SET element at position r of board at position c TO the string representation of mine_count  
            ELSE  
                SET element at position r of board at position c TO the character representing a revealed blank square  
                FOR i FROM r MINUS one TO r PLUS one  
                    FOR j FROM c MINUS one TO c PLUS one  
                        CALL dfs(i, j)  
                    END FOR  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET row TO element at position zero of click  
        SET col TO element at position one of click  
        IF element at position row of board at position col IS the character representing a mine  
            SET element at position row of board at position col TO the character representing a revealed mine explosion  
            RETURN board  
        END IF  
        
        CALL dfs(row, col)  
        RETURN board  
    END FUNCTION  
END CLASS
