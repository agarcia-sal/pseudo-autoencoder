CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            RETURN subtract the element at position zero of a MINUS the element at position zero of o MULTIPLIED BY subtract the element at position one of b MINUS the element at position one of o MINUS subtract the element at position one of a MINUS the element at position one of o MULTIPLIED BY subtract the element at position zero of b MINUS the element at position zero of o  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  
        
        SET orientation TO zero  
        
        FOR index i FROM zero TO n MINUS one  
            SET o TO the element at position i of points  
            SET a TO the element at position the remainder of index i PLUS one DIVIDED BY n of points  
            SET b TO the element at position the remainder of index i PLUS two DIVIDED BY n of points  
            SET cross TO call cross_product with parameters o, a, and b  
            
            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO minus one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS minus one AND cross GREATER THAN zero  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
