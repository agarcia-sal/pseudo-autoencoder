CLASS Solution  
    FUNCTION wordSquares(words)  
        ASSIGN pointer TO a new dictionary mapping from string prefix TO list of strings named prefix_to_words  
        
        FOR each word IN words  
            FOR index FROM zero TO the LENGTH OF word MINUS one  
                SET prefix TO the substring from position zero TO index MINUS one of word  
                APPEND word TO the list associated with prefix IN prefix_to_words  
            END FOR  
        END FOR  
        
        SET results TO an empty list  
        
        FUNCTION backtrack(square)  
            IF the LENGTH OF square EQUALS the LENGTH OF the first element of words THEN  
                APPEND square TO results  
                RETURN  
            END IF  
            
            SET prefix_parts TO an empty list  
            FOR each word IN square  
                APPEND the element at position LENGTH OF square of word TO prefix_parts  
            END FOR  
            SET prefix TO the concatenation of all elements of prefix_parts  
            
            FOR each candidate IN the list associated with prefix IN prefix_to_words  
                CALL backtrack with the argument as the concatenation of square and a list containing candidate  
            END FOR  
        END FUNCTION  
        
        FOR each word IN words  
            CALL backtrack with a list containing word  
        END FOR  
        
        RETURN results  
    END FUNCTION  
END CLASS
