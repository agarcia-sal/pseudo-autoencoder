CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        SET graph TO a default dictionary mapping each key to an empty list  
        SET indegree TO a list of zeros of length numCourses  
        
        FOR each pair course and prereq IN prerequisites  
            APPEND course TO the list at key prereq in graph  
            INCREMENT the element at position course of indegree BY one  
        END FOR  
        
        SET queue TO a deque containing all courses from zero TO numCourses MINUS one where the element at position course of indegree EQUALS zero  
        
        WHILE queue is not empty  
            REMOVE the element from the left of queue and ASSIGN TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN the list at key current_course of graph  
                DECREMENT the element at position next_course of indegree BY one  
                IF the element at position next_course of indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the test whether numCourses EQUALS zero  
    END FUNCTION  
END CLASS
