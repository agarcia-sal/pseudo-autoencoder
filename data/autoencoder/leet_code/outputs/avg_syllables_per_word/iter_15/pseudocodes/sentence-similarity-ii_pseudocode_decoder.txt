CLASS Solution  
    FUNCTION areSentencesSimilarTwo(sentence1, sentence2, similarPairs)  
        IF the LENGTH OF sentence1 IS NOT EQUAL TO the LENGTH OF sentence2  
            RETURN false  
        END IF  

        SET parent TO an empty mapping from elements to elements  

        FUNCTION find(x)  
            IF the element at key x of parent IS NOT EQUAL TO x  
                SET the element at key x of parent TO find(the element at key x of parent)  
            END IF  
            RETURN the element at key x of parent  
        END FUNCTION  

        FUNCTION union(x, y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX IS NOT EQUAL TO rootY  
                SET the element at key rootY of parent TO rootX  
            END IF  
        END FUNCTION  

        FOR each pair IN similarPairs  
            FOR each word IN pair  
                IF word IS NOT a key in parent  
                    SET the element at key word of parent TO word  
                END IF  
            END FOR  
        END FOR  

        FOR each x AND y IN similarPairs interpreted as pairs of elements  
            CALL union with parameters x and y  
        END FOR  

        FOR each w1 AND w2 IN pairs formed by corresponding elements of sentence1 and sentence2  
            IF w1 IS NOT a key in parent  
                SET the element at key w1 of parent TO w1  
            END IF  
            IF w2 IS NOT a key in parent  
                SET the element at key w2 of parent TO w2  
            END IF  
            IF find(w1) IS NOT EQUAL TO find(w2)  
                RETURN false  
            END IF  
        END FOR  

        RETURN true  
    END FUNCTION  
END CLASS
