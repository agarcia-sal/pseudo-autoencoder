CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET xor_all TO zero  
        FOR each num IN nums  
            SET xor_all TO xor_all XOR num  
        END FOR  
        SET diff_bit TO the BITWISE AND of xor_all AND the BITWISE NEGATION of xor_all  
        SET num1 TO zero  
        SET num2 TO zero  
        FOR each num IN nums  
            IF the BITWISE AND of num AND diff_bit DOES NOT EQUAL zero  
                SET num1 TO num1 XOR num  
            ELSE  
                SET num2 TO num2 XOR num  
            END IF  
        END FOR  
        RETURN list consisting of num1 AND num2  
    END FUNCTION  
END CLASS
