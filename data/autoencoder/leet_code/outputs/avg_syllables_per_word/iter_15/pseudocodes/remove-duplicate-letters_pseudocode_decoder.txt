CLASS Solution  
    FUNCTION removeDuplicateLetters(s)  
        SET last_occurrence TO an EMPTY DICTIONARY  
        FOR each index AND character IN s  
            SET the element associated WITH character IN last_occurrence TO index  
        END FOR  
        
        SET stack TO an EMPTY LIST  
        SET in_stack TO an EMPTY SET  
        
        FOR each index AND character IN s  
            IF character IS IN in_stack THEN  
                CONTINUE TO the next iteration  
            END IF  
            
            WHILE the LENGTH OF stack IS GREATER THAN zero AND character IS LESS THAN the last element at position the LENGTH OF stack MINUS one of stack AND index IS LESS THAN the element associated WITH the last element at position the LENGTH OF stack MINUS one of stack IN last_occurrence DO  
                SET removed_char TO the last element at position the LENGTH OF stack MINUS one of stack  
                REMOVE the last element at position the LENGTH OF stack MINUS one FROM stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            
            APPEND character TO stack  
            ADD character TO in_stack  
        END FOR  
        
        SET result_string TO the concatenation of all elements IN stack IN ORDER  
        RETURN result_string  
    END FUNCTION  
END CLASS
