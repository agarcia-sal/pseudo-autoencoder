CLASS Solution  
    FUNCTION minDistance(word1 STRING, word2 STRING) RETURNS INTEGER  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        SET dp TO a two dimensional list with m PLUS one ROWS and n PLUS one COLUMNS where each element is zero  
        
        FOR i FROM zero TO m  
            SET element at position zero of the element at position i of dp TO i  
        END FOR  
        
        FOR j FROM zero TO n  
            SET element at position j of the element at position zero of dp TO j  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2  
                    SET element at position j of the element at position i of dp TO element at position j MINUS one of the element at position i MINUS one of dp  
                ELSE  
                    SET deletion_cost TO element at position j of the element at position i MINUS one of dp  
                    SET insertion_cost TO element at position j MINUS one of the element at position i of dp  
                    SET replacement_cost TO element at position j MINUS one of the element at position i MINUS one of dp  
                    SET element at position j of the element at position i of dp TO one PLUS the MINIMUM of deletion_cost insertion_cost replacement_cost  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of the element at position m of dp  
    END FUNCTION  
END CLASS
