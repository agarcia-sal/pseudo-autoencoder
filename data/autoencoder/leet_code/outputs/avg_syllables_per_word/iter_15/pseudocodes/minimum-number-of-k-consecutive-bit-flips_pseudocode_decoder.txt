CLASS Solution  
    FUNCTION minKBitFlips(nums LIST OF INTEGER k INTEGER) RETURNS INTEGER  
        SET n TO the LENGTH OF nums  
        SET flip_count TO zero  
        SET flipped TO a LIST OF zeros of SIZE n PLUS one  
        SET flip_state TO zero  
        
        FOR index FROM zero TO n MINUS one  
            SET flip_state TO flip_state EXCLUSIVE OR element at position index of flipped  
            
            IF element at position index of nums EQUALS flip_state  
                IF index PLUS k GREATER THAN n  
                    RETURN negative one  
                END IF  
                SET flip_state TO flip_state EXCLUSIVE OR one  
                SET element at position index PLUS k of flipped TO element at position index PLUS k of flipped EXCLUSIVE OR one  
                INCREMENT flip_count BY one  
            END IF  
        END FOR  
        
        RETURN flip_count  
    END FUNCTION  
END CLASS
