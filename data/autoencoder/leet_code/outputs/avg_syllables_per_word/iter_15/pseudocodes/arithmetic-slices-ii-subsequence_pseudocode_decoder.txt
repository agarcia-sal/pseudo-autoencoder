CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n IS LESS THAN three  
            RETURN zero  
        END IF  
        
        SET dp TO a list containing for each index from zero TO n MINUS one a new default dictionary of integers  
        SET total_count TO zero  
        
        FOR each index i FROM zero TO n MINUS one  
            FOR each index j FROM zero TO i MINUS one  
                SET d TO the element at position i of nums MINUS the element at position j of nums  
                
                IF dp at position j CONTAINS key d  
                    INCREMENT the value at key d in dp at position i BY the value at key d in dp at position j PLUS one  
                    INCREMENT total_count BY the value at key d in dp at position j  
                ELSE  
                    INCREMENT the value at key d in dp at position i BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
