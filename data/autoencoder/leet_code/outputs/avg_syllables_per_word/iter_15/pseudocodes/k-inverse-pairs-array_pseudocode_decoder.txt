CLASS Solution  
    FUNCTION kInversePairs(n, k)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a two dimensional list of zeros with (n PLUS one) rows and (k PLUS one) columns  
        SET element at position zero zero of dp TO one  
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            FOR j FROM zero TO k  
                SET prefix_sum TO prefix_sum PLUS element at position i MINUS one j of dp  
                SET prefix_sum TO prefix_sum MODULO MOD  
                IF j GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO prefix_sum MINUS element at position i MINUS one j MINUS i of dp PLUS MOD  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                END IF  
                SET element at position i j of dp TO prefix_sum  
            END FOR  
        END FOR  
        RETURN element at position n k of dp  
    END FUNCTION  
END CLASS
