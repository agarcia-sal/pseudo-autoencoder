CLASS Solution  
    FUNCTION countArrangement WITH PARAMETERS n  
        FUNCTION backtrack WITH PARAMETERS used AND index  
            IF index EQUALS n PLUS one  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR i FROM one TO n  
                IF element at position i of used EQUALS NOT true AND (i MODULO index EQUALS zero OR index MODULO i EQUALS zero)  
                    SET element at position i of used TO true  
                    INCREMENT count BY backtrack WITH PARAMETERS used AND index PLUS one  
                    SET element at position i of used TO false  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET used TO a list of Boolean false repeated n PLUS one times  
        RETURN backtrack WITH PARAMETERS used AND one  
    END FUNCTION  
END CLASS
