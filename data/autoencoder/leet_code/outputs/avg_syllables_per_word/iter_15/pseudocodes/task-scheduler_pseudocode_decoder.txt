CLASS Solution  
    FUNCTION leastInterval(tasks AS list of strings, n AS integer) RETURNS integer  
        SET task_counts TO the frequency count of each element in tasks  
        SET max_heap TO a list formed by NEGATING each count in the values of task_counts  
        APPLY the heapify operation to max_heap to arrange elements as a heap  
        
        SET time TO zero  
        SET queue TO an empty list to hold pairs of cooling completion time and task count  
        
        WHILE max_heap is not empty OR queue is not empty  
            WHILE queue is not empty AND the first element of the first pair in queue EQUALS time  
                REMOVE the first element from queue and DESTRUCTURE it into unused variable and count  
                IF count is LESS THAN zero  
                    APPLY the heappush operation to add count back into max_heap  
                END IF  
            END WHILE  
            
            IF max_heap is not empty  
                REMOVE and RETURN the smallest element from max_heap and SET it to count  
                INCREMENT count BY one  
                IF count is LESS THAN zero  
                    APPEND a pair formed by the value of time PLUS n PLUS one and count TO queue  
                END IF  
            END IF  
            
            INCREMENT time BY one  
        END WHILE  
        
        RETURN time  
    END FUNCTION  
END CLASS
