CLASS Solution  
    FUNCTION arrayNesting(nums)  
        SET visited TO a list of logical false values with length equal to the LENGTH OF nums  
        SET max_length TO zero  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF the element at position index of visited EQUALS logical false  
                SET current_length TO zero  
                SET k TO index  
                WHILE the element at position k of visited EQUALS logical false  
                    SET the element at position k of visited TO logical true  
                    SET k TO the element at position k of nums  
                    INCREMENT current_length BY one  
                END WHILE  
                SET max_length TO maximum value BETWEEN max_length AND current_length  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
