CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters obtained by removing colon from time  
        SET hour AND minute TO integers obtained by splitting time at colon  
        
        FUNCTION generate_times(digits)  
            SET times TO empty list  
            FOR each first_hour_digit IN digits  
                FOR each second_hour_digit IN digits  
                    FOR each first_minute_digit IN digits  
                        FOR each second_minute_digit IN digits  
                            SET hour TO the integer formed by concatenating first_hour_digit and second_hour_digit as strings  
                            SET minute TO the integer formed by concatenating first_minute_digit and second_minute_digit as strings  
                            IF hour IS LESS THAN twenty four AND minute IS LESS THAN sixty  
                                APPEND the pair of hour and minute TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the sorted list times  
        END FUNCTION  
        
        SET all_times TO the result of calling generate_times with digits  
        SET current_time_index TO the position of the pair consisting of hour and minute in all_times  
        
        IF current_time_index PLUS one IS LESS THAN the length of all_times  
            SET next_hour AND next_minute TO the pair at position current_time_index PLUS one in all_times  
        ELSE  
            SET next_hour AND next_minute TO the first pair in all_times  
        END IF  
        
        RETURN the string formed by formatting next_hour and next_minute as two digits separated by colon  
    END FUNCTION  
END CLASS
