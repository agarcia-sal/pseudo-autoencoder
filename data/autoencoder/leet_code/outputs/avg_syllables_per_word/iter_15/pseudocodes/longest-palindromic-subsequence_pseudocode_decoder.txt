CLASS Solution  
    FUNCTION longestPalindromeSubseq(s)  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional array of size n BY n initialized WITH zero  
        
        FOR each i FROM zero TO n MINUS one  
            SET element at position i at position i of dp TO one  
        END FOR  
        
        FOR each length FROM two TO n  
            FOR each i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF character at position i of s EQUALS character at position j of s  
                    SET element at position i at position j of dp TO element at position i PLUS one at position j MINUS one of dp PLUS two  
                ELSE  
                    SET element at position i at position j of dp TO the MAXIMUM OF element at position i PLUS one at position j of dp AND element at position i at position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero at position n MINUS one of dp  
    END FUNCTION  
END CLASS
