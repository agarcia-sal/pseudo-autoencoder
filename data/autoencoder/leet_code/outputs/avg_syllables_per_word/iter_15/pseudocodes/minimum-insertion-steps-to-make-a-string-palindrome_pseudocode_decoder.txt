CLASS Solution  
    FUNCTION minInsertions(s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO the LENGTH OF s  
            SET dp TO a two dimensional array of size n BY n with all values zero  
            FOR index i FROM zero TO n MINUS one  
                SET element at position i and i of dp TO one  
            END FOR  
            FOR length_of_substring cl FROM two TO n  
                FOR start_index i FROM zero TO n MINUS cl  
                    SET end_index j TO i PLUS cl MINUS one  
                    IF character at position i of s EQUALS character at position j of s AND length_of_substring EQUALS two  
                        SET element at position i and j of dp TO two  
                    ELSE IF character at position i of s EQUALS character at position j of s  
                        SET element at position i and j of dp TO element at position i PLUS one and j MINUS one of dp PLUS two  
                    ELSE  
                        SET element at position i and j of dp TO the GREATER OF element at position i and j MINUS one of dp AND element at position i PLUS one and j of dp  
                    END IF  
                END FOR  
            END FOR  
            RETURN element at position zero and n MINUS one of dp  
        END FUNCTION  
        RETURN the LENGTH OF s MINUS longest_palindromic_subsequence(s)  
    END FUNCTION  
END CLASS
