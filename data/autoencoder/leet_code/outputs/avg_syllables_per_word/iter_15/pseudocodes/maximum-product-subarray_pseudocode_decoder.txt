CLASS Solution  
    FUNCTION maxProduct(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET max_product TO the element at position zero of nums  
        SET min_product TO the element at position zero of nums  
        SET result TO the element at position zero of nums  
        
        FOR each num IN the elements of nums starting from position one TO the end  
            IF num LESS THAN zero THEN  
                SET temporary TO max_product  
                SET max_product TO min_product  
                SET min_product TO temporary  
            END IF  
            
            IF num GREATER THAN max_product MULTIPLIED BY num THEN  
                SET max_product TO num  
            ELSE  
                SET max_product TO max_product MULTIPLIED BY num  
            END IF  
            
            IF num LESS THAN min_product MULTIPLIED BY num THEN  
                SET min_product TO num  
            ELSE  
                SET min_product TO min_product MULTIPLIED BY num  
            END IF  
            
            IF result LESS THAN max_product THEN  
                SET result TO max_product  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
