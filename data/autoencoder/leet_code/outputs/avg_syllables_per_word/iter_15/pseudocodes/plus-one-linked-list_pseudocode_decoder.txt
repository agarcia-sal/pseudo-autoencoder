CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET instance variable val TO val  
        SET instance variable next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO NEW ListNode CREATED WITH the first element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values STARTING FROM position one TO the end  
        SET node TO NEW ListNode CREATED WITH val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN whether the value variable of p1 EQUALS the value variable of p2 AND the result of is_same_list WITH THE next element of p1 AND THE next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne WITH PARAMETER head AS Optional ListNode  
        FUNCTION reverseList WITH PARAMETER node AS ListNode  
            SET prev TO None  
            WHILE node IS NOT None  
                SET next_node TO the next element of node  
                SET the next element of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO the result of reverseList WITH head  
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current IS NOT None AND carry EQUALS one  
            SET the value variable of current TO the value variable of current PLUS carry  
            SET carry TO the value variable of current DIVIDED BY ten  
            SET the value variable of current TO the value variable of current MODULO ten  
            
            IF the next element of current IS None AND carry EQUALS one  
                SET the next element of current TO NEW ListNode CREATED WITH carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO the next element of current  
        END WHILE  
        
        RETURN the result of reverseList WITH reversed_head  
    END FUNCTION  
END CLASS
