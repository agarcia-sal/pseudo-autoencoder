CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero COMMA left EQUALS none COMMA right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH THE element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE first element FROM queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO A NEW TreeNode WITH THE element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO A NEW TreeNode WITH THE element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p COMMA q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left COMMA q dot left AND is_same_tree WITH PARAMETERS p dot right COMMA q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues WITH PARAMETERS self COMMA root AS Optional TreeNode RETURNS List of integers  
        IF root IS none  
            RETURN an EMPTY list  
        END IF  
        SET queue TO A DOUBLE ENDED QUEUE INITIALIZED WITH root  
        SET largest_values TO AN EMPTY list  
        WHILE queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO NEGATIVE infinity  
            FOR index FROM one TO level_size  
                REMOVE first element FROM queue AND ASSIGN TO node  
                IF max_value IS LESS THAN node dot val  
                    SET max_value TO node dot val  
                END IF  
                IF node dot left IS NOT none  
                    APPEND node dot left TO queue  
                END IF  
                IF node dot right IS NOT none  
                    APPEND node dot right TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
