CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO list of pairs of zero and one, one and zero, zero and MINUS one, MINUS one and zero  
        SET min_heap TO list containing the tuple of zero zero zero representing cost zero position zero zero  
        SET visited TO set containing the tuple of zero zero representing starting position  
        
        WHILE min_heap is not empty  
            REMOVE the tuple of cost position x position y FROM min_heap as the element with smallest cost  
            
            IF the pair of x and y EQUALS the pair of m MINUS one and n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND the pair nx ny IS NOT IN visited  
                    ADD the pair nx ny TO visited  
                    SET new_cost TO cost PLUS the element at position nx of grid at position ny  
                    INSERT the tuple of new_cost nx ny INTO min_heap maintaining heap order  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the integer MINUS one  
    END FUNCTION  
END CLASS
