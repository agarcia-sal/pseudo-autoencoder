CLASS Solution  
    FUNCTION largestRectangleArea(heights)  
        SET stack TO a list containing negative one  
        SET max_area TO zero  
        APPEND zero TO heights  
        FOR each index i AND each height h IN heights  
            WHILE element at the last position of stack NOT EQUALS negative one AND height h LESS THAN element at the position at the last element of stack in heights  
                SET height TO element at the position FROM POPPED last element of stack in heights  
                SET width TO index i MINUS element at the last position of stack MINUS one  
                SET max_area TO the GREATER OF max_area AND height MULTIPLIED BY width  
            END WHILE  
            APPEND index i TO stack  
        END FOR  
        RETURN max_area  
    END FUNCTION  
END CLASS
