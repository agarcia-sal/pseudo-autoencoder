CLASS Solution  
    FUNCTION spiralMatrixIII(rows, cols, rStart, cStart)  
        SET directions TO list containing the tuple of zero and one representing east, the tuple of one and zero representing south, the tuple of zero and negative one representing west, the tuple of negative one and zero representing north  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO empty list  
        SET x TO rStart  
        SET y TO cStart  
        
        WHILE the LENGTH OF visited is LESS THAN rows MULTIPLIED BY cols  
            IF increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR each iteration FROM one TO steps  
                IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN rows AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN cols  
                    APPEND list containing x and y TO visited  
                END IF  
                SET dx TO element at position direction_index of directions and first element of that tuple  
                SET dy TO element at position direction_index of directions and second element of that tuple  
                SET x TO x PLUS dx  
                SET y TO y PLUS dy  
            END FOR  
            SET direction_index TO the REMAINDER when direction_index PLUS one IS DIVIDED BY four  
            INCREMENT increment BY one  
        END WHILE  
        
        RETURN visited  
    END FUNCTION  
END CLASS
