CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            ASSIGN iterator TO a sequential reader over s  
            IF every character of word appears in iterator in order  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SORT dictionary by descending length and then by ascending alphabetical order  
        
        FOR each word IN dictionary  
            IF is_subsequence with parameters word and s returns true  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN empty string  
    END FUNCTION  
END CLASS
