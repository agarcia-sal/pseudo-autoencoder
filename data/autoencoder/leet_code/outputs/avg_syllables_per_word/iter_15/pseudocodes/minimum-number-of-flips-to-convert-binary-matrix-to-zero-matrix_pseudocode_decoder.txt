CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix, x, y)  
            SET directions TO list of pairs consisting of negative one zero positive one zero zero negative one zero and zero one and zero zero  
            FOR each pair dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET element at position nx of matrix and element at position ny TO one MINUS element at position nx of matrix and element at position ny  
                END IF  
            END FOR  
        END FUNCTION  

        FUNCTION matrix_to_tuple(matrix)  
            RETURN a tuple constructed by converting each row in matrix into a tuple  
        END FUNCTION  

        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET target TO tuple constructed by creating tuples of zeros for each position from zero TO n MINUS one for each row from zero TO m MINUS one  
        SET initial_state TO matrix_to_tuple applied to mat  
        SET queue TO list containing a pair of initial_state and zero  
        SET visited TO empty set  
        ADD initial_state TO visited  

        WHILE queue is not empty  
            REMOVE the first element from queue and ASSIGN to current_state and steps  
            IF current_state EQUALS target  
                RETURN steps  
            END IF  

            SET current_matrix TO list constructed by converting each row in current_state to a list  
            
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO a deep copy of current_matrix  
                    CALL flip with next_matrix and i and j  
                    SET next_state TO matrix_to_tuple applied to next_matrix  
                    IF next_state IS NOT IN visited  
                        ADD next_state TO visited  
                        APPEND the pair of next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
