CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        SET tasks TO sorted list of pairs OF corresponding elements FROM plantTime AND growTime ORDERED BY the growTime element IN descending order  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair OF plant AND grow IN tasks  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            SET earliest_bloom TO the maximum OF earliest_bloom AND bloom_day  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
