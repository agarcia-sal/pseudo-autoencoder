CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero next EQUAL TO none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the input parameter values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO a new ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each element val IN values STARTING FROM the element at position one TO the end  
        SET node TO a new ListNode initialized with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the condition that the val element of p1 EQUALS the val element of p2 AND the result of is_same_list called recursively with the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList WITH PARAMETERS self head  
        IF head IS none OR the next element of head IS none  
            RETURN head  
        END IF  
        SET dummy TO a new ListNode initialized with the value minus five thousand and one  
        SET the next element of dummy TO head  
        SET current TO the next element of head  
        SET the next element of head TO none  
        WHILE current IS NOT none  
            SET next_node TO the next element of current  
            SET prev TO dummy  
            WHILE the next element of prev IS NOT none AND the val element of the next element of prev IS LESS THAN the val element of current  
                SET prev TO the next element of prev  
            END WHILE  
            SET the next element of current TO the next element of prev  
            SET the next element of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
