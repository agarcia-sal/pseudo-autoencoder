CLASS Solution  
    FUNCTION kMirror WITH PARAMETERS k AND n  
        FUNCTION is_palindrome WITH PARAMETER s  
            RETURN whether s IS EQUAL TO reverse of s  
        END FUNCTION  
        
        FUNCTION to_base_k WITH PARAMETERS num AND k  
            IF num IS EQUAL TO zero  
                RETURN string representation of zero  
            END IF  
            SET digits TO an empty list  
            WHILE num IS NOT EQUAL TO zero  
                APPEND the remainder of num DIVIDED BY k TO digits  
                SET num TO the integer quotient of num DIVIDED BY k  
            END WHILE  
            RETURN concatenation of string forms of each element in digits in reverse order  
        END FUNCTION  
        
        FUNCTION generate_palindromes WITH PARAMETER length  
            IF length IS EQUAL TO one  
                FOR each i FROM one TO nine  
                    YIELD string representation of i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO the POWER of the integer division of length MINUS one BY two  
                SET end TO ten RAISED TO the POWER of the integer division of length PLUS one BY two  
                FOR each half FROM start TO one LESS THAN end  
                    SET half_str TO string representation of half  
                    IF length MODULO two IS EQUAL TO zero  
                        SET full TO concatenation of half_str AND reverse of half_str  
                    ELSE  
                        SET full TO concatenation of half_str AND reverse of substring of half_str from the second last character TO first character  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO an empty list  
        SET length TO one  
        WHILE the LENGTH OF k_mirror_numbers IS LESS THAN n  
            FOR each pal IN generate_palindromes WITH length  
                SET num TO integer form of pal  
                IF is_palindrome OF to_base_k WITH num AND k RETURNS true  
                    APPEND num TO k_mirror_numbers  
                    IF LENGTH OF k_mirror_numbers IS EQUAL TO n  
                        BREAK out of the for loop  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN sum of all elements in k_mirror_numbers  
    END FUNCTION  
END CLASS
