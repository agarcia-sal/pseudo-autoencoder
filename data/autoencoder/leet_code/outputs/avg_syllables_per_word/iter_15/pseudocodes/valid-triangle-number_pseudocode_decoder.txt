CLASS Solution  
    FUNCTION triangleNumber(nums)  
        CALL sort ON nums  
        SET count TO zero  
        SET n TO the LENGTH OF nums  
          
        FOR k FROM n MINUS one DOWN TO two BY negative one  
            SET i TO zero  
            SET j TO k MINUS one  
            WHILE i IS LESS THAN j  
                IF element at position i of nums PLUS element at position j of nums IS GREATER THAN element at position k of nums  
                    INCREMENT count BY j MINUS i  
                    DECREMENT j BY one  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
        END FOR  
          
        RETURN count  
    END FUNCTION  
END CLASS
