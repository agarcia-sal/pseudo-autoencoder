CLASS Solution  
    FUNCTION findRepeatedDnaSequences with parameter s  
        SET sequence_count TO a new mapping from sequences to counts initialized with zero  
        FOR each index i FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO substring from position i TO i PLUS nine of s  
            INCREMENT the count at sequence in sequence_count BY one  
        END FOR  
        SET repeated_sequences TO a new list of sequences  
        FOR each sequence AND associated count IN the entries of sequence_count  
            IF the count IS GREATER THAN one  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
