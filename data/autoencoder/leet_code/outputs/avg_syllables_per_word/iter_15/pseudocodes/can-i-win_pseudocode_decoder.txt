CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the product of maxChoosableInteger AND the sum of maxChoosableInteger PLUS one DIVIDED BY two IS LESS THAN desiredTotal THEN  
            RETURN False  
        END IF  
        
        IF desiredTotal IS LESS THAN OR EQUAL TO zero THEN  
            RETURN True  
        END IF  
        
        SET memo TO an empty dictionary  
        
        FUNCTION can_win(current_total, used_numbers)  
            IF the pair current_total AND used_numbers IS IN memo THEN  
                RETURN the value in memo at the pair current_total AND used_numbers  
            END IF  
            
            FOR number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers AND the number one LEFT SHIFTED BY number IS EQUALS zero THEN  
                    IF the sum of current_total AND number IS GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win(the sum of current_total AND number, the bitwise OR of used_numbers AND the number one LEFT SHIFTED BY number) THEN  
                        SET memo at the pair current_total AND used_numbers TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET memo at the pair current_total AND used_numbers TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_win(zero, zero)  
    END FUNCTION  
END CLASS
