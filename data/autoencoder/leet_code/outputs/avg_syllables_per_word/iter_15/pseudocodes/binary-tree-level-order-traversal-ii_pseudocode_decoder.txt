CLASS TreeNode  
    FUNCTION __init__ with parameters self val default zero left default none right default none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values as list  
    IF the collection values is empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO a new double-ended queue  
    APPEND root TO queue  
    WHILE the collection queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i is less than the LENGTH OF values AND the element at position i of values is not none  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the LENGTH OF values AND the element at position i of values is not none  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF the value of p is NOT EQUAL TO the value of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of calling is_same_tree with the left child of p and the left child of q AND calling is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom with parameters self root as optional TreeNode  
        IF root is none  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a new double-ended queue initialized with the single element root  
        WHILE the collection queue is not empty  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR each index from zero TO level_size MINUS one  
                SET node TO the element removed from the front of queue  
                APPEND the value of node TO current_level  
                IF the left child of node is not none  
                    APPEND the left child of node TO queue  
                END IF  
                IF the right child of node is not none  
                    APPEND the right child of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN the reverse of the collection result  
    END FUNCTION  
END CLASS
