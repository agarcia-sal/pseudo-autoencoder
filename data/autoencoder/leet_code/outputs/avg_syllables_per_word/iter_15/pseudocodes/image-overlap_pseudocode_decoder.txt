CLASS Solution  
    FUNCTION largestOverlap(img1, img2)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an EMPTY list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of img1 EQUALS one  
                    APPEND the tuple consisting of i and j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an EMPTY list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of img2 EQUALS one  
                    APPEND the tuple consisting of i and j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a new DEFAULT DICTIONARY mapping from keys to integers with default zero  
        FOR each tuple x1 y1 IN points1  
            FOR each tuple x2 y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value at key consisting of dx and dy in overlap_count BY one  
            END FOR  
        END FOR  
        SET maximum_value TO zero  
        FOR each value IN the VALUES OF overlap_count  
            IF value GREATER THAN maximum_value  
                SET maximum_value TO value  
            END IF  
        END FOR  
        RETURN maximum_value  
    END FUNCTION  
END CLASS
