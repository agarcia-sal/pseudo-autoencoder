CLASS Solution  
    FUNCTION combinationSum4(nums AS List OF integer, target AS integer) RETURNS integer  
        SET dp TO a list OF integer WITH length target PLUS one WITH all elements SET TO zero  
        SET element at position zero of dp TO one  
        FOR amount FROM one TO target  
            FOR each num IN nums  
                IF amount GREATER THAN OR EQUAL TO num THEN  
                    INCREMENT element at position amount of dp BY element at position amount MINUS num of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position target of dp  
    END FUNCTION  
END CLASS
