CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulting to zero left defaulting to none right defaulting to none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values list  
    IF values is empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE the element from the left side of queue and ASSIGN TO node  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p is none AND q is none  
        RETURN true  
    ELSE IF p is none OR q is none  
        RETURN false  
    ELSE IF the value of p is NOT EQUAL TO the value of q  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with arguments the left child of p AND the left child of q AND the result of calling is_same_tree with arguments the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder with parameters self and root  
        IF root is none  
            RETURN an empty list  
        END IF  
        SET column_table TO a default dictionary mapping keys to empty lists  
        SET queue TO a double ended queue initialized with a tuple containing root AND zero  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE queue is not empty  
            REMOVE the element from the left side of queue and ASSIGN TO node AND column_index  
            IF node is not none  
                APPEND the value of node TO the list at key column_index in column_table  
                IF min_column is greater than column_index  
                    SET min_column TO column_index  
                END IF  
                IF max_column is less than column_index  
                    SET max_column TO column_index  
                END IF  
                APPEND a tuple containing the left child of node AND column_index MINUS one TO queue  
                APPEND a tuple containing the right child of node AND column_index PLUS one TO queue  
            END IF  
        END WHILE  
        RETURN a list of elements taken from column_table at each integer index FROM min_column TO max_column inclusive  
    END FUNCTION  
END CLASS
