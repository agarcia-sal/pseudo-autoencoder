CLASS TreeNode  
    FUNCTION __init__ with parameters val set to zero, left set to none, right set to none  
        SET the value property TO val  
        SET the left property TO left  
        SET the right property TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values which is a list  
    IF the values list is empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element obtained by removing from the left of queue  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the left property of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not none  
            SET the right property of node TO a new TreeNode initialized with the element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p is none AND q is none  
        RETURN True  
    ELSE IF p is none OR q is none  
        RETURN False  
    ELSE IF the value property of p is not equal to the value property of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left property of p and the left property of q AND the result of is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum with parameter root which is optionally a TreeNode  
        FUNCTION helper with parameter node  
            IF node is none  
                RETURN float negative infinity, float negative infinity  
            END IF  
            SET left_max and left_gain TO the result of calling helper with the left property of node  
            SET right_max and right_gain TO the result of calling helper with the right property of node  
            SET current_max TO the value property of node PLUS the maximum of left_gain and zero PLUS the maximum of right_gain and zero  
            SET current_gain TO the value property of node PLUS the maximum of left_gain, right_gain, and zero  
            SET global_max TO the maximum of left_max, right_max, and current_max  
            RETURN global_max, current_gain  
        END FUNCTION  
        RETURN the first element of the result of calling helper with root  
    END FUNCTION  
END CLASS
