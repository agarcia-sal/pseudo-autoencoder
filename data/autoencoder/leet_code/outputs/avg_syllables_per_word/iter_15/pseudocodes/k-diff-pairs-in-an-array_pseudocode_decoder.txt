CLASS Solution  
    FUNCTION findPairs(nums LIST OF INTEGERS, k INTEGER) RETURNS INTEGER  
        IF k LESS THAN zero THEN  
            RETURN zero  
        END IF  
        SET num_counts TO an empty mapping from integers to integers  
        FOR each num IN nums  
            SET current_count TO value associated with num in num_counts OR zero if num is not present  
            SET value associated with num in num_counts TO current_count PLUS one  
        END FOR  
        SET unique_pairs TO zero  
        FOR each num IN keys of num_counts  
            IF k EQUALS zero THEN  
                IF value associated with num in num_counts GREATER THAN one THEN  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF there EXISTS a key in num_counts EQUALS num PLUS k THEN  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
