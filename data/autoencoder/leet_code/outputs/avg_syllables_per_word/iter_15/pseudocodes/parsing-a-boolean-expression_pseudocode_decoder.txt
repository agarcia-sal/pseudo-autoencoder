CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO an empty list  
        FOR each character IN expression  
            IF character IS IN the collection consisting of the characters t f exclamation mark ampersand vertical bar  
                APPEND character TO stk  
            ELSE IF character EQUALS closing parenthesis  
                SET t TO zero  
                SET f TO zero  
                WHILE the element at the last position of stk IS IN the collection consisting of the characters t and f  
                    INCREMENT t BY one IF the element at the last position of stk EQUALS the character t  
                    INCREMENT f BY one IF the element at the last position of stk EQUALS the character f  
                    REMOVE the element at the last position FROM stk  
                END WHILE  
                SET op TO the element at the last position of stk  
                REMOVE the element at the last position FROM stk  
                IF op EQUALS exclamation mark  
                    SET character TO the character t IF f IS greater than zero OTHERWISE the character f  
                ELSE IF op EQUALS ampersand  
                    SET character TO the character f IF f IS greater than zero OTHERWISE the character t  
                ELSE IF op EQUALS vertical bar  
                    SET character TO the character t IF t IS greater than zero OTHERWISE the character f  
                END IF  
                APPEND character TO stk  
            END IF  
        END FOR  
        RETURN whether the element at the first position of stk EQUALS the character t  
    END FUNCTION  
END CLASS
