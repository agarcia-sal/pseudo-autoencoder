CLASS Solution  
    FUNCTION maximumGood(statements AS List OF List OF Integer) AS Integer  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR each integer i FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            SET good TO a list of boolean FALSE values of LENGTH n  
            SET bad TO a list of boolean FALSE values of LENGTH n  
            FOR each integer j FROM zero TO n MINUS one  
                IF the BITWISE AND of i AND two RAISED TO THE POWER OF j IS NOT EQUAL TO zero  
                    SET the element at position j of good TO TRUE  
                ELSE  
                    SET the element at position j of bad TO TRUE  
                END IF  
            END FOR  
            SET valid TO TRUE  
            FOR each integer j FROM zero TO n MINUS one  
                IF the element at position j of good IS TRUE  
                    FOR each integer k FROM zero TO n MINUS one  
                        IF (the element at position k of the element at position j of statements IS EQUAL TO zero AND the element at position k of good IS TRUE) OR (the element at position k of the element at position j of statements IS EQUAL TO one AND the element at position k of bad IS TRUE)  
                            SET valid TO FALSE  
                            BREAK FROM the inner loop  
                        END IF  
                    END FOR  
                    IF valid IS FALSE  
                        BREAK FROM the outer loop  
                    END IF  
                END IF  
            END FOR  
            IF valid IS TRUE  
                SET max_good TO the GREATER OF max_good AND the SUM OF the elements of good WHERE TRUE COUNTS AS one AND FALSE COUNTS AS zero  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
