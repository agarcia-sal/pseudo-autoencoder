CLASS Solution  
    FUNCTION rangeBitwiseAnd(left INTEGER PARAMETER right INTEGER PARAMETER) RETURNS INTEGER  
        SET shift TO zero  
        WHILE left IS LESS THAN right  
            SET left TO the integer division RESULT of left DIVIDED BY two  
            SET right TO the integer division RESULT of right DIVIDED BY two  
            INCREMENT shift BY one  
        END WHILE  
        RETURN the RESULT of left MULTIPLIED BY two RAISED TO THE POWER OF shift  
    END FUNCTION  
END CLASS
