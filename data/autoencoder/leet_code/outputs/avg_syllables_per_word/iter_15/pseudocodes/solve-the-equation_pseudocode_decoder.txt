CLASS Solution  
    FUNCTION solveEquation WITH PARAMETER equation  
        FUNCTION parse_side WITH PARAMETER side  
            SET tokens TO the result of replacing all occurrences of plus sign in side with delimiter and plus sign then replacing all occurrences of minus sign in side with delimiter and minus sign then splitting by delimiter  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token is empty string  
                    CONTINUE to next iteration  
                END IF  
                IF token contains character x  
                    IF token equals character x OR token equals plus sign followed by character x  
                        INCREMENT x_count BY one  
                    ELSE IF token equals minus sign followed by character x  
                        DECREMENT x_count BY one  
                    ELSE  
                        INCREMENT x_count BY the integer value of token after removing character x  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY the integer value of token  
                END IF  
            END FOR  
            RETURN x_count, num_sum  
        END FUNCTION  

        SET left_side TO the substring of equation before character equals sign  
        SET right_side TO the substring of equation after character equals sign  
        SET left_x, left_num TO the result of calling parse_side with left_side  
        SET right_x, right_num TO the result of calling parse_side with right_side  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN the string Infinite solutions  
            ELSE  
                RETURN the string No solution  
            END IF  
        ELSE  
            RETURN the string consisting of character x followed by equals sign followed by the integer result of total_num DIVIDED BY total_x using integer division  
        END IF  
    END FUNCTION  
END CLASS
