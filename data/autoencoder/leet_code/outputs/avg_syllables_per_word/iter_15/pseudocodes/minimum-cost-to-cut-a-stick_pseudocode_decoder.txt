CLASS Solution  
    FUNCTION minCost WITH PARAMETERS n AND cuts  
        SET cuts TO the RESULT OF sorting the CONCATENATION of cuts WITH zero AND n  
        
        FUNCTION dp WITH PARAMETERS left AND right  
            IF right MINUS left LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO positive infinity  
            FOR i FROM left PLUS one TO right MINUS one  
                SET cost TO element at position right of cuts MINUS element at position left of cuts PLUS dp CALLED WITH left AND i PLUS dp CALLED WITH i AND right  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp CALLED WITH zero AND the LENGTH OF cuts MINUS one  
END CLASS
