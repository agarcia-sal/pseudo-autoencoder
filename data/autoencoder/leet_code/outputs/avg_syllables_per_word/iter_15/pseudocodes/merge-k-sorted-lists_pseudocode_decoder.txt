CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the attribute val TO val  
        SET the attribute next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values from position one TO the end  
        SET node TO new ListNode initialized with val  
        SET the attribute next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the result of the logical AND between the equality of the attribute val of p1 AND the attribute val of p2 AND the result of is_same_list called with the attribute next of p1 AND the attribute next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists WITH PARAMETERS lists AS list of optional ListNode  
        SET min_heap TO empty list  
        FOR each index i AND element lst IN lists with enumeration  
            IF lst IS not none  
                PERFORM the operation to push the tuple containing the attribute val of lst, i, AND lst INTO min_heap using heap push operation  
            END IF  
        END FOR  
        SET dummy TO new ListNode initialized with default parameters  
        SET current TO dummy  
        WHILE min_heap IS not empty  
            PERFORM the operation to pop the smallest element FROM min_heap using heap pop operation AND ASSIGN the unpacked tuple to val, i, AND node  
            SET the attribute next of current TO node  
            SET current TO the attribute next of current  
            IF the attribute next of node IS not none  
                PERFORM the operation to push the tuple containing the attribute val of the attribute next of node, i, AND the attribute next of node INTO min_heap using heap push operation  
            END IF  
        END WHILE  
        RETURN the attribute next of dummy  
    END FUNCTION  
END CLASS
