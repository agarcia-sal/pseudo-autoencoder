CLASS Solution  
    FUNCTION isNStraightHand(hand AS List OF Integer, groupSize AS Integer) RETURNS Boolean  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        SET count TO the frequency count of each element IN hand  
        SET unique_cards TO the sorted list of all unique elements IN count  
        FOR each card IN unique_cards  
            IF the value at card IN count GREATER THAN zero THEN  
                SET needed TO the value at card IN count  
                FOR each index FROM zero TO groupSize MINUS one  
                    IF the value at the sum of card PLUS index IN count LESS THAN needed THEN  
                        RETURN False  
                    END IF  
                    SET the value at the sum of card PLUS index IN count TO the value at the sum of card PLUS index IN count MINUS needed  
                END FOR  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
