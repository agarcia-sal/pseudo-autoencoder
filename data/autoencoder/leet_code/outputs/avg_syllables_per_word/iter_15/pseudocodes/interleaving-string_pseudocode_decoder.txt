CLASS Solution  
    FUNCTION isInterleave WITH PARAMETERS s1 s2 s3  
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 NOT EQUALS the LENGTH OF s3  
            RETURN False  
        END IF  

        INITIALIZE dp AS a list of BOOLEAN False values with LENGTH the LENGTH OF s2 PLUS one  
        SET the element at position zero of dp TO True  

        FOR i FROM zero TO the LENGTH OF s1 INCLUSIVE  
            FOR j FROM zero TO the LENGTH OF s2 INCLUSIVE  
                IF i GREATER THAN zero  
                    SET the element at position j of dp TO the logical AND of the element at position j of dp AND the BOOLEAN value of the equality TEST between the character at position i MINUS one of s1 AND the character at position i PLUS j MINUS one of s3  
                END IF  
                IF j GREATER THAN zero  
                    SET the element at position j of dp TO the logical OR of the element at position j of dp AND the logical AND of the element at position j MINUS one of dp AND the BOOLEAN value of the equality TEST between the character at position j MINUS one of s2 AND the character at position i PLUS j MINUS one of s3  
                END IF  
            END FOR  
        END FOR  

        RETURN the element at position the LENGTH OF s2 of dp  
    END FUNCTION  
END CLASS
