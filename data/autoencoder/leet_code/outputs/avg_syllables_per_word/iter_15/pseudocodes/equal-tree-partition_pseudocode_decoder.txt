CLASS TreeNode  
    FUNCTION __init__ (val EQUALS zero COMMA left EQUALS None COMMA right EQUALS None)  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node (values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO first element removed from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO a new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO a new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree (p COMMA q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF value of p IS NOT EQUALS value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(value of left child of p COMMA value of left child of q) AND is_same_tree(value of right child of p COMMA value of right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree (root)  
        FUNCTION subtree_sum (node)  
            IF node IS None  
                RETURN zero  
            END IF  
            RETURN value of node PLUS subtree_sum(value of left child of node) PLUS subtree_sum(value of right child of node)  
        END FUNCTION  

        SET total_sum TO subtree_sum(root)  

        IF total_sum MODULO two IS NOT EQUALS zero  
            RETURN False  
        END IF  

        SET target TO total_sum DIVIDED BY two  
        SET found TO a list containing False  

        FUNCTION can_split (node)  
            IF node IS None  
                RETURN zero  
            END IF  
            SET current_sum TO value of node PLUS can_split(value of left child of node) PLUS can_split(value of right child of node)  
            IF current_sum EQUALS target AND node IS NOT root  
                SET element at position zero of found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  

        CALL can_split(root)  
        RETURN element at position zero of found  
    END FUNCTION  
END CLASS
