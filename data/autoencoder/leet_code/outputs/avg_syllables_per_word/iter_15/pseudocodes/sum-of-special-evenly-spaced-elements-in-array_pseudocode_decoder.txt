CLASS Solution  
    FUNCTION solve(nums AS list of integers, queries AS list of list of integers) RETURNS list of integers  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF nums  
        
        SET max_y TO the MAXIMUM value of the second element IN each element of queries  
        SET prefix_sums TO a list containing max_y PLUS one elements each initialized to a list of n zeros  
        
        FOR y FROM one TO max_y  
            FOR i FROM n MINUS one DOWN TO zero  
                IF the sum of i PLUS y IS LESS THAN n  
                    SET the element at position i of prefix_sums at position y TO the element at position i of nums PLUS the element at position i PLUS y of prefix_sums at position y  
                ELSE  
                    SET the element at position i of prefix_sums at position y TO the element at position i of nums  
                END IF  
            END FOR  
        END FOR  
        
        SET answer TO an empty list  
        FOR each pair of x AND y IN queries  
            APPEND the remainder of the element at position x of prefix_sums at position y DIVIDED BY MOD TO answer  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
