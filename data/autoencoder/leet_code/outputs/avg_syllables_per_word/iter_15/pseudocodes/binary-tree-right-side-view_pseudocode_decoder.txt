CLASS TreeNode  
    FUNCTION __init__ WITH parameters val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH parameter values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized WITH the element at position zero of values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO a new TreeNode initialized WITH the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO a new TreeNode initialized WITH the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH parameters p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree WITH p dot left AND q dot left AND the result of calling is_same_tree WITH p dot right AND q dot right  
    END ELSE  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView WITH parameters self root  
        IF root IS none  
            RETURN an empty list  
        END IF  
        SET right_side_view TO an empty list  
        SET queue TO a new double ended queue initialized WITH root as the only element  
        WHILE queue is not empty  
            SET level_length TO the LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO the element removed from the front of queue  
                IF i EQUALS level_length MINUS one  
                    APPEND node dot val TO right_side_view  
                END IF  
                IF node dot left IS NOT none  
                    APPEND node dot left TO queue  
                END IF  
                IF node dot right IS NOT none  
                    APPEND node dot right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
