CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s AS string, k AS integer) RETURNS integer  
        IF k EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET char_map TO empty collection mapping characters to integers  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            ASSIGN in char_map the element at position right of s TO right  
            
            IF the NUMBER OF keys in char_map GREATER THAN k THEN  
                SET leftmost_char TO the key with the MINIMUM associated value in char_map  
                SET left TO the value associated with leftmost_char in char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET current_window_length TO right MINUS left PLUS one  
            SET max_length TO the GREATER of max_length AND current_window_length  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
