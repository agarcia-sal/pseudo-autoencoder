CLASS Solution  
    FUNCTION leastOpsExpressTarget(x integer, target integer) RETURNS integer  
        DEFINE FUNCTION dp(target integer) RETURNS integer  
            IF target EQUALS zero THEN  
                RETURN negative one  
            END IF  
            IF target LESS THAN x THEN  
                RETURN the MINIMUM OF the value of two TIMES target MINUS one AND the value of two TIMES x MINUS target  
            END IF  
            
            SET power TO one  
            WHILE x RAISED TO power LESS THAN OR EQUAL TO target HOLDS  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            
            SET res TO the value of dp WITH argument target MINUS x RAISED TO power PLUS power  
            
            IF the value of x RAISED TO the value of power PLUS one MINUS target LESS THAN target THEN  
                SET res TO the MINIMUM OF res AND the value of dp WITH argument x RAISED TO the value of power PLUS one MINUS target PLUS power PLUS one  
            END IF  
            
            RETURN res  
        END FUNCTION  
        
        RETURN dp WITH argument target  
    END FUNCTION  
END CLASS
