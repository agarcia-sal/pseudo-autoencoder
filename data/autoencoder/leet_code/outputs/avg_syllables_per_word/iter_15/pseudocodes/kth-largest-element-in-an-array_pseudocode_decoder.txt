CLASS Solution  
    FUNCTION findKthLargest(nums AS list of integers, k AS integer) RETURNS integer  
        SET min_heap TO an empty collection capable of heap operations  
        FOR each num IN nums  
            INSERT num INTO min_heap USING heap push operation  
            IF the LENGTH OF min_heap IS GREATER THAN k  
                REMOVE the smallest element FROM min_heap USING heap pop operation  
            END IF  
        END FOR  
        RETURN the element at position one of min_heap  
    END FUNCTION  
END CLASS
