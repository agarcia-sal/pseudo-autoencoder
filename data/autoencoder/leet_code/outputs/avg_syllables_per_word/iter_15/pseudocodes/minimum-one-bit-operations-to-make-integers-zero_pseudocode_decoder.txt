CLASS Solution  
    FUNCTION minimumOneBitOperations(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE two RAISED TO THE POWER OF k PLUS one IS LESS THAN OR EQUAL TO n  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO two RAISED TO THE POWER OF k  
        RETURN two RAISED TO THE POWER OF k PLUS one MINUS one MINUS the RESULT OF minimumOneBitOperations OF self WITH PARAMETER THE RESULT OF n EXCLUSIVE OR most_significant_bit  
    END FUNCTION  
END CLASS
