CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        FOR each character IN s  
            IF character IS a digit  
                SET d TO concatenation of d and character  
            ELSE IF character IS an alphabetic letter  
                SET a TO concatenation of a and character  
            ELSE IF character IS equal to the character of opening square bracket  
                APPEND tuple of a and the integer conversion of d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF character IS equal to the character of closing square bracket  
                SET tuple p, n TO the last element removed from stack  
                SET a TO concatenation of p and a repeated n times  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
