CLASS Solution  
    FUNCTION wordsTyping(sentence LIST OF strings, rows INTEGER, cols INTEGER) RETURNS INTEGER  
        SET s TO the concatenation of all elements in sentence with a single space between each FOLLOWED BY a single space  
        SET total_length TO the LENGTH OF s  
        SET position TO zero  
        FOR each unused_element FROM zero TO rows MINUS one  
            INCREMENT position BY cols  
            IF the element at position MODULO total_length of s EQUALS a single space  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the element at position MINUS one MODULO total_length of s NOT EQUALS a single space  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        SET result TO the INTEGER DIVISION of position BY total_length  
        RETURN result  
    END FUNCTION  
END CLASS
