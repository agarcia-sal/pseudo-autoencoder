CLASS Solution  
    FUNCTION largestNumber(cost AS List OF integer, target AS integer) RETURNS string  
        CREATE a dictionary called cost_to_digit that maps each element c of cost to the string representation of the index of c PLUS one  
        SET dp TO a list containing an empty string FOLLOWED BY the string negative one repeated target times  
        FOR t FROM one TO target  
            FOR each cost_value AND digit_string IN cost_to_digit  
                IF t IS GREATER THAN OR EQUAL TO cost_value AND element at position t MINUS cost_value of dp IS NOT EQUAL TO string negative one  
                    SET candidate TO digit_string CONCATENATED WITH element at position t MINUS cost_value of dp  
                    IF element at position t of dp IS EQUAL TO string negative one OR the LENGTH OF candidate IS GREATER THAN the LENGTH OF element at position t of dp OR the LENGTH OF candidate IS EQUAL TO the LENGTH OF element at position t of dp AND candidate IS GREATER THAN element at position t of dp  
                        SET element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        IF element at position target of dp IS NOT EQUAL TO string negative one  
            RETURN element at position target of dp  
        ELSE  
            RETURN string zero  
        END IF  
    END FUNCTION  
END CLASS
