CLASS Solution  
    FUNCTION appealSum with parameter s as string  
        SET last TO list of integer values all set to negative one with the LENGTH OF twenty six  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each position i and character char IN s with indices  
            SET index TO the Unicode code point of character char MINUS the Unicode code point of lowercase letter a  
            SET current_appeal TO current_appeal PLUS the difference of position i MINUS element at position index of list last  
            SET total_appeal TO total_appeal PLUS current_appeal  
            SET element at position index of list last TO position i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
