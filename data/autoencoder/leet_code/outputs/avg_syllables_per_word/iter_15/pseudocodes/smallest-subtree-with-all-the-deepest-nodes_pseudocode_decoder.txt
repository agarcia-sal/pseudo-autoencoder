CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero, left EQUAL TO none, right EQUAL TO none  
        SET the attribute val OF self TO val  
        SET the attribute left OF self TO left  
        SET the attribute right OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the parameter values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO A NEW double-ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO THE FIRST ELEMENT REMOVED FROM THE LEFT OF queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the attribute left OF node TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the attribute left OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the attribute right OF node TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the attribute right OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the attribute val OF p IS NOT EQUAL TO the attribute val OF q  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS the attribute left OF p AND the attribute left OF q  
        AND the RESULT OF is_same_tree WITH PARAMETERS the attribute right OF p AND the attribute right OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION subtreeWithAllDeepest WITH PARAMETERS self, root  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS none  
                RETURN node AND zero  
            END IF  
            SET left_node AND left_depth TO THE RESULT OF dfs WITH THE ATTRIBUTE left OF node  
            SET right_node AND right_depth TO THE RESULT OF dfs WITH THE ATTRIBUTE right OF node  
            IF left_depth IS GREATER THAN right_depth  
                RETURN left_node AND left_depth PLUS one  
            ELSE IF right_depth IS GREATER THAN left_depth  
                RETURN right_node AND right_depth PLUS one  
            ELSE  
                RETURN node AND left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN THE FIRST ELEMENT OF THE RESULT OF dfs WITH PARAMETER root  
    END FUNCTION  
END CLASS
