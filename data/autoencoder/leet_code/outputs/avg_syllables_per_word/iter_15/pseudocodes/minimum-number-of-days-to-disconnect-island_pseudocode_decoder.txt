CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the TUPLE of x AND y IS IN visited OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            ADD the TUPLE of x AND y TO visited  
            FOR each pair dx AND dy IN the LIST containing the TUPLE minus one AND zero AND the TUPLE one AND zero AND the TUPLE zero AND minus one AND the TUPLE zero AND one  
                CALL dfs WITH x PLUS dx AND y PLUS dy AND visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an EMPTY SET  
            SET land_cells TO a LIST containing all TUPLES i AND j FOR each i FROM zero TO the LENGTH OF grid MINUS one AND for each j FROM zero TO the LENGTH OF element at position zero of grid MINUS one WHERE element at position i of grid at position j EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero  
                RETURN True  
            END IF  
            CALL dfs WITH the element at position zero of the first TUPLE of land_cells AND the element at position one of the first TUPLE of land_cells AND visited  
            RETURN the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF is_connected() IS NOT True  
            RETURN zero  
        END IF  

        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    SET element at position i of grid at position j TO zero  
                    IF is_connected() IS NOT True  
                        RETURN one  
                    END IF  
                    SET element at position i of grid at position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
