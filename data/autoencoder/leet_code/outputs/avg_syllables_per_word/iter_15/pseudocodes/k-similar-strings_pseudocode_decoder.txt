CLASS Solution  
    FUNCTION kSimilarity(s1, s2)  
        DEFINE FUNCTION neighbors(s)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND the element at position i of s EQUALS the element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF the element at position j of s EQUALS the element at position i of s2 AND the element at position j of s IS NOT EQUAL TO the element at position j of s2  
                    YIELD the concatenation of substring from position zero TO i MINUS one of s PLUS the element at position j of s PLUS substring from position i PLUS one TO j MINUS one of s PLUS the element at position i of s PLUS substring from position j PLUS one TO the end of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET queue TO a new double ended queue initialized with the tuple of s1 AND zero  
        SET visited TO a new set initialized with s1  
        
        WHILE queue IS NOT EMPTY  
            ASSIGN s AND swaps TO the element popped from the left of queue  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors(s)  
                IF neighbor IS NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND the tuple of neighbor AND swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
