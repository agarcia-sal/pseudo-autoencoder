CLASS Solution  
    FUNCTION removeKdigits WITH PARAMETERS num and k  
        SET stack TO empty list  
        
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND element at position the LENGTH OF stack MINUS one OF stack GREATER THAN digit  
                REMOVE last element FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        
        IF k GREATER THAN zero  
            SET final_stack TO the elements of stack UP TO the LENGTH OF stack MINUS k  
        ELSE  
            SET final_stack TO stack  
        END IF  
        
        SET result TO the joining of all elements in final_stack FOLLOWED BY the removal of all leading zeros  
        
        IF result IS NOT empty  
            RETURN result  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
