CLASS Solution  
    FUNCTION waysToDistribute(n, k)  
        SET MOD TO ten TO THE POWER OF nine PLUS seven  
        INITIALIZE dp AS two dimensional list WITH dimensions n PLUS one BY k PLUS one FILLED WITH zero  
        SET element at position zero zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET product_of_j_and_dp_element TO j MULTIPLIED BY element at position i MINUS one j of dp  
                SET sum_of_product_and_previous TO product_of_j_and_dp_element PLUS element at position i MINUS one j MINUS one of dp  
                SET element at position i j of dp TO remainder of sum_of_product_and_previous DIVIDED BY MOD  
            END FOR  
        END FOR  
        RETURN element at position n k of dp  
    END FUNCTION  
END CLASS
