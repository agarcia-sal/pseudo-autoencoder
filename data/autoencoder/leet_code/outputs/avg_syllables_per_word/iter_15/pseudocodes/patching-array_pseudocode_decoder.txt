CLASS Solution  
    FUNCTION minPatches(nums LIST OF INTEGER, n INTEGER) RETURNS INTEGER  
        SET missing TO one as the smallest number that cannot be formed  
        SET patches TO zero as the count of patches added  
        SET i TO zero as the current index in nums  
        WHILE missing IS LESS THAN OR EQUAL TO n  
            IF i IS LESS THAN the LENGTH OF nums AND element at position i of nums IS LESS THAN OR EQUAL TO missing  
                SET missing TO missing PLUS element at position i of nums  
                INCREMENT i BY one  
            ELSE  
                SET missing TO missing PLUS missing  
                INCREMENT patches BY one  
            END IF  
        END WHILE  
        RETURN patches  
    END FUNCTION  
END CLASS
