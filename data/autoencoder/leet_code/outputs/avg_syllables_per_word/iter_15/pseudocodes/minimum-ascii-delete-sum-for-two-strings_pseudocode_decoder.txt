CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        
        SET dp TO a two dimensional list of zeroes with dimensions m PLUS one BY n PLUS one  
        
        FOR j FROM one TO n  
            SET element at position j of element at position zero of dp TO element at position j MINUS one of element at position zero of dp PLUS the ASCII value of the element at position j MINUS one of s2  
        END FOR  
        
        FOR i FROM one TO m  
            SET element at position zero of element at position i of dp TO element at position zero of element at position i MINUS one of dp PLUS the ASCII value of the element at position i MINUS one of s1  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of s1 EQUALS the element at position j MINUS one of s2  
                    SET element at position j of element at position i of dp TO element at position j MINUS one of element at position i MINUS one of dp  
                ELSE  
                    SET element at position j of element at position i of dp TO the minimum of  
                        element at position j of element at position i MINUS one of dp PLUS the ASCII value of the element at position i MINUS one of s1  
                        AND element at position j MINUS one of element at position i of dp PLUS the ASCII value of the element at position j MINUS one of s2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of element at position m of dp  
    END FUNCTION  
END CLASS
