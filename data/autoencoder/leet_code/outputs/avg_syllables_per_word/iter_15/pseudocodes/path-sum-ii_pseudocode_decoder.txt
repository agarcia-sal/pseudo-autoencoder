CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self point dot val TO val  
        SET self point dot left TO left  
        SET self point dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the collection values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO instantiate TreeNode with the element at position zero of the collection values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT EMPTY  
        REMOVE the element from the front of the queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node point dot left TO instantiate TreeNode with the element at position i of values  
            APPEND node point dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node point dot right TO instantiate TreeNode with the element at position i of values  
            APPEND node point dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value of p IS NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN the result OF is_same_tree with the left child of p AND the left child of q AND the result OF is_same_tree with the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION dfs(node, current_sum, path)  
            IF node IS none  
                RETURN  
            END IF  
            APPEND the value of node TO path  
            SET current_sum TO current_sum PLUS the value of node  
            IF the left child of node IS none AND the right child of node IS none AND current_sum EQUALS targetSum  
                APPEND a copy of path TO result  
            END IF  
            CALL dfs with the left child of node, current_sum, AND path  
            CALL dfs with the right child of node, current_sum, AND path  
            REMOVE the last element FROM path  
        END FUNCTION  
        SET result TO an empty collection  
        CALL dfs with root, zero, AND an empty collection  
        RETURN result  
    END FUNCTION  
END CLASS
