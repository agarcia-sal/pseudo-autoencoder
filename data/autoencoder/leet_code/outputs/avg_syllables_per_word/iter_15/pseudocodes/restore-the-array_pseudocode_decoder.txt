CLASS Solution  
    FUNCTION numberOfArrays(s parameter s AS string, k parameter k AS integer) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        INITIALIZE dp AS a list of zeros with the LENGTH OF n PLUS one  
        SET element at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO i MINUS one  
                IF element at position j of s NOT EQUALS zero AND the integer value represented by the substring from position j PLUS one TO position i of s IS GREATER THAN OR EQUAL TO one AND IS LESS THAN OR EQUAL TO k THEN  
                    SET element at position i of dp TO the sum of element at position i of dp PLUS element at position j of dp TAKEN MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
