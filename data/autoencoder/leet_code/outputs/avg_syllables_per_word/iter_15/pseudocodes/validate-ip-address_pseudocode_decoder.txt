CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block consists entirely of decimal digits THEN  
                SET num TO the integer value of block  
                IF num LESS THAN OR EQUAL TO two hundred fifty five AND the string representation of num EQUALS block THEN  
                    RETURN True  
                ELSE  
                    RETURN False  
                END IF  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF the length of block IS GREATER THAN OR EQUAL TO one AND IS LESS THAN OR EQUAL TO four THEN  
                TRY  
                    CONVERT block TO an integer assuming base sixteen  
                    RETURN True  
                CATCH any conversion failure  
                    RETURN False  
                END TRY  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO the list of substrings obtained by splitting queryIP by the period character  
        SET parts_ipv6 TO the list of substrings obtained by splitting queryIP by the colon character  

        IF the number of elements in parts_ipv4 EQUALS four AND every element in parts_ipv4 SATISFIES is_valid_ipv4_block THEN  
            RETURN the string identifier IPv4  
        ELSE IF the number of elements in parts_ipv6 EQUALS eight AND every element in parts_ipv6 SATISFIES is_valid_ipv6_block THEN  
            RETURN the string identifier IPv6  
        ELSE  
            RETURN the string identifier Neither  
        END IF  
    END FUNCTION  
END CLASS
