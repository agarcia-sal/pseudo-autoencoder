CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO the list obtained by sorting each pair of the first element of element at position index of intervals AND the index in intervals  
        SET result TO empty list  
        FOR each element at any position AND variable end IN intervals  
            SET idx TO the position found by binary searching left for a pair starting with element end in start_times  
            IF idx is less than the length of start_times  
                APPEND the element at position one of element at position idx of start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
