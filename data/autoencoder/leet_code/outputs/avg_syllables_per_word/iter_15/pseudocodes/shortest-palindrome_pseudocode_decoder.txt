CLASS Solution  
    FUNCTION shortestPalindrome WITH parameter s  
        IF s EQUALS empty string  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome WITH parameter check_str  
            IF check_str EQUALS reverse of check_str  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        FOR index FROM the LENGTH OF s DOWN TO zero  
            IF is_palindrome WITH argument substring from the start TO position index of s RETURNS true  
                SET to_add TO reverse of substring from position index TO the end of s  
                RETURN concatenation of to_add FOLLOWED BY s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
