CLASS Solution  
    FUNCTION myPow(x, n)  
        FUNCTION qpow(a, n)  
            SET ans TO one  
            WHILE n IS NOT EQUAL TO zero  
                IF the remainder when n is divided by two EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO the integer division of n BY two  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n IS GREATER THAN OR EQUAL TO zero  
            RETURN qpow WITH PARAMETERS x AND n  
        ELSE  
            RETURN one DIVIDED BY qpow WITH PARAMETERS x AND the negation of n  
        END IF  
    END FUNCTION  
END CLASS
