CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            ASSIGN it TO an iterator over t  
            RETURN whether for every character c in s the character c appears in order within t using it  
        END FUNCTION  
        
        SORT strs by the length of each element in descending order followed by lexicographical order  
        
        FOR index i FROM zero TO the LENGTH OF strs MINUS one  
            ASSIGN word TO the element at position i of strs  
            SET unique TO true  
            FOR index j FROM zero TO the LENGTH OF strs MINUS one  
                ASSIGN other TO the element at position j of strs  
                IF i DOES NOT EQUAL j AND the function is_subsequence with s set to word and t set to other RETURNS true  
                    SET unique TO false  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF unique IS true  
                RETURN the LENGTH OF word  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
