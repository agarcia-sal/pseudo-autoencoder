CLASS Solution  
    FUNCTION maxValueOfCoins WITH PARAMETERS piles AND k  
        SET n TO the LENGTH OF piles  
        SET dp TO a LIST of LISTS of zeros with n PLUS one ROWS and k PLUS one COLUMNS representing maximum values for first i piles and exactly j coins chosen  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one of piles  
                SET current_value TO zero  
                FOR l FROM zero TO the MINIMUM OF j AND the LENGTH OF current_pile  
                    IF l IS GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one of current_pile  
                    END IF  
                    SET dp element at row i column j TO the MAXIMUM OF dp element at row i column j AND dp element at row i MINUS one column j MINUS l PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at row n column k  
    END FUNCTION  
END CLASS
