CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self DOT val TO val  
        SET self DOT left TO left  
        SET self DOT right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH value equal TO element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO THE ELEMENT REMOVED FROM THE FRONT OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node DOT left TO A NEW TreeNode WITH value equal TO element at position i of values  
            APPEND node DOT left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node DOT right TO A NEW TreeNode WITH value equal TO element at position i of values  
            APPEND node DOT right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p DOT val IS NOT EQUAL TO q DOT val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH arguments p DOT left AND q DOT left AND is_same_tree WITH arguments p DOT right AND q DOT right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestUnivaluePath(self, root)  
        FUNCTION dfs(node)  
            IF node IS none  
                RETURN zero AND zero  
            END IF  
            SET left_length AND left_value TO CALL dfs WITH argument node DOT left  
            SET right_length AND right_value TO CALL dfs WITH argument node DOT right  
            SET left_arrow TO zero  
            SET right_arrow TO zero  
            IF node DOT left IS NOT none AND node DOT left DOT val EQUALS node DOT val  
                SET left_arrow TO left_length PLUS one  
            END IF  
            IF node DOT right IS NOT none AND node DOT right DOT val EQUALS node DOT val  
                SET right_arrow TO right_length PLUS one  
            END IF  
            SET self DOT max_length TO THE GREATER OF self DOT max_length AND left_arrow PLUS right_arrow  
            RETURN THE GREATER OF left_arrow AND right_arrow AND node DOT val  
        END FUNCTION  
        SET self DOT max_length TO zero  
        CALL dfs WITH argument root  
        RETURN self DOT max_length  
    END FUNCTION  
END CLASS
