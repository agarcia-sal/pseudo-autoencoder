CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO a DICTIONARY mapping each character TO its last position IN the string s  
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO an empty LIST  
        FOR each index i AND character char IN the string s  
            SET end TO the GREATER OF end AND the value at key char IN last_occurrence  
            IF i EQUALS end  
                APPEND the value of end MINUS start PLUS one TO partition_sizes  
                SET start TO i PLUS one  
            END IF  
        END FOR  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
