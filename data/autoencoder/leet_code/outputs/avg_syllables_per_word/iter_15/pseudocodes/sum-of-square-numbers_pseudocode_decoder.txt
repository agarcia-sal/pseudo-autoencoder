CLASS Solution  
    FUNCTION judgeSquareSum(c)  
        SET left TO zero  
        SET right TO the INTEGER PART OF c RAISED TO THE POWER OF zero POINT five  
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET current_sum TO left MULTIPLIED BY left PLUS right MULTIPLIED BY right  
            IF current_sum EQUALS c  
                RETURN True  
            ELSE IF current_sum IS LESS THAN c  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
