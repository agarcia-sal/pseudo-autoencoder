CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET fractions TO the result of replacing every minus sign in expression WITH a plus sign followed by a minus sign  
        SET fractions TO the result of splitting fractions BY plus sign  
        SET result TO the sum of Fraction created from each fraction in fractions WHERE fraction is not empty  
        RETURN the string formed by concatenating the numerator of result followed by slash followed by the denominator of result  
    END FUNCTION  
END CLASS
