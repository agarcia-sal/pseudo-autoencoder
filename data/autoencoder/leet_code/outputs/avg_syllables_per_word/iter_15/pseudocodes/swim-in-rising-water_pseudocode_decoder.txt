        ADD the tuple of the number zero and the number zero TO visited  
        SET directions TO a list containing the tuples of the number zero and the number one the number one and the number zero the number zero and the NEGATIVE number one and the NEGATIVE number one and the number zero  
        WHILE min_heap is not empty  
            REMOVE the smallest element from min_heap and ASSIGN it TO the variables t i and j respectively  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            FOR each di dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj

                    ADD the tuple of ni and nj TO visited  
                    INSERT the tuple of the greater value BETWEEN t AND the element at position ni nj of grid the number ni and the number nj INTO min_heap maintaining the heap property  
                END IF  
            END FOR  
        END WHILE
