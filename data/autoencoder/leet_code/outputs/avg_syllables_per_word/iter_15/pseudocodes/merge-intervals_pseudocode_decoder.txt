CLASS Solution  
    FUNCTION merge(intervals)  
        SORT intervals BY the element at position zero of each element  
        SET merged TO an empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR the element at position one of the last element of merged IS LESS THAN the element at position zero of interval  
                APPEND interval TO merged  
            ELSE  
                SET the element at position one of the last element of merged TO the GREATER OF the element at position one of the last element of merged AND the element at position one of interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
