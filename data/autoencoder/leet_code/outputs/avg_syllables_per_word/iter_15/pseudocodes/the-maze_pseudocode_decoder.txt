CLASS Solution  
    FUNCTION hasPath WITH PARAMETERS maze, start, destination  
        SET directions TO a list containing the pairs zero and one zero and one zero and negative one and negative one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET visited TO an empty set  
        
        FUNCTION dfs WITH PARAMETERS x, y  
            IF the pair consisting of x and y IS IN visited  
                RETURN false  
            END IF  
            IF the list consisting of x and y EQUALS destination  
                RETURN true  
            END IF  
            ADD the pair consisting of x and y TO visited  
            
            FOR each pair dx, dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE the sum of nx and dx IS GREATER THAN OR EQUAL TO zero AND the sum of nx and dx IS LESS THAN m AND the sum of ny and dy IS GREATER THAN OR EQUAL TO zero AND the sum of ny and dy IS LESS THAN n AND the element at position sum of nx and dx of maze AND element at position sum of ny and dy OF that element EQUALS zero  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF the result of dfs WITH PARAMETERS nx AND ny IS true  
                    RETURN true  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN the result of dfs WITH PARAMETERS element at position zero of start AND element at position one of start  
    END FUNCTION  
END CLASS
