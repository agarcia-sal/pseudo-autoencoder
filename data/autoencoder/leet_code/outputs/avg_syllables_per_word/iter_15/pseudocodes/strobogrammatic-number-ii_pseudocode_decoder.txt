CLASS Solution  
    FUNCTION findStrobogrammatic WITH parameter n  
        FUNCTION dfs WITH parameter u  
            IF u EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF u EQUALS one  
                RETURN list containing string zero and string one and string eight  
            END IF  
            SET ans TO empty list  
            FOR each element v IN the result of calling dfs WITH argument u MINUS two  
                FOR each pair l and r IN the sequence of string one string one string eight string eight string six nine string nine six  
                    APPEND the concatenation of l PLUS v PLUS r TO ans  
                END FOR  
                IF u IS NOT EQUAL TO n  
                    APPEND the concatenation of string zero PLUS v PLUS string zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN dfs WITH argument n  
    END FUNCTION  
END CLASS
