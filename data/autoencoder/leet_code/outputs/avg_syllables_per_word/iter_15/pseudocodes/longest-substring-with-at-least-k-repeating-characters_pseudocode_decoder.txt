CLASS Solution  
    FUNCTION longestSubstring(s k)  
        IF the LENGTH OF s LESS THAN k  
            RETURN zero  
        END IF  
        
        SET count TO the frequency mapping of characters in s  
        
        FOR each character IN count  
            IF the frequency of character LESS THAN k  
                RETURN the MAXIMUM value among  
                    THE RESULT OF CALLING longestSubstring WITH each substring OBTAINED BY SPLITTING s BY character AND k  
            END IF  
        END FOR  
        
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
