CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the value property OF self TO val  
        SET the next property OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode CREATED WITH THE element at position zero OF values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO THE END  
        SET node TO A NEW ListNode CREATED WITH val  
        SET the next property OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the condition that the value property OF p1 EQUALS the value property OF p2 AND the recursive call to is_same_list WITH the next property OF p1 AND the next property OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers WITH PARAMETERS l1 AND l2  
        FUNCTION pushIntoStack WITH PARAMETER node  
            SET stack TO an empty list  
            WHILE node IS NOT none  
                APPEND the value property OF node TO stack  
                SET node TO the next property OF node  
            END WHILE  
            RETURN stack  
        END FUNCTION  
        
        SET stack1 TO the result of pushIntoStack WITH l1  
        SET stack2 TO the result of pushIntoStack WITH l2  
        
        SET carry TO zero  
        SET result TO none  
        
        WHILE the condition that stack1 is not empty OR stack2 is not empty OR carry IS NOT zero  
            IF stack1 IS NOT empty  
                SET val1 TO the last element REMOVED FROM stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            
            IF stack2 IS NOT empty  
                SET val2 TO the last element REMOVED FROM stack2  
            ELSE  
                SET val2 TO zero  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO the integer division of total BY ten  
            SET digit TO the remainder of total DIVIDED BY ten  
            
            SET new_node TO A NEW ListNode CREATED WITH digit  
            SET the next property OF new_node TO result  
            SET result TO new_node  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
