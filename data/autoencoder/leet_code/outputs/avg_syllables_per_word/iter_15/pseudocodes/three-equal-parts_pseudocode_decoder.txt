CLASS Solution  
    FUNCTION threeEqualParts(arr)  
        SET total_ones TO the SUM of all elements IN arr  
        IF total_ones EQUALS zero  
            RETURN list containing zero and two  
        END IF  
        IF the MODULO of total_ones BY three NOT EQUALS zero  
            RETURN list containing negative one and negative one  
        END IF  
        SET part_ones TO the INTEGER DIVISION of total_ones BY three  
        SET first_part_index TO zero  
        SET count_ones TO zero  
        FOR index FROM zero TO the LENGTH OF arr MINUS one  
            IF the element at position index of arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS one  
                    SET first_part_index TO index  
                END IF  
                IF count_ones EQUALS part_ones  
                    BREAK out of the loop  
                END IF  
            END IF  
        END FOR  
        SET second_part_index TO zero  
        SET count_ones TO zero  
        FOR index FROM zero TO the LENGTH OF arr MINUS one  
            IF the element at position index of arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS part_ones PLUS one  
                    SET second_part_index TO index  
                    BREAK out of the loop  
                END IF  
            END IF  
        END FOR  
        SET third_part_index TO zero  
        SET count_ones TO zero  
        FOR index FROM zero TO the LENGTH OF arr MINUS one  
            IF the element at position index of arr EQUALS one  
                INCREMENT count_ones BY one  
                IF count_ones EQUALS two MULTIPLIED BY part_ones PLUS one  
                    SET third_part_index TO index  
                    BREAK out of the loop  
                END IF  
            END IF  
        END FOR  
        SET n TO the LENGTH OF arr MINUS third_part_index  
        IF first_part_index PLUS n LESS THAN OR EQUAL TO the LENGTH OF arr AND second_part_index PLUS n LESS THAN OR EQUAL TO the LENGTH OF arr AND the SUBLIST of arr from position first_part_index TO first_part_index PLUS n MINUS one EQUALS the SUBLIST of arr from position second_part_index TO second_part_index PLUS n MINUS one AND EQUALS the SUBLIST of arr from position third_part_index TO the END OF arr  
            RETURN list containing first_part_index PLUS n MINUS one AND second_part_index PLUS n  
        END IF  
        RETURN list containing negative one AND negative one  
    END FUNCTION  
END CLASS
