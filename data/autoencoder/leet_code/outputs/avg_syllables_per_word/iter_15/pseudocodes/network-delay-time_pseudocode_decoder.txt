CLASS Solution  
    FUNCTION networkDelayTime(times AS list of list of integer, n AS integer, k AS integer) RETURNS integer  
        SET graph TO a new empty dictionary mapping integers to lists of pairs of integers  
        FOR i FROM one TO n  
            SET element at key i of graph TO an empty list  
        END FOR  
        FOR each triple u v w IN times  
            APPEND the pair of v and w TO the list at key u of graph  
        END FOR  
        
        SET dist TO a new empty dictionary mapping integers to floating point numbers  
        FOR i FROM one TO n  
            SET element at key i of dist TO positive infinity  
        END FOR  
        SET element at key k of dist TO zero  
        
        SET min_heap TO a new list containing the pair of zero and k  
        
        WHILE the LENGTH OF min_heap is greater than zero  
            POP the smallest element from min_heap into current_dist and node  
            
            IF current_dist GREATER THAN element at key node of dist  
                CONTINUE to next iteration of the while loop  
            END IF  
            
            FOR each pair neighbor weight IN the list at key node of graph  
                SET distance TO current_dist PLUS weight  
                
                IF distance LESS THAN element at key neighbor of dist  
                    SET element at key neighbor of dist TO distance  
                    PUSH the pair of distance and neighbor INTO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        SET max_time TO the maximum value among all values of dist  
        
        IF max_time EQUALS positive infinity  
            RETURN negative one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
