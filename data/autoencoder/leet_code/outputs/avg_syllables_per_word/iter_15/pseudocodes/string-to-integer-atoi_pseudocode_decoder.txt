CLASS Solution  
    FUNCTION myAtoi(s)  
        SET INT_MAX TO two RAISED TO THE POWER OF thirty one MINUS one  
        SET INT_MIN TO NEGATIVE one MULTIPLIED BY two RAISED TO THE POWER OF thirty one  
        SET result TO zero  
        SET index TO zero  
        SET n TO the LENGTH OF s  
        SET sign TO one  
        WHILE index IS LESS THAN n AND the element at position index of s EQUALS a space character  
            INCREMENT index BY one  
        END WHILE  
        IF index IS LESS THAN n AND the element at position index of s EQUALS a plus SIGN OR the element at position index of s EQUALS a minus SIGN  
            IF the element at position index of s EQUALS a minus SIGN  
                SET sign TO NEGATIVE one  
            ELSE  
                SET sign TO one  
            END IF  
            INCREMENT index BY one  
        END IF  
        WHILE index IS LESS THAN n AND the element at position index of s IS A DIGIT  
            SET digit TO the numeric VALUE OF the element at position index of s  
            IF result IS GREATER THAN INT_MAX MINUS digit DIVIDED BY ten  
                IF sign EQUALS one  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
                END IF  
            END IF  
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        END WHILE  
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
