CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND left EQUALS none AND right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode CREATED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element REMOVED FROM the FRONT OF queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO a new instance of TreeNode CREATED WITH the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO a new instance of TreeNode CREATED WITH the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value of p IS NOT EQUALS TO the value of q  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH the left child of p AND the left child of q AND the RESULT OF is_same_tree WITH the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree WITH PARAMETER root  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS none  
                RETURN positive infinity number REPRESENTING minimum VALUE AND negative infinity number REPRESENTING maximum VALUE AND zero REPRESENTING SIZE  
            END IF  
            SET left_min TO the first RETURN VALUE OF dfs CALLED WITH the left child of node  
            SET left_max TO the second RETURN VALUE OF dfs CALLED WITH the left child of node  
            SET left_size TO the third RETURN VALUE OF dfs CALLED WITH the left child of node  
            SET right_min TO the first RETURN VALUE OF dfs CALLED WITH the right child of node  
            SET right_max TO the second RETURN VALUE OF dfs CALLED WITH the right child of node  
            SET right_size TO the third RETURN VALUE OF dfs CALLED WITH the right child of node  
            IF left_max IS LESS THAN the value of node AND the value of node IS LESS THAN right_min  
                RETURN the minimum VALUE BETWEEN the value of node AND left_min AND the maximum VALUE BETWEEN the value of node AND right_max AND the sum of left_size PLUS right_size PLUS one  
            END IF  
            RETURN negative infinity number REPRESENTING minimum VALUE AND positive infinity number REPRESENTING maximum VALUE AND the maximum VALUE BETWEEN left_size AND right_size  
        END FUNCTION  
        RETURN the third RETURN VALUE OF dfs CALLED WITH root  
    END FUNCTION  
END CLASS
