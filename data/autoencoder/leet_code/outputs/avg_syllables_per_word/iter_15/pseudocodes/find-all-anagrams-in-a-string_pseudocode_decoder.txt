CLASS Solution  
    FUNCTION findAnagrams WITH PARAMETERS s AND p  
        SET len_s TO the LENGTH OF s  
        SET len_p TO the LENGTH OF p  
  
        IF len_p IS GREATER THAN len_s  
            RETURN an EMPTY LIST  
        END IF  
  
        SET p_count TO the COUNTER OF characters in p  
        SET s_count TO the COUNTER OF characters in the substring from position zero TO position len_p MINUS one of s  
  
        SET anagrams TO an EMPTY LIST  
  
        IF s_count EQUALS p_count  
            APPEND zero TO anagrams  
        END IF  
  
        FOR i FROM len_p TO len_s MINUS one  
            INCREMENT the COUNT of the character at position i of s IN s_count BY one  
            DECREMENT the COUNT of the character at position i MINUS len_p of s IN s_count BY one  
  
            IF the COUNT of the character at position i MINUS len_p of s IN s_count EQUALS zero  
                REMOVE the character at position i MINUS len_p of s FROM s_count  
            END IF  
  
            IF s_count EQUALS p_count  
                APPEND i MINUS len_p PLUS one TO anagrams  
            END IF  
        END FOR  
  
        RETURN anagrams  
    END FUNCTION  
END CLASS
