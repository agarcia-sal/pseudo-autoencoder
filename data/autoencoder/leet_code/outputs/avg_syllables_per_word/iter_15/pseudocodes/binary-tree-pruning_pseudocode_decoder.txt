CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero left EQUAL TO none right EQUAL TO none  
        SET the val ATTRIBUTE OF self TO val  
        SET the left ATTRIBUTE OF self TO left  
        SET the right ATTRIBUTE OF self TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW INSTANCE OF TreeNode WITH parameter element at position zero of values  
    SET i TO one  
    SET queue TO A NEW DEQUE COLLECTION  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO THE ELEMENT REMOVED FROM THE FRONT OF queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET the left ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode WITH parameter element at position i of values  
            APPEND the left ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET the right ATTRIBUTE OF node TO A NEW INSTANCE OF TreeNode WITH parameter element at position i of values  
            APPEND the right ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUALS TO the val ATTRIBUTE OF q  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS the left ATTRIBUTE OF p AND the left ATTRIBUTE OF q AND the RESULT OF is_same_tree WITH PARAMETERS the right ATTRIBUTE OF p AND the right ATTRIBUTE OF q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION pruneTree WITH PARAMETERS self root  
        IF root IS none  
            RETURN none  
        END IF  
        SET the left ATTRIBUTE OF root TO the RESULT OF self calling pruneTree WITH parameter the left ATTRIBUTE OF root  
        SET the right ATTRIBUTE OF root TO the RESULT OF self calling pruneTree WITH parameter the right ATTRIBUTE OF root  
        IF the val ATTRIBUTE OF root IS EQUALS TO zero AND the left ATTRIBUTE OF root IS none AND the right ATTRIBUTE OF root IS none  
            RETURN none  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
