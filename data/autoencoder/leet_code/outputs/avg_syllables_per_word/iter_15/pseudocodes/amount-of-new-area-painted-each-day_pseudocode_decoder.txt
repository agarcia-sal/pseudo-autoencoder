CLASS Solution  
    FUNCTION amountPainted WITH PARAMETER paint WHICH IS A LIST OF LIST OF INTEGER  
        SET max_end TO the maximum of each end element from each pair IN paint  
        SET painted TO a list of length max_end PLUS one where each element IS set TO negative one  
        SET worklog TO an empty list  
        
        FOR each pair of start and end IN paint  
            SET current_start TO start  
            SET new_paint TO zero  
            
            WHILE current_start IS LESS THAN end  
                IF the element at position current_start of painted EQUALS negative one  
                    SET the element at position current_start of painted TO end  
                    INCREMENT new_paint BY one  
                    INCREMENT current_start BY one  
                ELSE  
                    SET current_start TO the element at position current_start of painted  
                END IF  
            END WHILE  
            
            APPEND new_paint TO worklog  
        END FOR  
        
        RETURN worklog  
    END FUNCTION  
END CLASS
