CLASS Solution  
    FUNCTION grayCode WITH parameter n  
        IF n EQUALS zero  
            RETURN list containing zero  
        END IF  
        SET prev_gray TO grayCode called with n MINUS one  
        SET mask TO one LEFT SHIFTED BY n MINUS one  
        SET reversed_prev_gray TO reversed sequence of prev_gray  
        SET extended_gray TO empty list  
        FOR each num IN reversed_prev_gray  
            SET new_num TO mask OR num  
            APPEND new_num TO extended_gray  
        END FOR  
        SET current_gray TO concatenation of prev_gray FOLLOWED BY extended_gray  
        RETURN current_gray  
    END FUNCTION  
END CLASS
