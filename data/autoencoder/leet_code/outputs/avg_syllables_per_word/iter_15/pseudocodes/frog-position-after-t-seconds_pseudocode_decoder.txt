CLASS Solution  
    FUNCTION frogPosition WITH PARAMETERS n AS integer edges AS list of list of integer t AS integer target AS integer RETURNS float  
        IF n EQUALS one  
            RETURN one point zero  
        END IF  

        SET graph TO the result of calling defaultdict with parameter list  
        FOR each pair u v IN edges  
            APPEND v TO the element at position u of graph  
            APPEND u TO the element at position v of graph  
        END FOR  

        SET queue TO the result of calling deque with a list containing tuple of one current node one point zero current time zero  
        SET visited TO a set containing element one  

        WHILE queue is not empty  
            SET current_node TO the element at position zero of the element at position zero of queue  
            SET current_probability TO the element at position one of the element at position zero of queue  
            SET current_time TO the element at position two of the element at position zero of queue  
            REMOVE the element at position zero of queue  

            IF current_time EQUALS t OR (current_node NOT EQUALS one AND the length of the element at position current_node of graph EQUALS one)  
                IF current_node EQUALS target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE TO next iteration of the WHILE loop  
                END IF  
            END IF  

            SET possible_moves TO the length of the element at position current_node of graph MINUS (one IF current_node NOT EQUALS one ELSE zero)  

            FOR each neighbor IN the element at position current_node of graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND tuple of neighbor current_probability DIVIDED BY possible_moves current_time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  

        RETURN zero point zero  
    END FUNCTION  
END CLASS
