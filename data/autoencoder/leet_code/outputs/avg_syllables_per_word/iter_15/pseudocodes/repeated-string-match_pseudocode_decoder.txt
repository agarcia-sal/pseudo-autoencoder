CLASS Solution  
    FUNCTION repeatedStringMatch(a STRING, b STRING) RETURNS INTEGER  
        SET minimum repeats TO the negative value of the integer division of the negative of the length of b BY the length of a  
        IF the string b IS A SUBSTRING OF the string a REPEATED minimum repeats TIMES  
            RETURN minimum repeats  
        END IF  
        IF the string b IS A SUBSTRING OF the string a REPEATED minimum repeats PLUS one TIME  
            RETURN minimum repeats PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
