CLASS Solution  
    FUNCTION minimumTotal(triangle)  
        FOR row FROM the LENGTH OF triangle MINUS two DOWN TO zero  
            FOR col FROM zero TO the LENGTH OF element at position row of triangle MINUS one  
                SET element at position row at position col of triangle TO element at position row at position col of triangle PLUS the MINIMUM OF element at position row PLUS one at position col of triangle AND element at position row PLUS one at position col PLUS one of triangle  
            END FOR  
        END FOR  
        RETURN element at position zero at position zero of triangle  
    END FUNCTION  
END CLASS
