CLASS Solution  
    FUNCTION maxIncreaseKeepingSkyline(grid)  
        SET n TO the LENGTH OF grid  
        SET row_maxes TO an empty list  
        FOR each row IN grid  
            APPEND the maximum value IN row TO row_maxes  
        END FOR  
        SET col_maxes TO an empty list  
        FOR each column IN the TRANSPOSED grid  
            APPEND the maximum value IN column TO col_maxes  
        END FOR  
        SET total_increase TO zero  
        FOR each index i FROM zero TO n MINUS one  
            FOR each index j FROM zero TO n MINUS one  
                SET max_height TO the MINIMUM OF the element at position i of row_maxes AND the element at position j of col_maxes  
                INCREMENT total_increase BY the GREATER OF zero AND the DIFFERENCE BETWEEN max_height AND the element at position i of grid THEN at position j  
            END FOR  
        END FOR  
        RETURN total_increase  
    END FUNCTION  
END CLASS
