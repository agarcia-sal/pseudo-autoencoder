CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value property of p IS NOT EQUAL TO the value property of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with the left child of p and the left child of q AND the result of calling is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(root)  
        IF root IS None  
            RETURN  
        END IF  
        SET stack TO a new list containing root as its only element  
        WHILE stack IS NOT empty  
            SET node TO the element removed from the end of stack  
            IF the right child property of node IS NOT None  
                APPEND the right child property of node TO stack  
            END IF  
            IF the left child property of node IS NOT None  
                APPEND the left child property of node TO stack  
            END IF  
            SET the left child property of node TO None  
            IF stack IS NOT empty  
                SET the right child property of node TO the element at the last position of stack  
            ELSE  
                SET the right child property of node TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
