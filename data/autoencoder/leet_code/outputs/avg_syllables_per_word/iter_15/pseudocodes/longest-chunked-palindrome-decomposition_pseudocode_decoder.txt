CLASS Solution  
    FUNCTION longestDecomposition WITH PARAMETER text  
        SET left TO zero  
        SET right TO the LENGTH OF text MINUS one  
        SET left_part TO empty string  
        SET right_part TO empty string  
        SET k TO zero  
        
        WHILE left IS LESS THAN right  
            SET left_part TO left_part CONCATENATED WITH the element at position left of text  
            SET right_part TO the element at position right of text CONCATENATED WITH right_part  
            
            IF left_part EQUALS right_part  
                INCREMENT k BY two  
                SET left_part TO empty string  
                SET right_part TO empty string  
            END IF  
            
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
        
        IF left_part IS NOT empty OR left EQUALS right  
            INCREMENT k BY one  
        END IF  
        
        RETURN k  
    END FUNCTION  
END CLASS
