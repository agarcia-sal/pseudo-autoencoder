CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO a dictionary with key zero and value zero  
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each key d AND value y IN dp items  
                SET key_for_taller_support TO d PLUS rod  
                SET value_for_taller_support TO the maximum value BETWEEN element at key key_for_taller_support IN new_dp OR zero AND y  
                ASSIGN key_for_taller_support in new_dp TO value_for_taller_support  
                SET absolute_difference TO the absolute value of d MINUS rod  
                SET value_for_shorter_support TO the maximum value BETWEEN element at key absolute_difference IN new_dp OR zero AND y PLUS the minimum value BETWEEN d AND rod  
                ASSIGN absolute_difference in new_dp TO value_for_shorter_support  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN element at key zero in dp  
    END FUNCTION  
END CLASS
