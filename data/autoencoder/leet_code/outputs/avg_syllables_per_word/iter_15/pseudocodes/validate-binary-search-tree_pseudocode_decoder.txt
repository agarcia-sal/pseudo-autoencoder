CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty list  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH parameter val EQUALS element at position zero of values  
    SET i TO one  
    SET queue TO NEW empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from left end of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO NEW TreeNode WITH parameter val EQUALS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO NEW TreeNode WITH parameter val EQUALS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left AND q.left AND the result of is_same_tree with parameters p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low EQUALS negative infinity, high EQUALS positive infinity)  
            IF node IS none  
                RETURN True  
            END IF  
            IF the value of node.val IS LESS THAN OR EQUAL TO low OR the value of node.val IS GREATER THAN OR EQUAL TO high  
                RETURN False  
            END IF  
            RETURN the result of validate with parameters node.left, low, node.val AND the result of validate with parameters node.right, node.val, high  
        END FUNCTION  
        RETURN the result of validate with parameter root  
    END FUNCTION  
END CLASS
