CLASS Solution  
    FUNCTION maxProfit WITH PARAMETERS k AS integer AND prices AS list of integers  
        IF the collection prices is empty OR k EQUALS zero  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF the collection prices  
        
        IF k IS GREATER THAN OR EQUAL TO n DIVIDED BY two disregarding remainder  
            SET profit TO zero  
            FOR index FROM one TO n MINUS one  
                IF element at position index of prices IS GREATER THAN element at position index MINUS one of prices  
                    INCREMENT profit BY element at position index of prices MINUS element at position index MINUS one of prices  
                END IF  
            END FOR  
            RETURN profit  
        END IF  
        
        SET hold TO a collection of length k PLUS one FILLED WITH minus infinity  
        SET release TO a collection of length k PLUS one FILLED WITH zero  
        
        FOR each price IN prices  
            FOR i FROM one TO k  
                SET hold at position i TO the GREATER OF hold at position i AND release at position i MINUS one MINUS price  
                SET release at position i TO the GREATER OF release at position i AND hold at position i PLUS price  
            END FOR  
        END FOR  
        
        RETURN release at position k  
    END FUNCTION  
END CLASS
