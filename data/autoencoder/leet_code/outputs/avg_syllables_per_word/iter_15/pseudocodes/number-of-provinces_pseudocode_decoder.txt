CLASS Solution  
    FUNCTION findCircleNum WITH PARAMETER isConnected  
        FUNCTION dfs WITH PARAMETER city  
            FOR neighbor FROM zero TO the LENGTH OF isConnected MINUS one  
                IF the element at position city of isConnected AT position neighbor EQUALS one AND the element at position neighbor of visited EQUALS false  
                    SET the element at position neighbor of visited TO true  
                    CALL dfs WITH ARGUMENT neighbor  
                END IF  
            END FOR  
        END FUNCTION  
  
        SET n TO the LENGTH OF isConnected  
        INITIALIZE visited AS a list of the value false repeated n times  
        SET province_count TO zero  
  
        FOR city FROM zero TO n MINUS one  
            IF the element at position city of visited EQUALS false  
                CALL dfs WITH ARGUMENT city  
                INCREMENT province_count BY one  
            END IF  
        END FOR  
  
        RETURN province_count  
    END FUNCTION  
END CLASS
