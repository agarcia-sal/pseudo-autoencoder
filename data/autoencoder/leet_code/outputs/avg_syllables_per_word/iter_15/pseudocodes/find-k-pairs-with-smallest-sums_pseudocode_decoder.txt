CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k IS LESS THAN OR EQUAL TO zero  
            RETURN an EMPTY list  
        END IF  
        
        SET min_heap TO an EMPTY list  
        
        FOR index FROM zero TO the MINIMUM OF k MINUS one AND the LENGTH OF nums1 MINUS one  
            CALCULATE sum_of_elements AS element at position index of nums1 PLUS element at position zero of nums2  
            PUSH a TUPLE OF sum_of_elements AND index AND zero INTO min_heap USING THE HEAP INSERTION FUNCTION  
        END FOR  
        
        SET result TO an EMPTY list  
        
        WHILE min_heap IS NOT EMPTY AND the LENGTH OF result IS LESS THAN k  
            EXTRACT the TUPLE WITH the SMALLEST sum_of_elements FROM min_heap USING THE HEAP REMOVAL FUNCTION AND ASSIGN THE COMPONENTS TO variables IGNORE_SUM, i, j  
            APPEND a LIST CONTAINING element at position i of nums1 AND element at position j of nums2 TO result  
            
            IF j PLUS one IS LESS THAN the LENGTH OF nums2  
                CALCULATE next_sum AS element at position i of nums1 PLUS element at position j PLUS one of nums2  
                PUSH a TUPLE OF next_sum AND i AND j PLUS one INTO min_heap USING THE HEAP INSERTION FUNCTION  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
