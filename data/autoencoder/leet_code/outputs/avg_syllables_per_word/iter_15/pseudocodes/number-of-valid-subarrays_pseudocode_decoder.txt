CLASS Solution  
    FUNCTION validSubarrays(nums)  
        SET stack TO empty list  
        SET count TO zero  
        FOR each num IN nums  
            WHILE stack IS NOT empty AND element at last position of stack IS GREATER THAN num  
                REMOVE element at last position FROM stack  
            END WHILE  
            APPEND num TO stack  
            INCREMENT count BY the LENGTH OF stack  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
