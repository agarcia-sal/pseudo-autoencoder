CLASS Solution  
    FUNCTION closestToTarget(arr PARAMETER, target PARAMETER)  
        SET seen TO empty collection of unique values  
        SET min_diff TO positive infinity  
        FOR each num IN arr  
            SET temporary collection TO empty collection of unique values  
            FOR each x IN seen  
                APPEND the bitwise AND of num AND x TO temporary collection  
            END FOR  
            SET seen TO the combination of temporary collection AND the collection containing num  
            FOR each val IN seen  
                SET current_difference TO the absolute value of val MINUS target  
                IF current_difference LESS THAN min_diff  
                    SET min_diff TO current_difference  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
