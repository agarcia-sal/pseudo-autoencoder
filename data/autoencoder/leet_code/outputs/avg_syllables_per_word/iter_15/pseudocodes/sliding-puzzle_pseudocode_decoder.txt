CLASS Solution  
    FUNCTION slidingPuzzle WITH PARAMETERS board  
        SET start TO the tuple of tuples formed by converting each row of board into a tuple  
        SET target TO the tuple consisting of the tuples one two three and four five zero in two rows  
        
        SET neighbors TO a mapping where  
            zero maps to the set containing one and three  
            one maps to the set containing zero two and four  
            two maps to the set containing one and five  
            three maps to the set containing zero and four  
            four maps to the set containing one three and five  
            five maps to the set containing two and four  
        
        FOR each integer i FROM zero TO one  
            FOR each integer j FROM zero TO two  
                IF the element at position j of the element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        
        SET queue TO a deque initialized with a tuple of start zero_pos and zero  
        SET visited TO a set initialized with start  
        
        WHILE queue is not empty  
            REMOVE the leftmost element from queue and ASSIGN it to state zero_pos moves  
            
            IF state EQUALS target  
                RETURN moves  
            END IF  
            
            FOR each neighbor IN the elements of neighbors at key zero_pos  
                SET new_state TO a list formed by converting each element of state into a list  
                SET i TO the integer division of zero_pos by three  
                SET j TO the remainder of zero_pos divided by three  
                SET ni TO the integer division of neighbor by three  
                SET nj TO the remainder of neighbor divided by three  
                
                SWAP the element at position j of the element at position i of new_state WITH the element at position nj of the element at position ni of new_state  
                
                SET new_tuple_state TO the tuple of tuples formed by converting each row of new_state into a tuple  
                
                IF new_tuple_state is not in visited  
                    ADD new_tuple_state TO visited  
                    APPEND the tuple new_tuple_state neighbor moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
