CLASS Solution  
    FUNCTION numPermsDISequence WITH PARAMETER s  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional collection of zeros with n PLUS one rows and n PLUS one columns  
        FOR each j FROM zero TO n  
            SET element at position j of collection at position zero of dp TO one  
        END FOR  
        FOR each i FROM one TO n  
            SET prefix_sum TO zero  
            IF character at position i MINUS one of s EQUALS the letter I  
                FOR each j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO the REMAINDER OF prefix_sum PLUS element at position j of collection at position i MINUS one of dp DIVIDED BY MOD  
                    SET element at position j of collection at position i of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR each j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO the REMAINDER OF prefix_sum PLUS element at position j PLUS one of collection at position i MINUS one of dp DIVIDED BY MOD  
                    SET element at position j of collection at position i of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        SET result TO the SUM OF collection at position n of dp  
        RETURN the REMAINDER OF result DIVIDED BY MOD  
    END FUNCTION  
END CLASS
