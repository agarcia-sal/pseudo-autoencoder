CLASS Solution  
    FUNCTION validTicTacToe(board)  
        SET countX TO zero  
        FOR each row IN board  
            INCREMENT countX BY the NUMBER OF OCCURRENCES OF character X IN row  
        END FOR  
        SET countO TO zero  
        FOR each row IN board  
            INCREMENT countO BY the NUMBER OF OCCURRENCES OF character O IN row  
        END FOR  
        IF countX LESS THAN countO OR countX GREATER THAN countO PLUS one  
            RETURN False  
        END IF  

        FUNCTION check_win(player)  
            FOR each row IN board  
                IF row EQUALS player REPEATED THREE TIMES  
                    RETURN True  
                END IF  
            END FOR  
            FOR column_index FROM zero TO two  
                IF element at position column_index of first row EQUALS player AND element at position column_index of second row EQUALS player AND element at position column_index of third row EQUALS player  
                    RETURN True  
                END IF  
            END FOR  
            IF element at position zero of first row EQUALS player AND element at position one of second row EQUALS player AND element at position two of third row EQUALS player  
                RETURN True  
            END IF  
            IF element at position two of first row EQUALS player AND element at position one of second row EQUALS player AND element at position zero of third row EQUALS player  
                RETURN True  
            END IF  
            RETURN False  
        END FUNCTION  

        SET winX TO check_win(character X)  
        SET winO TO check_win(character O)  

        IF winX AND winO  
            RETURN False  
        END IF  
        IF winX AND countX EQUALS countO  
            RETURN False  
        END IF  
        IF winO AND countX NOT EQUALS countO  
            RETURN False  
        END IF  

        RETURN True  
    END FUNCTION  
END CLASS
