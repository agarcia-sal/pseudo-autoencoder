CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO a set created by CONVERTING each point in points into a tuple  
        SET min_x TO the minimum value of the first element at position zero of each point in unique_points  
        SET max_x TO the maximum value of the first element at position zero of each point in unique_points  
        SET line_of_reflection TO the sum of min_x AND max_x DIVIDED BY two  
        FOR each point consisting of first element named x AND second element named y IN unique_points  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the pair consisting of reflected_x AND y IS NOT an element of unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
