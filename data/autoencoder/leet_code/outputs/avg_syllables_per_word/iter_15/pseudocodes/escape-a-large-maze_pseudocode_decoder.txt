CLASS Solution  
    FUNCTION isEscapePossible(blocked, source, target)  
        IF blocked IS empty  
            RETURN True  
        END IF  
        
        SET blocked_set TO a set containing the tuple form of each block in blocked  
        SET n TO the LENGTH OF blocked  
        SET max_steps TO n MULTIPLIED BY n MINUS one DIVIDED BY two  
        
        FUNCTION bfs(start, end)  
            SET queue TO a deque initialized with the tuple form of start  
            SET visited TO a set initialized with the elements of queue  
            
            WHILE queue IS NOT empty  
                SET current_position TO the element removed from the left of queue  
                SET x TO the first element of current_position  
                SET y TO the second element of current_position  
                
                IF the LENGTH OF visited IS GREATER THAN max_steps OR x EQUALS the first element of end AND y EQUALS the second element of end  
                    RETURN True  
                END IF  
                
                FOR each pair dx AND dy IN the list of pairs zero and one zero and negative one one and zero negative one and zero  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN one million AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN one million AND the tuple containing nx and ny IS NOT IN blocked_set AND the tuple containing nx and ny IS NOT IN visited  
                        ADD the tuple containing nx and ny TO visited  
                        APPEND the tuple containing nx and ny TO queue  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN False  
        END FUNCTION  
        
        RETURN the result of bfs called with source and target AND the result of bfs called with target and source  
    END FUNCTION  
END CLASS
