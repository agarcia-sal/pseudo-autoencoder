CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF the element at position zero of cost  
        SET dp TO a two dimensional list with size1 plus one rows and with two TO THE POWER OF size2 columns each initialized TO positive infinity  
        SET the element at position zero of dp and zero TO zero  
        
        FOR each index i FROM zero TO size1 MINUS one  
            FOR each mask FROM zero TO two TO THE POWER OF size2 MINUS one  
                FOR each index j FROM zero TO size2 MINUS one  
                    SET position TO mask OR the value one SHIFTED LEFT BY j  
                    SET new_cost TO the element at position i and position j of cost PLUS the element at position i and position mask of dp  
                    IF the element at position i plus one and position position of dp IS GREATER THAN new_cost  
                        SET the element at position i plus one and position position of dp TO new_cost  
                    END IF  
                END FOR  
                
                FOR each index j FROM zero TO size2 MINUS one  
                    IF the bitwise AND of mask AND the value one SHIFTED LEFT BY j IS NOT EQUAL TO zero  
                        SET new_mask TO the bitwise XOR of mask AND the value one SHIFTED LEFT BY j  
                        SET possible_cost TO the element at position i plus one and position new_mask of dp PLUS the element at position i and position j of cost  
                        IF the element at position i plus one and position mask of dp IS GREATER THAN possible_cost  
                            SET the element at position i plus one and position mask of dp TO possible_cost  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN the element at position size1 and position two TO THE POWER OF size2 MINUS one of dp  
    END FUNCTION  
END CLASS
