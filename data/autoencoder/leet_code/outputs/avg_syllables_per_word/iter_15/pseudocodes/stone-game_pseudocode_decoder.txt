CLASS Solution  
    FUNCTION stoneGame WITH PARAMETER piles AS list of integers  
        FUNCTION dfs WITH PARAMETERS i AS integer AND j AS integer  
            IF i IS GREATER THAN j THEN  
                RETURN zero  
            END IF  
            SET left_score TO element at position i of piles MINUS dfs WITH PARAMETERS i PLUS one AND j  
            SET right_score TO element at position j of piles MINUS dfs WITH PARAMETERS i AND j MINUS one  
            RETURN the GREATER VALUE BETWEEN left_score AND right_score  
        END FUNCTION  
        SET result TO dfs WITH PARAMETERS zero AND the LENGTH OF piles MINUS one  
        RETURN result IS GREATER THAN zero  
    END FUNCTION  
END CLASS
