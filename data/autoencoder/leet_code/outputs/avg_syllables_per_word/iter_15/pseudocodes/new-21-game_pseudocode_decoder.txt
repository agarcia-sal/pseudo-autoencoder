CLASS Solution  
    FUNCTION new21Game(n, k, maxPts)  
        FUNCTION dfs(i)  
            IF i IS GREATER THAN OR EQUAL TO k  
                IF i IS LESS THAN OR EQUAL TO n  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF i IS EQUAL TO k MINUS one  
                SET difference TO n MINUS k PLUS one  
                IF difference IS LESS THAN maxPts  
                    SET numerator TO difference  
                ELSE  
                    SET numerator TO maxPts  
                END IF  
                RETURN numerator DIVIDED BY maxPts  
            END IF  
            SET first_call TO dfs(i PLUS one)  
            SET second_call TO dfs(i PLUS maxPts PLUS one)  
            SET recurrence_value TO first_call PLUS (first_call MINUS second_call) DIVIDED BY maxPts  
            RETURN recurrence_value  
        END FUNCTION  
        RETURN dfs(zero)  
    END FUNCTION  
END CLASS
