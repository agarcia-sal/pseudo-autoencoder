CLASS Solution  
    FUNCTION groupAnagrams WITH PARAMETER strs as list of strings  
        SET anagrams TO a new mapping with default value as an empty list  
        FOR each s IN strs  
            SET characters_in_s TO the list of characters IN s  
            SET sorted_characters TO characters_in_s SORTED in ascending order  
            SET sorted_str TO the concatenation of all characters IN sorted_characters in order  
            APPEND s TO the list WITH KEY sorted_str IN anagrams  
        END FOR  
        SET result TO the list of all values IN anagrams  
        RETURN result  
    END FUNCTION  
END CLASS
