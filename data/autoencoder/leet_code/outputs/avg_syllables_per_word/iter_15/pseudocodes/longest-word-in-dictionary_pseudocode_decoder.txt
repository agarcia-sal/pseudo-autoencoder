CLASS Solution  
    FUNCTION longestWord(words)  
        CALL words SORT BY length OF word AND lexicographically by word  
        SET valid_words TO empty set  
        SET longest_word TO empty string  
        FOR each word IN words  
            IF the LENGTH OF word EQUALS one OR the SUBSTRING OF word from position one TO the LENGTH OF word MINUS one IS IN valid_words  
                ADD word TO valid_words  
                IF the LENGTH OF word GREATER THAN the LENGTH OF longest_word  
                    SET longest_word TO word  
                END IF  
            END IF  
        END FOR  
        RETURN longest_word  
    END FUNCTION  
END CLASS
