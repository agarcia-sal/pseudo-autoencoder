CLASS Solution  
    FUNCTION minAbbreviation WITH PARAMETERS target STRING AND dictionary LIST OF STRING  
        FUNCTION is_valid_abbr WITH PARAMETERS s STRING AND abbr STRING  
            SET i TO zero  
            SET j TO zero  
            WHILE i LESS THAN the LENGTH OF s AND j LESS THAN the LENGTH OF abbr  
                IF the CHARACTER at position j of abbr IS A DIGIT  
                    IF the CHARACTER at position j of abbr EQUALS zero  
                        RETURN False  
                    END IF  
                    SET shift TO zero  
                    WHILE j LESS THAN the LENGTH OF abbr AND the CHARACTER at position j of abbr IS A DIGIT  
                        SET shift TO shift MULTIPLIED BY ten PLUS the INTEGER VALUE of the CHARACTER at position j of abbr  
                        INCREMENT j BY one  
                    END WHILE  
                    INCREMENT i BY shift  
                ELSE  
                    IF the CHARACTER at position i of s NOT EQUALS the CHARACTER at position j of abbr  
                        RETURN False  
                    END IF  
                    INCREMENT i BY one  
                    INCREMENT j BY one  
                END IF  
            END WHILE  
            RETURN i EQUALS the LENGTH OF s AND j EQUALS the LENGTH OF abbr  
        END FUNCTION  
        
        FUNCTION generate_abbrs WITH PARAMETERS word STRING length INTEGER abbr STRING DEFAULT EMPTY STRING index INTEGER DEFAULT zero skip_count INTEGER DEFAULT zero  
            IF index EQUALS length  
                IF skip_count GREATER THAN zero  
                    SET abbr TO abbr CONCATENATED WITH THE STRING REPRESENTATION OF skip_count  
                END IF  
                APPEND abbr TO abbrs  
                RETURN  
            END IF  
            CALL generate_abbrs WITH word length abbr index PLUS one skip_count PLUS one  
            IF skip_count GREATER THAN zero  
                SET abbr TO abbr CONCATENATED WITH THE STRING REPRESENTATION OF skip_count  
            END IF  
            CALL generate_abbrs WITH word length abbr CONCATENATED WITH the CHARACTER at position index of word index PLUS one zero  
        END FUNCTION  
        
        FUNCTION find_unique_abbr  
            FOR length FROM one TO the LENGTH OF target  
                SET abbrs TO an EMPTY LIST  
                CALL generate_abbrs WITH target LENGTH OF target  
                SET min_abbr TO None  
                SET min_length TO INFINITY  
                FOR each abbr IN abbrs  
                    IF for ALL each word IN filtered_dict the RESULT OF is_valid_abbr WITH PARAMETERS word AND abbr IS False  
                        IF the LENGTH OF abbr LESS THAN min_length  
                            SET min_length TO the LENGTH OF abbr  
                            SET min_abbr TO abbr  
                        END IF  
                    END IF  
                END FOR  
                IF min_abbr IS NOT None  
                    RETURN min_abbr  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET filtered_dict TO an EMPTY LIST  
        FOR each word IN dictionary  
            IF the LENGTH OF word EQUALS the LENGTH OF target  
                APPEND word TO filtered_dict  
            END IF  
        END FOR  
        
        SET abbrs TO an EMPTY LIST  
        RETURN the RESULT OF find_unique_abbr  
    END FUNCTION  
END CLASS
