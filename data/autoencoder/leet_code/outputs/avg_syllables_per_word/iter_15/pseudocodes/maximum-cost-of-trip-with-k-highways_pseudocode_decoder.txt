CLASS Solution  
    FUNCTION maximumCost(n, highways, k)  
        CREATE adjacency list named graph mapping each city TO list of neighboring cities with their tolls  
        FOR each city1 city2 toll IN highways  
            APPEND the tuple of city2 and toll TO the list associated WITH city1 IN graph  
            APPEND the tuple of city1 and toll TO the list associated WITH city2 IN graph  
        END FOR  
        
        SET max_cost TO negative one  
        
        FUNCTION dfs(city, visited, cost, highways_used)  
            IF highways_used EQUALS k THEN  
                SET max_cost TO the greater value BETWEEN max_cost AND cost  
                RETURN  
            END IF  
            
            FOR each neighbor toll IN the list of neighbors associated WITH city IN graph  
                IF neighbor IS NOT IN visited THEN  
                    ADD neighbor TO visited  
                    CALL dfs WITH neighbor visited cost PLUS toll highways_used PLUS one  
                    REMOVE neighbor FROM visited  
                END IF  
            END FOR  
        END FUNCTION  
        
        FOR each start_city FROM zero TO n MINUS one  
            CREATE a set named visited containing start_city  
            CALL dfs WITH start_city visited zero zero  
        END FOR  
        
        RETURN max_cost  
    END FUNCTION  
END CLASS
