CLASS Solution  
    FUNCTION calculate WITH PARAMETER s  
        FUNCTION helper WITH PARAMETER s  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO the string plus  
            WHILE the LENGTH OF s GREATER THAN zero holds  
                SET char TO the element at position zero of s  
                REMOVE the element at position zero from s  
                IF char IS a digit THEN  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                IF char EQUALS the string left parenthesis THEN  
                    SET num TO the result of calling helper WITH PARAMETER s  
                END IF  
                IF char EQUALS any one of the strings plus minus multiplied by divided by left parenthesis right parenthesis OR the LENGTH OF s EQUALS zero THEN  
                    IF sign EQUALS the string plus THEN  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS the string minus THEN  
                        APPEND the negative of num TO stack  
                    ELSE IF sign EQUALS the string multiplied by THEN  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS the string divided by THEN  
                        SET the element at the last position of stack TO the integer value of the element at the last position of stack DIVIDED BY num with truncation towards zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS the string right parenthesis THEN  
                    BREAK the loop  
                END IF  
            END WHILE  
            RETURN the SUM OF all elements in stack  
        END FUNCTION  

        SET s TO the list of all characters in s  
        RETURN the result of calling helper WITH PARAMETER s  
    END FUNCTION  
END CLASS
