CLASS Solution  
    FUNCTION checkPossibility(nums)  
        SET modified TO false  
        FOR index FROM one TO the LENGTH OF nums MINUS one  
            IF element at position index of nums IS LESS THAN element at position index MINUS one of nums  
                IF modified IS true  
                    RETURN false  
                END IF  
                SET modified TO true  
                IF index IS LESS THAN two OR element at position index of nums IS GREATER THAN OR EQUAL TO element at position index MINUS two of nums  
                    SET element at position index MINUS one of nums TO element at position index of nums  
                ELSE  
                    SET element at position index of nums TO element at position index MINUS one of nums  
                END IF  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
