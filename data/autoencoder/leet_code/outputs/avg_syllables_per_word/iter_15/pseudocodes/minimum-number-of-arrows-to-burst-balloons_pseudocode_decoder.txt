CLASS Solution  
    FUNCTION findMinArrowShots(points AS List OF List OF Integer)  
        SORT points BY the element at position one of each balloon in ascending order  
        SET arrows TO zero  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF the element at position zero of balloon IS GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO the element at position one of balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
