CLASS Solution  
    FUNCTION maximumMinutes(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO the list containing the pairs one zero and MINUS one zero and zero one and zero MINUS one  
        
        FUNCTION can_escape(wait)  
            SET fire_times TO a two-dimensional list of size m by n filled with infinity values  
            SET queue TO an empty double-ended queue  
            
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    IF element at position i of grid EQUALS one  
                        APPEND the tuple of i j zero TO queue  
                        SET element at position i j of fire_times TO zero  
                    END IF  
                END FOR  
            END FOR  
            
            WHILE queue is not empty  
                REMOVE the first element FROM queue and ASSIGN its values respectively TO x y t  
                FOR each pair dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND element at position nx ny of grid EQUALS zero AND element at position nx ny of fire_times EQUALS infinity  
                        SET element at position nx ny of fire_times TO t PLUS one  
                        APPEND the tuple of nx ny t PLUS one TO queue  
                    END IF  
                END FOR  
            END WHILE  
            
            SET queue TO a double-ended queue initialized with the tuple zero zero wait  
            SET visited TO a set initialized with the tuple zero zero  
            
            WHILE queue is not empty  
                REMOVE the first element FROM queue and ASSIGN its values respectively TO x y t  
                FOR each pair dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND tuple nx ny IS NOT IN visited AND element at position nx ny of grid EQUALS zero  
                        IF nx EQUALS m MINUS one AND ny EQUALS n MINUS one  
                            IF element at position nx ny of fire_times IS GREATER THAN OR EQUAL TO t PLUS one  
                                RETURN True  
                            ELSE  
                                RETURN False  
                            END IF  
                        END IF  
                        IF element at position nx ny of fire_times IS GREATER THAN t PLUS one  
                            ADD tuple nx ny TO visited  
                            APPEND tuple nx ny t PLUS one TO queue  
                        END IF  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN False  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO m MULTIPLIED BY n  
        SET result TO MINUS one  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the integer division of left PLUS right BY two  
            IF can_escape(mid) IS True  
                SET result TO mid  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        
        IF result IS EQUAL TO m MULTIPLIED BY n  
            RETURN one FOLLOWED BY nine FOLLOWED BY zero FOLLOWED BY zero FOLLOWED BY zero FOLLOWED BY zero FOLLOWED BY zero FOLLOWED BY zero FOLLOWED BY zero FOLLOWED BY zero  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
