CLASS Solution  
    FUNCTION strobogrammaticInRange(low STRING, high STRING) RETURNS INTEGER  
        FUNCTION generate_strobogrammatic(n INTEGER, final_length INTEGER) RETURNS LIST OF STRING  
            IF n EQUALS zero THEN  
                RETURN list containing empty string  
            END IF  
            IF n EQUALS one THEN  
                RETURN list containing string zero string one string eight  
            END IF  
            SET middles TO generate_strobogrammatic(n MINUS two, final_length)  
            SET result TO empty list  
            FOR each middle IN middles  
                IF n NOT EQUAL TO final_length THEN  
                    APPEND concatenation of string zero with middle with string zero TO result  
                END IF  
                APPEND concatenation of string one with middle with string one TO result  
                APPEND concatenation of string six with middle with string nine TO result  
                APPEND concatenation of string eight with middle with string eight TO result  
                APPEND concatenation of string nine with middle with string six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        FUNCTION count_strobogrammatic(low STRING, high STRING) RETURNS INTEGER  
            SET count TO zero  
            FOR length FROM the LENGTH OF low TO the LENGTH OF high  
                FOR each num IN generate_strobogrammatic(length, length)  
                    IF (length EQUALS the LENGTH OF low AND num LESS THAN low) OR (length EQUALS the LENGTH OF high AND num GREATER THAN high) THEN  
                        CONTINUE to next iteration  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN count_strobogrammatic(low, high)  
    END FUNCTION  
END CLASS
