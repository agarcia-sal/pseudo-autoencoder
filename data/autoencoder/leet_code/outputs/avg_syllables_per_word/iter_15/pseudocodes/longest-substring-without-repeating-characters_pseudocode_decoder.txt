CLASS Solution  
    FUNCTION lengthOfLongestSubstring(with parameter s)  
        SET char_index_map TO an empty mapping of characters to indices  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each position end AND character char IN the enumeration of s  
            IF char IS IN char_index_map AND the element at character char of char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO the element at character char of char_index_map PLUS one  
            END IF  
            SET the element at character char of char_index_map TO end  
            SET max_length TO the GREATER OF max_length AND end MINUS start PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
