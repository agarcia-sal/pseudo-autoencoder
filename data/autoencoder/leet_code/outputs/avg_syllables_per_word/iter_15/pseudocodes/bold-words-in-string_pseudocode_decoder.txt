CLASS Solution  
    FUNCTION boldWords(words AS List of String, s AS String) RETURNS String  
        SET bold TO boolean list of the length of s all set to false  
        FOR each word IN words  
            SET start TO the position at which word first occurs in s  
            WHILE start DOES NOT EQUAL negative one  
                FOR each index FROM start TO start plus the LENGTH OF word MINUS one  
                    SET the element at position index of bold TO true  
                END FOR  
                SET start TO the position at which word next occurs in s starting from one position after start  
            END WHILE  
        END FOR  
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF s  
            IF the element at position i of bold IS true  
                APPEND the string lowercase be character to result  
                WHILE i LESS THAN the LENGTH OF s AND the element at position i of bold IS true  
                    APPEND the character at position i of s to result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string slash be characters to result  
            ELSE  
                APPEND the character at position i of s to result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
