CLASS Solution  
    FUNCTION fractionToDecimal(numerator INTEGER denominator INTEGER) RETURNS STRING  
        IF numerator LESS THAN zero EXCLUSIVE OR denominator LESS THAN zero  
            SET sign TO MINUS SIGN  
        ELSE  
            SET sign TO EMPTY STRING  
        END IF  
  
        SET numerator TO ABSOLUTE VALUE OF numerator  
        SET denominator TO ABSOLUTE VALUE OF denominator  
  
        SET quotient TO INTEGER RESULT OF numerator DIVIDED BY denominator  
        SET remainder TO REMAINDER OF numerator DIVIDED BY denominator  
        SET integer_part TO STRING REPRESENTATION OF quotient  
  
        IF remainder EQUALS zero  
            RETURN CONCATENATION OF sign AND integer_part  
        END IF  
  
        SET seen_remainders TO EMPTY DICTIONARY  
        SET fractional_part TO EMPTY LIST  
  
        WHILE remainder NOT EQUAL TO zero  
            IF remainder EXISTS AS A KEY IN seen_remainders  
                SET start TO VALUE ASSOCIATED WITH remainder IN seen_remainders  
                SET non_repeating TO CONCATENATION OF ELEMENTS FROM FIRST ELEMENT TO ELEMENT BEFORE start IN fractional_part  
                SET repeating TO CONCATENATION OF ELEMENTS FROM start TO LAST ELEMENT IN fractional_part  
                RETURN CONCATENATION OF sign AND integer_part AND A PERIOD AND non_repeating AND OPENING PARENTHESIS AND repeating AND CLOSING PARENTHESIS  
            END IF  
  
            ASSIGN remainder AS A KEY IN seen_remainders TO THE LENGTH OF fractional_part  
  
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO INTEGER RESULT OF remainder DIVIDED BY denominator  
            SET remainder TO REMAINDER OF remainder DIVIDED BY denominator  
            APPEND STRING REPRESENTATION OF quotient TO fractional_part  
        END WHILE  
  
        RETURN CONCATENATION OF sign AND integer_part AND A PERIOD AND CONCATENATION OF ALL ELEMENTS IN fractional_part  
    END FUNCTION  
END CLASS
