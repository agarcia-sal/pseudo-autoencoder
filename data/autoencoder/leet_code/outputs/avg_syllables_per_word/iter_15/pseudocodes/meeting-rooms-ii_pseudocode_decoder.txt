CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals IS EMPTY  
            RETURN zero  
        END IF  
        
        SORT intervals BY the element at position zero of each meeting period  
        
        SET min_heap TO empty list  
        
        FOR each meeting IN intervals  
            IF min_heap IS NOT EMPTY AND the element at position zero of meeting IS GREATER THAN OR EQUAL TO the element at position zero of min_heap  
                REMOVE the smallest element FROM min_heap  
            END IF  
            
            ADD the element at position one of meeting TO min_heap PRESERVING MINIMUM ORDER  
        END FOR  
        
        RETURN the number of elements IN min_heap  
    END FUNCTION  
END CLASS
