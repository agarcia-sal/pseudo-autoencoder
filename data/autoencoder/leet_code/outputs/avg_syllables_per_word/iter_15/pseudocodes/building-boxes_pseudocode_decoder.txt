CLASS Solution  
    FUNCTION minimumBoxes(n)  
        FUNCTION triangular_pyramid(k)  
            RETURN the product of k AND the sum of k PLUS one AND the sum of k PLUS two DIVIDED BY six  
        END FUNCTION  

        SET left TO zero  
        SET right TO n  
        WHILE the value of left IS LESS THAN the value of right  
            SET mid TO the integer division of the sum of left PLUS right PLUS one BY two  
            IF the result of triangular_pyramid with parameter mid IS LESS THAN OR EQUAL TO n  
                SET left TO mid  
            ELSE  
                SET right TO the difference of right MINUS one  
            END IF  
        END WHILE  

        SET boxes_used TO the result of triangular_pyramid with parameter left  
        SET remaining TO the difference of n MINUS boxes_used  
        SET floor_boxes TO the integer division of the product of left AND the sum of left PLUS one BY two  

        SET additional_boxes TO zero  
        WHILE remaining IS GREATER THAN zero  
            INCREMENT additional_boxes BY one  
            DECREMENT remaining BY additional_boxes  
        END WHILE  

        RETURN the sum of floor_boxes PLUS additional_boxes  
    END FUNCTION  
END CLASS
