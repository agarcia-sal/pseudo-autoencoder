CLASS Solution  
    FUNCTION findAllPeople(n AS integer, meetings AS list of list of integer, firstPerson AS integer) AS list of integer  
        SET know_secret TO new empty set  
        ADD zero TO know_secret  
        ADD firstPerson TO know_secret  
        
        SORT meetings BY the element at position two of each meeting  
        
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO the element at position two of the element at position i of meetings  
            SET graph TO a new empty dictionary with default values as empty lists  
            WHILE i LESS THAN the LENGTH OF meetings AND the element at position two of the element at position i of meetings EQUALS current_time  
                SET x TO the element at position zero of the element at position i of meetings  
                SET y TO the element at position one of the element at position i of meetings  
                APPEND y TO the list at key x IN graph  
                APPEND x TO the list at key y IN graph  
                INCREMENT i BY one  
            END WHILE  
            
            SET queue TO a new deque containing each person FROM know_secret if that person IS a key IN graph  
            SET visited TO a new empty set  
            WHILE the LENGTH OF queue IS GREATER THAN zero  
                SET person TO the first element removed FROM the left of queue  
                IF person IS IN visited  
                    CONTINUE to the next iteration of the loop  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list at key person IN graph  
                    IF neighbor IS NOT IN visited  
                        APPEND neighbor TO the right side of queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        
        RETURN the conversion of know_secret TO a list  
    END FUNCTION  
END CLASS
