CLASS Solution  
    FUNCTION canConvert(str1, str2)  
        IF str1 EQUALS str2  
            RETURN True  
        END IF  
        
        SET conversion_map TO an empty mapping from character to character  
        
        FOR each pair of char1 AND char2 IN corresponding positions of str1 AND str2  
            IF char1 IS A KEY IN conversion_map  
                IF the value at key char1 in conversion_map IS NOT EQUAL TO char2  
                    RETURN False  
                END IF  
            ELSE  
                SET the value at key char1 in conversion_map TO char2  
            END IF  
        END FOR  
        
        IF the number of unique characters in str2 EQUALS twenty six  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
