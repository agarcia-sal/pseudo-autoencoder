CLASS Solution  
    FUNCTION isOneEditDistance WITH PARAMETERS s AS string AND t AS string  
        IF the LENGTH OF s IS GREATER THAN the LENGTH OF t THEN  
            RETURN the RESULT OF CALLING isOneEditDistance WITH PARAMETERS t AND s  
        END IF  
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        IF diff IS GREATER THAN one THEN  
            RETURN false  
        END IF  
        IF diff EQUALS zero THEN  
            SET count_of_differences TO zero  
            FOR each pair of characters c1 and c2 IN corresponding positions OF s AND t  
                IF c1 IS NOT EQUAL TO c2 THEN  
                    INCREMENT count_of_differences BY one  
                END IF  
            END FOR  
            RETURN whether count_of_differences EQUALS one  
        ELSE  
            FOR each index i FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position i OF s IS NOT EQUAL TO the element at position i OF t THEN  
                    RETURN whether the substring from position i TO the END OF s IS EQUAL TO the substring from position i PLUS one TO the END OF t  
                END IF  
            END FOR  
            RETURN true  
        END IF  
    END FUNCTION  
END CLASS
