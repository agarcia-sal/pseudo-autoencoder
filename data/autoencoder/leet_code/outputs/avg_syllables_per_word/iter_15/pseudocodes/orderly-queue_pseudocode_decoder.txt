CLASS Solution  
    FUNCTION orderlyQueue with parameters s and k  
        IF k GREATER THAN one  
            RETURN the STRING formed by SORTING the characters of s in lexicographical order  
        ELSE  
            SET n TO the LENGTH OF s  
            SET smallest TO s  
            FOR index FROM one TO n MINUS one  
                SET rotated TO the CONCATENATION OF the substring from position index TO the end of s AND the substring from position zero TO index MINUS one of s  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
