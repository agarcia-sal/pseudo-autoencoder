CLASS Solution  
    FUNCTION minPushBox(grid)  
        SET directions TO list of pairs of numbers minus one zero zero zero one zero one zero  
        FOR index i FROM zero TO LENGTH OF grid MINUS one  
            FOR index j FROM zero TO LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS the character S  
                    SET player_start TO pair of i and j  
                ELSE IF element at position i of grid at position j EQUALS the character B  
                    SET box_start TO pair of i and j  
                ELSE IF element at position i of grid at position j EQUALS the character T  
                    SET target TO pair of i and j  
                END IF  
            END FOR  
        END FOR  
        ASSIGN pointer TO a new double ended queue with element as tuple of player_start box_start and zero  
        SET visited TO empty set  
        ADD tuple of player_start and box_start TO visited  
        WHILE the queue is not empty  
            REMOVE and ASSIGN front element of queue TO player_pos box_pos push_count  
            IF box_pos EQUALS target  
                RETURN push_count  
            END IF  
            FOR each pair dx dy IN directions  
                SET new_player_pos TO pair of first element of player_pos PLUS dx and second element of player_pos PLUS dy  
                IF is_valid_position with new_player_pos and grid returns true  
                    IF new_player_pos EQUALS box_pos  
                        SET new_box_pos TO pair of first element of box_pos PLUS dx and second element of box_pos PLUS dy  
                        IF is_valid_position with new_box_pos and grid returns true AND tuple new_player_pos new_box_pos NOT IN visited  
                            ADD tuple of new_player_pos and new_box_pos TO visited  
                            APPEND tuple of new_player_pos new_box_pos push_count PLUS one TO queue  
                        END IF  
                    ELSE  
                        IF tuple of new_player_pos and box_pos NOT IN visited  
                            ADD tuple of new_player_pos and box_pos TO visited  
                            APPEND tuple of new_player_pos box_pos push_count TO queue  
                        END IF  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  

    FUNCTION is_valid_position(position grid)  
        SET x TO first element of position  
        SET y TO second element of position  
        RETURN x GREATER THAN OR EQUAL TO zero AND x LESS THAN LENGTH OF grid AND y GREATER THAN OR EQUAL TO zero AND y LESS THAN LENGTH OF element at position zero of grid AND element at position x of grid at position y NOT EQUALS the character hash  
    END FUNCTION  
END CLASS
