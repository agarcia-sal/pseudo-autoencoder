CLASS Solution  
    FUNCTION findMaxValueOfEquation(points AS List OF List OF Integer, k AS Integer) RETURNS Integer  
        SET max_value TO the smallest possible number  
        INITIALIZE deque AS an empty double ended queue  
        FOR each pair xj and yj IN points  
            WHILE deque is not empty AND the difference between xj and the element at the front position one of deque is greater than k  
                REMOVE the element at the front position one of deque  
            END WHILE  
            IF deque is not empty  
                SET max_value TO the greater value between max_value AND the sum of yj PLUS xj PLUS the first element of the pair at the front position one of deque  
            END IF  
            WHILE deque is not empty AND the difference between yj AND xj is greater than or equal to the first element of the pair at the last position one of deque  
                REMOVE the element at the last position one of deque  
            END WHILE  
            APPEND the pair consisting of the difference between yj AND xj AND xj TO deque  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
