CLASS Solution  
    FUNCTION generateParenthesis WITH parameter n  
        FUNCTION backtrack WITH parameters s defaulting to empty string left defaulting to zero right defaulting to zero  
            IF the LENGTH OF s EQUALS the MULTIPLICATION OF two AND n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left IS LESS THAN n  
                CALL backtrack WITH s CONCATENATED WITH opening parenthesis character left INCREMENTED BY one right  
            END IF  
            IF right IS LESS THAN left  
                CALL backtrack WITH s CONCATENATED WITH closing parenthesis character left right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH default parameters  
        RETURN result  
    END FUNCTION  
END CLASS
