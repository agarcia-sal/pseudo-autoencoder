CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS None right EQUALS None  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO A NEW empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node dot left TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node dot right TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p dot val DOES NOT EQUAL q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left AND q dot left AND is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalTraversal WITH PARAMETERS self root  
        IF root IS None  
            RETURN AN EMPTY list  
        END IF  
        SET node_dict TO A NEW dictionary THAT RETURNS AN EMPTY list FOR MISSING KEYS  
        SET min_col TO zero  
        SET max_col TO zero  

        FUNCTION dfs WITH PARAMETERS node row col  
            IF node IS None  
                RETURN  
            END IF  
            APPEND the tuple consisting of row AND node dot val TO the list AT key col IN node_dict  
            SET min_col TO THE SMALLER OF min_col AND col  
            SET max_col TO THE LARGER OF max_col AND col  
            CALL dfs WITH PARAMETERS node dot left row PLUS one col MINUS one  
            CALL dfs WITH PARAMETERS node dot right row PLUS one col PLUS one  
        END FUNCTION  

        CALL dfs WITH PARAMETERS root zero zero  

        SET result TO AN EMPTY list  
        FOR each col FROM min_col TO max_col inclusive  
            SET sorted_list TO THE LIST OF VALUES FROM node_dict AT key col SORTED BY FIRST ELEMENT THEN SECOND ELEMENT IN EACH tuple  
            SET column_values TO AN EMPTY list  
            FOR each tuple_in_sorted_list IN sorted_list  
                APPEND the second element of tuple_in_sorted_list TO column_values  
            END FOR  
            APPEND column_values TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
