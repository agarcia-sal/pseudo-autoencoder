CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path IS GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num  
                IF the element at position start of num EQUALS zero AND end IS GREATER THAN start PLUS one  
                    CONTINUE TO next iteration  
                END IF  
                SET next_num TO the integer VALUE OF substring from position start TO end MINUS one of num  
                IF next_num IS GREATER THAN OR EQUAL TO two RAISED TO THE POWER OF thirty ONE  
                    BREAK out of the loop  
                END IF  
                IF the LENGTH OF path IS LESS THAN two OR next_num EQUALS the sum of the element at the position before last in path PLUS the element at the position two before last in path  
                    SET result TO backtrack(end, a new list by APPENDING next_num TO path)  
                    IF result IS NOT empty  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN an empty list  
        END FUNCTION  
        RETURN backtrack(zero, an empty list)  
    END FUNCTION  
END CLASS
