CLASS Solution  
    FUNCTION largestSumOfAverages(nums PARAMETERS LIST OF INTEGERS, k PARAMETERS INTEGER) RETURNS FLOAT  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of n PLUS one elements all initialized TO zero  
        FOR index FROM zero TO n MINUS one  
            SET element at position index PLUS one of prefix_sum TO element at position index of prefix_sum PLUS element at position index of nums  
        END FOR  
        
        FUNCTION average(i PARAMETERS INTEGER, j PARAMETERS INTEGER) RETURNS FLOAT  
            RETURN element at position j PLUS one of prefix_sum MINUS element at position i of prefix_sum DIVIDED BY j MINUS i PLUS one  
        END FUNCTION  
        
        SET dp TO a two dimensional list of size n PLUS one BY k PLUS one all initialized TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AND one in dp TO CALL average WITH i AND n MINUS one  
        END FOR  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    SET element at position i AND partitions in dp TO the GREATER OF element at position i AND partitions in dp AND element at position j AND partitions MINUS one in dp PLUS CALL average WITH i AND j MINUS one  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero AND k in dp  
    END FUNCTION  
END CLASS
