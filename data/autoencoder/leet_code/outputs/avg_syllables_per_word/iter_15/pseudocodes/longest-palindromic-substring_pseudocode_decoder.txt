CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO a two dimensional collection of boolean values initially set to true for each position from zero TO n MINUS one WITH dimensions n BY n  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET the element at position i and position j of f TO false  
                IF the element at position i of s EQUALS the element at position j of s  
                    SET the element at position i and position j of f TO the element at position i PLUS one and position j MINUS one of f  
                    IF the element at position i and position j of f IS true AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the substring from position k TO k PLUS mx MINUS one of s  
    END FUNCTION  
END CLASS
