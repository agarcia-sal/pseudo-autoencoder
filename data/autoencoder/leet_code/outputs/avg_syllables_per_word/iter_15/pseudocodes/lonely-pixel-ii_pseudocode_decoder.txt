CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture IS EMPTY OR element at position zero of picture IS EMPTY  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO a list of zero repeated m times  
        SET col_count TO a list of zero repeated n times  
        SET row_patterns TO a new defaultdict of integer  
        
        FOR r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of all elements at position c from zero TO n MINUS one of element at position r of picture  
            INCREMENT value of row_patterns at key pattern BY one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS black pixel  
                    INCREMENT element at position r of row_count BY one  
                    INCREMENT element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS black pixel  
                    SET pattern TO the concatenation of all elements at position i from zero TO n MINUS one of element at position r of picture  
                    IF element at position r of row_count EQUALS target AND element at position c of col_count EQUALS target AND value of row_patterns at key pattern EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
