CLASS Solution  
    FUNCTION maxSizeSlices(slices)  
        SET n TO the LENGTH OF slices DIVIDED BY three  
        
        FUNCTION dp(start, end, k)  
            IF k EQUALS zero OR start GREATER THAN end  
                RETURN zero  
            END IF  
            SET pick TO element at position start of slices PLUS dp with start PLUS two and end and k MINUS one  
            SET skip TO dp with start PLUS one and end and k  
            IF pick GREATER THAN skip  
                RETURN pick  
            ELSE  
                RETURN skip  
            END IF  
        END FUNCTION  
        
        SET case1 TO dp with one and the LENGTH OF slices MINUS one and n  
        SET case2 TO dp with zero and the LENGTH OF slices MINUS two and n  
        
        IF case1 GREATER THAN case2  
            RETURN case1  
        ELSE  
            RETURN case2  
        END IF  
    END FUNCTION  
END CLASS
