CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values of type list)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode with value element at position zero of values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        REMOVE the first element FROM queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO NEW TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO NEW TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p IS NOT EQUAL TO value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(self, root of type Optional TreeNode, k of type integer) RETURNS integer  
        SET stack TO an empty list  
        SET current TO root  
        SET count TO zero  
        WHILE stack IS NOT EMPTY OR current IS NOT none  
            WHILE current IS NOT none  
                APPEND current TO stack  
                SET current TO left child of current  
            END WHILE  
            REMOVE the last element FROM stack AND ASSIGN TO current  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN value of current  
            END IF  
            SET current TO right child of current  
        END WHILE  
    END FUNCTION  
END CLASS
