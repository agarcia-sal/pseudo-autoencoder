CLASS Solution  
    FUNCTION minMoves2(nums)  
        CALL the METHOD sort ON nums TO arrange the elements in ascending order  
        SET median TO the element at position the LENGTH OF nums DIVIDED BY two of nums  
        SET moves TO zero  
        FOR each num IN nums  
            INCREMENT moves BY the ABSOLUTE VALUE OF num MINUS median  
        END FOR  
        RETURN moves  
    END FUNCTION  
END CLASS
