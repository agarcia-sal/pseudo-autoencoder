CLASS Solution  
    FUNCTION findKthNumber(n integer k integer) RETURNS integer  
        FUNCTION count_steps(prefix integer n integer) RETURNS integer  
            SET next_prefix TO prefix PLUS one  
            SET steps TO zero  
            WHILE prefix IS LESS THAN OR EQUAL TO n  
                INCREMENT steps BY THE MINIMUM OF the DIFFERENCE BETWEEN n AND prefix PLUS one AND THE DIFFERENCE BETWEEN next_prefix AND prefix  
                SET prefix TO prefix MULTIPLIED BY ten  
                SET next_prefix TO next_prefix MULTIPLIED BY ten  
            END WHILE  
            RETURN steps  
        END FUNCTION  

        SET current TO one  
        DECREMENT k BY one  
        WHILE k IS GREATER THAN zero  
            SET steps TO count_steps WITH current AND n  
            IF steps IS LESS THAN OR EQUAL TO k  
                INCREMENT current BY one  
                DECREMENT k BY steps  
            ELSE  
                SET current TO current MULTIPLIED BY ten  
                DECREMENT k BY one  
            END IF  
        END WHILE  
        RETURN current  
    END FUNCTION  
END CLASS
