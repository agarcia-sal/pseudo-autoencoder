CLASS Solution  
    FUNCTION countSubstrings with parameter s  
        FUNCTION expand_around_center with parameters left and right  
            SET count TO zero  
            WHILE left IS GREATER THAN OR EQUAL TO zero AND right IS LESS THAN the LENGTH OF s AND character at position left of s EQUALS character at position right of s  
                INCREMENT count BY one  
                DECREMENT left BY one  
                INCREMENT right BY one  
            END WHILE  
            RETURN count  
        END FUNCTION  
        SET total_palindromes TO zero  
        FOR index i FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT total_palindromes BY the RESULT OF expand_around_center with arguments index i and index i  
            INCREMENT total_palindromes BY the RESULT OF expand_around_center with arguments index i and index i PLUS one  
        END FOR  
        RETURN total_palindromes  
    END FUNCTION  
END CLASS
