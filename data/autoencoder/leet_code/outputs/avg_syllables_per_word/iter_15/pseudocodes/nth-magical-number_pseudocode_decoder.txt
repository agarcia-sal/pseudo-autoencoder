CLASS Solution  
    FUNCTION nthMagicalNumber(n, a, b)  
        SET MOD TO one billion PLUS seven  
        
        FUNCTION gcd(x, y)  
            WHILE the value of y IS NOT zero  
                SET temporary_x TO the value of y  
                SET temporary_y TO the remainder of x DIVIDED BY y  
                SET x TO the value of temporary_x  
                SET y TO the value of temporary_y  
            END WHILE  
            RETURN x  
        END FUNCTION  
        
        FUNCTION lcm(x, y)  
            RETURN x MULTIPLIED BY y DIVIDED BY the result of gcd with parameters x and y  
        END FUNCTION  
        
        SET L TO the result of lcm with parameters a and b  
        
        SET left TO one  
        SET right TO n MULTIPLIED BY the smaller value BETWEEN a AND b  
        
        WHILE left IS LESS THAN right  
            SET mid TO the integer division result of left PLUS right DIVIDED BY two  
            SET count TO mid DIVIDED BY a PLUS mid DIVIDED BY b MINUS mid DIVIDED BY L  
            
            IF count IS LESS THAN n  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN the remainder of left DIVIDED BY MOD  
    END FUNCTION  
END CLASS
