CLASS Solution  
    FUNCTION findLength(nums1, nums2)  
        SET dp TO a two dimensional collection of integers with the number of rows equal to the LENGTH OF nums1 PLUS one and the number of columns equal to the LENGTH OF nums2 PLUS one where every element is zero  
        SET max_length TO zero  
        FOR index i FROM one TO the LENGTH OF nums1 INCLUSIVE  
            FOR index j FROM one TO the LENGTH OF nums2 INCLUSIVE  
                IF the element at position i MINUS one of nums1 EQUALS the element at position j MINUS one of nums2  
                    SET the element at row i column j of dp TO the element at row i MINUS one column j MINUS one of dp PLUS one  
                    IF max_length LESS THAN the element at row i column j of dp  
                        SET max_length TO the element at row i column j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
