CLASS Solution  
    FUNCTION advantageCount(nums1 AS List OF integers, nums2 AS List OF integers) RETURNS List OF integers  
        SORT nums1 IN ASCENDING ORDER  
        
        SET nums2_with_indices TO SORTED list OF pairs of value and original position FROM nums2 IN DESCENDING ORDER BASED ON value  
        
        SET result TO a list OF zeros WITH LENGTH EQUAL TO the LENGTH OF nums1  
        
        SET left TO zero  
        SET right TO the LENGTH OF nums1 MINUS one  
        
        FOR each pair OF value and index IN nums2_with_indices  
            IF element at position right of nums1 IS GREATER THAN value THEN  
                SET element at position index of result TO element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index of result TO element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
