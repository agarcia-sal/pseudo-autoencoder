CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO an empty list  
        SET min_value TO positive infinity  
        FOR each num IN nums  
            IF the remainder of num divided by two EQUALS one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            SET min_value TO the smaller value BETWEEN min_value AND num  
            ADD the negative of num TO max_heap using a heap push operation  
        END FOR  
        SET min_deviation TO positive infinity  
        WHILE the element at position zero of max_heap modulo two EQUALS zero  
            SET max_value TO the negative of the heap pop of max_heap  
            SET min_deviation TO the smaller value BETWEEN min_deviation AND max_value MINUS min_value  
            SET max_value TO max_value DIVIDED BY two using integer division  
            SET min_value TO the smaller value BETWEEN min_value AND max_value  
            ADD the negative of max_value TO max_heap using a heap push operation  
        END WHILE  
        SET min_deviation TO the smaller value BETWEEN min_deviation AND the negative of the element at position zero of max_heap MINUS min_value  
        RETURN min_deviation  
    END FUNCTION  
END CLASS
