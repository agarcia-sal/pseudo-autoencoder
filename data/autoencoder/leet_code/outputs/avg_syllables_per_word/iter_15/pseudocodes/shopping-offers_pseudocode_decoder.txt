CLASS Solution  
    FUNCTION shoppingOffers(price LIST OF INTEGER, special LIST OF LIST OF INTEGER, needs LIST OF INTEGER) RETURNS INTEGER  
        SET special TO A NEW LIST CONTAINING each offer IN special WHERE the SUM OF element at position from zero TO the position BEFORE last of offer IS GREATER THAN zero AND the element at the last position of offer IS LESS THAN the SUM OF the PRODUCT OF corresponding elements of price AND offer from zero TO the position BEFORE last  
        
        FUNCTION dfs(needs TUPLE OF INTEGER) RETURNS INTEGER  
            SET min_cost TO the SUM OF the PRODUCT OF corresponding elements of price AND needs  
            FOR each offer IN special  
                SET new_needs TO A NEW LIST CONTAINING each element at position index of needs MINUS element at position index of offer from zero TO the position BEFORE last  
                IF ALL elements of new_needs ARE GREATER THAN OR EQUAL TO zero  
                    SET min_cost TO THE LOWER VALUE BETWEEN min_cost AND the element at the last position of offer PLUS dfs(tuple of new_needs)  
                END IF  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dfs(tuple of needs)  
END CLASS
