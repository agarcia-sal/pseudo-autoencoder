CLASS Solution  
    FUNCTION maxPoints(points)  
        FUNCTION slope(p1, p2)  
            IF the element at position zero of p1 EQUALS the element at position zero of p2  
                RETURN positive infinity representing an unbounded value  
            END IF  
            SET numerator TO the element at position one of p1 MINUS the element at position one of p2  
            SET denominator TO the element at position zero of p1 MINUS the element at position zero of p2  
            SET slope_value TO numerator DIVIDED BY denominator  
            RETURN slope_value  
        END FUNCTION  

        SET max_points TO zero  

        FOR each index i FROM zero TO the length of points MINUS one  
            INITIALIZE slopes AS a mapping FROM keys TO zero  
            FOR each index j FROM zero TO the length of points MINUS one  
                IF index i NOT EQUALS index j  
                    SET s TO the result of slope called with element at position i of points AND element at position j of points  
                    INCREMENT the value at key s in slopes BY one  
                    IF the value at key s in slopes GREATER THAN max_points  
                        SET max_points TO the value at key s in slopes  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        IF points IS NOT empty  
            RETURN max_points PLUS one  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
