CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the attribute val TO val  
        SET the attribute next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode OBJECT CREATED WITH element at position zero of values  
    SET p TO head  
    FOR each val IN THE SUBLIST OF values FROM position one TO the END  
        SET node TO A NEW ListNode OBJECT CREATED WITH val  
        SET the attribute next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN THE RESULT OF THE LOGICAL AND OPERATION BETWEEN the attribute val of p1 EQUALS the attribute val of p2 AND THE CALL OF FUNCTION is_same_list WITH ARGUMENTS the attribute next of p1 AND the attribute next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween WITH PARAMETERS head AND left AND right  
        IF head IS none OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO A NEW ListNode OBJECT CREATED WITH zero  
        SET the attribute next of dummy TO head  
        SET prev TO dummy  
        FOR each count FROM one TO left MINUS one  
            SET prev TO the attribute next of prev  
        END FOR  
        SET current TO the attribute next of prev  
        FOR each count FROM one TO right MINUS left  
            SET next_node TO the attribute next of current  
            SET the attribute next of current TO the attribute next of next_node  
            SET the attribute next of next_node TO the attribute next of prev  
            SET the attribute next of prev TO next_node  
        END FOR  
        RETURN the attribute next of dummy  
    END FUNCTION  
END CLASS
