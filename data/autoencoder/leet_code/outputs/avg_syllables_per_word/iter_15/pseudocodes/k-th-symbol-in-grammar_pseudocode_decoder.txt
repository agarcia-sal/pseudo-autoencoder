CLASS Solution  
    FUNCTION kthGrammar(n, k)  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET length_of_previous_row TO two RAISED TO THE POWER OF the value of n MINUS two  
        IF k LESS THAN OR EQUAL TO length_of_previous_row  
            RETURN the CALL TO the FUNCTION kthGrammar WITH the values n MINUS one AND k OF the current object  
        ELSE  
            RETURN one MINUS the CALL TO the FUNCTION kthGrammar WITH the values n MINUS one AND k MINUS length_of_previous_row OF the current object  
        END IF  
    END FUNCTION  
END CLASS
