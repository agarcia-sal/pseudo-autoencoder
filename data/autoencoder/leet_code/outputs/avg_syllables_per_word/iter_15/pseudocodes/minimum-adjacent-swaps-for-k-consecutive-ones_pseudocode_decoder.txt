CLASS Solution  
    FUNCTION minMoves(nums AS List of integers, k AS integer) RETURNS integer  
        SET positions TO empty list  
        FOR each index AND number IN nums WITH their positions  
            IF number EQUALS one  
                APPEND index TO positions  
            END IF  
        END FOR  

        FUNCTION calculate_cost(start AS integer, end AS integer) RETURNS integer  
            SET mid TO the integer division result of start PLUS end BY two  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR each index FROM start TO end MINUS one  
                INCREMENT cost BY the absolute value of the element at position index of positions MINUS the result of median MINUS the result of mid MINUS index  
            END FOR  
            RETURN cost  
        END FUNCTION  

        SET min_cost TO positive infinity  
        FOR each index FROM zero TO the result of the length of positions MINUS k  
            SET current_cost TO the result of calculate_cost with parameters index AND index PLUS k  
            SET min_cost TO the smaller value between min_cost AND current_cost  
        END FOR  

        RETURN min_cost  
    END FUNCTION  
END CLASS
