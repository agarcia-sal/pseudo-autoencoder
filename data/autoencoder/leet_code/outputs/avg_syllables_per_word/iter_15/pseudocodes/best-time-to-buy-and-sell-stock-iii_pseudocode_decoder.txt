CLASS Solution  
    FUNCTION maxProfit(prices)  
        IF prices EQUALS nothing OR the LENGTH OF prices EQUALS zero  
            RETURN zero  
        END IF  
        SET min_price1 TO positive infinity  
        SET min_price2 TO positive infinity  
        SET max_profit1 TO zero  
        SET max_profit2 TO zero  
        FOR each price IN prices  
            IF min_price1 GREATER THAN price  
                SET min_price1 TO price  
            END IF  
            IF max_profit1 LESS THAN price MINUS min_price1  
                SET max_profit1 TO price MINUS min_price1  
            END IF  
            IF min_price2 GREATER THAN price MINUS max_profit1  
                SET min_price2 TO price MINUS max_profit1  
            END IF  
            IF max_profit2 LESS THAN price MINUS min_price2  
                SET max_profit2 TO price MINUS min_price2  
            END IF  
        END FOR  
        RETURN max_profit2  
    END FUNCTION  
END CLASS
