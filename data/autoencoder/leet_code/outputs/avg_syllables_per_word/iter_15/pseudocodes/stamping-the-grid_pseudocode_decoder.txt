CLASS Solution  
    FUNCTION possibleToStamp(grid, stampHeight, stampWidth)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero of grid  
        
        FUNCTION prefix_sum(matrix)  
            SET p TO a new matrix of zeros WITH dimensions m PLUS one BY n PLUS one  
            FOR r FROM one TO m  
                FOR c FROM one TO n  
                    SET element AT position r AND c OF p TO element at position r MINUS one AND c MINUS one OF matrix PLUS element AT position r MINUS one AND c OF p PLUS element AT position r AND c MINUS one OF p MINUS element AT position r MINUS one AND c MINUS one OF p  
                END FOR  
            END FOR  
            RETURN p  
        END FUNCTION  
        
        FUNCTION submatrix_sum(p, r1, c1, r2, c2)  
            RETURN element AT position r2 PLUS one AND c2 PLUS one OF p MINUS element AT position r2 PLUS one AND c1 OF p MINUS element AT position r1 AND c2 PLUS one OF p PLUS element AT position r1 AND c1 OF p  
        END FUNCTION  
        
        SET p_grid TO the RESULT OF prefix_sum WITH argument grid  
        
        SET p_stamps TO a new matrix of zeros WITH dimensions m PLUS one BY n PLUS one  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position r AND c OF grid EQUALS zero THEN  
                    IF r PLUS stampHeight IS LESS THAN OR EQUAL TO m AND c PLUS stampWidth IS LESS THAN OR EQUAL TO n THEN  
                        IF submatrix_sum WITH arguments p_grid, r, c, r PLUS stampHeight MINUS one, c PLUS stampWidth MINUS one EQUALS zero THEN  
                            INCREMENT element at position r AND c OF p_stamps BY one  
                        END IF  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET p_stamps TO the RESULT OF prefix_sum WITH argument p_stamps  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position r AND c OF grid EQUALS zero THEN  
                    IF submatrix_sum WITH arguments p_stamps, the maximum of zero AND r MINUS stampHeight PLUS one, the maximum of zero AND c MINUS stampWidth PLUS one, r, c EQUALS zero THEN  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
