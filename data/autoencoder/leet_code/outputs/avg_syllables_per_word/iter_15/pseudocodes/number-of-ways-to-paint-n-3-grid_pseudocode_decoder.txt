CLASS Solution  
    FUNCTION numOfWays with parameter n  
        SET MODULO TO one billion PLUS seven  
        SET variable a TO six  
        SET variable b TO six  
        FOR index FROM two TO n  
            SET temporary_a TO the result of three MULTIPLIED BY a PLUS two MULTIPLIED BY b MODULO MODULO  
            SET temporary_b TO the result of two MULTIPLIED BY a PLUS two MULTIPLIED BY b MODULO MODULO  
            SET a TO temporary_a  
            SET b TO temporary_b  
        END FOR  
        SET result TO the result of a PLUS b MODULO MODULO  
        RETURN result  
    END FUNCTION  
END CLASS
