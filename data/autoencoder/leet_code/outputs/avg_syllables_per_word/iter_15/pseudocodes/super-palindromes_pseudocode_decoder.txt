CLASS Solution  
    FUNCTION superpalindromesInRange(left AS string, right AS string) RETURNS integer  
        FUNCTION is_palindrome(num) RETURNS boolean  
            RETURN the STRING representation of num EQUALS the STRING representation of num REVERSED  
        END FUNCTION  
        
        FUNCTION generate_palindromes(limit) RETURNS list of integers  
            SET palindromes TO an empty list  
            FOR i FROM one TO one hundred thousand MINUS one  
                SET s TO the STRING representation of i  
                APPEND the integer value of s CONCATENATED WITH the substring of s FROM the second last character TO the first character IN REVERSE ORDER TO palindromes  
                APPEND the integer value of s CONCATENATED WITH s REVERSED TO palindromes  
            END FOR  
            RETURN all elements p IN palindromes WHERE p IS LESS THAN OR EQUAL TO limit  
        END FUNCTION  
        
        SET left TO the integer value of left  
        SET right TO the integer value of right  
        SET limit TO the integer value of the square root of right PLUS one  
        SET palindromes TO generate_palindromes(limit)  
        
        SET count TO zero  
        FOR each p IN palindromes  
            SET square TO p MULTIPLIED BY p  
            IF left IS LESS THAN OR EQUAL TO square AND square IS LESS THAN OR EQUAL TO right AND is_palindrome(square) IS true  
                INCREMENT count BY one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
