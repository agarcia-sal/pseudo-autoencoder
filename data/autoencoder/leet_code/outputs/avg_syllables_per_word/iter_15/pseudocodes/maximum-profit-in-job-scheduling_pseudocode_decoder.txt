CLASS Solution  
    FUNCTION jobScheduling(startTime list OF INTEGER endTime list OF INTEGER profit list OF INTEGER)  
        SET jobs TO the LIST OF triples COMPOSED OF corresponding elements of startTime endTime and profit SORTED BY the second element of each triple in ascending order  
        SET end_times TO LIST OF the second elements of each job in jobs  
        SET dp TO LIST OF zero VALUES with LENGTH OF jobs PLUS one  
        FOR i FROM one TO the LENGTH OF jobs  
            SET j TO the POSITION WHERE the element at position i MINUS one of jobs FIRST element WOULD BE INSERTED IN end_times MAINTAINING ASCENDING ORDER  
            SET dp AT position i TO THE GREATER OF dp AT position i MINUS one AND THE SUM OF dp AT position j AND the third element of the job at position i MINUS one of jobs  
        END FOR  
        RETURN the LAST element of dp  
    END FUNCTION  
END CLASS
