CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUALS the character plus OR token EQUALS the character minus OR token EQUALS the character multiplied by OR token EQUALS the character divided by  
                SET b TO the last element REMOVED FROM stack  
                SET a TO the last element REMOVED FROM stack  
                IF token EQUALS the character plus  
                    APPEND a PLUS b TO stack  
                ELSE IF token EQUALS the character minus  
                    APPEND a MINUS b TO stack  
                ELSE IF token EQUALS the character multiplied by  
                    APPEND a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS the character divided by  
                    APPEND integer part of a DIVIDED BY b TO stack  
                END IF  
            ELSE  
                APPEND integer value of token TO stack  
            END IF  
        END FOR  
        RETURN the element at position zero of stack  
    END FUNCTION  
END CLASS
