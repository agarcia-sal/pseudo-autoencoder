CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours AND minutes TO the result of converting the parts of time separated by the colon to integers  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
        
        SET minutes_list TO the list created by applying to_minutes to each element in timePoints  
        
        SORT minutes_list in ascending order  
        
        SET min_diff TO a very large number representing infinity  
        
        FOR index FROM one TO the LENGTH OF minutes_list MINUS one  
            SET current_difference TO the element at position index of minutes_list MINUS the element at position index MINUS one of minutes_list  
            SET min_diff TO the smaller value between min_diff AND current_difference  
        END FOR  
        
        SET circular_difference TO the sum of subtracting the element at the last position of minutes_list FROM one thousand four hundred forty AND the element at position zero of minutes_list  
        
        SET min_diff TO the smaller value between min_diff AND circular_difference  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
