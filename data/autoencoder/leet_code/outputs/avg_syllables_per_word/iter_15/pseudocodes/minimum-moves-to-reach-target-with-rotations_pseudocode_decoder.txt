CLASS Solution  
    FUNCTION minimumMoves(grid)  
        SET n TO the LENGTH OF grid  
        ASSIGN pointer/reference TO a double ended queue initialized with a tuple containing zero zero zero one zero  
        ASSIGN pointer/reference TO a set containing a tuple of zero zero zero one named visited  
        
        WHILE the dequeued queue is not empty  
            REMOVE the first element FROM queue and ASSIGN to r1 c1 r2 c2 steps  
            
            IF r1 EQUALS n MINUS one AND c1 EQUALS n MINUS two AND r2 EQUALS n MINUS one AND c2 EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            IF c2 PLUS one LESS THAN n AND element at position r1 of grid at position c1 PLUS one EQUALS zero AND element at position r2 of grid at position c2 PLUS one EQUALS zero AND the tuple r1 c1 PLUS one r2 c2 PLUS one NOT IN visited  
                ADD the tuple r1 c1 PLUS one r2 c2 PLUS one TO visited  
                APPEND the tuple r1 c1 PLUS one r2 c2 PLUS one steps PLUS one TO queue  
            END IF  
            
            IF r2 PLUS one LESS THAN n AND element at position r1 PLUS one of grid at position c1 EQUALS zero AND element at position r2 PLUS one of grid at position c2 EQUALS zero AND the tuple r1 PLUS one c1 r2 PLUS one c2 NOT IN visited  
                ADD the tuple r1 PLUS one c1 r2 PLUS one c2 TO visited  
                APPEND the tuple r1 PLUS one c1 r2 PLUS one c2 steps PLUS one TO queue  
            END IF  
            
            IF r1 EQUALS r2 AND r1 PLUS one LESS THAN n AND element at position r1 PLUS one of grid at position c1 EQUALS zero AND element at position r1 PLUS one of grid at position c2 EQUALS zero AND the tuple r1 c1 r1 PLUS one c1 NOT IN visited  
                ADD the tuple r1 c1 r1 PLUS one c1 TO visited  
                APPEND the tuple r1 c1 r1 PLUS one c1 steps PLUS one TO queue  
            END IF  
            
            IF c1 EQUALS c2 AND c1 PLUS one LESS THAN n AND element at position r1 of grid at position c1 PLUS one EQUALS zero AND element at position r2 of grid at position c2 PLUS one EQUALS zero AND the tuple r1 c1 r1 c1 PLUS one NOT IN visited  
                ADD the tuple r1 c1 r1 c1 PLUS one TO visited  
                APPEND the tuple r1 c1 r1 c1 PLUS one steps PLUS one TO queue  
            END IF  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
