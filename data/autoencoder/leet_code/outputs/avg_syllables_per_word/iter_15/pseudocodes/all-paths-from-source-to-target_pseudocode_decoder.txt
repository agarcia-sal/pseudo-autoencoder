CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        SET q TO a new double ended queue initialized with a list containing zero  
        SET ans TO an empty list  
        WHILE the double ended queue q is not empty  
            SET path TO the element removed from the left end of the double ended queue q  
            SET u TO the element at the last position of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            FOR each v IN the element at position u of graph  
                APPEND a new list obtained by concatenating path and a list containing v TO the double ended queue q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
