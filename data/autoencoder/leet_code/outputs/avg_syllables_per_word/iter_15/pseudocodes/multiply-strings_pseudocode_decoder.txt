CLASS Solution  
    FUNCTION multiply(num1 string, num2 string) RETURNS string  
        IF num1 EQUALS string zero OR num2 EQUALS string zero  
            RETURN string zero  
        END IF  
        SET result TO list OF zero repeated the sum of the LENGTH OF num1 AND the LENGTH OF num2 times  
        SET num1 TO the reverse order of characters of num1  
        SET num2 TO the reverse order of characters of num2  
        FOR each index i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR each index j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the difference between the character code at position i of num1 AND the character code of character zero MULTIPLIED BY the difference between the character code at position j of num2 AND the character code of character zero  
                SET p1 TO the sum of i AND j  
                SET p2 TO the sum of i AND j AND one  
                SET sum_ TO the sum of mul AND the element at position p1 of result  
                SET the element at position p1 of result TO the remainder of the division of sum_ BY ten  
                INCREMENT the element at position p2 of result BY the quotient of the division of sum_ BY ten  
            END FOR  
        END FOR  
        SET result_str TO the concatenation of all elements of result in reverse order converted to string  
        RETURN result_str after removing all leading characters equal to zero character  
    END FUNCTION  
END CLASS
