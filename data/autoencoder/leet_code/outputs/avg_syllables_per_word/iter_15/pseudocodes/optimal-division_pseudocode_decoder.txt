CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position zero of nums converted to string  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the element at position zero of nums converted to string CONCATENATED WITH the string slash CONCATENATED WITH the element at position one of nums converted to string  
        END IF  
        RETURN the element at position zero of nums converted to string CONCATENATED WITH the string slash CONCATENATED WITH the string left parenthesis CONCATENATED WITH the result of joining the string versions of elements from position one to the end of nums separated by the string slash CONCATENATED WITH the string right parenthesis  
    END FUNCTION  
END CLASS
