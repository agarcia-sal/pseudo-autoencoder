CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS None  
                RETURN zero  
            END IF  
            SET x1 TO the QUOTIENT OF the VALUE OF the UNICODE CODE OF c1 MINUS the UNICODE CODE OF uppercase letter A DIVIDED BY six  
            SET y1 TO the REMAINDER OF the VALUE OF the UNICODE CODE OF c1 MINUS the UNICODE CODE OF uppercase letter A DIVIDED BY six  
            SET x2 TO the QUOTIENT OF the VALUE OF the UNICODE CODE OF c2 MINUS the UNICODE CODE OF uppercase letter A DIVIDED BY six  
            SET y2 TO the REMAINDER OF the VALUE OF the UNICODE CODE OF c2 MINUS the UNICODE CODE OF uppercase letter A DIVIDED BY six  
            RETURN the ABSOLUTE VALUE OF x1 MINUS x2 PLUS the ABSOLUTE VALUE OF y1 MINUS y2  
        END FUNCTION  

        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the LENGTH OF word  
                RETURN zero  
            END IF  
            SET c TO the ELEMENT AT POSITION i OF word  
            SET d1 TO the RESULT OF distance GIVEN f1 AND c PLUS the RESULT OF dp GIVEN i PLUS one AND c AND f2  
            SET d2 TO the RESULT OF distance GIVEN f2 AND c PLUS the RESULT OF dp GIVEN i PLUS one AND f1 AND c  
            RETURN the MINIMUM OF d1 AND d2  
        END FUNCTION  

        RETURN the RESULT OF dp GIVEN zero AND None AND None  
    END FUNCTION  
END CLASS
