CLASS Solution  
    FUNCTION minStickers(stickers AS List OF String, target AS String) RETURNS Integer  
        SET stickers TO A LIST THAT CONTAINS THE FREQUENCY COUNTER OBJECT FOR EACH sticker IN stickers  
        SORT stickers BY THE NEGATIVE VALUE OF THE HIGHEST FREQUENCY VALUE OF EACH COUNTER OBJECT  
        ASSIGN queue TO A DOUBLE-ENDED QUEUE INITIALIZED WITH A PAIR OF target AND ZERO  
        ASSIGN memo TO A MAPPING WITH target AS KEY AND ZERO AS VALUE  
        WHILE queue IS NOT EMPTY  
            ASSIGN current_target AND used_stickers TO THE FIRST ELEMENT REMOVED FROM queue  
            FOR each sticker IN stickers  
                IF THE FIRST CHARACTER OF current_target IS NOT PRESENT IN sticker  
                    CONTINUE TO THE NEXT ITERATION OF THE LOOP  
                END IF  
                ASSIGN remaining TO A FREQUENCY COUNTER OBJECT OF current_target  
                SUBTRACT THE FREQUENCY COUNTER OF sticker FROM remaining  
                ASSIGN new_target TO A STRING COMPOSED BY JOINING THE SORTED ELEMENTS OF remaining  
                IF new_target IS EMPTY  
                    RETURN used_stickers PLUS ONE  
                END IF  
                IF new_target IS NOT PRESENT AS A KEY IN memo  
                    SET memo AT new_target TO used_stickers PLUS ONE  
                    APPEND THE PAIR OF new_target AND used_stickers PLUS ONE TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN NEGATIVE ONE  
    END FUNCTION  
END CLASS
