CLASS Solution  
    FUNCTION multiply(mat1 AS List OF List OF integer, mat2 AS List OF List OF integer) RETURNS List OF List OF integer  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF element at position zero of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF element at position zero of mat2  
        SET result TO a new list of lists of integers with m rows and n columns each initialized to zero  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                FOR index p FROM zero TO k MINUS one  
                    SET element at position i of result at position j TO element at position i of mat1 at position p MULTIPLIED BY element at position p of mat2 at position j PLUS element at position i of result at position j  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
