CLASS Solution  
    FUNCTION numPoints(darts, r)  
        FUNCTION distance(p1, p2)  
            SET first_coordinate_difference TO element at position zero of p1 MINUS element at position zero of p2  
            SET second_coordinate_difference TO element at position one of p1 MINUS element at position one of p2  
            SET square_of_first_difference TO first_coordinate_difference MULTIPLIED BY first_coordinate_difference  
            SET square_of_second_difference TO second_coordinate_difference MULTIPLIED BY second_coordinate_difference  
            SET sum_of_squares TO square_of_first_difference PLUS square_of_second_difference  
            SET result TO the square root of sum_of_squares  
            RETURN result  
        END FUNCTION  
        
        FUNCTION circle_center(p1, p2, r)  
            SET d TO distance of p1 and p2  
            IF d GREATER THAN two MULTIPLIED BY r  
                RETURN none  
            END IF  
            SET half_distance_squared TO d DIVIDED BY two MULTIPLIED BY d DIVIDED BY two  
            SET radius_squared TO r MULTIPLIED BY r  
            SET a TO the square root of radius_squared MINUS half_distance_squared  
            SET midpoint TO list composed of element at position zero of p1 PLUS element at position zero of p2 DIVIDED BY two AND element at position one of p1 PLUS element at position one of p2 DIVIDED BY two  
            SET dx TO element at position one of p2 MINUS element at position one of p1 MULTIPLIED BY a DIVIDED BY d  
            SET dy TO element at position zero of p2 MINUS element at position zero of p1 MULTIPLIED BY a DIVIDED BY d  
            SET first_center TO tuple composed of element at position zero of midpoint MINUS dx AND element at position one of midpoint PLUS dy  
            SET second_center TO tuple composed of element at position zero of midpoint PLUS dx AND element at position one of midpoint MINUS dy  
            RETURN list composed of first_center AND second_center  
        END FUNCTION  
        
        FUNCTION is_inside(circle, point, r)  
            SET distance_between_circle_and_point TO distance of circle and point  
            IF distance_between_circle_and_point LESS THAN OR EQUAL TO r  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET n TO the LENGTH OF darts  
        IF n EQUALS one  
            RETURN one  
        END IF  
        
        SET max_count TO one  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET centers TO circle_center of element at position i of darts AND element at position j of darts AND r  
                IF centers IS NOT none  
                    FOR each center IN centers  
                        SET count TO zero  
                        FOR each dart IN darts  
                            IF is_inside of center AND dart AND r IS true  
                                INCREMENT count BY one  
                            END IF  
                        END FOR  
                        IF count GREATER THAN max_count  
                            SET max_count TO count  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_count  
    END FUNCTION  
END CLASS
