CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the TUPLE of index AND current_sum EXISTS AS A KEY IN memo  
                RETURN the VALUE AT the TUPLE of index AND current_sum IN memo  
            END IF  
            SET positive TO the CALL TO dfs with index INCREMENTED BY one AND current_sum INCREMENTED BY the element at position index of nums AND memo  
            SET negative TO the CALL TO dfs with index INCREMENTED BY one AND current_sum DECREMENTED BY the element at position index of nums AND memo  
            SET the VALUE AT the TUPLE of index AND current_sum IN memo TO positive PLUS negative  
            RETURN the VALUE AT the TUPLE of index AND current_sum IN memo  
        END FUNCTION  
        RETURN the CALL TO dfs with zero FOR index AND zero FOR current_sum AND an EMPTY dictionary FOR memo  
    END FUNCTION  
END CLASS
