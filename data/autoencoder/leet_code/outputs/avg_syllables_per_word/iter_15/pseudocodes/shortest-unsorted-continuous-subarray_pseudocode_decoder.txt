CLASS Solution  
    FUNCTION findUnsortedSubarray(nums)  
        SET sorted_nums TO the ARRAY OF nums SORTED IN NON DECREASING ORDER  
        SET start TO negative one  
        SET end TO negative two  
        FOR each index i FROM zero TO the LENGTH OF nums MINUS one  
            IF element at position i of nums IS NOT EQUALS TO element at position i of sorted_nums  
                IF start EQUALS TO negative one  
                    SET start TO i  
                END IF  
                SET end TO i  
            END IF  
        END FOR  
        SET result_length TO end MINUS start PLUS one  
        RETURN result_length  
    END FUNCTION  
END CLASS
