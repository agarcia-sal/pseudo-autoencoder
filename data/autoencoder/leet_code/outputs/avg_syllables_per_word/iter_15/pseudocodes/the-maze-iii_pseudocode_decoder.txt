CLASS Solution  
    FUNCTION findShortestWay WITH PARAMETERS maze AS list of list of integers, ball AS list of integers, hole AS list of integers  
        SET directions TO list OF tuples each containing zero AND one AND the character r, one AND zero AND the character d, zero AND minus one AND the character l, minus one AND zero AND the character u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero OF maze  
        SET start TO the tuple constructed from ball  
        SET hole TO the tuple constructed from hole  
        SET pq TO a list containing a tuple composed of zero AS distance AND an empty string AS path AND start AS position  
        SET visited TO an empty set  
        WHILE pq IS NOT empty  
            REMOVE and RETURN the element WITH the smallest distance FROM pq AS dist, path, AND position unpacked as x AND y  
            IF the tuple composed of x AND y EQUALS hole  
                RETURN path  
            END IF  
            IF the tuple composed of x AND y IS IN visited  
                CONTINUE to the next iteration  
            END IF  
            ADD the tuple composed of x AND y TO visited  
            FOR each tuple dx, dy, direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the element at position new_x PLUS dx OF maze AT position new_y PLUS dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                    IF the tuple composed of new_x AND new_y EQUALS hole  
                        BREAK the loop  
                    END IF  
                END WHILE  
                IF the tuple composed of new_x AND new_y IS NOT IN visited  
                    ADD the tuple composed of dist PLUS count, the concatenation of path AND direction, AND the tuple composed of new_x AND new_y TO pq using the heap push operation  
                END IF  
            END FOR  
        END WHILE  
        RETURN the string impossible  
    END FUNCTION  
END CLASS
