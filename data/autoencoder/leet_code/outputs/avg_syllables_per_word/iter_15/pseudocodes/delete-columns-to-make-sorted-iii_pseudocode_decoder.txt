CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero of strs  
        SET dp TO a list of the LENGTH OF n with all elements equal to one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF FOR every k FROM zero TO m MINUS one the element at position k of element at position j of strs IS LESS THAN OR EQUAL TO the element at position k of element at position i of strs  
                    SET dp at position i TO the GREATER OF dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        RETURN n MINUS the GREATEST element IN dp  
    END FUNCTION  
END CLASS
