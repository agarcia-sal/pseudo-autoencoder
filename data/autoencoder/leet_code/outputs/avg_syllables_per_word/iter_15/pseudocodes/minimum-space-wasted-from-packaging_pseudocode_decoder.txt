CLASS Solution  
    FUNCTION minWastedSpace(packages AS List of integers, boxes AS List of List of integers) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        SORT the list packages IN ascending order  
        SET n TO the LENGTH OF packages  
        
        SET prefix_sum TO a list of zeros with LENGTH EQUAL TO n PLUS one  
        FOR index FROM zero TO n MINUS one  
            SET the element at position index PLUS one of prefix_sum TO the element at position index of prefix_sum PLUS the element at position index of packages  
        END FOR  
        
        SET min_wasted_space TO positive infinity  
        
        FOR each box IN boxes  
            SORT the list box IN ascending order  
            IF the element at the last position of box IS LESS THAN the element at the last position of packages THEN  
                CONTINUE to the next iteration of the FOR loop  
            END IF  
            
            SET total_wasted TO zero  
            SET last_index TO zero  
            
            FOR each box_size IN box  
                SET index TO last_index  
                WHILE index IS LESS THAN n AND the element at position index of packages IS LESS THAN OR EQUAL TO box_size  
                    INCREMENT index BY one  
                END WHILE  
                
                IF index IS GREATER THAN last_index THEN  
                    SET total_wasted TO total_wasted PLUS the PRODUCT of the difference between index AND last_index AND box_size MINUS the difference between the element at position index of prefix_sum AND the element at position last_index of prefix_sum  
                    SET last_index TO index  
                END IF  
            END FOR  
            
            IF last_index EQUALS n THEN  
                SET min_wasted_space TO the MINIMUM OF min_wasted_space AND total_wasted  
            END IF  
        END FOR  
        
        IF min_wasted_space IS NOT EQUAL TO positive infinity THEN  
            RETURN min_wasted_space MODULO MOD  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
