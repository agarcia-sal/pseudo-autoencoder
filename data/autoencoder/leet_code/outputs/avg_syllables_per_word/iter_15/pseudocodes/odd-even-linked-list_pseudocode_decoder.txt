CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero next EQUAL TO none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode WITH PARAMETER element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO the end  
        SET node TO A NEW ListNode WITH PARAMETER val  
        SET the element at position next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the value at p1 EQUALS the value at p2 AND is_same_list WITH PARAMETERS the element at position next of p1 the element at position next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList WITH PARAMETERS self head  
        IF head IS none OR the element at position next of head IS none OR the element at position next of the element at position next of head IS none  
            RETURN head  
        END IF  

        SET odd TO head  
        SET even TO the element at position next of head  
        SET even_head TO even  

        WHILE even IS NOT none AND the element at position next of even IS NOT none  
            SET the element at position next of odd TO the element at position next of even  
            SET odd TO the element at position next of odd  
            SET the element at position next of even TO the element at position next of odd  
            SET even TO the element at position next of even  
        END WHILE  

        SET the element at position next of odd TO even_head  

        RETURN head  
    END FUNCTION  
END CLASS
