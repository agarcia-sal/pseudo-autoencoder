CLASS Solution  
    FUNCTION expressiveWords WITH PARAMETERS s AND words  
        FUNCTION compress WITH PARAMETER word  
            IF word EQUALS an empty sequence  
                RETURN an empty list  
            END IF  
            SET count TO one  
            SET compressed TO an empty list  
            FOR index FROM one TO the LENGTH OF word MINUS one  
                IF element at position index of word EQUALS element at position index MINUS one of word  
                    INCREMENT count BY one  
                ELSE  
                    APPEND a pair of element at position index MINUS one of word AND count TO compressed  
                    SET count TO one  
                END IF  
            END FOR  
            APPEND a pair of element at the last position of word AND count TO compressed  
            RETURN compressed  
        END FUNCTION  
  
        FUNCTION is_stretchy WITH PARAMETERS s_compressed AND word_compressed  
            IF the LENGTH OF s_compressed NOT EQUALS the LENGTH OF word_compressed  
                RETURN false  
            END IF  
            FOR each pair of elements sc and sc_count IN s_compressed AND wc and wc_count IN word_compressed taken correspondingly  
                IF sc NOT EQUALS wc  
                    RETURN false  
                END IF  
                IF sc_count NOT EQUALS wc_count AND (sc_count LESS THAN three OR sc_count LESS THAN wc_count)  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
  
        SET s_compressed TO the result of compress WITH ARGUMENT s  
        SET stretchy_count TO zero  
  
        FOR each word IN words  
            SET word_compressed TO the result of compress WITH ARGUMENT word  
            IF is_stretchy WITH ARGUMENTS s_compressed AND word_compressed  
                INCREMENT stretchy_count BY one  
            END IF  
        END FOR  
  
        RETURN stretchy_count  
    END FUNCTION  
END CLASS
