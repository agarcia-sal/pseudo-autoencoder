CLASS Solution  
    FUNCTION canSeePersonsCount(heights)  
        SET n TO the LENGTH OF heights  
        SET answer TO a list of zeros with LENGTH n  
        SET stack TO an empty list  
        FOR index i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position element at the last position of stack of heights IS LESS THAN the element at position i of heights  
                INCREMENT the element at position the element REMOVED FROM THE last position of stack in answer BY one  
            END WHILE  
            IF the LENGTH OF stack IS GREATER THAN zero  
                INCREMENT the element at position the element at the last position of stack in answer BY one  
            END IF  
            APPEND i TO stack  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
