CLASS Solution  
    FUNCTION getXORSum(arr1 list of integers arr2 list of integers) RETURNS integer  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        FOR each num IN arr1  
            SET xor_sum_arr1 TO the BITWISE XOR of xor_sum_arr1 WITH num  
        END FOR  
        FOR each num IN arr2  
            SET xor_sum_arr2 TO the BITWISE XOR of xor_sum_arr2 WITH num  
        END FOR  
        RETURN the BITWISE AND of xor_sum_arr1 WITH xor_sum_arr2  
    END FUNCTION  
END CLASS
