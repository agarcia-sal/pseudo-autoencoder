CLASS Solution  
    FUNCTION minSwap(nums1 LIST OF INTEGER nums2 LIST OF INTEGER) RETURNS INTEGER  
        SET n TO the LENGTH OF nums1  
        SET dp TO a list of n elements each consisting of a list of two zeros representing minimum swaps without swap and with swap respectively at each position  
        SET element at position zero and zero of dp TO zero representing no swap needed for the first element  
        SET element at position zero and one of dp TO one representing one swap needed for the first element  
        FOR index i FROM one TO n MINUS one  
            IF element at position i of nums1 GREATER THAN element at position i MINUS one of nums1 AND element at position i of nums2 GREATER THAN element at position i MINUS one of nums2 THEN  
                IF element at position i of nums1 GREATER THAN element at position i MINUS one of nums2 AND element at position i of nums2 GREATER THAN element at position i MINUS one of nums1 THEN  
                    SET element at position i and zero of dp TO the MINIMUM OF element at position i MINUS one and zero of dp AND element at position i MINUS one and one of dp  
                    SET element at position i and one of dp TO element at position i and zero of dp PLUS one  
                ELSE  
                    SET element at position i and zero of dp TO element at position i MINUS one and zero of dp  
                    SET element at position i and one of dp TO element at position i MINUS one and one of dp PLUS one  
                END IF  
            ELSE  
                IF element at position i of nums1 GREATER THAN element at position i MINUS one of nums2 AND element at position i of nums2 GREATER THAN element at position i MINUS one of nums1 THEN  
                    SET element at position i and zero of dp TO element at position i MINUS one and one of dp  
                    SET element at position i and one of dp TO element at position i MINUS one and zero of dp PLUS one  
                ELSE  
                    SET element at position i and zero of dp TO element at position i MINUS one and one of dp  
                    SET element at position i and one of dp TO element at position i MINUS one and zero of dp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN the MINIMUM OF element at position n MINUS one and zero of dp AND element at position n MINUS one and one of dp  
    END FUNCTION  
END CLASS
