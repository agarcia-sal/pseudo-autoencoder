CLASS Solution  
    FUNCTION findSubstring with parameters s and words  
        IF s EQUALS not AVAILABLE OR words EQUALS not AVAILABLE OR element at position zero of words EQUALS not AVAILABLE  
            RETURN an EMPTY list  
        END IF  
        
        SET word_length TO the LENGTH OF element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the RESULT OF Counter FUNCTION APPLIED TO words  
        SET result TO an EMPTY list  
        
        FOR index i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO the RESULT OF Counter FUNCTION APPLIED TO an EMPTY COLLECTION  
            FOR index j FROM i TO i PLUS total_length MINUS one IN STEPS OF word_length  
                SET word TO the SUBSTRING from position j TO j PLUS word_length MINUS one OF s  
                INCREMENT the COUNT OF word IN seen BY one  
                IF the COUNT OF word IN seen GREATER THAN the COUNT OF word IN word_count  
                    EXIT the INNER LOOP  
                END IF  
            END FOR  
            IF the INNER LOOP WAS NOT EXITED ABRUPTLY  
                APPEND i TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
