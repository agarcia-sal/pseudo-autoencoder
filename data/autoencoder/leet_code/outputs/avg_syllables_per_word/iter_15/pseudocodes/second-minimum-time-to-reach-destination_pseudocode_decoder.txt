CLASS Solution  
    FUNCTION secondMinimum(n, edges, time, change)  
        SET graph TO a list of empty lists with length equal to n PLUS one  
        FOR each pair u AND v IN edges  
            APPEND v TO the element at position u of graph  
            APPEND u TO the element at position v of graph  
        END FOR  

        SET queue TO a double ended queue initialized with a tuple of one, zero, zero representing the current vertex, current time, and count of shortest or second shortest paths  
        SET minTimes TO a list of lists where each sublist contains two elements each set to positive infinity with length equal to n PLUS one  
        SET the element at position one and zero of minTimes TO zero  

        WHILE queue holds elements  
            REMOVE and ASSIGN the leftmost tuple of queue TO current, currentTime, count  

            IF current EQUALS n AND count EQUALS one  
                RETURN currentTime  
            END IF  

            IF the integer division of currentTime BY change MODULO two EQUALS zero  
                SET nextTime TO currentTime PLUS time  
            ELSE  
                SET nextTime TO the product of the integer division of currentTime BY change PLUS one AND change PLUS time  
            END IF  

            FOR each neighbor IN the element at position current of graph  
                IF the element at position neighbor and zero of minTimes GREATER THAN nextTime  
                    SET the element at position neighbor and one of minTimes TO the element at position neighbor and zero of minTimes  
                    SET the element at position neighbor and zero of minTimes TO nextTime  
                    APPEND the tuple of neighbor, nextTime, zero TO queue  
                ELSE IF the element at position neighbor and zero of minTimes LESS THAN nextTime AND nextTime LESS THAN the element at position neighbor and one of minTimes  
                    SET the element at position neighbor and one of minTimes TO nextTime  
                    APPEND the tuple of neighbor, nextTime, one TO queue  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
