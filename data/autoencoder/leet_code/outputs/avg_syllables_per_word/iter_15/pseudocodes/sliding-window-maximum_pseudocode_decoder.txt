CLASS Solution  
    FUNCTION maxSlidingWindow(nums LIST OF INTEGER, k INTEGER) RETURNS LIST OF INTEGER  
        IF nums IS EMPTY OR k EQUALS zero THEN  
            RETURN empty list  
        END IF  
        SET deq TO an empty double ended queue  
        SET result TO an empty list  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT EMPTY AND the element at position zero of deq IS LESS THAN i MINUS k PLUS one THEN  
                REMOVE the element from the left end of deq  
            END IF  
            WHILE deq IS NOT EMPTY AND the element at position element at position the last position of deq of nums IS LESS THAN the element at position i of nums  
                REMOVE the element from the right end of deq  
            END WHILE  
            APPEND i TO deq  
            IF i IS GREATER THAN OR EQUAL TO k MINUS one THEN  
                APPEND the element at position element at position zero of deq of nums TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
