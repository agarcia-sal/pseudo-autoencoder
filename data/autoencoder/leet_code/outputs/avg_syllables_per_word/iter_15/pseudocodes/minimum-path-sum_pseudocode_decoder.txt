CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid EQUALS empty OR the LENGTH OF grid EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        ASSIGN dp TO a two dimensional LIST OF zero WITH m rows AND n columns  
        
        SET element at row zero and column zero of dp TO element at row zero and column zero of grid  
        
        FOR each index j FROM one TO n MINUS one  
            SET element at row zero and column j of dp TO element at row zero and column j MINUS one of dp PLUS element at row zero and column j of grid  
        END FOR  
        
        FOR each index i FROM one TO m MINUS one  
            SET element at row i and column zero of dp TO element at row i MINUS one and column zero of dp PLUS element at row i and column zero of grid  
        END FOR  
        
        FOR each index i FROM one TO m MINUS one  
            FOR each index j FROM one TO n MINUS one  
                SET element at row i and column j of dp TO the MINIMUM OF element at row i MINUS one and column j of dp AND element at row i and column j MINUS one of dp PLUS element at row i and column j of grid  
            END FOR  
        END FOR  
        
        RETURN element at row m MINUS one and column n MINUS one of dp  
    END FUNCTION  
END CLASS
