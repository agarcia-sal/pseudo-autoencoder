CLASS Solution  
    FUNCTION validUtf8(data)  
        FUNCTION get_byte_count(byte)  
            IF the result of shifting byte seven positions to the right EQUALS zero  
                RETURN one  
            ELSE IF the result of shifting byte five positions to the right EQUALS one one zero in binary  
                RETURN two  
            ELSE IF the result of shifting byte four positions to the right EQUALS one one one zero in binary  
                RETURN three  
            ELSE IF the result of shifting byte three positions to the right EQUALS one one one one zero in binary  
                RETURN four  
            ELSE  
                RETURN negative one  
            END IF  
        END FUNCTION  

        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF data  
            SET byte_count TO the result of calling get_byte_count with element at position i of data  
            IF byte_count EQUALS negative one  
                RETURN False  
            END IF  
            IF i PLUS byte_count IS GREATER THAN the LENGTH OF data  
                RETURN False  
            END IF  
            FOR j FROM i PLUS one TO i PLUS byte_count MINUS one  
                IF the result of shifting element at position j of data six positions to the right DOES NOT EQUAL one zero in binary  
                    RETURN False  
                END IF  
            END FOR  
            INCREMENT i BY byte_count  
        END WHILE  
        RETURN True  
    END FUNCTION  
END CLASS
