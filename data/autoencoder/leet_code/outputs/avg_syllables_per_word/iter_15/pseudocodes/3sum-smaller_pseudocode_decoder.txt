CLASS Solution  
    FUNCTION threeSumSmaller(nums, target)  
        SORT the list nums in ascending order  
        SET count TO zero  
        SET n TO the LENGTH OF list nums  
        
        FOR index i FROM zero TO n MINUS two  
            SET left TO i PLUS one  
            SET right TO n MINUS one  
            WHILE the value of left IS LESS THAN the value of right  
                SET current_sum TO the element at position i of nums PLUS the element at position left of nums PLUS the element at position right of nums  
                IF current_sum IS LESS THAN target  
                    INCREMENT count BY the value of right MINUS the value of left  
                    INCREMENT left BY one  
                ELSE  
                    DECREMENT right BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
