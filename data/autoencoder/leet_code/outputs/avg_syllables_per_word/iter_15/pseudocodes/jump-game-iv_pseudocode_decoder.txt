CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO a new empty mapping from values TO empty lists  
        FOR each pair of index and value IN arr  
            APPEND index TO value_indices value  
        END FOR  
        
        SET queue TO a new empty double-ended collection initialized with the pair of zero and zero representing current index and current steps respectively  
        SET visited TO a new empty set containing zero  
        
        WHILE queue IS NOT EMPTY  
            REMOVE and RETURN the element from the front of queue ASSIGN TO current_index and steps  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO a new list containing current_index PLUS one and current_index MINUS one concatenated with the list at value_indices of the element at position current_index of arr  
            
            FOR each next_index IN next_indices  
                IF next_index IS GREATER THAN OR EQUAL TO zero AND next_index IS LESS THAN n AND next_index IS NOT IN visited  
                    ADD next_index TO visited  
                    APPEND the pair of next_index and steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list at value_indices of the element at position current_index of arr TO an empty list  
        END WHILE  
        
        RETURN the integer negative one  
    END FUNCTION  
END CLASS
