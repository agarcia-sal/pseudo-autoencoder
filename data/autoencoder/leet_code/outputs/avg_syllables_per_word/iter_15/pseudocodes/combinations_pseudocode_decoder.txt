CLASS Solution  
    FUNCTION combine WITH PARAMETERS n INTEGER k INTEGER  
        FUNCTION backtrack WITH PARAMETERS start INTEGER path LIST OF INTEGER  
            IF the LENGTH OF path EQUALS k  
                APPEND a COPY of path TO result  
                RETURN  
            END IF  
            FOR i FROM start TO n  
                APPEND i TO path  
                CALL backtrack WITH PARAMETERS the INCREMENTED BY one VALUE OF i AND path  
                REMOVE the LAST ELEMENT FROM path  
            END FOR  
        END FUNCTION  
        SET result TO an EMPTY LIST  
        CALL backtrack WITH PARAMETERS one AND an EMPTY LIST  
        RETURN result  
    END FUNCTION  
END CLASS
