CLASS Solution  
    FUNCTION parseTernary with parameter expression  
        FUNCTION evaluate with parameter index  
            IF index EQUALS the LENGTH OF expression MINUS one OR the element at position index PLUS one of expression EQUALS the character colon  
                RETURN the element at position index of expression AND index PLUS two  
            END IF  
            SET true_part comma next_index TO the RESULT OF evaluate with index PLUS two  
            SET false_part comma final_index TO the RESULT OF evaluate with next_index  
            IF the element at position index of expression EQUALS the character T  
                RETURN true_part AND final_index  
            ELSE  
                RETURN false_part AND final_index  
            END IF  
        END FUNCTION  
        SET result comma unused_variable TO the RESULT OF evaluate with zero  
        RETURN result  
    END FUNCTION  
END CLASS
