CLASS Solution  
    FUNCTION minInterval(intervals, queries)  
        CALL sort the intervals by their start time in ascending order  
        SET sorted_queries TO the sorted pairs of each query and its original index in ascending order by query value  
        SET min_heap TO an empty list for storing interval size and interval end pairs  
        SET result TO a list of length equal to the length of queries filled with negative one  
        SET interval_index TO zero  
        FOR each pair of query and original_index IN sorted_queries  
            WHILE interval_index is less than the length of intervals AND the start element at position zero of the interval at position interval_index is less than or equal to query  
                SET start TO the start element at position zero of the interval at position interval_index  
                SET end TO the end element at position one of the interval at position interval_index  
                SET interval_size TO end MINUS start PLUS one  
                PUSH the pair of interval_size and end INTO min_heap maintaining heap property  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE min_heap is not empty AND the second element at position one of the first pair in min_heap is less than query  
                POP the smallest pair from min_heap maintaining heap property  
            END WHILE  
            IF min_heap is not empty  
                SET the element at position original_index of result TO the first element at position zero of the first pair in min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
