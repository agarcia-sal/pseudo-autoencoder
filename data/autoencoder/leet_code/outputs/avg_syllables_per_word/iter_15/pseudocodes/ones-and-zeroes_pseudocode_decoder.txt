CLASS Solution  
    FUNCTION findMaxForm(strs AS list of strings, m AS integer, n AS integer) RETURNS integer  
        SET dp TO a two-dimensional list of integers with dimensions m PLUS one by n PLUS one initialized to zero  
        FOR each string s IN strs  
            SET count_zero TO the number of zero characters in string s  
            SET count_one TO the number of one characters in string s  
            FOR i FROM m DOWN TO count_zero  
                FOR j FROM n DOWN TO count_one  
                    SET dp element at position i and j TO the greater of dp element at position i and j AND the value of dp element at position i MINUS count_zero and j MINUS count_one PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN the element at position m and n of dp  
    END FUNCTION  
END CLASS
