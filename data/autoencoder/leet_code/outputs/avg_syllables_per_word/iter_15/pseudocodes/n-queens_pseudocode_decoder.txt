CLASS Solution  
    FUNCTION solveNQueens WITH PARAMETER n  
        FUNCTION is_not_under_attack WITH PARAMETERS row AND col  
            RETURN NOT THE SUM OF element at position col of cols PLUS element at position row MINUS col of hills PLUS element at position row PLUS col of dales EQUALS zero  
        END FUNCTION  
        
        FUNCTION place_queen WITH PARAMETERS row AND col  
            ADD the pair consisting of row and col TO queens  
            SET element at position col of cols TO one  
            SET element at position row MINUS col of hills TO one  
            SET element at position row PLUS col of dales TO one  
        END FUNCTION  
        
        FUNCTION remove_queen WITH PARAMETERS row AND col  
            REMOVE the pair consisting of row and col FROM queens  
            SET element at position col of cols TO zero  
            SET element at position row MINUS col of hills TO zero  
            SET element at position row PLUS col of dales TO zero  
        END FUNCTION  
        
        FUNCTION add_solution  
            CREATE empty list solution  
            FOR each pair of any value AND col IN queens SORTED BY row  
                APPEND string consisting of a sequence of dot characters repeated col times PLUS character Q PLUS a sequence of dot characters repeated n MINUS col MINUS one times TO solution  
            END FOR  
            APPEND solution TO output  
        END FUNCTION  
        
        FUNCTION backtrack WITH PARAMETER row DEFAULTED TO zero  
            FOR col FROM zero TO n MINUS one  
                IF is_not_under_attack WITH row AND col RETURNS true  
                    CALL place_queen WITH row AND col  
                    IF row PLUS one EQUALS n  
                        CALL add_solution  
                    ELSE  
                        CALL backtrack WITH row PLUS one  
                    END IF  
                    CALL remove_queen WITH row AND col  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET cols TO a list of n zeros  
        SET hills TO a list of two times n MINUS one zeros representing hill diagonals  
        SET dales TO a list of two times n MINUS one zeros representing dale diagonals  
        SET queens TO an empty set representing positions of queens  
        SET output TO an empty list  
        CALL backtrack  
        RETURN output  
    END FUNCTION  
END CLASS
