CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO a mapping from each integer course from zero TO numCourses MINUS one ASSIGNED TO an empty list  
        SET in_degrees TO a list of zero repeated numCourses times  
        
        FOR each pair consisting of destination course AND source course IN prerequisites  
            APPEND destination course TO the list at key source course in adj_list  
            INCREMENT the element at position destination course in in_degrees BY one  
        END FOR  
        
        SET queue TO a deque initialized with each course FROM zero TO numCourses MINUS one WHERE the element at position course in in_degrees EQUALS zero  
        SET topological_order TO an empty list  
        
        WHILE queue IS NOT empty  
            SET current TO the element removed from the left end of queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN the list at key current in adj_list  
                DECREMENT the element at position neighbor in in_degrees BY one  
                IF the element at position neighbor in in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the length of topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
