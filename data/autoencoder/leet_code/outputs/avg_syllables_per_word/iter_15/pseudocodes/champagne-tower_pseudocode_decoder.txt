CLASS Solution  
    FUNCTION champagneTower(poured, query_row, query_glass)  
        SET current_row TO a list containing poured as its only element  
        FOR each integer row FROM one TO query_row  
            SET next_row TO a list of zeros with LENGTH equal TO row PLUS one  
            FOR each index i FROM zero TO the LENGTH OF current_row MINUS one  
                IF the element at position i of current_row IS GREATER THAN one  
                    SET excess TO the element at position i of current_row MINUS one DIVIDED BY two  
                    INCREMENT the element at position i of next_row BY excess  
                    INCREMENT the element at position i PLUS one of next_row BY excess  
                END IF  
            END FOR  
            ASSIGN current_row TO next_row  
        END FOR  
        IF the element at position query_glass of current_row IS LESS THAN one  
            RETURN the element at position query_glass of current_row  
        ELSE  
            RETURN one  
        END IF  
    END FUNCTION  
END CLASS
