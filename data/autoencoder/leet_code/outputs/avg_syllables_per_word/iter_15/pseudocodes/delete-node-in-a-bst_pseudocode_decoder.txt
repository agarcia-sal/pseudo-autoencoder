CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero left EQUAL TO none right EQUAL TO none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values AS list  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO A NEW double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO FIRST ELEMENT REMOVED FROM THE LEFT OF queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUAL TO q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left q dot left AND is_same_tree WITH PARAMETERS p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode WITH PARAMETERS self root key  
        IF root IS none  
            RETURN none  
        END IF  
        IF root dot val IS GREATER THAN key  
            SET root dot left TO self dot deleteNode WITH PARAMETERS root dot left key  
            RETURN root  
        END IF  
        IF root dot val IS LESS THAN key  
            SET root dot right TO self dot deleteNode WITH PARAMETERS root dot right key  
            RETURN root  
        END IF  
        IF root dot left IS none  
            RETURN root dot right  
        END IF  
        IF root dot right IS none  
            RETURN root dot left  
        END IF  
        SET node TO root dot right  
        WHILE node dot left IS NOT none  
            SET node TO node dot left  
        END WHILE  
        SET node dot left TO root dot left  
        SET root TO root dot right  
        RETURN root  
    END FUNCTION  
END CLASS
