CLASS Solution  
    FUNCTION search(nums WITH TYPE List OF integer, target WITH TYPE integer) RETURNS boolean  
        IF the LENGTH OF nums EQUALS zero  
            RETURN False  
        END IF  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS the DIFFERENCE BETWEEN right AND left DIVIDED BY two  
            IF the element at position mid of nums EQUALS target  
                RETURN True  
            END IF  
            IF the element at position left of nums EQUALS the element at position mid of nums AND the element at position mid of nums EQUALS the element at position right of nums  
                INCREMENT left BY one  
                DECREMENT right BY one  
            ELSE IF the element at position left of nums IS LESS THAN OR EQUAL TO the element at position mid of nums  
                IF the element at position left of nums IS LESS THAN OR EQUAL TO target AND target IS LESS THAN the element at position mid of nums  
                    SET right TO mid MINUS one  
                ELSE  
                    SET left TO mid PLUS one  
                END IF  
            ELSE  
                IF the element at position mid of nums IS LESS THAN target AND target IS LESS THAN OR EQUAL TO the element at position right of nums  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
