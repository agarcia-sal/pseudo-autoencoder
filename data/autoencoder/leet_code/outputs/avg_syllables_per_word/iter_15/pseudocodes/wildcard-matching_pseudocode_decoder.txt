CLASS Solution  
    FUNCTION isMatch(s string, p string) RETURNS boolean  
        SET dp TO a two dimensional collection of boolean values with the number of rows EQUAL TO the length of s PLUS one AND the number of columns EQUAL TO the length of p PLUS one  
        SET each element in dp TO false  
        
        SET element at position zero row zero column of dp TO true  
        
        FOR each position j FROM one TO the length of p  
            IF the element at position j MINUS one of p EQUALS the character star  
                SET the element at position zero row j column of dp TO the element at position zero row j MINUS one column of dp  
            END IF  
        END FOR  
        
        FOR each position i FROM one TO the length of s  
            FOR each position j FROM one TO the length of p  
                IF the element at position j MINUS one of p EQUALS the character star  
                    SET the element at position i row j column of dp TO the logical OR of the element at position i MINUS one row j column of dp AND the element at position i row j MINUS one column of dp  
                ELSE IF the element at position j MINUS one of p EQUALS the character question mark OR the element at position i MINUS one of s EQUALS the element at position j MINUS one of p  
                    SET the element at position i row j column of dp TO the element at position i MINUS one row j MINUS one column of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position the length of s row the length of p column of dp  
    END FUNCTION  
END CLASS
