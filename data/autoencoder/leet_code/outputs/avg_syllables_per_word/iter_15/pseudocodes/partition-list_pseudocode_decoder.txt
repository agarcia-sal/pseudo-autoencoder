CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A new ListNode CREATED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN values STARTING FROM position one TO the END  
        SET node TO A new ListNode CREATED WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND is_same_list CALLED WITH the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION partition WITH PARAMETERS head AS Optional ListNode AND x AS integer RETURNING Optional ListNode  
        SET less_head TO A new ListNode CREATED WITH zero  
        SET greater_head TO A new ListNode CREATED WITH zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current IS NOT none  
            IF the val ATTRIBUTE OF current IS LESS THAN x  
                SET the next ATTRIBUTE OF less TO current  
                SET less TO the next ATTRIBUTE OF less  
            ELSE  
                SET the next ATTRIBUTE OF greater TO current  
                SET greater TO the next ATTRIBUTE OF greater  
            END IF  
            SET current TO the next ATTRIBUTE OF current  
        END WHILE  
        SET the next ATTRIBUTE OF less TO the next ATTRIBUTE OF greater_head  
        SET the next ATTRIBUTE OF greater TO none  
        RETURN the next ATTRIBUTE OF less_head  
    END FUNCTION  
END CLASS
