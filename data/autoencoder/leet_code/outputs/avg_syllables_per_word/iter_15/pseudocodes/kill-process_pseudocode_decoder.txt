CLASS Solution  
    FUNCTION killProcess(pid AS List OF integer, ppid AS List OF integer, kill AS integer) RETURNS List OF integer  
        SET tree TO a mapping from parent process identifiers TO lists of child process identifiers with default empty lists  
        FOR each child process identifier and corresponding parent process identifier IN parallel iteration of pid AND ppid  
            APPEND child process identifier TO the list associated with parent process identifier in tree  
        END FOR  
        SET queue TO a list containing kill  
        SET killed_processes TO an empty list  
        WHILE the list queue holds any elements  
            REMOVE the element at position zero FROM queue AND ASSIGN TO current  
            APPEND current TO killed_processes  
            EXTEND queue BY the list of child process identifiers associated with current in tree  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
