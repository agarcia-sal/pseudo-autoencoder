CLASS Solution  
    FUNCTION frequencySort WITH parameter s  
        ASSIGN pointer TO a counter object created from the elements of s named frequency  
        ASSIGN pointer TO a list of elements corresponding to the keys of frequency sorted such that elements with higher associated frequency come before elements with lower associated frequency named sorted_characters  
        SET result TO empty string  
        FOR each character IN sorted_characters  
            SET repeated_sequence TO character REPEATED the number of times EQUAL TO element at position character of frequency  
            SET result TO result CONCATENATED WITH repeated_sequence  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
