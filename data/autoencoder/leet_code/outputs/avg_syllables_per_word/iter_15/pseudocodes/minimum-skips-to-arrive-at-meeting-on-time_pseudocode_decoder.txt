CLASS Solution  
    FUNCTION minSkips(dist AS List of integers, speed AS integer, hoursBefore AS integer) RETURNS integer  
        SET n TO the LENGTH OF dist  
        
        FUNCTION initialize_dp_matrix(rows AS integer, columns AS integer) RETURNS two dimensional list of infinity values  
            CREATE a two dimensional list with the given number of rows and columns  
            EACH element IS SET TO infinity  
            RETURN the created list  
        END FUNCTION  
        
        SET dp TO initialize_dp_matrix WITH parameters n AND n PLUS one  
        
        SET total_dist TO the SUM OF all elements in dist  
        
        IF total_dist DIVIDED BY speed IS GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  
        
        SET element at position zero zero of dp TO element at position zero of dist  
        
        FOR j FROM one TO n MINUS one  
            SET element at position zero j of dp TO element at position zero of dist  
        END FOR  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j IS LESS THAN i THEN  
                    SET element at position i j of dp TO the CEILING of element at position i MINUS one j of dp DIVIDED BY speed MULTIPLIED BY speed PLUS element at position i of dist  
                END IF  
                IF j IS GREATER THAN zero THEN  
                    SET element at position i j of dp TO the MINIMUM OF element at position i j of dp AND element at position i MINUS one j MINUS one of dp PLUS element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            IF element at position n MINUS one j of dp IS LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
