CLASS Solution  
    FUNCTION numMatchingSubseq WITH PARAMETERS s AND words  
        SET waiting TO a DICTIONARY MAPPING each UNIQUE CHARACTER IN s TO an EMPTY DOUBLE ENDED QUEUE  
        FOR each word IN words  
            ASSIGN it TO an ITERATOR OVER word  
            SET first_letter TO the NEXT ELEMENT OF it  
            IF first_letter IS A KEY IN waiting  
                APPEND it TO the DOUBLE ENDED QUEUE ASSOCIATED WITH first_letter IN waiting  
            END IF  
        END FOR  
        SET match_count TO zero  
        FOR each character c IN s  
            IF c IS A KEY IN waiting  
                ASSIGN current_iterators TO the DOUBLE ENDED QUEUE ASSOCIATED WITH c IN waiting  
                SET the DOUBLE ENDED QUEUE ASSOCIATED WITH c IN waiting TO a NEW EMPTY DOUBLE ENDED QUEUE  
                FOR each it IN current_iterators  
                    SET next_char TO the NEXT ELEMENT OF it OR TO NONE IF NO ELEMENT EXISTS  
                    IF next_char IS NONE  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char IS A KEY IN waiting  
                            APPEND it TO the DOUBLE ENDED QUEUE ASSOCIATED WITH next_char IN waiting  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        RETURN match_count  
    END FUNCTION  
END CLASS
