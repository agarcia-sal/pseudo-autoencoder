CLASS Solution  
    FUNCTION largestPathValue(colors, edges)  
        SET n TO the LENGTH OF colors  
        SET graph TO a new mapping with default empty list for missing keys  
        SET in_degree TO a list of zeros with the LENGTH n  
        
        FOR each pair u AND v IN edges  
            APPEND element v TO the list associated with key u IN graph  
            INCREMENT element at position v of in_degree BY one  
        END FOR  
        
        SET queue TO a new double ended queue containing each integer i FROM zero TO n MINUS one WHERE the element at position i of in_degree EQUALS zero  
        SET dp TO a list of n elements each being a list of twenty six zeros  
        SET visited TO zero  
        SET max_color_value TO zero  
        
        WHILE queue IS NOT empty  
            REMOVE the first element FROM queue AND ASSIGN it TO node  
            INCREMENT element at position corresponding to the ordinal value of the element at position node of colors MINUS ordinal value of letter a of dp at position node BY one  
            SET max_color_value TO the greater value between max_color_value AND element at position corresponding to the ordinal value of the element at position node of colors MINUS ordinal value of letter a of dp at position node  
            INCREMENT visited BY one  
            
            FOR each neighbor IN the list associated with key node IN graph  
                FOR each integer c FROM zero TO twenty five  
                    SET element at position c of dp at position neighbor TO the greater value between element at position c of dp at position neighbor AND element at position c of dp at position node  
                END FOR  
                DECREMENT element at position neighbor of in_degree BY one  
                IF element at position neighbor of in_degree EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF visited EQUALS n  
            RETURN max_color_value  
        ELSE  
            RETURN the integer negative one  
        END IF  
    END FUNCTION  
END CLASS
