CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each index AND character IN the enumeration of s  
                IF character EQUALS the character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND the concatenation of the character one PLUS the result of calling dfs with the substring from position start PLUS one TO index PLUS the character zero TO specials  
                    SET start TO index PLUS one  
                END IF  
            END FOR  
            SORT specials IN reverse lexicographical order  
            RETURN the concatenation of all elements in specials as a single string  
        END FUNCTION  
        RETURN the result of calling dfs with s  
    END FUNCTION  
END CLASS
