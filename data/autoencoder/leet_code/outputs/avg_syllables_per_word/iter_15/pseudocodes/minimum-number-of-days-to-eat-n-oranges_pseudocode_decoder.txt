CLASS Solution  
    FUNCTION minDays WITH PARAMETER n  
        FUNCTION dp WITH PARAMETER remaining  
            IF remaining IS LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            SET option_one TO remainder of remaining DIVIDED BY two PLUS dp of remaining INTEGER DIVIDED BY two  
            SET option_two TO remainder of remaining DIVIDED BY three PLUS dp of remaining INTEGER DIVIDED BY three  
            RETURN one PLUS the MINIMUM OF option_one AND option_two  
        END FUNCTION  
        RETURN dp of n  
    END FUNCTION  
END CLASS
