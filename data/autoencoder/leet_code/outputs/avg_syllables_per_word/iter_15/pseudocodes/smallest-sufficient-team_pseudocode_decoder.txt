CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills AS List OF STRING, people AS List OF List OF STRING) RETURNS List OF INTEGER  
        CREATE dictionary d THAT MAPS each skill TO its index IN req_skills BY enumerating req_skills  
        SET m TO the LENGTH OF req_skills  
        SET n TO the LENGTH OF people  
        CREATE list p WITH n elements ALL SET TO zero  
        FOR each index i AND skill_list ss IN people BY enumerating people  
            FOR each skill s IN ss  
                SET p at position i TO the BITWISE OR OF p at position i AND one LEFT SHIFTED BY the index of skill s from dictionary d  
            END FOR  
        END FOR  
        CREATE list f WITH size equal TO two RAISED TO THE power m FILLED WITH the value infinity  
        CREATE list g WITH size equal TO two RAISED TO THE power m FILLED WITH zero  
        CREATE list h WITH size equal TO two RAISED TO THE power m FILLED WITH zero  
        SET the element at position zero in list f TO zero  
        FOR i FROM zero TO two RAISED TO THE power m MINUS one  
            IF the element at position i in list f EQUALS infinity  
                CONTINUE to the next iteration  
            END IF  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i in list f PLUS one IS LESS THAN the element at position result of the BITWISE OR of i AND p at position j in list f  
                    SET the element at position BITWISE OR of i AND p at position j in list f TO the element at position i in list f PLUS one  
                    SET the element at position BITWISE OR of i AND p at position j in list g TO j  
                    SET the element at position BITWISE OR of i AND p at position j in list h TO i  
                END IF  
            END FOR  
        END FOR  
        SET variable i TO two RAISED TO THE power m MINUS one  
        CREATE empty list ans  
        WHILE variable i IS NOT zero  
            APPEND the element at position i in list g TO list ans  
            SET i TO the element at position i in list h  
        END WHILE  
        RETURN list ans  
    END FUNCTION  
END CLASS
