CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1, val2)  
            RETURN the condition that val1 is GREATER THAN zero AND val2 is GREATER THAN zero OR val1 is LESS THAN zero AND val2 is LESS THAN zero  
        END FUNCTION  
        
        SET n TO the LENGTH OF nums  
        
        FOR i FROM zero TO n MINUS one  
            SET slow TO i  
            SET fast TO i  
            
            IF the element at position i of nums EQUALS zero  
                CONTINUE to next iteration of the FOR loop  
            END IF  
            
            WHILE TRUE  
                SET slow TO the remainder of the SUM of slow PLUS the element at position slow of nums DIVIDED BY n  
                IF slow is LESS THAN zero  
                    INCREMENT slow BY n  
                END IF  
                
                SET fast TO the remainder of the SUM of fast PLUS the element at position fast of nums DIVIDED BY n  
                IF fast is LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  
                
                SET fast TO the remainder of the SUM of fast PLUS the element at position fast of nums DIVIDED BY n  
                IF fast is LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  
                
                IF the condition that the direction given by the element at position slow of nums and the element at position i of nums ARE NOT the same OR the direction given by the element at position fast of nums and the element at position i of nums ARE NOT the same  
                    BREAK the WHILE loop  
                END IF  
                
                IF slow EQUALS fast  
                    IF slow EQUALS the remainder of the SUM of slow PLUS the element at position slow of nums DIVIDED BY n  
                        BREAK the WHILE loop  
                    END IF  
                    RETURN TRUE  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN FALSE  
    END FUNCTION  
END CLASS
