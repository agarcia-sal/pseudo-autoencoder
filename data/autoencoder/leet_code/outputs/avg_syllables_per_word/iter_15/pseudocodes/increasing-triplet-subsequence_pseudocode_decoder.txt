CLASS Solution  
    FUNCTION increasingTriplet WITH parameter nums  
        SET first TO positive infinity  
        SET second TO positive infinity  
        FOR each num IN nums  
            IF num IS LESS THAN OR EQUAL TO first THEN  
                SET first TO num  
            ELSE IF num IS LESS THAN OR EQUAL TO second THEN  
                SET second TO num  
            ELSE  
                RETURN true  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
