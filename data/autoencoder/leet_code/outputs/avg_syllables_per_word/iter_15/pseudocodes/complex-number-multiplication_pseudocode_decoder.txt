CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO the result of splitting the string num around the plus sign  
            SET real TO the integer converted from the element at position zero of parts  
            SET imaginary TO the integer converted from the substring from position zero TO the length of the element at position one of parts MINUS one  
            RETURN real AND imaginary  
        END FUNCTION  

        SET real1 AND imag1 TO the result of calling parse_complex with argument num1  
        SET real2 AND imag2 TO the result of calling parse_complex with argument num2  

        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  

        RETURN the concatenation of the string form of real_part PLUS the plus sign PLUS the string form of imaginary_part PLUS the character i  
    END FUNCTION  
END CLASS
