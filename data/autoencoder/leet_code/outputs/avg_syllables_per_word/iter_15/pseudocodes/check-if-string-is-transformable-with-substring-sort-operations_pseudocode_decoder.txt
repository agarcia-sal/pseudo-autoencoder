CLASS Solution  
    FUNCTION isTransformable WITH PARAMETERS s AND t  
        SET pos TO a new DEFAULT DICTIONARY of lists  
        FOR each index i AND character c IN s  
            PUSH the index i ONTO the heap corresponding TO the integer value of character c IN pos  
        END FOR  
        FOR each index i AND character c IN t  
            SET digit TO the integer value of character c  
            IF the heap corresponding TO digit IN pos IS empty  
                RETURN False  
            END IF  
            FOR each smaller_digit FROM zero TO digit MINUS one  
                IF the heap corresponding TO smaller_digit IN pos IS NOT empty AND the smallest element at position zero of the heap corresponding TO smaller_digit IN pos IS LESS THAN the smallest element at position zero of the heap corresponding TO digit IN pos  
                    RETURN False  
                END IF  
            END FOR  
            REMOVE the smallest element FROM the heap corresponding TO digit IN pos  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
