CLASS Solution  
    FUNCTION maximumScore(nums LIST OF INTEGER, multipliers LIST OF INTEGER) RETURNS INTEGER  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a new LIST OF INTEGER of size m PLUS one with all elements SET TO zero  
        FOR i FROM m MINUS one DOWN TO zero  
            SET new_dp TO a new LIST OF INTEGER of size m PLUS one with all elements SET TO zero  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS i PLUS left  
                SET mul TO element at position i of multipliers  
                SET score_if_left_chosen TO mul MULTIPLIED BY element at position left of nums PLUS element at position left PLUS one of dp  
                SET score_if_right_chosen TO mul MULTIPLIED BY element at position right of nums PLUS element at position left of dp  
                SET element at position left of new_dp TO the GREATER OF score_if_left_chosen AND score_if_right_chosen  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
