CLASS Solution  
    FUNCTION validTree WITH PARAMETERS n AND edges  
        IF the LENGTH OF edges IS NOT EQUAL TO n MINUS one THEN  
            RETURN false  
        END IF  
        SET adj_list TO an empty mapping from each integer FROM zero TO n MINUS one TO an empty list  
        FOR each pair a AND b IN edges  
            APPEND b TO the element at position a of adj_list  
            APPEND a TO the element at position b of adj_list  
        END FOR  
        SET visited TO an empty set  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the element at position node of adj_list  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        CALL dfs WITH zero  
        RETURN the LENGTH OF visited IS EQUAL TO n  
    END FUNCTION  
END CLASS
