CLASS Solution  
    FUNCTION orderOfLargestPlusSign WITH PARAMETERS n AND mines  
        SET mines_set TO a SET CREATED FROM each mine CONVERTED TO tuple IN mines  
        SET dp TO a TWO DIMENSIONAL LIST OF SIZE n BY n WHERE EACH ELEMENT IS A LIST OF FOUR ZEROES representing counts in four directions  
        SET directions TO a LIST OF FOUR TUPLES representing zero and one and minus one displacements for right down left up in that order  

        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the PAIR of i AND j IS NOT PRESENT IN mines_set  
                    IF j IS GREATER THAN zero  
                        SET element at position zero of element at position j of element at position i of dp TO the VALUE OF element at position zero of element at position j MINUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i IS GREATER THAN zero  
                        SET element at position one of element at position j of element at position i of dp TO the VALUE OF element at position one of element at position j of element at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR

        SET result TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the PAIR of i AND j IS NOT PRESENT IN mines_set  
                    IF j IS LESS THAN n MINUS one  
                        SET element at position two of element at position j of element at position i of dp TO the VALUE OF element at position two of element at position j PLUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i IS LESS THAN n MINUS one  
                        SET element at position three of element at position j of element at position i of dp TO the VALUE OF element at position three of element at position j of element at position i PLUS one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the MINIMUM OF all FOUR VALUES in element at position j of element at position i of dp  
                    SET result TO the MAXIMUM OF result AND plus_sign_order  
                END IF  
            END FOR  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
