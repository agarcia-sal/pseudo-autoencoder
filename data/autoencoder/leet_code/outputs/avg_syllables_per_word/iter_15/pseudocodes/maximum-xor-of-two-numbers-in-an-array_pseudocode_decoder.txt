CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO zero  
        FOR bit_position FROM thirty-one TO zero DECREMENTING BY one  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            CREATE empty set called prefixes  
            FOR each num IN nums  
                SET prefix TO integer division of num BY two RAISED TO THE POWER OF bit_position  
                ADD prefix TO prefixes  
            END FOR  
            FOR each prefix IN prefixes  
                IF result of logical XOR of current_candidate AND prefix IS IN prefixes  
                    SET max_xor TO current_candidate  
                    EXIT the current loop  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
