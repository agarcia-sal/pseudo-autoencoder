CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict LIST OF strings word1 string word2 string) RETURNS integer  
        SET min_distance TO positive infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO word1 EQUALS word2  
        
        FOR each index i AND corresponding word IN the enumeration of wordsDict  
            IF word EQUALS word1  
                IF same_word  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO i  
            END IF  
            
            IF word EQUALS word2  
                IF NOT same_word  
                    SET prev_word2 TO i  
                END IF  
            END IF  
            
            IF prev_word1 NOT EQUALS negative one AND prev_word2 NOT EQUALS negative one  
                SET min_distance TO the smaller value BETWEEN min_distance AND the absolute value of prev_word1 MINUS prev_word2  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
