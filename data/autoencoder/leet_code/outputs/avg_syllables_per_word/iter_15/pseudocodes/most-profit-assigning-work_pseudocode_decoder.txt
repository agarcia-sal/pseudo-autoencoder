CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty list OF integer profit list OF integer worker list OF integer) RETURNS integer  
        SET jobs TO RESULT OF zip FUNCTION TAKING difficulty AND profit AS ARGUMENTS CONVERTED TO list  
        CALL sort METHOD ON jobs  
        CALL sort METHOD ON worker  
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO LENGTH OF jobs  
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND element at position job_index of jobs AT position zero LESS THAN OR EQUAL TO ability HOLDS  
                SET best_profit TO THE GREATER BETWEEN best_profit AND element at position job_index of jobs AT position one  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        RETURN max_profit  
    END FUNCTION  
END CLASS
