CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals BY the element at position one of each interval AND BY the negative of the element at position zero of each interval  
        SET first TO the element at position one of the interval at position zero MINUS one  
        SET second TO the element at position one of the interval at position zero  
        SET count TO two  
        FOR each pair start and end IN the slice of intervals from position one TO the end  
            IF start IS GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
