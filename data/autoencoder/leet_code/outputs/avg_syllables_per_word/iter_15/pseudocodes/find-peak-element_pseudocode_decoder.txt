CLASS Solution  
    FUNCTION findPeakElement(nums)  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        WHILE left LESS THAN right  
            SET mid TO the SUM of left AND right DIVIDED BY two rounded down to the nearest integer  
            IF the element at position mid of nums GREATER THAN the element at position mid PLUS one of nums  
                SET right TO mid  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
