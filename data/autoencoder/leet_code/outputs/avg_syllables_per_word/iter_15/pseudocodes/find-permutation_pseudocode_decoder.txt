CLASS Solution  
    FUNCTION findPermutation WITH parameter s  
        SET result TO a list containing integers from one TO the LENGTH OF s PLUS one IN ascending order  
        SET start TO zero  
        FOR each end FROM zero TO the LENGTH OF s inclusive  
            IF end EQUALS the LENGTH OF s OR the element at position end of s EQUALS the uppercase letter I  
                SET the segment of result from position start TO end inclusive TO the reversed order of itself  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
