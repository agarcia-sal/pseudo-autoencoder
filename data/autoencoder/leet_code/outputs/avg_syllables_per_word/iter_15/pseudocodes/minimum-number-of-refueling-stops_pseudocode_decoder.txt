CLASS Solution  
    FUNCTION minRefuelStops WITH parameters target AND startFuel AND stations  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a list containing target AND zero TO stations  
        FOR each position AND fuel IN stations  
            WHILE max_heap IS NOT empty AND current_fuel IS LESS THAN position  
                SET largest_fuel TO NEGATIVE of the REMOVAL OF element WITH the smallest value FROM max_heap  
                INCREMENT current_fuel BY largest_fuel  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel IS LESS THAN position  
                RETURN negative one  
            END IF  
            INSERT NEGATIVE of fuel INTO max_heap maintaining heap properties  
        END FOR  
        RETURN stops  
    END FUNCTION  
END CLASS
