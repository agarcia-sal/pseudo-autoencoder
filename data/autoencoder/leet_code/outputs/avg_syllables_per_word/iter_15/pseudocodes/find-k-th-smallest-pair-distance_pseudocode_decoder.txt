CLASS Solution  
    FUNCTION smallestDistancePair(nums AS List OF integer, k AS integer) AS integer  
        CALL sorting OF nums TO arrange elements in ascending order  
        SET n TO the LENGTH OF nums  
        
        FUNCTION count(mid AS integer) AS integer  
            SET count TO zero  
            SET left TO zero  
            FOR each right FROM zero TO the LENGTH OF nums MINUS one  
                WHILE the value at position right of nums MINUS the value at position left of nums IS GREATER THAN mid  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT count BY the value of right MINUS left  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the value at the last position of nums MINUS the value at the first position of nums  
        
        WHILE left IS LESS THAN right  
            SET mid TO the integer division of left PLUS right BY two  
            IF the result of calling count with mid AS argument IS LESS THAN k  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
