CLASS Solution  
    FUNCTION exist(board, word)  
        IF board EQUALS empty  
            RETURN False  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        SET visited TO a two dimensional list of FALSE values with size m BY n  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS the LENGTH OF word  
                RETURN True  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of visited AND element at position y of that element IS True OR element at position x of board AND element at position y of that element NOT EQUALS element at position index of word  
                RETURN False  
            END IF  

            SET element at position x of visited AND element at position y of that element TO True  
            FOR each pair dx AND dy IN the collection of pairs consisting of MINUS one AND zero, one AND zero, zero AND MINUS one, zero AND one  
                IF dfs(x PLUS dx, y PLUS dy, index PLUS one)  
                    RETURN True  
                END IF  
            END FOR  
            SET element at position x of visited AND element at position y of that element TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs(i, j, zero)  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
