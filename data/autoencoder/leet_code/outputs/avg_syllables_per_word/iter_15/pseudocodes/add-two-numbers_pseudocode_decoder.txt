CLASS ListNode  
    FUNCTION __init__ with parameters val EQUALS zero and next EQUALS none  
        SET the attribute val TO val  
        SET the attribute next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position one TO the end  
        SET node TO a new ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 and p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the expression that the val attribute of p1 EQUALS the val attribute of p2 AND the result of is_same_list with the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers with parameters l1 and l2  
        SET dummy_head TO a new ListNode with zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT none OR l2 IS NOT none OR carry IS NOT zero  
            IF l1 IS NOT none  
                SET val1 TO the val attribute of l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT none  
                SET val2 TO the val attribute of l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO the integer division of total BY ten  
            SET the next attribute of current TO a new ListNode with the remainder of total DIVIDED BY ten  
            SET current TO the next attribute of current  
            
            IF l1 IS NOT none  
                SET l1 TO the next attribute of l1  
            END IF  
            IF l2 IS NOT none  
                SET l2 TO the next attribute of l2  
            END IF  
        END WHILE  
        
        RETURN the next attribute of dummy_head  
    END FUNCTION  
END CLASS
