CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO the LENGTH OF nums1  
        CREATE dp AS a list of size TWO RAISED TO THE POWER OF n WITH all elements SET TO infinity  
        SET element at position zero of dp TO zero  
        FOR mask FROM zero TO TWO RAISED TO THE POWER OF n MINUS one  
            SET bit_count TO the COUNT OF ONES IN THE BINARY REPRESENTATION OF mask  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND of mask AND TWO RAISED TO THE POWER OF j IS NOT ZERO  
                    SET candidate_value TO the element at position mask XOR TWO RAISED TO THE POWER OF j of dp PLUS the BITWISE XOR of element at position bit_count MINUS one of nums1 AND element at position j of nums2  
                    SET element at position mask of dp TO the MINIMUM OF the current element at position mask of dp AND candidate_value  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position TWO RAISED TO THE POWER OF n MINUS one of dp  
    END FUNCTION  
END CLASS
