CLASS Solution  
    FUNCTION findItinerary WITH PARAMETER tickets AS List OF List OF strings  
        SET graph TO a new mapping from keys to lists with default empty list  
        FOR each pair consisting of departure and arrival IN the reversal of the sorted tickets list  
            APPEND the arrival TO the list at key departure in graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit WITH PARAMETER airport  
            WHILE the list at key airport in graph holds at least one element  
                CALL visit WITH argument the element removed from the end of the list at key airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit WITH argument the string representing the airport code J F K  
        RETURN the reversal of itinerary  
    END FUNCTION  
END CLASS
