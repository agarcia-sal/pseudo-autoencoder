CLASS Solution  
    FUNCTION checkValidString WITH PARAMETER s  
        SET low TO zero  
        SET high TO zero  
        FOR each character IN s  
            IF character EQUALS open parenthesis  
                INCREMENT low BY one  
                INCREMENT high BY one  
            ELSE IF character EQUALS close parenthesis  
                DECREMENT low BY one  
                DECREMENT high BY one  
            ELSE  
                DECREMENT low BY one  
                INCREMENT high BY one  
            END IF  
            IF high LESS THAN zero  
                RETURN false  
            END IF  
            IF low LESS THAN zero  
                SET low TO zero  
            END IF  
        END FOR  
        IF low EQUALS zero  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
