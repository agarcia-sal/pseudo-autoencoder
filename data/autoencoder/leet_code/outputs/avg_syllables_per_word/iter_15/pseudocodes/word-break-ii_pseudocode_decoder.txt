CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO the SET of elements in wordDict  
        SET memo TO an empty DICTIONARY  
        
        FUNCTION backtrack(start)  
            IF start IS A KEY IN memo  
                RETURN the element at key start in memo  
            END IF  
            IF start EQUALS the LENGTH OF s  
                RETURN a LIST CONTAINING the EMPTY STRING  
            END IF  
            
            SET res TO an EMPTY LIST  
            FOR each end FROM start PLUS one TO the LENGTH OF s  
                SET word TO the SUBSTRING of s from position start PLUS one TO position end  
                IF word IS AN ELEMENT IN wordSet  
                    FOR each sentence IN backtrack with argument end  
                        IF sentence IS NOT EMPTY  
                            APPEND the CONCATENATION of word PLUS one SPACE PLUS sentence TO res  
                        ELSE  
                            APPEND word TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            
            SET the element at key start in memo TO res  
            RETURN res  
        END FUNCTION  
        
        RETURN backtrack with argument zero  
    END FUNCTION  
END CLASS
