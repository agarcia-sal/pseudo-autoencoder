CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET val ATTRIBUTE TO val  
        SET next ATTRIBUTE TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode WITH PARAMETER element at position zero of values  
    SET p TO head  
    FOR each val IN elements from position one TO the end of values  
        SET node TO A NEW ListNode WITH PARAMETER val  
        SET the next ATTRIBUTE of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the val ATTRIBUTE of p1 EQUALS the val ATTRIBUTE of p2 AND is_same_list WITH PARAMETERS the next ATTRIBUTE of p1 AND the next ATTRIBUTE of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight WITH PARAMETERS head AS optional ListNode AND k AS integer  
        IF head IS none OR the next ATTRIBUTE of head IS none OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE the next ATTRIBUTE of tail HOLDS  
            SET tail TO the next ATTRIBUTE of tail  
            INCREMENT length BY one  
        END WHILE  
        SET the next ATTRIBUTE of tail TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR each index FROM zero TO steps_to_new_tail MINUS one  
            SET new_tail TO the next ATTRIBUTE of new_tail  
        END FOR  
        SET new_head TO the next ATTRIBUTE of new_tail  
        SET the next ATTRIBUTE of new_tail TO none  
        RETURN new_head  
    END FUNCTION  
END CLASS
