CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary mapping zero TO negative one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each index i AND each number num IN nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count EXISTS AS KEY IN prefix_sum_index  
                SET possible_length TO i MINUS element at position count of prefix_sum_index  
                SET max_length TO THE GREATER OF max_length AND possible_length  
            ELSE  
                SET element at position count of prefix_sum_index TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
