CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count_of_elements_less_than_or_equal_to_x TO zero  
            FOR each element v IN nums  
                IF element v is less than or equal to x THEN  
                    INCREMENT count_of_elements_less_than_or_equal_to_x BY one  
                END IF  
            END FOR  
            IF count_of_elements_less_than_or_equal_to_x is greater than x THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        RETURN the result of performing binary search with bisect_left over the range from zero TO the length of nums minus one using function f as the key function for comparison  
    END FUNCTION  
END CLASS
