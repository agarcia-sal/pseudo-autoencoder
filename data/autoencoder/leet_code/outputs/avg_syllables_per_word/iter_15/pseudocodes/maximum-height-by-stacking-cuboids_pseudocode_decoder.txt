CLASS Solution  
    FUNCTION maxHeight WITH PARAMETERS cuboids  
        FOR each cuboid IN cuboids  
            SORT the elements of cuboid in ascending order  
        END FOR  
        SORT the cuboids in lexicographical ascending order  
        SET n TO the LENGTH OF cuboids  
        INITIALIZE dp AS a list of n zeros  
        FOR index i FROM zero TO n MINUS one  
            SET dp at position i TO element at position two of cuboids at position i  
            FOR index j FROM zero TO i MINUS one  
                IF element at position zero of cuboids at position j IS LESS THAN OR EQUAL TO element at position zero of cuboids at position i  
                AND element at position one of cuboids at position j IS LESS THAN OR EQUAL TO element at position one of cuboids at position i  
                AND element at position two of cuboids at position j IS LESS THAN OR EQUAL TO element at position two of cuboids at position i THEN  
                    SET dp at position i TO the GREATER VALUE BETWEEN dp at position i AND dp at position j PLUS element at position two of cuboids at position i  
                END IF  
            END FOR  
        END FOR  
        RETURN the MAXIMUM VALUE IN dp  
    END FUNCTION  
END CLASS
