CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MOD TO one billion PLUS seven  
        SORT nums IN ascending order  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        SET power TO a list of size n FILLED WITH one  
        
        FOR index FROM one TO n MINUS one  
            SET element at position index of power TO the REMAINDER AFTER DIVIDING the PRODUCT OF element at position index MINUS one of power AND two BY MOD  
        END FOR  
        
        FOR index FROM zero TO n MINUS one  
            SET total_sum TO the REMAINDER AFTER DIVIDING the SUM OF total_sum AND the PRODUCT OF element at position index of nums AND the DIFFERENCE BETWEEN element at position index of power AND element at position n MINUS index MINUS one of power BY MOD  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
