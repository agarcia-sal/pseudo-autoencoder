CLASS Solution  
    FUNCTION checkSubarraySum WITH PARAMETERS nums AND k  
        SET remainder_map TO A DICTIONARY WITH KEY zero AND VALUE negative one  
        SET current_sum TO zero  
        FOR each index i AND element num IN nums  
            INCREMENT current_sum BY num  
            IF k IS NOT EQUALS TO zero  
                SET current_sum TO current_sum MODULO k  
            END IF  
            IF remainder_map CONTAINS KEY current_sum  
                IF i MINUS remainder_map AT KEY current_sum IS GREATER THAN one  
                    RETURN True  
                END IF  
            ELSE  
                SET remainder_map AT KEY current_sum TO i  
            END ELSE  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
