CLASS Solution  
    FUNCTION sortTransformedArray(nums AS List OF Integer, a AS Integer, b AS Integer, c AS Integer) RETURNS List OF Integer  
        FUNCTION quadratic(x AS Integer) RETURNS Integer  
            RETURN a MULTIPLIED BY x MULTIPLIED BY x PLUS b MULTIPLIED BY x PLUS c  
        END FUNCTION  

        SET n TO the LENGTH OF nums  
        SET result TO a LIST OF zero WITH LENGTH n  
        SET left TO zero  
        SET right TO n MINUS one  

        IF a IS GREATER THAN OR EQUAL TO zero  
            SET index TO n MINUS one  
            WHILE left IS LESS THAN OR EQUAL TO right  
                IF quadratic OF the element at position left of nums IS GREATER THAN quadratic OF the element at position right of nums  
                    SET the element at position index of result TO quadratic OF the element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET the element at position index of result TO quadratic OF the element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                DECREMENT index BY one  
            END WHILE  
        ELSE  
            SET index TO zero  
            WHILE left IS LESS THAN OR EQUAL TO right  
                IF quadratic OF the element at position left of nums IS LESS THAN quadratic OF the element at position right of nums  
                    SET the element at position index of result TO quadratic OF the element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET the element at position index of result TO quadratic OF the element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                INCREMENT index BY one  
            END WHILE  
        END IF  

        RETURN result  
    END FUNCTION  
END CLASS
