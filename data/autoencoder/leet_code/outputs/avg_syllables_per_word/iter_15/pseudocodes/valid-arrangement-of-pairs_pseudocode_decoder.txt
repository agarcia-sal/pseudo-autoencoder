CLASS Solution  
    FUNCTION validArrangement WITH parameters pairs AS List OF List OF Integer  
        SET graph TO a new mapping from element to list initialized as empty list  
        SET out_degree TO a new mapping from element to count initialized as zero  
        SET in_degree TO a new mapping from element to count initialized as zero  
        
        FOR each element u AND element v IN pairs  
            APPEND element v TO the list at key u in graph  
            INCREMENT count at key u in out_degree BY one  
            INCREMENT count at key v in in_degree BY one  
        END FOR  
        
        SET start_node TO nothing  
        FOR each element node IN graph  
            IF the count at key node in out_degree EQUALS the count at key node in in_degree PLUS one  
                SET start_node TO node  
                EXIT the current loop  
            END IF  
        END FOR  
        IF start_node IS nothing  
            SET start_node TO the first element obtained from graph keys iterator  
        END IF  
        
        SET path TO an empty list  
        SET stack TO a list containing start_node  
        
        WHILE the LENGTH OF stack IS greater than zero  
            SET u TO the element at the last position of stack  
            IF the list at key u in graph IS NOT empty  
                SET v TO and REMOVE the last element from the list at key u in graph  
                APPEND element v TO stack  
            ELSE  
                REMOVE element from the last position of stack AND APPEND it TO path  
            END IF  
        END WHILE  
        
        REVERSE the order of elements in path  
        
        SET result TO an empty list  
        FOR each index i FROM zero TO the LENGTH OF path MINUS one MINUS one  
            APPEND a pair consisting of the element at position i of path AND the element at position i PLUS one of path TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
