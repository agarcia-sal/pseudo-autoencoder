    FUNCTION kthSmallestPrimeFraction WITH PARAMETERS arr AND k  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an EMPTY LIST  
        FOR i FROM zero TO n MINUS one MINUS one  
            PUSH onto min_heap the TUPLE OF the RATIO OF element at position i of arr DIVIDED BY element at position n MINUS one of arr AND i AND n MINUS one  
        END FOR  
        FOR each IGNORE IN range FROM zero TO k MINUS one MINUS one  
            POP the SMALLEST element FROM min_heap AND ASSIGN TO IGNORE VARIABLE unnamed AND i AND j  
            IF j MINUS one IS GREATER THAN i  
                PUSH onto min_heap the TUPLE OF the RATIO OF element at position i of arr DIVIDED BY element at position j MINUS one of arr AND i AND j MINUS one  
            END IF  
        END FOR  
        POP the SMALLEST element FROM min_heap AND ASSIGN TO IGNORE VARIABLE unnamed AND i AND j  
        RETURN LIST composed OF element at position i of arr FOLLOWED BY element at position j of arr  
    END FUNCTION  
END CLASS
