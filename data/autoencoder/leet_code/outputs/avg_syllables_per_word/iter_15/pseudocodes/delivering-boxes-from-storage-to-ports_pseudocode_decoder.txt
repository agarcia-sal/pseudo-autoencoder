CLASS Solution  
    FUNCTION boxDelivering(boxes AS List of List of integer, portsCount AS integer, maxBoxes AS integer, maxWeight AS integer) RETURNS integer  
        SET n TO the LENGTH OF boxes  
        INITIALIZE dp AS list of length n PLUS one with all elements set TO infinity  
        SET element at position zero of dp TO zero  
        
        SET left TO zero  
        SET current_weight TO zero  
        SET extra_trips TO zero  
        
        FOR right FROM zero TO n MINUS one  
            INCREMENT current_weight BY the element at position one of element at position right of boxes  
            IF right IS GREATER THAN zero AND the element at position zero of element at position right of boxes IS NOT EQUAL TO the element at position zero of element at position right MINUS one of boxes THEN  
                INCREMENT extra_trips BY one  
            END IF  
            
            WHILE left IS LESS THAN OR EQUAL TO right AND (the difference between right AND left PLUS one IS GREATER THAN maxBoxes OR current_weight IS GREATER THAN maxWeight) HOLDS  
                DECREMENT current_weight BY the element at position one of element at position left of boxes  
                IF left IS GREATER THAN zero AND the element at position zero of element at position left of boxes IS NOT EQUAL TO the element at position zero of element at position left MINUS one of boxes THEN  
                    DECREMENT extra_trips BY one  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET element at position right PLUS one of dp TO the sum of the element at position left of dp PLUS two PLUS extra_trips  
        END FOR  
        
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
