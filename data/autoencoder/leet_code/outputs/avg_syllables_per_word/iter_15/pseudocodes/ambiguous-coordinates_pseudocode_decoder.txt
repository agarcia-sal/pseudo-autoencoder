CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO substring from position one TO d of frag  
                SET right TO substring from position d PLUS one TO n of frag  
                IF left DOES NOT START WITH character zero OR left EQUALS character zero AND right DOES NOT END WITH character zero THEN  
                    YIELD concatenation of left PLUS decimal point IF d IS NOT EQUAL TO n OTHERWISE empty string PLUS right  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO substring from position two TO the LENGTH OF s MINUS one of s  
        RETURN list constructed by FOR i FROM one TO the LENGTH OF s MINUS one FOR each a IN make(substring from position one TO i of s) FOR each b IN make(substring from position i PLUS one TO the LENGTH OF s of s) COLLECT concatenation of open parenthesis PLUS a PLUS comma PLUS space PLUS b PLUS close parenthesis  
    END FUNCTION  
END CLASS
