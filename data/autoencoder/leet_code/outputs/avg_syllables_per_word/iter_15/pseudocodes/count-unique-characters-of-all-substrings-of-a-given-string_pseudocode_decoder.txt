CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO ten POWERED BY nine PLUS one  
        SET n TO the LENGTH OF s  
        SET last TO a LIST OF integer minus one REPEATED twenty six TIMES  
        SET last2 TO a LIST OF integer minus one REPEATED twenty six TIMES  
        SET result TO zero  
        FOR each index i AND character char IN the ENUMERATED collection s  
            SET idx TO the UNICODE NUMERIC VALUE OF char MINUS the UNICODE NUMERIC VALUE OF capital letter A  
            SET result TO result PLUS the DIFFERENCE BETWEEN i AND the ELEMENT at position idx OF last MULTIPLIED BY the DIFFERENCE BETWEEN the ELEMENT at position idx OF last AND the ELEMENT at position idx OF last2  
            SET result TO the REMAINDER OF result DIVIDED BY MOD  
            SET the ELEMENT at position idx OF last2 TO the ELEMENT at position idx OF last  
            SET the ELEMENT at position idx OF last TO i  
        END FOR  
        FOR each integer i FROM zero TO twenty five  
            SET result TO result PLUS the DIFFERENCE BETWEEN n AND the ELEMENT at position i OF last MULTIPLIED BY the DIFFERENCE BETWEEN the ELEMENT at position i OF last AND the ELEMENT at position i OF last2  
            SET result TO the REMAINDER OF result DIVIDED BY MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
