CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO list of boolean FALSE values with length equal to n PLUS one  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE the square of j IS LESS THAN OR EQUAL TO i  
                IF the boolean value at position i MINUS the square of j of dp IS FALSE  
                    SET the boolean value at position i of dp TO TRUE  
                    BREAK out of the WHILE loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN the boolean value at position n of dp  
    END FUNCTION  
END CLASS
