CLASS Solution  
    FUNCTION findMaximizedCapital(k, w, profits, capital)  
        SET projects TO collection of pairs formed by each element at position index in capital paired with element at position index in profits  
        SORT projects in ascending order based on the first element of each pair representing capital required  
        SET max_heap TO empty collection representing a max-heap for profits  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        FOR each count FROM one TO k  
            WHILE i LESS THAN n AND the first element of the pair at position i in projects LESS THAN OR EQUAL TO w  
                PUSH the negation of the second element of the pair at position i in projects INTO max_heap to simulate a max-heap  
                INCREMENT i BY one  
            END WHILE  
            IF max_heap is empty  
                BREAK the loop  
            END IF  
            SET w TO w PLUS the negation of the element obtained by removing the top element from max_heap  
        END FOR  
        RETURN w  
    END FUNCTION  
END CLASS
