CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO CONVERT wordDict INTO A SET OF UNIQUE WORDS  
        SET dp TO A LIST OF FALSE with LENGTH EQUAL TO THE LENGTH OF s PLUS ONE  
        SET element at position zero of dp TO TRUE  
        FOR index i FROM one TO THE LENGTH OF s  
            FOR index j FROM zero TO ONE LESS THAN i  
                IF element at position j of dp EQUALS TRUE AND substring from position j TO i of s IS AN ELEMENT OF wordSet  
                    SET element at position i of dp TO TRUE  
                    EXIT THE INNER LOOP  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position LENGTH OF s of dp  
    END FUNCTION  
END CLASS
