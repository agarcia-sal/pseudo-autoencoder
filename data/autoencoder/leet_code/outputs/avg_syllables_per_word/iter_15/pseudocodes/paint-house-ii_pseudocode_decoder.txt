CLASS Solution  
    FUNCTION minCostII(costs)  
        IF costs EQUALS empty THEN  
            RETURN zero  
        END IF  

        SET n TO the LENGTH OF costs  
        SET k TO the LENGTH OF element at position zero of costs  

        IF k EQUALS one THEN  
            IF n EQUALS one THEN  
                RETURN element at position zero position zero of costs  
            ELSE  
                RETURN positive infinity  
            END IF  
        END IF  

        FOR i FROM one TO n MINUS one  
            SET min_cost TO positive infinity  
            SET min_cost_index TO negative one  
            SET second_min_cost TO positive infinity  

            FOR j FROM zero TO k MINUS one  
                IF element at position i MINUS one position j of costs LESS THAN min_cost THEN  
                    SET second_min_cost TO min_cost  
                    SET min_cost TO element at position i MINUS one position j of costs  
                    SET min_cost_index TO j  
                ELSE IF element at position i MINUS one position j of costs LESS THAN second_min_cost THEN  
                    SET second_min_cost TO element at position i MINUS one position j of costs  
                END IF  
            END FOR  

            FOR j FROM zero TO k MINUS one  
                IF j EQUALS min_cost_index THEN  
                    SET element at position i position j of costs TO element at position i position j of costs PLUS second_min_cost  
                ELSE  
                    SET element at position i position j of costs TO element at position i position j of costs PLUS min_cost  
                END IF  
            END FOR  
        END FOR  

        SET result TO positive infinity  
        FOR each cost IN element at position n MINUS one of costs  
            IF cost LESS THAN result THEN  
                SET result TO cost  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
