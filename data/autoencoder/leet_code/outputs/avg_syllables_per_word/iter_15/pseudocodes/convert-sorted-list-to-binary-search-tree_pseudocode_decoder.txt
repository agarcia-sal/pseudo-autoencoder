CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS None  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values OF TYPE list  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode INITIALIZED WITH element at position zero of values  
    SET p TO head  
    FOR each val IN elements from position one TO THE END OF values  
        SET node TO A NEW ListNode INITIALIZED WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the condition that the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND the result of is_same_list WITH PARAMETERS the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND left EQUALS None AND right EQUALS None  
        SET the val ATTRIBUTE OF self TO val  
        SET the left ATTRIBUTE OF self TO left  
        SET the right ATTRIBUTE OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values OF TYPE list  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO A NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element REMOVED FROM THE FRONT OF queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET the left ATTRIBUTE OF node TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND the left ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET the right ATTRIBUTE OF node TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND the right ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUALS TO the val ATTRIBUTE OF q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS the left ATTRIBUTE OF p AND the left ATTRIBUTE OF q AND the result of is_same_tree WITH PARAMETERS the right ATTRIBUTE OF p AND the right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST WITH PARAMETERS self AND head OF TYPE Optional OF ListNode TO RETURN Optional OF TreeNode  
        IF head EQUALS None  
            RETURN None  
        END IF  
        
        FUNCTION findMiddle WITH PARAMETERS left AND right  
            SET slow TO left  
            SET fast TO left  
            WHILE fast IS NOT EQUALS TO right AND the next ATTRIBUTE OF fast IS NOT EQUALS TO right  
                SET slow TO the next ATTRIBUTE OF slow  
                SET fast TO the next ATTRIBUTE OF the next ATTRIBUTE OF fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  
        
        FUNCTION convertListToBST WITH PARAMETERS left AND right  
            IF left IS EQUALS TO right  
                RETURN None  
            END IF  
            
            SET mid TO findMiddle WITH PARAMETERS left AND right  
            
            SET node TO A NEW TreeNode INITIALIZED WITH the val ATTRIBUTE OF mid  
            SET the left ATTRIBUTE OF node TO convertListToBST WITH PARAMETERS left AND mid  
            SET the right ATTRIBUTE OF node TO convertListToBST WITH PARAMETERS the next ATTRIBUTE OF mid AND right  
            RETURN node  
        END FUNCTION  
        
        RETURN convertListToBST WITH PARAMETERS head AND None  
    END FUNCTION  
END CLASS
