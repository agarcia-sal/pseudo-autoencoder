CLASS Solution  
    FUNCTION matrixScore(grid)  
        FOR each row IN grid  
            IF element at position zero of row EQUALS zero  
                FOR index FROM zero TO LENGTH OF row MINUS one  
                    SET element at position index of row TO one MINUS element at position index of row  
                END FOR  
            END IF  
        END FOR  

        FOR col FROM one TO LENGTH OF element at position zero of grid MINUS one  
            SET count_ones TO zero  
            FOR row FROM zero TO LENGTH OF grid MINUS one  
                INCREMENT count_ones BY element at position col of element at position row of grid  
            END FOR  
            IF count_ones LESS THAN LENGTH OF grid DIVIDED BY two  
                FOR row FROM zero TO LENGTH OF grid MINUS one  
                    SET element at position col of element at position row of grid TO one MINUS element at position col of element at position row of grid  
                END FOR  
            END IF  
        END FOR  

        SET score TO zero  
        FOR each row IN grid  
            SET binary_number TO concatenation of textual representation of each element in row  
            INCREMENT score BY integer interpretation of binary_number in base two  
        END FOR  

        RETURN score  
    END FUNCTION  
END CLASS
