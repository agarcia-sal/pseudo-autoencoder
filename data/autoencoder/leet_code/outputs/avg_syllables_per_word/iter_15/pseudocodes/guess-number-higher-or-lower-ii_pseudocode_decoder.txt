CLASS Solution  
    FUNCTION getMoneyAmount(n)  
        FUNCTION dp(left, right)  
            IF left GREATER THAN OR EQUAL TO right THEN  
                RETURN zero  
            END IF  
            SET min_cost TO positive infinity  
            FOR pivot FROM left TO right  
                SET cost TO pivot PLUS the GREATER VALUE BETWEEN the RESULT OF dp WITH PARAMETERS left AND pivot MINUS one AND the RESULT OF dp WITH PARAMETERS pivot PLUS one AND right  
                SET min_cost TO the LESSER VALUE BETWEEN min_cost AND cost  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN the RESULT OF dp WITH PARAMETERS one AND n  
    END FUNCTION  
END CLASS
