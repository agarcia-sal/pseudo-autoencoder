CLASS Solution  
    FUNCTION racecar(target)  
        SET queue TO a double ended queue initialized with a tuple consisting of zero position one speed and zero steps  
        SET visited TO a set initialized with a tuple consisting of zero position and one speed  
        
        WHILE queue is not empty  
            REMOVE and ASSIGN the first tuple of queue TO position speed and steps  
            
            IF position EQUALS target  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF a tuple consisting of next_position and next_speed is not contained in visited AND next_position is greater than or equal to zero AND next_position is less than two MULTIPLIED BY target  
                ADD the tuple consisting of next_position and next_speed TO visited  
                APPEND a tuple consisting of next_position next_speed and steps PLUS one TO queue  
            END IF  
            
            IF speed is greater than zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF a tuple consisting of position and next_speed is not contained in visited AND position is greater than or equal to zero AND position is less than two MULTIPLIED BY target  
                ADD the tuple consisting of position and next_speed TO visited  
                APPEND a tuple consisting of position next_speed and steps PLUS one TO queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
