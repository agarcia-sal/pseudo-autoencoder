CLASS Solution  
    FUNCTION maxNumOfSubstrings(s)  
        SET first TO empty dictionary  
        SET last TO empty dictionary  
        FOR each index AND character IN enumeration of string s  
            IF character NOT IN first  
                SET element at position character of first TO index  
            END IF  
            SET element at position character of last TO index  
        END FOR  
          
        SET intervals TO empty list  
        FOR each character IN keys of first  
            SET start TO element at position character of first  
            SET end TO element at position character of last  
            SET i TO start  
            WHILE i IS LESS THAN OR EQUAL TO end  
                IF element at position element at position i of string s of first IS LESS THAN start  
                    BREAK the loop  
                END IF  
                SET end TO the GREATER OF end AND element at position element at position i of string s of last  
                INCREMENT i BY one  
            END WHILE  
            IF loop did not break  
                APPEND pair consisting of start and end TO intervals  
            END IF  
        END FOR  
          
        SORT intervals BY the second element of each pair in ascending order  
          
        SET result TO empty list  
        SET prev_end TO negative one  
        FOR each pair start and end IN intervals  
            IF start IS GREATER THAN prev_end  
                APPEND substring from position start TO end of string s TO result  
                SET prev_end TO end  
            END IF  
        END FOR  
          
        RETURN result  
    END FUNCTION  
END CLASS
