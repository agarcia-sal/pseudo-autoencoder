CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO one billion PLUS seven  
        SET dp TO list containing zero zero zero  
        FOR each num IN nums  
            IF num EQUALS zero  
                SET element at position zero of dp TO the remainder of the sum of two TIMES element at position zero of dp PLUS one DIVIDED BY MOD  
            ELSE IF num EQUALS one  
                SET element at position one of dp TO the remainder of the sum of two TIMES element at position one of dp PLUS element at position zero of dp DIVIDED BY MOD  
            ELSE IF num EQUALS two  
                SET element at position two of dp TO the remainder of the sum of two TIMES element at position two of dp PLUS element at position one of dp DIVIDED BY MOD  
            END IF  
        END FOR  
        RETURN element at position two of dp  
    END FUNCTION  
END CLASS
