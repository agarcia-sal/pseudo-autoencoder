CLASS Solution  
    FUNCTION topKFrequent(words AS List OF strings, k AS integer) RETURNS List OF strings  
        SET count TO the result of counting frequency occurrences of each word in words  
        SET heap TO an empty list  
        FOR each word and frequency IN count  
            APPEND a pair consisting of negative frequency and word TO heap  
        END FOR  
        CALL a function to transform heap into a heap structure  
        SET result TO an empty list  
        FOR each number FROM one TO k  
            REMOVE the smallest element from heap  
            APPEND the word part of the removed element TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
