CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i GREATER THAN j  
                RETURN zero  
            END IF  
            IF pair consisting of i AND j IS IN memo  
                RETURN value associated with pair consisting of i AND j IN memo  
            END IF  
            SET result TO call dp with parameters i, j MINUS one, AND memo PLUS one  
            FOR each k FROM i TO j MINUS one  
                IF character at position k of s EQUALS character at position j of s  
                    SET candidate TO call dp with parameters i, k MINUS one, AND memo PLUS call dp with parameters k, j MINUS one, AND memo  
                    IF candidate LESS THAN result  
                        SET result TO candidate  
                    END IF  
                END IF  
            END FOR  
            SET value associated with pair consisting of i AND j IN memo TO result  
            RETURN result  
        END FUNCTION  
        RETURN call dp with parameters zero, length of s MINUS one, AND an empty memo structure  
    END FUNCTION  
END CLASS
