CLASS Solution  
    FUNCTION countComponents WITH PARAMETERS n edges  
        SET graph TO an empty mapping from integer to list of integers  
        FOR each integer i FROM zero TO n MINUS one  
            SET an empty list as the value at position i of graph  
        END FOR  
        FOR each pair a and b IN edges  
            APPEND b TO the list at position a of graph  
            APPEND a TO the list at position b of graph  
        END FOR  
        
        SET visited TO an empty set  
        
        FUNCTION dfs WITH PARAMETER node  
            IF node IS IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list at position node of graph  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        
        SET components TO zero  
        FOR each integer node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                CALL dfs WITH node  
                INCREMENT components BY one  
            END IF  
        END FOR  
        
        RETURN components  
    END FUNCTION  
END CLASS
