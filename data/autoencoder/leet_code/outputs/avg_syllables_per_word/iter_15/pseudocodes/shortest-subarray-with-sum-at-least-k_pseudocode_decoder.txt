CLASS Solution  
    FUNCTION shortestSubarray(nums AS list of integers, k AS integer) RETURNS integer  
        SET prefix_sum TO a list containing zero as the only element  
        FOR each num IN nums  
            APPEND the sum of the element at the last position of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        ASSIGN pointer dq TO an empty double ended queue  
        SET min_length TO infinite positive value  
        FOR each index i WITH the corresponding element current_sum IN prefix_sum  
            WHILE the double ended queue dq is not empty AND the difference between current_sum AND the element at the position at the first place of dq of prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET min_length TO the lesser value BETWEEN min_length AND the difference between i AND the element popped from the first position of dq  
            END WHILE  
            WHILE the double ended queue dq is not empty AND current_sum IS LESS THAN OR EQUAL TO the element at the position at the last place of dq of prefix_sum  
                REMOVE the element from the last position of dq  
            END WHILE  
            APPEND i TO dq  
        END FOR  
        IF min_length IS NOT EQUAL TO infinite positive value  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
