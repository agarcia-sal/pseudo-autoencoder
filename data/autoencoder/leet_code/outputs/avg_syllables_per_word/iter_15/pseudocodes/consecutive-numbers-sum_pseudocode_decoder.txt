CLASS Solution  
    FUNCTION consecutiveNumbersSum(n)  
        SET count TO zero  
        FOR k FROM one TO n INCLUSIVE  
            IF the remainder of two MULTIPLIED BY n DIVIDED BY k EQUALS zero  
                SET m TO the quantity of two MULTIPLIED BY n DIVIDED BY k MINUS k PLUS one all DIVIDED BY two  
                IF m IS GREATER THAN zero AND m EQUALS m CONVERTED TO INTEGER  
                    INCREMENT count BY one  
                END IF  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
