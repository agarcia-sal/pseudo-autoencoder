CLASS Solution  
    FUNCTION addOperators(num, target) RETURNS list of strings  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO substring from position index TO position i MINUS one of num  
                IF the LENGTH OF operand GREATER THAN one AND the element at position zero of operand EQUALS character zero  
                    CONTINUE TO the next iteration  
                END IF  
                
                IF index EQUALS zero  
                    CALL dfs with parameters i, operand, CONVERT operand TO integer, CONVERT operand TO integer  
                ELSE  
                    CALL dfs with parameters i, path CONCATENATED WITH plus SIGN CONCATENATED WITH operand, value PLUS the integer value of operand, the integer value of operand  
                    CALL dfs with parameters i, path CONCATENATED WITH minus SIGN CONCATENATED WITH operand, value MINUS the integer value of operand, the NEGATIVE of the integer value of operand  
                    CALL dfs with parameters i, path CONCATENATED WITH asterisk SIGN CONCATENATED WITH operand, value MINUS prev PLUS prev MULTIPLIED BY the integer value of operand, prev MULTIPLIED BY the integer value of operand  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL dfs with parameters zero, empty string, zero, zero  
        RETURN result  
END CLASS
