CLASS Solution  
    FUNCTION convert WITH PARAMETERS s AND numRows  
        IF numRows EQUALS one  
            RETURN s  
        END IF  
        SET rows TO list of empty strings with length numRows  
        SET current_row TO zero  
        SET going_down TO false  
        FOR each character IN s  
            APPEND character TO element at position current_row of rows  
            IF current_row EQUALS zero OR current_row EQUALS numRows MINUS one  
                SET going_down TO NOT going_down  
            END IF  
            IF going_down IS true  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
        SET result TO concatenation of all elements in rows  
        RETURN result  
    END FUNCTION  
END CLASS
