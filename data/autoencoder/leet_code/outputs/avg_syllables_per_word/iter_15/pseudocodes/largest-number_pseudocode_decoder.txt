CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO the result of applying the conversion of each element in nums to string and collecting all results into a list  
        
        FUNCTION compare(x, y)  
            IF the concatenation of x followed by y is GREATER THAN the concatenation of y followed by x  
                RETURN negative one  
            ELSE IF the concatenation of x followed by y is LESS THAN the concatenation of y followed by x  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str using the comparison function compare converted to a key function
        
        IF the element at position zero of nums_str EQUALS the string zero  
            RETURN the string zero  
        END IF  
        
        SET result_string TO the concatenation of all elements in nums_str in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
