CLASS Solution  
    FUNCTION minCost(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO a list of pairs containing zero and one zero and MINUS one zero and one and MINUS one and zero  
        SET costs TO a two dimensional list with m rows and n columns where each element is set to infinity value  
        SET element at position zero zero of costs TO zero  
        SET pq TO a list containing a tuple of zero zero zero  
        SET visited TO an empty set  
        
        WHILE pq is not empty  
            REMOVE and RETURN the smallest element from pq and ASSIGN to cost i j  
            
            IF the pair of i and j is IN visited  
                CONTINUE to the next iteration of WHILE  
            END IF  
            ADD the pair of i and j to visited  
            
            IF i EQUALS m MINUS one AND j EQUALS n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each index k and the pair di dj in directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN m AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n  
                    IF k PLUS one IS NOT EQUAL TO element at position i j of grid  
                        SET additional_cost TO one  
                    ELSE  
                        SET additional_cost TO zero  
                    END IF  
                    SET new_cost TO cost PLUS additional_cost  
                    IF new_cost IS LESS THAN element at position ni nj of costs  
                        SET element at position ni nj of costs TO new_cost  
                        ADD the tuple of new_cost ni nj TO pq maintaining smallest element order  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
