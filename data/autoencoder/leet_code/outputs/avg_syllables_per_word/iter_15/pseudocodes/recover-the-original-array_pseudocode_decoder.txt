CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL sort METHOD ON nums  
        SET n TO the LENGTH OF nums  
        SET first TO the ELEMENT at position zero of nums  
        FOR i FROM one TO n MINUS one  
            SET k TO the INTEGER RESULT OF the ELEMENT at position i of nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS the ELEMENT at position i of nums  
                CONTINUE TO the NEXT iteration  
            END IF  
            SET count TO CALL Counter FUNCTION WITH nums AS ARGUMENT  
            SET arr TO an empty list  
            FOR each x IN nums  
                IF the VALUE ASSOCIATED WITH x IN count EQUALS zero  
                    CONTINUE TO the NEXT iteration  
                END IF  
                IF the VALUE ASSOCIATED WITH the ELEMENT LOCATED AT x PLUS two MULTIPLIED BY k IN count EQUALS zero  
                    BREAK out of the FOR loop  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT the VALUE ASSOCIATED WITH x IN count BY one  
                DECREMENT the VALUE ASSOCIATED WITH the ELEMENT LOCATED AT x PLUS two MULTIPLIED BY k IN count BY one  
            END FOR  
            IF the LENGTH OF arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an empty list  
    END FUNCTION  
END CLASS
