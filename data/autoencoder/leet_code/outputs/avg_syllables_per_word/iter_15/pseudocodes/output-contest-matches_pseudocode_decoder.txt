CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO a list of string representations of the numbers from one TO n  
        WHILE n GREATER THAN one  
            FOR each index i FROM zero TO n DIVIDED BY two MINUS one  
                SET the element at position i of teams TO an opening parenthesis FOLLOWED BY the element at position i of teams FOLLOWED BY a comma FOLLOWED BY the element at position n MINUS one MINUS i of teams FOLLOWED BY a closing parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two  
        END WHILE  
        RETURN the element at position zero of teams  
    END FUNCTION  
END CLASS
