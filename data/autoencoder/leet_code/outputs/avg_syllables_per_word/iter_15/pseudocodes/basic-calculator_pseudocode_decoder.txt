CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET sign TO plus character  
            SET num TO zero  
            
            WHILE the LENGTH OF s IS GREATER THAN zero  
                SET char TO element at position zero of s  
                REMOVE element at position zero FROM s  
                
                IF char IS a digit character  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                
                IF char IS an opening parenthesis character  
                    SET num TO helper invoked with s as argument  
                END IF  
                
                IF char IS NOT a digit character AND char IS NOT a whitespace character OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS plus character  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus character  
                        APPEND the negation of num TO stack  
                    ELSE IF sign EQUALS multiplication character  
                        SET element at the last position of stack TO element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS division character  
                        SET element at the last position of stack TO the integer conversion of element at the last position of stack DIVIDED BY num  
                    END IF  
                    
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                
                IF char IS a closing parenthesis character  
                    BREAK the loop  
                END IF  
            END WHILE  
            
            RETURN the sum of all elements in stack  
        END FUNCTION  
        
        RETURN helper invoked with the list conversion of s as argument  
    END FUNCTION  
END CLASS
