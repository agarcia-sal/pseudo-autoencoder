CLASS Solution  
    FUNCTION getHint(secret WITH TYPE string, guess WITH TYPE string) RETURNS string  
        SET bulls TO zero  
        SET cows TO zero  
        ASSIGN secret_count TO AN EMPTY DICTIONARY THAT PROVIDES zero FOR MISSING KEYS  
        ASSIGN guess_count TO AN EMPTY DICTIONARY THAT PROVIDES zero FOR MISSING KEYS  
        FOR each pair s AND g IN PARALLEL ITERATION OVER secret AND guess  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the VALUE ASSOCIATED WITH key s IN secret_count BY one  
                INCREMENT the VALUE ASSOCIATED WITH key g IN guess_count BY one  
            END IF  
        END FOR  
        FOR each digit IN THE COLLECTION OF KEYS IN secret_count  
            INCREMENT cows BY THE MINIMUM OF the VALUE ASSOCIATED WITH digit IN secret_count AND the VALUE ASSOCIATED WITH digit IN guess_count  
        END FOR  
        RETURN A STRING REPRESENTATION OF bulls CONCATENATED WITH uppercase A CONCATENATED WITH cows CONCATENATED WITH uppercase B  
    END FUNCTION  
END CLASS
