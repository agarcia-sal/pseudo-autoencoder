CLASS Solution  
    FUNCTION maximizeSweetness(sweetness AS List OF Integer, k AS Integer) RETURNS Integer  
        FUNCTION canDivide(min_sweet AS Integer) RETURNS Boolean  
            SET pieces TO zero  
            SET current_sweet TO zero  
            FOR each sweet IN sweetness  
                SET current_sweet TO current_sweet PLUS sweet  
                IF current_sweet GREATER THAN OR EQUAL TO min_sweet  
                    INCREMENT pieces BY one  
                    SET current_sweet TO zero  
                    IF pieces GREATER THAN k  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            RETURN pieces GREATER THAN k  
        END FUNCTION  

        SET left TO the MINIMUM value from sweetness  
        SET right TO the SUM of all elements in sweetness DIVIDED BY the sum of k PLUS one  
        SET result TO zero  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS right DIVIDED BY two  
            IF canDivide(mid)  
                SET result TO mid  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
