CLASS Solution  
    FUNCTION maxVacationDays(flights AS List OF List OF integer, days AS List OF List OF integer) RETURNS integer  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the element at position zero of days  
        
        SET dp TO a list of size n with all elements set to negative one  
        SET the element at position zero of dp TO zero  
        
        SET graph TO a new mapping from integer to list of integers  
        FOR each integer i FROM zero TO n MINUS one  
            ASSIGN an empty list TO the element at key i of graph  
        END FOR  
        
        FOR each integer i FROM zero TO n MINUS one  
            FOR each integer j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of flights EQUALS one THEN  
                    APPEND j TO the list at key i of graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR each integer week FROM zero TO k MINUS one  
            SET new_dp TO a list of size n with all elements set to negative one  
            FOR each integer i FROM zero TO n MINUS one  
                IF the element at position i of dp EQUALS negative one THEN  
                    CONTINUE to next iteration  
                END IF  
                IF the element at position i of new_dp LESS THAN dp at position i PLUS the element at position week of the element at position i of days THEN  
                    SET the element at position i of new_dp TO the element at position i of dp PLUS the element at position week of the element at position i of days  
                END IF  
                FOR each integer j IN the list at key i of graph  
                    IF the element at position j of new_dp LESS THAN dp at position i PLUS the element at position week of the element at position j of days THEN  
                        SET the element at position j of new_dp TO dp at position i PLUS the element at position week of the element at position j of days  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the maximum value in dp  
    END FUNCTION  
END CLASS
