CLASS Solution  
    FUNCTION longestPath(parent AS List OF integer, s AS string) RETURNS integer  
        INITIALIZE tree AS a mapping from integer TO list of integers WITH default empty list  
        
        FOR each index i AND element p IN enumeration of parent  
            IF p IS NOT EQUAL TO negative one  
                APPEND i TO element at position p OF tree  
            END IF  
        END FOR  
        
        SET result TO one  
        
        FUNCTION dfs(node AS integer) RETURNS integer  
            SET max1 TO zero  
            SET max2 TO zero  
            
            FOR each child IN element at position node OF tree  
                SET child_length TO CALL dfs WITH argument child  
                
                IF character at position child OF s IS NOT EQUAL TO character at position node OF s  
                    IF child_length IS GREATER THAN max1  
                        SET max2 TO max1  
                        SET max1 TO child_length  
                    ELSE IF child_length IS GREATER THAN max2  
                        SET max2 TO child_length  
                    END IF  
                END IF  
            END FOR  
            
            SET result TO the GREATER OF result AND max1 PLUS max2 PLUS one  
            
            RETURN max1 PLUS one  
        END FUNCTION  
        
        CALL dfs WITH argument zero  
        
        RETURN result  
    END FUNCTION  
END CLASS
