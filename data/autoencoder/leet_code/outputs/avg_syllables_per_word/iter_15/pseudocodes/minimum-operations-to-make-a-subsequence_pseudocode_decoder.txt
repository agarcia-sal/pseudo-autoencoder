CLASS Solution  
    FUNCTION minOperations(target AS List of integers ARR arr AS List of integers) RETURNS integer  
        SET index_map TO a mapping from each integer value to its corresponding index position in target  
        SET transformed_arr TO an empty list of integers  
        FOR each value IN arr  
            IF value EXISTS IN index_map  
                APPEND the element at position value in index_map TO transformed_arr  
            END IF  
        END FOR  
        
        SET lis TO an empty list of integers  
        FOR each num IN transformed_arr  
            SET pos TO the position at which num can be inserted in lis to maintain ascending order found by taking the leftmost insertion index of num in lis  
            IF pos EQUALS the length of lis  
                APPEND num TO lis  
            ELSE  
                SET the element at position pos of lis TO num  
            END IF  
        END FOR  
        
        SET lcs_length TO the length of lis  
        RETURN the length of target MINUS lcs_length  
    END FUNCTION  
END CLASS
