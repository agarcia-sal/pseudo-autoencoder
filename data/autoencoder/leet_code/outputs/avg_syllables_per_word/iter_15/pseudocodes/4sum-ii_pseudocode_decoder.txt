CLASS Solution  
    FUNCTION fourSumCount(nums1 LIST OF INTEGER nums2 LIST OF INTEGER nums3 LIST OF INTEGER nums4 LIST OF INTEGER) RETURNS INTEGER  
        SET sum_ab TO an empty dictionary mapping from integer to integer  
        FOR each element a IN nums1  
            FOR each element b IN nums2  
                SET current_sum TO a PLUS b  
                IF current_sum EXISTS AS A KEY IN sum_ab THEN  
                    INCREMENT the value at key current_sum IN sum_ab BY one  
                ELSE  
                    SET the value at key current_sum IN sum_ab TO one  
                END IF  
            END FOR  
        END FOR  
        SET count TO zero  
        FOR each element c IN nums3  
            FOR each element d IN nums4  
                SET target TO the NEGATION OF the sum of c PLUS d  
                IF target EXISTS AS A KEY IN sum_ab THEN  
                    INCREMENT count BY the value at key target IN sum_ab  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
