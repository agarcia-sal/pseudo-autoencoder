CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a new map with key zero assigned to position minus one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        FOR each index i AND character char IN the enumeration of s  
            SET digit TO the integer conversion of character char  
            SET prefix_xor TO the bitwise exclusive OR of prefix_xor WITH the value obtained by one shifted left by digit positions  
            IF prefix_xor EXISTS as a key in prefix_map  
                SET max_length TO the maximum value BETWEEN max_length AND the difference BETWEEN i AND the element at key prefix_xor of prefix_map  
            END IF  
            FOR each integer j FROM zero TO nine  
                SET flipped TO the bitwise exclusive OR of prefix_xor WITH the value obtained by one shifted left by j positions  
                IF flipped EXISTS as a key in prefix_map  
                    SET max_length TO the maximum value BETWEEN max_length AND the difference BETWEEN i AND the element at key flipped of prefix_map  
                END IF  
            END FOR  
            IF prefix_xor DOES NOT EXIST as a key in prefix_map  
                ASSIGN the element at key prefix_xor of prefix_map TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
