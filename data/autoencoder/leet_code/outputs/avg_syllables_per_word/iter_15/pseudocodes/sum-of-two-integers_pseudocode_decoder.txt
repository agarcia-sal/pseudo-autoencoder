CLASS Solution  
    FUNCTION getSum(a INTEGER_PARAMETER b INTEGER_PARAMETER) RETURNS INTEGER  
        SET MAX TO the integer value seven million one hundred seventy-four thousand eight hundred seventy-five  
        SET mask TO the integer value four billion two hundred ninety-four million nine hundred sixty-seven thousand two hundred ninety-five  
        WHILE the value of b IS NOT EQUAL TO zero  
            SET carry TO the bitwise AND of a AND b MULTIPLIED BY two  
            SET a TO the bitwise XOR of a AND b AND ANDed WITH mask  
            SET b TO carry AND ANDed WITH mask  
        END WHILE  
        IF a IS LESS THAN OR EQUAL TO MAX  
            RETURN the value of a  
        ELSE  
            RETURN the bitwise NOT of the bitwise XOR of a AND mask  
        END IF  
    END FUNCTION  
END CLASS
