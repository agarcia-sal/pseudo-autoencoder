CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO the LENGTH OF jobDifficulty  
        IF n LESS THAN d  
            RETURN negative one  
        END IF  
        FUNCTION dp(i, d)  
            IF d EQUALS one  
                RETURN the MAXIMUM value IN the SUBLIST of jobDifficulty FROM position i TO the END  
            END IF  
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR each index j FROM i TO n MINUS d PLUS one  
                SET current_max TO the MAXIMUM value BETWEEN current_max AND the ELEMENT at position j OF jobDifficulty  
                SET min_difficulty TO the MINIMUM value BETWEEN min_difficulty AND current_max PLUS the RESULT of dp WITH PARAMETERS j PLUS one AND d MINUS one  
            END FOR  
            RETURN min_difficulty  
        END FUNCTION  
        RETURN dp WITH PARAMETERS zero AND d  
    END FUNCTION  
END CLASS
