CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET i TO the LENGTH OF nums MINUS two  
        WHILE i IS GREATER THAN OR EQUAL TO zero AND the element at position i of nums IS GREATER THAN OR EQUAL TO the element at position i PLUS one of nums  
            DECREMENT i BY one  
        END WHILE  
        IF i IS GREATER THAN OR EQUAL TO zero  
            SET j TO the LENGTH OF nums MINUS one  
            WHILE the element at position j of nums IS LESS THAN OR EQUAL TO the element at position i of nums  
                DECREMENT j BY one  
            END WHILE  
            SET temporary_value TO the element at position i of nums  
            SET the element at position i of nums TO the element at position j of nums  
            SET the element at position j of nums TO temporary_value  
        END IF  
        CALL reverseSubsequence(nums, i PLUS one, the LENGTH OF nums MINUS one)  
    END FUNCTION  

    FUNCTION reverseSubsequence(nums, start_position, end_position)  
        SET left_pointer TO start_position  
        SET right_pointer TO end_position  
        WHILE left_pointer IS LESS THAN right_pointer  
            SET temporary_value TO the element at position left_pointer of nums  
            SET the element at position left_pointer of nums TO the element at position right_pointer of nums  
            SET the element at position right_pointer of nums TO temporary_value  
            INCREMENT left_pointer BY one  
            DECREMENT right_pointer BY one  
        END WHILE  
    END FUNCTION  
END CLASS
