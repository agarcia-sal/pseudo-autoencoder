CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero COMMA left EQUALS none COMMA right EQUALS none  
        SET the val ATTRIBUTE OF self TO val  
        SET the left ATTRIBUTE OF self TO left  
        SET the right ATTRIBUTE OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values OF TYPE list  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue data structure  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element REMOVED FROM the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left ATTRIBUTE OF node TO a new instance of TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the left ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right ATTRIBUTE OF node TO a new instance of TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the right ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p COMMA q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUAL TO the val ATTRIBUTE OF q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree CALLED WITH the left ATTRIBUTE OF p COMMA the left ATTRIBUTE OF q AND the result of is_same_tree CALLED WITH the right ATTRIBUTE OF p COMMA the right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow WITH PARAMETERS root OF TYPE Optional OF TreeNode COMMA val OF TYPE integer COMMA depth OF TYPE integer  
        IF depth EQUALS one  
            RETURN a new instance of TreeNode INITIALIZED WITH val AND left ATTRIBUTE SET TO root  
        END IF  
        SET queue TO a new double ended queue CONTAINING the TUPLE OF root AND integer one  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET node AND level TO the elements REMOVED FROM the front of queue  
            IF level EQUALS depth MINUS one  
                SET the left ATTRIBUTE OF node TO a new instance of TreeNode INITIALIZED WITH val AND left ATTRIBUTE SET TO the current left ATTRIBUTE OF node  
                SET the right ATTRIBUTE OF node TO a new instance of TreeNode INITIALIZED WITH val AND right ATTRIBUTE SET TO the current right ATTRIBUTE OF node  
            ELSE IF level IS LESS THAN depth MINUS one  
                IF the left ATTRIBUTE OF node IS NOT none  
                    APPEND the TUPLE OF the left ATTRIBUTE OF node AND level PLUS one TO queue  
                END IF  
                IF the right ATTRIBUTE OF node IS NOT none  
                    APPEND the TUPLE OF the right ATTRIBUTE OF node AND level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
