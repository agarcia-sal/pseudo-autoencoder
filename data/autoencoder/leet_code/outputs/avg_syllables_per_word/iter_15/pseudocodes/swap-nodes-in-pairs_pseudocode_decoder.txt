CLASS ListNode  
    FUNCTION __init__ with parameters val EQUALS zero and next EQUALS none  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the list values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO a new instance of ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values from position one TO the end  
        SET node TO a new instance of ListNode initialized with val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 and p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND the recursive call of is_same_list with parameters the next ATTRIBUTE OF p1 and the next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs with parameters self and head  
        SET dummy TO a new instance of ListNode initialized with zero  
        SET the next ATTRIBUTE OF dummy TO head  
        SET current TO dummy  
        WHILE the next ATTRIBUTE OF current IS NOT none AND the next ATTRIBUTE OF the next ATTRIBUTE OF current IS NOT none  
            SET first TO the next ATTRIBUTE OF current  
            SET second TO the next ATTRIBUTE OF the next ATTRIBUTE OF current  
            SET the next ATTRIBUTE OF current TO second  
            SET the next ATTRIBUTE OF first TO the next ATTRIBUTE OF second  
            SET the next ATTRIBUTE OF second TO first  
            SET current TO the next ATTRIBUTE OF the next ATTRIBUTE OF current  
        END WHILE  
        RETURN the next ATTRIBUTE OF dummy  
    END FUNCTION  
END CLASS
