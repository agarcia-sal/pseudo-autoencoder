CLASS Solution  
    FUNCTION sortColors(nums)  
        SET low TO zero  
        SET mid TO zero  
        SET high TO the LENGTH OF nums MINUS one  
        WHILE mid LESS THAN OR EQUAL TO high  
            IF element at position mid of nums EQUALS zero  
                SET temporary variable TO element at position mid of nums  
                SET element at position mid of nums TO element at position low of nums  
                SET element at position low of nums TO temporary variable  
                INCREMENT low BY one  
                INCREMENT mid BY one  
            ELSE IF element at position mid of nums EQUALS one  
                INCREMENT mid BY one  
            ELSE  
                SET temporary variable TO element at position mid of nums  
                SET element at position mid of nums TO element at position high of nums  
                SET element at position high of nums TO temporary variable  
                DECREMENT high BY one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
