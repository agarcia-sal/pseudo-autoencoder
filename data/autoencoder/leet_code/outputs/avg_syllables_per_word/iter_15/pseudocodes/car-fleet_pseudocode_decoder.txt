CLASS Solution  
    FUNCTION carFleet(target, position, speed)  
        SET paired_positions_and_speeds TO the list of pairs where each pair consists of element at position index of position and element at position index of speed for all indices FROM zero TO the length of position MINUS one  
        SORT paired_positions_and_speeds IN DESCENDING ORDER BASED ON the element at position zero of each pair  
        SET time_to_target TO an empty list  
        FOR each pair IN paired_positions_and_speeds  
            SET the local variable p TO the element at position zero of pair  
            SET the local variable s TO the element at position one of pair  
            SET the local variable time_to_reach TO target MINUS p DIVIDED BY s  
            APPEND time_to_reach TO time_to_target  
        END FOR  
        
        SET fleets TO zero  
        SET current_max_time TO zero  
        
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time THEN  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
        
        RETURN fleets  
    END FUNCTION  
END CLASS
