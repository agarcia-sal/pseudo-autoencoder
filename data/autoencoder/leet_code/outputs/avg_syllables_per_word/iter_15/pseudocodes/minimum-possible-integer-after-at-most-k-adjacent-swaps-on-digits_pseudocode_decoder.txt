CLASS Solution  
    FUNCTION minInteger(num AS string, k AS integer) AS string  
        SET num TO the LIST formed by the characters of num  
        SET n TO the LENGTH OF num  
        
        WHILE k GREATER THAN zero holds  
            SET swapped TO FALSE  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF the ELEMENT at position i of num GREATER THAN the ELEMENT at position i PLUS one of num  
                    SET min_digit TO the character nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO the MINIMUM OF i PLUS k PLUS one AND n MINUS one  
                        IF the ELEMENT at position j of num LESS THAN min_digit  
                            SET min_digit TO the ELEMENT at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET the SUBLIST of num FROM position i TO min_index PLUS one TO the LIST formed by the ELEMENT at position min_index of num FOLLOWED BY the SUBLIST of num FROM position i TO min_index MINUS one  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO TRUE  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            
            IF swapped EQUALS FALSE  
                BREAK the WHILE loop  
            END IF  
        END WHILE  
        
        SET result_string TO the STRING formed by joining all elements of num  
        RETURN result_string  
    END FUNCTION  
END CLASS
