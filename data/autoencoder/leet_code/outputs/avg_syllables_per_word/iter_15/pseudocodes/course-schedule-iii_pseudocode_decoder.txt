CLASS Solution  
    FUNCTION scheduleCourse(courses AS List OF List OF Integer) RETURNS Integer  
        CALL sort courses BY the element at position two of each course  
        SET max_heap TO empty list  
        SET current_end_day TO zero  
        FOR each course IN courses  
            SET duration TO the element at position one of course  
            SET last_day TO the element at position two of course  
            IF current_end_day PLUS duration IS LESS THAN OR EQUAL TO last_day THEN  
                CALL push negative duration INTO max_heap USING heap operation  
                INCREMENT current_end_day BY duration  
            ELSE IF max_heap IS NOT empty AND negative of the element at position one of max_heap IS GREATER THAN duration THEN  
                INCREMENT current_end_day BY duration PLUS removal of the smallest element from max_heap using heap operation  
                CALL push negative duration INTO max_heap USING heap operation  
            END IF  
        END FOR  
        RETURN the LENGTH OF max_heap  
    END FUNCTION  
END CLASS
