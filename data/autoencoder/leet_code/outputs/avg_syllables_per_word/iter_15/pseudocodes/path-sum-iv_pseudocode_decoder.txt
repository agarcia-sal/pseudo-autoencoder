CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
        FOR each num IN nums  
            SET depth TO the integer division of num BY one hundred  
            SET pos TO the integer division of num BY ten MODULO ten  
            SET value TO num MODULO ten  
            SET the element at the tuple of depth and pos in tree TO value  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the value associated with the tuple of depth and pos in tree  
            IF current_value IS NONE  
                RETURN zero  
            END IF  
            
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child TO the tuple of depth PLUS one and pos MULTIPLIED BY two MINUS one  
            SET right_child TO the tuple of depth PLUS one and pos MULTIPLIED BY two  
            
            IF left_child IS NOT IN tree AND right_child IS NOT IN tree  
                RETURN new_path_sum  
            END IF  
            
            SET left_sum TO dfs with the first element of left_child and the second element of left_child and new_path_sum  
            SET right_sum TO dfs with the first element of right_child and the second element of right_child and new_path_sum  
            
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN dfs with one and one and zero  
    END FUNCTION  
END CLASS
