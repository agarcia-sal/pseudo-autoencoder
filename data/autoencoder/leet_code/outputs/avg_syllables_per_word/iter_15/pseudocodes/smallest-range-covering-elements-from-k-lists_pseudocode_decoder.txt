CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        FOR index FROM zero TO LENGTH OF nums MINUS one  
            PUSH the element at position zero of the list at position index of nums TO min_heap WITH the associated index and zero as element index  
            SET max_value TO the GREATER THAN BETWEEN max_value AND the element at position zero of the list at position index of nums  
        END FOR  
        SET smallest_range TO infinity  
        SET result TO the list containing minus one hundred thousand AND one hundred thousand  
        WHILE min_heap is not empty  
            REMOVE the smallest value element FROM min_heap ASSIGN TO min_value list_index element_index  
            IF max_value MINUS min_value IS LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO the list containing min_value AND max_value  
            END IF  
            IF element_index PLUS one IS LESS THAN the LENGTH OF the list at position list_index of nums  
                SET next_value TO the element at position element_index PLUS one of the list at position list_index of nums  
                PUSH next_value AND list_index AND element_index PLUS one TO min_heap  
                SET max_value TO the GREATER THAN BETWEEN max_value AND next_value  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
