CLASS Solution  
    FUNCTION insert(intervals AS List OF List OF integer, newInterval AS List OF integer) AS List OF List OF integer  
        IF intervals EQUALS an empty list THEN  
            RETURN a list containing newInterval  
        END IF  
        SET merged TO an empty list  
        SET i TO zero  
        SET n TO the LENGTH OF intervals  
        WHILE i LESS THAN n AND the element at position one of element at position i of intervals LESS THAN the element at position zero of newInterval holds  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        WHILE i LESS THAN n AND the element at position zero of element at position i of intervals LESS THAN OR EQUAL TO the element at position one of newInterval holds  
            SET the element at position zero of newInterval TO the LOWER OF the element at position zero of newInterval AND the element at position zero of element at position i of intervals  
            SET the element at position one of newInterval TO the HIGHER OF the element at position one of newInterval AND the element at position one of element at position i of intervals  
            INCREMENT i BY one  
        END WHILE  
        APPEND newInterval TO merged  
        WHILE i LESS THAN n  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        RETURN merged  
    END FUNCTION  
END CLASS
