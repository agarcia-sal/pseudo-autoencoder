CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN true  
            END IF  
            SET expected TO the STRING REPRESENTATION OF first PLUS second  
            SET length TO the LENGTH OF expected  
            IF start PLUS length IS GREATER THAN the LENGTH OF num OR the SUBSTRING of num FROM position start TO position start PLUS length MINUS one IS NOT EQUALS expected  
                RETURN false  
            END IF  
            RETURN the RESULT OF is_valid_sequence WITH PARAMETERS start PLUS length SECOND AND the INTEGER VALUE OF expected  
        END FUNCTION  
        SET n TO the LENGTH OF num  
        FOR i FROM one TO the RESULT OF integer DIVISION OF n BY two PLUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET first TO the SUBSTRING of num FROM position zero TO position i MINUS one  
                SET second TO the SUBSTRING of num FROM position i TO position j MINUS one  
                IF the LENGTH OF first IS GREATER THAN one AND the ELEMENT at position zero of first EQUALS zero OR the LENGTH OF second IS GREATER THAN one AND the ELEMENT at position zero of second EQUALS zero  
                    CONTINUE to the next iteration  
                END IF  
                IF the RESULT OF is_valid_sequence WITH PARAMETERS j integer VALUE OF first AND integer VALUE OF second IS true  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
