CLASS Solution  
    FUNCTION findLongestChain(pairs)  
        CALL pairs SORTED BASED ON element at position two of each pair  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each pair consisting of start and end IN pairs  
            IF start IS GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
