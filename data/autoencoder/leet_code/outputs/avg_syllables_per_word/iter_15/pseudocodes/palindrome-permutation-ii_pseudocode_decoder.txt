CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND its count v IN cnt  
                IF v GREATER THAN one  
                    DECREMENT the count of c IN cnt BY two  
                    CALL dfs WITH the CONCATENATION of c PLUS t PLUS c  
                    INCREMENT the count of c IN cnt BY two  
                END IF  
            END FOR  
        END FUNCTION  

        SET cnt TO the COUNTER representation of s  
        SET mid TO empty string  
        FOR each character c AND its count v IN cnt  
            IF v MODULO two EQUALS one  
                IF mid IS NOT empty  
                    RETURN empty list  
                END IF  
                SET mid TO c  
                DECREMENT the count of c IN cnt BY one  
            END IF  
        END FOR  
        SET ans TO empty list  
        CALL dfs WITH mid  
        RETURN ans  
    END FUNCTION  
END CLASS
