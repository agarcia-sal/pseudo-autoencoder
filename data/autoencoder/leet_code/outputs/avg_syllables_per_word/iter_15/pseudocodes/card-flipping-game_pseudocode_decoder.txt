CLASS Solution  
    FUNCTION flipgame(fronts AS List OF integers, backs AS List OF integers) RETURNS integer  
        SET same TO an empty set  
        FOR each pair x and y IN the pairing of fronts and backs  
            IF x EQUALS y  
                ADD x TO the set same  
            END IF  
        END FOR  
        SET all_numbers TO the union of fronts and backs as a set  
        SET good_numbers TO the difference of all_numbers MINUS same  
        IF the set good_numbers is not empty  
            SET minimum_good_number TO the smallest value in good_numbers  
            RETURN minimum_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
