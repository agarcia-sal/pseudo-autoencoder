CLASS Solution  
    FUNCTION maxNumber(nums1, nums2, k)  
        FUNCTION maxSingleArray(nums, k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop IS GREATER THAN zero AND stack IS NOT empty AND element at the last position of stack IS LESS THAN num  
                    REMOVE element at the last position of stack FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the first k elements of stack  
        END FUNCTION  

        FUNCTION merge(arr1, arr2)  
            SET merged_list TO an empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR iteration FROM one TO total_length  
                IF arr1 IS GREATER THAN arr2 in lexicographical comparison WHEN CONSIDERING elements FROM their first positions  
                    APPEND element at position one of arr1 TO merged_list  
                    REMOVE element at position one FROM arr1  
                ELSE  
                    APPEND element at position one of arr2 TO merged_list  
                    REMOVE element at position one FROM arr2  
                END IF  
            END FOR  
            RETURN merged_list  
        END FUNCTION  

        SET max_number TO an empty list  
        FOR i FROM zero TO k  
            IF i IS LESS THAN OR EQUAL TO the LENGTH OF nums1 AND the result of k MINUS i IS LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO the result of merging maxSingleArray with nums1 and i AND maxSingleArray with nums2 and k MINUS i  
                IF max_number IS LESS THAN candidate in lexicographical comparison  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
