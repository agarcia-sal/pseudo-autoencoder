CLASS TreeNode  
    FUNCTION __init__(val SET TO zero, left SET TO none, right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO A new TreeNode with value equal to the element at position zero of values  
    SET i TO one  
    SET queue TO A new double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        REMOVE element FROM queue at the left end AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO A new TreeNode with value equal to the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO A new TreeNode with value equal to the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the value of p EQUALS the value of q IS False  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters element at attribute left of p AND element at attribute left of q AND the result of is_same_tree with parameters element at attribute right of p AND element at attribute right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF preorder IS empty OR inorder IS empty  
            RETURN none  
        END IF  
        SET root_val TO the element at position zero of preorder  
        SET root TO A new TreeNode with value root_val  
        SET root_index_inorder TO the position of root_val in inorder  
        SET root.left TO the result of calling buildTree on the elements from position one TO root_index_inorder PLUS one of preorder AND the elements from position zero TO root_index_inorder of inorder  
        SET root.right TO the result of calling buildTree on the elements from position root_index_inorder PLUS one TO the end of preorder AND the elements from position root_index_inorder PLUS one TO the end of inorder  
        RETURN root  
    END FUNCTION  
END CLASS
