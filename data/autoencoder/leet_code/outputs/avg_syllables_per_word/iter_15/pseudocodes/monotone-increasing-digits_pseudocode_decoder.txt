CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO list of characters representing the number n  
        SET length TO the LENGTH OF digits  
        SET marker TO length  
        FOR index FROM length MINUS one DOWN TO one  
            IF digit at position index of digits LESS THAN digit at position index MINUS one of digits  
                SET digit at position index MINUS one of digits TO string representation of the integer value of digit at position index MINUS one of digits MINUS one  
                SET marker TO index  
            END IF  
        END FOR  
        FOR index FROM marker TO length MINUS one  
            SET digit at position index of digits TO character nine  
        END FOR  
        SET result_string TO concatenation of all elements of digits  
        RETURN integer value of result_string  
    END FUNCTION  
END CLASS
