CLASS Solution  
    FUNCTION outerTrees(trees)  
        FUNCTION dist_sq(p1, p2)  
            RETURN the SQUARE of the DIFFERENCE between element at position zero of p1 AND element at position zero of p2 PLUS the SQUARE of the DIFFERENCE between element at position one of p1 AND element at position one of p2  
        END FUNCTION  
  
        FUNCTION circumcenter(p1, p2, p3)  
            SET ax TO element at position zero of p1  
            SET ay TO element at position one of p1  
            SET bx TO element at position zero of p2  
            SET by TO element at position one of p2  
            SET cx TO element at position zero of p3  
            SET cy TO element at position one of p3  
            SET d TO two MULTIPLIED BY the SUM of ax MULTIPLIED BY the DIFFERENCE of by MINUS cy PLUS bx MULTIPLIED BY the DIFFERENCE of cy MINUS ay PLUS cx MULTIPLIED BY the DIFFERENCE of ay MINUS by  
            SET ux TO the SUM of the SQUARE of ax PLUS the SQUARE of ay MULTIPLIED BY the DIFFERENCE of by MINUS cy PLUS the SUM of the SQUARE of bx PLUS the SQUARE of by MULTIPLIED BY the DIFFERENCE of cy MINUS ay PLUS the SUM of the SQUARE of cx PLUS the SQUARE of cy MULTIPLIED BY the DIFFERENCE of ay MINUS by DIVIDED BY d  
            SET uy TO the SUM of the SQUARE of ax PLUS the SQUARE of ay MULTIPLIED BY the DIFFERENCE of cx MINUS bx PLUS the SUM of the SQUARE of bx PLUS the SQUARE of by MULTIPLIED BY the DIFFERENCE of ax MINUS cx PLUS the SUM of the SQUARE of cx PLUS the SQUARE of cy MULTIPLIED BY the DIFFERENCE of bx MINUS ax DIVIDED BY d  
            RETURN the list containing ux AND uy, and the SQUARE ROOT of the SUM of the SQUARE of the DIFFERENCE between ux AND ax AND the SQUARE of the DIFFERENCE between uy AND ay  
        END FUNCTION  
  
        FUNCTION welzl(points, boundary, n)  
            IF n EQUALS zero OR the LENGTH OF boundary EQUALS three  
                IF the LENGTH OF boundary EQUALS zero  
                    RETURN the list containing zero AND zero, and zero  
                ELSE IF the LENGTH OF boundary EQUALS one  
                    RETURN the sole element of boundary, and zero  
                ELSE IF the LENGTH OF boundary EQUALS two  
                    SET center TO the list containing the SUM of element at position zero of element at position zero of boundary AND element at position zero of element at position one of boundary DIVIDED BY two AND the SUM of element at position one of element at position zero of boundary AND element at position one of element at position one of boundary DIVIDED BY two  
                    SET radius TO the SQUARE ROOT of the DISTANCE SQUARED BETWEEN element at position zero of boundary AND element at position one of boundary DIVIDED BY two  
                    RETURN center AND radius  
                ELSE  
                    RETURN the result of calling circumcenter with unpacked elements of boundary  
                END IF  
            END IF  
  
            SET p TO element at position n MINUS one of points  
            SET center, radius TO the result of calling welzl with points, boundary, AND n MINUS one  
            IF the DISTANCE SQUARED BETWEEN center AND p IS LESS THAN OR EQUAL TO the SQUARE of radius  
                RETURN center AND radius  
            END IF  
  
            RETURN the result of calling welzl with points, boundary with p appended, AND n MINUS one  
        END FUNCTION  
  
        CALL the shuffling operation on trees TO RANDOMLY rearrange the order of elements  
        SET center, radius TO the result of calling welzl with trees, an empty list, AND the LENGTH OF trees  
        RETURN the list containing element at position zero of center, element at position one of center, AND radius  
    END FUNCTION  
END CLASS
