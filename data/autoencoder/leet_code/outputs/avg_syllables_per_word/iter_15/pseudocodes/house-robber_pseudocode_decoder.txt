CLASS Solution  
    FUNCTION rob(nums)  
        IF nums EQUALS NOTHING  
            RETURN zero  
        END IF  
        IF THE LENGTH OF nums EQUALS one  
            RETURN element at position one of nums  
        END IF  
        SET dp TO a list of zeros with the LENGTH OF nums  
        SET element at position one of dp TO element at position one of nums  
        SET element at position two of dp TO the GREATER OF element at position one of nums AND element at position two of nums  
        FOR i FROM three TO THE LENGTH OF nums  
            SET element at position i of dp TO the GREATER OF element at position i MINUS one of dp AND element at position i MINUS two of dp PLUS element at position i of nums  
        END FOR  
        RETURN element at the LAST POSITION of dp  
    END FUNCTION  
END CLASS
