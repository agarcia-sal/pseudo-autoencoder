CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO an empty mapping associating each integer FROM zero TO n MINUS one WITH an empty list  
        FOR each element containing u v w IN flights  
            APPEND the tuple of neighbor v and price w TO the list at position u of graph  
        END FOR  
        
        SET min_heap TO a list containing the tuple of zero cost zero minus one as stops and source city src  
        SET min_cost TO a mapping associating source city src WITH minus one as the number of stops  
        
        WHILE min_heap is not empty  
            EXTRACT the tuple of cost stops and city with minimum cost FROM min_heap  
            
            IF city EQUALS dst THEN  
                RETURN cost  
            END IF  
            
            IF stops IS GREATER THAN OR EQUAL TO k THEN  
                CONTINUE to the next iteration of the loop  
            END IF  
            
            FOR each element containing neighbor and price IN the list at position city of graph  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  
                
                IF neighbor is not in min_cost OR next_stops IS LESS THAN the value at neighbor in min_cost OR next_cost IS LESS THAN the minimum cost in the first element of the first tuple inside min_heap THEN  
                    SET min_cost at neighbor TO next_stops  
                    INSERT the tuple of next_cost next_stops and neighbor INTO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one as an indicator of no valid path found  
    END FUNCTION  
END CLASS
