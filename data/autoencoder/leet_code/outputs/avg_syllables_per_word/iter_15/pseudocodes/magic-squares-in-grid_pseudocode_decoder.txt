CLASS Solution  
    FUNCTION numMagicSquaresInside(grid)  
        FUNCTION is_magic(i, j)  
            IF element at position i j of grid IS NOT EQUAL TO five  
                RETURN false  
            END IF  
            SET numbers TO an empty set  
            FOR x FROM i MINUS one TO i PLUS one  
                FOR y FROM j MINUS one TO j PLUS one  
                    IF element at position x y of grid IS LESS THAN one OR element at position x y of grid IS GREATER THAN nine OR element at position x y of grid IS IN numbers  
                        RETURN false  
                    END IF  
                    ADD element at position x y of grid TO numbers  
                END FOR  
            END FOR  
            FOR start FROM i MINUS one TO i PLUS one  
                IF the sum of elements at positions start j MINUS one AND start j AND start j PLUS one of grid IS NOT EQUAL TO fifteen OR the sum of elements at positions i MINUS one start AND i start AND i PLUS one start of grid IS NOT EQUAL TO fifteen  
                    RETURN false  
                END IF  
            END FOR  
            IF the sum of elements at positions i MINUS one j MINUS one AND i j AND i PLUS one j PLUS one of grid IS NOT EQUAL TO fifteen OR the sum of elements at positions i MINUS one j PLUS one AND i j AND i PLUS one j MINUS one of grid IS NOT EQUAL TO fifteen  
                RETURN false  
            END IF  
            RETURN true  
        END FUNCTION  
        SET count TO zero  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        FOR i FROM one TO rows MINUS one MINUS one  
            FOR j FROM one TO cols MINUS one MINUS one  
                IF is_magic(i j)  
                    INCREMENT count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
