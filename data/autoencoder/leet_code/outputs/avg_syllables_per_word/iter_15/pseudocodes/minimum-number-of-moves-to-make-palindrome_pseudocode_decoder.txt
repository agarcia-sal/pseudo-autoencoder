CLASS Solution  
    FUNCTION minMovesToMakePalindrome WITH PARAMETER s  
        SET s TO a list conversion of s  
        SET n TO the length of s  
        SET moves TO zero  
        
        WHILE the length of s is greater than one  
            FOR index i FROM the length of s minus one DOWN TO one  
                IF the element at position i of s EQUALS the element at position zero of s  
                    FOR index j FROM i TO the length of s minus two  
                        SWAP the element at position j of s WITH the element at position j plus one of s  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE the element at position zero FROM s  
                    REMOVE the element at the last position FROM s  
                    EXIT the FOR loop  
                END IF  
            END FOR  
            IF no matching character was found in the FOR loop  
                REMOVE the element at position zero FROM s  
            END IF  
        END WHILE  
        
        RETURN moves  
    END FUNCTION  
END CLASS
