CLASS Solution  
    FUNCTION minimumTimeRequired(jobs AS List OF integer, k AS integer) RETURNS integer  
        SET n TO the LENGTH OF jobs  
        
        SET subset_sums TO a list of zero replicated TWO RAISED TO THE POWER OF n times  
        
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO TWO RAISED TO THE POWER OF i MINUS one  
                SET element at position mask OR BITWISE OR OF ONE LEFT SHIFTED BY i OF subset_sums TO element at position mask of subset_sums PLUS element at position i of jobs  
            END FOR  
        END FOR  
        
        FUNCTION dp(k AS integer, mask AS integer) RETURNS integer  
            IF k EQUALS one THEN  
                RETURN element at position mask of subset_sums  
            END IF  
            
            SET res TO infinity  
            
            SET submask TO mask  
            WHILE submask IS GREATER THAN zero  
                SET temp TO THE MAXIMUM OF CALL dp WITH PARAMETERS k MINUS one AND mask XOR submask AND element at position submask of subset_sums  
                SET res TO THE MINIMUM OF res AND temp  
                SET submask TO bitwise AND OF submask MINUS one AND mask  
            END WHILE  
            
            RETURN res  
        END FUNCTION  
        
        RETURN CALL dp WITH PARAMETERS k AND TWO RAISED TO THE POWER OF n MINUS one  
    END FUNCTION  
END CLASS
