CLASS Solution  
    FUNCTION countBattleships WITH parameter board AS list of list of string  
        SET count TO zero  
        FOR each index i FROM zero TO the LENGTH OF board MINUS one  
            FOR each index j FROM zero TO the LENGTH OF element at position i of board MINUS one  
                IF the element at position i of board at position j EQUALS character X  
                    IF the index i EQUALS zero OR the element at position i MINUS one of board at position j EQUALS character DOT AND the index j EQUALS zero OR the element at position i of board at position j MINUS one EQUALS character DOT  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
