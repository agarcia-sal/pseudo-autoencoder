CLASS Solution  
    FUNCTION shortestDistance(maze, start, destination)  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO the LIST of PAIRS consisting of one and zero MINUS one and zero zero and one zero and MINUS one  
        SET start TO the TUPLE representation of start  
        SET destination TO the TUPLE representation of destination  
        SET visited TO an EMPTY SET  
        SET pq TO a LIST containing a PAIR of zero and start  

        WHILE pq is not empty  
            EXTRACT the element with the LOWEST distance and ASSIGN to dist and position pair x and y from pq  

            IF the PAIR x and y EQUALS destination  
                RETURN dist  
            END IF  

            IF the PAIR x and y IS IN visited  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  

            ADD the PAIR x and y TO visited  

            FOR each PAIR dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  

                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the element at position new_x PLUS dx of maze AT position new_y PLUS dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  

                IF the PAIR new_x and new_y IS NOT IN visited  
                    ADD a PAIR of dist PLUS count and a PAIR of new_x and new_y TO pq PRESERVING priority order BY distance  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
