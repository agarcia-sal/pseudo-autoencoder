CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO A NEW ListNode WITH value AS element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO THE END  
        SET node TO A NEW ListNode WITH value AS val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN boolean result of the comparison of the val element of p1 EQUALS the val element of p2  
          AND recursive call to is_same_list with the next element of p1 and the next element of p2  
END FUNCTION

CLASS Solution  
    FUNCTION reorderList(self, head)  
        IF head EQUALS None OR the next element of head EQUALS None  
            RETURN  
        END IF  

        SET slow TO head  
        SET fast TO head  
        WHILE fast IS NOT None AND the next element of fast IS NOT None  
            SET slow TO the next element of slow  
            SET fast TO the element two positions after fast  
        END WHILE  

        SET prev TO None  
        SET curr TO slow  
        WHILE curr IS NOT None  
            SET next_temp TO the next element of curr  
            SET the next element of curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  

        SET first TO head  
        SET second TO prev  
        WHILE the next element of second IS NOT None  
            SET temp1 TO the next element of first  
            SET temp2 TO the next element of second  
            SET the next element of first TO second  
            SET the next element of second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
