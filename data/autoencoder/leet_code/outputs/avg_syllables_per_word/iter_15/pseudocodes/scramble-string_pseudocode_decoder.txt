CLASS Solution  
    FUNCTION isScramble(s1 AS string, s2 AS string) RETURNS boolean  
        IF s1 EQUALS s2 THEN  
            RETURN true  
        END IF  
        
        IF the SORTED characters of s1 NOT EQUALS the SORTED characters of s2 THEN  
            RETURN false  
        END IF  
        
        SET n TO the LENGTH OF s1  
        FOR i FROM one TO n MINUS one  
            IF the CALL to isScramble with the SUBSTRING of s1 from position one TO i AND the SUBSTRING of s2 from position one TO i RETURNS true  
            AND the CALL to isScramble with the SUBSTRING of s1 from position i PLUS one TO n AND the SUBSTRING of s2 from position i PLUS one TO n RETURNS true THEN  
                RETURN true  
            END IF  
            
            IF the CALL to isScramble with the SUBSTRING of s1 from position one TO i AND the SUBSTRING of s2 from position n MINUS i PLUS one TO n RETURNS true  
            AND the CALL to isScramble with the SUBSTRING of s1 from position i PLUS one TO n AND the SUBSTRING of s2 from position one TO n MINUS i RETURNS true THEN  
                RETURN true  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
