CLASS Solution  
    FUNCTION longestConsecutive(nums)  
        IF the collection nums is empty  
            RETURN zero  
        END IF  
        SET num_set TO the set of unique elements from nums  
        SET longest_streak TO zero  
        FOR each num IN num_set  
            IF the element equal to num MINUS one is NOT IN num_set  
                SET current_num TO num  
                SET current_streak TO one  
                WHILE the element equal to current_num PLUS one is IN num_set  
                    SET current_num TO current_num PLUS one  
                    INCREMENT current_streak BY one  
                END WHILE  
                SET longest_streak TO the maximum OF longest_streak AND current_streak  
            END IF  
        END FOR  
        RETURN longest_streak  
    END FUNCTION  
END CLASS
