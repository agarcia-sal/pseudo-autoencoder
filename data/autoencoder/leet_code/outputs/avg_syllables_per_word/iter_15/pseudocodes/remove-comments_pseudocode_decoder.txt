CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO False  
        SET result TO empty list  
        SET current_line TO empty list  
        FOR each line IN source  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF line  
                IF in_block_comment  
                    IF the element at position i OF line FOLLOWED BY the element at position i plus one OF line EQUALS the two characters A STERISK AND A SLASH in that order  
                        SET in_block_comment TO False  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF the element at position i OF line FOLLOWED BY the element at position i plus one OF line EQUALS the two characters FORWARD SLASH AND FORWARD SLASH in that order  
                        BREAK out of the WHILE loop  
                    ELSE IF the element at position i OF line FOLLOWED BY the element at position i plus one OF line EQUALS the two characters FORWARD SLASH AND A STERISK in that order  
                        SET in_block_comment TO True  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND the element at position i OF line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  
            IF current_line IS NOT empty AND in_block_comment IS False  
                APPEND the concatenation of all elements in current_line TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
