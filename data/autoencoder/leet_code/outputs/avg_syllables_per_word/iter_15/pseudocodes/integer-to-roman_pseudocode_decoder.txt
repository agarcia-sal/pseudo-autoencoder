CLASS Solution  
    FUNCTION intToRoman WITH parameter num  
        SET val TO a list containing one thousand nine hundred five hundred four hundred one hundred ninety fifty forty ten nine five four one  
        SET syms TO a list containing the string representing the Roman numeral for one thousand the Roman numeral for nine hundred the Roman numeral for five hundred the Roman numeral for four hundred the Roman numeral for one hundred the Roman numeral for ninety the Roman numeral for fifty the Roman numeral for forty the Roman numeral for ten the Roman numeral for nine the Roman numeral for five the Roman numeral for four the Roman numeral for one  
        SET roman_numeral TO an empty string  
        SET i TO zero  
        WHILE num IS GREATER THAN zero  
            FOR each element IN the sequence of the integer part of the division of num BY the element at position i of val  
                APPEND the element at position i of syms TO roman_numeral  
                DECREMENT num BY the element at position i of val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
