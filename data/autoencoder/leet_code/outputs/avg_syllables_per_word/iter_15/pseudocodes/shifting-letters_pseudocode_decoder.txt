CLASS Solution  
    FUNCTION shiftingLetters(s AS string, shifts AS list of integers) RETURNS string  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one DOWN TO zero  
            INCREMENT total_shifts BY the element at position index of shifts  
            SET the element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character_position TO the Unicode code point of the element at position index of s MINUS the Unicode code point of lowercase letter a  
            SET shifted_position TO character_position PLUS the element at position index of shifts  
            SET wrapped_position TO the remainder of shifted_position DIVIDED BY twenty six  
            SET new_character_code TO wrapped_position PLUS the Unicode code point of lowercase letter a  
            SET new_character TO the character represented by the Unicode code point new_character_code  
            APPEND new_character TO result  
        END FOR  
        RETURN the concatenation of all elements in result as a single string  
    END FUNCTION  
END CLASS
