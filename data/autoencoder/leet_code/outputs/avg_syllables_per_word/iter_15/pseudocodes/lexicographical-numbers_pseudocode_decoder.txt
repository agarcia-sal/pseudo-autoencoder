CLASS Solution  
    FUNCTION lexicalOrder WITH PARAMETER n  
        SET result TO an empty list  
        
        FUNCTION dfs WITH PARAMETER current  
            IF current IS GREATER THAN n  
                RETURN  
            END IF  
            APPEND current TO result  
            FOR each i FROM zero TO nine  
                SET next_num TO current MULTIPLIED BY ten PLUS i  
                IF next_num IS GREATER THAN n  
                    RETURN  
                END IF  
                CALL dfs WITH next_num  
            END FOR  
        END FUNCTION  
        
        FOR each i FROM one TO nine  
            CALL dfs WITH i  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
