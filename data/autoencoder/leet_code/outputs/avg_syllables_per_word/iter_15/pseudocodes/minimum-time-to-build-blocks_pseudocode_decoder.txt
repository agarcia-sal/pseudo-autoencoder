CLASS Solution  
    FUNCTION minBuildTime(blocks LIST OF INTEGER split INTEGER) RETURNS INTEGER  
        PERFORM heapify OPERATION ON blocks TO CONVERT IT INTO A MINIMUM HEAP  
        WHILE THE LENGTH OF blocks IS GREATER THAN one  
            SET first TO THE SMALLEST ELEMENT REMOVED FROM blocks USING HEAP REMOVAL  
            SET second TO THE NEXT SMALLEST ELEMENT REMOVED FROM blocks USING HEAP REMOVAL  
            SET combined_time TO split PLUS second  
            INSERT combined_time INTO blocks USING HEAP INSERTION  
        END WHILE  
        RETURN THE ELEMENT AT POSITION zero OF blocks  
    END FUNCTION  
END CLASS
