CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with value equal to element at position zero of values  
    SET i TO one  
    INITIALIZE queue AS an empty double ended queue  
    ADD root TO the end of queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        REMOVE element from the front of queue and SET TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO a new TreeNode with value equal to element at position i of values  
            ADD node.left TO the end of queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO a new TreeNode with value equal to element at position i of values  
            ADD node.right TO the end of queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(element at position left of p, element at position left of q) AND is_same_tree(element at position right of p, element at position right of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root IS none  
            RETURN an empty list  
        END IF  
        INITIALIZE result AS an empty list  
        INITIALIZE queue AS a double ended queue initialized with root as the only element  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            INITIALIZE current_level AS an empty list  
            FOR each number FROM one TO level_size  
                REMOVE element from the front of queue and SET TO node  
                APPEND element at position val of node TO current_level  
                IF element at position left of node IS NOT none  
                    ADD element at position left of node TO the end of queue  
                END IF  
                IF element at position right of node IS NOT none  
                    ADD element at position right of node TO the end of queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
