CLASS Solution  
    FUNCTION maxRunTime(n, batteries)  
        FUNCTION canRunFor(time)  
            SET total TO zero  
            FOR each battery IN batteries  
                SET total TO total PLUS the SMALLER BETWEEN battery AND time  
            END FOR  
            RETURN total GREATER THAN OR EQUAL TO time MULTIPLIED BY n  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the INTEGER RESULT OF the SUM OF all elements IN batteries DIVIDED BY n  
        
        WHILE left LESS THAN right  
            SET mid TO the INTEGER RESULT OF left PLUS right PLUS one DIVIDED BY two  
            IF canRunFor(mid) IS true  
                SET left TO mid  
            ELSE  
                SET right TO right MINUS one  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
