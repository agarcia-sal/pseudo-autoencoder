CLASS Solution  
    FUNCTION smallestGoodBase(n)  
        SET n TO the INTEGER CONVERSION OF n  
        SET max_length TO the INTEGER PART OF the BASE TWO LOGARITHM OF n INCREMENTED BY one  
        FOR m FROM max_length DOWN TO two  
            SET k TO the INTEGER PART OF n RAISED TO THE POWER OF one DIVIDED BY m MINUS one  
            IF k LESS THAN two THEN  
                CONTINUE TO the next iteration  
            END IF  
            SET num TO the INTEGER RESULT OF (k RAISED TO THE POWER OF m MINUS one) DIVIDED BY (k MINUS one)  
            IF num EQUALS n THEN  
                RETURN the STRING CONVERSION OF k  
            END IF  
        END FOR  
        RETURN the STRING CONVERSION OF n MINUS one  
    END FUNCTION  
END CLASS
