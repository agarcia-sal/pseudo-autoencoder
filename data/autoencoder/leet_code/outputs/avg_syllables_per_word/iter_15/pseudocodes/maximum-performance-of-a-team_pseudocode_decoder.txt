CLASS Solution  
    FUNCTION maxPerformance WITH PARAMETERS n speed efficiency k  
        SET engineers TO the LIST OF PAIRS OF corresponding elements FROM efficiency AND speed  
        SORT engineers IN ORDER OF descending efficiency  
        INITIALIZE speed_heap AS an empty collection to represent a minimum heap of speeds  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair eff AND spd IN engineers  
            ADD spd TO the speed_heap AND UPDATE its internal order to maintain minimum heap property  
            INCREMENT speed_sum BY spd  
            IF the LENGTH OF speed_heap IS GREATER THAN k  
                REMOVE the smallest element FROM speed_heap AND DECREMENT speed_sum BY that removed speed  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance IS GREATER THAN max_performance  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        RETURN the remainder WHEN max_performance IS DIVIDED BY one billion PLUS seven  
    END FUNCTION  
END CLASS
