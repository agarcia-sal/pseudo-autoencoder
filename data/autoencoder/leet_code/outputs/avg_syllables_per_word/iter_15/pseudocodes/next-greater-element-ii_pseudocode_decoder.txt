CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a list of the LENGTH OF n filled with negative one  
        SET stack TO an empty list  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO i MODULO n  
            WHILE the LENGTH OF stack is greater than zero AND the element at position the last element of stack in nums is LESS THAN the element at position index in nums  
                SET popped_index TO the element REMOVED FROM the end of stack  
                SET the element at position popped_index in result TO the element at position index in nums  
            END WHILE  
            IF i is LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
