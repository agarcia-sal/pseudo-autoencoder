CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        SET dp TO a two dimensional array OF SIZE m PLUS one BY n PLUS one FILLED WITH zeroes  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2  
                    SET the element at position i at position j of dp TO the element at position i MINUS one at position j MINUS one of dp PLUS one  
                ELSE  
                    SET the element at position i at position j of dp TO the GREATER OF the element at position i MINUS one at position j of dp AND the element at position i at position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        SET lcs_length TO the element at position m at position n of dp  
        RETURN the SUM of m MINUS lcs_length AND n MINUS lcs_length  
    END FUNCTION  
END CLASS
