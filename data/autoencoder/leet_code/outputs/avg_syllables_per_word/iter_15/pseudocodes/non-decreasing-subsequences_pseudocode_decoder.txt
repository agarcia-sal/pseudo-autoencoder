CLASS Solution  
    FUNCTION findSubsequences(nums)  
        FUNCTION dfs(u, last, t)  
            IF u EQUALS the LENGTH OF nums  
                IF the LENGTH OF t GREATER THAN one  
                    APPEND a COPY OF t TO ans  
                END IF  
                RETURN  
            END IF  
            IF the element at position u of nums GREATER THAN OR EQUAL TO last  
                APPEND the element at position u of nums TO t  
                CALL dfs WITH u INCREMENTED BY one AND last SET TO the element at position u of nums AND t AS PARAMETERS  
                REMOVE the last element FROM t  
            END IF  
            IF the element at position u of nums NOT EQUAL TO last  
                CALL dfs WITH u INCREMENTED BY one AND last AND t AS PARAMETERS  
            END IF  
        END FUNCTION  
        SET ans TO an empty list  
        CALL dfs WITH zero AND negative one thousand AND an empty list AS PARAMETERS  
        RETURN ans  
    END FUNCTION  
END CLASS
