CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a DICTIONARY mapping each character from a TO z TO zero  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND (the POSITION of the character at position index of s MINUS the POSITION of the character at position index MINUS one of s EQUALS one OR the character at position index of s EQUALS character a AND the character at position index MINUS one of s EQUALS character z)  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET max_length at key the character at position index of s TO the GREATER VALUE BETWEEN max_length at key the character at position index of s AND current_length  
        END FOR  
        SET total_unique_substrings TO the SUM OF all VALUES IN max_length  
        RETURN total_unique_substrings  
    END FUNCTION  
END CLASS
