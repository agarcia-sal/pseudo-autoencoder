CLASS Solution  
    FUNCTION numberOfWays WITH parameter corridor  
        SET MOD TO one billion PLUS seven  
        SET seats TO an empty list  
        FOR each position AND character IN corridor WITH enumeration starting from zero  
            IF character EQUALS uppercase letter S  
                APPEND position TO seats  
            END IF  
        END FOR  
        IF the LENGTH OF seats MODULO two NOT EQUAL TO zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM two TO the LENGTH OF seats MINUS one STEP two  
            SET difference TO element at position index of seats MINUS element at position index MINUS one of seats  
            SET ways TO ways MULTIPLIED BY difference MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
