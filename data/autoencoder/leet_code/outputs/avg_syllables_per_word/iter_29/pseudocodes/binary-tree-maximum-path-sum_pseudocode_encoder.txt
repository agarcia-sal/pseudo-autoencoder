CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EQUAL TO None OR the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode WITH the element at position one of values  
    SET i TO one  
    SET queue TO an EMPTY double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element REMOVED from the front of queue  
        IF i IS LESS THAN OR EQUAL TO the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node dot left TO A NEW TreeNode WITH the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN OR EQUAL TO the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node dot right TO A NEW TreeNode WITH the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p dot val IS NOT EQUAL TO q dot val  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH p dot left AND q dot left AND the RESULT OF is_same_tree WITH p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum(self, root)  
        FUNCTION helper(node)  
            IF node IS None  
                RETURN negative infinity, negative infinity  
            END IF  
            SET left_max, left_gain TO helper WITH node dot left  
            SET right_max, right_gain TO helper WITH node dot right  
            SET current_max TO node dot val PLUS the LARGER OF left_gain AND zero PLUS the LARGER OF right_gain AND zero  
            SET current_gain TO node dot val PLUS the LARGEST OF left_gain, right_gain, AND zero  
            SET global_max TO the LARGEST OF left_max, right_max, AND current_max  
            RETURN global_max, current_gain  
        END FUNCTION  
        RETURN the FIRST element OF helper WITH root  
    END FUNCTION  
END CLASS
