CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the list of substrings from splitting path by the forward slash character  
        SET stack TO an empty list  
        FOR each component IN components  
            IF the component EQUALS an empty string OR the component EQUALS a single period character  
                CONTINUE to the next iteration of the loop  
            ELSE IF the component EQUALS two period characters  
                IF the stack is not empty  
                    REMOVE the last element FROM the stack  
                END IF  
            ELSE  
                APPEND the component TO the stack  
            END IF  
        END FOR  
        SET simplified_path TO the string formed by the forward slash character concatenated WITH the concatenation of elements of stack separated by the forward slash character  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
