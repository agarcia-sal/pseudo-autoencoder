CLASS Solution  
    FUNCTION subsets(nums)  
        IF the collection nums IS empty  
            RETURN a list containing an empty list  
        END IF  
        SET subsets_without_first TO the result of calling subsets with the collection nums excluding the first element  
        SET subsets_with_first TO an empty list  
        FOR each subset IN subsets_without_first  
            SET new_subset TO a list created by putting the first element of nums followed by the elements of subset  
            APPEND new_subset TO subsets_with_first  
        END FOR  
        RETURN a new list created by concatenating subsets_without_first and subsets_with_first  
    END FUNCTION  
END CLASS
