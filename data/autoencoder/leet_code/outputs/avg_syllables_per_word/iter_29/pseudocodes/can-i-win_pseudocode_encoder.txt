CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the product of maxChoosableInteger AND the sum of maxChoosableInteger AND one divided by two IS LESS THAN desiredTotal  
            RETURN false  
        END IF  
          
        IF desiredTotal IS LESS THAN OR EQUAL TO zero  
            RETURN true  
        END IF  
          
        SET memo TO an empty collection of key value pairs  
          
        FUNCTION can_win(current_total, used_numbers)  
            IF the pair of current_total AND used_numbers IS PRESENT AS A KEY IN memo  
                RETURN the value associated with the key of current_total AND used_numbers in memo  
            END IF  
              
            FOR number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers AND the result of one shifted LEFT BY number EQUALS zero  
                    IF the sum of current_total AND number IS GREATER THAN OR EQUAL TO desiredTotal  
                        OR the negation of the result of calling can_win WITH the sum of current_total AND number AND the bitwise OR of used_numbers AND the result of one shifted LEFT BY number  
                        SET the value associated with the key of current_total AND used_numbers in memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
              
            SET the value associated with the key of current_total AND used_numbers in memo TO false  
            RETURN false  
        END FUNCTION  
          
        RETURN the result of calling can_win WITH zero AND zero  
    END FUNCTION  
END CLASS
