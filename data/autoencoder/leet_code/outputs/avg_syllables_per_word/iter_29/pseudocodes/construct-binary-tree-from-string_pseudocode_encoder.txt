CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty collection THEN  
        RETURN none  
    END IF  
    SET head TO NEW ListNode WITH the element at position one of values  
    SET p TO head  
    FOR each val IN the collection of elements of values starting from position two TO the end  
        SET node TO NEW ListNode WITH val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none THEN  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none THEN  
        RETURN false  
    END IF  
    RETURN the condition that the val field of p1 EQUALS the val field of p2 AND the recursive call to is_same_list WITH the next field of p1 AND the next field of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty collection THEN  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH the element at position one of values  
    SET i TO one  
    INITIALIZE an empty double-ended queue called queue  
    APPEND root TO queue  
    WHILE queue HOLDS elements DO  
        REMOVE the element from the front of queue and ASSIGN it TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values DOES NOT EQUAL none THEN  
            SET the left field of node TO NEW TreeNode WITH the element at position i of values  
            APPEND the left field of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values DOES NOT EQUAL none THEN  
            SET the right field of node TO NEW TreeNode WITH the element at position i of values  
            APPEND the right field of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN false  
    ELSE IF the val field of p DOES NOT EQUAL the val field of q THEN  
        RETURN false  
    ELSE  
        RETURN the logical AND of the recursive call to is_same_tree WITH the left field of p AND the left field of q AND the recursive call to is_same_tree WITH the right field of p AND the right field of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index IS GREATER THAN OR EQUAL TO the LENGTH OF s THEN  
                RETURN none AND index  
            END IF  
            SET start TO index  
            WHILE index IS LESS THAN the LENGTH OF s AND (the character at position index of s IS a digit OR the character at position index of s IS a minus sign) DO  
                INCREMENT index BY one  
            END WHILE  
            SET node TO NEW TreeNode WITH the number represented by substring from position start TO position index MINUS one of string s converted to integer  
            IF index IS LESS THAN the LENGTH OF s AND the character at position index of s EQUALS left parenthesis THEN  
                SET node dot left AND index TO the result of parse_tree WITH argument index PLUS one  
            END IF  
            IF index IS LESS THAN the LENGTH OF s AND the character at position index of s EQUALS left parenthesis THEN  
                SET node dot right AND index TO the result of parse_tree WITH argument index PLUS one  
            END IF  
            IF index IS LESS THAN the LENGTH OF s AND the character at position index of s EQUALS right parenthesis THEN  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND ignored_value TO parse_tree WITH argument zero  
        RETURN root  
    END FUNCTION  
END CLASS
