CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the list values IS empty  
        RETURN no value  
    END IF  
    SET head TO a new ListNode constructed with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two TO the end  
        SET node TO a new ListNode constructed with val  
        SET the next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS no value AND p2 IS no value  
        RETURN true  
    END IF  
    IF p1 IS no value OR p2 IS no value  
        RETURN false  
    END IF  
    RETURN the logical AND of the condition that the val of p1 EQUALS the val of p2 AND the recursive call of is_same_list with the next of p1 and the next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight(head, k)  
        IF head IS no value OR the next of head IS no value OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE the next of tail IS a value  
            SET tail TO the next of tail  
            INCREMENT length BY one  
        END WHILE  
        SET the next of tail TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR each number FROM one TO steps_to_new_tail  
            SET new_tail TO the next of new_tail  
        END FOR  
        SET new_head TO the next of new_tail  
        SET the next of new_tail TO no value  
        RETURN new_head  
    END FUNCTION  
END CLASS
