CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF houses EQUALS empty collection  
                RETURN zero  
            END IF  
            IF the LENGTH OF houses EQUALS one  
                RETURN element at position one of houses  
            END IF  
            SET dp TO a collection of zeros with the LENGTH OF houses  
            SET element at position one of dp TO element at position one of houses  
            SET element at position two of dp TO the GREATER OF element at position one of houses AND element at position two of houses  
            FOR index FROM three TO the LENGTH OF houses  
                SET element at position index of dp TO the GREATER OF element at position index MINUS one of dp AND the SUM OF element at position index MINUS two of dp AND element at position index of houses  
            END FOR  
            RETURN element at the last position of dp  
        END FUNCTION  
        IF nums EQUALS empty collection  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN element at position one of nums  
        END IF  
        RETURN the GREATER OF rob_linear of the substring from position one TO the position BEFORE the last position of nums AND rob_linear of the substring from position two TO the last position of nums  
    END FUNCTION  
END CLASS
