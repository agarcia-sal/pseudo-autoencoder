CLASS Solution  
    FUNCTION numRescueBoats(people_list OF integers, limit_value) RETURNS integer  
        SORT people_list IN ascending order by their values  
        SET left_pointer TO zero  
        SET right_pointer TO the LENGTH OF people_list MINUS one  
        SET boat_count TO zero  
        WHILE left_pointer LESS THAN OR EQUAL TO right_pointer  
            IF the value at position left_pointer of people_list PLUS the value at position right_pointer of people_list LESS THAN OR EQUAL TO limit_value  
                INCREMENT left_pointer BY one  
            END IF  
            DECREMENT right_pointer BY one  
            INCREMENT boat_count BY one  
        END WHILE  
        RETURN boat_count  
    END FUNCTION  
END CLASS
