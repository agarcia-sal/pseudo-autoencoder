CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode WITH parameter first element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO end of values  
        SET node TO new ListNode WITH parameter val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the expression that p1 dot val EQUALS p2 dot val AND is_same_list WITH parameters p1 dot next AND p2 dot next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1 l2)  
        SET dummy_head TO new ListNode WITH value zero  
        SET current TO dummy_head  
        SET carry TO zero  
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry IS GREATER THAN zero  
            IF l1 IS NOT None  
                SET val1 TO l1 dot val  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT None  
                SET val2 TO l2 dot val  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO the sum of val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten TAKING THE INTEGER PART  
            SET current dot next TO new ListNode WITH the remainder of total DIVIDED BY ten  
            SET current TO current dot next  
            IF l1 IS NOT None  
                SET l1 TO l1 dot next  
            END IF  
            IF l2 IS NOT None  
                SET l2 TO l2 dot next  
            END IF  
        END WHILE  
        RETURN dummy_head dot next  
    END FUNCTION  
END CLASS
