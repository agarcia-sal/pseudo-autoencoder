CLASS Solution  
    FUNCTION minKBitFlips(nums list_of_integers k integer) RETURNS integer  
        SET n TO the LENGTH OF nums  
        SET flip_count TO zero  
        SET flipped TO a new list of zeroes with the LENGTH OF n PLUS one  
        SET flip_state TO zero  
        
        FOR index FROM zero TO n MINUS one  
            SET flip_state TO the bitwise Exclusive OR of flip_state AND the element at position index of flipped  
            
            IF the element at position index of nums EQUALS flip_state THEN  
                
                IF index PLUS k GREATER THAN n THEN  
                    RETURN the negative of one  
                END IF  
                
                SET flip_state TO the bitwise Exclusive OR of flip_state AND one  
                SET the element at position index PLUS k of flipped TO the bitwise Exclusive OR of the element at position index PLUS k of flipped AND one  
                INCREMENT flip_count BY one  
            END IF  
        END FOR  
        
        RETURN flip_count  
    END FUNCTION  
END CLASS
