CLASS Solution  
    FUNCTION lastRemaining(n)  
        SET remaining TO n  
        SET step TO one  
        SET left_to_right TO True  
        SET head TO one  
        
        WHILE remaining GREATER THAN one  
            IF left_to_right EQUALS True OR the remainder of remaining DIVIDED BY two EQUALS one  
                INCREMENT head BY step  
            END IF  
            SET step TO step MULTIPLIED BY two  
            SET remaining TO the quotient of remaining DIVIDED BY two  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        
        RETURN head  
    END FUNCTION  
END CLASS
