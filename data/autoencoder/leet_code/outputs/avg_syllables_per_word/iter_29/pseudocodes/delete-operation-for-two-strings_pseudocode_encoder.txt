CLASS Solution  
    FUNCTION minDistance(word1 word1 STRING word2 word2 STRING) RETURNS integer  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        SET dp TO a two dimensional collection of zeroes with the first dimension size m PLUS one and the second dimension size n PLUS one  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the character at position i MINUS one of word1 EQUALS the character at position j MINUS one of word2  
                    SET the element at position i of the element at position j of dp TO the element at position i MINUS one of the element at position j MINUS one of dp PLUS one  
                ELSE  
                    SET the element at position i of the element at position j of dp TO the GREATER OF the element at position i MINUS one of the element at position j of dp AND the element at position i of the element at position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        SET lcs_length TO the element at position m of the element at position n of dp  
        RETURN m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  
END CLASS
