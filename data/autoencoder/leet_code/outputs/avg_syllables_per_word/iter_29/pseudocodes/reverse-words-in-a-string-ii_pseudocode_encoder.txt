CLASS Solution  
    FUNCTION reverseWords(list_of_characters)  
        REVERSE the order of elements IN list_of_characters  
        SET start_position TO zero  
        FOR each index_position FROM zero TO the LENGTH OF list_of_characters MINUS one  
            IF the element at position index_position of list_of_characters EQUALS a space character  
                REVERSE the order of elements from position start_position TO position index_position MINUS one of list_of_characters  
                SET start_position TO index_position PLUS one  
            END IF  
        END FOR  
        REVERSE the order of elements from position start_position TO the last position of list_of_characters  
    END FUNCTION  
END CLASS
