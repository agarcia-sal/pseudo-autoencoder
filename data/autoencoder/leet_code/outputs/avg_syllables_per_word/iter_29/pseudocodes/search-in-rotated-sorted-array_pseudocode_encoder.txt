CLASS Solution  
    FUNCTION search(nums left_to_right_list, target search_value) RETURNS integer  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS the result of right MINUS left DIVIDED BY two using integer division  
            
            IF the element at position mid of nums EQUALS search_value  
                RETURN mid  
            END IF  
            
            IF the element at position left of nums LESS THAN OR EQUAL TO the element at position mid of nums  
                IF the element at position left of nums LESS THAN OR EQUAL TO search_value AND search_value LESS THAN the element at position mid of nums  
                    SET right TO mid MINUS one  
                ELSE  
                    SET left TO mid PLUS one  
                END IF  
            ELSE  
                IF the element at position mid of nums LESS THAN search_value AND search_value LESS THAN OR EQUAL TO the element at position right of nums  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END IF  
        END WHILE  
        
        RETURN the integer negative one  
    END FUNCTION  
END CLASS
