CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s t)  
            SET iterator TO the ITERATOR OF t  
            RETURN whether for ALL characters in s each character IS CONTAINED IN iterator in sequence  
        END FUNCTION  
        
        SORT strs BY the NEGATION OF the LENGTH OF each element FOLLOWED BY the ELEMENT itself in LEXICOGRAPHICAL ORDER  
        
        FOR each index i AND word IN the ENUMERATION OF strs  
            SET unique TO true  
            FOR each index j AND other IN the ENUMERATION OF strs  
                IF i NOT EQUALS j AND is_subsequence(word other) IS true  
                    SET unique TO false  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF unique IS true  
                RETURN the LENGTH OF word  
            END IF  
        END FOR  
        
        RETURN the NEGATION OF one  
    END FUNCTION  
END CLASS
