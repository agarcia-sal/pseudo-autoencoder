CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO the set of elements in wordDict  
        SET dp TO a list of boolean values each set to false with the length equal to the length of s PLUS one  
        SET element at position zero of dp TO true  
        FOR i FROM one TO the length of s  
            FOR j FROM zero TO i MINUS one  
                IF element at position j of dp EQUALS true AND the substring from position j PLUS one TO position i of string s IS IN wordSet  
                    SET element at position i of dp TO true  
                    BREAK  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the length of s of dp  
    END FUNCTION  
END CLASS
