CLASS Solution  
    FUNCTION wordsTyping(sentence rows cols)  
        SET concatenated_sentence TO the concatenation of all elements of sentence separated by a single space and followed by a single space  
        SET total_length TO the LENGTH OF concatenated_sentence  
        SET position TO zero  
        
        FOR each iteration FROM one TO rows  
            INCREMENT position BY cols  
            
            IF the character at position MODULO total_length of concatenated_sentence EQUALS a space character THEN  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the character at the position MINUS one MODULO total_length of concatenated_sentence DOES NOT EQUAL a space character  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        
        RETURN position DIVIDED BY total_length  
    END FUNCTION  
END CLASS
