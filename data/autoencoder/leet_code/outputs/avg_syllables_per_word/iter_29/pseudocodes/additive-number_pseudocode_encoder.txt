CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start position OF number_string, first number, second number)  
            IF start position EQUALS the LENGTH OF number_string  
                RETURN true  
            END IF  
            
            SET sum_result TO first number PLUS second number  
            SET expected_string TO the string representation of sum_result  
            SET expected_length TO the LENGTH OF expected_string  
            
            IF start position PLUS expected_length GREATER THAN the LENGTH OF number_string OR the substring of number_string from start position TO start position PLUS expected_length DOES NOT EQUAL expected_string  
                RETURN false  
            END IF  
            
            RETURN the RESULT OF is_valid_sequence WITH start position INCREMENTED BY expected_length SECOND number AS first number AND the integer conversion of expected_string AS second number  
        END FUNCTION  
        
        SET total_length TO the LENGTH OF num  
        FOR i FROM one TO total_length DIVIDED BY two INCLUSIVE  
            FOR j FROM i PLUS one TO total_length MINUS one  
                SET first_substring TO the substring of num from position zero TO position i  
                SET second_substring TO the substring of num from position i TO position j  
                
                IF the LENGTH OF first_substring GREATER THAN one AND the character at position zero OF first_substring EQUALS zero OR the LENGTH OF second_substring GREATER THAN one AND the character at position zero OF second_substring EQUALS zero  
                    CONTINUE  
                END IF  
                
                IF is_valid_sequence WITH start position j FIRST number AS the integer conversion of first_substring SECOND number AS the integer conversion of second_substring RETURNS true  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
