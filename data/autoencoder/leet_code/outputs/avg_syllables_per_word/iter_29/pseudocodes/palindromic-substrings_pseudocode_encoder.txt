CLASS Solution  
    FUNCTION countSubstrings(s)  
        FUNCTION expand_around_center(left, right)  
            SET count TO zero  
            WHILE left IS GREATER THAN OR EQUAL TO zero AND right IS LESS THAN the LENGTH OF s AND character at position left of s EQUALS character at position right of s  
                INCREMENT count BY one  
                DECREMENT left BY one  
                INCREMENT right BY one  
            END WHILE  
            RETURN count  
        END FUNCTION  
        SET total_palindromes TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT total_palindromes BY expand_around_center(index, index)  
            INCREMENT total_palindromes BY expand_around_center(index, index PLUS one)  
        END FOR  
        RETURN total_palindromes  
    END FUNCTION  
END CLASS
