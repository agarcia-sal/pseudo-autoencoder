CLASS Solution  
    FUNCTION stoneGameVIII(stones)  
        SET prefix_sum TO a list of zeros with length EQUAL TO the LENGTH OF stones  
        SET element at position one of prefix_sum TO element at position one of stones  
        FOR index FROM two TO the LENGTH OF stones  
            SET element at position index of prefix_sum TO element at position index minus one of prefix_sum PLUS element at position index of stones  
        END FOR  
        SET dp TO element at the last position of prefix_sum  
        FOR index FROM the LENGTH OF stones MINUS two DOWN TO two DECREMENTING BY one  
            IF dp GREATER THAN OR EQUAL TO element at position index of prefix_sum MINUS dp THEN  
                SET dp TO dp  
            ELSE  
                SET dp TO element at position index of prefix_sum MINUS dp  
            END IF  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
