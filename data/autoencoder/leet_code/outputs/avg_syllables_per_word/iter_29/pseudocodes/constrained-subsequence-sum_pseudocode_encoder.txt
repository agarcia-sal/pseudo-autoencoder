CLASS Solution  
    FUNCTION constrainedSubsetSum(nums, k)  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zero repeated n times  
        SET element at position zero of dp TO element at position zero of nums  
        SET max_sum TO element at position zero of nums  
        SET q TO an empty double ended queue initialized with element zero  
        FOR i FROM one TO n MINUS one  
            IF element at position at the front of q of dp GREATER THAN zero  
                SET value_to_add TO element at position at the front of q of dp  
            ELSE  
                SET value_to_add TO zero  
            END IF  
            SET element at position i of dp TO element at position i of nums PLUS value_to_add  
            IF max_sum LESS THAN element at position i of dp  
                SET max_sum TO element at position i of dp  
            END IF  
            WHILE q IS NOT empty AND element at position i of dp GREATER THAN OR EQUAL TO element at position at the last position of q of dp  
                REMOVE element at the last position of q  
            END WHILE  
            APPEND i TO q  
            IF element at position at the front of q EQUALS i MINUS k  
                REMOVE element at the front of q  
            END IF  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
