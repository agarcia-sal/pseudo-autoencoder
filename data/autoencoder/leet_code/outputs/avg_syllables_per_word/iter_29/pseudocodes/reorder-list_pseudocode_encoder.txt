CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the INPUT list values IS empty  
        RETURN nothing  
    END IF  
    SET head TO a new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two TO the end  
        SET node TO a new ListNode with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS nothing AND p2 IS nothing  
        RETURN true  
    END IF  
    IF p1 IS nothing OR p2 IS nothing  
        RETURN false  
    END IF  
    RETURN the condition that the val element of p1 EQUALS the val element of p2 AND the recursive call to is_same_list with the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList(self, head)  
        IF head IS nothing OR the next element of head IS nothing  
            RETURN  
        END IF  

        SET slow TO head  
        SET fast TO head  
        WHILE fast IS NOT nothing AND the next element of fast IS NOT nothing  
            SET slow TO the next element of slow  
            SET fast TO the element two steps ahead from fast via next elements  
        END WHILE  

        SET prev TO nothing  
        SET curr TO slow  
        WHILE curr IS NOT nothing  
            SET next_temp TO the next element of curr  
            SET the next element of curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  

        SET first TO head  
        SET second TO prev  
        WHILE the next element of second IS NOT nothing  
            SET temp1 TO the next element of first  
            SET temp2 TO the next element of second  
            SET the next element of first TO second  
            SET the next element of second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
