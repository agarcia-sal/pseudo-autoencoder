CLASS Solution  
    FUNCTION maxDotProduct(nums1 list_of_integers nums2 list_of_integers) RETURNS integer  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        
        SET dp TO a two dimensional list with m PLUS one rows and n PLUS one columns WHERE each element IS the NEGATIVE INFINITY value  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                
                SET current_product TO the element at position i MINUS one of nums1 MULTIPLIED BY the element at position j MINUS one of nums2  
                
                SET dp element at position i and position j TO THE maximum value among the following  
                    current_product  
                    dp element at position i MINUS one and position j  
                    dp element at position i and position j MINUS one  
                    dp element at position i MINUS one and position j MINUS one PLUS current_product  
                
            END FOR  
        END FOR  
        
        RETURN dp element at position m and position n  
    END FUNCTION  
END CLASS
