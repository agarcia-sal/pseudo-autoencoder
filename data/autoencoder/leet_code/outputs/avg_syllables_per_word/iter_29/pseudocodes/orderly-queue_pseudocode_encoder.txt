CLASS Solution  
    FUNCTION orderlyQueue(parameter s parameter k)  
        IF parameter k GREATER THAN one  
            SET sorted_characters TO the characters of parameter s ARRANGED IN ascending alphabetical order  
            SET result_string TO concatenation of sorted_characters in their arranged order  
            RETURN result_string  
        END IF  
        SET length_of_string TO the LENGTH OF parameter s  
        SET smallest_rotation TO parameter s  
        FOR index FROM one TO length_of_string MINUS one  
            SET rotated_string TO substring from position index TO position length_of_string MINUS one of parameter s CONCATENATED WITH substring from position zero TO position index MINUS one of parameter s  
            IF rotated_string LESS THAN smallest_rotation  
                SET smallest_rotation TO rotated_string  
            END IF  
        END FOR  
        RETURN smallest_rotation  
    END FUNCTION  
END CLASS
