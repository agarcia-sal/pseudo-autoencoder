CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO a new mapping with default integer zero value  
        
        FOR each cpdomain IN cpdomains  
            SET count_and_domain TO the result of splitting cpdomain by whitespace  
            SET count_string TO element at position zero of count_and_domain  
            SET domain_string TO element at position one of count_and_domain  
            SET count TO the integer value represented by count_string  
            SET subdomains TO the result of splitting domain_string by the character period  
            
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one  
                SET subdomain_parts TO the elements of subdomains from position index TO the end  
                SET subdomain TO joining subdomain_parts by the character period as separator  
                INCREMENT the value at key subdomain in domain_count BY count  
            END FOR  
        END FOR  
        
        SET result_list TO an empty list  
        FOR each domain WITH associated count IN domain_count  
            SET domain_count_string TO the string representation of count  
            SET domain_string TO domain  
            SET combined_string TO the concatenation of domain_count_string PLUS one space character PLUS domain_string  
            APPEND combined_string TO result_list  
        END FOR  
        
        RETURN result_list  
    END FUNCTION  
END CLASS
