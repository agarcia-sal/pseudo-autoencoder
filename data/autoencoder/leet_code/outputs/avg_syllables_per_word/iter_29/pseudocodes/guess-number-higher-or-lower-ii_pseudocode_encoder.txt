CLASS Solution  
    FUNCTION getMoneyAmount(n)  
        FUNCTION dp(left, right)  
            IF left GREATER THAN OR EQUAL TO right  
                RETURN zero  
            END IF  
            SET min_cost TO positive infinity  
            FOR pivot FROM left TO right  
                SET cost TO pivot PLUS the GREATER OF dp(left, pivot MINUS one) AND dp(pivot PLUS one, right)  
                SET min_cost TO the SMALLER OF min_cost AND cost  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN dp(one, n)  
    END FUNCTION  
END CLASS
