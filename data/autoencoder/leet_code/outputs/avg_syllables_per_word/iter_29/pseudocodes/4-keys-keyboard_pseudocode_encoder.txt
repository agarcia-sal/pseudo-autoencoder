CLASS Solution  
    FUNCTION maxA(n)  
        IF n LESS THAN OR EQUAL TO three THEN  
            RETURN n  
        END IF  
        SET dp TO a list of zero repeated n PLUS one times  
        FOR i FROM one TO three  
            SET element at position i of dp TO i  
        END FOR  
        FOR i FROM four TO n  
            FOR j FROM i MINUS three DOWN TO one  
                SET element at position i of dp TO the maximum BETWEEN element at position i of dp AND the product of element at position j of dp AND i MINUS j MINUS one  
            END FOR  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
