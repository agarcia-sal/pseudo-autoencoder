CLASS Solution  
    FUNCTION minWindow(s1 s1_parameter, s2 s2_parameter)  
        SET min_len TO infinite large number  
        SET min_start TO minus one  
        SET len1 TO the LENGTH OF s1_parameter  
        SET len2 TO the LENGTH OF s2_parameter  
        
        FOR index i FROM zero TO len1 MINUS one  
            SET pointer j TO zero  
            FOR index k FROM i TO len1 MINUS one  
                IF the character at position k of s1_parameter EQUALS the character at position j of s2_parameter  
                    INCREMENT j BY one  
                    IF j EQUALS len2  
                        IF the value of k MINUS i PLUS one LESS THAN min_len  
                            SET min_len TO the value of k MINUS i PLUS one  
                            SET min_start TO i  
                        END IF  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF min_start EQUALS minus one  
            RETURN empty string description  
        ELSE  
            RETURN the substring from position min_start TO position min_start PLUS min_len MINUS one of s1_parameter  
        END IF  
    END FUNCTION  
END CLASS
