CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO TWO POWERED BY n MINUS one  
            SET good TO a list of FALSE repeated n times  
            SET bad TO a list of FALSE repeated n times  
            FOR j FROM zero TO n MINUS one  
                IF the result of the BITWISE AND between i and TWO POWERED BY j IS NOT EQUAL TO zero  
                    SET the element at position j of good TO TRUE  
                ELSE  
                    SET the element at position j of bad TO TRUE  
                END IF  
            END FOR  
            SET valid TO TRUE  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j of good IS TRUE  
                    FOR k FROM zero TO n MINUS one  
                        IF (the element at position j of statements AT position k EQUALS zero AND the element at position k of good IS TRUE) OR (the element at position j of statements AT position k EQUALS one AND the element at position k of bad IS TRUE)  
                            SET valid TO FALSE  
                            BREAK  
                        END IF  
                    END FOR  
                    IF valid IS FALSE  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            IF valid IS TRUE  
                SET max_good TO the GREATER OF max_good AND the SUM OF elements of good that ARE TRUE  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
