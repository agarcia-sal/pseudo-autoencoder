CLASS Solution  
    FUNCTION change(amount coins)  
        SET dp TO a list of zeroes with the length of amount PLUS one  
        SET element at position zero of dp TO one  
        FOR each coin IN coins  
            FOR x FROM coin TO amount  
                INCREMENT element at position x of dp BY element at position x MINUS coin of dp  
            END FOR  
        END FOR  
        RETURN element at position amount of dp  
    END FUNCTION  
END CLASS
