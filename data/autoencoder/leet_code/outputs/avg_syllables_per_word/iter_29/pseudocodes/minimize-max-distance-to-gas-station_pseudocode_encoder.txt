CLASS Solution  
    FUNCTION minmaxGasDist(stations, k)  
        FUNCTION possible(max_dist)  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                SET distance_between_stations TO element at position index of stations MINUS element at position index MINUS one of stations  
                SET stations_needed TO distance_between_stations DIVIDED BY max_dist CONVERTED TO WHOLE NUMBER  
                INCREMENT additional_stations BY stations_needed  
            END FOR  
            RETURN additional_stations LESS THAN OR EQUAL TO k  
        END FUNCTION  

        SET left TO zero  
        SET right TO ten POWERED BY eight  
        WHILE right MINUS left GREATER THAN one DIVIDED BY ten POWERED BY six  
            SET mid TO left PLUS right DIVIDED BY two  
            IF possible(mid)  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
