CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1 list_of_integers arr2 list_of_integers) RETURNS integer  
        PERFORM sorting of list arr2 in ascending order  
        
        SET dp TO a dictionary with key negative one VALUE zero representing zero operations for empty subsequence  
        
        FOR each number IN arr1  
            SET new_dp TO an empty dictionary  
            
            FOR each last_number VALUE operations IN items of dp  
                IF number GREATER THAN last_number THEN  
                    IF number NOT IN new_dp THEN  
                        SET new_dp at key number TO operations  
                    ELSE  
                        SET new_dp at key number TO the MINIMUM of existing value at new_dp key number AND operations  
                    END IF  
                END IF  
                
                SET index TO the position of the first element in arr2 strictly GREATER THAN last_number found by binary search for insertion point to the right  
                
                IF index LESS THAN the LENGTH OF arr2 THEN  
                    SET replacement_value TO element at position index of arr2  
                    IF replacement_value NOT IN new_dp THEN  
                        SET new_dp at key replacement_value TO operations PLUS one  
                    ELSE  
                        SET new_dp at key replacement_value TO the MINIMUM of existing value at new_dp key replacement_value AND operations PLUS one  
                    END IF  
                END IF  
            END FOR  
            
            IF new_dp IS empty THEN  
                RETURN negative one  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the MINIMUM of all values in dp  
    END FUNCTION  
END CLASS
