CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        
        SET dp TO a new collection where each element is a mapping from difference to count initialized as zero and the number of elements is n  
        SET total_count TO zero  
        
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO i MINUS one  
                SET difference d TO element at position i of nums MINUS element at position j of nums  
                
                IF dp at position j CONTAINS the key difference d  
                    INCREMENT dp at position i for key difference d BY the value of dp at position j for key difference d PLUS one  
                    INCREMENT total_count BY the value of dp at position j for key difference d  
                ELSE  
                    INCREMENT dp at position i for key difference d BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
