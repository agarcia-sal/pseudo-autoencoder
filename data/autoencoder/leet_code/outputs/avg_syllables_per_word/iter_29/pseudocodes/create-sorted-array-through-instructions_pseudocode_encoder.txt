CLASS Solution  
    FUNCTION createSortedArray(instructions)  
        SET MODULO TO ten POWERED BY nine PLUS one  
        SET MAXIMUM TO the MAXIMUM value IN instructions PLUS one  
        SET binary_indexed_tree TO a list OF zero VALUES WITH length MAXIMUM  
        
        FUNCTION update(index_parameter value_parameter)  
            WHILE index_parameter IS LESS THAN MAXIMUM  
                INCREMENT the element at position index_parameter OF binary_indexed_tree BY value_parameter  
                INCREMENT index_parameter BY the bitwise AND of index_parameter AND the NEGATION of index_parameter  
            END WHILE  
        END FUNCTION  
        
        FUNCTION query(index_parameter)  
            SET running_total TO zero  
            WHILE index_parameter IS GREATER THAN zero  
                INCREMENT running_total BY the element at position index_parameter OF binary_indexed_tree  
                DECREMENT index_parameter BY the bitwise AND of index_parameter AND the NEGATION of index_parameter  
            END WHILE  
            RETURN running_total  
        END FUNCTION  
        
        SET total_cost TO zero  
        FOR each pair index_variable number_variable IN the enumeration of instructions  
            SET count_less_than TO the result of calling query WITH argument number_variable MINUS one  
            SET count_greater_than TO index_variable MINUS the result of calling query WITH argument number_variable  
            IF count_less_than IS LESS THAN count_greater_than  
                SET cost_value TO count_less_than  
            ELSE  
                SET cost_value TO count_greater_than  
            END IF  
            SET total_cost TO the remainder of total_cost PLUS cost_value DIVIDED BY MODULO  
            CALL update WITH arguments number_variable AND one  
        END FOR  
        RETURN total_cost  
    END FUNCTION  
END CLASS
