CLASS Solution  
    FUNCTION topKFrequent(words list_of_strings k integer) RETURNS list_of_strings  
        SET count TO a collection that counts each unique string in words with their frequencies  
        SET heap TO an empty list  
        FOR each word and its frequency IN the collection count  
            APPEND a pair consisting of the negative of the frequency and the word TO heap  
        END FOR  
        TRANSFORM heap INTO a heap structure that arranges elements by their first item ascending and second item ascending  
        SET result TO an empty list  
        FOR iteration FROM one TO k  
            REMOVE the smallest element from heap and EXTRACT the word component  
            APPEND the extracted word TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
