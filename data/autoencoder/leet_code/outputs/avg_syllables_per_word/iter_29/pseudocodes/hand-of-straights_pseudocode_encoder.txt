CLASS Solution  
    FUNCTION isNStraightHand(hand group_of_integers groupSize integer) RETURNS boolean  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero  
            RETURN false  
        END IF  
        SET count TO A FREQUENCY MAPPING OF EACH ELEMENT IN hand  
        SET unique_cards TO A LIST OF KEYS OF count SORTED IN ASCENDING ORDER  
        FOR each card IN unique_cards  
            IF the FREQUENCY OF card IN count GREATER THAN zero  
                SET needed TO the FREQUENCY OF card IN count  
                FOR each index FROM zero TO groupSize MINUS one  
                    IF the FREQUENCY OF the ELEMENT AT POSITION card PLUS index IN count LESS THAN needed  
                        RETURN false  
                    END IF  
                    DECREMENT the FREQUENCY OF the ELEMENT AT POSITION card PLUS index IN count BY needed  
                END FOR  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
