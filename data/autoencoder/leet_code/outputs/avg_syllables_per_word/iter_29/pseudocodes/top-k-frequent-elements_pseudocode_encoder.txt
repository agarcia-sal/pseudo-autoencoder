CLASS Solution  
    FUNCTION topKFrequent(nums k)  
        SET frequency_map TO a new mapping with default value zero for each key  
        FOR each num IN nums  
            INCREMENT the value at key num in frequency_map BY one  
        END FOR  
        SET max_frequency TO the maximum value among all values in frequency_map  
        SET buckets TO a list of empty lists with the number of empty lists equal TO max_frequency PLUS one  
        FOR each pair of num and freq IN the entries of frequency_map  
            APPEND num TO the list at position freq of buckets  
        END FOR  
        SET result TO an empty list  
        FOR freq FROM max_frequency DOWN TO one  
            IF the length of the list at position freq of buckets IS GREATER THAN zero  
                EXTEND result BY all elements in the list at position freq of buckets  
                IF the length of result IS GREATER THAN OR EQUAL TO k  
                    RETURN the sublist consisting of the first k elements of result  
                END IF  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
