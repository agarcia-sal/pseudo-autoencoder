CLASS Solution  
    FUNCTION numberOfBoomerangs(points)  
        SET answer TO zero  
        FOR each point_one IN points  
            SET count_map TO a new empty counter collection  
            FOR each point_two IN points  
                SET delta_x TO the element at position zero of point_one MINUS the element at position zero of point_two  
                SET delta_y TO the element at position one of point_one MINUS the element at position one of point_two  
                SET distance_squared TO delta_x MULTIPLIED BY delta_x PLUS delta_y MULTIPLIED BY delta_y  
                INCREMENT answer BY the value associated with distance_squared in count_map  
                INCREMENT the value associated with distance_squared in count_map BY one  
            END FOR  
        END FOR  
        RETURN answer MULTIPLIED BY two  
    END FUNCTION  
END CLASS
