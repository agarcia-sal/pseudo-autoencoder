CLASS Solution  
    FUNCTION findMinHeightTrees(n, edges)  
        IF n EQUALS one  
            RETURN list containing zero  
        END IF  
          
        SET graph TO a mapping with default empty list for each key  
        SET degree TO a list of zeros with length n  
          
        FOR each pair of vertices u and v IN edges  
            APPEND v TO the list associated with u in graph  
            APPEND u TO the list associated with v in graph  
            INCREMENT the element at position u of degree BY one  
            INCREMENT the element at position v of degree BY one  
        END FOR  
          
        SET leaves TO a double ended queue initialized with each index i FROM zero TO one LESS THAN n WHERE the element at position i of degree EQUALS one  
          
        SET remaining_nodes TO n  
        WHILE remaining_nodes GREATER THAN two  
            SET leaves_count TO the length OF leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR each count FROM one TO leaves_count  
                REMOVE the first element FROM leaves AND ASSIGN it TO leaf  
                FOR each neighbor IN the list at key leaf OF graph  
                    DECREMENT the element at position neighbor of degree BY one  
                    IF the element at position neighbor of degree EQUALS one  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
          
        RETURN a list constructed from leaves  
    END FUNCTION  
END CLASS
