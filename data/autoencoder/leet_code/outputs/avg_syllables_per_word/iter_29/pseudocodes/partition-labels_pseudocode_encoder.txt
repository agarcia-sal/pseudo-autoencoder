CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO an empty dictionary mapping characters to positions  
        FOR each index AND character IN the sequence of positions and characters in s  
            SET last_occurrence at character TO index  
        END FOR  
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO an empty list  
        FOR each index AND character IN the sequence of positions and characters in s  
            SET end TO the greater value BETWEEN end AND the last_occurrence at character  
            IF index EQUALS end THEN  
                APPEND the difference BETWEEN end AND start PLUS one TO partition_sizes  
                SET start TO index PLUS one  
            END IF  
        END FOR  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
