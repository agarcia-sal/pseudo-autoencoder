CLASS Solution  
    FUNCTION checkValidString(s)  
        SET low TO zero  
        SET high TO zero  
        FOR each character IN s  
            IF character EQUALS opening parenthesis  
                INCREMENT low BY one  
                INCREMENT high BY one  
            ELSE IF character EQUALS closing parenthesis  
                DECREMENT low BY one  
                DECREMENT high BY one  
            ELSE  
                DECREMENT low BY one  
                INCREMENT high BY one  
            END ELSE IF  
            IF high LESS THAN zero  
                RETURN false  
            END IF  
            SET low TO the GREATER VALUE BETWEEN low AND zero  
        END FOR  
        RETURN whether low EQUALS zero  
    END FUNCTION  
END CLASS
