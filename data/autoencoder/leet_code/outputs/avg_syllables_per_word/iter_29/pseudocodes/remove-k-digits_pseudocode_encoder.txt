CLASS Solution  
    FUNCTION removeKdigits(num string, k integer) RETURNS string  
        SET stack TO an empty collection  
        
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND last element of stack GREATER THAN digit  
                REMOVE last element FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        
        IF k EQUALS zero  
            SET final_stack TO stack  
        ELSE  
            SET final_stack TO all elements of stack EXCEPT the last k elements  
        END IF  
        
        SET joined_result TO the concatenation of all elements in final_stack IN ORDER  
        
        SET result TO joined_result AFTER REMOVING all leading characters that ARE zero  
        
        IF result IS empty  
            RETURN the string representing zero  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
