CLASS Solution  
    FUNCTION catMouseGame(graph)  
        SET DRAW TO zero  
        SET MOUSE_WIN TO one  
        SET CAT_WIN TO two  
        SET n TO the LENGTH OF graph  
        
        FUNCTION dfs(turn mouse_pos cat_pos)  
            IF turn EQUALS two MULTIPLIED BY n THEN  
                RETURN DRAW  
            END IF  
            IF mouse_pos EQUALS zero THEN  
                RETURN MOUSE_WIN  
            END IF  
            IF mouse_pos EQUALS cat_pos THEN  
                RETURN CAT_WIN  
            END IF  
            
            IF turn MODULO two EQUALS zero THEN  
                SET current_player TO mouse  
            ELSE  
                SET current_player TO cat  
            END IF  
            
            IF current_player EQUALS mouse THEN  
                SET result TO CAT_WIN  
                FOR each next_pos IN the ELEMENTS connected to mouse_pos IN graph  
                    SET next_result TO dfs WITH turn INCREMENTED BY one AND next_pos AND cat_pos  
                    IF next_result EQUALS MOUSE_WIN THEN  
                        RETURN MOUSE_WIN  
                    ELSE IF next_result EQUALS DRAW THEN  
                        SET result TO DRAW  
                    END ELSE IF  
                END FOR  
                RETURN result  
            ELSE  
                SET result TO MOUSE_WIN  
                FOR each next_pos IN the ELEMENTS connected to cat_pos IN graph  
                    IF next_pos EQUALS zero THEN  
                        CONTINUE TO the next iteration  
                    END IF  
                    SET next_result TO dfs WITH turn INCREMENTED BY one AND mouse_pos AND next_pos  
                    IF next_result EQUALS CAT_WIN THEN  
                        RETURN CAT_WIN  
                    ELSE IF next_result EQUALS DRAW THEN  
                        SET result TO DRAW  
                    END ELSE IF  
                END FOR  
                RETURN result  
            END IF  
        END FUNCTION  
        
        RETURN dfs WITH zero mouse position one AND cat position two  
    END FUNCTION  
END CLASS
