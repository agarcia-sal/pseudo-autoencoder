CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the value of values IS EQUAL TO nothing THEN  
        RETURN nothing  
    END IF  
    SET head TO a new ListNode initialized with the element at position one of values  
    SET p TO head  
    FOR each val in the elements of values starting from position two TO the last element  
        SET node TO a new ListNode initialized with val  
        SET the next field of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS EQUAL TO nothing AND p2 IS EQUAL TO nothing THEN  
        RETURN true  
    END IF  
    IF p1 IS EQUAL TO nothing OR p2 IS EQUAL TO nothing THEN  
        RETURN false  
    END IF  
    RETURN the value field of p1 IS EQUAL TO the value field of p2 AND the result of calling is_same_list with the next field of p1 and the next field of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList(self, head)  
        IF head IS EQUAL TO nothing OR the next field of head IS EQUAL TO nothing THEN  
            RETURN head  
        END IF  

        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO the next field of head  
            WHILE fast IS NOT EQUAL TO nothing AND the next field of fast IS NOT EQUAL TO nothing  
                SET slow TO the next field of slow  
                SET fast TO the next field of the next field of fast  
            END WHILE  
            SET mid TO the next field of slow  
            SET the next field of slow TO nothing  
            RETURN head AND mid  
        END FUNCTION  

        FUNCTION merge(l1, l2)  
            SET dummy TO a new ListNode initialized with no arguments  
            SET current TO dummy  
            WHILE l1 IS NOT EQUAL TO nothing AND l2 IS NOT EQUAL TO nothing  
                IF the value field of l1 IS LESS THAN the value field of l2 THEN  
                    SET the next field of current TO l1  
                    SET l1 TO the next field of l1  
                ELSE  
                    SET the next field of current TO l2  
                    SET l2 TO the next field of l2  
                END IF  
                SET current TO the next field of current  
            END WHILE  
            IF l1 IS NOT EQUAL TO nothing THEN  
                SET the next field of current TO l1  
            ELSE  
                SET the next field of current TO l2  
            END IF  
            RETURN the next field of dummy  
        END FUNCTION  

        SET left TO the first output of calling split with head  
        SET right TO the second output of calling split with head  
        SET left TO the result of calling the method sortList of self with left  
        SET right TO the result of calling the method sortList of self with right  

        RETURN the result of calling merge with left and right  
    END FUNCTION  
END CLASS
