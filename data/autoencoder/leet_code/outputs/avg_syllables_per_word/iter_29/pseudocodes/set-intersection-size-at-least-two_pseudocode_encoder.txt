CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals BY the value at position two of each interval IN ascending order AND BY the negative value at position one of each interval IN ascending order  
        SET first TO the value at position two of the first interval MINUS one  
        SET second TO the value at position two of the first interval  
        SET count TO two  
        FOR each pair of start and end IN intervals starting from the second element TO the last element  
            IF start IS GREATER THAN second THEN  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first THEN  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END ELSE IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
