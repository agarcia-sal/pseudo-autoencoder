CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        WHILE n IS GREATER THAN length MULTIPLIED BY count  
            DECREMENT n BY length MULTIPLIED BY count  
            INCREMENT length BY one  
            MULTIPLY count BY ten  
            MULTIPLY start BY ten  
        END WHILE  
        INCREMENT start BY the INTEGER DIVISION OF (n MINUS one) BY length  
        RETURN the INTEGER VALUE OF the CHARACTER AT POSITION the REMAINDER OF (n MINUS one) DIVIDED BY length OF the STRING REPRESENTATION OF start  
    END FUNCTION  
END CLASS
