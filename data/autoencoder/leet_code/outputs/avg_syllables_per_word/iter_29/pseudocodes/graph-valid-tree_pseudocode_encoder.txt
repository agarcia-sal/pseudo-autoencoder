CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF the LENGTH OF edges NOT EQUALS n MINUS one  
            RETURN false  
        END IF  
        
        SET adjacency_list TO an empty mapping FROM each integer i FROM zero TO n MINUS one TO an empty list  
        FOR each pair of integers a and b IN edges  
            APPEND b TO the list at position a of adjacency_list  
            APPEND a TO the list at position b of adjacency_list  
        END FOR  
        
        SET visited_nodes TO an empty set  
        
        FUNCTION dfs(node)  
            IF node IS IN visited_nodes  
                RETURN  
            END IF  
            ADD node TO visited_nodes  
            FOR each neighbor IN the list at position node of adjacency_list  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        
        CALL dfs WITH zero  
        
        RETURN the LENGTH OF visited_nodes EQUALS n  
    END FUNCTION  
END CLASS
