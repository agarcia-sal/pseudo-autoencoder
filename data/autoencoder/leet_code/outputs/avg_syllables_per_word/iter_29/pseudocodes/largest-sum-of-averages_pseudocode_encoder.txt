CLASS Solution  
    FUNCTION largestSumOfAverages(nums parameter k parameter)  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of zero repeated the number n PLUS one TIMES  
        FOR index i FROM zero TO n MINUS one  
            SET element at position i PLUS one OF prefix_sum TO element at position i OF prefix_sum PLUS element at position i OF nums  
        END FOR  
        
        FUNCTION average(i parameter j parameter)  
            SET numerator TO element at position j PLUS one OF prefix_sum MINUS element at position i OF prefix_sum  
            SET denominator TO j MINUS i PLUS one  
            RETURN numerator DIVIDED BY denominator  
        END FUNCTION  
        
        SET dp TO a two dimensional list of zeroes with the number of rows equal to n PLUS one and the number of columns equal to k PLUS one  
        
        FOR index i FROM zero TO n MINUS one  
            SET element at position i AT column one OF dp TO average(i parameter n MINUS one)  
        END FOR  
        
        FOR index i FROM n MINUS one DOWNTO zero  
            FOR partitions FROM two TO k  
                FOR index j FROM i PLUS one TO n MINUS one  
                    SET current_value TO element at position i AT column partitions OF dp  
                    SET candidate_value TO element at position j AT column partitions MINUS one OF dp PLUS average(i parameter j MINUS one)  
                    SET element at position i AT column partitions OF dp TO the GREATER VALUE BETWEEN current_value AND candidate_value  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero AT column k OF dp  
    END FUNCTION  
END CLASS
