CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulting TO zero and left defaulting TO none and right defaulting TO none  
        SET the instance variable val TO val  
        SET the instance variable left TO left  
        SET the instance variable right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values which is a list  
    IF values EQUALS nothing  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode initialized with element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE and RETURN first element FROM queue and ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is not nothing  
            SET the left property of node TO a new TreeNode initialized with element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is not nothing  
            SET the right property of node TO a new TreeNode initialized with element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN true  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN false  
    ELSE IF the val property of p NOT EQUALS the val property of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree with the left property of p and the left property of q AND is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers with parameter root which is optionally a TreeNode  
        FUNCTION dfs with parameters node and current_number  
            IF node EQUALS nothing  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS the val property of node  
            IF the left property of node EQUALS nothing AND the right property of node EQUALS nothing  
                RETURN current_number  
            END IF  
            SET left_sum TO dfs called with the left property of node and current_number  
            SET right_sum TO dfs called with the right property of node and current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN dfs called with root and zero  
    END FUNCTION  
END CLASS
