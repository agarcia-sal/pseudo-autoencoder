CLASS Solution  
    FUNCTION canTransform(start string_parameter end string_parameter) RETURNS boolean  
        SET start_filtered TO an empty list  
        SET end_filtered TO an empty list  
        
        FOR each index position FROM zero TO the LENGTH OF start string_parameter MINUS one  
            SET character TO character at position index position of start string_parameter  
            IF character NOT EQUALS character X  
                APPEND the pair of character and index position TO start_filtered  
            END IF  
        END FOR  
        
        FOR each index position FROM zero TO the LENGTH OF end string_parameter MINUS one  
            SET character TO character at position index position of end string_parameter  
            IF character NOT EQUALS character X  
                APPEND the pair of character and index position TO end_filtered  
            END IF  
        END FOR  
        
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        
        FOR each pair of elements paired as first_pair and second_pair FROM start_filtered AND end_filtered  
            SET c1 TO the first element of first_pair  
            SET i1 TO the second element of first_pair  
            SET c2 TO the first element of second_pair  
            SET i2 TO the second element of second_pair  
            
            IF c1 NOT EQUALS c2  
                RETURN false  
            END IF  
            
            IF c1 EQUALS character L AND i1 LESS THAN i2  
                RETURN false  
            END IF  
            
            IF c1 EQUALS character R AND i1 GREATER THAN i2  
                RETURN false  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
