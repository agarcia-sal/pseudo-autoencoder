CLASS Solution  
    FUNCTION customSortString(order, s)  
        CREATE an empty dictionary to map characters to their positions  
        FOR each position AND character IN the sequence of characters of order  
            SET the character as a key in the dictionary with the associated position as its value  
        END FOR  
        SORT the sequence of characters of s using a key function that for each character returns the associated position in the dictionary if present OTHERWISE returns zero  
        JOIN the sorted sequence of characters into a single sequence of characters  
        RETURN the joined sequence of characters  
    END FUNCTION  
END CLASS
