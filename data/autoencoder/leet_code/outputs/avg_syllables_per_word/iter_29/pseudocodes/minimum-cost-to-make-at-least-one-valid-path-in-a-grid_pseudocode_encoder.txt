CLASS Solution  
    FUNCTION minCost(grid)  
        SET number_of_rows TO the LENGTH OF grid  
        SET number_of_columns TO the LENGTH OF element at position zero of grid  
        SET directions TO a collection of four pairs representing zero AND one then zero AND negative one then one AND zero then negative one AND zero  
        SET costs TO a two dimensional collection with number_of_rows rows AND number_of_columns columns where each element IS infinity  
        SET element at position zero row AND zero column of costs TO zero  
        SET priority_queue TO a collection containing one tuple of zero zero zero  
        SET visited_positions TO an empty set  
        
        WHILE priority_queue IS NOT empty  
            EXTRACT the element with the smallest cost FROM priority_queue AND ASSIGN cost TO first component AND i TO second component AND j TO third component  
            
            IF the pair of i AND j IS IN visited_positions  
                CONTINUE to the next iteration of the loop  
            END IF  
            ADD the pair of i AND j TO visited_positions  
            
            IF i EQUALS number_of_rows MINUS one AND j EQUALS number_of_columns MINUS one  
                RETURN cost  
            END IF  
            
            FOR each index k FROM zero TO the LENGTH OF directions MINUS one  
                SET delta_i TO the first component of element at position k of directions  
                SET delta_j TO the second component of element at position k of directions  
                SET new_i TO i PLUS delta_i  
                SET new_j TO j PLUS delta_j  
                IF new_i GREATER THAN OR EQUAL TO zero AND new_i LESS THAN number_of_rows AND new_j GREATER THAN OR EQUAL TO zero AND new_j LESS THAN number_of_columns  
                    IF k PLUS one NOT EQUALS element at position i row AND j column of grid  
                        SET additional_cost TO one  
                    ELSE  
                        SET additional_cost TO zero  
                    END IF  
                    SET new_cost TO cost PLUS additional_cost  
                    IF new_cost LESS THAN element at position new_i row AND new_j column of costs  
                        SET element at position new_i row AND new_j column of costs TO new_cost  
                        ADD the tuple of new_cost new_i AND new_j TO priority_queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
