CLASS Solution  
    FUNCTION characterReplacement(s k)  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        SET count TO a new mapping from character TO integer with default zero  
        
        FOR each position right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the entry corresponding TO character at position right of s IN count BY one  
            SET max_count TO the GREATER OF max_count AND the entry corresponding TO character at position right of s IN count  
            
            IF the LENGTH OF the current substring FROM position left TO position right MINUS max_count IS GREATER THAN k THEN  
                DECREMENT the entry corresponding TO character at position left of s IN count BY one  
                INCREMENT left BY one  
            END IF  
            
            SET max_length TO the GREATER OF max_length AND the LENGTH OF the current substring FROM position left TO position right  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
