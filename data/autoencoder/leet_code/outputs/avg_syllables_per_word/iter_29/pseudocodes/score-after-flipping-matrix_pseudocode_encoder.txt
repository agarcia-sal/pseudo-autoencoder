CLASS Solution  
    FUNCTION matrixScore(grid)  
        FOR each row IN grid  
            IF the element at position zero of row EQUALS zero  
                FOR index FROM zero TO the LENGTH OF row MINUS one  
                    SET the element at position index of row TO one MINUS the element at position index of row  
                END FOR  
            END IF  
        END FOR  
        
        FOR column FROM one TO the LENGTH OF the element at position zero of grid MINUS one  
            SET count_of_ones TO zero  
            FOR row_index FROM zero TO the LENGTH OF grid MINUS one  
                INCREMENT count_of_ones BY the element at position column of the element at position row_index of grid  
            END FOR  
            IF count_of_ones LESS THAN the LENGTH OF grid DIVIDED BY two  
                FOR row_index FROM zero TO the LENGTH OF grid MINUS one  
                    SET the element at position column of the element at position row_index of grid TO one MINUS the element at position column of the element at position row_index of grid  
                END FOR  
            END IF  
        END FOR  
        
        SET total_score TO zero  
        FOR each row IN grid  
            SET binary_representation TO an empty string  
            FOR each bit IN row  
                APPEND the string form of bit TO binary_representation  
            END FOR  
            SET numeric_value TO the integer value of binary_representation interpreted as a binary number  
            INCREMENT total_score BY numeric_value  
        END FOR  
        
        RETURN total_score  
    END FUNCTION  
END CLASS
