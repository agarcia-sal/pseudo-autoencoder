CLASS Solution  
    FUNCTION boldWords(words LIST OF STRINGS s STRING) RETURNS STRING  
        SET bold_marks TO a list of boolean values all set to FALSE with the LENGTH OF s elements  
        FOR each word IN words  
            SET start_position TO the POSITION OF word IN s  
            WHILE start_position NOT EQUALS negative one  
                FOR index FROM start_position TO start_position PLUS the LENGTH OF word MINUS one  
                    SET the element at position index of bold_marks TO TRUE  
                END FOR  
                SET start_position TO the POSITION OF word IN s STARTING FROM start_position PLUS one  
            END WHILE  
        END FOR  
        SET result_collection TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF s  
            IF the element at position i of bold_marks IS TRUE  
                APPEND the string indication of opening bold tag TO result_collection  
                WHILE i LESS THAN the LENGTH OF s AND the element at position i of bold_marks IS TRUE  
                    APPEND the character at position i of s TO result_collection  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string indication of closing bold tag TO result_collection  
            ELSE  
                APPEND the character at position i of s TO result_collection  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation of all elements in result_collection as a single string  
    END FUNCTION  
END CLASS
