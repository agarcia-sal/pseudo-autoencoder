CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        FOR each number IN the reverse of nums  
            IF number IS LESS THAN second  
                RETURN true  
            END IF  
            WHILE the stack IS NOT empty AND element at the last position of stack IS LESS THAN number  
                SET second TO the element removed from the last position of stack  
            END WHILE  
            APPEND number TO stack  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
