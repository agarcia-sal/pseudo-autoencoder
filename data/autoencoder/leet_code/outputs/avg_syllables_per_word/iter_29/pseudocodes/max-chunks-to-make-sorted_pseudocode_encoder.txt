CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET maximum_value_observed_so_far TO zero  
        SET count_of_chunks TO zero  
        FOR index FROM zero TO the LENGTH OF arr MINUS one  
            SET maximum_value_observed_so_far TO the GREATER OF maximum_value_observed_so_far AND element at position index of arr  
            IF maximum_value_observed_so_far EQUALS index  
                INCREMENT count_of_chunks BY one  
            END IF  
        END FOR  
        RETURN count_of_chunks  
    END FUNCTION  
END CLASS
