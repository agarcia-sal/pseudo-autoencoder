CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the collection values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the first element removed from the left side of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value of p IS NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree applied to the left child of p AND the left child of q AND the result of is_same_tree applied to the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree(self, nums)  
        IF nums IS EMPTY  
            RETURN none  
        END IF  
        SET max_val TO the maximum element in nums  
        SET max_index TO the position of the first occurrence of max_val in nums  
        SET root TO a new TreeNode initialized with max_val  
        SET root.left TO the result of calling constructMaximumBinaryTree on the subcollection from the first position TO the position max_index MINUS one of nums  
        SET root.right TO the result of calling constructMaximumBinaryTree on the subcollection from the position max_index PLUS one TO the end of nums  
        RETURN root  
    END FUNCTION  
END CLASS
