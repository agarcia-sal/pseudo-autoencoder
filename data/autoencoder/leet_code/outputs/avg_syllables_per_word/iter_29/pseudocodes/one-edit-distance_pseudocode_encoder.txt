CLASS Solution  
    FUNCTION isOneEditDistance(s string_parameter t string_parameter) RETURNS boolean  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF calling isOneEditDistance WITH s AS t AND t AS s  
        END IF  

        SET difference_in_length TO the LENGTH OF t MINUS the LENGTH OF s  

        IF difference_in_length GREATER THAN one  
            RETURN boolean value false  
        END IF  

        IF difference_in_length EQUALS zero  
            SET count_of_differences TO zero  
            FOR each position index FROM zero TO the LENGTH OF s MINUS one  
                IF the character at position index of s NOT EQUALS the character at position index of t  
                    INCREMENT count_of_differences BY one  
                END IF  
            END FOR  
            RETURN boolean value of count_of_differences EQUALS one  
        ELSE  
            FOR each position index FROM zero TO the LENGTH OF s MINUS one  
                IF the character at position index of s NOT EQUALS the character at position index of t  
                    RETURN boolean value of the substring from position index TO the END OF s EQUALS the substring from position index PLUS one TO the END OF t  
                END IF  
            END FOR  
            RETURN boolean value true  
        END IF  
    END FUNCTION  
END CLASS
