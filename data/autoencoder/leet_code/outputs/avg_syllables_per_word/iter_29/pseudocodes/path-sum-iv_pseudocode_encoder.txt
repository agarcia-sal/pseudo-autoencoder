CLASS Solution  
    FUNCTION pathSum(list_of_numbers)  
        SET tree TO empty dictionary  
        
        FOR each number IN list_of_numbers  
            SET depth TO number DIVIDED BY one hundred  
            SET position TO number DIVIDED BY ten THEN MODULO ten  
            SET value TO number MODULO ten  
            SET element identified by depth and position IN tree TO value  
        END FOR  
        
        FUNCTION dfs(depth_value, position_value, path_sum_value)  
            SET current_value TO value identified by depth_value and position_value IN tree  
            IF current_value IS NOT present  
                RETURN zero  
            END IF  
            
            SET new_path_sum TO path_sum_value PLUS current_value  
            
            SET left_child_depth TO depth_value PLUS one  
            SET left_child_position TO position_value MULTIPLIED BY two MINUS one  
            SET right_child_depth TO depth_value PLUS one  
            SET right_child_position TO position_value MULTIPLIED BY two  
            
            IF element identified by left_child_depth and left_child_position IS NOT present IN tree AND element identified by right_child_depth and right_child_position IS NOT present IN tree  
                RETURN new_path_sum  
            END IF  
            
            SET left_sum TO CALL dfs WITH left_child_depth, left_child_position, new_path_sum  
            SET right_sum TO CALL dfs WITH right_child_depth, right_child_position, new_path_sum  
            
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN CALL dfs WITH one, one, zero  
    END FUNCTION  
END CLASS
