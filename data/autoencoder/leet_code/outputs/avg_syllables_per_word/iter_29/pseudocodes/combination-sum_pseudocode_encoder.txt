CLASS Solution  
    FUNCTION combinationSum(candidates, target)  
        FUNCTION backtrack(start, target_value, current_path)  
            IF target_value EQUALS zero  
                APPEND current_path TO result_collection  
                RETURN  
            END IF  
            IF target_value LESS THAN zero  
                RETURN  
            END IF  
            FOR index FROM start TO the LENGTH OF candidates MINUS one  
                CALL backtrack WITH arguments index, target_value MINUS element at position index of candidates, current_path WITH element at position index of candidates APPENDED  
            END FOR  
        END FUNCTION  
        SET result_collection TO an empty collection  
        CALL backtrack WITH arguments zero, target, an empty collection  
        RETURN result_collection  
    END FUNCTION  
END CLASS
