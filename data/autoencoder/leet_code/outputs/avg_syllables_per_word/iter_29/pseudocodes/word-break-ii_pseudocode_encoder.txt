CLASS Solution  
    FUNCTION wordBreak(s word_string wordDict list_of_words) RETURNS list_of_sentences  
        SET wordSet TO a new set created from wordDict  
        SET memo TO an empty dictionary  

        FUNCTION backtrack(start_index) RETURNS list_of_sentences  
            IF start_index EXISTS AS A KEY IN memo  
                RETURN the value associated with start_index in memo  
            END IF  

            IF start_index EQUALS the LENGTH OF s  
                RETURN a list containing an empty string  
            END IF  

            SET res TO an empty list  
            FOR each end_index FROM start_index PLUS one TO the LENGTH OF s PLUS one  
                SET word TO the substring of s from position start_index TO position end_index MINUS one  
                IF word EXISTS IN wordSet  
                    FOR each sentence IN the result of calling backtrack with end_index  
                        IF sentence IS NOT an empty string  
                            APPEND the concatenation of word PLUS a space PLUS sentence TO res  
                        ELSE  
                            APPEND word TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  

            SET the value associated with start_index in memo TO res  
            RETURN res  
        END FUNCTION  

        RETURN the result of calling backtrack with zero  
    END FUNCTION  
END CLASS
