CLASS Solution  
    FUNCTION numOfWays(n)  
        SET MODULO_VALUE TO ten POWERED BY nine PLUS seven  
        SET count_type_a TO six  
        SET count_type_b TO six  
        FOR counter FROM two TO n  
            SET new_count_type_a TO count_type_a MULTIPLIED BY three PLUS count_type_b MULTIPLIED BY two  
            SET new_count_type_a TO new_count_type_a MODULO MODULO_VALUE  
            SET new_count_type_b TO count_type_a MULTIPLIED BY two PLUS count_type_b MULTIPLIED BY two  
            SET new_count_type_b TO new_count_type_b MODULO MODULO_VALUE  
            SET count_type_a TO new_count_type_a  
            SET count_type_b TO new_count_type_b  
        END FOR  
        SET total_count TO count_type_a PLUS count_type_b  
        SET total_count TO total_count MODULO MODULO_VALUE  
        RETURN total_count  
    END FUNCTION  
END CLASS
