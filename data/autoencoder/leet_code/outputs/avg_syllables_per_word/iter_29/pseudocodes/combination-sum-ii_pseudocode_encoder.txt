CLASS Solution  
    FUNCTION combinationSum2(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                IF i GREATER THAN start AND element at position i of candidates EQUALS element at position i MINUS one of candidates  
                    CONTINUE  
                END IF  
                CALL backtrack WITH i PLUS one AS start AND target MINUS element at position i of candidates AS target AND path WITH element at position i of candidates appended AS path  
            END FOR  
        END FUNCTION  
        SORT candidates IN NON DECREASING ORDER  
        SET result TO empty list  
        CALL backtrack WITH zero AS start AND target AS target AND empty list AS path  
        RETURN result  
    END FUNCTION  
END CLASS
