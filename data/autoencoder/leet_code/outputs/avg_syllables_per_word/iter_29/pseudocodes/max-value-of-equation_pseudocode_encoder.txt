CLASS Solution  
    FUNCTION findMaxValueOfEquation(points LIST OF LIST OF INTEGER k INTEGER) RETURNS INTEGER  
        SET max_value TO negative infinity  
        SET deque_ TO an empty double ended queue  
        
        FOR each element in points  
            SET xj TO the first element of element  
            SET yj TO the second element of element  
            
            WHILE the deque_ is not empty AND the difference of xj MINUS the second element of the first pair in deque_ is GREATER THAN k  
                REMOVE the first element from deque_  
            END WHILE  
            
            IF the deque_ is not empty  
                SET max_value TO the larger value BETWEEN max_value AND the sum of yj PLUS xj PLUS the first element of the first pair in deque_  
            END IF  
            
            WHILE the deque_ is not empty AND the difference of yj MINUS xj IS GREATER THAN OR EQUAL TO the first element of the last pair in deque_  
                REMOVE the last element from deque_  
            END WHILE  
            
            APPEND the pair consisting of the difference of yj MINUS xj AND xj TO deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
