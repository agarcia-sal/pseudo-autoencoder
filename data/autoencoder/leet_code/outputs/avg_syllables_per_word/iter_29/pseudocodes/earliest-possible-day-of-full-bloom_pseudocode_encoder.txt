CLASS Solution  
    FUNCTION earliestFullBloom(plantTime list_of_integers growTime list_of_integers)  
        SET tasks TO sorting the list formed by pairing each element of plantTime with the corresponding element of growTime ordering by the second element of each pair in descending order  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair in tasks  
            SET plant TO the first element of the pair  
            SET grow TO the second element of the pair  
            INCREMENT current_day BY plant  
            SET bloom_day TO the sum of current_day and grow  
            IF earliest_bloom LESS THAN bloom_day  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
