CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN the substring sub EQUALS the reverse of the substring sub  
        END FUNCTION  
        
        SET n TO the LENGTH OF the string s  
        
        SET is_pal TO a two dimensional collection of dimensions n BY n FILLED WITH false  
        
        FOR index i FROM n MINUS one DOWN TO zero  
            FOR index j FROM i TO n MINUS one  
                IF the character at position i of string s EQUALS the character at position j of string s AND  
                   the difference between j AND i IS LESS THAN OR EQUAL TO one OR is_pal at position i PLUS one AND position j MINUS one IS true  
                THEN  
                    SET is_pal at position i AND position j TO true  
                END IF  
            END FOR  
        END FOR  
        
        FOR index i FROM one TO n MINUS two  
            IF is_pal at position zero AND position i MINUS one IS true  
            THEN  
                FOR index j FROM i TO n MINUS two  
                    IF is_pal at position i AND position j IS true AND is_pal at position j PLUS one AND position n MINUS one IS true  
                    THEN  
                        RETURN true  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
