CLASS Solution  
    FUNCTION maxSubArrayLen(nums k)  
        SET cumulative_sum_index TO a dictionary with key zero mapped to value negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        FOR each index i AND element num IN the enumeration of nums  
            INCREMENT cumulative_sum BY num  
            IF the result of cumulative_sum MINUS k is a key in cumulative_sum_index  
                SET max_length TO the greater value between max_length AND the difference of i MINUS the value mapped by the key of cumulative_sum MINUS k in cumulative_sum_index  
            END IF  
            IF cumulative_sum is not a key in cumulative_sum_index  
                SET the value mapped by the key cumulative_sum in cumulative_sum_index TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
