CLASS ListNode  
    FUNCTION __init__(val WITH DEFAULT zero next WITH DEFAULT none)  
        SET the attribute val OF the current instance TO val  
        SET the attribute next OF the current instance TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode CREATED WITH the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values STARTING FROM position two TO the end  
        SET node TO a new ListNode CREATED WITH val  
        SET the attribute next OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the expression that the attribute val OF p1 EQUALS the attribute val OF p2 AND the result of is_same_list CALLED WITH the attribute next OF p1 AND the attribute next OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd(head n)  
        SET dummy TO a new ListNode CREATED WITH zero  
        SET the attribute next OF dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one MINUS one  
            SET first TO the attribute next OF first  
        END FOR  
        WHILE first IS not none  
            SET first TO the attribute next OF first  
            SET second TO the attribute next OF second  
        END WHILE  
        SET the attribute next OF second TO the attribute next OF the attribute next OF second  
        RETURN the attribute next OF dummy  
    END FUNCTION  
END CLASS
