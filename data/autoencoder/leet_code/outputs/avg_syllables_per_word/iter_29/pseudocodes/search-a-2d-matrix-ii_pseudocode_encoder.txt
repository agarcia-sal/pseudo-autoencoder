CLASS Solution  
    FUNCTION searchMatrix(matrix matrix_of_lists_of_integers target integer) RETURNS boolean  
        IF matrix IS empty OR the FIRST element of matrix IS empty  
            RETURN false  
        END IF  
        SET row TO zero  
        SET col TO the LENGTH OF the FIRST element of matrix MINUS one  
        WHILE row IS LESS THAN the LENGTH OF matrix AND col IS GREATER THAN OR EQUAL TO zero  
            IF element at position row of matrix AT position col EQUALS target  
                RETURN true  
            ELSE IF element at position row of matrix AT position col IS GREATER THAN target  
                DECREMENT col BY one  
            ELSE  
                INCREMENT row BY one  
            END ELSE IF  
        END WHILE  
        RETURN false  
    END FUNCTION  
END CLASS
