CLASS Solution  
    FUNCTION minOperations(target_parameter LIST OF integers ARR_PARAMETER LIST OF integers)  
        SET index_map TO empty collection mapping values TO positions  
        FOR each index AND each value IN the enumeration of target_parameter  
            SET the value key in index_map TO the index  
        END FOR  
        
        SET transformed_arr TO empty list  
        FOR each value IN arr_parameter  
            IF the value IS PRESENT AS A KEY IN index_map  
                APPEND the index_map value corresponding to the value TO transformed_arr  
            END IF  
        END FOR  
        
        SET lis TO empty list  
        FOR each number IN transformed_arr  
            SET position TO the result of finding the insertion point on the left in lis for number using binary search  
            IF position EQUALS the LENGTH OF lis  
                APPEND number TO lis  
            ELSE  
                SET the element at position in lis TO number  
            END IF  
        END FOR  
        
        SET lcs_length TO the LENGTH OF lis  
        SET minimum_operations TO the LENGTH OF target_parameter MINUS lcs_length  
        RETURN minimum_operations  
    END FUNCTION  
END CLASS
