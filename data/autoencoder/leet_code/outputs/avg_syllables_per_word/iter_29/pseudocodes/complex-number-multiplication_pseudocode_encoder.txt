CLASS Solution  
    FUNCTION complexNumberMultiply(num1 num1 STRING num2 num2 STRING) RETURNS STRING  
        FUNCTION parse_complex(num num STRING)  
            SET parts TO the list of substrings obtained by splitting num BY the character plus  
            SET real TO the integer conversion of the first element of parts  
            SET imaginary_substring TO the substring from the first character TO the last character MINUS one of the second element of parts  
            SET imaginary TO the integer conversion of imaginary_substring  
            RETURN real AND imaginary  
        END FUNCTION  
        
        SET real1 AND imag1 TO the result of calling parse_complex WITH num1  
        SET real2 AND imag2 TO the result of calling parse_complex WITH num2  
        
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        
        RETURN the string formed by concatenating the textual representation of real_part THEN the character plus THEN the textual representation of imaginary_part THEN the character i  
    END FUNCTION  
END CLASS
