CLASS Solution  
    FUNCTION isScramble(s1 s1_parameter STRING, s2 s2_parameter STRING) RETURNS BOOLEAN  
        IF the entire string s1_parameter IS EQUAL TO the entire string s2_parameter  
            RETURN true  
        END IF  
        
        IF the sequence of characters obtained by arranging all characters of string s1_parameter in ascending order IS NOT EQUAL TO the sequence of characters obtained by arranging all characters of string s2_parameter in ascending order  
            RETURN false  
        END IF  
        
        SET n TO the LENGTH OF string s1_parameter  
        FOR i FROM one TO n MINUS one  
            IF the result of calling isScramble with the substring of s1_parameter from position one TO position i IS true AND the result of calling isScramble with the substring of s2_parameter from position one TO position i IS true  
                RETURN true  
            END IF  
            IF the result of calling isScramble with the substring of s1_parameter from position one TO position i IS true AND the result of calling isScramble with the substring of s2_parameter from position n MINUS i PLUS one TO position n IS true AND the result of calling isScramble with the substring of s1_parameter from position i PLUS one TO position n IS true AND the result of calling isScramble with the substring of s2_parameter from position one TO position n MINUS i IS true  
                RETURN true  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
