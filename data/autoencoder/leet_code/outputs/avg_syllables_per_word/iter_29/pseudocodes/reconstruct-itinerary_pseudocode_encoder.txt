CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a mapping from keys to empty lists by default  
        FOR each pair of departure_airport AND arrival_airport IN the list of tickets sorted in ascending order and then reversed to descending order  
            APPEND arrival_airport TO the list associated with departure_airport in graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit(airport)  
            WHILE the list associated with airport in graph IS NOT empty  
                CALL visit with the last element removed from the list associated with airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit with the airport named as the sequence of characters J F K  
        RETURN the reverse of the list itinerary  
    END FUNCTION  
END CLASS
