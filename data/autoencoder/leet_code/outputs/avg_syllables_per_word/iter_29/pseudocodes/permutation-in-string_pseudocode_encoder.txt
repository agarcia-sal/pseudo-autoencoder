CLASS Solution  
    FUNCTION checkInclusion(string_parameter_one string_parameter_two) RETURNS boolean  
        SET length_of_string_one TO the LENGTH OF string_parameter_one  
        SET length_of_string_two TO the LENGTH OF string_parameter_two  
        
        IF length_of_string_one GREATER THAN length_of_string_two  
            RETURN false  
        END IF  
        
        SET frequency_count_of_string_one TO the frequency count of each character in string_parameter_one  
        SET frequency_count_of_first_window_of_string_two TO the frequency count of each character in the substring from position one TO position length_of_string_one of string_parameter_two  
        
        IF frequency_count_of_string_one EQUALS frequency_count_of_first_window_of_string_two  
            RETURN true  
        END IF  
        
        FOR each index FROM length_of_string_one PLUS one TO length_of_string_two  
            INCREMENT the count of character at position index of string_parameter_two IN frequency_count_of_first_window_of_string_two BY one  
            
            DECREMENT the count of character at position index MINUS length_of_string_one of string_parameter_two IN frequency_count_of_first_window_of_string_two BY one  
            
            IF the count of character at position index MINUS length_of_string_one of string_parameter_two IN frequency_count_of_first_window_of_string_two EQUALS zero  
                REMOVE the character at position index MINUS length_of_string_one of string_parameter_two FROM frequency_count_of_first_window_of_string_two  
            END IF  
            
            IF frequency_count_of_string_one EQUALS frequency_count_of_first_window_of_string_two  
                RETURN true  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
