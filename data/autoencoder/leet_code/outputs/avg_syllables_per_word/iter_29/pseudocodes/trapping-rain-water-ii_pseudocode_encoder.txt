CLASS Solution  
    FUNCTION trapRainWater(heightMap)  
        IF heightMap EQUALS nothing OR the LENGTH OF heightMap EQUALS zero OR the LENGTH OF element at position zero of heightMap EQUALS zero  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF heightMap  
        SET n TO the LENGTH OF element at position zero of heightMap  
        SET visited TO a two dimensional list of Boolean false WITH m ROWS AND n COLUMNS  
        SET min_heap TO an empty list  
        FOR i FROM zero TO m MINUS one  
            ADD a tuple consisting of element at position i of heightMap AT position zero AND i AND zero TO min_heap PRESERVING heap property  
            ADD a tuple consisting of element at position i of heightMap AT position n MINUS one AND i AND n MINUS one TO min_heap PRESERVING heap property  
            SET element at position i AT position zero of visited TO true  
            SET element at position i AT position n MINUS one of visited TO true  
        END FOR  
        FOR j FROM zero TO n MINUS one  
            ADD a tuple consisting of element at position zero of heightMap AT position j AND zero AND j TO min_heap PRESERVING heap property  
            ADD a tuple consisting of element at position m MINUS one of heightMap AT position j AND m MINUS one AND j TO min_heap PRESERVING heap property  
            SET element at position zero AT position j of visited TO true  
            SET element at position m MINUS one AT position j of visited TO true  
        END FOR  
        SET directions TO a list containing the tuples negative one AND zero AND one AND zero AND zero AND negative one AND zero AND one  
        SET water_trapped TO zero  
        WHILE min_heap is not empty  
            REMOVE the smallest element FROM min_heap PRESERVING heap property AND UNPACK TO height AND x AND y  
            FOR each pair dx AND dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND element at position nx AT position ny of visited EQUALS false  
                    SET element at position nx AT position ny of visited TO true  
                    INCREMENT water_trapped BY the GREATER OF zero AND height MINUS element at position nx AT position ny of heightMap  
                    ADD a tuple consisting of the GREATER OF height AND element at position nx AT position ny of heightMap AND nx AND ny TO min_heap PRESERVING heap property  
                END IF  
            END FOR  
        END WHILE  
        RETURN water_trapped  
    END FUNCTION  
END CLASS
