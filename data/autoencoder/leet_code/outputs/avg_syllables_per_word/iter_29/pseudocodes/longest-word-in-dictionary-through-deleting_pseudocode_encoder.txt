CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator_of_s TO an iterator over s  
            RETURN the result of checking for every character in word whether that character is contained in iterator_of_s in sequence  
        END FUNCTION  
        
        SORT dictionary so that words are ordered by length from the longest to the shortest AND for words with the same length ordered in ascending lexicographical order  
        
        FOR each word IN dictionary  
            IF the function is_subsequence returns true when called with word and s  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN the empty string  
    END FUNCTION  
END CLASS
