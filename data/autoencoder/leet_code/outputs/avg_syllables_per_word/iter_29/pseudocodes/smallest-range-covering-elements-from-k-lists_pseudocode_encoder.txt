CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            CALL heap push ON min_heap WITH the tuple of element at position zero of list at position index of nums THEN index THEN zero  
            SET max_value TO the GREATER OF max_value AND element at position zero of list at position index of nums  
        END FOR  
        SET smallest_range TO positive infinity  
        SET result TO the list containing negative one hundred thousand THEN positive one hundred thousand  
        WHILE min_heap IS NOT empty  
            CALL heap pop ON min_heap AND ASSIGN result TO min_value THEN list_index THEN element_index  
            IF max_value MINUS min_value IS LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO the list containing min_value THEN max_value  
            END IF  
            IF element_index PLUS one IS LESS THAN the LENGTH OF list at position list_index of nums  
                SET next_value TO element at position element_index PLUS one of list at position list_index of nums  
                CALL heap push ON min_heap WITH the tuple of next_value THEN list_index THEN element_index PLUS one  
                SET max_value TO the GREATER OF max_value AND next_value  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
