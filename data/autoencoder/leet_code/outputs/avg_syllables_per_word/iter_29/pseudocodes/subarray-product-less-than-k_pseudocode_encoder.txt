CLASS Solution  
    FUNCTION numSubarrayProductLessThanK(nums left_to_right_list OF integers, k single_integer) RETURNS single_integer  
        IF k LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        SET product_value TO one  
        SET left_pointer TO zero  
        SET count_subarrays TO zero  
        FOR right_pointer FROM zero TO the LENGTH OF nums MINUS one  
            SET product_value TO product_value MULTIPLIED BY element at position right_pointer of nums  
            WHILE product_value GREATER THAN OR EQUAL TO k AND left_pointer LESS THAN OR EQUAL TO right_pointer  
                SET product_value TO product_value DIVIDED BY element at position left_pointer of nums  
                INCREMENT left_pointer BY one  
            END WHILE  
            INCREMENT count_subarrays BY right_pointer MINUS left_pointer PLUS one  
        END FOR  
        RETURN count_subarrays  
    END FUNCTION  
END CLASS
