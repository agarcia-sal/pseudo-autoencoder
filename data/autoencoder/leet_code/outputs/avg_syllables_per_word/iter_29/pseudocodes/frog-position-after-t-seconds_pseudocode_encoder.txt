CLASS Solution  
    FUNCTION frogPosition(n, edges, t, target)  
        IF n EQUALS one  
            RETURN one  
        END IF  
        
        SET graph TO an empty mapping from elements TO empty lists  
        FOR each element u AND each element v IN edges  
            APPEND v TO the list associated with u IN graph  
            APPEND u TO the list associated with v IN graph  
        END FOR  
        
        SET queue TO a double ended queue containing a single element consisting of the value one the number one point zero and zero  
        SET visited TO a set containing the element one  
        
        WHILE the queue is not empty  
            REMOVE the first element from the queue AND ASSIGN the three components TO current_node current_probability current_time  
            
            IF current_time EQUALS t OR current_node NOT EQUALS one AND the number of elements in the list associated with current_node IN graph EQUALS one  
                IF current_node EQUALS target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE  
                END IF  
            END IF  
            
            SET possible_moves TO the number of elements in the list associated with current_node IN graph MINUS one if current_node is not equal to one otherwise zero  
            
            FOR each neighbor IN the list associated with current_node IN graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    ADD to the end of queue a triple consisting of neighbor current_probability DIVIDED BY possible_moves current_time INCREMENTED BY one  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN zero point zero  
    END FUNCTION  
END CLASS
