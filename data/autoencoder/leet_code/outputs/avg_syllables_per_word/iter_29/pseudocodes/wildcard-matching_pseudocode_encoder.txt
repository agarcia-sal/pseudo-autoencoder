CLASS Solution  
    FUNCTION isMatch(s pattern_string p pattern_string) RETURNS boolean  
        SET dp_table TO a two dimensional list with the count of elements in s PLUS one rows and the count of elements in p PLUS one columns initialized to false  
        
        SET element at position zero of dp_table and position zero of dp_table TO true  
        
        FOR index j FROM one TO the count of elements in p  
            IF the character at position index j MINUS one of pattern_string EQUALS the character representing an asterisk  
                SET element at position zero of dp_table and position j of dp_table TO the value of element at position zero of dp_table and position j MINUS one of dp_table  
            END IF  
        END FOR  
        
        FOR index i FROM one TO the count of elements in s  
            FOR index j FROM one TO the count of elements in p  
                IF the character at position index j MINUS one of pattern_string EQUALS the character representing an asterisk  
                    SET element at position i of dp_table and position j of dp_table TO the logical OR of the value of element at position i MINUS one of dp_table and position j of dp_table AND the value of element at position i of dp_table and position j MINUS one of dp_table  
                ELSE IF the character at position index j MINUS one of pattern_string EQUALS the character representing a question mark OR the character at position index i MINUS one of s EQUALS the character at position index j MINUS one of pattern_string  
                    SET element at position i of dp_table and position j of dp_table TO the value of element at position i MINUS one of dp_table and position j MINUS one of dp_table  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the value of element at position the count of elements in s of dp_table and position the count of elements in p of dp_table  
    END FUNCTION  
END CLASS
