CLASS Solution  
    FUNCTION isRationalEqual(s, t)  
        SET ratios TO a list containing one followed by one DIVIDED BY nine followed by one DIVIDED BY ninety nine followed by one DIVIDED BY nine hundred ninety nine followed by one DIVIDED BY nine thousand nine hundred ninety nine  
        
        FUNCTION valueOf(s)  
            IF the position of the character open parenthesis in s EQUALS negative one THEN  
                RETURN the conversion of s TO a floating point number  
            END IF  
            
            SET leftParenIndex TO the position of the character open parenthesis in s  
            SET rightParenIndex TO the position of the character close parenthesis in s  
            SET dotIndex TO the position of the character period in s  
            
            SET integerAndNonRepeating TO the conversion TO floating point number OF the substring of s from position zero UP TO position leftParenIndex MINUS one  
            SET nonRepeatingLength TO leftParenIndex MINUS dotIndex MINUS one  
            
            SET repeating TO the conversion TO integer OF the substring of s from position leftParenIndex PLUS one UP TO position rightParenIndex MINUS one  
            SET repeatingLength TO rightParenIndex MINUS leftParenIndex MINUS one  
            
            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY one TENTH POWERED BY nonRepeatingLength MULTIPLIED BY the element at position repeatingLength OF the list ratios  
        END FUNCTION  
        
        RETURN the absolute value of valueOf OF s MINUS valueOf OF t LESS THAN one TIMES ten POWERED BY negative nine  
    END FUNCTION  
END CLASS
