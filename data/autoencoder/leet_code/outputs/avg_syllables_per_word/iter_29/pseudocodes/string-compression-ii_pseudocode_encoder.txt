CLASS Solution  
    FUNCTION getLengthOfOptimalCompression(string_s parameter_k)  
        FUNCTION dp(parameter_i parameter_k parameter_prev_char parameter_prev_count)  
            IF parameter_k LESS THAN zero  
                RETURN infinity  
            END IF  
            IF parameter_i EQUALS the LENGTH OF string_s  
                RETURN zero  
            END IF  
            IF the CHARACTER at position parameter_i of string_s EQUALS parameter_prev_char  
                IF parameter_prev_count EQUALS one OR parameter_prev_count EQUALS nine OR parameter_prev_count EQUALS ninety nine  
                    SET increment TO one  
                ELSE  
                    SET increment TO zero  
                END IF  
                RETURN increment PLUS the RESULT OF dp with arguments parameter_i PLUS one parameter_k parameter_prev_char parameter_prev_count PLUS one  
            ELSE  
                SET keep TO one PLUS the RESULT OF dp with arguments parameter_i PLUS one parameter_k the CHARACTER at position parameter_i of string_s one  
                SET delete TO the RESULT OF dp with arguments parameter_i PLUS one parameter_k MINUS one parameter_prev_char parameter_prev_count  
                IF keep LESS THAN delete  
                    RETURN keep  
                ELSE  
                    RETURN delete  
                END IF  
            END IF  
        END FUNCTION  
        RETURN the RESULT OF dp with arguments zero parameter_k empty string zero  
    END FUNCTION  
END CLASS
