CLASS TreeNode  
    FUNCTION __init__ (val defaulted TO zero left defaulted TO none right defaulted TO none)  
        SET the value field TO val  
        SET the left field TO left  
        SET the right field TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(list_of_values)  
    IF the list_of_values IS empty  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode initialized with the element at position one of list_of_values  
    SET index TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        REMOVE and ASSIGN the element at the front of queue TO node  
        IF index IS LESS THAN the LENGTH OF list_of_values AND the element at position index of list_of_values IS NOT none  
            SET the left field of node TO a new instance of TreeNode initialized with the element at position index of list_of_values  
            APPEND the left field of node TO queue  
        END IF  
        INCREMENT index BY one  
        IF index IS LESS THAN the LENGTH OF list_of_values AND the element at position index of list_of_values IS NOT none  
            SET the right field of node TO a new instance of TreeNode initialized with the element at position index of list_of_values  
            APPEND the right field of node TO queue  
        END IF  
        INCREMENT index BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p_node q_node)  
    IF p_node IS none AND q_node IS none  
        RETURN True  
    ELSE IF p_node IS none OR q_node IS none  
        RETURN False  
    ELSE IF the value field of p_node IS NOT EQUALS TO the value field of q_node  
        RETURN False  
    ELSE   
        RETURN the result of is_same_tree called with the left field of p_node AND the left field of q_node AND the result of is_same_tree called with the right field of p_node AND the right field of q_node  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(root_node k_value)  
        SET stack_structure TO an empty list  
        SET current_node TO root_node  
        SET count_tracker TO zero  
        WHILE stack_structure IS NOT empty OR current_node IS NOT none  
            WHILE current_node IS NOT none  
                APPEND current_node TO stack_structure  
                SET current_node TO the left field of current_node  
            END WHILE  
            REMOVE and ASSIGN the last element of stack_structure TO current_node  
            INCREMENT count_tracker BY one  
            IF count_tracker IS EQUALS TO k_value  
                RETURN the value field of current_node  
            END IF  
            SET current_node TO the right field of current_node  
        END WHILE  
    END FUNCTION  
END CLASS
