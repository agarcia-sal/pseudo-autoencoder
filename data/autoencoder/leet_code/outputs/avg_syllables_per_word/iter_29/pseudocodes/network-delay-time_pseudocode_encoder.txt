CLASS Solution  
    FUNCTION networkDelayTime(times WITH TYPE List OF List OF Integer, n WITH TYPE Integer, k WITH TYPE Integer) RETURNS Integer  
        CREATE an empty collection named graph where each key is each integer from one TO n and each value is an empty list  
        FOR each trio of values named u v w IN times  
            APPEND the pair consisting of neighbor v AND weight w TO the list at key u IN graph  
        END FOR  
        
        CREATE a collection named dist where each key is each integer from one TO n and each value is infinity  
        SET the value at key k IN dist TO zero  
        
        CREATE a min-heap named min_heap containing a single pair consisting of zero AND k  
        
        WHILE min_heap is not empty  
            REMOVE the smallest pair from min_heap and ASSIGN the first value TO current_dist and the second value TO node  
            
            IF current_dist IS GREATER THAN the value at key node IN dist  
                CONTINUE to the next iteration of the loop  
            END IF  
            
            FOR each pair consisting of neighbor AND weight IN the list at key node IN graph  
                SET distance TO current_dist PLUS weight  
                
                IF distance IS LESS THAN the value at key neighbor IN dist  
                    SET the value at key neighbor IN dist TO distance  
                    ADD the pair consisting of distance AND neighbor TO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        SET max_time TO the maximum value among all values IN dist  
        
        IF max_time IS EQUAL TO infinity  
            RETURN minus one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
