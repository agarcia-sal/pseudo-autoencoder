CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO an empty list  
        FOR index FROM one TO n  
            APPEND the string form of index TO teams  
        END FOR  
        WHILE n IS GREATER THAN one  
            FOR i FROM zero TO the value of n DIVIDED BY two MINUS one  
                SET the element at position i of teams TO a string formed by concatenating an opening parenthesis followed by the element at position i of teams followed by a comma followed by the element at position n MINUS one MINUS i of teams followed by a closing parenthesis  
            END FOR  
            SET n TO the value of n DIVIDED BY two  
        END WHILE  
        RETURN the element at position zero of teams  
    END FUNCTION  
END CLASS
