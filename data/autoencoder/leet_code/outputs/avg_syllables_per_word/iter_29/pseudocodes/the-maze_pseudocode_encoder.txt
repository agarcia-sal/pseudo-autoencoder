CLASS Solution  
    FUNCTION hasPath(maze, start, destination)  
        SET directions TO a list containing the pairs of zero and one for right direction the pairs of one and zero for down direction the pairs of zero and minus one for left direction the pairs of minus one and zero for up direction  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero of maze  
        SET visited TO an empty collection for storing unique elements  
        
        FUNCTION dfs(x, y)  
            IF the pair of x and y IS IN visited THEN  
                RETURN False  
            END IF  
            IF the list containing x and y EQUALS destination THEN  
                RETURN True  
            END IF  
            ADD the pair of x and y TO visited  
            
            FOR each pair of dx and dy IN directions  
                SET nx TO x  
                SET ny TO y  
                
                WHILE the value of nx plus dx IS GREATER THAN OR EQUAL TO zero AND the value of nx plus dx IS LESS THAN m AND the value of ny plus dy IS GREATER THAN OR EQUAL TO zero AND the value of ny plus dy IS LESS THAN n AND the element at position nx plus dx of maze at position ny plus dy EQUALS zero  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                
                IF dfs(nx, ny) RETURNS True THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN dfs(the element at position zero of start, the element at position one of start)  
    END FUNCTION  
END CLASS
