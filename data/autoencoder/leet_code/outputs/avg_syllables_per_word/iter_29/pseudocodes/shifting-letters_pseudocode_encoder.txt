CLASS Solution  
    FUNCTION shiftingLetters(s shifts)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one TO zero STEP minus one  
            INCREMENT total_shifts BY the element at position index of shifts  
            SET the element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO an empty collection  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character_code_difference TO the Unicode code point of the character at position index of s MINUS the Unicode code point of the lowercase letter a  
            SET shifted_code TO the remainder after dividing the sum of character_code_difference PLUS the element at position index of shifts BY twenty six  
            SET new_character_code TO shifted_code PLUS the Unicode code point of the lowercase letter a  
            SET new_character TO the character with Unicode code point new_character_code  
            APPEND new_character TO result  
        END FOR  
        RETURN the concatenation of all elements in result as a single string  
    END FUNCTION  
END CLASS
