CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode with the element at position one of values AS val  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element popped from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new instance of TreeNode with the element at position i of values AS val  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new instance of TreeNode with the element at position i of values AS val  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the val attribute of p IS NOT EQUALS TO the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree called with the left attribute of p AND left attribute of q AND the result of is_same_tree called with the right attribute of p AND right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder(self, root)  
        IF root IS none  
            RETURN empty list  
        END IF  
        SET column_table TO a mapping from default keys as empty lists  
        SET queue TO a double ended queue initialized with a tuple containing root AND zero  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE queue is not empty  
            SET node, column_index TO the tuple popped from the left of queue  
            IF node IS NOT none  
                APPEND the val attribute of node TO the list at key column_index in column_table  
                IF column_index IS LESS THAN min_column  
                    SET min_column TO column_index  
                END IF  
                IF column_index IS GREATER THAN max_column  
                    SET max_column TO column_index  
                END IF  
                APPEND a tuple of the left attribute of node AND column_index MINUS one TO queue  
                APPEND a tuple of the right attribute of node AND column_index PLUS one TO queue  
            END IF  
        END WHILE  
        SET result TO an empty list  
        FOR column FROM min_column TO max_column  
            APPEND the list at key column in column_table TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
