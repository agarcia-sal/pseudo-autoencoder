CLASS Solution  
    FUNCTION findAllPeople(n number_of_people meetings list_of_meetings firstPerson initial_person) RETURNS list_of_people  
        SET know_secret TO a set containing zero and initial_person  
        SORT list_of_meetings according to the time value at position three of each meeting  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF list_of_meetings  
            SET current_time TO the time value at position three of the meeting at position i of list_of_meetings  
            SET graph TO a new mapping with default value as an empty list  
            WHILE i IS LESS THAN the LENGTH OF list_of_meetings AND the time value at position three of the meeting at position i of list_of_meetings EQUALS current_time  
                SET x TO the element at position one of the meeting at position i of list_of_meetings  
                SET y TO the element at position two of the meeting at position i of list_of_meetings  
                APPEND y TO the list associated with key x in graph  
                APPEND x TO the list associated with key y in graph  
                INCREMENT i BY one  
            END WHILE  
            SET queue TO a deque initialized with each person from know_secret who is a key in graph  
            SET visited TO an empty set  
            WHILE queue is NOT empty  
                REMOVE and GET the leftmost person from queue  
                IF person is in visited  
                    CONTINUE to next iteration of while loop  
                END IF  
                ADD person to visited  
                ADD person to know_secret  
                FOR each neighbor IN the list associated with key person in graph  
                    IF neighbor is NOT in visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN a list converted from know_secret  
    END FUNCTION  
END CLASS
