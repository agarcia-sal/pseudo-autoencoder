CLASS Solution  
    FUNCTION threeSumSmaller(nums list_of_numbers target target_number) RETURNS integer  
        CALL sort ON nums  
        SET count TO zero  
        SET n TO the LENGTH OF nums  
        
        FOR i FROM zero TO n MINUS two MINUS one  
            SET left TO i PLUS one  
            SET right TO n MINUS one  
            
            WHILE left LESS THAN right  
                SET current_sum TO the element at position i of nums PLUS the element at position left of nums PLUS the element at position right of nums  
                
                IF current_sum LESS THAN target  
                    INCREMENT count BY right MINUS left  
                    INCREMENT left BY one  
                ELSE  
                    DECREMENT right BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
