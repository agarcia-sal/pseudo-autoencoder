CLASS Solution  
    FUNCTION reachNumber(target)  
        SET target TO the absolute value of target  
        SET k TO zero  
        SET sum_steps TO zero  
        WHILE sum_steps IS LESS THAN target  
            INCREMENT k BY one  
            INCREMENT sum_steps BY k  
        END WHILE  
        IF the difference of sum_steps MINUS target IS EVEN  
            RETURN k  
        ELSE  
            IF k MODULO two EQUALS zero  
                RETURN k PLUS one  
            ELSE  
                RETURN k PLUS two  
            END IF  
        END IF  
    END FUNCTION  
END CLASS
