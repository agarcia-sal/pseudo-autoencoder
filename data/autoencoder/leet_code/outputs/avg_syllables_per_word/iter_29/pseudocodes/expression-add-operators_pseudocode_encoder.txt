CLASS Solution  
    FUNCTION addOperators(num string_parameter target integer_parameter) RETURNS list_of_strings  
        FUNCTION dfs(index integer_parameter path string_parameter value integer_parameter prev integer_parameter)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
              
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO substring from position index TO position i MINUS one of num  
                IF the LENGTH OF operand GREATER THAN one AND character at position zero of operand EQUALS zero  
                    CONTINUE  
                END IF  
                  
                IF index EQUALS zero  
                    CALL dfs WITH parameters i operand the INTEGER VALUE OF operand the INTEGER VALUE OF operand  
                ELSE  
                    CALL dfs WITH parameters i concatenation of path PLUS symbol PLUS operand value PLUS the INTEGER VALUE OF operand the INTEGER VALUE OF operand  
                    CALL dfs WITH parameters i concatenation of path MINUS symbol PLUS operand value MINUS the INTEGER VALUE OF operand NEGATIVE the INTEGER VALUE OF operand  
                    CALL dfs WITH parameters i concatenation of path MULTIPLICATION symbol PLUS operand value MINUS prev PLUS prev MULTIPLIED BY the INTEGER VALUE OF operand prev MULTIPLIED BY the INTEGER VALUE OF operand  
                END IF  
            END FOR  
        END FUNCTION  
          
        SET result TO an empty list  
        CALL dfs WITH parameters zero empty string zero zero  
        RETURN result  
    END FUNCTION  
END CLASS
