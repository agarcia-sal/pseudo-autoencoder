CLASS Solution  
    FUNCTION spiralMatrixIII(rows direction_rows cols direction_cols rStart row_start cStart column_start)  
        SET directions TO a list of pairs as follows: pair of zero AND one representing east direction pair of one AND zero representing south direction pair of zero AND negative one representing west direction pair of negative one AND zero representing north direction  
        SET direction_index TO zero representing the starting direction facing east  
        SET steps TO zero representing the number of steps to take in the current direction  
        SET increment TO zero representing when to increase the number of steps  
        SET visited TO an empty list representing visited positions  
        SET x TO row_start  
        SET y TO column_start  
        
        WHILE the LENGTH OF visited IS LESS THAN direction_rows MULTIPLIED BY direction_cols  
            IF increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR each count FROM one TO steps  
                IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN direction_rows AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN direction_cols  
                    APPEND a list containing x AND y TO visited  
                END IF  
                SET dx TO the first element of directions AT position direction_index  
                SET dy TO the second element of directions AT position direction_index  
                INCREMENT x BY dx  
                INCREMENT y BY dy  
            END FOR  
            SET direction_index TO the remainder of the division of the sum of direction_index AND one BY four  
            INCREMENT increment BY one  
        END WHILE  
        
        RETURN visited  
    END FUNCTION  
END CLASS
