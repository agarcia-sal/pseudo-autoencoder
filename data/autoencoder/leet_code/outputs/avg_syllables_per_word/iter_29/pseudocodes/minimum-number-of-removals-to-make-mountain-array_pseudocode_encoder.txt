CLASS Solution  
    FUNCTION minimumMountainRemovals(nums)  
        SET n TO the LENGTH OF nums  
        
        SET lis TO a list of the value one REPEATED n times  
        FOR each index i FROM zero TO n MINUS one  
            FOR each index j FROM zero TO i MINUS one  
                IF the element at position i of nums GREATER THAN the element at position j of nums THEN  
                    SET lis at position i TO the GREATER OF the current value at lis position i AND the value at lis position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        
        SET lds TO a list of the value one REPEATED n times  
        FOR each index i FROM n MINUS one DOWN TO zero  
            FOR each index j FROM i PLUS one TO n MINUS one  
                IF the element at position i of nums GREATER THAN the element at position j of nums THEN  
                    SET lds at position i TO the GREATER OF the current value at lds position i AND the value at lds position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        
        SET max_mountain_length TO zero  
        FOR each index i FROM one TO n MINUS two  
            IF the value at lis position i GREATER THAN one AND the value at lds position i GREATER THAN one THEN  
                SET max_mountain_length TO the GREATER OF max_mountain_length AND the value at lis position i PLUS the value at lds position i MINUS one  
            END IF  
        END FOR  
        
        RETURN n MINUS max_mountain_length  
    END FUNCTION  
END CLASS
