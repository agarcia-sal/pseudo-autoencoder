CLASS Solution  
    FUNCTION reverse(x)  
        SET sign TO negative one IF x IS LESS THAN zero OTHERWISE positive one  
        SET absolute_value_string TO the string representation OF the absolute value OF x  
        SET reversed_string TO the reverse order OF characters IN absolute_value_string  
        SET reversed_x TO the integer conversion OF reversed_string  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result IS LESS THAN negative two POWERED BY thirty ONE OR result IS GREATER THAN positive two POWERED BY thirty ONE MINUS one  
            RETURN zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
