CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        CREATE an empty dictionary named cost_to_digit  
        FOR each index i AND each value c IN the enumeration of cost  
            SET the entry at key c in cost_to_digit TO the string representing the digit of i plus one  
        END FOR  
        
        CREATE a list named dp starting with an empty string FOLLOWED BY the string minus one repeated target TIMES  
        
        FOR t FROM one TO target  
            FOR each key c AND each value d IN the key value pairs of cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND the element at position t minus c in dp IS NOT equal to string minus one THEN  
                    CREATE a string named candidate BY concatenating d FOLLOWED BY the element at position t minus c in dp  
                    IF the element at position t in dp EQUALS string minus one OR  
                       the LENGTH OF candidate GREATER THAN the LENGTH OF the element at position t in dp OR  
                       the LENGTH OF candidate EQUALS the LENGTH OF the element at position t in dp AND candidate GREATER THAN the element at position t in dp THEN  
                        SET the element at position t in dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF the element at position target in dp IS NOT equal to string minus one THEN  
            RETURN the element at position target in dp  
        ELSE  
            RETURN the string zero  
        END IF  
    END FUNCTION  
END CLASS
