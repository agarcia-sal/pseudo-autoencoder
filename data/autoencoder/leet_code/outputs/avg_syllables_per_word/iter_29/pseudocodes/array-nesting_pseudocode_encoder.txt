CLASS Solution  
    FUNCTION arrayNesting(nums)  
        SET visited_list TO a list of Boolean FALSE values with the LENGTH OF nums elements  
        SET maximum_length TO zero  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF the element at position index of visited_list EQUALS Boolean FALSE  
                SET current_length TO zero  
                SET k TO index  
                WHILE the element at position k of visited_list EQUALS Boolean FALSE  
                    SET the element at position k of visited_list TO Boolean TRUE  
                    SET k TO the element at position k of nums  
                    INCREMENT current_length BY one  
                END WHILE  
                IF maximum_length LESS THAN current_length  
                    SET maximum_length TO current_length  
                END IF  
            END IF  
        END FOR  
        RETURN maximum_length  
    END FUNCTION  
END CLASS
