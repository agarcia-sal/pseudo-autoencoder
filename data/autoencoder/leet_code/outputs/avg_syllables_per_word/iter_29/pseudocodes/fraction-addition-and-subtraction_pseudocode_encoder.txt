CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET modified_expression TO expression WITH every occurrence of the character dash REPLACED BY the character plus FOLLOWED BY the character dash  
        SET list_of_fractions TO the SPLIT of modified_expression BY the character plus  
        SET filtered_fractions TO an EMPTY collection  
        FOR each fraction_string IN list_of_fractions  
            IF fraction_string IS NOT an EMPTY string  
                APPEND fraction_string TO filtered_fractions  
            END IF  
        END FOR  
        SET sum_of_fractions TO the RESULT OF adding all elements OBTAINED BY CONVERTING each fraction_string IN filtered_fractions INTO a fraction object BY the USE of the fraction arithmetic class  
        SET numerator_value TO the numerator COMPONENT OF sum_of_fractions  
        SET denominator_value TO the denominator COMPONENT OF sum_of_fractions  
        SET result_string TO the STRING made by the CONCATENATION of the STRING representation of numerator_value FOLLOWED BY the character slash FOLLOWED BY the STRING representation of denominator_value  
        RETURN result_string  
    END FUNCTION  
END CLASS
