CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num EQUALS one  
            RETURN one  
        END IF  

        SET factors TO an empty list  
        FOR digit FROM nine DOWN TO two  
            WHILE the remainder of num DIVIDED BY digit EQUALS zero  
                APPEND digit TO factors  
                SET num TO the quotient of num DIVIDED BY digit  
            END WHILE  
        END FOR  

        IF num NOT EQUALS one  
            RETURN zero  
        END IF  

        SORT factors IN ascending order  
        SET result TO the integer formed by concatenating all elements in factors in order  

        IF result GREATER THAN two POWERED BY thirty ONE MINUS one  
            RETURN zero  
        END IF  

        RETURN result  
    END FUNCTION  
END CLASS
