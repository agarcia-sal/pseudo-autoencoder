CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value equal to element at position one of values  
    SET i TO one  
    CREATE a new empty double ended queue called queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE the element from the left of queue and ASSIGN IT TO node  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO a new TreeNode with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO a new TreeNode with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value of p IS NOT EQUAL TO the value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left and q.left AND the result of is_same_tree with parameters p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION recoverTree(root)  
        SET first TO None  
        SET second TO None  
        SET prev TO None  
        SET curr TO root  
        
        WHILE curr IS NOT None  
            IF curr.left IS None  
                IF prev IS NOT None AND the value of prev IS GREATER THAN the value of curr  
                    IF first IS None  
                        SET first TO prev  
                    END IF  
                    SET second TO curr  
                END IF  
                SET prev TO curr  
                SET curr TO curr.right  
            ELSE  
                SET pred TO curr.left  
                WHILE pred.right IS NOT None AND pred.right IS NOT curr  
                    SET pred TO pred.right  
                END WHILE  
                IF pred.right IS None  
                    SET pred.right TO curr  
                    SET curr TO curr.left  
                ELSE  
                    SET pred.right TO None  
                    IF prev IS NOT None AND the value of prev IS GREATER THAN the value of curr  
                        IF first IS None  
                            SET first TO prev  
                        END IF  
                        SET second TO curr  
                    END IF  
                    SET prev TO curr  
                    SET curr TO curr.right  
                END IF  
            END IF  
        END WHILE  
        
        IF first IS NOT None AND second IS NOT None  
            SET temporary_value TO the value of first  
            SET the value of first TO the value of second  
            SET the value of second TO temporary_value  
        END IF  
    END FUNCTION  
END CLASS
