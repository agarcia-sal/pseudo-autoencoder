CLASS Solution  
    FUNCTION maximumScore(scores list_of_integers scores parameter, edges list_of_lists_of_integers edges parameter)  
        SET graph TO an empty dictionary with default value as empty list  
        FOR each pair of integers u AND v IN edges parameter  
            APPEND a tuple of the score at position v of scores parameter AND the integer v TO the list at position u of graph  
            APPEND a tuple of the score at position u of scores parameter AND the integer u TO the list at position v of graph  
        END FOR  
        FOR each node IN the keys of graph  
            SET the list at position node of graph TO the result of selecting the three largest elements from the list at position node of graph according to the first element of each tuple  
        END FOR  
        SET max_score TO negative one  
        FOR each pair of integers u AND v IN edges parameter  
            FOR each tuple of integer score1 AND integer x IN the list at position u of graph  
                FOR each tuple of integer score2 AND integer y IN the list at position v of graph  
                    IF the number of distinct integers in the set containing the integers x y u AND v EQUALS four  
                        SET max_score TO the larger value between max_score AND the sum of score1 PLUS score2 PLUS the score at position u of scores parameter PLUS the score at position v of scores parameter  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN max_score  
    END FUNCTION  
END CLASS
