CLASS Solution  
    FUNCTION maxProduct(nums)  
        IF the list nums IS empty THEN  
            RETURN zero  
        END IF  
        SET maximum product TO element at position zero of nums  
        SET minimum product TO element at position zero of nums  
        SET result TO element at position zero of nums  
        FOR each number IN the sublist of nums starting from position one to the end  
            IF number LESS THAN zero THEN  
                SWAP the values of maximum product AND minimum product  
            END IF  
            SET maximum product TO the GREATER OF number AND maximum product MULTIPLIED BY number  
            SET minimum product TO the SMALLER OF number AND minimum product MULTIPLIED BY number  
            SET result TO the GREATER OF result AND maximum product  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
