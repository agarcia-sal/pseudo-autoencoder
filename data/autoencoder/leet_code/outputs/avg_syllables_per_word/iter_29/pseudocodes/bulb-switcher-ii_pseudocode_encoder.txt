CLASS Solution  
    FUNCTION flipLights(n number_of_lights presses number_of_presses) RETURNS number  
        SET number_of_lights TO the smaller value BETWEEN number_of_lights AND six  
        
        IF number_of_presses EQUALS zero  
            RETURN one  
        END IF  
        
        IF number_of_lights EQUALS one  
            RETURN two  
        END IF  
        
        IF number_of_lights EQUALS two  
            IF number_of_presses EQUALS one  
                RETURN three  
            ELSE  
                RETURN four  
            END IF  
        END IF  
        
        IF number_of_presses EQUALS one  
            RETURN four  
        END IF  
        
        IF number_of_presses EQUALS two  
            RETURN seven  
        END IF  
        
        RETURN eight  
    END FUNCTION  
END CLASS
