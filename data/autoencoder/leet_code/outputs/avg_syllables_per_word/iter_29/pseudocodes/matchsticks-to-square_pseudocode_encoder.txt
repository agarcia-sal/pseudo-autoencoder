CLASS Solution  
    FUNCTION makesquare(matchsticks)  
        SET total_length TO the SUM OF all elements IN matchsticks  
        IF the REMAINDER OF total_length DIVIDED BY four IS NOT equal to zero  
            RETURN false  
        END IF  
        SET side_length TO total_length DIVIDED BY four  
        SORT matchsticks IN descending order  
        
        FUNCTION can_form_sides(list_of_sides index)  
            IF index IS EQUAL TO the LENGTH OF matchsticks  
                FOR each side IN list_of_sides  
                    IF side IS NOT EQUAL TO side_length  
                        RETURN false  
                    END IF  
                END FOR  
                RETURN true  
            END IF  
            
            FOR position FROM zero TO three  
                IF the element at position position OF list_of_sides PLUS the element at position index OF matchsticks IS LESS THAN OR EQUAL TO side_length  
                    INCREMENT the element at position position OF list_of_sides BY the element at position index OF matchsticks  
                    IF can_form_sides(list_of_sides index PLUS one) RETURNS true  
                        RETURN true  
                    END IF  
                    DECREMENT the element at position position OF list_of_sides BY the element at position index OF matchsticks  
                END IF  
                
                IF the element at position position OF list_of_sides IS EQUAL TO zero  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        RETURN can_form_sides(a list of four zeros zero)  
    END FUNCTION  
END CLASS
