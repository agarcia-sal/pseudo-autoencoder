CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        SORT points IN ASCENDING ORDER BY the element at position two of each item  
        SET arrows TO zero  
        SET last_arrow_position TO the smallest possible real number representing negative infinity  
        FOR each balloon IN points  
            IF the element at position one of balloon IS GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO the element at position two of balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
