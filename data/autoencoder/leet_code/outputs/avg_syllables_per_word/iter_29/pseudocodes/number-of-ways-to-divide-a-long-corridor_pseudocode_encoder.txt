CLASS Solution  
    FUNCTION numberOfWays(corridor)  
        SET MOD TO one BILLION PLUS seven  
        SET seats TO an empty list  
        SET position_index TO zero  
        WHILE position_index LESS THAN the LENGTH OF corridor  
            IF character at position position_index of corridor EQUALS the character representing seat  
                APPEND position_index TO seats  
            END IF  
            INCREMENT position_index BY one  
        END WHILE  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        SET index TO two  
        WHILE index LESS THAN the LENGTH OF seats  
            SET difference TO the element at position index of seats MINUS the element at position index MINUS one of seats  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways MODULO MOD  
            INCREMENT index BY two  
        END WHILE  
        RETURN ways  
    END FUNCTION  
END CLASS
