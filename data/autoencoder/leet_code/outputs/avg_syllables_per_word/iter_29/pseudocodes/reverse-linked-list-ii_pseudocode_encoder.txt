CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val defaulted TO zero next defaulted TO none  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values list  
    IF values IS empty list  
        RETURN none  
    END IF  
    SET head TO a new ListNode WITH val being element at position one of values  
    SET p TO head  
    FOR each val IN the sublist of values from position two TO the last position  
        SET node TO a new ListNode WITH val being val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN p1 dot val EQUALS p2 dot val AND the RESULT OF calling is_same_list WITH p1 dot next AND p2 dot next  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween WITH PARAMETERS self head left right  
        IF head IS none OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO a new ListNode WITH val being zero  
        SET dummy dot next TO head  
        SET prev TO dummy  
        FOR counter FROM one TO left MINUS one  
            SET prev TO prev dot next  
        END FOR  
        SET current TO prev dot next  
        FOR counter FROM one TO right MINUS left  
            SET next_node TO current dot next  
            SET current dot next TO next_node dot next  
            SET next_node dot next TO prev dot next  
            SET prev dot next TO next_node  
        END FOR  
        RETURN dummy dot next  
    END FUNCTION  
END CLASS
