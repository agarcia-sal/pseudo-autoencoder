CLASS Solution  
    FUNCTION minMoves(nums k)  
        SET positions TO an empty list  
        FOR each index FROM zero TO the length of nums MINUS one  
            IF the element at position index of nums EQUALS one  
                APPEND index TO positions  
            END IF  
        END FOR  
        
        FUNCTION calculate_cost(start end)  
            SET mid TO the integer division result of start PLUS end MINUS one DIVIDED BY two  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR i FROM start TO end MINUS one  
                SET distance TO the absolute value of element at position i of positions MINUS median PLUS mid MINUS i  
                INCREMENT cost BY distance  
            END FOR  
            RETURN cost  
        END FUNCTION  
        
        SET min_cost TO positive infinity  
        FOR i FROM zero TO the length of positions MINUS k  
            SET current_cost TO calculate_cost(i i PLUS k)  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
