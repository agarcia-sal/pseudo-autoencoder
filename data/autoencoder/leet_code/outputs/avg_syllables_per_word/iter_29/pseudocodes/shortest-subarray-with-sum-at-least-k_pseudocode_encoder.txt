CLASS Solution  
    FUNCTION shortestSubarray(list_of_numbers NUMS parameter k)  
        SET prefix_sum_list TO a list containing zero as the only element  
        FOR each number IN list_of_numbers NUMS  
            APPEND the sum of the last element of prefix_sum_list PLUS the current number TO prefix_sum_list  
        END FOR  
        ASSIGN reference TO an empty double ended queue named dq  
        SET minimum_length TO positive infinity  
        FOR each index i AND corresponding current_sum IN enumeration of prefix_sum_list  
            WHILE dq is not empty AND the difference between current_sum AND the element at position zero of prefix_sum_list at the index at position zero of dq is GREATER THAN OR EQUAL TO k  
                SET minimum_length TO the smaller value between minimum_length AND the difference between i AND the element removed from the left end of dq  
            END WHILE  
            WHILE dq is not empty AND current_sum is LESS THAN OR EQUAL TO the element at position zero of prefix_sum_list at the index at the right end of dq  
                REMOVE the element at the right end FROM dq  
            END WHILE  
            APPEND i TO dq  
        END FOR  
        IF minimum_length IS NOT EQUAL TO positive infinity  
            RETURN minimum_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
