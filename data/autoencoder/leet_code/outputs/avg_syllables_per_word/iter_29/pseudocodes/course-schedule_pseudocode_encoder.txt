CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        CREATE an empty mapping from keys to empty lists NAMED graph  
        CREATE a list named indegree WITH size equal to numCourses FILLED with zero values  
        
        FOR each pair consisting of a course and a prerequisite IN prerequisites  
            APPEND course TO the list associated with prerequisite in graph  
            INCREMENT the value at position course in indegree BY one  
        END FOR  
        
        CREATE a queue named queue containing each course FROM zero TO numCourses MINUS one WHERE the value at position course in indegree EQUALS zero  
        
        WHILE the queue is not empty  
            REMOVE the first element from queue AND ASSIGN it TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN the list associated with current_course in graph  
                DECREMENT the value at position next_course in indegree BY one  
                IF the value at position next_course in indegree EQUALS zero  
                    APPEND next_course TO the queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the expression numCourses EQUALS zero  
    END FUNCTION  
END CLASS
