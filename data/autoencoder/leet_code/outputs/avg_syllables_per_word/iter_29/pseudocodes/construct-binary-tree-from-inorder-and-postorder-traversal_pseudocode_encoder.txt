CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value as element at position one of values  
    SET index i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO element removed from the front of queue  
        IF index i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT EQUAL TO none  
            SET left child of node TO new TreeNode with value as element at position i of values  
            APPEND left child of node TO queue  
        END IF  
        INCREMENT index i BY one  
        IF index i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT EQUAL TO none  
            SET right child of node TO new TreeNode with value as element at position i of values  
            APPEND right child of node TO queue  
        END IF  
        INCREMENT index i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF val of p IS NOT EQUAL TO val of q  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree with parameters left child of p AND left child of q AND the RESULT OF is_same_tree with parameters right child of p AND right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, inorder, postorder)  
        IF inorder IS EMPTY OR postorder IS EMPTY  
            RETURN none  
        END IF  
        SET root_val TO element removed from the end of postorder  
        SET root TO new TreeNode with value root_val  
        SET root_index TO the POSITION OF root_val IN inorder  
        SET right subtree of root TO the RESULT OF calling buildTree with parameters the SUBLIST of inorder from position root_index plus one TO the end AND postorder  
        SET left subtree of root TO the RESULT OF calling buildTree with parameters the SUBLIST of inorder from position one TO position root_index minus one AND postorder  
        RETURN root  
    END FUNCTION  
END CLASS
