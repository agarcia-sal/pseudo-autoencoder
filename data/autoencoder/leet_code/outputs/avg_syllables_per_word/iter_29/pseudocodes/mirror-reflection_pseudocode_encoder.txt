CLASS Solution  
    FUNCTION mirrorReflection(p q)  
        SET greatest_common_divisor TO the greatest common divisor OF p AND q  
        SET least_common_multiple TO p MULTIPLIED BY q DIVIDED BY greatest_common_divisor  
        SET vertical_reflections TO least_common_multiple DIVIDED BY p  
        SET horizontal_reflections TO least_common_multiple DIVIDED BY q  
        IF the remainder when vertical_reflections is divided by two EQUALS zero  
            RETURN zero  
        ELSE IF the remainder when horizontal_reflections is divided by two EQUALS one  
            RETURN one  
        ELSE  
            RETURN two  
        END ELSE  
    END FUNCTION  
END CLASS
