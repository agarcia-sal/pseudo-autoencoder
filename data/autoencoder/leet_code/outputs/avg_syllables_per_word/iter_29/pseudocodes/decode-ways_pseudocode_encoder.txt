CLASS Solution  
    FUNCTION numDecodings(string_input)  
        IF string_input IS EMPTY OR character at position one of string_input EQUALS character zero  
            RETURN zero  
        END IF  
        SET decoding_ways_list TO a new list of zeros with the LENGTH OF string_input PLUS one  
        SET element at position zero of decoding_ways_list TO one  
        SET element at position one of decoding_ways_list TO one  
        FOR index FROM two TO the LENGTH OF string_input PLUS one  
            IF character at position index MINUS one of string_input NOT EQUALS character zero  
                INCREMENT element at position index of decoding_ways_list BY element at position index MINUS one of decoding_ways_list  
            END IF  
            SET two_digit_number TO the number represented by substring from position index MINUS two TO position index MINUS one of string_input  
            IF two_digit_number IS GREATER THAN OR EQUAL TO ten AND two_digit_number IS LESS THAN OR EQUAL TO twenty six  
                INCREMENT element at position index of decoding_ways_list BY element at position index MINUS two of decoding_ways_list  
            END IF  
        END FOR  
        RETURN element at the last position of decoding_ways_list  
    END FUNCTION  
END CLASS
