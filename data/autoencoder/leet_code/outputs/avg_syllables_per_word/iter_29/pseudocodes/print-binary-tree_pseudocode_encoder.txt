CLASS TreeNode  
    FUNCTION __init__(val WITH DEFAULT zero, left WITH DEFAULT None, right WITH DEFAULT None)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS None OR the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH parameter element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node dot left TO NEW TreeNode WITH parameter element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node dot right TO NEW TreeNode WITH parameter element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN the RESULT of is_same_tree WITH parameters p dot left AND q dot left AND the RESULT of is_same_tree WITH parameters p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION printTree(root)  
        FUNCTION getHeight(node)  
            IF node EQUALS None  
                RETURN zero  
            END IF  
            RETURN one PLUS the GREATER value BETWEEN getHeight OF node dot left AND getHeight OF node dot right  
        END FUNCTION  
        
        SET height TO getHeight WITH parameter root  
        SET width TO TWO POWERED BY height MINUS one  
        SET res TO a list where EACH element IS a list of empty strings repeated width times REPEATED height times  
        
        FUNCTION placeNode(node, row, left, right)  
            IF node EQUALS None  
                RETURN  
            END IF  
            SET mid TO the INTEGER DIVISION of left PLUS right BY two  
            SET the element at position mid of the list at position row of res TO the string representation of node dot val  
            CALL placeNode WITH parameters node dot left, row PLUS one, left, mid  
            CALL placeNode WITH parameters node dot right, row PLUS one, mid, right  
        END FUNCTION  
        
        CALL placeNode WITH parameters root, zero, zero, width  
        RETURN res  
    END FUNCTION  
END CLASS
