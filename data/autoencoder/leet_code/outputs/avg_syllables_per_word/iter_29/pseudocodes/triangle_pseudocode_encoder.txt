CLASS Solution  
    FUNCTION minimumTotal(triangle)  
        FOR row FROM the LENGTH OF triangle MINUS two DOWN TO zero DECREMENT BY one  
            FOR col FROM zero TO the LENGTH OF element at position row of triangle MINUS one INCREMENT BY one  
                SET element at position row and position col of triangle TO the element at position row and position col of triangle PLUS the MINIMUM OF the element at position row PLUS one and position col of triangle AND the element at position row PLUS one and position col PLUS one of triangle  
            END FOR  
        END FOR  
        RETURN the element at position zero and position zero of triangle  
    END FUNCTION  
END CLASS
