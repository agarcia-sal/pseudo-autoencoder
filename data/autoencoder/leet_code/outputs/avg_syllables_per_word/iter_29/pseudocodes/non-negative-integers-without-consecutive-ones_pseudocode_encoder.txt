CLASS Solution  
    FUNCTION findIntegers(n)  
        SET binary_representation TO the binary form of n WITHOUT the first character which represents the binary prefix  
        SET length_of_representation TO the LENGTH OF binary_representation  
        
        INITIALIZE a list dp OF size length_of_representation PLUS one WITH all elements SET TO zero  
        SET element at position zero OF dp TO one  
        SET element at position one OF dp TO two  
        
        FOR index FROM two TO length_of_representation  
            SET element at position index OF dp TO element at position index MINUS one OF dp PLUS element at position index MINUS two OF dp  
        END FOR  
        
        SET result TO zero  
        SET previous_bit TO zero  
        
        FOR index FROM zero TO length_of_representation MINUS one  
            IF character at position index OF binary_representation EQUALS one THEN  
                INCREMENT result BY element at position length_of_representation MINUS index MINUS one OF dp  
                IF previous_bit EQUALS one THEN  
                    DECREMENT result BY one  
                    BREAK  
                END IF  
            END IF  
            SET previous_bit TO one IF character at position index OF binary_representation EQUALS one OTHERWISE zero  
        END FOR  
        
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
