CLASS Solution  
    FUNCTION longestWord(words)  
        CALL sort ON words BY the LENGTH OF word FOLLOWED BY the value of word in lexicographical order  
        SET valid_words TO an empty set  
        SET longest_word TO an empty value representing no characters  
        
        FOR each word IN words  
            IF the LENGTH OF word EQUALS one OR the substring of word from position one TO the LENGTH OF word MINUS one EXISTS IN valid_words  
                ADD word TO valid_words  
                IF the LENGTH OF word GREATER THAN the LENGTH OF longest_word  
                    SET longest_word TO word  
                END IF  
            END IF  
        END FOR  
        
        RETURN longest_word  
    END FUNCTION  
END CLASS
