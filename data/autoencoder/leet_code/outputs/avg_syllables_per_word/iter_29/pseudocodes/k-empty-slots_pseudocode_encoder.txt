CLASS Solution  
    FUNCTION kEmptySlots(list_of_bulbs, integer_k)  
        SET list_of_bloom_days TO an empty list  
        FOR each pair of day and position WITH day starting from one IN the enumeration of list_of_bulbs  
            SET insertion_index TO the position at which position SHOULD BE inserted IN list_of_bloom_days TO maintain sorted order USING binary search left side  
            IF insertion_index GREATER THAN zero AND the result of position MINUS the element at position insertion_index MINUS one IN list_of_bloom_days MINUS one EQUALS the integer_k  
                RETURN day  
            END IF  
            IF insertion_index LESS THAN the LENGTH OF list_of_bloom_days AND the result of the element at position insertion_index IN list_of_bloom_days MINUS position MINUS one EQUALS the integer_k  
                RETURN day  
            END IF  
            INSERT position INTO list_of_bloom_days AT insertion_index  
        END FOR  
        RETURN the negative of one  
    END FUNCTION  
END CLASS
