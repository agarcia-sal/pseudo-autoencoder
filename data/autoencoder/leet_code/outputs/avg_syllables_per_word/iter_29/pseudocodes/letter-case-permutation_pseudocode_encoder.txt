CLASS Solution  
    FUNCTION letterCasePermutation(string_input)  
        FUNCTION dfs(index_position)  
            IF index_position GREATER THAN OR EQUAL TO the LENGTH OF temporary_list  
                APPEND the JOINED elements of temporary_list INTO answer_list  
                RETURN  
            END IF  
            CALL dfs WITH index_position INCREMENTED BY one  
            IF character at position index_position of temporary_list IS an alphabetical character  
                SET character at position index_position of temporary_list TO the character obtained by toggling the case of the original character at the same position USING bitwise XOR with thirty two  
                CALL dfs WITH index_position INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET temporary_list TO the LIST OF characters in string_input  
        SET answer_list TO an empty list  
        CALL dfs WITH zero  
        RETURN answer_list  
    END FUNCTION  
END CLASS
