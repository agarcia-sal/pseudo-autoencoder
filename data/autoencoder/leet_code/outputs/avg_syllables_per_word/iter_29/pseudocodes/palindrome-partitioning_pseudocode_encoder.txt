CLASS Solution  
    FUNCTION partition(s)  
        FUNCTION is_palindrome(sub)  
            RETURN the value of sub EQUALS the value of substring from position the LENGTH OF sub MINUS one DOWN TO position zero OF sub  
        END FUNCTION  
        
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF s  
                APPEND path TO result  
                RETURN  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF s  
                IF the value of is_palindrome with argument the substring from position start TO position end MINUS one OF s HOLDS  
                    CALL backtrack WITH arguments end AND the list path APPENDED BY the substring from position start TO position end MINUS one OF s  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO empty list  
        CALL backtrack WITH arguments zero AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
