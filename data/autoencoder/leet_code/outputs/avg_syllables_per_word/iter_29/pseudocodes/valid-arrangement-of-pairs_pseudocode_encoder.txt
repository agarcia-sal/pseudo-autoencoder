CLASS Solution  
    FUNCTION validArrangement(pairs)  
        SET graph TO an empty mapping from keys to list of values  
        SET out_degree TO an empty counting mapping from keys to counts  
        SET in_degree TO an empty counting mapping from keys to counts  
        
        FOR each pair consisting of variable u AND variable v IN pairs  
            APPEND variable v TO the list at key u in graph  
            INCREMENT count of u in out_degree BY one  
            INCREMENT count of v in in_degree BY one  
        END FOR  
        
        SET start_node TO no value assigned  
        FOR each node IN keys of graph  
            IF count of node in out_degree EQUALS count of node in in_degree PLUS one THEN  
                SET start_node TO node  
                BREAK the loop  
            END IF  
        END FOR  
        
        IF start_node HAS no value assigned THEN  
            SET start_node TO the first element in keys of graph  
        END IF  
        
        SET path TO an empty list  
        SET stack TO a list containing only start_node  
        
        WHILE stack is not empty  
            SET u TO the last element of stack  
            IF the list at key u in graph is not empty THEN  
                REMOVE the last element v from the list at key u in graph  
                APPEND v TO stack  
            ELSE  
                REMOVE the last element from stack AND APPEND it TO path  
            END IF  
        END WHILE  
        
        REVERSE the order of elements in path  
        
        SET result TO an empty list  
        FOR each index i FROM zero TO the length of path MINUS one  
            APPEND a pair consisting of element at position i of path AND element at position i PLUS one of path TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
