CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SPLIT time INTO hours_part AND minutes_part USING the character colon  
            CONVERT hours_part TO number hours  
            CONVERT minutes_part TO number minutes  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
      
        SET minutes_list TO an empty list  
        FOR each time_point IN timePoints  
            APPEND to_minutes of time_point TO minutes_list  
        END FOR  
      
        SORT minutes_list IN ascending order  
      
        SET min_diff TO a very large positive number representing infinity  
      
        FOR index FROM one TO the LENGTH OF minutes_list MINUS one  
            SET current_difference TO element at position index of minutes_list MINUS element at position index MINUS one of minutes_list  
            IF current_difference IS LESS THAN min_diff THEN  
                SET min_diff TO current_difference  
            END IF  
        END FOR  
      
        SET circular_difference TO the number one thousand four hundred forty MINUS element at last position of minutes_list PLUS element at position zero of minutes_list  
      
        IF circular_difference IS LESS THAN min_diff THEN  
            SET min_diff TO circular_difference  
        END IF  
      
        RETURN min_diff  
    END FUNCTION  
END CLASS
