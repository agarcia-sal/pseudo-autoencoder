CLASS Solution  
    FUNCTION minTotalDistance(grid)  
        SET list_of_row_indices TO empty list  
        SET list_of_column_indices TO empty list  
        FOR index_i FROM zero TO the LENGTH OF grid MINUS one  
            FOR index_j FROM zero TO the LENGTH OF element at position index_i of grid MINUS one  
                IF element at position index_i of grid at position index_j EQUALS one  
                    APPEND index_i TO list_of_row_indices  
                    APPEND index_j TO list_of_column_indices  
                END IF  
            END FOR  
        END FOR  
        SORT list_of_row_indices IN ASCENDING ORDER  
        SORT list_of_column_indices IN ASCENDING ORDER  
        SET middle_index_for_rows TO the LENGTH OF list_of_row_indices DIVIDED BY two  
        SET middle_index_for_columns TO the LENGTH OF list_of_column_indices DIVIDED BY two  
        SET median_row TO element at position middle_index_for_rows of list_of_row_indices  
        SET median_column TO element at position middle_index_for_columns of list_of_column_indices  
        SET minimum_total_distance TO zero  
        FOR each row_index IN list_of_row_indices  
            INCREMENT minimum_total_distance BY the ABSOLUTE VALUE OF row_index MINUS median_row  
        END FOR  
        FOR each column_index IN list_of_column_indices  
            INCREMENT minimum_total_distance BY the ABSOLUTE VALUE OF column_index MINUS median_column  
        END FOR  
        RETURN minimum_total_distance  
    END FUNCTION  
END CLASS
