CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with value equal to element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN OR EQUAL TO the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO a new TreeNode with value equal to element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN OR EQUAL TO the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO a new TreeNode with value equal to element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value attribute of p IS NOT EQUALS the value attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree called with the left child of p AND the left child of q AND the result of is_same_tree called with the right child of p AND the right child of q  
    END ELSE  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST(self, root, low, high)  
        IF root IS none  
            RETURN none  
        END IF  
        IF the value attribute of root IS LESS THAN low  
            RETURN the result of calling trimBST on self with the right child of root AND low AND high  
        END IF  
        IF the value attribute of root IS GREATER THAN high  
            RETURN the result of calling trimBST on self with the left child of root AND low AND high  
        END IF  
        SET the left child of root TO the result of calling trimBST on self with the left child of root AND low AND high  
        SET the right child of root TO the result of calling trimBST on self with the right child of root AND low AND high  
        RETURN root  
    END FUNCTION  
END CLASS
