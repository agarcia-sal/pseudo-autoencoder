CLASS ListNode  
    FUNCTION __init__ (val OPTIONAL, next OPTIONAL)  
        SET self.val TO val OR zero  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO new instance of ListNode with value equal TO element at position one of values  
    SET p TO head  
    FOR each val IN values starting from position two TO the end  
        SET node TO new instance of ListNode with value equal TO val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the condition that value at p1 EQUALS value at p2 AND the result of is_same_list with p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1, l2)  
        FUNCTION pushIntoStack(node)  
            SET stack TO empty list  
            WHILE node IS NOT None  
                APPEND value at node TO stack  
                SET node TO node.next  
            END WHILE  
            RETURN stack  
        END FUNCTION  
        
        SET stack1 TO the result of pushIntoStack with l1  
        SET stack2 TO the result of pushIntoStack with l2  
        
        SET carry TO zero  
        SET result TO None  
        
        WHILE stack1 IS NOT empty OR stack2 IS NOT empty OR carry IS NOT zero  
            IF stack1 IS NOT empty  
                SET val1 TO the element popped from stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF stack2 IS NOT empty  
                SET val2 TO the element popped from stack2  
            ELSE  
                SET val2 TO zero  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  
            
            SET new_node TO new instance of ListNode with value equal TO digit  
            SET new_node.next TO result  
            SET result TO new_node  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
