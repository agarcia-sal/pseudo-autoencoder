CLASS Solution  
    FUNCTION superEggDrop(k n)  
        SET list_dp TO list of zeros with length k PLUS one  
        SET moves TO zero  
        WHILE the element at position k of list_dp IS LESS THAN n  
            INCREMENT moves BY one  
            FOR eggs FROM k DOWN TO one  
                SET element at position eggs of list_dp TO one PLUS element at position eggs MINUS one of list_dp PLUS element at position eggs of list_dp  
            END FOR  
        END WHILE  
        RETURN moves  
    END FUNCTION  
END CLASS
