CLASS Solution  
    FUNCTION visiblePoints(points angle location)  
        SET list_of_angles TO empty list  
        SET count_of_extra_points TO zero  
        FOR each point_x AND point_y IN points  
            IF point_x EQUALS element at position zero of location AND point_y EQUALS element at position one of location  
                INCREMENT count_of_extra_points BY one  
                CONTINUE to next iteration  
            END IF  
            SET angle_in_radians TO arctangent of the difference of point_y and element at position one of location DIVIDED BY the difference of point_x and element at position zero of location  
            SET angle_in_degrees TO conversion of angle_in_radians from radians to degrees  
            APPEND angle_in_degrees TO list_of_angles  
        END FOR  
        SORT list_of_angles IN ascending order  
        SET list_of_angles TO concatenation of list_of_angles AND list of each element in list_of_angles INCREMENTED BY three hundred sixty  
        SET maximum_number_of_visible_points TO zero  
        SET left_pointer TO zero  
        FOR right_pointer FROM zero TO the length of list_of_angles MINUS one  
            WHILE element at position right_pointer of list_of_angles MINUS element at position left_pointer of list_of_angles GREATER THAN angle  
                INCREMENT left_pointer BY one  
            END WHILE  
            SET current_window_size TO right_pointer MINUS left_pointer PLUS one  
            IF current_window_size GREATER THAN maximum_number_of_visible_points  
                SET maximum_number_of_visible_points TO current_window_size  
            END IF  
        END FOR  
        RETURN sum of maximum_number_of_visible_points AND count_of_extra_points  
    END FUNCTION  
END CLASS
