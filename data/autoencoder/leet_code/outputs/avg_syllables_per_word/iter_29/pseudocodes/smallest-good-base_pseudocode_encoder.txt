CLASS Solution  
    FUNCTION smallestGoodBase(n)  
        SET integer_n TO the integer value converted from n  
        SET maximum_length TO the integer part of the base two logarithm of integer_n INCREMENTED BY one  
        FOR each m FROM maximum_length DOWN TO two  
            SET candidate_base TO the integer part of integer_n RAISED TO THE POWER OF one DIVIDED BY the quantity m MINUS one  
            IF candidate_base LESS THAN two  
                CONTINUE TO the next iteration  
            END IF  
            SET geometric_sum TO the integer division of the quantity candidate_base RAISED TO THE POWER OF m MINUS one BY the quantity candidate_base MINUS one  
            IF geometric_sum EQUALS integer_n  
                RETURN the string representation of candidate_base  
            END IF  
        END FOR  
        RETURN the string representation of integer_n MINUS one  
    END FUNCTION  
END CLASS
