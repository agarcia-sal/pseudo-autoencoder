CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO a dictionary mapping zero TO zero  
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each key_value_pair in dp  
                SET d TO the key of key_value_pair  
                SET y TO the value of key_value_pair  
                SET sum_for_taller_support TO the maximum BETWEEN the value in new_dp at key which is the sum of d AND rod OR zero AND y  
                ASSIGN to new_dp at key which is the sum of d AND rod the value sum_for_taller_support  
                SET absolute_difference TO the absolute value of the difference BETWEEN d AND rod  
                SET sum_for_shorter_support TO the maximum BETWEEN the value in new_dp at key which is absolute_difference OR zero AND the sum of y AND the minimum BETWEEN d AND rod  
                ASSIGN to new_dp at key which is absolute_difference the value sum_for_shorter_support  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        RETURN the value in dp at key zero  
    END FUNCTION  
END CLASS
