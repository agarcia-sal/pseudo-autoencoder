CLASS Solution  
    FUNCTION maxPerformance(n, speed, efficiency, k)  
        SET engineers TO a list of pairs where each pair CONTAINS element from efficiency paired WITH corresponding element from speed  
        SORT engineers IN descending order BY the first element of each pair representing efficiency  
        SET speed_heap TO an empty collection representing a minimum heap  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        
        FOR each pair eff and spd IN engineers  
            ADD spd TO speed_heap USING heap push operation  
            INCREMENT speed_sum BY spd  
            
            IF the LENGTH OF speed_heap GREATER THAN k  
                REMOVE the smallest element FROM speed_heap USING heap pop operation  
                DECREMENT speed_sum BY the removed smallest element  
            END IF  
            
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            SET max_performance TO the larger value BETWEEN max_performance AND current_performance  
        END FOR  
        
        RETURN remainder of max_performance DIVIDED BY the sum of one billion PLUS seven  
    END FUNCTION  
END CLASS
