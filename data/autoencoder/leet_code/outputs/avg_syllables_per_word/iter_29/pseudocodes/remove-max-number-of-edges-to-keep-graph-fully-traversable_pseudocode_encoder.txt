CLASS UnionFind  
    FUNCTION __init__(n)  
        SET parent TO a list containing numbers from zero TO n inclusive  
        SET rank TO a list containing zero repeated n plus one times  
        SET count TO n  
    END FUNCTION  
      
    FUNCTION find(u)  
        IF element at position u of parent NOT EQUALS u THEN  
            SET element at position u of parent TO the result of calling find with parameter element at position u of parent  
        END IF  
        RETURN element at position u of parent  
    END FUNCTION  
      
    FUNCTION union(u, v)  
        SET rootU TO the result of calling find with parameter u  
        SET rootV TO the result of calling find with parameter v  
        IF rootU NOT EQUALS rootV THEN  
            IF element at position rootU of rank GREATER THAN element at position rootV of rank THEN  
                SET element at position rootV of parent TO rootU  
            ELSE IF element at position rootU of rank LESS THAN element at position rootV of rank THEN  
                SET element at position rootU of parent TO rootV  
            ELSE  
                SET element at position rootV of parent TO rootU  
                INCREMENT element at position rootU of rank BY one  
            END IF  
            DECREMENT count BY one  
            RETURN true  
        END IF  
        RETURN false  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION maxNumEdgesToRemove(n, edges)  
        SET uf_common TO a new UnionFind object initialized with n  
        SET uf_alice TO a new UnionFind object initialized with n  
        SET uf_bob TO a new UnionFind object initialized with n  
        
        SET used_edges TO zero  
        
        FOR each element consisting of t, u, v IN edges  
            IF t EQUALS three THEN  
                IF calling union on uf_common with parameters u and v RETURNS true THEN  
                    CALL union on uf_alice with parameters u and v  
                    CALL union on uf_bob with parameters u and v  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  
        
        FOR each element consisting of t, u, v IN edges  
            IF t EQUALS one AND calling union on uf_alice with parameters u and v RETURNS true THEN  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND calling union on uf_bob with parameters u and v RETURNS true THEN  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  
        
        IF count in uf_alice EQUALS one AND count in uf_bob EQUALS one THEN  
            RETURN the LENGTH OF edges MINUS used_edges  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
