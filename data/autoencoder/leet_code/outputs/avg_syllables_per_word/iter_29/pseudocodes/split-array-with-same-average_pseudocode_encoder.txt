CLASS Solution  
    FUNCTION splitArraySameAverage(nums)  
        SET n TO the LENGTH OF nums  
        SET total_sum TO the SUM OF nums  
        IF n EQUALS one  
            RETURN False  
        END IF  
        SET dp TO a LIST of SETS where each SET is empty for each number FROM zero TO the INTEGER DIVISION of n BY two PLUS one  
        ADD zero TO the SET at position zero of dp  
        FOR each number IN nums  
            FOR size FROM the INTEGER DIVISION of n BY two DOWN TO one  
                FOR each s IN the SET at position size MINUS one of dp  
                    SET new_sum TO s PLUS number  
                    IF new_sum MULTIPLIED BY n EQUALS size MULTIPLIED BY total_sum  
                        RETURN True  
                    END IF  
                    ADD new_sum TO the SET at position size of dp  
                END FOR  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
