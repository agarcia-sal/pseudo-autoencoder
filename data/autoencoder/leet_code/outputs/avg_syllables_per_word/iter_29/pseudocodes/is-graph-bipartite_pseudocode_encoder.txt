CLASS Solution  
    FUNCTION isBipartite(graph)  
        SET number_of_nodes TO the LENGTH OF graph  
        SET color_list TO a list of zeros repeated number_of_nodes times  
        
        FUNCTION dfs(node, current_color)  
            SET the element at position node of color_list TO current_color  
            FOR each neighbor IN the element at position node of graph  
                IF the element at position neighbor of color_list EQUALS current_color THEN  
                    RETURN False  
                END IF  
                IF the element at position neighbor of color_list EQUALS zero THEN  
                    IF the result of dfs with neighbor and NEGATIVE one MULTIPLIED BY current_color IS False THEN  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        FOR index FROM zero TO number_of_nodes MINUS one  
            IF the element at position index of color_list EQUALS zero THEN  
                IF the result of dfs with index and one IS False THEN  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
