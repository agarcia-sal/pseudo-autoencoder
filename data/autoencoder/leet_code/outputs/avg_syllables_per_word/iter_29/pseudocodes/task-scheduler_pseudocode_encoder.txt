CLASS Solution  
    FUNCTION leastInterval(tasks list_of_strings n integer)  
        SET task_counts TO the frequency COUNTER of tasks  
        SET max_heap TO a new list by TAKING for each count IN the VALUES OF task_counts the negative of count  
        TRANSFORM max_heap INTO a heap structure  
        
        SET time_counter TO zero  
        SET queue TO an empty list representing pairs of ready_time AND task_count  
        
        WHILE max_heap IS NOT empty OR queue IS NOT empty  
            WHILE queue IS NOT empty AND the first element at position zero of queue at position zero EQUALS time_counter  
                REMOVE the element at position zero FROM queue and ASSIGN its second element to count  
                IF count IS LESS THAN zero  
                    ADD count TO max_heap maintaining the heap property  
                END IF  
            END WHILE  
            
            IF max_heap IS NOT empty  
                REMOVE the smallest element FROM max_heap and ASSIGN it to count  
                INCREMENT count BY one  
                IF count IS LESS THAN zero  
                    APPEND a pair consisting of time_counter PLUS n PLUS one AND count TO queue  
                END IF  
            END IF  
            
            INCREMENT time_counter BY one  
        END WHILE  
        
        RETURN time_counter  
    END FUNCTION  
END CLASS
