CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF the value at position i and j of board EXCLUSIVE OR the value at position i and zero of board EXCLUSIVE OR the value at position zero and j of board EXCLUSIVE OR the value at position zero and zero of board EQUALS one  
                    RETURN negative one  
                END IF  
            END FOR  
            
            INCREMENT rowSum BY the value at position i and zero of board  
            INCREMENT colSum BY the value at position zero and i of board  
            INCREMENT rowSwap BY one IF the value at position i and zero of board EQUALS the remainder of i DIVIDED BY two ELSE zero  
            INCREMENT colSwap BY one IF the value at position zero and i of board EQUALS the remainder of i DIVIDED BY two ELSE zero  
        END FOR  
        
        IF the integer division of N by two GREATER THAN rowSum OR rowSum GREATER THAN the integer division of N plus one by two  
            RETURN negative one  
        END IF  
        IF the integer division of N by two GREATER THAN colSum OR colSum GREATER THAN the integer division of N plus one by two  
            RETURN negative one  
        END IF  
        
        IF the remainder of N DIVIDED BY two EQUALS one  
            IF the remainder of rowSwap DIVIDED BY two EQUALS one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of colSwap DIVIDED BY two EQUALS one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the minimum value BETWEEN rowSwap AND N MINUS rowSwap  
            SET colSwap TO the minimum value BETWEEN colSwap AND N MINUS colSwap  
        END IF  
        
        RETURN the integer division of the sum of rowSwap AND colSwap by two  
    END FUNCTION  
END CLASS
