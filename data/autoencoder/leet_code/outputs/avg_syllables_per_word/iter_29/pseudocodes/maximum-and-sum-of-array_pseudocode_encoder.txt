CLASS Solution  
    FUNCTION maximumANDSum(nums PARAMETER, numSlots PARAMETER)  
        FUNCTION dp(index PARAMETER, slots PARAMETER)  
            IF index EQUALS the LENGTH OF nums THEN  
                RETURN zero  
            END IF  
            
            SET max_and_sum TO zero  
            SET slots_list TO a LIST CONSTRUCTED FROM slots  
            
            FOR i FROM zero TO numSlots MINUS one  
                IF the ELEMENT AT POSITION i OF slots_list LESS THAN two THEN  
                    INCREMENT the ELEMENT AT POSITION i OF slots_list BY one  
                    SET and_sum TO the BITWISE AND OF the ELEMENT AT POSITION index OF nums AND i PLUS one PLUS THE RESULT OF CALLING dp WITH index PLUS one AND A TUPLE CONSTRUCTED FROM slots_list  
                    IF and_sum GREATER THAN max_and_sum THEN  
                        SET max_and_sum TO and_sum  
                    END IF  
                    DECREMENT the ELEMENT AT POSITION i OF slots_list BY one  
                END IF  
            END FOR  
            
            RETURN max_and_sum  
        END FUNCTION  
        
        SET initial_slots TO A TUPLE CONSTRUCTED FROM A LIST OF zero REPEATED numSlots TIMES  
        RETURN THE RESULT OF CALLING dp WITH zero AND initial_slots  
    END FUNCTION  
END CLASS
