CLASS Solution  
    FUNCTION getSum(a b)  
        SET MAX TO the cardinal value seven followed by seven times the hexadecimal letter F  
        SET mask TO the cardinal value fifteen followed by eight times the hexadecimal letter F  
        WHILE b NOT EQUALS zero HOLDS  
            SET carry TO the bitwise AND of a AND b SHIFTED LEFT BY one position  
            SET a TO the bitwise XOR of a AND b AND THEN the bitwise AND of that result AND mask  
            SET b TO the bitwise AND of carry AND mask  
        END WHILE  
        IF a LESS THAN OR EQUAL TO MAX THEN  
            RETURN a  
        ELSE  
            RETURN the bitwise negation of the bitwise XOR of a AND mask  
        END IF  
    END FUNCTION  
END CLASS
