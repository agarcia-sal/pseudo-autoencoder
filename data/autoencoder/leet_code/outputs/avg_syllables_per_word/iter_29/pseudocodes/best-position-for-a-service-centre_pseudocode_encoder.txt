CLASS Solution  
    FUNCTION getMinDistSum(positions)  
        SET quantity TO the LENGTH OF positions  
        SET x TO zero  
        SET y TO zero  
        FOR each pair_of_coordinates IN positions  
            SET x_coordinate TO the first element OF pair_of_coordinates  
            SET y_coordinate TO the second element OF pair_of_coordinates  
            INCREMENT x BY x_coordinate  
            INCREMENT y BY y_coordinate  
        END FOR  
        SET x TO x DIVIDED BY quantity  
        SET y TO y DIVIDED BY quantity  
        SET decay TO zero point nine nine nine  
        SET epsilon TO zero point zero zero zero zero zero one  
        SET alpha TO zero point five  
        WHILE true HOLDS  
            SET gradient_x TO zero  
            SET gradient_y TO zero  
            SET total_distance TO zero  
            FOR each pair_of_coordinates IN positions  
                SET x_coordinate TO the first element OF pair_of_coordinates  
                SET y_coordinate TO the second element OF pair_of_coordinates  
                SET difference_x TO x MINUS x_coordinate  
                SET difference_y TO y MINUS y_coordinate  
                SET current_distance TO THE SQUARE ROOT OF difference_x MULTIPLIED BY difference_x PLUS difference_y MULTIPLIED BY difference_y  
                INCREMENT gradient_x BY difference_x DIVIDED BY current_distance PLUS zero point zero zero zero zero zero zero one  
                INCREMENT gradient_y BY difference_y DIVIDED BY current_distance PLUS zero point zero zero zero zero zero zero one  
                INCREMENT total_distance BY current_distance  
            END FOR  
            SET delta_x TO gradient_x MULTIPLIED BY alpha  
            SET delta_y TO gradient_y MULTIPLIED BY alpha  
            DECREMENT x BY delta_x  
            DECREMENT y BY delta_y  
            MULTIPLY alpha BY decay  
            IF the ABSOLUTE VALUE OF delta_x IS LESS THAN OR EQUAL TO epsilon AND the ABSOLUTE VALUE OF delta_y IS LESS THAN OR EQUAL TO epsilon THEN  
                RETURN total_distance  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
