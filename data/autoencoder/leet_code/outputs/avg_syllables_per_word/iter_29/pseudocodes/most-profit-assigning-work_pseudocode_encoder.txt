CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO the list formed by pairing each element at corresponding position from difficulty and profit  
        SORT the list jobs IN ascending order by the first element of each paired item representing difficulty  
        SORT the list worker IN ascending order by worker ability  
        
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the LENGTH OF jobs  
        
        FOR each ability IN worker  
            WHILE job_index IS LESS THAN n AND the first element of the paired item at position job_index in jobs IS LESS THAN OR EQUAL TO ability  
                SET best_profit TO the larger value BETWEEN best_profit AND the second element of the paired item at position job_index in jobs representing profit  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
