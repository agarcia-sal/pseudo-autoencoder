CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET list_of_start_time_and_index TO the list of pairs where each pair is the first element of the element of intervals and the index of that element in intervals  
        SORT list_of_start_time_and_index IN ASCENDING ORDER BY the first element of each pair  
        
        SET list_of_results TO an empty list  
        
        FOR each element IN intervals  
            SET current_end_time TO the second element of the current element  
            
            SET position TO the position of the leftmost pair in list_of_start_time_and_index where the first element of the pair is GREATER THAN OR EQUAL TO current_end_time USING binary search  
            
            IF position IS LESS THAN the LENGTH OF list_of_start_time_and_index  
                APPEND the second element of the pair at position in list_of_start_time_and_index TO list_of_results  
            ELSE  
                APPEND negative one TO list_of_results  
            END IF  
        END FOR  
        
        RETURN list_of_results  
    END FUNCTION  
END CLASS
