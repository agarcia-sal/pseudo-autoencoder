CLASS TreeNode  
    FUNCTION __init__(val WITH DEFAULT zero, left WITH DEFAULT none, right WITH DEFAULT none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(list_of_values)  
    IF list_of_values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH value AS element at position one of list_of_values  
    SET index TO one  
    SET queue TO AN EMPTY double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF index IS LESS THAN the LENGTH OF list_of_values AND element at position index of list_of_values IS NOT none  
            SET node.left TO A NEW TreeNode WITH value AS element at position index of list_of_values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT index BY one  
        IF index IS LESS THAN the LENGTH OF list_of_values AND element at position index of list_of_values IS NOT none  
            SET node.right TO A NEW TreeNode WITH value AS element at position index of list_of_values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT index BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the value attribute of p IS NOT EQUALS TO the value attribute of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree(root)  
        FUNCTION subtree_sum(node)  
            IF node IS none  
                RETURN zero  
            END IF  
            RETURN the value attribute of node PLUS subtree_sum(the left attribute of node) PLUS subtree_sum(the right attribute of node)  
        END FUNCTION  
        
        SET total_sum TO subtree_sum(root)  
        
        IF total_sum MODULO two IS NOT EQUALS TO zero  
            RETURN False  
        END IF  
        
        SET target TO total_sum DIVIDED BY two  
        SET found TO A LIST CONTAINING False  
        
        FUNCTION can_split(node)  
            IF node IS none  
                RETURN zero  
            END IF  
            SET current_sum TO the value attribute of node PLUS can_split(the left attribute of node) PLUS can_split(the right attribute of node)  
            IF current_sum EQUALS target AND node IS NOT EQUALS TO root  
                SET the first element of found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  
        
        CALL can_split(root)  
        RETURN the first element of found  
    END FUNCTION  
END CLASS
