CLASS Solution  
    FUNCTION insert(intervals WITH TYPE List OF List OF integers newInterval WITH TYPE List OF integers) RETURNS List OF List OF integers  
        IF the input intervals list IS EMPTY  
            RETURN a new list containing newInterval  
        END IF  
          
        SET merged TO an empty list  
        SET i TO zero  
        SET n TO the LENGTH OF intervals  
          
        WHILE i IS LESS THAN n AND the end element of the interval at position i in intervals IS LESS THAN the start element of newInterval  
            APPEND the interval at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
          
        WHILE i IS LESS THAN n AND the start element of the interval at position i in intervals IS LESS THAN OR EQUAL TO the end element of newInterval  
            SET the start element of newInterval TO the MINIMUM between the start element of newInterval AND the start element of the interval at position i in intervals  
            SET the end element of newInterval TO the MAXIMUM between the end element of newInterval AND the end element of the interval at position i in intervals  
            INCREMENT i BY one  
        END WHILE  
          
        APPEND newInterval TO merged  
          
        WHILE i IS LESS THAN n  
            APPEND the interval at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
          
        RETURN merged  
    END FUNCTION  
END CLASS
