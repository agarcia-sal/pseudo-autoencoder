CLASS Solution  
    FUNCTION mincostToHireWorkers(quality LIST OF INTEGERS wage LIST OF INTEGERS k INTEGER) RETURNS FLOAT  
        SET workers TO the LIST OF PAIRS consisting of WAGE DIVIDED BY QUALITY AND QUALITY FOR each pair of WAGE AND QUALITY obtained by BITWISE PAIRING WAGE AND QUALITY  
        SORT workers BY the FIRST ELEMENT of each pair IN ASCENDING ORDER  
        
        SET max_heap TO an empty LIST  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        
        FOR each ratio AND q IN workers  
            PUSH the NEGATION OF q INTO max_heap  
            INCREMENT total_quality BY q  
            
            IF the LENGTH OF max_heap IS GREATER THAN k  
                SET removed_quality TO the VALUE obtained by REMOVING the SMALLEST ELEMENT FROM max_heap  
                INCREMENT total_quality BY removed_quality  
            END IF  
            
            IF the LENGTH OF max_heap IS EQUAL TO k  
                SET current_cost TO total_quality MULTIPLIED BY ratio  
                IF current_cost IS LESS THAN min_cost  
                    SET min_cost TO current_cost  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
