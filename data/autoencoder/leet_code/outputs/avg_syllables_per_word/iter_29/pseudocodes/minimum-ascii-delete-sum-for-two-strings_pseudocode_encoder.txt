CLASS Solution  
    FUNCTION minimumDeleteSum(s1 s1_string_parameter s2 s2_string_parameter)  
        SET m TO the LENGTH OF s1_string_parameter  
        SET n TO the LENGTH OF s2_string_parameter  
        
        SET dp TO a two dimensional array of zeroes with the first dimension size m PLUS one and the second dimension size n PLUS one  
        
        FOR j FROM one TO n  
            SET element at position zero AND position j of dp TO the element at position zero AND position j MINUS one of dp PLUS the ASCII value of character at position j MINUS one of s2_string_parameter  
        END FOR  
        
        FOR i FROM one TO m  
            SET element at position i AND position zero of dp TO the element at position i MINUS one AND position zero of dp PLUS the ASCII value of character at position i MINUS one of s1_string_parameter  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one of s1_string_parameter EQUALS character at position j MINUS one of s2_string_parameter  
                    SET element at position i AND position j of dp TO the element at position i MINUS one AND position j MINUS one of dp  
                ELSE  
                    SET element at position i AND position j of dp TO the MINIMUM OF (the element at position i MINUS one AND position j of dp PLUS the ASCII value of character at position i MINUS one of s1_string_parameter) AND (the element at position i AND position j MINUS one of dp PLUS the ASCII value of character at position j MINUS one of s2_string_parameter)  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position m AND position n of dp  
    END FUNCTION  
END CLASS
