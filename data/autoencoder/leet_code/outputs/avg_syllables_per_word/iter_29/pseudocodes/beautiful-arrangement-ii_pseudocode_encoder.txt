CLASS Solution  
    FUNCTION constructArray(parameter n, parameter k)  
        SET result TO an empty list  
        FOR current_number FROM one TO the difference between n AND k AND THEN DECREMENTED BY one  
            APPEND current_number TO result  
        END FOR  
        SET left_pointer TO the difference between n AND k  
        SET right_pointer TO n  
        WHILE left_pointer IS LESS THAN OR EQUAL TO right_pointer  
            IF left_pointer EQUALS right_pointer  
                APPEND left_pointer TO result  
            ELSE  
                APPEND left_pointer TO result  
                APPEND right_pointer TO result  
            END IF  
            INCREMENT left_pointer BY one  
            DECREMENT right_pointer BY one  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
