CLASS Solution  
    FUNCTION canPartition(list_of_numbers)  
        SET total_sum TO the SUM OF all elements IN list_of_numbers  
        
        IF the REMAINDER WHEN total_sum IS DIVIDED BY two IS NOT EQUAL TO zero THEN  
            RETURN false  
        END IF  
        
        SET target TO the QUOTIENT WHEN total_sum IS DIVIDED BY two  
        SET n TO the LENGTH OF list_of_numbers  
        
        CREATE a list dp OF BOOLEAN values WITH LENGTH target PLUS one WHERE each element IS false  
        SET the element at position zero OF dp TO true  
        
        FOR each number IN list_of_numbers  
            FOR index FROM target DOWN TO the VALUE OF number  
                SET dp at position index TO dp at position index OR dp at position index MINUS number  
            END FOR  
        END FOR  
        
        RETURN the element at position target OF dp  
    END FUNCTION  
END CLASS
