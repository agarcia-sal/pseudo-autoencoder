CLASS Solution  
    FUNCTION decodeAtIndex(s k)  
        SET total_length TO zero  
        
        FOR each character IN s  
            IF character IS A DIGIT  
                MULTIPLY total_length BY the integer value of character  
            ELSE  
                INCREMENT total_length BY one  
            END IF  
        END FOR  
        
        FOR each character IN the reverse order of s  
            SET k TO the remainder of k DIVIDED BY total_length  
            
            IF k EQUALS zero AND character IS AN ALPHABETICAL LETTER  
                RETURN character  
            END IF  
            
            IF character IS A DIGIT  
                DIVIDE total_length BY the integer value of character  
            ELSE  
                DECREMENT total_length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
