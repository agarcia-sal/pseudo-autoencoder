CLASS Solution  
    FUNCTION numDupDigitsAtMostN(n)  
        FUNCTION count_unique_digits(x)  
            SET string_representation TO the string form of x  
            SET length_of_string TO the LENGTH OF string_representation  
            SET count TO zero  
            
            FOR each integer i FROM one TO length_of_string MINUS one  
                SET count TO count PLUS the product of nine AND perm(nine MINUS i PLUS one, i MINUS one)  
            END FOR  
            
            SET seen TO an empty set  
            SET position TO zero  
            WHILE position LESS THAN length_of_string  
                SET current_digit TO the integer value of the character at position position of string_representation  
                
                FOR each integer d FROM zero TO current_digit MINUS one  
                    IF position EQUALS zero AND d EQUALS zero  
                        CONTINUE to the next iteration of the FOR loop  
                    END IF  
                    IF d NOT IN seen  
                        SET count TO count PLUS perm(nine MINUS position, length_of_string MINUS position MINUS one)  
                    END IF  
                END FOR  
                
                IF current_digit IN seen  
                    BREAK the WHILE loop  
                END IF  
                
                ADD current_digit TO seen  
                INCREMENT position BY one  
            END WHILE  
            
            IF position EQUALS length_of_string  
                INCREMENT count BY one  
            END IF  
            
            RETURN count  
        END FUNCTION  
        
        SET result TO n MINUS count_unique_digits(n)  
        RETURN result  
    END FUNCTION  
      
    FUNCTION perm(m, n)  
        IF n EQUALS zero  
            RETURN one  
        END IF  
        SET result TO one  
        FOR each integer i FROM zero TO n MINUS one  
            SET result TO result MULTIPLIED BY m MINUS i  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
