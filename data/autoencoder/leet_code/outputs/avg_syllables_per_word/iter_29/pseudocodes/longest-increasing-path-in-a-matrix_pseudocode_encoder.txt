CLASS Solution  
    FUNCTION longestIncreasingPath(matrix)  
        IF matrix EQUALS empty OR the LENGTH OF matrix EQUALS zero OR the LENGTH OF element at position zero of matrix EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET directions TO a collection containing a pair of one and zero a pair of negative one and zero a pair of zero and one and a pair of zero and negative one  
        SET dp TO a matrix of size m BY n FILLED WITH negative one  
        
        FUNCTION dfs(x y)  
            IF element at position x of dp element at position y NOT EQUALS negative one THEN  
                RETURN element at position x of dp element at position y  
            END IF  
            
            SET max_length TO one  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx of matrix element at position ny GREATER THAN element at position x of matrix element at position y THEN  
                    SET max_length TO the GREATER VALUE BETWEEN max_length AND one PLUS dfs(nx ny)  
                END IF  
            END FOR  
            
            SET element at position x of dp element at position y TO max_length  
            RETURN max_length  
        END FUNCTION  
        
        SET longest_path TO zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET longest_path TO the GREATER VALUE BETWEEN longest_path AND dfs(i j)  
            END FOR  
        END FOR  
        
        RETURN longest_path  
    END FUNCTION  
END CLASS
