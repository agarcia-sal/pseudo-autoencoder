CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, left WITH DEFAULT nothing, right WITH DEFAULT nothing)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the list values IS empty THEN  
        RETURN nothing  
    END IF  
    SET root TO a new instance of TreeNode initialized with element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT nothing THEN  
            SET node dot left TO a new instance of TreeNode initialized with element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT nothing THEN  
            SET node dot right TO a new instance of TreeNode initialized with element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS nothing AND q IS nothing THEN  
        RETURN truth  
    ELSE IF p IS nothing OR q IS nothing THEN  
        RETURN false  
    ELSE IF p dot val IS NOT EQUAL TO q dot val THEN  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree applied to p dot left AND q dot left AND the RESULT OF is_same_tree applied to p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root IS nothing THEN  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a double ended queue initialized with root as the only element  
        WHILE the queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR each index FROM one TO level_size  
                SET node TO the element removed from the front of queue  
                APPEND node dot val TO current_level  
                IF node dot left IS NOT nothing THEN  
                    APPEND node dot left TO queue  
                END IF  
                IF node dot right IS NOT nothing THEN  
                    APPEND node dot right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
