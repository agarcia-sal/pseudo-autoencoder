CLASS Solution  
    FUNCTION kMirror(k parameter_for_base  n parameter_for_count)  
        FUNCTION is_palindrome(s parameter_for_string)  
            RETURN the value of s parameter_for_string EQUALS the reverse of s parameter_for_string  
        END FUNCTION  
  
        FUNCTION to_base_k(num parameter_for_number  k parameter_for_base_conversion)  
            IF num parameter_for_number EQUALS zero  
                RETURN the string representation of zero  
            END IF  
            SET digits TO an empty list  
            WHILE num parameter_for_number is truthy meaning NOT zero  
                APPEND the remainder of num parameter_for_number DIVIDED BY k parameter_for_base_conversion converted to integer TO digits  
                SET num parameter_for_number TO the integer division of num parameter_for_number BY k parameter_for_base_conversion  
            END WHILE  
            RETURN the concatenation of the string conversion of each element IN digits taken in reverse order  
        END FUNCTION  
  
        FUNCTION generate_palindromes(length parameter_for_length)  
            IF length parameter_for_length EQUALS one  
                FOR i parameter_for_digit FROM one TO nine inclusive  
                    YIELD the string representation of i parameter_for_digit  
                END FOR  
            ELSE  
                SET start TO ten POWERED BY the integer division of length parameter_for_length MINUS one BY two  
                SET end TO ten POWERED BY the integer division of length parameter_for_length PLUS one BY two  
                FOR half parameter_for_half FROM start TO end EXCLUDING end  
                    SET half_str TO the string representation of half parameter_for_half  
                    IF length parameter_for_length MODULO two EQUALS zero  
                        SET full TO the concatenation of half_str WITH the reverse of half_str  
                    ELSE  
                        SET full TO the concatenation of half_str WITH the reverse of half_str except the last character  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
  
        SET k_mirror_numbers TO an empty list  
        SET length TO one  
        WHILE the LENGTH OF k_mirror_numbers IS LESS THAN n parameter_for_count  
            FOR each pal IN generate_palindromes WITH parameter length  
                CONVERT pal TO an integer AND SET TO num  
                IF is_palindrome called with to_base_k called with num AND k parameter_for_base RETURNS true  
                    APPEND num TO k_mirror_numbers  
                    IF the LENGTH OF k_mirror_numbers EQUALS n parameter_for_count  
                        BREAK the current loop  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
  
        RETURN the SUM OF all elements IN k_mirror_numbers  
    END FUNCTION  
END CLASS
