CLASS Solution  
    FUNCTION maxIncreaseKeepingSkyline(grid)  
        SET n TO the LENGTH OF grid  
        SET row_maxes TO a new empty list  
        FOR each row IN grid  
            FIND the maximum value IN the current row  
            APPEND that maximum value TO row_maxes  
        END FOR  
        SET col_maxes TO a new empty list  
        FOR each column_index FROM zero TO n MINUS one  
            SET column_values TO a new empty list  
            FOR each row_index FROM zero TO n MINUS one  
                APPEND element at position column_index of element at position row_index of grid TO column_values  
            END FOR  
            FIND the maximum value IN column_values  
            APPEND that maximum value TO col_maxes  
        END FOR  
        SET total_increase TO zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                FIND the minimum value BETWEEN element at position i of row_maxes AND element at position j of col_maxes  
                SET max_height TO that minimum value  
                FIND the difference BETWEEN max_height AND element at position i of element at position j of grid  
                IF that difference IS GREATER THAN zero  
                    INCREMENT total_increase BY that difference  
                END IF  
            END FOR  
        END FOR  
        RETURN total_increase  
    END FUNCTION  
END CLASS
