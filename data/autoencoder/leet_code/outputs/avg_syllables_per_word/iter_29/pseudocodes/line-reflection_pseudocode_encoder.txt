CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the set of tuples constructed from each point in points  
        SET min_x TO the minimum value among the first elements of each point in unique_points  
        SET max_x TO the maximum value among the first elements of each point in unique_points  
        SET line_of_reflection TO min_x PLUS max_x DIVIDED BY two  
        FOR each pair x and y IN unique_points  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the tuple consisting of reflected_x and y IS NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
