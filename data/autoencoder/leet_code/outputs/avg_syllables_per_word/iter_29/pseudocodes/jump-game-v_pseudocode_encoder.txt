CLASS Solution  
    FUNCTION maxJumps(arr AS list_of_integers d AS integer) RETURNS integer  
        FUNCTION dp(i AS integer) RETURNS integer  
            SET max_jumps TO one  
            FOR each integer x FROM one TO d  
                IF the sum of i AND x IS LESS THAN the LENGTH OF arr THEN  
                    IF the element at position the sum of i AND x of arr IS GREATER THAN OR EQUAL TO the element at position i of arr THEN  
                        BREAK  
                    END IF  
                    SET max_jumps TO the GREATER VALUE BETWEEN max_jumps AND the sum of one AND the RESULT OF dp WITH argument the sum of i AND x  
                END IF  
            END FOR  
            FOR each integer x FROM one TO d  
                IF the difference of i AND x IS GREATER THAN OR EQUAL TO zero THEN  
                    IF the element at position the difference of i AND x of arr IS GREATER THAN OR EQUAL TO the element at position i of arr THEN  
                        BREAK  
                    END IF  
                    SET max_jumps TO the GREATER VALUE BETWEEN max_jumps AND the sum of one AND the RESULT OF dp WITH argument the difference of i AND x  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        RETURN the GREATER VALUE AMONG all RESULTS OF dp WITH argument i FOR each integer i FROM zero TO the LENGTH OF arr MINUS one  
    END FUNCTION  
END CLASS
