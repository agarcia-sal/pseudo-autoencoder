CLASS Solution  
    FUNCTION stoneGameV(stoneValue)  
        SET prefix_sum TO a list of zeroes with the LENGTH OF stoneValue PLUS one  
        FOR index FROM zero TO the LENGTH OF stoneValue MINUS one  
            SET element at position index PLUS one of prefix_sum TO element at position index of prefix_sum PLUS element at position index of stoneValue  
        END FOR  
        
        FUNCTION get_range_sum(left, right)  
            RETURN element at position right PLUS one of prefix_sum MINUS element at position left of prefix_sum  
        END FUNCTION  
        
        FUNCTION dp(left, right)  
            IF left EQUALS right THEN  
                RETURN zero  
            END IF  
            SET max_score TO zero  
            FOR i FROM left TO right MINUS one  
                SET left_sum TO get_range_sum called with left and i  
                SET right_sum TO get_range_sum called with i PLUS one and right  
                IF left_sum LESS THAN right_sum THEN  
                    SET max_score TO the maximum value BETWEEN max_score AND left_sum PLUS dp called with left and i  
                ELSE IF left_sum GREATER THAN right_sum THEN  
                    SET max_score TO the maximum value BETWEEN max_score AND right_sum PLUS dp called with i PLUS one and right  
                ELSE  
                    SET max_score TO the maximum value AMONG max_score AND left_sum PLUS dp called with left and i AND right_sum PLUS dp called with i PLUS one and right  
                END IF  
            END FOR  
            RETURN max_score  
        END FUNCTION  
        
        RETURN dp called with zero and the LENGTH OF stoneValue MINUS one  
    END FUNCTION  
END CLASS
