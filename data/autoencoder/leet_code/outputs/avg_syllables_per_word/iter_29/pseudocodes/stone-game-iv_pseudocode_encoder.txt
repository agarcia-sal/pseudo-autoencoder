CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO a list of False values of length n PLUS one  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE j MULTIPLIED BY j IS LESS THAN OR EQUAL TO i  
                IF the value at position i MINUS j MULTIPLIED BY j of dp EQUALS False  
                    SET the value at position i of dp TO True  
                    BREAK out of the WHILE loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN the value at position n of dp  
    END FUNCTION  
END CLASS
