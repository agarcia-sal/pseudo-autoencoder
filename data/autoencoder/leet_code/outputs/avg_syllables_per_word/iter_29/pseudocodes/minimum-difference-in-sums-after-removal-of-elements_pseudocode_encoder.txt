CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
        
        SET min_heap TO an empty collection representing a minimum heap  
        SET min_sums TO a collection of zeros with size equal to two TIMES n PLUS one  
        
        FOR each index i FROM zero TO two TIMES n MINUS one  
            PUSH the NEGATION OF the element at position i of nums INTO min_heap AS maximum heap behavior  
            IF the LENGTH OF min_heap IS GREATER THAN n  
                REMOVE the ROOT ELEMENT FROM min_heap  
            END IF  
            IF the LENGTH OF min_heap EQUALS n  
                SET the element at position i of min_sums TO the NEGATION OF the SUM OF all elements in min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO an empty collection representing a minimum heap  
        SET max_sums TO a collection of zeros with size equal to two TIMES n PLUS one  
        
        FOR each index i FROM the value of three TIMES n MINUS one DOWN TO n  
            PUSH the element at position i of nums INTO max_heap AS minimum heap behavior  
            IF the LENGTH OF max_heap IS GREATER THAN n  
                REMOVE the ROOT ELEMENT FROM max_heap  
            END IF  
            IF the LENGTH OF max_heap EQUALS n  
                SET the element at position i of max_sums TO the SUM OF all elements in max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO a very large positive value representing infinity  
        FOR each index i FROM n MINUS one TO two TIMES n MINUS one  
            SET min_diff TO the SMALLER OF min_diff AND the DIFFERENCE BETWEEN the element at position i of min_sums AND the element at position i PLUS one of max_sums  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
