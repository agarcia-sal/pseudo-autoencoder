CLASS Solution  
    FUNCTION maxNiceDivisors(primeFactors)  
        SET MODULO_DIVISOR TO ten POWERED BY nine PLUS seven  
        
        IF primeFactors LESS THAN OR EQUAL TO three  
            RETURN primeFactors  
        END IF  
        
        SET number_of_threes TO primeFactors DIVIDED BY three  
        SET remainder_after_division TO primeFactors MODULO three  
        
        IF remainder_after_division EQUALS zero  
            RETURN three POWERED BY number_of_threes MODULO modulo_divisor  
        ELSE IF remainder_after_division EQUALS one  
            RETURN three POWERED BY number_of_threes MINUS one MODULO modulo_divisor MULTIPLIED BY four MODULO modulo_divisor  
        ELSE  
            RETURN three POWERED BY number_of_threes MODULO modulo_divisor MULTIPLIED BY two MODULO modulo_divisor  
        END ELSE IF  
    END FUNCTION  
END CLASS
