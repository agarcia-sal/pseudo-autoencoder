CLASS Solution  
    FUNCTION uniquePathsWithObstacles(obstacleGrid)  
        IF obstacleGrid IS empty OR the element at position zero zero of obstacleGrid EQUALS one THEN  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF obstacleGrid  
        SET n TO the LENGTH OF element at position zero of obstacleGrid  
        
        CREATE dp AS a two dimensional list of zeros with m rows and n columns  
        
        SET the element at position zero zero of dp TO one  
        
        FOR each j FROM one TO n MINUS one  
            IF the element at position zero j of obstacleGrid EQUALS zero THEN  
                SET the element at position zero j of dp TO the element at position zero j MINUS one of dp  
            END IF  
        END FOR  
        
        FOR each i FROM one TO m MINUS one  
            IF the element at position i zero of obstacleGrid EQUALS zero THEN  
                SET the element at position i zero of dp TO the element at position i MINUS one zero of dp  
            END IF  
        END FOR  
        
        FOR each i FROM one TO m MINUS one  
            FOR each j FROM one TO n MINUS one  
                IF the element at position i j of obstacleGrid EQUALS zero THEN  
                    SET the element at position i j of dp TO the element at position i MINUS one j of dp PLUS the element at position i j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position m MINUS one n MINUS one of dp  
    END FUNCTION  
END CLASS
