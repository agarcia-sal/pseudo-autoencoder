CLASS ListNode  
    FUNCTION __init__ (val WITH DEFAULT zero, next WITH DEFAULT none)  
        SET the value property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode WITH the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position two TO the end  
        SET node TO A NEW ListNode WITH val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN the value property of p1 EQUALS the value property of p2 AND is_same_list WITH the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs(head)  
        SET dummy TO A NEW ListNode WITH zero  
        SET the next property of dummy TO head  
        SET current TO dummy  
        WHILE the next property of current IS NOT none AND the next property of the next property of current IS NOT none  
            SET first TO the next property of current  
            SET second TO the next property of the next property of current  
            SET the next property of current TO second  
            SET the next property of first TO the next property of second  
            SET the next property of second TO first  
            SET current TO the next property of the next property of current  
        END WHILE  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
