CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF character at position start of num EQUALS character zero AND end GREATER THAN start PLUS one  
                    CONTINUE TO the next iteration of the FOR loop  
                END IF  
                SET next_num TO the integer value formed by the substring from position start TO position end MINUS one of num  
                IF next_num GREATER THAN OR EQUAL TO two POWERED BY thirty ONE  
                    BREAK out of the FOR loop  
                END IF  
                IF the LENGTH OF path LESS THAN two OR next_num EQUALS the SUM OF the last element of path AND the second to last element of path  
                    SET result TO the value returned by calling backtrack with parameters end AND the list formed by appending next_num to path  
                    IF result IS NOT empty  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN an empty list  
        END FUNCTION  
        RETURN the value returned by calling backtrack with parameters zero AND an empty list  
    END FUNCTION  
END CLASS
