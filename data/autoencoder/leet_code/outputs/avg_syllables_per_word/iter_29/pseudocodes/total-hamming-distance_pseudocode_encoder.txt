CLASS Solution  
    FUNCTION totalHammingDistance(nums)  
        SET total_distance TO zero  
        FOR i FROM zero TO thirty one  
            SET count_ones TO zero  
            FOR each number IN nums  
                INCREMENT count_ones BY the value of the bit at position i of number where the bit value is determined by shifting number right by i positions AND comparing the least significant bit to one  
            END FOR  
            SET count_zeros TO the LENGTH OF nums MINUS count_ones  
            INCREMENT total_distance BY count_ones MULTIPLIED BY count_zeros  
        END FOR  
        RETURN total_distance  
    END FUNCTION  
END CLASS
