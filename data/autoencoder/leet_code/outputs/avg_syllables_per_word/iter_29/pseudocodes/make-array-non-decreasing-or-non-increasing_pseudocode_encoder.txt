CLASS Solution  
    FUNCTION convertArray(nums)  
        FUNCTION min_operations_to_non_decreasing(nums)  
            SET max_heap TO empty list  
            SET operations TO zero  
            FOR each number IN nums  
                IF max_heap IS NOT empty AND the NEGATIVE of element at position zero of max_heap IS GREATER THAN number  
                    INCREMENT operations BY the NEGATIVE of the result of removing the smallest element from max_heap MINUS number  
                    ADD the NEGATIVE of number TO max_heap  
                END IF  
                ADD the NEGATIVE of number TO max_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
        
        FUNCTION min_operations_to_non_increasing(nums)  
            SET min_heap TO empty list  
            SET operations TO zero  
            FOR each number IN nums  
                IF min_heap IS NOT empty AND element at position zero of min_heap IS LESS THAN number  
                    INCREMENT operations BY number MINUS the result of removing the smallest element from min_heap  
                    ADD number TO min_heap  
                END IF  
                ADD number TO min_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
        
        RETURN the smaller value BETWEEN the result of calling min_operations_to_non_decreasing with nums AND the result of calling min_operations_to_non_increasing with nums  
    END FUNCTION  
END CLASS
