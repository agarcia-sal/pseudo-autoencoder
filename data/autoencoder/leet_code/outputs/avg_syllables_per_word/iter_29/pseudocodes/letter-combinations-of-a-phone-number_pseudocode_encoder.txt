CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string THEN  
            RETURN empty list  
        END IF  
        
        SET phone_map TO mapping where  
            character two MAPS TO string abc  
            character three MAPS TO string def  
            character four MAPS TO string ghi  
            character five MAPS TO string jkl  
            character six MAPS TO string mno  
            character seven MAPS TO string pqrs  
            character eight MAPS TO string tuv  
            character nine MAPS TO string wxyz  
        END MAPPING  
        
        FUNCTION backtrack(combination, next_digits)  
            IF the LENGTH OF next_digits EQUALS zero THEN  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the string that phone_map maps from character at position one of next_digits  
                    CALL backtrack WITH the CONCATENATION of combination AND letter AND the substring of next_digits from position two TO the end  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO empty list  
        CALL backtrack WITH empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
