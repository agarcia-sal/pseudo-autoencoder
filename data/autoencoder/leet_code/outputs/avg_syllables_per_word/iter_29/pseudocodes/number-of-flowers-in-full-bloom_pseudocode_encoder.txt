CLASS Solution  
    FUNCTION fullBloomFlowers(flowers list_of_lists_of_integers people list_of_integers) RETURNS list_of_integers  
        SET start_times TO the list of starting moments extracted from each element in flowers  
        SORT start_times in ascending order  
        SET end_times TO the list of ending moments extracted from each element in flowers  
        SORT end_times in ascending order  
        SET result TO an empty list  
        FOR each person IN people  
            SET started TO the count of elements in start_times that are less than or equal to person  
            SET ended TO the count of elements in end_times that are less than person  
            SET flowers_in_bloom TO started MINUS ended  
            APPEND flowers_in_bloom TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
