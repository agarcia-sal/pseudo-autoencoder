CLASS Solution  
    FUNCTION maxSlidingWindow(nums list_of_integers k integer) RETURNS list_of_integers  
        IF nums IS EMPTY OR k EQUALS zero  
            RETURN an empty list  
        END IF  
        SET deq TO an empty double ended queue  
        SET result TO an empty list  
        FOR each i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT EMPTY AND element at position zero of deq IS LESS THAN i MINUS k PLUS one  
                REMOVE the element at position zero FROM deq  
            END IF  
            WHILE deq IS NOT EMPTY AND element at position last of nums AT the element at position last of deq IS LESS THAN element at position i of nums  
                REMOVE the element at position last FROM deq  
            END WHILE  
            APPEND i TO deq  
            IF i IS GREATER THAN OR EQUAL TO k MINUS one  
                APPEND element at position zero of deq AT the element at position zero of nums TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
