CLASS Solution  
    FUNCTION smallestSufficientTeam(self req_skills AS list_of_strings people AS list_of_list_of_strings) RETURNS list_of_integers  
        CREATE empty mapping skill_to_index  
        SET skill_index TO zero  
        FOR each skill IN req_skills  
            ASSIGN skill_index TO skill_to_index AT skill  
            INCREMENT skill_index BY one  
        END FOR  
        SET total_skills TO the LENGTH OF req_skills  
        SET total_people TO the LENGTH OF people  
        CREATE list_of_integers skill_masks WITH total_people ELEMENTS ALL SET TO zero  
        SET person_index TO zero  
        FOR each skill_list IN people  
            FOR each skill IN skill_list  
                SET skill_bit TO one POWERED BY skill_to_index AT skill  
                SET skill_masks AT person_index TO skill_masks AT person_index OR skill_bit  
            END FOR  
            INCREMENT person_index BY one  
        END FOR  
        CREATE list_of_integers min_team_size WITH SIZE one POWERED BY total_skills ALL SET TO positive_infinity  
        CREATE list_of_integers last_person WITH SIZE one POWERED BY total_skills ALL SET TO zero  
        CREATE list_of_integers previous_state WITH SIZE one POWERED BY total_skills ALL SET TO zero  
        SET min_team_size AT zero TO zero  
        SET team_state TO zero  
        WHILE team_state LESS THAN one POWERED BY total_skills  
            IF min_team_size AT team_state EQUALS positive_infinity  
                CONTINUE TO next iteration  
            END IF  
            SET person_id TO zero  
            WHILE person_id LESS THAN total_people  
                SET combined_state TO team_state OR skill_masks AT person_id  
                IF min_team_size AT team_state PLUS one LESS THAN min_team_size AT combined_state  
                    SET min_team_size AT combined_state TO min_team_size AT team_state PLUS one  
                    SET last_person AT combined_state TO person_id  
                    SET previous_state AT combined_state TO team_state  
                END IF  
                INCREMENT person_id BY one  
            END WHILE  
            INCREMENT team_state BY one  
        END WHILE  
        SET current_state TO one POWERED BY total_skills MINUS one  
        CREATE empty list answer  
        WHILE current_state NOT EQUAL TO zero  
            APPEND last_person AT current_state TO answer  
            SET current_state TO previous_state AT current_state  
        END WHILE  
        RETURN answer  
    END FUNCTION  
END CLASS
