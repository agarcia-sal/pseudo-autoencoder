CLASS Solution  
    FUNCTION uniqueLetterString(parameter s)  
        SET MODULO TO ten POWERED BY nine PLUS seven  
        SET length_of_s TO the LENGTH OF s  
        
        SET last_positions TO a list containing negative one repeated twenty six times  
        SET last_positions_two TO a list containing negative one repeated twenty six times  
        
        SET total_result TO zero  
        
        FOR each index WITH character IN s WITH enumeration starting at zero  
            SET character_index TO the UNICODE VALUE OF character MINUS the UNICODE VALUE OF uppercase A  
            
            SET total_result TO total_result PLUS the PRODUCT OF the difference between index AND the element at position character_index of last_positions AND the difference between the element at position character_index of last_positions AND the element at position character_index of last_positions_two  
            SET total_result TO the remainder of total_result DIVIDED BY MODULO  
            
            SET the element at position character_index of last_positions_two TO the element at position character_index of last_positions  
            SET the element at position character_index of last_positions TO index  
        END FOR  
        
        FOR each index FROM zero TO twenty five  
            SET total_result TO total_result PLUS the PRODUCT OF the difference between length_of_s AND the element at position index of last_positions AND the difference between the element at position index of last_positions AND the element at position index of last_positions_two  
            SET total_result TO the remainder of total_result DIVIDED BY MODULO  
        END FOR  
        
        RETURN total_result  
    END FUNCTION  
END CLASS
