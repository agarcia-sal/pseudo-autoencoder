CLASS Solution  
    FUNCTION increasingTriplet(nums)  
        SET first TO positive infinity  
        SET second TO positive infinity  
        
        FOR each number IN nums  
            IF number LESS THAN OR EQUAL TO first  
                SET first TO number  
            ELSE IF number LESS THAN OR EQUAL TO second  
                SET second TO number  
            ELSE  
                RETURN true  
            END ELSE  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
