CLASS Solution  
    FUNCTION advantageCount(nums1, nums2)  
        SORT nums1 IN ascending order  
        SET nums2_with_indices TO list of pairs each containing a value and its original position taken FROM nums2 arranged in descending order by the value  
        SET result TO list of zeroes with the same length AS nums1  
        SET left TO zero  
        SET right TO the length of nums1 MINUS one  
        FOR each pair containing value and index IN nums2_with_indices  
            IF element at position right of nums1 GREATER THAN value THEN  
                SET element at position index of result TO element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index of result TO element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
