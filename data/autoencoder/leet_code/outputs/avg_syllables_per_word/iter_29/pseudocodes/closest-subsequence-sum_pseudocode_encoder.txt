CLASS Solution  
    FUNCTION minAbsDifference(nums goal)  
        SET n TO the LENGTH OF nums  
        SET mid TO n DIVIDED BY two rounded down to nearest integer  
        
        SET left_sums TO a list containing zero  
        FOR each num IN the SUBLIST of nums from position zero TO position mid MINUS one  
            CREATE a list of sums by ADDING num TO each element in left_sums  
            EXTEND left_sums BY the created list of sums  
        END FOR  
        
        SET right_sums TO a list containing zero  
        FOR each num IN the SUBLIST of nums from position mid TO the LAST position  
            CREATE a list of sums by ADDING num TO each element in right_sums  
            EXTEND right_sums BY the created list of sums  
        END FOR  
        
        SORT right_sums in ascending order  
        
        SET min_diff TO positive infinity  
        
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO the insertion position in right_sums where target SHOULD BE inserted to maintain sorted order found by binary search  
            
            IF pos LESS THAN the LENGTH OF right_sums  
                SET min_diff TO the MINIMUM of min_diff AND the ABSOLUTE VALUE of left_sum PLUS the element at position pos of right_sums MINUS goal  
            END IF  
            
            IF pos GREATER THAN zero  
                SET min_diff TO the MINIMUM of min_diff AND the ABSOLUTE VALUE of left_sum PLUS the element at position pos MINUS one of right_sums MINUS goal  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
