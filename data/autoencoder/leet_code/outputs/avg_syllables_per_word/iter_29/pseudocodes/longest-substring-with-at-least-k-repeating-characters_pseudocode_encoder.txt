CLASS Solution  
    FUNCTION longestSubstring(s k)  
        IF the LENGTH OF s IS LESS THAN k  
            RETURN zero  
        END IF  
        SET count TO the mapping of each character TO its frequency IN s  
        FOR each character IN the keys OF count  
            IF the frequency of character IN count IS LESS THAN k  
                RETURN the maximum value OF applying the FUNCTION longestSubstring TO each substring OBTAINED BY splitting s BY character WITH argument k  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
