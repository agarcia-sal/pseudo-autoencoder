CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s PLUS one MINUS one  
            APPEND index PLUS one TO result  
        END FOR  
        SET start TO zero  
        FOR end FROM zero TO the LENGTH OF s  
            IF end EQUALS the LENGTH OF s OR the character at position end of string s EQUALS the character indicating increase  
                REVERSE the subsequence of result from position start TO position end PLUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
