CLASS ListNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, next WITH DEFAULT None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO NEW ListNode WITH first element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position two TO THE END  
        SET node TO NEW ListNode WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND is_same_list of the reference to the next element of p1 AND the reference to the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates(self, head)  
        SET dummy TO NEW ListNode WITH zero AND head  
        SET prev TO dummy  
        WHILE head IS NOT None  
            IF the reference to the next element of head IS NOT None AND the value of head EQUALS the value of the reference to the next element of head  
                WHILE the reference to the next element of head IS NOT None AND the value of head EQUALS the value of the reference to the next element of head  
                    SET head TO the reference to the next element of head  
                END WHILE  
                SET the reference to the next element of prev TO the reference to the next element of head  
            ELSE  
                SET prev TO the reference to the next element of prev  
            END IF  
            SET head TO the reference to the next element of head  
        END WHILE  
        RETURN the reference to the next element of dummy  
    END FUNCTION  
END CLASS
