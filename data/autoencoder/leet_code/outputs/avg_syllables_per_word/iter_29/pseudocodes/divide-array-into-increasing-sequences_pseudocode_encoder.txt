CLASS Solution  
    FUNCTION canDivideIntoSubsequences(list_of_numbers NUMS list_of_integers k INTEGER) RETURNS boolean  
        SET frequency_counter TO the mapping of each element in list_of_numbers to its frequency count  
        SET maximum_frequency TO the greatest frequency value among all values in frequency_counter  
        SET length_of_list TO the LENGTH OF list_of_numbers  
        SET quotient TO length_of_list DIVIDED BY k  
        IF quotient GREATER THAN OR EQUAL TO maximum_frequency  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
