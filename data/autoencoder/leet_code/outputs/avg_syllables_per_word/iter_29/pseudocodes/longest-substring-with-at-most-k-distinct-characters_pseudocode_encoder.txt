CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s k)  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
          
        SET character_to_position_map TO empty mapping  
        SET left_pointer TO zero  
        SET maximum_length TO zero  
          
        FOR each index right_pointer FROM zero TO the LENGTH OF s MINUS one  
            ASSIGN rightmost_character TO character at position right_pointer of s  
            SET the value at key rightmost_character in character_to_position_map TO right_pointer  
              
            IF the SIZE OF character_to_position_map GREATER THAN k  
                SET leftmost_character TO the key in character_to_position_map whose value is the minimum among all values in character_to_position_map  
                SET left_pointer TO the value at key leftmost_character in character_to_position_map PLUS one  
                REMOVE the key leftmost_character FROM character_to_position_map  
            END IF  
              
            SET current_window_length TO right_pointer MINUS left_pointer PLUS one  
            SET maximum_length TO the GREATER BETWEEN maximum_length AND current_window_length  
        END FOR  
          
        RETURN maximum_length  
    END FUNCTION  
END CLASS
