CLASS Solution  
    FUNCTION minPatches(nums list_of_integers n integer)  
        SET missing TO one which is the smallest number we cannot form  
        SET patches TO zero which is the number of patches added  
        SET i TO zero which is the current index in the list of numbers  
        
        WHILE missing IS LESS THAN OR EQUAL TO n  
            IF i IS LESS THAN the LENGTH OF nums AND the element at position i of nums IS LESS THAN OR EQUAL TO missing  
                INCREMENT missing BY the element at position i of nums  
                INCREMENT i BY one  
            ELSE  
                INCREMENT missing BY missing  
                INCREMENT patches BY one  
            END IF  
        END WHILE  
        
        RETURN patches  
    END FUNCTION  
END CLASS
