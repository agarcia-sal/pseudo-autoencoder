CLASS Solution  
    FUNCTION longestValidParentheses(input_string)  
        SET stack TO a list containing the integer negative one  
        SET maximum_length TO zero  
        FOR each index AND character IN input_string enumerated from zero  
            IF character EQUALS the opening parenthesis character  
                APPEND index TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF the LENGTH OF stack EQUALS zero  
                    APPEND index TO stack  
                ELSE  
                    SET current_length TO index MINUS the last element in stack  
                    IF current_length GREATER THAN maximum_length  
                        SET maximum_length TO current_length  
                    END IF  
                END IF  
            END IF  
        END FOR  
        RETURN maximum_length  
    END FUNCTION  
END CLASS
