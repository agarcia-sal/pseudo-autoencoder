CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty OR values IS null  
        RETURN null  
    END IF  
    SET root TO a new instance of TreeNode with value equal to element at position one of values  
    SET i TO one  
    SET queue TO a new deque data structure  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the result of removing the element from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT null  
            SET node.left TO a new instance of TreeNode with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT null  
            SET node.right TO a new instance of TreeNode with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS null AND q IS null  
        RETURN True  
    ELSE IF p IS null OR q IS null  
        RETURN False  
    ELSE IF value of p IS NOT EQUALS value of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with p.left and q.left AND the result of calling is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree(self, root)  
        IF root IS null  
            RETURN zero  
        END IF  
        SET queue TO a new deque data structure containing a single tuple of root and zero  
        SET max_width TO zero  
        WHILE queue IS NOT empty  
            SET level_length TO the LENGTH OF queue  
            SET first_index TO the second element of the first tuple in queue  
            FOR each integer _ FROM zero TO level_length MINUS one  
                SET node TO the first element of the result of removing the element from the front of queue  
                SET index TO the second element of the result of removing the element from the front of queue  
                IF _ EQUALS level_length MINUS one  
                    SET max_width TO the greater of max_width AND index MINUS first_index PLUS one  
                END IF  
                IF node.left IS NOT null  
                    APPEND the tuple of node.left and the product of two AND index TO queue  
                END IF  
                IF node.right IS NOT null  
                    APPEND the tuple of node.right and the sum of the product of two AND index AND one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
