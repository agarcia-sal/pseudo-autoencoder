CLASS Solution  
    FUNCTION shortestDistance(maze, start, destination)  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO a collection consisting of the pair of one and zero the pair of negative one and zero the pair of zero and one and the pair of zero and negative one  
        SET start TO a tuple formed from the input start list  
        SET destination TO a tuple formed from the input destination list  
        SET visited TO an empty set  
        SET pq TO a collection consisting of a pair of zero and start representing distance and position  
        
        WHILE pq is not empty  
            REMOVE from pq the pair with the smallest distance and ASSIGN the first element to dist and the second element unpacked as x and y  
            
            IF the pair of x and y EQUALS destination  
                RETURN dist  
            END IF  
            
            IF the pair of x and y IS CONTAINED IN visited  
                CONTINUE to the next iteration of the loop  
            END IF  
            
            ADD the pair of x and y TO visited  
            
            FOR each pair dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE the sum of new_x and dx IS GREATER THAN OR EQUAL TO zero AND the sum of new_x and dx IS LESS THAN m AND the sum of new_y and dy IS GREATER THAN OR EQUAL TO zero AND the sum of new_y and dy IS LESS THAN n AND the element at position of new_x plus dx of the row at position of new_y plus dy of maze EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF the pair of new_x and new_y IS NOT CONTAINED IN visited  
                    ADD to pq a pair of the sum of dist and count and the pair of new_x and new_y  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
