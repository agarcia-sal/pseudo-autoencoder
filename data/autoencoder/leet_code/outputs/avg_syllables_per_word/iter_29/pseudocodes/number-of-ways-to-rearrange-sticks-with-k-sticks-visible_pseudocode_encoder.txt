CLASS Solution  
    FUNCTION rearrangeSticks(n number_of_sticks k number_of_visible_sticks)  
        SET MODULO_VALUE TO ten POWERED BY nine PLUS seven  
        
        INITIALIZE two_dimensional_collection dp WITH zero VALUES WITH dimensions number_of_sticks PLUS one BY number_of_visible_sticks PLUS one  
        
        SET element at position zero OF the first list in dp TO one  
        
        FOR index i FROM one TO number_of_sticks  
            FOR index j FROM one TO number_of_visible_sticks  
                SET current_value TO the element at position i MINUS one OF dp AT position j MINUS one PLUS the product of i MINUS one AND the element at position i MINUS one OF dp AT position j  
                SET element at position i OF dp AT position j TO the remainder of current_value DIVIDED BY MODULO_VALUE  
            END FOR  
        END FOR  
        
        RETURN the element at position number_of_sticks OF dp AT position number_of_visible_sticks  
    END FUNCTION  
END CLASS
