CLASS Solution  
    FUNCTION findClosestElements(arr list_of_integers k integer x integer) RETURNS list_of_integers  
        SET left_pointer TO the POSITION where x would be inserted to keep arr sorted MINUS one  
        SET right_pointer TO left_pointer PLUS one  
        WHILE right_pointer MINUS left_pointer MINUS one IS LESS THAN k  
            IF left_pointer EQUALS minus one OR right_pointer IS LESS THAN the LENGTH OF arr AND the ABSOLUTE VALUE of element at position right_pointer of arr MINUS x IS LESS THAN the ABSOLUTE VALUE of element at position left_pointer of arr MINUS x  
                INCREMENT right_pointer BY one  
            ELSE  
                DECREMENT left_pointer BY one  
            END IF  
        END WHILE  
        RETURN the SUBLIST of arr FROM position left_pointer PLUS one TO position right_pointer MINUS one inclusive  
    END FUNCTION  
END CLASS
