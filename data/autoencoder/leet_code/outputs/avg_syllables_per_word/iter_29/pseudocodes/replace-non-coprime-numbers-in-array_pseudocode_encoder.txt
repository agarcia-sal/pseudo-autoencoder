CLASS Solution  
    FUNCTION replaceNonCoprimes(nums)  
        SET stack TO empty list  
        FOR each number IN nums  
            WHILE the LENGTH OF stack IS greater than zero  
                SET greatest_common_divisor TO the greatest common divisor OF element at the last position of stack AND number  
                IF greatest_common_divisor EQUALS one  
                    BREAK the WHILE loop  
                END IF  
                SET number TO element at the last position of stack MULTIPLIED BY number DIVIDED BY greatest_common_divisor  
                REMOVE element at the last position FROM stack  
            END WHILE  
            APPEND number TO stack  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
