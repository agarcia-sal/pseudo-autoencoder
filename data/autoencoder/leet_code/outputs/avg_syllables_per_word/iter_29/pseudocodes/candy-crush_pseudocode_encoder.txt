CLASS Solution  
    FUNCTION candyCrush(board)  
        SET number_of_rows TO the LENGTH OF board  
        SET number_of_columns TO the LENGTH OF element at position zero of board  
        SET stable TO False  
        
        WHILE the NEGATION of stable HOLDS  
            SET stable TO True  
            SET positions_to_crush TO an empty set  
            
            FOR each row_index FROM zero TO number_of_rows MINUS one  
                FOR each column_index FROM zero TO number_of_columns MINUS two  
                    IF the absolute value of element at position row_index of board at position column_index EQUALS the absolute value of element at position row_index of board at position column_index PLUS one  
                    AND the absolute value of element at position row_index of board at position column_index PLUS one EQUALS the absolute value of element at position row_index of board at position column_index PLUS two  
                    AND the absolute value of element at position row_index of board at position column_index IS NOT EQUAL TO zero  
                    THEN  
                        SET stable TO False  
                        ADD the tuple of row_index and column_index TO positions_to_crush  
                        ADD the tuple of row_index and column_index PLUS one TO positions_to_crush  
                        ADD the tuple of row_index and column_index PLUS two TO positions_to_crush  
                    END IF  
                END FOR  
            END FOR  
            
            FOR each row_index FROM zero TO number_of_rows MINUS two  
                FOR each column_index FROM zero TO number_of_columns MINUS one  
                    IF the absolute value of element at position row_index of board at position column_index EQUALS the absolute value of element at position row_index PLUS one of board at position column_index  
                    AND the absolute value of element at position row_index PLUS one of board at position column_index EQUALS the absolute value of element at position row_index PLUS two of board at position column_index  
                    AND the absolute value of element at position row_index of board at position column_index IS NOT EQUAL TO zero  
                    THEN  
                        SET stable TO False  
                        ADD the tuple of row_index and column_index TO positions_to_crush  
                        ADD the tuple of row_index PLUS one and column_index TO positions_to_crush  
                        ADD the tuple of row_index PLUS two and column_index TO positions_to_crush  
                    END IF  
                END FOR  
            END FOR  
            
            FOR each position IN positions_to_crush  
                SET the element at position row_index of board at position column_index TO zero  
                WHERE row_index AND column_index ARE EXTRACTED FROM position  
            END FOR  
            
            FOR each column_index FROM zero TO number_of_columns MINUS one  
                SET write_row TO number_of_rows MINUS one  
                FOR each row_index FROM number_of_rows MINUS one DOWN TO zero  
                    IF element at position row_index of board at position column_index IS NOT EQUAL TO zero  
                    THEN  
                        SET element at position write_row of board at position column_index TO element at position row_index of board at position column_index  
                        DECREMENT write_row BY one  
                    END IF  
                END FOR  
                FOR each write_row_index FROM write_row DOWN TO zero  
                    SET element at position write_row_index of board at position column_index TO zero  
                END FOR  
            END FOR  
        
        END WHILE  
        
        RETURN board  
    END FUNCTION  
END CLASS
