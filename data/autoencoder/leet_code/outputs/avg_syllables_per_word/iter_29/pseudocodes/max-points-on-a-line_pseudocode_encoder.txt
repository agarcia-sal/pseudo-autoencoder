CLASS Solution  
    FUNCTION maxPoints(points)  
        FUNCTION slope(p1, p2)  
            IF the first element of p1 EQUALS the first element of p2  
                RETURN positive infinity  
            END IF  
            RETURN the difference between the second element of p1 AND the second element of p2 DIVIDED BY the difference between the first element of p1 AND the first element of p2  
        END FUNCTION  
        
        SET max_points TO zero  
        
        FOR index i FROM zero TO the LENGTH OF points MINUS one  
            SET slopes TO a new default dictionary with integer default  
            FOR index j FROM zero TO the LENGTH OF points MINUS one  
                IF i NOT EQUALS j  
                    SET s TO the result of slope function with points at position i AND points at position j  
                    INCREMENT the value at key s in slopes BY one  
                    IF the value at key s in slopes GREATER THAN max_points  
                        SET max_points TO the value at key s in slopes  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF the LENGTH OF points EQUALS zero  
            RETURN zero  
        ELSE  
            RETURN max_points PLUS one  
        END IF  
    END FUNCTION  
END CLASS
