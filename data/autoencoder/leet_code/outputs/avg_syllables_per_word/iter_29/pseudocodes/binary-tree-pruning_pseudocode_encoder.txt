CLASS TreeNode  
    FUNCTION __init__(val default TO zero left default TO no value right default TO no value)  
        SET the attribute val OF this instance TO val  
        SET the attribute left OF this instance TO left  
        SET the attribute right OF this instance TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values HAS NO ELEMENTS  
        RETURN no value  
    END IF  
    SET root TO a new instance of TreeNode initialized with element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue HOLDS ELEMENTS  
        SET node TO the element REMOVED FROM THE FRONT OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET the attribute left OF node TO a new instance of TreeNode initialized with element at position i of values  
            APPEND the attribute left OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET the attribute right OF node TO a new instance of TreeNode initialized with element at position i of values  
            APPEND the attribute right OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value  
        RETURN True  
    ELSE IF p IS no value OR q IS no value  
        RETURN False  
    ELSE IF the attribute val OF p IS NOT EQUAL TO the attribute val OF q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the attribute left OF p and the attribute left OF q AND the result of is_same_tree with the attribute right OF p and the attribute right OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree(root)  
        IF root IS no value  
            RETURN no value  
        END IF  
        SET the attribute left OF root TO the result of pruneTree with the attribute left OF root  
        SET the attribute right OF root TO the result of pruneTree with the attribute right OF root  
        IF the attribute val OF root EQUALS zero AND the attribute left OF root IS no value AND the attribute right OF root IS no value  
            RETURN no value  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
