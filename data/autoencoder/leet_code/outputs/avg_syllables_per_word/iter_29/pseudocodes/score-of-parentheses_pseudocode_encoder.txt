CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO an empty list  
        SET score TO zero  
        FOR each character IN s  
            IF the character EQUALS an opening parenthesis  
                APPEND the current value of score TO stack  
                SET score TO zero  
            ELSE  
                SET score TO the last element removed FROM stack PLUS one IF the current score EQUALS zero OTHERWISE two TIMES the current score  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
