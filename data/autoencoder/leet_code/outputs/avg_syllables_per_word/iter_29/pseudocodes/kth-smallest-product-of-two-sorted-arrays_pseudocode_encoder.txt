CLASS Solution  
    FUNCTION kthSmallestProduct(nums1 nums1_list OF integers nums2 nums2_list OF integers k kth_integer) RETURNS integer  
        FUNCTION count(p p_integer) RETURNS integer  
            SET cnt count_integer TO zero  
            SET n length_of_nums2 TO the LENGTH OF nums2_list  
            FOR each x IN nums1_list  
                IF x GREATER THAN zero  
                    INCREMENT cnt BY the POSITION TO INSERT p DIVIDED BY x IN nums2_list TO THE RIGHT  
                ELSE IF x LESS THAN zero  
                    INCREMENT cnt BY n MINUS the POSITION TO INSERT p DIVIDED BY x IN nums2_list TO THE LEFT  
                ELSE  
                    IF p GREATER THAN OR EQUAL TO zero  
                        INCREMENT cnt BY n  
                    END IF  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  
        SET mx maximum_value TO the MULTIPLICATION OF the MAXIMUM BETWEEN the ABSOLUTE VALUE OF the FIRST element of nums1_list AND the ABSOLUTE VALUE of the LAST element of nums1_list AND the MAXIMUM BETWEEN the ABSOLUTE VALUE of the FIRST element of nums2_list AND the ABSOLUTE VALUE of the LAST element of nums2_list  
        SET position_index TO the POSITION TO INSERT kth_integer IN the integer range from NEGATIVE mx TO mx INCLUSIVE USING count AS THE KEY FUNCTION  
        RETURN position_index MINUS mx  
    END FUNCTION  
END CLASS
