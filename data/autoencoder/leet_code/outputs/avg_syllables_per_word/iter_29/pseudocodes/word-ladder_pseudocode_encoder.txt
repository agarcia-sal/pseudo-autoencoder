CLASS Solution  
    FUNCTION ladderLength(beginWord word_for_starting_word endWord word_for_ending_word wordList list_of_words)  
        IF the word_for_ending_word IS NOT IN the list_of_words  
            RETURN zero  
        END IF  

        SET word_dict TO an empty dictionary that defaults to an empty list  
        SET word_length TO the LENGTH OF the word_for_starting_word  

        FOR each word IN the list_of_words  
            FOR index FROM zero TO one LESS THAN the word_length  
                SET intermediate_word TO the concatenation of the substring from position zero TO position index MINUS one of word PLUS a literal asterisk PLUS the substring from position index PLUS one TO the end of word  
                APPEND word TO the list at key intermediate_word in word_dict  
            END FOR  
        END FOR  

        SET queue TO a double ended queue containing a tuple of the word_for_starting_word and one  
        SET visited TO a set containing the word_for_starting_word  

        WHILE the queue IS NOT empty  
            REMOVE and RETURN the first element FROM the queue and ASSIGN TO current_word and current_length  
            
            FOR index FROM zero TO one LESS THAN the word_length  
                SET intermediate_word TO the concatenation of the substring from position zero TO position index MINUS one of current_word PLUS a literal asterisk PLUS the substring from position index PLUS one TO the end of current_word  
                FOR each word IN the list at key intermediate_word in word_dict  
                    IF word EQUALS the word_for_ending_word  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word IS NOT IN visited  
                        ADD word TO visited  
                        APPEND a tuple of word and current_length PLUS one TO the queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN zero  
    END FUNCTION  
END CLASS
