CLASS Solution  
    FUNCTION isValidPalindrome(s k)  
        FUNCTION longest_palindromic_subseq(string_input)  
            SET length_of_string TO the LENGTH OF string_input  
            SET dp_matrix TO a new two dimensional collection with each element set to zero with dimension length_of_string BY length_of_string  
            
            FOR index FROM zero TO length_of_string MINUS one  
                SET element at position index AT element at position index OF dp_matrix TO one  
            END FOR  
            
            FOR current_length FROM two TO length_of_string  
                FOR start_index FROM zero TO length_of_string MINUS current_length  
                    SET end_index TO start_index PLUS current_length MINUS one  
                    IF character at position start_index OF string_input EQUALS character at position end_index OF string_input  
                        SET element at position start_index AT element at position end_index OF dp_matrix TO element at position start_index PLUS one AT element at position end_index MINUS one OF dp_matrix PLUS two  
                    ELSE  
                        SET element at position start_index AT element at position end_index OF dp_matrix TO the MAXIMUM OF element at position start_index PLUS one AT element at position end_index OF dp_matrix AND element at position start_index AT element at position end_index MINUS one OF dp_matrix  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN element at position zero AT element at position length_of_string MINUS one OF dp_matrix  
        END FUNCTION  
        
        SET longest_palindrome_length TO the RESULT OF longest_palindromic_subseq APPLIED TO s  
        RETURN the LENGTH OF s MINUS longest_palindrome_length LESS THAN OR EQUAL TO k  
    END FUNCTION  
END CLASS
