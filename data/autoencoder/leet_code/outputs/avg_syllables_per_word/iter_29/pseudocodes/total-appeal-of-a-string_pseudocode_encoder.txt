CLASS Solution  
    FUNCTION appealSum(s)  
        SET last_seen_indices TO a collection of twenty six elements each SET TO negative one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        SET position TO zero  
        FOR each character IN s  
            SET character_index TO the numerical position of character in the alphabet counting from zero where character lowercase a equals zero  
            SET difference TO position MINUS the element at position character_index of last_seen_indices  
            INCREMENT current_appeal BY difference  
            INCREMENT total_appeal BY current_appeal  
            SET the element at position character_index of last_seen_indices TO position  
            INCREMENT position BY one  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
