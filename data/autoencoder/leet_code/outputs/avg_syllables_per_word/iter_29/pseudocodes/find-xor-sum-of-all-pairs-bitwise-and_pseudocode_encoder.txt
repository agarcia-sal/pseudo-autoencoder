CLASS Solution  
    FUNCTION getXORSum arr1 arr2  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        FOR each number IN arr1  
            SET xor_sum_arr1 TO the result of xor_sum_arr1 EXCLUSIVE OR number  
        END FOR  
        FOR each number IN arr2  
            SET xor_sum_arr2 TO the result of xor_sum_arr2 EXCLUSIVE OR number  
        END FOR  
        RETURN the result of xor_sum_arr1 AND xor_sum_arr2  
    END FUNCTION  
END CLASS
