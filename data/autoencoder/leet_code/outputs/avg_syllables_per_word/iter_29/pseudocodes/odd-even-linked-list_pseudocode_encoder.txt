CLASS ListNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, next WITH DEFAULT none)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS none OR values IS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode INITIALIZED WITH the element at position one of values  
    SET p TO head  
    FOR each val IN values FROM position two TO the end  
        SET node TO a new ListNode INITIALIZED WITH val  
        SET the next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN (the val of p1 EQUALS the val of p2) AND is_same_list(the next of p1, the next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList(self, head)  
        IF head IS none OR the next of head IS none OR the next of the next of head IS none  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO the next of head  
        SET even_head TO even  
        WHILE even IS NOT none AND the next of even IS NOT none  
            SET the next of odd TO the next of even  
            SET odd TO the next of odd  
            SET the next of even TO the next of odd  
            SET even TO the next of even  
        END WHILE  
        SET the next of odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
