CLASS Solution  
    FUNCTION countRoutes(locations PARAMETER list_of_integers start PARAMETER integer finish PARAMETER integer fuel PARAMETER integer) RETURNS integer  
        SET MODULO_VALUE TO ten POWERED BY nine PLUS one  
        SET number_of_locations TO the LENGTH OF locations  
        
        FUNCTION dp(city PARAMETER integer remaining_fuel PARAMETER integer) RETURNS integer  
            IF remaining_fuel IS LESS THAN zero THEN  
                RETURN zero  
            END IF  
            SET route_count TO zero  
            IF city EQUALS finish THEN  
                SET route_count TO one  
            END IF  
            FOR next_city FROM zero TO number_of_locations MINUS one  
                IF next_city IS NOT EQUAL TO city THEN  
                    SET fuel_cost TO the ABSOLUTE VALUE OF the element at position city of locations MINUS the element at position next_city of locations  
                    IF remaining_fuel IS GREATER THAN OR EQUAL TO fuel_cost THEN  
                        SET route_count TO the REMAINDER OF route_count PLUS the RESULT OF dp with parameters next_city and remaining_fuel MINUS fuel_cost DIVIDED BY MODULO_VALUE  
                    END IF  
                END IF  
            END FOR  
            RETURN route_count  
        END FUNCTION  
        
        RETURN the RESULT OF dp with parameters start and fuel  
    END FUNCTION  
END CLASS
