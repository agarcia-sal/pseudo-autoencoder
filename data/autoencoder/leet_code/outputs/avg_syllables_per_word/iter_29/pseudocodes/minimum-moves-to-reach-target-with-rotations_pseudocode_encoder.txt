CLASS Solution  
    FUNCTION minimumMoves(grid)  
        SET n TO the LENGTH OF grid  
        SET queue TO a new double ended queue with the single element consisting of zero zero zero one zero representing tail row tail column head row head column steps  
        SET visited TO a new empty set containing the single tuple zero zero zero one representing visited states  
        WHILE queue is not empty  
            SET r1 c1 r2 c2 steps TO the FIRST element REMOVED FROM the LEFT OF queue  
            IF r1 EQUALS n MINUS one AND c1 EQUALS n MINUS two AND r2 EQUALS n MINUS one AND c2 EQUALS n MINUS one  
                RETURN steps  
            END IF  
            IF c2 PLUS one LESS THAN n AND the element at position r1 of grid at position c1 PLUS one EQUALS zero AND the element at position r2 of grid at position c2 PLUS one EQUALS zero AND the tuple r1 c1 PLUS one r2 c2 PLUS one NOT IN visited  
                ADD the tuple r1 c1 PLUS one r2 c2 PLUS one TO visited  
                APPEND the tuple r1 c1 PLUS one r2 c2 PLUS one steps PLUS one TO queue  
            END IF  
            IF r2 PLUS one LESS THAN n AND the element at position r1 PLUS one of grid at position c1 EQUALS zero AND the element at position r2 PLUS one of grid at position c2 EQUALS zero AND the tuple r1 PLUS one c1 r2 PLUS one c2 NOT IN visited  
                ADD the tuple r1 PLUS one c1 r2 PLUS one c2 TO visited  
                APPEND the tuple r1 PLUS one c1 r2 PLUS one c2 steps PLUS one TO queue  
            END IF  
            IF r1 EQUALS r2 AND r1 PLUS one LESS THAN n AND the element at position r1 PLUS one of grid at position c1 EQUALS zero AND the element at position r1 PLUS one of grid at position c2 EQUALS zero AND the tuple r1 c1 r1 PLUS one c1 NOT IN visited  
                ADD the tuple r1 c1 r1 PLUS one c1 TO visited  
                APPEND the tuple r1 c1 r1 PLUS one c1 steps PLUS one TO queue  
            END IF  
            IF c1 EQUALS c2 AND c1 PLUS one LESS THAN n AND the element at position r1 of grid at position c1 PLUS one EQUALS zero AND the element at position r2 of grid at position c2 PLUS one EQUALS zero AND the tuple r1 c1 r1 c1 PLUS one NOT IN visited  
                ADD the tuple r1 c1 r1 c1 PLUS one TO visited  
                APPEND the tuple r1 c1 r1 c1 PLUS one steps PLUS one TO queue  
            END IF  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
