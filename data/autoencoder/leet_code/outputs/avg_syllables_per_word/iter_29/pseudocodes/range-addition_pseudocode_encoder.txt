CLASS Solution  
    FUNCTION getModifiedArray(length_parameter, updates_parameter)  
        SET diff_array TO a new list of zeros with the size of length_parameter PLUS one  
        FOR each update_element CONTAINING start_index_element end_index_element and increment_element IN updates_parameter  
            INCREMENT element at position start_index_element of diff_array BY increment_element  
            IF start_index_element PLUS one LESS THAN OR EQUAL TO end_index_element  
                IF end_index_element PLUS one LESS THAN length_parameter  
                    DECREMENT element at position end_index_element PLUS one of diff_array BY increment_element  
                END IF  
            END IF  
        END FOR  
        SET result_array TO a new list of zeros with the size of length_parameter  
        SET element at position zero of result_array TO element at position zero of diff_array  
        FOR index_variable FROM one TO length_parameter MINUS one  
            SET element at position index_variable of result_array TO element at position index_variable MINUS one of result_array PLUS element at position index_variable of diff_array  
        END FOR  
        RETURN result_array  
    END FUNCTION  
END CLASS
