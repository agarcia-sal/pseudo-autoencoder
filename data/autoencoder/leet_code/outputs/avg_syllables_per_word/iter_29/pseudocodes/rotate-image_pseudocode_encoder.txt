CLASS Solution  
    FUNCTION rotate(matrix)  
        SET size TO the LENGTH OF matrix  
        FOR index_i FROM zero TO size MINUS one  
            FOR index_j FROM index_i TO size MINUS one  
                SET temporary_value TO the element at position index_i of the element at position index_j of matrix  
                SET the element at position index_j of the element at position index_i of matrix TO the element at position index_i of the element at position index_j of matrix  
                SET the element at position index_i of the element at position index_j of matrix TO temporary_value  
            END FOR  
        END FOR  
        FOR index_i FROM zero TO size MINUS one  
            REVERSE the element at position index_i of matrix  
        END FOR  
    END FUNCTION  
END CLASS
