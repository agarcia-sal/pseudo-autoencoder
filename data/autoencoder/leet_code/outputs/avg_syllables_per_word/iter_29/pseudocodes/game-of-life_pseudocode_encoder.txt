CLASS Solution  
    FUNCTION gameOfLife(board)  
        IF board IS empty OR the first element of board IS empty  
            RETURN  
        END IF  
        SET number_of_rows TO the LENGTH OF board  
        SET number_of_columns TO the LENGTH OF the first element of board  
        SET list_of_directions TO the list of pairs each consisting of minus one and minus one, minus one and zero, minus one and one, zero and minus one, zero and one, one and minus one, one and zero, and one and one  
        FOR each row_index FROM zero TO number_of_rows MINUS one  
            FOR each column_index FROM zero TO number_of_columns MINUS one  
                SET count_of_live_neighbors TO zero  
                FOR each direction_pair IN list_of_directions  
                    SET neighbor_row TO row_index PLUS the first element of direction_pair  
                    SET neighbor_column TO column_index PLUS the second element of direction_pair  
                    IF neighbor_row IS GREATER THAN OR EQUAL TO zero AND neighbor_row IS LESS THAN number_of_rows AND neighbor_column IS GREATER THAN OR EQUAL TO zero AND neighbor_column IS LESS THAN number_of_columns AND (element at position neighbor_row of board AT position neighbor_column EQUALS one OR element at position neighbor_row of board AT position neighbor_column EQUALS three)  
                        INCREMENT count_of_live_neighbors BY one  
                    END IF  
                END FOR  
                IF element at position row_index of board AT position column_index EQUALS one  
                    IF count_of_live_neighbors IS LESS THAN two OR count_of_live_neighbors IS GREATER THAN three  
                        SET element at position row_index of board AT position column_index TO three  
                    END IF  
                ELSE  
                    IF count_of_live_neighbors EQUALS three  
                        SET element at position row_index of board AT position column_index TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        FOR each row_index FROM zero TO number_of_rows MINUS one  
            FOR each column_index FROM zero TO number_of_columns MINUS one  
                IF element at position row_index of board AT position column_index EQUALS two  
                    SET element at position row_index of board AT position column_index TO one  
                ELSE IF element at position row_index of board AT position column_index EQUALS three  
                    SET element at position row_index of board AT position column_index TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
