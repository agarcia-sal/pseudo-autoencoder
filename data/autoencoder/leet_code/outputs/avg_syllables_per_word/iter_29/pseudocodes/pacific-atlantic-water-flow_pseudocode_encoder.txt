CLASS Solution  
    FUNCTION pacificAtlantic(heights)  
        FUNCTION bfs(queue_structure visited_set)  
            WHILE the LENGTH OF queue_structure IS GREATER THAN zero  
                FOR a number of TIMES EQUAL TO the LENGTH OF queue_structure  
                    SET current_indices TO the REMOVAL of the FIRST ELEMENT from queue_structure  
                    SET i TO the FIRST ELEMENT of current_indices  
                    SET j TO the SECOND ELEMENT of current_indices  
                    FOR each pair IN the LIST consisting of the PAIRS zero MINUS one zero ONE zero AND MINUS one zero  
                        SET a TO the FIRST ELEMENT of pair  
                        SET b TO the SECOND ELEMENT of pair  
                        SET x TO i PLUS a  
                        SET y TO j PLUS b  
                        IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN m AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN n AND the PAIR x y IS NOT IN visited_set AND the ELEMENT at position x OF heights AT position y IS GREATER THAN OR EQUAL TO the ELEMENT at position i OF heights AT position j  
                            ADD the PAIR x y TO visited_set  
                            ADD the PAIR x y TO the END of queue_structure  
                        END IF  
                    END FOR  
                END FOR  
            END WHILE  
        END FUNCTION  

        SET m TO the LENGTH OF heights  
        SET n TO the LENGTH OF the ELEMENT at position zero OF heights  
        SET vis1 TO an EMPTY SET  
        SET vis2 TO an EMPTY SET  
        SET q1 TO a NEW DOUBLE ENDED QUEUE  
        SET q2 TO a NEW DOUBLE ENDED QUEUE  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF i EQUALS zero OR j EQUALS zero  
                    ADD the PAIR i j TO vis1  
                    ADD the PAIR i j TO the END of q1  
                END IF  
                IF i EQUALS m MINUS one OR j EQUALS n MINUS one  
                    ADD the PAIR i j TO vis2  
                    ADD the PAIR i j TO the END of q2  
                END IF  
            END FOR  
        END FOR  
        CALL bfs WITH PARAMETERS q1 AND vis1  
        CALL bfs WITH PARAMETERS q2 AND vis2  
        SET result TO an EMPTY LIST  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the PAIR i j IS IN vis1 AND the PAIR i j IS IN vis2  
                    ADD the PAIR i j TO result  
                END IF  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
