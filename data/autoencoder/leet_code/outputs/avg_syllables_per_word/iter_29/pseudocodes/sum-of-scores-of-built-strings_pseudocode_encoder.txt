CLASS Solution  
    FUNCTION sumScores(s)  
        SET n TO the LENGTH OF s  
        INITIALIZE lcp AS a list of zeroes with the LENGTH OF s  

        SET l TO zero  
        SET r TO zero  
        
        FOR index FROM one TO n MINUS one  
            IF index IS LESS THAN OR EQUAL TO r THEN  
                SET lcp AT position index TO the MINIMUM OF r MINUS index PLUS one AND lcp AT position index MINUS l  
            END IF  
            
            WHILE index PLUS lcp AT position index IS LESS THAN n AND character at position lcp AT position index OF s EQUALS character at position index PLUS lcp AT position index OF s DO  
                INCREMENT lcp AT position index BY one  
            END WHILE  
            
            IF index PLUS lcp AT position index MINUS one IS GREATER THAN r THEN  
                SET l TO index  
                SET r TO index PLUS lcp AT position index MINUS one  
            END IF  
        END FOR  
        
        SET total_score TO the SUM OF all ELEMENTS IN lcp PLUS n  
        RETURN total_score  
    END FUNCTION  
END CLASS
