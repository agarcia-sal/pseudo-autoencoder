CLASS Solution  
    FUNCTION findLength(nums1 list_of_integers nums2 list_of_integers) RETURNS integer  
        SET dp TO a two-dimensional collection with the number of rows equal to the LENGTH OF nums1 PLUS one and the number of columns equal to the LENGTH OF nums2 PLUS one each element initialized to zero  
        SET max_length TO zero  
        FOR index_i FROM one TO the LENGTH OF nums1  
            FOR index_j FROM one TO the LENGTH OF nums2  
                IF the element at position index_i MINUS one of nums1 EQUALS the element at position index_j MINUS one of nums2  
                    SET the element at row index_i column index_j of dp TO the element at row index_i MINUS one column index_j MINUS one of dp PLUS one  
                    IF max_length LESS THAN the element at row index_i column index_j of dp  
                        SET max_length TO the element at row index_i column index_j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
