CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUALS plus OR token EQUALS minus OR token EQUALS multiplied by OR token EQUALS divided by  
                SET value_b TO the last element REMOVED FROM stack  
                SET value_a TO the last element REMOVED FROM stack  
                IF token EQUALS plus  
                    APPEND value_a PLUS value_b TO stack  
                ELSE IF token EQUALS minus  
                    APPEND value_a MINUS value_b TO stack  
                ELSE IF token EQUALS multiplied by  
                    APPEND value_a MULTIPLIED BY value_b TO stack  
                ELSE IF token EQUALS divided by  
                    SET division_result TO value_a DIVIDED BY value_b  
                    SET truncated_result TO the integer part of division_result that truncates towards zero  
                    APPEND truncated_result TO stack  
                END ELSE IF  
            ELSE  
                CONVERT token TO integer_value  
                APPEND integer_value TO stack  
            END IF  
        END FOR  
        RETURN the first element at position one of stack  
    END FUNCTION  
END CLASS
