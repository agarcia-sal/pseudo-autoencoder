CLASS Solution  
    FUNCTION maxProfit(prices)  
        SET maximum profit TO zero  
        FOR index FROM two TO the LENGTH OF prices  
            IF element at position index of prices IS GREATER THAN element at position index MINUS one of prices  
                INCREMENT maximum profit BY element at position index of prices MINUS element at position index MINUS one of prices  
            END IF  
        END FOR  
        RETURN maximum profit  
    END FUNCTION  
END CLASS
