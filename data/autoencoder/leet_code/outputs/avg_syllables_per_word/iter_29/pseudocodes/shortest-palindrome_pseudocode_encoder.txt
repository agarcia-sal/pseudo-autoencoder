CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS the empty string THEN  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS the reverse of check_str  
        END FUNCTION  
        
        FOR index FROM the LENGTH OF s DOWN TO zero BY negative one  
            IF is_palindrome(substring from position zero TO position index MINUS one of s) THEN  
                SET to_add TO the reverse of the substring from position index TO the end of s  
                RETURN the concatenation of to_add WITH s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
