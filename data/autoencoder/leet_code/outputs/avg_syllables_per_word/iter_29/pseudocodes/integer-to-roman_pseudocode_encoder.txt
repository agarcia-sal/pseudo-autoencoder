CLASS Solution  
    FUNCTION intToRoman(num)  
        SET list_of_integer_values TO list containing one thousand nine hundred ninety nine hundred four hundred one hundred ninety fifty forty ten nine five four one  
        SET list_of_roman_symbols TO list containing the roman numeral for one thousand the roman numeral for nine hundred the roman numeral for five hundred the roman numeral for four hundred the roman numeral for one hundred the roman numeral for ninety the roman numeral for fifty the roman numeral for forty the roman numeral for ten the roman numeral for nine the roman numeral for five the roman numeral for four the roman numeral for one  
        SET roman_numeral_string TO empty string  
        SET index_variable TO zero  
        WHILE num GREATER THAN zero  
            SET quotient_variable TO the integer division of num BY element at position index_variable of list_of_integer_values  
            FOR each iteration FROM one TO quotient_variable  
                APPEND element at position index_variable of list_of_roman_symbols TO roman_numeral_string  
                DECREMENT num BY element at position index_variable of list_of_integer_values  
            END FOR  
            INCREMENT index_variable BY one  
        END WHILE  
        RETURN roman_numeral_string  
    END FUNCTION  
END CLASS
