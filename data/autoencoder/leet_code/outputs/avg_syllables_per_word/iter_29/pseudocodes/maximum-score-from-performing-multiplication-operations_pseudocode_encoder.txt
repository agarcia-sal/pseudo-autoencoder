CLASS Solution  
    FUNCTION maximumScore(nums multipliers)  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zero repeated m PLUS one times  
        FOR i FROM m MINUS one DOWN TO zero  
            SET new_dp TO a list of zero repeated m PLUS one times  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS i MINUS left  
                SET mul TO element at position i of multipliers  
                SET score_when_taking_left TO mul MULTIPLIED BY element at position left of nums PLUS element at position left PLUS one of dp  
                SET score_when_taking_right TO mul MULTIPLIED BY element at position right of nums PLUS element at position left of dp  
                IF score_when_taking_left GREATER THAN score_when_taking_right  
                    SET element at position left of new_dp TO score_when_taking_left  
                ELSE  
                    SET element at position left of new_dp TO score_when_taking_right  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
