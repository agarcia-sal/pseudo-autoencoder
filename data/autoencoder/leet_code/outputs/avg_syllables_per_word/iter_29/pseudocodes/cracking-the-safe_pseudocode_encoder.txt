CLASS Solution  
    FUNCTION crackSafe(n k)  
        SET visited TO an empty set  
        SET sequence TO an empty list  

        FUNCTION dfs(node)  
            FOR each integer i FROM zero TO k MINUS one  
                SET edge TO concatenation of node AND string representation of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs with argument substring of edge FROM position two TO position length of edge  
                    APPEND string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  

        SET start_node TO a string consisting of n MINUS one repetitions of character zero  
        CALL dfs with argument start_node  
        SET full_sequence TO concatenation of the reverse of sequence AND start_node  
        RETURN full_sequence  
    END FUNCTION  
END CLASS
