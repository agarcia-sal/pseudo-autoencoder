CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO an empty list  
        
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the current asteroid IS LESS THAN zero AND the element at position the last index of stack IS GREATER THAN zero  
                IF the element at position the last index of stack IS LESS THAN the NEGATION OF the current asteroid  
                    REMOVE the element at position the last index FROM stack  
                    CONTINUE with the next iteration of the WHILE loop  
                ELSE IF the element at position the last index of stack EQUALS the NEGATION OF the current asteroid  
                    REMOVE the element at position the last index FROM stack  
                END IF  
                BREAK the WHILE loop  
            END WHILE  
            IF the WHILE loop was not broken by a BREAK statement  
                APPEND the current asteroid TO stack  
            END IF  
        END FOR  
        
        RETURN stack  
    END FUNCTION  
END CLASS
