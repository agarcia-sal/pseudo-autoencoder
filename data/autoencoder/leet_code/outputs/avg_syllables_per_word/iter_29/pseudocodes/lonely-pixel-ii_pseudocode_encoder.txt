CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture IS empty OR the LENGTH OF picture AT position zero IS zero THEN  
            RETURN zero  
        END IF  

        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF picture AT position zero  
        SET row_count TO a list of size m FILLED WITH zero  
        SET col_count TO a list of size n FILLED WITH zero  
        INITIALIZE row_patterns AS a mapping from patterns to counts WITH default integer zero  

        FOR each r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of all elements AT positions zero TO n MINUS one of picture AT r  
            INCREMENT the count of pattern IN row_patterns BY one  
            FOR each c FROM zero TO n MINUS one  
                IF the element at position c of picture at position r EQUALS the character B THEN  
                    INCREMENT the element at position r of row_count BY one  
                    INCREMENT the element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  

        SET lonely_pixels TO zero  

        FOR each r FROM zero TO m MINUS one  
            FOR each c FROM zero TO n MINUS one  
                IF the element at position c of picture at position r EQUALS the character B THEN  
                    SET pattern TO the concatenation of all elements AT positions zero TO n MINUS one of picture AT r  
                    IF the element at position r of row_count EQUALS target AND the element at position c of col_count EQUALS target AND the count of pattern IN row_patterns EQUALS target THEN  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
