CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty collection for key value pairs  
        
        FUNCTION can_win_helper(state)  
            IF state IS A KEY IN memo  
                RETURN the value associated with state in memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                SET first_character TO character at position index of state  
                SET second_character TO character at position index PLUS one of state  
                IF first_character EQUALS the plus character AND second_character EQUALS the plus character  
                    SET new_state TO the concatenation of the substring from position zero TO position index MINUS one of state  
                    PLUS the two minus characters  
                    PLUS the substring from position index PLUS two TO the end of state  
                    IF the NEGATION of the result of calling can_win_helper WITH new_state  
                        SET the value associated with state in memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value associated with state in memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN the result of calling can_win_helper WITH currentState  
    END FUNCTION  
END CLASS
