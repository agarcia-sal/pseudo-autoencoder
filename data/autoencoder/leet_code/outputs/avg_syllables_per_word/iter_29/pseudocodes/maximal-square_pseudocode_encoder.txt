CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET dp TO a two dimensional collection of zeroes with size m PLUS one BY n PLUS one  
        SET max_side TO zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of matrix AND element at position j MINUS one of element at position i MINUS one of matrix EQUALS one as string  
                    SET dp at position i AND position j TO the MINIMUM OF dp at position i MINUS one AND position j AND dp at position i AND position j MINUS one AND dp at position i MINUS one AND position j MINUS one PLUS one  
                    SET max_side TO the MAXIMUM OF max_side AND dp at position i AND position j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
