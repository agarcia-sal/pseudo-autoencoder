CLASS Solution  
    FUNCTION kSmallestPairs(nums1 list_of_integers nums2 list_of_integers k integer) RETURNS list_of_list_of_integers  
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k IS LESS THAN OR EQUAL TO zero THEN  
            RETURN empty list  
        END IF  
        
        SET min_heap TO empty list  
        
        FOR index i FROM zero TO the MINIMUM OF k AND the LENGTH OF nums1 MINUS one  
            SET sum_value TO element at position i of nums1 PLUS element at position zero of nums2  
            PUSH the tuple containing sum_value AND i AND zero INTO min_heap USING heap push operation  
        END FOR  
        
        SET result TO empty list  
        
        WHILE min_heap IS NOT EMPTY AND the LENGTH OF result IS LESS THAN k  
            POP from min_heap USING heap pop operation AND IGNORE first element ASSIGN second element TO i AND third element TO j  
            APPEND a list containing element at position i of nums1 AND element at position j of nums2 TO result  
            
            IF j PLUS one IS LESS THAN the LENGTH OF nums2 THEN  
                SET new_sum TO element at position i of nums1 PLUS element at position j PLUS one of nums2  
                PUSH the tuple containing new_sum AND i AND j PLUS one INTO min_heap USING heap push operation  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
