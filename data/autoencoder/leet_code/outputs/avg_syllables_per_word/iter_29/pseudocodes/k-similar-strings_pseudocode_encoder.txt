CLASS Solution  
    FUNCTION kSimilarity(s1 s1 STRING, s2 s2 STRING) RETURNS integer  
        FUNCTION neighbors(s s STRING)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND character at position i of s EQUALS character at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR each index j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF character at position j of s EQUALS character at position i of s2 AND character at position j of s NOT EQUALS character at position j of s2  
                    YIELD concatenation of substring from position zero TO position i MINUS one of s PLUS character at position j of s PLUS substring from position i PLUS one TO position j MINUS one of s PLUS character at position i of s PLUS substring from position j PLUS one TO end of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET queue TO an empty double ended queue containing a tuple of s1 and zero  
        SET visited TO a set containing s1  
        
        WHILE queue is not empty  
            REMOVE the first element from queue and ASSIGN TO s and swaps  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN the result of calling neighbors with s  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND a tuple of neighbor and swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
