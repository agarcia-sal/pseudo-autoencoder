CLASS Solution  
    FUNCTION dailyTemperatures(temperatures)  
        SET number_of_temperatures TO the LENGTH OF temperatures  
        SET answer TO a list OF number_of_temperatures ELEMENTS ALL SET TO zero  
        SET stack TO an empty list  
        FOR index FROM zero TO number_of_temperatures MINUS one  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position index of temperatures IS GREATER THAN the element at position the element at the last position of stack of temperatures  
                SET previous_index TO the element REMOVED FROM the end of stack  
                SET the element at position previous_index of answer TO index MINUS previous_index  
            END WHILE  
            APPEND index TO stack  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
