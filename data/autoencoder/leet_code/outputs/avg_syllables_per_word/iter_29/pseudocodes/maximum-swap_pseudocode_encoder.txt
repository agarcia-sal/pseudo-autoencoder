CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits_list TO the LIST OF THE characters OF the STRING REPRESENTATION OF num  
        SET last_occurrence_dictionary TO an EMPTY DICTIONARY  
        FOR each index AND digit_character IN the ENUMERATED digits_list  
            SET digit_number TO the INTEGER EQUIVALENT OF digit_character  
            SET the VALUE at key digit_number IN last_occurrence_dictionary TO index  
        END FOR  
        FOR each index AND digit_character IN the ENUMERATED digits_list  
            SET digit_number TO the INTEGER EQUIVALENT OF digit_character  
            FOR candidate_digit FROM nine DOWN TO the digit_number PLUS one  
                IF the VALUE ASSOCIATED WITH candidate_digit IN last_occurrence_dictionary GREATER THAN index  
                    SET temporary_variable TO the ELEMENT AT POSITION index OF digits_list  
                    SET the ELEMENT AT POSITION index OF digits_list TO the ELEMENT AT POSITION THE VALUE ASSOCIATED WITH candidate_digit IN last_occurrence_dictionary OF digits_list  
                    SET the ELEMENT AT POSITION THE VALUE ASSOCIATED WITH candidate_digit IN last_occurrence_dictionary OF digits_list TO temporary_variable  
                    SET swapped_number_string TO the CONCATENATION OF ALL ELEMENTS IN digits_list  
                    SET swapped_number TO the INTEGER EQUIVALENT OF swapped_number_string  
                    RETURN swapped_number  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
