CLASS Solution  
    FUNCTION generateParenthesis(parameter n)  
        FUNCTION backtrack(parameter s default to empty string parameter left default to zero parameter right default to zero)  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s CONCATENATED WITH opening parenthesis AND left INCREMENTED BY one AND right UNCHANGED  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s CONCATENATED WITH closing parenthesis AND left UNCHANGED AND right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH no arguments  
        RETURN result  
    END FUNCTION  
END CLASS
