CLASS Solution  
    FUNCTION findStrobogrammatic(n)  
        FUNCTION dfs(u)  
            IF u EQUALS zero  
                RETURN list containing the empty string  
            END IF  
            IF u EQUALS one  
                RETURN list containing the string consisting of character zero and the string consisting of character one and the string consisting of character eight  
            END IF  
            SET answer_list TO empty list  
            FOR each string v IN the result of calling dfs WITH parameter u MINUS two  
                FOR each pair of characters left_character and right_character IN the collection of pairs consisting of character one and character one and character eight and character eight and character six and character nine and character nine and character six  
                    APPEND the concatenation of left_character followed by string v followed by right_character TO answer_list  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND the concatenation of character zero followed by string v followed by character zero TO answer_list  
                END IF  
            END FOR  
            RETURN answer_list  
        END FUNCTION  
        RETURN the result of calling dfs WITH parameter n  
    END FUNCTION  
END CLASS
