CLASS Solution  
    FUNCTION minimumTime(n, relations, time)  
        SET graph TO a new mapping where each key corresponds to a list of values  
        FOR each pair of previous course and next course IN relations  
            ADD previous course TO the list corresponding to next course in graph  
        END FOR  
        
        SET memo TO a list of length n PLUS one where each element is set TO negative one  
        
        FUNCTION dfs(course)  
            IF the element at position course of memo NOT EQUALS negative one  
                RETURN the element at position course of memo  
            END IF  
            
            SET maximum prerequisite time TO zero  
            FOR each previous course IN the list corresponding to course in graph  
                SET current time TO dfs(previous course)  
                IF current time GREATER THAN maximum prerequisite time  
                    SET maximum prerequisite time TO current time  
                END IF  
            END FOR  
            
            SET the element at position course of memo TO maximum prerequisite time PLUS the element at position course MINUS one of time  
            RETURN the element at position course of memo  
        END FUNCTION  
        
        SET total minimum time TO zero  
        FOR course FROM one TO n  
            SET current time TO dfs(course)  
            IF current time GREATER THAN total minimum time  
                SET total minimum time TO current time  
            END IF  
        END FOR  
        
        RETURN total minimum time  
    END FUNCTION  
END CLASS
