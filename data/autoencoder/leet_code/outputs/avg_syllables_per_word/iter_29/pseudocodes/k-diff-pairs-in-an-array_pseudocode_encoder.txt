CLASS Solution  
    FUNCTION findPairs(nums k)  
        IF k LESS THAN zero  
            RETURN zero  
        END IF  
        SET num_counts TO empty map  
        FOR each num IN nums  
            SET current_count TO zero  
            IF num IS A KEY IN num_counts  
                SET current_count TO value associated with num in num_counts  
            END IF  
            SET value associated with num in num_counts TO current_count PLUS one  
        END FOR  
        SET unique_pairs TO zero  
        FOR each num IN keys of num_counts  
            IF k EQUALS zero  
                IF value associated with num in num_counts GREATER THAN one  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                SET target TO num PLUS k  
                IF target IS A KEY IN num_counts  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
