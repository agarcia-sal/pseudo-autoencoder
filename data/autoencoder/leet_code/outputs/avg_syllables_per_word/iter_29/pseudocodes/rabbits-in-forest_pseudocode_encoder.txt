CLASS Solution  
    FUNCTION numRabbits(answers)  
        SET answer_counts TO a collection that counts the number of occurrences of each element in answers  
        SET total_rabbits TO zero  
        FOR each answer WITH its corresponding count IN answer_counts  
            SET group_size TO answer PLUS one  
            SET groups TO the integer division of the sum of count PLUS answer BY group_size  
            INCREMENT total_rabbits BY the product of groups AND group_size  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
