CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MODULO_VALUE TO ten POWERED BY nine PLUS seven  
        SET dp_zero TO zero  
        SET dp_one TO zero  
        SET dp_two TO zero  
        
        FOR each number IN nums  
            IF number EQUALS zero  
                SET dp_zero TO dp_zero MULTIPLIED BY two PLUS one  
                SET dp_zero TO the remainder of dp_zero DIVIDED BY MODULO_VALUE  
            ELSE IF number EQUALS one  
                SET dp_one TO dp_one MULTIPLIED BY two PLUS dp_zero  
                SET dp_one TO the remainder of dp_one DIVIDED BY MODULO_VALUE  
            ELSE IF number EQUALS two  
                SET dp_two TO dp_two MULTIPLIED BY two PLUS dp_one  
                SET dp_two TO the remainder of dp_two DIVIDED BY MODULO_VALUE  
            END IF  
        END FOR  
        
        RETURN dp_two  
    END FUNCTION  
END CLASS
