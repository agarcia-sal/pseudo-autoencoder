CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET string_length TO the LENGTH OF s  
        SET palindrome_table TO a two dimensional collection WHERE each element AT position row AND column IS TRUE WITH dimensions string_length BY string_length  
        SET start_index TO zero  
        SET maximum_length TO one  
        FOR index_i FROM string_length MINUS two DOWN TO zero STEP minus one  
            FOR index_j FROM index_i PLUS one TO string_length MINUS one  
                SET the element AT position index_i AND index_j OF palindrome_table TO FALSE  
                IF the character AT position index_i OF s EQUALS the character AT position index_j OF s  
                    SET the element AT position index_i AND index_j OF palindrome_table TO the element AT position index_i PLUS one AND index_j MINUS one OF palindrome_table  
                    IF the element AT position index_i AND index_j OF palindrome_table IS TRUE AND maximum_length LESS THAN index_j MINUS index_i PLUS one  
                        SET start_index TO index_i  
                        SET maximum_length TO index_j MINUS index_i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the substring FROM position start_index TO position start_index PLUS maximum_length MINUS one OF s  
    END FUNCTION  
END CLASS
