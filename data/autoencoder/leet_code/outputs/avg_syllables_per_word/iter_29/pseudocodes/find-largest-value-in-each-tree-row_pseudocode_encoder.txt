CLASS TreeNode  
    FUNCTION __init__ (self val SET TO zero left SET TO none right SET TO none)  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH value AS element at position one of values  
    SET i TO one  
    SET queue TO A NEW empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO THE ELEMENT OBTAINED BY REMOVING FROM THE FRONT OF queue  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node left TO A NEW TreeNode WITH value AS element at position i of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND element at position i of values IS NOT none  
            SET node right TO A NEW TreeNode WITH value AS element at position i of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF value of p IS NOT EQUALS TO value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters value of left subtree of p and value of left subtree of q AND the result of is_same_tree with parameters value of right subtree of p and value of right subtree of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues(self root)  
        IF root IS none  
            RETURN an empty list  
        END IF  
        SET queue TO A NEW double ended queue initialized with root as its only element  
        SET largest_values TO AN EMPTY list  
        WHILE queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO NEGATIVE infinity  
            FOR each count FROM one TO level_size  
                SET node TO THE ELEMENT OBTAINED BY REMOVING FROM THE FRONT OF queue  
                IF max_value IS LESS THAN value of node  
                    SET max_value TO value of node  
                END IF  
                IF left subtree of node IS NOT none  
                    APPEND left subtree of node TO queue  
                END IF  
                IF right subtree of node IS NOT none  
                    APPEND right subtree of node TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
