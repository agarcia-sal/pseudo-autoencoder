CLASS Solution  
    FUNCTION kIncreasing(arr k)  
        FUNCTION longest_increasing_subsequence(subarr)  
            SET lis TO empty list  
            FOR each number IN subarr  
                IF lis IS EMPTY OR number GREATER THAN OR EQUAL TO element at position LENGTH OF lis MINUS one OF lis  
                    APPEND number TO lis  
                ELSE  
                    SET pos TO the position where number can be inserted on the right side to maintain sorted order in lis  
                    SET element at position pos OF lis TO number  
                END IF  
            END FOR  
            RETURN the LENGTH OF lis  
        END FUNCTION  
        
        SET total_operations TO zero  
        SET n TO the LENGTH OF arr  
        
        FOR start FROM zero TO k MINUS one  
            SET subarr TO empty list  
            FOR i FROM start TO n MINUS one STEP BY k  
                APPEND element at position i OF arr TO subarr  
            END FOR  
            
            SET lis_length TO longest_increasing_subsequence(subarr)  
            INCREMENT total_operations BY the LENGTH OF subarr MINUS lis_length  
        END FOR  
        
        RETURN total_operations  
    END FUNCTION  
END CLASS
