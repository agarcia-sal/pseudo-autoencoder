CLASS Solution  
    FUNCTION jobScheduling(startTime list_of_integers endTime list_of_integers profit list_of_integers) RETURNS integer  
        SET jobs TO the list obtained by sorting the list formed by pairing corresponding elements of startTime endTime and profit using the end time of each pair as the sorting key  
        SET end_times TO the list of the second element of each job in jobs  
        SET dp TO a list of zeros with the length of the total number of jobs PLUS one  
        FOR i FROM one TO the length of jobs  
            SET j TO the position returned by the binary search that finds the insertion point to the right in end_times for the first element of the job at position i MINUS one in jobs  
            SET dp at position i TO the maximum value between dp at position i MINUS one AND the sum of dp at position j AND the third element of the job at position i MINUS one in jobs  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
