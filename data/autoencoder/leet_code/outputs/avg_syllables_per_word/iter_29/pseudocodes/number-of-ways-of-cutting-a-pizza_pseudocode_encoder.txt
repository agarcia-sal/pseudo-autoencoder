CLASS Solution  
    FUNCTION ways(pizza, k)  
        FUNCTION dfs(i, j, k)  
            IF k EQUALS zero  
                IF number of apple slices in the subrectangle from position i TO position m AND from position j TO position n IS GREATER THAN zero  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            SET ans TO zero  
            FOR x FROM i PLUS one TO m MINUS one  
                IF number of apple slices in the subrectangle from position x TO position m AND from position j TO position n IS GREATER THAN zero  
                    INCREMENT ans BY dfs(x, j, k MINUS one)  
                END IF  
            END FOR  
            FOR y FROM j PLUS one TO n MINUS one  
                IF number of apple slices in the subrectangle from position i TO position m AND from position y TO position n IS GREATER THAN zero  
                    INCREMENT ans BY dfs(i, y, k MINUS one)  
                END IF  
            END FOR  
            RETURN ans MODULO mod  
        END FUNCTION  
        
        SET mod TO one billion PLUS seven  
        SET m TO LENGTH OF pizza  
        SET n TO LENGTH OF element at position zero of pizza  
        INITIALIZE s AS a two dimensional list of zeros WITH m PLUS one rows AND n PLUS one columns  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET s at position i and j TO s at position i MINUS one and j PLUS s at position i and j MINUS one MINUS s at position i MINUS one and j MINUS one PLUS one IF character at position j MINUS one of string at position i MINUS one of pizza EQUALS character A OTHERWISE zero  
            END FOR  
        END FOR  
        RETURN dfs(zero, zero, k MINUS one)  
    END FUNCTION  
END CLASS
