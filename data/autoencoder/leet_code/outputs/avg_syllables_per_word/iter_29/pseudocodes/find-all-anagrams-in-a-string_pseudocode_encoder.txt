CLASS Solution  
    FUNCTION findAnagrams(parameter s AS string_parameter parameter p AS string_parameter) RETURNS list_of_integers  
        SET length_of_s TO the LENGTH OF s  
        SET length_of_p TO the LENGTH OF p  
        
        IF length_of_p GREATER THAN length_of_s  
            RETURN an empty list  
        END IF  
        
        SET character_frequency_in_p TO the FREQUENCY COUNTER OF all characters IN p  
        SET character_frequency_in_s_window TO the FREQUENCY COUNTER OF characters IN the substring of s FROM position one TO position length_of_p  
        
        SET list_of_starting_indices_of_anagrams TO an empty list  
        
        IF character_frequency_in_s_window EQUALS character_frequency_in_p  
            APPEND zero TO list_of_starting_indices_of_anagrams  
        END IF  
        
        FOR index FROM length_of_p PLUS one TO length_of_s  
            SET current_character_to_add TO the character at position index of s  
            INCREMENT the count of current_character_to_add IN character_frequency_in_s_window BY one  
            
            SET character_to_remove FROM the character at position index MINUS length_of_p of s  
            DECREMENT the count of character_to_remove IN character_frequency_in_s_window BY one  
            
            IF the count of character_to_remove IN character_frequency_in_s_window EQUALS zero  
                REMOVE character_to_remove FROM character_frequency_in_s_window  
            END IF  
            
            IF character_frequency_in_s_window EQUALS character_frequency_in_p  
                SET starting_index TO index MINUS length_of_p PLUS one  
                APPEND starting_index TO list_of_starting_indices_of_anagrams  
            END IF  
        END FOR  
        
        RETURN list_of_starting_indices_of_anagrams  
    END FUNCTION  
END CLASS
