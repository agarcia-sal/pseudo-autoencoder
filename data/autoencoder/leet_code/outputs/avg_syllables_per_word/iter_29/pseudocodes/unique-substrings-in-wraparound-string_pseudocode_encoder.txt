CLASS Solution  
    FUNCTION findSubstringInWraproundString(string s)  
        SET max_length_dictionary TO a new dictionary where each lowercase letter from a to z maps TO zero  
        SET current_valid_substring_length TO zero  
        FOR index FROM zero TO the LENGTH OF string s MINUS one  
            IF index GREATER THAN zero AND (the character code of character at position index of string s MINUS the character code of character at position index MINUS one of string s EQUALS one OR character at position index of string s EQUALS character a AND character at position index MINUS one of string s EQUALS character z)  
                INCREMENT current_valid_substring_length BY one  
            ELSE  
                SET current_valid_substring_length TO one  
            END IF  
            SET the value associated with character at position index of string s in max_length_dictionary TO the GREATER OF the current value associated with character at position index of string s in max_length_dictionary AND current_valid_substring_length  
        END FOR  
        RETURN the SUM OF all values in max_length_dictionary  
    END FUNCTION  
END CLASS
