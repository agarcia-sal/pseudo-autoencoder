CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS ONLY OF DIGITS THEN  
                SET num TO the INTEGER CONVERSION OF block  
                RETURN whether num IS LESS THAN OR EQUAL TO two hundred and fifty five AND whether the STRING REPRESENTATION OF num EQUALS block  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF the LENGTH OF block IS GREATER THAN OR EQUAL TO one AND the LENGTH OF block IS LESS THAN OR EQUAL TO four THEN  
                TRY TO CONVERT block TO an INTEGER USING base sixteen  
                IF CONVERSION SUCCEEDS THEN  
                    RETURN true  
                ELSE  
                    RETURN false  
                END IF  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO the RESULT OF SPLITTING queryIP BY the CHARACTER dot  
        SET parts_ipv6 TO the RESULT OF SPLITTING queryIP BY the CHARACTER colon  

        IF the LENGTH OF parts_ipv4 EQUALS four AND for EACH part IN parts_ipv4 the VALUE OF is_valid_ipv4_block WITH part IS true THEN  
            RETURN the TEXT IPvFour  
        ELSE IF the LENGTH OF parts_ipv6 EQUALS eight AND for EACH part IN parts_ipv6 the VALUE OF is_valid_ipv6_block WITH part IS true THEN  
            RETURN the TEXT IPvSix  
        ELSE  
            RETURN the TEXT Neither  
        END IF  
    END FUNCTION  
END CLASS
