CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        SET radiant TO an empty double ended queue  
        SET dire TO an empty double ended queue  
        SET length_of_senate TO the LENGTH OF senate  
        
        FOR each index_position and senator_character IN the ENUMERATE of senate  
            IF senator_character EQUALS the character representing Radiant party  
                APPEND index_position TO radiant  
            ELSE  
                APPEND index_position TO dire  
            END IF  
        END FOR  
        
        WHILE radiant is NOT empty AND dire is NOT empty  
            REMOVE the element at the front FROM radiant and SET TO radiant_index  
            REMOVE the element at the front FROM dire and SET TO dire_index  
            
            IF radiant_index LESS THAN dire_index  
                APPEND radiant_index PLUS length_of_senate TO radiant  
            ELSE  
                APPEND dire_index PLUS length_of_senate TO dire  
            END IF  
        END WHILE  
        
        IF radiant is NOT empty  
            RETURN the string representing Radiant victory  
        ELSE  
            RETURN the string representing Dire victory  
        END IF  
    END FUNCTION  
END CLASS
