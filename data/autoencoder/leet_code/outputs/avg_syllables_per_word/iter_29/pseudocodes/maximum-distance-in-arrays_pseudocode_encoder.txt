CLASS Solution  
    FUNCTION maxDistance(arrays)  
        SET min_val TO element at position zero of element at position zero of arrays  
        SET max_val TO element at position zero of element at position position equal to the LENGTH OF element at position zero of arrays MINUS one of arrays  
        SET max_distance TO zero  
        
        FOR index FROM one TO the LENGTH OF arrays MINUS one  
            SET current_array TO element at position index of arrays  
            SET current_min TO element at position zero of current_array  
            SET current_max TO element at position position equal to the LENGTH OF current_array MINUS one of current_array  
            
            SET max_distance TO the maximum of the current max_distance AND the absolute value of current_max MINUS min_val AND the absolute value of max_val MINUS current_min  
            
            SET min_val TO the minimum of min_val AND current_min  
            SET max_val TO the maximum of max_val AND current_max  
        END FOR  
        
        RETURN max_distance  
    END FUNCTION  
END CLASS
