CLASS Solution  
    FUNCTION digitsCount(digit_value parameter D low_value parameter LOW high_value parameter HIGH)  
        FUNCTION count_digit(parameter N digit_value parameter DIGIT)  
            SET count TO zero  
            SET power_of_ten TO one  
            SET current_value TO N  
            WHILE current_value GREATER THAN zero  
                SET last_digit TO the remainder of current_value DIVIDED BY ten  
                SET current_value TO the quotient of current_value DIVIDED BY ten  
                SET full_cycles TO current_value  
                IF DIGIT EQUALS zero  
                    DECREMENT full_cycles BY one  
                END IF  
                INCREMENT count BY full_cycles MULTIPLIED BY power_of_ten  
                IF last_digit GREATER THAN DIGIT  
                    INCREMENT count BY power_of_ten  
                ELSE IF last_digit EQUALS DIGIT  
                    INCREMENT count BY the remainder of N DIVIDED BY power_of_ten PLUS one  
                END ELSE IF  
                MULTIPLY power_of_ten BY ten  
            END WHILE  
            RETURN count  
        END FUNCTION  

        RETURN the result of count_digit with arguments HIGH and D MINUS the result of count_digit with arguments LOW MINUS one and D  
    END FUNCTION  
END CLASS
