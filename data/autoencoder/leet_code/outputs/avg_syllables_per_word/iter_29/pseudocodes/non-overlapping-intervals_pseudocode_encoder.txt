CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF the LENGTH OF intervals EQUALS zero THEN  
            RETURN zero  
        END IF  
        SORT intervals BY the second element at position one of each sublist in intervals  
        SET end TO the second element at position one of the first element of intervals  
        SET count TO zero  
        FOR index FROM one TO the LENGTH OF intervals MINUS one  
            IF the first element at position zero of the element at position index of intervals GREATER THAN OR EQUAL TO end THEN  
                SET end TO the second element at position one of the element at position index of intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
