CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET xor_all TO zero  
        FOR each num IN nums  
            SET xor_all TO xor_all XOR num  
        END FOR  
        SET diff_bit TO the result of performing BITWISE AND between xor_all AND the NEGATION of xor_all  
        SET num1 TO zero  
        SET num2 TO zero  
        FOR each num IN nums  
            IF the result of performing BITWISE AND between num AND diff_bit IS GREATER THAN zero  
                SET num1 TO num1 XOR num  
            ELSE  
                SET num2 TO num2 XOR num  
            END IF  
        END FOR  
        RETURN a collection containing num1 AND num2  
    END FUNCTION  
END CLASS
