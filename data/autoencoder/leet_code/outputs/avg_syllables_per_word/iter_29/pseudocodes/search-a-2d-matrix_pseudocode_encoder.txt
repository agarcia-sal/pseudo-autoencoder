CLASS Solution  
    FUNCTION searchMatrix(matrix target)  
        IF matrix EQUALS nothing OR element at position zero of matrix EQUALS nothing  
            RETURN false  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER PART OF left PLUS right DIVIDED BY two  
            SET mid_value TO element at position mid DIVIDED BY n of matrix AT position mid MODULO n  
            
            IF mid_value EQUALS target  
                RETURN true  
            ELSE IF mid_value LESS THAN target  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END ELSE IF  
        END WHILE  
        
        RETURN false  
    END FUNCTION  
END CLASS
