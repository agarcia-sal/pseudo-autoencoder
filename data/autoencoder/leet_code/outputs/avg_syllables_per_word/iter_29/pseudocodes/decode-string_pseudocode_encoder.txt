CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty collection  
        SET a TO empty string representation  
        SET d TO empty string representation  
        FOR each character IN s  
            IF character IS a digit character  
                SET d TO the concatenation of d AND character  
            ELSE IF character IS an alphabetic character  
                SET a TO the concatenation of a AND character  
            ELSE IF character EQUALS the left bracket character  
                APPEND the tuple consisting of a AND the value of d converted to a number TO stack  
                SET a TO empty string representation  
                SET d TO empty string representation  
            ELSE IF character EQUALS the right bracket character  
                SET p AND n TO the last element REMOVED from stack  
                SET a TO the concatenation of p AND the repetition of a by n times  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
