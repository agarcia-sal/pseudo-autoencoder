CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self DOT val TO val  
        SET self DOT left TO left  
        SET self DOT right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the list values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node DOT left TO a new TreeNode with the element at position i of values  
            APPEND node DOT left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node DOT right TO a new TreeNode with the element at position i of values  
            APPEND node DOT right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value of p IS NOT EQUAL TO the value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p DOT left AND q DOT left AND the result of is_same_tree with p DOT right AND q DOT right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION lowestCommonAncestor(self, root, p, q)  
        IF root IS None OR root IS p OR root IS q  
            RETURN root  
        END IF  
        SET left TO the result of calling lowestCommonAncestor on self with root DOT left AND p AND q  
        SET right TO the result of calling lowestCommonAncestor on self with root DOT right AND p AND q  
        IF left IS NOT None AND right IS NOT None  
            RETURN root  
        END IF  
        IF left IS NOT None  
            RETURN left  
        ELSE  
            RETURN right  
        END IF  
    END FUNCTION  
END CLASS
