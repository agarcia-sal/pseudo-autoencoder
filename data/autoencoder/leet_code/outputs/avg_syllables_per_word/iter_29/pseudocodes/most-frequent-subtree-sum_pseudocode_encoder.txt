CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, left WITH DEFAULT none, right WITH DEFAULT none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the list values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode created WITH the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element removed FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode created WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode created WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value field of p IS NOT EQUALS TO the value field of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree WITH the left field of p AND the left field of q AND the result of is_same_tree WITH the right field of p AND the right field of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS none  
                RETURN zero  
            END IF  
            SET current_sum TO the value field of node PLUS the result of subtree_sum WITH the left field of node PLUS the result of subtree_sum WITH the right field of node  
            INCREMENT the frequency count associated WITH current_sum BY one  
            RETURN current_sum  
        END FUNCTION  
        
        SET count TO a new mapping with default value zero for any key  
        CALL subtree_sum WITH root  
        
        SET max_freq TO the greatest frequency value present IN count OR zero IF count IS empty  
        
        SET result TO a list of all keys IN count WHERE the frequency value IS EQUALS TO max_freq  
        
        RETURN result  
    END FUNCTION  
END CLASS
