CLASS Solution  
    FUNCTION minMovesToMakePalindrome(string_s)  
        SET list_s TO list converted from string_s  
        SET length_n TO the LENGTH OF list_s  
        SET count_moves TO zero  
          
        WHILE the LENGTH OF list_s GREATER THAN one  
            SET match_found TO false  
            FOR index_i FROM the LENGTH OF list_s MINUS one DOWN TO one DECREMENTING BY one  
                IF element at position index_i of list_s EQUALS element at position zero of list_s  
                    FOR index_j FROM index_i TO the LENGTH OF list_s MINUS two INCREMENTING BY one  
                        SET temporary_variable TO element at position index_j PLUS one of list_s  
                        SET element at position index_j plus one of list_s TO element at position index_j of list_s  
                        SET element at position index_j of list_s TO temporary_variable  
                        INCREMENT count_moves BY one  
                    END FOR  
                    REMOVE element at position zero FROM list_s  
                    REMOVE element at the last position FROM list_s  
                    SET match_found TO true  
                    BREAK out of FOR loop  
                END IF  
            END FOR  
            IF match_found EQUALS false  
                REMOVE element at position zero FROM list_s  
            END IF  
        END WHILE  
        RETURN count_moves  
    END FUNCTION  
END CLASS
