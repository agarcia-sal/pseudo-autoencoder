CLASS Solution  
    FUNCTION getFactors(n)  
        FUNCTION backtrack(start, target)  
            FOR i FROM start TO the INTEGER EQUIVALENT OF the SQUARE ROOT OF target INCREMENTED BY one  
                IF the REMAINDER WHEN target IS DIVIDED BY i EQUALS zero  
                    APPEND the concatenation OF path WITH the list CONSISTING OF i AND target DIVIDED BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH i AND target DIVIDED BY i  
                    REMOVE the LAST ELEMENT FROM path  
                END IF  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        SET path TO an empty list  
        CALL backtrack WITH two AND n  
        RETURN result  
    END FUNCTION  
END CLASS
