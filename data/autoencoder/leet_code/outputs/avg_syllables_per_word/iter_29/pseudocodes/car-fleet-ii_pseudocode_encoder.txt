CLASS Solution  
    FUNCTION getCollisionTimes(cars)  
        SET total_number_of_cars TO the LENGTH OF cars  
        SET collision_times TO a list containing total_number_of_cars elements all set TO negative one point zero  
        SET stack_of_car_indices TO an empty list  
        
        FOR index FROM total_number_of_cars MINUS one TO zero stepping by negative one  
            SET current_position TO the element at position zero of the element at position index of cars  
            SET current_speed TO the element at position one of the element at position index of cars  
            
            WHILE the LENGTH OF stack_of_car_indices IS GREATER THAN zero  
                SET next_car_index TO the element at the last position of stack_of_car_indices  
                SET next_car_position TO the element at position zero of the element at position next_car_index of cars  
                SET next_car_speed TO the element at position one of the element at position next_car_index of cars  
                
                IF current_speed IS LESS THAN OR EQUAL TO next_car_speed  
                    REMOVE the last element FROM stack_of_car_indices  
                    CONTINUE TO the next iteration of WHILE  
                END IF  
                
                SET potential_collision_time TO next_car_position MINUS current_position DIVIDED BY current_speed MINUS next_car_speed  
                
                IF the element at position next_car_index of collision_times IS GREATER THAN zero AND potential_collision_time IS GREATER THAN OR EQUAL TO the element at position next_car_index of collision_times  
                    REMOVE the last element FROM stack_of_car_indices  
                    CONTINUE TO the next iteration of WHILE  
                END IF  
                
                SET the element at position index of collision_times TO potential_collision_time  
                BREAK the WHILE loop  
            END WHILE  
            
            APPEND index TO stack_of_car_indices  
        END FOR  
        
        RETURN collision_times  
    END FUNCTION  
END CLASS
