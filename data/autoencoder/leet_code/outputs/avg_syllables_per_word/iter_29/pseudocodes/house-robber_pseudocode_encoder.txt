CLASS Solution  
    FUNCTION rob(nums)  
        IF the LIST nums IS empty  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position zero of nums  
        END IF  
        CREATE a LIST dp WITH the LENGTH OF nums FILLED WITH zero  
        SET the element at position zero of dp TO the element at position zero of nums  
        SET the element at position one of dp TO the GREATER VALUE BETWEEN the element at position zero of nums AND the element at position one of nums  
        FOR index FROM two TO the LENGTH OF nums MINUS one  
            SET the element at position index of dp TO the GREATER VALUE BETWEEN the element at position index MINUS one of dp AND the SUM OF the element at position index MINUS two of dp AND the element at position index of nums  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
