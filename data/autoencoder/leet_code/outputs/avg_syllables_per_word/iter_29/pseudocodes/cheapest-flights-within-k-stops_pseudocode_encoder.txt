CLASS Solution  
    FUNCTION findCheapestPrice(n number_of_cities flights list_of_lists_of_integers src source_city dst destination_city k maximum_number_of_stops RETURNS integer  
        SET graph TO an empty collection mapping each integer from zero TO n MINUS one TO an empty list  
        FOR each flight IN flights  
            SET u TO element at position zero of flight  
            SET v TO element at position one of flight  
            SET w TO element at position two of flight  
            APPEND the tuple of v and w TO the list mapped by u in graph  
        END FOR  
  
        SET min_heap TO a list containing one tuple of zero cost MINUS one stops AND source_city  
        SET min_cost TO a mapping from source_city TO minus one  
  
        WHILE min_heap is not empty  
            REMOVE and ASSIGN the smallest tuple FROM min_heap TO cost stops city  
  
            IF city EQUALS destination_city  
                RETURN cost  
            END IF  
  
            IF stops GREATER THAN OR EQUAL TO k  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
  
            FOR each neighbor price IN the list mapped by city in graph  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  
  
                IF neighbor NOT IN min_cost OR next_stops LESS THAN the value mapped by neighbor in min_cost OR next_cost LESS THAN the cost element of the first tuple in min_heap  
                    SET the value mapped by neighbor in min_cost TO next_stops  
                    ADD the tuple of next_cost next_stops neighbor INTO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
  
        RETURN minus one  
    END FUNCTION  
END CLASS
