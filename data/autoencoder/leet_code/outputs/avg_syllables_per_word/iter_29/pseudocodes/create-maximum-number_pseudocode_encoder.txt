CLASS Solution  
    FUNCTION maxNumber(nums1 list_of_integers nums2 list_of_integers k integer) RETURNS list_of_integers  
        FUNCTION maxSingleArray(nums list_of_integers k integer) RETURNS list_of_integers  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO empty list  
            FOR each num IN nums  
                WHILE drop IS GREATER THAN zero AND stack IS NOT empty AND the LAST element of stack IS LESS THAN num  
                    REMOVE the LAST element FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the FIRST k elements OF stack  
        END FUNCTION  
        
        FUNCTION merge(arr1 list_of_integers arr2 list_of_integers) RETURNS list_of_integers  
            SET merged_list TO empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR count FROM one TO total_length  
                IF arr1 IS GREATER THAN arr2 starting FROM their CURRENT POSITIONS when COMPARED lexicographically  
                    APPEND the FIRST element OF arr1 TO merged_list  
                    REMOVE the FIRST element FROM arr1  
                ELSE  
                    APPEND the FIRST element OF arr2 TO merged_list  
                    REMOVE the FIRST element FROM arr2  
                END IF  
            END FOR  
            RETURN merged_list  
        END FUNCTION  
        
        SET max_number TO empty list  
        FOR i FROM zero TO k INCLUSIVE  
            IF i IS LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i IS LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO merge(maxSingleArray(nums1 i) maxSingleArray(nums2 k MINUS i))  
                IF max_number IS LESS THAN candidate lexicographically  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
