CLASS Solution  
    FUNCTION subarraySum(list_of_numbers WITH ELEMENTS OF INTEGER parameter nums INTEGER parameter k) RETURNS INTEGER  
        SET dictionary_of_cumulative_sum_frequencies TO dictionary WITH key zero AND value one  
        SET accumulated_sum TO zero  
        SET count_of_subarrays TO zero  
        FOR each number IN list_of_numbers WITH ELEMENTS OF INTEGER nums  
            INCREMENT accumulated_sum BY number  
            IF the accumulated_sum MINUS k EXISTS AS A KEY IN dictionary_of_cumulative_sum_frequencies  
                INCREMENT count_of_subarrays BY the value AT KEY of accumulated_sum MINUS k IN dictionary_of_cumulative_sum_frequencies  
            END IF  
            IF accumulated_sum EXISTS AS A KEY IN dictionary_of_cumulative_sum_frequencies  
                INCREMENT the value AT KEY accumulated_sum IN dictionary_of_cumulative_sum_frequencies BY one  
            ELSE  
                SET the value AT KEY accumulated_sum IN dictionary_of_cumulative_sum_frequencies TO one  
            END ELSE  
        END FOR  
        RETURN count_of_subarrays  
    END FUNCTION  
END CLASS
