CLASS Solution  
    FUNCTION distanceToCycle(n edges)  
        SET graph TO a new default dictionary with list as default value  
        FOR each pair of values u and v IN edges  
            APPEND value v TO the list at key u IN graph  
            APPEND value u TO the list at key v IN graph  
        END FOR  
        
        FUNCTION dfs(node parent path visited)  
            IF node IS IN visited THEN  
                SET cycle_start TO the position of node IN path  
                RETURN the sublist of path from position cycle_start TO the end  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN the list at key node IN graph  
                IF neighbor IS NOT EQUAL TO parent THEN  
                    SET cycle TO dfs(neighbor node path visited)  
                    IF cycle is not None THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE the last element FROM path  
            RETURN None  
        END FUNCTION  
        
        SET visited TO an empty set  
        SET cycle TO None  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                SET cycle TO dfs(node negative one empty list visited)  
                IF cycle is not None THEN  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
        
        SET distance TO a list of zero repeated n times  
        SET queue TO a new deque initialized with cycle  
        SET visited TO a set initialized with cycle  
        
        SET level TO zero  
        WHILE queue is not empty  
            SET current_length TO the length of queue  
            FOR count FROM one TO current_length  
                SET node TO the removal of the first element from queue  
                SET the element at position node of distance TO level  
                FOR each neighbor IN the list at key node IN graph  
                    IF neighbor IS NOT IN visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  
        
        RETURN distance  
    END FUNCTION  
END CLASS
