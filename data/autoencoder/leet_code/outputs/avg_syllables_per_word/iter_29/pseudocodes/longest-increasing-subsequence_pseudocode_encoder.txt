CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums EQUALS empty collection  
            RETURN zero  
        END IF  
        SET dp TO empty collection  
        FOR each number IN nums  
            SET idx TO the result of finding the leftmost position to insert number into dp maintaining sorted order  
            IF idx EQUALS the LENGTH OF dp  
                APPEND number TO dp  
            ELSE  
                SET the element at position idx of dp TO number  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
