CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix, x, y)  
            SET directions TO the collection of coordinate pairs consisting of negative one zero positive one zero zero negative one zero one zero zero  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET element at position nx of matrix element at position ny TO one MINUS element at position nx of matrix element at position ny  
                END IF  
            END FOR  
        END FUNCTION  

        FUNCTION matrix_to_tuple(matrix)  
            RETURN the immutable collection created by converting each row of matrix as an immutable collection  
        END FUNCTION  

        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET target TO the immutable collection consisting of m elements each being an immutable collection consisting of n zeros  
        SET initial_state TO matrix_to_tuple(mat)  
        SET queue TO a collection containing the tuple consisting of initial_state and zero  
        SET visited TO an empty set  
        ADD initial_state TO visited  

        WHILE queue is not empty  
            REMOVE the first element of queue and SET it to a tuple current_state and steps  
            IF current_state EQUALS target  
                RETURN steps  
            END IF  

            SET current_matrix TO a new collection where each element is a list converted from each element of current_state  

            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO a deep copy of current_matrix  
                    CALL flip with arguments next_matrix i j  
                    SET next_state TO matrix_to_tuple(next_matrix)  
                    IF next_state NOT IN visited  
                        ADD next_state TO visited  
                        APPEND the tuple consisting of next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
