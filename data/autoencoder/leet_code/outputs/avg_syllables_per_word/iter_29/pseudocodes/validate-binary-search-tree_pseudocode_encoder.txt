CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value element at position one of values with left set to None and right set to None  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO a new TreeNode with value element at position i of values with left set to None and right set to None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO a new TreeNode with value element at position i of values with left set to None and right set to None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value property of p IS NOT EQUAL TO the value property of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with parameters the left child property of p and the left child property of q AND the result of calling is_same_tree with parameters the right child property of p and the right child property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low, high)  
            IF node IS None  
                RETURN True  
            END IF  
            IF the value property of node IS LESS THAN OR EQUAL TO low OR the value property of node IS GREATER THAN OR EQUAL TO high  
                RETURN False  
            END IF  
            RETURN the result of calling validate with parameters the left child property of node, low, and the value property of node AND the result of calling validate with parameters the right child property of node, the value property of node, and high  
        END FUNCTION  
        RETURN the result of calling validate with parameters root, negative infinity, positive infinity  
    END FUNCTION  
END CLASS
