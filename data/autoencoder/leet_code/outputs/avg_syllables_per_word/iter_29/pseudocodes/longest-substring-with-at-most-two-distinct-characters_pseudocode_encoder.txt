CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        SET character count TO empty dictionary with default value zero  
        SET left TO zero  
        SET maximum length TO zero  
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the count of character at position right of s IN character count BY one  
            WHILE the NUMBER OF keys IN character count IS GREATER THAN two  
                DECREMENT the count of character at position left of s IN character count BY one  
                IF the count of character at position left of s IN character count EQUALS zero  
                    REMOVE the character at position left of s FROM character count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            SET maximum length TO THE GREATER VALUE BETWEEN maximum length AND the RESULT OF right MINUS left PLUS one  
        END FOR  
        RETURN maximum length  
    END FUNCTION  
END CLASS
