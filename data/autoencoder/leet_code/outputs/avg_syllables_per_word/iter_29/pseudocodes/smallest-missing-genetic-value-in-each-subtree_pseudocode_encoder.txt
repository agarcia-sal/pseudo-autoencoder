CLASS Solution  
    FUNCTION smallestMissingValueSubtree(parents, nums)  
        SET n TO the LENGTH OF parents  
        SET tree TO a new mapping from keys to lists with default empty list values  
        
        FOR index FROM one TO n MINUS one  
            APPEND index TO the list at key element at position index of parents IN tree  
        END FOR  
        
        SET node_with_one TO negative one  
        FOR index FROM zero TO n MINUS one  
            IF element at position index of nums EQUALS one  
                SET node_with_one TO index  
                BREAK the loop  
            END IF  
        END FOR  
        
        IF node_with_one EQUALS negative one  
            RETURN a list of length n where every element is one  
        END IF  
        
        SET result TO a list of length n where every element is one  
        
        FUNCTION dfs(node)  
            ADD node TO the visited collection  
            SET genetic_values TO a new collection containing the element at position node of nums  
            FOR each child IN the list at key node IN tree  
                IF child NOT IN visited  
                    SET returned_values TO dfs(child)  
                    ADD every element of returned_values TO genetic_values  
                END IF  
            END FOR  
            SET smallest_missing TO one  
            WHILE smallest_missing IS AN ELEMENT OF genetic_values  
                INCREMENT smallest_missing BY one  
            END WHILE  
            SET element at position node of result TO smallest_missing  
            RETURN genetic_values  
        END FUNCTION  
        
        SET visited TO an empty collection  
        
        SET path TO an empty list  
        SET current TO node_with_one  
        WHILE current NOT EQUALS negative one  
            APPEND current TO path  
            SET current TO element at position current of parents  
        END WHILE  
        
        SET path_set TO a new collection containing every element of path  
        FOR each node IN the reversed sequence of path  
            IF node IN visited  
                CONTINUE to the next iteration  
            END IF  
            CALL dfs with node  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
