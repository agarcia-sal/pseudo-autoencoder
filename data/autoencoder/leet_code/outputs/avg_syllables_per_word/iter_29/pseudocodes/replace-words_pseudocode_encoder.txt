CLASS Solution  
    FUNCTION replaceWords(dictionary list_of_strings sentence string)  
        SORT dictionary BY the LENGTH OF each element IN ascending order  
        SET words TO the list OF elements obtained BY splitting sentence INTO words BY spaces  
        
        FUNCTION replace(word string)  
            FOR each root IN dictionary  
                IF the word STARTS WITH the root  
                    RETURN root  
                END IF  
            END FOR  
            RETURN word  
        END FUNCTION  
        
        SET replaced_words TO a new list  
        FOR each word IN words  
            APPEND the result of replace(word) TO replaced_words  
        END FOR  
        
        SET result_sentence TO the concatenation of all elements IN replaced_words SEPARATED BY spaces  
        RETURN result_sentence  
    END FUNCTION  
END CLASS
