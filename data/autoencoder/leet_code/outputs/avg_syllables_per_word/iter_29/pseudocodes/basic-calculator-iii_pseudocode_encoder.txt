CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s_list)  
            SET stack TO empty list  
            SET number TO zero  
            SET sign TO the character plus  
            WHILE the LENGTH OF s_list GREATER THAN zero  
                SET character TO the element at position zero of s_list  
                REMOVE the element at position zero FROM s_list  
                IF the character is a digit  
                    SET number TO number MULTIPLIED BY ten PLUS the numeric value of character  
                END IF  
                IF the character EQUALS the character open parenthesis  
                    SET number TO the result of calling helper with s_list  
                END IF  
                IF the character is one of the characters plus minus multiply divide open parenthesis close parenthesis OR the LENGTH OF s_list EQUALS zero  
                    IF sign EQUALS the character plus  
                        APPEND number TO stack  
                    ELSE IF sign EQUALS the character minus  
                        APPEND the negative of number TO stack  
                    ELSE IF sign EQUALS the character multiply  
                        SET the last element of stack TO the last element of stack MULTIPLIED BY number  
                    ELSE IF sign EQUALS the character divide  
                        SET the last element of stack TO the integer part of the last element of stack DIVIDED BY number truncated towards zero  
                    END IF  
                    SET sign TO the character  
                    SET number TO zero  
                END IF  
                IF the character EQUALS the character close parenthesis  
                    BREAK the loop  
                END IF  
            END WHILE  
            RETURN the sum of all elements in stack  
        END FUNCTION  
        SET s TO the list of characters from the input string s  
        RETURN the result of calling helper with s  
    END FUNCTION  
END CLASS
