CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET length_of_dominoes TO the LENGTH OF dominoes  
        SET list_of_forces TO a new list of zeros with length length_of_dominoes  

        SET force_magnitude TO zero  
        FOR index FROM zero TO length_of_dominoes MINUS one  
            IF the character at position index of dominoes EQUALS the character R  
                SET force_magnitude TO length_of_dominoes  
            ELSE IF the character at position index of dominoes EQUALS the character L  
                SET force_magnitude TO zero  
            ELSE  
                SET force_magnitude TO the GREATER OF force_magnitude MINUS one AND zero  
            END IF  
            INCREMENT the element at position index of list_of_forces BY force_magnitude  
        END FOR  

        SET force_magnitude TO zero  
        FOR index FROM length_of_dominoes MINUS one DOWN TO zero  
            IF the character at position index of dominoes EQUALS the character L  
                SET force_magnitude TO length_of_dominoes  
            ELSE IF the character at position index of dominoes EQUALS the character R  
                SET force_magnitude TO zero  
            ELSE  
                SET force_magnitude TO the GREATER OF force_magnitude MINUS one AND zero  
            END IF  
            DECREMENT the element at position index of list_of_forces BY force_magnitude  
        END FOR  

        SET list_of_results TO an empty list  
        FOR each force_value IN list_of_forces  
            IF force_value GREATER THAN zero  
                APPEND the character R TO list_of_results  
            ELSE IF force_value LESS THAN zero  
                APPEND the character L TO list_of_results  
            ELSE  
                APPEND the character representing a dot TO list_of_results  
            END IF  
        END FOR  

        RETURN the concatenation of all elements in list_of_results as a string  
    END FUNCTION  
END CLASS
