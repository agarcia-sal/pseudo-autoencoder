CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word representing zero  
        END IF  
        
        SET list below_twenty TO the list of words representing numbers from zero up to nineteen in order starting with empty string for zero  
        SET list tens TO the list of words representing multiples of ten from twenty to ninety in order starting with empty strings for zero and ten  
        SET list thousands TO the list of scale words including empty string for units thousand million billion in order  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            END IF  
            IF n LESS THAN twenty  
                RETURN element at position n of below_twenty FOLLOWED BY a space  
            END IF  
            IF n LESS THAN one hundred  
                SET quotient TO n DIVIDED BY ten  
                SET remainder TO n MODULO ten  
                RETURN element at position quotient of tens FOLLOWED BY a space FOLLOWED BY the result of calling helper WITH remainder  
            END IF  
            SET quotient TO n DIVIDED BY one hundred  
            SET remainder TO n MODULO one hundred  
            RETURN element at position quotient of below_twenty FOLLOWED BY the word Hundred FOLLOWED BY a space FOLLOWED BY the result of calling helper WITH remainder  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero  
            SET remainder TO num MODULO one thousand  
            IF remainder NOT EQUALS zero  
                SET result TO the result of calling helper WITH remainder FOLLOWED BY element at position i of thousands FOLLOWED BY a space FOLLOWED BY result  
            END IF  
            SET num TO num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN result WITH all leading and trailing spaces removed  
    END FUNCTION  
END CLASS
