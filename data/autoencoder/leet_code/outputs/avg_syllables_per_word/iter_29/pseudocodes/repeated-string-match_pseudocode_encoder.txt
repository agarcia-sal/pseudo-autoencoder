CLASS Solution  
    FUNCTION repeatedStringMatch(a b)  
        SET minimum_repeats TO the smallest integer greater than or equal to the LENGTH OF b DIVIDED BY the LENGTH OF a  
        IF the string b IS CONTAINED WITHIN the string formed by concatenating the string a repeated minimum_repeats times  
            RETURN minimum_repeats  
        END IF  
        IF the string b IS CONTAINED WITHIN the string formed by concatenating the string a repeated the result of minimum_repeats PLUS one times  
            RETURN the result of minimum_repeats PLUS one  
        END IF  
        RETURN the integer negative one  
    END FUNCTION  
END CLASS
