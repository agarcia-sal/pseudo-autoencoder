CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each element l IN left  
                FOR each element r IN right  
                    IF operator EQUALS the addition symbol THEN  
                        APPEND the sum of l AND r TO results  
                    ELSE IF operator EQUALS the subtraction symbol THEN  
                        APPEND the difference of l MINUS r TO results  
                    ELSE IF operator EQUALS the multiplication symbol THEN  
                        APPEND the product of l MULTIPLIED BY r TO results  
                    END ELSE IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        FUNCTION helper(sub_expr)  
            IF sub_expr CONSISTS ONLY OF DIGITS THEN  
                RETURN a list containing the integer conversion OF sub_expr  
            END IF  
            
            SET results TO empty list  
            FOR index FROM zero TO the LENGTH OF sub_expr MINUS one  
                IF the character at position index OF sub_expr IS one of the symbols for addition subtraction OR multiplication THEN  
                    SET left_results TO the result of calling helper WITH the substring FROM position zero TO position index MINUS one OF sub_expr  
                    SET right_results TO the result of calling helper WITH the substring FROM position index PLUS one TO the END OF sub_expr  
                    APPEND all elements FROM the result of calling compute WITH left_results right_results AND the character at position index OF sub_expr TO results  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        RETURN the result of calling helper WITH expression  
    END FUNCTION  
END CLASS
