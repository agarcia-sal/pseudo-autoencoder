CLASS Solution  
    FUNCTION minRefuelStops(target_parameter startFuel_parameter stations_parameter)  
        SET max_heap TO an empty list  
        SET current_fuel TO startFuel_parameter  
        SET stops TO zero  
        APPEND a list containing target_parameter and zero TO stations_parameter  
        
        FOR each position_variable and fuel_variable IN stations_parameter  
            WHILE max_heap is not empty AND current_fuel LESS THAN position_variable  
                DECREMENT current_fuel BY the element obtained by removing the smallest element from max_heap  
                INCREMENT stops BY one  
            END WHILE  
            
            IF current_fuel LESS THAN position_variable  
                RETURN negative one  
            END IF  
            
            ADD negative fuel_variable TO max_heap maintaining heap property  
        END FOR  
        
        RETURN stops  
    END FUNCTION  
END CLASS
