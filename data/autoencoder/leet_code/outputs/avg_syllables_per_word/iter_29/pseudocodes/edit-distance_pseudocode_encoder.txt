CLASS Solution  
    FUNCTION minDistance(word1 word1_parameter word2 word2_parameter)  
        SET m TO the LENGTH OF word1_parameter  
        SET n TO the LENGTH OF word2_parameter  
  
        CREATE a two dimensional list dp with dimensions m PLUS one BY n PLUS one FILLED WITH zeroes  
  
        FOR i FROM zero TO m  
            SET element at position i AT position zero OF dp TO i  
        END FOR  
  
        FOR j FROM zero TO n  
            SET element at position zero AT position j OF dp TO j  
        END FOR  
  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one OF word1_parameter EQUALS character at position j MINUS one OF word2_parameter  
                    SET element at position i AT position j OF dp TO element at position i MINUS one AT position j MINUS one OF dp  
                ELSE  
                    SET element at position i AT position j OF dp TO one PLUS the MINIMUM of element at position i MINUS one AT position j OF dp AND element at position i AT position j MINUS one OF dp AND element at position i MINUS one AT position j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
  
        RETURN element at position m AT position n OF dp  
    END FUNCTION  
END CLASS
