CLASS Solution  
    FUNCTION palindromePartition(s k)  
        FUNCTION min_changes_to_palindrome(sub)  
            SET i TO zero  
            SET j TO the LENGTH OF sub MINUS one  
            SET changes TO zero  
            WHILE i LESS THAN j  
                IF character at position i of sub NOT EQUALS character at position j of sub  
                    INCREMENT changes BY one  
                END IF  
                INCREMENT i BY one  
                DECREMENT j BY one  
            END WHILE  
            RETURN changes  
        END FUNCTION  

        SET n TO the LENGTH OF s  
        INITIALIZE dp AS a two dimensional collection with n PLUS one rows AND k PLUS one columns FILLED WITH infinite value  
        SET element at row zero column zero of dp TO zero  

        FOR i FROM one TO n INCLUSIVE  
            FOR j FROM one TO the MINIMUM OF k AND i INCLUSIVE  
                FOR start FROM zero TO i MINUS one INCLUSIVE  
                    SET changes TO CALL min_changes_to_palindrome WITH substring from position start TO position i MINUS one of s  
                    SET element at row i column j of dp TO the MINIMUM OF element at row i column j of dp AND element at row start column j MINUS one of dp PLUS changes  
                END FOR  
            END FOR  
        END FOR  

        RETURN element at row n column k of dp  
    END FUNCTION  
END CLASS
