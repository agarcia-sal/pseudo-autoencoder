CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a new collection mapping zero TO negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        FOR each position i AND character char IN s  
            SET digit TO the integer value representation of character at position i of s  
            SET prefix_xor TO prefix_xor EXCLUSIVE OR with one shifted LEFT BY digit positions  
            IF prefix_xor IS A KEY IN prefix_map  
                SET max_length TO the greater value BETWEEN max_length AND position i MINUS the value corresponding to prefix_xor in prefix_map  
            END IF  
            FOR variable j FROM zero TO nine  
                SET flipped TO prefix_xor EXCLUSIVE OR with one shifted LEFT BY j positions  
                IF flipped IS A KEY IN prefix_map  
                    SET max_length TO the greater value BETWEEN max_length AND position i MINUS the value corresponding to flipped in prefix_map  
                END IF  
            END FOR  
            IF prefix_xor IS NOT A KEY IN prefix_map  
                SET the value corresponding to prefix_xor in prefix_map TO position i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
