CLASS Solution  
    FUNCTION rangeBitwiseAnd(left_parameter WITH TYPE integer_parameter, right_parameter WITH TYPE integer_parameter) RETURNS integer_parameter  
        SET shift_variable TO zero  
        WHILE left_parameter IS LESS THAN right_parameter  
            SET left_parameter TO the result of integer division of left_parameter BY two  
            SET right_parameter TO the result of integer division of right_parameter BY two  
            INCREMENT shift_variable BY one  
        END WHILE  
        RETURN the result of multiplication of left_parameter BY the value of two POWERED BY shift_variable  
    END FUNCTION  
END CLASS
