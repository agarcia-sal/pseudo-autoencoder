CLASS Solution  
    FUNCTION isTransformable(parameter s AS string, parameter t AS string) RETURNS boolean  
        CREATE a mapping from digits TO lists of integer positions NAMED pos with a default of an empty list  
        FOR each index i AND character c IN sequence s  
            LET digit_value BE the integer value represented by character c  
            PUSH the index i INTO the list of positions for digit_value within pos such that the smallest index can be retrieved easily  
        END FOR  
        FOR each index i AND character c IN sequence t  
            LET digit_value BE the integer value represented by character c  
            IF the list of positions for digit_value in pos IS empty THEN  
                RETURN false  
            END IF  
            FOR each integer smaller_digit FROM zero UP TO but NOT INCLUDING digit_value  
                IF the list of positions for smaller_digit in pos IS NOT empty AND the first element of the list of positions for smaller_digit IS LESS THAN the first element of the list of positions for digit_value THEN  
                    RETURN false  
                END IF  
            END FOR  
            REMOVE the smallest position from the list of positions for digit_value within pos  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
