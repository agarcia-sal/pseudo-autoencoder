CLASS Solution  
    FUNCTION combinationSum4(nums list_of_integers target integer) RETURNS integer  
        SET dp_list TO a list filled with zero repeated the LENGTH OF nums PLUS one times  
        SET the element at position zero of dp_list TO one  
        FOR each amount FROM one TO target  
            FOR each number IN nums  
                IF amount GREATER THAN OR EQUAL TO number THEN  
                    INCREMENT the element at position amount of dp_list BY the element at position amount MINUS number of dp_list  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position target of dp_list  
    END FUNCTION  
END CLASS
