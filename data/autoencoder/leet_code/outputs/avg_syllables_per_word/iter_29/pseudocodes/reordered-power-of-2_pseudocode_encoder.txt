CLASS Solution  
    FUNCTION reorderedPowerOf2(n)  
        SET collection_of_sorted_digit_tuples_of_powers_of_two TO an empty set  
        SET current_power_of_two TO one  
        WHILE current_power_of_two LESS THAN OR EQUAL TO one billion  
            CONVERT current_power_of_two TO a string_of_digits  
            CONVERT string_of_digits TO a list_of_characters  
            SORT list_of_characters IN ascending order  
            CONVERT sorted list_of_characters TO a tuple_of_characters  
            ADD tuple_of_characters TO collection_of_sorted_digit_tuples_of_powers_of_two  
            SET current_power_of_two TO current_power_of_two MULTIPLIED BY two  
        END WHILE  
        CONVERT n TO a string_of_digits  
        CONVERT string_of_digits TO a list_of_characters  
        SORT list_of_characters IN ascending order  
        CONVERT sorted list_of_characters TO tuple_of_characters_of_n  
        RETURN a boolean indicating whether tuple_of_characters_of_n IS A MEMBER OF collection_of_sorted_digit_tuples_of_powers_of_two  
    END FUNCTION  
END CLASS
