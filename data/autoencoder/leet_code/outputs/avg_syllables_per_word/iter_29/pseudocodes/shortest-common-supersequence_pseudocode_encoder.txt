CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1, str2)  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        
        CREATE a two dimensional list named dp WITH m PLUS one ROWS AND n PLUS one COLUMNS FILLED WITH zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one of str1 EQUALS character at position j MINUS one of str2 THEN  
                    SET element at position i AND j of dp TO element at position i MINUS one AND j MINUS one of dp PLUS one  
                ELSE  
                    SET element at position i AND j of dp TO the GREATER OF element at position i MINUS one AND j of dp AND element at position i AND j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        SET i TO m  
        SET j TO n  
        CREATE an empty list named scs  
        
        WHILE i GREATER THAN zero AND j GREATER THAN zero  
            IF character at position i MINUS one of str1 EQUALS character at position j MINUS one of str2 THEN  
                APPEND character at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF element at position i MINUS one AND j of dp GREATER THAN element at position i AND j MINUS one of dp THEN  
                APPEND character at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND character at position j MINUS one of str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i GREATER THAN zero  
            APPEND character at position i MINUS one of str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        
        WHILE j GREATER THAN zero  
            APPEND character at position j MINUS one of str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        
        REVERSE the order of elements in scs  
        
        RETURN the concatenation of all characters in scs forming a single string  
    END FUNCTION  
END CLASS
