CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS no elements  
        RETURN nothing  
    END IF  
    SET head TO a new ListNode with value as the element at position one of values and next as nothing  
    SET p TO head  
    FOR each val IN the list of elements of values starting from position two TO the last element  
        SET node TO a new ListNode with value as val and next as nothing  
        SET the next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS nothing AND p2 EQUALS nothing  
        RETURN true  
    END IF  
    IF p1 EQUALS nothing OR p2 EQUALS nothing  
        RETURN false  
    END IF  
    RETURN the condition that the val of p1 EQUALS the val of p2 AND the result of calling is_same_list recursively with the next of p1 AND the next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists(self, lists)  
        SET min_heap TO an empty collection that functions as a min-heap  
        SET index TO zero  
        FOR each lst IN lists WHILE index IS less than the LENGTH OF lists  
            IF lst IS NOT nothing  
                ADD to the min_heap a tuple consisting of the val of lst AS first element the current index AS second element AND lst AS third element  
            END IF  
            INCREMENT index BY one  
        END FOR  
        SET dummy TO a new ListNode with val as zero and next as nothing  
        SET current TO dummy  
        WHILE the min_heap is not empty  
            REMOVE from min_heap the tuple with the smallest first element and ASSIGN its elements respectively TO val AS first element index AS second element AND node AS third element  
            SET the next of current TO node  
            SET current TO the next of current  
            IF the next of node IS NOT nothing  
                ADD to the min_heap a tuple consisting of the val of the next of node AS first element index AS second element AND the next of node AS third element  
            END IF  
        END WHILE  
        RETURN the next of dummy  
    END FUNCTION  
END CLASS
