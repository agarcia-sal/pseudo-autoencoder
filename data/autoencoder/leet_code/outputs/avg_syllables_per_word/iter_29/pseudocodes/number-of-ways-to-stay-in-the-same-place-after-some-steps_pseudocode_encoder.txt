CLASS Solution  
    FUNCTION numWays(steps PARAMETER, arrLen PARAMETER)  
        SET MODULO TO one billion PLUS seven  
        SET maximum reachable index TO the minimum OF steps DIVIDED BY two AND arrLen MINUS one  
        SET dp TO a list of zeros WITH LENGTH maximum reachable index PLUS one  
        SET element at position zero OF dp TO one  
        FOR each step FROM one TO steps  
            SET previous dp TO a copy OF dp  
            FOR each index FROM zero TO maximum reachable index  
                SET element at position index OF dp TO element at position index OF previous dp  
                IF index GREATER THAN zero  
                    SET element at position index OF dp TO element at position index OF dp PLUS element at position index MINUS one OF previous dp THEN TAKEN MODULO modulo  
                END IF  
                IF index LESS THAN maximum reachable index  
                    SET element at position index OF dp TO element at position index OF dp PLUS element at position index PLUS one OF previous dp THEN TAKEN MODULO modulo  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position zero OF dp  
    END FUNCTION  
END CLASS
