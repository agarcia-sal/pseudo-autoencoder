CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid IS empty OR the element at position zero of grid IS empty  
            RETURN zero  
        END IF  

        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero of grid  

        CREATE a two dimensional list named dp with m rows each containing n elements all set TO zero  

        SET the element at position zero of the element at position zero of dp TO the element at position zero of the element at position zero of grid  

        FOR j FROM one TO n MINUS one  
            SET the element at position j of the element at position zero of dp TO the element at position j MINUS one of the element at position zero of dp PLUS the element at position j of the element at position zero of grid  
        END FOR  

        FOR i FROM one TO m MINUS one  
            SET the element at position zero of the element at position i of dp TO the element at position zero of the element at position i MINUS one of dp PLUS the element at position zero of the element at position i of grid  
        END FOR  

        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET the element at position j of the element at position i of dp TO the minimum VALUE BETWEEN the element at position j of the element at position i MINUS one of dp AND the element at position j MINUS one of the element at position i of dp PLUS the element at position j of the element at position i of grid  
            END FOR  
        END FOR  

        RETURN the element at position n MINUS one of the element at position m MINUS one of dp  
    END FUNCTION  
END CLASS
