CLASS Solution  
    FUNCTION fullJustify(words, maxWidth)  
        FUNCTION justify_line(line, num_of_letters, is_last_line)  
            IF the LENGTH OF line EQUALS one OR is_last_line IS true  
                SET number_of_spaces_to_add TO maxWidth MINUS num_of_letters MINUS the LENGTH OF line PLUS one  
                RETURN the CONCATENATION OF all words in line SEPARATED BY a single space PLUS number_of_spaces_to_add MANY spaces  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET number_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO total_spaces DIVIDED BY number_of_gaps  
                SET extra_spaces TO total_spaces MODULO number_of_gaps  
                FOR index FROM zero TO number_of_gaps MINUS one  
                    IF index IS LESS THAN extra_spaces  
                        APPEND spaces_between_words PLUS one MANY spaces TO the word at position index of line  
                    ELSE  
                        APPEND spaces_between_words MANY spaces TO the word at position index of line  
                    END IF  
                END FOR  
                RETURN the CONCATENATION OF all words in line WITH no separator  
            END IF  
        END FUNCTION  
        
        SET result TO an empty list  
        SET line TO an empty list  
        SET num_of_letters TO zero  
        
        FOR each word IN words  
            SET current_line_length TO num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line  
            IF current_line_length IS GREATER THAN maxWidth  
                APPEND the result of calling justify_line WITH line num_of_letters AND false TO result  
                SET line TO an empty list  
                SET num_of_letters TO zero  
            END IF  
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
        
        APPEND the result of calling justify_line WITH line num_of_letters AND true TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
