CLASS Solution  
    FUNCTION findKthNumber(m, n, k)  
        FUNCTION enough(x)  
            SET count TO zero  
            FOR i FROM one TO m  
                INCREMENT count BY the smaller value BETWEEN the integer division of x BY i AND n  
            END FOR  
            RETURN whether count IS GREATER THAN OR EQUAL TO k  
        END FUNCTION  

        SET low TO one  
        SET high TO the product of m AND n  
        WHILE low IS LESS THAN high  
            SET mid TO the integer division of the sum of low AND high BY two  
            IF calling enough WITH mid RETURNS true  
                SET high TO mid  
            ELSE  
                SET low TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
