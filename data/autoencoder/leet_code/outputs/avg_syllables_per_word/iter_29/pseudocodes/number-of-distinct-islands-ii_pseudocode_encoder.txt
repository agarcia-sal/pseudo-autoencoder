CLASS Solution  
    FUNCTION numDistinctIslands2(grid)  
        FUNCTION dfs(x y island)  
            IF x IS LESS THAN zero OR x IS GREATER THAN OR EQUAL TO the LENGTH OF grid OR y IS LESS THAN zero OR y IS GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            SET element at position x of grid at position y TO zero  
            APPEND tuple consisting of x and y TO island  
            CALL dfs WITH x PLUS one y island  
            CALL dfs WITH x MINUS one y island  
            CALL dfs WITH x y PLUS one island  
            CALL dfs WITH x y MINUS one island  
        END FUNCTION  

        FUNCTION normalize(island)  
            SET shapes TO empty list  
            FOR each pair dx dy IN list consisting of tuple one one tuple one negative one tuple negative one one tuple negative one negative one  
                SET new_island TO empty list  
                FOR each pair x y IN island  
                    APPEND tuple consisting of x MULTIPLIED BY dx and y MULTIPLIED BY dy TO new_island  
                END FOR  
                SORT new_island in ascending order  
                SET min_x TO the minimum x value among all pairs x y in new_island  
                SET min_y TO the minimum y value among all pairs x y in new_island  
                APPEND tuple consisting of pairs where each x MINUS min_x and y MINUS min_y for each pair x y in new_island TO shapes  
            END FOR  
            FOR index FROM zero TO the LENGTH OF shapes MINUS one  
                APPEND tuple consisting of pairs where y and NEGATIVE x for each pair x y in element at position index of shapes TO shapes  
            END FOR  
            RETURN the minimum element in shapes according to lexicographical order  
        END FUNCTION  

        SET islands TO empty set  
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    SET island TO empty list  
                    CALL dfs WITH i j island  
                    ADD normalize(island) TO islands  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the LENGTH OF islands  
    END FUNCTION  
END CLASS
