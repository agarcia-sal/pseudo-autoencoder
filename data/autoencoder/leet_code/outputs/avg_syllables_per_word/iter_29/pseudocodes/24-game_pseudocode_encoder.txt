CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the DIFFERENCE BETWEEN the element at position zero of nums AND twenty four IS LESS THAN one TEN MILLIONTH  
            END IF  
            
            FOR each index i FROM zero TO the LENGTH OF nums MINUS one  
                FOR each index j FROM zero TO the LENGTH OF nums MINUS one  
                    IF index i NOT EQUALS index j  
                        SET new_nums TO an empty list  
                        FOR each index k FROM zero TO the LENGTH OF nums MINUS one  
                            IF index k NOT EQUALS index i AND index k NOT EQUALS index j  
                                APPEND the element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR each operation op IN the list consisting of true division multiplication addition and subtraction  
                            IF operation op IS addition OR operation op IS multiplication AND index j GREATER THAN index i  
                                CONTINUE to the next iteration  
                            END IF  
                            
                            IF operation op IS NOT true division OR the element at position j of nums NOT EQUALS zero  
                                APPEND the result of applying operation op TO the elements at positions i and j of nums TO new_nums  
                                IF solve(new_nums) RETURNS true  
                                    RETURN true  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        IMPORT true division multiplication addition subtraction FROM operator  
        RETURN the result of solve applied TO cards  
    END FUNCTION  
END CLASS
