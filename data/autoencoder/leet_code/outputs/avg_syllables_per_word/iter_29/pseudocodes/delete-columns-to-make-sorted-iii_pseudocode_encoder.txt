CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET number_of_rows TO the LENGTH OF strs  
        SET number_of_columns TO the LENGTH OF element at position zero of strs  
        SET dp TO a list of the number one repeated number_of_columns times  
        FOR index_i FROM number one TO number_of_columns MINUS one  
            FOR index_j FROM zero TO index_i MINUS one  
                SET condition_to_check TO true  
                FOR index_k FROM zero TO number_of_rows MINUS one  
                    IF character at position index_j of element at position index_k of strs IS GREATER THAN character at position index_i of element at position index_k of strs  
                        SET condition_to_check TO false  
                        BREAK  
                    END IF  
                END FOR  
                IF condition_to_check EQUALS true  
                    IF element at position index_j of dp PLUS one IS GREATER THAN element at position index_i of dp  
                        SET element at position index_i of dp TO element at position index_j of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        SET longest_non_decreasing_subsequence_length TO the MAXIMUM value IN dp  
        SET result TO number_of_columns MINUS longest_non_decreasing_subsequence_length  
        RETURN result  
    END FUNCTION  
END CLASS
