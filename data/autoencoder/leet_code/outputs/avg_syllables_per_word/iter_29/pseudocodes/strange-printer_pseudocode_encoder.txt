CLASS Solution  
    FUNCTION strangePrinter(string_of_characters)  
        FUNCTION dp(start_index end_index memo_dictionary)  
            IF start_index GREATER THAN end_index  
                RETURN zero  
            END IF  
            IF the tuple of start_index AND end_index IS A KEY IN memo_dictionary  
                RETURN the value at key of tuple of start_index AND end_index in memo_dictionary  
            END IF  
            SET result TO the VALUE returned by dp with start_index AND end_index MINUS one AND memo_dictionary PLUS one  
            FOR each index_k FROM start_index TO end_index MINUS one  
                IF character at position index_k of string_of_characters EQUALS character at position end_index of string_of_characters  
                    SET candidate_value TO the VALUE returned by dp with start_index AND index_k MINUS one AND memo_dictionary PLUS the VALUE returned by dp with index_k AND end_index MINUS one AND memo_dictionary  
                    SET result TO the MINIMUM OF result AND candidate_value  
                END IF  
            END FOR  
            SET the value at key of tuple of start_index AND end_index in memo_dictionary TO result  
            RETURN result  
        END FUNCTION  
        RETURN the VALUE returned by dp with zero AND the LENGTH OF string_of_characters MINUS one AND an empty dictionary  
    END FUNCTION  
END CLASS
