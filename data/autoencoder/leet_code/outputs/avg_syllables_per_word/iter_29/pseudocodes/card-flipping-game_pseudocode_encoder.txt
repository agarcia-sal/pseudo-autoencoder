CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same_numbers TO the set of elements x FOR each pair of elements x and y TAKEN FROM fronts AND backs WHERE x EQUALS y  
        SET all_numbers TO the set of all elements FROM fronts combined WITH all elements FROM backs  
        SET good_numbers TO the set of all_numbers EXCLUDING all elements THAT ARE IN same_numbers  
        IF good_numbers IS NOT empty  
            RETURN the minimum element FROM good_numbers  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
