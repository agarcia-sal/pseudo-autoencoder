CLASS Solution  
    FUNCTION splitLoopedString(list_of_strings)  
        FOR index FROM zero TO the LENGTH OF list_of_strings MINUS one  
            SET reversed_string TO the string at position index of list_of_strings REVERSED  
            IF reversed_string GREATER THAN the string at position index of list_of_strings  
                SET the string at position index of list_of_strings TO reversed_string  
            END IF  
        END FOR  

        SET doubled_list_of_strings TO the concatenation of list_of_strings WITH list_of_strings  
        SET maximum_string TO the empty string  

        FOR index FROM zero TO the LENGTH OF list_of_strings MINUS one  
            FOR each boolean_value IN the collection of False AND True  
                IF boolean_value IS True  
                    SET current_string TO the string at position index of list_of_strings REVERSED  
                ELSE  
                    SET current_string TO the string at position index of list_of_strings  
                END IF  

                SET remaining_string TO the concatenation of the elements from position index PLUS one TO position index PLUS the LENGTH OF list_of_strings MINUS one of doubled_list_of_strings  

                FOR cut_position FROM zero TO the LENGTH OF current_string  
                    SET candidate_string TO the concatenation of the substring from position cut_position TO the end of current_string WITH remaining_string WITH the substring from position zero TO position cut_position MINUS one of current_string  
                    IF candidate_string GREATER THAN maximum_string  
                        SET maximum_string TO candidate_string  
                    END IF  
                END FOR  
            END FOR  
        END FOR  

        RETURN maximum_string  
    END FUNCTION  
END CLASS
