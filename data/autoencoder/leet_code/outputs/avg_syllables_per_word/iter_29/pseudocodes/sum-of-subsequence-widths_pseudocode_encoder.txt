CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET constant MOD TO ten POWERED BY nine PLUS seven  
        SORT nums IN NON-DECREASING ORDER  
        SET variable n TO the LENGTH OF nums  
        SET variable total_sum TO zero  
        SET list power TO a new list of size n FILLED WITH value one  
        
        FOR index i FROM one TO n MINUS one  
            SET element at position i of power TO the element at position i MINUS one of power MULTIPLIED BY two THEN MODULO MOD  
        END FOR  
        
        FOR index i FROM zero TO n MINUS one  
            SET total_sum TO total_sum PLUS the element at position i of nums MULTIPLIED BY (the element at position i of power MINUS the element at position n MINUS i MINUS one of power) THEN MODULO MOD  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
