CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO a list obtained by converting each number in nums to its string representation  
        
        FUNCTION compare(x, y)  
            SET concatenation_one TO the string resulting from appending string y to string x  
            SET concatenation_two TO the string resulting from appending string x to string y  
            IF concatenation_one lexicographically GREATER THAN concatenation_two  
                RETURN negative one  
            ELSE IF concatenation_one lexicographically LESS THAN concatenation_two  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str USING the comparison function compare as the key for ordering  
        
        IF the first element of nums_str EQUALS the string zero  
            RETURN the string zero  
        END IF  
        
        SET result_string TO the concatenation of all elements in nums_str in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
