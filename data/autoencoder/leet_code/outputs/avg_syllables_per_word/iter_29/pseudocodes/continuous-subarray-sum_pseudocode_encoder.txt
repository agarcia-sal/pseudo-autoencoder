CLASS Solution  
    FUNCTION checkSubarraySum(nums parameter_list, k parameter_name) RETURNS boolean  
        SET remainder_map TO a new dictionary with key zero associated with value negative one  
        SET current_sum TO zero  
        FOR each index i AND each number num IN enumeration of nums parameter_list  
            INCREMENT current_sum BY number num  
            IF k parameter_name NOT EQUALS zero  
                SET current_sum TO remainder of current_sum DIVIDED BY k parameter_name  
            END IF  
            IF current_sum IS A KEY IN remainder_map  
                IF difference between index i AND value at key current_sum in remainder_map IS GREATER THAN one  
                    RETURN true  
                END IF  
            ELSE  
                SET value at key current_sum in remainder_map TO index i  
            END ELSE  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
