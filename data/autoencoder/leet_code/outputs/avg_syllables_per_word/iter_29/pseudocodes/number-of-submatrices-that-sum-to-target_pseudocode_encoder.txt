CLASS Solution  
    FUNCTION numSubmatrixSumTarget(matrix WITH parameter name matrix AS list of list of integers, target WITH parameter name target AS integer) RETURNS integer  
        FUNCTION subarraySum(nums WITH parameter name nums AS list of integers, target WITH parameter name target AS integer) RETURNS integer  
            SET count TO zero  
            SET seen_sums TO a default dictionary mapping integer keys to integer values  
            SET cumulative_sum TO zero  
            FOR each number IN nums  
                INCREMENT cumulative_sum BY number  
                IF cumulative_sum EQUALS target THEN  
                    INCREMENT count BY one  
                END IF  
                IF the difference of cumulative_sum MINUS target IS A KEY IN seen_sums THEN  
                    INCREMENT count BY the value associated with the key of the difference of cumulative_sum MINUS target in seen_sums  
                END IF  
                INCREMENT the value associated with the key cumulative_sum in seen_sums BY one  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET rows TO the LENGTH OF matrix  
        SET cols TO the LENGTH OF the element at position zero of matrix  
        SET total_count TO zero  
        
        FOR top FROM zero TO rows MINUS one  
            SET current_cols TO a list of zeros of length cols  
            FOR bottom FROM top TO rows MINUS one  
                FOR col FROM zero TO cols MINUS one  
                    INCREMENT the element at position col of current_cols BY the element at position bottom of matrix then the element at position col of that matrix element  
                END FOR  
                INCREMENT total_count BY the result of calling subarraySum with current_cols and target  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
