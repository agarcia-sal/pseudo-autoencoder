CLASS Solution  
    FUNCTION minTrioDegree(parameter n AS integer PARAMETER edges AS list OF list OF integer)  
        SET graph TO an empty dictionary from integer TO set of integers  
        FOR each integer i FROM one TO n  
            SET to the graph AT key i an empty set  
        END FOR  

        FOR each pair of integers u AND v IN edges  
            ADD v TO the set referenced by graph AT key u  
            ADD u TO the set referenced by graph AT key v  
        END FOR  

        SET min_degree TO positive infinity  
        SET trio_found TO false  

        FOR each integer i FROM one TO n  
            FOR each integer j FROM i PLUS one TO n  
                FOR each integer k FROM j PLUS one TO n  
                    IF the set referenced by graph AT key i CONTAINS j AND the set referenced by graph AT key i CONTAINS k AND the set referenced by graph AT key j CONTAINS k THEN  
                        SET trio_found TO true  
                        SET trio_degree TO the LENGTH OF the set referenced by graph AT key i PLUS the LENGTH OF the set referenced by graph AT key j PLUS the LENGTH OF the set referenced by graph AT key k MINUS six  
                        IF trio_degree LESS THAN min_degree THEN  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  

        IF trio_found THEN  
            RETURN min_degree  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
