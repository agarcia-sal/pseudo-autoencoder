CLASS Solution  
    FUNCTION maxHeight(cuboids)  
        FOR each cuboid IN cuboids  
            SORT the dimensions of the cuboid from smallest TO largest  
        END FOR  
        SORT the list of cuboids lexicographically by their dimensions in ascending order  
        SET number_of_cuboids TO the LENGTH OF cuboids  
        CREATE a list dp WITH number_of_cuboids elements INITIALLY all zero  
        FOR index i FROM zero TO number_of_cuboids MINUS one  
            SET dp at position i TO the element at position three of cuboid at position i  
            FOR index j FROM zero TO i MINUS one  
                IF the element at position one of cuboid at position j IS LESS THAN OR EQUAL TO the element at position one of cuboid at position i  
                AND the element at position two of cuboid at position j IS LESS THAN OR EQUAL TO the element at position two of cuboid at position i  
                AND the element at position three of cuboid at position j IS LESS THAN OR EQUAL TO the element at position three of cuboid at position i  
                    SET dp at position i TO the GREATER OF dp at position i AND dp at position j PLUS the element at position three of cuboid at position i  
                END IF  
            END FOR  
        END FOR  
        RETURN the MAXIMUM value contained in the list dp  
    END FUNCTION  
END CLASS
