CLASS Solution  
    FUNCTION findSubsequences(nums)  
        FUNCTION dfs(current_index, last_value, temporary_list)  
            IF current_index EQUALS the LENGTH OF nums THEN  
                IF the LENGTH OF temporary_list GREATER THAN one THEN  
                    APPEND a copy OF temporary_list TO ans  
                END IF  
                RETURN  
            END IF  
            IF the element at position current_index of nums GREATER THAN OR EQUAL TO last_value THEN  
                APPEND the element at position current_index of nums TO temporary_list  
                CALL dfs WITH arguments current_index PLUS one last_value REPLACED BY the element at position current_index of nums temporary_list  
                REMOVE the last element FROM temporary_list  
            END IF  
            IF the element at position current_index of nums NOT EQUALS last_value THEN  
                CALL dfs WITH arguments current_index PLUS one last_value temporary_list  
            END IF  
        END FUNCTION  
        SET ans TO an empty list  
        CALL dfs WITH arguments zero MINUS one thousand temporary_list AS an empty list  
        RETURN ans  
    END FUNCTION  
END CLASS
