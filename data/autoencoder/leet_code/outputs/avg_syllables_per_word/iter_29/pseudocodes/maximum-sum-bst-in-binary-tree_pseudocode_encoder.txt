CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value  
        RETURN true  
    ELSE IF p IS no value OR q IS no value  
        RETURN false  
    ELSE IF the value attribute of p IS NOT EQUALS the value attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree applied to the left attribute of p and the left attribute of q AND the result of is_same_tree applied to the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST(self, root)  
        SET self.max_sum TO zero  
        
        FUNCTION helper(node)  
            IF node IS no value  
                RETURN true WITH zero WITH positive infinity WITH negative infinity  
            END IF  
            
            SET left_is_bst, left_sum, left_min, left_max TO the result of calling helper with the left attribute of node  
            SET right_is_bst, right_sum, right_min, right_max TO the result of calling helper with the right attribute of node  
            
            IF left_is_bst IS true AND right_is_bst IS true AND left_max IS LESS THAN the value attribute of node AND the value attribute of node IS LESS THAN right_min  
                SET current_sum TO left_sum PLUS right_sum PLUS the value attribute of node  
                SET current_min TO the lesser of the value attribute of node AND left_min  
                SET current_max TO the greater of the value attribute of node AND right_max  
                SET self.max_sum TO the greater of self.max_sum AND current_sum  
                RETURN true WITH current_sum WITH current_min WITH current_max  
            END IF  
            
            RETURN false WITH zero WITH positive infinity WITH negative infinity  
        END FUNCTION  
        
        CALL helper with root  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
