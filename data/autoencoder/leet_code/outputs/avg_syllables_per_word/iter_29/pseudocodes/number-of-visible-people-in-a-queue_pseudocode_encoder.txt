CLASS Solution  
    FUNCTION canSeePersonsCount(heights)  
        SET number_of_people TO the LENGTH OF heights  
        SET answer_list TO a new list of zeros with size number_of_people  
        SET stack_list TO an empty list  
        
        FOR index FROM zero TO number_of_people MINUS one  
            WHILE the stack_list is not empty AND the element at position element at last position of stack_list of heights IS LESS THAN the element at position index of heights  
                SET popped_index TO the last element removed from stack_list  
                INCREMENT the element at position popped_index of answer_list BY one  
            END WHILE  
            IF the stack_list is not empty  
                INCREMENT the element at position element at last position of stack_list of answer_list BY one  
            END IF  
            APPEND index TO stack_list  
        END FOR  
        
        RETURN answer_list  
    END FUNCTION  
END CLASS
