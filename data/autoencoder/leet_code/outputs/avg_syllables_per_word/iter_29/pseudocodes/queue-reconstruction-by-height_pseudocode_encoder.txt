CLASS Solution  
    FUNCTION reconstructQueue(people)  
        SORT the list people BY the following criteria in order  
            height OF person MULTIPLIED BY negative one FOR descending order  
            value k OF person FOR ascending order  
        SET queue TO an empty list  
        FOR each person IN people  
            INSERT person INTO queue AT position value k OF person  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
