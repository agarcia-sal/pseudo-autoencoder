CLASS Solution  
    FUNCTION splitArray(nums k)  
        FUNCTION canSplit(max_sum)  
            SET current_sum TO zero  
            SET subarrays TO one  
            FOR each number IN nums  
                IF current_sum PLUS number GREATER THAN max_sum  
                    INCREMENT subarrays BY one  
                    SET current_sum TO number  
                ELSE  
                    INCREMENT current_sum BY number  
                END IF  
            END FOR  
            RETURN subarrays LESS THAN OR EQUAL TO k  
        END FUNCTION  
      
        SET left TO the MAXIMUM VALUE IN nums  
        SET right TO the SUM OF all elements IN nums  
        WHILE left LESS THAN right  
            SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
            IF canSplit(mid)  
                SET right TO mid  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
