CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO a mapping with default value zero for any key  
        FOR each first_number IN nums  
            FOR each second_number IN nums  
                SET bitwise_and_result TO first_number BITWISE AND second_number  
                INCREMENT the value at key bitwise_and_result in pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each k_number IN nums  
            FOR each key_value_pair IN the entries of pair_and_count  
                SET pair_and TO the key of key_value_pair  
                SET count TO the value of key_value_pair  
                IF pair_and BITWISE AND k_number EQUALS zero THEN  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
