CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO a new TreeNode object initialized with the element at position one of values  
    SET i TO one  
    CREATE an empty queue collection  
    ADD root TO the queue  
    WHILE the queue IS NOT EMPTY  
        SET node TO the element REMOVED TEMPORARILY from the FRONT of the queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT equal TO None  
            SET node dot left TO a new TreeNode object initialized with the element at position i of values  
            ADD node dot left TO the queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT equal TO None  
            SET node dot right TO a new TreeNode object initialized with the element at position i of values  
            ADD node dot right TO the queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with p dot left AND q dot left AND the result of calling is_same_tree with p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView(self, root)  
        IF root IS None  
            RETURN an empty list  
        END IF  

        CREATE an empty list called right_side_view  
        CREATE a queue collection initialized with root as its only element  

        WHILE the queue IS NOT EMPTY  
            SET level_length TO the LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO the element REMOVED TEMPORARILY from the FRONT of the queue  
                IF i EQUALS level_length MINUS one  
                    ADD node dot val TO right_side_view  
                END IF  
                IF node dot left IS NOT None  
                    ADD node dot left TO the queue  
                END IF  
                IF node dot right IS NOT None  
                    ADD node dot right TO the queue  
                END IF  
            END FOR  
        END WHILE  

        RETURN right_side_view  
    END FUNCTION  
END CLASS
