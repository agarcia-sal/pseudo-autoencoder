CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a collection of n elements each SET TO negative one  
        SET stack TO an empty collection  
        FOR i FROM zero TO two TIMES n MINUS one  
            SET index TO the remainder of i DIVIDED BY n  
            WHILE the stack IS NOT empty AND the element at position element at the last position of stack of nums IS LESS THAN the element at position index of nums  
                SET popped_index TO the last element removed from stack  
                SET the element at position popped_index of result TO the element at position index of nums  
            END WHILE  
            IF i IS LESS THAN n  
                ADD index TO the end of stack  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
