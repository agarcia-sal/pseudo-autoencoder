CLASS Solution  
    FUNCTION rearrangeString(s k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO a mapping of each character in s TO its frequency count  
        
        SET max_heap TO an empty list  
        FOR each character AND its count IN freq  
            APPEND the pair of the negation of count AND character TO max_heap  
        END FOR  
        APPLY heapification TO max_heap  
        
        SET wait_queue TO an empty list  
        
        SET result TO an empty list  
        
        WHILE max_heap IS NOT empty  
            EXTRACT the smallest element FROM max_heap ASSIGNING TO count AND char  
            APPEND char TO result  
            
            APPEND the pair of count INCREMENTED BY one AND char TO wait_queue  
            
            IF the LENGTH OF wait_queue EQUALS k  
                REMOVE the first element FROM wait_queue ASSIGNING TO count AND char  
                IF count LESS THAN zero  
                    INSERT the pair count AND char INTO max_heap ACCORDING TO heap rules  
                END IF  
            END IF  
        END WHILE  
        
        IF the LENGTH OF result IS NOT EQUAL TO the LENGTH OF s  
            RETURN an empty string  
        END IF  
        
        RETURN the concatenation of all elements IN result  
    END FUNCTION  
END CLASS
