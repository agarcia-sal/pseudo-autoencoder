CLASS Solution  
    FUNCTION secondMinimum(n time_parameter edges list_of_lists_of_integers time change) RETURNS integer  
        SET graph TO a list containing empty lists repeated the number of time which is the sum of n AND one  
        FOR each pair of integers u and v IN edges  
            APPEND v TO the list element at position u of graph  
            APPEND u TO the list element at position v of graph  
        END FOR  
        CREATE an empty queue initialized with a tuple containing the integer one the integer zero and the integer zero  
        SET minTimes TO a list containing a list of two elements each being positive infinity repeated the number of time which is the sum of n AND one  
        SET the element at position one THEN zero in minTimes TO zero  
        WHILE the queue is not empty  
            REMOVE the element at the front of queue and UNPACK into variables current currentTime and count  
            IF current EQUALS n AND count EQUALS one  
                RETURN currentTime  
            END IF  
            IF the quotient of currentTime DIVIDED BY change MODULO two EQUALS zero  
                SET nextTime TO currentTime PLUS time  
            ELSE  
                SET nextTime TO the sum of the product of the sum of the quotient of currentTime DIVIDED BY change PLUS one AND change AND time  
            END IF  
            FOR each neighbor IN the list element at position current of graph  
                IF the element at position neighbor THEN zero in minTimes IS GREATER THAN nextTime  
                    SET the element at position neighbor THEN one in minTimes TO the element at position neighbor THEN zero in minTimes  
                    SET the element at position neighbor THEN zero in minTimes TO nextTime  
                    APPEND the tuple of neighbor nextTime and zero TO queue  
                ELSE IF the element at position neighbor THEN zero in minTimes IS LESS THAN nextTime AND nextTime IS LESS THAN the element at position neighbor THEN one in minTimes  
                    SET the element at position neighbor THEN one in minTimes TO nextTime  
                    APPEND the tuple of neighbor nextTime and one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
