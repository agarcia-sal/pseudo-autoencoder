CLASS Solution  
    FUNCTION minSkips(dist LIST OF INTEGERS speed INTEGER hoursBefore INTEGER) RETURNS INTEGER  
        SET n TO the LENGTH OF dist  
        SET dp TO a two dimensional list with n rows and n PLUS one columns where each element IS a representation OF infinity  
        SET total_dist TO the SUM OF all elements IN dist  
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  
        
        SET the element at position zero of row zero IN dp TO the element at position zero OF dist  
        
        FOR each j FROM one TO n MINUS one  
            SET the element at position j of row zero IN dp TO the element at position zero OF dist  
        END FOR  
        
        FOR each i FROM one TO n MINUS one  
            FOR each j FROM zero TO i  
                IF j LESS THAN i THEN  
                    SET the element at position j of row i IN dp TO the CEILING OF the element at position j of row i MINUS one IN dp DIVIDED BY speed MULTIPLIED BY speed PLUS the element at position i OF dist  
                END IF  
                IF j GREATER THAN zero THEN  
                    SET the element at position j of row i IN dp TO the MINIMUM OF the current value at the element at position j of row i IN dp AND the element at position j MINUS one of row i MINUS one IN dp PLUS the element at position i OF dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR each j FROM zero TO n MINUS one  
            IF the element at position j of row n MINUS one IN dp LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
