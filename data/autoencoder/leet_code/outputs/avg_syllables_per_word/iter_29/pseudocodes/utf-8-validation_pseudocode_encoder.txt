CLASS Solution  
    FUNCTION validUtf8(data)  
        FUNCTION get_byte_count(byte)  
            IF the result of shifting byte to the right by seven positions EQUALS zero  
                RETURN one  
            ELSE IF the result of shifting byte to the right by five positions EQUALS the binary value one one zero  
                RETURN two  
            ELSE IF the result of shifting byte to the right by four positions EQUALS the binary value one one one zero  
                RETURN three  
            ELSE IF the result of shifting byte to the right by three positions EQUALS the binary value one one one one zero  
                RETURN four  
            ELSE  
                RETURN negative one  
            END ELSE  
        END FUNCTION  

        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF data  
            SET byte_count TO the result of get_byte_count applied to the element at position i of data  
            IF byte_count EQUALS negative one  
                RETURN False  
            END IF  
            IF the sum of i AND byte_count GREATER THAN the LENGTH OF data  
                RETURN False  
            END IF  
            FOR each j FROM the sum of i and one TO the difference of the sum of i and byte_count and one inclusive  
                IF the result of shifting the element at position j of data to the right by six positions NOT EQUALS the binary value one zero  
                    RETURN False  
                END IF  
            END FOR  
            INCREMENT i BY byte_count  
        END WHILE  
        RETURN True  
    END FUNCTION  
END CLASS
