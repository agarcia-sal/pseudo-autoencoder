CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        SET q TO an empty double ended queue containing a single element that is a list containing zero  
        SET ans TO an empty list  
        WHILE the double ended queue q is not empty  
            SET path TO the first element removed from the left side of the double ended queue q  
            SET u TO the element at the last position of the list path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            FOR each v IN the list at position u of graph  
                APPEND the list formed by concatenating path and a list containing v TO the double ended queue q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
