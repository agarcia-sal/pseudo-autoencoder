CLASS Solution  
    FUNCTION exist(board, word)  
        IF board EQUALS empty THEN  
            RETURN False  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        SET visited TO a two dimensional collection of size m BY n filled with False  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS the LENGTH OF word THEN  
                RETURN True  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of visited AT position y IS True OR element at position x of board AT position y DOES NOT EQUAL character at position index of word THEN  
                RETURN False  
            END IF  

            SET element at position x of visited AT position y TO True  
            FOR each pair dx and dy IN the collection composed of pair of minus one AND zero AND pair of one AND zero AND pair of zero AND minus one AND pair of zero AND one  
                IF dfs(x PLUS dx, y PLUS dy, index PLUS one) IS True THEN  
                    RETURN True  
                END IF  
            END FOR  
            SET element at position x of visited AT position y TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs(i, j, zero) IS True THEN  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
