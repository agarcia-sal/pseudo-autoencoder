CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the STRING REPRESENTATION OF element at position one of nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the STRING REPRESENTATION OF element at position one of nums DIVIDED BY the STRING REPRESENTATION OF element at position two of nums  
        END IF  
        RETURN the STRING REPRESENTATION OF element at position one of nums DIVIDED BY the OPENING PARENTHESIS concatenated WITH the STRING REPRESENTATION OF element at position two of nums concatenated WITH the DIVISION SYMBOL concatenated WITH the concatenation of the STRING REPRESENTATIONS of each element from position three TO the last position of nums separated BY the DIVISION SYMBOL concatenated WITH the CLOSING PARENTHESIS  
    END FUNCTION  
END CLASS
