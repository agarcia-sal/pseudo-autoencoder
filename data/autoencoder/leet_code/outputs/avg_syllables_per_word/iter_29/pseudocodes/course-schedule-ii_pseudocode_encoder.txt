CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        CREATE an empty map named adjacency list to map each course TO an empty list  
        FOR each course FROM zero TO numCourses MINUS one  
            SET the entry for course in adjacency list TO an empty list  
        END FOR  
        
        CREATE a list named in degrees with the length of numCourses FILLED WITH zeroes  
        
        FOR each pair of destination course AND source course IN prerequisites  
            APPEND destination course TO the list at key source course in adjacency list  
            INCREMENT the in degree at position destination course BY one  
        END FOR  
        
        CREATE a queue containing all courses FROM zero TO numCourses MINUS one WHERE the in degree at position course EQUALS zero  
        CREATE an empty list named topological order  
        
        WHILE the queue is not empty  
            REMOVE the first element from the queue and SET it TO current  
            APPEND current TO topological order  
            
            FOR each neighbor IN the list at key current in adjacency list  
                DECREMENT the in degree at position neighbor BY one  
                IF the in degree at position neighbor EQUALS zero  
                    APPEND neighbor TO the queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF topological order EQUALS numCourses  
            RETURN topological order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
