CLASS Solution  
    FUNCTION isMatch(string_s, string_p)  
        SET dp_table TO a list of lists of False with number of rows EQUAL TO the LENGTH OF string_s PLUS one AND number of columns EQUAL TO the LENGTH OF string_p PLUS one  
        
        SET element at position zero zero of dp_table TO True  
        
        FOR index_j FROM one TO the LENGTH OF string_p inclusive  
            SET position_in_pattern TO index_j MINUS one  
            IF character at position position_in_pattern of string_p EQUALS the character representing asterisk THEN  
                SET element at position zero index_j of dp_table TO the element at position zero index_j MINUS two of dp_table  
            END IF  
        END FOR  
        
        FOR index_i FROM one TO the LENGTH OF string_s inclusive  
            FOR index_j FROM one TO the LENGTH OF string_p inclusive  
                SET position_in_string TO index_i MINUS one  
                SET position_in_pattern TO index_j MINUS one  
                IF character at position position_in_pattern of string_p EQUALS character at position position_in_string of string_s OR character at position position_in_pattern of string_p EQUALS the character representing dot THEN  
                    SET element at position index_i index_j of dp_table TO the element at position index_i MINUS one index_j MINUS one of dp_table  
                ELSE IF character at position position_in_pattern of string_p EQUALS the character representing asterisk THEN  
                    SET element at position index_i index_j of dp_table TO the element at position index_i index_j MINUS two of dp_table  
                    IF character at position position_in_pattern MINUS one of string_p EQUALS character at position position_in_string of string_s OR character at position position_in_pattern MINUS one of string_p EQUALS the character representing dot THEN  
                        SET element at position index_i index_j of dp_table TO element at position index_i index_j of dp_table OR element at position index_i MINUS one index_j of dp_table  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position the LENGTH OF string_s the LENGTH OF string_p of dp_table  
    END FUNCTION  
END CLASS
