CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF the LENGTH OF remaining EQUALS zero  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR index FROM zero TO the LENGTH OF remaining MINUS one  
                SET new_path TO the CONCATENATION OF path AND the element at position index of remaining  
                SET new_remaining TO the CONCATENATION OF the SUBLIST of remaining FROM position zero TO position index MINUS one AND the SUBLIST of remaining FROM position index PLUS one TO the END OF remaining  
                CALL backtrack WITH new_path AND new_remaining  
            END FOR  
        END FUNCTION  
        SET permutations TO an empty list  
        CALL backtrack WITH an empty list AND nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
