CLASS Solution  
    FUNCTION generatePalindromes(string_input)  
        FUNCTION dfs(current_string)  
            IF the LENGTH OF current_string EQUALS the LENGTH OF string_input  
                APPEND current_string TO answer_collection  
                RETURN  
            END IF  
            FOR each character_key AND character_value IN the items of count_dictionary  
                IF character_value GREATER THAN one  
                    DECREMENT the value associated with character_key in count_dictionary BY two  
                    CALL dfs WITH the argument character_key CONCATENATED WITH current_string CONCATENATED WITH character_key  
                    INCREMENT the value associated with character_key in count_dictionary BY two  
                END IF  
            END FOR  
        END FUNCTION  
  
        SET count_dictionary TO the COUNTER COLLECTION of string_input  
        SET middle_character TO an empty string  
        FOR each character_key AND character_value IN the items of count_dictionary  
            IF character_value MODULO two EQUALS one  
                IF middle_character IS NOT an empty string  
                    RETURN an empty list  
                END IF  
                SET middle_character TO character_key  
                DECREMENT the value associated with character_key in count_dictionary BY one  
            END IF  
        END FOR  
        SET answer_collection TO an empty list  
        CALL dfs WITH the argument middle_character  
        RETURN answer_collection  
    END FUNCTION  
END CLASS
