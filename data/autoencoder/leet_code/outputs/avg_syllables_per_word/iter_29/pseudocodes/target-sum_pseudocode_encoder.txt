CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END ELSE  
            END IF  
            IF the TUPLE consisting of index AND current_sum EXISTS AS A KEY IN memo  
                RETURN the VALUE ASSOCIATED WITH the TUPLE consisting of index AND current_sum IN memo  
            END IF  
            SET positive TO dfs with index INCREMENTED BY one AND current_sum INCREMENTED BY element at position index of nums AND memo  
            SET negative TO dfs with index INCREMENTED BY one AND current_sum DECREMENTED BY element at position index of nums AND memo  
            SET the VALUE ASSOCIATED WITH the TUPLE consisting of index AND current_sum IN memo TO positive PLUS negative  
            RETURN the VALUE ASSOCIATED WITH the TUPLE consisting of index AND current_sum IN memo  
        END FUNCTION  
        RETURN dfs with zero AS index AND zero AS current_sum AND an EMPTY memo  
    END FUNCTION  
END CLASS
