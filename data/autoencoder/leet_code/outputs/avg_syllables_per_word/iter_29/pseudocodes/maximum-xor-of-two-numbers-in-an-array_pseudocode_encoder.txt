CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO zero  
        FOR i FROM thirty-one DOWN TO zero  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            SET prefixes TO an empty set  
            FOR each number IN nums  
                SET prefix TO the integer division of number BY two POWERED BY i  
                ADD prefix TO prefixes  
            END FOR  
            FOR each prefix_element IN prefixes  
                IF the bitwise XOR of current_candidate AND prefix_element IS an element IN prefixes THEN  
                    SET max_xor TO current_candidate  
                    BREAK  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
