CLASS Solution  
    FUNCTION superpalindromesInRange(left_parameter right_parameter)  
  
        FUNCTION is_palindrome(number_parameter)  
            RETURN the string representation of number_parameter EQUALS the string representation of number_parameter REVERSED  
        END FUNCTION  
  
        FUNCTION generate_palindromes(limit_parameter)  
            SET palindromes_list TO an empty list  
            FOR integer_value FROM one TO one hundred thousand MINUS one  
                SET string_value TO the string representation of integer_value  
                APPEND to palindromes_list the integer conversion of the concatenation of string_value AND the substring of string_value from the second last character TO the first character in reverse order for odd length palindromes  
                APPEND to palindromes_list the integer conversion of the concatenation of string_value AND the reverse of string_value for even length palindromes  
            END FOR  
            RETURN the list of each palindrome in palindromes_list WHERE the palindrome IS LESS THAN OR EQUAL TO limit_parameter  
        END FUNCTION  
  
        SET left TO the integer conversion of left_parameter  
        SET right TO the integer conversion of right_parameter  
        SET limit TO the integer conversion of right POWERED BY one half PLUS one  
        SET palindromes TO the result of generate_palindromes with limit_parameter equal to limit  
  
        SET count TO zero  
        FOR each palindrome_value IN palindromes  
            SET square TO palindrome_value MULTIPLIED BY palindrome_value  
            IF left IS LESS THAN OR EQUAL TO square AND square IS LESS THAN OR EQUAL TO right AND is_palindrome with number_parameter equal to square RETURNS true  
                INCREMENT count BY one  
            END IF  
        END FOR  
  
        RETURN count  
    END FUNCTION  
END CLASS
