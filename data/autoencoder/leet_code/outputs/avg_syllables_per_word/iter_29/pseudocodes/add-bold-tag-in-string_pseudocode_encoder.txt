CLASS Solution  
    FUNCTION addBoldTag(s WORDS, words LIST OF WORDS) RETURNS WORDS  
        IF the LENGTH OF words EQUALS zero THEN  
            RETURN s  
        END IF  
          
        SET n TO the LENGTH OF s  
        SET mask TO a new list of Boolean FALSE values with a size EQUAL TO n  
          
        FOR each word IN words  
            SET start TO zero  
            WHILE start IS LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the POSITION OF word FOUND IN s STARTING FROM start  
                IF start EQUALS negative one THEN  
                    BREAK the WHILE loop  
                END IF  
                FOR each index i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i of mask TO TRUE  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
          
        SET result TO an empty list  
        SET i TO zero  
        WHILE i IS LESS THAN n  
            IF the element at position i of mask IS TRUE THEN  
                APPEND the text for the opening bold tag TO result  
                WHILE i IS LESS THAN n AND the element at position i of mask IS TRUE  
                    APPEND the character at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the text for the closing bold tag TO result  
            ELSE  
                APPEND the character at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
          
        RETURN the concatenation of all elements in result as a single string  
    END FUNCTION  
END CLASS
