CLASS Solution  
    FUNCTION findDerangement(n)  
        SET MODULO TO ten TO THE POWER OF nine PLUS seven  
        
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        IF n EQUALS two  
            RETURN one  
        END IF  
        
        SET prev2 TO one  
        SET prev1 TO zero  
        
        FOR i FROM three TO n  
            SET current TO the product of the quantity of i MINUS one AND the quantity of prev1 PLUS prev2 THEN TAKEN MODULO MODULO  
            SET prev2 TO prev1  
            SET prev1 TO current  
        END FOR  
        
        RETURN prev1  
    END FUNCTION  
END CLASS
