CLASS Solution  
    FUNCTION maxSatisfaction(satisfaction)  
        SORT the list satisfaction IN descending order  
        SET variable max_satisfaction TO zero  
        SET variable current_satisfaction TO zero  
        SET variable total_satisfaction TO zero  
        FOR each element s IN the list satisfaction  
            INCREMENT current_satisfaction BY s  
            INCREMENT total_satisfaction BY current_satisfaction  
            SET max_satisfaction TO the greater value BETWEEN max_satisfaction AND total_satisfaction  
        END FOR  
        RETURN max_satisfaction  
    END FUNCTION  
END CLASS
