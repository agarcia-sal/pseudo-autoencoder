CLASS Solution  
    FUNCTION numDistinctIslands(grid)  
        FUNCTION dfs(i, j, direction)  
            IF i GREATER THAN OR EQUAL TO zero AND i LESS THAN the LENGTH OF grid AND j GREATER THAN OR EQUAL TO zero AND j LESS THAN the LENGTH OF element at position zero of grid AND element at position i of grid at position j EQUALS one  
                SET element at position i of grid at position j TO zero  
                APPEND direction TO path  
                CALL dfs WITH parameters i PLUS one j direction REPRESENTING down  
                CALL dfs WITH parameters i MINUS one j direction REPRESENTING up  
                CALL dfs WITH parameters i j PLUS one direction REPRESENTING right  
                CALL dfs WITH parameters i j MINUS one direction REPRESENTING left  
                APPEND a letter representing backtracking TO path  
            END IF  
            RETURN path  
        END FUNCTION  
        
        SET unique_islands TO an empty collection that stores unique elements  
        
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    SET path TO an empty list  
                    ADD the sequence returned by calling dfs WITH i j and a letter representing the origin converted to an immutable sequence TO unique_islands  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the NUMBER OF ELEMENTS IN unique_islands  
    END FUNCTION  
END CLASS
