CLASS Solution  
    FUNCTION smallestDistancePair(nums list_of_integers k integer) RETURNS integer  
        EXECUTE the OPERATION of SORTING the list_of_integers nums IN ascending order  
        SET n TO the LENGTH OF nums  
        
        FUNCTION count(mid integer) RETURNS integer  
            SET count TO zero  
            SET left TO zero  
            FOR each right FROM zero TO n MINUS one  
                WHILE the VALUE AT position right OF nums MINUS the VALUE AT position left OF nums IS GREATER THAN mid  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT count BY right MINUS left  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the VALUE AT last POSITION OF nums MINUS the VALUE AT first POSITION OF nums  
        
        WHILE left IS LESS THAN right  
            SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
            IF the RESULT OF count(mid) IS LESS THAN k  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
