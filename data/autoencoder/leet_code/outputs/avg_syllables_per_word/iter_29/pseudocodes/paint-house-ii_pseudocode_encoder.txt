CLASS Solution  
    FUNCTION minCostII(costs)  
        IF costs EQUALS empty collection THEN  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF costs  
        SET k TO the LENGTH OF the element at position zero of costs  
        
        IF k EQUALS one THEN  
            IF n EQUALS one THEN  
                RETURN element at position zero at position zero of costs  
            ELSE  
                RETURN positive infinity  
            END IF  
        END IF  
        
        FOR each i FROM one TO n MINUS one  
            SET min_cost TO positive infinity  
            SET min_cost_index TO negative one  
            SET second_min_cost TO positive infinity  
            
            FOR each j FROM zero TO k MINUS one  
                IF element at position i MINUS one at position j of costs LESS THAN min_cost THEN  
                    SET second_min_cost TO min_cost  
                    SET min_cost TO element at position i MINUS one at position j of costs  
                    SET min_cost_index TO j  
                ELSE IF element at position i MINUS one at position j of costs LESS THAN second_min_cost THEN  
                    SET second_min_cost TO element at position i MINUS one at position j of costs  
                END ELSE IF  
            END FOR  
            
            FOR each j FROM zero TO k MINUS one  
                IF j EQUALS min_cost_index THEN  
                    INCREMENT element at position i at position j of costs BY second_min_cost  
                ELSE  
                    INCREMENT element at position i at position j of costs BY min_cost  
                END IF  
            END FOR  
        END FOR  
        
        SET last_row TO element at position n MINUS one of costs  
        SET minimum_cost TO positive infinity  
        FOR each cost_value IN last_row  
            IF cost_value LESS THAN minimum_cost THEN  
                SET minimum_cost TO cost_value  
            END IF  
        END FOR  
        
        RETURN minimum_cost  
    END FUNCTION  
END CLASS
