CLASS Solution  
    FUNCTION carFleet(target position LIST OF integers speed LIST OF integers) RETURNS integer  
        SET list_of_pairs TO list of pairs made by pairing each element of position with corresponding element of speed  
        SORT list_of_pairs IN ORDER from the largest first element TO the smallest first element  
        SET list_of_times TO empty list  
        FOR each pair IN list_of_pairs  
            SET position_value TO the first element of pair  
            SET speed_value TO the second element of pair  
            SET time_to_reach TO target MINUS position_value DIVIDED BY speed_value  
            APPEND time_to_reach TO list_of_times  
        END FOR  
        SET fleet_count TO zero  
        SET current_maximum_time TO zero  
        FOR each time_value IN list_of_times  
            IF time_value GREATER THAN current_maximum_time  
                INCREMENT fleet_count BY one  
                SET current_maximum_time TO time_value  
            END IF  
        END FOR  
        RETURN fleet_count  
    END FUNCTION  
END CLASS
