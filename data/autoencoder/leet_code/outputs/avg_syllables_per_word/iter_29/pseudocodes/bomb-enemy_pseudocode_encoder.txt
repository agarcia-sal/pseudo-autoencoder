CLASS Solution  
    FUNCTION maxKilledEnemies(grid)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN zero  
        END IF  
        
        SET number_of_rows TO the LENGTH OF grid  
        SET number_of_columns TO the LENGTH OF element at position zero of grid  
        SET maximum_kills TO zero  
        
        SET row_kills TO zero  
        SET column_kills TO a list OF number_of_columns elements all set TO zero  
        
        FOR each row_index FROM zero TO number_of_rows MINUS one  
            FOR each column_index FROM zero TO number_of_columns MINUS one  
                
                IF column_index EQUALS zero OR the element at position column_index MINUS one of row at position row_index of grid EQUALS character WALL  
                    SET row_kills TO zero  
                    FOR each k FROM column_index TO number_of_columns MINUS one  
                        IF the element at position k of row at position row_index of grid EQUALS character ENEMY  
                            INCREMENT row_kills BY one  
                        ELSE IF the element at position k of row at position row_index of grid EQUALS character WALL  
                            BREAK the inner FOR loop  
                        END IF  
                    END FOR  
                END IF  
                
                IF row_index EQUALS zero OR the element at position row_index MINUS one of column at position column_index of grid EQUALS character WALL  
                    SET the element at position column_index of column_kills TO zero  
                    FOR each k FROM row_index TO number_of_rows MINUS one  
                        IF the element at position k of column at position column_index of grid EQUALS character ENEMY  
                            INCREMENT the element at position column_index of column_kills BY one  
                        ELSE IF the element at position k of column at position column_index of grid EQUALS character WALL  
                            BREAK the inner FOR loop  
                        END IF  
                    END FOR  
                END IF  
                
                IF the element at position column_index of row at position row_index of grid EQUALS character EMPTY  
                    SET maximum_kills TO the greater value BETWEEN maximum_kills AND the sum of row_kills AND the element at position column_index of column_kills  
                END IF  
            END FOR  
        END FOR  
        
        RETURN maximum_kills  
    END FUNCTION  
END CLASS
