CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1 c2)  
            IF c1 EQUALS None  
                RETURN zero  
            END IF  
            SET x1 TO the INTEGER DIVISION of the NUMERIC VALUE OF c1 MINUS the NUMERIC VALUE OF uppercase A BY six  
            SET y1 TO the REMAINDER of the NUMERIC VALUE OF c1 MINUS the NUMERIC VALUE OF uppercase A DIVIDED BY six  
            SET x2 TO the INTEGER DIVISION of the NUMERIC VALUE OF c2 MINUS the NUMERIC VALUE OF uppercase A BY six  
            SET y2 TO the REMAINDER of the NUMERIC VALUE OF c2 MINUS the NUMERIC VALUE OF uppercase A DIVIDED BY six  
            RETURN the ABSOLUTE VALUE OF x1 MINUS x2 PLUS the ABSOLUTE VALUE OF y1 MINUS y2  
        END FUNCTION  
        
        FUNCTION dp(i f1 f2)  
            IF i EQUALS the LENGTH OF word  
                RETURN zero  
            END IF  
            SET c TO the CHARACTER at position i of word  
            SET d1 TO distance of f1 AND c PLUS dp of i PLUS one AND c AND f2  
            SET d2 TO distance of f2 AND c PLUS dp of i PLUS one AND f1 AND c  
            IF d1 LESS THAN d2  
                RETURN d1  
            ELSE  
                RETURN d2  
            END IF  
        END FUNCTION  
        
        RETURN dp of zero AND None AND None  
    END FUNCTION  
END CLASS
