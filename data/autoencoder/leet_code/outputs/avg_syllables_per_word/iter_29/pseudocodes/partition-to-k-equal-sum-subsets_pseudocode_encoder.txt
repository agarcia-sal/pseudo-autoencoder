CLASS Solution  
    FUNCTION canPartitionKSubsets(nums PARAMETER k)  
        SET total_sum TO the sum of the elements of nums  
        IF the remainder of total_sum DIVIDED BY k IS NOT EQUAL TO zero  
            RETURN False  
        END IF  
        SET target TO total_sum DIVIDED BY k  
        SORT nums IN descending order  
        
        FUNCTION can_partition(index PARAMETER k PARAMETER current_sum PARAMETER used)  
            IF k EQUALS zero  
                RETURN True  
            END IF  
            IF current_sum EQUALS target  
                RETURN the result of calling can_partition with zero AS index AND k MINUS one AS k AND zero AS current_sum AND used AS used  
            END IF  
            
            FOR each i FROM index TO the LENGTH OF nums MINUS one  
                IF the value at position i of used IS False AND current_sum PLUS the element at position i of nums IS LESS THAN OR EQUAL TO target  
                    SET the value at position i of used TO True  
                    IF the result of calling can_partition with i PLUS one AS index AND k AS k AND current_sum PLUS the element at position i of nums AS current_sum AND used AS used IS True  
                        RETURN True  
                    END IF  
                    SET the value at position i of used TO False  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET used TO a list of Boolean False values of the same length as nums  
        RETURN the result of calling can_partition with zero AS index AND k AS k AND zero AS current_sum AND used AS used  
    END FUNCTION  
END CLASS
