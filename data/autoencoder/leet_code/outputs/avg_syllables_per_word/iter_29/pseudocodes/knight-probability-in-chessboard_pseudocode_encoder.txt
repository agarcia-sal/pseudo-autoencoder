CLASS Solution  
    FUNCTION knightProbability(n k row column)  
        SET moves TO the list of pairs consisting of negative two and negative one negative two and one negative one and negative two negative one and two one and negative two one and two two and negative one two and one  
        SET dp TO two dimensional list of size n by n FILLED WITH zero  
        SET element at position row and position column of dp TO one  
        FOR each move_count FROM one TO k  
            SET new_dp TO two dimensional list of size n by n FILLED WITH zero  
            FOR each r FROM zero TO n MINUS one  
                FOR each c FROM zero TO n MINUS one  
                    IF element at position r and position c of dp EQUALS zero THEN  
                        CONTINUE  
                    END IF  
                    FOR each pair dr dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n THEN  
                            INCREMENT element at position nr and position nc of new_dp BY element at position r and position c of dp DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        SET total_probability TO zero  
        FOR each row_list IN dp  
            FOR each probability IN row_list  
                INCREMENT total_probability BY probability  
            END FOR  
        END FOR  
        RETURN total_probability  
    END FUNCTION  
END CLASS
