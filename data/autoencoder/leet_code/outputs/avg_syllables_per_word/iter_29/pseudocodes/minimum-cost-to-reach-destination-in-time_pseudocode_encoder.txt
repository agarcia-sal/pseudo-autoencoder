CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO a list of empty lists repeated n times  
        FOR each x y time IN edges  
            APPEND the pair of y and time TO graph element at position x  
            APPEND the pair of x and time TO graph element at position y  
        END FOR  
        
        SET priority_queue TO a list containing a triple composed of element at position zero of passingFees the number zero and the number zero  
        SET visited_time TO a list containing the representation of infinity repeated n times  
        SET element at position zero of visited_time TO zero  
        
        WHILE priority_queue is not empty  
            EXTRACT the smallest element from priority_queue into current_cost current_city and current_time  
            
            IF current_city EQUALS n MINUS one THEN  
                RETURN current_cost  
            END IF  
            
            FOR each neighbor travel_time IN graph element at position current_city  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS passingFees element at position neighbor  
                
                IF new_time LESS THAN OR EQUAL TO maxTime AND the expression new_time LESS THAN element at position neighbor of visited_time OR new_cost LESS THAN current_cost holds THEN  
                    SET element at position neighbor of visited_time TO new_time  
                    ADD the triple composed of new_cost neighbor and new_time TO priority_queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the cardinal word negative one  
    END FUNCTION  
END CLASS
