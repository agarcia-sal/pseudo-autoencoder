CLASS Solution  
    FUNCTION findMaximizedCapital(k, w, profits, capital)  
        SET projects TO list of pairs of elements from capital and profits combining each element at matching position  
        SORT projects BY the first element of each pair representing capital required  
        
        SET max_heap TO empty collection representing a maximum priority queue of profits  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        
        FOR loop_variable FROM one TO k  
            WHILE i LESS THAN n AND first element of element at position i of projects LESS THAN OR EQUAL TO w  
                ADD the negative of the second element of element at position i of projects TO max_heap maintaining max-heap property  
                INCREMENT i BY one  
            END WHILE  
            
            IF max_heap IS empty  
                BREAK the loop  
            END IF  
            
            SET w TO w PLUS the negation of REMOVING the largest element FROM max_heap  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
