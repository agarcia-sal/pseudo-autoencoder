CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr list_of_integers k integer) RETURNS list_of_integers  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  

        FOR i FROM zero TO n MINUS one MINUS one  
            INSERT INTO min_heap a tuple containing the result of element at position i of arr DIVIDED BY element at position n MINUS one of arr followed by i followed by n MINUS one  
        END FOR  

        FOR count FROM zero TO k MINUS one MINUS one  
            REMOVE the smallest tuple FROM min_heap AND UNPACK INTO variables ignored numerator_index denominator_index  
            IF denominator_index MINUS one GREATER THAN numerator_index  
                INSERT INTO min_heap a tuple containing the result of element at position numerator_index of arr DIVIDED BY element at position denominator_index MINUS one of arr followed by numerator_index followed by denominator_index MINUS one  
            END IF  
        END FOR  

        REMOVE the smallest tuple FROM min_heap AND UNPACK INTO variables ignored numerator_index denominator_index  
        RETURN a list containing element at position numerator_index of arr FOLLOWED BY element at position denominator_index of arr  
    END FUNCTION  
END CLASS
