CLASS Solution  
    FUNCTION findReplaceString(s indices sources targets)  
        SET list_of_replacements TO a list of tuples each containing an element from indices an element from sources and an element from targets sorted by the first element representing the index  
        SET result_list TO an empty list  
        SET previous_end_index TO zero  
        FOR each tuple composed of current_index current_source and current_target IN list_of_replacements  
            APPEND the substring of s from position previous_end_index TO position current_index MINUS one inclusive TO result_list  
            IF the substring of s from position current_index TO position current_index PLUS the LENGTH OF current_source MINUS one inclusive EQUALS current_source  
                APPEND current_target TO result_list  
            ELSE  
                APPEND the substring of s from position current_index TO position current_index PLUS the LENGTH OF current_source MINUS one inclusive TO result_list  
            END IF  
            SET previous_end_index TO current_index PLUS the LENGTH OF current_source  
        END FOR  
        APPEND the substring of s from position previous_end_index TO the end of s TO result_list  
        RETURN the concatenation of all elements in result_list as a single string  
    END FUNCTION  
END CLASS
