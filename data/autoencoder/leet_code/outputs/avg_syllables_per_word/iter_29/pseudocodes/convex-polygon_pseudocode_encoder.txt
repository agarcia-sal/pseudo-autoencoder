CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(origin_point parameter a_point parameter b_point)  
            SET first_difference_x TO the element at position zero of a_point MINUS the element at position zero of origin_point  
            SET second_difference_y TO the element at position one of b_point MINUS the element at position one of origin_point  
            SET third_difference_y TO the element at position one of a_point MINUS the element at position one of origin_point  
            SET fourth_difference_x TO the element at position zero of b_point MINUS the element at position zero of origin_point  
            RETURN first_difference_x MULTIPLIED BY second_difference_y MINUS third_difference_y MULTIPLIED BY fourth_difference_x  
        END FUNCTION  
        
        SET number_of_points TO the LENGTH OF points  
        IF number_of_points LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  
        
        SET orientation TO zero  
        
        FOR index FROM zero TO number_of_points MINUS one  
            SET origin TO the element at position index of points  
            SET point_a TO the element at position the remainder of the sum of index PLUS one DIVIDED BY number_of_points of points  
            SET point_b TO the element at position the remainder of the sum of index PLUS two DIVIDED BY number_of_points of points  
            SET cross TO the result of cross_product WITH origin AND point_a AND point_b  
            
            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS negative one AND cross GREATER THAN zero  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
