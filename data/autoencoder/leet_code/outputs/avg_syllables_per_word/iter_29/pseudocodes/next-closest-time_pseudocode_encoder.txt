CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters obtained by removing the character colon from time  
        SET hour TO the integer value represented by the substring consisting of character at position one and character at position two of time respectively converted to number  
        SET minute TO the integer value represented by the substring consisting of character at position four and character at position five of time respectively converted to number  
        
        FUNCTION generate_times(digit_set)  
            SET times TO an empty list  
            FOR each character h1 IN digit_set  
                FOR each character h2 IN digit_set  
                    FOR each character m1 IN digit_set  
                        FOR each character m2 IN digit_set  
                            SET combined_hour TO the integer value of concatenating character h1 followed by character h2 converted to number  
                            SET combined_minute TO the integer value of concatenating character m1 followed by character m2 converted to number  
                            IF combined_hour LESS THAN twenty four AND combined_minute LESS THAN sixty  
                                APPEND the tuple consisting of combined_hour and combined_minute TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the list times sorted in ascending order  
        END FUNCTION  
        
        SET all_times TO the result of calling generate_times with digits  
        SET current_time_index TO the position of the tuple consisting of hour and minute in all_times  
        
        IF current_time_index PLUS one LESS THAN the length of all_times  
            SET next_hour TO the hour element of the tuple at position current_time_index PLUS one of all_times  
            SET next_minute TO the minute element of the tuple at position current_time_index PLUS one of all_times  
        ELSE  
            SET next_hour TO the hour element of the tuple at position zero of all_times  
            SET next_minute TO the minute element of the tuple at position zero of all_times  
        END IF  
        
        RETURN the string formed by converting next_hour to a two digit string including leading zero if needed concatenated with the character colon concatenated with the two digit string representation of next_minute including leading zero if needed  
    END FUNCTION  
END CLASS
