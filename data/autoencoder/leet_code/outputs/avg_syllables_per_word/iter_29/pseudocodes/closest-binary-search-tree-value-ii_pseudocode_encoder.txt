CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, left WITH DEFAULT None, right WITH DEFAULT None)  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values_list)  
    IF values_list IS empty OR values_list IS None THEN  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH value AS element at position one of values_list  
    SET index i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element REMOVED FROM queue  
        IF index i IS LESS THAN the LENGTH OF values_list AND element at position i of values_list IS NOT None THEN  
            SET node left TO NEW TreeNode WITH value AS element at position i of values_list  
            APPEND node left TO queue  
        END IF  
        INCREMENT index i BY one  
        IF index i IS LESS THAN the LENGTH OF values_list AND element at position i of values_list IS NOT None THEN  
            SET node right TO NEW TreeNode WITH value AS element at position i of values_list  
            APPEND node right TO queue  
        END IF  
        INCREMENT index i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF the value of p IS NOT EQUAL TO the value of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues(root, target_value, k)  
        FUNCTION inorder_traversal(node)  
            IF node IS None THEN  
                RETURN  
            END IF  
            CALL inorder_traversal WITH the left child of node  
            IF the LENGTH OF result IS LESS THAN k THEN  
                APPEND the value of node TO result  
            ELSE  
                IF the absolute value of the value of node MINUS target_value IS LESS THAN the absolute value of the element at position one of result MINUS target_value THEN  
                    REMOVE the first element FROM result  
                    APPEND the value of node TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal WITH the right child of node  
        END FUNCTION  
        
        SET result TO a new empty double ended queue  
        CALL inorder_traversal WITH root  
        RETURN a list CONVERTED FROM result  
    END FUNCTION  
END CLASS
