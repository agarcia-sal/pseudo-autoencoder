CLASS Solution  
    FUNCTION merge(intervals)  
        SORT intervals BY the first element of each interval in increasing order  
        SET merged TO an empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR the second element at the last interval in merged IS LESS THAN the first element of the current interval  
                APPEND interval TO merged  
            ELSE  
                SET the second element at the last interval in merged TO the greater VALUE BETWEEN the second element at the last interval in merged AND the second element in the current interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
