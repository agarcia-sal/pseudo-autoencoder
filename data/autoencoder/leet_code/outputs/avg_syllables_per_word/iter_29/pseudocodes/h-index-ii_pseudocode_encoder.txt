CLASS Solution  
    FUNCTION hIndex(citations)  
        SET total_citations TO the LENGTH OF citations  
        SET left TO zero  
        SET right TO total_citations MINUS one  
        SET h_index TO zero  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS the INTEGER DIVISION OF right MINUS left BY two  
            IF the element at position mid of citations IS GREATER THAN OR EQUAL TO total_citations MINUS mid  
                SET h_index TO total_citations MINUS mid  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        
        RETURN h_index  
    END FUNCTION  
END CLASS
