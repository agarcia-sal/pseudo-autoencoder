CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH parameter value as element at position one of values  
    SET i TO one  
    SET queue TO NEW double ended queue  
    APPEND root TO queue  
    WHILE queue is NOT empty  
        SET node TO the element REMOVED FROM the LEFT of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is NOT None  
            SET node.left TO NEW TreeNode WITH parameter value as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is NOT None  
            SET node.right TO NEW TreeNode WITH parameter value as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF value attribute of p NOT EQUALS value attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH parameters as left child attribute of p AND left child attribute of q  
        AND  
        the result of is_same_tree WITH parameters as right child attribute of p AND right child attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST(root, val)  
        FUNCTION dfs(root)  
            IF root IS None  
                RETURN NEW TreeNode WITH parameter value as val  
            END IF  
            IF value attribute of root LESS THAN val  
                SET right child attribute of root TO dfs WITH parameter as right child attribute of root  
            ELSE  
                SET left child attribute of root TO dfs WITH parameter as left child attribute of root  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN dfs WITH parameter root  
    END FUNCTION  
END CLASS
