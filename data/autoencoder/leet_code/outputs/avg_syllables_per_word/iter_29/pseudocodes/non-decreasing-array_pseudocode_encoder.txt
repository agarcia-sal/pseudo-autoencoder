CLASS Solution  
    FUNCTION checkPossibility(list_of_numbers)  
        SET modification_made TO false  
        FOR index FROM two TO the LENGTH OF list_of_numbers  
            IF the element at position index of list_of_numbers IS LESS THAN the element at position index MINUS one of list_of_numbers  
                IF modification_made IS true  
                    RETURN false  
                END IF  
                SET modification_made TO true  
                IF index IS LESS THAN two OR the element at position index of list_of_numbers IS GREATER THAN OR EQUAL TO the element at position index MINUS two of list_of_numbers  
                    SET the element at position index MINUS one of list_of_numbers TO the element at position index of list_of_numbers  
                ELSE  
                    SET the element at position index of list_of_numbers TO the element at position index MINUS one of list_of_numbers  
                END IF  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
