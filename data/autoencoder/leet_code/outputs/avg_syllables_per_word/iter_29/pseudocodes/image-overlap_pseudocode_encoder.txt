CLASS Solution  
    FUNCTION largestOverlap(img1, img2)  
        SET matrix_size TO the LENGTH OF img1  
        SET list_of_points_one TO an empty list  
        FOR each row_index FROM zero TO matrix_size MINUS one  
            FOR each column_index FROM zero TO matrix_size MINUS one  
                IF the element at position row_index of img1 at position column_index EQUALS one  
                    APPEND the pair of row_index AND column_index TO list_of_points_one  
                END IF  
            END FOR  
        END FOR  
        SET list_of_points_two TO an empty list  
        FOR each row_index FROM zero TO matrix_size MINUS one  
            FOR each column_index FROM zero TO matrix_size MINUS one  
                IF the element at position row_index of img2 at position column_index EQUALS one  
                    APPEND the pair of row_index AND column_index TO list_of_points_two  
                END IF  
            END FOR  
        END FOR  
        SET dictionary_of_overlap_counts TO a mapping from pairs TO integers defaulting to zero  
        FOR each pair x1 AND y1 IN list_of_points_one  
            FOR each pair x2 AND y2 IN list_of_points_two  
                SET delta_x TO x2 MINUS x1  
                SET delta_y TO y2 MINUS y1  
                INCREMENT the value at key of delta_x AND delta_y IN dictionary_of_overlap_counts BY one  
            END FOR  
        END FOR  
        IF dictionary_of_overlap_counts IS empty  
            RETURN zero  
        ELSE  
            RETURN the maximum VALUE in dictionary_of_overlap_counts  
        END IF  
    END FUNCTION  
END CLASS
