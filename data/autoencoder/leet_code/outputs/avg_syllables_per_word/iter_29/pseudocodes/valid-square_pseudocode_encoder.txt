CLASS Solution  
    FUNCTION validSquare(p1 parameter list_of_integers p2 parameter list_of_integers p3 parameter list_of_integers p4 parameter list_of_integers) RETURNS boolean  
        FUNCTION squared_distance(point_one parameter list_of_integers point_two parameter list_of_integers)  
            SET difference_in_first_coordinate TO element at position zero of point_one MINUS element at position zero of point_two  
            SET difference_in_second_coordinate TO element at position one of point_one MINUS element at position one of point_two  
            SET square_of_first_difference TO difference_in_first_coordinate POWERED BY two  
            SET square_of_second_difference TO difference_in_second_coordinate POWERED BY two  
            SET sum_of_squares TO square_of_first_difference PLUS square_of_second_difference  
            RETURN sum_of_squares  
        END FUNCTION  
        
        SET points TO list composed of p1 p2 p3 p4  
        SET distances TO empty set  
        
        FOR i FROM zero TO three  
            FOR j FROM i PLUS one TO three  
                SET current_distance TO CALL squared_distance WITH element at position i of points AND element at position j of points  
                IF current_distance EQUALS zero THEN  
                    RETURN boolean false  
                END IF  
                ADD current_distance TO distances  
            END FOR  
        END FOR  
        
        IF the number of unique elements in distances EQUALS two THEN  
            RETURN boolean true  
        ELSE  
            RETURN boolean false  
        END IF  
    END FUNCTION  
END CLASS
