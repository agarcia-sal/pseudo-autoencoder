CLASS Solution  
    FUNCTION countRangeSum(nums lower upper)  
        SET prefix_sums TO a list containing zero  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            INCREMENT current_sum BY num  
            
            SET left TO current_sum MINUS upper  
            SET right TO current_sum MINUS lower  
            
            SET count TO count PLUS the DIFFERENCE BETWEEN the POSITION WHERE right WOULD BE INSERTED TO THE RIGHT IN prefix_sums AND the POSITION WHERE left WOULD BE INSERTED TO THE LEFT IN prefix_sums  
            
            INSERT current_sum INTO prefix_sums IN ORDERED POSITION  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
