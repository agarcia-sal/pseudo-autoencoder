CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET ones TO zero  
        SET twos TO zero  
        FOR each number IN nums  
            SET ones TO the result of the bitwise exclusive OR of ones AND number AND the bitwise NOT of twos  
            SET twos TO the result of the bitwise exclusive OR of twos AND number AND the bitwise NOT of ones  
        END FOR  
        RETURN ones  
    END FUNCTION  
END CLASS
