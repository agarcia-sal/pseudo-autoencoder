CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF the LENGTH OF nums LESS THAN two THEN  
            RETURN zero  
        END IF  

        FUNCTION countingSort(arr, exp)  
            SET n TO the LENGTH OF arr  
            SET output TO a list of zeros with LENGTH n  
            SET count TO a list of zeros with LENGTH ten  

            FOR each index FROM zero TO n MINUS one  
                SET digit_index TO the INTEGER DIVISION of element at position index of arr DIVIDED BY exp  
                INCREMENT element at position digit_index MODULO ten of count BY one  
            END FOR  

            FOR each index FROM one TO nine  
                INCREMENT element at position index of count BY element at position index MINUS one of count  
            END FOR  

            SET i TO n MINUS one  
            WHILE i GREATER THAN OR EQUAL TO zero DO  
                SET digit_index TO the INTEGER DIVISION of element at position i of arr DIVIDED BY exp  
                SET element at position element at position digit_index MODULO ten of count MINUS one of output TO element at position i of arr  
                DECREMENT element at position digit_index MODULO ten of count BY one  
                DECREMENT i BY one  
            END WHILE  

            FOR each index FROM zero TO n MINUS one  
                SET element at position index of arr TO element at position index of output  
            END FOR  
        END FUNCTION  

        FUNCTION radixSort(arr)  
            SET max1 TO the maximum element of arr  
            SET exp TO one  
            WHILE the INTEGER DIVISION of max1 BY exp GREATER THAN zero DO  
                CALL countingSort WITH arr AND exp  
                SET exp TO exp MULTIPLIED BY ten  
            END WHILE  
        END FUNCTION  

        CALL radixSort WITH nums  

        SET max_diff TO zero  
        FOR each index FROM one TO the LENGTH OF nums MINUS one  
            SET current_difference TO element at position index of nums MINUS element at position index MINUS one of nums  
            IF current_difference GREATER THAN max_diff THEN  
                SET max_diff TO current_difference  
            END IF  
        END FOR  

        RETURN max_diff  
    END FUNCTION  
END CLASS
