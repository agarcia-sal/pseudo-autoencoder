CLASS Solution  
    FUNCTION multiply(mat1 matrix_of_lists_of_integers mat2 matrix_of_lists_of_integers) RETURNS matrix_of_lists_of_integers  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF element at position zero of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF element at position zero of mat2  
        SET result TO a new matrix with m rows AND n columns filled with zero  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                FOR index p FROM zero TO k MINUS one  
                    INCREMENT element at position i of result at position j BY element at position i of mat1 at position p MULTIPLIED BY element at position p of mat2 at position j  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
