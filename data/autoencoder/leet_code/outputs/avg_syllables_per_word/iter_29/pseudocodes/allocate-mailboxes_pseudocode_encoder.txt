CLASS Solution  
    FUNCTION minDistance(houses LIST OF integer PARAMETERS, k integer PARAMETER)  
        CALL sort ON houses  
        SET n TO the LENGTH OF houses  
          
        SET cost TO a two dimensional list OF zeroes WITH n BY n dimensions  
          
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost element at position i and position j TO the SUM OF cost element at position i PLUS one and position j MINUS one PLUS the VALUE AT position j OF houses MINUS the VALUE AT position i OF houses  
            END FOR  
        END FOR  
          
        SET dp TO a two dimensional list FILLED WITH infinite VALUES WITH n PLUS one BY k PLUS one dimensions  
        SET dp element at position zero and position zero TO zero  
          
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET candidate_value TO dp element at position p and position j MINUS one PLUS cost element at position p and position i MINUS one  
                    IF candidate_value LESS THAN dp element at position i and position j  
                        SET dp element at position i and position j TO candidate_value  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
          
        RETURN dp element at position n and position k  
    END FUNCTION  
END CLASS
