CLASS Solution  
    FUNCTION findLonelyPixel(picture)  
        IF picture EQUALS nothing OR the LENGTH OF picture EQUALS zero OR the LENGTH OF element at position one of picture EQUALS zero  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position one of picture  
        SET row_count TO a new list of length m FILLED WITH zero  
        SET col_count TO a new list of length n FILLED WITH zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i of picture and position j of element at position i of picture EQUALS the character representing black pixel  
                    INCREMENT the element at position i of row_count BY one  
                    INCREMENT the element at position j of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        SET lonely_pixel_count TO zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i of picture and position j of element at position i of picture EQUALS the character representing black pixel AND the element at position i of row_count EQUALS one AND the element at position j of col_count EQUALS one  
                    INCREMENT lonely_pixel_count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
