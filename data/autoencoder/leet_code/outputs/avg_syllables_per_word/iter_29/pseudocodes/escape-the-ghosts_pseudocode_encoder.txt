CLASS Solution  
    FUNCTION escapeGhosts(ghosts, target)  
        SET player_distance TO the absolute value of character at position zero of target PLUS the absolute value of character at position one of target  
        FOR each ghost IN ghosts  
            SET ghost_distance TO the absolute value of the difference between character at position zero of ghost AND character at position zero of target PLUS the absolute value of the difference between character at position one of ghost AND character at position one of target  
            IF ghost_distance LESS THAN OR EQUAL TO player_distance  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
