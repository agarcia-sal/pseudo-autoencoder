CLASS TreeNode  
    FUNCTION __init__(val WITH DEFAULT zero left WITH DEFAULT none right WITH DEFAULT none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed with the element at position one of values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element removed from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO a new TreeNode constructed with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO a new TreeNode constructed with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val IS NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p dot left, q dot left) AND is_same_tree(p dot right, q dot right)  
    END ELSE  
END FUNCTION

CLASS Solution  
    FUNCTION deleteNode(root, key)  
        IF root IS none  
            RETURN none  
        END IF  
        IF root dot val IS GREATER THAN key  
            SET root dot left TO the result of calling deleteNode with root dot left and key  
            RETURN root  
        END IF  
        IF root dot val IS LESS THAN key  
            SET root dot right TO the result of calling deleteNode with root dot right and key  
            RETURN root  
        END IF  
        IF root dot left IS none  
            RETURN root dot right  
        END IF  
        IF root dot right IS none  
            RETURN root dot left  
        END IF  
        SET node TO root dot right  
        WHILE node dot left IS NOT none  
            SET node TO node dot left  
        END WHILE  
        SET node dot left TO root dot left  
        SET root TO root dot right  
        RETURN root  
    END FUNCTION  
END CLASS
