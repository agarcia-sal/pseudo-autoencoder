CLASS Solution  
    FUNCTION shortestPath(grid k)  
        IF grid IS empty OR element at position zero of grid IS empty  
            RETURN the negative of one  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        IF k IS GREATER THAN OR EQUAL TO m PLUS n MINUS two  
            RETURN m PLUS n MINUS two  
        END IF  
        
        SET directions TO a list containing the pair of zero and one the pair of one and zero the pair of zero and negative one and the pair of negative one and zero  
        
        SET queue TO a new empty double ended queue initialized with a tuple consisting of zero zero k and zero  
        
        SET visited TO a new set initialized with a tuple consisting of zero zero and k  
        
        WHILE queue is not empty  
            REMOVE and ASSIGN the first element of queue TO x y remaining_k and steps  
            
            IF x EQUALS m MINUS one AND y EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET new_k TO remaining_k MINUS element at position nx and ny of grid  
                    IF new_k IS GREATER THAN OR EQUAL TO zero AND the tuple consisting of nx ny and new_k IS NOT IN visited  
                        ADD the tuple consisting of nx ny and new_k TO visited  
                        APPEND the tuple consisting of nx ny new_k and steps PLUS one TO queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the negative of one  
    END FUNCTION  
END CLASS
