CLASS TreeNode  
    FUNCTION __init__ WITH val AND left AND right  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH values  
    IF the values collection is empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode initialized with the element at position one of values  
    SET i TO one  
    INITIALIZE queue as a new deque  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i is less than the length of values AND the element at position i of values is not None  
            SET node dot left TO a new TreeNode initialized with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not None  
            SET node dot right TO a new TreeNode initialized with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH p AND q  
    IF p is None AND q is None  
        RETURN True  
    ELSE IF p is None OR q is None  
        RETURN False  
    ELSE IF the value attribute of p is not equal to the value attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree applied to the left attribute of p AND the left attribute of q  
        AND the result of is_same_tree applied to the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree WITH root  
        
        FUNCTION dfs WITH node  
            IF node is None  
                RETURN positive infinity FOR the minimum value AND negative infinity FOR the maximum value AND zero FOR the size  
            END IF  
            SET left_min AND left_max AND left_size TO the result of dfs applied to the left attribute of node  
            SET right_min AND right_max AND right_size TO the result of dfs applied to the right attribute of node  
            IF the maximum value from the left is less than the value attribute of node AND the value attribute of node is less than the minimum value from the right  
                RETURN the lesser value between the value attribute of node AND the left minimum value  
                AND the greater value between the value attribute of node AND the right maximum value  
                AND the sum of left_size AND right_size AND one  
            END IF  
            RETURN negative infinity FOR the minimum value AND positive infinity FOR the maximum value  
            AND the greater value between left_size AND right_size  
        END FUNCTION  
        
        RETURN the third element of the tuple returned by dfs applied to root  
    END FUNCTION  
END CLASS
