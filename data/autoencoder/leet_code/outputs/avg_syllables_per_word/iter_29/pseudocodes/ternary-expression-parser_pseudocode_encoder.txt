CLASS Solution  
    FUNCTION parseTernary(expression)  
        FUNCTION evaluate(index)  
            IF index EQUALS the length of expression MINUS one OR the character at position index PLUS one of expression EQUALS the character colon  
                RETURN the character at position index of expression AND the index PLUS two  
            END IF  
            SET true_part AND next_index TO the result of calling evaluate with the argument index PLUS two  
            SET false_part AND final_index TO the result of calling evaluate with the argument next_index  
            IF the character at position index of expression EQUALS the character T  
                RETURN true_part AND final_index  
            ELSE  
                RETURN false_part AND final_index  
            END ELSE  
        END FUNCTION  
        SET result AND an unused_value TO the result of calling evaluate with zero  
        RETURN result  
    END FUNCTION  
END CLASS
