CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO a default dictionary that maps values to empty lists  
        SET i TO zero  
        WHILE i LESS THAN n  
            SET value TO element at position i of arr  
            APPEND i TO the list mapped by value in value_indices  
            INCREMENT i BY one  
        END WHILE  
        
        SET queue TO a double ended queue initialized with a pair of zero and zero representing current index and current steps  
        SET visited TO a set containing zero  
        
        WHILE queue is not empty  
            REMOVE the first element from queue and ASSIGN current_index and steps from that element  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO a new list composed of current_index PLUS one, current_index MINUS one, and all indices mapped by the value at position current_index of arr in value_indices  
            
            FOR each next_index IN next_indices  
                IF next_index GREATER THAN OR EQUAL TO zero AND next_index LESS THAN n AND next_index NOT IN visited  
                    ADD next_index TO visited  
                    APPEND the pair of next_index and steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list mapped by the value at position current_index of arr in value_indices TO an empty list  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
