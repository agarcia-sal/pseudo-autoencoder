CLASS Solution  
    FUNCTION isInterleave(s1 string_parameter s2 string_parameter s3 string_parameter) RETURNS boolean  
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 NOT EQUALS the LENGTH OF s3  
            RETURN false  
        END IF  

        SET dp TO a list of boolean values of size the LENGTH OF s2 PLUS one with all values set to false  
        SET the element at position zero of dp TO true  

        FOR i FROM zero TO the LENGTH OF s1  
            FOR j FROM zero TO the LENGTH OF s2  
                IF i GREATER THAN zero  
                    SET the element at position j of dp TO the logical AND of the current element at position j of dp AND the condition that the character at position i MINUS one of s1 EQUALS the character at position i PLUS j MINUS one of s3  
                END IF  
                IF j GREATER THAN zero  
                    SET the element at position j of dp TO the logical OR of the current element at position j of dp AND the logical AND of the element at position j MINUS one of dp AND the condition that the character at position j MINUS one of s2 EQUALS the character at position i PLUS j MINUS one of s3  
                END IF  
            END FOR  
        END FOR  

        RETURN the element at position the LENGTH OF s2 of dp  
    END FUNCTION  
END CLASS
