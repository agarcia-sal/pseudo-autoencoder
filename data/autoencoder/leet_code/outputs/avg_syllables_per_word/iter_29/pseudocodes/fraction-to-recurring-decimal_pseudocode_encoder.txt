CLASS Solution  
    FUNCTION fractionToDecimal(numerator denominator)  
        IF (numerator LESS THAN zero) XOR (denominator LESS THAN zero)  
            SET sign TO the string containing only the minus symbol  
        ELSE  
            SET sign TO the empty string  
        END IF  

        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  

        SET quotient TO the integer result of numerator DIVIDED BY denominator  
        SET remainder TO the remainder after numerator DIVIDED BY denominator  
        SET integer_part TO the string representation of quotient  

        IF remainder EQUALS zero  
            RETURN the concatenation of sign with integer_part  
        END IF  

        SET seen_remainders TO an empty dictionary mapping integers to positions  
        SET fractional_part TO an empty list of characters  

        WHILE remainder NOT EQUALS zero  
            IF remainder IS a key in seen_remainders  
                SET start TO the position mapped to remainder in seen_remainders  
                SET non_repeating TO the concatenation of all characters in fractional_part from position zero UP TO but NOT INCLUDING position start  
                SET repeating TO the concatenation of all characters in fractional_part from position start TO the end  
                RETURN the concatenation of sign with integer_part with the full stop character and non_repeating and the opening parenthesis character and repeating and the closing parenthesis character  
            END IF  

            SET in seen_remainders at remainder TO the current length of fractional_part  

            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO the integer result of remainder DIVIDED BY denominator  
            SET remainder TO the remainder after remainder DIVIDED BY denominator  
            APPEND the string representation of quotient TO fractional_part  
        END WHILE  

        RETURN the concatenation of sign with integer_part with the full stop character and the concatenation of all characters in fractional_part  
    END FUNCTION  
END CLASS
