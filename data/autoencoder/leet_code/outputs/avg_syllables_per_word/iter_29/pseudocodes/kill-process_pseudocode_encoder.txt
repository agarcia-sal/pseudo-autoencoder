CLASS Solution  
    FUNCTION killProcess(pid list_of_integers ppid list_of_integers kill integer) RETURNS list_of_integers  
        SET tree TO a dictionary where each key maps to an empty list  
        FOR each child and parent taken together from the list_of_integers pid and the list_of_integers ppid in order  
            APPEND child TO the list in tree associated WITH parent  
        END FOR  
        SET queue TO a new list containing the single element kill  
        SET killed_processes TO a new empty list  
        WHILE the list queue is NOT empty  
            REMOVE the element at position zero of queue AND ASSIGN it TO current  
            APPEND current TO the list killed_processes  
            EXTEND the list queue BY all elements in the list associated WITH current in tree  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
