CLASS Solution  
    FUNCTION fallingSquares(positions)  
        SET max_heights TO an empty list  
        SET intervals TO an empty list  
        SET current_max_height TO zero  
        FOR each position IN positions  
            SET left TO the first element of position  
            SET side_length TO the second element of position  
            SET right TO left PLUS side_length MINUS one  
            SET height TO side_length  
            FOR each interval IN intervals  
                SET interval_left TO the first element of interval  
                SET interval_right TO the second element of interval  
                SET interval_height TO the third element of interval  
                IF left LESS THAN OR EQUAL TO interval_right AND right GREATER THAN OR EQUAL TO interval_left  
                    SET height TO the GREATER OF height AND side_length PLUS interval_height  
                END IF  
            END FOR  
            APPEND a tuple consisting of left right and height TO intervals  
            SET current_max_height TO the GREATER OF current_max_height AND height  
            APPEND current_max_height TO max_heights  
        END FOR  
        RETURN max_heights  
    END FUNCTION  
END CLASS
