CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO positive infinity  
        FOR each number IN nums  
            IF remainder of number DIVIDED BY two EQUALS one THEN  
                SET number TO number MULTIPLIED BY two  
            END IF  
            SET min_value TO the smaller value BETWEEN min_value AND number  
            ADD the negative of number TO max_heap  
        END FOR  
        
        SET min_deviation TO positive infinity  
        
        WHILE remainder of element at position zero of max_heap DIVIDED BY two EQUALS zero  
            SET max_value TO the negative of the result of removing and returning the largest element from max_heap  
            SET min_deviation TO the smaller value BETWEEN min_deviation AND the difference BETWEEN max_value AND min_value  
            SET max_value TO the quotient of max_value DIVIDED BY two rounded down to the nearest whole number  
            SET min_value TO the smaller value BETWEEN min_value AND max_value  
            ADD the negative of max_value TO max_heap  
        END WHILE  
        
        SET min_deviation TO the smaller value BETWEEN min_deviation AND the difference BETWEEN the negative of element at position zero of max_heap AND min_value  
        
        RETURN min_deviation  
    END FUNCTION  
END CLASS
