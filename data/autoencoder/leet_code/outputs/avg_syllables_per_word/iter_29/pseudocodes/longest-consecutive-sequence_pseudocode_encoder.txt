CLASS Solution  
    FUNCTION longestConsecutive(nums)  
        IF nums IS EMPTY  
            RETURN zero  
        END IF  
        SET num_set TO the SET OF elements IN nums  
        SET longest_streak TO zero  
        FOR each number IN num_set  
            IF the ELEMENT obtained by SUBTRACTING one FROM number IS NOT IN num_set  
                SET current_num TO number  
                SET current_streak TO one  
                WHILE the ELEMENT obtained by ADDING one TO current_num IS IN num_set  
                    INCREMENT current_num BY one  
                    INCREMENT current_streak BY one  
                END WHILE  
                IF current_streak IS GREATER THAN longest_streak  
                    SET longest_streak TO current_streak  
                END IF  
            END IF  
        END FOR  
        RETURN longest_streak  
    END FUNCTION  
END CLASS
