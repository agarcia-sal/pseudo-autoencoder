CLASS Solution  
    FUNCTION minDistance(height width tree squirrel nuts)  
        FUNCTION distance(point_one point_two)  
            RETURN the absolute value of the difference between element at position zero of point_one AND element at position zero of point_two PLUS the absolute value of the difference between element at position one of point_one AND element at position one of point_two  
        END FUNCTION  
        
        SET total_distance TO zero  
        SET max_save TO negative infinity  
        
        FOR each nut IN nuts  
            INCREMENT total_distance BY two MULTIPLIED BY distance(nut tree)  
            
            SET first_nut_distance TO distance(squirrel nut) PLUS distance(nut tree)  
            
            SET later_nut_distance TO two MULTIPLIED BY distance(nut tree)  
            
            SET save TO later_nut_distance MINUS first_nut_distance  
            
            SET max_save TO the greater value BETWEEN max_save AND save  
        END FOR  
        
        RETURN total_distance MINUS max_save  
    END FUNCTION  
END CLASS
