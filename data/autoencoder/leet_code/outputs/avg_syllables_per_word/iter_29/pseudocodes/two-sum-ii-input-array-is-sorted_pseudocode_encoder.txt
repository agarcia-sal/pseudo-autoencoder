CLASS Solution  
    FUNCTION twoSum(numbers_list OF integers, target_value OF integer) RETURNS list of integers  
        SET left TO zero  
        SET right TO the LENGTH OF numbers_list MINUS one  
        WHILE left IS LESS THAN right  
            SET current_sum TO element at position left of numbers_list PLUS element at position right of numbers_list  
            IF current_sum EQUALS target_value  
                RETURN list containing left PLUS one AND right PLUS one  
            ELSE IF current_sum IS LESS THAN target_value  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END ELSE  
        END WHILE  
    END FUNCTION  
END CLASS
