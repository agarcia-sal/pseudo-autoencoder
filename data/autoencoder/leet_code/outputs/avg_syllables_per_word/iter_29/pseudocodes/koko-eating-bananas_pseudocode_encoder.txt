CLASS Solution  
    FUNCTION minEatingSpeed(piles parameter, h parameter)  
        FUNCTION canFinish(k parameter)  
            SET hours TO zero  
            FOR each pile IN piles parameter  
                INCREMENT hours BY the smallest integer greater than or equal to the result of pile DIVIDED BY k parameter  
            END FOR  
            RETURN hours LESS THAN OR EQUAL TO h parameter  
        END FUNCTION  
        
        SET left TO one  
        SET right TO the maximum value among elements IN piles parameter  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the integer division of the sum of left AND right BY two  
            IF canFinish(mid)  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
