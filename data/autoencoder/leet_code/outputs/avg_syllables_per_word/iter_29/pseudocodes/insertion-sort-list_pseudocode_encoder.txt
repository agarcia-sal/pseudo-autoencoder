CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the value property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN nothing  
    END IF  
    SET head TO a new instance of ListNode initialized WITH the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting FROM the second element TO the last element  
        SET node TO a new instance of ListNode initialized WITH val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS nothing AND p2 IS nothing  
        RETURN true  
    END IF  
    IF p1 IS nothing OR p2 IS nothing  
        RETURN false  
    END IF  
    RETURN the comparison that the value property of p1 EQUALS the value property of p2 AND recursive call to is_same_list WITH the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList WITH PARAMETERS head  
        IF head IS nothing OR the next property of head IS nothing  
            RETURN head  
        END IF  
        
        SET dummy TO a new instance of ListNode initialized WITH a value much less than any possible node value  
        SET the next property of dummy TO head  
        SET current TO the next property of head  
        SET the next property of head TO nothing  
        
        WHILE current IS NOT nothing  
            SET next_node TO the next property of current  
            
            SET prev TO dummy  
            WHILE the next property of prev IS NOT nothing AND the value property of the next property of prev IS LESS THAN the value property of current  
                SET prev TO the next property of prev  
            END WHILE  
            
            SET the next property of current TO the next property of prev  
            SET the next property of prev TO current  
            
            SET current TO next_node  
        END WHILE  
        
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
