CLASS Solution  
    FUNCTION bestRotation(nums)  
        SET n TO the LENGTH OF nums  
        SET delta TO a list of zeroes of LENGTH n  
        FOR each index i AND element num IN nums WITH enumeration  
            IF num LESS THAN OR EQUAL TO i  
                INCREMENT the element at position zero OF delta BY one  
                IF the result of i MINUS num PLUS one LESS THAN n  
                    DECREMENT the element at position the result of i MINUS num PLUS one OF delta BY one  
                END IF  
            END IF  
            IF i PLUS one LESS THAN n  
                INCREMENT the element at position i PLUS one OF delta BY one  
                IF the result of i PLUS n MINUS num PLUS one LESS THAN n  
                    DECREMENT the element at position the result of i PLUS n MINUS num PLUS one OF delta BY one  
                END IF  
            END IF  
        END FOR  
        SET score TO zero  
        SET max_score TO zero  
        SET best_k TO zero  
        FOR each k FROM zero TO n MINUS one  
            INCREMENT score BY the element at position k OF delta  
            IF score GREATER THAN max_score  
                SET max_score TO score  
                SET best_k TO k  
            END IF  
        END FOR  
        RETURN best_k  
    END FUNCTION  
END CLASS
