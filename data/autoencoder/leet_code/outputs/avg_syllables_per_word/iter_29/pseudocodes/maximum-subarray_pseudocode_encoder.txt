CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET maximum sum TO element at position one of nums  
        SET current sum TO element at position one of nums  
        FOR each number IN nums starting from position two  
            SET current sum TO the greater value BETWEEN number AND current sum PLUS number  
            SET maximum sum TO the greater value BETWEEN maximum sum AND current sum  
        END FOR  
        RETURN maximum sum  
    END FUNCTION  
END CLASS
