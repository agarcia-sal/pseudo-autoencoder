CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET list_of_reversed_words TO the SORTED list CREATED BY reversing each word IN words  
        SET total_length TO zero  
        SET number_of_reversed_words TO the LENGTH OF list_of_reversed_words  
        FOR index FROM zero TO number_of_reversed_words MINUS one  
            IF index PLUS one EQUALS number_of_reversed_words OR the word at position index PLUS one of list_of_reversed_words DOES NOT START WITH the word at position index of list_of_reversed_words  
                INCREMENT total_length BY the LENGTH OF the word at position index of list_of_reversed_words PLUS one  
            END IF  
        END FOR  
        RETURN total_length  
    END FUNCTION  
END CLASS
