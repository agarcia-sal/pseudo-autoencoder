CLASS Solution  
    FUNCTION validSubarrays(nums)  
        SET stack TO an empty list  
        SET count TO zero  
        FOR each number IN nums  
            WHILE the stack IS NOT empty AND the last element in the stack IS GREATER THAN number  
                REMOVE the last element FROM the stack  
            END WHILE  
            APPEND number TO the stack  
            INCREMENT count BY the LENGTH OF the stack  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
