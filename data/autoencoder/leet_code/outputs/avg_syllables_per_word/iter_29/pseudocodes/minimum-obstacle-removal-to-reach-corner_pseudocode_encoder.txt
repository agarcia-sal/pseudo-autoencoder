CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO the LIST OF four pairs each containing zero and one or one and zero or zero and negative one or negative one and zero respectively  
        SET min_heap TO the LIST CONTAINING a TRIPLE of zero zero and zero representing cost and x coordinate and y coordinate  
        SET visited TO a SET CONTAINING the PAIR of zero and zero  
        
        WHILE min_heap is NOT empty  
            REMOVE AND RETURN the SMALLEST element from min_heap ASSIGNING its values TO cost x and y  
            
            IF the PAIR of x and y EQUALS the PAIR of m MINUS one and n MINUS one THEN  
                RETURN cost  
            END IF  
            
            FOR each dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the PAIR of nx and ny NOT IN visited THEN  
                    ADD the PAIR of nx and ny TO visited  
                    SET new_cost TO cost PLUS element at position nx of grid AT position ny  
                    INSERT the TRIPLE of new_cost nx and ny INTO min_heap MAINTAINING HEAP PROPERTY  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
