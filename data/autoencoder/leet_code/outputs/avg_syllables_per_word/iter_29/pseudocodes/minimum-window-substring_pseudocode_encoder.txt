CLASS Solution  
    FUNCTION minWindow(s string_parameter t string_parameter) RETURNS string_parameter  
        SET t_count TO the count of each character contained in t string_parameter  
        SET required TO the LENGTH OF the set of unique keys in t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO the count of each character currently in the window as empty initially  
        SET min_length TO positive infinity  
        SET min_start TO zero  
        WHILE right IS LESS THAN the LENGTH OF s string_parameter  
            INCREMENT the count of character at position right of s string_parameter IN window_counts BY one  
            IF the character at position right of s string_parameter IS A KEY IN t_count AND the count of character at position right of s string_parameter IN window_counts EQUALS the count of character at position right of s string_parameter IN t_count THEN  
                INCREMENT formed BY one  
            END IF  
            WHILE left IS LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO the character at position left of s string_parameter  
                IF right MINUS left PLUS one IS LESS THAN min_length THEN  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                DECREMENT the count of char IN window_counts BY one  
                IF char IS A KEY IN t_count AND the count of char IN window_counts IS LESS THAN the count of char IN t_count THEN  
                    DECREMENT formed BY one  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            INCREMENT right BY one  
        END WHILE  
        IF min_length EQUALS positive infinity THEN  
            RETURN an empty string  
        ELSE  
            RETURN the substring of s string_parameter from position min_start TO position min_start PLUS min_length MINUS one  
        END IF  
    END FUNCTION  
END CLASS
