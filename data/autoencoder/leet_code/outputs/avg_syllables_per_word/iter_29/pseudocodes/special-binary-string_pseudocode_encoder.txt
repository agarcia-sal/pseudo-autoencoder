CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR index AND character IN position AND character OF s  
                IF character EQUALS the character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    SET substring_special TO the character one CONCATENATED WITH the RESULT OF CALLING dfs WITH the substring from position start PLUS one TO position index MINUS one OF s CONCATENATED WITH the character zero  
                    APPEND substring_special TO specials  
                    SET start TO index PLUS one  
                END IF  
            END FOR  
            SORT specials IN reverse lexicographical order  
            RETURN the concatenation of all elements IN specials WITH no delimiter  
        END FUNCTION  
        RETURN dfs WITH argument s  
    END FUNCTION  
END CLASS
