CLASS Solution  
    FUNCTION rotate(nums LIST OF integers k integer)  
        SET n TO the LENGTH OF nums  
        SET k TO k MODULO n  
        
        FUNCTION reverse(start integer end integer)  
            WHILE start IS LESS THAN end  
                SET temporary_value TO element at position start of nums  
                SET element at position start of nums TO element at position end of nums  
                SET element at position end of nums TO temporary_value  
                INCREMENT start BY one  
                DECREMENT end BY one  
            END WHILE  
        END FUNCTION  
        
        CALL reverse WITH start AS zero AND end AS n MINUS one  
        CALL reverse WITH start AS zero AND end AS k MINUS one  
        CALL reverse WITH start AS k AND end AS n MINUS one  
    END FUNCTION  
END CLASS
