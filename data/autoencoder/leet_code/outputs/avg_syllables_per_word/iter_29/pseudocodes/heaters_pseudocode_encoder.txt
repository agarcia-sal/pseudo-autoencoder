CLASS Solution  
    FUNCTION findRadius(houses, heaters)  
        CALL sort on houses  
        CALL sort on heaters  
        
        SET min_radius TO zero  
        SET j TO zero  
        
        FOR each house IN houses  
            WHILE j LESS THAN the LENGTH OF heaters MINUS one AND the ABSOLUTE VALUE OF the difference BETWEEN element at position j of heaters AND house GREATER THAN OR EQUAL TO the ABSOLUTE VALUE OF the difference BETWEEN element at position j PLUS one of heaters AND house  
                INCREMENT j BY one  
            END WHILE  
            
            SET min_radius TO the GREATER VALUE BETWEEN min_radius AND the ABSOLUTE VALUE OF the difference BETWEEN element at position j of heaters AND house  
        END FOR  
        
        RETURN min_radius  
    END FUNCTION  
END CLASS
