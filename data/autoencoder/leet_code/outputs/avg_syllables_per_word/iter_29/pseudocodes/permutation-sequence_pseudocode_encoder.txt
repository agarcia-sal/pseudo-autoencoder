CLASS Solution  
    FUNCTION getPermutation(n k)  
        IMPORT mathematical calculations  
        SET list_of_numbers TO list containing elements from one TO n  
        SET adjusted_k TO k MINUS one  
        SET result_list TO empty list  
        FOR i FROM n TO one DECREMENT BY one  
            SET factorial_value TO factorial of i MINUS one USING mathematical calculations  
            SET index_value TO adjusted_k DIVIDED BY factorial_value  
            APPEND element at position index_value of list_of_numbers TO result_list  
            REMOVE element at position index_value FROM list_of_numbers  
            SET adjusted_k TO adjusted_k MODULO factorial_value  
        END FOR  
        SET final_result TO joining all elements of result_list as continuous string  
        RETURN final_result  
    END FUNCTION  
END CLASS
