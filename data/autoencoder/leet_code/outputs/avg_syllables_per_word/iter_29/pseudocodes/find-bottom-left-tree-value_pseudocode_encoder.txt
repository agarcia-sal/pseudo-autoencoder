CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty OR values EQUALS nothing  
        RETURN nothing  
    END IF  
    SET root TO A NEW TreeNode WITH value AT position one OF values WITH left AS nothing AND right AS nothing  
    SET i TO one  
    SET queue TO A NEW empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND value AT position i OF values IS NOT nothing  
            SET node.left TO A NEW TreeNode WITH value AT position i OF values WITH left AS nothing AND right AS nothing  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND value AT position i OF values IS NOT nothing  
            SET node.right TO A NEW TreeNode WITH value AT position i OF values WITH left AS nothing AND right AS nothing  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS nothing AND q IS nothing  
        RETURN True  
    ELSE IF p IS nothing OR q IS nothing  
        RETURN False  
    ELSE IF value OF p IS NOT EQUALS TO value OF q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child OF p, left child OF q) AND is_same_tree(right child OF p, right child OF q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self, root)  
        SET queue TO A NEW double ended queue CONTAINING root  
        SET leftmost_value TO nothing  
        WHILE queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one  
                SET node TO REMOVE FIRST ELEMENT FROM queue  
                IF i EQUALS zero  
                    SET leftmost_value TO value OF node  
                END IF  
                IF left child OF node IS NOT nothing  
                    APPEND left child OF node TO queue  
                END IF  
                IF right child OF node IS NOT nothing  
                    APPEND right child OF node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
