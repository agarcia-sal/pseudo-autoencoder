CLASS Solution  
    FUNCTION myPow(x float_parameter n integer_parameter) RETURNS float  
        FUNCTION qpow(a float_parameter n integer_parameter) RETURNS float  
            SET ans TO one  
            WHILE n GREATER THAN zero  
                IF the result of the bitwise AND of n AND one EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO the result of the floor division of n BY two  
            END WHILE  
            RETURN ans  
        END FUNCTION  

        IF n GREATER THAN OR EQUAL TO zero  
            RETURN the result of calling qpow WITH a AS x AND n AS n  
        ELSE  
            RETURN one DIVIDED BY the result of calling qpow WITH a AS x AND n AS the negation of n  
        END IF  
    END FUNCTION  
END CLASS
