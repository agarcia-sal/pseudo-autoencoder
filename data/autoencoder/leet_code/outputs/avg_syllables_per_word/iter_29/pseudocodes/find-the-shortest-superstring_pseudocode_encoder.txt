CLASS Solution  
    FUNCTION shortestSuperstring(words)  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap(i j)  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i of words AND the LENGTH OF element at position j of words DOWN TO one BY minus one  
                IF the SUBSTRING of element at position i of words starting at the LENGTH OF element at position i of words MINUS k PLUS one TO the LENGTH OF element at position i of words EQUALS the SUBSTRING of element at position j of words starting at one TO k THEN  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp(mask i)  
            IF mask EQUALS two POWERED BY n MINUS one THEN  
                RETURN zero AND empty string  
            END IF  
            
            SET min_len TO a very large number representing infinity  
            SET best_path TO empty string  
            FOR j FROM zero TO n MINUS one  
                IF the bitwise AND of mask AND two POWERED BY j EQUALS zero THEN  
                    SET length_and_path TO dp(the bitwise OR of mask AND two POWERED BY j i)  
                    SET length TO the first element of length_and_path  
                    SET path TO the second element of length_and_path  
                    INCREMENT length BY the LENGTH OF element at position j of words MINUS overlap(i j)  
                    IF length LESS THAN min_len THEN  
                        SET min_len TO length  
                        SET best_path TO the SUBSTRING of element at position j of words starting at overlap(i j) PLUS one TO the LENGTH OF element at position j of words CONCATENATED WITH path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO a very large number representing infinity  
        SET shortest_path TO empty string  
        FOR i FROM zero TO n MINUS one  
            SET length_and_path TO dp(two POWERED BY i i)  
            SET length TO the first element of length_and_path  
            SET path TO the second element of length_and_path  
            INCREMENT length BY the LENGTH OF element at position i of words  
            IF length LESS THAN min_len THEN  
                SET min_len TO length  
                SET shortest_path TO element at position i of words CONCATENATED WITH path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
