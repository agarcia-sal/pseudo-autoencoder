CLASS Solution  
    FUNCTION maximumRequests(n number_of_buildings requests list_of_lists_of_numbers)  
        SET maximum_number_of_requests TO zero  
        SET total_number_of_requests TO the LENGTH OF requests  
        
        FOR mask FROM zero TO the NUMBER ONE SHIFTED LEFT BY total_number_of_requests MINUS one  
            SET balance_list TO a list of zeros OF LENGTH number_of_buildings  
            SET current_request_count TO zero  
            
            FOR index FROM zero TO total_number_of_requests MINUS one  
                IF the BITWISE AND of mask AND the NUMBER ONE SHIFTED LEFT BY index IS NOT EQUALS zero THEN  
                    SET from_building TO the element at position zero of the element at position index of requests  
                    SET to_building TO the element at position one of the element at position index of requests  
                    DECREMENT the element at position from_building of balance_list BY one  
                    INCREMENT the element at position to_building of balance_list BY one  
                    INCREMENT current_request_count BY one  
                END IF  
            END FOR  
            
            IF every element of balance_list EQUALS zero THEN  
                SET maximum_number_of_requests TO the GREATER OF maximum_number_of_requests AND current_request_count  
            END IF  
        END FOR  
        
        RETURN maximum_number_of_requests  
    END FUNCTION  
END CLASS
