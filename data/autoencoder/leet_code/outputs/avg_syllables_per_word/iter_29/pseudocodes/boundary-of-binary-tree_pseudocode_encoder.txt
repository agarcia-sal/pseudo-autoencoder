CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the input list values IS empty  
        RETURN no value  
    END IF  
    SET root TO new instance of TreeNode with value equal to element at position one of values and no left or right children  
    SET variable i TO one  
    SET variable queue TO a new double-ended queue  
    ADD root TO the end of queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET node.left TO a new instance of TreeNode with value equal to element at position i of values and no left or right children  
            ADD node.left TO the end of queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET node.right TO a new instance of TreeNode with value equal to element at position i of values and no left or right children  
            ADD node.right TO the end of queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value  
        RETURN truth  
    ELSE IF p IS no value OR q IS no value  
        RETURN falsehood  
    ELSE IF value of p IS NOT EQUAL TO value of q  
        RETURN falsehood  
    ELSE  
        RETURN the result of is_same_tree with p.left AND q.left AND the result of is_same_tree with p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION boundaryOfBinaryTree(root)  
        IF root IS no value  
            RETURN an empty list  
        END IF  
        
        FUNCTION is_leaf(node)  
            RETURN node.left IS no value AND node.right IS no value  
        END FUNCTION  
        
        SET left_boundary TO an empty list  
        SET current TO root.left  
        WHILE current IS NOT no value  
            IF is_leaf(current) IS falsehood  
                ADD value of current TO left_boundary  
            END IF  
            IF current.left IS NOT no value  
                SET current TO current.left  
            ELSE  
                SET current TO current.right  
            END IF  
        END WHILE  
        
        SET right_boundary TO an empty list  
        SET current TO root.right  
        WHILE current IS NOT no value  
            IF is_leaf(current) IS falsehood  
                ADD value of current TO right_boundary  
            END IF  
            IF current.right IS NOT no value  
                SET current TO current.right  
            ELSE  
                SET current TO current.left  
            END IF  
        END WHILE  
        
        SET leaves TO an empty list  
        FUNCTION collect_leaves(node)  
            IF node IS no value  
                RETURN  
            END IF  
            IF is_leaf(node)  
                ADD value of node TO leaves  
            END IF  
            CALL collect_leaves with node.left  
            CALL collect_leaves with node.right  
        END FUNCTION  
        
        CALL collect_leaves with root  
        
        IF is_leaf(root)  
            SET leaves TO an empty list  
        END IF  
        
        RETURN a new list consisting of value of root followed by all elements of left_boundary followed by all elements of leaves followed by all elements of right_boundary in the reverse order  
    END FUNCTION  
END CLASS
