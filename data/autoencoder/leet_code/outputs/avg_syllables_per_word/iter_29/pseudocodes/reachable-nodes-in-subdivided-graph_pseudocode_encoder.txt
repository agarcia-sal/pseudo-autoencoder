CLASS Solution  
    FUNCTION reachableNodes(edges, maxMoves, n)  
        SET graph TO a new empty map from node indices TO empty lists  
        FOR each index i FROM zero TO n MINUS one  
            SET the element at position i of graph TO empty list  
        END FOR  
        FOR each triple of values u v cnt IN edges  
            APPEND the pair of v and cnt TO the list at key u in graph  
            APPEND the pair of u and cnt TO the list at key v in graph  
        END FOR  
        
        SET distances TO a new list with length n FILLED WITH positive infinity  
        SET the element at position zero of distances TO zero  
        SET priority_queue TO a new list CONTAINING the pair of zero and zero  
  
        WHILE priority_queue is not empty  
            REMOVE and OBTAIN the pair of dist and node with smallest dist FROM priority_queue  
            IF dist IS GREATER THAN the element at position node of distances  
                CONTINUE to next iteration of WHILE loop  
            END IF  
            FOR each pair of neighbor and cnt IN the list at key node of graph  
                SET new_dist TO dist PLUS cnt PLUS one  
                IF new_dist IS LESS THAN the element at position neighbor of distances  
                    SET the element at position neighbor of distances TO new_dist  
                    ADD the pair of new_dist and neighbor INTO priority_queue maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        SET reachable_nodes TO zero  
        FOR each distance_value IN distances  
            IF distance_value IS LESS THAN OR EQUAL TO maxMoves  
                INCREMENT reachable_nodes BY one  
            END IF  
        END FOR  
        
        SET used_edges TO a new empty map  
        FOR each triple of values u v cnt IN edges  
            SET moves_u TO the maximum value BETWEEN zero AND maxMoves MINUS the element at position u of distances  
            SET moves_v TO the maximum value BETWEEN zero AND maxMoves MINUS the element at position v of distances  
            SET reachable_new_nodes TO the minimum value BETWEEN cnt AND moves_u PLUS moves_v  
            INCREMENT reachable_nodes BY reachable_new_nodes  
            ASSIGN reachable_new_nodes TO the key which is the pair of u and v in used_edges  
        END FOR  
  
        RETURN reachable_nodes  
    END FUNCTION  
END CLASS
