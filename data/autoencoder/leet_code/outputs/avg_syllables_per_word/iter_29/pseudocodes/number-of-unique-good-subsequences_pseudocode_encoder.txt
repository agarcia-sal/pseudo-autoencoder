CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MODULO TO ten POWERED BY nine PLUS seven  
        SET count_of_subsequences_ending_with_zero TO zero  
        SET count_of_subsequences_ending_with_one TO zero  
        SET contains_zero TO whether the binary string CONTAINS character zero  
        FOR each bit IN binary string  
            IF bit EQUALS character one  
                SET count_of_subsequences_ending_with_one TO the REMAINDER AFTER DIVIDING the SUM of count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one PLUS one BY MODULO  
            ELSE  
                SET count_of_subsequences_ending_with_zero TO the REMAINDER AFTER DIVIDING the SUM of count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one BY MODULO  
            END IF  
        END FOR  
        SET total_unique_good_subsequences TO the REMAINDER AFTER DIVIDING the SUM of count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one PLUS contains_zero BY MODULO  
        RETURN total_unique_good_subsequences  
    END FUNCTION  
END CLASS
