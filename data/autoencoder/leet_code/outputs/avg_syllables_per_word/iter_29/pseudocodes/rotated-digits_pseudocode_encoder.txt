CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO collection containing elements two five six nine  
        SET same_digits TO collection containing elements zero one eight  
        
        FUNCTION is_good_number(num)  
            CONVERT num TO string representation named num_str  
            IF every character in num_str IS AN ELEMENT OF the union of change_digits AND same_digits THEN  
                IF at least one character in num_str IS AN ELEMENT OF change_digits THEN  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR each integer i FROM one TO n  
            IF is_good_number(i) EQUALS true THEN  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        
        RETURN good_count  
    END FUNCTION  
END CLASS
