CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO the list of characters representing the string form of n  
        SET length TO the LENGTH OF digits  
        SET marker TO length  
        FOR i FROM length MINUS one TO one IN DECREASING order  
            IF the character at position i of digits converted to number IS LESS THAN the character at position i MINUS one of digits converted to number  
                SET the character at position i MINUS one of digits TO the string form of the number obtained by subtracting one from the character at position i MINUS one of digits converted to number  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one  
            SET the character at position i of digits TO the string form of the number nine  
        END FOR  
        RETURN the integer form of the string obtained by concatenating all characters in digits  
    END FUNCTION  
END CLASS
