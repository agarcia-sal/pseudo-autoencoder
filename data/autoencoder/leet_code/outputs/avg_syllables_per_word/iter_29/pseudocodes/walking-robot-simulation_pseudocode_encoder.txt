CLASS Solution  
    FUNCTION robotSim(commands, obstacles)  
        SET directions TO a list containing the pairs zero and one one and zero zero and negative one negative one and zero representing north east south and west respectively  
        SET position_x TO zero  
        SET position_y TO zero  
        SET direction_index TO zero indicating facing north  
        SET obstacles_set TO an empty set  
        FOR each obstacle IN obstacles  
            APPEND tuple of obstacle to obstacles_set  
        END FOR  
        SET maximum_distance TO zero  
        
        FOR each command IN commands  
            IF command EQUALS negative two THEN  
                SET direction_index TO direction_index MINUS one MODULO four  
            ELSE IF command EQUALS negative one THEN  
                SET direction_index TO direction_index PLUS one MODULO four  
            ELSE  
                SET delta_x TO element at position direction_index of directions FIRST element  
                SET delta_y TO element at position direction_index of directions SECOND element  
                FOR count FROM one TO command  
                    SET next_x TO position_x PLUS delta_x  
                    SET next_y TO position_y PLUS delta_y  
                    IF tuple of next_x and next_y IS IN obstacles_set THEN  
                        BREAK  
                    END IF  
                    SET position_x TO next_x  
                    SET position_y TO next_y  
                END FOR  
                SET current_distance TO position_x MULTIPLIED BY position_x PLUS position_y MULTIPLIED BY position_y  
                IF current_distance GREATER THAN maximum_distance THEN  
                    SET maximum_distance TO current_distance  
                END IF  
            END IF  
        END FOR  
        
        RETURN maximum_distance  
    END FUNCTION  
END CLASS
