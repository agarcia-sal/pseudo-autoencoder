CLASS Solution  
    FUNCTION minInteger(num AS string_parameter, k AS integer_parameter) RETURNS string  
        SET list_of_characters TO list representation of string_parameter  
        SET length_of_list TO the LENGTH OF list_of_characters  
        
        WHILE integer_parameter GREATER THAN zero  
            SET swap_occurred TO false  
            FOR index FROM zero TO length_of_list MINUS one MINUS one  
                IF character at position index of list_of_characters GREATER THAN character at position index PLUS one of list_of_characters  
                    SET smallest_character TO character nine  
                    SET position_of_smallest TO negative one  
                    FOR position FROM index TO the MINIMUM OF index PLUS integer_parameter PLUS one AND length_of_list  
                        IF character at position of list_of_characters LESS THAN smallest_character  
                            SET smallest_character TO character at position of list_of_characters  
                            SET position_of_smallest TO position  
                        END IF  
                    END FOR  
                    
                    SET sublist_to_be_replaced TO list consisting of character at position position_of_smallest of list_of_characters FOLLOWED BY characters from position index TO position position_of_smallest MINUS one of list_of_characters  
                    REPLACE elements from position index TO position position_of_smallest of list_of_characters WITH sublist_to_be_replaced  
                    DECREMENT integer_parameter BY position_of_smallest MINUS index  
                    SET swap_occurred TO true  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            
            IF swap_occurred IS false  
                BREAK the WHILE loop  
            END IF  
        END WHILE  
        
        RETURN concatenation of all characters in list_of_characters  
    END FUNCTION  
END CLASS
