CLASS Solution  
    FUNCTION computeArea(ax1 ax1 INPUT ay1 ay1 INPUT ax2 ax2 INPUT ay2 ay2 INPUT bx1 bx1 INPUT by1 by1 INPUT bx2 bx2 INPUT by2 by2 INPUT)  
        SET area1 TO the difference between ax2 AND ax1 MULTIPLIED BY the difference between ay2 AND ay1  
        SET area2 TO the difference between bx2 AND bx1 MULTIPLIED BY the difference between by2 AND by1  
        SET overlap_width TO the greater value BETWEEN zero AND the result of the lesser value BETWEEN ax2 AND bx2 MINUS the greater value BETWEEN ax1 AND bx1  
        SET overlap_height TO the greater value BETWEEN zero AND the result of the lesser value BETWEEN ay2 AND by2 MINUS the greater value BETWEEN ay1 AND by1  
        SET overlap_area TO overlap_width MULTIPLIED BY overlap_height  
        SET total_area TO area1 PLUS area2 MINUS overlap_area  
        RETURN total_area  
    END FUNCTION  
END CLASS
