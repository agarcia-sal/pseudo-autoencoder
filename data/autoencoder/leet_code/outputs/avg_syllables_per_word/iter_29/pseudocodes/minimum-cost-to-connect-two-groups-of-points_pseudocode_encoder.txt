CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size one TO the LENGTH OF cost  
        SET size two TO the LENGTH OF element at position zero of cost  
        
        SET dp TO an empty two dimensional list with the FIRST dimension being size one PLUS one  
        FOR each index i FROM zero TO size one  
            SET a list of length POWERED BY two TO size two filled with positive infinity as inf TO dp at position i  
        END FOR  
        
        SET element at position zero of dp at position zero TO zero  
        
        FOR each index i FROM zero TO size one MINUS one  
            FOR each integer mask FROM zero TO the VALUE OF two POWERED BY size two MINUS one  
                FOR each index j FROM zero TO size two MINUS one  
                    SET position new mask TO the BITWISE OR of mask WITH the VALUE OF two POWERED BY j  
                    SET element at position i PLUS one at position new mask of dp TO THE MINIMUM OF element at position i PLUS one at position new mask of dp AND element at position i at position mask of dp PLUS the element at position i at position j of cost  
                END FOR  
                
                FOR each index j FROM zero TO size two MINUS one  
                    IF the BITWISE AND of mask AND the VALUE OF two POWERED BY j IS NOT EQUAL TO zero  
                        SET element at position i PLUS one at position mask of dp TO THE MINIMUM OF element at position i PLUS one at position mask of dp AND element at position i PLUS one at the BITWISE XOR of mask AND the VALUE OF two POWERED BY j of dp PLUS the element at position i at position j of cost  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position size one at the VALUE OF two POWERED BY size two MINUS one of dp  
    END FUNCTION  
END CLASS
