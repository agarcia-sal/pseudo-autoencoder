CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the PAIR consisting of x and y IS IN visited OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            ADD the PAIR consisting of x and y TO visited  
            FOR each PAIR dx and dy IN the collection consisting of PAIR minus one and zero PLUS PAIR one and zero PLUS PAIR zero and minus one PLUS PAIR zero and one  
                CALL dfs WITH arguments x PLUS dx AND y PLUS dy AND visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an empty set  
            SET land_cells TO an empty list  
            FOR each index i FROM zero TO the LENGTH OF grid MINUS one  
                FOR each index j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                    IF element at position i of grid at position j EQUALS one  
                        APPEND the PAIR consisting of i and j TO land_cells  
                    END IF  
                END FOR  
            END FOR  
            IF LENGTH OF land_cells EQUALS zero  
                RETURN True  
            END IF  
            CALL dfs WITH arguments the first element of the first pair in land_cells AND the second element of the first pair in land_cells AND visited  
            RETURN WHETHER the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF NOT is_connected()  
            RETURN zero  
        END IF  

        FOR each index i FROM zero TO the LENGTH OF grid MINUS one  
            FOR each index j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    SET element at position i of grid at position j TO zero  
                    IF NOT is_connected()  
                        RETURN one  
                    END IF  
                    SET element at position i of grid at position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
