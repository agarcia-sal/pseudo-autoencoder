CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, left WITH DEFAULT nothing, right WITH DEFAULT nothing)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS nothing OR the LENGTH OF values EQUALS zero  
        RETURN nothing  
    END IF  
    SET root TO A NEW TreeNode WITH val EQUALS element at position zero of values  
    SET i TO one  
    SET queue TO A NEW empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element REMOVED FROM THE FRONT OF queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT nothing  
            SET node dot left TO A NEW TreeNode WITH val EQUALS element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT nothing  
            SET node dot right TO A NEW TreeNode WITH val EQUALS element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN True  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN False  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p dot left, q dot left) AND is_same_tree(p dot right, q dot right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF preorder EQUALS nothing OR the LENGTH OF preorder EQUALS zero OR inorder EQUALS nothing OR the LENGTH OF inorder EQUALS zero  
            RETURN nothing  
        END IF  
        SET root_val TO element at position zero of preorder  
        SET root TO A NEW TreeNode WITH val EQUALS root_val  
        SET root_index_inorder TO the POSITION OF root_val IN inorder  
        SET root dot left TO self dot buildTree(sublist of preorder FROM position one TO position root_index_inorder PLUS one EXCLUSIVE, sublist of inorder FROM position zero TO position root_index_inorder EXCLUSIVE)  
        SET root dot right TO self dot buildTree(sublist of preorder FROM position root_index_inorder PLUS one TO the END, sublist of inorder FROM position root_index_inorder PLUS one TO the END)  
        RETURN root  
    END FUNCTION  
END CLASS
