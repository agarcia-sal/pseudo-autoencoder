CLASS Solution  
    FUNCTION countArrangement(n)  
        FUNCTION backtrack(used_collection index)  
            IF index EQUALS n PLUS one  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR i FROM one TO n  
                IF element at position i of used_collection EQUALS False AND (i MODULO index EQUALS zero OR index MODULO i EQUALS zero)  
                    SET element at position i of used_collection TO True  
                    INCREMENT count BY backtrack(used_collection index PLUS one)  
                    SET element at position i of used_collection TO False  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET used_collection TO a collection of False elements of size n PLUS one  
        RETURN backtrack(used_collection one)  
    END FUNCTION  
END CLASS
