CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS EMPTY OR the LENGTH OF stones IS LESS THAN two  
            RETURN False  
        END IF  
        
        SET stone_set TO the SET OF elements FROM stones  
        
        SET memo TO an EMPTY MAP  
        
        FUNCTION can_jump_to(position jump)  
            IF position EQUALS the LAST ELEMENT of stones  
                RETURN True  
            END IF  
            
            IF the PAIR position AND jump IS A KEY IN memo  
                RETURN the VALUE ASSOCIATED WITH the PAIR position AND jump IN memo  
            END IF  
            
            FOR each possible_jump FROM jump MINUS one TO jump PLUS one  
                IF possible_jump IS GREATER THAN zero AND possible_jump PLUS position IS IN stone_set  
                    IF can_jump_to(possible_jump PLUS position possible_jump)  
                        SET the VALUE ASSOCIATED WITH the PAIR position AND jump IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the VALUE ASSOCIATED WITH the PAIR position AND jump IN memo TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(the FIRST ELEMENT OF stones zero)  
    END FUNCTION  
END CLASS
