CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO the result of splitting input by the newline character  
        SET stack TO an empty collection  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO the number of tab characters contained in line  
            SET name TO the substring of line starting from position depth to the end of line  
            
            WHILE the number of elements in stack IS GREATER THAN depth  
                REMOVE the last element from stack  
            END WHILE  
            
            IF the number of elements in stack IS GREATER THAN zero  
                SET current_length TO the last element in stack PLUS the length of name PLUS one  
            ELSE  
                SET current_length TO the length of name  
            END IF  
            
            IF the name contains the character representing a period  
                IF current_length IS GREATER THAN max_length  
                    SET max_length TO current_length  
                END IF  
            ELSE  
                ADD current_length TO the end of stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
