CLASS Solution  
    FUNCTION findKthLargest(list_of_numbers K)  
        SET min_heap TO an empty list  
        FOR each number IN list_of_numbers  
            CALL a procedure to push number INTO min_heap preserving heap property  
            IF the LENGTH OF min_heap GREATER THAN K  
                CALL a procedure to remove the smallest element FROM min_heap preserving heap property  
            END IF  
        END FOR  
        RETURN the element at position one of min_heap  
    END FUNCTION  
END CLASS
