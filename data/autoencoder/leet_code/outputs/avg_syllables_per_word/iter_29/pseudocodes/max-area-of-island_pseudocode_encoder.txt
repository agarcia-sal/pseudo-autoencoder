CLASS Solution  
    FUNCTION maxAreaOfIsland(grid)  
        IF grid EQUALS an empty collection  
            RETURN zero  
        END IF  
        
        SET max_area TO zero  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        
        FUNCTION dfs(row, column)  
            IF row LESS THAN zero OR row GREATER THAN OR EQUAL TO rows OR column LESS THAN zero OR column GREATER THAN OR EQUAL TO cols OR element at position row of grid at position column EQUALS zero  
                RETURN zero  
            END IF  
            SET element at position row of grid at position column TO zero  
            SET area TO one  
            INCREMENT area BY dfs(row PLUS one, column)  
            INCREMENT area BY dfs(row MINUS one, column)  
            INCREMENT area BY dfs(row, column PLUS one)  
            INCREMENT area BY dfs(row, column MINUS one)  
            RETURN area  
        END FUNCTION  
        
        FOR row FROM zero TO rows MINUS one  
            FOR column FROM zero TO cols MINUS one  
                IF element at position row of grid at position column EQUALS one  
                    SET max_area TO the GREATER OF max_area AND dfs(row, column)  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
