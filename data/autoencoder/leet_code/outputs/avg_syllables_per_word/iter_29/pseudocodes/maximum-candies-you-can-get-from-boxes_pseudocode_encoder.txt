CLASS Solution  
    FUNCTION maxCandies(status LIST OF INTEGER comma candies LIST OF INTEGER comma keys LIST OF LIST OF INTEGER comma containedBoxes LIST OF LIST OF INTEGER comma initialBoxes LIST OF INTEGER)  
        SET boxes_we_have TO a new set containing each element from initialBoxes  
        SET keys_we_have TO an empty set  
        SET total_candies TO zero  
        SET progress_made TO true  
        WHILE progress_made EQUALS true  
            SET progress_made TO false  
            FOR each box IN a list created from boxes_we_have  
                IF the element at position box of status EQUALS one OR box IS IN keys_we_have  
                    INCREMENT total_candies BY the element at position box of candies  
                    ADD each element from the list at position box of keys TO keys_we_have  
                    ADD each element from the list at position box of containedBoxes TO boxes_we_have  
                    REMOVE box FROM boxes_we_have  
                    SET progress_made TO true  
                END IF  
            END FOR  
        END WHILE  
        RETURN total_candies  
    END FUNCTION  
END CLASS
