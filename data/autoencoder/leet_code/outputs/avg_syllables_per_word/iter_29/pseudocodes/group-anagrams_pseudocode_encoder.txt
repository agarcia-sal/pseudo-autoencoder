CLASS Solution  
    FUNCTION groupAnagrams(list_of_strings)  
        SET anagrams_mapping TO a new default dictionary where each value is a list  
        FOR each string IN list_of_strings  
            SET characters_of_string TO the collection of characters that compose string  
            SET sorted_characters TO the result of arranging characters_of_string in ascending alphabetical order  
            SET sorted_string_key TO the concatenation of all characters in sorted_characters in sequence  
            APPEND string TO the list at key sorted_string_key in anagrams_mapping  
        END FOR  
        SET list_of_anagram_groups TO the list of all values contained in anagrams_mapping  
        RETURN list_of_anagram_groups  
    END FUNCTION  
END CLASS
