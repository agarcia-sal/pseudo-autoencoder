CLASS Solution  
    FUNCTION isPossible(list_of_numbers)  
        SET num_count TO a counter of each number in list_of_numbers representing the frequency of each number  
        SET end_with TO a counter representing the count of subsequences ending with each number  
        
        FOR each number IN list_of_numbers  
            IF the frequency of the current number in num_count EQUALS zero  
                CONTINUE to next iteration  
            END IF  
            
            IF the count of subsequences ending with the number that is one less than the current number IN end_with IS GREATER THAN zero  
                DECREMENT the count of subsequences ending with the number that is one less than the current number IN end_with BY one  
                INCREMENT the count of subsequences ending with the current number IN end_with BY one  
            ELSE IF the frequency of the number that is one greater than the current number IN num_count IS GREATER THAN zero AND the frequency of the number that is two greater than the current number IN num_count IS GREATER THAN zero  
                DECREMENT the frequency of the number that is one greater than the current number IN num_count BY one  
                DECREMENT the frequency of the number that is two greater than the current number IN num_count BY one  
                INCREMENT the count of subsequences ending with the number that is two greater than the current number IN end_with BY one  
            ELSE  
                RETURN false  
            END IF  
            
            DECREMENT the frequency of the current number IN num_count BY one  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
