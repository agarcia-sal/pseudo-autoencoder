CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET variable i TO the LENGTH OF nums MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND the element at position i of nums GREATER THAN OR EQUAL TO the element at position i PLUS one of nums  
            DECREMENT i BY one  
        END WHILE  
        IF i GREATER THAN OR EQUAL TO zero  
            SET variable j TO the LENGTH OF nums MINUS one  
            WHILE the element at position j of nums LESS THAN OR EQUAL TO the element at position i of nums  
                DECREMENT j BY one  
            END WHILE  
            SET temporary variable temp TO the element at position i of nums  
            SET the element at position i of nums TO the element at position j of nums  
            SET the element at position j of nums TO temporary variable temp  
        END IF  
        SET variable start TO i PLUS one  
        SET variable end TO the LENGTH OF nums MINUS one  
        WHILE start LESS THAN end  
            SET temporary variable temp TO the element at position start of nums  
            SET the element at position start of nums TO the element at position end of nums  
            SET the element at position end of nums TO temporary variable temp  
            INCREMENT start BY one  
            DECREMENT end BY one  
        END WHILE  
    END FUNCTION  
END CLASS
