CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count_to_x_or_less TO zero  
            FOR each value IN nums  
                IF value LESS THAN OR EQUAL TO x  
                    INCREMENT count_to_x_or_less BY one  
                END IF  
            END FOR  
            RETURN count_to_x_or_less GREATER THAN x  
        END FUNCTION  
        RETURN the smallest index IN the range FROM zero TO the LENGTH OF nums THAT SATISFIES that function f RETURNS true  
    END FUNCTION  
END CLASS
