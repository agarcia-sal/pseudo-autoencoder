CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO a new mapping from file content to empty lists  
        FOR each path IN paths  
            SPLIT path INTO parts BY space  
            SET directory TO the first element of parts  
            FOR each file_info IN parts starting from the second element to the last  
                SPLIT file_info INTO file_name AND content PARTS WHERE content BEGINS after the opening parenthesis  
                REMOVE the last character from content to eliminate the closing parenthesis  
                SET full_file_path TO the concatenation of directory FOLLOWED BY a forward slash FOLLOWED BY file_name  
                APPEND full_file_path TO the list associated with content in content_to_paths  
            END FOR  
        END FOR  
        SET duplicate_groups TO an empty list  
        FOR each group_of_paths IN all the lists in content_to_paths  
            IF the LENGTH OF group_of_paths IS GREATER THAN one  
                APPEND group_of_paths TO duplicate_groups  
            END IF  
        END FOR  
        RETURN duplicate_groups  
    END FUNCTION  
END CLASS
