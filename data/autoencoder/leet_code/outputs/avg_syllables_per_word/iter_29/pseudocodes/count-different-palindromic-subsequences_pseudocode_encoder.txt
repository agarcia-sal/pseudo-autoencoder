CLASS Solution  
    FUNCTION countPalindromicSubsequences(s)  
        SET MOD TO ten POWERED BY nine PLUS seven  
        SET n TO the LENGTH OF s  
        
        SET next_occurrence TO a two dimensional collection with four rows each containing n elements all SET TO negative one  
        SET prev_occurrence TO a two dimensional collection with four rows each containing n elements all SET TO negative one  
        
        FOR each c FROM zero TO three  
            SET character TO the letter a PLUS c positions in the alphabet  
            SET last TO negative one  
            FOR each i FROM zero TO n MINUS one  
                IF the character at position i of s EQUALS character THEN  
                    SET last TO i  
                END IF  
                SET element at position i of the row at position c of prev_occurrence TO last  
            END FOR  
            
            SET last TO negative one  
            FOR each i FROM n MINUS one DOWNTO zero  
                IF the character at position i of s EQUALS character THEN  
                    SET last TO i  
                END IF  
                SET element at position i of the row at position c of next_occurrence TO last  
            END FOR  
        END FOR  
        
        SET dp TO a two dimensional collection with n rows each containing n elements all SET TO zero  
        
        FOR each i FROM zero TO n MINUS one  
            SET element at position i and position i of dp TO one  
        END FOR  
        
        FOR each length FROM two TO n  
            FOR each i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                FOR each c FROM zero TO three  
                    SET character TO the letter a PLUS c positions in the alphabet  
                    SET left TO element at position i of the row at position c of next_occurrence  
                    SET right TO element at position j of the row at position c of prev_occurrence  
                    
                    IF left GREATER THAN j OR right LESS THAN i THEN  
                        CONTINUE to the next iteration of the character loop  
                    ELSE IF left EQUALS right THEN  
                        SET element at position i and position j of dp TO element at position i and position j of dp PLUS one MODULO MOD  
                    ELSE  
                        SET element at position i and position j of dp TO element at position i and position j of dp PLUS element at position left PLUS one and position right MINUS one of dp PLUS two MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero and position n MINUS one of dp  
    END FUNCTION  
END CLASS
