CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN a list containing zero  
        END IF  
        SET prev_gray TO the result of calling grayCode with the argument n MINUS one  
        SET mask TO one SHIFTED LEFT BY n MINUS one positions  
        SET reversed_prev_gray TO the list prev_gray in reverse order  
        SET extended_list TO an empty list  
        FOR each number IN reversed_prev_gray  
            SET appended_number TO mask BITWISE OR number  
            APPEND appended_number TO extended_list  
        END FOR  
        SET current_gray TO the concatenation of prev_gray FOLLOWED BY extended_list  
        RETURN current_gray  
    END FUNCTION  
END CLASS
