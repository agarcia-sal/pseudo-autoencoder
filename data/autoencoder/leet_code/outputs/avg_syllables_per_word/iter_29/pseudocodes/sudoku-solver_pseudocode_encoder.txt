CLASS Solution  
    FUNCTION solveSudoku(board)  
        
        FUNCTION is_valid(board_parameter row_parameter col_parameter num_parameter)  
            FOR index_i FROM zero TO eight  
                IF element at position row_parameter of board_parameter AT position index_i EQUALS num_parameter  
                    RETURN false  
                END IF  
            END FOR  
            
            FOR index_i FROM zero TO eight  
                IF element at position index_i of board_parameter AT position col_parameter EQUALS num_parameter  
                    RETURN false  
                END IF  
            END FOR  
            
            SET start_row TO three MULTIPLIED BY the INTEGER DIVISION of row_parameter BY three  
            SET start_col TO three MULTIPLIED BY the INTEGER DIVISION of col_parameter BY three  
            
            FOR index_i FROM zero TO two  
                FOR index_j FROM zero TO two  
                    IF element at position start_row PLUS index_i of board_parameter AT position start_col PLUS index_j EQUALS num_parameter  
                        RETURN false  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN true  
        END FUNCTION  
        
        FUNCTION solve(board_parameter)  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF element at position row of board_parameter AT position col EQUALS the single character representing empty cell  
                        FOR num_ascii FROM the character representing one TO the character representing nine  
                            IF is_valid(board_parameter row col num_ascii)  
                                SET element at position row of board_parameter AT position col TO num_ascii  
                                IF solve(board_parameter)  
                                    RETURN true  
                                END IF  
                                SET element at position row of board_parameter AT position col TO the single character representing empty cell  
                            END IF  
                        END FOR  
                        RETURN false  
                    END IF  
                END FOR  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        CALL solve WITH board  
    END FUNCTION  
END CLASS
