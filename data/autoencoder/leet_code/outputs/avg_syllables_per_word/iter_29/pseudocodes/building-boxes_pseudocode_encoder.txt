CLASS Solution  
    FUNCTION minimumBoxes(n)  
        FUNCTION triangular_pyramid(k)  
            SET result TO k MULTIPLIED BY the sum of k AND one MULTIPLIED BY the sum of k AND two DIVIDED BY the natural number six  
            RETURN result  
        END FUNCTION  
      
        SET left TO zero  
        SET right TO n  
        WHILE left LESS THAN right  
            SET mid TO the sum of left AND right AND one DIVIDED BY the natural number two  
            IF the value of triangular_pyramid called with mid IS LESS THAN OR EQUAL TO n  
                SET left TO mid  
            ELSE  
                SET right TO the difference between mid AND one  
            END IF  
        END WHILE  
      
        SET boxes_used TO the value of triangular_pyramid called with left  
        SET remaining TO the difference between n AND boxes_used  
        SET floor_boxes TO left MULTIPLIED BY the sum of left AND one DIVIDED BY the natural number two  
      
        SET additional_boxes TO zero  
        WHILE remaining IS GREATER THAN zero  
            INCREMENT additional_boxes BY one  
            DECREMENT remaining BY additional_boxes  
        END WHILE  
      
        RETURN the sum of floor_boxes AND additional_boxes  
    END FUNCTION  
END CLASS
