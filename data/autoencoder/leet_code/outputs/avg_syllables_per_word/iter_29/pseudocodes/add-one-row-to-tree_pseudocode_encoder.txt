CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET this instance val TO val  
        SET this instance left TO left  
        SET this instance right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values_list)  
    IF the values_list IS empty OR DOES NOT exist  
        RETURN None  
    END IF  
    SET root_node TO a new TreeNode with the element at position one OF values_list  
    SET index i TO one  
    SET queue_structure TO an empty double ended queue  
    APPEND root_node TO queue_structure  
    WHILE the queue_structure IS NOT empty  
        REMOVE the element from the front of queue_structure AND ASSIGN it TO node  
        IF index i IS LESS THAN the LENGTH OF values_list AND the element at position i OF values_list IS NOT None  
            SET node left TO a new TreeNode with the element at position i OF values_list  
            APPEND the node left TO queue_structure  
        END IF  
        INCREMENT i BY one  
        IF index i IS LESS THAN the LENGTH OF values_list AND the element at position i OF values_list IS NOT None  
            SET node right TO a new TreeNode with the element at position i OF values_list  
            APPEND the node right TO queue_structure  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root_node  
END FUNCTION  

FUNCTION is_same_tree(first_tree_node, second_tree_node)  
    IF first_tree_node DOES NOT exist AND second_tree_node DOES NOT exist  
        RETURN True  
    ELSE IF first_tree_node DOES NOT exist OR second_tree_node DOES NOT exist  
        RETURN False  
    ELSE IF the val attribute of first_tree_node IS NOT EQUAL TO the val attribute of second_tree_node  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of is_same_tree with the left children of first_tree_node and second_tree_node AND the result of is_same_tree with the right children of first_tree_node and second_tree_node  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow(root_tree_node, val_to_add, depth_level)  
        IF depth_level EQUALS one  
            RETURN a new TreeNode with val_to_add AS val AND left child AS root_tree_node  
        END IF  
        SET queue_structure TO a double ended queue containing the tuple of root_tree_node AND one  
        WHILE queue_structure IS NOT empty  
            REMOVE the element from the front of queue_structure AND ASSIGN it TO a tuple of node AND current_level  
            IF current_level EQUALS depth_level MINUS one  
                SET node left TO a new TreeNode with val_to_add AS val AND left child AS the existing node left  
                SET node right TO a new TreeNode with val_to_add AS val AND right child AS the existing node right  
            ELSE IF current_level IS LESS THAN depth_level MINUS one  
                IF node left EXISTS  
                    APPEND the tuple of node left AND current_level PLUS one TO queue_structure  
                END IF  
                IF node right EXISTS  
                    APPEND the tuple of node right AND current_level PLUS one TO queue_structure  
                END IF  
            END IF  
        END WHILE  
        RETURN root_tree_node  
    END FUNCTION  
END CLASS
