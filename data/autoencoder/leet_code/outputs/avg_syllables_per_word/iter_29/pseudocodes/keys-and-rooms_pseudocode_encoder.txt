CLASS Solution  
    FUNCTION canVisitAllRooms(rooms)  
        SET visited TO an empty collection representing a set  
        SET stack TO a collection containing only zero  
        WHILE the collection stack is not empty  
            REMOVE the last element from stack and ASSIGN it to current_room  
            ADD current_room TO the collection visited  
            FOR each key IN the collection at position current_room of rooms  
                IF the key is NOT IN the collection visited  
                    ADD the key TO the collection stack  
                END IF  
            END FOR  
        END WHILE  
        RETURN the LENGTH OF the collection visited EQUALS the LENGTH OF rooms  
    END FUNCTION  
END CLASS
