CLASS Solution  
    FUNCTION maxTaskAssign(tasks WORKERS pills STRENGTH)  
        SORT tasks IN ASCENDING ORDER  
        SORT workers IN ASCENDING ORDER  
        
        FUNCTION canAssignTasks(k)  
            SET available_workers TO the last k elements of workers  
            SET pills_used TO zero  
            
            FOR index FROM k MINUS one DOWN TO zero DECREMENT BY one  
                SET task TO element at position index of tasks  
                SET worker_index TO the position where task can be inserted in available_workers to maintain order using binary search  
                IF worker_index IS LESS THAN the LENGTH OF available_workers THEN  
                    REMOVE element at position worker_index FROM available_workers  
                ELSE IF pills_used IS LESS THAN pills THEN  
                    SET worker_index TO the position where task MINUS strength can be inserted in available_workers to maintain order using binary search  
                    IF worker_index IS LESS THAN the LENGTH OF available_workers THEN  
                        REMOVE element at position worker_index FROM available_workers  
                        INCREMENT pills_used BY one  
                    ELSE  
                        RETURN False  
                    END IF  
                ELSE  
                    RETURN False  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the MINIMUM OF the LENGTH OF tasks AND the LENGTH OF workers PLUS one  
        WHILE left IS LESS THAN right  
            SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
            IF canAssignTasks(mid) IS True THEN  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN left MINUS one  
    END FUNCTION  
END CLASS
