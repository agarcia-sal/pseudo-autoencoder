CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO a tuple of tuples each created from each row of board  
        SET target TO a tuple of tuples consisting of one two three and four five zero  
        SET neighbors TO a mapping from zero TO a set of one and three from one TO a set of zero and two and four from two TO a set of one and five from three TO a set of zero and four from four TO a set of one and three and five from five TO a set of two and four  
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF the element at position i of start at position j EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a double ended queue initialized with a tuple consisting of start zero_pos and zero  
        SET visited TO a set initialized with start  
        WHILE queue holds elements  
            REMOVE the leftmost element of queue and ASSIGN its components to state zero_pos and moves  
            IF state EQUALS target  
                RETURN moves  
            END IF  
            FOR each neighbor IN the set of neighbors at key zero_pos  
                SET new_state TO a list of lists each created from each row of state  
                SET i TO the integer division of zero_pos by three  
                SET j TO the remainder of zero_pos divided by three  
                SET ni TO the integer division of neighbor by three  
                SET nj TO the remainder of neighbor divided by three  
                SWAP the element at position i of new_state at position j WITH the element at position ni of new_state at position nj  
                SET new_tuple_state TO a tuple of tuples each created from each row of new_state  
                IF new_tuple_state is not in visited  
                    ADD new_tuple_state to visited  
                    APPEND a tuple consisting of new_tuple_state neighbor and moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
