CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET mapping_of_characters_to_indexes TO an empty collection  
        SET maximum_length TO zero  
        SET start_position TO zero  
        FOR each index_and_character FROM zero TO the LENGTH OF s MINUS one  
            SET end_position TO the current index from index_and_character  
            SET char TO the current character from index_and_character  
            IF char IS IN mapping_of_characters_to_indexes AND the element at char in mapping_of_characters_to_indexes IS GREATER THAN OR EQUAL TO start_position  
                SET start_position TO the element at char in mapping_of_characters_to_indexes PLUS one  
            END IF  
            SET the element at char in mapping_of_characters_to_indexes TO end_position  
            SET maximum_length TO the GREATER VALUE BETWEEN maximum_length AND end_position MINUS start_position PLUS one  
        END FOR  
        RETURN maximum_length  
    END FUNCTION  
END CLASS
