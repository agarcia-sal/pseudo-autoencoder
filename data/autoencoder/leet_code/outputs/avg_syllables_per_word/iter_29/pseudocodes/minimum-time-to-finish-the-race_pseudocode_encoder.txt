CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of infinity repeated fifteen times  
        FOR each pair of base_time and rate IN tires  
            SET total_time TO zero  
            FOR each index FROM zero TO fourteen  
                SET lap_time TO base_time MULTIPLIED BY rate POWERED BY index  
                IF lap_time GREATER THAN changeTime PLUS base_time  
                    BREAK the inner loop  
                END IF  
                INCREMENT total_time BY lap_time  
                SET min_times at position index TO the minimum BETWEEN min_times at position index AND total_time  
            END FOR  
        END FOR  
        
        SET dp TO a list of infinity repeated numLaps plus one times  
        SET dp at position zero TO zero  
        
        FOR each i FROM one TO numLaps inclusive  
            FOR each j FROM zero TO the minimum BETWEEN i MINUS one AND fourteen  
                SET dp at position i TO the minimum BETWEEN dp at position i AND dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
            END FOR  
        END FOR  
        
        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
