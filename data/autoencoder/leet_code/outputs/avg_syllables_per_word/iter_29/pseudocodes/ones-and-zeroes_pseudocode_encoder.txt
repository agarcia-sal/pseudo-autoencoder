CLASS Solution  
    FUNCTION findMaxForm(strs m n)  
        CREATE a two dimensional list named dp with the number of rows equal to m incremented by one and the number of columns equal to n incremented by one each element initialized to zero  
        FOR each string s IN strs  
            SET count_0 TO the number of times character zero occurs in s  
            SET count_1 TO the number of times character one occurs in s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp element at row i column j TO the maximum value BETWEEN dp element at row i column j AND dp element at row equal to i minus count_0 column equal to j minus count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at row m column n  
    END FUNCTION  
END CLASS
