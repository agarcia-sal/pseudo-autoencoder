CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the string representation of count TO path  
                END IF  
                APPEND the concatenation of all elements in path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH arguments index PLUS one AND the concatenation of path WITH the string representation of count AND the character at position index of word AND zero  
                ELSE  
                    CALL backtrack WITH arguments index PLUS one AND the concatenation of path WITH the character at position index of word AND zero  
                END IF  
                CALL backtrack WITH arguments index PLUS one AND path AND count PLUS one  
            END IF  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL backtrack WITH arguments zero AND an empty list AND zero  
        RETURN result  
    END FUNCTION  
END CLASS
