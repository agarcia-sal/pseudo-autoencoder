CLASS Solution  
    FUNCTION getHint(secret_string secret, guess_string guess)  
        SET bulls TO zero  
        SET cows TO zero  
        INITIALIZE a dictionary secret_count to count occurrences of characters with default zero  
        INITIALIZE a dictionary guess_count to count occurrences of characters with default zero  
        FOR each pair of characters s and g TAKEN SIMULTANEOUSLY FROM secret_string secret AND guess_string guess  
            IF character s EQUALS character g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value associated with character s IN secret_count BY one  
                INCREMENT the value associated with character g IN guess_count BY one  
            END IF  
        END FOR  
        FOR each character digit IN the keys of secret_count  
            INCREMENT cows BY the lesser of the value associated with character digit IN secret_count AND the value associated with character digit IN guess_count  
        END FOR  
        RETURN a string composed by the value of bulls FOLLOWED BY the character A FOLLOWED BY the value of cows FOLLOWED BY the character B  
    END FUNCTION  
END CLASS
