CLASS Solution  
    FUNCTION solveNQueens(n)  
        FUNCTION is_not_under_attack(row, col)  
            RETURN NOT the sum of the value at position col of cols PLUS the value at position row MINUS col of hills PLUS the value at position row PLUS col of dales EQUALS zero  
        END FUNCTION  
      
        FUNCTION place_queen(row, col)  
            ADD the pair consisting of row and col TO the set queens  
            SET the value at position col of cols TO one  
            SET the value at position row MINUS col of hills TO one  
            SET the value at position row PLUS col of dales TO one  
        END FUNCTION  
      
        FUNCTION remove_queen(row, col)  
            REMOVE the pair consisting of row and col FROM the set queens  
            SET the value at position col of cols TO zero  
            SET the value at position row MINUS col of hills TO zero  
            SET the value at position row PLUS col of dales TO zero  
        END FUNCTION  
      
        FUNCTION add_solution()  
            SET solution TO an empty list  
            FOR each pair of ignored value and col IN the collection queens sorted by the first element of each pair  
                APPEND to solution the concatenation of a string consisting of a period repeated col times PLUS the character Q PLUS a string consisting of a period repeated the difference between n MINUS col MINUS one times  
            END FOR  
            APPEND solution TO output  
        END FUNCTION  
      
        FUNCTION backtrack(row)  
            FOR col FROM zero TO n MINUS one  
                IF is_not_under_attack(row, col)  
                    CALL place_queen WITH arguments row and col  
                    IF the sum of row PLUS one EQUALS n  
                        CALL add_solution  
                    ELSE  
                        CALL backtrack WITH the argument of the sum of row PLUS one  
                    END IF  
                    CALL remove_queen WITH arguments row and col  
                END IF  
            END FOR  
        END FUNCTION  
      
        SET cols TO a list of zero repeated n times  
        SET hills TO a list of zero repeated the result of two MULTIPLIED BY n MINUS one times  
        SET dales TO a list of zero repeated the result of two MULTIPLIED BY n MINUS one times  
        SET queens TO an empty set  
        SET output TO an empty list  
        CALL backtrack WITH zero  
        RETURN output  
    END FUNCTION  
END CLASS
