CLASS TreeNode  
    FUNCTION __init__ with parameters self val WITH DEFAULT zero left WITH DEFAULT None right WITH DEFAULT None  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH parameter values_list  
    IF the values_list IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with element at position one of values_list  
    SET index_counter TO one  
    SET queue_collection TO an empty double ended queue  
    APPEND root TO queue_collection  
    WHILE the queue_collection IS not empty  
        SET node TO the element removed from the left end of queue_collection  
        IF index_counter IS LESS THAN the LENGTH OF values_list AND the element at position index_counter of values_list IS NOT None  
            SET node left TO a new TreeNode with element at position index_counter of values_list  
            APPEND node left TO queue_collection  
        END IF  
        INCREMENT index_counter BY one  
        IF index_counter IS LESS THAN the LENGTH OF values_list AND the element at position index_counter of values_list IS NOT None  
            SET node right TO a new TreeNode with element at position index_counter of values_list  
            APPEND node right TO queue_collection  
        END IF  
        INCREMENT index_counter BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH parameters p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val attribute of p IS NOT EQUALS TO the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree WITH the left attribute of p AND the left attribute of q AND the result of calling is_same_tree WITH the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten WITH parameters self root  
        IF root IS None  
            RETURN  
        END IF  
        SET stack_collection TO a list containing root  
        WHILE stack_collection IS not empty  
            SET node TO the element removed from the end of stack_collection  
            IF the right attribute of node EXISTS  
                APPEND the right attribute of node TO stack_collection  
            END IF  
            IF the left attribute of node EXISTS  
                APPEND the left attribute of node TO stack_collection  
            END IF  
            SET the left attribute of node TO None  
            IF stack_collection IS not empty  
                SET the right attribute of node TO the element at the last position of stack_collection  
            ELSE  
                SET the right attribute of node TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
