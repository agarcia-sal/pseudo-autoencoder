CLASS Solution  
    FUNCTION numDistinct(s string_parameter t string_parameter)  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
        
        CREATE a two dimensional collection named dp with m PLUS one elements each containing collections of n PLUS one elements all set TO zero  
        
        FOR index i FROM zero TO m  
            SET the element at position i in dp at position zero TO one  
        END FOR  
        
        FOR index i FROM one TO m  
            FOR index j FROM one TO n  
                IF the character at position i MINUS one of s EQUALS the character at position j MINUS one of t THEN  
                    SET the element at position i in dp at position j TO the element at position i MINUS one in dp at position j MINUS one PLUS the element at position i MINUS one in dp at position j  
                ELSE  
                    SET the element at position i in dp at position j TO the element at position i MINUS one in dp at position j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position m in dp at position n  
    END FUNCTION  
END CLASS
