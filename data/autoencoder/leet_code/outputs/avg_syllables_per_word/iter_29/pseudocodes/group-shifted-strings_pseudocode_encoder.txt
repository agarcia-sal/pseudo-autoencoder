CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF the string s is empty  
                RETURN the empty string  
            END IF  
            SET a list called canonical_form TO an empty list  
            SET first_character_code TO the position in the alphabet of the character at position one of string s  
            FOR each character c IN string s  
                SET character_code TO the position in the alphabet of character c  
                SET difference TO character_code MINUS first_character_code  
                IF difference IS LESS THAN zero  
                    INCREMENT difference BY twenty six  
                END IF  
                APPEND the string representation of difference TO canonical_form  
            END FOR  
            SET joined_canonical_form TO the concatenation of each element in canonical_form separated by the character comma  
            RETURN joined_canonical_form  
        END FUNCTION  
        
        SET groups TO a dictionary that returns an empty list when the key does not exist  
        
        FOR each string s IN the list strings  
            SET canonical_form TO get_canonical_form applied to string s  
            APPEND string s TO the list at key canonical_form in groups  
        END FOR  
        
        RETURN the list of all values in groups  
    END FUNCTION  
END CLASS
