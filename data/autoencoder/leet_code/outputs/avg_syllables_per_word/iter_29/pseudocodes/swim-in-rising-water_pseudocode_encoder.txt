CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an EMPTY set  
        SET min_heap TO a COLLECTION CONTAINING a TUPLE of the element at position zero zero of grid the NUMBER zero and the NUMBER zero  
        ADD the TUPLE of the NUMBER zero and the NUMBER zero TO visited  
        
        SET directions TO a COLLECTION CONTAINING the TUPLE of the NUMBER zero and the NUMBER one the TUPLE of the NUMBER one and the NUMBER zero the TUPLE of the NUMBER zero and the NEGATIVE NUMBER one the TUPLE of the NEGATIVE NUMBER one and the NUMBER zero  
        
        WHILE min_heap is NOT EMPTY  
            REMOVE the element with the SMALLEST first value FROM min_heap AND ASSIGN TO t i j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            
            FOR each di dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND the TUPLE of ni nj IS NOT IN visited  
                    ADD the TUPLE of ni nj TO visited  
                    INSERT a TUPLE of the GREATER VALUE BETWEEN t AND the element at position ni nj of grid ni nj INTO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the NEGATIVE NUMBER one  
    END FUNCTION  
END CLASS
