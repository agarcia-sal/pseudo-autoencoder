CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO a new empty set  
        FOR each mine IN mines  
            CONVERT mine to a tuple form  
            ADD the tuple form of mine TO mines_set  
        END FOR  
        
        SET dp TO a two dimensional list of size n BY n  
        EACH element of dp IS a list containing four zeros representing counts in four directions  
        
        SET directions TO a list containing the pairs zero and one one and zero zero and negative one negative one and zero  
        
        FOR integer i FROM zero TO n MINUS one  
            FOR integer j FROM zero TO n MINUS one  
                IF the pair of i and j IS NOT IN mines_set  
                    IF j GREATER THAN zero  
                        SET the element at position zero of dp at position i and j TO the element at position zero of dp at position i and position j MINUS one PLUS one  
                    END IF  
                    IF i GREATER THAN zero  
                        SET the element at position one of dp at position i and j TO the element at position one of dp at position i MINUS one and j PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR integer i FROM n MINUS one DOWN TO zero  
            FOR integer j FROM n MINUS one DOWN TO zero  
                IF the pair of i and j IS NOT IN mines_set  
                    IF j LESS THAN n MINUS one  
                        SET the element at position two of dp at position i and j TO the element at position two of dp at position i and position j PLUS one PLUS one  
                    END IF  
                    IF i LESS THAN n MINUS one  
                        SET the element at position three of dp at position i and j TO the element at position three of dp at position i PLUS one and j PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum of the four elements in dp at position i and j  
                    SET result TO the maximum of result AND plus_sign_order  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
