CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO a list of zeros WITH the LENGTH OF batchSize  
        FOR each group IN groups  
            INCREMENT the element at position the remainder of group DIVIDED BY batchSize OF remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder, variable_number_of_remainders)  
            SET remainder_count TO a list CONTAINING each element of variable_number_of_remainders  
            SET max_happy TO zero  
            
            FOR i FROM zero TO batchSize MINUS one  
                IF the element at position i OF remainder_count EQUALS zero  
                    CONTINUE TO the next iteration  
                END IF  
                
                DECREMENT the element at position i OF remainder_count BY one  
                SET new_remainder TO the remainder of the difference of remainder MINUS i DIVIDED BY batchSize  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS the result of dp CALLED WITH new_remainder AND the elements of remainder_count  
                ELSE  
                    SET happy_groups TO the result of dp CALLED WITH new_remainder AND the elements of remainder_count  
                END IF  
                
                SET max_happy TO the GREATER OF max_happy AND happy_groups  
                
                INCREMENT the element at position i OF remainder_count BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN dp CALLED WITH zero AND the elements of remainder_count  
    END FUNCTION  
END CLASS
