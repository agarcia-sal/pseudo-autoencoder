CLASS Solution  
    FUNCTION numberOfArrays(input_string, limit_k)  
        SET MOD TO ten POWERED BY nine PLUS seven  
        SET length_n TO the LENGTH OF input_string  
        SET dp_list TO a list of zeros of size length_n PLUS one  
        SET element at position zero of dp_list TO one  
        FOR position_i FROM one TO length_n  
            FOR position_j FROM zero TO position_i MINUS one  
                SET character_at_position_j TO character at position position_j of input_string  
                SET substring_from_j_to_i TO substring from position position_j TO position position_i MINUS one of input_string  
                IF character_at_position_j NOT EQUALS zero AND converting substring_from_j_to_i TO number IS GREATER THAN OR EQUAL TO one AND IS LESS THAN OR EQUAL TO limit_k  
                    SET element at position position_i of dp_list TO element at position position_i of dp_list PLUS element at position position_j of dp_list  
                    THEN SET element at position position_i of dp_list TO element at position position_i of dp_list MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position length_n of dp_list  
    END FUNCTION  
END CLASS
