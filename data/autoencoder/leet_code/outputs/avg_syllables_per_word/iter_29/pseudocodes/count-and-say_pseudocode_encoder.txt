CLASS Solution  
    FUNCTION countAndSay(n)  
        IF n EQUALS one  
            RETURN the string representing one  
        END IF  
        
        SET current_sequence TO the string representing one  
        
        FOR each iteration_index FROM two TO n  
            SET next_sequence TO the empty string  
            SET count TO one  
            SET previous_char TO the first character of current_sequence  
            
            FOR each position_index FROM two TO the LENGTH OF current_sequence  
                IF the character at position position_index of current_sequence EQUALS previous_char  
                    INCREMENT count BY one  
                ELSE  
                    APPEND the string representation of count TO next_sequence  
                    APPEND previous_char TO next_sequence  
                    SET previous_char TO the character at position position_index of current_sequence  
                    SET count TO one  
                END IF  
            END FOR  
            
            APPEND the string representation of count TO next_sequence  
            APPEND previous_char TO next_sequence  
            
            SET current_sequence TO next_sequence  
        END FOR  
        
        RETURN current_sequence  
    END FUNCTION  
END CLASS
