CLASS Solution  
    FUNCTION longestPalindromeSubseq(s)  
        SET n TO the LENGTH OF s  
        CREATE dp AS a two dimensional array of size n BY n FILLED WITH zeros  
        FOR each index i FROM zero TO the number one LESS THAN n  
            SET element at row i column i of dp TO one  
        END FOR  
        FOR each length FROM two TO n  
            FOR each index i FROM zero TO n MINUS length PLUS one LESS THAN n  
                SET j TO i PLUS length MINUS one  
                IF the character at position i of s EQUALS the character at position j of s THEN  
                    SET element at row i column j of dp TO element at row i PLUS one column j MINUS one of dp PLUS two  
                ELSE  
                    SET element at row i column j of dp TO the GREATER VALUE BETWEEN element at row i PLUS one column j of dp AND element at row i column j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at row zero column n MINUS one of dp  
    END FUNCTION  
END CLASS
