CLASS Solution  
    FUNCTION firstMissingPositive(nums)  
        SET n TO the LENGTH OF nums  
        
        FOR i FROM zero TO n MINUS one  
            WHILE the value at position i of nums IS GREATER THAN OR EQUAL TO one AND the value at position i of nums IS LESS THAN OR EQUAL TO n AND the value at position indexed by the value at position i of nums MINUS one of nums IS NOT EQUAL TO the value at position i of nums  
                SET correct_index TO the value at position i of nums MINUS one  
                SET temporary_value TO the value at position correct_index of nums  
                SET the value at position correct_index of nums TO the value at position i of nums  
                SET the value at position i of nums TO temporary_value  
            END WHILE  
        END FOR  
        
        FOR i FROM zero TO n MINUS one  
            IF the value at position i of nums IS NOT EQUAL TO i PLUS one  
                RETURN i PLUS one  
            END IF  
        END FOR  
        
        RETURN n PLUS one  
    END FUNCTION  
END CLASS
