CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s_list)  
            SET stack TO an empty list  
            SET sign TO the plus symbol  
            SET number TO zero  
            
            WHILE the LENGTH OF s_list IS GREATER THAN zero  
                SET character TO the element at position zero of s_list  
                REMOVE the element at position zero FROM s_list  
                
                IF character represents a digit  
                    SET number TO number MULTIPLIED BY ten PLUS the numeric value represented by character  
                END IF  
                
                IF character EQUALS the opening parenthesis symbol  
                    SET number TO the result of calling helper with s_list  
                END IF  
                
                IF character DOES NOT represent a digit AND character IS NOT a whitespace character OR the LENGTH OF s_list EQUALS zero  
                    IF sign EQUALS the plus symbol  
                        APPEND number TO stack  
                    ELSE IF sign EQUALS the minus symbol  
                        APPEND the negation of number TO stack  
                    ELSE IF sign EQUALS the multiplication symbol  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY number  
                    ELSE IF sign EQUALS the division symbol  
                        SET the element at the last position of stack TO the integer part of the element at the last position of stack DIVIDED BY number  
                    END IF  
                    
                    SET sign TO character  
                    SET number TO zero  
                END IF  
                
                IF character EQUALS the closing parenthesis symbol  
                    BREAK the loop  
                END IF  
            END WHILE  
            
            RETURN the sum of all elements in stack  
        END FUNCTION  
        
        RETURN the result of calling helper with the list formed by each character in s  
    END FUNCTION  
END CLASS
