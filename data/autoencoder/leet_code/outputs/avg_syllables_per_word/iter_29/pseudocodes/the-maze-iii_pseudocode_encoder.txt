CLASS Solution  
    FUNCTION findShortestWay(maze, ball, hole)  
        SET directions TO list of elements each consisting of zero zero and letter r then one zero and letter d then zero minus one and letter l then minus one zero and letter u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        ASSIGN start TO the tuple CONSTRUCTED FROM ball  
        ASSIGN hole TO the tuple CONSTRUCTED FROM hole  
        SET pq TO a list containing a tuple consisting of zero distance an empty string path and start position  
        SET visited TO an empty set  
        
        WHILE pq is NOT empty  
            SET dist path and position x y TO the result of REMOVING the SMALLEST ELEMENT from pq  
            
            IF the position x y EQUALS hole  
                RETURN path  
            END IF  
            
            IF the position x y IS IN visited  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            
            ADD the position x y TO visited  
            
            FOR each dx dy and direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x plus dx IS GREATER THAN OR EQUAL TO zero AND new_x plus dx IS LESS THAN m AND new_y plus dy IS GREATER THAN OR EQUAL TO zero AND new_y plus dy IS LESS THAN n AND the element at position new_x plus dx of maze at position new_y plus dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                    IF the position new_x new_y EQUALS hole  
                        BREAK out of the WHILE loop  
                    END IF  
                END WHILE  
                
                IF the position new_x new_y IS NOT IN visited  
                    ADD a tuple consisting of dist plus count the concatenation of path and direction and the tuple consisting of new_x and new_y TO pq as a priority QUEUE element  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
