CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict word_list word1 target_word1 word2 target_word2) RETURNS integer  
        SET min_distance TO positive infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO the result of comparison that target_word1 EQUALS target_word2  
        
        FOR index FROM zero TO the LENGTH OF word_list MINUS one  
            SET word TO element at position index of word_list  
            
            IF word EQUALS target_word1  
                IF same_word IS true  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO index  
            END IF  
            
            IF word EQUALS target_word2  
                IF same_word IS false  
                    SET prev_word2 TO index  
                END IF  
            END IF  
            
            IF prev_word1 IS NOT EQUAL TO negative one AND prev_word2 IS NOT EQUAL TO negative one  
                SET current_distance TO the absolute value of prev_word1 MINUS prev_word2  
                IF current_distance LESS THAN min_distance  
                    SET min_distance TO current_distance  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
