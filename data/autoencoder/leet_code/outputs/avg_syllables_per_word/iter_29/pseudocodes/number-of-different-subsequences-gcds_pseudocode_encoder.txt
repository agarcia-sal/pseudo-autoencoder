CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET maximum_number TO the maximum value among elements of nums  
        SET number_set TO the set of all elements in nums  
        SET count_of_gcds TO zero  
        FOR each integer g FROM one TO maximum_number INCLUSIVE  
            SET gcd_value TO zero  
            FOR each integer m FROM g TO maximum_number INCLUSIVE IN STEPS OF g  
                IF m IS AN ELEMENT OF number_set THEN  
                    SET gcd_value TO the greatest common divisor of gcd_value AND m  
                    IF gcd_value EQUALS g THEN  
                        INCREMENT count_of_gcds BY one  
                        BREAK OUT OF the inner FOR loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count_of_gcds  
    END FUNCTION  
END CLASS
