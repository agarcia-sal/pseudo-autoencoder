CLASS Solution  
    FUNCTION minimumOneBitOperations(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        
        SET k TO zero  
        WHILE one POWERED BY k PLUS one POWERED BY one IS LESS THAN OR EQUAL TO n  
            INCREMENT k BY one  
        END WHILE  
        
        SET most_significant_bit TO one POWERED BY k  
        
        RETURN (one POWERED BY k PLUS one POWERED BY one) MINUS one MINUS self CALL minimumOneBitOperations WITH ARGUMENT n EXCLUSIVE OR most_significant_bit  
    END FUNCTION  
END CLASS
