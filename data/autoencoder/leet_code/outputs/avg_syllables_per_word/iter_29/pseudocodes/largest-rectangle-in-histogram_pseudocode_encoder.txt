CLASS Solution  
    FUNCTION largestRectangleArea(heights)  
        SET stack TO a list containing negative one  
        SET max_area TO zero  
        APPEND zero TO heights  
        FOR each index i AND each height h IN heights with their enumeration  
            WHILE the last element of stack DOES NOT EQUAL negative one AND height h IS LESS THAN the element at position of the last element of stack in heights  
                SET height TO the element at the position popped from stack in heights  
                SET width TO index i MINUS the last element of stack MINUS one  
                SET max_area TO the greater value BETWEEN max_area AND height MULTIPLIED BY width  
            END WHILE  
            APPEND index i TO stack  
        END FOR  
        RETURN max_area  
    END FUNCTION  
END CLASS
