CLASS Solution  
    FUNCTION setZeroes(matrix)  
        IF matrix IS empty OR the LENGTH OF matrix IS zero OR the LENGTH OF element at position zero of matrix IS zero  
            RETURN  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        
        SET first_row_has_zero TO a boolean VALUE THAT IS true IF any element at position j of element at position zero of matrix EQUALS zero FOR each j FROM zero TO n MINUS one ELSE false  
        SET first_col_has_zero TO a boolean VALUE THAT IS true IF any element at position i of matrix at position zero EQUALS zero FOR each i FROM zero TO m MINUS one ELSE false  
        
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF element at position i of matrix at position j EQUALS zero  
                    SET element at position i of matrix at position zero TO zero  
                    SET element at position zero of matrix at position j TO zero  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF element at position i of matrix at position zero EQUALS zero OR element at position zero of matrix at position j EQUALS zero  
                    SET element at position i of matrix at position j TO zero  
                END IF  
            END FOR  
        END FOR  
        
        IF first_row_has_zero IS true  
            FOR j FROM zero TO n MINUS one  
                SET element at position zero of matrix at position j TO zero  
            END FOR  
        END IF  
        
        IF first_col_has_zero IS true  
            FOR i FROM zero TO m MINUS one  
                SET element at position i of matrix at position zero TO zero  
            END FOR  
        END IF  
    END FUNCTION  
END CLASS
