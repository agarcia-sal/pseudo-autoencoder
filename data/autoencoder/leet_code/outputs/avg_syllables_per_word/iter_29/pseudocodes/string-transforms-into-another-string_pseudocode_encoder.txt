CLASS Solution  
    FUNCTION canConvert(str1 str2)  
        IF str1 EQUALS str2  
            RETURN true  
        END IF  
        SET conversion_map TO empty mapping  
        FOR each character_one AND character_two IN pairs formed by corresponding positions of str1 AND str2  
            IF character_one IS FOUND IN conversion_map  
                IF value associated with character_one in conversion_map DOES NOT EQUAL character_two  
                    RETURN false  
                END IF  
            ELSE  
                SET value associated with character_one in conversion_map TO character_two  
            END ELSE  
        END FOR  
        IF the number of unique characters contained in str2 EQUALS twenty six  
            RETURN false  
        END IF  
        RETURN true  
    END FUNCTION  
END CLASS
