CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET length_of_frag TO the LENGTH OF frag  
            FOR digit_position FROM one TO length_of_frag  
                SET left_part TO the SUBSTRING OF frag FROM position one TO digit_position  
                SET right_part TO the SUBSTRING OF frag FROM position digit_position PLUS one TO the END OF frag  
                IF the FIRST CHARACTER of left_part IS NOT zero OR left_part EQUALS zero AND the LAST CHARACTER of right_part IS NOT zero  
                    YIELD the CONCATENATION of left_part PLUS the CHARACTER period IF digit_position IS NOT EQUAL TO length_of_frag OTHERWISE empty STRING PLUS right_part  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO the SUBSTRING OF s FROM position two TO the SECOND LAST CHARACTER OF s  

        RETURN the LIST of CONCATENATIONS of left_element PLUS the CHARACTER comma PLUS space PLUS right_element ENCLOSED IN PARENTHESES FOR each index FROM one TO the LENGTH OF s MINUS one FOR each left_element AND right_element IN the CARTESIAN PRODUCT OF make OF the SUBSTRING OF s FROM position one TO index AND make OF the SUBSTRING OF s FROM position index PLUS one TO the END OF s  
    END FUNCTION  
END CLASS
