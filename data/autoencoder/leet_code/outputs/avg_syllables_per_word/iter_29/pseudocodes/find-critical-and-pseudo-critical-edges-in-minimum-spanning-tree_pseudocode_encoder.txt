CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO a list of numbers FROM zero TO n MINUS one  
        SET self.rank TO a list containing the number one repeated exactly n times  
    END FUNCTION  
  
    FUNCTION find(self, u)  
        IF the element at position u of self.parent NOT EQUALS u  
            SET the element at position u of self.parent TO the result of calling find with the element at position u of self.parent  
        END IF  
        RETURN the element at position u of self.parent  
    END FUNCTION  
  
    FUNCTION union(self, u, v)  
        SET rootU TO the result of calling find with u  
        SET rootV TO the result of calling find with v  
        
        IF rootU NOT EQUALS rootV  
            IF the element at position rootU of self.rank GREATER THAN the element at position rootV of self.rank  
                SET the element at position rootV of self.parent TO rootU  
            ELSE IF the element at position rootU of self.rank LESS THAN the element at position rootV of self.rank  
                SET the element at position rootU of self.parent TO rootV  
            ELSE  
                SET the element at position rootV of self.parent TO rootU  
                INCREMENT the element at position rootU of self.rank BY one  
            END ELSE IF  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)  
        FOR index FROM zero TO the LENGTH OF edges MINUS one  
            APPEND index TO the element at position index of edges  
        END FOR  
  
        SORT edges BY the value at position two of each element IN ascending order  
  
        FUNCTION mst(exclude, include)  
            SET uf TO a new instance of UnionFind with parameter n  
            SET weight TO zero  
            
            IF include IS NOT None  
                CALL the union method of uf WITH parameters the element at position zero of include AND the element at position one of include  
                INCREMENT weight BY the element at position two of include  
            END IF  
            
            FOR each element u, v, w, idx IN edges  
                IF idx EQUALS exclude  
                    CONTINUE to the next iteration  
                END IF  
                IF the result of calling the union method of uf WITH parameters u AND v IS True  
                    INCREMENT weight BY w  
                END IF  
            END FOR  
            
            SET root TO the result of calling the find method of uf WITH parameter zero  
            FOR i FROM zero TO n MINUS one  
                IF the result of calling the find method of uf WITH parameter i NOT EQUALS root  
                    RETURN positive infinity  
                END IF  
            END FOR  
            
            RETURN weight  
        END FUNCTION  
  
        SET mst_weight TO the result of calling mst WITH no parameters passed  
          
        SET critical TO an empty list  
        SET pseudo_critical TO an empty list  
  
        FOR each element u, v, w, idx IN edges  
            IF the result of calling mst WITH exclude equal to idx GREATER THAN mst_weight  
                APPEND idx TO critical  
            ELSE IF the result of calling mst WITH include equal to the tuple containing u, v, w EQUALS mst_weight  
                APPEND idx TO pseudo_critical  
            END ELSE IF  
        END FOR  
  
        RETURN a list containing critical FOLLOWED BY pseudo_critical  
    END FUNCTION  
END CLASS
