CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO an empty list  
        SET lower_bound TO negative infinity  
        FOR each value IN preorder  
            IF value IS LESS THAN lower_bound  
                RETURN false  
            END IF  
            WHILE the stack IS NOT empty AND value IS GREATER THAN the element at position last of stack  
                SET lower_bound TO the result of removing and returning the last element from stack  
            END WHILE  
            APPEND value TO stack  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
