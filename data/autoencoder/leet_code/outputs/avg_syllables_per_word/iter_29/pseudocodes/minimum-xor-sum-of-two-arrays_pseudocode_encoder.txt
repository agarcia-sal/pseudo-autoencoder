CLASS Solution  
    FUNCTION minimumXORSum(nums1 PARAMETER, nums2 PARAMETER)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a collection of length equal to two POWERED BY n with all elements set TO infinity  
        SET element at position zero of dp TO zero  
        
        FOR each mask FROM zero TO two POWERED BY n MINUS one  
            SET bit_count TO the COUNT OF characters EQUAL TO one IN the BINARY REPRESENTATION OF mask  
            FOR each j FROM zero TO n MINUS one  
                IF the BITWISE AND of mask AND two POWERED BY j IS NOT zero  
                    SET dp at position mask TO the MINIMUM OF dp at position mask AND the SUM OF dp at position the BITWISE XOR of mask AND the BITWISE NOT of two POWERED BY j AND the BITWISE XOR of element at position bit_count MINUS one of nums1 AND element at position j of nums2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position two POWERED BY n MINUS one of dp  
    END FUNCTION  
END CLASS
