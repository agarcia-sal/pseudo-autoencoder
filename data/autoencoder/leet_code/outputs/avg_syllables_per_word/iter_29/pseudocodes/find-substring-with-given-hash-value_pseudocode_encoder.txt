CLASS Solution  
    FUNCTION subStrHash(s power modulo k hashValue)  
        FUNCTION val(character)  
            RETURN the Unicode code point of character MINUS the Unicode code point of lowercase letter a PLUS one  
        END FUNCTION  
          
        SET length_of_s TO the LENGTH OF s  
        SET current_hash TO zero  
        SET power_to_zero TO one  
          
        FOR index FROM length_of_s MINUS k TO length_of_s MINUS one  
            SET current_hash TO current_hash PLUS val of character at position index of s MULTIPLIED BY power_to_zero THEN MODULO modulo  
            IF index LESS THAN length_of_s MINUS one  
                SET power_to_zero TO power_to_zero MULTIPLIED BY power THEN MODULO modulo  
            END IF  
        END FOR  
          
        SET start_index TO length_of_s MINUS k  
        IF current_hash EQUALS hashValue  
            RETURN substring from position start_index TO position start_index PLUS k MINUS one of s  
        END IF  
          
        SET power_to_k_minus_one TO power_to_zero  
        FOR index FROM length_of_s MINUS k MINUS one DOWN TO zero DECREMENT BY one  
            SET current_hash TO current_hash MINUS val of character at position index PLUS k of s MULTIPLIED BY power_to_k_minus_one THEN MODULO modulo  
            SET current_hash TO current_hash MULTIPLIED BY power PLUS val of character at position index of s THEN MODULO modulo  
            IF current_hash EQUALS hashValue  
                SET start_index TO index  
            END IF  
        END FOR  
          
        RETURN substring from position start_index TO position start_index PLUS k MINUS one of s  
    END FUNCTION  
END CLASS
