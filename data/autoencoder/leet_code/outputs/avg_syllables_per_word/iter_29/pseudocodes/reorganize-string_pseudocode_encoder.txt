CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO the frequency count of each character in s  
        SET max_heap TO the list of pairs where each pair consists of the negative of the frequency and the corresponding character for each character and frequency in count  
        CONVERT max_heap INTO a heap structure to allow efficient retrieval of the smallest element  
        SET prev_char TO none  
        SET prev_freq TO zero  
        SET result TO an empty list  
        WHILE the heap max_heap is not empty OR prev_freq is less than zero  
            IF the heap max_heap is empty AND prev_freq is less than zero  
                RETURN an empty string  
            END IF  
            REMOVE the element with the smallest value from max_heap and ASSIGN the first part to freq and the second part to char  
            APPEND char TO result  
            INCREMENT freq BY one  
            IF prev_freq is less than zero  
                ADD the pair consisting of prev_freq and prev_char back into max_heap maintaining the heap property  
            END IF  
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        RETURN the concatenation of all characters in result  
    END FUNCTION  
END CLASS
