CLASS Solution  
    FUNCTION countVowelPermutation(n)  
        SET MODULO_DIVISOR TO ten POWERED BY nine PLUS seven  
        SET count_of_a TO one  
        SET count_of_e TO one  
        SET count_of_i TO one  
        SET count_of_o TO one  
        SET count_of_u TO one  
        FOR each integer FROM two TO n  
            SET next_count_of_a TO count_of_e PLUS count_of_i PLUS count_of_u  
            SET next_count_of_e TO count_of_a PLUS count_of_i  
            SET next_count_of_i TO count_of_e PLUS count_of_o  
            SET next_count_of_o TO count_of_i  
            SET next_count_of_u TO count_of_i PLUS count_of_o  
            SET count_of_a TO next_count_of_a MODULO MODULO_DIVISOR  
            SET count_of_e TO next_count_of_e MODULO MODULO_DIVISOR  
            SET count_of_i TO next_count_of_i MODULO MODULO_DIVISOR  
            SET count_of_o TO next_count_of_o MODULO MODULO_DIVISOR  
            SET count_of_u TO next_count_of_u MODULO MODULO_DIVISOR  
        END FOR  
        SET total_count TO count_of_a PLUS count_of_e PLUS count_of_i PLUS count_of_o PLUS count_of_u  
        SET result TO total_count MODULO MODULO_DIVISOR  
        RETURN result  
    END FUNCTION  
END CLASS
