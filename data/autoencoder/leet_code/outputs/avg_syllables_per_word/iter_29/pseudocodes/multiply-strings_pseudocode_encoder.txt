CLASS Solution  
    FUNCTION multiply(num1 num1_string num2 num2_string) RETURNS string  
        IF num1 EQUALS the string zero OR num2 EQUALS the string zero  
            RETURN the string zero  
        END IF  

        SET result_list TO a list of zeros with the LENGTH OF num1 PLUS the LENGTH OF num2  

        SET num1 TO the reverse of string num1  
        SET num2 TO the reverse of string num2  

        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the difference between the character code at position i of num1 AND the character code of character zero MULTIPLIED BY the difference between the character code at position j of num2 AND the character code of character zero  
                SET p1 TO the sum of i AND j  
                SET p2 TO the sum of i AND j AND one  
                SET sum_value TO mul PLUS the element at position p1 of result_list  

                SET the element at position p1 of result_list TO the remainder when sum_value IS DIVIDED BY ten  
                INCREMENT the element at position p2 of result_list BY the quotient when sum_value IS DIVIDED BY ten  
            END FOR  
        END FOR  

        SET reversed_result_string TO the concatenation of the elements of result_list in reverse order converted to their string representations  

        RETURN reversed_result_string with the leading characters equal to the string zero removed  
    END FUNCTION  
END CLASS
