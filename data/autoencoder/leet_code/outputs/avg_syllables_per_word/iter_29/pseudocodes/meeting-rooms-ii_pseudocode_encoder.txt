CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF the list intervals IS empty  
            RETURN zero  
        END IF  

        SORT the list intervals by the first element of each sublist representing start time  

        SET min_heap TO an empty list representing a minimum heap of meeting end times  

        FOR each meeting IN intervals  
            IF the list min_heap IS NOT empty AND the first element of meeting representing start time IS GREATER THAN OR EQUAL TO the first element of min_heap representing the earliest meeting end time  
                REMOVE the smallest element from min_heap representing freeing up a meeting room  
            END IF  
            ADD the second element of meeting representing meeting end time TO min_heap using heap push operation  
        END FOR  

        RETURN the LENGTH of min_heap representing the number of meeting rooms required  
    END FUNCTION  
END CLASS
