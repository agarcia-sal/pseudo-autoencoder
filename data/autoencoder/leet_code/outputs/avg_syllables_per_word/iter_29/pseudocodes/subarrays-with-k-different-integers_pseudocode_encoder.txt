CLASS Solution  
    FUNCTION subarraysWithKDistinct(nums k)  
        FUNCTION atMostKDistinct(t)  
            SET count TO a new default dictionary with integer default  
            SET left TO zero  
            SET result TO zero  
            FOR each right FROM zero TO the LENGTH OF nums MINUS one  
                INCREMENT the value associated with element at position right of nums IN count BY one  
                WHILE the number of keys in count GREATER THAN t  
                    DECREMENT the value associated with element at position left of nums IN count BY one  
                    IF the value associated with element at position left of nums IN count EQUALS zero  
                        REMOVE the key element at position left of nums FROM count  
                    END IF  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT result BY right MINUS left PLUS one  
            END FOR  
            RETURN result  
        END FUNCTION  
        RETURN the value returned by atMostKDistinct with argument k MINUS the value returned by atMostKDistinct with argument k MINUS one  
    END FUNCTION  
END CLASS
