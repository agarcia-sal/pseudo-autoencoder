CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        CREATE dp AS a two dimensional collection of size n PLUS one by n PLUS one filled with zeroes  

        FOR j FROM zero TO n INCLUSIVE  
            SET element at position zero in dp at position j TO one  
        END FOR  

        FOR i FROM one TO n INCLUSIVE  
            SET prefix_sum TO zero  
            IF character at position i MINUS one of s EQUALS the character representing increasing  
                FOR j FROM zero TO n MINUS i INCLUSIVE  
                    SET prefix_sum TO prefix_sum PLUS element at position i MINUS one in dp at position j THEN TAKE REMAINDER AFTER DIVISION BY MOD  
                    SET element at position i in dp at position j TO prefix_sum  
                END FOR  
            ELSE  
                FOR j FROM n MINUS i DOWN TO zero INCLUSIVE  
                    SET prefix_sum TO prefix_sum PLUS element at position i MINUS one in dp at position j PLUS one THEN TAKE REMAINDER AFTER DIVISION BY MOD  
                    SET element at position i in dp at position j TO prefix_sum  
                END FOR  
            END IF  
        END FOR  

        SET result TO the SUM OF elements in the collection at position n in dp THEN TAKE REMAINDER AFTER DIVISION BY MOD  
        RETURN result  
    END FUNCTION  
END CLASS
