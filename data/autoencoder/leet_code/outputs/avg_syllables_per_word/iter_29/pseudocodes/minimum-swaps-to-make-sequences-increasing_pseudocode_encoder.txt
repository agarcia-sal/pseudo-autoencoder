CLASS Solution  
    FUNCTION minSwap(nums1 list_of_integers nums2 list_of_integers) RETURNS integer  
        SET n TO the LENGTH OF nums1  
        SET dp TO a new list with n elements each containing a list of two zeros  
        
        SET element at position zero at index zero of dp TO zero  
        SET element at position zero at index one of dp TO one  
        
        FOR i FROM one TO n MINUS one  
            IF the element at position i of nums1 GREATER THAN the element at position i MINUS one of nums1 AND the element at position i of nums2 GREATER THAN the element at position i MINUS one of nums2 THEN  
                IF the element at position i of nums1 GREATER THAN the element at position i MINUS one of nums2 AND the element at position i of nums2 GREATER THAN the element at position i MINUS one of nums1 THEN  
                    SET element at position i at index zero of dp TO the MINIMUM OF element at position i MINUS one at index zero of dp AND element at position i MINUS one at index one of dp  
                    SET element at position i at index one of dp TO the element at position i at index zero of dp PLUS one  
                ELSE  
                    SET element at position i at index zero of dp TO element at position i MINUS one at index zero of dp  
                    SET element at position i at index one of dp TO element at position i MINUS one at index one of dp PLUS one  
                END IF  
            ELSE  
                IF the element at position i of nums1 GREATER THAN the element at position i MINUS one of nums2 AND the element at position i of nums2 GREATER THAN the element at position i MINUS one of nums1 THEN  
                    SET element at position i at index zero of dp TO element at position i MINUS one at index one of dp  
                    SET element at position i at index one of dp TO element at position i MINUS one at index zero of dp PLUS one  
                ELSE  
                    SET element at position i at index zero of dp TO element at position i MINUS one at index one of dp  
                    SET element at position i at index one of dp TO element at position i MINUS one at index zero of dp PLUS one  
                END IF  
            END IF  
        END FOR  
        
        RETURN the MINIMUM OF element at position n MINUS one at index zero of dp AND element at position n MINUS one at index one of dp  
    END FUNCTION  
END CLASS
