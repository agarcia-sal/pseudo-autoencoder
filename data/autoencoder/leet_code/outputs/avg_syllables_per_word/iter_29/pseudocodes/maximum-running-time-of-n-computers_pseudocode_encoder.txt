CLASS Solution  
    FUNCTION maxRunTime(n, batteries)  
        FUNCTION canRunFor(time)  
            SET total TO zero  
            FOR each battery IN batteries  
                INCREMENT total BY the smaller VALUE BETWEEN battery AND time  
            END FOR  
            RETURN WHETHER total IS GREATER THAN OR EQUAL TO time MULTIPLIED BY n  
        END FUNCTION  
        SET left TO zero  
        SET right TO the INTEGER PART OF the SUM OF batteries DIVIDED BY n  
        WHILE left IS LESS THAN right  
            SET mid TO the INTEGER PART OF left PLUS right PLUS one DIVIDED BY two  
            IF canRunFor(mid) IS TRUE  
                SET left TO mid  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
