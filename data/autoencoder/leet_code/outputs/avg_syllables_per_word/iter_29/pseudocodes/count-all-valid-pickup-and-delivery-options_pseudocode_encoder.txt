CLASS Solution  
    FUNCTION countOrders(n)  
        SET MOD to ten POWERED BY nine PLUS one  
        SET count to one  
        FOR i FROM two TO n INCLUSIVE  
            SET count to count MULTIPLIED BY i MULTIPLIED BY the result of two MULTIPLIED BY i MINUS one  
            SET count to count MODULO MOD  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
