CLASS Solution  
    FUNCTION tilingRectangle(n, m)  
        FUNCTION backtrack(list_of_heights, current_moves)  
            IF every element IN list_of_heights EQUALS n THEN  
                SET min_moves TO the MINIMUM OF min_moves AND current_moves  
                RETURN  
            END IF  
            
            IF current_moves GREATER THAN OR EQUAL TO min_moves THEN  
                RETURN  
            END IF  
            
            SET minimum_height TO the MINIMUM VALUE IN list_of_heights  
            SET left_position TO the POSITION OF minimum_height IN list_of_heights  
            SET right_position TO left_position  
            WHILE right_position LESS THAN m AND element at position right_position of list_of_heights EQUALS minimum_height DO  
                INCREMENT right_position BY one  
            END WHILE  
            
            FOR size FROM right_position MINUS left_position DOWN TO one DO  
                IF for EACH index i FROM zero TO size MINUS one THE element at position left_position PLUS i of list_of_heights PLUS size LESS THAN OR EQUAL TO n THEN  
                    SET new_list_of_heights TO a COPY OF list_of_heights  
                    FOR index i FROM zero TO size MINUS one DO  
                        INCREMENT element at position left_position PLUS i of new_list_of_heights BY size  
                    END FOR  
                    CALL backtrack WITH new_list_of_heights AND current_moves PLUS one  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET min_moves TO a very large number representing infinity  
        CALL backtrack WITH a list consisting of zero repeated m times AND zero  
        RETURN min_moves  
    END FUNCTION  
END CLASS
