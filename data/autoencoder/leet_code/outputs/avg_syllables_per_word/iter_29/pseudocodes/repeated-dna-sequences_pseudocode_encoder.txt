CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO an empty dictionary with default value zero  
        FOR each index FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO the substring from position index TO position index PLUS ten MINUS one of string s  
            INCREMENT the value associated with sequence in sequence_count BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each sequence_key AND sequence_value IN all key value pairs of sequence_count  
            IF sequence_value GREATER THAN one  
                APPEND sequence_key TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
