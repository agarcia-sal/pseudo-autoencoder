CLASS Solution  
    FUNCTION waysToPartition(nums WITH TYPE List OF integers, k WITH TYPE integer) RETURNS integer  
        SET total_sum TO the SUM OF elements IN nums  
        SET n TO the LENGTH OF nums  
        
        SET left_diff TO a new default dictionary mapping keys TO zero  
        SET right_diff TO a new default dictionary mapping keys TO zero  
        
        SET current_sum TO zero  
        FOR index FROM zero TO n MINUS one MINUS one  
            INCREMENT current_sum BY element at position index of nums  
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            INCREMENT value at key diff IN right_diff BY one  
        END FOR  
        
        SET max_ways TO the value at key zero IN right_diff  
        
        SET current_sum TO zero  
        FOR index FROM zero TO n MINUS one  
            INCREMENT current_sum BY element at position index of nums  
            
            SET diff_change TO k MINUS element at position index of nums  
            
            SET new_ways TO the value at key which is the NEGATIVE of diff_change IN left_diff PLUS the value at key diff_change IN right_diff  
            IF new_ways GREATER THAN max_ways  
                SET max_ways TO new_ways  
            END IF  
            
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            DECREMENT value at key diff IN right_diff BY one  
            INCREMENT value at key diff IN left_diff BY one  
        END FOR  
        
        RETURN max_ways  
    END FUNCTION  
END CLASS
