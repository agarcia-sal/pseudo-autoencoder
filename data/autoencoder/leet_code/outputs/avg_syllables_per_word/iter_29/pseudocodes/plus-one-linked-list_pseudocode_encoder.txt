CLASS ListNode  
    FUNCTION __init__(val WITH DEFAULT zero, next WITH DEFAULT none)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the list values IS empty  
        RETURN none  
    END IF  
    SET head TO a new instance of ListNode with the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values from position one TO the last position  
        SET node TO a new instance of ListNode with val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the element val at p1 EQUALS the element val at p2 AND the result of calling is_same_list with the element next at p1 and the element next at p2  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne(head)  
        FUNCTION reverseList(node)  
            SET prev TO none  
            WHILE node IS NOT none  
                SET next_node TO the element next at node  
                SET the element next at node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO the result of calling reverseList with head  
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current IS NOT none AND carry EQUALS one  
            SET the element val at current TO the element val at current PLUS carry  
            SET carry TO the result of dividing the element val at current BY ten using integer division  
            SET the element val at current TO the remainder of the element val at current DIVIDED BY ten  
            
            IF the element next at current EQUALS none AND carry EQUALS one  
                SET the element next at current TO a new instance of ListNode with carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO the element next at current  
        END WHILE  
        
        RETURN the result of calling reverseList with reversed_head  
    END FUNCTION  
END CLASS
