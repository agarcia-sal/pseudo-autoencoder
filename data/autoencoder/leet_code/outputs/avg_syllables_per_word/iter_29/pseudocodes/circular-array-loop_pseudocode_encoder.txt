CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1 val2)  
            RETURN (val1 GREATER THAN zero AND val2 GREATER THAN zero) OR (val1 LESS THAN zero AND val2 LESS THAN zero)  
        END FUNCTION  
      
        SET n TO the LENGTH OF nums  
      
        FOR i FROM zero TO n MINUS one  
            SET slow TO i  
            SET fast TO i  
          
            IF element at position i of nums EQUALS zero THEN  
                CONTINUE  
            END IF  
          
            WHILE true  
                SET slow TO the REMAINDER WHEN slow PLUS element at position slow of nums IS DIVIDED BY n  
                IF slow LESS THAN zero THEN  
                    INCREMENT slow BY n  
                END IF  
              
                SET fast TO the REMAINDER WHEN fast PLUS element at position fast of nums IS DIVIDED BY n  
                IF fast LESS THAN zero THEN  
                    INCREMENT fast BY n  
                END IF  
                SET fast TO the REMAINDER WHEN fast PLUS element at position fast of nums IS DIVIDED BY n  
                IF fast LESS THAN zero THEN  
                    INCREMENT fast BY n  
                END IF  
              
                IF NOT sameDirection(element at position slow of nums element at position i of nums) OR NOT sameDirection(element at position fast of nums element at position i of nums) THEN  
                    BREAK  
                END IF  
              
                IF slow EQUALS fast THEN  
                    IF slow EQUALS the REMAINDER WHEN slow PLUS element at position slow of nums IS DIVIDED BY n THEN  
                        BREAK  
                    END IF  
                    RETURN true  
                END IF  
            END WHILE  
        END FOR  
      
        RETURN false  
    END FUNCTION  
END CLASS
