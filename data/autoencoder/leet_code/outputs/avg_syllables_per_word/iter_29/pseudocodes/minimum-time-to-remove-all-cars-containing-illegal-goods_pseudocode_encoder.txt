CLASS Solution  
    FUNCTION minimumTime(s)  
        SET n TO the LENGTH OF s  
        SET left_costs TO a list of zero elements of size n  
        SET right_costs TO a list of zero elements of size n  
        
        IF character at position one of s EQUALS character one  
            SET element at position one of left_costs TO one  
        ELSE  
            SET element at position one of left_costs TO zero  
        END IF  
        
        FOR i FROM two TO n  
            IF character at position i of s EQUALS character zero  
                SET element at position i of left_costs TO element at position i MINUS one of left_costs  
            ELSE  
                SET element at position i of left_costs TO the minimum of i PLUS one AND element at position i MINUS one of left_costs PLUS two  
            END IF  
        END FOR  
        
        IF character at position n of s EQUALS character one  
            SET element at position n of right_costs TO one  
        ELSE  
            SET element at position n of right_costs TO zero  
        END IF  
        
        FOR i FROM the number n MINUS one DOWN TO one  
            IF character at position i of s EQUALS character zero  
                SET element at position i of right_costs TO element at position i PLUS one of right_costs  
            ELSE  
                SET element at position i of right_costs TO the minimum of the number n MINUS i PLUS one AND element at position i PLUS one of right_costs PLUS two  
            END IF  
        END FOR  
        
        SET min_time TO the minimum of element at position n of left_costs AND element at position one of right_costs  
        
        FOR i FROM one TO the number n MINUS one  
            SET min_time TO the minimum of min_time AND element at position i of left_costs PLUS element at position i PLUS one of right_costs  
        END FOR  
        
        RETURN min_time  
    END FUNCTION  
END CLASS
