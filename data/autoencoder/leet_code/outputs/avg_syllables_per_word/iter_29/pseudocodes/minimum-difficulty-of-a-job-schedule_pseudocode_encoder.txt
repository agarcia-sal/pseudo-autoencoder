CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty list_of_integers d integer)  
        SET n TO the LENGTH OF jobDifficulty  
        IF n LESS THAN d THEN  
            RETURN negative one  
        END IF  
        FUNCTION dp(i integer d integer)  
            IF d EQUALS one THEN  
                RETURN the MAXIMUM value IN the SUBLIST OF jobDifficulty FROM position i TO the END  
            END IF  
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR j FROM i TO n MINUS d  
                SET current_max TO the MAXIMUM value BETWEEN current_max AND the ELEMENT AT position j OF jobDifficulty  
                SET min_difficulty TO the MINIMUM value BETWEEN min_difficulty AND THE SUM OF current_max AND THE RESULT OF dp WITH PARAMETERS j PLUS one AND d MINUS one  
            END FOR  
            RETURN min_difficulty  
        END FUNCTION  
        RETURN dp WITH PARAMETERS zero AND d  
END CLASS
