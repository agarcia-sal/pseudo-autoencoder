CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten POWERED BY nine PLUS one  
        SET dp TO a list of zeros with the length of the LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS the character star  
                RETURN nine  
            ELSE IF char EQUALS the character zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END ELSE  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair(char1 char2)  
            IF char1 EQUALS the character star AND char2 EQUALS the character star  
                RETURN fifteen  
            ELSE IF char1 EQUALS the character star  
                IF character at char2 LESS THAN OR EQUAL TO the character six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END ELSE  
            ELSE IF char2 EQUALS the character star  
                IF char1 EQUALS the character one  
                    RETURN nine  
                ELSE IF char1 EQUALS the character two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END ELSE  
            ELSE  
                SET num TO the integer value of the concatenation of char1 and char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END ELSE  
            END ELSE  
        END FUNCTION  
        
        FOR index FROM one TO the LENGTH OF s  
            SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS one of dp MULTIPLIED BY the result of ways_to_decode_single of the character at position index MINUS one of s  
            SET element at position index of dp TO element at position index of dp MODULO MOD  
            
            IF index GREATER THAN one  
                SET element at position index of dp TO element at position index of dp PLUS element at position index MINUS two of dp MULTIPLIED BY the result of ways_to_decode_pair of the character at position index MINUS two of s AND the character at position index MINUS one of s  
                SET element at position index of dp TO element at position index of dp MODULO MOD  
            END IF  
        END FOR  
        
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
