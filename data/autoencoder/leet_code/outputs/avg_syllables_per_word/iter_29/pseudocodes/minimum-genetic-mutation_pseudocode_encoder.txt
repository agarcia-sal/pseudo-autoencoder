CLASS Solution  
    FUNCTION minMutation(start STRING end STRING bank LIST OF STRING) RETURNS INTEGER  
        SET s TO the SET OF elements IN bank  
        SET q TO the DOUBLE ENDED QUEUE containing a PAIR of start AND zero  
        SET mp TO a MAPPING where A MAPS TO the STRING formed by the characters T C G T MAPS TO the STRING formed by the characters A C G C MAPS TO the STRING formed by the characters A T G G MAPS TO the STRING formed by the characters A T C  
        WHILE q IS NOT EMPTY  
            REMOVE and RETURN the FIRST element FROM q AND ASSIGN the FIRST part TO t AND the SECOND part TO step  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR each index i AND character v AT position i IN t  
                FOR each character j IN the STRING that mp MAPS FROM v  
                    SET next TO the STRING formed by the substring of t from position zero TO position i MINUS one CONCATENATED WITH character j CONCATENATED WITH the substring of t from position i PLUS one TO the END OF t  
                    IF next IS AN ELEMENT OF s  
                        APPEND the PAIR of next AND step PLUS one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
