CLASS Solution  
    FUNCTION wiggleSort(nums)  
        CALL a procedure to SORT the list nums IN ASCENDING ORDER  
        SET sorted_nums TO a new list CREATED AS a COPY OF every element IN nums FROM the FIRST position TO THE LAST position  
        SET n TO the LENGTH OF nums  
        SET left TO THE INTEGER PART OF THE RESULT OF n MINUS one DIVIDED BY two  
        SET right TO n MINUS one  
        FOR index FROM zero TO n MINUS one  
            IF the REMAINDER OF index DIVIDED BY two EQUALS zero  
                SET the element at position index OF nums TO the element at position left OF sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET the element at position index OF nums TO the element at position right OF sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
