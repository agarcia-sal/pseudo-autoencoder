CLASS Solution  
    FUNCTION findCircleNum(isConnected)  
        FUNCTION dfs(city)  
            FOR neighbor FROM zero TO n MINUS one  
                IF element at position city of isConnected at position neighbor EQUALS one AND the element at position neighbor of visited NOT EQUALS True  
                    SET element at position neighbor of visited TO True  
                    CALL dfs WITH argument neighbor  
                END IF  
            END FOR  
        END FUNCTION  

        SET n TO the LENGTH OF isConnected  
        SET visited TO a list of False repeated n TIMES  
        SET province_count TO zero  

        FOR city FROM zero TO n MINUS one  
            IF the element at position city of visited NOT EQUALS True  
                CALL dfs WITH argument city  
                INCREMENT province_count BY one  
            END IF  
        END FOR  

        RETURN province_count  
    END FUNCTION  
END CLASS
