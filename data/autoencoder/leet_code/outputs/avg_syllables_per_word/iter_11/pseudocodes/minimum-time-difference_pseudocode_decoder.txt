CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours TO integer VALUE OF substring before colon IN time  
            SET minutes TO integer VALUE OF substring after colon IN time  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  

        SET minutes_list TO empty list  
        FOR each time IN timePoints  
            APPEND result of to_minutes(time) TO minutes_list  
        END FOR  

        SORT minutes_list IN ascending order  

        SET min_diff TO infinity  
        FOR index FROM one TO LENGTH OF minutes_list MINUS one  
            SET current_diff TO element at position index of minutes_list MINUS element at position index MINUS one of minutes_list  
            IF current_diff LESS THAN min_diff  
                SET min_diff TO current_diff  
            END IF  
        END FOR  

        SET circular_diff TO one thousand four hundred forty MINUS last element of minutes_list PLUS first element of minutes_list  
        IF circular_diff LESS THAN min_diff  
            SET min_diff TO circular_diff  
        END IF  

        RETURN min_diff  
    END FUNCTION  
END CLASS
