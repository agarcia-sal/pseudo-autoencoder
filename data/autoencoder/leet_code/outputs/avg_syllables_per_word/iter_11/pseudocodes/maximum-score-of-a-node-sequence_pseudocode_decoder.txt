CLASS Solution  
    FUNCTION maximumScore(scores, edges)  
        SET graph TO a new default dictionary with list as default type  
        FOR each pair u and v IN edges  
            APPEND tuple of score at position v of scores and v TO the list at position u in graph  
            APPEND tuple of score at position u of scores and u TO the list at position v in graph  
        END FOR  
        FOR each node IN graph  
            SET graph at position node TO the three largest elements in graph at position node based on the first element of each tuple  
        END FOR  
        SET max_score TO negative one  
        FOR each pair u and v IN edges  
            FOR each tuple of score1 and x IN graph at position u  
                FOR each tuple of score2 and y IN graph at position v  
                    IF the length of the set consisting of elements x, y, u, and v EQUALS four  
                        SET max_score TO the greater value between max_score and the sum of score1, score2, score at position u of scores, and score at position v of scores  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN max_score  
    END FUNCTION  
END CLASS
