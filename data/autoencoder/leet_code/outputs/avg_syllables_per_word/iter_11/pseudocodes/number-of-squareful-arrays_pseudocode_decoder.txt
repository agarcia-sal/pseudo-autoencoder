CLASS Solution  
    FUNCTION numSquarefulPerms(nums)  
        FUNCTION is_square(n)  
            RETURN the integer part of the square root of n MULTIPLIED BY the integer part of the square root of n EQUALS n  
        END FUNCTION  

        FUNCTION backtrack(path, count)  
            IF the length of path EQUALS the length of nums  
                RETURN one  
            END IF  

            SET ans TO zero  
            FOR each element x IN the keys of count  
                IF the value of count at x GREATER THAN zero AND (path IS empty OR is_square(the last element of path PLUS x))  
                    DECREMENT the value of count at x BY one  
                    INCREMENT ans BY the result of backtrack(the path appended with x, count)  
                    INCREMENT the value of count at x BY one  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  

        SET count TO the occurrence count of elements in nums  
        RETURN backtrack(an empty list, count)  
    END FUNCTION  
END CLASS
