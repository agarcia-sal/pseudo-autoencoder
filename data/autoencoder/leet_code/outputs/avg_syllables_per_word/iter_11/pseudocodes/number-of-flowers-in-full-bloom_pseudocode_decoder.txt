CLASS Solution
    FUNCTION fullBloomFlowers(flowers, people)
        SET start_times TO an empty list
        FOR each flower IN flowers
            APPEND first element of flower TO start_times
        END FOR
        SORT start_times in ascending order

        SET end_times TO an empty list
        FOR each flower IN flowers
            APPEND second element of flower TO end_times
        END FOR
        SORT end_times in ascending order

        SET result TO an empty list

        FOR each person IN people
            SET started TO the count of elements in start_times that are LESS THAN OR EQUAL TO person
            SET ended TO the count of elements in end_times that are LESS THAN person
            SET in_bloom TO started MINUS ended
            APPEND in_bloom TO result
        END FOR

        RETURN result
    END FUNCTION
END CLASS
