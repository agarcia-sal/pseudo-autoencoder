CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO empty dictionary  
        FOR course FROM zero TO numCourses MINUS one  
            SET element at position course of adj_list TO empty list  
        END FOR
        
        SET in_degrees TO list containing numCourses elements each set TO zero
        
        FOR each element [dest, src] IN prerequisites  
            APPEND dest TO element at position src of adj_list  
            INCREMENT element at position dest of in_degrees BY one  
        END FOR
        
        SET queue TO deque containing all courses FROM zero TO numCourses MINUS one WHERE element at position course of in_degrees EQUALS zero
        
        SET topological_order TO empty list
        
        WHILE queue is not empty  
            SET current TO element removed from the start of queue  
            APPEND current TO topological_order
            
            FOR each neighbor IN element at position current of adj_list  
                DECREMENT element at position neighbor of in_degrees BY one
                IF element at position neighbor of in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE
        
        IF length of topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN empty list  
        END IF  
    END FUNCTION  
END CLASS
