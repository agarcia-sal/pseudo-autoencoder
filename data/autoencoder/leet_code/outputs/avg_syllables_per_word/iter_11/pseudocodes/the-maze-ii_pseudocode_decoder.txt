CLASS Solution  
    FUNCTION shortestDistance(maze, start, destination)  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero of maze  
        SET directions TO list containing elements of pair one zero, pair negative one zero, pair zero one, pair zero negative one  
        SET start TO tuple constructed from start  
        SET destination TO tuple constructed from destination  
        SET visited TO an empty set  
        SET pq TO list containing a pair of zero and start  

        WHILE pq is not empty  
            REMOVE element with smallest first value FROM pq  
            SET dist TO the first element of removed pair  
            SET x TO the first element of the second element of removed pair  
            SET y TO the second element of the second element of removed pair  
            
            IF pair x and y EQUALS destination  
                RETURN dist  
            END IF
            
            IF pair x and y IN visited  
                CONTINUE to next iteration of WHILE  
            END IF
            
            ADD pair x and y TO visited  
            
            FOR each pair dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x plus dx IS GREATER THAN OR EQUAL TO zero AND new_x plus dx IS LESS THAN m AND new_y plus dy IS GREATER THAN OR EQUAL TO zero AND new_y plus dy IS LESS THAN n AND element at new_x plus dx, new_y plus dy of maze EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE
                
                IF pair new_x and new_y NOT IN visited  
                    ADD pair dist plus count and pair new_x and new_y TO pq with priority according to first value  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
