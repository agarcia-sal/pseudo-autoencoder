CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL nums SORT  
        SET n TO LENGTH OF nums  
        SET first TO element at position zero of nums  
        FOR i FROM one TO n MINUS one  
            SET k TO (element at position i of nums MINUS first) DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at position i of nums  
                CONTINUE  
            END IF  
            SET count TO a counter of nums  
            SET arr TO empty list  
            FOR each x IN nums  
                IF count at x EQUALS zero  
                    CONTINUE  
                END IF  
                IF count at x PLUS two MULTIPLIED BY k EQUALS zero  
                    BREAK  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT count at x BY one  
                DECREMENT count at x PLUS two MULTIPLIED BY k BY one  
            END FOR  
            IF LENGTH OF arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN empty list  
    END FUNCTION  
END CLASS
