CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO the set containing element two PLUS element five PLUS element six PLUS element nine  
        SET same_digits TO the set containing element zero PLUS element one PLUS element eight  
        
        FUNCTION is_good_number(num)  
            SET num_str TO the string representation of num  
            IF all digit IN the union of change_digits AND same_digits FOR each digit IN num_str THEN  
                RETURN any digit IN change_digits FOR each digit IN num_str  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR i FROM one TO n  
            IF is_good_number(i) EQUALS true THEN  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
