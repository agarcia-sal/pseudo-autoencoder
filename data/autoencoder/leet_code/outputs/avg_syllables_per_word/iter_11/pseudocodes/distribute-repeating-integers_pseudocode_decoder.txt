CLASS Solution  
    FUNCTION canDistribute(nums, quantity)  
        SET count TO the list of values of the frequency counter of nums  
        SORT quantity IN descending order  
        
        FUNCTION can_satisfy(index)  
            IF index EQUALS the LENGTH OF quantity  
                RETURN true  
            END IF  
            FOR i FROM zero TO the LENGTH OF count MINUS one  
                IF the element at position i of count GREATER THAN OR EQUAL TO the element at position index of quantity  
                    DECREMENT the element at position i of count BY the element at position index of quantity  
                    IF can_satisfy(index PLUS one)  
                        RETURN true  
                    END IF  
                    INCREMENT the element at position i of count BY the element at position index of quantity  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN can_satisfy(zero)  
    END FUNCTION  
END CLASS
