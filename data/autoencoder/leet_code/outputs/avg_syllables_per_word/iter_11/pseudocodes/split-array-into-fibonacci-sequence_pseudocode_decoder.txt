CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF the character at position start of num EQUALS zero AND end GREATER THAN start PLUS one  
                    CONTINUE TO the next iteration  
                END IF  
                SET next_num TO the INTEGER value of the substring of num FROM position start TO position end MINUS one  
                IF next_num GREATER THAN OR EQUAL TO two POWERED BY thirty ONE  
                    BREAK the loop  
                END IF  
                IF the LENGTH OF path LESS THAN two OR next_num EQUALS the SUM of the elements at position (the LENGTH OF path MINUS one) AND position (the LENGTH OF path MINUS two) of path  
                    SET result TO the CALL of backtrack WITH parameters end AND path APPENDED WITH next_num  
                    IF result IS NOT EMPTY  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN an empty list  
        END FUNCTION  
        RETURN the CALL of backtrack WITH zero AND an empty list  
    END FUNCTION  
END CLASS
