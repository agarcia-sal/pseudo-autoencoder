CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            
            FOR each index i AND character char IN s  
                IF char EQUALS character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                
                IF count EQUALS zero  
                    APPEND concatenation of character one concatenated WITH result of dfs applied to substring of s from position start PLUS one TO position i concatenated WITH character zero TO specials  
                    SET start TO i PLUS one  
                END IF  
            END FOR  
            
            SORT specials IN reverse lexicographical order  
            RETURN the concatenation of all elements in specials combined into a single string  
        END FUNCTION  
        
        RETURN result of dfs applied to s  
    END FUNCTION  
END CLASS
