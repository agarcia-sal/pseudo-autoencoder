CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO a mapping of each element in nums to its frequency  
        SET unique_nums TO the sorted list of keys in count  
        SET dp TO a list of zeros with length the LENGTH OF unique_nums PLUS one  
        SET the element at position one of dp TO the element at position zero of unique_nums MULTIPLIED BY the count at element at position zero of unique_nums  
        FOR index FROM two TO the LENGTH OF unique_nums  
            SET num TO the element at position index MINUS one of unique_nums  
            IF num EQUALS the element at position index MINUS two of unique_nums PLUS one  
                SET the element at position index of dp TO the MAXIMUM OF the element at position index MINUS one of dp AND the element at position index MINUS two of dp PLUS num MULTIPLIED BY the count at num  
            ELSE  
                SET the element at position index of dp TO the element at position index MINUS one of dp PLUS num MULTIPLIED BY the count at num  
            END IF  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
