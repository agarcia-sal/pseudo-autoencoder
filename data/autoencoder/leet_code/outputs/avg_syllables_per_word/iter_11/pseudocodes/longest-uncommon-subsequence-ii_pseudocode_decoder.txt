CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator TO the iterator of t  
            RETURN whether for every character c in s the character c appears in sequence in iterator  
        END FUNCTION  
  
        SORT strs BY length of each element in descending order AND lexicographically ascending  
  
        FOR each index i AND element word IN strs  
            SET unique TO true  
            FOR each index j AND element other IN strs  
                IF i NOT EQUALS j AND is_subsequence with parameters word and other RETURNS true  
                    SET unique TO false  
                    BREAK inner loop  
                END IF  
            END FOR  
  
            IF unique EQUALS true  
                RETURN the length of word  
            END IF  
        END FOR  
  
        RETURN negative one  
    END FUNCTION  
END CLASS
