CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a map with key zero and value minus one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each index i, character char IN s  
            SET digit TO the integer value of character char  
            SET prefix_xor TO prefix_xor XOR one shifted left by digit positions  
            
            IF prefix_xor EXISTS in prefix_map  
                SET max_length TO the greater of max_length AND index i MINUS the value of prefix_map at key prefix_xor  
            END IF  
            
            FOR each index j FROM zero TO nine  
                SET flipped TO prefix_xor XOR one shifted left by j positions  
                IF flipped EXISTS in prefix_map  
                    SET max_length TO the greater of max_length AND index i MINUS the value of prefix_map at key flipped  
                END IF  
            END FOR  
            
            IF prefix_xor DOES NOT EXIST in prefix_map  
                SET the value of prefix_map at key prefix_xor TO index i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
