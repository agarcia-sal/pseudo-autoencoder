CLASS Solution  
    FUNCTION nthMagicalNumber(n, a, b)  
        SET MOD TO one billion PLUS seven  
        
        FUNCTION gcd(x, y)  
            WHILE y IS NOT zero  
                SET temporary TO y  
                SET y TO x MINUS y MULTIPLIED BY the INTEGER DIVISION OF x BY y  
                SET x TO temporary  
            END WHILE  
            RETURN x  
        END FUNCTION  
        
        FUNCTION lcm(x, y)  
            RETURN x MULTIPLIED BY y DIVIDED BY gcd(x, y)  
        END FUNCTION  
        
        SET L TO lcm(a, b)  
        
        SET left TO one  
        SET right TO n MULTIPLIED BY the smaller OF a AND b  
        
        WHILE left IS LESS THAN right  
            SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
            SET count TO the INTEGER DIVISION OF mid BY a PLUS the INTEGER DIVISION OF mid BY b MINUS the INTEGER DIVISION OF mid BY L  
            
            IF count IS LESS THAN n  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left MODULO MOD  
    END FUNCTION  
END CLASS
