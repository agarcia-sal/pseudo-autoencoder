CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO the result of counting frequency of each character in s  
        SET max_heap TO empty list  
        FOR each character and frequency IN count  
            APPEND the pair of negative frequency and character TO max_heap  
        END FOR  
        CALL procedure to transform max_heap into a heap  
        
        SET prev_char TO none  
        SET prev_freq TO zero  
        SET result TO empty list  
        
        WHILE max_heap is not empty OR prev_freq is less than zero  
            IF max_heap is empty AND prev_freq is less than zero  
                RETURN empty string  
            END IF  
            
            REMOVE smallest element FROM max_heap AND UNPACK to freq and char  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq is less than zero  
                ADD the pair of prev_freq and prev_char TO max_heap  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN the concatenation of all characters in result as a single string  
    END FUNCTION  
END CLASS
