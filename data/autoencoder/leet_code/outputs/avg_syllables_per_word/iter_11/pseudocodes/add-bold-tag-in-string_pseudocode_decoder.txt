CLASS Solution  
    FUNCTION addBoldTag(s, words)  
        IF words EQUALS empty  
            RETURN s  
        END IF  
          
        SET n TO the LENGTH OF s  
        SET mask TO list of False with size n  
          
        FOR each word IN words  
            SET start TO zero  
            WHILE start LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the POSITION OF word IN s STARTING FROM start  
                IF start EQUALS negative one  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET mask element at position i TO True  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
          
        SET result TO empty list  
        SET i TO zero  
        WHILE i LESS THAN n  
            IF mask element at position i EQUALS True  
                APPEND the string "<b>" TO result  
                WHILE i LESS THAN n AND mask element at position i EQUALS True  
                    APPEND element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string "</b>" TO result  
            ELSE  
                APPEND element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
          
        RETURN the JOINED result as a string  
    END FUNCTION  
END CLASS
