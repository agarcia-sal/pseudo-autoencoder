CLASS Solution  
    FUNCTION reachingPoints(sx, sy, tx, ty)  
        WHILE tx IS GREATER THAN OR EQUAL TO sx AND ty IS GREATER THAN OR EQUAL TO sy  
            IF tx IS EQUALS TO sx AND ty IS EQUALS TO sy  
                RETURN True  
            END IF  
            IF tx IS GREATER THAN ty  
                IF ty IS GREATER THAN sy  
                    SET tx TO tx MODULO ty  
                ELSE  
                    RETURN the result of tx MINUS sx MODULO ty IS EQUALS TO zero  
                END IF  
            ELSE  
                IF tx IS GREATER THAN sx  
                    SET ty TO ty MODULO tx  
                ELSE  
                    RETURN the result of ty MINUS sy MODULO tx IS EQUALS TO zero  
                END IF  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
