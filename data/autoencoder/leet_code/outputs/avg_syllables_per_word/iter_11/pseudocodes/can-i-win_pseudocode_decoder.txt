CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF maxChoosableInteger MULTIPLIED BY (maxChoosableInteger PLUS one) DIVIDED BY two LESS THAN desiredTotal  
            RETURN False  
        END IF  

        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN True  
        END IF  

        SET memo TO an empty dictionary  

        FUNCTION can_win(current_total, used_numbers)  
            IF the pair of current_total AND used_numbers EXISTS IN memo  
                RETURN the value in memo at the pair of current_total AND used_numbers  
            END IF  

            FOR number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers AND one shifted LEFT BY number EQUALS zero  
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win(current_total PLUS number, the bitwise OR of used_numbers AND one shifted LEFT BY number)  
                        SET memo at the pair of current_total AND used_numbers TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  

            SET memo at the pair of current_total AND used_numbers TO False  
            RETURN False  
        END FUNCTION  

        RETURN can_win(zero, zero)  
    END FUNCTION  
END CLASS
