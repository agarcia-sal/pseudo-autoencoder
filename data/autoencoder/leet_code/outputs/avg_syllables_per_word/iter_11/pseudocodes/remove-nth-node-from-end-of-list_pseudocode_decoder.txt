CLASS ListNode  
    FUNCTION __init__ with parameters val, next  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with parameter the element at position zero of values  
    SET p TO head  
    FOR each val IN the slice of values from position one to the end  
        SET node TO new ListNode with parameter val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1, p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the expression p1.val EQUALS p2.val AND the result of calling is_same_list with parameters p1.next, p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd with parameters self, head, n  
        SET dummy TO new ListNode with parameter zero  
        SET dummy.next TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one  
            SET first TO first.next  
        END FOR  
        WHILE first NOT EQUALS None  
            SET first TO first.next  
            SET second TO second.next  
        END WHILE  
        SET second.next TO second.next.next  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
