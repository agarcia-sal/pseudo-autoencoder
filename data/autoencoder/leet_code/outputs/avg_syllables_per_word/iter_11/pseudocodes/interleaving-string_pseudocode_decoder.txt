CLASS Solution  
    FUNCTION isInterleave(s1, s2, s3)  
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 NOT EQUALS the LENGTH OF s3  
            RETURN False  
        END IF  

        SET dp TO a list of False values of length the LENGTH OF s2 PLUS one  
        SET the first element at position zero of dp TO True  

        FOR i FROM zero TO the LENGTH OF s1  
            FOR j FROM zero TO the LENGTH OF s2  
                IF i GREATER THAN zero  
                    SET dp at position j TO dp at position j AND element at position i MINUS one of s1 EQUALS element at position i PLUS j MINUS one of s3  
                END IF  
                IF j GREATER THAN zero  
                    SET dp at position j TO dp at position j OR (dp at position j MINUS one AND element at position j MINUS one of s2 EQUALS element at position i PLUS j MINUS one of s3)  
                END IF  
            END FOR  
        END FOR  

        RETURN dp at position the LENGTH OF s2  
    END FUNCTION  
END CLASS
