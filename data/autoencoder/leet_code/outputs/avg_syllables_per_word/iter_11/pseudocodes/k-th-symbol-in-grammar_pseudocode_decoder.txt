CLASS Solution  
    FUNCTION kthGrammar(n, k)  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET length_of_previous_row TO two RAISED TO THE POWER OF (n MINUS two)  
        IF k LESS THAN OR EQUAL TO length_of_previous_row  
            RETURN self DOT kthGrammar WITH PARAMETERS (n MINUS one, k)  
        ELSE  
            SET previous_value TO self DOT kthGrammar WITH PARAMETERS (n MINUS one, k MINUS length_of_previous_row)  
            SET complement_value TO one MINUS previous_value  
            RETURN complement_value  
        END IF  
    END FUNCTION  
END CLASS
