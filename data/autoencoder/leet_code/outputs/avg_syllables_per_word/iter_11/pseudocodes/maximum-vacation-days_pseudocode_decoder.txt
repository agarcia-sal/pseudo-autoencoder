CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the first element of days  
        
        SET dp TO a list of length n with all elements SET TO negative one  
        SET the element at position zero of dp TO zero  
        
        SET graph TO an empty dictionary  
        FOR index i FROM zero TO n MINUS one  
            SET the element at position i of graph TO an empty list  
        END FOR  
        
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF the element at position i of flights at position j EQUALS one  
                    APPEND j TO the element at position i of graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR index week FROM zero TO k MINUS one  
            SET new_dp TO a list of length n with all elements SET TO negative one  
            FOR index i FROM zero TO n MINUS one  
                IF the element at position i of dp EQUALS negative one  
                    CONTINUE TO next iteration  
                END IF  
                SET the element at position i of new_dp TO the maximum of the element at position i of new_dp AND the element at position i of dp PLUS the element at position i of days at position week  
                FOR each element j IN the element at position i of graph  
                    SET the element at position j of new_dp TO the maximum of the element at position j of new_dp AND the element at position i of dp PLUS the element at position j of days at position week  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the maximum element in dp  
    END FUNCTION  
END CLASS
