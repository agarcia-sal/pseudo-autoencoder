CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET it TO iterator over s  
            FOR each character IN word  
                IF character NOT IN it  
                    RETURN False  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        SORT dictionary BY descending order of LENGTH OF each element AND ascending lexicographical order of each element  
        
        FOR each word IN dictionary  
            IF is_subsequence(word, s) EQUALS True  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN empty string  
    END FUNCTION  
END CLASS
