CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        CALL points SORTED BY the second element of each point  
        SET arrows TO zero  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF the first element of balloon GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO the second element of balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
