CLASS Solution  
    FUNCTION isBipartite(graph)  
        SET n TO the LENGTH OF graph  
        SET color TO a list of LENGTH n FILLED WITH zero  
        
        FUNCTION dfs(node, c)  
            SET color at position node TO c  
            FOR each neighbor IN graph at position node  
                IF color at position neighbor EQUALS c  
                    RETURN False  
                END IF  
                IF color at position neighbor EQUALS zero  
                    IF NOT dfs(neighbor, NEGATIVE c)  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        FOR index i FROM zero TO n MINUS one  
            IF color at position i EQUALS zero  
                IF NOT dfs(i, one)  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
