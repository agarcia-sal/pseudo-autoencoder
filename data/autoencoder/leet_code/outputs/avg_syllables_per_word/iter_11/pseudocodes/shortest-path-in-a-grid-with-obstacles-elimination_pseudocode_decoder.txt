CLASS Solution  
    FUNCTION shortestPath(grid, k)  
        IF grid EQUALS empty OR element at position zero of grid EQUALS empty  
            RETURN negative one  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        IF k GREATER THAN OR EQUAL TO m PLUS n MINUS two  
            RETURN m PLUS n MINUS two  
        END IF  
        
        SET directions TO list containing  
            tuple of zero and one,  
            tuple of one and zero,  
            tuple of zero and negative one,  
            tuple of negative one and zero  
        
        SET queue TO double-ended queue initialized with tuple containing zero, zero, k, zero  
        
        SET visited TO set initialized with tuple containing zero, zero, k  
        
        WHILE queue IS NOT empty  
            SET x, y, remaining_k, steps TO the first element REMOVED from the left OF queue  
            
            IF x EQUALS m MINUS one AND y EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            FOR each tuple dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n  
                    SET new_k TO remaining_k MINUS element at position nx of element at position ny of grid  
                    
                    IF new_k GREATER THAN OR EQUAL TO zero AND tuple containing nx, ny, new_k NOT IN visited  
                        ADD tuple containing nx, ny, new_k TO visited  
                        APPEND tuple containing nx, ny, new_k, steps PLUS one TO queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
