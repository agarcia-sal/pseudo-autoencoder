CLASS Solution  
    FUNCTION countPairs(nums, k)  
        SET gcd_count TO RESULT OF Counter()  
        
        FOR each num IN nums  
            SET current_gcd TO RESULT OF gcd OF num AND k  
            INCREMENT the value at key current_gcd IN gcd_count BY one  
        END FOR  
        
        SET total_pairs TO zero  
        
        SET gcd_values TO list of keys IN gcd_count  
        SET n TO length OF gcd_values  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET gcd1 TO element at position i OF gcd_values  
                SET gcd2 TO element at position j OF gcd_values  
                
                IF the product OF gcd1 AND gcd2 MODULO k EQUALS zero  
                    IF i EQUALS j  
                        INCREMENT total_pairs BY value at key gcd1 IN gcd_count MULTIPLIED BY value at key gcd1 IN gcd_count MINUS one DIVIDED BY two  
                    ELSE  
                        INCREMENT total_pairs BY value at key gcd1 IN gcd_count MULTIPLIED BY value at key gcd2 IN gcd_count  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_pairs  
    END FUNCTION  
END CLASS
