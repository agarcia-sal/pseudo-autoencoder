CLASS Solution  
    FUNCTION characterReplacement(s, k)  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        CALL defaultdict WITH int AND SET TO count  
        
        FOR right FROM zero TO LENGTH OF s MINUS one  
            INCREMENT element at position right of s IN count BY one  
            SET max_count TO the GREATER THAN BETWEEN max_count AND the element at position right of s IN count  
            
            IF the EXPRESSION (right MINUS left PLUS one) MINUS max_count IS GREATER THAN k  
                DECREMENT element at position left of s IN count BY one  
                INCREMENT left BY one  
            END IF  
            
            SET max_length TO the GREATER THAN BETWEEN max_length AND (right MINUS left PLUS one)  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
