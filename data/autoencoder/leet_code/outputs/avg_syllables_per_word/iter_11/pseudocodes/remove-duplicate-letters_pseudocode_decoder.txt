CLASS Solution  
    FUNCTION removeDuplicateLetters(s)  
        SET last_occurrence TO empty dictionary  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position index of s  
            SET last_occurrence at key character TO index  
        END FOR  
  
        SET stack TO empty list  
        SET in_stack TO empty set  
  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position index of s  
  
            IF character IN in_stack  
                CONTINUE to next iteration  
            END IF  
  
            WHILE the LENGTH OF stack GREATER THAN zero AND character LESS THAN the last element in stack AND index LESS THAN last_occurrence at key the last element in stack  
                SET removed_char TO the last element in stack  
                REMOVE last element FROM stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
  
            APPEND character TO stack  
            ADD character TO in_stack  
        END FOR  
  
        SET result TO concatenation of all elements in stack in order  
        RETURN result  
    END FUNCTION  
END CLASS
