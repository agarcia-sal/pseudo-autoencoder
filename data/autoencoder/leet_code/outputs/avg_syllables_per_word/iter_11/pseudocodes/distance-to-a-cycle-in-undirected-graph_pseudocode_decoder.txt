CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO a new default dictionary with list as default value  
        FOR each pair of u and v IN edges  
            APPEND v TO the element at key u of graph  
            APPEND u TO the element at key v of graph  
        END FOR  
        
        FUNCTION dfs(node, parent, path, visited)  
            IF node IN visited THEN  
                SET cycle_start TO the position of node in path  
                RETURN the sublist of path from position cycle_start to the end  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN the element at key node of graph  
                IF neighbor NOT EQUALS parent THEN  
                    SET cycle TO dfs(neighbor, node, path, visited)  
                    IF cycle IS NOT NONE THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE last element FROM path  
            RETURN NONE  
        END FUNCTION  
        
        SET visited TO an empty set  
        SET cycle TO NONE  
        FOR node FROM zero TO n MINUS one  
            IF node NOT IN visited THEN  
                SET cycle TO dfs(node, negative one, an empty list, visited)  
                IF cycle IS NOT NONE THEN  
                    BREAK  
                END IF  
            END IF  
        END FOR  
        
        SET distance TO a list of zero repeated n times  
        SET queue TO a new deque initialized with cycle  
        SET visited TO a new set initialized with cycle  
        
        SET level TO zero  
        WHILE queue IS NOT EMPTY  
            SET current_length TO the LENGTH OF queue  
            FOR index FROM one TO current_length  
                REMOVE first element FROM queue AND SET TO node  
                SET element at position node of distance TO level  
                FOR each neighbor IN the element at key node of graph  
                    IF neighbor NOT IN visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  
        
        RETURN distance  
    END FUNCTION  
END CLASS
