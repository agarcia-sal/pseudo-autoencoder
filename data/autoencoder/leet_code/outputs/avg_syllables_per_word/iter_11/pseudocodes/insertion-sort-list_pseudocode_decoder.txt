CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with value as element at position zero of values  
    SET p TO head  
    FOR each val IN all elements of values from position one to end  
        SET node TO new ListNode with value val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN element val of p1 EQUALS element val of p2 AND is_same_list(element next of p1, element next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self, head)  
        IF head EQUALS None OR element next of head EQUALS None  
            RETURN head  
        END IF  
        SET dummy TO new ListNode with value MINUS five thousand AND one  
        SET dummy.next TO head  
        SET current TO element next of head  
        SET element next of head TO None  
        WHILE current IS NOT None  
            SET next_node TO element next of current  
            SET prev TO dummy  
            WHILE element next of prev IS NOT None AND element val of element next of prev LESS THAN element val of current  
                SET prev TO element next of prev  
            END WHILE  
            SET element next of current TO element next of prev  
            SET element next of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN element next of dummy  
    END FUNCTION  
END CLASS
