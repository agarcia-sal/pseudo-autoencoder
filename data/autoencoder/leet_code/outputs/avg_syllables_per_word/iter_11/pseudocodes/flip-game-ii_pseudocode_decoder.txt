CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF state EXISTS IN memo  
                RETURN value at state IN memo  
            END IF  
            
            FOR i FROM zero TO the LENGTH OF state MINUS one MINUS one  
                SET substring_at_i TO element at position i of state PLUS element at position i PLUS one of state  
                IF substring_at_i EQUALS two PLUS signs  
                    SET new_state TO concatenation of substring of state BEFORE position i PLUS two MINUS signs PLUS substring of state AFTER position i PLUS two  
                    IF NOT can_win_helper(new_state)  
                        SET value at state IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET value at state IN memo TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_win_helper(currentState)  
    END FUNCTION  
END CLASS
