CLASS Solution  
    FUNCTION exist(board, word)  
        IF board EQUALS empty  
            RETURN False  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        SET visited TO a list of lists of False values with m rows and n columns  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS the LENGTH OF word  
                RETURN True  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of visited at position y IS True OR element at position x of board at position y NOT EQUALS element at position index of word  
                RETURN False  
            END IF  

            SET element at position x of visited at position y TO True  
            FOR each pair dx, dy IN the list of pairs where pairs are minus one and zero, one and zero, zero and minus one, zero and one  
                IF dfs(x PLUS dx, y PLUS dy, index PLUS one) EQUALS True  
                    RETURN True  
                END IF  
            END FOR  
            SET element at position x of visited at position y TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs(i, j, zero) EQUALS True  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  

        RETURN False  
    END FUNCTION  
END CLASS
