CLASS Solution  
    FUNCTION minDistance(houses, k)  
        CALL houses SORT  
        SET n TO the LENGTH OF houses  
        
        SET cost TO a new two-dimensional list of size n BY n FILLED WITH zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost at position i and j TO cost at position i plus one and j minus one PLUS the element at position j of houses MINUS the element at position i of houses  
            END FOR  
        END FOR  
        
        SET dp TO a new two-dimensional list of size n PLUS one BY k PLUS one FILLED WITH infinity  
        SET dp at position zero and zero TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET dp at position i and j TO the MINIMUM OF dp at position i and j AND dp at position p and j minus one PLUS cost at position p and i minus one  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp at position n and k  
    END FUNCTION  
END CLASS
