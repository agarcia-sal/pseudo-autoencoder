CLASS Solution  
    FUNCTION getPermutation(n, k)  
        IMPORT math  
        SET numbers TO list of integers from one TO n inclusive  
        DECREMENT k BY one  
        SET result TO empty list  
        FOR i FROM n DOWN TO one  
            SET fact TO factorial of i MINUS one USING math  
            SET index TO k DIVIDED BY fact  
            APPEND string of element at position index OF numbers TO result  
            REMOVE element at position index FROM numbers  
            SET k TO k MODULO fact  
        END FOR  
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
