CLASS Solution  
    FUNCTION isMatch(s, p)  
        CALL initialize_dp_table WITH parameters s, p  
        SET dp TO the result of initialize_dp_table  
        
        SET first element at position zero and position zero of dp TO True  
        
        FOR index j FROM one TO the LENGTH OF p  
            IF element at position j minus one of p EQUALS the character star  
                SET element at position zero and position j of dp TO element at position zero and position j minus one of dp  
            END IF  
        END FOR  
        
        FOR index i FROM one TO the LENGTH OF s  
            FOR index j FROM one TO the LENGTH OF p  
                IF element at position j minus one of p EQUALS the character star  
                    SET element at position i and position j of dp TO element at position i minus one and position j of dp OR element at position i and position j minus one of dp  
                ELSE IF element at position j minus one of p EQUALS the character question mark OR element at position i minus one of s EQUALS element at position j minus one of p  
                    SET element at position i and position j of dp TO element at position i minus one and position j minus one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position LENGTH OF s and position LENGTH OF p of dp  
    END FUNCTION  
    
    FUNCTION initialize_dp_table(s, p)  
        SET dp TO empty list  
        FOR index FROM zero TO LENGTH OF s  
            SET row TO empty list  
            FOR index j FROM zero TO LENGTH OF p  
                APPEND False TO row  
            END FOR  
            APPEND row TO dp  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
