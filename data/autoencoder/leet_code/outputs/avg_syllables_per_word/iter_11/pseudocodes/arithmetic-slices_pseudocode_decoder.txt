CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        IF the LENGTH OF nums LESS THAN 3  
            RETURN zero  
        END IF  
        
        SET total_slices TO zero  
        SET current_length TO two  
        
        FOR i FROM two TO the LENGTH OF nums MINUS one  
            IF element at position i of nums MINUS element at position i MINUS one of nums EQUALS element at position i MINUS one of nums MINUS element at position i MINUS two of nums  
                INCREMENT current_length BY one  
                INCREMENT total_slices BY current_length MINUS two  
            ELSE  
                SET current_length TO two  
            END IF  
        END FOR  
        
        RETURN total_slices  
    END FUNCTION  
END CLASS
