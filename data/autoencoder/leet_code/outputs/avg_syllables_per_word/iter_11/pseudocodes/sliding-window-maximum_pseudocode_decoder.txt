CLASS Solution  
    FUNCTION maxSlidingWindow(nums, k)  
        IF nums EQUALS empty OR k EQUALS zero  
            RETURN empty list  
        END IF  
        
        SET deq TO empty double-ended queue  
        SET result TO empty list  
        
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT empty AND the element at position zero of deq IS LESS THAN i MINUS k PLUS one  
                REMOVE first element FROM deq  
            END IF  
            
            WHILE deq IS NOT empty AND the element at position element at last position of deq of nums IS LESS THAN the element at position i of nums  
                REMOVE last element FROM deq  
            END WHILE  
            
            APPEND i TO deq  
            
            IF i IS GREATER THAN OR EQUAL TO k MINUS one  
                APPEND the element at position element at first position of deq of nums TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
