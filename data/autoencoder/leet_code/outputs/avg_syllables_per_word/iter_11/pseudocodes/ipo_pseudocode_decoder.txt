CLASS Solution  
    FUNCTION findMaximizedCapital(k, w, profits, capital)  
        SET projects TO list of pairs of elements at same positions of capital and profits  
        SORT projects in ascending order by first element of each pair  
        
        SET max_heap TO empty list  
        
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        
        FOR iteration FROM one TO k  
            WHILE i LESS THAN n AND element at position i of projects at position zero LESS THAN OR EQUAL TO w  
                PUSH the negative of element at position i of projects at position one TO max_heap  
                INCREMENT i BY one  
            END WHILE  
            
            IF max_heap IS empty  
                BREAK the loop  
            END IF  
            
            SET w TO w PLUS the negative of the element popped from max_heap  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
