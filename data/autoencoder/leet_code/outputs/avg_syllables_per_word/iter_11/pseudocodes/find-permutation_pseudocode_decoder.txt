CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO empty list  
        FOR index FROM zero TO length of s PLUS one MINUS one  
            APPEND index PLUS one TO result  
        END FOR  
        SET start TO zero  
        FOR end FROM zero TO length of s  
            IF end EQUALS length of s OR element at position end of s EQUALS character I  
                REVERSE the segment of result FROM position start TO position end PLUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
