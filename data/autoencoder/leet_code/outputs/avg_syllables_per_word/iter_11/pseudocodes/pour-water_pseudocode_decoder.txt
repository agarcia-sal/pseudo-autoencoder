CLASS Solution  
    FUNCTION pourWater(heights, volume, k)  
        SET n TO the LENGTH OF heights  
        
        FOR each iteration FROM one TO volume  
            SET i TO k  
            WHILE i GREATER THAN zero AND the element at position i MINUS one of heights LESS THAN OR EQUAL TO the element at position i of heights  
                DECREMENT i BY one  
            END WHILE  
            
            WHILE i LESS THAN k AND the element at position i of heights EQUALS the element at position i PLUS one of heights  
                INCREMENT i BY one  
            END WHILE  
            
            IF i LESS THAN k  
                INCREMENT the element at position i of heights BY one  
                CONTINUE to the next iteration  
            END IF  
            
            SET i TO k  
            WHILE i LESS THAN n MINUS one AND the element at position i PLUS one of heights LESS THAN OR EQUAL TO the element at position i of heights  
                INCREMENT i BY one  
            END WHILE  
            
            WHILE i GREATER THAN k AND the element at position i of heights EQUALS the element at position i MINUS one of heights  
                DECREMENT i BY one  
            END WHILE  
            
            IF i GREATER THAN k  
                INCREMENT the element at position i of heights BY one  
                CONTINUE to the next iteration  
            END IF  
            
            INCREMENT the element at position k of heights BY one  
        END FOR  
        
        RETURN heights  
    END FUNCTION  
END CLASS
