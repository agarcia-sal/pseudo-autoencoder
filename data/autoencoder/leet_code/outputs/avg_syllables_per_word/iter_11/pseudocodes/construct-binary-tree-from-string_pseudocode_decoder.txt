CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with val as element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO end  
        SET node TO new ListNode with val as val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val as element at position zero of values  
    SET i TO one  
    CREATE queue as deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode with val as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode with val as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO LENGTH OF s  
                RETURN None, index  
            END IF  
          
            SET start TO index  
            WHILE index LESS THAN LENGTH OF s AND (element at position index of s IS digit OR element at position index of s EQUALS minus sign)  
                INCREMENT index BY one  
            END WHILE  
          
            SET node TO new TreeNode with val as integer parsed from substring of s from position start TO index minus one  
          
            IF index LESS THAN LENGTH OF s AND element at position index of s EQUALS left parenthesis  
                SET node.left, index TO parse_tree(index PLUS one)  
            END IF  
          
            IF index LESS THAN LENGTH OF s AND element at position index of s EQUALS left parenthesis  
                SET node.right, index TO parse_tree(index PLUS one)  
            END IF  
          
            IF index LESS THAN LENGTH OF s AND element at position index of s EQUALS right parenthesis  
                INCREMENT index BY one  
            END IF  
          
            RETURN node, index  
        END FUNCTION  
      
        SET root, _ TO parse_tree(zero)  
        RETURN root  
    END FUNCTION  
END CLASS
