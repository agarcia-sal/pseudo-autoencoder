CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        SET cost_to_digit TO an empty dictionary  
        FOR index FROM zero TO the LENGTH OF cost MINUS one  
            SET c TO element at position index of cost  
            SET digit_string TO string representation of index PLUS one  
            SET element at key c of cost_to_digit TO digit_string  
        END FOR  
        
        SET dp TO a list with the first element as empty string and the next target elements as string "-1"  
        
        FOR t FROM one TO target  
            FOR each key c AND value d IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND element at position t MINUS c of dp DOES NOT EQUAL string "-1" THEN  
                    SET candidate TO concatenation of d FOLLOWED BY element at position t MINUS c of dp  
                    IF element at position t of dp EQUALS string "-1" OR LENGTH OF candidate GREATER THAN LENGTH OF element at position t of dp OR (LENGTH OF candidate EQUALS LENGTH OF element at position t of dp AND candidate GREATER THAN element at position t of dp) THEN  
                        SET element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF element at position target of dp DOES NOT EQUAL string "-1" THEN  
            RETURN element at position target of dp  
        ELSE  
            RETURN string "0"  
        END IF  
    END FUNCTION  
END CLASS
