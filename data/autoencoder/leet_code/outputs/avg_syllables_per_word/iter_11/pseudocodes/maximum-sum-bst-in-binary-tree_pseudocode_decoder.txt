CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val as element at position zero of values  
    SET i TO one  
    SET queue TO a new deque collection  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed FROM queue  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with val as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with val as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST(self, root)  
        SET self.max_sum TO zero  
        
        FUNCTION helper(node)  
            IF node EQUALS None  
                RETURN True, zero, positive infinity, negative infinity  
            END IF  
            
            SET left_is_bst, left_sum, left_min, left_max TO helper(node.left)  
            SET right_is_bst, right_sum, right_min, right_max TO helper(node.right)  
            
            IF left_is_bst AND right_is_bst AND left_max LESS THAN node.val AND node.val LESS THAN right_min  
                SET current_sum TO left_sum PLUS right_sum PLUS node.val  
                SET current_min TO the smaller of node.val AND left_min  
                SET current_max TO the greater of node.val AND right_max  
                SET self.max_sum TO the greater of self.max_sum AND current_sum  
                RETURN True, current_sum, current_min, current_max  
            END IF  
            
            RETURN False, zero, positive infinity, negative infinity  
        END FUNCTION  
        
        CALL helper with root  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
