CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO list of characters in string representation of n  
        SET length TO the LENGTH OF digits  
          
        SET i TO length MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND element at position i of digits GREATER THAN OR EQUAL TO element at position i PLUS one of digits  
            DECREMENT i BY one  
        END WHILE  
          
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
          
        SET j TO length MINUS one  
        WHILE element at position j of digits LESS THAN OR EQUAL TO element at position i of digits  
            DECREMENT j BY one  
        END WHILE  
          
        SET temporary variable TO element at position i of digits  
        SET element at position i of digits TO element at position j of digits  
        SET element at position j of digits TO temporary variable  
          
        SET prefix TO the SUBLIST of digits FROM start TO position i PLUS one  
        SET suffix TO the SUBLIST of digits FROM position i PLUS one TO end  
        REVERSE the order of elements in suffix  
        SET digits TO concatenation of prefix AND suffix  
          
        SET result TO integer converted from concatenation of all elements in digits  
          
        IF result GREATER THAN the number two RAISED TO the power THIRTY ONE MINUS one  
            RETURN negative one  
        END IF  
          
        RETURN result  
    END FUNCTION  
END CLASS
