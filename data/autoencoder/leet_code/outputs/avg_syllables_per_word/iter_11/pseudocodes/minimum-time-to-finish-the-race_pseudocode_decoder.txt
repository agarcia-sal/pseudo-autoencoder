CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of fifteen elements each set TO infinity  
        FOR each pair of values f and r IN tires  
            SET total_time TO zero  
            FOR index i FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER OF i  
                IF lap_time GREATER THAN changeTime PLUS f  
                    BREAK out of the inner loop  
                END IF  
                INCREMENT total_time BY lap_time  
                SET min_times at position i TO the lesser of min_times at position i AND total_time  
            END FOR  
        END FOR  
        
        SET dp TO a list of length numLaps PLUS one each element set TO infinity  
        SET dp at position zero TO zero  
        
        FOR index i FROM one TO numLaps  
            FOR index j FROM zero TO the lesser of i MINUS one AND fourteen  
                SET dp at position i TO the lesser of dp at position i AND the sum of dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
            END FOR  
        END FOR  
        
        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
