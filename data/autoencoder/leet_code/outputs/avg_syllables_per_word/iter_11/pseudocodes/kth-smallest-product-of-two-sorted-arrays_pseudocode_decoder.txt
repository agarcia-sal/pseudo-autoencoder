CLASS Solution  
    FUNCTION kthSmallestProduct(nums1, nums2, k)  
        FUNCTION count(p)  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each x IN nums1  
                IF x GREATER THAN zero  
                    INCREMENT cnt BY the position IN nums2 WHERE p DIVIDED BY x WOULD BE INSERTED TO KEEP ORDER ON THE RIGHT  
                ELSE IF x LESS THAN zero  
                    INCREMENT cnt BY n MINUS the position IN nums2 WHERE p DIVIDED BY x WOULD BE INSERTED TO KEEP ORDER ON THE LEFT  
                ELSE  
                    INCREMENT cnt BY n MULTIPLIED BY one IF p GREATER THAN OR EQUAL TO zero OTHERWISE zero  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  

        SET mx TO the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums1 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums1  
        SET mx TO mx MULTIPLIED BY the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT OF nums2 AND the ABSOLUTE VALUE OF the LAST ELEMENT OF nums2  
        SET result TO the POSITION IN THE RANGE FROM NEGATIVE mx TO mx INCLUDING mx WHERE k WOULD BE INSERTED TO KEEP ORDER ON THE LEFT USING count AS THE KEY FUNCTION MINUS mx  
        RETURN result  
    END FUNCTION  
END CLASS
