CLASS Solution  
    FUNCTION numRescueBoats(people, limit)  
        CALL sort ON people  
        SET left TO zero  
        SET right TO the LENGTH OF people MINUS one  
        SET boats TO zero  
        WHILE left LESS THAN OR EQUAL TO right  
            IF element at position left of people PLUS element at position right of people LESS THAN OR EQUAL TO limit  
                INCREMENT left BY one  
            END IF  
            DECREMENT right BY one  
            INCREMENT boats BY one  
        END WHILE  
        RETURN boats  
    END FUNCTION  
END CLASS
