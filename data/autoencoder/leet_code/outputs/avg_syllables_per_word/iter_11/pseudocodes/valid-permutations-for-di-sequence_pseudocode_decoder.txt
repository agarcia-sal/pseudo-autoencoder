CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a list of lists each containing zero with size n PLUS one repeated n PLUS one times  
        
        FOR j FROM zero TO n  
            SET the element at position j of the list at position zero of dp TO one  
        END FOR  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            IF the element at position i MINUS one of s EQUALS the character 'I'  
                FOR j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS the element at position j of the list at position i MINUS one of dp  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                    SET the element at position j of the list at position i of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO prefix_sum PLUS the element at position j PLUS one of the list at position i MINUS one of dp  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                    SET the element at position j of the list at position i of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO the sum of the list at position n of dp  
        SET result TO result MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
