CLASS Solution  
    FUNCTION removeKdigits(num, k)  
        SET stack TO empty list  
        
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND stack NOT empty AND last element of stack GREATER THAN digit  
                REMOVE last element FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        
        IF k GREATER THAN zero  
            SET final_stack TO all elements of stack EXCEPT last k elements  
        ELSE  
            SET final_stack TO stack  
        END IF  
        
        SET result TO the concatenation of all elements in final_stack with leading zeros REMOVED  
        
        IF result IS NOT empty string  
            RETURN result  
        ELSE  
            RETURN string "0"  
        END IF  
    END FUNCTION  
END CLASS
