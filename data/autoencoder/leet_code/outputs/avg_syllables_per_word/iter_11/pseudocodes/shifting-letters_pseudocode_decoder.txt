CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one TO zero STEPPING BY minus one  
            INCREMENT total_shifts BY the element at position index of shifts  
            SET the element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET original_position TO the Unicode code of the element at position index of s MINUS the Unicode code of letter a  
            SET shifted_position TO original_position PLUS the element at position index of shifts  
            SET wrapped_position TO shifted_position MODULO twenty six  
            SET new_character_code TO wrapped_position PLUS the Unicode code of letter a  
            SET new_char TO the character represented by new_character_code  
            APPEND new_char TO result  
        END FOR  
        RETURN the concatenation of the elements of result  
    END FUNCTION  
END CLASS
