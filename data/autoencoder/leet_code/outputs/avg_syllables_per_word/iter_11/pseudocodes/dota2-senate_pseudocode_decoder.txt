CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        IMPORT deque FROM collections  
        SET radiant TO a new empty deque  
        SET dire TO a new empty deque  
        
        FOR each position i AND character s IN senate  
            IF s EQUALS the character R  
                APPEND i TO radiant  
            ELSE  
                APPEND i TO dire  
            END IF  
        END FOR  
        
        WHILE radiant IS NOT empty AND dire IS NOT empty  
            SET r_index TO the removal of first element FROM radiant  
            SET d_index TO the removal of first element FROM dire  
            
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO dire  
            END IF  
        END WHILE  
        
        IF radiant IS NOT empty  
            RETURN the string Radiant  
        ELSE  
            RETURN the string Dire  
        END IF  
    END FUNCTION  
END CLASS
