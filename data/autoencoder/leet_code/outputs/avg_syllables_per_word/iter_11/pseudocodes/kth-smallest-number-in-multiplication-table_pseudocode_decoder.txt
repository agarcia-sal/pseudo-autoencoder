CLASS Solution  
    FUNCTION findKthNumber(m, n, k)  
        FUNCTION enough(x)  
            SET count TO zero  
            FOR i FROM one TO m  
                SET count TO count PLUS the smaller value BETWEEN x DIVIDED BY i AND n  
            END FOR  
            RETURN count GREATER THAN OR EQUAL TO k  
        END FUNCTION  
        SET low TO one  
        SET high TO m MULTIPLIED BY n  
        WHILE low LESS THAN high  
            SET mid TO the integer division result of low PLUS high DIVIDED BY two  
            IF enough(mid)  
                SET high TO mid  
            ELSE  
                SET low TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
