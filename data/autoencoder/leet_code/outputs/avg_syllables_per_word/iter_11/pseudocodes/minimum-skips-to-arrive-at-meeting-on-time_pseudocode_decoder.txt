CLASS Solution  
    FUNCTION minSkips(dist, speed, hoursBefore)  
        SET n TO the LENGTH OF dist  
        SET dp TO a list of length n where each element is a list of length n PLUS one filled with infinite value  
        SET total_dist TO the SUM OF all elements IN dist  
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore  
            RETURN negative one  
        END IF  
        
        SET element at position zero of dp at position zero TO element at position zero of dist  
        FOR j FROM one TO n MINUS one  
            SET element at position zero of dp at position j TO element at position zero of dist  
        END FOR  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j LESS THAN i  
                    SET element at position i of dp at position j TO the CEILING OF (element at position i MINUS one of dp at position j DIVIDED BY speed) MULTIPLIED BY speed PLUS element at position i of dist  
                END IF  
                IF j GREATER THAN zero  
                    SET element at position i of dp at position j TO the MINIMUM OF previous value of element at position i of dp at position j AND element at position i MINUS one of dp at position j MINUS one PLUS element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            IF element at position n MINUS one of dp at position j LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
