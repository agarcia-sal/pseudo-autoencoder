CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO element at position zero of nums  
        SET current_sum TO element at position zero of nums  
        FOR each num IN sublist of nums starting from position one to the end  
            IF num GREATER THAN OR EQUAL TO current_sum PLUS num  
                SET current_sum TO num  
            ELSE  
                SET current_sum TO current_sum PLUS num  
            END IF  
            IF max_sum LESS THAN current_sum  
                SET max_sum TO current_sum  
            END IF  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
