CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1, val2)  
            IF val1 GREATER THAN zero AND val2 GREATER THAN zero OR val1 LESS THAN zero AND val2 LESS THAN zero  
                RETURN True  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        SET n TO LENGTH OF nums  

        FOR i FROM zero TO n MINUS one  
            SET slow TO i  
            SET fast TO i  

            IF element at position i of nums EQUALS zero  
                CONTINUE TO next iteration  
            END IF  

            WHILE True  
                SET slow TO slow PLUS element at position slow of nums  
                SET slow TO slow MODULO n  
                IF slow LESS THAN zero  
                    INCREMENT slow BY n  
                END IF  

                SET fast TO fast PLUS element at position fast of nums  
                SET fast TO fast MODULO n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  

                SET fast TO fast PLUS element at position fast of nums  
                SET fast TO fast MODULO n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  

                IF NOT sameDirection(element at position slow of nums, element at position i of nums) OR NOT sameDirection(element at position fast of nums, element at position i of nums)  
                    BREAK FROM WHILE  
                END IF  

                IF slow EQUALS fast  
                    IF slow EQUALS (slow PLUS element at position slow of nums) MODULO n  
                        BREAK FROM WHILE  
                    END IF  
                    RETURN True  
                END IF  
            END WHILE  
        END FOR  

        RETURN False  
    END FUNCTION  
END CLASS
