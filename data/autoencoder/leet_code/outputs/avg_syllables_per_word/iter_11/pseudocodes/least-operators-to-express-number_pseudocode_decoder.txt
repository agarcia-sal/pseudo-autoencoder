CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        FUNCTION dp(target)  
            IF target EQUALS zero  
                RETURN minus one  
            END IF  
            IF target LESS THAN x  
                RETURN minimum of (two MULTIPLIED BY target MINUS one) AND (two MULTIPLIED BY (x MINUS target))  
            END IF  
            SET power TO one  
            WHILE x RAISED TO power LESS THAN OR EQUAL TO target  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET res TO dp(target MINUS x RAISED TO power) PLUS power  
            IF (x RAISED TO (power PLUS one)) MINUS target LESS THAN target  
                SET res TO minimum of res AND dp(x RAISED TO (power PLUS one) MINUS target) PLUS power PLUS one  
            END IF  
            RETURN res  
        END FUNCTION  
        RETURN dp(target)  
    END FUNCTION  
END CLASS
