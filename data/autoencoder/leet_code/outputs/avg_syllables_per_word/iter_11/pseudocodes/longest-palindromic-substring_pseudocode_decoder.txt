CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO a two-dimensional list of size n BY n with all elements SET TO True USING helper FUNCTION initializeTwoDimensionalList WITH parameters n, n, True  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET element at position i and j of f TO False  
                IF element at position i of s EQUALS element at position j of s  
                    SET element at position i and j of f TO element at position i PLUS one and j MINUS one of f  
                    IF element at position i and j of f AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the substring of s STARTING AT position k WITH LENGTH mx  
    END FUNCTION

    FUNCTION initializeTwoDimensionalList(rows, columns, value)  
        SET result TO empty list  
        FOR each row_index FROM zero TO rows MINUS one  
            SET current_row TO empty list  
            FOR each column_index FROM zero TO columns MINUS one  
                APPEND value TO current_row  
            END FOR  
            APPEND current_row TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
