CLASS Solution  
    FUNCTION integerBreak(n)  
        SET dp TO list of zeros with length n PLUS one  
        SET element at position one of dp TO one  
        FOR i FROM two TO n  
            SET max_product TO zero  
            FOR j FROM one TO i MINUS one  
                SET candidate_one TO j MULTIPLIED BY element at position i MINUS j of dp  
                SET candidate_two TO j MULTIPLIED BY (i MINUS j)  
                SET max_product TO the maximum of max_product, candidate_one, candidate_two  
            END FOR  
            SET element at position i of dp TO max_product  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
