CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        WHILE n GREATER THAN length MULTIPLIED BY count  
            SET n TO n MINUS length MULTIPLIED BY count  
            INCREMENT length BY one  
            SET count TO count MULTIPLIED BY ten  
            SET start TO start MULTIPLIED BY ten  
        END WHILE  
        SET start TO start PLUS the INTEGER DIVISION of (n MINUS one) BY length  
        SET digit_position TO the REMAINDER of (n MINUS one) DIVIDED BY length  
        SET start_string TO the STRING representation OF start  
        SET desired_digit_character TO the CHARACTER at position digit_position OF start_string  
        SET desired_digit TO the INTEGER conversion OF desired_digit_character  
        RETURN desired_digit  
    END FUNCTION  
END CLASS
