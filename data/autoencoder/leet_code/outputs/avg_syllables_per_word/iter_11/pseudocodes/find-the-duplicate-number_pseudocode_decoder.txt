CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count TO zero  
            FOR each v IN nums  
                IF v LESS THAN OR EQUAL TO x  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            IF count GREATER THAN x  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        RETURN the result of the binary search for the leftmost position in the range from zero TO the LENGTH OF nums MINUS one where f RETURNS true  
    END FUNCTION  
END CLASS
