CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS plus sign  
                        APPEND l PLUS r TO results  
                    ELSE IF operator EQUALS minus sign  
                        APPEND l MINUS r TO results  
                    ELSE IF operator EQUALS asterisk sign  
                        APPEND l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        FUNCTION helper(sub_expr)  
            IF sub_expr IS composed only of digits  
                RETURN list containing the integer value of sub_expr  
            END IF  
            
            SET results TO empty list  
            FOR index FROM zero TO LENGTH OF sub_expr MINUS one  
                IF character at position index of sub_expr IS one of plus sign, minus sign, asterisk sign  
                    SET left_results TO helper(substring of sub_expr FROM start TO index MINUS one)  
                    SET right_results TO helper(substring of sub_expr FROM index PLUS one TO end)  
                    EXTEND results BY compute(left_results, right_results, character at position index of sub_expr)  
                END IF  
            END FOR  
            
            RETURN results  
        END FUNCTION  
        
        RETURN helper(expression)  
    END FUNCTION  
END CLASS
