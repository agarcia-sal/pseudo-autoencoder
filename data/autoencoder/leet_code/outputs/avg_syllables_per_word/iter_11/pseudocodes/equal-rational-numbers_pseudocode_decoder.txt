CLASS Solution  
    FUNCTION isRationalEqual(s, t)  
        SET ratios TO list containing one, one DIVIDED BY nine, one DIVIDED BY ninety-nine, one DIVIDED BY nine hundred ninety-nine, one DIVIDED BY nine thousand nine hundred ninety-nine  

        FUNCTION valueOf(s)  
            IF the position of character open parenthesis IN s EQUALS negative one  
                RETURN the floating-point number converted from s  
            END IF  

            SET leftParenIndex TO the position of character open parenthesis IN s  
            SET rightParenIndex TO the position of character close parenthesis IN s  
            SET dotIndex TO the position of character dot IN s  

            SET integerAndNonRepeating TO the floating-point number converted from the substring of s from the beginning UP TO leftParenIndex  
            SET nonRepeatingLength TO leftParenIndex MINUS dotIndex MINUS one  

            SET repeating TO the integer converted from the substring of s starting from leftParenIndex PLUS one UP TO rightParenIndex  
            SET repeatingLength TO rightParenIndex MINUS leftParenIndex MINUS one  

            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY zero point one RAISED TO nonRepeatingLength MULTIPLIED BY the element at position repeatingLength OF ratios  
        END FUNCTION  

        RETURN the absolute value of valueOf(s) MINUS valueOf(t) LESS THAN one times ten RAISED TO the power of negative nine  
    END FUNCTION  
END CLASS
