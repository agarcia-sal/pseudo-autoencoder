CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val equal to element at position zero of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with val equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with val equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree with parameters p, q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left, q.left AND the result of is_same_tree with parameters p.right, q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION findLeaves with parameter root  
        FUNCTION collectLeaves with parameter node  
            IF node EQUALS None  
                RETURN negative one  
            END IF  
            SET left_height TO the result of collectLeaves with parameter node.left  
            SET right_height TO the result of collectLeaves with parameter node.right  
            SET current_height TO the maximum of left_height and right_height PLUS one  
            IF current_height GREATER THAN OR EQUAL TO the LENGTH OF result  
                APPEND empty list TO result  
            END IF  
            APPEND node.val TO the element at position current_height of result  
            RETURN current_height  
        END FUNCTION  
        SET result TO empty list  
        CALL collectLeaves with parameter root  
        RETURN result  
    END FUNCTION  
END CLASS
