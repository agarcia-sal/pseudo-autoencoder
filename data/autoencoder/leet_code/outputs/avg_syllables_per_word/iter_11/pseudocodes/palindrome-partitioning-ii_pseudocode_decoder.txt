CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        
        CREATE is_palindrome AS a two-dimensional list of size n by n with all values SET TO False  
        
        FOR i FROM zero TO n MINUS one  
            SET the element at position i and i of is_palindrome TO True  
        END FOR  
        
        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET the element at position start and end of is_palindrome TO the comparison of character at position start of s EQUALS character at position end of s  
                ELSE  
                    SET the element at position start and end of is_palindrome TO the logical AND of the comparison of character at position start of s EQUALS character at position end of s AND the element at position start PLUS one and end MINUS one of is_palindrome  
                END IF  
            END FOR  
        END FOR  
        
        CREATE min_cuts AS a list of size n with all values SET TO zero  
        FOR i FROM zero TO n MINUS one  
            IF the element at position zero and i of is_palindrome  
                SET the element at position i of min_cuts TO zero  
            ELSE  
                SET the element at position i of min_cuts TO infinity  
                FOR j FROM one TO i  
                    IF the element at position j and i of is_palindrome AND the element at position j MINUS one of min_cuts PLUS one LESS THAN the element at position i of min_cuts  
                        SET the element at position i of min_cuts TO the element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN the element at position n MINUS one of min_cuts  
    END FUNCTION  
END CLASS
