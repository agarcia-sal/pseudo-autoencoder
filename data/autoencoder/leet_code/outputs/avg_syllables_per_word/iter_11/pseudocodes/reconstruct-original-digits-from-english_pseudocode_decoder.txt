CLASS Solution  
    FUNCTION originalDigits(s)  
        IMPORT Counter FROM collections  
        
        SET count TO Counter APPLIED TO s  
        
        SET digits TO a list of ten zeros  
        
        SET element at position zero of digits TO the element of count at character 'z'  
        SET element at position two of digits TO the element of count at character 'w'  
        SET element at position four of digits TO the element of count at character 'u'  
        SET element at position six of digits TO the element of count at character 'x'  
        SET element at position eight of digits TO the element of count at character 'g'  
        
        SET element at position three of digits TO the element of count at character 'h' MINUS the element at position eight of digits  
        SET element at position five of digits TO the element of count at character 'f' MINUS the element at position four of digits  
        SET element at position seven of digits TO the element of count at character 's' MINUS the element at position six of digits  
        SET element at position nine of digits TO the element of count at character 'i' MINUS the element at position five of digits MINUS the element at position six of digits MINUS the element at position eight of digits  
        SET element at position one of digits TO the element of count at character 'o' MINUS the element at position zero of digits MINUS the element at position two of digits MINUS the element at position four of digits  
        
        SET result TO an empty list  
        FOR i FROM zero TO nine  
            APPEND the string representation of i repeated the element at position i of digits times TO result  
        END FOR  
        
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
