CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val DEFAULT zero, next DEFAULT none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode INITIALIZED WITH element at position zero of values  
    SET p TO head  
    FOR each val IN values STARTING FROM position one TO the END  
        SET node TO new ListNode INITIALIZED WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1, p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the RESULT of p1.val EQUALS p2.val AND is_same_list WITH PARAMETERS p1.next, p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween WITH PARAMETERS head, left, right  
        IF head EQUALS none OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO new ListNode INITIALIZED WITH zero  
        SET dummy.next TO head  
        SET prev TO dummy  
        FOR index FROM zero TO left MINUS one MINUS one  
            SET prev TO prev.next  
        END FOR  
        SET current TO prev.next  
        FOR index FROM zero TO right MINUS left MINUS one  
            SET next_node TO current.next  
            SET current.next TO next_node.next  
            SET next_node.next TO prev.next  
            SET prev.next TO next_node  
        END FOR  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
