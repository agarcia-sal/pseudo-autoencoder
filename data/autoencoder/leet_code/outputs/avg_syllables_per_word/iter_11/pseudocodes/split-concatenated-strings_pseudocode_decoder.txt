CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the element at position index of strs REVERSED  
            IF reversed_str GREATER THAN the element at position index of strs  
                SET the element at position index of strs TO reversed_str  
            END IF  
        END FOR  
        
        SET doubled_strs TO strs APPENDED TO strs  
        
        SET max_string TO empty string  
        
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN the list containing false AND true  
                IF is_reversed EQUALS true  
                    SET current TO the element at position index of strs REVERSED  
                ELSE  
                    SET current TO the element at position index of strs  
                END IF  
                
                SET remaining TO the concatenation of elements from position index PLUS one TO position index PLUS the LENGTH OF strs MINUS one of doubled_strs  
                
                FOR position FROM zero TO the LENGTH OF current  
                    SET candidate TO the substring of current from position TO the END CONCATENATED WITH remaining CONCATENATED WITH the substring of current from zero TO position  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
