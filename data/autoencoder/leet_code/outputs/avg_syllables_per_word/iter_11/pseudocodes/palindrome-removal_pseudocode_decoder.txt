CLASS Solution  
    FUNCTION minimumMoves(arr)  
        SET n TO the LENGTH OF arr  
        SET dp TO a two-dimensional list of zeros with dimensions n by n  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i and i of dp TO one  
        END FOR  
        
        FOR i FROM zero TO n MINUS two  
            IF element at position i of arr EQUALS element at position i PLUS one of arr  
                SET element at position i and i PLUS one of dp TO one  
            ELSE  
                SET element at position i and i PLUS one of dp TO two  
            END IF  
        END FOR  
        
        FOR length FROM three TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of arr EQUALS element at position j of arr  
                    SET element at position i and j of dp TO element at position i PLUS one and j MINUS one of dp  
                ELSE  
                    SET element at position i and j of dp TO infinity  
                    FOR k FROM i TO j MINUS one  
                        SET temp_sum TO element at position i and k of dp PLUS element at position k PLUS one and j of dp  
                        IF temp_sum LESS THAN element at position i and j of dp  
                            SET element at position i and j of dp TO temp_sum  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero and n MINUS one of dp  
    END FUNCTION  
END CLASS
