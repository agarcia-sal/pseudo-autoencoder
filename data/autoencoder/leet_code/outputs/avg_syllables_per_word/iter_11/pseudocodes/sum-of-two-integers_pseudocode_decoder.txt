CLASS Solution  
    FUNCTION getSum(a, b)  
        SET MAX TO seven thousand two hundred thirty four million seven hundred forty eight thousand seven hundred fifty five  
        SET mask TO four billion two hundred ninety four million nine hundred sixty seven thousand two hundred ninety five  
        WHILE b NOT EQUALS zero  
            SET carry TO the bitwise AND of a and b MULTIPLIED BY two  
            SET a TO the bitwise XOR of a and b AND mask  
            SET b TO carry AND mask  
        END WHILE  
        IF a LESS THAN OR EQUAL TO MAX  
            RETURN a  
        ELSE  
            RETURN the bitwise NOT of the bitwise XOR of a and mask  
        END IF  
    END FUNCTION  
END CLASS
