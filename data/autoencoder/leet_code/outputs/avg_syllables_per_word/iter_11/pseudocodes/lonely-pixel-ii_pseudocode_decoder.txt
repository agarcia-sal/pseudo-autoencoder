CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture EQUALS empty OR element at position zero of picture EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO list of zero values of size m  
        SET col_count TO list of zero values of size n  
        SET row_patterns TO empty mapping with default integer zero  
        
        FOR index r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of all elements at position r of picture  
            INCREMENT the value in row_patterns at key pattern BY one  
            FOR index c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS the character B  
                    INCREMENT element at position r of row_count BY one  
                    INCREMENT element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR index r FROM zero TO m MINUS one  
            FOR index c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS the character B  
                    SET pattern TO the concatenation of all elements at position r of picture  
                    IF element at position r of row_count EQUALS target AND element at position c of col_count EQUALS target AND value in row_patterns at key pattern EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
