CLASS Solution  
    FUNCTION kSimilarity(s1, s2)  
        FUNCTION neighbors(s)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND element at position i of s EQUALS element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF element at position j of s EQUALS element at position i of s2 AND element at position j of s NOT EQUALS element at position j of s2  
                    YIELD concatenation of substring of s from start TO i MINUS one PLUS element at position j of s PLUS substring of s from i PLUS one TO j MINUS one PLUS element at position i of s PLUS substring of s from j PLUS one TO end  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET queue TO double ended queue CONTAINING tuple of s1 and zero  
        SET visited TO set CONTAINING s1  
        
        WHILE queue IS NOT empty  
            SET s, swaps TO left most element of queue REMOVED  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors(s)  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND tuple of neighbor and swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
