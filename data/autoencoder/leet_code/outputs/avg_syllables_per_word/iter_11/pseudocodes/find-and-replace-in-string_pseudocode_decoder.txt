CLASS Solution  
    FUNCTION findReplaceString(s, indices, sources, targets)  
        SET replacements TO the list of tuples of element at position i of indices, element at position i of sources, element at position i of targets FOR each i FROM zero TO the LENGTH OF indices MINUS one, sorted BY the first element of each tuple  
        SET result TO an empty list  
        SET prev_end TO zero  
        FOR each tuple of index, source, target IN replacements  
            APPEND the substring of s from prev_end TO index MINUS one TO result  
            IF the substring of s from index TO index PLUS the LENGTH OF source MINUS one EQUALS source  
                APPEND target TO result  
            ELSE  
                APPEND the substring of s from index TO index PLUS the LENGTH OF source MINUS one TO result  
            END IF  
            SET prev_end TO index PLUS the LENGTH OF source  
        END FOR  
        APPEND the substring of s from prev_end TO the end TO result  
        RETURN the concatenation of all elements of result as a single string  
    END FUNCTION  
END CLASS
