CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR index i FROM zero TO N MINUS one  
            FOR index j FROM zero TO N MINUS one  
                IF the element at position i of board AND XOR element at position zero of the row i of board AND XOR element at position j of the row zero of board AND XOR element at position zero of the row zero of board EQUALS one  
                    RETURN negative one  
                END IF  
            END FOR  
            INCREMENT rowSum BY the element at position zero of the row i of board  
            INCREMENT colSum BY the element at position zero of the row i of board at position zero swapped to column i of board zero row  
            INCREMENT rowSwap BY ONE IF the element at position zero of the row i of board EQUALS the remainder of i DIVIDED BY two ELSE zero  
            INCREMENT colSwap BY ONE IF the element at position zero of the row zero of board at position i EQUALS the remainder of i DIVIDED BY two ELSE zero  
        END FOR  
        IF integer division of N BY two GREATER THAN rowSum OR rowSum GREATER THAN integer division of the sum of N AND one BY two  
            RETURN negative one  
        END IF  
        IF integer division of N BY two GREATER THAN colSum OR colSum GREATER THAN integer division of the sum of N AND one BY two  
            RETURN negative one  
        END IF  
        IF the remainder of N DIVIDED BY two EQUALS one  
            IF the remainder of rowSwap DIVIDED BY two EQUALS one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of colSwap DIVIDED BY two EQUALS one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the MINIMUM OF rowSwap AND N MINUS rowSwap  
            SET colSwap TO the MINIMUM OF colSwap AND N MINUS colSwap  
        END IF  
        RETURN integer division of the sum of rowSwap AND colSwap BY two  
    END FUNCTION  
END CLASS
