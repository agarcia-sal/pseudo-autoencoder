CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO an empty list  
        SET max_value TO negative infinity  
      
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            CALL helper_push_heap WITH min_heap AND the element at position zero of the element at position i of nums, i, and zero  
            SET max_value TO the GREATER THAN of max_value AND the element at position zero of the element at position i of nums  
        END FOR  
      
        SET smallest_range TO positive infinity  
        SET result TO the list containing negative one hundred thousand, and one hundred thousand  
      
        WHILE min_heap is not empty  
            CALL helper_pop_heap WITH min_heap AND STORE the results into min_value, list_index, element_index  
          
            IF max_value MINUS min_value LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO the list containing min_value AND max_value  
            END IF  
          
            IF element_index PLUS one LESS THAN the LENGTH OF the element at position list_index of nums  
                SET next_value TO the element at position element_index PLUS one of the element at position list_index of nums  
                CALL helper_push_heap WITH min_heap AND next_value, list_index, element_index PLUS one  
                SET max_value TO the GREATER THAN of max_value AND next_value  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
      
        RETURN result  
    END FUNCTION  
  
    FUNCTION helper_push_heap(heap, value1, value2, value3)  
        PERFORM the action of pushing the tuple containing value1, value2, and value3 into heap as a min-heap  
    END FUNCTION  
  
    FUNCTION helper_pop_heap(heap)  
        PERFORM the action of popping the smallest element from heap as a min-heap AND RETURN its elements as value1, value2, value3  
    END FUNCTION  
END CLASS
