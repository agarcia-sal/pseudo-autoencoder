CLASS Solution  
    FUNCTION uniquePaths(m, n)  
        SET dp TO the result of calling createMatrix WITH parameters m AND n  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET element at position i, j of dp TO the element at position i MINUS one, j of dp PLUS the element at position i, j MINUS one of dp  
            END FOR  
        END FOR  
        RETURN the element at position m MINUS one, n MINUS one of dp  
    END FUNCTION  

    FUNCTION createMatrix(rows, columns)  
        SET matrix TO an empty list  
        FOR each index FROM zero TO rows MINUS one  
            SET row TO an empty list  
            FOR each index FROM zero TO columns MINUS one  
                APPEND one TO row  
            END FOR  
            APPEND row TO matrix  
        END FOR  
        RETURN matrix  
    END FUNCTION  
END CLASS
