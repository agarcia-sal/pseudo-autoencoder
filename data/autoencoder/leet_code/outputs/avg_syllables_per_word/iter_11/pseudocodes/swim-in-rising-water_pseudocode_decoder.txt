CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an empty set  
        SET min_heap TO a list containing a tuple of the element at position zero zero of grid, zero, zero  
        ADD the tuple of zero zero TO visited  
        
        SET directions TO a list containing tuples zero one, one zero, zero MINUS one, MINUS one zero  
        
        WHILE min_heap is not empty  
            REMOVE the smallest element FROM min_heap AND DESTRUCTURE it INTO t, i, j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            
            FOR each tuple di, dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND the tuple ni, nj NOT IN visited  
                    ADD the tuple ni, nj TO visited  
                    ADD to min_heap the tuple of the maximum of t AND the element at position ni nj of grid, ni, nj  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN MINUS one  
    END FUNCTION  
END CLASS
