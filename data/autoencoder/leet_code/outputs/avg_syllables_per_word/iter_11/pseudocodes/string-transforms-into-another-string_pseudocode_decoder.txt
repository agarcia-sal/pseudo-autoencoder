CLASS Solution  
    FUNCTION canConvert(str1, str2)  
        IF str1 EQUALS str2  
            RETURN True  
        END IF  
  
        SET conversion_map TO empty mapping  
  
        FOR each char1 AND char2 IN pairs of characters from str1 AND str2  
            IF char1 EXISTS IN conversion_map  
                IF the value at char1 in conversion_map NOT EQUALS char2  
                    RETURN False  
                END IF  
            ELSE  
                SET the value at char1 in conversion_map TO char2  
            END IF  
        END FOR  
  
        IF the LENGTH OF the set of characters in str2 EQUALS twenty six  
            RETURN False  
        END IF  
  
        RETURN True  
    END FUNCTION  
END CLASS
