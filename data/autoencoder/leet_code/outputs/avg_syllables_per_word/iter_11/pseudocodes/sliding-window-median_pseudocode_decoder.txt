CLASS Solution  
    FUNCTION medianSlidingWindow(nums, k)  
        SET lo TO empty list  
        SET hi TO empty list  

        FUNCTION balance_heaps()  
            IF the LENGTH OF lo IS GREATER THAN the LENGTH OF hi PLUS one  
                CALL heappush WITH hi AND NEGATIVE OF the RESULT OF heappop FROM lo  
            ELSE IF the LENGTH OF hi IS GREATER THAN the LENGTH OF lo  
                CALL heappush WITH lo AND NEGATIVE OF the RESULT OF heappop FROM hi  
            END IF  
        END FUNCTION  

        FUNCTION add_num(num)  
            IF lo IS empty OR num IS LESS THAN OR EQUAL TO NEGATIVE OF the FIRST element OF lo  
                CALL heappush WITH lo AND NEGATIVE OF num  
            ELSE  
                CALL heappush WITH hi AND num  
            END IF  
            CALL balance_heaps()  
        END FUNCTION  

        FUNCTION remove_num(num)  
            IF num IS LESS THAN OR EQUAL TO NEGATIVE OF the FIRST element OF lo  
                REMOVE NEGATIVE OF num FROM lo  
                IF lo IS NOT empty  
                    CALL heapify WITH lo  
                END IF  
            ELSE  
                REMOVE num FROM hi  
                IF hi IS NOT empty  
                    CALL heapify WITH hi  
                END IF  
            END IF  
            CALL balance_heaps()  
        END FUNCTION  

        FUNCTION get_median()  
            IF the LENGTH OF lo IS GREATER THAN the LENGTH OF hi  
                RETURN NEGATIVE OF the FIRST element OF lo  
            END IF  
            RETURN THE SUM OF NEGATIVE OF the FIRST element OF lo AND the FIRST element OF hi DIVIDED BY two  
        END FUNCTION  

        FOR index FROM zero TO k MINUS one  
            CALL add_num WITH element at position index OF nums  
        END FOR  

        SET medians TO a list containing the RESULT OF get_median()  

        FOR index FROM k TO the LENGTH OF nums MINUS one  
            CALL add_num WITH element at position index OF nums  
            CALL remove_num WITH element at position index MINUS k OF nums  
            APPEND the RESULT OF get_median() TO medians  
        END FOR  

        RETURN medians  
    END FUNCTION  
END CLASS
