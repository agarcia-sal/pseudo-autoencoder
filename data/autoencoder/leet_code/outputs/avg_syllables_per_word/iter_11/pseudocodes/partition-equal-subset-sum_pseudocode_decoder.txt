CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO zero  
        FOR each number IN nums  
            INCREMENT total_sum BY number  
        END FOR  
        
        IF total_sum MODULO two NOT EQUALS zero  
            RETURN False  
        END IF  
        
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        
        CALL initialize_dp WITH target  
        
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET dp at position i TO dp at position i OR dp at position i MINUS num  
            END FOR  
        END FOR  
        
        RETURN dp at position target  
    END FUNCTION
    
    FUNCTION initialize_dp(target)  
        CREATE dp AS list of False values with LENGTH target PLUS one  
        SET dp at position zero TO True  
    END FUNCTION  
END CLASS
