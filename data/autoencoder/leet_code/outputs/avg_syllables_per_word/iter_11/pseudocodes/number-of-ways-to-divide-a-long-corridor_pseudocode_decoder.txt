CLASS Solution  
    FUNCTION numberOfWays(corridor)  
        SET MOD TO one billion PLUS seven  
        SET seats TO empty list  
        FOR each index, character IN corridor with index  
            IF character EQUALS letter S  
                APPEND index TO seats  
            END IF  
        END FOR  

        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  

        SET ways TO one  
        FOR index FROM two TO the LENGTH OF seats MINUS one STEP two  
            SET ways TO ways MULTIPLIED BY the difference BETWEEN element at position index of seats AND element at position index MINUS one of seats  
            SET ways TO ways MODULO MOD  
        END FOR  

        RETURN ways  
    END FUNCTION  
END CLASS
