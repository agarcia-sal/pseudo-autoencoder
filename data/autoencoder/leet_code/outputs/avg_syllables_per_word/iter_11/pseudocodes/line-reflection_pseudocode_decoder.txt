CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the SET OF converting each element of points TO a tuple  
        SET min_x TO the MINIMUM OF the element at position zero of each point IN unique_points  
        SET max_x TO the MAXIMUM OF the element at position zero of each point IN unique_points  
        SET line_of_reflection TO min_x PLUS max_x DIVIDED BY two  
        FOR each pair of x and y IN unique_points  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the pair of reflected_x and y IS NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
