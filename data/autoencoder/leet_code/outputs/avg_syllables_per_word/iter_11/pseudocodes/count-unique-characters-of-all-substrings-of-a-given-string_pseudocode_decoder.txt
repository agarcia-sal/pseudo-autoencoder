CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        
        SET last TO a list of negative one repeated twenty six times  
        SET last2 TO a list of negative one repeated twenty six times  
        
        SET result TO zero  
        
        FOR each index i AND character char IN s  
            SET idx TO the code of char MINUS the code of uppercase letter A  
            
            SET result TO the remainder after dividing the sum of result AND the product of the difference of i MINUS the element at position idx of last AND the difference of the element at position idx of last MINUS the element at position idx of last2 BY MOD  
            
            SET the element at position idx of last2 TO the element at position idx of last  
            SET the element at position idx of last TO i  
        END FOR  
        
        FOR i FROM zero TO twenty five  
            SET result TO the remainder after dividing the sum of result AND the product of the difference of n MINUS the element at position i of last AND the difference of the element at position i of last MINUS the element at position i of last2 BY MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
