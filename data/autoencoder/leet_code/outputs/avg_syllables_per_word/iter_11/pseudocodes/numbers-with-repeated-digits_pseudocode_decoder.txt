CLASS Solution  
    FUNCTION numDupDigitsAtMostN(n)  
        FUNCTION count_unique_digits(x)  
            SET s TO the string representation of x  
            SET length TO the length of s  
            SET count TO zero  
            FOR i FROM one TO length MINUS one  
                INCREMENT count BY nine MULTIPLIED BY perm(nine, i MINUS one)  
            END FOR  
            SET seen TO empty set  
            FOR each position i and character digit AT position i OF s  
                FOR d FROM one IF i EQUALS zero ELSE zero TO digit converted to integer MINUS one  
                    IF d NOT IN seen  
                        INCREMENT count BY perm(nine MINUS i, length MINUS i MINUS one)  
                    END IF  
                END FOR  
                IF digit converted to integer IN seen  
                    BREAK out of the FOR loop  
                END IF  
                ADD digit converted to integer TO seen  
            END FOR  
            OTHERWISE  
                INCREMENT count BY one  
            RETURN count  
        END FUNCTION  
  
        RETURN n MINUS count_unique_digits(n)  
    END FUNCTION  
  
    FUNCTION perm(m, n)  
        IF n EQUALS zero  
            RETURN one  
        END IF  
        SET result TO one  
        FOR i FROM zero TO n MINUS one  
            SET result TO result MULTIPLIED BY (m MINUS i)  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
