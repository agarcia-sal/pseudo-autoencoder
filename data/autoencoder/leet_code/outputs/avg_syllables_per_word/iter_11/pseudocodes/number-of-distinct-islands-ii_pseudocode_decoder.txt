CLASS Solution  
    FUNCTION numDistinctIslands2(grid)  
        FUNCTION dfs(x, y, island)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF the ELEMENT AT position zero OF grid OR the ELEMENT AT position x OF grid AT position y EQUALS zero  
                RETURN  
            END IF  
            SET the ELEMENT AT position x OF grid AT position y TO zero  
            APPEND tuple of x AND y TO island  
            CALL dfs WITH x PLUS one, y, island  
            CALL dfs WITH x MINUS one, y, island  
            CALL dfs WITH x, y PLUS one, island  
            CALL dfs WITH x, y MINUS one, island  
        END FUNCTION  

        FUNCTION normalize(island)  
            SET shapes TO empty list  
            FOR each dx AND dy IN list of tuples of one AND one, one AND MINUS one, MINUS one AND one, MINUS one AND MINUS one  
                SET new_island TO empty list  
                FOR each x AND y IN island  
                    APPEND tuple of x MULTIPLIED BY dx AND y MULTIPLIED BY dy TO new_island  
                END FOR  
                SORT new_island  
                SET min_x TO the MINIMUM x value FROM elements x AND y IN new_island  
                SET min_y TO the MINIMUM y value FROM elements x AND y IN new_island  
                APPEND tuple of each x MINUS min_x AND y MINUS min_y FOR each x AND y IN new_island TO shapes  
            END FOR  
            FOR index FROM zero TO the LENGTH OF shapes MINUS one  
                APPEND tuple of each y AND MINUS x FOR each x AND y IN the ELEMENT AT position index OF shapes TO shapes  
            END FOR  
            RETURN the MINIMUM element FROM shapes  
        END FUNCTION  

        SET islands TO empty set  
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF the ELEMENT AT position zero OF grid MINUS one  
                IF the ELEMENT AT position i OF grid AT position j EQUALS one  
                    SET island TO empty list  
                    CALL dfs WITH i, j, island  
                    ADD the RESULT OF normalize WITH island TO islands  
                END IF  
            END FOR  
        END FOR  
        RETURN the LENGTH OF islands  
    END FUNCTION  
END CLASS
