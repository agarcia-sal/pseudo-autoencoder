CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO one billion PLUS seven  
        SET dp at position zero TO zero  
        SET dp at position one TO zero  
        SET dp at position two TO zero  
        
        FOR each num IN nums  
            IF num EQUALS zero  
                SET dp at position zero TO the remainder of two MULTIPLIED BY dp at position zero PLUS one DIVIDED BY MOD  
            ELSE IF num EQUALS one  
                SET dp at position one TO the remainder of two MULTIPLIED BY dp at position one PLUS dp at position zero DIVIDED BY MOD  
            ELSE IF num EQUALS two  
                SET dp at position two TO the remainder of two MULTIPLIED BY dp at position two PLUS dp at position one DIVIDED BY MOD  
            END IF  
        END FOR  
        
        RETURN dp at position two  
    END FUNCTION  
END CLASS
