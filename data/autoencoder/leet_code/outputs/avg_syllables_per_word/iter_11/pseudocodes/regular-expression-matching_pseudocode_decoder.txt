CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO result of create_two_dimensional_list WITH parameters length of s PLUS one, length of p PLUS one, False  
        SET element at position zero zero of dp TO True  
        
        FOR j FROM one TO length of p  
            IF element at position j MINUS one of p EQUALS '*'
                SET element at position zero j of dp TO element at position zero j MINUS two of dp
            END IF
        END FOR  
        
        FOR i FROM one TO length of s  
            FOR j FROM one TO length of p  
                IF element at position j MINUS one of p EQUALS element at position i MINUS one of s OR element at position j MINUS one of p EQUALS '.'  
                    SET element at position i j of dp TO element at position i MINUS one j MINUS one of dp  
                ELSE IF element at position j MINUS one of p EQUALS '*'  
                    SET element at position i j of dp TO element at position i j MINUS two of dp  
                    IF element at position j MINUS two of p EQUALS element at position i MINUS one of s OR element at position j MINUS two of p EQUALS '.'  
                        SET element at position i j of dp TO element at position i j of dp OR element at position i MINUS one j of dp  
                    END IF
                END IF
            END FOR  
        END FOR  
        
        RETURN element at position length of s length of p of dp  
    END FUNCTION

    FUNCTION create_two_dimensional_list(rows, columns, initial_value)  
        SET result TO empty list  
        FOR each index FROM zero TO rows MINUS one  
            SET row TO empty list  
            FOR each index_column FROM zero TO columns MINUS one  
                APPEND initial_value TO row  
            END FOR  
            APPEND row TO result  
        END FOR  
        RETURN result  
    END FUNCTION
END CLASS
