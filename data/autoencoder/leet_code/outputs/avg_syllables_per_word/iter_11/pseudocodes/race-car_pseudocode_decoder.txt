CLASS Solution  
    FUNCTION racecar(target)  
        SET queue TO a new double-ended queue containing a tuple of zero position, one speed, zero steps  
        SET visited TO a new set containing a tuple of zero position and one speed  
        
        WHILE queue is not empty  
            REMOVE first element FROM queue AND UNPACK INTO position, speed, steps  
            
            IF position EQUALS target  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF tuple of next_position and next_speed NOT IN visited AND next_position GREATER THAN OR EQUAL TO zero AND next_position LESS THAN two MULTIPLIED BY target  
                ADD tuple of next_position and next_speed TO visited  
                APPEND tuple of next_position, next_speed, steps PLUS one TO queue  
            END IF  
            
            IF speed GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF tuple of position and next_speed NOT IN visited AND position GREATER THAN OR EQUAL TO zero AND position LESS THAN two MULTIPLIED BY target  
                ADD tuple of position and next_speed TO visited  
                APPEND tuple of position, next_speed, steps PLUS one TO queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
