CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF nums1 EQUALS empty OR nums2 EQUALS empty OR k LESS THAN OR EQUAL TO zero  
            RETURN empty list  
        END IF  
        
        SET min_heap TO empty list  
        
        FOR i FROM zero TO minimum OF k AND length of nums1 MINUS one  
            CALL heap push FUNCTION TO insert the tuple of sum of element at position i of nums1 PLUS element at position zero of nums2, i, and zero INTO min_heap  
        END FOR  
        
        SET result TO empty list  
        
        WHILE min_heap IS NOT empty AND length of result LESS THAN k  
            CALL heap pop FUNCTION TO remove and GET smallest element AS tuple underscore, i, and j FROM min_heap  
            APPEND list containing element at position i of nums1 AND element at position j of nums2 TO result  
            
            IF j PLUS one LESS THAN length of nums2  
                CALL heap push FUNCTION TO insert the tuple of sum of element at position i of nums1 PLUS element at position j PLUS one of nums2, i, and j PLUS one INTO min_heap  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
