CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO a new default dictionary with default value as empty list  
        FOR each string s IN strs  
            SET sorted_str TO the concatenation of characters in the sorted version of s  
            APPEND s TO the list at key sorted_str in anagrams  
        END FOR  
        SET result TO the list of all values in anagrams  
        RETURN result  
    END FUNCTION  
END CLASS
