CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO CONVERT wordDict TO set  
        SET memo TO empty dictionary  
          
        FUNCTION backtrack(start)  
            IF start IS IN memo  
                RETURN value AT key start IN memo  
            END IF  
            IF start EQUALS the LENGTH OF s  
                RETURN a list containing an empty string  
            END IF  
              
            SET res TO empty list  
            FOR end FROM start PLUS one TO the LENGTH OF s PLUS one  
                SET word TO substring of s FROM start TO end  
                IF word IS IN wordSet  
                    FOR each sentence IN backtrack(end)  
                        IF sentence IS NOT empty  
                            APPEND concatenation of word, space, and sentence TO res  
                        ELSE  
                            APPEND word TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
              
            SET memo AT key start TO res  
            RETURN res  
        END FUNCTION  
          
        RETURN backtrack(zero)  
END CLASS
