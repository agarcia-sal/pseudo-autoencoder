CLASS Solution  
    FUNCTION findIntegers(n)  
        SET bin_rep TO the binary representation of n starting from the second character  
        SET length TO the LENGTH OF bin_rep  
        
        CALL initialize_dp_list(length) AND STORE RESULT IN dp  
        
        FOR index FROM two TO length  
            SET element at position index of dp TO element at position index MINUS one of dp PLUS element at position index MINUS two of dp  
        END FOR  
        
        SET result TO zero  
        SET prev_bit TO zero  
        
        FOR index FROM zero TO length MINUS one  
            IF element at position index of bin_rep EQUALS one as character  
                INCREMENT result BY element at position length MINUS index MINUS one of dp  
                IF prev_bit EQUALS one  
                    DECREMENT result BY one  
                    BREAK the loop  
                END IF  
            END IF  
            SET prev_bit TO one IF element at position index of bin_rep EQUALS one as character OTHERWISE zero  
        END FOR  
        
        RETURN result PLUS one  
    END FUNCTION  
    
    FUNCTION initialize_dp_list(length)  
        CREATE a list dp with length PLUS one elements all SET TO zero  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO two  
        RETURN dp  
    END FUNCTION  
END CLASS
