CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO a new empty dictionary  
        FOR each character c AND index i IN order starting from zero  
            SET the value at key c in d TO i  
        END FOR  
        SET sorted_list TO a new list containing the elements of s sorted by the value obtained from d at key element or zero if key not found  
        SET result_string TO the concatenation of all elements in sorted_list in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
