CLASS Solution  
    FUNCTION kEmptySlots(bulbs, k)  
        SET bloom_days TO empty list  
        FOR each day and position IN bulbs with day STARTING FROM one  
            SET i TO the index where position SHOULD BE inserted in bloom_days to KEEP IT sorted  
            IF i GREATER THAN zero AND position MINUS element at position i MINUS one EQUALS k  
                RETURN day  
            END IF  
            IF i LESS THAN the LENGTH OF bloom_days AND element at position i OF bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position INTO bloom_days AT position i  
        END FOR  
        RETURN minus one  
    END FUNCTION  
END CLASS
