CLASS Solution  
    FUNCTION numMagicSquaresInside(grid)  
        FUNCTION is_magic(i, j)  
            IF element at position i comma j of grid NOT EQUALS five  
                RETURN false  
            END IF  
            
            SET numbers TO empty set  
            FOR x FROM i MINUS one TO i PLUS one  
                FOR y FROM j MINUS one TO j PLUS one  
                    IF element at position x comma y of grid LESS THAN one OR element at position x comma y of grid GREATER THAN nine OR element at position x comma y of grid IN numbers  
                        RETURN false  
                    END IF  
                    ADD element at position x comma y of grid TO numbers  
                END FOR  
            END FOR  
            
            FOR start FROM i MINUS one TO i PLUS one  
                IF element at position start comma j MINUS one of grid PLUS element at position start comma j of grid PLUS element at position start comma j PLUS one of grid NOT EQUALS fifteen OR element at position i MINUS one comma start of grid PLUS element at position i comma start of grid PLUS element at position i PLUS one comma start of grid NOT EQUALS fifteen  
                    RETURN false  
                END IF  
            END FOR  
            
            IF element at position i MINUS one comma j MINUS one of grid PLUS element at position i comma j of grid PLUS element at position i PLUS one comma j PLUS one of grid NOT EQUALS fifteen OR element at position i MINUS one comma j PLUS one of grid PLUS element at position i comma j of grid PLUS element at position i PLUS one comma j MINUS one of grid NOT EQUALS fifteen  
                RETURN false  
            END IF  
            
            RETURN true  
        END FUNCTION  
        
        SET count TO zero  
        SET rows TO LENGTH OF grid  
        SET cols TO LENGTH OF element at position zero of grid  
        
        FOR i FROM one TO rows MINUS two  
            FOR j FROM one TO cols MINUS two  
                IF is_magic(i, j)  
                    INCREMENT count BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
