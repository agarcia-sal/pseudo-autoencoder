CLASS Solution  
    FUNCTION topKFrequent WITH parameters words, k  
        SET count TO RESULT OF CALLING Counter WITH words  
        SET heap TO EMPTY LIST  
        FOR each word AND freq IN count ITEMS  
            APPEND TUPLE OF negative freq AND word TO heap  
        END FOR  
        CALL heapify FUNCTION WITH heap  
        SET result TO EMPTY LIST  
        FOR _ FROM one TO k  
            SET top_element TO RESULT OF heap pop FROM heap  
            SET word_at_top TO second element OF top_element  
            APPEND word_at_top TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
