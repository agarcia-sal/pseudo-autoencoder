CLASS Solution  
    FUNCTION canVisitAllRooms WITH PARAMETER rooms  
        SET visited TO new empty set  
        SET stack TO new list containing zero  
        WHILE stack is not empty  
            SET current_room TO the element removed from the end of stack  
            ADD current_room TO visited  
            FOR each key IN the element at position current_room of rooms  
                IF key is not in visited  
                    APPEND key TO stack  
                END IF  
            END FOR  
        END WHILE  
        RETURN the LENGTH OF visited EQUALS the LENGTH OF rooms  
    END FUNCTION  
END CLASS
