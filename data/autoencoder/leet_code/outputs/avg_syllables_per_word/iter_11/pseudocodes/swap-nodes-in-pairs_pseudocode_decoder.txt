CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with val equal to element at position zero of values and next equal to None  
    SET p TO head  
    FOR each val IN elements from position one TO end of values  
        SET node TO new ListNode with val equal to val and next equal to None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN (value of p1 EQUALS value of p2) AND is_same_list(next element of p1, next element of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs(self, head)  
        SET dummy TO new ListNode with val equal to zero and next equal to None  
        SET dummy.next TO head  
        SET current TO dummy  
        
        WHILE next element of current IS NOT None AND next of next element of current IS NOT None  
            SET first TO next element of current  
            SET second TO next of next element of current  
            
            SET current.next TO second  
            SET first.next TO next element of second  
            SET second.next TO first  
            
            SET current TO next of next element of current  
        END WHILE  
        
        RETURN next element of dummy  
    END FUNCTION  
END CLASS
