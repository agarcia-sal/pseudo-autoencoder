CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO an empty set  
        FOR each mine IN mines  
            APPEND the tuple of mine TO mines_set  
        END FOR  
        
        SET dp TO a list containing n elements where each element is a list containing n elements where each element is a list with four zeros representing directions right, down, left, up  
        
        SET directions TO a list containing the tuples zero and one, one and zero, zero and minus one, minus one and zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the tuple of i and j NOT IN mines_set  
                    IF j GREATER THAN zero  
                        SET the element at position zero of the element at position j of the element at position i of dp TO the element at position zero of the element at position j MINUS one of the element at position i of dp PLUS one  
                    END IF  
                    IF i GREATER THAN zero  
                        SET the element at position one of the element at position j of the element at position i of dp TO the element at position one of the element at position j of the element at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the tuple of i and j NOT IN mines_set  
                    IF j LESS THAN n MINUS one  
                        SET the element at position two of the element at position j of the element at position i of dp TO the element at position two of the element at position j PLUS one of the element at position i of dp PLUS one  
                    END IF  
                    IF i LESS THAN n MINUS one  
                        SET the element at position three of the element at position j of the element at position i of dp TO the element at position three of the element at position j of the element at position i PLUS one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum value of the list at the element at position j of the element at position i of dp  
                    IF plus_sign_order GREATER THAN result  
                        SET result TO plus_sign_order  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
