CLASS Solution  
    FUNCTION wiggleMaxLength(nums)  
        IF the LENGTH OF nums LESS THAN 2  
            RETURN the LENGTH OF nums  
        END IF  
        
        SET up TO one  
        SET down TO one  
        
        FOR index FROM one TO the LENGTH OF nums MINUS one  
            IF element at position index of nums GREATER THAN element at position index MINUS one of nums  
                SET up TO down PLUS one  
            ELSE IF element at position index of nums LESS THAN element at position index MINUS one of nums  
                SET down TO up PLUS one  
            END IF  
        END FOR  
        
        RETURN the GREATER OF up AND down  
    END FUNCTION  
END CLASS
