CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a list containing minus one  
        SET max_length TO zero  
        FOR each index i AND character char IN s  
            IF char EQUALS open parenthesis  
                APPEND i TO stack  
            ELSE  
                REMOVE last element FROM stack  
                IF stack IS empty  
                    APPEND i TO stack  
                ELSE  
                    SET max_length TO the greater OF max_length AND i MINUS the last element OF stack  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
