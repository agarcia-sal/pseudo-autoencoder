CLASS Solution  
    FUNCTION countArrangement(n)  
        FUNCTION backtrack(used, index)  
            IF index EQUALS n PLUS one  
                RETURN one  
            END IF  
            SET count TO zero  
            FOR i FROM one TO n  
                IF USED element at position i EQUALS False AND (i MODULO index EQUALS zero OR index MODULO i EQUALS zero)  
                    SET USED element at position i TO True  
                    INCREMENT count BY backtrack(used, index PLUS one)  
                    SET USED element at position i TO False  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        SET used TO a list of False repeated n PLUS one times  
        RETURN backtrack(used, one)  
    END FUNCTION  
END CLASS
