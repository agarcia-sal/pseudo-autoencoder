CLASS Solution  
    FUNCTION reorderedPowerOf2(n)  
        SET power_of_2_sorted_digits TO empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 LESS THAN OR EQUAL TO ten raised to the power of nine  
            APPEND tuple of sorted string representation of power_of_2 TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  
        SET sorted_n_digits TO tuple of sorted string representation of n  
        RETURN sorted_n_digits IS IN power_of_2_sorted_digits  
    END FUNCTION  
END CLASS
