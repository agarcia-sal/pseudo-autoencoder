CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val EQUALS element at position zero of values  
    SET i TO one  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element of queue removed  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with val EQUALS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with val EQUALS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION helper(node, current_sum, prefix_sums)  
            IF node EQUALS None  
                RETURN zero  
            END IF  
            INCREMENT current_sum BY node.val  
            SET paths_to_here TO value at key current_sum MINUS targetSum in prefix_sums OR zero if key not present  
            SET value at key current_sum in prefix_sums TO value at key current_sum in prefix_sums OR zero if key not present PLUS one  
            SET paths_in_left TO helper(node.left, current_sum, prefix_sums)  
            SET paths_in_right TO helper(node.right, current_sum, prefix_sums)  
            DECREMENT value at key current_sum in prefix_sums BY one  
            RETURN paths_to_here PLUS paths_in_left PLUS paths_in_right  
        END FUNCTION  
        SET prefix_sums TO dictionary with key zero and value one  
        RETURN helper(root, zero, prefix_sums)  
    END FUNCTION  
END CLASS
