CLASS Solution  
    FUNCTION minMovesToMakePalindrome(s)  
        SET s TO list of characters in s  
        SET n TO the LENGTH OF s  
        SET moves TO zero  
        
        WHILE the LENGTH OF s is GREATER THAN one  
            SET matched TO false  
            FOR i FROM the LENGTH OF s MINUS one DOWN TO one  
                IF element at position i of s EQUALS element at position zero of s  
                    SET matched TO true  
                    FOR j FROM i TO the LENGTH OF s MINUS two  
                        SWAP element at position j of s WITH element at position j PLUS one of s  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE first element FROM s  
                    REMOVE last element FROM s  
                    BREAK  
                END IF  
            END FOR  
            IF matched EQUALS false  
                REMOVE first element FROM s  
            END IF  
        END WHILE  
        
        RETURN moves  
    END FUNCTION  
END CLASS
