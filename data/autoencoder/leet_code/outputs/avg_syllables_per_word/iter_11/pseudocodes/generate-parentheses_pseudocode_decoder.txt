CLASS Solution  
    FUNCTION generateParenthesis WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS s, left, right  
            IF LENGTH OF s EQUALS two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s PLUS open parenthesis character, left INCREMENTED BY one, right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s PLUS close parenthesis character, left, right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        
        SET result TO empty list  
        CALL backtrack WITH empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
