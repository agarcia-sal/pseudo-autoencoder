CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO the frequency count of digits  
        SET total_sum TO the sum of all digits  
        SET remainder TO total_sum MODULO three  
        
        IF remainder EQUALS one  
            FOR each d IN list containing one, four, seven  
                IF the count of d GREATER THAN zero  
                    DECREMENT the count of d BY one  
                    BREAK  
                END IF  
            END FOR  
            ELSE  
                FOR each d IN list containing two, five, eight  
                    IF the count of d GREATER THAN one  
                        DECREMENT the count of d BY two  
                        BREAK  
                    END IF  
                END FOR  
        END IF  
        
        IF remainder EQUALS two  
            FOR each d IN list containing two, five, eight  
                IF the count of d GREATER THAN zero  
                    DECREMENT the count of d BY one  
                    BREAK  
                END IF  
            END FOR  
            ELSE  
                FOR each d IN list containing one, four, seven  
                    IF the count of d GREATER THAN one  
                        DECREMENT the count of d BY two  
                        BREAK  
                    END IF  
                END FOR  
        END IF  
        
        SET result TO an empty list  
        FOR d FROM nine DOWN TO zero  
            APPEND the string form of digit d repeated the count of d times TO result  
        END FOR  
        
        SET final_number TO the concatenation of result elements  
        
        IF final_number IS NOT empty AND the element at position zero of final_number EQUALS zero  
            RETURN the string zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
