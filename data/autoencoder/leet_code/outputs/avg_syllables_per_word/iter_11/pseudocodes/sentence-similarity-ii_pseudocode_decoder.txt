CLASS Solution  
    FUNCTION areSentencesSimilarTwo(sentence1, sentence2, similarPairs)  
        IF the LENGTH OF sentence1 NOT EQUALS the LENGTH OF sentence2  
            RETURN False  
        END IF  

        SET parent TO empty dictionary  

        FUNCTION find(x)  
            IF the element at key x of parent NOT EQUALS x  
                SET the element at key x of parent TO the RESULT OF find of the element at key x of parent  
            END IF  
            RETURN the element at key x of parent  
        END FUNCTION  

        FUNCTION union(x, y)  
            SET rootX TO the RESULT OF find with parameter x  
            SET rootY TO the RESULT OF find with parameter y  
            IF rootX NOT EQUALS rootY  
                SET the element at key rootY of parent TO rootX  
            END IF  
        END FUNCTION  

        FOR each pair IN similarPairs  
            FOR each word IN pair  
                IF word NOT IN parent  
                    SET the element at key word of parent TO word  
                END IF  
            END FOR  
        END FOR  

        FOR each x, y IN similarPairs  
            CALL union with parameters x and y  
        END FOR  

        FOR each w1, w2 IN pairs of elements at corresponding positions from sentence1 and sentence2  
            IF w1 NOT IN parent  
                SET the element at key w1 of parent TO w1  
            END IF  
            IF w2 NOT IN parent  
                SET the element at key w2 of parent TO w2  
            END IF  
            IF the RESULT OF find with parameter w1 NOT EQUALS the RESULT OF find with parameter w2  
                RETURN False  
            END IF  
        END FOR  

        RETURN True  
    END FUNCTION  
END CLASS
