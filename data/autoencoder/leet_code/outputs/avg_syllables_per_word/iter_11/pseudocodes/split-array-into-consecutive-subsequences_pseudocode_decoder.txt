CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO the frequency count of elements in nums  
        SET end_with TO an empty frequency count dictionary  

        FOR each num IN nums  
            IF the frequency of num in num_count EQUALS zero  
                CONTINUE  
            END IF  

            IF the frequency of (num MINUS one) in end_with GREATER THAN zero  
                DECREMENT the frequency of (num MINUS one) in end_with BY one  
                INCREMENT the frequency of num in end_with BY one  
            ELSE IF the frequency of (num PLUS one) in num_count GREATER THAN zero AND the frequency of (num PLUS two) in num_count GREATER THAN zero  
                DECREMENT the frequency of (num PLUS one) in num_count BY one  
                DECREMENT the frequency of (num PLUS two) in num_count BY one  
                INCREMENT the frequency of (num PLUS two) in end_with BY one  
            ELSE  
                RETURN False  
            END IF  

            DECREMENT the frequency of num in num_count BY one  
        END FOR  

        RETURN True  
    END FUNCTION  
END CLASS
