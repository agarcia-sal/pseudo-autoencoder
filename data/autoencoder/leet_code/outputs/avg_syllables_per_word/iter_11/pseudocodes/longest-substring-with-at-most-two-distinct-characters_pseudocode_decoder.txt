CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        INITIALIZE char_count AS auxiliary FUNCTION defaultdict_initialized_to_zero  
        SET left TO zero  
        SET max_length TO zero  
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the count of character at position right of s IN char_count BY one  
            WHILE the NUMBER OF UNIQUE KEYS IN char_count IS GREATER THAN two  
                DECREMENT the count of character at position left of s IN char_count BY one  
                IF the count of character at position left of s IN char_count EQUALS zero  
                    REMOVE this character FROM char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            SET max_length TO the GREATER OF max_length AND right MINUS left PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
