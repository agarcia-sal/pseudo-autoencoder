CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the TUPLE of index and current_sum IS IN memo  
                RETURN the ELEMENT at TUPLE of index and current_sum IN memo  
            END IF  
            SET positive TO CALL dfs WITH index PLUS one, current_sum PLUS the ELEMENT at position index OF nums, memo  
            SET negative TO CALL dfs WITH index PLUS one, current_sum MINUS the ELEMENT at position index OF nums, memo  
            SET the ELEMENT at TUPLE of index and current_sum IN memo TO positive PLUS negative  
            RETURN the ELEMENT at TUPLE of index and current_sum IN memo  
        END FUNCTION  
        RETURN CALL dfs WITH zero, zero, an EMPTY COLLECTION  
    END FUNCTION  
END CLASS
