CLASS Solution  
    FUNCTION scheduleCourse(courses)  
        SORT courses BY the second element of each course (end day) in ascending order  
        INITIALIZE max_heap AS empty list  
        SET current_end_day TO zero  
        
        FOR each course IN courses  
            SET duration TO the first element of course  
            SET last_day TO the second element of course  
            
            IF current_end_day PLUS duration LESS THAN OR EQUAL TO last_day  
                PUSH the negative of duration INTO max_heap using a heap push operation  
                INCREMENT current_end_day BY duration  
            ELSE  
                IF max_heap IS NOT empty AND the negative of the first element of max_heap GREATER THAN duration  
                    INCREMENT current_end_day BY duration PLUS heap pop of max_heap  
                    PUSH the negative of duration INTO max_heap using a heap push operation  
                END IF  
            END IF  
        END FOR  
        
        RETURN the length of max_heap  
    END FUNCTION  
END CLASS
