CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  

        FOR i FROM zero TO n MINUS one MINUS one  
            PUSH a tuple consisting of the element at position i of arr DIVIDED BY the element at position n MINUS one of arr, i, and n MINUS one INTO min_heap  
        END FOR  

        FOR underscore FROM zero TO k MINUS one MINUS one  
            REMOVE the smallest element FROM min_heap AND DESTRUCTURE IT INTO _, i, j  
            IF j MINUS one GREATER THAN i  
                PUSH a tuple consisting of the element at position i of arr DIVIDED BY the element at position j MINUS one of arr, i, and j MINUS one INTO min_heap  
            END IF  
        END FOR  

        REMOVE the smallest element FROM min_heap AND DESTRUCTURE IT INTO _, i, j  
        RETURN a list consisting of the element at position i of arr and the element at position j of arr  
    END FUNCTION  
END CLASS
