CLASS Solution  
    FUNCTION findKthLargest WITH parameters nums, k  
        SET min_heap TO empty list  
        FOR each num IN nums  
            INSERT num INTO min_heap MAINTAINING HEAP PROPERTY  
            IF the LENGTH OF min_heap GREATER THAN k  
                REMOVE the SMALLEST ELEMENT FROM min_heap  
            END IF  
        END FOR  
        RETURN the ELEMENT at position zero OF min_heap  
    END FUNCTION  
END CLASS
