CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO plus sign  
            WHILE the LENGTH OF s GREATER THAN zero  
                SET char TO the first element of s REMOVED FROM s  
                IF char IS a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                IF char EQUALS opening parenthesis  
                    SET num TO the result of CALLING helper WITH s  
                END IF  
                IF char IS IN the list plus sign, minus sign, multiplication sign, division sign, opening parenthesis, closing parenthesis OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS plus sign  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus sign  
                        APPEND the negation of num TO stack  
                    ELSE IF sign EQUALS multiplication sign  
                        SET the last element of stack TO the last element of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS division sign  
                        SET the last element of stack TO the integer truncation towards zero of the last element of stack DIVIDED BY num  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS closing parenthesis  
                    BREAK the WHILE loop  
                END IF  
            END WHILE  
            RETURN the sum of elements in stack  
        END FUNCTION  
        SET s TO the list of characters from s  
        RETURN the result of CALLING helper WITH s  
    END FUNCTION  
END CLASS
