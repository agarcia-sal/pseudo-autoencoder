CLASS Solution  
    FUNCTION minSwap(nums1, nums2)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a new list with n elements each containing a list of two zeros  
        SET the element at position zero and zero in dp TO zero  
        SET the element at position zero and one in dp TO one  
        FOR index i FROM one TO n MINUS one  
            IF the element at position i of nums1 GREATER THAN the element at position i MINUS one of nums1 AND the element at position i of nums2 GREATER THAN the element at position i MINUS one of nums2  
                IF the element at position i of nums1 GREATER THAN the element at position i MINUS one of nums2 AND the element at position i of nums2 GREATER THAN the element at position i MINUS one of nums1  
                    SET the element at position i and zero in dp TO the MINIMUM OF the element at position i MINUS one and zero in dp AND the element at position i MINUS one and one in dp  
                    SET the element at position i and one in dp TO the element at position i and zero in dp PLUS one  
                ELSE  
                    SET the element at position i and zero in dp TO the element at position i MINUS one and zero in dp  
                    SET the element at position i and one in dp TO the element at position i MINUS one and one in dp PLUS one  
                END IF  
            ELSE  
                IF the element at position i of nums1 GREATER THAN the element at position i MINUS one of nums2 AND the element at position i of nums2 GREATER THAN the element at position i MINUS one of nums1  
                    SET the element at position i and zero in dp TO the element at position i MINUS one and one in dp  
                    SET the element at position i and one in dp TO the element at position i MINUS one and zero in dp PLUS one  
                ELSE  
                    SET the element at position i and zero in dp TO the element at position i MINUS one and one in dp  
                    SET the element at position i and one in dp TO the element at position i MINUS one and zero in dp PLUS one  
                END IF  
            END IF  
        END FOR  
        RETURN the MINIMUM OF the element at position n MINUS one and zero in dp AND the element at position n MINUS one and one in dp  
    END FUNCTION  
END CLASS
