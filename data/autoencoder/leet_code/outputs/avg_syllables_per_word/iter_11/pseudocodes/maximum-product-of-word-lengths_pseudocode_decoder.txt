CLASS Solution  
    FUNCTION maxProduct(words)  
        SET word_sets TO an empty list  
        FOR each word IN words  
            APPEND the set of characters in word TO word_sets  
        END FOR  
        
        SET max_product TO zero  
        
        FOR index i FROM zero TO the length of words MINUS one  
            FOR index j FROM i PLUS one TO the length of words MINUS one  
                IF the intersection of element at position i of word_sets AND element at position j of word_sets EQUALS empty set  
                    SET product TO the length of element at position i of words MULTIPLIED BY the length of element at position j of words  
                    IF product GREATER THAN max_product  
                        SET max_product TO product  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_product  
    END FUNCTION  
END CLASS
