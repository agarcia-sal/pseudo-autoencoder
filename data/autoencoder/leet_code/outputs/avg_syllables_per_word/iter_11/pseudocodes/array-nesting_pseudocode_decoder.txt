CLASS Solution  
    FUNCTION arrayNesting(nums)  
        SET visited TO list of False values with the LENGTH OF nums  
        SET max_length TO zero  
        FOR index FROM zero TO LENGTH OF nums MINUS one  
            IF element at position index of visited EQUALS False  
                SET current_length TO zero  
                SET k TO index  
                WHILE element at position k of visited EQUALS False  
                    SET element at position k of visited TO True  
                    SET k TO element at position k of nums  
                    INCREMENT current_length BY one  
                END WHILE  
                IF max_length LESS THAN current_length  
                    SET max_length TO current_length  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
