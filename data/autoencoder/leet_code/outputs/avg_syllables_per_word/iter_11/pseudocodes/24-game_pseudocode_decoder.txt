CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the element at position zero of nums MINUS twenty FOUR LESS THAN one E MULTIPLIED BY MINUS six  
            END IF  
            
            FOR i FROM zero TO the LENGTH OF nums MINUS one  
                FOR j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO empty list  
                        FOR k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND the element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR each op IN the collection of truediv, mul, add, sub  
                            IF (op EQUALS add OR op EQUALS mul) AND j GREATER THAN i  
                                CONTINUE TO next iteration  
                            END IF  
                            
                            IF op NOT EQUALS truediv OR the element at position j of nums NOT EQUALS zero  
                                APPEND the result of op APPLIED TO the elements at positions i and j of nums TO new_nums  
                                IF solve(new_nums)  
                                    RETURN True  
                                END IF  
                                REMOVE last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        IMPORT truediv, mul, add, sub FROM operator  
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
