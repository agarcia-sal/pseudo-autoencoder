CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with parameter element at position zero of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element FROM queue  
        REMOVE first element FROM queue  
        IF i LESS THAN length OF values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode with parameter element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length OF values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode with parameter element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree with parameters p, q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF element val of p NOT EQUALS element val of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters element left of p, element left of q AND is_same_tree with parameters element right of p, element right of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION lowestCommonAncestor with parameters self, root, p, q  
        IF root IS None OR root EQUALS p OR root EQUALS q  
            RETURN root  
        END IF  
        SET left TO call lowestCommonAncestor on self with parameters element left of root, p, q  
        SET right TO call lowestCommonAncestor on self with parameters element right of root, p, q  
        IF left IS NOT None AND right IS NOT None  
            RETURN root  
        END IF  
        IF left IS NOT None  
            RETURN left  
        ELSE  
            RETURN right  
        END IF  
    END FUNCTION  
END CLASS
