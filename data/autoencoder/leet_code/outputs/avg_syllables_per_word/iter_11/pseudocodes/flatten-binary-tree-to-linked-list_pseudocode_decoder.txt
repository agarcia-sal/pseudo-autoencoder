CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self, val EQUALS zero, left EQUALS None, right EQUALS None  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH val AS element at position zero of values  
    SET i TO one  
    CALL new deque AND SET TO queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE first element FROM queue AND SET TO node  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode WITH val AS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode WITH val AS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left, q.left AND is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten WITH PARAMETERS self, root  
        IF root IS None  
            RETURN  
        END IF  
        SET stack TO list containing root  
        WHILE stack IS NOT empty  
            REMOVE last element FROM stack AND SET TO node  
            IF node.right IS NOT None  
                APPEND node.right TO stack  
            END IF  
            IF node.left IS NOT None  
                APPEND node.left TO stack  
            END IF  
            SET node.left TO None  
            IF stack IS NOT empty  
                SET node.right TO last element of stack  
            ELSE  
                SET node.right TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
