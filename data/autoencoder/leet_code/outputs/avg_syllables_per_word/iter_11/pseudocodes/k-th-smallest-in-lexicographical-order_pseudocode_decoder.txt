CLASS Solution  
    FUNCTION findKthNumber(n, k)  
        FUNCTION count_steps(prefix, n)  
            SET next_prefix TO prefix PLUS one  
            SET steps TO zero  
            WHILE prefix LESS THAN OR EQUAL TO n  
                SET difference TO n MINUS prefix PLUS one  
                SET interval TO next_prefix MINUS prefix  
                IF difference LESS THAN interval  
                    INCREMENT steps BY difference  
                ELSE  
                    INCREMENT steps BY interval  
                END IF  
                SET prefix TO prefix MULTIPLIED BY ten  
                SET next_prefix TO next_prefix MULTIPLIED BY ten  
            END WHILE  
            RETURN steps  
        END FUNCTION  

        SET current TO one  
        DECREMENT k BY one  
        WHILE k GREATER THAN zero  
            SET steps TO count_steps(current, n)  
            IF steps LESS THAN OR EQUAL TO k  
                INCREMENT current BY one  
                DECREMENT k BY steps  
            ELSE  
                SET current TO current MULTIPLIED BY ten  
                DECREMENT k BY one  
            END IF  
        END WHILE  
        RETURN current  
    END FUNCTION  
END CLASS
