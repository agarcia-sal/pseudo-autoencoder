CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
          
        CALL construct_2D_list WITH parameters m PLUS one AND n PLUS one  
        SET dp TO the RESULT OF the CALL  
          
        FOR j FROM one TO n  
            SET value_at_position zero and j OF dp TO the VALUE at position zero and j MINUS one OF dp PLUS the ASCII VALUE OF the CHARACTER at position j MINUS one OF s2  
        END FOR  
          
        FOR i FROM one TO m  
            SET value_at_position i and zero OF dp TO the VALUE at position i MINUS one and zero OF dp PLUS the ASCII VALUE OF the CHARACTER at position i MINUS one OF s1  
        END FOR  
          
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the CHARACTER at position i MINUS one OF s1 EQUALS the CHARACTER at position j MINUS one OF s2  
                    SET value_at_position i and j OF dp TO the VALUE at position i MINUS one and j MINUS one OF dp  
                ELSE  
                    SET value_at_position i and j OF dp TO the MINIMUM OF the VALUE at position i MINUS one and j OF dp PLUS the ASCII VALUE OF the CHARACTER at position i MINUS one OF s1 AND the VALUE at position i and j MINUS one OF dp PLUS the ASCII VALUE OF the CHARACTER at position j MINUS one OF s2  
                END IF  
            END FOR  
        END FOR  
          
        RETURN the VALUE at position m and n OF dp  
    END FUNCTION  
  
    FUNCTION construct_2D_list(rows, columns)  
        DESCRIBE creating a two-dimensional list with rows NUMBER OF rows AND columns NUMBER OF columns FILLED WITH zero  
        RETURN the constructed two-dimensional list  
    END FUNCTION  
END CLASS
