CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO a set containing zero AND firstPerson  
        SORT meetings BY the element at position two of each meeting  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO the element at position two of the meeting at position i of meetings  
            CALL FUNCTION defaultdict WITH list TO CREATE graph  
            WHILE i LESS THAN the LENGTH OF meetings AND the element at position two of the meeting at position i of meetings EQUALS current_time  
                SET x TO the element at position zero of the meeting at position i of meetings  
                SET y TO the element at position one of the meeting at position i of meetings  
                APPEND y TO the list at key x IN graph  
                APPEND x TO the list at key y IN graph  
                INCREMENT i BY one  
            END WHILE  
            CALL FUNCTION deque WITH a list of persons IN know_secret THAT ARE IN graph TO CREATE queue  
            SET visited TO an empty set  
            WHILE queue is not empty  
                SET person TO first element REMOVED FROM queue  
                IF person IN visited  
                    CONTINUE to the next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list at key person IN graph  
                    IF neighbor NOT IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN a list CREATED FROM know_secret  
    END FUNCTION  
END CLASS
