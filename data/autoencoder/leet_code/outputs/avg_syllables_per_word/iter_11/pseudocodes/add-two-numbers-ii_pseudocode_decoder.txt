CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with value at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting from position one  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN (value at p1 EQUALS value at p2) AND is_same_list(next of p1, next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1, l2)  
        FUNCTION pushIntoStack(node)  
            SET stack TO empty list  
            WHILE node IS NOT None  
                APPEND value at node TO stack  
                SET node TO next of node  
            END WHILE  
            RETURN stack  
        END FUNCTION  
        
        SET stack1 TO pushIntoStack(l1)  
        SET stack2 TO pushIntoStack(l2)  
        
        SET carry TO zero  
        SET result TO None  
        
        WHILE stack1 IS NOT empty OR stack2 IS NOT empty OR carry NOT EQUALS zero  
            IF stack1 IS NOT empty  
                SET val1 TO last element popped from stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF stack2 IS NOT empty  
                SET val2 TO last element popped from stack2  
            ELSE  
                SET val2 TO zero  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  
            
            SET new_node TO new ListNode with digit  
            SET new_node.next TO result  
            SET result TO new_node  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
