CLASS Solution  
    FUNCTION lastSubstring(s)  
        SET n TO the LENGTH OF s  
        SET max_char TO the MAXIMUM CHARACTER IN s  
        SET max_indices TO an EMPTY LIST  
        FOR index FROM zero TO n MINUS one  
            IF the ELEMENT AT position index OF s EQUALS max_char  
                APPEND index TO max_indices  
            END IF  
        END FOR  
        SET max_substring TO empty string  
        FOR each start IN max_indices  
            SET current_substring TO the SUBSTRING OF s FROM start TO the END  
            IF current_substring GREATER THAN max_substring  
                SET max_substring TO current_substring  
            END IF  
        END FOR  
        RETURN max_substring  
    END FUNCTION  
END CLASS
