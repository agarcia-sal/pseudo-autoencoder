CLASS Solution  
    FUNCTION dieSimulator(n, rollMax)  
        SET MOD TO one billion PLUS seven  
        
        SET dp TO a three-dimensional list with dimensions n PLUS one, six, and maximum value in rollMax PLUS one, filled with zero  
        
        FOR j FROM zero TO five  
            SET element at position one of dp at position one, j, one TO one  
        END FOR  
        
        FOR i FROM two TO n  
            FOR j FROM zero TO five  
                FOR k FROM one TO element at position j of rollMax  
                    IF k GREATER THAN one  
                        SET element at position i, j, k of dp TO element at position i MINUS one, j, k MINUS one of dp  
                    ELSE  
                        SET element at position i, j, k of dp TO zero  
                        FOR x FROM zero TO five  
                            IF x NOT EQUALS j  
                                FOR y FROM one TO element at position x of rollMax  
                                    INCREMENT element at position i, j, k of dp BY element at position i MINUS one, x, y of dp  
                                END FOR  
                            END IF  
                        END FOR  
                        SET element at position i, j, k of dp TO element at position i, j, k of dp MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR j FROM zero TO five  
            FOR k FROM one TO element at position j of rollMax  
                INCREMENT result BY element at position n, j, k of dp  
            END FOR  
        END FOR  
        
        RETURN result MODULO MOD  
    END FUNCTION  
END CLASS
