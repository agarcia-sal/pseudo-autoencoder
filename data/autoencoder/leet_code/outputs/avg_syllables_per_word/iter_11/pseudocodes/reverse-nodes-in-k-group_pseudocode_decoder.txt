CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO new ListNode with value the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values STARTING FROM position one TO the last element  
        SET node TO new ListNode with value val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN (the element val of p1 EQUALS the element val of p2) AND is_same_list(the element next of p1, the element next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseKGroup(head, k)  
        FUNCTION reverseLinkedList(start, end)  
            SET prev TO None  
            SET curr TO start  
            WHILE curr NOT EQUALS end  
                SET next_node TO the element next of curr  
                SET the element next of curr TO prev  
                SET prev TO curr  
                SET curr TO next_node  
            END WHILE  
            SET the element next of curr TO prev  
            RETURN end, start  
        END FUNCTION  

        SET dummy TO new ListNode with value zero  
        SET the element next of dummy TO head  
        SET prev_group_end TO dummy  

        WHILE True  
            SET kth_node TO prev_group_end  
            FOR count FROM one TO k  
                SET kth_node TO the element next of kth_node  
                IF kth_node EQUALS None  
                    RETURN the element next of dummy  
                END IF  
            END FOR  
            SET next_group_start TO the element next of kth_node  
            SET the element next of kth_node TO None  
            SET new_group_start, new_group_end TO reverseLinkedList(the element next of prev_group_end, kth_node)  
            SET the element next of prev_group_end TO new_group_start  
            SET the element next of new_group_end TO next_group_start  
            SET prev_group_end TO new_group_end  
        END WHILE  

        RETURN the element next of dummy  
    END FUNCTION  
END CLASS
