CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH val as element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode WITH val as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode WITH val as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val DOES NOT EQUAL q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left AND q.left AND is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION findFrequentTreeSum WITH PARAMETERS root  
        FUNCTION subtree_sum WITH PARAMETER node  
            IF node IS None  
                RETURN zero  
            END IF  
            SET current_sum TO node.val PLUS subtree_sum WITH PARAMETER node.left PLUS subtree_sum WITH PARAMETER node.right  
            INCREMENT count AT current_sum BY one  
            RETURN current_sum  
        END FUNCTION  
        SET count TO a new default dictionary with integer default  
        CALL subtree_sum WITH PARAMETER root  
        SET max_freq TO the maximum VALUE IN count VALUES OR zero IF count IS empty  
        SET result TO an empty list  
        FOR each key, freq IN count ITEMS  
            IF freq EQUALS max_freq  
                APPEND key TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
