CLASS Solution
    FUNCTION minimumDistance(word)
        FUNCTION distance(c1, c2)
            IF c1 EQUALS None
                RETURN zero
            END IF
            SET x1 TO the result of dividing the ordinal value of c1 MINUS the ordinal value of letter A BY six WITH quotient part
            SET y1 TO the result of dividing the ordinal value of c1 MINUS the ordinal value of letter A BY six WITH remainder part
            SET x2 TO the result of dividing the ordinal value of c2 MINUS the ordinal value of letter A BY six WITH quotient part
            SET y2 TO the result of dividing the ordinal value of c2 MINUS the ordinal value of letter A BY six WITH remainder part
            RETURN the absolute value of x1 MINUS x2 PLUS the absolute value of y1 MINUS y2
        END FUNCTION

        FUNCTION dp(i, f1, f2)
            IF i EQUALS the length of word
                RETURN zero
            END IF
            SET c TO the element at position i of word
            SET d1 TO the sum of distance from f1 to c PLUS dp of i plus one, c, f2
            SET d2 TO the sum of distance from f2 to c PLUS dp of i plus one, f1, c
            RETURN the lesser of d1 and d2
        END FUNCTION

        RETURN dp of zero, None, None
    END FUNCTION
END CLASS
