CLASS Solution  
    FUNCTION digitsCount(d, low, high)  
        FUNCTION count_digit(n, digit)  
            SET count TO zero  
            SET power_of_10 TO one  
            SET current TO n  
            WHILE current GREATER THAN zero  
                SET last TO the remainder of current DIVIDED BY ten  
                SET current TO the quotient of current DIVIDED BY ten  
                SET full_cycles TO current  
                IF digit EQUALS zero  
                    DECREMENT full_cycles BY one  
                END IF  
                INCREMENT count BY full_cycles MULTIPLIED BY power_of_10  
                IF last GREATER THAN digit  
                    INCREMENT count BY power_of_10  
                ELSE IF last EQUALS digit  
                    INCREMENT count BY the remainder of n DIVIDED BY power_of_10 PLUS one  
                END IF  
                SET power_of_10 TO power_of_10 MULTIPLIED BY ten  
            END WHILE  
            RETURN count  
        END FUNCTION  
        RETURN the result of count_digit called with high and d MINUS the result of count_digit called with low MINUS one and d  
    END FUNCTION  
END CLASS
