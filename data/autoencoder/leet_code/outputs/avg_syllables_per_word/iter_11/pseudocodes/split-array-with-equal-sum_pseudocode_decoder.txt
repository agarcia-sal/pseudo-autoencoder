CLASS Solution  
    FUNCTION splitArray(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN 7  
            RETURN False  
        END IF  
        
        SET prefix_sum TO a list of size n with all elements zero  
        SET element at position zero of prefix_sum TO element at position zero of nums  
        FOR i FROM 1 TO n MINUS 1  
            SET element at position i of prefix_sum TO element at position i MINUS 1 of prefix_sum PLUS element at position i of nums  
        END FOR  
        
        FOR j FROM 3 TO n MINUS 3  
            SET seen TO an empty set  
            FOR i FROM 1 TO j MINUS 1  
                IF element at position i MINUS 1 of prefix_sum EQUALS element at position j MINUS 1 of prefix_sum MINUS element at position i of prefix_sum  
                    ADD element at position i MINUS 1 of prefix_sum TO seen  
                END IF  
            END FOR  
            
            FOR k FROM j PLUS 2 TO n MINUS 1  
                IF element at position n MINUS 1 of prefix_sum MINUS element at position k of prefix_sum EQUALS element at position k MINUS 1 of prefix_sum MINUS element at position j of prefix_sum AND element at position k MINUS 1 of prefix_sum MINUS element at position j of prefix_sum IS IN seen  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
