CLASS Solution  
    FUNCTION combinationSum(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO LENGTH OF candidates MINUS one  
                CALL backtrack WITH i, target MINUS element at position i of candidates, path CONCATENATED WITH element at position i of candidates  
            END FOR  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH zero, target, empty list  
        RETURN result  
    END FUNCTION  
END CLASS
