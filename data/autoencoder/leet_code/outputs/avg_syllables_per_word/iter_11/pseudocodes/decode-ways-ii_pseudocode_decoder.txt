CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a list of zeros with LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  

        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS star THEN  
                RETURN nine  
            ELSE IF char EQUALS zero THEN  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  

        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS star AND char2 EQUALS star THEN  
                RETURN fifteen  
            ELSE IF char1 EQUALS star THEN  
                IF char2 LESS THAN OR EQUAL TO six THEN  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS star THEN  
                IF char1 EQUALS one THEN  
                    RETURN nine  
                ELSE IF char1 EQUALS two THEN  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer value formed by concatenating char1 and char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six THEN  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  

        FOR i FROM one TO LENGTH OF s  
            SET element at position i of dp TO element at position i of dp PLUS element at position i MINUS one of dp MULTIPLIED BY ways_to_decode_single(element at position i MINUS one of s) MODULO MOD  
            IF i GREATER THAN one THEN  
                SET element at position i of dp TO element at position i of dp PLUS element at position i MINUS two of dp MULTIPLIED BY ways_to_decode_pair(element at position i MINUS two of s, element at position i MINUS one of s) MODULO MOD  
            END IF  
        END FOR  

        RETURN element at last position of dp  
    END FUNCTION  
END CLASS
