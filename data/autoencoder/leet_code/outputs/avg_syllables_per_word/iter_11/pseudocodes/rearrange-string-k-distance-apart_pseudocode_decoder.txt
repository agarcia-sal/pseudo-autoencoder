CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO the count of each character in s  
        
        SET max_heap TO an empty list  
        FOR each character and count IN freq  
            APPEND the pair of negative count and character TO max_heap  
        END FOR  
        
        PERFORM heapify operation on max_heap  
        
        SET wait_queue TO an empty list  
        
        SET result TO an empty list  
        
        WHILE max_heap is not empty  
            EXTRACT the smallest element FROM max_heap AS count and char  
            APPEND char TO result  
            
            APPEND the pair of count plus one and char TO wait_queue  
            
            IF the length of wait_queue EQUALS k  
                REMOVE first element FROM wait_queue AS count and char  
                IF count LESS THAN zero  
                    PUSH the pair of count and char INTO max_heap  
                END IF  
            END IF  
        END WHILE  
        
        IF the length of result NOT EQUALS the length of s  
            RETURN empty string  
        END IF  
        
        JOIN all characters in result INTO a single string  
        RETURN the joined string  
    END FUNCTION  
END CLASS
