CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF element at position zero of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF element at position zero of mat2  
        CALL initialize_result_matrix WITH parameters m, n RETURNING result  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                FOR index p FROM zero TO k MINUS one  
                    SET element at position j of element at position i of result TO the ELEMENT at position j of element at position i of result PLUS the ELEMENT at position p of element at position i of mat1 MULTIPLIED BY the ELEMENT at position j of element at position p of mat2  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  

    FUNCTION initialize_result_matrix(m, n)  
        SET result TO empty list  
        FOR index i FROM zero TO m MINUS one  
            SET row TO empty list  
            FOR index j FROM zero TO n MINUS one  
                APPEND zero TO row  
            END FOR  
            APPEND row TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
