CLASS Solution  
    FUNCTION convert(s, numRows)  
        IF numRows EQUALS one  
            RETURN s  
        END IF  
  
        SET rows TO a list of empty strings with length numRows  
        SET current_row TO zero  
        SET going_down TO false  
  
        FOR each char IN s  
            SET element at position current_row of rows TO element at position current_row of rows PLUS char  
            IF current_row EQUALS zero OR current_row EQUALS numRows MINUS one  
                SET going_down TO NOT going_down  
            END IF  
            IF going_down  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
  
        SET result TO the concatenation of all elements in rows in order  
        RETURN result  
    END FUNCTION  
END CLASS
