CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO a list of zero repeated batchSize times  
        FOR each group IN groups  
            INCREMENT the element at position (group MODULO batchSize) OF remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder, variable number of remainder_count_elements)  
            SET remainder_count TO a list constructed from remainder_count_elements  
            SET max_happy TO zero  
            
            FOR index FROM zero TO batchSize MINUS one  
                IF the element at position index OF remainder_count EQUALS zero  
                    CONTINUE to next iteration  
                END IF  
                
                DECREMENT the element at position index OF remainder_count BY one  
                SET new_remainder TO (remainder MINUS index) MODULO batchSize  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS dp(new_remainder, all elements of remainder_count unpacked)  
                ELSE  
                    SET happy_groups TO dp(new_remainder, all elements of remainder_count unpacked)  
                END IF  
                
                SET max_happy TO the maximum of max_happy AND happy_groups  
                INCREMENT the element at position index OF remainder_count BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN dp(zero, all elements of remainder_count unpacked)  
END CLASS
