CLASS Solution  
    FUNCTION findMedianSortedArrays(nums1, nums2)  
        IF the LENGTH OF nums1 GREATER THAN the LENGTH OF nums2  
            SET temporary variable TO nums1  
            SET nums1 TO nums2  
            SET nums2 TO temporary variable  
        END IF  
          
        SET x TO the LENGTH OF nums1  
        SET y TO the LENGTH OF nums2  
        SET low TO zero  
        SET high TO x  
          
        WHILE low LESS THAN OR EQUAL TO high  
            SET partitionX TO the INTEGER DIVISION of (low PLUS high) BY two  
            SET partitionY TO the INTEGER DIVISION of (x PLUS y PLUS one) BY two MINUS partitionX  
              
            IF partitionX EQUALS zero  
                SET maxLeftX TO negative infinity  
            ELSE  
                SET maxLeftX TO element at position (partitionX MINUS one) of nums1  
            END IF  
              
            IF partitionX EQUALS x  
                SET minRightX TO positive infinity  
            ELSE  
                SET minRightX TO element at position partitionX of nums1  
            END IF  
              
            IF partitionY EQUALS zero  
                SET maxLeftY TO negative infinity  
            ELSE  
                SET maxLeftY TO element at position (partitionY MINUS one) of nums2  
            END IF  
              
            IF partitionY EQUALS y  
                SET minRightY TO positive infinity  
            ELSE  
                SET minRightY TO element at position partitionY of nums2  
            END IF  
              
            IF maxLeftX LESS THAN OR EQUAL TO minRightY AND maxLeftY LESS THAN OR EQUAL TO minRightX  
                IF (x PLUS y) MODULO two EQUALS zero  
                    SET max_of_left TO the GREATER OF maxLeftX AND maxLeftY  
                    SET min_of_right TO the LESSER OF minRightX AND minRightY  
                    RETURN the SUM of max_of_left AND min_of_right DIVIDED BY two  
                ELSE  
                    RETURN the GREATER OF maxLeftX AND maxLeftY  
                END IF  
            ELSE IF maxLeftX GREATER THAN minRightY  
                SET high TO partitionX MINUS one  
            ELSE  
                SET low TO partitionX PLUS one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
