CLASS Solution  
    FUNCTION numRabbits(answers)  
        SET answer_counts TO result of calling Counter with answers  
        SET total_rabbits TO zero  
        FOR each answer and count IN answer_counts items  
            SET groups TO the result of (count PLUS answer) DIVIDED BY (answer PLUS one) using integer division  
            INCREMENT total_rabbits BY groups MULTIPLIED BY (answer PLUS one)  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
