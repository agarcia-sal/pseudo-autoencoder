CLASS Solution  
    FUNCTION isScramble(s1, s2)  
        IF s1 EQUALS s2  
            RETURN True  
        END IF  
        
        IF the sorted characters of s1 NOT EQUALS the sorted characters of s2  
            RETURN False  
        END IF  
        
        SET n TO the LENGTH OF s1  
        FOR i FROM one TO n MINUS one  
            IF self dot isScramble with parameters the substring of s1 from start TO i AND the substring of s2 from start TO i AND self dot isScramble with parameters the substring of s1 from i TO end AND the substring of s2 from i TO end  
                RETURN True  
            END IF  
            
            IF self dot isScramble with parameters the substring of s1 from start TO i AND the substring of s2 from the negative i TO end AND self dot isScramble with parameters the substring of s1 from i TO end AND the substring of s2 from start TO the negative i  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
