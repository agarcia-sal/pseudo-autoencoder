CLASS Solution  
    FUNCTION numMatchingSubseq(s, words)  
        SET waiting TO an empty dictionary mapping each unique character in s TO an empty deque  
        FOR each character c IN the set of characters in s  
            SET the value at key c in waiting TO an empty deque  
        END FOR  
        
        FOR each word IN words  
            SET it TO an iterator over word  
            SET first_letter TO the next element of it  
            IF first_letter IS IN waiting  
                APPEND it TO the deque at key first_letter in waiting  
            END IF  
        END FOR  
        
        SET match_count TO zero  
        
        FOR each character c IN s  
            IF c IS IN waiting  
                SET current_iterators TO the deque at key c in waiting  
                SET waiting at key c TO a new empty deque  
                FOR each it IN current_iterators  
                    SET next_char TO the next element of it OR None if no next element exists  
                    IF next_char IS None  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char IS IN waiting  
                            APPEND it TO the deque at key next_char in waiting  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN match_count  
    END FUNCTION  
END CLASS
