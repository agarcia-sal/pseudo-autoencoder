CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a new defaultdict of lists  
        FOR each pair of frm and to IN the reversed order of tickets sorted in ascending order  
            APPEND to TO the list at key frm in graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit(airport)  
            WHILE the list at key airport in graph is not empty  
                CALL visit with the result of removing last element FROM the list at key airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit with the string "JFK"  
        RETURN the reverse of itinerary  
    END FUNCTION  
END CLASS
