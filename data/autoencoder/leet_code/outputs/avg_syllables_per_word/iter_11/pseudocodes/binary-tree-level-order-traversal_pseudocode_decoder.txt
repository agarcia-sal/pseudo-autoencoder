CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element removed from queue  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the val attribute of p IS NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree applied to the left attribute of p AND the left attribute of q  
            AND the result of is_same_tree applied to the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root IS none  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a new double-ended queue initialized with root as its only element  
        WHILE queue IS NOT empty  
            SET level_size TO the length of queue  
            SET current_level TO an empty list  
            FOR _ FROM zero TO level_size MINUS one  
                SET node TO the first element removed from queue  
                APPEND the val attribute of node TO current_level  
                IF the left attribute of node IS NOT none  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node IS NOT none  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
