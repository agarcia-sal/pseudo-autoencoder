CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            RETURN (element at position zero of a MINUS element at position zero of o) MULTIPLIED BY (element at position one of b MINUS element at position one of o) MINUS (element at position one of a MINUS element at position one of o) MULTIPLIED BY (element at position zero of b MINUS element at position zero of o)  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  
        
        SET orientation TO zero  
        
        FOR index FROM zero TO n MINUS one  
            SET o TO element at position index of points  
            SET a TO element at position (index PLUS one) MODULO n of points  
            SET b TO element at position (index PLUS two) MODULO n of points  
            SET cross TO cross_product(o, a, b)  
            
            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS negative one AND cross GREATER THAN zero  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
