CLASS Solution  
    FUNCTION canDivideIntoSubsequences(nums, k)  
        SET freq TO result of calling Counter on nums  
        SET max_freq TO the maximum value in freq values  
        IF the length of nums DIVIDED BY k IS GREATER THAN OR EQUAL TO max_freq  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
