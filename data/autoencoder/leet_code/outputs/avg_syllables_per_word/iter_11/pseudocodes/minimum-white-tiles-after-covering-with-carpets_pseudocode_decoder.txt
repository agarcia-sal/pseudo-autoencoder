CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO the LENGTH OF floor  
        SET dp TO the result of calling initialize_dp WITH parameters n PLUS one AND numCarpets PLUS one  
        
        FOR i FROM one TO n  
            SET dp at position i at zero TO dp at position i minus one at zero PLUS the VALUE OF the CONDITION that the element at position i minus one of floor EQUALS the character one  
        END FOR  
        
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp at position i at j TO dp at position i minus one at j PLUS the VALUE OF the CONDITION that the element at position i minus one of floor EQUALS the character one  
                SET dp at position i at j TO the MINIMUM OF dp at position i at j AND dp at position the MAXIMUM OF zero AND i MINUS carpetLen at position j minus one  
            END FOR  
        END FOR  
        
        RETURN dp at position n at numCarpets  
    END FUNCTION  
    
    FUNCTION initialize_dp(rows, cols)  
        CREATE a list named result  
        FOR each index FROM zero TO rows MINUS one  
            CREATE a list named row with cols elements ALL SET TO zero  
            APPEND row TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
