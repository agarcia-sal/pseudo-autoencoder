CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        FOR each num IN the elements of nums in reversed order  
            IF num LESS THAN second  
                RETURN True  
            END IF  
            WHILE the LENGTH OF stack GREATER THAN zero AND element at position last of stack LESS THAN num  
                SET second TO the removal of last element FROM stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
