CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO list of characters of string representation of num  
        SET last TO empty map  
        FOR each index i AND element d IN digits  
            SET integer value of d AT key i IN last  
        END FOR  
        FOR each index i AND element d IN digits  
            FOR d2 FROM nine DOWN TO integer value of d PLUS one  
                IF value at key d2 IN last EXISTS AND value at key d2 IN last GREATER THAN i  
                    SET temporary TO element at position i OF digits  
                    SET element at position i OF digits TO element at position value at key d2 IN last OF digits  
                    SET element at position value at key d2 IN last OF digits TO temporary  
                    RETURN integer value of concatenation of elements of digits  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
