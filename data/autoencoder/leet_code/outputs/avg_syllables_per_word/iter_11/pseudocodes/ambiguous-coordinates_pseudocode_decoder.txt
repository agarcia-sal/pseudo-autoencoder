CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO the SUBSTRING of frag FROM start TO position d minus one  
                SET right TO the SUBSTRING of frag FROM position d TO the end  
                IF left DOES NOT START WITH zero OR left EQUALS zero THEN  
                    IF right DOES NOT END WITH zero THEN  
                        YIELD left CONCATENATED WITH a period IF d IS NOT EQUAL TO n ELSE empty string CONCATENATED WITH right  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
        SET s TO the SUBSTRING of s FROM position one TO the position BEFORE the last character  
        RETURN list of strings constructed as open parenthesis CONCATENATED WITH a CONCATENATED WITH comma and space CONCATENATED WITH b CONCATENATED WITH close parenthesis  
        FOR i FROM one TO the LENGTH OF s MINUS one  
        FOR each a IN make called with the SUBSTRING of s FROM start TO position i minus one  
        FOR each b IN make called with the SUBSTRING of s FROM position i TO the end  
        END FOR  
        END FOR  
        END FOR  
    END FUNCTION  
END CLASS
