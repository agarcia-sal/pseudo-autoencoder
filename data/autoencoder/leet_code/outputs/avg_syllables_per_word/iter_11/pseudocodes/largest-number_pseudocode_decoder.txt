CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO list of string representations of elements in nums  
        
        FUNCTION compare(x, y)  
            IF concatenation of x and y GREATER THAN concatenation of y and x  
                RETURN negative one  
            ELSE IF concatenation of x and y LESS THAN concatenation of y and x  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str USING custom comparison defined by compare  
        
        IF element at position zero of nums_str EQUALS string zero  
            RETURN string zero  
        END IF  
        
        SET result TO concatenation of all elements in nums_str  
        RETURN result  
    END FUNCTION  
END CLASS
