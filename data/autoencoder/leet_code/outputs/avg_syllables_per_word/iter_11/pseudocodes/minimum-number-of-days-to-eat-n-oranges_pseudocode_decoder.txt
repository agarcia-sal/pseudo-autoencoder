CLASS Solution  
    FUNCTION minDays(n)  
        FUNCTION dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            SET option_one TO remaining MODULO two PLUS dp(remaining DIVIDED BY two)  
            SET option_two TO remaining MODULO three PLUS dp(remaining DIVIDED BY three)  
            RETURN one PLUS the MINIMUM OF option_one AND option_two  
        END FUNCTION  
        RETURN dp(n)  
END CLASS
