CLASS Solution  
    FUNCTION minWindow(s, t)  
        FROM collections IMPORT Counter  
  
        SET t_count TO Counter OF t  
        SET required TO LENGTH OF t_count  
  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO Counter OF NO ELEMENTS  
  
        SET min_length TO infinity  
        SET min_start TO zero  
  
        WHILE right LESS THAN LENGTH OF s  
            INCREMENT window_counts OF element at position right of s BY one  
  
            IF element at position right of s IN t_count AND window_counts OF element at position right of s EQUALS t_count OF element at position right of s  
                INCREMENT formed BY one  
            END IF  
  
            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO element at position left of s  
  
                IF right MINUS left PLUS one LESS THAN min_length  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
  
                DECREMENT window_counts OF char BY one  
                IF char IN t_count AND window_counts OF char LESS THAN t_count OF char  
                    DECREMENT formed BY one  
                END IF  
  
                INCREMENT left BY one  
            END WHILE  
  
            INCREMENT right BY one  
        END WHILE  
  
        IF min_length EQUALS infinity  
            RETURN empty string  
        ELSE  
            RETURN substring of s starting at min_start and of length min_length  
        END IF  
    END FUNCTION  
END CLASS
