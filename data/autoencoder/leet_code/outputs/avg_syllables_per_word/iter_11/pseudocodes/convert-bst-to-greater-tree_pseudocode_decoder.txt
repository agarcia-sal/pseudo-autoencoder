CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS nothing  
        RETURN nothing  
    END IF  
    SET root TO new TreeNode with parameter element at position zero of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing  
            SET node.left TO new TreeNode with parameter element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing  
            SET node.right TO new TreeNode with parameter element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN true  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree with parameters p.left, q.left AND is_same_tree with parameters p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION convertBST with parameter root  
        SET self.sum TO zero  
        FUNCTION traverse with parameter node  
            IF node EQUALS nothing  
                RETURN  
            END IF  
            CALL traverse with parameter node.right  
            INCREMENT self.sum BY node.val  
            SET node.val TO self.sum  
            CALL traverse with parameter node.left  
        END FUNCTION  
        CALL traverse with parameter root  
        RETURN root  
    END FUNCTION  
END CLASS
