CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO the result of splitting num by the character PLUS  
            SET real TO the integer value of the element at position zero of parts  
            SET imaginary TO the integer value of the element at position one of parts with the last character REMOVED  
            RETURN real, imaginary  
        END FUNCTION  
        SET real1, imag1 TO the result of calling parse_complex with num1  
        SET real2, imag2 TO the result of calling parse_complex with num2  
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        RETURN the concatenation of real_part converted to string PLUS the character PLUS PLUS imaginary_part converted to string PLUS the character i  
    END FUNCTION  
END CLASS
