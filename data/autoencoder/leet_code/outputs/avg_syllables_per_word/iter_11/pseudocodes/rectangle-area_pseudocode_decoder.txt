CLASS Solution  
    FUNCTION computeArea(ax1, ay1, ax2, ay2, bx1, by1, bx2, by2)  
        SET area1 TO ax2 MINUS ax1 MULTIPLIED BY ay2 MINUS ay1  
        SET area2 TO bx2 MINUS bx1 MULTIPLIED BY by2 MINUS by1  
        IF ax2 LESS THAN bx2 THEN  
            SET min_right TO ax2  
        ELSE  
            SET min_right TO bx2  
        END IF  
        IF ax1 GREATER THAN bx1 THEN  
            SET max_left TO ax1  
        ELSE  
            SET max_left TO bx1  
        END IF  
        SET overlap_width_candidate TO min_right MINUS max_left  
        IF overlap_width_candidate GREATER THAN zero THEN  
            SET overlap_width TO overlap_width_candidate  
        ELSE  
            SET overlap_width TO zero  
        END IF  
        IF ay2 LESS THAN by2 THEN  
            SET min_top TO ay2  
        ELSE  
            SET min_top TO by2  
        END IF  
        IF ay1 GREATER THAN by1 THEN  
            SET max_bottom TO ay1  
        ELSE  
            SET max_bottom TO by1  
        END IF  
        SET overlap_height_candidate TO min_top MINUS max_bottom  
        IF overlap_height_candidate GREATER THAN zero THEN  
            SET overlap_height TO overlap_height_candidate  
        ELSE  
            SET overlap_height TO zero  
        END IF  
        SET overlap_area TO overlap_width MULTIPLIED BY overlap_height  
        SET total_area TO area1 PLUS area2 MINUS overlap_area  
        RETURN total_area  
    END FUNCTION  
END CLASS
