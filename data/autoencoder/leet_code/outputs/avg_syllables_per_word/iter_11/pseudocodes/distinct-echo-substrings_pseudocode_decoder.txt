CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO the LENGTH OF text  
        SET substrings TO an empty set  
        
        FOR length FROM one TO n DIVIDED BY two  
            FOR start FROM zero TO n MINUS two MULTIPLIED BY length  
                SET a TO the substring of text from position start TO start PLUS length MINUS one  
                SET b TO the substring of text from position start PLUS length TO start PLUS two MULTIPLIED BY length MINUS one  
                IF a EQUALS b  
                    ADD a PLUS b TO substrings  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the SIZE OF substrings  
    END FUNCTION  
END CLASS
