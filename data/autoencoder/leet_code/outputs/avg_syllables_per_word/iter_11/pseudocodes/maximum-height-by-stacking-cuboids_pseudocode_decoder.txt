CLASS Solution  
    FUNCTION maxHeight(cuboids)  
        FOR each cuboid IN cuboids  
            SORT dimensions of cuboid in ascending order  
        END FOR  
        
        SORT cuboids lexicographically by their dimensions  
        
        SET n TO the LENGTH OF cuboids  
        INITIALIZE dp as a list of zeroes with LENGTH n  
        
        FOR i FROM zero TO n MINUS one  
            SET dp at position i TO the height dimension at position two of cuboid at position i  
            
            FOR j FROM zero TO i MINUS one  
                IF dimension at position zero of cuboid at position j LESS THAN OR EQUAL TO dimension at position zero of cuboid at position i  
                    AND dimension at position one of cuboid at position j LESS THAN OR EQUAL TO dimension at position one of cuboid at position i  
                    AND dimension at position two of cuboid at position j LESS THAN OR EQUAL TO dimension at position two of cuboid at position i  
                    SET dp at position i TO the GREATER OF dp at position i AND dp at position j PLUS height dimension of cuboid at position i  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the MAXIMUM VALUE IN dp  
    END FUNCTION  
END CLASS
