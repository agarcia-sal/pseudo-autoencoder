CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value as the first element of values  
    SET i TO one  
    INITIALIZE queue as an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE first element FROM queue AND SET NODE TO it  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET left property of NODE TO a new TreeNode with value element at position i of values  
            APPEND the left property of NODE TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET right property of NODE TO a new TreeNode with value element at position i of values  
            APPEND the right property of NODE TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value property of p IS NOT EQUAL TO the value property of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters left property of p and left property of q AND is_same_tree with parameters right property of p and right property of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION subtreeWithAllDeepest WITH PARAMETERS root  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS None  
                RETURN node, zero  
            END IF  
            SET left_node, left_depth TO the result of dfs with parameter left property of node  
            SET right_node, right_depth TO the result of dfs with parameter right property of node  
            IF left_depth IS GREATER THAN right_depth  
                RETURN left_node, left_depth PLUS one  
            ELSE IF right_depth IS GREATER THAN left_depth  
                RETURN right_node, right_depth PLUS one  
            ELSE  
                RETURN node, left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN the first element of the result of dfs with parameter root  
    END FUNCTION  
END CLASS
