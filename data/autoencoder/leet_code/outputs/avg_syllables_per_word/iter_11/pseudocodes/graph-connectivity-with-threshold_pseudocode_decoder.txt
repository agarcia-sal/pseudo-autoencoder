CLASS Solution  
    FUNCTION areConnected(n, threshold, queries)  
        IF threshold EQUALS zero  
            RETURN list of True repeated LENGTH OF queries times  
        END IF  
  
        SET parent TO list of integers from zero TO n inclusive  
  
        FUNCTION find(x)  
            IF element at position x of parent NOT EQUALS x  
                SET element at position x of parent TO find(element at position x of parent)  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
  
        FUNCTION union(x, y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX NOT EQUALS rootY  
                SET element at position rootY of parent TO rootX  
            END IF  
        END FUNCTION  
  
        FOR i FROM threshold PLUS one TO n  
            FOR j FROM two TIMES i TO n STEP i  
                union(i, j)  
            END FOR  
        END FOR  
  
        SET result TO empty list  
        FOR each pair consisting of a and b IN queries  
            APPEND condition that find(a) EQUALS find(b) TO result  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
