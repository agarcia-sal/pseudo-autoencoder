CLASS Solution  
    FUNCTION minimumOneBitOperations(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE one SHIFTED LEFT BY (k PLUS one) LESS THAN OR EQUAL TO n  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO one SHIFTED LEFT BY k  
        RETURN one SHIFTED LEFT BY (k PLUS one) MINUS one MINUS THE RESULT OF minimumOneBitOperations OF (n XOR most_significant_bit)  
    END FUNCTION  
END CLASS
