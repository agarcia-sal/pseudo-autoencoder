CLASS Solution  
    FUNCTION amountPainted(paint)  
        SET max_end TO the maximum value of end FOR each pair of start and end IN paint  
        SET painted TO a list containing negative one REPEATED max_end PLUS one TIMES  
        SET worklog TO an empty list  
        
        FOR each start and end IN paint  
            SET current_start TO start  
            SET new_paint TO zero  
            
            WHILE current_start LESS THAN end  
                IF the element at position current_start IN painted EQUALS negative one  
                    SET the element at position current_start IN painted TO end  
                    INCREMENT new_paint BY one  
                    INCREMENT current_start BY one  
                ELSE  
                    SET current_start TO the element at position current_start IN painted  
                END IF  
            END WHILE  
            
            APPEND new_paint TO worklog  
        END FOR  
        
        RETURN worklog  
    END FUNCTION  
END CLASS
