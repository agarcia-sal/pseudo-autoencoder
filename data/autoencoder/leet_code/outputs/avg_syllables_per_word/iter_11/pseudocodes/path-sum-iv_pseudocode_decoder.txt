CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO empty dictionary  
        
        FOR each num IN nums  
            SET depth TO num DIVIDED BY one hundred  
            SET pos TO the INTEGER RESULT OF num DIVIDED BY ten MODULO ten  
            SET value TO num MODULO ten  
            SET element at key (depth, pos) of tree TO value  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the value at key (depth, pos) of tree OR none if not present  
            IF current_value EQUALS none  
                RETURN zero  
            END IF  
            
            SET new_path_sum TO path_sum PLUS current_value  
            
            SET left_child TO pair of depth PLUS one AND pos MULTIPLIED BY two MINUS one  
            SET right_child TO pair of depth PLUS one AND pos MULTIPLIED BY two  
            
            IF the key left_child NOT PRESENT IN tree AND the key right_child NOT PRESENT IN tree  
                RETURN new_path_sum  
            END IF  
            
            SET left_sum TO dfs(element at position zero of left_child, element at position one of left_child, new_path_sum)  
            SET right_sum TO dfs(element at position zero of right_child, element at position one of right_child, new_path_sum)  
            
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN dfs(one, one, zero)  
    END FUNCTION  
END CLASS
