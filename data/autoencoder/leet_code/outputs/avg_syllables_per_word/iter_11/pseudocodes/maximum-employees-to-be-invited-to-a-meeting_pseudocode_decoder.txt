CLASS Solution  
    FUNCTION maximumInvitations(favorite)  
        SET n TO the LENGTH OF favorite  
        SET graph TO a new default dictionary of list  
        SET in_degree TO a list of zero values with LENGTH n  
        
        FOR index FROM zero TO n MINUS one  
            APPEND index TO the list at key favorite element at position index of favorite in graph  
            INCREMENT element at position favorite element at position index of favorite IN in_degree BY one  
        END FOR  
        
        SET mutual_chains TO zero  
        SET visited TO a list of False values with LENGTH n  
        
        FOR index FROM zero TO n MINUS one  
            IF element at position favorite element at position index of favorite EQUALS index AND index LESS THAN favorite element at position index of favorite THEN  
                SET a TO index  
                SET b TO favorite element at position index of favorite  
                SET chain_a TO CALL find_chain_length with parameters a, graph, visited  
                SET chain_b TO CALL find_chain_length with parameters b, graph, visited  
                INCREMENT mutual_chains BY chain_a PLUS chain_b  
            END IF  
        END FOR  
        
        SET longest_cycle TO zero  
        SET visited TO a list of False values with LENGTH n  
        
        FOR index FROM zero TO n MINUS one  
            IF element at position index of visited EQUALS False THEN  
                SET cycle_length TO CALL find_cycle_length with parameters index, graph, visited, favorite  
                SET longest_cycle TO the GREATER OF longest_cycle AND cycle_length  
            END IF  
        END FOR  
        
        RETURN the GREATER OF mutual_chains AND longest_cycle  
    END FUNCTION  
    
    FUNCTION find_chain_length(start, graph, visited)  
        SET length TO zero  
        SET queue TO a new double-ended queue containing start  
        SET element at position start of visited TO True  
        
        WHILE queue is not empty  
            REMOVE first element FROM queue AND SET TO current  
            FOR each neighbor IN the list at key current in graph  
                IF element at position neighbor of visited EQUALS False THEN  
                    SET element at position neighbor of visited TO True  
                    APPEND neighbor TO queue  
                    INCREMENT length BY one  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN length  
    END FUNCTION  
    
    FUNCTION find_cycle_length(start, graph, visited, favorite)  
        IF element at position start of visited EQUALS True THEN  
            RETURN zero  
        END IF  
        
        SET stack TO an empty list  
        SET current TO start  
        
        WHILE element at position current of visited EQUALS False  
            SET element at position current of visited TO True  
            APPEND current TO stack  
            SET current TO element at position current of favorite  
        END WHILE  
        
        IF current NOT IN stack THEN  
            RETURN zero  
        END IF  
        
        SET cycle_start_index TO the POSITION OF current IN stack  
        SET cycle_length TO the LENGTH OF stack MINUS cycle_start_index  
        
        RETURN cycle_length  
    END FUNCTION  
END CLASS
