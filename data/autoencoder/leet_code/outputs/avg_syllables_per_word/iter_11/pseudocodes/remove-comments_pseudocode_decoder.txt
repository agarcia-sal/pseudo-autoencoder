CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO False  
        SET result TO empty list  
        SET current_line TO empty list  
        FOR each line IN source  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF line  
                IF in_block_comment EQUALS True  
                    IF the element at position i of line AND the element at position i PLUS one of line EQUALS the string "ASTERISK SLASH"  
                        SET in_block_comment TO False  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF the element at position i of line AND the element at position i PLUS one of line EQUALS the string "SLASH SLASH"  
                        BREAK  
                    ELSE IF the element at position i of line AND the element at position i PLUS one of line EQUALS the string "SLASH ASTERISK"  
                        SET in_block_comment TO True  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND the element at position i of line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  
            IF the LENGTH OF current_line GREATER THAN zero AND in_block_comment EQUALS False  
                APPEND the concatenation of current_line TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
