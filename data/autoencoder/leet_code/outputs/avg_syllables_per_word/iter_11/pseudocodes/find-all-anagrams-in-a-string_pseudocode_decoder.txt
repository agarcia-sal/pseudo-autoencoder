CLASS Solution  
    FUNCTION findAnagrams(s, p)  
        SET len_s TO the LENGTH OF s  
        SET len_p TO the LENGTH OF p  
        
        IF len_p GREATER THAN len_s  
            RETURN empty list  
        END IF  
        
        SET p_count TO the frequency COUNTER of characters IN p  
        SET s_count TO the frequency COUNTER of characters IN the SUBSTRING of s FROM zero TO len_p MINUS one  
        
        SET anagrams TO empty list  
        
        IF s_count EQUALS p_count  
            APPEND zero TO anagrams  
        END IF  
        
        FOR i FROM len_p TO len_s MINUS one  
            INCREMENT count of character at position i of s IN s_count BY one  
            DECREMENT count of character at position i MINUS len_p of s IN s_count BY one  
            
            IF count of character at position i MINUS len_p of s IN s_count EQUALS zero  
                REMOVE count of character at position i MINUS len_p of s FROM s_count  
            END IF  
            
            IF s_count EQUALS p_count  
                APPEND i MINUS len_p PLUS one TO anagrams  
            END IF  
        END FOR  
        
        RETURN anagrams  
    END FUNCTION  
END CLASS
