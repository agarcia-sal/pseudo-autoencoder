CLASS Solution  
    FUNCTION countComponents(n, edges)  
        SET graph TO an empty mapping from integer TO empty list  
        FOR index FROM zero TO n MINUS one  
            SET the element at position index of graph TO an empty list  
        END FOR  
        FOR each pair a, b IN edges  
            APPEND b TO the list at position a of graph  
            APPEND a TO the list at position b of graph  
        END FOR  
  
        SET visited TO an empty set  
  
        FUNCTION dfs(node)  
            IF node IN visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list at position node of graph  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
  
        SET components TO zero  
        FOR node FROM zero TO n MINUS one  
            IF node NOT IN visited  
                CALL dfs WITH node  
                INCREMENT components BY one  
            END IF  
        END FOR  
  
        RETURN components  
    END FUNCTION  
END CLASS
