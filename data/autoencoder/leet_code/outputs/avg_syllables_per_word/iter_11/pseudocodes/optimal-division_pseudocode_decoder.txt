CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the STRING REPRESENTATION OF the element at position one of nums  
        END IF  

        IF the LENGTH OF nums EQUALS two  
            RETURN the STRING REPRESENTATION OF the element at position one of nums CONCATENATED WITH the STRING FORWARD SLASH CONCATENATED WITH the STRING REPRESENTATION OF the element at position two of nums  
        END IF  

        RETURN the STRING REPRESENTATION OF the element at position one of nums CONCATENATED WITH the STRING FORWARD SLASH CONCATENATED WITH the OPENING PARENTHESIS CONCATENATED WITH the JOINING of the STRING REPRESENTATION of each element from position two TO the END OF nums WITH the STRING FORWARD SLASH AS SEPARATOR CONCATENATED WITH the CLOSING PARENTHESIS  
    END FUNCTION  
END CLASS
