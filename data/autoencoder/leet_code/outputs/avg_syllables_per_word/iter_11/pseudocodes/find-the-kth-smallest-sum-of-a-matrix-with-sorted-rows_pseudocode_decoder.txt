CLASS Solution  
    FUNCTION kthSmallest(mat, k)  
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET min_heap TO a list containing a tuple of the SUM of element at position zero of each row in mat, AND a tuple of zero repeated m TIMES  
        SET visited TO a set containing the second element of the first tuple in min_heap  
        
        WHILE k GREATER THAN zero  
            REMOVE the smallest element FROM min_heap and DESTRUCTURE it INTO current_sum and indices  
            DECREMENT k BY one  
            
            IF k EQUALS zero  
                RETURN current_sum  
            END IF  
            
            FOR i FROM zero TO m MINUS one  
                IF element at position i of indices PLUS one LESS THAN n  
                    SET new_indices TO a list COPY OF indices  
                    INCREMENT element at position i of new_indices BY one  
                    SET new_indices_tuple TO tuple FROM new_indices  
                    SET new_sum TO current_sum MINUS element at position i and at position element at position i of indices of mat PLUS element at position i and at position element at position i of new_indices of mat  
                    
                    IF new_indices_tuple NOT IN visited  
                        ADD new_indices_tuple TO visited  
                        INSERT the tuple of new_sum and new_indices_tuple INTO min_heap PRESERVING heap property  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
