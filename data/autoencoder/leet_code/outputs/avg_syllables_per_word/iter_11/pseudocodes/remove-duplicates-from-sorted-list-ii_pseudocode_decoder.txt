CLASS ListNode  
    FUNCTION __init__(val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with value at position zero of values  
    SET p TO head  
    FOR each val IN values starting from position one to the end  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN (value at position val of p1 EQUALS value at position val of p2) AND is_same_list(value at position next of p1, value at position next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates(head)  
        SET dummy TO new ListNode with zero and next equal to head  
        SET prev TO dummy  
        WHILE head IS NOT none  
            IF value at position next of head IS NOT none AND value at position val of head EQUALS value at position val of value at position next of head  
                WHILE value at position next of head IS NOT none AND value at position val of head EQUALS value at position val of value at position next of head  
                    SET head TO value at position next of head  
                END WHILE  
                SET value at position next of prev TO value at position next of head  
            ELSE  
                SET prev TO value at position next of prev  
            END IF  
            SET head TO value at position next of head  
        END WHILE  
        RETURN value at position next of dummy  
    END FUNCTION  
END CLASS
