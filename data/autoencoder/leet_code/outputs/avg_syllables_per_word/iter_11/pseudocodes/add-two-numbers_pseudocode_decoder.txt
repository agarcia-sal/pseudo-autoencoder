CLASS ListNode  
    FUNCTION __init__ with parameters val, next  
        SET val TO zero if not provided  
        SET next TO none if not provided  
        SET the instance val TO val  
        SET the instance next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with the first element of values  
    SET p TO head  
    FOR each val IN values starting from second element to end  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1, p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the condition that p1.val EQUALS p2.val AND is_same_list with parameters p1.next, p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers with parameters l1, l2  
        SET dummy_head TO new ListNode with zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT none OR l2 IS NOT none OR carry IS NOT zero  
            SET val1 TO l1.val IF l1 IS NOT none OTHERWISE zero  
            SET val2 TO l2.val IF l2 IS NOT none OTHERWISE zero  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET current.next TO new ListNode with total MODULO ten  
            SET current TO current.next  
            
            IF l1 IS NOT none  
                SET l1 TO l1.next  
            END IF  
            IF l2 IS NOT none  
                SET l2 TO l2.next  
            END IF  
        END WHILE  
        
        RETURN dummy_head.next  
    END FUNCTION  
END CLASS
