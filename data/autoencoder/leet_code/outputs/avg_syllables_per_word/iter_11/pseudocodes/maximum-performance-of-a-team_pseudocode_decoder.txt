CLASS Solution  
    FUNCTION maxPerformance(n, speed, efficiency, k)  
        SET engineers TO list combining elements of efficiency and speed as pairs  
        SORT engineers in descending order by first element of each pair (efficiency)  
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  

        FOR each pair eff and spd IN engineers  
            ADD spd TO speed_heap maintaining min-heap property  
            INCREMENT speed_sum BY spd  

            IF the LENGTH OF speed_heap GREATER THAN k  
                REMOVE the smallest element FROM speed_heap and DECREMENT speed_sum BY that value  
            END IF  

            SET current_performance TO speed_sum MULTIPLIED BY eff  
            SET max_performance TO the greater of max_performance and current_performance  
        END FOR  

        RETURN max_performance MODULO one billion PLUS seven  
    END FUNCTION  
END CLASS
