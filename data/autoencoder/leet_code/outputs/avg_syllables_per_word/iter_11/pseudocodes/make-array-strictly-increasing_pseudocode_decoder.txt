CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL sort ON arr2  
        SET dp TO dictionary with single entry key MINUS ONE and value ZERO  
        FOR each num IN arr1  
            SET new_dp TO empty dictionary  
            FOR each last_num and ops IN entries of dp  
                IF num GREATER THAN last_num  
                    SET current_value TO value at key num in new_dp or infinity if absent  
                    SET new_dp at key num TO minimum of current_value and ops  
                END IF  
                SET idx TO result of binary search for smallest element GREATER THAN last_num in arr2 using bisect_right  
                IF idx LESS THAN LENGTH OF arr2  
                    SET replacement TO element at position idx of arr2  
                    SET current_value_replacement TO value at key replacement in new_dp or infinity if absent  
                    SET new_dp at key replacement TO minimum of current_value_replacement and ops PLUS ONE  
                END IF  
            END FOR  
            IF new_dp is empty  
                RETURN MINUS ONE  
            END IF  
            SET dp TO new_dp  
        END FOR  
        RETURN minimum value among all values in dp  
    END FUNCTION  
END CLASS
