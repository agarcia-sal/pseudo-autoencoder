CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO the LIST OF tuples of element at position zero of interval and index i FOR each index i AND interval IN intervals SORTED BY the element at position zero of interval  
        SET result TO empty list  
        FOR each element at position one of interval IN intervals  
            SET idx TO the POSITION OF element at position zero EQUAL TO element at position one IN start_times USING BINARY SEARCH FOR LEFTMOST MATCH  
            IF idx LESS THAN THE LENGTH OF start_times  
                APPEND element at position one OF element at position idx OF start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
