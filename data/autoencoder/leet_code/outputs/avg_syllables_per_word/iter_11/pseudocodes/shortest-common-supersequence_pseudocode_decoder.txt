CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1, str2)  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
          
        CALL initialize_2D_list WITH parameters m PLUS one, n PLUS one RETURNING dp  
          
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2  
                    SET the element at position i and j of dp TO the element at position i MINUS one and j MINUS one of dp PLUS one  
                ELSE  
                    SET the element at position i and j of dp TO the GREATER OF the element at position i MINUS one and j of dp AND the element at position i and j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
          
        SET i TO m  
        SET j TO n  
        SET scs TO an empty list  
          
        WHILE i GREATER THAN zero AND j GREATER THAN zero  
            IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF the element at position i MINUS one and j of dp GREATER THAN the element at position i and j MINUS one of dp  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND the element at position j MINUS one of str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
          
        WHILE i GREATER THAN zero  
            APPEND the element at position i MINUS one of str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
          
        WHILE j GREATER THAN zero  
            APPEND the element at position j MINUS one of str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
          
        CALL reverse_list WITH parameter scs  
          
        RETURN concatenation of all elements IN scs as a string  
    END FUNCTION  
      
    FUNCTION initialize_2D_list(rows, columns)  
        SET dp TO an empty list  
        FOR each number FROM zero TO rows MINUS one  
            SET row TO an empty list  
            FOR each number FROM zero TO columns MINUS one  
                APPEND zero TO row  
            END FOR  
            APPEND row TO dp  
        END FOR  
        RETURN dp  
    END FUNCTION  
      
    FUNCTION reverse_list(list_to_reverse)  
        SET start TO zero  
        SET end TO the LENGTH OF list_to_reverse MINUS one  
        WHILE start LESS THAN end  
            SWAP the element at position start WITH the element at position end in list_to_reverse  
            INCREMENT start BY one  
            DECREMENT end BY one  
        END WHILE  
    END FUNCTION  
END CLASS
