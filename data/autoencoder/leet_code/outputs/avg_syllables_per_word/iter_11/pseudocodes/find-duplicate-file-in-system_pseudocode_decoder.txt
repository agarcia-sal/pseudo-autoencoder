CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO empty mapping from content TO list of file paths  
        FOR each path IN paths  
            SET parts TO the list of elements obtained by splitting path by whitespace  
            SET directory TO element at position zero of parts  
            FOR each file_info IN the elements of parts from position one TO the end  
                SET file_name TO the substring before the first occurrence of the character '(' in file_info  
                SET content TO the substring of file_info from the character after '(' up to but excluding the last character  
                APPEND the concatenation of directory, a forward slash, and file_name TO the list stored at content in content_to_paths  
            END FOR  
        END FOR  
        SET result TO empty list  
        FOR each paths_list IN the values of content_to_paths  
            IF the length of paths_list GREATER THAN one  
                APPEND paths_list TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
