CLASS Solution  
    FUNCTION replaceNonCoprimes(nums)  
        SET stack TO empty list  
        FOR each num IN nums  
            WHILE stack IS NOT empty  
                SET g TO greatest common divisor OF element at last position OF stack AND num  
                IF g EQUALS one  
                    BREAK the WHILE loop  
                END IF  
                SET num TO element at last position OF stack MULTIPLIED BY num DIVIDED BY g  
                REMOVE last element FROM stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
