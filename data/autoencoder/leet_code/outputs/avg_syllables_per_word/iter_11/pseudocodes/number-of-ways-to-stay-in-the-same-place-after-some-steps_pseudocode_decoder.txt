CLASS Solution  
    FUNCTION numWays(steps, arrLen)  
        SET MOD TO one billion PLUS seven  
        SET max_index TO minimum OF steps DIVIDED BY two AND arrLen MINUS one  
        SET dp TO a list of zeros with length max_index PLUS one  
        SET element at position zero of dp TO one  
        FOR each step FROM one TO steps  
            SET prev_dp TO a copy of dp  
            FOR each i FROM zero TO max_index  
                SET element at position i of dp TO element at position i of prev_dp  
                IF i GREATER THAN zero  
                    SET element at position i of dp TO element at position i of dp PLUS element at position i MINUS one of prev_dp  
                    SET element at position i of dp TO element at position i of dp MODULO MOD  
                END IF  
                IF i LESS THAN max_index  
                    SET element at position i of dp TO element at position i of dp PLUS element at position i PLUS one of prev_dp  
                    SET element at position i of dp TO element at position i of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
