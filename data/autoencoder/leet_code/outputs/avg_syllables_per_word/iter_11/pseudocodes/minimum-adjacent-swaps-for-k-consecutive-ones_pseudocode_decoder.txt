CLASS Solution  
    FUNCTION minMoves(nums, k)  
        SET positions TO empty list  
        FOR each index i FROM zero TO the LENGTH OF nums MINUS one  
            IF the element at position i of nums EQUALS one  
                APPEND i TO positions  
            END IF  
        END FOR  
          
        FUNCTION calculate_cost(start, end)  
            SET mid TO start PLUS end DIVIDED BY two USING INTEGER DIVISION  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR each index i FROM start TO end MINUS one  
                SET distance TO positions at position i MINUS median PLUS mid MINUS i  
                IF distance LESS THAN zero  
                    SET distance TO NEGATION OF distance  
                END IF  
                INCREMENT cost BY distance  
            END FOR  
            RETURN cost  
        END FUNCTION  
          
        SET min_cost TO positive infinity  
        FOR each index i FROM zero TO the LENGTH OF positions MINUS k  
            SET current_cost TO calculate_cost(i, i PLUS k)  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
          
        RETURN min_cost  
    END FUNCTION  
END CLASS
