CLASS Solution  
    FUNCTION isNStraightHand(hand, groupSize)  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero  
            RETURN False  
        END IF  
        SET count TO CALL Counter WITH hand  
        SET unique_cards TO CALL sorted WITH the KEYS OF count  
        FOR each card IN unique_cards  
            IF the VALUE AT count FOR card GREATER THAN zero  
                SET needed TO the VALUE AT count FOR card  
                FOR i FROM zero TO groupSize MINUS one  
                    IF the VALUE AT count FOR card PLUS i LESS THAN needed  
                        RETURN False  
                    END IF  
                    SET the VALUE AT count FOR card PLUS i TO the VALUE AT count FOR card PLUS i MINUS needed  
                END FOR  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
