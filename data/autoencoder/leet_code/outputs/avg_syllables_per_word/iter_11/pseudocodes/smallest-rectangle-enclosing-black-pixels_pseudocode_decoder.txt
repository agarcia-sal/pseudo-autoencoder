CLASS Solution  
    FUNCTION minArea(image, x, y)  
        IF image EQUALS empty OR the first element of image EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF image  
        SET n TO the LENGTH OF the first element of image  
        
        FUNCTION hasBlackPixelInRow(row)  
            FOR each pixel IN the element at position row of image  
                IF pixel EQUALS one  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION hasBlackPixelInColumn(col)  
            FOR row FROM zero TO m MINUS one  
                IF the element at position col of the element at position row of image EQUALS one  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION findTop()  
            SET low TO zero  
            SET high TO x  
            WHILE low LESS THAN high  
                SET mid TO low PLUS high DIVIDED BY two USING INTEGER DIVISION  
                IF hasBlackPixelInRow(mid)  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findBottom()  
            SET low TO x  
            SET high TO m MINUS one  
            WHILE low LESS THAN high  
                SET mid TO low PLUS high PLUS one DIVIDED BY two USING INTEGER DIVISION  
                IF hasBlackPixelInRow(mid)  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findLeft()  
            SET low TO zero  
            SET high TO y  
            WHILE low LESS THAN high  
                SET mid TO low PLUS high DIVIDED BY two USING INTEGER DIVISION  
                IF hasBlackPixelInColumn(mid)  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findRight()  
            SET low TO y  
            SET high TO n MINUS one  
            WHILE low LESS THAN high  
                SET mid TO low PLUS high PLUS one DIVIDED BY two USING INTEGER DIVISION  
                IF hasBlackPixelInColumn(mid)  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        SET top TO findTop()  
        SET bottom TO findBottom()  
        SET left TO findLeft()  
        SET right TO findRight()  
        
        RETURN bottom MINUS top PLUS one MULTIPLIED BY right MINUS left PLUS one  
    END FUNCTION  
END CLASS
