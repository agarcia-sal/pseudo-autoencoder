CLASS Solution  
    FUNCTION findLongestChain(pairs)  
        CALL pairs SORTED BY element at position two of each pair IN ascending order  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each pair IN pairs  
            SET start TO first element of pair  
            SET end TO second element of pair  
            IF start GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
