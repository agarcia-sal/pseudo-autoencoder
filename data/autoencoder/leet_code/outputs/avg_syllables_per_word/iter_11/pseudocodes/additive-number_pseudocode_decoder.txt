CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            
            SET expected TO the STRING REPRESENTATION OF first PLUS second  
            SET length TO the LENGTH OF expected  
            
            IF start PLUS length GREATER THAN the LENGTH OF num OR the SUBSTRING OF num FROM start TO start PLUS length NOT EQUALS expected  
                RETURN False  
            END IF  
            
            RETURN the RESULT OF is_valid_sequence WITH PARAMETERS start PLUS length, second, AND the INTEGER VALUE OF expected  
        END FUNCTION  
        
        SET n TO the LENGTH OF num  
        FOR i FROM one TO n DIVIDED BY two PLUS one  
            FOR j FROM i PLUS one TO n MINUS one PLUS one  
                SET first TO the SUBSTRING OF num FROM ZERO TO i  
                SET second TO the SUBSTRING OF num FROM i TO j  
                
                IF the LENGTH OF first GREATER THAN one AND the FIRST CHARACTER OF first EQUALS zero OR the LENGTH OF second GREATER THAN one AND the FIRST CHARACTER OF second EQUALS zero  
                    CONTINUE TO next iteration OF the inner FOR loop  
                END IF  
                
                IF the RESULT OF is_valid_sequence WITH PARAMETERS j, the INTEGER VALUE OF first, AND the INTEGER VALUE OF second  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
