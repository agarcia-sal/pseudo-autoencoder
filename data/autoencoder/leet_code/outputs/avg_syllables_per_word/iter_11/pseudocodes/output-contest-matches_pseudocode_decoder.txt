CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO list of strings representing numbers from one TO n  
        WHILE n GREATER THAN one  
            FOR i FROM zero TO n DIVIDED BY two MINUS one  
                SET element at position i of teams TO concatenation of open parenthesis, element at position i of teams, comma, element at position n MINUS one MINUS i of teams, close parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
