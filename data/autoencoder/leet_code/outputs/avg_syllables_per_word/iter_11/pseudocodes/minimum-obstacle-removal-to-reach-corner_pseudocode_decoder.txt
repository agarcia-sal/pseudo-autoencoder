CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO list containing tuple zero and one, tuple one and zero, tuple zero and negative one, tuple negative one and zero  
        SET min_heap TO list containing tuple zero, zero, zero  
        SET visited TO set containing tuple zero and zero  
        
        WHILE min_heap is NOT EMPTY  
            REMOVE the element with smallest cost FROM min_heap and DESTRUCTURE into cost, x, y  
            
            IF tuple x and y EQUALS tuple m MINUS one and n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each tuple dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF zero LESS THAN OR EQUAL TO nx AND nx LESS THAN m AND zero LESS THAN OR EQUAL TO ny AND ny LESS THAN n AND tuple nx and ny NOT IN visited  
                    ADD tuple nx and ny TO visited  
                    SET new_cost TO cost PLUS element at position nx of grid at position ny  
                    ADD tuple new_cost, nx, ny TO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
