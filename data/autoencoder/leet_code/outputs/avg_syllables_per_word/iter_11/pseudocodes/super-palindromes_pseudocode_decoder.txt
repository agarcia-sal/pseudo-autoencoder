CLASS Solution  
    FUNCTION superpalindromesInRange(left, right)  
        FUNCTION is_palindrome(num)  
            RETURN string representation of num EQUALS reverse of string representation of num  
        END FUNCTION  
        
        FUNCTION generate_palindromes(limit)  
            SET palindromes TO empty list  
            FOR i FROM one TO one hundred thousand MINUS one  
                SET s TO string representation of i  
                APPEND integer converted from concatenation of s AND reverse of substring of s from second last character to first character TO palindromes  
                APPEND integer converted from concatenation of s AND reverse of s TO palindromes  
            END FOR  
            RETURN list of elements p from palindromes WHERE p IS LESS THAN OR EQUAL TO limit  
        END FUNCTION  
        
        SET left TO integer converted from left  
        SET right TO integer converted from right  
        SET limit TO integer converted from square root of right PLUS one  
        SET palindromes TO generate_palindromes(limit)  
        
        SET count TO zero  
        FOR each p IN palindromes  
            SET square TO p MULTIPLIED BY p  
            IF left IS LESS THAN OR EQUAL TO square AND square IS LESS THAN OR EQUAL TO right AND is_palindrome(square)  
                INCREMENT count BY one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
