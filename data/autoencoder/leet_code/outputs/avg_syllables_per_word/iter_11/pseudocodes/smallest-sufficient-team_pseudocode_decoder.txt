CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        SET d TO an empty mapping from skill to index  
        FOR each index i FROM zero TO LENGTH OF req_skills MINUS one  
            SET skill s TO element at position i of req_skills  
            SET mapping d at skill s TO i  
        END FOR  
        
        SET m TO LENGTH OF req_skills  
        SET n TO LENGTH OF people  
        
        SET p TO a list of zero repeated n times  
        
        FOR each index i FROM zero TO n MINUS one  
            SET ss TO element at position i of people  
            FOR each skill s IN ss  
                SET p at position i TO p at position i OR one SHIFTED LEFT BY d at skill s  
            END FOR  
        END FOR  
        
        SET f TO a list of inf repeated two POWER m times  
        SET g TO a list of zero repeated two POWER m times  
        SET h TO a list of zero repeated two POWER m times  
        
        SET f at position zero TO zero  
        
        FOR each i FROM zero TO two POWER m MINUS one  
            IF f at position i EQUALS inf  
                CONTINUE  
            END IF  
            FOR each j FROM zero TO n MINUS one  
                IF f at position i PLUS one LESS THAN f at position i OR p at position j  
                    SET f at position i OR p at position j TO f at position i PLUS one  
                    SET g at position i OR p at position j TO j  
                    SET h at position i OR p at position j TO i  
                END IF  
            END FOR  
        END FOR  
        
        SET i TO two POWER m MINUS one  
        SET ans TO an empty list  
        WHILE i NOT EQUALS zero  
            APPEND g at position i TO ans  
            SET i TO h at position i  
        END WHILE  
        
        RETURN ans  
    END FUNCTION  
END CLASS
