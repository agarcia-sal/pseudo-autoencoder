CLASS Solution
    FUNCTION solveEquation(equation)
        FUNCTION parse_side(side)
            SET tokens TO the result of replacing plus signs in side with hash plus sign and replacing minus signs in side with hash minus sign and then splitting by hash character
            SET x_count TO zero
            SET num_sum TO zero
            FOR each token IN tokens
                IF token EQUALS empty string
                    CONTINUE to next iteration
                END IF
                IF token CONTAINS the character x
                    IF token EQUALS the character x OR token EQUALS plus sign followed by x
                        INCREMENT x_count BY one
                    ELSE IF token EQUALS minus sign followed by x
                        DECREMENT x_count BY one
                    ELSE
                        SET x_count TO x_count PLUS the integer value of token after replacing x with empty string
                    END IF
                ELSE
                    INCREMENT num_sum BY the integer value of token
                END IF
            END FOR
            RETURN x_count, num_sum
        END FUNCTION

        SET left_side, right_side TO the two parts of equation separated by equal sign
        SET left_x, left_num TO the result of calling parse_side with left_side
        SET right_x, right_num TO the result of calling parse_side with right_side

        SET total_x TO left_x MINUS right_x
        SET total_num TO right_num MINUS left_num

        IF total_x EQUALS zero
            IF total_num EQUALS zero
                RETURN the string Infinite solutions
            ELSE
                RETURN the string No solution
            END IF
        ELSE
            RETURN the string consisting of the character x followed by equal sign followed by the integer result of total_num DIVIDED BY total_x
        END IF
    END FUNCTION
END CLASS
