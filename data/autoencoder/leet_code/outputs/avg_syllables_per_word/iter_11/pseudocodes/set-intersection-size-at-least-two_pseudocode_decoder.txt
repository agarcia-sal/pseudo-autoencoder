CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        CALL intervals SORTED BY end element of interval AND IF ends ARE EQUAL, THEN BY negative start element of interval  
        SET first TO element at position one of first interval MINUS one  
        SET second TO element at position one of first interval  
        SET count TO two  
        FOR each start, end IN intervals FROM position one TO the last element  
            IF start GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
