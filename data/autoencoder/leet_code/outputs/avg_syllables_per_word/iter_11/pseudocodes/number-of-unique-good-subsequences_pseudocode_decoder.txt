CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MOD TO one billion PLUS seven  
        SET end_with_0 TO zero  
        SET end_with_1 TO zero  
        SET has_zero TO true IF the STRING binary CONTAINS character zero OTHERWISE false  
        FOR each bit IN binary  
            IF bit EQUALS character one  
                SET end_with_1 TO the REMAINDER AFTER DIVIDING the SUM of end_with_0 PLUS end_with_1 PLUS one BY MOD  
            ELSE  
                SET end_with_0 TO the REMAINDER AFTER DIVIDING the SUM of end_with_0 PLUS end_with_1 BY MOD  
            END IF  
        END FOR  
        RETURN the REMAINDER AFTER DIVIDING the SUM of end_with_0 PLUS end_with_1 PLUS has_zero BY MOD  
    END FUNCTION  
END CLASS
