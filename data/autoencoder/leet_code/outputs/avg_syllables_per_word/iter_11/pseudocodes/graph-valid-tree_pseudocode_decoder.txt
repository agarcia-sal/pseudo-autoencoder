CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF the LENGTH OF edges NOT EQUALS n MINUS one  
            RETURN false  
        END IF  
        SET adj_list TO an empty mapping from integers to lists  
        FOR index FROM zero TO n MINUS one  
            SET element at position index of adj_list TO an empty list  
        END FOR  
        FOR each pair a and b IN edges  
            APPEND b TO the list at key a in adj_list  
            APPEND a TO the list at key b in adj_list  
        END FOR  
        SET visited TO an empty set  
        FUNCTION dfs(node)  
            IF node IS IN visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list at key node in adj_list  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        CALL dfs WITH zero  
        RETURN the LENGTH OF visited EQUALS n  
    END FUNCTION  
END CLASS
