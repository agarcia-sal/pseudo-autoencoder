CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO empty list  
        SET score TO zero  
        FOR each char IN s  
            IF char EQUALS open parenthesis  
                APPEND score TO stack  
                SET score TO zero  
            ELSE  
                SET score TO last element removed FROM stack PLUS (one IF score EQUALS zero ELSE two MULTIPLIED BY score)  
            END IF  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
