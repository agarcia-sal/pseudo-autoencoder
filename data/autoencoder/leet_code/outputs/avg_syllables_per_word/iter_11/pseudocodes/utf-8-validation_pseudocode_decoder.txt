CLASS Solution  
    FUNCTION validUtf8(data)  
        FUNCTION get_byte_count(byte)  
            IF the value of byte SHIFTED RIGHT BY seven EQUALS zero  
                RETURN one  
            ELSE IF the value of byte SHIFTED RIGHT BY five EQUALS the binary value one one zero  
                RETURN two  
            ELSE IF the value of byte SHIFTED RIGHT BY four EQUALS the binary value one one one zero  
                RETURN three  
            ELSE IF the value of byte SHIFTED RIGHT BY three EQUALS the binary value one one one one zero  
                RETURN four  
            ELSE  
                RETURN negative one  
            END IF  
        END FUNCTION  
  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF data  
            SET byte_count TO get_byte_count(element at position i of data)  
            IF byte_count EQUALS negative one  
                RETURN false  
            END IF  
            IF i PLUS byte_count GREATER THAN the LENGTH OF data  
                RETURN false  
            END IF  
            FOR j FROM i PLUS one TO i PLUS byte_count MINUS one  
                IF the value of element at position j of data SHIFTED RIGHT BY six NOT EQUALS the binary value one zero  
                    RETURN false  
                END IF  
            END FOR  
            INCREMENT i BY byte_count  
        END WHILE  
        RETURN true  
    END FUNCTION  
END CLASS
