CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO a new ListNode initialized with element at position zero of values  
    SET p TO head  
    FOR each val IN elements from position one to the end of values  
        SET node TO a new ListNode initialized with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN element val of p1 EQUALS element val of p2 AND is_same_list(element next of p1, element next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION partition(self, head, x)  
        SET less_head TO a new ListNode initialized with zero  
        SET greater_head TO a new ListNode initialized with zero  

        SET less TO less_head  
        SET greater TO greater_head  

        SET current TO head  
        WHILE current IS NOT None  
            IF element val of current LESS THAN x  
                SET less.next TO current  
                SET less TO element next of less  
            ELSE  
                SET greater.next TO current  
                SET greater TO element next of greater  
            END IF  
            SET current TO element next of current  
        END WHILE  

        SET less.next TO element next of greater_head  
        SET greater.next TO None  

        RETURN element next of less_head  
    END FUNCTION  
END CLASS
