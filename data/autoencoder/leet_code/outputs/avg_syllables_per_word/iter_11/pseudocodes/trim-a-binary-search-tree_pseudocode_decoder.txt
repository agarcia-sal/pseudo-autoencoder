CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero, left EQUALS none, right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH val EQUAL TO element at position zero of values  
    SET i TO one  
    SET queue TO NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN length of values AND element at position i of values IS NOT none  
            SET node.left TO NEW TreeNode WITH val EQUAL TO element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values IS NOT none  
            SET node.right TO NEW TreeNode WITH val EQUAL TO element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left, q.left AND is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST WITH PARAMETERS self, root, low, high  
        IF root IS none  
            RETURN none  
        END IF  
        IF root.val LESS THAN low  
            RETURN self.trimBST WITH PARAMETERS root.right, low, high  
        END IF  
        IF root.val GREATER THAN high  
            RETURN self.trimBST WITH PARAMETERS root.left, low, high  
        END IF  
        SET root.left TO self.trimBST WITH PARAMETERS root.left, low, high  
        SET root.right TO self.trimBST WITH PARAMETERS root.right, low, high  
        RETURN root  
    END FUNCTION  
END CLASS
