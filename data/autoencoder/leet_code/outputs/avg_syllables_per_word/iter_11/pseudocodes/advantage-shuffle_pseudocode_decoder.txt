CLASS Solution
    FUNCTION advantageCount(nums1, nums2)
        SORT nums1 IN ASCENDING ORDER
        SET nums2_with_indices TO a list of pairs of each element and its index from nums2
        SORT nums2_with_indices IN DESCENDING ORDER BY element value
        SET result TO a list of zeros with the LENGTH OF nums1
        SET left TO zero
        SET right TO LENGTH OF nums1 MINUS one
        FOR each pair value and index IN nums2_with_indices
            IF element at position right of nums1 GREATER THAN value
                SET element at position index of result TO element at position right of nums1
                DECREMENT right BY one
            ELSE
                SET element at position index of result TO element at position left of nums1
                INCREMENT left BY one
            END IF
        END FOR
        RETURN result
    END FUNCTION
END CLASS
