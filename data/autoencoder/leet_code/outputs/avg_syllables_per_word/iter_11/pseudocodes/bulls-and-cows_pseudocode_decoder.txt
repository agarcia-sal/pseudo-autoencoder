CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        CALL initialize_default_dictionary FOR secret_count  
        CALL initialize_default_dictionary FOR guess_count  
        FOR each pair s AND g IN zipped secret AND guess  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the count of s IN secret_count BY one  
                INCREMENT the count of g IN guess_count BY one  
            END IF  
        END FOR  
        FOR each digit IN keys of secret_count  
            INCREMENT cows BY the minimum OF the count of digit IN secret_count AND the count of digit IN guess_count  
        END FOR  
        RETURN string COMPOSED OF bulls FOLLOWED BY letter A FOLLOWED BY cows FOLLOWED BY letter B  
    END FUNCTION  
END CLASS
