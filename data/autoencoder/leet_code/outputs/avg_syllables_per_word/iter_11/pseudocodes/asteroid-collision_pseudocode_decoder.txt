CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO empty list  
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack is greater than zero AND asteroid LESS THAN zero AND element at position LENGTH OF stack MINUS one of stack GREATER THAN zero  
                IF element at position LENGTH OF stack MINUS one of stack LESS THAN the NEGATION OF asteroid  
                    REMOVE last element FROM stack  
                    CONTINUE  
                ELSE IF element at position LENGTH OF stack MINUS one of stack EQUALS the NEGATION OF asteroid  
                    REMOVE last element FROM stack  
                END IF  
                BREAK  
            END WHILE  
            IF the WHILE loop DID NOT break prematurely  
                APPEND asteroid TO stack  
            END IF  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
