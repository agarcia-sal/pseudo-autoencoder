CLASS Solution  
    FUNCTION getMinDistSum(positions)  
        SET n TO the LENGTH OF positions  
        SET x TO zero  
        SET y TO zero  
        FOR each pair x1, y1 IN positions  
            INCREMENT x BY x1  
            INCREMENT y BY y1  
        END FOR  
        SET x TO x DIVIDED BY n  
        SET y TO y DIVIDED BY n  
        SET decay TO zero POINT nine nine nine  
        SET eps TO one TIMES ten RAISED TO negative six  
        SET alpha TO zero POINT five  
        WHILE true  
            SET grad_x TO zero  
            SET grad_y TO zero  
            SET dist TO zero  
            FOR each pair x1, y1 IN positions  
                SET a TO x MINUS x1  
                SET b TO y MINUS y1  
                SET c TO the SQUARE ROOT OF a MULTIPLIED BY a PLUS b MULTIPLIED BY b  
                INCREMENT grad_x BY a DIVIDED BY c PLUS one TIMES ten RAISED TO negative eight  
                INCREMENT grad_y BY b DIVIDED BY c PLUS one TIMES ten RAISED TO negative eight  
                INCREMENT dist BY c  
            END FOR  
            SET dx TO grad_x MULTIPLIED BY alpha  
            SET dy TO grad_y MULTIPLIED BY alpha  
            DECREMENT x BY dx  
            DECREMENT y BY dy  
            MULTIPLY alpha BY decay AND SET alpha TO the RESULT  
            IF the ABSOLUTE VALUE OF dx LESS THAN OR EQUAL TO eps AND the ABSOLUTE VALUE OF dy LESS THAN OR EQUAL TO eps  
                RETURN dist  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
