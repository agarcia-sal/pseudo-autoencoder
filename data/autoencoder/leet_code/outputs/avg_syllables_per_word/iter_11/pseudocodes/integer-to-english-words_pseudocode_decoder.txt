CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the string "Zero"  
        END IF  
  
        SET below_twenty TO the list of strings empty string, "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine", "Ten", "Eleven", "Twelve", "Thirteen", "Fourteen", "Fifteen", "Sixteen", "Seventeen", "Eighteen", "Nineteen"  
        SET tens TO the list of strings empty string, empty string, "Twenty", "Thirty", "Forty", "Fifty", "Sixty", "Seventy", "Eighty", "Ninety"  
        SET thousands TO the list of strings empty string, "Thousand", "Million", "Billion"  
  
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            ELSE IF n LESS THAN twenty  
                RETURN element at position n of below_twenty PLUS the string with a single space  
            ELSE IF n LESS THAN one hundred  
                RETURN element at position n DIVIDED BY ten of tens PLUS the string with a single space PLUS helper(n MODULO ten)  
            ELSE  
                RETURN element at position n DIVIDED BY one hundred of below_twenty PLUS the string " Hundred " PLUS helper(n MODULO one hundred)  
            END IF  
        END FUNCTION  
  
        SET result TO empty string  
        SET i TO zero  
  
        WHILE num GREATER THAN zero  
            IF num MODULO one thousand NOT EQUALS zero  
                SET result TO helper(num MODULO one thousand) PLUS element at position i of thousands PLUS the string with a single space PLUS result  
            END IF  
            SET num TO num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  
  
        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
