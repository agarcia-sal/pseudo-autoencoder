CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones EQUALS empty OR the LENGTH OF stones LESS THAN 2  
            RETURN False  
        END IF  
        
        SET stone_set TO a set containing all elements of stones  
        SET memo TO an empty mapping  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the element at the last position of stones  
                RETURN True  
            END IF  
            
            IF the pair position AND jump IS IN memo  
                RETURN the value in memo at key pair position AND jump  
            END IF  
            
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump GREATER THAN zero AND position PLUS next_jump IS IN stone_set  
                    IF can_jump_to(position PLUS next_jump, next_jump)  
                        SET the value in memo at key pair position AND jump TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the value in memo at key pair position AND jump TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_jump_to(the element at position zero of stones, zero)  
    END FUNCTION  
END CLASS
