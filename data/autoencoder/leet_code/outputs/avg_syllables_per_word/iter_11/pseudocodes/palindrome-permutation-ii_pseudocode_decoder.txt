CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND its count v IN cnt  
                IF v GREATER THAN 1  
                    DECREMENT the count of character c IN cnt BY two  
                    CALL dfs WITH the concatenation of character c PLUS t PLUS character c  
                    INCREMENT the count of character c IN cnt BY two  
                END IF  
            END FOR  
        END FUNCTION  

        SET cnt TO a mapping from characters to their counts in s  
        SET mid TO an empty string  
        FOR each character c AND its count v IN cnt  
            IF v MODULO two EQUALS one  
                IF mid IS NOT empty  
                    RETURN an empty list  
                END IF  
                SET mid TO character c  
                DECREMENT the count of character c IN cnt BY one  
            END IF  
        END FOR  
        SET ans TO an empty list  
        CALL dfs WITH mid  
        RETURN ans  
    END FUNCTION  
END CLASS
