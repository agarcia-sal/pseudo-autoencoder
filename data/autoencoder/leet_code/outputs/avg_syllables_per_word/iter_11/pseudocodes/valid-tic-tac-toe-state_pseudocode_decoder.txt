CLASS Solution  
    FUNCTION validTicTacToe(board)  
        SET countX TO zero  
        FOR each row IN board  
            INCREMENT countX BY the number of occurrences of character X in row  
        END FOR  
        SET countO TO zero  
        FOR each row IN board  
            INCREMENT countO BY the number of occurrences of character O in row  
        END FOR  
        IF countX LESS THAN countO OR countX GREATER THAN countO PLUS one  
            RETURN False  
        END IF  

        FUNCTION check_win(player)  
            FOR each row IN board  
                IF row EQUALS player REPEATED three times  
                    RETURN True  
                END IF  
            END FOR  
            FOR col FROM zero TO two  
                IF element at position zero of board at col EQUALS player AND element at position one of board at col EQUALS player AND element at position two of board at col EQUALS player  
                    RETURN True  
                END IF  
            END FOR  
            IF element at position zero of board at zero EQUALS player AND element at position one of board at one EQUALS player AND element at position two of board at two EQUALS player  
                RETURN True  
            END IF  
            IF element at position zero of board at two EQUALS player AND element at position one of board at one EQUALS player AND element at position two of board at zero EQUALS player  
                RETURN True  
            END IF  
            RETURN False  
        END FUNCTION  

        SET winX TO the result of check_win with argument character X  
        SET winO TO the result of check_win with argument character O  

        IF winX AND winO  
            RETURN False  
        END IF  
        IF winX AND countX EQUALS countO  
            RETURN False  
        END IF  
        IF winO AND countX NOT EQUALS countO  
            RETURN False  
        END IF  

        RETURN True  
    END FUNCTION  
END CLASS
