CLASS Solution  
    FUNCTION waysToDistribute(n, k)  
        SET MOD TO one billion PLUS seven  
        CALL initialize_dp_array WITH n PLUS one AND k PLUS one AND SET RESULT TO dp  
        SET element at position zero of dp TO list of length k PLUS one WITH all elements zero  
        SET element at position zero of dp at position zero TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET product_of_j_and_element_at_i_minus_one_j_of_dp TO j MULTIPLIED BY element at position i MINUS one of dp at position j  
                SET sum_of_product_and_element_at_i_minus_one_j_minus_one_of_dp TO product_of_j_and_element_at_i_minus_one_j_of_dp PLUS element at position i MINUS one of dp at position j MINUS one  
                SET element at position i of dp at position j TO sum_of_product_and_element_at_i_minus_one_j_minus_one_of_dp MODULO MOD  
            END FOR  
        END FOR  
        RETURN element at position n of dp at position k  
    END FUNCTION  
  
    FUNCTION initialize_dp_array(n_length, k_length)  
        SET list_result TO empty list  
        FOR i FROM zero TO n_length MINUS one  
            SET element_to_append TO list of length k_length WITH all elements zero  
            APPEND element_to_append TO list_result  
        END FOR  
        RETURN list_result  
    END FUNCTION  
END CLASS
