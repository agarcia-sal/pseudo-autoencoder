CLASS Solution  
    FUNCTION numPoints(darts, r)  
        FUNCTION distance(p1, p2)  
            COMPUTE squared difference of first element of p1 MINUS first element of p2  
            COMPUTE squared difference of second element of p1 MINUS second element of p2  
            COMPUTE square root of sum of above two squared differences  
            RETURN result  
        END FUNCTION  
        
        FUNCTION circle_center(p1, p2, r)  
            SET d TO distance of p1 AND p2  
            IF d GREATER THAN 2 MULTIPLIED BY r  
                RETURN None  
            END IF  
            SET a TO square root of r MULTIPLIED BY r MINUS (d DIVIDED BY 2) MULTIPLIED BY (d DIVIDED BY 2)  
            SET h TO list containing first element as sum of first elements of p1 AND p2 DIVIDED BY 2 AND second element as sum of second elements of p1 AND p2 DIVIDED BY 2  
            SET dx TO (second element of p2 MINUS second element of p1) MULTIPLIED BY a DIVIDED BY d  
            SET dy TO (first element of p2 MINUS first element of p1) MULTIPLIED BY a DIVIDED BY d  
            RETURN list containing tuple of (first element of h MINUS dx, second element of h PLUS dy) AND tuple of (first element of h PLUS dx, second element of h MINUS dy)  
        END FUNCTION  
        
        FUNCTION is_inside(circle, point, r)  
            RETURN distance of circle AND point LESS THAN OR EQUAL TO r  
        END FUNCTION  
        
        SET n TO length of darts  
        IF n EQUALS 1  
            RETURN 1  
        END IF  
        
        SET max_count TO 1  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET centers TO circle_center of element at position i of darts, element at position j of darts, AND r  
                IF centers IS NOT None  
                    FOR each center IN centers  
                        SET count TO zero  
                        FOR each dart IN darts  
                            IF is_inside of center, dart, AND r  
                                INCREMENT count BY one  
                            END IF  
                        END FOR  
                        SET max_count TO the greater of max_count AND count  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_count  
    END FUNCTION  
END CLASS
