CLASS Solution
    FUNCTION numDistinct(s, t)
        SET m TO the LENGTH OF s
        SET n TO the LENGTH OF t
        
        CALL dp INITIALIZATION WITH m PLUS one AND n PLUS one
        
        FOR i FROM zero TO m
            SET the element at position i and zero of dp TO one
        END FOR
        
        FOR i FROM one TO m
            FOR j FROM one TO n
                IF the character at position i MINUS one of s EQUALS the character at position j MINUS one of t
                    SET the element at position i and j of dp TO the element at position i MINUS one and j MINUS one of dp PLUS the element at position i MINUS one and j of dp
                ELSE
                    SET the element at position i and j of dp TO the element at position i MINUS one and j of dp
                END IF
            END FOR
        END FOR
        
        RETURN the element at position m and n of dp
    END FUNCTION

    FUNCTION dp INITIALIZATION WITH rows, columns
        CREATE a two dimensional list named dp_structure WITH rows number of lists EACH consisting of columns number of zeros
        RETURN dp_structure
    END FUNCTION
END CLASS
