CLASS Solution  
    FUNCTION closestToTarget(arr, target)  
        SET seen TO empty set  
        SET min_diff TO positive infinity  
        FOR each num IN arr  
            SET updated_set TO empty set  
            FOR each x IN seen  
                APPEND bitwise AND of num AND x TO updated_set  
            END FOR  
            SET seen TO union of updated_set AND set containing num  
            FOR each val IN seen  
                SET current_diff TO absolute value of val MINUS target  
                IF current_diff LESS THAN min_diff  
                    SET min_diff TO current_diff  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
