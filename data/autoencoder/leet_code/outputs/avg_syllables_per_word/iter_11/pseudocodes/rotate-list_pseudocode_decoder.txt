CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with val AS element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting from position one TO end  
        SET node TO new ListNode with val AS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN (element val of p1 EQUALS element val of p2) AND is_same_list(element next of p1, element next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight(self, head, k)  
        IF head EQUALS None OR element next of head EQUALS None OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE element next of tail IS NOT None  
            SET tail TO element next of tail  
            INCREMENT length BY one  
        END WHILE  
        SET element next of tail TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR each _ FROM zero TO steps_to_new_tail MINUS one  
            SET new_tail TO element next of new_tail  
        END FOR  
        SET new_head TO element next of new_tail  
        SET element next of new_tail TO None  
        RETURN new_head  
    END FUNCTION  
END CLASS
