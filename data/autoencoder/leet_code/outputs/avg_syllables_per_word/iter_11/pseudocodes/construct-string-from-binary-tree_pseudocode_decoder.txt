CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value equal to element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO a new TreeNode with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO a new TreeNode with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str(self, root)  
        FUNCTION helper(node)  
            IF node EQUALS None  
                RETURN empty string  
            END IF  
            SET result TO the string representation of node.val  
            IF node.left NOT EQUALS None OR node.right NOT EQUALS None  
                SET result TO result PLUS an opening parenthesis PLUS helper(node.left) PLUS a closing parenthesis  
            END IF  
            IF node.right NOT EQUALS None  
                SET result TO result PLUS an opening parenthesis PLUS helper(node.right) PLUS a closing parenthesis  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN helper(root)  
    END FUNCTION  
END CLASS
