CLASS Solution  
    FUNCTION longestPalindrome(word1, word2)  
        SET s TO concatenation of word1 AND word2  
        SET n TO LENGTH OF s  
        SET dp TO two dimensional list with dimensions n BY n FILLED WITH zero  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AND i OF dp TO one  
        END FOR  
        
        SET max_len TO zero  
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i OF s EQUALS element at position j OF s  
                    SET element at position i AND j OF dp TO element at position i PLUS one AND j MINUS one OF dp PLUS two  
                    IF i LESS THAN LENGTH OF word1 AND j GREATER THAN OR EQUAL TO LENGTH OF word1  
                        SET max_len TO greater value BETWEEN max_len AND element at position i AND j OF dp  
                    END IF  
                ELSE  
                    SET element at position i AND j OF dp TO greater value BETWEEN element at position i PLUS one AND j OF dp AND element at position i AND j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_len  
    END FUNCTION  
END CLASS
