CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO empty set  
        FOR each pair of x and y IN corresponding elements of fronts and backs  
            IF x EQUALS y  
                ADD x TO same  
            END IF  
        END FOR  
        SET all_numbers TO set containing all elements from fronts and all elements from backs  
        SET good_numbers TO the result of all_numbers MINUS same  
        IF the LENGTH OF good_numbers IS GREATER THAN zero  
            RETURN the MINIMUM element OF good_numbers  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
