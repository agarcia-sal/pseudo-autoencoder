CLASS Solution  
    FUNCTION minSubArrayLen(target, nums)  
        SET left TO zero  
        SET current_sum TO zero  
        SET min_length TO infinity  
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            SET current_sum TO current_sum PLUS the element at position right of nums  
            WHILE current_sum GREATER THAN OR EQUAL TO target  
                IF min_length GREATER THAN right MINUS left PLUS one  
                    SET min_length TO right MINUS left PLUS one  
                END IF  
                SET current_sum TO current_sum MINUS the element at position left of nums  
                INCREMENT left BY one  
            END WHILE  
        END FOR  
        IF min_length EQUALS infinity  
            RETURN zero  
        ELSE  
            RETURN min_length  
        END IF  
    END FUNCTION  
END CLASS
