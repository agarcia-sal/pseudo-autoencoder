CLASS Solution  
    FUNCTION canSeePersonsCount(heights)  
        SET n TO the LENGTH OF heights  
        SET answer TO a new list of zeroes of length n  
        SET stack TO an empty list  
        
        FOR i FROM zero TO n MINUS one  
            WHILE stack is NOT empty AND element at position last element of stack of heights is LESS THAN element at position i of heights  
                SET index TO the last element REMOVED FROM stack  
                INCREMENT element at position index of answer BY one  
            END WHILE  
            IF stack is NOT empty  
                INCREMENT element at position last element of stack of answer BY one  
            END IF  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
