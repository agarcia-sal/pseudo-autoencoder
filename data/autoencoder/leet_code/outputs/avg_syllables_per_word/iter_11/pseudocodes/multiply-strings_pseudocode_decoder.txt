CLASS Solution  
    FUNCTION multiply(num1, num2)  
        IF num1 EQUALS string zero OR num2 EQUALS string zero  
            RETURN string zero  
        END IF  
        SET result TO a list of zero WITH the length EQUAL TO the LENGTH OF num1 PLUS the LENGTH OF num2  
        SET num1 TO the reverse order OF num1  
        SET num2 TO the reverse order OF num2  
        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the numeric value OF character at position i OF num1 MINUS the numeric value OF character zero MULTIPLIED BY the numeric value OF character at position j OF num2 MINUS the numeric value OF character zero  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS element at position p1 OF result  
                SET element at position p1 OF result TO sum_ MODULO ten  
                INCREMENT element at position p2 OF result BY sum_ DIVIDED BY ten  
            END FOR  
        END FOR  
        SET result_str TO the concatenation OF string representations OF elements OF result IN reverse order  
        RETURN result_str AFTER REMOVING leading occurrences OF character zero  
    END FUNCTION  
END CLASS
