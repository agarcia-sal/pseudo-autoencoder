CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO empty mapping  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each end FROM zero TO LENGTH OF s MINUS one  
            SET char TO element at position end of s  
            IF char EXISTS IN char_index_map AND value at key char in char_index_map GREATER THAN OR EQUAL TO start  
                SET start TO value at key char in char_index_map PLUS one  
            END IF  
            SET value at key char in char_index_map TO end  
            IF max_length LESS THAN end MINUS start PLUS one  
                SET max_length TO end MINUS start PLUS one  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
