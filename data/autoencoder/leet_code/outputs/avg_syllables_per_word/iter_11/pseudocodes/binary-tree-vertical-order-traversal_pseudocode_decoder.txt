CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero, left EQUALS none, right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode WITH val equal to element at position zero of values  
    SET i TO one  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.left TO new TreeNode WITH val equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.right TO new TreeNode WITH val equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p.left, q.left AND is_same_tree WITH PARAMETERS p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder WITH PARAMETERS root  
        IF root EQUALS none  
            RETURN empty list  
        END IF  
        SET column_table TO new defaultdict of list  
        SET queue TO new deque containing one tuple of root AND zero  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE queue is not empty  
            SET node, column_index TO first element removed FROM queue  
            IF node NOT EQUALS none  
                APPEND node.val TO value at key column_index of column_table  
                IF column_index LESS THAN min_column  
                    SET min_column TO column_index  
                END IF  
                IF column_index GREATER THAN max_column  
                    SET max_column TO column_index  
                END IF  
                APPEND tuple of node.left AND column_index MINUS one TO queue  
                APPEND tuple of node.right AND column_index PLUS one TO queue  
            END IF  
        END WHILE  
        SET result TO empty list  
        FOR x FROM min_column TO max_column  
            APPEND value at key x of column_table TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
