CLASS ListNode  
    FUNCTION __init__ (val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node (values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode with val equal to the element at position zero of values  
    SET p TO head  
    FOR each val IN values starting from position one to the end  
        SET node TO a new ListNode with val equal to val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list (p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the element val of p1 EQUALS the element val of p2 AND is_same_list with arguments element next of p1 AND element next of p2  
END FUNCTION

CLASS Solution  
    FUNCTION plusOne (head)  
        FUNCTION reverseList (node)  
            SET prev TO None  
            WHILE node IS NOT None  
                SET next_node TO element next of node  
                SET element next of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION

        SET reversed_head TO reverseList with argument head  
        SET carry TO one  
        SET current TO reversed_head  
        WHILE current IS NOT None AND carry EQUALS one  
            SET element val of current TO element val of current PLUS carry  
            SET carry TO element val of current DIVIDED BY ten  
            SET element val of current TO element val of current MODULO ten  
            IF element next of current EQUALS None AND carry EQUALS one  
                SET element next of current TO a new ListNode with val equal to carry  
                SET carry TO zero  
            END IF  
            SET prev TO current  
            SET current TO element next of current  
        END WHILE  
        RETURN reverseList with argument reversed_head  
    END FUNCTION  
END CLASS
