CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the SET OF characters IN time REPLACING colon WITH empty string  
        SET hour AND minute TO the INTEGER VALUES FROM splitting time BY colon  
        
        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each h1 IN digits  
                FOR each h2 IN digits  
                    FOR each m1 IN digits  
                        FOR each m2 IN digits  
                            SET hour TO the INTEGER VALUE OF concatenating h1 AND h2  
                            SET minute TO the INTEGER VALUE OF concatenating m1 AND m2  
                            IF hour LESS THAN 24 AND minute LESS THAN 60  
                                APPEND the PAIR hour AND minute TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the sorted list times  
        END FUNCTION  
        
        SET all_times TO generate_times(digits)  
        
        SET current_time_index TO the POSITION OF the PAIR hour AND minute IN all_times  
        
        IF current_time_index PLUS one LESS THAN the LENGTH OF all_times  
            SET next_hour AND next_minute TO the ELEMENT AT POSITION current_time_index PLUS one IN all_times  
        ELSE  
            SET next_hour AND next_minute TO the FIRST ELEMENT IN all_times  
        END IF  
        
        RETURN the STRING REPRESENTATION OF next_hour FORMATTED AS two digits CONCATENATED WITH colon CONCATENATED WITH next_minute FORMATTED AS two digits  
    END FUNCTION  
END CLASS
