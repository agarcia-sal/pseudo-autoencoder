CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        IF the SUM of m AND n MINUS one MODULO two NOT EQUALS zero  
            RETURN False  
        END IF  
        
        FUNCTION initialize_dp()  
            CREATE a three-dimensional list dp WITH dimensions (m PLUS one) BY (n PLUS one) BY (m PLUS n PLUS one) FILLED WITH False  
            RETURN dp  
        END FUNCTION  
        
        SET dp TO the RESULT OF initialize_dp()  
        
        SET the element at position zero, one, zero of dp TO True  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR k FROM zero TO m PLUS n  
                    IF the element at position i, j PLUS one, k of dp OR the element at position i PLUS one, j, k of dp IS True  
                        IF the element at position i, j of grid EQUALS open parenthesis AND k PLUS one LESS THAN m PLUS n PLUS one  
                            SET the element at position i PLUS one, j PLUS one, k PLUS one of dp TO True  
                        ELSE IF the element at position i, j of grid EQUALS close parenthesis AND k MINUS one GREATER THAN OR EQUAL TO zero  
                            SET the element at position i PLUS one, j PLUS one, k MINUS one of dp TO True  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN the element at position m, n, zero of dp  
    END FUNCTION  
END CLASS
