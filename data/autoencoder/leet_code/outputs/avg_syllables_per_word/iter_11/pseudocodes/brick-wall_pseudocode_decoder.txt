CLASS Solution  
    FUNCTION leastBricks(wall)  
        CALL initialize gap_counts AS a mapping FROM integer TO integer WITH default zero  
        
        FOR each row IN wall  
            SET total TO zero  
            FOR each brick IN all elements of row EXCEPT the last element  
                SET total TO total PLUS brick  
                INCREMENT the value at key total IN gap_counts BY one  
            END FOR  
        END FOR  
        
        SET max_gaps TO the maximum value AMONG all values IN gap_counts WITH default zero  
        
        RETURN the length of wall MINUS max_gaps  
    END FUNCTION  
END CLASS
