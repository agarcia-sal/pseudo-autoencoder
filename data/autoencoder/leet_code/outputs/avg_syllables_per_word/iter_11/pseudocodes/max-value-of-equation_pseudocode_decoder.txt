CLASS Solution  
    FUNCTION findMaxValueOfEquation(points, k)  
        SET max_value TO negative infinity  
        SET deque_ TO empty double ended queue  
        
        FOR each pair xj and yj IN points  
            WHILE deque_ is not empty AND xj MINUS element at position zero and one of deque_ GREATER THAN k  
                REMOVE first element FROM deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO maximum of max_value AND yj PLUS xj PLUS element at position zero and zero of deque_  
            END IF  
            
            WHILE deque_ is not empty AND yj MINUS xj GREATER THAN OR EQUAL TO element at last position and zero of deque_  
                REMOVE last element FROM deque_  
            END WHILE  
            
            APPEND pair of yj MINUS xj AND xj TO deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
