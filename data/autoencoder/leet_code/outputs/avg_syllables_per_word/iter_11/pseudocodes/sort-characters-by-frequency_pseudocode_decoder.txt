CLASS Solution  
    FUNCTION frequencySort(s)  
        SET frequency TO the result of calling Counter with s  
        SET sorted_characters TO the list of keys in frequency sorted in such a way that for each character x, the frequency of x is considered as the sorting key in descending order  
        SET result TO empty string  
        FOR each character IN sorted_characters  
            APPEND character REPEATED frequency of character TIMES TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
