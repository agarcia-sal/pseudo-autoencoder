CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF calling isOneEditDistance WITH arguments t AND s  
        END IF  
  
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
  
        IF diff GREATER THAN one  
            RETURN False  
        END IF  
  
        IF diff EQUALS zero  
            SET count_of_differences TO zero  
            FOR each pair of characters c1 AND c2 AT the SAME POSITION IN s AND t  
                IF c1 NOT EQUALS c2  
                    INCREMENT count_of_differences BY one  
                END IF  
            END FOR  
            RETURN whether count_of_differences EQUALS one  
        ELSE  
            FOR index FROM zero TO the LENGTH OF s MINUS one  
                IF the CHARACTER at position index of s NOT EQUALS the CHARACTER at position index of t  
                    RETURN whether the SUBSTRING OF s FROM position index TO END EQUALS the SUBSTRING OF t FROM position index PLUS one TO END  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
