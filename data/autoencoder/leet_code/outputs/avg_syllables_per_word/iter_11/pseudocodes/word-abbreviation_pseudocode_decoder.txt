CLASS Solution  
    FUNCTION wordsAbbreviation(words)  
        FUNCTION abbreviate(word, prefix_length)  
            IF length of word MINUS prefix_length LESS THAN OR EQUAL TO two  
                RETURN word  
            END IF  
            RETURN concatenation of substring of word from start TO prefix_length, string representation of length of word MINUS prefix_length MINUS one, and last character of word  
        END FUNCTION  

        FUNCTION get_unique_abbreviations(words_with_indices, prefix_length)  
            SET abbs TO new default dictionary with list as default value  
            FOR each index, word IN words_with_indices  
                APPEND tuple of index and word TO abbs at key obtained by calling abbreviate with word and prefix_length  
            END FOR  
            RETURN abbs  
        END FUNCTION  

        SET n TO length of words  
        SET ans TO list of empty strings repeated n times  
        SET words_with_indices TO list of tuples of each index and word by enumerating words  

        SET abbs TO result of calling get_unique_abbreviations with words_with_indices and one  

        FOR each abb, group IN abbs items  
            IF length of group EQUALS one  
                SET index, word TO first element of group  
                SET element at position index of ans TO abb  
            ELSE  
                SET new_group TO empty list  
                FOR each index, word IN group  
                    APPEND tuple of index and word TO new_group  
                END FOR  
                SET prefix_length TO two  
                WHILE length of new_group GREATER THAN zero  
                    SET new_abbs TO result of calling get_unique_abbreviations with new_group and prefix_length  
                    SET new_group TO empty list  
                    FOR each new_abb, new_group_items IN new_abbs items  
                        IF length of new_group_items EQUALS one  
                            SET index, word TO first element of new_group_items  
                            SET element at position index of ans TO new_abb  
                        ELSE  
                            EXTEND new_group BY new_group_items  
                        END IF  
                    END FOR  
                    INCREMENT prefix_length BY one  
                END WHILE  
            END IF  
        END FOR  

        RETURN ans  
    END FUNCTION  
END CLASS
