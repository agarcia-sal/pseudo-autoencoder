CLASS Solution  
    FUNCTION countBattleships(board)  
        SET count TO zero  
        FOR i FROM zero TO the LENGTH OF board MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of board MINUS one  
                IF the element at position i of board at position j EQUALS the character X  
                    IF (i EQUALS zero OR the element at position i MINUS one of board at position j EQUALS the character dot) AND (j EQUALS zero OR the element at position i of board at position j MINUS one EQUALS the character dot)  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
