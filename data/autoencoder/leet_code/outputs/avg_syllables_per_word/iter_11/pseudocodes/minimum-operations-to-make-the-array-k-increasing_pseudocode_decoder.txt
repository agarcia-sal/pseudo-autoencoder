CLASS Solution  
    FUNCTION kIncreasing(arr, k)  
        FUNCTION longest_increasing_subsequence(subarr)  
            SET lis TO an empty list  
            FOR each num IN subarr  
                IF lis IS empty OR num GREATER THAN OR EQUAL TO the last element of lis  
                    APPEND num TO lis  
                ELSE  
                    SET pos TO the position where num can replace the first number greater than num in lis  
                    SET the element at position pos of lis TO num  
                END IF  
            END FOR  
            RETURN the length of lis  
        END FUNCTION  
        
        SET total_operations TO zero  
        SET n TO the length of arr  
        
        FOR start FROM zero TO k MINUS one  
            SET subarr TO an empty list  
            FOR i FROM start TO n MINUS one IN STEPS OF k  
                APPEND the element at position i of arr TO subarr  
            END FOR  
            
            SET lis_length TO longest_increasing_subsequence(subarr)  
            INCREMENT total_operations BY the length of subarr MINUS lis_length  
        END FOR  
        
        RETURN total_operations  
    END FUNCTION  
END CLASS
