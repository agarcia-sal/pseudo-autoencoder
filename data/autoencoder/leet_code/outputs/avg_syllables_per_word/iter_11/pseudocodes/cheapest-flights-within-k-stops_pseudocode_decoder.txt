CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO an empty dictionary mapping each integer from zero TO n MINUS one TO an empty list  
        FOR each element u, v, w IN flights  
            APPEND the pair of v AND w TO the list at key u OF graph  
        END FOR  
        
        SET min_heap TO a list containing the triple zero, MINUS one, AND src  
        SET min_cost TO a dictionary with key src mapped TO MINUS one  
        
        WHILE min_heap is not empty  
            EXTRACT from min_heap the element with the smallest first value INTO cost, stops, city  
            
            IF city EQUALS dst  
                RETURN cost  
            END IF  
            
            IF stops GREATER THAN OR EQUAL TO k  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            
            FOR each pair neighbor, price IN the list at key city OF graph  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  
                
                IF neighbor NOT IN min_cost OR next_stops LESS THAN min_cost at key neighbor OR next_cost LESS THAN the first element of the first tuple in min_heap  
                    SET min_cost at key neighbor TO next_stops  
                    INSERT the triple next_cost, next_stops, neighbor INTO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN MINUS one  
    END FUNCTION  
END CLASS
