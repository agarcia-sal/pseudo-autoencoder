CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        CALL initialize sequence_count AS a dictionary with default integer zero  
        FOR index FROM zero TO the LENGTH OF s MINUS ten PLUS one  
            SET sequence TO the substring of s STARTING AT position index FOR ten characters  
            INCREMENT the value in sequence_count at key sequence BY one  
        END FOR  
        CALL initialize repeated_sequences AS an empty list  
        FOR each sequence AND count IN sequence_count  
            IF count GREATER THAN one  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
