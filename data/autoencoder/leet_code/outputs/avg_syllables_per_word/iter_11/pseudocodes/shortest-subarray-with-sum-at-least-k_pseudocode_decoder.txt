CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO list containing zero  
        FOR each num IN nums  
            APPEND sum of last element of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        INITIALIZE dq AS empty double ended queue  
        SET min_length TO positive infinity  
        FOR each index i, element current_sum IN enumeration of prefix_sum  
            WHILE dq is not empty AND difference of current_sum MINUS element at position front of dq in prefix_sum is GREATER THAN OR EQUAL TO k  
                SET min_length TO smaller of min_length AND difference of i MINUS element removed from front of dq  
            END WHILE  
            WHILE dq is not empty AND current_sum is LESS THAN OR EQUAL TO element at position back of dq in prefix_sum  
                REMOVE last element FROM dq  
            END WHILE  
            APPEND i TO dq  
        END FOR  
        IF min_length is NOT EQUAL TO positive infinity  
            RETURN min_length  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
