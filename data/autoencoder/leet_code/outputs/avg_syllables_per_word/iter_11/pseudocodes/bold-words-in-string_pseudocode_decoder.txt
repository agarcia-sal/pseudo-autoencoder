CLASS Solution  
    FUNCTION boldWords(words, s)  
        SET bold TO a list of False values of the same length as s  
        
        FOR each word IN words  
            SET start TO the position of word in s  
            WHILE start is not equal to minus one  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i of bold TO True  
                END FOR  
                SET start TO the position of word in s starting from start PLUS one  
            END WHILE  
        END FOR  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i is less than the LENGTH OF s  
            IF the element at position i of bold is True  
                APPEND the string less than character b greater than character TO result  
                WHILE i is less than the LENGTH OF s AND the element at position i of bold is True  
                    APPEND the element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string less than character slash character b greater than character TO result  
            ELSE  
                APPEND the element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
