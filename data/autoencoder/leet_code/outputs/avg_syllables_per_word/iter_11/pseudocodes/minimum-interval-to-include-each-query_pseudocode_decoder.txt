CLASS Solution  
    FUNCTION minInterval(intervals, queries)  
        SORT intervals IN ASCENDING ORDER BY element at position zero of each interval  
        SET sorted_queries TO a list of pairs of each element and its position in queries sorted by the element  
        
        SET min_heap TO an empty list  
        SET result TO a list containing negative one repeated the LENGTH OF queries times  
        
        SET interval_index TO zero  
        FOR each query_and_index IN sorted_queries  
            SET query TO the first element of query_and_index  
            SET original_index TO the second element of query_and_index  
            
            WHILE interval_index LESS THAN the LENGTH OF intervals AND the element at position zero of the interval at position interval_index LESS THAN OR EQUAL TO query  
                SET start TO the element at position zero of the interval at position interval_index  
                SET end TO the element at position one of the interval at position interval_index  
                SET interval_size TO end MINUS start PLUS one  
                ADD the pair of interval_size and end TO min_heap using heap push operation  
                INCREMENT interval_index BY one  
            END WHILE  
            
            WHILE the LENGTH OF min_heap IS GREATER THAN zero AND the second element of the first item in min_heap LESS THAN query  
                REMOVE the first item from min_heap using heap pop operation  
            END WHILE  
            
            IF the LENGTH OF min_heap IS GREATER THAN zero  
                SET the element at position original_index of result TO the first element of the first item in min_heap  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
