CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO result of calling initialize_defaultdict_with_int()  
        FOR each i IN nums  
            FOR each j IN nums  
                SET current_and TO bitwise AND of i AND j  
                INCREMENT value at key current_and IN pair_and_count BY one  
            END FOR  
        END FOR  
        
        SET triplet_count TO zero  
        FOR each k IN nums  
            FOR each pair_and AND count IN items of pair_and_count  
                IF bitwise AND of pair_and AND k EQUALS zero  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        
        RETURN triplet_count  
    END FUNCTION  
END CLASS
