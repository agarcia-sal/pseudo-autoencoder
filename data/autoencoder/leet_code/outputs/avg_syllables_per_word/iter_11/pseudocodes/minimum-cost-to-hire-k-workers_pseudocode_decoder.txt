CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO list of pairs of wage at position index DIVIDED BY quality at position index and quality at position index FOR each index FROM zero TO LENGTH OF quality MINUS one SORTED ascending BY first element  
        SET max_heap TO empty list  
        SET total_quality TO zero  
        SET min_cost TO infinity  
        FOR each pair ratio, q IN workers  
            PUSH negative of q INTO max_heap  
            INCREMENT total_quality BY q  
            IF LENGTH OF max_heap GREATER THAN k  
                SET total_quality TO total_quality PLUS value popped from max_heap  
            END IF  
            IF LENGTH OF max_heap EQUALS k  
                SET current_cost TO total_quality MULTIPLIED BY ratio  
                IF current_cost LESS THAN min_cost  
                    SET min_cost TO current_cost  
                END IF  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
