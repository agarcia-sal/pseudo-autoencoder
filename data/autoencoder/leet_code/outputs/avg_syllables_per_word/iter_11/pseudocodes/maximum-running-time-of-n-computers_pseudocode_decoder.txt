CLASS Solution  
    FUNCTION maxRunTime(n, batteries)  
        FUNCTION canRunFor(time)  
            SET total TO zero  
            FOR each battery IN batteries  
                INCREMENT total BY the smaller OF battery AND time  
            END FOR  
            RETURN total GREATER THAN OR EQUAL TO time MULTIPLIED BY n  
        END FUNCTION  

        SET left TO zero  
        SET right TO the sum OF batteries DIVIDED BY n  
        WHILE left LESS THAN right  
            SET mid TO the sum OF left AND right AND one DIVIDED BY two  
            IF canRunFor(mid)  
                SET left TO mid  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
