CLASS Solution  
    FUNCTION combinationSum2(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                IF i GREATER THAN start AND element at position i of candidates EQUALS element at position i MINUS one of candidates  
                    CONTINUE TO next iteration  
                END IF  
                CALL backtrack WITH parameters i PLUS one, target MINUS element at position i of candidates, path PLUS element at position i of candidates  
            END FOR  
        END FUNCTION  
        CALL sort ON candidates  
        SET result TO an empty list  
        CALL backtrack WITH parameters zero, target, an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
