CLASS Solution  
    FUNCTION ipToCIDR(ip, n)  
        FUNCTION ip_to_int(ip)  
            SET parts TO list of integer conversions of splitting ip by period  
            SET res TO zero  
            FOR each part IN parts  
                SET res TO res MULTIPLIED BY two HUNDRED FIFTY SIX PLUS part  
            END FOR  
            RETURN res  
        END FUNCTION  

        FUNCTION int_to_ip(num)  
            SET list_of_strings TO empty list  
            FOR i FROM three DOWN TO zero  
                APPEND string conversion of num DIVIDED BY two HUNDRED FIFTY SIX TO THE POWER OF i MODULO two HUNDRED FIFTY SIX TO list_of_strings  
            END FOR  
            RETURN joining elements of list_of_strings by period  
        END FUNCTION  

        FUNCTION largest_cidr_size(start, n)  
            IF start MODULO two EQUALS one  
                RETURN one  
            END IF  
            SET x TO one  
            WHILE x LESS THAN OR EQUAL TO n AND start BITWISE AND (x MINUS one) EQUALS zero  
                SET x TO x MULTIPLIED BY two  
            END WHILE  
            RETURN x DIVIDED BY two  
        END FUNCTION  

        SET result TO empty list  
        SET start_ip TO ip_to_int(ip)  

        WHILE n GREATER THAN zero  
            SET size TO largest_cidr_size(start_ip, n)  
            APPEND concatenation of int_to_ip(start_ip), forward slash, thirty TWO MINUS (bit length of size MINUS one) TO result  
            SET start_ip TO start_ip PLUS size  
            SET n TO n MINUS size  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
