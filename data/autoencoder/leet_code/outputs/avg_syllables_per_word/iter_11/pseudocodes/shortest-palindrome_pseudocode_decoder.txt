CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        END IF  
  
        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS the reverse of check_str  
        END FUNCTION  
  
        FOR i FROM the LENGTH OF s DOWN TO zero  
            IF is_palindrome(the substring of s from start TO position i MINUS one)  
                SET to_add TO the reverse of the substring of s from position i TO the end  
                RETURN to_add CONCATENATED WITH s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
