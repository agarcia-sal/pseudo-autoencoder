CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            WHILE the element at position i of nums NOT EQUALS the element at position the element at position i of nums MINUS one of nums  
                SET temp TO the element at position i of nums  
                SET the element at position i of nums TO the element at position temp MINUS one of nums  
                SET the element at position temp MINUS one of nums TO temp  
            END WHILE  
        END FOR  
        SET result_list TO empty list  
        FOR each index and value IN enumeration of nums  
            IF value NOT EQUALS index PLUS one  
                APPEND value TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
