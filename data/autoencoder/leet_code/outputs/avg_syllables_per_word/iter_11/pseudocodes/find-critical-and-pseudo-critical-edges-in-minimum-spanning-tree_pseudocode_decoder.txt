CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO the list of numbers from zero TO n MINUS one  
        SET self.rank TO a list containing the number one repeated n times  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF the element at position u of self.parent NOT EQUALS u  
            SET the element at position u of self.parent TO the result of self.find with the element at position u of self.parent  
        END IF  
        RETURN the element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET rootU TO the result of self.find with u  
        SET rootV TO the result of self.find with v  
        
        IF rootU NOT EQUALS rootV  
            IF the element at position rootU of self.rank GREATER THAN the element at position rootV of self.rank  
                SET the element at position rootV of self.parent TO rootU  
            ELSE IF the element at position rootU of self.rank LESS THAN the element at position rootV of self.rank  
                SET the element at position rootU of self.parent TO rootV  
            ELSE  
                SET the element at position rootV of self.parent TO rootU  
                INCREMENT the element at position rootU of self.rank BY one  
            END IF  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)  
        FOR each index i AND element edge IN the enumeration of edges  
            APPEND i TO edge  
        END FOR  
        
        SORT edges based on the element at position two of each edge  
        
        FUNCTION mst(exclude NONE, include NONE)  
            SET uf TO a new UnionFind object constructed with n  
            SET weight TO zero  
            
            IF include NOT EQUALS None  
                CALL uf.union with the element at position zero of include AND the element at position one of include  
                INCREMENT weight BY the element at position two of include  
            END IF  
            
            FOR each u, v, w, idx IN edges  
                IF idx EQUALS exclude  
                    CONTINUE to next iteration  
                END IF  
                IF the result of uf.union with u AND v IS True  
                    INCREMENT weight BY w  
                END IF  
            END FOR  
            
            SET root TO the result of uf.find with zero  
            FOR i FROM zero TO n MINUS one  
                IF the result of uf.find with i NOT EQUALS root  
                    RETURN positive infinity  
                END IF  
            END FOR  
            
            RETURN weight  
        END FUNCTION  
        
        SET mst_weight TO the result of mst  
        
        SET critical TO an empty list  
        SET pseudo_critical TO an empty list  
        
        FOR each u, v, w, idx IN edges  
            IF the result of mst with exclude equal to idx GREATER THAN mst_weight  
                APPEND idx TO critical  
            ELSE IF the result of mst with include equal to the tuple of u, v, w EQUALS mst_weight  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR  
        
        RETURN the list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
