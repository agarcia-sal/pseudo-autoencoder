CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO list of one thousand, nine hundred, five hundred, four hundred, one hundred, ninety, fifty, forty, ten, nine, five, four, one  
        SET syms TO list of character M, character CM, character D, character CD, character C, character XC, character L, character XL, character X, character IX, character V, character IV, character I  
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero  
            FOR each underscore FROM one TO num DIVIDED BY element at position i of val  
                APPEND element at position i of syms TO roman_numeral  
                DECREMENT num BY element at position i of val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
