CLASS Solution  
    FUNCTION ladderLength(beginWord, endWord, wordList)  
        IF endWord NOT IN wordList  
            RETURN zero  
        END IF  
        
        SET word_dict TO result of calling helper FUNCTION create default dictionary mapping to empty list  
        SET word_length TO LENGTH OF beginWord  
        
        FOR each word IN wordList  
            FOR index FROM zero TO word_length MINUS one  
                SET intermediate_key TO concatenation of substring of word from start TO index MINUS one PLUS character star PLUS substring of word from index PLUS one TO end  
                APPEND word TO list at intermediate_key in word_dict  
            END FOR  
        END FOR  
        
        SET queue TO result of calling helper FUNCTION create deque containing tuple of beginWord and one  
        SET visited TO set containing beginWord  
        
        WHILE queue IS NOT empty  
            SET current_word and current_length TO result of removing first element FROM queue  
            
            FOR index FROM zero TO word_length MINUS one  
                SET intermediate_word TO concatenation of substring of current_word from start TO index MINUS one PLUS character star PLUS substring of current_word from index PLUS one TO end  
                
                FOR each word IN list at intermediate_word in word_dict  
                    IF word EQUALS endWord  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited  
                        ADD word TO visited  
                        APPEND tuple of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN zero  
    END FUNCTION  
END CLASS
