CLASS Solution  
    FUNCTION isValid(code)  
        IMPORT regular expression as re  
        
        SET cdata_pattern TO string representing CDATA regular expression  
        
        SET code TO result of replacing all substrings matching cdata_pattern in code with empty string  
        
        SET stack TO empty list  
        
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF code  
            IF element at position i of code EQUALS the character less than  
                SET j TO position of the character greater than in code starting from position i PLUS one  
                IF j EQUALS minus one  
                    RETURN False  
                END IF  
                
                SET tag_content TO substring of code from position i PLUS one TO position j MINUS one  
                
                IF tag_content IS empty  
                    RETURN False  
                END IF  
                
                IF first character of tag_content EQUALS the character slash  
                    SET tag_name TO substring of tag_content from position one TO the end  
                    IF stack IS empty OR last element of stack NOT EQUALS tag_name  
                        RETURN False  
                    END IF  
                    REMOVE last element FROM stack  
                ELSE IF first character of tag_content EQUALS the character exclamation mark  
                    RETURN False  
                ELSE  
                    IF the LENGTH OF tag_content LESS THAN one OR the LENGTH OF tag_content GREATER THAN nine OR tag_content IS NOT composed entirely of uppercase letters  
                        RETURN False  
                    END IF  
                    APPEND tag_content TO stack  
                END IF  
                
                SET i TO j PLUS one  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        SET valid_tags_count TO count of substrings in code that match pattern of opening tag enclosing uppercase letters followed by closing tag of the same letters  
        
        RETURN True IF stack IS empty AND valid_tags_count GREATER THAN zero ELSE False  
    END FUNCTION  
END CLASS
