CLASS Solution  
    FUNCTION new21Game(n, k, maxPts)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k  
                RETURN integer value of (i LESS THAN OR EQUAL TO n)  
            END IF  
            IF i EQUALS k MINUS one  
                RETURN the MINIMUM OF (n MINUS k PLUS one) AND maxPts DIVIDED BY maxPts  
            END IF  
            SET result TO dfs(i PLUS one) PLUS (dfs(i PLUS one) MINUS dfs(i PLUS maxPts PLUS one)) DIVIDED BY maxPts  
            RETURN result  
        END FUNCTION  
        RETURN dfs(zero)  
END CLASS
