CLASS Solution  
    FUNCTION isSolvable(words, result)  
        SET unique_chars TO the SET OF all characters IN the CONCATENATION OF words AND result  
        IF the LENGTH OF unique_chars IS GREATER THAN ten  
            RETURN False  
        END IF  
        
        SET non_zero_chars TO the SET OF first character OF each word IN words WHERE the LENGTH OF word IS GREATER THAN one  
        IF the LENGTH OF result IS GREATER THAN one  
            ADD first character OF result TO non_zero_chars  
        END IF  
        
        SET char_to_digit TO an EMPTY dictionary  
        SET digit_to_char TO an EMPTY dictionary  
        
        FUNCTION can_assign(char, digit)  
            RETURN char NOT IN char_to_digit AND (digit NOT IN digit_to_char OR digit_to_char[digit] EQUALS char)  
        END FUNCTION  
        
        FUNCTION assign(char, digit)  
            SET char_to_digit[char] TO digit  
            SET digit_to_char[digit] TO char  
        END FUNCTION  
        
        FUNCTION unassign(char, digit)  
            REMOVE char FROM char_to_digit  
            REMOVE digit FROM digit_to_char  
        END FUNCTION  
        
        FUNCTION word_value(word)  
            SET total_value TO zero  
            FOR each position i FROM zero TO the LENGTH OF word MINUS one  
                SET char TO the character AT position the LENGTH OF word MINUS one MINUS i OF word  
                INCREMENT total_value BY char_to_digit[char] MULTIPLIED BY ten RAISED TO THE power OF i  
            END FOR  
            RETURN total_value  
        END FUNCTION  
        
        FUNCTION backtrack(index, col, sum_col)  
            IF col IS GREATER THAN OR EQUAL TO the MAXIMUM LENGTH OF words AND col IS GREATER THAN OR EQUAL TO the LENGTH OF result  
                RETURN sum_col EQUALS zero  
            END IF  
            
            IF index EQUALS the LENGTH OF words  
                IF col IS LESS THAN the LENGTH OF result  
                    SET digit TO the character AT position the LENGTH OF result MINUS one MINUS col OF result  
                    IF digit IS IN char_to_digit  
                        IF char_to_digit[digit] EQUALS sum_col MODULO ten  
                            RETURN backtrack(zero, col PLUS one, sum_col INTEGER DIVIDED BY ten)  
                        ELSE  
                            RETURN False  
                        END IF  
                    ELSE IF can_assign(digit, sum_col MODULO ten) AND (sum_col MODULO ten NOT EQUALS zero OR digit NOT IN non_zero_chars)  
                        CALL assign(digit, sum_col MODULO ten)  
                        IF backtrack(zero, col PLUS one, sum_col INTEGER DIVIDED BY ten) RETURNS True  
                            RETURN True  
                        END IF  
                        CALL unassign(digit, sum_col MODULO ten)  
                    END IF  
                ELSE  
                    RETURN sum_col EQUALS zero  
                END IF  
                RETURN False  
            END IF  
            
            SET word TO the element AT position index OF words  
            IF col IS GREATER THAN OR EQUAL TO the LENGTH OF word  
                RETURN backtrack(index PLUS one, col, sum_col)  
            END IF  
            
            SET char TO the character AT position the LENGTH OF word MINUS one MINUS col OF word  
            IF char IS IN char_to_digit  
                RETURN backtrack(index PLUS one, col, sum_col PLUS char_to_digit[char])  
            END IF  
            
            FOR digit FROM zero TO nine  
                IF can_assign(char, digit) AND (digit NOT EQUALS zero OR char NOT IN non_zero_chars)  
                    CALL assign(char, digit)  
                    IF backtrack(index PLUS one, col, sum_col PLUS digit) RETURNS True  
                        RETURN True  
                    END IF  
                    CALL unassign(char, digit)  
                END IF  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        RETURN backtrack(zero, zero, zero)  
    END FUNCTION  
END CLASS
