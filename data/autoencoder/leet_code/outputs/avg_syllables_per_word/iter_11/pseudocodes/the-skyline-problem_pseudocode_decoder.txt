CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO empty list  
        FOR each building IN buildings  
            SET left TO element at position zero of building  
            SET right TO element at position one of building  
            SET height TO element at position two of building  
            APPEND tuple of left, negative height, right TO events  
            APPEND tuple of right, zero, zero TO events  
        END FOR  
        SORT events in ascending order by first element then second element then third element  
        SET result TO list containing list of zero and zero  
        SET max_heap TO list containing tuple of zero and positive infinity  
        FOR each tuple x, negH, r IN events  
            WHILE first element of first tuple in max_heap at position one IS LESS THAN OR EQUAL TO x  
                REMOVE first element FROM max_heap using heap removal of smallest element  
            END WHILE  
            IF negH IS NOT EQUAL TO zero  
                ADD tuple of negH and r TO max_heap using heap addition  
            END IF  
            IF second element of last element in result IS NOT EQUAL TO negative of first element of first tuple in max_heap at position zero  
                APPEND list of x and negative of first element of first tuple in max_heap at position zero TO result  
            END IF  
        END FOR  
        IF first element of result IS equal to list of zero and zero  
            REMOVE first element FROM result  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
