CLASS Solution  
    FUNCTION robotSim(commands, obstacles)  
        SET directions TO list of tuples zero and one and one and zero and zero and negative one and negative one and zero representing North East South West directions  
        SET x TO zero  
        SET y TO zero  
        SET dir_index TO zero  
        SET obstacles_set TO a set containing each tuple converted from elements of obstacles  
        SET max_distance TO zero  
        
        FOR each command IN commands  
            IF command EQUALS negative two  
                SET dir_index TO the remainder of dir_index MINUS one DIVIDED BY four  
            ELSE IF command EQUALS negative one  
                SET dir_index TO the remainder of dir_index PLUS one DIVIDED BY four  
            ELSE  
                SET dx TO the element at position dir_index of directions at position zero  
                SET dy TO the element at position dir_index of directions at position one  
                FOR index FROM one TO command  
                    SET next_x TO x PLUS dx  
                    SET next_y TO y PLUS dy  
                    IF the tuple next_x and next_y IS IN obstacles_set  
                        BREAK  
                    END IF  
                    SET x TO next_x  
                    SET y TO next_y  
                END FOR  
                SET max_distance TO the maximum of max_distance AND the sum of x MULTIPLIED BY x PLUS y MULTIPLIED BY y  
            END IF  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
