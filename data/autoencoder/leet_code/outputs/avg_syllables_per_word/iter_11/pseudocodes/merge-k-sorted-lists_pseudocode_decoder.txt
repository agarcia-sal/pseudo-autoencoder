CLASS ListNode  
    FUNCTION __init__ with parameters val, next  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode with parameter val equal to the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values from position one TO the end  
        SET node TO a new ListNode with parameter val equal to val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1, p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the element at attribute val of p1 EQUALS the element at attribute val of p2 AND is_same_list with parameters the element at attribute next of p1, the element at attribute next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists with parameter lists  
        SET min_heap TO an empty list  
        FOR each i, lst IN enumeration of lists  
            IF lst NOT EQUALS None  
                INSERT a tuple of the element at attribute val of lst, i, lst INTO min_heap maintaining min-heap property  
            END IF  
        END FOR  
        SET dummy TO a new ListNode with default parameters  
        SET current TO dummy  
        WHILE the LENGTH OF min_heap IS GREATER THAN zero  
            EXTRACT the minimum element FROM min_heap AND DESTRUCTURE it INTO val, i, node  
            SET current.next TO node  
            SET current TO the element at attribute next of current  
            IF the element at attribute next of node NOT EQUALS None  
                INSERT a tuple of the element at attribute val of the element at attribute next of node, i, the element at attribute next of node INTO min_heap maintaining min-heap property  
            END IF  
        END WHILE  
        RETURN the element at attribute next of dummy  
    END FUNCTION  
END CLASS
