CLASS Solution  
    FUNCTION maxStudents(seats)  
        SET m TO the LENGTH OF seats  
        SET n TO the LENGTH OF the element at position zero of seats  
        
        SET broken TO a list of zeros with LENGTH m  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of seats EQUALS the character hash  
                    SET the element at position i of broken TO the bitwise OR of the element at position i of broken and one LEFT SHIFTED BY j  
                END IF  
            END FOR  
        END FOR  
        
        FUNCTION is_valid(arrangement)  
            RETURN the bitwise AND of arrangement and the result of arrangement RIGHT SHIFTED BY one EQUALS zero  
        END FUNCTION  
        
        SET dp TO a list of empty dictionaries with LENGTH m  
        
        FOR arrangement FROM zero TO one LEFT SHIFTED BY n MINUS one  
            IF is_valid(arrangement) AND the bitwise AND of arrangement and the element at position zero of broken EQUALS zero  
                SET the value at key arrangement of the element at position zero of dp TO the COUNT OF ones in the binary representation of arrangement  
            END IF  
        END FOR  
        
        FOR index i FROM one TO m MINUS one  
            FOR arrangement FROM zero TO one LEFT SHIFTED BY n MINUS one  
                IF is_valid(arrangement) AND the bitwise AND of arrangement and the element at position i of broken EQUALS zero  
                    SET max_students TO zero  
                    FOR each prev_arrangement IN the keys of the element at position i MINUS one of dp  
                        IF the bitwise AND of arrangement and the result of prev_arrangement RIGHT SHIFTED BY one EQUALS zero AND the bitwise AND of arrangement and the result of prev_arrangement LEFT SHIFTED BY one EQUALS zero  
                            IF the value at key prev_arrangement of the element at position i MINUS one of dp GREATER THAN max_students  
                                SET max_students TO the value at key prev_arrangement of the element at position i MINUS one of dp  
                            END IF  
                        END IF  
                    END FOR  
                    SET the value at key arrangement of the element at position i of dp TO max_students PLUS the COUNT OF ones in the binary representation of arrangement  
                END IF  
            END FOR  
        END FOR  
        
        IF the element at position m MINUS one of dp IS NOT empty  
            RETURN the maximum value among values of the element at position m MINUS one of dp  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
