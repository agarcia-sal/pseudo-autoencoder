CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO an empty list  
        FOR each num IN target  
            APPEND the negative of num TO max_heap  
        END FOR  
        CALL heapify on max_heap to arrange it as a heap  
        SET total_sum TO the sum of target  
        WHILE the element at position zero of max_heap IS LESS THAN the negative of one  
            SET largest TO the negative of the element removed from max_heap by heappop  
            SET rest_sum TO total_sum MINUS largest  
            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero  
                RETURN False  
            END IF  
            SET new_value TO largest MODULO rest_sum  
            IF new_value EQUALS zero AND rest_sum NOT EQUALS one  
                RETURN False  
            END IF  
            SET total_sum TO rest_sum PLUS new_value  
            PUSH the negative of new_value INTO max_heap using heappush  
        END WHILE  
        RETURN True  
    END FUNCTION  
END CLASS
