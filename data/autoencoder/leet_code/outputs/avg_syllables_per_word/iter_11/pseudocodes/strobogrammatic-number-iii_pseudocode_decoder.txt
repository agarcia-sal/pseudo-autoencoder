CLASS Solution
    FUNCTION strobogrammaticInRange(low, high)
        FUNCTION generate_strobogrammatic(n, final_length)
            IF n EQUALS zero
                RETURN list containing empty string
            END IF

            IF n EQUALS one
                RETURN list containing string "0", string "1", string "8"
            END IF

            SET middles TO generate_strobogrammatic(n MINUS two, final_length)
            SET result TO empty list
            FOR each middle IN middles
                IF n NOT EQUALS final_length
                    APPEND concatenation of string "0", middle, string "0" TO result
                END IF
                APPEND concatenation of string "1", middle, string "1" TO result
                APPEND concatenation of string "6", middle, string "9" TO result
                APPEND concatenation of string "8", middle, string "8" TO result
                APPEND concatenation of string "9", middle, string "6" TO result
            END FOR
            RETURN result
        END FUNCTION

        FUNCTION count_strobogrammatic(low, high)
            SET count TO zero
            FOR length FROM the LENGTH OF low TO the LENGTH OF high
                FOR each num IN generate_strobogrammatic(length, length)
                    IF (length EQUALS the LENGTH OF low AND num LESS THAN low) OR (length EQUALS the LENGTH OF high AND num GREATER THAN high)
                        CONTINUE to next iteration
                    END IF
                    INCREMENT count BY one
                END FOR
            END FOR
            RETURN count
        END FUNCTION

        RETURN count_strobogrammatic(low, high)
    END FUNCTION
END CLASS
