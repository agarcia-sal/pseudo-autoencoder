CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty list  
        RETURN None  
    END IF  
    SET head TO new ListNode with val equal to element at position zero of values and next equal to None  
    SET p TO head  
    FOR each val IN elements of values from position one to the end  
        SET node TO new ListNode with val equal to val and next equal to None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN (p1.val EQUALS p2.val) AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList(self, head)  
        IF head EQUALS None OR head.next EQUALS None OR head.next.next EQUALS None  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO head.next  
        SET even_head TO even  
        WHILE even IS NOT None AND even.next IS NOT None  
            SET odd.next TO even.next  
            SET odd TO odd.next  
            SET even.next TO odd.next  
            SET even TO even.next  
        END WHILE  
        SET odd.next TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
