CLASS Solution  
    FUNCTION killProcess(pid, ppid, kill)  
        CALL create a default dictionary named tree with lists as default values  
        FOR each child process at position index and parent process at position index in parallel from pid and ppid  
            APPEND child process TO the list at key parent process in tree  
        END FOR  
        SET queue TO a list containing kill  
        SET killed_processes TO an empty list  
        WHILE the LENGTH OF queue is greater than zero  
            REMOVE first element FROM queue and SET it TO current  
            APPEND current TO killed_processes  
            FOR each child IN the list at key current in tree  
                APPEND child TO queue  
            END FOR  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
