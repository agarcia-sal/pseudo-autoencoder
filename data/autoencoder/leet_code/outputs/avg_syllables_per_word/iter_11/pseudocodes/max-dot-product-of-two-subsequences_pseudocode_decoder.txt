CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        CALL initializeDPTable WITH m PLUS one AND n PLUS one AND STORE THE RESULT IN dp  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO element at position i MINUS one of nums1 MULTIPLIED BY element at position j MINUS one of nums2  
                SET dp at position i and j TO the MAXIMUM OF current_product, dp at position i MINUS one and j, dp at position i and j MINUS one, dp at position i MINUS one and j MINUS one PLUS current_product  
            END FOR  
        END FOR  
        RETURN dp at position m and n  
    END FUNCTION  

    FUNCTION initializeDPTable(rows, columns)  
        SET dp_table TO an empty list  
        FOR each index FROM zero TO rows MINUS one  
            SET row TO an empty list  
            FOR each index_j FROM zero TO columns MINUS one  
                APPEND the NEGATIVE infinity value TO row  
            END FOR  
            APPEND row TO dp_table  
        END FOR  
        RETURN dp_table  
    END FUNCTION  
END CLASS
