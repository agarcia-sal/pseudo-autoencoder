CLASS Solution
    FUNCTION distinctSubseqII(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a list of zeros with the LENGTH OF n PLUS one  
        SET element at position zero of dp TO one  
        SET last TO an empty mapping  
        
        FOR i FROM one TO n  
            SET char TO element at position i MINUS one of s  
            SET element at position i of dp TO the product of two AND element at position i MINUS one of dp MODULO MOD  
            
            IF char IS a key in last  
                SET element at position i of dp TO element at position i of dp MINUS element at position last at char MINUS one of dp PLUS MOD MODULO MOD  
            END IF  
            
            SET last at key char TO i  
        END FOR  
        
        RETURN element at position n of dp MINUS one MODULO MOD  
    END FUNCTION
END CLASS
