CLASS Solution  
    FUNCTION confusingNumberII(n)  
        SET rotate_map TO mapping of zero TO zero, one TO one, six TO nine, eight TO eight, nine TO six  
        SET valid_digits TO list of zero, one, six, eight, nine  
        
        FUNCTION is_confusing(num)  
            SET original_num TO num  
            SET rotated_num TO zero  
            WHILE num GREATER THAN zero  
                SET digit TO remainder of num DIVIDED BY ten  
                SET rotated_num TO rotated_num MULTIPLIED BY ten PLUS value at key digit OF rotate_map  
                SET num TO quotient of num DIVIDED BY ten  
            END WHILE  
            RETURN rotated_num NOT EQUALS original_num  
        END FUNCTION  
        
        FUNCTION count_confusing_numbers(limit, current)  
            IF current GREATER THAN limit  
                RETURN zero  
            END IF  
            SET count TO one IF current NOT EQUALS zero AND is_confusing(current) ELSE zero  
            FOR each digit IN valid_digits  
                SET new_number TO current MULTIPLIED BY ten PLUS digit  
                IF new_number EQUALS zero  
                    CONTINUE TO next digit  
                END IF  
                SET count TO count PLUS count_confusing_numbers(limit, new_number)  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN count_confusing_numbers(n, zero)  
    END FUNCTION  
END CLASS
