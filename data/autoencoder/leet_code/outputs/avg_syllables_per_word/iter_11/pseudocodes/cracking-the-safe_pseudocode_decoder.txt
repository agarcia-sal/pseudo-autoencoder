CLASS Solution  
    FUNCTION crackSafe(n, k)  
        SET visited TO an empty set  
        SET sequence TO an empty list  

        FUNCTION dfs(node)  
            FOR i FROM zero TO k MINUS one  
                SET edge TO concatenation of node AND string representation of i  
                IF edge NOT IN visited  
                    ADD edge TO visited  
                    CALL dfs WITH parameter substring of edge FROM position one TO end  
                    APPEND string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  

        SET start_node TO string of zero CHARACTER repeated n MINUS one times  
        CALL dfs WITH parameter start_node  

        SET result TO concatenation of elements IN sequence IN reverse order FOLLOWED BY start_node  
        RETURN result  
    END FUNCTION  
END CLASS
