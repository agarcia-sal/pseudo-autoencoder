CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of elements in nums  
        SET dp TO a list of zeroes of length two RAISED TO the power ten  
        SET element at position zero of dp TO one  
        
        FOR each num IN the keys of count  
            IF num EQUALS one  
                SET element at position zero of dp TO element at position zero of dp MULTIPLIED BY two RAISED TO the power the value of count at key one  
                SET element at position zero of dp TO element at position zero of dp MODULO MOD  
                CONTINUE  
            END IF  
            
            IF num NOT IN prime_factors  
                CONTINUE  
            END IF  
            
            SET factors TO the value of prime_factors at key num  
            SET mask TO zero  
            
            FOR each prime IN factors  
                SET mask TO mask BITWISE OR two RAISED TO the power prime MINUS one  
            END FOR  
            
            FOR m FROM (two RAISED TO the power ten) MINUS one DOWN TO zero  
                IF m BITWISE AND mask EQUALS zero  
                    SET element at position m BITWISE OR mask of dp TO the sum of element at position m BITWISE OR mask of dp PLUS element at position m of dp MULTIPLIED BY count at key num MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of elements of dp FROM position one TO the end MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
