CLASS Solution  
    FUNCTION maxNiceDivisors(primeFactors)  
        SET MOD TO one billion PLUS seven  
        
        IF primeFactors LESS THAN OR EQUAL TO three  
            RETURN primeFactors  
        END IF  
        
        SET num_threes TO primeFactors DIVIDED BY three  
        SET remainder TO primeFactors MODULO three  
        
        IF remainder EQUALS zero  
            RETURN the result of three RAISED TO THE POWER num_threes WITH MODULO MOD  
        ELSE IF remainder EQUALS one  
            RETURN the result of three RAISED TO THE POWER num_threes MINUS one WITH MODULO MOD MULTIPLIED BY four WITH MODULO MOD  
        ELSE  
            RETURN the result of three RAISED TO THE POWER num_threes WITH MODULO MOD MULTIPLIED BY two WITH MODULO MOD  
        END IF  
    END FUNCTION  
END CLASS
