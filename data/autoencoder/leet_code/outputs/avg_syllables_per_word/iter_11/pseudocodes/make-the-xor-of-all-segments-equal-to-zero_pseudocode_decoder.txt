CLASS Solution  
    FUNCTION minChanges(nums, k)  
        SET groups TO call defaultdict WITH list  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            APPEND element at position i of nums TO element at position i MODULO k of groups  
        END FOR  
        
        SET freq TO list of length k FILLED WITH call defaultdict WITH integer zero  
        SET min_size TO list of length k FILLED WITH positive infinity  
        FOR i FROM zero TO k MINUS one  
            FOR each num IN element at position i of groups  
                INCREMENT element at key num of element at position i of freq BY one  
            END FOR  
            SET element at position i of min_size TO the LENGTH OF element at position i of groups  
        END FOR  
        
        SET dp TO list FILLED WITH positive infinity OF length TWO TO THE POWER OF ten  
        SET element at position zero of dp TO zero  
        
        FOR i FROM zero TO k MINUS one  
            SET new_dp TO list FILLED WITH positive infinity OF length TWO TO THE POWER OF ten  
            SET min_dp TO the MINIMUM VALUE IN dp  
            FOR j FROM zero TO TWO TO THE POWER OF ten MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position i of min_size  
                FOR each pair num AND count IN element at position i of freq  
                    SET candidate TO element at position j XOR num of dp PLUS element at position i of min_size MINUS count  
                    IF candidate LESS THAN element at position j of new_dp  
                        SET element at position j of new_dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
