CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string  
            RETURN empty list  
        END IF  
          
        SET phone_map TO a mapping from string keys TO string values with pairs as follows  
            SET key '2' TO value 'abc'  
            SET key '3' TO value 'def'  
            SET key '4' TO value 'ghi'  
            SET key '5' TO value 'jkl'  
            SET key '6' TO value 'mno'  
            SET key '7' TO value 'pqrs'  
            SET key '8' TO value 'tuv'  
            SET key '9' TO value 'wxyz'  
          
        FUNCTION backtrack(combination, next_digits)  
            IF LENGTH OF next_digits EQUALS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the string at position zero of next_digits IN phone_map  
                    CALL backtrack WITH combination concatenated with letter, and substring of next_digits from position one to the end  
                END FOR  
            END IF  
        END FUNCTION  
          
        SET output TO empty list  
        CALL backtrack WITH empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
