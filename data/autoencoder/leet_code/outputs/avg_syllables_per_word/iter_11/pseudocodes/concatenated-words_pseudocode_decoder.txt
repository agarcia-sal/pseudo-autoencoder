CLASS Trie  
    FUNCTION __init__()  
        SET children TO a list of twenty-six None elements  
        SET is_end TO False  
    END FUNCTION  

    FUNCTION insert(w)  
        SET node TO self  
        FOR each character c IN w  
            SET idx TO the Unicode code point of c MINUS the Unicode code point of character a  
            IF the element at position idx of node.children EQUALS None THEN  
                SET the element at position idx of node.children TO a new Trie instance  
            END IF  
            SET node TO the element at position idx of node.children  
        END FOR  
        SET node.is_end TO True  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION findAllConcatenatedWordsInADict(words)  
        FUNCTION dfs(w, start)  
            IF start EQUALS the LENGTH OF w THEN  
                RETURN True  
            END IF  
            SET node TO trie  
            FOR index i FROM start TO the LENGTH OF w MINUS one  
                SET idx TO the Unicode code point of the character at position i of w MINUS the Unicode code point of character a  
                IF the element at position idx of node.children EQUALS None THEN  
                    RETURN False  
                END IF  
                SET node TO the element at position idx of node.children  
                IF node.is_end IS True AND dfs(w, i PLUS one) IS True THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  

        SET trie TO a new Trie instance  
        SET ans TO an empty list  
        SORT words IN ascending order by the LENGTH OF each word  
        FOR each word w IN words  
            IF w IS NOT an empty string THEN  
                IF dfs(w, zero) IS True THEN  
                    APPEND w TO ans  
                ELSE  
                    CALL the insert function of trie with parameter w  
                END IF  
            END IF  
        END FOR  
        RETURN ans  
    END FUNCTION  
END CLASS
