CLASS Solution

    FUNCTION findShortestWay(maze, ball, hole)
        SET directions TO list containing (zero, one, 'r'), (one, zero, 'd'), (zero, minus one, 'l'), (minus one, zero, 'u')
        SET m TO the LENGTH OF maze
        SET n TO the LENGTH OF the FIRST element of maze
        SET start TO tuple OF ball
        SET hole TO tuple OF hole
        SET pq TO list containing a tuple of zero, empty string, and start
        SET visited TO empty set

        WHILE pq IS NOT empty
            SET dist, path, coordinate TO pop the smallest element FROM pq
            SET x TO the FIRST element of coordinate
            SET y TO the SECOND element of coordinate

            IF x EQUALS the FIRST element of hole AND y EQUALS the SECOND element of hole
                RETURN path
            END IF

            IF coordinate IS IN visited
                CONTINUE to the next iteration
            END IF

            ADD coordinate TO visited

            FOR each dx, dy, direction IN directions
                SET new_x TO x
                SET new_y TO y
                SET count TO zero

                WHILE new_x PLUS dx is GREATER THAN OR EQUAL TO zero AND new_x PLUS dx is LESS THAN m AND new_y PLUS dy is GREATER THAN OR EQUAL TO zero AND new_y PLUS dy is LESS THAN n AND the element at position new_x PLUS dx and new_y PLUS dy of maze EQUALS zero
                    INCREMENT new_x BY dx
                    INCREMENT new_y BY dy
                    INCREMENT count BY one

                    IF new_x EQUALS the FIRST element of hole AND new_y EQUALS the SECOND element of hole
                        BREAK from this WHILE loop
                    END IF
                END WHILE

                IF the tuple of new_x and new_y IS NOT IN visited
                    ADD a tuple of dist PLUS count, path concatenated WITH direction, and the tuple of new_x and new_y TO pq
                END IF
            END FOR
        END WHILE

        RETURN "impossible"
    END FUNCTION

END CLASS
