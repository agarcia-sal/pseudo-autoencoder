CLASS Solution  
    FUNCTION kMirror(k, n)  
        FUNCTION is_palindrome(s)  
            RETURN s EQUALS the REVERSE OF s  
        END FUNCTION  
        
        FUNCTION to_base_k(num, k)  
            IF num EQUALS zero  
                RETURN string "0"  
            END IF  
            SET digits TO empty list  
            WHILE num GREATER THAN zero  
                APPEND the REMAINDER OF num DIVIDED BY k TO digits  
                SET num TO the QUOTIENT OF num DIVIDED BY k  
            END WHILE  
            RETURN the CONCATENATION OF the STRINGS of the REVERSED digits  
        END FUNCTION  
        
        FUNCTION generate_palindromes(length)  
            IF length EQUALS one  
                FOR i FROM one TO nine  
                    YIELD the STRING of i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO the INTEGER DIVISION OF (length MINUS one) BY two  
                SET end TO ten RAISED TO the INTEGER DIVISION OF (length PLUS one) BY two  
                FOR half FROM start TO end MINUS one  
                    SET half_str TO the STRING of half  
                    IF length MODULO two EQUALS zero  
                        SET full TO half_str CONCATENATED WITH the REVERSE OF half_str  
                    ELSE  
                        SET full TO half_str CONCATENATED WITH the REVERSE OF half_str EXCLUDING the LAST CHARACTER  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO empty list  
        SET length TO one  
        WHILE the LENGTH OF k_mirror_numbers LESS THAN n  
            FOR each pal IN generate_palindromes(length)  
                SET num TO the INTEGER VALUE OF pal  
                IF is_palindrome(to_base_k(num, k))  
                    APPEND num TO k_mirror_numbers  
                    IF the LENGTH OF k_mirror_numbers EQUALS n  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN the SUM OF k_mirror_numbers  
    END FUNCTION  
END CLASS
