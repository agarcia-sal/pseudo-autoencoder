CLASS Solution
    FUNCTION letterCasePermutation(s)
        FUNCTION dfs(i)
            IF i GREATER THAN OR EQUAL TO the LENGTH OF t
                APPEND the concatenation of elements of t TO ans
                RETURN
            END IF
            CALL dfs WITH i PLUS one
            IF element at position i of t IS an alphabetic character
                SET element at position i of t TO the character with ASCII code EQUAL TO the ASCII code of element at position i of t EXCLUSIVE OR thirty-two
                CALL dfs WITH i PLUS one
            END IF
        END FUNCTION

        SET t TO the list of characters of s
        SET ans TO an empty list
        CALL dfs WITH zero
        RETURN ans
    END FUNCTION
END CLASS
