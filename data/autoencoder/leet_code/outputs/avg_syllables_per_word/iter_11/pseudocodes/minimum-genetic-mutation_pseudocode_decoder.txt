CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO the SET constructed from bank  
        SET q TO an empty double-ended queue  
        APPEND the PAIR of start and zero TO q  
        SET mp TO a MAP with keys and corresponding string values as follows:  
            character 'A' MAPS TO string 'TCG'  
            character 'T' MAPS TO string 'ACG'  
            character 'C' MAPS TO string 'ATG'  
            character 'G' MAPS TO string 'ATC'  
        WHILE q is NOT empty  
            REMOVE first element FROM q and SET it TO the PAIR t and step  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR index i FROM zero TO length of t MINUS one  
                SET v TO element at position i of t  
                FOR each character j IN the string mp at key v  
                    SET next TO concatenation of substring of t from start TO i MINUS one, character j, and substring of t from i PLUS one TO end  
                    IF next IS IN s  
                        APPEND the PAIR of next and step PLUS one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
