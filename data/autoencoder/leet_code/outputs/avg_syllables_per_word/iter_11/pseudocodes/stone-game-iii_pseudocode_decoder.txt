CLASS Solution  
    FUNCTION stoneGameIII(stoneValue)  
        SET n TO the LENGTH OF stoneValue  
        SET dp TO a list of zeros with LENGTH n PLUS one  
          
        FOR index FROM n MINUS one DOWN TO zero  
            SET element at position index of dp TO element at position index of stoneValue MINUS element at position index PLUS one of dp  
            IF index PLUS two LESS THAN OR EQUAL TO n  
                SET element at position index of dp TO the GREATER OF element at position index of dp AND the SUM of element at position index of stoneValue AND element at position index PLUS one of stoneValue MINUS element at position index PLUS two of dp  
            END IF  
            IF index PLUS three LESS THAN OR EQUAL TO n  
                SET element at position index of dp TO the GREATER OF element at position index of dp AND the SUM of element at position index of stoneValue PLUS element at position index PLUS one of stoneValue PLUS element at position index PLUS two of stoneValue MINUS element at position index PLUS three of dp  
            END IF  
        END FOR  
          
        IF element at position zero of dp GREATER THAN zero  
            RETURN the string "Alice"  
        ELSE IF element at position zero of dp LESS THAN zero  
            RETURN the string "Bob"  
        ELSE  
            RETURN the string "Tie"  
        END IF  
    END FUNCTION  
END CLASS
