CLASS Solution  
    FUNCTION minRefuelStops(target, startFuel, stations)  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND list consisting of target and zero TO stations  

        FOR each element consisting of position and fuel IN stations  
            WHILE max_heap is not empty AND current_fuel LESS THAN position  
                SET largest_fuel TO the result of removing the smallest element from max_heap  
                SET current_fuel TO current_fuel MINUS largest_fuel  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel LESS THAN position  
                RETURN negative one  
            END IF  
            ADD negative fuel TO max_heap maintaining heap property  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
