CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with element at position zero of values  
    SET i TO one  
    INITIALIZE queue as an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element popped from the left end of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO a new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO a new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        SET left_same TO the result of is_same_tree with parameters p.left and q.left  
        SET right_same TO the result of is_same_tree with parameters p.right and q.right  
        RETURN left_same AND right_same  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues with parameters self, root  
        IF root IS None  
            RETURN an empty list  
        END IF  
        INITIALIZE queue as a double-ended queue containing root  
        INITIALIZE largest_values as an empty list  
        WHILE queue IS NOT empty  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR index FROM zero TO level_size MINUS one  
                SET node TO the element popped from the left end of queue  
                IF max_value IS LESS THAN node.val  
                    SET max_value TO node.val  
                END IF  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
