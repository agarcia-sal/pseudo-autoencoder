CLASS Solution  
    FUNCTION minNumberOfSemesters(n, relations, k)  
        SET graph TO a new mapping with default value as empty list  
        SET in_degree TO list of zeros with length n PLUS one  
        
        FOR each prevCourse, nextCourse IN relations  
            APPEND nextCourse TO the list at key prevCourse of graph  
            INCREMENT the element at position nextCourse of in_degree BY one  
        END FOR  
        
        SET queue TO a new empty double ended queue  
        FOR course FROM one TO n  
            IF the element at position course of in_degree EQUALS zero  
                APPEND course TO queue  
            END IF  
        END FOR  
        
        SET semesters TO zero  
        
        WHILE queue is not empty  
            SET limit TO the smaller value between k and the LENGTH OF queue  
            FOR _ FROM one TO limit  
                SET current_course TO the first element removed FROM queue  
                FOR each next_course IN the list at key current_course of graph  
                    DECREMENT the element at position next_course of in_degree BY one  
                    IF the element at position next_course of in_degree EQUALS zero  
                        APPEND next_course TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT semesters BY one  
        END WHILE  
        
        RETURN semesters  
    END FUNCTION  
END CLASS
