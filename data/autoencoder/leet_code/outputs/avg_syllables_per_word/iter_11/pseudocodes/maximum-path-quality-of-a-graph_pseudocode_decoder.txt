CLASS Solution  
    FUNCTION maximalPathQuality(values, edges, maxTime)  
        SET graph TO result of calling create default dictionary of lists  
        FOR each element edge IN edges  
            SET u TO element at position zero of edge  
            SET v TO element at position one of edge  
            SET time TO element at position two of edge  
            APPEND tuple of v and time TO the list at key u in graph  
            APPEND tuple of u and time TO the list at key v in graph  
        END FOR  
        
        SET visited TO result of creating empty set  
        SET max_quality TO zero  
        
        FUNCTION dfs(node, current_time, current_quality)  
            IF node EQUALS zero  
                SET max_quality TO maximum of max_quality AND current_quality  
            END IF  
            
            FOR each element neighbor_time_pair IN graph at key node  
                SET neighbor TO element at position zero of neighbor_time_pair  
                SET time TO element at position one of neighbor_time_pair  
                IF time LESS THAN OR EQUAL TO current_time  
                    IF neighbor NOT IN visited  
                        ADD neighbor TO visited  
                        CALL dfs with neighbor, current_time MINUS time, current_quality PLUS element at position neighbor of values  
                        REMOVE neighbor FROM visited  
                    ELSE  
                        CALL dfs with neighbor, current_time MINUS time, current_quality  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
        
        ADD zero TO visited  
        CALL dfs with zero, maxTime, element at position zero of values  
        
        RETURN max_quality  
    END FUNCTION  
END CLASS
