CLASS Solution  
    FUNCTION updateMatrix(mat)  
        IF mat EQUALS null OR element at position zero of mat EQUALS null  
            RETURN mat  
        END IF  
        
        SET rows TO the LENGTH OF mat  
        SET cols TO the LENGTH OF element at position zero of mat  
        SET dist TO the RESULT OF FUNCTION initializeDistanceMatrix WITH PARAMETERS rows, cols  
        SET queue TO an empty double-ended queue  
        
        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF element at position r of mat at position c EQUALS zero  
                    SET element at position r of dist at position c TO zero  
                    APPEND tuple of r AND c TO queue  
                END IF  
            END FOR  
        END FOR  
        
        SET directions TO the RESULT OF FUNCTION initializeDirections  
        
        WHILE queue IS NOT empty  
            SET r, c TO the first element REMOVED FROM queue  
            FOR each dr, dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN rows AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN cols AND element at position nr of dist at position nc GREATER THAN element at position r of dist at position c PLUS one  
                    SET element at position nr of dist at position nc TO element at position r of dist at position c PLUS one  
                    APPEND tuple of nr AND nc TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN dist  
    END FUNCTION  
  
    FUNCTION initializeDistanceMatrix(rows, cols)  
        CREATE matrix OF size rows BY cols  
        FOR r FROM zero TO rows MINUS one  
            CREATE list row_list  
            FOR c FROM zero TO cols MINUS one  
                APPEND infinity TO row_list  
            END FOR  
            APPEND row_list TO matrix  
        END FOR  
        RETURN matrix  
    END FUNCTION  
  
    FUNCTION initializeDirections  
        RETURN list containing tuples of negative one AND zero, one AND zero, zero AND negative one, zero AND one  
    END FUNCTION  
END CLASS
