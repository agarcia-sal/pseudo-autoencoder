CLASS Solution
    FUNCTION spiralOrder(matrix)
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty
            RETURN empty list
        END IF

        SET result TO empty list
        SET top TO zero
        SET bottom TO the LENGTH OF matrix MINUS one
        SET left TO zero
        SET right TO the LENGTH OF element at position zero of matrix MINUS one

        WHILE top LESS THAN OR EQUAL TO bottom AND left LESS THAN OR EQUAL TO right
            FOR i FROM left TO right
                APPEND element at position top of matrix at position i TO result
            END FOR
            INCREMENT top BY one

            FOR i FROM top TO bottom
                APPEND element at position i of matrix at position right TO result
            END FOR
            DECREMENT right BY one

            IF top LESS THAN OR EQUAL TO bottom
                FOR i FROM right DOWN TO left
                    APPEND element at position bottom of matrix at position i TO result
                END FOR
                DECREMENT bottom BY one
            END IF

            IF left LESS THAN OR EQUAL TO right
                FOR i FROM bottom DOWN TO top
                    APPEND element at position i of matrix at position left TO result
                END FOR
                INCREMENT left BY one
            END IF
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
