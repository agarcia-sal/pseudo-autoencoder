CLASS Solution  
    FUNCTION openLock(deadends, target)  
        SET deadends_set TO a new set containing all elements of deadends  
        
        IF the string "0000" IS IN deadends_set  
            RETURN negative one  
        END IF  
        
        IF target EQUALS the string "0000"  
            RETURN zero  
        END IF  
        
        SET queue TO a new deque initialized with a tuple of the string "0000" and zero  
        SET visited TO a new set containing the string "0000"  
        
        FUNCTION neighbors(state)  
            SET result TO an empty list  
            FOR index FROM zero TO three  
                SET digit TO the integer value of the character at position index of state  
                FOR each move IN the list containing negative one and one  
                    SET new_digit TO the sum of digit and move MODULO ten  
                    SET new_state TO the concatenation of the substring of state from start to position index MINUS one, the string representation of new_digit, and the substring of state from position index PLUS one to end  
                    APPEND new_state TO result  
                END FOR  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        WHILE queue is not empty  
            SET current_state, steps TO the result of removing the first element from queue  
            
            FOR each neighbor IN the result of calling neighbors with current_state  
                IF neighbor IS NOT IN visited AND neighbor IS NOT IN deadends_set  
                    IF neighbor EQUALS target  
                        RETURN steps PLUS one  
                    END IF  
                    ADD neighbor TO visited  
                    APPEND a tuple of neighbor and steps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
