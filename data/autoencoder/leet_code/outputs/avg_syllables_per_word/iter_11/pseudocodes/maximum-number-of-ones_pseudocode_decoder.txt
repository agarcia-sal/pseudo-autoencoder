CLASS Solution  
    FUNCTION maximumNumberOfOnes(width, height, sideLength, maxOnes)  
        SET freq TO a list of zero repeated sideLength MULTIPLIED BY sideLength times  
        FOR i FROM zero TO width MINUS one  
            FOR j FROM zero TO height MINUS one  
                SET pos TO remainder of i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS remainder of j DIVIDED BY sideLength  
                INCREMENT value at position pos of freq BY one  
            END FOR  
        END FOR  
        SORT freq IN descending order  
        SET result TO sum of first maxOnes elements of freq  
        RETURN result  
    END FUNCTION  
END CLASS
