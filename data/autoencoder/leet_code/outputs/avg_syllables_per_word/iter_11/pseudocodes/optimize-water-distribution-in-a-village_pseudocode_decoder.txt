CLASS Solution  
    FUNCTION minCostToSupplyWater(n, wells, pipes)  
        FOR index FROM zero TO n MINUS one  
            APPEND list containing zero, index PLUS one, element at position index of wells TO pipes  
        END FOR  
        SORT pipes BY the third element of each pipe in ascending order  
        SET parent TO list of integers from zero TO n inclusive  
        FUNCTION find(x)  
            IF element at position x of parent NOT EQUALS x  
                SET element at position x of parent TO find(element at position x of parent)  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        FUNCTION union(x, y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX NOT EQUALS rootY  
                SET element at position rootX of parent TO rootY  
                RETURN True  
            END IF  
            RETURN False  
        END FUNCTION  
        SET min_cost TO zero  
        FOR each pipe IN pipes  
            SET house1 TO first element of pipe  
            SET house2 TO second element of pipe  
            SET cost TO third element of pipe  
            IF union(house1, house2)  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
