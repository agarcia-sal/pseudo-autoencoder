CLASS Solution  
    FUNCTION isTransformable(s, t)  
        CREATE a default mapping from digits to empty lists named pos  
        FOR each index i FROM zero TO the LENGTH OF s MINUS one  
            SET c TO the element at position i of s  
            ADD i TO the heap associated with the integer value of c in pos  
        END FOR  
        
        FOR each index i FROM zero TO the LENGTH OF t MINUS one  
            SET c TO the element at position i of t  
            SET digit TO the integer value of c  
            
            IF the heap associated with digit in pos IS empty  
                RETURN false  
            END IF  
            
            FOR each smaller FROM zero TO digit MINUS one  
                IF the heap associated with smaller in pos IS NOT empty AND the first element of the heap associated with smaller in pos IS LESS THAN the first element of the heap associated with digit in pos  
                    RETURN false  
                END IF  
            END FOR  
            
            REMOVE the first element FROM the heap associated with digit in pos  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
