CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MODULO TO one billion PLUS seven  
        SORT nums IN ASCENDING ORDER  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        INITIALIZE power AS a list OF size n FILLED WITH one  
        
        FOR index FROM one TO n MINUS one  
            SET element at position index OF power TO the REMAINDER OF the MULTIPLICATION of element at position index MINUS one OF power BY two DIVIDED BY MODULO  
        END FOR  
        
        FOR index FROM zero TO n MINUS one  
            SET increment_value TO element at position index OF nums MULTIPLIED BY the DIFFERENCE BETWEEN element at position index OF power AND element at position n MINUS index MINUS one OF power  
            SET total_sum TO the REMAINDER OF the ADDITION of total_sum AND increment_value DIVIDED BY MODULO  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
