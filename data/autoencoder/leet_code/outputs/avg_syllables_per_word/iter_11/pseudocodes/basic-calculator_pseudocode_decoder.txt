CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET sign TO plus symbol  
            SET num TO zero  
            
            WHILE the LENGTH OF s GREATER THAN zero  
                SET char TO first element of s  
                REMOVE first element FROM s  
                
                IF char IS digit character  
                    SET num TO num MULTIPLIED BY ten PLUS numeric value of char  
                END IF  
                
                IF char EQUALS open parenthesis symbol  
                    SET num TO RESULT OF helper(s)  
                END IF  
                
                IF char IS NOT digit character AND char IS NOT whitespace character OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS plus symbol  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus symbol  
                        APPEND negative num TO stack  
                    ELSE IF sign EQUALS multiplication symbol  
                        SET last element of stack TO last element of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS division symbol  
                        SET last element of stack TO integer division of last element of stack BY num  
                    END IF  
                    
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                
                IF char EQUALS close parenthesis symbol  
                    BREAK WHILE loop  
                END IF  
            END WHILE  
            
            RETURN sum of elements in stack  
        END FUNCTION  
        
        RETURN RESULT OF helper(the list of characters of s)  
    END FUNCTION  
END CLASS
