CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS OF digits ONLY  
                SET num TO integer CONVERSION OF block  
                IF num LESS THAN OR EQUAL TO two hundred fifty five AND string CONVERSION OF num EQUALS block  
                    RETURN True  
                ELSE  
                    RETURN False  
                END IF  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF length OF block GREATER THAN OR EQUAL TO one AND length OF block LESS THAN OR EQUAL TO four  
                TRY  
                    CONVERT block TO integer BASE sixteen  
                    RETURN True  
                EXCEPT ValueError  
                    RETURN False  
                END TRY  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO result OF splitting queryIP BY period  
        SET parts_ipv6 TO result OF splitting queryIP BY colon  

        IF length OF parts_ipv4 EQUALS four AND for EACH part IN parts_ipv4 the CALL is_valid_ipv4_block(part) RETURNS True  
            RETURN string "IPv4"  
        ELSE IF length OF parts_ipv6 EQUALS eight AND for EACH part IN parts_ipv6 the CALL is_valid_ipv6_block(part) RETURNS True  
            RETURN string "IPv6"  
        ELSE  
            RETURN string "Neither"  
        END IF  
    END FUNCTION  
END CLASS
