CLASS Solution  
    FUNCTION findNumOfValidWords(words, puzzles)  
        FUNCTION to_mask(s)  
            SET mask TO zero  
            FOR each char IN s  
                SET mask TO mask OR ONE SHIFTED LEFT BY the ORDINAL value of char MINUS the ORDINAL value of letter a  
            END FOR  
            RETURN mask  
        END FUNCTION  
  
        SET word_count TO defaultdict of integer  
        FOR each word IN words  
            SET mask TO to_mask of word  
            IF the NUMBER OF ONES IN mask IS LESS THAN OR EQUAL TO seven  
                INCREMENT the value at key mask IN word_count BY one  
            END IF  
        END FOR  
  
        SET result TO empty list  
        FOR each puzzle IN puzzles  
            SET first_char_mask TO ONE SHIFTED LEFT BY the ORDINAL value of the FIRST character of puzzle MINUS the ORDINAL value of letter a  
            SET puzzle_mask TO to_mask of the SUBSTRING of puzzle FROM position two TO the END  
            SET count TO the value at key first_char_mask IN word_count  
            SET subset TO puzzle_mask  
            WHILE subset IS NOT zero  
                INCREMENT count BY the value at key (subset OR first_char_mask) IN word_count  
                SET subset TO (subset MINUS one) AND puzzle_mask  
            END WHILE  
            APPEND count TO result  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
