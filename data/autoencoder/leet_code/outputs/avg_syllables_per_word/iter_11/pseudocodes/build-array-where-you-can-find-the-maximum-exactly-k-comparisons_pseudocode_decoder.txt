CLASS Solution  
    FUNCTION numOfArrays(n, m, k)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a three-dimensional list of zeros with dimensions n PLUS one, m PLUS one, and k PLUS one  
        FOR j FROM one TO m  
            SET element at position one, j, and one of dp TO one  
        END FOR  
        FOR i FROM two TO n  
            FOR j FROM one TO m  
                FOR p FROM one TO k  
                    FOR x FROM one TO j MINUS one  
                        SET element at position i, j, and p of dp TO element at position i, j, and p of dp PLUS element at position i MINUS one, x, and p MINUS one of dp  
                        SET element at position i, j, and p of dp TO element at position i, j, and p of dp MODULO MOD  
                    END FOR  
                    SET element at position i, j, and p of dp TO element at position i, j, and p of dp PLUS element at position i MINUS one, j, and p of dp MULTIPLIED BY j  
                    SET element at position i, j, and p of dp TO element at position i, j, and p of dp MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        SET result TO zero  
        FOR j FROM one TO m  
            SET result TO result PLUS element at position n, j, and k of dp  
            SET result TO result MODULO MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
