CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET dp TO a list of lists of zero with m PLUS one rows and n PLUS one columns  
        SET max_side TO zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of matrix and position j MINUS one EQUALS character one  
                    SET dp at position i and position j TO the MINIMUM OF dp at position i MINUS one and position j, dp at position i and position j MINUS one, dp at position i MINUS one and position j MINUS one PLUS one  
                    SET max_side TO the GREATER OF max_side AND dp at position i and position j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
