CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
  
        SET value_indices TO an empty mapping from values TO lists  
        SET index TO zero  
        FOR each value IN arr  
            APPEND index TO the list corresponding TO value IN value_indices  
            INCREMENT index BY one  
        END FOR  
  
        SET queue TO a double-ended queue containing a pair of zero AND zero  
        SET visited TO a set containing zero  
  
        WHILE queue IS NOT empty  
            REMOVE first pair FROM queue AND DESTRUCTURE TO current_index AND steps  
  
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
  
            SET next_indices TO a list containing current_index PLUS one, current_index MINUS one, AND all indices IN the list mapped by arr AT current_index IN value_indices  
  
            FOR each next_index IN next_indices  
                IF next_index IS GREATER THAN OR EQUAL TO zero AND next_index IS LESS THAN n AND next_index IS NOT IN visited  
                    ADD next_index TO visited  
                    APPEND a pair of next_index AND steps PLUS one TO queue  
                END IF  
            END FOR  
  
            SET the list mapped by arr AT current_index IN value_indices TO an empty list  
        END WHILE  
  
        RETURN minus one  
    END FUNCTION  
END CLASS
