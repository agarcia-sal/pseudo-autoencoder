CLASS Solution  
    FUNCTION kInversePairs(n, k)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a new two-dimensional list of size n PLUS one by k PLUS one initialized with zero  
        SET element at position zero of the list at position zero of dp TO one  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            FOR j FROM zero TO k  
                SET prefix_sum TO prefix_sum PLUS element at position j of the list at position i MINUS one of dp  
                SET prefix_sum TO prefix_sum MODULO MOD  
                IF j IS GREATER THAN OR EQUAL TO i  
                    SET prefix_sum TO prefix_sum MINUS element at position j MINUS i of the list at position i MINUS one of dp PLUS MOD  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                END IF  
                SET element at position j of the list at position i of dp TO prefix_sum  
            END FOR  
        END FOR  
        
        RETURN element at position k of the list at position n of dp  
    END FUNCTION  
END CLASS
