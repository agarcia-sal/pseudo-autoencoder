CLASS Solution  
    FUNCTION findPairs(nums, k)  
        IF k LESS THAN zero  
            RETURN zero  
        END IF  
        
        SET num_counts TO empty dictionary  
        FOR each num IN nums  
            IF num IN num_counts  
                SET current_count TO the value associated with num in num_counts  
            ELSE  
                SET current_count TO zero  
            END IF  
            SET the value associated with num in num_counts TO current_count PLUS one  
        END FOR  
        
        SET unique_pairs TO zero  
        FOR each num IN keys of num_counts  
            IF k EQUALS zero  
                IF the value associated with num in num_counts GREATER THAN one  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF num PLUS k IN keys of num_counts  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
