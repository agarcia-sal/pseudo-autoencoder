CLASS Solution  
    FUNCTION rob(nums)  
        IF nums EQUALS empty  
            RETURN zero  
        END IF  
        IF LENGTH OF nums EQUALS one  
            RETURN element at position zero of nums  
        END IF  
        SET dp TO auxiliary FUNCTION initialize_list WITH parameters zero AND LENGTH OF nums  
        SET element at position zero of dp TO element at position zero of nums  
        IF element at position zero of nums GREATER THAN OR EQUAL TO element at position one of nums  
            SET element at position one of dp TO element at position zero of nums  
        ELSE  
            SET element at position one of dp TO element at position one of nums  
        END IF  
        FOR i FROM two TO LENGTH OF nums MINUS one  
            IF element at position i MINUS one of dp GREATER THAN OR EQUAL TO element at position i MINUS two of dp PLUS element at position i of nums  
                SET element at position i of dp TO element at position i MINUS one of dp  
            ELSE  
                SET element at position i of dp TO element at position i MINUS two of dp PLUS element at position i of nums  
            END IF  
        END FOR  
        RETURN element at last position of dp  
    END FUNCTION  
END CLASS
