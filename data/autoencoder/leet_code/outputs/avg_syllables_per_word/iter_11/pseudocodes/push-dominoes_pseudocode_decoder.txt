CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a list of zero repeated n times  
        
        SET force TO zero  
        FOR i FROM zero TO n MINUS one  
            IF element at position i of dominoes EQUALS character R  
                SET force TO n  
            ELSE IF element at position i of dominoes EQUALS character L  
                SET force TO zero  
            ELSE  
                SET force TO the MAXIMUM OF force MINUS one AND zero  
            END IF  
            INCREMENT element at position i of forces BY force  
        END FOR  
        
        SET force TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            IF element at position i of dominoes EQUALS character L  
                SET force TO n  
            ELSE IF element at position i of dominoes EQUALS character R  
                SET force TO zero  
            ELSE  
                SET force TO the MAXIMUM OF force MINUS one AND zero  
            END IF  
            DECREMENT element at position i of forces BY force  
        END FOR  
        
        SET result TO an empty list  
        FOR each f IN forces  
            IF f GREATER THAN zero  
                APPEND character R TO result  
            ELSE IF f LESS THAN zero  
                APPEND character L TO result  
            ELSE  
                APPEND character period TO result  
            END IF  
        END FOR  
        
        RETURN the CONCATENATION OF elements in result  
    END FUNCTION  
END CLASS
