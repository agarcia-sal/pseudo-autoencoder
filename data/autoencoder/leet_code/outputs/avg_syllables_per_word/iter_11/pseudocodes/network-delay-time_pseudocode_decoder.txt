CLASS Solution
    FUNCTION networkDelayTime(times, n, k)
        SET graph TO result of calling construct_graph with times and n
        SET dist TO result of calling initialize_distances with n and k

        SET min_heap TO list containing tuple of zero and k

        WHILE min_heap is not empty
            SET current_dist, node TO result of removing smallest element from min_heap

            IF current_dist GREATER THAN value of dist at key node
                CONTINUE to next iteration
            END IF

            FOR each neighbor, weight IN list at key node in graph
                SET distance TO current_dist PLUS weight

                IF distance LESS THAN value of dist at key neighbor
                    SET value of dist at key neighbor TO distance
                    INSERT tuple of distance and neighbor into min_heap maintaining order
                END IF
            END FOR
        END WHILE

        SET max_time TO maximum value in dist

        IF max_time EQUALS infinity
            RETURN negative one
        ELSE
            RETURN max_time
        END IF
    END FUNCTION

    FUNCTION construct_graph(times, n)
        SET graph TO empty mapping from integers to lists for keys from one TO n
        FOR each u, v, w IN times
            APPEND tuple of v and w TO list at key u in graph
        END FOR
        RETURN graph
    END FUNCTION

    FUNCTION initialize_distances(n, k)
        SET dist TO empty mapping from integers to infinity for keys from one TO n
        SET value of dist at key k TO zero
        RETURN dist
    END FUNCTION
END CLASS
