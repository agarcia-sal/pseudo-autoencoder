CLASS Solution  
    FUNCTION checkRecord(n)  
        SET MOD TO one billion PLUS seven  
        
        CALL initialize_dp WITH n RETURNING dp
        
        SET element at position zero OF element at position zero OF element at position zero OF dp TO one  
        
        FOR i FROM one TO n  
            FOR j FROM zero TO one  
                FOR k FROM zero TO two  
                    SET element at position zero OF element at position j OF element at position i OF dp TO the value of element at position zero OF element at position j OF element at position i OF dp PLUS element at position k OF element at position j OF element at position i MINUS one OF dp  
                    SET element at position zero OF element at position j OF element at position i OF dp TO element at position zero OF element at position j OF element at position i OF dp MODULO MOD  
                    
                    IF j EQUALS zero  
                        SET element at position zero OF element at position one OF element at position i OF dp TO element at position zero OF element at position one OF element at position i OF dp PLUS element at position k OF element at position zero OF element at position i MINUS one OF dp  
                        SET element at position zero OF element at position one OF element at position i OF dp TO element at position zero OF element at position one OF element at position i OF dp MODULO MOD  
                    END IF  
                    
                    IF k LESS THAN two  
                        SET element at position k PLUS one OF element at position j OF element at position i OF dp TO element at position k PLUS one OF element at position j OF element at position i OF dp PLUS element at position k OF element at position j OF element at position i MINUS one OF dp  
                        SET element at position k PLUS one OF element at position j OF element at position i OF dp TO element at position k PLUS one OF element at position j OF element at position i OF dp MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        
        FOR j FROM zero TO one  
            FOR k FROM zero TO two  
                SET result TO result PLUS element at position k OF element at position j OF element at position n OF dp  
                SET result TO result MODULO MOD  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
    
    FUNCTION initialize_dp(n)  
        SET dp TO empty list  
        FOR i FROM zero TO n  
            SET list_j TO empty list  
            FOR j FROM zero TO one  
                SET list_k TO empty list  
                FOR k FROM zero TO two  
                    APPEND zero TO list_k  
                END FOR  
                APPEND list_k TO list_j  
            END FOR  
            APPEND list_j TO dp  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
