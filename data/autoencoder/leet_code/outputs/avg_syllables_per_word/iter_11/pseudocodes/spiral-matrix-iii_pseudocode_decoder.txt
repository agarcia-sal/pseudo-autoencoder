CLASS Solution
    FUNCTION spiralMatrixIII(rows, cols, rStart, cStart)
        SET directions TO list containing tuple of zero and one, tuple of one and zero, tuple of zero and negative one, tuple of negative one and zero
        SET direction_index TO zero
        SET steps TO zero
        SET increment TO zero
        SET visited TO empty list
        SET x TO rStart
        SET y TO cStart

        WHILE the LENGTH OF visited LESS THAN rows MULTIPLIED BY cols
            IF increment MODULO two EQUALS zero
                INCREMENT steps BY one
            END IF

            FOR each iteration FROM one TO steps
                IF x GREATER THAN OR EQUAL TO zero AND x LESS THAN rows AND y GREATER THAN OR EQUAL TO zero AND y LESS THAN cols
                    APPEND list containing x and y TO visited
                END IF

                SET dx TO element at position direction_index AND zero of directions
                SET dy TO element at position direction_index AND one of directions
                SET x TO x PLUS dx
                SET y TO y PLUS dy
            END FOR

            SET direction_index TO (direction_index PLUS one) MODULO four
            INCREMENT increment BY one
        END WHILE

        RETURN visited
    END FUNCTION
END CLASS
