CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO an empty mapping from characters to zero for each character from 'a' to 'z'  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND ( the character code of element at position index of s MINUS the character code of element at position index MINUS one of s EQUALS one OR the element at position index of s EQUALS character 'a' AND the element at position index MINUS one of s EQUALS character 'z' )  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET max_length at key element at position index of s TO the maximum of max_length at key element at position index of s AND current_length  
        END FOR  
        SET total_unique_substrings TO the sum of all values in max_length  
        RETURN total_unique_substrings  
    END FUNCTION  
END CLASS
