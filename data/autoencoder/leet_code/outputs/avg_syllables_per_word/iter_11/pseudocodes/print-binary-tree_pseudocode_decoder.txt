CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH parameter value at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO NEW TreeNode WITH parameter element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO NEW TreeNode WITH parameter element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left and q.left AND the result of is_same_tree with parameters p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION printTree(self, root)  
        FUNCTION getHeight(node)  
            IF node IS None  
                RETURN zero  
            END IF  
            RETURN one PLUS the MAXIMUM of getHeight with parameter node.left AND getHeight with parameter node.right  
        END FUNCTION  
        
        SET height TO getHeight with parameter root  
        SET width TO two RAISED TO THE POWER OF height MINUS one  
        SET res TO a list of height elements each consisting of width elements all SET TO an empty string  
        
        FUNCTION placeNode(node, row, left, right)  
            IF node IS None  
                RETURN  
            END IF  
            SET mid TO the INTEGER DIVISION of the SUM of left AND right BY two  
            SET the element at position mid of the element at position row of res TO the string representation of the value of node  
            CALL placeNode WITH PARAMETERS node.left, row PLUS one, left, mid  
            CALL placeNode WITH PARAMETERS node.right, row PLUS one, mid, right  
        END FUNCTION  
        
        CALL placeNode WITH PARAMETERS root, zero, zero, width  
        RETURN res  
    END FUNCTION  
END CLASS
