CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO sorted list of cuts appended with zero and n  
        
        FUNCTION dp(left, right)  
            IF right MINUS left LESS THAN OR EQUAL TO one  
                RETURN zero  
            END IF  
            
            SET min_cost TO positive infinity  
            FOR i FROM left PLUS one TO right MINUS one  
                SET cost TO element at position right of cuts MINUS element at position left of cuts PLUS dp(left, i) PLUS dp(i, right)  
                IF cost LESS THAN min_cost  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp(zero, LENGTH OF cuts MINUS one)  
    END FUNCTION  
END CLASS
