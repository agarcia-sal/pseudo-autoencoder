CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix, x, y)  
            SET directions TO list containing tuple of minus one and zero, tuple of one and zero, tuple of zero and minus one, tuple of zero and one, tuple of zero and zero  
            FOR each tuple dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n  
                    SET matrix element at position nx and ny TO one MINUS matrix element at position nx and ny  
                END IF  
            END FOR  
        END FUNCTION  

        FUNCTION matrix_to_tuple(matrix)  
            RETURN tuple containing for each row IN matrix a tuple constructed from row  
        END FUNCTION  

        SET m TO LENGTH OF mat  
        SET n TO LENGTH OF element at position zero of mat  
        SET target TO tuple containing for index from zero TO m MINUS one a tuple containing for index from zero TO n MINUS one a zero  
        SET initial_state TO matrix_to_tuple applied to mat  
        SET queue TO list containing tuple of initial_state and zero  
        SET visited TO empty set  
        ADD initial_state TO visited  

        WHILE queue is not empty  
            REMOVE first element FROM queue and SET TO tuple current_state and steps  
            IF current_state EQUALS target  
                RETURN steps  
            END IF  

            SET current_matrix TO list containing for each row IN current_state a list constructed from row  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO deep copy of current_matrix  
                    CALL flip with next_matrix, i, j  
                    SET next_state TO matrix_to_tuple applied to next_matrix  
                    IF next_state NOT IN visited  
                        ADD next_state TO visited  
                        APPEND tuple of next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN minus one  
    END FUNCTION  
END CLASS
