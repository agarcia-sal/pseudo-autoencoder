CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO a list of SIZE elements from zero to size MINUS one  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u  
            SET element at position u of self.parent TO the result of self.find(element at position u of self.parent)  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET root_u TO self.find(u)  
        SET root_v TO self.find(v)  
        IF root_u NOT EQUALS root_v  
            SET element at position root_u of self.parent TO root_v  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION matrixRankTransform(self, matrix)  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET value_to_indices TO an empty mapping with default value as empty list  

        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                APPEND the pair of r and c TO element at key element at position r and c of matrix of value_to_indices  
            END FOR  
        END FOR  

        SET row_rank TO a list of m zeros  
        SET col_rank TO a list of n zeros  
        SET result TO a list of m elements each being a list of n zeros  

        FOR each value IN the keys of value_to_indices sorted in ascending order  
            SET uf TO a new UnionFind with size m PLUS n  
            SET rank TO an empty mapping with default zero  

            FOR each pair r, c IN element at key value of value_to_indices  
                CALL uf.union with arguments r and c PLUS m  
            END FOR  

            FOR each pair r, c IN element at key value of value_to_indices  
                SET root TO uf.find(r)  
                SET element at key root of rank TO the maximum of element at key root of rank and the maximum of element at position r of row_rank and element at position c of col_rank PLUS one  
            END FOR  

            FOR each pair r, c IN element at key value of value_to_indices  
                SET root TO uf.find(r)  
                SET element at position r and c of result TO element at key root of rank  
                SET element at position r of row_rank TO element at key root of rank  
                SET element at position c of col_rank TO element at key root of rank  
            END FOR  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
