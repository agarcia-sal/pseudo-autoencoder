CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each char IN s  
            IF char IS a digit  
                SET d TO d concatenated with char  
            ELSE IF char IS an alphabetic character  
                SET a TO a concatenated with char  
            ELSE IF char EQUALS the character left bracket  
                APPEND tuple of a and d converted to integer TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char EQUALS the character right bracket  
                SET p, n TO the last element removed FROM stack  
                SET a TO p concatenated with a repeated n times  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
