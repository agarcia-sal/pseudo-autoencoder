CLASS Solution  
    FUNCTION maxProfit(k, prices)  
        IF prices EQUALS empty OR k EQUALS zero  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF prices  
        
        IF k GREATER THAN OR EQUAL TO n DIVIDED BY two  
            SET profit TO zero  
            FOR i FROM one TO n MINUS one  
                IF element at position i of prices GREATER THAN element at position i MINUS one of prices  
                    INCREMENT profit BY element at position i of prices MINUS element at position i MINUS one of prices  
                END IF  
            END FOR  
            RETURN profit  
        END IF  
        
        SET hold TO a list of k PLUS one elements each SET TO negative infinity  
        SET release TO a list of k PLUS one elements each SET TO zero  
        
        FOR each price IN prices  
            FOR i FROM one TO k  
                SET hold at position i TO the maximum OF hold at position i AND release at position i MINUS one MINUS price  
                SET release at position i TO the maximum OF release at position i AND hold at position i PLUS price  
            END FOR  
        END FOR  
        
        RETURN element at position k of release  
    END FUNCTION  
END CLASS
