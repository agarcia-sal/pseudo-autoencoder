CLASS Solution  
    FUNCTION myPow WITH PARAMETERS x, n  
        FUNCTION qpow WITH PARAMETERS a, n  
            SET ans TO one  
            WHILE n NOT EQUALS zero  
                IF n BITWISE AND one EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two DISCARDED REMAINDER  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n GREATER THAN OR EQUAL TO zero  
            RETURN qpow WITH PARAMETERS x, n  
        ELSE  
            RETURN one DIVIDED BY qpow WITH PARAMETERS x, n NEGATED  
        END IF  
    END FUNCTION  
END CLASS
