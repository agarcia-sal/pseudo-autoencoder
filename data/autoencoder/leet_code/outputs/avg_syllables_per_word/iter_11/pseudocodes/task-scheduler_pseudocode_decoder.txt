CLASS Solution  
    FUNCTION leastInterval(tasks, n)  
        SET task_counts TO CALL Counter WITH tasks  
        SET max_heap TO an empty list  
        FOR each count IN the values of task_counts  
            APPEND negative count TO max_heap  
        END FOR  
        CALL heapify ON max_heap  
        
        SET time TO zero  
        SET queue TO an empty list  
        
        WHILE max_heap IS NOT empty OR queue IS NOT empty  
            WHILE queue IS NOT empty AND the first element at position zero of queue at position zero EQUALS time  
                SET pair TO REMOVE first element FROM queue  
                SET _, count TO pair  
                IF count LESS THAN zero  
                    CALL heappush ON max_heap AND count  
                END IF  
            END WHILE  
            
            IF max_heap IS NOT empty  
                SET count TO CALL heappop ON max_heap  
                INCREMENT count BY one  
                IF count LESS THAN zero  
                    APPEND tuple OF (time PLUS n PLUS one, count) TO queue  
                END IF  
            END IF  
            
            INCREMENT time BY one  
        END WHILE  
        
        RETURN time  
    END FUNCTION  
END CLASS
