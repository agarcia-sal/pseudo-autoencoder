CLASS TreeNode  
    FUNCTION __init__ with parameters val, left, right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue IS not empty  
        SET node TO the first element REMOVED FROM queue  
        IF i IS less than the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO a new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS less than the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO a new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p, q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree with arguments p.left AND q.left AND is_same_tree with arguments p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView with parameter root  
        IF root IS None  
            RETURN an empty list  
        END IF  
        SET right_side_view TO an empty list  
        SET queue TO a deque initialized with root  
        WHILE queue IS not empty  
            SET level_length TO the LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO the first element REMOVED FROM queue  
                IF i IS EQUAL TO level_length MINUS one  
                    APPEND node.val TO right_side_view  
                END IF  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
