CLASS Solution  
    FUNCTION minAbsDifference(nums, goal)  
        SET n TO the LENGTH OF nums  
        SET mid TO n DIVIDED BY two  
        
        SET left_sums TO a list containing zero  
        FOR each num IN the SUBLIST of nums FROM the START TO position mid MINUS one  
            SET new_sums TO an empty list  
            FOR each s IN left_sums  
                APPEND s PLUS num TO new_sums  
            END FOR  
            APPEND each element IN new_sums TO left_sums  
        END FOR  
        
        SET right_sums TO a list containing zero  
        FOR each num IN the SUBLIST of nums FROM position mid TO the END  
            SET new_sums TO an empty list  
            FOR each s IN right_sums  
                APPEND s PLUS num TO new_sums  
            END FOR  
            APPEND each element IN new_sums TO right_sums  
        END FOR  
        
        SORT right_sums IN ascending order  
        
        SET min_diff TO positive infinity  
        
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO the RESULT of binary searching the POSITION TO INSERT target in right_sums maintaining order  
            
            IF pos LESS THAN the LENGTH OF right_sums  
                SET temp_diff TO the ABSOLUTE VALUE OF left_sum PLUS the element at position pos of right_sums MINUS goal  
                IF temp_diff LESS THAN min_diff  
                    SET min_diff TO temp_diff  
                END IF  
            END IF  
            
            IF pos GREATER THAN zero  
                SET temp_diff TO the ABSOLUTE VALUE OF left_sum PLUS the element at position pos MINUS one of right_sums MINUS goal  
                IF temp_diff LESS THAN min_diff  
                    SET min_diff TO temp_diff  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
