CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the result of splitting path BY the character slash  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS single period  
                CONTINUE to the next iteration  
            ELSE IF component EQUALS double period  
                IF the LENGTH OF stack IS GREATER THAN zero  
                    REMOVE last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO concatenation of slash AND concatenation of elements of stack joined BY slash  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
