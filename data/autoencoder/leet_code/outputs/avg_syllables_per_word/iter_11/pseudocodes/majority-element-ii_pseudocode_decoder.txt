CLASS Solution  
    FUNCTION majorityElement(nums)  
        SET n1 TO zero  
        SET n2 TO zero  
        SET m1 TO zero  
        SET m2 TO one  
        FOR each m IN nums  
            IF m EQUALS m1  
                INCREMENT n1 BY one  
            ELSE IF m EQUALS m2  
                INCREMENT n2 BY one  
            ELSE IF n1 EQUALS zero  
                SET m1 TO m  
                SET n1 TO one  
            ELSE IF n2 EQUALS zero  
                SET m2 TO m  
                SET n2 TO one  
            ELSE  
                SET n1 TO n1 MINUS one  
                SET n2 TO n2 MINUS one  
            END IF  
        END FOR  
        RETURN list of each m IN list of m1 and m2 WHERE count of m IN nums GREATER THAN LENGTH OF nums DIVIDED BY three  
    END FUNCTION  
END CLASS
