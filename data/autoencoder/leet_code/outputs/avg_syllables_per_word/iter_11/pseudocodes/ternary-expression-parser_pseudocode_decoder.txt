CLASS Solution  
    FUNCTION parseTernary(expression)  
        FUNCTION evaluate(index)  
            IF index EQUALS the LENGTH OF expression MINUS one OR the character at position index PLUS one of expression EQUALS colon  
                RETURN the character at position index of expression, index PLUS two  
            END IF  
            SET true_part, next_index TO the RESULT of calling evaluate with index PLUS two  
            SET false_part, final_index TO the RESULT of calling evaluate with next_index  
            IF the character at position index of expression EQUALS the letter T  
                RETURN true_part, final_index  
            ELSE  
                RETURN false_part, final_index  
            END IF  
        END FUNCTION  
        SET result, unused TO the RESULT of calling evaluate with zero  
        RETURN result  
    END FUNCTION  
END CLASS
