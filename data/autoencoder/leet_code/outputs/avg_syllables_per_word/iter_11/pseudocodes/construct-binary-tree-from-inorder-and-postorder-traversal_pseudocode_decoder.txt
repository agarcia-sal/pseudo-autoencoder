CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized WITH the element at position zero of values  
    SET i TO one  
    CREATE queue AS an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE first element FROM queue AND SET TO node  
        IF i LESS THAN LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode initialized WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode initialized WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, inorder, postorder)  
        IF inorder EQUALS none OR the LENGTH OF inorder EQUALS zero OR postorder EQUALS none OR the LENGTH OF postorder EQUALS zero  
            RETURN none  
        END IF  
        REMOVE last element FROM postorder AND SET TO root_val  
        SET root TO a new TreeNode initialized WITH root_val  
        FIND the index of root_val in inorder AND SET TO root_index  
        SET root.right TO result of calling self.buildTree with parameters: subsection of inorder from position root_index PLUS one TO end, postorder  
        SET root.left TO result of calling self.buildTree with parameters: subsection of inorder from start TO position root_index MINUS one, postorder  
        RETURN root  
    END FUNCTION  
END CLASS
