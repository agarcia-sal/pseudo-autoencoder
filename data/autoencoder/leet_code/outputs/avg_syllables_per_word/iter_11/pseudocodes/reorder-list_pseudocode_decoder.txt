CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode with val as element at position zero of values and next as None  
    SET p TO head  
    FOR each val IN elements of values starting from position one TO end  
        SET node TO new ListNode with val as val and next as None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN expression that element val at p1 EQUALS element val at p2 AND call is_same_list with parameters element next at p1 AND element next at p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList(self, head)  
        IF head EQUALS None OR element next at head EQUALS None THEN  
            RETURN  
        END IF  

        SET slow TO head  
        SET fast TO head  
        WHILE fast IS NOT None AND element next at fast IS NOT None  
            SET slow TO element next at slow  
            SET fast TO element next at element next at fast  
        END WHILE  

        SET prev TO None  
        SET curr TO slow  
        WHILE curr IS NOT None  
            SET next_temp TO element next at curr  
            SET element next at curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  

        SET first TO head  
        SET second TO prev  
        WHILE element next at second IS NOT None  
            SET temp1 TO element next at first  
            SET temp2 TO element next at second  
            SET element next at first TO second  
            SET element next at second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
