CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO LENGTH OF word1  
        SET n TO LENGTH OF word2  
        SET dp TO CALL createTwoDimensionalArray WITH PARAMETERS m PLUS one, n PLUS one  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of word1 EQUALS element at position j MINUS one of word2  
                    SET element at position i, position j of dp TO element at position i MINUS one, position j MINUS one of dp PLUS one  
                ELSE  
                    SET element at position i, position j of dp TO the GREATER OF element at position i MINUS one, position j of dp AND element at position i, position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        SET lcs_length TO element at position m, position n of dp  
        RETURN m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  
  
    FUNCTION createTwoDimensionalArray(rows, columns)  
        CREATE array WITH rows NUMBER OF elements  
        FOR row_index FROM zero TO rows MINUS one  
            SET element at position row_index of array TO list OF columns NUMBER OF zeros  
        END FOR  
        RETURN array  
    END FUNCTION  
END CLASS
