CLASS Solution  
    FUNCTION scoreOfStudents(s, answers)  
        FUNCTION evaluate_expression(expression)  
            SET tokens TO empty list  
            SET num TO zero  
            FOR each char IN expression  
                IF char IS a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                ELSE  
                    APPEND num TO tokens  
                    APPEND char TO tokens  
                    SET num TO zero  
                END IF  
            END FOR  
            APPEND num TO tokens  
            
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens  
                IF the element at position i of tokens EQUALS multiplication symbol  
                    SET element at position i MINUS one of tokens TO element at position i MINUS one of tokens MULTIPLIED BY element at position i PLUS one of tokens  
                    REMOVE element at position i and element at position i PLUS one FROM tokens  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
            
            SET result TO the element at position zero of tokens  
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens  
                IF the element at position i of tokens EQUALS addition symbol  
                    SET result TO result PLUS element at position i PLUS one of tokens  
                END IF  
                INCREMENT i BY two  
            END WHILE  
            
            RETURN result  
        END FUNCTION  
        
        FUNCTION get_possible_answers(expression)  
            IF expression CONTAINS only digits  
                RETURN a set containing the integer value of expression  
            END IF  
            
            SET results TO empty set  
            FOR i FROM one TO the LENGTH OF expression MINUS two STEP two  
                SET left_results TO get_possible_answers(the substring of expression from start TO position i MINUS one)  
                SET right_results TO get_possible_answers(the substring of expression from position i PLUS one TO end)  
                FOR each left IN left_results  
                    FOR each right IN right_results  
                        IF the element at position i of expression EQUALS addition symbol  
                            SET result TO left PLUS right  
                        ELSE  
                            SET result TO left MULTIPLIED BY right  
                        END IF  
                        IF result LESS THAN OR EQUAL TO one thousand  
                            ADD result TO results  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        SET correct_answer TO evaluate_expression(s)  
        SET possible_answers TO get_possible_answers(s)  
        
        SET points TO zero  
        FOR each answer IN answers  
            IF answer EQUALS correct_answer  
                INCREMENT points BY five  
            ELSE IF answer IS IN possible_answers  
                INCREMENT points BY two  
            END IF  
        END FOR  
        
        RETURN points  
    END FUNCTION  
END CLASS
