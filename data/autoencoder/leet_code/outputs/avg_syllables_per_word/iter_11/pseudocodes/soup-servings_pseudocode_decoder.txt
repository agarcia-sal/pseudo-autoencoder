CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN 4800  
            RETURN one point zero  
        END IF  
        
        SET n TO the quantity of n PLUS twenty four DIVIDED BY twenty five  
        
        SET memo TO an empty collection  
        
        FUNCTION dp(a, b)  
            IF the pair of a and b IS IN memo  
                RETURN the value stored at the pair of a and b in memo  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN zero point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one point zero  
            END IF  
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero point zero  
            END IF  
            
            SET prob TO zero point two five MULTIPLIED BY the sum of dp of a MINUS four and b, dp of a MINUS three and b MINUS one, dp of a MINUS two and b MINUS two, and dp of a MINUS one and b MINUS three  
            
            SET memo at the pair of a and b TO prob  
            RETURN prob  
        END FUNCTION  
        
        RETURN dp of n and n  
    END FUNCTION  
END CLASS
