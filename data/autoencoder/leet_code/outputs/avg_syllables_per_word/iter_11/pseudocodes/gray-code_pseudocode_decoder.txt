CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN a list containing zero  
        END IF  
        SET prev_gray TO the result of calling grayCode with argument n MINUS one  
        SET mask TO one MULTIPLIED BY two RAISED TO the power of n MINUS one  
        SET current_gray TO prev_gray CONCATENATED WITH a new list  
        FOR each num IN the elements of prev_gray taken in reverse order  
            APPEND mask OR num TO current_gray  
        END FOR  
        RETURN current_gray  
    END FUNCTION  
END CLASS
