CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuples where each tuple is a row from board  
        SET target TO tuple containing tuple one two three and tuple four five zero  
        
        SET neighbors TO dictionary with keys and sets as values:  
            key zero maps to set containing one and three  
            key one maps to set containing zero two and four  
            key two maps to set containing one and five  
            key three maps to set containing zero and four  
            key four maps to set containing one three and five  
            key five maps to set containing two and four  
        
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position j of element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        
        INITIALIZE queue WITH element containing start zero_pos and zero  
        INITIALIZE visited AS set containing start  
        
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND UNPACK INTO state zero_pos and moves  
            
            IF state EQUALS target  
                RETURN moves  
            END IF  
            
            FOR each neighbor IN element at position zero_pos of neighbors  
                CONVERT state TO list of lists AND SET TO new_state  
                SET i TO zero_pos DIVIDED BY three  
                SET j TO zero_pos MODULO three  
                SET ni TO neighbor DIVIDED BY three  
                SET nj TO neighbor MODULO three  
                
                SWAP element at position j of element at position i OF new_state WITH element at position nj of element at position ni OF new_state  
                
                SET new_tuple_state TO tuple of tuples where each tuple is a row from new_state  
                
                IF new_tuple_state IS NOT IN visited  
                    ADD new_tuple_state TO visited  
                    APPEND element containing new_tuple_state neighbor and moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
