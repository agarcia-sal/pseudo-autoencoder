CLASS Solution  
    FUNCTION maxEnvelopes(envelopes)  
        SORT envelopes BY the first element of each envelope IN ascending order AND BY the negation of the second element IN ascending order  
        SET dp TO an empty list  
        FOR each element in envelopes AS underscore AND h  
            SET idx TO the position found by binary searching for h IN dp from the left  
            IF idx EQUALS the length of dp  
                APPEND h TO dp  
            ELSE  
                SET the element at position idx of dp TO h  
            END IF  
        END FOR  
        RETURN the length of dp  
    END FUNCTION  
END CLASS
