CLASS Solution  
    FUNCTION canJump(nums)  
        SET farthest TO zero  
        FOR i FROM zero TO LENGTH OF nums MINUS one  
            IF i GREATER THAN farthest  
                RETURN False  
            END IF  
            SET farthest TO the GREATER OF farthest AND i PLUS element at position i of nums  
        END FOR  
        RETURN farthest GREATER THAN OR EQUAL TO LENGTH OF nums MINUS one  
    END FUNCTION  
END CLASS
