CLASS Solution  
    FUNCTION setZeroes(matrix)  
        IF matrix EQUALS nothing OR the LENGTH OF matrix EQUALS zero OR the LENGTH OF the element at position zero of matrix EQUALS zero  
            RETURN  
        END IF  

        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF the element at position zero of matrix  
        SET first_row_has_zero TO false  
        FOR j FROM zero TO n MINUS one  
            IF the element at position zero of matrix AT position j EQUALS zero  
                SET first_row_has_zero TO true  
            END IF  
        END FOR  

        SET first_col_has_zero TO false  
        FOR i FROM zero TO m MINUS one  
            IF the element at position i of matrix AT position zero EQUALS zero  
                SET first_col_has_zero TO true  
            END IF  
        END FOR  

        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF the element at position i of matrix AT position j EQUALS zero  
                    SET the element at position i of matrix AT position zero TO zero  
                    SET the element at position zero of matrix AT position j TO zero  
                END IF  
            END FOR  
        END FOR  

        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF the element at position i of matrix AT position zero EQUALS zero OR the element at position zero of matrix AT position j EQUALS zero  
                    SET the element at position i of matrix AT position j TO zero  
                END IF  
            END FOR  
        END FOR  

        IF first_row_has_zero  
            FOR j FROM zero TO n MINUS one  
                SET the element at position zero of matrix AT position j TO zero  
            END FOR  
        END IF  

        IF first_col_has_zero  
            FOR i FROM zero TO m MINUS one  
                SET the element at position i of matrix AT position zero TO zero  
            END FOR  
        END IF  
    END FUNCTION  
END CLASS
