CLASS Solution  
    FUNCTION canCompleteCircuit(gas, cost)  
        IF the SUM OF gas IS LESS THAN the SUM OF cost  
            RETURN negative one  
        END IF  
        
        SET start TO zero  
        SET total_tank TO zero  
        
        FOR i FROM zero TO the LENGTH OF gas MINUS one  
            SET total_tank TO total_tank PLUS the element at position i of gas MINUS the element at position i of cost  
            
            IF total_tank IS LESS THAN zero  
                SET start TO i PLUS one  
                SET total_tank TO zero  
            END IF  
        END FOR  
        
        RETURN start  
    END FUNCTION  
END CLASS
