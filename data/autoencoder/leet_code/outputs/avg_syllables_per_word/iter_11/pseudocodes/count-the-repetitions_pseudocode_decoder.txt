CLASS Solution  
    FUNCTION getMaxRepetitions(s1, n1, s2, n2)  
        IF n1 EQUALS zero  
            RETURN zero  
        END IF  

        SET s2_count TO a list of zeros with the LENGTH EQUAL TO the LENGTH OF s2 PLUS one  
        SET index_map TO a list of zeros with the LENGTH EQUAL TO the LENGTH OF s2 PLUS one  

        SET current_index_s2 TO zero  
        SET s1_count TO zero  

        WHILE s1_count LESS THAN n1  
            INCREMENT s1_count BY one  
            FOR each char IN s1  
                IF char EQUALS the character at position current_index_s2 of s2  
                    INCREMENT current_index_s2 BY one  
                    IF current_index_s2 EQUALS the LENGTH OF s2  
                        SET the element at position s1_count of s2_count TO the element at position s1_count MINUS one of s2_count PLUS one  
                        SET current_index_s2 TO zero  
                    END IF  
                END IF  
                SET the element at position s1_count of index_map TO current_index_s2  
            END FOR  

            IF current_index_s2 IS IN the slice of index_map from the first element to the element at position s1_count MINUS one  
                SET start TO the position of current_index_s2 in index_map  
                SET cycle_length TO s1_count MINUS start  
                SET cycles_fit TO the integer DIVISION OF the difference of n1 MINUS start BY cycle_length  
                SET remainder TO the MODULO OF the difference of n1 MINUS start BY cycle_length  
                SET total_s2_count TO the element at position start of s2_count PLUS cycles_fit MULTIPLIED BY the difference of the element at position start PLUS cycle_length of s2_count MINUS the element at position start of s2_count PLUS the difference of the element at position start PLUS remainder of s2_count MINUS the element at position start of s2_count  
                RETURN the integer DIVISION OF total_s2_count BY n2  
            END IF  
        END WHILE  

        RETURN the integer DIVISION OF the element at position s1_count of s2_count BY n2  
    END FUNCTION  
END CLASS
