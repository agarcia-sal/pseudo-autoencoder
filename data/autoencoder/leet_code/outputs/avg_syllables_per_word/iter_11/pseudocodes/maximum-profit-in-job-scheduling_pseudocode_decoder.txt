CLASS Solution  
    FUNCTION jobScheduling(startTime, endTime, profit)  
        SET jobs TO the list of tuples of elements at position index of startTime, endTime, and profit sorted by the second element of each tuple in ascending order  
        SET end_times TO the list of the second element of each job in jobs  
        SET dp TO the list of zeroes with length the LENGTH OF jobs PLUS one  
        FOR i FROM one TO the LENGTH OF jobs  
            SET j TO the position where the element at position i MINUS one of jobs start time would be inserted in end_times to maintain sorted order  
            SET dp element at position i TO the maximum value between dp element at position i MINUS one AND dp element at position j PLUS the third element of job at position i MINUS one of jobs  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
