CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a list of n elements each SET TO negative one  
        SET stack TO an empty list  

        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO i MODULO n  
            WHILE stack is not empty AND element at position last element of stack in nums IS LESS THAN element at position index in nums  
                SET popped_index TO the REMOVAL of last element FROM stack  
                SET element at position popped_index in result TO element at position index in nums  
            END WHILE  
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
