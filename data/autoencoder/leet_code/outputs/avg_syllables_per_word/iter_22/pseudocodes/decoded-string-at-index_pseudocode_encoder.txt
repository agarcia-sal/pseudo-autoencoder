CLASS Solution  
    FUNCTION decodeAtIndex(s k)  
        SET length TO zero  
        FOR each char IN s  
            IF char IS a digit character  
                SET length TO length MULTIPLIED BY the numeric value of char  
            ELSE  
                INCREMENT length BY one  
            END IF  
        END FOR  
        FOR each char IN s in reverse order  
            SET k TO k MODULO length  
            IF k EQUALS zero AND char IS an alphabetic character  
                RETURN char  
            END IF  
            IF char IS a digit character  
                SET length TO length DIVIDED BY the numeric value of char  
            ELSE  
                DECREMENT length BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
