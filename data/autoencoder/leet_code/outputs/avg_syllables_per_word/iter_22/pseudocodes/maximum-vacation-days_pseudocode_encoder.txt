CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the FIRST element of days  
        
        SET dp TO a list of n elements each SET TO negative one  
        SET element at position zero of dp TO zero  
        
        SET graph TO a mapping from each integer i FROM zero TO n MINUS one ASSIGNED TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of flights EQUALS one THEN  
                    APPEND j TO element at position i of graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR week FROM zero TO k MINUS one  
            SET new_dp TO a list of n elements each SET TO negative one  
            FOR i FROM zero TO n MINUS one  
                IF element at position i of dp EQUALS negative one THEN  
                    CONTINUE TO next iteration  
                END IF  
                SET element at position i of new_dp TO the GREATER OF element at position i of new_dp AND element at position i of dp PLUS element at position i of days PLUS week position  
                
                FOR each city j IN element at position i of graph  
                    SET element at position j of new_dp TO the GREATER OF element at position j of new_dp AND element at position i of dp PLUS element at position j of days PLUS week position  
                END FOR  
            END FOR  
            ASSIGN reference of new_dp TO dp  
        END FOR  
        
        RETURN the GREATER element of dp  
    END FUNCTION  
END CLASS
