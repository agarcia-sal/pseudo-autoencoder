CLASS Solution  
    FUNCTION convert(s numRows)  
        IF numRows EQUALS one  
            RETURN s  
        END IF  
        SET rows TO a list containing numRows empty strings  
        SET current_row TO zero  
        SET going_down TO false  
        FOR each character IN s  
            SET element at position current_row of rows TO the concatenation of element at position current_row of rows PLUS character  
            IF current_row EQUALS zero OR current_row EQUALS numRows MINUS one  
                SET going_down TO NOT going_down  
            END IF  
            IF going_down  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
        SET result_string TO the concatenation of all elements in rows  
        RETURN result_string  
    END FUNCTION  
END CLASS
