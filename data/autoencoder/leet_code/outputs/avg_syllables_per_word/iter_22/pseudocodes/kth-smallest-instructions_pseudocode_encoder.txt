CLASS Solution  
    FUNCTION kthSmallestPath(destination PARAMETER List of Integer, k PARAMETER Integer) RETURNS String  
        SET row TO element at position one of destination  
        SET col TO element at position two of destination  
        SET path TO empty list  
        SET total_moves TO col PLUS row  
        FOR i FROM one TO total_moves  
            IF col GREATER THAN zero THEN  
                SET paths_with_H TO number of combinations of row PLUS col MINUS one TAKEN col MINUS one  
                IF k LESS THAN OR EQUAL TO paths_with_H THEN  
                    APPEND letter H TO path  
                    DECREMENT col BY one  
                ELSE  
                    APPEND letter V TO path  
                    DECREMENT row BY one  
                    DECREMENT k BY paths_with_H  
                END IF  
            ELSE  
                APPEND letter V TO path  
                DECREMENT row BY one  
            END IF  
        END FOR  
        RETURN concatenation of all elements in path as a single string  
    END FUNCTION  
END CLASS
