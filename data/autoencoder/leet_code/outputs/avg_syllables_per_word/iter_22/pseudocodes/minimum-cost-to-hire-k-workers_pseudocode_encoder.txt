CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO list of pairs of wage element DIVIDED BY quality element AND quality element FOR each pair of wage element AND quality element taken simultaneously from wage AND quality  
        SORT workers BY the first element of each pair in ascending order  
        SET max_heap TO empty list  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each pair ratio AND q IN workers  
            PUSH the negative of q INTO max_heap  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap GREATER THAN k THEN  
                REMOVE the smallest element FROM max_heap AND ADD it TO total_quality  
            END IF  
            IF the LENGTH OF max_heap EQUALS k THEN  
                SET current_cost TO total_quality MULTIPLIED BY ratio  
                IF current_cost LESS THAN min_cost THEN  
                    SET min_cost TO current_cost  
                END IF  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
