CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO A new TreeNode WITH the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT EMPTY  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.left TO A new TreeNode WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.right TO A new TreeNode WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree(self, root)  
        
        FUNCTION dfs(node)  
            IF node IS none THEN  
                RETURN positive infinity, negative infinity, zero  
            END IF  
            SET left_min, left_max, left_size TO dfs(the left child of node)  
            SET right_min, right_max, right_size TO dfs(the right child of node)  
            IF left_max LESS THAN node.val AND node.val LESS THAN right_min THEN  
                RETURN the minimum of node.val AND left_min, the maximum of node.val AND right_max, left_size PLUS right_size PLUS one  
            END IF  
            RETURN negative infinity, positive infinity, the maximum of left_size AND right_size  
        END FUNCTION  
        
        RETURN the third element of dfs(root)  
    END FUNCTION  
END CLASS
