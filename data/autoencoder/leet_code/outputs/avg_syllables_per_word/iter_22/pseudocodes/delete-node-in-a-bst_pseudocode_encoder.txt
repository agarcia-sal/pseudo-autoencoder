CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the list values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty deque collection  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        REMOVE the first element FROM queue AND ASSIGN it TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree with arguments p.left AND q.left AND the RESULT OF is_same_tree with arguments p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode(self, root, key)  
        IF root IS None  
            RETURN None  
        END IF  
        IF root.val IS GREATER THAN key  
            SET root.left TO the RESULT OF calling deleteNode of self with arguments root.left AND key  
            RETURN root  
        END IF  
        IF root.val IS LESS THAN key  
            SET root.right TO the RESULT OF calling deleteNode of self with arguments root.right AND key  
            RETURN root  
        END IF  
        IF root.left IS None  
            RETURN root.right  
        END IF  
        IF root.right IS None  
            RETURN root.left  
        END IF  
        SET node TO root.right  
        WHILE node.left IS NOT None  
            SET node TO node.left  
        END WHILE  
        SET node.left TO root.left  
        SET root TO root.right  
        RETURN root  
    END FUNCTION  
END CLASS
