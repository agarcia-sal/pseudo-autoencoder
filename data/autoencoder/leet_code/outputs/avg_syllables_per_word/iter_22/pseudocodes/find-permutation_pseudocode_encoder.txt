CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO empty list  
        FOR index FROM zero TO the LENGTH OF s PLUS one  
            APPEND index PLUS one TO result  
        END FOR  
        SET start TO zero  
        FOR end TO index FROM zero TO the LENGTH OF s  
            IF end EQUALS the LENGTH OF s OR element at position end of s EQUALS the character I  
                SET segment TO the elements from position start TO position end PLUS one of result  
                REVERSE the order of elements in segment  
                SET the elements from position start TO position end PLUS one of result TO segment  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
