CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        FOR each char IN s  
            IF char represents a digit  
                SET d TO d PLUS char  
            ELSE IF char represents an alphabetic letter  
                SET a TO a PLUS char  
            ELSE IF char EQUALS the character that marks the beginning of a bracket  
                APPEND the pair consisting of a and the integer value of d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char EQUALS the character that marks the end of a bracket  
                SET p and n TO the last element removed FROM stack  
                SET a TO p PLUS a repeated n times  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
