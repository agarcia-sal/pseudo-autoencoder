CLASS Solution  
    FUNCTION numDistinctIslands2(grid)  
        FUNCTION dfs(x, y, island)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR element at position x of grid at position y EQUALS zero THEN  
                RETURN  
            END IF  
            SET element at position x of grid at position y TO zero  
            APPEND tuple of x AND y TO island  
            CALL dfs with arguments x PLUS one AND y AND island  
            CALL dfs with arguments x MINUS one AND y AND island  
            CALL dfs with arguments x AND y PLUS one AND island  
            CALL dfs with arguments x AND y MINUS one AND island  
        END FUNCTION  

        FUNCTION normalize(island)  
            SET shapes TO empty list  
            FOR each pair dx AND dy IN list of pairs consisting of one AND one PLUS one AND MINUS one MINUS one AND one MINUS one AND MINUS one  
                SET new_island TO empty list  
                FOR each pair x AND y IN island  
                    APPEND tuple of x MULTIPLIED BY dx AND y MULTIPLIED BY dy TO new_island  
                END FOR  
                SORT new_island  
                SET min_x TO minimum value of x for each pair x AND y IN new_island  
                SET min_y TO minimum value of y for each pair x AND y IN new_island  
                APPEND tuple of pairs of x MINUS min_x AND y MINUS min_y for each pair x AND y IN new_island TO shapes  
            END FOR  
            FOR i FROM zero TO the LENGTH OF shapes MINUS one  
                APPEND tuple of pairs of y AND MINUS x for each pair x AND y IN element at position i of shapes TO shapes  
            END FOR  
            RETURN minimum element of shapes  
        END FUNCTION  

        SET islands TO empty set  
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one THEN  
                    SET island TO empty list  
                    CALL dfs with arguments i AND j AND island  
                    ADD result of calling normalize with argument island TO islands  
                END IF  
            END FOR  
        END FOR  
        RETURN the LENGTH OF islands  
    END FUNCTION  
END CLASS
