CLASS Solution  
    FUNCTION search(nums target)  
        IF the COLLECTION nums DOES NOT HAVE ANY ELEMENTS  
            RETURN false  
        END IF  
        
        SET variable left TO zero  
        SET variable right TO the LENGTH OF the COLLECTION nums MINUS one  
        
        WHILE variable left IS LESS THAN OR EQUAL TO variable right  
            SET variable mid TO variable left PLUS the DIFFERENCE OF variable right MINUS variable left DIVIDED BY two  
            
            IF the ELEMENT at position variable mid of the COLLECTION nums EQUALS variable target  
                RETURN true  
            END IF  
            
            IF the ELEMENT at position variable left of the COLLECTION nums EQUALS the ELEMENT at position variable mid of the COLLECTION nums AND the ELEMENT at position variable mid of the COLLECTION nums EQUALS the ELEMENT at position variable right of the COLLECTION nums  
                INCREMENT variable left BY one  
                DECREMENT variable right BY one  
            ELSE IF the ELEMENT at position variable left of the COLLECTION nums IS LESS THAN OR EQUAL TO the ELEMENT at position variable mid of the COLLECTION nums  
                IF the ELEMENT at position variable left of the COLLECTION nums IS LESS THAN OR EQUAL TO variable target AND variable target IS LESS THAN the ELEMENT at position variable mid of the COLLECTION nums  
                    SET variable right TO variable mid MINUS one  
                ELSE  
                    SET variable left TO variable mid PLUS one  
                END IF  
            ELSE  
                IF the ELEMENT at position variable mid of the COLLECTION nums IS LESS THAN variable target AND variable target IS LESS THAN OR EQUAL TO the ELEMENT at position variable right of the COLLECTION nums  
                    SET variable left TO variable mid PLUS one  
                ELSE  
                    SET variable right TO variable mid MINUS one  
                END IF  
            END IF  
        END WHILE  
        
        RETURN false  
    END FUNCTION  
END CLASS
