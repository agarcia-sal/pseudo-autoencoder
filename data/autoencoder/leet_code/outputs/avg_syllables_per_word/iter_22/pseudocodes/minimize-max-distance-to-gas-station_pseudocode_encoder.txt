CLASS Solution  
    FUNCTION minmaxGasDist(stations FOR List OF INTEGER, k) RETURNS FLOAT  
        FUNCTION possible(max_dist) RETURNS BOOLEAN  
            SET additional_stations TO zero  
            FOR index FROM one TO LENGTH OF stations MINUS one  
                INCREMENT additional_stations BY the INTEGER PART OF the DIFFERENCE BETWEEN the element at position index of stations AND the element at position index MINUS one of stations DIVIDED BY max_dist  
            END FOR  
            RETURN additional_stations LESS THAN OR EQUAL TO k  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE the DIFFERENCE BETWEEN right AND left GREATER THAN one MILLIONTH  
            SET mid TO the SUM OF left AND right DIVIDED BY two  
            IF possible(mid)  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
