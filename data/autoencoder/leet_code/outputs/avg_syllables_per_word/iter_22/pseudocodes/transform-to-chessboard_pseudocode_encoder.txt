CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF the ELEMENT at position i OF board AND THE ELEMENT at position j OF THAT ROW EXCLUSIVE OR the ELEMENT at position i OF board AND THE ELEMENT at position zero OF THAT ROW EXCLUSIVE OR the ELEMENT at position zero OF board AND THE ELEMENT at position j OF THAT ROW EXCLUSIVE OR the ELEMENT at position zero OF board AND THE ELEMENT at position zero OF THAT ROW  
                    RETURN minus one  
                END IF  
            END FOR  
            INCREMENT rowSum BY the ELEMENT at position i OF board AND THE ELEMENT at position zero OF THAT ROW  
            INCREMENT colSum BY the ELEMENT at position zero OF board AND THE ELEMENT at position i OF THAT ROW  
            INCREMENT rowSwap BY WHETHER the ELEMENT at position i OF board AND THE ELEMENT at position zero OF THAT ROW EQUALS the REMAINDER when i DIVIDED BY two  
            INCREMENT colSwap BY WHETHER the ELEMENT at position zero OF board AND THE ELEMENT at position i OF THAT ROW EQUALS the REMAINDER when i DIVIDED BY two  
        END FOR  
        IF the INTEGER DIVISION of N BY two IS GREATER THAN rowSum OR rowSum IS GREATER THAN the INTEGER DIVISION of the SUM of N PLUS one BY two  
            RETURN minus one  
        END IF  
        IF the INTEGER DIVISION of N BY two IS GREATER THAN colSum OR colSum IS GREATER THAN the INTEGER DIVISION of the SUM of N PLUS one BY two  
            RETURN minus one  
        END IF  
        IF the REMAINDER when N DIVIDED BY two IS NOT EQUALS zero  
            IF the REMAINDER when rowSwap DIVIDED BY two IS NOT EQUALS zero  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the REMAINDER when colSwap DIVIDED BY two IS NOT EQUALS zero  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO THE MINIMUM OF rowSwap AND N MINUS rowSwap  
            SET colSwap TO THE MINIMUM OF colSwap AND N MINUS colSwap  
        END IF  
        SET result TO the INTEGER DIVISION of the SUM of rowSwap PLUS colSwap BY two  
        RETURN result  
    END FUNCTION  
END CLASS
