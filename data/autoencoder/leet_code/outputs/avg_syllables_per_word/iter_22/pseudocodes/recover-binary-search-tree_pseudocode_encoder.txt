CLASS TreeNode  
    FUNCTION __init__(self val EQUALS zero left EQUALS None right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value being the element at position one of values  
    SET i TO one  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode with value being the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode with value being the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with arguments p.left AND q.left AND the result of calling is_same_tree with arguments p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION recoverTree(root)  
        SET first TO None  
        SET second TO None  
        SET prev TO None  
        SET curr TO root  
        WHILE curr IS NOT None  
            IF curr.left EQUALS None  
                IF prev IS NOT None AND prev.val GREATER THAN curr.val  
                    IF first EQUALS None  
                        SET first TO prev  
                    END IF  
                    SET second TO curr  
                END IF  
                SET prev TO curr  
                SET curr TO curr.right  
            ELSE  
                SET pred TO curr.left  
                WHILE pred.right IS NOT None AND pred.right NOT EQUALS curr  
                    SET pred TO pred.right  
                END WHILE  
                IF pred.right EQUALS None  
                    SET pred.right TO curr  
                    SET curr TO curr.left  
                ELSE  
                    SET pred.right TO None  
                    IF prev IS NOT None AND prev.val GREATER THAN curr.val  
                        IF first EQUALS None  
                            SET first TO prev  
                        END IF  
                        SET second TO curr  
                    END IF  
                    SET prev TO curr  
                    SET curr TO curr.right  
                END IF  
            END IF  
        END WHILE  
        IF first IS NOT None AND second IS NOT None  
            SET a temporary variable TO first.val  
            SET first.val TO second.val  
            SET second.val TO the temporary variable  
        END IF  
        RETURN None  
    END FUNCTION  
END CLASS
