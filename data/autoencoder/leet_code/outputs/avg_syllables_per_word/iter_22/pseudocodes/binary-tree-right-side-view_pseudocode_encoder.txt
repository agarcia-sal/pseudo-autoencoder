CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty DO  
        SET node TO the element removed from the left of queue  
        IF i is LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value THEN  
        RETURN boolean true  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN boolean false  
    ELSE IF the value of p IS NOT EQUALS the value of q THEN  
        RETURN boolean false  
    ELSE  
        RETURN the result of is_same_tree called with the left child of p AND the left child of q AND the result of is_same_tree called with the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView(self, root)  
        IF root IS no value THEN  
            RETURN an empty list  
        END IF  
        SET right_side_view TO an empty list  
        SET queue TO a double ended queue initialized with root as its only element  
        WHILE the queue is not empty DO  
            SET level_length TO the LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one DO  
                SET node TO the element removed from the left of queue  
                IF i EQUALS level_length MINUS one THEN  
                    APPEND the value of node TO right_side_view  
                END IF  
                IF the left child of node IS NOT no value THEN  
                    APPEND the left child of node TO queue  
                END IF  
                IF the right child of node IS NOT no value THEN  
                    APPEND the right child of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
