CLASS Solution  
    FUNCTION rotate(nums parameter of type list of integer, k parameter of type integer) RETURNS none  
        SET n TO the LENGTH OF nums  
        SET k TO k MODULO n  
        
        FUNCTION reverse(start parameter of type integer, end parameter of type integer) RETURNS none  
            WHILE start LESS THAN end  
                SET temporary TO element at position start of nums  
                SET element at position start of nums TO element at position end of nums  
                SET element at position end of nums TO temporary  
                INCREMENT start BY one  
                DECREMENT end BY one  
            END WHILE  
        END FUNCTION  
        
        CALL reverse WITH start AS zero AND end AS n MINUS one  
        CALL reverse WITH start AS zero AND end AS k MINUS one  
        CALL reverse WITH start AS k AND end AS n MINUS one  
    END FUNCTION  
END CLASS
