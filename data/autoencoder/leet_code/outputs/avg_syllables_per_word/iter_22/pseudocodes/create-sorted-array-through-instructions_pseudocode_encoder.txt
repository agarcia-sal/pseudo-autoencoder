CLASS Solution  
    FUNCTION createSortedArray(instructions)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET MAX TO the maximum element in instructions PLUS one  
        SET bit TO list of zero of length MAX  

        FUNCTION update(idx val)  
            WHILE idx LESS THAN MAX  
                INCREMENT element at position idx of bit BY val  
                SET idx TO idx PLUS the result of bitwise AND of idx and the negative of idx  
            END WHILE  
        END FUNCTION  

        FUNCTION query(idx)  
            SET total TO zero  
            WHILE idx GREATER THAN zero  
                INCREMENT total BY element at position idx of bit  
                SET idx TO idx MINUS the result of bitwise AND of idx and the negative of idx  
            END WHILE  
            RETURN total  
        END FUNCTION  

        SET total_cost TO zero  
        FOR each index i AND element num IN the enumeration of instructions  
            SET less_than TO query of num MINUS one  
            SET greater_than TO i MINUS query of num  
            IF less_than LESS THAN greater_than  
                SET cost TO less_than  
            ELSE  
                SET cost TO greater_than  
            END IF  
            SET total_cost TO the remainder of total_cost PLUS cost DIVIDED BY MOD  
            CALL update with num AND one  
        END FOR  

        RETURN total_cost  
    END FUNCTION  
END CLASS
