CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO list of elements obtained by splitting path at each occurrence of the character slash  
        SET stack TO empty list  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS single character period  
                CONTINUE to next iteration of the loop  
            ELSE IF component EQUALS two characters periods  
                IF the LENGTH OF stack IS GREATER THAN zero  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO the character slash CONCATENATED WITH all elements of stack joined by the character slash as delimiter  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
