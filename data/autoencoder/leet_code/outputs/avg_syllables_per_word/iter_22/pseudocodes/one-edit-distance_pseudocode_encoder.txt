CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t  
            RETURN self DOT isOneEditDistance with the first argument t AND the second argument s  
        END IF  
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        IF diff GREATER THAN one  
            RETURN False  
        END IF  
        IF diff EQUALS zero  
            SET count_of_differences TO zero  
            FOR each index FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position index of s NOT EQUALS the element at position index of t  
                    INCREMENT count_of_differences BY one  
                END IF  
            END FOR  
            RETURN whether count_of_differences EQUALS one  
        ELSE  
            FOR each index FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position index of s NOT EQUALS the element at position index of t  
                    RETURN whether the substring of s from position index TO the END EQUALS the substring of t from position index PLUS one TO the END  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
