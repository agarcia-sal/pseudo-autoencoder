CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals by the value of element at position two of each interval AND by the negative value of element at position one of each interval in descending order  
        SET first TO the value of element at position two of the first interval MINUS one  
        SET second TO the value of element at position two of the first interval  
        SET count TO two  
        FOR each interval in intervals starting from the second interval  
            SET start TO the value of element at position one of the interval  
            SET end TO the value of element at position two of the interval  
            IF start IS GREATER THAN second THEN  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first THEN  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
