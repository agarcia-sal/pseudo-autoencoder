CLASS Solution  
    FUNCTION longestSubstring(s k)  
        IF the LENGTH OF s LESS THAN k THEN  
            RETURN zero  
        END IF  
        ASSIGN count TO a MAPPING of characters TO their OCCURRENCE COUNT in s  
        FOR each character IN the KEYS OF count  
            IF the value MAPPED FROM character IN count LESS THAN k THEN  
                RETURN the MAXIMUM OF longestSubstring(substring k) FOR each substring IN the RESULT OF SPLITTING s BY character  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
