CLASS Solution  
    FUNCTION exclusiveTime(n, logs)  
        SET exclusive_time TO a list of the value zero repeated n times  
        SET stack TO an empty list  
        
        FOR each log IN logs  
            SET function_id, action, timestamp TO the three parts of log separated by the character colon  
            SET function_id TO function_id converted to an integer  
            SET timestamp TO timestamp converted to an integer  
            
            IF action EQUALS the word start  
                IF the LENGTH OF stack IS greater than zero  
                    SET prev_function_id, prev_timestamp TO the last element in stack  
                    SET exclusive_time at position prev_function_id TO exclusive_time at position prev_function_id PLUS timestamp MINUS prev_timestamp  
                END IF  
                APPEND the pair of function_id and timestamp TO stack  
            ELSE  
                SET prev_function_id, prev_timestamp TO the last element in stack removed from stack  
                SET exclusive_time at position prev_function_id TO exclusive_time at position prev_function_id PLUS timestamp MINUS prev_timestamp PLUS one  
                IF the LENGTH OF stack IS greater than zero  
                    SET the last element of stack TO the pair of the first element of the last element of stack and timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
