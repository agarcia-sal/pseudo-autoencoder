CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        IF the sum of m PLUS n MINUS one MODULO two NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        
        SET dp TO a three dimensional list of False VALUES with dimensions from zero TO m PLUS one FOR the first dimension, from zero TO n PLUS one FOR the second dimension, and from zero TO m PLUS n PLUS one FOR the third dimension  
        
        SET element at position zero of element at position one of dp TO True  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR k FROM zero TO m PLUS n  
                    IF element at position i of element at position j PLUS one of dp OR element at position i PLUS one of element at position j of dp IS True THEN  
                        IF element at position i of element at position j of grid EQUALS open parenthesis AND k PLUS one LESS THAN m PLUS n PLUS one THEN  
                            SET element at position i PLUS one of element at position j PLUS one of dp at position k PLUS one TO True  
                        ELSE IF element at position i of element at position j of grid EQUALS close parenthesis AND k MINUS one GREATER THAN OR EQUAL TO zero THEN  
                            SET element at position i PLUS one of element at position j PLUS one of dp at position k MINUS one TO True  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position m of element at position n of dp at position zero  
    END FUNCTION  
END CLASS
