CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO a new mapping from values to empty lists  
        FOR each index named i AND each element named value IN arr with index  
            APPEND i TO the list at key value IN value_indices  
        END FOR  
        
        SET queue TO a new double ended queue containing the pair of zero and zero  
        SET visited TO a new set containing zero  
        
        WHILE queue is not empty  
            REMOVE FROM the front OF queue AND ASSIGN TO current_index AND steps  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO the list containing current_index PLUS one, current_index MINUS one, concatenated WITH the list retrieved from value_indices at the element at position current_index of arr  
            
            FOR each next_index IN next_indices  
                IF next_index GREATER THAN OR EQUAL TO zero AND next_index LESS THAN n AND next_index NOT IN visited  
                    ADD next_index TO visited  
                    APPEND the pair of next_index AND steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list at key element at position current_index of arr IN value_indices TO an empty list  
        END WHILE  
        
        RETURN the negative of one  
    END FUNCTION  
END CLASS
