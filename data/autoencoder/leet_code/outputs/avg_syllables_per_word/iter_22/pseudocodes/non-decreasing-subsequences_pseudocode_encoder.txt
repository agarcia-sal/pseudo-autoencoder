CLASS Solution  
    FUNCTION findSubsequences(nums)  
        FUNCTION dfs(u, last, t)  
            IF u EQUALS the LENGTH OF nums  
                IF the LENGTH OF t GREATER THAN one  
                    APPEND a COPY OF t TO ans  
                END IF  
                RETURN  
            END IF  
            IF element at position u of nums GREATER THAN OR EQUAL TO last  
                APPEND element at position u of nums TO t  
                CALL dfs WITH arguments u PLUS one last element at position u of nums t  
                REMOVE the last element FROM t  
            END IF  
            IF element at position u of nums NOT EQUALS last  
                CALL dfs WITH arguments u PLUS one last t  
            END IF  
        END FUNCTION  
        SET ans TO an empty list  
        CALL dfs WITH arguments zero negative one thousand t as an empty list  
        RETURN ans  
    END FUNCTION  
END CLASS
