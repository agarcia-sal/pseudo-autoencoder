CLASS Solution  
    FUNCTION reachingPoints WITH PARAMETERS sx sy tx ty  
        WHILE the value of tx IS GREATER THAN OR EQUAL TO the value of sx AND the value of ty IS GREATER THAN OR EQUAL TO the value of sy  
            IF the value of tx EQUALS the value of sx AND the value of ty EQUALS the value of sy  
                RETURN True  
            END IF  
            IF the value of tx IS GREATER THAN the value of ty  
                IF the value of ty IS GREATER THAN the value of sy  
                    SET tx TO the remainder of the value of tx DIVIDED BY the value of ty  
                ELSE  
                    RETURN the remainder of the value of tx MINUS the value of sx DIVIDED BY the value of ty EQUALS zero  
                END IF  
            ELSE  
                IF the value of tx IS GREATER THAN the value of sx  
                    SET ty TO the remainder of the value of ty DIVIDED BY the value of tx  
                ELSE  
                    RETURN the remainder of the value of ty MINUS the value of sy DIVIDED BY the value of tx EQUALS zero  
                END IF  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
