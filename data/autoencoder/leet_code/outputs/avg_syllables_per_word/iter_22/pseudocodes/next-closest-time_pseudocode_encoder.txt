CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters in time after removing the colon character  
        SET hour TO the integer value of the substring before the colon character in time  
        SET minute TO the integer value of the substring after the colon character in time  
        
        FUNCTION generate_times(digits)  
            SET times TO empty list  
            FOR each first hour digit IN digits  
                FOR each second hour digit IN digits  
                    FOR each first minute digit IN digits  
                        FOR each second minute digit IN digits  
                            SET combined_hour_string TO concatenation of first hour digit and second hour digit  
                            SET combined_minute_string TO concatenation of first minute digit and second minute digit  
                            SET hour_value TO the integer value of combined_hour_string  
                            SET minute_value TO the integer value of combined_minute_string  
                            IF hour_value LESS THAN twenty four AND minute_value LESS THAN sixty THEN  
                                APPEND tuple of hour_value and minute_value TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN times sorted in ascending order  
        END FUNCTION  
        
        SET all_times TO result of calling generate_times with digits  
        SET current_time_index TO the position of the tuple of hour and minute in all_times  
        
        IF current_time_index PLUS one LESS THAN the length of all_times THEN  
            SET next_hour, next_minute TO the element at the position current_time_index PLUS one in all_times  
        ELSE  
            SET next_hour, next_minute TO the element at the first position in all_times  
        END IF  
        
        RETURN the string formed by formatting next_hour as two digits character length PLUS colon character PLUS next_minute as two digits character length  
    END FUNCTION  
END CLASS
