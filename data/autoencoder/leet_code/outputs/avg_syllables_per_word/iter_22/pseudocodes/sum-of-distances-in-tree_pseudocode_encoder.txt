CLASS Solution  
    FUNCTION sumOfDistancesInTree(n, edges)  
        SET graph TO a mapping from nodes to empty lists  
        FOR each pair of nodes a AND b IN edges  
            APPEND node b TO the list of neighbors of node a in graph  
            APPEND node a TO the list of neighbors of node b in graph  
        END FOR  

        SET count TO a list of size n where each element is one  
        SET dist TO a list of size n where each element is zero  

        FUNCTION dfs1(u, p)  
            FOR each node v IN the list of neighbors of node u in graph  
                IF node v NOT EQUALS node p THEN  
                    CALL dfs1 with node v and node u  
                    INCREMENT count at position u BY count at position v  
                    INCREMENT dist at position u BY dist at position v PLUS count at position v  
                END IF  
            END FOR  
        END FUNCTION  

        CALL dfs1 with zero and negative one  

        FUNCTION dfs2(u, p)  
            FOR each node v IN the list of neighbors of node u in graph  
                IF node v NOT EQUALS node p THEN  
                    SET dist at position v TO dist at position u MINUS count at position v PLUS the total number of nodes n MINUS count at position v  
                    CALL dfs2 with node v and node u  
                END IF  
            END FOR  
        END FUNCTION  

        CALL dfs2 with zero and negative one  

        RETURN dist  
    END FUNCTION  
END CLASS
