CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO empty list  
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND asteroid IS LESS THAN zero AND the element at position the last index of stack IS GREATER THAN zero  
                IF the element at position the last index of stack IS LESS THAN the NEGATION OF asteroid  
                    REMOVE the element at position the last index FROM stack  
                    CONTINUE to the next iteration of the WHILE loop  
                ELSE IF the element at position the last index of stack EQUALS the NEGATION OF asteroid  
                    REMOVE the element at position the last index FROM stack  
                END IF  
                BREAK from the WHILE loop  
            END WHILE  
            IF the condition of the WHILE loop IS NOT true  
                APPEND asteroid TO stack  
            END IF  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
