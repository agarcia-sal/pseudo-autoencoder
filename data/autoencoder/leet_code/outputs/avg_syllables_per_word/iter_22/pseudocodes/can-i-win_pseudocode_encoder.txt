CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the product of maxChoosableInteger AND the sum of maxChoosableInteger PLUS one IS DIVIDED BY two LESS THAN desiredTotal  
            RETURN false  
        END IF  
  
        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN true  
        END IF  
  
        SET memo TO an empty dictionary  
  
        FUNCTION can_win(current_total, used_numbers)  
            IF a pair of current_total AND used_numbers IS A KEY IN memo  
                RETURN the value in memo corresponding to the pair of current_total AND used_numbers  
            END IF  
  
            FOR number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers AND the number one LEFT SHIFTED BY number IS EQUALS zero  
                    IF the sum of current_total PLUS number IS GREATER THAN OR EQUAL TO desiredTotal OR the negation of can_win of the sum of current_total PLUS number AND the bitwise OR of used_numbers AND the number one LEFT SHIFTED BY number  
                        SET the value in memo corresponding to the pair of current_total AND used_numbers TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
  
            SET the value in memo corresponding to the pair of current_total AND used_numbers TO false  
            RETURN false  
        END FUNCTION  
  
        RETURN the result of can_win with zero AND zero  
    END FUNCTION  
END CLASS
