CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num EQUALS one  
            RETURN one  
        END IF  
        
        SET factors TO an empty list  
        FOR digit FROM nine TO two DECREMENT BY one  
            WHILE num MODULO digit EQUALS zero  
                APPEND digit TO factors  
                SET num TO num DIVIDED BY digit  
            END WHILE  
        END FOR  
        
        IF num NOT EQUALS one  
            RETURN zero  
        END IF  
        
        SORT factors IN ascending order  
        SET result TO the integer formed by concatenating each element of factors expressed in natural language in order  
        
        IF result GREATER THAN two to the power of thirty-one MINUS one  
            RETURN zero  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
