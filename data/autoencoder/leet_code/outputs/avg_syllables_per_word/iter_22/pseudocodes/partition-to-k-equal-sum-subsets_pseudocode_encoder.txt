CLASS Solution  
    FUNCTION canPartitionKSubsets(nums, k)  
        SET total_sum TO the sum of the elements in nums  
        IF the remainder when total_sum is divided by k NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        SET target TO the quotient of total_sum DIVIDED BY k  
        SORT nums in descending order  
        
        FUNCTION can_partition(index, k, current_sum, used)  
            IF k EQUALS zero THEN  
                RETURN True  
            END IF  
            IF current_sum EQUALS target THEN  
                RETURN can_partition with parameters zero, k MINUS one, zero, and used  
            END IF  
            
            FOR i FROM index TO the length of nums MINUS one  
                IF element at position i of used EQUALS False AND current_sum PLUS element at position i of nums LESS THAN OR EQUAL TO target THEN  
                    SET element at position i of used TO True  
                    IF can_partition with parameters i PLUS one, k, current_sum PLUS element at position i of nums, and used RETURNS True THEN  
                        RETURN True  
                    END IF  
                    SET element at position i of used TO False  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET used TO a list of False values with the same length as nums  
        RETURN can_partition with parameters zero, k, zero, and used  
    END FUNCTION  
END CLASS
