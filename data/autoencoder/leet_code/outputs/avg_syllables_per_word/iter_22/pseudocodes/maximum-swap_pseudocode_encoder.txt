CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO a list of the characters representing num converted to string  
        SET last TO an empty dictionary  
        FOR each index i AND digit d IN digits  
            SET last at the integer value of digit d TO index i  
        END FOR  
        FOR each index i AND digit d IN digits  
            FOR each digit_candidate FROM nine DOWN TO the integer value of digit d PLUS one DECREMENTING BY one  
                IF last at digit_candidate EXISTS AND last at digit_candidate GREATER THAN i  
                    SET temp TO element at position i of digits  
                    SET element at position i of digits TO element at position last at digit_candidate of digits  
                    SET element at position last at digit_candidate of digits TO temp  
                    RETURN the integer conversion of the concatenation of all elements in digits  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
