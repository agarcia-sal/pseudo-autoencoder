CLASS Solution  
    FUNCTION lexicalOrder WITH PARAMETER n  
        SET result TO empty list  
        
        FUNCTION dfs WITH PARAMETER current  
            IF current GREATER THAN n  
                RETURN  
            END IF  
            APPEND current TO result  
            FOR i FROM zero TO nine  
                SET next_num TO current MULTIPLIED BY ten PLUS i  
                IF next_num GREATER THAN n  
                    RETURN  
                END IF  
                CALL dfs WITH PARAMETER next_num  
            END FOR  
        END FUNCTION  
        
        FOR i FROM one TO nine  
            CALL dfs WITH PARAMETER i  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
