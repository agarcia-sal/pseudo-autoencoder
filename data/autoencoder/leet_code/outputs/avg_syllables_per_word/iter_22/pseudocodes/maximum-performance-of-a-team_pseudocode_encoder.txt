CLASS Solution  
    FUNCTION maxPerformance(n speed efficiency k)  
        SET engineers TO list of pairs each consisting of element at position index of efficiency AND element at position index of speed FOR each index FROM zero TO n MINUS one  
        SORT engineers into descending order by the first element of each pair  
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair eff spd IN engineers  
            ADD spd TO speed_heap using heap push operation  
            INCREMENT speed_sum BY spd  
            IF the length of speed_heap GREATER THAN k THEN  
                SET removed_speed TO element extracted from speed_heap using heap pop operation  
                DECREMENT speed_sum BY removed_speed  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance GREATER THAN max_performance THEN  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        RETURN max_performance MODULO value one billion PLUS seven  
    END FUNCTION  
END CLASS
