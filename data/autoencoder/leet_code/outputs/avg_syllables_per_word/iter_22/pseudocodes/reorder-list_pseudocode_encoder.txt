CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the member named val TO the parameter val  
        SET the member named next TO the parameter next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the parameter values is empty  
        RETURN no value  
    END IF  
    SET head TO a new ListNode object initialized with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two to the end  
        SET node TO a new ListNode object initialized with val  
        SET the member next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 IS no value AND p2 IS no value  
        RETURN true  
    END IF  
    IF p1 IS no value OR p2 IS no value  
        RETURN false  
    END IF  
    RETURN the condition that the member val of p1 EQUALS the member val of p2 AND the recursive call of is_same_list with the members next of p1 AND p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList with parameter head  
        IF head IS no value OR the member next of head IS no value  
            RETURN no value  
        END IF  
        SET slow TO head  
        SET fast TO head  
        WHILE fast IS NOT no value AND the member next of fast IS NOT no value  
            SET slow TO the member next of slow  
            SET fast TO the member next of the member next of fast  
        END WHILE  
        SET prev TO no value  
        SET curr TO slow  
        WHILE curr IS NOT no value  
            SET next_temp TO the member next of curr  
            SET the member next of curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  
        SET first TO head  
        SET second TO prev  
        WHILE the member next of second IS NOT no value  
            SET temp1 TO the member next of first  
            SET temp2 TO the member next of second  
            SET the member next of first TO second  
            SET the member next of second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
