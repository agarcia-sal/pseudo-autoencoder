CLASS Solution  
    FUNCTION topKFrequent(words AS list of strings, k AS integer) RETURNS list of strings  
        SET count TO the frequency mapping of each word in words  
        SET heap TO an empty list  
        FOR each word and its frequency IN count entries  
            APPEND a pair of negative frequency and the word TO heap  
        END FOR  
        TRANSFORM heap INTO a heap data structure  
        SET result TO an empty list  
        FOR index FROM one TO k  
            REMOVE the smallest element FROM heap  
            SET word_at_second_position TO the second element of the removed pair  
            APPEND word_at_second_position TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
