CLASS Solution  
    FUNCTION shortestPath(grid k)  
        IF grid EQUALS absence OF value OR element at position zero of grid EQUALS absence OF value  
            RETURN value negative one  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        IF k GREATER THAN OR EQUAL TO m PLUS n MINUS two  
            RETURN m PLUS n MINUS two  
        END IF  
        
        SET directions TO the list containing the pairs zero AND one AND one AND zero AND zero AND negative one AND negative one AND zero  
        
        SET queue TO a double ended queue initialized with the tuple of zero zero k and zero  
        
        SET visited TO a set initialized with the tuple of zero zero k  
        
        WHILE the LENGTH OF queue GREATER THAN zero  
            REMOVE AND SET x y remaining_k steps TO the element removed from the left of queue  
            
            IF x EQUALS m MINUS one AND y EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            FOR each dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n  
                    SET new_k TO remaining_k MINUS the element at position nx of grid at position ny  
                    IF new_k GREATER THAN OR EQUAL TO zero AND the tuple nx ny new_k NOT IN visited  
                        ADD the tuple nx ny new_k TO visited  
                        APPEND the tuple nx ny new_k steps PLUS one TO queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN value negative one  
    END FUNCTION  
END CLASS
