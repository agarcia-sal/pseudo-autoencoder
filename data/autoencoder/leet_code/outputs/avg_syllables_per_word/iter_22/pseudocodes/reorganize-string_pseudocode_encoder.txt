CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO the frequency count of each character IN s  
        SET max_heap TO an empty list  
        FOR each character AND its frequency IN count  
            APPEND the pair of the negative value of frequency AND character TO max_heap  
        END FOR  
        CALL the function to transform max_heap into a heap structure  
        
        SET prev_char TO no value  
        SET prev_freq TO zero  
        SET result TO an empty list  
        
        WHILE max_heap contains elements OR prev_freq is less than zero  
            IF max_heap does not contain elements AND prev_freq is less than zero  
                RETURN the empty string  
            END IF  
            
            EXTRACT the element with the smallest value FROM max_heap and ASSIGN frequency TO freq AND character TO char  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq is less than zero  
                INSERT the pair of prev_freq AND prev_char INTO max_heap maintaining heap properties  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        JOIN all elements IN result INTO a single sequence and RETURN this sequence  
    END FUNCTION  
END CLASS
