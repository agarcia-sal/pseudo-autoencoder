CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO an empty dictionary that returns an empty list for missing keys  
        FOR each string_element IN strs  
            SET sorted_string TO the string obtained by ordering all characters of string_element in ascending alphabetical order and then joining them into a single continuous sequence  
            APPEND string_element TO the list associated with sorted_string in anagrams  
        END FOR  
        RETURN the list of all values contained in anagrams  
    END FUNCTION  
END CLASS
