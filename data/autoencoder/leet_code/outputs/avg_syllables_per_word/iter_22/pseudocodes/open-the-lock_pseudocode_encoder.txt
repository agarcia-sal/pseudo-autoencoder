CLASS Solution  
    FUNCTION openLock(deadends, target)  
        SET deadends_set TO the set constructed from deadends  
        IF the element consisting of four zeros EXISTS IN deadends_set THEN  
            RETURN the integer negative one  
        END IF  
        IF target EQUALS the element consisting of four zeros THEN  
            RETURN the integer zero  
        END IF  
        SET queue TO a double ended queue containing a tuple with the element consisting of four zeros and the integer zero  
        SET visited TO the set containing the element consisting of four zeros  
        
        FUNCTION neighbors(state)  
            SET result TO an empty list  
            FOR index FROM zero TO three  
                SET digit TO the integer representation of the element at position index of state  
                FOR move IN the list containing negative one and one  
                    SET new_digit TO the remainder after division by ten of the sum of digit PLUS move  
                    SET new_state TO the concatenation of substring from position zero TO position index minus one of state  
                                     PLUS the string representation of new_digit  
                                     PLUS substring from position index plus one TO the end of state  
                    APPEND new_state TO result  
                END FOR  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        WHILE queue is not empty  
            REMOVE the leftmost element from queue and assign it to current_state and steps  
            FOR each neighbor IN neighbors(current_state)  
                IF neighbor NOT IN visited AND neighbor NOT IN deadends_set THEN  
                    IF neighbor EQUALS target THEN  
                        RETURN steps PLUS one  
                    END IF  
                    ADD neighbor TO visited  
                    APPEND the tuple consisting of neighbor and steps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN the integer negative one  
    END FUNCTION  
END CLASS
