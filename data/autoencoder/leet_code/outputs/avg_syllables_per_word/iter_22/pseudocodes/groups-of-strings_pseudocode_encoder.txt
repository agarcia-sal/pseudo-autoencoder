CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO list of integers from zero TO size MINUS one  
        SET self.rank TO list of zeroes with length size  
        SET self.size TO list of ones with length size  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u  
            SET element at position u of self.parent TO self.find(element at position u of self.parent)  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET rootU TO self.find(u)  
        SET rootV TO self.find(v)  
        IF rootU NOT EQUALS rootV  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.size BY element at position rootV of self.size  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
                INCREMENT element at position rootV of self.size BY element at position rootU of self.size  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.size BY element at position rootV of self.size  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION groupStrings(self, words)  
        FUNCTION bitmask(word)  
            SET mask TO zero  
            FOR each character IN word  
                SET position TO numerical order of character MINUS numerical order of lowercase a  
                SET mask TO mask OR one shifted left BY position  
            END FOR  
            RETURN mask  
        END FUNCTION  

        SET word_map TO a new mapping from keys to lists  
        SET n TO the LENGTH OF words  
        SET uf TO a new instance of UnionFind with parameter n  

        FOR each index i and word IN words  
            SET mask TO bitmask(word)  
            APPEND i TO list at key mask in word_map  
        END FOR  

        FOR each index i and word IN words  
            SET mask TO bitmask(word)  
            FOR each integer j FROM zero TO twenty five  
                SET new_mask TO mask XOR one shifted left BY j  
                IF new_mask IS a key in word_map  
                    FOR each element k IN list at key new_mask in word_map  
                        CALL uf.union with parameters i and k  
                    END FOR  
                END IF  

                SET bit_check TO mask AND one shifted left BY j  
                IF bit_check EQUALS zero  
                    SET new_mask TO mask OR one shifted left BY j  
                    IF new_mask IS a key in word_map  
                        FOR each element k IN list at key new_mask in word_map  
                            CALL uf.union with parameters i and k  
                        END FOR  
                    END IF  
                END IF  

                IF bit_check NOT EQUALS zero  
                    FOR each integer k FROM zero TO twenty five  
                        SET bit_check_k TO mask AND one shifted left BY k  
                        IF bit_check_k EQUALS zero  
                            SET new_mask TO mask XOR one shifted left BY j  
                            SET new_mask TO new_mask OR one shifted left BY k  
                            IF new_mask IS a key in word_map  
                                FOR each element l IN list at key new_mask in word_map  
                                    CALL uf.union with parameters i and l  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        SET root_sizes TO a new mapping from keys to integers with default zero  
        FOR each integer i FROM zero TO n MINUS one  
            SET root TO uf.find(i)  
            INCREMENT element at key root of root_sizes BY one  
        END FOR  

        SET max_group_size TO the maximum value in root_sizes  
        SET num_groups TO the count of keys in root_sizes  

        RETURN list containing num_groups followed by max_group_size  
    END FUNCTION  
END CLASS
