CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            SET difference_of_first_elements_of_a_and_o TO element at position zero of a MINUS element at position zero of o  
            SET difference_of_second_elements_of_b_and_o TO element at position one of b MINUS element at position one of o  
            SET difference_of_second_elements_of_a_and_o TO element at position one of a MINUS element at position one of o  
            SET difference_of_first_elements_of_b_and_o TO element at position zero of b MINUS element at position zero of o  
            RETURN difference_of_first_elements_of_a_and_o MULTIPLIED BY difference_of_second_elements_of_b_and_o MINUS difference_of_second_elements_of_a_and_o MULTIPLIED BY difference_of_first_elements_of_b_and_o  
        END FUNCTION  
        
        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  
        
        SET orientation TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET o TO element at position i of points  
            SET a TO element at position remainder of i PLUS one DIVIDED BY n of points  
            SET b TO element at position remainder of i PLUS two DIVIDED BY n of points  
            SET cross TO cross_product WITH o a b  
            
            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero  
                    RETURN False  
                ELSE IF orientation EQUALS negative one AND cross GREATER THAN zero  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
