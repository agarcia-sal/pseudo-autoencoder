CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO an empty dictionary with default integer values  
        FOR index i FROM zero TO the length of s MINUS ten PLUS one  
            SET sequence TO the substring from position i TO position i PLUS ten MINUS one of s  
            INCREMENT the value at key sequence in sequence_count BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each sequence IN the keys of sequence_count  
            IF the value at key sequence in sequence_count GREATER THAN one  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
