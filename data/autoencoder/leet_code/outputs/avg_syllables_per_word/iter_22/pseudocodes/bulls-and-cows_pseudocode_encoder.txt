CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        SET secret_count TO empty dictionary with default integer zero  
        SET guess_count TO empty dictionary with default integer zero  
        
        FOR each pair of characters s AND g FROM secret AND guess IN PARALLEL  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value mapped by s IN secret_count BY one  
                INCREMENT the value mapped by g IN guess_count BY one  
            END IF  
        END FOR  
        
        FOR each digit IN keys of secret_count  
            SET minimum_occurrence TO the smaller VALUE between the value mapped by digit IN secret_count AND the value mapped by digit IN guess_count  
            INCREMENT cows BY minimum_occurrence  
        END FOR  
        
        RETURN string describing bulls FOLLOWED BY uppercase letter A FOLLOWED BY cows FOLLOWED BY uppercase letter B  
    END FUNCTION  
END CLASS
