CLASS Solution  
    FUNCTION findLength(nums1, nums2)  
        SET dp TO a two dimensional list where each element is zero with the number of rows equal to the length of nums1 PLUS one and the number of columns equal to the length of nums2 PLUS one  
        SET max_length TO zero  
        FOR i FROM one TO the length of nums1  
            FOR j FROM one TO the length of nums2  
                IF element at position i MINUS one of nums1 EQUALS element at position j MINUS one of nums2  
                    SET element at position i and position j of dp TO element at position i MINUS one and position j MINUS one of dp PLUS one  
                    IF max_length LESS THAN element at position i and position j of dp  
                        SET max_length TO element at position i and position j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
