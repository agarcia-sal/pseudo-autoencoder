CLASS ListNode  
    FUNCTION __init__(self, val ZERO, next NONE)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty list  
        RETURN NONE  
    END IF  
    SET head TO new ListNode with element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two TO the end  
        SET node TO new ListNode with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS NONE AND p2 EQUALS NONE  
        RETURN True  
    END IF  
    IF p1 EQUALS NONE OR p2 EQUALS NONE  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND is_same_list(the next element of p1, the next element of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne(self, head)  
        FUNCTION reverseList(node)  
            SET prev TO NONE  
            WHILE node NOT EQUALS NONE  
                SET next_node TO the next element of node  
                SET the next element of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO reverseList(head)  
        
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current NOT EQUALS NONE AND carry EQUALS one  
            SET current.val TO current.val PLUS carry  
            SET carry TO the integer division of current.val BY ten  
            SET current.val TO the remainder of current.val MODULO ten  
            
            IF the next element of current EQUALS NONE AND carry EQUALS one  
                SET the next element of current TO new ListNode with carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO the next element of current  
        END WHILE  
        
        RETURN reverseList(reversed_head)  
    END FUNCTION  
END CLASS
