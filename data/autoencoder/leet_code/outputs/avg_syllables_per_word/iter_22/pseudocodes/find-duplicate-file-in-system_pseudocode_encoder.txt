CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content TO paths mapping to empty list  
        FOR each path IN paths  
            SET parts TO the list of substrings in path separated by space  
            SET directory TO element at position zero of parts  
            FOR each file_info IN elements from position one TO the end of parts  
                SET file_name TO substring of file_info before the first occurrence of the opening parenthesis  
                SET content_key TO substring of file_info from position after the opening parenthesis TO the position before the closing parenthesis  
                APPEND concatenation of directory and the forward slash and file_name TO content at key content_key  
            END FOR  
        END FOR  
        SET result TO empty list  
        FOR each group IN the values of content  
            IF the LENGTH OF group GREATER THAN one  
                APPEND group TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
