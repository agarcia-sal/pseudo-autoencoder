CLASS Solution  
    FUNCTION canConvert(str1, str2)  
        IF str1 EQUALS str2  
            RETURN True  
        END IF  
        SET conversion_map TO an empty collection of mappings  
        FOR each character1 IN str1 AND corresponding character2 IN str2  
            IF character1 IS A KEY IN conversion_map  
                IF the value AT character1 IN conversion_map NOT EQUALS character2  
                    RETURN False  
                END IF  
            ELSE  
                SET the value AT character1 IN conversion_map TO character2  
            END IF  
        END FOR  
        IF the number of unique characters IN str2 EQUALS twenty six  
            RETURN False  
        END IF  
        RETURN True  
    END FUNCTION  
END CLASS
