CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EQUALS empty THEN  
        RETURN None  
    END IF  
    SET root TO TreeNode(element at position one of values)  
    SET i TO one  
    SET queue TO a new empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO REMOVE element from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT EQUALS None THEN  
            SET node.left TO TreeNode(element at position i of values)  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT EQUALS None THEN  
            SET node.right TO TreeNode(element at position i of values)  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS EQUALS None AND q IS EQUALS None THEN  
        RETURN True  
    ELSE IF p IS EQUALS None OR q IS EQUALS None THEN  
        RETURN False  
    ELSE IF the val attribute of p IS NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalTraversal(self, root)  
        IF root IS EQUALS None THEN  
            RETURN an empty list  
        END IF  
        SET node_dict TO a new default dictionary with lists as default  
        SET min_col TO zero  
        SET max_col TO zero  

        FUNCTION dfs(node, row, col)  
            IF node IS EQUALS None THEN  
                RETURN  
            END IF  
            APPEND the pair consisting of row AND the val attribute of node TO the list at key col in node_dict  
            IF col IS LESS THAN min_col THEN  
                SET min_col TO col  
            END IF  
            IF col IS GREATER THAN max_col THEN  
                SET max_col TO col  
            END IF  
            CALL dfs(the left attribute of node, row PLUS one, col MINUS one)  
            CALL dfs(the right attribute of node, row PLUS one, col PLUS one)  
        END FUNCTION  

        CALL dfs(root, zero, zero)  

        SET result TO an empty list  
        FOR each col FROM min_col TO max_col  
            SET sorted_pairs TO node_dict[col] sorted first by the first element of each pair AND THEN by the second element of each pair in ascending order  
            SET column_values TO an empty list  
            FOR each pair IN sorted_pairs  
                APPEND the second element of pair TO column_values  
            END FOR  
            APPEND column_values TO result  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
