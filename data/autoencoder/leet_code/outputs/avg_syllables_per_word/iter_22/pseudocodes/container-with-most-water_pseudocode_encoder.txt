CLASS Solution  
    FUNCTION maxArea(height)  
        SET left TO zero  
        SET right TO the LENGTH OF height MINUS one  
        SET max_area TO zero  
        
        WHILE left LESS THAN right  
            SET current_area TO the MINIMUM OF the element at position left of height AND the element at position right of height MULTIPLIED BY right MINUS left  
            SET max_area TO the MAXIMUM OF max_area AND current_area  
            
            IF the element at position left of height LESS THAN the element at position right of height  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
