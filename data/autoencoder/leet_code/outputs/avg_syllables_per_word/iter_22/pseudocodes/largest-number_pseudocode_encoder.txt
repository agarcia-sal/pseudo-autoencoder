CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO an empty list  
        FOR each element IN nums  
            APPEND string representation of element TO nums_str  
        END FOR  

        FUNCTION compare(x, y)  
            IF concatenation of x AND y GREATER THAN concatenation of y AND x THEN  
                RETURN negative one  
            ELSE IF concatenation of x AND y LESS THAN concatenation of y AND x THEN  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  

        SORT nums_str USING comparison defined by compare  
        
        IF element at position zero of nums_str EQUALS string zero THEN  
            RETURN string zero  
        END IF  

        SET result TO empty string  
        FOR each element IN nums_str  
            SET result TO result PLUS element  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
