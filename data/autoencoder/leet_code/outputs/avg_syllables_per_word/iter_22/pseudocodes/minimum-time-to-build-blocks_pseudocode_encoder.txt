CLASS Solution  
    FUNCTION minBuildTime(blocks AS list OF integers split AS integer) RETURNS integer  
        CONVERT the list blocks INTO a min heap structure  
        WHILE the LENGTH OF blocks IS GREATER THAN one  
            SET first TO the element removed from blocks THAT HAS the smallest value  
            SET second TO the element removed from blocks THAT HAS the smallest value  
            SET combined_time TO the value of split PLUS the value of second  
            INSERT combined_time INTO blocks AS part of the heap  
        END WHILE  
        RETURN the element at position one of blocks  
    END FUNCTION  
END CLASS
