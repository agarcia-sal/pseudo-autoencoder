CLASS Solution  
    FUNCTION scheduleCourse(courses)  
        SORT courses BY the second element of each course in ascending order  
        SET max_heap TO empty list  
        SET current_end_day TO zero  
        FOR each element IN courses  
            SET duration TO the first element of element  
            SET last_day TO the second element of element  
            IF current_end_day PLUS duration LESS THAN OR EQUAL TO last_day THEN  
                ADD negative duration TO max_heap using heap push operation  
                INCREMENT current_end_day BY duration  
            ELSE  
                IF max_heap IS NOT empty AND the negative of the first element of max_heap GREATER THAN duration THEN  
                    INCREMENT current_end_day BY duration PLUS the heap pop operation from max_heap  
                    ADD negative duration TO max_heap using heap push operation  
                END IF  
            END IF  
        END FOR  
        RETURN the length of max_heap  
    END FUNCTION  
END CLASS
