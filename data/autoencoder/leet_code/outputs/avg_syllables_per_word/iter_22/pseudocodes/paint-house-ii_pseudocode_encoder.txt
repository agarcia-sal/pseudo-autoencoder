CLASS Solution  
    FUNCTION minCostII(costs)  
        IF costs EQUALS empty  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF costs  
        SET k TO the LENGTH OF element at position zero of costs  
        
        IF k EQUALS one  
            IF n EQUALS one  
                RETURN element at position zero zero of costs  
            ELSE  
                RETURN infinity  
            END IF  
        END IF  
        
        FOR i FROM one TO n MINUS one  
            SET min_cost TO infinity  
            SET min_cost_index TO minus one  
            SET second_min_cost TO infinity  
            
            FOR j FROM zero TO k MINUS one  
                IF element at position i MINUS one j of costs LESS THAN min_cost  
                    SET second_min_cost TO min_cost  
                    SET min_cost TO element at position i MINUS one j of costs  
                    SET min_cost_index TO j  
                ELSE IF element at position i MINUS one j of costs LESS THAN second_min_cost  
                    SET second_min_cost TO element at position i MINUS one j of costs  
                END IF  
            END FOR  
            
            FOR j FROM zero TO k MINUS one  
                IF j EQUALS min_cost_index  
                    SET element at position i j of costs TO element at position i j of costs PLUS second_min_cost  
                ELSE  
                    SET element at position i j of costs TO element at position i j of costs PLUS min_cost  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO infinity  
        FOR each value IN element at position n MINUS one of costs  
            IF value LESS THAN result  
                SET result TO value  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
