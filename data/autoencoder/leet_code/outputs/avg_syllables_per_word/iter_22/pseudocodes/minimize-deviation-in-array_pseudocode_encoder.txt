CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO infinite large value  
        FOR each num IN nums  
            IF element at position one of num MODULO element at position one of integer two EQUALS element at position one of integer one  
                SET num TO num MULTIPLIED BY integer two  
            END IF  
            SET min_value TO THE SMALLER OF min_value AND num  
            ADD negative num TO max_heap USING heap insertion  
        END FOR  
        
        SET min_deviation TO infinite large value  
        
        WHILE element at position one of the first element of max_heap MODULO element at position one of integer two EQUALS element at position one of integer zero  
            REMOVE and GET the largest element FROM max_heap USING heap removal  
            SET max_value TO negative of the removed element  
            SET min_deviation TO THE SMALLER OF min_deviation AND max_value MINUS min_value  
            SET max_value TO max_value DIVIDED BY integer two  
            SET min_value TO THE SMALLER OF min_value AND max_value  
            ADD negative max_value TO max_heap USING heap insertion  
        END WHILE  
        
        SET min_deviation TO THE SMALLER OF min_deviation AND negative of the first element of max_heap MINUS min_value  
        
        RETURN min_deviation  
    END FUNCTION  
END CLASS
