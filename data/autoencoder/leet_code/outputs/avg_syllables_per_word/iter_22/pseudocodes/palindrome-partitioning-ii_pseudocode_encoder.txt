CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        
        CREATE a two-dimensional table named is_palindrome with dimensions n by n filled with FALSE values  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i of is_palindrome at position i TO TRUE  
        END FOR  
        
        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET element at position start of is_palindrome at position end TO the condition that element at position start of s EQUALS element at position end of s  
                ELSE  
                    SET element at position start of is_palindrome at position end TO the condition that element at position start of s EQUALS element at position end of s AND element at position start PLUS one of is_palindrome at position end MINUS one IS TRUE  
                END IF  
            END FOR  
        END FOR  
        
        CREATE a list min_cuts with length n filled with zero values  
        FOR i FROM zero TO n MINUS one  
            IF element at position zero of is_palindrome at position i IS TRUE  
                SET element at position i of min_cuts TO zero  
            ELSE  
                SET element at position i of min_cuts TO a very large number representing infinity  
                FOR j FROM one TO i  
                    IF element at position j of is_palindrome at position i IS TRUE AND element at position j MINUS one of min_cuts PLUS one LESS THAN element at position i of min_cuts  
                        SET element at position i of min_cuts TO element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN element at position n MINUS one of min_cuts  
    END FUNCTION  
END CLASS
