CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO list of elements in num  
        SET n TO the LENGTH OF num  
        
        WHILE k GREATER THAN zero holds  
            SET swapped TO false  
            FOR i FROM zero TO the LENGTH OF num MINUS one MINUS one  
                IF element at position i of num GREATER THAN element at position i PLUS one of num holds  
                    SET min_digit TO character nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO the MINIMUM OF i PLUS k PLUS one AND n MINUS one  
                        IF element at position j of num LESS THAN min_digit holds  
                            SET min_digit TO element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    PERFORM rotation of elements from position i TO position min_index of num such that element at position min_index moves to position i and all other elements shift right by one position  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO true  
                    BREAK  
                END IF  
            END FOR  
            
            IF swapped EQUALS false holds  
                BREAK  
            END IF  
        END WHILE  
        
        SET result TO concatenation of all elements in num as a single string  
        RETURN result  
    END FUNCTION  
END CLASS
