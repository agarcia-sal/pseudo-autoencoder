CLASS Solution  
    FUNCTION networkDelayTime(times times, n n, k k)  
        SET graph TO the empty dictionary  
        FOR each integer i FROM one TO n  
            ASSIGN reference TO the empty list for element at position i of graph  
        END FOR  
        FOR each element u, v, w IN times  
            APPEND the pair of v and w TO element at position u of graph  
        END FOR  
        SET dist TO the empty dictionary  
        FOR each integer i FROM one TO n  
            ASSIGN reference TO infinity for element at position i of dist  
        END FOR  
        SET element at position k of dist TO zero  
        SET min_heap TO the list containing the pair of zero and k  
        WHILE the LENGTH OF min_heap is greater than zero  
            REMOVE and ASSIGN first popped pair TO current_dist and node from min_heap using heap POP operation  
            IF current_dist GREATER THAN element at position node of dist  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            FOR each pair neighbor and weight IN element at position node of graph  
                SET distance TO current_dist PLUS weight  
                IF distance LESS THAN element at position neighbor of dist  
                    SET element at position neighbor of dist TO distance  
                    INSERT the pair of distance and neighbor INTO min_heap using heap PUSH operation  
                END IF  
            END FOR  
        END WHILE  
        SET max_time TO the maximum value in the values of dist  
        IF max_time NOT EQUALS infinity  
            RETURN max_time  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
