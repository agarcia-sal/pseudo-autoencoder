CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position one of values  
    SET i TO one  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the left of queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS none  
            SET node.left TO new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS none  
            SET node.right TO new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN logical AND of is_same_tree of p.left and q.left AND is_same_tree of p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node EQUALS none  
                RETURN zero  
            END IF  
            SET current_sum TO node.val PLUS subtree_sum of node.left PLUS subtree_sum of node.right  
            INCREMENT count at current_sum BY one  
            RETURN current_sum  
        END FUNCTION  
        
        SET count TO a new default dictionary with default integer zero  
        CALL subtree_sum with root  
        
        SET max_freq TO the maximum value among all values in count or zero if empty  
        
        SET result TO an empty list  
        FOR each key and freq IN items in count  
            IF freq EQUALS max_freq  
                APPEND key TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
