CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one DOWN TO zero  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET original_position TO the position of element at position index of s IN the alphabet starting at letter a  
            SET shifted_position TO original_position PLUS element at position index of shifts  
            SET wrapped_position TO the remainder of shifted_position DIVIDED BY the number of letters in the alphabet  
            SET new_character TO the letter at position wrapped_position in the alphabet starting at letter a  
            APPEND new_character TO result  
        END FOR  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
