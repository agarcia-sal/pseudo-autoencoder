CLASS Solution  
    FUNCTION maxA(n)  
        IF n LESS THAN OR EQUAL TO three  
            RETURN n  
        END IF  

        SET dp TO a list of zeros with LENGTH equal TO n PLUS one  

        FOR i FROM one TO three  
            SET element at position i of dp TO i  
        END FOR  

        FOR i FROM four TO n  
            FOR j FROM i MINUS three DOWN TO one  
                SET value_at_j TO element at position j of dp  
                SET product TO value_at_j MULTIPLIED BY (i MINUS j MINUS one)  
                IF product GREATER THAN element at position i of dp  
                    SET element at position i of dp TO product  
                END IF  
            END FOR  
        END FOR  

        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
