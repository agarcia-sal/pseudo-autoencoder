CLASS Solution  
    FUNCTION minDays(n)  
        FUNCTION dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            RETURN one PLUS min(the remainder of remaining DIVIDED BY two PLUS dp(the quotient of remaining DIVIDED BY two), the remainder of remaining DIVIDED BY three PLUS dp(the quotient of remaining DIVIDED BY three))  
        END FUNCTION  
        RETURN dp(n)  
    END FUNCTION  
END CLASS
