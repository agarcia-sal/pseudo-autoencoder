CLASS Solution  
    FUNCTION wallsAndGates(rooms)  
        IF rooms EQUALS empty OR the element at position zero of rooms EQUALS empty THEN  
            RETURN  
        END IF  
        
        SET rows TO the LENGTH OF rooms  
        SET cols TO the LENGTH OF the element at position zero of rooms  
        ASSIGN reference TO an empty double-ended queue named queue  
        
        FOR each r FROM zero TO rows MINUS one  
            FOR each c FROM zero TO cols MINUS one  
                IF the element at position r of rooms AT position c EQUALS zero THEN  
                    APPEND the tuple consisting of r and c TO queue  
                END IF  
            END FOR  
        END FOR  
        
        SET directions TO the list containing the tuples representing zero PLUS one zero zero MINUS one and MINUS one zero  
        
        WHILE queue is not empty  
            REMOVE the first element from queue AND ASSIGN reference TO r and c from that element  
            FOR each dr and dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN rows AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN cols AND the element at position nr of rooms AT position nc EQUALS two billion one hundred forty seven million four hundred eighty three thousand six hundred forty seven THEN  
                    SET the element at position nr of rooms AT position nc TO the element at position r of rooms AT position c PLUS one  
                    APPEND the tuple consisting of nr and nc TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
