CLASS Solution  
    FUNCTION addBoldTag WITH PARAMETERS s words  
        IF the LENGTH OF words EQUALS zero  
            RETURN s  
        END IF  
        SET n TO the LENGTH OF s  
        SET mask TO a list of the BOOLEAN value false REPEATED n TIMES  
        FOR each word IN words  
            SET start TO zero  
            WHILE start LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the POSITION of the FIRST OCCURRENCE of word IN s STARTING FROM start  
                IF start EQUALS negative one  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i OF mask TO true  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN n  
            IF the element at position i OF mask IS true  
                APPEND the string beginning tag FOR bold TO result  
                WHILE i LESS THAN n AND the element at position i OF mask IS true  
                    APPEND the element at position i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string ending tag FOR bold TO result  
            ELSE  
                APPEND the element at position i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation OF all elements IN result AS a single string  
    END FUNCTION  
END CLASS
