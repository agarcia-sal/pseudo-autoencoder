CLASS Solution  
    FUNCTION orderlyQueue(s parameter of type string k parameter of type integer) RETURNS string  
        IF k GREATER THAN one THEN  
            RETURN concatenate all characters IN the sequence OF the characters of s sorted in ascending order  
        ELSE  
            SET n TO the length of s  
            SET smallest TO s  
            FOR index i FROM one TO n MINUS one  
                SET rotated TO concatenation of substring from position i TO the end of s AND substring from position zero TO position i MINUS one of s  
                IF rotated LESS THAN smallest THEN  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
