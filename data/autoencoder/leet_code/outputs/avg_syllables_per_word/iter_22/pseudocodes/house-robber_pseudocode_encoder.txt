CLASS Solution  
    FUNCTION rob(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one THEN  
            RETURN the element at position one of nums  
        END IF  
        SET dp TO a list of zeros with the LENGTH OF nums  
        SET the element at position one of dp TO the element at position one of nums  
        SET the element at position two of dp TO the GREATER VALUE BETWEEN the element at position one of nums AND the element at position two of nums  
        FOR index i FROM three TO the LENGTH OF nums  
            SET the element at position i of dp TO the GREATER VALUE BETWEEN the element at position i MINUS one of dp AND the SUM OF the element at position i MINUS two of dp AND the element at position i of nums  
        END FOR  
        RETURN the element at the LAST POSITION of dp  
    END FUNCTION  
END CLASS
