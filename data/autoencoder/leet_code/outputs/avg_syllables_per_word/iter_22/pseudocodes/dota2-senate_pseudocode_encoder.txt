CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        INITIALIZE radiant AS empty double ended queue  
        INITIALIZE dire AS empty double ended queue  
        FOR each position index AND senator_character IN the enumeration of senate  
            IF senator_character EQUALS character R  
                APPEND position index TO radiant  
            ELSE  
                APPEND position index TO dire  
            END IF  
        END FOR  
        WHILE the LENGTH OF radiant IS GREATER THAN zero AND the LENGTH OF dire IS GREATER THAN zero  
            REMOVE AND RETURN the first element FROM radiant AND ASSIGN TO r_index  
            REMOVE AND RETURN the first element FROM dire AND ASSIGN TO d_index  
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO dire  
            END IF  
        END WHILE  
        IF the LENGTH OF radiant IS GREATER THAN zero  
            RETURN the word Radiant  
        ELSE  
            RETURN the word Dire  
        END IF  
    END FUNCTION  
END CLASS
