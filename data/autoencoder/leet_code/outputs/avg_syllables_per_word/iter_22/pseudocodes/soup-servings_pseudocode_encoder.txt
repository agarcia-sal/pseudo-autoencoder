CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one  
        END IF  
        SET n TO n PLUS twenty four DIVIDED BY twenty five rounded down to nearest integer  
        SET memo TO empty collection of key value pairs  
        FUNCTION dp(a b)  
            IF the pair a comma b EXISTS AS a key IN memo  
                RETURN the value FOR the key a comma b IN memo  
            END IF  
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one  
            END IF  
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero  
            END IF  
            SET prob TO point two five MULTIPLIED BY the sum OF  
                dp of a MINUS four b  
                PLUS dp of a MINUS three b MINUS one  
                PLUS dp of a MINUS two b MINUS two  
                PLUS dp of a MINUS one b MINUS three  
            SET memo at key a comma b TO prob  
            RETURN prob  
        END FUNCTION  
        RETURN dp of n n  
    END FUNCTION  
END CLASS
