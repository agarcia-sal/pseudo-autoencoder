CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET xor_all TO zero  
        FOR each num IN nums  
            SET xor_all TO the result of xor_all XOR num  
        END FOR  
        SET diff_bit TO the bit position corresponding to the rightmost bit that is set in xor_all  
        SET num1 TO zero  
        SET num2 TO zero  
        FOR each num IN nums  
            IF the bit at position diff_bit of num EQUALS one  
                SET num1 TO the result of num1 XOR num  
            ELSE  
                SET num2 TO the result of num2 XOR num  
            END IF  
        END FOR  
        RETURN list containing num1 and num2  
    END FUNCTION  
END CLASS
