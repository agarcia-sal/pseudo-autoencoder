CLASS Solution  
    FUNCTION minEatingSpeed(piles, h)  
        FUNCTION canFinish(k)  
            SET hours TO zero  
            FOR each pile IN piles  
                INCREMENT hours BY the smallest integer greater than or equal to the result of pile DIVIDED BY k  
            END FOR  
            RETURN whether hours IS LESS THAN OR EQUAL TO h  
        END FUNCTION  
        
        SET left TO one  
        SET right TO the largest element IN piles  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO the integer result of left PLUS right DIVIDED BY two ignoring any remainder  
            IF canFinish(mid) IS TRUE  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
