CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with value the element at position zero of values  
    SET i TO one  
    INITIALIZE queue as an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater than zero  
        REMOVE the first element from queue and ASSIGN it TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not no value THEN  
            SET node.left TO a new TreeNode with value the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not no value THEN  
            SET node.right TO a new TreeNode with value the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS no value AND q EQUALS no value THEN  
        RETURN True  
    ELSE IF p EQUALS no value OR q EQUALS no value THEN  
        RETURN False  
    ELSE IF the value of p DOES NOT EQUAL the value of q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with arguments p.left and q.left AND the result of is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION printTree(self, root)  
        FUNCTION getHeight(node)  
            IF node EQUALS no value THEN  
                RETURN zero  
            END IF  
            RETURN one PLUS the greater of the values returned by getHeight called with node.left AND getHeight called with node.right  
        END FUNCTION  

        SET height TO getHeight called with root  
        SET width TO two RAISED TO THE POWER OF height MINUS one  
        SET res TO a list of height elements where each element is a list of width elements each SET TO empty string  

        FUNCTION placeNode(node, row, left, right)  
            IF node EQUALS no value THEN  
                RETURN  
            END IF  
            SET mid TO the integer division of the sum of left AND right BY two  
            SET the element at position mid of the list at position row of res TO the string form of the value of node  
            CALL placeNode with arguments node.left, row PLUS one, left, mid  
            CALL placeNode with arguments node.right, row PLUS one, mid, right  
        END FUNCTION  

        CALL placeNode with arguments root, zero, zero, width  
        RETURN res  
    END FUNCTION  
END CLASS
