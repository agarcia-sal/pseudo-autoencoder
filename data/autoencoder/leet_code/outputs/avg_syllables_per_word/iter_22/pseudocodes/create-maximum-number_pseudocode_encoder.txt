CLASS Solution  
    FUNCTION maxNumber(nums1 nums2 k)  
        FUNCTION maxSingleArray(nums k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop is greater than zero AND the LENGTH OF stack is greater than zero AND the element at the last position of stack is LESS THAN num  
                    REMOVE the element at the last position FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the first k elements of stack  
        END FUNCTION  
        
        FUNCTION merge(arr1 arr2)  
            SET result TO an empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR index FROM one TO total_length  
                SET max_array TO the array which is lexicographically greater between arr1 and arr2  
                REMOVE the element at the first position FROM max_array AND APPEND it TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        SET max_number TO an empty list  
        FOR i FROM zero TO k  
            IF i is LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i is LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO merge(the result of maxSingleArray with nums1 and i, the result of maxSingleArray with nums2 and k MINUS i)  
                SET max_number TO the array which is lexicographically greater between max_number and candidate  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
