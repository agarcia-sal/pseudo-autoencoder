CLASS Solution  
    FUNCTION largestRectangleArea(heights)  
        SET stack TO a list containing negative one  
        SET max_area TO zero  
        APPEND zero TO the end of heights  
        FOR each index i and height h AS the enumeration of heights  
            WHILE the last element of stack NOT EQUALS negative one AND height h LESS THAN the element at the position of the last element of stack in heights  
                SET height TO the element at the position obtained by removing the last element from stack in heights  
                SET width TO index i MINUS the last element of stack MINUS one  
                SET max_area TO the larger value BETWEEN max_area AND height MULTIPLIED BY width  
            END WHILE  
            APPEND index i TO stack  
        END FOR  
        RETURN max_area  
    END FUNCTION  
END CLASS
