CLASS Solution  
    FUNCTION getMinDistSum(positions)  
        SET n TO the LENGTH OF positions  
        SET x TO zero  
        SET y TO zero  
        FOR each position_x AND position_y IN positions  
            INCREMENT x BY position_x  
            INCREMENT y BY position_y  
        END FOR  
        SET x TO x DIVIDED BY n  
        SET y TO y DIVIDED BY n  
        SET decay TO zero point nine nine nine  
        SET epsilon TO one times ten TO THE POWER OF MINUS six  
        SET alpha TO zero point five  
        WHILE true holds  
            SET grad_x TO zero  
            SET grad_y TO zero  
            SET dist TO zero  
            FOR each position_x AND position_y IN positions  
                SET a TO x MINUS position_x  
                SET b TO y MINUS position_y  
                SET c TO the square root of a MULTIPLIED BY a PLUS b MULTIPLIED BY b  
                INCREMENT grad_x BY a DIVIDED BY c PLUS one times ten TO THE POWER OF MINUS eight  
                INCREMENT grad_y BY b DIVIDED BY c PLUS one times ten TO THE POWER OF MINUS eight  
                INCREMENT dist BY c  
            END FOR  
            SET dx TO grad_x MULTIPLIED BY alpha  
            SET dy TO grad_y MULTIPLIED BY alpha  
            DECREMENT x BY dx  
            DECREMENT y BY dy  
            MULTIPLY alpha BY decay  
            IF the absolute value of dx LESS THAN OR EQUAL TO epsilon AND the absolute value of dy LESS THAN OR EQUAL TO epsilon  
                RETURN dist  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
