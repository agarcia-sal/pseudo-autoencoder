CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO an empty list  
        SET max_value TO negative infinity  
        
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            PUSH the tuple of element at position zero of element at position index of nums AND index AND zero TO min_heap USING a heap push operation  
            SET max_value TO the GREATER THAN of max_value AND element at position zero of element at position index of nums  
        END FOR  
        
        SET smallest_range TO infinity  
        SET result TO the list consisting of a large negative number AND a large positive number  
        
        WHILE min_heap is not empty  
            REMOVE the smallest element FROM min_heap USING a heap pop operation AND ASSIGN the first element TO min_value AND second element TO list_index AND third element TO element_index  
            
            IF max_value MINUS min_value IS LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO the list consisting of min_value AND max_value  
            END IF  
            
            IF element_index PLUS one IS LESS THAN the LENGTH OF element at position list_index of nums  
                SET next_value TO element at position element_index PLUS one of element at position list_index of nums  
                PUSH the tuple of next_value AND list_index AND element_index PLUS one TO min_heap USING a heap push operation  
                SET max_value TO the GREATER THAN of max_value AND next_value  
            ELSE  
                BREAK the loop  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
