CLASS Solution  
    FUNCTION subarraySum(nums k)  
        SET cumulative_sum_freq TO dictionary with key zero and value one  
        SET current_sum TO zero  
        SET count TO zero  
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF the key of the value current_sum MINUS k EXISTS IN cumulative_sum_freq  
                INCREMENT count BY the value for the key current_sum MINUS k IN cumulative_sum_freq  
            END IF  
            IF the key current_sum EXISTS IN cumulative_sum_freq  
                INCREMENT the value for the key current_sum IN cumulative_sum_freq BY one  
            ELSE  
                SET the value for the key current_sum IN cumulative_sum_freq TO one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
