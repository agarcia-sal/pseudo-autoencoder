CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            SET good TO a list of FALSE values of LENGTH n  
            SET bad TO a list of FALSE values of LENGTH n  
            FOR j FROM zero TO n MINUS one  
                IF the bitwise AND of i AND two RAISED TO THE POWER OF j EQUALS zero  
                    SET bad element at position j TO TRUE  
                ELSE  
                    SET good element at position j TO TRUE  
                END IF  
            END FOR  
            SET valid TO TRUE  
            FOR j FROM zero TO n MINUS one  
                IF good element at position j EQUALS TRUE  
                    FOR k FROM zero TO n MINUS one  
                        IF statement at position j at position k EQUALS zero AND good element at position k EQUALS TRUE OR statement at position j at position k EQUALS one AND bad element at position k EQUALS TRUE  
                            SET valid TO FALSE  
                            BREAK the inner loop  
                        END IF  
                    END FOR  
                    IF valid EQUALS FALSE  
                        BREAK the outer loop  
                    END IF  
                END IF  
            END FOR  
            IF valid EQUALS TRUE  
                SET count_good TO zero  
                FOR each element IN good  
                    IF element EQUALS TRUE  
                        INCREMENT count_good BY one  
                    END IF  
                END FOR  
                IF count_good GREATER THAN max_good  
                    SET max_good TO count_good  
                END IF  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
