CLASS Solution  
    FUNCTION findRotateSteps(ring key)  
        SET char_positions TO a new dictionary with default value as empty list  
        SET i TO zero  
        FOR each character char IN ring  
            APPEND i TO the list at element character char of char_positions  
            INCREMENT i BY one  
        END FOR  
        SET n TO the LENGTH OF ring  
        
        FUNCTION dp(i prev_pos)  
            IF i EQUALS the LENGTH OF key THEN  
                RETURN zero  
            END IF  
            
            SET min_steps TO positive infinity  
            FOR each position pos IN the list at element character at position i of key in char_positions  
                SET clockwise TO the remainder of (pos MINUS prev_pos) DIVIDED BY n  
                SET anticlockwise TO the remainder of (prev_pos MINUS pos) DIVIDED BY n  
                IF clockwise LESS THAN anticlockwise THEN  
                    SET distance TO clockwise  
                ELSE  
                    SET distance TO anticlockwise  
                END IF  
                SET steps TO distance PLUS one PLUS the result of dp with parameters (i PLUS one) and pos  
                IF steps LESS THAN min_steps THEN  
                    SET min_steps TO steps  
                END IF  
            END FOR  
            RETURN min_steps  
        END FUNCTION  
        
        RETURN dp with parameters zero and zero  
    END FUNCTION  
END CLASS
