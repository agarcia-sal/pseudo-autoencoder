CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO none right SET TO none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS nothing THEN  
        RETURN nothing  
    END IF  
    SET root TO NEW TreeNode WITH value as element at position zero of values  
    SET i TO one  
    SET queue TO NEW empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty DO  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT nothing THEN  
            SET node dot left TO NEW TreeNode WITH value as element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT nothing THEN  
            SET node dot right TO NEW TreeNode WITH value as element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS nothing AND q EQUALS nothing THEN  
        RETURN True  
    ELSE IF p EQUALS nothing OR q EQUALS nothing THEN  
        RETURN False  
    ELSE IF p dot val NOT EQUALS q dot val THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH p dot left AND q dot left AND the result of is_same_tree WITH p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom(self root)  
        IF root EQUALS nothing THEN  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a new double ended queue INITIALIZED WITH root  
        WHILE queue IS NOT empty DO  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR underscore FROM zero TO level_size MINUS one DO  
                SET node TO REMOVE first element FROM queue  
                APPEND node dot val TO current_level  
                IF node dot left IS NOT nothing THEN  
                    APPEND node dot left TO queue  
                END IF  
                IF node dot right IS NOT nothing THEN  
                    APPEND node dot right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result IN REVERSED ORDER  
    END FUNCTION  
END CLASS
