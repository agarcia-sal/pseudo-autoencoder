CLASS Solution  
    FUNCTION minInsertions(s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO the LENGTH OF s  
            CREATE dp AS a two-dimensional list of size n BY n FILLED WITH zero  
            FOR i FROM zero TO n MINUS one  
                SET element at position i OF the list at position i OF dp TO one  
            END FOR  
            FOR cl FROM two TO n  
                FOR i FROM zero TO n MINUS cl  
                    SET j TO i PLUS cl MINUS one  
                    IF element at position i of s EQUALS element at position j of s AND cl EQUALS two  
                        SET element at position i of the list at position j of dp TO two  
                    ELSE IF element at position i of s EQUALS element at position j of s  
                        SET element at position i of the list at position j of dp TO element at position i PLUS one of the list at position j MINUS one of dp PLUS two  
                    ELSE  
                        SET element at position i of the list at position j of dp TO the GREATER OF element at position i of the list at position j MINUS one of dp AND element at position i PLUS one of the list at position j of dp  
                    END IF  
                END FOR  
            END FOR  
            RETURN element at position zero of the list at position n MINUS one of dp  
        END FUNCTION  
        RETURN the LENGTH OF s MINUS longest_palindromic_subsequence(s)  
    END FUNCTION  
END CLASS
