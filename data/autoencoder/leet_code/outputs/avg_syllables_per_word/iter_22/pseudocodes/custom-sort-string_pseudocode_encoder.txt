CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO an empty dictionary  
        FOR each character c AND index i IN order WITH their respective positions starting from the first position  
            SET the value at key c in dictionary d TO i  
        END FOR  
        SET sorted_characters TO the result of sorting s by assigning to each character x the value obtained by LOOKUP of x in dictionary d with default zero if x is not found  
        SET result_string TO the concatenation of all characters in sorted_characters in sequence  
        RETURN result_string  
    END FUNCTION  
END CLASS
