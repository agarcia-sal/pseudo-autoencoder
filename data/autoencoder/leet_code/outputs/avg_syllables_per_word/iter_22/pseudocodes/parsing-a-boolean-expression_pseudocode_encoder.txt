CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO empty list  
        FOR each character c IN expression  
            IF character c EQUALS one of the elements t or f or logical NOT or logical AND or logical OR  
                APPEND character c TO stk  
            ELSE IF character c EQUALS closing parenthesis  
                SET count_true TO zero  
                SET count_false TO zero  
                WHILE element at the last position of stk EQUALS one of the elements t or f  
                    INCREMENT count_true BY one IF element at the last position of stk EQUALS t  
                    INCREMENT count_false BY one IF element at the last position of stk EQUALS f  
                    REMOVE element at the last position FROM stk  
                END WHILE  
                SET operator TO element at the last position of stk  
                REMOVE element at the last position FROM stk  
                IF operator EQUALS logical NOT  
                    SET c TO t IF count_false GREATER THAN zero EQUALS false ELSE f  
                ELSE IF operator EQUALS logical AND  
                    SET c TO f IF count_false GREATER THAN zero ELSE t  
                ELSE IF operator EQUALS logical OR  
                    SET c TO t IF count_true GREATER THAN zero ELSE f  
                END IF  
                APPEND c TO stk  
            END IF  
        END FOR  
        RETURN element at the first position of stk EQUALS t  
    END FUNCTION  
END CLASS
