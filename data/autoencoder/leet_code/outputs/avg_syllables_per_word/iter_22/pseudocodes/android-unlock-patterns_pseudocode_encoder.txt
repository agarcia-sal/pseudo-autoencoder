CLASS Solution
    FUNCTION numberOfPatterns(m parameter n parameter)
        SET jumps TO a mapping from pairs of dots to a dot that must be visited when jumping over
            the pair from one to three maps to two
            the pair from one to seven maps to four
            the pair from one to nine maps to five
            the pair from two to eight maps to five
            the pair from three to one maps to two
            the pair from three to seven maps to five
            the pair from three to nine maps to six
            the pair from four to six maps to five
            the pair from five to five maps to five
            the pair from six to four maps to five
            the pair from seven to one maps to four
            the pair from seven to three maps to five
            the pair from seven to nine maps to eight
            the pair from eight to two maps to five
            the pair from nine to one maps to five
            the pair from nine to three maps to six
            the pair from nine to seven maps to eight

        FUNCTION countPatternsFromDot(current parameter visited parameter length parameter)
            IF length EQUALS n THEN
                RETURN one
            END IF
            SET count TO one IF length GREATER THAN OR EQUAL TO m OTHERWISE zero
            FOR next_dot FROM one TO nine
                IF element at position next_dot of visited EQUALS false THEN
                    SET jump TO the value in jumps for the pair current and next_dot
                    IF jump IS NOT DEFINED OR element at position jump of visited EQUALS true THEN
                        SET element at position next_dot of visited TO true
                        INCREMENT count BY the result of countPatternsFromDot with parameters next_dot visited length PLUS one
                        SET element at position next_dot of visited TO false
                    END IF
                END IF
            END FOR
            RETURN count
        END FUNCTION

        SET visited TO a list of ten elements all set to false
        SET result TO zero
        FOR i FROM one TO nine
            SET element at position i of visited TO true
            INCREMENT result BY the result of countPatternsFromDot with parameters i visited one
            SET element at position i of visited TO false
        END FOR

        RETURN result
    END FUNCTION
END CLASS
