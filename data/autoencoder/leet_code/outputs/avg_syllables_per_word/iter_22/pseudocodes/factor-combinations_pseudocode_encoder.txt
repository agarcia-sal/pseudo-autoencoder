CLASS Solution  
    FUNCTION getFactors(n)  
        FUNCTION backtrack(start, target)  
            FOR i FROM start TO the INTEGER PART OF the SQUARE ROOT OF target PLUS one  
                IF the MODULO of target BY i EQUALS zero  
                    APPEND the concatenation of path PLUS the list containing i AND target DIVIDED BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH i AND target DIVIDED BY i  
                    REMOVE the last element FROM path  
                END IF  
            END FOR  
        END FUNCTION  

        SET result TO an empty list  
        SET path TO an empty list  
        CALL backtrack WITH two AND n  
        RETURN result  
    END FUNCTION  
END CLASS
