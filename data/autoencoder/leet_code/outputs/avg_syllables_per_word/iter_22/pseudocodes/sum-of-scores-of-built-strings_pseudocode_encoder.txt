CLASS Solution  
    FUNCTION sumScores WITH PARAMETER s  
        SET n TO the LENGTH OF s  
        SET lcp TO a LIST OF zeroes OF LENGTH n  

        SET l TO zero  
        SET r TO zero  
        FOR i FROM one TO n MINUS one  
            IF i LESS THAN OR EQUAL TO r  
                SET lcp at position i TO the MINIMUM OF r MINUS i PLUS one AND lcp at position i MINUS l  
            END IF  

            WHILE i PLUS lcp at position i LESS THAN n AND element at position lcp at position i of s EQUALS element at position i PLUS lcp at position i of s  
                INCREMENT lcp at position i BY one  
            END WHILE  

            IF i PLUS lcp at position i MINUS one GREATER THAN r  
                SET l TO i  
                SET r TO i PLUS lcp at position i MINUS one  
            END IF  
        END FOR  

        SET total_score TO the SUM OF all elements IN lcp PLUS n  
        RETURN total_score  
    END FUNCTION  
END CLASS
