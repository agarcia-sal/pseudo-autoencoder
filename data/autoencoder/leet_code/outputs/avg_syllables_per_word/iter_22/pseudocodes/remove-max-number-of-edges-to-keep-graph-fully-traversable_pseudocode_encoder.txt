CLASS UnionFind  
    FUNCTION __init__ (self n)  
        SET self.parent TO a list of integers from zero TO n  
        SET self.rank TO a list of zeros with length n PLUS one  
        SET self.count TO n  
    END FUNCTION  
  
    FUNCTION find (self u)  
        IF element at position u of self.parent NOT EQUALS u  
            SET element at position u of self.parent TO the result of calling find on self.parent at position u  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  
  
    FUNCTION union (self u v)  
        SET rootU TO the result of calling find on u  
        SET rootV TO the result of calling find on v  
        IF rootU NOT EQUALS rootV  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            DECREMENT self.count BY one  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION maxNumEdgesToRemove (self n edges)  
        ASSIGN reference uf_common TO a new UnionFind with parameter n  
        ASSIGN reference uf_alice TO a new UnionFind with parameter n  
        ASSIGN reference uf_bob TO a new UnionFind with parameter n  
  
        SET used_edges TO zero  
  
        FOR each element from the start to the end in edges  
            DESTRUCTURE element into t u v  
            IF t EQUALS three  
                IF call union method on uf_common with parameters u and v RETURNS True  
                    CALL union method on uf_alice with parameters u and v  
                    CALL union method on uf_bob with parameters u and v  
                    INCREMENT used_edges BY one  
                END IF  
            END IF  
        END FOR  
  
        FOR each element from the start to the end in edges  
            DESTRUCTURE element into t u v  
            IF t EQUALS one AND call union method on uf_alice with parameters u and v RETURNS True  
                INCREMENT used_edges BY one  
            ELSE IF t EQUALS two AND call union method on uf_bob with parameters u and v RETURNS True  
                INCREMENT used_edges BY one  
            END IF  
        END FOR  
  
        IF uf_alice.count EQUALS one AND uf_bob.count EQUALS one  
            RETURN the LENGTH OF edges MINUS used_edges  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
