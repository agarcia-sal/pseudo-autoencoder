CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET the value attribute OF self TO val  
        SET the left attribute OF self TO left  
        SET the right attribute OF self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the values list IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        REMOVE the element from the front of queue AND ASSIGN TO node  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT None  
            SET the left attribute OF node TO a new TreeNode with the element at position i of values  
            APPEND the left attribute OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the length of values AND the element at position i of values IS NOT None  
            SET the right attribute OF node TO a new TreeNode with the element at position i of values  
            APPEND the right attribute OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value attribute OF p IS NOT EQUALS TO the value attribute OF q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left attribute OF p and the left attribute OF q AND the result of is_same_tree with the right attribute OF p and the right attribute OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST(self, root, val)  
        FUNCTION dfs(root)  
            IF root IS None  
                RETURN a new TreeNode with val  
            END IF  
            IF the value attribute OF root IS LESS THAN val  
                SET the right attribute OF root TO the result of dfs with the right attribute OF root  
            ELSE  
                SET the left attribute OF root TO the result of dfs with the left attribute OF root  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN the result of dfs with root  
    END FUNCTION  
END CLASS
