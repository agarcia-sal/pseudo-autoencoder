CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string  
            RETURN empty list  
        END IF  
        
        SET phone_map TO mapping from character two TO string consisting of letters a b and c AND character three TO string consisting of letters d e and f AND character four TO string consisting of letters g h and i AND character five TO string consisting of letters j k and l AND character six TO string consisting of letters m n and o AND character seven TO string consisting of letters p q r and s AND character eight TO string consisting of letters t u and v AND character nine TO string consisting of letters w x y and z  
        
        FUNCTION backtrack(combination, next_digits)  
            IF the length of next_digits EQUALS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the string mapped by the first character of next_digits in phone_map  
                    CALL backtrack with first argument as combination concatenated with letter AND second argument as substring of next_digits from second position TO the end  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO empty list  
        CALL backtrack with empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
