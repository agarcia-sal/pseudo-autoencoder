CLASS Solution  
    FUNCTION secondMinimum(n time edges change)  
        SET graph TO a list containing empty lists repeated the value of n PLUS one times  
        FOR each pair u v IN edges  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
        
        ASSIGN reference TO a double ended queue containing one tuple consisting of integer one and integer zero and integer zero  
        SET minTimes TO a list containing sublists each sublist containing two elements of infinity repeated the value of n PLUS one times  
        SET element at position one first element of minTimes TO zero  
        
        WHILE the queue is not empty holds  
            REMOVE leftmost element from queue and ASSIGN TO current currentTime count  
            
            IF current EQUALS n AND count EQUALS one THEN  
                RETURN currentTime  
            END IF  
            
            IF integer division of currentTime BY change MODULO two EQUALS zero THEN  
                SET nextTime TO currentTime PLUS time  
            ELSE  
                SET nextTime TO the sum of the product of the integer division of currentTime BY change PLUS one and change PLUS time  
            END IF  
            
            FOR each neighbor IN element at position current of graph  
                IF element at position neighbor first element of minTimes GREATER THAN nextTime THEN  
                    SET element at position neighbor second element of minTimes TO element at position neighbor first element of minTimes  
                    SET element at position neighbor first element of minTimes TO nextTime  
                    APPEND a tuple consisting of neighbor nextTime zero TO queue  
                ELSE IF element at position neighbor first element of minTimes LESS THAN nextTime AND nextTime LESS THAN element at position neighbor second element of minTimes THEN  
                    SET element at position neighbor second element of minTimes TO nextTime  
                    APPEND a tuple consisting of neighbor nextTime one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
