CLASS Solution  
    FUNCTION maximizeXor(nums WITH TYPE List OF integers, queries WITH TYPE List OF List OF integers) WITH RETURN TYPE List OF integers  
        SORT nums IN ASCENDING ORDER  
        SET sorted_queries TO the LIST OF pairs of index AND query FROM queries sorted by the second element of the query  
        SET result TO a LIST of the LENGTH OF queries with all elements SET TO negative one  
        SET trie TO an EMPTY mapping  
        SET num_index TO zero  
        
        FOR each query_index AND pair of xi AND mi IN sorted_queries  
            WHILE num_index LESS THAN the LENGTH OF nums AND the element at position num_index of nums LESS THAN OR EQUAL TO mi  
                SET num TO the element at position num_index of nums  
                SET node TO trie  
                FOR i FROM thirty ONE DOWN TO zero  
                    SET bit TO the value of the bit at position i of num  
                    IF bit NOT IN node  
                        SET element at key bit of node TO an EMPTY mapping  
                    END IF  
                    SET node TO the element at key bit of node  
                END FOR  
                INCREMENT num_index BY one  
            END WHILE  
            
            IF trie IS EMPTY  
                CONTINUE TO next iteration of the FOR loop  
            END IF  
            
            SET node TO trie  
            SET max_xor TO zero  
            FOR i FROM thirty ONE DOWN TO zero  
                SET bit TO the value of the bit at position i of xi  
                SET toggle_bit TO one MINUS bit  
                IF toggle_bit IN node  
                    ADD value of one shifted left by i TO max_xor  
                    SET node TO the element at key toggle_bit of node  
                ELSE  
                    SET node TO the element at key bit of node  
                END IF  
            END FOR  
            
            SET element at position query_index of result TO max_xor  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
