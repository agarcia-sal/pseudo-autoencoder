CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF the LENGTH OF houses EQUALS zero  
                RETURN zero  
            END IF  
            IF the LENGTH OF houses EQUALS one  
                RETURN the element at position one of houses  
            END IF  
            SET dp TO a list of zeroes with the LENGTH OF houses  
            SET the element at position one of dp TO the element at position one of houses  
            SET the element at position two of dp TO the GREATER VALUE BETWEEN the element at position one of houses AND the element at position two of houses  
            FOR i FROM three TO the LENGTH OF houses  
                SET the element at position i of dp TO the GREATER VALUE BETWEEN the element at position i MINUS one of dp AND the SUM OF the element at position i MINUS two of dp AND the element at position i of houses  
            END FOR  
            RETURN the element at the last position of dp  
        END FUNCTION  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position one of nums  
        END IF  
        RETURN the GREATER VALUE BETWEEN rob_linear of the substring of nums from position one TO the POSITION BEFORE the last position AND rob_linear of the substring of nums from position two TO the last position  
    END FUNCTION  
END CLASS
