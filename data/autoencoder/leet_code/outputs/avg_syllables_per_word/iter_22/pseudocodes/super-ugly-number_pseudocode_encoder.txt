CLASS Solution  
    FUNCTION nthSuperUglyNumber(n, primes)  
        SET ugly_numbers TO a list containing the single element one  
        SET heap TO a list of pairs where each pair contains an element from primes and its corresponding position in primes  
        CALL heapify ON heap  
        SET indices TO a list of zeros with the same length as primes  
        WHILE the LENGTH OF ugly_numbers LESS THAN n  
            CALL heappop ON heap AND ASSIGN results TO next_ugly AND prime_index  
            IF next_ugly NOT EQUALS the last element of ugly_numbers  
                APPEND next_ugly TO ugly_numbers  
            END IF  
            INCREMENT the element at position prime_index of indices BY one  
            SET next_multiple TO the element at position prime_index of primes MULTIPLIED BY the element at position element at position prime_index of indices of ugly_numbers  
            CALL heappush ON heap WITH the pair next_multiple and prime_index  
        END WHILE  
        RETURN the last element of ugly_numbers  
    END FUNCTION  
END CLASS
