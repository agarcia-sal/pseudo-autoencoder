CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary that maps zero TO negative one  
        SET max_length TO zero  
        SET count TO zero  
        
        FOR each i index AND each num element IN enumeration of nums  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            
            IF count IS A KEY IN prefix_sum_index  
                SET max_length TO the greater value BETWEEN max_length AND i MINUS the element at count of prefix_sum_index  
            ELSE  
                SET the element at count of prefix_sum_index TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
