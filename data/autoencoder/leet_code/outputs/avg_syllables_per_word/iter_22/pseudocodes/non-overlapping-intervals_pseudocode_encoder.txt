CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals EQUALS nothing OR the LENGTH OF intervals EQUALS zero THEN  
            RETURN zero  
        END IF  
        SORT intervals BY the element at position two of each interval  
        SET end TO the element at position two of the first interval in intervals  
        SET count TO zero  
        FOR each index i FROM one TO the LENGTH OF intervals MINUS one  
            IF the element at position one of the interval at position i OF intervals IS GREATER THAN OR EQUAL TO end THEN  
                SET end TO the element at position two of the interval at position i OF intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
