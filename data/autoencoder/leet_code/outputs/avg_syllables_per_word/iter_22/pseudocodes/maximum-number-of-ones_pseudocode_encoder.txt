CLASS Solution  
    FUNCTION maximumNumberOfOnes(width height sideLength maxOnes)  
        SET freq TO a list of zeros with the size equal to the product of sideLength PLUS sideLength  
        FOR i FROM zero TO width MINUS one  
            FOR j FROM zero TO height MINUS one  
                SET rowPosition TO the remainder of i DIVIDED BY sideLength  
                SET columnPosition TO the remainder of j DIVIDED BY sideLength  
                SET pos TO rowPosition MULTIPLIED BY sideLength PLUS columnPosition  
                INCREMENT the element at position pos of freq BY one  
            END FOR  
        END FOR  
        SORT freq in descending order  
        SET total ONES TO zero  
        FOR index FROM zero TO maxOnes MINUS one  
            INCREMENT total ONES BY the element at position index of freq  
        END FOR  
        RETURN total ONES  
    END FUNCTION  
END CLASS
