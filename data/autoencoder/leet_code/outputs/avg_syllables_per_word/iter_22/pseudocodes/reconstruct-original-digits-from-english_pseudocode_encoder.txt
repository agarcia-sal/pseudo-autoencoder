CLASS Solution  
    FUNCTION originalDigits(s)  
        SET count TO the frequency count of each character in the input string s  
        SET digits TO a list of ten zeros representing counts for digits zero through nine  
        
        SET count at character representing the letter z TO digits at zero position  
        SET count at character representing the letter w TO digits at two position  
        SET count at character representing the letter u TO digits at four position  
        SET count at character representing the letter x TO digits at six position  
        SET count at character representing the letter g TO digits at eight position  
        
        SET count at character representing the letter h MINUS digits at eight position TO digits at three position  
        SET count at character representing the letter f MINUS digits at four position TO digits at five position  
        SET count at character representing the letter s MINUS digits at six position TO digits at seven position  
        SET count at character representing the letter i MINUS digits at five position MINUS digits at six position MINUS digits at eight position TO digits at nine position  
        SET count at character representing the letter o MINUS digits at zero position MINUS digits at two position MINUS digits at four position TO digits at one position  
        
        SET result TO an empty list  
        FOR index FROM zero TO nine  
            APPEND the string representing index repeated digits at index position times TO result  
        END FOR  
        
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
