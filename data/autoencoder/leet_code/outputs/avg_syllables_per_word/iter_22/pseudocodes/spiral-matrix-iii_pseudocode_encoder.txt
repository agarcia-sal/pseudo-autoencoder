CLASS Solution  
    FUNCTION spiralMatrixIII(rows integer, cols integer, rStart integer, cStart integer) RETURNS list of list of integer  
        SET directions TO list of pairs representing zero and one zero and one zero and negative one negative one and zero  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO empty list  
        SET x TO rStart  
        SET y TO cStart  
      
        WHILE the LENGTH OF visited LESS THAN rows MULTIPLIED BY cols  
            IF increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR each element in the RANGE from zero TO steps MINUS one  
                IF x GREATER THAN OR EQUAL TO zero AND x LESS THAN rows AND y GREATER THAN OR EQUAL TO zero AND y LESS THAN cols  
                    APPEND list containing x and y TO visited  
                END IF  
                SET dx TO element at position direction_index in directions at position zero  
                SET dy TO element at position direction_index in directions at position one  
                INCREMENT x BY dx  
                INCREMENT y BY dy  
            END FOR  
            SET direction_index TO the remainder of direction_index PLUS one DIVIDED BY four  
            INCREMENT increment BY one  
        END WHILE  
      
        RETURN visited  
    END FUNCTION  
END CLASS
