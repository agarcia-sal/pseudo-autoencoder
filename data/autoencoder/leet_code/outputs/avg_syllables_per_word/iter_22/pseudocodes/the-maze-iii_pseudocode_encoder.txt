    FUNCTION findShortestWay maze ball hole  
        SET directions TO list of tuples of zero AND one AND character r AND one AND zero AND character d AND zero AND negative one AND character l AND negative one AND zero AND character u  
        SET m TO the length of maze  
        SET n TO the length of the element at position zero of maze  
        ASSIGN reference TO a tuple of elements of ball  
        ASSIGN reference TO a tuple of elements of hole  
        SET pq TO list containing tuple of zero AND empty string AND start  
        SET visited TO empty set  
        
        WHILE the LENGTH OF pq is greater than zero  
            EXTRACT the element with smallest distance FROM pq ASSIGNING TO dist AND path AND position unpacked into x AND y  
            
            IF x EQUALS the first element of hole AND y EQUALS the second element of hole  
                RETURN path  
            END IF  
            
            IF tuple of x AND y IS IN visited  
                CONTINUE to next iteration  
            END IF  
            
            ADD tuple of x AND y TO visited  
            
            FOR each tuple dx AND dy AND direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the element at position new_x PLUS dx AND new_y PLUS dy of maze EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                    IF new_x EQUALS the first element of hole AND new_y EQUALS the second element of hole  
                        BREAK from the while loop  
                    END IF  
                END WHILE  
                
                IF tuple of new_x AND new_y IS NOT IN visited  
                    PUSH tuple of dist PLUS count AND concatenation of path AND direction AND tuple of new_x AND new_y INTO pq maintaining heap order  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
