CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO empty mapping with default empty list  
        FOR each pair u and v IN edges  
            APPEND v TO list at key u in graph  
            APPEND u TO list at key v in graph  
        END FOR  
  
        FUNCTION dfs(node, parent, path, visited)  
            IF node IS IN visited THEN  
                SET cycle_start TO the position of node in path  
                RETURN the sublist of path from position cycle_start TO the end  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN list at key node in graph  
                IF neighbor IS NOT EQUAL TO parent THEN  
                    SET cycle TO dfs(neighbor, node, path, visited)  
                    IF cycle IS NOT NONE THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE last element FROM path  
            RETURN NONE  
        END FUNCTION  
  
        SET visited TO empty set  
        SET cycle TO NONE  
        FOR each node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                SET cycle TO dfs(node, negative one, empty list, visited)  
                IF cycle IS NOT NONE THEN  
                    BREAK  
                END IF  
            END IF  
        END FOR  
  
        SET distance TO list of zeros WITH length n  
        SET queue TO double ended queue initialized with elements of cycle  
        SET visited TO set initialized with elements of cycle  
  
        SET level TO zero  
        WHILE queue is not empty  
            SET iteration_count TO the current length of queue  
            FOR each iteration_index FROM one TO iteration_count  
                REMOVE the element from the front of queue and assign it TO node  
                SET element at position node of distance TO level  
                FOR each neighbor IN list at key node in graph  
                    IF neighbor IS NOT IN visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  
  
        RETURN distance  
    END FUNCTION  
END CLASS
