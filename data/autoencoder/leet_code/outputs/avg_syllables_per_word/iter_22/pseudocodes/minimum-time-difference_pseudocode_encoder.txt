CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours TO integer value of substring from position zero TO position before the character representing the separator in time  
            SET minutes TO integer value of substring from position after the character representing the separator TO the end of time  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
        
        SET minutes_list TO list of applying function to_minutes on each element IN timePoints  
        SORT minutes_list IN ascending order  
        SET min_diff TO positive infinity  
        FOR index FROM one TO the length of minutes_list MINUS one  
            SET current_difference TO element at position index of minutes_list MINUS element at position index MINUS one of minutes_list  
            IF current_difference LESS THAN min_diff  
                SET min_diff TO current_difference  
            END IF  
        END FOR  
        SET circular_difference TO one thousand four hundred forty MINUS element at last position of minutes_list PLUS element at first position of minutes_list  
        IF circular_difference LESS THAN min_diff  
            SET min_diff TO circular_difference  
        END IF  
        RETURN min_diff  
    END FUNCTION  
END CLASS
