CLASS Solution  
    FUNCTION flipgame(fronts list OF integer backs list OF integer) RETURNS integer  
        SET same TO empty set  
        FOR each pair x AND y IN the pairs formed by elements at corresponding positions of fronts AND backs  
            IF x EQUALS y  
                ADD x TO same  
            END IF  
        END FOR  
        SET all_numbers TO empty set  
        FOR each number IN fronts  
            ADD number TO all_numbers  
        END FOR  
        FOR each number IN backs  
            ADD number TO all_numbers  
        END FOR  
        SET good_numbers TO empty set  
        FOR each number IN all_numbers  
            IF number NOT IN same  
                ADD number TO good_numbers  
            END IF  
        END FOR  
        IF good_numbers IS NOT empty  
            SET smallest_good_number TO the smallest element IN good_numbers  
            RETURN smallest_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
