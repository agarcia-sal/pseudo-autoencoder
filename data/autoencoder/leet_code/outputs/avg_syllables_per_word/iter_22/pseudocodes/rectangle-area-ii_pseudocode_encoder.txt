CLASS Solution  
    FUNCTION rectangleArea(rectangles)  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS one  
        SET y_coords TO EMPTY SET  
        FOR each x1 y1 x2 y2 IN rectangles  
            ADD y1 TO y_coords  
            ADD y2 TO y_coords  
        END FOR  
        SET y_coords TO SORTED LIST OF ELEMENTS IN y_coords  
        SET y_index TO EMPTY DICTIONARY  
        SET i TO zero  
        FOR each y_value IN y_coords  
            SET y_index AT y_value TO i  
            INCREMENT i BY one  
        END FOR  
        SET count TO LIST OF ZERO VALUES WITH LENGTH EQUAL TO LENGTH OF y_coords  
        SET events TO EMPTY LIST  
        FOR each x1 y1 x2 y2 IN rectangles  
            APPEND A TUPLE OF x1 ONE y1 y2 TO events  
            APPEND A TUPLE OF x2 NEGATIVE ONE y1 y2 TO events  
        END FOR  
        SORT events BY THE FIRST ELEMENT OF EACH TUPLE IN ASCENDING ORDER  
        SET prev_x TO zero  
        SET area TO zero  
        FOR each x eventType y1 y2 IN events  
            SET current_length TO zero  
            FOR i FROM one TO LENGTH OF count MINUS one  
                IF ELEMENT AT POSITION i MINUS one OF count IS GREATER THAN zero  
                    INCREMENT current_length BY ELEMENT AT POSITION i OF y_coords MINUS ELEMENT AT POSITION i MINUS one OF y_coords  
                END IF  
            END FOR  
            INCREMENT area BY current_length MULTIPLIED BY x MINUS prev_x  
            SET area TO area MODULO MOD  
            IF eventType EQUALS one  
                FOR i FROM y_index AT y1 TO y_index AT y2 MINUS one  
                    INCREMENT ELEMENT AT POSITION i OF count BY one  
                END FOR  
            ELSE  
                FOR i FROM y_index AT y1 TO y_index AT y2 MINUS one  
                    DECREMENT ELEMENT AT POSITION i OF count BY one  
                END FOR  
            END IF  
            SET prev_x TO x  
        END FOR  
        RETURN area  
    END FUNCTION  
END CLASS
