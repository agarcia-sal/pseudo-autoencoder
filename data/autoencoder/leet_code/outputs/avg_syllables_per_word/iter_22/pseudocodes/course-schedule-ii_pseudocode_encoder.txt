CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO an empty mapping of course TO empty list for each course FROM zero TO numCourses MINUS one  
        SET in_degrees TO a list of zeros of length numCourses  
        
        FOR each pair dest PLUS src IN prerequisites  
            APPEND dest TO element at position src of adj_list  
            INCREMENT element at position dest of in_degrees BY one  
        END FOR  
        
        SET queue TO an empty double ended queue  
        FOR each course FROM zero TO numCourses MINUS one  
            IF element at position course of in_degrees EQUALS zero  
                APPEND course TO queue  
            END IF  
        END FOR  
        
        SET topological_order TO an empty list  
        
        WHILE queue is not empty  
            REMOVE the first element FROM queue AND SET IT TO current  
            APPEND current TO topological_order  
            
            FOR each neighbor IN element at position current of adj_list  
                DECREMENT element at position neighbor of in_degrees BY one  
                IF element at position neighbor of in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
