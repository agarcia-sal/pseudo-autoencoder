CLASS Solution  
    FUNCTION numOfArrays(n m k)  
        SET MOD TO one billion PLUS seven  
        INITIALIZE three dimensional list dp WITH zeros WITH dimensions n PLUS one BY m PLUS one BY k PLUS one  
        FOR j FROM one TO m  
            SET element at position j AT position one AT position one OF dp TO one  
        END FOR  
        FOR i FROM two TO n  
            FOR j FROM one TO m  
                FOR p FROM one TO k  
                    FOR x FROM one TO j MINUS one  
                        SET element at position j AT position i AT position p OF dp TO element at position j AT position i AT position p OF dp PLUS element at position x AT position i MINUS one AT position p MINUS one OF dp MODULO MOD  
                    END FOR  
                    SET element at position j AT position i AT position p OF dp TO element at position j AT position i AT position p OF dp PLUS element at position j AT position i MINUS one AT position p OF dp MULTIPLIED BY j MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        SET result TO zero  
        FOR j FROM one TO m  
            SET result TO result PLUS element at position j AT position n AT position k OF dp MODULO MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
