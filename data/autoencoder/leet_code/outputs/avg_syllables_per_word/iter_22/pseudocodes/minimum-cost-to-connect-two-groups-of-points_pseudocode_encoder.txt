CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF element at position zero of cost  
        SET dp TO a list of lists each containing infinity repeated TWO RAISED TO THE POWER OF size2 elements with length size1 PLUS ONE  
        SET element at position zero zero of dp TO zero  
        
        FOR i FROM zero TO size1 MINUS one  
            FOR mask FROM zero TO TWO RAISED TO THE POWER OF size2 MINUS one  
                FOR j FROM zero TO size2 MINUS one  
                    SET current_mask TO mask OR TWO RAISED TO THE POWER OF j  
                    SET value at position i PLUS ONE current_mask of dp TO the MINIMUM OF value at position i PLUS ONE current_mask of dp AND the SUM OF value at position i mask of dp AND element at position i j of cost  
                END FOR  
                
                FOR j FROM zero TO size2 MINUS one  
                    IF the BITWISE AND of mask AND TWO RAISED TO THE POWER OF j is GREATER THAN zero  
                        SET new_mask TO the BITWISE XOR of mask AND TWO RAISED TO THE POWER OF j  
                        SET value at position i PLUS ONE mask of dp TO the MINIMUM OF value at position i PLUS ONE mask of dp AND the SUM OF value at position i PLUS ONE new_mask of dp AND element at position i j of cost  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN value at position size1 TWO RAISED TO THE POWER OF size2 MINUS one of dp  
    END FUNCTION  
END CLASS
