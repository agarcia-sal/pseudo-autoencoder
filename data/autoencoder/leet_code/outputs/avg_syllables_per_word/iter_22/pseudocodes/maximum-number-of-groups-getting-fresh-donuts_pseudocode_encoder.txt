CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO a list of zero repeated batchSize times  
        FOR each group IN groups  
            INCREMENT the element at position corresponding to the remainder of group DIVIDED BY batchSize in remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder, variable number of remainder_count arguments)  
            CONVERT remainder_count arguments TO a list named remainder_count  
            SET max_happy TO zero  
            
            FOR index FROM zero TO batchSize MINUS one  
                IF the element at position index of remainder_count EQUALS zero  
                    CONTINUE to next iteration  
                END IF  
                
                DECREMENT the element at position index of remainder_count BY one  
                SET new_remainder TO the remainder of remainder MINUS index DIVIDED BY batchSize  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS the RESULT OF calling dp WITH new_remainder and contents of remainder_count  
                ELSE  
                    SET happy_groups TO the RESULT OF calling dp WITH new_remainder and contents of remainder_count  
                END IF  
                
                IF happy_groups GREATER THAN max_happy  
                    SET max_happy TO happy_groups  
                END IF  
                
                INCREMENT the element at position index of remainder_count BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN the RESULT OF calling dp WITH zero and contents of remainder_count  
    END FUNCTION  
END CLASS
