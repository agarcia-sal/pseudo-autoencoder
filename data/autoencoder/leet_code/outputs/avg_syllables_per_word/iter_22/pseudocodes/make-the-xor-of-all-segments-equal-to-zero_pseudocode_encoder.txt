CLASS Solution  
    FUNCTION minChanges(nums k)  
        SET groups TO empty map with default empty list  
        FOR i FROM zero TO LENGTH OF nums MINUS one  
            APPEND element at position i of nums TO element at position i MODULO k of groups  
        END FOR  
        
        SET freq TO list containing k empty maps with default zero integer  
        SET min_size TO list containing k elements each set to positive infinity  
        FOR i FROM zero TO k MINUS one  
            FOR each num IN element at position i of groups  
                INCREMENT element at key num of element at position i of freq BY one  
            END FOR  
            SET element at position i of min_size TO LENGTH OF element at position i of groups  
        END FOR  
        
        SET dp TO list containing two to the power of ten elements each set to positive infinity  
        SET element at position zero of dp TO zero  
        
        FOR i FROM zero TO k MINUS one  
            SET new_dp TO list containing two to the power of ten elements each set to positive infinity  
            SET min_dp TO the minimum value in dp  
            FOR j FROM zero TO two to the power of ten MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position i of min_size  
                FOR each num count IN elements of element at position i of freq  
                    SET candidate TO element at position j BITWISE XOR num of dp PLUS element at position i of min_size MINUS count  
                    IF candidate LESS THAN element at position j of new_dp  
                        SET element at position j of new_dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
