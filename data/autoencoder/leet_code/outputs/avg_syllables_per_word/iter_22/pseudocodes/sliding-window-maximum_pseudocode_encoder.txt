CLASS Solution  
    FUNCTION maxSlidingWindow(nums k)  
        IF the collection nums IS EMPTY OR k EQUALS zero  
            RETURN an empty list  
        END IF  
        ASSIGN reference TO an empty double ended queue named deq  
        ASSIGN reference TO an empty list named result  
        FOR index i FROM zero TO the LENGTH OF nums MINUS one  
            IF the double ended queue deq IS NOT EMPTY AND the element at position zero of deq IS LESS THAN i MINUS k PLUS one  
                REMOVE the element from the left end of deq  
            END IF  
            WHILE the double ended queue deq IS NOT EMPTY AND the element at position element at position last of deq IN nums IS LESS THAN the element at position i in nums  
                REMOVE the element from the right end of deq  
            END WHILE  
            APPEND the integer i TO the right end of deq  
            IF i IS GREATER THAN OR EQUAL TO k MINUS one  
                APPEND the element at position element at position zero of deq IN nums TO the list result  
            END IF  
        END FOR  
        RETURN the list result  
    END FUNCTION  
END CLASS
