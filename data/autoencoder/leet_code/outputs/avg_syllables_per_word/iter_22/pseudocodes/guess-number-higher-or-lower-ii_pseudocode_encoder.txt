CLASS Solution  
    FUNCTION getMoneyAmount(n)  
        FUNCTION dp(left, right)  
            IF left GREATER THAN OR EQUAL TO right THEN  
                RETURN zero  
            END IF  
            SET minimum cost TO infinity  
            FOR pivot FROM left TO right DO  
                SET cost TO pivot PLUS maximum of dp of left and pivot MINUS one AND dp of pivot PLUS one and right  
                SET minimum cost TO minimum of minimum cost AND cost  
            END FOR  
            RETURN minimum cost  
        END FUNCTION  
        RETURN dp of one and n  
    END FUNCTION  
END CLASS
