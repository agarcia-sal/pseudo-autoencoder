CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator TO iterable over s  
            RETURN logical AND of each character of word being contained in iterator  
        END FUNCTION  
        
        SORT dictionary BY descending LENGTH OF each element THEN ascending LEXICOGRAPHICAL ORDER OF each element  
        
        FOR each word IN dictionary  
            IF is_subsequence(word, s) EQUALS true THEN  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN empty string  
    END FUNCTION  
END CLASS
