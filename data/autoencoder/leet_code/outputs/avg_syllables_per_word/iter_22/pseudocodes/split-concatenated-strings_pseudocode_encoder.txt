CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR i FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the string formed by the characters of element at position i of strs in reverse order  
            IF reversed_str GREATER THAN element at position i of strs  
                SET element at position i of strs TO reversed_str  
            END IF  
        END FOR  
        
        SET doubled_strs TO the list formed by concatenating strs WITH strs  
        
        SET max_string TO the empty string  
        
        FOR i FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN the list containing false AND true  
                IF is_reversed EQUALS true  
                    SET current TO the string formed by the characters of element at position i of strs in reverse order  
                ELSE  
                    SET current TO element at position i of strs  
                END IF  
                
                SET remaining TO the string formed by concatenating the elements of doubled_strs STARTING FROM position i PLUS one TO position i PLUS the LENGTH OF strs MINUS one IN ORDER  
                
                FOR j FROM zero TO the LENGTH OF current  
                    SET candidate TO the string formed by concatenating the substring of current FROM position j TO the end WITH remaining WITH the substring of current FROM the beginning TO position j MINUS one  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
