CLASS ListNode  
    FUNCTION __init__ with parameters val DEFAULT zero AND next DEFAULT None  
        SET val attribute TO val  
        SET next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET head TO a new ListNode created with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from the second element TO the last element  
        SET node TO a new ListNode created with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the condition that the val attribute of p1 EQUALS the val attribute of p2 AND the result of calling is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ with parameters val DEFAULT zero AND left DEFAULT None AND right DEFAULT None  
        SET val attribute TO val  
        SET left attribute TO left  
        SET right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode created with the element at position one of values  
    SET i TO one  
    INITIALIZE an empty deque named queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the first element REMOVED FROM queue  
        IF i LESS THAN OR EQUAL TO the LENGTH OF values AND the element at position i of values DOES NOT EQUAL None THEN  
            SET the left attribute of node TO a new TreeNode created with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN OR EQUAL TO the LENGTH OF values AND the element at position i of values DOES NOT EQUAL None THEN  
            SET the right attribute of node TO a new TreeNode created with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p AND q  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with the left attributes of p AND q AND the result of calling is_same_tree with the right attributes of p AND q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree with parameter s  
        FUNCTION parse_tree with parameter index  
            IF index GREATER THAN OR EQUAL TO the LENGTH OF s THEN  
                RETURN None AND index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the LENGTH OF s AND (the element at position index of s IS a digit OR the element at position index of s EQUALS the character dash)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO a new TreeNode created with the integer value derived from the substring of s from position start TO position index MINUS one  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character representing open parenthesis THEN  
                SET the left attribute of node AND index TO the two VALUES RETURNED by calling parse_tree with index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character representing open parenthesis THEN  
                SET the right attribute of node AND index TO the two VALUES RETURNED by calling parse_tree with index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character representing close parenthesis THEN  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND ignored_variable TO the two VALUES RETURNED by calling parse_tree with zero  
        RETURN root  
    END FUNCTION  
END CLASS
