CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty collection  
            RETURN zero  
        END IF  
        SORT intervals BY value at position zero of each element in ascending order  
        SET min_heap TO empty collection  
        FOR each meeting IN intervals  
            IF min_heap NOT EQUALS empty collection AND value at position zero of meeting GREATER THAN OR EQUAL TO element at position zero of min_heap  
                CALL remove the smallest element from min_heap  
            END IF  
            CALL add the value at position one of meeting TO min_heap preserving order  
        END FOR  
        RETURN the number of elements in min_heap  
    END FUNCTION  
END CLASS
