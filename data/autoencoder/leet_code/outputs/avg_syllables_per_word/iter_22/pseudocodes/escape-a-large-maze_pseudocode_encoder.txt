CLASS Solution  
    FUNCTION isEscapePossible(blocked, source, target)  
        IF blocked EQUALS empty  
            RETURN True  
        END IF  

        SET blocked_set TO the set of tuples of each block element in blocked  
        SET n TO the LENGTH OF blocked  
        SET max_steps TO n MULTIPLIED BY n MINUS one DIVIDED BY two  

        FUNCTION bfs(start, end)  
            SET queue TO a double-ended queue initialized with the tuple of start  
            SET visited TO a set initialized with the elements in queue  

            WHILE queue is not empty  
                REMOVE the element from the front of queue and ASSIGN it to x and y  

                IF the LENGTH OF visited GREATER THAN max_steps OR x EQUALS the element at position zero of end AND y EQUALS the element at position one of end  
                    RETURN True  
                END IF  

                FOR each pair dx and dy IN the list of pairs zero and one zero and minus one one and zero and minus one and zero  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  

                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN ten TO THE POWER OF six AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN ten TO THE POWER OF six AND the tuple of nx and ny NOT IN blocked_set AND the tuple of nx and ny NOT IN visited  
                        ADD the tuple of nx and ny TO visited  
                        APPEND the tuple of nx and ny TO queue  
                    END IF  
                END FOR  
            END WHILE  

            RETURN False  
        END FUNCTION  

        RETURN the result of bfs with source and target AND the result of bfs with target and source  
    END FUNCTION  
END CLASS
