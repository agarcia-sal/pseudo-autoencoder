CLASS Solution  
    FUNCTION minPatches(nums parameter LIST OF INTEGER n parameter INTEGER)  
        SET missing TO the integer one representing the smallest number that cannot be formed  
        SET patches TO zero representing the count of patches added  
        SET i TO zero representing the index in nums being considered  
        WHILE the value of missing is less than or equal to the value of n  
            IF the value of i is less than the length of the collection nums AND the element at position i of nums is less than or equal to the value of missing  
                SET missing TO the value of missing PLUS the element at position i of nums  
                INCREMENT i BY one  
            ELSE  
                SET missing TO the value of missing PLUS the value of missing  
                INCREMENT patches BY one  
            END IF  
        END WHILE  
        RETURN the value of patches  
    END FUNCTION  
END CLASS
