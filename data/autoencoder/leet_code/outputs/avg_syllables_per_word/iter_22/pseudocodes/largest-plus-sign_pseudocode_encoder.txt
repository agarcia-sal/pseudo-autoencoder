CLASS Solution  
    FUNCTION orderOfLargestPlusSign WITH PARAMETERS n AND mines  
        SET mines_set TO a new empty set  
        FOR each mine_element IN mines  
            ADD the tuple of mine_element TO mines_set  
        END FOR  
        
        SET dp TO a new two-dimensional list of size n BY n where each element is a list of four zeros representing directions  
        
        SET directions TO a list containing the pairs of zeros and ones and minus ones representing right down left and up directions  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair of i AND j is NOT IN mines_set THEN  
                    IF j GREATER THAN zero THEN  
                        SET element at position zero of the element at position j MINUS one of the element at position i of dp TO the value of that element PLUS one  
                        SET element at position zero of the element at position j of the element at position i of dp TO element at position zero of the element at position j MINUS one of the element at position i of dp PLUS one  
                    END IF  
                    IF i GREATER THAN zero THEN  
                        SET element at position one of the element at position j of the element at position i of dp TO element at position one of the element at position j of the element at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the pair of i AND j is NOT IN mines_set THEN  
                    IF j LESS THAN n MINUS one THEN  
                        SET element at position two of the element at position j PLUS one of the element at position i of dp TO the value of that element  
                        SET element at position two of the element at position j of the element at position i of dp TO element at position two of the element at position j PLUS one of the element at position i of dp PLUS one  
                    END IF  
                    IF i LESS THAN n MINUS one THEN  
                        SET element at position three of the element at position j of the element at position i PLUS one of dp TO the value of that element  
                        SET element at position three of the element at position j of the element at position i of dp TO element at position three of the element at position j of the element at position i PLUS one of dp PLUS one  
                    END IF  
                    
                    SET plus_sign_order TO the minimum value among the list of four directional values at the element at position j of element at position i of dp  
                    IF plus_sign_order GREATER THAN result THEN  
                        SET result TO plus_sign_order  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
