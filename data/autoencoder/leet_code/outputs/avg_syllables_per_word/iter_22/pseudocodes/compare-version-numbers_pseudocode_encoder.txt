CLASS Solution  
    FUNCTION compareVersion(version1, version2)  
        SET v1_revisions TO the list of substrings of version1 divided by the full stop character  
        SET v2_revisions TO the list of substrings of version2 divided by the full stop character  
        SET max_length TO the greater value BETWEEN the length of v1_revisions AND the length of v2_revisions  
        FOR index FROM zero TO max_length MINUS one  
            IF the index IS LESS THAN the length of v1_revisions  
                SET v1 TO the integer value of the element at position index of v1_revisions  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF the index IS LESS THAN the length of v2_revisions  
                SET v2 TO the integer value of the element at position index of v2_revisions  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 IS LESS THAN v2  
                RETURN negative one  
            ELSE IF v1 IS GREATER THAN v2  
                RETURN positive one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
