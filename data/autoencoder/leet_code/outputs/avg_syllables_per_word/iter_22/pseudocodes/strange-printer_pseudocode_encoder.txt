CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            IF the TUPLE of i AND j IS IN memo THEN  
                RETURN the VALUE ASSOCIATED WITH the TUPLE of i AND j IN memo  
            END IF  
            SET result TO the VALUE RETURNED BY dp WITH PARAMETERS i AND j MINUS one AND memo PLUS one  
            FOR k FROM i TO j MINUS one  
                IF the CHARACTER AT POSITION k OF s EQUALS the CHARACTER AT POSITION j OF s THEN  
                    SET candidate TO the VALUE RETURNED BY dp WITH PARAMETERS i AND k MINUS one AND memo  
                    SET candidate TO candidate PLUS the VALUE RETURNED BY dp WITH PARAMETERS k AND j MINUS one AND memo  
                    IF candidate LESS THAN result THEN  
                        SET result TO candidate  
                    END IF  
                END IF  
            END FOR  
            SET the VALUE ASSOCIATED WITH the TUPLE of i AND j IN memo TO result  
            RETURN result  
        END FUNCTION  
        RETURN the VALUE RETURNED BY dp WITH PARAMETERS zero AND the LENGTH OF s MINUS one AND an EMPTY COLLECTION  
    END FUNCTION  
END CLASS
