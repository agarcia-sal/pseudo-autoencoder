CLASS Solution  
    FUNCTION totalNQueens with parameter n  
        FUNCTION is_not_under_attack with parameters row and col  
            RETURN the logical negation of the sum of element at position col of the list named cols PLUS element at position row MINUS col of the list named hills PLUS element at position row PLUS col of the list named dales EQUALS zero  
        END FUNCTION  
        
        FUNCTION place_queen with parameters row and col  
            SET element at position col of the list named cols TO one  
            SET element at position row MINUS col of the list named hills TO one  
            SET element at position row PLUS col of the list named dales TO one  
        END FUNCTION  
        
        FUNCTION remove_queen with parameters row and col  
            SET element at position col of the list named cols TO zero  
            SET element at position row MINUS col of the list named hills TO zero  
            SET element at position row PLUS col of the list named dales TO zero  
        END FUNCTION  
        
        FUNCTION backtrack with parameters row defaulted to zero and count defaulted to zero  
            FOR col FROM zero TO n MINUS one  
                IF is_not_under_attack with arguments row and col EQUALS true  
                    CALL place_queen with arguments row and col  
                    IF row PLUS one EQUALS n  
                        INCREMENT count BY one  
                    ELSE  
                        SET count TO the result of calling backtrack with arguments row PLUS one and count  
                    END IF  
                    CALL remove_queen with arguments row and col  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET cols TO a list of length n filled with zeros  
        SET hills TO a list of length two MULTIPLIED BY n MINUS one filled with zeros  
        SET dales TO a list of length two MULTIPLIED BY n MINUS one filled with zeros  
        
        RETURN the result of calling backtrack with no arguments  
    END FUNCTION  
END CLASS
