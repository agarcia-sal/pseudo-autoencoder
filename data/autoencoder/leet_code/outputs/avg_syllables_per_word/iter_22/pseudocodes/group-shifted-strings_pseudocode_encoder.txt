CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF the LENGTH OF s EQUALS zero  
                RETURN the empty string  
            END IF  
            SET canonical_form TO an empty list  
            FOR each character c IN s  
                SET difference TO the ASCII numeric value of character c MINUS the ASCII numeric value of the first character of s  
                SET wrapped_difference TO the remainder of difference DIVIDED BY twenty six  
                CONVERT wrapped_difference TO its string representation  
                APPEND this string representation TO canonical_form  
            END FOR  
            RETURN the concatenation of all elements in canonical_form JOINED BY the string consisting of the comma character  
        END FUNCTION  
        
        SET groups TO a new dictionary that returns an empty list for missing keys  
        FOR each string s IN strings  
            SET canonical_form TO the result of calling get_canonical_form WITH s  
            APPEND s TO the list stored at the key canonical_form in groups  
        END FOR  
        RETURN the list of all values in groups  
    END FUNCTION  
END CLASS
