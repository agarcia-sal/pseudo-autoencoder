CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN a null reference  
    END IF  
    SET head TO a new ListNode with the element at position one of values and a null reference  
    SET p TO head  
    FOR each val IN the sublist of values starting from the second element TO the last element  
        SET node TO a new ListNode with val and a null reference  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS a null reference AND p2 IS a null reference  
        RETURN True  
    END IF  
    IF p1 IS a null reference OR p2 IS a null reference  
        RETURN False  
    END IF  
    RETURN the condition that the value of p1 EQUALS the value of p2 AND the result of calling is_same_list with the next element of p1 and the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList(self, head)  
        IF head IS a null reference OR the next element of head IS a null reference  
            RETURN head  
        END IF  

        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO the next element of head  
            WHILE fast IS NOT a null reference AND the next element of fast IS NOT a null reference  
                SET slow TO the next element of slow  
                SET fast TO the element that is two steps after fast in the linked sequence  
            END WHILE  
            SET mid TO the next element of slow  
            SET the next element of slow TO a null reference  
            RETURN head and mid  
        END FUNCTION  

        FUNCTION merge(l1, l2)  
            SET dummy TO a new ListNode with default value zero and a null reference  
            SET current TO dummy  
            WHILE l1 IS NOT a null reference AND l2 IS NOT a null reference  
                IF the value of l1 IS LESS THAN the value of l2  
                    SET the next element of current TO l1  
                    SET l1 TO the next element of l1  
                ELSE  
                    SET the next element of current TO l2  
                    SET l2 TO the next element of l2  
                END IF  
                SET current TO the next element of current  
            END WHILE  
            IF l1 IS NOT a null reference  
                SET the next element of current TO l1  
            ELSE  
                SET the next element of current TO l2  
            END IF  
            RETURN the next element of dummy  
        END FUNCTION  

        SET left and right TO the result of calling split with head  
        SET left TO the result of calling sortList with left  
        SET right TO the result of calling sortList with right  

        RETURN the result of calling merge with left and right  
    END FUNCTION  
END CLASS
