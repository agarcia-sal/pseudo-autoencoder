CLASS Solution  
    FUNCTION outerTrees(trees)  
        FUNCTION cross(i, j, k)  
            SET a TO element at position i of trees  
            SET b TO element at position j of trees  
            SET c TO element at position k of trees  
            RETURN (element at position zero of b MINUS element at position zero of a) MULTIPLIED BY (element at position one of c MINUS element at position one of b) MINUS (element at position one of b MINUS element at position one of a) MULTIPLIED BY (element at position zero of c MINUS element at position zero of b)  
        END FUNCTION  
        
        SET n TO the LENGTH OF trees  
        IF n LESS THAN four  
            RETURN trees  
        END IF  
        SORT trees IN ascending order  
        SET vis TO a list of n elements all assigned to False  
        SET stk TO a list with element zero as its only item  
        FOR i FROM one TO n MINUS one  
            WHILE the LENGTH OF stk GREATER THAN one AND cross(element at position LENGTH OF stk MINUS two of stk, element at position LENGTH OF stk MINUS one of stk, i) LESS THAN zero holds  
                SET removed_index TO REMOVE the last element FROM stk  
                SET element at position removed_index of vis TO False  
            END WHILE  
            SET element at position i of vis TO True  
            APPEND i TO stk  
        END FOR  
        SET m TO the LENGTH OF stk  
        FOR i FROM n MINUS two DOWN TO zero  
            IF element at position i of vis EQUALS True  
                CONTINUE to next iteration  
            END IF  
            WHILE the LENGTH OF stk GREATER THAN m AND cross(element at position LENGTH OF stk MINUS two of stk, element at position LENGTH OF stk MINUS one of stk, i) LESS THAN zero holds  
                REMOVE the last element FROM stk  
            END WHILE  
            APPEND i TO stk  
        END FOR  
        REMOVE the last element FROM stk  
        SET result TO an empty list  
        FOR each index IN stk  
            APPEND element at position index of trees TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
