CLASS Solution  
    FUNCTION ipToCIDR(ip, n)  
        FUNCTION ip_to_int(ip)  
            SET parts TO the LIST OF integer values converted from each element in the sequence obtained by splitting the ip string at each occurrence of the period word  
            SET res TO zero  
            FOR each part IN parts  
                SET res TO res MULTIPLIED BY two HUNDRED FIFTY SIX PLUS part  
            END FOR  
            RETURN res  
        END FUNCTION  

        FUNCTION int_to_ip(num)  
            SET octets TO empty list  
            FOR index FROM three DOWN TO zero  
                APPEND the remainder of the integer division of num DIVIDED BY two HUNDRED FIFTY SIX RAISED TO the power of index AND THEN MODULO two HUNDRED FIFTY SIX TO octets  
            END FOR  
            SET ip_string TO the concatenation of the string conversion of each element in octets joined with the period word as separator  
            RETURN ip_string  
        END FUNCTION  

        FUNCTION largest_cidr_size(start, n)  
            IF the bitwise AND of start AND one EQUALS one  
                RETURN one  
            END IF  
            SET x TO one  
            WHILE x IS LESS THAN OR EQUAL TO n AND the bitwise AND of start AND the subtraction of x MINUS one EQUALS zero  
                SET x TO x MULTIPLIED BY two  
            END WHILE  
            RETURN the integer division of x BY two  
        END FUNCTION  

        SET result TO empty list  
        SET start_ip TO the result of calling ip_to_int with ip as argument  

        WHILE n IS GREATER THAN zero  
            SET size TO the result of calling largest_cidr_size with start_ip AND n as arguments  
            SET cidr_suffix TO thirty TWO MINUS the difference of the bit length of size MINUS one  
            SET cidr_block TO the concatenation of the result of calling int_to_ip with start_ip as argument, the forward slash word, AND the string conversion of cidr_suffix  
            APPEND cidr_block TO result  
            SET start_ip TO start_ip PLUS size  
            SET n TO n MINUS size  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
