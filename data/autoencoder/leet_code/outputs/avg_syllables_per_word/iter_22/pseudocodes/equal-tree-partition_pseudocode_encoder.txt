CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET val TO val  
        SET left TO left  
        SET right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with element at position one of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO a new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO a new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree of value left of p AND is_same_tree of value left of q AND is_same_tree of value right of p AND is_same_tree of value right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree with parameters self root  
        FUNCTION subtree_sum with parameter node  
            IF node EQUALS None  
                RETURN zero  
            END IF  
            RETURN value of node PLUS subtree_sum of value left of node PLUS subtree_sum of value right of node  
        END FUNCTION  

        SET total_sum TO subtree_sum of root  

        IF total_sum MODULO two NOT EQUALS zero  
            RETURN False  
        END IF  

        SET target TO total_sum DIVIDED BY two  
        SET found TO a list containing False  

        FUNCTION can_split with parameter node  
            IF node EQUALS None  
                RETURN zero  
            END IF  
            SET current_sum TO value of node PLUS can_split of value left of node PLUS can_split of value right of node  
            IF current_sum EQUALS target AND node NOT EQUALS root  
                SET element at position one of found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  

        CALL can_split with root  
        RETURN element at position one of found  
    END FUNCTION  
END CLASS
