CLASS Solution  
    FUNCTION kSimilarity(s1 s2)  
        FUNCTION neighbors(s)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND the element at position i of s EQUALS the element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF the element at position j of s EQUALS the element at position i of s2 AND the element at position j of s NOT EQUALS the element at position j of s2  
                    YIELD concatenation of  
                        substring from position zero TO position i MINUS one of s,  
                        element at position j of s,  
                        substring from position i PLUS one TO position j MINUS one of s,  
                        element at position i of s,  
                        substring from position j PLUS one TO the end of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        ASSIGN reference TO an empty double ended queue initialized with a tuple containing s1 and zero  
        ASSIGN reference TO a set initialized with s1  
        
        WHILE the double ended queue is not empty  
            REMOVE and ASSIGN tuple containing s and swaps from the left end of the double ended queue  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors(s)  
                IF neighbor is not an element of the visited set  
                    ADD neighbor TO the visited set  
                    APPEND a tuple containing neighbor and swaps PLUS one TO the right end of the double ended queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
