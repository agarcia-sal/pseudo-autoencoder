CLASS Solution  
    FUNCTION frequencySort(s)  
        SET frequency TO the result of counting each character occurring in s  
        SET sorted_characters TO the list of characters in frequency sorted in order where the character with a greater frequency comes before a character with a lesser frequency  
        SET result TO empty string  
        FOR each character IN sorted_characters  
            APPEND the character repeated the number of times equal to the frequency of character TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
