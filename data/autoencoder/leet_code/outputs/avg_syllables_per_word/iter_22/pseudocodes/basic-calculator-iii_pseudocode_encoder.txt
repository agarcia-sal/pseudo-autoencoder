CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO the character plus  
            WHILE the LENGTH OF s GREATER THAN zero  
                SET char TO the element at position zero of s  
                REMOVE element at position zero FROM s  
                IF char represents a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                IF char EQUALS the character of opening parenthesis  
                    SET num TO the result of calling helper with argument s  
                END IF  
                IF char EQUALS the character plus OR char EQUALS the character minus OR char EQUALS the character multiplied OR char EQUALS the character divided OR char EQUALS the character of opening parenthesis OR char EQUALS the character of closing parenthesis OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS the character plus  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS the character minus  
                        APPEND the negation of num TO stack  
                    ELSE IF sign EQUALS the character multiplied  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS the character divided  
                        SET the element at the last position of stack TO the integer part of the element at the last position of stack DIVIDED BY num with truncation towards zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS the character of closing parenthesis  
                    BREAK  
                END IF  
            END WHILE  
            RETURN the sum of all elements in stack  
        END FUNCTION  

        SET s TO the list form of s where each character is an element  
        RETURN the result of calling helper with argument s  
    END FUNCTION  
END CLASS
