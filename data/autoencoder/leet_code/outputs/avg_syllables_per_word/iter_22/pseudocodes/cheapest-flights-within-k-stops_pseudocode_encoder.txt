CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO a map from integer TO empty list for every integer FROM zero TO n MINUS one  
        FOR each element consisting of u v w IN flights  
            APPEND the pair consisting of v AND w TO the list corresponding to key u IN graph  
        END FOR  
        SET min_heap TO a list containing a tuple of zero cost, minus one stops, AND src city  
        SET min_cost TO a map with key src assigned to minus one  
        WHILE the list min_heap is not empty  
            REMOVE and RETRIEVE from min_heap the element with the smallest cost, ASSIGNING cost, stops, AND city  
            IF city EQUALS dst THEN  
                RETURN cost  
            END IF  
            IF stops GREATER THAN OR EQUAL TO k THEN  
                CONTINUE to next iteration of the WHILE loop  
            END IF  
            FOR each element consisting of neighbor AND price IN the list corresponding to key city IN graph  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  
                IF neighbor NOT IN keys of min_cost OR next_stops LESS THAN min_cost at neighbor OR next_cost LESS THAN the first element of the smallest tuple currently in min_heap  
                    ASSIGN key neighbor in min_cost TO next_stops  
                    INSERT the tuple consisting of next_cost, next_stops, AND neighbor INTO min_heap maintaining heap order by cost  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
