CLASS Solution  
    FUNCTION mirrorReflection(p, q)  
        SET lcm TO the product of p AND q DIVIDED BY the greatest common divisor of p AND q  
        SET vertical_reflections TO lcm DIVIDED BY p  
        SET horizontal_reflections TO lcm DIVIDED BY q  
        IF the remainder of vertical_reflections DIVIDED BY two EQUALS zero  
            RETURN zero  
        ELSE IF the remainder of horizontal_reflections DIVIDED BY two EQUALS one  
            RETURN one  
        ELSE  
            RETURN two  
        END IF  
    END FUNCTION  
END CLASS
