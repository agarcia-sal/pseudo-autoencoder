CLASS Solution  
    FUNCTION kSmallestPairs(nums1 nums1 OF TYPE List OF Integer nums2 nums2 OF TYPE List OF Integer k k OF TYPE Integer) RETURNS List OF List OF Integer  
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k LESS THAN OR EQUAL TO zero THEN  
            RETURN an EMPTY list  
        END IF  
        SET min_heap TO an EMPTY list  
        FOR each i FROM zero TO the SMALLEST VALUE BETWEEN k MINUS one AND the LENGTH OF nums1 MINUS one  
            PUSH ONTO min_heap a TRIPLE CONSISTING OF the ELEMENT AT POSITION i OF nums1 PLUS the ELEMENT AT POSITION zero OF nums2 THE VALUE i AND zero  
        END FOR  
        SET result TO an EMPTY list  
        WHILE min_heap IS NOT EMPTY AND the LENGTH OF result IS LESS THAN k DO  
            POP the SMALLEST element FROM min_heap ASSIGN ITS VALUES TO _, i, j  
            APPEND a LIST CONSISTING OF the ELEMENT AT POSITION i OF nums1 AND the ELEMENT AT POSITION j OF nums2 TO result  
            IF j PLUS one IS LESS THAN the LENGTH OF nums2 THEN  
                PUSH ONTO min_heap a TRIPLE CONSISTING OF the ELEMENT AT POSITION i OF nums1 PLUS the ELEMENT AT POSITION j PLUS one OF nums2 THE VALUE i AND j PLUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
