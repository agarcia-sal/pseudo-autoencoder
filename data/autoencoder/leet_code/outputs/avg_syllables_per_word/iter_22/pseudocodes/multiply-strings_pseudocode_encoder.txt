CLASS Solution  
    FUNCTION multiply(num1 as string num2 as string) RETURNS string  
        IF num1 EQUALS the character zero OR num2 EQUALS the character zero  
            RETURN the character zero  
        END IF  

        SET result TO a list of zeros with the length equal to the LENGTH OF num1 PLUS the LENGTH OF num2  

        SET num1 TO the string num1 reversed in order of characters  
        SET num2 TO the string num2 reversed in order of characters  

        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the value of the character at position i of num1 converted to its numeric digit value MULTIPLIED BY the value of the character at position j of num2 converted to its numeric digit value  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS the element at position p1 of result  

                SET the element at position p1 of result TO sum_ MODULO ten  
                INCREMENT the element at position p2 of result BY sum_ DIVIDED BY ten  
            END FOR  
        END FOR  

        SET result_str TO the concatenation of all numeric elements in result reversed in order converted to string form  
        RETURN the string result_str with all leading characters equal to the character zero removed  
    END FUNCTION  
END CLASS
