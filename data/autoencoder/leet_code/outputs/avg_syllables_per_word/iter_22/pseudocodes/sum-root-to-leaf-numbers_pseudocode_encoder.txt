CLASS TreeNode  
    FUNCTION __init__ with parameters val DEFAULT zero left DEFAULT None right DEFAULT None  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode initialized with element at position one of values  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO element removed from the left end of queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left q.left AND the result of is_same_tree with parameters p.right q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers with parameters self root  
        FUNCTION dfs with parameters node current_number  
            IF node EQUALS None  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS node.val  
            IF node.left EQUALS None AND node.right EQUALS None  
                RETURN current_number  
            END IF  
            SET left_sum TO result of dfs with parameters node.left current_number  
            SET right_sum TO result of dfs with parameters node.right current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN result of dfs with parameters root zero  
    END FUNCTION  
END CLASS
