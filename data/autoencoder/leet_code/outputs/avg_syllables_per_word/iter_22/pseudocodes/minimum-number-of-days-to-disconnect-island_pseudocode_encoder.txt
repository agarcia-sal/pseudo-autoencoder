CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the ELEMENT AT position x OF grid AND position y OF THAT ELEMENT EQUALS zero OR the PAIR x AND y IS IN visited THEN  
                RETURN  
            END IF  
            ADD the PAIR x AND y TO visited  
            FOR each pair dx AND dy IN the LIST OF PAIRS representing movement one UP AND zero horizontal, one DOWN AND zero horizontal, zero vertical AND one LEFT, zero vertical AND one RIGHT  
                CALL dfs WITH the ARGUMENTS the SUM OF x AND dx, the SUM OF y AND dy, AND visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an empty set  
            SET land_cells TO an empty list  
            FOR each i FROM zero TO the LENGTH OF grid MINUS one  
                FOR each j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                    IF the ELEMENT AT position i OF grid AND position j OF THAT ELEMENT EQUALS one THEN  
                        APPEND the PAIR i AND j TO land_cells  
                    END IF  
                END FOR  
            END FOR  
            IF the LENGTH OF land_cells EQUALS zero THEN  
                RETURN True  
            END IF  
            CALL dfs WITH the ARGUMENTS the ELEMENT AT position zero OF land_cells AND the ELEMENT AT position one OF that ELEMENT AND visited  
            RETURN whether the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF NOT is_connected() THEN  
            RETURN zero  
        END IF  

        FOR each i FROM zero TO the LENGTH OF grid MINUS one  
            FOR each j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF the ELEMENT AT position i OF grid AND position j OF THAT ELEMENT EQUALS one THEN  
                    SET the ELEMENT AT position i OF grid AND position j OF THAT ELEMENT TO zero  
                    IF NOT is_connected() THEN  
                        RETURN one  
                    END IF  
                    SET the ELEMENT AT position i OF grid AND position j OF THAT ELEMENT TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
