CLASS Solution  
    FUNCTION isTransformable(s AS string, t AS string) RETURNS boolean  
        SET pos TO a mapping from digit to list of positions initialized empty  
        FOR each position i AND character c IN s  
            CONVERT character c TO digit  
            APPEND position i TO list of positions FOR that digit in pos  
        END FOR  
        
        FOR each position i AND character c IN t  
            CONVERT character c TO digit  
            IF list of positions FOR digit in pos IS empty THEN  
                RETURN false  
            END IF  
            FOR each smaller digit FROM zero TO digit MINUS one  
                IF list of positions FOR smaller digit in pos IS not empty AND first element in that list IS LESS THAN first element in list of positions FOR digit in pos THEN  
                    RETURN false  
                END IF  
            END FOR  
            REMOVE first element FROM list of positions FOR digit in pos  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
