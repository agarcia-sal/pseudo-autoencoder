CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO ten raised to the power of nine PLUS seven  
        SET n TO the LENGTH OF s  
        INITIALIZE dp AS a two dimensional list with n PLUS one rows and n PLUS one columns filled with zero  
        
        FOR j FROM zero TO n  
            SET element at position j of the first row of dp TO one  
        END FOR  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            
            IF element at position i MINUS one of s EQUALS the character I THEN  
                FOR j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position j of row i MINUS one of dp MODULO MOD  
                    SET element at position j of row i of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR j FROM n MINUS i DOWN TO zero STEP minus one  
                    SET prefix_sum TO prefix_sum PLUS element at position j PLUS one of row i MINUS one of dp MODULO MOD  
                    SET element at position j of row i of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO zero  
        FOR each value IN row n of dp  
            INCREMENT result BY value  
        END FOR  
        SET result TO result MODULO MOD  
        
        RETURN result  
    END FUNCTION  
END CLASS
