CLASS Solution  
    FUNCTION ambiguousCoordinates WITH PARAMETER s  
        FUNCTION make WITH PARAMETER frag  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO the SUBSTRING of frag from position one TO position d  
                SET right TO the SUBSTRING of frag from position d PLUS one TO the END of frag  
                IF NOT the first character of left EQUALS zero OR left EQUALS zero AND NOT the last character of right EQUALS zero  
                    IF d NOT EQUALS n  
                        YIELD the CONCATENATION of left PLUS the character point PLUS right  
                    ELSE  
                        YIELD the CONCATENATION of left PLUS right  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO the SUBSTRING of s from position two TO the POSITION of the SECOND LAST character of s  
        RETURN the LIST of CONCATENATIONS of an opening parenthesis PLUS a PLUS a comma PLUS a space PLUS b PLUS a closing parenthesis FOR i FROM one TO the LENGTH OF s MINUS one FOR each a IN make with the SUBSTRING of s from position one TO position i FOR each b IN make with the SUBSTRING of s from position i PLUS one TO the END of s  
    END FUNCTION  
END CLASS
