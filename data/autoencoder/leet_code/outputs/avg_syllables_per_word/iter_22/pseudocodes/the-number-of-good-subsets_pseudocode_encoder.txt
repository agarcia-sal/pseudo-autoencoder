CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of each number in nums  
        SET dp TO a list of zeroes of length two POWER ten  
        SET element at position zero of dp TO one  
        
        FOR each num IN count  
            IF num EQUALS one  
                SET element at position zero of dp TO element at position zero of dp MULTIPLIED BY two POWER count at key one  
                SET element at position zero of dp TO element at position zero of dp MODULO ten POWER nine PLUS seven  
                CONTINUE  
            END IF  
            
            IF num NOT IN prime_factors  
                CONTINUE  
            END IF  
            
            SET factors TO prime_factors at key num  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask PLUS two POWER prime MINUS one  
            END FOR  
            
            FOR m FROM two POWER ten MINUS one DOWNTO zero  
                IF the bitwise AND of m AND mask EQUALS zero  
                    SET element at position m OR mask of dp TO (element at position m OR mask of dp PLUS element at position m of dp MULTIPLIED BY count at key num) MODULO ten POWER nine PLUS seven  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of elements from position one to the end of dp MODULO ten POWER nine PLUS seven  
        RETURN result  
    END FUNCTION  
END CLASS
