CLASS Solution  
    FUNCTION shortestPathLength(graph)  
        IF the LENGTH OF graph EQUALS zero  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF graph  
        
        SET queue TO a new deque containing pairs of each node AND the value of one shifted left by the node for node FROM zero TO n MINUS one  
        SET visited TO a new set containing pairs of each node AND the value of one shifted left by the node for node FROM zero TO n MINUS one  
        
        SET steps TO zero  
        SET all_nodes_visited TO the value of one shifted left by n MINUS one  
        
        WHILE queue is not empty  
            SET current_level_size TO the LENGTH OF queue  
            
            FOR each index FROM zero TO current_level_size MINUS one  
                REMOVE the first element FROM queue AND DESTRUCTURE into current_node AND visited_nodes  
                
                IF visited_nodes EQUALS all_nodes_visited  
                    RETURN steps  
                END IF  
                
                FOR each neighbor IN the collection at position current_node of graph  
                    SET next_visited_nodes TO visited_nodes OR the value of one shifted left by neighbor  
                    SET next_state TO the pair of neighbor AND next_visited_nodes  
                    
                    IF next_state NOT IN visited  
                        ADD next_state TO visited  
                        ADD next_state TO queue  
                    END IF  
                END FOR  
            END FOR  
            
            INCREMENT steps BY one  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
