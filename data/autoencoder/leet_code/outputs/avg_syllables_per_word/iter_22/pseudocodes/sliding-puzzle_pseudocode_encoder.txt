CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO a tuple of tuples each constructed from each row in board  
        SET target TO a tuple of tuples representing a two by three matrix with elements one two three in the first row and four five zero in the second row  
        SET neighbors TO a mapping of zero position indices to their respective sets of adjacent indices as follows  
            zero mapped to the set containing one and three  
            one mapped to the set containing zero two and four  
            two mapped to the set containing one and five  
            three mapped to the set containing zero and four  
            four mapped to the set containing one three and five  
            five mapped to the set containing two and four  
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position j of element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a deque initialized with a tuple containing start zero_pos and zero  
        SET visited TO a set containing start  
        WHILE the LENGTH OF queue is greater than zero  
            REMOVE and ASSIGN the leftmost element of queue to state zero_pos moves  
            IF state EQUALS target  
                RETURN moves  
            END IF  
            FOR each neighbor IN the set mapped from zero_pos in neighbors  
                SET new_state TO a list of lists each converted from the tuples in state  
                SET i TO the integer division of zero_pos by three  
                SET j TO zero_pos MODULO three  
                SET ni TO the integer division of neighbor by three  
                SET nj TO neighbor MODULO three  
                SWAP the elements at positions j and i with elements at positions nj and ni in new_state  
                SET new_tuple_state TO a tuple of tuples each constructed from each row in new_state  
                IF new_tuple_state NOT IN visited  
                    ADD new_tuple_state TO visited  
                    APPEND a tuple containing new_tuple_state neighbor and moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
