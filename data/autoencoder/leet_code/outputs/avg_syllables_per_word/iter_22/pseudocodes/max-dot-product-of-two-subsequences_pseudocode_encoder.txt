CLASS Solution  
    FUNCTION maxDotProduct(nums1 nums2)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        SET dp TO a two dimensional list with (m PLUS one) rows and (n PLUS one) columns EACH FILLED WITH the smallest possible number  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO the element at position (i MINUS one) of nums1 MULTIPLIED BY the element at position (j MINUS one) of nums2  
                SET dp element at position i and position j TO the maximum of the following values:  
                    current_product  
                    the element at position (i MINUS one) and position j of dp  
                    the element at position i and position (j MINUS one) of dp  
                    the element at position (i MINUS one) and position (j MINUS one) of dp PLUS current_product  
            END FOR  
        END FOR  
        RETURN the element at position m and position n of dp  
    END FUNCTION  
END CLASS
