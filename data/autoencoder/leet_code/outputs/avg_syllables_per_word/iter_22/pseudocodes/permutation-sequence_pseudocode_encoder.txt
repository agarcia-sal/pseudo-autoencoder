CLASS Solution  
    FUNCTION getPermutation(n k)  
        SET numbers TO list of numbers from one TO n inclusive  
        DECREMENT k BY one  
        SET result TO empty list  
        FOR i FROM n DOWN TO one  
            SET fact TO factorial of i MINUS one  
            SET index TO k DIVIDED BY fact  
            APPEND string representation of element at position index of numbers TO result  
            REMOVE element at position index FROM numbers  
            SET k TO k MODULO fact  
        END FOR  
        SET joined_result TO concatenation of all elements in result as a single string  
        RETURN joined_result  
    END FUNCTION  
END CLASS
