CLASS Solution  
    FUNCTION movesToStamp(stamp target)  
        SET m TO the LENGTH OF stamp  
        SET n TO the LENGTH OF target  
        SET indeg TO a list containing m repeated the quantity of n MINUS m PLUS one times  
        SET q TO an empty double ended queue  
        SET g TO a list containing empty lists repeated the quantity of n times  
        FOR i FROM zero TO n MINUS m  
            FOR j FROM zero TO the LENGTH OF stamp MINUS one  
                SET c TO the element at position j of stamp  
                IF the element at position i PLUS j of target EQUALS c  
                    DECREMENT the element at position i of indeg BY one  
                    IF the element at position i of indeg EQUALS zero  
                        APPEND i TO q  
                    END IF  
                ELSE  
                    APPEND i TO the element at position i PLUS j of g  
                END IF  
            END FOR  
        END FOR  
        SET ans TO an empty list  
        SET vis TO a list containing False repeated the quantity of n times  
        WHILE q is not empty  
            SET i TO the first element REMOVED FROM q  
            APPEND i TO ans  
            FOR j FROM zero TO m MINUS one  
                IF the element at position i PLUS j of vis EQUALS False  
                    SET the element at position i PLUS j of vis TO True  
                    FOR each k IN the element at position i PLUS j of g  
                        DECREMENT the element at position k of indeg BY one  
                        IF the element at position k of indeg EQUALS zero  
                            APPEND k TO q  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  
        IF all elements of vis are True  
            RETURN ans in reverse order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
