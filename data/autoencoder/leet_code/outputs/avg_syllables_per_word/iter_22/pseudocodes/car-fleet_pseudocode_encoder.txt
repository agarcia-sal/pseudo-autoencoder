CLASS Solution  
    FUNCTION carFleet(target position speed)  
        SET paired_position_speed TO list of pairs of element at position index of position and element at position index of speed FOR each index FROM zero TO the LENGTH OF position MINUS one  
        SET sorted_pairs TO paired_position_speed SORTED in ORDER from greatest position element TO smallest position element  
        SET time_to_target TO empty list  
        FOR each pair IN sorted_pairs  
            SET position_element TO the first element of pair  
            SET speed_element TO the second element of pair  
            SET time_value TO target MINUS position_element DIVIDED BY speed_element  
            APPEND time_value TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time_element IN time_to_target  
            IF time_element GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time_element  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
