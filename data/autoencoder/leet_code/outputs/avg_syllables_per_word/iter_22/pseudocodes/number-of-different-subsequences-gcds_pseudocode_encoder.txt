CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET max_num TO the maximum value among elements of nums  
        SET num_set TO the set of elements of nums  
        SET count TO zero  
        
        FOR each g FROM one TO max_num INCLUSIVE  
            SET gcd_val TO zero  
            FOR each m FROM g TO max_num INCLUSIVE STEP g  
                IF element at position m of num_set IS PRESENT  
                    SET gcd_val TO the greatest common divisor of gcd_val AND m  
                    IF gcd_val EQUALS g  
                        INCREMENT count BY one  
                        BREAK the inner loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
