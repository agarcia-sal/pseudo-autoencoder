CLASS Solution  
    FUNCTION isScramble(s1, s2)  
        IF s1 EQUALS s2 THEN  
            RETURN True  
        END IF  
        IF sorted characters of s1 NOT EQUALS sorted characters of s2 THEN  
            RETURN False  
        END IF  
        SET n TO the LENGTH OF s1  
        FOR i FROM one TO n MINUS one  
            IF isScramble of substring from position one TO position i of s1 AND isScramble of substring from position one TO position i of s2 THEN  
                RETURN True  
            END IF  
            IF isScramble of substring from position one TO position i of s1 AND isScramble of substring from position n MINUS i PLUS one TO position n of s2 THEN  
                RETURN True  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
