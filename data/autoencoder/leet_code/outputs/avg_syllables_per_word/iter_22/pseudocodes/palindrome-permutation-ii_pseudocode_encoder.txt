CLASS Solution  
    FUNCTION generatePalindromes WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER t  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c and count v IN the ITEMS OF cnt  
                IF v GREATER THAN one  
                    DECREMENT cnt AT c BY two  
                    CALL dfs WITH PARAMETER the concatenation of c PLUS t PLUS c  
                    INCREMENT cnt AT c BY two  
                END IF  
            END FOR  
        END FUNCTION  
        
        ASSIGN cnt TO the COUNTER OF s  
        SET mid TO empty string  
        FOR each character c and count v IN the ITEMS OF cnt  
            IF v MODULO two NOT EQUALS zero  
                IF mid NOT EQUALS empty string  
                    RETURN empty list  
                END IF  
                SET mid TO c  
                DECREMENT cnt AT c BY one  
            END IF  
        END FOR  
        SET ans TO empty list  
        CALL dfs WITH PARAMETER mid  
        RETURN ans  
    END FUNCTION  
END CLASS
