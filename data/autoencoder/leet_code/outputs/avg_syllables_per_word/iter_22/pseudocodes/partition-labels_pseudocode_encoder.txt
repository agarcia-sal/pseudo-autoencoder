CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO an empty dictionary  
        FOR each index AND character IN the enumeration of s  
            SET the character in last_occurrence TO the index  
        END FOR  
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO an empty list  
        FOR each index AND character IN the enumeration of s  
            SET end TO the greater of end AND the value of character in last_occurrence  
            IF index EQUALS end  
                SET partition_size TO end MINUS start PLUS one  
                APPEND partition_size TO partition_sizes  
                SET start TO index PLUS one  
            END IF  
        END FOR  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
