CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each element l IN left  
                FOR each element r IN right  
                    IF operator EQUALS the plus symbol  
                        APPEND the sum of l PLUS r TO results  
                    ELSE IF operator EQUALS the minus symbol  
                        APPEND the difference of l MINUS r TO results  
                    ELSE IF operator EQUALS the multiplication symbol  
                        APPEND the product of l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  

        FUNCTION helper(sub_expr)  
            IF sub_expr consists only of digit characters  
                RETURN a list containing the integer value of sub_expr  
            END IF  
            SET results TO empty list  
            FOR index i FROM zero TO the length of sub_expr MINUS one  
                IF the element at position i of sub_expr EQUALS one of the plus symbol OR the minus symbol OR the multiplication symbol  
                    SET left_results TO the result of calling helper with the substring of sub_expr from position zero TO position i MINUS one  
                    SET right_results TO the result of calling helper with the substring of sub_expr from position i PLUS one TO the end  
                    EXTEND results BY the result of calling compute with left_results right_results and the element at position i of sub_expr  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  

        RETURN the result of calling helper with expression  
    END FUNCTION  
END CLASS
