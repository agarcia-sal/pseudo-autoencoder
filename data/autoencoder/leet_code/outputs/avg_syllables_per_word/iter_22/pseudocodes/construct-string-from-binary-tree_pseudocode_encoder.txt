CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    ASSIGN reference root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    CREATE an empty queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        REMOVE the first element FROM queue AND ASSIGN reference TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str(self, root)  
        FUNCTION helper(node)  
            IF node EQUALS None  
                RETURN an empty string  
            END IF  
            SET result TO the string representation of the value of node  
            IF the left child of node IS NOT None OR the right child of node IS NOT None  
                SET result TO result PLUS an opening parenthesis PLUS helper(the left child of node) PLUS a closing parenthesis  
            END IF  
            IF the right child of node IS NOT None  
                SET result TO result PLUS an opening parenthesis PLUS helper(the right child of node) PLUS a closing parenthesis  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN helper(root)  
    END FUNCTION  
END CLASS
