CLASS Solution  
    FUNCTION killProcess(pid parameter as list of integers ppid parameter as list of integers kill parameter as integer) RETURNS list of integers  
        SET tree TO a default dictionary with default value as empty list  
        FOR each pair of child and parent IN parallel iteration over pid and ppid  
            APPEND child TO the list at parent key in tree  
        END FOR  
        SET queue TO list containing kill  
        SET killed_processes TO empty list  
        WHILE the LENGTH OF queue is greater than zero  
            REMOVE the element at position zero from queue and ASSIGN it TO current  
            APPEND current TO killed_processes  
            EXTEND queue BY elements in the list at key current in tree  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
