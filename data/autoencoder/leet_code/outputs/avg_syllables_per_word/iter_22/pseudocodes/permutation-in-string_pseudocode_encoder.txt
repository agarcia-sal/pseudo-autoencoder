CLASS Solution  
    FUNCTION checkInclusion(s1 parameter, s2 parameter) RETURNS boolean  
        SET length of s1 TO length_s1  
        SET length of s2 TO length_s2  
        
        IF length_s1 GREATER THAN length_s2  
            RETURN false  
        END IF  
        
        SET frequency count of characters in s1 TO count_s1  
        SET frequency count of characters in substring from position zero TO position length_s1 MINUS one of s2 TO count_s2  
        
        IF count_s1 EQUALS count_s2  
            RETURN true  
        END IF  
        
        FOR variable i FROM length_s1 TO length_s2 MINUS one  
            INCREMENT count of element at position i of s2 IN count_s2 BY one  
            DECREMENT count of element at position i MINUS length_s1 of s2 IN count_s2 BY one  
            
            IF count of element at position i MINUS length_s1 of s2 IN count_s2 EQUALS zero  
                REMOVE element at position i MINUS length_s1 of s2 FROM count_s2  
            END IF  
            
            IF count_s1 EQUALS count_s2  
                RETURN true  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
