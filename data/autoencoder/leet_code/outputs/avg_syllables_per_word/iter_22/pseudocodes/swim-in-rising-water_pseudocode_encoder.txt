CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        CREATE visited AS an empty set  
        CREATE min_heap AS a new list containing a single tuple of the element at position zero zero of grid AND zero AND zero  
        ADD the tuple of zero zero TO visited  
        SET directions TO a new list containing the tuples zero one one zero zero MINUS one zero MINUS one zero  
        WHILE min_heap is not empty  
            REMOVE and RETURN the tuple with the smallest first element from min_heap ASSIGNING it to t i j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            FOR each element di dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND the tuple ni nj NOT IN visited  
                    ADD the tuple ni nj TO visited  
                    ADD to min_heap the tuple of the greater value between t AND the element at position ni nj of grid AND ni AND nj  
                END IF  
            END FOR  
        END WHILE  
        RETURN the negative integer one  
    END FUNCTION  
END CLASS
