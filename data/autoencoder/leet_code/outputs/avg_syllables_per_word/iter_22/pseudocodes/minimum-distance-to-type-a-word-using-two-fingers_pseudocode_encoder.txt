CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS None  
                RETURN zero  
            END IF  
            SET x1 TO integer division of the difference between the character code of c1 and the character code of capital letter A BY six  
            SET y1 TO remainder of the difference between the character code of c1 and the character code of capital letter A BY six  
            SET x2 TO integer division of the difference between the character code of c2 and the character code of capital letter A BY six  
            SET y2 TO remainder of the difference between the character code of c2 and the character code of capital letter A BY six  
            RETURN the absolute value of x1 MINUS x2 PLUS the absolute value of y1 MINUS y2  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the length of word  
                RETURN zero  
            END IF  
            SET c TO element at position i of word  
            SET d1 TO the result of distance with parameters f1 and c PLUS the result of dp with parameters i PLUS one c and f2  
            SET d2 TO the result of distance with parameters f2 and c PLUS the result of dp with parameters i PLUS one f1 and c  
            RETURN the minimum of d1 and d2  
        END FUNCTION  
        
        RETURN the result of dp with parameters zero None and None  
    END FUNCTION  
END CLASS
