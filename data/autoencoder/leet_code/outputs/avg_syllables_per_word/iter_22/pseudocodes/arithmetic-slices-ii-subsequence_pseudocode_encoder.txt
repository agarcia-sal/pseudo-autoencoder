CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        
        SET dp TO a list containing for each index from zero TO n MINUS one a new mapping from differences to counts initialized with zero  
        SET total_count TO zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                SET d TO the element at position i of nums MINUS the element at position j of nums  
                
                IF the difference d IS a key in the mapping at position j in dp  
                    INCREMENT the value associated with key d in the mapping at position i in dp BY the value associated with key d in the mapping at position j in dp PLUS one  
                    INCREMENT total_count BY the value associated with key d in the mapping at position j in dp  
                ELSE  
                    INCREMENT the value associated with key d in the mapping at position i in dp BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
