CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET dp TO a TWO DIMENSIONAL collection of zeros with dimensions m PLUS one BY n PLUS one  
        SET max_side TO zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of matrix AND element at position j MINUS one of that element EQUALS the character one  
                    SET dp element at position i AND j TO the MINIMUM AMONG dp element at position i MINUS one AND j AND dp element at position i AND j MINUS one AND dp element at position i MINUS one AND j MINUS one PLUS one  
                    SET max_side TO the MAXIMUM BETWEEN max_side AND dp element at position i AND j  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
