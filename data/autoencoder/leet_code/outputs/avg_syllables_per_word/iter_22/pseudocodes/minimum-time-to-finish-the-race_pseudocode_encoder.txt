CLASS Solution  
    FUNCTION minimumFinishTime WITH parameters tires AND changeTime AND numLaps  
        SET min_times TO a list of positive infinity repeated fifteen times  
        FOR each pair of first_factor AND reduction_factor IN tires  
            SET total_time TO zero  
            FOR index FROM zero TO fourteen  
                SET lap_time TO first_factor MULTIPLIED BY reduction_factor RAISED TO THE POWER OF index  
                IF lap_time GREATER THAN changeTime PLUS first_factor  
                    BREAK the loop  
                END IF  
                INCREMENT total_time BY lap_time  
                IF total_time LESS THAN element at position index of min_times  
                    SET element at position index of min_times TO total_time  
                END IF  
            END FOR  
        END FOR  

        SET dp TO a list of positive infinity repeated numLaps PLUS one times  
        SET element at position zero of dp TO zero  

        FOR lap_count FROM one TO numLaps  
            FOR segment_length FROM zero TO the smaller of lap_count MINUS one AND fourteen  
                SET candidate_time TO element at position lap_count MINUS segment_length MINUS one of dp PLUS element at position segment_length of min_times PLUS changeTime  
                IF candidate_time LESS THAN element at position lap_count of dp  
                    SET element at position lap_count of dp TO candidate_time  
                END IF  
            END FOR  
        END FOR  

        RETURN element at position numLaps of dp MINUS changeTime  
    END FUNCTION  
END CLASS
