CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO a list of boolean values false repeated the length of n PLUS one times  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE j MULTIPLIED BY j LESS THAN OR EQUAL TO i  
                IF dp at position i MINUS j MULTIPLIED BY j EQUALS false  
                    SET dp at position i TO true  
                    BREAK out of the WHILE loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN dp at position n  
    END FUNCTION  
END CLASS
