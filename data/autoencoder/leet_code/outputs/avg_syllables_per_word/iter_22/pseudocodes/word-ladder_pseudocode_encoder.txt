CLASS Solution  
    FUNCTION ladderLength(beginWord endWord wordList)  
        IF endWord NOT IN wordList  
            RETURN zero  
        END IF  

        SET word_dict TO a dictionary with default list values  
        SET word_length TO the LENGTH OF beginWord  

        FOR each word IN wordList  
            FOR index FROM zero TO word_length MINUS one  
                SET intermediate_word TO the concatenation of substring from position zero TO position index MINUS one of word PLUS the character star PLUS substring from position index PLUS one TO the end of word  
                APPEND word TO the list stored at key intermediate_word IN word_dict  
            END FOR  
        END FOR  

        SET queue TO a double ended queue initialized with the tuple of beginWord and one  
        SET visited TO a set initialized with beginWord  

        WHILE queue is not empty  
            REMOVE and RETURN the first tuple from queue AS current_word and current_length  

            FOR index FROM zero TO word_length MINUS one  
                SET intermediate_word TO the concatenation of substring from position zero TO position index MINUS one of current_word PLUS the character star PLUS substring from position index PLUS one TO the end of current_word  

                FOR each word IN the list stored at key intermediate_word IN word_dict  
                    IF word EQUALS endWord  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited  
                        ADD word TO visited  
                        APPEND the tuple of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN zero  
    END FUNCTION  
END CLASS
