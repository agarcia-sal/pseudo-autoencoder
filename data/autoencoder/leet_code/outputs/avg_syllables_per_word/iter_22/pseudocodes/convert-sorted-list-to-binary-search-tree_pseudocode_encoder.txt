CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS None  
        SET the member val TO val  
        SET the member next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the values is empty  
        RETURN None  
    END IF  
    SET head TO a new ListNode instantiated with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two to the end  
        SET node TO a new ListNode instantiated with val  
        SET the member next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the logical expression that the member val of p1 EQUALS the member val of p2 AND the result of calling is_same_list with parameters the member next of p1 AND the member next of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND left EQUALS None AND right EQUALS None  
        SET the member val TO val  
        SET the member left TO left  
        SET the member right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the values is empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode instantiated with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN OR EQUAL TO the length of values AND the element at position i of values NOT EQUALS None  
            SET the member left of node TO a new TreeNode instantiated with the element at position i of values  
            APPEND the member left of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN OR EQUAL TO the length of values AND the element at position i of values NOT EQUALS None  
            SET the member right of node TO a new TreeNode instantiated with the element at position i of values  
            APPEND the member right of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the member val of p NOT EQUALS the member val of q  
        RETURN False  
    ELSE  
        RETURN the logical expression that the result of calling is_same_tree with parameters the member left of p AND the member left of q AND the result of calling is_same_tree with parameters the member right of p AND the member right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST WITH PARAMETER head  
        IF head EQUALS None  
            RETURN None  
        END IF  

        FUNCTION findMiddle WITH PARAMETERS left AND right  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUAL TO right AND the member next of fast NOT EQUAL TO right  
                SET slow TO the member next of slow  
                SET fast TO the member next of the member next of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  

        FUNCTION convertListToBST WITH PARAMETERS left AND right  
            IF left EQUALS right  
                RETURN None  
            END IF  
            SET mid TO the result of calling findMiddle with parameters left AND right  
            SET node TO a new TreeNode instantiated with the member val of mid  
            SET the member left of node TO the result of calling convertListToBST with parameters left AND mid  
            SET the member right of node TO the result of calling convertListToBST with parameters the member next of mid AND right  
            RETURN node  
        END FUNCTION  

        RETURN the result of calling convertListToBST with parameters head AND None  
    END FUNCTION  
END CLASS
