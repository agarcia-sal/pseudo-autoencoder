CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO list of pairs where each pair is the first element of each interval PLUS the position of that interval IN intervals  
        SORT start_times BY the first element of each pair IN ascending order  
        SET result TO empty list  
        FOR each pair IGNORE the first element AND keep second element AS end IN intervals  
            SET idx TO position found USING binary search to locate the first pair IN start_times WHERE its first element IS GREATER THAN OR EQUAL TO end  
            IF idx IS LESS THAN the LENGTH OF start_times  
                APPEND the second element of the pair AT position idx IN start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
