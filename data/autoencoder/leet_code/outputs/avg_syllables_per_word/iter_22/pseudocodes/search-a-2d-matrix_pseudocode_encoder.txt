CLASS Solution  
    FUNCTION searchMatrix(matrix target)  
        IF matrix EQUALS nothing OR element at position zero of matrix EQUALS nothing THEN  
            RETURN false  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
            SET mid_value TO element at position the INTEGER DIVISION OF mid BY n of matrix at position the REMAINDER OF mid DIVIDED BY n  
            IF mid_value EQUALS target THEN  
                RETURN true  
            ELSE IF mid_value LESS THAN target THEN  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN false  
    END FUNCTION  
END CLASS
