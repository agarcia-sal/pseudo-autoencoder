CLASS Solution  
    FUNCTION merge(intervals)  
        CALL sort intervals BY the element at position zero of each interval  
        SET merged TO an empty list  
        FOR each interval IN intervals  
            IF the LENGTH OF merged EQUALS zero OR the element at position one of the last interval in merged IS LESS THAN the element at position zero of interval  
                APPEND interval TO merged  
            ELSE  
                SET the element at position one of the last interval in merged TO the greater VALUE BETWEEN the element at position one of the last interval in merged AND the element at position one of interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
