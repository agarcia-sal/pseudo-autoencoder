CLASS Solution  
    FUNCTION maxSubArrayLen(nums k)  
        SET cumulative_sum_index TO dictionary with key zero associated with value negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        
        FOR each index i AND element num IN enumeration of nums  
            SET cumulative_sum TO cumulative_sum PLUS num  
            
            IF cumulative_sum MINUS k EXISTS AS key IN cumulative_sum_index  
                SET max_length TO the greater value BETWEEN max_length AND i MINUS the value associated with key cumulative_sum MINUS k IN cumulative_sum_index  
            END IF  
            
            IF cumulative_sum DOES NOT EXIST AS key IN cumulative_sum_index  
                ASSIGN cumulative_sum_index at key cumulative_sum TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
