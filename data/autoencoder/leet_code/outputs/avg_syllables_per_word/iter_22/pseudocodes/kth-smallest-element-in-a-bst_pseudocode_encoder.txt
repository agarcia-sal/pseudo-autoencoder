CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the list values EQUALS empty  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode with value at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the length of values AND the element at position i of values IS NOT nothing  
            SET the left attribute of node TO a new TreeNode with value at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values IS NOT nothing  
            SET the right attribute of node TO a new TreeNode with value at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p IS nothing AND q IS nothing  
        RETURN true  
    ELSE IF p IS nothing OR q IS nothing  
        RETURN false  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with parameters the left attribute of p and the left attribute of q AND the result of is_same_tree with parameters the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest with parameters root k  
        SET stack TO an empty list  
        SET current TO root  
        SET count TO zero  
        WHILE the list stack is not empty OR current IS NOT nothing  
            WHILE current IS NOT nothing  
                APPEND current TO stack  
                SET current TO the left attribute of current  
            END WHILE  
            SET current TO the last element removed from stack  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN the value attribute of current  
            END IF  
            SET current TO the right attribute of current  
        END WHILE  
    END FUNCTION  
END CLASS
