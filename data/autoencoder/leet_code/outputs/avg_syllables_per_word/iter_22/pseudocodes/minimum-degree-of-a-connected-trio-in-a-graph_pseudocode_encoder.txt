CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        SET graph TO empty mapping from integer TO empty set  
        FOR i FROM one TO n  
            ASSIGN reference TO empty set for element at position i of graph  
        END FOR  
        FOR each pair u v IN edges  
            ADD v TO set associated with u in graph  
            ADD u TO set associated with v in graph  
        END FOR  
        SET min_degree TO positive infinity  
        SET trio_found TO false  
        FOR i FROM one TO n  
            FOR j FROM i PLUS one TO n  
                FOR k FROM j PLUS one TO n  
                    IF element j EXISTS IN set associated with i in graph AND element k EXISTS IN set associated with i in graph AND element k EXISTS IN set associated with j in graph  
                        SET trio_found TO true  
                        SET trio_degree TO the LENGTH OF set associated with i in graph PLUS the LENGTH OF set associated with j in graph PLUS the LENGTH OF set associated with k in graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF trio_found EQUALS true  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
