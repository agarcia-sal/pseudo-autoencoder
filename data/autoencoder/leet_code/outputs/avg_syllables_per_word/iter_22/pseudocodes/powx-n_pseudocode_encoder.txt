CLASS Solution  
    FUNCTION myPow(x FLOAT, n INTEGER) RETURNS FLOAT  
        FUNCTION qpow(a FLOAT, n INTEGER) RETURNS FLOAT  
            SET ans TO one  
            WHILE n NOT EQUALS zero  
                IF the LEAST SIGNIFICANT BIT of n EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two DISCARDING REMAINDER  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n GREATER THAN OR EQUAL TO zero  
            RETURN qpow WITH a REPLACED BY x AND n REPLACED BY n  
        ELSE  
            RETURN one DIVIDED BY qpow WITH a REPLACED BY x AND n REPLACED BY the NEGATION OF n  
        END IF  
    END FUNCTION  
END CLASS
