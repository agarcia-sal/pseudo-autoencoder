CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO a Counter initialized with nums  
        SET end_with TO an empty Counter  
        
        FOR each num IN nums  
            IF the count of num in num_count EQUALS zero  
                CONTINUE to the next iteration  
            END IF  
            
            IF the count of the element that is num MINUS one in end_with GREATER THAN zero  
                DECREMENT the count of the element that is num MINUS one in end_with BY one  
                INCREMENT the count of num in end_with BY one  
            ELSE IF the count of the element that is num PLUS one in num_count GREATER THAN zero AND the count of the element that is num PLUS two in num_count GREATER THAN zero  
                DECREMENT the count of the element that is num PLUS one in num_count BY one  
                DECREMENT the count of the element that is num PLUS two in num_count BY one  
                INCREMENT the count of the element that is num PLUS two in end_with BY one  
            ELSE  
                RETURN False  
            END IF  
            
            DECREMENT the count of num in num_count BY one  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
