CLASS Solution  
    FUNCTION removeBoxes WITH PARAMETER boxes  
        FUNCTION dp WITH PARAMETERS l r k memo  
            IF l GREATER THAN r  
                RETURN zero  
            END IF  
            IF the TUPLE of l r k EXISTS IN memo  
                RETURN the ELEMENT CORRESPONDING TO the TUPLE of l r k IN memo  
            END IF  
            WHILE r GREATER THAN l AND element at position r of boxes EQUALS element at position r MINUS one of boxes  
                DECREMENT r BY one  
                INCREMENT k BY one  
            END WHILE  
            SET res TO dp WITH PARAMETERS l r MINUS one zero memo PLUS the PRODUCT of the SUM of k PLUS one AND the SUM of k PLUS one  
            FOR i FROM l TO r MINUS one  
                IF element at position i of boxes EQUALS element at position r of boxes  
                    SET res TO the GREATER OF res AND the SUM of dp WITH PARAMETERS l i k PLUS one memo AND dp WITH PARAMETERS i PLUS one r MINUS one zero memo  
                END IF  
            END FOR  
            SET the ELEMENT CORRESPONDING TO the TUPLE of l r k IN memo TO res  
            RETURN res  
        END FUNCTION  
        RETURN dp WITH PARAMETERS zero the LENGTH OF boxes MINUS one zero an EMPTY DICTIONARY  
    END FUNCTION  
END CLASS
