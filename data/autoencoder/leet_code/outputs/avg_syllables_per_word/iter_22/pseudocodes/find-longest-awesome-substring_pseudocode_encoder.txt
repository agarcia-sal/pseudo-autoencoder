CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a mapping with the key zero associated with the value negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each index i AND character char IN the enumeration of s  
            SET digit TO the integer value of character char  
            SET prefix_xor TO prefix_xor XORED WITH one shifted LEFT BY digit positions  
            
            IF prefix_xor EXISTS AS a key IN prefix_map  
                SET max_length TO the greater OF max_length AND the difference BETWEEN i AND the value associated with prefix_xor in prefix_map  
            END IF  
            
            FOR each integer j FROM zero TO nine  
                SET flipped TO prefix_xor XORED WITH one shifted LEFT BY j positions  
                IF flipped EXISTS AS a key IN prefix_map  
                    SET max_length TO the greater OF max_length AND the difference BETWEEN i AND the value associated with flipped in prefix_map  
                END IF  
            END FOR  
            
            IF prefix_xor DOES NOT EXIST AS a key IN prefix_map  
                ASSIGN prefix_map at key prefix_xor TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
