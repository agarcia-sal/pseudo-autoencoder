CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO the LENGTH OF floor  
        SET dp TO a list of lists with the dimension n PLUS one BY numCarpets PLUS one with all elements initialized to zero  
        
        FOR i FROM one TO n  
            SET dp element at position i and position zero TO dp element at position i MINUS one and position zero PLUS the condition that element at position i MINUS one of floor EQUALS the character one  
        END FOR  
        
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp element at position i and position j TO dp element at position i MINUS one and position j PLUS the condition that element at position i MINUS one of floor EQUALS the character one  
                SET dp element at position i and position j TO the MINIMUM of dp element at position i and position j AND dp element at position MAXIMUM OF zero AND i MINUS carpetLen and position j MINUS one  
            END FOR  
        END FOR  
        
        RETURN dp element at position n and position numCarpets  
    END FUNCTION  
END CLASS
