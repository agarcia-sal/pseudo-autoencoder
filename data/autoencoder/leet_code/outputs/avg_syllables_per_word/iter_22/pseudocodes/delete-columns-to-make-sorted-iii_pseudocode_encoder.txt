CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET total_rows TO the LENGTH OF strs  
        SET total_columns TO the LENGTH OF element at position zero of strs  
        SET dp TO a list of total_columns elements each with the value one  
        FOR index i FROM one TO total_columns MINUS one  
            FOR index j FROM zero TO i MINUS one  
                IF for every index k FROM zero TO total_rows MINUS one the element at position j of element at position k of strs IS LESS THAN OR EQUAL TO the element at position i of element at position k of strs  
                    SET dp at position i TO the GREATER VALUE BETWEEN the current value of dp at position i AND the value of dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        SET longest_non_decreasing_subsequence_length TO the GREATEST VALUE IN dp  
        RETURN total_columns MINUS longest_non_decreasing_subsequence_length  
    END FUNCTION  
END CLASS
