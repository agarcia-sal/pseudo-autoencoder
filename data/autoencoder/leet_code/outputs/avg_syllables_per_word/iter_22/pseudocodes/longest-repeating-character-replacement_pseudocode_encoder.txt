CLASS Solution  
    FUNCTION characterReplacement(s k)  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        ASSIGN count TO a new mapping with default integer zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the count AT element at position right of s BY one  
            SET max_count TO the maximum value BETWEEN max_count AND the count AT element at position right of s  
            
            IF the quantity of right MINUS left PLUS one MINUS max_count IS GREATER THAN k THEN  
                DECREMENT the count AT element at position left of s BY one  
                INCREMENT left BY one  
            END IF  
            
            SET max_length TO the maximum value BETWEEN max_length AND the quantity of right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
