CLASS Solution  
    FUNCTION maximumCost(n, highways, k)  
        ASSIGN reference TO a new empty mapping from city to list of pairs of city and toll named graph  
        FOR each element consisting of city1 city2 toll IN highways  
            APPEND pair of city2 and toll TO element at position city1 of graph  
            APPEND pair of city1 and toll TO element at position city2 of graph  
        END FOR  
        
        SET max_cost TO negative one  
        
        FUNCTION dfs(city, visited, cost, highways_used)  
            IF highways_used EQUALS k THEN  
                SET max_cost TO the greater value BETWEEN max_cost AND cost  
                RETURN  
            END IF  
            
            FOR each pair consisting of neighbor and toll IN element at position city of graph  
                IF neighbor NOT IN visited THEN  
                    ADD neighbor TO visited  
                    CALL dfs WITH neighbor visited cost PLUS toll highways_used PLUS one  
                    REMOVE neighbor FROM visited  
                END IF  
            END FOR  
        END FUNCTION  
        
        FOR each start_city FROM zero TO n MINUS one  
            ASSIGN new set containing start_city TO visited  
            CALL dfs WITH start_city visited zero zero  
        END FOR  
        
        RETURN max_cost  
    END FUNCTION  
END CLASS
