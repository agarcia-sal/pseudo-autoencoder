CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO a two dimensional list of boolean value False with number of rows equal to the LENGTH OF s PLUS one and number of columns equal to the LENGTH OF p PLUS one  
        SET element at position zero zero of dp TO True  
        FOR j FROM one TO the LENGTH OF p  
            IF element at position j MINUS one of string p EQUALS the character star  
                SET element at position zero j of dp TO element at position zero j MINUS one of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM one TO the LENGTH OF p  
                IF element at position j MINUS one of string p EQUALS the character star  
                    SET element at position i j of dp TO element at position i MINUS one j of dp OR element at position i j MINUS one of dp  
                ELSE IF element at position j MINUS one of string p EQUALS the character question mark OR element at position i MINUS one of string s EQUALS element at position j MINUS one of string p  
                    SET element at position i j of dp TO element at position i MINUS one j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position LENGTH OF s LENGTH OF p of dp  
    END FUNCTION  
END CLASS
