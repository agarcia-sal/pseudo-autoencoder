CLASS Solution  
    FUNCTION shortestSubarray(nums LIST OF INTEGER k INTEGER) RETURNS INTEGER  
        SET prefix_sum TO list containing zero as the first element  
        FOR each num IN nums  
            APPEND element at position LENGTH OF prefix_sum MINUS one of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        
        ASSIGN reference TO an empty double-ended queue dq  
        SET min_length TO positive infinity  
        
        FOR each index i AND current_sum IN enumeration of prefix_sum  
            WHILE dq is not empty AND current_sum MINUS element at position element at position zero of dq of prefix_sum GREATER THAN OR EQUAL TO k  
                SET min_length TO the smaller of min_length AND i MINUS the element removed from the left end of dq  
            END WHILE  
            
            WHILE dq is not empty AND current_sum LESS THAN OR EQUAL TO element at position element at the right end of dq of prefix_sum  
                REMOVE element from the right end of dq  
            END WHILE  
            
            APPEND i TO dq  
        END FOR  
        
        IF min_length NOT EQUALS positive infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
