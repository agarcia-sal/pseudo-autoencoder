CLASS Solution  
    FUNCTION findSubstring(s parameter s OF TYPE string, words parameter words OF TYPE list of strings) RETURNS list of integers  
        IF s EQUALS empty string OR words EQUALS empty list OR element at position zero of words EQUALS empty string  
            RETURN empty list  
        END IF  
        SET word_length TO the LENGTH OF element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the frequency COUNTER of words  
        SET result TO empty list  
        FOR index i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO empty frequency COUNTER  
            FOR index j FROM i TO i PLUS total_length MINUS one IN STEPS OF word_length  
                SET word TO the substring of s FROM position j TO position j PLUS word_length MINUS one  
                INCREMENT the count of word in seen BY one  
                IF the count of word in seen GREATER THAN the count of word in word_count  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF the inner loop completed without break  
                APPEND i TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
