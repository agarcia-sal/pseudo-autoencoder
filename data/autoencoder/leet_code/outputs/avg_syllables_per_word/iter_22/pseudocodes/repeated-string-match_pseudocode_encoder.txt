CLASS Solution  
    FUNCTION repeatedStringMatch(a string_parameter b string_parameter) RETURNS integer  
        SET minimum_number_of_repetitions TO the ceiling value of the length of b DIVIDED BY the length of a  
        IF the string b IS A SUBSTRING OF the string a REPEATED minimum_number_of_repetitions TIMES  
            RETURN minimum_number_of_repetitions  
        END IF  
        IF the string b IS A SUBSTRING OF the string a REPEATED the sum of minimum_number_of_repetitions PLUS one TIMES  
            RETURN the sum of minimum_number_of_repetitions PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
