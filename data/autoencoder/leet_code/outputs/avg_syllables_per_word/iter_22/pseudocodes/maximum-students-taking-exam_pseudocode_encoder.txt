CLASS Solution  
    FUNCTION maxStudents(seats)  
        SET m TO the LENGTH OF seats  
        SET n TO the LENGTH OF element at position zero of seats  
        
        SET broken TO a new list of length m with all elements set to zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of element at position i of seats EQUALS the character indicating broken seat  
                    SET element at position i of broken TO element at position i of broken OR the value one LEFT SHIFTED BY j  
                END IF  
            END FOR  
        END FOR  
        
        FUNCTION is_valid(arrangement)  
            RETURN arrangement AND arrangement RIGHT SHIFTED BY one EQUALS zero  
        END FUNCTION  
        
        SET dp TO a new list of length m where each element is an empty dictionary  
        
        FOR arrangement FROM zero TO the value one LEFT SHIFTED BY n MINUS one  
            IF is_valid(arrangement) AND arrangement AND element at position zero of broken EQUALS zero  
                SET element at key arrangement of element at position zero of dp TO the count of ones in the binary representation of arrangement  
            END IF  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            FOR arrangement FROM zero TO the value one LEFT SHIFTED BY n MINUS one  
                IF is_valid(arrangement) AND arrangement AND element at position i of broken EQUALS zero  
                    SET max_students TO zero  
                    FOR each prev_arrangement IN the keys of element at position i MINUS one of dp  
                        IF arrangement AND prev_arrangement RIGHT SHIFTED BY one EQUALS zero AND arrangement AND prev_arrangement LEFT SHIFTED BY one EQUALS zero  
                            IF element at key prev_arrangement of element at position i MINUS one of dp GREATER THAN max_students  
                                SET max_students TO element at key prev_arrangement of element at position i MINUS one of dp  
                            END IF  
                        END IF  
                    END FOR  
                    SET element at key arrangement of element at position i of dp TO max_students PLUS the count of ones in the binary representation of arrangement  
                END IF  
            END FOR  
        END FOR  
        
        IF element at position m MINUS one of dp IS NOT empty  
            RETURN the maximum value among the values of element at position m MINUS one of dp  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
