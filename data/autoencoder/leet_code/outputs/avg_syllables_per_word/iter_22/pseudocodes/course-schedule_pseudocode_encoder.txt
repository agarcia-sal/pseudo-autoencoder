CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        ASSIGN reference TO a mapping from keys to empty lists named graph  
        SET indegree TO a list of zeros with length numCourses  
        
        FOR each pair consisting of course and prereq IN prerequisites  
            APPEND course TO the list associated with prereq in graph  
            INCREMENT the element at position course of indegree BY one  
        END FOR  
        
        SET queue TO a double ended queue containing all course for course from zero TO numCourses MINUS one WHERE the element at position course of indegree EQUALS zero  
        
        WHILE the queue is not empty  
            REMOVE from the left the element of queue and ASSIGN it TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN the list associated with current_course in graph  
                DECREMENT the element at position next_course of indegree BY one  
                IF the element at position next_course of indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the result of numCourses EQUALS zero  
    END FUNCTION  
END CLASS
