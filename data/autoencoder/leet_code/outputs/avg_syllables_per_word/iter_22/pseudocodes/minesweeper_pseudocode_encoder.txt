CLASS Solution  
    FUNCTION updateBoard(board parameter, click parameter)  
        FUNCTION count_adjacent_mines(r parameter, c parameter)  
            SET count TO zero  
            FOR i FROM r MINUS one TO r PLUS two MINUS one  
                FOR j FROM c MINUS one TO c PLUS two MINUS one  
                    IF i GREATER THAN OR EQUAL TO zero AND i LESS THAN the LENGTH OF board AND j GREATER THAN OR EQUAL TO zero AND j LESS THAN the LENGTH OF element at position zero of board AND element at position i of board at position j EQUALS the character M  
                        INCREMENT count BY one  
                    END IF  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
  
        FUNCTION dfs(r parameter, c parameter)  
            IF NOT r GREATER THAN OR EQUAL TO zero AND r LESS THAN the LENGTH OF board AND c GREATER THAN OR EQUAL TO zero AND c LESS THAN the LENGTH OF element at position zero of board  
                RETURN  
            END IF  
            IF element at position r of board at position c NOT EQUALS the character E  
                RETURN  
            END IF  
            SET mine_count TO count_adjacent_mines with arguments r and c  
            IF mine_count GREATER THAN zero  
                SET element at position r of board at position c TO the STRING REPRESENTATION OF mine_count  
            ELSE  
                SET element at position r of board at position c TO the character B  
                FOR i FROM r MINUS one TO r PLUS two MINUS one  
                    FOR j FROM c MINUS one TO c PLUS two MINUS one  
                        CALL dfs with arguments i and j  
                    END FOR  
                END FOR  
            END IF  
        END FUNCTION  
  
        SET row TO element at position zero of click  
        SET col TO element at position one of click  
        IF element at position row of board at position col EQUALS the character M  
            SET element at position row of board at position col TO the character X  
            RETURN board  
        END IF  
  
        CALL dfs with arguments row and col  
        RETURN board  
    END FUNCTION  
END CLASS
