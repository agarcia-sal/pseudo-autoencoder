CLASS Solution  
    FUNCTION numOfWays(nums)  
        SET MOD TO ten raised to the power of nine PLUS one  
        
        FUNCTION count_ways(arr)  
            IF the LENGTH of arr LESS THAN OR EQUAL TO one THEN  
                RETURN one  
            END IF  
            
            SET root TO element at position one of arr  
            SET left_subtree TO an empty list  
            SET right_subtree TO an empty list  
            
            FOR each element x IN arr  
                IF x LESS THAN root THEN  
                    APPEND x TO left_subtree  
                ELSE IF x GREATER THAN root THEN  
                    APPEND x TO right_subtree  
                END IF  
            END FOR  
            
            SET ways_left TO result of calling count_ways with left_subtree  
            SET ways_right TO result of calling count_ways with right_subtree  
            
            SET total_ways TO the result of calling combination function with the sum of the LENGTH of left_subtree PLUS the LENGTH of right_subtree AND the LENGTH of left_subtree  
            
            RETURN the remainder of the product of ways_left MULTIPLIED BY ways_right MULTIPLIED BY total_ways when DIVIDED BY MOD  
        END FUNCTION  
        
        SET result TO the remainder of the difference of the result of calling count_ways with nums MINUS one when DIVIDED BY MOD  
        RETURN result  
    END FUNCTION  
END CLASS
