CLASS ListNode  
    FUNCTION __init__ with parameters val and next  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET head TO a new ListNode constructed with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values from position one TO the end  
        SET node TO a new ListNode constructed with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 and p2  
    IF p1 EQUALS no value AND p2 EQUALS no value THEN  
        RETURN true  
    END IF  
    IF p1 EQUALS no value OR p2 EQUALS no value THEN  
        RETURN false  
    END IF  
    RETURN the condition that the val attribute of p1 EQUALS the val attribute of p2 AND the result of calling is_same_list with the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd with parameters head and n  
        SET dummy TO a new ListNode constructed with zero  
        SET the next attribute of dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one  
            SET first TO the next attribute of first  
        END FOR  
        WHILE first DOES NOT EQUAL no value  
            SET first TO the next attribute of first  
            SET second TO the next attribute of second  
        END WHILE  
        SET the next attribute of second TO the next attribute of the next attribute of second  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
