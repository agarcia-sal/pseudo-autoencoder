CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a new list containing the single element minus one  
        SET max_length TO zero  
        FOR each index i AND character char IN the string s  
            IF the character char EQUALS the opening parenthesis character  
                APPEND the index i TO the list stack  
            ELSE  
                REMOVE the last element FROM the list stack  
                IF the list stack IS empty  
                    APPEND the index i TO the list stack  
                ELSE  
                    SET current_length TO the index i MINUS the last element in the list stack  
                    IF current_length GREATER THAN max_length  
                        SET max_length TO current_length  
                    END IF  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
