CLASS ListNode  
    FUNCTION __init__(self val DEFAULT zero next DEFAULT None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the last element of values  
        SET node TO ListNode with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN that the value element of p1 EQUALS the value element of p2 AND is_same_list of the next element of p1 and the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1 l2)  
        SET dummy_head TO ListNode with zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 NOT EQUALS None OR l2 NOT EQUALS None OR carry NOT EQUALS zero  
            IF l1 NOT EQUALS None  
                SET val1 TO the value element of l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 NOT EQUALS None  
                SET val2 TO the value element of l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET the next element of current TO ListNode with total MODULO ten  
            SET current TO the next element of current  
            
            IF l1 NOT EQUALS None  
                SET l1 TO the next element of l1  
            END IF  
            IF l2 NOT EQUALS None  
                SET l2 TO the next element of l2  
            END IF  
        END WHILE  
        
        RETURN the next element of dummy_head  
    END FUNCTION  
END CLASS
