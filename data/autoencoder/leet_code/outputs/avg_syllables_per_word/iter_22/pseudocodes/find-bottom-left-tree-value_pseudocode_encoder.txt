CLASS TreeNode  
    FUNCTION __init__ (self val left right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node (values)  
    IF values EQUALS empty list THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value of element at position one of values and no left and right children  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO remove first element from queue  
        IF i LESS THAN LENGTH OF values AND element at position i of values is not none THEN  
            SET node.left TO new TreeNode with value of element at position i of values and no left and right children  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i of values is not none THEN  
            SET node.right TO new TreeNode with value of element at position i of values and no left and right children  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree (p q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree of p.left AND q.left AND is_same_tree of p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue (self root)  
        SET queue TO double ended queue containing root  
        SET leftmost_value TO none  
        WHILE queue is not empty  
            SET level_size TO LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one  
                SET node TO remove first element from queue  
                IF i EQUALS zero THEN  
                    SET leftmost_value TO node.val  
                END IF  
                IF node.left NOT EQUALS none THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right NOT EQUALS none THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
