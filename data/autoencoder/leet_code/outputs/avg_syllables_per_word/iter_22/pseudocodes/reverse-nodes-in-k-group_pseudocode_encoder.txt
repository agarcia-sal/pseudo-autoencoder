CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the values list EQUALS empty  
        RETURN nothing  
    END IF  
    SET head TO a new instance of ListNode initialized with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values from position two TO the last element  
        SET node TO a new instance of ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS nothing AND p2 EQUALS nothing  
        RETURN True  
    END IF  
    IF p1 EQUALS nothing OR p2 EQUALS nothing  
        RETURN False  
    END IF  
    RETURN the logical AND of the condition that the value attribute of p1 EQUALS the value attribute of p2 AND the recursive call to is_same_list with parameters the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseKGroup with parameters head AND k  
        FUNCTION reverseLinkedList with parameters start AND end  
            SET prev TO nothing  
            SET curr TO start  
            WHILE the condition that curr NOT EQUALS end holds  
                SET next_node TO the next attribute of curr  
                SET the next attribute of curr TO prev  
                SET prev TO curr  
                SET curr TO next_node  
            END WHILE  
            SET the next attribute of curr TO prev  
            RETURN end AND start  
        END FUNCTION  

        SET dummy TO a new instance of ListNode initialized with zero  
        SET the next attribute of dummy TO head  
        SET prev_group_end TO dummy  

        WHILE True holds  
            SET kth_node TO prev_group_end  
            FOR each _ FROM one TO k  
                SET kth_node TO the next attribute of kth_node  
                IF kth_node EQUALS nothing  
                    RETURN the next attribute of dummy  
                END IF  
            END FOR  
            SET next_group_start TO the next attribute of kth_node  
            SET the next attribute of kth_node TO nothing  
            SET new_group_start AND new_group_end TO the result of reverseLinkedList called with parameters the next attribute of prev_group_end AND kth_node  
            SET the next attribute of prev_group_end TO new_group_start  
            SET the next attribute of new_group_end TO next_group_start  
            SET prev_group_end TO new_group_end  
        END WHILE  

        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
