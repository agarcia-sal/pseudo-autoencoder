CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO new TreeNode with value at position one of values and no left and no right  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.left TO new TreeNode with value at position i of values and no left and no right  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.right TO new TreeNode with value at position i of values and no left and no right  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value THEN  
        RETURN true  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN false  
    ELSE IF the val property of p NOT EQUALS the val property of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with the left property of p and the left property of q AND the result of is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues(self, root, target, k)  
        FUNCTION inorder_traversal(node)  
            IF node IS no value THEN  
                RETURN  
            END IF  
            CALL inorder_traversal with the left property of node  
            IF the LENGTH OF result LESS THAN k THEN  
                APPEND the val property of node TO result  
            ELSE  
                IF the absolute value of the val property of node MINUS target LESS THAN the absolute value of the element at position one of result MINUS target THEN  
                    REMOVE the element from the front of result  
                    APPEND the val property of node TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal with the right property of node  
        END FUNCTION  
        SET result TO new empty double ended queue  
        CALL inorder_traversal with root  
        RETURN list converted from result  
    END FUNCTION  
END CLASS
