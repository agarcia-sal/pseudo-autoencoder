CLASS Solution  
    FUNCTION countPyramids(grid)  
        FUNCTION count_pyramids_from_top(grid)  
            IF grid EQUALS empty OR the first element of grid EQUALS empty  
                RETURN zero  
            END IF  
            SET m TO the LENGTH OF grid  
            SET n TO the LENGTH OF the first element of grid  
            SET dp TO a new list of lists with m elements each containing n zeros  
            SET count TO zero  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    IF element at position i of grid AND element at position j EQUALS one THEN  
                        IF i EQUALS zero OR j EQUALS zero OR j EQUALS n MINUS one THEN  
                            SET element at position i of dp AND element at position j TO one  
                        ELSE  
                            SET element at position i of dp AND element at position j TO the MINIMUM OF element at position i MINUS one of dp AND element at position j MINUS one AND element at position i MINUS one of dp AND element at position j AND element at position i MINUS one of dp AND element at position j PLUS one PLUS one  
                        END IF  
                        IF element at position i of dp AND element at position j GREATER THAN one THEN  
                            INCREMENT count BY element at position i of dp AND element at position j MINUS one  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        FUNCTION count_pyramids_from_bottom(grid)  
            IF grid EQUALS empty OR the first element of grid EQUALS empty  
                RETURN zero  
            END IF  
            SET m TO the LENGTH OF grid  
            SET n TO the LENGTH OF the first element of grid  
            SET dp TO a new list of lists with m elements each containing n zeros  
            SET count TO zero  
            FOR i FROM m MINUS one DOWN TO zero  
                FOR j FROM zero TO n MINUS one  
                    IF element at position i of grid AND element at position j EQUALS one THEN  
                        IF i EQUALS m MINUS one OR j EQUALS zero OR j EQUALS n MINUS one THEN  
                            SET element at position i of dp AND element at position j TO one  
                        ELSE  
                            SET element at position i of dp AND element at position j TO the MINIMUM OF element at position i PLUS one of dp AND element at position j MINUS one AND element at position i PLUS one of dp AND element at position j AND element at position i PLUS one of dp AND element at position j PLUS one PLUS one  
                        END IF  
                        IF element at position i of dp AND element at position j GREATER THAN one THEN  
                            INCREMENT count BY element at position i of dp AND element at position j MINUS one  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN the RESULT OF count_pyramids_from_top WITH grid PLUS the RESULT OF count_pyramids_from_bottom WITH grid  
    END FUNCTION  
END CLASS
