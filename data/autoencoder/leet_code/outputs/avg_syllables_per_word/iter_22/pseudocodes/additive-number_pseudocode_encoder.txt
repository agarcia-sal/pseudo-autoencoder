CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            
            SET expected TO the STRING REPRESENTATION OF first PLUS second  
            SET length TO the LENGTH OF expected  
            
            IF start PLUS length GREATER THAN the LENGTH OF num OR the SUBSTRING OF num from position start TO position start PLUS length NOT EQUALS expected  
                RETURN False  
            END IF  
            
            RETURN is_valid_sequence(start PLUS length, second, the INTEGER REPRESENTATION OF expected)  
        END FUNCTION  
        
        SET n TO the LENGTH OF num  
        
        FOR i FROM one TO the INTEGER DIVISION of n BY two PLUS one  
            FOR j FROM i PLUS one TO n  
                SET first TO the SUBSTRING OF num from the beginning TO position i  
                SET second TO the SUBSTRING OF num from position i TO position j  
                
                IF the LENGTH OF first GREATER THAN one AND the CHARACTER at position one of first EQUALS zero OR the LENGTH OF second GREATER THAN one AND the CHARACTER at position one of second EQUALS zero  
                    CONTINUE  
                END IF  
                
                IF is_valid_sequence(j, the INTEGER REPRESENTATION OF first, the INTEGER REPRESENTATION OF second)  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
