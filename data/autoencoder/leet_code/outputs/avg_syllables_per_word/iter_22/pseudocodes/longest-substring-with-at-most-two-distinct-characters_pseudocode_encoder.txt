CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        SET character count TO an empty dictionary that returns zero for missing keys  
        SET left TO zero  
        SET maximum length TO zero  
        
        FOR right FROM zero TO the length of s MINUS one  
            INCREMENT the count for the character at position right of s BY one  
            
            WHILE the number of keys in character count IS GREATER THAN two  
                DECREMENT the count for the character at position left of s BY one  
                IF the count for the character at position left of s EQUALS zero  
                    REMOVE the character at position left of s FROM character count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET maximum length TO the greater of maximum length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN maximum length  
    END FUNCTION  
END CLASS
