CLASS Solution  
    FUNCTION rangeBitwiseAnd(left PARAMETER, right PARAMETER)  
        SET shift TO zero  
        WHILE left LESS THAN right  
            SET left TO left DIVIDED BY two discarding remainder  
            SET right TO right DIVIDED BY two discarding remainder  
            INCREMENT shift BY one  
        END WHILE  
        SET result TO left MULTIPLIED BY two RAISED TO THE POWER OF shift  
        RETURN result  
    END FUNCTION  
END CLASS
