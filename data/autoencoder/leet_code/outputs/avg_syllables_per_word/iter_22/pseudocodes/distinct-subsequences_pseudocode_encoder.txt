CLASS Solution  
    FUNCTION numDistinct(s, t)  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
          
        CREATE a two dimensional list dp with m PLUS one rows and n PLUS one columns FILLED WITH zeros  
          
        FOR i FROM zero TO m  
            SET the element at position i and position zero of dp TO one  
        END FOR  
          
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of s EQUALS the element at position j MINUS one of t THEN  
                    SET the element at position i and position j of dp TO the element at position i MINUS one and position j MINUS one of dp PLUS the element at position i MINUS one and position j of dp  
                ELSE  
                    SET the element at position i and position j of dp TO the element at position i MINUS one and position j of dp  
                END IF  
            END FOR  
        END FOR  
          
        RETURN the element at position m and position n of dp  
    END FUNCTION  
END CLASS
