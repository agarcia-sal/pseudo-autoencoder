CLASS Solution  
    FUNCTION closestToTarget(arr target)  
        SET seen TO empty collection of unique values  
        SET min_diff TO positive infinity  
        FOR each num IN arr  
            SET new_seen TO empty collection of unique values  
            FOR each x IN seen  
                APPEND bitwise AND of num AND x TO new_seen  
            END FOR  
            SET seen TO unique values in new_seen combined WITH collection containing num  
            FOR each val IN seen  
                SET current_difference TO absolute value of val MINUS target  
                IF current_difference LESS THAN min_diff  
                    SET min_diff TO current_difference  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
