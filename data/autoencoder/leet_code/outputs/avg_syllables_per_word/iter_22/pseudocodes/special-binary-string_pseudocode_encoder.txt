CLASS Solution  
    FUNCTION makeLargestSpecial WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER s  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each index i AND character char IN elements of s  
                IF character char EQUALS character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND concatenation of character one PLUS result of dfs WITH parameter substring from position start PLUS one TO position i PLUS character zero TO specials  
                    SET start TO index i PLUS one  
                END IF  
            END FOR  
            SORT elements of specials IN descending lexicographical order  
            RETURN concatenation of elements IN specials  
        END FUNCTION  
        RETURN result of dfs WITH parameter s  
    END FUNCTION  
END CLASS
