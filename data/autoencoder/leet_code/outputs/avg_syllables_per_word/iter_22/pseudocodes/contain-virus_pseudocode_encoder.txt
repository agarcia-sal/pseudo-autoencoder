CLASS Solution  
    FUNCTION containVirus(isInfected)  
        FUNCTION neighbors(r, c)  
            RETURN list containing pair of r PLUS one and c  
            APPEND pair of r MINUS one and c  
            APPEND pair of r and c PLUS one  
            APPEND pair of r and c MINUS one  
        END FUNCTION  

        FUNCTION bfs(r, c)  
            SET infected TO empty set  
            SET threatened TO empty set  
            SET queue TO a double ended queue initialized with pair of r and c  
            SET visited TO a set initialized with pair of r and c  
            WHILE queue holds elements  
                REMOVE and RETURN leftmost element of queue INTO x and y  
                ADD pair of x and y TO infected  
                FOR each nx and ny IN neighbors of x and y  
                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND pair of nx and ny NOT IN visited THEN  
                        ADD pair of nx and ny TO visited  
                        IF element at position nx of isInfected AND element at position ny of element at position nx of isInfected EQUALS one THEN  
                            APPEND pair of nx and ny TO queue  
                        ELSE IF element at position nx of isInfected AND element at position ny of element at position nx of isInfected EQUALS zero THEN  
                            ADD pair of nx and ny TO threatened  
                        END IF  
                    END IF  
                END FOR  
            END WHILE  
            RETURN infected AND threatened  
        END FUNCTION  

        FUNCTION build_walls(infected)  
            SET walls TO zero  
            FOR each x and y IN infected  
                FOR each nx and ny IN neighbors of x and y  
                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx of isInfected AND element at position ny of element at position nx of isInfected EQUALS zero THEN  
                        INCREMENT walls BY one  
                    END IF  
                END FOR  
            END FOR  
            RETURN walls  
        END FUNCTION  

        FUNCTION spread_virus(threatened)  
            FOR each x and y IN threatened  
                SET element at position x of isInfected AND element at position y of element at position x of isInfected TO one  
            END FOR  
        END FUNCTION  

        SET m TO the LENGTH OF isInfected  
        SET n TO the LENGTH OF element at position zero of isInfected  
        SET total_walls TO zero  

        WHILE true holds  
            SET regions TO empty list  
            SET visited TO empty set  
            FOR r FROM zero TO m MINUS one  
                FOR c FROM zero TO n MINUS one  
                    IF element at position r of isInfected AND element at position c of element at position r of isInfected EQUALS one AND pair of r and c NOT IN visited THEN  
                        CALL bfs with r and c INTO infected and threatened  
                        APPEND pair of infected and threatened TO regions  
                        ASSIGN visited TO the union of visited AND infected  
                    END IF  
                END FOR  
            END FOR  

            IF the LENGTH OF regions EQUALS zero THEN  
                BREAK  
            END IF  

            SORT regions in descending order based on the LENGTH OF the second element of each region  

            INCREMENT total_walls BY the result of build_walls with the first element of the first element of regions  
            FOR each x and y IN the first element of the first element of regions  
                SET element at position x of isInfected AND element at position y of element at position x of isInfected TO negative one  
            END FOR  

            FOR i FROM one TO the LENGTH OF regions MINUS one  
                CALL spread_virus with the second element of the element at position i of regions  
            END FOR  
        END WHILE  

        RETURN total_walls  
    END FUNCTION  
END CLASS
