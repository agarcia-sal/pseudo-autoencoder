CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the value list is empty EQUALS True THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode constructed with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty DO  
        SET node TO the element removed from the front of queue  
        IF i less than the length of values AND the element at position i of values is not equal to None THEN  
            SET the left child of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the left child of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i less than the length of values AND the element at position i of values is not equal to None THEN  
            SET the right child of node TO a new TreeNode constructed with the element at position i of values  
            APPEND the right child of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p is None AND q is None THEN  
        RETURN True  
    ELSE IF p is None OR q is None THEN  
        RETURN False  
    ELSE IF the value of p is not equal to the value of q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left child of p and the left child of q AND the result of is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree(self, root)  
        IF root is None OR the left child of root is None THEN  
            RETURN root  
        END IF  
        SET new_root TO the result of calling upsideDownBinaryTree on the left child of root  
        SET the left child of the left child of root TO the right child of root  
        SET the right child of the left child of root TO root  
        SET the left child of root TO None  
        SET the right child of root TO None  
        RETURN new_root  
    END FUNCTION  
END CLASS
