CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO the LENGTH OF text  
        SET substrings TO an EMPTY SET  
        FOR length FROM one TO n DIVIDED BY two INCLUSIVE  
            FOR start FROM zero TO n MINUS two MULTIPLIED BY length INCLUSIVE  
                SET substring_a TO the substring FROM position start TO position start PLUS length MINUS one OF text  
                SET substring_b TO the substring FROM position start PLUS length TO position start PLUS two MULTIPLIED BY length MINUS one OF text  
                IF substring_a EQUALS substring_b  
                    ADD the concatenation of substring_a AND substring_b TO substrings  
                END IF  
            END FOR  
        END FOR  
        RETURN the NUMBER OF ELEMENTS IN substrings  
    END FUNCTION  
END CLASS
