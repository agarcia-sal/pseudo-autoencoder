CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO list of integer negative one repeated twenty six times  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each i WITH each char IN enumeration of s  
            SET index TO numerical value of char MINUS numerical value of character a  
            INCREMENT current_appeal BY i MINUS element at position index of last  
            INCREMENT total_appeal BY current_appeal  
            SET element at position index of last TO i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
