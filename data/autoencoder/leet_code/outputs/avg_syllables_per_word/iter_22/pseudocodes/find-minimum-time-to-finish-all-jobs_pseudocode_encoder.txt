CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        SET subset_sums TO a list of zeros with LENGTH equal TO TWO RAISED TO THE POWER OF n  
        FOR i FROM zero TO n MINUS one  
            FOR mask FROM zero TO TWO RAISED TO THE POWER OF i MINUS one  
                SET element at position mask OR TWO RAISED TO THE POWER OF i IN subset_sums TO element at position mask IN subset_sums PLUS element at position i IN jobs  
            END FOR  
        END FOR  
        
        FUNCTION dp(k, mask)  
            IF k EQUALS one THEN  
                RETURN element at position mask IN subset_sums  
            END IF  
            SET res TO positive infinity  
            SET submask TO mask  
            WHILE submask IS NOT zero  
                SET candidate TO THE GREATER OF dp(k MINUS one, mask EXCLUSIVE OR submask) AND element at position submask IN subset_sums  
                IF candidate LESS THAN res THEN  
                    SET res TO candidate  
                END IF  
                SET submask TO bitwise AND OF submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        RETURN dp(k, TWO RAISED TO THE POWER OF n MINUS one)  
    END FUNCTION  
END CLASS
