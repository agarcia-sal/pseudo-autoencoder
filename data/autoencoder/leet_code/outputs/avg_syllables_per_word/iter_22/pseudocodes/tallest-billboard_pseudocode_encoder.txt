CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO a dictionary with key zero and value zero  
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each key d and value y IN dp  
                SET sum_for_taller_support TO d PLUS rod  
                IF new_dp contains key sum_for_taller_support  
                    SET new_dp at key sum_for_taller_support TO the greater value BETWEEN new_dp at key sum_for_taller_support AND y  
                ELSE  
                    SET new_dp at key sum_for_taller_support TO y  
                END IF  
                SET absolute_difference TO the absolute value of d MINUS rod  
                SET sum_for_shorter_support TO y PLUS the smaller value BETWEEN d AND rod  
                IF new_dp contains key absolute_difference  
                    SET new_dp at key absolute_difference TO the greater value BETWEEN new_dp at key absolute_difference AND sum_for_shorter_support  
                ELSE  
                    SET new_dp at key absolute_difference TO sum_for_shorter_support  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN the value in dp at key zero  
    END FUNCTION  
END CLASS
