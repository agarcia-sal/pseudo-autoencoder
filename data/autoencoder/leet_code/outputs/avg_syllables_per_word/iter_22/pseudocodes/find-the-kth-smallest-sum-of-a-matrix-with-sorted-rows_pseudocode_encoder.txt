CLASS Solution  
    FUNCTION kthSmallest(mat WITH TYPE List OF List OF Integer, k WITH TYPE Integer) RETURNS Integer  
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF the element at position zero of mat  
        SET initial_sum TO the SUM OF element at position zero of each row in mat  
        SET initial_indices TO a tuple consisting of zero repeated m times  
        SET min_heap TO a list containing a single element consisting of initial_sum AND initial_indices  
        SET visited TO a set containing initial_indices  
        
        WHILE k IS GREATER THAN zero  
            REMOVE AND RETURN the smallest element from min_heap ASSIGN TO current_sum AND indices  
            DECREMENT k BY one  
            
            IF k EQUALS zero  
                RETURN current_sum  
            END IF  
            
            FOR i FROM zero TO m MINUS one  
                IF the element at position i of indices PLUS one IS LESS THAN n  
                    SET new_indices TO a list COPY OF indices  
                    INCREMENT the element at position i of new_indices BY one  
                    SET new_indices_tuple TO a tuple CONVERTED FROM new_indices  
                    SET new_sum TO current_sum MINUS the element at position i AND the element at position i of indices OF mat PLUS the element at position i AND the element at position i of new_indices OF mat  
                    
                    IF new_indices_tuple IS NOT IN visited  
                        ADD new_indices_tuple TO visited  
                        INSERT the tuple consisting of new_sum AND new_indices_tuple INTO min_heap MAINTAINING heap property  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
