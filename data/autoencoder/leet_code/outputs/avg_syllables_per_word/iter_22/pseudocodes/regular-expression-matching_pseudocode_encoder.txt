CLASS Solution  
    FUNCTION isMatch(s  
                     p)  
        SET dp TO a two dimensional list with the number of rows equal to the length of s plus one and the number of columns equal to the length of p plus one where each element is False  
        SET element at position zero zero of dp TO True  
        FOR j FROM one TO the length of p  
            IF the element at position j minus one of p EQUALS the character star  
                SET element at position zero j of dp TO the value of element at position zero j minus two of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF the element at position j minus one of p EQUALS the element at position i minus one of s OR the element at position j minus one of p EQUALS the character dot  
                    SET element at position i j of dp TO the value of element at position i minus one j minus one of dp  
                ELSE IF the element at position j minus one of p EQUALS the character star  
                    SET element at position i j of dp TO the value of element at position i j minus two of dp  
                    IF the element at position j minus two of p EQUALS the element at position i minus one of s OR the element at position j minus two of p EQUALS the character dot  
                        SET element at position i j of dp TO the value of element at position i j of dp OR the value of element at position i minus one j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the value of element at position length of s length of p of dp  
    END FUNCTION  
END CLASS
