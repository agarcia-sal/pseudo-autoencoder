CLASS Solution  
    FUNCTION minimumDeleteSum(s1 s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        
        CREATE dp AS a TWO DIMENSIONAL LIST WITH m PLUS one ROWS AND n PLUS one COLUMNS FILLED WITH zero  
        
        FOR j FROM one TO n  
            SET element at position zero AND position j of dp TO element at position zero AND position j MINUS one of dp PLUS the ASCII VALUE OF element at position j MINUS one of s2  
        END FOR  
        
        FOR i FROM one TO m  
            SET element at position i AND position zero of dp TO element at position i MINUS one AND position zero of dp PLUS the ASCII VALUE OF element at position i MINUS one of s1  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of s1 EQUALS element at position j MINUS one of s2  
                    SET element at position i AND position j of dp TO element at position i MINUS one AND position j MINUS one of dp  
                ELSE  
                    SET element at position i AND position j of dp TO the MINIMUM OF element at position i MINUS one AND position j of dp PLUS the ASCII VALUE OF element at position i MINUS one of s1 AND element at position i AND position j MINUS one of dp PLUS the ASCII VALUE OF element at position j MINUS one of s2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position m AND position n of dp  
    END FUNCTION  
END CLASS
