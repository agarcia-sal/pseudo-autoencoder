CLASS Solution  
    FUNCTION kthGrammar(n IN INTEGER, k IN INTEGER) RETURNS INTEGER  
        IF n EQUALS one THEN  
            RETURN zero  
        END IF  
        SET length_of_previous_row TO two RAISED TO THE POWER OF n MINUS two  
        IF k LESS THAN OR EQUAL TO length_of_previous_row THEN  
            RETURN self DOT kthGrammar WITH PARAMETERS n MINUS one AND k  
        ELSE  
            SET value TO self DOT kthGrammar WITH PARAMETERS n MINUS one AND k MINUS length_of_previous_row  
            RETURN one MINUS value  
        END IF  
    END FUNCTION  
END CLASS
