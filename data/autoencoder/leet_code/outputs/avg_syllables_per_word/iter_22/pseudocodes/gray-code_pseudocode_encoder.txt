CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN list containing zero  
        END IF  
        SET prev_gray TO CALL grayCode WITH n MINUS one  
        SET mask TO one MULTIPLIED BY two RAISED TO THE POWER OF n MINUS one  
        SET reversed_prev_gray TO reverse the order of elements in prev_gray  
        SET reflected_list TO empty list  
        FOR each num IN reversed_prev_gray  
            SET computed_value TO mask OR num  
            APPEND computed_value TO reflected_list  
        END FOR  
        SET current_gray TO the concatenation of prev_gray FOLLOWED BY reflected_list  
        RETURN current_gray  
    END FUNCTION  
END CLASS
