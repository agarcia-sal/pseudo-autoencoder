CLASS Solution  
    FUNCTION findMaxValueOfEquation(points PARAMETER points OF TYPE List OF List OF integer, k PARAMETER k OF TYPE integer) RETURNS integer  
        SET max_value TO the smallest possible number  
        CREATE an empty double ended queue named deque_  
        FOR each pair consisting of xj and yj IN points  
            WHILE deque_ is not empty AND the value of xj MINUS the second element of the first pair in deque_ is GREATER THAN k  
                REMOVE the first element from deque_  
            END WHILE  
            IF deque_ is not empty  
                SET max_value TO the larger of max_value AND the sum of yj PLUS xj PLUS the first element of the first pair in deque_  
            END IF  
            WHILE deque_ is not empty AND the value of yj MINUS xj IS GREATER THAN OR EQUAL TO the first element of the last pair in deque_  
                REMOVE the last element from deque_  
            END WHILE  
            APPEND the pair consisting of yj MINUS xj AND xj TO deque_  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
