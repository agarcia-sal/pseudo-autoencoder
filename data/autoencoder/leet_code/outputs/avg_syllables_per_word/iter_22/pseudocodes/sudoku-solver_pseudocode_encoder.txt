CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR each i FROM zero TO eight  
                IF element at position i of row at position row of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            FOR each i FROM zero TO eight  
                IF element at position col of row at position i of board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY the INTEGER DIVISION of row BY three  
            SET start_col TO three MULTIPLIED BY the INTEGER DIVISION of col BY three  
            FOR each i FROM zero TO two  
                FOR each j FROM zero TO two  
                    IF element at position start_col PLUS j of row at position start_row PLUS i of board EQUALS num  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
        FUNCTION solve(board)  
            FOR each row FROM zero TO eight  
                FOR each col FROM zero TO eight  
                    IF element at position col of row at position row of board EQUALS a dot character  
                        FOR each num IN the sequence of string representations of numbers from one TO nine  
                            IF is_valid(board, row, col, num)  
                                SET element at position col of row at position row of board TO num  
                                IF solve(board)  
                                    RETURN True  
                                END IF  
                                SET element at position col of row at position row of board TO a dot character  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
        CALL solve(board)  
    END FUNCTION  
END CLASS
