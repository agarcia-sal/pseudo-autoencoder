CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a dictionary mapping each character from a to z TO zero  
        SET current_length TO zero  
        FOR index FROM zero TO the length of s MINUS one  
            IF index GREATER THAN zero AND (the code number of the element at position index of s MINUS the code number of the element at position index MINUS one of s EQUALS one OR the element at position index of s EQUALS character a AND the element at position index MINUS one of s EQUALS character z)  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET the value in max_length for the element at position index of s TO the greater value between the current value in max_length for the element at position index of s AND current_length  
        END FOR  
        SET total_sum TO zero  
        FOR each value IN the values of max_length  
            INCREMENT total_sum BY value  
        END FOR  
        RETURN total_sum  
    END FUNCTION  
END CLASS
