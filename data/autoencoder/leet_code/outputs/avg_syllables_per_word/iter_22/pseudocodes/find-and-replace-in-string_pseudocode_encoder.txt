CLASS Solution  
    FUNCTION findReplaceString(s parameter, indices parameter, sources parameter, targets parameter)  
        SET replacements TO a list of elements each consisting of element at position zero of zipped elements from indices and sources and targets SORTED BY element at position zero of each element  
        SET result TO an empty list  
        SET prev_end TO zero  
        FOR each element consisting of index comma source comma target IN replacements  
            APPEND the substring of s from position prev_end TO position index MINUS one TO result  
            IF the substring of s from position index TO position index PLUS the LENGTH OF source MINUS one EQUALS source THEN  
                APPEND target TO result  
            ELSE  
                APPEND the substring of s from position index TO position index PLUS the LENGTH OF source MINUS one TO result  
            END IF  
            SET prev_end TO index PLUS the LENGTH OF source  
        END FOR  
        APPEND the substring of s from position prev_end TO the end TO result  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
