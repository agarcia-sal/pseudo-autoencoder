CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET n TO the LENGTH OF s  
        
        SET last TO a list of twenty six elements each SET TO negative one  
        SET last2 TO a list of twenty six elements each SET TO negative one  
        
        SET result TO zero  
        
        FOR each index i FROM zero TO n MINUS one  
            SET char TO element at position i of s  
            SET idx TO the Unicode code point of char MINUS the Unicode code point of uppercase A  
            
            SET contribution TO (i MINUS element at position idx of last) MULTIPLIED BY (element at position idx of last MINUS element at position idx of last2)  
            SET result TO result PLUS contribution  
            SET result TO result MODULO MOD  
            
            SET element at position idx of last2 TO element at position idx of last  
            SET element at position idx of last TO i  
        END FOR  
        
        FOR each index i FROM zero TO twenty five  
            SET contribution TO (n MINUS element at position i of last) MULTIPLIED BY (element at position i of last MINUS element at position i of last2)  
            SET result TO result PLUS contribution  
            SET result TO result MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
