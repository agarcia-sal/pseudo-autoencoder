CLASS Solution  
    FUNCTION matrixScore(grid)  
        FOR each row IN grid  
            IF element at position zero of row EQUALS zero  
                FOR index FROM zero TO the LENGTH OF row MINUS one  
                    SET element at position index of row TO one MINUS element at position index of row  
                END FOR  
            END IF  
        END FOR  
        FOR column_index FROM one TO the LENGTH OF element at position zero of grid MINUS one  
            SET count_of_ones TO zero  
            FOR row_index FROM zero TO the LENGTH OF grid MINUS one  
                INCREMENT count_of_ones BY element at position row_index of grid AT position column_index  
            END FOR  
            IF count_of_ones LESS THAN the LENGTH OF grid DIVIDED BY two  
                FOR row_index FROM zero TO the LENGTH OF grid MINUS one  
                    SET element at position row_index of grid AT position column_index TO one MINUS element at position row_index of grid AT position column_index  
                END FOR  
            END IF  
        END FOR  
        SET score TO zero  
        FOR each row IN grid  
            SET binary_representation TO empty string  
            FOR each element IN row  
                APPEND string representation of element TO binary_representation  
            END FOR  
            SET numeric_value TO convert binary_representation from base two to decimal number  
            INCREMENT score BY numeric_value  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
