CLASS Solution  
    FUNCTION findMedianSortedArrays(nums1 PARAMETER, nums2 PARAMETER)  
        IF the LENGTH OF nums1 IS GREATER THAN the LENGTH OF nums2 THEN  
            SET temporary variable nums1 TO nums2  
            SET temporary variable nums2 TO nums1  
        END IF  
        
        SET variable x TO the LENGTH OF nums1  
        SET variable y TO the LENGTH OF nums2  
        SET variable low TO zero  
        SET variable high TO x  
        
        WHILE low IS LESS THAN OR EQUAL TO high  
            SET variable partitionX TO the INTEGER DIVISION OF low PLUS high BY two  
            SET variable partitionY TO the INTEGER DIVISION OF x PLUS y PLUS one BY two MINUS partitionX  
            
            IF partitionX EQUALS zero THEN  
                SET variable maxLeftX TO negative infinity  
            ELSE  
                SET variable maxLeftX TO the element at position partitionX MINUS one OF nums1  
            END IF  
            
            IF partitionX EQUALS x THEN  
                SET variable minRightX TO positive infinity  
            ELSE  
                SET variable minRightX TO the element at position partitionX OF nums1  
            END IF  
            
            IF partitionY EQUALS zero THEN  
                SET variable maxLeftY TO negative infinity  
            ELSE  
                SET variable maxLeftY TO the element at position partitionY MINUS one OF nums2  
            END IF  
            
            IF partitionY EQUALS y THEN  
                SET variable minRightY TO positive infinity  
            ELSE  
                SET variable minRightY TO the element at position partitionY OF nums2  
            END IF  
            
            IF maxLeftX IS LESS THAN OR EQUAL TO minRightY AND maxLeftY IS LESS THAN OR EQUAL TO minRightX THEN  
                IF the REMAINDER WHEN x PLUS y IS DIVIDED BY two EQUALS zero THEN  
                    RETURN the SUM OF the GREATER VALUE BETWEEN maxLeftX AND maxLeftY AND the LESSER VALUE BETWEEN minRightX AND minRightY DIVIDED BY two  
                ELSE  
                    RETURN the GREATER VALUE BETWEEN maxLeftX AND maxLeftY  
                END IF  
            ELSE IF maxLeftX IS GREATER THAN minRightY THEN  
                SET high TO partitionX MINUS one  
            ELSE  
                SET low TO partitionX PLUS one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
