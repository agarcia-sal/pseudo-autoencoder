CLASS Solution  
    FUNCTION generateParenthesis WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS s WITH DEFAULT empty string left WITH DEFAULT zero right WITH DEFAULT zero  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s CONCATENATED WITH the character for opening parenthesis left PLUS one right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s CONCATENATED WITH the character for closing parenthesis left right PLUS one  
            END IF  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH NO ARGUMENTS  
        RETURN result  
    END FUNCTION  
END CLASS
