CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO list of characters representing the number n converted to string  
        SET length TO the number of elements in digits  
        SET marker TO length  
        FOR i FROM length MINUS one TO one IN DECREASING order  
            IF element at position i of digits LESS THAN element at position i MINUS one of digits  
                SET element at position i MINUS one of digits TO string representation of integer value of element at position i MINUS one of digits MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one  
            SET element at position i of digits TO character representing digit nine  
        END FOR  
        SET result_string TO concatenation of all elements of digits in order  
        SET result_integer TO integer representation of result_string  
        RETURN result_integer  
    END FUNCTION  
END CLASS
