CLASS Solution  
    FUNCTION knightProbability(n k row column)  
        SET moves TO a list containing pairs of integers as follows  
            a pair with first element equal to minus two and second element equal to minus one  
            a pair with first element equal to minus two and second element equal to one  
            a pair with first element equal to minus one and second element equal to minus two  
            a pair with first element equal to minus one and second element equal to two  
            a pair with first element equal to one and second element equal to minus two  
            a pair with first element equal to one and second element equal to two  
            a pair with first element equal to two and second element equal to minus one  
            a pair with first element equal to two and second element equal to one  
        
        SET dp TO a two dimensional list of size n by n where each element is set to zero  
        SET element at position row of dp and position column of that sublist TO one  
        
        FOR each iteration FROM one TO k  
            SET new_dp TO a two dimensional list of size n by n where each element is set to zero  
            
            FOR r FROM zero TO n minus one  
                FOR c FROM zero TO n minus one  
                    IF element at position r of dp and position c of that sublist NOT EQUALS zero THEN  
                        FOR each pair dr dc IN moves  
                            SET nr TO r PLUS dr  
                            SET nc TO c PLUS dc  
                            IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n THEN  
                                INCREMENT element at position nr of new_dp and position nc of that sublist BY element at position r of dp and position c of that sublist DIVIDED BY eight  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            ASSIGN dp TO new_dp  
        END FOR  
        
        SET total_probability TO zero  
        FOR each row_list IN dp  
            SET row_sum TO zero  
            FOR each probability_value IN row_list  
                INCREMENT row_sum BY probability_value  
            END FOR  
            INCREMENT total_probability BY row_sum  
        END FOR  
        
        RETURN total_probability  
    END FUNCTION  
END CLASS
