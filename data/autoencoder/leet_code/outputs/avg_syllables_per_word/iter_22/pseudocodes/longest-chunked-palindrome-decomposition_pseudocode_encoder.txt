CLASS Solution  
    FUNCTION longestDecomposition(text)  
        SET left TO zero  
        SET right TO the LENGTH OF text MINUS one  
        SET left_part TO an empty sequence of characters  
        SET right_part TO an empty sequence of characters  
        SET k TO zero  
        
        WHILE the value of left IS LESS THAN the value of right  
            SET left_part TO left_part PLUS the element at position left of text  
            SET right_part TO the element at position right of text PLUS right_part  
            
            IF left_part EQUALS right_part  
                INCREMENT k BY two  
                SET left_part TO an empty sequence of characters  
                SET right_part TO an empty sequence of characters  
            END IF  
            
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
        
        IF the sequence left_part IS NOT empty OR the value of left EQUALS the value of right  
            INCREMENT k BY one  
        END IF  
        
        RETURN k  
    END FUNCTION  
END CLASS
