CLASS Solution  
    FUNCTION coinChange(coins amount)  
        SET dp TO a list of length amount plus one where each element is amount plus one  
        SET element at position zero of dp TO zero  
        FOR each a FROM one TO amount  
            FOR each coin IN coins  
                IF a MINUS coin GREATER THAN OR EQUAL TO zero  
                    SET candidate TO element at position a MINUS coin of dp PLUS one  
                    IF candidate LESS THAN element at position a of dp  
                        SET element at position a of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        IF element at position amount of dp NOT EQUALS amount PLUS one  
            RETURN element at position amount of dp  
        ELSE  
            RETURN minus one  
        END IF  
    END FUNCTION  
END CLASS
