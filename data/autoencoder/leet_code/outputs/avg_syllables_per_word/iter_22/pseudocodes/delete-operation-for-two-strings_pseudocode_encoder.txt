CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        INITIALIZE dp AS a two dimensional list with m PLUS one rows AND n PLUS one columns FILLED WITH zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of word1 EQUALS element at position j MINUS one of word2  
                    SET element at position i AND j of dp TO element at position i MINUS one AND j MINUS one of dp PLUS one  
                ELSE  
                    SET element at position i AND j of dp TO the GREATER VALUE between element at position i MINUS one AND j of dp AND element at position i AND j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        SET lcs_length TO element at position m AND n of dp  
        RETURN m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  
END CLASS
