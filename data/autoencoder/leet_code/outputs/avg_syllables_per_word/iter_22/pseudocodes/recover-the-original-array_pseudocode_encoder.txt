CLASS Solution  
    FUNCTION recoverArray(nums)  
        SORT nums IN ascending order  
        SET n TO the LENGTH OF nums  
        
        SET first TO element at position one of nums  
        
        FOR i FROM two TO n  
            SET k TO element at position i of nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at position i of nums  
                CONTINUE TO next iteration  
            END IF  
            
            SET count TO a collection mapping each number IN nums TO its frequency  
            SET arr TO an empty list  
            
            FOR each x IN nums  
                IF the frequency of x in count EQUALS zero  
                    CONTINUE TO next iteration  
                END IF  
                IF the frequency of x PLUS two MULTIPLIED BY k in count EQUALS zero  
                    BREAK out of loop  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT the frequency of x in count BY one  
                DECREMENT the frequency of x PLUS two MULTIPLIED BY k in count BY one  
            END FOR  
            
            IF the LENGTH OF arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        
        RETURN an empty list  
    END FUNCTION  
END CLASS
