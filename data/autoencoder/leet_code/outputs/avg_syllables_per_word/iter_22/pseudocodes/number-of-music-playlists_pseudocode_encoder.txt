CLASS Solution  
    FUNCTION numMusicPlaylists(n goal k)  
        SET MOD TO ten POWER nine PLUS one  
        CREATE dp AS two dimensional list with dimensions goal PLUS one BY n PLUS one FILLED WITH zero  
        SET element at position zero AND zero of dp TO one  
        FOR i FROM one TO goal  
            FOR j FROM one TO n  
                INCREMENT element at position i AND j of dp BY element at position i MINUS one AND j MINUS one of dp MULTIPLIED BY n MINUS j MINUS one PLUS one  
                SET element at position i AND j of dp TO element at position i AND j of dp MODULO MOD  
                IF j GREATER THAN k THEN  
                    INCREMENT element at position i AND j of dp BY element at position i MINUS one AND j of dp MULTIPLIED BY j MINUS k  
                    SET element at position i AND j of dp TO element at position i AND j of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position goal AND n of dp  
    END FUNCTION  
END CLASS
