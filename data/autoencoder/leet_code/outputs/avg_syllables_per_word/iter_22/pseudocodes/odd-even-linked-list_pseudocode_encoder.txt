CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET head TO a new ListNode initialized with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from the second element TO the last element  
        SET node TO a new ListNode initialized with val  
        SET the next reference of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the equality predicate where the value stored in p1 EQUALS the value stored in p2 AND the recursive call to is_same_list with the next reference of p1 AND the next reference of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList(head)  
        IF head EQUALS None OR the next reference of head EQUALS None OR the next reference of the next reference of head EQUALS None THEN  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO the next reference of head  
        SET even_head TO even  
        WHILE even IS NOT None AND the next reference of even IS NOT None  
            SET the next reference of odd TO the next reference of even  
            SET odd TO the next reference of odd  
            SET the next reference of even TO the next reference of odd  
            SET even TO the next reference of even  
        END WHILE  
        SET the next reference of odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
