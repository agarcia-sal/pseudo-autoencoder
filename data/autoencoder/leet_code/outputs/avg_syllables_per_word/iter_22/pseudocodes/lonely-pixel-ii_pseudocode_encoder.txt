CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture EQUALS empty OR element at position zero of picture EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO a list of zeros of size m  
        SET col_count TO a list of zeros of size n  
        SET row_patterns TO a mapping from string TO integer with default zero  
        
        FOR r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of all elements in element at position r of picture as a single string  
            INCREMENT value at key pattern in row_patterns BY one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS the letter B  
                    INCREMENT element at position r of row_count BY one  
                    INCREMENT element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS the letter B  
                    SET pattern TO the concatenation of all elements in element at position r of picture as a single string  
                    IF element at position r of row_count EQUALS target AND element at position c of col_count EQUALS target AND value at key pattern in row_patterns EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
