CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MODULO TO ten raised to the power of nine PLUS one  
        SORT the list nums in ascending order  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        INITIALIZE list power WITH n elements all equal to one  
        
        FOR index i FROM one TO n MINUS one  
            SET element at position i of power TO element at position i MINUS one of power MULTIPLIED BY two MODULO MODULO  
        END FOR  
        
        FOR index i FROM zero TO n MINUS one  
            SET total_sum TO total_sum PLUS element at position i of nums MULTIPLIED BY element at position i of power MINUS element at position n MINUS i MINUS one of power  
            SET total_sum TO total_sum MODULO MODULO  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
