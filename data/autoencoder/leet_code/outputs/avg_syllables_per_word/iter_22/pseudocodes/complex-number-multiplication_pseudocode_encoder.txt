CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO result of splitting num at each occurrence of the plus character  
            SET real TO integer converted from element at position zero of parts  
            SET imaginary TO integer converted from substring of element at position one of parts from first position TO position before last character  
            RETURN real AND imaginary  
        END FUNCTION  
        
        SET real1 AND imag1 TO result of calling parse_complex with num1  
        SET real2 AND imag2 TO result of calling parse_complex with num2  
        
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        
        RETURN string formed by concatenating string representation of real_part plus plus character plus string representation of imaginary_part plus character i  
    END FUNCTION  
END CLASS
