CLASS Solution  
    FUNCTION uniquePathsIII(grid)  
        FUNCTION dfs(x, y, steps)  
            IF element at position x of grid AT position y EQUALS two THEN  
                RETURN steps EQUALS zero  
            END IF  
            SET original TO element at position x of grid AT position y  
            SET element at position x of grid AT position y TO negative two  
            SET count TO zero  
            FOR each pair dx dy IN the list of pairs negative one zero AND one zero AND zero negative one AND zero one  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx of grid AT position ny GREATER THAN OR EQUAL TO zero THEN  
                    INCREMENT count BY the result of dfs WITH nx ny AND steps MINUS one  
                END IF  
            END FOR  
            SET element at position x of grid AT position y TO original  
            RETURN count  
        END FUNCTION  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET start_x TO zero  
        SET start_y TO zero  
        SET empty_squares TO one  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of grid AT position j EQUALS one THEN  
                    SET start_x TO i  
                    SET start_y TO j  
                ELSE IF element at position i of grid AT position j EQUALS zero THEN  
                    INCREMENT empty_squares BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN the result of dfs WITH start_x start_y AND empty_squares  
    END FUNCTION  
END CLASS
