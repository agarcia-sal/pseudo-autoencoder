CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
        FOR each num IN nums  
            SET depth TO num DIVIDED BY one hundred  
            SET pos TO num DIVIDED BY ten MODULO ten  
            SET value TO num MODULO ten  
            ASSIGN reference OF value TO the element at pair of depth and pos in tree  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the value obtained from tree at pair of depth and pos  
            IF current_value EQUALS no value exists THEN  
                RETURN zero  
            END IF  
            
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child TO the pair of depth PLUS one AND pos MULTIPLIED BY two MINUS one  
            SET right_child TO the pair of depth PLUS one AND pos MULTIPLIED BY two  
            
            IF left_child NOT IN tree AND right_child NOT IN tree THEN  
                RETURN new_path_sum  
            END IF  
            
            SET left_sum TO dfs with parameters first element of left_child, second element of left_child, new_path_sum  
            SET right_sum TO dfs with parameters first element of right_child, second element of right_child, new_path_sum  
            
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN dfs with parameters one, one, zero  
    END FUNCTION  
END CLASS
