CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS None  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode WITH val EQUALS element at position one of values  
    SET p TO head  
    FOR each val IN elements of values from position two TO the end  
        SET node TO new ListNode WITH val EQUALS val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the predicate that p1 dot val EQUALS p2 dot val AND the result of calling is_same_list WITH PARAMETERS p1 dot next p2 dot next  
END FUNCTION  

CLASS Solution  
    FUNCTION partition WITH PARAMETERS self head x  
        SET less_head TO new ListNode WITH val EQUALS zero  
        SET greater_head TO new ListNode WITH val EQUALS zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current IS NOT None  
            IF the predicate that current dot val LESS THAN x  
                SET less dot next TO current  
                SET less TO less dot next  
            ELSE  
                SET greater dot next TO current  
                SET greater TO greater dot next  
            END IF  
            SET current TO current dot next  
        END WHILE  
        SET less dot next TO greater_head dot next  
        SET greater dot next TO None  
        RETURN less_head dot next  
    END FUNCTION  
END CLASS
