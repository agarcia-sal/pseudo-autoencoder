CLASS Solution  
    FUNCTION findPairs(nums k)  
        IF k LESS THAN zero  
            RETURN zero  
        END IF  
        SET num_counts TO an empty mapping  
        FOR each num IN nums  
            IF num EXISTS AS A KEY IN num_counts  
                SET current_count TO value ASSOCIATED WITH num IN num_counts  
            ELSE  
                SET current_count TO zero  
            END IF  
            SET updated_count TO current_count PLUS one  
            ASSIGN updated_count TO key num IN num_counts  
        END FOR  
        SET unique_pairs TO zero  
        FOR each num IN the set OF keys IN num_counts  
            IF k EQUALS zero  
                IF value ASSOCIATED WITH num IN num_counts GREATER THAN one  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF num PLUS k EXISTS AS A KEY IN num_counts  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
