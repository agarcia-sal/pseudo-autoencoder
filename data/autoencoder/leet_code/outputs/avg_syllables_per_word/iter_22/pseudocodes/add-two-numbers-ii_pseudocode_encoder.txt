CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO A new ListNode with element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two TO the end  
        SET node TO A new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN self.val EQUALS p2.val AND is_same_list(p1.next p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1 l2)  
        FUNCTION pushIntoStack(node)  
            SET stack TO an empty list  
            WHILE node DOES NOT EQUAL None  
                APPEND node.val TO stack  
                SET node TO node.next  
            END WHILE  
            RETURN stack  
        END FUNCTION  
        
        SET stack1 TO pushIntoStack(l1)  
        SET stack2 TO pushIntoStack(l2)  
        
        SET carry TO zero  
        SET result TO None  
        
        WHILE the LENGTH OF stack1 GREATER THAN zero OR the LENGTH OF stack2 GREATER THAN zero OR carry NOT EQUALS zero  
            IF the LENGTH OF stack1 GREATER THAN zero  
                SET val1 TO the last element of stack1 and REMOVE it from stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            
            IF the LENGTH OF stack2 GREATER THAN zero  
                SET val2 TO the last element of stack2 and REMOVE it from stack2  
            ELSE  
                SET val2 TO zero  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  
            
            SET new_node TO A new ListNode with digit  
            SET new_node.next TO result  
            SET result TO new_node  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
