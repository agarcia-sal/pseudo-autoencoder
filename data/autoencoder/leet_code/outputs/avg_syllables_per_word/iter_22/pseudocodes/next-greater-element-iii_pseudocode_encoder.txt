CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO list of the characters representing the decimal digits of n  
        SET length TO the count of elements in digits  
        SET i TO length MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND element at position i of digits GREATER THAN OR EQUAL TO element at position i PLUS one of digits  
            DECREMENT i BY one  
        END WHILE  
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
        SET j TO length MINUS one  
        WHILE element at position j of digits LESS THAN OR EQUAL TO element at position i of digits  
            DECREMENT j BY one  
        END WHILE  
        SET temporary TO element at position i of digits  
        SET element at position i of digits TO element at position j of digits  
        SET element at position j of digits TO temporary  
        SET digits TO concatenation of substring of digits from position zero TO position i PLUS one AND the reverse of substring of digits from position i PLUS one TO the end  
        SET result TO the integer value represented by concatenating all elements in digits in order  
        IF result GREATER THAN the value of two raised to the power of thirty one MINUS one  
            RETURN negative one  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
