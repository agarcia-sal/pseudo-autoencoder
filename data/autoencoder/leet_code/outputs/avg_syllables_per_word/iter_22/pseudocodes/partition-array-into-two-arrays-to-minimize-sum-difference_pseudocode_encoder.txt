CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY two  
        SET total_sum TO the SUM OF all elements IN nums  
        SET target TO total_sum DIVIDED BY two  
        SET min_diff TO positive infinity  
        
        SET left_half TO the ELEMENTS AT positions from the first UP TO position n MINUS one OF nums  
        SET right_half TO the ELEMENTS AT positions from position n TO the last ELEMENT OF nums  
        
        SET left_sums TO a list of empty dictionaries FOR each index FROM zero TO n INCLUSIVE  
        SET right_sums TO a list of empty dictionaries FOR each index FROM zero TO n INCLUSIVE  
        
        FOR each index i FROM zero TO n INCLUSIVE  
            FOR each combination combo OF i ELEMENTS FROM left_half  
                SET subset_sum TO the SUM OF elements IN combo  
                IF subset_sum DOES NOT EXIST AS a key IN left_sums AT position i THEN  
                    SET left_sums AT position i AT key subset_sum TO zero  
                END IF  
                INCREMENT left_sums AT position i AT key subset_sum BY one  
            END FOR  
            
            FOR each combination combo OF i ELEMENTS FROM right_half  
                SET subset_sum TO the SUM OF elements IN combo  
                IF subset_sum DOES NOT EXIST AS a key IN right_sums AT position i THEN  
                    SET right_sums AT position i AT key subset_sum TO zero  
                END IF  
                INCREMENT right_sums AT position i AT key subset_sum BY one  
            END FOR  
        END FOR  
        
        FOR each index i FROM zero TO n INCLUSIVE  
            SET left_values TO the sorted list OF keys FROM left_sums AT position i  
            SET right_values TO the sorted list OF keys FROM right_sums AT position n MINUS i  
            
            SET j TO zero  
            SET k TO the LENGTH OF right_values MINUS one  
            
            WHILE j IS LESS THAN the LENGTH OF left_values AND k IS GREATER THAN OR EQUAL TO zero  
                SET current_sum TO the element at position j OF left_values PLUS the element at position k OF right_values  
                SET min_diff TO the smaller value BETWEEN min_diff AND the absolute value OF total_sum MINUS two TIMES current_sum  
                
                IF current_sum IS LESS THAN target THEN  
                    INCREMENT j BY one  
                ELSE IF current_sum IS GREATER THAN target THEN  
                    DECREMENT k BY one  
                ELSE  
                    RETURN zero  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
