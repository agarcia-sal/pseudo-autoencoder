CLASS Solution  
    FUNCTION fullJustify(words maxWidth)  
        FUNCTION justify_line(line num_of_letters is_last_line)  
            IF the LENGTH OF line EQUALS one OR is_last_line IS true  
                RETURN concatenate all elements of line separated by a single space FOLLOWED BY a number of spaces EQUAL TO maxWidth MINUS num_of_letters MINUS the LENGTH OF line PLUS one  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps  
                SET extra_spaces TO total_spaces MODULO num_of_gaps  
                
                FOR index FROM zero TO num_of_gaps MINUS one  
                    IF index LESS THAN extra_spaces  
                        APPEND spaces BETWEEN words PLUS one number of spaces TO element at position index OF line  
                    ELSE  
                        APPEND spaces BETWEEN words number of spaces TO element at position index OF line  
                    END IF  
                END FOR  
                
                RETURN concatenation of all elements in line without separators  
            END IF  
        END FUNCTION  
        
        SET result TO empty list  
        SET line TO empty list  
        SET num_of_letters TO zero  
        
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth  
                APPEND the result of calling justify_line with line num_of_letters and false TO result  
                SET line TO empty list  
                SET num_of_letters TO zero  
            END IF  
            
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
        
        APPEND the result of calling justify_line with line num_of_letters and true TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
