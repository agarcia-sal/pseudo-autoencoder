CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the collection values is empty  
        RETURN no value  
    END IF  
    SET root TO creating a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO removing the element from the left of queue  
        IF i is less than the length of values AND the element at position i of values is not no value  
            SET the left attribute of node TO creating a new TreeNode with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not no value  
            SET the right attribute of node TO creating a new TreeNode with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p is no value AND q is no value  
        RETURN true  
    ELSE IF p is no value OR q is no value  
        RETURN false  
    ELSE IF the value attribute of p is NOT EQUALS the value attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with the left attribute of p and the left attribute of q AND the result of calling is_same_tree with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree with parameter root  
        IF root is no value  
            RETURN no value  
        END IF  
        SET the left attribute of root TO the result of calling pruneTree with the left attribute of root  
        SET the right attribute of root TO the result of calling pruneTree with the right attribute of root  
        IF the value attribute of root EQUALS zero AND the left attribute of root is no value AND the right attribute of root is no value  
            RETURN no value  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
