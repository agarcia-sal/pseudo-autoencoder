CLASS Solution  
    FUNCTION kInversePairs(n parameter n EQUALS integer, k parameter k EQUALS integer) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        SET dp TO a list of lists where for each number from zero to n INCLUSIVE there is a list of zeroes with the length of k PLUS one  
        SET the element at position zero of the element at position zero of dp TO one  
        FOR each integer i FROM one TO n INCLUSIVE  
            SET prefix_sum TO zero  
            FOR each integer j FROM zero TO k INCLUSIVE  
                SET prefix_sum TO prefix_sum PLUS the element at position j of the element at position i MINUS one of dp  
                APPLY modulo MOD TO prefix_sum  
                IF j GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO prefix_sum MINUS the element at position j MINUS i of the element at position i MINUS one of dp  
                    PLUS MOD  
                    APPLY modulo MOD TO prefix_sum  
                END IF  
                SET the element at position j of the element at position i of dp TO prefix_sum  
            END FOR  
        END FOR  
        RETURN the element at position n of the element at position k of dp  
    END FUNCTION  
END CLASS
