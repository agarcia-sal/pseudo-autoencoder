CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of integers from zero TO n MINUS one  
        SET self.rank TO list of ones of length n  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u THEN  
            SET element at position u of self.parent TO the result of self.find with argument element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET rootU TO the result of self.find with argument u  
        SET rootV TO the result of self.find with argument v  

        IF rootU NOT EQUALS rootV THEN  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank THEN  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank THEN  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)  
        FOR each index i AND element edge IN edges WITH their positions  
            APPEND i TO edge  
        END FOR  

        SORT edges according to the value of element at position two of each edge in ascending order  

        FUNCTION mst(exclude, include)  
            SET uf TO new instance of UnionFind with argument n  
            SET weight TO zero  

            IF include IS NOT None THEN  
                CALL uf.union with arguments element at position zero of include AND element at position one of include  
                INCREMENT weight BY element at position two of include  
            END IF  

            FOR each element u, v, w, idx IN edges  
                IF idx EQUALS exclude THEN  
                    CONTINUE to next iteration  
                END IF  
                IF uf.union with arguments u AND v RETURNS True THEN  
                    INCREMENT weight BY w  
                END IF  
            END FOR  

            SET root TO the result of uf.find with argument zero  
            FOR each integer i FROM zero TO n MINUS one  
                IF uf.find with argument i NOT EQUALS root THEN  
                    RETURN infinity  
                END IF  
            END FOR  
            RETURN weight  
        END FUNCTION  

        SET mst_weight TO the result of calling mst with no arguments  

        SET critical TO empty list  
        SET pseudo_critical TO empty list  

        FOR each u, v, w, idx IN edges  
            IF mst with argument exclude equal to idx GREATER THAN mst_weight THEN  
                APPEND idx TO critical  
            ELSE IF mst with argument include equal to tuple of u, v, w EQUALS mst_weight THEN  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR  

        RETURN list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
