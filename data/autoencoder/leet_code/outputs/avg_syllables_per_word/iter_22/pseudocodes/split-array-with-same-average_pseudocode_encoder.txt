CLASS Solution  
    FUNCTION splitArraySameAverage(nums)  
        SET n TO the LENGTH OF nums  
        SET total_sum TO the SUM of all elements IN nums  
        IF n EQUALS one  
            RETURN False  
        END IF  
        INITIALIZE dp AS a list of sets with LENGTH equal TO n DIVIDED BY two PLUS one  
        ADD zero TO the set at position zero in dp  
        FOR each num IN nums  
            FOR size FROM n DIVIDED BY two DOWN TO one  
                FOR each s IN the set at position size MINUS one in dp  
                    SET new_sum TO s PLUS num  
                    IF new_sum MULTIPLIED BY n EQUALS size MULTIPLIED BY total_sum  
                        RETURN True  
                    END IF  
                    ADD new_sum TO the set at position size in dp  
                END FOR  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
