CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix x y)  
            SET directions TO list containing element at negative one zero position element at one zero position element at zero negative one position element at zero one position element at zero zero position  
            FOR each element dx AND each element dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n THEN  
                    SET element at position nx of matrix at position ny TO one MINUS element at position nx of matrix at position ny  
                END IF  
            END FOR  
        END FUNCTION  
        
        FUNCTION matrix_to_tuple(matrix)  
            RETURN tuple of tuples each containing elements of each row of matrix  
        END FUNCTION  
        
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET target TO tuple of tuples each containing zero repeated n times repeated m times  
        SET initial_state TO matrix_to_tuple applied to mat  
        SET queue TO list containing pair of initial_state AND zero  
        SET visited TO empty set  
        ADD initial_state TO visited  
        
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND ASSIGN TO current_state AND steps  
            IF current_state EQUALS target THEN  
                RETURN steps  
            END IF  
            
            SET current_matrix TO list of lists each containing elements of each row in current_state  
            
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO deep copy of current_matrix  
                    CALL flip WITH next_matrix AND i AND j  
                    SET next_state TO matrix_to_tuple applied to next_matrix  
                    IF next_state NOT IN visited THEN  
                        ADD next_state TO visited  
                        APPEND pair of next_state AND steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
