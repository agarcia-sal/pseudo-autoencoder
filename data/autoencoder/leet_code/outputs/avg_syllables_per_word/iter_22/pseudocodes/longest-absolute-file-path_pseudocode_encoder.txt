CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO list of substrings obtained by splitting input string at each newline character  
        SET stack TO empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO count of tab characters in line  
            SET name TO substring of line from position depth plus one TO the end of line  
            
            WHILE the length of stack is GREATER THAN depth  
                REMOVE last element FROM stack  
            END WHILE  
            
            IF the length of stack IS GREATER THAN zero  
                SET current_length TO the element at last position of stack PLUS the length of name PLUS one  
            ELSE  
                SET current_length TO the length of name  
            END IF  
            
            IF name CONTAINS the period character  
                IF current_length IS GREATER THAN max_length  
                    SET max_length TO current_length  
                END IF  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
