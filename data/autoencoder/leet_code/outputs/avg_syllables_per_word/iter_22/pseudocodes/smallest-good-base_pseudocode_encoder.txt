CLASS Solution  
    FUNCTION smallestGoodBase WITH PARAMETER n  
        SET n TO the integer conversion of n  
        SET max_length TO the integer value of the logarithm base two of n PLUS one  
        FOR m FROM max_length DOWN TO two  
            SET k TO the integer value of n RAISED TO THE POWER OF one DIVIDED BY m MINUS one  
            IF k LESS THAN two  
                CONTINUE to the next iteration  
            END IF  
            SET num TO the integer value of k RAISED TO THE POWER OF m MINUS one DIVIDED BY k MINUS one  
            IF num EQUALS n  
                RETURN the string conversion of k  
            END IF  
        END FOR  
        RETURN the string conversion of n MINUS one  
    END FUNCTION  
END CLASS
