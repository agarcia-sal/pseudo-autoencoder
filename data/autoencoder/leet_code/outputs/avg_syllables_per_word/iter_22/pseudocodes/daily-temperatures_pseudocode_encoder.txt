CLASS Solution  
    FUNCTION dailyTemperatures(temperatures)  
        SET n TO the LENGTH OF temperatures  
        SET answer TO a list of zeroes of size n  
        SET stack TO an empty list of indices  
        
        FOR i FROM zero TO n MINUS one  
            WHILE stack IS NOT empty AND the element at position i of temperatures IS GREATER THAN the element at position at the last position of stack in temperatures  
                SET prev_index TO the last element REMOVED FROM stack  
                SET the element at position prev_index of answer TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
