CLASS Solution  
    FUNCTION stoneGame WITH parameter piles  
        FUNCTION dfs WITH parameters i j  
            IF i GREATER THAN j  
                RETURN zero  
            END IF  
            SET result_from_left TO element at position i of piles MINUS dfs WITH i PLUS one AND j  
            SET result_from_right TO element at position j of piles MINUS dfs WITH i AND j MINUS one  
            RETURN the GREATER OF result_from_left AND result_from_right  
        END FUNCTION  
        SET final_result TO dfs WITH zero AND the LENGTH OF piles MINUS one  
        RETURN final_result GREATER THAN zero  
    END FUNCTION  
END CLASS
