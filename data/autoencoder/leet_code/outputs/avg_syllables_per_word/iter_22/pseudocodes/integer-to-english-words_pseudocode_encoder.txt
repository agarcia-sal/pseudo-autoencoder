CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  
  
        SET below_twenty TO list of words where first element is empty word followed by words One through Nineteen in order  
        SET tens TO list of words where first two elements are empty words followed by Twenty, Thirty, Forty, Fifty, Sixty, Seventy, Eighty, Ninety in order  
        SET thousands TO list of words where first element is empty word followed by Thousand, Million, Billion in order  
  
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty word  
            ELSE IF n LESS THAN twenty  
                RETURN the element at position n of below_twenty concatenated with a space character  
            ELSE IF n LESS THAN one hundred  
                RETURN the element at position n DIVIDED BY ten of tens concatenated with a space THEN concatenated with the result of calling helper with the remainder of n MODULO ten  
            ELSE  
                RETURN the element at position n DIVIDED BY one hundred of below_twenty concatenated with the word Hundred and a space THEN concatenated with the result of calling helper with the remainder of n MODULO one hundred  
            END IF  
        END FUNCTION  
  
        SET result TO empty word  
        SET i TO zero  
  
        WHILE num GREATER THAN zero  
            IF remainder of num MODULO one thousand NOT EQUALS zero  
                SET result TO result of helper with remainder of num MODULO one thousand concatenated with the element at position i of thousands concatenated with a space then concatenated with current result  
            END IF  
            SET num TO num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  
  
        RETURN result with all leading and trailing spaces removed  
    END FUNCTION  
END CLASS
