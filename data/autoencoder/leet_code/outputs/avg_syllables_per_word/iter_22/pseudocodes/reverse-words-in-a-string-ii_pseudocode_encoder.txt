CLASS Solution  
    FUNCTION reverseWords(s)  
        CALL reverse ENTIRE list s  
        SET start TO zero  
        FOR end FROM zero TO the LENGTH OF s MINUS one  
            IF element at position end of s EQUALS a space character  
                CALL reverse SUBLIST OF s FROM position start TO position end MINUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        CALL reverse SUBLIST OF s FROM position start TO the LAST position  
    END FUNCTION  
END CLASS
