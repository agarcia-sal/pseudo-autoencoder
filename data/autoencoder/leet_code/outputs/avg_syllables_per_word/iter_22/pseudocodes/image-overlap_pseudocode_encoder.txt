CLASS Solution  
    FUNCTION largestOverlap(img1, img2)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF element at position i of img1 and then element at position j EQUALS one  
                    APPEND the pair of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO n MINUS one  
                IF element at position i of img2 and then element at position j EQUALS one  
                    APPEND the pair of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO an empty mapping DEFAULTING TO zero  
        FOR each pair of x1 AND y1 IN points1  
            FOR each pair of x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value at key pair of dx AND dy IN overlap_count BY one  
            END FOR  
        END FOR  
        SET maximum_value TO zero  
        FOR each value IN overlap_count  
            IF value GREATER THAN maximum_value  
                SET maximum_value TO value  
            END IF  
        END FOR  
        RETURN maximum_value  
    END FUNCTION  
END CLASS
