CLASS Solution  
    FUNCTION checkWays(pairs)  
        SET graph TO a default dictionary with empty sets as default values  
        FOR each element named x COMMA y IN pairs  
            APPEND y TO the set associated with x in graph  
            APPEND x TO the set associated with y in graph  
        END FOR  
        
        SET max_degree TO the element in graph having the greatest size of the associated set  
        SET n TO the number of keys in graph  
        
        IF the size of the set associated with max_degree in graph NOT EQUALS n MINUS one THEN  
            RETURN zero  
        END IF  
        
        FUNCTION can_be_parent(u, v)  
            RETURN for all neighbor in the set associated with v in graph the condition neighbor is in the set associated with u in graph OR neighbor EQUALS u holds  
        END FUNCTION  
        
        SET ways TO one  
        
        FOR each element u IN graph  
            IF u EQUALS max_degree THEN  
                CONTINUE to next iteration  
            END IF  
            
            SET parent TO None  
            FOR each element v IN the set associated with u in graph  
                IF v IN graph AND the size of the set associated with v in graph GREATER THAN OR EQUAL TO the size of the set associated with u in graph AND can_be_parent v u is true THEN  
                    IF parent IS None OR the size of the set associated with v in graph LESS THAN the size of the set associated with parent in graph THEN  
                        SET parent TO v  
                    END IF  
                END IF  
            END FOR  
            
            IF parent IS None THEN  
                RETURN zero  
            END IF  
            
            IF the size of the set associated with parent in graph EQUALS the size of the set associated with u in graph THEN  
                SET ways TO two  
            END IF  
        END FOR  
        
        RETURN ways  
    END FUNCTION  
END CLASS
