CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  
  
FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode with the element at position one of values and next set to None  
    SET p TO head  
    FOR each val IN the elements from position two TO the last element of values  
        SET node TO a new ListNode with val and next set to None  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  
  
FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the condition that the val element of p1 EQUALS the val element of p2 AND the result of the call to is_same_list with the next element of p1 AND the next element of p2  
END FUNCTION  
  
CLASS Solution  
    FUNCTION swapPairs(self, head)  
        SET dummy TO a new ListNode with zero and next set to None  
        SET the next element of dummy TO head  
        SET current TO dummy  
        WHILE the next element of current DOES NOT EQUAL None AND the next element of the next element of current DOES NOT EQUAL None  
            SET first TO the next element of current  
            SET second TO the next element of the next element of current  
            SET the next element of current TO second  
            SET the next element of first TO the next element of second  
            SET the next element of second TO first  
            SET current TO the next element of the next element of current  
        END WHILE  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
