CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator_of_t TO iterator over t  
            RETURN boolean indicating whether all characters in s are contained in iterator_of_t following the order of s  
        END FUNCTION  
        
        SORT strs by descending order of the length of each element PLUS by lexicographical order of each element  
        
        FOR each index i AND element word IN strs  
            SET unique TO true  
            FOR each index j AND element other IN strs  
                IF i NOT EQUALS j AND is_subsequence of word IN other holds true  
                    SET unique TO false  
                    BREAK inner loop  
                END IF  
            END FOR  
            IF unique is true  
                RETURN length of word  
            END IF  
        END FOR  
        
        RETURN the negation of one  
    END FUNCTION  
END CLASS
