CLASS ListNode  
    FUNCTION __init__(self val DEFAULT zero next DEFAULT none)  
        SET self val TO val  
        SET self next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO new instance of ListNode with value as the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values from position one TO the last position  
        SET node TO new instance of ListNode with value as val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN TRUE  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN FALSE  
    END IF  
    RETURN the condition that the value of p1 EQUALS the value of p2 AND the call to is_same_list with the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self head)  
        IF head EQUALS none OR the next element of head EQUALS none  
            RETURN head  
        END IF  
        SET dummy TO new instance of ListNode with value as negative five thousand and one  
        SET the next element of dummy TO head  
        SET current TO the next element of head  
        SET the next element of head TO none  
        WHILE current holds  
            SET next_node TO the next element of current  
            SET prev TO dummy  
            WHILE the next element of prev holds AND the value of the next element of prev LESS THAN the value of current  
                SET prev TO the next element of prev  
            END WHILE  
            SET the next element of current TO the next element of prev  
            SET the next element of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
