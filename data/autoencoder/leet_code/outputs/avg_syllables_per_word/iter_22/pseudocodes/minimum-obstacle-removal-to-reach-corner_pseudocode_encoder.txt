CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO a list of elements each describing a pair of zero and one, one and zero, zero and minus one, minus one and zero  
        SET min_heap TO a list containing a tuple of zero zero zero representing cost and position coordinates  
        SET visited TO a set containing a tuple of zero zero representing visited position coordinates  
        
        WHILE min_heap HAS elements  
            REMOVE from min_heap the element with the smallest cost and UNPACK it INTO cost x y  
            
            IF the tuple of x y EQUALS the tuple of m MINUS one and n MINUS one THEN  
                RETURN cost  
            END IF  
            
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the tuple of nx ny NOT IN visited THEN  
                    ADD the tuple of nx ny TO visited  
                    SET new_cost TO cost PLUS element at position nx of grid THEN element at position ny of that element  
                    ADD the tuple of new_cost nx ny TO min_heap using a priority insertion method maintaining order by cost  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
