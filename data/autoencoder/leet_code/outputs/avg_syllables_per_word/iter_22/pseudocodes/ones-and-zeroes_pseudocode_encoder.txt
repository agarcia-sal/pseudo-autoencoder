CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        SET dp TO a two dimensional list of zeros with the number of rows equal to m PLUS one and the number of columns equal to n PLUS one  
        
        FOR each s IN strs  
            SET count_0 TO the number of elements equal to zero character in s  
            SET count_1 TO the number of elements equal to one character in s  
            
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp element at position i and j TO the greater value BETWEEN dp element at position i and j AND dp element at position i MINUS count_0 and j MINUS count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp element at position m and n  
    END FUNCTION  
END CLASS
