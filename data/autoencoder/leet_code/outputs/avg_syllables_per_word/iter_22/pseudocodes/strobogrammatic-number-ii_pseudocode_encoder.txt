CLASS Solution  
    FUNCTION findStrobogrammatic WITH PARAMETER n  
        FUNCTION dfs WITH PARAMETER u  
            IF u EQUALS zero  
                RETURN list CONTAINING empty string  
            END IF  
            IF u EQUALS one  
                RETURN list CONTAINING string zero AND string one AND string eight  
            END IF  
            SET ans TO empty list  
            FOR each v IN CALL dfs WITH PARAMETER u MINUS two  
                FOR each pair OF left_character AND right_character IN list CONTAINING string one one AND string eight eight AND string six nine AND string nine six  
                    APPEND concatenation OF left_character PLUS v PLUS right_character TO ans  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND concatenation OF string zero PLUS v PLUS string zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN CALL dfs WITH PARAMETER n  
    END FUNCTION  
END CLASS
