CLASS Solution  
    FUNCTION isBipartite(graph)  
        SET n TO the LENGTH OF graph  
        SET color TO a collection of zero values with the LENGTH OF n  
        
        FUNCTION dfs(node, c)  
            SET the element at position node of color TO c  
            FOR each neighbor IN the element at position node of graph  
                IF the element at position neighbor of color EQUALS c THEN  
                    RETURN False  
                END IF  
                IF the element at position neighbor of color EQUALS zero THEN  
                    IF the result of dfs with neighbor AND the NEGATION OF c DOES NOT EQUAL True THEN  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
            RETURN True  
        END FUNCTION  
        
        FOR i FROM zero TO n MINUS one  
            IF the element at position i of color EQUALS zero THEN  
                IF the result of dfs with i AND one DOES NOT EQUAL True THEN  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
