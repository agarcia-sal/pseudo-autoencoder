CLASS Solution  
    FUNCTION reorderedPowerOf2(n)  
        SET power_of_2_sorted_digits TO a new empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 LESS THAN OR EQUAL TO one billion  
            ADD the tuple of the sorted list of characters in the string representation of power_of_2 TO power_of_2_sorted_digits  
            SET power_of_2 TO power_of_2 MULTIPLIED BY two  
        END WHILE  
        SET sorted_n_digits TO the tuple of the sorted list of characters in the string representation of n  
        IF sorted_n_digits IS IN power_of_2_sorted_digits  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
