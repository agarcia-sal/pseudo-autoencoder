CLASS Solution  
    FUNCTION boldWords(words, s)  
        SET bold TO list of boolean values False with LENGTH OF s  
        FOR each word IN words  
            SET start TO position of first occurrence of word in s  
            WHILE start NOT EQUALS negative one  
                FOR i FROM start TO start PLUS LENGTH OF word MINUS one  
                    SET element at position i of bold TO True  
                END FOR  
                SET start TO position of next occurrence of word in s starting from start PLUS one  
            END WHILE  
        END FOR  
        SET result TO empty list  
        SET i TO zero  
        WHILE i LESS THAN LENGTH OF s  
            IF element at position i of bold EQUALS True  
                APPEND the substring representing opening bold tag TO result  
                WHILE i LESS THAN LENGTH OF s AND element at position i of bold EQUALS True  
                    APPEND element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the substring representing closing bold tag TO result  
            ELSE  
                APPEND element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
