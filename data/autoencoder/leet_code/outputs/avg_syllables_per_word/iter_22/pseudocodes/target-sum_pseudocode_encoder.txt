CLASS Solution  
    FUNCTION findTargetSumWays(nums target)  
        FUNCTION dfs(index current_sum memo)  
            IF index EQUALS the LENGTH OF nums THEN  
                IF current_sum EQUALS target THEN  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the TUPLE of index AND current_sum IS IN memo THEN  
                RETURN the ELEMENT CORRESPONDING TO the TUPLE of index AND current_sum IN memo  
            END IF  
            SET positive TO the RESULT OF dfs with PARAMETERS index PLUS one current_sum PLUS the ELEMENT at POSITION index of nums memo  
            SET negative TO the RESULT OF dfs with PARAMETERS index PLUS one current_sum MINUS the ELEMENT at POSITION index of nums memo  
            SET the ELEMENT CORRESPONDING TO the TUPLE of index AND current_sum IN memo TO positive PLUS negative  
            RETURN the ELEMENT CORRESPONDING TO the TUPLE of index AND current_sum IN memo  
        END FUNCTION  
        RETURN the RESULT OF dfs with PARAMETERS zero zero an EMPTY DICTIONARY  
    END FUNCTION  
END CLASS
