CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO an empty list  
        SET lower_bound TO negative infinity  
        
        FOR each value IN preorder  
            IF value LESS THAN lower_bound  
                RETURN False  
            END IF  
            
            WHILE the LENGTH OF stack is greater than zero AND value GREATER THAN the element at the last position of stack  
                SET lower_bound TO the last element removed from stack  
            END WHILE  
            
            APPEND value TO stack  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
