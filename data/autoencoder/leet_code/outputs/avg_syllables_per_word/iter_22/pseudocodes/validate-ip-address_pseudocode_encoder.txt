CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS ONLY OF DIGITS  
                SET num TO the integer value of block  
                IF num IS LESS THAN OR EQUAL TO two hundred fifty five AND the string representation of num EQUALS block  
                    RETURN True  
                ELSE  
                    RETURN False  
                END IF  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF the LENGTH OF block IS GREATER THAN OR EQUAL TO one AND the LENGTH OF block IS LESS THAN OR EQUAL TO four  
                TRY  
                    CONVERT block TO hexadecimal integer  
                    RETURN True  
                EXCEPT conversion fails  
                    RETURN False  
                END TRY  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO the list of substrings of queryIP SPLIT BY the period character  
        SET parts_ipv6 TO the list of substrings of queryIP SPLIT BY the colon character  

        IF the LENGTH OF parts_ipv4 EQUALS four AND for every part IN parts_ipv4 the call to is_valid_ipv4_block WITH part RETURNS True  
            RETURN the string IPv4  
        ELSE IF the LENGTH OF parts_ipv6 EQUALS eight AND for every part IN parts_ipv6 the call to is_valid_ipv6_block WITH part RETURNS True  
            RETURN the string IPv6  
        ELSE  
            RETURN the string Neither  
        END IF  
    END FUNCTION  
END CLASS
