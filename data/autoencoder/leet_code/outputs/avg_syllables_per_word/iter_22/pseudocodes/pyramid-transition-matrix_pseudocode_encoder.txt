CLASS Solution  
    FUNCTION pyramidTransition(bottom, allowed)  
        SET allowed_map TO an empty mapping from character to mapping from character to list of characters  
        FOR each rule IN allowed  
            SET left TO element at position zero of rule  
            SET right TO element at position one of rule  
            SET top TO element at position two of rule  
            APPEND top TO the list in allowed_map associated with left and right  
        END FOR  
        
        FUNCTION can_build_pyramid(current_bottom, current_top)  
            IF the LENGTH OF current_bottom EQUALS one  
                RETURN true  
            END IF  
            IF the LENGTH OF current_top EQUALS the LENGTH OF current_bottom MINUS one  
                RETURN the result of calling can_build_pyramid with current_top and an empty string  
            END IF  
            
            SET position TO the LENGTH OF current_top  
            SET left TO element at position position of current_bottom  
            SET right TO element at position position PLUS one of current_bottom  
            IF allowed_map CONTAINS left AND the mapping for left CONTAINS right  
                FOR each top IN the list allowed_map associated with left and right  
                    IF the result of calling can_build_pyramid with current_bottom and current_top PLUS top EQUALS true  
                        RETURN true  
                    END IF  
                END FOR  
            END IF  
            RETURN false  
        END FUNCTION  
        
        RETURN the result of calling can_build_pyramid with bottom and an empty string  
    END FUNCTION  
END CLASS
