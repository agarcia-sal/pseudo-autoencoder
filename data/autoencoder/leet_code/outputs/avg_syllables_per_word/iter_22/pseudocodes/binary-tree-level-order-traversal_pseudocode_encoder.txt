CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at position one of values  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the first element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.left TO new TreeNode with value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.right TO new TreeNode with value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments p.left AND q.left AND the result of is_same_tree with arguments p.right AND q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root EQUALS none  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO new double ended queue containing root  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO empty list  
            FOR underscore FROM one TO level_size  
                SET node TO the first element removed from the left side of queue  
                APPEND the value of node TO current_level  
                IF node.left NOT EQUALS none  
                    APPEND node.left TO queue  
                END IF  
                IF node.right NOT EQUALS none  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
