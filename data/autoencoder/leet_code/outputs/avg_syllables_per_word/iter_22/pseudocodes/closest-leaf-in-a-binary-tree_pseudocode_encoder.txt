CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the list values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode created with the element at position one of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO the element removed from the left side of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode created with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode created with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value field of p IS NOT EQUALS the value field of q  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with the left field of p and the left field of q AND the result of calling is_same_tree with the right field of p and the right field of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self, root, k)  
        IF the left field of root IS none AND the right field of root IS none  
            RETURN the value field of root  
        END IF  
        SET graph TO a new default dictionary where each value is a list  
        FUNCTION dfs(node, parent EQUALS none)  
            IF parent IS NOT none  
                APPEND the value field of parent TO the list at key equal to the value field of node in graph  
                APPEND the value field of node TO the list at key equal to the value field of parent in graph  
            END IF  
            IF the left field of node IS NOT none  
                CALL dfs with the left field of node AND node  
            END IF  
            IF the right field of node IS NOT none  
                CALL dfs with the right field of node AND node  
            END IF  
        END FUNCTION  
        CALL dfs with root  
        SET queue TO a new deque initialized with k  
        SET visited TO an empty set  
        WHILE the queue IS NOT empty  
            SET current TO the element removed from the left side of queue  
            ADD current TO visited  
            IF the LENGTH OF the list at key current in graph IS EQUALS one AND current IS NOT EQUALS the value field of root  
                RETURN current  
            END IF  
            FOR each neighbor IN the list at key current in graph  
                IF neighbor IS NOT IN visited  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
