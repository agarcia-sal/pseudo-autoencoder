CLASS Solution  
    FUNCTION combinationSum2(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                IF i GREATER THAN start AND element at position i of candidates EQUALS element at position i MINUS one of candidates  
                    CONTINUE  
                END IF  
                CALL backtrack WITH i PLUS one AS start PARAMETER AND target MINUS element at position i of candidates AS target PARAMETER AND path PLUS element at position i of candidates AS path PARAMETER  
            END FOR  
        END FUNCTION  
        SORT candidates IN non-decreasing order  
        SET result TO an empty list  
        CALL backtrack WITH zero AS start PARAMETER AND target AS target PARAMETER AND an empty list AS path PARAMETER  
        RETURN result  
    END FUNCTION  
END CLASS
