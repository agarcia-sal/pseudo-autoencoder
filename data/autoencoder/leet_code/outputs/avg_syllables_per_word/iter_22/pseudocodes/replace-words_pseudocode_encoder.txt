CLASS Solution  
    FUNCTION replaceWords WITH PARAMETERS dictionary sentence  
        SORT dictionary BY the LENGTH OF each element FROM the SMALLEST TO the LARGEST  
        SET words TO the list of individual elements obtained by SPLITTING sentence BY spaces  
        
        FUNCTION replace WITH PARAMETER word  
            FOR each root IN dictionary  
                IF word STARTS WITH root THEN  
                    RETURN root  
                END IF  
            END FOR  
            RETURN word  
        END FUNCTION  
        
        SET replaced_words TO an empty list  
        FOR each word IN words  
            APPEND the result of replace WITH word TO replaced_words  
        END FOR  
        
        SET result_sentence TO the concatenation of all elements in replaced_words SEPARATED BY spaces  
        RETURN result_sentence  
    END FUNCTION  
END CLASS
