CLASS Solution  
    FUNCTION frogPosition(n, edges, t, target)  
        IF n EQUALS one  
            RETURN one point zero  
        END IF  

        SET graph TO a mapping with default value as an empty list  
        FOR each pair u and v IN edges  
            APPEND v TO the list at key u in graph  
            APPEND u TO the list at key v in graph  
        END FOR  

        SET queue TO a double-ended queue containing the tuple of current_node as one current_probability as one point zero and current_time as zero  
        SET visited TO a set containing one  

        WHILE queue has elements  
            REMOVE and SET current_node current_probability current_time FROM the leftmost element of queue  

            IF current_time EQUALS t OR current_node NOT EQUALS one AND the LENGTH of the list at key current_node in graph EQUALS one  
                IF current_node EQUALS target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE to the next iteration  
                END IF  
            END IF  

            SET possible_moves TO the LENGTH of the list at key current_node in graph MINUS if current_node NOT EQUALS one THEN one ELSE zero  

            FOR each neighbor IN the list at key current_node in graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND tuple of neighbor current_probability DIVIDED BY possible_moves current_time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  

        RETURN zero point zero  
    END FUNCTION  
END CLASS
