CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO a new default dictionary with default value zero  
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain INTO two parts count_text AND domain_text by the first whitespace  
            SET count TO the integer value of count_text  
            SPLIT domain_text INTO a list of subdomains separated by the word dot  
            FOR index FROM zero TO the length of subdomains MINUS one  
                JOIN the elements from position index TO the last element of subdomains USING the word dot INTO subdomain  
                INCREMENT the value associated with subdomain in domain_count BY count  
            END FOR  
        END FOR  
        CREATE a new empty list result  
        FOR each pair of domain_key AND count_value IN the items of domain_count  
            SET formatted_string TO the concatenation of the string representation of count_value FOLLOWED BY a space FOLLOWED BY domain_key  
            APPEND formatted_string TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
