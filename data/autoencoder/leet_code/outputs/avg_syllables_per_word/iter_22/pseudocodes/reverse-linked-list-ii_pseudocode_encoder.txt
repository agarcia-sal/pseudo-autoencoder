CLASS ListNode  
    FUNCTION __init__ WITH parameters self val SET TO zero next SET TO none  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH parameter values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two TO the last element  
        SET node TO a new ListNode with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH parameters p1 p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND call is_same_list WITH the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween WITH parameters head left right  
        IF head EQUALS none OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO a new ListNode with zero  
        SET the next element of dummy TO head  
        SET prev TO dummy  
        FOR underscore FROM one TO left MINUS one  
            SET prev TO the next element of prev  
        END FOR  
        SET current TO the next element of prev  
        FOR underscore FROM one TO right MINUS left  
            SET next_node TO the next element of current  
            SET the next element of current TO the next element of next_node  
            SET the next element of next_node TO the next element of prev  
            SET the next element of prev TO next_node  
        END FOR  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
