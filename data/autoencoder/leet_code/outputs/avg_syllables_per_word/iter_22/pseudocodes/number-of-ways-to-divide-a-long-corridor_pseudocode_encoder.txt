CLASS Solution  
    FUNCTION numberOfWays(corridor)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET seats TO empty list  
        FOR each index AND character IN corridor enumeration  
            IF character EQUALS capital S  
                APPEND index TO seats  
            END IF  
        END FOR  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR each index FROM three TO the LENGTH OF seats STEP two  
            SET difference TO element at position index of seats MINUS element at position index MINUS one of seats  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
