CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val equal to element at position one of values AND left equal to none AND right equal to none  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode with val equal to element at position i of values AND left equal to none AND right equal to none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode with val equal to element at position i of values AND left equal to none AND right equal to none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION longestUnivaluePath(self, root)  
        FUNCTION dfs(node)  
            IF node IS none  
                RETURN zero AND zero  
            END IF  
            SET left_length AND left_value TO dfs(node.left)  
            SET right_length AND right_value TO dfs(node.right)  
            SET left_arrow TO zero  
            SET right_arrow TO zero  
            IF node.left IS NOT none AND node.left.val EQUALS node.val  
                SET left_arrow TO left_length PLUS one  
            END IF  
            IF node.right IS NOT none AND node.right.val EQUALS node.val  
                SET right_arrow TO right_length PLUS one  
            END IF  
            SET self.max_length TO the GREATER OF self.max_length AND left_arrow PLUS right_arrow  
            RETURN the GREATER OF left_arrow AND right_arrow AND node.val  
        END FUNCTION  
        SET self.max_length TO zero  
        CALL dfs WITH root  
        RETURN self.max_length  
    END FUNCTION  
END CLASS
