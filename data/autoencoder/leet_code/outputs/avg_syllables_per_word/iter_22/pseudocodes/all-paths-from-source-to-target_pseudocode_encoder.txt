CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        SET q TO a new double ended queue containing a list with the element zero  
        SET ans TO an empty list  
        WHILE the LENGTH OF q GREATER THAN zero holds  
            SET path TO the first element REMOVED FROM q  
            SET u TO the last element of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the loop  
            END IF  
            FOR each v IN the element at position u of graph  
                APPEND path PLUS a list containing v TO q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
