CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO empty collection of key to value pairs  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each end and char IN the enumeration of s  
            IF char EXISTS IN char_index_map AND the value at char in char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO the value at char in char_index_map PLUS one  
            END IF  
            SET the value at char in char_index_map TO end  
            IF max_length IS LESS THAN max_length PLUS end MINUS start PLUS one MINUS max_length  
                SET max_length TO max_length PLUS end MINUS start PLUS one MINUS max_length  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
