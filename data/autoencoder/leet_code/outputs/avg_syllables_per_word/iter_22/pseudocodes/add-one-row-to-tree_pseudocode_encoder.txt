CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with value as the element at position one of values  
    SET i TO one  
    CREATE an empty queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        REMOVE the first element FROM queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode with value as the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode with value as the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left AND q.left AND the result of is_same_tree with parameters p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION addOneRow(self, root, val, depth)  
        IF depth EQUALS one  
            RETURN a new TreeNode with value val AND left subtree as root  
        END IF  
        CREATE an empty queue  
        APPEND the tuple of root AND one TO queue  
        WHILE the queue is not empty  
            REMOVE the first tuple FROM queue AND ASSIGN TO node AND level  
            IF level EQUALS depth MINUS one  
                SET node.left TO a new TreeNode with value val AND left subtree as node.left  
                SET node.right TO a new TreeNode with value val AND right subtree as node.right  
            ELSE IF level LESS THAN depth MINUS one  
                IF node.left IS NOT none  
                    APPEND the tuple of node.left AND level PLUS one TO queue  
                END IF  
                IF node.right IS NOT none  
                    APPEND the tuple of node.right AND level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
