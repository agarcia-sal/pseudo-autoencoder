CLASS Solution  
    FUNCTION removeKdigits(num k)  
        SET stack TO an empty list  
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND the LENGTH OF stack GREATER THAN zero AND the last element at position LENGTH OF stack MINUS one of stack GREATER THAN digit  
                REMOVE the last element FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        IF k EQUALS zero  
            SET final_stack TO stack  
        ELSE  
            SET final_stack TO all elements FROM start TO the LENGTH OF stack MINUS k of stack  
        END IF  
        SET result TO the concatenation of all elements in final_stack with leading zeros removed  
        IF result IS NOT empty  
            RETURN result  
        ELSE  
            RETURN zero as a string  
        END IF  
    END FUNCTION  
END CLASS
