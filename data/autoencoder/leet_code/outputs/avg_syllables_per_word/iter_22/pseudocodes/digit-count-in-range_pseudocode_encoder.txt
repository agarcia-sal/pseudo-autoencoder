CLASS Solution  
    FUNCTION digitsCount(d, low, high)  
        FUNCTION count_digit(n, digit)  
            SET count TO zero  
            SET power_of_10 TO one  
            SET current TO n  
            WHILE current GREATER THAN zero  
                SET last TO element at position representing remainder of current DIVIDED BY ten of current  
                SET current TO element at position representing integer division of current BY ten of current  
                SET full_cycles TO current  
                IF digit EQUALS zero  
                    DECREMENT full_cycles BY one  
                END IF  
                INCREMENT count BY full_cycles MULTIPLIED BY power_of_10  
                IF last GREATER THAN digit  
                    INCREMENT count BY power_of_10  
                ELSE IF last EQUALS digit  
                    INCREMENT count BY remainder of n DIVIDED BY power_of_10 PLUS one  
                END IF  
                SET power_of_10 TO power_of_10 MULTIPLIED BY ten  
            END WHILE  
            RETURN count  
        END FUNCTION  

        RETURN count_digit(high, d) MINUS count_digit(low MINUS one, d)  
    END FUNCTION  
END CLASS
