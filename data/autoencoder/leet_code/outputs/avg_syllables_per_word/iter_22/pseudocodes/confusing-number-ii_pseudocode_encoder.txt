CLASS Solution  
    FUNCTION confusingNumberII(n)  
        SET rotate_map TO mapping where zero maps to zero and one maps to one and six maps to nine and eight maps to eight and nine maps to six  
        SET valid_digits TO list of zero one six eight nine  
        
        FUNCTION is_confusing(num)  
            SET original_num TO num  
            SET rotated_num TO zero  
            WHILE num GREATER THAN zero holds  
                SET digit TO the remainder of num DIVIDED BY ten  
                SET rotated_num TO rotated_num MULTIPLIED BY ten PLUS the value from rotate_map corresponding to digit  
                SET num TO the quotient of num DIVIDED BY ten ignoring the remainder  
            END WHILE  
            RETURN rotated_num NOT EQUALS original_num  
        END FUNCTION  
        
        FUNCTION count_confusing_numbers(limit current)  
            IF current GREATER THAN limit THEN  
                RETURN zero  
            END IF  
            SET count TO one if current NOT EQUALS zero AND is_confusing(current) is true otherwise zero  
            FOR each digit IN valid_digits  
                SET new_number TO current MULTIPLIED BY ten PLUS digit  
                IF new_number EQUALS zero THEN  
                    CONTINUE to next iteration  
                END IF  
                INCREMENT count BY the result of count_confusing_numbers(limit new_number)  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN the result of count_confusing_numbers with n as argument  
    END FUNCTION  
END CLASS
