CLASS Solution  
    FUNCTION canDivideIntoSubsequences(nums, k)  
        SET frequency_dictionary TO the frequency count of each element in nums  
        SET maximum_frequency TO the largest value in the frequency_dictionary  
        SET length_of_nums TO the count of elements in nums  
        IF length_of_nums DIVIDED BY k IS GREATER THAN OR EQUAL TO maximum_frequency  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
