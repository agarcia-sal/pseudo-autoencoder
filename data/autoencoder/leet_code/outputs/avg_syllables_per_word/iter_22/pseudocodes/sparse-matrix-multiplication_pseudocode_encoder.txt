CLASS Solution  
    FUNCTION multiply(mat1 mat2)  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF the element at position one of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF the element at position one of mat2  
        SET result TO an empty list  
        FOR each index i FROM one TO m  
            SET row TO an empty list  
            FOR each index _ FROM one TO n  
                APPEND zero TO row  
            END FOR  
            APPEND row TO result  
        END FOR  
        FOR each index i FROM one TO m  
            FOR each index j FROM one TO n  
                FOR each index p FROM one TO k  
                    SET element at position j of element at position i of result TO element at position j of element at position i of result PLUS element at position p of element at position i of mat1 MULTIPLIED BY element at position j of element at position p of mat2  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
