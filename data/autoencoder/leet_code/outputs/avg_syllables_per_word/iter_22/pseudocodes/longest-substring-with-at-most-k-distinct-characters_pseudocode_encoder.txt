CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct WITH PARAMETERS s k  
        IF k EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET char_map TO empty collection of key value pairs  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            ASSIGN reference TO right TO element at position right of s IN char_map  
            
            IF the LENGTH OF char_map GREATER THAN k THEN  
                SET leftmost_char TO the key in char_map WITH THE MINIMUM value  
                SET left TO the value associated with leftmost_char IN char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET max_length TO the GREATER VALUE BETWEEN max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
