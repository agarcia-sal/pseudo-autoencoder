CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN zero  
            SET sign TO negative one  
        ELSE  
            SET sign TO one  
        END IF  
        SET absolute_value_string TO string representation of absolute value of x  
        SET reversed_string TO the characters of absolute_value_string in reverse order  
        SET reversed_x TO integer conversion of reversed_string  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative two RAISED TO the power of thirty one OR result GREATER THAN two RAISED TO the power of thirty one MINUS one  
            RETURN zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
