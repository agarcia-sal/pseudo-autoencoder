CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the absence of values holds  
        RETURN the absence of an object  
    END IF  
    SET head TO a new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the last position of values  
        SET node TO a new ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 is the absence of an object AND p2 is the absence of an object  
        RETURN truth  
    END IF  
    IF the absence of p1 OR the absence of p2 holds  
        RETURN falsehood  
    END IF  
    RETURN the condition that the value attribute of p1 EQUALS the value attribute of p2  
        AND the result of is_same_list with parameters the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists with parameter lists  
        SET min_heap TO an empty collection  
        FOR each pair of index i AND element lst IN lists with element enumeration  
            IF lst exists  
                APPLY the heap push operation to min_heap and the triple consisting of the value attribute of lst, i, AND lst  
            END IF  
        END FOR  
        SET dummy TO a new ListNode with default initialization  
        SET current TO dummy  
        WHILE min_heap is not empty  
            SET the triple of val i node TO the heap pop operation applied to min_heap  
            SET the next attribute of current TO node  
            SET current TO the next attribute of current  
            IF the next attribute of node exists  
                APPLY the heap push operation to min_heap and the triple consisting of the value attribute of the next attribute of node, i, AND the next attribute of node  
            END IF  
        END WHILE  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
