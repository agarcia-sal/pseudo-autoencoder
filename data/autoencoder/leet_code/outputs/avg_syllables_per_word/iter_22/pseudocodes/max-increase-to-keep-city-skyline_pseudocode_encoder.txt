CLASS Solution  
    FUNCTION maxIncreaseKeepingSkyline(grid)  
        SET n TO the LENGTH OF grid  
        SET row_maxes TO an empty list  
        FOR each row IN grid  
            APPEND the MAXIMUM value IN row TO row_maxes  
        END FOR  
        SET col_maxes TO an empty list  
        FOR each column formed by taking element at position of each row across grid  
            APPEND the MAXIMUM value IN that column TO col_maxes  
        END FOR  
        SET total_increase TO zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET max_height TO the LESSER OF the element at position i of row_maxes AND the element at position j of col_maxes  
                IF max_height GREATER THAN the element at position i of grid AND the element at position j of grid  
                    INCREMENT total_increase BY max_height MINUS the element at position i of grid AT position j  
                ELSE IF max_height LESS THAN OR EQUAL TO the element at position i of grid AT position j  
                    INCREMENT total_increase BY zero  
                END IF  
            END FOR  
        END FOR  
        RETURN total_increase  
    END FUNCTION  
END CLASS
