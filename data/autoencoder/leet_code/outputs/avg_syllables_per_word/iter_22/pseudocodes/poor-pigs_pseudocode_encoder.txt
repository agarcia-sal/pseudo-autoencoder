CLASS Solution  
    FUNCTION poorPigs(buckets minutesToDie minutesToTest)  
        SET number_of_states TO the INTEGER DIVISION OF minutesToTest BY minutesToDie PLUS one  
        SET logarithm_of_buckets TO the logarithm of buckets WITH base natural  
        SET logarithm_of_states TO the logarithm of number_of_states WITH base natural  
        SET minimum_number_of_pigs TO the smallest integer GREATER THAN OR EQUAL TO the division of logarithm_of_buckets BY logarithm_of_states  
        RETURN minimum_number_of_pigs  
    END FUNCTION  
END CLASS
