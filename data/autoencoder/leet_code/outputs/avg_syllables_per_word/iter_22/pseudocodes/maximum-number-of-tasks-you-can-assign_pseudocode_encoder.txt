CLASS Solution  
    FUNCTION maxTaskAssign(tasks LIST OF INTEGER, workers LIST OF INTEGER, pills INTEGER, strength INTEGER) RETURNS INTEGER  
        SORT tasks IN ASCENDING ORDER  
        SORT workers IN ASCENDING ORDER  
          
        FUNCTION canAssignTasks(k INTEGER) RETURNS BOOLEAN  
            SET available_workers TO last k elements of workers  
            SET pills_used TO zero  
              
            FOR i FROM k MINUS one DOWN TO zero  
                SET task TO element at position i of tasks  
                SET worker_idx TO the position of the smallest element IN available_workers THAT IS GREATER THAN OR EQUAL TO task  
                IF worker_idx IS LESS THAN the LENGTH OF available_workers THEN  
                    REMOVE element at position worker_idx FROM available_workers  
                ELSE IF pills_used IS LESS THAN pills THEN  
                    SET worker_idx TO the position of the smallest element IN available_workers THAT IS GREATER THAN OR EQUAL TO task MINUS strength  
                    IF worker_idx IS LESS THAN the LENGTH OF available_workers THEN  
                        REMOVE element at position worker_idx FROM available_workers  
                        INCREMENT pills_used BY one  
                    ELSE  
                        RETURN false  
                    END IF  
                ELSE  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
          
        SET left TO zero  
        SET right TO the smaller of the LENGTH OF tasks AND the LENGTH OF workers PLUS one  
        WHILE left IS LESS THAN right  
            SET mid TO left PLUS right DIVIDED BY two using integer division  
            IF canAssignTasks(mid) IS true THEN  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN left MINUS one  
    END FUNCTION  
END CLASS
