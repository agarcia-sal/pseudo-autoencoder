CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the value attribute TO val  
        SET the left child attribute TO left  
        SET the right child attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the values list is empty  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode constructed with element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i is less than the length of values AND the element at position i of values is not nothing  
            SET the left child attribute of node TO a new TreeNode constructed with element at position i of values  
            APPEND the left child attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not nothing  
            SET the right child attribute of node TO a new TreeNode constructed with element at position i of values  
            APPEND the right child attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p is nothing AND q is nothing  
        RETURN true  
    ELSE IF p is nothing OR q is nothing  
        RETURN false  
    ELSE IF the value attribute of p is NOT EQUALS to the value attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with parameters the left child attribute of p and the left child attribute of q AND the result of calling is_same_tree with parameters the right child attribute of p and the right child attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves with parameters root  
        FUNCTION collectLeaves with parameter node  
            IF node is nothing  
                RETURN minus one  
            END IF  
            SET left_height TO the result of calling collectLeaves with parameter the left child attribute of node  
            SET right_height TO the result of calling collectLeaves with parameter the right child attribute of node  
            SET current_height TO the maximum of left_height AND right_height PLUS one  
            IF current_height is GREATER THAN OR EQUAL TO the length of result  
                APPEND an empty list TO result  
            END IF  
            APPEND the value attribute of node TO the element at position current_height of result  
            RETURN current_height  
        END FUNCTION  
        SET result TO an empty list  
        CALL collectLeaves with parameter root  
        RETURN result  
    END FUNCTION  
END CLASS
