CLASS BIT  
    FUNCTION __init__(self, n)  
        SET self.size TO n  
        SET self.tree TO a list of zero repeated the size of n PLUS one  
    END FUNCTION  

    FUNCTION update(self, idx, delta)  
        WHILE idx LESS THAN OR EQUAL TO self.size  
            INCREMENT element at position idx of self.tree BY delta  
            SET idx TO idx PLUS the bitwise AND of idx and the negative idx  
        END WHILE  
    END FUNCTION  

    FUNCTION query(self, idx)  
        SET result TO zero  
        WHILE idx GREATER THAN zero  
            INCREMENT result BY element at position idx of self.tree  
            SET idx TO idx MINUS the bitwise AND of idx and the negative idx  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION goodTriplets(self, nums1, nums2)  
        SET n TO the length of nums1  

        SET pos TO a list of zero repeated the size of n  
        FOR each pair of i AND num IN enumeration of nums1  
            SET element at position num of pos TO i  
        END FOR  

        SET compressed TO a list containing element at position num of pos FOR each num IN nums2  

        ASSIGN reference TO new BIT object with parameter n AS bit  

        SET smaller_count TO a list of zero repeated the size of n  
        SET larger_count TO a list of zero repeated the size of n  

        FOR i FROM zero TO n MINUS one  
            SET element at position i of smaller_count TO bit.query(element at position i of compressed)  
            CALL bit.update with parameters element at position i of compressed PLUS one AND one  
        END FOR  

        ASSIGN reference TO new BIT object with parameter n AS bit  

        FOR i FROM n MINUS one TO zero DECREMENT BY one  
            SET element at position i of larger_count TO bit.query(n) MINUS bit.query(element at position i of compressed PLUS one)  
            CALL bit.update with parameters element at position i of compressed PLUS one AND one  
        END FOR  

        SET good_triplets TO zero  
        FOR i FROM zero TO n MINUS one  
            INCREMENT good_triplets BY element at position i of smaller_count MULTIPLIED BY element at position i of larger_count  
        END FOR  

        RETURN good_triplets  
    END FUNCTION  
END CLASS
