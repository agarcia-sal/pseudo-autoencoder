CLASS Solution
    FUNCTION fractionToDecimal(numerator INTEGER denominator INTEGER)  
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero  
            SET sign TO the string consisting of a minus character  
        ELSE  
            SET sign TO the empty string  
        END IF  
        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  
        SET quotient TO the integer division result of numerator DIVIDED BY denominator  
        SET remainder TO the remainder after numerator is divided by denominator  
        SET integer_part TO the string representation of quotient  
        IF remainder EQUALS zero  
            RETURN the concatenation of sign and integer_part  
        END IF  
        CREATE an empty dictionary named seen_remainders  
        CREATE an empty list named fractional_part  
        WHILE remainder NOT EQUALS zero  
            IF remainder IS A KEY IN seen_remainders  
                SET start TO the value mapped by remainder in seen_remainders  
                SET non_repeating TO the concatenation of the elements in fractional_part from the first position UP TO but NOT INCLUDING position start  
                SET repeating TO the concatenation of the elements in fractional_part from position start TO the last position  
                RETURN the concatenation of sign integer_part the single character representing a decimal point non_repeating a single opening parenthesis repeating and a single closing parenthesis  
            END IF  
            MAP remainder TO the length of fractional_part in seen_remainders  
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO the integer division result of remainder DIVIDED BY denominator  
            SET remainder TO the remainder after remainder is divided by denominator  
            APPEND the string representation of quotient TO fractional_part  
        END WHILE  
        RETURN the concatenation of sign integer_part the single character representing a decimal point and the concatenation of all elements in fractional_part  
    END FUNCTION
END CLASS
