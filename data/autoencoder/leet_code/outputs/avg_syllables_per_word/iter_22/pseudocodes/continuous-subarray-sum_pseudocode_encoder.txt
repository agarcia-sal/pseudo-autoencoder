CLASS Solution  
    FUNCTION checkSubarraySum(nums k)  
        SET remainder_map TO dictionary with key zero mapped to negative one  
        SET current_sum TO zero  
        FOR each index i AND each element num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF k NOT EQUALS zero THEN  
                SET current_sum TO current_sum MODULO k  
            END IF  
            IF current_sum IS A KEY IN remainder_map THEN  
                IF i MINUS the element at key current_sum in remainder_map IS GREATER THAN one THEN  
                    RETURN True  
                END IF  
            ELSE  
                ASSIGN the element at key current_sum in remainder_map TO i  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
