CLASS Solution  
    FUNCTION crackSafe(n, k)  
        SET visited TO an empty set  
        SET sequence TO an empty list  
        
        FUNCTION dfs(node)  
            FOR i FROM zero TO k MINUS one  
                SET edge TO concatenation of node AND string representation of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs with argument substring of edge from position one TO the end of edge  
                    APPEND string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO string consisting of character zero REPEATED n MINUS one TIMES  
        CALL dfs with argument start_node  
        SET result TO concatenation of elements of sequence IN REVERSE ORDER FOLLOWED BY start_node  
        RETURN result  
    END FUNCTION  
END CLASS
