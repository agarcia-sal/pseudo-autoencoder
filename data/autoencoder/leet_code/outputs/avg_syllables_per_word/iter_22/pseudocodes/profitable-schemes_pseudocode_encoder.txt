CLASS Solution  
    FUNCTION profitableSchemes(n profit group profit)  
        SET MOD TO one billion PLUS seven  
        CREATE two dimensional list dp WITH size n PLUS one BY minProfit PLUS one FILLED WITH zero  
        FOR index i FROM zero TO n  
            SET element at position zero of element at position i of dp TO one  
        END FOR  
        FOR each element p AND element g IN parallel from profit AND group  
            FOR index i FROM n DOWN TO g  
                FOR index j FROM minProfit DOWN TO zero  
                    SET element at position j of element at position i of dp TO the sum of element at position j of element at position i of dp AND element at position the greater of zero AND j MINUS p of element at position i MINUS g of dp MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position minProfit of element at position n of dp  
    END FUNCTION  
END CLASS
