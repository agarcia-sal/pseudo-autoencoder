CLASS Solution  
    FUNCTION strobogrammaticInRange(low  
                                    high)  
        FUNCTION generate_strobogrammatic(n  
                                         final_length)  
            IF n EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF n EQUALS one  
                RETURN list containing string zero string one string eight  
            END IF  
            SET middles TO generate_strobogrammatic(n MINUS two  
                                                    final_length)  
            SET result TO empty list  
            FOR each middle IN middles  
                IF n NOT EQUALS final_length  
                    APPEND concatenation of string zero plus middle plus string zero TO result  
                END IF  
                APPEND concatenation of string one plus middle plus string one TO result  
                APPEND concatenation of string six plus middle plus string nine TO result  
                APPEND concatenation of string eight plus middle plus string eight TO result  
                APPEND concatenation of string nine plus middle plus string six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        FUNCTION count_strobogrammatic(low  
                                       high)  
            SET count TO zero  
            FOR length FROM length of low TO length of high  
                FOR each num IN generate_strobogrammatic(length  
                                                        length)  
                    IF (length EQUALS length of low AND num LESS THAN low) OR (length EQUALS length of high AND num GREATER THAN high)  
                        CONTINUE to next iteration  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN count_strobogrammatic(low  
                                     high)  
    END FUNCTION  
END CLASS
