CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1, str2)  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        SET dp TO a list of lists where each inner list has n PLUS one zeros and there are m PLUS one such inner lists  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2 THEN  
                    SET the element at position i and position j of dp TO the element at position i MINUS one and position j MINUS one of dp PLUS one  
                ELSE  
                    IF the element at position i MINUS one and position j of dp GREATER THAN the element at position i and position j MINUS one of dp THEN  
                        SET the element at position i and position j of dp TO the element at position i MINUS one and position j of dp  
                    ELSE  
                        SET the element at position i and position j of dp TO the element at position i and position j MINUS one of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        SET i TO m  
        SET j TO n  
        SET scs TO an empty list  
        WHILE i GREATER THAN zero AND j GREATER THAN zero  
            IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2 THEN  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF the element at position i MINUS one and position j of dp GREATER THAN the element at position i and position j MINUS one of dp THEN  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND the element at position j MINUS one of str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        WHILE i GREATER THAN zero  
            APPEND the element at position i MINUS one of str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        WHILE j GREATER THAN zero  
            APPEND the element at position j MINUS one of str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        REVERSE the list scs  
        SET result_string TO the concatenation of all elements in scs without any separator  
        RETURN result_string  
    END FUNCTION  
END CLASS
