CLASS Solution  
    FUNCTION myAtoi(s)  
        SET INT_MAX TO two RAISED TO THE POWER OF thirty ONE MINUS one  
        SET INT_MIN TO NEGATIVE two RAISED TO THE POWER OF thirty ONE  
        SET result TO zero  
        SET index TO zero  
        SET n TO the LENGTH OF s  
        SET sign TO one  
        WHILE index LESS THAN n AND the character at position index of s EQUALS a space character HOLDS  
            INCREMENT index BY one  
        END WHILE  
        IF index LESS THAN n AND (the character at position index of s EQUALS a plus character OR the character at position index of s EQUALS a minus character) HOLDS  
            IF the character at position index of s EQUALS a minus character HOLDS  
                SET sign TO NEGATIVE one  
            ELSE  
                SET sign TO one  
            END IF  
            INCREMENT index BY one  
        END IF  
        WHILE index LESS THAN n AND the character at position index of s IS A DIGIT HOLDS  
            SET digit TO the integer value of the character at position index of s  
            IF result GREATER THAN INT_MAX MINUS digit DIVIDED BY ten HOLDS  
                IF sign EQUALS one HOLDS  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
                END IF  
            END IF  
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        END WHILE  
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
