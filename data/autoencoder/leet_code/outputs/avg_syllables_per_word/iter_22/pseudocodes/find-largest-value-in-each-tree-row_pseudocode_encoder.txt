CLASS TreeNode  
    FUNCTION __init__(self WITH val DEFAULTS TO zero WITH left DEFAULTS TO none WITH right DEFAULTS TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS no elements THEN  
        RETURN none  
    END IF  
    SET root TO new instance of TreeNode with element at position one of values  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE length of queue GREATER THAN zero HOLD  
        SET node TO first element removed from queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS none THEN  
            SET node.left TO new instance of TreeNode with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS none THEN  
            SET node.right TO new instance of TreeNode with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p WITH q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN False  
    ELSE IF value of p NOT EQUALS value of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p WITH left child of q) AND is_same_tree(right child of p WITH right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues(self WITH root)  
        IF root EQUALS none THEN  
            RETURN empty list  
        END IF  
        SET queue TO double ended queue initialized with root  
        SET largest_values TO empty list  
        WHILE length of queue GREATER THAN zero HOLD  
            SET level_size TO length of queue  
            SET max_value TO negative infinity  
            FOR index FROM one TO level_size DO  
                SET node TO first element removed from queue  
                SET max_value TO greater of max_value AND value of node  
                IF left child of node NOT EQUALS none THEN  
                    APPEND left child of node TO queue  
                END IF  
                IF right child of node NOT EQUALS none THEN  
                    APPEND right child of node TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
