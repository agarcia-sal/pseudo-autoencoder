CLASS Solution  
    FUNCTION smallestSufficientTeam(self, req_skills, people)  
        CREATE empty dictionary named d  
        SET index_counter TO zero  
        FOR each skill s IN req_skills  
            ASSIGN reference TO index_counter AS value for key s IN dictionary d  
            INCREMENT index_counter BY one  
        END FOR  
        SET m TO the LENGTH OF req_skills  
        SET n TO the LENGTH OF people  
        CREATE list p with n elements each SET TO zero  
        FOR each index i AND skill_set ss IN people WITH index  
            FOR each skill s IN skill_set ss  
                SET p at position i TO the VALUE OF p at position i PLUS ONE MULTIPLIED BY TWO RAISED TO THE POWER OF the VALUE in d FOR key s  
            END FOR  
        END FOR  
        CREATE list f WITH LENGTH equal TO TWO RAISED TO THE POWER OF m EACH ELEMENT SET TO infinite  
        CREATE list g WITH LENGTH equal TO TWO RAISED TO THE POWER OF m EACH ELEMENT SET TO zero  
        CREATE list h WITH LENGTH equal TO TWO RAISED TO THE POWER OF m EACH ELEMENT SET TO zero  
        SET f at position zero TO zero  
        FOR each number i FROM zero TO TWO RAISED TO THE POWER OF m MINUS one  
            IF f at position i EQUALS infinite  
                CONTINUE TO NEXT ITERATION  
            END IF  
            FOR each number j FROM zero TO n MINUS one  
                SET combined_index TO the BITWISE OR OF i AND p at position j  
                IF f at position i PLUS one LESS THAN f at position combined_index  
                    SET f at position combined_index TO f at position i PLUS one  
                    SET g at position combined_index TO j  
                    SET h at position combined_index TO i  
                END IF  
            END FOR  
        END FOR  
        SET i TO TWO RAISED TO THE POWER OF m MINUS one  
        CREATE empty list named ans  
        WHILE i NOT EQUALS zero  
            APPEND g at position i TO list ans  
            SET i TO h at position i  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
