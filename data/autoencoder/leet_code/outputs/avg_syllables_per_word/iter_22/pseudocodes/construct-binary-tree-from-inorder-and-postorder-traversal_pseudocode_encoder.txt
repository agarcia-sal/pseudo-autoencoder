CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO none right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode with element at position one of values  
    SET i TO one  
    SET queue TO NEW double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element REMOVED from the FRONT of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.left TO NEW TreeNode with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.right TO NEW TreeNode with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left attribute of p AND the left attribute of q AND the result of is_same_tree with the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION buildTree(self inorder postorder)  
        IF the LENGTH OF inorder EQUALS zero OR the LENGTH OF postorder EQUALS zero  
            RETURN none  
        END IF  
        SET root_val TO the element REMOVED from the END of postorder  
        SET root TO NEW TreeNode with root_val  
        SET root_index TO the POSITION of root_val in inorder  
        SET root.right TO the result of buildTree with the subsection of inorder from the position root_index PLUS one TO the END AND postorder  
        SET root.left TO the result of buildTree with the subsection of inorder from the START TO the position root_index MINUS one AND postorder  
        RETURN root  
    END FUNCTION  
END CLASS
