CLASS Solution  
    FUNCTION numRescueBoats people limit  
        SORT the list people in ascending order by their weight  
        SET left TO zero  
        SET right TO the length of the list people MINUS one  
        SET boats TO zero  
        WHILE left LESS THAN OR EQUAL TO right  
            IF element at position left of the list people PLUS element at position right of the list people LESS THAN OR EQUAL TO limit  
                INCREMENT left BY one  
            END IF  
            DECREMENT right BY one  
            INCREMENT boats BY one  
        END WHILE  
        RETURN boats  
    END FUNCTION  
END CLASS
