CLASS Solution  
    FUNCTION countComponents(n number, edges list of list of number) RETURNS number  
        SET graph TO empty mapping from number to empty list  
        FOR index FROM zero TO n MINUS one  
            ASSIGN reference TO empty list IN graph at position index  
        END FOR  
        FOR each pair consisting of element a AND element b IN edges  
            APPEND element b TO list at position a in graph  
            APPEND element a TO list at position b in graph  
        END FOR  
        SET visited TO empty set  
        FUNCTION dfs(node number)  
            IF node IS IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN list at position node in graph  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        SET components TO zero  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                CALL dfs WITH node  
                INCREMENT components BY one  
            END IF  
        END FOR  
        RETURN components  
    END FUNCTION  
END CLASS
