CLASS Solution  
    FUNCTION new21Game WITH PARAMETERS n k maxPts  
        FUNCTION dfs WITH PARAMETER i  
            IF i GREATER THAN OR EQUAL TO k  
                IF i LESS THAN OR EQUAL TO n  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF i EQUALS k MINUS one  
                SET difference TO n MINUS k PLUS one  
                IF difference GREATER THAN maxPts  
                    SET numerator TO maxPts  
                ELSE  
                    SET numerator TO difference  
                END IF  
                RETURN numerator DIVIDED BY maxPts  
            END IF  
            SET value1 TO dfs OF i PLUS one  
            SET value2 TO dfs OF i PLUS maxPts PLUS one  
            SET difference TO value1 MINUS value2  
            RETURN value1 PLUS difference DIVIDED BY maxPts  
        END FUNCTION  
        RETURN dfs OF zero  
    END FUNCTION  
END CLASS
