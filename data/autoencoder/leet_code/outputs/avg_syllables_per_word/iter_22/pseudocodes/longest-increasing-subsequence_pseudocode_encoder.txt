CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums EQUALS empty list  
            RETURN zero  
        END IF  
        SET dp TO empty list  
        FOR each num IN nums  
            SET idx TO the position to insert num in dp such that dp remains sorted in ascending order  
            IF idx EQUALS the LENGTH OF dp  
                APPEND num TO dp  
            ELSE  
                SET element at position idx of dp TO num  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
