CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO A NEW TreeNode WITH val EQUALS the element at position zero of values AND left EQUALS no value AND right EQUALS no value  
    SET i TO one  
    SET queue TO AN EMPTY double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO THE first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.left TO A NEW TreeNode WITH val EQUALS the element at position i of values AND left EQUALS no value AND right EQUALS no value  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.right TO A NEW TreeNode WITH val EQUALS the element at position i of values AND left EQUALS no value AND right EQUALS no value  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value THEN  
        RETURN true  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree WITH p.left AND q.left AND the result of is_same_tree WITH p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION countUnivalSubtrees(self, root)  
        FUNCTION helper(node)  
            IF node IS no value THEN  
                RETURN true AND zero  
            END IF  
            SET is_left_unival, left_count TO helper WITH node.left  
            SET is_right_unival, right_count TO helper WITH node.right  
            IF is_left_unival AND is_right_unival THEN  
                IF (node.left IS no value OR the val of node.left EQUALS the val of node) AND (node.right IS no value OR the val of node.right EQUALS the val of node) THEN  
                    RETURN true AND left_count PLUS right_count PLUS one  
                END IF  
            END IF  
            RETURN false AND left_count PLUS right_count  
        END FUNCTION  
        SET _, count TO helper WITH root  
        RETURN count  
    END FUNCTION  
END CLASS
