CLASS Solution  
    FUNCTION minOperations(nums)  
        SET n TO the LENGTH OF nums  
        SET unique_nums TO the SORTED SET OF nums  
        SET min_operations TO n  
        SET j TO zero  
        FOR i FROM zero TO the LENGTH OF unique_nums MINUS one  
            WHILE j IS LESS THAN the LENGTH OF unique_nums AND element at position j of unique_nums IS LESS THAN element at position i of unique_nums PLUS n  
                INCREMENT j BY one  
            END WHILE  
            SET current_window_size TO j MINUS i  
            SET min_operations TO the MINIMUM OF min_operations AND n MINUS current_window_size  
        END FOR  
        RETURN min_operations  
    END FUNCTION  
END CLASS
