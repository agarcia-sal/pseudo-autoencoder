CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO count frequency of each character in s  
        
        SET max_heap TO list of pairs of negative frequency and character from freq  
        TRANSFORM max_heap into a heap structure  
        
        SET wait_queue TO empty list  
        
        SET result TO empty list  
        
        WHILE max_heap is not empty  
            EXTRACT the element with highest priority from max_heap and assign its frequency to count and character to char  
            APPEND char TO result  
            
            APPEND the pair of count PLUS one and char TO wait_queue  
            
            IF the length of wait_queue EQUALS k  
                REMOVE the first element from wait_queue and assign its frequency to count and character to char  
                IF count LESS THAN zero  
                    ADD the pair of count and char TO max_heap  
                END IF  
            END IF  
        END WHILE  
        
        IF the length of result NOT EQUALS the length of s  
            RETURN an empty string  
        END IF  
        
        RETURN the concatenation of characters in result  
    END FUNCTION  
END CLASS
