CLASS ListNode  
    FUNCTION __init__ with parameters val next  
        SET the attribute val TO val  
        SET the attribute next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO a new ListNode with element at position one of values  
    SET p TO head  
    FOR each val IN the sublist of values from position two TO the end  
        SET node TO a new ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the attribute val of p1 EQUALS the attribute val of p2 AND the result of calling is_same_list with the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight with parameters head k  
        IF head EQUALS None OR the next attribute of head EQUALS None OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE the next attribute of tail IS NOT None  
            SET tail TO the next attribute of tail  
            INCREMENT length BY one  
        END WHILE  
        SET the next attribute of tail TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR index FROM zero TO steps_to_new_tail MINUS one  
            SET new_tail TO the next attribute of new_tail  
        END FOR  
        SET new_head TO the next attribute of new_tail  
        SET the next attribute of new_tail TO None  
        RETURN new_head  
    END FUNCTION  
END CLASS
