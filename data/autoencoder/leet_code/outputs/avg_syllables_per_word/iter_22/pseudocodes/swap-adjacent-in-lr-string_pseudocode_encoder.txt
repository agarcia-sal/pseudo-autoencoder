CLASS Solution  
    FUNCTION canTransform(start  
                         end)  
        SET start_filtered TO an empty list  
        FOR each index i AND character c IN the enumeration of start  
            IF c NOT EQUALS the character X  
                APPEND the pair consisting of c AND i TO start_filtered  
            END IF  
        END FOR  
        SET end_filtered TO an empty list  
        FOR each index i AND character c IN the enumeration of end  
            IF c NOT EQUALS the character X  
                APPEND the pair consisting of c AND i TO end_filtered  
            END IF  
        END FOR  
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        FOR each pair consisting of character c1 AND position i1 FROM start_filtered AND each pair consisting of character c2 AND position i2 FROM end_filtered TAKEN IN PARALLEL  
            IF c1 NOT EQUALS c2  
                RETURN false  
            END IF  
            IF c1 EQUALS the character L AND i1 LESS THAN i2  
                RETURN false  
            END IF  
            IF c1 EQUALS the character R AND i1 GREATER THAN i2  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
