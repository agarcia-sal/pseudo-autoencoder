CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode constructed with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value  
        RETURN true  
    ELSE IF p IS no value OR q IS no value  
        RETURN false  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree applied to the left child of p and the left child of q AND the result of is_same_tree applied to the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST(self, root)  
        SET self.max_sum TO zero  
        
        FUNCTION helper(node)  
            IF node IS no value  
                RETURN true, zero, positive infinity, negative infinity  
            END IF  
            
            SET left_is_bst, left_sum, left_min, left_max TO the result of helper applied to the left child of node  
            SET right_is_bst, right_sum, right_min, right_max TO the result of helper applied to the right child of node  
            
            IF left_is_bst EQUALS true AND right_is_bst EQUALS true AND left_max LESS THAN the value of node AND the value of node LESS THAN right_min  
                SET current_sum TO left_sum PLUS right_sum PLUS the value of node  
                SET current_min TO the smaller of the value of node and left_min  
                SET current_max TO the larger of the value of node and right_max  
                SET self.max_sum TO the larger of self.max_sum and current_sum  
                RETURN true, current_sum, current_min, current_max  
            END IF  
            
            RETURN false, zero, positive infinity, negative infinity  
        END FUNCTION  
        
        CALL helper with root  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
