CLASS Solution  
    FUNCTION waysToBuildRooms(prevRoom)  
        SET n TO the LENGTH OF prevRoom  
        ASSIGN graph TO a new mapping with default value as an empty list  
        
        FOR room FROM one TO n MINUS one  
            APPEND room TO the list AT element at position room of prevRoom IN graph  
        END FOR  
        
        FUNCTION dp(node)  
            IF the list AT node IN graph IS empty  
                RETURN one AND one  
            END IF  
            
            SET total_ways TO one  
            SET total_size TO one  
            ASSIGN child_results TO an empty list  
            
            FOR each child IN the list AT node IN graph  
                SET ways AND size TO the result of dp called with child  
                APPEND the pair of ways and size TO child_results  
                INCREMENT total_size BY size  
            END FOR  
            
            SET current_size TO one  
            FOR each pair ways AND size IN child_results  
                SET total_ways TO total_ways MULTIPLIED BY the binomial coefficient of current_size PLUS size MINUS one AND size MULTIPLIED BY ways MODULO MOD  
                INCREMENT current_size BY size  
            END FOR  
            
            RETURN total_ways AND total_size  
        END FUNCTION  
        
        RETURN the first element of the result of dp called with zero  
    END FUNCTION  
END CLASS
