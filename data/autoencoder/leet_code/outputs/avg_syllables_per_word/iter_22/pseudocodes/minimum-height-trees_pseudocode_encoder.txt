CLASS Solution  
    FUNCTION findMinHeightTrees WITH PARAMETERS n edges  
        IF n EQUALS one THEN  
            RETURN list containing zero  
        END IF  
        SET graph TO a mapping from each key to empty list by default  
        SET degree TO list of zero repeated n times  
        FOR each pair u and v IN edges  
            APPEND v TO list at key u in graph  
            APPEND u TO list at key v in graph  
            INCREMENT degree at position u BY one  
            INCREMENT degree at position v BY one  
        END FOR  
        SET leaves TO a double-ended queue containing all indices i from zero TO n MINUS one WHERE degree at position i EQUALS one  
        SET remaining_nodes TO n  
        WHILE remaining_nodes GREATER THAN two  
            SET leaves_count TO the LENGTH OF leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR iteration FROM one TO leaves_count  
                SET leaf TO the element removed from the front of leaves  
                FOR each neighbor IN list at key leaf in graph  
                    DECREMENT degree at position neighbor BY one  
                    IF degree at position neighbor EQUALS one THEN  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN list converted from leaves  
    END FUNCTION  
END CLASS
