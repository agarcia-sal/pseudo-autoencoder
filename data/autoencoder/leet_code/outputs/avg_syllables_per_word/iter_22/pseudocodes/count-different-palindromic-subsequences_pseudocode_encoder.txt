CLASS Solution  
    FUNCTION countPalindromicSubsequences(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        
        SET next_occurrence TO a two dimensional list of size four by n FILLED WITH negative one  
        SET prev_occurrence TO a two dimensional list of size four by n FILLED WITH negative one  
        
        FOR each c FROM zero TO three  
            SET char TO the character at the position corresponding to the alphabetical order of lowercase a PLUS c  
            SET last TO negative one  
            FOR each i FROM zero TO n MINUS one  
                IF element at position i of s EQUALS char THEN  
                    SET last TO i  
                END IF  
                SET element at position c and i of prev_occurrence TO last  
            END FOR  
            
            SET last TO negative one  
            FOR each i FROM n MINUS one TO zero DECREMENT BY one  
                IF element at position i of s EQUALS char THEN  
                    SET last TO i  
                END IF  
                SET element at position c and i of next_occurrence TO last  
            END FOR  
        END FOR  
        
        SET dp TO a two dimensional list of size n by n FILLED WITH zero  
        
        FOR each i FROM zero TO n MINUS one  
            SET element at position i and i of dp TO one  
        END FOR  
        
        FOR each length FROM two TO n  
            FOR each i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                
                FOR each c FROM zero TO three  
                    SET char TO the character at the position corresponding to the alphabetical order of lowercase a PLUS c  
                    SET left TO element at position c and i of next_occurrence  
                    SET right TO element at position c and j of prev_occurrence  
                    
                    IF left GREATER THAN j OR right LESS THAN i THEN  
                        CONTINUE to the next iteration of the innermost loop  
                    ELSE IF left EQUALS right THEN  
                        SET element at position i and j of dp TO the remainder of the sum of element at position i and j of dp PLUS one DIVIDED BY MOD  
                    ELSE  
                        SET element at position i and j of dp TO the remainder of the sum of element at position i and j of dp PLUS element at position left PLUS one and right MINUS one of dp PLUS two DIVIDED BY MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero and n MINUS one of dp  
    END FUNCTION  
END CLASS
