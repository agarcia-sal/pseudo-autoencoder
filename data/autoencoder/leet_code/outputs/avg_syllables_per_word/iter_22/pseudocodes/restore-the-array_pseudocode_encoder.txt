CLASS Solution  
    FUNCTION numberOfArrays(s string_parameter k integer_parameter) RETURNS integer  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF string_parameter  
        INITIALIZE dp AS a list of zeros with LENGTH n PLUS one  
        SET element at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO i MINUS one  
                IF the character at position j of string_parameter NOT EQUALS the character zero AND  
                   the integer value of the substring from position j TO position i MINUS one of string_parameter GREATER THAN OR EQUAL TO one AND  
                   the integer value of the substring from position j TO position i MINUS one of string_parameter LESS THAN OR EQUAL TO integer_parameter  
                THEN  
                    SET element at position i of dp TO the sum of element at position i of dp PLUS element at position j of dp  
                    SET element at position i of dp TO element at position i of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
