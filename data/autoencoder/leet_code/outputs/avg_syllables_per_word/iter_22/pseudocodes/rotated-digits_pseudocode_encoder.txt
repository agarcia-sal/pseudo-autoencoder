CLASS Solution  
    FUNCTION rotatedDigits WITH PARAMETER n  
        SET change_digits TO the collection of elements two five six nine  
        SET same_digits TO the collection of elements zero one eight  
        
        FUNCTION is_good_number WITH PARAMETER num  
            SET num_str TO the representation of num as a sequence of characters  
            IF for each digit IN num_str every digit is contained in the union of change_digits and same_digits THEN  
                IF there exists at least one digit IN num_str that is contained in change_digits THEN  
                    RETURN true  
                END IF  
                RETURN false  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR each i FROM one TO n  
            IF is_good_number WITH argument i EQUALS true THEN  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
