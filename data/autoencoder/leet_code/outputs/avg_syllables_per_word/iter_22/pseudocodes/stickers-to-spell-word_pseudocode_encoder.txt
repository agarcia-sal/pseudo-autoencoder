CLASS Solution  
    FUNCTION minStickers(stickers LIST OF STRINGS target STRING) RETURNS INTEGER  
        SET stickers TO list of frequency mappings obtained by COUNTING each element IN stickers  
        SORT stickers BY the NEGATION OF the maximum frequency value IN each frequency mapping  
        SET queue TO a double ended queue initialized with a tuple containing target and zero  
        SET memo TO a mapping with target associated TO zero  
        WHILE the queue HAS elements  
            REMOVE and ASSIGN the first element of queue TO current_target and used_stickers  
            FOR each sticker IN stickers  
                IF the first character of current_target IS NOT PRESENT IN the keys of sticker  
                    CONTINUE to next iteration  
                END IF  
                SET remaining TO a copy of frequency mapping obtained by COUNTING characters IN current_target  
                DECREMENT each frequency in remaining BY corresponding frequency IN sticker  
                REMOVE all characters FROM remaining WITH frequency LESS THAN OR EQUAL TO zero  
                SET new_target TO a string formed by sorting all characters FROM remaining REPEATED by their frequencies  
                IF new_target IS AN EMPTY STRING  
                    RETURN used_stickers PLUS one  
                END IF  
                IF new_target IS NOT a key IN memo  
                    SET memo[new_target] TO used_stickers PLUS one  
                    APPEND the tuple of new_target and used_stickers PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
