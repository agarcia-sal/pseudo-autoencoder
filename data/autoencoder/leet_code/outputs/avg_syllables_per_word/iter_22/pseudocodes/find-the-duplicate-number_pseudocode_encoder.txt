CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count TO zero  
            FOR each value IN nums  
                IF value LESS THAN OR EQUAL TO x  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            RETURN count GREATER THAN x  
        END FUNCTION  
        RETURN the smallest position IN the sequence from zero TO the LENGTH OF nums MINUS one WHERE f(position) IS True  
    END FUNCTION  
END CLASS
