CLASS Solution  
    FUNCTION intToRoman(num)  
        SET list_of_values TO list of integer one thousand PLUS nine hundred PLUS five hundred PLUS four hundred PLUS one hundred PLUS ninety PLUS fifty PLUS forty PLUS ten PLUS nine PLUS five PLUS four PLUS one  
        SET list_of_symbols TO list of string em PLUS string ce em PLUS string de PLUS string ce de PLUS string ce PLUS string ex ce PLUS string el PLUS string ex el PLUS string ex PLUS string ix PLUS string ve PLUS string iv PLUS string i  
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero  
            SET number_of_times TO integer division of num DIVIDED BY element at position i of list_of_values  
            FOR count FROM one TO number_of_times  
                APPEND element at position i of list_of_symbols TO roman_numeral  
                DECREMENT num BY element at position i of list_of_values  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
