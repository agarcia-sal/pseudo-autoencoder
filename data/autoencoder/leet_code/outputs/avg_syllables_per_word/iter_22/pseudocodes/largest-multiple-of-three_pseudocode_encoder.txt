CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO the frequency count of digits  
        SET total_sum TO the sum of all elements in digits  
        SET remainder TO the total_sum MODULO three  

        IF remainder EQUALS one  
            FOR each digit_element IN the list containing one four seven  
                IF the count of digit_element IS GREATER THAN zero  
                    DECREMENT the count of digit_element BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit_element IN the list containing two five eight  
                    IF the count of digit_element IS GREATER THAN one  
                        DECREMENT the count of digit_element BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
        END IF  

        IF remainder EQUALS two  
            FOR each digit_element IN the list containing two five eight  
                IF the count of digit_element IS GREATER THAN zero  
                    DECREMENT the count of digit_element BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit_element IN the list containing one four seven  
                    IF the count of digit_element IS GREATER THAN one  
                        DECREMENT the count of digit_element BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
        END IF  

        SET result TO an empty list  
        FOR digit_value FROM nine TO zero DECREASING BY one  
            APPEND the string representation of digit_value REPEATED the count of digit_value times TO result  
        END FOR  

        SET final_number TO the concatenation of all elements in result  

        IF the length of final_number IS GREATER THAN zero AND the element at position zero of final_number EQUALS zero  
            RETURN zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
