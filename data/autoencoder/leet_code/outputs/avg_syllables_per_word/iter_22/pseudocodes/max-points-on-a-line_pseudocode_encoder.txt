CLASS Solution  
    FUNCTION maxPoints(points)  
        FUNCTION slope(p1 p2)  
            IF element at position zero of p1 EQUALS element at position zero of p2  
                RETURN positive infinity  
            END IF  
            SET numerator TO element at position one of p1 MINUS element at position one of p2  
            SET denominator TO element at position zero of p1 MINUS element at position zero of p2  
            RETURN numerator DIVIDED BY denominator  
        END FUNCTION  
        
        SET max_points TO zero  
        
        FOR index i FROM zero TO the LENGTH OF points MINUS one  
            SET slopes TO an empty map with default value zero  
            FOR index j FROM zero TO the LENGTH OF points MINUS one  
                IF i NOT EQUALS j  
                    SET s TO the RESULT OF slope WITH arguments element at position i of points AND element at position j of points  
                    INCREMENT the value mapped by s in slopes BY one  
                    IF the value mapped by s in slopes GREATER THAN max_points  
                        SET max_points TO the value mapped by s in slopes  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF the LENGTH OF points GREATER THAN zero  
            RETURN max_points PLUS one  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
