CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones EQUALS empty OR the LENGTH OF stones LESS THAN two  
            RETURN False  
        END IF  
        SET stone_set TO the COLLECTION of DISTINCT elements FROM stones  
        SET memo TO an EMPTY mapping  
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the element at the last position of stones  
                RETURN True  
            END IF  
            IF the PAIR of position AND jump EXISTS IN memo  
                RETURN the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo  
            END IF  
            FOR each next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump GREATER THAN zero AND the SUM of position AND next_jump IS AN ELEMENT OF stone_set  
                    IF can_jump_to(the SUM of position AND next_jump, next_jump)  
                        SET the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            SET the VALUE ASSOCIATED WITH the PAIR of position AND jump IN memo TO False  
            RETURN False  
        END FUNCTION  
        RETURN can_jump_to(the element at position zero of stones, zero)  
    END FUNCTION  
END CLASS
