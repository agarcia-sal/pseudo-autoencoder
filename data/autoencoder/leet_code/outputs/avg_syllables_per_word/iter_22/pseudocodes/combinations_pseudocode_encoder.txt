CLASS Solution  
    FUNCTION combine(n k)  
        FUNCTION backtrack(start path)  
            IF the LENGTH OF path EQUALS k  
                APPEND a COPY OF path TO result  
                RETURN  
            END IF  
            FOR i FROM start TO n PLUS one MINUS one  
                APPEND element at position i TO path  
                CALL backtrack WITH the first argument equal TO i PLUS one AND the second argument equal TO path  
                REMOVE the last element FROM path  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH the first argument equal TO one AND the second argument equal TO an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
