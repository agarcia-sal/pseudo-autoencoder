CLASS Solution  
    FUNCTION minMoves(nums k)  
        SET positions TO empty list  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF element at position index of nums EQUALS one  
                APPEND index TO positions  
            END IF  
        END FOR  

        FUNCTION calculate_cost(start end)  
            SET mid TO start PLUS end DIVIDED BY two using integer division  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR i FROM start TO end MINUS one  
                SET distance TO element at position i of positions MINUS median PLUS mid MINUS i  
                IF distance LESS THAN zero  
                    SET distance TO distance MULTIPLIED BY negative one  
                END IF  
                INCREMENT cost BY distance  
            END FOR  
            RETURN cost  
        END FUNCTION  

        SET min_cost TO a very large number representing infinity  
        FOR i FROM zero TO the LENGTH OF positions MINUS k  
            SET current_cost TO calculate_cost(i i PLUS k)  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
