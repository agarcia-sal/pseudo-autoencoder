CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty list  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val EQUALS element at position one of values  
    SET i TO one  
    SET queue TO new deque  
    CALL queue APPEND with root  
    WHILE the LENGTH OF queue GREATER THAN zero holds  
        SET node TO the first element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.left TO new TreeNode with val EQUALS element at position i of values  
            CALL queue APPEND with node.left  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.right TO new TreeNode with val EQUALS element at position i of values  
            CALL queue APPEND with node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p.left AND q.left AND the result of is_same_tree with p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root EQUALS none  
            RETURN  
        END IF  
        SET stack TO list containing root  
        WHILE the LENGTH OF stack GREATER THAN zero holds  
            SET node TO the last element removed from stack  
            IF node.right NOT EQUALS none  
                CALL stack APPEND with node.right  
            END IF  
            IF node.left NOT EQUALS none  
                CALL stack APPEND with node.left  
            END IF  
            SET node.left TO none  
            IF the LENGTH OF stack GREATER THAN zero  
                SET node.right TO the last element of stack  
            ELSE  
                SET node.right TO none  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
