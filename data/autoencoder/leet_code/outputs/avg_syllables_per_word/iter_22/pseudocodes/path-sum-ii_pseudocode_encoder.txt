CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position zero of values  
    SET i TO one  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the front of queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS none  
            SET node.left TO new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS none  
            SET node.right TO new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree with arguments p.left and q.left AND is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION dfs(node, current_sum, path)  
            IF node EQUALS none  
                RETURN  
            END IF  
            APPEND node.val TO path  
            SET current_sum TO current_sum PLUS node.val  
            IF node.left EQUALS none AND node.right EQUALS none AND current_sum EQUALS targetSum  
                APPEND a copy of path TO result  
            END IF  
            CALL dfs with arguments node.left current_sum path  
            CALL dfs with arguments node.right current_sum path  
            REMOVE the last element FROM path  
        END FUNCTION  
        SET result TO empty list  
        CALL dfs with arguments root zero empty list  
        RETURN result  
    END FUNCTION  
END CLASS
