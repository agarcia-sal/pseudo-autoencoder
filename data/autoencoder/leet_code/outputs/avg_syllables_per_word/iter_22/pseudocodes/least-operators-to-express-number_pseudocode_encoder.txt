CLASS Solution  
    FUNCTION leastOpsExpressTarget(x integer_param target integer_param) RETURNS integer  
        FUNCTION dp(target integer_param) RETURNS integer  
            IF target EQUALS zero  
                RETURN negative one  
            END IF  
            IF target LESS THAN x  
                RETURN the minimum BETWEEN the value of two MULTIPLIED BY target MINUS one AND the value of two MULTIPLIED BY x MINUS target  
            END IF  
            SET power TO one  
            WHILE x RAISED TO THE power LESS THAN OR EQUAL TO target  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET res TO dp(target MINUS x RAISED TO THE power) PLUS power  
            IF x RAISED TO THE power PLUS one MINUS target LESS THAN target  
                SET res TO the minimum BETWEEN res AND dp(x RAISED TO THE power PLUS one MINUS target) PLUS power PLUS one  
            END IF  
            RETURN res  
        END FUNCTION  
        RETURN dp(target)  
    END FUNCTION  
END CLASS
