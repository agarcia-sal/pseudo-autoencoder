CLASS Solution  
    FUNCTION shortestSuperstring(words)  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap(i, j)  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i of words AND the LENGTH OF element at position j of words DOWN TO one  
                IF the SUBSTRING of the element at position i of words from the POSITION of the LENGTH OF the element at position i of words MINUS k PLUS one TO the END EQUALS the SUBSTRING of the element at position j of words from the START TO the POSITION k THEN  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp(mask, i)  
            IF mask EQUALS the result of ONE LEFT SHIFTED BY n MINUS one THEN  
                RETURN a TUPLE of zero AND empty string  
            END IF  
            
            SET min_len TO positive infinity  
            SET best_path TO empty string  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND of mask AND the result of ONE LEFT SHIFTED BY j EQUALS zero THEN  
                    SET result TO the CALL TO dp with the BITWISE OR of mask AND the result of ONE LEFT SHIFTED BY j AND j  
                    SET length TO the FIRST ELEMENT of result PLUS the LENGTH OF element at position j of words MINUS the CALL TO overlap with i AND j  
                    SET path TO the SECOND ELEMENT of result  
                    IF length LESS THAN min_len THEN  
                        SET min_len TO length  
                        SET best_path TO the SUBSTRING of element at position j of words from the POSITION of the CALL TO overlap with i AND j TO the END CONCATENATED WITH path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN a TUPLE of min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO positive infinity  
        SET shortest_path TO empty string  
        FOR i FROM zero TO n MINUS one  
            SET result TO the CALL TO dp with the result of ONE LEFT SHIFTED BY i AND i  
            SET length TO the FIRST ELEMENT of result PLUS the LENGTH OF element at position i of words  
            SET path TO the SECOND ELEMENT of result  
            IF length LESS THAN min_len THEN  
                SET min_len TO length  
                SET shortest_path TO the ELEMENT at position i of words CONCATENATED WITH path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
