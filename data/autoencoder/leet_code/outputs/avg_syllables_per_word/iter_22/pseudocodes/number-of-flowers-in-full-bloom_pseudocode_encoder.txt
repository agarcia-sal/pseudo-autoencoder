CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET list_of_start_times TO a list consisting of the first element of each pair in flowers  
        SET list_of_end_times TO a list consisting of the second element of each pair in flowers  
        SET start_times TO list_of_start_times sorted in ascending order  
        SET end_times TO list_of_end_times sorted in ascending order  
        SET result TO an empty list  
        FOR each person IN people  
            SET number_of_started_flowers TO the position where person would be inserted to keep start_times sorted such that all elements to the left are less than or equal to person  
            SET number_of_ended_flowers TO the position where person would be inserted to keep end_times sorted such that all elements to the left are strictly less than person  
            SET number_of_flowers_in_bloom TO number_of_started_flowers MINUS number_of_ended_flowers  
            APPEND number_of_flowers_in_bloom TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
