CLASS Solution  
    FUNCTION hasPath(maze, start, destination)  
        SET directions TO list containing pair of zero and one pair of one and zero pair of zero and negative one pair of negative one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET visited TO empty set  
        
        FUNCTION dfs(x, y)  
            IF pair of x and y EXISTS IN visited THEN  
                RETURN False  
            END IF  
            IF list containing x and y EQUALS destination THEN  
                RETURN True  
            END IF  
            ADD pair of x and y TO visited  
            
            FOR each pair of dx and dy IN directions  
                SET nx TO x  
                SET ny TO y  
                
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND element at position nx PLUS dx of maze AND element at position ny PLUS dy of that maze EQUALS zero DO  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                
                IF dfs(nx, ny) THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN dfs(element at position zero of start, element at position one of start)  
    END FUNCTION  
END CLASS
