CLASS Solution  
    FUNCTION kMirror(k n)  
        FUNCTION is_palindrome(s)  
            RETURN s EQUALS the REVERSE of s  
        END FUNCTION  
        
        FUNCTION to_base_k(num k)  
            IF num EQUALS zero THEN  
                RETURN the string representing zero  
            END IF  
            SET digits TO empty list  
            WHILE num GREATER THAN zero DO  
                APPEND the remainder of num DIVIDED BY k TO digits  
                SET num TO the integer part of num DIVIDED BY k  
            END WHILE  
            RETURN the concatenation of the string representation of each element in digits in reverse order  
        END FUNCTION  
        
        FUNCTION generate_palindromes(length)  
            IF length EQUALS one THEN  
                FOR i FROM one TO nine DO  
                    YIELD the string representation of i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO THE power of the integer part of the division of length MINUS one BY two  
                SET end TO ten RAISED TO THE power of the integer part of the division of length PLUS one BY two  
                FOR half FROM start TO one LESS THAN end DO  
                    SET half_str TO the string representation of half  
                    IF length MODULO two EQUALS zero THEN  
                        SET full TO half_str PLUS the reverse of half_str  
                    ELSE  
                        SET full TO half_str PLUS the reverse of half_str EXCLUDING the last character  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO empty list  
        SET length TO one  
        WHILE the LENGTH of k_mirror_numbers LESS THAN n DO  
            FOR each pal IN generate_palindromes(length) DO  
                SET num TO the integer representation of pal  
                IF is_palindrome(to_base_k(num k)) THEN  
                    APPEND num TO k_mirror_numbers  
                    IF the LENGTH of k_mirror_numbers EQUALS n THEN  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN the SUM of all elements IN k_mirror_numbers  
    END FUNCTION  
END CLASS
