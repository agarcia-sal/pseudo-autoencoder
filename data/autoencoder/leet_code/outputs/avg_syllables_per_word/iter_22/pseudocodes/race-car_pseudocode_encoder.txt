CLASS Solution  
    FUNCTION racecar(target)  
        ASSIGN reference TO a double ended queue containing a single triple of zero position then one speed then zero steps  
        ASSIGN visited TO a set containing a single tuple of zero position and one speed  
        WHILE the queue is not empty  
            ASSIGN position speed steps TO the first triple removed from the left end of the queue  
            IF position EQUALS target  
                RETURN steps  
            END IF  
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple of next_position and next_speed NOT IN visited AND position greater than or equal to zero AND position less than two MULTIPLIED BY target  
                ADD the tuple of next_position and next_speed TO visited  
                APPEND the triple of next_position then next_speed then steps PLUS one TO the right end of the queue  
            END IF  
            IF speed GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple of position and next_speed NOT IN visited AND position greater than or equal to zero AND position less than two MULTIPLIED BY target  
                ADD the tuple of position and next_speed TO visited  
                APPEND the triple of position then next_speed then steps PLUS one TO the right end of the queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
