CLASS Solution  
    FUNCTION isNStraightHand(hand parameter which is a list of integers groupSize parameter as integer) RETURNS boolean  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN false  
        END IF  
        
        SET count TO the frequency mapping of each element in hand  
        SET unique_cards TO the list of unique elements in count arranged in ascending order  
        
        FOR each card IN unique_cards  
            IF the value associated with card in count GREATER THAN zero THEN  
                SET needed TO the value associated with card in count  
                FOR index FROM zero TO groupSize MINUS one  
                    IF the value associated with the element at position card PLUS index in count LESS THAN needed THEN  
                        RETURN false  
                    END IF  
                    DECREMENT the value associated with the element at position card PLUS index in count BY needed  
                END FOR  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
