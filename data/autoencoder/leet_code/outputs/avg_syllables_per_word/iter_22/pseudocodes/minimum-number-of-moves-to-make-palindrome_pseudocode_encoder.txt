CLASS Solution  
    FUNCTION minMovesToMakePalindrome WITH parameter s  
        SET s TO the list of characters of s  
        SET n TO the LENGTH OF s  
        SET moves TO zero  
        WHILE the LENGTH OF s GREATER THAN one  
            SET match_found TO false  
            FOR i FROM the LENGTH OF s MINUS one DOWN TO one  
                IF the element at position i of s EQUALS the element at position zero of s  
                    FOR j FROM i TO the LENGTH OF s MINUS two  
                        SWAP the element at position j of s WITH the element at position j PLUS one of s  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE the element at position zero FROM s  
                    REMOVE the last element FROM s  
                    SET match_found TO true  
                    BREAK out of the FOR loop  
                END IF  
            END FOR  
            IF match_found IS false  
                REMOVE the element at position zero FROM s  
            END IF  
        END WHILE  
        RETURN moves  
    END FUNCTION  
END CLASS
