CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr parameter, k parameter)  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  
        FOR i FROM zero TO n MINUS one MINUS one  
            PUSH onto min_heap a tuple OF the element at position i of arr DIVIDED BY the element at position n MINUS one of arr the value i the value n MINUS one  
        END FOR  
        FOR dummy_variable FROM zero TO k MINUS one MINUS one  
            REMOVE the smallest element FROM min_heap AND DESTRUCTURE INTO discard_variable i j  
            IF j MINUS one GREATER THAN i  
                PUSH onto min_heap a tuple OF the element at position i of arr DIVIDED BY the element at position j MINUS one of arr the value i the value j MINUS one  
            END IF  
        END FOR  
        REMOVE the smallest element FROM min_heap AND DESTRUCTURE INTO discard_variable i j  
        RETURN a list CONTAINING the element at position i of arr and the element at position j of arr  
    END FUNCTION  
END CLASS
