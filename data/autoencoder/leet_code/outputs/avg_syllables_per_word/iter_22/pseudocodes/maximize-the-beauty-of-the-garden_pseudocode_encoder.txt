CLASS Solution  
    FUNCTION maximumBeauty(flowers)  
        SET indices TO a new dictionary with default value as an empty list  
        SET prefix_sum TO a list of zeros with the length equal to the length of flowers plus one  
        FOR i FROM zero TO the length of flowers MINUS one  
            SET element_at_position i PLUS one of prefix_sum TO element_at_position i of prefix_sum PLUS the greater of element_at_position i of flowers AND zero  
            APPEND i TO the list referenced by element_at_position i of flowers in indices  
        END FOR  
        SET max_beauty TO negative infinity  
        FOR each beauty IN the keys of indices  
            IF the length of the list referenced by beauty in indices IS GREATER THAN OR EQUAL TO two  
                SET first TO element at position zero of the list referenced by beauty in indices  
                SET last TO element at position the last element of the list referenced by beauty in indices  
                SET current_beauty TO two MULTIPLIED BY beauty PLUS element at position last of prefix_sum MINUS element at position first PLUS one of prefix_sum  
                IF current_beauty IS GREATER THAN max_beauty  
                    SET max_beauty TO current_beauty  
                END IF  
            END IF  
        END FOR  
        RETURN max_beauty  
    END FUNCTION  
END CLASS
