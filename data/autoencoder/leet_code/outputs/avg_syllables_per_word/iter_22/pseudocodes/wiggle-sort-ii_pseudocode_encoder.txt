CLASS Solution  
    FUNCTION wiggleSort(nums)  
        CALL sort nums IN ascending order  
        SET sorted_nums TO a copy of nums  
        SET n TO the LENGTH OF nums  
        SET left TO the INTEGER DIVISION OF the MINUS one FROM n BY two  
        SET right TO the MINUS one FROM n  
        FOR i FROM zero TO n MINUS one  
            IF the REMAINDER WHEN i IS DIVIDED BY two EQUALS zero  
                SET element at position i of nums TO element at position left of sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET element at position i of nums TO element at position right of sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
