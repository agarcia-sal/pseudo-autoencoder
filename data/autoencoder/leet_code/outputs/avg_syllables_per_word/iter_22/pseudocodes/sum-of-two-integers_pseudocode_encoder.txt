CLASS Solution  
    FUNCTION getSum(a parameter a INTEGER b parameter b INTEGER) RETURNS INTEGER  
        SET MAX TO the maximum thirty two bits signed integer value  
        SET mask TO the value representing thirty two bits all set to one  
        WHILE b NOT EQUALS zero holds  
            SET carry TO the bitwise AND of a and b MULTIPLIED BY two  
            SET a TO the bitwise XOR of a and b AND the value of mask  
            SET b TO the bitwise AND of carry and mask  
        END WHILE  
        IF a LESS THAN OR EQUAL TO MAX holds  
            RETURN a  
        ELSE  
            RETURN the bitwise complement of the bitwise XOR of a and mask  
        END IF  
    END FUNCTION  
END CLASS
