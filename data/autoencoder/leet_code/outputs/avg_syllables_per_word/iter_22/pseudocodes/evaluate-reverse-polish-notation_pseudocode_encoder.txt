CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUALS plus OR token EQUALS minus OR token EQUALS multiplied by OR token EQUALS divided by  
                SET b TO element at last position of stack  
                REMOVE element at last position FROM stack  
                SET a TO element at last position of stack  
                REMOVE element at last position FROM stack  
                IF token EQUALS plus  
                    APPEND sum of a PLUS b TO stack  
                ELSE IF token EQUALS minus  
                    APPEND difference of a MINUS b TO stack  
                ELSE IF token EQUALS multiplied by  
                    APPEND product of a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS divided by  
                    APPEND integer value of a DIVIDED BY b TO stack  
                END IF  
            ELSE  
                APPEND integer value of token TO stack  
            END IF  
        END FOR  
        RETURN element at first position of stack  
    END FUNCTION  
END CLASS
