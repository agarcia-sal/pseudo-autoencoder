CLASS Solution  
    FUNCTION minRefuelStops WITH PARAMETERS target startFuel stations  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND list composed of target AND zero TO stations  

        FOR each element composed of position AND fuel IN stations  
            WHILE the LENGTH OF max_heap is greater than zero AND current_fuel LESS THAN position  
                REMOVE the element of largest value FROM max_heap AND SUBTRACT it FROM current_fuel  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel LESS THAN position  
                RETURN negative one  
            END IF  
            ADD negative fuel TO max_heap maintaining heap property  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
