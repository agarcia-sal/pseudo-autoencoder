CLASS Solution  
    FUNCTION findMaximizedCapital(k INTEGER_PARAMETER w INTEGER_PARAMETER profits LIST_PARAMETER capital LIST_PARAMETER) RETURNS INTEGER  
        SET projects TO the LIST OF pairs of element at position index of capital AND element at position index of profits FOR index FROM the START TO the LENGTH OF capital MINUS one  
        SORT projects BY the first element of each pair IN ascending order  
        SET max_heap TO an EMPTY LIST  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        FOR loop_variable FROM one TO k  
            WHILE i LESS THAN n AND the first element of the pair at position i of projects LESS THAN OR EQUAL TO w  
                ADD the NEGATIVE of the second element of the pair at position i of projects TO max_heap USING a HEAP PUSH OPERATION  
                INCREMENT i BY one  
            END WHILE  
            IF max_heap IS empty  
                BREAK the FOR LOOP  
            END IF  
            SET w TO w PLUS the NEGATIVE of the HEAP POP from max_heap  
        END FOR  
        RETURN w  
    END FUNCTION  
END CLASS
