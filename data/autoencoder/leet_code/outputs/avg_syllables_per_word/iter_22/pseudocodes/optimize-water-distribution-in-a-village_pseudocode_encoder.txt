CLASS Solution  
    FUNCTION minCostToSupplyWater(n, wells, pipes)  
        FOR index FROM zero TO n MINUS one  
            APPEND a list composed of zero AND index PLUS one AND element at position index of wells TO pipes  
        END FOR  
        SORT pipes IN ascending order based on the element at position two of each sublist  
        SET parent TO a list containing integers from zero TO n inclusive  
        FUNCTION find(x)  
            IF element at position x of parent NOT EQUALS x  
                SET element at position x of parent TO find(element at position x of parent)  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        FUNCTION union(x, y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX NOT EQUALS rootY  
                SET element at position rootX of parent TO rootY  
                RETURN true  
            END IF  
            RETURN false  
        END FUNCTION  
        SET min_cost TO zero  
        FOR each element consisting of house1 AND house2 AND cost IN pipes  
            IF union(house1, house2) EQUALS true  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
