CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO an empty list  
        FOR each num IN the elements of nums traversed from the end to the beginning  
            IF num LESS THAN second  
                RETURN true  
            END IF  
            WHILE the stack is not empty AND element at the last position of stack LESS THAN num  
                SET second TO element removed from the last position of stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
