CLASS Solution  
    FUNCTION findNumOfValidWords(words, puzzles)  
        FUNCTION to_mask(s)  
            SET mask TO zero  
            FOR each character IN s  
                SET mask TO mask OR one LEFT SHIFT position of character in alphabet starting from letter a  
            END FOR  
            RETURN mask  
        END FUNCTION  
        
        ASSIGN word_count TO a new mapping with default value zero  
        FOR each word IN words  
            SET mask TO to_mask of word  
            IF the number of set bits in mask IS LESS THAN OR EQUAL TO seven THEN  
                INCREMENT value at key mask in word_count BY one  
            END IF  
        END FOR  
        
        SET result TO an empty list  
        FOR each puzzle IN puzzles  
            SET first_char_mask TO one LEFT SHIFT position of first character in puzzle in alphabet starting from letter a  
            SET puzzle_mask TO to_mask of substring from second character to end of puzzle  
            SET count TO the value from word_count at key first_char_mask  
            
            SET subset TO puzzle_mask  
            WHILE subset IS NOT EQUAL TO zero  
                SET count TO count PLUS the value from word_count at key subset OR first_char_mask  
                SET subset TO subset MINUS one AND puzzle_mask  
            END WHILE  
            
            APPEND count TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
