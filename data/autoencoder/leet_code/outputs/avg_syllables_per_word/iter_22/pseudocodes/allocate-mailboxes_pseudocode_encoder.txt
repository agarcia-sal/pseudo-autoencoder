CLASS Solution  
    FUNCTION minDistance(houses, k)  
        SORT the list houses in ascending order  
        SET n TO the LENGTH OF houses  
        
        INITIALIZE cost AS a two dimensional list with n rows and n columns each filled with zero  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET element at position i and position j of cost TO element at position i PLUS one and position j MINUS one of cost PLUS element at position j of houses MINUS element at position i of houses  
            END FOR  
        END FOR  
        
        INITIALIZE dp AS a two dimensional list with n PLUS one rows and k PLUS one columns each filled with a very large number representing infinity  
        SET element at position zero and position zero of dp TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET element at position i and position j of dp TO the smaller value BETWEEN element at position i and position j of dp AND element at position p and position j MINUS one of dp PLUS element at position p and position i MINUS one of cost  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position n and position k of dp  
    END FUNCTION  
END CLASS
