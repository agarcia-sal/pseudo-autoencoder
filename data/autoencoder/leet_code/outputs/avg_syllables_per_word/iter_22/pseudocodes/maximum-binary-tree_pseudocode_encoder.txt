CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode CREATED WITH the element at position one of values  
    SET i TO one  
    SET queue TO a new deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element REMOVED FROM THE FRONT OF queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO a new TreeNode CREATED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO a new TreeNode CREATED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the val ATTRIBUTE OF p NOT EQUALS the val ATTRIBUTE OF q  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS the left ATTRIBUTE OF p AND the left ATTRIBUTE OF q AND the RESULT OF is_same_tree WITH PARAMETERS the right ATTRIBUTE OF p AND the right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree WITH PARAMETERS self nums  
        IF the LENGTH OF nums EQUALS zero  
            RETURN none  
        END IF  
        SET max_val TO the maximum value IN nums  
        SET max_index TO the position OF max_val IN nums  
        SET root TO a new TreeNode CREATED WITH max_val  
        SET root.left TO the RESULT OF self.constructMaximumBinaryTree WITH the SUBLIST OF nums FROM position one TO position max_index MINUS one  
        SET root.right TO the RESULT OF self.constructMaximumBinaryTree WITH the SUBLIST OF nums FROM position max_index PLUS one TO THE END OF nums  
        RETURN root  
    END FUNCTION  
END CLASS
