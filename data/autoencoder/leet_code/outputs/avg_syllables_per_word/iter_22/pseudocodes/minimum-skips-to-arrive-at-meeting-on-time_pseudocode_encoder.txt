CLASS Solution  
    FUNCTION minSkips(dist speed hoursBefore)  
        SET n TO the LENGTH OF dist  
        CREATE dp AS A two dimensional list with n rows AND n PLUS one columns FILLED WITH infinite time values  
        SET total_dist TO the SUM OF all elements IN dist  
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore  
            RETURN negative one  
        END IF  
        
        SET the element at position zero of the first row IN dp TO the element at position zero of dist  
        FOR j FROM one TO n MINUS one  
            SET the element at position zero of the jth column IN dp TO the element at position zero of dist  
        END FOR  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j LESS THAN i  
                    SET dp at position i j TO the smallest integer NOT LESS THAN dp at position i MINUS one j DIVIDED BY speed MULTIPLIED BY speed PLUS the element at position i of dist  
                END IF  
                IF j GREATER THAN zero  
                    SET dp at position i j TO the minimum BETWEEN dp at position i j AND dp at position i MINUS one j MINUS one PLUS the element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            IF dp at position n MINUS one j LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
