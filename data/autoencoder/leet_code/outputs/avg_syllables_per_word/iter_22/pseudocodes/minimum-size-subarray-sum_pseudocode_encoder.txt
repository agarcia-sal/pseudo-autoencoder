CLASS Solution  
    FUNCTION minSubArrayLen(target parameter, nums parameter)  
        SET left TO zero  
        SET current_sum TO zero  
        SET min_length TO infinity value  
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            INCREMENT current_sum BY the element at position right of nums  
            WHILE current_sum GREATER THAN OR EQUAL TO target  
                IF min_length LESS THAN min_length AND min_length LESS THAN OR EQUAL TO right MINUS left PLUS one THEN  
                    SET min_length TO min_length  
                ELSE  
                    SET min_length TO right MINUS left PLUS one  
                END IF  
                DECREMENT current_sum BY the element at position left of nums  
                INCREMENT left BY one  
            END WHILE  
        END FOR  
        IF min_length NOT EQUALS infinity value THEN  
            RETURN min_length  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
