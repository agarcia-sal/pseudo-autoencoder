CLASS Solution  
    FUNCTION alienOrder(words)  
        SET graph TO a mapping of characters to an empty set of characters  
        SET indegree TO a mapping of characters to zero  
        SET all_chars TO the set of all characters appearing in any word of words  
        FOR index FROM zero TO the LENGTH OF words MINUS one MINUS one  
            SET word1 TO the element at position index of words  
            SET word2 TO the element at position index PLUS one of words  
            SET min_length TO the minimum of the LENGTH OF word1 AND the LENGTH OF word2  
            SET found TO false  
            FOR position FROM zero TO min_length MINUS one  
                IF the character at position of word1 NOT EQUALS the character at position of word2  
                    IF the character at position of word2 NOT IN the set at key character at position of word1 in graph  
                        ADD the character at position of word2 TO the set at key character at position of word1 in graph  
                        INCREMENT the value at key character at position of word2 in indegree BY one  
                    END IF  
                    SET found TO true  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF found EQUALS false AND the LENGTH OF word1 GREATER THAN the LENGTH OF word2  
                RETURN an empty string  
            END IF  
        END FOR  
        SET queue TO a double-ended queue initialized with all characters in all_chars for which the value in indegree EQUALS zero  
        SET result TO an empty list  
        WHILE the queue is not empty  
            REMOVE and RETURN the element from the front of queue and assign it to char  
            APPEND char TO result  
            FOR each neighbor IN the set at key char in graph  
                DECREMENT the value at key neighbor in indegree BY one  
                IF the value at key neighbor in indegree EQUALS zero  
                    APPEND neighbor TO the end of queue  
                END IF  
            END FOR  
        END WHILE  
        IF the LENGTH OF result NOT EQUALS the LENGTH OF all_chars  
            RETURN an empty string  
        END IF  
        SET joined_result TO the concatenation of all characters in result as a single string  
        RETURN joined_result  
    END FUNCTION  
END CLASS
