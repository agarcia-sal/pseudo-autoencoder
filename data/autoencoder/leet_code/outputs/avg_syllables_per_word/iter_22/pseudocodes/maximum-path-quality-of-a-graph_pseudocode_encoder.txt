CLASS Solution  
    FUNCTION maximalPathQuality(values, edges, maxTime)  
        SET graph TO a new mapping from elements to empty lists  
        FOR each element u, element v, element time IN edges  
            APPEND the pair of element v and element time TO the list at key u in graph  
            APPEND the pair of element u and element time TO the list at key v in graph  
        END FOR  
          
        SET visited TO a new empty set  
        SET max_quality TO zero  
          
        FUNCTION dfs(node, current_time, current_quality)  
            IF node EQUALS zero THEN  
                IF current_quality GREATER THAN max_quality THEN  
                    SET max_quality TO current_quality  
                END IF  
            END IF  
              
            FOR each pair neighbor and time IN the list at key node in graph  
                IF time LESS THAN OR EQUAL TO current_time THEN  
                    IF neighbor NOT IN visited THEN  
                        ADD neighbor TO visited  
                        CALL dfs with neighbor, current_time MINUS time, current_quality PLUS element at position neighbor of values  
                        REMOVE neighbor FROM visited  
                    ELSE  
                        CALL dfs with neighbor, current_time MINUS time, current_quality  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
          
        ADD zero TO visited  
        CALL dfs with zero, maxTime, element at position zero of values  
          
        RETURN max_quality  
    END FUNCTION  
END CLASS
