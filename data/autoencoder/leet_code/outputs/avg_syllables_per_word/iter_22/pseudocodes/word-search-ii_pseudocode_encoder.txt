CLASS Trie  
    FUNCTION __init__(self)  
        SET self.children TO a list containing twenty-six times none  
        SET self.ref TO negative one  
    END FUNCTION  

    FUNCTION insert(self, w, ref)  
        SET node TO self  
        FOR each character c IN w  
            SET idx TO the numerical position of character c in the alphabet MINUS the numerical position of letter a in the alphabet  
            IF the element at position idx of node.children EQUALS none  
                SET the element at position idx of node.children TO a new instance of Trie  
            END IF  
            SET node TO the element at position idx of node.children  
        END FOR  
        SET node.ref TO ref  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION findWords(self, board, words)  

        FUNCTION dfs(node, i, j)  
            SET idx TO the numerical position of the element at position i of the elements at position j of board MINUS the numerical position of letter a in the alphabet  
            IF the element at position idx of node.children EQUALS none  
                RETURN  
            END IF  
            SET node TO the element at position idx of node.children  
            IF node.ref GREATER THAN OR EQUAL TO zero  
                APPEND the element at position node.ref of words TO ans  
                SET node.ref TO negative one  
            END IF  
            SET c TO the element at position i of the elements at position j of board  
            SET the element at position i of the elements at position j of board TO the character hash sign  
            FOR each pair a and b taken successively from the sequence negative one zero one zero negative one  
                SET x TO i PLUS a  
                SET y TO j PLUS b  
                IF x GREATER THAN OR EQUAL TO zero AND x LESS THAN m AND y GREATER THAN OR EQUAL TO zero AND y LESS THAN n AND the element at position x of the elements at position y of board NOT EQUALS the character hash sign  
                    CALL dfs WITH node AND x AND y  
                END IF  
            END FOR  
            SET the element at position i of the elements at position j of board TO c  
        END FUNCTION  

        SET tree TO a new instance of Trie  
        FOR each index i and element w enumerated from words  
            CALL insert method of tree WITH w AND i  
        END FOR  
        SET m TO the number of elements in board  
        SET n TO the number of elements in the element at position zero of board  
        SET ans TO an empty list  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                CALL dfs WITH tree AND i AND j  
            END FOR  
        END FOR  
        RETURN ans  
    END FUNCTION  
END CLASS
