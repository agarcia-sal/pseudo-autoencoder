CLASS Solution  
    FUNCTION shortestDistance(maze AS List of List of integers start AS List of integers destination AS List of integers) RETURNS integer  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO list of pairs where first pair is one and zero and second pair is negative one and zero and third pair is zero and one and fourth pair is zero and negative one  
        SET start TO tuple conversion of start  
        SET destination TO tuple conversion of destination  
        SET visited TO an empty set  
        SET pq TO a list containing a pair where first element is zero and second element is start  

        WHILE pq is not empty  
            REMOVE and ASSIGN the element with smallest first value of pq TO dist and the pair x and y  
            
            IF x EQUALS the first element of destination AND y EQUALS the second element of destination  
                RETURN dist  
            END IF  
            
            IF the pair x and y IS IN visited  
                CONTINUE to next iteration  
            END IF  
            
            ADD the pair x and y TO visited  
            
            FOR each pair dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE the sum of new_x and dx IS GREATER THAN OR EQUAL TO zero AND the sum of new_x and dx IS LESS THAN m AND the sum of new_y and dy IS GREATER THAN OR EQUAL TO zero AND the sum of new_y and dy IS LESS THAN n AND the element at position the sum of new_x and dx of maze AND at position the sum of new_y and dy IS EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF the pair new_x and new_y IS NOT IN visited  
                    ADD the pair where first element is the sum of dist PLUS count and second element is the pair new_x and new_y TO pq with heap push operation  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
