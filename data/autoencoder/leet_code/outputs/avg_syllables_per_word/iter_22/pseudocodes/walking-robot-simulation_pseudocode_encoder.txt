CLASS Solution  
    FUNCTION robotSim(commands, obstacles)  
        SET directions TO list of pairs where first pair is zero and one second pair is one and zero third pair is zero and negative one fourth pair is negative one and zero  
        SET x TO zero  
        SET y TO zero  
        SET dir_index TO zero  
        SET obstacles_set TO empty set  
        FOR each obstacle IN obstacles  
            CONVERT element at position zero of obstacle and element at position one of obstacle TO pair  
            ADD the pair TO obstacles_set  
        END FOR  
        SET max_distance TO zero  
        FOR each command IN commands  
            IF command EQUALS negative two  
                SET dir_index TO dir_index MINUS one MODULO four  
            ELSE IF command EQUALS negative one  
                SET dir_index TO dir_index PLUS one MODULO four  
            ELSE  
                SET dx TO element at position zero of element at position dir_index of directions  
                SET dy TO element at position one of element at position dir_index of directions  
                FOR index FROM one TO command  
                    SET next_x TO x PLUS dx  
                    SET next_y TO y PLUS dy  
                    IF pair of next_x and next_y IS IN obstacles_set  
                        BREAK the loop  
                    END IF  
                    SET x TO next_x  
                    SET y TO next_y  
                END FOR  
                SET current_distance TO x MULTIPLIED BY x PLUS y MULTIPLIED BY y  
                IF current_distance GREATER THAN max_distance  
                    SET max_distance TO current_distance  
                END IF  
            END IF  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
