CLASS Solution  
    FUNCTION findItinerary WITH PARAMETER tickets  
        SET graph TO a new default dictionary that returns an empty list for missing keys  
        FOR each pair of departure and arrival IN the list of tickets sorted in ascending order and then reversed  
            APPEND arrival TO the list at key departure in graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit WITH PARAMETER airport  
            WHILE the list at key airport in graph is not empty  
                CALL visit WITH the last element removed from the list at key airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit WITH argument the string consisting of the letters J F K  
        RETURN the list itinerary in reversed order  
    END FUNCTION  
END CLASS
