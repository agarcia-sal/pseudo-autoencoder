CLASS Solution  
    FUNCTION findKthNumber WITH PARAMETERS m n k  
        FUNCTION enough WITH PARAMETER x  
            SET count TO zero  
            FOR i FROM one TO m  
                SET quotient TO the number x DIVIDED BY i  
                SET minimum_value TO the smaller BETWEEN quotient AND n  
                INCREMENT count BY minimum_value  
            END FOR  
            IF count GREATER THAN OR EQUAL TO k  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET low TO one  
        SET high TO the number m MULTIPLIED BY n  
        WHILE the value low LESS THAN high  
            SET mid TO the number low PLUS high DIVIDED BY two  
            IF the result of enough WITH mid EQUALS true  
                SET high TO mid  
            ELSE  
                SET low TO the number mid PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
