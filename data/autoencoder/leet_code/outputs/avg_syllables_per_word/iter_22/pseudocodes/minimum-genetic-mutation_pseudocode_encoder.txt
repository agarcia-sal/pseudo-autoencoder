CLASS Solution  
    FUNCTION minMutation(start STRING end STRING bank LIST OF STRING) RETURNS INTEGER  
        SET s TO the SET OF elements IN bank  
        SET q TO a double ended queue CONTAINING a TUPLE OF start AND zero  
        SET mp TO a MAPPING FROM character A TO the STRING of characters T C G AND FROM character T TO the STRING of characters A C G AND FROM character C TO the STRING of characters A T G AND FROM character G TO the STRING of characters A T C  
        WHILE q is not empty  
            SET t TO the first element of the first element of q  
            SET step TO the second element of the first element of q  
            REMOVE the first element from q  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR each index i AND character v IN the sequence of characters IN t  
                FOR each character j IN the string mapped from v IN mp  
                    SET next TO the concatenation of the substring of t from position start TO position i MINUS one PLUS character j PLUS the substring of t from position i PLUS one TO the end  
                    IF next IS IN s  
                        APPEND a TUPLE OF next AND step PLUS one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN integer negative one  
    END FUNCTION  
END CLASS
