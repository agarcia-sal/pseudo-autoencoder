CLASS Solution  
    FUNCTION addOperators(num AS string, target AS integer) RETURNS list OF string  
        FUNCTION dfs(index AS integer, path AS string, value AS integer, prev AS integer)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO substring from position index TO position i MINUS one of num  
                IF the LENGTH OF operand GREATER THAN one AND element at position one of operand EQUALS character zero  
                    CONTINUE  
                END IF  
                
                IF index EQUALS zero  
                    CALL dfs WITH arguments i, operand, integer VALUE OF operand, integer VALUE OF operand  
                ELSE  
                    CALL dfs WITH arguments i, path PLUS the character plus PLUS operand, value PLUS integer VALUE OF operand, integer VALUE OF operand  
                    CALL dfs WITH arguments i, path PLUS the character minus PLUS operand, value MINUS integer VALUE OF operand, NEGATIVE integer VALUE OF operand  
                    CALL dfs WITH arguments i, path PLUS the character asterisk PLUS operand, value MINUS prev PLUS prev MULTIPLIED BY integer VALUE OF operand, prev MULTIPLIED BY integer VALUE OF operand  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL dfs WITH arguments zero, an empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
