CLASS Solution  
    FUNCTION findIntegers WITH parameter n  
        SET binary representation string TO the binary form of n starting from the second character to the end  
        SET length TO the length of the binary representation string  
        INITIALIZE a list named dp with zeros of the size length plus one  
        SET the element at position zero of dp TO one  
        SET the element at position one of dp TO two  
        FOR index i FROM two TO length  
            SET the element at position i of dp TO the sum of the elements at positions i minus one AND i minus two of dp  
        END FOR  
        SET result TO zero  
        SET previous bit TO zero  
        FOR index i FROM zero TO length minus one  
            IF the character at position i of the binary representation string EQUALS one  
                INCREMENT result BY the element at position length MINUS i MINUS one of dp  
                IF previous bit EQUALS one  
                    DECREMENT result BY one  
                    EXIT THE LOOP  
                END IF  
            END IF  
            SET previous bit TO one IF the character at position i of the binary representation string EQUALS one OTHERWISE zero  
        END FOR  
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
