CLASS Solution  
    FUNCTION maxJumps(arr parameter, d parameter)  
        FUNCTION dp(i parameter)  
            SET max jumps TO one  
            FOR x parameter FROM one TO d parameter  
                IF sum of i parameter PLUS x parameter LESS THAN the LENGTH OF arr parameter  
                    IF element at position sum of i parameter PLUS x parameter of arr parameter GREATER THAN OR EQUAL TO element at position i parameter of arr parameter  
                        BREAK the loop  
                    END IF  
                    SET max jumps TO the GREATER OF max jumps AND the sum of one PLUS the RESULT OF dp WITH argument sum of i parameter PLUS x parameter  
                END IF  
            END FOR  
            FOR x parameter FROM one TO d parameter  
                IF difference of i parameter MINUS x parameter GREATER THAN OR EQUAL TO zero  
                    IF element at position difference of i parameter MINUS x parameter of arr parameter GREATER THAN OR EQUAL TO element at position i parameter of arr parameter  
                        BREAK the loop  
                    END IF  
                    SET max jumps TO the GREATER OF max jumps AND the sum of one PLUS the RESULT OF dp WITH argument difference of i parameter MINUS x parameter  
                END IF  
            END FOR  
            RETURN max jumps  
        END FUNCTION  
        RETURN the GREATER VALUE among the RESULTS OF dp FOR each i parameter FROM zero TO the LENGTH OF arr parameter MINUS one  
    END FUNCTION  
END CLASS
