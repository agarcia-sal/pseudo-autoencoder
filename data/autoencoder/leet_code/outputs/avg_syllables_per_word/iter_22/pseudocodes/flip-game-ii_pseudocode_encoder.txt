CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF state EXISTS AS A KEY IN memo THEN  
                RETURN the element at key state of memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring from position index TO position index PLUS one of state EQUALS the concatenation of two plus signs THEN  
                    SET new_state TO the concatenation of the substring from the start position TO position index MINUS one of state PLUS two minus signs PLUS the substring from position index PLUS two TO the end position of state  
                    IF NOT the result of calling can_win_helper with argument new_state THEN  
                        SET the element at key state of memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the element at key state of memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN the result of calling can_win_helper with argument currentState  
    END FUNCTION  
END CLASS
