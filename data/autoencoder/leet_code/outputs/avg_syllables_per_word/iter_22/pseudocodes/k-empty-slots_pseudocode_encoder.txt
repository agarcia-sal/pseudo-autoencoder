CLASS Solution  
    FUNCTION kEmptySlots(bulbs, k)  
        SET bloom_days TO an empty list  
        FOR each day and position IN enumeration of bulbs starting from one  
            SET i TO the leftmost position where the current position can be inserted in bloom_days to keep it sorted  
            IF i is greater than zero AND position MINUS element at position i MINUS one of bloom_days MINUS one EQUALS k THEN  
                RETURN day  
            END IF  
            IF i is less than the LENGTH OF bloom_days AND element at position i of bloom_days MINUS position MINUS one EQUALS k THEN  
                RETURN day  
            END IF  
            INSERT position INTO bloom_days at position i  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
