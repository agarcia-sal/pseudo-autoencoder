CLASS Solution  
    FUNCTION largestNumber(cost parameter of type List of integers target parameter of type integer) RETURNS string  
        SET cost_to_digit TO an empty mapping from integer to string  
        FOR each index i and element c IN cost with enumeration starting from zero  
            ASSIGN to cost_to_digit at key c the string representation of index i PLUS one  
        END FOR  
        
        SET dp TO a list containing one empty string followed by a sequence of strings each equal to string minus one repeated target times  
        
        FOR each integer t FROM one TO target inclusive  
            FOR each key c and value d IN cost_to_digit items  
                IF t GREATER THAN OR EQUAL TO c AND element at position t MINUS c of dp NOT EQUALS string minus one  
                    SET candidate TO the concatenation of d followed by element at position t MINUS c of dp  
                    IF element at position t of dp EQUALS string minus one OR length of candidate GREATER THAN length of element at position t of dp OR (length of candidate EQUALS length of element at position t of dp AND candidate GREATER THAN element at position t of dp)  
                        SET element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF element at position target of dp NOT EQUALS string minus one  
            RETURN element at position target of dp  
        ELSE  
            RETURN string zero  
        END IF  
    END FUNCTION  
END CLASS
