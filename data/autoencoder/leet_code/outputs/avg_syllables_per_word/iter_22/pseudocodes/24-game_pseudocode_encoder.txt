CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the ELEMENT at position one of nums MINUS twenty four LESS THAN one times ten TO THE POWER OF negative six  
            END IF  
            
            FOR i FROM one TO the LENGTH OF nums  
                FOR j FROM one TO the LENGTH OF nums  
                    IF i NOT EQUALS j  
                        SET new_nums TO an empty list  
                        FOR k FROM one TO the LENGTH OF nums  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND the ELEMENT at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR op IN the OPERATIONS truediv mul add sub  
                            IF (op EQUALS add OR op EQUALS mul) AND j GREATER THAN i  
                                CONTINUE to the next iteration of the loop  
                            END IF  
                            
                            IF op NOT EQUALS truediv OR the ELEMENT at position j of nums NOT EQUALS zero  
                                APPEND the RESULT of APPLYING op TO the ELEMENT at position i of nums AND the ELEMENT at position j of nums TO new_nums  
                                IF solve(new_nums) HOLDS TRUE  
                                    RETURN true  
                                END IF  
                                REMOVE the LAST ELEMENT FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
