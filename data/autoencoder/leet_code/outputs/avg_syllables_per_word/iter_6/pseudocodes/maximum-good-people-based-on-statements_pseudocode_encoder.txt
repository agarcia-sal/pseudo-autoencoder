CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO LENGTH OF statements  
        SET max_good TO 0  
        
        FOR i FROM 0 TO (2 RAISED TO THE POWER OF n) MINUS 1  
            SET good TO list of n elements all set to False  
            SET bad TO list of n elements all set to False  
            
            FOR j FROM 0 TO n MINUS 1  
                IF binary representation of i HAS bit j SET THEN  
                    SET good[j] TO True  
                ELSE  
                    SET bad[j] TO True  
            
            SET valid TO True  
            FOR j FROM 0 TO n MINUS 1  
                IF good[j] IS True THEN  
                    FOR k FROM 0 TO n MINUS 1  
                        IF (statements[j][k] EQUALS 0 AND good[k] IS True) OR (statements[j][k] EQUALS 1 AND bad[k] IS True) THEN  
                            SET valid TO False  
                            BREAK inner loop  
                    IF valid IS False THEN  
                        BREAK outer loop  
            
            IF valid IS True THEN  
                SET count_good TO SUM OF True values IN good  
                IF count_good GREATER THAN max_good THEN  
                    SET max_good TO count_good  
        
        RETURN max_good  
    END FUNCTION  
END CLASS
