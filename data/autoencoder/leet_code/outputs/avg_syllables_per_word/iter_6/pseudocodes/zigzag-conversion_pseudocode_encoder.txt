CLASS Solution  
    FUNCTION convert(s, numRows)  
        IF numRows EQUALS 1  
            RETURN s  
        END IF  

        SET rows TO list of empty strings with length numRows  
        SET current_row TO 0  
        SET going_down TO FALSE  

        FOR each char IN s  
            APPEND char TO rows AT current_row  
            IF current_row EQUALS 0 OR current_row EQUALS numRows MINUS 1  
                SET going_down TO NOT going_down  
            END IF  
            IF going_down IS TRUE  
                INCREMENT current_row BY 1  
            ELSE  
                DECREMENT current_row BY 1  
            END IF  
        END FOR  

        SET result TO concatenation of all strings in rows  
        RETURN result  
    END FUNCTION  
END CLASS
