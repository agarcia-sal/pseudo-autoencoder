CLASS Solution  
    FUNCTION kMirror(k, n)  
        FUNCTION is_palindrome(s)  
            RETURN s EQUALS s REVERSED  
        END FUNCTION  
        
        FUNCTION to_base_k(num, k)  
            IF num EQUALS 0 THEN  
                RETURN "0"  
            END IF  
            SET digits TO empty list  
            WHILE num IS GREATER THAN 0  
                APPEND remainder of num DIVIDED BY k TO digits  
                SET num TO num DIVIDED BY k (integer division)  
            END WHILE  
            RETURN concatenation of digits REVERSED as strings  
        END FUNCTION  
        
        FUNCTION generate_palindromes(length)  
            IF length EQUALS 1 THEN  
                FOR i FROM 1 TO 9  
                    YIELD string representation of i  
                END FOR  
            ELSE  
                SET start TO TEN RAISED TO power of (length MINUS 1) DIVIDED BY 2 (integer division)  
                SET end TO TEN RAISED TO power of (length PLUS 1) DIVIDED BY 2 (integer division)  
                FOR half FROM start TO end MINUS 1  
                    SET half_str TO string representation of half  
                    IF length MODULO 2 EQUALS 0 THEN  
                        SET full TO half_str CONCATENATED WITH half_str REVERSED  
                    ELSE  
                        SET full TO half_str CONCATENATED WITH half_str EXCLUDING LAST character REVERSED  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO empty list  
        SET length TO 1  
        WHILE LENGTH OF k_mirror_numbers IS LESS THAN n  
            FOR each pal IN generate_palindromes(length)  
                SET num TO numeric value of pal  
                IF is_palindrome(to_base_k(num, k)) IS TRUE THEN  
                    APPEND num TO k_mirror_numbers  
                    IF LENGTH OF k_mirror_numbers EQUALS n THEN  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY 1  
        END WHILE  
        
        RETURN sum of k_mirror_numbers  
    END FUNCTION  
END CLASS
