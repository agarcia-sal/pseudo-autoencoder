CLASS Solution  
    FUNCTION tilingRectangle(n, m)  
        FUNCTION backtrack(heights, moves)  
            IF all elements IN heights EQUAL n THEN  
                SET min_moves TO minimum OF min_moves AND moves  
                RETURN  
            END IF  
            
            IF moves GREATER THAN OR EQUAL TO min_moves THEN  
                RETURN  
            END IF  
            
            SET min_height TO minimum element IN heights  
            SET left TO index OF min_height IN heights  
            SET right TO left  
            WHILE right LESS THAN m AND heights AT right EQUALS min_height  
                INCREMENT right BY 1  
            END WHILE  
            
            FOR size FROM right MINUS left DOWNTO 1  
                IF for ALL i FROM 0 TO size MINUS 1 heights AT left PLUS i PLUS size LESS THAN OR EQUAL TO n THEN  
                    SET new_heights TO COPY OF heights  
                    FOR i FROM 0 TO size MINUS 1  
                        INCREMENT new_heights AT left PLUS i BY size  
                    END FOR  
                    CALL backtrack WITH new_heights AND moves PLUS 1  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET min_moves TO infinity  
        CALL backtrack WITH list OF m ZEROS AND 0  
        RETURN min_moves  
    END FUNCTION  
END CLASS
