CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO list containing zero  
        FOR each num IN nums  
            APPEND last element of prefix_sum PLUS num TO prefix_sum  
        SET dq TO empty double-ended queue  
        SET min_length TO infinity  
        FOR index i, current_sum IN enumeration of prefix_sum  
            WHILE dq is not empty AND current_sum MINUS prefix_sum at front of dq GREATER THAN OR EQUAL TO k  
                SET min_length TO minimum of min_length AND i MINUS front element of dq  
                REMOVE front element FROM dq  
            END WHILE  
            WHILE dq is not empty AND current_sum LESS THAN OR EQUAL TO prefix_sum at back of dq  
                REMOVE back element FROM dq  
            END WHILE  
            APPEND i TO dq  
        END FOR  
        IF min_length EQUALS infinity  
            RETURN negative one  
        ELSE  
            RETURN min_length  
        END IF  
    END FUNCTION  
END CLASS
