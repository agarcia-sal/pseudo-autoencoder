CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO NEW TreeNode WITH val AS first element of values  
    SET i TO 1  
    INITIALIZE queue AS empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from front of queue  
        IF i LESS THAN length of values AND values at i IS NOT None  
            SET node.left TO NEW TreeNode WITH val AS values at i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values at i IS NOT None  
            SET node.right TO NEW TreeNode WITH val AS values at i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(root)  
        INITIALIZE queue AS deque containing root  
        SET leftmost_value TO None  
        WHILE queue IS NOT empty  
            SET level_size TO number of elements in queue  
            FOR i FROM 0 TO level_size MINUS 1  
                SET node TO element removed from front of queue  
                IF i EQUALS 0  
                    SET leftmost_value TO node.val  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
