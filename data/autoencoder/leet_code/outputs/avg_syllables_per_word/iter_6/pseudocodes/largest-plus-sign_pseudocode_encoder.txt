CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO set of tuples created from each mine in mines  
        INITIALIZE dp as a two-dimensional list of size n by n where each element is a list of four zeros  
        SET directions TO list of four direction pairs: right, down, left, up  
  
        FOR i FROM 0 TO n MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF pair (i, j) IS NOT IN mines_set THEN  
                    IF j GREATER THAN 0 THEN  
                        SET dp at position (i, j) index 0 TO dp at position (i, j MINUS 1) index 0 PLUS 1  
                    IF i GREATER THAN 0 THEN  
                        SET dp at position (i, j) index 1 TO dp at position (i MINUS 1, j) index 1 PLUS 1  
  
        SET result TO 0  
        FOR i FROM n MINUS 1 DOWN TO 0  
            FOR j FROM n MINUS 1 DOWN TO 0  
                IF pair (i, j) IS NOT IN mines_set THEN  
                    IF j LESS THAN n MINUS 1 THEN  
                        SET dp at position (i, j) index 2 TO dp at position (i, j PLUS 1) index 2 PLUS 1  
                    IF i LESS THAN n MINUS 1 THEN  
                        SET dp at position (i, j) index 3 TO dp at position (i PLUS 1, j) index 3 PLUS 1  
                    SET plus_sign_order TO minimum value among dp at position (i, j)  
                    SET result TO maximum value BETWEEN result AND plus_sign_order  
  
        RETURN result  
    END FUNCTION  
END CLASS
