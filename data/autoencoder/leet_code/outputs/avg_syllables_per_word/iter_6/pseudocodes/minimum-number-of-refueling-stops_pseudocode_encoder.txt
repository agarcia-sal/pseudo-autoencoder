CLASS Solution  
    FUNCTION minRefuelStops(target, startFuel, stations)  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO 0  
        APPEND list consisting of target AND 0 TO stations  

        FOR each position, fuel IN stations  
            WHILE max_heap IS NOT empty AND current_fuel LESS THAN position  
                SET largest_fuel TO REMOVE the maximum element FROM max_heap  
                DECREMENT current_fuel BY largest_fuel  
                INCREMENT stops BY 1  
            IF current_fuel LESS THAN position  
                RETURN minus one  
            ADD negative of fuel TO max_heap  

        RETURN stops  
    END FUNCTION  
END CLASS
