CLASS Solution  
    FUNCTION minArea(image, x, y)  
        IF image IS EMPTY OR first row of image IS EMPTY  
            RETURN 0  
        
        SET m TO LENGTH OF image  
        SET n TO LENGTH OF first row of image  
        
        FUNCTION hasBlackPixelInRow(row)  
            FOR each pixel IN image at row  
                IF pixel EQUALS '1'  
                    RETURN True  
            RETURN False  
        END FUNCTION  
        
        FUNCTION hasBlackPixelInColumn(col)  
            FOR index FROM 0 TO m MINUS 1  
                IF image at index and col EQUALS '1'  
                    RETURN True  
            RETURN False  
        END FUNCTION  
        
        FUNCTION findTop()  
            SET low TO 0  
            SET high TO x  
            WHILE low LESS THAN high  
                SET mid TO (low PLUS high) DIVIDED BY 2  
                IF hasBlackPixelInRow(mid) IS True  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS 1  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findBottom()  
            SET low TO x  
            SET high TO m MINUS 1  
            WHILE low LESS THAN high  
                SET mid TO (low PLUS high PLUS 1) DIVIDED BY 2  
                IF hasBlackPixelInRow(mid) IS True  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS 1  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findLeft()  
            SET low TO 0  
            SET high TO y  
            WHILE low LESS THAN high  
                SET mid TO (low PLUS high) DIVIDED BY 2  
                IF hasBlackPixelInColumn(mid) IS True  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS 1  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findRight()  
            SET low TO y  
            SET high TO n MINUS 1  
            WHILE low LESS THAN high  
                SET mid TO (low PLUS high PLUS 1) DIVIDED BY 2  
                IF hasBlackPixelInColumn(mid) IS True  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS 1  
            RETURN low  
        END FUNCTION  
        
        SET top TO CALL findTop()  
        SET bottom TO CALL findBottom()  
        SET left TO CALL findLeft()  
        SET right TO CALL findRight()  
        
        RETURN (bottom MINUS top PLUS 1) MULTIPLIED BY (right MINUS left PLUS 1)  
    END FUNCTION  
END CLASS
