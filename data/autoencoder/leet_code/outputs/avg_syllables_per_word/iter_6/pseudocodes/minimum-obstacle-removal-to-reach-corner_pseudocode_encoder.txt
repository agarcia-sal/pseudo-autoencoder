CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF first element IN grid  
        SET directions TO list of tuples (0, 1), (1, 0), (0, -1), and (-1, 0)  
        SET min_heap TO list containing tuple (0, 0, 0) representing cost zero at coordinates zero zero  
        SET visited TO set containing tuple (0, 0)  
        
        WHILE min_heap IS NOT EMPTY  
            CALL function to remove and return smallest element FROM min_heap, SET TO cost, x, y  
            
            IF x EQUALS m MINUS 1 AND y EQUALS n MINUS 1  
                RETURN cost  
            
            FOR each dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND tuple (nx, ny) NOT IN visited  
                    ADD tuple (nx, ny) TO visited  
                    SET new_cost TO cost PLUS value IN grid AT nx, ny  
                    ADD tuple (new_cost, nx, ny) TO min_heap preserving heap order  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
