CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO first element IN nums  
        SET current_sum TO first element IN nums  
        FOR each num IN nums starting from second element  
            SET current_sum TO maximum OF num AND current_sum PLUS num  
            SET max_sum TO maximum OF max_sum AND current_sum  
        RETURN max_sum  
    END FUNCTION  
END CLASS
