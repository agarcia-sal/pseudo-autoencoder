CLASS Solution  
    FUNCTION advantageCount(nums1, nums2)  
        CALL nums1 TO BE SORTED IN ASCENDING ORDER  
        
        SET nums2_with_indices TO list of pairs each containing element from nums2 and its original index, SORTED BY element IN DESCENDING ORDER  

        SET result TO a list of zeros with LENGTH EQUAL TO LENGTH OF nums1  

        SET left TO 0  
        SET right TO LENGTH OF nums1 MINUS 1  

        FOR each pair value, index IN nums2_with_indices  
            IF element at nums1[right] IS GREATER THAN value  
                SET result at index TO element at nums1[right]  
                DECREMENT right BY 1  
            ELSE  
                SET result at index TO element at nums1[left]  
                INCREMENT left BY 1  
        
        RETURN result  
    END FUNCTION  
END CLASS
