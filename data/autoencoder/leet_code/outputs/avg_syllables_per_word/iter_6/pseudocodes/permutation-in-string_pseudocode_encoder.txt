CLASS Solution  
    FUNCTION checkInclusion(s1, s2)  
        SET len_s1 TO LENGTH OF s1  
        SET len_s2 TO LENGTH OF s2  
        
        IF len_s1 GREATER THAN len_s2  
            RETURN False  
        END IF  
        
        SET count_s1 TO frequency count of characters in s1  
        SET count_s2 TO frequency count of characters in first len_s1 characters of s2  
        
        IF count_s1 EQUALS count_s2  
            RETURN True  
        END IF  
        
        FOR index FROM len_s1 TO len_s2 MINUS 1  
            INCREMENT count_s2 at character s2 at position index BY 1  
            DECREMENT count_s2 at character s2 at position (index MINUS len_s1) BY 1  
            
            IF count_s2 at character s2 at position (index MINUS len_s1) EQUALS 0  
                REMOVE count_s2 entry for character s2 at position (index MINUS len_s1)  
            END IF  
            
            IF count_s1 EQUALS count_s2  
                RETURN True  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
