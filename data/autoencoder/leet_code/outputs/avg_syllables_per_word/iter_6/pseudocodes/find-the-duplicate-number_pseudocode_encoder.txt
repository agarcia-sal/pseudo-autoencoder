CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count TO 0  
            FOR each v IN nums  
                IF v LESS THAN OR EQUAL TO x THEN  
                    INCREMENT count BY 1  
            RETURN count GREATER THAN x  
        END FUNCTION

        CALL bisect_left WITH arguments:  
            collection AS range FROM 0 TO LENGTH OF nums MINUS 1  
            value AS True  
            key AS f  
        RETURN result of bisect_left  
    END FUNCTION  
END CLASS
