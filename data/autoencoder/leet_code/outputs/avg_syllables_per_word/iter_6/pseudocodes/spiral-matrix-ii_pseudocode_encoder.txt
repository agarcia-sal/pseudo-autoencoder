CLASS Solution  
    FUNCTION generateMatrix(n)  
        SET matrix TO new two-dimensional list of size n by n filled with zeros  
        SET top TO 0  
        SET bottom TO n MINUS 1  
        SET left TO 0  
        SET right TO n MINUS 1  
        SET num TO 1  
        
        WHILE top LESS THAN OR EQUAL TO bottom AND left LESS THAN OR EQUAL TO right  
            FOR i FROM left TO right  
                SET element at matrix position top and i TO num  
                INCREMENT num BY 1  
            INCREMENT top BY 1  
            
            FOR i FROM top TO bottom  
                SET element at matrix position i and right TO num  
                INCREMENT num BY 1  
            DECREMENT right BY 1  
            
            IF top LESS THAN OR EQUAL TO bottom  
                FOR i FROM right DOWN TO left  
                    SET element at matrix position bottom and i TO num  
                    INCREMENT num BY 1  
                DECREMENT bottom BY 1  
            
            IF left LESS THAN OR EQUAL TO right  
                FOR i FROM bottom DOWN TO top  
                    SET element at matrix position i and left TO num  
                    INCREMENT num BY 1  
                INCREMENT left BY 1  
        
        RETURN matrix  
    END FUNCTION  
END CLASS
