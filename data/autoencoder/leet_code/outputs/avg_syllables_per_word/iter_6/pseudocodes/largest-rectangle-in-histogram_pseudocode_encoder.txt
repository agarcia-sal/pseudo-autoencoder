CLASS Solution  
    FUNCTION largestRectangleArea(heights)  
        SET stack TO list containing single element negative one  
        SET max_area TO zero  
        APPEND zero TO heights  
        
        FOR each index i AND element h IN heights  
            WHILE last element of stack NOT EQUALS negative one AND h LESS THAN element of heights AT last element of stack  
                SET height TO element of heights AT result of removing last element from stack  
                SET width TO i MINUS last element of stack MINUS one  
                SET max_area TO greater value BETWEEN max_area AND height MULTIPLIED BY width  
            APPEND i TO stack  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
