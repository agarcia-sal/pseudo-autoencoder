CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF FIRST ELEMENT IN grid  
        
        IF (m PLUS n MINUS 1) MODULO 2 NOT EQUALS 0 THEN  
            RETURN False  
        
        INITIALIZE dp AS three-dimensional list of dimensions (m PLUS 1) BY (n PLUS 1) BY (m PLUS n PLUS 1) FILLED WITH False  
        
        SET dp[0][1][0] TO True  
        
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                FOR k FROM 0 TO m PLUS n DO  
                    IF dp[i][j PLUS 1][k] OR dp[i PLUS 1][j][k] IS True THEN  
                        IF grid[i][j] EQUALS '(' AND k PLUS 1 LESS THAN m PLUS n PLUS 1 THEN  
                            SET dp[i PLUS 1][j PLUS 1][k PLUS 1] TO True  
                        ELSE IF grid[i][j] EQUALS ')' AND k MINUS 1 GREATER THAN OR EQUAL TO 0 THEN  
                            SET dp[i PLUS 1][j PLUS 1][k MINUS 1] TO True  
        
        RETURN dp[m][n][0]  
    END FUNCTION  
END CLASS
