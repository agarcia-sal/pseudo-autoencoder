CLASS Solution  
    FUNCTION firstMissingPositive(nums)  
        SET n TO LENGTH OF nums  
        
        FOR i FROM 0 TO n MINUS 1  
            WHILE nums[i] GREATER THAN OR EQUAL TO 1 AND nums[i] LESS THAN OR EQUAL TO n AND nums[nums[i] MINUS 1] NOT EQUALS nums[i]  
                SET correct_index TO nums[i] MINUS 1  
                SWAP nums[i] WITH nums[correct_index]  
        
        FOR i FROM 0 TO n MINUS 1  
            IF nums[i] NOT EQUALS i PLUS 1  
                RETURN i PLUS 1  
        
        RETURN n PLUS 1  
    END FUNCTION  
END CLASS
