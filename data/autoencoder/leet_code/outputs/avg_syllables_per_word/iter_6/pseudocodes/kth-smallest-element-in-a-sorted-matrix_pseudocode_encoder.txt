CLASS Solution  
    FUNCTION kthSmallest(matrix, k)  
        FUNCTION countLessEqual(mid)  
            SET count TO 0  
            SET row TO n MINUS 1  
            SET col TO 0  
            WHILE row GREATER THAN OR EQUAL TO 0 AND col LESS THAN n  
                IF element at matrix[row][col] LESS THAN OR EQUAL TO mid  
                    INCREMENT count BY row PLUS 1  
                    INCREMENT col BY 1  
                ELSE  
                    DECREMENT row BY 1  
            RETURN count  
        END FUNCTION  
          
        SET n TO LENGTH OF matrix  
        SET low TO element at matrix[0][0]  
        SET high TO element at matrix[n MINUS 1][n MINUS 1]  
          
        WHILE low LESS THAN high  
            SET mid TO low PLUS high DIVIDED BY 2 USING INTEGER DIVISION  
            IF CALL countLessEqual WITH mid LESS THAN k  
                SET low TO mid PLUS 1  
            ELSE  
                SET high TO mid  
        RETURN low  
    END FUNCTION  
END CLASS
