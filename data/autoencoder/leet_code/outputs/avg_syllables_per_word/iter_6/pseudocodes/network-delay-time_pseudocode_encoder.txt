CLASS Solution  
    FUNCTION networkDelayTime(times, n, k)  
        SET graph TO an empty dictionary mapping each integer from 1 TO n TO an empty list  
        FOR each element u, v, w IN times  
            APPEND pair (v, w) TO graph[u]  
        
        SET dist TO a dictionary mapping each integer from 1 TO n TO infinity  
        SET dist[k] TO 0  
        
        SET min_heap TO a list containing pair (0, k)  
        
        WHILE min_heap IS NOT empty  
            CALL extracting the pair with smallest first element FROM min_heap AS current_dist, node  
            
            IF current_dist GREATER THAN dist[node]  
                CONTINUE  
            
            FOR each neighbor, weight IN graph[node]  
                SET distance TO current_dist PLUS weight  
                
                IF distance LESS THAN dist[neighbor]  
                    SET dist[neighbor] TO distance  
                    CALL inserting pair (distance, neighbor) INTO min_heap  
        
        SET max_time TO the maximum value among all values in dist  
        
        IF max_time EQUALS infinity  
            RETURN -1  
        ELSE  
            RETURN max_time  
    END FUNCTION  
END CLASS
