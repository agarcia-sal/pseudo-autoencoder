CLASS Solution  
    FUNCTION largestPalindrome(n)  
        IF n EQUALS 1  
            RETURN 9  
        END IF  

        SET upper_limit TO 10 RAISED TO POWER n MINUS 1  
        SET lower_limit TO 10 RAISED TO POWER (n MINUS 1)  

        SET max_palindrome TO 0  

        FOR i FROM upper_limit DOWN TO lower_limit  
            FOR j FROM i DOWN TO lower_limit  
                SET product TO i MULTIPLIED BY j  
                IF product LESS THAN OR EQUAL TO max_palindrome  
                    BREAK  
                END IF  
                SET str_product TO CONVERT product TO STRING  
                IF str_product EQUALS REVERSE OF str_product  
                    SET max_palindrome TO product  
                END IF  
            END FOR  
        END FOR  

        RETURN max_palindrome MODULO 1337  
    END FUNCTION  
END CLASS
