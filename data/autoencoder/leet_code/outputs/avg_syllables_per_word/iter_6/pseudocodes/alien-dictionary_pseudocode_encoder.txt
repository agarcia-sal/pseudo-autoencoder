CLASS Solution  
    FUNCTION alienOrder(words)  
        SET graph TO a mapping from character to set of characters  
        SET indegree TO a mapping from character to integer  
  
        SET all_chars TO the set of all characters found in all words  
  
        FOR index FROM 0 TO LENGTH OF words MINUS 2  
            SET word1 TO words at index  
            SET word2 TO words at index PLUS 1  
            SET min_length TO the smaller length between word1 and word2  
            SET found TO FALSE  
            FOR position FROM 0 TO min_length MINUS 1  
                IF character at position in word1 NOT EQUALS character at position in word2 THEN  
                    IF character at position in word2 NOT IN graph at character at position in word1 THEN  
                        ADD character at position in word2 TO graph at character at position in word1  
                        INCREMENT indegree at character at position in word2 BY 1  
                    SET found TO TRUE  
                    BREAK the loop  
            IF found IS FALSE AND length of word1 GREATER THAN length of word2 THEN  
                RETURN empty string  
  
        SET queue TO a double-ended queue containing all characters in all_chars WHERE indegree is zero or not set  
  
        SET result TO empty list  
  
        WHILE queue IS NOT empty  
            SET char TO REMOVE from front of queue  
            APPEND char TO result  
            FOR each neighbor IN graph at char  
                DECREMENT indegree at neighbor BY 1  
                IF indegree at neighbor EQUALS zero THEN  
                    APPEND neighbor TO queue  
  
        IF LENGTH OF result NOT EQUALS LENGTH OF all_chars THEN  
            RETURN empty string  
  
        RETURN concatenation of all characters in result  
    END FUNCTION  
END CLASS
