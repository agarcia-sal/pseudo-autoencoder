CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF LENGTH OF t EQUALS LENGTH OF s THEN  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND count v IN cnt  
                IF v GREATER THAN 1 THEN  
                    DECREMENT cnt[c] BY 2  
                    CALL dfs WITH argument c PLUS t PLUS c  
                    INCREMENT cnt[c] BY 2  
                END IF  
            END FOR  
        END FUNCTION  

        SET cnt TO mapping of character counts in s  
        SET mid TO empty string  
        FOR each character c AND count v IN cnt  
            IF v MODULO 2 NOT EQUALS 0 THEN  
                IF mid NOT EQUALS empty string THEN  
                    RETURN empty list  
                END IF  
                SET mid TO c  
                DECREMENT cnt[c] BY 1  
            END IF  
        END FOR  
        SET ans TO empty list  
        CALL dfs WITH argument mid  
        RETURN ans  
    END FUNCTION  
END CLASS
