CLASS Solution
    FUNCTION getSum(a, b)
        SET MAX TO the maximum 32-bit integer value
        SET mask TO the value that selects 32 bits

        WHILE b NOT EQUALS 0
            SET carry TO the bitwise AND of a and b shifted left by one bit
            SET a TO the bitwise XOR of a and b AND mask
            SET b TO carry AND mask
        END WHILE

        IF a LESS THAN OR EQUAL TO MAX
            RETURN a
        ELSE
            RETURN the bitwise NOT of the bitwise XOR of a and mask
        END IF
    END FUNCTION
END CLASS
