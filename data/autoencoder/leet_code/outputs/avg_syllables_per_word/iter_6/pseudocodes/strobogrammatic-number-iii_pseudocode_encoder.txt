CLASS Solution  
    FUNCTION strobogrammaticInRange(low, high)  
        FUNCTION generate_strobogrammatic(n, final_length)  
            IF n EQUALS 0 THEN  
                RETURN list containing empty string  
            ELSE IF n EQUALS 1 THEN  
                RETURN list containing "0", "1", and "8"  
            END IF  
            
            SET middles TO CALL generate_strobogrammatic WITH n MINUS 2 AND final_length  
            SET result TO empty list  
            FOR each middle IN middles  
                IF n NOT EQUALS final_length THEN  
                    APPEND string "0" PLUS middle PLUS string "0" TO result  
                END IF  
                APPEND string "1" PLUS middle PLUS string "1" TO result  
                APPEND string "6" PLUS middle PLUS string "9" TO result  
                APPEND string "8" PLUS middle PLUS string "8" TO result  
                APPEND string "9" PLUS middle PLUS string "6" TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        FUNCTION count_strobogrammatic(low, high)  
            SET count TO 0  
            FOR length FROM LENGTH OF low TO LENGTH OF high  
                FOR each num IN CALL generate_strobogrammatic WITH length AND length  
                    IF (length EQUALS LENGTH OF low AND num LESS THAN low) OR (length EQUALS LENGTH OF high AND num GREATER THAN high) THEN  
                        CONTINUE  
                    END IF  
                    INCREMENT count BY 1  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN CALL count_strobogrammatic WITH low AND high  
    END FUNCTION  
END CLASS
