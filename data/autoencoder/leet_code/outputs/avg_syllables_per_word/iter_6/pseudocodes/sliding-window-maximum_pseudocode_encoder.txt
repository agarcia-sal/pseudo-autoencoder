CLASS Solution  
    FUNCTION maxSlidingWindow(nums, k)  
        IF nums IS empty OR k EQUALS 0 THEN  
            RETURN empty list  
        
        SET deq TO empty double-ended queue  
        SET result TO empty list  
        
        FOR i FROM 0 TO length of nums MINUS 1  
            IF deq IS NOT empty AND front element of deq LESS THAN i MINUS k PLUS 1 THEN  
                REMOVE front element FROM deq  
            
            WHILE deq IS NOT empty AND element at index at back of deq IN nums LESS THAN element at index i IN nums DO  
                REMOVE back element FROM deq  
            
            APPEND i TO back of deq  
            
            IF i GREATER THAN OR EQUAL TO k MINUS 1 THEN  
                APPEND element at index front element of deq IN nums TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
