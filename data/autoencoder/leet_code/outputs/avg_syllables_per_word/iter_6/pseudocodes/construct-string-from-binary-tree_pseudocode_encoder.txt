CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS 0, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO CALL TreeNode WITH values AT INDEX 0  
    SET i TO 1  
    SET queue TO NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO CALL queue.popleft  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.left TO CALL TreeNode WITH values AT INDEX i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.right TO CALL TreeNode WITH values AT INDEX i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str(self, root)  
        FUNCTION helper(node)  
            IF node IS None  
                RETURN empty string  
            SET result TO string representation of node.val  
            IF node.left IS NOT None OR node.right IS NOT None  
                SET result TO CONCATENATION OF result, OPEN PARENTHESIS, CALL helper WITH node.left, CLOSE PARENTHESIS  
            IF node.right IS NOT None  
                SET result TO CONCATENATION OF result, OPEN PARENTHESIS, CALL helper WITH node.right, CLOSE PARENTHESIS  
            RETURN result  
        END FUNCTION  
        RETURN CALL helper WITH root  
    END FUNCTION  
END CLASS
