CLASS Solution  
    FUNCTION minCostToSupplyWater(n, wells, pipes)  
        FOR index FROM 0 TO n MINUS 1  
            APPEND list containing zero, index PLUS one, and wells AT index TO pipes  
        
        SORT pipes BY third element (cost) IN ascending order  
        
        SET parent TO list of integers from zero TO n  
        
        FUNCTION find(x)  
            IF parent AT x NOT EQUALS x  
                SET parent AT x TO CALL find WITH parent AT x  
            RETURN parent AT x  
        END FUNCTION  
        
        FUNCTION union(x, y)  
            SET rootX TO CALL find WITH x  
            SET rootY TO CALL find WITH y  
            IF rootX NOT EQUALS rootY  
                SET parent AT rootX TO rootY  
                RETURN true  
            RETURN false  
        END FUNCTION  
        
        SET min_cost TO zero  
        FOR each [house1, house2, cost] IN pipes  
            IF CALL union WITH house1, house2 IS true  
                INCREMENT min_cost BY cost  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
