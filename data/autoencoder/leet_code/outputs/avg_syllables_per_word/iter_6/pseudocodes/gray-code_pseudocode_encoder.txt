CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS 0  
            RETURN list containing 0  
        END IF  
        SET prev_gray TO CALL grayCode WITH n MINUS 1  
        SET mask TO 1 MULTIPLIED BY 2 RAISED TO THE POWER OF n MINUS 1  
        SET current_gray TO prev_gray PLUS list constructed by FOR each number IN REVERSED prev_gray APPENDING mask OR number  
        RETURN current_gray  
    END FUNCTION  
END CLASS
