CLASS Solution  
    FUNCTION findMedianSortedArrays(nums1, nums2)  
        IF LENGTH OF nums1 GREATER THAN LENGTH OF nums2 THEN  
            SET temporary TO nums1  
            SET nums1 TO nums2  
            SET nums2 TO temporary  
        SET x TO LENGTH OF nums1  
        SET y TO LENGTH OF nums2  
        SET low TO 0  
        SET high TO x  
  
        WHILE low LESS THAN OR EQUAL TO high  
            SET partitionX TO (low PLUS high) DIVIDED BY 2  
            SET partitionY TO (x PLUS y PLUS 1) DIVIDED BY 2 MINUS partitionX  
  
            IF partitionX EQUALS 0 THEN  
                SET maxLeftX TO negative infinity  
            ELSE  
                SET maxLeftX TO ELEMENT AT (partitionX MINUS 1) IN nums1  
  
            IF partitionX EQUALS x THEN  
                SET minRightX TO positive infinity  
            ELSE  
                SET minRightX TO ELEMENT AT partitionX IN nums1  
  
            IF partitionY EQUALS 0 THEN  
                SET maxLeftY TO negative infinity  
            ELSE  
                SET maxLeftY TO ELEMENT AT (partitionY MINUS 1) IN nums2  
  
            IF partitionY EQUALS y THEN  
                SET minRightY TO positive infinity  
            ELSE  
                SET minRightY TO ELEMENT AT partitionY IN nums2  
  
            IF maxLeftX LESS THAN OR EQUAL TO minRightY AND maxLeftY LESS THAN OR EQUAL TO minRightX THEN  
                IF (x PLUS y) MODULO 2 EQUALS 0 THEN  
                    SET leftMax TO maxLeftX  
                    IF maxLeftY GREATER THAN leftMax THEN  
                        SET leftMax TO maxLeftY  
                    SET rightMin TO minRightX  
                    IF minRightY LESS THAN rightMin THEN  
                        SET rightMin TO minRightY  
                    RETURN (leftMax PLUS rightMin) DIVIDED BY 2  
                ELSE  
                    IF maxLeftX GREATER THAN maxLeftY THEN  
                        RETURN maxLeftX  
                    ELSE  
                        RETURN maxLeftY  
            ELSE IF maxLeftX GREATER THAN minRightY THEN  
                SET high TO partitionX MINUS 1  
            ELSE  
                SET low TO partitionX PLUS 1  
        END WHILE  
    END FUNCTION  
END CLASS
