CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO positive infinity  

        FOR each num IN nums  
            IF num MODULO 2 EQUALS 1  
                SET num TO num MULTIPLIED BY 2  
            END IF  
            IF num LESS THAN min_value  
                SET min_value TO num  
            END IF  
            CALL heap push ON max_heap WITH NEGATIVE num  
        END FOR  

        SET min_deviation TO positive infinity  

        WHILE NEGATIVE max_heap[0] MODULO 2 EQUALS 0  
            SET max_value TO NEGATIVE heap pop FROM max_heap  
            IF max_value MINUS min_value LESS THAN min_deviation  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY 2  
            IF max_value LESS THAN min_value  
                SET min_value TO max_value  
            END IF  
            CALL heap push ON max_heap WITH NEGATIVE max_value  
        END WHILE  

        IF NEGATIVE max_heap[0] MINUS min_value LESS THAN min_deviation  
            SET min_deviation TO NEGATIVE max_heap[0] MINUS min_value  
        END IF  

        RETURN min_deviation  
    END FUNCTION  
END CLASS
