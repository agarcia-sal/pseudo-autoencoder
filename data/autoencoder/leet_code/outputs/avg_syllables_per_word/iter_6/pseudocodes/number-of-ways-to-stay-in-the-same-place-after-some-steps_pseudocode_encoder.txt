CLASS Solution  
    FUNCTION numWays(steps, arrLen)  
        SET MOD TO ten power nine PLUS seven  
        SET max_index TO minimum of steps DIVIDED BY two AND arrLen MINUS one  
        SET dp TO list of zeros with length max_index PLUS one  
        SET dp at position zero TO one  
        
        FOR iteration FROM zero TO steps MINUS one  
            SET prev_dp TO copy of dp  
            FOR i FROM zero TO max_index  
                SET dp at position i TO prev_dp at position i  
                IF i GREATER THAN zero  
                    SET dp at position i TO dp at position i PLUS prev_dp at position i MINUS one  
                    SET dp at position i TO dp at position i MODULO MOD  
                END IF  
                IF i LESS THAN max_index  
                    SET dp at position i TO dp at position i PLUS prev_dp at position i PLUS one  
                    SET dp at position i TO dp at position i MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN dp at position zero  
    END FUNCTION  
END CLASS
