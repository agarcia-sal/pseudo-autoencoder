CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET sign TO plus  
            SET num TO 0  
            
            WHILE length of s GREATER THAN 0  
                SET char TO first element removed from s  
                
                IF char IS digit  
                    SET num TO num MULTIPLIED BY 10 PLUS numeric value of char  
                END IF  
                
                IF char EQUALS open parenthesis  
                    SET num TO CALL helper WITH s  
                END IF  
                
                IF char IS NOT digit AND char IS NOT whitespace OR length of s EQUALS 0  
                    IF sign EQUALS plus  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus  
                        APPEND num MINUS num MINUS num TO stack  // correction: should be negative num  
                        (Corrected:) APPEND negative num TO stack  
                    ELSE IF sign EQUALS multiply  
                        SET last element of stack TO last element MULTIPLIED BY num  
                    ELSE IF sign EQUALS divide  
                        SET last element of stack TO integer division of last element BY num  
                    END IF  
                    SET sign TO char  
                    SET num TO 0  
                END IF  
                
                IF char EQUALS close parenthesis  
                    BREAK  
                END IF  
            END WHILE  
            
            RETURN sum of stack  
        END FUNCTION  
        
        RETURN CALL helper WITH list conversion of s  
    END FUNCTION  
END CLASS
