CLASS Solution  
    FUNCTION rearrangeSticks(n, k)  
        SET MOD TO 10 raised to the power of 9 PLUS 7  
        
        INITIALIZE dp AS a two-dimensional list with n PLUS 1 rows AND k PLUS 1 columns FILLED WITH 0  
        
        SET dp at row 0 column 0 TO 1  
        
        FOR i FROM 1 TO n  
            FOR j FROM 1 TO k  
                SET dp at row i column j TO (dp at row i minus 1 column j minus 1 PLUS (i minus 1) MULTIPLIED BY dp at row i minus 1 column j) MODULO MOD  
        
        RETURN dp at row n column k  
    END FUNCTION  
END CLASS
