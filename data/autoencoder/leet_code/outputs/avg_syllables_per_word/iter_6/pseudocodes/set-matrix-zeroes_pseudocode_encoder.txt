CLASS Solution  
    FUNCTION setZeroes(matrix)  
        IF matrix IS EMPTY OR matrix FIRST ELEMENT IS EMPTY  
            RETURN  
        SET m TO LENGTH OF matrix  
        SET n TO LENGTH OF matrix FIRST ELEMENT  
        SET first_row_has_zero TO FALSE  
        FOR j FROM 0 TO n MINUS 1  
            IF matrix FIRST ROW AT j EQUALS 0  
                SET first_row_has_zero TO TRUE  
                BREAK  
        SET first_col_has_zero TO FALSE  
        FOR i FROM 0 TO m MINUS 1  
            IF matrix AT i COLUMN ZERO EQUALS 0  
                SET first_col_has_zero TO TRUE  
                BREAK  
        FOR i FROM 1 TO m MINUS 1  
            FOR j FROM 1 TO n MINUS 1  
                IF matrix AT i j EQUALS 0  
                    SET matrix AT i COLUMN ZERO TO 0  
                    SET matrix FIRST ROW AT j TO 0  
        FOR i FROM 1 TO m MINUS 1  
            FOR j FROM 1 TO n MINUS 1  
                IF matrix AT i COLUMN ZERO EQUALS 0 OR matrix FIRST ROW AT j EQUALS 0  
                    SET matrix AT i j TO 0  
        IF first_row_has_zero IS TRUE  
            FOR j FROM 0 TO n MINUS 1  
                SET matrix FIRST ROW AT j TO 0  
        IF first_col_has_zero IS TRUE  
            FOR i FROM 0 TO m MINUS 1  
                SET matrix AT i COLUMN ZERO TO 0  
        RETURN  
    END FUNCTION  
END CLASS
