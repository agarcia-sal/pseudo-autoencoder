CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO a dictionary with key zero and value zero  
        
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each key d and value y IN dp  
                SET key_for_taller TO d PLUS rod  
                SET value_for_taller TO maximum between new_dp value at key_for_taller or zero AND y  
                SET new_dp at key_for_taller TO value_for_taller  
                
                SET difference TO absolute value of d MINUS rod  
                SET value_for_shorter TO maximum between new_dp value at difference or zero AND y PLUS minimum of d AND rod  
                SET new_dp at difference TO value_for_shorter  
            SET dp TO new_dp  
        
        RETURN dp value at zero  
    END FUNCTION  
END CLASS
