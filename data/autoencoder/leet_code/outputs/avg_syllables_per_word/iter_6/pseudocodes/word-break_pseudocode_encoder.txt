CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO a set containing all elements of wordDict  
        SET dp TO a list of length length of s PLUS one, filled with False values  
        SET dp at position zero TO True  
        
        FOR i FROM 1 TO length of s  
            FOR j FROM 0 TO i MINUS one  
                IF dp at position j IS True AND the substring of s from position j TO i IS IN wordSet  
                    SET dp at position i TO True  
                    BREAK the inner loop  
        RETURN dp at position length of s  
    END FUNCTION  
END CLASS
