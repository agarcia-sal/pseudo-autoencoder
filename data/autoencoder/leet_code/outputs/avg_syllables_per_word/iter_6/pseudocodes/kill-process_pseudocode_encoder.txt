CLASS Solution  
    FUNCTION killProcess(pid, ppid, kill)  
        SET tree TO a default dictionary mapping each key to an empty list  
        FOR each child AND parent IN pid AND ppid respectively  
            APPEND child TO tree[parent]  
        
        SET queue TO a list containing the element kill  
        SET killed_processes TO an empty list  
        
        WHILE queue IS NOT empty  
            REMOVE the first element FROM queue AND SET TO current  
            APPEND current TO killed_processes  
            FOR each child IN tree[current]  
                APPEND child TO queue  
        
        RETURN killed_processes  
    END FUNCTION  
END CLASS
