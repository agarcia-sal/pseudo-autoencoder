CLASS Solution  
    FUNCTION maxKilledEnemies(grid)  
        IF grid IS EMPTY OR grid FIRST ELEMENT IS EMPTY  
            RETURN 0  
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF grid FIRST ELEMENT  
        SET max_kills TO 0  
        SET row_kills TO 0  
        SET col_kills TO LIST OF LENGTH n WITH ALL VALUES 0  
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF j EQUALS 0 OR grid ELEMENT AT i AND j MINUS 1 EQUALS 'W'  
                    SET row_kills TO 0  
                    FOR k FROM j TO n MINUS 1  
                        IF grid ELEMENT AT i AND k EQUALS 'E'  
                            INCREMENT row_kills BY 1  
                        ELSE IF grid ELEMENT AT i AND k EQUALS 'W'  
                            BREAK  
                IF i EQUALS 0 OR grid ELEMENT AT i MINUS 1 AND j EQUALS 'W'  
                    SET col_kills AT j TO 0  
                    FOR k FROM i TO m MINUS 1  
                        IF grid ELEMENT AT k AND j EQUALS 'E'  
                            INCREMENT col_kills AT j BY 1  
                        ELSE IF grid ELEMENT AT k AND j EQUALS 'W'  
                            BREAK  
                IF grid ELEMENT AT i AND j EQUALS '0'  
                    SET max_kills TO MAXIMUM OF max_kills AND row_kills PLUS col_kills AT j  
        RETURN max_kills  
    END FUNCTION  
END CLASS
