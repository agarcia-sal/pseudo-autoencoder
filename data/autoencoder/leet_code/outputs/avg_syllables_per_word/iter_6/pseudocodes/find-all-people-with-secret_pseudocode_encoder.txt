CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO set containing 0 and firstPerson  
        CALL sort meetings BY time element in each meeting  
        
        SET i TO 0  
        WHILE i LESS THAN LENGTH OF meetings  
            SET current_time TO time element of meetings at index i  
            SET graph TO empty mapping of list  
            
            WHILE i LESS THAN LENGTH OF meetings AND time element of meetings at index i EQUALS current_time  
                SET x TO first person in meetings at index i  
                SET y TO second person in meetings at index i  
                APPEND y TO graph at key x  
                APPEND x TO graph at key y  
                INCREMENT i BY 1  
            
            SET queue TO collection of persons IN know_secret THAT ARE KEYS IN graph  
            SET visited TO empty set  
            
            WHILE queue IS NOT empty  
                REMOVE first person FROM queue AND SET TO person  
                IF person IS IN visited  
                    CONTINUE  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN graph at key person  
                    IF neighbor NOT IN visited  
                        APPEND neighbor TO queue  
        
        RETURN list converted from know_secret  
    END FUNCTION  
END CLASS
