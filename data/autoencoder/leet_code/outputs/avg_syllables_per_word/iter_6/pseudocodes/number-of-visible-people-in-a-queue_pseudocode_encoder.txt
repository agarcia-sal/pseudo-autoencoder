CLASS Solution  
    FUNCTION canSeePersonsCount(heights)  
        SET n TO LENGTH OF heights  
        SET answer TO list of zeros of length n  
        SET stack TO empty list  
        
        FOR i FROM 0 TO n MINUS 1  
            WHILE stack IS NOT EMPTY AND height AT last element of stack IS LESS THAN height AT i  
                SET index TO REMOVE last element FROM stack  
                INCREMENT answer AT index BY 1  
            IF stack IS NOT EMPTY  
                INCREMENT answer AT last element of stack BY 1  
            APPEND i TO stack  
        
        RETURN answer  
    END FUNCTION  
END CLASS
