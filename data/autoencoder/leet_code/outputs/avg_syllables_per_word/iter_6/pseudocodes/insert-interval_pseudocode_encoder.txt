CLASS Solution  
    FUNCTION insert(intervals, newInterval)  
        IF intervals IS EMPTY  
            RETURN list containing newInterval  
        
        SET merged TO empty list  
        SET i TO 0  
        SET n TO length of intervals  
        
        WHILE i LESS THAN n AND end of intervals at index i LESS THAN start of newInterval  
            APPEND intervals at index i TO merged  
            INCREMENT i BY 1  
        
        WHILE i LESS THAN n AND start of intervals at index i LESS THAN OR EQUAL TO end of newInterval  
            SET start of newInterval TO minimum of start of newInterval AND start of intervals at index i  
            SET end of newInterval TO maximum of end of newInterval AND end of intervals at index i  
            INCREMENT i BY 1  
        
        APPEND newInterval TO merged  
        
        WHILE i LESS THAN n  
            APPEND intervals at index i TO merged  
            INCREMENT i BY 1  
        
        RETURN merged  
    END FUNCTION  
END CLASS
