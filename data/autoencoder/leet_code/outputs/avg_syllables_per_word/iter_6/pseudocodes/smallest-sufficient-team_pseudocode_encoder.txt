CLASS Solution  
    FUNCTION smallestSufficientTeam(req_skills, people)  
        SET d TO an empty mapping  
        FOR index FROM 0 TO LENGTH OF req_skills MINUS 1  
            SET d AT req_skills AT index TO index  
        SET m TO LENGTH OF req_skills  
        SET n TO LENGTH OF people  
        SET p TO a list of zeros with length n  
        FOR i FROM 0 TO n MINUS 1  
            FOR each skill IN people AT i  
                SET p AT i TO p AT i OR SHIFT 1 LEFT BY d AT skill  
        SET f TO a list filled with infinity of length SHIFT 1 LEFT BY m  
        SET g TO a list of zeros with length SHIFT 1 LEFT BY m  
        SET h TO a list of zeros with length SHIFT 1 LEFT BY m  
        SET f AT 0 TO 0  
        FOR i FROM 0 TO (SHIFT 1 LEFT BY m) MINUS 1  
            IF f AT i EQUALS infinity  
                CONTINUE  
            FOR j FROM 0 TO n MINUS 1  
                IF f AT i PLUS 1 LESS THAN f AT (i OR p AT j)  
                    SET f AT (i OR p AT j) TO f AT i PLUS 1  
                    SET g AT (i OR p AT j) TO j  
                    SET h AT (i OR p AT j) TO i  
        SET i TO (SHIFT 1 LEFT BY m) MINUS 1  
        SET ans TO an empty list  
        WHILE i NOT EQUALS 0  
            APPEND g AT i TO ans  
            SET i TO h AT i  
        RETURN ans  
    END FUNCTION  
END CLASS
