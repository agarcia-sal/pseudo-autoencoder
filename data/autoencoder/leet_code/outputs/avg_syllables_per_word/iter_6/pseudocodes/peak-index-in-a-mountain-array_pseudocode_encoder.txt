CLASS Solution  
    FUNCTION peakIndexInMountainArray(arr)  
        SET left TO 0  
        SET right TO LENGTH OF arr MINUS 1  
        WHILE left IS LESS THAN right  
            SET mid TO INTEGER DIVISION OF (left PLUS right) BY 2  
            IF element AT mid IN arr IS LESS THAN element AT mid PLUS 1 IN arr  
                SET left TO mid PLUS 1  
            ELSE  
                SET right TO mid  
        RETURN left  
    END FUNCTION  
END CLASS
