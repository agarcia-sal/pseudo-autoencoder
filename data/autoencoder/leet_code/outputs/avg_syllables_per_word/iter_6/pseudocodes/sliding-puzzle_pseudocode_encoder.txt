CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuples constructed from board rows  
        SET target TO tuple containing (1, 2, 3) and (4, 5, 0)  
        
        SET neighbors TO map of positions:
            0 maps to set containing 1 and 3
            1 maps to set containing 0, 2, and 4
            2 maps to set containing 1 and 5
            3 maps to set containing 0 and 4
            4 maps to set containing 1, 3, and 5
            5 maps to set containing 2 and 4  
        
        FOR i FROM 0 TO 1  
            FOR j FROM 0 TO 2  
                IF element at start position i, j EQUALS 0  
                    SET zero_pos TO i MULTIPLIED BY 3 PLUS j  
        
        INITIALIZE queue TO deque containing tuple of start, zero_pos, and 0 moves  
        INITIALIZE visited TO set containing start  
        
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND ASSIGN TO state, zero_pos, moves  
            
            IF state EQUALS target  
                RETURN moves  
            
            FOR each neighbor IN neighbors at zero_pos  
                CONVERT state TO list of lists AND ASSIGN TO new_state  
                SET i TO zero_pos DIVIDED BY 3  
                SET j TO zero_pos MODULO 3  
                SET ni TO neighbor DIVIDED BY 3  
                SET nj TO neighbor MODULO 3  
                
                SWAP new_state at i, j WITH new_state at ni, nj  
                
                CONVERT new_state TO tuple of tuples AND ASSIGN TO new_tuple_state  
                
                IF new_tuple_state NOT IN visited  
                    ADD new_tuple_state TO visited  
                    APPEND tuple of new_tuple_state, neighbor, moves PLUS 1 TO queue  
        
        RETURN negative 1  
    END FUNCTION  
END CLASS
