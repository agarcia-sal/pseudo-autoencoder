CLASS Solution  
    FUNCTION minAbsDifference(nums, goal)  
        SET n TO LENGTH OF nums  
        SET mid TO n DIVIDED BY 2  
        
        SET left_sums TO LIST CONTAINING 0  
        FOR each num IN SUBLIST OF nums FROM START TO mid EXCLUSIVE  
            SET new_sums TO EMPTY LIST  
            FOR each s IN left_sums  
                APPEND s PLUS num TO new_sums  
            EXTEND left_sums BY new_sums  
        
        SET right_sums TO LIST CONTAINING 0  
        FOR each num IN SUBLIST OF nums FROM mid TO END  
            SET new_sums TO EMPTY LIST  
            FOR each s IN right_sums  
                APPEND s PLUS num TO new_sums  
            EXTEND right_sums BY new_sums  
        
        SORT right_sums IN ASCENDING ORDER  
        
        SET min_diff TO positive infinity  
        
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO CALL bisect_left WITH right_sums AND target  
            
            IF pos LESS THAN LENGTH OF right_sums  
                SET candidate_diff TO ABSOLUTE VALUE OF left_sum PLUS right_sums AT pos MINUS goal  
                IF candidate_diff LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
            IF pos GREATER THAN 0  
                SET candidate_diff TO ABSOLUTE VALUE OF left_sum PLUS right_sums AT pos MINUS 1 MINUS goal  
                IF candidate_diff LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
