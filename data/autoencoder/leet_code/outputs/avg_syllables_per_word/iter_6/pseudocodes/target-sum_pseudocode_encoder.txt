CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS LENGTH OF nums THEN  
                IF current_sum EQUALS target THEN  
                    RETURN 1  
                ELSE  
                    RETURN 0  
            IF memo CONTAINS KEY (index, current_sum) THEN  
                RETURN VALUE FROM memo FOR KEY (index, current_sum)  
            SET positive TO CALL dfs WITH (index PLUS 1, current_sum PLUS nums AT index, memo)  
            SET negative TO CALL dfs WITH (index PLUS 1, current_sum MINUS nums AT index, memo)  
            SET memo AT KEY (index, current_sum) TO positive PLUS negative  
            RETURN VALUE FROM memo FOR KEY (index, current_sum)  
        END FUNCTION  
        RETURN CALL dfs WITH (0, 0, EMPTY COLLECTION)  
    END FUNCTION  
END CLASS
