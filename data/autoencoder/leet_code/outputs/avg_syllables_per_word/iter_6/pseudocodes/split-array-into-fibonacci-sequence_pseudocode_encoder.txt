CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS LENGTH OF num AND LENGTH OF path GREATER THAN OR EQUAL TO 3  
                RETURN path  
            FOR end FROM start PLUS 1 TO LENGTH OF num PLUS 1  
                IF character at position start in num EQUALS '0' AND end GREATER THAN start PLUS 1  
                    CONTINUE to next iteration  
                SET next_num TO integer value of substring of num FROM start TO end  
                IF next_num GREATER THAN OR EQUAL TO 2 TO THE POWER OF 31  
                    BREAK loop  
                IF LENGTH OF path LESS THAN 2 OR next_num EQUALS sum of last two elements in path  
                    SET result TO CALL backtrack WITH end AND path APPENDED WITH next_num  
                    IF result IS NOT EMPTY  
                        RETURN result  
            RETURN empty list  
        END FUNCTION  
        RETURN CALL backtrack WITH 0 AND empty list  
    END FUNCTION  
END CLASS
