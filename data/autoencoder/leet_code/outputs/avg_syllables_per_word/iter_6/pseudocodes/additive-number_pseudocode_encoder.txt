CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS LENGTH OF num THEN  
                RETURN True  
            END IF  
            
            SET expected TO STRING REPRESENTATION OF (first PLUS second)  
            SET length TO LENGTH OF expected  
            
            IF start PLUS length GREATER THAN LENGTH OF num OR SUBSTRING OF num FROM start TO start PLUS length NOT EQUALS expected THEN  
                RETURN False  
            END IF  
            
            RETURN CALL is_valid_sequence WITH start PLUS length, second, INTEGER VALUE OF expected  
        END FUNCTION  
        
        SET n TO LENGTH OF num  
        FOR i FROM 1 TO (n DIVIDED BY 2) PLUS 1  
            FOR j FROM i PLUS 1 TO n  
                SET first TO SUBSTRING OF num FROM START TO i  
                SET second TO SUBSTRING OF num FROM i TO j  
                
                IF (LENGTH OF first GREATER THAN 1 AND FIRST CHARACTER OF first EQUALS CHARACTER ZERO) OR (LENGTH OF second GREATER THAN 1 AND FIRST CHARACTER OF second EQUALS CHARACTER ZERO) THEN  
                    CONTINUE  
                END IF  
                
                IF CALL is_valid_sequence WITH j, INTEGER VALUE OF first, INTEGER VALUE OF second THEN  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
