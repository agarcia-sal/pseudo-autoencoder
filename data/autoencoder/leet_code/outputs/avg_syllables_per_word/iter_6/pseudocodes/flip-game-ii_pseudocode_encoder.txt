CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF state IS IN memo THEN  
                RETURN memo[state]  
            
            FOR index FROM 0 TO LENGTH OF state MINUS 2  
                IF SUBSTRING of state FROM index TO index PLUS 2 EQUALS "++" THEN  
                    SET new_state TO CONCATENATION of SUBSTRING of state UP TO index, "--", and SUBSTRING of state FROM index PLUS 2  
                    IF NOT CALL can_win_helper WITH new_state THEN  
                        SET memo[state] TO True  
                        RETURN True  
            
            SET memo[state] TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN CALL can_win_helper WITH currentState  
    END FUNCTION  
END CLASS
