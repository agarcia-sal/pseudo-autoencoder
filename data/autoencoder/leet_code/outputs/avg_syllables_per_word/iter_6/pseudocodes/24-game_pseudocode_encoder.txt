CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF LENGTH OF nums EQUALS 1  
                RETURN absolute value of (first element of nums MINUS 24) LESS THAN 0.000001  
            END IF  
            
            FOR i FROM 0 TO LENGTH OF nums MINUS 1  
                FOR j FROM 0 TO LENGTH OF nums MINUS 1  
                    IF i NOT EQUALS j  
                        SET new_nums TO empty list  
                        FOR k FROM 0 TO LENGTH OF nums MINUS 1  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND element at index k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR each operation IN (division as true division, multiplication, addition, subtraction)  
                            IF (operation IS addition OR operation IS multiplication) AND j GREATER THAN i  
                                CONTINUE to next operation  
                            END IF  
                            
                            IF operation IS NOT division OR element at index j of nums NOT EQUALS 0  
                                APPEND result of applying operation to elements at index i and j of nums TO new_nums  
                                IF CALL solve WITH new_nums RETURNS true  
                                    RETURN true  
                                END IF  
                                REMOVE last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN false  
        END FUNCTION  
        
        RETURN CALL solve WITH cards  
    END FUNCTION  
END CLASS
