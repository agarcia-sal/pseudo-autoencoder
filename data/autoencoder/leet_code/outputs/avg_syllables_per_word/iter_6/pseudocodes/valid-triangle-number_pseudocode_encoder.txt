CLASS Solution  
    FUNCTION triangleNumber(nums)  
        CALL nums SORT  
        SET count TO 0  
        SET n TO LENGTH OF nums  
  
        FOR k FROM n MINUS 1 TO 2 BY MINUS 1  
            SET i TO 0  
            SET j TO k MINUS 1  
  
            WHILE i LESS THAN j  
                IF nums AT i PLUS nums AT j GREATER THAN nums AT k  
                    INCREMENT count BY j MINUS i  
                    DECREMENT j BY 1  
                ELSE  
                    INCREMENT i BY 1  
            END WHILE  
        END FOR  
  
        RETURN count  
    END FUNCTION  
END CLASS
