CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS 0, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH values AT INDEX 0  
    SET i TO 1  
    INITIALIZE queue AS EMPTY QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.left TO NEW TreeNode WITH values AT INDEX i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.right TO NEW TreeNode WITH values AT INDEX i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive(self, root)  
        FUNCTION dfs(node)  
            IF node IS None  
                RETURN 0, 0  
            SET inc_len TO 1  
            SET dec_len TO 1  
            SET left_inc, left_dec TO CALL dfs WITH node.left  
            SET right_inc, right_dec TO CALL dfs WITH node.right  
            IF node.left IS NOT None  
                IF node.val EQUALS node.left.val PLUS 1  
                    SET dec_len TO left_dec PLUS 1  
                ELSE IF node.val EQUALS node.left.val MINUS 1  
                    SET inc_len TO left_inc PLUS 1  
            IF node.right IS NOT None  
                IF node.val EQUALS node.right.val PLUS 1  
                    SET dec_len TO MAXIMUM OF dec_len AND right_dec PLUS 1  
                ELSE IF node.val EQUALS node.right.val MINUS 1  
                    SET inc_len TO MAXIMUM OF inc_len AND right_inc PLUS 1  
            SET max_len TO MAXIMUM OF max_len AND inc_len PLUS dec_len MINUS 1  
            RETURN inc_len, dec_len  
        END FUNCTION  
        SET max_len TO 0  
        CALL dfs WITH root  
        RETURN max_len  
    END FUNCTION  
END CLASS
