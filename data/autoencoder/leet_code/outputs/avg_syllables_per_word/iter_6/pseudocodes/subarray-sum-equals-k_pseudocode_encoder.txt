CLASS Solution  
    FUNCTION subarraySum(nums, k)  
        SET cumulative_sum_freq TO dictionary with key zero mapped to one  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF current_sum MINUS k EXISTS IN cumulative_sum_freq  
                INCREMENT count BY value mapped to current_sum MINUS k IN cumulative_sum_freq  
            IF current_sum EXISTS IN cumulative_sum_freq  
                INCREMENT value mapped to current_sum IN cumulative_sum_freq BY one  
            ELSE  
                SET value mapped to current_sum IN cumulative_sum_freq TO one  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
