CLASS Solution  
    FUNCTION findIntegers(n)  
        SET bin_rep TO binary representation of n starting from second character  
        SET length TO length of bin_rep  
        
        INITIALIZE dp AS list of zeros of size length PLUS 1  
        SET dp at index 0 TO 1  
        SET dp at index 1 TO 2  
        
        FOR i FROM 2 TO length  
            SET dp at index i TO dp at index (i MINUS 1) PLUS dp at index (i MINUS 2)  
        
        SET result TO 0  
        SET prev_bit TO 0  
        
        FOR i FROM 0 TO length MINUS 1  
            IF character at index i of bin_rep EQUALS '1' THEN  
                INCREMENT result BY dp at index (length MINUS i MINUS 1)  
                IF prev_bit EQUALS 1 THEN  
                    DECREMENT result BY 1  
                    BREAK  
            END IF  
            SET prev_bit TO 1 IF character at index i of bin_rep EQUALS '1' ELSE 0  
        END FOR  
        
        RETURN result PLUS 1  
    END FUNCTION  
END CLASS
