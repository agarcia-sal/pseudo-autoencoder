CLASS Solution  
    FUNCTION findKthLargest(nums, k)  
        SET min_heap TO empty list  
        
        FOR each num IN nums  
            CALL add num TO min_heap maintaining heap property  
            IF length OF min_heap GREATER THAN k  
                CALL remove smallest element FROM min_heap maintaining heap property  
        RETURN first element OF min_heap  
    END FUNCTION  
END CLASS
