CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF nums1 IS EMPTY OR nums2 IS EMPTY OR k LESS THAN OR EQUAL TO 0  
            RETURN empty list  
        
        SET min_heap TO empty list  
        
        FOR index FROM 0 TO MINIMUM OF k AND LENGTH OF nums1 MINUS 1  
            CALL operation TO ADD TO min_heap THE TRIPLE OF THE SUM OF nums1 AT index PLUS nums2 AT 0, index, AND 0  
        
        SET result TO empty list  
        
        WHILE min_heap IS NOT EMPTY AND LENGTH OF result LESS THAN k  
            CALL operation TO REMOVE SMALLEST ELEMENT FROM min_heap AND ASSIGN TO temporary variables sum_value, i, j  
            APPEND list CONSISTING OF nums1 AT i AND nums2 AT j TO result  
            IF j PLUS 1 LESS THAN LENGTH OF nums2  
                CALL operation TO ADD TO min_heap THE TRIPLE OF THE SUM OF nums1 AT i PLUS nums2 AT j PLUS 1, i, AND j PLUS 1  
        
        RETURN result  
    END FUNCTION  
END CLASS
