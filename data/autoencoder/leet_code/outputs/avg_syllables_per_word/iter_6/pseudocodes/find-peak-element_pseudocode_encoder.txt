CLASS Solution  
    FUNCTION findPeakElement(nums)  
        SET left TO 0  
        SET right TO LENGTH OF nums MINUS 1  
        
        WHILE left LESS THAN right  
            SET mid TO (left PLUS right) DIVIDED BY 2  
            IF element at index mid in nums GREATER THAN element at index mid PLUS 1 in nums  
                SET right TO mid  
            ELSE  
                SET left TO mid PLUS 1  
        RETURN left  
    END FUNCTION  
END CLASS
