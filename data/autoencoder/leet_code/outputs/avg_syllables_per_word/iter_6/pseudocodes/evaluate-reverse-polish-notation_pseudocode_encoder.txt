CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUALS PLUS OR token EQUALS MINUS OR token EQUALS MULTIPLY OR token EQUALS DIVIDE  
                SET b TO REMOVE last element FROM stack  
                SET a TO REMOVE last element FROM stack  
                IF token EQUALS PLUS  
                    APPEND sum of a AND b TO stack  
                ELSE IF token EQUALS MINUS  
                    APPEND difference of a AND b TO stack  
                ELSE IF token EQUALS MULTIPLY  
                    APPEND product of a AND b TO stack  
                ELSE IF token EQUALS DIVIDE  
                    APPEND integer division of a BY b truncated towards zero TO stack  
            ELSE  
                APPEND integer value of token TO stack  
        RETURN first element OF stack  
    END FUNCTION  
END CLASS
