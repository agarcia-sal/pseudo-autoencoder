CLASS Solution  
    FUNCTION longestSubstring(s, k)  
        IF LENGTH OF s LESS THAN k THEN  
            RETURN 0  
        END IF  

        SET count TO CALL Counter WITH s  

        FOR each char IN count  
            IF count AT char LESS THAN k THEN  
                RETURN MAXIMUM OF CALL self.longestSubstring WITH substring AND k FOR each substring IN RESULT OF SPLITTING s BY char  
            END IF  
        END FOR  

        RETURN LENGTH OF s  
    END FUNCTION  
END CLASS
