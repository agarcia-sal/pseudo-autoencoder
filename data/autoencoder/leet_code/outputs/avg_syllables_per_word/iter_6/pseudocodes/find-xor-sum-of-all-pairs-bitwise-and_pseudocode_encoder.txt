CLASS Solution  
    FUNCTION getXORSum(arr1, arr2)  
        SET xor_sum_arr1 TO 0  
        SET xor_sum_arr2 TO 0  
        
        FOR each num IN arr1  
            SET xor_sum_arr1 TO xor_sum_arr1 XOR num  
        
        FOR each num IN arr2  
            SET xor_sum_arr2 TO xor_sum_arr2 XOR num  
        
        SET result TO bitwise AND of xor_sum_arr1 AND xor_sum_arr2  
        RETURN result  
    END FUNCTION  
END CLASS
