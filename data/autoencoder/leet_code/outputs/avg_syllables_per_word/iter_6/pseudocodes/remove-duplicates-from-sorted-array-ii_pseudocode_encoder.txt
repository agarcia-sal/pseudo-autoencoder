CLASS Solution  
    FUNCTION removeDuplicates(nums)  
        IF nums IS EMPTY  
            RETURN 0  
        
        SET write_index TO 0  
        
        FOR i FROM 0 TO MINIMUM OF 2 AND LENGTH OF nums  
            SET element AT write_index IN nums TO element AT i IN nums  
            INCREMENT write_index BY 1  
        
        FOR i FROM 2 TO LENGTH OF nums MINUS 1  
            IF element AT i IN nums NOT EQUALS element AT write_index MINUS 2 IN nums  
                SET element AT write_index IN nums TO element AT i IN nums  
                INCREMENT write_index BY 1  
        
        RETURN write_index  
    END FUNCTION  
END CLASS
