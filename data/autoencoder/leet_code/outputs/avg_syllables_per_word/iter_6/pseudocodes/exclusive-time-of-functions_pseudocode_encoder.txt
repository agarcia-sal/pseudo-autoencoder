CLASS Solution  
    FUNCTION exclusiveTime(n, logs)  
        SET exclusive_time TO list of zeroes of length n  
        SET stack TO empty list  
        
        FOR each log IN logs  
            SPLIT log INTO function_id, action, timestamp  
            SET function_id TO integer value of function_id  
            SET timestamp TO integer value of timestamp  
            
            IF action EQUALS "start"  
                IF stack IS NOT empty  
                    SET prev_function_id, prev_timestamp TO last element in stack  
                    INCREMENT exclusive_time at prev_function_id BY timestamp MINUS prev_timestamp  
                APPEND tuple of function_id and timestamp TO stack  
            ELSE  
                SET prev_function_id, prev_timestamp TO REMOVE last element of stack  
                INCREMENT exclusive_time at prev_function_id BY timestamp MINUS prev_timestamp PLUS one  
                IF stack IS NOT empty  
                    SET top_function_id, _ TO last element in stack  
                    REPLACE last element in stack WITH tuple of top_function_id and timestamp PLUS one  
        
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
