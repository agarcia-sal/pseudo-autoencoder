CLASS Solution  
    FUNCTION ladderLength(beginWord, endWord, wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN 0  
        
        SET word_dict TO dictionary with default empty list  
        SET word_length TO LENGTH OF beginWord  
        
        FOR each word IN wordList  
            FOR i FROM 0 TO word_length MINUS 1  
                SET intermediate_key TO CONCATENATION OF substring of word FROM start TO i MINUS 1 PLUS "*" PLUS substring of word FROM i PLUS 1 TO end  
                APPEND word TO word_dict AT intermediate_key  
        
        SET queue TO new queue containing tuple of beginWord and 1  
        SET visited TO set containing beginWord  
        
        WHILE queue IS NOT EMPTY  
            SET current_word, current_length TO REMOVE front element FROM queue  
            
            FOR i FROM 0 TO word_length MINUS 1  
                SET intermediate_word TO CONCATENATION OF substring of current_word FROM start TO i MINUS 1 PLUS "*" PLUS substring of current_word FROM i PLUS 1 TO end  
                
                FOR each word IN word_dict AT intermediate_word  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS 1  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        ADD tuple of word and current_length PLUS 1 TO queue  
        
        RETURN 0  
    END FUNCTION  
END CLASS
