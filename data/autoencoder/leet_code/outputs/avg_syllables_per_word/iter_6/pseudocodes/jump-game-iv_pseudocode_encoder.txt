CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO LENGTH OF arr  
        IF n EQUALS 1  
            RETURN 0  
        END IF  
        
        CREATE value_indices AS mapping FROM values TO list of indices  
        FOR i FROM 0 TO n MINUS 1  
            APPEND i TO value_indices AT key arr AT i  
        END FOR  
        
        CREATE queue AS an empty deque CONTAINING tuple (0, 0) representing current index and steps  
        CREATE visited AS set CONTAINING 0  
        
        WHILE queue IS NOT empty  
            REMOVE and RETURN first element FROM queue INTO current_index, steps  
            
            IF current_index EQUALS n MINUS 1  
                RETURN steps  
            END IF  
            
            SET next_indices TO list CONTAINING current_index PLUS 1, current_index MINUS 1  
            APPEND all elements FROM value_indices AT key arr AT current_index TO next_indices  
            
            FOR each next_index IN next_indices  
                IF next_index GREATER THAN OR EQUAL TO 0 AND next_index LESS THAN n AND next_index NOT IN visited  
                    ADD next_index TO visited  
                    APPEND tuple (next_index, steps PLUS 1) TO queue  
                END IF  
            END FOR  
            
            SET value_indices AT key arr AT current_index TO empty list  
        END WHILE  
        
        RETURN negative 1  
    END FUNCTION  
END CLASS
