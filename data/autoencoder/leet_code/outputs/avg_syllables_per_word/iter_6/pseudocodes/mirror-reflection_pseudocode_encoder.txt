CLASS Solution  
    FUNCTION mirrorReflection(p, q)  
        CALL gcd WITH p AND q AND STORE RESULT IN gcd_value  
        SET lcm TO p MULTIPLIED BY q DIVIDED BY gcd_value  
        
        SET vertical_reflections TO lcm DIVIDED BY p  
        SET horizontal_reflections TO lcm DIVIDED BY q  
        
        IF vertical_reflections MODULO 2 EQUALS 0  
            RETURN 0  
        ELSE IF horizontal_reflections MODULO 2 EQUALS 1  
            RETURN 1  
        ELSE  
            RETURN 2  
    END FUNCTION  
END CLASS
