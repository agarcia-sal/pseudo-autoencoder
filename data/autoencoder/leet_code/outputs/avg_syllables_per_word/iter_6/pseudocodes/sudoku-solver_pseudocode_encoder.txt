CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR i FROM 0 TO 8  
                IF board AT row AND i EQUALS num  
                    RETURN False  
            
            FOR i FROM 0 TO 8  
                IF board AT i AND col EQUALS num  
                    RETURN False  
            
            SET start_row TO 3 MULTIPLIED BY (row DIVIDED BY 3)  
            SET start_col TO 3 MULTIPLIED BY (col DIVIDED BY 3)  
            FOR i FROM 0 TO 2  
                FOR j FROM 0 TO 2  
                    IF board AT (start_row PLUS i) AND (start_col PLUS j) EQUALS num  
                        RETURN False  
            
            RETURN True  
        END FUNCTION  
        
        FUNCTION solve(board)  
            FOR row FROM 0 TO 8  
                FOR col FROM 0 TO 8  
                    IF board AT row AND col EQUALS '.'  
                        FOR num FROM 1 TO 9  
                            SET num_str TO CONVERT num TO STRING  
                            IF CALL is_valid WITH board, row, col, num_str RETURNS True  
                                SET board AT row AND col TO num_str  
                                IF CALL solve WITH board RETURNS True  
                                    RETURN True  
                                SET board AT row AND col TO '.'  
                        RETURN False  
            RETURN True  
        END FUNCTION  
        
        CALL solve WITH board  
    END FUNCTION  
END CLASS
