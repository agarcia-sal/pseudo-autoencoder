CLASS Solution  
    FUNCTION makesquare(matchsticks)  
        SET total_length TO sum of matchsticks  
        IF total_length MODULO 4 NOT EQUALS 0  
            RETURN False  
        
        SET side_length TO total_length DIVIDED BY 4  
        SORT matchsticks IN descending order  
        
        FUNCTION can_form_sides(sides, index)  
            IF index EQUALS LENGTH OF matchsticks  
                RETURN all elements in sides EQUAL side_length  
            
            FOR i FROM 0 TO 3  
                IF sides at i PLUS matchsticks at index LESS THAN OR EQUAL TO side_length  
                    INCREMENT sides at i BY matchsticks at index  
                    IF CALL can_form_sides WITH sides AND index PLUS 1 RETURNS True  
                        RETURN True  
                    DECREMENT sides at i BY matchsticks at index  
                
                IF sides at i EQUALS 0  
                    BREAK  
            
            RETURN False  
        END FUNCTION  
        
        RETURN CALL can_form_sides WITH list of four zeros AND 0  
    END FUNCTION  
END CLASS
