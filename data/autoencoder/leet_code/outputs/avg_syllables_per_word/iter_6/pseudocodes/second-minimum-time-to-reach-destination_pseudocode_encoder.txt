CLASS Solution  
    FUNCTION secondMinimum(n, edges, time, change)  
        SET graph TO list of empty list for index FROM 0 TO n  
        FOR each pair (u, v) IN edges  
            APPEND v TO graph at index u  
            APPEND u TO graph at index v  
        
        SET queue TO deque initialized WITH tuple (1, 0, 0)  
        SET minTimes TO list of lists filled WITH infinity value of length 2 for index FROM 0 TO n  
        SET minTimes at index 1 and 0 TO 0  
        
        WHILE queue IS NOT empty  
            SET current, currentTime, count TO queue popped from the left  
            
            IF current EQUALS n AND count EQUALS 1  
                RETURN currentTime  
            
            IF (currentTime DIVIDED BY change) MODULO 2 EQUALS 0  
                SET nextTime TO currentTime PLUS time  
            ELSE  
                SET nextTime TO ((currentTime DIVIDED BY change) PLUS 1) MULTIPLIED BY change PLUS time  
            
            FOR each neighbor IN graph at index current  
                IF minTimes at neighbor index 0 GREATER THAN nextTime  
                    SET minTimes at neighbor index 1 TO minTimes at neighbor index 0  
                    SET minTimes at neighbor index 0 TO nextTime  
                    APPEND tuple (neighbor, nextTime, 0) TO queue  
                ELSE IF minTimes at neighbor index 0 LESS THAN nextTime AND nextTime LESS THAN minTimes at neighbor index 1  
                    SET minTimes at neighbor index 1 TO nextTime  
                    APPEND tuple (neighbor, nextTime, 1) TO queue  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
