CLASS Solution  
    FUNCTION fullJustify(words, maxWidth)  
        FUNCTION justify_line(line, num_of_letters, is_last_line)  
            IF LENGTH OF line EQUALS 1 OR is_last_line IS TRUE  
                RETURN concatenation OF elements IN line SEPARATED BY single space PLUS number OF spaces EQUAL TO maxWidth MINUS num_of_letters MINUS LENGTH OF line PLUS 1  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO LENGTH OF line MINUS 1  
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps  
                SET extra_spaces TO total_spaces MODULO num_of_gaps  
                
                FOR index FROM 0 TO num_of_gaps MINUS 1  
                    IF index LESS THAN extra_spaces  
                        APPEND spaces_between_words PLUS 1 spaces TO line AT index  
                    ELSE  
                        APPEND spaces_between_words spaces TO line AT index  
                
                RETURN concatenation OF elements IN line WITHOUT separator  
        END FUNCTION  
        
        SET result TO empty list  
        SET line TO empty list  
        SET num_of_letters TO 0  
        
        FOR each word IN words  
            IF num_of_letters PLUS LENGTH OF word PLUS LENGTH OF line GREATER THAN maxWidth  
                APPEND CALL to justify_line WITH line, num_of_letters, FALSE TO result  
                SET line TO empty list  
                SET num_of_letters TO 0  
            APPEND word TO line  
            INCREMENT num_of_letters BY LENGTH OF word  
        
        APPEND CALL to justify_line WITH line, num_of_letters, TRUE TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
