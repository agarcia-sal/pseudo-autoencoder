CLASS Solution  
    FUNCTION countComponents(n, edges)  
        SET graph TO a mapping from each integer from 0 TO n MINUS 1 TO an empty list  
        FOR each pair a, b IN edges  
            APPEND b TO graph at key a  
            APPEND a TO graph at key b  
        
        SET visited TO an empty set  
        
        FUNCTION dfs(node)  
            IF node IS IN visited  
                RETURN  
            ADD node TO visited  
            FOR each neighbor IN graph at key node  
                CALL dfs WITH neighbor  
        END FUNCTION  
        
        SET components TO 0  
        FOR node FROM 0 TO n MINUS 1  
            IF node IS NOT IN visited  
                CALL dfs WITH node  
                INCREMENT components BY 1  
        
        RETURN components  
    END FUNCTION  
END CLASS
