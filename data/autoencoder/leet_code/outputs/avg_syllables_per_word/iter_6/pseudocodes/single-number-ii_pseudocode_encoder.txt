CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET ones TO 0  
        SET twos TO 0  
        FOR each num IN nums  
            SET ones TO bits_in_ones EXCLUSIVE OR bits_in_num AND NOT bits_in_twos  
            SET twos TO bits_in_twos EXCLUSIVE OR bits_in_num AND NOT bits_in_ones  
        RETURN ones  
    END FUNCTION  
END CLASS
