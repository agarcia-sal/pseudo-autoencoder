CLASS ListNode  
    FUNCTION __init__(val WITH DEFAULT 0, next WITH DEFAULT None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET head TO CALL ListNode WITH values AT INDEX 0  
    SET p TO head  
    FOR each val IN values STARTING FROM INDEX 1  
        SET node TO CALL ListNode WITH val  
        SET p.next TO node  
        SET p TO node  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN (p1.val EQUALS p2.val) AND CALL is_same_list WITH p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs(head)  
        SET dummy TO CALL ListNode WITH 0  
        SET dummy.next TO head  
        SET current TO dummy  
        WHILE current.next IS NOT None AND current.next.next IS NOT None  
            SET first TO current.next  
            SET second TO current.next.next  
            SET current.next TO second  
            SET first.next TO second.next  
            SET second.next TO first  
            SET current TO current.next.next  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
