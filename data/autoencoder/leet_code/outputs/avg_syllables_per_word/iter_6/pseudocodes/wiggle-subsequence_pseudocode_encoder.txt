CLASS Solution  
    FUNCTION wiggleMaxLength(nums)  
        IF LENGTH OF nums LESS THAN 2 THEN  
            RETURN LENGTH OF nums  
        END IF  

        SET up TO 1  
        SET down TO 1  

        FOR index FROM 1 TO LENGTH OF nums MINUS 1  
            IF nums AT index GREATER THAN nums AT index MINUS 1 THEN  
                SET up TO down PLUS 1  
            ELSE IF nums AT index LESS THAN nums AT index MINUS 1 THEN  
                SET down TO up PLUS 1  
            END IF  
        END FOR  

        RETURN MAXIMUM OF up AND down  
    END FUNCTION  
END CLASS
