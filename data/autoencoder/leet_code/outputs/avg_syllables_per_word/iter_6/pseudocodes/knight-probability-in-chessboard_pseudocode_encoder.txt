CLASS Solution  
    FUNCTION knightProbability(n, k, row, column)  
        SET moves TO list of coordinate pairs representing knight moves:  
            (-2, MINUS 1), (-2, 1), (-1, MINUS 2), (-1, 2), (1, MINUS 2), (1, 2), (2, MINUS 1), (2, 1)  
        
        INITIALIZE dp as a two-dimensional list of size n by n with all values set to 0  
        SET dp at position row and column TO 1  
        
        FOR each move FROM 1 TO k  
            INITIALIZE new_dp as a two-dimensional list of size n by n with all values set to 0  
            
            FOR r FROM 0 TO n MINUS 1  
                FOR c FROM 0 TO n MINUS 1  
                    IF dp at r and c EQUALS 0  
                        CONTINUE to next iteration  
                    FOR each dr, dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr GREATER THAN OR EQUAL TO 0 AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO 0 AND nc LESS THAN n  
                            INCREMENT new_dp at nr and nc BY dp at r and c DIVIDED BY 8  
            SET dp TO new_dp  
        
        SET total_probability TO 0  
        FOR each row_list IN dp  
            INCREMENT total_probability BY sum of row_list  
        RETURN total_probability  
    END FUNCTION  
END CLASS
