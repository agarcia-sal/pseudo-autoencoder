CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO sum of all elements in nums  
        
        IF total_sum MODULO 2 NOT EQUALS 0 THEN  
            RETURN False  
        
        SET target TO total_sum DIVIDED BY 2  
        SET n TO length of nums  
        
        INITIALIZE dp AS list of length target PLUS 1 with all values False  
        SET dp at index 0 TO True  
        
        FOR each num IN nums  
            FOR i FROM target DOWNTO num  
                SET dp at index i TO dp at index i OR dp at index (i MINUS num)  
        
        RETURN dp at index target  
    END FUNCTION  
END CLASS
