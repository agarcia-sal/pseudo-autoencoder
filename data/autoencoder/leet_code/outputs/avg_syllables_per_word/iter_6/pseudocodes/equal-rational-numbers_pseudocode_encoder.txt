CLASS Solution  
    FUNCTION isRationalEqual(s, t)  
        SET ratios TO list containing values 1, 1 DIVIDED BY 9, 1 DIVIDED BY 99, 1 DIVIDED BY 999, 1 DIVIDED BY 9999  

        FUNCTION valueOf(s)  
            IF s DOES NOT CONTAIN character '(' THEN  
                RETURN s CONVERTED TO FLOAT  
            END IF  

            SET leftParenIndex TO POSITION OF character '(' IN s  
            SET rightParenIndex TO POSITION OF character ')' IN s  
            SET dotIndex TO POSITION OF character '.' IN s  

            SET integerAndNonRepeating TO FLOAT OF SUBSTRING OF s FROM START TO leftParenIndex  
            SET nonRepeatingLength TO leftParenIndex MINUS dotIndex MINUS 1  

            SET repeating TO INTEGER OF SUBSTRING OF s FROM leftParenIndex PLUS 1 TO rightParenIndex  
            SET repeatingLength TO rightParenIndex MINUS leftParenIndex MINUS 1  

            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY 0.1 TO THE POWER OF nonRepeatingLength MULTIPLIED BY ratios AT INDEX repeatingLength  
        END FUNCTION  

        RETURN ABSOLUTE VALUE OF valueOf APPLIED TO s MINUS valueOf APPLIED TO t LESS THAN 0.000000001  
    END FUNCTION  
END CLASS
