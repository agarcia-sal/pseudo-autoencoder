CLASS Solution  
    FUNCTION isMatch(s, p)  
        INITIALIZE dp AS two-dimensional list of False with dimensions LENGTH OF s PLUS 1 BY LENGTH OF p PLUS 1  
        SET dp AT position 0,0 TO True  

        FOR j FROM 1 TO LENGTH OF p  
            IF character AT position j MINUS 1 IN p EQUALS '*'  
                SET dp AT position 0,j TO dp AT position 0,j MINUS 1  

        FOR i FROM 1 TO LENGTH OF s  
            FOR j FROM 1 TO LENGTH OF p  
                IF character AT position j MINUS 1 IN p EQUALS '*'  
                    SET dp AT position i,j TO dp AT position i MINUS 1,j OR dp AT position i,j MINUS 1  
                ELSE IF character AT position j MINUS 1 IN p EQUALS '?' OR character AT position i MINUS 1 IN s EQUALS character AT position j MINUS 1 IN p  
                    SET dp AT position i,j TO dp AT position i MINUS 1,j MINUS 1  

        RETURN dp AT position LENGTH OF s,LENGTH OF p  
    END FUNCTION  
END CLASS
