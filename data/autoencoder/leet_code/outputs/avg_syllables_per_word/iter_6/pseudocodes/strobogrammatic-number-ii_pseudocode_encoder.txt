CLASS Solution  
    FUNCTION findStrobogrammatic(n)  
        FUNCTION dfs(u)  
            IF u EQUALS 0  
                RETURN list containing empty string  
            ELSE IF u EQUALS 1  
                RETURN list containing strings "0", "1", and "8"  
            SET ans TO empty list  
            FOR each v IN CALL dfs WITH argument u MINUS 2  
                FOR each pair (l, r) IN list of pairs ("1","1"), ("8","8"), ("6","9"), ("9","6")  
                    APPEND concatenation of l PLUS v PLUS r TO ans  
                IF u NOT EQUALS n  
                    APPEND concatenation of "0" PLUS v PLUS "0" TO ans  
            RETURN ans  
        END FUNCTION  
        RETURN CALL dfs WITH argument n  
    END FUNCTION  
END CLASS
