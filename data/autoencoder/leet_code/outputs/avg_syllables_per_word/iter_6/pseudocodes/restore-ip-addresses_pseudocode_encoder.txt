CLASS Solution  
    FUNCTION restoreIpAddresses(s)  
        FUNCTION is_valid(segment)  
            IF LENGTH OF segment EQUALS 1 OR (FIRST CHARACTER OF segment NOT EQUALS '0' AND CONVERT segment TO NUMBER LESS THAN OR EQUAL TO 255)  
                RETURN TRUE  
            ELSE  
                RETURN FALSE  
            END IF  
        END FUNCTION  

        FUNCTION backtrack(start, path)  
            IF LENGTH OF path EQUALS 4  
                IF start EQUALS LENGTH OF s  
                    APPEND JOIN path WITH '.' TO result  
                END IF  
                RETURN  
            END IF  
            FOR length FROM 1 TO 3  
                IF start PLUS length LESS THAN OR EQUAL TO LENGTH OF s  
                    SET segment TO SUBSTRING OF s FROM start TO start PLUS length  
                    IF CALL is_valid WITH segment  
                        CALL backtrack WITH start PLUS length AND path PLUS segment  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  

        SET result TO EMPTY LIST  
        CALL backtrack WITH 0 AND EMPTY LIST  
        RETURN result  
    END FUNCTION  
END CLASS
