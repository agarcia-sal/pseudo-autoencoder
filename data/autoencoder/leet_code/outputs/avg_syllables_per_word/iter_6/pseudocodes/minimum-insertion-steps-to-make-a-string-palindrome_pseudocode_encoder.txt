CLASS Solution  
    FUNCTION minInsertions(s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO length of s  
            INITIALIZE dp AS 2D array of size n BY n with all values zero  
            FOR i FROM 0 TO n MINUS 1  
                SET dp at position i, i TO 1  
            FOR cl FROM 2 TO n  
                FOR i FROM 0 TO n MINUS cl  
                    SET j TO i PLUS cl MINUS 1  
                    IF character at s position i EQUALS character at s position j AND cl EQUALS 2  
                        SET dp at i, j TO 2  
                    ELSE IF character at s position i EQUALS character at s position j  
                        SET dp at i, j TO dp at i PLUS 1, j MINUS 1 PLUS 2  
                    ELSE  
                        SET dp at i, j TO maximum of dp at i, j MINUS 1 AND dp at i PLUS 1, j  
            RETURN dp at 0, n MINUS 1  
        END FUNCTION  
        RETURN length of s MINUS CALL longest_palindromic_subsequence WITH s  
    END FUNCTION  
END CLASS
