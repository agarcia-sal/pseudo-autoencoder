CLASS Solution  
    FUNCTION repeatedStringMatch(a, b)  
        SET min_repeats TO negative division of length of b BY length of a rounded up  
        IF b IS substring OF a repeated min_repeats times  
            RETURN min_repeats  
        ELSE IF b IS substring OF a repeated (min_repeats PLUS 1) times  
            RETURN min_repeats PLUS 1  
        ELSE  
            RETURN negative one  
    END FUNCTION  
END CLASS
