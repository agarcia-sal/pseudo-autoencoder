CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO dictionary mapping each integer from 0 TO n MINUS 1 TO empty list  
        FOR each flight IN flights  
            SET u TO first element of flight  
            SET v TO second element of flight  
            SET w TO third element of flight  
            APPEND pair of v and w TO graph at key u  
        
        SET min_heap TO list containing tuple of zero cost, minus one stops, and src city  
        SET min_cost TO dictionary with key src mapped to minus one  
        
        WHILE min_heap IS NOT empty  
            CALL function heap pop on min_heap ASSIGNING cost, stops, city  
            
            IF city EQUALS dst  
                RETURN cost  
            
            IF stops GREATER THAN OR EQUAL TO k  
                CONTINUE  
            
            FOR each neighbor and price IN graph at key city  
                SET next_cost TO cost PLUS price  
                SET next_stops TO stops PLUS one  
                
                IF neighbor NOT IN min_cost OR next_stops LESS THAN min_cost at key neighbor OR next_cost LESS THAN first element of first tuple in min_heap  
                    SET min_cost at key neighbor TO next_stops  
                    CALL function heap push on min_heap WITH tuple of next_cost, next_stops, neighbor  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
