CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MOD TO 10 RAISED TO THE POWER OF 9 PLUS 7  
        CALL sort ON nums  
        SET n TO LENGTH OF nums  
        SET total_sum TO 0  
        SET power TO list of length n FILLED WITH 1  
        
        FOR i FROM 1 TO n MINUS 1  
            SET power AT i TO power AT (i MINUS 1) MULTIPLIED BY 2 MODULO MOD  
        
        FOR i FROM 0 TO n MINUS 1  
            SET total_sum TO total_sum PLUS nums AT i MULTIPLIED BY (power AT i MINUS power AT (n MINUS i MINUS 1)) MODULO MOD  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
