CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO set of characters from time with colon removed  
        SET hour, minute TO integer values obtained by splitting time at colon  

        FUNCTION generate_times(digits)  
            SET times TO empty list  
            FOR each first_hour_digit IN digits  
                FOR each second_hour_digit IN digits  
                    FOR each first_minute_digit IN digits  
                        FOR each second_minute_digit IN digits  
                            SET hour_value TO integer formed by concatenating first_hour_digit and second_hour_digit  
                            SET minute_value TO integer formed by concatenating first_minute_digit and second_minute_digit  
                            IF hour_value LESS THAN 24 AND minute_value LESS THAN 60  
                                APPEND tuple of hour_value and minute_value TO times  
            RETURN times sorted in ascending order  
        END FUNCTION  

        SET all_times TO CALL generate_times WITH digits  
        SET current_time_index TO position of tuple (hour, minute) IN all_times  

        IF current_time_index PLUS 1 LESS THAN length of all_times  
            SET next_hour, next_minute TO element at current_time_index PLUS 1 IN all_times  
        ELSE  
            SET next_hour, next_minute TO first element IN all_times  
        RETURN string formatted as two-digit next_hour plus colon plus two-digit next_minute  
    END FUNCTION  
END CLASS
