CLASS Solution  
    FUNCTION profitableSchemes(n, minProfit, group, profit)  
        SET MOD TO one billion PLUS seven  
        INITIALIZE dp AS two-dimensional list with dimensions (n PLUS one) BY (minProfit PLUS one), FILLED WITH zero  
        FOR i FROM zero TO n  
            SET dp at position i and zero TO one  
        FOR each p, g IN pairs from profit and group  
            FOR i FROM n DOWN TO g  
                FOR j FROM minProfit DOWN TO zero  
                    SET dp at position i and j TO sum of dp at position i and j PLUS dp at position (i MINUS g) and maximum of zero and (j MINUS p), MODULO MOD  
        RETURN dp at position n and minProfit  
    END FUNCTION  
END CLASS
