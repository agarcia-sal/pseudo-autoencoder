CLASS Solution  
    FUNCTION bestRotation(nums)  
        SET n TO LENGTH OF nums  
        SET delta TO list of zeros with length n  
        
        FOR i FROM 0 TO n MINUS 1  
            SET num TO nums at index i  
            IF num LESS THAN OR EQUAL TO i  
                INCREMENT delta at index 0 BY 1  
                IF i MINUS num PLUS 1 LESS THAN n  
                    DECREMENT delta at index (i MINUS num PLUS 1) BY 1  
            END IF  
            IF i PLUS 1 LESS THAN n  
                INCREMENT delta at index (i PLUS 1) BY 1  
                IF (i PLUS n MINUS num) PLUS 1 LESS THAN n  
                    DECREMENT delta at index ((i PLUS n MINUS num) PLUS 1) BY 1  
                END IF  
            END IF  
        END FOR  
        
        SET score TO 0  
        SET max_score TO 0  
        SET best_k TO 0  
        
        FOR k FROM 0 TO n MINUS 1  
            INCREMENT score BY delta at index k  
            IF score GREATER THAN max_score  
                SET max_score TO score  
                SET best_k TO k  
            END IF  
        END FOR  
        
        RETURN best_k  
    END FUNCTION  
END CLASS
