CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        
        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS reverse of check_str  
        END FUNCTION  
        
        FOR i FROM length of s DOWNTO 0  
            IF CALL is_palindrome WITH s from start TO index i  
                SET to_add TO reverse of substring of s from index i TO end  
                RETURN concatenation of to_add AND s  
        END FOR  
    END FUNCTION  
END CLASS
