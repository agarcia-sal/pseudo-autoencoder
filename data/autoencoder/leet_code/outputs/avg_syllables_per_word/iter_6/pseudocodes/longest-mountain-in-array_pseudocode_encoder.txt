CLASS Solution  
    FUNCTION longestMountain(arr)  
        SET n TO LENGTH OF arr  
        SET longest TO 0  
        
        SET i TO 1  
        WHILE i LESS THAN n MINUS 1  
            IF arr AT i MINUS 1 LESS THAN arr AT i AND arr AT i GREATER THAN arr AT i PLUS 1  
                SET left TO i MINUS 1  
                SET right TO i PLUS 1  
                
                WHILE left GREATER THAN 0 AND arr AT left MINUS 1 LESS THAN arr AT left  
                    DECREMENT left BY 1  
                
                WHILE right LESS THAN n MINUS 1 AND arr AT right GREATER THAN arr AT right PLUS 1  
                    INCREMENT right BY 1  
                
                SET longest TO MAXIMUM OF longest AND right MINUS left PLUS 1  
                
                SET i TO right  
            ELSE  
                INCREMENT i BY 1  
        
        RETURN longest  
    END FUNCTION  
END CLASS
