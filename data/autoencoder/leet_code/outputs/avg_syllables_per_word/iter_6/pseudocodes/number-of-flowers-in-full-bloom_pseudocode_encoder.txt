CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO sorted list of first elements from each flower in flowers  
        SET end_times TO sorted list of second elements from each flower in flowers  
        SET result TO empty list  
        FOR each person IN people  
            SET started TO CALL bisect_right WITH start_times AND person  
            SET ended TO CALL bisect_left WITH end_times AND person  
            APPEND started MINUS ended TO result  
        RETURN result  
    END FUNCTION  
END CLASS
