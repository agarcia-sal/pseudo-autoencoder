CLASS Solution  
    FUNCTION myAtoi(s)  
        SET INT_MAX TO two raised to the power of thirty-one MINUS one  
        SET INT_MIN TO negative two raised to the power of thirty-one  
        
        SET result TO zero  
        SET index TO zero  
        SET n TO length of s  
        SET sign TO one  
        
        WHILE index LESS THAN n AND character at s[index] EQUALS whitespace  
            INCREMENT index BY one  
        
        IF index LESS THAN n AND (character at s[index] EQUALS plus OR character at s[index] EQUALS minus)  
            IF character at s[index] EQUALS minus  
                SET sign TO negative one  
            ELSE  
                SET sign TO one  
            INCREMENT index BY one  
        
        WHILE index LESS THAN n AND character at s[index] IS a digit  
            SET digit TO numeric value of character at s[index]  
            
            IF result GREATER THAN (INT_MAX MINUS digit) DIVIDED BY ten  
                IF sign EQUALS one  
                    RETURN INT_MAX  
                ELSE  
                    RETURN INT_MIN  
            
            SET result TO result MULTIPLIED BY ten PLUS digit  
            INCREMENT index BY one  
        
        RETURN sign MULTIPLIED BY result  
    END FUNCTION  
END CLASS
