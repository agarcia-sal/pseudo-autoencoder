CLASS Solution  
    FUNCTION nthSuperUglyNumber(n, primes)  
        SET ugly_numbers TO list containing 1  
        SET heap TO list of pairs each containing a prime and its index from primes  
        CALL heapify ON heap  
        SET indices TO list of zeros with length equal to length of primes  
        
        WHILE length of ugly_numbers IS LESS THAN n  
            CALL heap pop on heap AND ASSIGN TO next_ugly AND prime_index  
            
            IF next_ugly NOT EQUALS last element in ugly_numbers  
                APPEND next_ugly TO ugly_numbers  
            END IF  
            
            INCREMENT indices at position prime_index BY 1  
            SET next_multiple TO prime at prime_index MULTIPLIED BY ugly_numbers at indices at prime_index  
            CALL heap push with pair (next_multiple, prime_index) ON heap  
        END WHILE  
        
        RETURN last element in ugly_numbers  
    END FUNCTION  
END CLASS
