CLASS Solution  
    FUNCTION boxDelivering(boxes, portsCount, maxBoxes, maxWeight)  
        SET n TO LENGTH OF boxes  
        SET dp TO list of size (n PLUS 1) FILLED WITH infinity  
        SET dp[0] TO 0  
        
        SET left TO 0  
        SET current_weight TO 0  
        SET extra_trips TO 0  
        
        FOR right FROM 0 TO n MINUS 1  
            INCREMENT current_weight BY weight part of boxes at index right  
            
            IF right GREATER THAN 0 AND port part of boxes at index right NOT EQUALS port part of boxes at index right MINUS 1  
                INCREMENT extra_trips BY 1  
            END IF  
            
            WHILE left LESS THAN OR EQUAL TO right AND (number of boxes in current window GREATER THAN maxBoxes OR current_weight GREATER THAN maxWeight)  
                DECREMENT current_weight BY weight part of boxes at index left  
                
                IF left GREATER THAN 0 AND port part of boxes at index left NOT EQUALS port part of boxes at index left MINUS 1  
                    DECREMENT extra_trips BY 1  
                END IF  
                
                INCREMENT left BY 1  
            END WHILE  
            
            SET dp at position right PLUS 1 TO dp at position left PLUS 2 PLUS extra_trips  
        END FOR  
        
        RETURN dp at position n  
    END FUNCTION  
END CLASS
