CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i GREATER THAN j THEN  
                RETURN 0  
            IF memo CONTAINS KEY (i, j) THEN  
                RETURN VALUE FROM memo AT (i, j)  
              
            SET result TO CALL dp WITH (i, j MINUS 1, memo) PLUS 1  
            
            FOR k FROM i TO j MINUS 1  
                IF CHARACTER AT s[k] EQUALS CHARACTER AT s[j] THEN  
                    SET temp_result TO CALL dp WITH (i, k MINUS 1, memo) PLUS CALL dp WITH (k, j MINUS 1, memo)  
                    IF temp_result LESS THAN result THEN  
                        SET result TO temp_result  
            
            SET memo AT (i, j) TO result  
            RETURN result  
        END FUNCTION  
        
        RETURN CALL dp WITH (0, LENGTH OF s MINUS 1, EMPTY DICTIONARY)  
    END FUNCTION  
END CLASS
