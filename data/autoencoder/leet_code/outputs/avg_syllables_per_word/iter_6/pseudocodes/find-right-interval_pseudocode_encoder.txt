CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO sorted list of tuples containing (interval start time, interval index) FOR each interval WITH index IN intervals  
        SET result TO empty list  
        FOR each interval IN intervals  
            SET end TO interval end time  
            SET idx TO position found by binary search to insert (end) INTO start_times to maintain order  
            IF idx LESS THAN length of start_times  
                APPEND start_times at position idx second element TO result  
            ELSE  
                APPEND negative one TO result  
        RETURN result  
    END FUNCTION  
END CLASS
