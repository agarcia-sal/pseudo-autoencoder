CLASS Solution  
    FUNCTION findKthNumber(m, n, k)  
        FUNCTION enough(x)  
            SET count TO 0  
            FOR i FROM 1 TO m  
                INCREMENT count BY minimum of integer division of x by i AND n  
            IF count GREATER THAN OR EQUAL TO k  
                RETURN true  
            ELSE  
                RETURN false  
        END FUNCTION  

        SET low TO 1  
        SET high TO m MULTIPLIED BY n  
        WHILE low LESS THAN high  
            SET mid TO integer division of low PLUS high BY 2  
            IF CALL enough WITH mid IS true  
                SET high TO mid  
            ELSE  
                SET low TO mid PLUS 1  
        RETURN low  
    END FUNCTION  
END CLASS
