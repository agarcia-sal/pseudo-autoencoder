CLASS Solution  
    FUNCTION carFleet(target, position, speed)  
        CALL sorting of pairs of position and speed BY descending order  
        SET time_to_target TO empty list  
        FOR each pair (p, s) IN sorted pairs  
            COMPUTE time as target MINUS p DIVIDED BY s  
            APPEND time TO time_to_target  
        SET fleets TO 0  
        SET current_max_time TO 0  
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time  
                INCREMENT fleets BY 1  
                SET current_max_time TO time  
        RETURN fleets  
    END FUNCTION  
END CLASS
