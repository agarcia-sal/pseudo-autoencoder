CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SPLIT time INTO hours AND minutes BY delimiter colon  
            SET hours TO converted integer of hours  
            SET minutes TO converted integer of minutes  
            RETURN hours MULTIPLIED BY 60 PLUS minutes  
        END FUNCTION  
        
        SET minutes_list TO list of CALL to_minutes WITH each element IN timePoints  
        SORT minutes_list  
        
        SET min_diff TO infinity  
        FOR index FROM 1 TO LENGTH OF minutes_list MINUS 1  
            SET difference TO minutes_list AT index MINUS minutes_list AT index MINUS 1  
            IF difference LESS THAN min_diff THEN  
                SET min_diff TO difference  
            END IF  
        END FOR  
        
        SET circular_difference TO 1440 MINUS minutes_list AT last index PLUS minutes_list AT first index  
        IF circular_difference LESS THAN min_diff THEN  
            SET min_diff TO circular_difference  
        END IF  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
