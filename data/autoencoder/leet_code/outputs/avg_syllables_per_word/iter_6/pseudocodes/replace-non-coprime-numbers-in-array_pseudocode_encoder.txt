CLASS Solution  
    FUNCTION replaceNonCoprimes(nums)  
        SET stack TO empty list  
        FOR each num IN nums  
            WHILE stack IS NOT empty  
                SET g TO CALL gcd WITH stack last element AND num  
                IF g EQUALS 1  
                    BREAK  
                SET num TO stack last element MULTIPLIED BY num DIVIDED BY g  
                REMOVE last element FROM stack  
            APPEND num TO stack  
        RETURN stack  
    END FUNCTION  
END CLASS
