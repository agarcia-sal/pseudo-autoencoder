CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO a list of FALSE repeated n PLUS 1 times  
        FOR i FROM 1 TO n  
            SET j TO 1  
            WHILE j MULTIPLIED BY j LESS THAN OR EQUAL TO i  
                IF dp AT i MINUS j MULTIPLIED BY j EQUALS FALSE  
                    SET dp AT i TO TRUE  
                    BREAK  
                INCREMENT j BY 1  
        RETURN dp AT n  
    END FUNCTION  
END CLASS
