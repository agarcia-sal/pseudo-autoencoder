CLASS Solution  
    FUNCTION myPow(x, n)  
        FUNCTION qpow(a, n)  
            SET ans TO 1  
            WHILE n GREATER THAN 0  
                IF n MODULO 2 EQUALS 1  
                    SET ans TO ans MULTIPLIED BY a  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY 2 (integer division)  
            RETURN ans  
        END FUNCTION  
        
        IF n GREATER THAN OR EQUAL TO 0  
            RETURN CALL qpow WITH a AS x AND n AS n  
        ELSE  
            RETURN 1 DIVIDED BY CALL qpow WITH a AS x AND n AS NEGATION OF n  
    END FUNCTION  
END CLASS
