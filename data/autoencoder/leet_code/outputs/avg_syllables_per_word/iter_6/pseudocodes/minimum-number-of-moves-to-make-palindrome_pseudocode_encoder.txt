CLASS Solution  
    FUNCTION minMovesToMakePalindrome(s)  
        CONVERT s TO list of characters  
        SET n TO LENGTH OF s  
        SET moves TO 0  
        
        WHILE LENGTH OF s GREATER THAN 1  
            SET found_match TO FALSE  
            FOR i FROM LENGTH OF s MINUS 1 DOWN TO 1  
                IF s AT i EQUALS s AT 0  
                    SET found_match TO TRUE  
                    FOR j FROM i TO LENGTH OF s MINUS 2  
                        SWAP s AT j WITH s AT j PLUS 1  
                        INCREMENT moves BY 1  
                    REMOVE first element FROM s  
                    REMOVE last element FROM s  
                    BREAK  
            IF found_match IS FALSE  
                REMOVE first element FROM s  
        
        RETURN moves  
    END FUNCTION  
END CLASS
