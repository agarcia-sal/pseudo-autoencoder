CLASS Solution  
    FUNCTION numOfArrays(n, m, k)  
        SET MOD TO ten raised to the ninth power PLUS seven  
        
        INITIALIZE three-dimensional list dp with dimensions (n PLUS one) by (m PLUS one) by (k PLUS one) FILLED WITH zero  
        
        FOR j FROM one TO m  
            SET dp at position [one][j][one] TO one  
        
        FOR i FROM two TO n  
            FOR j FROM one TO m  
                FOR p FROM one TO k  
                    FOR x FROM one TO j MINUS one  
                        SET dp at position [i][j][p] TO dp at position [i][j][p] PLUS dp at position [i MINUS one][x][p MINUS one]  
                        SET dp at position [i][j][p] TO dp at position [i][j][p] MODULO MOD  
                    
                    SET dp at position [i][j][p] TO dp at position [i][j][p] PLUS dp at position [i MINUS one][j][p] MULTIPLIED BY j  
                    SET dp at position [i][j][p] TO dp at position [i][j][p] MODULO MOD  
        
        SET result TO zero  
        FOR j FROM one TO m  
            SET result TO result PLUS dp at position [n][j][k]  
            SET result TO result MODULO MOD  
        
        RETURN result  
    END FUNCTION  
END CLASS
