CLASS Solution  
    FUNCTION stoneGameVIII(stones)  
        SET prefix_sum TO list of zeros with length EQUAL TO length of stones  
        SET prefix_sum at index 0 TO stones at index 0  
        FOR index FROM 1 TO length of stones MINUS 1  
            SET prefix_sum at index TO prefix_sum at index MINUS 1 PLUS stones at index  
        SET dp TO last element of prefix_sum  
        FOR index FROM length of stones MINUS 2 DOWN TO 1  
            IF dp GREATER THAN OR EQUAL TO prefix_sum at index MINUS dp  
                SET dp TO dp  
            ELSE  
                SET dp TO prefix_sum at index MINUS dp  
        RETURN dp  
    END FUNCTION  
END CLASS
