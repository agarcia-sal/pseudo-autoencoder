CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO result of replacing plus signs with delimiter plus sign in side, and replacing minus signs with delimiter minus sign, then splitting by delimiter  
            SET x_count TO 0  
            SET num_sum TO 0  
            FOR each token IN tokens  
                IF token IS empty  
                    CONTINUE to next iteration  
                IF token CONTAINS 'x'  
                    IF token EQUALS 'x' OR token EQUALS plus sign PLUS 'x'  
                        INCREMENT x_count BY 1  
                    ELSE IF token EQUALS minus sign PLUS 'x'  
                        DECREMENT x_count BY 1  
                    ELSE  
                        SET coefficient TO integer value of token with 'x' removed  
                        INCREMENT x_count BY coefficient  
                ELSE  
                    INCREMENT num_sum BY integer value of token  
            RETURN x_count, num_sum  
        END FUNCTION  

        SPLIT equation INTO left_side AND right_side BY equal sign  
        CALL parse_side WITH left_side AND STORE RESULTS IN left_x AND left_num  
        CALL parse_side WITH right_side AND STORE RESULTS IN right_x AND right_num  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS 0  
            IF total_num EQUALS 0  
                RETURN "Infinite solutions"  
            ELSE  
                RETURN "No solution"  
        ELSE  
            RETURN string constructed by concatenating "x=" AND total_num DIVIDED BY total_x as integer division  
    END FUNCTION  
END CLASS
