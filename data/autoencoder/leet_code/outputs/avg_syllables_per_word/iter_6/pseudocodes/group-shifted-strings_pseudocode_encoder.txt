CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s IS EMPTY  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET difference TO (numeric position of c MINUS numeric position of first character of s) MODULO 26  
                APPEND string representation of difference TO canonical_form  
            END FOR  
            SET joined_form TO concatenation of elements in canonical_form separated by comma  
            RETURN joined_form  
        END FUNCTION  

        SET groups TO a mapping from keys to lists with default empty list  
        FOR each string s IN strings  
            SET canonical_form TO CALL get_canonical_form WITH s  
            APPEND s TO groups AT key canonical_form  
        END FOR  
        RETURN list of values in groups  
    END FUNCTION  
END CLASS
