CLASS Solution  
    FUNCTION findLength(nums1, nums2)  
        SET dp TO new two-dimensional array of zeros with dimensions LENGTH OF nums1 PLUS 1 BY LENGTH OF nums2 PLUS 1  
        SET max_length TO 0  
        FOR i FROM 1 TO LENGTH OF nums1  
            FOR j FROM 1 TO LENGTH OF nums2  
                IF element at index i MINUS 1 in nums1 EQUALS element at index j MINUS 1 in nums2 THEN  
                    SET dp element at i and j TO dp element at i MINUS 1 and j MINUS 1 PLUS 1  
                    IF dp element at i and j GREATER THAN max_length THEN  
                        SET max_length TO dp element at i and j  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
