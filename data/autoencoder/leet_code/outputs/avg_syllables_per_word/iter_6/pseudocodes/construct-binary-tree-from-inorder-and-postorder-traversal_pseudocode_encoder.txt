CLASS TreeNode  
    FUNCTION __init__(val SET TO 0, left SET TO None, right SET TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH values AT ZERO  
    SET i TO 1  
    SET queue TO NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.left TO NEW TreeNode WITH values AT i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.right TO NEW TreeNode WITH values AT i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, inorder, postorder)  
        IF inorder IS EMPTY OR postorder IS EMPTY  
            RETURN None  
        SET root_val TO REMOVE LAST ELEMENT FROM postorder  
        SET root TO NEW TreeNode WITH root_val  
        SET root_index TO POSITION OF root_val IN inorder  
        SET root.right TO CALL self.buildTree WITH SUBLIST OF inorder FROM root_index PLUS ONE TO END AND postorder  
        SET root.left TO CALL self.buildTree WITH SUBLIST OF inorder FROM START TO root_index AND postorder  
        RETURN root  
    END FUNCTION  
END CLASS
