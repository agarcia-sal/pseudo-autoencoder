CLASS Solution  
    FUNCTION numRescueBoats(people, limit)  
        CALL sort ON people  
        SET left TO 0  
        SET right TO LENGTH OF people MINUS 1  
        SET boats TO 0  
        WHILE left IS LESS THAN OR EQUAL TO right  
            IF people AT left PLUS people AT right IS LESS THAN OR EQUAL TO limit  
                INCREMENT left BY 1  
            END IF  
            DECREMENT right BY 1  
            INCREMENT boats BY 1  
        END WHILE  
        RETURN boats  
    END FUNCTION  
END CLASS
