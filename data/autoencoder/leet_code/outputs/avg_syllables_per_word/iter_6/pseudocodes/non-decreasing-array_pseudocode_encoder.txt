CLASS Solution  
    FUNCTION checkPossibility(nums)  
        SET modified TO False  
        FOR index FROM 1 TO LENGTH OF nums MINUS 1  
            IF element AT nums AT index LESS THAN element AT nums AT index MINUS 1 THEN  
                IF modified IS True THEN  
                    RETURN False  
                SET modified TO True  
                IF index LESS THAN 2 OR element AT nums AT index GREATER THAN OR EQUAL TO element AT nums AT index MINUS 2 THEN  
                    SET element AT nums AT index MINUS 1 TO element AT nums AT index  
                ELSE  
                    SET element AT nums AT index TO element AT nums AT index MINUS 1  
        RETURN True  
    END FUNCTION  
END CLASS
