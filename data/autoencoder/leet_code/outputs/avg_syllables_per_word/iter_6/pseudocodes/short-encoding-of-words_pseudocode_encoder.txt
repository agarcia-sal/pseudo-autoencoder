CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET reversed_words TO sorted list of each word reversed from words  
        SET total_length TO 0  
        FOR index FROM 0 TO LENGTH OF reversed_words MINUS 1  
            IF index PLUS 1 EQUALS LENGTH OF reversed_words OR reversed_words AT index PLUS 1 DOES NOT START WITH reversed_words AT index  
                INCREMENT total_length BY LENGTH OF reversed_words AT index PLUS 1  
        RETURN total_length  
    END FUNCTION  
END CLASS
