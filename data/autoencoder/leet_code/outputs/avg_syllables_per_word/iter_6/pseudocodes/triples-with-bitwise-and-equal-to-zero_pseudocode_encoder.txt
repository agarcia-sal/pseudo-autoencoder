CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO an empty map with default value zero  
        
        FOR each element i IN nums  
            FOR each element j IN nums  
                SET bitwise_and_result TO bitwise AND of i and j  
                INCREMENT value at key bitwise_and_result IN pair_and_count BY one  
        
        SET triplet_count TO zero  
        
        FOR each element k IN nums  
            FOR each key-value pair pair_and, count IN pair_and_count  
                IF bitwise AND of pair_and and k EQUALS zero THEN  
                    INCREMENT triplet_count BY count  
        
        RETURN triplet_count  
    END FUNCTION  
END CLASS
