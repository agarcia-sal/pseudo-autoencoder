CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture IS EMPTY OR picture AT INDEX 0 IS EMPTY  
            RETURN 0  
        
        SET m TO LENGTH OF picture  
        SET n TO LENGTH OF picture AT INDEX 0  
        SET row_count TO list of zeros with size m  
        SET col_count TO list of zeros with size n  
        SET row_patterns TO empty mapping with default integer zero  
        
        FOR r FROM 0 TO m MINUS 1  
            SET pattern TO concatenation of elements in picture AT INDEX r  
            INCREMENT row_patterns AT KEY pattern BY 1  
            FOR c FROM 0 TO n MINUS 1  
                IF picture AT INDEX r AT INDEX c EQUALS 'B'  
                    INCREMENT row_count AT INDEX r BY 1  
                    INCREMENT col_count AT INDEX c BY 1  
        
        SET lonely_pixels TO 0  
        
        FOR r FROM 0 TO m MINUS 1  
            FOR c FROM 0 TO n MINUS 1  
                IF picture AT INDEX r AT INDEX c EQUALS 'B'  
                    SET pattern TO concatenation of elements in picture AT INDEX r  
                    IF row_count AT INDEX r EQUALS target AND col_count AT INDEX c EQUALS target AND row_patterns AT KEY pattern EQUALS target  
                        INCREMENT lonely_pixels BY 1  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
