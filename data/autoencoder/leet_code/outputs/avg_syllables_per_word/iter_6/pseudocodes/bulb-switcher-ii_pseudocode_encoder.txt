CLASS Solution  
    FUNCTION flipLights(n, presses)  
        SET n TO minimum of n AND 6  
        
        IF presses EQUALS 0  
            RETURN 1  
        
        IF n EQUALS 1  
            RETURN 2  
        
        IF n EQUALS 2  
            IF presses EQUALS 1  
                RETURN 3  
            ELSE  
                RETURN 4  
        
        IF presses EQUALS 1  
            RETURN 4  
        
        IF presses EQUALS 2  
            RETURN 7  
        
        RETURN 8  
    END FUNCTION  
END CLASS
