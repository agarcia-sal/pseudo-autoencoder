CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET head TO CALL ListNode WITH values at index zero, None  
    SET p TO head  
    FOR each val IN values FROM second element TO end  
        SET node TO CALL ListNode WITH val, None  
        SET p.next TO node  
        SET p TO node  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN p1.val EQUALS p2.val AND CALL is_same_list WITH p1.next, p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd(head, n)  
        SET dummy TO CALL ListNode WITH zero, None  
        SET dummy.next TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n INCLUSIVE  
            SET first TO first.next  
        WHILE first IS NOT None  
            SET first TO first.next  
            SET second TO second.next  
        SET second.next TO second.next.next  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
