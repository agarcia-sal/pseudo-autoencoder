CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS EMPTY OR LENGTH OF stones LESS THAN 2  
            RETURN False  
        
        SET stone_set TO set of stones  
        SET memo TO empty map  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS last element of stones  
                RETURN True  
            
            IF (position, jump) EXISTS IN memo  
                RETURN memo value for (position, jump)  
            
            FOR next_jump FROM jump MINUS 1 TO jump PLUS 1  
                IF next_jump GREATER THAN 0 AND position PLUS next_jump EXISTS IN stone_set  
                    IF CALL can_jump_to WITH position PLUS next_jump AND next_jump RETURNS True  
                        SET memo value for (position, jump) TO True  
                        RETURN True  
            
            SET memo value for (position, jump) TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN CALL can_jump_to WITH first element of stones AND 0  
    END FUNCTION  
END CLASS
