CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO LENGTH OF t  
                APPEND CONCATENATION OF ELEMENTS IN t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH i PLUS 1  
            IF CHARACTER AT t[i] IS ALPHABETIC  
                SET t[i] TO CHARACTER WITH ASCII VALUE EQUAL TO ASCII VALUE OF t[i] XORED WITH 32  
                CALL dfs WITH i PLUS 1  
            END IF  
        END FUNCTION  

        SET t TO LIST OF CHARACTERS IN s  
        SET ans TO EMPTY LIST  
        CALL dfs WITH 0  
        RETURN ans  
    END FUNCTION  
END CLASS
