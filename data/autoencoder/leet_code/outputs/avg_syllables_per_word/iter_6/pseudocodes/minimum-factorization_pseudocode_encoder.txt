CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num EQUALS 1  
            RETURN 1  
        END IF  
        
        SET factors TO empty list  
        
        FOR digit FROM 9 DOWN TO 2  
            WHILE num MODULO digit EQUALS 0  
                APPEND digit TO factors  
                SET num TO num DIVIDED BY digit  
            END WHILE  
        END FOR  
        
        IF num NOT EQUALS 1  
            RETURN 0  
        END IF  
        
        SORT factors IN ascending order  
        
        SET result TO concatenation of factors as a number  
        
        IF result GREATER THAN 2 MULTIPLIED BY 31st power MINUS 1  
            RETURN 0  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
