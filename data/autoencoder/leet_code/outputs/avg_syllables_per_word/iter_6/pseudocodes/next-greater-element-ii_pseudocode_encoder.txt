CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO LENGTH OF nums  
        SET result TO list of length n with all elements EQUAL TO -1  
        SET stack TO empty list  

        FOR i FROM 0 TO (2 MULTIPLIED BY n) MINUS 1  
            SET index TO i MODULO n  
            WHILE stack IS NOT empty AND nums VALUE AT stack LAST ELEMENT LESS THAN nums VALUE AT index  
                SET popped_index TO REMOVE last element FROM stack  
                SET result VALUE AT popped_index TO nums VALUE AT index  
            IF i LESS THAN n  
                APPEND index TO stack  

        RETURN result  
    END FUNCTION  
END CLASS
