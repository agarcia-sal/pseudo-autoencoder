CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the set of points converted to tuples from points  
        SET min_x TO the minimum x-coordinate value among unique_points  
        SET max_x TO the maximum x-coordinate value among unique_points  
        SET line_of_reflection TO min_x PLUS max_x DIVIDED BY 2  
        FOR each point WITH coordinates x AND y IN unique_points  
            SET reflected_x TO 2 MULTIPLIED BY line_of_reflection MINUS x  
            IF the point with coordinates reflected_x AND y IS NOT IN unique_points  
                RETURN False  
        RETURN True  
    END FUNCTION  
END CLASS
