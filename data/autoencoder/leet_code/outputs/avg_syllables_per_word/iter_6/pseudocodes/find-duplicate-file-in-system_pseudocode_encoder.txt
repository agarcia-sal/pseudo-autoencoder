CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO a new mapping with default empty list  
        
        FOR each path IN paths  
            SET parts TO result of splitting path by whitespace  
            SET directory TO first element of parts  
            
            FOR each file_info IN parts starting from second element  
                SET file_name, content TO result of splitting file_info by the first occurrence of '('  
                REMOVE the last character from content  
                
                APPEND concatenation of directory PLUS "/" PLUS file_name TO content_to_paths at key content  
        
        SET result TO empty list  
        FOR each group_of_paths IN values of content_to_paths  
            IF length of group_of_paths GREATER THAN 1  
                APPEND group_of_paths TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
