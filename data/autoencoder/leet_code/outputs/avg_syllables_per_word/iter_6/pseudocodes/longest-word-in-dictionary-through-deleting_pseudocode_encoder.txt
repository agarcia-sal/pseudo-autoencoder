CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator TO ITERATOR OVER s  
            FOR each character IN word  
                IF character IS NOT FOUND IN iterator  
                    RETURN FALSE  
            RETURN TRUE  
        END FUNCTION  
        
        CALL dictionary SORTED BY DESCENDING LENGTH AND ASCENDING LEXICOGRAPHICAL ORDER
        
        FOR each word IN dictionary  
            IF CALL is_subsequence WITH word AND s RETURNS TRUE  
                RETURN word  
        END FOR  
        
        RETURN empty string  
    END FUNCTION  
END CLASS
