CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO CALL TreeNode WITH values at index zero, None, None  
    SET i TO 1  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.left TO CALL TreeNode WITH values at index i, None, None  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values at index i IS NOT None  
            SET node.right TO CALL TreeNode WITH values at index i, None, None  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left, q.left AND CALL is_same_tree WITH p.right, q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree(root)  
        FUNCTION dfs(node)  
            IF node IS None  
                RETURN positive infinity, negative infinity, zero  
            SET left_min, left_max, left_size TO CALL dfs WITH node.left  
            SET right_min, right_max, right_size TO CALL dfs WITH node.right  
            IF left_max LESS THAN node.val AND node.val LESS THAN right_min  
                RETURN minimum of node.val AND left_min, maximum of node.val AND right_max, left_size PLUS right_size PLUS one  
            ELSE  
                RETURN negative infinity, positive infinity, maximum of left_size AND right_size  
            END IF  
        END FUNCTION  
        RETURN third element FROM CALL dfs WITH root  
    END FUNCTION  
END CLASS
