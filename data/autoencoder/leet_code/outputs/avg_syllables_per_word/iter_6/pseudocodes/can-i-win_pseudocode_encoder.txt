CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF (maxChoosableInteger MULTIPLIED BY (maxChoosableInteger PLUS 1)) DIVIDED BY 2 LESS THAN desiredTotal  
            RETURN False  
        END IF  
        
        IF desiredTotal LESS THAN OR EQUAL TO 0  
            RETURN True  
        END IF  
        
        SET memo TO empty dictionary  
        
        FUNCTION can_win(current_total, used_numbers)  
            IF pair (current_total, used_numbers) EXISTS IN memo  
                RETURN memo VALUE FOR pair (current_total, used_numbers)  
            END IF  
            
            FOR number FROM 1 TO maxChoosableInteger  
                IF number IS NOT MARKED AS USED IN used_numbers  
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT CALL can_win WITH (current_total PLUS number, used_numbers MARKED WITH number AS USED)  
                        SET memo VALUE FOR pair (current_total, used_numbers) TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET memo VALUE FOR pair (current_total, used_numbers) TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN CALL can_win WITH (0, 0)  
    END FUNCTION  
END CLASS
