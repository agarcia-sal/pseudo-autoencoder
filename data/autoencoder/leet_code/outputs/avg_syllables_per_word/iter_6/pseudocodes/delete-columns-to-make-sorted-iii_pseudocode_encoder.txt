CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO LENGTH OF strs  
        SET n TO LENGTH OF first element IN strs  
        SET dp TO list of size n with all elements set to 1  
        
        FOR i FROM 1 TO n MINUS 1  
            FOR j FROM 0 TO i MINUS 1  
                IF all characters at position j in each string are LESS THAN OR EQUAL TO characters at position i in the same string  
                    SET dp at i TO maximum of dp at i AND dp at j PLUS 1  
        
        RETURN n MINUS maximum value in dp  
    END FUNCTION  
END CLASS
