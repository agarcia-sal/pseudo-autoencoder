CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET i TO LENGTH OF nums MINUS 2  
        WHILE i GREATER THAN OR EQUAL TO 0 AND element AT nums i GREATER THAN OR EQUAL TO element AT nums i PLUS 1 IS TRUE  
            DECREMENT i BY 1  

        IF i GREATER THAN OR EQUAL TO 0  
            SET j TO LENGTH OF nums MINUS 1  
            WHILE element AT nums j LESS THAN OR EQUAL TO element AT nums i IS TRUE  
                DECREMENT j BY 1  
            SWAP elements AT nums i AND nums j  
        END IF  

        REVERSE elements IN nums FROM POSITION i PLUS 1 TO END  
    END FUNCTION  
END CLASS
