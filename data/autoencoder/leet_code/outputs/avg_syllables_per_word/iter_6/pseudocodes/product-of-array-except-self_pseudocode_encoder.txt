CLASS Solution  
    FUNCTION productExceptSelf(nums)  
        SET n TO LENGTH OF nums  
        SET answer TO list of size n filled with value 1  
        
        SET left_product TO 1  
        FOR i FROM 0 TO n MINUS 1  
            SET answer at position i TO left_product  
            SET left_product TO left_product MULTIPLIED BY nums at position i  
        
        SET right_product TO 1  
        FOR i FROM n MINUS 1 DOWN TO 0  
            SET answer at position i TO answer at position i MULTIPLIED BY right_product  
            SET right_product TO right_product MULTIPLIED BY nums at position i  
        
        RETURN answer  
    END FUNCTION  
END CLASS
