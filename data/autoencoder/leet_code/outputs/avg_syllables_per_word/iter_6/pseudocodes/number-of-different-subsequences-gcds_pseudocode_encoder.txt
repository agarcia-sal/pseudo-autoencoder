CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET max_num TO maximum value IN nums  
        SET num_set TO set containing elements FROM nums  
        SET count TO 0  
        
        FOR g FROM 1 TO max_num  
            SET gcd_val TO 0  
            
            FOR m FROM g TO max_num STEP g  
                IF m IS IN num_set THEN  
                    SET gcd_val TO greatest common divisor OF gcd_val AND m  
                    IF gcd_val EQUALS g THEN  
                        INCREMENT count BY 1  
                        BREAK  
            
        RETURN count  
    END FUNCTION  
END CLASS
