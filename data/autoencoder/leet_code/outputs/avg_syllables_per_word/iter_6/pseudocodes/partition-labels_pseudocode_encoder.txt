CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO empty dictionary  
        FOR index FROM 0 TO LENGTH OF s MINUS 1  
            SET character TO element AT index IN s  
            SET last_occurrence AT character TO index  
        SET start TO 0  
        SET end TO 0  
        SET partition_sizes TO empty list  
        FOR index FROM 0 TO LENGTH OF s MINUS 1  
            SET character TO element AT index IN s  
            IF last_occurrence AT character GREATER THAN end  
                SET end TO last_occurrence AT character  
            IF index EQUALS end  
                CALCULATE partition_size AS end MINUS start PLUS 1  
                APPEND partition_size TO partition_sizes  
                SET start TO index PLUS 1  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
