CLASS Solution  
    FUNCTION findMaximizedCapital(k, w, profits, capital)  
        SET projects TO zipped pairs of capital and profits  
        SORT projects BY capital  
        
        SET max_heap TO empty list  
        
        SET i TO 0  
        SET n TO length of projects  
        
        FOR each iteration FROM 1 TO k  
            WHILE i LESS THAN n AND project capital AT i LESS THAN OR EQUAL TO w IS TRUE  
                CALL push negative profit AT i INTO max_heap  
                INCREMENT i BY 1  
            
            IF max_heap IS empty THEN  
                BREAK loop  
            
            SET w TO w PLUS negative value popped from max_heap  
        
        RETURN w  
    END FUNCTION  
END CLASS
