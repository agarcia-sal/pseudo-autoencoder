CLASS Solution  
    FUNCTION minimumEffort(tasks)  
        CALL tasks SORTED BY difference BETWEEN second element AND first element IN descending order  
        SET current_energy TO 0  
        SET required_initial_energy TO 0  
        FOR each pair actual, minimum IN tasks  
            IF current_energy LESS THAN minimum  
                INCREMENT required_initial_energy BY difference BETWEEN minimum AND current_energy  
                SET current_energy TO minimum  
            END IF  
            DECREMENT current_energy BY actual  
        END FOR  
        RETURN required_initial_energy  
    END FUNCTION  
END CLASS
