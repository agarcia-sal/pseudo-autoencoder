CLASS Solution  
    FUNCTION maxJumps(arr, d)  
        FUNCTION dp(i)  
            SET max_jumps TO 1  
            FOR x FROM 1 TO d  
                IF i PLUS x IS LESS THAN LENGTH OF arr  
                    IF VALUE AT arr AT POSITION i PLUS x IS GREATER THAN OR EQUAL TO VALUE AT arr AT POSITION i  
                        BREAK LOOP  
                    END IF  
                    SET max_jumps TO MAXIMUM OF max_jumps AND ONE PLUS CALL dp WITH i PLUS x  
                END IF  
            END FOR  
            FOR x FROM 1 TO d  
                IF i MINUS x IS GREATER THAN OR EQUAL TO ZERO  
                    IF VALUE AT arr AT POSITION i MINUS x IS GREATER THAN OR EQUAL TO VALUE AT arr AT POSITION i  
                        BREAK LOOP  
                    END IF  
                    SET max_jumps TO MAXIMUM OF max_jumps AND ONE PLUS CALL dp WITH i MINUS x  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  

        SET max_result TO ZERO  
        FOR i FROM ZERO TO LENGTH OF arr MINUS ONE  
            SET current_jumps TO CALL dp WITH i  
            SET max_result TO MAXIMUM OF max_result AND current_jumps  
        END FOR  
        RETURN max_result  
    END FUNCTION  
END CLASS
