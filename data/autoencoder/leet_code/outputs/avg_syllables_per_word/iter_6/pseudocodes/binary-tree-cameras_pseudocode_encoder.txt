CLASS TreeNode  
    FUNCTION __init__(self, val DEFAULTS TO 0, left DEFAULTS TO None, right DEFAULTS TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH values AT INDEX 0  
    SET i TO 1  
    SET queue TO NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.left TO NEW TreeNode WITH values AT INDEX i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.right TO NEW TreeNode WITH values AT INDEX i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover(self, root)  
        FUNCTION dfs(node)  
            IF node IS None  
                RETURN 0, 0, positive infinity  
            SET left_no_camera, left_monitored, left_camera TO CALL dfs WITH node.left  
            SET right_no_camera, right_monitored, right_camera TO CALL dfs WITH node.right  
            SET no_camera TO left_monitored PLUS right_monitored  
            SET monitored TO MINIMUM OF  
                left_camera PLUS MINIMUM OF right_monitored AND right_camera,  
                right_camera PLUS MINIMUM OF left_monitored AND left_camera,  
                left_camera PLUS right_camera  
            SET camera TO 1 PLUS MINIMUM OF left_no_camera, left_monitored, left_camera PLUS MINIMUM OF right_no_camera, right_monitored, right_camera  
            RETURN no_camera, monitored, camera  
        END FUNCTION  
        SET root_no_camera, root_monitored, root_camera TO CALL dfs WITH root  
        RETURN MINIMUM OF root_monitored AND root_camera  
    END FUNCTION  
END CLASS
