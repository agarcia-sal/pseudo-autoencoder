CLASS Solution  
    FUNCTION uniquePaths(m, n)  
        CREATE two_dimensional_array dp WITH m ROWS AND n COLUMNS WHERE EACH ELEMENT IS SET TO 1  
        FOR i FROM 1 TO m MINUS 1  
            FOR j FROM 1 TO n MINUS 1  
                SET dp AT i AND j TO dp AT i MINUS 1 AND j PLUS dp AT i AND j MINUS 1  
        RETURN dp AT m MINUS 1 AND n MINUS 1  
    END FUNCTION  
END CLASS
