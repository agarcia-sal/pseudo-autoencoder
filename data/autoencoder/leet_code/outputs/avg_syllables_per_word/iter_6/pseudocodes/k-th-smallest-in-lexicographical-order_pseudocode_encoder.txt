CLASS Solution  
    FUNCTION findKthNumber(n, k)  
        FUNCTION count_steps(prefix, n)  
            SET next_prefix TO prefix PLUS 1  
            SET steps TO 0  
            WHILE prefix IS LESS THAN OR EQUAL TO n  
                SET steps TO steps PLUS MINIMUM OF (n MINUS prefix PLUS 1) AND (next_prefix MINUS prefix)  
                SET prefix TO prefix MULTIPLIED BY 10  
                SET next_prefix TO next_prefix MULTIPLIED BY 10  
            RETURN steps  
        END FUNCTION  

        SET current TO 1  
        DECREMENT k BY 1  
        WHILE k IS GREATER THAN 0  
            SET steps TO CALL count_steps WITH current AND n  
            IF steps IS LESS THAN OR EQUAL TO k  
                INCREMENT current BY 1  
                DECREMENT k BY steps  
            ELSE  
                SET current TO current MULTIPLIED BY 10  
                DECREMENT k BY 1  
        RETURN current  
    END FUNCTION  
END CLASS
