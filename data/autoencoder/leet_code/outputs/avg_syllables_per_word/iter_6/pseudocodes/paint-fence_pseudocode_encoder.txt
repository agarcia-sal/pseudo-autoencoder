CLASS Solution  
    FUNCTION numWays(n, k)  
        IF n EQUALS 0 THEN  
            RETURN 0  
        ELSE IF n EQUALS 1 THEN  
            RETURN k  
        ELSE IF n EQUALS 2 THEN  
            RETURN k MULTIPLIED BY k  
        
        SET same TO k  
        SET diff TO k MULTIPLIED BY (k MINUS 1)  
        SET total TO same PLUS diff  
        
        FOR i FROM 3 TO n DO  
            SET new_same TO diff  
            SET new_diff TO total MULTIPLIED BY (k MINUS 1)  
            SET same TO new_same  
            SET diff TO new_diff  
            SET total TO same PLUS diff  
        
        RETURN total  
    END FUNCTION  
END CLASS
