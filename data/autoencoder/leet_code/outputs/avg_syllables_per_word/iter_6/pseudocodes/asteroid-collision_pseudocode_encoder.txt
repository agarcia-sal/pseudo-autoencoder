CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO empty list  
        
        FOR each asteroid IN asteroids  
            WHILE stack IS NOT empty AND asteroid LESS THAN 0 AND last element in stack GREATER THAN 0  
                IF last element in stack LESS THAN negative of asteroid  
                    REMOVE last element FROM stack  
                    CONTINUE  
                ELSE IF last element in stack EQUALS negative of asteroid  
                    REMOVE last element FROM stack  
                BREAK  
            ELSE  
                APPEND asteroid TO stack  
        
        RETURN stack  
    END FUNCTION  
END CLASS
