CLASS Solution  
    FUNCTION topKFrequent(nums, k)  
        SET frequency_map TO mapping with default integer zero  
        FOR each num IN nums  
            INCREMENT frequency_map value for num BY 1  
        SET max_frequency TO maximum value in frequency_map values  
        SET buckets TO list of empty lists with length max_frequency PLUS one  
        FOR each num, freq IN frequency_map items  
            APPEND num TO buckets at index freq  
        SET result TO empty list  
        FOR freq FROM max_frequency DOWN TO 1  
            IF length of buckets at freq IS GREATER THAN 0  
                EXTEND result BY elements in buckets at freq  
                IF length of result IS GREATER THAN OR EQUAL TO k  
                    RETURN first k elements of result  
        RETURN result  
    END FUNCTION  
END CLASS
