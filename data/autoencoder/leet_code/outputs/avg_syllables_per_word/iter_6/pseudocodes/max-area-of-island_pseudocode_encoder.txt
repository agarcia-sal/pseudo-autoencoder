CLASS Solution  
    FUNCTION maxAreaOfIsland(grid)  
        IF grid IS EMPTY  
            RETURN 0  
        SET max_area TO 0  
        SET rows TO LENGTH OF grid  
        SET cols TO LENGTH OF grid AT 0  

        FUNCTION dfs(r, c)  
            IF r IS LESS THAN 0 OR r IS GREATER THAN OR EQUAL TO rows OR c IS LESS THAN 0 OR c IS GREATER THAN OR EQUAL TO cols OR grid AT r,c EQUALS 0  
                RETURN 0  
            SET grid AT r,c TO 0  
            SET area TO 1  
            INCREMENT area BY CALL dfs WITH r PLUS 1, c  
            INCREMENT area BY CALL dfs WITH r MINUS 1, c  
            INCREMENT area BY CALL dfs WITH r, c PLUS 1  
            INCREMENT area BY CALL dfs WITH r, c MINUS 1  
            RETURN area  
        END FUNCTION

        FOR r FROM 0 TO rows MINUS 1  
            FOR c FROM 0 TO cols MINUS 1  
                IF grid AT r,c EQUALS 1  
                    SET max_area TO MAXIMUM OF max_area AND CALL dfs WITH r, c  
        RETURN max_area  
    END FUNCTION  
END CLASS
