CLASS Solution  
    FUNCTION new21Game(n, k, maxPts)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k THEN  
                RETURN value of i LESS THAN OR EQUAL TO n converted to integer  
            ELSE IF i EQUALS k MINUS 1 THEN  
                RETURN minimum of (n MINUS k PLUS 1) AND maxPts DIVIDED BY maxPts  
            ELSE  
                RETURN dfs called WITH i PLUS 1 PLUS  
                       (dfs called WITH i PLUS 1 MINUS dfs called WITH i PLUS maxPts PLUS 1) DIVIDED BY maxPts  
            END IF  
        END FUNCTION  
        RETURN dfs called WITH 0  
    END FUNCTION  
END CLASS
