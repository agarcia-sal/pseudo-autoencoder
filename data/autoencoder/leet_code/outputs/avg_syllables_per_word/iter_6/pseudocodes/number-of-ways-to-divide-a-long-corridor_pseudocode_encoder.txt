CLASS Solution  
    FUNCTION numberOfWays(corridor)  
        SET MOD TO 1000000007  
        SET seats TO empty list  
        FOR index FROM 0 TO LENGTH OF corridor MINUS 1  
            IF character AT corridor WITH index EQUALS 'S'  
                APPEND index TO seats  
        IF LENGTH OF seats MODULO 2 NOT EQUALS 0 OR LENGTH OF seats LESS THAN 2  
            RETURN 0  
        SET ways TO 1  
        FOR i FROM 2 TO LENGTH OF seats MINUS 1 STEP 2  
            SET difference TO seats AT i MINUS seats AT i MINUS 1  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways MODULO MOD  
        RETURN ways  
    END FUNCTION  
END CLASS
