CLASS Solution  
    FUNCTION closestToTarget(arr, target)  
        SET seen TO empty set  
        SET min_diff TO positive infinity  
        
        FOR each num IN arr  
            SET new_seen TO empty set  
            FOR each x IN seen  
                COMPUTE bitwise AND of num AND x  
                ADD result TO new_seen  
            UNION new_seen WITH set containing num  
            SET seen TO new_seen  
            
            FOR each val IN seen  
                COMPUTE absolute difference BETWEEN val AND target  
                IF difference LESS THAN min_diff  
                    SET min_diff TO difference  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
