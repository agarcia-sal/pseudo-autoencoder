CLASS Solution  
    FUNCTION minSwap(nums1, nums2)  
        SET n TO LENGTH OF nums1  
        INITIALIZE dp AS list of n elements each containing list of two zeros  
        
        SET dp at index 0 position 0 TO 0  
        SET dp at index 0 position 1 TO 1  
        
        FOR i FROM 1 TO n MINUS 1  
            IF nums1 at i GREATER THAN nums1 at i MINUS 1 AND nums2 at i GREATER THAN nums2 at i MINUS 1 THEN  
                IF nums1 at i GREATER THAN nums2 at i MINUS 1 AND nums2 at i GREATER THAN nums1 at i MINUS 1 THEN  
                    SET dp at i position 0 TO minimum of dp at i MINUS 1 position 0 AND dp at i MINUS 1 position 1  
                    SET dp at i position 1 TO dp at i position 0 PLUS 1  
                ELSE  
                    SET dp at i position 0 TO dp at i MINUS 1 position 0  
                    SET dp at i position 1 TO dp at i MINUS 1 position 1 PLUS 1  
            ELSE  
                IF nums1 at i GREATER THAN nums2 at i MINUS 1 AND nums2 at i GREATER THAN nums1 at i MINUS 1 THEN  
                    SET dp at i position 0 TO dp at i MINUS 1 position 1  
                    SET dp at i position 1 TO dp at i MINUS 1 position 0 PLUS 1  
                ELSE  
                    SET dp at i position 0 TO dp at i MINUS 1 position 1  
                    SET dp at i position 1 TO dp at i MINUS 1 position 0 PLUS 1  
        
        RETURN minimum of dp at n MINUS 1 position 0 AND dp at n MINUS 1 position 1  
    END FUNCTION  
END CLASS
