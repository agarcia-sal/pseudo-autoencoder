CLASS Solution  
    FUNCTION maxIncreaseKeepingSkyline(grid)  
        SET n TO LENGTH OF grid  
        SET row_maxes TO EMPTY LIST  
        FOR each row IN grid  
            APPEND maximum value IN row TO row_maxes  
        SET col_maxes TO EMPTY LIST  
        FOR each column IN grid transposed  
            APPEND maximum value IN column TO col_maxes  
        SET total_increase TO 0  
        FOR i FROM 0 TO n MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                SET max_height TO minimum OF row_maxes AT i AND col_maxes AT j  
                IF max_height GREATER THAN grid AT i AND j  
                    INCREMENT total_increase BY max_height MINUS grid AT i AND j  
        RETURN total_increase  
    END FUNCTION  
END CLASS
