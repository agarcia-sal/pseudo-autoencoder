CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1, str2)  
        SET m TO LENGTH OF str1  
        SET n TO LENGTH OF str2  
        
        SET dp TO two-dimensional list of size (m PLUS 1) by (n PLUS 1) FILLED WITH 0  
        
        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF character at position (i MINUS 1) of str1 EQUALS character at position (j MINUS 1) of str2  
                    SET dp at position i,j TO dp at position (i MINUS 1),(j MINUS 1) PLUS 1  
                ELSE  
                    SET dp at position i,j TO maximum of dp at position (i MINUS 1),j AND dp at position i,(j MINUS 1)  
                
        SET i TO m  
        SET j TO n  
        SET scs TO empty list  
        
        WHILE i GREATER THAN 0 AND j GREATER THAN 0  
            IF character at position (i MINUS 1) of str1 EQUALS character at position (j MINUS 1) of str2  
                APPEND character at position (i MINUS 1) of str1 TO scs  
                DECREMENT i BY 1  
                DECREMENT j BY 1  
            ELSE IF dp at position (i MINUS 1),j GREATER THAN dp at position i,(j MINUS 1)  
                APPEND character at position (i MINUS 1) of str1 TO scs  
                DECREMENT i BY 1  
            ELSE  
                APPEND character at position (j MINUS 1) of str2 TO scs  
                DECREMENT j BY 1  
        
        WHILE i GREATER THAN 0  
            APPEND character at position (i MINUS 1) of str1 TO scs  
            DECREMENT i BY 1  
        
        WHILE j GREATER THAN 0  
            APPEND character at position (j MINUS 1) of str2 TO scs  
            DECREMENT j BY 1  
        
        REVERSE scs  
        
        SET result TO concatenation of all characters in scs  
        RETURN result  
    END FUNCTION  
END CLASS
