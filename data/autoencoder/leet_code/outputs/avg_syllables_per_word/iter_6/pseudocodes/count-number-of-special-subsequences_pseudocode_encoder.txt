CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET dp TO list containing three zeros  
        
        FOR each num IN nums  
            IF num EQUALS zero  
                SET dp at position zero TO remainder of (two MULTIPLIED BY dp at position zero PLUS one) divided by MOD  
            ELSE IF num EQUALS one  
                SET dp at position one TO remainder of (two MULTIPLIED BY dp at position one PLUS dp at position zero) divided by MOD  
            ELSE IF num EQUALS two  
                SET dp at position two TO remainder of (two MULTIPLIED BY dp at position two PLUS dp at position one) divided by MOD  
        RETURN dp at position two  
    END FUNCTION  
END CLASS
