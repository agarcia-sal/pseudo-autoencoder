CLASS Solution  
    FUNCTION smallestGoodBase(n)  
        SET n TO integer value of n  
        SET max_length TO integer part of logarithm base 2 of n PLUS 1  
        
        FOR m FROM max_length DOWN TO 2  
            SET k TO integer part of nth root of n with degree (m MINUS 1)  
            
            IF k LESS THAN 2  
                CONTINUE  
            
            SET num TO result of geometric series sum with base k and length m  
            
            IF num EQUALS n  
                RETURN string representation of k  
        
        RETURN string representation of n MINUS 1  
    END FUNCTION  
END CLASS
