CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO LENGTH OF mat1  
        SET k TO LENGTH OF mat1 AT FIRST ELEMENT  
        SET k TO LENGTH OF mat2  
        SET n TO LENGTH OF mat2 AT FIRST ELEMENT  
        
        CALL initialize_result_matrix WITH m AND n AND SET TO result  
        
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                FOR p FROM 0 TO k MINUS 1  
                    SET result AT i AT j TO result AT i AT j PLUS mat1 AT i AT p MULTIPLIED BY mat2 AT p AT j  
        RETURN result  
    END FUNCTION  
    
    FUNCTION initialize_result_matrix(m, n)  
        CREATE matrix WITH m ROWS AND n COLUMNS FILLED WITH ZEROES  
        RETURN matrix  
    END FUNCTION  
END CLASS
