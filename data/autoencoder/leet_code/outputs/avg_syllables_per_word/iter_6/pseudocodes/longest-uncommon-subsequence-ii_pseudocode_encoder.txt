CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET it TO an iterator over t  
            RETURN whether all characters in s appear in it in order  
        END FUNCTION  
        
        SORT strs BY descending length AND lexicographical order  
        
        FOR each index i FROM 0 TO length of strs MINUS 1  
            SET word TO strs at position i  
            SET unique TO true  
            FOR each index j FROM 0 TO length of strs MINUS 1  
                IF i NOT EQUALS j AND CALL is_subsequence WITH word AND strs at position j IS true  
                    SET unique TO false  
                    BREAK inner loop  
            IF unique IS true  
                RETURN length of word  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
