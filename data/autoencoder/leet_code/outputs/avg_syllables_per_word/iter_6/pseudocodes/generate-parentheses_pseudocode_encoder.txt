CLASS Solution  
    FUNCTION generateParenthesis(n)  
        FUNCTION backtrack(s, left, right)  
            IF LENGTH OF s EQUALS 2 MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s PLUS '(' AS s, left PLUS 1 AS left, right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s PLUS ')' AS s, left, right PLUS 1 AS right  
            END IF  
        END FUNCTION  

        SET result TO empty list  
        CALL backtrack WITH empty string AS s, 0 AS left, 0 AS right  
        RETURN result  
    END FUNCTION  
END CLASS
