CLASS Solution  
    FUNCTION integerReplacement(n)  
        SET count TO 0  
        WHILE n NOT EQUALS 1  
            IF n MODULO 2 EQUALS 0  
                SET n TO n DIVIDED BY 2  
            ELSE  
                IF n EQUALS 3  
                    SET n TO n MINUS 1  
                ELSE IF (n PLUS 1) MODULO 4 EQUALS 0  
                    SET n TO n PLUS 1  
                ELSE  
                    SET n TO n MINUS 1  
            INCREMENT count BY 1  
        RETURN count  
    END FUNCTION  
END CLASS
