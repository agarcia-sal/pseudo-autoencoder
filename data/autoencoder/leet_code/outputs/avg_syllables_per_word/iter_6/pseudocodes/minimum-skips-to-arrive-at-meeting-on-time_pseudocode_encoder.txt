CLASS Solution  
    FUNCTION minSkips(dist, speed, hoursBefore)  
        SET n TO LENGTH OF dist  
        INITIALIZE dp AS two-dimensional list WITH size n BY (n PLUS 1) FILLED WITH infinity  
        SET total_dist TO SUM OF dist  
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore  
            RETURN -1  
        
        SET dp AT index 0 AT 0 TO dist AT 0  
        FOR j FROM 1 TO n MINUS 1  
            SET dp AT index 0 AT j TO dist AT 0  
        
        FOR i FROM 1 TO n MINUS 1  
            FOR j FROM 0 TO i  
                IF j LESS THAN i  
                    SET dp AT i AT j TO CEILING OF (dp AT i MINUS 1 AT j DIVIDED BY speed) MULTIPLIED BY speed PLUS dist AT i  
                IF j GREATER THAN 0  
                    SET dp AT i AT j TO MINIMUM OF dp AT i AT j AND dp AT i MINUS 1 AT j MINUS 1 PLUS dist AT i  
        
        FOR j FROM 0 TO n MINUS 1  
            IF dp AT n MINUS 1 AT j LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed  
                RETURN j  
        
        RETURN -1  
    END FUNCTION  
END CLASS
