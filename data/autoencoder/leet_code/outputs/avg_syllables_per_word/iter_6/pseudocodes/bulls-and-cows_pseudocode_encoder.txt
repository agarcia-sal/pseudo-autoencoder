CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO 0  
        SET cows TO 0  
        CALL initialize secret_count AS dictionary with default integer zero  
        CALL initialize guess_count AS dictionary with default integer zero  

        FOR each pair s, g IN secret and guess simultaneously  
            IF s EQUALS g  
                INCREMENT bulls BY 1  
            ELSE  
                INCREMENT secret_count AT key s BY 1  
                INCREMENT guess_count AT key g BY 1  
        
        FOR each digit IN keys of secret_count  
            INCREMENT cows BY the minimum of secret_count AT digit AND guess_count AT digit  
        
        RETURN string formed by concatenating bulls, "A", cows, "B"  
    END FUNCTION  
END CLASS
