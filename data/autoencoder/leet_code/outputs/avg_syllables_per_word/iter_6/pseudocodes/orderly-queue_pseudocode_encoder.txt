CLASS Solution  
    FUNCTION orderlyQueue(s, k)  
        IF k GREATER THAN 1  
            RETURN string formed by sorting characters of s in ascending order  
        ELSE  
            SET n TO length of s  
            SET smallest TO s  
            FOR i FROM 1 TO n MINUS 1  
                SET rotated TO substring of s from index i TO end CONCATENATED WITH substring of s from start TO index i MINUS 1  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
            RETURN smallest  
    END FUNCTION  
END CLASS
