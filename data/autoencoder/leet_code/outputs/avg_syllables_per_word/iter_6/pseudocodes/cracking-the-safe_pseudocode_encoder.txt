CLASS Solution  
    FUNCTION crackSafe(n, k)  
        SET visited TO empty set  
        SET sequence TO empty list  
        
        FUNCTION dfs(node)  
            FOR i FROM 0 TO k MINUS 1  
                SET edge TO concatenation of node AND string representation of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH argument substring of edge FROM second character TO end  
                    APPEND string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO string of '0' REPEATED n MINUS 1 times  
        CALL dfs WITH argument start_node  
        RETURN concatenation of all elements of sequence IN REVERSE ORDER PLUS start_node  
    END FUNCTION  
END CLASS
