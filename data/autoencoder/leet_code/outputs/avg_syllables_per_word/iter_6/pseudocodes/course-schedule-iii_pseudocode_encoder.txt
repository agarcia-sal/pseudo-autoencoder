CLASS Solution  
    FUNCTION scheduleCourse(courses)  
        CALL courses SORTED BY second element OF each course IN ASCENDING ORDER  
        SET max_heap TO empty list  
        SET current_end_day TO 0  
        
        FOR each course IN courses  
            SET duration TO first element OF course  
            SET last_day TO second element OF course  
            
            IF current_end_day PLUS duration LESS THAN OR EQUAL TO last_day THEN  
                CALL insert negative duration INTO max_heap USING heap push operation  
                INCREMENT current_end_day BY duration  
            ELSE IF max_heap IS NOT empty AND negative of first element OF max_heap GREATER THAN duration THEN  
                INCREMENT current_end_day BY duration PLUS value obtained by removing largest duration from max_heap  
                CALL insert negative duration INTO max_heap USING heap push operation  
            END IF  
        END FOR  
        
        RETURN size OF max_heap  
    END FUNCTION  
END CLASS
