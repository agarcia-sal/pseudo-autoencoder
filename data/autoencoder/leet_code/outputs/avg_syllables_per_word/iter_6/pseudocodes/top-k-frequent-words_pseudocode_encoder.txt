CLASS Solution  
    FUNCTION topKFrequent(words, k)  
        SET count TO CALL Counter WITH words  
        SET heap TO empty list  
        FOR each word AND freq IN count items  
            APPEND tuple of negative freq AND word TO heap  
        CALL heapify ON heap  
        SET result TO empty list  
        FOR index FROM 1 TO k  
            SET top_element TO CALL heappop ON heap  
            APPEND second element OF top_element TO result  
        RETURN result  
    END FUNCTION  
END CLASS
