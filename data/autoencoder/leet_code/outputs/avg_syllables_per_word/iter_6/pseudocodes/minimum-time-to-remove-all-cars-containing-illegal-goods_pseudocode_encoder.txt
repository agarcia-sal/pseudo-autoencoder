CLASS Solution  
    FUNCTION minimumTime(s)  
        SET n TO LENGTH OF s  
        SET left_costs TO list of zeros WITH LENGTH n  
        SET right_costs TO list of zeros WITH LENGTH n  

        IF s AT POSITION 0 EQUALS '1' THEN  
            SET left_costs AT POSITION 0 TO 1  
        ELSE  
            SET left_costs AT POSITION 0 TO 0  
        END IF  

        FOR i FROM 1 TO n MINUS 1  
            IF s AT POSITION i EQUALS '0' THEN  
                SET left_costs AT POSITION i TO left_costs AT POSITION i MINUS 1  
            ELSE  
                SET left_costs AT POSITION i TO MINIMUM OF (i PLUS 1) AND (left_costs AT POSITION i MINUS 1 PLUS 2)  
            END IF  
        END FOR  

        IF s AT POSITION n MINUS 1 EQUALS '1' THEN  
            SET right_costs AT POSITION n MINUS 1 TO 1  
        ELSE  
            SET right_costs AT POSITION n MINUS 1 TO 0  
        END IF  

        FOR i FROM n MINUS 2 DOWN TO 0  
            IF s AT POSITION i EQUALS '0' THEN  
                SET right_costs AT POSITION i TO right_costs AT POSITION i PLUS 1  
            ELSE  
                SET right_costs AT POSITION i TO MINIMUM OF (n MINUS i) AND (right_costs AT POSITION i PLUS 1 PLUS 2)  
            END IF  
        END FOR  

        SET min_time TO MINIMUM OF left_costs AT POSITION n MINUS 1 AND right_costs AT POSITION 0  

        FOR i FROM 0 TO n MINUS 2  
            SET min_time TO MINIMUM OF min_time AND (left_costs AT POSITION i PLUS right_costs AT POSITION i PLUS 1)  
        END FOR  

        RETURN min_time  
    END FUNCTION  
END CLASS
