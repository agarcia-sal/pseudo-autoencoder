CLASS Solution  
    FUNCTION jobScheduling(startTime, endTime, profit)  
        SET jobs TO list of tuples combining startTime, endTime, and profit SORTED BY each tuple's second element  
        SET end_times TO list of second elements FROM each tuple IN jobs  
        SET dp TO list of zeros WITH length EQUAL TO length of jobs PLUS one  

        FOR index FROM 1 TO length of jobs  
            SET j TO result of binary search for rightmost position TO insert jobs at index minus one’s first element INTO end_times  
            SET dp at index TO maximum of dp at index minus one AND dp at j PLUS jobs at index minus one’s third element  
        RETURN last element in dp  
    END FUNCTION  
END CLASS
