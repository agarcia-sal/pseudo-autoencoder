CLASS Solution  
    FUNCTION minInterval(intervals, queries)  
        CALL intervals SORTED BY start time  
        SET sorted_queries TO list of tuples (query_value, original_index) SORTED by query_value  
        
        SET min_heap TO empty heap  
        SET result TO list of length equal to queries filled with -1  
        
        SET interval_index TO 0  
        FOR each query, original_index IN sorted_queries  
            WHILE interval_index LESS THAN length of intervals AND intervals[interval_index] start LESS THAN OR EQUAL TO query  
                SET start, end TO intervals[interval_index] values  
                SET interval_size TO end MINUS start PLUS 1  
                ADD tuple (interval_size, end) TO min_heap  
                INCREMENT interval_index BY 1  
            
            WHILE min_heap IS NOT empty AND smallest end value in min_heap LESS THAN query  
                REMOVE smallest element FROM min_heap  
            
            IF min_heap IS NOT empty  
                SET result at original_index TO smallest interval_size in min_heap  
        
        RETURN result  
    END FUNCTION  
END CLASS
