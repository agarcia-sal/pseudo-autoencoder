CLASS Solution  
    FUNCTION numDistinct(s, t)  
        SET m TO LENGTH OF s  
        SET n TO LENGTH OF t  
        
        CALL dp INITIALIZE AS 2D ARRAY OF SIZE (m PLUS 1) BY (n PLUS 1) FILLED WITH ZEROES  
        
        FOR i FROM 0 TO m  
            SET dp AT i, 0 TO 1  
        
        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF CHARACTER AT s AT i MINUS 1 EQUALS CHARACTER AT t AT j MINUS 1  
                    SET dp AT i, j TO dp AT i MINUS 1, j MINUS 1 PLUS dp AT i MINUS 1, j  
                ELSE  
                    SET dp AT i, j TO dp AT i MINUS 1, j  
        
        RETURN dp AT m, n  
    END FUNCTION  
END CLASS
