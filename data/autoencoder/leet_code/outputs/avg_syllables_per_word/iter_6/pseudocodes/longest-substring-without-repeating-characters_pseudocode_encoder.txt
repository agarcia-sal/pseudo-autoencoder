CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO empty mapping  
        SET max_length TO 0  
        SET start TO 0  
        
        FOR each index end AND character char IN s  
            IF char EXISTS IN char_index_map AND char_index_map[char] GREATER THAN OR EQUAL TO start  
                SET start TO char_index_map[char] PLUS 1  
            END IF  
            SET char_index_map[char] TO end  
            SET max_length TO the greater value BETWEEN max_length AND end MINUS start PLUS 1  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
