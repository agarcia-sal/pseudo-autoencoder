CLASS Solution  
    FUNCTION minSubArrayLen(target, nums)  
        SET left TO 0  
        SET current_sum TO 0  
        SET min_length TO infinity  
        
        FOR right FROM 0 TO LENGTH OF nums MINUS 1  
            INCREMENT current_sum BY element AT index right IN nums  
            
            WHILE current_sum GREATER THAN OR EQUAL TO target  
                SET min_length TO minimum OF min_length AND right MINUS left PLUS 1  
                DECREMENT current_sum BY element AT index left IN nums  
                INCREMENT left BY 1  
        
        IF min_length NOT EQUALS infinity  
            RETURN min_length  
        ELSE  
            RETURN 0  
    END FUNCTION  
END CLASS
