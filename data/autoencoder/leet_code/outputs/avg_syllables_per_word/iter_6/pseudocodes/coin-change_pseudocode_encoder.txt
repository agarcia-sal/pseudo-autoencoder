CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        SET dp TO new list of length amount PLUS 1 with all values SET TO amount PLUS 1  
        SET dp at index 0 TO 0  
        
        FOR each a FROM 1 TO amount  
            FOR each coin IN coins  
                IF a MINUS coin GREATER THAN OR EQUAL TO 0 THEN  
                    SET dp at index a TO the minimum BETWEEN dp at index a AND dp at index (a MINUS coin) PLUS 1  
        
        IF dp at index amount NOT EQUALS amount PLUS 1 THEN  
            RETURN dp at index amount  
        ELSE  
            RETURN negative 1  
    END FUNCTION  
END CLASS
