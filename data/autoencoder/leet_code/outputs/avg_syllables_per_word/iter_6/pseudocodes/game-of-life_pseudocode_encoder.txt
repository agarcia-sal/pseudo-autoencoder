CLASS Solution  
    FUNCTION gameOfLife(board)  
        IF board IS EMPTY OR first element of board IS EMPTY  
            RETURN  
        
        SET m TO LENGTH OF board  
        SET n TO LENGTH OF first element of board  
        
        SET directions TO list of eight neighbor coordinate offsets  
        
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                SET live_neighbors TO 0  
                FOR each dx, dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni GREATER THAN OR EQUAL TO 0 AND ni LESS THAN m AND nj GREATER THAN OR EQUAL TO 0 AND nj LESS THAN n  
                        IF board element at ni, nj EQUALS 1 OR board element at ni, nj EQUALS 3  
                            INCREMENT live_neighbors BY 1  
                IF board element at i, j EQUALS 1  
                    IF live_neighbors LESS THAN 2 OR live_neighbors GREATER THAN 3  
                        SET board element at i, j TO 3  
                ELSE  
                    IF live_neighbors EQUALS 3  
                        SET board element at i, j TO 2  
        
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF board element at i, j EQUALS 2  
                    SET board element at i, j TO 1  
                ELSE IF board element at i, j EQUALS 3  
                    SET board element at i, j TO 0  
    END FUNCTION  
END CLASS
