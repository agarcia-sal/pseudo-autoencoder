CLASS Solution  
    FUNCTION merge(intervals)  
        CALL sort intervals BY starting point  
        SET merged TO empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR ending point of last element in merged LESS THAN starting point of interval  
                APPEND interval TO merged  
            ELSE  
                SET ending point of last element in merged TO maximum of ending point of last element in merged AND ending point of interval  
        RETURN merged  
    END FUNCTION  
END CLASS
