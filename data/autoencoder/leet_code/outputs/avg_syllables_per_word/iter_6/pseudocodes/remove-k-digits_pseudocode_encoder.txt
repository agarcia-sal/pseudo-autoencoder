CLASS Solution  
    FUNCTION removeKdigits(num, k)  
        SET stack TO empty list  
        
        FOR each digit IN num  
            WHILE k GREATER THAN 0 AND stack IS NOT empty AND last element of stack GREATER THAN digit  
                REMOVE last element FROM stack  
                DECREMENT k BY 1  
            APPEND digit TO stack  
        
        IF k GREATER THAN 0  
            SET final_stack TO all elements of stack EXCLUDING last k elements  
        ELSE  
            SET final_stack TO stack  
        
        SET result TO string formed by concatenating all elements of final_stack  
        REMOVE leading zero characters FROM result  
        
        IF result IS empty  
            RETURN "0"  
        ELSE  
            RETURN result  
    END FUNCTION  
END CLASS
