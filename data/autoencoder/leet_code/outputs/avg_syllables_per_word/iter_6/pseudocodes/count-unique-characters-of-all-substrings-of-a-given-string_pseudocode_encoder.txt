CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO one billion plus seven  
        SET n TO LENGTH OF s  
        SET last TO list of negative one repeated twenty-six times  
        SET last2 TO list of negative one repeated twenty-six times  
        SET result TO zero  
        
        FOR each index i and character char IN s  
            SET idx TO ordinal value of char MINUS ordinal value of 'A'  
            SET contribution TO (i MINUS last at idx) MULTIPLIED BY (last at idx MINUS last2 at idx)  
            SET result TO (result PLUS contribution) MODULO MOD  
            SET last2 at idx TO last at idx  
            SET last at idx TO i  
        
        FOR each index i FROM zero TO twenty-five  
            SET contribution TO (n MINUS last at i) MULTIPLIED BY (last at i MINUS last2 at i)  
            SET result TO (result PLUS contribution) MODULO MOD  
        
        RETURN result  
    END FUNCTION  
END CLASS
