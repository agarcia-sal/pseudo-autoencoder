CLASS TreeNode  
    FUNCTION __init__(val EQUALS 0, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH val EQUALS values AT INDEX 0  
    SET i TO 1  
    SET queue TO NEW empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.left TO NEW TreeNode WITH val EQUALS values AT INDEX i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.right TO NEW TreeNode WITH val EQUALS values AT INDEX i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers(root)  
        FUNCTION dfs(node, current_number)  
            IF node IS None  
                RETURN 0  
            SET current_number TO current_number MULTIPLIED BY 10 PLUS node.val  
            IF node.left IS None AND node.right IS None  
                RETURN current_number  
            SET left_sum TO CALL dfs WITH node.left AND current_number  
            SET right_sum TO CALL dfs WITH node.right AND current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN CALL dfs WITH root AND 0  
    END FUNCTION  
END CLASS
