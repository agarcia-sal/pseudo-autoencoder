CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            RETURN (a at index 0 MINUS o at index 0) MULTIPLIED BY (b at index 1 MINUS o at index 1) MINUS (a at index 1 MINUS o at index 1) MULTIPLIED BY (b at index 0 MINUS o at index 0)  
        END FUNCTION  
        
        SET n TO LENGTH OF points  
        IF n LESS THAN OR EQUAL TO 3 THEN  
            RETURN True  
        END IF  
        
        SET orientation TO 0  
        
        FOR i FROM 0 TO n MINUS 1  
            SET o TO points at index i  
            SET a TO points at index ((i PLUS 1) MODULO n)  
            SET b TO points at index ((i PLUS 2) MODULO n)  
            SET cross TO CALL cross_product WITH o, a, b  
            
            IF cross NOT EQUALS 0 THEN  
                IF orientation EQUALS 0 THEN  
                    IF cross GREATER THAN 0 THEN  
                        SET orientation TO 1  
                    ELSE  
                        SET orientation TO MINUS 1  
                    END IF  
                ELSE IF (orientation EQUALS 1 AND cross LESS THAN 0) OR (orientation EQUALS MINUS 1 AND cross GREATER THAN 0) THEN  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
