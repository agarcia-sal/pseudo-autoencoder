CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO LENGTH OF nums1  
        SET n TO LENGTH OF nums2  

        INITIALIZE dp AS TWO-DIMENSIONAL LIST OF SIZE (m PLUS 1) BY (n PLUS 1)  
        SET EACH ELEMENT OF dp TO NEGATIVE INFINITY  

        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                SET current_product TO ELEMENT AT INDEX (i MINUS 1) OF nums1 MULTIPLIED BY ELEMENT AT INDEX (j MINUS 1) OF nums2  

                SET dp ELEMENT AT i,j TO MAXIMUM OF  
                    current_product,  
                    dp ELEMENT AT (i MINUS 1), j,  
                    dp ELEMENT AT i, (j MINUS 1),  
                    dp ELEMENT AT (i MINUS 1), (j MINUS 1) PLUS current_product  

        RETURN dp ELEMENT AT m,n  
    END FUNCTION  
END CLASS
