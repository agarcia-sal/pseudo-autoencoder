CLASS Solution  
    FUNCTION kIncreasing(arr, k)  
        FUNCTION longest_increasing_subsequence(subarr)  
            SET lis TO empty list  
            FOR each num IN subarr  
                IF lis IS empty OR num GREATER THAN OR EQUAL TO last element IN lis  
                    APPEND num TO lis  
                ELSE  
                    SET pos TO the position where num can be inserted on the right to keep lis sorted  
                    SET element at pos IN lis TO num  
            RETURN length of lis  
        END FUNCTION  
        
        SET total_operations TO 0  
        SET n TO length of arr  
        
        FOR start FROM 0 TO k MINUS 1  
            SET subarr TO empty list  
            FOR i FROM start TO n MINUS 1 STEP k  
                APPEND element at i IN arr TO subarr  
            
            SET lis_length TO CALL longest_increasing_subsequence WITH subarr  
            INCREMENT total_operations BY length of subarr MINUS lis_length  
        
        RETURN total_operations  
    END FUNCTION  
END CLASS
