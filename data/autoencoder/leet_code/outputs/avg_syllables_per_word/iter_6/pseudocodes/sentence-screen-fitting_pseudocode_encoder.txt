CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO concatenation of sentence words separated by a space PLUS a trailing space  
        SET total_length TO length of s  
        SET position TO 0  
        
        FOR count FROM 1 TO rows  
            INCREMENT position BY cols  
            
            IF character at position modulo total_length in s EQUALS space  
                INCREMENT position BY 1  
            ELSE  
                WHILE position GREATER THAN 0 AND character at (position MINUS 1) modulo total_length in s NOT EQUALS space  
                    DECREMENT position BY 1  
            
        RETURN position DIVIDED BY total_length  
    END FUNCTION  
END CLASS
