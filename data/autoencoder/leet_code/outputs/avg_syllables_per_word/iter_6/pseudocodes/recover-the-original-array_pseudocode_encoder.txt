CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL nums SORTED IN ASCENDING ORDER  
        SET n TO LENGTH OF nums  
        
        SET first TO nums AT INDEX 0  
        
        FOR index FROM 1 TO n MINUS 1  
            SET k TO (nums AT index MINUS first) DIVIDED BY 2  
            IF k LESS THAN OR EQUAL TO 0 OR first PLUS 2 MULTIPLIED BY k NOT EQUALS nums AT index THEN  
                CONTINUE  
            END IF  
            
            SET count TO CALL Counter WITH nums  
            SET arr TO EMPTY LIST  
            
            FOR each x IN nums  
                IF count AT x EQUALS 0 THEN  
                    CONTINUE  
                END IF  
                IF count AT (x PLUS 2 MULTIPLIED BY k) EQUALS 0 THEN  
                    BREAK  
                END IF  
                APPEND (x PLUS k) TO arr  
                DECREMENT count AT x BY 1  
                DECREMENT count AT (x PLUS 2 MULTIPLIED BY k) BY 1  
            END FOR  
            
            IF LENGTH OF arr EQUALS n DIVIDED BY 2 THEN  
                RETURN arr  
            END IF  
        END FOR  
        
        RETURN EMPTY LIST  
    END FUNCTION  
END CLASS
