CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO list of size 26 with all values set TO -1  
        SET total_appeal TO 0  
        SET current_appeal TO 0  
        
        FOR i FROM 0 TO LENGTH OF s MINUS 1  
            SET char TO character at position i in s  
            SET index TO numeric position of char in alphabet starting from 0  
            INCREMENT current_appeal BY i MINUS value at last position index  
            INCREMENT total_appeal BY current_appeal  
            SET value at last position index TO i  
        
        RETURN total_appeal  
    END FUNCTION  
END CLASS
