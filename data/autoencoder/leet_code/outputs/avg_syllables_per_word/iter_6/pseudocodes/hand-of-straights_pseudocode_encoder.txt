CLASS Solution  
    FUNCTION isNStraightHand(hand, groupSize)  
        IF LENGTH OF hand MODULO groupSize NOT EQUALS 0  
            RETURN False  
        
        SET count TO frequency mapping of elements in hand  
        SET unique_cards TO sorted list of keys in count  
        
        FOR each card IN unique_cards  
            IF count of card GREATER THAN 0  
                SET needed TO count of card  
                FOR index FROM 0 TO groupSize MINUS 1  
                    IF count of card PLUS index LESS THAN needed  
                        RETURN False  
                    DECREMENT count of card PLUS index BY needed  
        
        RETURN True  
    END FUNCTION  
END CLASS
