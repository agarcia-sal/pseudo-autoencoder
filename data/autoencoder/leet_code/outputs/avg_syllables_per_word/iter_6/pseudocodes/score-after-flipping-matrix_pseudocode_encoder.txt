CLASS Solution  
    FUNCTION matrixScore(grid)  
        FOR each row IN grid  
            IF row AT index 0 EQUALS 0  
                FOR index FROM 0 TO LENGTH OF row MINUS 1  
                    SET row AT index TO 1 MINUS row AT index  
        
        FOR col FROM 1 TO LENGTH OF grid AT 0 MINUS 1  
            SET count_ones TO 0  
            FOR row_index FROM 0 TO LENGTH OF grid MINUS 1  
                IF grid AT row_index AT col EQUALS 1  
                    INCREMENT count_ones BY 1  
            IF count_ones LESS THAN LENGTH OF grid DIVIDED BY 2  
                FOR row_index FROM 0 TO LENGTH OF grid MINUS 1  
                    SET grid AT row_index AT col TO 1 MINUS grid AT row_index AT col  
        
        SET score TO 0  
        FOR each row IN grid  
            SET binary_number TO concatenation of elements of row converted to string  
            CONVERT binary_number TO integer_value as base 2  
            INCREMENT score BY integer_value  
        
        RETURN score  
    END FUNCTION  
END CLASS
