CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS 0, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH val EQUALS first element of values  
    SET i TO 1  
    INITIALIZE queue AS deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN length of values AND values at i IS NOT None  
            SET node.left TO NEW TreeNode WITH val EQUALS values at i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values at i IS NOT None  
            SET node.right TO NEW TreeNode WITH val EQUALS values at i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left, q.left AND CALL is_same_tree WITH p.right, q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self, root)  
        IF root IS None  
            RETURN empty list  
        SET result TO empty list  
        INITIALIZE queue AS deque WITH root  
        SET left_to_right TO True  
        WHILE queue IS NOT EMPTY  
            SET level_size TO length of queue  
            INITIALIZE current_level AS empty deque  
            FOR _ FROM 1 TO level_size  
                SET node TO REMOVE first element FROM queue  
                IF left_to_right IS True  
                    APPEND node.val TO current_level  
                ELSE  
                    APPEND node.val TO FRONT OF current_level  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
            END FOR  
            APPEND list form of current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
