CLASS Solution  
    FUNCTION rangeBitwiseAnd(left, right)  
        SET shift TO 0  
        WHILE left IS LESS THAN right  
            SET left TO left DIVIDED BY 2  
            SET right TO right DIVIDED BY 2  
            INCREMENT shift BY 1  
        SET result TO left MULTIPLIED BY 2 RAISED TO THE POWER OF shift  
        RETURN result  
    END FUNCTION  
END CLASS
