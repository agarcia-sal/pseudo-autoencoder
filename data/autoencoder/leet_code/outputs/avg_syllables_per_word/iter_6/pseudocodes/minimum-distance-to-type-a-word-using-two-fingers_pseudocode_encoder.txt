CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS None THEN  
                RETURN 0  
            SET x1 TO integer division of (ordinal value of c1 MINUS ordinal value of 'A') DIVIDED BY 6  
            SET y1 TO remainder of (ordinal value of c1 MINUS ordinal value of 'A') DIVIDED BY 6  
            SET x2 TO integer division of (ordinal value of c2 MINUS ordinal value of 'A') DIVIDED BY 6  
            SET y2 TO remainder of (ordinal value of c2 MINUS ordinal value of 'A') DIVIDED BY 6  
            RETURN absolute value of (x1 MINUS x2) PLUS absolute value of (y1 MINUS y2)  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2)  
            IF i EQUALS length of word THEN  
                RETURN 0  
            SET c TO character at position i in word  
            SET d1 TO CALL distance WITH arguments f1, c PLUS CALL dp WITH arguments i PLUS 1, c, f2  
            SET d2 TO CALL distance WITH arguments f2, c PLUS CALL dp WITH arguments i PLUS 1, f1, c  
            RETURN minimum of d1 AND d2  
        END FUNCTION  
        
        RETURN CALL dp WITH arguments 0, None, None  
    END FUNCTION  
END CLASS
