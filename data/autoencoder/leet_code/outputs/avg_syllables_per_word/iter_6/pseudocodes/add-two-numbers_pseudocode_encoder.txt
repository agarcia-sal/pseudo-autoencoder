CLASS ListNode  
    FUNCTION __init__(val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET head TO CALL ListNode WITH values FIRST ELEMENT AND None  
    SET p TO head  
    FOR each val IN values FROM SECOND ELEMENT TO END  
        SET node TO CALL ListNode WITH val AND None  
        SET p.next TO node  
        SET p TO node  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN p1.val EQUALS p2.val AND CALL is_same_list WITH p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1, l2)  
        SET dummy_head TO CALL ListNode WITH 0 AND None  
        SET current TO dummy_head  
        SET carry TO 0  
        
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry GREATER THAN 0  
            IF l1 IS NOT None  
                SET val1 TO l1.val  
            ELSE  
                SET val1 TO 0  
            IF l2 IS NOT None  
                SET val2 TO l2.val  
            ELSE  
                SET val2 TO 0  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY 10  
            SET new_node_value TO total MODULO 10  
            SET current.next TO CALL ListNode WITH new_node_value AND None  
            SET current TO current.next  
            
            IF l1 IS NOT None  
                SET l1 TO l1.next  
            IF l2 IS NOT None  
                SET l2 TO l2.next  
        
        RETURN dummy_head.next  
    END FUNCTION  
END CLASS
