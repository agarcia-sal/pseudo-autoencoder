CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN 0  
            SET sign TO -1  
        ELSE  
            SET sign TO 1  
        SET absolute_string TO string representation of absolute value of x  
        SET reversed_string TO reversal of absolute_string  
        SET reversed_x TO integer value of reversed_string  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative two POWER thirty-one OR result GREATER THAN two POWER thirty-one MINUS one  
            RETURN 0  
        RETURN result  
    END FUNCTION  
END CLASS
