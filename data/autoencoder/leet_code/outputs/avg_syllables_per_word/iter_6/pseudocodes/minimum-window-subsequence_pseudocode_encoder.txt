CLASS Solution  
    FUNCTION minWindow(s1, s2)  
        SET min_len TO infinity  
        SET min_start TO negative one  
        SET len1 TO LENGTH OF s1  
        SET len2 TO LENGTH OF s2  
        
        FOR i FROM zero TO len1 MINUS one  
            SET j TO zero  
            FOR k FROM i TO len1 MINUS one  
                IF CHARACTER AT s1 POSITION k EQUALS CHARACTER AT s2 POSITION j THEN  
                    INCREMENT j BY one  
                    IF j EQUALS len2 THEN  
                        IF k MINUS i PLUS one LESS THAN min_len THEN  
                            SET min_len TO k MINUS i PLUS one  
                            SET min_start TO i  
                        END IF  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF min_start EQUALS negative one THEN  
            RETURN empty string  
        ELSE  
            RETURN SUBSTRING OF s1 FROM min_start TO min_start PLUS min_len MINUS one  
        END IF  
    END FUNCTION  
END CLASS
