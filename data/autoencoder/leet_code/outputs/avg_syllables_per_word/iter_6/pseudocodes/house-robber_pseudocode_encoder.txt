CLASS Solution  
    FUNCTION rob(nums)  
        IF nums IS EMPTY  
            RETURN 0  
        ELSE IF LENGTH OF nums EQUALS 1  
            RETURN nums AT 0  
        END IF  

        SET dp TO list of zeros WITH LENGTH OF nums  
        SET dp AT 0 TO nums AT 0  
        SET dp AT 1 TO MAXIMUM OF nums AT 0 AND nums AT 1  

        FOR index FROM 2 TO LENGTH OF nums MINUS 1  
            SET dp AT index TO MAXIMUM OF dp AT index MINUS 1 AND dp AT index MINUS 2 PLUS nums AT index  
        END FOR  

        RETURN dp AT LAST INDEX  
    END FUNCTION  
END CLASS
