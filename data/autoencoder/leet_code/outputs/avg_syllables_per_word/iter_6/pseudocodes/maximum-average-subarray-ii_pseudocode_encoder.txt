CLASS Solution  
    FUNCTION findMaxAverage(nums, k)  
        FUNCTION canFindAverage(mid)  
            SET prefix_sums TO list of zeros with length LENGTH OF nums PLUS 1  
            SET min_prefix_sum TO list of zeros with length LENGTH OF nums PLUS 1  
            
            FOR i FROM 0 TO LENGTH OF nums MINUS 1  
                SET prefix_sums AT INDEX i PLUS 1 TO prefix_sums AT INDEX i PLUS nums AT INDEX i MINUS mid  
                
                IF i GREATER THAN OR EQUAL TO k THEN  
                    SET min_prefix_sum AT INDEX i PLUS 1 TO MINIMUM OF min_prefix_sum AT INDEX i AND prefix_sums AT INDEX i PLUS 1 MINUS k  
                ELSE  
                    SET min_prefix_sum AT INDEX i PLUS 1 TO min_prefix_sum AT INDEX i  
                
                IF i GREATER THAN OR EQUAL TO k MINUS 1 AND prefix_sums AT INDEX i PLUS 1 MINUS min_prefix_sum AT INDEX i PLUS 1 GREATER THAN OR EQUAL TO 0 THEN  
                    RETURN True  
            RETURN False  
        END FUNCTION  
        
        SET low TO MINIMUM VALUE IN nums  
        SET high TO MAXIMUM VALUE IN nums  
        
        WHILE high MINUS low GREATER THAN 0.00001  
            SET mid TO low PLUS high DIVIDED BY 2  
            IF CALL canFindAverage WITH mid IS True THEN  
                SET low TO mid  
            ELSE  
                SET high TO mid  
        RETURN low  
    END FUNCTION  
END CLASS
