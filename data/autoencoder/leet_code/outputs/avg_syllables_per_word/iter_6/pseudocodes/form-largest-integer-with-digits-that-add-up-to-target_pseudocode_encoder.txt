CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        SET cost_to_digit TO new empty dictionary  
        FOR index FROM 0 TO LENGTH OF cost MINUS 1  
            SET c TO element AT index IN cost  
            SET digit TO string representation OF index PLUS 1  
            SET cost_to_digit AT c TO digit  

        SET dp TO new list WITH first element AS empty string AND next target elements AS string "-1"  

        FOR t FROM 1 TO target  
            FOR each cost_value, digit_value IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO cost_value AND dp AT (t MINUS cost_value) NOT EQUALS TO "-1" THEN  
                    SET candidate TO concatenation OF digit_value PLUS dp AT (t MINUS cost_value)  
                    IF dp AT t EQUALS "-1" OR length OF candidate GREATER THAN length OF dp AT t OR (length OF candidate EQUALS length OF dp AT t AND candidate GREATER THAN dp AT t) THEN  
                        SET dp AT t TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        IF dp AT target NOT EQUALS TO "-1" THEN  
            RETURN dp AT target  
        ELSE  
            RETURN "0"  
        END IF  
    END FUNCTION  
END CLASS
