CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MOD TO ten raised to the power of nine PLUS seven  
        SET end_with_0 TO zero  
        SET end_with_1 TO zero  
        SET has_zero TO TRUE if binary CONTAINS character '0' ELSE FALSE  
        FOR each bit IN binary  
            IF bit EQUALS character '1'  
                SET end_with_1 TO (end_with_0 PLUS end_with_1 PLUS one) MODULO MOD  
            ELSE  
                SET end_with_0 TO (end_with_0 PLUS end_with_1) MODULO MOD  
        RETURN (end_with_0 PLUS end_with_1 PLUS has_zero) MODULO MOD  
    END FUNCTION  
END CLASS
