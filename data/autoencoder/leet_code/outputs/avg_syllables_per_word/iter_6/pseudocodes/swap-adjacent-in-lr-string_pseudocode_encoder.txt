CLASS Solution  
    FUNCTION canTransform(start, end)  
        SET start_filtered TO empty list  
        FOR index FROM 0 TO LENGTH OF start MINUS 1  
            IF CHARACTER AT start AT index NOT EQUALS 'X'  
                APPEND pair OF (CHARACTER AT start AT index, index) TO start_filtered  
        SET end_filtered TO empty list  
        FOR index FROM 0 TO LENGTH OF end MINUS 1  
            IF CHARACTER AT end AT index NOT EQUALS 'X'  
                APPEND pair OF (CHARACTER AT end AT index, index) TO end_filtered  
        IF LENGTH OF start_filtered NOT EQUALS LENGTH OF end_filtered  
            RETURN False  
        FOR each pair_of_pairs IN zip of start_filtered AND end_filtered  
            SET (c1, i1) TO first element of pair_of_pairs  
            SET (c2, i2) TO second element of pair_of_pairs  
            IF c1 NOT EQUALS c2  
                RETURN False  
            IF c1 EQUALS 'L' AND i1 LESS THAN i2  
                RETURN False  
            IF c1 EQUALS 'R' AND i1 GREATER THAN i2  
                RETURN False  
        RETURN True  
    END FUNCTION  
END CLASS
