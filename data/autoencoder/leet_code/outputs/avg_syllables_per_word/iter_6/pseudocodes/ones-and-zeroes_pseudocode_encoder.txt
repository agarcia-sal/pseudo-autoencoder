CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        INITIALIZE dp AS two-dimensional list with dimensions (m PLUS 1) BY (n PLUS 1) FILLED WITH zero  
        
        FOR each s IN strs  
            SET count_0 TO the number of zero characters in s  
            SET count_1 TO the number of one characters in s  
            
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp[i][j] TO the maximum of dp[i][j] AND dp[i MINUS count_0][j MINUS count_1] PLUS one  
        
        RETURN dp[m][n]  
    END FUNCTION  
END CLASS
