CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO set created from bank  
        SET q TO new deque containing tuple of start and zero  
        SET mp TO dictionary mapping 'A' TO 'TCG', 'T' TO 'ACG', 'C' TO 'ATG', 'G' TO 'ATC'  
        WHILE q IS NOT empty  
            SET t, step TO first element removed from q  
            IF t EQUALS end  
                RETURN step  
            FOR each index i AND character v IN t  
                FOR each character j IN value of mp at key v  
                    SET next TO string composed of substring of t from start to i PLUS j PLUS substring of t from i plus one to end  
                    IF next IS IN s  
                        APPEND tuple of next and step PLUS one TO q  
                        REMOVE next FROM s  
        RETURN negative one  
    END FUNCTION  
END CLASS
