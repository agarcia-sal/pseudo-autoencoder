CLASS Solution  
    FUNCTION exist(board, word)  
        IF board IS EMPTY  
            RETURN False  
        SET m TO LENGTH OF board  
        SET n TO LENGTH OF board AT INDEX 0  
        SET visited TO new two-dimensional list of size m by n with all values False  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS LENGTH OF word  
                RETURN True  
            IF x LESS THAN 0 OR x GREATER THAN OR EQUAL TO m OR y LESS THAN 0 OR y GREATER THAN OR EQUAL TO n  
                OR visited at x and y IS True  
                OR board at x and y NOT EQUALS character at word at index  
                RETURN False  

            SET visited at x and y TO True  
            FOR each (dx, dy) IN list of coordinate pairs: (-1, 0), (1, 0), (0, -1), (0, 1)  
                IF CALL dfs WITH x PLUS dx, y PLUS dy, index PLUS 1 RETURNS True  
                    RETURN True  
            SET visited at x and y TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF CALL dfs WITH i, j, 0 RETURNS True  
                    RETURN True  
        RETURN False  
    END FUNCTION  
END CLASS
