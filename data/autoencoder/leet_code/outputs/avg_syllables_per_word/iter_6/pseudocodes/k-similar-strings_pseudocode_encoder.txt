CLASS Solution  
    FUNCTION kSimilarity(s1, s2)  
        FUNCTION neighbors(s)  
            SET i TO 0  
            WHILE i LESS THAN LENGTH OF s AND CHARACTER AT s AND i EQUALS CHARACTER AT s2 AND i  
                INCREMENT i BY 1  
            FOR j FROM i PLUS 1 TO LENGTH OF s MINUS 1  
                IF CHARACTER AT s AND j EQUALS CHARACTER AT s2 AND i AND CHARACTER AT s AND j NOT EQUALS CHARACTER AT s2 AND j  
                    YIELD string FORMED BY CONCATENATING  
                        SUBSTRING OF s FROM START TO i MINUS 1,  
                        CHARACTER AT s AND j,  
                        SUBSTRING OF s FROM i PLUS 1 TO j MINUS 1,  
                        CHARACTER AT s AND i,  
                        SUBSTRING OF s FROM j PLUS 1 TO END  
        CREATE queue AS A DOUBLE-ENDED QUEUE CONTAINING PAIR OF s1 AND 0  
        CREATE visited AS A SET CONTAINING s1  
        WHILE queue IS NOT EMPTY  
            REMOVE FIRST ELEMENT FROM queue AND ASSIGN TO s AND swaps  
            IF s EQUALS s2  
                RETURN swaps  
            FOR each neighbor IN CALL neighbors WITH s  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND PAIR OF neighbor AND swaps PLUS 1 TO queue  
    END FUNCTION  
END CLASS
