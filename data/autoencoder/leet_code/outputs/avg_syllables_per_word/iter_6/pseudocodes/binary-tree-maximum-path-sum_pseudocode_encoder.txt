CLASS TreeNode  
    FUNCTION __init__(val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO NEW TreeNode WITH val AS values FIRST ELEMENT  
    SET i TO 1  
    CREATE empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.left TO NEW TreeNode WITH val AS values AT i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.right TO NEW TreeNode WITH val AS values AT i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum(root)  
        FUNCTION helper(node)  
            IF node IS None  
                RETURN negative infinity, negative infinity  
            SET left_max, left_gain TO CALL helper WITH node.left  
            SET right_max, right_gain TO CALL helper WITH node.right  
            SET current_max TO node.val PLUS maximum OF left_gain AND zero PLUS maximum OF right_gain AND zero  
            SET current_gain TO node.val PLUS maximum OF left_gain, right_gain, AND zero  
            SET global_max TO maximum OF left_max, right_max, AND current_max  
            RETURN global_max, current_gain  
        END FUNCTION  
        RETURN FIRST ITEM OF CALL helper WITH root  
    END FUNCTION  
END CLASS
