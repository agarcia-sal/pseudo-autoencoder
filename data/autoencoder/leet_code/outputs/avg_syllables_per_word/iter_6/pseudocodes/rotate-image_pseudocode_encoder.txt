CLASS Solution  
    FUNCTION rotate(matrix)  
        SET n TO LENGTH OF matrix  
        
        FOR i FROM 0 TO n MINUS 1  
            FOR j FROM i TO n MINUS 1  
                SET temporary_value TO matrix AT i AND j  
                SET matrix AT i AND j TO matrix AT j AND i  
                SET matrix AT j AND i TO temporary_value  
        
        FOR i FROM 0 TO n MINUS 1  
            CALL reverse_row WITH matrix AT i  
    END FUNCTION  
  
    FUNCTION reverse_row(row)  
        SET start_index TO 0  
        SET end_index TO LENGTH OF row MINUS 1  
        
        WHILE start_index LESS THAN end_index  
            SET temporary_value TO row AT start_index  
            SET row AT start_index TO row AT end_index  
            SET row AT end_index TO temporary_value  
            INCREMENT start_index BY 1  
            DECREMENT end_index BY 1  
    END FUNCTION  
END CLASS
