CLASS Solution  
    FUNCTION maxSum(nums1, nums2)  
        SET i TO 0  
        SET j TO 0  
        SET sum1 TO 0  
        SET sum2 TO 0  
        SET MOD TO 10 raised to the power of 9 PLUS 7  

        WHILE i LESS THAN LENGTH OF nums1 AND j LESS THAN LENGTH OF nums2  
            IF element at index i in nums1 LESS THAN element at index j in nums2  
                INCREMENT sum1 BY element at index i in nums1  
                INCREMENT i BY 1  
            ELSE IF element at index i in nums1 GREATER THAN element at index j in nums2  
                INCREMENT sum2 BY element at index j in nums2  
                INCREMENT j BY 1  
            ELSE  
                SET sum1 TO maximum of sum1 and sum2 PLUS element at index i in nums1  
                SET sum2 TO sum1  
                INCREMENT i BY 1  
                INCREMENT j BY 1  
        
        WHILE i LESS THAN LENGTH OF nums1  
            INCREMENT sum1 BY element at index i in nums1  
            INCREMENT i BY 1  
        
        WHILE j LESS THAN LENGTH OF nums2  
            INCREMENT sum2 BY element at index j in nums2  
            INCREMENT j BY 1  
        
        RETURN maximum of sum1 and sum2 MODULO MOD  
    END FUNCTION  
END CLASS
