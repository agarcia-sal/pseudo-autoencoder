CLASS Solution  
    FUNCTION canConvert(str1, str2)  
        IF str1 EQUALS str2  
            RETURN True  
        
        SET conversion_map TO empty mapping  
        
        FOR each pair char1, char2 IN corresponding characters of str1 AND str2  
            IF char1 EXISTS IN conversion_map  
                IF conversion_map value FOR char1 NOT EQUALS char2  
                    RETURN False  
            ELSE  
                SET conversion_map value FOR char1 TO char2  
        
        IF number of unique characters IN str2 EQUALS 26  
            RETURN False  
        
        RETURN True  
    END FUNCTION  
END CLASS
