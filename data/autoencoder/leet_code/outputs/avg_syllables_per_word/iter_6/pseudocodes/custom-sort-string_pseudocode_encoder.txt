CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO empty dictionary  
        FOR index FROM 0 TO LENGTH OF order MINUS 1  
            SET character TO character AT position index IN order  
            SET d[character] TO index  
        SET sorted_characters TO CALL sorted WITH s AND key FUNCTION THAT TAKES x AND RETURNS d GET x OR 0  
        SET result_string TO JOIN sorted_characters INTO single string  
        RETURN result_string  
    END FUNCTION  
END CLASS
