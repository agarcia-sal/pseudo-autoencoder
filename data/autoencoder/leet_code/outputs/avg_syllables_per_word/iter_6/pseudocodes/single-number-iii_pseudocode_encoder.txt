CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET xor_all TO 0  
        FOR each num IN nums  
            SET xor_all TO xor_all XOR num  
        
        SET diff_bit TO the rightmost set bit in xor_all  
        
        SET num1 TO 0  
        SET num2 TO 0  
        
        FOR each num IN nums  
            IF num AND diff_bit IS NOT EQUAL TO 0  
                SET num1 TO num1 XOR num  
            ELSE  
                SET num2 TO num2 XOR num  
        
        RETURN list containing num1 and num2  
    END FUNCTION  
END CLASS
