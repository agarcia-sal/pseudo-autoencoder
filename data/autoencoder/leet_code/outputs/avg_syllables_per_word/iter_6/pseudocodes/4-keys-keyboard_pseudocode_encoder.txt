CLASS Solution  
    FUNCTION maxA(n)  
        IF n LESS THAN OR EQUAL TO 3  
            RETURN n  
        END IF  
        
        SET dp TO list of zeros with length n PLUS 1  
        
        FOR i FROM 1 TO 3  
            SET dp at index i TO i  
        END FOR  
        
        FOR i FROM 4 TO n  
            FOR j FROM i MINUS 3 DOWN TO 1  
                SET candidate TO dp at index j MULTIPLIED BY (i MINUS j MINUS 1)  
                IF candidate GREATER THAN dp at index i  
                    SET dp at index i TO candidate  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at index n  
    END FUNCTION  
END CLASS
