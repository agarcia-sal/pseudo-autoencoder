CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO LENGTH OF text  
        SET substrings TO empty set  
        FOR length FROM 1 TO n DIVIDED BY 2  
            FOR start FROM 0 TO n MINUS (2 MULTIPLIED BY length)  
                SET a TO substring of text FROM start TO start PLUS length  
                SET b TO substring of text FROM start PLUS length TO start PLUS (2 MULTIPLIED BY length)  
                IF a EQUALS b  
                    ADD concatenation of a AND b TO substrings  
        RETURN SIZE OF substrings  
    END FUNCTION  
END CLASS
