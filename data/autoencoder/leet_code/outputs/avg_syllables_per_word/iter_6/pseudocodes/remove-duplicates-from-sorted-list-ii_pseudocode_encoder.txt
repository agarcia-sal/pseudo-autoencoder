CLASS ListNode  
    FUNCTION __init__(val = 0, next = None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET head TO CALL ListNode WITH values AT FIRST POSITION  
    SET p TO head  
    FOR each val IN values FROM SECOND ELEMENT TO END  
        SET node TO CALL ListNode WITH val  
        SET p.next TO node  
        SET p TO node  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN p1.val EQUALS p2.val AND CALL is_same_list WITH p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates(head)  
        SET dummy TO CALL ListNode WITH 0 AND head  
        SET prev TO dummy  
        
        WHILE head IS NOT None  
            IF head.next IS NOT None AND head.val EQUALS head.next.val  
                WHILE head.next IS NOT None AND head.val EQUALS head.next.val  
                    SET head TO head.next  
                SET prev.next TO head.next  
            ELSE  
                SET prev TO prev.next  
            SET head TO head.next  
        
        RETURN dummy.next  
    END FUNCTION  
END CLASS
