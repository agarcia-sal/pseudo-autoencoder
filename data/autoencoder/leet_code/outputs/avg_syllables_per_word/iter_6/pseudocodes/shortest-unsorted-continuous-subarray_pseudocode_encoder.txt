CLASS Solution  
    FUNCTION findUnsortedSubarray(nums)  
        SET sorted_nums TO CALL sorted WITH nums  
        SET start TO minus one  
        SET end TO minus two  
        FOR index FROM zero TO LENGTH OF nums MINUS one  
            IF nums AT index NOT EQUALS sorted_nums AT index  
                IF start EQUALS minus one  
                    SET start TO index  
                END IF  
                SET end TO index  
            END IF  
        END FOR  
        RETURN end MINUS start PLUS one  
    END FUNCTION  
END CLASS
