CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO list containing single element negative one  
        SET max_length TO zero  
        
        FOR each index i AND character char IN s  
            IF char EQUALS '('  
                APPEND i TO stack  
            ELSE  
                REMOVE last element FROM stack  
                IF stack IS empty  
                    APPEND i TO stack  
                ELSE  
                    SET max_length TO maximum BETWEEN max_length AND i MINUS last element IN stack  
        RETURN max_length  
    END FUNCTION  
END CLASS
