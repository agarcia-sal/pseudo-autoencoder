CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO dictionary with key 0 mapped to value -1  
        SET max_length TO 0  
        SET count TO 0  
        
        FOR each index i AND element num IN nums  
            IF num EQUALS 1  
                INCREMENT count BY 1  
            ELSE  
                DECREMENT count BY 1  
            END IF  
            
            IF prefix_sum_index CONTAINS key count  
                SET max_length TO the greater value BETWEEN max_length AND (i MINUS prefix_sum_index at key count)  
            ELSE  
                SET prefix_sum_index at key count TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
