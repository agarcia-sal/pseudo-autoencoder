CLASS Solution  
    FUNCTION boldWords(words, s)  
        SET bold TO list of False repeated LENGTH OF s times  
        
        FOR each word IN words  
            SET start TO position of word IN s  
            WHILE start NOT EQUALS negative one  
                FOR index FROM start TO start PLUS LENGTH OF word MINUS one  
                    SET bold at index TO True  
                SET start TO position of word IN s starting from start PLUS one  
        
        SET result TO empty list  
        SET i TO zero  
        WHILE i LESS THAN LENGTH OF s  
            IF bold at i IS True  
                APPEND string "less than b greater than" TO result  
                WHILE i LESS THAN LENGTH OF s AND bold at i IS True  
                    APPEND s at i TO result  
                    INCREMENT i BY one  
                APPEND string "less than slash b greater than" TO result  
            ELSE  
                APPEND s at i TO result  
                INCREMENT i BY one  
        
        RETURN concatenation of result  
    END FUNCTION  
END CLASS
