CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO frequency count of elements in nums  
        INITIALIZE dp AS list of zeros with length equal to two raised to the power ten  
        SET dp at index zero TO one  

        FOR each num IN count  
            IF num EQUALS one THEN  
                SET dp at index zero TO dp at index zero MULTIPLIED BY two raised to the power count at num, MODULO MOD  
                CONTINUE  
            END IF  
            
            IF num NOT IN prime_factors THEN  
                CONTINUE  
            END IF  
            
            SET factors TO prime_factors at num  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask OR two raised to the power (prime MINUS one)  
            END FOR  
            
            FOR m FROM (two raised to the power ten MINUS one) DOWN TO zero  
                IF bitwise AND of m and mask EQUALS zero THEN  
                    SET dp at index (m OR mask) TO (dp at index (m OR mask) PLUS dp at index m MULTIPLIED BY count at num) MODULO MOD  
                END IF  
            END FOR  
        END FOR  

        SET result TO sum of dp elements from index one to end, MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
