CLASS Solution  
    FUNCTION isTransformable(s, t)  
        CREATE pos AS default dictionary mapping to empty list  
        FOR index FROM 0 TO LENGTH OF s MINUS 1  
            CALL heappush ON pos WITH key AS integer value of s AT index AND value AS index  
        
        FOR index FROM 0 TO LENGTH OF t MINUS 1  
            SET digit TO integer value of t AT index  
            IF pos AT digit IS empty  
                RETURN False  
            FOR smaller FROM 0 TO digit MINUS 1  
                IF pos AT smaller IS NOT empty AND first element IN pos AT smaller IS LESS THAN first element IN pos AT digit  
                    RETURN False  
            CALL heappop ON pos AT digit  
        
        RETURN True  
    END FUNCTION  
END CLASS
