CLASS ListNode  
    FUNCTION __init__(self, val = 0, next = None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET head TO new ListNode WITH value AT values FIRST ELEMENT  
    SET p TO head  
    FOR each val IN values FROM SECOND ELEMENT TO END  
        SET node TO new ListNode WITH val  
        SET p.next TO node  
        SET p TO node  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN p1.val EQUALS p2.val AND CALL is_same_list WITH p1.next, p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1, l2)  
        FUNCTION pushIntoStack(node)  
            SET stack TO empty list  
            WHILE node IS NOT None  
                APPEND node.val TO stack  
                SET node TO node.next  
            RETURN stack  
        END FUNCTION  
        
        SET stack1 TO CALL pushIntoStack WITH l1  
        SET stack2 TO CALL pushIntoStack WITH l2  
        
        SET carry TO 0  
        SET result TO None  
        
        WHILE stack1 IS NOT EMPTY OR stack2 IS NOT EMPTY OR carry IS GREATER THAN 0  
            IF stack1 IS NOT EMPTY  
                SET val1 TO LAST ELEMENT OF stack1 AND REMOVE IT  
            ELSE  
                SET val1 TO 0  
            IF stack2 IS NOT EMPTY  
                SET val2 TO LAST ELEMENT OF stack2 AND REMOVE IT  
            ELSE  
                SET val2 TO 0  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY 10  
            SET digit TO total MODULO 10  
            
            SET new_node TO new ListNode WITH digit  
            SET new_node.next TO result  
            SET result TO new_node  
        
        RETURN result  
    END FUNCTION  
END CLASS
