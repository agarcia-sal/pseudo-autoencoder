CLASS Solution  
    FUNCTION countArrangement(n)  
        FUNCTION backtrack(used, index)  
            IF index EQUALS n PLUS 1 THEN  
                RETURN 1  
            SET count TO 0  
            FOR i FROM 1 TO n  
                IF used AT i IS FALSE AND (i MODULO index EQUALS 0 OR index MODULO i EQUALS 0) THEN  
                    SET used AT i TO TRUE  
                    INCREMENT count BY CALL backtrack WITH used AND index PLUS 1  
                    SET used AT i TO FALSE  
            RETURN count  
        END FUNCTION  
        
        SET used TO list of FALSE with length n PLUS 1  
        RETURN CALL backtrack WITH used AND 1  
    END FUNCTION  
END CLASS
