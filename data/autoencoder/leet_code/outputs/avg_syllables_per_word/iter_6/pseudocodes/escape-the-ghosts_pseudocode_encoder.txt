CLASS Solution  
    FUNCTION escapeGhosts(ghosts, target)  
        SET player_distance TO absolute value of target at position zero PLUS absolute value of target at position one  
        
        FOR each ghost IN ghosts  
            SET ghost_distance TO absolute value of ghost at position zero MINUS target at position zero PLUS absolute value of ghost at position one MINUS target at position one  
            IF ghost_distance LESS THAN OR EQUAL TO player_distance  
                RETURN False  
        RETURN True  
    END FUNCTION  
END CLASS
