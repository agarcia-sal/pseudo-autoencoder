CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO result of splitting input BY newline  
        SET stack TO empty list  
        SET max_length TO 0  
        
        FOR each line IN lines  
            SET depth TO count of tab characters in line  
            SET name TO substring of line starting from depth position  
            
            WHILE length of stack IS GREATER THAN depth  
                REMOVE last element FROM stack  
            
            IF stack IS NOT empty  
                SET current_length TO last element of stack PLUS length of name PLUS 1  
            ELSE  
                SET current_length TO length of name  
            
            IF name CONTAINS dot character  
                SET max_length TO greater of max_length AND current_length  
            ELSE  
                APPEND current_length TO stack  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
