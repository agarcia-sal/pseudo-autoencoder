CLASS Solution  
    FUNCTION minTaps(n, ranges)  
        SET max_range TO a list of zeros with length n PLUS 1  
        
        FOR i FROM 0 TO n  
            IF value at ranges at i GREATER THAN 0  
                SET left TO the greater value between 0 AND i MINUS ranges at i  
                SET right TO the lesser value between n AND i PLUS ranges at i  
                SET max_range at left TO the greater value between max_range at left AND right  
            END IF  
        END FOR  
        
        SET taps TO 0  
        SET current_end TO 0  
        SET farthest TO 0  
        
        FOR i FROM 0 TO n  
            IF i GREATER THAN farthest  
                RETURN negative 1  
            END IF  
            IF i GREATER THAN current_end  
                INCREMENT taps BY 1  
                SET current_end TO farthest  
            END IF  
            SET farthest TO the greater value between farthest AND max_range at i  
        END FOR  
        
        RETURN taps  
    END FUNCTION  
END CLASS
