CLASS Solution
    FUNCTION ambiguousCoordinates(s)
        FUNCTION make(frag)
            SET n TO LENGTH OF frag
            FOR d FROM 1 TO n
                SET left TO SUBSTRING OF frag FROM START TO d
                SET right TO SUBSTRING OF frag FROM d TO END
                IF (left DOES NOT START WITH '0' OR left EQUALS '0') AND right DOES NOT END WITH '0'
                    YIELD CONCATENATION OF left AND IF d NOT EQUALS n THEN '.' ELSE '' AND right
                END IF
            END FOR
        END FUNCTION

        SET s TO SUBSTRING OF s FROM SECOND CHARACTER TO SECOND LAST CHARACTER
        RETURN LIST OF STRINGS FORMATTED AS "(" CONCAT a CONCAT ", " CONCAT b CONCAT ")" 
            FOR i FROM 1 TO LENGTH OF s MINUS 1
            FOR EACH a, b IN CARTESIAN PRODUCT OF make(SUBSTRING OF s FROM START TO i) AND make(SUBSTRING OF s FROM i TO END)
    END FUNCTION
END CLASS
