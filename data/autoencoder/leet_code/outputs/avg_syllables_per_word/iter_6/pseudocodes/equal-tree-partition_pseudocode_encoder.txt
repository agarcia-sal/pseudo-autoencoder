CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH val AS values[0]  
    SET i TO 1  
    SET queue TO EMPTY QUEUE  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None  
            SET node.left TO NEW TreeNode WITH val AS values[i]  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None  
            SET node.right TO NEW TreeNode WITH val AS values[i]  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree(root)  
        FUNCTION subtree_sum(node)  
            IF node IS None  
                RETURN 0  
            RETURN node.val PLUS CALL subtree_sum WITH node.left PLUS CALL subtree_sum WITH node.right  
        END FUNCTION  

        SET total_sum TO CALL subtree_sum WITH root  

        IF total_sum MODULO 2 NOT EQUALS 0  
            RETURN False  

        SET target TO total_sum DIVIDED BY 2  
        SET found TO LIST WITH VALUE False  

        FUNCTION can_split(node)  
            IF node IS None  
                RETURN 0  
            SET current_sum TO node.val PLUS CALL can_split WITH node.left PLUS CALL can_split WITH node.right  
            IF current_sum EQUALS target AND node NOT EQUALS root  
                SET found[0] TO True  
            RETURN current_sum  
        END FUNCTION  

        CALL can_split WITH root  
        RETURN found[0]  
    END FUNCTION  
END CLASS
