CLASS Solution  
    FUNCTION numberOfWays(numPeople)  
        SET MOD TO one billion PLUS seven  
        INITIALIZE dp AS list of zeros WITH size numPeople PLUS one  
        SET dp at index zero TO one  
        SET dp at index two TO one  
        
        FOR i FROM four TO numPeople STEP two  
            FOR j FROM zero TO i MINUS one STEP two  
                SET dp at index i TO dp at index i PLUS dp at index j MULTIPLIED BY dp at index i MINUS two MINUS j  
                SET dp at index i TO dp at index i MODULO MOD  
        
        RETURN dp at index numPeople  
    END FUNCTION  
END CLASS
