CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO CALL Counter WITH s  
        SET max_heap TO EMPTY LIST  
        FOR each char, freq IN count items  
            APPEND tuple OF negative freq AND char TO max_heap  
        CALL heapify ON max_heap  
        
        SET prev_char TO None  
        SET prev_freq TO 0  
        SET result TO EMPTY LIST  
        
        WHILE max_heap IS NOT EMPTY OR prev_freq LESS THAN 0 IS TRUE  
            IF max_heap IS EMPTY AND prev_freq LESS THAN 0 IS TRUE  
                RETURN empty string  
            
            SET freq, char TO CALL heappop ON max_heap  
            APPEND char TO result  
            
            SET freq TO freq PLUS 1  
            
            IF prev_freq LESS THAN 0 IS TRUE  
                CALL heappush ON max_heap WITH tuple OF prev_freq AND prev_char  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        
        RETURN concatenation OF all characters IN result  
    END FUNCTION  
END CLASS
