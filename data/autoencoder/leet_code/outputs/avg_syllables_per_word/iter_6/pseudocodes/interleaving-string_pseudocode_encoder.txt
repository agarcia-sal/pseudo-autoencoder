CLASS Solution  
    FUNCTION isInterleave(s1, s2, s3)  
        IF LENGTH OF s1 PLUS LENGTH OF s2 NOT EQUALS LENGTH OF s3  
            RETURN False  
        END IF  

        SET dp TO list of False values with LENGTH OF s2 PLUS 1  
        SET dp at index 0 TO True  

        FOR i FROM 0 TO LENGTH OF s1  
            FOR j FROM 0 TO LENGTH OF s2  
                IF i GREATER THAN 0  
                    SET dp at index j TO dp at index j AND value at s1 position i MINUS 1 EQUALS value at s3 position i PLUS j MINUS 1  
                END IF  
                IF j GREATER THAN 0  
                    SET dp at index j TO dp at index j OR (dp at index j MINUS 1 AND value at s2 position j MINUS 1 EQUALS value at s3 position i PLUS j MINUS 1)  
                END IF  
            END FOR  
        END FOR  

        RETURN dp at index LENGTH OF s2  
    END FUNCTION  
END CLASS
