CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO list of characters representing n  
        SET length TO length of digits  
          
        SET i TO length MINUS 2  
        WHILE i GREATER THAN OR EQUAL TO 0 AND digits at i GREATER THAN OR EQUAL TO digits at i PLUS 1 IS TRUE  
            DECREMENT i BY 1  
          
        IF i EQUALS -1 THEN  
            RETURN -1  
          
        SET j TO length MINUS 1  
        WHILE digits at j LESS THAN OR EQUAL TO digits at i IS TRUE  
            DECREMENT j BY 1  
          
        SWAP digits at i WITH digits at j  
          
        SET digits TO concatenation of digits from start to i PLUS 1 AND reversed sublist of digits from i PLUS 1 to end  
          
        SET result TO integer formed by concatenating elements of digits  
          
        IF result GREATER THAN 2 POWER 31 MINUS 1 THEN  
            RETURN -1  
          
        RETURN result  
    END FUNCTION  
END CLASS
