CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO empty list  
        FOR index FROM 0 TO LENGTH OF s PLUS 1 MINUS 1  
            APPEND index PLUS 1 TO result  
        SET start TO 0  
        FOR end FROM 0 TO LENGTH OF s  
            IF end EQUALS LENGTH OF s OR CHARACTER AT s AT end EQUALS 'I'  
                REVERSE SUBLIST OF result FROM start TO end INCLUSIVE  
                SET start TO end PLUS 1  
        RETURN result  
    END FUNCTION  
END CLASS
