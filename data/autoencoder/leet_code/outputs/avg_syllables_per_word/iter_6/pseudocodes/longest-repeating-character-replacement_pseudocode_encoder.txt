CLASS Solution  
    FUNCTION characterReplacement(s, k)  
        INITIALIZE max_length TO 0  
        INITIALIZE max_count TO 0  
        INITIALIZE left TO 0  
        INITIALIZE count TO empty mapping with default value zero  
        
        FOR right FROM 0 TO LENGTH OF s MINUS 1  
            INCREMENT count VALUE FOR s AT POSITION right BY 1  
            SET max_count TO the maximum OF max_count AND count VALUE FOR s AT POSITION right  
            
            IF (right MINUS left PLUS 1) MINUS max_count GREATER THAN k THEN  
                DECREMENT count VALUE FOR s AT POSITION left BY 1  
                INCREMENT left BY 1  
            END IF  
            
            SET max_length TO the maximum OF max_length AND (right MINUS left PLUS 1)  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
