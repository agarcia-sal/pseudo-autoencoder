CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each char IN s  
            IF char IS digit THEN  
                SET d TO d CONCATENATED WITH char  
            ELSE IF char IS alphabetic THEN  
                SET a TO a CONCATENATED WITH char  
            ELSE IF char EQUALS open bracket THEN  
                APPEND tuple of (a, CONVERT d TO integer) TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char EQUALS close bracket THEN  
                SET pair TO REMOVE last element FROM stack  
                SET p TO first element OF pair  
                SET n TO second element OF pair  
                SET a TO p CONCATENATED WITH a REPEATED n TIMES  
        RETURN a  
    END FUNCTION  
END CLASS
