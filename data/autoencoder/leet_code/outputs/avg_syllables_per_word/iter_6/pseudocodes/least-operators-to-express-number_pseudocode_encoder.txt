CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        DEFINE FUNCTION dp(target)  
            IF target EQUALS 0 THEN  
                RETURN negative one  
            ELSE IF target LESS THAN x THEN  
                RETURN minimum of (two MULTIPLIED BY target MINUS one) AND (two MULTIPLIED BY (x MINUS target))  
            END IF  
            
            SET power TO one  
            WHILE x RAISED TO power LESS THAN OR EQUAL TO target IS TRUE  
                INCREMENT power BY one  
            DECREMENT power BY one  
            
            SET res TO CALL dp WITH (target MINUS x RAISED TO power) PLUS power  
            
            IF (x RAISED TO (power PLUS one) MINUS target) LESS THAN target THEN  
                SET res TO minimum of res AND (CALL dp WITH (x RAISED TO (power PLUS one) MINUS target) PLUS power PLUS one)  
            END IF  
            
            RETURN res  
        END FUNCTION  
        
        RETURN CALL dp WITH target  
    END FUNCTION  
END CLASS
