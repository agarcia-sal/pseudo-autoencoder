CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO list of characters representing n  
        SET length TO length of digits  
        SET marker TO length  
        FOR i FROM length MINUS 1 DOWN TO 1  
            IF digits[i] LESS THAN digits[i MINUS 1]  
                SET digits[i MINUS 1] TO string of (integer value of digits[i MINUS 1] MINUS 1)  
                SET marker TO i  
        FOR i FROM marker TO length MINUS 1  
            SET digits[i] TO character '9'  
        SET result_string TO concatenation of all elements in digits  
        SET result TO integer value of result_string  
        RETURN result  
    END FUNCTION  
END CLASS
