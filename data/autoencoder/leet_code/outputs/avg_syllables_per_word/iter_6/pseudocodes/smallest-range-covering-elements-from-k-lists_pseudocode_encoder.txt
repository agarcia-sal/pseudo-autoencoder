CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 1  
            CALL heap push WITH arguments min_heap AND tuple of (first element of nums at i, i, 0)  
            IF max_value LESS THAN first element of nums at i  
                SET max_value TO first element of nums at i  
        
        SET smallest_range TO infinity  
        SET result TO list of -100000 AND 100000  
        
        WHILE min_heap IS NOT EMPTY  
            CALL heap pop WITH argument min_heap AND SET TO min_value, list_index, element_index  
            
            IF max_value MINUS min_value LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO list of min_value AND max_value  
            
            IF element_index PLUS 1 LESS THAN LENGTH OF nums at list_index  
                SET next_value TO element at position element_index PLUS 1 in nums at list_index  
                CALL heap push WITH arguments min_heap AND tuple of (next_value, list_index, element_index PLUS 1)  
                IF max_value LESS THAN next_value  
                    SET max_value TO next_value  
            ELSE  
                BREAK  
        
        RETURN result  
    END FUNCTION  
END CLASS
