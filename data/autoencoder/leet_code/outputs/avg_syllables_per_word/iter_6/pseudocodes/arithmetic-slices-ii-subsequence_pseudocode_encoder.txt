CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO LENGTH OF nums  
        IF n LESS THAN 3  
            RETURN 0  
        
        SET dp TO list of empty mappings from integer to integer with size n  
        SET total_count TO 0  
        
        FOR i FROM 0 TO n MINUS 1  
            FOR j FROM 0 TO i MINUS 1  
                SET d TO nums AT i MINUS nums AT j  
                
                IF d EXISTS IN dp AT j  
                    INCREMENT dp AT i AT d BY dp AT j AT d PLUS 1  
                    INCREMENT total_count BY dp AT j AT d  
                ELSE  
                    INCREMENT dp AT i AT d BY 1  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
