CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF LENGTH OF nums LESS THAN 2 THEN  
            RETURN 0  
        END IF  

        FUNCTION countingSort(arr, exp)  
            SET n TO LENGTH OF arr  
            INITIALIZE output AS list of zeros of length n  
            INITIALIZE count AS list of zeros of length 10  
            
            FOR i FROM 0 TO n MINUS 1  
                SET index TO arr AT i DIVIDED BY exp  
                INCREMENT count AT index MODULO 10 BY 1  
            END FOR  
            
            FOR i FROM 1 TO 9  
                SET count AT i TO count AT i PLUS count AT (i MINUS 1)  
            END FOR  
            
            SET i TO n MINUS 1  
            WHILE i GREATER THAN OR EQUAL TO 0  
                SET index TO arr AT i DIVIDED BY exp  
                SET output AT (count AT index MODULO 10 MINUS 1) TO arr AT i  
                DECREMENT count AT index MODULO 10 BY 1  
                DECREMENT i BY 1  
            END WHILE  
            
            FOR i FROM 0 TO n MINUS 1  
                SET arr AT i TO output AT i  
            END FOR  
        END FUNCTION  

        FUNCTION radixSort(arr)  
            SET max1 TO maximum value in arr  
            SET exp TO 1  
            WHILE max1 DIVIDED BY exp GREATER THAN 0  
                CALL countingSort WITH arr AND exp  
                SET exp TO exp MULTIPLIED BY 10  
            END WHILE  
        END FUNCTION  

        CALL radixSort WITH nums  
        
        SET max_diff TO 0  
        FOR i FROM 1 TO LENGTH OF nums MINUS 1  
            IF (nums AT i MINUS nums AT (i MINUS 1)) GREATER THAN max_diff THEN  
                SET max_diff TO nums AT i MINUS nums AT (i MINUS 1)  
            END IF  
        END FOR  
        
        RETURN max_diff  
    END FUNCTION  
END CLASS
