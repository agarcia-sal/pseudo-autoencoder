CLASS Solution  
    FUNCTION canCompleteCircuit(gas, cost)  
        IF sum of gas IS LESS THAN sum of cost THEN  
            RETURN -1  
        SET start TO 0  
        SET total_tank TO 0  
        FOR index FROM 0 TO LENGTH of gas MINUS 1  
            SET total_tank TO total_tank PLUS gas AT index MINUS cost AT index  
            IF total_tank IS LESS THAN 0 THEN  
                SET start TO index PLUS 1  
                SET total_tank TO 0  
        RETURN start  
    END FUNCTION  
END CLASS
