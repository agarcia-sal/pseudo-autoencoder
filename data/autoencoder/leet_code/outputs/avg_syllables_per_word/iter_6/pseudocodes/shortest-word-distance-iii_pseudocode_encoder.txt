CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict, word1, word2)  
        SET min_distance TO infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO whether word1 EQUALS word2  
        
        FOR index FROM zero TO LENGTH OF wordsDict MINUS one  
            SET word TO element AT index IN wordsDict  
            
            IF word EQUALS word1  
                IF same_word IS TRUE  
                    SET prev_word2 TO prev_word1  
                SET prev_word1 TO index  
            END IF  
            
            IF word EQUALS word2  
                IF same_word IS FALSE  
                    SET prev_word2 TO index  
                END IF  
            END IF  
            
            IF prev_word1 NOT EQUALS negative one AND prev_word2 NOT EQUALS negative one  
                SET distance TO absolute value of prev_word1 MINUS prev_word2  
                IF distance LESS THAN min_distance  
                    SET min_distance TO distance  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
