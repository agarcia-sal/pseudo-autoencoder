CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO LENGTH OF floor  
        CALL FUNCTION createDpArray WITH n AND numCarpets, RETURNING dp  
  
        FOR i FROM 1 TO n  
            SET dp[i][0] TO dp[i MINUS 1][0] PLUS IF floor AT POSITION i MINUS 1 EQUALS '1' THEN 1 ELSE 0  
  
        FOR j FROM 1 TO numCarpets  
            FOR i FROM 1 TO n  
                SET dp[i][j] TO dp[i MINUS 1][j] PLUS IF floor AT POSITION i MINUS 1 EQUALS '1' THEN 1 ELSE 0  
                SET dp[i][j] TO MINIMUM OF dp[i][j] AND dp[MAXIMUM OF 0 AND i MINUS carpetLen][j MINUS 1]  
  
        RETURN dp[n][numCarpets]  
    END FUNCTION  
  
    FUNCTION createDpArray(n, numCarpets)  
        CREATE two dimensional list dp WITH n PLUS 1 ROWS AND numCarpets PLUS 1 COLUMNS INITIALIZED TO ZERO  
        RETURN dp  
    END FUNCTION  
END CLASS
