CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO LENGTH OF arr  
        INITIALIZE min_heap AS empty list  
        
        FOR i FROM 0 TO n MINUS 2  
            CALCULATE fraction_value AS arr[i] DIVIDED BY arr[n MINUS 1]  
            ADD tuple of fraction_value, i, n MINUS 1 TO min_heap USING heap push operation  
        
        FOR count FROM 1 TO k MINUS 1  
            REMOVE smallest fraction tuple FROM min_heap USING heap pop operation  
            DESTRUCTURE tuple INTO _, i, j  
            IF j MINUS 1 GREATER THAN i  
                CALCULATE new_fraction AS arr[i] DIVIDED BY arr[j MINUS 1]  
                ADD tuple of new_fraction, i, j MINUS 1 TO min_heap USING heap push operation  
        
        REMOVE smallest fraction tuple FROM min_heap USING heap pop operation  
        DESTRUCTURE tuple INTO _, i, j  
        RETURN list containing arr[i] FOLLOWED BY arr[j]  
    END FUNCTION  
END CLASS
