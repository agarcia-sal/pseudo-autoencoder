CLASS Solution  
    FUNCTION kthGrammar(n, k)  
        IF n EQUALS 1  
            RETURN 0  
        SET length_of_previous_row TO 2 MULTIPLIED BY (2 RAISED TO THE POWER OF (n MINUS 2) DIVIDED BY 2)  
        SET length_of_previous_row TO 2 RAISED TO THE POWER OF (n MINUS 2)  
        IF k LESS THAN OR EQUAL TO length_of_previous_row  
            RETURN CALL kthGrammar WITH (n MINUS 1, k)  
        ELSE  
            SET previous_value TO CALL kthGrammar WITH (n MINUS 1, k MINUS length_of_previous_row)  
            SET complement_value TO 1 MINUS previous_value  
            RETURN complement_value  
    END FUNCTION  
END CLASS
