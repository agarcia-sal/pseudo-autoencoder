CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO LENGTH OF s  
        SET f TO two-dimensional list of size n by n FILLED WITH True  
        SET k TO 0  
        SET mx TO 1  
        FOR i FROM n MINUS 2 DOWN TO 0  
            FOR j FROM i PLUS 1 TO n MINUS 1  
                SET f at position i and j TO False  
                IF character at s position i EQUALS character at s position j THEN  
                    SET f at position i and j TO value of f at position i PLUS 1 and j MINUS 1  
                    IF f at position i and j IS True AND mx LESS THAN j MINUS i PLUS 1 THEN  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS 1  
        RETURN substring of s FROM position k LENGTH mx  
    END FUNCTION  
END CLASS
