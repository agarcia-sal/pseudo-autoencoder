CLASS Solution  
    FUNCTION minimumTotal(triangle)  
        FOR row FROM LENGTH OF triangle MINUS 2 DOWN TO 0  
            FOR col FROM 0 TO LENGTH OF triangle AT row MINUS 1  
                SET smallest_adjacent TO MINIMUM OF triangle AT row PLUS 1 AND col, AND triangle AT row PLUS 1 AND col PLUS 1  
                SET triangle AT row AND col TO triangle AT row AND col PLUS smallest_adjacent  
            END FOR  
        END FOR  
        RETURN triangle AT 0 AND 0  
    END FUNCTION  
END CLASS
