CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO empty list  
        FOR each num IN target  
            APPEND negative num TO max_heap  
        CALL heapify ON max_heap  
  
        SET total_sum TO sum of target  
  
        WHILE first element of max_heap IS LESS THAN negative one  
            CALL heappop ON max_heap AND SET largest TO negative of result  
            SET rest_sum TO total_sum MINUS largest  
  
            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero  
                RETURN False  
            END IF  
  
            SET new_value TO largest MODULO rest_sum  
            IF new_value EQUALS zero AND rest_sum NOT EQUAL TO one  
                RETURN False  
            END IF  
  
            SET total_sum TO rest_sum PLUS new_value  
            CALL heappush ON max_heap WITH negative new_value  
        END WHILE  
  
        RETURN True  
    END FUNCTION  
END CLASS
