CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO False  
        SET result TO empty list  
        SET current_line TO empty list  

        FOR each line IN source  
            SET i TO 0  
            WHILE i LESS THAN length of line  
                IF in_block_comment IS True  
                    IF substring of line FROM i TO i PLUS 2 EQUALS "*/"  
                        SET in_block_comment TO False  
                        INCREMENT i BY 1  
                ELSE  
                    IF substring of line FROM i TO i PLUS 2 EQUALS "//"  
                        BREAK  
                    ELSE IF substring of line FROM i TO i PLUS 2 EQUALS "/*"  
                        SET in_block_comment TO True  
                        INCREMENT i BY 1  
                    ELSE  
                        APPEND character at line[i] TO current_line  
                INCREMENT i BY 1  
            
            IF current_line IS NOT empty AND in_block_comment IS False  
                APPEND concatenation of current_line TO result  
                SET current_line TO empty list  

        RETURN result  
    END FUNCTION  
END CLASS
