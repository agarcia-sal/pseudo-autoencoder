CLASS Solution  
    FUNCTION findClosestElements(arr, k, x)  
        SET left TO bisect left insertion position of x in arr MINUS 1  
        SET right TO left PLUS 1  
        
        WHILE right MINUS left MINUS 1 LESS THAN k IS TRUE  
            IF left EQUALS MINUS 1 OR (right LESS THAN length of arr AND absolute value of arr at right MINUS x LESS THAN absolute value of arr at left MINUS x) THEN  
                INCREMENT right BY 1  
            ELSE  
                DECREMENT left BY 1  
        RETURN sublist of arr from left PLUS 1 TO right  
    END FUNCTION  
END CLASS
