CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO LENGTH OF dominoes  
        SET forces TO list of zeros with length n  

        SET force TO 0  
        FOR i FROM 0 TO n MINUS 1  
            IF dominoes AT i EQUALS 'R'  
                SET force TO n  
            ELSE IF dominoes AT i EQUALS 'L'  
                SET force TO 0  
            ELSE  
                SET force TO MAXIMUM OF force MINUS 1 AND 0  
            ADD force TO forces AT i  
        
        SET force TO 0  
        FOR i FROM n MINUS 1 DOWN TO 0  
            IF dominoes AT i EQUALS 'L'  
                SET force TO n  
            ELSE IF dominoes AT i EQUALS 'R'  
                SET force TO 0  
            ELSE  
                SET force TO MAXIMUM OF force MINUS 1 AND 0  
            SUBTRACT force FROM forces AT i  

        SET result TO empty list  
        FOR each f IN forces  
            IF f GREATER THAN 0  
                APPEND 'R' TO result  
            ELSE IF f LESS THAN 0  
                APPEND 'L' TO result  
            ELSE  
                APPEND '.' TO result  
        
        RETURN CONCATENATION OF all elements in result  
    END FUNCTION  
END CLASS
