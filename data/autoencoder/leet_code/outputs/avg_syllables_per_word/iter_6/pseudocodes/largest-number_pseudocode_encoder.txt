CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO list of strings converted from nums  
        
        FUNCTION compare(x, y)  
            IF concatenation of x PLUS y IS GREATER THAN concatenation of y PLUS x  
                RETURN negative one  
            ELSE IF concatenation of x PLUS y IS LESS THAN concatenation of y PLUS x  
                RETURN one  
            ELSE  
                RETURN zero  
        END FUNCTION  
        
        CALL sort ON nums_str WITH key AS comparison based on compare function  
        
        IF first element of nums_str EQUALS string "0"  
            RETURN string "0"  
        
        SET result TO concatenation of all elements in nums_str  
        RETURN result  
    END FUNCTION  
END CLASS
