CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS 0, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH values AT FIRST POSITION  
    SET i TO 1  
    SET queue TO NEW deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.left TO NEW TreeNode WITH values AT i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.right TO NEW TreeNode WITH values AT i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder(root)  
        IF root IS None  
            RETURN EMPTY LIST  
        SET column_table TO NEW DEFAULTDICTIONARY WITH LIST AS DEFAULT  
        SET queue TO NEW deque WITH ELEMENT (root, 0)  
        SET min_column TO 0  
        SET max_column TO 0  
        WHILE queue IS NOT EMPTY  
            SET node, column_index TO REMOVE FIRST ELEMENT FROM queue  
            IF node IS NOT None  
                APPEND node.val TO column_table AT column_index  
                SET min_column TO MINIMUM OF min_column AND column_index  
                SET max_column TO MAXIMUM OF max_column AND column_index  
                APPEND (node.left, column_index MINUS 1) TO queue  
                APPEND (node.right, column_index PLUS 1) TO queue  
        SET result TO EMPTY LIST  
        FOR x FROM min_column TO max_column  
            APPEND column_table AT x TO result  
        RETURN result  
    END FUNCTION  
END CLASS
