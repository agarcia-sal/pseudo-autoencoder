CLASS Solution  
    FUNCTION kInversePairs(n, k)  
        SET MOD TO 10 POWERED TO 9 PLUS 7  
        CREATE dp AS list of lists with dimensions (n PLUS 1) BY (k PLUS 1) FILLED WITH 0  
        SET dp AT index 0,0 TO 1  
        
        FOR i FROM 1 TO n  
            SET prefix_sum TO 0  
            FOR j FROM 0 TO k  
                SET prefix_sum TO prefix_sum PLUS dp AT index i MINUS 1, j  
                SET prefix_sum TO prefix_sum MODULO MOD  
                IF j GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO prefix_sum MINUS dp AT index i MINUS 1, j MINUS i PLUS MOD  
                    SET prefix_sum TO prefix_sum MODULO MOD  
                END IF  
                SET dp AT index i, j TO prefix_sum  
            END FOR  
        END FOR  
        
        RETURN dp AT index n, k  
    END FUNCTION  
END CLASS
