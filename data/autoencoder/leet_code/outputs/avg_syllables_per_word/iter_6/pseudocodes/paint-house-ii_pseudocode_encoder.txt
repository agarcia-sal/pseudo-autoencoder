CLASS Solution  
    FUNCTION minCostII(costs)  
        IF costs IS EMPTY  
            RETURN 0  
        
        SET n TO LENGTH OF costs  
        SET k TO LENGTH OF costs AT INDEX 0  
        
        IF k EQUALS 1  
            IF n EQUALS 1  
                RETURN costs AT 0 AT 0  
            ELSE  
                RETURN infinity  
        
        FOR i FROM 1 TO n MINUS 1  
            SET min_cost TO infinity  
            SET min_cost_index TO negative one  
            SET second_min_cost TO infinity  
            
            FOR j FROM 0 TO k MINUS 1  
                IF costs AT i MINUS 1 AT j LESS THAN min_cost  
                    SET second_min_cost TO min_cost  
                    SET min_cost TO costs AT i MINUS 1 AT j  
                    SET min_cost_index TO j  
                ELSE IF costs AT i MINUS 1 AT j LESS THAN second_min_cost  
                    SET second_min_cost TO costs AT i MINUS 1 AT j  
            
            FOR j FROM 0 TO k MINUS 1  
                IF j EQUALS min_cost_index  
                    INCREMENT costs AT i AT j BY second_min_cost  
                ELSE  
                    INCREMENT costs AT i AT j BY min_cost  
        
        RETURN MINIMUM VALUE IN costs AT LAST INDEX  
    END FUNCTION  
END CLASS
