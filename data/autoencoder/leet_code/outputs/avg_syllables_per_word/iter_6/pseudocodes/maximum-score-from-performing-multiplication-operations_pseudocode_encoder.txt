CLASS Solution  
    FUNCTION maximumScore(nums, multipliers)  
        SET m TO LENGTH OF multipliers  
        SET n TO LENGTH OF nums  
        
        SET dp TO list of zeroes with length m PLUS 1  
        
        FOR i FROM m MINUS 1 DOWN TO 0  
            SET new_dp TO list of zeroes with length m PLUS 1  
            FOR left FROM 0 TO i  
                SET right TO n MINUS 1 MINUS (i MINUS left)  
                SET mul TO multipliers AT i  
                SET new_dp AT left TO MAXIMUM OF (mul MULTIPLIED BY nums AT left PLUS dp AT left PLUS 1) AND (mul MULTIPLIED BY nums AT right PLUS dp AT left)  
            SET dp TO new_dp  
        
        RETURN dp AT 0  
    END FUNCTION  
END CLASS
