CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH val AS values AT index zero  
    SET i TO 1  
    SET queue TO NEW empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.left TO NEW TreeNode WITH val AS values AT i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.right TO NEW TreeNode WITH val AS values AT i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION subtreeWithAllDeepest(root)  
        FUNCTION dfs(node)  
            IF node IS None  
                RETURN node, zero  
            SET left_node, left_depth TO CALL dfs WITH node.left  
            SET right_node, right_depth TO CALL dfs WITH node.right  
            IF left_depth GREATER THAN right_depth  
                RETURN left_node, left_depth PLUS one  
            ELSE IF right_depth GREATER THAN left_depth  
                RETURN right_node, right_depth PLUS one  
            ELSE  
                RETURN node, left_depth PLUS one  
        END FUNCTION  

        RETURN FIRST ELEMENT OF CALL dfs WITH root  
    END FUNCTION  
END CLASS
