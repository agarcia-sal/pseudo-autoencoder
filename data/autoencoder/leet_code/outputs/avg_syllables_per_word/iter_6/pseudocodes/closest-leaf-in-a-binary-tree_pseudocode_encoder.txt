CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH val EQUALS first element of values  
    SET i TO 1  
    CREATE queue as empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO first element removed from queue  
        IF i IS LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.left TO NEW TreeNode WITH val EQUALS values at i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.right TO NEW TreeNode WITH val EQUALS values at i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(root, k)  
        IF root.left IS None AND root.right IS None  
            RETURN root.val  
        
        CREATE graph AS empty mapping from integer TO list  
        
        FUNCTION dfs(node, parent)  
            IF parent IS NOT None  
                APPEND parent.val TO graph at node.val  
                APPEND node.val TO graph at parent.val  
            IF node.left IS NOT None  
                CALL dfs WITH node.left AND node  
            IF node.right IS NOT None  
                CALL dfs WITH node.right AND node  
        END FUNCTION  
        
        CALL dfs WITH root AND None  
        
        CREATE queue AS deque initialized with k  
        CREATE visited AS empty set  
        
        WHILE queue IS NOT EMPTY  
            SET current TO first element removed from queue  
            ADD current TO visited  
            
            IF LENGTH OF graph at current EQUALS 1 AND current NOT EQUALS root.val  
                RETURN current  
            
            FOR each neighbor IN graph at current  
                IF neighbor NOT IN visited  
                    APPEND neighbor TO queue  
        END WHILE  
        
        RETURN -1  
    END FUNCTION  
END CLASS
