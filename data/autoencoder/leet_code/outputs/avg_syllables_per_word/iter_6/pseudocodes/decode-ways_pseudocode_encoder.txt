CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s IS empty OR first character of s EQUALS '0'  
            RETURN 0  
        
        INITIALIZE dp AS list of zeros WITH length OF s PLUS 1  
        SET dp at position zero TO 1  
        SET dp at position one TO 1  
        
        FOR index FROM 2 TO length OF s PLUS 1  
            IF character at position index MINUS 1 OF s NOT EQUALS '0'  
                INCREMENT dp at position index BY dp at position index MINUS 1  
            
            SET two_digit TO integer value OF substring OF s FROM position index MINUS 2 TO position index  
            IF two_digit GREATER THAN OR EQUAL TO 10 AND two_digit LESS THAN OR EQUAL TO 26  
                INCREMENT dp at position index BY dp at position index MINUS 2  
        
        RETURN value AT last position OF dp  
    END FUNCTION  
END CLASS
