CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF LENGTH OF edges NOT EQUALS n MINUS 1  
            RETURN False  
        
        SET adj_list TO a mapping from each integer from 0 TO n MINUS 1 TO an empty list  
        
        FOR each pair a, b IN edges  
            APPEND b TO adj_list at key a  
            APPEND a TO adj_list at key b  
        
        SET visited TO an empty set  
        
        FUNCTION dfs(node)  
            IF node IS IN visited  
                RETURN  
            ADD node TO visited  
            FOR each neighbor IN adj_list at key node  
                CALL dfs WITH neighbor  
        END FUNCTION  
        
        CALL dfs WITH 0  
        
        RETURN whether LENGTH OF visited EQUALS n  
    END FUNCTION  
END CLASS
