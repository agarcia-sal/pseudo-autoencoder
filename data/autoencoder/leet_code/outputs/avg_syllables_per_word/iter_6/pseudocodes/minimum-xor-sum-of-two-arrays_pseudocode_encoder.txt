CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO LENGTH OF nums1  
        SET dp TO list with size EQUAL TO two RAISED TO n FILLED WITH infinity  
        SET dp AT position zero TO zero  
        
        FOR mask FROM zero TO two RAISED TO n MINUS one  
            SET bit_count TO number of set bits IN mask  
            FOR j FROM zero TO n MINUS one  
                IF bit j IN mask IS SET  
                    SET dp AT mask TO the minimum OF dp AT mask AND (dp AT mask WITH bit j UNSET PLUS bitwise XOR OF nums1 AT bit_count MINUS one AND nums2 AT j)  
        
        RETURN dp AT position two RAISED TO n MINUS one  
    END FUNCTION  
END CLASS
