CLASS Solution  
    FUNCTION compareVersion(version1, version2)  
        SET v1_revisions TO CALL split version1 BY period  
        SET v2_revisions TO CALL split version2 BY period  
        SET max_length TO GREATER OF LENGTH OF v1_revisions AND LENGTH OF v2_revisions  
        FOR index FROM 0 TO max_length MINUS 1  
            IF index LESS THAN LENGTH OF v1_revisions  
                SET v1 TO CALL convert v1_revisions AT index TO integer  
            ELSE  
                SET v1 TO 0  
            IF index LESS THAN LENGTH OF v2_revisions  
                SET v2 TO CALL convert v2_revisions AT index TO integer  
            ELSE  
                SET v2 TO 0  
            IF v1 LESS THAN v2  
                RETURN minus one  
            ELSE IF v1 GREATER THAN v2  
                RETURN one  
        RETURN zero  
    END FUNCTION  
END CLASS
