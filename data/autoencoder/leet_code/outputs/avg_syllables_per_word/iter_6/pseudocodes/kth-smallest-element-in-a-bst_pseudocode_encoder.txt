CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO CALL TreeNode WITH values AT POSITION zero  
    SET i TO one  
    INITIALIZE queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT POSITION i IS NOT None  
            SET node.left TO CALL TreeNode WITH values AT POSITION i  
            APPEND node.left TO queue  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND values AT POSITION i IS NOT None  
            SET node.right TO CALL TreeNode WITH values AT POSITION i  
            APPEND node.right TO queue  
        INCREMENT i BY one  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(root, k)  
        INITIALIZE stack TO empty list  
        SET current TO root  
        SET count TO zero  
        WHILE stack IS NOT EMPTY OR current IS NOT None  
            WHILE current IS NOT None  
                APPEND current TO stack  
                SET current TO current.left  
            SET current TO REMOVE LAST ELEMENT FROM stack  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN current.val  
            SET current TO current.right  
    END FUNCTION  
END CLASS
