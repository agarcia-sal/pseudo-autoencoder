CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO CALL Counter WITH nums  
        SET end_with TO CALL Counter WITH no arguments  
        
        FOR each num IN nums  
            IF num_count[num] EQUALS 0  
                CONTINUE  
            
            IF end_with[num MINUS 1] GREATER THAN 0  
                DECREMENT end_with[num MINUS 1] BY 1  
                INCREMENT end_with[num] BY 1  
            ELSE IF num_count[num PLUS 1] GREATER THAN 0 AND num_count[num PLUS 2] GREATER THAN 0  
                DECREMENT num_count[num PLUS 1] BY 1  
                DECREMENT num_count[num PLUS 2] BY 1  
                INCREMENT end_with[num PLUS 2] BY 1  
            ELSE  
                RETURN False  
            
            DECREMENT num_count[num] BY 1  
        
        RETURN True  
    END FUNCTION  
END CLASS
