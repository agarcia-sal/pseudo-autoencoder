CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        SET graph TO a mapping from each integer from 1 TO n TO an empty set  
        
        FOR each pair u, v IN edges  
            ADD v TO the set mapped by u in graph  
            ADD u TO the set mapped by v in graph  
        
        SET min_degree TO positive infinity  
        SET trio_found TO FALSE  
        
        FOR i FROM 1 TO n  
            FOR j FROM i PLUS 1 TO n  
                FOR k FROM j PLUS 1 TO n  
                    IF j IS IN the set mapped by i in graph AND k IS IN the set mapped by i in graph AND k IS IN the set mapped by j in graph  
                        SET trio_found TO TRUE  
                        SET trio_degree TO the sum of the sizes of the sets mapped by i, j, and k in graph MINUS 6  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
        
        IF trio_found IS TRUE  
            RETURN min_degree  
        ELSE  
            RETURN -1  
    END FUNCTION  
END CLASS
