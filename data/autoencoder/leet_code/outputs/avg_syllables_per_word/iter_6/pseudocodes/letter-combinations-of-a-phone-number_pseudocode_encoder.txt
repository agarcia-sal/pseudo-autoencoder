CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string THEN  
            RETURN empty list  
        END IF  

        SET phone_map TO mapping of digits TO corresponding letters  
            '2' TO 'abc'  
            '3' TO 'def'  
            '4' TO 'ghi'  
            '5' TO 'jkl'  
            '6' TO 'mno'  
            '7' TO 'pqrs'  
            '8' TO 'tuv'  
            '9' TO 'wxyz'  

        FUNCTION backtrack(combination, next_digits)  
            IF LENGTH OF next_digits EQUALS 0 THEN  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN phone_map MAPPED BY first element OF next_digits  
                    CALL backtrack WITH combination CONCATENATED WITH letter, substring of next_digits FROM second character TO end  
                END FOR  
            END IF  
        END FUNCTION  

        SET output TO empty list  
        CALL backtrack WITH empty string, digits  
        RETURN output  
    END FUNCTION  
END CLASS
