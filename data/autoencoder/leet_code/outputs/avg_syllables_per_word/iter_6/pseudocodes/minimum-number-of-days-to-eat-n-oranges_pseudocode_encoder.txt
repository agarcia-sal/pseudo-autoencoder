CLASS Solution  
    FUNCTION minDays(n)  
        DEFINE dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO 1  
                RETURN remaining  
            RETURN 1 PLUS MINIMUM OF  
                remaining MODULO 2 PLUS CALL dp WITH remaining DIVIDED BY 2,  
                remaining MODULO 3 PLUS CALL dp WITH remaining DIVIDED BY 3  
        END FUNCTION  
        RETURN CALL dp WITH n  
    END FUNCTION  
END CLASS
