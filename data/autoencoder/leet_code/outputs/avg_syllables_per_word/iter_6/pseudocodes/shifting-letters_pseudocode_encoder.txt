CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO 0  
        FOR index FROM length of shifts MINUS 1 DOWN TO 0  
            INCREMENT total_shifts BY shifts AT index  
            SET shifts AT index TO total_shifts  
        
        SET result TO empty list  
        FOR index FROM 0 TO length of s MINUS 1  
            SET new_character_code TO (character code of s AT index MINUS character code of 'a' PLUS shifts AT index) MODULO 26 PLUS character code of 'a'  
            SET new_character TO character represented by new_character_code  
            APPEND new_character TO result  
        
        RETURN concatenation of characters in result  
    END FUNCTION  
END CLASS
