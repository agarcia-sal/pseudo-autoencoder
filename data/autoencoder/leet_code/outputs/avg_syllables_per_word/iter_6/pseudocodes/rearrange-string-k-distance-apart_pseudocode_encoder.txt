CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO 1 THEN  
            RETURN s  
        END IF  
        
        SET freq TO CALL Counter WITH s  
        
        SET max_heap TO EMPTY LIST  
        FOR each character, count IN freq  
            APPEND tuple OF NEGATION OF count AND character TO max_heap  
        END FOR  
        CALL heapify ON max_heap  
        
        SET wait_queue TO EMPTY LIST  
        SET result TO EMPTY LIST  
        
        WHILE max_heap IS NOT EMPTY  
            SET count, char TO CALL heappop ON max_heap  
            APPEND char TO result  
            APPEND tuple OF count PLUS 1 AND char TO wait_queue  
            
            IF LENGTH OF wait_queue EQUALS k THEN  
                SET count, char TO FIRST ELEMENT OF wait_queue  
                REMOVE FIRST ELEMENT FROM wait_queue  
                IF count LESS THAN 0 THEN  
                    CALL heappush ON max_heap WITH tuple OF count AND char  
                END IF  
            END IF  
        END WHILE  
        
        IF LENGTH OF result NOT EQUALS LENGTH OF s THEN  
            RETURN EMPTY STRING  
        END IF  
        
        RETURN CONCATENATION OF ELEMENTS IN result  
    END FUNCTION  
END CLASS
