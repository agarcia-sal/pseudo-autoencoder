CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO new dictionary with default value zero  
        FOR index FROM 0 TO LENGTH OF s MINUS 10  
            SET sequence TO substring of s FROM index TO index PLUS 10  
            INCREMENT sequence_count AT sequence BY 1  
        SET repeated_sequences TO empty list  
        FOR each sequence, count IN sequence_count  
            IF count GREATER THAN 1  
                APPEND sequence TO repeated_sequences  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
