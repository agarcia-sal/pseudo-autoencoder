CLASS Solution  
    FUNCTION dailyTemperatures(temperatures)  
        SET n TO LENGTH OF temperatures  
        INITIALIZE answer AS list OF n ZEROS  
        INITIALIZE stack AS empty list  
        
        FOR i FROM 0 TO n MINUS 1  
            WHILE stack IS NOT EMPTY AND temperatures AT i GREATER THAN temperatures AT index ON TOP OF stack  
                SET prev_index TO REMOVE last element FROM stack  
                SET answer AT prev_index TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
