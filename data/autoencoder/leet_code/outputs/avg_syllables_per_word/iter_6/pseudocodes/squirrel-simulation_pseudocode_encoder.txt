CLASS Solution  
    FUNCTION minDistance(height, width, tree, squirrel, nuts)  
        FUNCTION distance(p1, p2)  
            RETURN absolute value of p1 at index 0 MINUS p2 at index 0 PLUS absolute value of p1 at index 1 MINUS p2 at index 1  
        END FUNCTION  
        
        SET total_distance TO 0  
        SET max_save TO negative infinity  
        
        FOR each nut IN nuts  
            INCREMENT total_distance BY 2 MULTIPLIED BY CALL distance WITH nut AND tree  
            SET first_nut_distance TO CALL distance WITH squirrel AND nut PLUS CALL distance WITH nut AND tree  
            SET later_nut_distance TO 2 MULTIPLIED BY CALL distance WITH nut AND tree  
            SET save TO later_nut_distance MINUS first_nut_distance  
            IF save GREATER THAN max_save  
                SET max_save TO save  
            END IF  
        END FOR  
        
        RETURN total_distance MINUS max_save  
    END FUNCTION  
END CLASS
