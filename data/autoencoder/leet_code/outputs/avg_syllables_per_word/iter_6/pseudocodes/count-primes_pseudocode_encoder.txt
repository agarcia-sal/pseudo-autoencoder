CLASS Solution  
    FUNCTION countPrimes(n)  
        IF n LESS THAN OR EQUAL TO 2 THEN  
            RETURN 0  
        END IF  

        SET is_prime TO a list of TRUE values of length n  
        SET is_prime at position 0 TO FALSE  
        SET is_prime at position 1 TO FALSE  

        FOR start FROM 2 TO INTEGER SQUARE ROOT OF n PLUS 1  
            IF is_prime at position start IS TRUE THEN  
                FOR multiple FROM start MULTIPLIED BY start TO n MINUS 1 STEP start  
                    SET is_prime at position multiple TO FALSE  
                END FOR  
            END IF  
        END FOR  

        SET prime_count TO sum of all TRUE values in is_prime  
        RETURN prime_count  
    END FUNCTION  
END CLASS
