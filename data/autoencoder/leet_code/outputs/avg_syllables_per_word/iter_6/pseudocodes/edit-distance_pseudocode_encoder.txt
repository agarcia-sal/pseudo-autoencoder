CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO LENGTH OF word1  
        SET n TO LENGTH OF word2  
          
        CALL create_2d_list WITH (m PLUS 1, n PLUS 1) RETURNING dp  
          
        FOR i FROM 0 TO m  
            SET dp AT i AND 0 TO i  
        FOR j FROM 0 TO n  
            SET dp AT 0 AND j TO j  
          
        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF CHARACTER AT word1 AT (i MINUS 1) EQUALS CHARACTER AT word2 AT (j MINUS 1)  
                    SET dp AT i AND j TO dp AT (i MINUS 1) AND (j MINUS 1)  
                ELSE  
                    SET dp AT i AND j TO 1 PLUS MINIMUM OF dp AT (i MINUS 1) AND j, dp AT i AND (j MINUS 1), dp AT (i MINUS 1) AND (j MINUS 1)  
        RETURN dp AT m AND n  
    END FUNCTION  
      
    FUNCTION create_2d_list(rows, columns)  
        RETURN a two-dimensional list with rows number of lists each containing columns number of zeros  
    END FUNCTION  
END CLASS
