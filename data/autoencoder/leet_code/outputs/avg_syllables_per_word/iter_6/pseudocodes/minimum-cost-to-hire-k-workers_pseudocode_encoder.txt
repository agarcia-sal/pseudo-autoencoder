CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO sorted list of pairs (wage DIVIDED BY quality, quality) for each corresponding element in wage and quality  
        INITIALIZE max_heap TO empty list  
        SET total_quality TO 0  
        SET min_cost TO infinity  
        
        FOR each ratio, q IN workers  
            CALL heappush ON max_heap WITH negative q  
            INCREMENT total_quality BY q  
            
            IF length of max_heap GREATER THAN k THEN  
                SET total_quality TO total_quality PLUS heappop FROM max_heap  
            END IF  
            
            IF length of max_heap EQUALS k THEN  
                SET min_cost TO minimum of min_cost AND total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
