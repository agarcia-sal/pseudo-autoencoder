CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals IS EMPTY  
            RETURN 0  
        END IF  

        CALL intervals SORTED BY start time

        SET min_heap TO empty list

        FOR each meeting IN intervals  
            IF min_heap IS NOT EMPTY AND meeting start time GREATER THAN OR EQUAL TO minimum element IN min_heap  
                CALL remove minimum element FROM min_heap  
            END IF  

            CALL add meeting end time TO min_heap  
        END FOR  

        RETURN size OF min_heap  
    END FUNCTION  
END CLASS
