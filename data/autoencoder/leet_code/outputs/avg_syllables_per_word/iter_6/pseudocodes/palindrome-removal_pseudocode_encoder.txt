CLASS Solution  
    FUNCTION minimumMoves(arr)  
        SET n TO LENGTH OF arr  
        SET dp TO two-dimensional list of size n BY n initialized with zeros  
        
        FOR i FROM 0 TO n MINUS 1  
            SET dp at position [i][i] TO 1  
        
        FOR i FROM 0 TO n MINUS 2  
            IF element at arr[i] EQUALS element at arr[i PLUS 1]  
                SET dp at position [i][i PLUS 1] TO 1  
            ELSE  
                SET dp at position [i][i PLUS 1] TO 2  
        
        FOR length FROM 3 TO n  
            FOR i FROM 0 TO n MINUS length  
                SET j TO i PLUS length MINUS 1  
                IF element at arr[i] EQUALS element at arr[j]  
                    SET dp at position [i][j] TO dp at position [i PLUS 1][j MINUS 1]  
                ELSE  
                    SET dp at position [i][j] TO infinity  
                    FOR k FROM i TO j MINUS 1  
                        SET dp at position [i][j] TO minimum OF dp at position [i][j] AND sum of dp at position [i][k] PLUS dp at position [k PLUS 1][j]  
        
        RETURN dp at position [0][n MINUS 1]  
    END FUNCTION  
END CLASS
