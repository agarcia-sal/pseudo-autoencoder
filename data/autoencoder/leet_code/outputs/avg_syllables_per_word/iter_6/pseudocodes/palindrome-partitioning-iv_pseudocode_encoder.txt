CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS sub REVERSED  
        END FUNCTION  

        SET n TO LENGTH OF s  
        INITIALIZE is_pal AS two-dimensional list of FALSE with dimensions n BY n  

        FOR i FROM n MINUS 1 DOWN TO 0  
            FOR j FROM i TO n MINUS 1  
                IF s AT i EQUALS s AT j AND (j MINUS i LESS THAN OR EQUAL TO 1 OR is_pal AT i PLUS 1 AT j MINUS 1 IS TRUE)  
                    SET is_pal AT i AT j TO TRUE  
        
        FOR i FROM 1 TO n MINUS 2  
            IF is_pal AT 0 AT i MINUS 1 IS TRUE  
                FOR j FROM i TO n MINUS 2  
                    IF is_pal AT i AT j IS TRUE AND is_pal AT j PLUS 1 AT n MINUS 1 IS TRUE  
                        RETURN TRUE  
        RETURN FALSE  
    END FUNCTION  
END CLASS
