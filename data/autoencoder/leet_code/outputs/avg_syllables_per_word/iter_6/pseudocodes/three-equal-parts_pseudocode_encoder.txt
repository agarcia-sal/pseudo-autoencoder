CLASS Solution  
    FUNCTION threeEqualParts(arr)  
        SET total_ones TO sum of all elements in arr  
        
        IF total_ones EQUALS 0  
            RETURN list containing 0 and 2  
        END IF  
        
        IF total_ones MODULO 3 NOT EQUALS 0  
            RETURN list containing -1 and -1  
        END IF  
        
        SET part_ones TO total_ones DIVIDED BY 3  
        
        SET first_part_index TO 0  
        SET count_ones TO 0  
        FOR index FROM 0 TO LENGTH OF arr MINUS 1  
            IF arr AT index EQUALS 1  
                INCREMENT count_ones BY 1  
                IF count_ones EQUALS 1  
                    SET first_part_index TO index  
                END IF  
                IF count_ones EQUALS part_ones  
                    BREAK  
                END IF  
            END IF  
        END FOR  
        
        SET second_part_index TO 0  
        SET count_ones TO 0  
        FOR index FROM 0 TO LENGTH OF arr MINUS 1  
            IF arr AT index EQUALS 1  
                INCREMENT count_ones BY 1  
                IF count_ones EQUALS part_ones PLUS 1  
                    SET second_part_index TO index  
                    BREAK  
                END IF  
            END IF  
        END FOR  
        
        SET third_part_index TO 0  
        SET count_ones TO 0  
        FOR index FROM 0 TO LENGTH OF arr MINUS 1  
            IF arr AT index EQUALS 1  
                INCREMENT count_ones BY 1  
                IF count_ones EQUALS 2 MULTIPLIED BY part_ones PLUS 1  
                    SET third_part_index TO index  
                    BREAK  
                END IF  
            END IF  
        END FOR  
        
        SET n TO LENGTH OF arr MINUS third_part_index  
        IF first_part_index PLUS n LESS THAN OR EQUAL TO LENGTH OF arr AND  
           second_part_index PLUS n LESS THAN OR EQUAL TO LENGTH OF arr AND  
           SUBLIST of arr FROM first_part_index TO first_part_index PLUS n MINUS 1 EQUALS SUBLIST FROM second_part_index TO second_part_index PLUS n MINUS 1 AND  
           SUBLIST of arr FROM second_part_index TO second_part_index PLUS n MINUS 1 EQUALS SUBLIST FROM third_part_index TO END  
            RETURN list containing first_part_index PLUS n MINUS 1 AND second_part_index PLUS n  
        ELSE  
            RETURN list containing -1 AND -1  
        END IF  
    END FUNCTION  
END CLASS
