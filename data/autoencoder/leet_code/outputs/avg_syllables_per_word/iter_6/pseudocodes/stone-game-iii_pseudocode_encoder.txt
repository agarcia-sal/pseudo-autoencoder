CLASS Solution  
    FUNCTION stoneGameIII(stoneValue)  
        SET n TO LENGTH OF stoneValue  
        SET dp TO list of zeros with length n PLUS 1  
        
        FOR i FROM n MINUS 1 DOWNTO 0  
            SET dp at index i TO stoneValue at index i MINUS dp at index i PLUS 1  
            IF i PLUS 2 IS LESS THAN OR EQUAL TO n  
                SET dp at index i TO the maximum of dp at index i AND stoneValue at index i PLUS stoneValue at index i PLUS 1 MINUS dp at index i PLUS 2  
            IF i PLUS 3 IS LESS THAN OR EQUAL TO n  
                SET dp at index i TO the maximum of dp at index i AND stoneValue at index i PLUS stoneValue at index i PLUS 1 PLUS stoneValue at index i PLUS 2 MINUS dp at index i PLUS 3  
        
        IF dp at index 0 IS GREATER THAN 0  
            RETURN "Alice"  
        ELSE IF dp at index 0 IS LESS THAN 0  
            RETURN "Bob"  
        ELSE  
            RETURN "Tie"  
    END FUNCTION  
END CLASS
