CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET max_so_far TO empty list  
        SET current_max TO first element of arr  
        FOR each num IN arr  
            IF num GREATER THAN current_max  
                SET current_max TO num  
            APPEND current_max TO max_so_far  
        
        SET sorted_arr TO arr sorted in ascending order  
        SET chunks TO 0  
        SET max_chunk TO 0  
        
        FOR index FROM 0 TO length of arr MINUS 1  
            IF arr at index GREATER THAN max_chunk  
                SET max_chunk TO arr at index  
            IF max_chunk EQUALS sorted_arr at index  
                INCREMENT chunks BY 1  
        
        RETURN chunks  
    END FUNCTION  
END CLASS
