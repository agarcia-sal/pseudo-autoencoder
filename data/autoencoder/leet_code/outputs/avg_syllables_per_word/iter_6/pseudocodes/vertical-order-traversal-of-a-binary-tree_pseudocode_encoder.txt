CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS 0, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO CALL TreeNode WITH values AT INDEX 0  
    SET i TO 1  
    SET queue TO NEW empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.left TO CALL TreeNode WITH values AT INDEX i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.right TO CALL TreeNode WITH values AT INDEX i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left, q.left AND CALL is_same_tree WITH p.right, q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalTraversal(self, root)  
        IF root IS None  
            RETURN empty list  

        SET node_dict TO new dictionary with default as empty list  
        SET min_col TO 0  
        SET max_col TO 0  

        FUNCTION dfs(node, row, col)  
            IF node IS None  
                RETURN  
            APPEND (row, node.val) TO node_dict AT key col  
            SET min_col TO minimum OF min_col AND col  
            SET max_col TO maximum OF max_col AND col  
            CALL dfs WITH node.left, row PLUS 1, col MINUS 1  
            CALL dfs WITH node.right, row PLUS 1, col PLUS 1  
        END FUNCTION  

        CALL dfs WITH root, 0, 0  

        SET result TO empty list  
        FOR col FROM min_col TO max_col  
            SET sorted_list TO sorted values of node_dict AT col BY row THEN val  
            SET col_vals TO list of val FROM sorted_list  
            APPEND col_vals TO result  

        RETURN result  
    END FUNCTION  
END CLASS
