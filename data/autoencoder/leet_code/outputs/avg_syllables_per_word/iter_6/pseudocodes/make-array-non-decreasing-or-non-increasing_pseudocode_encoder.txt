CLASS Solution  
    FUNCTION convertArray(nums)  
        FUNCTION min_operations_to_non_decreasing(nums)  
            SET max_heap TO empty list  
            SET operations TO 0  
            FOR each num IN nums  
                IF max_heap IS NOT empty AND negative of first element in max_heap GREATER THAN num  
                    INCREMENT operations BY negative of value popped from max_heap MINUS num  
                    PUSH negative of num TO max_heap  
                END IF  
                PUSH negative of num TO max_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
      
        FUNCTION min_operations_to_non_increasing(nums)  
            SET min_heap TO empty list  
            SET operations TO 0  
            FOR each num IN nums  
                IF min_heap IS NOT empty AND first element in min_heap LESS THAN num  
                    INCREMENT operations BY num MINUS value popped from min_heap  
                    PUSH num TO min_heap  
                END IF  
                PUSH num TO min_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
      
        RETURN minimum of CALLING min_operations_to_non_decreasing WITH nums AND CALLING min_operations_to_non_increasing WITH nums  
    END FUNCTION  
END CLASS
