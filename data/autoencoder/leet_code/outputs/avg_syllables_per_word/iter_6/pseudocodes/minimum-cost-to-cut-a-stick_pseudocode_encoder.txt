CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO sorted list of cuts PLUS 0 PLUS n  
        
        FUNCTION dp(left, right)  
            IF right MINUS left IS LESS THAN OR EQUAL TO 1 THEN  
                RETURN 0  
            END IF  
            
            SET min_cost TO infinity  
            
            FOR i FROM left PLUS 1 TO right MINUS 1  
                SET cost TO cuts AT position right MINUS cuts AT position left PLUS CALL dp WITH left AND i PLUS CALL dp WITH i AND right  
                IF cost IS LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN CALL dp WITH 0 AND LENGTH OF cuts MINUS 1  
    END FUNCTION  
END CLASS
