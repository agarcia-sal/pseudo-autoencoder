CLASS Solution
    FUNCTION soupServings(n)
        IF n GREATER THAN 4800
            RETURN 1.0
        END IF

        SET n TO (n PLUS 24) DIVIDED BY 25

        SET memo TO empty dictionary

        FUNCTION dp(a, b)
            IF (a, b) IS IN memo
                RETURN memo AT (a, b)
            END IF

            IF a LESS THAN OR EQUAL TO 0 AND b LESS THAN OR EQUAL TO 0
                RETURN 0.5
            ELSE IF a LESS THAN OR EQUAL TO 0
                RETURN 1.0
            ELSE IF b LESS THAN OR EQUAL TO 0
                RETURN 0.0
            END IF

            SET prob TO 0.25 MULTIPLIED BY (
                CALL dp WITH (a MINUS 4, b)
                PLUS CALL dp WITH (a MINUS 3, b MINUS 1)
                PLUS CALL dp WITH (a MINUS 2, b MINUS 2)
                PLUS CALL dp WITH (a MINUS 1, b MINUS 3)
            )

            SET memo AT (a, b) TO prob
            RETURN prob
        END FUNCTION

        RETURN CALL dp WITH (n, n)
    END FUNCTION
END CLASS
