CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode WITH val AS first element of values  
    SET i TO 1  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN length of values AND values[i] IS NOT None  
            SET node.left TO new TreeNode WITH val as values[i]  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values[i] IS NOT None  
            SET node.right TO new TreeNode WITH val as values[i]  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST(root, low, high)  
        IF root IS None  
            RETURN None  

        IF root.val LESS THAN low  
            RETURN CALL trimBST WITH root.right, low, high  

        IF root.val GREATER THAN high  
            RETURN CALL trimBST WITH root.left, low, high  

        SET root.left TO CALL trimBST WITH root.left, low, high  
        SET root.right TO CALL trimBST WITH root.right, low, high  

        RETURN root  
    END FUNCTION  
END CLASS
