CLASS Solution  
    FUNCTION countAndSay(n)  
        IF n EQUALS 1  
            RETURN string "1"  
        END IF  
        
        SET current_sequence TO string "1"  
        
        FOR iteration FROM 1 TO n MINUS 1  
            SET next_sequence TO empty string  
            SET count TO 1  
            SET previous_char TO first character OF current_sequence  
            
            FOR index FROM 1 TO length OF current_sequence MINUS 1  
                IF character AT index OF current_sequence EQUALS previous_char  
                    INCREMENT count BY 1  
                ELSE  
                    SET next_sequence TO next_sequence PLUS string representation OF count PLUS previous_char  
                    SET previous_char TO character AT index OF current_sequence  
                    SET count TO 1  
                END IF  
            END FOR  
            
            SET next_sequence TO next_sequence PLUS string representation OF count PLUS previous_char  
            
            SET current_sequence TO next_sequence  
        END FOR  
        
        RETURN current_sequence  
    END FUNCTION  
END CLASS
