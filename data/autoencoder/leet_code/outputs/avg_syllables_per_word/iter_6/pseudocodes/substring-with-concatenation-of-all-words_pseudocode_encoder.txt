CLASS Solution  
    FUNCTION findSubstring(s, words)  
        IF s IS EMPTY OR words IS EMPTY OR words AT INDEX 0 IS EMPTY THEN  
            RETURN empty list  
        END IF  
        
        SET word_length TO LENGTH OF words AT INDEX 0  
        SET num_words TO LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO CALL Counter WITH words  
        SET result TO empty list  
        
        FOR i FROM 0 TO LENGTH OF s MINUS total_length PLUS 1  
            SET seen TO CALL Counter WITH empty collection  
            FOR j FROM i TO i PLUS total_length MINUS 1 STEP word_length  
                SET word TO SUBSTRING OF s FROM j TO j PLUS word_length  
                INCREMENT value for word IN seen BY 1  
                IF value for word IN seen GREATER THAN value for word IN word_count THEN  
                    BREAK inner FOR loop  
                END IF  
            END FOR  
            IF inner FOR loop COMPLETED WITHOUT BREAK THEN  
                APPEND i TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
