CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM 0 TO LENGTH OF strs MINUS 1  
            SET reversed_str TO reverse of strs at index  
            IF reversed_str GREATER THAN strs at index  
                SET strs at index TO reversed_str  
        
        SET doubled_strs TO concatenation of strs WITH strs  
        SET max_string TO empty string  
        
        FOR index FROM 0 TO LENGTH OF strs MINUS 1  
            FOR is_reversed IN [FALSE, TRUE]  
                IF is_reversed IS TRUE  
                    SET current TO reverse of strs at index  
                ELSE  
                    SET current TO strs at index  
                
                SET remaining TO concatenation of elements in doubled_strs from index PLUS 1 TO index PLUS LENGTH OF strs MINUS 1  
                
                FOR cut_position FROM 0 TO LENGTH OF current  
                    SET candidate TO concatenation of substring of current FROM cut_position TO end PLUS remaining PLUS substring of current FROM 0 TO cut_position  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
