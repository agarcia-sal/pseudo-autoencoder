CLASS Solution  
    FUNCTION subsets(nums)  
        IF nums IS EMPTY  
            RETURN list containing empty list  
        END IF  
        
        SET subsets_without_first TO CALL subsets WITH nums EXCLUDING FIRST ELEMENT  
        
        SET subsets_with_first TO empty list  
        FOR each subset IN subsets_without_first  
            SET new_subset TO list with nums FIRST ELEMENT APPENDED TO subset  
            APPEND new_subset TO subsets_with_first  
        END FOR  
        
        RETURN concatenation of subsets_without_first AND subsets_with_first  
    END FUNCTION  
END CLASS
