CLASS Solution  
    FUNCTION shortestDistance(maze, start, destination)  
        SET m TO LENGTH OF maze  
        SET n TO LENGTH OF maze AT INDEX zero  
        SET directions TO list containing four pairs: (one, zero), (negative one, zero), (zero, one), (zero, negative one)  
        CONVERT start TO tuple  
        CONVERT destination TO tuple  
        SET visited TO empty set  
        SET pq TO list containing pair of zero and start  

        WHILE pq IS NOT empty  
            CALL function to remove element with smallest distance FROM pq AND ASSIGN TO dist AND (x, y)  
            
            IF (x, y) EQUALS destination  
                RETURN dist  
            END IF  
            
            IF (x, y) IS IN visited  
                CONTINUE  
            END IF  
            
            ADD (x, y) TO visited  
            
            FOR each (dx, dy) IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND maze AT (new_x PLUS dx) AND (new_y PLUS dy) EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF (new_x, new_y) IS NOT IN visited  
                    ADD pair of dist PLUS count AND (new_x, new_y) TO pq  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
