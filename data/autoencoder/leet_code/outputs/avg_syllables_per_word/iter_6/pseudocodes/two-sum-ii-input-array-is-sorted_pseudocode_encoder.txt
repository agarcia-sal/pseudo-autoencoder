CLASS Solution  
    FUNCTION twoSum(numbers, target)  
        SET left TO 0  
        SET right TO LENGTH OF numbers MINUS 1  
        
        WHILE left LESS THAN right IS TRUE  
            SET current_sum TO VALUE AT numbers AT left PLUS VALUE AT numbers AT right  
            IF current_sum EQUALS target THEN  
                RETURN list CONTAINING left PLUS 1 AND right PLUS 1  
            ELSE IF current_sum LESS THAN target THEN  
                INCREMENT left BY 1  
            ELSE  
                DECREMENT right BY 1  
        END WHILE  
    END FUNCTION  
END CLASS
