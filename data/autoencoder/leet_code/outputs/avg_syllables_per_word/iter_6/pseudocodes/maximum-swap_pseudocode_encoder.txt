CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO list of characters of string representation of num  
        SET last TO dictionary mapping each digit as integer TO its last index in digits  
        
        FOR i FROM 0 TO LENGTH OF digits MINUS 1  
            SET d TO integer value of digits at index i  
            FOR d2 FROM 9 DOWN TO d PLUS 1  
                IF last CONTAINS d2 AND last at d2 GREATER THAN i  
                    SWAP digits at i WITH digits at last at d2  
                    RETURN integer parsed from concatenation of digits  
        RETURN num  
    END FUNCTION  
END CLASS
