CLASS Solution  
    FUNCTION hasPath(maze, start, destination)  
        SET directions TO list of coordinate pairs representing right, down, left, and up  
        SET m TO LENGTH OF maze  
        SET n TO LENGTH OF first element IN maze  
        SET visited TO empty set  
        
        FUNCTION dfs(x, y)  
            IF pair (x, y) IS IN visited THEN  
                RETURN False  
            IF list of x and y EQUALS destination THEN  
                RETURN True  
            ADD pair (x, y) TO visited  
            
            FOR each dx, dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND LESS THAN n AND maze value AT nx PLUS dx AND ny PLUS dy EQUALS zero  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                IF CALL dfs WITH nx AND ny RETURNS True THEN  
                    RETURN True  
            RETURN False  
        END FUNCTION  
        
        RETURN CALL dfs WITH start at position zero AND start at position one  
    END FUNCTION  
END CLASS
