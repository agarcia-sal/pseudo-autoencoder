CLASS Solution  
    FUNCTION maxSizeSlices(slices)  
        SET n TO LENGTH OF slices DIVIDED BY 3  
        
        FUNCTION dp(start, end, k)  
            IF k EQUALS 0 OR start GREATER THAN end THEN  
                RETURN 0  
            END IF  
            SET pick TO slices AT start PLUS CALL dp WITH start PLUS 2, end, k MINUS 1  
            SET skip TO CALL dp WITH start PLUS 1, end, k  
            RETURN THE GREATER OF pick AND skip  
        END FUNCTION  
        
        SET case1 TO CALL dp WITH 1, LENGTH OF slices MINUS 1, n  
        SET case2 TO CALL dp WITH 0, LENGTH OF slices MINUS 2, n  
        
        RETURN THE GREATER OF case1 AND case2  
    END FUNCTION  
END CLASS
