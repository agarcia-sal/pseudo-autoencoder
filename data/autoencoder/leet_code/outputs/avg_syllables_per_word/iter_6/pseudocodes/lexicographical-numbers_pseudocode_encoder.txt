CLASS Solution  
    FUNCTION lexicalOrder(n)  
        SET result TO empty list  
        
        FUNCTION dfs(current)  
            IF current GREATER THAN n  
                RETURN  
            APPEND current TO result  
            FOR i FROM 0 TO 9  
                SET next_num TO current MULTIPLIED BY 10 PLUS i  
                IF next_num GREATER THAN n  
                    RETURN  
                CALL dfs WITH next_num  
            END FOR  
        END FUNCTION  
        
        FOR i FROM 1 TO 9  
            CALL dfs WITH i  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
