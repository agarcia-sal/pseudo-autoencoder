CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS plus sign  
                        APPEND l PLUS r TO results  
                    ELSE IF operator EQUALS minus sign  
                        APPEND l MINUS r TO results  
                    ELSE IF operator EQUALS multiplication sign  
                        APPEND l MULTIPLIED BY r TO results  
            RETURN results  
        END FUNCTION  
        
        FUNCTION helper(sub_expr)  
            IF sub_expr consists entirely of digits  
                RETURN list containing integer value of sub_expr  
            
            SET results TO empty list  
            FOR index FROM zero TO LENGTH OF sub_expr MINUS one  
                IF character at sub_expr index EQUALS one of plus sign OR minus sign OR multiplication sign  
                    SET left_results TO CALL helper WITH substring of sub_expr from start TO index minus one  
                    SET right_results TO CALL helper WITH substring of sub_expr from index plus one TO end  
                    EXTEND results BY CALL compute WITH left_results, right_results, character at sub_expr index  
            RETURN results  
        END FUNCTION  
        
        RETURN CALL helper WITH expression  
    END FUNCTION  
END CLASS
