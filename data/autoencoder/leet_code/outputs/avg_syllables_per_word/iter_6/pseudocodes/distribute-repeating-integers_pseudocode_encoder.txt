CLASS Solution  
    FUNCTION canDistribute(nums, quantity)  
        SET count TO list of frequency values of elements in nums  
        SORT quantity in descending order  

        FUNCTION can_satisfy(index)  
            IF index EQUALS length of quantity THEN  
                RETURN True  
            FOR i FROM 0 TO length of count MINUS 1  
                IF count at i GREATER THAN OR EQUAL TO quantity at index THEN  
                    DECREMENT count at i BY quantity at index  
                    IF CALL can_satisfy WITH index PLUS 1 RETURNS True THEN  
                        RETURN True  
                    INCREMENT count at i BY quantity at index  
            RETURN False  
        END FUNCTION  

        RETURN CALL can_satisfy WITH 0  
    END FUNCTION  
END CLASS
