CLASS Solution  
    FUNCTION findMin(nums)  
        SET left TO 0  
        SET right TO LENGTH OF nums MINUS 1  
        
        WHILE left LESS THAN right  
            SET mid TO (left PLUS right) DIVIDED BY 2  
            
            IF nums AT mid GREATER THAN nums AT right  
                SET left TO mid PLUS 1  
            ELSE IF nums AT mid LESS THAN nums AT right  
                SET right TO mid  
            ELSE  
                DECREMENT right BY 1  
        
        RETURN nums AT left  
    END FUNCTION  
END CLASS
