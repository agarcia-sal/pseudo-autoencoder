CLASS ListNode  
    FUNCTION __init__(val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET head TO CALL ListNode WITH values AT FIRST POSITION AND None  
    SET p TO head  
    FOR each val IN values STARTING FROM SECOND ELEMENT  
        SET node TO CALL ListNode WITH val AND None  
        SET p.next TO node  
        SET p TO node  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN p1.val EQUALS p2.val AND CALL is_same_list WITH p1.next AND p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList(head)  
        IF head IS None OR head.next IS None OR head.next.next IS None  
            RETURN head  
        SET odd TO head  
        SET even TO head.next  
        SET even_head TO even  
        WHILE even IS NOT None AND even.next IS NOT None  
            SET odd.next TO even.next  
            SET odd TO odd.next  
            SET even.next TO odd.next  
            SET even TO even.next  
        SET odd.next TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
