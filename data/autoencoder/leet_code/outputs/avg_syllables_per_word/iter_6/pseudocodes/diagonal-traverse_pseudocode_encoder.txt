CLASS Solution
    FUNCTION findDiagonalOrder(mat)
        IF mat IS EMPTY OR first element of mat IS EMPTY THEN
            RETURN empty list
        END IF

        SET m TO length of mat
        SET n TO length of first element of mat
        SET result TO empty list
        SET r TO 0
        SET c TO 0
        SET direction TO 1

        FOR index FROM 0 TO m MULTIPLIED BY n MINUS 1
            APPEND mat element at row r and column c TO result
            IF direction EQUALS 1 THEN
                SET new_r TO r MINUS 1
                SET new_c TO c PLUS 1
            ELSE
                SET new_r TO r PLUS 1
                SET new_c TO c MINUS 1
            END IF

            IF new_r LESS THAN 0 OR new_r EQUALS m OR new_c LESS THAN 0 OR new_c EQUALS n THEN
                IF direction EQUALS 1 THEN
                    IF new_c EQUALS n THEN
                        SET new_r TO r PLUS 1
                        SET new_c TO c
                    ELSE
                        SET new_r TO 0
                        SET new_c TO c PLUS 1
                    END IF
                ELSE
                    IF new_r EQUALS m THEN
                        SET new_r TO r
                        SET new_c TO c PLUS 1
                    ELSE
                        SET new_r TO r PLUS 1
                        SET new_c TO 0
                    END IF
                END IF
                SET direction TO NEGATION OF direction
            END IF

            SET r TO new_r
            SET c TO new_c
        END FOR

        RETURN result
    END FUNCTION
END CLASS
