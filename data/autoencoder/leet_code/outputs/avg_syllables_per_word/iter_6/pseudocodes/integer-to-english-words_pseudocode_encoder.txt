CLASS Solution
    FUNCTION numberToWords(num)
        IF num EQUALS 0 THEN
            RETURN "Zero"
        END IF

        SET below_twenty TO list of strings: "", "One", "Two", "Three", "Four", "Five", "Six", "Seven", "Eight", "Nine", "Ten",
                                      "Eleven", "Twelve", "Thirteen", "Fourteen", "Fifteen", "Sixteen", "Seventeen", "Eighteen", "Nineteen"
        SET tens TO list of strings: "", "", "Twenty", "Thirty", "Forty", "Fifty", "Sixty", "Seventy", "Eighty", "Ninety"
        SET thousands TO list of strings: "", "Thousand", "Million", "Billion"
        
        FUNCTION helper(n)
            IF n EQUALS 0 THEN
                RETURN empty string
            ELSE IF n LESS THAN 20 THEN
                RETURN below_twenty element at index n PLUS a space
            ELSE IF n LESS THAN 100 THEN
                RETURN tens element at index (n DIVIDED BY 10) PLUS a space PLUS CALL helper WITH (n MODULO 10)
            ELSE
                RETURN below_twenty element at index (n DIVIDED BY 100) PLUS " Hundred " PLUS CALL helper WITH (n MODULO 100)
            END IF
        END FUNCTION
        
        SET result TO empty string
        SET i TO 0
        
        WHILE num GREATER THAN 0 DO
            IF (num MODULO 1000) NOT EQUALS 0 THEN
                SET result TO CALL helper WITH (num MODULO 1000) PLUS thousands element at index i PLUS a space PLUS result
            END IF
            SET num TO num DIVIDED BY 1000
            INCREMENT i BY 1
        END WHILE
        
        RETURN result with leading and trailing spaces removed
    END FUNCTION
END CLASS
