CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET max_so_far TO 0  
        SET chunks TO 0  
        
        FOR i FROM 0 TO LENGTH OF arr MINUS 1  
            SET max_so_far TO maximum of max_so_far AND arr AT i  
            IF max_so_far EQUALS i  
                INCREMENT chunks BY 1  
        
        RETURN chunks  
    END FUNCTION  
END CLASS
