CLASS TreeNode  
    FUNCTION __init__(self, val DEFAULTS TO 0, left DEFAULTS TO None, right DEFAULTS TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO NEW TreeNode WITH values AT INDEX 0  
    SET i TO 1  
    SET queue TO NEW empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.left TO NEW TreeNode WITH values AT INDEX i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT INDEX i IS NOT None  
            SET node.right TO NEW TreeNode WITH values AT INDEX i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree(self, root)  
        IF root IS None OR root.left IS None  
            RETURN root  
        SET new_root TO CALL self.upsideDownBinaryTree WITH root.left  
        SET root.left.left TO root.right  
        SET root.left.right TO root  
        SET root.left TO None  
        SET root.right TO None  
        RETURN new_root  
    END FUNCTION  
END CLASS
