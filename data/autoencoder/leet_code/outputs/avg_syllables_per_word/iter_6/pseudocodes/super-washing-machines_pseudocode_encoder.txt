CLASS Solution  
    FUNCTION findMinMoves(machines)  
        SET total_dresses TO sum of elements in machines  
        SET n TO length of machines  
        
        IF total_dresses MODULO n NOT EQUALS 0  
            RETURN negative one  
        
        SET target TO total_dresses DIVIDED BY n  
        SET left_sum TO zero  
        SET max_moves TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET left_excess TO left_sum MINUS target MULTIPLIED BY i  
            SET right_excess TO (total_dresses MINUS left_sum MINUS machines at i) MINUS target MULTIPLIED BY (n MINUS i MINUS one)  
            
            SET current_moves TO maximum of absolute value of left_excess, absolute value of right_excess, machines at i MINUS target  
            SET max_moves TO maximum of max_moves and current_moves  
            
            INCREMENT left_sum BY machines at i  
        
        RETURN max_moves  
    END FUNCTION  
END CLASS
