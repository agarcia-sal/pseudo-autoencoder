CLASS Solution  
    FUNCTION arrayNesting(nums)  
        SET visited TO list of False values with length equal to LENGTH OF nums  
        SET max_length TO 0  
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 1  
            IF visited at position i IS FALSE  
                SET current_length TO 0  
                SET k TO i  
                WHILE visited at position k IS FALSE  
                    SET visited at position k TO TRUE  
                    SET k TO value at position k in nums  
                    INCREMENT current_length BY 1  
                END WHILE  
                IF current_length GREATER THAN max_length  
                    SET max_length TO current_length  
                END IF  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
