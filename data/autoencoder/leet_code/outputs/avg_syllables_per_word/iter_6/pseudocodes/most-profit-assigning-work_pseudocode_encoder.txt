CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO list of pairs combining difficulty and profit  
        CALL sort ON jobs by difficulty  
        CALL sort ON worker by ability  
        
        SET max_profit TO 0  
        SET best_profit TO 0  
        SET job_index TO 0  
        SET n TO LENGTH OF jobs  
        
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND difficulty AT jobs[job_index] LESS THAN OR EQUAL TO ability  
                SET best_profit TO the greater value BETWEEN best_profit AND profit AT jobs[job_index]  
                INCREMENT job_index BY 1  
            INCREMENT max_profit BY best_profit  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
