CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO LENGTH OF passingFees  
        SET graph TO list of empty lists with size n  
        FOR each x, y, time IN edges  
            APPEND pair of y and time TO graph at index x  
            APPEND pair of x and time TO graph at index y  
          
        SET pq TO list containing tuple of passingFees at 0, 0, and 0  
        SET visited_time TO list of infinite values with size n  
        SET visited_time at 0 TO 0  
        
        WHILE pq IS NOT EMPTY  
            CALL heappop ON pq AND ASSIGN TO current_cost, current_city, current_time  
            
            IF current_city EQUALS n MINUS 1  
                RETURN current_cost  
            
            FOR each neighbor, travel_time IN graph at current_city  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS passingFees at neighbor  
                
                IF new_time LESS THAN OR EQUAL TO maxTime AND (new_time LESS THAN visited_time at neighbor OR new_cost LESS THAN current_cost)  
                    SET visited_time at neighbor TO new_time  
                    CALL heappush ON pq WITH tuple of new_cost, neighbor, new_time  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
