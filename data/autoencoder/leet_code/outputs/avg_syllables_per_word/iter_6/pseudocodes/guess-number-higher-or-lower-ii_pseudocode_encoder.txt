CLASS Solution  
    FUNCTION getMoneyAmount(n)  
        DEFINE FUNCTION dp(left, right)  
            IF left GREATER THAN OR EQUAL TO right THEN  
                RETURN 0  
            SET min_cost TO infinity  
            FOR pivot FROM left TO right DO  
                CALL dp WITH left AND pivot MINUS 1 RETURNING left_cost  
                CALL dp WITH pivot PLUS 1 AND right RETURNING right_cost  
                SET cost TO pivot PLUS maximum OF left_cost AND right_cost  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
            RETURN min_cost  
        END FUNCTION  
        RETURN CALL dp WITH 1 AND n  
    END FUNCTION  
END CLASS
