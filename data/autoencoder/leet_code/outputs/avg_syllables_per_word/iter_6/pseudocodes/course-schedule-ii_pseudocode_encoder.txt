CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO a mapping from each course in range from zero TO numCourses MINUS one TO an empty list  
        SET in_degrees TO a list of zeros with length equal to numCourses  
        
        FOR each pair dest, src IN prerequisites  
            APPEND dest TO adj_list at key src  
            INCREMENT in_degrees at index dest BY one  
        
        SET queue TO a collection of courses from zero TO numCourses MINUS one WHERE in_degrees at course EQUALS zero  
        SET topological_order TO empty list  
        
        WHILE queue IS NOT empty  
            SET current TO the first element removed from queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN adj_list at key current  
                DECREMENT in_degrees at index neighbor BY one  
                IF in_degrees at neighbor EQUALS zero  
                    APPEND neighbor TO queue  
        
        IF length of topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN empty list  
    END FUNCTION  
END CLASS
