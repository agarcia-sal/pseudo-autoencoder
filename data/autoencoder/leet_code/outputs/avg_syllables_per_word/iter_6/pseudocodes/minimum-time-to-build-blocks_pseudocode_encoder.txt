CLASS Solution  
    FUNCTION minBuildTime(blocks, split)  
        CALL heapify WITH blocks  
        WHILE LENGTH OF blocks GREATER THAN 1  
            SET first TO CALL heappop WITH blocks  
            SET second TO CALL heappop WITH blocks  
            SET combined_time TO split PLUS second  
            CALL heappush WITH blocks AND combined_time  
        RETURN blocks AT POSITION 0  
    END FUNCTION  
END CLASS
