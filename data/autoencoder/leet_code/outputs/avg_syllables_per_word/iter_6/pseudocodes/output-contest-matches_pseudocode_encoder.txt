CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO list of strings from 1 TO n  
        WHILE n GREATER THAN 1 IS TRUE  
            FOR i FROM 0 TO n DIVIDED BY 2 MINUS 1  
                SET teams AT i TO concatenation of open parenthesis PLUS teams AT i PLUS comma PLUS teams AT n MINUS 1 MINUS i PLUS close parenthesis  
            SET n TO n DIVIDED BY 2  
        RETURN teams AT 0  
    END FUNCTION  
END CLASS
