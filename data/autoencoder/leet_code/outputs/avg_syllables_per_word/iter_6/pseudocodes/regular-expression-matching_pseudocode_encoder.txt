CLASS Solution  
    FUNCTION isMatch(s, p)  
        CREATE dp AS two-dimensional list of False with dimensions (LENGTH OF s PLUS 1) BY (LENGTH OF p PLUS 1)  
        SET dp at position zero zero TO True  
        
        FOR j FROM 1 TO LENGTH OF p  
            IF character at position j MINUS 1 in p EQUALS '*'
                SET dp at position zero j TO dp at position zero j MINUS 2  
        
        FOR i FROM 1 TO LENGTH OF s  
            FOR j FROM 1 TO LENGTH OF p  
                IF character at position j MINUS 1 in p EQUALS character at position i MINUS 1 in s OR character at position j MINUS 1 in p EQUALS '.'  
                    SET dp at position i j TO dp at position i MINUS 1 j MINUS 1  
                ELSE IF character at position j MINUS 1 in p EQUALS '*'  
                    SET dp at position i j TO dp at position i j MINUS 2  
                    IF character at position j MINUS 2 in p EQUALS character at position i MINUS 1 in s OR character at position j MINUS 2 in p EQUALS '.'  
                        SET dp at position i j TO dp at position i j OR dp at position i MINUS 1 j  
        
        RETURN dp at position LENGTH OF s LENGTH OF p  
    END FUNCTION  
END CLASS
