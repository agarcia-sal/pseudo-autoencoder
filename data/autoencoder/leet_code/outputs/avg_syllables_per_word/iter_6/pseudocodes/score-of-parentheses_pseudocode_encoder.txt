CLASS Solution  
    FUNCTION scoreOfParentheses(s)  
        SET stack TO empty list  
        SET score TO 0  
        
        FOR each char IN s  
            IF char EQUALS '('  
                APPEND score TO stack  
                SET score TO 0  
            ELSE  
                SET score TO POP last element FROM stack PLUS (IF score EQUALS 0 THEN 1 ELSE 2 MULTIPLIED BY score)  
        RETURN score  
    END FUNCTION  
END CLASS
