CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO LENGTH OF jobDifficulty  
        
        IF n LESS THAN d  
            RETURN negative one  
        
        FUNCTION dp(i, d)  
            IF d EQUALS one  
                RETURN maximum value IN jobDifficulty FROM index i TO end  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR j FROM i TO n MINUS d  
                SET current_max TO maximum OF current_max AND jobDifficulty AT index j  
                SET candidate TO current_max PLUS CALL dp WITH arguments j PLUS one, d MINUS one  
                SET min_difficulty TO minimum OF min_difficulty AND candidate  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN CALL dp WITH arguments zero, d  
    END FUNCTION  
END CLASS
