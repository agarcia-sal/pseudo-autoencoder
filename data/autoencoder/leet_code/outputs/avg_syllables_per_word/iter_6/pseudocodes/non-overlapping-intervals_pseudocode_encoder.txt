CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals IS EMPTY  
            RETURN 0  
        
        CALL intervals SORTED BY each element's second value  
        
        SET end TO second value of first element in intervals  
        SET count TO 0  
        
        FOR index FROM 1 TO LENGTH OF intervals MINUS 1  
            IF first value of intervals at index GREATER THAN OR EQUAL TO end  
                SET end TO second value of intervals at index  
            ELSE  
                INCREMENT count BY 1  
        
        RETURN count  
    END FUNCTION  
END CLASS
