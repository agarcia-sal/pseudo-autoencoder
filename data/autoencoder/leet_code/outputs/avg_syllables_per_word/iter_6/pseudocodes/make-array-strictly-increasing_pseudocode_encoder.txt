CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL arr2.sort()  
        SET dp TO dictionary with key -1 and value 0  
        
        FOR each num IN arr1  
            SET new_dp TO empty dictionary  
            FOR each last_num, ops IN dp items  
                IF num GREATER THAN last_num  
                    SET new_dp[num] TO minimum of new_dp value for num OR infinity, AND ops  
                END IF  
                
                CALL bisect_search_right FUNCTION TO FIND idx OF smallest element in arr2 GREATER THAN last_num  
                IF idx LESS THAN LENGTH OF arr2  
                    SET new_dp value for arr2 at idx TO minimum of new_dp value for that element OR infinity, AND ops PLUS 1  
                END IF  
            END FOR  
            
            IF new_dp IS empty  
                RETURN -1  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN minimum value among dp values  
    END FUNCTION  
END CLASS
