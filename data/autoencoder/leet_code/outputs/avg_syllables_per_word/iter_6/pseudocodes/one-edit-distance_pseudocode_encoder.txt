CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF LENGTH OF s GREATER THAN LENGTH OF t  
            RETURN CALL isOneEditDistance WITH arguments t, s  
        
        SET diff TO LENGTH OF t MINUS LENGTH OF s  
        
        IF diff GREATER THAN 1  
            RETURN False  
        
        IF diff EQUALS 0  
            SET count TO 0  
            FOR each index FROM 0 TO LENGTH OF s MINUS 1  
                IF CHARACTER AT s index NOT EQUALS CHARACTER AT t index  
                    INCREMENT count BY 1  
            RETURN count EQUALS 1  
        
        ELSE  
            FOR each index FROM 0 TO LENGTH OF s MINUS 1  
                IF CHARACTER AT s index NOT EQUALS CHARACTER AT t index  
                    RETURN SUBSTRING OF s FROM index TO END EQUALS SUBSTRING OF t FROM index PLUS 1 TO END  
            RETURN True  
    END FUNCTION  
END CLASS
