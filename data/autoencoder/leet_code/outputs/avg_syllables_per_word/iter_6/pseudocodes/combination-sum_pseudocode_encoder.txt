CLASS Solution  
    FUNCTION combinationSum(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS 0  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN 0  
                RETURN  
            END IF  
            FOR i FROM start TO LENGTH OF candidates MINUS 1  
                CALL backtrack WITH i, target MINUS candidates AT i, path PLUS candidates AT i  
            END FOR  
        END FUNCTION  

        SET result TO empty list  
        CALL backtrack WITH 0, target, empty list  
        RETURN result  
    END FUNCTION  
END CLASS
