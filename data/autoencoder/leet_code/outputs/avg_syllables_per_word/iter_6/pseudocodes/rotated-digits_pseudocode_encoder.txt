CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO collection of '2', '5', '6', '9'  
        SET same_digits TO collection of '0', '1', '8'  
        
        FUNCTION is_good_number(num)  
            SET num_str TO string representation of num  
            IF all digits IN num_str ARE IN union of change_digits AND same_digits THEN  
                IF any digit IN num_str IS IN change_digits THEN  
                    RETURN TRUE  
                END IF  
            END IF  
            RETURN FALSE  
        END FUNCTION  
        
        SET good_count TO 0  
        FOR i FROM 1 TO n  
            IF CALL is_good_number WITH i IS TRUE THEN  
                INCREMENT good_count BY 1  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
