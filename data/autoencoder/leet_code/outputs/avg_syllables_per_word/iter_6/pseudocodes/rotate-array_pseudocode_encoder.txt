CLASS Solution  
    FUNCTION rotate(nums, k)  
        SET n TO LENGTH OF nums  
        SET k TO k MODULO n  
        
        FUNCTION reverse(start, end)  
            WHILE start LESS THAN end IS TRUE  
                SWAP nums AT start WITH nums AT end  
                INCREMENT start BY 1  
                DECREMENT end BY 1  
            END WHILE  
        END FUNCTION  
        
        CALL reverse WITH 0 AND n MINUS 1  
        CALL reverse WITH 0 AND k MINUS 1  
        CALL reverse WITH k AND n MINUS 1  
        
    END FUNCTION  
END CLASS
