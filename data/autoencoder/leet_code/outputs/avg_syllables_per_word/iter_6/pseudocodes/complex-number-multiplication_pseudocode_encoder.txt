CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SPLIT num INTO parts AROUND the plus sign  
            SET real TO the integer conversion of the first part  
            SET imaginary TO the integer conversion of the second part AFTER REMOVING the last character  
            RETURN real, imaginary  
        END FUNCTION  
  
        CALL parse_complex WITH num1 AND SET TO real1, imag1  
        CALL parse_complex WITH num2 AND SET TO real2, imag2  
  
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
  
        RETURN the string formed by concatenating real_part, plus sign, imaginary_part, and character 'i'  
    END FUNCTION  
END CLASS
