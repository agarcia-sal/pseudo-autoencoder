CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS LENGTH OF word THEN  
                IF count GREATER THAN 0 THEN  
                    APPEND string representation of count TO path  
                APPEND concatenation of elements in path TO result  
            ELSE  
                IF count GREATER THAN 0 THEN  
                    CALL backtrack WITH arguments index PLUS 1, path PLUS string representation of count PLUS word at position index, zero  
                ELSE  
                    CALL backtrack WITH arguments index PLUS 1, path PLUS word at position index, zero  
                CALL backtrack WITH arguments index PLUS 1, path, count PLUS 1  
            END IF  
        END FUNCTION  
  
        SET result TO empty list  
        CALL backtrack WITH arguments zero, empty list, zero  
        RETURN result  
    END FUNCTION  
END CLASS
