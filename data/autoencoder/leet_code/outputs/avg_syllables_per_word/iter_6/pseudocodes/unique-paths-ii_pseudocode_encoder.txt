CLASS Solution
    FUNCTION uniquePathsWithObstacles(obstacleGrid)
        IF obstacleGrid IS EMPTY OR obstacleGrid AT POSITION zero zero EQUALS 1
            RETURN 0
        SET m TO LENGTH OF obstacleGrid
        SET n TO LENGTH OF obstacleGrid AT POSITION zero
        INITIALIZE dp AS TWO-DIMENSIONAL LIST OF SIZE m BY n WITH ALL ZEROES
        
        SET dp AT POSITION zero zero TO 1
        
        FOR j FROM 1 TO n MINUS 1
            IF obstacleGrid AT POSITION zero j EQUALS 0
                SET dp AT POSITION zero j TO dp AT POSITION zero j MINUS 1
        
        FOR i FROM 1 TO m MINUS 1
            IF obstacleGrid AT POSITION i zero EQUALS 0
                SET dp AT POSITION i zero TO dp AT POSITION i MINUS 1 zero
        
        FOR i FROM 1 TO m MINUS 1
            FOR j FROM 1 TO n MINUS 1
                IF obstacleGrid AT POSITION i j EQUALS 0
                    SET dp AT POSITION i j TO dp AT POSITION i MINUS 1 j PLUS dp AT POSITION i j MINUS 1
        
        RETURN dp AT POSITION m MINUS 1 n MINUS 1
    END FUNCTION
END CLASS
