CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO 1  
        SET count TO 9  
        SET start TO 1  
        
        WHILE n GREATER THAN length MULTIPLIED BY count  
            DECREMENT n BY length MULTIPLIED BY count  
            INCREMENT length BY 1  
            MULTIPLY count BY 10  
            MULTIPLY start BY 10  
        
        INCREMENT start BY (n MINUS 1) DIVIDED BY length  
        SET digit_string TO CONVERT start TO STRING  
        SET digit_index TO (n MINUS 1) MODULO length  
        SET digit_character TO CHARACTER AT digit_index IN digit_string  
        RETURN CONVERT digit_character TO INTEGER  
    END FUNCTION  
END CLASS
