CLASS Solution  
    FUNCTION waysToDistribute(n, k)  
        SET MOD TO ten raised to the power nine PLUS seven  
        INITIALIZE dp AS two-dimensional list WITH dimensions (n PLUS one) BY (k PLUS one) FILLED WITH zeroes  
        SET dp at zero zero TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET dp at i j TO (j MULTIPLIED BY dp at (i MINUS one) j) PLUS dp at (i MINUS one) (j MINUS one) MODULO MOD  
        RETURN dp at n k  
    END FUNCTION  
END CLASS
