CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        CALL intervals SORTED BY end OF interval AND BY start IN DESCENDING ORDER WHEN ends ARE EQUAL  
        
        SET first TO end OF first interval MINUS 1  
        SET second TO end OF first interval  
        SET count TO 2  
        
        FOR each interval FROM second interval TO last interval  
            SET start TO start OF interval  
            SET end TO end OF interval  
            
            IF start GREATER THAN second THEN  
                SET first TO end MINUS 1  
                SET second TO end  
                INCREMENT count BY 2  
            ELSE IF start GREATER THAN first THEN  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY 1  
        
        RETURN count  
    END FUNCTION  
END CLASS
