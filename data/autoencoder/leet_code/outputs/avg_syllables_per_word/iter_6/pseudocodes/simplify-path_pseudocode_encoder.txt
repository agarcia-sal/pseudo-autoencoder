CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO result of splitting path by 'slash'  
        SET stack TO empty list  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS dot  
                CONTINUE to next iteration  
            ELSE IF component EQUALS double dot  
                IF stack IS NOT empty  
                    REMOVE last item FROM stack  
            ELSE  
                APPEND component TO stack  
        SET simplified_path TO string formed by concatenating slash WITH all items in stack joined by slash  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
