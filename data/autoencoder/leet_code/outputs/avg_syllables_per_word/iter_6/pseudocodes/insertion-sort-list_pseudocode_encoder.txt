CLASS ListNode  
    FUNCTION __init__(self, val EQUALS 0, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET head TO CALL ListNode WITH values AT FIRST POSITION  
    SET p TO head  
    FOR each val IN values FROM SECOND POSITION TO END  
        SET node TO CALL ListNode WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN p1.val EQUALS p2.val AND CALL is_same_list WITH p1.next, p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self, head)  
        IF head IS None OR head.next IS None  
            RETURN head  
        
        SET dummy TO CALL ListNode WITH -5001  
        SET dummy.next TO head  
        SET current TO head.next  
        SET head.next TO None  
        
        WHILE current IS NOT None  
            SET next_node TO current.next  
            SET prev TO dummy  
            WHILE prev.next IS NOT None AND prev.next.val LESS THAN current.val  
                SET prev TO prev.next  
            END WHILE  
            SET current.next TO prev.next  
            SET prev.next TO current  
            SET current TO next_node  
        END WHILE  
        
        RETURN dummy.next  
    END FUNCTION  
END CLASS
