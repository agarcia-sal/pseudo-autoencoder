CLASS Solution  
    FUNCTION originalDigits(s)  
        CALL function Counter WITH s RETURNING count  
        
        INITIALIZE digits AS list of ten zeros  
        
        SET digits AT index 0 TO count AT character 'z'  
        SET digits AT index 2 TO count AT character 'w'  
        SET digits AT index 4 TO count AT character 'u'  
        SET digits AT index 6 TO count AT character 'x'  
        SET digits AT index 8 TO count AT character 'g'  
        
        SET digits AT index 3 TO count AT character 'h' MINUS digits AT index 8  
        SET digits AT index 5 TO count AT character 'f' MINUS digits AT index 4  
        SET digits AT index 7 TO count AT character 's' MINUS digits AT index 6  
        SET digits AT index 9 TO count AT character 'i' MINUS digits AT index 5 MINUS digits AT index 6 MINUS digits AT index 8  
        SET digits AT index 1 TO count AT character 'o' MINUS digits AT index 0 MINUS digits AT index 2 MINUS digits AT index 4  
        
        INITIALIZE result AS empty list  
        FOR i FROM 0 TO 9  
            APPEND string representation OF i REPEATED digits AT index i TIMES TO result  
        
        RETURN concatenated string OF all elements IN result  
    END FUNCTION  
END CLASS
