CLASS Solution  
    FUNCTION searchRange(nums, target)  
        FUNCTION find_left(nums, target)  
            SET left TO 0  
            SET right TO LENGTH OF nums MINUS 1  
            WHILE left LESS THAN OR EQUAL TO right  
                SET mid TO left PLUS right DIVIDED BY 2 (integer division)  
                IF element AT mid IN nums LESS THAN target  
                    SET left TO mid PLUS 1  
                ELSE  
                    SET right TO mid MINUS 1  
            RETURN left  
        END FUNCTION  

        FUNCTION find_right(nums, target)  
            SET left TO 0  
            SET right TO LENGTH OF nums MINUS 1  
            WHILE left LESS THAN OR EQUAL TO right  
                SET mid TO left PLUS right DIVIDED BY 2 (integer division)  
                IF element AT mid IN nums LESS THAN OR EQUAL TO target  
                    SET left TO mid PLUS 1  
                ELSE  
                    SET right TO mid MINUS 1  
            RETURN right  
        END FUNCTION  

        SET left_index TO CALL find_left WITH nums, target  
        SET right_index TO CALL find_right WITH nums, target  

        IF left_index LESS THAN OR EQUAL TO right_index AND right_index GREATER THAN OR EQUAL TO 0 AND right_index LESS THAN LENGTH OF nums AND element AT left_index IN nums EQUALS target  
            RETURN list CONTAINING left_index AND right_index  
        ELSE  
            RETURN list CONTAINING MINUS 1 AND MINUS 1  
    END FUNCTION  
END CLASS
