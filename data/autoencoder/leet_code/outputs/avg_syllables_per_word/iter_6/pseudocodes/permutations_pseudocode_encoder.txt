CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF remaining IS EMPTY  
                APPEND path TO permutations  
                RETURN  
            FOR index FROM 0 TO LENGTH OF remaining MINUS 1  
                SET new_path TO path PLUS remaining AT index  
                SET new_remaining TO remaining FROM START TO index MINUS 1 CONCATENATED WITH remaining FROM index PLUS 1 TO END  
                CALL backtrack WITH new_path, new_remaining  
            END FOR  
        END FUNCTION  
      
        SET permutations TO EMPTY LIST  
        CALL backtrack WITH EMPTY LIST, nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
