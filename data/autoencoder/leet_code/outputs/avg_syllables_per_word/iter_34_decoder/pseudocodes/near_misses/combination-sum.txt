CLASS Solution  
    FUNCTION combinationSum(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                CALL backtrack WITH i AND target MINUS element at position i of candidates AND path PLUS element at position i of candidates  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH zero AND target AND an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
