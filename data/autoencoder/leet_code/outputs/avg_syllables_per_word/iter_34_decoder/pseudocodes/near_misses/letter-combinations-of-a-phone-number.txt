CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string THEN  
            RETURN empty list  
        END IF  
        
        SET phone_map TO mapping from each digit from two to nine TO corresponding letters as follows  
            digit two maps to letters a b c  
            digit three maps to letters d e f  
            digit four maps to letters g h i  
            digit five maps to letters j k l  
            digit six maps to letters m n o  
            digit seven maps to letters p q r s  
            digit eight maps to letters t u v  
            digit nine maps to letters w x y z  
        
        FUNCTION backtrack(combination, next_digits)  
            IF the length of next_digits EQUALS zero THEN  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the letters mapped by the first element of next_digits in phone_map  
                    CALL backtrack with concatenation of combination PLUS letter, and substring of next_digits from position two TO the end  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO empty list  
        CALL backtrack with empty string and digits  
        RETURN output  
    END FUNCTION  
END CLASS
