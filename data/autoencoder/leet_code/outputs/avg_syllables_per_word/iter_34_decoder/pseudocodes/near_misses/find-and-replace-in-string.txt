CLASS Solution  
    FUNCTION findReplaceString(s string parameter WITH name s indices parameter WITH name indices sources parameter WITH name sources targets parameter WITH name targets)  
        SET replacements TO list CREATED BY arranging TOGETHER elements FROM indices sources AND targets AND SORTED BY elements at first position  
        SET result TO empty list  
        SET prev_end TO zero  
        FOR each element IN replacements WITH names index source target  
            APPEND substring of s FROM position prev_end TO position index TO result  
            IF substring of s FROM position index TO position index PLUS LENGTH OF source EQUALS source THEN  
                APPEND target TO result  
            ELSE  
                APPEND substring of s FROM position index TO position index PLUS LENGTH OF source TO result  
            END IF  
            SET prev_end TO index PLUS LENGTH OF source  
        END FOR  
        APPEND substring of s FROM position prev_end TO end TO result  
        RETURN string FORMED BY JOINING all elements IN result  
    END FUNCTION  
END CLASS
