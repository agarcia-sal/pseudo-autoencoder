CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF the LENGTH OF nums LESS THAN two THEN  
            RETURN zero  
        END IF  
        
        FUNCTION countingSort(arr, exp)  
            SET n TO the LENGTH OF arr  
            SET output TO a list of zeros with the LENGTH n  
            SET count TO a list of zeros with the LENGTH ten  
            
            FOR index FROM zero TO n MINUS one  
                SET digit_index TO the INTEGER DIVISION of the element at position index of arr DIVIDED BY exp  
                INCREMENT the element at position digit_index MODULO ten of count BY one  
            END FOR  
            
            FOR index FROM one TO nine  
                INCREMENT the element at position index of count BY the element at position index MINUS one of count  
            END FOR  
            
            SET index TO n MINUS one  
            WHILE index GREATER THAN OR EQUAL TO zero  
                SET digit_index TO the INTEGER DIVISION of the element at position index of arr DIVIDED BY exp  
                SET the element at position the element at position digit_index MODULO ten of count MINUS one of output TO the element at position index of arr  
                DECREMENT the element at position digit_index MODULO ten of count BY one  
                DECREMENT index BY one  
            END WHILE  
            
            FOR index FROM zero TO n MINUS one  
                SET the element at position index of arr TO the element at position index of output  
            END FOR  
        END FUNCTION  
        
        FUNCTION radixSort(arr)  
            SET max_value TO the maximum element in arr  
            SET exp TO one  
            WHILE the INTEGER DIVISION of max_value BY exp GREATER THAN zero  
                CALL countingSort WITH arr, exp  
                SET exp TO exp MULTIPLIED BY ten  
            END WHILE  
        END FUNCTION  
        
        CALL radixSort WITH nums  
        
        SET max_diff TO zero  
        FOR index FROM one TO the LENGTH OF nums MINUS one  
            SET current_diff TO the element at position index of nums MINUS the element at position index MINUS one of nums  
            IF current_diff GREATER THAN max_diff THEN  
                SET max_diff TO current_diff  
            END IF  
        END FOR  
        
        RETURN max_diff  
    END FUNCTION  
END CLASS
