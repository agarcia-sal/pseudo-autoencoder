CLASS Solution  
    FUNCTION getPermutation(n k)  
        SET numbers TO list of all integers from one TO n  
        DECREMENT k BY one  
        SET result TO empty list  
        FOR i FROM n TO one DECREMENT BY one  
            SET fact TO factorial of i MINUS one  
            SET index TO k DIVIDED BY fact  
            APPEND element at position index of numbers CONVERTED TO string TO result  
            REMOVE element at position index FROM numbers  
            SET k TO k MODULO fact  
        END FOR  
        RETURN concatenation of all elements of result AS a single string  
    END FUNCTION  
END CLASS
