CLASS Solution  
    FUNCTION findMinHeightTrees(n parameter, edges parameter)  
        IF n parameter EQUALS one  
            RETURN the list containing the single element zero  
        END IF  
        
        SET graph TO a new mapping with default value as empty list  
        SET degree TO a list of zero values repeated n parameter times  
        
        FOR each pair of values u parameter and v parameter IN edges parameter  
            APPEND v parameter TO the list at key u parameter in graph  
            APPEND u parameter TO the list at key v parameter in graph  
            INCREMENT the value at position u parameter in degree BY one  
            INCREMENT the value at position v parameter in degree BY one  
        END FOR  
        
        SET leaves TO a new double ended queue containing elements i FOR each i FROM zero TO n parameter MINUS one WHERE the value at position i in degree EQUALS one  
        
        SET remaining_nodes TO n parameter  
        WHILE remaining_nodes GREATER THAN two  
            SET leaves_count TO the length of leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR each index FROM zero TO leaves_count MINUS one  
                SET leaf TO the element removed from the front of leaves  
                FOR each neighbor IN the list at key leaf in graph  
                    DECREMENT the value at position neighbor in degree BY one  
                    IF the value at position neighbor in degree EQUALS one  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN the list converted from leaves  
    END FUNCTION  
END CLASS
