CLASS Solution  
    FUNCTION shortestSuperstring(words)  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap(i, j)  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i of words AND the LENGTH OF element at position j of words DOWN TO one  
                IF the SUBSTRING consisting of the last k characters of element at position i of words EQUALS the SUBSTRING consisting of the first k characters of element at position j of words  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp(mask, i)  
            IF mask EQUALS the INTEGER ONE SHIFTED LEFT BY n MINUS one  
                RETURN a TUPLE of zero and empty string  
            END IF  
            
            SET min_len TO positive infinity  
            SET best_path TO empty string  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND of mask AND integer one SHIFTED LEFT BY j EQUALS zero  
                    SET length_and_path TO the RESULT OF dp where first parameter is the BITWISE OR of mask AND integer one SHIFTED LEFT BY j AND second parameter j  
                    SET length TO the first element of length_and_path  
                    SET path TO the second element of length_and_path  
                    SET length TO length PLUS the LENGTH OF element at position j of words MINUS the RESULT OF overlap with parameters i and j  
                    IF length LESS THAN min_len  
                        SET min_len TO length  
                        SET best_path TO the SUBSTRING of element at position j of words starting from the position EQUAL TO the RESULT OF overlap with parameters i and j TO the END CONCATENATED WITH path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN a TUPLE of min_len and best_path  
        END FUNCTION  
        
        SET min_len TO positive infinity  
        SET shortest_path TO empty string  
        FOR i FROM zero TO n MINUS one  
            SET length_and_path TO the RESULT OF dp where first parameter is integer one SHIFTED LEFT BY i AND second parameter i  
            SET length TO the first element of length_and_path  
            SET path TO the second element of length_and_path  
            SET length TO length PLUS the LENGTH OF element at position i of words  
            IF length LESS THAN min_len  
                SET min_len TO length  
                SET shortest_path TO element at position i of words CONCATENATED WITH path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
