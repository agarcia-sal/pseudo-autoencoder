CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO an empty mapping that assigns an empty list for missing keys  
        FOR each pair u and v IN edges  
            APPEND v TO the list corresponding to u in graph  
            APPEND u TO the list corresponding to v in graph  
        END FOR  

        FUNCTION dfs(node, parent, path, visited)  
            IF node IS IN visited  
                SET cycle_start TO the position of node in path  
                RETURN the sublist of path FROM position cycle_start TO the end  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN the list corresponding to node in graph  
                IF neighbor IS NOT EQUAL TO parent  
                    SET cycle TO the result of calling dfs WITH neighbor, node, path, visited  
                    IF cycle IS NOT NONE  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE the last element FROM path  
            RETURN NONE  
        END FUNCTION  

        SET visited TO an empty set  
        SET cycle TO NONE  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited  
                SET cycle TO the result of calling dfs WITH node, negative one, an empty list, visited  
                IF cycle IS NOT NONE  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  

        SET distance TO a list of zeros with the length n  
        SET queue TO a double-ended queue initialized with cycle  
        SET visited TO a set initialized with cycle  

        SET level TO zero  
        WHILE the queue IS NOT empty  
            SET current_length TO the LENGTH of queue  
            FOR index FROM zero TO current_length MINUS one  
                REMOVE the element from the front of queue and ASSIGN it TO node  
                SET the element at position node of distance TO level  
                FOR each neighbor IN the list corresponding to node in graph  
                    IF neighbor IS NOT IN visited  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  

        RETURN distance  
    END FUNCTION  
END CLASS
