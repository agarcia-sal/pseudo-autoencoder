CLASS Solution  
    FUNCTION openLock(deadends, target)  
        SET deadends_set TO empty set  
        FOR each deadend IN deadends  
            ADD deadend TO deadends_set  
        END FOR  
        
        IF deadends_set CONTAINS state consisting of four zeros  
            RETURN negative one  
        END IF  
        
        IF target EQUALS state consisting of four zeros  
            RETURN zero  
        END IF  
        
        SET queue TO new empty double ended queue  
        APPEND pair consisting of state consisting of four zeros AND zero TO queue  
        SET visited TO empty set  
        ADD state consisting of four zeros TO visited  
        
        FUNCTION neighbors(state)  
            SET result TO empty list  
            FOR index FROM zero TO three  
                SET digit_string TO element at position index of state  
                SET digit TO digit_string CONVERTED TO integer  
                FOR move IN list consisting of negative one AND one  
                    SET new_digit TO digit PLUS move  
                    IF new_digit LESS THAN zero  
                        SET new_digit TO nine  
                    ELSE IF new_digit GREATER THAN nine  
                        SET new_digit TO zero  
                    END IF  
                    SET new_state TO concatenation of substring from position zero TO position index MINUS one of state  
                                     PLUS string representation of new_digit  
                                     PLUS substring from position index PLUS one TO end of state  
                    APPEND new_state TO result  
                END FOR  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        WHILE queue IS NOT empty  
            REMOVE first element FROM queue AND ASSIGN TO pair current_state AND steps  
            FOR each neighbor_state IN neighbors(current_state)  
                IF neighbor_state NOT IN visited AND neighbor_state NOT IN deadends_set  
                    IF neighbor_state EQUALS target  
                        RETURN steps PLUS one  
                    END IF  
                    ADD neighbor_state TO visited  
                    APPEND pair consisting of neighbor_state AND steps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
