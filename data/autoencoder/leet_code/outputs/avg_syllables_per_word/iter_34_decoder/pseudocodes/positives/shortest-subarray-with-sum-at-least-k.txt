CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO list containing zero as its only element  
        FOR each num IN nums  
            APPEND to prefix_sum the element at position one less than the LENGTH OF prefix_sum PLUS num  
        END FOR  
        ASSIGN reference TO empty double ended queue dq  
        SET min_length TO infinite value  
        FOR each i and current_sum IN enumeration of prefix_sum  
            WHILE dq is not empty AND current_sum MINUS element at position element at the front of dq of prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET min_length TO the smaller value BETWEEN min_length AND i MINUS removed element from the front of dq  
            END WHILE  
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO element at position element at the back of dq of prefix_sum  
                REMOVE element from the back of dq  
            END WHILE  
            APPEND i TO dq  
        END FOR  
        IF min_length IS NOT EQUAL TO infinite value  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
