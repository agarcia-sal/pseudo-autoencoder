CLASS Solution  
    FUNCTION removeKdigits(num parameter AS string, k parameter AS integer) RETURNS string  
        SET stack TO an empty list  
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND the element at the last position of stack GREATER THAN digit  
                REMOVE the element at the last position of stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        IF k GREATER THAN zero  
            SET final_stack TO the elements of stack from the first position TO the position that is the LENGTH OF stack MINUS k  
        ELSE  
            SET final_stack TO stack  
        END IF  
        SET result TO the string formed by concatenating all elements of final_stack WITH all leading zero characters REMOVED  
        IF the LENGTH OF result EQUALS zero  
            RETURN the string that represents zero  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
