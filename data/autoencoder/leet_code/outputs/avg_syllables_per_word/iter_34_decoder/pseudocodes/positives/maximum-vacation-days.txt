CLASS Solution  
    FUNCTION maxVacationDays(flights days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the FIRST element of days  
        
        SET dp TO a list of length n with all elements SET TO negative one  
        SET the element at position zero of dp TO zero  
        
        SET graph TO an empty mapping from each index from zero TO n MINUS one TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of flights AT position j EQUALS one  
                    APPEND j TO the list mapped from i in graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR week FROM zero TO k MINUS one  
            SET new_dp TO a list of length n with all elements SET TO negative one  
            FOR i FROM zero TO n MINUS one  
                IF element at position i of dp EQUALS negative one  
                    CONTINUE to next iteration  
                END IF  
                SET element at position i of new_dp TO the GREATER BETWEEN element at position i of new_dp AND element at position i of dp PLUS element at position i of days AT position week  
                FOR each j IN the list mapped from i in graph  
                    SET element at position j of new_dp TO the GREATER BETWEEN element at position j of new_dp AND element at position i of dp PLUS element at position j of days AT position week  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        
        SET maximum_vacation_days TO the GREATEST VALUE in dp  
        RETURN maximum_vacation_days  
    END FUNCTION  
END CLASS
