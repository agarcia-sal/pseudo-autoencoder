CLASS Solution  
    FUNCTION removeBoxes(boxes)  
        FUNCTION dp(l, r, k, memo)  
            IF l GREATER THAN r THEN  
                RETURN zero  
            END IF  
            IF the tuple of l r k IS IN memo THEN  
                RETURN the element at the tuple of l r k IN memo  
            END IF  
            WHILE r GREATER THAN l AND the element at position r of boxes EQUALS the element at position r MINUS one of boxes DO  
                DECREMENT r BY one  
                INCREMENT k BY one  
            END WHILE  
            SET res TO the result of dp WITH parameters l r MINUS one zero memo PLUS the product of the sum of k PLUS one AND the sum of k PLUS one  
            FOR i FROM l TO r MINUS one DO  
                IF the element at position i of boxes EQUALS the element at position r of boxes THEN  
                    SET res TO the maximum OF res AND the sum of dp WITH parameters l i k PLUS one memo AND dp WITH parameters i PLUS one r MINUS one zero memo  
                END IF  
            END FOR  
            SET the element at the tuple of l r k IN memo TO res  
            RETURN res  
        END FUNCTION  
        RETURN dp WITH parameters zero the length OF boxes MINUS one zero an empty dictionary  
    END FUNCTION  
END CLASS
