CLASS Solution  
    FUNCTION validUtf8(data)  
        FUNCTION get_byte_count(byte)  
            IF the value obtained by shifting byte right BY seven BITS EQUALS zero  
                RETURN one  
            ELSE IF the value obtained by shifting byte right BY five BITS EQUALS the binary representation of one one zero  
                RETURN two  
            ELSE IF the value obtained by shifting byte right BY four BITS EQUALS the binary representation of one one one zero  
                RETURN three  
            ELSE IF the value obtained by shifting byte right BY three BITS EQUALS the binary representation of one one one one zero  
                RETURN four  
            ELSE  
                RETURN minus one  
            END IF  
        END FUNCTION  
      
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF data  
            SET byte_count TO the result of get_byte_count with the element at position i of data  
            IF byte_count EQUALS minus one  
                RETURN False  
            END IF  
            IF i PLUS byte_count GREATER THAN the LENGTH OF data  
                RETURN False  
            END IF  
            FOR j FROM i PLUS one TO i PLUS byte_count MINUS one  
                IF the value obtained by shifting the element at position j of data right BY six BITS NOT EQUALS the binary representation of one zero  
                    RETURN False  
                END IF  
            END FOR  
            INCREMENT i BY byte_count  
        END WHILE  
        RETURN True  
    END FUNCTION  
END CLASS
