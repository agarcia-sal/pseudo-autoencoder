CLASS Solution  
    FUNCTION findMaxValueOfEquation(points, k)  
        SET max_value TO negative infinity  
        ASSIGN reference TO a new empty double ended queue named deque_  
        
        FOR each pair of values named xj and yj IN points  
            WHILE deque_ is not empty AND the value of xj MINUS the second element at the front position of deque_ is GREATER THAN k  
                REMOVE the front element FROM deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO the greater value BETWEEN max_value AND the sum of yj PLUS xj PLUS the first element at the front position of deque_  
            END IF  
            
            WHILE deque_ is not empty AND the difference of yj MINUS xj is GREATER THAN OR EQUAL TO the first element at the back position of deque_  
                REMOVE the back element FROM deque_  
            END WHILE  
            
            APPEND the pair consisting of the difference yj MINUS xj AND the value xj TO deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
