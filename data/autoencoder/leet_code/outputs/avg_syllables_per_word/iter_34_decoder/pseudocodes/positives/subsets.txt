CLASS Solution  
    FUNCTION subsets(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN a LIST containing an EMPTY LIST  
        END IF  
        SET subsets_without_first TO the result of CALLING subsets with the LIST of elements of nums FROM position two TO the end  
        SET subsets_with_first TO an EMPTY LIST  
        FOR each subset IN subsets_without_first  
            SET new_subset TO a LIST with the element at position one of nums FOLLOWED BY the elements of subset  
            APPEND new_subset TO subsets_with_first  
        END FOR  
        SET combined_subsets TO the concatenation of subsets_without_first FOLLOWED BY subsets_with_first  
        RETURN combined_subsets  
    END FUNCTION  
END CLASS
