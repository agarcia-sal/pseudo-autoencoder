CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator_variable TO an iterator over the characters of s  
            RETURN the result of testing for each character in word whether this character appears in the remaining elements of iterator_variable in order  
        END FUNCTION  
  
        SORT the dictionary by ordering the words first by the negative value of their length so that longer words come before shorter ones AND if two words have the same length by the natural alphabetical order from the beginning of the words  
  
        FOR each word IN the dictionary  
            IF the word is a subsequence of s according to is_subsequence function  
                RETURN the current word  
            END IF  
        END FOR  
  
        RETURN an empty string  
    END FUNCTION  
END CLASS
