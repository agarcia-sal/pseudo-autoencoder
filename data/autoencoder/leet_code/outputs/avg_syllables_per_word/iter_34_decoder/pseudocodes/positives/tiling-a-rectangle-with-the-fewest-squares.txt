CLASS Solution  
    FUNCTION tilingRectangle(n parameter, m parameter)  
        FUNCTION backtrack(heights parameter, moves parameter)  
            IF all elements of heights ARE EQUAL TO n parameter THEN  
                SET min_moves TO the smaller value BETWEEN min_moves AND moves parameter  
                RETURN  
            END IF  
            
            IF moves parameter IS GREATER THAN OR EQUAL TO min_moves THEN  
                RETURN  
            END IF  
            
            SET min_height TO the smallest value IN heights parameter  
            SET left TO the position of min_height IN heights parameter  
            SET right TO left  
            WHILE right IS LESS THAN m parameter AND element at position right of heights parameter IS EQUAL TO min_height DO  
                INCREMENT right BY one  
            END WHILE  
            
            FOR size FROM the difference BETWEEN right AND left DOWN TO one DO  
                IF for every i FROM zero TO size MINUS one the element at position left PLUS i of heights parameter PLUS size IS LESS THAN OR EQUAL TO n parameter THEN  
                    SET new_heights TO a copy OF heights parameter  
                    FOR i FROM zero TO size MINUS one DO  
                        INCREMENT element at position left PLUS i of new_heights BY size  
                    END FOR  
                    CALL backtrack WITH new_heights AND moves parameter PLUS one  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET min_moves TO positive infinity  
        CALL backtrack WITH a list of m parameter zeros AND zero  
        RETURN min_moves  
    END FUNCTION  
END CLASS
