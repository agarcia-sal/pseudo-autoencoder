CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO an empty DEFAULT DICTIONARY mapping to lists  
        FOR each i FROM zero TO n MINUS one  
            APPEND i TO the list at key element at position i of arr in value_indices  
        END FOR  
        
        SET queue TO an empty DOUBLE ENDED QUEUE  
        APPEND a tuple of zero AND zero TO queue  
        SET visited TO a set containing zero  
        
        WHILE queue is not empty  
            REMOVE the first element from queue AND ASSIGN its first item to current_index AND second item to steps  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO a list containing current_index PLUS one AND current_index MINUS one  
            APPEND all elements in the list at key element at position current_index of arr in value_indices TO next_indices  
            
            FOR each next_index IN next_indices  
                IF next_index GREATER THAN OR EQUAL TO zero AND next_index LESS THAN n AND next_index NOT IN visited  
                    ADD next_index TO visited  
                    APPEND a tuple of next_index AND steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list at key element at position current_index of arr in value_indices TO an empty list  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
