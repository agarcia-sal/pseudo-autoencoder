CLASS Solution  
    FUNCTION wiggleSort(nums)  
        SET less TO true  
        FOR i FROM zero TO the LENGTH OF nums MINUS one MINUS one  
            IF less EQUALS true  
                IF element at position i of nums GREATER THAN element at position i PLUS one of nums  
                    SET temporary TO element at position i of nums  
                    SET element at position i of nums TO element at position i PLUS one of nums  
                    SET element at position i PLUS one of nums TO temporary  
                END IF  
            ELSE  
                IF element at position i of nums LESS THAN element at position i PLUS one of nums  
                    SET temporary TO element at position i of nums  
                    SET element at position i of nums TO element at position i PLUS one of nums  
                    SET element at position i PLUS one of nums TO temporary  
                END IF  
            END IF  
            SET less TO NOT less  
        END FOR  
    END FUNCTION  
END CLASS
