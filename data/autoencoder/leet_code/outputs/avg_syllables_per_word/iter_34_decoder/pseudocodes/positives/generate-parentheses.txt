CLASS Solution  
    FUNCTION generateParenthesis WITH parameter n  
        FUNCTION backtrack WITH parameters s AS an empty string left AS zero right AS zero  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s CONCATENATED WITH an opening parenthesis left INCREMENTED BY one right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s CONCATENATED WITH a closing parenthesis left right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH no arguments  
        RETURN result  
    END FUNCTION  
END CLASS
