CLASS Solution  
    FUNCTION addBoldTag(s STRING, words LIST OF STRING) RETURNS STRING  
        IF the LENGTH OF words EQUALS zero THEN  
            RETURN s  
        END IF  
        
        SET n TO the LENGTH OF s  
        SET mask TO a list of boolean False values of length n  
        
        FOR each word IN words  
            SET start TO zero  
            WHILE start IS LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the POSITION OF word IN s STARTING FROM start  
                IF start EQUALS negative one THEN  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i OF mask TO True  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i IS LESS THAN n  
            IF the element at position i OF mask IS True THEN  
                APPEND the string indicating the start of a bold tag TO result  
                WHILE i IS LESS THAN n AND the element at position i OF mask IS True  
                    APPEND the element at position i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string indicating the end of a bold tag TO result  
            ELSE  
                APPEND the element at position i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
