CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO the list of characters representing the string form of n  
        SET length TO the number of elements in digits  
        SET marker TO length  
        FOR i FROM length MINUS one DOWN TO one  
            IF the element at position i of digits converted to number LESS THAN the element at position i MINUS one of digits converted to number  
                SET the element at position i MINUS one of digits TO the string form of the element at position i MINUS one of digits converted to number MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one  
            SET the element at position i of digits TO the character representing number nine  
        END FOR  
        SET result_string TO the concatenation of all elements of digits in order  
        SET result_number TO the numeric form of result_string  
        RETURN result_number  
    END FUNCTION  
END CLASS
