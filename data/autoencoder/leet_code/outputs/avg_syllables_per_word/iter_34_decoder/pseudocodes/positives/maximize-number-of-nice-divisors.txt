CLASS Solution  
    FUNCTION maxNiceDivisors(primeFactors)  
        SET MOD TO ten raised to the power of nine PLUS one  
        
        IF primeFactors LESS THAN OR EQUAL TO three  
            RETURN primeFactors  
        END IF  
        
        SET num_threes TO integer division of primeFactors BY three  
        SET remainder TO primeFactors MINUS num_threes MULTIPLIED BY three  
        
        IF remainder EQUALS zero  
            RETURN the result of three raised to the power of num_threes taken modulo MOD  
        ELSE IF remainder EQUALS one  
            RETURN the result of three raised to the power of num_threes MINUS one taken modulo MOD MULTIPLIED BY four taken modulo MOD  
        ELSE  
            RETURN the result of three raised to the power of num_threes taken modulo MOD MULTIPLIED BY two taken modulo MOD  
        END IF  
    END FUNCTION  
END CLASS
