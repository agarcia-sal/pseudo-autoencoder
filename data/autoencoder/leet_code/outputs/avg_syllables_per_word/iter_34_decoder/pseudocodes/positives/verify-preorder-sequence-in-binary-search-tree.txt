CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO empty list  
        SET lower_bound TO the smallest possible number  
        
        FOR each value IN preorder  
            IF value LESS THAN lower_bound  
                RETURN false  
            END IF  
            
            WHILE the stack is not empty AND value GREATER THAN element at the last position of the stack  
                SET lower_bound TO element at the last position of the stack  
                REMOVE the element at the last position FROM stack  
            END WHILE  
            
            APPEND value TO stack  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
