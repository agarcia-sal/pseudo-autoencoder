CLASS Solution  
    FUNCTION isMatch(s string p string) RETURNS boolean  
        SET dp TO a two dimensional list of boolean values all set to false with the number of rows equal to the length of s PLUS one and the number of columns equal to the length of p PLUS one  
        SET element at position zero at position zero of dp TO true  
        FOR j FROM one TO the LENGTH OF p  
            IF element at position j MINUS one of p EQUALS the character Asterisk  
                SET element at position zero at position j of dp TO element at position zero at position j MINUS two of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM one TO the LENGTH OF p  
                IF element at position j MINUS one of p EQUALS element at position i MINUS one of s OR element at position j MINUS one of p EQUALS the character Dot  
                    SET element at position i at position j of dp TO element at position i MINUS one at position j MINUS one of dp  
                ELSE IF element at position j MINUS one of p EQUALS the character Asterisk  
                    SET element at position i at position j of dp TO element at position i at position j MINUS two of dp  
                    IF element at position j MINUS two of p EQUALS element at position i MINUS one of s OR element at position j MINUS two of p EQUALS the character Dot  
                        SET element at position i at position j of dp TO element at position i at position j of dp OR element at position i MINUS one at position j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position the LENGTH OF s at position the LENGTH OF p of dp  
    END FUNCTION  
END CLASS
