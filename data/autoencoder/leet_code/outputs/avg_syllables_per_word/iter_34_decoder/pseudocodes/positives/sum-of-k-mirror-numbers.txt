CLASS Solution  
    FUNCTION kMirror WITH PARAMETERS k n  
        FUNCTION is_palindrome WITH PARAMETER s  
            RETURN s EQUALS reverse of s  
        END FUNCTION  
        
        FUNCTION to_base_k WITH PARAMETERS num k  
            IF num EQUALS zero THEN  
                RETURN string representing zero  
            END IF  
            SET digits TO empty list  
            WHILE num NOT EQUALS zero  
                APPEND remainder of num DIVIDED BY k converted to integer TO digits  
                SET num TO num DIVIDED BY k converted to integer by floor division  
            END WHILE  
            RETURN concatenation of strings converted from elements in reverse order of digits  
        END FUNCTION  
        
        FUNCTION generate_palindromes WITH PARAMETER length  
            IF length EQUALS one THEN  
                FOR i FROM one TO nine  
                    YIELD string representing i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO the power of integer division of (length MINUS one) BY two  
                SET end TO ten RAISED TO the power of integer division of (length PLUS one) BY two  
                FOR half FROM start TO end MINUS one  
                    SET half_str TO string representing half  
                    IF length MODULO two EQUALS zero THEN  
                        SET full TO concatenation of half_str AND reverse of half_str  
                    ELSE  
                        SET full TO concatenation of half_str AND substring of half_str from the position of the second last character to the first character in reverse order  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO empty list  
        SET length TO one  
        WHILE LENGTH OF k_mirror_numbers LESS THAN n  
            FOR each pal IN generate_palindromes WITH PARAMETER length  
                SET num TO integer converted from pal  
                IF is_palindrome WITH PARAMETER to_base_k WITH PARAMETERS num k RETURNS true THEN  
                    APPEND num TO k_mirror_numbers  
                    IF LENGTH OF k_mirror_numbers EQUALS n THEN  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN sum of elements in k_mirror_numbers  
    END FUNCTION  
END CLASS
