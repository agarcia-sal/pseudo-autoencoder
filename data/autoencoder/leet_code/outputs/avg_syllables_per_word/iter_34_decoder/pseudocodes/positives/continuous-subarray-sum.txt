CLASS Solution  
    FUNCTION checkSubarraySum(nums k)  
        SET remainder_map TO a dictionary mapping zero TO negative one  
        SET current_sum TO zero  
        FOR each index i AND each number num IN enumeration of nums  
            INCREMENT current_sum BY num  
            IF k NOT EQUALS zero  
                SET current_sum TO current_sum MODULO k  
            END IF  
            IF current_sum EXISTS AS A KEY IN remainder_map  
                IF i MINUS value at key current_sum in remainder_map GREATER THAN one  
                    RETURN true  
                END IF  
            ELSE  
                ASSIGN value at key current_sum in remainder_map TO i  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
