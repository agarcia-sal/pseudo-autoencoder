CLASS Solution  
    FUNCTION frogPosition(n, edges, t, target)  
        IF n EQUALS one  
            RETURN one  
        END IF  
        
        SET graph TO a new default dictionary mapping each key to an empty list  
        FOR each pair of values u AND v IN edges  
            APPEND v TO the list associated with u in graph  
            APPEND u TO the list associated with v in graph  
        END FOR  
        
        SET queue TO a new double ended queue containing a tuple of the integer one the floating point number one and the integer zero  
        SET visited TO a new set containing the integer one  
        
        WHILE queue is not empty  
            REMOVE AND RETURN the leftmost element of queue assigning it in order to current_node current_probability AND current_time  
            
            IF current_time EQUALS t OR current_node NOT EQUALS one AND the LENGTH OF the list associated with current_node in graph EQUALS one  
                IF current_node EQUALS target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE to the next iteration of the WHILE loop  
                END IF  
            END IF  
            
            IF current_node EQUALS one  
                SET possible_moves TO the LENGTH OF the list associated with current_node in graph  
            ELSE  
                SET possible_moves TO the LENGTH OF the list associated with current_node in graph MINUS one  
            END IF  
            
            FOR each neighbor IN the list associated with current_node in graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND a tuple of neighbor current_probability DIVIDED BY possible_moves AND current_time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN zero  
    END FUNCTION  
END CLASS
