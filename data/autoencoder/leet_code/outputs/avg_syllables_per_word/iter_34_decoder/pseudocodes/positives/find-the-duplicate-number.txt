CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count_of_values_less_than_or_equal_to_x TO zero  
            FOR each value IN nums  
                IF value LESS THAN OR EQUAL TO x  
                    INCREMENT count_of_values_less_than_or_equal_to_x BY one  
                END IF  
            END FOR  
            IF count_of_values_less_than_or_equal_to_x GREATER THAN x  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        SET left_boundary TO zero  
        SET right_boundary TO LENGTH OF nums MINUS one  
        WHILE left_boundary LESS THAN right_boundary  
            SET middle_point TO left_boundary PLUS right_boundary MINUS left_boundary DIVIDED BY two  
            IF f(middle_point) EQUALS true  
                SET right_boundary TO middle_point  
            ELSE  
                SET left_boundary TO middle_point PLUS one  
            END IF  
        END WHILE  
        RETURN left_boundary  
    END FUNCTION  
END CLASS
