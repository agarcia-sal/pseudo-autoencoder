CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO sorted list of the first element of each pair in flowers  
        SET end_times TO sorted list of the second element of each pair in flowers  
        SET result TO empty list  
        FOR each person IN people  
            SET started TO the count of elements in start_times that are less than or equal to person by binary searching the position to the right of person  
            SET ended TO the count of elements in end_times that are less than person by binary searching the position to the left of person  
            APPEND started MINUS ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
