CLASS Solution  
    FUNCTION getSum(a b)  
        SET MAX TO the maximum positive integer value for thirty two bits  
        SET mask TO the bit mask representing thirty two bits all set to one  
        WHILE b NOT EQUALS zero  
            SET carry TO the bitwise AND of a and b MULTIPLIED BY two  
            SET a TO the bitwise XOR of a and b ANDed with mask  
            SET b TO the bitwise AND of carry and mask  
        END WHILE  
        IF a LESS THAN OR EQUAL TO MAX  
            RETURN a  
        ELSE  
            RETURN the bitwise negation of the bitwise XOR of a and mask  
        END IF  
    END FUNCTION  
END CLASS
