CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the tuple of index AND current_sum IS a KEY IN memo  
                RETURN the value CORRESPONDING TO the tuple of index AND current_sum IN memo  
            END IF  
            SET positive TO the RESULT OF dfs WITH index INCREMENTED BY one AND current_sum INCREMENTED BY the element at position index OF nums AND memo  
            SET negative TO the RESULT OF dfs WITH index INCREMENTED BY one AND current_sum DECREMENTED BY the element at position index OF nums AND memo  
            SET the tuple of index AND current_sum IN memo TO positive PLUS negative  
            RETURN the value CORRESPONDING TO the tuple of index AND current_sum IN memo  
        END FUNCTION  
        RETURN the RESULT OF dfs WITH zero AS index AND zero AS current_sum AND an empty dictionary AS memo  
    END FUNCTION  
END CLASS
