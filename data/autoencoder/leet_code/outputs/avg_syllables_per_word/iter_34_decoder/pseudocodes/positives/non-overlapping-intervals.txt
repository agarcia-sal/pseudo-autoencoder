CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals EQUALS empty collection  
            RETURN zero  
        END IF  
        SORT intervals BY the end element at position two of each interval  
        SET end TO the end element at position two of the first interval in intervals  
        SET count TO zero  
        FOR index FROM two TO the LENGTH OF intervals  
            IF the start element at position one of the interval at index IN intervals IS GREATER THAN OR EQUAL TO end  
                SET end TO the end element at position two of the interval at index IN intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
