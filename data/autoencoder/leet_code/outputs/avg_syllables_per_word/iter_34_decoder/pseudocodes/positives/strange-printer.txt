CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            IF memo CONTAINS key consisting of i AND j THEN  
                RETURN value AT key consisting of i AND j IN memo  
            END IF  
            SET result TO CALL dp WITH i AND j MINUS one AND memo PLUS one  
            FOR k FROM i TO j MINUS one  
                IF character AT position k OF s EQUALS character AT position j OF s THEN  
                    SET temp TO CALL dp WITH i AND k MINUS one AND memo PLUS CALL dp WITH k AND j MINUS one AND memo  
                    IF temp LESS THAN result THEN  
                        SET result TO temp  
                    END IF  
                END IF  
            END FOR  
            SET value AT key consisting of i AND j IN memo TO result  
            RETURN result  
        END FUNCTION  
        RETURN CALL dp WITH zero AND the LENGTH OF s MINUS one AND an empty dictionary  
    END FUNCTION  
END CLASS
