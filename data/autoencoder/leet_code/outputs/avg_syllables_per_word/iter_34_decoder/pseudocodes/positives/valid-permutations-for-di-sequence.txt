CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO ten TO THE POWER OF nine PLUS seven  
        SET n TO the LENGTH OF s  
        CREATE dp AS a two dimensional list of zeroes with n PLUS one rows and n PLUS one columns  
        
        FOR j FROM zero TO n  
            SET element at position j of row zero of dp TO one  
        END FOR  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            IF the element at position i MINUS one of s EQUALS the character I THEN  
                FOR j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS the element at position j of row i MINUS one of dp MODULO MOD  
                    SET element at position j of row i of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO prefix_sum PLUS the element at position j PLUS one of row i MINUS one of dp MODULO MOD  
                    SET element at position j of row i of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO the SUM OF the elements at row n of dp MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
