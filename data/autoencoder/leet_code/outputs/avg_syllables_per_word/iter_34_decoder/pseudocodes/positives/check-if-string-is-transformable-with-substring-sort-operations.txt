CLASS Solution  
    FUNCTION isTransformable WITH PARAMETERS s AND t  
        CREATE a mapping from digit to a list of positions called pos  
        FOR each position i AND character c IN s  
            ADD position i TO the list corresponding to the digit represented by character c in pos  
        END FOR  
        FOR each position i AND character c IN t  
            SET digit TO the digit represented by character c  
            IF the list corresponding to digit in pos HAS NO elements THEN  
                RETURN false  
            END IF  
            FOR each smaller digit FROM zero TO digit MINUS one  
                IF the list corresponding to smaller digit in pos IS NOT empty AND the first element in that list IS LESS THAN the first element in the list for digit THEN  
                    RETURN false  
                END IF  
            END FOR  
            REMOVE the first element from the list corresponding to digit in pos  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
