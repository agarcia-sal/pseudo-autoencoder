CLASS Solution  
    FUNCTION spiralMatrixIII(rows rStart cols cStart)  
        SET directions TO list of pairs zero one one zero zero minus one minus one zero  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO empty list  
        SET x TO rStart  
        SET y TO cStart  
        WHILE the LENGTH OF visited LESS THAN rows MULTIPLIED BY cols  
            IF increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR each element FROM one TO steps  
                IF x GREATER THAN OR EQUAL TO zero AND x LESS THAN rows AND y GREATER THAN OR EQUAL TO zero AND y LESS THAN cols  
                    APPEND list containing x and y TO visited  
                END IF  
                SET dx TO element at position direction_index in directions first element  
                SET dy TO element at position direction_index in directions second element  
                SET x TO x PLUS dx  
                SET y TO y PLUS dy  
            END FOR  
            SET direction_index TO (direction_index PLUS one) MODULO four  
            INCREMENT increment BY one  
        END WHILE  
        RETURN visited  
    END FUNCTION  
END CLASS
