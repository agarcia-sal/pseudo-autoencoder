CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO empty list  
        FOR each c IN expression  
            IF c EQUALS character representing true OR c EQUALS character representing false OR c EQUALS character representing not OR c EQUALS character representing and OR c EQUALS character representing or  
                APPEND c TO stk  
            ELSE IF c EQUALS character representing closing parenthesis  
                SET count_true TO zero  
                SET count_false TO zero  
                WHILE the element at the last position of stk EQUALS character representing true OR the element at the last position of stk EQUALS character representing false  
                    IF the element at the last position of stk EQUALS character representing true  
                        INCREMENT count_true BY one  
                    END IF  
                    IF the element at the last position of stk EQUALS character representing false  
                        INCREMENT count_false BY one  
                    END IF  
                    REMOVE the element at the last position FROM stk  
                END WHILE  
                SET op TO the element at the last position of stk  
                REMOVE the element at the last position FROM stk  
                IF op EQUALS character representing not  
                    IF count_false GREATER THAN zero  
                        SET c TO character representing true  
                    ELSE  
                        SET c TO character representing false  
                    END IF  
                ELSE IF op EQUALS character representing and  
                    IF count_false GREATER THAN zero  
                        SET c TO character representing false  
                    ELSE  
                        SET c TO character representing true  
                    END IF  
                ELSE IF op EQUALS character representing or  
                    IF count_true GREATER THAN zero  
                        SET c TO character representing true  
                    ELSE  
                        SET c TO character representing false  
                    END IF  
                END IF  
                APPEND c TO stk  
            END IF  
        END FOR  
        RETURN the element at position zero of stk EQUALS character representing true  
    END FUNCTION  
END CLASS
