CLASS Solution  
    FUNCTION kSimilarity(s1 AS string s2 AS string) RETURNS integer  
        FUNCTION neighbors(s AS string)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND element at position i of s EQUALS element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF element at position j of s EQUALS element at position i of s2 AND element at position j of s NOT EQUALS element at position j of s2  
                    YIELD a string formed by concatenating substring from position zero TO position i MINUS one of s PLUS element at position j of s PLUS substring from position i PLUS one TO position j MINUS one of s PLUS element at position i of s PLUS substring from position j PLUS one TO the end of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET queue TO a double ended queue containing a tuple of s1 and zero  
        SET visited TO a set initialized with s1  
        
        WHILE queue is not empty  
            REMOVE the first element from queue and ASSIGN to variables s and swaps  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors(s)  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND a tuple of neighbor and swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
