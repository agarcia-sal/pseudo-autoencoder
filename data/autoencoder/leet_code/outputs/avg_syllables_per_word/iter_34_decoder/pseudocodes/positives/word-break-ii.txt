CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO the SET of all elements IN wordDict  
        SET memo TO an empty MAPPING FROM integers TO lists of strings  
        
        FUNCTION backtrack(start)  
            IF start IS PRESENT IN memo THEN  
                RETURN the VALUE ASSOCIATED WITH start IN memo  
            END IF  
            IF start EQUALS the LENGTH OF s THEN  
                RETURN a LIST CONTAINING an EMPTY STRING  
            END IF  
            
            SET res TO an EMPTY LIST  
            FOR end FROM start PLUS one TO the LENGTH OF s INCLUSIVE  
                SET word TO the SUBSTRING OF s FROM position start TO position end MINUS one  
                IF word IS PRESENT IN wordSet THEN  
                    FOR each sentence IN backtrack(end)  
                        IF sentence EQUALS an EMPTY STRING THEN  
                            APPEND word TO res  
                        ELSE  
                            APPEND word PLUS a SINGLE SPACE PLUS sentence TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            
            SET memo AT start TO res  
            RETURN res  
        END FUNCTION  
        
        RETURN backtrack WITH argument zero  
    END FUNCTION  
END CLASS
