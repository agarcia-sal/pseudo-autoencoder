CLASS Solution  
    FUNCTION wallsAndGates(rooms)  
        IF rooms EQUALS empty OR the element at position zero of rooms EQUALS empty THEN  
            RETURN  
        END IF  
        
        SET rows TO the LENGTH OF rooms  
        SET cols TO the LENGTH OF the element at position zero of rooms  
        ASSIGN reference TO an empty double ended queue named queue  
        
        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF the element at position r of rooms AT position c EQUALS zero THEN  
                    APPEND the tuple of r AND c TO queue  
                END IF  
            END FOR  
        END FOR  
        
        ASSIGN reference TO a list named directions CONTAINING the tuples of zero AND one, one AND zero, zero AND negative one, negative one AND zero  
        
        WHILE queue is NOT empty  
            REMOVE AND RETURN the leftmost tuple FROM queue AND UNPACK INTO r AND c  
            FOR each tuple in directions UNPACKED INTO dr AND dc  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF zero LESS THAN OR EQUAL TO nr AND nr LESS THAN rows AND zero LESS THAN OR EQUAL TO nc AND nc LESS THAN cols AND the element at position nr of rooms AT position nc EQUALS the integer two billion one hundred forty seven million four hundred eighty three thousand six hundred forty seven THEN  
                    SET the element at position nr of rooms AT position nc TO the element at position r of rooms AT position c PLUS one  
                    APPEND the tuple of nr AND nc TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
