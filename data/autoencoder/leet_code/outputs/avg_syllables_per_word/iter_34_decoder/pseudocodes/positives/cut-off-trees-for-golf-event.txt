CLASS Solution  
    FUNCTION cutOffTree(forest)  
        IF the LENGTH OF forest EQUALS zero OR the LENGTH OF element at position zero of forest EQUALS zero THEN  
            RETURN the negative of one  
        END IF  
          
        SET trees TO an empty list  
        FOR index i FROM zero TO the LENGTH OF forest MINUS one  
            FOR index j FROM zero TO the LENGTH OF element at position zero of forest MINUS one  
                IF the element at position i of forest at position j GREATER THAN one THEN  
                    APPEND a tuple consisting of the element at position i of forest at position j FOLLOWED BY i FOLLOWED BY j TO trees  
                END IF  
            END FOR  
        END FOR  
        SORT trees in ascending order  
          
        FUNCTION bfs(start, end)  
            IF start EQUALS end THEN  
                RETURN zero  
            END IF  
            SET m TO the LENGTH OF forest  
            SET n TO the LENGTH OF element at position zero of forest  
            SET queue TO a new deque containing start  
            SET visited TO a new set containing start  
            SET steps TO zero  
              
            WHILE the queue is not empty DO  
                FOR each element in the current queue size DO  
                    SET x TO the first element of the dequeued position  
                    SET y TO the second element of the dequeued position  
                    FOR each pair dx, dy in the list of pairs consisting of negative one and zero THEN one and zero THEN zero and negative one THEN zero and one DO  
                        SET nx TO x PLUS dx  
                        SET ny TO y PLUS dy  
                        IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the tuple consisting of nx and ny is not in visited AND the element at position nx of forest at position ny NOT EQUALS zero THEN  
                            IF the tuple consisting of nx and ny EQUALS end THEN  
                                RETURN steps PLUS one  
                            END IF  
                            ADD the tuple consisting of nx and ny TO visited  
                            APPEND the tuple consisting of nx and ny TO queue  
                        END IF  
                    END FOR  
                END FOR  
                INCREMENT steps BY one  
            END WHILE  
              
            RETURN the negative of one  
        END FUNCTION  
          
        SET x TO zero  
        SET y TO zero  
        SET total_steps TO zero  
          
        FOR each tuple consisting of height, tx, ty IN trees DO  
            SET steps TO bfs with parameters the tuple consisting of x and y and the tuple consisting of tx and ty  
            IF steps EQUALS the negative of one THEN  
                RETURN the negative of one  
            END IF  
            INCREMENT total_steps BY steps  
            SET x TO tx  
            SET y TO ty  
        END FOR  
          
        RETURN total_steps  
    END FUNCTION  
END CLASS
