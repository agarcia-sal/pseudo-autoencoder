CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS OF only DIGITS THEN  
                SET num TO the INTEGER VALUE OF block  
                IF num IS LESS THAN OR EQUAL TO two hundred fifty five AND the STRING REPRESENTATION OF num EQUALS block THEN  
                    RETURN TRUE  
                ELSE  
                    RETURN FALSE  
                END IF  
            ELSE  
                RETURN FALSE  
            END IF  
        END FUNCTION  
      
        FUNCTION is_valid_ipv6_block(block)  
            IF the LENGTH OF block IS GREATER THAN OR EQUAL TO one AND the LENGTH OF block IS LESS THAN OR EQUAL TO four THEN  
                TRY  
                    CONVERT block TO an INTEGER VALUE USING BASE sixteen  
                    RETURN TRUE  
                CATCH conversion FAILURE  
                    RETURN FALSE  
                END TRY  
            ELSE  
                RETURN FALSE  
            END IF  
        END FUNCTION  
      
        SET parts_ipv4 TO the LIST OF SUBSTRINGS OF queryIP SPLIT BY the DOT CHARACTER  
        SET parts_ipv6 TO the LIST OF SUBSTRINGS OF queryIP SPLIT BY the COLON CHARACTER  
      
        IF the LENGTH OF parts_ipv4 EQUALS four AND for EACH part IN parts_ipv4 the CALL TO is_valid_ipv4_block(part) RETURNS TRUE THEN  
            RETURN the STRING indicating Internet Protocol version four  
        ELSE IF the LENGTH OF parts_ipv6 EQUALS eight AND for EACH part IN parts_ipv6 the CALL TO is_valid_ipv6_block(part) RETURNS TRUE THEN  
            RETURN the STRING indicating Internet Protocol version six  
        ELSE  
            RETURN the STRING indicating neither  
        END IF  
    END FUNCTION  
END CLASS
