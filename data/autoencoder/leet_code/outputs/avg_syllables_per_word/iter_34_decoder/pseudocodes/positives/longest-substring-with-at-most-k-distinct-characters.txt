CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s parameter, k parameter)  
        IF k parameter EQUALS zero  
            RETURN zero  
        END IF  
        
        SET char_map TO empty mapping  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s parameter MINUS one  
            SET element at position right of s parameter IN char_map TO right  
            
            IF the LENGTH OF char_map GREATER THAN k parameter  
                SET leftmost_char TO the key in char_map with the minimum value  
                SET left TO the value corresponding to leftmost_char in char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET current_length TO right MINUS left PLUS one  
            IF current_length GREATER THAN max_length  
                SET max_length TO current_length  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
