CLASS Solution  
    FUNCTION maxJumps(arr parameter AS list OF integer, d parameter AS integer) RETURNS integer  
        FUNCTION dp(i parameter AS integer) RETURNS integer  
            SET max_jumps TO one  
            FOR x FROM one TO d parameter  
                IF the sum of i parameter AND x is LESS THAN the length of arr parameter THEN  
                    IF the element at position the sum of i parameter AND x of arr parameter is GREATER THAN OR EQUAL TO the element at position i parameter of arr parameter THEN  
                        BREAK  
                    END IF  
                    SET max_jumps TO the greater value BETWEEN max_jumps AND the sum of one AND the result of dp function called with the sum of i parameter AND x  
                END IF  
            END FOR  
            FOR x FROM one TO d parameter  
                IF the difference of i parameter MINUS x is GREATER THAN OR EQUAL TO zero THEN  
                    IF the element at position the difference of i parameter MINUS x of arr parameter is GREATER THAN OR EQUAL TO the element at position i parameter of arr parameter THEN  
                        BREAK  
                    END IF  
                    SET max_jumps TO the greater value BETWEEN max_jumps AND the sum of one AND the result of dp function called with the difference of i parameter MINUS x  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        RETURN the maximum value of dp function called with i parameter FOR i parameter FROM zero TO the difference between the length of arr parameter AND one  
    END FUNCTION  
END CLASS
