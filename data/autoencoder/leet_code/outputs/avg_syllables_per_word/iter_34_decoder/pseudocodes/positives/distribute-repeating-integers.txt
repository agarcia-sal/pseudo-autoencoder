CLASS Solution  
    FUNCTION canDistribute(nums PARAMETER, quantity PARAMETER)  
        SET count TO the list of frequencies of each element in nums  
        SORT quantity IN descending order  
        
        FUNCTION can_satisfy(index PARAMETER)  
            IF index EQUALS the length of quantity THEN  
                RETURN True  
            END IF  
            FOR i FROM zero TO the length of count MINUS one  
                IF the element at position i of count GREATER THAN OR EQUAL TO the element at position index of quantity THEN  
                    DECREMENT the element at position i of count BY the element at position index of quantity  
                    IF can_satisfy(index PLUS one) RETURNS True THEN  
                        RETURN True  
                    END IF  
                    INCREMENT the element at position i of count BY the element at position index of quantity  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN can_satisfy(zero)  
    END FUNCTION  
END CLASS
