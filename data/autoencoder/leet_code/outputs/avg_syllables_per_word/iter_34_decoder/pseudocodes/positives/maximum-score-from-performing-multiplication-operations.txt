CLASS Solution  
    FUNCTION maximumScore(nums LIST OF INTEGER multipliers LIST OF INTEGER) RETURNS INTEGER  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a LIST OF INTEGER OF SIZE m PLUS one FILLED WITH zero  
        FOR i FROM m MINUS one TO zero DECREMENT BY one  
            SET new_dp TO a LIST OF INTEGER OF SIZE m PLUS one FILLED WITH zero  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS i MINUS left MINUS MINUS left  
                SET mul TO the element at position i of multipliers  
                SET value_from_left TO mul MULTIPLIED BY the element at position left of nums PLUS the element at position left PLUS one of dp  
                SET value_from_right TO mul MULTIPLIED BY the element at position right of nums PLUS the element at position left of dp  
                IF value_from_left GREATER THAN value_from_right  
                    SET the element at position left of new_dp TO value_from_left  
                ELSE  
                    SET the element at position left of new_dp TO value_from_right  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN the element at position zero of dp  
    END FUNCTION  
END CLASS
