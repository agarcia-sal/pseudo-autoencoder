CLASS Solution  
    FUNCTION findMedianSortedArrays(nums1 nums2)  
        IF the LENGTH OF nums1 GREATER THAN the LENGTH OF nums2 THEN  
            SET temporary variable first TO nums1  
            SET nums1 TO nums2  
            SET nums2 TO first  
        END IF  
        
        SET x TO the LENGTH OF nums1  
        SET y TO the LENGTH OF nums2  
        SET low TO zero  
        SET high TO x  
        
        WHILE low LESS THAN OR EQUAL TO high  
            SET partitionX TO the INTEGER DIVISION OF the SUM of low PLUS high BY two  
            SET partitionY TO the INTEGER DIVISION OF the SUM of x PLUS y PLUS one BY two MINUS partitionX  
            
            IF partitionX EQUALS zero THEN  
                SET maxLeftX TO NEGATIVE_INFINITY  
            ELSE  
                SET maxLeftX TO the element at position partitionX MINUS one OF nums1  
            END IF  
            
            IF partitionX EQUALS x THEN  
                SET minRightX TO POSITIVE_INFINITY  
            ELSE  
                SET minRightX TO the element at position partitionX OF nums1  
            END IF  
            
            IF partitionY EQUALS zero THEN  
                SET maxLeftY TO NEGATIVE_INFINITY  
            ELSE  
                SET maxLeftY TO the element at position partitionY MINUS one OF nums2  
            END IF  
            
            IF partitionY EQUALS y THEN  
                SET minRightY TO POSITIVE_INFINITY  
            ELSE  
                SET minRightY TO the element at position partitionY OF nums2  
            END IF  
            
            IF maxLeftX LESS THAN OR EQUAL TO minRightY AND maxLeftY LESS THAN OR EQUAL TO minRightX THEN  
                IF the MODULO of the SUM of x PLUS y EQUALS zero THEN  
                    SET maximum_left TO maxLeftX  
                    IF maxLeftY GREATER THAN maximum_left THEN  
                        SET maximum_left TO maxLeftY  
                    END IF  
                    SET minimum_right TO minRightX  
                    IF minRightY LESS THAN minimum_right THEN  
                        SET minimum_right TO minRightY  
                    END IF  
                    RETURN the SUM of maximum_left PLUS minimum_right DIVIDED BY two  
                ELSE  
                    IF maxLeftX GREATER THAN maxLeftY THEN  
                        RETURN maxLeftX  
                    ELSE  
                        RETURN maxLeftY  
                    END IF  
                END IF  
            ELSE IF maxLeftX GREATER THAN minRightY THEN  
                SET high TO partitionX MINUS one  
            ELSE  
                SET low TO partitionX PLUS one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
