CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO a new list containing string conversions of each element in nums  
        
        FUNCTION compare(x, y)  
            IF the concatenation of x WITH y is GREATER THAN the concatenation of y WITH x  
                RETURN negative one  
            ELSE IF the concatenation of x WITH y is LESS THAN the concatenation of y WITH x  
                RETURN positive one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        CALL the sort method on nums_str USING the comparison function compare as the key  
        
        IF element at position zero of nums_str EQUALS the string zero  
            RETURN the string zero  
        END IF  
        
        RETURN the concatenation of all elements in nums_str in order  
    END FUNCTION  
END CLASS
