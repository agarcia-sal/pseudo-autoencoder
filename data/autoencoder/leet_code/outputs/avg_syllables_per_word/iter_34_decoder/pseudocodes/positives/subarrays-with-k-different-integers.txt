CLASS Solution  
    FUNCTION subarraysWithKDistinct(nums AS list OF integer, k AS integer) RETURNS integer  
        FUNCTION atMostKDistinct(t AS integer) RETURNS integer  
            SET count TO an empty mapping with default integer zero  
            SET left TO zero  
            SET result TO zero  
            FOR each right FROM zero TO the length of nums MINUS one  
                INCREMENT the value in count for the element at position right of nums BY one  
                WHILE the number of keys in count IS GREATER THAN t  
                    DECREMENT the value in count for the element at position left of nums BY one  
                    IF the value in count for the element at position left of nums EQUALS zero  
                        REMOVE the element at position left of nums FROM count  
                    END IF  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT result BY right MINUS left PLUS one  
            END FOR  
            RETURN result  
        END FUNCTION  
        RETURN the result of atMostKDistinct with parameter k MINUS the result of atMostKDistinct with parameter k MINUS one  
    END FUNCTION  
END CLASS
