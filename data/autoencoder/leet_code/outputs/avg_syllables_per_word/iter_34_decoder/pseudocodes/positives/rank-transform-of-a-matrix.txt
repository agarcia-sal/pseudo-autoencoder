CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO list of integers from zero TO size MINUS one  
    END FUNCTION  

    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u THEN  
            SET element at position u of self.parent TO the result of calling self.find with element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union(self, u, v)  
        SET root_u TO the result of calling self.find with u  
        SET root_v TO the result of calling self.find with v  
        IF root_u NOT EQUALS root_v THEN  
            SET element at position root_u of self.parent TO root_v  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION matrixRankTransform(self, matrix)  
        SET m TO length of matrix  
        SET n TO length of element at position zero of matrix  
        SET value_to_indices TO a new mapping with default of empty list  
          
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                APPEND tuple consisting of r and c TO the list at key element at position r of matrix AT position c of value_to_indices  
            END FOR  
        END FOR  
          
        SET row_rank TO a list of zeros of length m  
        SET col_rank TO a list of zeros of length n  
        SET result TO a list of lists consisting of zeros with m elements each containing n elements  
          
        FOR value IN the sorted list of keys in value_to_indices  
            ASSIGN uf TO a new UnionFind object initialized with m PLUS n  
            SET rank TO a new mapping with default of integer zero  
              
            FOR each tuple consisting of r and c IN the list at key value of value_to_indices  
                CALL uf.union with r AND c PLUS m  
            END FOR  
              
            FOR each tuple consisting of r and c IN the list at key value of value_to_indices  
                SET root TO the result of calling uf.find with r  
                SET element at key root of rank TO the maximum between element at key root of rank AND the maximum between element at position r of row_rank AND element at position c of col_rank PLUS one  
            END FOR  
              
            FOR each tuple consisting of r and c IN the list at key value of value_to_indices  
                SET root TO the result of calling uf.find with r  
                SET element at position r of result at position c TO element at key root of rank  
                SET element at position r of row_rank TO element at key root of rank  
                SET element at position c of col_rank TO element at key root of rank  
            END FOR  
        END FOR  
          
        RETURN result  
    END FUNCTION  
END CLASS
