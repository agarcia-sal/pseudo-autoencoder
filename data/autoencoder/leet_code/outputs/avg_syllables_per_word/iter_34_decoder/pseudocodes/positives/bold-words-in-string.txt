CLASS Solution
    FUNCTION boldWords(words, s)
        SET bold TO a list of Boolean false values with the length equal to the length of s
        FOR each word IN words
            SET start TO the position of the first occurrence of word in s
            WHILE start is not equal to negative one
                FOR i FROM start TO start PLUS the length of word MINUS one
                    SET element at position i of bold TO true
                END FOR
                SET start TO the position of the next occurrence of word in s starting from position start PLUS one
            END WHILE
        END FOR

        SET result TO an empty list
        SET i TO zero
        WHILE i is less than the length of s
            IF the element at position i of bold is true
                APPEND the string denoting the opening bold tag to result
                WHILE i is less than the length of s AND the element at position i of bold is true
                    APPEND the element at position i of s to result
                    INCREMENT i BY one
                END WHILE
                APPEND the string denoting the closing bold tag to result
            ELSE
                APPEND the element at position i of s to result
                INCREMENT i BY one
            END IF
        END WHILE

        RETURN the concatenation of all elements in result
    END FUNCTION
END CLASS
