CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF the collection of houses IS EMPTY THEN  
                RETURN zero  
            END IF  
            IF the LENGTH OF houses EQUALS one THEN  
                RETURN the element at position one of houses  
            END IF  
            SET dp TO a new list of zeros with the LENGTH OF houses  
            SET the element at position one of dp TO the element at position one of houses  
            SET the element at position two of dp TO the GREATER OF the element at position one of houses AND the element at position two of houses  
            FOR i FROM three TO the LENGTH OF houses  
                SET the element at position i of dp TO the GREATER OF the element at position i MINUS one of dp AND the SUM OF the element at position i MINUS two of dp PLUS the element at position i of houses  
            END FOR  
            RETURN the element at the last position of dp  
        END FUNCTION  
        IF the collection nums IS EMPTY THEN  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one THEN  
            RETURN the element at position one of nums  
        END IF  
        RETURN the GREATER OF the RESULT OF rob_linear with the SUBSEQUENCE of nums FROM position one TO the position BEFORE the last element AND the RESULT OF rob_linear with the SUBSEQUENCE of nums FROM the second element TO the last element  
    END FUNCTION  
END CLASS
