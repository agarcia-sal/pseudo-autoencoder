CLASS Solution  
    FUNCTION isValidPalindrome(s, k)  
        FUNCTION longest_palindromic_subseq(s)  
            SET n TO the LENGTH OF s  
            SET dp TO a two dimensional list of size n by n FILLED WITH zero  
            
            FOR index i FROM zero TO n MINUS one  
                SET element at position i at position i of dp TO one  
            END FOR  
            
            FOR length FROM two TO n  
                FOR index i FROM zero TO n MINUS length  
                    SET index j TO i PLUS length MINUS one  
                    IF element at position i of s EQUALS element at position j of s  
                        SET element at position i at position j of dp TO element at position i PLUS one at position j MINUS one of dp PLUS two  
                    ELSE  
                        SET element at position i at position j of dp TO the GREATER OF element at position i PLUS one at position j of dp AND element at position i at position j MINUS one of dp  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN element at position zero at position n MINUS one of dp  
        END FUNCTION  
        
        SET longest_palindrome_length TO longest_palindromic_subseq(s)  
        RETURN the LENGTH OF s MINUS longest_palindrome_length LESS THAN OR EQUAL TO k  
    END FUNCTION  
END CLASS
