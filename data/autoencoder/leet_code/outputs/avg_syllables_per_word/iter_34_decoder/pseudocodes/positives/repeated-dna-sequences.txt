CLASS Solution  
    FUNCTION findRepeatedDnaSequences WITH PARAMETER s  
        SET sequence_count TO an empty dictionary that defaults to zero for any new key  
        FOR index i FROM zero TO the length of s MINUS nine  
            SET sequence TO the substring from position i TO position i PLUS ten MINUS one of s  
            INCREMENT the value associated with sequence in sequence_count BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each key sequence AND each value count IN the items of sequence_count  
            IF count GREATER THAN one THEN  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
