CLASS Solution  
    FUNCTION getFactors WITH parameter n  
        FUNCTION backtrack WITH parameters start AND target  
            FOR i FROM start TO the INTEGER PART OF the SQUARE ROOT OF target PLUS one  
                IF the MODULO of target BY i EQUALS zero  
                    APPEND the CONCATENATION of path AND the LIST containing i AND the QUOTIENT of target DIVIDED BY i TO result  
                    APPEND i TO path  
                    CALL backtrack WITH i AND the QUOTIENT of target DIVIDED BY i  
                    REMOVE the LAST element FROM path  
                END IF  
            END FOR  
        END FUNCTION  
      
        SET result TO an empty list  
        SET path TO an empty list  
        CALL backtrack WITH two AND n  
        RETURN result  
    END FUNCTION  
END CLASS
