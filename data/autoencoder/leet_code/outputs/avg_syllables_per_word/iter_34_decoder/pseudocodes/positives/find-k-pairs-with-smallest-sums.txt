CLASS Solution  
    FUNCTION kSmallestPairs(nums1 nums2 k)  
        IF the LENGTH OF nums1 EQUALS zero OR the LENGTH OF nums2 EQUALS zero OR k LESS THAN OR EQUAL TO zero  
            RETURN an empty list  
        END IF  

        SET min_heap TO an empty list  

        FOR i FROM zero TO the MINIMUM OF k AND the LENGTH OF nums1 MINUS one  
            PUSH ONTO min_heap the tuple of the element at position i of nums1 PLUS the element at position zero of nums2 AND i AND zero  
        END FOR  

        SET result TO an empty list  

        WHILE min_heap is not empty AND the LENGTH OF result LESS THAN k  
            POP the smallest tuple from min_heap INTO underscore AND i AND j  
            APPEND the list containing the element at position i of nums1 AND the element at position j of nums2 TO result  
            IF j PLUS one LESS THAN the LENGTH OF nums2  
                PUSH ONTO min_heap the tuple of the element at position i of nums1 PLUS the element at position j PLUS one of nums2 AND i AND j PLUS one  
            END IF  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
