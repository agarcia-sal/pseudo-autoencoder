CLASS Solution  
    FUNCTION minSubArrayLen(target parameter, nums parameter)  
        SET left TO zero  
        SET current sum TO zero  
        SET min length TO a quantity greater than any real number  
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            INCREMENT current sum BY the element at position right of nums  
            WHILE current sum GREATER THAN OR EQUAL TO target parameter holds  
                SET min length TO the lesser BETWEEN min length AND right MINUS left PLUS one  
                DECREMENT current sum BY the element at position left of nums  
                INCREMENT left BY one  
            END WHILE  
        END FOR  
        IF min length NOT EQUALS a quantity greater than any real number THEN  
            RETURN min length  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
