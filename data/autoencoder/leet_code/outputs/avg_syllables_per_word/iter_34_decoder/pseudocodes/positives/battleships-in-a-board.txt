CLASS Solution  
    FUNCTION countBattleships WITH PARAMETER board  
        SET count TO zero  
        FOR i FROM zero TO LENGTH OF board MINUS one  
            FOR j FROM zero TO LENGTH OF element at position zero of board MINUS one  
                IF element at position i of board at position j EQUALS letter X THEN  
                    IF i EQUALS zero OR element at position i MINUS one of board at position j EQUALS dot AND j EQUALS zero OR element at position i of board at position j MINUS one EQUALS dot THEN  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
