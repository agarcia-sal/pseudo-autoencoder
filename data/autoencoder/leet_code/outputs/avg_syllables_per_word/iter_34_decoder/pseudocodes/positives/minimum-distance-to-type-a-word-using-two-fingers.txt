CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1 c2)  
            IF c1 EQUALS None  
                RETURN zero  
            END IF  
            SET x1 TO the result of dividing the numerical Unicode position of character c1 MINUS the numerical Unicode position of character A by six  
            SET y1 TO the remainder of the numerical Unicode position of character c1 MINUS the numerical Unicode position of character A DIVIDED BY six  
            SET x2 TO the result of dividing the numerical Unicode position of character c2 MINUS the numerical Unicode position of character A by six  
            SET y2 TO the remainder of the numerical Unicode position of character c2 MINUS the numerical Unicode position of character A DIVIDED BY six  
            RETURN the absolute value of x1 MINUS x2 PLUS the absolute value of y1 MINUS y2  
        END FUNCTION  
          
        FUNCTION dp(i f1 f2)  
            IF i EQUALS the length of word  
                RETURN zero  
            END IF  
            SET c TO the element at position i of word  
            SET d1 TO the result of distance between f1 and c PLUS the result of dp with parameters i PLUS one c f2  
            SET d2 TO the result of distance between f2 and c PLUS the result of dp with parameters i PLUS one f1 c  
            IF d1 LESS THAN d2  
                RETURN d1  
            ELSE  
                RETURN d2  
            END IF  
        END FUNCTION  
          
        RETURN dp with parameters zero None None  
    END FUNCTION  
END CLASS
