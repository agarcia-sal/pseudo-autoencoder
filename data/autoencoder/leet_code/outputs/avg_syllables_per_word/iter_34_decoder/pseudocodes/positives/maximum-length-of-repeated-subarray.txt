CLASS Solution  
    FUNCTION findLength(nums1 nums2)  
        SET dp TO a two dimensional list with the number of rows EQUAL TO the length of nums1 PLUS one AND the number of columns EQUAL TO the length of nums2 PLUS one FILLED WITH zero  
        SET max_length TO zero  
        FOR i FROM one TO the length of nums1  
            FOR j FROM one TO the length of nums2  
                IF the element at position i MINUS one of nums1 EQUALS the element at position j MINUS one of nums2  
                    SET the element at position i and position j of dp TO the element at position i MINUS one and position j MINUS one of dp PLUS one  
                    IF max_length LESS THAN the element at position i and position j of dp  
                        SET max_length TO the element at position i and position j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
