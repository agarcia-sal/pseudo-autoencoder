CLASS Solution  
    FUNCTION countSmaller(nums)  
        SET sorted_nums TO a sorted list of unique elements of nums  
        SET rank TO an empty dictionary  
        SET index TO one  
        FOR each value IN sorted_nums  
            ASSIGN rank at key value TO index  
            INCREMENT index BY one  
        END FOR  
        
        SET bit TO a list of zeros of length the LENGTH OF sorted_nums PLUS one  
        
        FUNCTION get_sum(idx)  
            SET res TO zero  
            WHILE idx GREATER THAN zero  
                INCREMENT res BY element at position idx of bit  
                SET idx TO idx MINUS the bitwise AND of idx and the negative of idx (described as the largest power of two dividing idx)  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        FUNCTION update(idx, delta)  
            WHILE idx LESS THAN the LENGTH OF bit  
                INCREMENT element at position idx of bit BY delta  
                SET idx TO idx PLUS the bitwise AND of idx and the negative of idx (described as the largest power of two dividing idx)  
            END WHILE  
        END FUNCTION  
        
        SET counts TO an empty list  
        FOR each number IN nums iterated from end to start  
            SET r TO rank at key number  
            APPEND the result of get_sum of r MINUS one TO counts  
            CALL update with parameters r and one  
        END FOR  
        
        RETURN counts reversed in order  
    END FUNCTION  
END CLASS
