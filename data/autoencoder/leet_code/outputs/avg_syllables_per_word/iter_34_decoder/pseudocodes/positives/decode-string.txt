CLASS Solution  
    FUNCTION decodeString WITH PARAMETER s  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each character IN s  
            IF character IS a digit THEN  
                SET d TO d concatenated with character  
            ELSE IF character IS an alphabetic letter THEN  
                SET a TO a concatenated with character  
            ELSE IF character IS equal to the left square bracket character THEN  
                APPEND pair of a and the integer value of d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF character IS equal to the right square bracket character THEN  
                SET pair TO the last element removed from stack  
                SET p TO the first element of pair  
                SET n TO the second element of pair  
                SET a TO p concatenated with a repeated n times  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
