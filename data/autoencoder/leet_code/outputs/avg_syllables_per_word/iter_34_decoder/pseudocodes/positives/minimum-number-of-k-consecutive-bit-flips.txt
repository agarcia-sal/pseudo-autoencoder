CLASS Solution  
    FUNCTION minKBitFlips(nums k)  
        SET n TO the LENGTH OF nums  
        SET flip_count TO zero  
        SET flipped TO a list of zeros of LENGTH n PLUS one  
        SET flip_state TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET flip_state TO flip_state XOR element at position i of flipped  
            
            IF element at position i of nums EQUALS flip_state  
                IF i PLUS k GREATER THAN n  
                    RETURN negative one  
                END IF  
                SET flip_state TO flip_state XOR one  
                SET element at position i PLUS k of flipped TO element at position i PLUS k of flipped XOR one  
                INCREMENT flip_count BY one  
            END IF  
        END FOR  
        
        RETURN flip_count  
    END FUNCTION  
END CLASS
