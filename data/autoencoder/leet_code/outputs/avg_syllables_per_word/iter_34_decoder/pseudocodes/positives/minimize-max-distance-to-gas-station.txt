CLASS Solution  
    FUNCTION minmaxGasDist(stations parameter OF TYPE List of integers k parameter OF TYPE integer) RETURNS floating point number  
        FUNCTION possible(max_dist parameter OF TYPE floating point number) RETURNS boolean  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                INCREMENT additional_stations BY the INTEGER PART OF the element at position index of stations MINUS the element at position index MINUS one of stations DIVIDED BY max_dist  
            END FOR  
            RETURN additional_stations LESS THAN OR EQUAL TO k  
        END FUNCTION  

        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE right MINUS left GREATER THAN zero POINT zero zero zero zero zero one  
            SET mid TO left PLUS right DIVIDED BY two  
            IF possible(mid) IS true  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
