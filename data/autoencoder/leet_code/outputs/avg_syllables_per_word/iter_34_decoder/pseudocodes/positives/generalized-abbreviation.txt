CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the STRING REPRESENTATION OF count TO path  
                END IF  
                APPEND the CONCATENATION OF ELEMENTS IN path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH index PLUS one AS index argument  
                        WITH path PLUS the STRING REPRESENTATION OF count PLUS the element at position index of word AS path argument  
                        WITH zero AS count argument  
                ELSE  
                    CALL backtrack WITH index PLUS one AS index argument  
                        WITH path PLUS the element at position index of word AS path argument  
                        WITH zero AS count argument  
                END IF  
                CALL backtrack WITH index PLUS one AS index argument  
                    WITH path AS path argument  
                    WITH count PLUS one AS count argument  
            END IF  
        END FUNCTION  

        SET result TO an empty list  
        CALL backtrack WITH zero AS index argument  
            WITH an empty list AS path argument  
            WITH zero AS count argument  
        RETURN result  
    END FUNCTION  
END CLASS
