CLASS Solution  
    FUNCTION minTotalDistance(grid)  
        SET rows TO empty list  
        SET cols TO empty list  
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position i of grid MINUS one  
                IF element at position i of element at position j of grid EQUALS one  
                    APPEND i TO rows  
                    APPEND j TO cols  
                END IF  
            END FOR  
        END FOR  
        SORT rows  
        SORT cols  
        SET median_row TO element at position the LENGTH OF rows DIVIDED BY two of rows  
        SET median_col TO element at position the LENGTH OF cols DIVIDED BY two of cols  
        SET min_distance TO zero  
        FOR each row IN rows  
            SET difference TO row MINUS median_row  
            IF difference LESS THAN zero  
                SET difference TO difference MULTIPLIED BY negative one  
            END IF  
            INCREMENT min_distance BY difference  
        END FOR  
        FOR each col IN cols  
            SET difference TO col MINUS median_col  
            IF difference LESS THAN zero  
                SET difference TO difference MULTIPLIED BY negative one  
            END IF  
            INCREMENT min_distance BY difference  
        END FOR  
        RETURN min_distance  
    END FUNCTION  
END CLASS
