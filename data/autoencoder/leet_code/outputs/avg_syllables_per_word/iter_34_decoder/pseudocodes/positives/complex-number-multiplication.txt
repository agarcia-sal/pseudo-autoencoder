CLASS Solution  
    FUNCTION complexNumberMultiply(num1  
                                   num2)  
        FUNCTION parse_complex(num)  
            SET parts TO the result of splitting num BY the plus sign  
            SET real TO the integer representation of element at position zero of parts  
            SET imaginary_string TO element at position one of parts  
            SET imaginary TO the integer representation of the substring of imaginary_string FROM position zero TO the position before the last character  
            RETURN real AND imaginary  
        END FUNCTION  
        
        SET real1 AND imag1 TO the result of calling parse_complex WITH num1  
        SET real2 AND imag2 TO the result of calling parse_complex WITH num2  
        
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        
        RETURN the concatenation of the string representation of real_part PLUS the plus sign PLUS the string representation of imaginary_part PLUS the letter i  
    END FUNCTION  
END CLASS
