CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO empty mapping from content to list of file paths  
        FOR each path IN paths  
            SET parts TO the list of elements obtained by splitting path at spaces  
            SET directory TO the first element in parts  
            FOR each file_info IN the elements of parts starting from the second element TO the last element  
                SET file_name TO the substring of file_info before the opening parenthesis  
                SET content TO the substring of file_info from the position after the opening parenthesis TO the position before the closing parenthesis  
                APPEND the concatenation of directory word slash word file_name TO the list in content_to_paths at key content  
            END FOR  
        END FOR  
        SET result TO empty list  
        FOR each group_of_paths IN the values of content_to_paths  
            IF the length of group_of_paths is GREATER THAN one  
                APPEND group_of_paths TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
