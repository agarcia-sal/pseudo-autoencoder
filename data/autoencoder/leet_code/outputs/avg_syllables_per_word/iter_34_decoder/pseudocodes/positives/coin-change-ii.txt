CLASS Solution  
    FUNCTION change(amount parameter, coins parameter)  
        SET a list named dp TO a list of zeros with the length equal to amount parameter PLUS one  
        SET element at position zero of dp TO one  
        FOR each coin element IN coins parameter  
            FOR index x FROM the value of coin element TO amount parameter  
                INCREMENT element at position x of dp BY element at position x MINUS coin element of dp  
            END FOR  
        END FOR  
        RETURN element at position amount parameter of dp  
    END FUNCTION  
END CLASS
