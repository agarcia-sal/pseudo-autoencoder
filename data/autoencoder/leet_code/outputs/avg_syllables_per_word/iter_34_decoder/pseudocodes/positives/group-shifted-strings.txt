CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s EQUALS empty string  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET difference TO the character code of c MINUS the character code of element at position one of s  
                SET wrapped_difference TO difference MODULO the cardinal number of letters in the English alphabet  
                APPEND the string representation of wrapped_difference TO canonical_form  
            END FOR  
            SET joined_form TO the concatenation of elements of canonical_form separated by the single character comma  
            RETURN joined_form  
        END FUNCTION  
        
        SET groups TO a new default dictionary with default type of empty list  
        
        FOR each string s IN strings  
            SET canonical_form TO get_canonical_form with argument s  
            APPEND s TO element at key canonical_form OF groups  
        END FOR  
        
        RETURN the list of all values contained in groups  
    END FUNCTION  
END CLASS
