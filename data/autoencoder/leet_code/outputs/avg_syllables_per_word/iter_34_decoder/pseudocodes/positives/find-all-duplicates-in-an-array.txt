    FUNCTION findDuplicates(nums)  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            WHILE element at position index of nums NOT EQUALS element at position the element at position index of nums MINUS one of nums  
                SET temporary_value TO element at position index of nums  
                SET element at position index of nums TO element at position the element at position index of nums MINUS one of nums  
                SET element at position the temporary_value MINUS one of nums TO temporary_value  
            END WHILE  
        END FOR  
        SET result_list TO empty list  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            SET current_value TO element at position index of nums  
            IF current_value NOT EQUALS index PLUS one
