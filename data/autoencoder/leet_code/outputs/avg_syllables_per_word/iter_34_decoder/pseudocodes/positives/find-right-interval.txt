CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO an empty list  
        FOR index FROM zero TO the LENGTH OF intervals MINUS one  
            SET interval TO element at position index of intervals  
            APPEND a pair consisting of the element at position zero of interval AND index TO start_times  
        END FOR  
        SORT start_times BY the first element of each pair in ascending order  
        
        SET result TO an empty list  
        FOR each interval IN intervals  
            SET end_value TO the element at position one of interval  
            SET search_index TO the position found by binary searching for the smallest first element in start_times GREATER THAN OR EQUAL TO end_value  
            IF search_index IS LESS THAN the LENGTH OF start_times THEN  
                APPEND the second element of the pair at position search_index of start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
