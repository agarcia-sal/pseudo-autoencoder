CLASS Solution  
    FUNCTION minMoves2(nums)  
        CALL sort on nums to arrange elements in non decreasing order  
        SET median TO element at position integer division of the LENGTH OF nums by two of nums  
        SET moves TO zero  
        FOR each num IN nums  
            INCREMENT moves BY the absolute value of num MINUS median  
        END FOR  
        RETURN moves  
    END FUNCTION  
END CLASS
