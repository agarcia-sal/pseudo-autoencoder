CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN the substring sub EQUALS the reverse of substring sub  
        END FUNCTION  
        
        SET n TO the LENGTH OF s  
        
        SET is_pal TO a two dimensional collection of size n by n with all values set TO false  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF the character at position i of s EQUALS the character at position j of s AND  
                   the difference of j MINUS i LESS THAN OR EQUAL TO one OR the value at position i PLUS one and j MINUS one in is_pal EQUALS true  
                THEN  
                    SET the value at position i and j in is_pal TO true  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM one TO n MINUS two  
            IF the value at position zero and i MINUS one in is_pal EQUALS true  
            THEN  
                FOR j FROM i TO n MINUS two  
                    IF the value at position i and j in is_pal EQUALS true AND the value at position j PLUS one and n MINUS one in is_pal EQUALS true  
                    THEN  
                        RETURN true  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
