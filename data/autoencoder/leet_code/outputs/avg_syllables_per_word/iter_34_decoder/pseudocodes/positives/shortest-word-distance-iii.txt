CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict word_list OF STRINGS word1 word_string word2 word_string) RETURNS INTEGER  
        SET min_distance TO positive infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO word1 EQUALS word2  
        
        FOR each index i FROM zero TO the LENGTH OF wordsDict MINUS one  
            SET word TO element at position i of wordsDict  
            
            IF word EQUALS word1  
                IF same_word IS true  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO i  
            END IF  
            
            IF word EQUALS word2  
                IF same_word IS false  
                    SET prev_word2 TO i  
                END IF  
            END IF  
            
            IF prev_word1 NOT EQUALS negative one AND prev_word2 NOT EQUALS negative one  
                IF min_distance GREATER THAN absolute value of prev_word1 MINUS prev_word2  
                    SET min_distance TO absolute value of prev_word1 MINUS prev_word2  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
