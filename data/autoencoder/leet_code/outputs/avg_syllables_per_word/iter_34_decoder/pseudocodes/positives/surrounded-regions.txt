CLASS Solution  
    FUNCTION solve(board)  
        IF board EQUALS empty OR the element at position zero of board EQUALS empty THEN  
            RETURN  
        END IF  
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF the element at position zero of board  
        FUNCTION dfs(x, y)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR the element at position x of board at position y NOT EQUALS string O THEN  
                RETURN  
            END IF  
            SET the element at position x of board at position y TO string S  
            CALL dfs with x PLUS one and y  
            CALL dfs with x MINUS one and y  
            CALL dfs with x and y PLUS one  
            CALL dfs with x and y MINUS one  
        END FUNCTION  
        FOR i FROM zero TO m MINUS one  
            CALL dfs with i and zero  
            CALL dfs with i and n MINUS one  
        END FOR  
        FOR j FROM zero TO n MINUS one  
            CALL dfs with zero and j  
            CALL dfs with m MINUS one and j  
        END FOR  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i of board at position j EQUALS string O THEN  
                    SET the element at position i of board at position j TO string X  
                END IF  
            END FOR  
        END FOR  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i of board at position j EQUALS string S THEN  
                    SET the element at position i of board at position j TO string O  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
