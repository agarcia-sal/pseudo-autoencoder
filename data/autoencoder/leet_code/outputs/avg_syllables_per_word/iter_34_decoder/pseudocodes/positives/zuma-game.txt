CLASS Solution  
    FUNCTION findMinStep(board, hand)  
        FUNCTION clean(board)  
            SET stack TO empty list  
            FOR each ball IN the sequence of characters in board  
                IF stack is not empty AND the first element at the last position of stack NOT EQUALS ball AND the second element at the last position of stack GREATER THAN OR EQUAL TO three  
                    REMOVE the last element FROM stack  
                END IF  
                IF stack is empty OR the first element at the last position of stack NOT EQUALS ball  
                    APPEND a pair of ball and one TO stack  
                ELSE  
                    INCREMENT the second element at the last position of stack BY one  
                END IF  
            END FOR  
            IF stack is not empty AND the second element at the last position of stack GREATER THAN OR EQUAL TO three  
                REMOVE the last element FROM stack  
            END IF  
            SET result TO an empty string  
            FOR each pair of ball and count IN stack  
                APPEND the sequence of ball repeated count times TO result  
            END FOR  
            RETURN result  
        END FUNCTION  

        FUNCTION dfs(board, hand_counter)  
            IF the sequence board is empty  
                RETURN zero  
            END IF  
            SET min_steps TO positive infinity  
            SET i TO zero  
            WHILE i LESS THAN the length of board  
                SET j TO i PLUS one  
                WHILE j LESS THAN the length of board AND the element at position i of board EQUALS the element at position j of board  
                    INCREMENT j BY one  
                END WHILE  
                SET increment TO three MINUS the difference of j MINUS i  
                IF the value corresponding to the element at position i of board IN hand_counter GREATER THAN OR EQUAL TO increment  
                    SET new_hand_counter TO a copy of hand_counter  
                    DECREMENT the value corresponding to the element at position i of board IN new_hand_counter BY increment  
                    SET new_board TO clean(the substring from position zero TO position i of board CONCATENATED WITH substring from position j TO the end of board)  
                    SET steps TO dfs(new_board, new_hand_counter)  
                    IF steps GREATER THAN OR EQUAL TO zero  
                        SET min_steps TO the lesser of min_steps AND steps PLUS increment  
                    END IF  
                END IF  
                SET i TO j  
            END WHILE  
            IF min_steps EQUALS positive infinity  
                RETURN negative one  
            ELSE  
                RETURN min_steps  
            END IF  
        END FUNCTION  

        SET hand_counter TO the count mapping of elements in hand  
        RETURN dfs(board, hand_counter)  
    END FUNCTION  
END CLASS
