CLASS Solution  
    FUNCTION minimumMoves(arr)  
        SET n TO the LENGTH OF arr  
        SET dp TO a two dimensional list of zeros with dimensions n BY n  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AT position i OF dp TO one  
        END FOR  
        
        FOR i FROM zero TO n MINUS two  
            IF element at position i OF arr EQUALS element at position i PLUS one OF arr THEN  
                SET element at position i AT position i PLUS one OF dp TO one  
            ELSE  
                SET element at position i AT position i PLUS one OF dp TO two  
            END IF  
        END FOR  
        
        FOR length FROM three TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i OF arr EQUALS element at position j OF arr THEN  
                    SET element at position i AT position j OF dp TO element at position i PLUS one AT position j MINUS one OF dp  
                ELSE  
                    SET element at position i AT position j OF dp TO infinity  
                    FOR k FROM i TO j MINUS one  
                        SET candidate_value TO element at position i AT position k OF dp PLUS element at position k PLUS one AT position j OF dp  
                        IF candidate_value LESS THAN element at position i AT position j OF dp THEN  
                            SET element at position i AT position j OF dp TO candidate_value  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position zero AT position n MINUS one OF dp  
    END FUNCTION  
END CLASS
