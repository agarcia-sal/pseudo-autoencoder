CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            
            SET expected TO the STRING REPRESENTATION OF the SUM OF first PLUS second  
            SET length TO the LENGTH OF expected  
            
            IF the SUM OF start PLUS length GREATER THAN the LENGTH OF num OR the SUBSTRING of num from position start TO the POSITION OF start PLUS length NOT EQUALS expected  
                RETURN False  
            END IF  
            
            RETURN the RESULT OF is_valid_sequence WITH the ARGUMENTS start PLUS length, second, the INTEGER REPRESENTATION OF expected  
        END FUNCTION  
        
        SET n TO the LENGTH OF num  
        FOR i FROM one TO the INTEGER DIVISION OF n BY two PLUS one  
            FOR j FROM i PLUS one TO n  
                SET first TO the SUBSTRING of num from position zero TO position i  
                SET second TO the SUBSTRING of num from position i TO position j  
                
                IF the LENGTH OF first GREATER THAN one AND the FIRST CHARACTER of first EQUALS the CHARACTER zero OR the LENGTH OF second GREATER THAN one AND the FIRST CHARACTER of second EQUALS the CHARACTER zero  
                    CONTINUE TO the NEXT ITERATION  
                END IF  
                
                IF is_valid_sequence WITH the ARGUMENTS j, the INTEGER REPRESENTATION OF first, the INTEGER REPRESENTATION OF second EQUALS True  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
