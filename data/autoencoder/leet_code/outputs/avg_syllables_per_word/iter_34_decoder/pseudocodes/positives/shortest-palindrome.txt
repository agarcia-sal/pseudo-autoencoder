CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS nothing  
            RETURN s  
        END IF  
          
        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS reverse_of check_str  
        END FUNCTION  
          
        FOR i FROM the LENGTH OF s DOWN TO zero  
            IF is_palindrome(substring from position zero TO position i MINUS one of s)  
                SET to_add TO reverse_of substring from position i TO the END OF s  
                RETURN to_add PLUS s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
