CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        SET tasks TO the collection of pairs composed of each element at position index of plantTime AND at position index of growTime FOR each index FROM zero TO the length of plantTime MINUS one  
        SORT tasks in the order that the second element of each pair IS GREATER THAN for all comparisons  
        
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        
        FOR each pair IN tasks  
            SET plant TO the first element of pair  
            SET grow TO the second element of pair  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF bloom_day IS GREATER THAN earliest_bloom THEN  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
