CLASS Solution  
    FUNCTION longestSubstring(s k)  
        IF the LENGTH OF s LESS THAN k  
            RETURN zero  
        END IF  
        SET count TO the FREQUENCY MAP OF characters IN s  
        FOR each character IN the KEYS OF count  
            IF the FREQUENCY OF character IN count LESS THAN k  
                RETURN the MAXIMUM VALUE OVER the RESULTS OF longestSubstring CALLED WITH each substring OBTAINED BY SPLITTING s BY character AND k AS PARAMETERS  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
