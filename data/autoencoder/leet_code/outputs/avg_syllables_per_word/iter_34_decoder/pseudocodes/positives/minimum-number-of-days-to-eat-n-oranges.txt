CLASS Solution  
    FUNCTION minDays WITH PARAMETER n  
        FUNCTION dp WITH PARAMETER remaining  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            RETURN one PLUS THE MINIMUM OF the REMAINDER OF remaining DIVIDED BY two PLUS THE RESULT OF dp WITH PARAMETER remaining DIVIDED BY two AND the REMAINDER OF remaining DIVIDED BY three PLUS THE RESULT OF dp WITH PARAMETER remaining DIVIDED BY three  
        END FUNCTION  
        RETURN dp WITH PARAMETER n  
    END FUNCTION  
END CLASS
