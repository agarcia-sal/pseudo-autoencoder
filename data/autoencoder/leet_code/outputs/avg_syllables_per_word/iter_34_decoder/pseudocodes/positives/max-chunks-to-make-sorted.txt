CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET max_so_far TO zero  
        SET chunks TO zero  
        FOR index i FROM zero TO the LENGTH OF arr MINUS one  
            SET element_at_i TO element at position i of arr  
            SET max_so_far TO the GREATER THAN value BETWEEN max_so_far AND element_at_i  
            IF max_so_far EQUALS i THEN  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        RETURN chunks  
    END FUNCTION  
END CLASS
