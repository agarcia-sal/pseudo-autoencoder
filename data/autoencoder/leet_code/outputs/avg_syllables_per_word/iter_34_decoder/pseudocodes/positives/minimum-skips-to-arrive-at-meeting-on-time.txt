CLASS Solution  
    FUNCTION minSkips(dist speed hoursBefore)  
        SET n TO the LENGTH OF dist  
        CREATE a two dimensional list dp with n rows and n PLUS one columns where each element IS set TO infinity  
        SET total_dist TO the SUM OF all elements IN dist  
        
        IF total_dist DIVIDED BY speed IS GREATER THAN hoursBefore  
            RETURN negative one  
        END IF  
        
        SET element at position zero of row zero IN dp TO element at position zero of dist  
        FOR j FROM one TO n MINUS one  
            SET element at position zero of row j IN dp TO element at position zero of dist  
        END FOR  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j IS LESS THAN i  
                    SET element at position i of row j IN dp TO the CEILING OF the element at position i MINUS one of row j IN dp DIVIDED BY speed MULTIPLIED BY speed PLUS element at position i of dist  
                END IF  
                IF j IS GREATER THAN zero  
                    SET element at position i of row j IN dp TO the MINIMUM VALUE BETWEEN element at position i of row j IN dp AND element at position i MINUS one of row j MINUS one IN dp PLUS element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            IF element at position n MINUS one of row j IN dp IS LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
