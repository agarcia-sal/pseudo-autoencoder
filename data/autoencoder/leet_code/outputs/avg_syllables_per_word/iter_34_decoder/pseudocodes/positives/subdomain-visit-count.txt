CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        ASSIGN reference TO a new mapping with default integer zero named domain_count  
        
        FOR each cpdomain IN cpdomains  
            SET count_and_domain TO the SPLIT of cpdomain BY whitespace  
            SET count TO the conversion of the first element of count_and_domain TO integer  
            SET domain TO the second element of count_and_domain  
            SET subdomains TO the SPLIT of domain BY the period word  
            
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one  
                SET subdomain_parts TO the slice of subdomains FROM index TO the end  
                SET subdomain TO the JOIN of subdomain_parts USING the period word  
                INCREMENT the value at subdomain in domain_count BY count  
            END FOR  
        END FOR  
        
        SET result_list TO an empty list  
        FOR each domain and count IN the items of domain_count  
            SET formatted_string TO the concatenation of the string representation of count, a single space word, and the domain  
            APPEND formatted_string TO result_list  
        END FOR  
        
        RETURN result_list  
    END FUNCTION  
END CLASS
