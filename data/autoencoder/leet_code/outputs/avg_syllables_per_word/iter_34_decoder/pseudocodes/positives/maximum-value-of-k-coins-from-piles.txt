CLASS Solution  
    FUNCTION maxValueOfCoins(piles, k)  
        SET n TO the LENGTH OF piles  
        SET dp TO a LIST OF LISTS of zero WITH the SIZE of n PLUS one BY k PLUS one  

        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one of piles  
                SET current_value TO zero  
                FOR l FROM zero TO the MINIMUM OF j AND the LENGTH OF current_pile  
                    IF l GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one of current_pile  
                    END IF  
                    IF dp at position i and j LESS THAN dp at position i MINUS one and j MINUS l PLUS current_value  
                        SET dp at position i and j TO dp at position i MINUS one and j MINUS l PLUS current_value  
                    END IF  
                END FOR  
            END FOR  
        END FOR  

        RETURN dp at position n and k  
    END FUNCTION  
END CLASS
