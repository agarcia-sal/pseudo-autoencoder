CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a list of the LENGTH OF n FILLED WITH minus one  
        SET stack TO an empty list  
        
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO the remainder of i DIVIDED BY n  
            WHILE the LENGTH OF stack is GREATER THAN zero AND the element at position the last element of stack of nums is LESS THAN the element at position index of nums  
                SET top_index TO the last element of stack  
                REMOVE the last element FROM stack  
                SET the element at position top_index of result TO the element at position index of nums  
            END WHILE  
            IF i is LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
