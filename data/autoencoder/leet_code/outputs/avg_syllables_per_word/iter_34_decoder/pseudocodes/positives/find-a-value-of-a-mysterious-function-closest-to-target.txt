CLASS Solution  
    FUNCTION closestToTarget(arr, target)  
        SET seen TO an empty set  
        SET min_diff TO a value representing positive infinity  
        FOR each num IN arr  
            SET new_seen TO an empty set  
            FOR each x IN seen  
                APPEND the bitwise AND of num AND x TO new_seen  
            END FOR  
            SET seen TO the union of new_seen AND a set containing num  
            FOR each val IN seen  
                SET current_difference TO the absolute value of val MINUS target  
                IF current_difference LESS THAN min_diff  
                    SET min_diff TO current_difference  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
