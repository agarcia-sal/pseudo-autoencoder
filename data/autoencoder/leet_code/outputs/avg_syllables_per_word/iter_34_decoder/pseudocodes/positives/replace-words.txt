CLASS Solution  
    FUNCTION replaceWords(dictionary PARAMETER AS List of strings, sentence PARAMETER AS string) RETURNS string  
        SORT dictionary BY the LENGTH OF each element FROM the SMALLEST TO the LARGEST  
        SET words TO the LIST OF WORDS OBTAINED BY SPLITTING sentence BY SPACES  
        FUNCTION replace(word PARAMETER AS string) RETURNS string  
            FOR each root IN dictionary  
                IF the BEGINNING OF word EQUALS root THEN  
                    RETURN root  
                END IF  
            END FOR  
            RETURN word  
        END FUNCTION  
        SET replaced_words TO an EMPTY list  
        FOR each word IN words  
            APPEND the RESULT OF replace APPLIED TO word TO replaced_words  
        END FOR  
        SET result_sentence TO the STRING OBTAINED BY JOINING replaced_words WITH SPACES AS SEPARATORS  
        RETURN result_sentence  
    END FUNCTION  
END CLASS
