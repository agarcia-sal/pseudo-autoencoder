CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        CREATE graph AS a mapping from a key to a list of values with default empty list  
        SET indegree TO a list of zeros with length equal to numCourses  
        
        FOR each pair of course and prereq IN prerequisites  
            APPEND course TO list associated with prereq in graph  
            INCREMENT value at position course of indegree BY one  
        END FOR  
        
        CREATE queue AS a double ended queue initialized with each course in range from zero TO numCourses MINUS one WHERE value at position course of indegree EQUALS zero  
        
        WHILE queue is not empty  
            REMOVE and RETURN leftmost element from queue TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN list associated with current_course in graph  
                DECREMENT value at position next_course of indegree BY one  
                IF value at position next_course of indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN whether numCourses EQUALS zero  
    END FUNCTION  
END CLASS
