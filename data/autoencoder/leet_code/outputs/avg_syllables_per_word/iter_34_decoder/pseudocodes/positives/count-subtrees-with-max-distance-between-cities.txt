CLASS Solution  
    FUNCTION countSubgraphsForEachDiameter(n parameter, edges parameter)  
        SET graph TO an empty mapping from integer TO list of integers  
        FOR each integer i FROM one TO n parameter  
            SET the element at key i of graph TO an empty list  
        END FOR  
        FOR each pair u parameter AND v parameter IN edges parameter  
            APPEND v parameter TO the element at key u parameter of graph  
            APPEND u parameter TO the element at key v parameter of graph  
        END FOR  
        
        FUNCTION bfs(subtree parameter)  
            IMPORT deque FROM collections  
            SET start TO an arbitrary element chosen from subtree parameter  
            SET queue TO a deque initialized with start  
            SET visited TO a set initialized with start  
            SET distance TO zero  
            SET farthest_node TO start  
            
            WHILE queue is not empty  
                REMOVE the element from the front of queue AND ASSIGN it TO node  
                FOR each neighbor IN the element at key node of graph  
                    IF neighbor is in subtree parameter AND neighbor is not in visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                        SET farthest_node TO neighbor  
                    END IF  
                END FOR  
            END WHILE  
            
            SET queue TO a deque initialized with farthest_node  
            SET visited TO a set initialized with farthest_node  
            SET max_distance TO zero  
            
            WHILE queue is not empty  
                SET size TO the number of elements in queue  
                FOR counter FROM one TO size  
                    REMOVE the element from the front of queue AND ASSIGN it TO node  
                    FOR each neighbor IN the element at key node of graph  
                        IF neighbor is in subtree parameter AND neighbor is not in visited THEN  
                            ADD neighbor TO visited  
                            APPEND neighbor TO queue  
                        END IF  
                    END FOR  
                END FOR  
                IF queue is not empty THEN  
                    INCREMENT max_distance BY one  
                END IF  
            END WHILE  
            
            RETURN max_distance  
        END FUNCTION  
        
        SET result TO a list of zeros with length n parameter MINUS one  
        
        FOR size FROM two TO n parameter  
            FOR each subset IN combinations of the sequence of integers from one TO n parameter taken size at a time  
                SET visited TO an empty set  
                SET stack TO a list initialized with the first element of subset  
                WHILE stack is not empty  
                    REMOVE the last element from stack AND ASSIGN it TO node  
                    IF node is not in visited THEN  
                        ADD node TO visited  
                        FOR each neighbor IN the element at key node of graph  
                            IF neighbor is in subset AND neighbor is not in visited THEN  
                                APPEND neighbor TO stack  
                            END IF  
                        END FOR  
                    END IF  
                END WHILE  
                
                IF visited EQUALS the set constructed from subset THEN  
                    SET diameter TO bfs called with the set constructed from subset  
                    IF diameter GREATER THAN zero THEN  
                        INCREMENT the element at position diameter MINUS one of result BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
