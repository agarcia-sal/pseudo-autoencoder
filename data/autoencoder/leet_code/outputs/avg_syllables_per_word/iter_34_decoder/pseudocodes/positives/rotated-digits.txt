CLASS Solution  
    FUNCTION rotatedDigits WITH parameter n  
        SET change_digits TO the collection containing the characters two five six nine  
        SET same_digits TO the collection containing the characters zero one eight  
        
        FUNCTION is_good_number WITH parameter num  
            SET num_str TO the string representation of num  
            IF each character IN num_str belongs TO the union of change_digits and same_digits THEN  
                IF at least one character IN num_str belongs TO change_digits THEN  
                    RETURN true  
                ELSE  
                    RETURN false  
                END IF  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR each i FROM one TO n  
            IF is_good_number WITH argument i RETURNS true THEN  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
