CLASS Solution  
    FUNCTION largestSumOfAverages(nums PARAMETER, k PARAMETER)  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of zeros with LENGTH n PLUS one  
        FOR i FROM zero TO n MINUS one  
            SET element at position i PLUS one of prefix_sum TO element at position i of prefix_sum PLUS element at position i of nums  
        END FOR  
        
        FUNCTION average(i PARAMETER, j PARAMETER)  
            SET total_sum TO element at position j PLUS one of prefix_sum MINUS element at position i of prefix_sum  
            SET length_subarray TO j MINUS i PLUS one  
            SET average_value TO total_sum DIVIDED BY length_subarray  
            RETURN average_value  
        END FUNCTION  
        
        SET dp TO a two dimensional list of zeros with dimensions n PLUS one BY k PLUS one  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position one of element at position i of dp TO CALL average WITH arguments i AND n MINUS one  
        END FOR  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    SET current_value TO element at position partitions of element at position i of dp  
                    SET candidate_value TO element at position partitions MINUS one of element at position j of dp PLUS CALL average WITH arguments i AND j MINUS one  
                    IF candidate_value GREATER THAN current_value THEN  
                        SET element at position partitions of element at position i of dp TO candidate_value  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position k of element at position zero of dp  
    END FUNCTION  
END CLASS
