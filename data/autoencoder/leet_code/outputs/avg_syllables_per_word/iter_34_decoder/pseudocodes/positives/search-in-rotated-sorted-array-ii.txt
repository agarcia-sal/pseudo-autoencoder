CLASS Solution  
    FUNCTION search(nums target)  
        IF the collection nums EQUALS empty  
            RETURN false  
        END IF  
        SET left TO zero  
        SET right TO the LENGTH OF the collection nums MINUS one  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS the DIFFERENCE OF right MINUS left DIVIDED BY two  
            IF element at position mid of the collection nums EQUALS target  
                RETURN true  
            END IF  
            IF element at position left of the collection nums EQUALS element at position mid of the collection nums AND element at position mid of the collection nums EQUALS element at position right of the collection nums  
                INCREMENT left BY one  
                DECREMENT right BY one  
            ELSE IF element at position left of the collection nums LESS THAN OR EQUAL TO element at position mid of the collection nums  
                IF element at position left of the collection nums LESS THAN OR EQUAL TO target AND target LESS THAN element at position mid of the collection nums  
                    SET right TO mid MINUS one  
                ELSE  
                    SET left TO mid PLUS one  
                END IF  
            ELSE  
                IF element at position mid of the collection nums LESS THAN target AND target LESS THAN OR EQUAL TO element at position right of the collection nums  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END IF  
        END WHILE  
        RETURN false  
    END FUNCTION  
END CLASS
