CLASS Solution  
    FUNCTION ipToCIDR(ip, n)  
        FUNCTION ip_to_int(ip)  
            SET parts TO list of integers obtained by splitting ip string by the dot character  
            SET res TO zero  
            FOR each part IN parts  
                SET res TO res MULTIPLIED BY two HUNDRED FIFTY SIX PLUS part  
            END FOR  
            RETURN res  
        END FUNCTION  
        
        FUNCTION int_to_ip(num)  
            SET ip_parts TO empty list  
            FOR i FROM three DOWN TO zero  
                SET part TO the element obtained by dividing num RIGHT SHIFTED BY eight MULTIPLIED BY i THEN MODULO two HUNDRED FIFTY SIX  
                APPEND string representation of part TO ip_parts  
            END FOR  
            RETURN all elements of ip_parts joined by the dot character  
        END FUNCTION  
        
        FUNCTION largest_cidr_size(start, n)  
            IF the bitwise AND of start AND one EQUALS one THEN  
                RETURN one  
            END IF  
            SET x TO one  
            WHILE x LESS THAN OR EQUAL TO n AND the bitwise AND of start AND x MINUS one EQUALS zero DO  
                SET x TO x MULTIPLIED BY two  
            END WHILE  
            RETURN x DIVIDED BY two  
        END FUNCTION  
        
        SET result TO empty list  
        SET start_ip TO ip_to_int applied to ip  
        
        WHILE n GREATER THAN zero DO  
            SET size TO largest_cidr_size applied to start_ip AND n  
            SET cidr_suffix TO thirty TWO MINUS the bit length of size MINUS one  
            SET cidr_string TO the int_to_ip applied to start_ip concatenated with the forward slash character concatenated with string representation of cidr_suffix  
            APPEND cidr_string TO result  
            INCREMENT start_ip BY size  
            DECREMENT n BY size  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
