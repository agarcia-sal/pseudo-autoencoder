CLASS Solution  
    FUNCTION maxNumber(nums1 nums2 k)  
        FUNCTION maxSingleArray(nums k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop GREATER THAN zero AND the stack IS NOT empty AND the element at the last position of stack LESS THAN num  
                    REMOVE the last element FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the FIRST k elements OF stack  
        END FUNCTION  
  
        FUNCTION merge(arr1 arr2)  
            SET result TO an empty list  
            WHILE the LENGTH OF arr1 PLUS the LENGTH OF arr2 GREATER THAN zero  
                IF arr1 GREATER THAN arr2 IN LEXICOGRAPHICAL ORDER  
                    APPEND the element at position zero of arr1 TO result  
                    REMOVE the element at position zero FROM arr1  
                ELSE  
                    APPEND the element at position zero of arr2 TO result  
                    REMOVE the element at position zero FROM arr2  
                END IF  
            END WHILE  
            RETURN result  
        END FUNCTION  
  
        SET max_number TO an empty list  
        FOR i FROM zero TO k  
            IF i LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO the result of merging the result of maxSingleArray with nums1 and i AND the result of maxSingleArray with nums2 and k MINUS i  
                IF max_number LESS THAN candidate IN LEXICOGRAPHICAL ORDER  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
