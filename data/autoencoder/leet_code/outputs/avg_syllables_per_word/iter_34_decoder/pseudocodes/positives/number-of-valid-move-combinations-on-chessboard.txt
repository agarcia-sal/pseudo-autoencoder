CLASS Solution  
    FUNCTION countCombinations(pieces, positions)  
        SET directions TO dictionary mapping rook TO list of pairs of one and zero negative one and zero zero and one zero and negative one   
        ADD queen TO dictionary with list of pairs of one and zero negative one and zero zero and one zero and negative one one and one negative one and one one and negative one negative one and negative one  
        ADD bishop TO dictionary with list of pairs of one and one negative one and one one and negative one negative one and negative one  
        
        SET positions TO new list by setting each row MINUS one and column MINUS one in positions  
        
        FUNCTION get_destinations(start, piece_type)  
            SET r TO element at first position of start  
            SET c TO element at second position of start  
            SET dests TO list containing start  
            FOR each pair dr and dc IN directions at key piece_type  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                WHILE nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN eight AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN eight  
                    APPEND pair nr and nc TO dests  
                    INCREMENT nr BY dr  
                    INCREMENT nc BY dc  
                END WHILE  
            END FOR  
            RETURN dests  
        END FUNCTION  
        
        SET all_destinations TO empty list  
        FOR each pos and piece IN pairs from positions and pieces  
            APPEND get_destinations with pos and piece TO all_destinations  
        END FOR  
        
        FUNCTION is_valid_combination(combination)  
            SET pos TO copy of positions  
            SET n TO length of pos  
            WHILE true  
                SET unique_positions_count TO zero  
                CREATE empty set visited_positions  
                FOR each position IN pos  
                    IF position NOT IN visited_positions  
                        ADD position TO visited_positions  
                        INCREMENT unique_positions_count BY one  
                    END IF  
                END FOR  
                IF unique_positions_count LESS THAN n  
                    RETURN false  
                END IF  
                SET all_reached TO true  
                FOR i FROM zero TO n MINUS one  
                    IF element at position i of pos EQUALS element at position i of combination  
                        CONTINUE to next iteration  
                    END IF  
                    SET all_reached TO false  
                    SET r TO element at first position of element at position i of pos  
                    SET c TO element at second position of element at position i of pos  
                    IF element at first position of element at position i of combination GREATER THAN r  
                        SET dr TO one  
                    ELSE IF element at first position of element at position i of combination LESS THAN r  
                        SET dr TO negative one  
                    ELSE  
                        SET dr TO zero  
                    END IF  
                    IF element at second position of element at position i of combination GREATER THAN c  
                        SET dc TO one  
                    ELSE IF element at second position of element at position i of combination LESS THAN c  
                        SET dc TO negative one  
                    ELSE  
                        SET dc TO zero  
                    END IF  
                    SET element at position i of pos TO pair r PLUS dr and c PLUS dc  
                END FOR  
                IF all_reached  
                    RETURN true  
                END IF  
            END WHILE  
        END FUNCTION  
        
        SET valid_count TO zero  
        FOR each combination IN Cartesian product of elements in all_destinations  
            IF is_valid_combination with combination  
                INCREMENT valid_count BY one  
            END IF  
        END FOR  
        
        RETURN valid_count  
    END FUNCTION  
END CLASS
