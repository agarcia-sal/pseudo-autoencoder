CLASS Solution  
    FUNCTION minWindow(s parameter OF TYPE string, t parameter OF TYPE string) RETURNS string  
        SET t_count TO a mapping of characters to their frequency counts in t  
        SET required TO the length of the set of unique characters in t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO an empty mapping of characters to their frequency counts  
        SET min_length TO a representation of infinity  
        SET min_start TO zero  
        
        WHILE right is LESS THAN the length of s  
            INCREMENT the frequency count of the character at position right of s in window_counts BY one  
            
            IF the character at position right of s is a key in t_count AND the frequency count of the character at position right of s in window_counts EQUALS the frequency count of that character in t_count THEN  
                INCREMENT formed BY one  
            END IF  
            
            WHILE left is LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO the character at position left of s  
                
                IF right MINUS left PLUS one is LESS THAN min_length THEN  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                
                DECREMENT the frequency count of char in window_counts BY one  
                
                IF char is a key in t_count AND the frequency count of char in window_counts is LESS THAN the frequency count of char in t_count THEN  
                    DECREMENT formed BY one  
                END IF  
                
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT right BY one  
        END WHILE  
        
        IF min_length EQUALS infinity THEN  
            RETURN an empty string  
        ELSE  
            RETURN the substring of s from position min_start TO position min_start PLUS min_length MINUS one  
        END IF  
    END FUNCTION  
END CLASS
