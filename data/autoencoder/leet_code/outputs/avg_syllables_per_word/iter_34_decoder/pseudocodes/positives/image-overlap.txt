CLASS Solution  
    FUNCTION largestOverlap(img1, img2)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of img1 at position j EQUALS one THEN  
                    APPEND a pair of i and j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of img2 at position j EQUALS one THEN  
                    APPEND a pair of i and j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a default dictionary with integer default values  
        FOR each pair x1 and y1 IN points1  
            FOR each pair x2 and y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value at key of a pair dx and dy in overlap_count BY one  
            END FOR  
        END FOR  
        SET max_overlap TO zero  
        FOR each value count IN the values of overlap_count  
            IF count GREATER THAN max_overlap THEN  
                SET max_overlap TO count  
            END IF  
        END FOR  
        RETURN max_overlap  
    END FUNCTION  
END CLASS
