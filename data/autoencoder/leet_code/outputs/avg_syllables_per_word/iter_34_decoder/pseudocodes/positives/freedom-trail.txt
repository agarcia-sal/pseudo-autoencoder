CLASS Solution  
    FUNCTION findRotateSteps(ring, key)  
        CREATE empty dictionary named char_positions with default value as an empty list  
        FOR each index i AND character char IN the ring string  
            APPEND i TO the list at key char in char_positions  
        END FOR  
        
        SET n TO the LENGTH OF ring  
        
        FUNCTION dp(i, prev_pos)  
            IF i EQUALS the LENGTH OF key THEN  
                RETURN zero  
            END IF  
            
            SET min_steps TO a very large number representing infinite  
            FOR each position pos IN the list at key at character at position i in key in char_positions  
                SET clockwise TO position MINUS prev_pos MODULO n  
                SET anticlockwise TO prev_pos MINUS position MODULO n  
                SET steps TO the smaller value BETWEEN clockwise AND anticlockwise PLUS one PLUS dp with arguments i PLUS one AND position  
                SET min_steps TO the smaller value BETWEEN min_steps AND steps  
            END FOR  
            
            RETURN min_steps  
        END FUNCTION  
        
        RETURN dp with arguments zero AND zero  
    END FUNCTION  
END CLASS
