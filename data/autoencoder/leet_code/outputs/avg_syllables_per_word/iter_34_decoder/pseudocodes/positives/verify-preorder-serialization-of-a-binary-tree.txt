CLASS Solution  
    FUNCTION isValidSerialization(preorder)  
        SET nodes TO the list obtained by splitting preorder at each comma  
        SET slots TO one  
        FOR each node IN nodes  
            IF slots EQUALS zero  
                RETURN False  
            END IF  
            DECREMENT slots BY one  
            IF node NOT EQUALS the single character hash symbol indicating null node  
                INCREMENT slots BY two  
            END IF  
        END FOR  
        RETURN slots EQUALS zero  
    END FUNCTION  
END CLASS
