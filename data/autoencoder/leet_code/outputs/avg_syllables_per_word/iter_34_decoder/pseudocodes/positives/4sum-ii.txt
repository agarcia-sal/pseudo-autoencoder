CLASS Solution  
    FUNCTION fourSumCount(nums1  
                          nums2  
                          nums3  
                          nums4)  
        SET sum_ab TO empty dictionary  
        FOR each element a IN nums1  
            FOR each element b IN nums2  
                SET current_sum TO a PLUS b  
                IF current_sum EXISTS AS A KEY IN sum_ab THEN  
                    INCREMENT value at key current_sum IN sum_ab BY one  
                ELSE  
                    SET value at key current_sum IN sum_ab TO one  
                END IF  
            END FOR  
        END FOR  
        SET count TO zero  
        FOR each element c IN nums3  
            FOR each element d IN nums4  
                SET target TO negative OF c PLUS d  
                IF target EXISTS AS A KEY IN sum_ab THEN  
                    INCREMENT count BY value at key target IN sum_ab  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
