CLASS Solution  
    FUNCTION canVisitAllRooms(rooms)  
        SET visited TO empty set  
        SET stack TO list containing zero  
        WHILE the LENGTH OF stack GREATER THAN zero  
            SET current_room TO the last element of stack  
            REMOVE the last element FROM stack  
            ADD current_room TO visited  
            FOR each key IN the element at position current_room of rooms  
                IF key NOT IN visited  
                    ADD key TO stack  
                END IF  
            END FOR  
        END WHILE  
        RETURN the LENGTH OF visited EQUALS the LENGTH OF rooms  
    END FUNCTION  
END CLASS
