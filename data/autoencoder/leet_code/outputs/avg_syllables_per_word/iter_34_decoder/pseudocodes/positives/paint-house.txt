CLASS Solution  
    FUNCTION minCost(costs)  
        IF costs EQUALS no elements  
            RETURN zero  
        END IF  
        
        SET prev_red TO element at position zero first element of costs  
        SET prev_blue TO element at position zero second element of costs  
        SET prev_green TO element at position zero third element of costs  
        
        FOR index i FROM one TO the LENGTH OF costs MINUS one  
            SET current_red TO element at position i first element of costs PLUS THE MINIMUM OF prev_blue AND prev_green  
            SET current_blue TO element at position i second element of costs PLUS THE MINIMUM OF prev_red AND prev_green  
            SET current_green TO element at position i third element of costs PLUS THE MINIMUM OF prev_red AND prev_blue  
            
            SET prev_red TO current_red  
            SET prev_blue TO current_blue  
            SET prev_green TO current_green  
        END FOR  
        
        RETURN THE MINIMUM OF prev_red AND prev_blue AND prev_green  
    END FUNCTION  
END CLASS
