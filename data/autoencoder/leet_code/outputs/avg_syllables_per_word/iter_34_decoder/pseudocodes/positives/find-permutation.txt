CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO an empty list  
        FOR each index i FROM zero TO the LENGTH OF s PLUS one MINUS one  
            APPEND the value of index i PLUS one TO result  
        END FOR  
        SET start TO zero  
        FOR each index end FROM zero TO the LENGTH OF s PLUS one MINUS one  
            IF the value of index end EQUALS the LENGTH OF s OR the element at position end of s EQUALS the uppercase letter I THEN  
                REVERSE the sublist of result FROM position start TO position end PLUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
