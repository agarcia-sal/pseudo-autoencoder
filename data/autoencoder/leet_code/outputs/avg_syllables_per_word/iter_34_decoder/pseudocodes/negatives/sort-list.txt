CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode with val equal to element at position one of values and next equal to None  
    SET p TO head  
    FOR each val IN elements of values starting from position two TO the last position  
        SET node TO new ListNode with val equal to val and next equal to None  
        SET the element next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the element val of p1 EQUALS the element val of p2 AND is_same_list(the element next of p1, the element next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList(self, head)  
        IF head EQUALS None OR the element next of head EQUALS None THEN  
            RETURN head  
        END IF  
        
        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO the element next of head  
            WHILE fast NOT EQUALS None AND the element next of fast NOT EQUALS None  
                SET slow TO the element next of slow  
                SET fast TO the element next of the element next of fast  
            END WHILE  
            SET mid TO the element next of slow  
            SET the element next of slow TO None  
            RETURN head, mid  
        END FUNCTION  
        
        FUNCTION merge(l1, l2)  
            SET dummy TO new ListNode with val equal to zero and next equal to None  
            SET current TO dummy  
            WHILE l1 NOT EQUALS None AND l2 NOT EQUALS None  
                IF the element val of l1 LESS THAN the element val of l2 THEN  
                    SET the element next of current TO l1  
                    SET l1 TO the element next of l1  
                ELSE  
                    SET the element next of current TO l2  
                    SET l2 TO the element next of l2  
                END IF  
                SET current TO the element next of current  
            END WHILE  
            IF l1 NOT EQUALS None THEN  
                SET the element next of current TO l1  
            ELSE  
                SET the element next of current TO l2  
            END IF  
            RETURN the element next of dummy  
        END FUNCTION  
        
        SET left, right TO split(head)  
        SET left TO self.sortList(left)  
        SET right TO self.sortList(right)  
        
        RETURN merge(left, right)  
    END FUNCTION  
END CLASS
