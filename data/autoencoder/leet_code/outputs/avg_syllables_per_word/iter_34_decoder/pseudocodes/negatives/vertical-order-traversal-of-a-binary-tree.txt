CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is GREATER THAN zero  
        SET node TO the element removed from the front of queue  
        IF i is LESS THAN the LENGTH OF values AND the element at position i of values is NOT EQUALS none  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is LESS THAN the LENGTH OF values AND the element at position i of values is NOT EQUALS none  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF the value of p IS NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of is_same_tree on the left child of p and the left child of q AND the result of is_same_tree on the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalTraversal(self, root)  
        IF root EQUALS none  
            RETURN an empty list  
        END IF  
        SET node_dict TO a default dictionary mapping keys to empty lists  
        SET min_col TO zero  
        SET max_col TO zero  

        FUNCTION dfs(node, row, col)  
            IF node EQUALS none  
                RETURN  
            END IF  
            APPEND a tuple consisting of row and the value of node TO the list at key col in node_dict  
            SET min_col TO the smaller of min_col and col  
            SET max_col TO the larger of max_col and col  
            CALL dfs with the left child of node, row PLUS one, col MINUS one  
            CALL dfs with the right child of node, row PLUS one, col PLUS one  
        END FUNCTION  

        CALL dfs with root, zero, zero  

        SET result TO an empty list  
        FOR each col FROM min_col TO max_col  
            SET values_list TO an empty list  
            FOR each pair consisting of row_value and node_value IN the list at key col in node_dict sorted by first element then second element  
                APPEND node_value TO values_list  
            END FOR  
            APPEND values_list TO result  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
