CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO ListNode element at position one of values  
    SET p TO head  
    FOR each val IN values from position two TO the end  
        SET node TO ListNode val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN self.val EQUALS p2.val AND is_same_list p1.next p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1 l2)  
        SET dummy_head TO ListNode zero  
        SET current TO dummy_head  
        SET carry TO zero  
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry DOES NOT EQUAL zero  
            IF l1 IS NOT None  
                SET val1 TO l1.val  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT None  
                SET val2 TO l2.val  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET current.next TO ListNode total MODULO ten  
            SET current TO current.next  
            IF l1 IS NOT None  
                SET l1 TO l1.next  
            END IF  
            IF l2 IS NOT None  
                SET l2 TO l2.next  
            END IF  
        END WHILE  
        RETURN dummy_head.next  
    END FUNCTION  
END CLASS
