CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO none right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with value being the element at position zero of values  
    SET i TO one  
    SET queue TO a new deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.left TO a new TreeNode with value being the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.right TO a new TreeNode with value being the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left q.left) AND is_same_tree(p.right q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self root k)  
        IF root.left EQUALS none AND root.right EQUALS none  
            RETURN root.val  
        END IF  
        SET graph TO a new default dictionary with list as default value  
        FUNCTION dfs(node parent SET TO none)  
            IF parent NOT EQUALS none  
                APPEND parent.val TO element at key node.val in graph  
                APPEND node.val TO element at key parent.val in graph  
            END IF  
            IF node.left NOT EQUALS none  
                CALL dfs with node.left and node  
            END IF  
            IF node.right NOT EQUALS none  
                CALL dfs with node.right and node  
            END IF  
        END FUNCTION  
        CALL dfs with root  
        SET queue TO a new deque initialized with element k  
        SET visited TO a new empty set  
        WHILE queue is not empty  
            SET current TO the element removed from the front of queue  
            ADD current TO visited  
            IF the LENGTH OF element at key current in graph EQUALS one AND current NOT EQUALS root.val  
                RETURN current  
            END IF  
            FOR each neighbor IN element at key current in graph  
                IF neighbor NOT IN visited  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
