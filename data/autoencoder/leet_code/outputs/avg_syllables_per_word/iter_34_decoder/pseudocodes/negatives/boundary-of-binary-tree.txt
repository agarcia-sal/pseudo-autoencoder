CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position zero of values  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue holds elements  
        SET node TO remove the first element from queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.left TO new TreeNode with value element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node.right TO new TreeNode with value element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION boundaryOfBinaryTree(self, root)  
        IF root EQUALS none  
            RETURN empty list  
        END IF  
        
        FUNCTION is_leaf(node)  
            RETURN left child of node EQUALS none AND right child of node EQUALS none  
        END FUNCTION  
        
        SET left_boundary TO empty list  
        SET current TO left child of root  
        WHILE current holds value  
            IF NOT is_leaf(current)  
                APPEND value of current TO left_boundary  
            END IF  
            IF left child of current holds value  
                SET current TO left child of current  
            ELSE  
                SET current TO right child of current  
            END IF  
        END WHILE  
        
        SET right_boundary TO empty list  
        SET current TO right child of root  
        WHILE current holds value  
            IF NOT is_leaf(current)  
                APPEND value of current TO right_boundary  
            END IF  
            IF right child of current holds value  
                SET current TO right child of current  
            ELSE  
                SET current TO left child of current  
            END IF  
        END WHILE  
        
        SET leaves TO empty list  
        FUNCTION collect_leaves(node)  
            IF node EQUALS none  
                RETURN  
            END IF  
            IF is_leaf(node)  
                APPEND value of node TO leaves  
            END IF  
            CALL collect_leaves(left child of node)  
            CALL collect_leaves(right child of node)  
        END FUNCTION  
        
        CALL collect_leaves(root)  
        
        IF is_leaf(root)  
            SET leaves TO empty list  
        END IF  
        
        RETURN list created by concatenating value of root WITH left_boundary WITH leaves WITH elements of right_boundary in reverse order  
    END FUNCTION  
END CLASS
