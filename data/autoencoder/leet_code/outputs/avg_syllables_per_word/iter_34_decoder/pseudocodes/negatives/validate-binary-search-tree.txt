CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulted to zero AND left defaulted to none AND right defaulted to none  
        SET the val property TO val  
        SET the left property TO left  
        SET the right property TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the values collection is empty THEN  
        RETURN none  
    END IF  
    SET root TO a new TreeNode object initialized with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty DO  
        SET node TO the element obtained by removing the leftmost element of queue  
        IF i is less than the length of values AND the element at position i of values is not none THEN  
            SET the left property of node TO a new TreeNode object initialized with the element at position i of values  
            APPEND the left property of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not none THEN  
            SET the right property of node TO a new TreeNode object initialized with the element at position i of values  
            APPEND the right property of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p AND q  
    IF p is none AND q is none THEN  
        RETURN true  
    ELSE IF p is none OR q is none THEN  
        RETURN false  
    ELSE IF the val property of p NOT EQUALS the val property of q THEN  
        RETURN false  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with the left property of p AND the left property of q AND the result of calling is_same_tree with the right property of p AND the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST with parameter root  
        FUNCTION validate with parameters node AND low defaulted to negative infinity AND high defaulted to positive infinity  
            IF node is none THEN  
                RETURN true  
            END IF  
            IF the val property of node is LESS THAN OR EQUAL TO low OR the val property of node is GREATER THAN OR EQUAL TO high THEN  
                RETURN false  
            END IF  
            RETURN the logical AND of the result of calling validate with the left property of node AND low AND the val property of node AND the result of calling validate with the right property of node AND the val property of node AND high  
        END FUNCTION  
        RETURN the result of calling validate with root  
    END FUNCTION  
END CLASS
