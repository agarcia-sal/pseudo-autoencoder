CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the value attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node WITH PARAMETER values  
    IF the values list EQUALS empty list  
        RETURN None  
    END IF  
    SET head TO a new instance of ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values from position one TO the end  
        SET node TO a new instance of ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the logical AND of the equality of the value attribute of p1 AND the value attribute of p2 AND the result of calling is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION

CLASS Solution  
    FUNCTION removeNthFromEnd WITH PARAMETERS head AND n  
        SET dummy TO a new instance of ListNode initialized with zero  
        SET the next attribute of dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one  
            SET first TO the next attribute of first  
        END FOR  
        WHILE first NOT EQUALS None  
            SET first TO the next attribute of first  
            SET second TO the next attribute of second  
        END WHILE  
        SET the next attribute of second TO the next attribute of the next attribute of second  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
