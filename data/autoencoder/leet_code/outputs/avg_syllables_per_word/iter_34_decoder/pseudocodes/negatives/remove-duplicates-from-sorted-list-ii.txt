CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET the value property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node with parameter values  
    IF the values collection IS EMPTY  
        RETURN None  
    END IF  
    SET head TO a new ListNode with the element at position one of the values collection  
    SET p TO head  
    FOR each val IN the values collection from position two TO the last position  
        SET node TO a new ListNode with val as the value  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the logical AND of the condition that the value property of p1 EQUALS the value property of p2 AND the result of is_same_list invoked with the next property of p1 AND the next property of p2  
END FUNCTION

CLASS Solution  
    FUNCTION deleteDuplicates with parameters self AND head  
        SET dummy TO a new ListNode with zero as the value AND head as the next property  
        SET prev TO dummy  
        WHILE head IS NOT None  
            IF the next property of head IS NOT None AND the value property of head EQUALS the value property of the next property of head  
                WHILE the next property of head IS NOT None AND the value property of head EQUALS the value property of the next property of head  
                    SET head TO the next property of head  
                END WHILE  
                SET the next property of prev TO the next property of head  
            ELSE  
                SET prev TO the next property of prev  
            END IF  
            SET head TO the next property of head  
        END WHILE  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
