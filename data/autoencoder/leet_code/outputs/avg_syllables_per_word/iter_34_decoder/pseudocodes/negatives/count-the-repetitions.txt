CLASS Solution  
    FUNCTION getMaxRepetitions(s1 s one, n1 n one, s2 s two, n2 n two)  
        IF n one EQUALS zero THEN  
            RETURN zero  
        END IF  

        SET s two count TO a list of zeros with the length EQUAL TO the LENGTH OF s two PLUS one  
        SET index map TO a list of zeros with the length EQUAL TO the LENGTH OF s two PLUS one  

        SET current index s two TO zero  
        SET s one count TO zero  

        WHILE s one count LESS THAN n one  
            INCREMENT s one count BY one  
            FOR each character IN s1  
                IF character EQUALS element at position current index s two of s two THEN  
                    INCREMENT current index s two BY one  
                    IF current index s two EQUALS the LENGTH OF s two THEN  
                        SET element at position s one count of s two count TO element at position s one count MINUS one of s two count PLUS one  
                        SET current index s two TO zero  
                    END IF  
                END IF  
                SET element at position s one count of index map TO current index s two  
            END FOR  

            IF current index s two IS IN elements from position one TO position s one count MINUS one of index map THEN  
                SET start TO the POSITION OF current index s two IN index map  
                SET cycle length TO s one count MINUS start  
                SET cycles fit TO (n one MINUS start) DIVIDED BY cycle length  
                SET remainder TO (n one MINUS start) MODULO cycle length  
                SET total s two count TO element at position start of s two count PLUS cycles fit MULTIPLIED BY (element at position start PLUS cycle length of s two count MINUS element at position start of s two count) PLUS element at position start PLUS remainder of s two count MINUS element at position start of s two count  
                RETURN total s two count DIVIDED BY n two  
            END IF  
        END WHILE  

        RETURN element at position s one count of s two count DIVIDED BY n two  
    END FUNCTION  
END CLASS
