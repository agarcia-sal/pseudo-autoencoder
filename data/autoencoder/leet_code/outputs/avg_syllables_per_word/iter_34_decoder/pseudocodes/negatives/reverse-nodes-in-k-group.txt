CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty list  
        RETURN none  
    END IF  
    SET head TO new ListNode object with value as element at position one of values and next as none  
    SET p TO head  
    FOR each val IN elements of values starting from position two to the end  
        SET node TO new ListNode object with value as val and next as none  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the expression that element val at p1 EQUALS element val at p2 AND function call is_same_list with arguments element next at p1 and element next at p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseKGroup(head, k)  
        FUNCTION reverseLinkedList(start, end)  
            SET prev TO none  
            SET curr TO start  
            WHILE curr NOT EQUALS end  
                SET next_node TO element next at curr  
                SET next of curr TO prev  
                SET prev TO curr  
                SET curr TO next_node  
            END WHILE  
            SET next of curr TO prev  
            RETURN end AND start  
        END FUNCTION  

        SET dummy TO new ListNode object with value zero and next as none  
        SET next of dummy TO head  
        SET prev_group_end TO dummy  

        WHILE true holds  
            SET kth_node TO prev_group_end  
            FOR index FROM one TO k  
                SET kth_node TO element next at kth_node  
                IF kth_node EQUALS none  
                    RETURN element next at dummy  
                END IF  
            END FOR  

            SET next_group_start TO element next at kth_node  
            SET next of kth_node TO none  
            CALL reverseLinkedList with arguments element next at prev_group_end and kth_node and ASSIGN results TO new_group_start AND new_group_end  

            SET next of prev_group_end TO new_group_start  
            SET next of new_group_end TO next_group_start  
            SET prev_group_end TO new_group_end  
        END WHILE  

        RETURN element next at dummy  
    END FUNCTION  
END CLASS
