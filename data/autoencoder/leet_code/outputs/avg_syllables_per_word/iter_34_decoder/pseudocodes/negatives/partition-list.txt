CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self val TO val  
        SET self next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO NEW ListNode WITH element at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO the end of values  
        SET node TO NEW ListNode WITH val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND is_same_list(next of p1 next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION partition(self head x)  
        SET less_head TO NEW ListNode WITH zero  
        SET greater_head TO NEW ListNode WITH zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current NOT EQUALS None  
            IF the value of current LESS THAN x  
                SET next of less TO current  
                SET less TO next of less  
            ELSE  
                SET next of greater TO current  
                SET greater TO next of greater  
            END IF  
            SET current TO next of current  
        END WHILE  
        SET next of less TO next of greater_head  
        SET next of greater TO None  
        RETURN next of less_head  
    END FUNCTION  
END CLASS
