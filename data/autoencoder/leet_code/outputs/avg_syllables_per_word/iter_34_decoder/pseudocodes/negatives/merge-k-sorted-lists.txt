CLASS ListNode  
    FUNCTION __init__(self val SET TO zero next SET TO None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET head TO a new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values from position two TO the last position  
        SET node TO a new ListNode with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the element val of p1 EQUALS the element val of p2 AND is_same_list(the element next of p1 the element next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists(self lists)  
        SET min_heap TO an empty list  
        SET i TO one  
        FOR each lst IN lists WITH INDEX i FROM one TO the LENGTH OF lists  
            IF lst NOT EQUALS None THEN  
                ADD the triple of the element val of lst AND i AND lst TO min_heap using heap push operation  
            END IF  
            INCREMENT i BY one  
        END FOR  
        SET dummy TO a new ListNode with default values  
        SET current TO dummy  
        WHILE min_heap is NOT empty DO  
            REMOVE the triple with the smallest first element from min_heap using heap pop operation and SET it TO val i node  
            SET the next property of current TO node  
            SET current TO the next property of current  
            IF the element next of node NOT EQUALS None THEN  
                ADD the triple of the element val of the element next of node AND i AND the element next of node TO min_heap using heap push operation  
            END IF  
        END WHILE  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
