CLASS Solution  
    FUNCTION findNumOfValidWords(words words LIST OF STRING puzzles puzzles LIST OF STRING) RETURNS LIST OF INTEGER  
        FUNCTION to_mask(s s STRING) RETURNS INTEGER  
            SET mask TO zero  
            FOR each char IN s  
                SET mask TO mask OR one SHIFTED LEFT BY the numeric position of char MINUS the numeric position of letter a  
            END FOR  
            RETURN mask  
        END FUNCTION  
        
        SET word_count TO a dictionary with default value zero  
        FOR each word IN words  
            SET mask TO to_mask with word  
            IF the number of bits set in mask IS LESS THAN OR EQUAL TO seven  
                INCREMENT the value in word_count at mask BY one  
            END IF  
        END FOR  
        
        SET result TO an empty list  
        FOR each puzzle IN puzzles  
            SET first_char_mask TO one SHIFTED LEFT BY the numeric position of element at position zero of puzzle MINUS the numeric position of letter a  
            SET puzzle_mask TO to_mask of the substring from position one TO the end of puzzle  
            SET count TO the value in word_count at first_char_mask  
            
            SET subset TO puzzle_mask  
            WHILE subset IS NOT zero  
                SET count TO count PLUS the value in word_count at the bitwise OR of subset AND first_char_mask  
                SET subset TO the bitwise AND of subset MINUS one AND puzzle_mask  
            END WHILE  
            
            APPEND count TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
