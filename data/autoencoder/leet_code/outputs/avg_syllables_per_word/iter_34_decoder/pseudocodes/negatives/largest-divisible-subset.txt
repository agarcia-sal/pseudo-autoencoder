CLASS Solution  
    FUNCTION largestDivisibleSubset(nums)  
        SORT nums IN ascending order  
        SET n TO the LENGTH OF nums  
        SET f TO a list of length n with all elements equal TO one  
        SET k TO zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF the element at position i of nums MODULO the element at position j of nums EQUALS zero THEN  
                    SET f at position i TO the GREATER OF the value at position i of f AND the value at position j of f PLUS one  
                END IF  
            END FOR  
            IF the value at position k of f IS LESS THAN the value at position i of f THEN  
                SET k TO i  
            END IF  
        END FOR  
        SET m TO the value at position k of f  
        SET i TO k  
        SET ans TO an empty list  
        WHILE m IS GREATER THAN zero  
            IF the element at position k of nums MODULO the element at position i of nums EQUALS zero AND the value at position i of f EQUALS m THEN  
                APPEND the element at position i of nums TO ans  
                SET k TO i  
                DECREMENT m BY one  
            END IF  
            DECREMENT i BY one  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
