CLASS ListNode  
    FUNCTION __init__(self val DEFAULT zero next DEFAULT none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements from position one TO the last position of values  
        SET node TO A NEW ListNode WITH val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN False  
    END IF  
    RETURN the condition that the val attribute of p1 EQUALS the val attribute of p2 AND is_same_list called with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight(head k)  
        IF head EQUALS none OR the next attribute of head EQUALS none OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE the next attribute of tail IS NOT none  
            SET tail TO the next attribute of tail  
            INCREMENT length BY one  
        END WHILE  
        SET the next attribute of tail TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR each index FROM zero TO steps_to_new_tail MINUS one  
            SET new_tail TO the next attribute of new_tail  
        END FOR  
        SET new_head TO the next attribute of new_tail  
        SET the next attribute of new_tail TO none  
        RETURN new_head  
    END FUNCTION  
END CLASS
