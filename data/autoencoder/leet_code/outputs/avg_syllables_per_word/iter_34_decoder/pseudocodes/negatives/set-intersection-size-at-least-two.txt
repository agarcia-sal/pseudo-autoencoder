CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals BY the element at position two of each interval AND BY the negative of the element at position one of each interval IN descending order  
        SET first TO the element at position two of the first interval MINUS one  
        SET second TO the element at position two of the first interval  
        SET count TO two  
        FOR each interval FROM the second interval TO the last interval  
            LET start BE the element at position one of the current interval  
            LET end BE the element at position two of the current interval  
            IF start IS GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
