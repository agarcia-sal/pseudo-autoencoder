CLASS Solution  
    FUNCTION minChanges(nums AS list OF integers, k AS integer) RETURNS integer  
        SET groups TO a dictionary having default empty list  
        FOR i FROM zero TO the length of nums MINUS one  
            APPEND element at position i of nums TO element at position i MINUS the integer division of i BY k MULTIPLIED BY k of groups  
        END FOR  
        
        SET freq TO a list containing k dictionaries having default zero integer  
        SET min_size TO a list containing k elements each initialized to infinity  
        
        FOR i FROM zero TO k MINUS one  
            FOR each num IN element at position i of groups  
                INCREMENT element at key num of element at position i of freq BY one  
            END FOR  
            SET element at position i of min_size TO the length of element at position i of groups  
        END FOR  
        
        SET dp TO a list containing two raised to the power of ten elements each initialized to infinity  
        SET element at position zero of dp TO zero  
        
        FOR i FROM zero TO k MINUS one  
            SET new_dp TO a list containing two raised to the power of ten elements each initialized to infinity  
            SET min_dp TO the minimum value in dp  
            FOR j FROM zero TO two raised to the power of ten MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position i of min_size  
                FOR each pair num and count IN element at position i of freq  
                    SET candidate TO element at position j of dp PLUS element at position i of min_size MINUS count  
                    SET candidate TO candidate PLUS zero  // Expression to emphasize the computation as natural number  
                    IF candidate LESS THAN element at position j of new_dp  
                        SET element at position j of new_dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
