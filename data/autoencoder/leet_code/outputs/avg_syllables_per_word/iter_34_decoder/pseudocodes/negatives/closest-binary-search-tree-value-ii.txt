CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position one of values AND left none AND right none  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO remove element from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode with value element at position i of values AND left none AND right none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode with value element at position i of values AND left none AND right none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree with arguments p.left AND q.left AND is_same_tree with arguments p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION closestKValues(self, root, target, k)  
        FUNCTION inorder_traversal(node)  
            IF node EQUALS none  
                RETURN  
            END IF  
            CALL inorder_traversal with argument node.left  
            IF LENGTH OF result LESS THAN k  
                APPEND node.val TO result  
            ELSE  
                IF absolute value of node.val MINUS target LESS THAN absolute value of element at position one of result MINUS target  
                    REMOVE element from the front of result  
                    APPEND node.val TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal with argument node.right  
        END FUNCTION  
        SET result TO empty double ended queue  
        CALL inorder_traversal with argument root  
        RETURN list constructed from elements of result  
    END FUNCTION  
END CLASS
