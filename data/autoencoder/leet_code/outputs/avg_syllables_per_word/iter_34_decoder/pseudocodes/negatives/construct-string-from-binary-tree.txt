CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater than zero  
        REMOVE the first element FROM queue and ASSIGN IT TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.left TO a new TreeNode constructed with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.right TO a new TreeNode constructed with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the value field of p NOT EQUALS the value field of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(the left field of p, the left field of q) AND is_same_tree(the right field of p, the right field of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str(root)  
        FUNCTION helper(node)  
            IF node EQUALS none  
                RETURN empty string  
            END IF  
            SET result TO the string conversion of the value field of node  
            IF the left field of node EXISTS OR (the left field of node DOES NOT EXIST AND the right field of node EXISTS)  
                SET result TO result PLUS an opening parenthesis PLUS helper(the left field of node) PLUS a closing parenthesis  
            END IF  
            IF the right field of node EXISTS  
                SET result TO result PLUS an opening parenthesis PLUS helper(the right field of node) PLUS a closing parenthesis  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN helper(root)  
    END FUNCTION  
END CLASS
