CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode CREATED WITH the element at position one of values  
    SET i TO one  
    SET queue TO a new double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the FIRST element REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET the left child of node TO a new TreeNode CREATED WITH the element at position i of values  
            APPEND the left child of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET the right child of node TO a new TreeNode CREATED WITH the element at position i of values  
            APPEND the right child of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val attribute of p IS NOT EQUALS TO the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the AND of the result of is_same_tree CALLED WITH the left child of p AND the left child of q AND the result of is_same_tree CALLED WITH the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom WITH PARAMETERS self root  
        IF root IS None  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a new double-ended queue CONTAINING root  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR counter FROM zero TO level_size MINUS one  
                SET node TO the FIRST element REMOVED FROM queue  
                APPEND the val attribute of node TO current_level  
                IF the left child of node IS NOT None  
                    APPEND the left child of node TO queue  
                END IF  
                IF the right child of node IS NOT None  
                    APPEND the right child of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result IN THE ORDER FROM LAST ELEMENT TO FIRST ELEMENT  
    END FUNCTION  
END CLASS
