CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    INITIALIZE queue as an empty double-ended collection  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not None THEN  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not None THEN  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p.left and q.left AND the result of is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS None THEN  
                RETURN zero  
            END IF  
            SET current_sum TO node.val PLUS subtree_sum with node.left PLUS subtree_sum with node.right  
            INCREMENT the value at key current_sum in count BY one  
            RETURN current_sum  
        END FUNCTION  
        
        INITIALIZE count as a default dictionary with integer default value  
        CALL subtree_sum with root  
        
        SET max_freq TO the maximum value among all values in count with a default of zero  
        
        SET result TO a list of keys from count where the corresponding value EQUALS max_freq  
        
        RETURN result  
    END FUNCTION  
END CLASS
