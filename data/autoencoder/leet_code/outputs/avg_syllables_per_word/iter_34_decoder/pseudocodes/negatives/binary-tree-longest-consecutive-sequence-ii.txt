CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value THEN  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value THEN  
        RETURN TRUE  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN FALSE  
    ELSE IF the value of p IS NOT EQUAL TO the value of q THEN  
        RETURN FALSE  
    ELSE  
        RETURN the logical AND of the call to is_same_tree with p.left and q.left AND the call to is_same_tree with p.right and q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION longestConsecutive(self, root)  
        FUNCTION dfs(node)  
            IF node IS no value THEN  
                RETURN zero, zero  
            END IF  
            SET inc_len TO one  
            SET dec_len TO one  
            SET left_inc, left_dec TO the result of dfs called with node.left  
            SET right_inc, right_dec TO the result of dfs called with node.right  
            IF node.left IS NOT no value THEN  
                IF the value of node EQUALS the value of node.left PLUS one THEN  
                    SET dec_len TO left_dec PLUS one  
                ELSE IF the value of node EQUALS the value of node.left MINUS one THEN  
                    SET inc_len TO left_inc PLUS one  
                END IF  
            END IF  
            IF node.right IS NOT no value THEN  
                IF the value of node EQUALS the value of node.right PLUS one THEN  
                    SET dec_len TO the maximum of dec_len AND right_dec PLUS one  
                ELSE IF the value of node EQUALS the value of node.right MINUS one THEN  
                    SET inc_len TO the maximum of inc_len AND right_inc PLUS one  
                END IF  
            END IF  
            SET max_len TO the maximum of max_len AND inc_len PLUS dec_len MINUS one  
            RETURN inc_len, dec_len  
        END FUNCTION  
        SET max_len TO zero  
        CALL dfs with root  
        RETURN max_len  
    END FUNCTION  
END CLASS
