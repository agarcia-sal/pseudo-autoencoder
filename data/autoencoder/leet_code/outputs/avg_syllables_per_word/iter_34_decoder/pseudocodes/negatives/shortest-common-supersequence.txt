CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1 parameter, str2 parameter)  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        SET dp TO a two dimensional list of zeroes with dimensions m PLUS one by n PLUS one  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2  
                    SET element at position i and j of dp TO element at position i MINUS one and j MINUS one of dp PLUS one  
                ELSE  
                    SET element at position i and j of dp TO the GREATER OF element at position i MINUS one and j of dp AND element at position i and j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        SET i TO m  
        SET j TO n  
        SET scs TO an empty list  
        
        WHILE i GREATER THAN zero AND j GREATER THAN zero  
            IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF element at position i MINUS one and j of dp GREATER THAN element at position i and j MINUS one of dp  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND the element at position j MINUS one of str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i GREATER THAN zero  
            APPEND the element at position i MINUS one of str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        
        WHILE j GREATER THAN zero  
            APPEND the element at position j MINUS one of str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        
        REVERSE the order of scs  
        
        RETURN the concatenation of all elements in scs  
    END FUNCTION  
END CLASS
