CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET head TO a new ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the end of values  
        SET node TO a new ListNode with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value of the element p1 EQUALS the value of the element p2 AND is_same_list of the next element of p1 WITH the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(self, l1, l2)  
        FUNCTION pushIntoStack(node)  
            SET stack TO an empty list  
            WHILE node DOES NOT EQUAL None  
                APPEND the value of the element node TO stack  
                SET node TO the next element of node  
            END WHILE  
            RETURN stack  
        END FUNCTION  

        SET stack1 TO pushIntoStack with l1  
        SET stack2 TO pushIntoStack with l2  

        SET carry TO zero  
        SET result TO None  

        WHILE the LENGTH OF stack1 GREATER THAN zero OR the LENGTH OF stack2 GREATER THAN zero OR carry NOT EQUALS zero  
            IF the LENGTH OF stack1 GREATER THAN zero  
                SET val1 TO the last element of stack1  
                REMOVE the last element FROM stack1  
            ELSE  
                SET val1 TO zero  
            END IF  

            IF the LENGTH OF stack2 GREATER THAN zero  
                SET val2 TO the last element of stack2  
                REMOVE the last element FROM stack2  
            ELSE  
                SET val2 TO zero  
            END IF  

            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO the result of total DIVIDED BY ten rounded down to the nearest integer  
            SET digit TO the remainder of total MODULO ten  

            SET new_node TO a new ListNode with digit  
            SET the next element of new_node TO result  
            SET result TO new_node  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
