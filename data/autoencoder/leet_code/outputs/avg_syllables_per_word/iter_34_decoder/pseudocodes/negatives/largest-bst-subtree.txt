CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with value element at position one of values, left no value, right no value  
    SET i TO one  
    SET queue TO a new empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value THEN  
            SET node.left TO a new TreeNode with value element at position i of values, left no value, right no value  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value THEN  
            SET node.right TO a new TreeNode with value element at position i of values, left no value, right no value  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value THEN  
        RETURN true  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with arguments p.left and q.left AND the result of is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree(self, root)  
        FUNCTION dfs(node)  
            IF node IS no value THEN  
                RETURN positive infinity, negative infinity, zero  
            END IF  
            SET left_min, left_max, left_size TO the result of dfs with argument node.left  
            SET right_min, right_max, right_size TO the result of dfs with argument node.right  
            IF left_max LESS THAN node.val AND node.val LESS THAN right_min THEN  
                RETURN the minimum of node.val and left_min, the maximum of node.val and right_max, left_size PLUS right_size PLUS one  
            END IF  
            RETURN negative infinity, positive infinity, the maximum of left_size and right_size  
        END FUNCTION  
        RETURN the third element of the result of dfs with argument root  
    END FUNCTION  
END CLASS
