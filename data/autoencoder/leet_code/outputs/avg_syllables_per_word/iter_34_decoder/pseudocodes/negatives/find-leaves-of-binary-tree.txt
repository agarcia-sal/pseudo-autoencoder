CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO new instance of TreeNode initialized with the element at position one of values  
    SET i TO one  
    INITIALIZE queue as an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty DO  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node dot left TO new instance of TreeNode initialized with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node dot right TO new instance of TreeNode initialized with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of calling is_same_tree with the left attribute of p AND the left attribute of q AND calling is_same_tree with the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves(self, root)  
        FUNCTION collectLeaves(node)  
            IF node EQUALS none THEN  
                RETURN minus one  
            END IF  
            SET left_height TO the result of calling collectLeaves with the left attribute of node  
            SET right_height TO the result of calling collectLeaves with the right attribute of node  
            SET current_height TO the greater value between left_height AND right_height PLUS one  
            IF current_height GREATER THAN OR EQUAL TO the LENGTH OF result THEN  
                APPEND an empty list TO result  
            END IF  
            APPEND the val attribute of node TO the element at position current_height of result  
            RETURN current_height  
        END FUNCTION  
        SET result TO an empty list  
        CALL collectLeaves with root  
        RETURN result  
    END FUNCTION  
END CLASS
