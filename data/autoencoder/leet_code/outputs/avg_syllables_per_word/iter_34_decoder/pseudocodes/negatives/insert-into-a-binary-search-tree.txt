CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET val attribute of self TO val  
        SET left attribute of self TO left  
        SET right attribute of self TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS nothing  
        RETURN nothing  
    END IF  
    SET root TO new instance of TreeNode with parameter element at position one of values  
    SET i TO one  
    SET queue TO new double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from left side of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing  
            SET left attribute of node TO new instance of TreeNode with parameter element at position i of values  
            APPEND left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing  
            SET right attribute of node TO new instance of TreeNode with parameter element at position i of values  
            APPEND right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN True  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN False  
    ELSE IF val attribute of p NOT EQUALS val attribute of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters left attribute of p left attribute of q AND is_same_tree with parameters right attribute of p right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST with parameters root val  
        FUNCTION dfs with parameter root  
            IF root EQUALS nothing  
                RETURN new instance of TreeNode with parameter val  
            END IF  
            IF val attribute of root LESS THAN val  
                SET right attribute of root TO dfs with parameter right attribute of root  
            ELSE  
                SET left attribute of root TO dfs with parameter left attribute of root  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN dfs with parameter root  
    END FUNCTION  
END CLASS
