CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten MULTIPLIED BY ten PLUS one  
        SET dp TO list of zeros with LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  
  
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS the character star  
                RETURN nine  
            ELSE IF char EQUALS the character zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
  
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS the character star AND char2 EQUALS the character star  
                RETURN fifteen  
            ELSE IF char1 EQUALS the character star  
                IF char2 LESS THAN OR EQUAL TO the character six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS the character star  
                IF char1 EQUALS the character one  
                    RETURN nine  
                ELSE IF char1 EQUALS the character two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer value of the concatenation of char1 and char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
  
        FOR i FROM one TO LENGTH OF s  
            SET dp at position i TO dp at position i PLUS dp at position i MINUS one MULTIPLIED BY ways_to_decode_single(the element at position i MINUS one of s)  
            SET dp at position i TO dp at position i MODULO MOD  
            IF i GREATER THAN one  
                SET dp at position i TO dp at position i PLUS dp at position i MINUS two MULTIPLIED BY ways_to_decode_pair(the element at position i MINUS two of s, the element at position i MINUS one of s)  
                SET dp at position i TO dp at position i MODULO MOD  
            END IF  
        END FOR  
  
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
