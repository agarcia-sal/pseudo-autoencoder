CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO a collection mapping each character in s to its frequency  
        SET max_heap TO an empty list  
        FOR each character and frequency IN the collection of count  
            APPEND the pair of the negative of the frequency and the character TO max_heap  
        END FOR  
        TRANSFORM max_heap INTO a heap structure preserving the heap invariant  
        
        SET prev_char TO a value representing none  
        SET prev_freq TO zero  
        SET result TO an empty list  
        
        WHILE the heap max_heap is not empty OR prev_freq is less than zero holds  
            IF the heap max_heap is empty AND prev_freq is less than zero holds  
                RETURN an empty string  
            END IF  
            
            REMOVE the element with the smallest frequency value FROM max_heap and ASSIGN its frequency TO freq and character TO char  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq is less than zero holds  
                ADD the pair of prev_freq and prev_char INTO the heap max_heap  
            END IF  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        
        RETURN the concatenation of all characters in result as a string  
    END FUNCTION  
END CLASS
