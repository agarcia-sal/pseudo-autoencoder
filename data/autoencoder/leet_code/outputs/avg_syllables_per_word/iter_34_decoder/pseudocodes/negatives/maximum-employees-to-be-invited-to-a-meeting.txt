CLASS Solution  
    FUNCTION maximumInvitations(favorite)  
        SET n TO the LENGTH OF favorite  
        SET graph TO an EMPTY DEFAULTDICTIONARY OF LISTS  
        SET in_degree TO a LIST OF n ZEROS  
        
        FOR i FROM zero TO n MINUS one  
            APPEND i TO the LIST at element at position i of favorite IN graph  
            INCREMENT the element at position i of favorite IN in_degree BY one  
        END FOR  
        
        SET mutual_chains TO zero  
        SET visited TO a LIST OF n FALSE VALUES  
        
        FOR i FROM zero TO n MINUS one  
            IF the element at position the element at position i of favorite IN favorite EQUALS i AND i LESS THAN the element at position i of favorite  
                SET a TO i  
                SET b TO the element at position i of favorite  
                SET chain_a TO CALL find_chain_length WITH a graph visited  
                SET chain_b TO CALL find_chain_length WITH b graph visited  
                INCREMENT mutual_chains BY chain_a PLUS chain_b  
            END IF  
        END FOR  
        
        SET longest_cycle TO zero  
        SET visited TO a LIST OF n FALSE VALUES  
        
        FOR i FROM zero TO n MINUS one  
            IF the element at position i IN visited EQUALS FALSE  
                SET cycle_length TO CALL find_cycle_length WITH i graph visited favorite  
                IF cycle_length GREATER THAN longest_cycle  
                    SET longest_cycle TO cycle_length  
                END IF  
            END IF  
        END FOR  
        
        IF mutual_chains GREATER THAN longest_cycle  
            RETURN mutual_chains  
        ELSE  
            RETURN longest_cycle  
        END IF  
    END FUNCTION  
    
    FUNCTION find_chain_length(start graph visited)  
        SET length TO zero  
        SET queue TO a NEW DEQUE CONTAINING start  
        SET the element at position start IN visited TO TRUE  
        
        WHILE queue IS NOT EMPTY  
            REMOVE the FIRST ELEMENT FROM queue AND ASSIGN TO current  
            FOR each neighbor IN the LIST at element at position current IN graph  
                IF the element at position neighbor IN visited EQUALS FALSE  
                    SET the element at position neighbor IN visited TO TRUE  
                    APPEND neighbor TO queue  
                    INCREMENT length BY one  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN length  
    END FUNCTION  
    
    FUNCTION find_cycle_length(start graph visited favorite)  
        IF the element at position start IN visited EQUALS TRUE  
            RETURN zero  
        END IF  
        
        SET stack TO an EMPTY LIST  
        SET current TO start  
        
        WHILE the element at position current IN visited EQUALS FALSE  
            SET the element at position current IN visited TO TRUE  
            APPEND current TO stack  
            SET current TO the element at position current IN favorite  
        END WHILE  
        
        IF current NOT IN stack  
            RETURN zero  
        END IF  
        
        SET cycle_start_index TO the POSITION OF current IN stack  
        SET cycle_length TO the LENGTH OF stack MINUS cycle_start_index  
        
        RETURN cycle_length  
    END FUNCTION  
END CLASS
