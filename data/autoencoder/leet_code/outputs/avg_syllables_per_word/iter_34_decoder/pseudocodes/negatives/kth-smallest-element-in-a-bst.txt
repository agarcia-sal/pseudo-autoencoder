CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        ASSIGN reference TO self.left AS left  
        ASSIGN reference TO self.right AS right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with the element at position one of values as val and no left or right  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater than zero  
        REMOVE the element at the front of queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            ASSIGN reference TO node.left AS a new TreeNode with the element at position i of values as val and no left or right  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            ASSIGN reference TO node.right AS a new TreeNode with the element at position i of values as val and no left or right  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF the value stored in p.val NOT EQUALS the value stored in q.val THEN  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with p.left and q.left AND the result of calling is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(self, root, k)  
        SET stack TO an empty list  
        ASSIGN reference TO current AS root  
        SET count TO zero  
        WHILE the LENGTH OF stack GREATER THAN zero OR current IS NOT none  
            WHILE current IS NOT none  
                APPEND current TO stack  
                ASSIGN reference TO current AS the left child of current  
            END WHILE  
            REMOVE the last element FROM stack AND ASSIGN TO current  
            INCREMENT count BY one  
            IF count EQUALS k THEN  
                RETURN the value stored in current.val  
            END IF  
            ASSIGN reference TO current AS the right child of current  
        END WHILE  
    END FUNCTION  
END CLASS
