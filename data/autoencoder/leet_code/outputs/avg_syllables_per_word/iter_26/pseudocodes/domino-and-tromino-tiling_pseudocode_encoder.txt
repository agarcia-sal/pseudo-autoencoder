CLASS Solution
    FUNCTION numTilings(parameter n AS integer)
        SET MODULO TO ten RAISED TO THE power OF nine PLUS one

        CREATE list dp WITH length n PLUS one WITH default value zero
        CREATE list dp1 WITH length n PLUS one WITH default value zero

        SET element at position zero OF list dp TO one
        SET element at position one OF list dp TO one

        IF n GREATER THAN OR EQUAL TO two HOLDS
            SET element at position two OF list dp TO two
            SET element at position two OF list dp1 TO one
        END IF

        FOR index i FROM three TO n INCLUSIVE
            SET element at position i OF list dp TO element at position i MINUS one OF list dp PLUS element at position i MINUS two OF list dp PLUS two MULTIPLIED BY element at position i MINUS one OF list dp1 CONFINED TO MODULO MODULO
            SET element at position i OF list dp1 TO element at position i MINUS two OF list dp PLUS element at position i MINUS one OF list dp1 CONFINED TO MODULO MODULO
        END FOR

        RETURN element at position n OF list dp
    END FUNCTION
END CLASS
