CLASS Solution
    FUNCTION minimumDistance(word)
        FUNCTION distance(character_one, character_two)
            IF character_one EQUALS None
                RETURN zero
            END IF
            SET position_one_row TO the INTEGER DIVISION OF the UNICODE CODE POINT OF character_one MINUS the UNICODE CODE POINT OF uppercase A BY six
            SET position_one_column TO the REMAINDER OF the UNICODE CODE POINT OF character_one MINUS the UNICODE CODE POINT OF uppercase A DIVIDED BY six
            SET position_two_row TO the INTEGER DIVISION OF the UNICODE CODE POINT OF character_two MINUS the UNICODE CODE POINT OF uppercase A BY six
            SET position_two_column TO the REMAINDER OF the UNICODE CODE POINT OF character_two MINUS the UNICODE CODE POINT OF uppercase A DIVIDED BY six
            RETURN the ABSOLUTE VALUE OF position_one_row MINUS position_two_row PLUS the ABSOLUTE VALUE OF position_one_column MINUS position_two_column
        END FUNCTION
        
        DECORATE dp WITH least recently used cache WITH unlimited capacity
        
        FUNCTION dp(index, finger_one_character, finger_two_character)
            IF index EQUALS the LENGTH OF word
                RETURN zero
            END IF
            SET current_character TO the ELEMENT at position index OF word
            SET distance_one TO the CALL OF distance WITH finger_one_character AND current_character PLUS the CALL OF dp WITH index PLUS one AND current_character AND finger_two_character
            SET distance_two TO the CALL OF distance WITH finger_two_character AND current_character PLUS the CALL OF dp WITH index PLUS one AND finger_one_character AND current_character
            RETURN the MINIMUM OF distance_one AND distance_two
        END FUNCTION
        
        RETURN the CALL OF dp WITH zero AND None AND None
END CLASS
