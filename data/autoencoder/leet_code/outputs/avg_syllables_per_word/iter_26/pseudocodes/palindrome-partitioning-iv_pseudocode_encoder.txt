CLASS Solution
    FUNCTION checkPartitioning(s)
        
        FUNCTION is_palindrome(sub)
            RETURN sub EQUALS the REVERSED FORM OF sub
        END FUNCTION
        
        SET n TO the LENGTH OF s
        
        SET is_pal TO a two dimensional list of boolean FALSE values with dimensions n BY n
        
        FOR i FROM n MINUS one DOWN TO zero INCLUSIVE
            FOR j FROM i TO n MINUS one INCLUSIVE
                IF the element at position i OF s EQUALS the element at position j OF s AND 
                   (the DIFFERENCE BETWEEN j AND i IS LESS THAN OR EQUAL TO one OR the element at position i PLUS one AND position j MINUS one OF is_pal HOLDS TRUE)
                    SET the element at row i AND column j OF is_pal TO TRUE
                END IF
            END FOR
        END FOR
        
        FOR i FROM one TO n MINUS two INCLUSIVE
            IF the element at row zero AND column i MINUS one OF is_pal HOLDS TRUE
                FOR j FROM i TO n MINUS two INCLUSIVE
                    IF the element at row i AND column j OF is_pal HOLDS TRUE AND the element at row j PLUS one AND column n MINUS one OF is_pal HOLDS TRUE
                        RETURN TRUE
                    END IF
                END FOR
            END IF
        END FOR
        
        RETURN FALSE
    END FUNCTION
END CLASS
