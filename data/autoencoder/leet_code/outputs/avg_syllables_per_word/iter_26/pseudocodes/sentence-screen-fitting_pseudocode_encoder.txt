CLASS Solution
    FUNCTION wordsTyping(sentence, rows, cols)
        SET s TO the concatenation of all elements of sentence separated by a single space PLUS a single space character at the end
        SET total_length TO the LENGTH OF s
        SET position TO zero

        FOR each unused_index FROM one TO rows INCLUSIVE
            INCREMENT position BY cols

            IF the element at position MODULO total_length of s EQUALS a single space character
                INCREMENT position BY one
            ELSE
                WHILE position GREATER THAN zero AND the element at (position MINUS one) MODULO total_length of s NOT EQUALS a single space character HOLDS
                    DECREMENT position BY one
                END WHILE
            END IF
        END FOR

        RETURN position DIVIDED BY total_length
    END FUNCTION
END CLASS
