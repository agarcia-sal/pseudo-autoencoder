CLASS Solution
    FUNCTION minTaps(number_n, list_ranges)
        SET list_max_range TO a new list of zeroes with LENGTH OF number_n PLUS one

        FOR index_i FROM zero TO number_n INCLUSIVE
            IF element at position index_i OF list_ranges GREATER THAN zero THEN
                SET value_left TO the GREATER OF zero AND index_i MINUS element at position index_i OF list_ranges
                SET value_right TO the LESSER OF number_n AND index_i PLUS element at position index_i OF list_ranges
                SET element at position value_left OF list_max_range TO the GREATER OF element at position value_left OF list_max_range AND value_right
            END IF
        END FOR

        SET variable_taps TO zero
        SET variable_current_end TO zero
        SET variable_farthest TO zero

        FOR index_i FROM zero TO number_n INCLUSIVE
            IF index_i GREATER THAN variable_farthest THEN
                RETURN the negative of one
            END IF
            IF index_i GREATER THAN variable_current_end THEN
                INCREMENT variable_taps BY one
                SET variable_current_end TO variable_farthest
            END IF
            SET variable_farthest TO the GREATER OF variable_farthest AND element at position index_i OF list_max_range
        END FOR

        RETURN variable_taps
    END FUNCTION
END CLASS
