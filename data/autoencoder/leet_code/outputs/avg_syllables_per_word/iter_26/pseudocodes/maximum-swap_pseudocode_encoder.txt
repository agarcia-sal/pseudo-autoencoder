CLASS Solution
    FUNCTION maximumSwap(num)
        SET digits TO the LIST OF THE STRING REPRESENTATION OF num
        SET last TO a DICTIONARY MAPPING each digit converted TO a number TO its last POSITION index IN digits
        
        FOR each index i AND digit d IN the ENUMERATION OF digits
            FOR each digit_candidate d2 FROM nine DOWN TO digit d PLUS one EXCLUSIVE BY negative one
                IF the VALUE ASSOCIATED TO digit_candidate d2 IN last EXISTING OR zero GREATER THAN index i HOLDS
                    EXCHANGE the elements at position i AND the element at position last[d2] IN digits
                    RETURN the INTEGER CONVERSION OF the CONCATENATION OF all elements IN digits
                END IF
            END FOR
        END FOR
        
        RETURN num
    END FUNCTION
END CLASS
