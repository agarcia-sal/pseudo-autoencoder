CLASS Solution
    FUNCTION minCostToSupplyWater(n, wells, pipes)
        FOR index FROM zero TO n MINUS one INCLUSIVE
            APPEND list containing zero, index PLUS one, element at position index OF wells TO pipes
        END FOR

        SORT pipes ACCORDING TO the element at position two OF each item IN ascending order

        SET parent TO list of elements FROM zero TO n INCLUSIVE

        FUNCTION find(x)
            IF element at position x OF parent NOT EQUALS x THEN
                SET element at position x OF parent TO the RESULT OF find(element at position x OF parent)
            END IF
            RETURN element at position x OF parent
        END FUNCTION

        FUNCTION union(x, y)
            SET rootX TO RESULT OF find(x)
            SET rootY TO RESULT OF find(y)
            IF rootX NOT EQUALS rootY THEN
                SET element at position rootX OF parent TO rootY
                RETURN true
            END IF
            RETURN false
        END FUNCTION

        SET min_cost TO zero
        FOR each element house1, house2, cost IN pipes
            IF union(house1, house2) HOLDS THEN
                INCREMENT min_cost BY cost
            END IF
        END FOR

        RETURN min_cost
    END FUNCTION
END CLASS
