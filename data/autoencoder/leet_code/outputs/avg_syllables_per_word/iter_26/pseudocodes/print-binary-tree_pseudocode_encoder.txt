CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty THEN
        RETURN none
    END IF
    SET root TO A NEW TreeNode WITH value AS the FIRST element OF values
    SET i TO one
    SET queue TO AN EMPTY double ended queue
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO THE FIRST element REMOVED FROM queue
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none THEN
            SET node.left TO A NEW TreeNode WITH value AS element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none THEN
            SET node.right TO A NEW TreeNode WITH value AS element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none THEN
        RETURN true
    ELSE IF p IS none OR q IS none THEN
        RETURN false
    ELSE IF p.val IS NOT EQUAL TO q.val THEN
        RETURN false
    ELSE
        RETURN is_same_tree WITH parameters p.left AND q.left AND is_same_tree WITH parameters p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION printTree(self, root)
        FUNCTION getHeight(node)
            IF node IS none THEN
                RETURN zero
            END IF
            RETURN one PLUS THE GREATER OF getHeight WITH parameter node.left AND getHeight WITH parameter node.right
        END FUNCTION

        SET height TO getHeight WITH parameter root
        SET width TO two RAISED TO THE POWER OF height MINUS one
        SET res TO A LIST OF height ELEMENTS WHERE EACH ELEMENT IS A LIST OF width ELEMENTS CONTAINING EMPTY STRINGS

        FUNCTION placeNode(node, row, left, right)
            IF node IS none THEN
                RETURN
            END IF
            SET mid TO THE INTEGER RESULT OF left PLUS right DIVIDED BY two
            SET element at position mid OF element at position row OF res TO THE STRING REPRESENTATION OF node.val
            CALL placeNode WITH parameters node.left, row PLUS one, left, mid
            CALL placeNode WITH parameters node.right, row PLUS one, mid, right
        END FUNCTION

        CALL placeNode WITH parameters root, zero, zero, width
        RETURN res
    END FUNCTION
END CLASS
