CLASS Solution
    FUNCTION strangePrinter(s)
        FUNCTION dp(i, j, memo)
            IF i GREATER THAN j THEN
                RETURN zero
            END IF
            IF the PAIR OF i AND j IS IN memo THEN
                RETURN the VALUE ASSOCIATED WITH the PAIR OF i AND j IN memo
            END IF

            SET result TO the CALL OF dp WITH i AND j MINUS one AND memo PLUS one

            FOR k FROM i TO j MINUS one INCLUSIVE
                IF the CHARACTER AT POSITION k OF s EQUALS the CHARACTER AT POSITION j OF s THEN
                    SET result TO the MINIMUM OF result AND the SUM OF the CALL OF dp WITH i AND k MINUS one AND memo AND the CALL OF dp WITH k AND j MINUS one AND memo
                END IF
            END FOR

            SET the VALUE ASSOCIATED WITH the PAIR OF i AND j IN memo TO result
            RETURN result
        END FUNCTION

        RETURN the CALL OF dp WITH zero AND the LENGTH OF s MINUS one AND an EMPTY DICTIONARY
    END FUNCTION
END CLASS
