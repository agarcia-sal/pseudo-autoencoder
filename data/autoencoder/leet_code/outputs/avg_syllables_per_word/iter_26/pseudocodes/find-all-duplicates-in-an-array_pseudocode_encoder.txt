CLASS Solution
    FUNCTION findDuplicates(list_of_numbers)
        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one INCLUSIVE
            WHILE the element at position index OF list_of_numbers NOT EQUALS the element at the position of the element at position index OF list_of_numbers MINUS one OF list_of_numbers HOLDS
                SET temporary_first_element TO the element at position index OF list_of_numbers
                SET temporary_second_element TO the element at the position of the element at position index OF list_of_numbers MINUS one OF list_of_numbers
                SET the element at the position of the element at position index OF list_of_numbers MINUS one OF list_of_numbers TO temporary_first_element
                SET the element at position index OF list_of_numbers TO temporary_second_element
            END WHILE
        END FOR
        SET result_list TO an empty list
        FOR each pair_of_index_and_value IN the enumeration of list_of_numbers
            IF the value NOT EQUALS the index PLUS one THEN
                APPEND the value TO result_list
            END IF
        END FOR
        RETURN result_list
    END FUNCTION
END CLASS
