CLASS Solution
    FUNCTION combinationSum2(candidates, target)
        FUNCTION backtrack(start, target_remaining, path)
            IF target_remaining EQUALS zero
                APPEND path TO result
                RETURN
            END IF
            IF target_remaining LESS THAN zero
                RETURN
            END IF
            FOR index FROM start TO the LENGTH OF candidates MINUS one INCLUSIVE
                IF index GREATER THAN start AND element at position index OF candidates EQUALS element at position index MINUS one OF candidates
                    CONTINUE
                END IF
                CALL backtrack WITH arguments index PLUS one, target_remaining MINUS element at position index OF candidates, path CONCATENATED WITH element at position index OF candidates
            END FOR
        END FUNCTION

        CALL sort ON candidates
        SET result TO an empty list
        CALL backtrack WITH arguments zero, target, an empty list
        RETURN result
    END FUNCTION
END CLASS
