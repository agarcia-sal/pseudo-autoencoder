CLASS Solution
    FUNCTION ipToCIDR(ip, n)
        
        FUNCTION ip_to_int(ip_address)
            SET parts TO the list obtained by mapping to integer each element obtained by splitting ip_address by the dot character
            SET result_value TO zero
            FOR each part_element IN parts
                SET result_value TO result_value MULTIPLIED BY two HUNDRED FIFTY-SIX PLUS part_element
            END FOR
            RETURN result_value
        END FUNCTION
        
        FUNCTION int_to_ip(number_value)
            RETURN the concatenation with the dot character as separator of the string representations of the element at position index OF the sequence for each index counting from three DOWNTO zero INCLUSIVE obtained by the modulo of number_value SHIFTED RIGHT BY the product of eight AND index WITH two HUNDRED FIFTY-SIX
        END FUNCTION
        
        FUNCTION largest_cidr_size(start_value, count_number)
            IF the bitwise AND of start_value AND one EQUALS one
                RETURN one
            END IF
            SET x_value TO one
            WHILE x_value LESS THAN OR EQUAL TO count_number AND the bitwise AND of start_value AND x_value MINUS one EQUALS zero
                SET x_value TO x_value MULTIPLIED BY two
            END WHILE
            RETURN x_value DIVIDED BY two
        END FUNCTION
        
        SET result TO empty list
        SET start_ip TO ip_to_int(ip)
        
        WHILE n GREATER THAN zero HOLDS
            SET size TO largest_cidr_size(start_ip, n)
            APPEND the concatenation of int_to_ip(start_ip) WITH the forward slash character AND the string representation of thirty-two MINUS the bit length of size MINUS one TO result
            SET start_ip TO start_ip PLUS size
            SET n TO n MINUS size
        END WHILE
        
        RETURN result
    END FUNCTION
END CLASS
