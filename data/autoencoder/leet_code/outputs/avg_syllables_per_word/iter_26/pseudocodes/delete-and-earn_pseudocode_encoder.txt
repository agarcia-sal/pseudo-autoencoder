CLASS Solution
    FUNCTION deleteAndEarn(list_of_numbers)
        SET count TO the counting of occurrences of each element IN list_of_numbers
        SET unique_nums TO the sorted list of distinct elements IN count

        SET dp TO a list of zeroes with the length of the LENGTH OF unique_nums PLUS one

        SET element_at position one OF dp TO element_at position zero OF unique_nums MULTIPLIED BY the value associated to element_at position zero OF unique_nums IN count

        FOR index FROM two TO the LENGTH OF unique_nums PLUS one INCLUSIVE
            SET num TO element_at position index MINUS one OF unique_nums
            IF num EQUALS the element_at position index MINUS two OF unique_nums PLUS one THEN
                SET element_at position index OF dp TO the GREATER BETWEEN the element_at position index MINUS one OF dp AND the sum OF element_at position index MINUS two OF dp PLUS num MULTIPLIED BY the value associated to num IN count
            ELSE
                SET element_at position index OF dp TO the sum OF element_at position index MINUS one OF dp PLUS num MULTIPLIED BY the value associated to num IN count
            END IF
        END FOR

        RETURN the last element OF dp
    END FUNCTION
END CLASS
