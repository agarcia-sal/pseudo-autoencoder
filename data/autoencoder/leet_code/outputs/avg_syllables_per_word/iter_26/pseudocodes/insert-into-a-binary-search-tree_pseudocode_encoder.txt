CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN none
    END IF
    SET root TO A NEW TreeNode WITH value EQUALS element at position zero OF values
    SET i TO one
    SET queue TO A NEW DEQUE COLLECTION
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO THE FIRST ELEMENT REMOVED FROM queue
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none
            SET node.left TO A NEW TreeNode WITH value EQUALS element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none
            SET node.right TO A NEW TreeNode WITH value EQUALS element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF p.val IS NOT EQUALS q.val
        RETURN false
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION insertIntoBST(root, val)
        FUNCTION dfs(root)
            IF root IS none
                RETURN A NEW TreeNode WITH value EQUALS val
            END IF
            IF root.val IS LESS THAN val
                SET root.right TO dfs(root.right)
            ELSE
                SET root.left TO dfs(root.left)
            END IF
            RETURN root
        END FUNCTION

        RETURN dfs(root)
    END FUNCTION
END CLASS
