CLASS Solution
    FUNCTION findStrobogrammatic(n)
        FUNCTION dfs(u)
            IF u EQUALS zero
                RETURN list containing the empty string
            END IF
            IF u EQUALS one
                RETURN list containing string zero, string one, string eight
            END IF
            SET answer_list TO empty list
            FOR each value IN the RESULT OF dfs with argument u MINUS two
                FOR each pair_of_characters IN the list consisting of the strings eleven, eighty eight, sixty nine, ninety six
                    LET left_character BE the element at position one OF pair_of_characters
                    LET right_character BE the element at position two OF pair_of_characters
                    SET new_string TO the concatenation of left_character PLUS value PLUS right_character
                    APPEND new_string TO answer_list
                END FOR
                IF u NOT EQUALS n
                    SET zero_string TO the concatenation of string zero PLUS value PLUS string zero
                    APPEND zero_string TO answer_list
                END IF
            END FOR
            RETURN answer_list
        END FUNCTION
        RETURN the RESULT OF dfs with argument n
    END FUNCTION
END CLASS
