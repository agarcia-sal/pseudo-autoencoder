CLASS Solution
    FUNCTION uniquePaths(parameter m, parameter n)
        SET dp TO a two dimensional list where each element is one and there are n elements in each inner list and there are m such inner lists
        FOR index i FROM one TO m GREATER THAN OR EQUAL TO one THEN NOT INCLUDING m
            FOR index j FROM one TO n GREATER THAN OR EQUAL TO one THEN NOT INCLUDING n
                SET the element at position i OF dp TO a list where the element at position j OF that list IS EQUAL TO the element at position i MINUS one OF dp at position j PLUS the element at position i OF dp at position j MINUS one
            END FOR
        END FOR
        RETURN the element at position m MINUS one OF dp at position n MINUS one
    END FUNCTION
END CLASS
