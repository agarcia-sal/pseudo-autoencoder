CLASS Solution
    FUNCTION minCost(grid)
        SET m TO the LENGTH OF grid
        SET n TO the LENGTH OF the element at position zero OF grid
        SET directions TO a LIST OF TUPLES representing the FOUR cardinal directions with zero AND one OR zero AND negative one OR one AND zero OR negative one AND zero
        SET costs TO a TWO DIMENSIONAL LIST where EACH element IS positive infinity WITH m ROWS AND n COLUMNS
        SET the element at position zero AT position zero OF costs TO zero
        
        SET pq TO a LIST CONTAINING a TUPLE OF zero AND zero AND zero
        SET visited TO an empty SET
        
        WHILE pq IS NOT empty
            EXTRACT the SMALLEST element FROM pq AND ASSIGN cost TO the FIRST element OF the TUPLE AND i TO the SECOND element AND j TO the THIRD element
            
            IF the TUPLE consisting OF i AND j IS IN visited
                CONTINUE
            END IF
            ADD the TUPLE consisting OF i AND j TO visited
            
            IF i EQUALS m MINUS one AND j EQUALS n MINUS one
                RETURN cost
            END IF
            
            FOR k AND the TUPLE OF di AND dj IN directions WITH enumeration starting AT zero
                SET ni TO i PLUS di
                SET nj TO j PLUS dj
                
                IF zero IS LESS THAN OR EQUAL TO ni AND ni IS LESS THAN m AND zero IS LESS THAN OR EQUAL TO nj AND nj IS LESS THAN n
                    IF k PLUS one DOES NOT EQUAL the element at position i AT position j OF grid
                        SET new_cost TO cost PLUS one
                    ELSE
                        SET new_cost TO cost
                    END IF
                    IF new_cost IS LESS THAN the element at position ni AT position nj OF costs
                        SET the element at position ni AT position nj OF costs TO new_cost
                        ADD the TUPLE OF new_cost AND ni AND nj INTO pq USING the HEAP PUSH operation
                    END IF
                END IF
            END FOR
        END WHILE
        
        RETURN the NEGATIVE of one
    END FUNCTION
END CLASS
