CLASS Solution
    FUNCTION maximizeSweetness(sweetness, k)
        FUNCTION canDivide(min_sweet)
            SET pieces TO zero
            SET current_sweet TO zero
            FOR each sweet IN sweetness
                INCREMENT current_sweet BY sweet
                IF current_sweet GREATER THAN OR EQUAL TO min_sweet
                    INCREMENT pieces BY one
                    SET current_sweet TO zero
                    IF pieces GREATER THAN k
                        RETURN true
                    END IF
                END IF
            END FOR
            RETURN pieces GREATER THAN k
        END FUNCTION

        SET left TO the MINIMUM ELEMENT OF sweetness
        SET right TO the SUM OF sweetness DIVIDED BY the SUM OF k PLUS one
        SET result TO zero
        WHILE left LESS THAN OR EQUAL TO right HOLDS
            SET mid TO left PLUS right DIVIDED BY two
            IF canDivide(mid) IS true
                SET result TO mid
                SET left TO mid PLUS one
            ELSE
                SET right TO mid MINUS one
            END IF
        END WHILE
        RETURN result
    END FUNCTION
END CLASS
