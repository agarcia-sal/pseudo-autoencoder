CLASS Solution
    FUNCTION maxProfit(prices, fee)
        SET cash TO zero
        SET hold TO negative infinity
        
        FOR each price IN prices
            SET potential_cash_after_selling TO hold PLUS price MINUS fee
            SET cash TO the GREATER OF cash AND potential_cash_after_selling
            SET potential_hold_after_buying TO cash MINUS price
            SET hold TO the GREATER OF hold AND potential_hold_after_buying
        END FOR
        
        RETURN cash
    END FUNCTION
END CLASS
