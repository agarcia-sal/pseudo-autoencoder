CLASS Solution
    FUNCTION constructArray(parameter n, parameter k)
        SET result TO the LIST OF all INTEGERS FROM the NUMBER one TO the NUMBER n MINUS the NUMBER k MINUS the NUMBER one INCLUSIVE
        SET left TO the NUMBER n MINUS the NUMBER k
        SET right TO the NUMBER n
        WHILE left IS LESS THAN OR EQUAL TO right HOLDS
            IF left EQUALS right THEN
                APPEND left TO result
            ELSE
                APPEND left TO result
                APPEND right TO result
            END IF
            INCREMENT left BY the NUMBER one
            DECREMENT right BY the NUMBER one
        END WHILE
        RETURN result
    END FUNCTION
END CLASS
