CLASS Solution
    FUNCTION minDistance(word1, word2)
        SET m TO the LENGTH OF word1
        SET n TO the LENGTH OF word2

        SET dp TO a two dimensional list of zero values with the first dimension of size m PLUS one and the second dimension of size n PLUS one

        FOR i FROM one TO m INCLUSIVE
            FOR j FROM one TO n INCLUSIVE
                IF the element at position i MINUS one OF word1 EQUALS the element at position j MINUS one OF word2
                    SET the element at position i and j OF dp TO the element at position i MINUS one and j MINUS one OF dp PLUS one
                ELSE
                    SET the element at position i and j OF dp TO the maximum value BETWEEN the element at position i MINUS one and j OF dp AND the element at position i and j MINUS one OF dp
                END IF
            END FOR
        END FOR

        SET lcs_length TO the element at position m and n OF dp

        RETURN m MINUS lcs_length PLUS n MINUS lcs_length
    END FUNCTION
END CLASS
