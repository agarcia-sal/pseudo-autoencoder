CLASS Solution
    FUNCTION intersectionSizeTwo(intervals AS list of list of integer) RETURNS integer
        CALL intervals SORTED BY the element at position two OF element IN intervals AND THEN BY the NEGATION OF the element at position one OF element DESCENDING
        SET first TO the element at position two OF the element at position one OF intervals MINUS one
        SET second TO the element at position two OF the element at position one OF intervals
        SET count TO two
        FOR each pair start and end IN the elements from position two TO the END OF intervals
            IF start GREATER THAN second
                SET first TO end MINUS one
                SET second TO end
                INCREMENT count BY two
            ELSE IF start GREATER THAN first
                SET first TO second
                SET second TO end
                INCREMENT count BY one
            END IF
        END FOR
        RETURN count
    END FUNCTION
END CLASS
