CLASS Solution
    FUNCTION generatePalindromes(string_input)
        FUNCTION dfs(current_string)
            IF the LENGTH OF current_string EQUALS the LENGTH OF string_input
                APPEND current_string TO ans
                RETURN
            END IF
            FOR each character_and_value IN the ITEMS OF cnt
                SET character TO the FIRST element OF character_and_value
                SET value TO the SECOND element OF character_and_value
                IF value GREATER THAN one
                    DECREMENT cnt AT character BY two
                    CALL dfs WITH the CONCATENATION OF character PLUS current_string PLUS character
                    INCREMENT cnt AT character BY two
                END IF
            END FOR
        END FUNCTION

        ASSIGN cnt TO the COUNTER OF string_input
        SET mid TO empty string
        FOR each character_and_value IN the ITEMS OF cnt
            SET character TO the FIRST element OF character_and_value
            SET value TO the SECOND element OF character_and_value
            IF the RESULT OF value AND one EQUALS one
                IF mid NOT EQUALS empty string
                    RETURN empty list
                END IF
                SET mid TO character
                DECREMENT cnt AT character BY one
            END IF
        END FOR
        SET ans TO empty list
        CALL dfs WITH mid
        RETURN ans
    END FUNCTION
END CLASS
