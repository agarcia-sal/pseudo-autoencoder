CLASS Solution
    FUNCTION parseTernary(expression)
        FUNCTION evaluate(current_position)
            IF current_position EQUALS the LENGTH OF expression MINUS one OR the CHARACTER at position current_position PLUS one OF expression EQUALS a colon
                RETURN the CHARACTER at position current_position OF expression AND current_position PLUS two
            END IF
            
            SET true_part AND updated_position TO the RESULT OF evaluate WITH current_position PLUS two
            SET false_part AND final_position TO the RESULT OF evaluate WITH updated_position
            
            IF the CHARACTER at position current_position OF expression EQUALS the uppercase letter T
                RETURN true_part AND final_position
            ELSE
                RETURN false_part AND final_position
            END IF
        END FUNCTION
        
        SET result AND unused_position TO the RESULT OF evaluate WITH zero
        RETURN result
    END FUNCTION
END CLASS
