CLASS Solution
    FUNCTION nextGreaterElement(n)
        SET digits TO the LIST of the STRING form of n
        SET length TO the LENGTH OF digits
        
        SET i TO length MINUS two
        WHILE i GREATER THAN OR EQUAL TO zero AND the element at position i OF digits GREATER THAN OR EQUAL TO the element at position i PLUS one OF digits HOLDS
            DECREMENT i BY one
        END WHILE
        
        IF i EQUALS negative one THEN
            RETURN negative one
        END IF
        
        SET j TO length MINUS one
        WHILE the element at position j OF digits LESS THAN OR EQUAL TO the element at position i OF digits HOLDS
            DECREMENT j BY one
        END WHILE
        
        SET temporary TO the element at position i OF digits
        SET the element at position i OF digits TO the element at position j OF digits
        SET the element at position j OF digits TO temporary
        
        SET digits TO the concatenation of the SUBLIST OF digits FROM position zero TO position i INCLUSIVE AND the REVERSED SUBLIST OF digits FROM position i PLUS one TO the LAST ELEMENT INCLUSIVE
        
        SET result_string TO the STRING formed by JOINING all characters of digits
        SET result TO the INTEGER form of result_string
        
        IF result GREATER THAN the INTEGER form of two RAISED TO THE power of thirty ONE MINUS one THEN
            RETURN negative one
        END IF
        
        RETURN result
    END FUNCTION
END CLASS
