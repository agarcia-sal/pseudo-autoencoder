CLASS Solution
    FUNCTION findUnsortedSubarray(list_of_numbers)
        SET sorted_list TO the list_of_numbers ARRANGED IN ASCENDING ORDER
        SET start_pointer TO negative one
        SET end_pointer TO negative two
        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one INCLUSIVE
            IF the element at position index OF list_of_numbers NOT EQUALS the element at position index OF sorted_list
                IF start_pointer EQUALS negative one
                    SET start_pointer TO index
                END IF
                SET end_pointer TO index
            END IF
        END FOR
        SET subarray_length TO end_pointer MINUS start_pointer PLUS one
        RETURN subarray_length
    END FUNCTION
END CLASS
