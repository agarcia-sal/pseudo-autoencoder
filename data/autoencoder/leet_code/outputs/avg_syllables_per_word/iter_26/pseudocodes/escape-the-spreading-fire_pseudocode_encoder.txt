CLASS Solution
    FUNCTION maximumMinutes(grid)
        SET m TO the LENGTH OF grid
        SET n TO the LENGTH OF element at position zero OF grid
        SET directions TO the LIST OF PAIRS consisting of ONE AND ZERO, MINUS ONE AND ZERO, ZERO AND ONE, ZERO AND MINUS ONE

        FUNCTION can_escape(wait)
            SET fire_times TO a TWO DIMENSIONAL LIST of dimension m BY n WHERE EACH element IS positive infinity
            ASSIGN reference TO a new empty double ended queue NAMED queue

            FOR i FROM zero TO m MINUS one INCLUSIVE
                FOR j FROM zero TO n MINUS one INCLUSIVE
                    IF element at position i OF grid EQUALS one THEN
                        APPEND the TRIPLE consisting of i, j, and zero TO queue
                        SET element at position i OF element at position j OF fire_times TO zero
                    END IF
                END FOR
            END FOR

            WHILE queue HOLDS
                REMOVE the element at the FRONT OF queue AND ASSIGN IT TO x, y, t
                FOR each pair dx, dy IN directions
                    SET nx TO x PLUS dx
                    SET ny TO y PLUS dy
                    IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND element at position nx OF element at position ny OF grid EQUALS zero AND element at position nx OF element at position ny OF fire_times IS positive infinity THEN
                        SET element at position nx OF element at position ny OF fire_times TO t PLUS one
                        APPEND the TRIPLE consisting of nx, ny, and t PLUS one TO queue
                    END IF
                END FOR
            END WHILE

            ASSIGN reference TO a new double ended queue containing the TRIPLE consisting of zero, zero, and wait NAMED queue
            ASSIGN reference TO a new set containing the PAIR consisting of zero and zero NAMED visited

            WHILE queue HOLDS
                REMOVE the element at the FRONT OF queue AND ASSIGN IT TO x, y, t
                FOR each pair dx, dy IN directions
                    SET nx TO x PLUS dx
                    SET ny TO y PLUS dy
                    IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND the PAIR consisting of nx and ny IS NOT IN visited AND element at position nx OF element at position ny OF grid EQUALS zero THEN
                        IF nx EQUALS m MINUS one AND ny EQUALS n MINUS one THEN
                            IF element at position nx OF element at position ny OF fire_times IS GREATER THAN OR EQUAL TO t PLUS one THEN
                                RETURN True
                            ELSE
                                RETURN False
                            END IF
                        END IF
                        IF element at position nx OF element at position ny OF fire_times IS GREATER THAN t PLUS one THEN
                            ADD the PAIR consisting of nx and ny TO visited
                            APPEND the TRIPLE consisting of nx, ny, and t PLUS one TO queue
                        END IF
                    END IF
                END FOR
            END WHILE

            RETURN False
        END FUNCTION

        SET left TO zero
        SET right TO m MULTIPLIED BY n
        SET result TO negative one

        WHILE left IS LESS THAN OR EQUAL TO right
            SET mid TO the INTEGER division result of left PLUS right DIVIDED BY two
            IF can_escape(mid) HOLDS THEN
                SET result TO mid
                SET left TO mid PLUS one
            ELSE
                SET right TO mid MINUS one
            END IF
        END WHILE

        IF result NOT EQUALS m MULTIPLIED BY n THEN
            RETURN result
        ELSE
            RETURN ten RAISED TO THE POWER OF nine
        END IF
    END FUNCTION
END CLASS
