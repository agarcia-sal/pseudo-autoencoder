CLASS Solution
    FUNCTION complexNumberMultiply(num1, num2)
        FUNCTION parse_complex(num)
            SET parts TO the result of splitting num by the plus sign
            SET real TO the integer converted FROM the element at position zero OF parts
            SET imaginary_string TO the element at position one OF parts
            SET imaginary TO the integer converted FROM the substring FROM position zero TO the length of imaginary_string MINUS one OF imaginary_string
            RETURN real, imaginary
        END FUNCTION

        SET real1, imag1 TO the result of calling parse_complex WITH num1
        SET real2, imag2 TO the result of calling parse_complex WITH num2

        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2

        RETURN the concatenation of the string form of real_part PLUS the plus sign PLUS the string form of imaginary_part PLUS the character i
    END FUNCTION
END CLASS
