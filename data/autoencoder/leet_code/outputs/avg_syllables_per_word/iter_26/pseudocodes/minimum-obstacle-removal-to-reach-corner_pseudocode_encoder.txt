CLASS Solution
    FUNCTION minimumObstacles(grid)
        SET number_of_rows TO the LENGTH OF grid
        SET number_of_columns TO the LENGTH OF element at position zero OF grid
        SET directions TO the LIST OF the TUPLES of zero and one, one and zero, zero and MINUS one, MINUS one and zero
        SET minimum_heap TO the LIST containing the TUPLE of zero, zero, zero representing cost and coordinates
        SET visited_positions TO the SET containing the TUPLE of zero and zero
        
        WHILE minimum_heap HOLDS elements
            REMOVE the element WITH the SMALLEST cost FROM minimum_heap AND ASSIGN TO cost, x_coordinate, y_coordinate
            
            IF the TUPLE of x_coordinate and y_coordinate EQUALS the TUPLE of number_of_rows MINUS one and number_of_columns MINUS one THEN
                RETURN cost
            END IF
            
            FOR each direction_change_x, direction_change_y IN directions
                SET new_x_coordinate TO x_coordinate PLUS direction_change_x
                SET new_y_coordinate TO y_coordinate PLUS direction_change_y
                
                IF new_x_coordinate IS GREATER THAN OR EQUAL TO zero AND new_x_coordinate IS LESS THAN number_of_rows AND new_y_coordinate IS GREATER THAN OR EQUAL TO zero AND new_y_coordinate IS LESS THAN number_of_columns AND the TUPLE of new_x_coordinate and new_y_coordinate IS NOT IN visited_positions THEN
                    ADD the TUPLE of new_x_coordinate and new_y_coordinate TO visited_positions
                    SET new_cost TO cost PLUS the element at position new_x_coordinate OF grid AT position new_y_coordinate
                    ADD the TUPLE of new_cost, new_x_coordinate, new_y_coordinate TO minimum_heap MAINTAINING HEAP PROPERTY
                END IF
            END FOR
        END WHILE
        
        RETURN the NEGATIVE of one
    END FUNCTION
END CLASS
