CLASS Solution
    FUNCTION minTrioDegree(n, edges)
        SET graph TO a mapping FROM each number FROM one TO n INCLUSIVE TO an empty set
        
        FOR each pair u and v IN edges
            ADD v TO the set associated WITH u IN graph
            ADD u TO the set associated WITH v IN graph
        END FOR
        
        SET min_degree TO positive infinity
        SET trio_found TO false
        
        FOR i FROM one TO n INCLUSIVE
            FOR j FROM the number immediately after i TO n INCLUSIVE
                FOR k FROM the number immediately after j TO n INCLUSIVE
                    IF the element j IS IN the set associated WITH i IN graph AND the element k IS IN the set associated WITH i IN graph AND the element k IS IN the set associated WITH j IN graph
                        SET trio_found TO true
                        SET trio_degree TO the LENGTH OF the set associated WITH i IN graph PLUS the LENGTH OF the set associated WITH j IN graph PLUS the LENGTH OF the set associated WITH k IN graph MINUS six
                        IF trio_degree IS LESS THAN min_degree
                            SET min_degree TO trio_degree
                        END IF
                    END IF
                END FOR
            END FOR
        END FOR
        
        IF trio_found IS true
            RETURN min_degree
        ELSE
            RETURN negative one
        END IF
    END FUNCTION
END CLASS
