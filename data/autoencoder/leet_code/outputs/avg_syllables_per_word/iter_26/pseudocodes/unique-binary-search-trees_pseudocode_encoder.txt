CLASS Solution
    FUNCTION numTrees(parameter_n)
        SET list_dp TO a list of zeros of the LENGTH OF parameter_n PLUS one
        SET element at position zero OF list_dp TO one
        SET element at position one OF list_dp TO one
        FOR index_i FROM two TO parameter_n INCLUSIVE
            FOR index_j FROM one TO index_i INCLUSIVE
                SET element at position index_i OF list_dp TO element at position index_i OF list_dp PLUS element at position index_j MINUS one OF list_dp MULTIPLIED BY element at position index_i MINUS index_j OF list_dp
            END FOR
        END FOR
        RETURN element at position parameter_n OF list_dp
    END FUNCTION
END CLASS
