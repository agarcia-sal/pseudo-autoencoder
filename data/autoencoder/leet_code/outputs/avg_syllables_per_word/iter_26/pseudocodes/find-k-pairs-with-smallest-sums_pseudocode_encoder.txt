CLASS Solution
    FUNCTION kSmallestPairs(nums1, nums2, k)
        IF nums1 IS empty OR nums2 IS empty OR k IS LESS THAN OR EQUAL TO zero
            RETURN empty list
        END IF
        
        SET min_heap TO empty list
        
        FUNCTION minimum_of_two_values(value_one, value_two)
            IF value_one IS LESS THAN value_two
                RETURN value_one
            ELSE
                RETURN value_two
            END IF
        END FUNCTION
        
        SET limit TO minimum_of_two_values(k, the LENGTH OF nums1)
        
        FOR index i FROM zero TO limit MINUS one INCLUSIVE
            SET first_sum TO element at position i OF nums1 PLUS element at position zero OF nums2
            CALL pushing of tuple containing first_sum, i, zero INTO min_heap USING heap push procedure
        END FOR
        
        SET result TO empty list
        
        WHILE min_heap IS NOT empty AND the LENGTH OF result IS LESS THAN k
            CALL popping FROM min_heap USING heap pop procedure AND DESTRUCTURE TO discard_sum, i, j
            APPEND list containing element at position i OF nums1 AND element at position j OF nums2 TO result
            IF j PLUS one IS LESS THAN the LENGTH OF nums2
                SET next_sum TO element at position i OF nums1 PLUS element at position j PLUS one OF nums2
                CALL pushing of tuple containing next_sum, i, j PLUS one INTO min_heap USING heap push procedure
            END IF
        END WHILE
        
        RETURN result
    END FUNCTION
END CLASS
