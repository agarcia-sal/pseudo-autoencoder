CLASS Solution
    FUNCTION letterCasePermutation(s)
        FUNCTION dfs(i)
            IF i GREATER THAN OR EQUAL TO the LENGTH OF t HOLDS
                APPEND the concatenation of all elements OF t TO ans
                RETURN
            END IF
            CALL dfs WITH argument i PLUS one
            IF the element at position i OF t IS an alphabetical character HOLDS
                SET the element at position i OF t TO the character FOUND BY TOGGLING the case OF the element at position i OF t
                CALL dfs WITH argument i PLUS one
            END IF
        END FUNCTION

        SET t TO the list of elements OF s
        SET ans TO an empty list
        CALL dfs WITH argument zero
        RETURN ans
    END FUNCTION
END CLASS
