CLASS Solution
    FUNCTION maxIncreaseKeepingSkyline(grid AS list of list of integers) RETURNS integer
        SET n TO the LENGTH OF grid
        SET row_maxes TO an empty list
        FOR each row IN grid
            APPEND the MAXIMUM VALUE IN row TO row_maxes
        END FOR
        SET col_maxes TO an empty list
        SET transposed_grid TO the collection of elements taken by pairing each element at the same position FROM each row IN grid
        FOR each column IN transposed_grid
            APPEND the MAXIMUM VALUE IN column TO col_maxes
        END FOR

        SET total_increase TO zero
        FOR i FROM zero TO n MINUS one INCLUSIVE
            FOR j FROM zero TO n MINUS one INCLUSIVE
                SET max_height TO the MINIMUM OF the element at position i OF row_maxes AND the element at position j OF col_maxes
                SET current_height TO the element at position i OF the element at position j OF grid
                SET height_difference TO max_height MINUS current_height
                IF height_difference GREATER THAN zero
                    INCREMENT total_increase BY height_difference
                END IF
            END FOR
        END FOR

        RETURN total_increase
    END FUNCTION
END CLASS
