CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values EQUALS empty
        RETURN none
    END IF
    SET root TO A NEW TreeNode WITH the element at position zero OF values
    SET i TO one
    SET queue TO AN EMPTY double ended queue
    APPEND root TO queue
    WHILE queue HOLDS
        SET node TO THE FIRST ELEMENT REMOVED FROM queue
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.left TO A NEW TreeNode WITH the element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.right TO A NEW TreeNode WITH the element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF p.val NOT EQUALS q.val
        RETURN false
    ELSE
        RETURN the RESULT OF is_same_tree WITH p.left AND q.left AND the RESULT OF is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION tree2str(root)
        FUNCTION helper(node)
            IF node IS none
                RETURN empty string
            END IF
            SET result TO the STRING REPRESENTATION OF node.val
            IF node.left IS NOT none OR node.right IS NOT none
                SET result TO result CONCATENATED WITH an OPENING PARENTHESIS CONCATENATED WITH helper WITH node.left CONCATENATED WITH a CLOSING PARENTHESIS
            END IF
            IF node.right IS NOT none
                SET result TO result CONCATENATED WITH an OPENING PARENTHESIS CONCATENATED WITH helper WITH node.right CONCATENATED WITH a CLOSING PARENTHESIS
            END IF
            RETURN result
        END FUNCTION
        RETURN helper WITH root
    END FUNCTION
END CLASS
