CLASS Solution
    FUNCTION maxDistance(arrays)
        SET minimum_value TO element at position zero OF element at position zero OF arrays
        SET maximum_value TO element at position zero OF element at position at last position OF arrays
        SET maximum_distance TO zero

        FOR index FROM one TO the LENGTH OF arrays MINUS one INCLUSIVE
            SET current_array TO element at position index OF arrays
            SET current_minimum TO element at position zero OF current_array
            SET current_maximum TO element at position at last position OF current_array

            SET maximum_distance TO the GREATER OF maximum_distance AND the GREATER OF the ABSOLUTE VALUE OF current_maximum MINUS minimum_value AND the ABSOLUTE VALUE OF maximum_value MINUS current_minimum

            SET minimum_value TO the SMALLER OF minimum_value AND current_minimum
            SET maximum_value TO the GREATER OF maximum_value AND current_maximum
        END FOR

        RETURN maximum_distance
    END FUNCTION
END CLASS
