CLASS Solution
    FUNCTION removeKdigits(num, k)
        SET stack TO empty list
        FOR each digit IN num
            WHILE k GREATER THAN zero AND stack IS NOT empty AND the element at the last position OF stack GREATER THAN digit
                REMOVE the element at the last position FROM stack
                DECREMENT k BY one
            END WHILE
            APPEND digit TO stack
        END FOR

        IF k EQUALS zero
            SET final_stack TO stack
        ELSE
            SET final_stack TO all elements FROM position one TO the length of stack MINUS k positions OF stack
        END IF

        SET result TO the concatenation of all elements in final_stack with leading zero characters REMOVED

        IF result IS NOT empty
            RETURN result
        ELSE
            RETURN a single character digit zero expressed as a string
        END IF
    END FUNCTION
END CLASS
