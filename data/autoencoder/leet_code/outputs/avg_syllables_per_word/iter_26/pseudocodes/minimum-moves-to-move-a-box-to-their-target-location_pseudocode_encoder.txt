CLASS Solution
    FUNCTION minPushBox(grid)
        SET directions TO the list of four pairs consisting of the pair of negative one and zero, the pair of one and zero, the pair of zero and negative one, and the pair of zero and one
        
        FOR each index i FROM zero TO the LENGTH OF grid MINUS one INCLUSIVE
            FOR each index j FROM zero TO the LENGTH OF element at position i OF grid MINUS one INCLUSIVE
                IF element at position i OF grid AT element at position j EQUALS the character labeled S
                    SET player_start TO the pair consisting of i and j
                ELSE IF element at position i OF grid AT element at position j EQUALS the character labeled B
                    SET box_start TO the pair consisting of i and j
                ELSE IF element at position i OF grid AT element at position j EQUALS the character labeled T
                    SET target TO the pair consisting of i and j
                END IF
            END FOR
        END FOR
        
        INITIALIZE queue TO a double ended queue containing the tuple of player_start, box_start, and zero
        INITIALIZE visited TO an empty set
        ADD the tuple of player_start and box_start TO visited
        
        WHILE queue HOLDS
            REMOVE and ASSIGN the element from the front of queue TO player_pos, box_pos, push_count
            
            IF player_pos EQUALS target
                RETURN push_count
            END IF
            
            FOR each pair dx, dy IN directions
                SET new_player_pos TO the pair consisting of the element at position zero OF player_pos PLUS dx and the element at position one OF player_pos PLUS dy
                
                IF the call to method is_valid_position with arguments new_player_pos and grid HOLDS TRUE
                    IF new_player_pos EQUALS box_pos
                        SET new_box_pos TO the pair consisting of the element at position zero OF box_pos PLUS dx and the element at position one OF box_pos PLUS dy
                        
                        IF the call to method is_valid_position with arguments new_box_pos and grid HOLDS TRUE AND the tuple of new_player_pos and new_box_pos IS NOT IN visited
                            ADD the tuple of new_player_pos and new_box_pos TO visited
                            APPEND the tuple of new_player_pos, new_box_pos, and push_count PLUS one TO queue
                        END IF
                    ELSE
                        IF the tuple of new_player_pos and box_pos IS NOT IN visited
                            ADD the tuple of new_player_pos and box_pos TO visited
                            APPEND the tuple of new_player_pos, box_pos, and push_count TO queue
                        END IF
                    END IF
                END IF
            END FOR
        END WHILE
        
        RETURN negative one
    END FUNCTION
    
    FUNCTION is_valid_position(position, grid)
        SET x TO the element at position zero OF position
        SET y TO the element at position one OF position
        
        RETURN the logical expression that x GREATER THAN OR EQUAL TO zero AND x LESS THAN the LENGTH OF grid AND y GREATER THAN OR EQUAL TO zero AND y LESS THAN the LENGTH OF element at position zero OF grid AND element at position x OF grid AT element at position y NOT EQUALS the character labeled with the hash symbol
    END FUNCTION
END CLASS
