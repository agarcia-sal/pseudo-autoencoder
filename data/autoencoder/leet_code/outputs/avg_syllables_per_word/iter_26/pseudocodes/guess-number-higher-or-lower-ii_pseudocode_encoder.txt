CLASS Solution
    FUNCTION getMoneyAmount(n)
        IMPORT the LEAST RECENTLY USED CACHE decorator

        FUNCTION dp(left, right)
            IF left GREATER THAN OR EQUAL TO right HOLDS
                RETURN zero
            END IF

            SET minimal_cost TO positive infinity

            FOR pivot FROM left TO right INCLUSIVE
                SET cost_value TO pivot PLUS the GREATER OF dp(left, pivot MINUS one) AND dp(pivot PLUS one, right)
                SET minimal_cost TO the SMALLER OF minimal_cost AND cost_value
            END FOR

            RETURN minimal_cost
        END FUNCTION

        RETURN dp(one, n)
    END FUNCTION
END CLASS
