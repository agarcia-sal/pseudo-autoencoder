CLASS Solution
    FUNCTION kMirror(k, n)
        FUNCTION is_palindrome(s)
            RETURN s EQUALS the REVERSE OF s
        END FUNCTION

        FUNCTION to_base_k(num, k)
            IF num EQUALS zero THEN
                RETURN the string REPRESENTING zero
            END IF
            SET digits TO an empty list
            WHILE num HOLDS
                APPEND the integer value OF num MODULO k TO digits
                SET num TO num DIVIDED BY k USING INTEGER DIVISION
            END WHILE
            RETURN the concatenation OF the string representations OF the elements IN digits IN THE REVERSE ORDER
        END FUNCTION

        FUNCTION generate_palindromes(length)
            IF length EQUALS one THEN
                FOR i FROM one TO nine INCLUSIVE
                    YIELD the string REPRESENTATION OF i
                END FOR
            ELSE
                SET start TO ten RAISED TO THE POWER OF the INTEGER DIVISION OF (length MINUS one) BY two
                SET end TO ten RAISED TO THE POWER OF the INTEGER DIVISION OF (length PLUS one) BY two
                FOR half FROM start TO end MINUS one INCLUSIVE
                    SET half_str TO the string REPRESENTATION OF half
                    IF length MODULO two EQUALS zero THEN
                        SET full TO the concatenation OF half_str AND the REVERSE OF half_str
                    ELSE
                        SET full TO the concatenation OF half_str AND the substring OF half_str FROM the SECOND TO LAST POSITION TO the FIRST POSITION IN REVERSE ORDER
                    END IF
                    YIELD full
                END FOR
            END IF
        END FUNCTION

        SET k_mirror_numbers TO an empty list
        SET length TO one
        WHILE the LENGTH OF k_mirror_numbers IS LESS THAN n
            FOR each pal IN generate_palindromes(length)
                SET num TO the integer REPRESENTATION OF pal
                IF is_palindrome(to_base_k(num, k)) THEN
                    APPEND num TO k_mirror_numbers
                    IF the LENGTH OF k_mirror_numbers EQUALS n THEN
                        BREAK the inner loop
                    END IF
                END IF
            END FOR
            INCREMENT length BY one
        END WHILE

        RETURN the SUM OF the elements IN k_mirror_numbers
    END FUNCTION
END CLASS
