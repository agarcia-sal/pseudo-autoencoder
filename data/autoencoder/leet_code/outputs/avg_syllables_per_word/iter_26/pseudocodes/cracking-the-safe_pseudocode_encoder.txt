CLASS Solution
    FUNCTION crackSafe(n_parameter, k_parameter)
        SET visited_set TO empty set
        SET sequence_list TO empty list

        FUNCTION dfs(node_parameter)
            FOR index_i FROM zero TO k_parameter MINUS one INCLUSIVE
                SET edge_string TO CONCATENATION OF node_parameter AND the string representation of index_i
                IF edge_string NOT IN visited_set
                    ADD edge_string TO visited_set
                    CALL dfs WITH the substring OF edge_string FROM position one TO the END
                    APPEND the string representation of index_i TO sequence_list
                END IF
            END FOR
        END FUNCTION

        SET start_node TO the string consisting of n_parameter MINUS one repetitions OF character zero
        CALL dfs WITH start_node

        RETURN the concatenation OF the elements IN sequence_list IN ORDER FOLLOWED BY start_node
    END FUNCTION
END CLASS
