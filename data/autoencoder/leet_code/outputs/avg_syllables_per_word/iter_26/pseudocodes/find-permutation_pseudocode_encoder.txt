CLASS Solution
    FUNCTION findPermutation(s)
        SET result TO list of numbers from one TO the LENGTH OF s PLUS one INCLUSIVE
        SET start TO zero
        FOR end FROM zero TO the LENGTH OF s INCLUSIVE
            IF end EQUALS the LENGTH OF s OR the element at position end OF s EQUALS the character I
                SET the segment of result FROM position start TO position end PLUS one INCLUSIVE TO the reverse of the segment of result FROM position start TO position end PLUS one INCLUSIVE
                SET start TO end PLUS one
            END IF
        END FOR
        RETURN result
    END FUNCTION
END CLASS
