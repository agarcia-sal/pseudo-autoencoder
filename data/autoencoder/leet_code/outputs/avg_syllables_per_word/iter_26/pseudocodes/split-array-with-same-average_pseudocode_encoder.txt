CLASS Solution
    FUNCTION splitArraySameAverage(nums)
        SET n TO the LENGTH OF nums
        SET total_sum TO the SUM OF nums

        IF n EQUALS one
            RETURN False
        END IF

        SET dp TO a LIST OF SETS WHERE each SET CORRESPONDS TO a subset size FROM zero TO n DIVIDED BY two INCLUSIVE
        ADD zero TO the SET at position zero OF dp

        FOR each number IN nums
            FOR size FROM n DIVIDED BY two DOWN TO one INCLUSIVE
                FOR each sum_value IN the SET at position size MINUS one OF dp
                    SET new_sum TO sum_value PLUS number
                    IF new_sum MULTIPLIED BY n EQUALS size MULTIPLIED BY total_sum
                        RETURN True
                    END IF
                    ADD new_sum TO the SET at position size OF dp
                END FOR
            END FOR
        END FOR

        RETURN False
    END FUNCTION
END CLASS
