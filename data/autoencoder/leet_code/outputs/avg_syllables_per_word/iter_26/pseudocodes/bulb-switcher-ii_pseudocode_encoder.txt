CLASS Solution
    FUNCTION flipLights(n, presses)
        SET n TO the MINIMUM OF n AND six
        
        IF presses EQUALS zero THEN
            RETURN one
        END IF
        
        IF n EQUALS one THEN
            RETURN two
        END IF
        
        IF n EQUALS two THEN
            IF presses EQUALS one THEN
                RETURN three
            ELSE
                RETURN four
            END IF
        END IF
        
        IF presses EQUALS one THEN
            RETURN four
        END IF
        
        IF presses EQUALS two THEN
            RETURN seven
        END IF
        
        RETURN eight
    END FUNCTION
END CLASS
