CLASS Solution
    FUNCTION judgePoint24(cards)
        FUNCTION solve(nums)
            IF the LENGTH OF nums EQUALS one
                RETURN the ABSOLUTE VALUE OF element at position zero OF nums MINUS twenty four LESS THAN one multiplied BY ten TO THE POWER OF negative six
            END IF
            
            FOR index i FROM zero TO the LENGTH OF nums MINUS one INCLUSIVE
                FOR index j FROM zero TO the LENGTH OF nums MINUS one INCLUSIVE
                    IF i NOT EQUALS j
                        SET new_nums TO an EMPTY list
                        FOR index k FROM zero TO the LENGTH OF nums MINUS one INCLUSIVE
                            IF k NOT EQUALS i AND k NOT EQUALS j
                                APPEND element at position k OF nums TO new_nums
                            END IF
                        END FOR
                        
                        FOR each op IN the collection consisting of true division, multiplication, addition, subtraction
                            IF (op IS addition OR op IS multiplication) AND j GREATER THAN i
                                CONTINUE to next iteration of operation loop
                            END IF
                            
                            IF op IS NOT true division OR element at position j OF nums NOT EQUALS zero
                                APPEND the RESULT OF applying op TO element at position i OF nums AND element at position j OF nums TO new_nums
                                IF solve(new_nums) HOLDS
                                    RETURN true
                                END IF
                                REMOVE the last element FROM new_nums
                            END IF
                        END FOR
                    END IF
                END FOR
            END FOR
            
            RETURN false
        END FUNCTION
        
        IMPORT true division, multiplication, addition, subtraction FROM the operator collection
        RETURN solve(cards)
    END FUNCTION
END CLASS
