CLASS Solution
    FUNCTION findSubsequences(list_of_numbers)
        FUNCTION dfs(current_index, previous_number, temporary_list)
            IF current_index EQUALS the LENGTH OF list_of_numbers
                IF the LENGTH OF temporary_list GREATER THAN one
                    APPEND a COPY OF temporary_list TO answer_list
                END IF
                RETURN
            END IF
            IF the element at position current_index OF list_of_numbers GREATER THAN OR EQUAL TO previous_number
                APPEND the element at position current_index OF list_of_numbers TO temporary_list
                CALL dfs WITH current_index INCREMENTED BY one AND the element at position current_index OF list_of_numbers AS previous_number AND temporary_list
                REMOVE the last element FROM temporary_list
            END IF
            IF the element at position current_index OF list_of_numbers NOT EQUALS previous_number
                CALL dfs WITH current_index INCREMENTED BY one AND previous_number AND temporary_list
            END IF
        END FUNCTION

        SET answer_list TO an empty list
        CALL dfs WITH zero AS current_index AND minus one thousand AS previous_number AND an empty list AS temporary_list
        RETURN answer_list
    END FUNCTION
END CLASS
