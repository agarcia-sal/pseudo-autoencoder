CLASS Solution
    FUNCTION fullBloomFlowers(flowers AS List OF List, people AS List) AS List
        SET start_times TO the LIST OF start times taken from each pair in flowers SORTED in ascending order
        SET end_times TO the LIST OF end times taken from each pair in flowers SORTED in ascending order
        
        SET result TO an empty list
        
        FOR each person IN people
            SET started TO the NUMBER OF elements in start_times that are LESS THAN OR EQUAL TO person USING binary search for right insertion position
            SET ended TO the NUMBER OF elements in end_times that are LESS THAN person USING binary search for left insertion position
            APPEND the difference between started and ended TO result
        END FOR
        
        RETURN result
    END FUNCTION
END CLASS
