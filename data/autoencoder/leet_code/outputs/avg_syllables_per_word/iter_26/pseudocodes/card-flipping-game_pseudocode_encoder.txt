CLASS Solution
    FUNCTION flipgame(fronts, backs)
        SET same TO the SET OF each number x FOR each pair x AND y IN the ZIP OF fronts AND backs WHERE x EQUALS y
        SET all_numbers TO the SET OF the CONCATENATION OF fronts AND backs
        SET good_numbers TO the DIFFERENCE BETWEEN all_numbers AND same
        IF the LENGTH OF good_numbers GREATER THAN zero
            SET minimum_good_number TO the MINIMUM VALUE IN good_numbers
            RETURN minimum_good_number
        ELSE
            RETURN zero
        END IF
    END FUNCTION
END CLASS
