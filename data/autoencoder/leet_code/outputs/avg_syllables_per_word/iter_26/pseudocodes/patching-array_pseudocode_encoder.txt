CLASS Solution
    FUNCTION minPatches(nums, n)
        SET missing TO the smallest positive integer one
        SET patches TO zero representing the count of added patches
        SET i TO zero representing the current index in the list nums

        WHILE missing IS LESS THAN OR EQUAL TO n HOLDS
            IF i IS LESS THAN the LENGTH OF nums AND the element at position i OF nums IS LESS THAN OR EQUAL TO missing
                INCREMENT missing BY the element at position i OF nums
                INCREMENT i BY one
            ELSE
                INCREMENT missing BY missing itself
                INCREMENT patches BY one
            END IF
        END WHILE

        RETURN patches
    END FUNCTION
END CLASS
