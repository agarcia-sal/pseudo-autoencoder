CLASS Solution
    FUNCTION search(list_of_numbers, target_value)
        SET left TO zero
        SET right TO the LENGTH OF list_of_numbers MINUS one
        
        WHILE left LESS THAN OR EQUAL TO right HOLDS
            SET middle TO left PLUS the DIFFERENCE BETWEEN right AND left DIVIDED BY two
            
            IF the ELEMENT at position middle OF list_of_numbers EQUALS target_value THEN
                RETURN middle
            END IF
            
            IF the ELEMENT at position left OF list_of_numbers LESS THAN OR EQUAL TO the ELEMENT at position middle OF list_of_numbers THEN
                IF the ELEMENT at position left OF list_of_numbers LESS THAN OR EQUAL TO target_value AND target_value LESS THAN the ELEMENT at position middle OF list_of_numbers THEN
                    SET right TO middle MINUS one
                ELSE
                    SET left TO middle PLUS one
                END IF
            ELSE
                IF the ELEMENT at position middle OF list_of_numbers LESS THAN target_value AND target_value LESS THAN OR EQUAL TO the ELEMENT at position right OF list_of_numbers THEN
                    SET left TO middle PLUS one
                ELSE
                    SET right TO middle MINUS one
                END IF
            END IF
        END WHILE
        
        RETURN negative one
    END FUNCTION
END CLASS
