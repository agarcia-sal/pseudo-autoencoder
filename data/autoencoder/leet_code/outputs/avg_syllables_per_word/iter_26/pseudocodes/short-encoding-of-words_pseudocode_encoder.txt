CLASS Solution
    FUNCTION minimumLengthEncoding(words)
        SET reversed_words TO the LIST OF each word IN words TRANSFORMED BY REVERSING the CHARACTERS OF word AND THEN SORTED IN ASCENDING ORDER
        SET total_length TO zero
        FOR index FROM zero TO the LENGTH OF reversed_words MINUS one INCLUSIVE
            IF index PLUS one EQUALS the LENGTH OF reversed_words OR NOT the STRING AT position index PLUS one OF reversed_words STARTS WITH the STRING AT position index OF reversed_words HOLDS
                INCREMENT total_length BY the LENGTH OF the STRING AT position index OF reversed_words PLUS one
            END IF
        END FOR
        RETURN total_length
    END FUNCTION
END CLASS
