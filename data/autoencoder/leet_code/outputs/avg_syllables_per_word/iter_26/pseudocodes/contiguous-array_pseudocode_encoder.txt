CLASS Solution
    FUNCTION findMaxLength(list_of_numbers)
        SET dictionary_of_prefix_sum_positions TO a dictionary with zero mapped to negative one
        SET maximum_length TO zero
        SET count_tracker TO zero

        FOR each index_number, element_value IN enumeration of list_of_numbers
            IF element_value EQUALS one
                INCREMENT count_tracker BY one
            ELSE
                DECREMENT count_tracker BY one
            END IF

            IF count_tracker EXISTS AS A KEY IN dictionary_of_prefix_sum_positions
                SET maximum_length TO THE GREATER OF maximum_length AND index_number MINUS the element at count_tracker key OF dictionary_of_prefix_sum_positions
            ELSE
                SET the element at count_tracker key OF dictionary_of_prefix_sum_positions TO index_number
            END IF
        END FOR

        RETURN maximum_length
    END FUNCTION
END CLASS
