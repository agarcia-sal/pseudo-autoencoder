CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN none
    END IF
    SET root TO a new instance of TreeNode with value as element at position zero OF values
    SET i TO one
    SET queue TO a new empty double ended queue
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO the element removed from the front OF queue
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.left TO a new instance of TreeNode with value as element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.right TO a new instance of TreeNode with value as element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF p.val NOT EQUALS q.val
        RETURN false
    ELSE
        RETURN the result of is_same_tree with arguments p.left and q.left AND the result of is_same_tree with arguments p.right and q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION zigzagLevelOrder(self, root)
        IF root IS none
            RETURN an empty list
        END IF
        SET result TO an empty list
        SET queue TO a new double ended queue initialized with root
        SET left_to_right TO true
        WHILE queue IS NOT empty
            SET level_size TO the LENGTH OF queue
            SET current_level TO a new empty double ended queue
            FOR each _ IN the range FROM zero TO level_size EXCLUSIVE
                SET node TO the element removed from the front OF queue
                IF left_to_right
                    APPEND node.val TO current_level
                ELSE
                    APPEND node.val TO the front OF current_level
                END IF
                IF node.left IS NOT none
                    APPEND node.left TO queue
                END IF
                IF node.right IS NOT none
                    APPEND node.right TO queue
                END IF
            END FOR
            APPEND the list converted from current_level TO result
            SET left_to_right TO NOT left_to_right
        END WHILE
        RETURN result
    END FUNCTION
END CLASS
