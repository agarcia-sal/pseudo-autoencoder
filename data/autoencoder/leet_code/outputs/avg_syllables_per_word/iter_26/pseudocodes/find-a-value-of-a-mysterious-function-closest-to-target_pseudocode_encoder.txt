CLASS Solution
    FUNCTION closestToTarget(array_of_numbers, target_value)
        SET seen TO empty set of unique values
        SET minimum_difference TO positive infinity

        FOR each number IN array_of_numbers
            SET temporary_set TO empty set
            FOR each element IN seen
                APPEND bitwise AND of number AND element TO temporary_set
            END FOR
            SET seen TO the union of temporary_set AND the set containing number

            FOR each value IN seen
                SET current_difference TO the absolute value of value MINUS target_value
                IF current_difference LESS THAN minimum_difference
                    SET minimum_difference TO current_difference
                END IF
            END FOR
        END FOR

        RETURN minimum_difference
    END FUNCTION
END CLASS
