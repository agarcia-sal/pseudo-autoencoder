CLASS Solution
    FUNCTION singleNumber(list_of_numbers)
        SET ones TO zero
        SET twos TO zero
        FOR each number IN list_of_numbers
            SET ones TO the bitwise exclusive OR of ones AND number AND the bitwise negation of twos
            SET twos TO the bitwise exclusive OR of twos AND number AND the bitwise negation of ones
        END FOR
        RETURN ones
    END FUNCTION
END CLASS
