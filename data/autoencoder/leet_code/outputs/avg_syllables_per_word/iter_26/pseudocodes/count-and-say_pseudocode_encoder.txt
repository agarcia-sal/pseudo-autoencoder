CLASS Solution
    FUNCTION countAndSay(n)
        IF n EQUALS one THEN
            RETURN the string consisting of character one
        END IF

        SET current_sequence TO the string consisting of character one

        FOR each iteration_index FROM two TO n INCLUSIVE
            SET next_sequence TO the empty string
            SET count TO one
            SET previous_char TO the element at position one OF current_sequence

            FOR each position_index FROM two TO the LENGTH OF current_sequence INCLUSIVE
                IF the element at position position_index OF current_sequence EQUALS previous_char THEN
                    INCREMENT count BY one
                ELSE
                    APPEND the string formed by the textual representation of count PLUS previous_char TO next_sequence
                    SET previous_char TO the element at position position_index OF current_sequence
                    SET count TO one
                END IF
            END FOR

            APPEND the string formed by the textual representation of count PLUS previous_char TO next_sequence
            SET current_sequence TO next_sequence
        END FOR

        RETURN current_sequence
    END FUNCTION
END CLASS
