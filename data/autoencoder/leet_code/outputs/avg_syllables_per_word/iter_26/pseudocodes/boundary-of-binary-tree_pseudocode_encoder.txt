CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values EQUALS none OR the LENGTH OF values EQUALS zero THEN
        RETURN none
    END IF
    SET root TO A NEW TreeNode WITH the element at position zero OF values
    SET i TO one
    SET queue TO A NEW DEQUE
    APPEND root TO queue
    WHILE the LENGTH OF queue GREATER THAN zero HOLDS
        SET node TO THE FIRST element REMOVED FROM queue
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values NOT EQUALS none THEN
            SET node.left TO A NEW TreeNode WITH the element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values NOT EQUALS none THEN
            SET node.right TO A NEW TreeNode WITH the element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p EQUALS none AND q EQUALS none THEN
        RETURN true
    ELSE IF p EQUALS none OR q EQUALS none THEN
        RETURN false
    ELSE IF p.val NOT EQUALS q.val THEN
        RETURN false
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION boundaryOfBinaryTree(self, root)
        IF root EQUALS none THEN
            RETURN an empty list
        END IF

        FUNCTION is_leaf(node)
            RETURN node.left EQUALS none AND node.right EQUALS none
        END FUNCTION

        SET left_boundary TO an empty list
        SET current TO root.left
        WHILE current NOT EQUALS none HOLDS
            IF NOT is_leaf(current) THEN
                APPEND current.val TO left_boundary
            END IF
            IF current.left NOT EQUALS none THEN
                SET current TO current.left
            ELSE
                SET current TO current.right
            END IF
        END WHILE

        SET right_boundary TO an empty list
        SET current TO root.right
        WHILE current NOT EQUALS none HOLDS
            IF NOT is_leaf(current) THEN
                APPEND current.val TO right_boundary
            END IF
            IF current.right NOT EQUALS none THEN
                SET current TO current.right
            ELSE
                SET current TO current.left
            END IF
        END WHILE

        SET leaves TO an empty list
        FUNCTION collect_leaves(node)
            IF node EQUALS none THEN
                RETURN
            END IF
            IF is_leaf(node) THEN
                APPEND node.val TO leaves
            END IF
            CALL collect_leaves WITH node.left
            CALL collect_leaves WITH node.right
        END FUNCTION

        CALL collect_leaves WITH root

        IF is_leaf(root) THEN
            SET leaves TO an empty list
        END IF

        RETURN a list CONCATENATING an element list containing root.val PLUS the elements of left_boundary PLUS the elements of leaves PLUS the elements of right_boundary IN REVERSE ORDER
    END FUNCTION
END CLASS
