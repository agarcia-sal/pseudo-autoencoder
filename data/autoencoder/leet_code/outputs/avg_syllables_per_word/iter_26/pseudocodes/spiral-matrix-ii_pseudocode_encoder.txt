CLASS Solution
    FUNCTION generateMatrix(parameter n)
        SET matrix TO a collection of n elements EACH BEING a collection of n elements FILLED WITH zero
        SET top TO zero
        SET bottom TO n MINUS one
        SET left TO zero
        SET right TO n MINUS one
        SET num TO one
        WHILE top LESS THAN OR EQUAL TO bottom AND left LESS THAN OR EQUAL TO right HOLDS
            FOR i FROM left TO right INCLUSIVE
                SET the element at position top IN matrix TO a collection element at position i TO num
                INCREMENT num BY one
            END FOR
            INCREMENT top BY one
            FOR i FROM top TO bottom INCLUSIVE
                SET the element at position i IN matrix TO a collection element at position right TO num
                INCREMENT num BY one
            END FOR
            DECREMENT right BY one
            IF top LESS THAN OR EQUAL TO bottom HOLDS
                FOR i FROM right DOWN TO left INCLUSIVE
                    SET the element at position bottom IN matrix TO a collection element at position i TO num
                    INCREMENT num BY one
                END FOR
                DECREMENT bottom BY one
            END IF
            IF left LESS THAN OR EQUAL TO right HOLDS
                FOR i FROM bottom DOWN TO top INCLUSIVE
                    SET the element at position i IN matrix TO a collection element at position left TO num
                    INCREMENT num BY one
                END FOR
                INCREMENT left BY one
            END IF
        END WHILE
        RETURN matrix
    END FUNCTION
END CLASS
