CLASS Solution
    FUNCTION dailyTemperatures(temperatures)
        SET n TO the LENGTH OF temperatures
        SET answer TO a list OF zero REPEATED n times
        SET stack TO an empty list TO STORE indices OF temperatures

        FOR index FROM zero TO n MINUS one INCLUSIVE
            WHILE stack IS NOT empty AND element at position index OF temperatures IS GREATER THAN element at position element at last position OF stack OF temperatures
                SET previous_index TO the REMOVAL of last element FROM stack
                SET element at position previous_index OF answer TO index MINUS previous_index
            END WHILE
            APPEND index TO stack
        END FOR

        RETURN answer
    END FUNCTION
END CLASS
