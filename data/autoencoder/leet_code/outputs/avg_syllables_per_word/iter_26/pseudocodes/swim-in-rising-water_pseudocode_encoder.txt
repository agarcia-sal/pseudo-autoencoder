CLASS Solution
    FUNCTION swimInWater(grid)
        SET n TO the LENGTH OF grid
        SET visited TO an empty set
        SET min_heap TO a list containing a tuple of element at position zero zero OF grid, zero, zero
        ADD the tuple of zero and zero TO visited
        
        SET directions TO a list containing tuples of zero and one, one and zero, zero and negative one, negative one and zero
        
        WHILE the LENGTH OF min_heap IS GREATER THAN zero HOLDS
            EXTRACT the smallest element FROM min_heap INTO variables t, i, j
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one THEN
                RETURN t
            END IF
            
            FOR each pair di, dj IN directions
                SET ni TO i PLUS di
                SET nj TO j PLUS dj
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND the tuple of ni and nj IS NOT IN visited THEN
                    ADD the tuple of ni and nj TO visited
                    PUSH the tuple of the GREATER OF t AND element at position ni nj OF grid, ni, nj INTO min_heap
                END IF
            END FOR
        END WHILE
        
        RETURN negative one
    END FUNCTION
END CLASS
