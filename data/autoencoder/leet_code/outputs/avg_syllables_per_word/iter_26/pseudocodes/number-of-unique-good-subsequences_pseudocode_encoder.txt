CLASS Solution
    FUNCTION numberOfUniqueGoodSubsequences(binary)
        SET MODULO TO ten POWERED BY nine PLUS seven

        SET end_with_zero TO zero
        SET end_with_one TO zero

        SET has_zero TO WHETHER the CHARACTER zero EXISTS IN binary

        FOR each bit IN binary
            IF bit EQUALS CHARACTER one
                SET end_with_one TO the REMAINDER OF the SUM OF end_with_zero PLUS end_with_one PLUS one DIVIDED BY MODULO
            ELSE
                SET end_with_zero TO the REMAINDER OF the SUM OF end_with_zero PLUS end_with_one DIVIDED BY MODULO
            END IF
        END FOR

        RETURN the REMAINDER OF the SUM OF end_with_zero PLUS end_with_one PLUS has_zero DIVIDED BY MODULO
    END FUNCTION
END CLASS
