CLASS Solution
    FUNCTION singleNumber(list_of_numbers)
        SET xor_of_all_numbers TO zero
        FOR each number IN list_of_numbers
            SET xor_of_all_numbers TO xor_of_all_numbers XOR number
        END FOR

        SET differing_bit TO the BITWISE AND of xor_of_all_numbers AND the NEGATION of xor_of_all_numbers

        SET first_unique_number TO zero
        SET second_unique_number TO zero

        FOR each number IN list_of_numbers
            IF the BITWISE AND of number AND differing_bit IS NOT zero
                SET first_unique_number TO first_unique_number XOR number
            ELSE
                SET second_unique_number TO second_unique_number XOR number
            END IF
        END FOR

        RETURN list CONTAINING first_unique_number AND second_unique_number
    END FUNCTION
END CLASS
