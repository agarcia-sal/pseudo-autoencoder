CLASS Solution

    FUNCTION makeLargestSpecial(s)
    
        FUNCTION dfs(special_string)
            SET count TO zero
            SET start TO zero
            SET specials TO empty list

            FOR each index i AND character char IN special_string
                IF char EQUALS character one
                    INCREMENT count BY one
                ELSE
                    DECREMENT count BY one
                END IF

                IF count EQUALS zero
                    APPEND concatenation of character one PLUS dfs(substring FROM position start PLUS one TO position i MINUS one OF special_string) PLUS character zero TO specials
                    SET start TO i PLUS one
                END IF
            END FOR

            SORT specials IN reverse lexicographical order
            SET joined_specials TO concatenation of all elements IN specials
            RETURN joined_specials
        END FUNCTION

        RETURN dfs(s)
    END FUNCTION

END CLASS
