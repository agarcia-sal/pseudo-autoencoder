CLASS Solution
    FUNCTION frogPosition(n, edges, t, target)
        IF n EQUALS one
            RETURN one
        END IF

        SET graph TO a new mapping with default empty list

        FOR each pair consisting of node_u and node_v IN edges
            APPEND node_v TO element at position node_u OF graph
            APPEND node_u TO element at position node_v OF graph
        END FOR

        SET queue TO a new double ended queue containing the single element consisting of one, one, and zero as current_node, current_probability, and current_time respectively
        SET visited TO a new set containing the single element one

        WHILE queue HOLDS
            REMOVE and ASSIGN the first element of queue TO current_node, current_probability, current_time

            IF current_time EQUALS t OR (current_node NOT EQUALS one AND the LENGTH OF element at position current_node OF graph EQUALS one)
                IF current_node EQUALS target
                    RETURN current_probability
                ELSE
                    CONTINUE to the next iteration of the WHILE
                END IF
            END IF

            SET possible_moves TO the LENGTH OF element at position current_node OF graph MINUS one IF current_node NOT EQUALS one ELSE zero

            FOR each neighbor IN element at position current_node OF graph
                IF neighbor NOT IN visited
                    ADD neighbor TO visited
                    APPEND the tuple consisting of neighbor, current_probability DIVIDED BY possible_moves, and current_time PLUS one TO queue
                END IF
            END FOR
        END WHILE

        RETURN zero
    END FUNCTION
END CLASS
