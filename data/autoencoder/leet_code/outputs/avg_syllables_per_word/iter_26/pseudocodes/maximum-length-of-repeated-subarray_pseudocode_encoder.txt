CLASS Solution
    FUNCTION findLength(nums1, nums2)
        SET dp TO a two dimensional list WITH the number of rows EQUAL TO the length of nums1 PLUS one AND the number of columns EQUAL TO the length of nums2 PLUS one FILLED WITH zero values
        SET max_length TO zero
        FOR index i FROM one TO the length of nums1 INCLUSIVE
            FOR index j FROM one TO the length of nums2 INCLUSIVE
                IF the element at position i MINUS one OF nums1 EQUALS the element at position j MINUS one OF nums2
                    SET the element at position i AND j OF dp TO the element at position i MINUS one AND j MINUS one OF dp PLUS one
                    SET max_length TO the GREATER value BETWEEN max_length AND the element at position i AND j OF dp
                END IF
            END FOR
        END FOR
        RETURN max_length
    END FUNCTION
END CLASS
