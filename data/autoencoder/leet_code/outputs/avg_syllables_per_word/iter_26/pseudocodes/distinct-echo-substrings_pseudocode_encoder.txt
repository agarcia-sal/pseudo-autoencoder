CLASS Solution
    FUNCTION distinctEchoSubstrings(text)
        SET n TO the LENGTH OF text
        ASSIGN substrings TO an EMPTY SET
        
        FOR length FROM one TO the INTEGER DIVISION OF n BY two PLUS one EXCLUSIVE
            FOR start FROM zero TO n MINUS two MULTIPLIED BY length PLUS one EXCLUSIVE
                SET a TO the SUBSTRING FROM position start TO position start PLUS length OF text
                SET b TO the SUBSTRING FROM position start PLUS length TO position start PLUS two MULTIPLIED BY length OF text
                IF a EQUALS b THEN
                    ADD the CONCATENATION OF a AND b TO substrings
                END IF
            END FOR
        END FOR
        
        RETURN the SIZE OF substrings
    END FUNCTION
END CLASS
