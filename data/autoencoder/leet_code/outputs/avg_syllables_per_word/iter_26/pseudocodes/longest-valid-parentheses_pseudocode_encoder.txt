CLASS Solution
    FUNCTION longestValidParentheses(string s)
        SET stack TO a new list containing the negative integer one as the initial base value
        SET max_length TO zero

        FOR each index i AND character char IN the enumeration of string s
            IF character char EQUALS the opening parenthesis symbol
                APPEND index i TO stack
            ELSE
                REMOVE the last element FROM stack
                IF stack IS empty
                    APPEND index i TO stack
                ELSE
                    SET difference TO index i MINUS the last element in stack
                    SET max_length TO the greater value BETWEEN max_length AND difference
                END IF
            END IF
        END FOR

        RETURN max_length
    END FUNCTION
END CLASS
