CLASS Solution
    FUNCTION maxProfit(k, prices)
        IF prices IS EMPTY OR k EQUALS zero
            RETURN zero
        END IF
        
        SET n TO the LENGTH OF prices
        
        IF k GREATER THAN OR EQUAL TO n DIVIDED BY two
            SET profit TO zero
            FOR index FROM two TO n EXCLUSIVE
                IF element at position index OF prices GREATER THAN element at position index MINUS one OF prices
                    INCREMENT profit BY element at position index OF prices MINUS element at position index MINUS one OF prices
                END IF
            END FOR
            RETURN profit
        END IF
        
        INITIALIZE hold TO a LIST OF k PLUS one ELEMENTS EACH SET TO NEGATIVE infinity
        INITIALIZE release TO a LIST OF k PLUS one ELEMENTS EACH SET TO zero
        
        FOR each price IN prices
            FOR transaction_index FROM one TO k INCLUSIVE
                SET hold at position transaction_index TO THE GREATER OF hold at position transaction_index AND release at position transaction_index MINUS one MINUS price
                SET release at position transaction_index TO THE GREATER OF release at position transaction_index AND hold at position transaction_index PLUS price
            END FOR
        END FOR
        
        RETURN release at position k
    END FUNCTION
END CLASS
