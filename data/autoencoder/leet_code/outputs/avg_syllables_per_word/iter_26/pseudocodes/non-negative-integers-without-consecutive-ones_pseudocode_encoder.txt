CLASS Solution
    FUNCTION findIntegers(n)
        SET bin_representation TO the binary representation of n WITHOUT the first TWO characters
        SET length TO the LENGTH OF bin_representation

        CREATE a list named dp WITH length plus ONE elements ALL SET TO zero
        SET the element at position zero OF dp TO one
        SET the element at position one OF dp TO two

        FOR i FROM two TO length INCLUSIVE
            SET the element at position i OF dp TO the element at position i minus one OF dp PLUS the element at position i minus two OF dp
        END FOR

        SET result TO zero
        SET previous_bit TO zero

        FOR i FROM zero TO length MINUS one INCLUSIVE
            IF the element at position i OF bin_representation EQUALS one AS A CHARACTER
                INCREMENT result BY the element at position length MINUS i MINUS one OF dp
                IF previous_bit EQUALS one
                    DECREMENT result BY one
                    BREAK
                END IF
            END IF

            SET previous_bit TO one IF the element at position i OF bin_representation EQUALS one AS A CHARACTER ELSE zero
        END FOR

        RETURN result PLUS one
    END FUNCTION
END CLASS
