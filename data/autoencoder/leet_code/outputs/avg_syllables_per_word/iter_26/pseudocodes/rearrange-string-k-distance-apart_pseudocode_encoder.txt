CLASS Solution
    FUNCTION rearrangeString(string_s, integer_k)
        IF integer_k LESS THAN OR EQUAL TO one
            RETURN string_s
        END IF

        SET frequency_counter TO the count of each character IN string_s
        
        SET max_heap TO an empty list
        FOR each character AND its count IN frequency_counter items
            APPEND the tuple of NEGATIVE count AND character TO max_heap
        END FOR
        CALL the heapify operation ON max_heap

        SET wait_queue TO an empty list

        SET result TO an empty list

        WHILE max_heap HOLDS elements
            REMOVE the element with the highest priority FROM max_heap AND ASSIGN its count_component AND char_component
            APPEND char_component TO result

            APPEND the tuple of count_component PLUS one AND char_component TO wait_queue

            IF the LENGTH OF wait_queue EQUALS integer_k
                REMOVE the first element FROM wait_queue AND ASSIGN its count_component AND char_component
                IF count_component LESS THAN zero
                    PUSH the tuple of count_component AND char_component INTO max_heap
                END IF
            END IF
        END WHILE

        IF the LENGTH OF result NOT EQUALS the LENGTH OF string_s
            RETURN an empty string
        END IF

        RETURN the concatenation of all elements IN result AS a single string
    END FUNCTION
END CLASS
