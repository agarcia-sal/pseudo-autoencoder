CLASS Solution
    FUNCTION myPow(x_float, n_integer) RETURNS float
        FUNCTION qpow(a_float, n_integer) RETURNS float
            SET ans TO one
            WHILE n_integer GREATER THAN zero HOLDS
                IF the BITWISE AND OF n_integer AND one EQUALS one
                    SET ans TO ans MULTIPLIED BY a_float
                END IF
                SET a_float TO a_float MULTIPLIED BY a_float
                SET n_integer TO n_integer DIVIDED BY two USING INTEGER DIVISION
            END WHILE
            RETURN ans
        END FUNCTION

        IF n_integer GREATER THAN OR EQUAL TO zero
            RETURN qpow WITH PARAMETERS x_float AND n_integer
        ELSE
            RETURN one DIVIDED BY qpow WITH PARAMETERS x_float AND the NEGATION OF n_integer
        END IF
    END FUNCTION
END CLASS
