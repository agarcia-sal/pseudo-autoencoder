CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values DOES NOT HOLD ANY ELEMENTS
        RETURN none
    END IF
    SET root TO a NEW TreeNode INITIALIZED WITH the element at position zero OF values
    SET i TO one
    SET queue TO a new empty double ended queue
    APPEND root TO queue
    WHILE queue HOLDS ELEMENTS
        SET node TO the element REMOVED FROM the LEFT OF queue
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.left TO a NEW TreeNode INITIALIZED WITH the element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.right TO a NEW TreeNode INITIALIZED WITH the element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN false
    ELSE
        RETURN the RESULT OF is_same_tree WITH PARAMETERS p.left AND q.left AND the RESULT OF is_same_tree WITH PARAMETERS p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION levelOrder(self, root)
        IF root IS none
            RETURN an empty list
        END IF
        SET result TO an empty list
        SET queue TO a new double ended queue INITIALIZED WITH root as its single element
        WHILE queue HOLDS ELEMENTS
            SET level_size TO the LENGTH OF queue
            SET current_level TO an empty list
            FOR each number FROM zero TO level_size MINUS one INCLUSIVE
                SET node TO the element REMOVED FROM the LEFT OF queue
                APPEND node.val TO current_level
                IF node.left HOLDS AN ELEMENT
                    APPEND node.left TO queue
                END IF
                IF node.right HOLDS AN ELEMENT
                    APPEND node.right TO queue
                END IF
            END FOR
            APPEND current_level TO result
        END WHILE
        RETURN result
    END FUNCTION
END CLASS
