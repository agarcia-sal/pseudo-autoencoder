CLASS Solution
    FUNCTION isReflected(points)
        SET unique_points TO the SET OF each TUPLE CONVERSION OF each element IN points
        SET minimum_x TO the SMALLEST value of the element AT position zero OF each point IN unique_points
        SET maximum_x TO the LARGEST value of the element AT position zero OF each point IN unique_points
        SET line_of_reflection TO minimum_x PLUS maximum_x DIVIDED BY two
        FOR each point IN unique_points
            SET x TO the element AT position zero OF point
            SET y TO the element AT position one OF point
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x
            IF the TUPLE OF reflected_x AND y IS NOT IN unique_points
                RETURN false
            END IF
        END FOR
        RETURN true
    END FUNCTION
END CLASS
