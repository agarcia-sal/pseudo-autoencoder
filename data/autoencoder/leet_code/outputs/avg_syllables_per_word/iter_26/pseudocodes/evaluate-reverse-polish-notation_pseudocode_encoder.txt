CLASS Solution
    FUNCTION evalRPN(tokens)
        SET stack TO an empty list
        FOR each token IN tokens
            IF token EQUALS plus OR token EQUALS minus OR token EQUALS multiplied by OR token EQUALS divided by
                SET b TO the element removed from the end of stack
                SET a TO the element removed from the end of stack
                IF token EQUALS plus
                    APPEND the result of a PLUS b TO stack
                ELSE IF token EQUALS minus
                    APPEND the result of a MINUS b TO stack
                ELSE IF token EQUALS multiplied by
                    APPEND the result of a MULTIPLIED BY b TO stack
                ELSE IF token EQUALS divided by
                    APPEND the integer result of a DIVIDED BY b TRUNCATED TOWARDS ZERO TO stack
                END IF
            ELSE
                APPEND the integer value of token TO stack
            END IF
        END FOR
        RETURN the element at position zero OF stack
    END FUNCTION
END CLASS
