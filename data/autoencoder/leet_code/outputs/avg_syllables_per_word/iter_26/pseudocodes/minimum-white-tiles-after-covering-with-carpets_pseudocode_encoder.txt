CLASS Solution
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)
        SET n TO the LENGTH OF floor

        SET dp TO a TWO DIMENSIONAL list OF zeros with the NUMBER OF ROWS EQUAL TO n PLUS one AND the NUMBER OF COLUMNS EQUAL TO numCarpets PLUS one

        FOR i FROM one TO n INCLUSIVE
            SET dp at position i and zero TO dp at position i MINUS one and zero PLUS the RESULT OF checking WHETHER the element at position i MINUS one OF floor EQUALS the character one
        END FOR

        FOR j FROM one TO numCarpets INCLUSIVE
            FOR i FROM one TO n INCLUSIVE
                SET dp at position i and j TO dp at position i MINUS one and j PLUS the RESULT OF checking WHETHER the element at position i MINUS one OF floor EQUALS the character one
                SET dp at position i and j TO THE MINIMUM BETWEEN dp at position i and j AND dp at position the GREATER OF zero AND i MINUS carpetLen and position j MINUS one
            END FOR
        END FOR

        RETURN dp at position n and numCarpets
    END FUNCTION
END CLASS
