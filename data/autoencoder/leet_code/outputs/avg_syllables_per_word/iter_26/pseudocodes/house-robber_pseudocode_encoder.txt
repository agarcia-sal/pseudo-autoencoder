CLASS Solution
    FUNCTION rob(parameter nums AS list of integers) RETURNS integer
        IF NOT nums HOLDS
            RETURN zero
        END IF

        IF the LENGTH OF nums EQUALS one
            RETURN the element at position zero OF nums
        END IF

        SET dp TO a new list of zeros WITH the LENGTH OF nums

        SET the element at position zero OF dp TO the element at position zero OF nums

        SET the element at position one OF dp TO THE GREATER VALUE BETWEEN the element at position zero OF nums AND the element at position one OF nums

        FOR index i FROM two TO the LENGTH OF nums MINUS one INCLUSIVE
            SET the element at position i OF dp TO THE GREATER VALUE BETWEEN the element at position i MINUS one OF dp AND the sum of the element at position i MINUS two OF dp PLUS the element at position i OF nums
        END FOR

        RETURN the element at the last position OF dp
    END FUNCTION
END CLASS
