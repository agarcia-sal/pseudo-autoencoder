CLASS Solution
    FUNCTION combinationSum(candidates, target)
        FUNCTION backtrack(start, target_value, path_list)
            IF target_value EQUALS zero
                APPEND path_list TO result
                RETURN
            END IF
            IF target_value LESS THAN zero
                RETURN
            END IF
            FOR index FROM start TO the LENGTH OF candidates MINUS one INCLUSIVE
                CALL backtrack WITH arguments index, target_value MINUS the element at position index OF candidates, path_list PLUS a new list containing the element at position index OF candidates
            END FOR
        END FUNCTION

        SET result TO an empty list
        CALL backtrack WITH arguments zero, target, an empty list
        RETURN result
    END FUNCTION
END CLASS
