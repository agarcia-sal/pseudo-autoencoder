CLASS Solution
    FUNCTION findMinDifference(timePoints)
        FUNCTION to_minutes(time)
            DESCRIBE hours AND minutes AS the TWO PARTS FROM SPLITTING time AT THE COLON
            SET hours TO the INTEGER CONVERSION OF THE FIRST PART OF time SPLIT AT THE COLON
            SET minutes TO the INTEGER CONVERSION OF THE SECOND PART OF time SPLIT AT THE COLON
            RETURN hours MULTIPLIED BY the NUMBER OF MINUTES IN AN HOUR PLUS minutes
        END FUNCTION

        SET minutes_list TO the LIST RESULT OF APPLYING to_minutes TO EACH ELEMENT OF timePoints
        SORT minutes_list IN ASCENDING ORDER

        SET min_diff TO a VALUE REPRESENTING INFINITY

        FOR i FROM the NUMBER ONE TO the LENGTH OF minutes_list MINUS ONE INCLUSIVE
            SET current_difference TO the ELEMENT AT POSITION i OF minutes_list MINUS the ELEMENT AT POSITION i MINUS one OF minutes_list
            SET min_diff TO THE SMALLER VALUE BETWEEN min_diff AND current_difference
        END FOR

        SET circular_difference TO the NUMBER OF MINUTES IN A DAY MINUS the ELEMENT AT THE LAST POSITION OF minutes_list PLUS the ELEMENT AT THE FIRST POSITION OF minutes_list
        SET min_diff TO THE SMALLER VALUE BETWEEN min_diff AND circular_difference

        RETURN min_diff
    END FUNCTION
END CLASS
