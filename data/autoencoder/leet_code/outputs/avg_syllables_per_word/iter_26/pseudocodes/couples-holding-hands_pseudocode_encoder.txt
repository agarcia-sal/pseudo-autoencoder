CLASS Solution
    FUNCTION minSwapsCouples(row OF LIST OF INTEGER) RETURNS INTEGER
        SET position TO an EMPTY dictionary
        FOR index FROM zero TO the LENGTH OF row MINUS one INCLUSIVE
            SET person TO the element at position index OF row
            SET position AT key person TO index
        END FOR
        SET swaps TO zero
        FOR index FROM zero TO the LENGTH OF row MINUS one IN STEPS OF two INCLUSIVE
            SET correct_pair TO the element at position index OF row EXCLUSIVE OR with one
            IF the element at position index PLUS one OF row NOT EQUALS correct_pair THEN
                SET swap_idx TO the value AT key correct_pair IN position
                SET temporary_person TO the element at position index PLUS one OF row
                SET the element at position index PLUS one OF row TO the element at position swap_idx OF row
                SET the element at position swap_idx OF row TO temporary_person
                SET position AT key the element at position swap_idx OF row TO swap_idx
                SET position AT key the element at position index PLUS one OF row TO index PLUS one
                INCREMENT swaps BY one
            END IF
        END FOR
        RETURN swaps
    END FUNCTION
END CLASS
