CLASS Solution
    FUNCTION winnerSquareGame(number_of_stones)
        SET dynamic_programming_list TO a list of boolean false values of length number_of_stones PLUS one
        FOR index FROM ONE TO number_of_stones INCLUSIVE
            SET square_root_candidate TO ONE
            WHILE square_root_candidate MULTIPLIED BY square_root_candidate LESS THAN OR EQUAL TO index HOLDS
                IF NOT the element at position index MINUS square_root_candidate MULTIPLIED BY square_root_candidate OF dynamic_programming_list
                    SET the element at position index OF dynamic_programming_list TO true
                    BREAK the inner WHILE loop
                END IF
                INCREMENT square_root_candidate BY ONE
            END WHILE
        END FOR
        RETURN the element at position number_of_stones OF dynamic_programming_list
    END FUNCTION
END CLASS
