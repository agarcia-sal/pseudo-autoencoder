CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS NOT present HOLDS false
        RETURN none
    END IF
    SET root TO A NEW TreeNode WITH the element at position zero OF values
    SET i TO one
    SET queue TO AN EMPTY DEQUE
    APPEND root TO queue
    WHILE queue IS NOT EMPTY HOLDS
        SET node TO THE FIRST element REMOVED FROM queue
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none HOLDS true
            SET node.left TO A NEW TreeNode WITH the element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none HOLDS true
            SET node.right TO A NEW TreeNode WITH the element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS NOT present HOLDS false AND q IS NOT present HOLDS false
        RETURN true
    ELSE IF p IS NOT present HOLDS false OR q IS NOT present HOLDS false
        RETURN false
    ELSE IF THE val ATTRIBUTE OF p IS NOT EQUALS TO THE val ATTRIBUTE OF q
        RETURN false
    ELSE
        RETURN THE RESULT OF is_same_tree WITH THE left ATTRIBUTE OF p AND THE left ATTRIBUTE OF q AND THE RESULT OF is_same_tree WITH THE right ATTRIBUTE OF p AND THE right ATTRIBUTE OF q
    END IF
END FUNCTION

CLASS Solution
    FUNCTION sumNumbers(self, root)
        FUNCTION dfs(node, current_number)
            IF node IS NOT present HOLDS false
                RETURN zero
            END IF
            SET current_number TO current_number MULTIPLIED BY ten PLUS THE val ATTRIBUTE OF node

            IF THE left ATTRIBUTE OF node IS NOT present HOLDS false AND THE right ATTRIBUTE OF node IS NOT present HOLDS false
                RETURN current_number
            END IF

            SET left_sum TO THE RESULT OF dfs WITH THE left ATTRIBUTE OF node AND current_number
            SET right_sum TO THE RESULT OF dfs WITH THE right ATTRIBUTE OF node AND current_number

            RETURN left_sum PLUS right_sum
        END FUNCTION

        RETURN dfs WITH root AND zero
    END FUNCTION
END CLASS
