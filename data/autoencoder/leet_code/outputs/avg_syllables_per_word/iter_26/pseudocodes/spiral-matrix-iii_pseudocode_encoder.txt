CLASS Solution
    FUNCTION spiralMatrixIII(rows, cols, rStart, cStart)
        SET directions TO list of pairs zero AND one, one AND zero, zero AND negative one, negative one AND zero
        SET direction_index TO zero
        SET steps TO zero
        SET increment TO zero
        SET visited TO empty list
        SET x TO rStart
        SET y TO cStart
        
        WHILE the LENGTH OF visited IS LESS THAN rows MULTIPLIED BY cols HOLDS
            IF increment MODULO two EQUALS zero
                INCREMENT steps BY one
            END IF
            FOR each iteration FROM one TO steps INCLUSIVE
                IF zero LESS THAN OR EQUAL TO x AND x LESS THAN rows AND zero LESS THAN OR EQUAL TO y AND y LESS THAN cols
                    APPEND list containing x AND y TO visited
                END IF
                SET dx TO element at position direction_index OF directions FIRST element
                SET dy TO element at position direction_index OF directions SECOND element
                SET x TO x PLUS dx
                SET y TO y PLUS dy
            END FOR
            SET direction_index TO the remainder of the sum of direction_index AND one DIVIDED BY four
            INCREMENT increment BY one
        END WHILE
        
        RETURN visited
    END FUNCTION
END CLASS
