CLASS Solution
    FUNCTION killProcess(pid LIST OF INTEGERS, ppid LIST OF INTEGERS, kill INTEGER) RETURNS LIST OF INTEGERS
        SET tree TO a dictionary with default empty list for missing keys
        FOR each pair of child process and parent process FROM PAIRS OF pid AND ppid
            APPEND child process TO the list at key parent process IN tree
        END FOR

        SET queue TO a list containing kill process
        SET killed_processes TO an empty list

        WHILE the list queue IS NOT empty
            SET current TO the element at position zero OF queue
            REMOVE the element at position zero FROM queue
            APPEND current TO killed_processes
            EXTEND queue BY the list of child processes corresponding to current IN tree
        END WHILE

        RETURN killed_processes
    END FUNCTION
END CLASS
