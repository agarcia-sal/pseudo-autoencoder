CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN none
    END IF
    SET root TO NEW TreeNode WITH value THE element AT position zero OF values
    SET i TO one
    SET queue TO an EMPTY double ended queue
    APPEND root TO queue
    WHILE queue HOLDS elements
        SET node TO the FIRST element REMOVED FROM queue
        IF i IS LESS THAN THE LENGTH OF values AND the element AT position i OF values IS NOT none
            SET node.left TO NEW TreeNode WITH value THE element AT position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN THE LENGTH OF values AND the element AT position i OF values IS NOT none
            SET node.right TO NEW TreeNode WITH value THE element AT position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF p.val NOT EQUALS q.val
        RETURN false
    ELSE
        RETURN is_same_tree WITH parameters p.left AND q.left AND is_same_tree WITH parameters p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION findFrequentTreeSum(root)
        FUNCTION subtree_sum(node)
            IF node IS none
                RETURN zero
            END IF
            SET current_sum TO node.val PLUS subtree_sum WITH parameter node.left PLUS subtree_sum WITH parameter node.right
            INCREMENT count AT key current_sum BY one
            RETURN current_sum
        END FUNCTION

        SET count TO an EMPTY default dictionary WITH integer default
        CALL subtree_sum WITH parameter root
        SET max_freq TO the MAXIMUM value AMONG values OF count WITH default zero
        SET result TO an EMPTY list
        FOR each key FREQUENCY freq IN the ITEMS OF count
            IF freq EQUALS max_freq
                APPEND key TO result
            END IF
        END FOR
        RETURN result
    END FUNCTION
END CLASS
