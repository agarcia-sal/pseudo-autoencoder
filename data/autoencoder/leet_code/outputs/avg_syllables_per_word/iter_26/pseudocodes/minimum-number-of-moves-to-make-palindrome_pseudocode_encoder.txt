CLASS Solution
    FUNCTION minMovesToMakePalindrome(parameter s OF TYPE string)
        SET s TO list CONVERTED FROM s
        SET n TO the LENGTH OF s
        SET moves TO zero

        WHILE the LENGTH OF s GREATER THAN one HOLDS
            FOR i FROM the LENGTH OF s MINUS one DOWN TO one INCLUSIVE
                IF the element at position i OF s EQUALS the element at position zero OF s
                    FOR j FROM i TO the LENGTH OF s MINUS two INCLUSIVE
                        SWAP the element at position j OF s WITH the element at position j PLUS one OF s
                        INCREMENT moves BY one
                    END FOR
                    REMOVE the element at position zero FROM s
                    REMOVE the element at the LAST POSITION FROM s
                    BREAK out of the FOR loop
                END IF
            END FOR
            IF no matching character WAS FOUND in the FOR loop
                REMOVE the element at position zero FROM s
            END IF
        END WHILE

        RETURN moves
    END FUNCTION
END CLASS
