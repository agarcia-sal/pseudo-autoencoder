CLASS Solution
    FUNCTION lengthLongestPath(input)
        SET lines TO the result of splitting input BY newline character
        SET stack TO empty list
        SET max_length TO zero

        FOR each line IN lines
            SET depth TO the number of tab characters IN line
            SET name TO the substring of line FROM position depth TO the end

            WHILE the LENGTH OF stack GREATER THAN depth HOLDS
                REMOVE the last element FROM stack
            END WHILE

            IF the LENGTH OF stack IS GREATER THAN zero
                SET current_length TO the last element in stack PLUS the LENGTH OF name PLUS one
            ELSE
                SET current_length TO the LENGTH OF name
            END IF

            IF the character for a period IS PRESENT IN name
                SET max_length TO the greater value BETWEEN max_length AND current_length
            ELSE
                APPEND current_length TO stack
            END IF
        END FOR

        RETURN max_length
    END FUNCTION
END CLASS
