CLASS Solution
    FUNCTION isConvex(points)
        FUNCTION cross_product(point_o, point_a, point_b)
            RETURN THE quantity OF the element at position zero OF point_a MINUS the element at position zero OF point_o MULTIPLIED BY the element at position one OF point_b MINUS the element at position one OF point_o MINUS THE quantity OF the element at position one OF point_a MINUS the element at position one OF point_o MULTIPLIED BY the element at position zero OF point_b MINUS the element at position zero OF point_o
        END FUNCTION

        SET n TO the LENGTH OF points
        IF n LESS THAN OR EQUAL TO three THEN
            RETURN true
        END IF

        SET orientation TO zero

        FOR index i FROM zero TO n MINUS one INCLUSIVE
            SET o TO the element at position i OF points
            SET a TO the element at position THE remainder OF i PLUS one DIVIDED BY n OF points
            SET b TO the element at position THE remainder OF i PLUS two DIVIDED BY n OF points
            SET cross TO cross_product WITH argument point_o o AND point_a a AND point_b b

            IF cross NOT EQUALS zero THEN
                IF orientation EQUALS zero THEN
                    IF cross GREATER THAN zero THEN
                        SET orientation TO one
                    ELSE
                        SET orientation TO negative one
                    END IF
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS negative one AND cross GREATER THAN zero THEN
                    RETURN false
                END IF
            END IF
        END FOR

        RETURN true
    END FUNCTION
END CLASS
