CLASS Solution
    FUNCTION topKFrequent(words, k)
        SET count TO the FREQUENCY COUNT OF each element IN words USING COUNTER
        SET heap TO a LIST OF PAIRS OF NEGATION OF frequency AND word FOR each word AND frequency IN count
        TRANSFORM heap INTO A HEAP STRUCTURE USING HEAPIFY
        SET result TO an EMPTY LIST
        FOR each index FROM one TO k INCLUSIVE
            APPEND the second element OF THE RESULT OF REMOVING THE SMALLEST ELEMENT FROM heap TO result
        END FOR
        RETURN result
    END FUNCTION
END CLASS
