CLASS Solution
    FUNCTION findBlackPixel(picture AS List OF List OF string, target AS integer) AS integer
        IF picture IS EMPTY OR element at position zero OF picture IS EMPTY THEN
            RETURN zero
        END IF

        SET m TO the LENGTH OF picture
        SET n TO the LENGTH OF element at position zero OF picture
        INITIALIZE row_count AS a list OF length m FILLED WITH zero
        INITIALIZE col_count AS a list OF length n FILLED WITH zero
        INITIALIZE row_patterns AS a mapping FROM string TO integer WITH DEFAULT zero

        FOR each r FROM zero TO m MINUS one INCLUSIVE
            SET pattern TO the concatenation OF all elements IN element at position r OF picture
            INCREMENT the value mapped by pattern IN row_patterns BY one
            FOR each c FROM zero TO n MINUS one INCLUSIVE
                IF element at position c OF element at position r OF picture EQUALS character B THEN
                    INCREMENT the element at position r OF row_count BY one
                    INCREMENT the element at position c OF col_count BY one
                END IF
            END FOR
        END FOR

        SET lonely_pixels TO zero

        FOR each r FROM zero TO m MINUS one INCLUSIVE
            FOR each c FROM zero TO n MINUS one INCLUSIVE
                IF element at position c OF element at position r OF picture EQUALS character B THEN
                    SET pattern TO the concatenation OF all elements IN element at position r OF picture
                    IF the element at position r OF row_count EQUALS target AND the element at position c OF col_count EQUALS target AND the value mapped by pattern IN row_patterns EQUALS target THEN
                        INCREMENT lonely_pixels BY one
                    END IF
                END IF
            END FOR
        END FOR

        RETURN lonely_pixels
    END FUNCTION
END CLASS
