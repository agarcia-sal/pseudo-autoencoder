CLASS Solution
    FUNCTION nextGreaterElements(nums)
        SET n TO the LENGTH OF nums
        SET result TO a LIST OF n ELEMENTS each SET TO negative one
        SET stack TO an EMPTY LIST

        FOR i FROM zero TO n MULTIPLIED BY two MINUS one INCLUSIVE
            SET index TO i MODULO n
            WHILE stack IS NOT EMPTY AND element AT position element AT position the LAST element OF stack OF nums IS LESS THAN element AT position index OF nums HOLDS
                SET popped_index TO REMOVE the LAST element FROM stack
                SET element AT position popped_index OF result TO element AT position index OF nums
            END WHILE
            IF i IS LESS THAN n THEN
                APPEND index TO stack
            END IF
        END FOR

        RETURN result
    END FUNCTION
END CLASS
