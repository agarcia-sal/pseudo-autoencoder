CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS NOT PRESENT
        RETURN none
    END IF
    SET root TO A NEW TreeNode WITH the element at position zero OF values
    SET i TO one
    SET queue TO A NEW deque
    APPEND root TO queue
    WHILE queue HOLDS
        SET node TO THE FIRST ELEMENT REMOVED FROM queue
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.left TO A NEW TreeNode WITH the element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.right TO A NEW TreeNode WITH the element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS NOT PRESENT AND q IS NOT PRESENT
        RETURN true
    ELSE IF p IS NOT PRESENT OR q IS NOT PRESENT
        RETURN false
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN false
    ELSE
        SET left_comparison TO THE RESULT OF is_same_tree WITH p.left AND q.left
        SET right_comparison TO THE RESULT OF is_same_tree WITH p.right AND q.right
        RETURN left_comparison AND right_comparison
    END IF
END FUNCTION

CLASS Solution
    FUNCTION pathSum(self, root, targetSum)
        FUNCTION dfs(node, current_sum, path)
            IF node IS NOT PRESENT
                RETURN
            END IF
            APPEND node.val TO path
            SET current_sum TO current_sum PLUS node.val
            IF node.left IS NOT PRESENT AND node.right IS NOT PRESENT AND current_sum IS EQUAL TO targetSum
                APPEND A COPY OF path TO result
            END IF
            CALL dfs WITH node.left, current_sum, path
            CALL dfs WITH node.right, current_sum, path
            REMOVE THE LAST ELEMENT FROM path
        END FUNCTION

        SET result TO AN EMPTY LIST
        CALL dfs WITH root, zero, AN EMPTY LIST
        RETURN result
    END FUNCTION
END CLASS
