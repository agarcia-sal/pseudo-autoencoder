CLASS Solution
    FUNCTION maxRotateFunction(nums)
        SET n TO the LENGTH OF nums
        IF n EQUALS one
            RETURN zero
        END IF

        SET total_sum TO the SUM OF nums
        SET current_sum TO the SUM OF each element AT position i MULTIPLIED BY the element AT position i OF nums FOR each index i AND element num IN nums WITH i AND num RESPECTIVELY

        SET max_value TO current_sum

        FOR k FROM one TO n MINUS one INCLUSIVE
            SET current_sum TO current_sum PLUS total_sum MINUS n MULTIPLIED BY the element AT position n MINUS k OF nums
            SET max_value TO the GREATER OF max_value AND current_sum
        END FOR

        RETURN max_value
    END FUNCTION
END CLASS
