CLASS Solution
    FUNCTION combinationSum3(parameter k, parameter n)
        FUNCTION backtrack(parameter start, parameter path, parameter target, parameter k)
            IF target EQUALS zero AND k EQUALS zero
                APPEND a COPY OF path TO result
                RETURN
            END IF
            IF target LESS THAN zero OR k LESS THAN zero
                RETURN
            END IF
            FOR variable i FROM start TO nine INCLUSIVE
                APPEND i TO path
                CALL backtrack WITH arguments the SUM OF i AND one, path, target MINUS i, k MINUS one
                REMOVE the LAST ELEMENT FROM path
            END FOR
        END FUNCTION

        SET result TO empty list
        CALL backtrack WITH arguments one, empty list, n, k
        RETURN result
    END FUNCTION
END CLASS
