CLASS Solution
    FUNCTION findRepeatedDnaSequences(string_s)
        SET sequence_count TO a new default dictionary with integer default

        FOR index_i FROM zero TO the LENGTH OF string_s MINUS nine MINUS one INCLUSIVE
            SET sequence TO substring FROM position index_i TO position index_i PLUS ten MINUS one OF string_s
            INCREMENT the value AT key sequence IN sequence_count BY one
        END FOR

        SET repeated_sequences TO an empty list
        FOR each sequence_key AND count_value IN sequence_count items
            IF count_value GREATER THAN one
                APPEND sequence_key TO repeated_sequences
            END IF
        END FOR

        RETURN repeated_sequences
    END FUNCTION
END CLASS
