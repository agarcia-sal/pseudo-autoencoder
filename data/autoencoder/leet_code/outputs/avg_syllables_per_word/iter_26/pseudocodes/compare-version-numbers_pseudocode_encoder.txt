CLASS Solution
    FUNCTION compareVersion(parameter version1 OF TYPE string, parameter version2 OF TYPE string) RETURNS integer
        SET v1_revisions TO the LIST OF SUBSTRINGS OF version1 SPLIT BY the CHARACTER dot
        SET v2_revisions TO the LIST OF SUBSTRINGS OF version2 SPLIT BY the CHARACTER dot

        SET max_length TO the GREATER OF the LENGTH OF v1_revisions AND the LENGTH OF v2_revisions

        FOR index i FROM zero TO max_length MINUS one INCLUSIVE
            IF i LESS THAN the LENGTH OF v1_revisions
                SET v1 TO the WHOLE NUMBER CONVERSION OF the ELEMENT AT POSITION i OF v1_revisions
            ELSE
                SET v1 TO zero
            END IF

            IF i LESS THAN the LENGTH OF v2_revisions
                SET v2 TO the WHOLE NUMBER CONVERSION OF the ELEMENT AT POSITION i OF v2_revisions
            ELSE
                SET v2 TO zero
            END IF

            IF v1 LESS THAN v2
                RETURN negative one
            ELSE IF v1 GREATER THAN v2
                RETURN positive one
            END IF
        END FOR

        RETURN zero
    END FUNCTION
END CLASS
