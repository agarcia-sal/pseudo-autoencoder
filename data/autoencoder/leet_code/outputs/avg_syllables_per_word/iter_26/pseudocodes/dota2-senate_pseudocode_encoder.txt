CLASS Solution
    FUNCTION predictPartyVictory(parameter senate AS string)
        IMPORT deque FROM collections

        SET radiant TO a new empty deque
        SET dire TO a new empty deque

        FOR each position index AND senator_symbol IN the enumeration OF senate
            IF senator_symbol EQUALS uppercase letter R
                APPEND position index TO radiant
            ELSE
                APPEND position index TO dire
            END IF
        END FOR

        WHILE radiant IS NOT empty AND dire IS NOT empty
            SET radiant_index TO the element removed from the front of radiant
            SET dire_index TO the element removed from the front of dire

            IF radiant_index IS LESS THAN dire_index
                APPEND radiant_index PLUS the LENGTH OF senate TO radiant
            ELSE
                APPEND dire_index PLUS the LENGTH OF senate TO dire
            END IF
        END WHILE

        IF radiant IS NOT empty
            RETURN the word Radiant
        ELSE
            RETURN the word Dire
        END IF
    END FUNCTION
END CLASS
