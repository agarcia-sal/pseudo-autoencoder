CLASS Solution
    FUNCTION largestNumber(cost, target)
        SET cost_to_digit TO an empty dictionary mapping each element c IN the enumeration of cost TO the string form of the addition of index i and one
        SET dp TO a list starting with an empty string FOLLOWED BY a repetition of a string representing impossible state MINUS one FOR each value FROM one TO target INCLUSIVE
        
        FOR each integer t FROM one TO target INCLUSIVE
            FOR each key c AND corresponding digit d IN the entries OF cost_to_digit
                IF t GREATER THAN OR EQUAL TO c AND the element at position t MINUS c OF dp NOT EQUALS the string representing impossible state MINUS one
                    SET candidate TO the concatenation of d FOLLOWED BY element at position t MINUS c OF dp
                    IF the element at position t OF dp EQUALS the string representing impossible state MINUS one OR the LENGTH OF candidate GREATER THAN the LENGTH OF the element at position t OF dp OR the LENGTH OF candidate EQUALS the LENGTH OF the element at position t OF dp AND candidate GREATER THAN the element at position t OF dp
                        SET the element at position t OF dp TO candidate
                    END IF
                END IF
            END FOR
        END FOR
        
        IF the element at position target OF dp NOT EQUALS the string representing impossible state MINUS one
            RETURN the element at position target OF dp
        ELSE
            RETURN the string zero
        END IF
    END FUNCTION
END CLASS
