CLASS Solution
    FUNCTION mirrorReflection(p, q)
        FROM the mathematical functions IMPORT greatest common divisor
        SET least common multiple TO p MULTIPLIED BY q DIVIDED BY greatest common divisor OF p AND q
        SET vertical reflections TO least common multiple DIVIDED BY p
        SET horizontal reflections TO least common multiple DIVIDED BY q
        IF vertical reflections MODULO two EQUALS zero
            RETURN zero
        ELSE IF horizontal reflections MODULO two NOT EQUALS zero
            RETURN one
        ELSE
            RETURN two
        END IF
    END FUNCTION
END CLASS
