CLASS Solution
    FUNCTION minDistance(word1, word2)
        SET m TO the LENGTH OF word1
        SET n TO the LENGTH OF word2
        
        CREATE a two dimensional list named dp WITH dimensions m PLUS one BY n PLUS one FILLED WITH zero
        
        FOR i FROM zero TO m INCLUSIVE
            SET the element at position i AT position zero OF dp TO i
        END FOR
        FOR j FROM zero TO n INCLUSIVE
            SET the element at position zero AT position j OF dp TO j
        END FOR
        
        FOR i FROM one TO m INCLUSIVE
            FOR j FROM one TO n INCLUSIVE
                IF the element at position i MINUS one OF word1 EQUALS the element at position j MINUS one OF word2
                    SET the element at position i AT position j OF dp TO the element at position i MINUS one AT position j MINUS one OF dp
                ELSE
                    SET the element at position i AT position j OF dp TO one PLUS the MINIMUM OF the element at position i MINUS one AT position j OF dp AND the element at position i AT position j MINUS one OF dp AND the element at position i MINUS one AT position j MINUS one OF dp
                END IF
            END FOR
        END FOR
        
        RETURN the element at position m AT position n OF dp
    END FUNCTION
END CLASS
