CLASS Solution
    FUNCTION minChanges(nums, k)
        SET groups TO a new default dictionary mapping keys to empty lists
        FOR i FROM zero TO the LENGTH OF nums MINUS ONE INCLUSIVE
            APPEND element at position i OF nums TO the list AT key of the remainder of i DIVIDED BY k OF groups
        END FOR

        SET freq TO a list containing k new default dictionaries mapping keys to zero
        SET min_size TO a list of k elements each set to infinity
        FOR i FROM zero TO k MINUS ONE INCLUSIVE
            FOR each number IN the list AT key i OF groups
                INCREMENT the value AT key number OF freq at position i BY one
            END FOR
            SET element at position i OF min_size TO the LENGTH OF the list AT key i OF groups
        END FOR

        SET dp TO a list of length two RAISED TO THE POWER OF ten each element set to infinity
        SET element at position zero OF dp TO zero

        FOR i FROM zero TO k MINUS ONE INCLUSIVE
            SET new_dp TO a list of length two RAISED TO_THE_POWER_OF ten each element set to infinity
            SET min_dp TO the MINIMUM VALUE in dp
            FOR j FROM zero TO two RAISED_TO_THE_POWER_OF ten MINUS ONE INCLUSIVE
                SET element at position j OF new_dp TO min_dp PLUS element at position i OF min_size
                FOR each number, count PAIR IN freq at position i
                    SET current_value TO element at position j EXCLUSIVE_OR number OF dp PLUS element at position i OF min_size MINUS count
                    IF current_value IS LESS THAN element at position j OF new_dp
                        SET element at position j OF new_dp TO current_value
                    END IF
                END FOR
            END FOR
            ASSIGN dp TO new_dp
        END FOR

        RETURN element at position zero OF dp
    END FUNCTION
END CLASS
