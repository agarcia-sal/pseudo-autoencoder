CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS none OR the LENGTH OF values EQUALS zero
        RETURN none
    END IF
    SET root TO AN INSTANCE OF TreeNode WITH val EQUALS the element at position zero OF values
    SET i TO one
    CREATE queue AS AN EMPTY double ended queue
    APPEND root TO queue
    WHILE the LENGTH OF queue IS GREATER THAN zero HOLDS
        REMOVE the element at the FRONT OF queue AND ASSIGN TO node
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.left TO AN INSTANCE OF TreeNode WITH val EQUALS the element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.right TO AN INSTANCE OF TreeNode WITH val EQUALS the element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF p.val NOT EQUALS q.val
        RETURN false
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION minCameraCover(self, root)
        FUNCTION dfs(node)
            IF node IS none
                RETURN zero, zero, positive infinity
            END IF
            SET left_no_camera, left_monitored, left_camera TO dfs(node.left)
            SET right_no_camera, right_monitored, right_camera TO dfs(node.right)
            SET no_camera TO left_monitored PLUS right_monitored
            SET monitored TO THE MINIMUM OF
                left_camera PLUS THE MINIMUM OF right_monitored AND right_camera,
                right_camera PLUS THE MINIMUM OF left_monitored AND left_camera,
                left_camera PLUS right_camera
            SET camera TO one PLUS THE MINIMUM OF left_no_camera, left_monitored, left_camera PLUS THE MINIMUM OF right_no_camera, right_monitored, right_camera
            RETURN no_camera, monitored, camera
        END FUNCTION
        SET root_no_camera, root_monitored, root_camera TO dfs(root)
        RETURN THE MINIMUM OF root_monitored AND root_camera
    END FUNCTION
END CLASS
