CLASS Solution
    FUNCTION canTransform(start, end)
        SET start_filtered TO an empty list
        FOR each character AND index IN start
            IF character NOT EQUALS the character X
                APPEND the tuple of character AND index TO start_filtered
            END IF
        END FOR
        
        SET end_filtered TO an empty list
        FOR each character AND index IN end
            IF character NOT EQUALS the character X
                APPEND the tuple of character AND index TO end_filtered
            END IF
        END FOR
        
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered
            RETURN False
        END IF
        
        FOR each pair of tuples (character one AND index one) AND (character two AND index two) FROM start_filtered AND end_filtered
            IF character one NOT EQUALS character two
                RETURN False
            END IF
            IF character one EQUALS the character L AND index one LESS THAN index two
                RETURN False
            END IF
            IF character one EQUALS the character R AND index one GREATER THAN index two
                RETURN False
            END IF
        END FOR
        
        RETURN True
    END FUNCTION
END CLASS
