CLASS Solution
    FUNCTION mincostToHireWorkers(quality, wage, k)
        # Calculate the ratio of wage to quality for each worker
        SET workers TO the list that is the SORTING OF the pairs of wage DIVIDED BY quality AND quality FOR each wage and quality PAIRED FROM THE ZIP OF wage AND quality
        
        SET max_heap TO an empty list to represent a max HEAP storing NEGATIVE qualities
        SET total_quality TO zero representing the sum of qualities in the current group
        SET min_cost TO positive infinity representing the minimum cost found so far
        
        FOR each pair of ratio AND q IN workers
            INSERT the NEGATIVE of q INTO max_heap to simulate max HEAP behavior
            INCREMENT total_quality BY q
            
            IF the LENGTH OF max_heap IS GREATER THAN k
                SET total_quality TO total_quality PLUS the REMOVAL of the largest NEGATIVE quality from max_heap by HEAP POP
            END IF
            
            IF the LENGTH OF max_heap IS EQUAL TO k
                SET min_cost TO the MINIMUM OF min_cost AND total_quality MULTIPLIED BY ratio
            END IF
        END FOR
        
        RETURN min_cost
    END FUNCTION
END CLASS
