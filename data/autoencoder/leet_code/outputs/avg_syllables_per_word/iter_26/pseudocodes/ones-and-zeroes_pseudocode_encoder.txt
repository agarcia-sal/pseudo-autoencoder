CLASS Solution
    FUNCTION findMaxForm(strs, m, n)
        SET dp TO a two dimensional list with dimensions m plus one BY n plus one FILLED WITH zero
        
        FOR each string s IN strs
            SET count_0 TO the number of zero characters IN string s
            SET count_1 TO the number of one characters IN string s
            
            FOR index i FROM m DOWN TO count_0
                FOR index j FROM n DOWN TO count_1
                    SET dp element at position i THEN j TO the GREATER OF dp element at position i THEN j AND dp element at position i MINUS count_0 THEN j MINUS count_1 PLUS one
                END FOR
            END FOR
        END FOR
        
        RETURN dp element at position m THEN n
    END FUNCTION
END CLASS
