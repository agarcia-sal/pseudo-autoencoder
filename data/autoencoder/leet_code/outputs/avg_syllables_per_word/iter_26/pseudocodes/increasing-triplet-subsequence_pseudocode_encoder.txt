CLASS Solution
    FUNCTION increasingTriplet(list_of_integers nums)
        SET variable first TO positive infinity
        SET variable second TO positive infinity

        FOR each number IN nums
            IF number LESS THAN OR EQUAL TO first
                SET first TO number
            ELSE IF number LESS THAN OR EQUAL TO second
                SET second TO number
            ELSE
                RETURN true
            END IF
        END FOR

        RETURN false
    END FUNCTION
END CLASS
