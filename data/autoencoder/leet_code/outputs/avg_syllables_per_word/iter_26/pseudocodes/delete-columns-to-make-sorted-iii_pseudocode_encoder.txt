CLASS Solution
    FUNCTION minDeletionSize(strs)
        SET m TO the LENGTH OF strs
        SET n TO the LENGTH OF element at position zero OF strs

        SET dp TO a list of length n WHERE each element is one

        FOR i FROM one TO n MINUS one INCLUSIVE
            FOR j FROM zero TO i MINUS one INCLUSIVE
                IF FOR each k FROM zero TO m MINUS one INCLUSIVE THE element at position k AND j OF strs IS LESS THAN OR EQUAL TO element at position k AND i OF strs HOLDS
                    SET dp at position i TO the GREATER OF dp at position i AND dp at position j PLUS one
                END IF
            END FOR
        END FOR

        RETURN n MINUS the MAXIMUM value IN dp
    END FUNCTION
END CLASS
