CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN none
    END IF
    SET root TO a new TreeNode CREATED WITH element at position zero OF values
    SET i TO one
    SET queue TO a new deque COLLECTION
    APPEND root TO queue
    WHILE queue HOLDS elements
        SET node TO the FIRST element REMOVED FROM queue
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.left TO a new TreeNode CREATED WITH the element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none
            SET node.right TO a new TreeNode CREATED WITH the element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUALS TO the val ATTRIBUTE OF q
        RETURN false
    ELSE
        RETURN is_same_tree(the left ATTRIBUTE OF p, the left ATTRIBUTE OF q) AND is_same_tree(the right ATTRIBUTE OF p, the right ATTRIBUTE OF q)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION longestConsecutive(self, root)
        IF root IS none
            RETURN zero
        END IF

        FUNCTION dfs(node, parent, current_length)
            IF node IS none
                RETURN current_length
            END IF

            IF parent IS NOT none AND the val ATTRIBUTE OF node IS EQUALS TO parent PLUS one
                INCREMENT current_length BY one
            ELSE
                SET current_length TO one
            END IF

            SET left_length TO dfs(the left ATTRIBUTE OF node, the val ATTRIBUTE OF node, current_length)
            SET right_length TO dfs(the right ATTRIBUTE OF node, the val ATTRIBUTE OF node, current_length)

            RETURN the MAXIMUM OF current_length, left_length, right_length
        END FUNCTION

        RETURN dfs(root, none, zero)
    END FUNCTION
END CLASS
