CLASS Solution
    FUNCTION minCostII(costs)
        IF costs EQUALS an empty collection
            RETURN zero
        END IF

        SET n TO the LENGTH OF costs
        SET k TO the LENGTH OF the element at position zero OF costs

        IF k EQUALS one
            IF n EQUALS one
                RETURN the element at position zero AT position zero OF costs
            ELSE
                RETURN positive infinity
            END IF
        END IF

        FOR i FROM one TO n MINUS one INCLUSIVE
            SET min_cost TO positive infinity
            SET min_cost_index TO negative one
            SET second_min_cost TO positive infinity
            
            FOR j FROM zero TO k MINUS one INCLUSIVE
                IF the element at position i MINUS one AT position j OF costs LESS THAN min_cost
                    SET second_min_cost TO min_cost
                    SET min_cost TO the element at position i MINUS one AT position j OF costs
                    SET min_cost_index TO j
                ELSE IF the element at position i MINUS one AT position j OF costs LESS THAN second_min_cost
                    SET second_min_cost TO the element at position i MINUS one AT position j OF costs
                END IF
            END FOR

            FOR j FROM zero TO k MINUS one INCLUSIVE
                IF j EQUALS min_cost_index
                    SET the element at position i AT position j OF costs TO the element at position i AT position j OF costs PLUS second_min_cost
                ELSE
                    SET the element at position i AT position j OF costs TO the element at position i AT position j OF costs PLUS min_cost
                END IF
            END FOR
        END FOR

        SET last_row TO the element at the last position OF costs
        SET minimum_cost TO positive infinity
        FOR each cost IN last_row
            IF cost LESS THAN minimum_cost
                SET minimum_cost TO cost
            END IF
        END FOR

        RETURN minimum_cost
    END FUNCTION
END CLASS
