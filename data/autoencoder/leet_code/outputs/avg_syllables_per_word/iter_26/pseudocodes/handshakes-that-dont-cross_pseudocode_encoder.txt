CLASS Solution
    FUNCTION numberOfWays(numPeople)
        SET MOD TO ten TO THE POWER OF nine PLUS seven
        SET dp TO a list OF zeros WITH length equal TO numPeople PLUS one
        SET element at position zero OF dp TO one
        SET element at position two OF dp TO one

        FOR i FROM four TO numPeople INCLUSIVE IN STEPS OF two
            FOR j FROM zero TO i MINUS one IN STEPS OF two
                SET element at position i OF dp TO element at position i OF dp PLUS element at position j OF dp MULTIPLIED BY element at position i MINUS two MINUS j OF dp
                SET element at position i OF dp TO element at position i OF dp MODULO BY MOD
            END FOR
        END FOR

        RETURN element at position numPeople OF dp
    END FUNCTION
END CLASS
