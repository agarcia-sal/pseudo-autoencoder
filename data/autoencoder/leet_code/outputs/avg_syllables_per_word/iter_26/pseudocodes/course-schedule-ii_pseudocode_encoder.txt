CLASS Solution
    FUNCTION findOrder(numCourses, prerequisites)
        SET adjacency_list TO empty dictionary mapping each course FROM zero TO numCourses MINUS one TO empty list
        SET in_degrees TO list of zeros with length numCourses

        FOR each pair destination, source IN prerequisites
            APPEND destination TO adjacency_list at key source
            INCREMENT in_degrees element at position destination BY one
        END FOR

        SET queue TO a double ended queue containing each course FROM zero TO numCourses MINUS one WHERE the element at corresponding position in in_degrees EQUALS zero
        SET topological_order TO empty list

        WHILE queue HOLDS
            SET current TO REMOVE the element from the front of queue
            APPEND current TO topological_order

            FOR each neighbor IN adjacency_list at key current
                DECREMENT in_degrees element at position neighbor BY one
                IF in_degrees element at position neighbor EQUALS zero THEN
                    APPEND neighbor TO queue
                END IF
            END FOR
        END WHILE

        IF the LENGTH OF topological_order EQUALS numCourses THEN
            RETURN topological_order
        ELSE
            RETURN empty list
        END IF
    END FUNCTION
END CLASS
