CLASS Solution
    FUNCTION boldWords(words, s)
        SET bold TO a list of Boolean FALSE repeated the LENGTH OF s times
        FOR each word IN words
            SET start TO the FIRST POSITION OF word IN s
            WHILE start NOT EQUALS negative one HOLDS
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one INCLUSIVE
                    SET the element at position i OF bold TO TRUE
                END FOR
                SET start TO the FIRST POSITION OF word IN s STARTING FROM start PLUS one
            END WHILE
        END FOR
        
        SET result TO an empty list
        SET i TO zero
        WHILE i LESS THAN the LENGTH OF s HOLDS
            IF the element at position i OF bold IS TRUE
                APPEND the string consisting of the characters less than B greater than TO result
                WHILE i LESS THAN the LENGTH OF s AND the element at position i OF bold IS TRUE HOLDS
                    APPEND the element at position i OF s TO result
                    INCREMENT i BY one
                END WHILE
                APPEND the string consisting of the characters less than slash B greater than TO result
            ELSE
                APPEND the element at position i OF s TO result
                INCREMENT i BY one
            END IF
        END WHILE
        
        RETURN the concatenation OF all elements IN result
    END FUNCTION
END CLASS
