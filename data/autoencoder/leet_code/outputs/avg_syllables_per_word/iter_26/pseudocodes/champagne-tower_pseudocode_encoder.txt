CLASS Solution
    FUNCTION champagneTower(poured, query_row, query_glass)
        SET current_row TO a list containing poured
        
        FOR row FROM one TO query_row INCLUSIVE
            SET next_row TO a list containing zero repeated row plus one times
            
            FOR i FROM zero TO the LENGTH OF current_row MINUS one INCLUSIVE
                IF element at position i OF current_row GREATER THAN one
                    SET excess TO element at position i OF current_row MINUS one DIVIDED BY two
                    INCREMENT element at position i OF next_row BY excess
                    INCREMENT element at position i plus one OF next_row BY excess
                END IF
            END FOR

            ASSIGN current_row TO next_row
        END FOR

        RETURN the MINIMUM OF element at position query_glass OF current_row AND one
    END FUNCTION
END CLASS
