CLASS Solution
    FUNCTION visiblePoints(points, angle, location)
        SET list_of_angles TO an empty list
        SET count_of_extra_points TO zero

        FOR each pair_of_coordinates x and y IN points
            IF x EQUALS the element at position zero OF location AND y EQUALS the element at position one OF location
                INCREMENT count_of_extra_points BY one
                CONTINUE to next iteration
            END IF

            CALCULATE angle_in_radians USING arctangent of the difference between y and the element at position one OF location OVER the difference between x and the element at position zero OF location
            CONVERT angle_in_radians TO degrees AS angle_in_degrees
            APPEND angle_in_degrees TO list_of_angles
        END FOR

        SORT list_of_angles

        CREATE extended_angles BY CONCATENATING list_of_angles WITH A NEW LIST WHERE each element a IN list_of_angles IS INCREMENTED BY three hundred sixty
        ASSIGN extended_angles TO list_of_angles

        SET maximum_visible_points TO zero
        SET left_pointer TO zero

        FOR right_pointer FROM zero TO the LENGTH OF list_of_angles MINUS one INCLUSIVE
            WHILE the element at position right_pointer OF list_of_angles MINUS the element at position left_pointer OF list_of_angles IS GREATER THAN angle HOLDS
                INCREMENT left_pointer BY one
            END WHILE

            SET maximum_visible_points TO the GREATER VALUE BETWEEN maximum_visible_points AND the difference BETWEEN right_pointer AND left_pointer PLUS one
        END FOR

        RETURN the SUM OF maximum_visible_points AND count_of_extra_points
    END FUNCTION
END CLASS
