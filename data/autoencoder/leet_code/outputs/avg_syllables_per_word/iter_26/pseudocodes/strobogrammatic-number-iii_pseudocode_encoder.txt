CLASS Solution
    FUNCTION strobogrammaticInRange(low, high)
        FUNCTION generate_strobogrammatic(n, final_length)
            IF n EQUALS zero THEN
                RETURN a list containing an empty string
            END IF
            IF n EQUALS one THEN
                RETURN a list containing the strings zero, one, and eight
            END IF

            SET middles TO generate_strobogrammatic(n MINUS two, final_length)
            SET result TO an empty list
            FOR each middle IN middles
                IF n NOT EQUALS final_length THEN
                    APPEND the concatenation of string zero, middle, and string zero TO result
                END IF
                APPEND the concatenation of string one, middle, and string one TO result
                APPEND the concatenation of string six, middle, and string nine TO result
                APPEND the concatenation of string eight, middle, and string eight TO result
                APPEND the concatenation of string nine, middle, and string six TO result
            END FOR
            RETURN result
        END FUNCTION

        FUNCTION count_strobogrammatic(low, high)
            SET count TO zero
            FOR length FROM the LENGTH OF low TO the LENGTH OF high INCLUSIVE
                FOR each num IN generate_strobogrammatic(length, length)
                    IF (length EQUALS the LENGTH OF low AND num LESS THAN low) OR (length EQUALS the LENGTH OF high AND num GREATER THAN high) THEN
                        CONTINUE to the next iteration of the loop
                    END IF
                    INCREMENT count BY one
                END FOR
            END FOR
            RETURN count
        END FUNCTION

        RETURN count_strobogrammatic(low, high)
    END FUNCTION
END CLASS
