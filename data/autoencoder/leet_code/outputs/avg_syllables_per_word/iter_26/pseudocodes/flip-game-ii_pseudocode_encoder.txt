CLASS Solution
    FUNCTION canWin(currentState)
        SET memo TO an empty dictionary

        FUNCTION can_win_helper(state)
            IF state IS IN memo THEN
                RETURN the value mapped to state IN memo
            END IF

            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one INCLUSIVE
                IF the substring FROM position index TO position index PLUS one OF state EQUALS two PLUS characters THEN
                    SET new_state TO the substring FROM start TO position index MINUS one OF state CONCATENATED WITH two MINUS characters CONCATENATED WITH the substring FROM position index PLUS two TO the end OF state
                    IF it IS NOT the case THAT can_win_helper(new_state) HOLDS THEN
                        SET the value mapped to state IN memo TO true
                        RETURN true
                    END IF
                END IF
            END FOR

            SET the value mapped to state IN memo TO false
            RETURN false
        END FUNCTION

        RETURN can_win_helper(currentState)
    END FUNCTION
END CLASS
