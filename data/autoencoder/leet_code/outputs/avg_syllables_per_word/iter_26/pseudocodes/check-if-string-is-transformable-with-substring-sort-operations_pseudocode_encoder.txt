CLASS Solution
    FUNCTION isTransformable(parameter s OF TYPE string, parameter t OF TYPE string) RETURNS boolean
        FROM the collection module IMPORT the default dictionary behavior
        FROM the heap queue module IMPORT the functions to pop from and push onto a heap structure
        
        SET variable pos TO a default dictionary of lists
        
        FOR each pair of index i and character c IN the enumeration of s
            PUSH onto the heap associated with the integer conversion of character c in pos the value i
        END FOR
        
        FOR each pair of index i and character c IN the enumeration of t
            SET variable digit TO the integer conversion of character c
            
            IF the list associated with digit in pos IS empty
                RETURN false
            END IF
            
            FOR each smaller FROM zero TO digit MINUS one INCLUSIVE
                IF the list associated with smaller in pos IS NOT empty AND the first element at position zero of the list associated with smaller in pos IS LESS THAN the first element at position zero of the list associated with digit in pos
                    RETURN false
                END IF
            END FOR
            
            POP the smallest element from the heap associated with digit in pos
        END FOR
        
        RETURN true
    END FUNCTION
END CLASS
