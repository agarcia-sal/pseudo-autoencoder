CLASS Solution
    FUNCTION getHint(secret, guess)
        IMPORT FUNCTION default dictionary FROM collections
        
        SET bulls TO zero
        SET cows TO zero
        
        SET secret_count TO a default dictionary INITIALIZED WITH integer
        SET guess_count TO a default dictionary INITIALIZED WITH integer
        
        FOR each pair of character s AND character g IN the ZIP OF secret AND guess
            IF s EQUALS g
                INCREMENT bulls BY one
            ELSE
                INCREMENT the value AT key s IN secret_count BY one
                INCREMENT the value AT key g IN guess_count BY one
            END IF
        END FOR
        
        FOR each digit IN the KEYS OF secret_count
            INCREMENT cows BY the MINIMUM OF the value AT key digit IN secret_count AND the value AT key digit IN guess_count
        END FOR
        
        RETURN the CONCATENATION OF bulls WITH the character A AND cows WITH the character B
    END FUNCTION
END CLASS
