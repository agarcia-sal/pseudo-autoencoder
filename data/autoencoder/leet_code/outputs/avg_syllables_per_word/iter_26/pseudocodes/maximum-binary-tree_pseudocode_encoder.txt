CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values OF TYPE list)
    IF values HOLDS NO ELEMENTS
        RETURN None
    END IF
    SET root TO A NEW TreeNode WITH element at position one OF values
    SET i TO one
    SET queue TO AN EMPTY double ended queue
    APPEND root TO queue
    WHILE queue HOLDS ELEMENTS
        SET node TO THE FIRST ELEMENT REMOVED FROM queue
        IF i IS LESS THAN OR EQUAL TO the LENGTH OF values AND element at position i OF values IS NOT None
            SET node.left TO A NEW TreeNode WITH element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN OR EQUAL TO the LENGTH OF values AND element at position i OF values IS NOT None
            SET node.right TO A NEW TreeNode WITH element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN THE RESULT OF is_same_tree WITH p.left AND q.left AND THE RESULT OF is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION constructMaximumBinaryTree(self, nums OF TYPE List of integers)
        IF nums HOLDS NO ELEMENTS
            RETURN None
        END IF

        SET max_val TO THE MAXIMUM VALUE OF nums
        SET max_index TO THE POSITION OF max_val IN nums

        SET root TO A NEW TreeNode WITH max_val

        SET root.left TO THE RESULT OF self.constructMaximumBinaryTree WITH THE SUBLIST OF nums FROM THE FIRST POSITION TO THE POSITION BEFORE max_index
        SET root.right TO THE RESULT OF self.constructMaximumBinaryTree WITH THE SUBLIST OF nums FROM THE POSITION AFTER max_index TO THE LAST POSITION

        RETURN root
    END FUNCTION
END CLASS
