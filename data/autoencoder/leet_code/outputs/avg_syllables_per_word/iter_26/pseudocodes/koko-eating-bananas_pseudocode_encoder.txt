CLASS Solution
    FUNCTION minEatingSpeed(piles, h)
        FUNCTION canFinish(k)
            SET hours TO zero
            FOR each pile IN piles
                INCREMENT hours BY the ceiling of pile DIVIDED BY k
            END FOR
            RETURN hours LESS THAN OR EQUAL TO h
        END FUNCTION

        SET left TO one
        SET right TO the maximum element OF piles

        WHILE left LESS THAN OR EQUAL TO right HOLDS
            SET mid TO the integer division of left PLUS right BY two
            IF canFinish(mid) HOLDS
                SET right TO mid MINUS one
            ELSE
                SET left TO mid PLUS one
            END IF
        END WHILE

        RETURN left
    END FUNCTION
END CLASS
