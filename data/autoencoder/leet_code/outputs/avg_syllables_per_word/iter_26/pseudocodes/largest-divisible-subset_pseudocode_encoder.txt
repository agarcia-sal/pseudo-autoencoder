CLASS Solution
    FUNCTION largestDivisibleSubset(nums)
        PERFORM sorting IN PLACE on nums
        SET n TO the LENGTH OF nums
        SET f TO a list of the number one REPEATED n TIMES
        SET k TO zero
        FOR i FROM zero TO n MINUS one INCLUSIVE
            FOR j FROM zero TO i MINUS one INCLUSIVE
                IF the element at position i OF nums MODULO element at position j OF nums EQUALS zero
                    SET element at position i OF f TO the GREATER OF element at position i OF f AND element at position j OF f PLUS one
                END IF
            END FOR
            IF element at position k OF f LESS THAN element at position i OF f
                SET k TO i
            END IF
        END FOR
        SET m TO element at position k OF f
        SET i TO k
        SET ans TO an empty list
        WHILE m HOLDS
            IF element at position k OF nums MODULO element at position i OF nums EQUALS zero AND element at position i OF f EQUALS m
                APPEND element at position i OF nums TO ans
                SET k TO i
                DECREMENT m BY one
            END IF
            DECREMENT i BY one
        END WHILE
        RETURN ans
    END FUNCTION
END CLASS
