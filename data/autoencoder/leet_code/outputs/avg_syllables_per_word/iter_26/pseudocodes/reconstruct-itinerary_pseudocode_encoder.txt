CLASS Solution
    FUNCTION findItinerary(tickets AS list of lists of strings) RETURNS list of strings
        SET graph TO a mapping from keys to empty lists by default
        FOR each tuple formed by airport_from AND airport_to IN the reverse order of the sorted list of tickets
            APPEND airport_to TO the list at key airport_from IN graph
        END FOR
        SET itinerary TO an empty list

        FUNCTION visit(airport AS string)
            WHILE the list at key airport IN graph HOLDS elements
                CALL visit WITH the last element REMOVED FROM the list at key airport IN graph
            END WHILE
            APPEND airport TO itinerary
        END FUNCTION

        CALL visit WITH the string representing the airport code J F K
        RETURN the reverse order of itinerary
    END FUNCTION
END CLASS
