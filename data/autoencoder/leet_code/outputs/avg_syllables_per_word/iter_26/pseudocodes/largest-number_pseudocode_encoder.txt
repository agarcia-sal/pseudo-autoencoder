CLASS Solution
    FUNCTION largestNumber(nums)
        SET nums_str TO the LIST OBTAINED BY APPLYING THE STRING CONVERSION TO each element IN nums

        FUNCTION compare(x, y)
            IF the CONCATENATION OF x AND y IS GREATER THAN the CONCATENATION OF y AND x
                RETURN negative one
            ELSE IF the CONCATENATION OF x AND y IS LESS THAN the CONCATENATION OF y AND x
                RETURN positive one
            ELSE
                RETURN zero
            END IF
        END FUNCTION

        CALL the SORT METHOD OF nums_str WITH THE KEY FUNCTION OF THE RESULT OF cmp_to_key APPLIED TO compare

        IF the ELEMENT at position zero OF nums_str EQUALS the STRING representation OF zero
            RETURN the STRING representation OF zero
        END IF

        RETURN the STRING OBTAINED BY JOINING all ELEMENTS OF nums_str WITH NO SEPARATOR
    END FUNCTION
END CLASS
