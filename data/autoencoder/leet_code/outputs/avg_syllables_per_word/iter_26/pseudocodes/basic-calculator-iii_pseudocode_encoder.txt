CLASS Solution
    FUNCTION calculate(s)
        FUNCTION helper(list_of_characters)
            SET stack TO an empty list
            SET number TO zero
            SET sign TO the character plus
            WHILE the LENGTH OF list_of_characters IS GREATER THAN zero HOLDS
                SET character TO the FIRST ELEMENT OF list_of_characters
                REMOVE the FIRST ELEMENT FROM list_of_characters
                IF character IS A DIGIT THEN
                    SET number TO number MULTIPLIED BY ten PLUS the integer VALUE OF character
                END IF
                IF character EQUALS the character opening parenthesis THEN
                    SET number TO the RESULT OF helper WITH list_of_characters AS ARGUMENT
                END IF
                IF character IS one OF the characters plus MINUS MULTIPLY DIVIDE opening parenthesis closing parenthesis OR the LENGTH OF list_of_characters EQUALS zero THEN
                    IF sign EQUALS the character plus THEN
                        APPEND number TO stack
                    ELSE IF sign EQUALS the character minus THEN
                        APPEND the NEGATIVE OF number TO stack
                    ELSE IF sign EQUALS the character multiplied BY THEN
                        SET the ELEMENT at the LAST POSITION OF stack TO the ELEMENT at the LAST POSITION OF stack MULTIPLIED BY number
                    ELSE IF sign EQUALS the character divided BY THEN
                        SET the ELEMENT at the LAST POSITION OF stack TO the INTEGER PART OF the ELEMENT at the LAST POSITION OF stack DIVIDED BY number with truncation towards zero
                    END IF
                    SET sign TO character
                    SET number TO zero
                END IF
                IF character EQUALS the character closing parenthesis THEN
                    BREAK OUT OF the WHILE LOOP
                END IF
            END WHILE
            RETURN the SUM OF all ELEMENTS IN stack
        END FUNCTION

        SET s TO the CONVERSION OF s INTO a LIST OF CHARACTERS
        RETURN the RESULT OF helper WITH s AS ARGUMENT
    END FUNCTION
END CLASS
