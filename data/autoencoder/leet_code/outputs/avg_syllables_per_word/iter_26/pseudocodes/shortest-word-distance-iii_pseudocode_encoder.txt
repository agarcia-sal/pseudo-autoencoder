CLASS Solution
    FUNCTION shortestWordDistance(wordsDict, word1, word2)
        SET min_distance TO infinity
        SET prev_word1 TO negative one
        SET prev_word2 TO negative one
        SET same_word TO the RESULT OF word1 EQUALS word2
        FOR each index and word IN the ENUMERATE OF wordsDict
            IF word EQUALS word1
                IF same_word HOLDS
                    SET prev_word2 TO prev_word1
                END IF
                SET prev_word1 TO index
            END IF
            IF word EQUALS word2
                IF same_word DOES NOT HOLD
                    SET prev_word2 TO index
                END IF
            END IF
            IF prev_word1 DOES NOT EQUAL negative one AND prev_word2 DOES NOT EQUAL negative one
                SET min_distance TO the MINIMUM OF min_distance AND the ABSOLUTE VALUE OF prev_word1 MINUS prev_word2
            END IF
        END FOR
        RETURN min_distance
    END FUNCTION
END CLASS
