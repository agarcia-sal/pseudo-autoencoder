CLASS Solution
    FUNCTION nthMagicalNumber(n, a, b)
        SET MOD TO ten raised to the power of nine PLUS one
        
        FUNCTION gcd(x, y)
            WHILE y HOLDS and y NOT EQUALS zero
                SET temporary_x TO x
                SET temporary_y TO y
                SET x TO temporary_y
                SET y TO the REMAINDER WHEN temporary_x IS DIVIDED BY temporary_y
            END WHILE
            RETURN x
        END FUNCTION
        
        FUNCTION lcm(x, y)
            RETURN x MULTIPLIED BY y DIVIDED BY gcd(x, y)
        END FUNCTION
        
        SET L TO lcm(a, b)
        
        SET left TO one
        SET right TO n MULTIPLIED BY the MINIMUM OF a AND b
        
        WHILE left LESS THAN right
            SET mid TO the INTEGER DIVISION RESULT OF left PLUS right DIVIDED BY two
            SET count TO the INTEGER DIVISION RESULT OF mid DIVIDED BY a PLUS the INTEGER DIVISION RESULT OF mid DIVIDED BY b MINUS the INTEGER DIVISION RESULT OF mid DIVIDED BY L
            
            IF count LESS THAN n
                SET left TO mid PLUS one
            ELSE
                SET right TO mid
            END IF
        END WHILE
        
        RETURN the REMAINDER WHEN left IS DIVIDED BY MOD
    END FUNCTION
END CLASS
