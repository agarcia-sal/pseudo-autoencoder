CLASS Solution
    FUNCTION grayCode(parameter n)
        IF n EQUALS zero
            RETURN a list containing zero
        END IF
        SET prev_gray TO the result of calling grayCode with parameter n MINUS one
        SET mask TO one BIT LEFT SHIFTED BY n MINUS one
        SET current_gray TO the concatenation of prev_gray AND a list constructed by
            FOR each number IN the reverse of prev_gray
                compute mask OR number
            END FOR
        RETURN current_gray
    END FUNCTION
END CLASS
