CLASS Solution
    FUNCTION diffWaysToCompute(expression)
        FUNCTION compute(left_list, right_list, operator_symbol)
            SET results TO empty list
            FOR each left_element IN left_list
                FOR each right_element IN right_list
                    IF operator_symbol EQUALS the PLUS symbol
                        APPEND left_element PLUS right_element TO results
                    ELSE IF operator_symbol EQUALS the MINUS symbol
                        APPEND left_element MINUS right_element TO results
                    ELSE IF operator_symbol EQUALS the MULTIPLICATION symbol
                        APPEND left_element MULTIPLIED BY right_element TO results
                    END IF
                END FOR
            END FOR
            RETURN results
        END FUNCTION

        FUNCTION helper(sub_expression)
            IF the entirety of sub_expression CONTAINS ONLY digit characters
                RETURN a new list containing the integer value OF sub_expression
            END IF

            SET results TO empty list
            FOR index FROM zero TO the LENGTH OF sub_expression MINUS one INCLUSIVE
                IF the element at position index OF sub_expression IS one of the symbols PLUS OR MINUS OR MULTIPLICATION
                    SET left_results TO the value returned BY helper WITH the substring FROM position zero TO position index MINUS one OF sub_expression
                    SET right_results TO the value returned BY helper WITH the substring FROM position index PLUS one TO the last position OF sub_expression
                    EXTEND results BY the value returned BY compute WITH left_results, right_results, and the element at position index OF sub_expression
                END IF
            END FOR
            RETURN results
        END FUNCTION

        RETURN the value returned BY helper WITH expression
    END FUNCTION
END CLASS
