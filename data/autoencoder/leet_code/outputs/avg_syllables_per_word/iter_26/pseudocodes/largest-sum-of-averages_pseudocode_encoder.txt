CLASS Solution
    FUNCTION largestSumOfAverages(nums, k)
        SET n TO the LENGTH OF nums
        
        SET prefix_sum TO a list of zeros with LENGTH OF n PLUS one
        FOR index i FROM zero TO n MINUS one INCLUSIVE
            SET element at position i PLUS one OF prefix_sum TO element at position i OF prefix_sum PLUS element at position i OF nums
        END FOR
        
        FUNCTION average(i, j)
            RETURN element at position j PLUS one OF prefix_sum MINUS element at position i OF prefix_sum DIVIDED BY j MINUS i PLUS one
        END FUNCTION
        
        SET dp TO a two dimensional list of zeros with LENGTH OF n PLUS one by k PLUS one
        
        FOR index i FROM zero TO n MINUS one INCLUSIVE
            SET element at position i AT position one OF dp TO average(i, n MINUS one)
        END FOR
        
        FOR index i FROM n MINUS one DOWN TO zero INCLUSIVE
            FOR variable partitions FROM two TO k INCLUSIVE
                FOR index j FROM i PLUS one TO n MINUS one INCLUSIVE
                    SET element at position i AT position partitions OF dp TO the GREATER OF element at position i AT position partitions OF dp AND element at position j AT position partitions MINUS one OF dp PLUS average(i, j MINUS one)
                END FOR
            END FOR
        END FOR
        
        RETURN element at position zero AT position k OF dp
    END FUNCTION
END CLASS
