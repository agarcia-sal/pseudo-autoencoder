CLASS Solution
    FUNCTION makesquare(matchsticks)
        SET total_length TO the SUM OF matchsticks
        IF the MODULO OF total_length BY four NOT EQUALS zero
            RETURN false
        END IF

        SET side_length TO total_length DIVIDED BY four
        SORT matchsticks IN descending order

        FUNCTION can_form_sides(sides, index)
            IF index EQUALS the LENGTH OF matchsticks
                RETURN whether ALL elements IN sides ARE EQUAL TO side_length
            END IF

            FOR i FROM zero TO three INCLUSIVE
                IF the ELEMENT AT position i OF sides PLUS the ELEMENT AT position index OF matchsticks IS LESS THAN OR EQUAL TO side_length
                    INCREMENT the ELEMENT AT position i OF sides BY the ELEMENT AT position index OF matchsticks
                    IF can_form_sides(sides, index PLUS one) HOLDS
                        RETURN true
                    END IF
                    DECREMENT the ELEMENT AT position i OF sides BY the ELEMENT AT position index OF matchsticks
                END IF

                IF the ELEMENT AT position i OF sides EQUALS zero
                    BREAK the FOR loop
                END IF
            END FOR

            RETURN false
        END FUNCTION

        RETURN can_form_sides(a LIST OF four zeros, zero)
    END FUNCTION
END CLASS
