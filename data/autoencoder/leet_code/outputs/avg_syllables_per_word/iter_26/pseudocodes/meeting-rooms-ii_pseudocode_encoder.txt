CLASS Solution
    FUNCTION minMeetingRooms(intervals AS List OF List OF integers)
        IF intervals IS EMPTY
            RETURN zero
        END IF

        PERFORM sorting OF intervals BY the element AT position zero OF each interval IN ASCENDING ORDER

        SET min_heap TO an empty list representing a minimum heap structure

        FOR each meeting IN intervals
            IF min_heap IS NOT EMPTY AND the element AT position zero OF meeting IS GREATER THAN OR EQUAL TO the element AT position zero OF min_heap
                REMOVE the smallest element FROM min_heap
            END IF
            ADD the element AT position one OF meeting TO min_heap USING the heap push operation
        END FOR

        RETURN the LENGTH OF min_heap
    END FUNCTION
END CLASS
