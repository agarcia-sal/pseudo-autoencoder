CLASS Solution
    FUNCTION customSortString(order, s)
        CREATE a DICTIONARY named d TO STORE each character IN order MAPPED TO its POSITION index IN order
        CREATE a LIST named sorted_characters BY SORTING the characters IN s BY ACCESSING the POSITION index USING d WITH DEFAULT VALUE zero IF character NOT FOUND IN d
        CREATE a STRING named result BY JOINING all characters IN sorted_characters IN SEQUENCE
        RETURN result
    END FUNCTION
END CLASS
