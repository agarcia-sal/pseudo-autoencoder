CLASS Solution
    FUNCTION numberOfGoodSubsets(nums)
        SET count TO the FREQUENCY COUNTER OF elements IN nums
        SET dp TO a list of zero elements WITH LENGTH the number represented by one SHIFTED LEFT by ten positions
        SET the element at position zero OF dp TO one
        
        FOR each num IN count
            IF num EQUALS one
                SET the element at position zero OF dp TO the element at position zero OF dp MULTIPLIED BY the number represented by one SHIFTED LEFT by the value of the element at position corresponding to one IN count
                CONTINUE TO the next iteration
            END IF
            
            IF num NOT IN prime_factors
                CONTINUE TO the next iteration
            END IF
            
            SET factors TO the list OF prime_factors AT key num
            SET mask TO zero
            FOR each prime IN factors
                SET mask TO mask BITWISE OR the number represented by one SHIFTED LEFT by the value of prime MINUS one
            END FOR
            
            FOR m FROM the number represented by one SHIFTED LEFT by ten MINUS one DOWN TO zero INCLUSIVE
                IF THE result OF bitwise AND BETWEEN m AND mask EQUALS zero
                    SET the element at position the bitwise OR BETWEEN m AND mask OF dp TO the sum OF the element at position the bitwise OR BETWEEN m AND mask OF dp AND the element at position m OF dp MULTIPLIED BY the value at position num OF count MODULO MOD
                END IF
            END FOR
        END FOR
        
        SET result TO the sum OF elements FROM position one TO the end OF dp MODULO MOD
        RETURN result
    END FUNCTION
END CLASS
