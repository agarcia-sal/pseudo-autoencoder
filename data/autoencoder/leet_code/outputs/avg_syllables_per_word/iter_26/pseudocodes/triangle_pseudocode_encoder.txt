CLASS Solution
    FUNCTION minimumTotal(triangle)
        FOR row FROM the LENGTH OF triangle MINUS two TO zero INCLUSIVE DECREASING BY one
            FOR col FROM zero TO the LENGTH OF the element at position row OF triangle MINUS one INCLUSIVE
                SET element at position col OF element at position row OF triangle TO 
                    the element at position col OF element at position row OF triangle PLUS 
                    the MINIMUM OF the element at position col OF element at position row plus one OF triangle AND 
                    the element at position col plus one OF element at position row plus one OF triangle
            END FOR
        END FOR
        RETURN the element at position zero OF element at position zero OF triangle
    END FUNCTION
END CLASS
