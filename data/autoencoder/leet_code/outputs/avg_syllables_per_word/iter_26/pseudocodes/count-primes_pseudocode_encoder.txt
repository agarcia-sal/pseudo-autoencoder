CLASS Solution
    FUNCTION countPrimes(n)
        IF n LESS THAN OR EQUAL TO two THEN
            RETURN zero
        END IF
        SET is_prime TO a list of boolean true repeated n times
        SET element at position zero OF is_prime TO boolean false
        SET element at position one OF is_prime TO boolean false
        FOR start FROM two TO the integer part of the square root of n PLUS one EXCLUSIVE
            IF element at position start OF is_prime IS boolean true THEN
                FOR multiple FROM start MULTIPLIED BY start TO n EXCLUSIVE STEPPING BY start
                    SET element at position multiple OF is_prime TO boolean false
                END FOR
            END IF
        END FOR
        RETURN the sum of all boolean true values IN is_prime
    END FUNCTION
END CLASS
