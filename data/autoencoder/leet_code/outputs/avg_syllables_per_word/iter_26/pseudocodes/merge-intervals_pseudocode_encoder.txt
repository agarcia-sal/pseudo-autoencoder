CLASS Solution
    FUNCTION merge(intervals AS list OF list OF integer) AS list OF list OF integer
        CALL the METHOD sort ON intervals WITH a KEY THAT TAKES an element NAMED x AND RETURNS the ELEMENT AT position zero OF x
        SET merged_list TO an empty list
        FOR each interval IN intervals
            IF merged_list IS empty OR the ELEMENT AT position one OF the LAST ELEMENT OF merged_list IS LESS THAN the ELEMENT AT position zero OF interval
                APPEND interval TO merged_list
            ELSE
                SET the ELEMENT AT position one OF the LAST ELEMENT OF merged_list TO the GREATER OF the ELEMENT AT position one OF the LAST ELEMENT OF merged_list AND the ELEMENT AT position one OF interval
            END IF
        END FOR
        RETURN merged_list
    END FUNCTION
END CLASS
