CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN none
    END IF
    SET root TO A NEW TreeNode CREATED WITH element at position zero OF values
    SET i TO one
    SET queue TO A NEW deque OBJECT
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO THE FIRST element REMOVED FROM queue
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none
            SET node.left TO A NEW TreeNode CREATED WITH element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none
            SET node.right TO A NEW TreeNode CREATED WITH element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN false
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION findClosestLeaf(self, root, k)
        IF root.left IS none AND root.right IS none
            RETURN root.val
        END IF

        SET graph TO A NEW default dictionary OF list

        FUNCTION dfs(node, parent EQUALS none)
            IF parent IS NOT none
                APPEND parent.val TO graph[element node.val]
                APPEND node.val TO graph[element parent.val]
            END IF
            IF node.left IS NOT none
                CALL dfs(node.left, node)
            END IF
            IF node.right IS NOT none
                CALL dfs(node.right, node)
            END IF
        END FUNCTION

        CALL dfs(root)

        SET queue TO A NEW deque WITH element k INSIDE
        SET visited TO A NEW empty set

        WHILE queue IS NOT empty
            SET current TO THE FIRST element REMOVED FROM queue
            ADD current TO visited

            IF THE LENGTH OF graph[element current] IS EQUAL TO one AND current IS NOT EQUAL TO root.val
                RETURN current
            END IF

            FOR each neighbor IN graph[element current]
                IF neighbor IS NOT IN visited
                    APPEND neighbor TO queue
                END IF
            END FOR
        END WHILE

        RETURN minus one
    END FUNCTION
END CLASS
