CLASS Solution
    FUNCTION wallsAndGates(rooms)
        IF rooms EQUALS empty OR the LENGTH OF element at position zero OF rooms EQUALS zero
            RETURN
        END IF

        SET rows TO the LENGTH OF rooms
        SET cols TO the LENGTH OF element at position zero OF rooms
        ASSIGN reference TO an empty double ended queue NAMED queue

        FOR each r FROM zero TO rows MINUS one INCLUSIVE
            FOR each c FROM zero TO cols MINUS one INCLUSIVE
                IF element at position r OF rooms AT position c EQUALS zero
                    APPEND the tuple consisting of r and c TO queue
                END IF
            END FOR
        END FOR

        ASSIGN reference TO the list of tuples consisting of zero and one, one and zero, zero and negative one, and negative one and zero NAMED directions

        WHILE queue HOLDS elements
            SET r, c TO the first element REMOVED FROM the left OF queue
            FOR each dr, dc IN directions
                SET nr TO r PLUS dr
                SET nc TO c PLUS dc
                IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN rows AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN cols AND element at position nr OF rooms AT position nc EQUALS two BILLION one hundred forty-seven MILLION four hundred eighty-three thousand six hundred forty seven
                    SET element at position nr OF rooms AT position nc TO element at position r OF rooms AT position c PLUS one
                    APPEND the tuple consisting of nr and nc TO queue
                END IF
            END FOR
        END WHILE
    END FUNCTION
END CLASS
