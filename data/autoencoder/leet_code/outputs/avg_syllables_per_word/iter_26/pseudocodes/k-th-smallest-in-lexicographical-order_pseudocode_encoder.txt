CLASS Solution
    FUNCTION findKthNumber(parameter n, parameter k)
        FUNCTION count_steps(parameter prefix, parameter n)
            SET next_prefix TO prefix PLUS one
            SET steps TO zero
            WHILE prefix LESS THAN OR EQUAL TO n HOLDS
                SET difference_one TO n MINUS prefix PLUS one
                SET difference_two TO next_prefix MINUS prefix
                SET minimal_difference TO the MINIMUM OF difference_one AND difference_two
                INCREMENT steps BY minimal_difference
                SET prefix TO prefix MULTIPLIED BY ten
                SET next_prefix TO next_prefix MULTIPLIED BY ten
            END WHILE
            RETURN steps
        END FUNCTION

        SET current TO one
        DECREMENT k BY one
        WHILE k GREATER THAN zero HOLDS
            SET steps TO count_steps WITH arguments current AND n
            IF steps LESS THAN OR EQUAL TO k THEN
                INCREMENT current BY one
                DECREMENT k BY steps
            ELSE
                SET current TO current MULTIPLIED BY ten
                DECREMENT k BY one
            END IF
        END WHILE
        RETURN current
    END FUNCTION
END CLASS
