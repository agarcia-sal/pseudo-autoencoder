CLASS Solution
    FUNCTION numPoints(darts, r)
        FUNCTION distance(p1, p2)
            RETURN the SQUARE ROOT OF the SUM OF the SQUARE of the DIFFERENCE BETWEEN element at position zero OF p1 AND element at position zero OF p2 AND the SQUARE of the DIFFERENCE BETWEEN element at position one OF p1 AND element at position one OF p2
        END FUNCTION

        FUNCTION circle_center(p1, p2, r)
            SET d TO CALL distance WITH p1 AND p2
            IF d GREATER THAN TWO MULTIPLIED BY r
                RETURN nothing
            END IF
            SET a TO the SQUARE ROOT OF r MULTIPLIED BY r MINUS the MULTIPLICATION of the DIVISION OF d BY two BY itself
            SET h TO list CONTAINING the DIVISION OF the SUM OF element at position zero OF p1 AND element at position zero OF p2 BY two AND the DIVISION OF the SUM OF element at position one OF p1 AND element at position one OF p2 BY two
            SET dx TO the MULTIPLICATION of the DIFFERENCE BETWEEN element at position one OF p2 AND element at position one OF p1 BY a DIVIDED BY d
            SET dy TO the MULTIPLICATION of the DIFFERENCE BETWEEN element at position zero OF p2 AND element at position zero OF p1 BY a DIVIDED BY d
            RETURN list CONTAINING a TUPLE OF the DIFFERENCE BETWEEN element at position zero OF h AND dx AND the SUM OF element at position one OF h AND dy AND a TUPLE OF the SUM OF element at position zero OF h AND dx AND the DIFFERENCE BETWEEN element at position one OF h AND dy
        END FUNCTION

        FUNCTION is_inside(circle, point, r)
            RETURN CALL distance WITH circle AND point LESS THAN OR EQUAL TO r
        END FUNCTION

        SET n TO the LENGTH OF darts
        IF n EQUALS one
            RETURN one
        END IF

        SET max_count TO one

        FOR i FROM zero TO n MINUS one INCLUSIVE
            FOR j FROM i PLUS one TO n MINUS one INCLUSIVE
                SET centers TO CALL circle_center WITH element at position i OF darts AND element at position j OF darts AND r
                IF centers IS NOT nothing
                    FOR each center IN centers
                        SET count TO zero
                        FOR each dart IN darts
                            IF CALL is_inside WITH center AND dart AND r HOLDS
                                INCREMENT count BY one
                            END IF
                        END FOR
                        IF max_count LESS THAN count
                            SET max_count TO count
                        END IF
                    END FOR
                END IF
            END FOR
        END FOR

        RETURN max_count
    END FUNCTION
END CLASS
