CLASS Solution
    FUNCTION decodeAtIndex(input_string, target_index)
        SET decoded_length TO zero
        
        FOR each character IN input_string
            IF character IS A DIGIT
                SET decoded_length TO decoded_length MULTIPLIED BY the NUMERIC VALUE OF character
            ELSE
                INCREMENT decoded_length BY one
            END IF
        END FOR
        
        FOR each character IN the REVERSE ORDER OF input_string
            SET target_index TO the REMAINDER OF target_index DIVIDED BY decoded_length
            
            IF target_index EQUALS zero AND character IS AN ALPHABET LETTER
                RETURN character
            END IF
            
            IF character IS A DIGIT
                SET decoded_length TO decoded_length DIVIDED BY the NUMERIC VALUE OF character
            ELSE
                DECREMENT decoded_length BY one
            END IF
        END FOR
    END FUNCTION
END CLASS
