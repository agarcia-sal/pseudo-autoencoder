CLASS Solution
    FUNCTION findMaximumXOR(nums)
        SET max_xor TO zero

        FOR i FROM the integer THIRTY ONE DOWN TO zero INCLUSIVE
            SET max_xor TO max_xor MULTIPLIED BY two
            SET current_candidate TO max_xor PLUS one

            CREATE an empty set CALLED prefixes
            FOR each num IN nums
                SET prefix TO num DIVIDED BY two TO THE POWER OF i DISCARDED THE FRACTIONAL PART
                ADD prefix TO prefixes
            END FOR

            FOR each prefix IN prefixes
                IF the logical XOR of current_candidate AND prefix EXISTS IN prefixes THEN
                    SET max_xor TO current_candidate
                    BREAK
                END IF
            END FOR
        END FOR

        RETURN max_xor
    END FUNCTION
END CLASS
