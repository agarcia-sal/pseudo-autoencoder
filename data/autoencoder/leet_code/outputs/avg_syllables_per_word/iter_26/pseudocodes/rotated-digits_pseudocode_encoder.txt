CLASS Solution
    FUNCTION rotatedDigits(n)
        SET change_digits TO the set containing the characters two, five, six, and nine
        SET same_digits TO the set containing the characters zero, one, and eight

        FUNCTION is_good_number(num)
            SET num_str TO the string representation OF num
            IF every digit IN num_str IS IN the UNION of change_digits AND same_digits
                IF there EXISTS at least one digit IN num_str THAT IS IN change_digits
                    RETURN true
                END IF
                RETURN false
            ELSE
                RETURN false
            END IF
        END FUNCTION

        SET good_count TO the sum of is_good_number applied TO each integer i FROM one TO n INCLUSIVE
        RETURN good_count
    END FUNCTION
END CLASS
