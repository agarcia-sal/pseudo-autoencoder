CLASS Solution
    FUNCTION totalHammingDistance(list_of_numbers)
        SET total_distance TO zero
        FOR each bit_position FROM zero TO thirty one INCLUSIVE
            SET count_ones TO zero
            FOR each number IN list_of_numbers
                INCREMENT count_ones BY the value of the bit at position bit_position OF number
            END FOR
            SET count_zeros TO the LENGTH OF list_of_numbers MINUS count_ones
            INCREMENT total_distance BY count_ones MULTIPLIED BY count_zeros
        END FOR
        RETURN total_distance
    END FUNCTION
END CLASS
