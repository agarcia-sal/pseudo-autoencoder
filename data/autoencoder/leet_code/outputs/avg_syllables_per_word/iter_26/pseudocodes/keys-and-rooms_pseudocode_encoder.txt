CLASS Solution
    FUNCTION canVisitAllRooms(rooms)
        SET visited TO empty collection representing a set
        SET stack TO collection containing only the element zero

        WHILE the LENGTH OF stack GREATER THAN zero HOLDS
            SET current_room TO the element obtained by removing the last element FROM stack
            ADD current_room TO visited
            FOR each key IN the collection at position current_room OF rooms
                IF key NOT IN visited THEN
                    APPEND key TO stack
                END IF
            END FOR
        END WHILE

        RETURN the LENGTH OF visited EQUALS the LENGTH OF rooms
    END FUNCTION
END CLASS
