CLASS Solution
    FUNCTION findNthDigit(parameter n)
        SET length TO one
        SET count TO nine
        SET start TO one
        
        WHILE n GREATER THAN length MULTIPLIED BY count HOLDS
            DECREMENT n BY length MULTIPLIED BY count
            INCREMENT length BY one
            MULTIPLY count BY ten AND ASSIGN TO count
            MULTIPLY start BY ten AND ASSIGN TO start
        END WHILE
        
        INCREMENT start BY the INTEGER RESULT OF the SUBTRACTION of one FROM n DIVIDED BY length
        
        RETURN the integer CONVERTED FROM the CHARACTER at the INTEGER RESULT OF the MODULO of the SUBTRACTION of one FROM n BY length IN THE STRING CONVERTED FROM start
    END FUNCTION
END CLASS
