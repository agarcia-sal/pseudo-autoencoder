CLASS Solution
    FUNCTION findMaximizedCapital(k, w, profits, capital)
        SET projects TO the LIST OF TUPLES FOR each PAIR formed BY COUPLING element AT the SAME POSITION OF capital AND profits
        SORT projects BY the FIRST element OF each TUPLE IN ascending ORDER

        SET max_heap TO an EMPTY LIST

        SET i TO zero
        SET n TO the LENGTH OF projects

        FOR each iteration FROM one TO k INCLUSIVE
            WHILE i LESS THAN n AND the FIRST element OF the TUPLE AT position i OF projects LESS THAN OR EQUAL TO w HOLDS
                PUSH the NEGATIVE of the SECOND element OF the TUPLE AT position i OF projects TO max_heap USING a HEAP PUSH operation
                INCREMENT i BY one
            END WHILE

            IF max_heap IS empty THEN
                BREAK the FOR loop
            END IF

            SET w TO w PLUS the NEGATIVE of the HEAP POP from max_heap
        END FOR

        RETURN w
    END FUNCTION
END CLASS
