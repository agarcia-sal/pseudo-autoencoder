CLASS Solution
    FUNCTION findSubstringInWraproundString(string_s)
        SET max_length_dictionary TO an empty dictionary WITH keys AS each character IN the sequence of lowercase alphabets FROM a TO z AND values AS zero
        
        SET current_length TO zero
        
        FOR index_i FROM zero TO the LENGTH OF string_s MINUS one INCLUSIVE
            IF index_i GREATER THAN zero AND 
               (the UNICODE code of the element at position index_i OF string_s MINUS the UNICODE code of the element at position index_i MINUS one OF string_s EQUALS one OR
                the element at position index_i OF string_s EQUALS character a AND the element at position index_i MINUS one OF string_s EQUALS character z)
                INCREMENT current_length BY one
            ELSE
                SET current_length TO one
            END IF
            
            SET key_character TO the element at position index_i OF string_s
            SET current_maximum TO the element in max_length_dictionary AT key_character
            IF current_length GREATER THAN current_maximum
                SET the element in max_length_dictionary AT key_character TO current_length
            END IF
        END FOR
        
        SET total_unique_substrings TO zero
        FOR each value_length IN the VALUES OF max_length_dictionary
            INCREMENT total_unique_substrings BY value_length
        END FOR
        
        RETURN total_unique_substrings
    END FUNCTION
END CLASS
