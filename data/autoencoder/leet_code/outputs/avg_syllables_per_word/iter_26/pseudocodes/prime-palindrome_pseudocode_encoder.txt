CLASS Solution
    FUNCTION primePalindrome(n)
        FUNCTION is_prime(num)
            IF num LESS THAN two
                RETURN False
            END IF
            IF num EQUALS two OR num EQUALS three
                RETURN True
            END IF
            IF num MODULO two EQUALS zero OR num MODULO three EQUALS zero
                RETURN False
            END IF
            SET i TO five
            WHILE i MULTIPLIED BY i LESS THAN OR EQUAL TO num HOLDS
                IF num MODULO i EQUALS zero OR num MODULO i PLUS two EQUALS zero
                    RETURN False
                END IF
                INCREMENT i BY six
            END WHILE
            RETURN True
        END FUNCTION

        FUNCTION is_palindrome(num)
            RETURN the STRING REPRESENTATION OF num EQUALS the STRING REPRESENTATION OF num REVERSED
        END FUNCTION

        IF n LESS THAN OR EQUAL TO two
            RETURN two
        END IF

        IF n GREATER THAN OR EQUAL TO ten AND n LESS THAN OR EQUAL TO eleven
            RETURN eleven
        END IF

        WHILE True HOLDS
            IF n MODULO two EQUALS zero
                INCREMENT n BY one
                CONTINUE
            END IF
            IF is_palindrome(n) AND is_prime(n)
                RETURN n
            END IF
            IF the LENGTH OF the STRING REPRESENTATION OF n MODULO two EQUALS zero
                SET n TO ten RAISED TO THE POWER OF the LENGTH OF the STRING REPRESENTATION OF n PLUS one
            ELSE
                INCREMENT n BY two
            END IF
        END WHILE
    END FUNCTION
END CLASS
