CLASS Solution
    FUNCTION shiftingLetters(string s, list_of_integer shifts)
        SET total_shifts TO zero
        FOR index i FROM the LENGTH OF shifts MINUS one DOWN TO zero INCLUSIVE
            INCREMENT total_shifts BY the element at position i OF shifts
            SET the element at position i OF shifts TO total_shifts
        END FOR

        SET result TO empty list
        FOR index i FROM zero TO the LENGTH OF s MINUS one INCLUSIVE
            SET original_character_position TO the position in the alphabet of element at position i OF s MINUS the position in the alphabet of the lowercase letter 'a'
            SET shifted_position TO original_character_position PLUS the element at position i OF shifts
            SET wrapped_position TO shifted_position MODULO the quantity of letters in the English alphabet
            SET new_character TO the lowercase letter at position wrapped_position starting from the lowercase letter 'a'
            APPEND new_character TO result
        END FOR

        RETURN the concatenation of all elements IN result AS a single string
    END FUNCTION
END CLASS
