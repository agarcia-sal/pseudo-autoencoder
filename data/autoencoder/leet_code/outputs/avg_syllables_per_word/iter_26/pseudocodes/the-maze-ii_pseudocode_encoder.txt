CLASS Solution
    FUNCTION shortestDistance(maze, start, destination)
        SET m TO the LENGTH OF maze
        SET n TO the LENGTH OF element at position zero OF maze
        SET directions TO the list containing tuples of one AND zero, MINUS one AND zero, zero AND one, zero AND MINUS one
        SET start TO the tuple form OF start
        SET destination TO the tuple form OF destination
        SET visited TO an empty set
        SET pq TO a list containing one tuple of zero AND start

        WHILE pq HOLDS
            ASSIGN dist AND a tuple of x AND y TO the result of REMOVING the smallest element FROM pq using the heap pop operation

            IF the tuple of x AND y EQUALS destination
                RETURN dist
            END IF

            IF the tuple of x AND y IS IN visited
                CONTINUE the WHILE loop
            END IF

            ADD the tuple of x AND y TO visited

            FOR each tuple of dx AND dy IN directions
                SET new_x TO x
                SET new_y TO y
                SET count TO zero

                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND the element at position new_x PLUS dx OF the element at position new_y PLUS dy OF maze EQUALS zero
                    INCREMENT new_x BY dx
                    INCREMENT new_y BY dy
                    INCREMENT count BY one
                END WHILE

                IF the tuple of new_x AND new_y IS NOT IN visited
                    PUSH onto pq the tuple of dist PLUS count AND the tuple of new_x AND new_y using the heap push operation
                END IF
            END FOR
        END WHILE

        RETURN MINUS one
    END FUNCTION
END CLASS
