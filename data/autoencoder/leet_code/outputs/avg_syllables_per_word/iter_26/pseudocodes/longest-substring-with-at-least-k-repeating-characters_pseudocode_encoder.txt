CLASS Solution
    FUNCTION longestSubstring(s, k)
        IF the LENGTH OF s LESS THAN k HOLDS
            RETURN zero
        END IF

        ASSIGN reference TO Counter OBJECT CREATED FROM s CALLED count
        
        FOR each character IN count
            IF the FREQUENCY OF character IN count LESS THAN k HOLDS
                RETURN the MAXIMUM OF the RESULTS OF CALLING longestSubstring WITH each substring OBTAINED BY SPLITTING s BY character AND k AS PARAMETERS
            END IF
        END FOR

        RETURN the LENGTH OF s
    END FUNCTION
END CLASS
