CLASS Solution
    FUNCTION checkInclusion(s1 AS string, s2 AS string) RETURNS boolean
        IMPORT Counter FROM collections
        
        SET length_of_s1 TO the LENGTH OF s1
        SET length_of_s2 TO the LENGTH OF s2
        
        IF length_of_s1 GREATER THAN length_of_s2
            RETURN False
        END IF
        
        SET count_s1 TO a Counter OF the characters IN s1
        SET count_s2 TO a Counter OF the characters IN the substring FROM position zero TO position length_of_s1 MINUS one OF s2
        
        IF count_s1 EQUALS count_s2
            RETURN True
        END IF
        
        FOR index FROM length_of_s1 TO length_of_s2 MINUS one INCLUSIVE
            SET character_to_add TO the element at position index OF s2
            INCREMENT the count OF character_to_add IN count_s2 BY one
            
            SET character_to_remove TO the element at position index MINUS length_of_s1 OF s2
            DECREMENT the count OF character_to_remove IN count_s2 BY one
            
            IF the count OF character_to_remove IN count_s2 EQUALS zero
                REMOVE character_to_remove FROM count_s2
            END IF
            
            IF count_s1 EQUALS count_s2
                RETURN True
            END IF
        END FOR
        
        RETURN False
    END FUNCTION
END CLASS
