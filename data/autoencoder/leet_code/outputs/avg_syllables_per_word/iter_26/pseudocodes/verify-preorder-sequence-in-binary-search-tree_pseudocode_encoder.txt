CLASS Solution
    FUNCTION verifyPreorder(preorder)
        SET stack TO an empty list
        SET lower_bound TO negative infinity

        FOR each value IN preorder
            IF value LESS THAN lower_bound
                RETURN False
            END IF

            WHILE the LENGTH OF stack is greater than zero AND value GREATER THAN the element at position the last index of stack OF stack
                SET lower_bound TO the element popped from the end of stack
            END WHILE

            APPEND value TO stack
        END FOR

        RETURN True
    END FUNCTION
END CLASS
