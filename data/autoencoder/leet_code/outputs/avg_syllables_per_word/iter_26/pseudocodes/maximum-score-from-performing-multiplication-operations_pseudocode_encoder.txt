CLASS Solution
    FUNCTION maximumScore(nums, multipliers)
        SET m TO the LENGTH OF multipliers
        SET n TO the LENGTH OF nums

        SET dp TO a list of zero VALUES with the LENGTH OF m PLUS one

        FOR index i FROM m MINUS one TO zero INCLUSIVE DECREASING BY one
            SET new_dp TO a list of zero VALUES with the LENGTH OF m PLUS one
            FOR index left FROM zero TO i INCLUSIVE
                SET right TO n MINUS one MINUS (i MINUS left)
                SET mul TO the element at position i OF multipliers
                SET value_one TO mul MULTIPLIED BY the element at position left OF nums PLUS the element at position left PLUS one OF dp
                SET value_two TO mul MULTIPLIED BY the element at position right OF nums PLUS the element at position left OF dp
                SET the element at position left OF new_dp TO the GREATER OF value_one AND value_two
            END FOR
            ASSIGN dp TO new_dp
        END FOR

        RETURN the element at position zero OF dp
    END FUNCTION
END CLASS
