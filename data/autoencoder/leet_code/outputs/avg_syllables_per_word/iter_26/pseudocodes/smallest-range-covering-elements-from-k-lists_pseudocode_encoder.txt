CLASS Solution
    FUNCTION smallestRange(nums)
        SET min_heap TO an empty list
        SET max_value TO negative infinity
        
        FOR index FROM zero TO the LENGTH OF nums MINUS one INCLUSIVE
            CALL the heap push operation TO INSERT a tuple of element at position zero OF list at position index OF nums, index, and zero INTO min_heap
            SET max_value TO the GREATER THAN value BETWEEN max_value AND element at position zero OF list at position index OF nums
        END FOR
        
        SET smallest_range TO positive infinity
        SET result TO a list containing a very large negative number and a very large positive number
        
        WHILE min_heap HOLDS
            CALL the heap pop operation TO REMOVE AND RETURN the minimum element FROM min_heap AS min_value, list_index, element_index
            
            IF max_value MINUS min_value LESS THAN smallest_range THEN
                SET smallest_range TO max_value MINUS min_value
                SET result TO a list containing min_value and max_value
            END IF
            
            IF element_index PLUS one LESS THAN the LENGTH OF list at position list_index OF nums THEN
                SET next_value TO element at position element_index PLUS one OF list at position list_index OF nums
                CALL the heap push operation TO INSERT a tuple of next_value, list_index, and element_index PLUS one INTO min_heap
                SET max_value TO the GREATER THAN value BETWEEN max_value AND next_value
            ELSE
                BREAK the WHILE loop
            END IF
        END WHILE
        
        RETURN result
    END FUNCTION
END CLASS
