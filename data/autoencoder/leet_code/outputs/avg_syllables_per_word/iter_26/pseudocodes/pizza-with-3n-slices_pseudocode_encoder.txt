CLASS Solution
    FUNCTION maxSizeSlices(parameter slices OF TYPE list of integers)
        SET n TO the LENGTH OF slices DIVIDED BY three

        FUNCTION dp(parameter start, parameter end, parameter k)
            IF k EQUALS zero OR start GREATER THAN end THEN
                RETURN zero
            END IF
            SET pick TO the element at position start OF slices PLUS dp(start PLUS two, end, k MINUS one)
            SET skip TO dp(start PLUS one, end, k)
            RETURN the GREATER VALUE BETWEEN pick AND skip
        END FUNCTION

        SET case1 TO dp(one, the LENGTH OF slices MINUS one, n)
        SET case2 TO dp(zero, the LENGTH OF slices MINUS two, n)
        RETURN the GREATER VALUE BETWEEN case1 AND case2
END CLASS
