CLASS Solution
    FUNCTION fourSum(nums, target)
        CALL the SORT METHOD ON nums
        SET n TO the LENGTH OF nums
        SET quadruplets TO an empty list
        
        FOR i FROM zero TO n MINUS three MINUS one INCLUSIVE
            IF i GREATER THAN zero AND the element at position i OF nums EQUALS the element at position i MINUS one OF nums
                CONTINUE to the next iteration
            END IF
            
            FOR j FROM i PLUS one TO n MINUS two MINUS one INCLUSIVE
                IF j GREATER THAN i PLUS one AND the element at position j OF nums EQUALS the element at position j MINUS one OF nums
                    CONTINUE to the next iteration
                END IF
                
                SET left TO j PLUS one
                SET right TO n MINUS one
                
                WHILE left LESS THAN right
                    SET total TO the element at position i OF nums PLUS the element at position j OF nums PLUS the element at position left OF nums PLUS the element at position right OF nums
                    
                    IF total EQUALS target
                        APPEND a list containing the element at position i OF nums, the element at position j OF nums, the element at position left OF nums, and the element at position right OF nums TO quadruplets
                        
                        WHILE left LESS THAN right AND the element at position left OF nums EQUALS the element at position left PLUS one OF nums
                            INCREMENT left BY one
                        END WHILE
                        
                        WHILE left LESS THAN right AND the element at position right OF nums EQUALS the element at position right MINUS one OF nums
                            DECREMENT right BY one
                        END WHILE
                        
                        INCREMENT left BY one
                        DECREMENT right BY one
                        
                    ELSE IF total LESS THAN target
                        INCREMENT left BY one
                        
                    ELSE
                        DECREMENT right BY one
                        
                    END IF
                END WHILE
            END FOR
        END FOR
        
        RETURN quadruplets
    END FUNCTION
END CLASS
