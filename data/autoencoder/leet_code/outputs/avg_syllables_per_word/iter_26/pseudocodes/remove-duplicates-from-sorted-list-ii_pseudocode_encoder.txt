CLASS ListNode
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values AS list)
    IF values IS NOT TRUTHY
        RETURN none
    END IF
    SET head TO A NEW ListNode WITH element at position zero OF values
    SET p TO head
    FOR each val IN the SUBLIST OF values FROM position one TO the END
        SET node TO A NEW ListNode WITH val
        SET p.next TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS none AND p2 IS none
        RETURN true
    END IF
    IF p1 IS NOT TRUTHY OR p2 IS NOT TRUTHY
        RETURN false
    END IF
    RETURN (element val AT p1 EQUALS element val AT p2) AND the RESULT OF is_same_list(element next AT p1, element next AT p2)
END FUNCTION

CLASS Solution
    FUNCTION deleteDuplicates(self, head AS Optional ListNode) RETURNS Optional ListNode
        SET dummy TO A NEW ListNode WITH val EQUALS zero AND next EQUALS head
        SET prev TO dummy

        WHILE head HOLDS
            IF element next AT head IS TRUTHY AND element val AT head EQUALS element val AT element next AT head
                WHILE element next AT head IS TRUTHY AND element val AT head EQUALS element val AT element next AT head
                    SET head TO element next AT head
                END WHILE
                SET element next AT prev TO element next AT head
            ELSE
                SET prev TO element next AT prev
            END IF
            SET head TO element next AT head
        END WHILE

        RETURN element next AT dummy
    END FUNCTION
END CLASS
