CLASS Solution
    FUNCTION isPossible(target)
        SET max_heap TO empty list
        FOR each number IN target
            APPEND negative of number TO max_heap
        END FOR
        CALL heapify ON max_heap

        SET total_sum TO the sum of all elements IN target

        WHILE element at position zero OF max_heap LESS THAN negative one HOLDS
            SET largest TO negative of the result of popping the smallest element FROM max_heap
            SET rest_sum TO total_sum MINUS largest

            IF largest LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero THEN
                RETURN False
            END IF

            SET new_value TO largest MODULO rest_sum
            IF new_value EQUALS zero AND rest_sum NOT EQUALS one THEN
                RETURN False
            END IF

            SET total_sum TO rest_sum PLUS new_value
            PUSH negative of new_value INTO max_heap
        END WHILE

        RETURN True
    END FUNCTION
END CLASS
