CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS EMPTY
        RETURN None
    END IF
    SET root TO new TreeNode WITH val EQUALS element at position zero OF values
    SET i TO one
    SET queue TO new double ended queue
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO the first element removed FROM queue
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None
            SET node.left TO new TreeNode WITH val EQUALS element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None
            SET node.right TO new TreeNode WITH val EQUALS element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUALS TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION largestBSTSubtree(self, root)
        FUNCTION dfs(node)
            IF node IS None
                RETURN positive infinity, negative infinity, zero
            END IF
            SET left_min, left_max, left_size TO dfs(node.left)
            SET right_min, right_max, right_size TO dfs(node.right)
            IF left_max IS LESS THAN node.val AND node.val IS LESS THAN right_min
                RETURN the lesser OF node.val AND left_min, the greater OF node.val AND right_max, left_size PLUS right_size PLUS one
            END IF
            RETURN negative infinity, positive infinity, the greater OF left_size AND right_size
        END FUNCTION
        RETURN the third element OF dfs(root)
    END FUNCTION
END CLASS
