CLASS Solution
    FUNCTION soupServings(n)
        IF n GREATER THAN four thousand eight hundred
            RETURN one
        END IF

        SET n TO n PLUS twenty four DIVIDED BY twenty five rounding up to the nearest whole number

        SET memo TO an empty associative collection

        FUNCTION dp(a, b)
            IF the pair of a and b EXISTS IN memo
                RETURN the value ASSOCIATED WITH the pair of a and b IN memo
            END IF

            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero
                RETURN point five
            END IF

            IF a LESS THAN OR EQUAL TO zero
                RETURN one
            END IF

            IF b LESS THAN OR EQUAL TO zero
                RETURN zero
            END IF

            SET prob TO point two five MULTIPLIED BY the SUM OF
                dp(a MINUS four, b),
                dp(a MINUS three, b MINUS one),
                dp(a MINUS two, b MINUS two),
                dp(a MINUS one, b MINUS three)

            SET the value ASSOCIATED WITH the pair of a and b IN memo TO prob
            RETURN prob
        END FUNCTION

        RETURN dp(n, n)
END CLASS
