CLASS Solution
    FUNCTION minimumMountainRemovals(list_of_numbers)
        SET length_n TO the LENGTH OF list_of_numbers
        
        SET longest_increasing_subsequence TO a list WITH length_n ELEMENTS EACH SET TO one
        FOR index_i FROM zero TO length_n MINUS one INCLUSIVE
            FOR index_j FROM zero TO index_i MINUS one INCLUSIVE
                IF the element at position index_i OF list_of_numbers GREATER THAN the element at position index_j OF list_of_numbers
                    SET the element at position index_i OF longest_increasing_subsequence TO the GREATER VALUE BETWEEN the element at position index_i OF longest_increasing_subsequence AND the element at position index_j OF longest_increasing_subsequence PLUS one
                END IF
            END FOR
        END FOR
        
        SET longest_decreasing_subsequence TO a list WITH length_n ELEMENTS EACH SET TO one
        FOR index_i FROM length_n MINUS one DOWN TO zero INCLUSIVE
            FOR index_j FROM index_i PLUS one TO length_n MINUS one INCLUSIVE
                IF the element at position index_i OF list_of_numbers GREATER THAN the element at position index_j OF list_of_numbers
                    SET the element at position index_i OF longest_decreasing_subsequence TO the GREATER VALUE BETWEEN the element at position index_i OF longest_decreasing_subsequence AND the element at position index_j OF longest_decreasing_subsequence PLUS one
                END IF
            END FOR
        END FOR
        
        SET maximum_mountain_length TO zero
        FOR index_i FROM one TO length_n MINUS two INCLUSIVE
            IF the element at position index_i OF longest_increasing_subsequence GREATER THAN one AND the element at position index_i OF longest_decreasing_subsequence GREATER THAN one
                SET maximum_mountain_length TO the GREATER VALUE BETWEEN maximum_mountain_length AND the SUM OF the element at position index_i OF longest_increasing_subsequence AND the element at position index_i OF longest_decreasing_subsequence MINUS one
            END IF
        END FOR
        
        SET minimum_number_of_removals TO length_n MINUS maximum_mountain_length
        RETURN minimum_number_of_removals
    END FUNCTION
END CLASS
