CLASS Solution
    FUNCTION findDuplicate(paths)
        SET content_to_paths TO an empty dictionary mapping to empty lists
        
        FOR each path IN paths
            SET parts TO the list of substrings obtained by splitting the path by whitespace
            SET directory TO the element at position zero OF parts
            
            FOR each file_info IN the elements FROM position one TO the last OF parts
                SET file_name TO the substring BEFORE the first open parenthesis OF file_info
                SET content TO the substring BETWEEN the first open parenthesis AND the last close parenthesis OF file_info
                
                APPEND the concatenation of directory, a forward slash, and file_name TO the list in content_to_paths under the key content
            END FOR
        END FOR
        
        SET result TO an empty list
        FOR each list_of_paths IN the values OF content_to_paths
            IF the LENGTH OF list_of_paths GREATER THAN one
                APPEND list_of_paths TO result
            END IF
        END FOR
        
        RETURN result
    END FUNCTION
END CLASS
