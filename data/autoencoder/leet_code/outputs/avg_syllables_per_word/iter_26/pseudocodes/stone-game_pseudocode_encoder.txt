CLASS Solution
    FUNCTION stoneGame(piles)
        FUNCTION dfs(i, j)
            IF i GREATER THAN j
                RETURN zero
            END IF
            RETURN the MAXIMUM OF the element at position i OF piles MINUS dfs(i INCREMENTED BY one, j) AND the element at position j OF piles MINUS dfs(i, j DECREMENTED BY one)
        END FUNCTION
        RETURN dfs(zero, the LENGTH OF piles MINUS one) GREATER THAN zero
    END FUNCTION
END CLASS
