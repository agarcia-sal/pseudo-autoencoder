CLASS Solution
    FUNCTION minDays(grid)
        FUNCTION dfs(x, y, visited)
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero OF grid OR the ordered pair of x AND y IS IN visited OR element at position x OF grid AT position y EQUALS zero THEN
                RETURN
            END IF
            ADD the ordered pair of x AND y TO visited
            FOR each ordered pair dx AND dy IN the collection of ordered pairs consisting of negative one AND zero, positive one AND zero, zero AND negative one, AND zero AND positive one
                CALL dfs WITH x PLUS dx, y PLUS dy, visited AS arguments
            END FOR
        END FUNCTION

        FUNCTION is_connected
            SET visited TO an empty set
            SET land_cells TO the collection of ordered pairs i AND j FOR each i FROM zero TO the LENGTH OF grid MINUS one INCLUSIVE AND for each j FROM zero TO the LENGTH OF element at position zero OF grid MINUS one INCLUSIVE WHERE element at position i OF grid AT position j EQUALS one
            IF land_cells IS empty THEN
                RETURN True
            END IF
            CALL dfs WITH the first element of the ordered pair AT position zero OF the first element OF land_cells, THE second element of the ordered pair AT position one OF the first element OF land_cells, AND visited AS arguments
            RETURN whether the SIZE OF visited EQUALS the SIZE OF land_cells
        END FUNCTION

        IF NOT is_connected HOLDS THEN
            RETURN zero
        END IF

        FOR i FROM zero TO the LENGTH OF grid MINUS one INCLUSIVE
            FOR j FROM zero TO the LENGTH OF element at position zero OF grid MINUS one INCLUSIVE
                IF element at position i OF grid AT position j EQUALS one THEN
                    SET element at position i OF grid AT position j TO zero
                    IF NOT is_connected HOLDS THEN
                        RETURN one
                    END IF
                    SET element at position i OF grid AT position j TO one
                END IF
            END FOR
        END FOR

        RETURN two
    END FUNCTION
END CLASS
