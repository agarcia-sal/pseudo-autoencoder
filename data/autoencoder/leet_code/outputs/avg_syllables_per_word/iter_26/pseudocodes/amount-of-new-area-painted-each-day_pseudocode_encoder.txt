CLASS Solution
    FUNCTION amountPainted(list_of_paint_intervals)
        SET maximum_end_position TO the maximum end value extracted FROM each pair OF list_of_paint_intervals
        SET painted_status_list TO a list FILLED WITH minus one OF length maximum_end_position PLUS one
        SET worklog_list TO an empty list
        
        FOR each pair start_position AND end_position IN list_of_paint_intervals
            SET current_start_position TO start_position
            SET new_paint_amount TO zero
            
            WHILE current_start_position IS LESS THAN end_position
                IF element at position current_start_position OF painted_status_list EQUALS minus one
                    SET element at position current_start_position OF painted_status_list TO end_position
                    INCREMENT new_paint_amount BY one
                    INCREMENT current_start_position BY one
                ELSE
                    SET current_start_position TO element at position current_start_position OF painted_status_list
                END IF
            END WHILE
            
            APPEND new_paint_amount TO worklog_list
        END FOR
        
        RETURN worklog_list
    END FUNCTION
END CLASS
