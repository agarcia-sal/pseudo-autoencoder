CLASS Solution
    FUNCTION minDays(n)
        FUNCTION dp(remaining)
            IF remaining LESS THAN OR EQUAL TO one
                RETURN remaining
            END IF
            SET option_one TO the remainder when remaining IS DIVIDED BY two PLUS dp(the quotient when remaining IS DIVIDED BY two)
            SET option_two TO the remainder when remaining IS DIVIDED BY three PLUS dp(the quotient when remaining IS DIVIDED BY three)
            RETURN one PLUS the lesser of option_one AND option_two
        END FUNCTION
        RETURN dp(n)
END CLASS
