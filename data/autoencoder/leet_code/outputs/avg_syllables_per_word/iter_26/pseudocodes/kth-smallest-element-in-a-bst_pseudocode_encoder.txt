CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self dot val TO val
        SET self dot left TO left
        SET self dot right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS NONE OR the LENGTH OF values EQUALS zero THEN
        RETURN none
    END IF
    SET root TO A NEW TreeNode WITH the element at position zero OF values
    SET i TO one
    SET queue TO AN EMPTY double ended queue
    APPEND root TO queue

    WHILE queue IS NOT EMPTY
        SET node TO REMOVE THE FIRST ELEMENT FROM queue
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none THEN
            SET node dot left TO A NEW TreeNode WITH the element at position i OF values
            APPEND node dot left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF values AND the element at position i OF values IS NOT none THEN
            SET node dot right TO A NEW TreeNode WITH the element at position i OF values
            APPEND node dot right TO queue
        END IF
        INCREMENT i BY one
    END WHILE

    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none THEN
        RETURN true
    ELSE IF p IS none OR q IS none THEN
        RETURN false
    ELSE IF p dot val IS NOT EQUALS TO q dot val THEN
        RETURN false
    ELSE
        RETURN is_same_tree WITH p dot left AND q dot left AND is_same_tree WITH p dot right AND q dot right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION kthSmallest(self, root, k)
        SET stack TO AN EMPTY list
        SET current TO root
        SET count TO zero

        WHILE stack IS NOT EMPTY OR current IS NOT none
            WHILE current IS NOT none
                APPEND current TO stack
                SET current TO current dot left
            END WHILE
            SET current TO THE LAST ELEMENT REMOVED FROM stack
            INCREMENT count BY one
            IF count IS EQUALS TO k THEN
                RETURN current dot val
            END IF
            SET current TO current dot right
        END WHILE
    END FUNCTION
END CLASS
