CLASS ListNode
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values)
    IF values IS EMPTY
        RETURN None
    END IF
    SET head TO NEW ListNode WITH element at position zero OF values
    SET p TO head
    FOR each val IN the SLICE OF values FROM position one TO END
        SET node TO NEW ListNode WITH val
        SET p.next TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS None AND p2 IS None
        RETURN True
    END IF
    IF p1 IS None OR p2 IS None
        RETURN False
    END IF
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)
END FUNCTION

CLASS Solution
    FUNCTION swapPairs(self, head)
        # Create a dummy node to simplify edge cases
        SET dummy TO NEW ListNode WITH zero
        SET dummy.next TO head
        SET current TO dummy

        WHILE current.next IS NOT None AND current.next.next IS NOT None
            # Nodes to be swapped
            SET first TO current.next
            SET second TO current.next.next

            # Swapping
            SET current.next TO second
            SET first.next TO second.next
            SET second.next TO first

            # Move to the next pair
            SET current TO current.next.next
        END WHILE

        RETURN dummy.next
    END FUNCTION
END CLASS
