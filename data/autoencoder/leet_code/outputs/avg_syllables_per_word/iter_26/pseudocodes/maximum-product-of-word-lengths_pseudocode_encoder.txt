CLASS Solution
    FUNCTION maxProduct(words)
        SET word_sets TO an empty list
        FOR each word IN words
            APPEND the set of characters IN word TO word_sets
        END FOR

        SET max_product TO zero

        FOR i FROM zero TO the LENGTH OF words MINUS one INCLUSIVE
            FOR j FROM i PLUS one TO the LENGTH OF words MINUS one INCLUSIVE
                IF NOT the set of characters element at position i OF word_sets INTERSECTED WITH the set of characters element at position j OF word_sets HOLDS
                    SET product TO the LENGTH OF element at position i OF words MULTIPLIED BY the LENGTH OF element at position j OF words
                    IF product GREATER THAN max_product
                        SET max_product TO product
                    END IF
                END IF
            END FOR
        END FOR

        RETURN max_product
    END FUNCTION
END CLASS
