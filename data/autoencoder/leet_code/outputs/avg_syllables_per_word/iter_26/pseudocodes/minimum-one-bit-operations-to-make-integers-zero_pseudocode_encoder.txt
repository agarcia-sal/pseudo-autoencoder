CLASS Solution
    FUNCTION minimumOneBitOperations(n)
        IF n EQUALS zero
            RETURN zero
        END IF

        SET k TO zero
        WHILE one LEFT SHIFTED BY the quantity of k PLUS one IS LESS THAN OR EQUAL TO n HOLDS
            INCREMENT k BY one
        END WHILE

        SET most_significant_bit TO one LEFT SHIFTED BY k

        RETURN one LEFT SHIFTED BY the quantity of k PLUS one MINUS one MINUS CALL minimumOneBitOperations WITH n EXCLUSIVE OR most_significant_bit ON self
    END FUNCTION
END CLASS
