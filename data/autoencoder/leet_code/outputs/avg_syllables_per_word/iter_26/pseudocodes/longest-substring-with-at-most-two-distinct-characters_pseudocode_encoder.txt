CLASS Solution
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)
        CREATE a dictionary named character_count with default zero integer values
        SET left_pointer TO zero
        SET maximum_substring_length TO zero
        
        FOR right_pointer FROM zero TO the LENGTH OF s MINUS one INCLUSIVE
            INCREMENT the count in character_count FOR the character at position right_pointer OF s BY one
            
            WHILE the NUMBER OF KEYS IN character_count IS GREATER THAN two HOLDS
                DECREMENT the count in character_count FOR the character at position left_pointer OF s BY one
                IF the count in character_count FOR the character at position left_pointer OF s EQUALS zero HOLDS
                    REMOVE the character at position left_pointer OF s FROM character_count
                END IF
                INCREMENT left_pointer BY one
            END WHILE

            SET maximum_substring_length TO the GREATER OF maximum_substring_length AND right_pointer MINUS left_pointer PLUS one
        END FOR
        
        RETURN maximum_substring_length
    END FUNCTION
END CLASS
