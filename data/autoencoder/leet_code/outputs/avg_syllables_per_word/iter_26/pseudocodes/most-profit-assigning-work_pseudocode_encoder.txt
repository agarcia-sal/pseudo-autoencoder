CLASS Solution
    FUNCTION maxProfitAssignment(difficulty, profit, worker)
        SET jobs TO a LIST OF TUPLES combining each element from difficulty AND corresponding element from profit
        SORT jobs in ascending order BY the first element of each tuple representing difficulty
        SORT worker in ascending order by ability
        
        SET max_profit TO zero
        SET best_profit TO zero
        SET job_index TO zero
        SET n TO the LENGTH OF jobs
        
        FOR each ability IN worker
            WHILE job_index LESS THAN n AND the first element at position job_index OF jobs LESS THAN OR EQUAL TO ability HOLDS
                SET best_profit TO the GREATER BETWEEN best_profit AND the second element at position job_index OF jobs
                INCREMENT job_index BY one
            END WHILE
            INCREMENT max_profit BY best_profit
        END FOR
        
        RETURN max_profit
    END FUNCTION
END CLASS
