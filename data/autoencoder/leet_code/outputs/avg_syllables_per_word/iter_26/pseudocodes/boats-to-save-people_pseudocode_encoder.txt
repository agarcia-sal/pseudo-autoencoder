CLASS Solution
    FUNCTION numRescueBoats(list_of_people WEIGHTED AS LIST OF INTEGERS, weight_limit AS INTEGER) RETURNS INTEGER
        CALL SORT on list_of_people TO ARRANGE elements IN ASCENDING ORDER
        SET left_pointer TO zero
        SET right_pointer TO the LENGTH OF list_of_people MINUS one
        SET boat_count TO zero
        WHILE left_pointer IS LESS THAN OR EQUAL TO right_pointer HOLDS
            IF the ELEMENT AT POSITION left_pointer OF list_of_people PLUS the ELEMENT AT POSITION right_pointer OF list_of_people IS LESS THAN OR EQUAL TO weight_limit
                INCREMENT left_pointer BY one
            END IF
            DECREMENT right_pointer BY one
            INCREMENT boat_count BY one
        END WHILE
        RETURN boat_count
    END FUNCTION
END CLASS
