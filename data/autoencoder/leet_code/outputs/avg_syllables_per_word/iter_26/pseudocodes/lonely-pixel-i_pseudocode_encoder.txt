CLASS Solution
    FUNCTION findLonelyPixel(two_dimensional_list picture)
        IF picture IS empty OR the LENGTH OF picture IS zero OR the LENGTH OF element at position zero OF picture IS zero
            RETURN zero
        END IF

        SET variable m TO the LENGTH OF picture
        SET variable n TO the LENGTH OF element at position zero OF picture

        SET list row_count TO a list of zero repeated m times
        SET list col_count TO a list of zero repeated n times

        FOR index i FROM zero TO m MINUS one INCLUSIVE
            FOR index j FROM zero TO n MINUS one INCLUSIVE
                IF element at position j OF element at position i OF picture EQUALS character B
                    INCREMENT element at position i OF row_count BY one
                    INCREMENT element at position j OF col_count BY one
                END IF
            END FOR
        END FOR

        SET variable lonely_pixel_count TO zero

        FOR index i FROM zero TO m MINUS one INCLUSIVE
            FOR index j FROM zero TO n MINUS one INCLUSIVE
                IF element at position j OF element at position i OF picture EQUALS character B AND element at position i OF row_count EQUALS one AND element at position j OF col_count EQUALS one
                    INCREMENT lonely_pixel_count BY one
                END IF
            END FOR
        END FOR

        RETURN lonely_pixel_count
    END FUNCTION
END CLASS
