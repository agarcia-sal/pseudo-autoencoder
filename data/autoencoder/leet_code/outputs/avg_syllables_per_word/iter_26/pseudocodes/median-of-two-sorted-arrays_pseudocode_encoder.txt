CLASS Solution
    FUNCTION findMedianSortedArrays(nums1, nums2)
        IF the LENGTH OF nums1 IS GREATER THAN the LENGTH OF nums2 THEN
            SET temporary_reference TO nums1
            SET nums1 TO nums2
            SET nums2 TO temporary_reference
        END IF

        SET x TO the LENGTH OF nums1
        SET y TO the LENGTH OF nums2

        SET low TO zero
        SET high TO x

        WHILE low IS LESS THAN OR EQUAL TO high HOLDS
            SET partitionX TO the INTEGER PART OF the SUM OF low AND high DIVIDED BY two
            SET partitionY TO the INTEGER PART OF the SUM OF x AND y AND one DIVIDED BY two MINUS partitionX

            IF partitionX EQUALS zero THEN
                SET maxLeftX TO NEGATIVE infinity
            ELSE
                SET maxLeftX TO the element at position partitionX MINUS one OF nums1
            END IF

            IF partitionX EQUALS x THEN
                SET minRightX TO POSITIVE infinity
            ELSE
                SET minRightX TO the element at position partitionX OF nums1
            END IF

            IF partitionY EQUALS zero THEN
                SET maxLeftY TO NEGATIVE infinity
            ELSE
                SET maxLeftY TO the element at position partitionY MINUS one OF nums2
            END IF

            IF partitionY EQUALS y THEN
                SET minRightY TO POSITIVE infinity
            ELSE
                SET minRightY TO the element at position partitionY OF nums2
            END IF

            IF maxLeftX IS LESS THAN OR EQUAL TO minRightY AND maxLeftY IS LESS THAN OR EQUAL TO minRightX THEN
                IF the MODULO of the SUM OF x AND y AND two EQUALS zero THEN
                    RETURN the SUM OF the GREATER OF maxLeftX AND maxLeftY AND the LESSER OF minRightX AND minRightY DIVIDED BY two
                ELSE
                    RETURN the GREATER OF maxLeftX AND maxLeftY
                END IF
            ELSE IF maxLeftX IS GREATER THAN minRightY THEN
                SET high TO partitionX MINUS one
            ELSE
                SET low TO partitionX PLUS one
            END IF
        END WHILE
    END FUNCTION
END CLASS
