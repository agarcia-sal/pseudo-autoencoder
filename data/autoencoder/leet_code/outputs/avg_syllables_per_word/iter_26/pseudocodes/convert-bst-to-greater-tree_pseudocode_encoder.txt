CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN none
    END IF
    SET root TO NEW TreeNode WITH value OF element at position zero OF values
    SET i TO one
    SET queue TO AN EMPTY double ended queue
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO THE FIRST element REMOVED FROM queue
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none
            SET node.left TO NEW TreeNode WITH value OF element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none
            SET node.right TO NEW TreeNode WITH value OF element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF value OF p IS NOT EQUALS TO value OF q
        RETURN false
    ELSE
        RETURN is_same_tree(left child OF p, left child OF q) AND is_same_tree(right child OF p, right child OF q)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION convertBST(self, root)
        SET self.sum TO zero
        
        FUNCTION traverse(node)
            IF node IS none
                RETURN
            END IF
            CALL traverse WITH right child OF node
            INCREMENT self.sum BY value OF node
            SET value OF node TO self.sum
            CALL traverse WITH left child OF node
        END FUNCTION
        
        CALL traverse WITH root
        RETURN root
    END FUNCTION
END CLASS
