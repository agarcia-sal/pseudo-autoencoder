CLASS Solution
    FUNCTION minWindow(s1, s2)
        SET minimum_window_length TO a very large number representing infinity
        SET minimum_window_start_index TO negative one
        SET length_of_s1 TO the LENGTH OF s1
        SET length_of_s2 TO the LENGTH OF s2

        FOR index_i FROM zero TO length_of_s1 MINUS one INCLUSIVE
            SET pointer_j TO zero
            FOR index_k FROM index_i TO length_of_s1 MINUS one INCLUSIVE
                IF the element at position index_k OF s1 EQUALS the element at position pointer_j OF s2 THEN
                    INCREMENT pointer_j BY one
                    IF pointer_j EQUALS length_of_s2 THEN
                        IF index_k MINUS index_i PLUS one LESS THAN minimum_window_length THEN
                            SET minimum_window_length TO index_k MINUS index_i PLUS one
                            SET minimum_window_start_index TO index_i
                        END IF
                        BREAK the inner loop
                    END IF
                END IF
            END FOR
        END FOR

        IF minimum_window_start_index EQUALS negative one THEN
            RETURN an empty string
        ELSE
            RETURN the substring FROM position minimum_window_start_index TO position minimum_window_start_index PLUS minimum_window_length MINUS one OF s1
        END IF
    END FUNCTION
END CLASS
