CLASS Solution
    FUNCTION calculateMinimumHP(dungeon)
        SET number_of_rows TO the LENGTH OF dungeon
        SET number_of_columns TO the LENGTH OF element at position zero OF dungeon

        CREATE dp AS a two-dimensional list WITH number_of_rows INCREMENTED BY one rows AND number_of_columns INCREMENTED BY one columns
        INITIALIZE each element OF dp TO positive infinity

        SET element at position number_of_rows OF dp AT position number_of_columns MINUS one TO one
        SET element at position number_of_rows MINUS one OF dp AT position number_of_columns TO one

        FOR row_index FROM number_of_rows MINUS one DOWN TO zero INCLUSIVE
            FOR column_index FROM number_of_columns MINUS one DOWN TO zero INCLUSIVE
                SET minimum_health_needed TO the MINIMUM OF element at position row_index INCREMENTED BY one OF dp AT position column_index AND element at position row_index OF dp AT position column_index INCREMENTED BY one
                DECREMENT minimum_health_needed BY element at position row_index OF dungeon AT position column_index
                SET element at position row_index OF dp AT position column_index TO the MAXIMUM OF minimum_health_needed AND one
            END FOR
        END FOR

        RETURN element at position zero OF dp AT position zero
    END FUNCTION
END CLASS
