CLASS Solution
    FUNCTION fractionToDecimal(numerator, denominator)
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero THEN
            SET sign TO a minus symbol
        ELSE
            SET sign TO empty string
        END IF

        SET numerator TO the absolute value of numerator
        SET denominator TO the absolute value of denominator

        SET quotient AND remainder TO the result of dividing numerator BY denominator with remainder
        SET integer_part TO the string representation of quotient

        IF remainder EQUALS zero THEN
            RETURN the concatenation of sign AND integer_part
        END IF

        SET seen_remainders TO an empty dictionary to store previously encountered remainders and their positions
        SET fractional_part TO an empty list to accumulate fractional digits

        WHILE remainder NOT EQUALS zero HOLDS
            IF remainder IS A KEY IN seen_remainders THEN
                SET start TO the value in seen_remainders at key remainder
                SET non_repeating TO the concatenation of the elements from the zero position UP TO position start NOT INCLUDING start IN fractional_part
                SET repeating TO the concatenation of the elements from position start TO the end IN fractional_part
                RETURN the concatenation of sign, integer_part, a dot symbol, non_repeating, an opening parenthesis symbol, repeating, and a closing parenthesis symbol
            END IF

            SET the value in seen_remainders at key remainder TO the LENGTH OF fractional_part

            MULTIPLY remainder BY ten
            SET quotient AND remainder TO the result of dividing remainder BY denominator with remainder
            APPEND the string representation of quotient TO fractional_part
        END WHILE

        RETURN the concatenation of sign, integer_part, a dot symbol, and the concatenation of all elements IN fractional_part
    END FUNCTION
END CLASS
