CLASS Solution
    FUNCTION calculate(s)
        FUNCTION helper(list_of_characters)
            SET stack TO empty list
            SET sign TO the character PLUS
            SET num TO zero

            WHILE the LENGTH OF list_of_characters GREATER THAN zero HOLDS
                SET char TO the element AT position zero OF list_of_characters
                REMOVE element AT position zero FROM list_of_characters

                IF char IS A DIGIT
                    SET num TO num MULTIPLIED BY ten PLUS the integer value OF char
                END IF

                IF char EQUALS the character OPENING PARENTHESIS
                    SET num TO the result OF calling helper WITH list_of_characters
                END IF

                IF NOT char IS A DIGIT AND NOT char IS A WHITESPACE OR the LENGTH OF list_of_characters EQUALS zero
                    IF sign EQUALS the character PLUS
                        APPEND num TO stack
                    ELSE IF sign EQUALS the character MINUS
                        APPEND the NEGATION OF num TO stack
                    ELSE IF sign EQUALS the character MULTIPLICATION SIGN
                        SET the element AT the last position OF stack TO the element AT the last position OF stack MULTIPLIED BY num
                    ELSE IF sign EQUALS the character DIVISION SIGN
                        SET the element AT the last position OF stack TO the integer division OF the element AT the last position OF stack DIVIDED BY num
                    END IF

                    SET sign TO char
                    SET num TO zero
                END IF

                IF char EQUALS the character CLOSING PARENTHESIS
                    BREAK out of the WHILE loop
                END IF
            END WHILE

            RETURN the SUM OF all elements IN stack
        END FUNCTION

        RETURN the result OF calling helper WITH the list form OF s
    END FUNCTION
END CLASS
