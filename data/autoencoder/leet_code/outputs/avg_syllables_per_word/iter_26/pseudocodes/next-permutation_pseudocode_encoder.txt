CLASS Solution
    FUNCTION nextPermutation(nums)
        SET index_i TO the LENGTH OF nums MINUS one MINUS one
        WHILE index_i IS GREATER THAN OR EQUAL TO zero AND element at position index_i OF nums IS GREATER THAN OR EQUAL TO element at position index_i PLUS one OF nums HOLDS
            DECREMENT index_i BY one
        END WHILE

        IF index_i IS GREATER THAN OR EQUAL TO zero THEN
            SET index_j TO the LENGTH OF nums MINUS one
            WHILE element at position index_j OF nums IS LESS THAN OR EQUAL TO element at position index_i OF nums HOLDS
                DECREMENT index_j BY one
            END WHILE
            SWAP element at position index_i OF nums WITH element at position index_j OF nums
        END IF

        REVERSE the SUBSEQUENCE of nums FROM position index_i PLUS one TO the LAST position INCLUSIVE
    END FUNCTION
END CLASS
