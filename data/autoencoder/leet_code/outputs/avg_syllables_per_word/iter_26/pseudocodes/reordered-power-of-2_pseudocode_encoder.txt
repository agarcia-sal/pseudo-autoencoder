CLASS Solution
    FUNCTION reorderedPowerOf2(n)
        SET power_of_2_sorted_digits TO an empty set
        SET power_of_2 TO one
        WHILE power_of_2 IS LESS THAN OR EQUAL TO ten RAISED TO THE POWER OF nine HOLDS
            ADD the tuple of the digits sorted in ascending order FROM the string representation OF power_of_2 TO power_of_2_sorted_digits
            SET power_of_2 TO power_of_2 MULTIPLIED BY two
        END WHILE
        SET sorted_n_digits TO the tuple of the digits sorted in ascending order FROM the string representation OF n
        RETURN whether sorted_n_digits IS IN power_of_2_sorted_digits
    END FUNCTION
END CLASS
