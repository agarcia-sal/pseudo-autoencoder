CLASS Solution
    FUNCTION minSkips(dist WITH TYPE List OF integers, speed WITH TYPE integer, hoursBefore WITH TYPE integer)
        SET n TO the LENGTH OF dist
        
        SET dp TO a two dimensional list with n rows EACH containing n PLUS one elements EACH set TO positive infinity
        
        SET total_dist TO the SUM OF all elements IN dist
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore
            RETURN negative one
        END IF
        
        SET element at position zero AT row zero OF dp TO element at position zero OF dist
        
        FOR j FROM one TO n MINUS one INCLUSIVE
            SET element at position j AT row zero OF dp TO element at position zero OF dist
        END FOR
        
        FOR i FROM one TO n MINUS one INCLUSIVE
            FOR j FROM zero TO i INCLUSIVE
                IF j LESS THAN i
                    SET element at position j AT row i OF dp TO the CEILING OF element at position j AT row i MINUS one OF dp DIVIDED BY speed MULTIPLIED BY speed PLUS element at position i OF dist
                END IF
                
                IF j GREATER THAN zero
                    SET element at position j AT row i OF dp TO the MINIMUM OF element at position j AT row i OF dp AND element at position j MINUS one AT row i MINUS one OF dp PLUS element at position i OF dist
                END IF
            END FOR
        END FOR
        
        FOR j FROM zero TO n MINUS one INCLUSIVE
            IF element at position j AT row n MINUS one OF dp LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed
                RETURN j
            END IF
        END FOR
        
        RETURN negative one
    END FUNCTION
END CLASS
