CLASS Solution
    FUNCTION threeSum(list_of_numbers)
        CALL the SORT method ON list_of_numbers
        SET result_list TO an empty list
        SET length_of_numbers TO the LENGTH OF list_of_numbers
        
        FOR index i FROM zero TO length_of_numbers MINUS two MINUS one INCLUSIVE
            IF index i GREATER THAN zero AND element at position i OF list_of_numbers EQUALS element at position i MINUS one OF list_of_numbers
                CONTINUE to the next iteration
            END IF

            SET left_pointer TO index i PLUS one
            SET right_pointer TO length_of_numbers MINUS one
            
            WHILE left_pointer LESS THAN right_pointer HOLDS
                SET total_sum TO element at position i OF list_of_numbers PLUS element at position left_pointer OF list_of_numbers PLUS element at position right_pointer OF list_of_numbers
                
                IF total_sum EQUALS zero
                    APPEND the list CONTAINING element at position i OF list_of_numbers, element at position left_pointer OF list_of_numbers, element at position right_pointer OF list_of_numbers TO result_list

                    WHILE left_pointer LESS THAN right_pointer AND element at position left_pointer OF list_of_numbers EQUALS element at position left_pointer PLUS one OF list_of_numbers HOLDS
                        INCREMENT left_pointer BY one
                    END WHILE

                    WHILE left_pointer LESS THAN right_pointer AND element at position right_pointer OF list_of_numbers EQUALS element at position right_pointer MINUS one OF list_of_numbers HOLDS
                        DECREMENT right_pointer BY one
                    END WHILE

                    INCREMENT left_pointer BY one
                    DECREMENT right_pointer BY one

                ELSE IF total_sum LESS THAN zero
                    INCREMENT left_pointer BY one

                ELSE
                    DECREMENT right_pointer BY one
                END IF
            END WHILE
        END FOR

        RETURN result_list
    END FUNCTION
END CLASS
