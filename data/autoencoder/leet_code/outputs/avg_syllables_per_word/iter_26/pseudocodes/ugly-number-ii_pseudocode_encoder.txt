CLASS Solution
    FUNCTION nthUglyNumber(number n)
        SET list ugly_numbers TO a list containing the number one
        SET integer i2 TO zero
        SET integer i3 TO zero
        SET integer i5 TO zero
        WHILE the LENGTH OF ugly_numbers LESS THAN n HOLDS
            SET integer next_multiple_of_2 TO the element at position i2 OF ugly_numbers MULTIPLIED BY the number two
            SET integer next_multiple_of_3 TO the element at position i3 OF ugly_numbers MULTIPLIED BY the number three
            SET integer next_multiple_of_5 TO the element at position i5 OF ugly_numbers MULTIPLIED BY the number five
            SET integer next_ugly_number TO the SMALLEST OF next_multiple_of_2 AND next_multiple_of_3 AND next_multiple_of_5
            APPEND next_ugly_number TO ugly_numbers
            IF next_ugly_number EQUALS next_multiple_of_2
                INCREMENT i2 BY one
            END IF
            IF next_ugly_number EQUALS next_multiple_of_3
                INCREMENT i3 BY one
            END IF
            IF next_ugly_number EQUALS next_multiple_of_5
                INCREMENT i5 BY one
            END IF
        END WHILE
        RETURN the element at position n MINUS one OF ugly_numbers
    END FUNCTION
END CLASS
