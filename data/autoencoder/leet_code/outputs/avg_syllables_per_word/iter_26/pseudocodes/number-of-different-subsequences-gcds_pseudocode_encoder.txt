CLASS Solution
    FUNCTION countDifferentSubsequenceGCDs(nums)
        SET maximum_number TO the maximum element IN nums
        SET number_set TO the set CONTAINING all elements IN nums
        SET count TO zero
        
        FOR g FROM one TO maximum_number INCLUSIVE
            SET gcd_value TO zero
            FOR m FROM g TO maximum_number INCLUSIVE STEPPED BY g
                IF element m IS IN number_set
                    SET gcd_value TO the greatest common divisor OF gcd_value AND m
                    IF gcd_value EQUALS g
                        INCREMENT count BY one
                        BREAK out of the inner loop
                    END IF
                END IF
            END FOR
        END FOR
        
        RETURN count
    END FUNCTION
END CLASS
