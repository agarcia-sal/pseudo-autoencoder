CLASS Solution
    FUNCTION minJumps(list_of_integers arr)
        SET length_of_arr TO the LENGTH OF arr
        IF length_of_arr EQUALS one
            RETURN zero
        END IF

        SET value_indices TO a DEFAULT DICTIONARY with LIST as default value
        FOR each position_index AND element_value IN the ENUMERATION OF arr
            APPEND position_index TO the LIST MAPPED BY element_value IN value_indices
        END FOR

        SET queue TO a DOUBLE ENDED QUEUE initialized with the TUPLE of zero AND zero
        SET visited TO a SET containing zero

        WHILE queue HOLDS
            REMOVE and RETURN the LEFTMOST element of queue INTO current_index AND steps

            IF current_index EQUALS length_of_arr MINUS one
                RETURN steps
            END IF

            SET next_indices TO a LIST containing current_index PLUS one AND current_index MINUS one CONCATENATED WITH the LIST MAPPED BY the element at position current_index OF arr IN value_indices

            FOR each next_index IN next_indices
                IF next_index IS GREATER THAN OR EQUAL TO zero AND next_index IS LESS THAN length_of_arr AND next_index IS NOT IN visited
                    ADD next_index TO visited
                    APPEND the TUPLE of next_index AND steps PLUS one TO queue
                END IF
            END FOR

            SET the LIST MAPPED BY the element at position current_index OF arr IN value_indices TO an EMPTY LIST
        END WHILE

        RETURN negative one
    END FUNCTION
END CLASS
