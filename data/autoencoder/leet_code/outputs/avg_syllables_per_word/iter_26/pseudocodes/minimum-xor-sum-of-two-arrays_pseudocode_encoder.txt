CLASS Solution
    FUNCTION minimumXORSum(nums1 AS List OF Integer, nums2 AS List OF Integer)
        SET n TO the LENGTH OF nums1
        SET dp TO a list OF infinity REPEATED as many times AS two RAISED TO the POWER OF n
        SET element at position zero OF dp TO zero

        FOR each mask FROM zero TO two RAISED TO the POWER OF n MINUS one INCLUSIVE
            SET bit_count TO the NUMBER OF ONES IN the BINARY REPRESENTATION OF mask
            FOR each j FROM zero TO n MINUS one INCLUSIVE
                IF the BITWISE AND OF mask AND two RAISED TO the POWER OF j HOLDS AS nonzero
                    SET dp at position mask TO the MINIMUM OF dp at position mask AND 
                        dp at position BITWISE XOR OF mask AND two RAISED TO the POWER OF j PLUS 
                        the BITWISE XOR OF the element at position bit_count MINUS one OF nums1 AND the element at position j OF nums2
                END IF
            END FOR
        END FOR

        RETURN the element at position two RAISED TO the POWER OF n MINUS one OF dp
    END FUNCTION
END CLASS
