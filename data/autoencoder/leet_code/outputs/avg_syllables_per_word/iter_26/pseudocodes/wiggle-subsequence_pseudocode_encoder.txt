CLASS Solution
    FUNCTION wiggleMaxLength(list_of_numbers)
        IF the LENGTH OF list_of_numbers LESS THAN two HOLDS
            RETURN the LENGTH OF list_of_numbers
        END IF

        SET upward_count TO one
        SET downward_count TO one

        FOR index FROM one TO the LENGTH OF list_of_numbers MINUS one INCLUSIVE
            IF the element at position index OF list_of_numbers GREATER THAN the element at position index MINUS one OF list_of_numbers HOLDS
                SET upward_count TO downward_count PLUS one
            ELSE IF the element at position index OF list_of_numbers LESS THAN the element at position index MINUS one OF list_of_numbers HOLDS
                SET downward_count TO upward_count PLUS one
            END IF
        END FOR

        RETURN the GREATER VALUE BETWEEN upward_count AND downward_count
    END FUNCTION
END CLASS
