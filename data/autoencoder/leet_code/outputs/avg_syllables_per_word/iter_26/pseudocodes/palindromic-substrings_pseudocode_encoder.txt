CLASS Solution

    FUNCTION countSubstrings(s AS string)
    
        FUNCTION expand_around_center(left AS integer, right AS integer)
            SET count TO zero
            WHILE left GREATER THAN OR EQUAL TO zero AND right LESS THAN the LENGTH OF s AND the ELEMENT AT POSITION left OF s EQUALS the ELEMENT AT POSITION right OF s HOLDS
                INCREMENT count BY one
                DECREMENT left BY one
                INCREMENT right BY one
            END WHILE
            RETURN count
        END FUNCTION
        
        SET total_palindromes TO zero
        FOR i FROM zero TO the LENGTH OF s MINUS one INCLUSIVE
            INCREMENT total_palindromes BY the RESULT OF expand_around_center WITH PARAMETERS i AND i
            INCREMENT total_palindromes BY the RESULT OF expand_around_center WITH PARAMETERS i AND i PLUS one
        END FOR
        RETURN total_palindromes

    END FUNCTION

END CLASS
