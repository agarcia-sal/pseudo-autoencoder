CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS NOT PRESENT THEN
        RETURN none
    END IF
    SET root TO A NEW TreeNode WITH the ELEMENT AT position zero OF values
    SET i TO one
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE
    APPEND root TO queue
    WHILE queue HOLDS
        SET node TO THE FIRST ELEMENT REMOVED FROM queue
        IF i IS LESS THAN THE LENGTH OF values AND THE ELEMENT AT position i OF values IS PRESENT THEN
            SET node.left TO A NEW TreeNode WITH THE ELEMENT AT position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN THE LENGTH OF values AND THE ELEMENT AT position i OF values IS PRESENT THEN
            SET node.right TO A NEW TreeNode WITH THE ELEMENT AT position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS NOT PRESENT AND q IS NOT PRESENT THEN
        RETURN true
    ELSE IF p IS NOT PRESENT OR q IS NOT PRESENT THEN
        RETURN false
    ELSE IF THE val ATTRIBUTE OF p IS NOT EQUALS THE val ATTRIBUTE OF q THEN
        RETURN false
    ELSE
        RETURN is_same_tree(the left ATTRIBUTE OF p, the left ATTRIBUTE OF q) AND is_same_tree(the right ATTRIBUTE OF p, the right ATTRIBUTE OF q)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION subtreeWithAllDeepest(self, root)
        FUNCTION dfs(node)
            IF node IS NOT PRESENT THEN
                RETURN node, zero
            END IF
            SET left_node, left_depth TO dfs(the left ATTRIBUTE OF node)
            SET right_node, right_depth TO dfs(the right ATTRIBUTE OF node)
            IF left_depth IS GREATER THAN right_depth THEN
                RETURN left_node, left_depth PLUS one
            ELSE IF right_depth IS GREATER THAN left_depth THEN
                RETURN right_node, right_depth PLUS one
            ELSE
                RETURN node, left_depth PLUS one
            END IF
        END FUNCTION
        RETURN THE FIRST ELEMENT OF dfs(root)
    END FUNCTION
END CLASS
