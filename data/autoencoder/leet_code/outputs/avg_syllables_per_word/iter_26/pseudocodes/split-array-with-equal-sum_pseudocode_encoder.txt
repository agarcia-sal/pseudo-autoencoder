CLASS Solution
    FUNCTION splitArray(nums)
        SET n TO the LENGTH OF nums
        IF n LESS THAN seven THEN
            RETURN false
        END IF

        SET prefix_sum TO a list OF the LENGTH OF n FILLED WITH zeroes
        SET the element at position zero OF prefix_sum TO the element at position zero OF nums
        
        FOR i FROM one TO n MINUS one EXCLUSIVE
            SET the element at position i OF prefix_sum TO the element at position i MINUS one OF prefix_sum PLUS the element at position i OF nums
        END FOR

        FOR j FROM three TO n MINUS three EXCLUSIVE
            SET seen TO an empty set
            
            FOR i FROM one TO j MINUS one EXCLUSIVE
                IF the element at position i MINUS one OF prefix_sum EQUALS the element at position j MINUS one OF prefix_sum MINUS the element at position i OF prefix_sum THEN
                    ADD the element at position i MINUS one OF prefix_sum TO the set seen
                END IF
            END FOR

            FOR k FROM j PLUS two TO n MINUS one EXCLUSIVE
                IF the element at position n MINUS one OF prefix_sum MINUS the element at position k OF prefix_sum EQUALS the element at position k MINUS one OF prefix_sum MINUS the element at position j OF prefix_sum AND the element at position k MINUS one OF prefix_sum MINUS the element at position j OF prefix_sum IS IN the set seen THEN
                    RETURN true
                END IF
            END FOR
        END FOR

        RETURN false
    END FUNCTION
END CLASS
