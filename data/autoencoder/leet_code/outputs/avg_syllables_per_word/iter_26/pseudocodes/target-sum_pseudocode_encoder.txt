CLASS Solution
    FUNCTION findTargetSumWays(list_of_numbers, target_value)
        FUNCTION dfs(current_index, current_sum_value, memo_dictionary)
            IF current_index EQUALS the LENGTH OF list_of_numbers
                IF current_sum_value EQUALS target_value
                    RETURN one
                ELSE
                    RETURN zero
                END IF
            END IF

            IF the PAIR OF current_index AND current_sum_value IS A KEY IN memo_dictionary
                RETURN the VALUE AT the PAIR OF current_index AND current_sum_value IN memo_dictionary
            END IF

            SET positive_result TO the RESULT OF dfs WITH current_index INCREMENTED BY one, current_sum_value PLUS element at position current_index OF list_of_numbers, AND memo_dictionary
            SET negative_result TO the RESULT OF dfs WITH current_index INCREMENTED BY one, current_sum_value MINUS element at position current_index OF list_of_numbers, AND memo_dictionary

            SET the VALUE AT the PAIR OF current_index AND current_sum_value IN memo_dictionary TO positive_result PLUS negative_result

            RETURN the VALUE AT the PAIR OF current_index AND current_sum_value IN memo_dictionary
        END FUNCTION

        RETURN the RESULT OF dfs WITH zero AS current_index, zero AS current_sum_value, AND an EMPTY dictionary AS memo_dictionary
    END FUNCTION
END CLASS
