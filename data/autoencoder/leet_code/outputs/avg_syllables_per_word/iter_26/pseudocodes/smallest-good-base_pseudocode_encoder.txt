CLASS Solution
    FUNCTION smallestGoodBase(n)
        SET n TO the INTEGER VALUE OF n
        SET maximum_length TO the INTEGER VALUE OF the LOGARITHM BASE two OF n PLUS one
        FOR m FROM maximum_length DOWN TO two INCLUSIVE
            SET candidate_base TO the INTEGER VALUE OF n RAISED TO THE POWER OF one DIVIDED BY (m MINUS one)
            IF candidate_base LESS THAN two
                CONTINUE TO the NEXT ITERATION
            END IF
            SET geometric_sum TO the INTEGER VALUE OF candidate_base RAISED TO THE POWER OF m MINUS one DIVIDED BY candidate_base MINUS one
            IF geometric_sum EQUALS n
                RETURN the STRING REPRESENTATION OF candidate_base
            END IF
        END FOR
        RETURN the STRING REPRESENTATION OF n MINUS one
    END FUNCTION
END CLASS
