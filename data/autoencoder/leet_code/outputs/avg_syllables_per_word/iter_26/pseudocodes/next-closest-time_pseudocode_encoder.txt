CLASS Solution
    FUNCTION nextClosestTime(time)
        SET digits TO the SET OF characters FROM time WITH the COLON CHARACTER REMOVED
        ASSIGN hour TO the INTEGER VALUE OF the substring FROM position zero TO position one OF time
        ASSIGN minute TO the INTEGER VALUE OF the substring FROM position three TO position four OF time

        FUNCTION generate_times(set_of_digits)
            SET times TO an empty LIST
            FOR each h1 IN set_of_digits
                FOR each h2 IN set_of_digits
                    FOR each m1 IN set_of_digits
                        FOR each m2 IN set_of_digits
                            ASSIGN hour_value TO the INTEGER VALUE OF the CONCATENATION OF h1 AND h2
                            ASSIGN minute_value TO the INTEGER VALUE OF the CONCATENATION OF m1 AND m2
                            IF hour_value IS LESS THAN twenty FOUR AND minute_value IS LESS THAN sixty
                                APPEND the TUPLE OF hour_value AND minute_value TO times
                            END IF
                        END FOR
                    END FOR
                END FOR
            END FOR
            RETURN the SORTED times LIST
        END FUNCTION

        ASSIGN all_times TO generate_times(digits)

        ASSIGN current_time_index TO the POSITION OF the TUPLE OF hour AND minute IN all_times

        IF current_time_index PLUS one IS LESS THAN the LENGTH OF all_times
            ASSIGN next_hour TO the FIRST ELEMENT OF the ELEMENT AT position current_time_index PLUS one OF all_times
            ASSIGN next_minute TO the SECOND ELEMENT OF the ELEMENT AT position current_time_index PLUS one OF all_times
        ELSE
            ASSIGN next_hour TO the FIRST ELEMENT OF the FIRST ELEMENT OF all_times
            ASSIGN next_minute TO the SECOND ELEMENT OF the FIRST ELEMENT OF all_times
        END IF

        RETURN the STRING REPRESENTATION OF next_hour WITH two DIGITS FOLLOWED BY a COLON FOLLOWED BY the STRING REPRESENTATION OF next_minute WITH two DIGITS
    END FUNCTION
END CLASS
