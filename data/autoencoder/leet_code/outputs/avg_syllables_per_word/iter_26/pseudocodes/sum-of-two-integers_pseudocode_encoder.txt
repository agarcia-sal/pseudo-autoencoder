CLASS Solution
    FUNCTION getSum(parameter a, parameter b)
        SET MAX TO the maximum value of a thirty two bits integer
        SET mask TO the value to extract thirty two bits
        
        WHILE b NOT EQUALS zero HOLDS
            SET carry TO the bitwise AND of a AND b MULTIPLIED BY two
            SET a TO the bitwise XOR of a AND b AND mask
            SET b TO the bitwise AND of carry AND mask
        END WHILE
        
        IF a LESS THAN OR EQUAL TO MAX THEN
            RETURN a
        ELSE
            RETURN the bitwise complement of the bitwise XOR of a AND mask
        END IF
    END FUNCTION
END CLASS
