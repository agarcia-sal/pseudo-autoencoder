CLASS Solution
    FUNCTION numWays(steps, arrLen)
        SET MOD TO the number obtained by raising ten TO the power of nine PLUS seven
        SET max_index TO the MINIMUM OF the INTEGER DIVISION OF steps BY two AND arrLen MINUS one
        
        SET dp TO a list of zeros WITH length max_index PLUS one
        SET the first element at position zero OF dp TO one
        
        FOR each repetition FROM the first TO steps TIMES INCLUSIVE
            SET prev_dp TO a copy OF dp
            FOR each index i FROM zero TO max_index INCLUSIVE
                SET the element at position i OF dp TO the element at position i OF prev_dp
                IF i GREATER THAN zero THEN
                    SET the element at position i OF dp TO the sum of the element at position i OF dp AND the element at position i MINUS one OF prev_dp MODULO MOD
                END IF
                IF i LESS THAN max_index THEN
                    SET the element at position i OF dp TO the sum of the element at position i OF dp AND the element at position i PLUS one OF prev_dp MODULO MOD
                END IF
            END FOR
        END FOR
        
        RETURN the element at position zero OF dp
    END FUNCTION
END CLASS
