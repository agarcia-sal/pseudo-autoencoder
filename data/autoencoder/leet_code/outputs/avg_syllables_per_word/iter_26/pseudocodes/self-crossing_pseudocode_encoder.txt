CLASS Solution
    FUNCTION isSelfCrossing(distance)
        SET number_of_distances TO the LENGTH OF distance
        IF number_of_distances LESS THAN four THEN
            RETURN False
        END IF

        FOR index FROM three TO number_of_distances MINUS one INCLUSIVE
            IF element at position index OF distance GREATER THAN OR EQUAL TO element at position index MINUS two OF distance AND element at position index MINUS one OF distance LESS THAN OR EQUAL TO element at position index MINUS three OF distance THEN
                RETURN True
            END IF

            IF index GREATER THAN OR EQUAL TO four AND element at position index MINUS one OF distance EQUALS element at position index MINUS three OF distance AND element at position index PLUS element at position index MINUS four OF distance GREATER THAN OR EQUAL TO element at position index MINUS two OF distance THEN
                RETURN True
            END IF

            IF index GREATER THAN OR EQUAL TO five AND element at position index MINUS two OF distance GREATER THAN OR EQUAL TO element at position index MINUS four OF distance AND element at position index MINUS three OF distance GREATER THAN OR EQUAL TO element at position index MINUS one OF distance AND element at position index MINUS one OF distance PLUS element at position index MINUS five OF distance GREATER THAN OR EQUAL TO element at position index MINUS three OF distance AND element at position index PLUS element at position index MINUS four OF distance GREATER THAN OR EQUAL TO element at position index MINUS two OF distance THEN
                RETURN True
            END IF
        END FOR

        RETURN False
    END FUNCTION
END CLASS
