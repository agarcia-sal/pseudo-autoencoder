CLASS Solution
    FUNCTION findReplaceString(s, indices, sources, targets)
        SET replacements TO a list of tuples combining elements AT matching positions FROM indices, sources, and targets SORTED BY the element at position zero OF each tuple
        SET result TO an empty list
        SET prev_end TO zero
        FOR each tuple (index, source, target) IN replacements
            APPEND the substring FROM position prev_end TO position index MINUS one OF string s TO result
            IF the substring FROM position index TO the position index PLUS the LENGTH OF source MINUS one OF string s EQUALS source
                APPEND target TO result
            ELSE
                APPEND the substring FROM position index TO the position index PLUS the LENGTH OF source MINUS one OF string s TO result
            END IF
            SET prev_end TO index PLUS the LENGTH OF source
        END FOR
        APPEND the substring FROM position prev_end TO the end OF string s TO result
        RETURN the concatenation OF all elements IN list result
    END FUNCTION
END CLASS
