CLASS Solution
    FUNCTION maxCoins(parameter nums AS List of integers) RETURNS integer
        SET nums TO the LIST CONSISTING OF the element one PLUS THE elements OF nums PLUS THE element one
        SET n TO the LENGTH OF nums

        SET dp TO a two dimensional LIST of size n by n FILLED WITH zero

        FOR length FROM two TO n MINUS one INCLUSIVE
            FOR i FROM zero TO n MINUS length MINUS one INCLUSIVE
                SET j TO i PLUS length
                FOR k FROM i PLUS one TO j MINUS one INCLUSIVE
                    SET coins TO the element at position i OF nums MULTIPLIED BY the element at position k OF nums MULTIPLIED BY the element at position j OF nums PLUS the element at position i AND k OF dp PLUS the element at position k AND j OF dp
                    IF the element at position i AND j OF dp LESS THAN coins
                        SET the element at position i AND j OF dp TO coins
                    END IF
                END FOR
            END FOR
        END FOR

        RETURN the element at position zero AND n MINUS one OF dp
    END FUNCTION
END CLASS
