CLASS Solution
    FUNCTION findRadius(houses, heaters)
        SORT houses IN ASCENDING ORDER
        SORT heaters IN ASCENDING ORDER
        
        SET min_radius TO zero
        SET j TO zero
        
        FOR each house IN houses
            WHILE j IS LESS THAN THE LENGTH OF heaters MINUS one AND
                  THE ABSOLUTE VALUE OF element at position j OF heaters MINUS house IS GREATER THAN OR EQUAL TO
                  THE ABSOLUTE VALUE OF element at position j PLUS one OF heaters MINUS house
                INCREMENT j BY one
            END WHILE
            
            SET min_radius TO THE GREATER VALUE BETWEEN min_radius AND THE ABSOLUTE VALUE OF element at position j OF heaters MINUS house
        END FOR
        
        RETURN min_radius
    END FUNCTION
END CLASS
