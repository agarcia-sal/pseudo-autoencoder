CLASS Solution
    FUNCTION subsets(nums)
        IF nums EQUALS empty list HOLDS
            RETURN list CONTAINING empty list
        END IF

        SET subsets_without_first TO CALL subsets WITH argument as the SUBLIST OF nums FROM position one TO THE END

        SET subsets_with_first TO empty list
        FOR each subset IN subsets_without_first
            SET new_subset TO list CONTAINING element at position zero OF nums FOLLOWED BY subset
            APPEND new_subset TO subsets_with_first
        END FOR

        RETURN list CONCATENATION OF subsets_without_first AND subsets_with_first
    END FUNCTION
END CLASS
