CLASS Solution
    FUNCTION numDecodings(string s)
        IF s IS EMPTY OR the element at position zero OF string s EQUALS the character zero
            RETURN zero
        END IF

        SET list dp TO a list of zeroes with LENGTH EQUAL TO the LENGTH OF string s PLUS one

        SET the element at position zero OF list dp TO one

        SET the element at position one OF list dp TO one

        FOR index i FROM two TO the LENGTH OF string s INCLUSIVE
            IF the element at position i MINUS one OF string s NOT EQUALS the character zero
                INCREMENT the element at position i OF list dp BY the element at position i MINUS one OF list dp
            END IF

            SET two_digit_number TO the natural number represented BY the substring FROM position i MINUS two TO position i EXCLUSIVE OF string s

            IF two_digit_number GREATER THAN OR EQUAL TO ten AND two_digit_number LESS THAN OR EQUAL TO twenty six
                INCREMENT the element at position i OF list dp BY the element at position i MINUS two OF list dp
            END IF
        END FOR

        RETURN the last element OF list dp
    END FUNCTION
END CLASS
