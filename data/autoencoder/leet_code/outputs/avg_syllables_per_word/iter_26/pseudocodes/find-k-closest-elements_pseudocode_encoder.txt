CLASS Solution
    FUNCTION findClosestElements(arr, k, x)
        SET left TO the POSITION OF the LEFTMOST PLACE WHERE x WOULD BE INSERTED TO KEEP arr SORTED MINUS one
        SET right TO left PLUS one
        
        WHILE right MINUS left MINUS one IS LESS THAN k HOLDS
            IF left EQUALS negative one OR right IS LESS THAN the LENGTH OF arr AND the ABSOLUTE VALUE OF the ELEMENT AT POSITION right OF arr MINUS x IS LESS THAN the ABSOLUTE VALUE OF the ELEMENT AT POSITION left OF arr MINUS x
                INCREMENT right BY one
            ELSE
                DECREMENT left BY one
            END IF
        END WHILE
        
        RETURN the SUBSEQUENCE FROM POSITION left PLUS one TO POSITION right EXCLUSIVE OF arr
    END FUNCTION
END CLASS
