CLASS Solution
    FUNCTION minimumDifference(list_of_numbers)
        SET n TO the LENGTH OF list_of_numbers DIVIDED BY three
        
        INITIALIZE empty list min_heap
        INITIALIZE list min_sums WITH zeroes of LENGTH two TIMES n PLUS one
        
        FOR i FROM zero TO two TIMES n MINUS one INCLUSIVE
            PUSH NEGATIVE of element at position i OF list_of_numbers INTO min_heap using heap insertion
            IF the LENGTH OF min_heap IS GREATER THAN n
                POP the SMALLEST element FROM min_heap using heap removal
            END IF
            IF the LENGTH OF min_heap EQUALS n
                SET element at position i OF min_sums TO NEGATIVE of the SUM OF all elements IN min_heap
            END IF
        END FOR
        
        INITIALIZE empty list max_heap
        INITIALIZE list max_sums WITH zeroes of LENGTH two TIMES n PLUS one
        
        FOR i FROM three TIMES n MINUS one DOWN TO n INCLUSIVE IN DECREMENTAL STEPS OF one
            PUSH element at position i OF list_of_numbers INTO max_heap using heap insertion
            IF the LENGTH OF max_heap IS GREATER THAN n
                POP the SMALLEST element FROM max_heap using heap removal
            END IF
            IF the LENGTH OF max_heap EQUALS n
                SET element at position i OF max_sums TO the SUM OF all elements IN max_heap
            END IF
        END FOR
        
        SET min_diff TO infinity
        
        FOR i FROM n MINUS one TO two Times n MINUS one INCLUSIVE
            SET min_diff TO the SMALLER OF min_diff AND the DIFFERENCE BETWEEN element at position i OF min_sums AND element at position i PLUS one OF max_sums
        END FOR
        
        RETURN min_diff
    END FUNCTION
END CLASS
