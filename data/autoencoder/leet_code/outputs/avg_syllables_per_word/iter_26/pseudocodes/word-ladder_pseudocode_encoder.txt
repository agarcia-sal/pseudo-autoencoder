CLASS Solution
    FUNCTION ladderLength(beginWord, endWord, wordList)
        IF endWord NOT IN wordList
            RETURN zero
        END IF
        
        SET word_dict TO a new default dictionary WHERE each key MAPS TO an empty list
        SET word_length TO the LENGTH OF beginWord
        
        FOR each word IN wordList
            FOR index FROM zero TO word_length MINUS one INCLUSIVE
                SET intermediate_key TO the CONCATENATION OF the SUBSTRING FROM zero TO index MINUS one OF word PLUS the ASTERISK SYMBOL PLUS the SUBSTRING FROM index PLUS one TO the END OF word
                APPEND word TO the LIST at key intermediate_key IN word_dict
            END FOR
        END FOR
        
        SET queue TO a new double ended queue INITIALIZED WITH a tuple CONTAINING beginWord AND one
        SET visited TO a new set INITIALIZED WITH beginWord
        
        WHILE the queue HOLDS
            REMOVE and RETURN the first element FROM queue AND ASSIGN the first part TO current_word AND the second part TO current_length
            
            FOR index FROM zero TO word_length MINUS one INCLUSIVE
                SET intermediate_word TO the CONCATENATION OF the SUBSTRING FROM zero TO index MINUS one OF current_word PLUS the ASTERISK SYMBOL PLUS the SUBSTRING FROM index PLUS one TO the END OF current_word
                
                FOR each word IN the LIST at key intermediate_word IN word_dict
                    IF word EQUALS endWord
                        RETURN current_length PLUS one
                    END IF
                    IF word NOT IN visited
                        ADD word TO visited
                        APPEND a tuple CONTAINING word AND current_length PLUS one TO queue
                    END IF
                END FOR
            END FOR
        END WHILE
        
        RETURN zero
    END FUNCTION
END CLASS
