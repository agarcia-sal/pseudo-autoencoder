CLASS Solution
    FUNCTION largestPalindrome(number_n)
        IF number_n EQUALS one
            RETURN nine
        END IF

        SET upper_limit TO ten RAISED TO THE POWER OF number_n MINUS one
        SET lower_limit TO ten RAISED TO THE POWER OF number_n MINUS one

        SET max_palindrome TO zero

        FOR number_i FROM upper_limit DOWN TO lower_limit INCLUSIVE
            FOR number_j FROM number_i DOWN TO lower_limit INCLUSIVE
                SET product TO number_i MULTIPLIED BY number_j
                IF product LESS THAN OR EQUAL TO max_palindrome
                    BREAK
                END IF
                SET str_product TO the string representation OF product
                IF str_product EQUALS the string obtained by reversing str_product
                    SET max_palindrome TO product
                END IF
            END FOR
        END FOR

        RETURN the remainder AFTER DIVIDING max_palindrome BY one thousand three hundred thirty seven
    END FUNCTION
END CLASS
