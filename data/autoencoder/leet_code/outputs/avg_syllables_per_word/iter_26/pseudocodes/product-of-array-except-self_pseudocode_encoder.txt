CLASS Solution
    FUNCTION productExceptSelf(nums)
        SET length_of_nums TO the LENGTH OF nums
        SET answer TO a list of the value one REPEATED length_of_nums TIMES
        
        SET left_product TO one
        FOR index FROM zero TO length_of_nums MINUS one INCLUSIVE
            SET element at position index OF answer TO left_product
            SET left_product TO left_product MULTIPLIED BY element at position index OF nums
        END FOR
        
        SET right_product TO one
        FOR index FROM length_of_nums MINUS one DOWN TO zero INCLUSIVE
            SET element at position index OF answer TO element at position index OF answer MULTIPLIED BY right_product
            SET right_product TO right_product MULTIPLIED BY element at position index OF nums
        END FOR
        
        RETURN answer
    END FUNCTION
END CLASS
