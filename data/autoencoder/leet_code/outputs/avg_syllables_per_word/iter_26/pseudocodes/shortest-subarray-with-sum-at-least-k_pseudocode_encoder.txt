CLASS Solution
    FUNCTION shortestSubarray(list_of_numbers, threshold_k)
        SET prefix_sum_array TO list containing zero
        FOR each number IN list_of_numbers
            SET last_element_of_prefix_sum TO element at position the LENGTH OF prefix_sum_array MINUS one OF prefix_sum_array
            SET new_sum TO last_element_of_prefix_sum PLUS number
            APPEND new_sum TO prefix_sum_array
        END FOR

        ASSIGN empty double ended queue TO dq
        SET minimum_length TO infinite value

        FOR each index_and_value IN enumeration OF prefix_sum_array
            ASSIGN index TO the index part OF index_and_value
            ASSIGN current_sum TO the value part OF index_and_value

            WHILE dq is not empty AND current_sum MINUS element at position the front element OF dq OF prefix_sum_array GREATER THAN OR EQUAL TO threshold_k HOLDS
                SET candidate_length TO index MINUS element removed from the front OF dq
                IF candidate_length LESS THAN minimum_length
                    SET minimum_length TO candidate_length
                END IF
            END WHILE

            WHILE dq is not empty AND current_sum LESS THAN OR EQUAL TO element at position the last element OF dq OF prefix_sum_array HOLDS
                REMOVE element from the back OF dq
            END WHILE

            APPEND index TO dq
        END FOR

        IF minimum_length IS EQUAL TO infinite value
            RETURN negative one
        ELSE
            RETURN minimum_length
        END IF
    END FUNCTION
END CLASS
