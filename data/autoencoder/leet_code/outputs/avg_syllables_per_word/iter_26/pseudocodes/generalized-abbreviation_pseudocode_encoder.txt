CLASS Solution
    FUNCTION generateAbbreviations(word)
        FUNCTION backtrack(index, path, count)
            IF index EQUALS the LENGTH OF word
                IF count GREATER THAN zero
                    APPEND the TEXT FORM OF count TO path
                END IF
                APPEND the CONCATENATION OF all ELEMENTS IN path TO result
            ELSE
                IF count GREATER THAN zero
                    CALL backtrack WITH index INCREMENTED BY one, path EXTENDED BY the TEXT FORM OF count AND the ELEMENT AT position index OF word, AND zero AS count
                ELSE
                    CALL backtrack WITH index INCREMENTED BY one, path EXTENDED BY the ELEMENT AT position index OF word, AND zero AS count
                END IF
                CALL backtrack WITH index INCREMENTED BY one, path UNCHANGED, AND count INCREMENTED BY one
            END IF
        END FUNCTION
        SET result TO an EMPTY LIST
        CALL backtrack WITH zero AS index, an EMPTY LIST AS path, AND zero AS count
        RETURN result
    END FUNCTION
END CLASS
