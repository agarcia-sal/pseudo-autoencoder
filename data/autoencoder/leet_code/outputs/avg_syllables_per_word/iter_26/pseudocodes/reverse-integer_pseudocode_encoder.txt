CLASS Solution
    FUNCTION reverse(parameter x)
        SET sign TO negative one IF x LESS THAN zero ELSE positive one
        SET reversed_x TO convert TO integer THE reverse OF the string OF the absolute value OF x
        SET result TO sign MULTIPLIED BY reversed_x
        IF result LESS THAN negative two TO THE POWER OF thirty ONE OR result GREATER THAN positive two TO THE POWER OF thirty ONE MINUS one
            RETURN zero
        END IF
        RETURN result
    END FUNCTION
END CLASS
