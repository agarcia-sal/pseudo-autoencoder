CLASS Solution
    FUNCTION makeArrayIncreasing(arr1 AS list OF integers, arr2 AS list OF integers) RETURNS integer
        CALL the SORT METHOD ON arr2 TO arrange elements IN ascending order

        SET dp TO a dictionary WITH key AS minus one AND value AS zero TO represent an empty subsequence

        FOR each number IN arr1
            SET new_dp TO an empty dictionary

            FOR each last_number AND operations IN dp items
                IF number GREATER THAN last_number THEN
                    IF number is NOT a key IN new_dp THEN
                        SET the value for number IN new_dp TO INFINITY
                    END IF
                    SET the value for number IN new_dp TO the MINIMUM OF the value for number IN new_dp AND operations
                END IF

                SET index TO the result of finding the insertion position TO THE RIGHT of last_number IN arr2 USING binary search

                IF index LESS THAN the LENGTH OF arr2 THEN
                    IF element at position index OF arr2 is NOT a key IN new_dp THEN
                        SET the value for element at position index OF arr2 IN new_dp TO INFINITY
                    END IF
                    SET the value for element at position index OF arr2 IN new_dp TO the MINIMUM OF the value for element at position index OF arr2 IN new_dp AND the sum of operations AND one
                END IF
            END FOR

            IF new_dp IS empty THEN
                RETURN minus one
            END IF

            SET dp TO new_dp
        END FOR

        RETURN the MINIMUM value AMONG all values IN dp
    END FUNCTION
END CLASS
