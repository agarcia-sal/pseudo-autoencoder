CLASS Solution
    FUNCTION repeatedStringMatch(a AS string, b AS string)
        SET minimum_repetitions TO the SMALLEST INTEGER GREATER THAN OR EQUAL TO the LENGTH OF b DIVIDED BY the LENGTH OF a
        IF the string b IS A SUBSTRING OF the STRING a REPEATED minimum_repetitions TIMES HOLDS
            RETURN minimum_repetitions
        END IF
        IF the string b IS A SUBSTRING OF the STRING a REPEATED the SUM OF minimum_repetitions AND one TIMES HOLDS
            RETURN the SUM OF minimum_repetitions AND one
        END IF
        RETURN NEGATIVE one
    END FUNCTION
END CLASS
