CLASS Solution
    FUNCTION minSteps(n)
        IF n EQUALS one THEN
            RETURN zero
        END IF

        SET operations TO zero
        SET factor TO two

        WHILE n GREATER THAN one HOLDS
            WHILE n MODULO factor EQUALS zero HOLDS
                INCREMENT operations BY factor
                SET n TO n DIVIDED BY factor
            END WHILE
            INCREMENT factor BY one
        END WHILE

        RETURN operations
    END FUNCTION
END CLASS
