CLASS Solution
    FUNCTION carFleet(parameter target, parameter position, parameter speed)
        SET list_of_time_to_target TO an empty list
        SET list_of_position_and_speed TO a list formed by pairing each element of position with the corresponding element of speed
        SORT list_of_position_and_speed IN descending order BASED ON the first element of each pair
        FOR each pair OF position_value AND speed_value IN list_of_position_and_speed
            SET time_needed TO the result of target MINUS position_value DIVIDED BY speed_value
            APPEND time_needed TO list_of_time_to_target
        END FOR

        SET fleets TO zero
        SET current_maximum_time TO zero

        FOR each time_value IN list_of_time_to_target
            IF time_value GREATER THAN current_maximum_time THEN
                INCREMENT fleets BY one
                SET current_maximum_time TO time_value
            END IF
        END FOR

        RETURN fleets
    END FUNCTION
END CLASS
