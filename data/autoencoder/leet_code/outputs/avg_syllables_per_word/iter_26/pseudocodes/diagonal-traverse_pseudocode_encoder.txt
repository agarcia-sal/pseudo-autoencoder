CLASS Solution
    FUNCTION findDiagonalOrder(mat)
        IF mat IS empty OR the LENGTH OF mat IS zero OR the LENGTH OF element at position zero OF mat IS zero
            RETURN an EMPTY LIST
        END IF

        SET m TO the LENGTH OF mat
        SET n TO the LENGTH OF element at position zero OF mat
        SET result TO an EMPTY LIST
        SET r TO zero
        SET c TO zero
        SET direction TO one

        FOR each iteration FROM zero TO the product OF m AND n MINUS one INCLUSIVE
            APPEND element at position r OF mat AT position c TO result

            IF direction EQUALS one
                SET new_r TO r MINUS one
                SET new_c TO c PLUS one
            ELSE
                SET new_r TO r PLUS one
                SET new_c TO c MINUS one
            END IF

            IF new_r LESS THAN zero OR new_r GREATER THAN OR EQUAL TO m OR new_c LESS THAN zero OR new_c GREATER THAN OR EQUAL TO n
                IF direction EQUALS one
                    IF new_c EQUALS n
                        SET new_r TO r PLUS one
                        SET new_c TO c
                    ELSE
                        SET new_r TO zero
                        SET new_c TO c PLUS one
                    END IF
                ELSE
                    IF new_r EQUALS m
                        SET new_r TO r
                        SET new_c TO c PLUS one
                    ELSE
                        SET new_r TO r PLUS one
                        SET new_c TO zero
                    END IF
                END IF
                SET direction TO NEGATIVE direction
            END IF

            SET r TO new_r
            SET c TO new_c
        END FOR

        RETURN result
    END FUNCTION
END CLASS
