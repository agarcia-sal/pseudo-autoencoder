CLASS Solution
    FUNCTION lengthOfLongestSubstring(parameter s OF TYPE string)
        SET char_index_map TO empty mapping
        SET max_length TO zero
        SET start TO zero
        
        FOR each index end AND each character char IN parameter s WITH their indexes
            IF char EXISTS AS A KEY IN char_index_map AND the VALUE IN char_index_map AT key char IS GREATER THAN OR EQUAL TO start
                SET start TO the VALUE IN char_index_map AT key char PLUS one
            END IF
            SET the VALUE IN char_index_map AT key char TO end
            SET max_length TO the GREATER OF max_length AND the DIFFERENCE BETWEEN end AND start PLUS one
        END FOR
        
        RETURN max_length
    END FUNCTION
END CLASS
