CLASS Solution
    FUNCTION smallestDistancePair(nums, k)
        CALL nums SORT IN ASCENDING ORDER
        SET n TO the LENGTH OF nums

        FUNCTION count(mid)
            SET count TO zero
            SET left TO zero
            FOR each right FROM zero TO n MINUS one INCLUSIVE
                WHILE the ELEMENT AT POSITION right OF nums MINUS the ELEMENT AT POSITION left OF nums IS GREATER THAN mid HOLDS
                    INCREMENT left BY one
                END WHILE
                INCREMENT count BY right MINUS left
            END FOR
            RETURN count
        END FUNCTION

        SET left TO zero
        SET right TO the ELEMENT AT THE LAST POSITION OF nums MINUS the ELEMENT AT THE FIRST POSITION OF nums

        WHILE left IS LESS THAN right HOLDS
            SET mid TO left PLUS right DIVIDED BY two USING INTEGER DIVISION
            IF the RESULT OF count CALLED WITH mid IS LESS THAN k
                SET left TO mid PLUS one
            ELSE
                SET right TO mid
            END IF
        END WHILE

        RETURN left
    END FUNCTION
END CLASS
