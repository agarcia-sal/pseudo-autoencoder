CLASS Solution
    FUNCTION fullJustify(words, maxWidth)
        FUNCTION justify_line(line, num_of_letters, is_last_line)
            IF the LENGTH OF line EQUALS one OR is_last_line HOLDS
                RETURN the concatenation of all elements in line WITH single space BETWEEN each PLUS spaces REPEATED maxWidth MINUS num_of_letters MINUS the LENGTH OF line PLUS one TIMES
            ELSE
                SET total_spaces TO maxWidth MINUS num_of_letters
                SET num_of_gaps TO the LENGTH OF line MINUS one
                SET spaces_between_words TO total_spaces DIVIDED BY num_of_gaps
                SET extra_spaces TO total_spaces MODULO num_of_gaps
                
                FOR i FROM zero TO num_of_gaps MINUS one INCLUSIVE
                    IF i LESS THAN extra_spaces HOLDS
                        APPEND spaces REPEATED spaces_between_words PLUS one TIMES TO element at position i OF line
                    ELSE
                        APPEND spaces REPEATED spaces_between_words TIMES TO element at position i OF line
                    END IF
                END FOR
                
                RETURN the concatenation of all elements in line WITHOUT ANY SEPARATION
            END IF
        END FUNCTION
        
        SET result TO empty list
        SET line TO empty list
        SET num_of_letters TO zero
        
        FOR each word IN words
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth HOLDS
                APPEND the RESULT OF justify_line WITH parameters line, num_of_letters, false TO result
                SET line TO empty list
                SET num_of_letters TO zero
            END IF
            
            APPEND word TO line
            INCREMENT num_of_letters BY the LENGTH OF word
        END FOR
        
        APPEND the RESULT OF justify_line WITH parameters line, num_of_letters, true TO result
        
        RETURN result
    END FUNCTION
END CLASS
