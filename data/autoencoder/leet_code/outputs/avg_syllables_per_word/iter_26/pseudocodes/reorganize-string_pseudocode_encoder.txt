CLASS Solution
    FUNCTION reorganizeString(s)
        SET count TO the frequency COUNT OF each CHARACTER IN s
        SET max_heap TO the LIST OF PAIRS consisting OF the NEGATIVE OF frequency AND the CHARACTER FOR each CHARACTER AND frequency IN the ITEMS OF count
        PERFORM heapify OPERATION ON max_heap
        
        SET prev_char TO none
        SET prev_freq TO zero
        SET result TO empty LIST
        
        WHILE max_heap IS NOT empty OR prev_freq IS LESS THAN zero HOLDS
            IF max_heap IS empty AND prev_freq IS LESS THAN zero HOLDS
                RETURN the empty STRING
            END IF
            
            SET freq AND char TO the ELEMENT REMOVED FROM THE HEAP max_heap
            APPEND char TO result
            
            SET freq TO freq PLUS one
            
            IF prev_freq IS LESS THAN zero HOLDS
                PUSH the PAIR consisting OF prev_freq AND prev_char INTO THE HEAP max_heap
            END IF
            
            SET prev_char TO char
            SET prev_freq TO freq
        END WHILE
        
        RETURN the STRING FORMED BY JOINING all ELEMENTS OF result
    END FUNCTION
END CLASS
