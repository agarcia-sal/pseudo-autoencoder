CLASS Solution
    FUNCTION minimumDeviation(nums AS List)
        SET max_heap TO an empty list
        SET min_value TO positive infinity
        FOR each num IN nums
            IF element at position zero OF num MODULO two EQUALS one
                SET num TO num MULTIPLIED BY two
            END IF
            SET min_value TO the smaller BETWEEN min_value AND num
            INSERT negative num INTO max_heap USING heap push operation
        END FOR
        SET min_deviation TO positive infinity
        WHILE element at position zero OF max_heap MODULO two EQUALS zero HOLDS
            SET max_value TO the negative of the result of heap pop from max_heap
            SET min_deviation TO the smaller BETWEEN min_deviation AND max_value MINUS min_value
            SET max_value TO the quotient of max_value DIVIDED BY two using integer division
            SET min_value TO the smaller BETWEEN min_value AND max_value
            INSERT negative max_value INTO max_heap USING heap push operation
        END WHILE
        SET min_deviation TO the smaller BETWEEN min_deviation AND the negative of element at position zero OF max_heap MINUS min_value
        RETURN min_deviation
    END FUNCTION
END CLASS
