CLASS Solution
    FUNCTION longestMountain(arr)
        SET n TO the LENGTH OF arr
        SET longest TO zero
        
        SET i TO one
        WHILE i LESS THAN n MINUS one HOLDS
            IF element at position i MINUS one OF arr LESS THAN element at position i OF arr AND element at position i OF arr GREATER THAN element at position i PLUS one OF arr
                SET left TO i MINUS one
                SET right TO i PLUS one
                
                WHILE left GREATER THAN zero AND element at position left MINUS one OF arr LESS THAN element at position left OF arr HOLDS
                    DECREMENT left BY one
                END WHILE
                
                WHILE right LESS THAN n MINUS one AND element at position right OF arr GREATER THAN element at position right PLUS one OF arr HOLDS
                    INCREMENT right BY one
                END WHILE
                
                SET longest TO the GREATER OF longest AND right MINUS left PLUS one
                
                SET i TO right
            ELSE
                INCREMENT i BY one
            END IF
        END WHILE
        
        RETURN longest
    END FUNCTION
END CLASS
