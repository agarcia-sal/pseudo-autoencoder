CLASS Solution
    FUNCTION movesToChessboard(board)
        SET N TO the LENGTH OF board
        
        SET rowSum TO zero
        SET colSum TO zero
        SET rowSwap TO zero
        SET colSwap TO zero
        
        FOR each index i FROM zero TO N MINUS one INCLUSIVE
            FOR each index j FROM zero TO N MINUS one INCLUSIVE
                IF the ELEMENT at position i OF board EXCLUSIVE OR the ELEMENT at position zero OF the ELEMENT at position i OF board EXCLUSIVE OR the ELEMENT at position j OF the ELEMENT at position zero OF board EXCLUSIVE OR the ELEMENT at position zero OF the ELEMENT at position zero OF board EQUALS one
                    RETURN negative one
                END IF
            END FOR
            
            INCREMENT rowSum BY the ELEMENT at position zero OF the ELEMENT at position i OF board
            INCREMENT colSum BY the ELEMENT at position i OF the ELEMENT at position zero OF board
            
            IF the ELEMENT at position zero OF the ELEMENT at position i OF board EQUALS the REMAINDER OF i DIVIDED BY two
                INCREMENT rowSwap BY one
            END IF
            
            IF the ELEMENT at position i OF the ELEMENT at position zero OF board EQUALS the REMAINDER OF i DIVIDED BY two
                INCREMENT colSwap BY one
            END IF
        END FOR
        
        IF the INTEGER DIVISION of N BY two GREATER THAN rowSum OR rowSum GREATER THAN the INTEGER DIVISION of N PLUS one BY two
            RETURN negative one
        END IF
        
        IF the INTEGER DIVISION of N BY two GREATER THAN colSum OR colSum GREATER THAN the INTEGER DIVISION of N PLUS one BY two
            RETURN negative one
        END IF
        
        IF the REMAINDER OF N DIVIDED BY two EQUALS one
            IF the REMAINDER OF rowSwap DIVIDED BY two EQUALS one
                SET rowSwap TO N MINUS rowSwap
            END IF
            
            IF the REMAINDER OF colSwap DIVIDED BY two EQUALS one
                SET colSwap TO N MINUS colSwap
            END IF
        ELSE
            SET rowSwap TO the MINIMUM OF rowSwap AND N MINUS rowSwap
            SET colSwap TO the MINIMUM OF colSwap AND N MINUS colSwap
        END IF
        
        RETURN the INTEGER DIVISION OF the SUM of rowSwap AND colSwap BY two
    END FUNCTION
END CLASS
