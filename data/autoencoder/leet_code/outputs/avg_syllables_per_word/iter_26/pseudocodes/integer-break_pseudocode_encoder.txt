CLASS Solution
    FUNCTION integerBreak(parameter n)
        SET dynamic_programming_list TO a list of zeros WITH the LENGTH OF n PLUS one
        SET the element at position one OF dynamic_programming_list TO one
        FOR integer i FROM two TO n INCLUSIVE
            SET maximum_product TO zero
            FOR integer j FROM one TO i MINUS one INCLUSIVE
                SET candidate_one TO j MULTIPLIED BY the element at position i MINUS j OF dynamic_programming_list
                SET candidate_two TO j MULTIPLIED BY i MINUS j
                SET maximum_product TO the GREATEST VALUE AMONG maximum_product, candidate_one, AND candidate_two
            END FOR
            SET the element at position i OF dynamic_programming_list TO maximum_product
        END FOR
        RETURN the element at position n OF dynamic_programming_list
    END FUNCTION
END CLASS
