CLASS Solution
    FUNCTION findShortestWay(maze AS list of list of integers, ball AS list of integers, hole AS list of integers) RETURNS string
        SET directions TO a list containing the quadruples of zero and one with corresponding letters right down left up as the direction labels
        SET m TO the LENGTH OF maze
        SET n TO the LENGTH OF element at position zero OF maze
        ASSIGN start TO the TUPLE of ball
        ASSIGN hole TO the TUPLE of hole
        SET pq TO a list containing one triple of zero distance empty string path and start position
        SET visited TO an empty set
        
        WHILE pq HOLDS
            REMOVE the smallest element FROM pq AND ASSIGN distance TO dist path string TO path AND a tuple of coordinates TO x comma y
            
            IF the tuple of x and y EQUALS hole THEN
                RETURN path
            END IF
            
            IF the tuple of x and y IS IN visited THEN
                CONTINUE to next iteration
            END IF
            
            ADD the tuple of x and y TO visited
            
            FOR each triple of dx dy direction IN directions
                SET new_x TO x
                SET new_y TO y
                SET count TO zero
                
                WHILE the sum of new_x and dx IS GREATER THAN OR EQUAL TO zero AND the sum of new_x and dx IS LESS THAN m AND the sum of new_y and dy IS GREATER THAN OR EQUAL TO zero AND the sum of new_y and dy IS LESS THAN n AND the element at position sum of new_x and dx OF maze sees the element at position sum of new_y and dy EQUALS zero
                    INCREMENT new_x BY dx
                    INCREMENT new_y BY dy
                    INCREMENT count BY one
                    
                    IF the tuple of new_x and new_y EQUALS hole THEN
                        BREAK out of this WHILE loop
                    END IF
                END WHILE
                
                IF the tuple of new_x and new_y IS NOT IN visited THEN
                    ADD to pq the triple of the sum of dist and count the concatenation of path and direction and the tuple of new_x and new_y maintaining the heap property
                END IF
            END FOR
        END WHILE
        
        RETURN the literal string impossible
    END FUNCTION
END CLASS
