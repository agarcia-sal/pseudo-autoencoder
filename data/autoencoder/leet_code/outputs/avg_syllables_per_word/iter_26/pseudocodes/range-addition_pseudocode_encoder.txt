CLASS Solution
    FUNCTION getModifiedArray(length, updates)
        SET difference_array TO a list of zeros with the size of length PLUS one

        FOR each update_triplet IN updates
            ASSIGN start_index TO element at position zero OF update_triplet
            ASSIGN end_index TO element at position one OF update_triplet
            ASSIGN increment_value TO element at position two OF update_triplet
            INCREMENT element at position start_index OF difference_array BY increment_value

            IF end_index PLUS one LESS THAN length HOLDS
                DECREMENT element at position end_index PLUS one OF difference_array BY increment_value
            END IF
        END FOR

        SET result_array TO a list of zeros with the size of length
        SET element at position zero OF result_array TO element at position zero OF difference_array

        FOR index FROM one TO length MINUS one INCLUSIVE
            SET element at position index OF result_array TO element at position index MINUS one OF result_array PLUS element at position index OF difference_array
        END FOR

        RETURN result_array
    END FUNCTION
END CLASS
