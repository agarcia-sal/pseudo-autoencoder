CLASS Solution
    FUNCTION decodeString(s)
        SET stack TO empty list
        SET a TO empty sequence of letters
        SET d TO empty sequence of digits
        
        FOR each character IN s
            IF character IS a digit
                APPEND character TO d
            ELSE IF character IS a letter of the alphabet
                APPEND character TO a
            ELSE IF character IS the opening bracket symbol
                APPEND the pair consisting of a and the integer value represented by d TO stack
                SET a TO empty sequence of letters
                SET d TO empty sequence of digits
            ELSE IF character IS the closing bracket symbol
                SET pair TO the last element removed FROM stack
                SET p TO the first element of pair
                SET n TO the second element of pair
                SET a TO p CONCATENATED WITH a REPEATED n TIMES
            END IF
        END FOR
        
        RETURN a
    END FUNCTION
END CLASS
