CLASS ListNode
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values)
    IF values IS EQUAL TO none OR the LENGTH OF values IS EQUAL TO zero
        RETURN none
    END IF
    SET head TO A NEW ListNode WITH value AS the ELEMENT AT POSITION zero OF values
    SET p TO head
    FOR each val IN the SUBLIST OF values FROM position one TO the END
        SET node TO A NEW ListNode WITH value AS val
        SET p.next TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS EQUAL TO none AND p2 IS EQUAL TO none
        RETURN true
    END IF
    IF p1 IS EQUAL TO none OR p2 IS EQUAL TO none
        RETURN false
    END IF
    RETURN p1.val IS EQUAL TO p2.val AND is_same_list(p1.next, p2.next)
END FUNCTION

CLASS Solution
    FUNCTION removeNthFromEnd(self, head, n)
        SET dummy TO A NEW ListNode WITH value AS zero
        SET dummy.next TO head
        SET first TO dummy
        SET second TO dummy

        FOR i FROM zero TO n INCLUSIVE
            SET first TO first.next
        END FOR

        WHILE first IS NOT EQUAL TO none HOLDS
            SET first TO first.next
            SET second TO second.next
        END WHILE

        SET second.next TO the ELEMENT AT POSITION one AFTER second.next

        RETURN dummy.next
    END FUNCTION
END CLASS
