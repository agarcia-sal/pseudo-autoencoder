CLASS ListNode
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values)
    IF values IS empty
        RETURN none
    END IF
    SET head TO A NEW ListNode WITH element at position zero OF values
    SET p TO head
    FOR each val IN THE SUBLIST OF values FROM position one TO THE END
        SET node TO A NEW ListNode WITH val
        SET the next element OF p TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS none AND p2 IS none
        RETURN true
    END IF
    IF p1 IS none OR p2 IS none
        RETURN false
    END IF
    RETURN THE VALUE OF element val OF p1 EQUALS THE VALUE OF element val OF p2 
        AND is_same_list(THE VALUE OF element next OF p1, THE VALUE OF element next OF p2)
END FUNCTION

CLASS Solution
    FUNCTION insertionSortList(self, head)
        IF head IS none OR the next element OF head IS none
            RETURN head
        END IF

        SET dummy TO A NEW ListNode WITH value MINUS five thousand AND one
        SET the next element OF dummy TO head
        SET current TO the next element OF head
        SET the next element OF head TO none

        WHILE current IS NOT none
            SET next_node TO the next element OF current
            SET prev TO dummy

            WHILE the next element OF prev IS NOT none AND THE VALUE OF element val OF the next element OF prev IS LESS THAN THE VALUE OF element val OF current
                SET prev TO the next element OF prev
            END WHILE

            SET the next element OF current TO the next element OF prev
            SET the next element OF prev TO current
            SET current TO next_node
        END WHILE

        RETURN the next element OF dummy
    END FUNCTION
END CLASS
