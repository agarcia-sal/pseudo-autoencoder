CLASS Solution

    FUNCTION superEggDrop(parameter k, parameter n)
        SET dp TO list of zeros OF length k PLUS one
        SET moves TO zero

        WHILE element at position k OF dp IS LESS THAN parameter n HOLDS
            INCREMENT moves BY one

            FOR eggs FROM parameter k DOWN TO one INCLUSIVE
                SET element at position eggs OF dp TO one PLUS element at position eggs MINUS one OF dp PLUS element at position eggs OF dp
            END FOR
        END WHILE

        RETURN moves
    END FUNCTION

END CLASS
