CLASS Solution
    FUNCTION isValidPalindrome(string s, integer k) RETURNS boolean
        FUNCTION longest_palindromic_subseq(string s) RETURNS integer
            SET length_of_string TO the LENGTH OF s
            INITIALIZE two dimensional list dp TO a list of lists each containing zero repeated length_of_string times repeated length_of_string times
            
            FOR index i FROM zero TO length_of_string MINUS one INCLUSIVE
                SET element at position i AT element at position i OF dp TO one
            END FOR
            
            FOR length_value FROM two TO length_of_string INCLUSIVE
                FOR index i FROM zero TO length_of_string MINUS length_value INCLUSIVE
                    SET index j TO index i PLUS length_value MINUS one
                    IF element at position i OF s EQUALS element at position j OF s THEN
                        SET element at position i AT element at position j OF dp TO element at position i PLUS one AT element at position j MINUS one OF dp PLUS two
                    ELSE
                        SET element at position i AT element at position j OF dp TO the GREATER OF element at position i PLUS one AT element at position j OF dp AND element at position i AT element at position j MINUS one OF dp
                    END IF
                END FOR
            END FOR
            
            RETURN element at position zero AT element at position length_of_string MINUS one OF dp
        END FUNCTION
        
        SET longest_palindrome_length TO the RESULT OF longest_palindromic_subseq WITH PARAMETER s
        RETURN the LENGTH OF s MINUS longest_palindrome_length LESS THAN OR EQUAL TO k
    END FUNCTION
END CLASS
