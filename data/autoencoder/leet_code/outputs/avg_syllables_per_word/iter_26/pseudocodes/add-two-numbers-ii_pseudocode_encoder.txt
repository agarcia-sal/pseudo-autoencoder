CLASS ListNode
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values LIST)
    IF values IS NOT PRESENT
        RETURN none
    END IF
    SET head TO A NEW ListNode WITH THE ELEMENT at POSITION one OF values
    SET p TO head
    FOR each val IN THE SUBLIST OF values FROM POSITION two TO THE END
        SET node TO A NEW ListNode WITH val
        SET p.next TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS none AND p2 IS none
        RETURN true
    END IF
    IF p1 IS NOT PRESENT OR p2 IS NOT PRESENT
        RETURN false
    END IF
    RETURN THE EXPRESSION THAT THE ELEMENT val OF p1 EQUALS THE ELEMENT val OF p2 AND THE FUNCTION is_same_list CALLED WITH PARAMETERS THE ELEMENT next OF p1 AND THE ELEMENT next OF p2
END FUNCTION

CLASS Solution
    FUNCTION addTwoNumbers(l1 OPTIONAL ListNode, l2 OPTIONAL ListNode) RETURNS OPTIONAL ListNode

        FUNCTION pushIntoStack(node)
            SET stack TO AN EMPTY LIST
            WHILE node HOLDS
                APPEND THE ELEMENT val OF node TO stack
                SET node TO THE ELEMENT next OF node
            END WHILE
            RETURN stack
        END FUNCTION

        SET stack1 TO THE RESULT OF pushIntoStack WITH PARAMETER l1
        SET stack2 TO THE RESULT OF pushIntoStack WITH PARAMETER l2

        SET carry TO zero
        SET result TO none

        WHILE THE EXPRESSION THAT stack1 IS NOT EMPTY OR stack2 IS NOT EMPTY OR carry IS GREATER THAN zero HOLDS
            IF stack1 IS NOT EMPTY
                SET val1 TO THE LAST ELEMENT REMOVED FROM stack1
            ELSE
                SET val1 TO zero
            END IF
            IF stack2 IS NOT EMPTY
                SET val2 TO THE LAST ELEMENT REMOVED FROM stack2
            ELSE
                SET val2 TO zero
            END IF

            SET total TO THE SUM OF val1 PLUS val2 PLUS carry
            SET carry TO total DIVIDED BY ten
            SET digit TO total MODULO ten

            SET new_node TO A NEW ListNode WITH digit
            SET new_node.next TO result
            SET result TO new_node

        END WHILE

        RETURN result
    END FUNCTION
END CLASS
