CLASS Solution
    FUNCTION minCost(costs)
        IF costs EQUALS empty
            RETURN zero
        END IF

        SET previous red cost TO element at position zero OF element at position zero OF costs
        SET previous blue cost TO element at position one OF element at position zero OF costs
        SET previous green cost TO element at position two OF element at position zero OF costs

        FOR index FROM one TO the LENGTH OF costs MINUS one INCLUSIVE
            SET current red cost TO element at position index OF costs AT position zero PLUS THE MINIMUM OF previous blue cost AND previous green cost
            SET current blue cost TO element at position index OF costs AT position one PLUS THE MINIMUM OF previous red cost AND previous green cost
            SET current green cost TO element at position index OF costs AT position two PLUS THE MINIMUM OF previous red cost AND previous blue cost

            SET previous red cost TO current red cost
            SET previous blue cost TO current blue cost
            SET previous green cost TO current green cost
        END FOR

        RETURN THE MINIMUM OF previous red cost AND previous blue cost AND previous green cost
    END FUNCTION
END CLASS
