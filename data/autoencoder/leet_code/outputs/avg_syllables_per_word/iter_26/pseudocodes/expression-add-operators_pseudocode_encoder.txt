CLASS Solution
    FUNCTION addOperators(parameter num AS string, parameter target AS integer) RETURNS list of strings
        FUNCTION dfs(parameter index, parameter path, parameter value, parameter prev)
            IF index EQUALS the LENGTH OF num
                IF value EQUALS target
                    APPEND path TO result
                END IF
                RETURN
            END IF

            FOR i FROM index PLUS one TO the LENGTH OF num INCLUSIVE
                SET operand TO the substring FROM position index TO position i OF num
                IF the LENGTH OF operand GREATER THAN one AND the element at position zero OF operand EQUALS the character zero
                    CONTINUE
                END IF

                IF index EQUALS zero
                    CALL dfs WITH arguments i, operand, the numeric integer value OF operand, the numeric integer value OF operand
                ELSE
                    CALL dfs WITH arguments i, path CONCATENATED WITH the character plus CONCATENATED WITH operand, value PLUS the numeric integer value OF operand, the numeric integer value OF operand
                    CALL dfs WITH arguments i, path CONCATENATED WITH the character minus CONCATENATED WITH operand, value MINUS the numeric integer value OF operand, the negative of the numeric integer value OF operand
                    CALL dfs WITH arguments i, path CONCATENATED WITH the character asterisk CONCATENATED WITH operand, value MINUS prev PLUS prev MULTIPLIED BY the numeric integer value OF operand, prev MULTIPLIED BY the numeric integer value OF operand
                END IF
            END FOR
        END FUNCTION

        SET result TO an empty list
        CALL dfs WITH arguments zero, an empty string, zero, zero
        RETURN result
    END FUNCTION
END CLASS
