CLASS Solution
    FUNCTION jump(nums)
        IF the LENGTH OF nums LESS THAN OR EQUAL TO one HOLDS
            RETURN zero
        END IF

        SET jumps TO zero
        SET current_end TO zero
        SET farthest TO zero

        FOR index i FROM zero TO the LENGTH OF nums MINUS one MINUS one INCLUSIVE
            SET farthest TO the GREATER OF farthest AND index i PLUS element at position index i OF nums

            IF index i EQUALS current_end HOLDS
                INCREMENT jumps BY one
                SET current_end TO farthest

                IF current_end GREATER THAN OR EQUAL TO the LENGTH OF nums MINUS one HOLDS
                    BREAK
                END IF
            END IF
        END FOR

        RETURN jumps
    END FUNCTION
END CLASS
