CLASS Solution
    FUNCTION threeSumSmaller(parameter nums AS list_of_integers, parameter target AS integer) RETURNS integer
        CALL sort ON nums
        SET count TO zero
        SET n TO the LENGTH OF nums

        FOR i FROM zero TO n MINUS two MINUS one INCLUSIVE
            SET left TO i PLUS one
            SET right TO n MINUS one

            WHILE left LESS THAN right HOLDS
                SET current_sum TO element at position i OF nums PLUS element at position left OF nums PLUS element at position right OF nums

                IF current_sum LESS THAN target THEN
                    INCREMENT count BY right MINUS left
                    INCREMENT left BY one
                ELSE
                    DECREMENT right BY one
                END IF
            END WHILE
        END FOR

        RETURN count
    END FUNCTION
END CLASS
