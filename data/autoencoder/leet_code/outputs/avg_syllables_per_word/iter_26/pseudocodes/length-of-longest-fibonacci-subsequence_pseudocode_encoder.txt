CLASS Solution
    FUNCTION lenLongestFibSubseq(list_of_numbers)
        SET index_map TO an empty mapping from number to position
        FOR each position AND each number IN list_of_numbers WITH their respective positions
            SET the entry at key number TO position IN index_map
        END FOR

        SET dynamic_programming_map TO an empty mapping from pair of positions TO length
        SET maximum_length TO zero

        FOR position_k FROM zero TO the LENGTH OF list_of_numbers MINUS one INCLUSIVE
            FOR position_j FROM zero TO position_k MINUS one INCLUSIVE
                SET difference_number TO the element at position position_k MINUS the element at position position_j OF list_of_numbers
                SET position_i TO the value in index_map at key difference_number OR minus one if not present
                IF position_i IS GREATER THAN OR EQUAL TO zero AND position_i IS LESS THAN position_j THEN
                    SET dynamic_programming_map at key pair of position_j AND position_k TO the value in dynamic_programming_map at key pair of position_i AND position_j OR two IF not present PLUS one
                    SET maximum_length TO the greater of maximum_length AND dynamic_programming_map at key pair of position_j AND position_k
                END IF
            END FOR
        END FOR

        IF maximum_length IS GREATER THAN OR EQUAL TO three THEN
            RETURN maximum_length
        ELSE
            RETURN zero
        END IF
    END FUNCTION
END CLASS
