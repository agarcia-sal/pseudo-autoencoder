CLASS Solution
    FUNCTION maxA(n)
        IF n LESS THAN OR EQUAL TO three
            RETURN n
        END IF

        SET dp TO list OF zero repeated n PLUS one times

        FOR i FROM one TO three INCLUSIVE
            SET element at position i OF dp TO i
        END FOR

        FOR i FROM four TO n INCLUSIVE
            FOR j FROM i MINUS three DOWN TO one INCLUSIVE DECREMENTING BY one
                SET value_to_compare TO element at position j OF dp MULTIPLIED BY i MINUS j MINUS one
                SET element at position i OF dp TO the GREATER OF element at position i OF dp AND value_to_compare
            END FOR
        END FOR

        RETURN element at position n OF dp
    END FUNCTION
END CLASS
