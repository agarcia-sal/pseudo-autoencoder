CLASS Solution
    FUNCTION kEmptySlots(list_of_bulbs, integer_k)
        SET bloom_days TO empty list

        FOR each day_with_index FROM one TO the LENGTH OF list_of_bulbs INCLUSIVE
            SET position TO the element at position day_with_index MINUS one OF list_of_bulbs

            SET insertion_index TO the result of locating the leftmost position to insert position IN bloom_days USING binary search

            IF insertion_index GREATER THAN zero AND position MINUS the element at position insertion_index MINUS one OF bloom_days MINUS one EQUALS integer_k THEN
                RETURN day_with_index
            END IF

            IF insertion_index LESS THAN the LENGTH OF bloom_days AND the element at position insertion_index OF bloom_days MINUS position MINUS one EQUALS integer_k THEN
                RETURN day_with_index
            END IF

            INSERT position INTO bloom_days AT insertion_index
        END FOR

        RETURN negative one
    END FUNCTION
END CLASS
