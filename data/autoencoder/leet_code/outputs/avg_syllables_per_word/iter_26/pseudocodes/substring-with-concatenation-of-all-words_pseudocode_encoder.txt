CLASS Solution
    FUNCTION findSubstring(string_s, list_words)
        IF string_s IS EMPTY OR list_words IS EMPTY OR element at position zero OF list_words IS EMPTY
            RETURN empty list
        END IF

        SET word_length TO the LENGTH OF element at position zero OF list_words
        SET number_of_words TO the LENGTH OF list_words
        SET total_length TO word_length MULTIPLIED BY number_of_words
        SET word_count TO a COUNTER created from list_words
        SET result TO an empty list

        FOR index_i FROM zero TO the LENGTH OF string_s MINUS total_length PLUS one NOT INCLUDING one
            SET seen TO an empty COUNTER
            FOR index_j FROM index_i TO index_i PLUS total_length NOT INCLUDING index_i PLUS total_length IN STEPS OF word_length
                SET word TO the SUBSTRING FROM position index_j TO position index_j PLUS word_length OF string_s
                INCREMENT the COUNT OF word IN seen BY one
                IF the COUNT OF word IN seen IS GREATER THAN the COUNT OF word IN word_count
                    BREAK out of the inner loop
                END IF
            END FOR
            IF the inner loop COMPLETED WITHOUT a BREAK
                APPEND index_i TO result
            END IF
        END FOR

        RETURN result
    END FUNCTION
END CLASS
