CLASS Solution
    FUNCTION characterReplacement(s, k)
        FROM COLLECTIONS IMPORT DEFAULTDICT
        
        SET max_length TO zero
        SET max_count TO zero
        SET left TO zero
        SET count TO DEFAULTDICT OF integer
        
        FOR right FROM zero TO the LENGTH OF s MINUS one INCLUSIVE
            INCREMENT the ELEMENT AT POSITION right OF s IN count BY one
            SET max_count TO the GREATER VALUE OF max_count AND the ELEMENT AT POSITION right OF s IN count
            
            IF the VALUE OF right MINUS left PLUS one MINUS max_count GREATER THAN k HOLDS
                DECREMENT the ELEMENT AT POSITION left OF s IN count BY one
                INCREMENT left BY one
            END IF
            
            SET max_length TO the GREATER VALUE OF max_length AND right MINUS left PLUS one
        END FOR
        
        RETURN max_length
    END FUNCTION
END CLASS
