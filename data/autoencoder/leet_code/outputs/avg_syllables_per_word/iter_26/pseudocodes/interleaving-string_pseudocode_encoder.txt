CLASS Solution
    FUNCTION isInterleave(s1, s2, s3)
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 NOT EQUALS the LENGTH OF s3
            RETURN false
        END IF

        SET dp TO a list of boolean values all set to false with the LENGTH OF s2 PLUS one elements
        SET element at position zero OF dp TO true

        FOR i FROM zero TO the LENGTH OF s1 INCLUSIVE
            FOR j FROM zero TO the LENGTH OF s2 INCLUSIVE
                IF i GREATER THAN zero
                    SET element at position j OF dp TO the logical AND of element at position j OF dp AND the EQUALITY check of element at position i MINUS one OF s1 EQUALS element at position i PLUS j MINUS one OF s3
                END IF
                IF j GREATER THAN zero
                    SET element at position j OF dp TO the logical OR of element at position j OF dp AND the logical AND of element at position j MINUS one OF dp AND the EQUALITY check of element at position j MINUS one OF s2 EQUALS element at position i PLUS j MINUS one OF s3
                END IF
            END FOR
        END FOR

        RETURN element at position the LENGTH OF s2 OF dp
    END FUNCTION
END CLASS
