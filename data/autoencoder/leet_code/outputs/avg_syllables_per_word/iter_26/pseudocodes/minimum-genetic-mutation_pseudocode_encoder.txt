CLASS Solution
    FUNCTION minMutation(start AS string, end AS string, bank AS list OF string) RETURNS integer
        SET s TO the SET CREATED FROM bank
        SET q TO an EMPTY double ENDED QUEUE
        APPEND the PAIR OF start AND zero AS integer TO q
        SET mp TO a MAPPING WHERE
            key A IS MAPPED TO the STRING composed OF the CHARACTERS tee, cee, gee
            key tee IS MAPPED TO the STRING composed OF the CHARACTERS ay, cee, gee
            key cee IS MAPPED TO the STRING composed OF the CHARACTERS ay, tee, gee
            key gee IS MAPPED TO the STRING composed OF the CHARACTERS ay, tee, cee
        WHILE q IS NOT EMPTY
            REMOVE AND ASSIGN THE FIRST ELEMENT OF q TO the PAIR OF t AND step
            IF t EQUALS end
                RETURN step
            END IF
            FOR each INDEXED ELEMENT i AND CHARACTER v IN t
                FOR each CHARACTER j IN the VALUE MAPPED FROM v IN mp
                    SET next TO the STRING COMPOSED BY CONCATENATING the SUBSTRING OF t FROM the FIRST POSITION TO BEFORE position i, THE CHARACTER j, AND the SUBSTRING OF t FROM position i PLUS ONE TO THE END
                    IF next IS A MEMBER OF s
                        APPEND THE PAIR OF next AND step PLUS one TO q
                        REMOVE next FROM s
                    END IF
                END FOR
            END FOR
        END WHILE
        RETURN the NEGATIVE OF one
    END FUNCTION
END CLASS
