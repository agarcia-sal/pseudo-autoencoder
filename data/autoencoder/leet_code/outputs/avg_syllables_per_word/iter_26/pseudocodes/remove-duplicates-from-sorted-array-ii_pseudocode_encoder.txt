CLASS Solution
    FUNCTION removeDuplicates(nums)
        IF nums HOLDS NO ELEMENTS
            RETURN zero
        END IF

        SET write_index TO zero

        FOR i FROM zero TO THE MINIMUM OF two AND the LENGTH OF nums MINUS one INCLUSIVE
            SET element at position write_index OF nums TO element at position i OF nums
            INCREMENT write_index BY one
        END FOR

        FOR i FROM two TO the LENGTH OF nums MINUS one INCLUSIVE
            IF element at position i OF nums NOT EQUALS element at position write_index MINUS two OF nums
                SET element at position write_index OF nums TO element at position i OF nums
                INCREMENT write_index BY one
            END IF
        END FOR

        RETURN write_index
    END FUNCTION
END CLASS
