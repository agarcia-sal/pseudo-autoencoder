CLASS Solution
    FUNCTION groupAnagrams(list_of_strings)
        SET anagrams TO a default dictionary with default value of an empty list
        
        FOR each string_element IN list_of_strings
            SET sorted_string TO the concatenation of the characters of string_element arranged in ascending order
            APPEND string_element TO the list at key sorted_string IN anagrams
        END FOR
        
        SET result_list TO the list of all values contained in anagrams
        RETURN result_list
    END FUNCTION
END CLASS
