CLASS Solution
    FUNCTION countBattleships(board AS list of list of strings)
        SET count TO zero
        FOR index i FROM zero TO the LENGTH OF board MINUS one INCLUSIVE
            FOR index j FROM zero TO the LENGTH OF element at position zero OF board MINUS one INCLUSIVE
                IF element at position i OF board AT position j EQUALS the character denoting battleship
                    IF the index i EQUALS zero OR element at position i MINUS one OF board AT position j EQUALS the character denoting empty cell
                    AND the index j EQUALS zero OR element at position i OF board AT position j MINUS one EQUALS the character denoting empty cell
                        INCREMENT count BY one
                    END IF
                END IF
            END FOR
        END FOR
        RETURN count
    END FUNCTION
END CLASS
