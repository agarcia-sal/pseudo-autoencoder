CLASS Solution
    FUNCTION insert(intervals AS List OF List OF Integer, newInterval AS List OF Integer) AS List OF List OF Integer
        IF intervals EQUALS empty
            RETURN a new list containing newInterval
        END IF

        SET merged TO an empty list
        SET i TO zero
        SET n TO the LENGTH OF intervals

        WHILE i LESS THAN n AND the element at position one OF the element at position i OF intervals LESS THAN the element at position zero OF newInterval HOLDS
            APPEND the element at position i OF intervals TO merged
            INCREMENT i BY one
        END WHILE

        WHILE i LESS THAN n AND the element at position zero OF the element at position i OF intervals LESS THAN OR EQUAL TO the element at position one OF newInterval HOLDS
            SET the element at position zero OF newInterval TO the MINIMUM OF the element at position zero OF newInterval AND the element at position zero OF the element at position i OF intervals
            SET the element at position one OF newInterval TO the MAXIMUM OF the element at position one OF newInterval AND the element at position one OF the element at position i OF intervals
            INCREMENT i BY one
        END WHILE

        APPEND newInterval TO merged

        WHILE i LESS THAN n HOLDS
            APPEND the element at position i OF intervals TO merged
            INCREMENT i BY one
        END WHILE

        RETURN merged
    END FUNCTION
END CLASS
