CLASS Solution
    FUNCTION findKthNumber(m, n, k)
        FUNCTION enough(x)
            SET count TO zero
            FOR i FROM one TO m INCLUSIVE
                INCREMENT count BY the LESSER OF x DIVIDED BY i AND n
            END FOR
            RETURN count GREATER THAN OR EQUAL TO k
        END FUNCTION

        SET low TO one
        SET high TO m MULTIPLIED BY n
        WHILE low LESS THAN high HOLDS
            SET mid TO low PLUS high DIVIDED BY two
            IF enough(mid) HOLDS
                SET high TO mid
            ELSE
                SET low TO low PLUS one
            END IF
        END WHILE
        RETURN low
    END FUNCTION
END CLASS
