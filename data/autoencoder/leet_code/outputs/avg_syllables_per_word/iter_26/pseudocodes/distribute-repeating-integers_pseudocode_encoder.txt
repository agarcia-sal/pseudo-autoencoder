CLASS Solution
    FUNCTION canDistribute(nums PARAMETER, quantity PARAMETER)
        SET count TO the LIST OF the VALUES OF the COUNTER OF nums
        SORT quantity IN DESCENDING ORDER
        
        FUNCTION can_satisfy(index PARAMETER)
            IF index EQUALS the LENGTH OF quantity THEN
                RETURN TRUE
            END IF
            FOR i FROM zero TO the LENGTH OF count MINUS one INCLUSIVE
                IF the ELEMENT at POSITION i OF count IS GREATER THAN OR EQUAL TO the ELEMENT at POSITION index OF quantity THEN
                    DECREMENT the ELEMENT at POSITION i OF count BY the ELEMENT at POSITION index OF quantity
                    IF can_satisfy(index PLUS one) HOLDS THEN
                        RETURN TRUE
                    END IF
                    INCREMENT the ELEMENT at POSITION i OF count BY the ELEMENT at POSITION index OF quantity
                END IF
            END FOR
            RETURN FALSE
        END FUNCTION
        
        RETURN can_satisfy(zero)
    END FUNCTION
END CLASS
