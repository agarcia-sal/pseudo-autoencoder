CLASS Solution
    FUNCTION circularArrayLoop(parameter nums AS list of integers) RETURNS boolean
        FUNCTION sameDirection(parameter val1, parameter val2) RETURNS boolean
            RETURN val1 GREATER THAN zero AND val2 GREATER THAN zero OR val1 LESS THAN zero AND val2 LESS THAN zero
        END FUNCTION

        SET n TO the LENGTH OF nums

        FOR index i FROM zero TO n MINUS one INCLUSIVE
            SET slow TO i
            SET fast TO i

            IF the element AT position i OF nums EQUALS zero THEN
                CONTINUE
            END IF

            WHILE true HOLDS
                SET slow TO the REMAINDER OF slow PLUS the element AT position slow OF nums DIVIDED BY n
                IF slow LESS THAN zero THEN
                    INCREMENT slow BY n
                END IF

                SET fast TO the REMAINDER OF fast PLUS the element AT position fast OF nums DIVIDED BY n
                IF fast LESS THAN zero THEN
                    INCREMENT fast BY n
                END IF

                SET fast TO the REMAINDER OF fast PLUS the element AT position fast OF nums DIVIDED BY n
                IF fast LESS THAN zero THEN
                    INCREMENT fast BY n
                END IF

                IF NOT sameDirection(the element AT position slow OF nums, the element AT position i OF nums) OR NOT sameDirection(the element AT position fast OF nums, the element AT position i OF nums) THEN
                    BREAK
                END IF

                IF slow EQUALS fast THEN
                    IF slow EQUALS the REMAINDER OF slow PLUS the element AT position slow OF nums DIVIDED BY n THEN
                        BREAK
                    END IF
                    RETURN true
                END IF
            END WHILE
        END FOR

        RETURN false
    END FUNCTION
END CLASS
