CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values DOES NOT HOLD
        RETURN none
    END IF
    SET root TO A NEW TreeNode WITH element at position zero OF values
    SET i TO one
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE
    APPEND root TO queue
    WHILE queue HOLDS
        SET node TO THE FIRST ELEMENT REMOVED FROM queue
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none
            SET node.left TO A NEW TreeNode WITH element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none
            SET node.right TO A NEW TreeNode WITH element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p DOES NOT HOLD AND q DOES NOT HOLD
        RETURN true
    ELSE IF p DOES NOT HOLD OR q DOES NOT HOLD
        RETURN false
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN false
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION maxPathSum(self, root)
        FUNCTION helper(node)
            IF node DOES NOT HOLD
                RETURN negative infinity, negative infinity
            END IF
            SET left_max, left_gain TO helper(node.left)
            SET right_max, right_gain TO helper(node.right)
            SET current_max TO node.val PLUS THE GREATER OF left_gain AND zero PLUS THE GREATER OF right_gain AND zero
            SET current_gain TO node.val PLUS THE GREATEST OF left_gain, right_gain, AND zero
            SET global_max TO THE GREATEST OF left_max, right_max, AND current_max
            RETURN global_max, current_gain
        END FUNCTION
        RETURN THE FIRST ELEMENT OF helper(root)
    END FUNCTION
END CLASS
