CLASS Solution
    FUNCTION validTree(n, edges)
        IF the LENGTH OF edges NOT EQUALS n MINUS one THEN
            RETURN false
        END IF

        SET adj_list TO a new empty dictionary mapping each integer i FROM zero TO n MINUS one INCLUSIVE TO an empty list

        FOR each pair a AND b IN edges
            APPEND b TO the list associated with key a IN adj_list
            APPEND a TO the list associated with key b IN adj_list
        END FOR

        SET visited TO a new empty set

        FUNCTION dfs(node)
            IF node IS IN visited THEN
                RETURN
            END IF
            ADD node TO visited
            FOR each neighbor IN the list associated with key node IN adj_list
                CALL dfs WITH neighbor
            END FOR
        END FUNCTION

        CALL dfs WITH zero

        RETURN the LENGTH OF visited EQUALS n
    END FUNCTION
END CLASS
