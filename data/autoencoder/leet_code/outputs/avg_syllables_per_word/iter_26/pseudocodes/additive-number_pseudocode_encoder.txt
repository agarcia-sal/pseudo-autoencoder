CLASS Solution
    FUNCTION isAdditiveNumber(num)
        FUNCTION is_valid_sequence(start_position, first_number, second_number)
            IF start_position EQUALS the LENGTH OF num
                RETURN true
            END IF

            SET expected_string TO the STRING REPRESENTATION OF the SUM OF first_number PLUS second_number
            SET expected_length TO the LENGTH OF expected_string

            IF start_position PLUS expected_length GREATER THAN the LENGTH OF num OR the SUBSTRING FROM position start_position TO position start_position PLUS expected_length MINUS one OF num NOT EQUALS expected_string
                RETURN false
            END IF

            RETURN the RESULT OF is_valid_sequence WITH PARAMETERS start_position PLUS expected_length, second_number, the INTEGER REPRESENTATION OF expected_string
        END FUNCTION

        SET n TO the LENGTH OF num

        FOR index i FROM one TO n DIVIDED BY two INCLUSIVE
            FOR index j FROM i PLUS one TO n MINUS one INCLUSIVE
                SET first TO the SUBSTRING FROM position zero TO position i MINUS one OF num
                SET second TO the SUBSTRING FROM position i TO position j MINUS one OF num

                IF the LENGTH OF first GREATER THAN one AND the FIRST CHARACTER OF first EQUALS the CHARACTER named zero OR the LENGTH OF second GREATER THAN one AND the FIRST CHARACTER OF second EQUALS the CHARACTER named zero
                    CONTINUE TO the NEXT ITERATION
                END IF

                IF the RESULT OF is_valid_sequence WITH PARAMETERS j, the INTEGER REPRESENTATION OF first, the INTEGER REPRESENTATION OF second HOLDS TRUE
                    RETURN true
                END IF
            END FOR
        END FOR

        RETURN false
    END FUNCTION
END CLASS
