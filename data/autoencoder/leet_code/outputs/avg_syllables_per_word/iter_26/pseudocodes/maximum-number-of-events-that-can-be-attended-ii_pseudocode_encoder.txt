CLASS Solution
    FUNCTION maxValue(events, k)
        CALL sort ON events BY the end day element OF each event
        SET number_of_events TO the LENGTH OF events
        
        SET dp TO a two dimensional list WHERE each element IS zero 
            WITH the number of rows EQUAL TO number_of_events PLUS one
            AND the number of columns EQUAL TO k PLUS one
        
        FOR index_i FROM one TO number_of_events INCLUSIVE
            SET start_day TO the element at position zero OF the event at position index_i MINUS one OF events
            SET end_day TO the element at position one OF the event at position index_i MINUS one OF events
            SET event_value TO the element at position two OF the event at position index_i MINUS one OF events
            
            SET position_j TO the POSITION FOUND USING binary search ON events 
                WHERE the end day element IS GREATER THAN OR EQUAL TO start_day
            
            FOR count_l FROM one TO k INCLUSIVE
                SET dp element at row index_i and column count_l TO the GREATER OF
                    the dp element at row index_i MINUS one and column count_l
                    AND
                    the sum of the dp element at row position_j and column count_l MINUS one PLUS event_value
            END FOR
        END FOR
        
        RETURN the dp element at row number_of_events and column k
    END FUNCTION
END CLASS
