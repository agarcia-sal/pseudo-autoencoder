CLASS Solution
    FUNCTION longestAwesome(s)
        SET prefix_map TO a dictionary containing the key zero MAPPED TO the value negative one
        SET prefix_xor TO zero
        SET max_length TO zero

        FOR each index i AND character char IN the enumeration of s
            SET digit TO the integer value CORRESPONDING TO character char
            SET prefix_xor TO prefix_xor XORED BY the value one LEFT SHIFTED BY digit

            IF prefix_xor IS A KEY IN prefix_map
                SET max_length TO the GREATER OF max_length AND the DIFFERENCE BETWEEN i AND the VALUE CORRESPONDING TO prefix_xor IN prefix_map
            END IF

            FOR each integer j FROM zero TO nine INCLUSIVE
                SET flipped TO prefix_xor XORED BY the value one LEFT SHIFTED BY j
                IF flipped IS A KEY IN prefix_map
                    SET max_length TO the GREATER OF max_length AND the DIFFERENCE BETWEEN i AND the VALUE CORRESPONDING TO flipped IN prefix_map
                END IF
            END FOR

            IF prefix_xor IS NOT A KEY IN prefix_map
                SET the VALUE CORRESPONDING TO prefix_xor IN prefix_map TO i
            END IF
        END FOR

        RETURN max_length
    END FUNCTION
END CLASS
