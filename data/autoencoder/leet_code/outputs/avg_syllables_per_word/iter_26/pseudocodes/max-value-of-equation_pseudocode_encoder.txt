CLASS Solution
    FUNCTION findMaxValueOfEquation(points, k)
        SET maximum_value TO negative infinity
        ASSIGN reference TO an empty double ended queue named deque
        
        FOR each pair of elements xj AND yj IN points
            WHILE deque IS NOT empty AND the difference between xj AND the second element of the first pair IN deque IS GREATER THAN k
                REMOVE the first element FROM deque
            END WHILE
            
            IF deque IS NOT empty
                SET maximum_value TO the greater value BETWEEN maximum_value AND the sum of yj PLUS xj PLUS the first element OF the first pair IN deque
            END IF
            
            WHILE deque IS NOT empty AND the difference between yj AND xj IS GREATER THAN OR EQUAL TO the first element OF the last pair IN deque
                REMOVE the last element FROM deque
            END WHILE
            
            APPEND the pair consisting of the difference between yj AND xj AND the value xj TO deque
        END FOR
        
        RETURN maximum_value
    END FUNCTION
END CLASS
