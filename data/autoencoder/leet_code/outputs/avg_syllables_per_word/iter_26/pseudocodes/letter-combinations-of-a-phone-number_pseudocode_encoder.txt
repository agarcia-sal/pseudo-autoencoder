CLASS Solution
    FUNCTION letterCombinations(digits)
        IF digits EQUALS an empty sequence
            RETURN an empty list
        END IF

        SET phone_map TO a mapping FROM digit character TWO TO sequence of letters a b c,
                                      FROM digit character THREE TO sequence of letters d e f,
                                      FROM digit character FOUR TO sequence of letters g h i,
                                      FROM digit character FIVE TO sequence of letters j k l,
                                      FROM digit character SIX TO sequence of letters m n o,
                                      FROM digit character SEVEN TO sequence of letters p q r s,
                                      FROM digit character EIGHT TO sequence of letters t u v,
                                      FROM digit character NINE TO sequence of letters w x y z

        FUNCTION backtrack(combination, next_digits)
            IF the LENGTH OF next_digits EQUALS zero
                APPEND combination TO output
            ELSE
                FOR each letter IN the sequence of letters THAT phone_map maps FROM the element at position zero OF next_digits
                    CALL backtrack WITH the concatenation OF combination AND letter AS first argument AND the subsequence OF next_digits FROM position one TO the end AS second argument
                END FOR
            END IF
        END FUNCTION

        SET output TO an empty list
        CALL backtrack WITH empty string AS first argument AND digits AS second argument
        RETURN output
    END FUNCTION
END CLASS
