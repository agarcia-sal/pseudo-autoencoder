CLASS Solution
    FUNCTION maxNiceDivisors(primeFactors)
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven
        
        IF primeFactors LESS THAN OR EQUAL TO three HOLDS
            RETURN primeFactors
        END IF
        
        SET num_threes TO primeFactors DIVIDED BY three
        SET remainder TO primeFactors MODULO three
        
        IF remainder EQUALS zero HOLDS
            RETURN three RAISED TO THE POWER OF num_threes WITH MODULO MODULO MOD
        ELSE IF remainder EQUALS one HOLDS
            RETURN three RAISED TO THE POWER OF num_threes MINUS one WITH MODULO MODULO MOD MULTIPLIED BY four MODULO MOD
        ELSE  # remainder EQUALS two
            RETURN three RAISED TO THE POWER OF num_threes WITH MODULO MODULO MOD MULTIPLIED BY two MODULO MOD
        END IF
    END FUNCTION
END CLASS
