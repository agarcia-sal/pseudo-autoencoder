CLASS Solution
    FUNCTION isScramble(string_one, string_two)
        IF string_one EQUALS string_two
            RETURN True
        END IF

        IF the SORTED CHARACTERS OF string_one NOT EQUALS the SORTED CHARACTERS OF string_two
            RETURN False
        END IF

        SET length_n TO the LENGTH OF string_one

        FOR index_i FROM one TO length_n MINUS one INCLUSIVE
            IF self DOT isScramble WITH PARAMETERS substring FROM position zero TO position index_i MINUS one OF string_one AND substring FROM position zero TO position index_i MINUS one OF string_two
               AND self DOT isScramble WITH PARAMETERS substring FROM position index_i TO the END OF string_one AND substring FROM position index_i TO the END OF string_two
                RETURN True
            END IF

            IF self DOT isScramble WITH PARAMETERS substring FROM position zero TO position index_i MINUS one OF string_one AND substring FROM the POSITION negative index_i TO the END OF string_two
               AND self DOT isScramble WITH PARAMETERS substring FROM position index_i TO the END OF string_one AND substring FROM position zero TO position negative index_i MINUS one OF string_two
                RETURN True
            END IF
        END FOR

        RETURN False
    END FUNCTION
END CLASS
