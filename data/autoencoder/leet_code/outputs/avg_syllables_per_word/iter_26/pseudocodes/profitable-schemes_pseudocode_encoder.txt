CLASS Solution
    FUNCTION profitableSchemes(n, minProfit, group, profit)
        SET MOD TO ten raised to the power of nine PLUS one
        COMMENT dp at position i and j MEANS the number of ways to achieve at least j profit WITH i members
        SET dp TO a two dimensional list of zeroes WITH each inner list of length minProfit PLUS one AND the number of inner lists equal to n PLUS one
        COMMENT There is one way to achieve zero profit WITH zero members BY doing nothing
        FOR index i FROM zero TO n INCLUSIVE
            SET element at position zero OF dp at position i TO one
        END FOR

        FOR each profit_value AND group_value IN parallel FROM profit AND group
            COMMENT Traverse backwards TO avoid using the same group member multiple times
            FOR index i FROM n DOWN TO group_value INCLUSIVE STEPPING BY negative one
                FOR index j FROM minProfit DOWN TO zero INCLUSIVE STEPPING BY negative one
                    SET dp at position i and j TO the remainder of the sum of dp at position i and j PLUS dp at position i minus group_value and the maximum of zero AND j minus profit_value DIVIDED BY MODULO
                END FOR
            END FOR
        END FOR

        RETURN the element at position n and minProfit OF dp
    END FUNCTION
END CLASS
