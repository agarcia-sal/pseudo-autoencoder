CLASS Solution
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)
        IF maxChoosableInteger MULTIPLIED BY the SUM OF maxChoosableInteger AND one DIVIDED BY two LESS THAN desiredTotal
            RETURN false
        END IF

        IF desiredTotal LESS THAN OR EQUAL TO zero
            RETURN true
        END IF

        SET memo TO empty dictionary

        FUNCTION can_win(current_total, used_numbers)
            IF the PAIR OF current_total AND used_numbers IS IN memo
                RETURN the VALUE ASSOCIATED WITH the PAIR OF current_total AND used_numbers IN memo
            END IF

            FOR number FROM one TO maxChoosableInteger INCLUSIVE
                IF NOT the BITWISE AND OF used_numbers AND one LEFT SHIFTED BY number HOLDS
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win(current_total PLUS number, the BITWISE OR OF used_numbers AND one LEFT SHIFTED BY number)
                        SET the VALUE ASSOCIATED WITH the PAIR OF current_total AND used_numbers IN memo TO true
                        RETURN true
                    END IF
                END IF
            END FOR

            SET the VALUE ASSOCIATED WITH the PAIR OF current_total AND used_numbers IN memo TO false
            RETURN false
        END FUNCTION

        RETURN can_win(zero, zero)
    END FUNCTION
END CLASS
