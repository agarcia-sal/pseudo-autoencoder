CLASS Solution
    FUNCTION multiply(mat1, mat2)
        SET m TO the LENGTH OF mat1
        SET k TO the LENGTH OF the element at position zero OF mat1
        SET k TO the LENGTH OF mat2
        SET n TO the LENGTH OF the element at position zero OF mat2

        INITIALIZE result TO an empty list
        FOR i FROM zero TO m MINUS one INCLUSIVE
            INITIALIZE row TO an empty list
            FOR j FROM zero TO n MINUS one INCLUSIVE
                APPEND zero TO row
            END FOR
            APPEND row TO result
        END FOR

        FOR i FROM zero TO m MINUS one INCLUSIVE
            FOR j FROM zero TO n MINUS one INCLUSIVE
                FOR p FROM zero TO k MINUS one INCLUSIVE
                    SET current_value TO the element at position i OF result
                    SET updated_value TO the element at position j OF current_value PLUS the element at position i OF mat1 MULTIPLIED BY the element at position j OF the element at position p OF mat2
                    SET the element at position j OF current_value TO updated_value
                END FOR
            END FOR
        END FOR

        RETURN result
    END FUNCTION
END CLASS
