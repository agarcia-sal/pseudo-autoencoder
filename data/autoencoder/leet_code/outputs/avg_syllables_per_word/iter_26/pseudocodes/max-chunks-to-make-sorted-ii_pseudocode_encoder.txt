CLASS Solution
    FUNCTION maxChunksToSorted(arr LIST OF integers)
        SET max_so_far TO an empty list
        SET current_max TO element at position zero OF arr
        FOR each number IN arr
            SET current_max TO the GREATER OF current_max AND number
            APPEND current_max TO max_so_far
        END FOR

        SET sorted_arr TO arr IN ascending order
        SET chunks TO zero
        SET max_chunk TO zero

        FOR index FROM zero TO the LENGTH OF arr MINUS one INCLUSIVE
            SET max_chunk TO the GREATER OF max_chunk AND element at position index OF arr
            IF max_chunk EQUALS element at position index OF sorted_arr
                INCREMENT chunks BY one
            END IF
        END FOR

        RETURN chunks
    END FUNCTION
END CLASS
