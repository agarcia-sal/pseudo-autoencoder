CLASS Solution
    FUNCTION arrayNesting(list_of_numbers)
        SET visited_list TO a list of boolean false repeated for the LENGTH OF list_of_numbers
        SET maximum_length TO zero
        
        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one INCLUSIVE
            IF the boolean value at position index OF visited_list NOT EQUALS true
                SET current_length TO zero
                SET current_position TO index
                WHILE the boolean value at position current_position OF visited_list NOT EQUALS true HOLDS
                    SET the boolean value at position current_position OF visited_list TO true
                    SET current_position TO the element at position current_position OF list_of_numbers
                    INCREMENT current_length BY one
                END WHILE
                SET maximum_length TO the GREATER OF maximum_length AND current_length
            END IF
        END FOR
        
        RETURN maximum_length
    END FUNCTION
END CLASS
