CLASS Solution
    FUNCTION longestPalindrome(word1, word2)
        SET s TO the concatenation of word1 AND word2
        SET n TO the LENGTH OF s
        SET dp TO a two dimensional list of zeroes with dimensions n BY n

        FOR index i FROM zero TO n MINUS one INCLUSIVE
            SET element at position i AND position i OF dp TO one
        END FOR

        SET max_len TO zero
        FOR length FROM two TO n INCLUSIVE
            FOR index i FROM zero TO n MINUS length INCLUSIVE
                SET j TO i PLUS length MINUS one
                IF the element at position i OF s EQUALS the element at position j OF s
                    SET element at position i AND position j OF dp TO the element at position i PLUS one AND position j MINUS one OF dp PLUS two
                    IF i LESS THAN the LENGTH OF word1 AND j GREATER THAN OR EQUAL TO the LENGTH OF word1
                        SET max_len TO the GREATER VALUE BETWEEN max_len AND the element at position i AND position j OF dp
                    END IF
                ELSE
                    SET element at position i AND position j OF dp TO the GREATER VALUE BETWEEN the element at position i PLUS one AND position j OF dp AND the element at position i AND position j MINUS one OF dp
                END IF
            END FOR
        END FOR

        RETURN max_len
    END FUNCTION
END CLASS
