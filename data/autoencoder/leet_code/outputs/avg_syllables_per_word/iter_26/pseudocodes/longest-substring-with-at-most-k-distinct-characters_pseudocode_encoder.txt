CLASS Solution
    FUNCTION lengthOfLongestSubstringKDistinct(s AS string, k AS integer) RETURNS integer
        IF k EQUALS zero THEN
            RETURN zero
        END IF

        SET char_map TO empty mapping
        SET left TO zero
        SET max_length TO zero

        FOR right FROM zero TO the LENGTH OF s MINUS one INCLUSIVE
            SET character AT position right OF s IN char_map TO right

            IF the LENGTH OF char_map GREATER THAN k THEN
                SET leftmost_char TO the CHARACTER WITH the MINIMUM VALUE IN char_map BY ITS ASSOCIATED POSITION
                SET left TO the POSITION ASSOCIATED WITH leftmost_char IN char_map PLUS one
                REMOVE leftmost_char FROM char_map
            END IF

            SET max_length TO the GREATER OF max_length AND the DIFFERENCE BETWEEN right AND left PLUS one
        END FOR

        RETURN max_length
    END FUNCTION
END CLASS
