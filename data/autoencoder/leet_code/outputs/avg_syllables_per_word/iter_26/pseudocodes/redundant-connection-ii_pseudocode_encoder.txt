CLASS Solution
    FUNCTION findRedundantDirectedConnection(list_of_edges)
        FUNCTION find(list_of_parent, element_i)
            IF element at position element_i OF list_of_parent NOT EQUALS element_i THEN
                SET element at position element_i OF list_of_parent TO find(list_of_parent, element at position element_i OF list_of_parent)
            END IF
            RETURN element at position element_i OF list_of_parent
        END FUNCTION

        FUNCTION union(list_of_parent, list_of_rank, element_x, element_y)
            SET root_X TO find(list_of_parent, element_x)
            SET root_Y TO find(list_of_parent, element_y)
            IF root_X NOT EQUALS root_Y THEN
                IF element at position root_X OF list_of_rank GREATER THAN element at position root_Y OF list_of_rank THEN
                    SET element at position root_Y OF list_of_parent TO root_X
                ELSE IF element at position root_X OF list_of_rank LESS THAN element at position root_Y OF list_of_rank THEN
                    SET element at position root_X OF list_of_parent TO root_Y
                ELSE
                    SET element at position root_Y OF list_of_parent TO root_X
                    INCREMENT element at position root_X OF list_of_rank BY one
                END IF
                RETURN truth_value_true
            END IF
            RETURN truth_value_false
        END FUNCTION

        SET variable_n TO the LENGTH OF list_of_edges
        SET list_parent TO the list of integers from zero TO variable_n INCLUSIVE
        SET list_rank TO the list containing variable_n PLUS one occurrences of zero

        SET candidate1 TO none_null_value
        SET candidate2 TO none_null_value
        SET last TO none_null_value

        FOR each pair of elements named u and v IN list_of_edges
            IF element at position v OF list_parent NOT EQUALS v THEN
                SET candidate1 TO the list containing element at position v OF list_parent AND v
                SET candidate2 TO the list containing u AND v
            ELSE
                IF union(list_parent, list_rank, u, v) EQUALS truth_value_false THEN
                    SET last TO the list containing u AND v
                END IF
            END IF
        END FOR

        IF candidate1 EQUALS none_null_value THEN
            RETURN last
        END IF

        IF last NOT EQUALS none_null_value THEN
            RETURN candidate1
        END IF

        RETURN candidate2
    END FUNCTION
END CLASS
