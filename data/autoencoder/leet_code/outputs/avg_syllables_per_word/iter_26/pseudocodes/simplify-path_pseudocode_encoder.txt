CLASS Solution
    FUNCTION simplifyPath(path)
        SET components TO the LIST OF SUBSTRINGS OBTAINED BY SPLITTING path BY the CHARACTER slash
        SET stack TO an EMPTY LIST
        FOR each component IN components
            IF component EQUALS the EMPTY STRING OR component EQUALS the SINGLE CHARACTER period
                CONTINUE the current iteration
            ELSE IF component EQUALS the DOUBLE CHARACTER period
                IF the LENGTH OF stack IS GREATER THAN zero
                    REMOVE the LAST ELEMENT FROM stack
                END IF
            ELSE
                APPEND component TO stack
            END IF
        END FOR
        SET simplified_path TO the SINGLE CHARACTER slash CONCATENATED WITH the JOINING OF ELEMENTS OF stack WITH SINGLE CHARACTER slash AS SEPARATOR
        RETURN simplified_path
    END FUNCTION
END CLASS
