CLASS Solution
    FUNCTION hasPath(maze, start, destination)
        SET directions TO the list of pairs consisting of zero and one, one and zero, zero and negative one, negative one and zero
        SET m TO the LENGTH OF maze
        SET n TO the LENGTH OF element at position zero OF maze
        SET visited TO an empty set

        FUNCTION dfs(x, y)
            IF the pair consisting of x and y IS IN visited THEN
                RETURN False
            END IF
            IF the list consisting of x and y EQUALS destination THEN
                RETURN True
            END IF
            ADD the pair consisting of x and y TO visited
            
            FOR each pair dx and dy IN directions
                SET nx TO x
                SET ny TO y
                
                WHILE the value zero LESS THAN OR EQUAL TO nx PLUS dx AND nx PLUS dx LESS THAN m AND zero LESS THAN OR EQUAL TO ny PLUS dy AND ny PLUS dy LESS THAN n AND the element at position nx PLUS dx OF the element at position ny PLUS dy OF maze EQUALS zero HOLDS
                    INCREMENT nx BY dx
                    INCREMENT ny BY dy
                END WHILE
                
                IF dfs(nx, ny) THEN
                    RETURN True
                END IF
            END FOR
            RETURN False
        END FUNCTION
        
        RETURN dfs(the element at position zero OF start, the element at position one OF start)
    END FUNCTION
END CLASS
