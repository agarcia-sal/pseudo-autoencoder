CLASS Solution
    FUNCTION isMatch(s as string, p as string) RETURNS boolean
        SET dp TO a two dimensional list of boolean FALSE values WITH number of rows EQUAL TO the LENGTH OF s PLUS one AND number of columns EQUAL TO the LENGTH OF p PLUS one

        SET the element at position zero row AND position zero column OF dp TO TRUE

        FOR each position j FROM one TO the LENGTH OF p INCLUSIVE
            IF the character at position j MINUS one OF p EQUALS the character representing asterisk
                SET the element at position zero row AND position j column OF dp TO the value of the element at position zero row AND position j MINUS two column OF dp
            END IF
        END FOR

        FOR each position i FROM one TO the LENGTH OF s INCLUSIVE
            FOR each position j FROM one TO the LENGTH OF p INCLUSIVE
                IF the character at position j MINUS one OF p EQUALS the character at position i MINUS one OF s OR the character at position j MINUS one OF p EQUALS the character representing a single period
                    SET the element at position i row AND position j column OF dp TO the value of the element at position i MINUS one row AND position j MINUS one column OF dp
                ELSE IF the character at position j MINUS one OF p EQUALS the character representing asterisk
                    SET the element at position i row AND position j column OF dp TO the value of the element at position i row AND position j MINUS two column OF dp
                    IF the character at position j MINUS two OF p EQUALS the character at position i MINUS one OF s OR the character at position j MINUS two OF p EQUALS the character representing a single period
                        SET the element at position i row AND position j column OF dp TO the logical OR of the current value at position i row AND position j column OF dp AND the value of the element at position i MINUS one row AND position j column OF dp
                    END IF
                END IF
            END FOR
        END FOR

        RETURN the value of the element at position LENGTH OF s row AND position LENGTH OF p column OF dp
    END FUNCTION
END CLASS
