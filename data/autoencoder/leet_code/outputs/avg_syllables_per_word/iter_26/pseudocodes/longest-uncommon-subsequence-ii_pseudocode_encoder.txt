CLASS Solution
    FUNCTION findLUSlength(list_of_strings)
        FUNCTION is_subsequence(sequence_s, sequence_t)
            ASSIGN iterator_it TO an iterator over sequence_t
            RETURN the result of checking for every character c IN sequence_s WHETHER character c IS IN iterator_it
        END FUNCTION

        SORT list_of_strings BY the NEGATION OF the LENGTH OF each element AND THEN BY element IN ascending lexicographical order

        FOR each index i AND each word word IN list_of_strings BY enumeration
            SET unique TO true
            FOR each index j AND each other_string other IN list_of_strings BY enumeration
                IF index i IS NOT EQUAL TO index j AND is_subsequence(word, other) HOLDS THEN
                    SET unique TO false
                    BREAK the inner loop
                END IF
            END FOR
            IF unique HOLDS THEN
                RETURN the LENGTH OF word
            END IF
        END FOR

        RETURN the NEGATION OF one
    END FUNCTION
END CLASS
