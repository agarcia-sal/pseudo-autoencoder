CLASS ListNode
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values_list)
    IF values_list IS empty
        RETURN none
    END IF
    SET head TO A new ListNode CREATED WITH element at position zero OF values_list
    SET p TO head
    FOR each val IN the elements of values_list FROM position one TO the last position INCLUSIVE
        SET node TO A new ListNode CREATED WITH val
        SET p.next TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 EQUALS none AND p2 EQUALS none
        RETURN true
    END IF
    IF p1 EQUALS none OR p2 EQUALS none
        RETURN false
    END IF
    RETURN the CONDITION that element val OF p1 EQUALS element val OF p2 AND the RESULT OF is_same_list FOR element next OF p1 AND element next OF p2
END FUNCTION

CLASS Solution
    FUNCTION addTwoNumbers(l1, l2)
        SET dummy_head TO A new ListNode CREATED WITH zero
        SET current TO dummy_head
        SET carry TO zero
        
        WHILE l1 IS NOT none OR l2 IS NOT none OR carry IS NOT zero HOLDS
            SET val1 TO element val OF l1 IF l1 IS NOT none OTHERWISE zero
            SET val2 TO element val OF l2 IF l2 IS NOT none OTHERWISE zero
            SET total TO val1 PLUS val2 PLUS carry
            SET carry TO total DIVIDED BY the cardinal number ten
            SET current.next TO A new ListNode CREATED WITH total MODULO the cardinal number ten
            SET current TO element next OF current
            
            IF l1 IS NOT none
                SET l1 TO element next OF l1
            END IF
            IF l2 IS NOT none
                SET l2 TO element next OF l2
            END IF
        END WHILE
        
        RETURN element next OF dummy_head
    END FUNCTION
END CLASS
