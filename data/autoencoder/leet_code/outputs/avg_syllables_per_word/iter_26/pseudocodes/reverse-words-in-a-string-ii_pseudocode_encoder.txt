CLASS Solution
    FUNCTION reverseWords(list_of_characters)
        COMMENT Do not return anything, modify list_of_characters in place instead
        
        CALL the method to REVERSE the entire list_of_characters
        
        SET start TO zero
        FOR each end FROM zero TO the LENGTH OF list_of_characters MINUS one INCLUSIVE
            IF the element at position end OF list_of_characters EQUALS a space character
                CALL the method to REVERSE the subsection of list_of_characters FROM position start TO position end MINUS one INCLUSIVE
                SET start TO end PLUS one
            END IF
        END FOR
        
        CALL the method to REVERSE the subsection of list_of_characters FROM position start TO the end of the list
    END FUNCTION
END CLASS
