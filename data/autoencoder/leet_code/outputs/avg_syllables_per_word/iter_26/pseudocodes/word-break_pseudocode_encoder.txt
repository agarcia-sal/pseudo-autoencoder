CLASS Solution
    FUNCTION wordBreak(s, wordDict)
        SET wordSet TO the SET OF wordDict
        SET dp TO a LIST OF the BOOLEAN value FALSE REPEATED the LENGTH OF s PLUS one TIMES
        SET element at position zero OF dp TO the BOOLEAN value TRUE
        
        FOR index i FROM one TO the LENGTH OF s INCLUSIVE
            FOR index j FROM zero TO index i MINUS one INCLUSIVE
                IF element at position j OF dp EQUALS the BOOLEAN value TRUE AND the SUBSTRING FROM position j TO position i MINUS one OF string s IS an ELEMENT OF wordSet
                    SET element at position i OF dp TO the BOOLEAN value TRUE
                    BREAK
                END IF
            END FOR
        END FOR
        
        RETURN element at position the LENGTH OF s OF dp
    END FUNCTION
END CLASS
