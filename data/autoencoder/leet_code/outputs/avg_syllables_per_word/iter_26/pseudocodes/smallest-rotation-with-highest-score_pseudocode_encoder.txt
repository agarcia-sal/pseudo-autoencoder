CLASS Solution
    FUNCTION bestRotation(nums)
        SET n TO the LENGTH OF nums
        SET delta TO a LIST OF zero WITH THE LENGTH OF n

        FOR each index i AND each number num IN nums WITH ENUMERATE
            IF num LESS THAN OR EQUAL TO i THEN
                SET element at position zero OF delta TO element at position zero OF delta PLUS one
                IF i MINUS num PLUS one LESS THAN n THEN
                    SET element at position i MINUS num PLUS one OF delta TO element at position i MINUS num PLUS one OF delta MINUS one
                END IF
            END IF
            IF i PLUS one LESS THAN n THEN
                SET element at position i PLUS one OF delta TO element at position i PLUS one OF delta PLUS one
                IF i PLUS n MINUS num PLUS one LESS THAN n THEN
                    SET element at position i PLUS n MINUS num PLUS one OF delta TO element at position i PLUS n MINUS num PLUS one OF delta MINUS one
                END IF
            END IF
        END FOR

        SET score TO zero
        SET max_score TO zero
        SET best_k TO zero

        FOR each k FROM zero TO n MINUS one INCLUSIVE
            INCREMENT score BY element at position k OF delta
            IF score GREATER THAN max_score THEN
                SET max_score TO score
                SET best_k TO k
            END IF
        END FOR

        RETURN best_k
    END FUNCTION
END CLASS
