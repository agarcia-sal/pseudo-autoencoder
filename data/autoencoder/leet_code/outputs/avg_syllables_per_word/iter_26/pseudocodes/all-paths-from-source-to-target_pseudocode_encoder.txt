CLASS Solution
    FUNCTION allPathsSourceTarget(graph AS list of list of integers) AS list of list of integers
        SET n TO the LENGTH OF graph
        ASSIGN q TO a new double ended queue INITIALIZED WITH a list containing the single element zero
        SET ans TO an empty list
        WHILE q HOLDS
            ASSIGN path TO the FIRST element REMOVED FROM the left side OF q
            ASSIGN u TO the element at the LAST position OF path
            IF u EQUALS n MINUS one THEN
                APPEND path TO ans
                CONTINUE
            END IF
            FOR each vertex v IN the element at position u OF graph
                APPEND the concatenation of path PLUS the list containing v TO q
            END FOR
        END WHILE
        RETURN ans
    END FUNCTION
END CLASS
