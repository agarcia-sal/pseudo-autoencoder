CLASS Solution
    FUNCTION find132pattern(nums)
        SET second TO negative infinity
        SET stack TO empty list
        
        FOR each num IN the reversal of nums
            IF num LESS THAN second
                RETURN true
            END IF
            
            WHILE stack IS NOT empty AND the element at the last position OF stack LESS THAN num
                SET second TO the removal of the last element FROM stack
            END WHILE
            
            APPEND num TO stack
        END FOR
        
        RETURN false
    END FUNCTION
END CLASS
