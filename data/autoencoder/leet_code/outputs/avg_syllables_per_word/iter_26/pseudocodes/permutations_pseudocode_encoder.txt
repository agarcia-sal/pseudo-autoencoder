CLASS Solution
    FUNCTION permute(parameter nums AS List OF integers) RETURNS List OF List OF integers
        FUNCTION backtrack(parameter path, parameter remaining)
            IF remaining HOLDS NOT
                APPEND path TO permutations
                RETURN
            END IF
            FOR index FROM zero TO the LENGTH OF remaining MINUS one INCLUSIVE
                SET new_path TO path PLUS the element at position index OF remaining
                SET new_remaining TO the concatenation OF the substring FROM position zero TO position index MINUS one OF remaining AND the substring FROM position index PLUS one TO the last position OF remaining
                CALL backtrack WITH new_path AND new_remaining
            END FOR
        END FUNCTION

        SET permutations TO an empty list
        CALL backtrack WITH an empty list AND nums
        RETURN permutations
    END FUNCTION
END CLASS
