CLASS Solution
    FUNCTION minDifficulty(jobDifficulty, d)
        SET n TO the LENGTH OF jobDifficulty

        IF n LESS THAN d
            RETURN the MINUS ONE
        END IF

        FUNCTION dp(i, d)
            IF d EQUALS one
                RETURN the MAXIMUM VALUE FROM element AT position i TO the LAST element OF jobDifficulty
            END IF

            SET min_difficulty TO positive infinity
            SET current_max TO zero
            FOR index j FROM i TO n MINUS d INCLUSIVE
                SET current_max TO the MAXIMUM VALUE BETWEEN current_max AND element AT position j OF jobDifficulty
                SET min_difficulty TO the MINIMUM VALUE BETWEEN min_difficulty AND current_max PLUS CALL TO dp WITH arguments j PLUS one AND d MINUS one
            END FOR

            RETURN min_difficulty
        END FUNCTION

        RETURN CALL TO dp WITH arguments zero AND d
    END FUNCTION
END CLASS
