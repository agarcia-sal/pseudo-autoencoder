CLASS Solution
    FUNCTION rangeBitwiseAnd(parameter left WITH TYPE integer, parameter right WITH TYPE integer) RETURNS integer
        SET shift TO zero
        WHILE left IS LESS THAN right HOLDS
            SET left TO the result of integer division of left BY two
            SET right TO the result of integer division of right BY two
            INCREMENT shift BY one
        END WHILE
        RETURN the result of multiplication of left BY two RAISED TO THE POWER OF shift
    END FUNCTION
END CLASS
