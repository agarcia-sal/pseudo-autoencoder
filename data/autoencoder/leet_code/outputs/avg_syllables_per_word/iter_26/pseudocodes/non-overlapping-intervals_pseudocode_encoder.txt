CLASS Solution
    FUNCTION eraseOverlapIntervals(intervals)
        IF intervals IS EMPTY
            RETURN zero
        END IF

        SORT intervals BY the element at position two OF each interval

        SET end TO the element at position two OF the element at position one OF intervals
        SET count TO zero

        FOR index FROM two TO the LENGTH OF intervals MINUS one INCLUSIVE
            IF the element at position one OF the element at position index OF intervals IS GREATER THAN OR EQUAL TO end
                SET end TO the element at position two OF the element at position index OF intervals
            ELSE
                INCREMENT count BY one
            END IF
        END FOR

        RETURN count
    END FUNCTION
END CLASS
