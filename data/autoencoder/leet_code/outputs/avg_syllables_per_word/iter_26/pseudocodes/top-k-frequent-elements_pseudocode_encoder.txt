CLASS Solution
    FUNCTION topKFrequent(parameter nums OF type list OF integers, parameter k OF type integer) RETURNS list OF integers
        SET frequency_map TO a new default dictionary WITH default integer value
        FOR each number IN nums
            INCREMENT the element at position number OF frequency_map BY one
        END FOR

        SET max_frequency TO the largest value AMONG the values OF frequency_map
        SET buckets TO a new list CONTAINING empty lists FOR each index FROM zero TO max_frequency INCLUSIVE

        FOR each key_value_pair OF number AND frequency IN the items OF frequency_map
            APPEND number TO the element at position frequency OF buckets
        END FOR

        SET result TO an empty list
        FOR frequency FROM max_frequency DOWN TO one INCLUSIVE
            IF the LENGTH OF the element at position frequency OF buckets GREATER THAN zero HOLDS
                EXTEND result BY all elements OF the element at position frequency OF buckets
                IF the LENGTH OF result GREATER THAN OR EQUAL TO k HOLDS
                    RETURN the first k elements FROM result
                END IF
            END IF
        END FOR

        RETURN result
    END FUNCTION
END CLASS
