CLASS Solution
    FUNCTION spiralOrder(matrix)
        IF matrix IS empty OR element at position zero OF matrix IS empty
            RETURN an empty list
        END IF

        SET result TO an empty list
        SET top TO zero
        SET bottom TO the LENGTH OF matrix MINUS one
        SET left TO zero
        SET right TO the LENGTH OF element at position zero OF matrix MINUS one

        WHILE top IS LESS THAN OR EQUAL TO bottom AND left IS LESS THAN OR EQUAL TO right HOLDS
            FOR index i FROM left TO right INCLUSIVE
                APPEND element at position top OF matrix AND element at position i OF element at position top OF matrix TO result
            END FOR
            INCREMENT top BY one

            FOR index i FROM top TO bottom INCLUSIVE
                APPEND element at position i OF matrix AND element at position right OF element at position i OF matrix TO result
            END FOR
            DECREMENT right BY one

            IF top IS LESS THAN OR EQUAL TO bottom
                FOR index i FROM right DOWN TO left INCLUSIVE
                    APPEND element at position bottom OF matrix AND element at position i OF element at position bottom OF matrix TO result
                END FOR
                DECREMENT bottom BY one
            END IF

            IF left IS LESS THAN OR EQUAL TO right
                FOR index i FROM bottom DOWN TO top INCLUSIVE
                    APPEND element at position i OF matrix AND element at position left OF element at position i OF matrix TO result
                END FOR
                INCREMENT left BY one
            END IF
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
