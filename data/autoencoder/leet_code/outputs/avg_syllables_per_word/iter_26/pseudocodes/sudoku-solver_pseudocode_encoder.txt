CLASS Solution
    FUNCTION solveSudoku(board)
        FUNCTION is_valid(board, row, col, num)
            FOR i FROM zero TO eight INCLUSIVE
                IF element at position i OF element at position row OF board EQUALS num
                    RETURN FALSE
                END IF
            END FOR

            FOR i FROM zero TO eight INCLUSIVE
                IF element at position row OF element at position i OF board EQUALS num
                    RETURN FALSE
                END IF
            END FOR

            SET start_row TO three MULTIPLIED BY the INTEGER DIVISION OF row BY three
            SET start_col TO three MULTIPLIED BY the INTEGER DIVISION OF col BY three

            FOR i FROM zero TO two INCLUSIVE
                FOR j FROM zero TO two INCLUSIVE
                    IF element at position start_row PLUS i OF element at position start_col PLUS j OF board EQUALS num
                        RETURN FALSE
                    END IF
                END FOR
            END FOR

            RETURN TRUE
        END FUNCTION

        FUNCTION solve(board)
            FOR row FROM zero TO eight INCLUSIVE
                FOR col FROM zero TO eight INCLUSIVE
                    IF element at position row OF element at position col OF board EQUALS the character dot
                        FOR num IN the string representations of the numbers from one TO nine INCLUSIVE
                            IF is_valid(board, row, col, num) HOLDS TRUE
                                SET element at position row OF element at position col OF board TO num
                                IF solve(board) HOLDS TRUE
                                    RETURN TRUE
                                END IF
                                SET element at position row OF element at position col OF board TO the character dot
                            END IF
                        END FOR
                        RETURN FALSE
                    END IF
                END FOR
            END FOR
            RETURN TRUE
        END FUNCTION

        CALL solve WITH board AS argument
    END FUNCTION
END CLASS
