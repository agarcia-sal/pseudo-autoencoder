CLASS Solution
    FUNCTION wiggleSort(nums)
        CALL the SORT METHOD OF nums
        ASSIGN reference TO a COPY OF nums NAMED sorted_nums
        SET n TO the LENGTH OF nums
        SET left TO the WHOLE NUMBER RESULT OF n MINUS one DIVIDED BY two
        SET right TO n MINUS one
        FOR i FROM zero TO n MINUS one INCLUSIVE
            IF the REMAINDER OF i DIVIDED BY two EQUALS zero THEN
                SET the ELEMENT AT POSITION i OF nums TO the ELEMENT AT POSITION left OF sorted_nums
                DECREMENT left BY one
            ELSE
                SET the ELEMENT AT POSITION i OF nums TO the ELEMENT AT POSITION right OF sorted_nums
                DECREMENT right BY one
            END IF
        END FOR
    END FUNCTION
END CLASS
