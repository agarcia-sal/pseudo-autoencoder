CLASS Solution
    FUNCTION threeSumClosest(nums WITH TYPE List OF INTEGER, target WITH TYPE INTEGER) RETURNS INTEGER
        CALL nums TO SORT THEMSELVES IN ASCENDING ORDER
        SET closest_sum TO a positive infinity value

        FOR index i FROM zero TO the LENGTH OF nums MINUS two INCLUSIVE
            SET left TO i PLUS one
            SET right TO the LENGTH OF nums MINUS one

            WHILE left IS LESS THAN right HOLDS
                SET current_sum TO the ELEMENT at position i OF nums PLUS the ELEMENT at position left OF nums PLUS the ELEMENT at position right OF nums

                IF the ABSOLUTE VALUE OF current_sum MINUS target IS LESS THAN the ABSOLUTE VALUE OF closest_sum MINUS target THEN
                    SET closest_sum TO current_sum
                END IF

                IF current_sum IS LESS THAN target THEN
                    INCREMENT left BY one
                ELSE IF current_sum IS GREATER THAN target THEN
                    DECREMENT right BY one
                ELSE
                    RETURN current_sum
                END IF
            END WHILE
        END FOR

        RETURN closest_sum
    END FUNCTION
END CLASS
