CLASS Solution
    FUNCTION monotoneIncreasingDigits(parameter n)
        SET digits TO LIST OF CHARACTERS OF THE STRING REPRESENTATION OF n
        SET length TO the LENGTH OF digits
        SET marker TO length
        FOR index i FROM the LENGTH OF digits MINUS one DOWN TO one INCLUSIVE DECREASING BY one
            IF the element at position i OF digits IS LESS THAN the element at position i MINUS one OF digits
                SET the element at position i MINUS one OF digits TO the STRING REPRESENTATION OF the INTEGER VALUE OF the element at position i MINUS one OF digits MINUS one
                SET marker TO i
            END IF
        END FOR
        FOR index i FROM marker TO length MINUS one INCLUSIVE
            SET the element at position i OF digits TO the CHARACTER NINE
        END FOR
        RETURN the INTEGER VALUE OF THE CONCATENATION OF ALL ELEMENTS OF digits
    END FUNCTION
END CLASS
