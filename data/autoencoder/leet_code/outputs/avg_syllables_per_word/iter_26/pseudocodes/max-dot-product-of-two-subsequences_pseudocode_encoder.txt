CLASS Solution
    FUNCTION maxDotProduct(nums1, nums2)
        SET m TO the LENGTH OF nums1
        SET n TO the LENGTH OF nums2
        
        SET dp TO a list of lists where each inner list contains the very smallest possible number repeated the LENGTH OF nums2 PLUS one times and this is done for the LENGTH OF nums1 PLUS one times
        
        FOR i FROM one TO m INCLUSIVE
            FOR j FROM one TO n INCLUSIVE
                SET current_product TO element at position i MINUS one OF nums1 MULTIPLIED BY element at position j MINUS one OF nums2
                
                SET candidate_1 TO current_product
                SET candidate_2 TO element at position i MINUS one OF dp INDEX j
                SET candidate_3 TO element at position i OF dp INDEX j MINUS one
                SET candidate_4 TO element at position i MINUS one OF dp INDEX j MINUS one PLUS current_product
                
                SET element at position i OF dp INDEX j TO the MAXIMUM OF candidate_1 AND candidate_2 AND candidate_3 AND candidate_4
            END FOR
        END FOR
        
        RETURN element at position m OF dp INDEX n
    END FUNCTION
END CLASS
