CLASS Solution
    FUNCTION findMinArrowShots(points of list of list of integers)
        PERFORM SORT ON points BY THE element at position two OF each point
        SET arrows TO zero
        SET last_arrow_position TO negative infinity value
        FOR each balloon IN points
            IF the element at position one OF balloon GREATER THAN last_arrow_position HOLDS
                INCREMENT arrows BY one
                SET last_arrow_position TO the element at position two OF balloon
            END IF
        END FOR
        RETURN arrows
    END FUNCTION
END CLASS
