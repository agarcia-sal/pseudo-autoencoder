CLASS Solution
    FUNCTION isOneEditDistance(s STRING, t STRING) RETURNS BOOLEAN
        IF the LENGTH OF s GREATER THAN the LENGTH OF t
            RETURN the RESULT OF calling isOneEditDistance WITH PARAMETERS t AND s
        END IF

        SET diff TO the LENGTH OF t MINUS the LENGTH OF s

        IF diff GREATER THAN one
            RETURN FALSE
        END IF

        IF diff EQUALS zero
            RETURN the RESULT OF checking that the COUNT OF positions WHERE the character at position index OF s NOT EQUALS the character at position index OF t IS exactly one
        ELSE
            FOR index FROM zero TO the LENGTH OF s MINUS one INCLUSIVE
                IF the character at position index OF s NOT EQUALS the character at position index OF t
                    RETURN the RESULT OF checking that the SUBSTRING FROM position index TO the END OF s EQUALS the SUBSTRING FROM position index plus one TO the END OF t
                END IF
            END FOR
            RETURN TRUE
        END IF
    END FUNCTION
END CLASS
