CLASS Solution
    FUNCTION firstMissingPositive(nums)
        SET n TO the LENGTH OF nums
        
        FOR i FROM zero TO n MINUS one INCLUSIVE
            WHILE the element at position i OF nums IS GREATER THAN OR EQUAL TO one AND the element at position i OF nums IS LESS THAN OR EQUAL TO n AND the element at position the element at position i OF nums MINUS one OF nums IS NOT EQUALS the element at position i OF nums HOLDS
                SET correct_index TO the element at position i OF nums MINUS one
                SET temporary_value TO the element at position i OF nums
                SET the element at position i OF nums TO the element at position correct_index OF nums
                SET the element at position correct_index OF nums TO temporary_value
            END WHILE
        END FOR
        
        FOR i FROM zero TO n MINUS one INCLUSIVE
            IF the element at position i OF nums IS NOT EQUALS i PLUS one THEN
                RETURN i PLUS one
            END IF
        END FOR
        
        RETURN n PLUS one
    END FUNCTION
END CLASS
