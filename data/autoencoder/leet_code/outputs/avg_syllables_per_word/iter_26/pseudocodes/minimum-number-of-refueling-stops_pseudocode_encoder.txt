CLASS Solution
    FUNCTION minRefuelStops(target, startFuel, stations)
        SET max_heap TO an empty list
        SET current_fuel TO startFuel
        SET stops TO zero
        APPEND a list composed of target and zero TO stations

        FOR each pair named position and fuel IN stations
            WHILE max_heap IS NOT empty AND current_fuel IS LESS THAN position
                SET largest_fuel TO the negative of the element REMOVED FROM max_heap AS a heap
                DECREMENT current_fuel BY largest_fuel
                INCREMENT stops BY one
            END WHILE

            IF current_fuel IS LESS THAN position
                RETURN negative one
            END IF

            ADD the negative of fuel TO max_heap AS a heap element
        END FOR

        RETURN stops
    END FUNCTION
END CLASS
