CLASS Solution
    FUNCTION orderOfLargestPlusSign(n, mines)
        SET mines_set TO a new empty set
        FOR each mine IN mines
            APPEND the tuple formed by the element at position zero OF mine and the element at position one OF mine TO mines_set
        END FOR

        SET dp TO a new list containing n elements each being a list containing n elements each being a list of four zeros representing counts in directions right down left up

        SET directions TO a list containing the ordered pairs representing offsets zero one one zero zero minus one minus one zero

        FOR i FROM zero TO n MINUS one INCLUSIVE
            FOR j FROM zero TO n MINUS one INCLUSIVE
                IF the tuple of i and j NOT IN mines_set HOLDS
                    IF j GREATER THAN zero
                        SET element at position zero OF element at position j OF element at position i OF dp TO the element at position zero OF element at position j MINUS one OF element at position i OF dp PLUS one
                    END IF
                    IF i GREATER THAN zero
                        SET element at position one OF element at position j OF element at position i OF dp TO the element at position one OF element at position j OF element at position i MINUS one OF dp PLUS one
                    END IF
                END IF
            END FOR
        END FOR

        SET result TO zero

        FOR i FROM n MINUS one TO zero STEP MINUS one INCLUSIVE
            FOR j FROM n MINUS one TO zero STEP MINUS one INCLUSIVE
                IF the tuple of i and j NOT IN mines_set HOLDS
                    IF j LESS THAN n MINUS one
                        SET element at position two OF element at position j OF element at position i OF dp TO the element at position two OF element at position j PLUS one OF element at position i OF dp PLUS one
                    END IF
                    IF i LESS THAN n MINUS one
                        SET element at position three OF element at position j OF element at position i OF dp TO the element at position three OF element at position j OF element at position i PLUS one OF dp PLUS one
                    END IF
                    SET plus_sign_order TO the minimum value among all four elements of element at position j OF element at position i OF dp
                    IF plus_sign_order GREATER THAN result
                        SET result TO plus_sign_order
                    END IF
                END IF
            END FOR
        END FOR

        RETURN result
    END FUNCTION
END CLASS
