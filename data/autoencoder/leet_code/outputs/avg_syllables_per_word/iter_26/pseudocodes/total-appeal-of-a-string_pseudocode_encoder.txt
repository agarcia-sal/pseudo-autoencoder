CLASS Solution
    FUNCTION appealSum(string s)
        SET last_seen_indices TO a list of length twenty six FILLED WITH negative one
        SET total_appeal_value TO zero
        SET current_appeal_value TO zero
        
        FOR each position i AND character char IN the enumeration of string s
            SET character_index TO the Unicode code point of character char MINUS the Unicode code point of lowercase letter a
            INCREMENT current_appeal_value BY position i MINUS the element at position character_index OF last_seen_indices
            INCREMENT total_appeal_value BY current_appeal_value
            SET the element at position character_index OF last_seen_indices TO position i
        END FOR
        
        RETURN total_appeal_value
    END FUNCTION
END CLASS
