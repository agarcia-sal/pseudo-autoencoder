CLASS Solution
    FUNCTION permuteUnique(list_of_numbers)
        FUNCTION backtrack(current_path, list_of_used_flags, list_of_results)
            IF the LENGTH OF current_path EQUALS the LENGTH OF list_of_numbers
                APPEND a COPY OF current_path TO list_of_results
                RETURN
            END IF

            FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one INCLUSIVE
                IF the element at position index OF list_of_used_flags IS TRUE OR 
                   (index GREATER THAN zero AND 
                    the element at position index OF list_of_numbers EQUALS the element at position index MINUS one OF list_of_numbers AND 
                    the element at position index MINUS one OF list_of_used_flags IS FALSE)
                    CONTINUE TO next iteration
                END IF

                SET the element at position index OF list_of_used_flags TO TRUE
                APPEND the element at position index OF list_of_numbers TO current_path
                CALL backtrack WITH current_path, list_of_used_flags, list_of_results
                REMOVE the last element FROM current_path
                SET the element at position index OF list_of_used_flags TO FALSE
            END FOR
        END FUNCTION

        SORT list_of_numbers IN ascending order
        SET list_of_results TO an empty list
        SET list_of_used_flags TO a list of FALSE values with the same LENGTH AS list_of_numbers
        CALL backtrack WITH an empty list, list_of_used_flags, list_of_results
        RETURN list_of_results
    END FUNCTION
END CLASS
