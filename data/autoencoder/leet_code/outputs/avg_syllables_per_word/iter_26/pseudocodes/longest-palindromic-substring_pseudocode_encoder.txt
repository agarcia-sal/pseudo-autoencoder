CLASS Solution
    FUNCTION longestPalindrome(string s)
        SET n TO the LENGTH OF s
        SET f TO a TWO DIMENSIONAL list WHERE each element IS TRUE with dimensions n BY n
        SET k TO zero
        SET mx TO one
        FOR i FROM n MINUS two TO zero INCLUSIVE DECREMENTING BY one
            FOR j FROM i PLUS one TO n MINUS one INCLUSIVE
                SET element at position j OF element at position i OF f TO FALSE
                IF element at position i OF s EQUALS element at position j OF s
                    SET element at position j OF element at position i OF f TO element at position j MINUS one OF element at position i PLUS one OF f
                    IF element at position j OF element at position i OF f AND mx LESS THAN j MINUS i PLUS one
                        SET k TO i
                        SET mx TO j MINUS i PLUS one
                    END IF
                END IF
            END FOR
        END FOR
        RETURN the SUBSTRING OF s FROM position k TO position k PLUS mx EXCLUSIVE
    END FUNCTION
END CLASS
