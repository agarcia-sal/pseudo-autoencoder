CLASS Solution
    FUNCTION maxHeight(cuboids)
        FOR each cuboid IN cuboids
            CALL sort ON cuboid TO ARRANGE its dimensions IN ascending order
        END FOR
        
        CALL sort ON cuboids TO ARRANGE all cuboids IN lexicographical order
        
        SET number_of_cuboids TO the LENGTH OF cuboids
        INITIALIZE dp_list AS a list OF zeros WITH the LENGTH OF number_of_cuboids
        
        FOR index i FROM zero TO number_of_cuboids MINUS one INCLUSIVE
            SET element_at_position_i OF dp_list TO element at position three OF cuboid at position i IN cuboids
            
            FOR index j FROM zero TO i MINUS one INCLUSIVE
                IF element at position one OF cuboid at position j IN cuboids LESS THAN OR EQUAL TO element at position one OF cuboid at position i IN cuboids
                    AND element at position two OF cuboid at position j IN cuboids LESS THAN OR EQUAL TO element at position two OF cuboid at position i IN cuboids
                    AND element at position three OF cuboid at position j IN cuboids LESS THAN OR EQUAL TO element at position three OF cuboid at position i IN cuboids
                THEN
                    SET element_at_position_i OF dp_list TO the GREATER VALUE BETWEEN element_at_position_i OF dp_list AND element_at_position_j OF dp_list PLUS element at position three OF cuboid at position i IN cuboids
                END IF
            END FOR
        END FOR
        
        SET maximum_height TO the GREATER VALUE IN dp_list
        RETURN maximum_height
    END FUNCTION
END CLASS
