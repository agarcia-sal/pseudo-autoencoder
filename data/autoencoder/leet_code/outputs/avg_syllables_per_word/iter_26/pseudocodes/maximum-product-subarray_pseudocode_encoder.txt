CLASS Solution
    FUNCTION maxProduct(nums)
        IF nums IS EMPTY
            RETURN zero
        END IF

        SET max_product TO element at position zero OF nums
        SET min_product TO element at position zero OF nums
        SET result TO element at position zero OF nums

        FOR each number IN the elements of nums EXCLUDING the first element
            IF number IS LESS THAN zero
                SET temporary_reference TO max_product
                SET max_product TO min_product
                SET min_product TO temporary_reference
            END IF

            SET max_product TO the GREATER OF number AND max_product MULTIPLIED BY number
            SET min_product TO the LESSER OF number AND min_product MULTIPLIED BY number

            SET result TO the GREATER OF result AND max_product
        END FOR

        RETURN result
    END FUNCTION
END CLASS
