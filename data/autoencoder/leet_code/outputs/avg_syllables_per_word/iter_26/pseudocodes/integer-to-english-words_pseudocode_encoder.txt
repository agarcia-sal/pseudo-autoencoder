CLASS Solution
    FUNCTION numberToWords(num)
        IF num EQUALS zero
            RETURN the word Zero
        END IF

        SET below_twenty TO list holding empty string, and words One, Two, Three, Four, Five, Six, Seven, Eight, Nine, Ten,
            Eleven, Twelve, Thirteen, Fourteen, Fifteen, Sixteen, Seventeen, Eighteen, and Nineteen respectively
        SET tens TO list holding empty string, empty string, and words Twenty, Thirty, Forty, Fifty, Sixty, Seventy, Eighty, and Ninety respectively
        SET thousands TO list holding empty string, and words Thousand, Million, and Billion respectively

        FUNCTION helper(n)
            IF n EQUALS zero
                RETURN empty string
            ELSE IF n LESS THAN twenty
                RETURN the element at position n OF below_twenty PLUS a space
            ELSE IF n LESS THAN one hundred
                RETURN the element at position the integer division of n BY ten OF tens PLUS a space PLUS helper of the remainder of n DIVIDED BY ten
            ELSE
                RETURN the element at position the integer division of n BY one hundred OF below_twenty PLUS the word Hundred PLUS a space PLUS helper of the remainder of n DIVIDED BY one hundred
            END IF
        END FUNCTION

        SET result TO empty string
        SET i TO zero

        WHILE num GREATER THAN zero HOLDS
            IF the remainder of num DIVIDED BY one thousand NOT EQUALS zero
                SET result TO helper of the remainder of num DIVIDED BY one thousand PLUS the element at position i OF thousands PLUS a space PLUS result
            END IF
            SET num TO the integer division of num BY one thousand
            INCREMENT i BY one
        END WHILE

        RETURN result with surrounding spaces removed
    END FUNCTION
END CLASS
