CLASS Solution
    FUNCTION new21Game(n, k, maxPts)
        FUNCTION dfs(i)
            IF i GREATER THAN OR EQUAL TO k THEN
                RETURN the truth value OF i LESS THAN OR EQUAL TO n
            END IF

            IF i EQUALS k MINUS one THEN
                RETURN the smaller value BETWEEN n MINUS k PLUS one AND maxPts DIVIDED BY maxPts
            END IF

            RETURN the sum of dfs of i INCREMENTED BY one AND the difference BETWEEN dfs of i INCREMENTED BY one AND dfs of i INCREMENTED BY maxPts PLUS one DIVIDED BY maxPts
        END FUNCTION

        RETURN dfs of zero
    END FUNCTION
END CLASS
