CLASS Solution
    FUNCTION minMoves(list_of_numbers, integer_k)
        SET list_of_positions TO an empty list
        FOR each index AND each number IN enumeration of list_of_numbers
            IF number EQUALS one
                APPEND index TO list_of_positions
            END IF
        END FOR

        FUNCTION calculate_cost(integer_start, integer_end)
            SET integer_mid TO integer_start PLUS integer_end DIVIDED BY two
            SET integer_median TO element at position integer_mid OF list_of_positions
            SET integer_cost TO zero
            FOR each integer_i FROM integer_start TO integer_end MINUS one INCLUSIVE
                SET integer_cost TO integer_cost PLUS the absolute value of element at position integer_i OF list_of_positions MINUS integer_median MINUS integer_mid MINUS integer_i
            END FOR
            RETURN integer_cost
        END FUNCTION

        SET minimal_cost TO positive infinity
        FOR each integer_i FROM zero TO the LENGTH OF list_of_positions MINUS integer_k
            SET minimal_cost TO the lesser BETWEEN minimal_cost AND calculate_cost(integer_i, integer_i PLUS integer_k)
        END FOR

        RETURN minimal_cost
    END FUNCTION
END CLASS
