CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS NOT PRESENT
        RETURN none
    END IF
    SET root TO A NEW INSTANCE OF TreeNode WITH element at position zero OF values
    SET i TO one
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO THE ELEMENT REMOVED FROM THE LEFT OF queue
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none
            SET node.left TO A NEW INSTANCE OF TreeNode WITH element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none
            SET node.right TO A NEW INSTANCE OF TreeNode WITH element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS NOT PRESENT AND q IS NOT PRESENT
        RETURN true
    ELSE IF p IS NOT PRESENT OR q IS NOT PRESENT
        RETURN false
    ELSE IF p.val IS NOT EQUALS q.val
        RETURN false
    ELSE
        RETURN is_same_tree(element left OF p, element left OF q) AND is_same_tree(element right OF p, element right OF q)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION findBottomLeftValue(self, root)
        SET queue TO A DOUBLE ENDED QUEUE INITIALIZED WITH root
        SET leftmost_value TO none
        WHILE queue IS NOT EMPTY
            SET level_size TO THE LENGTH OF queue
            FOR i FROM zero TO level_size MINUS one INCLUSIVE
                SET node TO THE ELEMENT REMOVED FROM THE LEFT OF queue
                IF i EQUALS zero
                    SET leftmost_value TO node.val
                END IF
                IF node.left IS PRESENT
                    APPEND node.left TO queue
                END IF
                IF node.right IS PRESENT
                    APPEND node.right TO queue
                END IF
            END FOR
        END WHILE
        RETURN leftmost_value
    END FUNCTION
END CLASS
