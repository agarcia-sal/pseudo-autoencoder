CLASS Solution
    FUNCTION minDistance(houses, k)
        CALL sort ON houses
        SET n TO the LENGTH OF houses
        
        SET cost TO a two dimensional list of zeroes with n rows and n columns
        FOR i FROM n MINUS one DOWNTO zero INCLUSIVE
            FOR j FROM i PLUS one TO n MINUS one INCLUSIVE
                SET element at position i and position j OF cost TO element at position i PLUS one and position j MINUS one OF cost PLUS element at position j OF houses MINUS element at position i OF houses
            END FOR
        END FOR
        
        SET dp TO a two dimensional list filled with infinity values with n PLUS one rows and k PLUS one columns
        SET element at position zero and position zero OF dp TO zero
        
        FOR i FROM one TO n INCLUSIVE
            FOR j FROM one TO k INCLUSIVE
                FOR p FROM zero TO i MINUS one INCLUSIVE
                    SET element at position i and position j OF dp TO the MINIMUM OF element at position i and position j OF dp AND element at position p and position j MINUS one OF dp PLUS element at position p and position i MINUS one OF cost
                END FOR
            END FOR
        END FOR
        
        RETURN element at position n and position k OF dp
    END FUNCTION
END CLASS
