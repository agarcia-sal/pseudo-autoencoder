CLASS Solution
    FUNCTION numberOfWays(corridor)
        SET MODULO TO ten raised to the power of nine PLUS one
        SET seats TO an empty list
        FOR each position indexed BY index AND character indexed BY character IN corridor
            IF character EQUALS the uppercase letter ESS
                APPEND index TO seats
            END IF
        END FOR

        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two
            RETURN zero
        END IF

        SET ways TO one
        FOR index FROM two TO the LENGTH OF seats MINUS one IN STEPS OF two
            SET ways TO ways MULTIPLIED BY the difference BETWEEN the element at position index OF seats MINUS the element at position index MINUS one OF seats
            SET ways TO ways MODULO MODULO
        END FOR

        RETURN ways
    END FUNCTION
END CLASS
