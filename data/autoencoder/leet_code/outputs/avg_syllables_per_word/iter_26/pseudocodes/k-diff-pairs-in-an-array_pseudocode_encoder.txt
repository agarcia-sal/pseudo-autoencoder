CLASS Solution
    FUNCTION findPairs(nums, k)
        IF k LESS THAN zero HOLDS
            RETURN zero
        END IF

        SET num_counts TO an empty mapping object
        FOR each num IN nums
            SET current_count TO the VALUE CORRESPONDING TO num IN num_counts OR zero IF num IS NOT PRESENT
            SET updated_count TO current_count PLUS one
            ASSIGN num_counts AT num TO updated_count
        END FOR

        SET unique_pairs TO zero
        FOR each num IN the KEYS OF num_counts
            IF k EQUALS zero HOLDS
                IF the VALUE CORRESPONDING TO num IN num_counts GREATER THAN one HOLDS
                    INCREMENT unique_pairs BY one
                END IF
            ELSE
                SET target_num TO num PLUS k
                IF target_num IS A KEY IN num_counts HOLDS
                    INCREMENT unique_pairs BY one
                END IF
            END IF
        END FOR

        RETURN unique_pairs
    END FUNCTION
END CLASS
