CLASS Solution
    FUNCTION minArea(image, x, y)
        IF image IS EMPTY OR the LENGTH OF image AT position zero IS ZERO
            RETURN zero
        END IF

        SET m TO the LENGTH OF image
        SET n TO the LENGTH OF image AT position zero

        FUNCTION hasBlackPixelInRow(row)
            FOR each pixel IN the element at position row OF image
                IF pixel EQUALS one
                    RETURN TRUE
                END IF
            END FOR
            RETURN FALSE
        END FUNCTION

        FUNCTION hasBlackPixelInColumn(col)
            FOR row FROM zero TO m MINUS one INCLUSIVE
                IF the element at position col OF the element at position row OF image EQUALS one
                    RETURN TRUE
                END IF
            END FOR
            RETURN FALSE
        END FUNCTION

        FUNCTION findTop()
            SET low TO zero
            SET high TO x
            WHILE low LESS THAN high HOLDS
                SET mid TO the FLOOR OF low PLUS high DIVIDED BY two
                IF hasBlackPixelInRow(mid) IS TRUE
                    SET high TO mid
                ELSE
                    SET low TO mid PLUS one
                END IF
            END WHILE
            RETURN low
        END FUNCTION

        FUNCTION findBottom()
            SET low TO x
            SET high TO m MINUS one
            WHILE low LESS THAN high HOLDS
                SET mid TO the FLOOR OF low PLUS high PLUS one DIVIDED BY two
                IF hasBlackPixelInRow(mid) IS TRUE
                    SET low TO mid
                ELSE
                    SET high TO mid MINUS one
                END IF
            END WHILE
            RETURN low
        END FUNCTION

        FUNCTION findLeft()
            SET low TO zero
            SET high TO y
            WHILE low LESS THAN high HOLDS
                SET mid TO the FLOOR OF low PLUS high DIVIDED BY two
                IF hasBlackPixelInColumn(mid) IS TRUE
                    SET high TO mid
                ELSE
                    SET low TO mid PLUS one
                END IF
            END WHILE
            RETURN low
        END FUNCTION

        FUNCTION findRight()
            SET low TO y
            SET high TO n MINUS one
            WHILE low LESS THAN high HOLDS
                SET mid TO the FLOOR OF low PLUS high PLUS one DIVIDED BY two
                IF hasBlackPixelInColumn(mid) IS TRUE
                    SET low TO mid
                ELSE
                    SET high TO mid MINUS one
                END IF
            END WHILE
            RETURN low
        END FUNCTION

        SET top TO findTop()
        SET bottom TO findBottom()
        SET left TO findLeft()
        SET right TO findRight()

        RETURN the value of bottom MINUS top PLUS one MULTIPLIED BY right MINUS left PLUS one
    END FUNCTION
END CLASS
