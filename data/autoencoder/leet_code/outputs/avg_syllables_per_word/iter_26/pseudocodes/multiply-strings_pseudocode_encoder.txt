CLASS Solution
    FUNCTION multiply(num1, num2)
        IF num1 EQUALS string zero OR num2 EQUALS string zero
            RETURN string zero
        END IF

        SET result TO a list containing zero repeated the sum of the LENGTH OF num1 AND the LENGTH OF num2 times

        SET num1 TO the reverse order of characters IN num1
        SET num2 TO the reverse order of characters IN num2

        FOR index i FROM zero TO the LENGTH OF num1 MINUS one INCLUSIVE
            FOR index j FROM zero TO the LENGTH OF num2 MINUS one INCLUSIVE
                SET mul TO the result of (the numeric code of the element at position i OF num1 MINUS the numeric code of character zero) MULTIPLIED BY (the numeric code of the element at position j OF num2 MINUS the numeric code of character zero)
                SET p1 TO i PLUS j
                SET p2 TO i PLUS j PLUS one
                SET sum_ TO mul PLUS the element at position p1 OF result

                SET the element at position p1 OF result TO sum_ MODULO ten
                INCREMENT the element at position p2 OF result BY sum_ DIVIDED BY ten
            END FOR
        END FOR

        SET result_str TO the concatenation of the string representation of each element IN result traversed in reverse order

        RETURN result_str with all leading characters equal to string zero REMOVED
    END FUNCTION
END CLASS
