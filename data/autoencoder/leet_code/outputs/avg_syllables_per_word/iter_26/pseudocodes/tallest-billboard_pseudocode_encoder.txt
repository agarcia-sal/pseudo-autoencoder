CLASS Solution
    FUNCTION tallestBillboard(rods)
        SET dp TO a dictionary WITH zero MAPPED TO zero
        
        FOR each rod IN rods
            SET new_dp TO a copy OF dp
            FOR each key_value_pair IN dp
                ASSIGN d TO the key OF key_value_pair
                ASSIGN y TO the value OF key_value_pair
                
                SET new_key_taller TO d PLUS rod
                SET new_value_taller TO the maximum OF the value MAPPED BY new_key_taller IN new_dp OR zero AND y
                ASSIGN new_value_taller TO the value AT new_key_taller IN new_dp
                
                SET new_key_shorter TO the absolute value OF d MINUS rod
                SET new_value_shorter TO the maximum OF the value MAPPED BY new_key_shorter IN new_dp OR zero AND y PLUS the minimum OF d AND rod
                ASSIGN new_value_shorter TO the value AT new_key_shorter IN new_dp
            END FOR
            ASSIGN dp TO new_dp
        END FOR
        
        RETURN the value AT zero IN dp
    END FUNCTION
END CLASS
