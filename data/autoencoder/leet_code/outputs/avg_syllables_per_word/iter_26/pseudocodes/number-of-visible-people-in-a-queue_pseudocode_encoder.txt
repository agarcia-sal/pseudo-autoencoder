CLASS Solution
    FUNCTION canSeePersonsCount(parameter named heights AS list of integers)
        SET variable n TO the LENGTH OF parameter named heights
        SET variable answer TO a list containing zero repeated n times
        SET variable stack TO an empty list

        FOR index i FROM zero TO n MINUS one INCLUSIVE
            WHILE stack HOLDS AND element at position element at position negative one OF stack OF parameter named heights IS LESS THAN element at position i OF parameter named heights
                INCREMENT element at position stack POPPED FROM stack OF variable answer BY one
            END WHILE
            IF stack HOLDS
                INCREMENT element at position element at position negative one OF stack OF variable answer BY one
            END IF
            APPEND i TO variable stack
        END FOR

        RETURN variable answer
    END FUNCTION
END CLASS
