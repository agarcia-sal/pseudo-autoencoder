CLASS Solution
    FUNCTION findRightInterval(intervals)
        SET start_times TO a list of pairs of the starting element OF each interval and its position IN intervals SORTED BY the starting element
        SET result TO an empty list
        FOR each pair of ignored_value and end_value IN intervals
            SET search_index TO the position returned by the binary search that finds the lowest position WHERE the starting element is GREATER THAN OR EQUAL TO end_value IN start_times
            IF search_index IS LESS THAN the LENGTH OF start_times
                APPEND the second element AT position search_index OF start_times TO result
            ELSE
                APPEND the description of minus one TO result
            END IF
        END FOR
        RETURN result
    END FUNCTION
END CLASS
