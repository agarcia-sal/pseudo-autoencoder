CLASS Solution
    FUNCTION jobScheduling(startTime, endTime, profit)
        SET jobs TO the list of tuples each containing an element at position index of startTime, endTime, and profit RESPECTIVELY SORTED BY the second element of each tuple IN ASCENDING ORDER
        
        SET end_times TO an empty list
        FOR each job IN jobs
            APPEND the second element of job TO end_times
        END FOR
        
        SET dp TO a list of zeroes with the LENGTH OF jobs PLUS one
        
        FOR index i FROM one TO the LENGTH OF jobs INCLUSIVE
            SET j TO the result of performing a binary search for the rightmost position to insert the element at the first position of the job at position i MINUS one IN end_times
            
            SET dp at position i TO the greater value BETWEEN dp at position i MINUS one AND the sum of dp at position j AND the third element of the job at position i MINUS one
        END FOR
        
        RETURN the element at the last position OF dp
    END FUNCTION
END CLASS
