CLASS Solution
    FUNCTION isNStraightHand(hand OF List OF integers, groupSize OF integer) RETURNS boolean
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero
            RETURN False
        END IF

        SET count TO the COUNTER COLLECTION OF hand
        SET unique_cards TO the SORTED LIST OF the KEYS OF count

        FOR each card IN unique_cards
            IF the ELEMENT AT POSITION card OF count GREATER THAN zero
                SET needed TO the ELEMENT AT POSITION card OF count
                FOR index FROM zero TO groupSize MINUS one INCLUSIVE
                    IF the ELEMENT AT POSITION card PLUS index OF count LESS THAN needed
                        RETURN False
                    END IF
                    DECREMENT the ELEMENT AT POSITION card PLUS index OF count BY needed
                END FOR
            END IF
        END FOR

        RETURN True
    END FUNCTION
END CLASS
