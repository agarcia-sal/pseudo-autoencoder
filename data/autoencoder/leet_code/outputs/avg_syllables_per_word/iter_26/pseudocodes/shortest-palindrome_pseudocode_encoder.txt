CLASS Solution
    FUNCTION shortestPalindrome(s)
        IF s EQUALS the empty string
            RETURN s
        END IF

        FUNCTION is_palindrome(check_str)
            RETURN check_str EQUALS the reverse OF check_str
        END FUNCTION

        FOR index FROM the LENGTH OF s DOWN TO zero INCLUSIVE
            IF is_palindrome(the substring FROM the first character TO the position index MINUS one OF s)
                SET to_add TO the reverse OF the substring FROM position index TO the last character OF s
                RETURN the concatenation OF to_add FOLLOWED BY s
            END IF
        END FOR
    END FUNCTION
END CLASS
