CLASS Solution
    FUNCTION maxKilledEnemies(grid)
        IF grid EQUALS empty OR element at position zero OF grid EQUALS empty
            RETURN zero
        END IF

        SET m TO the LENGTH OF grid
        SET n TO the LENGTH OF element at position zero OF grid
        SET max_kills TO zero

        SET row_kills TO zero
        SET col_kills TO a LIST OF zeros WITH the LENGTH OF n

        FOR i FROM zero TO m MINUS one INCLUSIVE
            FOR j FROM zero TO n MINUS one INCLUSIVE
                IF j EQUALS zero OR element at position j MINUS one OF element at position i OF grid EQUALS the character W
                    SET row_kills TO zero
                    FOR k FROM j TO n MINUS one INCLUSIVE
                        IF element at position k OF element at position i OF grid EQUALS the character E
                            INCREMENT row_kills BY one
                        ELSE IF element at position k OF element at position i OF grid EQUALS the character W
                            BREAK
                        END IF
                    END FOR
                END IF

                IF i EQUALS zero OR element at position i MINUS one OF element at position j OF grid EQUALS the character W
                    SET element at position j OF col_kills TO zero
                    FOR k FROM i TO m MINUS one INCLUSIVE
                        IF element at position k OF element at position j OF grid EQUALS the character E
                            INCREMENT element at position j OF col_kills BY one
                        ELSE IF element at position k OF element at position j OF grid EQUALS the character W
                            BREAK
                        END IF
                    END FOR
                END IF

                IF element at position j OF element at position i OF grid EQUALS the character zero
                    SET max_kills TO the GREATER OF max_kills AND row_kills PLUS element at position j OF col_kills
                END IF
            END FOR
        END FOR

        RETURN max_kills
    END FUNCTION
END CLASS
