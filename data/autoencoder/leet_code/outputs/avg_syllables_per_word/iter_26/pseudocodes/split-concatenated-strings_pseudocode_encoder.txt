CLASS Solution
    FUNCTION splitLoopedString(strs AS List OF strings) RETURNS string
        FOR index FROM zero TO the LENGTH OF strs MINUS one INCLUSIVE
            SET reversed_string TO the ELEMENT AT position index OF strs REVERSED
            IF reversed_string GREATER THAN the ELEMENT AT position index OF strs THEN
                SET the ELEMENT AT position index OF strs TO reversed_string
            END IF
        END FOR

        SET doubled_strs TO the CONCATENATION OF strs AND strs

        SET max_string TO the empty string

        FOR index FROM zero TO the LENGTH OF strs MINUS one INCLUSIVE
            FOR each is_reversed IN the list containing FALSE and TRUE
                IF is_reversed EQUALS TRUE THEN
                    SET current TO the ELEMENT AT position index OF strs REVERSED
                ELSE
                    SET current TO the ELEMENT AT position index OF strs
                END IF

                SET remaining TO the CONCATENATION OF the ELEMENTS FROM position index PLUS one TO position index PLUS the LENGTH OF strs MINUS one INCLUSIVE OF doubled_strs

                FOR cut_position FROM zero TO the LENGTH OF current INCLUSIVE
                    SET candidate TO the CONCATENATION OF the SUBSTRING FROM cut_position TO the END OF current THEN remaining THEN the SUBSTRING FROM zero TO cut_position MINUS one OF current
                    IF candidate GREATER THAN max_string THEN
                        SET max_string TO candidate
                    END IF
                END FOR
            END FOR
        END FOR

        RETURN max_string
    END FUNCTION
END CLASS
