CLASS Solution
    FUNCTION findLongestWord(s, dictionary)
        FUNCTION is_subsequence(word, s)
            ASSIGN iterator TO an iterator over s
            RETURN whether for every character in word the character IS found IN iterator IN ORDER
        END FUNCTION

        SORT dictionary BY the NEGATIVE OF the LENGTH OF the word AND THEN BY the word in ASCENDING LEXICOGRAPHICAL ORDER

        FOR each word IN dictionary
            IF is_subsequence(word, s) HOLDS
                RETURN word
            END IF
        END FOR

        RETURN an empty string
    END FUNCTION
END CLASS
