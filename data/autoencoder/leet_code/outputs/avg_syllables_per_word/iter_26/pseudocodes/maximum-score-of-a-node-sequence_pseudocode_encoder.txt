CLASS Solution
    FUNCTION maximumScore(scores, edges)
        SET graph TO a new mapping with default empty list
        FOR each pair of elements u and v IN edges
            APPEND a tuple consisting of element at position v OF scores AND v TO the list at key u IN graph
            APPEND a tuple consisting of element at position u OF scores AND u TO the list at key v IN graph
        END FOR

        FOR each node IN the keys OF graph
            SET the list at key node IN graph TO the three largest elements from the list at key node IN graph based on the first element of each tuple
        END FOR

        SET max_score TO negative one

        FOR each pair of elements u and v IN edges
            FOR each tuple consisting of score1 and x IN the list at key u IN graph
                FOR each tuple consisting of score2 and y IN the list at key v IN graph
                    IF the size OF the set consisting of x, y, u, and v EQUALS four
                        SET max_score TO the greater value BETWEEN max_score AND the sum of score1 PLUS score2 PLUS element at position u OF scores PLUS element at position v OF scores
                    END IF
                END FOR
            END FOR
        END FOR

        RETURN max_score
    END FUNCTION
END CLASS
