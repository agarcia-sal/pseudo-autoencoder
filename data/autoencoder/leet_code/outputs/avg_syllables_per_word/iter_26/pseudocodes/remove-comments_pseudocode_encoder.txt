CLASS Solution
    FUNCTION removeComments(source)
        SET in_block_comment TO false
        SET result TO empty list
        SET current_line TO empty list

        FOR each line IN source
            SET i TO zero
            WHILE i LESS THAN the LENGTH OF line HOLDS
                IF in_block_comment
                    IF the substring FROM position i TO position i PLUS one OF line EQUALS the two character string consisting of forward slash followed by asterisk
                        SET in_block_comment TO false
                        INCREMENT i BY one
                    END IF
                ELSE
                    IF the substring FROM position i TO position i PLUS one OF line EQUALS the two character string consisting of forward slash followed by forward slash
                        BREAK out of WHILE loop
                    ELSE IF the substring FROM position i TO position i PLUS one OF line EQUALS the two character string consisting of forward slash followed by asterisk
                        SET in_block_comment TO true
                        INCREMENT i BY one
                    ELSE
                        APPEND the element at position i OF line TO current_line
                    END IF
                END IF
                INCREMENT i BY one
            END WHILE

            IF current_line IS NOT empty AND in_block_comment EQUALS false
                APPEND the concatenation of elements in current_line TO result
                SET current_line TO empty list
            END IF
        END FOR

        RETURN result
    END FUNCTION
END CLASS
