CLASS Solution
    FUNCTION originalDigits(string_s)
        FROM collections IMPORT Counter

        SET count TO Counter APPLIED TO string_s

        SET digits TO list of ten zeros

        SET element at position zero OF digits TO value at element 'z' OF count
        SET element at position two OF digits TO value at element 'w' OF count
        SET element at position four OF digits TO value at element 'u' OF count
        SET element at position six OF digits TO value at element 'x' OF count
        SET element at position eight OF digits TO value at element 'g' OF count

        SET element at position three OF digits TO value at element 'h' OF count MINUS element at position eight OF digits
        SET element at position five OF digits TO value at element 'f' OF count MINUS element at position four OF digits
        SET element at position seven OF digits TO value at element 's' OF count MINUS element at position six OF digits
        SET element at position nine OF digits TO value at element 'i' OF count MINUS element at position five OF digits MINUS element at position six OF digits MINUS element at position eight OF digits
        SET element at position one OF digits TO value at element 'o' OF count MINUS element at position zero OF digits MINUS element at position two OF digits MINUS element at position four OF digits

        SET result TO empty list
        FOR index FROM zero TO nine INCLUSIVE
            APPEND string representing index REPEATED element at position index OF digits TIMES TO result
        END FOR

        RETURN concatenation OF all elements IN result
    END FUNCTION
END CLASS
