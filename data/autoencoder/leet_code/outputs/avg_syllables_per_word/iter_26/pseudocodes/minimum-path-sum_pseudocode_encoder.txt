CLASS Solution
    FUNCTION minPathSum(grid)
        IF grid IS EQUAL TO empty OR element at position zero OF grid IS EQUAL TO empty THEN
            RETURN zero
        END IF
        SET m TO the LENGTH OF grid
        SET n TO the LENGTH OF element at position zero OF grid

        CREATE a two dimensional list named dp WITH m rows AND n columns FILLED WITH zero

        SET element at position zero OF element at position zero OF dp TO element at position zero OF element at position zero OF grid

        FOR j FROM one TO n MINUS one INCLUSIVE
            SET element at position j OF element at position zero OF dp TO element at position j MINUS one OF element at position zero OF dp PLUS element at position j OF element at position zero OF grid
        END FOR

        FOR i FROM one TO m MINUS one INCLUSIVE
            SET element at position i OF element at position zero OF dp TO element at position i MINUS one OF element at position zero OF dp PLUS element at position i OF element at position zero OF grid
        END FOR

        FOR i FROM one TO m MINUS one INCLUSIVE
            FOR j FROM one TO n MINUS one INCLUSIVE
                SET element at position j OF element at position i OF dp TO the MINIMUM OF element at position j OF element at position i MINUS one OF dp AND element at position j MINUS one OF element at position i OF dp PLUS element at position j OF element at position i OF grid
            END FOR
        END FOR

        RETURN element at position n MINUS one OF element at position m MINUS one OF dp
    END FUNCTION
END CLASS
