CLASS Solution
    FUNCTION countOrders(parameter n)
        SET MOD TO ten raised TO the power of nine PLUS seven
        SET count TO one
        FOR index i FROM two TO n INCLUSIVE
            SET count TO count MULTIPLIED BY i MULTIPLIED BY open parenthesis two MULTIPLIED BY i MINUS one close parenthesis MODULO MOD
        END FOR
        RETURN count
    END FUNCTION
END CLASS
