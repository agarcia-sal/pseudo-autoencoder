CLASS Solution
    FUNCTION maxPerformance(n, speed, efficiency, k)
        SET engineers TO the LIST OF pairs FOR each corresponding element IN efficiency AND speed COMBINED AS tuples
        SORT engineers IN DESCENDING ORDER ACCORDING TO the FIRST element OF each tuple
        
        SET speed_heap TO an EMPTY list AS a MINIMUM heap structure
        SET speed_sum TO zero
        SET max_performance TO zero
        
        FOR each pair OF efficiency AS eff AND speed AS spd IN engineers
            ADD spd TO speed_heap USING heap PUSH operation
            INCREMENT speed_sum BY spd
            
            IF the LENGTH OF speed_heap IS GREATER THAN k THEN
                REMOVE the SMALLEST element FROM speed_heap USING heap POP operation
                DECREMENT speed_sum BY the REMOVED element
            END IF
            
            SET current_performance TO speed_sum MULTIPLIED BY eff
            IF current_performance IS GREATER THAN max_performance THEN
                SET max_performance TO current_performance
            END IF
        END FOR
        
        RETURN max_performance MODULO the NUMBER one FOLLOWED BY nine zeros PLUS the NUMBER seven
    END FUNCTION
END CLASS
