CLASS Solution
    FUNCTION maxSubArray(nums)
        SET maximum_sum TO the element at the first position OF nums
        SET current_sum TO the element at the first position OF nums
        FOR each number IN the elements of nums starting FROM the second position TO the last position INCLUSIVE
            IF number GREATER THAN current_sum PLUS number
                SET current_sum TO number
            ELSE
                SET current_sum TO current_sum PLUS number
            END IF
            IF current_sum GREATER THAN maximum_sum
                SET maximum_sum TO current_sum
            END IF
        END FOR
        RETURN maximum_sum
    END FUNCTION
END CLASS
