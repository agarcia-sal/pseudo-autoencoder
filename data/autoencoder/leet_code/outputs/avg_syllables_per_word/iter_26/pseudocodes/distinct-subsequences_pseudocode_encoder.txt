CLASS Solution
    FUNCTION numDistinct(s, t)
        SET m TO the LENGTH OF s
        SET n TO the LENGTH OF t
        
        SET dp TO a NEW TWO DIMENSIONAL LIST OF ZEROES WITH DIMENSIONS m PLUS one BY n PLUS one
        
        FOR i FROM zero TO m INCLUSIVE
            SET the ELEMENT AT POSITION zero OF THE ELEMENT AT POSITION i OF dp TO one
        END FOR
        
        FOR i FROM one TO m INCLUSIVE
            FOR j FROM one TO n INCLUSIVE
                IF the ELEMENT AT POSITION i MINUS one OF s EQUALS the ELEMENT AT POSITION j MINUS one OF t THEN
                    SET the ELEMENT AT POSITION j OF THE ELEMENT AT POSITION i OF dp TO the ELEMENT AT POSITION j MINUS one OF THE ELEMENT AT POSITION i MINUS one OF dp PLUS the ELEMENT AT POSITION j OF THE ELEMENT AT POSITION i MINUS one OF dp
                ELSE
                    SET the ELEMENT AT POSITION j OF THE ELEMENT AT POSITION i OF dp TO the ELEMENT AT POSITION j OF THE ELEMENT AT POSITION i MINUS one OF dp
                END IF
            END FOR
        END FOR
        
        RETURN the ELEMENT AT POSITION n OF THE ELEMENT AT POSITION m OF dp
    END FUNCTION
END CLASS
