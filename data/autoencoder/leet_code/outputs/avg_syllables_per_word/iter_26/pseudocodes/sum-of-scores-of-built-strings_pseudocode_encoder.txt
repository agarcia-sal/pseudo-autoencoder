CLASS Solution
    FUNCTION sumScores(s)
        SET n TO the LENGTH OF s
        SET lcp TO a LIST of zero VALUES WITH SIZE n
        
        SET l TO zero
        SET r TO zero
        
        FOR i FROM one TO n MINUS one INCLUSIVE
            IF i LESS THAN OR EQUAL TO r THEN
                SET lcp AT POSITION i TO the MINIMUM OF r MINUS i PLUS one AND lcp AT POSITION i MINUS l
            END IF
            
            WHILE i PLUS lcp AT POSITION i LESS THAN n AND element at position lcp AT POSITION i OF s EQUALS element at position i PLUS lcp AT POSITION i OF s HOLDS
                INCREMENT lcp AT POSITION i BY one
            END WHILE
            
            IF i PLUS lcp AT POSITION i MINUS one GREATER THAN r THEN
                SET l TO i
                SET r TO i PLUS lcp AT POSITION i MINUS one
            END IF
        END FOR
        
        SET total_score TO the SUM OF all elements IN lcp PLUS n
        
        RETURN total_score
    END FUNCTION
END CLASS
