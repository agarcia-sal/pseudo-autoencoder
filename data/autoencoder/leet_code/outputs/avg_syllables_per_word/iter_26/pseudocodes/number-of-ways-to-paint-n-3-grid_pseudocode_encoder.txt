CLASS Solution
    FUNCTION numOfWays(parameter n)
        SET MODULO_VALUE TO ten RAISED TO THE POWER OF nine PLUS one

        SET variable a TO six
        SET variable b TO six

        FOR index FROM two TO parameter n INCLUSIVE
            SET temporary_a TO the sum of the product of three and variable a AND the product of two and variable b
            SET temporary_b TO the sum of the product of two and variable a AND the product of two and variable b
            SET variable a TO temporary_a MODULO MODULO_VALUE
            SET variable b TO temporary_b MODULO MODULO_VALUE
        END FOR

        SET result TO the sum of variable a AND variable b
        SET result TO result MODULO MODULO_VALUE

        RETURN result
    END FUNCTION
END CLASS
