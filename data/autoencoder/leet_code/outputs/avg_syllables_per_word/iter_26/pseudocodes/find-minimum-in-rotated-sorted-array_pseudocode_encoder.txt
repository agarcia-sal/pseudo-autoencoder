CLASS Solution
    FUNCTION findMin(list_of_numbers)
        SET left TO zero
        SET right TO the LENGTH OF list_of_numbers MINUS one
        
        IF the element at position left OF list_of_numbers IS LESS THAN the element at position right OF list_of_numbers
            RETURN the element at position left OF list_of_numbers
        END IF
        
        WHILE left IS LESS THAN right
            SET mid TO the INTEGER DIVISION OF the SUM of left AND right BY two
            
            IF the element at position mid OF list_of_numbers IS GREATER THAN the element at position right OF list_of_numbers
                SET left TO mid PLUS one
            ELSE
                SET right TO mid
            END IF
        END WHILE
        
        RETURN the element at position left OF list_of_numbers
    END FUNCTION
END CLASS
