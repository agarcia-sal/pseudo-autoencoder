CLASS Solution
    FUNCTION minSubArrayLen(target, nums)
        SET left TO zero
        SET current_sum TO zero
        SET min_length TO positive infinity

        FOR right FROM zero TO the LENGTH OF nums MINUS one INCLUSIVE
            INCREMENT current_sum BY the element at position right OF nums

            WHILE current_sum GREATER THAN OR EQUAL TO target HOLDS
                SET min_length TO the minimum of min_length AND the expression right MINUS left PLUS one
                DECREMENT current_sum BY the element at position left OF nums
                INCREMENT left BY one
            END WHILE
        END FOR

        IF min_length NOT EQUALS positive infinity THEN
            RETURN min_length
        ELSE
            RETURN zero
        END IF
    END FUNCTION
END CLASS
