CLASS Solution
    FUNCTION rotate(nums, k)
        SET n TO the LENGTH OF nums
        SET k TO k MODULO n
        
        FUNCTION reverse(start, end)
            WHILE start LESS THAN end HOLDS
                SWAP the element at position start OF nums WITH the element at position end OF nums
                INCREMENT start BY one
                DECREMENT end BY one
            END WHILE
        END FUNCTION
        
        CALL reverse WITH the arguments zero AND n MINUS one
        CALL reverse WITH the arguments zero AND k MINUS one
        CALL reverse WITH the arguments k AND n MINUS one
    END FUNCTION
END CLASS
