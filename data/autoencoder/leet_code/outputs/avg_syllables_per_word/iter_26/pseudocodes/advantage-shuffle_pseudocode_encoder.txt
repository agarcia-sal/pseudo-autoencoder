CLASS Solution
    FUNCTION advantageCount(nums1 AS List OF Integer, nums2 AS List OF Integer) AS List OF Integer
        SORT list nums1 IN NON-DECREASING ORDER

        SET list_of_pairs TO a list CREATED BY PAIRING each value WITH its original position IN nums2
        SORT list_of_pairs IN NON-INCREASING ORDER BASED ON the value PART OF each pair

        SET result_list TO a new list WITH the LENGTH OF nums1 FILLED WITH zeroes

        SET left_pointer TO zero
        SET right_pointer TO the LENGTH OF nums1 MINUS one

        FOR each pair IN list_of_pairs
            SET current_value TO the value PART OF pair
            SET current_index TO the original position PART OF pair

            IF element at position right_pointer OF nums1 IS GREATER THAN current_value THEN
                SET element at position current_index OF result_list TO element at position right_pointer OF nums1
                DECREMENT right_pointer BY one
            ELSE
                SET element at position current_index OF result_list TO element at position left_pointer OF nums1
                INCREMENT left_pointer BY one
            END IF
        END FOR

        RETURN result_list
    END FUNCTION
END CLASS
