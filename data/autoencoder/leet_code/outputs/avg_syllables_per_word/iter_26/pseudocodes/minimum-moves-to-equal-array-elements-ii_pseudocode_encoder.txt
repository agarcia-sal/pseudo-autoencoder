CLASS Solution
    FUNCTION minMoves2(nums AS List OF integers) RETURNS integer
        # Sort the array to find the median
        INVOKE sort METHOD ON nums
        # Find the median
        SET median TO element at position the LENGTH OF nums DIVIDED BY two OF nums
        # Calculate the total number of moves to make all elements equal to the median
        SET moves TO the SUM OF absolute value OF the DIFFERENCE BETWEEN each number IN nums AND median
        RETURN moves
    END FUNCTION
END CLASS
