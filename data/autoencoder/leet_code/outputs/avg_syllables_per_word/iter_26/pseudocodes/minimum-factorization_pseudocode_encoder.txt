CLASS Solution
    FUNCTION smallestFactorization(num)
        IF num EQUALS one THEN
            RETURN one
        END IF

        SET factors TO an empty list

        FOR digit FROM nine DOWN TO two INCLUSIVE
            WHILE num MODULO digit EQUALS zero HOLDS
                APPEND digit TO factors
                SET num TO num DIVIDED BY digit
            END WHILE
        END FOR

        IF num NOT EQUALS one THEN
            RETURN zero
        END IF

        SORT factors IN ascending order

        SET result TO the integer conversion of the concatenation of the string forms of each element IN factors

        IF result GREATER THAN two raised to the power of thirty one MINUS one THEN
            RETURN zero
        END IF

        RETURN result
    END FUNCTION
END CLASS
