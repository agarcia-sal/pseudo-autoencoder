CLASS Solution
    FUNCTION subarraySum(list_of_numbers, target_value)
        SET cumulative_sum_frequency TO a DICTIONARY with mapping FROM zero TO one
        SET current_cumulative_sum TO zero
        SET count_of_subarrays TO zero
        
        FOR each number IN list_of_numbers
            INCREMENT current_cumulative_sum BY number
            IF current_cumulative_sum MINUS target_value EXISTS AS key IN cumulative_sum_frequency
                INCREMENT count_of_subarrays BY the VALUE ASSOCIATED WITH the KEY current_cumulative_sum MINUS target_value IN cumulative_sum_frequency
            END IF

            IF current_cumulative_sum EXISTS AS key IN cumulative_sum_frequency
                INCREMENT the VALUE ASSOCIATED WITH the KEY current_cumulative_sum IN cumulative_sum_frequency BY one
            ELSE
                SET the VALUE ASSOCIATED WITH the KEY current_cumulative_sum IN cumulative_sum_frequency TO one
            END IF
        END FOR
        
        RETURN count_of_subarrays
    END FUNCTION
END CLASS
