CLASS Solution
    FUNCTION stoneGameVIII(stones)
        SET prefix_sum TO a list of zeros of LENGTH OF stones
        SET element at position zero OF prefix_sum TO element at position zero OF stones
        FOR index i FROM one TO the LENGTH OF stones MINUS one EXCLUSIVE
            SET element at position i OF prefix_sum TO element at position i MINUS one OF prefix_sum PLUS element at position i OF stones
        END FOR
        SET dp TO element at position LENGTH OF prefix_sum MINUS one OF prefix_sum
        FOR index i FROM the LENGTH OF stones MINUS two DOWN TO one EXCLUSIVE DECREMENTING BY one
            SET dp TO THE GREATER OF dp AND element at position i OF prefix_sum MINUS dp
        END FOR
        RETURN dp
    END FUNCTION
END CLASS
