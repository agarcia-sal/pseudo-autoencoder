CLASS Solution
    FUNCTION findMaxConsecutiveOnes(list_of_numbers)
        SET maximum_count TO zero
        SET current_count TO zero
        SET zero_count TO zero
        SET left TO zero
        
        FOR right FROM zero TO the LENGTH OF list_of_numbers MINUS one INCLUSIVE
            IF element at position right OF list_of_numbers EQUALS one
                INCREMENT current_count BY one
            ELSE IF zero_count LESS THAN one
                INCREMENT zero_count BY one
                INCREMENT current_count BY one
            ELSE
                SET maximum_count TO THE GREATER OF maximum_count AND current_count
                WHILE zero_count EQUALS one HOLDS
                    IF element at position left OF list_of_numbers EQUALS zero
                        DECREMENT zero_count BY one
                    END IF
                    DECREMENT current_count BY one
                    INCREMENT left BY one
                END WHILE
                INCREMENT current_count BY one
            END IF
        END FOR
        
        SET maximum_count TO THE GREATER OF maximum_count AND current_count
        RETURN maximum_count
    END FUNCTION
END CLASS
