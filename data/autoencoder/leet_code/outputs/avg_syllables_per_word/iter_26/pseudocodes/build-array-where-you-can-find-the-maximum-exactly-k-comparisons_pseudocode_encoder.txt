CLASS Solution
    FUNCTION numOfArrays(n, m, k)
        SET MODULO TO ten TO THE POWER OF nine PLUS seven
        
        CREATE three dimensional list dp WITH zero VALUES
            WITH first dimension FROM zero TO n INCLUSIVE
            WITH second dimension FROM zero TO m INCLUSIVE
            WITH third dimension FROM zero TO k INCLUSIVE
        
        FOR max_element FROM one TO m INCLUSIVE
            SET element AT position one OF list AT position max_element OF list AT position one OF dp TO one
        END FOR
        
        FOR array_length FROM two TO n INCLUSIVE
            FOR max_element FROM one TO m INCLUSIVE
                FOR search_cost FROM one TO k INCLUSIVE
                    FOR smaller_max FROM one TO max_element MINUS one INCLUSIVE
                        SET element AT position array_length OF list AT position max_element OF list AT position search_cost OF dp TO element AT position array_length OF list AT position max_element OF list AT position search_cost OF dp PLUS element AT position array_length MINUS one OF list AT position smaller_max OF list AT position search_cost MINUS one OF dp
                        SET element AT position array_length OF list AT position max_element OF list AT position search_cost OF dp TO element AT position array_length OF list AT position max_element OF list AT position search_cost OF dp MODULO MODULO
                    END FOR
                    SET element AT position array_length OF list AT position max_element OF list AT position search_cost OF dp TO element AT position array_length OF list AT position max_element OF list AT position search_cost OF dp PLUS element AT position array_length MINUS one OF list AT position max_element OF list AT position search_cost OF dp MULTIPLIED BY max_element
                    SET element AT position array_length OF list AT position max_element OF list AT position search_cost OF dp TO element AT position array_length OF list AT position max_element OF list AT position search_cost OF dp MODULO MODULO
                END FOR
            END FOR
        END FOR
        
        SET total_result TO zero
        FOR max_element FROM one TO m INCLUSIVE
            SET total_result TO total_result PLUS element AT position n OF list AT position max_element OF list AT position k OF dp
            SET total_result TO total_result MODULO MODULO
        END FOR
        
        RETURN total_result
    END FUNCTION
END CLASS
