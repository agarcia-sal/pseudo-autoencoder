CLASS Solution
    FUNCTION maxSlidingWindow(nums, k)
        IF nums IS EMPTY OR k EQUALS zero
            RETURN empty list
        END IF

        ASSIGN reference TO a new double ended queue named deq
        ASSIGN reference TO a new empty list named result

        FOR index i FROM zero TO the LENGTH OF nums MINUS one INCLUSIVE
            IF deq IS NOT EMPTY AND element at position zero OF deq IS LESS THAN i MINUS k PLUS one
                REMOVE the element at the front OF deq
            END IF

            WHILE deq IS NOT EMPTY AND element at position element at position last OF deq OF nums IS LESS THAN element at position i OF nums HOLDS
                REMOVE the element at the back OF deq
            END WHILE

            APPEND index i TO deq

            IF i IS GREATER THAN OR EQUAL TO k MINUS one
                APPEND element at position element at position zero OF deq OF nums TO result
            END IF
        END FOR

        RETURN result
    END FUNCTION
END CLASS
