CLASS Solution
    FUNCTION exist(board AS List OF List OF string, word AS string) RETURNS boolean
        IF board IS empty
            RETURN FALSE
        END IF

        SET m TO the LENGTH OF board
        SET n TO the LENGTH OF element at position zero OF board
        SET visited TO a list OF lists OF boolean VALUES all set to FALSE with m rows and n columns

        FUNCTION dfs(x AS integer, y AS integer, index AS integer) RETURNS boolean
            IF index EQUALS the LENGTH OF word
                RETURN TRUE
            END IF

            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR the VALUE at position x OF visited at position y IS TRUE OR the VALUE at position x OF board at position y NOT EQUALS the element at position index OF word
                RETURN FALSE
            END IF

            SET the VALUE at position x OF visited at position y TO TRUE

            FOR each pair (dx, dy) IN the list OF pairs: negative one and zero, positive one and zero, zero and negative one, zero and positive one
                IF dfs(x PLUS dx, y PLUS dy, index PLUS one) RETURNS TRUE
                    RETURN TRUE
                END IF
            END FOR

            SET the VALUE at position x OF visited at position y TO FALSE
            RETURN FALSE
        END FUNCTION

        FOR i FROM zero TO m MINUS one INCLUSIVE
            FOR j FROM zero TO n MINUS one INCLUSIVE
                IF dfs(i, j, zero) RETURNS TRUE
                    RETURN TRUE
                END IF
            END FOR
        END FOR

        RETURN FALSE
    END FUNCTION
END CLASS
