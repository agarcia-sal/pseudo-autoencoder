CLASS Solution
    FUNCTION triangleNumber(list_of_numbers)
        CALL sort ON list_of_numbers
        SET count TO zero
        SET n TO the LENGTH OF list_of_numbers
        
        FOR index_k FROM n MINUS one DOWN TO two INCLUSIVE
            SET index_i TO zero
            SET index_j TO index_k MINUS one
            WHILE index_i LESS THAN index_j HOLDS
                IF element at position index_i OF list_of_numbers PLUS element at position index_j OF list_of_numbers GREATER THAN element at position index_k OF list_of_numbers
                    INCREMENT count BY index_j MINUS index_i
                    DECREMENT index_j BY one
                ELSE
                    INCREMENT index_i BY one
                END IF
            END WHILE
        END FOR
        
        RETURN count
    END FUNCTION
END CLASS
