CLASS Solution
    FUNCTION maximumGood(statements)
        SET n TO the LENGTH OF statements
        SET max_good TO zero
        
        FOR each integer i FROM zero TO TWO RAISED TO THE POWER OF n MINUS one INCLUSIVE
            SET good TO a list of BOOLEAN FALSE values repeated n times
            SET bad TO a list of BOOLEAN FALSE values repeated n times
            
            FOR each integer j FROM zero TO n MINUS one INCLUSIVE
                IF the BITWISE AND of i AND TWO RAISED TO THE POWER OF j IS GREATER THAN zero THEN
                    SET the element at position j OF good TO TRUE
                ELSE
                    SET the element at position j OF bad TO TRUE
                END IF
            END FOR
            
            SET valid TO TRUE
            FOR each integer j FROM zero TO n MINUS one INCLUSIVE
                IF the element at position j OF good IS TRUE THEN
                    FOR each integer k FROM zero TO n MINUS one INCLUSIVE
                        IF the element at position k OF statements at position j EQUALS zero AND the element at position k OF good IS TRUE
                            OR the element at position k OF statements at position j EQUALS one AND the element at position k OF bad IS TRUE THEN
                            SET valid TO FALSE
                            BREAK the inner loop
                        END IF
                    END FOR
                    IF valid IS FALSE THEN
                        BREAK the outer loop
                    END IF
                END IF
            END FOR
            
            IF valid IS TRUE THEN
                SET max_good TO the GREATER OF max_good AND the SUM OF elements of good treated as integer values
            END IF
        END FOR
        
        RETURN max_good
    END FUNCTION
END CLASS
