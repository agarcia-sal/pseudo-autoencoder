CLASS Solution
    FUNCTION minAbbreviation(target, dictionary)
        FUNCTION is_valid_abbr(string_s, abbreviation_abbr)
            SET index_i TO zero
            SET index_j TO zero
            WHILE index_i LESS THAN the LENGTH OF string_s AND index_j LESS THAN the LENGTH OF abbreviation_abbr HOLDS
                IF the element at position index_j OF abbreviation_abbr IS A DIGIT
                    IF the element at position index_j OF abbreviation_abbr EQUALS zero as character
                        RETURN false
                    END IF
                    SET shift TO zero
                    WHILE index_j LESS THAN the LENGTH OF abbreviation_abbr AND the element at position index_j OF abbreviation_abbr IS A DIGIT HOLDS
                        SET shift TO shift MULTIPLIED BY ten PLUS the numeric VALUE OF the element at position index_j OF abbreviation_abbr
                        INCREMENT index_j BY one
                    END WHILE
                    INCREMENT index_i BY shift
                ELSE
                    IF the element at position index_i OF string_s NOT EQUALS the element at position index_j OF abbreviation_abbr
                        RETURN false
                    END IF
                    INCREMENT index_i BY one
                    INCREMENT index_j BY one
                END IF
            END WHILE
            RETURN whether index_i EQUALS the LENGTH OF string_s AND index_j EQUALS the LENGTH OF abbreviation_abbr
        END FUNCTION

        FUNCTION generate_abbrs(word, length_value, abbreviation_abbr, index_current, skip_count_value)
            IF index_current EQUALS length_value
                IF skip_count_value GREATER THAN zero
                    CONCATENATE the SPELLING OF skip_count_value TO abbreviation_abbr
                END IF
                APPEND abbreviation_abbr TO abbrs
                RETURN
            END IF
            CALL generate_abbrs WITH word, length_value, abbreviation_abbr, index_current PLUS one, skip_count_value PLUS one
            IF skip_count_value GREATER THAN zero
                CONCATENATE the SPELLING OF skip_count_value TO abbreviation_abbr
            END IF
            CONCATENATE the element at position index_current OF word TO abbreviation_abbr
            CALL generate_abbrs WITH word, length_value, abbreviation_abbr, index_current PLUS one, zero
        END FUNCTION

        FUNCTION find_unique_abbr()
            FOR length_value FROM one TO the LENGTH OF target INCLUSIVE
                CLEAR abbrs TO an empty list
                CALL generate_abbrs WITH target, the LENGTH OF target, an empty string, zero, zero
                SET minimum_abbr TO None
                SET minimum_length TO a very large number representation
                FOR each abbreviation_abbr IN abbrs
                    IF EVERY word IN filtered_dict DOES NOT satisfy is_valid_abbr WITH word, abbreviation_abbr
                        IF the LENGTH OF abbreviation_abbr LESS THAN minimum_length
                            SET minimum_length TO the LENGTH OF abbreviation_abbr
                            SET minimum_abbr TO abbreviation_abbr
                        END IF
                    END IF
                END FOR
                IF minimum_abbr IS NOT None
                    RETURN minimum_abbr
                END IF
            END FOR
        END FUNCTION

        SET filtered_dict TO an empty list
        FOR each word IN dictionary
            IF the LENGTH OF word EQUALS the LENGTH OF target
                APPEND word TO filtered_dict
            END IF
        END FOR

        SET abbrs TO an empty list
        RETURN the RESULT OF find_unique_abbr()
    END FUNCTION
END CLASS
