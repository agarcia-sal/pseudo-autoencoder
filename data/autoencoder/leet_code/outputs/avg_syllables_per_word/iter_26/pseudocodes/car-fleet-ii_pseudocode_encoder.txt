CLASS Solution
    FUNCTION getCollisionTimes(cars)
        SET number_of_cars TO the LENGTH OF cars
        SET collision_times TO a LIST OF negative one POINT zero REPEATED number_of_cars TIMES
        SET stack TO an EMPTY LIST
        
        FOR index FROM number_of_cars MINUS one DOWN TO zero INCLUSIVE
            ASSIGN position TO the element at position zero OF element at position index OF cars
            ASSIGN speed TO the element at position one OF element at position index OF cars
            
            WHILE stack HOLDS
                ASSIGN next_index TO the last element of stack
                ASSIGN next_position TO the element at position zero OF element at position next_index OF cars
                ASSIGN next_speed TO the element at position one OF element at position next_index OF cars
                
                IF speed LESS THAN OR EQUAL TO next_speed
                    REMOVE the last element FROM stack
                    CONTINUE the WHILE loop
                END IF
                
                SET collision_time TO next_position MINUS position DIVIDED BY speed MINUS next_speed
                
                IF the element at position next_index OF collision_times GREATER THAN zero AND collision_time GREATER THAN OR EQUAL TO the element at position next_index OF collision_times
                    REMOVE the last element FROM stack
                    CONTINUE the WHILE loop
                END IF
                
                SET the element at position index OF collision_times TO collision_time
                BREAK the WHILE loop
            END WHILE
            
            APPEND index TO stack
        END FOR
        
        RETURN collision_times
    END FUNCTION
END CLASS
