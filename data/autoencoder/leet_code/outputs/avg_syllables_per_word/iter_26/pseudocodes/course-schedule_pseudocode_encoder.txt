CLASS Solution
    FUNCTION canFinish(numCourses, prerequisites)
        FROM collections IMPORT defaultdict AND deque

        SET graph TO a DEFAULT DICTIONARY with default value as an empty list
        SET indegree TO a LIST of ZEROES with LENGTH EQUAL TO numCourses

        FOR each pair of course AND prereq IN prerequisites
            APPEND course TO the LIST at key prereq IN graph
            INCREMENT value at position course IN indegree BY one
        END FOR

        SET queue TO a DOUBLE ENDED QUEUE containing each course FOR course FROM zero TO numCourses MINUS one WHERE the value at position course IN indegree EQUALS zero

        WHILE queue HOLDS ITEMS
            SET current_course TO REMOVE the FIRST ELEMENT FROM queue
            DECREMENT numCourses BY one

            FOR each next_course IN the LIST at key current_course IN graph
                DECREMENT value at position next_course IN indegree BY one
                IF value at position next_course IN indegree EQUALS zero
                    APPEND next_course TO queue
                END IF
            END FOR
        END WHILE

        RETURN TRUE IF numCourses EQUALS zero OTHERWISE FALSE
    END FUNCTION
END CLASS
