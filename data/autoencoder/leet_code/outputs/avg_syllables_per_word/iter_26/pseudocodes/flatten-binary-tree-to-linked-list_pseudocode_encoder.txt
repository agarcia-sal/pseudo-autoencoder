CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self dot val TO val
        SET self dot left TO left
        SET self dot right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values AS list)
    IF THE values IS NOT PRESENT HOLDS
        RETURN none
    END IF
    SET root TO A NEW TreeNode CREATED WITH THE element at position zero OF values
    SET i TO the INTEGER ONE
    SET queue TO A NEW double ended queue
    APPEND root TO queue
    WHILE THE queue IS NOT EMPTY HOLDS
        SET node TO THE REMOVED LEFTMOST ELEMENT FROM queue
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none HOLDS
            SET node dot left TO A NEW TreeNode CREATED WITH element at position i OF values
            APPEND node dot left TO queue
        END IF
        INCREMENT i BY the INTEGER ONE
        IF i IS LESS THAN THE LENGTH OF values AND element at position i OF values IS NOT none HOLDS
            SET node dot right TO A NEW TreeNode CREATED WITH element at position i OF values
            APPEND node dot right TO queue
        END IF
        INCREMENT i BY the INTEGER ONE
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS NOT PRESENT AND q IS NOT PRESENT HOLDS
        RETURN true
    ELSE IF p IS NOT PRESENT OR q IS NOT PRESENT HOLDS
        RETURN false
    ELSE IF p dot val IS NOT EQUALS TO q dot val HOLDS
        RETURN false
    ELSE
        RETURN THE RESULT OF CALLING is_same_tree WITH p dot left AND q dot left AND THE RESULT OF CALLING is_same_tree WITH p dot right AND q dot right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION flatten(self, root AS Optional TreeNode)
        IF root IS NOT PRESENT HOLDS
            RETURN
        END IF
        SET stack TO A NEW list CONTAINING root
        WHILE stack IS NOT EMPTY HOLDS
            SET node TO THE LAST ELEMENT REMOVED FROM stack
            IF node dot right IS PRESENT HOLDS
                APPEND node dot right TO stack
            END IF
            IF node dot left IS PRESENT HOLDS
                APPEND node dot left TO stack
            END IF
            SET node dot left TO none
            SET node dot right TO THE LAST ELEMENT OF stack IF stack IS NOT EMPTY ELSE none
        END WHILE
    END FUNCTION
END CLASS
