CLASS Solution
    FUNCTION search(list_of_numbers, target_value)
        IF NOT the list_of_numbers IS NOT EMPTY
            RETURN false
        END IF
        
        SET left TO zero
        SET right TO the LENGTH OF list_of_numbers MINUS one
        
        WHILE left IS LESS THAN OR EQUAL TO right
            SET mid TO left PLUS the DIFFERENCE BETWEEN right AND left DIVIDED BY two
            
            IF the ELEMENT AT position mid OF list_of_numbers EQUALS target_value
                RETURN true
            END IF
            
            IF the ELEMENT AT position left OF list_of_numbers EQUALS the ELEMENT AT position mid OF list_of_numbers AND the ELEMENT AT position mid OF list_of_numbers EQUALS the ELEMENT AT position right OF list_of_numbers
                INCREMENT left BY one
                DECREMENT right BY one
            ELSE IF the ELEMENT AT position left OF list_of_numbers IS LESS THAN OR EQUAL TO the ELEMENT AT position mid OF list_of_numbers
                IF the ELEMENT AT position left OF list_of_numbers IS LESS THAN OR EQUAL TO target_value AND target_value IS LESS THAN the ELEMENT AT position mid OF list_of_numbers
                    SET right TO mid MINUS one
                ELSE
                    SET left TO mid PLUS one
                END IF
            ELSE
                IF the ELEMENT AT position mid OF list_of_numbers IS LESS THAN target_value AND target_value IS LESS THAN OR EQUAL TO the ELEMENT AT position right OF list_of_numbers
                    SET left TO mid PLUS one
                ELSE
                    SET right TO mid MINUS one
                END IF
            END IF
        END WHILE
        
        RETURN false
    END FUNCTION
END CLASS
