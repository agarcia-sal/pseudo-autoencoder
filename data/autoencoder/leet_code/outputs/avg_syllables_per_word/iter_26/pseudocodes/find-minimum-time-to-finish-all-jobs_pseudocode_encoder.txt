CLASS Solution
    FUNCTION minimumTimeRequired(jobs, k)
        SET n TO the LENGTH OF jobs
        
        # Precompute the sum of all possible subsets of jobs
        SET subset_sums TO a list of zeroes with LENGTH equal to two RAISED TO THE POWER OF n
        
        FOR i FROM zero TO n MINUS one INCLUSIVE
            FOR mask FROM zero TO two RAISED TO THE POWER OF i MINUS one INCLUSIVE
                SET element_at_position mask OR the element at position two RAISED TO THE POWER OF i OF subset_sums TO the ELEMENT at position mask OF subset_sums PLUS the ELEMENT at position i OF jobs
            END FOR
        END FOR
        
        FUNCTION dp(k_parameter, mask_parameter)
            IF k_parameter EQUALS one THEN
                RETURN the ELEMENT at position mask_parameter OF subset_sums
            END IF
            SET res TO positive infinity
            
            # Iterate over all subsets of the mask_parameter
            SET submask TO mask_parameter
            WHILE submask NOT EQUALS zero
                SET res TO the MINIMUM OF res AND the MAXIMUM OF dp(k_parameter MINUS one, mask_parameter XOR submask) AND the ELEMENT at position submask OF subset_sums
                SET submask TO the BITWISE AND OF submask MINUS one AND mask_parameter
            END WHILE
            
            RETURN res
        END FUNCTION
        
        RETURN dp(k, two RAISED TO THE POWER OF n MINUS one)
    END FUNCTION
END CLASS
