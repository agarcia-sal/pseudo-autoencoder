CLASS Solution
    FUNCTION wiggleSort(list_of_numbers)
        SET flag_less TO true
        
        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one MINUS one INCLUSIVE
            IF flag_less HOLDS
                IF the element at position index OF list_of_numbers GREATER THAN the element at position index plus one OF list_of_numbers
                    SWAP the element at position index OF list_of_numbers WITH the element at position index plus one OF list_of_numbers
                END IF
            ELSE
                IF the element at position index OF list_of_numbers LESS THAN the element at position index plus one OF list_of_numbers
                    SWAP the element at position index OF list_of_numbers WITH the element at position index plus one OF list_of_numbers
                END IF
            END IF
            SET flag_less TO NOT flag_less
        END FOR
    END FUNCTION
END CLASS
