CLASS Solution
    FUNCTION kthSmallestPath(destination, k)
        UNPACK destination INTO row AND col
        SET path TO an empty list
        
        SET total_moves TO col PLUS row
        
        FOR index FROM zero TO total_moves MINUS one INCLUSIVE
            IF col GREATER THAN zero
                SET paths_with_H TO the NUMBER OF COMBINATIONS OF row PLUS col MINUS one CHOOSE col MINUS one
                IF k LESS THAN OR EQUAL TO paths_with_H
                    APPEND character H TO path
                    DECREMENT col BY one
                ELSE
                    APPEND character V TO path
                    DECREMENT row BY one
                    DECREMENT k BY paths_with_H
                END IF
            ELSE
                APPEND character V TO path
                DECREMENT row BY one
            END IF
        END FOR
        
        RETURN the CONCATENATION OF all elements IN path AS a SINGLE STRING
    END FUNCTION
END CLASS
