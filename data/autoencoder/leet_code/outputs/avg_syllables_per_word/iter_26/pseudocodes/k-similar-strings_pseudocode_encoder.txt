CLASS Solution
    FUNCTION kSimilarity(s1, s2)
        IMPORT deque FROM collections
        
        FUNCTION neighbors(s)
            SET position i TO zero
            WHILE position i IS LESS THAN the LENGTH OF s AND element at position i OF s EQUALS element at position i OF s2 HOLDS
                INCREMENT position i BY one
            END WHILE
            FOR position j FROM position i PLUS one TO the LENGTH OF s MINUS one INCLUSIVE
                IF element at position j OF s EQUALS element at position i OF s2 AND element at position j OF s NOT EQUALS element at position j OF s2
                    YIELD concatenation OF substring FROM start TO position i MINUS one OF s PLUS element at position j OF s PLUS substring FROM position i PLUS one TO position j MINUS one OF s PLUS element at position i OF s PLUS substring FROM position j PLUS one TO end OF s
                END IF
            END FOR
        END FUNCTION
        
        SET queue TO a new deque containing a tuple of s1 AND zero
        SET visited TO a new set containing s1
        
        WHILE queue HOLDS
            REMOVE and ASSIGN the first tuple OF queue TO variables s AND swaps
            IF s EQUALS s2
                RETURN swaps
            END IF
            FOR each neighbor IN neighbors(s)
                IF neighbor NOT IN visited
                    ADD neighbor TO visited
                    APPEND a tuple OF neighbor AND swaps PLUS one TO queue
                END IF
            END FOR
        END WHILE
    END FUNCTION
END CLASS
