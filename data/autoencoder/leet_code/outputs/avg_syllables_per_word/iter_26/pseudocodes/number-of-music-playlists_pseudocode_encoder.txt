CLASS Solution
    FUNCTION numMusicPlaylists(n, goal, k)
        SET MOD TO ten raised to the power of nine PLUS one

        # dp i j : number of ways to create a playlist of length i using exactly j different songs
        SET dp TO a list of lists where each inner list has zero repeated the number of times one MORE THAN n
            AND the outer list has a number of these inner lists one MORE THAN goal
        SET the element at position zero AND zero OF dp TO one

        FOR i FROM one TO goal INCLUSIVE
            FOR j FROM one TO n INCLUSIVE
                # Case one : Add a new song to the playlist
                INCREMENT the element at position i AND j OF dp BY the element at position i MINUS one AND j MINUS one OF dp MULTIPLIED BY n MINUS the quantity j MINUS one
                SET the element at position i AND j OF dp TO the element at position i AND j OF dp MODULO MOD

                # Case two : Repeat a song that was played at least k songs ago
                IF j GREATER THAN k
                    INCREMENT the element at position i AND j OF dp BY the element at position i MINUS one AND j OF dp MULTIPLIED BY j MINUS k
                    SET the element at position i AND j OF dp TO the element at position i AND j OF dp MODULO MOD
                END IF
            END FOR
        END FOR

        RETURN the element at position goal AND n OF dp
    END FUNCTION
END CLASS
