CLASS Solution
    FUNCTION lengthOfLIS(parameter nums AS list of integers) RETURNS integer
        IF NOT the list nums HOLDS ANY ELEMENTS
            RETURN zero
        END IF

        SET dp TO an empty list

        FOR each num IN nums
            SET idx TO the position obtained by FINDING LEFTMOST INSERTION INDEX OF num IN the list dp USING BINARY SEARCH

            IF idx EQUALS the LENGTH OF the list dp
                APPEND num TO the list dp
            ELSE
                SET the element at position idx OF the list dp TO num
            END IF
        END FOR

        RETURN the LENGTH OF the list dp
    END FUNCTION
END CLASS
