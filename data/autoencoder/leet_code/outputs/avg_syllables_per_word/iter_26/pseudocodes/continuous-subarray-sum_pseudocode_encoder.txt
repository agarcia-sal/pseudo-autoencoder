CLASS Solution
    FUNCTION checkSubarraySum(list_of_numbers, integer_k)
        SET remainder_map TO a dictionary WITH key zero MAPPED TO minus one
        SET current_sum TO zero
        
        FOR each index i AND each number num IN enumeration OF list_of_numbers
            INCREMENT current_sum BY number num
            
            IF integer_k NOT EQUALS zero THEN
                SET current_sum TO current_sum MODULO integer_k
            END IF
            
            IF current_sum IS A KEY IN remainder_map THEN
                IF index i MINUS value AT key current_sum OF remainder_map GREATER THAN one THEN
                    RETURN True
                END IF
            ELSE
                SET value AT key current_sum OF remainder_map TO index i
            END IF
        END FOR
        
        RETURN False
    END FUNCTION
END CLASS
