CLASS Solution
    FUNCTION sortColors(parameter nums)
        SET low TO zero
        SET mid TO zero
        SET high TO the LENGTH OF nums MINUS one
        WHILE mid LESS THAN OR EQUAL TO high HOLDS
            IF the element at position mid OF nums EQUALS zero
                SET temporary_variable TO the element at position mid OF nums
                SET the element at position mid OF nums TO the element at position low OF nums
                SET the element at position low OF nums TO temporary_variable
                INCREMENT low BY one
                INCREMENT mid BY one
            ELSE IF the element at position mid OF nums EQUALS one
                INCREMENT mid BY one
            ELSE IF the element at position mid OF nums EQUALS two
                SET temporary_variable TO the element at position mid OF nums
                SET the element at position mid OF nums TO the element at position high OF nums
                SET the element at position high OF nums TO temporary_variable
                DECREMENT high BY one
            END IF
        END WHILE
    END FUNCTION
END CLASS
