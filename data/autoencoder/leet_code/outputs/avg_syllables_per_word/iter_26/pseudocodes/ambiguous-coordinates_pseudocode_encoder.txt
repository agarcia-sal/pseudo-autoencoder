CLASS Solution
    FUNCTION ambiguousCoordinates(s STRING) RETURNS list_of_strings
        FUNCTION make(frag STRING) YIELDS string_parts
            SET n TO the LENGTH OF frag
            FOR d FROM one TO n INCLUSIVE
                SET left_part TO the SUBSTRING FROM position one TO position d OF frag
                SET right_part TO the SUBSTRING FROM position d PLUS one TO the END OF frag
                IF (NOT the FIRST CHARACTER OF left_part EQUALS zero OR left_part EQUALS zero) AND (NOT the LAST CHARACTER OF right_part EQUALS zero)
                    IF d NOT EQUAL TO n
                        YIELD the CONCATENATION OF left_part, a period expressed AS a decimal point, and right_part
                    ELSE
                        YIELD left_part
                    END IF
                END IF
            END FOR
        END FUNCTION

        SET s TO the SUBSTRING FROM position two TO the PENULTIMATE POSITION OF s
        RETURN the LIST formed BY
            FOR i FROM one TO the LENGTH OF s MINUS one INCLUSIVE
                FOR each a IN the PRODUCT of make(the SUBSTRING FROM position one TO position i OF s) AND each b IN make(the SUBSTRING FROM position i PLUS one TO the END OF s)
                    FORM a STRING BY CONCATENATING an opening parenthesis, a, comma, a space, b, and a closing parenthesis
                END FOR
            END FOR
    END FUNCTION
END CLASS
