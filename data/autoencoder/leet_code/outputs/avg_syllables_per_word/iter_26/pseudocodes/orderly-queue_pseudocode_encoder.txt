CLASS Solution
    FUNCTION orderlyQueue(string_s, integer_k)
        IF integer_k GREATER THAN one
            RETURN the concatenation OF the sorted characters IN string_s
        ELSE
            SET integer_n TO the LENGTH OF string_s
            SET string_smallest TO string_s
            FOR integer_i FROM one TO integer_n MINUS one INCLUSIVE
                SET string_rotated TO the substring FROM position integer_i TO the END OF string_s CONCATENATED WITH the substring FROM position zero TO position integer_i OF string_s
                IF string_rotated LESS THAN string_smallest
                    SET string_smallest TO string_rotated
                END IF
            END FOR
            RETURN string_smallest
        END IF
    END FUNCTION
END CLASS
