CLASS Solution
    FUNCTION myAtoi(s AS string)
        SET INT_MAX TO the two RAISED TO THE power OF thirty ONE MINUS one
        SET INT_MIN TO the NEGATIVE of two RAISED TO THE power OF thirty ONE

        SET result TO zero
        SET index TO zero
        SET n TO the LENGTH OF s
        SET sign TO one

        WHILE index LESS THAN n AND the element at position index OF s EQUALS a space character HOLDS
            INCREMENT index BY one
        END WHILE

        IF index LESS THAN n AND (the element at position index OF s EQUALS the plus character OR the element at position index OF s EQUALS the minus character) HOLDS
            IF the element at position index OF s EQUALS the minus character HOLDS
                SET sign TO NEGATIVE one
            ELSE
                SET sign TO one
            END IF
            INCREMENT index BY one
        END IF

        WHILE index LESS THAN n AND the element at position index OF s IS A digit character HOLDS
            SET digit TO the integer value OF the element at position index OF s

            IF result GREATER THAN the quotient of (INT_MAX MINUS digit) DIVIDED BY ten HOLDS
                IF sign EQUALS one HOLDS
                    RETURN INT_MAX
                ELSE
                    RETURN INT_MIN
                END IF
            END IF

            SET result TO result MULTIPLIED BY ten PLUS digit
            INCREMENT index BY one
        END WHILE

        RETURN sign MULTIPLIED BY result
    END FUNCTION
END CLASS
