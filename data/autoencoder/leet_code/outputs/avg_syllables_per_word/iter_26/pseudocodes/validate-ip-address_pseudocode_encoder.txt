CLASS Solution
    FUNCTION validIPAddress(queryIP)
        FUNCTION is_valid_ipv4_block(block)
            IF block CONSISTS OF DIGITS ONLY HOLDS
                SET num TO THE INTEGER REPRESENTATION OF block
                IF num IS LESS THAN OR EQUAL TO two hundred fifty five AND THE STRING REPRESENTATION OF num EQUALS block HOLDS
                    RETURN TRUE
                ELSE
                    RETURN FALSE
                END IF
            ELSE
                RETURN FALSE
            END IF
        END FUNCTION

        FUNCTION is_valid_ipv6_block(block)
            IF THE LENGTH OF block IS GREATER THAN OR EQUAL TO one AND THE LENGTH OF block IS LESS THAN OR EQUAL TO four HOLDS
                TRY
                    ATTEMPT TO CONVERT block TO INTEGER WITH BASE sixteen
                    RETURN TRUE
                CATCH EXCEPTION OF TYPE ValueError
                    RETURN FALSE
                END TRY
            ELSE
                RETURN FALSE
            END IF
        END FUNCTION

        SET parts_ipv4 TO THE RESULT OF SPLITTING queryIP BY DOT
        SET parts_ipv6 TO THE RESULT OF SPLITTING queryIP BY COLON

        IF THE LENGTH OF parts_ipv4 EQUALS four AND ALL ELEMENTS IN parts_ipv4 SATISFY is_valid_ipv4_block HOLDS
            RETURN THE STRING IPvFour
        ELSE IF THE LENGTH OF parts_ipv6 EQUALS eight AND ALL ELEMENTS IN parts_ipv6 SATISFY is_valid_ipv6_block HOLDS
            RETURN THE STRING IPvSix
        ELSE
            RETURN THE STRING Neither
        END IF
    END FUNCTION
END CLASS
