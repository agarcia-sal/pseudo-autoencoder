CLASS Solution
    FUNCTION pourWater(heights, volume, k)
        SET n TO the LENGTH OF heights
        
        FOR each iteration FROM one TO volume INCLUSIVE
            SET i TO k
            WHILE i GREATER THAN zero AND element at position i MINUS one OF heights LESS THAN OR EQUAL TO element at position i OF heights HOLDS
                DECREMENT i BY one
            END WHILE
            WHILE i LESS THAN k AND element at position i OF heights EQUALS element at position i PLUS one OF heights HOLDS
                INCREMENT i BY one
            END WHILE
            IF i LESS THAN k THEN
                INCREMENT element at position i OF heights BY one
                CONTINUE TO next iteration of the FOR loop
            END IF
            
            SET i TO k
            WHILE i LESS THAN n MINUS one AND element at position i PLUS one OF heights LESS THAN OR EQUAL TO element at position i OF heights HOLDS
                INCREMENT i BY one
            END WHILE
            WHILE i GREATER THAN k AND element at position i OF heights EQUALS element at position i MINUS one OF heights HOLDS
                DECREMENT i BY one
            END WHILE
            IF i GREATER THAN k THEN
                INCREMENT element at position i OF heights BY one
                CONTINUE TO next iteration of the FOR loop
            END IF
            
            INCREMENT element at position k OF heights BY one
        END FOR
        
        RETURN heights
    END FUNCTION
END CLASS
