CLASS Solution
    FUNCTION subdomainVisits(cpdomains)
        ASSIGN domain_count TO a new default dictionary of integer counts
        
        FOR each cpdomain IN cpdomains
            SPLIT cpdomain INTO two parts count_string AND domain USING space AS delimiter
            CONVERT count_string TO integer AND SET TO count
            SPLIT domain INTO subdomains USING the period character AS delimiter
            
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one INCLUSIVE
                JOIN subdomains FROM position index TO the end USING the period character AND SET TO subdomain
                INCREMENT the value corresponding to subdomain IN domain_count BY count
            END FOR
        END FOR
        
        SET result_list TO a new empty list
        FOR each domain AND count IN the items OF domain_count
            CREATE a string consisting of the count FOLLOWED BY a space FOLLOWED BY the domain
            APPEND this string TO result_list
        END FOR
        
        RETURN result_list
    END FUNCTION
END CLASS
