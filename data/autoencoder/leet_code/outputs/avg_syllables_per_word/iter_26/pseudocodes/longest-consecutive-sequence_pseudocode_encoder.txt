CLASS Solution
    FUNCTION longestConsecutive(list_of_numbers)
        IF the list_of_numbers IS EMPTY HOLDS
            RETURN zero
        END IF

        SET number_set TO the SET OF elements IN list_of_numbers
        SET longest_streak TO zero

        FOR each number IN number_set
            IF the ELEMENT IDENTIFIED BY number MINUS one IS NOT IN number_set HOLDS
                SET current_number TO number
                SET current_streak TO one

                WHILE the ELEMENT IDENTIFIED BY current_number PLUS one IS IN number_set HOLDS
                    INCREMENT current_number BY one
                    INCREMENT current_streak BY one
                END WHILE

                SET longest_streak TO the GREATER OF longest_streak AND current_streak
            END IF
        END FOR

        RETURN longest_streak
    END FUNCTION
END CLASS
