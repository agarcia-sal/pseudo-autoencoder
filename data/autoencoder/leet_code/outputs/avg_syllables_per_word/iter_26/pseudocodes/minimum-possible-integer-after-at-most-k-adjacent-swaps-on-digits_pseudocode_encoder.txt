CLASS Solution
    FUNCTION minInteger(num, k)
        SET num TO the LIST FORM OF num
        SET n TO the LENGTH OF num

        WHILE k GREATER THAN zero HOLDS
            SET swapped TO False

            FOR i FROM zero TO n MINUS one MINUS one INCLUSIVE
                IF the ELEMENT AT POSITION i OF num GREATER THAN the ELEMENT AT POSITION i PLUS one OF num
                    SET min_digit TO the CHARACTER nine
                    SET min_index TO minus one

                    FOR j FROM i TO the MINIMUM OF i PLUS k PLUS one AND n MINUS one INCLUSIVE
                        IF the ELEMENT AT POSITION j OF num LESS THAN min_digit
                            SET min_digit TO the ELEMENT AT POSITION j OF num
                            SET min_index TO j
                        END IF
                    END FOR

                    SET the SUBLIST OF num FROM POSITION i TO POSITION min_index PLUS one TO a NEW LIST CONSISTING OF the ELEMENT AT POSITION min_index OF num FOLLOWED BY the SUBLIST OF num FROM POSITION i TO POSITION min_index MINUS one
                    DECREMENT k BY the VALUE OF min_index MINUS i
                    SET swapped TO True
                    BREAK
                END IF
            END FOR

            IF swapped EQUALS False
                BREAK
            END IF
        END WHILE

        RETURN the CONCATENATION OF all ELEMENTS OF num AS A STRING
    END FUNCTION
END CLASS
