CLASS Solution
    FUNCTION shortestSuperstring(words)
        SET n TO the LENGTH OF words

        FUNCTION overlap(i, j)
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i OF words AND the LENGTH OF element at position j OF words DOWN TO one INCLUSIVE
                IF the substring FROM the LENGTH OF element at position i OF words MINUS k PLUS one TO the LENGTH OF element at position i OF words OF element at position i OF words EQUALS the substring FROM one TO k OF element at position j OF words
                    RETURN k
                END IF
            END FOR
            RETURN zero
        END FUNCTION

        FUNCTION dp(mask, i)
            IF mask EQUALS the BITWISE LEFT SHIFT OF one BY n MINUS one
                RETURN the tuple OF zero AND the empty string
            END IF

            SET min_len TO positive infinity
            SET best_path TO the empty string
            FOR j FROM zero TO n MINUS one INCLUSIVE
                IF the BITWISE AND OF mask AND the BITWISE LEFT SHIFT OF one BY j HOLDS NOT
                    SET length_and_path TO dp the BITWISE OR OF mask AND the BITWISE LEFT SHIFT OF one BY j, j
                    SET length TO the FIRST ITEM OF length_and_path
                    SET path TO the SECOND ITEM OF length_and_path
                    SET length TO length PLUS the LENGTH OF element at position j OF words MINUS overlap of i AND j
                    IF length LESS THAN min_len
                        SET min_len TO length
                        SET best_path TO the substring FROM overlap of i AND j PLUS one TO the LENGTH OF element at position j OF words OF element at position j OF words CONCATENATED WITH path
                    END IF
                END IF
            END FOR

            RETURN the tuple OF min_len AND best_path
        END FUNCTION

        SET min_len TO positive infinity
        SET shortest_path TO the empty string
        FOR i FROM zero TO n MINUS one INCLUSIVE
            SET length_and_path TO dp the BITWISE LEFT SHIFT OF one BY i, i
            SET length TO the FIRST ITEM OF length_and_path
            SET path TO the SECOND ITEM OF length_and_path
            SET length TO length PLUS the LENGTH OF element at position i OF words
            IF length LESS THAN min_len
                SET min_len TO length
                SET shortest_path TO element at position i OF words CONCATENATED WITH path
            END IF
        END FOR

        RETURN shortest_path
    END FUNCTION
END CLASS
