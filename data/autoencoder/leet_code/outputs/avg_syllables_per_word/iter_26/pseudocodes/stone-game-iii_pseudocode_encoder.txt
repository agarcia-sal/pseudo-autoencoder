CLASS Solution

    FUNCTION stoneGameIII(stoneValue)
        SET n TO the LENGTH OF stoneValue
        SET dp TO a list of zeros with the LENGTH OF n PLUS one
        
        FOR i FROM n MINUS one TO zero INCLUSIVE DECREASING
            SET dp ELEMENT AT position i TO element at position i OF stoneValue MINUS dp ELEMENT AT position i PLUS one
            IF i PLUS two IS LESS THAN OR EQUAL TO n THEN
                SET dp ELEMENT AT position i TO the GREATER OF dp ELEMENT AT position i AND the SUM OF element at position i OF stoneValue AND element at position i PLUS one OF stoneValue MINUS dp ELEMENT AT position i PLUS two
            END IF
            IF i PLUS three IS LESS THAN OR EQUAL TO n THEN
                SET dp ELEMENT AT position i TO the GREATER OF dp ELEMENT AT position i AND the SUM OF element at position i OF stoneValue AND element at position i PLUS one OF stoneValue AND element at position i PLUS two OF stoneValue MINUS dp ELEMENT AT position i PLUS three
            END IF
        END FOR
        
        IF dp ELEMENT AT position zero IS GREATER THAN zero THEN
            RETURN Alice
        ELSE IF dp ELEMENT AT position zero IS LESS THAN zero THEN
            RETURN Bob
        ELSE
            RETURN Tie
        END IF

    END FUNCTION

END CLASS
