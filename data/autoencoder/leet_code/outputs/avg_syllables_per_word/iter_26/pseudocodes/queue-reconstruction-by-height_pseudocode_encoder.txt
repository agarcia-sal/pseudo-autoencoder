CLASS Solution
    FUNCTION reconstructQueue(people AS list of lists of integers)
        CALL sort ON people USING key FUNCTION WITH parameter x
            RETURN NEGATIVE OF element at position zero OF x, AND element at position one OF x
        END FUNCTION

        SET queue TO empty list

        FOR each person IN people
            INSERT person INTO queue AT position element at position one OF person
        END FOR

        RETURN queue
    END FUNCTION
END CLASS
