CLASS Solution
    FUNCTION minOperations(nums)
        SET n TO the LENGTH OF nums
        SET unique_nums TO the SORTED COLLECTION OF THE UNIQUE ELEMENTS FROM nums
        SET min_operations TO n
        SET j TO zero
        FOR i FROM zero TO the LENGTH OF unique_nums MINUS one INCLUSIVE
            WHILE j LESS THAN the LENGTH OF unique_nums AND the ELEMENT AT POSITION j OF unique_nums LESS THAN the ELEMENT AT POSITION i OF unique_nums PLUS n
                INCREMENT j BY one
            END WHILE
            SET candidate_operations TO n MINUS the DIFFERENCE BETWEEN j AND i
            IF candidate_operations LESS THAN min_operations
                SET min_operations TO candidate_operations
            END IF
        END FOR
        RETURN min_operations
    END FUNCTION
END CLASS
