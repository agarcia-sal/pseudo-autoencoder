CLASS Solution
    FUNCTION uniqueLetterString(s)
        SET MOD TO ten POWERED BY nine PLUS one
        SET n TO the LENGTH OF s
        
        SET last TO a LIST OF twenty six TIMES minus one
        SET last2 TO a LIST OF twenty six TIMES minus one
        
        SET result TO zero
        
        FOR each index i AND character char IN ENUMERATION OF s
            SET idx TO the ORDINAL VALUE OF char MINUS the ORDINAL VALUE OF character A
            
            SET result TO result PLUS the DIFFERENCE OF i MINUS element at position idx OF last MULTIPLIED BY the DIFFERENCE OF element at position idx OF last MINUS element at position idx OF last2
            SET result TO result MODULO MOD
            
            SET element at position idx OF last2 TO element at position idx OF last
            SET element at position idx OF last TO i
        END FOR
        
        FOR each index i FROM zero TO twenty five INCLUSIVE
            SET result TO result PLUS the DIFFERENCE OF n MINUS element at position i OF last MULTIPLIED BY the DIFFERENCE OF element at position i OF last MINUS element at position i OF last2
            SET result TO result MODULO MOD
        END FOR
        
        RETURN result
    END FUNCTION
END CLASS
