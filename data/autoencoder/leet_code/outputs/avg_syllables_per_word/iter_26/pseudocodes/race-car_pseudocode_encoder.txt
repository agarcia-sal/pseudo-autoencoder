CLASS Solution
    FUNCTION racecar(parameter target)
        SET queue TO a new empty double ended queue containing a single tuple of zero position, one speed, zero steps
        SET visited TO a new empty set containing a single tuple of zero position and one speed
        
        WHILE queue HOLDS
            DESTRUCTURE the first element REMOVED FROM queue INTO position, speed, steps
            
            IF position EQUALS target THEN
                RETURN steps
            END IF
            
            SET next_position TO position PLUS speed
            SET next_speed TO speed MULTIPLIED BY two
            IF the tuple of next_position and next_speed NOT IN visited AND next_position GREATER THAN OR EQUAL TO zero AND next_position LESS THAN two MULTIPLIED BY target THEN
                ADD the tuple of next_position and next_speed TO visited
                APPEND the tuple of next_position, next_speed, steps PLUS one TO queue
            END IF
            
            IF speed GREATER THAN zero THEN
                SET next_speed TO negative one
            ELSE
                SET next_speed TO one
            END IF
            IF the tuple of position and next_speed NOT IN visited AND position GREATER THAN OR EQUAL TO zero AND position LESS THAN two MULTIPLIED BY target THEN
                ADD the tuple of position and next_speed TO visited
                APPEND the tuple of position, next_speed, steps PLUS one TO queue
            END IF
        END WHILE
    END FUNCTION
END CLASS
