CLASS Solution
    FUNCTION countSpecialSubsequences(list_of_numbers)
        SET MODULO TO ten raised to the power of nine PLUS one
        
        SET dynamic_programming_list TO a list containing three zeros
        
        FOR each number IN list_of_numbers
            IF number EQUALS zero
                SET the first element of dynamic_programming_list TO the value of two MULTIPLIED BY the first element of dynamic_programming_list PLUS one THEN MODULO the result BY MODULO
            ELSE IF number EQUALS one
                SET the second element of dynamic_programming_list TO the value of two MULTIPLIED BY the second element of dynamic_programming_list PLUS the first element of dynamic_programming_list THEN MODULO the result BY MODULO
            ELSE IF number EQUALS two
                SET the third element of dynamic_programming_list TO the value of two MULTIPLIED BY the third element of dynamic_programming_list PLUS the second element of dynamic_programming_list THEN MODULO the result BY MODULO
            END IF
        END FOR
        
        RETURN the third element of dynamic_programming_list
    END FUNCTION
END CLASS
