CLASS Solution
    FUNCTION minimumMoves(grid)
        SET grid_size TO the LENGTH OF grid

        SET queue TO a new double ended queue containing a tuple composed of zero for row one, zero for column one, zero for row two, one for column two, and zero for steps
        
        SET visited TO a new set containing a tuple composed of zero for row one, zero for column one, zero for row two, and one for column two

        WHILE the queue HOLDS
            ASSIGN zero for row one, zero for column one, zero for row two, zero for column two, and zero for steps TO the RESULT OF removing the first element FROM queue

            IF row one EQUALS grid_size MINUS one AND column one EQUALS grid_size MINUS two AND row two EQUALS grid_size MINUS one AND column two EQUALS grid_size MINUS one
                RETURN steps
            END IF

            IF column two PLUS one LESS THAN grid_size AND the element at position row one AND element at position column one PLUS one OF grid EQUALS zero AND the element at position row two AND element at position column two PLUS one OF grid EQUALS zero AND the tuple composed of row one, column one PLUS one, row two, and column two PLUS one NOT IN visited
                ADD the tuple composed of row one, column one PLUS one, row two, and column two PLUS one TO visited
                APPEND the tuple composed of row one, column one PLUS one, row two, column two PLUS one, and steps PLUS one TO queue
            END IF

            IF row two PLUS one LESS THAN grid_size AND the element at position row one PLUS one AND element at position column one OF grid EQUALS zero AND the element at position row two PLUS one AND element at position column two OF grid EQUALS zero AND the tuple composed of row one PLUS one, column one, row two PLUS one, and column two NOT IN visited
                ADD the tuple composed of row one PLUS one, column one, row two PLUS one, and column two TO visited
                APPEND the tuple composed of row one PLUS one, column one, row two PLUS one, column two, and steps PLUS one TO queue
            END IF

            IF row one EQUALS row two AND row one PLUS one LESS THAN grid_size AND the element at position row one PLUS one AND element at position column one OF grid EQUALS zero AND the element at position row one PLUS one AND element at position column two OF grid EQUALS zero AND the tuple composed of row one, column one, row one PLUS one, and column one NOT IN visited
                ADD the tuple composed of row one, column one, row one PLUS one, and column one TO visited
                APPEND the tuple composed of row one, column one, row one PLUS one, column one, and steps PLUS one TO queue
            END IF

            IF column one EQUALS column two AND column one PLUS one LESS THAN grid_size AND the element at position row one AND element at position column one PLUS one OF grid EQUALS zero AND the element at position row two AND element at position column two PLUS one OF grid EQUALS zero AND the tuple composed of row one, column one, row one, and column one PLUS one NOT IN visited
                ADD the tuple composed of row one, column one, row one, and column one PLUS one TO visited
                APPEND the tuple composed of row one, column one, row one, column one PLUS one, and steps PLUS one TO queue
            END IF
        END WHILE

        RETURN a negative one as failure indicator
    END FUNCTION
END CLASS
