CLASS Solution
    FUNCTION largestMultipleOfThree(digits)
        SET count TO the frequency COUNTER OF digits
        SET total_sum TO the SUM OF digits
        SET remainder TO total_sum MODULO BY three

        IF remainder EQUALS one THEN
            FOR each digit IN the list consisting of one, four, seven
                IF the frequency OF digit IN count GREATER THAN zero THEN
                    DECREMENT the frequency OF digit IN count BY one
                    BREAK OUT OF the FOR loop
                END IF
            END FOR
            OTHERWISE
                FOR each digit IN the list consisting of two, five, eight
                    IF the frequency OF digit IN count GREATER THAN one THEN
                        DECREMENT the frequency OF digit IN count BY two
                        BREAK OUT OF the FOR loop
                    END IF
                END FOR
            END OTHERWISE
        END IF

        IF remainder EQUALS two THEN
            FOR each digit IN the list consisting of two, five, eight
                IF the frequency OF digit IN count GREATER THAN zero THEN
                    DECREMENT the frequency OF digit IN count BY one
                    BREAK OUT OF the FOR loop
                END IF
            END FOR
            OTHERWISE
                FOR each digit IN the list consisting of one, four, seven
                    IF the frequency OF digit IN count GREATER THAN one THEN
                        DECREMENT the frequency OF digit IN count BY two
                        BREAK OUT OF the FOR loop
                    END IF
                END FOR
            END OTHERWISE
        END IF

        SET result TO an empty list
        FOR digit FROM nine DOWNTO zero INCLUSIVE
            APPEND the string CONSTRUCTED BY REPEATING digit the frequency OF digit TIMES TO result
        END FOR

        SET final_number TO the concatenation OF the elements IN result

        IF final_number IS NOT empty AND the element AT position zero OF final_number EQUALS the character zero THEN
            RETURN the string zero
        ELSE
            RETURN final_number
        END IF
    END FUNCTION
END CLASS
