CLASS Solution
    FUNCTION numberOfArrays(string_s, integer_k) RETURNS integer
        SET MOD TO ten raised to the power of nine PLUS one
        SET n TO the LENGTH OF string_s

        CREATE list_dp OF LENGTH n PLUS one FILLED WITH zeroes
        SET element at position zero OF list_dp TO one

        FOR index_i FROM one TO n INCLUSIVE
            FOR index_j FROM zero TO index_i MINUS one INCLUSIVE
                IF element at position index_j OF string_s NOT EQUALS character zero
                   AND THE INTEGER VALUE OF the substring FROM position index_j TO position index_i MINUS one OF string_s IS GREATER THAN OR EQUAL TO one
                   AND THE INTEGER VALUE OF the substring FROM position index_j TO position index_i MINUS one OF string_s IS LESS THAN OR EQUAL TO integer_k
                THEN
                    SET element at position index_i OF list_dp TO element at position index_i OF list_dp PLUS element at position index_j OF list_dp MODULO MOD
                END IF
            END FOR
        END FOR

        RETURN element at position n OF list_dp
    END FUNCTION
END CLASS
