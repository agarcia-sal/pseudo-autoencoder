CLASS Solution
    FUNCTION minSwap(nums1, nums2)
        SET n TO the LENGTH OF nums1
        SET dp TO a list containing n elements each of which is a list of two zeros
        
        SET the element at position zero OF element at position zero OF dp TO zero
        SET the element at position one OF element at position zero OF dp TO one
        
        FOR index i FROM one TO n MINUS one INCLUSIVE
            IF the element at position i OF nums1 GREATER THAN the element at position i MINUS one OF nums1 AND the element at position i OF nums2 GREATER THAN the element at position i MINUS one OF nums2 THEN
                IF the element at position i OF nums1 GREATER THAN the element at position i MINUS one OF nums2 AND the element at position i OF nums2 GREATER THAN the element at position i MINUS one OF nums1 THEN
                    SET the element at position zero OF element at position i OF dp TO the MINIMUM OF the element at position zero OF element at position i MINUS one OF dp AND the element at position one OF element at position i MINUS one OF dp
                    SET the element at position one OF element at position i OF dp TO the element at position zero OF element at position i OF dp PLUS one
                ELSE
                    SET the element at position zero OF element at position i OF dp TO the element at position zero OF element at position i MINUS one OF dp
                    SET the element at position one OF element at position i OF dp TO the element at position one OF element at position i MINUS one OF dp PLUS one
                END IF
            ELSE
                IF the element at position i OF nums1 GREATER THAN the element at position i MINUS one OF nums2 AND the element at position i OF nums2 GREATER THAN the element at position i MINUS one OF nums1 THEN
                    SET the element at position zero OF element at position i OF dp TO the element at position one OF element at position i MINUS one OF dp
                    SET the element at position one OF element at position i OF dp TO the element at position zero OF element at position i MINUS one OF dp PLUS one
                ELSE
                    SET the element at position zero OF element at position i OF dp TO the element at position one OF element at position i MINUS one OF dp
                    SET the element at position one OF element at position i OF dp TO the element at position zero OF element at position i MINUS one OF dp PLUS one
                END IF
            END IF
        END FOR
        
        RETURN the MINIMUM OF the element at position zero OF element at position n MINUS one OF dp AND the element at position one OF element at position n MINUS one OF dp
    END FUNCTION
END CLASS
