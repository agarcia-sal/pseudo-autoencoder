CLASS Solution
    FUNCTION generateParenthesis(n)
        FUNCTION backtrack(s SET TO empty string, left SET TO zero, right SET TO zero)
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n
                APPEND s TO result
                RETURN
            END IF
            IF left LESS THAN n
                CALL backtrack WITH s CONCATENATED WITH open parenthesis, left INCREMENTED BY one, right
            END IF
            IF right LESS THAN left
                CALL backtrack WITH s CONCATENATED WITH close parenthesis, left, right INCREMENTED BY one
            END IF
        END FUNCTION

        SET result TO empty list
        CALL backtrack WITH no arguments
        RETURN result
    END FUNCTION
END CLASS
