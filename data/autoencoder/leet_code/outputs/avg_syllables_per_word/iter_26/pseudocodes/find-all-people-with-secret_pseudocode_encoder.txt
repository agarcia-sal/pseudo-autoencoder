CLASS Solution
    FUNCTION findAllPeople(n, meetings, firstPerson)
        SET know_secret TO the set containing zero AND firstPerson
        
        SORT meetings BY the time element of each meeting
        
        SET i TO zero
        WHILE i LESS THAN the LENGTH OF meetings HOLDS
            SET current_time TO the time element of the meeting at position i in meetings
            
            ASSIGN graph TO a new default dictionary of lists
            
            WHILE i LESS THAN the LENGTH OF meetings AND the time element of the meeting at position i EQUALS current_time HOLDS
                DESTRUCTURE the meeting at position i INTO x, y, and ignore the third element
                APPEND y TO the list at key x in graph
                APPEND x TO the list at key y in graph
                INCREMENT i BY one
            END WHILE
            
            ASSIGN queue TO a new deque initialized with each person in know_secret that is a key in graph
            ASSIGN visited TO an empty set
            
            WHILE queue IS NOT empty
                REMOVE and ASSIGN the leftmost element OF queue TO person
                IF person IS IN visited THEN
                    CONTINUE to next iteration
                END IF
                ADD person TO visited
                ADD person TO know_secret
                FOR each neighbor IN the list at key person in graph
                    IF neighbor IS NOT IN visited THEN
                        APPEND neighbor TO queue
                    END IF
                END FOR
            END WHILE
        END WHILE
        
        RETURN the list form OF know_secret
    END FUNCTION
END CLASS
