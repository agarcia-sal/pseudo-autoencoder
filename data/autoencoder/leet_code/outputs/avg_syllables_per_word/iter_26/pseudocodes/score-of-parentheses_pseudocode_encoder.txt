CLASS Solution
    FUNCTION scoreOfParentheses(string_s)
        SET stack TO empty list
        SET score TO zero

        FOR each character IN string_s
            IF character EQUALS the single left parenthesis symbol
                APPEND score TO stack
                SET score TO zero
            ELSE
                SET score TO the value popped FROM stack PLUS
                    IF score EQUALS zero THEN the integer one ELSE the integer two MULTIPLIED BY score
            END IF
        END FOR

        RETURN score
    END FUNCTION
END CLASS
