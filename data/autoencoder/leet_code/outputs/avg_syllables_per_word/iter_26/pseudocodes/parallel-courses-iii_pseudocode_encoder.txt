CLASS Solution
    FUNCTION minimumTime(n, relations, time)
        ASSIGN reference TO a new default dictionary with list as default value NAMED graph
        FOR each pair of previous course and next course IN relations
            APPEND previous course TO element at position next course OF graph
        END FOR

        SET memo TO a list of length n PLUS one FILLED with minus one

        FUNCTION dfs(course)
            IF element at position course OF memo NOT EQUALS minus one
                RETURN element at position course OF memo
            END IF

            SET max prerequisite time TO zero
            FOR each previous course IN element at position course OF graph
                SET max prerequisite time TO the GREATER OF max prerequisite time AND dfs(previous course)
            END FOR

            SET element at position course OF memo TO max prerequisite time PLUS element at position course MINUS one OF time
            RETURN element at position course OF memo
        END FUNCTION

        SET total minimum time TO zero
        FOR course FROM one TO n INCLUSIVE
            SET total minimum time TO the GREATER OF total minimum time AND dfs(course)
        END FOR

        RETURN total minimum time
    END FUNCTION
END CLASS
