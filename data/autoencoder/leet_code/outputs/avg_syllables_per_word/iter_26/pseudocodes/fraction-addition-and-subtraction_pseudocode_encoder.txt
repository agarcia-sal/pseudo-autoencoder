CLASS Solution
    FUNCTION fractionAddition(expression)
        SET modified_expression TO expression WITH all occurrences of character minus REPLACED BY character plus FOLLOWED BY character minus
        SET list_of_fractions TO the RESULT OF SPLITTING modified_expression BY character plus
        SET filtered_fractions TO the COLLECTION OF fractions IN list_of_fractions WHERE fraction IS NOT empty
        SET result_fraction TO the SUM OF APPLYING the Fraction REPRESENTATION TO each fraction IN filtered_fractions
        SET numerator_part TO the NUMERATOR COMPONENT OF result_fraction
        SET denominator_part TO the DENOMINATOR COMPONENT OF result_fraction
        RETURN the CONCATENATION OF numerator_part AS string FOLLOWED BY character slash FOLLOWED BY denominator_part AS string
    END FUNCTION
END CLASS
