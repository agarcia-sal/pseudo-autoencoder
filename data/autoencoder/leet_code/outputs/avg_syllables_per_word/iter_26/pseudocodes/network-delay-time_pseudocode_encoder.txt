CLASS Solution
    FUNCTION networkDelayTime(times AS List OF List OF Integer, n AS Integer, k AS Integer) RETURNS Integer
        SET graph TO a dictionary with keys FROM integer one TO integer n INCLUSIVE each mapping TO an empty list

        FOR each elements u, v, w IN times
            APPEND the tuple of elements v and w TO the list at key u IN graph
        END FOR

        SET dist TO a dictionary with keys FROM integer one TO integer n INCLUSIVE each mapping TO positive infinity
        SET the value at key k IN dist TO zero

        SET min_heap TO a list containing a single tuple of zero and k

        WHILE min_heap HOLDS
            REMOVE and ASSIGN the smallest element of min_heap TO the variables current_dist and node

            IF current_dist IS GREATER THAN the value at key node IN dist
                CONTINUE to the next iteration of WHILE
            END IF

            FOR each elements neighbor and weight IN the list at key node IN graph
                SET distance TO current_dist PLUS weight

                IF distance IS LESS THAN the value at key neighbor IN dist
                    SET the value at key neighbor IN dist TO distance
                    ADD the tuple of distance and neighbor TO min_heap maintaining heap order
                END IF
            END FOR
        END WHILE

        SET max_time TO the maximum value AMONG all values IN dist

        IF max_time IS EQUAL TO positive infinity
            RETURN negative one
        ELSE
            RETURN max_time
        END IF
    END FUNCTION
END CLASS
