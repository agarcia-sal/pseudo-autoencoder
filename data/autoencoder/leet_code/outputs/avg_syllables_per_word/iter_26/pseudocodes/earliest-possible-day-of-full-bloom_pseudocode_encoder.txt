CLASS Solution
    FUNCTION earliestFullBloom(plantTime, growTime)
        SET tasks TO the LIST OF PAIRS OF element AT position index OF plantTime AND element AT position index OF growTime FOR index FROM the FIRST POSITION TO the LAST POSITION OF plantTime SORTED BY the SECOND ELEMENT OF EACH PAIR IN DESCENDING ORDER
        SET current_day TO zero
        SET earliest_bloom TO zero
        FOR each pair OF plant AND grow IN tasks
            INCREMENT current_day BY plant
            SET bloom_day TO current_day PLUS grow
            IF earliest_bloom LESS THAN bloom_day
                SET earliest_bloom TO bloom_day
            END IF
        END FOR
        RETURN earliest_bloom
    END FUNCTION
END CLASS
