CLASS Solution
    FUNCTION hIndex(list_of_citations)
        SET total_number TO the LENGTH OF list_of_citations
        SET left_pointer TO zero
        SET right_pointer TO total_number MINUS one
        SET current_h_index TO zero

        WHILE left_pointer LESS THAN OR EQUAL TO right_pointer HOLDS
            SET middle_position TO left_pointer PLUS the INTEGER DIVISION OF the DIFFERENCE BETWEEN right_pointer AND left_pointer BY two
            IF the ELEMENT at position middle_position OF list_of_citations GREATER THAN OR EQUAL TO total_number MINUS middle_position
                SET current_h_index TO total_number MINUS middle_position
                SET right_pointer TO middle_position MINUS one
            ELSE
                SET left_pointer TO middle_position PLUS one
            END IF
        END WHILE

        RETURN current_h_index
    END FUNCTION
END CLASS
