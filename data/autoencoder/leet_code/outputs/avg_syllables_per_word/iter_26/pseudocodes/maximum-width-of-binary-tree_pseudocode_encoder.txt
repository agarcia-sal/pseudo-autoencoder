CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS none OR the LENGTH OF values EQUALS zero
        RETURN none
    END IF
    SET root TO A NEW TreeNode WITH element at position zero OF values
    SET i TO one
    SET queue TO AN EMPTY double ended queue
    APPEND root TO queue
    WHILE queue IS NOT empty HOLDS
        SET node TO THE FIRST element REMOVED FROM queue
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none
            SET node.left TO A NEW TreeNode WITH element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none
            SET node.right TO A NEW TreeNode WITH element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF p.val NOT EQUALS q.val
        RETURN false
    ELSE
        RETURN the RESULT OF is_same_tree WITH p.left AND q.left AND the RESULT OF is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION widthOfBinaryTree(self, root)
        IF root IS none
            RETURN zero
        END IF
        SET queue TO A double ended queue INITIALIZED WITH A TUPLE OF root AND zero
        SET max_width TO zero
        WHILE queue IS NOT empty HOLDS
            SET level_length TO the LENGTH OF queue
            SET first_index TO THE SECOND element OF THE FIRST element OF queue
            FOR times_index FROM zero TO level_length MINUS one INCLUSIVE
                SET node, index TO THE FIRST element REMOVED FROM queue
                IF times_index EQUALS level_length MINUS one
                    SET max_width TO THE GREATER OF max_width AND index MINUS first_index PLUS one
                END IF
                IF node.left IS NOT none
                    APPEND A TUPLE OF node.left AND index MULTIPLIED BY two TO queue
                END IF
                IF node.right IS NOT none
                    APPEND A TUPLE OF node.right AND index MULTIPLIED BY two PLUS one TO queue
                END IF
            END FOR
        END WHILE
        RETURN max_width
    END FUNCTION
END CLASS
