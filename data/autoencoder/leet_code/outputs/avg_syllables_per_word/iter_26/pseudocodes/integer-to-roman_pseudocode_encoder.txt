CLASS Solution
    FUNCTION intToRoman(parameter num)
        SET list_of_values TO a list containing one thousand PLUS nine hundred PLUS five hundred PLUS four hundred PLUS one hundred PLUS ninety PLUS fifty PLUS forty PLUS ten PLUS nine PLUS five PLUS four PLUS one
        SET list_of_symbols TO a list containing the Roman numeral symbol for one thousand PLUS the Roman numeral symbol for nine hundred PLUS the Roman numeral symbol for five hundred PLUS the Roman numeral symbol for four hundred PLUS the Roman numeral symbol for one hundred PLUS the Roman numeral symbol for ninety PLUS the Roman numeral symbol for fifty PLUS the Roman numeral symbol for forty PLUS the Roman numeral symbol for ten PLUS the Roman numeral symbol for nine PLUS the Roman numeral symbol for five PLUS the Roman numeral symbol for four PLUS the Roman numeral symbol for one

        SET roman_numeral_string TO an empty string
        SET index_i TO zero
        WHILE num GREATER THAN zero HOLDS
            SET how_many_times TO the integer division of num DIVIDED BY element at position index_i OF list_of_values
            FOR each count IN the range from zero TO how_many_times exclusive
                APPEND element at position index_i OF list_of_symbols TO roman_numeral_string
                DECREMENT num BY element at position index_i OF list_of_values
            END FOR
            INCREMENT index_i BY one
        END WHILE
        RETURN roman_numeral_string
    END FUNCTION
END CLASS
