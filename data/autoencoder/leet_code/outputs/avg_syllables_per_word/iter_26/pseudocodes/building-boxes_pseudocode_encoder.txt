CLASS Solution
    FUNCTION minimumBoxes(n)
        FUNCTION triangular_pyramid(k)
            RETURN k MULTIPLIED BY the sum of k PLUS one MULTIPLIED BY the sum of k PLUS two DIVIDED BY six
        END FUNCTION

        SET left TO zero
        SET right TO n
        WHILE left LESS THAN right HOLDS
            SET mid TO the sum of left PLUS right PLUS one DIVIDED BY two
            IF triangular_pyramid(mid) LESS THAN OR EQUAL TO n
                SET left TO mid
            ELSE
                SET right TO the difference of right MINUS one
            END IF
        END WHILE

        SET boxes_used TO triangular_pyramid(left)
        SET remaining TO n MINUS boxes_used
        SET floor_boxes TO left MULTIPLIED BY the sum of left PLUS one DIVIDED BY two

        SET additional_boxes TO zero
        WHILE remaining GREATER THAN zero HOLDS
            INCREMENT additional_boxes BY one
            DECREMENT remaining BY additional_boxes
        END WHILE

        RETURN floor_boxes PLUS additional_boxes
    END FUNCTION
END CLASS
