CLASS Solution
    FUNCTION maximumNumberOfOnes(width, height, sideLength, maxOnes)
        SET frequency_list TO a list of zeros with the LENGTH EQUAL TO sideLength MULTIPLIED BY sideLength
        FOR each index i FROM zero TO width MINUS one INCLUSIVE
            FOR each index j FROM zero TO height MINUS one INCLUSIVE
                SET position_in_submatrix TO the REMAINDER OF i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS the REMAINDER OF j DIVIDED BY sideLength
                INCREMENT the element at position position_in_submatrix OF frequency_list BY one
            END FOR
        END FOR
        SORT frequency_list INTO descending order
        SET sum_of_top_frequencies TO the SUM of the FIRST maxOnes ELEMENTS OF frequency_list
        RETURN sum_of_top_frequencies
    END FUNCTION
END CLASS
