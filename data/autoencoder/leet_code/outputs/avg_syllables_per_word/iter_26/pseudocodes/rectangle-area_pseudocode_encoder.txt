CLASS Solution
    FUNCTION computeArea(ax1, ay1, ax2, ay2, bx1, by1, bx2, by2)
        SET area1 TO the difference of element at position ax2 and element at position ax1 MULTIPLIED BY the difference of element at position ay2 and element at position ay1
        SET area2 TO the difference of element at position bx2 and element at position bx1 MULTIPLIED BY the difference of element at position by2 and element at position by1
        SET overlap_width TO the maximum of zero AND the difference of the minimum of element at position ax2 and element at position bx2 AND the maximum of element at position ax1 and element at position bx1
        SET overlap_height TO the maximum of zero AND the difference of the minimum of element at position ay2 and element at position by2 AND the maximum of element at position ay1 and element at position by1
        SET overlap_area TO overlap_width MULTIPLIED BY overlap_height
        SET total_area TO area1 PLUS area2 MINUS overlap_area
        RETURN total_area
    END FUNCTION
END CLASS
