CLASS Solution
    FUNCTION change(amount, coins)
        INITIALIZE list_of_ways TO a list containing zero repeated the number of times equal to amount PLUS one
        SET the element at position zero OF list_of_ways TO one

        FOR each coin IN coins
            FOR variable x FROM coin TO amount INCLUSIVE
                INCREMENT element at position x OF list_of_ways BY element at position x MINUS coin OF list_of_ways
            END FOR
        END FOR

        RETURN element at position amount OF list_of_ways
    END FUNCTION
END CLASS
