CLASS Solution
    FUNCTION findContestMatch(n)
        SET teams TO a list created by converting each number from one TO n inclusive INTO string form
        WHILE n GREATER THAN one HOLDS
            FOR i FROM zero TO n DIVIDED BY two MINUS one INCLUSIVE
                SET teams element at position i TO a string expressing an open parenthesis FOLLOWED BY the element at position i OF teams FOLLOWED BY a comma FOLLOWED BY the element at position n MINUS one MINUS i OF teams FOLLOWED BY a close parenthesis
            END FOR
            SET n TO n DIVIDED BY two
        END WHILE
        RETURN the element at position zero OF teams
    END FUNCTION
END CLASS
