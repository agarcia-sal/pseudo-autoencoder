CLASS Solution
    FUNCTION subsetsWithDup(list_of_numbers)
        CALL the SORT METHOD ON list_of_numbers TO ARRANGE numbers in ascending order
        SET result TO an empty list
        SET subset TO an empty list
        
        FUNCTION dfs(index)
            IF index GREATER THAN OR EQUAL TO the LENGTH OF list_of_numbers
                APPEND a COPY OF subset TO result
                RETURN
            END IF
            
            APPEND element at position index OF list_of_numbers TO subset
            CALL dfs WITH argument index PLUS one
            
            REMOVE the last element FROM subset
            WHILE index PLUS one LESS THAN the LENGTH OF list_of_numbers AND element at position index OF list_of_numbers EQUALS element at position index PLUS one OF list_of_numbers
                INCREMENT index BY one
            END WHILE
            CALL dfs WITH argument index PLUS one
        END FUNCTION
        
        CALL dfs WITH argument zero
        RETURN result
    END FUNCTION
END CLASS
