CLASS Solution
    FUNCTION compress(chars)
        IF the LENGTH OF chars EQUALS zero
            RETURN zero
        END IF

        SET write_index TO zero
        SET read_index TO zero

        WHILE read_index LESS THAN the LENGTH OF chars HOLDS
            SET char TO the element at position read_index OF chars
            SET count TO zero

            WHILE read_index LESS THAN the LENGTH OF chars AND the element at position read_index OF chars EQUALS char HOLDS
                INCREMENT read_index BY one
                INCREMENT count BY one
            END WHILE

            SET the element at position write_index OF chars TO char
            INCREMENT write_index BY one

            IF count GREATER THAN one
                FOR each digit_character IN the STRING REPRESENTATION OF count
                    SET the element at position write_index OF chars TO digit_character
                    INCREMENT write_index BY one
                END FOR
            END IF
        END WHILE

        RETURN write_index
    END FUNCTION
END CLASS
