CLASS Solution
    FUNCTION solve(board)
        IF board IS empty OR the element at position zero OF board IS empty
            RETURN
        END IF

        SET m TO the LENGTH OF board
        SET n TO the LENGTH OF the element at position zero OF board

        FUNCTION dfs(x, y)
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR the element at position x OF board AND position y NOT EQUALS letter O
                RETURN
            END IF

            SET the element at position x OF board AND position y TO letter S

            CALL dfs WITH x PLUS one AND y
            CALL dfs WITH x MINUS one AND y
            CALL dfs WITH x AND y PLUS one
            CALL dfs WITH x AND y MINUS one
        END FUNCTION

        FOR i FROM zero TO m MINUS one INCLUSIVE
            CALL dfs WITH i AND zero
            CALL dfs WITH i AND n MINUS one
        END FOR

        FOR j FROM zero TO n MINUS one INCLUSIVE
            CALL dfs WITH zero AND j
            CALL dfs WITH m MINUS one AND j
        END FOR

        FOR i FROM zero TO m MINUS one INCLUSIVE
            FOR j FROM zero TO n MINUS one INCLUSIVE
                IF the element at position i OF board AND position j EQUALS letter O
                    SET the element at position i OF board AND position j TO letter X
                END IF
            END FOR
        END FOR

        FOR i FROM zero TO m MINUS one INCLUSIVE
            FOR j FROM zero TO n MINUS one INCLUSIVE
                IF the element at position i OF board AND position j EQUALS letter S
                    SET the element at position i OF board AND position j TO letter O
                END IF
            END FOR
        END FOR
    END FUNCTION
END CLASS
