CLASS Solution
    FUNCTION maxChunksToSorted(arr)
        SET maximum_value_so_far TO zero
        SET chunk_count TO zero
        
        FOR index FROM zero TO the LENGTH OF arr MINUS one INCLUSIVE
            SET maximum_value_so_far TO the GREATER OF maximum_value_so_far AND the element at position index OF arr
            IF maximum_value_so_far EQUALS index
                INCREMENT chunk_count BY one
            END IF
        END FOR
        
        RETURN chunk_count
    END FUNCTION
END CLASS
