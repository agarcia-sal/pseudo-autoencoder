CLASS Solution
    FUNCTION addBoldTag(s STRING, words LIST OF STRING) RETURNS STRING
        IF the LIST words IS EMPTY
            RETURN s
        END IF

        SET n TO the LENGTH OF s
        SET mask TO a LIST OF boolean FALSE repeated n times

        FOR each word IN words
            SET start TO zero
            WHILE start IS LESS THAN OR EQUAL TO n MINUS the LENGTH OF word
                SET start TO the POSITION OF the FIRST OCCURRENCE OF word IN s STARTING FROM POSITION start
                IF start EQUALS negative one
                    BREAK the WHILE loop
                END IF
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one INCLUSIVE
                    SET the ELEMENT at POSITION i OF mask TO TRUE
                END FOR
                INCREMENT start BY one
            END WHILE
        END FOR

        SET result TO an EMPTY LIST
        SET i TO zero
        WHILE i IS LESS THAN n
            IF the ELEMENT at POSITION i OF mask IS TRUE
                APPEND the OPENING BOLD TAG TO result
                WHILE i IS LESS THAN n AND the ELEMENT at POSITION i OF mask IS TRUE
                    APPEND the ELEMENT at POSITION i OF s TO result
                    INCREMENT i BY one
                END WHILE
                APPEND the CLOSING BOLD TAG TO result
            ELSE
                APPEND the ELEMENT at POSITION i OF s TO result
                INCREMENT i BY one
            END IF
        END WHILE

        RETURN the CONCATENATION OF all ELEMENTS IN result
    END FUNCTION
END CLASS
