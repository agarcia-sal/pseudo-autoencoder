CLASS Solution
    FUNCTION optimalDivision(nums)
        IF the LENGTH OF nums EQUALS one
            RETURN the string representation OF the element at position zero OF nums
        END IF
        IF the LENGTH OF nums EQUALS two
            RETURN the string representation OF the element at position zero OF nums CONCATENATED WITH the division symbol CONCATENATED WITH the string representation OF the element at position one OF nums
        END IF
        RETURN the string representation OF the element at position zero OF nums CONCATENATED WITH the division symbol CONCATENATED WITH an open parenthesis CONCATENATED WITH the joining of the string representations OF each element FROM position one TO the last position OF nums SEPARATED BY the division symbol CONCATENATED WITH a close parenthesis
    END FUNCTION
END CLASS
