CLASS Solution
    FUNCTION minInsertions(string_input)
        FUNCTION longest_palindromic_subsequence(string_input)
            SET length_of_string TO the LENGTH OF string_input
            CREATE two_dimensional_list dp WITH dimensions length_of_string BY length_of_string FILLED WITH zero
            
            FOR index FROM zero TO length_of_string MINUS one INCLUSIVE
                SET element at position index AND position index OF dp TO one
            END FOR
            
            FOR current_length FROM two TO length_of_string INCLUSIVE
                FOR start_index FROM zero TO length_of_string MINUS current_length INCLUSIVE
                    SET end_index TO start_index PLUS current_length MINUS one
                    IF element at position start_index OF string_input EQUALS element at position end_index OF string_input AND current_length EQUALS two
                        SET element at position start_index AND position end_index OF dp TO two
                    ELSE IF element at position start_index OF string_input EQUALS element at position end_index OF string_input
                        SET element at position start_index AND position end_index OF dp TO element at position start_index PLUS one AND position end_index MINUS one OF dp PLUS two
                    ELSE
                        SET element at position start_index AND position end_index OF dp TO the GREATER BETWEEN element at position start_index AND position end_index MINUS one OF dp AND element at position start_index PLUS one AND position end_index OF dp
                    END IF
                END FOR
            END FOR
            
            RETURN element at position zero AND position length_of_string MINUS one OF dp
        END FUNCTION
        
        RETURN the LENGTH OF string_input MINUS longest_palindromic_subsequence(string_input)
    END FUNCTION
END CLASS
