CLASS Solution
    FUNCTION rob(parameter nums)
        FUNCTION rob_linear(parameter houses)
            IF houses IS EMPTY
                RETURN zero
            END IF
            
            IF the LENGTH OF houses EQUALS one
                RETURN the FIRST element OF houses
            END IF
            
            SET dp TO a list of zeroes WITH LENGTH EQUAL TO the LENGTH OF houses
            SET the FIRST element OF dp TO the FIRST element OF houses
            SET the SECOND element OF dp TO the GREATER VALUE OF the FIRST element OF houses AND the SECOND element OF houses
            
            FOR index i FROM two TO the LENGTH OF houses MINUS one INCLUSIVE
                SET the element AT position i OF dp TO the GREATER VALUE OF the element AT position i MINUS one OF dp AND the SUM OF the element AT position i MINUS two OF dp AND the element AT position i OF houses
            END FOR
            
            RETURN the LAST element OF dp
        END FUNCTION
        
        IF nums IS EMPTY
            RETURN zero
        END IF
        
        IF the LENGTH OF nums EQUALS one
            RETURN the FIRST element OF nums
        END IF
        
        RETURN the GREATER VALUE OF rob_linear(the SUBLIST OF nums FROM position zero TO the SECOND TO LAST element) AND rob_linear(the SUBLIST OF nums FROM position one TO the LAST element)
    END FUNCTION
END CLASS
