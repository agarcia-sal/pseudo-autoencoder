CLASS Solution
    FUNCTION rotate(matrix)
        SET size_n TO the LENGTH OF matrix

        FOR index_i FROM zero TO size_n MINUS one INCLUSIVE
            FOR index_j FROM index_i TO size_n MINUS one INCLUSIVE
                SET temporary_value TO the element at position index_i OF matrix AT position index_j
                SET the element at position index_i OF matrix AT position index_j TO the element at position index_j OF matrix AT position index_i
                SET the element at position index_j OF matrix AT position index_i TO temporary_value
            END FOR
        END FOR

        FOR index_i FROM zero TO size_n MINUS one INCLUSIVE
            REVERSE the order of elements IN the element at position index_i OF matrix
        END FOR
    END FUNCTION
END CLASS
