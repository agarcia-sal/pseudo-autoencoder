CLASS Solution
    FUNCTION validSubarrays(parameter nums AS list OF integers)
        SET stack TO empty list
        SET count TO zero

        FOR each num IN nums
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at the last position OF stack IS GREATER THAN num HOLDS
                REMOVE the element at the last position FROM stack
            END WHILE
            APPEND num TO stack
            INCREMENT count BY the LENGTH OF stack
        END FOR

        RETURN count
    END FUNCTION
END CLASS
