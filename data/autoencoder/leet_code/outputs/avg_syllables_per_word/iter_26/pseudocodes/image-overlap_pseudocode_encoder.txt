CLASS Solution
    FUNCTION largestOverlap(image_one AS List OF List OF integer, image_two AS List OF List OF integer) RETURNS integer
        SET dimension_n TO the LENGTH OF image_one
        SET list_points_one TO AN EMPTY list
        FOR index_i FROM zero TO dimension_n MINUS one INCLUSIVE
            FOR index_j FROM zero TO dimension_n MINUS one INCLUSIVE
                IF the element at position index_i OF image_one AT position index_j EQUALS one
                    APPEND the tuple of index_i AND index_j TO list_points_one
                END IF
            END FOR
        END FOR
        SET list_points_two TO AN EMPTY list
        FOR index_i FROM zero TO dimension_n MINUS one INCLUSIVE
            FOR index_j FROM zero TO dimension_n MINUS one INCLUSIVE
                IF the element at position index_i OF image_two AT position index_j EQUALS one
                    APPEND the tuple of index_i AND index_j TO list_points_two
                END IF
            END FOR
        END FOR
        ASSIGN overlap_counter TO AN EMPTY mapping FROM tuple TO integer WITH default value zero
        FOR each tuple_x1_y1 IN list_points_one
            DECOMPOSE tuple_x1_y1 INTO x1 AND y1
            FOR each tuple_x2_y2 IN list_points_two
                DECOMPOSE tuple_x2_y2 INTO x2 AND y2
                SET difference_x TO x2 MINUS x1
                SET difference_y TO y2 MINUS y1
                INCREMENT the value in overlap_counter AT key tuple of difference_x AND difference_y BY one
            END FOR
        END FOR
        SET maximum_overlap TO the greatest value AMONG all values IN overlap_counter OR zero IF overlap_counter IS empty
        RETURN maximum_overlap
    END FUNCTION
END CLASS
