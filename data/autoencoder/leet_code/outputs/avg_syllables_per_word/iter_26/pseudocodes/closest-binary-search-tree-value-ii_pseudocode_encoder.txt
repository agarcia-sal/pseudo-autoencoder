CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS EQUAL TO empty
        RETURN none
    END IF
    SET root TO A NEW TreeNode WITH value AS element at position zero OF values
    SET i TO one
    SET queue TO AN EMPTY deque COLLECTION
    APPEND root TO queue
    WHILE queue HOLDS
        SET node TO THE FIRST ELEMENT REMOVED FROM queue
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT EQUAL TO none
            SET node.left TO A NEW TreeNode WITH value AS element at position i OF values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT EQUAL TO none
            SET node.right TO A NEW TreeNode WITH value AS element at position i OF values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS EQUAL TO none AND q IS EQUAL TO none
        RETURN true
    ELSE IF p IS EQUAL TO none OR q IS EQUAL TO none
        RETURN false
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN false
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION closestKValues(self, root, target, k)
        FUNCTION inorder_traversal(node)
            IF node IS EQUAL TO none
                RETURN
            END IF
            CALL inorder_traversal WITH node.left
            IF the LENGTH OF result IS LESS THAN k
                APPEND node.val TO result
            ELSE
                IF the ABSOLUTE VALUE OF node.val MINUS target IS LESS THAN the ABSOLUTE VALUE OF element at position zero OF result MINUS target
                    REMOVE the FIRST ELEMENT FROM result
                    APPEND node.val TO result
                ELSE
                    RETURN
                END IF
            END IF
            CALL inorder_traversal WITH node.right
        END FUNCTION

        SET result TO AN EMPTY deque COLLECTION
        CALL inorder_traversal WITH root
        RETURN a LIST CONSTRUCTED FROM result
    END FUNCTION
END CLASS
