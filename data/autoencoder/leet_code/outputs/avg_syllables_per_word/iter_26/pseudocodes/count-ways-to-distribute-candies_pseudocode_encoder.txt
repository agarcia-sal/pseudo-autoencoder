CLASS Solution
    FUNCTION waysToDistribute(parameter n AS integer, parameter k AS integer) RETURNS integer
        SET MODULO_VALUE TO ten RAISED TO THE POWER OF nine PLUS seven
        
        # Initialize the dp two dimensional list
        SET dp TO a two dimensional list CONSTRUCTED BY
            FOR each index_i FROM zero TO n INCLUSIVE
                A list of zeros with the length of k PLUS one
            END FOR
        
        # Base case assignment
        SET element at position zero OF list at position zero OF dp TO one
        
        # Fill the dp two dimensional list
        FOR each index_i FROM one TO n INCLUSIVE
            FOR each index_j FROM one TO k INCLUSIVE
                SET element at position index_j OF list at position index_i OF dp TO the remainder WHEN the sum of
                    the product of index_j AND element at position index_j OF list at position index_i MINUS one OF dp
                    AND element at position index_j MINUS one OF list at position index_i MINUS one OF dp
                IS DIVIDED BY MODULO_VALUE
            END FOR
        END FOR
        
        RETURN element at position n OF list at position k OF dp
    END FUNCTION
END CLASS
