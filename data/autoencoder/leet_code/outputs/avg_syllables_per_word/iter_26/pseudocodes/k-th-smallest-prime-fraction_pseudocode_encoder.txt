CLASS Solution
    FUNCTION kthSmallestPrimeFraction(arr AS List, k)
        SET n TO the LENGTH OF arr
        SET min_heap TO an EMPTY list
        
        FOR i FROM zero TO n MINUS one MINUS one INCLUSIVE
            CALL pushing INTO min_heap THE TUPLE OF the element at position i OF arr DIVIDED BY the element at position n MINUS one OF arr, i, AND n MINUS one
        END FOR
        
        FOR each IN the RANGE FROM zero TO k MINUS one MINUS one INCLUSIVE
            SET underscore AND i AND j TO REMOVING the smallest element FROM min_heap
            IF j MINUS one GREATER THAN i HOLDS
                CALL pushing INTO min_heap THE TUPLE OF the element at position i OF arr DIVIDED BY the element at position j MINUS one OF arr, i, AND j MINUS one
            END IF
        END FOR
        
        SET underscore AND i AND j TO REMOVING the smallest element FROM min_heap
        RETURN a list CONTAINING the element at position i OF arr AND the element at position j OF arr
    END FUNCTION
END CLASS
