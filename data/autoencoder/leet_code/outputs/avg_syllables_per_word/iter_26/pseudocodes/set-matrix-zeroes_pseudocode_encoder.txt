CLASS Solution
    FUNCTION setZeroes(matrix)
        IF matrix IS EMPTY OR the LENGTH OF matrix AT POSITION zero IS ZERO
            RETURN
        END IF

        SET m TO the LENGTH OF matrix
        SET n TO the LENGTH OF matrix AT POSITION zero

        SET first_row_has_zero TO WHETHER ANY element AT POSITION j OF matrix AT POSITION zero EQUALS zero FOR each j FROM zero TO n MINUS one INCLUSIVE
        SET first_col_has_zero TO WHETHER ANY element AT POSITION zero OF matrix AT POSITION i EQUALS zero FOR each i FROM zero TO m MINUS one INCLUSIVE

        FOR each i FROM one TO m MINUS one INCLUSIVE
            FOR each j FROM one TO n MINUS one INCLUSIVE
                IF element AT POSITION i OF matrix AT POSITION j EQUALS zero
                    SET element AT POSITION i OF matrix AT POSITION zero TO zero
                    SET element AT POSITION zero OF matrix AT POSITION j TO zero
                END IF
            END FOR
        END FOR

        FOR each i FROM one TO m MINUS one INCLUSIVE
            FOR each j FROM one TO n MINUS one INCLUSIVE
                IF element AT POSITION i OF matrix AT POSITION zero EQUALS zero OR element AT POSITION zero OF matrix AT POSITION j EQUALS zero
                    SET element AT POSITION i OF matrix AT POSITION j TO zero
                END IF
            END FOR
        END FOR

        IF first_row_has_zero
            FOR each j FROM zero TO n MINUS one INCLUSIVE
                SET element AT POSITION zero OF matrix AT POSITION j TO zero
            END FOR
        END IF

        IF first_col_has_zero
            FOR each i FROM zero TO m MINUS one INCLUSIVE
                SET element AT POSITION i OF matrix AT POSITION zero TO zero
            END FOR
        END IF
    END FUNCTION
END CLASS
