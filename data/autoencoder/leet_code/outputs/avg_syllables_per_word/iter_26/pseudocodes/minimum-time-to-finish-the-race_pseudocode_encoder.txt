CLASS Solution
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)
        SET min_times TO a list of infinity repeated fifteen times

        FOR each pair first_factor, reduction_factor IN tires
            SET total_time TO zero

            FOR index FROM zero TO fourteen INCLUSIVE
                SET lap_time TO first_factor MULTIPLIED BY reduction_factor RAISED TO THE POWER OF index

                IF lap_time GREATER THAN changeTime PLUS first_factor
                    BREAK THE INNER LOOP
                END IF

                INCREMENT total_time BY lap_time
                SET min_times element at position index TO the minimum BETWEEN min_times element at position index AND total_time
            END FOR
        END FOR

        SET dp TO a list of infinity repeated numLaps PLUS one times
        SET dp element at position zero TO zero

        FOR index_i FROM one TO numLaps INCLUSIVE
            FOR index_j FROM zero TO the minimum BETWEEN index_i MINUS one AND fourteen INCLUSIVE
                SET candidate TO dp element at position index_i MINUS index_j MINUS one PLUS min_times element at position index_j PLUS changeTime
                SET dp element at position index_i TO the minimum BETWEEN dp element at position index_i AND candidate
            END FOR
        END FOR

        RETURN dp element at position numLaps MINUS changeTime
    END FUNCTION
END CLASS
