CLASS Solution
    FUNCTION minOperations(target LIST OF INTEGERS, arr LIST OF INTEGERS)
        SET index_map TO AN EMPTY MAPPING FROM INTEGER TO INTEGER
        FOR each index AND value IN THE ENUMERATION OF target
            SET the MAPPING OF value TO index IN index_map
        END FOR

        SET transformed_arr TO AN EMPTY LIST OF INTEGERS
        FOR each value IN arr
            IF value IS A KEY IN index_map HOLDS
                APPEND the MAPPING OF value FROM index_map TO transformed_arr
            END IF
        END FOR

        SET lis TO AN EMPTY LIST OF INTEGERS
        FOR each number IN transformed_arr
            SET pos TO THE LEFTMOST POSITION TO INSERT number IN lis TO KEEP lis SORTED IN NON-DECREASING ORDER
            IF pos EQUALS THE LENGTH OF lis HOLDS
                APPEND number TO lis
            ELSE
                SET THE ELEMENT AT POSITION pos OF lis TO number
            END IF
        END FOR

        SET lcs_length TO THE LENGTH OF lis
        SET minimum_operations TO THE LENGTH OF target MINUS lcs_length

        RETURN minimum_operations
    END FUNCTION
END CLASS
