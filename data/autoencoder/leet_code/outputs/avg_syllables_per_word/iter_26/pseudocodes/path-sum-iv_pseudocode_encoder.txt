CLASS Solution
    FUNCTION pathSum(nums)
        SET tree TO an empty dictionary

        FOR each num IN nums
            SET depth TO num DIVIDED BY one hundred
            SET pos TO the remainder of num DIVIDED BY one hundred DIVIDED BY ten
            SET value TO the remainder of num DIVIDED BY ten
            SET the element at key formed by depth and pos IN tree TO value
        END FOR

        FUNCTION dfs(depth, pos, path_sum)
            SET current_value TO the element at key formed by depth and pos IN tree OR None IF not present
            IF current_value IS None
                RETURN zero
            END IF

            SET new_path_sum TO path_sum PLUS current_value

            SET left_child TO the pair formed by depth PLUS one AND pos MULTIPLIED BY two MINUS one
            SET right_child TO the pair formed by depth PLUS one AND pos MULTIPLIED BY two

            IF left_child NOT IN tree AND right_child NOT IN tree
                RETURN new_path_sum
            END IF

            SET left_sum TO dfs(element at position zero OF left_child, element at position one OF left_child, new_path_sum)
            SET right_sum TO dfs(element at position zero OF right_child, element at position one OF right_child, new_path_sum)

            RETURN left_sum PLUS right_sum
        END FUNCTION

        RETURN dfs(one, one, zero)
    END FUNCTION
END CLASS
