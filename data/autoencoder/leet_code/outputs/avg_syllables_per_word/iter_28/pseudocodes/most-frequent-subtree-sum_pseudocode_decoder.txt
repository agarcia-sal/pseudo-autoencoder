CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty DO  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.left TO new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none THEN  
            SET node.right TO new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN false  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS none THEN  
                RETURN zero  
            END IF  
            SET current_sum TO the val attribute of node PLUS subtree_sum(the left attribute of node) PLUS subtree_sum(the right attribute of node)  
            INCREMENT the value associated with current_sum in count BY one  
            RETURN current_sum  
        END FUNCTION  
        
        SET count TO an empty dictionary with default integer zero  
        CALL subtree_sum with root  
        
        SET max_freq TO the maximum value in the values of count WITH default zero  
        
        SET result TO an empty list  
        FOR each key AND freq IN the items of count DO  
            IF freq EQUALS max_freq THEN  
                APPEND key TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
