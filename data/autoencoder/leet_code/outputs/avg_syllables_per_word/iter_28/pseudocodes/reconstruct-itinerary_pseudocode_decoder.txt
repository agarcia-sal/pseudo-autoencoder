CLASS Solution  
    FUNCTION findItinerary WITH PARAMETER tickets  
        CREATE a graph as a mapping FROM a location TO a list of destinations  
        FOR each pair of departure location AND arrival location IN tickets sorted in reverse lexicographical order  
            APPEND the arrival location TO the list of destinations for the departure location IN the graph  
        END FOR  
        CREATE an empty list named itinerary  
        FUNCTION visit WITH PARAMETER airport  
            WHILE there exists at least one destination for the airport IN the graph  
                CALL visit WITH the last destination REMOVED from the list of destinations for the airport IN the graph  
            END WHILE  
            APPEND the airport TO itinerary  
        END FUNCTION  
        CALL visit WITH the location named JFK  
        RETURN the itinerary in reverse order  
    END FUNCTION  
END CLASS
