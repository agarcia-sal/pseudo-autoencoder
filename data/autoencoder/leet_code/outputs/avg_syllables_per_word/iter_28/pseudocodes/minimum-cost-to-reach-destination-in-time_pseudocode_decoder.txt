CLASS Solution  
    FUNCTION minCost(maxTime passingFees edges)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO an empty list of lists with size n  
        FOR each element x y time IN edges  
            APPEND a tuple of neighbor y and travel time time TO the element at position x of graph  
            APPEND a tuple of neighbor x and travel time time TO the element at position y of graph  
        END FOR  
        SET pq TO a list containing a tuple of the element at position zero of passingFees the number zero and the number zero  
        SET visited_time TO a list of the infinite value replicated n times  
        SET the element at position zero of visited_time TO the number zero  
        WHILE the LENGTH OF pq is greater than zero  
            REMOVE and RETURN the smallest element from pq AND ASSIGN it TO current_cost current_city current_time  
            IF current_city EQUALS n MINUS one  
                RETURN current_cost  
            END IF  
            FOR each tuple neighbor travel_time IN the element at position current_city of graph  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS the element at position neighbor of passingFees  
                IF new_time LESS THAN OR EQUAL TO maxTime AND (new_time LESS THAN the element at position neighbor of visited_time OR new_cost LESS THAN current_cost)  
                    SET the element at position neighbor of visited_time TO new_time  
                    ADD the tuple of new_cost neighbor new_time TO pq maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
