CLASS Solution  
    FUNCTION robotSim(commands, obstacles)  
        SET directions TO list of pairs representing zero PLUS one NORTH zero EAST one zero SOUTH zero MINUS one WEST MINUS one zero  
        SET x TO zero  
        SET y TO zero  
        SET dir_index TO zero  
        SET obstacles_set TO empty set  
        FOR each obstacle IN obstacles  
            CONVERT obstacle TO tuple_form  
            ADD tuple_form TO obstacles_set  
        END FOR  
        SET max_distance TO zero  

        FOR each command IN commands  
            IF command EQUALS negative two THEN  
                SET dir_index TO dir_index MINUS one MODULO four  
            ELSE IF command EQUALS negative one THEN  
                SET dir_index TO dir_index PLUS one MODULO four  
            ELSE  
                SET dx TO the first element of element at position dir_index of directions  
                SET dy TO the second element of element at position dir_index of directions  
                FOR step FROM one TO command  
                    SET next_x TO x PLUS dx  
                    SET next_y TO y PLUS dy  
                    IF the pair next_x next_y IS IN obstacles_set THEN  
                        BREAK the loop  
                    END IF  
                    SET x TO next_x  
                    SET y TO next_y  
                END FOR  
                SET current_distance TO x MULTIPLIED BY x PLUS y MULTIPLIED BY y  
                IF current_distance GREATER THAN max_distance THEN  
                    SET max_distance TO current_distance  
                END IF  
            END IF  
        END FOR  
        RETURN max_distance  
    END FUNCTION  
END CLASS
