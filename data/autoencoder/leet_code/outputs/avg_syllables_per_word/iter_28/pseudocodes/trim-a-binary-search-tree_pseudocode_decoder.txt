CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS NOT present  
        RETURN none  
    END IF  
    SET root TO a new TreeNode created with element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the length of values AND element at position i of values IS NOT none  
            SET node.left TO a new TreeNode created with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values IS NOT none  
            SET node.right TO a new TreeNode created with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS NOT present AND q IS NOT present  
        RETURN True  
    ELSE IF p IS NOT present OR q IS NOT present  
        RETURN False  
    ELSE IF the value of p IS NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with the left child of p and the left child of q AND the result of calling is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST(self, root, low, high)  
        IF root IS NOT present  
            RETURN none  
        END IF  
        IF the value of root LESS THAN low  
            RETURN the result of calling trimBST with the right child of root, low, and high  
        END IF  
        IF the value of root GREATER THAN high  
            RETURN the result of calling trimBST with the left child of root, low, and high  
        END IF  
        SET the left child of root TO the result of calling trimBST with the left child of root, low, and high  
        SET the right child of root TO the result of calling trimBST with the right child of root, low, and high  
        RETURN root  
    END FUNCTION  
END CLASS
