CLASS Solution  
    FUNCTION calculateMinimumHP(dungeon)  
        SET m TO the LENGTH OF dungeon  
        SET n TO the LENGTH OF element at position zero of dungeon  
        SET dp TO A list OF lists EACH CONTAINING the floating point value of infinity WITH dimensions m PLUS one BY n PLUS one  
        SET element at position m of dp AT position n MINUS one TO the integer value one  
        SET element at position m MINUS one of dp AT position n TO the integer value one  
        FOR i FROM m MINUS one TO zero DECREMENTING BY one  
            FOR j FROM n MINUS one TO zero DECREMENTING BY one  
                SET min_health TO the MINIMUM OF the element at position i PLUS one of dp AT position j AND the element at position i of dp AT position j PLUS one MINUS the element at position i of dungeon AT position j  
                SET element at position i of dp AT position j TO the MAXIMUM OF min_health AND the integer value one  
            END FOR  
        END FOR  
        RETURN element at position zero of dp AT position zero  
    END FUNCTION  
END CLASS
