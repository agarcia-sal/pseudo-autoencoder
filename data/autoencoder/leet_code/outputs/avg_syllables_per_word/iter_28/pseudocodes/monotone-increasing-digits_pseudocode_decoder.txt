CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO the list of characters of the string representation of n  
        SET length TO the number of elements in digits  
        SET marker TO length  
        FOR i FROM length MINUS one TO one STEP MINUS one  
            IF the element at position i of digits LESS THAN the element at position i MINUS one of digits  
                SET the element at position i MINUS one of digits TO the character representation of the integer value of the element at position i MINUS one of digits MINUS one  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS one  
            SET the element at position i of digits TO the character representation of nine  
        END FOR  
        SET result_string TO the concatenation of all elements in digits in order  
        SET result_integer TO the integer value of result_string  
        RETURN result_integer  
    END FUNCTION  
END CLASS
