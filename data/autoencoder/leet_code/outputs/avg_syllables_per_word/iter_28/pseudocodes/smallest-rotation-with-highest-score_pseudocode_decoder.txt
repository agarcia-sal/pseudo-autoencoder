CLASS Solution  
    FUNCTION bestRotation(nums)  
        SET n TO the LENGTH OF nums  
        SET delta TO a LIST OF n elements SET TO zero  
        
        FOR each position i AND element num IN nums WITH positions AND elements  
            IF num LESS THAN OR EQUAL TO i  
                INCREMENT the element at position zero of delta BY one  
                IF the result of position i MINUS num PLUS one LESS THAN n  
                    DECREMENT the element at position result of i MINUS num PLUS one of delta BY one  
                END IF  
            END IF  
            IF position i PLUS one LESS THAN n  
                INCREMENT the element at position result of i PLUS one of delta BY one  
                IF the result of i PLUS n MINUS num PLUS one LESS THAN n  
                    DECREMENT the element at position result of i PLUS n MINUS num PLUS one of delta BY one  
                END IF  
            END IF  
        END FOR  
        
        SET score TO zero  
        SET max_score TO zero  
        SET best_k TO zero  
        
        FOR each index k FROM zero TO n MINUS one  
            INCREMENT score BY the element at position k of delta  
            IF score GREATER THAN max_score  
                SET max_score TO score  
                SET best_k TO k  
            END IF  
        END FOR  
        
        RETURN best_k  
    END FUNCTION  
END CLASS
