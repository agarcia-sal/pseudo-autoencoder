CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET list_of_start_time_and_index TO a list created by pairing the first element of each interval WITH its position in intervals  
        SORT list_of_start_time_and_index by the first element in each pair in ascending order  
        SET result_list TO an empty list  
        FOR each element_ignored and interval_end IN intervals  
            SET position TO the smallest index in list_of_start_time_and_index such that the first element of the pair at that index is GREATER THAN OR EQUAL TO interval_end  
            IF position IS LESS THAN the length of list_of_start_time_and_index  
                APPEND the second element at position in list_of_start_time_and_index TO result_list  
            ELSE  
                APPEND negative one TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
