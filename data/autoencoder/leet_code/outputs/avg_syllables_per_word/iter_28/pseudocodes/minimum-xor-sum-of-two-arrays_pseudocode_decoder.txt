CLASS Solution  
    FUNCTION minimumXORSum(nums1 PARAMETER, nums2 PARAMETER)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a LIST OF size TWO RAISED TO THE POWER OF n FILLED WITH INFINITY  
        SET element at position zero of dp TO zero  
        
        FOR mask FROM zero TO TWO RAISED TO THE POWER OF n MINUS one  
            SET bit_count TO the COUNT OF ONES IN THE BINARY REPRESENTATION OF mask  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND OF mask AND TWO RAISED TO THE POWER OF j DOES NOT EQUAL zero  
                    SET value_one TO element at position mask OF dp  
                    SET value_two TO element at position the BITWISE XOR OF mask AND TWO RAISED TO THE POWER OF j OF dp PLUS the BITWISE XOR OF element at position bit_count MINUS one of nums1 AND element at position j of nums2  
                    IF value_two IS LESS THAN value_one  
                        SET element at position mask of dp TO value_two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position TWO RAISED TO THE POWER OF n MINUS one of dp  
    END FUNCTION  
END CLASS
