CLASS Solution  
    FUNCTION canConvert(str1 str2)  
        IF str1 EQUALS str2  
            RETURN True  
        END IF  
        SET conversion_map TO empty mapping  
        FOR each character named char1 in str1 AND each character named char2 in str2 at the same position  
            IF char1 IS A KEY IN conversion_map  
                IF value MAPPED TO char1 IN conversion_map NOT EQUALS char2  
                    RETURN False  
                END IF  
            ELSE  
                SET value MAPPED TO char1 IN conversion_map TO char2  
            END IF  
        END FOR  
        IF THE NUMBER OF UNIQUE CHARACTERS IN str2 EQUALS twenty six  
            RETURN False  
        END IF  
        RETURN True  
    END FUNCTION  
END CLASS
