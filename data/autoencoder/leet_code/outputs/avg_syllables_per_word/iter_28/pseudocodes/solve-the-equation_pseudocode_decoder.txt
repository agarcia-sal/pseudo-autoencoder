CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO the result of replacing each addition sign in side with a separator followed by addition sign then replacing each subtraction sign in side with a separator followed by subtraction sign then splitting the resulting string by the separator  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token is an empty string  
                    CONTINUE to next iteration  
                END IF  
                IF token contains the character x  
                    IF token EQUALS the single character x OR token EQUALS the plus sign followed by x  
                        INCREMENT x_count BY one  
                    ELSE IF token EQUALS the minus sign followed by x  
                        DECREMENT x_count BY one  
                    ELSE  
                        INCREMENT x_count BY the integer value represented by token after removing the character x  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY the integer value represented by token  
                END IF  
            END FOR  
            RETURN x_count AND num_sum  
        END FUNCTION  

        SET left_side AND right_side TO the two parts of equation separated by the equals sign  
        SET left_x AND left_num TO the values returned from calling parse_side on left_side  
        SET right_x AND right_num TO the values returned from calling parse_side on right_side  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN the string Infinite solutions  
            ELSE  
                RETURN the string No solution  
            END IF  
        ELSE  
            RETURN the string x EQUALS the integer division of total_num BY total_x  
        END IF  
    END FUNCTION  
END CLASS
