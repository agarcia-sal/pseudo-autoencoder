CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        SET q TO a new deque CONTAINING a list CONTAINING zero  
        SET ans TO an empty list  
        WHILE the LENGTH OF q IS GREATER THAN zero  
            SET path TO the FIRST element REMOVED FROM q  
            SET u TO the element at the LAST position OF path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            FOR each v IN the element at position u OF graph  
                APPEND a new list CREATED BY concatenating path AND a list CONTAINING v TO q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
