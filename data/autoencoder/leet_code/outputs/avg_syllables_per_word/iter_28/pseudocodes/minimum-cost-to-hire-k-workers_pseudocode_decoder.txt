CLASS Solution  
    FUNCTION mincostToHireWorkers(quality parameter AS List of integers, wage parameter AS List of integers, k parameter AS integer) RETURNS float  
        SET workers TO sorted list of pairs where each pair consists of wage element at current position DIVIDED BY quality element at the same position AND the quality element at that position  
        SET max_heap TO empty list  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each pair in workers  
            LET ratio TO first element of the pair  
            LET q TO second element of the pair  
            PUSH the negative of q TO max_heap  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap GREATER THAN k THEN  
                SET popped_value TO HEAP POP the max_heap  
                INCREMENT total_quality BY popped_value  
            END IF  
            IF the LENGTH OF max_heap EQUALS k THEN  
                SET current_cost TO total_quality MULTIPLIED BY ratio  
                IF current_cost LESS THAN min_cost THEN  
                    SET min_cost TO current_cost  
                END IF  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
