CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET sign TO the plus character  
            SET num TO zero  
            
            WHILE the LENGTH OF s IS GREATER THAN zero  
                SET char TO the element at position zero of s  
                REMOVE the element at position zero FROM s  
                
                IF the character char is a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                
                IF the character char EQUALS the left parenthesis character  
                    SET num TO the result of CALLING helper WITH s  
                END IF  
                
                IF the character char is NOT a digit AND the character char is NOT a whitespace OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS the plus character  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS the minus character  
                        APPEND the negative value of num TO stack  
                    ELSE IF sign EQUALS the multiplication character  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS the division character  
                        SET the element at the last position of stack TO the integer value of the element at the last position of stack DIVIDED BY num  
                    END IF  
                    
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                
                IF the character char EQUALS the right parenthesis character  
                    BREAK  
                END IF  
            END WHILE  
            
            RETURN the sum of all elements in stack  
        END FUNCTION  
        
        RETURN the result of CALLING helper WITH the list of characters in s  
    END FUNCTION  
END CLASS
