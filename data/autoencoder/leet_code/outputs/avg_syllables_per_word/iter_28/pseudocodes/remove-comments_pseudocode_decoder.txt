CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO false  
        SET result TO empty list  
        SET current_line TO empty list  
        
        FOR each line IN source  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF line  
                IF in_block_comment EQUALS true  
                    IF the element at position i OF line EQUALS the element at position i PLUS one OF line EQUALS the character Asterisk FOLLOWED BY the character Forward Slash  
                        SET in_block_comment TO false  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF the element at position i OF line EQUALS the element at position i PLUS one OF line EQUALS the character Forward Slash FOLLOWED BY the character Forward Slash  
                        BREAK the WHILE loop  
                    ELSE IF the element at position i OF line EQUALS the element at position i PLUS one OF line EQUALS the character Forward Slash FOLLOWED BY the character Asterisk  
                        SET in_block_comment TO true  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND the element at position i OF line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  
            
            IF the LENGTH OF current_line GREATER THAN zero AND in_block_comment EQUALS false  
                APPEND the concatenation of all elements in current_line TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
