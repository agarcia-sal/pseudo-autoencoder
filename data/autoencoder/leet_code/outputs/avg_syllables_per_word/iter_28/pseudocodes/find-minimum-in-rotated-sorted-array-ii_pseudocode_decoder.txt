CLASS Solution  
    FUNCTION findMin(nums)  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        
        WHILE left LESS THAN right  
            SET mid TO the INTEGER PART OF the SUM of left PLUS right DIVIDED BY two  
            
            IF element at position mid of nums GREATER THAN element at position right of nums  
                SET left TO mid PLUS one  
            ELSE IF element at position mid of nums LESS THAN element at position right of nums  
                SET right TO mid  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        
        RETURN element at position left of nums  
    END FUNCTION  
END CLASS
