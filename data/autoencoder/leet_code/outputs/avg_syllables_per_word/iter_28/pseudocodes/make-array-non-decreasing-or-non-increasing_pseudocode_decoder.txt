CLASS Solution  
    FUNCTION convertArray(nums)  
        FUNCTION min_operations_to_non_decreasing(nums)  
            SET max_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF max_heap IS NOT empty AND the NEGATION OF element at position zero of max_heap IS GREATER THAN num  
                    INCREMENT operations BY the NEGATION OF REMOVE the element with the smallest value from max_heap MINUS num  
                    INSERT the NEGATION OF num INTO max_heap maintaining the heap property  
                END IF  
                INSERT the NEGATION OF num INTO max_heap maintaining the heap property  
            END FOR  
            RETURN operations  
        END FUNCTION  
        
        FUNCTION min_operations_to_non_increasing(nums)  
            SET min_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF min_heap IS NOT empty AND element at position zero of min_heap IS LESS THAN num  
                    INCREMENT operations BY num MINUS REMOVE the element with the smallest value from min_heap  
                    INSERT num INTO min_heap maintaining the heap property  
                END IF  
                INSERT num INTO min_heap maintaining the heap property  
            END FOR  
            RETURN operations  
        END FUNCTION  
        
        RETURN the MINIMUM of min_operations_to_non_decreasing called with nums AND min_operations_to_non_increasing called with nums  
    END FUNCTION  
END CLASS
