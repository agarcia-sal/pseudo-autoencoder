CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        SET graph TO an empty mapping from each integer from one to n inclusive TO an empty set  
        
        FOR each pair u v IN edges  
            ADD v TO the set at key u in graph  
            ADD u TO the set at key v in graph  
        END FOR  
        
        SET min_degree TO positive infinity  
        SET trio_found TO false  
        
        FOR i FROM one TO n inclusive  
            FOR j FROM the value after i TO n inclusive  
                FOR k FROM the value after j TO n inclusive  
                    IF the set at key i in graph CONTAINS j AND the set at key i in graph CONTAINS k AND the set at key j in graph CONTAINS k  
                        SET trio_found TO true  
                        SET trio_degree TO the size of the set at key i in graph PLUS the size of the set at key j in graph PLUS the size of the set at key k in graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        IF trio_found IS true  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
