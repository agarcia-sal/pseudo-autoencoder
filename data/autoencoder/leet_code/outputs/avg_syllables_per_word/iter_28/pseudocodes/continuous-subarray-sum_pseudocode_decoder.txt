CLASS Solution  
    FUNCTION checkSubarraySum(nums k)  
        SET remainder_map TO a dictionary with key zero assigned to value negative one  
        SET current_sum TO zero  
        FOR each index i AND element num IN the enumeration of nums  
            INCREMENT current_sum BY num  
            IF k NOT EQUALS zero THEN  
                SET current_sum TO the remainder of current_sum DIVIDED BY k  
            END IF  
            IF current_sum EXISTS AS a key IN remainder_map THEN  
                IF the difference between i AND the value at key current_sum IN remainder_map IS GREATER THAN one THEN  
                    RETURN True  
                END IF  
            ELSE  
                ASSIGN the value i TO the key current_sum IN remainder_map  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
