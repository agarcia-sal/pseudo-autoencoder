CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty THEN  
        RETURN none  
    END IF  
    SET head TO new instance of ListNode with the element at position one of values  
    SET p TO head  
    FOR each val IN values from position two to the end  
        SET node TO new instance of ListNode with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none THEN  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none THEN  
        RETURN false  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND is_same_list(the next element of p1, the next element of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList(self, head)  
        IF head EQUALS none OR the next element of head EQUALS none THEN  
            RETURN head  
        END IF  
      
        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO the next element of head  
            WHILE fast NOT EQUALS none AND the next element of fast NOT EQUALS none  
                SET slow TO the next element of slow  
                SET fast TO the element at position two steps after fast  
            END WHILE  
            SET mid TO the next element of slow  
            SET the next element of slow TO none  
            RETURN head and mid  
        END FUNCTION  
      
        FUNCTION merge(l1, l2)  
            SET dummy TO new instance of ListNode with default values  
            SET current TO dummy  
            WHILE l1 NOT EQUALS none AND l2 NOT EQUALS none  
                IF the value of l1 LESS THAN the value of l2 THEN  
                    SET the next element of current TO l1  
                    SET l1 TO the next element of l1  
                ELSE  
                    SET the next element of current TO l2  
                    SET l2 TO the next element of l2  
                END IF  
                SET current TO the next element of current  
            END WHILE  
            IF l1 NOT EQUALS none THEN  
                SET the next element of current TO l1  
            ELSE  
                SET the next element of current TO l2  
            END IF  
            RETURN the next element of dummy  
        END FUNCTION  
      
        SET left and right TO split(head)  
        SET left TO self.sortList(left)  
        SET right TO self.sortList(right)  
      
        RETURN merge(left, right)  
    END FUNCTION  
END CLASS
