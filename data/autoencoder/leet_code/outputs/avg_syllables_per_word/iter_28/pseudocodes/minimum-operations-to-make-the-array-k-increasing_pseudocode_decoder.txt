CLASS Solution  
    FUNCTION kIncreasing(arr parameter ONE, k parameter TWO)  
        FUNCTION longest_increasing_subsequence(subarr parameter ONE)  
            SET lis TO empty list  
            FOR each element named num IN subarr  
                IF lis is empty OR num GREATER THAN OR EQUAL TO element at the last position of lis THEN  
                    APPEND num TO lis  
                ELSE  
                    SET pos TO the position from the bisect right function where num would be inserted into lis  
                    SET element at position pos of lis TO num  
                END IF  
            END FOR  
            RETURN the LENGTH OF lis  
        END FUNCTION  

        SET total_operations TO zero  
        SET n TO the LENGTH OF arr  

        FOR start FROM zero TO k MINUS one  
            SET subarr TO empty list  
            FOR index i FROM start TO LESS THAN n STEP k  
                APPEND element at position i of arr TO subarr  
            END FOR  

            SET lis_length TO longest_increasing_subsequence(subarr)  
            SET total_operations TO total_operations PLUS the LENGTH OF subarr MINUS lis_length  
        END FOR  

        RETURN total_operations  
    END FUNCTION  
END CLASS
