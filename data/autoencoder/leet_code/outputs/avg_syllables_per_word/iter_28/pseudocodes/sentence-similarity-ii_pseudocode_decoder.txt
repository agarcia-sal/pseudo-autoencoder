CLASS Solution  
    FUNCTION areSentencesSimilarTwo(sentence1 sentence1 AS List of string sentence2 sentence2 AS List of string similarPairs similarPairs AS List of List of string) RETURNS boolean  
        IF the LENGTH OF sentence1 NOT EQUALS the LENGTH OF sentence2  
            RETURN False  
        END IF  

        SET parent TO an empty dictionary  

        FUNCTION find(x x) RETURNS the representative element  
            IF the element referenced by x in parent NOT EQUALS x  
                SET the element referenced by x in parent TO the result of find with the element referenced by x in parent  
            END IF  
            RETURN the element referenced by x in parent  
        END FUNCTION  

        FUNCTION union(x x y y)  
            SET rootX TO the result of find with x  
            SET rootY TO the result of find with y  
            IF rootX NOT EQUALS rootY  
                SET the element referenced by rootY in parent TO rootX  
            END IF  
        END FUNCTION  

        FOR each pair IN similarPairs  
            FOR each word IN pair  
                IF word NOT IN the keys of parent  
                    SET the element referenced by word in parent TO word  
                END IF  
            END FOR  
        END FOR  

        FOR each x y IN similarPairs  
            CALL union with x and y  
        END FOR  

        FOR each w1 w2 IN elements of sentence1 and sentence2 simultaneously  
            IF w1 NOT IN the keys of parent  
                SET the element referenced by w1 in parent TO w1  
            END IF  
            IF w2 NOT IN the keys of parent  
                SET the element referenced by w2 in parent TO w2  
            END IF  
            IF the result of find with w1 NOT EQUALS the result of find with w2  
                RETURN False  
            END IF  
        END FOR  

        RETURN True  
    END FUNCTION  
END CLASS
