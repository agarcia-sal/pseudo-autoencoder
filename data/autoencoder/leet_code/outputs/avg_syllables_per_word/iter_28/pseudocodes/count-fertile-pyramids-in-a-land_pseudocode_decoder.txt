CLASS Solution  
    FUNCTION countPyramids(grid)  
        FUNCTION count_pyramids_from_top(grid)  
            IF grid IS empty OR the first element of grid IS empty  
                RETURN zero  
            END IF  
            SET m TO the LENGTH OF grid  
            SET n TO the LENGTH OF the first element of grid  
            SET dp TO a new two dimensional list of zeros with m rows and n columns  
            SET count TO zero  
            FOR index i FROM zero TO m MINUS one  
                FOR index j FROM zero TO n MINUS one  
                    IF the element at position i of grid AND position j OF that element EQUALS one  
                        IF i EQUALS zero OR j EQUALS zero OR j EQUALS n MINUS one  
                            SET the element at position i of dp AND position j OF that element TO one  
                        ELSE  
                            SET the element at position i of dp AND position j OF that element TO THE MINIMUM OF the element at position i MINUS one of dp AND position j MINUS one OF that element AND the element at position i MINUS one of dp AND position j OF that element AND the element at position i MINUS one of dp AND position j PLUS one OF that element PLUS one  
                        END IF  
                        IF the element at position i of dp AND position j OF that element IS GREATER THAN one  
                            INCREMENT count BY the element at position i of dp AND position j OF that element MINUS one  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  

        FUNCTION count_pyramids_from_bottom(grid)  
            IF grid IS empty OR the first element of grid IS empty  
                RETURN zero  
            END IF  
            SET m TO the LENGTH OF grid  
            SET n TO the LENGTH OF the first element of grid  
            SET dp TO a new two dimensional list of zeros with m rows and n columns  
            SET count TO zero  
            FOR index i FROM m MINUS one DOWN TO zero  
                FOR index j FROM zero TO n MINUS one  
                    IF the element at position i of grid AND position j OF that element EQUALS one  
                        IF i EQUALS m MINUS one OR j EQUALS zero OR j EQUALS n MINUS one  
                            SET the element at position i of dp AND position j OF that element TO one  
                        ELSE  
                            SET the element at position i of dp AND position j OF that element TO THE MINIMUM OF the element at position i PLUS one of dp AND position j MINUS one OF that element AND the element at position i PLUS one of dp AND position j OF that element AND the element at position i PLUS one of dp AND position j PLUS one OF that element PLUS one  
                        END IF  
                        IF the element at position i of dp AND position j OF that element IS GREATER THAN one  
                            INCREMENT count BY the element at position i of dp AND position j OF that element MINUS one  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  

        RETURN THE RESULT OF count_pyramids_from_top WITH grid PLUS THE RESULT OF count_pyramids_from_bottom WITH grid  
    END FUNCTION  
END CLASS
