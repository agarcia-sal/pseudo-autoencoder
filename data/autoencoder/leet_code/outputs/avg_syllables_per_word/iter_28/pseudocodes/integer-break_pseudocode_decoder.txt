CLASS Solution  
    FUNCTION integerBreak WITH PARAMETER n  
        SET dp TO list of zero values with the length equal to n PLUS one  
        SET element at position one of dp TO one  
        FOR i FROM two TO n  
            SET max_product TO zero  
            FOR j FROM one TO i MINUS one  
                SET product_one TO j MULTIPLIED BY element at position i MINUS j of dp  
                SET product_two TO j MULTIPLIED BY i MINUS j  
                IF max_product LESS THAN product_one AND max_product LESS THAN product_two  
                    SET max_product TO the greater value among max_product product_one AND product_two  
                ELSE IF max_product LESS THAN product_one  
                    SET max_product TO product_one  
                ELSE IF max_product LESS THAN product_two  
                    SET max_product TO product_two  
                END IF  
            END FOR  
            SET element at position i of dp TO max_product  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
