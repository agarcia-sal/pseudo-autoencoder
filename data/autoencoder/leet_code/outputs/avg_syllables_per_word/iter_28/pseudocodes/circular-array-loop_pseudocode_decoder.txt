CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1 val2)  
            IF val1 GREATER THAN zero AND val2 GREATER THAN zero  
                RETURN True  
            ELSE IF val1 LESS THAN zero AND val2 LESS THAN zero  
                RETURN True  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  
        
        SET n TO the LENGTH OF nums  
        
        FOR i FROM zero TO n MINUS one  
            SET slow TO i  
            SET fast TO i  
            
            IF element at position i of nums EQUALS zero  
                CONTINUE  
            END IF  
            
            WHILE True  
                SET slow TO the REMAINDER OF the ADDITION OF slow AND element at position slow of nums DIVIDED BY n  
                IF slow LESS THAN zero  
                    INCREMENT slow BY n  
                END IF  
                
                SET fast TO the REMAINDER OF the ADDITION OF fast AND element at position fast of nums DIVIDED BY n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  
                
                SET fast TO the REMAINDER OF the ADDITION OF fast AND element at position fast of nums DIVIDED BY n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  
                
                IF NOT sameDirection(element at position slow of nums element at position i of nums) OR NOT sameDirection(element at position fast of nums element at position i of nums)  
                    BREAK  
                END IF  
                
                IF slow EQUALS fast  
                    IF slow EQUALS the REMAINDER OF the ADDITION OF slow AND element at position slow of nums DIVIDED BY n  
                        BREAK  
                    END IF  
                    RETURN True  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
