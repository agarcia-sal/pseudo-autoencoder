CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO A new TreeNode initialized WITH the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element REMOVED FROM the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.left TO A new TreeNode initialized WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node.right TO A new TreeNode initialized WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree WITH the left attribute of p AND the left attribute of q AND the result of is_same_tree WITH the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers(self, root)  
        FUNCTION dfs(node, current_number)  
            IF node EQUALS none  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS the val attribute of node  
            IF the left attribute of node EQUALS none AND the right attribute of node EQUALS none  
                RETURN current_number  
            END IF  
            SET left_sum TO dfs WITH the left attribute of node AND current_number  
            SET right_sum TO dfs WITH the right attribute of node AND current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN dfs WITH root AND zero  
    END FUNCTION  
END CLASS
