CLASS Solution  
    FUNCTION countComponents WITH PARAMETERS n edges  
        SET graph TO an empty mapping from each index from zero TO n MINUS one TO an empty list  
        FOR each pair a b IN edges  
            APPEND element b TO the list at key a in graph  
            APPEND element a TO the list at key b in graph  
        END FOR  
        SET visited TO an empty set  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list at key node in graph  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        SET components TO zero  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                CALL dfs WITH node  
                INCREMENT components BY one  
            END IF  
        END FOR  
        RETURN components  
    END FUNCTION  
END CLASS
