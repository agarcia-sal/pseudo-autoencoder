CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        SET secret_count TO a mapping from character to integer with default zero  
        SET guess_count TO a mapping from character to integer with default zero  
        FOR each pair of elements s and g taken at corresponding positions from secret and guess  
            IF element s EQUALS element g THEN  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value associated with key s in secret_count BY one  
                INCREMENT the value associated with key g in guess_count BY one  
            END IF  
        END FOR  
        FOR each digit IN the keys of secret_count  
            INCREMENT cows BY the lesser between the value associated with digit in secret_count AND the value associated with digit in guess_count  
        END FOR  
        RETURN a string formed by the value of bulls concatenated with letter A concatenated with the value of cows concatenated with letter B  
    END FUNCTION  
END CLASS
