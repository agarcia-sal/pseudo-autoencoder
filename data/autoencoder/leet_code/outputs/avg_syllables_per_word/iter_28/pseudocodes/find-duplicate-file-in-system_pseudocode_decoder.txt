CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO a mapping from content to list of file paths with default empty list  
        FOR each path IN paths  
            SET parts TO the list of words in path separated by whitespace  
            SET directory TO the first element of parts  
            FOR each file_info IN the elements of parts starting from the second element to the end  
                SET file_name TO the substring of file_info before the position of the first opening parenthesis  
                SET content TO the substring of file_info from the position after the first opening parenthesis TO the position before the last character of file_info  
                APPEND the concatenation of directory, a slash, and file_name TO the list in content_to_paths at key content  
            END FOR  
        END FOR  
        RETURN the list of lists of file paths from content_to_paths values WHERE the LENGTH of each list is greater than one  
    END FUNCTION  
END CLASS
