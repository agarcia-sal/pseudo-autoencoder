CLASS Solution  
    FUNCTION getMinDistSum(positions)  
        SET n TO the LENGTH OF positions  
        SET x TO zero  
        SET y TO zero  
        FOR each pair of values named x1 and y1 IN positions  
            INCREMENT x BY x1  
            INCREMENT y BY y1  
        END FOR  
        SET x TO x DIVIDED BY n  
        SET y TO y DIVIDED BY n  
        SET decay TO point nine nine nine  
        SET eps TO one TIMES ten TO THE MINUS six  
        SET alpha TO point five  
        WHILE TRUE  
            SET grad_x TO zero  
            SET grad_y TO zero  
            SET dist TO zero  
            FOR each pair of values named x1 and y1 IN positions  
                SET a TO x MINUS x1  
                SET b TO y MINUS y1  
                SET c TO the square root of a MULTIPLIED BY a PLUS b MULTIPLIED BY b  
                INCREMENT grad_x BY a DIVIDED BY the quantity of c PLUS one TIMES ten TO THE MINUS eight  
                INCREMENT grad_y BY b DIVIDED BY the quantity of c PLUS one Times ten TO THE MINUS eight  
                INCREMENT dist BY c  
            END FOR  
            SET dx TO grad_x MULTIPLIED BY alpha  
            SET dy TO grad_y MULTIPLIED BY alpha  
            DECREMENT x BY dx  
            DECREMENT y BY dy  
            SET alpha TO alpha MULTIPLIED BY decay  
            IF the absolute value of dx LESS THAN OR EQUAL TO eps AND the absolute value of dy LESS THAN OR EQUAL TO eps  
                RETURN dist  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
