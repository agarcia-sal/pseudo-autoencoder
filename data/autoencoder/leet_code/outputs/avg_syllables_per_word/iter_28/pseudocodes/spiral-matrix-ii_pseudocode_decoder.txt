CLASS Solution  
    FUNCTION generateMatrix(n)  
        SET matrix TO a two dimensional collection of size n BY n filled with zero  
        SET top TO zero  
        SET bottom TO n MINUS one  
        SET left TO zero  
        SET right TO n MINUS one  
        SET num TO one  
        WHILE top LESS THAN OR EQUAL TO bottom AND left LESS THAN OR EQUAL TO right  
            FOR index i FROM left TO right  
                SET element at position top of matrix at position i TO num  
                INCREMENT num BY one  
            END FOR  
            INCREMENT top BY one  
            FOR index i FROM top TO bottom  
                SET element at position i of matrix at position right TO num  
                INCREMENT num BY one  
            END FOR  
            DECREMENT right BY one  
            IF top LESS THAN OR EQUAL TO bottom  
                FOR index i FROM right DOWN TO left  
                    SET element at position bottom of matrix at position i TO num  
                    INCREMENT num BY one  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            IF left LESS THAN OR EQUAL TO right  
                FOR index i FROM bottom DOWN TO top  
                    SET element at position i of matrix at position left TO num  
                    INCREMENT num BY one  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        RETURN matrix  
    END FUNCTION  
END CLASS
