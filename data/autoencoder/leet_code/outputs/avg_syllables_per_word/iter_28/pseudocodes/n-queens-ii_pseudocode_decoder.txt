CLASS Solution  
    FUNCTION totalNQueens WITH PARAMETER n  
        FUNCTION is_not_under_attack WITH PARAMETERS row col  
            RETURN NOT value at position col of cols PLUS value at position of row MINUS col of hills PLUS value at position of row PLUS col of dales EQUALS zero  
        END FUNCTION  
        
        FUNCTION place_queen WITH PARAMETERS row col  
            SET value at position col of cols TO one  
            SET value at position of row MINUS col of hills TO one  
            SET value at position of row PLUS col of dales TO one  
        END FUNCTION  
        
        FUNCTION remove_queen WITH PARAMETERS row col  
            SET value at position col of cols TO zero  
            SET value at position of row MINUS col of hills TO zero  
            SET value at position of row PLUS col of dales TO zero  
        END FUNCTION  
        
        FUNCTION backtrack WITH PARAMETERS row count  
            FOR col FROM zero TO n MINUS one  
                IF is_not_under_attack WITH ARGUMENTS row col EQUALS true  
                    CALL place_queen WITH ARGUMENTS row col  
                    IF row PLUS one EQUALS n  
                        INCREMENT count BY one  
                    ELSE  
                        SET count TO backtrack WITH ARGUMENTS row PLUS one count  
                    END IF  
                    CALL remove_queen WITH ARGUMENTS row col  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET cols TO list of zero repeated n times  
        SET hills TO list of zero repeated two MULTIPLIED BY n MINUS one times  
        SET dales TO list of zero repeated two MULTIPLIED BY n MINUS one times  
        
        RETURN backtrack WITH ARGUMENTS zero zero  
    END FUNCTION  
END CLASS
