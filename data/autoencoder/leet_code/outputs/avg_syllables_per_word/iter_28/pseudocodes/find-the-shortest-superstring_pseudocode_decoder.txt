CLASS Solution  
    FUNCTION shortestSuperstring WITH PARAMETER words  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap WITH PARAMETERS i AND j  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i of words AND the LENGTH OF element at position j of words DOWN TO one  
                IF substring consisting of the last k characters of element at position i of words EQUALS substring consisting of the first k characters of element at position j of words THEN  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp WITH PARAMETERS mask AND i  
            IF mask EQUALS the value of one SHIFTED LEFT BY n MINUS one THEN  
                RETURN zero AND an empty string  
            END IF  
            
            SET min_len TO positive infinity  
            SET best_path TO an empty string  
            FOR j FROM zero TO n MINUS one  
                IF the bitwise AND of mask AND the value of one SHIFTED LEFT BY j EQUALS zero THEN  
                    SET length_and_path TO the RESULT OF dp WITH PARAMETERS bitwise OR of mask AND the value of one SHIFTED LEFT BY j AND j  
                    SET length TO the first element of length_and_path  
                    SET path TO the second element of length_and_path  
                    SET length TO length PLUS the LENGTH OF element at position j of words MINUS the RESULT OF overlap WITH PARAMETERS i AND j  
                    IF length LESS THAN min_len THEN  
                        SET min_len TO length  
                        SET best_path TO substring consisting of characters from position RESULT OF overlap WITH PARAMETERS i AND j PLUS one TO the end of element at position j of words CONCATENATED WITH path  
                    END IF  
                END IF  
            END FOR  
            RETURN min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO positive infinity  
        SET shortest_path TO an empty string  
        FOR i FROM zero TO n MINUS one  
            SET length_and_path TO the RESULT OF dp WITH PARAMETERS the value of one SHIFTED LEFT BY i AND i  
            SET length TO the first element of length_and_path  
            SET path TO the second element of length_and_path  
            SET length TO length PLUS the LENGTH OF element at position i of words  
            IF length LESS THAN min_len THEN  
                SET min_len TO length  
                SET shortest_path TO element at position i of words CONCATENATED WITH path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
