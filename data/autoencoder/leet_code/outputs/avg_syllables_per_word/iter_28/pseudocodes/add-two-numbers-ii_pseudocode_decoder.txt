CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new instance of ListNode with val equal to element at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO the last element of values  
        SET node TO new instance of ListNode with val equal to val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the predicate that element val of p1 EQUALS element val of p2 AND function call is_same_list with parameters element next of p1 AND element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(self, l1, l2)  
        FUNCTION pushIntoStack(node)  
            SET stack TO empty list  
            WHILE node IS NOT none  
                APPEND element val of node TO stack  
                SET node TO element next of node  
            END WHILE  
            RETURN stack  
        END FUNCTION  

        SET stack1 TO function call pushIntoStack with parameter l1  
        SET stack2 TO function call pushIntoStack with parameter l2  
        SET carry TO zero  
        SET result TO none  

        WHILE logical predicate that stack1 IS NOT empty OR stack2 IS NOT empty OR carry GREATER THAN zero holds  
            IF stack1 IS NOT empty  
                REMOVE the last element FROM stack1 AND SET val1 TO that element  
            ELSE  
                SET val1 TO zero  
            END IF  

            IF stack2 IS NOT empty  
                REMOVE the last element FROM stack2 AND SET val2 TO that element  
            ELSE  
                SET val2 TO zero  
            END IF  

            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  
            SET new_node TO new instance of ListNode with val equal to digit  
            SET element next of new_node TO result  
            SET result TO new_node  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
