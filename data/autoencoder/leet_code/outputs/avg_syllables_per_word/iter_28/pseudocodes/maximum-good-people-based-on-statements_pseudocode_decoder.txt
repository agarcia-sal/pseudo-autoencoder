CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO the VALUE ONE LEFT SHIFTED BY n MINUS one  
            SET good TO a list of n elements each SET TO false  
            SET bad TO a list of n elements each SET TO false  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND of i AND the VALUE ONE LEFT SHIFTED BY j EQUALS the VALUE ONE LEFT SHIFTED BY j  
                    SET element at position j of good TO true  
                ELSE  
                    SET element at position j of bad TO true  
                END IF  
            END FOR  
            SET valid TO true  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of good EQUALS true  
                    FOR k FROM zero TO n MINUS one  
                        IF (element at position k of element at position j of statements EQUALS zero AND element at position k of good EQUALS true) OR (element at position k of element at position j of statements EQUALS one AND element at position k of bad EQUALS true)  
                            SET valid TO false  
                            BREAK from the inner FOR loop  
                        END IF  
                    END FOR  
                    IF valid EQUALS false  
                        BREAK from the FOR loop over j  
                    END IF  
                END IF  
            END FOR  
            IF valid EQUALS true  
                SET count_good TO zero  
                FOR each person IN good  
                    IF person EQUALS true  
                        INCREMENT count_good BY one  
                    END IF  
                END FOR  
                IF count_good GREATER THAN max_good  
                    SET max_good TO count_good  
                END IF  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
