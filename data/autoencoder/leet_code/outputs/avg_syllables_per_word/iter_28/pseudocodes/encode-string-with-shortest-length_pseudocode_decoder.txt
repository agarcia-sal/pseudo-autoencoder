CLASS Solution  
    FUNCTION encode(s)  
        FUNCTION get_encoded_length(count, substring)  
            RETURN the LENGTH OF the STRING representation of count PLUS the LENGTH OF substring PLUS two  
        END FUNCTION  

        FUNCTION dp(i, j)  
            SET substring TO the substring from position i TO position j of s  
            SET n TO j MINUS i PLUS one  
            IF n LESS THAN five THEN  
                RETURN substring  
            END IF  

            SET shortest TO substring  
            FOR k FROM one TO the INTEGER PART OF n DIVIDED BY two PLUS one  
                IF n MODULO k EQUALS zero THEN  
                    SET repeat_count TO n DIVIDED BY k  
                    IF substring EQUALS the substring from position zero TO position k MINUS one of substring MULTIPLIED BY repeat_count THEN  
                        SET encoded_candidate TO the STRING representation of repeat_count FOLLOWED BY the LEFT SQUARE BRACKET CHARACTER FOLLOWED BY dp(i, i PLUS k MINUS one) FOLLOWED BY the RIGHT SQUARE BRACKET CHARACTER  
                        IF the LENGTH OF encoded_candidate LESS THAN the LENGTH OF shortest THEN  
                            SET shortest TO encoded_candidate  
                        END IF  
                    END IF  
                END IF  
            END FOR  

            FOR k FROM i TO j MINUS one  
                SET left_encoded TO dp(i, k)  
                SET right_encoded TO dp(k PLUS one, j)  
                SET combined TO left_encoded FOLLOWED BY right_encoded  
                IF the LENGTH OF combined LESS THAN the LENGTH OF shortest THEN  
                    SET shortest TO combined  
                END IF  
            END FOR  

            RETURN shortest  
        END FUNCTION  

        RETURN dp(zero, the LENGTH OF s MINUS one)  
    END FUNCTION  
END CLASS
