CLASS Solution  
    FUNCTION threeSumClosest(nums target)  
        SORT the list nums in ascending order  
        SET closest_sum TO infinity  
        FOR index i FROM zero TO the LENGTH OF nums MINUS two  
            SET left TO i PLUS one  
            SET right TO the LENGTH OF nums MINUS one  
            WHILE left LESS THAN right  
                SET current_sum TO the element at position i of nums PLUS the element at position left of nums PLUS the element at position right of nums  
                IF the absolute value of current_sum MINUS target LESS THAN the absolute value of closest_sum MINUS target  
                    SET closest_sum TO current_sum  
                END IF  
                IF current_sum LESS THAN target  
                    INCREMENT left BY one  
                ELSE IF current_sum GREATER THAN target  
                    DECREMENT right BY one  
                ELSE  
                    RETURN current_sum  
                END IF  
            END WHILE  
        END FOR  
        RETURN closest_sum  
    END FUNCTION  
END CLASS
