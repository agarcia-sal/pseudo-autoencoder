CLASS TreeNode  
    FUNCTION __init__(val WITH DEFAULT zero, left WITH DEFAULT none, right WITH DEFAULT none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR THE LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO new TreeNode WITH val EQUALS element at position zero of values  
    SET i TO one  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node dot left TO new TreeNode WITH val EQUALS element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node dot right TO new TreeNode WITH val EQUALS element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH ARGUMENTS p dot left AND q dot left AND is_same_tree WITH ARGUMENTS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum(root)  
        FUNCTION helper(node)  
            IF node EQUALS none  
                RETURN negative infinity, negative infinity  
            END IF  
            SET left_max, left_gain TO helper WITH ARGUMENT node dot left  
            SET right_max, right_gain TO helper WITH ARGUMENT node dot right  
            SET current_max TO node dot val PLUS the GREATER OF left_gain AND zero PLUS the GREATER OF right_gain AND zero  
            SET current_gain TO node dot val PLUS the GREATER OF left_gain, right_gain, AND zero  
            SET global_max TO the GREATER OF left_max, right_max, AND current_max  
            RETURN global_max, current_gain  
        END FUNCTION  
        RETURN the FIRST element of helper WITH ARGUMENT root  
    END FUNCTION  
END CLASS
