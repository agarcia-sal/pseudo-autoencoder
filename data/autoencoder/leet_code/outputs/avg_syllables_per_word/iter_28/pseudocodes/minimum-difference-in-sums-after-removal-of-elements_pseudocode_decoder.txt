CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
        
        SET min_heap TO empty list  
        SET min_sums TO a list of zeroes of size the DOUBLE of n PLUS one  
        
        FOR i FROM zero TO the DOUBLE of n MINUS one  
            PUSH the NEGATIVE of element at position i of nums INTO min_heap as heap  
            IF the LENGTH OF min_heap GREATER THAN n THEN  
                POP the smallest element FROM min_heap as heap  
            END IF  
            IF the LENGTH OF min_heap EQUALS n THEN  
                SET element at position i of min_sums TO the NEGATIVE of the SUM of all elements in min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO empty list  
        SET max_sums TO a list of zeroes of size the DOUBLE of n PLUS one  
        
        FOR i FROM the TRIPLE of n MINUS one DOWN TO n  
            PUSH element at position i of nums INTO max_heap as heap  
            IF the LENGTH OF max_heap GREATER THAN n THEN  
                POP the smallest element FROM max_heap as heap  
            END IF  
            IF the LENGTH OF max_heap EQUALS n THEN  
                SET element at position i of max_sums TO the SUM of all elements in max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO infinity  
        FOR i FROM n MINUS one TO the DOUBLE of n MINUS one  
            IF min_diff GREATER THAN element at position i of min_sums MINUS element at position i PLUS one of max_sums THEN  
                SET min_diff TO element at position i of min_sums MINUS element at position i PLUS one of max_sums  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
