CLASS Solution  
    FUNCTION findKthNumber(m n k)  
        FUNCTION enough(x)  
            SET count TO zero  
            FOR i FROM one TO m  
                INCREMENT count BY the smaller value BETWEEN the INTEGER RESULT OF x DIVIDED BY i AND n  
            END FOR  
            RETURN WHETHER count IS GREATER THAN OR EQUAL TO k  
        END FUNCTION  
        SET low TO one  
        SET high TO m MULTIPLIED BY n  
        WHILE low IS LESS THAN high  
            SET mid TO the INTEGER RESULT OF low PLUS high DIVIDED BY two  
            IF enough(mid) IS TRUE  
                SET high TO mid  
            ELSE  
                SET low TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
