CLASS Solution  
    FUNCTION isValidSerialization(preorder)  
        SET nodes TO list of elements obtained by splitting preorder string by the word comma  
        SET slots TO one  
        FOR each node IN nodes  
            IF slots EQUALS zero  
                RETURN false  
            END IF  
            DECREMENT slots BY one  
            IF node NOT EQUALS the word hash  
                INCREMENT slots BY two  
            END IF  
        END FOR  
        RETURN the logical condition that slots EQUALS zero  
    END FUNCTION  
END CLASS
