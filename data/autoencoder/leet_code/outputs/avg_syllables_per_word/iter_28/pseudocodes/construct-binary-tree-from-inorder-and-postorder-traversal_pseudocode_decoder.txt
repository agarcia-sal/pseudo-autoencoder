CLASS TreeNode  
    FUNCTION __init__(self, val HAS DEFAULT zero, left HAS DEFAULT None, right HAS DEFAULT None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode instance with value equal to element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater than zero  
        REMOVE and ASSIGN the first element of queue TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            ASSIGN node.left TO a new TreeNode instance with value equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            ASSIGN node.right TO a new TreeNode instance with value equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, inorder, postorder)  
        IF inorder EQUALS empty OR postorder EQUALS empty  
            RETURN None  
        END IF  
        SET root_val TO the last element of postorder removed from postorder  
        SET root TO a new TreeNode instance with value equal to root_val  
        SET root_index TO the position of root_val in inorder  
        ASSIGN root.right TO the result of calling self.buildTree with arguments being the sublist of inorder from position root_index PLUS one TO the end AND postorder  
        ASSIGN root.left TO the result of calling self.buildTree with arguments being the sublist of inorder from position zero TO position root_index MINUS one AND postorder  
        RETURN root  
    END FUNCTION  
END CLASS
