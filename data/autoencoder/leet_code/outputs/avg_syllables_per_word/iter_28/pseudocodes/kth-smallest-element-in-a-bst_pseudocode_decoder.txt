CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS nothing  
        RETURN nothing  
    END IF  
    SET root TO new TreeNode initialized with element at position one of values  
    SET i TO one  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the front of queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS nothing  
            SET node.left TO new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS nothing  
            SET node.right TO new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN true  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN false  
    ELSE IF the val property of p NOT EQUALS the val property of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of calling is_same_tree with the left property of p and the left property of q AND calling is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest with parameters self root k  
        SET stack TO new empty list  
        SET current TO root  
        SET count TO zero  
        WHILE the length of stack is greater than zero OR current NOT EQUALS nothing  
            WHILE current NOT EQUALS nothing  
                APPEND current TO stack  
                SET current TO the left property of current  
            END WHILE  
            SET current TO element removed from the end of stack  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN the val property of current  
            END IF  
            SET current TO the right property of current  
        END WHILE  
    END FUNCTION  
END CLASS
