CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO the LENGTH OF text  
        SET substrings TO an EMPTY COLLECTION OF UNIQUE ELEMENTS  
        FOR length FROM one TO n DIVIDED BY two INCLUSIVE  
            FOR start FROM zero TO n MINUS two MULTIPLIED BY length INCLUSIVE  
                SET a TO the SUBSTRING of text FROM position start TO position start PLUS length MINUS one  
                SET b TO the SUBSTRING of text FROM position start PLUS length TO position start PLUS two MULTIPLIED BY length MINUS one  
                IF a EQUALS b  
                    ADD the CONCATENATION of a AND b TO substrings  
                END IF  
            END FOR  
        END FOR  
        RETURN the SIZE OF substrings  
    END FUNCTION  
END CLASS
