CLASS Solution  
    FUNCTION movesToStamp(stamp target)  
        SET m TO the LENGTH OF stamp  
        SET n TO the LENGTH OF target  
        SET indeg TO a list of m repeated the number of times EQUAL TO n MINUS m PLUS one  
        ASSIGN q TO a new empty double ended queue  
        SET g TO a list containing empty lists repeated the number of times EQUAL TO n  
        FOR i FROM zero TO n MINUS m  
            FOR each j AND c IN the enumeration of stamp  
                IF the element at position i PLUS j of target EQUALS c  
                    DECREMENT the element at position i of indeg BY one  
                    IF the element at position i of indeg EQUALS zero  
                        APPEND i TO q  
                    END IF  
                ELSE  
                    APPEND i TO the list at position i PLUS j of g  
                END IF  
            END FOR  
        END FOR  
        SET ans TO an empty list  
        SET vis TO a list of False repeated the number of times EQUAL TO n  
        WHILE q is not empty  
            REMOVE and ASSIGN the element from the left of q TO i  
            APPEND i TO ans  
            FOR j FROM zero TO m MINUS one  
                IF the element at position i PLUS j of vis EQUALS False  
                    SET the element at position i PLUS j of vis TO True  
                    FOR each k IN the list at position i PLUS j of g  
                        DECREMENT the element at position k of indeg BY one  
                        IF the element at position k of indeg EQUALS zero  
                            APPEND k TO q  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  
        IF all elements in vis EQUAL True  
            RETURN ans in reversed order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
