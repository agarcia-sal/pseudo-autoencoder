CLASS Solution  
    FUNCTION stoneGameIII(stoneValue)  
        SET n TO the LENGTH OF stoneValue  
        CREATE list dp WITH n PLUS one ELEMENTS ALL SET TO zero  
        FOR i FROM n MINUS one TO zero INCLUSIVE DECREMENTING BY one  
            SET dp at position i TO element at position i of stoneValue MINUS dp at position i PLUS one  
            IF i PLUS two LESS THAN OR EQUAL TO n THEN  
                SET dp at position i TO THE MAXIMUM VALUE BETWEEN dp at position i AND element at position i of stoneValue PLUS element at position i PLUS one of stoneValue MINUS dp at position i PLUS two  
            END IF  
            IF i PLUS three LESS THAN OR EQUAL TO n THEN  
                SET dp at position i TO THE MAXIMUM VALUE BETWEEN dp at position i AND element at position i of stoneValue PLUS element at position i PLUS one of stoneValue PLUS element at position i PLUS two of stoneValue MINUS dp at position i PLUS three  
            END IF  
        END FOR  
        IF dp at position zero GREATER THAN zero THEN  
            RETURN Alice  
        ELSE IF dp at position zero LESS THAN zero THEN  
            RETURN Bob  
        ELSE  
            RETURN Tie  
        END IF  
    END FUNCTION  
END CLASS
