CLASS Solution  
    FUNCTION topKFrequent(words k)  
        SET count TO the frequency count of each word in words  
        SET heap TO an empty list  
        FOR each word AND its frequency IN count  
            APPEND a tuple consisting of the negative value of frequency AND the word TO heap  
        END FOR  
        TRANSFORM heap INTO a heap data structure according to its elements  
        SET result TO an empty list  
        FOR each number FROM one TO k  
            REMOVE the smallest element FROM heap  
            APPEND the second element of the removed tuple TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
