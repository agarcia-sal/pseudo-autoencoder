CLASS Solution  
    FUNCTION multiply(num1 string num2 string) RETURNS string  
        IF num1 EQUALS the string representing zero OR num2 EQUALS the string representing zero  
            RETURN the string representing zero  
        END IF  
        
        SET result TO a list of zeros with the count of the length of num1 PLUS the length of num2  
        
        SET num1 TO the string num1 reversed  
        SET num2 TO the string num2 reversed  
        
        FOR index i FROM zero TO the length of num1 MINUS one  
            FOR index j FROM zero TO the length of num2 MINUS one  
                SET mul TO the value of the character at position i of num1 converted to a digit MINUS the value of the character representing zero MULTIPLIED BY the value of the character at position j of num2 converted to a digit MINUS the value of the character representing zero  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS the element at position p1 of result  
                
                SET the element at position p1 of result TO sum_ MODULO ten  
                INCREMENT the element at position p2 of result BY sum_ DIVIDED BY ten  
            END FOR  
        END FOR  
        
        SET result_str TO the concatenation of the string representations of the elements of result reversed  
        RETURN the string result_str with all leading characters representing zero removed  
    END FUNCTION  
END CLASS
