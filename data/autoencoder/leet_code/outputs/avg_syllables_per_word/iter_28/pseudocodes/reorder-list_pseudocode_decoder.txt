CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the val property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the collection values DOES NOT CONTAIN any element  
        RETURN absence of value  
    END IF  
    SET head TO a new instance of ListNode initialized with the first element of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two TO the end  
        SET node TO a new instance of ListNode initialized with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS absence of value AND p2 IS absence of value  
        RETURN truth value true  
    END IF  
    IF p1 IS absence of value OR p2 IS absence of value  
        RETURN truth value false  
    END IF  
    RETURN the logical conjunction of the equality of the val property of p1 AND the val property of p2 AND the result of is_same_list called with the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList WITH PARAMETER head  
        IF head IS absence of value OR the next property of head IS absence of value  
            RETURN  
        END IF  
        SET slow TO head  
        SET fast TO head  
        WHILE fast IS NOT absence of value AND the next property of fast IS NOT absence of value  
            SET slow TO the next property of slow  
            SET fast TO the next property of the next property of fast  
        END WHILE  
        SET prev TO absence of value  
        SET curr TO slow  
        WHILE curr IS NOT absence of value  
            SET next_temp TO the next property of curr  
            SET the next property of curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  
        SET first TO head  
        SET second TO prev  
        WHILE the next property of second IS NOT absence of value  
            SET temp1 TO the next property of first  
            SET temp2 TO the next property of second  
            SET the next property of first TO second  
            SET the next property of second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
