CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each index i AND character char IN the elements and positions of s  
                IF character char EQUALS the character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND the concatenation of the character one PLUS the result of calling dfs with the substring of s from position start PLUS one TO position i MINUS one PLUS the character zero TO the list specials  
                    SET start TO index i PLUS one  
                END IF  
            END FOR  
            SORT the list specials IN reverse lexicographical order  
            RETURN the concatenation of all the elements in the list specials without separators  
        END FUNCTION  
        RETURN the result of calling dfs with s  
    END FUNCTION  
END CLASS
