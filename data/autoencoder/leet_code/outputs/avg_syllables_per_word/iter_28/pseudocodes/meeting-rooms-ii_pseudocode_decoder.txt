CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty list  
            RETURN zero  
        END IF  
        SORT intervals BY the element at position zero of each interval  
        SET min_heap TO empty list  
        FOR each meeting IN intervals  
            IF min_heap NOT EQUALS empty list AND the element at position zero of meeting GREATER THAN OR EQUAL TO the element at position zero of min_heap  
                REMOVE the smallest element FROM min_heap  
            END IF  
            ADD the element at position one of meeting TO min_heap AS smallest element  
        END FOR  
        RETURN the LENGTH OF min_heap  
    END FUNCTION  
END CLASS
