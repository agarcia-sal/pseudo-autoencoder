CLASS Solution  
    FUNCTION kSimilarity(s1, s2)  
        FUNCTION neighbors(s)  
            SET index i TO zero  
            WHILE index i IS LESS THAN the LENGTH OF s AND the element at position index i of s EQUALS the element at position index i of s2  
                INCREMENT index i BY one  
            END WHILE  
            FOR index j FROM index i PLUS one TO the LENGTH OF s MINUS one  
                IF the element at position index j of s EQUALS the element at position index i of s2 AND the element at position index j of s IS NOT EQUAL TO the element at position index j of s2  
                    YIELD the STRING MADE BY concatenating the substring from position zero TO position index i MINUS one of s PLUS the element at position index j of s PLUS the substring from position index i PLUS one TO position index j MINUS one of s PLUS the element at position index i of s PLUS the substring from position index j PLUS one TO the END of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        ASSIGN reference TO a double-ended queue containing a PAIR of s1 AND zero  
        ASSIGN reference TO a set containing s1  
        
        WHILE the double-ended queue IS NOT EMPTY  
            REMOVE AND ASSIGN the FIRST element of the double-ended queue TO variables s AND swaps  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors of s  
                IF neighbor IS NOT IN the set visited  
                    ADD neighbor TO the set visited  
                    APPEND a PAIR of neighbor AND swaps PLUS one TO the double-ended queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
