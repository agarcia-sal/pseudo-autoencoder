CLASS Solution  
    FUNCTION combinationSum2(candidates, target)  
        FUNCTION backtrack(start, target_value, path_list)  
            IF target_value EQUALS zero  
                APPEND path_list TO result  
                RETURN  
            END IF  
            IF target_value LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                IF i GREATER THAN start AND element at position i of candidates EQUALS element at position i MINUS one of candidates  
                    CONTINUE TO next iteration  
                END IF  
                CALL backtrack WITH arguments i PLUS one, target_value MINUS element at position i of candidates, path_list PLUS element at position i of candidates  
            END FOR  
        END FUNCTION  
        SORT candidates IN ASCENDING ORDER  
        SET result TO an empty list  
        CALL backtrack WITH arguments zero, target, an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
