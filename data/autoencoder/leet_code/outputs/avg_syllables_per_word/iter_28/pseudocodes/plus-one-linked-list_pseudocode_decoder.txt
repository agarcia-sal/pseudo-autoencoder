CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self val TO val  
        SET self next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty list THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode with value at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO the last position of values  
        SET node TO new ListNode with val  
        SET p next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the value of element p1 EQUALS the value of element p2 AND is_same_list(next element of p1 next element of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne(self head)  
        FUNCTION reverseList(node)  
            SET prev TO None  
            WHILE node NOT EQUALS None DO  
                SET next_node TO next element of node  
                SET next element of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO reverseList(head)  
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current NOT EQUALS None AND carry EQUALS one DO  
            SET current val TO current val PLUS carry  
            SET carry TO current val DIVIDED BY ten  
            SET current val TO current val MODULO ten  
            
            IF next element of current EQUALS None AND carry EQUALS one THEN  
                SET next element of current TO new ListNode with carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO next element of current  
        END WHILE  
        
        RETURN reverseList(reversed_head)  
    END FUNCTION  
END CLASS
