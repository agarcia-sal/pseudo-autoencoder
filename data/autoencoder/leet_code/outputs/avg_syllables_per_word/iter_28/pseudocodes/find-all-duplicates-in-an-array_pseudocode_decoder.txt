CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            WHILE the element at position i of nums NOT EQUALS the element at position the element at position i of nums MINUS one of nums  
                SET temporary_first TO the element at position i of nums  
                SET temporary_second TO the element at position the element at position i of nums MINUS one of nums  
                SET the element at position the element at position i of nums MINUS one of nums TO temporary_first  
                SET the element at position i of nums TO temporary_second  
            END WHILE  
        END FOR  
        SET result_list TO an empty list  
        FOR each pair of index i and value v IN nums with their corresponding index  
            IF v NOT EQUALS i PLUS one  
                APPEND v TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
