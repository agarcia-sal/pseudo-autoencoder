CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO new list containing element negative one  
        SET max_length TO zero  
        FOR each index i AND character char IN the enumeration of s  
            IF character char EQUALS opening parenthesis  
                APPEND index i TO stack  
            ELSE  
                REMOVE last element FROM stack  
                IF the LENGTH OF stack EQUALS zero  
                    APPEND index i TO stack  
                ELSE  
                    SET current_length TO index i MINUS element at position last in stack  
                    IF current_length GREATER THAN max_length  
                        SET max_length TO current_length  
                    END IF  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
