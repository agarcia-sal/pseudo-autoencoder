CLASS Solution  
    FUNCTION threeSumSmaller(nums PARAMETER, target PARAMETER) RETURNS integer  
        SORT nums IN ascending order  
        SET count TO zero  
        SET n TO the LENGTH OF nums  
        
        FOR i FROM zero TO the VALUE OF n MINUS two  
            SET left TO the VALUE OF i PLUS one  
            SET right TO the VALUE OF n MINUS one  
            
            WHILE left IS LESS THAN right  
                SET current_sum TO the ELEMENT AT POSITION i OF nums PLUS the ELEMENT AT POSITION left OF nums PLUS the ELEMENT AT POSITION right OF nums  
                
                IF current_sum IS LESS THAN target  
                    INCREMENT count BY the VALUE OF right MINUS left  
                    INCREMENT left BY one  
                ELSE  
                    DECREMENT right BY one  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
