CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO the SUM of all elements IN nums  
        IF the remainder WHEN total_sum IS DIVIDED BY two NOT EQUALS zero  
            RETURN false  
        END IF  
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of boolean values with size target PLUS one WHERE all elements ARE false  
        SET element at position zero of dp TO true  
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET element at position i of dp TO the element at position i of dp OR the element at position i MINUS num of dp  
            END FOR  
        END FOR  
        RETURN the element at position target of dp  
    END FUNCTION  
END CLASS
