CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO an empty collection of mappings from characters to indices  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each position end AND corresponding character char IN the sequential enumeration of string s  
            IF character char EXISTS IN the keys of char_index_map AND the value mapped from character char in char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO the value mapped from character char in char_index_map PLUS one  
            END IF  
            SET the value mapped from character char in char_index_map TO end  
            SET max_length TO the maximum of max_length AND the difference between end MINUS start PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
