CLASS Solution  
    FUNCTION minimumTimeRequired WITH PARAMETERS jobs k  
        SET n TO the LENGTH OF jobs  
        SET subset_sums TO a list of zeros with the LENGTH OF two RAISED TO the POWER OF n  
        FOR each i FROM zero TO n MINUS one  
            FOR each mask FROM zero TO two RAISED TO the POWER OF i MINUS one  
                SET the element at position the BITWISE OR of mask WITH two RAISED TO the POWER OF i of subset_sums TO the element at position mask of subset_sums PLUS the element at position i of jobs  
            END FOR  
        END FOR  

        FUNCTION dp WITH PARAMETERS k mask  
            IF k EQUALS one THEN  
                RETURN the element at position mask of subset_sums  
            END IF  
            SET res TO positive infinity  
            SET submask TO mask  
            WHILE submask IS NOT zero  
                SET res TO the minimum BETWEEN res AND the maximum BETWEEN dp with parameters k MINUS one mask BITWISE XOR submask AND the element at position submask of subset_sums  
                SET submask TO the result of the bitwise AND BETWEEN submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  

        RETURN dp with parameters k two RAISED TO the POWER OF n MINUS one  
    END FUNCTION  
END CLASS
