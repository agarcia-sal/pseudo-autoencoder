CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a list of LENGTH n with each element SET TO negative one  
        SET stack TO an empty list  
        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO i MODULO n  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position element at position the LENGTH OF stack MINUS one of stack of nums IS LESS THAN the element at position index of nums  
                SET popped_index TO the element removed from the end of stack  
                SET the element at position popped_index of result TO the element at position index of nums  
            END WHILE  
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
