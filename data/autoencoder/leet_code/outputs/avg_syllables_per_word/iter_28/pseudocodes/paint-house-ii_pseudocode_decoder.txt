CLASS Solution  
    FUNCTION minCostII(costs)  
        IF costs EQUALS empty  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF costs  
        SET k TO the LENGTH OF the element at position zero of costs  
        
        IF k EQUALS one  
            IF n EQUALS one  
                RETURN the element at position zero and zero of costs  
            ELSE  
                RETURN infinity  
            END IF  
        END IF  
        
        FOR i FROM one TO n MINUS one  
            SET min_cost TO infinity  
            SET min_cost_index TO negative one  
            SET second_min_cost TO infinity  
            
            FOR j FROM zero TO k MINUS one  
                IF the element at position i MINUS one and j of costs LESS THAN min_cost  
                    SET second_min_cost TO min_cost  
                    SET min_cost TO the element at position i MINUS one and j of costs  
                    SET min_cost_index TO j  
                ELSE IF the element at position i MINUS one and j of costs LESS THAN second_min_cost  
                    SET second_min_cost TO the element at position i MINUS one and j of costs  
                END IF  
            END FOR  
            
            FOR j FROM zero TO k MINUS one  
                IF j EQUALS min_cost_index  
                    SET the element at position i and j of costs TO the element at position i and j of costs PLUS second_min_cost  
                ELSE  
                    SET the element at position i and j of costs TO the element at position i and j of costs PLUS min_cost  
                END IF  
            END FOR  
        END FOR  
        
        SET answer TO infinity  
        FOR each cost IN the element at the last position of costs  
            IF cost LESS THAN answer  
                SET answer TO cost  
            END IF  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
