CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            IF tuple of i and j IS IN memo THEN  
                RETURN value at tuple of i and j IN memo  
            END IF  
            SET result TO call dp WITH i AND j MINUS one AND memo PLUS one  
            FOR k FROM i TO j MINUS one  
                IF element at position k of s EQUALS element at position j of s THEN  
                    SET result TO the MINIMUM BETWEEN result AND the SUM OF call dp WITH i AND k MINUS one AND memo AND call dp WITH k AND j MINUS one AND memo  
                END IF  
            END FOR  
            SET value at tuple of i and j IN memo TO result  
            RETURN result  
        END FUNCTION  
        RETURN call dp WITH zero AND LENGTH OF s MINUS one AND an empty dictionary  
    END FUNCTION  
END CLASS
