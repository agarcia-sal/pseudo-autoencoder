CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO the SUBSTRING from position one TO position d OF frag  
                SET right TO the SUBSTRING from position d PLUS one TO the LENGTH OF frag OF frag  
                IF (the left STARTS WITH zero IS FALSE OR left EQUALS zero) AND (the right ENDS WITH zero IS FALSE)  
                    YIELD left PLUS (dot IF d NOT EQUALS n ELSE empty string) PLUS right  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO the SUBSTRING from position two TO the LENGTH OF s MINUS one OF s  
        RETURN a LIST constructed by  
            FOR i FROM one TO the LENGTH OF s MINUS one  
                FOR each pair OF a AND b IN the CARTESIAN PRODUCT of make(the SUBSTRING from position one TO position i OF s) AND make(the SUBSTRING from position i PLUS one TO the LENGTH OF s OF s)  
                    THE STRING CONSISTING OF OPEN PARENTHESIS PLUS a PLUS comma PLUS space PLUS b PLUS CLOSE PARENTHESIS  
                END FOR  
            END FOR  
    END FUNCTION  
END CLASS
