CLASS Solution  
    FUNCTION hasPath(maze maze AS LIST OF LIST OF INTEGER start AS LIST OF INTEGER destination AS LIST OF INTEGER) RETURNS BOOLEAN  
        SET directions TO list of pairs where first pair is zero PLUS one second pair is one PLUS zero third pair is zero MINUS one fourth pair is MINUS one PLUS zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF the element at position zero of maze  
        SET visited TO empty set  
        
        FUNCTION dfs(x x AS INTEGER y y AS INTEGER) RETURNS BOOLEAN  
            IF the pair consisting of x AND y IS IN visited THEN  
                RETURN false  
            END IF  
            IF a list consisting of x AND y EQUALS destination THEN  
                RETURN true  
            END IF  
            ADD the pair consisting of x AND y TO visited  
            
            FOR each pair dx AND dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE zero LESS THAN OR EQUAL TO nx PLUS dx AND nx PLUS dx LESS THAN m AND zero LESS THAN OR EQUAL TO ny PLUS dy AND ny PLUS dy LESS THAN n AND the element at position nx PLUS dx of maze at position ny PLUS dy EQUALS zero  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF dfs(nx ny) THEN  
                    RETURN true  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN dfs(the element at position zero of start the element at position one of start)  
    END FUNCTION  
END CLASS
