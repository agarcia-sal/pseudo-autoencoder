CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with value element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values STARTING FROM position two TO the end  
        SET node TO new ListNode with value val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value at p1 EQUALS the value at p2 AND is_same_list(the next element of p1 the next element of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseKGroup(head k)  
        FUNCTION reverseLinkedList(start end)  
            SET prev TO None  
            SET curr TO start  
            WHILE curr NOT EQUALS end  
                SET next_node TO the next element of curr  
                SET the next element of curr TO prev  
                SET prev TO curr  
                SET curr TO next_node  
            END WHILE  
            SET the next element of curr TO prev  
            RETURN end start  
        END FUNCTION  

        SET dummy TO new ListNode with value zero  
        SET the next element of dummy TO head  
        SET prev_group_end TO dummy  

        WHILE True  
            SET kth_node TO prev_group_end  
            FOR index FROM one TO k  
                SET kth_node TO the next element of kth_node  
                IF kth_node EQUALS None  
                    RETURN the next element of dummy  
                END IF  
            END FOR  
            SET next_group_start TO the next element of kth_node  
            SET the next element of kth_node TO None  
            SET new_group_start new_group_end TO reverseLinkedList(the next element of prev_group_end kth_node)  
            SET the next element of prev_group_end TO new_group_start  
            SET the next element of new_group_end TO next_group_start  
            SET prev_group_end TO new_group_end  
        END WHILE  

        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
