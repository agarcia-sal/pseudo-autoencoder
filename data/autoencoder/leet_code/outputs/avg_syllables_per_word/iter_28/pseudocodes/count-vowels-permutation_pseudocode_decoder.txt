CLASS Solution  
    FUNCTION countVowelPermutation(n)  
        SET MOD TO one billion PLUS seven  
        SET a TO one  
        SET e TO one  
        SET i TO one  
        SET o TO one  
        SET u TO one  
        FOR each number FROM two TO n  
            SET a_next TO the sum of e PLUS i PLUS u  
            SET e_next TO the sum of a PLUS i  
            SET i_next TO the sum of e PLUS o  
            SET o_next TO i  
            SET u_next TO the sum of i PLUS o  
            SET a TO the remainder of a_next DIVIDED BY MOD  
            SET e TO the remainder of e_next DIVIDED BY MOD  
            SET i TO the remainder of i_next DIVIDED BY MOD  
            SET o TO the remainder of o_next DIVIDED BY MOD  
            SET u TO the remainder of u_next DIVIDED BY MOD  
        END FOR  
        SET total TO the sum of a PLUS e PLUS i PLUS o PLUS u  
        SET result TO the remainder of total DIVIDED BY MOD  
        RETURN result  
    END FUNCTION  
END CLASS
