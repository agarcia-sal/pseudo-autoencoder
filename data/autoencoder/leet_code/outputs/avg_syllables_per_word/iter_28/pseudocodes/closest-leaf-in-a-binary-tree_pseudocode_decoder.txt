CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, left WITH DEFAULT none, right WITH DEFAULT none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO A new instance of TreeNode WITH parameter element at position one of values  
    SET i TO one  
    SET queue TO A new empty double ended queue  
    APPEND root TO queue  
    WHILE queue HOLDS elements  
        SET node TO the element REMOVED from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.left TO A new instance of TreeNode WITH parameter element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.right TO A new instance of TreeNode WITH parameter element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF value of p NOT EQUALS value of q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH parameters value of p.left AND value of q.left  
        AND the result of is_same_tree WITH parameters value of p.right AND value of q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self, root, k)  
        IF root.left IS none AND root.right IS none THEN  
            RETURN value of root  
        END IF  
        
        SET graph TO a new default dictionary of lists  
        
        FUNCTION dfs(node, parent WITH DEFAULT none)  
            IF parent IS NOT none THEN  
                APPEND value of parent TO the list in graph at key value of node  
                APPEND value of node TO the list in graph at key value of parent  
            END IF  
            IF node.left IS NOT none THEN  
                CALL dfs WITH parameters node.left AND node  
            END IF  
            IF node.right IS NOT none THEN  
                CALL dfs WITH parameters node.right AND node  
            END IF  
        END FUNCTION  
        
        CALL dfs WITH parameter root  
        
        SET queue TO a new double ended queue initialized with element k  
        SET visited TO an empty set  
        
        WHILE queue HOLDS elements  
            SET current TO the element REMOVED from the left side of queue  
            ADD current TO visited  
            
            IF the LENGTH OF the list in graph at key current EQUALS one  
            AND current NOT EQUALS the value of root THEN  
                RETURN current  
            END IF  
            
            FOR each neighbor IN the list in graph at key current  
                IF neighbor NOT IN visited THEN  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
