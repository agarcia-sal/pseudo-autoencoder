CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO negative infinity  
        FOR index i FROM zero TO the LENGTH OF nums MINUS one  
            PUSH onto min_heap the tuple consisting of the element at position zero of the list at position i of nums and index i and zero  
            SET max_value TO the GREATER OF max_value AND the element at position zero of the list at position i of nums  
        END FOR  
        SET smallest_range TO positive infinity  
        SET result TO the list consisting of negative one hundred thousand and one hundred thousand  
        WHILE min_heap is not empty  
            POP from min_heap the tuple consisting of min_value and list_index and element_index  
            IF max_value MINUS min_value LESS THAN smallest_range THEN  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO the list consisting of min_value and max_value  
            END IF  
            IF element_index PLUS one LESS THAN the LENGTH OF the list at position list_index of nums THEN  
                SET next_value TO the element at position element_index PLUS one of the list at position list_index of nums  
                PUSH onto min_heap the tuple consisting of next_value and list_index and element_index PLUS one  
                SET max_value TO the GREATER OF max_value AND next_value  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
