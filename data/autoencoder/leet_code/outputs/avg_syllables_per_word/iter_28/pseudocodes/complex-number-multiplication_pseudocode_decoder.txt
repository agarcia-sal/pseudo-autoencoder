CLASS Solution  
    FUNCTION complexNumberMultiply(num1 parameter AS string parameter num2 parameter AS string) RETURNS string  
        FUNCTION parse_complex(num parameter AS string)  
            SET parts TO the result of splitting num at each occurrence of the plus character  
            SET real TO the integer conversion of the first element of parts  
            SET imaginary TO the integer conversion of the substring from the first character TO the character before the last character of the second element of parts  
            RETURN the pair of real and imaginary  
        END FUNCTION  
        
        SET real1 AND imag1 TO the result of invoking parse_complex with num1  
        SET real2 AND imag2 TO the result of invoking parse_complex with num2  
        
        SET real_part TO the value of real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO the value of real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        
        RETURN the concatenation of the string representation of real_part PLUS the plus character PLUS the string representation of imaginary_part PLUS the character i  
    END FUNCTION  
END CLASS
