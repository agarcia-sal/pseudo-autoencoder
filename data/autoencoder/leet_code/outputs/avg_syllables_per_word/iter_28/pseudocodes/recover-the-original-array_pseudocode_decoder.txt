CLASS Solution  
    FUNCTION recoverArray(nums)  
        SORT nums in ascending order  
        SET n TO the LENGTH OF nums  
        SET first TO the element at position zero of nums  
        FOR i FROM one TO n MINUS one  
            SET difference TO element at position i of nums MINUS first  
            SET k TO difference DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at position i of nums  
                CONTINUE to next iteration of the FOR loop  
            END IF  
            SET count TO a mapping of each element in nums to its frequency  
            SET arr TO an empty list  
            FOR each element x IN nums  
                IF the frequency of x in count EQUALS zero  
                    CONTINUE to next iteration of the FOR loop  
                END IF  
                IF the frequency of the element x PLUS two MULTIPLIED BY k in count EQUALS zero  
                    BREAK out of the FOR loop  
                END IF  
                APPEND the element x PLUS k TO arr  
                DECREMENT the frequency of x in count BY one  
                DECREMENT the frequency of the element x PLUS two MULTIPLIED BY k in count BY one  
            END FOR  
            IF the LENGTH OF arr EQUALS n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an empty list  
    END FUNCTION  
END CLASS
