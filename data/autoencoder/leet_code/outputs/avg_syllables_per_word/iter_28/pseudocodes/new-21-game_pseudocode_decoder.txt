CLASS Solution  
    FUNCTION new21Game WITH PARAMETERS n k maxPts  
        FUNCTION dfs WITH PARAMETER i  
            IF i GREATER THAN OR EQUAL TO k  
                IF i LESS THAN OR EQUAL TO n  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF i EQUALS k MINUS one  
                SET difference TO n MINUS k PLUS one  
                IF difference LESS THAN maxPts  
                    SET numerator TO difference  
                ELSE  
                    SET numerator TO maxPts  
                END IF  
                RETURN numerator DIVIDED BY maxPts  
            END IF  
            SET first_call TO dfs OF i PLUS one  
            SET second_call TO dfs OF i PLUS maxPts PLUS one  
            SET difference TO first_call MINUS second_call  
            RETURN first_call PLUS difference DIVIDED BY maxPts  
        END FUNCTION  
        RETURN dfs OF zero  
    END FUNCTION  
END CLASS
