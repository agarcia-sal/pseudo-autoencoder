CLASS Solution  
    FUNCTION shiftingLetters(s character string, shifts list of integers)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one DOWNTO zero  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO empty list of characters  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET original_character_value TO the alphabetical position of element at position index of s relative to character a  
            SET shifted_value TO original_character_value PLUS element at position index of shifts  
            SET wrapped_value TO shifted_value MODULO twenty six  
            SET new_character TO character corresponding to wrapped_value plus the alphabetical value of character a  
            APPEND new_character TO result  
        END FOR  
        RETURN concatenation of all characters in result  
    END FUNCTION  
END CLASS
