CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO the sorted list of all elements in cuts PLUS zero PLUS n  
        
        FUNCTION dp(left, right)  
            IF the difference of right MINUS left is LESS THAN OR EQUAL TO one THEN  
                RETURN zero  
            END IF  
            
            SET min_cost TO positive infinity  
            FOR i FROM the sum of left PLUS one TO the difference of right MINUS one  
                SET cost TO the element at position right of cuts MINUS the element at position left of cuts PLUS dp called with left and i PLUS dp called with i and right  
                IF cost LESS THAN min_cost THEN  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN dp called with zero and the difference of the length of cuts MINUS one  
    END FUNCTION  
END CLASS
