CLASS Solution  
    FUNCTION minChanges(nums k)  
        SET groups TO a new mapping from each key to an empty list  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            APPEND element at position index of nums TO the list in groups at key equal to index MODULO k  
        END FOR  

        SET freq TO a list containing k new mappings from each key to zero  
        SET min_size TO a list containing k instances of a very large number  
        FOR index FROM zero TO k MINUS one  
            FOR each number IN the list in groups at key equal to index  
                INCREMENT the value in freq at position index and key number BY one  
            END FOR  
            SET element at position index of min_size TO the LENGTH OF the list in groups at key equal to index  
        END FOR  

        SET dp TO a list of length equal to two raised to the power of ten each element being a very large number  
        SET element at position zero of dp TO zero  

        FOR index FROM zero TO k MINUS one  
            SET new_dp TO a list of length equal to two raised to the power of ten each element being a very large number  
            SET min_dp TO the minimum value in dp  
            FOR j FROM zero TO two raised to the power of ten MINUS one  
                SET element at position j of new_dp TO min_dp PLUS element at position index of min_size  
                FOR each key number and corresponding value count IN the mapping freq at position index  
                    SET candidate TO element at position j XOR number of dp PLUS element at position index of min_size MINUS count  
                    IF candidate LESS THAN element at position j of new_dp  
                        SET element at position j of new_dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  

        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
