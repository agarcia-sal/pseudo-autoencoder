CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO dictionary with default value zero for each key  
        FOR each element i IN nums  
            FOR each element j IN nums  
                SET and_result TO bitwise AND of element i AND element j  
                INCREMENT the value at key and_result IN pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each element k IN nums  
            FOR each key pair_and WITH its associated value count IN pair_and_count  
                IF bitwise AND of pair_and AND element k EQUALS zero THEN  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
