CLASS Solution  
    FUNCTION leastBricks(wall)  
        SET gap_counts TO a mapping with default integer zero  
        FOR each row IN wall  
            SET total TO zero  
            FOR each brick IN all elements of row except the last element  
                INCREMENT total BY brick  
                INCREMENT the value associated with total in gap_counts BY one  
            END FOR  
        END FOR  
        SET max_gaps TO the maximum value among all values in gap_counts or zero if gap_counts is empty  
        SET result TO the number of rows in wall MINUS max_gaps  
        RETURN result  
    END FUNCTION  
END CLASS
