CLASS Solution  
    FUNCTION characterReplacement(s k)  
        ASSIGN reference TO a new dictionary with default value zero named count  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the element at position right of s IN count BY one  
            SET max_count TO THE GREATER OF max_count AND the element at position right of s IN count  
            IF the LENGTH OF the current window from left TO right MINUS max_count IS GREATER THAN k  
                DECREMENT the element at position left of s IN count BY one  
                INCREMENT left BY one  
            END IF  
            SET max_length TO THE GREATER OF max_length AND the LENGTH OF the current window from left TO right  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
