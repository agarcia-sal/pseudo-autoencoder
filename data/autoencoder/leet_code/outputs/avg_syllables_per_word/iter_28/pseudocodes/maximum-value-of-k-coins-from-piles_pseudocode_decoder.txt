CLASS Solution  
    FUNCTION maxValueOfCoins(piles k)  
        SET n TO the LENGTH OF piles  
        SET dp TO a list containing n PLUS one elements each of which is a list containing k PLUS one zeros  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one of piles  
                SET current_value TO zero  
                SET limit TO the MINIMUM of j AND the LENGTH OF current_pile  
                FOR l FROM zero TO limit  
                    IF l GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one of current_pile  
                    END IF  
                    IF the element at position i and j of dp LESS THAN the element at position i MINUS one and j MINUS l of dp PLUS current_value  
                        SET element at position i and j of dp TO the element at position i MINUS one and j MINUS l of dp PLUS current_value  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position n and k of dp  
    END FUNCTION  
END CLASS
