CLASS Solution  
    FUNCTION distinctSubseqII(s)  
        SET MOD TO ten RAISED TO THE power of nine PLUS seven  
        SET n TO the LENGTH OF s  
        CREATE list dp WITH size n PLUS one FILLED WITH zero  
        SET element at position zero of dp TO one  
        CREATE dictionary last AS EMPTY  
        FOR i FROM one TO n  
            SET char TO element at position i MINUS one of s  
            SET element at position i of dp TO the element at position i MINUS one of dp MULTIPLIED BY two MODULO MOD  
            IF char EXISTS IN last  
                SET element at position i of dp TO element at position i of dp MINUS element at position last at char MINUS one of dp PLUS MOD MODULO MOD  
            END IF  
            SET element at position char of last TO i  
        END FOR  
        RETURN element at position n of dp MINUS one MODULO MOD  
    END FUNCTION  
END CLASS
