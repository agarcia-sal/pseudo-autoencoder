CLASS Solution  
    FUNCTION orderlyQueue(s parameter, k parameter)  
        IF k GREATER THAN one  
            RETURN the RESULT OF concatenating the ELEMENTS of the COLLECTION obtained by arranging in ascending order all CHARACTERS in s  
        ELSE  
            SET n TO the LENGTH OF s  
            SET smallest TO s  
            FOR i FROM one TO n MINUS one  
                SET rotated TO the CONCATENATION of the SUBSTRING of s from position i TO the END and the SUBSTRING of s from position zero TO position i MINUS one  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
