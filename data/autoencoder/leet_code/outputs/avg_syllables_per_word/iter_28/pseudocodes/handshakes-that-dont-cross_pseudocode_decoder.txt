CLASS Solution  
    FUNCTION numberOfWays(numPeople)  
        SET MOD TO one billion PLUS seven  
        SET dp TO list of zeros of size numPeople PLUS one  
        SET element at position zero of dp TO one  
        SET element at position two of dp TO one  
        
        FOR i FROM four TO numPeople IN STEPS OF two  
            FOR j FROM zero TO i MINUS one IN STEPS OF two  
                SET dp at position i TO dp at position i PLUS dp at position j MULTIPLIED BY dp at position i MINUS two MINUS j  
                SET dp at position i TO dp at position i MODULO MOD  
            END FOR  
        END FOR  
        
        RETURN element at position numPeople of dp  
    END FUNCTION  
END CLASS
