CLASS Solution  
    FUNCTION shortestPathLength(graph)  
        IF graph EQUALS empty  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF graph  
        SET queue TO a new empty double ended queue  
        FOR each node FROM zero TO n MINUS one  
            APPEND the pair of node AND one MULTIPLIED BY two RAISED TO the power of node TO queue  
        END FOR  
        
        SET visited TO a new empty set  
        FOR each node FROM zero TO n MINUS one  
            ADD the pair of node AND one MULTIPLIED BY two RAISED TO the power of node TO visited  
        END FOR  
        
        SET steps TO zero  
        SET all_nodes_visited TO one MULTIPLIED BY two RAISED TO the power of n MINUS one  
        
        WHILE queue IS NOT empty  
            SET current_level_size TO the LENGTH OF queue  
            FOR each iteration FROM zero TO current_level_size MINUS one  
                REMOVE and RETURN the element at the front of queue INTO current_node AND visited_nodes  
                
                IF visited_nodes EQUALS all_nodes_visited  
                    RETURN steps  
                END IF  
                
                FOR each neighbor IN the element at position current_node of graph  
                    SET next_visited_nodes TO visited_nodes OR one MULTIPLIED BY two RAISED TO the power of neighbor  
                    SET next_state TO the pair of neighbor AND next_visited_nodes  
                    
                    IF next_state NOT IN visited  
                        ADD next_state TO visited  
                        APPEND next_state TO queue  
                    END IF  
                END FOR  
            END FOR  
            
            INCREMENT steps BY one  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
