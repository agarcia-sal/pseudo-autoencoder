CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n  
                                    mines)  
        SET mines_set TO an empty set  
        FOR each mine IN mines  
            APPEND the tuple of mine TO mines_set  
        END FOR  
        
        SET dp TO a two dimensional list of n rows and n columns  
        FOR each row_index FROM zero TO n MINUS one  
            SET row TO an empty list  
            FOR each column_index FROM zero TO n MINUS one  
                APPEND a list containing four zeros TO row  
            END FOR  
            APPEND row TO dp  
        END FOR  
        
        SET directions TO a list of four pairs representing zero and one then one and zero then zero and negative one then negative one and zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the pair of i and j IS NOT IN mines_set THEN  
                    IF j IS GREATER THAN zero THEN  
                        SET element at position zero of the list at position j of the list at position i of dp TO the element at position zero of the list at position j MINUS one of the list at position i of dp PLUS one  
                    END IF  
                    IF i IS GREATER THAN zero THEN  
                        SET element at position one of the list at position j of the list at position i of dp TO the element at position one of the list at position j of the list at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                IF the pair of i and j IS NOT IN mines_set THEN  
                    IF j IS LESS THAN n MINUS one THEN  
                        SET element at position two of the list at position j of the list at position i of dp TO the element at position two of the list at position j PLUS one of the list at position i of dp PLUS one  
                    END IF  
                    IF i IS LESS THAN n MINUS one THEN  
                        SET element at position three of the list at position j of the list at position i of dp TO the element at position three of the list at position j of the list at position i PLUS one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the minimum value among the elements of the list at position j of the list at position i of dp  
                    IF plus_sign_order IS GREATER THAN result THEN  
                        SET result TO plus_sign_order  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
