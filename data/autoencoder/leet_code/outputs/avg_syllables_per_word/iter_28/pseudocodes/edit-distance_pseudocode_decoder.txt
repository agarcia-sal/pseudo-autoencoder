CLASS Solution  
    FUNCTION minDistance(word1 word1 AS STRING word2 word2 AS STRING) RETURNS INTEGER  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        SET dp TO a new list containing m PLUS one elements each initialized to a new list containing n PLUS one zeros  
        
        FOR i FROM zero TO m  
            SET the element at position zero of the element at position i of dp TO i  
        END FOR  
        
        FOR j FROM zero TO n  
            SET the element at position j of the element at position zero of dp TO j  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2  
                    SET the element at position j of the element at position i of dp TO the element at position j MINUS one of the element at position i MINUS one of dp  
                ELSE  
                    SET the element at position j of the element at position i of dp TO one PLUS the MINIMUM OF the element at position j of the element at position i MINUS one of dp AND the element at position j MINUS one of the element at position i of dp AND the element at position j MINUS one of the element at position i MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position n of the element at position m of dp  
    END FUNCTION  
END CLASS
