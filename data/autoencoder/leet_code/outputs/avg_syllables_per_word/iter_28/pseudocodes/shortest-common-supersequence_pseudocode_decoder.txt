CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1, str2)  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        
        SET dp TO a list containing m PLUS one elements each of which is a list containing n PLUS one zeros  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2  
                    SET the element at position i of dp AT position j TO the element at position i MINUS one of dp AT position j MINUS one PLUS one  
                ELSE  
                    IF the element at position i MINUS one of dp AT position j GREATER THAN the element at position i of dp AT position j MINUS one  
                        SET the element at position i of dp AT position j TO the element at position i MINUS one of dp AT position j  
                    ELSE  
                        SET the element at position i of dp AT position j TO the element at position i of dp AT position j MINUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET i TO m  
        SET j TO n  
        SET scs TO an empty list  
        
        WHILE i GREATER THAN zero AND j GREATER THAN zero  
            IF the element at position i MINUS one of str1 EQUALS the element at position j MINUS one of str2  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF the element at position i MINUS one of dp AT position j GREATER THAN the element at position i of dp AT position j MINUS one  
                APPEND the element at position i MINUS one of str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND the element at position j MINUS one of str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i GREATER THAN zero  
            APPEND the element at position i MINUS one of str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        
        WHILE j GREATER THAN zero  
            APPEND the element at position j MINUS one of str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        
        REVERSE the order of elements IN scs  
        
        RETURN the concatenation of all elements IN scs AS a string  
    END FUNCTION  
END CLASS
