CLASS Solution  
    FUNCTION totalHammingDistance(nums)  
        SET total_distance TO zero  
        FOR i FROM zero TO thirty-one  
            SET count_ones TO zero  
            FOR each num IN nums  
                INCREMENT count_ones BY the value of element at position i of the binary representation of num counted from the least significant bit where element equals one if bit at position i of num is one otherwise zero  
            END FOR  
            SET count_zeros TO the LENGTH OF nums MINUS count_ones  
            INCREMENT total_distance BY count_ones MULTIPLIED BY count_zeros  
        END FOR  
        RETURN total_distance  
    END FUNCTION  
END CLASS
