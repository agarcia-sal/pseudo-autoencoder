CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN a string representing the element at position one of nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN a string formed by the element at position one of nums FOLLOWED BY the word division FOLLOWED BY the element at position two of nums  
        END IF  
        RETURN a string formed by the element at position one of nums FOLLOWED BY the word division FOLLOWED BY an open parenthesis FOLLOWED BY the concatenation of strings representing each element from position two TO the last position of nums JOINED BY the word division FOLLOWED BY a close parenthesis  
    END FUNCTION  
END CLASS
