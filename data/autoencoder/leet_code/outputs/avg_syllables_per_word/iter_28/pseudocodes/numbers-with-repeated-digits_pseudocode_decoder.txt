CLASS Solution  
    FUNCTION numDupDigitsAtMostN(n)  
        FUNCTION count_unique_digits(x)  
            SET s TO the string representation of x  
            SET length TO the number of characters in s  
            SET count TO zero  
            FOR i FROM one TO length MINUS one  
                INCREMENT count BY nine MULTIPLIED BY perm of nine AND i MINUS one  
            END FOR  
            SET seen TO an empty set  
            FOR i FROM zero TO length MINUS one  
                SET digit TO the element at position i of s converted to an integer  
                FOR d FROM zero TO digit MINUS one  
                    IF i EQUALS zero THEN  
                        SET start_range TO one  
                    ELSE  
                        SET start_range TO zero  
                    END IF  
                    IF d GREATER THAN OR EQUAL TO start_range AND d NOT IN seen THEN  
                        INCREMENT count BY perm of nine MINUS i AND length MINUS i MINUS one  
                    END IF  
                END FOR  
                IF digit IN seen THEN  
                    BREAK the loop  
                END IF  
                ADD digit TO seen  
            END FOR  
            IF the loop finished without break THEN  
                INCREMENT count BY one  
            END IF  
            RETURN count  
        END FUNCTION  
        SET unique_count TO count_unique_digits(n)  
        RETURN n MINUS unique_count  
    END FUNCTION  
    
    FUNCTION perm(m, n)  
        IF n EQUALS zero THEN  
            RETURN one  
        END IF  
        SET result TO one  
        FOR i FROM zero TO n MINUS one  
            SET result TO result MULTIPLIED BY m MINUS i  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
