CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET max_num TO the maximum element IN nums  
        SET num_set TO the set containing all elements IN nums  
        SET count TO zero  
        FOR each integer g FROM one TO max_num INCLUSIVE  
            SET gcd_val TO zero  
            FOR each integer m STARTING FROM g TO max_num INCLUSIVE IN STEPS OF g  
                IF integer m IS AN ELEMENT OF num_set  
                    SET gcd_val TO the greatest common divisor OF gcd_val AND m  
                    IF gcd_val EQUALS g  
                        INCREMENT count BY one  
                        BREAK FROM the inner FOR loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
