CLASS Solution  
    FUNCTION combine(n k)  
        FUNCTION backtrack(start path)  
            IF the LENGTH OF path EQUALS k  
                APPEND a COPY OF path TO result  
                RETURN  
            END IF  
            FOR i FROM start TO n PLUS one  
                APPEND the element i TO path  
                CALL backtrack WITH the arguments the element i PLUS one AND path  
                REMOVE the last element FROM path  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH the arguments one AND an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
