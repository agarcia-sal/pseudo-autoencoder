CLASS Solution  
    FUNCTION search(nums WITH TYPE List OF integers, target WITH TYPE integer) RETURNS boolean  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN False  
        END IF  
  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
  
        WHILE left LESS THAN OR EQUAL TO right DO  
            SET mid TO left PLUS the result of right MINUS left DIVIDED BY two  
  
            IF the element at position mid of nums EQUALS target THEN  
                RETURN True  
            END IF  
  
            IF the element at position left of nums EQUALS the element at position mid of nums AND the element at position mid of nums EQUALS the element at position right of nums THEN  
                INCREMENT left BY one  
                DECREMENT right BY one  
            ELSE IF the element at position left of nums LESS THAN OR EQUAL TO the element at position mid of nums THEN  
                IF the element at position left of nums LESS THAN OR EQUAL TO target AND target LESS THAN the element at position mid of nums THEN  
                    SET right TO mid MINUS one  
                ELSE  
                    SET left TO mid PLUS one  
                END IF  
            ELSE  
                IF the element at position mid of nums LESS THAN target AND target LESS THAN OR EQUAL TO the element at position right of nums THEN  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END IF  
        END WHILE  
  
        RETURN False  
    END FUNCTION  
END CLASS
