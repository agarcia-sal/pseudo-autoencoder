CLASS Solution  
    FUNCTION addBoldTag(s parameter s AS string words parameter words AS list of strings) RETURNS string  
        IF the LENGTH OF words EQUALS zero  
            RETURN s  
        END IF  

        SET n TO the LENGTH OF s  
        SET mask TO a list OF boolean values all SET TO False WITH the LENGTH OF n  

        FOR each word IN words  
            SET start TO zero  
            WHILE start IS LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the position WHERE word FIRST OCCURS IN s STARTING FROM start  
                IF start EQUALS negative one  
                    BREAK  
                END IF  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i OF mask TO True  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  

        SET result TO an empty list  
        SET i TO zero  
        WHILE i IS LESS THAN n  
            IF the element at position i OF mask EQUALS True  
                APPEND the string consisting of the characters less than b greater than TO result  
                WHILE i IS LESS THAN n AND the element at position i OF mask EQUALS True  
                    APPEND the element at position i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string consisting of the characters less than slash b greater than TO result  
            ELSE  
                APPEND the element at position i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  

        RETURN the concatenation OF all elements IN result  
    END FUNCTION  
END CLASS
