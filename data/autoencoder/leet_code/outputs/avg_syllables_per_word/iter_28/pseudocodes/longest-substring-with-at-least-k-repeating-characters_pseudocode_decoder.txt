CLASS Solution  
    FUNCTION longestSubstring WITH parameters s k  
        IF the LENGTH OF s LESS THAN k  
            RETURN zero  
        END IF  
        ASSIGN count TO the FREQUENCY COUNTS OF each character IN s  
        FOR each char IN the KEYS OF count  
            IF the FREQUENCY OF char IN count LESS THAN k  
                RETURN the MAXIMUM value OF calling longestSubstring ON each substring OBTAINED BY SPLITTING s BY char WITH parameter k  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
