CLASS Solution  
    FUNCTION kthSmallest(mat parameter, k parameter) RETURNS integer  
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET initial_sum TO zero  
        FOR each row IN mat  
            INCREMENT initial_sum BY element at position zero of row  
        END FOR  
        SET initial_indices_list TO an empty list  
        FOR position FROM zero TO m MINUS one  
            APPEND zero TO initial_indices_list  
        END FOR  
        SET initial_indices_tuple TO the tuple formed from initial_indices_list  
        SET min_heap TO a list containing a pair of initial_sum AND initial_indices_tuple  
        SET visited TO a set containing initial_indices_tuple  
        
        WHILE k GREATER THAN zero  
            SET current_pair TO the removal and return of the smallest element from min_heap  
            SET current_sum TO the first element of current_pair  
            SET indices TO the second element of current_pair  
            DECREMENT k BY one  
            
            IF k EQUALS zero  
                RETURN current_sum  
            END IF  
            
            FOR i FROM zero TO m MINUS one  
                SET current_index TO element at position i of indices  
                IF current_index PLUS one LESS THAN n  
                    SET new_indices_list TO a list form of indices  
                    INCREMENT element at position i of new_indices_list BY one  
                    SET new_indices_tuple TO the tuple formed from new_indices_list  
                    SET old_value TO element at position i of mat AT position current_index  
                    SET new_value TO element at position i of mat AT position current_index PLUS one  
                    SET new_sum TO current_sum MINUS old_value PLUS new_value  
                    
                    IF new_indices_tuple NOT IN visited  
                        ADD new_indices_tuple TO visited  
                        ADD the pair of new_sum AND new_indices_tuple TO min_heap maintaining heap property  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
