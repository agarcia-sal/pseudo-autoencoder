CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF element at position one of mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF element at position one of mat2  
        SET result TO a new collection where for each from one TO m there is a list of n elements each set TO zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                FOR p FROM one TO k  
                    INCREMENT element at position i of result at position j BY the element at position i of mat1 at position p MULTIPLIED BY the element at position p of mat2 at position j  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
