CLASS Solution  
    FUNCTION maximizeXor(nums, queries)  
        SORT nums IN ascending order  
        
        SET sorted_queries TO the enumeration of queries sorted by the second element IN each query  
        
        SET result TO a list with the LENGTH OF queries elements each SET TO negative one  
        
        SET trie TO an empty collection  
        
        SET num_index TO zero  
        
        FOR each query_index and query_pair IN sorted_queries  
            SET xi TO the first element of query_pair  
            SET mi TO the second element of query_pair  
            
            WHILE num_index LESS THAN the LENGTH OF nums AND element at position num_index of nums LESS THAN OR EQUAL TO mi  
                SET num TO element at position num_index of nums  
                SET node TO trie  
                
                FOR i FROM thirty-one DOWN TO zero  
                    SET bit TO the value of the bit at position i of num  
                    IF bit NOT IN node  
                        SET element with key bit of node TO an empty collection  
                    END IF  
                    SET node TO element with key bit of node  
                END FOR  
                
                INCREMENT num_index BY one  
            END WHILE  
            
            IF trie IS empty  
                CONTINUE to next iteration  
            END IF  
            
            SET node TO trie  
            SET max_xor TO zero  
            
            FOR i FROM thirty-one DOWN TO zero  
                SET bit TO the value of the bit at position i of xi  
                SET toggle_bit TO one MINUS bit  
                
                IF toggle_bit IN node  
                    SET max_xor TO max_xor OR the value with one AT bit position i  
                    SET node TO element with key toggle_bit of node  
                ELSE  
                    SET node TO element with key bit of node  
                END IF  
            END FOR  
            
            SET element at position query_index of result TO max_xor  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
