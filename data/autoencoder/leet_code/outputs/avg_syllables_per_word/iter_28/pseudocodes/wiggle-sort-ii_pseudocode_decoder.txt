CLASS Solution  
    FUNCTION wiggleSort(nums)  
        CALL sort_nums_in_ascending_order(nums)  
        SET sorted_nums TO a new list with elements copied from nums  
        SET n TO the LENGTH OF nums  
        SET left TO the INTEGER DIVISION OF the DIFFERENCE BETWEEN n AND one BY two  
        SET right TO the DIFFERENCE BETWEEN n AND one  
        FOR index FROM zero TO the DIFFERENCE BETWEEN n AND one  
            IF the REMAINDER OF index DIVIDED BY two EQUALS zero  
                SET the element at position index of nums TO the element at position left of sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET the element at position index of nums TO the element at position right of sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
