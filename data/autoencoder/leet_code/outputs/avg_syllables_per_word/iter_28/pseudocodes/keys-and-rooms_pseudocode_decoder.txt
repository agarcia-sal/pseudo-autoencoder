CLASS Solution  
    FUNCTION canVisitAllRooms with parameter rooms  
        SET visited TO an empty set  
        SET stack TO a list containing zero  
        WHILE the stack is not empty  
            SET current_room TO the last element of stack removed from stack  
            ADD current_room TO visited  
            FOR each key IN the collection of elements at position current_room of rooms  
                IF key NOT IN visited THEN  
                    ADD key TO stack  
                END IF  
            END FOR  
        END WHILE  
        IF the number of elements in visited EQUALS the number of elements in rooms THEN  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
