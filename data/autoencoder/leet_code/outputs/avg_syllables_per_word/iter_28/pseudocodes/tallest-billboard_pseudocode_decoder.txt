CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO a dictionary with key zero and value zero  
        
        FOR each rod IN rods  
            SET new_dp TO a copy of dp dictionary  
            FOR each key d and value y IN dp items  
                SET key_for_taller_support TO d PLUS rod  
                SET value_for_taller_support TO the maximum value between the value at key_for_taller_support in new_dp or zero AND y  
                SET the value at key_for_taller_support in new_dp TO value_for_taller_support  
                
                IF d GREATER THAN rod  
                    SET difference_key TO d MINUS rod  
                ELSE IF rod GREATER THAN d  
                    SET difference_key TO rod MINUS d  
                ELSE  
                    SET difference_key TO zero  
                END IF  
                
                IF d LESS THAN rod  
                    SET minimum_value TO d  
                ELSE  
                    SET minimum_value TO rod  
                END IF  
                
                SET value_for_shorter_support TO the maximum value between the value at difference_key in new_dp or zero AND y PLUS minimum_value  
                SET the value at difference_key in new_dp TO value_for_shorter_support  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the value at key zero in dp  
    END FUNCTION  
END CLASS
