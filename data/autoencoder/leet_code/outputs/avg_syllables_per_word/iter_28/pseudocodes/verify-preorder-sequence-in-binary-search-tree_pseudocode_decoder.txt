CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO an empty list  
        SET lower_bound TO negative infinity representation  
        
        FOR each value IN preorder  
            IF value LESS THAN lower_bound  
                RETURN False  
            END IF  
            
            WHILE the stack is not empty AND value GREATER THAN element at the last position of the stack  
                SET lower_bound TO the element removed from the last position of the stack  
            END WHILE  
            
            APPEND value TO stack  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
