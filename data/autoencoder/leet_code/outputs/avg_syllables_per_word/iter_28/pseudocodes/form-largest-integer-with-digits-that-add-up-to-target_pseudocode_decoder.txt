CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        CREATE dictionary cost_to_digit mapping each element c in cost to string of position of c IN cost PLUS one  
        CREATE list dp with empty string as first element FOLLOWED BY the string dash one repeated for target times  
        FOR t FROM one TO target  
            FOR each key c AND value d IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND element at position t MINUS c of dp NOT EQUALS dash one  
                    SET candidate TO concatenation of d AND element at position t MINUS c of dp  
                    IF element at position t of dp EQUALS dash one OR LENGTH OF candidate GREATER THAN LENGTH OF element at position t of dp OR (LENGTH OF candidate EQUALS LENGTH OF element at position t of dp AND candidate GREATER THAN element at position t of dp)  
                        SET element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        IF element at position target of dp NOT EQUALS dash one  
            RETURN element at position target of dp  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
