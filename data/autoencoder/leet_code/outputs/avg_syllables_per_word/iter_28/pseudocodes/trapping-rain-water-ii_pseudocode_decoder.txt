CLASS Solution  
    FUNCTION trapRainWater(heightMap)  
        IF heightMap EQUALS empty OR element at position zero of heightMap EQUALS empty  
            RETURN zero  
        END IF  

        SET m TO the LENGTH OF heightMap  
        SET n TO the LENGTH OF element at position zero of heightMap  
        SET visited TO a two dimensional collection of size m BY n FILLED WITH false  
        SET min_heap TO an empty collection  

        FOR index i FROM zero TO m MINUS one  
            ADD a tuple consisting of the element at position i of heightMap at position zero AND i AND zero TO min_heap  
            ADD a tuple consisting of the element at position i of heightMap at position n MINUS one AND i AND n MINUS one TO min_heap  
            SET element at position i of visited at position zero TO true  
            SET element at position i of visited at position n MINUS one TO true  
        END FOR  

        FOR index j FROM zero TO n MINUS one  
            ADD a tuple consisting of the element at position zero of heightMap at position j AND zero AND j TO min_heap  
            ADD a tuple consisting of the element at position m MINUS one of heightMap at position j AND m MINUS one AND j TO min_heap  
            SET element at position zero of visited at position j TO true  
            SET element at position m MINUS one of visited at position j TO true  
        END FOR  

        SET directions TO a collection of four tuples consisting of negative one AND zero, positive one AND zero, zero AND negative one, zero AND positive one  
        SET water_trapped TO zero  

        WHILE min_heap is not empty  
            REMOVE and RETURN the smallest element FROM min_heap AND ASSIGN to variables height AND x AND y  

            FOR each tuple consisting of dx AND dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  

                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx of visited at position ny EQUALS false  
                    SET element at position nx of visited at position ny TO true  
                    IF height GREATER THAN element at position nx of heightMap at position ny  
                        INCREMENT water_trapped BY height MINUS element at position nx of heightMap at position ny  
                    END IF  
                    ADD a tuple consisting of the greater value BETWEEN height AND element at position nx of heightMap at position ny AND nx AND ny TO min_heap  
                END IF  
            END FOR  
        END WHILE  

        RETURN water_trapped  
    END FUNCTION  
END CLASS
