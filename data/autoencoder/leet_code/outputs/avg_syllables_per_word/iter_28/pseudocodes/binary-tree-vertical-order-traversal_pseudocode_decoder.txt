CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values EQUALS empty list  
        RETURN nothing  
    END IF  
    SET root TO new TreeNode initialized with element at position one of values  
    SET i TO one  
    CREATE an empty double ended queue assigned to queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater than zero  
        REMOVE and ASSIGN the first element of queue TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing  
            SET node.left TO new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS nothing  
            SET node.right TO new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p EQUALS nothing AND q EQUALS nothing  
        RETURN True  
    ELSE IF p EQUALS nothing OR q EQUALS nothing  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the logical AND of calling is_same_tree with p.left and q.left AND calling is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder WITH PARAMETERS root  
        IF root EQUALS nothing  
            RETURN empty list  
        END IF  
        CREATE a default dictionary of lists assigned to column_table  
        CREATE a double ended queue assigned to queue initialized with a tuple of root and zero  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE the LENGTH OF queue is greater than zero  
            REMOVE and ASSIGN the first element of queue TO node and column_index  
            IF node NOT EQUALS nothing  
                APPEND node.val TO the list at key column_index in column_table  
                IF column_index LESS THAN min_column  
                    SET min_column TO column_index  
                END IF  
                IF column_index GREATER THAN max_column  
                    SET max_column TO column_index  
                END IF  
                APPEND a tuple of node.left and column_index MINUS one TO queue  
                APPEND a tuple of node.right and column_index PLUS one TO queue  
            END IF  
        END WHILE  
        CREATE an empty list assigned to result  
        FOR x FROM min_column TO max_column  
            APPEND the list at key x in column_table TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
