CLASS Solution  
    FUNCTION generateParenthesis WITH PARAMETER n  
        FUNCTION backtrack WITH PARAMETERS s left right  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n  
                CALL backtrack WITH s CONCATENATED WITH opening parenthesis left INCREMENTED BY one right  
            END IF  
            IF right LESS THAN left  
                CALL backtrack WITH s CONCATENATED WITH closing parenthesis left right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH no arguments  
        RETURN result  
    END FUNCTION  
END CLASS
