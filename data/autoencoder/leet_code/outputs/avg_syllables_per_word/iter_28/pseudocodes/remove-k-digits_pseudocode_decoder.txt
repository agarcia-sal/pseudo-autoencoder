CLASS Solution  
    FUNCTION removeKdigits(num string k integer) RETURNS string  
        SET stack TO empty list  
        FOR each digit IN num  
            WHILE k GREATER THAN zero AND stack IS NOT empty AND element at last position of stack GREATER THAN digit  
                REMOVE element at last position FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        IF k EQUALS zero THEN  
            SET final_stack TO stack  
        ELSE  
            SET final_stack TO all elements of stack EXCEPT for the last k elements  
        END IF  
        SET result TO concatenation of all elements in final_stack WITH leading zeros REMOVED  
        IF result IS empty THEN  
            RETURN string zero  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
