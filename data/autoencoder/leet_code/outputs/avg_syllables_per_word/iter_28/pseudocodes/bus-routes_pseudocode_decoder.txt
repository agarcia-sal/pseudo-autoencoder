CLASS Solution  
    FUNCTION numBusesToDestination(routes, source, target)  
        IF source EQUALS target  
            RETURN zero  
        END IF  

        SET stop_to_buses TO a new dictionary where each key has a new list as a default value  

        SET bus_position TO zero  
        FOR each route IN routes  
            FOR each stop IN route  
                APPEND bus_position TO the list associated WITH stop in stop_to_buses  
            END FOR  
            INCREMENT bus_position BY one  
        END FOR  

        SET bus_queue TO a new double ended queue  
        SET visited_buses TO a new empty set  
        SET visited_stops TO a new empty set  

        FOR each bus IN the list associated WITH source in stop_to_buses  
            APPEND a tuple containing bus AND one TO bus_queue  
            ADD bus TO visited_buses  
            ADD source TO visited_stops  
        END FOR  

        WHILE bus_queue is not empty  
            REMOVE the first element FROM bus_queue AND ASSIGN the first part TO current_bus AND the second part TO bus_count  

            FOR each stop IN routes at position current_bus  
                IF stop EQUALS target  
                    RETURN bus_count  
                END IF  

                IF stop is not an element of visited_stops  
                    ADD stop TO visited_stops  
                    FOR each next_bus IN the list associated WITH stop in stop_to_buses  
                        IF next_bus is not an element of visited_buses  
                            ADD next_bus TO visited_buses  
                            APPEND a tuple containing next_bus AND bus_count PLUS one TO bus_queue  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
