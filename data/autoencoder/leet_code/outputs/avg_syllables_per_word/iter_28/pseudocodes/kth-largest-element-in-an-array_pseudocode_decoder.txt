CLASS Solution  
    FUNCTION findKthLargest(nums k)  
        SET min_heap TO empty list  
        FOR each number IN nums  
            ADD the current number TO min_heap maintaining heap property  
            IF the LENGTH OF min_heap GREATER THAN k  
                REMOVE the smallest element FROM min_heap maintaining heap property  
            END IF  
        END FOR  
        RETURN the element at position zero of min_heap  
    END FUNCTION  
END CLASS
