CLASS Solution  
    FUNCTION findIntegers WITH PARAMETER n  
        SET binary_representation TO the binary representation of n starting from the position after the first character  
        SET length TO the number of elements in binary_representation  
        DECLARE dp AS a list of zeros with the size of length PLUS one  
        SET the element at position zero of dp TO one  
        SET the element at position one of dp TO two  
        FOR i FROM two TO length  
            SET the element at position i of dp TO the element at position i MINUS one of dp PLUS the element at position i MINUS two of dp  
        END FOR  
        SET result TO zero  
        SET previous_bit TO zero  
        FOR i FROM zero TO length MINUS one  
            IF the element at position i of binary_representation EQUALS one AS A CHARACTER  
                INCREMENT result BY the element at position length MINUS i MINUS one of dp  
                IF previous_bit EQUALS one  
                    DECREMENT result BY one  
                    BREAK the loop  
                END IF  
            END IF  
            SET previous_bit TO one IF the element at position i of binary_representation EQUALS one AS A CHARACTER OTHERWISE zero  
        END FOR  
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
