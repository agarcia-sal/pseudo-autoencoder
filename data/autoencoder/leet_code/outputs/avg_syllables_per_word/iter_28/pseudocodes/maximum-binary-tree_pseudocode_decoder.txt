CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO A new TreeNode with value EQUALS element at position one of values  
    SET i TO one  
    SET queue TO AN empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element REMOVED from the left of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO A new TreeNode with value EQUALS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO A new TreeNode with value EQUALS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree(self, nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN None  
        END IF  
        SET max_val TO the maximum value IN nums  
        SET max_index TO the position of max_val IN nums  
        SET root TO A new TreeNode with value EQUALS max_val  
        SET root.left TO the result of calling self.constructMaximumBinaryTree WITH the elements FROM position one TO position max_index MINUS one OF nums  
        SET root.right TO the result of calling self.constructMaximumBinaryTree WITH the elements FROM position max_index PLUS one TO the end OF nums  
        RETURN root  
    END FUNCTION  
END CLASS
