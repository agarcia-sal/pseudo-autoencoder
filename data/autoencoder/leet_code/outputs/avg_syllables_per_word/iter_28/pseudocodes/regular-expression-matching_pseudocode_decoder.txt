CLASS Solution  
    FUNCTION isMatch(s string parameter p string parameter) RETURNS boolean  
        SET dp TO a two dimensional list of boolean values initialized to False with the number of rows equal to the length of string parameter s PLUS one and the number of columns equal to the length of string parameter p PLUS one  
        SET the element at row position zero and column position zero of dp TO True  
        FOR each j FROM one TO the length of string parameter p  
            IF the element at position j MINUS one of string parameter p EQUALS the character asterisk  
                SET the element at row position zero and column position j of dp TO the element at row position zero and column position j MINUS two of dp  
            END IF  
        END FOR  
        FOR each i FROM one TO the length of string parameter s  
            FOR each j FROM one TO the length of string parameter p  
                IF the element at position j MINUS one of string parameter p EQUALS the element at position i MINUS one of string parameter s OR the element at position j MINUS one of string parameter p EQUALS the character dot  
                    SET the element at row position i and column position j of dp TO the element at row position i MINUS one and column position j MINUS one of dp  
                ELSE IF the element at position j MINUS one of string parameter p EQUALS the character asterisk  
                    SET the element at row position i and column position j of dp TO the element at row position i and column position j MINUS two of dp  
                    IF the element at position j MINUS two of string parameter p EQUALS the element at position i MINUS one of string parameter s OR the element at position j MINUS two of string parameter p EQUALS the character dot  
                        SET the element at row position i and column position j of dp TO the logical OR of the current value of the element at row position i and column position j of dp AND the element at row position i MINUS one and column position j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at row position equal to the length of string parameter s and column position equal to the length of string parameter p of dp  
    END FUNCTION  
END CLASS
