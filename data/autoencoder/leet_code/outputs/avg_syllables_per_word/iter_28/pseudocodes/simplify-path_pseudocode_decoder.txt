CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the list of each element obtained by splitting path by the slash character  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS a single dot character  
                CONTINUE to the next iteration  
            ELSE IF component EQUALS two consecutive dot characters  
                IF the stack is not empty  
                    REMOVE the last element FROM the stack  
                END IF  
            ELSE  
                ADD component TO the end of the stack  
            END IF  
        END FOR  
        SET simplified_path TO the concatenation of a single slash character and the joining of all elements of stack separated by the slash character  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
