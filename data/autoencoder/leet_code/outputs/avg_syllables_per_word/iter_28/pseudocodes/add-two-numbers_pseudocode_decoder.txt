CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the collection values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO a new ListNode constructed with element at position one of the collection values  
    SET p TO head  
    FOR each element val IN the collection values starting from position two TO the end  
        SET node TO a new ListNode constructed with val  
        SET the element named next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN TRUE  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN FALSE  
    END IF  
    RETURN the condition that the element named val of p1 EQUALS the element named val of p2 AND is_same_list called with the element named next of p1 and the element named next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1 l2)  
        SET dummy_head TO a new ListNode constructed with zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry IS NOT zero  
            IF l1 IS NOT None  
                SET val1 TO the element named val of l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT None  
                SET val2 TO the element named val of l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO the integer result of total DIVIDED BY ten  
            SET the element named next of current TO a new ListNode constructed with the integer result of total MODULO ten  
            SET current TO the element named next of current  
            
            IF l1 IS NOT None  
                SET l1 TO the element named next of l1  
            END IF  
            IF l2 IS NOT None  
                SET l2 TO the element named next of l2  
            END IF  
        END WHILE  
        
        RETURN the element named next of dummy_head  
    END FUNCTION  
END CLASS
