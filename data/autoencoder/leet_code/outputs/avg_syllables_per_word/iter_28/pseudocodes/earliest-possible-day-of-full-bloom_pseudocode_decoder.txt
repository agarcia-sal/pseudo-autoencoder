CLASS Solution  
    FUNCTION earliestFullBloom(plantTime growTime)  
        SET tasks TO the list created by pairing each element at each corresponding position in plantTime and growTime and then sorting this list in the order determined by descending value of the second element in each pair  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair consisting of plant and grow IN tasks  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF bloom_day GREATER THAN earliest_bloom  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
