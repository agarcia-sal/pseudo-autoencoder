CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position one of values and left none and right none  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the front of queue  
        IF i LESS THAN the length of values AND element at position i of values is not none  
            SET node.left TO new TreeNode with value element at position i of values and left none and right none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values is not none  
            SET node.right TO new TreeNode with value element at position i of values and left none and right none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with parameters value of left child of p and value of left child of q AND the result of is_same_tree with parameters value of right child of p and value of right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self, root)  
        IF root EQUALS none  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO double ended queue initialized with root  
        SET left_to_right TO true  
        WHILE queue is not empty  
            SET level_size TO the length of queue  
            SET current_level TO empty double ended queue  
            FOR each index from zero TO level_size MINUS one  
                SET node TO element removed from the front of queue  
                IF left_to_right IS true  
                    APPEND value of node TO current_level  
                ELSE  
                    APPEND value of node TO the front of current_level  
                END IF  
                IF left child of node IS NOT none  
                    APPEND left child of node TO queue  
                END IF  
                IF right child of node IS NOT none  
                    APPEND right child of node TO queue  
                END IF  
            END FOR  
            APPEND the list form of current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
