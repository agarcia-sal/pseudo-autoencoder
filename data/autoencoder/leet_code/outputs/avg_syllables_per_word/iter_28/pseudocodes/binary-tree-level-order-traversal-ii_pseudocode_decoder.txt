CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode WITH the element at position one of values  
    SET i TO one  
    SET queue TO AN EMPTY double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero DO  
        SET node TO THE FIRST element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET node.left TO A NEW TreeNode WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET node.right TO A NEW TreeNode WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom(self, root)  
        IF root EQUALS None THEN  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO A double ended queue CONTAINING root  
        WHILE the LENGTH OF queue GREATER THAN zero DO  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR index FROM one TO level_size DO  
                SET node TO THE FIRST element REMOVED FROM queue  
                APPEND the val attribute of node TO current_level  
                IF the left attribute of node NOT EQUALS None THEN  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node NOT EQUALS None THEN  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN the elements of result IN REVERSE ORDER  
    END FUNCTION  
END CLASS
