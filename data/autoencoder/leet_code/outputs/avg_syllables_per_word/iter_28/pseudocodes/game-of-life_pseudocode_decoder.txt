CLASS Solution  
    FUNCTION gameOfLife(board)  
        IF board EQUALS none OR the LENGTH OF board EQUALS zero OR the LENGTH OF element at position zero of board EQUALS zero THEN  
            RETURN  
        END IF  
      
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
      
        SET directions TO a list consisting of pairs representing the eight neighbor direction offsets defined as first element MINUS one PLUS second element MINUS one then first element MINUS one PLUS zero then first element MINUS one PLUS one then first element zero PLUS second element MINUS one then first element zero PLUS second element PLUS one then first element PLUS one PLUS second element MINUS one then first element PLUS one PLUS zero then first element PLUS one PLUS one  
      
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET live_neighbors TO zero  
                FOR each pair dx dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN m AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND (element at position ni of board EQUALS one OR element at position ni of board EQUALS three) THEN  
                        INCREMENT live_neighbors BY one  
                    END IF  
                END FOR  
                
                IF element at position i of board EQUALS one THEN  
                    IF live_neighbors LESS THAN two OR live_neighbors GREATER THAN three THEN  
                        SET element at position i of board TO three  
                    END IF  
                ELSE  
                    IF live_neighbors EQUALS three THEN  
                        SET element at position i of board TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
      
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board EQUALS two THEN  
                    SET element at position i of board TO one  
                ELSE IF element at position i of board EQUALS three THEN  
                    SET element at position i of board TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
