CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF values EQUALS an empty list  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode initialized with element at position one of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element obtained by removing from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO a new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO a new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree WITH PARAMETERS p.left q.left AND the result of calling is_same_tree WITH PARAMETERS p.right q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree WITH PARAMETERS self root  
        FUNCTION dfs WITH PARAMETERS node  
            IF node IS None  
                RETURN positive infinity NEGATIVE infinity zero  
            END IF  
            SET left_min left_max left_size TO the result of calling dfs WITH PARAMETERS node.left  
            SET right_min right_max right_size TO the result of calling dfs WITH PARAMETERS node.right  
            IF left_max IS LESS THAN node.val AND node.val IS LESS THAN right_min  
                RETURN the lesser of node.val AND left_min the greater of node.val AND right_max left_size PLUS right_size PLUS one  
            END IF  
            RETURN negative infinity positive infinity the greater of left_size AND right_size  
        END FUNCTION  
        RETURN the third element of the result of calling dfs WITH PARAMETERS root  
    END FUNCTION  
END CLASS
