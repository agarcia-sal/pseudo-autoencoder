CLASS Solution  
    FUNCTION possibleToStamp(grid, stampHeight, stampWidth)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero of grid  
        
        FUNCTION prefix_sum(matrix)  
            SET p TO a two dimensional list of zeroes with the number of rows equal TO m PLUS one and the number of columns equal TO n PLUS one  
            FOR r FROM one TO m  
                FOR c FROM one TO n  
                    SET element_at_row_minus_one_column_minus_one_of_matrix TO element at position r MINUS one of matrix AT position c MINUS one  
                    SET p_at_position_r_and_c TO element_at_row_minus_one_column_minus_one_of_matrix PLUS p at position r MINUS one and c PLUS p at position r and c MINUS one MINUS p at position r MINUS one and c MINUS one  
                END FOR  
            END FOR  
            RETURN p  
        END FUNCTION  
        
        FUNCTION submatrix_sum(p, r1, c1, r2, c2)  
            SET sum_value TO p at position r2 PLUS one and c2 PLUS one MINUS p at position r2 PLUS one and c1 MINUS p at position r1 and c2 PLUS one PLUS p at position r1 and c1  
            RETURN sum_value  
        END FUNCTION  
        
        SET p_grid TO the result of prefix_sum with argument grid  
        
        SET p_stamps TO a two dimensional list of zeroes with the number of rows equal TO m PLUS one and the number of columns equal TO n PLUS one  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position r of grid AT position c EQUALS zero THEN  
                    IF r PLUS stampHeight LESS THAN OR EQUAL TO m AND c PLUS stampWidth LESS THAN OR EQUAL TO n THEN  
                        IF the result of submatrix_sum with arguments p_grid r c r PLUS stampHeight MINUS one c PLUS stampWidth MINUS one EQUALS zero THEN  
                            INCREMENT p_stamps at position r and c BY one  
                        END IF  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET p_stamps TO the result of prefix_sum with argument p_stamps  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position r of grid AT position c EQUALS zero THEN  
                    SET r_start TO the MAXIMUM OF zero AND r MINUS stampHeight PLUS one  
                    SET c_start TO the MAXIMUM OF zero AND c MINUS stampWidth PLUS one  
                    IF the result of submatrix_sum with arguments p_stamps r_start c_start r c EQUALS zero THEN  
                        RETURN False  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
