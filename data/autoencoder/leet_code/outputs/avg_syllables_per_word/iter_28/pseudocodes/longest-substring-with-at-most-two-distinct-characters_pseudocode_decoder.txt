CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        ASSIGN reference TO a dictionary with default integer zero values named char_count  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR each right FROM zero TO the length of s MINUS one  
            INCREMENT the count of element at position right of s in char_count BY one  
            
            WHILE the number of distinct keys in char_count GREATER THAN two  
                DECREMENT the count of element at position left of s in char_count BY one  
                IF the count of element at position left of s in char_count EQUALS zero  
                    REMOVE the entry for element at position left of s FROM char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_length TO the greater value BETWEEN max_length AND the result of right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
