CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count_of_values_less_than_or_equals_x TO zero  
            FOR each value IN nums  
                IF value LESS THAN OR EQUAL TO x  
                    INCREMENT count_of_values_less_than_or_equals_x BY one  
                END IF  
            END FOR  
            IF count_of_values_less_than_or_equals_x GREATER THAN x  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        RETURN the smallest index in the range from zero TO the length of nums MINUS one for which function f RETURNS true  
    END FUNCTION  
END CLASS
