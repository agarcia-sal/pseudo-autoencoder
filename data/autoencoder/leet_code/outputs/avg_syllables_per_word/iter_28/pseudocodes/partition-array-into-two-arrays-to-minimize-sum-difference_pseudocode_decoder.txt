CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY two  
        SET total_sum TO the SUM OF all elements IN nums  
        SET target TO total_sum DIVIDED BY two  
        SET min_diff TO positive infinity  
        
        SET left_half TO the ELEMENTS from position zero TO position n MINUS one OF nums  
        SET right_half TO the ELEMENTS from position n TO the END OF nums  
        
        SET left_sums TO a list containing empty maps for each index FROM zero TO n  
        SET right_sums TO a list containing empty maps for each index FROM zero TO n  
        
        FOR i FROM zero TO n  
            FOR each combination combo OF length i FROM left_half  
                SET subset_sum TO the SUM OF all elements IN combo  
                IF subset_sum NOT PRESENT AS a key IN left_sums at position i  
                    SET the value at key subset_sum IN left_sums at position i TO zero  
                END IF  
                INCREMENT the value at key subset_sum IN left_sums at position i BY one  
            END FOR  
            
            FOR each combination combo OF length i FROM right_half  
                SET subset_sum TO the SUM OF all elements IN combo  
                IF subset_sum NOT PRESENT AS a key IN right_sums at position i  
                    SET the value at key subset_sum IN right_sums at position i TO zero  
                END IF  
                INCREMENT the value at key subset_sum IN right_sums at position i BY one  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO n  
            SET left_values TO the sorted list of keys IN left_sums at position i  
            SET right_values TO the sorted list of keys IN right_sums at position n MINUS i  
            
            SET j TO zero  
            SET k TO the LENGTH OF right_values MINUS one  
            
            WHILE j LESS THAN the LENGTH OF left_values AND k GREATER THAN OR EQUAL TO zero  
                SET current_sum TO the element at position j OF left_values PLUS the element at position k OF right_values  
                SET min_diff TO the smaller value BETWEEN min_diff AND the absolute value of total_sum MINUS two MULTIPLIED BY current_sum  
                
                IF current_sum LESS THAN target  
                    INCREMENT j BY one  
                ELSE IF current_sum GREATER THAN target  
                    DECREMENT k BY one  
                ELSE  
                    RETURN zero  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
