CLASS Solution  
    FUNCTION nthMagicalNumber(n a b)  
        SET MOD TO ten raised to the power of nine PLUS one  
        
        FUNCTION gcd(x y)  
            WHILE y GREATER THAN zero  
                SET temporary_x TO y  
                SET temporary_y TO x MODULO y  
                SET x TO temporary_x  
                SET y TO temporary_y  
            END WHILE  
            RETURN x  
        END FUNCTION  
        
        FUNCTION lcm(x y)  
            RETURN x MULTIPLIED BY y DIVIDED BY gcd(x y)  
        END FUNCTION  
        
        SET L TO lcm(a b)  
        
        SET left TO one  
        SET right TO n MULTIPLIED BY the smaller value BETWEEN a AND b  
        
        WHILE left LESS THAN right  
            SET mid TO the sum of left AND right DIVIDED BY two  
            SET count TO mid DIVIDED BY a PLUS mid DIVIDED BY b MINUS mid DIVIDED BY L  
            
            IF count LESS THAN n  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left MODULO MOD  
    END FUNCTION  
END CLASS
