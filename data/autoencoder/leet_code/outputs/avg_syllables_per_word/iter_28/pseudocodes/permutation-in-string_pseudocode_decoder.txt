CLASS Solution  
    FUNCTION checkInclusion(s1, s2)  
        SET length_of_s1 TO the LENGTH OF s1  
        SET length_of_s2 TO the LENGTH OF s2  
        
        IF length_of_s1 GREATER THAN length_of_s2  
            RETURN false  
        END IF  
        
        SET count_s1 TO a FREQUENCY MAP OF characters IN s1  
        SET count_s2 TO a FREQUENCY MAP OF characters IN the SUBSTRING OF s2 FROM position zero TO position length_of_s1 MINUS one  
        
        IF count_s1 EQUALS count_s2  
            RETURN true  
        END IF  
        
        FOR index FROM length_of_s1 TO length_of_s2 MINUS one  
            INCREMENT the FREQUENCY OF the CHARACTER AT position index OF s2 IN count_s2 BY one  
            DECREMENT the FREQUENCY OF the CHARACTER AT position index MINUS length_of_s1 OF s2 IN count_s2 BY one  
            
            IF the FREQUENCY OF the CHARACTER AT position index MINUS length_of_s1 OF s2 IN count_s2 EQUALS zero  
                REMOVE the CHARACTER AT position index MINUS length_of_s1 OF s2 FROM count_s2  
            END IF  
            
            IF count_s1 EQUALS count_s2  
                RETURN true  
            END IF  
        END FOR  
        
        RETURN false  
    END FUNCTION  
END CLASS
