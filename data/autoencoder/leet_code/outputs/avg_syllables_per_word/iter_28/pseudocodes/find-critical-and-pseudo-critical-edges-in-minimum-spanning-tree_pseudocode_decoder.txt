CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of numbers from zero TO n MINUS one  
        SET self.rank TO list of ones with length n  
    END FUNCTION

    FUNCTION find(self, u)  
        IF element at position u of self.parent NOT EQUALS u THEN  
            SET element at position u of self.parent TO self.find(element at position u of self.parent)  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION

    FUNCTION union(self, u, v)  
        SET rootU TO self.find(u)  
        SET rootV TO self.find(v)  

        IF rootU NOT EQUALS rootV THEN  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank THEN  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank THEN  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS

CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)  
        FOR each index i AND element edge IN edges  
            APPEND i TO edge  
        END FOR

        SORT edges BY the third element of each edge in ascending order  

        FUNCTION mst(exclude, include)  
            SET uf TO new instance of UnionFind with parameter n  
            SET weight TO zero  

            IF include IS NOT None THEN  
                CALL uf.union with first element of include AND second element of include  
                INCREMENT weight BY third element of include  
            END IF  

            FOR each element u, v, w, idx IN edges  
                IF idx EQUALS exclude THEN  
                    CONTINUE to next iteration  
                END IF  

                IF uf.union(u, v) IS True THEN  
                    INCREMENT weight BY w  
                END IF  
            END FOR  

            SET root TO uf.find(zero)  

            FOR each index i FROM zero TO n MINUS one  
                IF uf.find(i) NOT EQUALS root THEN  
                    RETURN infinity value  
                END IF  
            END FOR  

            RETURN weight  
        END FUNCTION  

        SET mst_weight TO result of mst called with no parameters  

        SET critical TO empty list  
        SET pseudo_critical TO empty list  

        FOR each element u, v, w, idx IN edges  
            IF mst called with exclude equal to idx GREATER THAN mst_weight THEN  
                APPEND idx TO critical  
            ELSE IF mst called with include equal to tuple of u, v, w EQUALS mst_weight THEN  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR  

        RETURN list containing critical AND pseudo_critical  
    END FUNCTION  
END CLASS
