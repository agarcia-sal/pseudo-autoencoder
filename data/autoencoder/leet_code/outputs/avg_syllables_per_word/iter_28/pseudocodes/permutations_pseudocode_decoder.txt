CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF the LENGTH OF remaining EQUALS zero DOES NOT HOLD  
                FOR index FROM zero TO the LENGTH OF remaining MINUS one  
                    SET new_path TO path PLUS element at position index of remaining  
                    SET new_remaining TO concatenation of elements from position zero TO position index MINUS one of remaining PLUS elements from position index PLUS one TO the end of remaining  
                    CALL backtrack WITH new_path AND new_remaining  
                END FOR  
            ELSE  
                APPEND path TO permutations  
                RETURN  
            END IF  
        END FUNCTION  

        SET permutations TO empty list  
        CALL backtrack WITH empty list AND nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
