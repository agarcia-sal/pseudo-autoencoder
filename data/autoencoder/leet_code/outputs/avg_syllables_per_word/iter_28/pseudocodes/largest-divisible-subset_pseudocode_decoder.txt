CLASS Solution  
    FUNCTION largestDivisibleSubset(nums)  
        SORT nums IN ASCENDING ORDER  
        SET n TO the LENGTH OF nums  
        SET f TO a list of LENGTH n with each element SET TO one  
        SET k TO zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF the element at position i of nums MODULO the element at position j of nums EQUALS zero  
                    SET the element at position i of f TO the GREATER VALUE BETWEEN the element at position i of f AND the element at position j of f PLUS one  
                END IF  
            END FOR  
            IF the element at position k of f LESS THAN the element at position i of f  
                SET k TO i  
            END IF  
        END FOR  
        SET m TO the element at position k of f  
        SET i TO k  
        SET ans TO an empty list  
        WHILE m GREATER THAN zero  
            IF the element at position k of nums MODULO the element at position i of nums EQUALS zero AND the element at position i of f EQUALS m  
                APPEND the element at position i of nums TO ans  
                SET k TO i  
                DECREMENT m BY one  
            END IF  
            DECREMENT i BY one  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
