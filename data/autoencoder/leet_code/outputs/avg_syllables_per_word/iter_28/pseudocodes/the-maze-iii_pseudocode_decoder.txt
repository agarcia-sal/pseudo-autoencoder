CLASS Solution  
    FUNCTION findShortestWay(maze, ball, hole)  
        SET directions TO list of tuples containing zero and one with right direction string then one and zero with down direction string then zero and negative one with left direction string then negative one and zero with up direction string  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET start TO a tuple containing the values of ball  
        SET hole TO a tuple containing the values of hole  
        SET pq TO a list containing a tuple of zero distance empty string path and start position  
        SET visited TO an empty set  
        
        WHILE pq holds  
            REMOVE and RETURN the smallest element from pq and ASSIGN to dist path and a tuple containing x y  
            
            IF x EQUALS the first element of hole AND y EQUALS the second element of hole  
                RETURN path  
            END IF  
            
            IF a tuple containing x y IS IN visited  
                CONTINUE to the next iteration of the loop  
            END IF  
            
            ADD the tuple containing x y TO visited  
            
            FOR each dx dy direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE the sum of new_x and dx IS GREATER THAN OR EQUAL TO zero AND the sum of new_x and dx IS LESS THAN m AND the sum of new_y and dy IS GREATER THAN OR EQUAL TO zero AND the sum of new_y and dy IS LESS THAN n AND element at position sum of new_x and dx of maze AT position sum of new_y and dy EQUALS zero  
                    SET new_x TO the sum of new_x and dx  
                    SET new_y TO the sum of new_y and dy  
                    INCREMENT count BY one  
                    IF new_x EQUALS the first element of hole AND new_y EQUALS the second element of hole  
                        BREAK the loop  
                    END IF  
                END WHILE  
                
                IF a tuple containing new_x new_y IS NOT IN visited  
                    ADD a tuple containing the sum of dist and count the concatenation of path and direction and a tuple containing new_x new_y TO the priority queue pq  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
