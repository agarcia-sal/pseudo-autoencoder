CLASS Solution  
    FUNCTION fractionToDecimal(numerator INTEGER, denominator INTEGER) RETURNS STRING  
        IF numerator LESS THAN ZERO AND denominator GREATER THAN OR EQUAL TO ZERO OR numerator GREATER THAN OR EQUAL TO ZERO AND denominator LESS THAN ZERO  
            SET sign TO the string containing the minus character  
        ELSE  
            SET sign TO the empty string  
        END IF  
        
        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  
        
        SET quotient TO the integer result of numerator DIVIDED BY denominator  
        SET remainder TO the integer remainder of numerator DIVIDED BY denominator  
        SET integer_part TO the string representation of quotient  
        
        IF remainder EQUALS zero  
            RETURN the concatenation of sign and integer_part  
        END IF  
        
        SET seen_remainders TO an empty dictionary mapping integers to integers  
        SET fractional_part TO an empty list of strings  
        
        WHILE remainder NOT EQUALS zero  
            IF remainder IS a key in seen_remainders  
                SET start TO the value in seen_remainders associated with remainder  
                SET non_repeating TO the concatenation of all elements in fractional_part from position zero UP TO position start MINUS ONE  
                SET repeating TO the concatenation of all elements in fractional_part from position start TO the last element  
                RETURN the concatenation of sign, integer_part, the string containing a period character, non_repeating, the string containing an opening parenthesis character, repeating, and the string containing a closing parenthesis character  
            END IF  
            
            ASSIGN in seen_remainders the key remainder with the value being the LENGTH OF fractional_part  
            
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO the integer result of remainder DIVIDED BY denominator  
            SET remainder TO the integer remainder of remainder DIVIDED BY denominator  
            APPEND the string representation of quotient TO fractional_part  
        END WHILE  
        
        RETURN the concatenation of sign, integer_part, the string containing a period character, and the concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
