CLASS Solution  
    FUNCTION longestConsecutive(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN zero  
        END IF  
        SET num_set TO the SET CONTAINING EACH element IN nums  
        SET longest_streak TO zero  
        FOR each num IN num_set  
            IF the element EQUALS num MINUS one DOES NOT EXIST IN num_set THEN  
                SET current_num TO num  
                SET current_streak TO one  
                WHILE the element EQUALS current_num PLUS one EXISTS IN num_set  
                    INCREMENT current_num BY one  
                    INCREMENT current_streak BY one  
                END WHILE  
                IF longest_streak LESS THAN current_streak THEN  
                    SET longest_streak TO current_streak  
                END IF  
            END IF  
        END FOR  
        RETURN longest_streak  
    END FUNCTION  
END CLASS
