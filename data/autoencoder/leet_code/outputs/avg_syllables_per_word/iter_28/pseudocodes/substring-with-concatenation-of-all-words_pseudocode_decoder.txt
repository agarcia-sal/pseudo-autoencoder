CLASS Solution  
    FUNCTION findSubstring(s parameter s AS string, words parameter words AS list of strings) RETURNS list of integers  
        IF s EQUALS empty string OR words EQUALS empty list OR element at position zero of words EQUALS empty string  
            RETURN empty list  
        END IF  
        
        SET word_length TO the LENGTH OF element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the COUNTER of words  
        SET result TO empty list  
        
        FOR i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO an EMPTY COUNTER  
            FOR j FROM i TO i PLUS total_length MINUS one STEP word_length  
                SET word TO substring from position j TO position j PLUS word_length MINUS one OF s  
                INCREMENT the COUNT of word in seen BY one  
                IF the COUNT of word in seen GREATER THAN the COUNT of word in word_count  
                    BREAK the inner FOR loop  
                END IF  
            END FOR  
            IF the inner FOR loop did NOT BREAK  
                APPEND i TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
