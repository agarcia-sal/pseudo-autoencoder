CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO dictionary mapping each unique number in nums TO its frequency  
        SET dp TO list of zeroes with the LENGTH OF two raised to the power of ten  
        SET the element at position zero of dp TO one  
        
        FOR each num IN the KEYS OF count  
            IF num EQUALS one THEN  
                SET the element at position zero of dp TO the element at position zero of dp MULTIPLIED BY two raised to the power of the frequency of one in count  
                CONTINUE to next iteration  
            END IF  
            
            IF num NOT IN the KEYS OF prime_factors THEN  
                CONTINUE to next iteration  
            END IF  
            
            SET factors TO the value at key num in prime_factors  
            SET mask TO zero  
            
            FOR each prime IN factors  
                SET mask TO mask PLUS two raised to the power of prime MINUS one  
            END FOR  
            
            FOR m FROM two raised to the power of ten MINUS one DOWN TO zero  
                IF the result of bitwise AND between m and mask EQUALS zero THEN  
                    SET the element at position of bitwise OR between m and mask of dp TO the sum of the element at position of bitwise OR between m and mask of dp AND the product of the element at position m of dp AND the frequency of num in count MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of elements in dp starting from position one TO the end MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
