CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self val TO val  
        SET self next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode element at position one of values  
    SET p TO head  
    FOR each val IN elements of values from position two TO end  
        SET node TO new ListNode val  
        SET p next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN self val of p1 EQUALS self val of p2 AND is_same_list(self next of p1 self next of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList(self head)  
        IF head EQUALS None OR self next of head EQUALS None OR self next of self next of head EQUALS None  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO self next of head  
        SET even_head TO even  
        WHILE even IS NOT None AND self next of even IS NOT None  
            SET self next of odd TO self next of even  
            SET odd TO self next of odd  
            SET self next of even TO self next of odd  
            SET even TO self next of even  
        END WHILE  
        SET self next of odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
