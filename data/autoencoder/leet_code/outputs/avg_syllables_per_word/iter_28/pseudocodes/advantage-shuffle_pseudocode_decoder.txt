CLASS Solution  
    FUNCTION advantageCount(nums1, nums2)  
        SORT nums1 IN ascending order  
        SET nums2_with_indices TO list of pairs where each pair consists of each element of nums2 and its corresponding index IN nums2  
        SORT nums2_with_indices IN descending order based on the first element of each pair  
        SET result TO a list of zeros with the same length as nums1  
        SET left TO zero  
        SET right TO the length of nums1 MINUS one  
        FOR each pair consisting of value and index IN nums2_with_indices  
            IF element at position right of nums1 IS GREATER THAN value THEN  
                SET element at position index of result TO element at position right of nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index of result TO element at position left of nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
