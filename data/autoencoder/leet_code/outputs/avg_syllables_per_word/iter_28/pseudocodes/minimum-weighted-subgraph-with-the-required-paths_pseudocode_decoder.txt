CLASS Solution  
    FUNCTION minimumWeight(n src1 src2 dest edges)  
        FUNCTION dijkstra(graph start)  
            SET distances TO a list of length n with each element set to infinity  
            SET element at position start of distances TO zero  
            SET heap TO a list containing a pair of zero and start  
            
            WHILE heap holds  
                SET current_distance TO the first element of the pair returned by removing the smallest element from heap  
                SET current_node TO the second element of the pair returned by removing the smallest element from heap  
                
                IF current_distance GREATER THAN element at position current_node of distances  
                    CONTINUE to the next iteration of the WHILE loop  
                END IF  
                
                FOR each pair of neighbor and weight IN the element at position current_node of graph  
                    SET distance TO current_distance PLUS weight  
                    
                    IF distance LESS THAN element at position neighbor of distances  
                        SET element at position neighbor of distances TO distance  
                        ADD the pair of distance and neighbor TO heap  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN distances  
        END FUNCTION  
        
        SET graph TO a list of length n with each element set to an empty list  
        SET reverse_graph TO a list of length n with each element set to an empty list  
        
        FOR each triple of u v w IN edges  
            APPEND the pair of v and w TO element at position u of graph  
            APPEND the pair of u and w TO element at position v of reverse_graph  
        END FOR  
        
        SET dist_from_src1 TO dijkstra called with graph and src1  
        SET dist_from_src2 TO dijkstra called with graph and src2  
        SET dist_to_dest TO dijkstra called with reverse_graph and dest  
        
        SET min_weight TO infinity  
        
        FOR index i FROM zero TO n MINUS one  
            IF element at position i of dist_from_src1 NOT EQUALS infinity AND element at position i of dist_from_src2 NOT EQUALS infinity AND element at position i of dist_to_dest NOT EQUALS infinity  
                SET min_weight TO the smaller value of min_weight AND the sum of element at position i of dist_from_src1 PLUS element at position i of dist_from_src2 PLUS element at position i of dist_to_dest  
            END IF  
        END FOR  
        
        IF min_weight EQUALS infinity  
            RETURN minus one  
        ELSE  
            RETURN min_weight  
        END IF  
    END FUNCTION  
END CLASS
