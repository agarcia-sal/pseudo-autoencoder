CLASS Solution  
    FUNCTION isTransformable(s parameter s OF TYPE string, t parameter t OF TYPE string) RETURNS boolean  
        CREATE a default dictionary named position_map with default type list  
        FOR each index i and character c IN s with their enumeration  
            CONVERT character c TO integer named digit_value  
            ADD index i TO the heap list in position_map for digit_value  
        END FOR  
        
        FOR each index i and character c IN t with their enumeration  
            CONVERT character c TO integer named current_digit  
            IF the heap list in position_map for current_digit IS empty  
                RETURN False  
            END IF  
            FOR each integer smaller_digit FROM zero TO current_digit MINUS one  
                IF the heap list in position_map for smaller_digit IS NOT empty AND the first element in the heap list for smaller_digit IS LESS THAN the first element in the heap list for current_digit  
                    RETURN False  
                END IF  
            END FOR  
            REMOVE the smallest element FROM the heap list in position_map for current_digit  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
