CLASS Solution  
    FUNCTION maximumMinutes(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO list containing the pairs one and zero MINUS one and zero ZERO and one ZERO and MINUS one  
        
        FUNCTION can_escape(wait)  
            SET fire_times TO a two dimensional list with m rows and n columns FILLED WITH infinity  
            SET queue TO an empty double-ended queue  
            
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    IF element at position i of grid EQUALS one  
                        APPEND the tuple of i j and zero TO queue  
                        SET element at position i and j of fire_times TO zero  
                    END IF  
                END FOR  
            END FOR  
            
            WHILE queue is not empty  
                SET x y t TO the element REMOVED FROM THE FRONT OF queue  
                FOR each pair dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx and ny of grid EQUALS zero AND element at position nx and ny of fire_times EQUALS infinity  
                        SET element at position nx and ny of fire_times TO t PLUS one  
                        APPEND the tuple of nx ny t PLUS one TO queue  
                    END IF  
                END FOR  
            END WHILE  
            
            SET queue TO a double-ended queue containing the tuple zero zero and wait  
            SET visited TO a set containing the tuple zero zero  
            
            WHILE queue is not empty  
                SET x y t TO the element REMOVED FROM THE FRONT OF queue  
                FOR each pair dx dy IN directions  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the tuple nx ny is not in visited AND element at position nx and ny of grid EQUALS zero  
                        IF nx EQUALS m MINUS one AND ny EQUALS n MINUS one  
                            IF element at position nx and ny of fire_times GREATER THAN OR EQUAL TO t PLUS one  
                                RETURN True  
                            ELSE  
                                RETURN False  
                            END IF  
                        END IF  
                        IF element at position nx and ny of fire_times GREATER THAN t PLUS one  
                            ADD the tuple nx ny TO visited  
                            APPEND the tuple nx ny t PLUS one TO queue  
                        END IF  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN False  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO m MULTIPLIED BY n  
        SET result TO negative one  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the integer division result of left PLUS right DIVIDED BY two  
            IF can_escape(mid)  
                SET result TO mid  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        
        IF result EQUALS m MULTIPLIED BY n  
            RETURN one followed by zero MULTIPLIED BY nine  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
