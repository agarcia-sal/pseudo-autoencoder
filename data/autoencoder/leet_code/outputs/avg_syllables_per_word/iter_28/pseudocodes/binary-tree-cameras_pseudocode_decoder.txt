CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS no value, right EQUALS no value)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS no value or the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO a new instance of TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty deque collection  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS no value  
            SET node.left TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS no value  
            SET node.right TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS no value AND q EQUALS no value  
        RETURN true  
    ELSE IF p EQUALS no value OR q EQUALS no value  
        RETURN false  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the logical AND of the result of is_same_tree with the left attribute of p AND the left attribute of q AND the result of is_same_tree with the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION minCameraCover(self, root)  
        FUNCTION dfs(node)  
            IF node EQUALS no value  
                RETURN zero, zero, infinity  
            END IF  
            SET left_no_camera, left_monitored, left_camera TO the result of dfs with the left attribute of node  
            SET right_no_camera, right_monitored, right_camera TO the result of dfs with the right attribute of node  
            SET no_camera TO left_monitored PLUS right_monitored  
            SET monitored TO the minimum of the following three values  
                left_camera PLUS the minimum of right_monitored AND right_camera  
                right_camera PLUS the minimum of left_monitored AND left_camera  
                left_camera PLUS right_camera  
            SET camera TO one PLUS the minimum of left_no_camera, left_monitored, left_camera PLUS the minimum of right_no_camera, right_monitored, right_camera  
            RETURN no_camera, monitored, camera  
        END FUNCTION  
        SET root_no_camera, root_monitored, root_camera TO the result of dfs with root  
        RETURN the minimum of root_monitored AND root_camera  
    END FUNCTION  
END CLASS
