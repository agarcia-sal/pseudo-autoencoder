CLASS Solution  
    FUNCTION consecutiveNumbersSum(n)  
        SET count TO zero  
        FOR k FROM one TO n PLUS one MINUS one  
            IF two MULTIPLIED BY n MODULO k EQUALS zero  
                SET m TO the result of two MULTIPLIED BY n DIVIDED BY k MINUS k PLUS one DIVIDED BY two  
                IF m GREATER THAN zero AND m EQUALS the integer part of m  
                    INCREMENT count BY one  
                END IF  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
