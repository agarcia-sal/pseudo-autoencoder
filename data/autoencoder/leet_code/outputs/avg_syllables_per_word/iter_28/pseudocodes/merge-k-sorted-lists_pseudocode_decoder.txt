CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val SET TO zero next SET TO None  
        SET self val TO val  
        SET self next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values EQUALS an empty list  
        RETURN None  
    END IF  
    SET head TO new instance of ListNode with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values from position one TO the last position  
        SET node TO new instance of ListNode with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND the result of is_same_list with parameters the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists WITH PARAMETERS self lists  
        SET min_heap TO an empty list  
        FOR each index i AND element lst IN lists with their positions  
            IF lst DOES NOT EQUAL None  
                ADD the tuple consisting of the value of lst the index i AND lst TO min_heap maintaining heap order by the first element of the tuple  
            END IF  
        END FOR  
        SET dummy TO new instance of ListNode  
        SET current TO dummy  
        WHILE min_heap IS NOT empty  
            REMOVE the tuple with the smallest first element from min_heap AND ASSIGN its elements TO val i node  
            SET the next element of current TO node  
            SET current TO the next element of current  
            IF the next element of node DOES NOT EQUAL None  
                ADD the tuple consisting of the value of the next element of node the index i AND the next element of node TO min_heap maintaining heap order by the first element of the tuple  
            END IF  
        END WHILE  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
