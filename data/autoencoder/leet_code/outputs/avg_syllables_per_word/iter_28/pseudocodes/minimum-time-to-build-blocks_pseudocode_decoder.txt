CLASS Solution  
    FUNCTION minBuildTime(blocks, split)  
        TRANSFORM blocks INTO a minimum priority queue structure  
        WHILE the LENGTH OF blocks GREATER THAN one  
            EXTRACT the smallest element FROM blocks AND ASSIGN TO first  
            EXTRACT the next smallest element FROM blocks AND ASSIGN TO second  
            SET combined_time TO split PLUS second  
            INSERT combined_time INTO blocks  
        END WHILE  
        RETURN the element at position one of blocks  
    END FUNCTION  
END CLASS
