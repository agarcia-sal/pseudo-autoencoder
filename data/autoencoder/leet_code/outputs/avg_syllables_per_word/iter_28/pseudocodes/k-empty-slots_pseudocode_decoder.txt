CLASS Solution  
    FUNCTION kEmptySlots(bulbs, k)  
        SET bloom days TO an empty list  
        FOR each day and position IN the enumeration of bulbs starting from one  
            SET i TO the insertion position in bloom days where position can be placed maintaining sorted order  
            IF i greater than zero AND position MINUS the element at position i MINUS one of bloom days MINUS one EQUALS k  
                RETURN day  
            END IF  
            IF i less than the LENGTH OF bloom days AND the element at position i of bloom days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position INTO bloom days at position i  
        END FOR  
        RETURN minus one  
    END FUNCTION  
END CLASS
