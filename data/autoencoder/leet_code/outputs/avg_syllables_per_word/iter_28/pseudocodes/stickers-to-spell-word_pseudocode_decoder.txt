CLASS Solution  
    FUNCTION minStickers(stickers AS List of strings, target AS string) RETURNS integer  
        SET stickers TO list of frequency counters of each sticker string in stickers  
        SORT stickers BY the negative value of the maximum frequency count among the frequencies in each sticker  
        SET queue TO a double ended queue containing a single element which is a pair of target string and zero  
        SET memo TO a dictionary mapping target string TO zero  
        WHILE the queue contains elements  
            REMOVE the first element from queue and ASSIGN it TO current_target and used_stickers  
            FOR each sticker IN stickers  
                IF the first character of current_target is NOT IN sticker  
                    CONTINUE to next iteration of loop  
                END IF  
                SET remaining TO a frequency counter of current_target  
                DECREMENT remaining frequencies BY the frequencies in sticker  
                SET new_target TO the string formed by sorting and joining all characters repeated according to their frequencies in remaining  
                IF new_target is empty  
                    RETURN used_stickers PLUS one  
                END IF  
                IF new_target is NOT a key in memo  
                    SET memo at new_target TO used_stickers PLUS one  
                    APPEND a pair of new_target and used_stickers PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
