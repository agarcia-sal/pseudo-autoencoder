CLASS Solution  
    FUNCTION minDistance(houses k)  
        SORT houses IN ascending order  
        SET n TO the LENGTH OF houses  
        INITIALIZE cost AS a two dimensional list with dimensions n by n FILLED WITH zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost at position i at position j TO cost at position i PLUS one at position j MINUS one PLUS element at position j of houses MINUS element at position i of houses  
            END FOR  
        END FOR  
        INITIALIZE dp AS a two dimensional list with dimensions n PLUS one by k PLUS one FILLED WITH infinite value  
        SET element at position zero at position zero of dp TO zero  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET dp at position i at position j TO THE MINIMUM OF dp at position i at position j AND dp at position p at position j MINUS one PLUS cost at position p at position i MINUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp at position n at position k  
    END FUNCTION  
END CLASS
