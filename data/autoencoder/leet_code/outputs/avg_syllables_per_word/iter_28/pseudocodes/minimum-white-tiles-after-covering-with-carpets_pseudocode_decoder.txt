CLASS Solution  
    FUNCTION minimumWhiteTiles(floor numCarpets carpetLen)  
        SET n TO the LENGTH OF floor  
        INITIALIZE dp AS a two dimensional list of zero VALUES with the HEIGHT of n PLUS one and WIDTH of numCarpets PLUS one  
        FOR i FROM one TO n  
            SET dp at position i zero TO dp at position i MINUS one zero PLUS one IF the element at position i MINUS one of floor EQUALS one ELSE zero  
        END FOR  
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp at position i j TO dp at position i MINUS one j PLUS one IF the element at position i MINUS one of floor EQUALS one ELSE zero  
                IF dp at position i j IS GREATER THAN dp at position the MAXIMUM of zero AND i MINUS carpetLen j MINUS one  
                    SET dp at position i j TO dp at position the MAXIMUM of zero AND i MINUS carpetLen j MINUS one  
                END IF  
            END FOR  
        END FOR  
        RETURN dp at position n numCarpets  
    END FUNCTION  
END CLASS
