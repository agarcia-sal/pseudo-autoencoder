CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO an empty dictionary with default integer zero  
        FOR each cpdomain IN cpdomains  
            SET count_string AND domain_string TO the result of splitting cpdomain by the space character  
            SET count TO count_string converted to an integer  
            SET subdomains TO the result of splitting domain_string by the period character  
            
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one  
                SET subdomain TO the concatenation of the elements of subdomains from position index TO the last position separated by the period character  
                INCREMENT the value at key subdomain in domain_count BY count  
            END FOR  
        END FOR  
        
        SET result_list TO an empty list  
        FOR each key_domain AND value_count IN the key value pairs of domain_count  
            SET entry_string TO the text string of value_count concatenated with a space character concatenated with key_domain  
            APPEND entry_string TO result_list  
        END FOR  
        
        RETURN result_list  
    END FUNCTION  
END CLASS
