CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO the list of the first elements of each element in flowers  
        SORT start_times IN ascending order  
        SET end_times TO the list of the second elements of each element in flowers  
        SORT end_times IN ascending order  
        SET result TO an empty list  
        FOR each person IN people  
            SET started TO the number of elements in start_times that are LESS THAN OR EQUAL TO person using bisect right method  
            SET ended TO the number of elements in end_times that are LESS THAN person using bisect left method  
            APPEND started MINUS ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
