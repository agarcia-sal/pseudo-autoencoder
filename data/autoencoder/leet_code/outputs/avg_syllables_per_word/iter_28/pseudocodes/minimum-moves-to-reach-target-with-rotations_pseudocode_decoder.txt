CLASS Solution  
    FUNCTION minimumMoves(grid)  
        SET n TO the LENGTH OF grid  
        ASSIGN reference TO queue WITH a new empty double ended queue containing a tuple of the integer zero and integer zero and integer zero and integer one and integer zero  
        ASSIGN reference TO visited WITH a new empty set containing a tuple of the integer zero and integer zero and integer zero and integer one  

        WHILE the queue is not empty  
            REMOVE and ASSIGN the first element OF queue TO r1 c1 r2 c2 steps  

            IF r1 EQUALS n MINUS one AND c1 EQUALS n MINUS two AND r2 EQUALS n MINUS one AND c2 EQUALS n MINUS one THEN  
                RETURN steps  
            END IF  

            IF c2 PLUS one LESS THAN n AND the element at position r1 and position c1 PLUS one of grid EQUALS zero AND the element at position r2 and position c2 PLUS one of grid EQUALS zero AND the tuple of r1 and c1 PLUS one and r2 and c2 PLUS one is NOT IN visited THEN  
                ADD the tuple of r1 and c1 PLUS one and r2 and c2 PLUS one TO visited  
                APPEND the tuple of r1 and c1 PLUS one and r2 and c2 PLUS one and steps PLUS one TO queue  
            END IF  

            IF r2 PLUS one LESS THAN n AND the element at position r1 PLUS one and position c1 of grid EQUALS zero AND the element at position r2 PLUS one and position c2 of grid EQUALS zero AND the tuple of r1 PLUS one and c1 and r2 PLUS one and c2 is NOT IN visited THEN  
                ADD the tuple of r1 PLUS one and c1 and r2 PLUS one and c2 TO visited  
                APPEND the tuple of r1 PLUS one and c1 and r2 PLUS one and c2 and steps PLUS one TO queue  
            END IF  

            IF r1 EQUALS r2 AND r1 PLUS one LESS THAN n AND the element at position r1 PLUS one and position c1 of grid EQUALS zero AND the element at position r1 PLUS one and position c2 of grid EQUALS zero AND the tuple of r1 and c1 and r1 PLUS one and c1 is NOT IN visited THEN  
                ADD the tuple of r1 and c1 and r1 PLUS one and c1 TO visited  
                APPEND the tuple of r1 and c1 and r1 PLUS one and c1 and steps PLUS one TO queue  
            END IF  

            IF c1 EQUALS c2 AND c1 PLUS one LESS THAN n AND the element at position r1 and position c1 PLUS one of grid EQUALS zero AND the element at position r2 and position c2 PLUS one of grid EQUALS zero AND the tuple of r1 and c1 and r1 and c1 PLUS one is NOT IN visited THEN  
                ADD the tuple of r1 and c1 and r1 and c1 PLUS one TO visited  
                APPEND the tuple of r1 and c1 and r1 and c1 PLUS one and steps PLUS one TO queue  
            END IF  
        END WHILE  

        RETURN negative one  
    END FUNCTION  
END CLASS
