CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO a list of twenty-six elements each equal to negative one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each index i FROM zero TO the length of s MINUS one  
            SET char TO element at position i of s  
            SET index TO the Unicode code point of char MINUS the Unicode code point of the character a  
            INCREMENT current_appeal BY i MINUS the element at position index of last  
            INCREMENT total_appeal BY current_appeal  
            SET element at position index of last TO i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
