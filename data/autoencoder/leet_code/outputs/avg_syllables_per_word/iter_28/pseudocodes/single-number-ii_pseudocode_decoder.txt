CLASS Solution  
    FUNCTION singleNumber(nums)  
        SET ones TO zero  
        SET twos TO zero  
        FOR each num IN nums  
            SET ones TO the result of the bitwise exclusive OR of ones and num AND the bitwise negation of twos  
            SET twos TO the result of the bitwise exclusive OR of twos and num AND the bitwise negation of ones  
        END FOR  
        RETURN ones  
    END FUNCTION  
END CLASS
