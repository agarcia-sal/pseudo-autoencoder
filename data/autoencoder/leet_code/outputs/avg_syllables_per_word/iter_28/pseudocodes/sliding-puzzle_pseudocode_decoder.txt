CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuples where each tuple is each row converted from board  
        SET target TO tuple of tuple where first tuple is sequence one two three and second tuple is sequence four five zero  
        SET neighbors TO dictionary with keys zero to five and values sets of adjacent zero positions as described  
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position j of element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a new deque containing a tuple of start zero_pos and zero  
        SET visited TO a new set containing start  
        WHILE queue is not empty  
            REMOVE the leftmost element from queue and ASSIGN its elements respectively to state zero_pos moves  
            IF state EQUALS target  
                RETURN moves  
            END IF  
            FOR each neighbor IN neighbors at position zero_pos  
                SET new_state TO list of lists converted from state  
                SET i TO zero_pos DIVIDED BY three using integer division  
                SET j TO zero_pos MODULO three  
                SET ni TO neighbor DIVIDED BY three using integer division  
                SET nj TO neighbor MODULO three  
                SET temporary variable temp TO element at position j of element at position i of new_state  
                SET element at position j of element at position i of new_state TO element at position nj of element at position ni of new_state  
                SET element at position nj of element at position ni of new_state TO temporary variable temp  
                SET new_tuple_state TO tuple of tuples where each tuple is each row converted from new_state  
                IF new_tuple_state NOT IN visited  
                    ADD new_tuple_state TO visited  
                    APPEND tuple of new_tuple_state neighbor moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
