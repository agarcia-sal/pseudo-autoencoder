CLASS Solution  
    FUNCTION connectTwoGroups WITH PARAMETER cost  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF the element at position zero of cost  
        INITIALIZE dp AS a LIST of size size1 PLUS one WHERE EACH ELEMENT IS a LIST of size two RAISED TO THE power size2 WITH ALL ELEMENTS SET TO infinity  
        SET the element at position zero of dp AT position zero TO zero  
        
        FOR i FROM zero TO size1 MINUS one  
            FOR mask FROM zero TO two RAISED TO THE power size2 MINUS one  
                FOR j FROM zero TO size2 MINUS one  
                    SET new_mask TO mask OR two RAISED TO THE power j  
                    SET the element at position i PLUS one of dp AT position new_mask TO THE MINIMUM OF the element at position i PLUS one of dp AT position new_mask AND the element at position i of dp AT position mask PLUS the element at position i OF cost AT position j  
                END FOR  
                
                FOR j FROM zero TO size2 MINUS one  
                    IF the bitwise AND of mask AND two RAISED TO THE power j IS GREATER THAN zero  
                        SET the element at position i PLUS one of dp AT position mask TO THE MINIMUM OF the element at position i PLUS one of dp AT position mask AND the element at position i PLUS one of dp AT position mask XOR two RAISED TO THE power j PLUS the element at position i OF cost AT position j  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN the element at position size1 of dp AT position two RAISED TO THE power size2 MINUS one  
    END FUNCTION  
END CLASS
