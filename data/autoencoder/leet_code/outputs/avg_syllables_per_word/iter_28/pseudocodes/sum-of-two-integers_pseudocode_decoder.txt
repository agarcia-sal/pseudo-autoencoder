CLASS Solution  
    FUNCTION getSum(a b)  
        SET maximum TO the largest thirty two bits integer value  
        SET bit_mask TO the mask covering thirty two bits  

        WHILE b NOT EQUALS zero  
            SET carry TO the result of bitwise AND of a and b MULTIPLIED BY two  
            SET a TO the result of bitwise XOR of a and b AND bit_mask  
            SET b TO carry AND bit_mask  
        END WHILE  

        IF a LESS THAN OR EQUAL TO maximum  
            RETURN a  
        ELSE  
            RETURN the bitwise complement of the result of bitwise XOR of a and bit_mask  
        END IF  
    END FUNCTION  
END CLASS
