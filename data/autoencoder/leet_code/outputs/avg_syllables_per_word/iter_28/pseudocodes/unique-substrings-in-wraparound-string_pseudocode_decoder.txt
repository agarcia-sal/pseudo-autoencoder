CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a new dictionary mapping each character from a to z TO zero  
        SET current_length TO zero  
        FOR each index i FROM zero TO the length of s MINUS one  
            IF i is greater than zero AND (the difference between the character code of element at position i of s AND the character code of element at position i MINUS one of s EQUALS one OR element at position i of s EQUALS the character a AND element at position i MINUS one of s EQUALS the character z)  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET max_length at element at position i of s TO the greater value BETWEEN max_length at element at position i of s AND current_length  
        END FOR  
        RETURN the sum of all values in max_length  
    END FUNCTION  
END CLASS
