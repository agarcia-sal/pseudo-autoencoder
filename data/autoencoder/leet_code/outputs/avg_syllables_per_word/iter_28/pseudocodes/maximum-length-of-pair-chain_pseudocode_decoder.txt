CLASS Solution  
    FUNCTION findLongestChain(pairs)  
        SET pairs TO pairs SORTED BY element at position two of each pair  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each pair IN pairs  
            LET start BE element at position one of pair  
            LET end BE element at position two of pair  
            IF start GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
