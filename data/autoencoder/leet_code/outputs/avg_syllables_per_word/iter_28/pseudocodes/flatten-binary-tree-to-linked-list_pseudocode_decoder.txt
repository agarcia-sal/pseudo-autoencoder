CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        REMOVE the element at the front of queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left AND q.left AND the result of is_same_tree with parameters p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root EQUALS none THEN  
            RETURN  
        END IF  
        SET stack TO a new list containing root  
        WHILE the LENGTH OF stack GREATER THAN zero  
            REMOVE the last element of stack AND ASSIGN TO node  
            IF node.right NOT EQUALS none THEN  
                APPEND node.right TO stack  
            END IF  
            IF node.left NOT EQUALS none THEN  
                APPEND node.left TO stack  
            END IF  
            SET node.left TO none  
            IF the LENGTH OF stack GREATER THAN zero THEN  
                SET node.right TO the last element of stack  
            ELSE  
                SET node.right TO none  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
