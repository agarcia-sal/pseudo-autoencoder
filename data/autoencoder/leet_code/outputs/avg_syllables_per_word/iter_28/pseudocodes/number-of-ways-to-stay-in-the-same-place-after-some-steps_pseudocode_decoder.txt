CLASS Solution  
    FUNCTION numWays(steps, arrLen)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET max_index TO the smaller of the integer division of steps by two AND arrLen MINUS one  
        SET dp TO a new list of zeros of length max_index PLUS one  
        SET element at position zero of dp TO one  

        FOR each iteration FROM zero TO steps MINUS one  
            SET prev_dp TO a copy of dp  
            FOR each index i FROM zero TO max_index  
                SET element at position i of dp TO element at position i of prev_dp  
                IF index i GREATER THAN zero THEN  
                    SET element at position i of dp TO the sum of element at position i of dp AND element at position i MINUS one of prev_dp MODULO MOD  
                END IF  
                IF index i LESS THAN max_index THEN  
                    SET element at position i of dp TO the sum of element at position i of dp AND element at position i PLUS one of prev_dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  

        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
