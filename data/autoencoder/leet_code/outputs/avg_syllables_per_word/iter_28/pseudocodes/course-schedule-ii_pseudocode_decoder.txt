CLASS Solution  
    FUNCTION findOrder(numCourses parameter, prerequisites parameter)  
        SET adj_list TO empty mapping from course to empty list for each course from zero TO numCourses MINUS one  
        SET in_degrees TO list of zeros with the LENGTH OF numCourses  
        FOR each pair of destination_course AND source_course IN prerequisites parameter  
            APPEND destination_course TO the list at key source_course in adj_list  
            INCREMENT the element at position destination_course of in_degrees BY one  
        END FOR  
        SET queue TO a deque of courses from zero TO numCourses MINUS one WHERE the element at position course of in_degrees EQUALS zero  
        SET topological_order TO empty list  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET current TO the removal of the first element from queue  
            APPEND current TO topological_order  
            FOR each neighbor IN the list at key current in adj_list  
                DECREMENT the element at position neighbor of in_degrees BY one  
                IF the element at position neighbor of in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        IF the LENGTH OF topological_order EQUALS numCourses parameter  
            RETURN topological_order  
        ELSE  
            RETURN empty list  
        END IF  
    END FUNCTION  
END CLASS
