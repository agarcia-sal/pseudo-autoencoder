CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF the LENGTH OF intervals EQUALS zero  
            RETURN zero  
        END IF  
        SORT intervals BY the end position of each interval  
        SET end TO the end position of the first interval in intervals  
        SET count TO zero  
        FOR index FROM one TO the LENGTH OF intervals MINUS one  
            IF the start position of the interval at position index IN intervals GREATER THAN OR EQUALS TO end  
                SET end TO the end position of the interval at position index IN intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
