CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a mapping from zero TO minus one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        FOR each index i AND character char IN enumeration of s  
            SET digit TO the integer value of character char  
            SET prefix_xor TO prefix_xor XOR the value of one SHIFTED LEFT BY digit  
            IF the key prefix_xor EXISTS IN prefix_map  
                SET max_length TO the greater value BETWEEN max_length AND index i MINUS the value at key prefix_xor in prefix_map  
            END IF  
            FOR each integer j FROM zero TO nine  
                SET flipped TO prefix_xor XOR the value of one SHIFTED LEFT BY j  
                IF the key flipped EXISTS IN prefix_map  
                    SET max_length TO the greater value BETWEEN max_length AND index i MINUS the value at key flipped in prefix_map  
                END IF  
            END FOR  
            IF the key prefix_xor DOES NOT EXIST IN prefix_map  
                SET the value at key prefix_xor in prefix_map TO index i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
