CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  
        
        SET below_twenty TO list of words representing numbers from empty string at position zero followed by the words One Two Three Four Five Six Seven Eight Nine Ten Eleven Twelve Thirteen Fourteen Fifteen Sixteen Seventeen Eighteen Nineteen in order  
        SET tens TO list of words representing numbers from empty string at position zero followed by empty string at position one followed by the words Twenty Thirty Forty Fifty Sixty Seventy Eighty Ninety in order  
        SET thousands TO list of words representing empty string at position zero followed by Thousand Million Billion in order  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN empty string  
            ELSE IF n LESS THAN twenty  
                RETURN the element at position n of below_twenty CONCATENATED WITH a single space  
            ELSE IF n LESS THAN one hundred  
                RETURN the element at position the whole part of n DIVIDED BY ten of tens CONCATENATED WITH a single space CONCATENATED WITH the result of calling helper with the remainder of n DIVIDED BY ten  
            ELSE  
                RETURN the element at position the whole part of n DIVIDED BY one hundred of below_twenty CONCATENATED WITH the word Hundred CONCATENATED WITH a single space CONCATENATED WITH the result of calling helper with the remainder of n DIVIDED BY one hundred  
            END IF  
        END FUNCTION  
        
        SET result TO empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero  
            IF the remainder of num DIVIDED BY one thousand NOT EQUALS zero  
                SET result TO the result of calling helper with the remainder of num DIVIDED BY one thousand CONCATENATED WITH the element at position i of thousands CONCATENATED WITH a single space CONCATENATED WITH result  
            END IF  
            SET num TO the whole part of num DIVIDED BY one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
