CLASS Solution  
    FUNCTION largestOverlap(img1 parameter, img2 parameter)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR each index i FROM zero TO n MINUS one  
            FOR each index j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of img1 EQUALS one  
                    APPEND the pair composed of i and j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR each index i FROM zero TO n MINUS one  
            FOR each index j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of img2 EQUALS one  
                    APPEND the pair composed of i and j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a mapping from pairs of integers to integers with default value zero  
        FOR each pair consisting of x1 and y1 IN points1  
            FOR each pair consisting of x2 and y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value in overlap_count associated with the pair composed of dx and dy BY one  
            END FOR  
        END FOR  
        SET maximum_overlap TO zero  
        FOR each value IN the collection of values in overlap_count  
            IF value GREATER THAN maximum_overlap  
                SET maximum_overlap TO value  
            END IF  
        END FOR  
        RETURN maximum_overlap  
    END FUNCTION  
END CLASS
