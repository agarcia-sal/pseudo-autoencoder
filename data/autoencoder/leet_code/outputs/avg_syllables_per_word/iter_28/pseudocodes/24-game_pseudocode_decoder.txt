CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the element at position zero of nums MINUS twenty four LESS THAN one times ten TO THE POWER OF negative six  
            END IF  
            
            FOR i FROM zero TO the LENGTH OF nums MINUS one  
                FOR j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO an empty list  
                        FOR k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND the element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR each operation IN the list of division operation multiplication operation addition operation subtraction operation  
                            IF (the operation IS the addition operation OR the operation IS the multiplication operation) AND j GREATER THAN i  
                                CONTINUE to the next iteration of the operation loop  
                            END IF  
                            
                            IF the operation IS NOT the division operation OR the element at position j of nums NOT EQUALS zero  
                                APPEND the result of applying the operation TO the element at position i of nums AND the element at position j of nums TO new_nums  
                                IF solve(new_nums)  
                                    RETURN True  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
