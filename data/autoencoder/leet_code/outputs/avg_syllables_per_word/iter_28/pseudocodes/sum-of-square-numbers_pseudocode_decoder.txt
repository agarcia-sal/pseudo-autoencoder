CLASS Solution  
    FUNCTION judgeSquareSum WITH PARAMETER c  
        SET left TO zero  
        SET right TO the integer value of the square root of c  
        WHILE left LESS THAN OR EQUAL TO right  
            SET current_sum TO left MULTIPLIED BY left PLUS right MULTIPLIED BY right  
            IF current_sum EQUALS c  
                RETURN True  
            ELSE IF current_sum LESS THAN c  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
