CLASS Solution  
    FUNCTION numDistinct(s, t)  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
        
        CREATE dp AS a two dimensional array of the size of m PLUS one by n PLUS one filled with zeroes  
        
        FOR i FROM zero TO m  
            SET element at position i and position zero of dp TO one  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one of s EQUALS character at position j MINUS one of t  
                    SET element at position i and position j of dp TO element at position i MINUS one and position j MINUS one of dp PLUS element at position i MINUS one and position j of dp  
                ELSE  
                    SET element at position i and position j of dp TO element at position i MINUS one and position j of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position m and position n of dp  
    END FUNCTION  
END CLASS
