CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO the concatenation of all elements in sentence separated by single space followed by a single space at the end  
        SET total_length TO the LENGTH OF s  
        SET position TO zero  
        FOR each iteration FROM one TO rows  
            INCREMENT position BY cols  
            IF element at position MODULO total_length of s EQUALS a single space character  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND element at position MINUS one MODULO total_length of s NOT EQUALS a single space character  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        RETURN the INTEGER DIVISION of position BY total_length  
    END FUNCTION  
END CLASS
