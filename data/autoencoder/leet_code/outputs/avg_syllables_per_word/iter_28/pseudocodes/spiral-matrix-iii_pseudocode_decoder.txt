CLASS Solution  
    FUNCTION spiralMatrixIII(rows rrows cols ccols rStart rrStart cStart ccStart) RETURNS list of list of integer  
        SET directions TO list of pairs representing zero one PLUS zero one MINUS zero one MINUS one zero  
        SET direction_index TO zero  
        SET steps TO zero  
        SET increment TO zero  
        SET visited TO empty list  
        SET x TO rrStart  
        SET y TO ccStart  
        
        WHILE the LENGTH OF visited IS LESS THAN rrows MULTIPLIED BY ccols  
            IF increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR each step FROM one TO steps  
                IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN rrows AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN ccols  
                    APPEND list containing x AND y TO visited  
                END IF  
                SET dx TO element at position zero of element at position direction_index of directions  
                SET dy TO element at position one of element at position direction_index of directions  
                SET x TO x PLUS dx  
                SET y TO y PLUS dy  
            END FOR  
            SET direction_index TO direction_index PLUS one  
            SET direction_index TO direction_index MODULO four  
            INCREMENT increment BY one  
        END WHILE  
        
        RETURN visited  
    END FUNCTION  
END CLASS
