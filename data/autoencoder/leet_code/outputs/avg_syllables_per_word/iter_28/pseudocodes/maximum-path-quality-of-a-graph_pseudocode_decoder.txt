CLASS Solution  
    FUNCTION maximalPathQuality(values parameter, edges parameter, maxTime parameter)  
        SET graph TO an empty mapping from keys to lists  
        FOR each u parameter v parameter time parameter IN edges parameter  
            APPEND a pair of neighbor v parameter and time time parameter TO the list associated with key u parameter IN graph  
            APPEND a pair of neighbor u parameter and time time parameter TO the list associated with key v parameter IN graph  
        END FOR  
        SET visited TO an empty set  
        SET max_quality TO zero  
        FUNCTION dfs(node parameter, current_time parameter, current_quality parameter)  
            IF node parameter EQUALS zero THEN  
                SET max_quality TO the greater value BETWEEN max_quality AND current_quality parameter  
            END IF  
            FOR each neighbor parameter time parameter IN the list associated with key node parameter IN graph  
                IF time parameter LESS THAN OR EQUAL TO current_time parameter THEN  
                    IF neighbor parameter NOT IN visited THEN  
                        ADD neighbor parameter TO visited  
                        CALL dfs WITH neighbor parameter WITH current_time parameter MINUS time parameter AND current_quality parameter PLUS the element at position neighbor parameter OF values parameter  
                        REMOVE neighbor parameter FROM visited  
                    ELSE  
                        CALL dfs WITH neighbor parameter WITH current_time parameter MINUS time parameter AND current_quality parameter  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
        ADD zero TO visited  
        CALL dfs WITH zero WITH maxTime parameter AND the element at position zero OF values parameter  
        RETURN max_quality  
    END FUNCTION  
END CLASS
