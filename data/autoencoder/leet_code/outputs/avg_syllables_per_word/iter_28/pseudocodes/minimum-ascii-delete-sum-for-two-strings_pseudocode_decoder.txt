CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        INITIALIZE dp AS a two dimensional list with m PLUS one rows and n PLUS one columns FILLED WITH zero  
        
        FOR j FROM one TO n  
            SET dp element at position zero and position j TO dp element at position zero and position j MINUS one PLUS the ASCII value of element at position j MINUS one of s2  
        END FOR  
        
        FOR i FROM one TO m  
            SET dp element at position i and position zero TO dp element at position i MINUS one and position zero PLUS the ASCII value of element at position i MINUS one of s1  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of s1 EQUALS element at position j MINUS one of s2 THEN  
                    SET dp element at position i and position j TO dp element at position i MINUS one and position j MINUS one  
                ELSE  
                    SET dp element at position i and position j TO the MINIMUM OF dp element at position i MINUS one and position j PLUS the ASCII value of element at position i MINUS one of s1 AND dp element at position i and position j MINUS one PLUS the ASCII value of element at position j MINUS one of s2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp element at position m and position n  
    END FUNCTION  
END CLASS
