CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums EQUALS an empty list  
            RETURN zero  
        END IF  
        SET dp TO an empty list  
        FOR each num IN nums  
            SET idx TO the position found by binary searching for num in dp where num can be inserted to maintain sorted order from left to right  
            IF idx EQUALS the LENGTH OF dp  
                APPEND num TO dp  
            ELSE  
                SET the element at position idx of dp TO num  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
