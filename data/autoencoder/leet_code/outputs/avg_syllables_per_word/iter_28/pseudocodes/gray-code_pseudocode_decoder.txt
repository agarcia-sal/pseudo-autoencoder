CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN list containing zero  
        END IF  
        SET prev_gray TO CALL grayCode WITH argument n MINUS one  
        SET mask TO one MULTIPLIED BY two RAISED TO THE POWER OF n MINUS one  
        SET reversed_prev_gray TO reversed order of elements in prev_gray  
        SET appended_list TO empty list  
        FOR each number IN reversed_prev_gray  
            SET appended_element TO mask OR number  
            APPEND appended_element TO appended_list  
        END FOR  
        SET current_gray TO concatenation of prev_gray FOLLOWED BY appended_list  
        RETURN current_gray  
    END FUNCTION  
END CLASS
