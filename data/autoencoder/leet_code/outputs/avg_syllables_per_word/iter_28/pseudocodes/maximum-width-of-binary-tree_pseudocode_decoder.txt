CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO A new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero DO  
        SET node TO the first element REMOVED from queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.left TO A new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None THEN  
            SET node.right TO A new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with arguments p.left and q.left AND the result of is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree(self, root)  
        IF root IS None THEN  
            RETURN zero  
        END IF  
        SET queue TO a new double ended queue containing one tuple with elements root and zero  
        SET max_width TO zero  
        WHILE the LENGTH OF queue IS GREATER THAN zero DO  
            SET level_length TO the LENGTH OF queue  
            SET first_index TO the second element of the first tuple of queue  
            FOR counter FROM zero TO level_length MINUS one DO  
                SET node and index TO the elements of the first tuple REMOVED from queue  
                IF counter EQUALS level_length MINUS one THEN  
                    IF max_width LESS THAN index MINUS first_index PLUS one THEN  
                        SET max_width TO index MINUS first_index PLUS one  
                    END IF  
                END IF  
                IF node.left IS NOT None THEN  
                    APPEND a tuple with elements node.left and index MULTIPLIED BY two TO queue  
                END IF  
                IF node.right IS NOT None THEN  
                    APPEND a tuple with elements node.right and index MULTIPLIED BY two PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
