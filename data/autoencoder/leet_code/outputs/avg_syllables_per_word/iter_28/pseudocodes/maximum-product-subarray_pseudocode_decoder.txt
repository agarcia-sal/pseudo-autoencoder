CLASS Solution  
    FUNCTION maxProduct(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        SET max_product TO element at position one of nums  
        SET min_product TO element at position one of nums  
        SET result TO element at position one of nums  
        FOR each num IN elements from position two TO the end of nums  
            IF num LESS THAN zero  
                SET temporary_to_max TO max_product  
                SET max_product TO min_product  
                SET min_product TO temporary_to_max  
            END IF  
            SET max_product TO the GREATER VALUE BETWEEN num AND max_product MULTIPLIED BY num  
            SET min_product TO the SMALLER VALUE BETWEEN num AND min_product MULTIPLIED BY num  
            SET result TO the GREATER VALUE BETWEEN result AND max_product  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
