CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO the set of all elements in wordDict  
        SET dp TO a list of Boolean values of the length of s PLUS one where all elements are set to false  
        SET element at position zero of dp TO true  
        
        FOR i FROM one TO the length of s  
            FOR j FROM zero TO i MINUS one  
                IF element at position j of dp EQUALS true AND the substring from position j PLUS one TO position i of s IS AN ELEMENT OF wordSet  
                    SET element at position i of dp TO true  
                    BREAK the inner loop  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position equal to the length of s in dp  
    END FUNCTION  
END CLASS
