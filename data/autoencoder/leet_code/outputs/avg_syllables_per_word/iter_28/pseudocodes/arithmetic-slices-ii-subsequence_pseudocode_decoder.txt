CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        
        SET dp TO a list where each element is an empty mapping from integer keys to integer values of size n  
        SET total_count TO zero  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                SET d TO the element at position i of nums MINUS the element at position j of nums  
                
                IF the key d EXISTS IN the mapping at position j of dp  
                    INCREMENT the value at key d in the mapping at position i of dp BY the value at key d in the mapping at position j of dp PLUS one  
                    INCREMENT total_count BY the value at key d in the mapping at position j of dp  
                ELSE  
                    INCREMENT the value at key d in the mapping at position i of dp BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
