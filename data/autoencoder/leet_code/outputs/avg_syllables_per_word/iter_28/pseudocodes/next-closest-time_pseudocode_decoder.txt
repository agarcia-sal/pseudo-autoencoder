CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of all characters in time with each colon character removed  
        SET hour TO the integer value of the first two characters of time  
        SET minute TO the integer value of the characters of time at position four and five  

        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each h1 IN digits  
                FOR each h2 IN digits  
                    FOR each m1 IN digits  
                        FOR each m2 IN digits  
                            SET hour_value TO the integer formed by concatenating h1 PLUS h2  
                            SET minute_value TO the integer formed by concatenating m1 PLUS m2  
                            IF hour_value LESS THAN twenty four AND minute_value LESS THAN sixty  
                                APPEND the tuple of hour_value AND minute_value TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the list times sorted in ascending order  
        END FUNCTION  

        SET all_times TO the result of calling generate_times with argument digits  
        SET current_time_index TO the position of the tuple of hour AND minute in all_times  

        IF current_time_index PLUS one LESS THAN the length of all_times  
            SET next_hour TO the first element of the tuple at position current_time_index PLUS one in all_times  
            SET next_minute TO the second element of the tuple at position current_time_index PLUS one in all_times  
        ELSE  
            SET next_hour TO the first element of the tuple at position zero in all_times  
            SET next_minute TO the second element of the tuple at position zero in all_times  
        END IF  

        SET formatted_hour TO the string form of next_hour with a leading zero if next_hour is less than ten  
        SET formatted_minute TO the string form of next_minute with a leading zero if next_minute is less than ten  
        SET formatted_time TO the concatenation of formatted_hour PLUS the colon character PLUS formatted_minute  

        RETURN formatted_time  
    END FUNCTION  
END CLASS
