CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val zero left None right None  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values list  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO TreeNode of element at position one of values  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO TreeNode of element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO TreeNode of element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree of p.left AND q.left AND is_same_tree of p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves WITH PARAMETERS self root Optional TreeNode  
        FUNCTION collectLeaves WITH PARAMETER node  
            IF node EQUALS None  
                RETURN negative one  
            END IF  
            SET left_height TO collectLeaves of node.left  
            SET right_height TO collectLeaves of node.right  
            SET current_height TO the maximum of left_height AND right_height PLUS one  
            IF current_height GREATER THAN OR EQUAL TO the LENGTH OF result  
                APPEND empty list TO result  
            END IF  
            APPEND node.val TO element at position current_height of result  
            RETURN current_height  
        END FUNCTION  
        SET result TO empty list  
        CALL collectLeaves WITH root  
        RETURN result  
    END FUNCTION  
END CLASS
