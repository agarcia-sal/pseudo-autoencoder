CLASS Solution  
    FUNCTION canJump(nums)  
        SET farthest TO zero  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF i GREATER THAN farthest  
                RETURN False  
            END IF  
            SET sum_of_index_and_value TO i PLUS element at position i of nums  
            SET farthest TO the GREATER VALUE BETWEEN farthest AND sum_of_index_and_value  
        END FOR  
        SET last_index TO the LENGTH OF nums MINUS one  
        RETURN farthest GREATER THAN OR EQUAL TO last_index  
    END FUNCTION  
END CLASS
