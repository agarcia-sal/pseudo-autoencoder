CLASS Solution  
    FUNCTION killProcess(pid list of integers ppid list of integers kill integer) RETURNS list of integers  
        SET tree TO a dictionary with default empty list for each key  
        FOR each child and parent IN corresponding elements of pid AND ppid  
            APPEND child TO the list at key parent in tree  
        END FOR  
        SET queue TO a list containing kill  
        SET killed_processes TO an empty list  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET current TO the first element of queue  
            REMOVE the first element FROM queue  
            APPEND current TO killed_processes  
            EXTEND queue BY the list of elements at key current in tree  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
