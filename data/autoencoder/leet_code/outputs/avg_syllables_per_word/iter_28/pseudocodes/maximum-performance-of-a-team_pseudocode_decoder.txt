CLASS Solution  
    FUNCTION maxPerformance(n speed efficiency k)  
        SET engineers TO empty list  
        FOR index FROM zero TO the LENGTH OF efficiency MINUS one  
            APPEND tuple containing element at position index of efficiency AND element at position index of speed TO engineers  
        END FOR  
        SORT engineers IN descending order BASED ON first element of each tuple  
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each tuple containing eff AND spd IN engineers  
            ADD spd TO speed_heap USING heap push operation  
            INCREMENT speed_sum BY spd  
            IF the LENGTH OF speed_heap GREATER THAN k THEN  
                REMOVE the smallest speed FROM speed_heap USING heap pop operation  
                DECREMENT speed_sum BY the removed speed  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance GREATER THAN max_performance THEN  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        RETURN the remainder of max_performance DIVIDED BY the sum of one billion AND seven  
    END FUNCTION  
END CLASS
