CLASS Solution  
    FUNCTION minOperations(target LIST OF integers arr LIST OF integers) RETURNS integer  
        SET index_map TO AN EMPTY MAPPING FROM integer TO integer  
        SET current_index TO zero  
        FOR each value IN target  
            SET index_map AT value TO current_index  
            INCREMENT current_index BY one  
        END FOR  
        
        SET transformed_arr TO AN EMPTY LIST OF integers  
        FOR each value IN arr  
            IF index_map CONTAINS key value THEN  
                APPEND element at key value FROM index_map TO transformed_arr  
            END IF  
        END FOR  
        
        SET lis TO AN EMPTY LIST OF integers  
        FOR each num IN transformed_arr  
            SET pos TO POSITION TO INSERT num IN lis SUCH THAT lis REMAINS IN NON-DECREASING ORDER  
            IF pos EQUALS LENGTH OF lis THEN  
                APPEND num TO lis  
            ELSE  
                SET element at position pos OF lis TO num  
            END IF  
        END FOR  
        
        SET lcs_length TO LENGTH OF lis  
        RETURN LENGTH OF target MINUS lcs_length  
    END FUNCTION  
END CLASS
