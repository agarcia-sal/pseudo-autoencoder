CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix x y)  
            SET directions TO list containing pair of minus one and zero pair of one and zero pair of zero and minus one pair of zero and one pair of zero and zero  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n THEN  
                    SET element at position nx of matrix at position ny TO one MINUS element at position nx of matrix at position ny  
                END IF  
            END FOR  
        END FUNCTION  

        FUNCTION matrix_to_tuple(matrix)  
            SET result TO empty list  
            FOR each row IN matrix  
                APPEND tuple of elements of row TO result  
            END FOR  
            RETURN tuple of elements of result  
        END FUNCTION  

        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero of mat  
        SET target TO tuple of tuples each composed of zero repeated n times repeated m times  
        SET initial_state TO the result of matrix_to_tuple applied to mat  
        SET queue TO list containing tuple of initial_state and zero  
        SET visited TO empty set  
        ADD initial_state TO visited  

        WHILE the LENGTH OF queue GREATER THAN zero DO  
            REMOVE first element FROM queue AND ASSIGN it TO current_state and steps  
            IF current_state EQUALS target THEN  
                RETURN steps  
            END IF  

            SET current_matrix TO empty list  
            FOR each row IN current_state  
                APPEND list of elements of row TO current_matrix  
            END FOR  

            FOR i FROM zero TO m MINUS one DO  
                FOR j FROM zero TO n MINUS one DO  
                    SET next_matrix TO deep copy of current_matrix  
                    CALL flip with arguments next_matrix i j  
                    SET next_state TO the result of matrix_to_tuple applied to next_matrix  
                    IF next_state NOT IN visited THEN  
                        ADD next_state TO visited  
                        APPEND tuple of next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  

        RETURN minus one  
    END FUNCTION  
END CLASS
