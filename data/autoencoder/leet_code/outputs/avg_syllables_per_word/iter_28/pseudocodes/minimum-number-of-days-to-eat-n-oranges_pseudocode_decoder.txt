CLASS Solution  
    FUNCTION minDays(n)  
        FUNCTION dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            SET option_one TO remainder of remaining DIVIDED BY two PLUS dp of remaining DIVIDED BY two  
            SET option_two TO remainder of remaining DIVIDED BY three PLUS dp of remaining DIVIDED BY three  
            SET result TO one PLUS the smaller value BETWEEN option_one AND option_two  
            RETURN result  
        END FUNCTION  
        RETURN dp of n  
    END FUNCTION  
END CLASS
