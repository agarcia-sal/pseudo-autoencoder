CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string THEN  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS the reverse order of all characters in check_str  
        END FUNCTION  
        
        FOR i FROM the LENGTH OF s DOWN TO zero  
            IF is_palindrome(the substring from position zero TO position i MINUS one of s) THEN  
                SET to_add TO the reverse order of the substring from position i TO the last position of s  
                RETURN the concatenation of to_add AND s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
