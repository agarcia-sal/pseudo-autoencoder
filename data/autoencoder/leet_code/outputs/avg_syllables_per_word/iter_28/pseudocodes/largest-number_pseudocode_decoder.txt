CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO an empty list  
        FOR each number IN nums  
            CONVERT number TO string  
            APPEND that string TO nums_str  
        END FOR  
        
        FUNCTION compare(x, y)  
            SET concatenation_one TO the result of the string x FOLLOWED BY the string y  
            SET concatenation_two TO the result of the string y FOLLOWED BY the string x  
            IF concatenation_one GREATER THAN concatenation_two  
                RETURN negative one  
            ELSE IF concatenation_one LESS THAN concatenation_two  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT the list nums_str USING the custom comparison function compare  
        
        IF the element at position zero of nums_str EQUALS the string zero  
            RETURN the string zero  
        END IF  
        
        SET result TO the concatenation of all elements IN nums_str IN ORDER  
        RETURN result  
    END FUNCTION  
END CLASS
