CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with element at position one of values  
    SET p TO head  
    FOR each val IN elements of values starting from position two TO the end  
        SET node TO new ListNode with val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the val element of p1 EQUALS the val element of p2 AND is_same_list with the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs(self head)  
        SET dummy TO new ListNode with zero  
        SET dummy dot next TO head  
        SET current TO dummy  
        WHILE current dot next NOT EQUALS None AND current dot next dot next NOT EQUALS None  
            SET first TO current dot next  
            SET second TO current dot next dot next  
            SET current dot next TO second  
            SET first dot next TO second dot next  
            SET second dot next TO first  
            SET current TO current dot next dot next  
        END WHILE  
        RETURN dummy dot next  
    END FUNCTION  
END CLASS
