CLASS Solution  
    FUNCTION uniquePaths WITH PARAMETERS m n  
        SET dp TO an empty list  
        FOR i FROM zero TO the quantity m MINUS one  
            SET row TO an empty list  
            FOR j FROM zero TO the quantity n MINUS one  
                APPEND one TO row  
            END FOR  
            APPEND row TO dp  
        END FOR  
        FOR i FROM one TO the quantity m MINUS one  
            FOR j FROM one TO the quantity n MINUS one  
                SET element at position i of dp TO element at position i of dp  
                SET element at position j of element at position i of dp TO element at position i MINUS one of dp at position j PLUS element at position i of dp at position j MINUS one  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one of dp at position n MINUS one  
    END FUNCTION  
END CLASS
