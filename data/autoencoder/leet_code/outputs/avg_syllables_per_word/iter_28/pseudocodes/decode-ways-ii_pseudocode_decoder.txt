CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten RAISED TO the POWER OF nine PLUS one  
        SET dp TO a list of zeros with the LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS the asterisk character  
                RETURN nine  
            ELSE IF char EQUALS the character zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS the asterisk character AND char2 EQUALS the asterisk character  
                RETURN fifteen  
            ELSE IF char1 EQUALS the asterisk character  
                IF char2 LESS THAN OR EQUAL TO the character six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS the asterisk character  
                IF char1 EQUALS the character one  
                    RETURN nine  
                ELSE IF char1 EQUALS the character two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer value of concatenation of char1 and char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
        
        FOR i FROM one TO the LENGTH OF s  
            SET dp at position i TO the value of element at position i of dp PLUS the value of element at position i MINUS one of dp MULTIPLIED BY ways_to_decode_single(element at position i MINUS one of s) MODULO MOD  
            IF i GREATER THAN one  
                SET dp at position i TO the value of element at position i of dp PLUS the value of element at position i MINUS two of dp MULTIPLIED BY ways_to_decode_pair(element at position i MINUS two of s, element at position i MINUS one of s) MODULO MOD  
            END IF  
        END FOR  
        
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
