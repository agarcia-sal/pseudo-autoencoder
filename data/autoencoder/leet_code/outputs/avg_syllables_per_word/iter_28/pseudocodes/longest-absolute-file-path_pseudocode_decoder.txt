CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO the result of splitting input by newline  
        SET stack TO an empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO the count of tab characters in line  
            SET name TO the substring of line starting from position depth TO the end  
            
            WHILE the LENGTH OF stack GREATER THAN depth  
                REMOVE the last element FROM stack  
            END WHILE  
            
            IF the LENGTH OF stack GREATER THAN zero  
                SET current_length TO the last element of stack PLUS the LENGTH OF name PLUS one representing the path separator  
            ELSE  
                SET current_length TO the LENGTH OF name  
            END IF  
            
            IF name CONTAINS a period character  
                SET max_length TO the larger value of max_length AND current_length  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
