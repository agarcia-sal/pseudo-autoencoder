CLASS Solution  
    FUNCTION canTransform(start start AS string, end end AS string) RETURNS boolean  
        SET start_filtered TO an empty list  
        FOR each element c WITH index i IN start  
            IF c NOT EQUALS the character X  
                APPEND the pair of c and i TO start_filtered  
            END IF  
        END FOR  
        
        SET end_filtered TO an empty list  
        FOR each element c WITH index i IN end  
            IF c NOT EQUALS the character X  
                APPEND the pair of c and i TO end_filtered  
            END IF  
        END FOR  
        
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        
        FOR each pair of elements from start_filtered and end_filtered AS (c1 and i1) AND (c2 and i2)  
            IF c1 NOT EQUALS c2  
                RETURN false  
            END IF  
            IF c1 EQUALS the character L AND i1 LESS THAN i2  
                RETURN false  
            END IF  
            IF c1 EQUALS the character R AND i1 GREATER THAN i2  
                RETURN false  
            END IF  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
