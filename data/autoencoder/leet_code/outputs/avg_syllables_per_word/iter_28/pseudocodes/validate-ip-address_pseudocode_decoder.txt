CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS OF DIGITS ONLY  
                SET num TO the integer value of block  
                IF num LESS THAN OR EQUAL TO two hundred fifty five AND the string representation of num EQUALS block  
                    RETURN True  
                ELSE  
                    RETURN False  
                END IF  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  
  
        FUNCTION is_valid_ipv6_block(block)  
            IF the length of block GREATER THAN OR EQUAL TO one AND the length of block LESS THAN OR EQUAL TO four  
                TRY  
                    CONVERT block TO an integer interpreted as base sixteen  
                    RETURN True  
                CATCH a value error  
                    RETURN False  
                END TRY  
            ELSE  
                RETURN False  
            END IF  
        END FUNCTION  
  
        SET parts_ipv4 TO the result of splitting queryIP by the period word  
        SET parts_ipv6 TO the result of splitting queryIP by the colon word  
  
        IF the number of elements in parts_ipv4 EQUALS four AND for every element part IN parts_ipv4 the call to is_valid_ipv4_block with argument part RETURNS True  
            RETURN the word IPv4  
        ELSE IF the number of elements in parts_ipv6 EQUALS eight AND for every element part IN parts_ipv6 the call to is_valid_ipv6_block with argument part RETURNS True  
            RETURN the word IPv6  
        ELSE  
            RETURN the word Neither  
        END IF  
    END FUNCTION  
END CLASS
