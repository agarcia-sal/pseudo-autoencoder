CLASS Solution  
    FUNCTION nthSuperUglyNumber(n, primes)  
        SET ugly_numbers TO list containing the single element one  
        SET heap TO list of pairs where each pair contains an element from primes and its corresponding position index in primes starting from zero  
        TRANSFORM heap into a min-heap structure  
        SET indices TO list of zeros with the size equal to the length of primes  
        WHILE the LENGTH OF ugly_numbers LESS THAN n  
            REMOVE and OBTAIN the pair with the smallest first element from heap, CALL the first element next_ugly and the second element prime_index  
            IF next_ugly NOT EQUALS the last element of ugly_numbers  
                APPEND next_ugly TO ugly_numbers  
            END IF  
            INCREMENT the element at prime_index in indices BY one  
            SET next_multiple TO the element at prime_index in primes MULTIPLIED BY the element at the position given by the element at prime_index in indices in ugly_numbers  
            INSERT the pair next_multiple and prime_index INTO heap while maintaining the min-heap property  
        END WHILE  
        RETURN the last element of ugly_numbers  
    END FUNCTION  
END CLASS
