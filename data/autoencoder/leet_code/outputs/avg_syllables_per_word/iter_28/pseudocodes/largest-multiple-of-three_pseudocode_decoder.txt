CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO CALL Counter WITH digits  
        SET total_sum TO zero  
        FOR each digit IN digits  
            INCREMENT total_sum BY digit  
        END FOR  
        SET remainder TO total_sum MODULO three  
        IF remainder EQUALS one  
            FOR each digit IN the list consisting of one and four and seven  
                IF the element at position digit of count GREATER THAN zero  
                    DECREMENT the element at position digit of count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit IN the list consisting of two and five and eight  
                    IF the element at position digit of count GREATER THAN one  
                        DECREMENT the element at position digit of count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
        IF remainder EQUALS two  
            FOR each digit IN the list consisting of two and five and eight  
                IF the element at position digit of count GREATER THAN zero  
                    DECREMENT the element at position digit of count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit IN the list consisting of one and four and seven  
                    IF the element at position digit of count GREATER THAN one  
                        DECREMENT the element at position digit of count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
        SET result TO an empty list  
        FOR each digit FROM nine TO zero DECREMENTING BY one  
            FOR times FROM one TO the element at position digit of count  
                APPEND the string describing digit TO result  
            END FOR  
        END FOR  
        SET final_number TO the concatenation of all elements of result in order  
        IF the LENGTH OF final_number GREATER THAN zero AND the element at position one of final_number EQUALS zero  
            RETURN the string zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
