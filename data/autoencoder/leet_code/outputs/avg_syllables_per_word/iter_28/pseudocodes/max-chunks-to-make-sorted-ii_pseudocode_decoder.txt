CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET max_so_far TO empty list  
        SET current_max TO element at position one of arr  
        FOR each num IN arr  
            IF current_max LESS THAN num  
                SET current_max TO num  
            END IF  
            APPEND current_max TO max_so_far  
        END FOR  
        SET sorted_arr TO arr sorted in ascending order  
        SET chunks TO zero  
        SET max_chunk TO zero  
        FOR index FROM one TO the LENGTH OF arr  
            IF max_chunk LESS THAN element at position index of arr  
                SET max_chunk TO element at position index of arr  
            END IF  
            IF max_chunk EQUALS element at position index of sorted_arr  
                INCREMENT chunks BY one  
            END IF  
        END FOR  
        RETURN chunks  
    END FUNCTION  
END CLASS
