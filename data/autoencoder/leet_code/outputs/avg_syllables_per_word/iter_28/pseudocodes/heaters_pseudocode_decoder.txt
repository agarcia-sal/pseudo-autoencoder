CLASS Solution  
    FUNCTION findRadius(houses, heaters)  
        SORT houses in ascending order  
        SORT heaters in ascending order  
        SET min_radius TO zero  
        SET j TO zero  
        FOR each house IN houses  
            WHILE j LESS THAN the LENGTH OF heaters MINUS one AND the ABSOLUTE VALUE OF element at position j of heaters MINUS house GREATER THAN OR EQUAL TO the ABSOLUTE VALUE OF element at position j PLUS one of heaters MINUS house  
                INCREMENT j BY one  
            END WHILE  
            SET distance TO the ABSOLUTE VALUE OF element at position j of heaters MINUS house  
            IF distance GREATER THAN min_radius  
                SET min_radius TO distance  
            END IF  
        END FOR  
        RETURN min_radius  
    END FUNCTION  
END CLASS
