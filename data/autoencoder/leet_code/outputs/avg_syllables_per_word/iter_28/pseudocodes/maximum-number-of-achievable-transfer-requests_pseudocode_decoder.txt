CLASS Solution  
    FUNCTION maximumRequests(n requests)  
        SET max_requests TO zero  
        SET num_requests TO the LENGTH OF requests  
        FOR mask FROM zero TO two MULTIPLIED BY num_requests MINUS one  
            SET balance TO a list of zero repeated n times  
            SET count TO zero  
            FOR i FROM zero TO num_requests MINUS one  
                IF the element at position i of the binary representation of mask EQUALS one  
                    SET from_building TO the element at position zero of the element at position i of requests  
                    SET to_building TO the element at position one of the element at position i of requests  
                    DECREMENT the element at position from_building of balance BY one  
                    INCREMENT the element at position to_building of balance BY one  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            IF every element b IN balance EQUALS zero  
                IF max_requests LESS THAN count  
                    SET max_requests TO count  
                END IF  
            END IF  
        END FOR  
        RETURN max_requests  
    END FUNCTION  
END CLASS
