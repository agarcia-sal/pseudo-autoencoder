CLASS Solution  
    FUNCTION getXORSum(arr1 arr1 OF integers arr2 arr2 OF integers) RETURNS integer  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        FOR each num IN arr1  
            SET xor_sum_arr1 TO the result of XOR operation BETWEEN xor_sum_arr1 AND num  
        END FOR  
        FOR each num IN arr2  
            SET xor_sum_arr2 TO the result of XOR operation BETWEEN xor_sum_arr2 AND num  
        END FOR  
        RETURN the result of AND operation BETWEEN xor_sum_arr1 AND xor_sum_arr2  
    END FUNCTION  
END CLASS
