CLASS Solution  
    FUNCTION letterCasePermutation(s) RETURNS list of strings  
        FUNCTION dfs(i) RETURNS nothing  
            IF i GREATER THAN OR EQUAL TO the LENGTH OF t THEN  
                APPEND the concatenation of all elements of t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH i PLUS one  
            IF the element at position i of t IS an alphabetic letter THEN  
                SET the element at position i of t TO the character obtained by toggling the case of the element at position i of t  
                CALL dfs WITH i PLUS one  
            END IF  
        END FUNCTION  
        SET t TO list of all characters of s  
        SET ans TO empty list  
        CALL dfs WITH zero  
        RETURN ans  
    END FUNCTION  
END CLASS
