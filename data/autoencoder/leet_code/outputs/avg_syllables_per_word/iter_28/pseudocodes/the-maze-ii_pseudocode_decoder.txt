CLASS Solution  
    FUNCTION shortestDistance(maze AS List OF List OF Integer start AS List OF Integer destination AS List OF Integer) RETURNS Integer  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO list containing the pair one AND zero AND the pair MINUS one AND zero AND the pair zero AND one AND the pair zero AND MINUS one  
        SET start TO tuple constructed from start  
        SET destination TO tuple constructed from destination  
        SET visited TO empty set  
        SET pq TO a list containing the pair zero AND start  
        
        WHILE pq is not empty  
            REMOVE the smallest element from pq and ASSIGN to dist AND a pair of x AND y  
            
            IF the pair x AND y EQUALS destination THEN  
                RETURN dist  
            END IF  
            
            IF the pair x AND y IS an element of visited THEN  
                CONTINUE to next iteration of the WHILE loop  
            END IF  
            
            ADD the pair x AND y to visited  
            
            FOR each pair dx AND dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx is GREATER THAN OR EQUAL TO zero AND new_x PLUS dx is LESS THAN m AND new_y PLUS dy is GREATER THAN OR EQUAL TO zero AND new_y PLUS dy is LESS THAN n AND the element at position new_x PLUS dx of the element at position new_y PLUS dy of maze EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF the pair new_x AND new_y is NOT an element of visited THEN  
                    INSERT the pair dist PLUS count AND the pair new_x AND new_y into pq  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN MINUS one  
    END FUNCTION  
END CLASS
