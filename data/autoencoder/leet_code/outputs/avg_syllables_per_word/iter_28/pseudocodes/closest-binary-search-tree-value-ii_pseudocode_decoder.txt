CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET val TO the parameter val  
        SET left TO the parameter left  
        SET right TO the parameter right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode object with element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element REMOVED FROM THE LEFT of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT EQUALS None  
            ASSIGN TO node.left a new TreeNode object with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT EQUALS None  
            ASSIGN TO node.right a new TreeNode object with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the val attribute of p IS NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with p.left and q.left AND the result of calling is_same_tree with p.right and q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION closestKValues with parameters root target k  
        FUNCTION inorder_traversal with parameter node  
            IF node EQUALS None  
                RETURN  
            END IF  
            CALL inorder_traversal with node.left  
            IF the LENGTH OF result IS LESS THAN k  
                APPEND the val attribute of node TO result  
            ELSE  
                IF the absolute value of the val attribute of node MINUS target IS LESS THAN the absolute value of the element at position one of result MINUS target  
                    REMOVE element FROM THE LEFT OF result  
                    APPEND the val attribute of node TO result  
                ELSE  
                    RETURN  
                END IF  
            END IF  
            CALL inorder_traversal with node.right  
        END FUNCTION  
        SET result TO an empty double ended queue  
        CALL inorder_traversal with root  
        RETURN a list converted from result  
    END FUNCTION  
END CLASS
