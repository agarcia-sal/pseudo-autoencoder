CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the TUPLE of index AND current_sum IS IN memo  
                RETURN the VALUE of memo AT the TUPLE of index AND current_sum  
            END IF  
            SET positive TO the RESULT of dfs WITH arguments index PLUS one AND current_sum PLUS the ELEMENT at position index of nums AND memo  
            SET negative TO the RESULT of dfs WITH arguments index PLUS one AND current_sum MINUS the ELEMENT at position index of nums AND memo  
            SET the VALUE of memo AT the TUPLE of index AND current_sum TO positive PLUS negative  
            RETURN the VALUE of memo AT the TUPLE of index AND current_sum  
        END FUNCTION  
        RETURN the RESULT of dfs WITH arguments zero AND zero AND an EMPTY DICTIONARY  
    END FUNCTION  
END CLASS
