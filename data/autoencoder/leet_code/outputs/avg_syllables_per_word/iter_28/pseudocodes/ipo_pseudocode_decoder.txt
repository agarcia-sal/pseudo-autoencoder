CLASS Solution  
    FUNCTION findMaximizedCapital(k INTEGER, w INTEGER, profits LIST OF INTEGER, capital LIST OF INTEGER) RETURNS INTEGER  
        SET projects TO A LIST OF PAIRS OF ELEMENTS WHERE EACH PAIR CONSISTS OF element at position SAME INDEX OF capital AND element at position SAME INDEX OF profits FOR EACH INDEX IN RANGE FROM zero TO THE LENGTH OF profits MINUS one  
        SORT projects IN ASCENDING ORDER BY THE FIRST ELEMENT OF EACH PAIR  
        SET max_heap TO AN EMPTY LIST  
        SET i TO zero  
        SET n TO THE LENGTH OF projects  
        FOR each iteration FROM one TO k  
            WHILE i LESS THAN n AND THE FIRST ELEMENT OF ELEMENT at position i OF projects LESS THAN OR EQUAL TO w  
                ADD THE NEGATION OF THE SECOND ELEMENT OF ELEMENT at position i OF projects TO max_heap USING A HEAP INSERT OPERATION  
                INCREMENT i BY one  
            END WHILE  
            IF max_heap IS EMPTY  
                BREAK THE LOOP IMMEDIATELY  
            END IF  
            SET extracted_profit TO THE NEGATION OF THE RESULT OF REMOVING THE HIGHEST PRIORITY ELEMENT FROM max_heap USING A HEAP POP OPERATION  
            INCREMENT w BY extracted_profit  
        END FOR  
        RETURN w  
    END FUNCTION  
END CLASS
