CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode initialized with element at position one of values  
    SET i TO one  
    SET queue TO new double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed from the left of queue  
        IF i LESS THAN the length of values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree called with p.left AND q.left AND is_same_tree called with p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive with parameters self root  
        IF root EQUALS None  
            RETURN zero  
        END IF  
        FUNCTION dfs with parameters node parent current_length  
            IF node EQUALS None  
                RETURN current_length  
            END IF  
            IF parent IS NOT None AND node.val EQUALS parent PLUS one  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET left_length TO dfs called with node.left parent node.val current_length  
            SET right_length TO dfs called with node.right parent node.val current_length  
            RETURN the maximum of current_length left_length right_length  
        END FUNCTION  
        RETURN dfs called with root None zero  
    END FUNCTION  
END CLASS
