CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO an empty list  
        FOR each token IN tokens  
            IF token EQUALS the plus symbol OR token EQUALS the minus symbol OR token EQUALS the multiplication symbol OR token EQUALS the division symbol  
                SET b TO the last element removed from stack  
                SET a TO the last element removed from stack  
                IF token EQUALS the plus symbol  
                    APPEND the sum of a PLUS b TO stack  
                ELSE IF token EQUALS the minus symbol  
                    APPEND the difference of a MINUS b TO stack  
                ELSE IF token EQUALS the multiplication symbol  
                    APPEND the product of a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS the division symbol  
                    APPEND the integer result of the division of a DIVIDED BY b truncated towards zero TO stack  
                END IF  
            ELSE  
                APPEND the integer value of token TO stack  
            END IF  
        END FOR  
        RETURN the first element of stack  
    END FUNCTION  
END CLASS
