CLASS Solution  
    FUNCTION numSquares(n)  
        SET dp TO a list of length n PLUS one where each element is set to positive infinity  
        SET element at position zero of dp TO zero  
        SET squares TO an empty list  
        SET i TO one  
        WHILE the result of i MULTIPLIED BY i is less than or equal to n holds  
            APPEND the result of i MULTIPLIED BY i TO squares  
            INCREMENT i BY one  
        END WHILE  
        FOR i FROM one TO n inclusive  
            FOR each square IN squares  
                IF i is LESS THAN square THEN  
                    BREAK the inner loop  
                END IF  
                SET element at position i of dp TO the minimum of element at position i of dp AND element at position i MINUS square of dp PLUS one  
            END FOR  
        END FOR  
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
