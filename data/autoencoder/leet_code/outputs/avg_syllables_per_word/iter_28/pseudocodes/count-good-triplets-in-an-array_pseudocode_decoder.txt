CLASS BIT  
    FUNCTION __init__(self, n)  
        SET self.size TO n  
        SET self.tree TO a list of zeros with length n PLUS one  
    END FUNCTION  

    FUNCTION update(self, idx, delta)  
        WHILE idx LESS THAN OR EQUAL TO self.size  
            INCREMENT element at position idx of self.tree BY delta  
            SET idx TO idx PLUS the result of idx AND the bitwise negation of idx  
        END WHILE  
    END FUNCTION  

    FUNCTION query(self, idx)  
        SET result TO zero  
        WHILE idx GREATER THAN zero  
            INCREMENT result BY element at position idx of self.tree  
            SET idx TO idx MINUS the result of idx AND the bitwise negation of idx  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION goodTriplets(self, nums1, nums2)  
        SET n TO the LENGTH OF nums1  

        SET pos TO a list of zeros with length n  
        FOR each index i AND each element num IN nums1  
            SET element at position num of pos TO i  
        END FOR  

        SET compressed TO a list constructed by mapping each num in nums2 to element at position num of pos  

        SET bit TO a new instance of BIT initialized with n  

        SET smaller_count TO a list of zeros with length n  
        SET larger_count TO a list of zeros with length n  

        FOR each index i FROM zero TO n MINUS one  
            SET element at position i of smaller_count TO result of bit.query called with element at position i of compressed  
            CALL bit.update with element at position i of compressed PLUS one AND one  
        END FOR  

        SET bit TO a new instance of BIT initialized with n  

        FOR each index i FROM n MINUS one DOWN TO zero  
            SET element at position i of larger_count TO result of bit.query called with n MINUS result of bit.query called with element at position i of compressed PLUS one  
            CALL bit.update with element at position i of compressed PLUS one AND one  
        END FOR  

        SET good_triplets TO zero  
        FOR each index i FROM zero TO n MINUS one  
            INCREMENT good_triplets BY element at position i of smaller_count MULTIPLIED BY element at position i of larger_count  
        END FOR  

        RETURN good_triplets  
    END FUNCTION  
END CLASS
