CLASS Solution  
    FUNCTION jobScheduling(startTime, endTime, profit)  
        SET jobs TO the list of tuples each containing element at position index of startTime element at position index of endTime element at position index of profit sorted in ascending order by the second element of each tuple  
        SET end_times TO the list of the second element of each tuple in jobs  
        SET dp TO a list of zeros with the length equal to the length of jobs PLUS one  
        FOR index i FROM one TO the length of jobs  
            SET j TO the position where the element at position i MINUS one of jobs first element would be inserted in end_times to maintain sorted order  
            SET dp element at position i TO the greater value between dp element at position i MINUS one AND dp element at position j PLUS the third element of the tuple at position i MINUS one of jobs  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
