CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict word_list WITH TYPE List of string, word1 word_one WITH TYPE string, word2 word_two WITH TYPE string) RETURNS integer  
        SET minimum_distance TO positive infinity  
        SET previous_position_word1 TO negative one  
        SET previous_position_word2 TO negative one  
        SET words_are_same TO word_one EQUALS word_two  
        
        FOR each index_and_word IN enumeration of word_list  
            SET current_index TO the position part of index_and_word  
            SET current_word TO the word part of index_and_word  
            
            IF current_word EQUALS word_one  
                IF words_are_same  
                    SET previous_position_word2 TO previous_position_word1  
                END IF  
                SET previous_position_word1 TO current_index  
            END IF  
            
            IF current_word EQUALS word_two  
                IF NOT words_are_same  
                    SET previous_position_word2 TO current_index  
                END IF  
            END IF  
            
            IF previous_position_word1 NOT EQUALS negative one AND previous_position_word2 NOT EQUALS negative one  
                SET absolute_difference TO the absolute value of previous_position_word1 MINUS previous_position_word2  
                IF absolute_difference LESS THAN minimum_distance  
                    SET minimum_distance TO absolute_difference  
                END IF  
            END IF  
        END FOR  
        
        RETURN minimum_distance  
    END FUNCTION  
END CLASS
