CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the set of tuples formed by converting each element of points to a tuple  
        SET minimum_x_coordinate TO the smallest first element of each point in unique_points  
        SET maximum_x_coordinate TO the largest first element of each point in unique_points  
        SET line_of_reflection TO the sum of minimum_x_coordinate AND maximum_x_coordinate DIVIDED BY two  
        FOR each point IN unique_points  
            SET current_x TO the first element of point  
            SET current_y TO the second element of point  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS current_x  
            IF the tuple formed by reflected_x AND current_y is NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
