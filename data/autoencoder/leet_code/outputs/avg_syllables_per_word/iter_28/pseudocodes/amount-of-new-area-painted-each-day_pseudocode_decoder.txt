CLASS Solution  
    FUNCTION amountPainted(paint)  
        SET max_end TO the greatest value of the second element of each pair in paint  
        SET painted TO a list of the length max_end PLUS one with each element set TO negative one  
        SET worklog TO an empty list  
        
        FOR each pair start AND end IN paint  
            SET current_start TO start  
            SET new_paint TO zero  
            
            WHILE current_start LESS THAN end holds  
                IF element at position current_start of painted EQUALS negative one THEN  
                    SET element at position current_start of painted TO end  
                    INCREMENT new_paint BY one  
                    INCREMENT current_start BY one  
                ELSE  
                    SET current_start TO element at position current_start of painted  
                END IF  
            END WHILE  
            
            APPEND new_paint TO worklog  
        END FOR  
        
        RETURN worklog  
    END FUNCTION  
END CLASS
