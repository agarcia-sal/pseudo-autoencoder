CLASS Solution  
    FUNCTION search(nums WITH type List OF integers, target WITH type integer) RETURNS integer  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS the VALUE OF right MINUS left DIVIDED BY two  
            IF the element at position mid of nums IS EQUALS TO target  
                RETURN mid  
            END IF  
            IF the element at position left of nums IS LESS THAN OR EQUAL TO the element at position mid of nums  
                IF the element at position left of nums IS LESS THAN OR EQUAL TO target AND target IS LESS THAN the element at position mid of nums  
                    SET right TO mid MINUS one  
                ELSE  
                    SET left TO mid PLUS one  
                END IF  
            ELSE  
                IF the element at position mid of nums IS LESS THAN target AND target IS LESS THAN OR EQUAL TO the element at position right of nums  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END IF  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
