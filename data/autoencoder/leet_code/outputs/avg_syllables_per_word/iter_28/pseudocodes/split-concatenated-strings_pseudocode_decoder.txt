CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the string formed by elements of element at position index of strs in reverse order  
            IF reversed_str GREATER THAN element at position index of strs  
                SET element at position index of strs TO reversed_str  
            END IF  
        END FOR  
      
        SET doubled_strs TO the concatenation of strs with strs  
        SET max_string TO the empty string  
      
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each is_reversed IN the list containing False and True  
                IF is_reversed EQUALS True  
                    SET current TO the string formed by elements of element at position index of strs in reverse order  
                ELSE  
                    SET current TO element at position index of strs  
                END IF  
              
                SET remaining TO the concatenation of the elements of doubled_strs from position index plus one TO position index plus the LENGTH OF strs minus one  
              
                FOR position FROM zero TO the LENGTH OF current  
                    SET candidate TO the concatenation of the substring of current from position TO the end PLUS remaining PLUS the substring of current from the beginning TO position minus one  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
      
        RETURN max_string  
    END FUNCTION  
END CLASS
