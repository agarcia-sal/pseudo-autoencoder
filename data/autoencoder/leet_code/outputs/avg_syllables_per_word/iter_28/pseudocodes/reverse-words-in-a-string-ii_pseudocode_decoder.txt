CLASS Solution  
    FUNCTION reverseWords WITH parameter s  
        REVERSE the entire list s  
        SET start TO zero  
        FOR each end FROM zero TO the length of list s MINUS one  
            IF element at position end of list s EQUALS a space character  
                REVERSE the elements of list s from position start TO position end MINUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        REVERSE the elements of list s from position start TO the last position of list s  
    END FUNCTION  
END CLASS
