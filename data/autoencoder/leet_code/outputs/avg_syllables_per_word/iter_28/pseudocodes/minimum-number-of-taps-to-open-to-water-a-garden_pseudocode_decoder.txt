CLASS Solution  
    FUNCTION minTaps(n, ranges)  
        SET max_range TO new list of size n PLUS one with all elements zero  
        FOR i FROM zero TO n  
            IF element at position i of ranges GREATER THAN zero  
                SET left TO the GREATER of zero AND i MINUS element at position i of ranges  
                SET right TO the LESSER of n AND i PLUS element at position i of ranges  
                SET element at position left of max_range TO the GREATER of element at position left of max_range AND right  
            END IF  
        END FOR  
        SET taps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        FOR i FROM zero TO n  
            IF i GREATER THAN farthest  
                RETURN minus one  
            END IF  
            IF i GREATER THAN current_end  
                INCREMENT taps BY one  
                SET current_end TO farthest  
            END IF  
            SET farthest TO the GREATER of farthest AND element at position i of max_range  
        END FOR  
        RETURN taps  
    END FUNCTION  
END CLASS
