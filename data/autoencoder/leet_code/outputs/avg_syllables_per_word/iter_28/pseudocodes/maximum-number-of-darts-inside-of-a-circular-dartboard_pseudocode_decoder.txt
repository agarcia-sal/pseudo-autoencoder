CLASS Solution  
    FUNCTION numPoints(darts, r)  
        FUNCTION distance(p1, p2)  
            SET difference between element at position zero of p1 AND element at position zero of p2 TO delta_x  
            SET difference between element at position one of p1 AND element at position one of p2 TO delta_y  
            SET squared delta_x TO delta_x MULTIPLIED BY delta_x  
            SET squared delta_y TO delta_y MULTIPLIED BY delta_y  
            SET sum_of_squares TO squared delta_x PLUS squared delta_y  
            SET result TO square root of sum_of_squares  
            RETURN result  
        END FUNCTION  

        FUNCTION circle_center(p1, p2, r)  
            SET distance_between_points TO distance of p1 AND p2  
            IF distance_between_points GREATER THAN r MULTIPLIED BY two  
                RETURN nothing  
            END IF  
            SET a TO square root of r MULTIPLIED BY r MINUS (distance_between_points DIVIDED BY two) MULTIPLIED BY (distance_between_points DIVIDED BY two)  
            SET midpoint_x TO element at position zero of p1 PLUS element at position zero of p2 DIVIDED BY two  
            SET midpoint_y TO element at position one of p1 PLUS element at position one of p2 DIVIDED BY two  
            SET dx TO difference between element at position one of p2 AND element at position one of p1 MULTIPLIED BY a DIVIDED BY distance_between_points  
            SET dy TO difference between element at position zero of p2 AND element at position zero of p1 MULTIPLIED BY a DIVIDED BY distance_between_points  
            SET first_center TO pair of midpoint_x MINUS dx AND midpoint_y PLUS dy  
            SET second_center TO pair of midpoint_x PLUS dx AND midpoint_y MINUS dy  
            RETURN list containing first_center AND second_center  
        END FUNCTION  

        FUNCTION is_inside(circle, point, r)  
            SET distance_to_point TO distance of circle AND point  
            IF distance_to_point LESS THAN OR EQUAL TO r  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET n TO length of darts  
        IF n EQUALS one  
            RETURN one  
        END IF  

        SET max_count TO one  

        FOR i FROM zero TO n MINUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET centers TO circle_center of element at position i of darts AND element at position j of darts AND r  
                IF centers NOT EQUALS nothing  
                    FOR each center IN centers  
                        SET count TO zero  
                        FOR each dart IN darts  
                            IF is_inside of center AND dart AND r EQUALS true  
                                INCREMENT count BY one  
                            END IF  
                        END FOR  
                        IF count GREATER THAN max_count  
                            SET max_count TO count  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        RETURN max_count  
    END FUNCTION  
END CLASS
