CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET the attribute val TO val  
        SET the attribute left TO left  
        SET the attribute right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF the length of values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO a new instance of TreeNode initialized with the element at position zero of values  
    SET i TO one  
    CREATE an empty double-ended queue named queue  
    APPEND root TO queue  
    WHILE the length of queue GREATER THAN zero holds  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the length of values AND the element at position i of values DOES NOT EQUAL no value  
            SET the attribute left of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the attribute left of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values DOES NOT EQUAL no value  
            SET the attribute right of node TO a new instance of TreeNode initialized with the element at position i of values  
            APPEND the attribute right of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS no value AND q EQUALS no value  
        RETURN True  
    ELSE IF p EQUALS no value OR q EQUALS no value  
        RETURN False  
    ELSE IF the attribute val of p DOES NOT EQUAL the attribute val of q  
        RETURN False  
    ELSE  
        RETURN the value of calling is_same_tree with arguments the attribute left of p AND the attribute left of q AND the value of calling is_same_tree with arguments the attribute right of p AND the attribute right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues with parameters self root  
        IF root EQUALS no value  
            RETURN an empty list  
        END IF  
        CREATE a double-ended queue named queue initialized with root as its single element  
        CREATE an empty list named largest_values  
        WHILE the length of queue GREATER THAN zero holds  
            SET level_size TO the length of queue  
            SET max_value TO a value representing negative infinity  
            FOR each iteration FROM zero TO level_size MINUS one  
                SET node TO the element removed from the front of queue  
                IF max_value LESS THAN the attribute val of node  
                    SET max_value TO the attribute val of node  
                END IF  
                IF the attribute left of node DOES NOT EQUAL no value  
                    APPEND the attribute left of node TO queue  
                END IF  
                IF the attribute right of node DOES NOT EQUAL no value  
                    APPEND the attribute right of node TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
