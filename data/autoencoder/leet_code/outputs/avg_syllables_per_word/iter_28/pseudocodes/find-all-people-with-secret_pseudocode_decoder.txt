CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO a new set containing zero and firstPerson  
        SORT meetings BY the element at position three of each meeting in ascending order  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO the element at position three of the meeting at position i of meetings  
            SET graph TO a new map with default value as an empty list  
            WHILE i LESS THAN the LENGTH OF meetings AND the element at position three of the meeting at position i of meetings EQUALS current_time  
                SET x TO the element at position one of the meeting at position i of meetings  
                SET y TO the element at position two of the meeting at position i of meetings  
                APPEND y TO the list at key x in graph  
                APPEND x TO the list at key y in graph  
                INCREMENT i BY one  
            END WHILE  
            SET queue TO a new double ended queue initialized with each person in know_secret who is a key in graph  
            SET visited TO a new empty set  
            WHILE queue IS NOT empty  
                REMOVE and RETURN the element from the front of queue INTO person  
                IF person IS IN visited  
                    CONTINUE to the next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list at key person in graph  
                    IF neighbor IS NOT IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN the list conversion of know_secret  
    END FUNCTION  
END CLASS
