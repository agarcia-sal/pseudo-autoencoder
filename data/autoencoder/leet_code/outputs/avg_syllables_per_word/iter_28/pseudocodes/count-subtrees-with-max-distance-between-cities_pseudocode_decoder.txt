CLASS Solution  
    FUNCTION countSubgraphsForEachDiameter(n, edges)  
        SET graph TO empty mapping from integer to list  
        FOR integer i FROM one TO n  
            SET element at position i of graph TO empty list  
        END FOR  
        FOR each pair u v IN edges  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
        
        FUNCTION bfs(subtree)  
            SET start TO the first element from subtree  
            SET queue TO a new double-ended queue initialized with start  
            SET visited TO a new set initialized with start  
            SET distance TO zero  
            SET farthest_node TO start  
            
            WHILE queue IS NOT empty  
                REMOVE the leftmost element from queue and ASSIGN to node  
                FOR each neighbor IN element at position node of graph  
                    IF neighbor IS an element in subtree AND neighbor IS NOT an element in visited  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                        SET farthest_node TO neighbor  
                    END IF  
                END FOR  
            END WHILE  
            
            SET queue TO a new double-ended queue initialized with farthest_node  
            SET visited TO a new set initialized with farthest_node  
            SET max_distance TO zero  
            
            WHILE queue IS NOT empty  
                SET size TO the number of elements in queue  
                FOR each count FROM one TO size  
                    REMOVE the leftmost element from queue and ASSIGN to node  
                    FOR each neighbor IN element at position node of graph  
                        IF neighbor IS an element in subtree AND neighbor IS NOT an element in visited  
                            ADD neighbor TO visited  
                            APPEND neighbor TO queue  
                        END IF  
                    END FOR  
                END FOR  
                IF queue IS NOT empty  
                    INCREMENT max_distance BY one  
                END IF  
            END WHILE  
            
            RETURN max_distance  
        END FUNCTION  
        
        SET result TO a list of zeros with length n MINUS one  
        
        FOR size FROM two TO n  
            FOR each subset IN combinations of range from one TO n of size size  
                SET visited TO an empty set  
                SET stack TO a new list initialized with the first element of subset  
                WHILE stack IS NOT empty  
                    REMOVE the last element from stack and ASSIGN to node  
                    IF node IS NOT an element in visited  
                        ADD node TO visited  
                        FOR each neighbor IN element at position node of graph  
                            IF neighbor IS an element in subset AND neighbor IS NOT an element in visited  
                                APPEND neighbor TO stack  
                            END IF  
                        END FOR  
                    END IF  
                END WHILE  
                
                IF visited EQUALS the set of elements in subset  
                    SET diameter TO bfs(set of elements in subset)  
                    IF diameter GREATER THAN zero  
                        INCREMENT element at position diameter MINUS one of result BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
