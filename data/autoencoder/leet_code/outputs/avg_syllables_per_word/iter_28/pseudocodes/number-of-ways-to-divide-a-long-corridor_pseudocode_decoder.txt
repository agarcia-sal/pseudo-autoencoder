CLASS Solution  
    FUNCTION numberOfWays WITH parameter corridor  
        SET MOD TO one billion PLUS seven  
        SET seats TO empty list  
        FOR each position index AND character x IN corridor  
            IF character x EQUALS the letter S  
                APPEND position index TO seats  
            END IF  
        END FOR  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index i FROM two TO the LENGTH OF seats MINUS one STEP two  
            SET difference TO the element at position i of seats MINUS the element at position i MINUS one of seats  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
