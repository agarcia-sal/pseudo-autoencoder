CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO a two dimensional list of size n by n with all values SET TO true  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET element at position i and j of f TO false  
                IF the element at position i of s EQUALS the element at position j of s  
                    SET element at position i and j of f TO the element at position i PLUS one and j MINUS one of f  
                    IF the element at position i and j of f IS true AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the substring of s from position k TO position k PLUS mx MINUS one  
    END FUNCTION  
END CLASS
