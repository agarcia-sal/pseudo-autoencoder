CLASS Solution  
    FUNCTION maxProduct(words)  
        SET word_sets TO list of sets where each set contains the characters of the corresponding word in words  
        SET max_product TO zero  
        FOR index i FROM zero TO the LENGTH OF words MINUS one  
            FOR index j FROM i PLUS one TO the LENGTH OF words MINUS one  
                IF the set at position i of word_sets INTERSECTED WITH the set at position j of word_sets EQUALS empty set  
                    SET product TO the LENGTH OF the word at position i of words MULTIPLIED BY the LENGTH OF the word at position j of words  
                    IF product GREATER THAN max_product  
                        SET max_product TO product  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_product  
    END FUNCTION  
END CLASS
