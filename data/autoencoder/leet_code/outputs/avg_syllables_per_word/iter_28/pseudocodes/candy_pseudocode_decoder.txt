CLASS Solution  
    FUNCTION candy WITH parameter ratings  
        SET n TO the LENGTH OF ratings  
        SET candies TO a list of INTEGER one WITH the LENGTH OF n  
        FOR index i FROM one TO n MINUS one  
            IF element at position i of ratings GREATER THAN element at position i MINUS one of ratings  
                SET element at position i of candies TO element at position i MINUS one of candies PLUS one  
            END IF  
        END FOR  
        FOR index i FROM n MINUS two DOWN TO zero  
            IF element at position i of ratings GREATER THAN element at position i PLUS one of ratings  
                SET element at position i of candies TO the GREATER VALUE BETWEEN element at position i of candies AND element at position i PLUS one of candies PLUS one  
            END IF  
        END FOR  
        RETURN the SUM OF candies  
    END FUNCTION  
END CLASS
