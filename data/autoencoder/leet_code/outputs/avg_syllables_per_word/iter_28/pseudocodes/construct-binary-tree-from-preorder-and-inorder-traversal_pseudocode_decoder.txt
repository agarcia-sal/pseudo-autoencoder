CLASS TreeNode  
    FUNCTION __init__ with parameters self val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty list  
        RETURN None  
    END IF  
    SET root TO new TreeNode with element at position zero of values  
    SET i TO one  
    SET queue TO new double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the left of queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree with parameters p.left q.left AND is_same_tree with parameters p.right q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree with parameters self preorder inorder  
        IF preorder EQUALS empty list OR inorder EQUALS empty list  
            RETURN None  
        END IF  
        SET root_val TO element at position zero of preorder  
        SET root TO new TreeNode with root_val  
        SET root_index_inorder TO the position of root_val in inorder  
        SET root.left TO the result of calling self.buildTree with parameters the sublist of preorder from position one TO position root_index_inorder PLUS one AND the sublist of inorder from position zero TO position root_index_inorder MINUS one PLUS one  
        SET root.right TO the result of calling self.buildTree with parameters the sublist of preorder from position root_index_inorder PLUS one TO the end AND the sublist of inorder from position root_index_inorder PLUS one TO the end  
        RETURN root  
    END FUNCTION  
END CLASS
