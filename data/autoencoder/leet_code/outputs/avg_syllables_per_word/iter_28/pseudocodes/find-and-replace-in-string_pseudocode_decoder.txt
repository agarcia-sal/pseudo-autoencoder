CLASS Solution  
    FUNCTION findReplaceString(s, indices, sources, targets)  
        SET list_of_replacements TO list of tuples of corresponding elements of indices sources and targets  
        SET sorted_replacements TO list_of_replacements sorted in ascending order based on the first element of each tuple  
        SET result_list TO empty list  
        SET previous_end_index TO zero  
        FOR each tuple_element IN sorted_replacements  
            SET current_index TO first element of tuple_element  
            SET current_source TO second element of tuple_element  
            SET current_target TO third element of tuple_element  
            APPEND substring of s from position previous_end_index TO position current_index MINUS one TO result_list  
            IF substring of s from position current_index TO position current_index PLUS the LENGTH OF current_source MINUS one EQUALS current_source  
                APPEND current_target TO result_list  
            ELSE  
                APPEND substring of s from position current_index TO position current_index PLUS the LENGTH OF current_source MINUS one TO result_list  
            END IF  
            SET previous_end_index TO current_index PLUS the LENGTH OF current_source  
        END FOR  
        APPEND substring of s from position previous_end_index TO the end of s TO result_list  
        SET final_result TO concatenation of all elements in result_list into a single string  
        RETURN final_result  
    END FUNCTION  
END CLASS
