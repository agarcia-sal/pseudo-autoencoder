CLASS Solution  
    FUNCTION maximumNumberOfOnes(width parameter height parameter sideLength parameter maxOnes parameter)  
        SET freq TO list of zeros with length sideLength MULTIPLIED BY sideLength  
        FOR each i FROM zero TO width MINUS one  
            FOR each j FROM zero TO height MINUS one  
                SET position TO remainder of i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS remainder of j DIVIDED BY sideLength  
                INCREMENT element at position of freq BY one  
            END FOR  
        END FOR  
        SORT freq IN descending order  
        SET sum_value TO zero  
        FOR each index FROM zero TO maxOnes MINUS one  
            INCREMENT sum_value BY element at index of freq  
        END FOR  
        RETURN sum_value  
    END FUNCTION  
END CLASS
