CLASS Solution  
    FUNCTION wordSquares(words)  
        SET prefix_to_words TO an empty mapping from prefix to list of words  
        FOR each word IN words  
            FOR index FROM zero TO the length of word MINUS one  
                SET prefix TO substring from position zero TO position index MINUS one of word  
                APPEND word TO the list at prefix_to_words at key prefix  
            END FOR  
        END FOR  
        
        SET results TO an empty list  
        
        FUNCTION backtrack(square)  
            IF the length of square EQUALS the length of the first word in words THEN  
                APPEND square TO results  
                RETURN  
            END IF  
            
            SET prefix TO the concatenation of the element at position length of square of each word in square in order  
            FOR each candidate IN the list at prefix_to_words at key prefix  
                CALL backtrack with square PLUS a list containing candidate  
            END FOR  
        END FUNCTION  
        
        FOR each word IN words  
            CALL backtrack with a list containing word  
        END FOR  
        
        RETURN results  
    END FUNCTION  
END CLASS
