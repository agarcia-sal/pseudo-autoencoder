CLASS TreeNode  
    FUNCTION __init__(self, val equals zero, left equals None, right equals None)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new instance of TreeNode with value equal to element at position zero of values  
    SET i TO one  
    SET queue TO new double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the length of values AND element at position i of values is not None  
            SET node dot left TO new instance of TreeNode with value equal to element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values is not None  
            SET node dot right TO new instance of TreeNode with value equal to element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p dot left, q dot left) AND is_same_tree(p dot right, q dot right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low equals negative infinity, high equals infinity)  
            IF node EQUALS None  
                RETURN True  
            END IF  
            IF node dot val LESS THAN OR EQUAL TO low OR node dot val GREATER THAN OR EQUAL TO high  
                RETURN False  
            END IF  
            RETURN validate(node dot left, low, node dot val) AND validate(node dot right, node dot val, high)  
        END FUNCTION  
        RETURN validate(root)  
    END FUNCTION  
END CLASS
