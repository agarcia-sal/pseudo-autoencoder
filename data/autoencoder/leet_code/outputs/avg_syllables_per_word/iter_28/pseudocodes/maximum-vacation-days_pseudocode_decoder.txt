CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the element at position zero of days  
        
        SET dp TO a list of n elements each set to negative one  
        SET the element at position zero of dp TO zero  
        
        SET graph TO a mapping from each index i in the range zero TO n MINUS one TO an empty list  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i of flights at position j EQUALS one  
                    APPEND j TO the list at graph for key i  
                END IF  
            END FOR  
        END FOR  
        
        FOR week FROM zero TO k MINUS one  
            SET new_dp TO a list of n elements each set to negative one  
            FOR i FROM zero TO n MINUS one  
                IF the element at position i of dp EQUALS negative one  
                    CONTINUE to the next iteration  
                END IF  
                IF the element at position i of new_dp LESS THAN the sum of the element at position i of dp PLUS the element at position i of days at position week  
                    SET the element at position i of new_dp TO the sum of the element at position i of dp PLUS the element at position i of days at position week  
                END IF  
                FOR each city j IN the list at graph for key i  
                    IF the element at position j of new_dp LESS THAN the sum of the element at position i of dp PLUS the element at position j of days at position week  
                        SET the element at position j of new_dp TO the sum of the element at position i of dp PLUS the element at position j of days at position week  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        SET max_vacation TO the element with maximum value in dp  
        RETURN max_vacation  
    END FUNCTION  
END CLASS
