CLASS Solution  
    FUNCTION minTotalDistance(grid)  
        SET rows TO empty list  
        SET cols TO empty list  
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position i of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    APPEND i TO rows  
                    APPEND j TO cols  
                END IF  
            END FOR  
        END FOR  
        SORT rows IN ascending order  
        SORT cols IN ascending order  
        SET median_row TO element at position LENGTH OF rows DIVIDED BY two OF rows  
        SET median_col TO element at position LENGTH OF cols DIVIDED BY two OF cols  
        SET min_distance TO zero  
        FOR each row IN rows  
            INCREMENT min_distance BY the ABSOLUTE VALUE OF row MINUS median_row  
        END FOR  
        FOR each col IN cols  
            INCREMENT min_distance BY the ABSOLUTE VALUE OF col MINUS median_col  
        END FOR  
        RETURN min_distance  
    END FUNCTION  
END CLASS
