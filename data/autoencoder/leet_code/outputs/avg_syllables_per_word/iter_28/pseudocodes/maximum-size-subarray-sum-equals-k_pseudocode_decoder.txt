CLASS Solution  
    FUNCTION maxSubArrayLen(nums k)  
        SET cumulative_sum_index TO an empty mapping with zero key mapped TO negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        
        FOR each pair index i AND element num IN enumeration of nums  
            INCREMENT cumulative_sum BY num  
            
            IF the value of cumulative_sum MINUS k is a key in cumulative_sum_index  
                SET max_length TO the greater value BETWEEN max_length AND the difference of i MINUS the value at key cumulative_sum MINUS k in cumulative_sum_index  
            END IF  
            
            IF cumulative_sum is not a key in cumulative_sum_index  
                SET the value at key cumulative_sum in cumulative_sum_index TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
