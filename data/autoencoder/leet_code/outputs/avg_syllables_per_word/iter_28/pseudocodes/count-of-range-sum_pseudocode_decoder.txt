CLASS Solution  
    FUNCTION countRangeSum(nums, lower, upper)  
        SET prefix_sums TO a list containing zero  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            
            SET left TO current_sum MINUS upper  
            SET right TO current_sum MINUS lower  
            
            SET count TO count PLUS the result of calling bisect right in prefix_sums with right MINUS the result of calling bisect left in prefix_sums with left  
            
            CALL bisect insert into prefix_sums the value current_sum  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
