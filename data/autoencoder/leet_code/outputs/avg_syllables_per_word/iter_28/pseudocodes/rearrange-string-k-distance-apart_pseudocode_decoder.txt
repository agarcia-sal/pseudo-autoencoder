CLASS Solution  
    FUNCTION rearrangeString WITH PARAMETERS s k  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO frequency count of each character in s  
        
        SET max_heap TO list of pairs containing negative of count and character for each character and count in freq  
        CONVERT max_heap into a heap structure  
        
        SET wait_queue TO empty list  
        
        SET result TO empty list  
        
        WHILE max_heap is not empty  
            REMOVE the element with largest priority from max_heap and ASSIGN to count and char  
            APPEND char TO result  
            
            APPEND pair of count PLUS one and char TO wait_queue  
            
            IF the length of wait_queue EQUALS k  
                REMOVE first element from wait_queue and ASSIGN to count and char  
                IF count LESS THAN zero  
                    ADD pair of count and char back to max_heap  
                END IF  
            END IF  
        END WHILE  
        
        IF the length of result NOT EQUALS the length of s  
            RETURN empty string  
        END IF  
        
        RETURN concatenation of all characters in result  
    END FUNCTION  
END CLASS
