CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL sort ON arr2 TO ARRANGE ELEMENTS IN ASCENDING ORDER  
        SET dp TO a dictionary WITH key negative one AND value zero  
        
        FOR each number IN arr1  
            SET new_dp TO an empty dictionary  
            
            FOR each pair last_num AND ops IN dp  
                IF number GREATER THAN last_num THEN  
                    IF number NOT IN new_dp OR the value at key number IN new_dp GREATER THAN ops THEN  
                        SET the value at key number IN new_dp TO ops  
                    END IF  
                END IF  
                
                SET idx TO the position IN arr2 WHERE an element GREATER THAN last_num WOULD BE INSERTED TO THE RIGHT ACCORDING TO BINARY SEARCH  
                IF idx LESS THAN the LENGTH OF arr2 THEN  
                    SET replacement TO the element at position idx OF arr2  
                    IF replacement NOT IN new_dp OR the value at key replacement IN new_dp GREATER THAN ops PLUS one THEN  
                        SET the value at key replacement IN new_dp TO ops PLUS one  
                    END IF  
                END IF  
            END FOR  
            
            IF new_dp IS EMPTY THEN  
                RETURN negative one  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the MINIMUM value AMONG all VALUES IN dp  
    END FUNCTION  
END CLASS
