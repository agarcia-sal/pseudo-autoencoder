CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF the element at position start of num EQUALS the character zero AND end GREATER THAN start PLUS one  
                    CONTINUE to the next iteration of the FOR loop  
                END IF  
                SET next_num TO the integer value represented by the substring from position start TO position end MINUS one of num  
                IF next_num GREATER THAN OR EQUAL TO two RAISED TO THE power of thirty ONE  
                    BREAK the FOR loop  
                END IF  
                IF the LENGTH OF path LESS THAN two OR next_num EQUALS the element at position LENGTH OF path MINUS one of path PLUS the element at position LENGTH OF path MINUS two of path  
                    SET result TO the invocation of backtrack with arguments end and path concatenated with next_num as a new list element  
                    IF result is not empty  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN an empty list  
        END FUNCTION  
        RETURN the invocation of backtrack with arguments zero and an empty list  
    END FUNCTION  
END CLASS
