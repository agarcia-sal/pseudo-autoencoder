CLASS Solution  
    FUNCTION findMinHeightTrees(n parameter, edges parameter)  
        IF n parameter EQUALS one  
            RETURN a list containing zero  
        END IF  
        
        SET graph TO a mapping with default empty list  
        SET degree TO a list of zeros with the length of n parameter  
        
        FOR each pair of elements u parameter and v parameter IN edges parameter  
            APPEND element v parameter TO the list at element u parameter of graph  
            APPEND element u parameter TO the list at element v parameter of graph  
            INCREMENT element u parameter of degree BY one  
            INCREMENT element v parameter of degree BY one  
        END FOR  
        
        SET leaves TO a double ended queue containing elements i for each element i FROM zero TO n parameter MINUS one WHERE element i of degree EQUALS one  
        
        SET remaining_nodes TO n parameter  
        WHILE remaining_nodes GREATER THAN two  
            SET leaves_count TO the length of leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR each element in an iteration from one TO leaves_count  
                SET leaf TO the element removed from the left end of leaves  
                FOR each element neighbor IN the list at element leaf of graph  
                    DECREMENT element neighbor of degree BY one  
                    IF element neighbor of degree EQUALS one  
                        APPEND element neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN the list converted from leaves  
    END FUNCTION  
END CLASS
