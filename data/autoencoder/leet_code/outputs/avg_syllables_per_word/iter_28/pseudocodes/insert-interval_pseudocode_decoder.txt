CLASS Solution  
    FUNCTION insert(intervals PARAMETERS intervals OF TYPE list of list of integers newInterval PARAMETERS newInterval OF TYPE list of integers) RETURNS list of list of integers  
        IF the LENGTH OF intervals EQUALS zero  
            RETURN a new list containing newInterval  
        END IF  
        SET merged TO an empty list  
        SET i TO zero  
        SET n TO the LENGTH OF intervals  
        WHILE i LESS THAN n AND the element at position one of element at position i of intervals LESS THAN element at position zero of newInterval  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        WHILE i LESS THAN n AND the element at position zero of element at position i of intervals LESS THAN OR EQUAL TO element at position one of newInterval  
            SET element at position zero of newInterval TO the MINIMUM OF element at position zero of newInterval AND element at position zero of element at position i of intervals  
            SET element at position one of newInterval TO the MAXIMUM OF element at position one of newInterval AND element at position one of element at position i of intervals  
            INCREMENT i BY one  
        END WHILE  
        APPEND newInterval TO merged  
        WHILE i LESS THAN n  
            APPEND the element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        RETURN merged  
    END FUNCTION  
END CLASS
