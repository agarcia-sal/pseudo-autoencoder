CLASS Solution  
    FUNCTION findMaxForm(strs AS List OF strings, m AS integer, n AS integer) RETURNS integer  
        SET dp TO two dimensional list of integers with dimensions m PLUS one by n PLUS one with all elements set to zero  
        FOR each string s IN strs  
            SET count_0 TO the number of times character zero appears in s  
            SET count_1 TO the number of times character one appears in s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp element at position i and j TO the greater of the current value at dp position i and j AND the sum of dp element at position reduced by count_0 and reduced by count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position m and n  
    END FUNCTION  
END CLASS
