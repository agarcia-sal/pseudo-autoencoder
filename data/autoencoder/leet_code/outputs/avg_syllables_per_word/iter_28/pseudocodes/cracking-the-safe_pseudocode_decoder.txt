CLASS Solution  
    FUNCTION crackSafe(n k)  
        SET visited TO empty collection of unique items  
        SET sequence TO empty collection of items  
        
        FUNCTION dfs(node)  
            FOR index i FROM zero TO k MINUS one  
                SET edge TO concatenation of node and string representation of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs WITH substring of edge FROM position one TO the END  
                    APPEND string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO string consisting of zero repeated n MINUS one TIMES  
        CALL dfs WITH start_node  
        
        SET result TO concatenation of sequence elements in reverse order FOLLOWED BY start_node  
        RETURN result  
    END FUNCTION  
END CLASS
