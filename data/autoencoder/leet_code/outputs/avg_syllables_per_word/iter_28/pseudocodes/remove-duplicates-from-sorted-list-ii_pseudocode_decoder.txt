CLASS ListNode  
    FUNCTION __init__ with parameters self val AND next  
        SET self val TO val  
        SET self next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH parameter values  
    IF values EQUALS empty collection  
        RETURN no value  
    END IF  
    SET head TO new ListNode CREATED WITH the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values STARTING FROM the second element TO the last element  
        SET node TO new ListNode CREATED WITH val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH parameters p1 AND p2  
    IF p1 EQUALS no value AND p2 EQUALS no value  
        RETURN true  
    END IF  
    IF p1 EQUALS no value OR p2 EQUALS no value  
        RETURN false  
    END IF  
    RETURN the condition that the val attribute of p1 EQUALS the val attribute of p2 AND the result of calling is_same_list WITH the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates WITH parameters self AND head  
        SET dummy TO new ListNode CREATED WITH zero AND head  
        SET prev TO dummy  
        WHILE head HOLDS a value  
            IF the next attribute of head HOLDS a value AND the val attribute of head EQUALS the val attribute of the next attribute of head  
                WHILE the next attribute of head HOLDS a value AND the val attribute of head EQUALS the val attribute of the next attribute of head  
                    SET head TO the next attribute of head  
                END WHILE  
                SET the next attribute of prev TO the next attribute of head  
            ELSE  
                SET prev TO the next attribute of prev  
            END IF  
            SET head TO the next attribute of head  
        END WHILE  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
