CLASS Solution  
    FUNCTION minimumMountainRemovals(nums)  
        SET n TO the LENGTH OF nums  
        SET lis TO a list of size n FILLED WITH one  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF element at position i of nums GREATER THAN element at position j of nums  
                    SET lis at position i TO the MAXIMUM OF lis at position i AND lis at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        SET lds TO a list of size n FILLED WITH one  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                IF element at position i of nums GREATER THAN element at position j of nums  
                    SET lds at position i TO the MAXIMUM OF lds at position i AND lds at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        SET max_mountain_length TO zero  
        FOR i FROM one TO n MINUS two  
            IF element at position i of lis GREATER THAN one AND element at position i of lds GREATER THAN one  
                SET max_mountain_length TO the MAXIMUM OF max_mountain_length AND element at position i of lis PLUS element at position i of lds MINUS one  
            END IF  
        END FOR  
        RETURN n MINUS max_mountain_length  
    END FUNCTION  
END CLASS
