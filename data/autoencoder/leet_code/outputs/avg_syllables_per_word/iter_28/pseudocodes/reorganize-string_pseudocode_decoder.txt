CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO frequency mapping of characters in s  
        SET max_heap TO empty list  
        FOR each character AND each frequency IN count  
            APPEND pair of negative frequency AND character TO max_heap  
        END FOR  
        TRANSFORM max_heap INTO a heap structure  
        
        SET previous character TO nothing  
        SET previous frequency TO zero  
        SET result TO empty list  
        
        WHILE max_heap is not empty OR previous frequency is less than zero  
            IF max_heap is empty AND previous frequency is less than zero  
                RETURN empty string  
            END IF  
            
            REMOVE the element with lowest priority FROM max_heap AND ASSIGN to frequency AND character  
            APPEND character TO result  
            
            INCREMENT frequency BY one  
            
            IF previous frequency is less than zero  
                ADD previous frequency AND previous character INTO max_heap  
            END IF  
            
            SET previous character TO character  
            SET previous frequency TO frequency  
        END WHILE  
        
        RETURN concatenation of all elements in result INTO a single string  
    END FUNCTION  
END CLASS
