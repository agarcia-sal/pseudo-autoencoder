CLASS Solution  
    FUNCTION findMinStep(board, hand)  
        FUNCTION clean(board)  
            SET stack TO empty list  
            FOR each ball IN the sequence of characters in board  
                IF the stack is not empty AND the first element of the last pair in stack NOT EQUALS ball AND the second element of the last pair in stack GREATER THAN OR EQUAL TO three  
                    REMOVE the last element FROM stack  
                END IF  
                IF the stack is empty OR the first element of the last pair in stack NOT EQUALS ball  
                    APPEND the pair consisting of ball AND one TO stack  
                ELSE  
                    INCREMENT the second element of the last pair in stack BY one  
                END IF  
            END FOR  
            IF the stack is not empty AND the second element of the last pair in stack GREATER THAN OR EQUAL TO three  
                REMOVE the last element FROM stack  
            END IF  
            SET result_string TO empty string  
            FOR each pair of ball AND count IN stack  
                APPEND the repetition of ball by count times TO result_string  
            END FOR  
            RETURN result_string  
        END FUNCTION  

        FUNCTION dfs(board, hand_counter)  
            IF the board is empty  
                RETURN zero  
            END IF  
            SET min_steps TO positive infinity  
            SET i TO zero  
            WHILE i LESS THAN the length of board  
                SET j TO i PLUS one  
                WHILE j LESS THAN the length of board AND the character at position i of board EQUALS the character at position j of board  
                    INCREMENT j BY one  
                END WHILE  
                SET increment TO three MINUS the difference of j MINUS i  
                IF the count of the character at position i of board IN hand_counter GREATER THAN OR EQUAL TO increment  
                    SET new_hand_counter TO a copy of hand_counter  
                    DECREMENT the count of the character at position i of board IN new_hand_counter BY increment  
                    SET new_board TO the result of calling clean with the concatenation of the substring of board from the start TO position i AND the substring of board from position j TO the end  
                    SET steps TO the result of calling dfs with new_board AND new_hand_counter  
                    IF steps GREATER THAN OR EQUAL TO zero  
                        SET min_steps TO the minimum of min_steps AND steps PLUS increment  
                    END IF  
                END IF  
                SET i TO j  
            END WHILE  
            IF min_steps EQUALS positive infinity  
                RETURN minus one  
            ELSE  
                RETURN min_steps  
            END IF  
        END FUNCTION  

        SET hand_counter TO a collection counting the occurrences of characters in hand  
        RETURN the result of calling dfs with board AND hand_counter  
    END FUNCTION  
END CLASS
