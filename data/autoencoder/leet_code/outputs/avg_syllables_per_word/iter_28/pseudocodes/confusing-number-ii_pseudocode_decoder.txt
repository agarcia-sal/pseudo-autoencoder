CLASS Solution  
    FUNCTION confusingNumberII(n)  
        SET rotate_map TO mapping where zero maps to zero one maps to one six maps to nine eight maps to eight nine maps to six  
        SET valid_digits TO list containing zero one six eight nine  

        FUNCTION is_confusing(num)  
            SET original_num TO num  
            SET rotated_num TO zero  
            WHILE num GREATER THAN zero holds  
                SET digit TO remainder of num DIVIDED BY ten  
                SET rotated_num TO rotated_num MULTIPLIED BY ten PLUS the value in rotate_map for digit  
                SET num TO quotient of num DIVIDED BY ten  
            END WHILE  
            RETURN rotated_num NOT EQUALS original_num  
        END FUNCTION  

        FUNCTION count_confusing_numbers(limit current)  
            IF current GREATER THAN limit THEN  
                RETURN zero  
            END IF  
            IF current NOT EQUALS zero AND is_confusing(current) THEN  
                SET count TO one  
            ELSE  
                SET count TO zero  
            END IF  
            FOR each digit IN valid_digits  
                SET new_number TO current MULTIPLIED BY ten PLUS digit  
                IF new_number EQUALS zero THEN  
                    CONTINUE TO next iteration  
                END IF  
                INCREMENT count BY count_confusing_numbers(limit new_number)  
            END FOR  
            RETURN count  
        END FUNCTION  

        RETURN count_confusing_numbers(n zero)  
    END FUNCTION  
END CLASS
