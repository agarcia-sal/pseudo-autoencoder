CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS_EMPTY  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode with value of element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE the first element FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO NEW TreeNode with value of element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO NEW TreeNode with value of element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF value of p.val IS NOT EQUALS value of q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree with p.left and q.left AND is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST(self, root)  
        SET self.max_sum TO zero  
        
        FUNCTION helper(node)  
            IF node IS None  
                RETURN True AND zero AND positive infinity AND negative infinity  
            END IF  
            
            SET left_is_bst, left_sum, left_min, left_max TO helper with node.left  
            SET right_is_bst, right_sum, right_min, right_max TO helper with node.right  
            
            IF left_is_bst IS True AND right_is_bst IS True AND left_max IS LESS THAN value of node.val AND value of node.val IS LESS THAN right_min  
                SET current_sum TO left_sum PLUS right_sum PLUS value of node.val  
                SET current_min TO the smaller value BETWEEN value of node.val AND left_min  
                SET current_max TO the larger value BETWEEN value of node.val AND right_max  
                SET self.max_sum TO the larger value BETWEEN self.max_sum AND current_sum  
                RETURN True AND current_sum AND current_min AND current_max  
            END IF  
            
            RETURN False AND zero AND positive infinity AND negative infinity  
        END FUNCTION  
        
        CALL helper with root  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
