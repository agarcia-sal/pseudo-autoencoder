CLASS ListNode  
    FUNCTION __init__ WITH parameters val AND next  
        SET the value property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH parameter values  
    IF values EQUALS an empty collection  
        RETURN no value  
    END IF  
    SET head TO a new ListNode object WITH the first element in values AS val  
    SET p TO head  
    FOR each val IN values starting from the second element TO the last element  
        SET node TO a new ListNode object WITH val AS val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH parameters p1 AND p2  
    IF p1 EQUALS no value AND p2 EQUALS no value  
        RETURN true  
    END IF  
    IF p1 EQUALS no value OR p2 EQUALS no value  
        RETURN false  
    END IF  
    RETURN the value property of p1 EQUALS the value property of p2 AND the result of calling is_same_list WITH the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList WITH parameters head  
        IF head EQUALS no value OR the next property of head EQUALS no value  
            RETURN head  
        END IF  
        SET dummy TO a new ListNode object WITH the value negative five thousand and one  
        SET the next property of dummy TO head  
        SET current TO the next property of head  
        SET the next property of head TO no value  
        WHILE current holds a value  
            SET next_node TO the next property of current  
            SET prev TO dummy  
            WHILE the next property of prev holds a value AND the value property of the next property of prev LESS THAN the value property of current  
                SET prev TO the next property of prev  
            END WHILE  
            SET the next property of current TO the next property of prev  
            SET the next property of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
