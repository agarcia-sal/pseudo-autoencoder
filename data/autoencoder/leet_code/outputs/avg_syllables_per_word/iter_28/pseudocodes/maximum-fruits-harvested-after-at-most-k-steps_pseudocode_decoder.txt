CLASS Solution  
    FUNCTION maxTotalFruits(fruits LIST OF LIST OF INTEGER startPos INTEGER k INTEGER) RETURNS INTEGER  
        SET max_fruits TO zero  
        SET total_fruits TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO the LENGTH OF fruits MINUS one  
            SET position TO the element at position right in fruits at position zero  
            SET amount TO the element at position right in fruits at position one  
            INCREMENT total_fruits BY amount  
            
            WHILE left LESS THAN OR EQUAL TO right AND NOT (startPos MINUS k LESS THAN OR EQUAL TO the element at position left in fruits at position zero AND the element at position left in fruits at position zero LESS THAN OR EQUAL TO startPos PLUS k AND startPos MINUS k LESS THAN OR EQUAL TO the element at position right in fruits at position zero AND the element at position right in fruits at position zero LESS THAN OR EQUAL TO startPos PLUS k AND (THE MINIMUM OF THE ABSOLUTE VALUE OF (the element at position right in fruits at position zero MINUS startPos) AND THE ABSOLUTE VALUE OF (the element at position left in fruits at position zero MINUS startPos)) PLUS (the element at position right in fruits at position zero MINUS the element at position left in fruits at position zero) LESS THAN OR EQUAL TO k)  
                DECREMENT total_fruits BY the element at position left in fruits at position one  
                INCREMENT left BY one  
            END WHILE  
            
            SET max_fruits TO THE MAXIMUM OF max_fruits AND total_fruits  
        END FOR  
        
        RETURN max_fruits  
    END FUNCTION  
END CLASS
