CLASS Solution  
    FUNCTION ladderLength(beginWord endWord wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN zero  
        END IF  
        SET word_dict TO a dictionary with default list for missing keys  
        SET word_length TO the LENGTH OF beginWord  
        FOR each word IN wordList  
            FOR each index i FROM zero TO word_length MINUS one  
                SET intermediate_word TO the substring from position zero TO position i MINUS one of word PLUS the character representing any single letter PLUS the substring from position i PLUS one TO the end of word  
                APPEND word TO the list at key intermediate_word in word_dict  
            END FOR  
        END FOR  
        SET queue TO a double ended queue containing the tuple of beginWord AND one  
        SET visited TO a set containing beginWord  
        WHILE queue is not empty  
            REMOVE the first tuple FROM queue AND ASSIGN the first element TO current_word AND the second element TO current_length  
            FOR each index i FROM zero TO word_length MINUS one  
                SET intermediate_word TO the substring from position zero TO position i MINUS one of current_word PLUS the character representing any single letter PLUS the substring from position i PLUS one TO the end of current_word  
                FOR each word IN the list at key intermediate_word in word_dict  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND the tuple of word AND current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN zero  
    END FUNCTION  
END CLASS
