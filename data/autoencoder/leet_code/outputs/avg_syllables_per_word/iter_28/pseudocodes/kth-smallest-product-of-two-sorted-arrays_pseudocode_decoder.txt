CLASS Solution  
    FUNCTION kthSmallestProduct(nums1 PARAMETER, nums2 PARAMETER, k PARAMETER)  
        FUNCTION count(p PARAMETER)  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each x IN nums1  
                IF x GREATER THAN zero THEN  
                    INCREMENT cnt BY the POSITION obtained by FINDING the RIGHT POSITION OF p DIVIDED BY x IN nums2 sorted in ascending order  
                ELSE IF x LESS THAN zero THEN  
                    INCREMENT cnt BY n MINUS the POSITION obtained by FINDING the LEFT POSITION OF p DIVIDED BY x IN nums2 sorted in ascending order  
                ELSE  
                    INCREMENT cnt BY n MULTIPLIED BY the boolean VALUE of p GREATER THAN OR EQUAL TO zero  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  

        SET mx TO the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT of nums1 AND the ABSOLUTE VALUE of the LAST ELEMENT of nums1 MULTIPLIED BY the MAXIMUM OF the ABSOLUTE VALUE OF the FIRST ELEMENT of nums2 AND the ABSOLUTE VALUE of the LAST ELEMENT of nums2  
        RETURN the POSITION obtained by FINDING the LEFT POSITION OF k IN the RANGE starting FROM negative mx TO mx PLUS one with the KEY function count MINUS mx  
    END FUNCTION  
END CLASS
