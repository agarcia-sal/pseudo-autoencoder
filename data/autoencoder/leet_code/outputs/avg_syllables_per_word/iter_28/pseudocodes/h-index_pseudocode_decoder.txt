CLASS Solution  
    FUNCTION hIndex(citations)  
        SORT citations IN ORDER FROM the HIGHEST TO the LOWEST  
        FOR index FROM zero TO the LENGTH OF citations MINUS one  
            SET citation TO the element at position index of citations  
            IF citation IS LESS THAN index PLUS one  
                RETURN index  
            END IF  
        END FOR  
        RETURN the LENGTH OF citations  
    END FUNCTION  
END CLASS
