CLASS Solution  
    FUNCTION frequencySort WITH parameter s  
        SET frequency TO result of counting the occurrences of each character in s using a frequency counting mechanism  
        SET sorted_characters TO the collection of keys from frequency sorted according to the frequency value corresponding to each key in descending order  
        SET result TO an empty string  
        FOR each character IN sorted_characters  
            APPEND character repeated exactly the frequency value of the character times TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
