CLASS Solution  
    FUNCTION minCostToSupplyWater(n, wells, pipes)  
        FOR index FROM zero TO n MINUS one  
            APPEND list consisting of zero PLUS index PLUS one AND element at position index of wells TO pipes  
        END FOR  
        SORT pipes BY the element at position three of each pipe in ascending order  
        SET parent TO list of integers from zero TO n inclusive  
        FUNCTION find(x)  
            IF element at position x of parent NOT EQUALS x  
                SET element at position x of parent TO result of find WITH argument element at position x of parent  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        FUNCTION union(x, y)  
            SET rootX TO result of find WITH argument x  
            SET rootY TO result of find WITH argument y  
            IF rootX NOT EQUALS rootY  
                SET element at position rootX of parent TO rootY  
                RETURN true  
            END IF  
            RETURN false  
        END FUNCTION  
        SET min_cost TO zero  
        FOR each triple consisting of house1 AND house2 AND cost IN pipes  
            IF union WITH arguments house1 AND house2 RETURNS true  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
