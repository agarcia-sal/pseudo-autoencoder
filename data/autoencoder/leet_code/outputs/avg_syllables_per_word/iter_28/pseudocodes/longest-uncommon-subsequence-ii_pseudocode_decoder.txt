CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s t)  
            SET iterator TO an iterator over t  
            RETURN a value indicating that for each character in s the character is found in order within iterator  
        END FUNCTION  
        
        SORT strs by a negative value of the LENGTH OF each element PLUS lexicographical order ascending  
        
        FOR each index i AND element word IN strs  
            SET unique TO true  
            FOR each index j AND element other IN strs  
                IF i NOT EQUALS j AND is_subsequence with s as word AND t as other RETURNS true  
                    SET unique TO false  
                    BREAK out of the inner loop  
                END IF  
            END FOR  
            IF unique IS true  
                RETURN the LENGTH OF word  
            END IF  
        END FOR  
        
        RETURN a value representing negative one  
    END FUNCTION  
END CLASS
