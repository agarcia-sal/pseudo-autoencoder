CLASS Solution  
    FUNCTION dailyTemperatures WITH PARAMETER temperatures  
        SET n TO the LENGTH OF temperatures  
        SET answer TO a LIST OF zeros WITH the LENGTH n  
        SET stack TO an EMPTY LIST  
        
        FOR i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position i of temperatures IS GREATER THAN the element at position the element at the last position of stack of temperatures  
                SET prev_index TO the ELEMENT REMOVED FROM the LAST POSITION OF stack  
                SET the element at position prev_index of answer TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
