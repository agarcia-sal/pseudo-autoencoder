CLASS Solution  
    FUNCTION closestToTarget(arr parameter, target parameter)  
        SET seen TO an empty set of unique values  
        SET min_diff TO a value representing infinity  
        FOR each num IN arr parameter  
            SET temporary_set TO an empty set of unique values  
            FOR each x IN seen  
                ADD the result of bitwise AND of num AND x TO temporary_set  
            END FOR  
            SET seen TO the union of temporary_set AND a set containing num  
            FOR each val IN seen  
                SET current_difference TO the absolute value of val MINUS target parameter  
                IF current_difference LESS THAN min_diff THEN  
                    SET min_diff TO current_difference  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
