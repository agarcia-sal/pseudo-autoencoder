CLASS Solution  
    FUNCTION groupAnagrams WITH PARAMETER strs  
        SET anagrams TO a new default dictionary with default value as an empty list  
        FOR each string s IN strs  
            SET sorted_str TO the concatenation of characters of s sorted in ascending order  
            APPEND s TO the list that is the value associated with key sorted_str in anagrams  
        END FOR  
        RETURN the list of all values contained in anagrams  
    END FUNCTION  
END CLASS
