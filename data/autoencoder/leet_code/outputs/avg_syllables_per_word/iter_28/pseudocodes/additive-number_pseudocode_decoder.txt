CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start first second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            SET expected TO the string representation OF the SUM of first PLUS second  
            SET length TO the LENGTH OF expected  
            IF start PLUS length GREATER THAN the LENGTH OF num OR the substring OF num FROM position start TO position start PLUS length NOT EQUALS expected  
                RETURN False  
            END IF  
            RETURN is_valid_sequence WITH arguments start PLUS length second AND the integer conversion OF expected  
        END FUNCTION  
        
        SET n TO the LENGTH OF num  
        FOR i FROM one TO the result OF integer division OF n BY two PLUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET first TO the substring OF num FROM position zero TO position i  
                SET second TO the substring OF num FROM position i TO position j  
                IF the LENGTH OF first GREATER THAN one AND the element at position zero OF first EQUALS zero OR the LENGTH OF second GREATER THAN one AND the element at position zero OF second EQUALS zero  
                    CONTINUE to the next iteration  
                END IF  
                IF is_valid_sequence WITH arguments j AND the integer conversion OF first AND the integer conversion OF second EQUALS True  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
