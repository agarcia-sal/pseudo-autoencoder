CLASS Solution  
    FUNCTION kSmallestPairs(nums1 nums1 AS List of integers nums2 nums2 AS List of integers k k AS integer) RETURNS List of List of integers  
        IF nums1 EQUALS empty OR nums2 EQUALS empty OR k LESS THAN OR EQUAL TO zero THEN  
            RETURN empty list  
        END IF  
        SET min_heap TO empty list  
        FOR i FROM zero TO the SMALLER OF k AND the LENGTH OF nums1 MINUS one  
            CALL push_element_into_heap min_heap WITH value as the SUM of element at position i of nums1 PLUS element at position zero of nums2 AND indices i AND zero  
        END FOR  
        SET result TO empty list  
        WHILE min_heap IS NOT empty AND the LENGTH OF result LESS THAN k DO  
            CALL pop_element_from_heap min_heap AND ASSIGN returned values TO dummy_variable AND i AND j  
            APPEND a list containing element at position i of nums1 AND element at position j of nums2 TO result  
            IF j PLUS one LESS THAN the LENGTH OF nums2 THEN  
                CALL push_element_into_heap min_heap WITH value as the SUM of element at position i of nums1 PLUS element at position j PLUS one of nums2 AND indices i AND j PLUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
