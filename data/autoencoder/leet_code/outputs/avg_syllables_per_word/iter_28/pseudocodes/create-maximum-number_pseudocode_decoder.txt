CLASS Solution  
    FUNCTION maxNumber(nums1 PARAMETER, nums2 PARAMETER, k PARAMETER)  
        FUNCTION maxSingleArray(nums PARAMETER, k PARAMETER)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop IS GREATER THAN zero AND the LENGTH OF stack IS GREATER THAN zero AND element at the last position of stack IS LESS THAN num  
                    REMOVE the element at the last position of stack FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the first k elements of stack in order  
        END FUNCTION  
        
        FUNCTION merge(arr1 PARAMETER, arr2 PARAMETER)  
            SET result TO an empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR index FROM one TO total_length  
                IF array arr1 FROM current position TO end IS GREATER THAN array arr2 FROM current position TO end based on lexicographical comparison  
                    APPEND element at the first position of arr1 TO result  
                    REMOVE element at the first position of arr1 FROM arr1  
                ELSE  
                    APPEND element at the first position of arr2 TO result  
                    REMOVE element at the first position of arr2 FROM arr2  
                END IF  
            END FOR  
            RETURN result  
        END FUNCTION  
        
        SET max_number TO an empty list  
        FOR i FROM zero TO k  
            IF i IS LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i IS LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO merge(the result of maxSingleArray called with parameters nums1 and i, the result of maxSingleArray called with parameters nums2 and k MINUS i)  
                IF max_number IS LESS THAN candidate based on lexicographical comparison  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        
        RETURN max_number  
    END FUNCTION  
END CLASS
