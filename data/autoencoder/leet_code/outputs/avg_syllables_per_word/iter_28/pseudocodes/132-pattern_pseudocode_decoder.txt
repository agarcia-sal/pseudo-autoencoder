CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO a value smaller than any possible number  
        SET stack TO an empty list  
        FOR each number IN nums starting from the last element to the first element  
            IF the current number is less than the value stored in second  
                RETURN true  
            END IF  
            WHILE the stack is not empty AND the element at the last position of the stack is less than the current number  
                SET second TO the value obtained by removing the element at the last position of the stack  
            END WHILE  
            APPEND the current number TO the stack  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
