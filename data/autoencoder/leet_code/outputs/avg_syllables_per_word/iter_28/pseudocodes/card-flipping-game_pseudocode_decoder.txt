CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO empty set  
        FOR each pair of elements x and y FROM fronts and backs respectively  
            IF x EQUALS y  
                ADD x TO same  
            END IF  
        END FOR  
        SET all_numbers TO set containing all elements from fronts and all elements from backs  
        SET good_numbers TO all_numbers MINUS same  
        IF good_numbers is not empty  
            SET minimum_good_number TO the smallest element in good_numbers  
            RETURN minimum_good_number  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
