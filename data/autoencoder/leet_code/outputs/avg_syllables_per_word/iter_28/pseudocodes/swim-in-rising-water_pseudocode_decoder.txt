CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an EMPTY SET  
        SET min_heap TO a LIST CONTAINING a TUPLE OF the element at position zero AND zero of the element at position zero of grid FOLLOWED BY zero FOLLOWED BY zero  
        ADD the TUPLE of zero AND zero TO visited  
        
        SET directions TO a LIST CONTAINING the TUPLES of zero AND one AND one AND zero AND zero AND negative one AND negative one AND zero  
        
        WHILE min_heap HOLDS ELEMENTS  
            SET t TO the FIRST element OF the result of REMOVING the SMALLEST element FROM min_heap  
            SET i TO the SECOND element OF the result of REMOVING the SMALLEST element FROM min_heap  
            SET j TO the THIRD element OF the result of REMOVING the SMALLEST element FROM min_heap  
            
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            
            FOR each TUPLE of di AND dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND the TUPLE of ni AND nj NOT IN visited  
                    ADD the TUPLE of ni AND nj TO visited  
                    INSERT the TUPLE of the GREATER OF t AND the element at position ni AND nj of grid FOLLOWED BY ni FOLLOWED BY nj INTO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
