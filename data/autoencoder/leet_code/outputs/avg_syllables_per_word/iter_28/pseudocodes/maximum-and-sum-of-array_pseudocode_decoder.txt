CLASS Solution  
    FUNCTION maximumANDSum(nums parameter, numSlots parameter)  
        FUNCTION dp(index parameter, slots parameter)  
            IF index EQUALS the LENGTH OF nums  
                RETURN zero  
            END IF  
            
            SET max_and_sum TO zero  
            SET slots_list TO a LIST CREATED FROM slots  
            
            FOR i FROM zero TO numSlots MINUS one  
                IF element at position i of slots_list LESS THAN two  
                    INCREMENT element at position i of slots_list BY one  
                    SET and_sum TO the bitwise AND of element at position index of nums AND i PLUS one PLUS the RESULT OF dp WITH PARAMETERS index PLUS one AND a TUPLE CREATED FROM slots_list  
                    IF and_sum GREATER THAN max_and_sum  
                        SET max_and_sum TO and_sum  
                    END IF  
                    DECREMENT element at position i of slots_list BY one  
                END IF  
            END FOR  
            
            RETURN max_and_sum  
        END FUNCTION  
        
        SET initial_slots TO a TUPLE CREATED FROM a LIST OF zeros WITH LENGTH numSlots  
        RETURN dp WITH PARAMETERS zero AND initial_slots  
    END FUNCTION  
END CLASS
