CLASS Solution  
    FUNCTION carFleet(target position speed)  
        SET paired_positions_and_speeds TO list of pairs of elements from position and speed sorted in order from greatest element to smallest element based on position  
        SET time_to_target TO empty list  
        FOR each pair_of_position_and_speed IN paired_positions_and_speeds  
            SET position_element TO element at position one of pair_of_position_and_speed  
            SET speed_element TO element at position two of pair_of_position_and_speed  
            SET time_value TO the result of subtracting position_element FROM target THEN DIVIDED BY speed_element  
            APPEND time_value TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time_element IN time_to_target  
            IF time_element GREATER THAN current_max_time THEN  
                INCREMENT fleets BY one  
                SET current_max_time TO time_element  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
