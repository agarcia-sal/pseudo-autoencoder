CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO an empty dictionary  
        FOR each position AS idx AND character AS char IN s WITH index  
            SET the value of character char in last_occurrence TO idx  
        END FOR  
        
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO an empty list  
        
        FOR each position AS i AND character AS char IN s WITH index  
            IF end LESS THAN last_occurrence of character char THEN  
                SET end TO last_occurrence of character char  
            END IF  
            
            IF i EQUALS end THEN  
                SET partition_size TO end MINUS start PLUS one  
                APPEND partition_size TO partition_sizes  
                SET start TO i PLUS one  
            END IF  
        END FOR  
        
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
