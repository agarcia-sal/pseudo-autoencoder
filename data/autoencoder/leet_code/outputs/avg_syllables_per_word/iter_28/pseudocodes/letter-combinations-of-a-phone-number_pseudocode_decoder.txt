CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string  
            RETURN empty list  
        END IF  
        
        SET phone_map TO mapping from digit two TO string of letters a b c PLUS digit three TO string of letters d e f PLUS digit four TO string of letters g h i PLUS digit five TO string of letters j k l PLUS digit six TO string of letters m n o PLUS digit seven TO string of letters p q r s PLUS digit eight TO string of letters t u v PLUS digit nine TO string of letters w x y z  
        
        FUNCTION backtrack(combination, next_digits)  
            IF the LENGTH OF next_digits EQUALS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the value at key that is element at position zero of next_digits of phone_map  
                    CALL backtrack WITH the STRING CONCATENATION of combination PLUS letter AND the substring from position one TO the end of next_digits  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO empty list  
        CALL backtrack WITH empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
