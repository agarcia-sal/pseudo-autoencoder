CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO the list of characters of num  
        SET n TO the LENGTH OF num  
        
        WHILE k GREATER THAN zero holds  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF element at position i of num GREATER THAN element at position i PLUS one of num  
                    SET min_digit TO character nine  
                    SET min_index TO minus one  
                    FOR j FROM i TO the MINIMUM OF i PLUS k PLUS one AND n  
                        IF element at position j of num LESS THAN min_digit  
                            SET min_digit TO element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    CALL helper to perform the swap of elements of num from position i TO position min_index by moving element at position min_index to position i and shifting others right  
                    SET k TO k MINUS the difference between min_index and i  
                    SET swapped TO true  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            
            IF swapped EQUALS false  
                BREAK the WHILE loop  
            END IF  
        END WHILE  
        
        RETURN concatenation of all elements in num as a single string  
    END FUNCTION  
END CLASS
