CLASS Solution  
    FUNCTION constructArray(n parameter k parameter)  
        SET result TO list consisting of each element FROM one TO n MINUS k MINUS one INCLUSIVE  
        SET left TO n MINUS k  
        SET right TO n  
        WHILE left LESS THAN OR EQUAL TO right  
            IF left EQUALS right  
                APPEND element left TO result  
            ELSE  
                APPEND element left TO result  
                APPEND element right TO result  
            END IF  
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
