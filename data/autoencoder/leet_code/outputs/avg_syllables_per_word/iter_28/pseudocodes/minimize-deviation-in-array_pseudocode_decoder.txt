CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO infinity  
        FOR each num IN nums  
            IF the remainder when num is divided by two EQUALS one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            IF min_value GREATER THAN num  
                SET min_value TO num  
            END IF  
            INSERT negative num INTO max_heap with heap property  
        END FOR  
        SET min_deviation TO infinity  
        WHILE the remainder when the element at position zero of max_heap is divided by two EQUALS zero  
            EXTRACT the smallest element from max_heap and assign the negation to max_value  
            IF min_deviation GREATER THAN max_value MINUS min_value  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY two with integer division  
            IF min_value GREATER THAN max_value  
                SET min_value TO max_value  
            END IF  
            INSERT negative max_value INTO max_heap with heap property  
        END WHILE  
        IF min_deviation GREATER THAN the negation of the element at position zero of max_heap MINUS min_value  
            SET min_deviation TO the negation of the element at position zero of max_heap MINUS min_value  
        END IF  
        RETURN min_deviation  
    END FUNCTION  
END CLASS
