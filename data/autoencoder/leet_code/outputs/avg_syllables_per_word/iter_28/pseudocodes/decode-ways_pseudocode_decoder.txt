CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s EQUALS empty OR element at position zero of s EQUALS character zero  
            RETURN zero  
        END IF  
        SET dp TO a list of zero values WITH the LENGTH of s PLUS one  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO one  
        FOR i FROM two TO the LENGTH of s  
            IF element at position i MINUS one of s DOES NOT EQUAL character zero  
                SET element at position i of dp TO element at position i of dp PLUS element at position i MINUS one of dp  
            END IF  
            SET two_digit TO the INTEGER VALUE OF substring from position i MINUS two TO position i of s  
            IF two_digit GREATER THAN OR EQUAL TO ten AND two_digit LESS THAN OR EQUAL TO twenty six  
                SET element at position i of dp TO element at position i of dp PLUS element at position i MINUS two of dp  
            END IF  
        END FOR  
        RETURN element at position the LENGTH of dp MINUS one of dp  
    END FUNCTION  
END CLASS
