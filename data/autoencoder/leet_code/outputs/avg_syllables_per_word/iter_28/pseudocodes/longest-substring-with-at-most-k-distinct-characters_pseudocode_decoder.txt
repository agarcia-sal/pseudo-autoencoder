CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s parameter k parameter)  
        IF k EQUALS zero THEN  
            RETURN zero  
        END IF  
        SET char_map TO empty mapping  
        SET left TO zero  
        SET max_length TO zero  
        FOR right FROM zero TO the length of s MINUS one  
            ASSIGN reference TO right FOR the element at position right of s IN char_map  
            IF the size of char_map GREATER THAN k THEN  
                SET leftmost_char TO the key in char_map with the minimal associated value  
                SET left TO the value associated with leftmost_char in char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            SET current_window_length TO right MINUS left PLUS one  
            SET max_length TO the greater value BETWEEN max_length AND current_window_length  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
