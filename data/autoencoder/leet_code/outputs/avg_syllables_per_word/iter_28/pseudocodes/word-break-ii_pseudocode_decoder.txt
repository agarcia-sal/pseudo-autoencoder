CLASS Solution  
    FUNCTION wordBreak with parameters s and wordDict  
        SET wordSet TO the set of all elements in wordDict  
        SET memo TO an empty dictionary  
        
        FUNCTION backtrack with parameter start  
            IF start EXISTS AS A KEY IN memo THEN  
                RETURN the value associated with start in memo  
            END IF  
            IF start EQUALS the length of s THEN  
                RETURN a list containing the empty string  
            END IF  
            
            SET res TO an empty list  
            FOR end FROM start PLUS one TO the length of s PLUS one  
                SET word TO the substring of s from position start TO position end MINUS one  
                IF word EXISTS IN wordSet THEN  
                    FOR each sentence IN backtrack called with end  
                        IF sentence IS the empty string THEN  
                            APPEND word TO res  
                        ELSE  
                            APPEND word FOLLOWED BY a single space FOLLOWED BY sentence TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            
            SET the value associated with start in memo TO res  
            RETURN res  
        END FUNCTION  
        
        RETURN backtrack called with zero  
    END FUNCTION  
END CLASS
