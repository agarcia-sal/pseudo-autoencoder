CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values and left equal to None and right equal to None  
    SET i TO one  
    CREATE an empty queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO element removed from the front of queue  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with value equal to element at position i of values and left equal to None and right equal to None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with value equal to element at position i of values and left equal to None and right equal to None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p.left and q.left AND the result of is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestUnivaluePath(self, root)  
        FUNCTION dfs(node)  
            IF node EQUALS None  
                RETURN zero, zero  
            END IF  
            SET left_length, left_value TO the result of dfs with node.left  
            SET right_length, right_value TO the result of dfs with node.right  
            SET left_arrow TO zero  
            SET right_arrow TO zero  
            IF node.left NOT EQUALS None AND node.left.val EQUALS node.val  
                SET left_arrow TO left_length PLUS one  
            END IF  
            IF node.right NOT EQUALS None AND node.right.val EQUALS node.val  
                SET right_arrow TO right_length PLUS one  
            END IF  
            SET self.max_length TO the maximum of self.max_length AND left_arrow PLUS right_arrow  
            RETURN the maximum of left_arrow AND right_arrow, node.val  
        END FUNCTION  
        SET self.max_length TO zero  
        CALL dfs with root  
        RETURN self.max_length  
    END FUNCTION  
END CLASS
