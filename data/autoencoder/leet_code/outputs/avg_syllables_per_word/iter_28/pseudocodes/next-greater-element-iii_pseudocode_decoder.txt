CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO list of the sequence of characters representing the number n converted to text  
        SET length TO the number of elements in digits  
          
        SET i TO length MINUS two  
        WHILE i IS GREATER THAN OR EQUAL TO zero AND the element at position i of digits IS GREATER THAN OR EQUAL TO the element at position i PLUS one of digits  
            DECREMENT i BY one  
        END WHILE  
          
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
          
        SET j TO length MINUS one  
        WHILE the element at position j of digits IS LESS THAN OR EQUAL TO the element at position i of digits  
            DECREMENT j BY one  
        END WHILE  
          
        SET temporary TO the element at position i of digits  
        SET the element at position i of digits TO the element at position j of digits  
        SET the element at position j of digits TO temporary  
          
        SET digits TO the concatenation of the elements from the start of digits TO the element at position i of digits PLUS one AND the reverse order of the elements from the position i PLUS one TO the end of digits  
          
        SET result TO the integer corresponding to the concatenation of all elements in digits into a continuous text string  
          
        IF result IS GREATER THAN two RAISED TO THE power of thirty ONE MINUS one  
            RETURN negative one  
        END IF  
          
        RETURN result  
    END FUNCTION  
END CLASS
