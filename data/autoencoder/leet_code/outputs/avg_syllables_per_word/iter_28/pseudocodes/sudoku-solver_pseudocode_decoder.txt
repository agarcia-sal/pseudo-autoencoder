CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR iterator FROM zero TO eight  
                IF element at position iterator of the row at position row of board EQUALS num  
                    RETURN false  
                END IF  
            END FOR  
            FOR iterator FROM zero TO eight  
                IF element at position row of the column at position col of board EQUALS num  
                    RETURN false  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY the integer division of row BY three  
            SET start_col TO three MULTIPLIED BY the integer division of col BY three  
            FOR iterator_i FROM zero TO two  
                FOR iterator_j FROM zero TO two  
                    IF element at position start_row PLUS iterator_i of the row at position start_col PLUS iterator_j of board EQUALS num  
                        RETURN false  
                    END IF  
                END FOR  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        FUNCTION solve(board)  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF element at position col of the row at position row of board EQUALS the single character dot  
                        FOR num FROM the character representing one TO the character representing nine  
                            IF is_valid(board, row, col, num)  
                                SET the element at position col of the row at position row of board TO num  
                                IF solve(board)  
                                    RETURN true  
                                END IF  
                                SET the element at position col of the row at position row of board TO the single character dot  
                            END IF  
                        END FOR  
                        RETURN false  
                    END IF  
                END FOR  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        CALL solve WITH board  
    END FUNCTION  
END CLASS
