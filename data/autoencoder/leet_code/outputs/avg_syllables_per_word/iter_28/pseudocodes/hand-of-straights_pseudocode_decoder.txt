CLASS Solution  
    FUNCTION isNStraightHand(hand PARAMETER, groupSize PARAMETER)  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN false  
        END IF  
        SET count TO the frequency count of each element in hand  
        SET unique_cards TO the sorted list of the keys in count  
        FOR each card IN unique_cards  
            IF the value associated with card in count GREATER THAN zero THEN  
                SET needed TO the value associated with card in count  
                FOR i FROM zero TO groupSize MINUS one  
                    IF the value associated with the element at position card PLUS i in count LESS THAN needed THEN  
                        RETURN false  
                    END IF  
                    DECREMENT the value associated with the element at position card PLUS i in count BY needed  
                END FOR  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
