CLASS Solution  
    FUNCTION minMoves(nums, k)  
        SET positions TO empty list  
        FOR index, num IN each element and position OF nums  
            IF num EQUALS one  
                APPEND index TO positions  
            END IF  
        END FOR  
          
        FUNCTION calculate_cost(start, end)  
            SET mid TO start PLUS end DIVIDED BY two  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR i FROM start TO end MINUS one  
                INCREMENT cost BY the absolute value of element at position i of positions MINUS median MINUS mid PLUS i  
            END FOR  
            RETURN cost  
        END FUNCTION  
          
        SET min_cost TO positive infinity  
        FOR i FROM zero TO LENGTH OF positions MINUS k  
            SET current_cost TO calculate_cost(i, i PLUS k)  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
          
        RETURN min_cost  
    END FUNCTION  
END CLASS
