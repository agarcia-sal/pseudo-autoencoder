CLASS Solution  
    FUNCTION findAnagrams(s parameter, p parameter)  
        SET length of s TO len_s  
        SET length of p TO len_p  
        IF length of p GREATER THAN length of s  
            RETURN empty list  
        END IF  
        SET p_count TO frequency count of each character IN p  
        SET s_count TO frequency count of each character IN the substring of s FROM position one TO position len_p  
        SET anagrams TO empty list  
        IF s_count EQUALS p_count  
            APPEND zero TO anagrams  
        END IF  
        FOR i FROM len_p PLUS one TO len_s  
            INCREMENT the count of character at position i of s BY one IN s_count  
            DECREMENT the count of character at position i MINUS len_p of s BY one IN s_count  
            IF the count of character at position i MINUS len_p of s EQUALS zero IN s_count  
                REMOVE the character at position i MINUS len_p of s FROM s_count  
            END IF  
            IF s_count EQUALS p_count  
                APPEND to anagrams the value of i MINUS len_p PLUS one  
            END IF  
        END FOR  
        RETURN anagrams  
    END FUNCTION  
END CLASS
