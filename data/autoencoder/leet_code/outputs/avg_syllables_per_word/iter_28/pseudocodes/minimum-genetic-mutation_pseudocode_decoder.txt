CLASS Solution  
    FUNCTION minMutation(start STRING end STRING bank LIST OF STRING) RETURNS INTEGER  
        SET s TO the SET created FROM bank  
        SET q TO the DOUBLE ENDED QUEUE containing the PAIR consisting OF start AND zero  
        SET mp TO the MAPPING from character A TO the STRING consisting OF T C G AND from character T TO the STRING consisting OF A C G AND from character C TO the STRING consisting OF A T G AND from character G TO the STRING consisting OF A T C  
        WHILE the QUEUE q is NOT empty  
            SET t TO the FIRST element OF the FIRST PAIR in q  
            SET step TO the SECOND element OF the FIRST PAIR in q  
            REMOVE the FIRST PAIR FROM the QUEUE q  
            IF the STRING t EQUALS the STRING end  
                RETURN step  
            END IF  
            FOR index i FROM zero TO the LENGTH OF the STRING t MINUS one  
                SET v TO the CHARACTER at POSITION i OF the STRING t  
                FOR each character j IN the STRING mapped TO v IN mp  
                    SET next TO the CONCATENATION of the SUBSTRING of t FROM START TO POSITION i MINUS one PLUS the character j PLUS the SUBSTRING of t FROM POSITION i PLUS one TO END  
                    IF the STRING next IS IN the SET s  
                        APPEND the PAIR consisting OF next AND step PLUS one TO the QUEUE q  
                        REMOVE the STRING next FROM the SET s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN the INTEGER negative one  
    END FUNCTION  
END CLASS
