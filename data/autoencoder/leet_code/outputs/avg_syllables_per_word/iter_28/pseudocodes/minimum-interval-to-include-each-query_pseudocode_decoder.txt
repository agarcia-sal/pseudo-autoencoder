CLASS Solution  
    FUNCTION minInterval(intervals intervals OF LIST OF LIST OF INTEGER queries queries OF LIST OF INTEGER) RETURNS LIST OF INTEGER  
        SORT intervals IN INCREASING ORDER BY element at position zero of each interval  
        SET sorted_queries TO the LIST OF PAIRS OF each query AND its original position IN queries SORTED IN INCREASING ORDER BY the query value  
        INITIALIZE min_heap TO empty list  
        INITIALIZE result TO a list OF length EQUAL TO the LENGTH OF queries FILLED WITH negative one  
        SET interval_index TO zero  
        FOR each pair query AND original_index IN sorted_queries  
            WHILE interval_index LESS THAN the LENGTH OF intervals AND element at position zero of intervals at position interval_index LESS THAN OR EQUAL TO query  
                SET start TO element at position zero of intervals at position interval_index  
                SET end TO element at position one of intervals at position interval_index  
                SET interval_size TO end MINUS start PLUS one  
                ADD the PAIR OF interval_size AND end TO min_heap USING heap insertion  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE min_heap IS NOT empty AND the second element of the pair AT the TOP OF min_heap LESS THAN query  
                REMOVE the element AT the TOP OF min_heap USING heap removal  
            END WHILE  
            IF min_heap IS NOT empty  
                SET element at position original_index of result TO the first element of the pair AT the TOP OF min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
