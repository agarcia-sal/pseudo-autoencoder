CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
  
        SET value_indices TO a new mapping FROM values TO empty lists  
        FOR each position i AND corresponding element value IN arr  
            APPEND i TO the list associated WITH value IN value_indices  
        END FOR  
  
        SET queue TO a new double-ended queue CONTAINING a pair of zero AND zero representing current position AND steps taken  
        SET visited TO a new set CONTAINING zero representing visited positions  
  
        WHILE queue IS NOT empty  
            SET current_index AND steps TO the front element REMOVED FROM queue  
  
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
  
            SET next_indices TO a list CONTAINING current_index PLUS one AND current_index MINUS one PLUS all positions FROM the list associated WITH the value at position current_index IN arr  
  
            FOR each next_index IN next_indices  
                IF next_index GREATER THAN OR EQUAL TO zero AND next_index LESS THAN n AND next_index IS NOT IN visited  
                    ADD next_index TO visited  
                    APPEND the pair of next_index AND steps PLUS one TO queue  
                END IF  
            END FOR  
  
            SET the list associated WITH the value at position current_index IN arr TO an empty list  
        END WHILE  
  
        RETURN the integer negative one  
    END FUNCTION  
END CLASS
