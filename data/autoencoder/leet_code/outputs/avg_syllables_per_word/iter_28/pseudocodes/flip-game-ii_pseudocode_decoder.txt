CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty associative collection  
        
        FUNCTION can_win_helper(state)  
            IF state EQUALS an element IN memo  
                RETURN the value associated with state IN memo  
            END IF  
            
            FOR position FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring from position TO position PLUS one of state EQUALS two consecutive plus signs  
                    SET new_state TO the concatenation of the substring from the beginning TO position of state PLUS two consecutive minus signs PLUS the substring from position PLUS two TO the end of state  
                    IF the result of calling can_win_helper WITH new_state IS NOT true  
                        SET the value associated with state IN memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value associated with state IN memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN the result of calling can_win_helper WITH currentState  
    END FUNCTION  
END CLASS
