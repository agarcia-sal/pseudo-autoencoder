CLASS Solution  
    FUNCTION minSkips(dist LIST OF integers speed INTEGER hoursBefore INTEGER) RETURNS INTEGER  
        SET n TO the LENGTH OF dist  
        CREATE dp AS a list containing n elements each being a list of n PLUS one elements all set to infinity value  
        SET total_dist TO the SUM of all elements in dist  
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  
        
        SET the element at position zero and zero of dp TO the element at position zero of dist  
        
        FOR j FROM one TO n MINUS one  
            SET the element at position zero and j of dp TO the element at position zero of dist  
        END FOR  
        
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j LESS THAN i THEN  
                    SET the element at position i and j of dp TO the result of the ceiling of the element at position i MINUS one and j of dp DIVIDED BY speed MULTIPLIED BY speed PLUS the element at position i of dist  
                END IF  
                IF j GREATER THAN zero THEN  
                    SET the element at position i and j of dp TO the smaller value BETWEEN the current element at position i and j of dp AND the element at position i MINUS one and j MINUS one of dp PLUS the element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            IF the element at position n MINUS one and j of dp LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
