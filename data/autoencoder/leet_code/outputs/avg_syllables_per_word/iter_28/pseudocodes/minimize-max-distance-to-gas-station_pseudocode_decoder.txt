CLASS Solution  
    FUNCTION minmaxGasDist(stations PARAMETER k)  
        FUNCTION possible(max_dist)  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                INCREMENT additional_stations BY the INTEGER PART OF the VALUE OF element at position index of stations MINUS element at position index MINUS one of stations DIVIDED BY max_dist  
            END FOR  
            IF additional_stations LESS THAN OR EQUAL TO k  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE right MINUS left GREATER THAN zero POINT zero zero zero zero zero one  
            SET mid TO left PLUS right DIVIDED BY two  
            IF possible(mid)  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
