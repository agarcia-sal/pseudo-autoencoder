CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        CREATE dp AS a two dimensional list WITH each element SET TO zero AND dimensions n PLUS one BY n PLUS one  
        FOR j FROM zero TO n  
            SET element at position zero and position j of dp TO one  
        END FOR  
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            IF the element at position i MINUS one of s EQUALS the character I  
                FOR j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position i MINUS one and position j of dp MODULO MOD  
                    SET element at position i and position j of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO prefix_sum PLUS element at position i MINUS one and position j PLUS one of dp MODULO MOD  
                    SET element at position i and position j of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        SET result TO zero  
        FOR each element value IN element at position n of dp  
            SET result TO result PLUS value  
        END FOR  
        SET result TO result MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
