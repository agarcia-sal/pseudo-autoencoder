CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero of strs  
        SET dp TO a list of the LENGTH n where each element is one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF condition all elements at position k of strs at position j LESS THAN OR EQUAL TO elements at position k of strs at position i for k FROM zero TO m MINUS one holds  
                    SET dp at position i TO the GREATER VALUE BETWEEN dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        RETURN n MINUS the MAXIMUM VALUE IN dp  
    END FUNCTION  
END CLASS
