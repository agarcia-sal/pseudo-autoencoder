CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        SET jobs TO list of paired elements where each element is the combination of element at position index of difficulty AND element at position index of profit for each index in the range of the length of difficulty  
        SORT jobs in ascending order by the first element of each paired element  
        SORT worker in ascending order  
        
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the length of jobs  
        
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND the first element of the paired element at position job_index of jobs LESS THAN OR EQUAL TO ability  
                IF the second element of the paired element at position job_index of jobs GREATER THAN best_profit  
                    SET best_profit TO the second element of the paired element at position job_index of jobs  
                END IF  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
