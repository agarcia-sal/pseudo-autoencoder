CLASS Solution  
    FUNCTION isPossible(nums)  
        SET num_count TO a mapping of each number in nums to its frequency  
        SET end_with TO a mapping to count subsequences ending with particular numbers  
        
        FOR each num IN nums  
            IF the frequency of num in num_count EQUALS zero  
                CONTINUE to next iteration  
            END IF  
            
            IF the count of subsequences ending with the number one less than num in end_with IS GREATER THAN zero  
                DECREMENT the count of subsequences ending with the number one less than num in end_with BY one  
                INCREMENT the count of subsequences ending with num in end_with BY one  
            ELSE IF the frequency of the number one greater than num in num_count IS GREATER THAN zero AND the frequency of the number two greater than num in num_count IS GREATER THAN zero  
                DECREMENT the frequency of the number one greater than num in num_count BY one  
                DECREMENT the frequency of the number two greater than num in num_count BY one  
                INCREMENT the count of subsequences ending with the number two greater than num in end_with BY one  
            ELSE  
                RETURN False  
            END IF  
            
            DECREMENT the frequency of num in num_count BY one  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
