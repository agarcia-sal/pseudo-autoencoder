CLASS Solution  
    FUNCTION findClosestElements(arr k x)  
        SET left TO the result of bisect left insertion position of x IN arr MINUS one  
        SET right TO left PLUS one  
        WHILE the difference of right MINUS left MINUS one IS LESS THAN k  
            IF left EQUALS minus one OR the value of right IS LESS THAN the length of arr AND the absolute value of the element at position right of arr MINUS x IS LESS THAN the absolute value of the element at position left of arr MINUS x  
                INCREMENT right BY one  
            ELSE  
                DECREMENT left BY one  
            END IF  
        END WHILE  
        RETURN the sublist of arr FROM position left PLUS one TO position right  
    END FUNCTION  
END CLASS
