CLASS Solution  
    FUNCTION minEatingSpeed(piles, h)  
        FUNCTION canFinish(k)  
            SET hours TO zero  
            FOR each pile IN piles  
                INCREMENT hours BY the smallest integer greater than or equal to the division of pile BY k  
            END FOR  
            RETURN whether hours is less than or equal to h  
        END FUNCTION  

        SET left TO one  
        SET right TO the maximum element in piles  

        WHILE left is less than or equal to right  
            SET mid TO the integer division of the sum of left and right BY two  
            IF canFinish(mid) holds true  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  

        RETURN left  
    END FUNCTION  
END CLASS
