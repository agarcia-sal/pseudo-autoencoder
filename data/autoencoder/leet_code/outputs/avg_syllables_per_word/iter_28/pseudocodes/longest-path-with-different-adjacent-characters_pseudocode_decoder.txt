CLASS Solution  
    FUNCTION longestPath(parent, s)  
        CREATE tree AS a mapping from keys to lists with default as empty list  
        FOR each index i AND element p IN parent  
            IF p NOT EQUALS minus one  
                APPEND i TO the list at key p in tree  
            END IF  
        END FOR  
        
        SET self.result TO one  
        
        FUNCTION dfs(node)  
            SET max1 TO zero  
            SET max2 TO zero  
            
            FOR each child IN the list at key node in tree  
                SET child_length TO the result of dfs called with argument child  
                IF the character at position child of s NOT EQUALS the character at position node of s  
                    IF child_length GREATER THAN max1  
                        SET max2 TO max1  
                        SET max1 TO child_length  
                    ELSE IF child_length GREATER THAN max2  
                        SET max2 TO child_length  
                    END IF  
                END IF  
            END FOR  
            
            SET self.result TO the greater value BETWEEN self.result AND max1 PLUS max2 PLUS one  
            RETURN max1 PLUS one  
        END FUNCTION  
        
        CALL dfs with argument zero  
        RETURN self.result  
    END FUNCTION  
END CLASS
