CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val zero left none right none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode WITH parameter element at position one of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element REMOVED FROM the FRONT of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node dot left TO new TreeNode WITH parameter element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS none  
            SET node dot right TO new TreeNode WITH parameter element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left q dot left AND is_same_tree WITH PARAMETERS p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION subtreeWithAllDeepest WITH PARAMETERS self root  
        FUNCTION dfs WITH PARAMETER node  
            IF node EQUALS none  
                RETURN node AND zero  
            END IF  
            SET left_node AND left_depth TO dfs WITH PARAMETER node dot left  
            SET right_node AND right_depth TO dfs WITH PARAMETER node dot right  
            IF left_depth GREATER THAN right_depth  
                RETURN left_node AND left_depth PLUS one  
            ELSE IF right_depth GREATER THAN left_depth  
                RETURN right_node AND right_depth PLUS one  
            ELSE  
                RETURN node AND left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN the first element OF dfs WITH PARAMETER root  
    END FUNCTION  
END CLASS
