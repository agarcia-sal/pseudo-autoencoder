CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO empty list  
        FOR each building IN buildings  
            SET left TO first element of building  
            SET right TO second element of building  
            SET height TO third element of building  
            APPEND a tuple of left LEFT EDGE NEGATIVE height RIGHT EDGE right TO events  
            APPEND a tuple of right RIGHT EDGE ZERO ZERO TO events  
        END FOR  
        
        SORT events BY the following order first by the first element in ascending order THEN by the second element in ascending order THEN by the third element in ascending order  
        
        SET result TO a list containing a single element which is a list of zero zero  
        SET max_heap TO a list containing a single tuple of zero AND positive infinity  
        
        FOR each event IN events  
            SET x TO first element of event  
            SET negH TO second element of event  
            SET r TO third element of event  
            
            WHILE the second element of the first tuple in max_heap LESS THAN OR EQUAL TO x  
                REMOVE the first tuple from max_heap USING heap pop operation  
            END WHILE  
            
            IF negH NOT EQUALS zero  
                ADD the tuple of negH AND r TO max_heap USING heap push operation  
            END IF  
            
            IF the second element of the last list in result NOT EQUALS the negative of the first element of the first tuple in max_heap  
                APPEND a list containing x AND the negative of the first element of the first tuple in max_heap TO result  
            END IF  
        END FOR  
        
        IF the first list in result EQUALS a list of zero zero  
            REMOVE the first list from result  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
