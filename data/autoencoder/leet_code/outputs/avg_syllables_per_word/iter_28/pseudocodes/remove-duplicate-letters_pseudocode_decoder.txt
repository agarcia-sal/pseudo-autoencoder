CLASS Solution  
    FUNCTION removeDuplicateLetters WITH PARAMETER s  
        SET last_occurrence TO empty dictionary  
        FOR each index AND character IN enumeration of s  
            SET the character key in last_occurrence TO the index  
        END FOR  
        SET stack TO empty list  
        SET in_stack TO empty set  
        FOR each index AND character IN enumeration of s  
            IF the character IS IN in_stack  
                CONTINUE to the next iteration of the loop  
            END IF  
            WHILE stack IS NOT empty AND the character IS LESS THAN the element at the last position of stack AND index IS LESS THAN the value in last_occurrence with key the element at the last position of stack  
                SET removed_char TO the element removed from the end of stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            APPEND the character TO stack  
            ADD the character TO in_stack  
        END FOR  
        SET result_string TO the concatenation of all elements of stack in order  
        RETURN result_string  
    END FUNCTION  
END CLASS
