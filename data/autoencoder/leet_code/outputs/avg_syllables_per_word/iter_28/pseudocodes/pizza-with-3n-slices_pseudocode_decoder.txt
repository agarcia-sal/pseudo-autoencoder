CLASS Solution  
    FUNCTION maxSizeSlices WITH PARAMETER slices  
        SET n TO the LENGTH OF slices DIVIDED BY three  
        
        FUNCTION dp WITH PARAMETERS start end k  
            IF k EQUALS zero OR start GREATER THAN end  
                RETURN zero  
            END IF  
            
            SET pick TO the element at position start of slices PLUS dp CALLED WITH start PLUS two end k MINUS one  
            SET skip TO dp CALLED WITH start PLUS one end k  
            RETURN the greater value BETWEEN pick AND skip  
        END FUNCTION  
        
        SET case1 TO dp CALLED WITH one LENGTH OF slices MINUS one n  
        SET case2 TO dp CALLED WITH zero LENGTH OF slices MINUS two n  
        
        RETURN the greater value BETWEEN case1 AND case2  
    END FUNCTION  
END CLASS
