CLASS Solution  
    FUNCTION repeatedStringMatch(a string, b string) RETURNS integer  
        SET min_repeats TO the smallest integer greater than or equal to the LENGTH OF b DIVIDED BY the LENGTH OF a  
        IF the string b is contained within the string a repeated min_repeats times  
            RETURN min_repeats  
        END IF  
        IF the string b is contained within the string a repeated min_repeats PLUS one times  
            RETURN min_repeats PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
