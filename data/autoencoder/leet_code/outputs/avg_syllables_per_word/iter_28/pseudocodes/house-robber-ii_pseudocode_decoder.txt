CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF the collection houses EQUALS empty  
                RETURN zero  
            END IF  
            IF the LENGTH OF the collection houses EQUALS one  
                RETURN the element at position one of the collection houses  
            END IF  
            SET dp TO a new collection of zero values with the LENGTH OF the collection houses  
            SET the element at position one of dp TO the element at position one of the collection houses  
            SET the element at position two of dp TO the GREATER VALUE BETWEEN the element at position one of the collection houses AND the element at position two of the collection houses  
            FOR index i FROM three TO the LENGTH OF the collection houses  
                SET the element at position i of dp TO the GREATER VALUE BETWEEN the element at position i MINUS one of dp AND the sum of the element at position i MINUS two of dp PLUS the element at position i of the collection houses  
            END FOR  
            RETURN the element at the last position of dp  
        END FUNCTION  
        IF the collection nums EQUALS empty  
            RETURN zero  
        END IF  
        IF the LENGTH OF the collection nums EQUALS one  
            RETURN the element at position one of the collection nums  
        END IF  
        RETURN the GREATER VALUE BETWEEN rob_linear(the collection nums WITHOUT the last element) AND rob_linear(the collection nums WITHOUT the first element)  
    END FUNCTION  
END CLASS
