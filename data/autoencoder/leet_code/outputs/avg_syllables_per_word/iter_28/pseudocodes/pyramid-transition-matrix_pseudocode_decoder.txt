CLASS Solution  
    FUNCTION pyramidTransition bottom ALLOWED  
        SET allowed_map TO a mapping from string to a mapping from string to a list of strings with default empty lists  
        FOR each rule IN ALLOWED  
            SET left TO element at position one of rule  
            SET right TO element at position two of rule  
            SET top TO element at position three of rule  
            APPEND top TO the list at allowed_map at key left and key right  
        END FOR

        FUNCTION can_build_pyramid current_bottom current_top  
            IF the LENGTH OF current_bottom EQUALS one  
                RETURN True  
            END IF  
            IF the LENGTH OF current_top EQUALS the LENGTH OF current_bottom MINUS one  
                RETURN the result of calling can_build_pyramid with current_top and an empty list  
            END IF  

            SET left TO element at position the LENGTH OF current_top PLUS one of current_bottom  
            SET right TO element at position the LENGTH OF current_top PLUS two of current_bottom  
            IF left IS IN allowed_map AND right IS IN the mapping at allowed_map for key left  
                FOR each top IN the list at allowed_map for key left and key right  
                    IF calling can_build_pyramid with current_bottom and current_top appended with top RETURNS True  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
            RETURN False  
        END FUNCTION

        RETURN the result of calling can_build_pyramid with bottom and an empty list  
    END FUNCTION  
END CLASS
