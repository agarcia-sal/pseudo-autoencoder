CLASS Solution  
    FUNCTION rearrangeSticks(n k)  
        SET MOD TO ten to the power of nine PLUS seven  
        CREATE dp AS list of lists with dimensions n PLUS one by k PLUS one FILLED WITH zeroes  
        SET element at position zero of list at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET element at position j of list at position i of dp TO the sum of element at position j MINUS one of list at position i MINUS one of dp AND the product of i MINUS one AND element at position j of list at position i MINUS one of dp TAKEN MODULO MOD  
            END FOR  
        END FOR  
        RETURN element at position k of list at position n of dp  
    END FUNCTION  
END CLASS
