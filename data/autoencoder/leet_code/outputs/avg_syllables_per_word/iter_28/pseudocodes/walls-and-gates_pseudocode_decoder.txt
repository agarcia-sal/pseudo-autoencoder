CLASS Solution  
    FUNCTION wallsAndGates(rooms)  
        IF rooms EQUALS nothing OR element at position zero of rooms EQUALS nothing THEN  
            RETURN  
        END IF  
        SET rows TO the LENGTH OF rooms  
        SET cols TO the LENGTH OF element at position zero of rooms  
        ASSIGN reference TO new empty double ended queue named queue  
        FOR variable r FROM zero TO rows MINUS one  
            FOR variable c FROM zero TO cols MINUS one  
                IF element at position r of rooms AND element at position c of element at position r of rooms EQUALS zero THEN  
                    APPEND tuple of r AND c TO queue  
                END IF  
            END FOR  
        END FOR  
        ASSIGN reference TO list of tuples representing direction shifts up zero AND one down one AND zero left zero AND negative one right negative one AND zero TO variable named directions  
        WHILE queue holds any elements  
            REMOVE first element FROM queue AND ASSIGN first item to variable r AND second item to variable c  
            FOR each tuple of dr AND dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN rows AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN cols AND element at position nr of rooms AND element at position nc of element at position nr of rooms EQUALS the maximum integer value representable THEN  
                    SET element at position nr of rooms AND element at position nc of element at position nr of rooms TO element at position r of rooms AND element at position c of element at position r of rooms PLUS one  
                    APPEND tuple of nr AND nc TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
