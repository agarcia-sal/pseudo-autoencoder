CLASS Solution  
    FUNCTION exclusiveTime(n, logs)  
        SET exclusive_time TO a list of zero values with the length equal to n  
        SET stack TO an empty list  
        
        FOR each log IN logs  
            SPLIT log INTO function_id AND action AND timestamp by the delimiter colon  
            SET function_id TO function_id converted to integer  
            SET timestamp TO timestamp converted to integer  
            
            IF action EQUALS the word start  
                IF the stack is not empty  
                    SET prev_function_id TO the first element of the last item in the stack  
                    SET prev_timestamp TO the second element of the last item in the stack  
                    INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND a tuple consisting of function_id AND timestamp TO stack  
            ELSE  
                REMOVE the last item FROM stack AND ASSIGN its first element TO prev_function_id AND second element TO prev_timestamp  
                INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp PLUS one  
                IF the stack is not empty  
                    SET the second element of the last item in stack TO timestamp PLUS one while keeping the first element unchanged  
                END IF  
            END IF  
        END FOR  
        
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
