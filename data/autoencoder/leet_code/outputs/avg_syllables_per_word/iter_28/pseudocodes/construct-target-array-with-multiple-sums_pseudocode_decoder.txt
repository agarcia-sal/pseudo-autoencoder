CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO empty list  
        FOR each num IN target  
            APPEND negative of num TO max_heap  
        END FOR  
        TRANSFORM max_heap INTO a heap structure by heapifying it  
        
        SET total_sum TO the sum of all elements in target  
        
        WHILE the element at position one of max_heap LESS THAN negative one holds  
            SET largest TO negative of the element obtained by removing the smallest element from max_heap heap  
            SET rest_sum TO total_sum MINUS largest  
            
            IF largest LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero  
                RETURN False  
            END IF  
            
            SET new_value TO largest MODULO rest_sum  
            IF new_value EQUALS zero AND rest_sum NOT EQUALS one  
                RETURN False  
            END IF  
            
            SET total_sum TO rest_sum PLUS new_value  
            INSERT negative of new_value INTO max_heap heap  
        END WHILE  
        
        RETURN True  
    END FUNCTION  
END CLASS
