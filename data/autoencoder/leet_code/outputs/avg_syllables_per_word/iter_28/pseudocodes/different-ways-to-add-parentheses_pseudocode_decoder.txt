CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each element l IN left  
                FOR each element r IN right  
                    IF operator EQUALS the character representing plus sign  
                        APPEND the sum of l PLUS r TO results  
                    ELSE IF operator EQUALS the character representing minus sign  
                        APPEND the difference of l MINUS r TO results  
                    ELSE IF operator EQUALS the character representing multiplication sign  
                        APPEND the product of l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        FUNCTION helper(sub_expr)  
            IF the string sub_expr represents only digit characters  
                RETURN a list containing the integer value of sub_expr  
            END IF  
            
            SET results TO empty list  
            FOR index i FROM zero TO the length of sub_expr MINUS one  
                IF the character at position i of sub_expr EQUALS the character representing plus sign OR EQUALS the character representing minus sign OR EQUALS the character representing multiplication sign  
                    SET left_results TO the result of calling helper with the substring of sub_expr from the beginning TO the position before index i  
                    SET right_results TO the result of calling helper with the substring of sub_expr from the position after index i TO the end  
                    EXTEND results BY the result of calling compute with left_results, right_results, and the character at position i of sub_expr  
                END IF  
            END FOR  
            
            RETURN results  
        END FUNCTION  
        
        RETURN the result of calling helper with expression  
    END FUNCTION  
END CLASS
