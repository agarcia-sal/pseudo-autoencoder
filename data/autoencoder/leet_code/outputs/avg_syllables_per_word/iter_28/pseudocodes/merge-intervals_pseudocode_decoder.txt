CLASS Solution  
    FUNCTION merge(intervals)  
        SORT intervals BY the first element at position zero of each interval  
        SET merged TO an empty list  
        FOR each interval IN intervals  
            IF the list merged does not contain any elements OR the element at position one of the last interval in merged is LESS THAN the element at position zero of the current interval  
                APPEND the current interval TO merged  
            ELSE  
                SET the element at position one of the last interval in merged TO the GREATER THAN OR EQUAL TO value BETWEEN the element at position one of the last interval in merged AND the element at position one of the current interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
