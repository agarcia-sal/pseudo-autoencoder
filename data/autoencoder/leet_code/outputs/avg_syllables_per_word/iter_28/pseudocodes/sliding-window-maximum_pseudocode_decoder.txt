CLASS Solution  
    FUNCTION maxSlidingWindow(nums parameter FROM List OF integer, k parameter FROM integer) RETURNS List OF integer  
        IF nums EQUALS empty collection OR k EQUALS zero THEN  
            RETURN empty collection  
        END IF  
        SET deq TO new empty double ended queue  
        SET result TO new empty collection  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT empty AND the element at position zero of deq IS LESS THAN i MINUS k PLUS one THEN  
                REMOVE the element at position zero FROM deq  
            END IF  
            WHILE deq IS NOT empty AND the element at position element at the last position of deq of nums IS LESS THAN the element at position i of nums  
                REMOVE the element at the last position FROM deq  
            END WHILE  
            APPEND i TO deq  
            IF i IS GREATER THAN OR EQUAL TO k MINUS one THEN  
                APPEND the element at position element at position zero of deq of nums TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
