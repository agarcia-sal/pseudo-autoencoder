CLASS Solution  
    FUNCTION maxSum(nums1 nums1 AS List OF integers nums2 nums2 AS List OF integers) RETURNS integer  
        SET i TO zero  
        SET j TO zero  
        SET sum1 TO zero  
        SET sum2 TO zero  
        SET MOD TO ten RAISED TO the power of nine PLUS one  
        
        WHILE i LESS THAN the LENGTH OF nums1 AND j LESS THAN the LENGTH OF nums2  
            IF the element at position i of nums1 LESS THAN the element at position j of nums2  
                INCREMENT sum1 BY the element at position i of nums1  
                INCREMENT i BY one  
            ELSE IF the element at position i of nums1 GREATER THAN the element at position j of nums2  
                INCREMENT sum2 BY the element at position j of nums2  
                INCREMENT j BY one  
            ELSE IF the element at position i of nums1 EQUALS the element at position j of nums2  
                SET sum1 TO the maximum value BETWEEN sum1 AND sum2 PLUS the element at position i of nums1  
                SET sum2 TO sum1  
                INCREMENT i BY one  
                INCREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i LESS THAN the LENGTH OF nums1  
            INCREMENT sum1 BY the element at position i of nums1  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE j LESS THAN the LENGTH OF nums2  
            INCREMENT sum2 BY the element at position j of nums2  
            INCREMENT j BY one  
        END WHILE  
        
        RETURN the maximum value BETWEEN sum1 AND sum2 MODULO MOD  
    END FUNCTION  
END CLASS
