    FUNCTION kthSmallestPrimeFraction(arr parameter LIST OF INTEGERS k parameter INTEGER) RETURNS LIST OF INTEGERS  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an EMPTY list  
        FOR i FROM zero TO n MINUS one MINUS one  
            COMPUTE the fraction BETWEEN element at position i of arr DIVIDED BY element at position n MINUS one of arr  
            PUSH a tuple consisting of the computed fraction AND i AND n MINUS one INTO min_heap USING heap push operation  
        END FOR  
        FOR each iteration FROM zero TO k MINUS one MINUS one  
            POP the smallest element FROM min_heap USING heap pop operation INTO variables unused_fraction AND i AND j  
            IF j MINUS one GREATER THAN i THEN  
                COMPUTE the fraction BETWEEN element at position i of arr DIVIDED BY element at position j MINUS one of arr  
                PUSH a tuple consisting of the computed fraction AND i AND j MINUS one INTO min_heap USING heap push operation  
            END IF  
        END FOR  
        POP the smallest element FROM min_heap USING heap pop operation INTO variables unused_fraction AND i AND j  
        RETURN a list consisting of element at position i of arr AND element at position j of arr  
    END FUNCTION  
END CLASS
