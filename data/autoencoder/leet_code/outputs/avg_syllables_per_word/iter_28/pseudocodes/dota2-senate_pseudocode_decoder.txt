CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        ASSIGN reference TO a new empty double ended queue radiant  
        ASSIGN reference TO a new empty double ended queue dire  
        FOR each index i AND each character s IN senate  
            IF character s EQUALS letter R  
                APPEND index i TO queue radiant  
            ELSE  
                APPEND index i TO queue dire  
            END IF  
        END FOR  
        WHILE the queue radiant IS NOT empty AND the queue dire IS NOT empty  
            REMOVE and ASSIGN the element at the front of queue radiant TO r_index  
            REMOVE and ASSIGN the element at the front of queue dire TO d_index  
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the LENGTH OF senate TO queue radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO queue dire  
            END IF  
        END WHILE  
        IF the queue radiant IS NOT empty  
            RETURN the word Radiant  
        ELSE  
            RETURN the word Dire  
        END IF  
    END FUNCTION  
END CLASS
