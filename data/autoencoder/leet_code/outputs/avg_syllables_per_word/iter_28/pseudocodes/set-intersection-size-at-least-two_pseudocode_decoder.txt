CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals BY the position at the end of each interval ASCENDING AND the position at the start of each interval DESCENDING  
        SET first TO the position at the end of the first interval MINUS one  
        SET second TO the position at the end of the first interval  
        SET count TO two  
        FOR each start AND end IN intervals skipping the first element  
            IF start GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
