CLASS Solution  
    FUNCTION largestSumOfAverages(nums list of integers k integer) RETURNS float  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of zero values with the LENGTH OF n PLUS one  
        FOR i FROM zero TO n MINUS one  
            SET element at position i PLUS one of prefix_sum TO element at position i of prefix_sum PLUS element at position i of nums  
        END FOR  
        
        FUNCTION average(i integer j integer) RETURNS float  
            RETURN element at position j PLUS one of prefix_sum MINUS element at position i of prefix_sum DIVIDED BY j MINUS i PLUS one  
        END FUNCTION  
        
        SET dp TO a list of lists each containing zero values with the LENGTH OF k PLUS one for each element in a list of LENGTH n PLUS one  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i at index one of dp TO average(i n MINUS one)  
        END FOR  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    IF element at position i at index partitions of dp LESS THAN element at position j at index partitions MINUS one of dp PLUS average(i j MINUS one)  
                        SET element at position i at index partitions of dp TO element at position j at index partitions MINUS one of dp PLUS average(i j MINUS one)  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero at index k of dp  
    END FUNCTION  
END CLASS
