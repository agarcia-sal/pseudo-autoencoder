CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one  
        END IF  
        
        SET n TO n PLUS twenty four DIVIDED BY twenty five rounded down to integer division  
        
        SET memo TO an empty dictionary  
        
        FUNCTION dp(a, b)  
            IF the pair of a and b EXISTS IN memo  
                RETURN the value in memo FOR the pair of a and b  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN point five  
            END IF  
            
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one  
            END IF  
            
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero  
            END IF  
            
            SET prob TO point two five MULTIPLIED BY the sum of dp of a MINUS four and b PLUS dp of a MINUS three and b MINUS one PLUS dp of a MINUS two and b MINUS two PLUS dp of a MINUS one and b MINUS three  
            
            SET memo at the pair of a and b TO prob  
            RETURN prob  
        END FUNCTION  
        
        RETURN dp with arguments n and n  
    END FUNCTION  
END CLASS
