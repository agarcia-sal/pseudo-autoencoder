CLASS Solution  
    FUNCTION countSubranges(nums1 LIST OF integer nums2 LIST OF integer) RETURNS integer  
        SET n TO the LENGTH OF nums1  
        SET count TO zero  
        SET dp TO a list of LENGTH n where each element is a new empty mapping from integer TO integer with default value zero  
        
        FOR i FROM zero TO n MINUS one  
            INCREMENT the value at key element at position i of nums1 in the mapping at element at position i of dp BY one  
            INCREMENT the value at key NEGATIVE element at position i of nums2 in the mapping at element at position i of dp BY one  
            
            IF i GREATER THAN zero THEN  
                FOR each key balance AND value freq IN the mapping at element at position i MINUS one of dp  
                    SET the value at key balance PLUS element at position i of nums1 in the mapping at element at position i of dp TO the sum of the current value at the same key in the same mapping PLUS freq MODULO ten raised to the power of nine PLUS seven  
                    SET the value at key balance MINUS element at position i of nums2 in the mapping at element at position i of dp TO the sum of the current value at the same key in the same mapping PLUS freq MODULO ten raised to the power of nine PLUS seven  
                END FOR  
            END IF  
            
            SET count TO the sum of count PLUS the value at key zero in the mapping at element at position i of dp MODULO ten raised to the power of nine PLUS seven  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
