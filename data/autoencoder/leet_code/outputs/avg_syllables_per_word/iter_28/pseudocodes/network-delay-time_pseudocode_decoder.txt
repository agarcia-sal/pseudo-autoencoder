CLASS Solution  
    FUNCTION networkDelayTime(times, n, k)  
        SET graph TO an empty mapping from each integer from one TO n inclusive TO an empty list  
        FOR each element WITH components u v w IN times  
            APPEND a pair of neighbor v AND weight w TO the list at key u in graph  
        END FOR  
        
        SET dist TO a mapping from each integer from one TO n inclusive TO infinity  
        SET dist at key k TO zero  
        
        SET min_heap TO a list containing a pair of zero and k  
        
        WHILE min_heap is not empty  
            EXTRACT the smallest element from min_heap INTO variables current_dist AND node  
            
            IF current_dist GREATER THAN dist at key node  
                CONTINUE the loop  
            END IF  
            
            FOR each element WITH components neighbor AND weight IN the list at key node in graph  
                SET distance TO current_dist PLUS weight  
                
                IF distance LESS THAN dist at key neighbor  
                    SET dist at key neighbor TO distance  
                    INSERT the pair distance AND neighbor INTO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        SET max_time TO the maximum value among all values in dist  
        
        IF max_time EQUALS infinity  
            RETURN negative one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
