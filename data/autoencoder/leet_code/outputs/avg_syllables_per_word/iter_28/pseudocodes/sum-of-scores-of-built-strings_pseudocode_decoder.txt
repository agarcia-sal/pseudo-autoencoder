CLASS Solution  
    FUNCTION sumScores(s)  
        SET n TO the length of s  
        SET lcp TO an empty list of size n filled with zeroes  
        SET l TO zero  
        SET r TO zero  
        FOR i FROM one TO n MINUS one  
            IF i IS LESS THAN OR EQUAL TO r  
                SET lcp element at position i TO the smaller value BETWEEN r MINUS i PLUS one AND lcp element at position i MINUS l  
            END IF  
            WHILE i PLUS lcp element at position i IS LESS THAN n AND element at position lcp element at position i of s EQUALS element at position i PLUS lcp element at position i of s  
                INCREMENT lcp element at position i BY one  
            END WHILE  
            IF i PLUS lcp element at position i MINUS one IS GREATER THAN r  
                SET l TO i  
                SET r TO i PLUS lcp element at position i MINUS one  
            END IF  
        END FOR  
        RETURN the sum of all elements in lcp PLUS n  
    END FUNCTION  
END CLASS
