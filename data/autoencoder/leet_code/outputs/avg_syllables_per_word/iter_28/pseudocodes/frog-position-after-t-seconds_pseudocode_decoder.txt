CLASS Solution  
    FUNCTION frogPosition(n, edges, t, target)  
        IF n EQUALS one  
            RETURN one  
        END IF  
        
        SET graph TO a default dictionary mapping to a list  
        FOR each pair u v IN edges  
            APPEND v TO the list at graph indexed by u  
            APPEND u TO the list at graph indexed by v  
        END FOR  
        
        SET queue TO a double ended queue containing tuple of one current node current probability and current time with values one one one zero  
        SET visited TO a set containing one  
        
        WHILE queue is not empty  
            SET current node current probability current time TO the first element removed from queue  
            
            IF current time EQUALS t OR the current node does not equal one AND the number of elements in the list at graph indexed by current node EQUALS one  
                IF current node EQUALS target  
                    RETURN current probability  
                ELSE  
                    CONTINUE to the next iteration of the WHILE loop  
                END IF  
            END IF  
            
            SET possible moves TO the number of elements in the list at graph indexed by current node MINUS one if current node does not equal one otherwise zero  
            
            FOR each neighbor IN the list at graph indexed by current node  
                IF neighbor is not an element of visited  
                    ADD neighbor to visited  
                    APPEND tuple of neighbor current probability DIVIDED BY possible moves current time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN zero  
    END FUNCTION  
END CLASS
