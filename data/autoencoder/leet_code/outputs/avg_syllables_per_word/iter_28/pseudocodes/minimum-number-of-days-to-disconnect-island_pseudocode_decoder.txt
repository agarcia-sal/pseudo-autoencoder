CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the pair of x AND y IS IN visited OR the element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            ADD the pair of x AND y TO visited  
            FOR each pair of dx AND dy IN the list containing the pair of minus one AND zero the pair of one AND zero the pair of zero AND minus one the pair of zero AND one  
                CALL dfs WITH x PLUS dx AND y PLUS dy AND visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an empty set  
            SET land_cells TO a list containing each pair of i AND j such that i ranges FROM zero TO the LENGTH OF grid MINUS one AND j ranges FROM zero TO the LENGTH OF element at position zero of grid MINUS one AND the element at position i of grid at position j EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero  
                RETURN true  
            END IF  
            CALL dfs WITH the first element of land_cells AT position zero AND the first element of land_cells AT position one AND visited  
            RETURN the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF NOT is_connected()  
            RETURN zero  
        END IF  

        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF the element at position i of grid at position j EQUALS one  
                    SET the element at position i of grid at position j TO zero  
                    IF NOT is_connected()  
                        RETURN one  
                    END IF  
                    SET the element at position i of grid at position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
