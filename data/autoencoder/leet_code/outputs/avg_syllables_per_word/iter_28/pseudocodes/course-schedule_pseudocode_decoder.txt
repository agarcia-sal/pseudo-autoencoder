CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        ASSIGN reference TO an empty mapping from element TO a list  
        SET indegree TO a list of zeroes of length numCourses  
        FOR each pair of elements course AND prereq IN prerequisites  
            APPEND course TO the list at key prereq in the mapping  
            INCREMENT the element at position course of indegree BY one  
        END FOR  
        SET queue TO a double ended queue containing each element course FROM zero TO numCourses MINUS one WHERE the element at position course of indegree EQUALS zero  
        WHILE queue is not empty  
            SET current_course TO the removal of the element from the left of queue  
            DECREMENT numCourses BY one  
            FOR each element next_course IN the list at key current_course in the mapping  
                DECREMENT the element at position next_course of indegree BY one  
                IF the element at position next_course of indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN the result of numCourses EQUALS zero  
    END FUNCTION  
END CLASS
