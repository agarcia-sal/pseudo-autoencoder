CLASS Solution  
    FUNCTION numDistinctIslands2(grid)  
        FUNCTION dfs(x, y, island)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            SET element at position x of grid at position y TO zero  
            APPEND pair of x and y TO island  
            CALL dfs WITH x PLUS one AND y AND island  
            CALL dfs WITH x MINUS one AND y AND island  
            CALL dfs WITH x AND y PLUS one AND island  
            CALL dfs WITH x AND y MINUS one AND island  
        END FUNCTION  
        
        FUNCTION normalize(island)  
            SET shapes TO empty list  
            FOR each pair dx and dy IN list of pairs one and one PLUS one and MINUS one PLUS MINUS one and one PLUS MINUS one and MINUS one  
                SET new_island TO empty list  
                FOR each pair x and y IN island  
                    APPEND pair of x MULTIPLIED BY dx AND y MULTIPLIED BY dy TO new_island  
                END FOR  
                SORT new_island IN ascending order  
                SET min_x TO minimum x among elements in new_island  
                SET min_y TO minimum y among elements in new_island  
                APPEND tuple of pairs of x MINUS min_x AND y MINUS min_y FOR each pair x and y IN new_island TO shapes  
            END FOR  
            FOR index i FROM zero TO the LENGTH OF shapes MINUS one  
                APPEND tuple of pairs of y AND the NEGATION of x FOR each pair x and y IN element at position i of shapes TO shapes  
            END FOR  
            RETURN minimum element of shapes  
        END FUNCTION  
        
        SET islands TO empty set  
        FOR index i FROM zero TO the LENGTH OF grid MINUS one  
            FOR index j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    SET island TO empty list  
                    CALL dfs WITH i AND j AND island  
                    ADD the result of normalize called with island TO islands  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the LENGTH OF islands  
    END FUNCTION  
END CLASS
