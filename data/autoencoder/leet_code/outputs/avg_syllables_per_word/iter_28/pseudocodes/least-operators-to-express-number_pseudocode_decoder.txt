CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        FUNCTION dp(target)  
            IF target EQUALS zero  
                RETURN negative one  
            END IF  
            IF target LESS THAN x  
                RETURN the minimum OF the value of two MULTIPLIED BY target MINUS one AND two MULTIPLIED BY x MINUS target  
            END IF  
            SET power TO one  
            WHILE x RAISED TO THE power LESS THAN OR EQUAL TO target  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET res TO the value of dp OF target MINUS x RAISED TO THE power PLUS power  
            IF x RAISED TO THE power PLUS one MINUS target LESS THAN target  
                SET res TO the minimum OF res AND dp OF x RAISED TO THE power PLUS one MINUS target PLUS power PLUS one  
            END IF  
            RETURN res  
        END FUNCTION  
        RETURN dp OF target  
    END FUNCTION  
END CLASS
