CLASS Solution  
    FUNCTION visiblePoints(points OF LIST OF LIST OF INTEGER angle OF INTEGER location OF LIST OF INTEGER) RETURNS INTEGER  
        SET angles TO empty list  
        SET extra_points TO zero  
        FOR each x AND y IN points  
            IF x EQUALS element at position zero OF location AND y EQUALS element at position one OF location  
                INCREMENT extra_points BY one  
                CONTINUE to next iteration  
            END IF  
            SET angle_in_radians TO arc tangent of the difference of y MINUS element at position one OF location AND the difference of x MINUS element at position zero OF location  
            SET angle_in_degrees TO degrees_conversion OF angle_in_radians  
            APPEND angle_in_degrees TO angles  
        END FOR  
        SORT angles in ascending order  
        SET extended_angles TO angles concatenated with each element a PLUS three hundred sixty FOR each a IN angles  
        SET max_visible TO zero  
        SET left TO zero  
        FOR each right FROM zero TO one LESS THAN the LENGTH OF extended_angles  
            WHILE element at position right OF extended_angles MINUS element at position left OF extended_angles GREATER THAN angle  
                INCREMENT left BY one  
            END WHILE  
            SET window_size TO right MINUS left PLUS one  
            IF window_size GREATER THAN max_visible  
                SET max_visible TO window_size  
            END IF  
        END FOR  
        RETURN max_visible PLUS extra_points  
    END FUNCTION  
END CLASS
