CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY THEN  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode with val being element at position one of values and left being none and right being none  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO remove element from left end of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.left TO new TreeNode with val being element at position i of values and left being none and right being none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.right TO new TreeNode with val being element at position i of values and left being none and right being none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN True  
    ELSE IF p IS none OR q IS none THEN  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self, root)  
        SET queue TO new double ended queue containing root  
        SET leftmost_value TO none  
        WHILE queue IS NOT EMPTY  
            SET level_size TO the LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one  
                SET node TO remove element from left end of queue  
                IF i EQUALS zero THEN  
                    SET leftmost_value TO node.val  
                END IF  
                IF node.left IS NOT none THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT none THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
