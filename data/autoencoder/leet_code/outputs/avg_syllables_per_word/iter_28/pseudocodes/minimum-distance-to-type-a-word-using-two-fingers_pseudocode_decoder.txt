CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS None  
                RETURN zero  
            END IF  
            SET x1 TO the integer division of the Unicode code point of c1 MINUS the Unicode code point of the uppercase letter A DIVIDED BY six  
            SET y1 TO the remainder of the division of the Unicode code point of c1 MINUS the Unicode code point of the uppercase letter A BY six  
            SET x2 TO the integer division of the Unicode code point of c2 MINUS the Unicode code point of the uppercase letter A DIVIDED BY six  
            SET y2 TO the remainder of the division of the Unicode code point of c2 MINUS the Unicode code point of the uppercase letter A BY six  
            RETURN the absolute value of x1 MINUS x2 PLUS the absolute value of y1 MINUS y2  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the length of word  
                RETURN zero  
            END IF  
            SET c TO the element at position i of word  
            SET d1 TO the result of distance with arguments f1 and c PLUS the result of dp with arguments i PLUS one f1 c  
            SET d2 TO the result of distance with arguments f2 and c PLUS the result of dp with arguments i PLUS one f1 c  
            RETURN the smaller value between d1 AND d2  
        END FUNCTION  
        
        RETURN the result of dp with arguments zero None None  
    END FUNCTION  
END CLASS
