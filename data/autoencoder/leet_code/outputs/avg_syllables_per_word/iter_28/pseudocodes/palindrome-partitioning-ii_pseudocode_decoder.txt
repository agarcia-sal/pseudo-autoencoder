CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        
        SET is_palindrome TO a two dimensional list of size n BY n WHERE each element IS False  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i at position i of is_palindrome TO True  
        END FOR  
        
        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET element at position start at position end of is_palindrome TO the character at position start of s EQUALS the character at position end of s  
                ELSE  
                    SET element at position start at position end of is_palindrome TO the character at position start of s EQUALS the character at position end of s AND element at position start PLUS one at position end MINUS one of is_palindrome  
                END IF  
            END FOR  
        END FOR  
        
        SET min_cuts TO a list of size n WHERE each element IS zero  
        FOR i FROM zero TO n MINUS one  
            IF element at position zero at position i of is_palindrome  
                SET element at position i of min_cuts TO zero  
            ELSE  
                SET element at position i of min_cuts TO positive infinity  
                FOR j FROM one TO i  
                    IF element at position j at position i of is_palindrome AND element at position j MINUS one of min_cuts PLUS one LESS THAN element at position i of min_cuts  
                        SET element at position i of min_cuts TO element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN element at position n MINUS one of min_cuts  
    END FUNCTION  
END CLASS
