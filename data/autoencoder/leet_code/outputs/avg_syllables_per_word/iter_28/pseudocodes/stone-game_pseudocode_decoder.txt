CLASS Solution  
    FUNCTION stoneGame(piles)  
        FUNCTION dfs(i, j)  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            RETURN the greater value BETWEEN element at position i of piles MINUS dfs(i PLUS one, j) AND element at position j of piles MINUS dfs(i, j MINUS one)  
        END FUNCTION  
        RETURN dfs(zero, the length of piles MINUS one) GREATER THAN zero  
    END FUNCTION  
END CLASS
