CLASS Solution  
    FUNCTION rangeBitwiseAnd(left INTEGER_PARAMETER right INTEGER_PARAMETER) RETURNS INTEGER  
        SET shift TO zero  
        WHILE left LESS THAN right  
            SET left TO integer division of left BY two  
            SET right TO integer division of right BY two  
            INCREMENT shift BY one  
        END WHILE  
        RETURN left MULTIPLIED BY two RAISED TO THE POWER OF shift  
    END FUNCTION  
END CLASS
