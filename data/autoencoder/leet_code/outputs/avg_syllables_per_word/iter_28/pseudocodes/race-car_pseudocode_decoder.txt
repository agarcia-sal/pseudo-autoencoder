CLASS Solution  
    FUNCTION racecar(target)  
        SET queue TO a double ended queue initialized with a tuple of zero position one speed and zero steps  
        SET visited TO a set initialized with a tuple of zero position and one speed  
        
        WHILE the LENGTH OF queue is greater than zero  
            REMOVE and ASSIGN the element at the front of queue TO position speed and steps  
            
            IF position EQUALS target  
                RETURN steps  
            END IF  
            
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple of next_position and next_speed NOT IN visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of next_position and next_speed TO visited  
                APPEND the tuple of next_position next_speed and steps PLUS one TO queue  
            END IF  
            
            IF speed IS GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple of position and next_speed NOT IN visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target  
                ADD the tuple of position and next_speed TO visited  
                APPEND the tuple of position next_speed and steps PLUS one TO queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
