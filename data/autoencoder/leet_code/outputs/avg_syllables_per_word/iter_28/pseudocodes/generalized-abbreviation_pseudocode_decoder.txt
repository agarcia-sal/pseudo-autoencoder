CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the string form of count TO path  
                END IF  
                APPEND the concatenation of all elements in path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH index INCREMENTED BY one AND path APPENDED WITH the string form of count AND element at position index of word AND count SET TO zero  
                ELSE  
                    CALL backtrack WITH index INCREMENTED BY one AND path APPENDED WITH element at position index of word AND count SET TO zero  
                END IF  
                CALL backtrack WITH index INCREMENTED BY one AND path UNCHANGED AND count INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH zero AND an empty list AND zero  
        RETURN result  
    END FUNCTION  
END CLASS
