CLASS Solution  
    FUNCTION countAndSay WITH parameter n  
        IF n EQUALS one  
            RETURN the string representing one  
        END IF  
        SET current_sequence TO the string representing one  
        FOR each number from two TO n  
            SET next_sequence TO empty string  
            SET count TO one  
            SET previous_char TO element at position one of current_sequence  
            FOR each index i from two TO the LENGTH OF current_sequence  
                IF element at position i of current_sequence EQUALS previous_char  
                    INCREMENT count BY one  
                ELSE  
                    APPEND string representation of count TO next_sequence  
                    APPEND previous_char TO next_sequence  
                    SET previous_char TO element at position i of current_sequence  
                    SET count TO one  
                END IF  
            END FOR  
            APPEND string representation of count TO next_sequence  
            APPEND previous_char TO next_sequence  
            SET current_sequence TO next_sequence  
        END FOR  
        RETURN current_sequence  
    END FUNCTION  
END CLASS
