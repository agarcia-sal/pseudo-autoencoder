CLASS Solution  
    FUNCTION boldWords(words list OF strings, s string) RETURNS string  
        SET bold TO list of boolean values initialized to false with the LENGTH OF s  
        
        FOR each word IN words  
            SET start TO the POSITION OF the FIRST OCCURRENCE OF word IN s  
            WHILE start NOT EQUALS negative one  
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET element at position i OF bold TO true  
                END FOR  
                SET start TO the POSITION OF the NEXT OCCURRENCE OF word IN s STARTING FROM start PLUS one  
            END WHILE  
        END FOR  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF s  
            IF element at position i OF bold EQUALS true  
                APPEND the string that represents opening bold tag TO result  
                WHILE i LESS THAN the LENGTH OF s AND element at position i OF bold EQUALS true  
                    APPEND element at position i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string that represents closing bold tag TO result  
            ELSE  
                APPEND element at position i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements IN result  
    END FUNCTION  
END CLASS
