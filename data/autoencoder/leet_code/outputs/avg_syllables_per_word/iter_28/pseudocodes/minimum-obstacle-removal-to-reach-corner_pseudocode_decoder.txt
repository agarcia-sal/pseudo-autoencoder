CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO a list of four pairs representing right one step down one step left one step up one step  
        SET min_heap TO a list containing a triple of zero zero zero representing cost and positions  
        SET visited TO a set containing the pair zero zero representing starting position  
        
        WHILE min_heap is not empty  
            REMOVE and SET cost x y TO the element with the smallest cost from min_heap  
            
            IF x EQUALS m MINUS one AND y EQUALS n MINUS one THEN  
                RETURN cost  
            END IF  
            
            FOR each dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the pair nx ny NOT IN visited THEN  
                    ADD the pair nx ny TO visited  
                    SET new_cost TO cost PLUS element at position nx of grid AT position ny  
                    ADD the triple new_cost nx ny TO min_heap maintaining the min heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
