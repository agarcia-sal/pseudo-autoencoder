CLASS Solution  
    FUNCTION subsets(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN a list containing an empty list  
        END IF  
        SET subsets_without_first TO the result of calling subsets on the list of elements in nums from position two TO the end  
        SET subsets_with_first TO an empty list  
        FOR each subset IN subsets_without_first  
            SET new_subset TO a list formed by prepending the element at position one of nums TO subset  
            APPEND new_subset TO subsets_with_first  
        END FOR  
        RETURN the concatenation of subsets_without_first AND subsets_with_first  
    END FUNCTION  
END CLASS
