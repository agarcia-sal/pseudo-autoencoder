CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger parameter, desiredTotal parameter)  
        IF maxChoosableInteger MULTIPLIED BY maxChoosableInteger PLUS one DIVIDED BY two LESS THAN desiredTotal  
            RETURN false  
        END IF  
        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN true  
        END IF  
        SET memo TO an empty dictionary  

        FUNCTION can_win(current_total parameter, used_numbers parameter)  
            IF the tuple of current_total and used_numbers IS IN memo  
                RETURN the value in memo for the tuple of current_total and used_numbers  
            END IF  
            FOR each number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers and two RAISED TO the power of number IS EQUAL TO zero  
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT can_win current_total PLUS number and the bitwise OR of used_numbers and two RAISED TO the power of number  
                        SET memo at the tuple of current_total and used_numbers TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            SET memo at the tuple of current_total and used_numbers TO false  
            RETURN false  
        END FUNCTION  

        RETURN can_win zero and zero  
    END FUNCTION  
END CLASS
