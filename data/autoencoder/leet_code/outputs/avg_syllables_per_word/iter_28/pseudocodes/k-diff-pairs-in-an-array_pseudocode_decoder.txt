CLASS Solution  
    FUNCTION findPairs(nums k)  
        IF k LESS THAN zero THEN  
            RETURN zero  
        END IF  
        SET num_counts TO empty mapping  
        FOR each num IN nums  
            IF num EXISTS IN num_counts THEN  
                INCREMENT the value associated with num in num_counts BY one  
            ELSE  
                SET the value associated with num in num_counts TO one  
            END IF  
        END FOR  
        SET unique_pairs TO zero  
        FOR each num IN the keys of num_counts  
            IF k EQUALS zero THEN  
                IF the value associated with num in num_counts GREATER THAN one THEN  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF num PLUS k EXISTS IN num_counts THEN  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
