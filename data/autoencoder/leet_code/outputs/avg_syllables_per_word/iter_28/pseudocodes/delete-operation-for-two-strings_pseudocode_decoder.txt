CLASS Solution  
    FUNCTION minDistance(word1 word1 AS string word2 word2 AS string) RETURNS integer  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        
        CREATE a two dimensional list dp with m PLUS one rows AND n PLUS one columns FILLED WITH zeroes  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of word1 EQUALS element at position j MINUS one of word2  
                    SET element at row i column j of dp TO element at row i MINUS one column j MINUS one of dp PLUS one  
                ELSE  
                    SET element at row i column j of dp TO the GREATER THAN OR EQUAL TO VALUE BETWEEN element at row i MINUS one column j of dp AND element at row i column j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        SET lcs_length TO element at row m column n of dp  
        
        RETURN m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  
END CLASS
