CLASS Solution  
    FUNCTION minWindow(s parameter FOR string, t parameter FOR string) RETURNS string  
        SET t_count TO the frequency count of each character in t  
        SET required TO the number of unique keys in t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO an empty frequency count  
        SET min_length TO infinity  
        SET min_start TO zero  
        WHILE right is LESS THAN the length of s  
            INCREMENT the count of the element at position right of s in window_counts BY one  
            IF the element at position right of s EXISTS IN t_count AND the count of that element in window_counts EQUALS the count of that element in t_count  
                INCREMENT formed BY one  
            END IF  
            WHILE left is LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO the element at position left of s  
                IF right MINUS left PLUS one is LESS THAN min_length  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                DECREMENT the count of char in window_counts BY one  
                IF char EXISTS IN t_count AND the count of char in window_counts is LESS THAN the count of char in t_count  
                    DECREMENT formed BY one  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            INCREMENT right BY one  
        END WHILE  
        IF min_length EQUALS infinity  
            RETURN an empty string  
        ELSE  
            RETURN the substring of s from position min_start TO position min_start PLUS min_length MINUS one  
        END IF  
    END FUNCTION  
END CLASS
