CLASS Solution  
    FUNCTION smallestDistancePair(nums k)  
        SORT nums IN ASCENDING ORDER  
        SET n TO the LENGTH OF nums  
        
        FUNCTION count(mid)  
            SET count TO zero  
            SET left TO zero  
            FOR right FROM zero TO n MINUS one  
                WHILE the ELEMENT AT POSITION right OF nums MINUS the ELEMENT AT POSITION left OF nums GREATER THAN mid  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT count BY right MINUS left  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the ELEMENT AT THE LAST POSITION OF nums MINUS the ELEMENT AT THE FIRST POSITION OF nums  
        WHILE left LESS THAN right  
            SET mid TO the INTEGER VALUE OF left PLUS right DIVIDED BY two  
            IF the RESULT OF count GIVING mid LESS THAN k  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
