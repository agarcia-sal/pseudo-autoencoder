CLASS Solution  
    FUNCTION checkRecord(n)  
        SET MOD TO one billion PLUS seven  
        SET dp TO a three dimensional list of size n PLUS one by two by three initialized to zero  
        SET element at position zero of dp at position zero and position zero TO one  
        
        FOR i FROM one TO n  
            FOR j FROM zero TO one  
                FOR k FROM zero TO two  
                    SET element at position i of dp at position j and position zero TO element at position i of dp at position j and position zero PLUS element at position i MINUS one of dp at position j and position k  
                    SET element at position i of dp at position j and position zero TO element at position i of dp at position j and position zero MODULO MOD  
                    
                    IF j EQUALS zero  
                        SET element at position i of dp at position one and position zero TO element at position i of dp at position one and position zero PLUS element at position i MINUS one of dp at position zero and position k  
                        SET element at position i of dp at position one and position zero TO element at position i of dp at position one and position zero MODULO MOD  
                    END IF  
                    
                    IF k LESS THAN two  
                        SET element at position i of dp at position j and position k PLUS one TO element at position i of dp at position j and position k PLUS one PLUS element at position i MINUS one of dp at position j and position k  
                        SET element at position i of dp at position j and position k PLUS one TO element at position i of dp at position j and position k PLUS one MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR j FROM zero TO one  
            FOR k FROM zero TO two  
                SET result TO result PLUS element at position n of dp at position j and position k  
                SET result TO result MODULO MOD  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
