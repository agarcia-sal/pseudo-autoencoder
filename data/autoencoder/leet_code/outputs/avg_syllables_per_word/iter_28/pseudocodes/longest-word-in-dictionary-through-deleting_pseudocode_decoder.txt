CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator_over_s TO the iterator over s  
            FOR each character IN word  
                IF character NOT IN the remaining elements of iterator_over_s  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SORT dictionary BY the negative of the LENGTH OF each element AND THEN BY natural alphabetical order  
        
        FOR each word IN dictionary  
            IF is_subsequence(word, s) EQUALS true  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN the empty string  
    END FUNCTION  
END CLASS
