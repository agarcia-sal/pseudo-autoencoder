CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MOD TO ten EXPONENTIATED BY nine PLUS one  
        SET end_with_0 TO zero  
        SET end_with_1 TO zero  
        SET has_zero TO the BOOLEAN VALUE OF whether the character zero EXISTS IN the string binary  
        FOR each bit IN the string binary  
            IF the character bit EQUALS the character one  
                SET end_with_1 TO remainder AFTER DIVIDING the SUM of end_with_0 PLUS end_with_1 PLUS one BY MOD  
            ELSE  
                SET end_with_0 TO remainder AFTER DIVIDING the SUM of end_with_0 PLUS end_with_1 BY MOD  
            END IF  
        END FOR  
        RETURN remainder AFTER DIVIDING the SUM of end_with_0 PLUS end_with_1 PLUS the INTEGER VALUE OF has_zero BY MOD  
    END FUNCTION  
END CLASS
