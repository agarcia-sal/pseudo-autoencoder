CLASS Solution  
    FUNCTION reorderedPowerOf2(n)  
        SET power_of_2_sorted_digits TO an empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 LESS THAN OR EQUAL TO the number one billion holds  
            CONVERT power_of_2 TO string form  
            SORT all characters that represent digits in the string form of power_of_2 in ascending order  
            CONVERT the sorted characters into an ordered collection  
            ADD this ordered collection TO power_of_2_sorted_digits  
            MULTIPLY power_of_2 BY two and update power_of_2  
        END WHILE  
        CONVERT the input number n TO string form  
        SORT all characters that represent digits in the string form of n in ascending order  
        CONVERT the sorted characters into an ordered collection assigned to sorted_n_digits  
        IF sorted_n_digits EXISTS IN power_of_2_sorted_digits THEN  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
