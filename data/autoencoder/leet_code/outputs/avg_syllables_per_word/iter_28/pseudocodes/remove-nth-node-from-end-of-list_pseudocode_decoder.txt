CLASS ListNode  
    FUNCTION __init__(self val SET TO zero next SET TO none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the collection values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode initialized with element at position one of values  
    SET p TO head  
    FOR each val IN the collection from position two TO the last element of values  
        SET node TO a new ListNode initialized with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN that element val of p1 EQUALS element val of p2 AND the call to is_same_list with parameters element next of p1 and element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd(self head n)  
        SET dummy TO a new ListNode initialized with zero  
        SET the next element of dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR index i FROM zero TO the value of n PLUS one  
            SET first TO the next element of first  
        END FOR  
        WHILE first NOT EQUALS none  
            SET first TO the next element of first  
            SET second TO the next element of second  
        END WHILE  
        SET the next element of second TO the next element of the next element of second  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
