CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF the LENGTH OF s EQUALS zero  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET difference TO the position of character c in the alphabet MINUS the position of the first character in string s in the alphabet  
                SET wrapped_difference TO difference MODULO the number of letters in the alphabet  
                APPEND the textual form of wrapped_difference TO canonical_form  
            END FOR  
            RETURN the concatenation of the elements in canonical_form separated by the word comma  
        END FUNCTION  
        
        SET groups TO a new map from keys to empty lists  
        FOR each string s IN strings  
            SET canonical_form TO the result of calling get_canonical_form with s  
            APPEND s TO the list in groups with key canonical_form  
        END FOR  
        
        RETURN the list of all values contained in groups  
    END FUNCTION  
END CLASS
