CLASS Solution  
    FUNCTION numRabbits WITH parameter answers  
        SET answer_counts TO the count of occurrences of each element in answers  
        SET total_rabbits TO zero  
        FOR each answer AND count IN the collection of answer_counts items  
            SET groups TO the integer division of the sum of count AND answer BY the sum of answer AND one  
            INCREMENT total_rabbits BY the product of groups AND the sum of answer AND one  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
