CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each char IN s  
            IF char IS a digit character  
                SET d TO d PLUS char  
            ELSE IF char IS an alphabet character  
                SET a TO a PLUS char  
            ELSE IF char EQUALS the left bracket character  
                APPEND the pair consisting of a and the integer value of d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char EQUALS the right bracket character  
                SET p and n TO the last element removed from stack  
                SET a TO p PLUS a REPEATED n times  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
