CLASS Solution  
    FUNCTION myPow(x FLOATING POINT NUMBER n INTEGER) RETURNS FLOATING POINT NUMBER  
        FUNCTION qpow(a FLOATING POINT NUMBER n INTEGER) RETURNS FLOATING POINT NUMBER  
            SET ans TO one  
            WHILE n IS GREATER THAN zero  
                IF the LEAST SIGNIFICANT BIT OF n IS EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two DISCARDING REMAINDER  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n IS GREATER THAN OR EQUAL TO zero  
            RETURN the RESULT OF qpow WITH a ASSIGNED x AND n ASSIGNED n  
        ELSE  
            RETURN one DIVIDED BY the RESULT OF qpow WITH a ASSIGNED x AND n ASSIGNED the NEGATION OF n  
        END IF  
    END FUNCTION  
END CLASS
