CLASS Solution  
    FUNCTION longestPalindromeSubseq(s)  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional array of size n by n with all elements SET TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i at position i of dp TO one  
        END FOR  
        
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF the character at position i of s EQUALS the character at position j of s THEN  
                    SET element at position i at position j of dp TO the element at position i PLUS one at position j MINUS one of dp PLUS two  
                ELSE  
                    SET element at position i at position j of dp TO the maximum value BETWEEN the element at position i PLUS one at position j of dp AND the element at position i at position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position zero at position n MINUS one of dp  
    END FUNCTION  
END CLASS
