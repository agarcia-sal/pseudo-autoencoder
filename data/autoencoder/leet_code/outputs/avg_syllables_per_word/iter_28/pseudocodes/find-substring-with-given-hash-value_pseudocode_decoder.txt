CLASS Solution  
    FUNCTION subStrHash(s power modulo k hashValue)  
        FUNCTION val(c)  
            RETURN the position value of character c in the alphabet starting from one for letter a  
        END FUNCTION  
       
        SET n TO the LENGTH OF s  
        SET current_hash TO zero  
        SET p_pow TO one  
       
        FOR i FROM n MINUS k TO n MINUS one  
            SET current_hash TO the remainder of the SUM of current_hash PLUS the product of val of element at position i of s MULTIPLIED BY p_pow divided by modulo  
            IF i LESS THAN n MINUS one  
                SET p_pow TO the remainder of the product of p_pow MULTIPLIED BY power divided by modulo  
            END IF  
        END FOR  
       
        SET start_index TO n MINUS k  
        IF current_hash EQUALS hashValue  
            RETURN the substring of s from position start_index TO position start_index PLUS k MINUS one  
        END IF  
       
        SET p_pow_k_minus_1 TO p_pow  
       
        FOR i FROM n MINUS k MINUS one DOWN TO zero  
            SET current_hash TO the remainder of the difference of current_hash MINUS the product of val of element at position i PLUS k of s MULTIPLIED BY p_pow_k_minus_1 divided by modulo  
            SET current_hash TO the remainder of the sum of the product of current_hash MULTIPLIED BY power PLUS val of element at position i of s divided by modulo  
           
            IF current_hash EQUALS hashValue  
                SET start_index TO i  
            END IF  
        END FOR  
       
        RETURN the substring of s from position start_index TO position start_index PLUS k MINUS one  
    END FUNCTION  
END CLASS
