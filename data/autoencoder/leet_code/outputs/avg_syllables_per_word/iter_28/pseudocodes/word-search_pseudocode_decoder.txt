CLASS Solution  
    FUNCTION exist(board, word)  
        IF board EQUALS empty  
            RETURN False  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        SET visited TO a list containing m elements EACH of which is a list containing n elements ALL set TO False  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS the LENGTH OF word  
                RETURN True  
            END IF  

            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of visited AT position y EQUALS True OR element at position x of board AT position y NOT EQUALS element at position index of word  
                RETURN False  
            END IF  

            SET element at position x of visited AT position y TO True  
            FOR each pair dx, dy IN the list containing pairs minus one and zero PLUS one and zero PLUS zero and minus one PLUS zero and one  
                IF dfs the sum of x PLUS dx the sum of y PLUS dy the sum of index PLUS one RETURNS True  
                    RETURN True  
                END IF  
            END FOR  
            SET element at position x of visited AT position y TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs(i, j, zero) RETURNS True  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
