CLASS Solution  
    FUNCTION longestWord(words)  
        SORT words BY their LENGTH THEN BY their NATURAL ORDER  
        SET valid_words TO an empty collection of unique elements  
        SET longest_word TO an empty sequence of characters  
        
        FOR each word IN words  
            IF the LENGTH OF word EQUALS one OR the SUBSEQUENCE of word from the FIRST CHARACTER TO the SECOND TO LAST CHARACTER IS PRESENT IN valid_words THEN  
                ADD word TO valid_words  
                IF the LENGTH OF word IS GREATER THAN the LENGTH OF longest_word THEN  
                    SET longest_word TO word  
                END IF  
            END IF  
        END FOR  
        
        RETURN longest_word  
    END FUNCTION  
END CLASS
