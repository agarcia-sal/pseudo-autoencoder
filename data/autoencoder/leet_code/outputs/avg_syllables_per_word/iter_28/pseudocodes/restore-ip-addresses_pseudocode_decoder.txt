CLASS Solution  
    FUNCTION restoreIpAddresses(s)  
        FUNCTION is_valid(segment)  
            RETURN the LENGTH OF segment EQUALS one OR the FIRST element of segment NOT EQUALS zero AND the INTEGER VALUE OF segment LESS THAN OR EQUAL TO two hundred fifty-five  
        END FUNCTION  
        
        FUNCTION backtrack(start, path)  
            IF the LENGTH OF path EQUALS four  
                IF start EQUALS the LENGTH OF s  
                    APPEND the JOINED elements of path WITH period TO result  
                END IF  
                RETURN  
            END IF  
            FOR length FROM one TO three  
                IF start PLUS length LESS THAN OR EQUAL TO the LENGTH OF s  
                    SET segment TO the SUBSTRING of s FROM position start TO position start PLUS length MINUS one  
                    IF is_valid(segment)  
                        CALL backtrack WITH start PLUS length AND path PLUS segment AS a new list  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL backtrack WITH zero AND an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
