CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO an empty dictionary  
        SET i TO zero  
        FOR each character c IN order  
            SET dictionary element for character c TO i  
            INCREMENT i BY one  
        END FOR  
        SET sorted list TO the result of sorting string s such that for each character x the sort key is the value in dictionary d for character x if character x is present in dictionary d otherwise zero  
        SET result string TO the concatenation of all elements in sorted list in order  
        RETURN result string  
    END FUNCTION  
END CLASS
