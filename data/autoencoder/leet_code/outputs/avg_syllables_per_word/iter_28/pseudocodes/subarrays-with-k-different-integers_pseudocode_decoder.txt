CLASS Solution  
    FUNCTION subarraysWithKDistinct(nums, k)  
        FUNCTION atMostKDistinct(t)  
            SET count TO an empty mapping from elements to counts  
            SET left TO zero  
            SET result TO zero  
            FOR right FROM zero TO the LENGTH OF nums MINUS one  
                INCREMENT the count of the element at position right of nums BY one  
                WHILE the NUMBER OF keys in count IS GREATER THAN t  
                    DECREMENT the count of the element at position left of nums BY one  
                    IF the count of the element at position left of nums EQUALS zero  
                        REMOVE the element at position left of nums FROM count  
                    END IF  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT result BY the value of right MINUS the value of left PLUS one  
            END FOR  
            RETURN result  
        END FUNCTION  
        RETURN the value returned by atMostKDistinct with k MINUS the value returned by atMostKDistinct with k MINUS one  
    END FUNCTION  
END CLASS
