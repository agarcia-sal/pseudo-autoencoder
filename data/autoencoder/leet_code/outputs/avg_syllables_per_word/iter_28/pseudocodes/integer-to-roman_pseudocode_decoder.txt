    FUNCTION intToRoman(num)  
        SET val TO the list of one thousand PLUS nine hundred PLUS five hundred PLUS four hundred PLUS one hundred PLUS ninety PLUS fifty PLUS forty PLUS ten PLUS nine PLUS five PLUS four PLUS one in order  
        SET syms TO the list of the Roman numeral symbol for one thousand PLUS the Roman numeral symbol for nine hundred PLUS the Roman numeral symbol for five hundred PLUS the Roman numeral symbol for four hundred PLUS the Roman numeral symbol for one hundred PLUS the Roman numeral symbol for ninety PLUS the Roman numeral symbol for fifty PLUS the Roman numeral symbol for forty PLUS the Roman numeral symbol for ten PLUS the Roman numeral symbol for nine PLUS the Roman numeral symbol for five PLUS the Roman numeral symbol for four PLUS the Roman numeral symbol for one in order  
        SET roman_numeral TO an empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero  
            FOR each iteration IN the whole number division of num BY the element at position i of val  
                SET roman_numeral TO the concatenation of roman_numeral AND the element at position i of syms  
                DECREMENT num BY the element at position i of val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
