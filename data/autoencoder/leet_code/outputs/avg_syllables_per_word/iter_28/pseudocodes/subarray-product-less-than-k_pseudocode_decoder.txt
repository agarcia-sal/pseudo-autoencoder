CLASS Solution  
    FUNCTION numSubarrayProductLessThanK(nums parameter of list of integers k parameter of integer) RETURNS integer  
        IF k LESS THAN OR EQUAL TO one THEN  
            RETURN zero  
        END IF  
        
        SET product TO one  
        SET left TO zero  
        SET count TO zero  
        
        FOR each right FROM zero TO the LENGTH OF nums MINUS one  
            MULTIPLY product BY element at position right of nums  
            
            WHILE product GREATER THAN OR EQUAL TO k AND left LESS THAN OR EQUAL TO right  
                DIVIDE product BY element at position left of nums  
                INCREMENT left BY one  
            END WHILE  
            
            INCREMENT count BY right MINUS left PLUS one  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
