CLASS Solution  
    FUNCTION numberOfBoomerangs(points)  
        SET ans TO zero  
        FOR each p1 IN points  
            CREATE a new counter named cnt  
            FOR each p2 IN points  
                SET dx TO the element at position zero of p1 MINUS the element at position zero of p2  
                SET dy TO the element at position one of p1 MINUS the element at position one of p2  
                SET d TO dx MULTIPLIED BY dx PLUS dy MULTIPLIED BY dy  
                INCREMENT ans BY the value associated with d in cnt  
                INCREMENT the value associated with d in cnt BY one  
            END FOR  
        END FOR  
        RETURN ans MULTIPLIED BY two  
    END FUNCTION  
END CLASS
