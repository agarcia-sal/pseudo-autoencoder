CLASS Solution  
    FUNCTION updateMatrix WITH PARAMETER mat  
        IF mat EQUALS empty OR element at position zero of mat EQUALS empty  
            RETURN mat  
        END IF  
        SET rows TO the LENGTH OF mat  
        SET cols TO the LENGTH OF element at position zero of mat  
        SET dist TO a new two dimensional list with rows number of elements each containing cols number of elements all SET TO infinity  
        CREATE an empty queue named queue  
        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF element at position r of mat AND element at position c EQUALS zero  
                    SET element at position r of dist AND element at position c TO zero  
                    APPEND the tuple of r and c TO queue  
                END IF  
            END FOR  
        END FOR  
        SET directions TO the list containing the tuples of negative one and zero ONE and zero zero and negative one zero and ONE  
        WHILE queue IS NOT empty  
            EXTRACT the front element of queue INTO r and c  
            FOR each tuple dr and dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN rows AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN cols AND element at position nr of dist AND element at position nc GREATER THAN element at position r of dist AND element at position c PLUS one  
                    SET element at position nr of dist AND element at position nc TO element at position r of dist AND element at position c PLUS one  
                    APPEND the tuple of nr and nc TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN dist  
    END FUNCTION  
END CLASS
