CLASS TreeNode  
    FUNCTION __init__(val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element popped from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p.left AND q.left AND the result of is_same_tree with p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rob(root)  
        FUNCTION rob_from_node(node)  
            IF node EQUALS none  
                RETURN zero, zero  
            END IF  
            SET left_robbed, left_not_robbed TO the two values returned by rob_from_node called with node.left  
            SET right_robbed, right_not_robbed TO the two values returned by rob_from_node called with node.right  
            SET rob_this_node TO node.val PLUS left_not_robbed PLUS right_not_robbed  
            SET not_rob_this_node TO the sum of the maximum of left_robbed AND left_not_robbed PLUS the maximum of right_robbed AND right_not_robbed  
            RETURN rob_this_node, not_rob_this_node  
        END FUNCTION  
        RETURN the maximum of the two values returned by rob_from_node called with root  
    END FUNCTION  
END CLASS
