CLASS Solution  
    FUNCTION fourSumCount(nums1 parameter, nums2 parameter, nums3 parameter, nums4 parameter)  
        SET sum_ab TO an empty dictionary  
        FOR each element a IN nums1  
            FOR each element b IN nums2  
                SET current_sum TO a PLUS b  
                IF current_sum EQUALS a key in sum_ab  
                    INCREMENT the value at key current_sum in sum_ab BY one  
                ELSE  
                    SET the value at key current_sum in sum_ab TO one  
                END IF  
            END FOR  
        END FOR  
        SET count TO zero  
        FOR each element c IN nums3  
            FOR each element d IN nums4  
                SET target TO the negative of c PLUS d  
                IF target EQUALS a key in sum_ab  
                    INCREMENT count BY the value at key target in sum_ab  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
