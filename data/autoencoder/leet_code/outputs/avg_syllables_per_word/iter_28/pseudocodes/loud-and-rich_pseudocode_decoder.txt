CLASS Solution  
    FUNCTION loudAndRich(richer, quiet)  
        SET graph TO a default dictionary mapping to empty lists  
        FOR each pair of elements a AND b IN richer  
            APPEND element a TO the list at element b in graph  
        END FOR  
        
        SET n TO the LENGTH OF quiet  
        SET answer TO a list of length n WHERE each element is negative one  
        
        FUNCTION dfs(person)  
            IF element at position person of answer EQUALS negative one THEN  
                SET element at position person of answer TO person  
                FOR each neighbor IN the list at element person of graph  
                    SET candidate TO the RESULT OF dfs(neighbor)  
                    IF element at position candidate of quiet LESS THAN element at position element at position person of answer of quiet THEN  
                        SET element at position person of answer TO candidate  
                    END IF  
                END FOR  
            END IF  
            RETURN element at position person of answer  
        END FUNCTION  
        
        FOR index i FROM zero TO n MINUS one  
            CALL dfs WITH argument i  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
