CLASS Solution  
    FUNCTION countBattleships(board)  
        SET count TO zero  
        FOR i FROM zero TO the LENGTH OF board MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position i of board MINUS one  
                IF the element at position i of board at position j EQUALS the character representing a battleship THEN  
                    IF i EQUALS zero OR the element at position i MINUS one of board at position j EQUALS the empty character AND j EQUALS zero OR the element at position i of board at position j MINUS one EQUALS the empty character THEN  
                        INCREMENT count BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
