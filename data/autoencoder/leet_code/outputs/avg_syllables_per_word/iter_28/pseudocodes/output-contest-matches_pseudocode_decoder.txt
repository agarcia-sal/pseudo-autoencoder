CLASS Solution  
    FUNCTION findContestMatch WITH PARAMETER n  
        SET teams TO empty list  
        FOR each number FROM one TO n  
            APPEND the string representation of number TO teams  
        END FOR  
        WHILE n GREATER THAN one  
            FOR each index FROM zero TO the integer division of n BY two MINUS one  
                SET element at position index of teams TO the string concatenation of a left parenthesis PLUS element at position index of teams PLUS a comma PLUS element at position n MINUS one MINUS index of teams PLUS a right parenthesis  
            END FOR  
            SET n TO the integer division of n BY two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
