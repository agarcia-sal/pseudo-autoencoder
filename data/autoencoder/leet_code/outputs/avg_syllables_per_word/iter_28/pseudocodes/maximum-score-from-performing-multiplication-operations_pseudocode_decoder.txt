CLASS Solution  
    FUNCTION maximumScore(nums multipliers)  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zero repeated m PLUS one times  
        FOR i FROM m MINUS one TO zero DECREASING BY one  
            SET new_dp TO a list of zero repeated m PLUS one times  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS i MINUS left MINUS left ADDRESSED AS n MINUS one MINUS i MINUS left  
                SET mul TO the element at position i of multipliers  
                IF mul MULTIPLIED BY the element at position left of nums PLUS the element at position left PLUS one of dp IS GREATER THAN OR EQUAL TO mul MULTIPLIED BY the element at position right of nums PLUS the element at position left of dp  
                    SET the element at position left of new_dp TO mul MULTIPLIED BY the element at position left of nums PLUS the element at position left PLUS one of dp  
                ELSE  
                    SET the element at position left of new_dp TO mul MULTIPLIED BY the element at position right of nums PLUS the element at position left of dp  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN the element at position zero of dp  
    END FUNCTION  
END CLASS
