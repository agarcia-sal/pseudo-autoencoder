CLASS Solution  
    FUNCTION isScramble WITH PARAMETERS s1 s2  
        IF s1 EQUALS s2  
            RETURN True  
        END IF  
        IF the characters of s1 sorted ARE NOT EQUALS the characters of s2 sorted  
            RETURN False  
        END IF  
        SET n TO the LENGTH OF s1  
        FOR i FROM one TO n MINUS one  
            IF the result of calling isScramble WITH the substring of s1 from position one TO position i AND the substring of s2 from position one TO position i IS True AND the result of calling isScramble WITH the substring of s1 from position i PLUS one TO position n AND the substring of s2 from position i PLUS one TO position n IS True  
                RETURN True  
            END IF  
            IF the result of calling isScramble WITH the substring of s1 from position one TO position i AND the substring of s2 from position n MINUS i PLUS one TO position n IS True AND the result of calling isScramble WITH the substring of s1 from position i PLUS one TO position n AND the substring of s2 from position one TO position n MINUS i IS True  
                RETURN True  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
