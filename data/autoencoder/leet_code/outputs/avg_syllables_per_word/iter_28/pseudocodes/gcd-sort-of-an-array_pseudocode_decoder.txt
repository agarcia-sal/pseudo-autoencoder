CLASS UnionFind  
    FUNCTION __init__ with parameter size  
        SET self.parent TO list of integers from zero TO size MINUS one  
        SET self.rank TO list of ones with length size  
    END FUNCTION  

    FUNCTION find with parameter p  
        IF element at position p of self.parent NOT EQUALS p THEN  
            SET element at position p of self.parent TO the result of calling find with the element at position p of self.parent  
        END IF  
        RETURN element at position p of self.parent  
    END FUNCTION  

    FUNCTION union with parameters p q  
        SET rootP TO the result of calling find with p  
        SET rootQ TO the result of calling find with q  
        IF rootP EQUALS rootQ THEN  
            RETURN False  
        END IF  
        IF element at position rootP of self.rank GREATER THAN element at position rootQ of self.rank THEN  
            SET element at position rootQ of self.parent TO rootP  
        ELSE IF element at position rootP of self.rank LESS THAN element at position rootQ of self.rank THEN  
            SET element at position rootP of self.parent TO rootQ  
        ELSE  
            SET element at position rootQ of self.parent TO rootP  
            INCREMENT element at position rootP of self.rank BY one  
        END IF  
        RETURN True  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION gcdSort with parameter nums  
        SET sorted_nums TO nums sorted in ascending order  
        SET max_val TO the maximum value in nums  
        CREATE an instance uf of UnionFind with parameter max_val PLUS one  
        
        FOR each num IN nums  
            FOR each factor FROM two TO the integer part of the square root of num PLUS one  
                IF the remainder when num is divided by factor EQUALS zero THEN  
                    CALL uf.union with parameters num factor  
                    CALL uf.union with parameters num num DIVIDED BY factor  
                END IF  
            END FOR  
        END FOR  

        FOR each index i FROM zero TO the length of nums MINUS one  
            IF the result of calling uf.find with element at position i of nums NOT EQUALS the result of calling uf.find with element at position i of sorted_nums THEN  
                RETURN False  
            END IF  
        END FOR  

        RETURN True  
    END FUNCTION  
END CLASS
