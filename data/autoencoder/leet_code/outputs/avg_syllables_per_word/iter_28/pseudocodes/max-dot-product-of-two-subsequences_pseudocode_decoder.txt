CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        SET dp TO a new two dimensional list with m PLUS one rows and n PLUS one columns  
        FOR index_i FROM zero TO m  
            FOR index_j FROM zero TO n  
                SET element at position index_j of element at position index_i of dp TO negative infinity  
            END FOR  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO element at position i MINUS one of nums1 MULTIPLIED BY element at position j MINUS one of nums2  
                SET element at position j of element at position i of dp TO the maximum value among the following four values:  
                    current_product  
                    element at position j of element at position i MINUS one of dp  
                    element at position j MINUS one of element at position i of dp  
                    element at position j MINUS one of element at position i MINUS one of dp PLUS current_product  
            END FOR  
        END FOR  
        RETURN element at position n of element at position m of dp  
    END FUNCTION  
END CLASS
