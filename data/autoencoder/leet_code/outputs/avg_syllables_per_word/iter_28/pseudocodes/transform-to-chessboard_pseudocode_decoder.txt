CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR index i FROM zero TO N MINUS one  
            FOR index j FROM zero TO N MINUS one  
                IF the element at position i of board AND the element at position j of that element NOT EQUALS the element at position i of board AND zero PLUS the element at position zero of board AND j PLUS the element at position zero of board AND zero  
                    RETURN negative one  
                END IF  
            END FOR  
            SET rowSum TO rowSum PLUS the element at position i of board AND zero  
            SET colSum TO colSum PLUS the element at position zero of board AND i  
            IF the element at position i of board AND zero EQUALS the remainder of the division of i BY two  
                INCREMENT rowSwap BY one  
            END IF  
            IF the element at position zero of board AND i EQUALS the remainder of the division of i BY two  
                INCREMENT colSwap BY one  
            END IF  
        END FOR  
        IF the integer division of N BY two GREATER THAN rowSum OR rowSum GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the integer division of N BY two GREATER THAN colSum OR colSum GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the remainder of the division of N BY two EQUALS one  
            IF the remainder of the division of rowSwap BY two EQUALS one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of the division of colSwap BY two EQUALS one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            IF rowSwap LESS THAN N MINUS rowSwap  
                SET rowSwap TO rowSwap  
            ELSE  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF colSwap LESS THAN N MINUS colSwap  
                SET colSwap TO colSwap  
            ELSE  
                SET colSwap TO N MINUS colSwap  
            END IF  
        END IF  
        SET result TO rowSwap PLUS colSwap  
        SET result TO the integer division of result BY two  
        RETURN result  
    END FUNCTION  
END CLASS
