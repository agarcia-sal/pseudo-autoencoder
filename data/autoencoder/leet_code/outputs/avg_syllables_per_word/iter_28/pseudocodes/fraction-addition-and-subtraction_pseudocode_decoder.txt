CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET modified_expression TO expression WHERE every occurrence of the character indicating a minus sign IS REPLACED BY the character indicating a plus sign FOLLOWED BY a minus sign  
        SET fractions_list TO the elements obtained by SPLITTING modified_expression BY the character indicating a plus sign  
        SET filtered_fractions TO an empty list  
        FOR each element IN fractions_list  
            IF element NOT EQUALS an empty string  
                APPEND element TO filtered_fractions  
            END IF  
        END FOR  
        SET result TO the sum of the fractions obtained by CONSTRUCTING fraction objects for each element IN filtered_fractions  
        RETURN the numerator of result AS a string concatenated WITH a character indicating a forward slash concatenated WITH the denominator of result AS a string  
    END FUNCTION  
END CLASS
