CLASS Solution  
    FUNCTION profitableSchemes(n number_of_members, minProfit minimum_profit, group list_of_group_sizes, profit list_of_profits) RETURNS integer  
        SET MODULUS TO one billion PLUS seven  
        SET dp TO a list of lists each containing zero with length minimum_profit PLUS one repeated for each member count from zero TO number_of_members inclusive  
        FOR each i IN the range from zero TO number_of_members inclusive  
            SET element at position zero of dp at position i TO one  
        END FOR  
        
        FOR each profit_value AND group_size IN the paired elements of list_of_profits AND list_of_group_sizes  
            FOR each i IN the range from number_of_members DOWN TO group_size  
                FOR each j IN the range from minimum_profit DOWN TO zero  
                    SET dp at position i at position j TO the sum of dp at position i at position j AND dp at position i MINUS group_size at position maximum of zero AND j MINUS profit_value  
                    SET dp at position i at position j TO dp at position i at position j MODULO MODULUS  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp at position number_of_members at position minimum_profit  
    END FUNCTION  
END CLASS
