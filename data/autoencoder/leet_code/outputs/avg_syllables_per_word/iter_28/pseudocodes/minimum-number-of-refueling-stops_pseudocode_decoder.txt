CLASS Solution  
    FUNCTION minRefuelStops(target startFuel stations)  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND list containing target and zero TO stations  

        FOR each position_and_fuel IN stations  
            SET position TO element at position zero of position_and_fuel  
            SET fuel TO element at position one of position_and_fuel  

            WHILE max_heap is not empty AND current_fuel LESS THAN position  
                REMOVE the largest fuel amount from max_heap by extracting the smallest element and negate it  
                INCREMENT current_fuel BY the extracted fuel amount  
                INCREMENT stops BY one  
            END WHILE  

            IF current_fuel LESS THAN position  
                RETURN negative one  
            END IF  

            ADD the negative of fuel TO max_heap to maintain max heap property  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
