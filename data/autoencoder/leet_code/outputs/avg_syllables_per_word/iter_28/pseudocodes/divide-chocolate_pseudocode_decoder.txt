CLASS Solution  
    FUNCTION maximizeSweetness WITH PARAMETERS sweetness AND k  
        FUNCTION canDivide WITH PARAMETER min_sweet  
            SET pieces TO zero  
            SET current_sweet TO zero  
            FOR each sweet IN sweetness  
                INCREMENT current_sweet BY sweet  
                IF current_sweet GREATER THAN OR EQUAL TO min_sweet  
                    INCREMENT pieces BY one  
                    SET current_sweet TO zero  
                    IF pieces GREATER THAN k  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            RETURN pieces GREATER THAN k  
        END FUNCTION  
        
        SET left TO the minimum value in sweetness  
        SET right TO the sum of all elements in sweetness DIVIDED BY k PLUS one  
        SET result TO zero  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the integer division result of left PLUS right DIVIDED BY two  
            IF canDivide WITH mid AS ARGUMENT IS True  
                SET result TO mid  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
