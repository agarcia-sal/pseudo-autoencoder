CLASS Solution  
    FUNCTION maxRunTime(n, batteries)  
        FUNCTION canRunFor(time)  
            SET total TO zero  
            FOR each battery IN batteries  
                SET total TO total PLUS the smaller value BETWEEN battery AND time  
            END FOR  
            RETURN total GREATER THAN OR EQUAL TO time MULTIPLIED BY n  
        END FUNCTION  

        SET left TO zero  
        SET right TO the sum of elements in batteries DIVIDED BY n  
        WHILE left LESS THAN right  
            SET mid TO the integer division result of left PLUS right PLUS one DIVIDED BY two  
            IF canRunFor(mid)  
                SET left TO mid  
            ELSE  
                SET right TO right MINUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
