CLASS Solution  
    FUNCTION numWays(words parameter AS List of strings, target parameter AS string) RETURNS integer  
        SET MOD TO the integer one billion PLUS seven  
        SET n TO the LENGTH OF the element at position zero of words  
        SET m TO the LENGTH OF target  
        
        SET freq TO a list of size n where each element is an empty dictionary with default integer zero  
        
        FOR each word IN words  
            FOR each position j AND character char IN word at respective positions  
                INCREMENT the value in freq at position j for key char BY one  
            END FOR  
        END FOR  
        
        SET dp TO a list of size m PLUS one where each element is a list of size n PLUS one filled with zero integers  
        
        FOR each position j FROM zero TO n  
            SET the element at position zero at dp row TO one  
        END FOR  
        
        FOR each position i FROM one TO m  
            FOR each position j FROM one TO n  
                SET dp at position i and j TO the value at dp at position i and j MINUS one  
                IF the value in freq at position j MINUS one for the key at position i MINUS one of target is GREATER THAN zero  
                    INCREMENT dp at position i and j BY dp at position i MINUS one and j MINUS one MULTIPLIED BY freq at position j MINUS one for key at position i MINUS one of target  
                    SET dp at position i and j TO dp at position i and j MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position m and n of dp  
    END FUNCTION  
END CLASS
