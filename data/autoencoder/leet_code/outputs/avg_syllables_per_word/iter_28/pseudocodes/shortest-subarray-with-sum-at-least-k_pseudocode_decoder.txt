CLASS Solution  
    FUNCTION shortestSubarray(nums WITH TYPE List OF integer, k WITH TYPE integer) RETURNS integer  
        SET prefix_sum TO a list containing single element zero  
        FOR each num IN nums  
            APPEND element at the last position of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        
        ASSIGN reference TO a new empty double ended queue dq  
        SET min_length TO positive infinity  
        
        FOR each index i WITH associated element current_sum IN prefix_sum starting at zero  
            
            WHILE dq is not empty AND current_sum MINUS element at position at the first element of dq in prefix_sum GREATER THAN OR EQUAL TO k  
                SET min_length TO the lesser of min_length AND i MINUS the element removed from the front of dq  
            END WHILE  
            
            WHILE dq is not empty AND current_sum LESS THAN OR EQUAL TO element at position at the last element of dq in prefix_sum  
                REMOVE element from the end of dq  
            END WHILE  
            
            APPEND i TO dq  
        END FOR  
        
        IF min_length NOT EQUALS positive infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
