CLASS TreeNode  
    FUNCTION __init__(self WITH val WITH left WITH right)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN no value  
    END IF  
    SET root TO A new TreeNode CREATED WITH element at position one of values  
    SET i TO one  
    SET queue TO AN EMPTY double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO THE LEFTMOST element REMOVED FROM queue  
        IF i LESS THAN THE LENGTH OF values AND element at position i of values IS NOT no value THEN  
            SET node dot left TO A new TreeNode CREATED WITH element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN THE LENGTH OF values AND element at position i of values IS NOT no value THEN  
            SET node dot right TO A new TreeNode CREATED WITH element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p WITH q)  
    IF p IS no value AND q IS no value THEN  
        RETURN True  
    ELSE IF p IS no value OR q IS no value THEN  
        RETURN False  
    ELSE IF p dot val NOT EQUALS q dot val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p dot left WITH q dot left) AND is_same_tree(p dot right WITH q dot right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree(self WITH root)  
        IF root IS no value THEN  
            RETURN no value  
        END IF  
        SET root dot left TO self dot pruneTree(root dot left)  
        SET root dot right TO self dot pruneTree(root dot right)  
        IF root dot val EQUALS zero AND root dot left IS no value AND root dot right IS no value THEN  
            RETURN no value  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
