CLASS Solution  
    FUNCTION findMaxValueOfEquation(points LIST OF LIST OF INTEGER k INTEGER) RETURNS INTEGER  
        SET max_value TO negative infinity  
        ASSIGN reference TO an empty double ended queue named deque_  
        
        FOR each pair xj yj IN points  
            WHILE deque_ is not empty AND the difference between xj and the second element at the front of deque_ is GREATER THAN k  
                REMOVE the element at the front from deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO the maximum of max_value AND the sum of yj PLUS xj PLUS the first element at the front of deque_  
            END IF  
            
            WHILE deque_ is not empty AND the difference between yj and xj is GREATER THAN OR EQUAL TO the first element at the back of deque_  
                REMOVE the element at the back from deque_  
            END WHILE  
            
            APPEND the pair consisting of the difference between yj and xj AND xj TO the back of deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
