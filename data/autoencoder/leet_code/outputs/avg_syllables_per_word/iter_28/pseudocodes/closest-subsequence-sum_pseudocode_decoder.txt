CLASS Solution  
    FUNCTION minAbsDifference(nums goal)  
        SET n TO the LENGTH OF nums  
        SET mid TO n DIVIDED BY two  
        
        SET left_sums TO a list containing the single element zero  
        FOR each num IN the SUBLIST of nums from position zero TO position mid MINUS one  
            CREATE a new list composed of each element s in left_sums PLUS num  
            APPEND all elements of this new list TO left_sums  
        END FOR  
        
        SET right_sums TO a list containing the single element zero  
        FOR each num IN the SUBLIST of nums from position mid TO the END of nums  
            CREATE a new list composed of each element s in right_sums PLUS num  
            APPEND all elements of this new list TO right_sums  
        END FOR  
        
        SORT right_sums IN ascending order  
        
        SET min_diff TO positive infinity  
        
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO the result of BINARY SEARCH to find insertion position of target in right_sums  
            
            IF pos LESS THAN the LENGTH OF right_sums  
                SET candidate_diff TO the ABSOLUTE VALUE OF left_sum PLUS the element at position pos of right_sums MINUS goal  
                IF candidate_diff LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
            
            IF pos GREATER THAN zero  
                SET candidate_diff TO the ABSOLUTE VALUE OF left_sum PLUS the element at position pos MINUS one of right_sums MINUS goal  
                IF candidate_diff LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
