CLASS ListNode  
    FUNCTION __init__(self comma val EQUALS zero comma next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode with value of element at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO end of values  
        SET node TO new ListNode with value val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 comma p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next comma p2.next)  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self comma val EQUALS zero comma left EQUALS None comma right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value of element at position one of values  
    SET i TO one  
    SET queue TO new deque object  
    CALL append on queue passing root  
    WHILE LENGTH OF queue GREATER THAN zero  
        SET node TO result of pop from left of queue  
        IF i LESS THAN LENGTH OF values AND element at position i PLUS one of values NOT EQUALS None THEN  
            SET node.left TO new TreeNode with value of element at position i PLUS one of values  
            CALL append on queue passing node.left  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i PLUS one of values NOT EQUALS None THEN  
            SET node.right TO new TreeNode with value of element at position i PLUS one of values  
            CALL append on queue passing node.right  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p comma q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left comma q.left) AND is_same_tree(p.right comma q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST(self comma head)  
        IF head EQUALS None THEN  
            RETURN None  
        END IF  
        
        FUNCTION findMiddle(left comma right)  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUALS right AND fast.next NOT EQUALS right  
                SET slow TO slow.next  
                SET fast TO fast.next.next  
            END WHILE  
            RETURN slow  
        END FUNCTION  
        
        FUNCTION convertListToBST(left comma right)  
            IF left EQUALS right THEN  
                RETURN None  
            END IF  
            
            SET mid TO findMiddle(left comma right)  
            SET node TO new TreeNode with value mid.val  
            SET node.left TO convertListToBST(left comma mid)  
            SET node.right TO convertListToBST(mid.next comma right)  
            RETURN node  
        END FUNCTION  
        
        RETURN convertListToBST(head comma None)  
    END FUNCTION  
END CLASS
