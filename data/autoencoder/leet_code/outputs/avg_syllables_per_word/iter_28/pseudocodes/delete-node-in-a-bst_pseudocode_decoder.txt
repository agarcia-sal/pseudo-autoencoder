CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty list THEN  
        RETURN nothing  
    END IF  
    SET root TO new instance of TreeNode with parameter val set to element at position one of values and left set to nothing and right set to nothing  
    SET i TO one  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO remove and return the first element of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is not nothing THEN  
            SET node.left TO new instance of TreeNode with parameter val set to element at position i of values and left set to nothing and right set to nothing  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values is not nothing THEN  
            SET node.right TO new instance of TreeNode with parameter val set to element at position i of values and left set to nothing and right set to nothing  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS nothing AND q EQUALS nothing THEN  
        RETURN True  
    ELSE IF p EQUALS nothing OR q EQUALS nothing THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode(self, root, key)  
        IF root EQUALS nothing THEN  
            RETURN nothing  
        END IF  
        IF root.val GREATER THAN key THEN  
            SET root.left TO the result of calling self.deleteNode with parameters root.left and key  
            RETURN root  
        END IF  
        IF root.val LESS THAN key THEN  
            SET root.right TO the result of calling self.deleteNode with parameters root.right and key  
            RETURN root  
        END IF  
        IF root.left EQUALS nothing THEN  
            RETURN root.right  
        END IF  
        IF root.right EQUALS nothing THEN  
            RETURN root.left  
        END IF  
        SET node TO root.right  
        WHILE node.left is not nothing DO  
            SET node TO node.left  
        END WHILE  
        SET node.left TO root.left  
        SET root TO root.right  
        RETURN root  
    END FUNCTION  
END CLASS
