CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET list_of_fractions TO the result of replacing every occurrence of the minus sign in expression WITH the plus sign FOLLOWED BY the minus sign THEN splitting the resulting string by the plus sign  
        SET sum_of_fractions TO zero fraction  
        FOR each fraction_string IN list_of_fractions  
            IF fraction_string is not empty  
                INCREMENT sum_of_fractions BY converting fraction_string TO fraction  
            END IF  
        END FOR  
        RETURN string composed of numerator of sum_of_fractions FOLLOWED BY the forward slash character FOLLOWED BY denominator of sum_of_fractions  
    END FUNCTION  
END CLASS
