CLASS Solution  
    FUNCTION orderlyQueue(s parameter, k parameter)  
        IF k parameter GREATER THAN one  
            RETURN concatenation of elements in the string s parameter arranged in order from smallest to largest according to lexicographical sequence  
        ELSE  
            SET n TO the LENGTH OF string s parameter  
            SET smallest TO string s parameter  
            FOR index FROM one TO n MINUS one  
                SET rotated TO concatenation of substring from position index TO the END of string s parameter AND substring from position zero TO position index MINUS one of string s parameter  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
