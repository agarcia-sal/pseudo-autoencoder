CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode with value as element at position zero of values  
    SET i TO one  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO remove and retrieve element from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode with value as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode with value as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree of p.left AND q.left AND is_same_tree of p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root IS None  
            RETURN  
        END IF  
        SET stack TO list containing root  
        WHILE stack IS NOT EMPTY  
            SET node TO remove and retrieve the last element of stack  
            IF node.right IS NOT None  
                APPEND node.right TO stack  
            END IF  
            IF node.left IS NOT None  
                APPEND node.left TO stack  
            END IF  
            SET node.left TO None  
            IF stack IS NOT EMPTY  
                SET node.right TO element at the last position of stack  
            ELSE  
                SET node.right TO None  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
