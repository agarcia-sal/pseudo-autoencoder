CLASS Solution  
    FUNCTION closestToTarget(arr parameter_arr OF TYPE list of integer, target parameter_target OF TYPE integer) RETURNS integer  
        SET seen TO empty collection of unique elements  
        SET min_diff TO positive infinity  
        FOR each num IN parameter_arr  
            SET temporary_collection TO empty collection of unique elements  
            FOR each x IN seen  
                APPEND bitwise AND of num AND x TO temporary_collection  
            END FOR  
            SET seen TO UNION of temporary_collection AND collection containing num  
            FOR each val IN seen  
                IF absolute value of val MINUS parameter_target IS LESS THAN min_diff  
                    SET min_diff TO absolute value of val MINUS parameter_target  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
