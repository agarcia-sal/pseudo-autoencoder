CLASS Solution  
    FUNCTION kIncreasing(arr parameter LIST OF integer, k parameter integer) RETURNS integer  
        FUNCTION longest_increasing_subsequence(subarr parameter LIST OF integer) RETURNS integer  
            SET lis TO empty list  
            FOR each num IN subarr  
                IF lis EQUALS empty list OR num GREATER THAN OR EQUAL TO element at position LENGTH OF lis MINUS one of lis  
                    APPEND num TO lis  
                ELSE  
                    SET pos TO the position where num can be inserted to the right in lis to maintain sorted order  
                    SET element at position pos of lis TO num  
                END IF  
            END FOR  
            RETURN the LENGTH OF lis  
        END FUNCTION  
        
        SET total_operations TO zero  
        SET n TO the LENGTH OF arr  
        
        FOR start FROM zero TO k MINUS one  
            SET subarr TO empty list  
            FOR i FROM start TO n MINUS one STEP k  
                APPEND element at position i of arr TO subarr  
            END FOR  
            
            SET lis_length TO result of calling longest_increasing_subsequence with subarr  
            INCREMENT total_operations BY the LENGTH OF subarr MINUS lis_length  
        END FOR  
        
        RETURN total_operations  
    END FUNCTION  
END CLASS
