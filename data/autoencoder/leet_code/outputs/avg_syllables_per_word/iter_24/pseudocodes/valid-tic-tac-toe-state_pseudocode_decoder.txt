CLASS Solution  
    FUNCTION validTicTacToe(board)  
        SET countX TO zero  
        FOR each row IN board  
            INCREMENT countX BY the number of elements in row EQUALS TO character X  
        END FOR  
        SET countO TO zero  
        FOR each row IN board  
            INCREMENT countO BY the number of elements in row EQUALS TO character O  
        END FOR  
        
        IF countX LESS THAN countO OR countX GREATER THAN countO PLUS one  
            RETURN False  
        END IF  
        
        FUNCTION check_win(player)  
            FOR each row IN board  
                IF every element in row EQUALS TO player  
                    RETURN True  
                END IF  
            END FOR  
            FOR col FROM zero TO two  
                IF element at position col of row zero EQUALS TO player AND element at position col of row one EQUALS TO player AND element at position col of row two EQUALS TO player  
                    RETURN True  
                END IF  
            END FOR  
            IF element at position zero of row zero EQUALS TO player AND element at position one of row one EQUALS TO player AND element at position two of row two EQUALS TO player  
                RETURN True  
            END IF  
            IF element at position two of row zero EQUALS TO player AND element at position one of row one EQUALS TO player AND element at position zero of row two EQUALS TO player  
                RETURN True  
            END IF  
            RETURN False  
        END FUNCTION  
        
        SET winX TO the result of check_win with argument character X  
        SET winO TO the result of check_win with argument character O  
        
        IF winX AND winO  
            RETURN False  
        END IF  
        IF winX AND countX EQUALS TO countO  
            RETURN False  
        END IF  
        IF winO AND countX NOT EQUALS TO countO  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
