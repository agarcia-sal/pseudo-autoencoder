CLASS Solution  
    FUNCTION characterReplacement(s parameter STRING k parameter INTEGER) RETURNS INTEGER  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        SET count TO a mapping with default integer zero for any key  
        FOR right FROM zero TO the length of s MINUS one  
            INCREMENT the element at position right of s in count BY one  
            IF max_count LESS THAN the element at position right of s in count THEN  
                SET max_count TO the element at position right of s in count  
            END IF  
            IF the value of right MINUS left PLUS one MINUS max_count GREATER THAN k THEN  
                DECREMENT the element at position left of s in count BY one  
                INCREMENT left BY one  
            END IF  
            IF max_length LESS THAN the value of right MINUS left PLUS one THEN  
                SET max_length TO the value of right MINUS left PLUS one  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
