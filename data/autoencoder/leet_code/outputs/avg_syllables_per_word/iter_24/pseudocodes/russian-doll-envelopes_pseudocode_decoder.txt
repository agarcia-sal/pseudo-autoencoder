CLASS Solution  
    FUNCTION maxEnvelopes(envelopes)  
        SORT envelopes BY element at position zero of envelope IN ASCENDING ORDER AND BY NEGATIVE of element at position one of envelope IN ASCENDING ORDER  
        SET dp TO empty list  
        FOR each element IGNORE width AS h IN envelopes  
            SET idx TO position found by binary search on dp WHERE h should be inserted to MAINTAIN SORT ORDER  
            IF idx EQUALS the LENGTH OF dp THEN  
                APPEND h TO dp  
            ELSE  
                SET element at position idx of dp TO h  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
