CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO sorted list of the first elements of each pair in flowers  
        SET end_times TO sorted list of the second elements of each pair in flowers  
        SET result TO empty list  
        FOR each person IN people  
            SET started TO number of elements in start_times which are less than OR EQUAL TO person  
            SET ended TO number of elements in end_times which are strictly less than person  
            APPEND started MINUS ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
