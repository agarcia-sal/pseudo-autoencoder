CLASS Solution  
    FUNCTION compareVersion(version1 version1 IS OF TYPE string version1, version2 version2 IS OF TYPE string version2) RETURNS integer  
        SET list_of_parts_one TO the division of version1 by the character dot  
        SET list_of_parts_two TO the division of version2 by the character dot  
        SET maximum_length TO the greater of the length of list_of_parts_one AND the length of list_of_parts_two  
        FOR index FROM zero TO maximum_length MINUS one  
            IF index LESS THAN the length of list_of_parts_one  
                SET revision_one TO the conversion of the element at position index of list_of_parts_one TO integer  
            ELSE  
                SET revision_one TO zero  
            END IF  
            IF index LESS THAN the length of list_of_parts_two  
                SET revision_two TO the conversion of the element at position index of list_of_parts_two TO integer  
            ELSE  
                SET revision_two TO zero  
            END IF  
            IF revision_one LESS THAN revision_two  
                RETURN negative one  
            ELSE IF revision_one GREATER THAN revision_two  
                RETURN one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
