CLASS Solution  
    FUNCTION maxBuilding(n, restrictions)  
        APPEND a pair of building position one AND height zero TO restrictions  
        SORT restrictions by building position ascending  
        
        FOR i FROM one TO the LENGTH OF restrictions MINUS one  
            SET idx TO element at position zero of element at position i of restrictions  
            SET height TO element at position one of element at position i of restrictions  
            SET prev_idx TO element at position zero of element at position i MINUS one of restrictions  
            SET prev_height TO element at position one of element at position i MINUS one of restrictions  
            SET the height of element at position i of restrictions TO the smaller OF height AND prev_height PLUS idx MINUS prev_idx  
        END FOR  
        
        FOR i FROM the LENGTH OF restrictions MINUS two DOWN TO zero  
            SET idx TO element at position zero of element at position i of restrictions  
            SET height TO element at position one of element at position i of restrictions  
            SET next_idx TO element at position zero of element at position i PLUS one of restrictions  
            SET next_height TO element at position one of element at position i PLUS one of restrictions  
            SET the height of element at position i of restrictions TO the smaller OF height AND next_height PLUS next_idx MINUS idx  
        END FOR  
        
        SET max_height TO zero  
        SET prev_idx TO element at position zero of element at position zero of restrictions  
        SET prev_height TO element at position one of element at position zero of restrictions  
        
        FOR i FROM one TO the LENGTH OF restrictions MINUS one  
            SET idx TO element at position zero of element at position i of restrictions  
            SET height TO element at position one of element at position i of restrictions  
            
            SET candidate_height TO the integer division OF idx MINUS prev_idx PLUS prev_height PLUS height BY two  
            IF candidate_height GREATER THAN max_height  
                SET max_height TO candidate_height  
            END IF  
            
            SET prev_idx TO idx  
            SET prev_height TO height  
        END FOR  
        
        SET tail_height TO prev_height PLUS n MINUS prev_idx  
        IF tail_height GREATER THAN max_height  
            SET max_height TO tail_height  
        END IF  
        
        RETURN max_height  
    END FUNCTION  
END CLASS
