CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty STRING THEN  
            RETURN empty LIST  
        END IF  

        SET phone_map TO MAP of digit CHARACTER TWO TO string of letters a b c AND digit CHARACTER THREE TO string of letters d e f AND digit CHARACTER FOUR TO string of letters g h i AND digit CHARACTER FIVE TO string of letters j k l AND digit CHARACTER SIX TO string of letters m n o AND digit CHARACTER SEVEN TO string of letters p q r s AND digit CHARACTER EIGHT TO string of letters t u v AND digit CHARACTER NINE TO string of letters w x y z  

        FUNCTION backtrack(combination, next_digits)  
            IF LENGTH OF next_digits EQUALS zero THEN  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN phone_map at element at position one of next_digits  
                    CALL backtrack with combination PLUS letter AND substring of next_digits from position two TO end  
                END FOR  
            END IF  
        END FUNCTION  

        SET output TO empty LIST  
        CALL backtrack with empty STRING AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
