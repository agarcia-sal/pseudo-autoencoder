CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO result of splitting input by each occurrence of a newline character  
        SET stack TO empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO number of tab characters contained in line  
            SET name TO substring of line starting from position depth TO the end  
            
            WHILE the length of stack GREATER THAN depth  
                REMOVE last element FROM stack  
            END WHILE  
            
            IF the length of stack GREATER THAN zero  
                SET current_length TO last element of stack PLUS the length of name PLUS one  
            ELSE  
                SET current_length TO the length of name  
            END IF  
            
            IF name CONTAINS the single character representing a dot  
                SET max_length TO the greater value between max_length AND current_length  
            ELSE  
                ADD current_length TO the end of stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
