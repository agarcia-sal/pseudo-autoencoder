CLASS Solution  
    FUNCTION minDistance(houses AS List OF integer, k AS integer)  
        SORT houses IN ascending order  
        SET n TO the LENGTH OF houses  
        
        CREATE two dimensional list cost WITH dimensions n BY n FILLED WITH zero  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost at position i and position j TO cost at position i PLUS one and position j MINUS one PLUS element at position j of houses MINUS element at position i of houses  
            END FOR  
        END FOR  
        
        CREATE two dimensional list dp WITH dimensions n PLUS one BY k PLUS one FILLED WITH positive infinity  
        SET dp at position zero and position zero TO zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET dp at position i and position j TO the smaller VALUE BETWEEN dp at position i and position j AND dp at position p and position j MINUS one PLUS cost at position p and position i MINUS one  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp at position n and position k  
    END FUNCTION  
END CLASS
