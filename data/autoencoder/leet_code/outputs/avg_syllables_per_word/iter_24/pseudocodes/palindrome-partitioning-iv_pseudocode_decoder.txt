CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            IF sub EQUALS the reverse sequence of elements IN sub THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
  
        SET n TO the LENGTH OF s  
  
        CREATE two dimensional list is_pal WITH n ROWS AND n COLUMNS EACH INITIALIZED TO false  
  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF element at position i of s EQUALS element at position j of s AND  
                   (j MINUS i LESS THAN OR EQUAL TO one OR is_pal at row i PLUS one column j MINUS one EQUALS true) THEN  
                    SET is_pal at row i column j TO true  
                END IF  
            END FOR  
        END FOR  
  
        FOR i FROM one TO n MINUS two  
            IF is_pal at row zero column i MINUS one EQUALS true THEN  
                FOR j FROM i TO n MINUS two  
                    IF is_pal at row i column j EQUALS true AND is_pal at row j PLUS one column n MINUS one EQUALS true THEN  
                        RETURN true  
                    END IF  
                END FOR  
            END IF  
        END FOR  
  
        RETURN false  
    END FUNCTION  
END CLASS
