CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix x y)  
            SET directions TO list of pairs of negative one zero one zero zero and one zero minus one zero one  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n  
                    SET element at position nx of matrix at position ny TO one MINUS element at position nx of matrix at position ny  
                END IF  
            END FOR  
        END FUNCTION  

        FUNCTION matrix_to_tuple(matrix)  
            RETURN tuple of tuple of each row IN matrix  
        END FUNCTION  

        SET m TO LENGTH OF mat  
        SET n TO LENGTH OF element at position zero of mat  
        SET target TO tuple of tuple of zero repeated n times for each m times  
        SET initial_state TO matrix_to_tuple(mat)  
        SET queue TO list containing pair of initial_state and zero  
        SET visited TO empty set  
        ADD initial_state TO visited  

        WHILE LENGTH OF queue GREATER THAN zero  
            SET current_state TO element at position zero of element at position zero of queue  
            SET steps TO element at position one of element at position zero of queue  
            REMOVE element at position zero FROM queue  
            IF current_state EQUALS target  
                RETURN steps  
            END IF  
            SET current_matrix TO list of list of each row IN current_state  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO deepcopy of current_matrix  
                    CALL flip(next_matrix i j)  
                    SET next_state TO matrix_to_tuple(next_matrix)  
                    IF next_state NOT IN visited  
                        ADD next_state TO visited  
                        APPEND pair of next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
