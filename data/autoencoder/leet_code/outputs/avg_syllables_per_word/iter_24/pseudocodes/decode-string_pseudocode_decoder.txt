CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each character IN s  
            IF character IS a digit character  
                SET d TO d concatenated WITH character  
            ELSE IF character IS an alphabetic character  
                SET a TO a concatenated WITH character  
            ELSE IF character EQUALS the left bracket character  
                APPEND the tuple consisting of a and the integer conversion of d TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF character EQUALS the right bracket character  
                SET p comma n TO the last element removed from stack  
                SET a TO p concatenated WITH a repeated n times  
            END IF  
        END FOR  
        
        RETURN a  
    END FUNCTION  
END CLASS
