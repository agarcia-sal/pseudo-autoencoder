CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO list containing zero  
        FOR each num IN nums  
            APPEND the element at the last position of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        ASSIGN reference TO new deque named dq  
        SET min_length TO infinity  
        FOR each index and current_sum IN enumeration of prefix_sum  
            WHILE dq is not empty AND current_sum MINUS element at position element at first position of dq of prefix_sum GREATER THAN OR EQUAL TO k  
                SET min_length TO the smaller value BETWEEN min_length AND index MINUS the element removed from the front of dq  
            END WHILE  
            WHILE dq is not empty AND current_sum LESS THAN OR EQUAL TO element at position element at last position of dq of prefix_sum  
                REMOVE the element at the last position from dq  
            END WHILE  
            APPEND index TO dq  
        END FOR  
        IF min_length NOT EQUALS infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
