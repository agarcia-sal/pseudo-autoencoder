CLASS Solution  
    FUNCTION kSimilarity(s1 parameter s1 AS string s2 parameter s2 AS string) RETURNS integer  
        FUNCTION neighbors(s parameter s AS string)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND element at position i of s EQUALS element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF element at position j of s EQUALS element at position i of s2 AND element at position j of s NOT EQUALS element at position j of s2  
                    YIELD concatenation of substring of s from position zero TO position i MINUS one PLUS element at position j of s PLUS substring of s from position i PLUS one TO position j MINUS one PLUS element at position i of s PLUS substring of s from position j PLUS one TO the end  
                END IF  
            END FOR  
        END FUNCTION  
          
        SET queue TO new double ended queue containing tuple of s1 and zero  
        SET visited TO new set containing s1  
          
        WHILE queue is not empty  
            SET s, swaps TO the result of removing and returning the element at the start of queue  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR neighbor IN neighbors(s)  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND tuple of neighbor and swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
