CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO empty list  
        FOR each num IN target  
            APPEND negative of num TO max_heap  
        END FOR  
        TRANSFORM max_heap INTO a heap structure  
        
        SET total_sum TO the sum of all elements in target  
        
        WHILE the element at position zero of max_heap LESS THAN negative one holds  
            SET largest TO the negative of the element removed from max_heap at the root position  
            SET rest_sum TO total_sum MINUS largest  
            
            IF largest LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero THEN  
                RETURN False  
            END IF  
            
            SET new_value TO largest MODULO rest_sum  
            IF new_value EQUALS zero AND rest_sum NOT EQUALS one THEN  
                RETURN False  
            END IF  
            
            SET total_sum TO rest_sum PLUS new_value  
            ADD negative of new_value TO max_heap maintaining heap property  
        END WHILE  
        
        RETURN True  
    END FUNCTION  
END CLASS
