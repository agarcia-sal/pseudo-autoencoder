CLASS Solution  
    FUNCTION killProcess(pid AS List of integers ppid AS List of integers kill AS integer) RETURNS List of integers  
        SET tree TO a mapping from integer to a list of integers with default empty list  
        FOR each pair of child and parent IN the pairs formed by corresponding elements of pid and ppid  
            APPEND child TO the list in tree associated with parent  
        END FOR  
        
        SET queue TO a list containing kill  
        SET killed_processes TO an empty list  
        
        WHILE the LENGTH OF queue is GREATER THAN zero  
            SET current TO the element at position zero of queue  
            REMOVE the element at position zero FROM queue  
            APPEND current TO killed_processes  
            EXTEND queue BY the list in tree associated with current  
        END WHILE  
        
        RETURN killed_processes  
    END FUNCTION  
END CLASS
