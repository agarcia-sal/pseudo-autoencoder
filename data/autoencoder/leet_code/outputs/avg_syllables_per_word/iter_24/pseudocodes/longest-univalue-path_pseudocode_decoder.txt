CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue NOT EQUALS zero  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with p.left and q.left AND the result of is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestUnivaluePath(self, root)  
        FUNCTION dfs(node)  
            IF node EQUALS None  
                RETURN zero AND zero  
            END IF  
            SET left_length and left_value TO the two results returned by calling dfs on node.left  
            SET right_length and right_value TO the two results returned by calling dfs on node.right  
            SET left_arrow TO zero  
            SET right_arrow TO zero  
            IF node.left NOT EQUALS None AND node.left.val EQUALS node.val  
                SET left_arrow TO left_length PLUS one  
            END IF  
            IF node.right NOT EQUALS None AND node.right.val EQUALS node.val  
                SET right_arrow TO right_length PLUS one  
            END IF  
            SET self.max_length TO the maximum OF self.max_length AND left_arrow PLUS right_arrow  
            RETURN the maximum OF left_arrow AND right_arrow AND node.val  
        END FUNCTION  
        SET self.max_length TO zero  
        CALL dfs with root  
        RETURN self.max_length  
    END FUNCTION  
END CLASS
