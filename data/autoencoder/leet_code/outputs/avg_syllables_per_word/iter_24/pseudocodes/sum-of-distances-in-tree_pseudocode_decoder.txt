CLASS Solution  
    FUNCTION sumOfDistancesInTree(n, edges)  
        SET graph TO a mapping with default empty list  
        FOR each element a AND element b IN edges  
            APPEND element b TO the list mapped from key a in graph  
            APPEND element a TO the list mapped from key b in graph  
        END FOR  

        SET count TO a list of size n where each element is one  
        SET dist TO a list of size n where each element is zero  

        FUNCTION dfs1(u, p)  
            FOR each element v IN the list mapped from key u in graph  
                IF element v NOT EQUALS element p THEN  
                    CALL dfs1 with parameters element v and element u  
                    INCREMENT element at position u of count BY element at position v of count  
                    INCREMENT element at position u of dist BY element at position v of dist PLUS element at position v of count  
                END IF  
            END FOR  
        END FUNCTION  

        CALL dfs1 with parameters zero and negative one  

        FUNCTION dfs2(u, p)  
            FOR each element v IN the list mapped from key u in graph  
                IF element v NOT EQUALS element p THEN  
                    SET element at position v of dist TO element at position u of dist MINUS element at position v of count PLUS the difference of parameter n MINUS element at position v of count  
                    CALL dfs2 with parameters element v and element u  
                END IF  
            END FOR  
        END FUNCTION  

        CALL dfs2 with parameters zero and negative one  

        RETURN dist  
    END FUNCTION  
END CLASS
