CLASS Solution  
    FUNCTION canPartitionKSubsets(nums, k)  
        SET total_sum TO the sum of all elements in nums  
        IF the remainder when total_sum is divided by k is NOT EQUALS to zero  
            RETURN False  
        END IF  
        SET target TO the integer division of total_sum by k  
        SORT nums in descending order  
        
        FUNCTION can_partition(index, k, current_sum, used)  
            IF k EQUALS zero  
                RETURN True  
            END IF  
            IF current_sum EQUALS target  
                RETURN the result of calling can_partition with index SET TO zero k DECREMENTED BY one current_sum SET TO zero and used unchanged  
            END IF  
            
            FOR i FROM index TO the length of nums MINUS one  
                IF the element at position i of used IS False AND current_sum PLUS the element at position i of nums LESS THAN OR EQUAL TO target  
                    SET the element at position i of used TO True  
                    IF the result of calling can_partition with i INCREMENTED BY one k unchanged current_sum PLUS the element at position i of nums and used is True  
                        RETURN True  
                    END IF  
                    SET the element at position i of used TO False  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET used TO a list of False values with the length EQUALS to the length of nums  
        RETURN the result of calling can_partition with zero k zero and used  
    END FUNCTION  
END CLASS
