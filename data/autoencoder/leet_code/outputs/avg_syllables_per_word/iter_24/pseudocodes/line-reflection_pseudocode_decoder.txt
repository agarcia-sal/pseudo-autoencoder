CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO an empty set  
        FOR each point IN points  
            CONVERT point TO a tuple  
            ADD tuple point TO unique_points  
        END FOR  
        SET min_x TO the minimum value of element at position zero of each point IN unique_points  
        SET max_x TO the maximum value of element at position zero of each point IN unique_points  
        SET line_of_reflection TO min_x PLUS max_x DIVIDED BY two  
        FOR each point IN unique_points  
            SET x TO element at position zero of point  
            SET y TO element at position one of point  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF tuple of reflected_x and y NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
