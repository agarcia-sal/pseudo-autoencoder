CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY two  
        SET total_sum TO the SUM OF all elements IN nums  
        SET target TO total_sum DIVIDED BY two  
        SET min_diff TO positive infinity  
        
        SET left_half TO the first n elements OF nums  
        SET right_half TO the elements OF nums FROM position n TO the end  
        
        SET left_sums TO a list OF empty dictionaries WITH length n PLUS one  
        SET right_sums TO a list OF empty dictionaries WITH length n PLUS one  
        
        FOR i FROM zero TO n  
            FOR each combination OF left_half TAKEN i AT A TIME  
                SET subset_sum TO the SUM OF elements IN combination  
                IF subset_sum NOT IN left_sums AT position i  
                    SET left_sums AT position i WITH key subset_sum TO zero  
                END IF  
                INCREMENT left_sums AT position i WITH key subset_sum BY one  
            END FOR  
            
            FOR each combination OF right_half TAKEN i AT A TIME  
                SET subset_sum TO the SUM OF elements IN combination  
                IF subset_sum NOT IN right_sums AT position i  
                    SET right_sums AT position i WITH key subset_sum TO zero  
                END IF  
                INCREMENT right_sums AT position i WITH key subset_sum BY one  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO n  
            SET left_values TO the SORTED list OF keys OF left_sums AT position i  
            SET right_values TO the SORTED list OF keys OF right_sums AT position n MINUS i  
            
            SET j TO zero  
            SET k TO the LENGTH OF right_values MINUS one  
            
            WHILE j LESS THAN the LENGTH OF left_values AND k GREATER THAN OR EQUAL TO zero  
                SET current_sum TO the element AT position j OF left_values PLUS the element AT position k OF right_values  
                SET candidate_diff TO total_sum MINUS two MULTIPLIED BY current_sum  
                IF candidate_diff LESS THAN zero  
                    SET candidate_diff TO the NEGATION OF candidate_diff  
                END IF  
                IF candidate_diff LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
                
                IF current_sum LESS THAN target  
                    INCREMENT j BY one  
                ELSE IF current_sum GREATER THAN target  
                    DECREMENT k BY one  
                ELSE  
                    RETURN zero  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
