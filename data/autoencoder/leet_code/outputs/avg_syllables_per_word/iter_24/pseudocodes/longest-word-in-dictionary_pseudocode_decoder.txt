CLASS Solution  
    FUNCTION longestWord(words)  
        CALL words SORTED BY LENGTH OF element AND THEN BY element LEXICOGRAPHICALLY  
        SET valid_words TO an empty set  
        SET longest_word TO an empty string  
        FOR each word IN words  
            IF the LENGTH OF word EQUALS one OR the SUBSTRING of word FROM position zero TO the LENGTH OF word MINUS one IS IN valid_words  
                ADD word TO valid_words  
                IF the LENGTH OF word GREATER THAN the LENGTH OF longest_word  
                    SET longest_word TO word  
                END IF  
            END IF  
        END FOR  
        RETURN longest_word  
    END FUNCTION  
END CLASS
