CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an EMPTY SET  
        SET min_heap TO a LIST CONTAINING a TUPLE of the element at position zero zero of grid AND zero AND zero  
        ADD a TUPLE of zero AND zero TO visited  
        SET directions TO a LIST CONTAINING a TUPLE of zero AND one AND a TUPLE of one AND zero AND a TUPLE of zero AND negative one AND a TUPLE of negative one AND zero  
        WHILE the LENGTH OF min_heap IS GREATER THAN zero  
            REMOVE and RETURN the SMALLEST ELEMENT FROM min_heap AND UNPACK INTO t AND i AND j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            FOR each di AND dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND a TUPLE of ni AND nj IS NOT IN visited  
                    ADD a TUPLE of ni AND nj TO visited  
                    ADD a TUPLE of the GREATER VALUE BETWEEN t AND the element at position ni nj of grid AND ni AND nj TO min_heap MAINTAINING THE HEAP PROPERTY  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
