CLASS Solution  
    FUNCTION numberOfWays WITH parameter corridor  
        SET MOD TO the number one followed by nine zeros PLUS seven  
        SET seats TO an empty list  
        FOR each index AND character IN corridor  
            IF character EQUALS the letter S  
                APPEND index TO seats  
            END IF  
        END FOR  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM the number two TO the LENGTH OF seats MINUS one, STEP two  
            SET difference TO the element at position index of seats MINUS the element at position index MINUS one of seats  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
