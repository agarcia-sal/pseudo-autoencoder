CLASS ListNode  
    FUNCTION __init__ with parameters val EQUALS zero and next EQUALS none  
        SET the val attribute TO val  
        SET the next attribute TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS an empty collection  
        RETURN none  
    END IF  
    SET head TO a new instance of ListNode with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position one to the end  
        SET node TO a new instance of ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 and p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the expression that the val attribute of p1 EQUALS the val attribute of p2 AND is_same_list called with the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ with parameters val EQUALS zero and left EQUALS none and right EQUALS none  
        SET the val attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS an empty collection  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element removed from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values DOES NOT EQUAL none  
            SET the left attribute of node TO a new instance of TreeNode with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values DOES NOT EQUAL none  
            SET the right attribute of node TO a new instance of TreeNode with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN false  
    ELSE  
        RETURN the expression that is_same_tree called with the left attribute of p and the left attribute of q AND is_same_tree called with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST with parameter head  
        IF head EQUALS none  
            RETURN none  
        END IF  

        FUNCTION findMiddle with parameters left and right  
            SET slow TO left  
            SET fast TO left  
            WHILE fast DOES NOT EQUAL right AND the next attribute of fast DOES NOT EQUAL right  
                SET slow TO the next attribute of slow  
                SET fast TO the next attribute of the next attribute of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  

        FUNCTION convertListToBST with parameters left and right  
            IF left EQUALS right  
                RETURN none  
            END IF  
            SET mid TO the result of findMiddle called with left and right  
            SET node TO a new instance of TreeNode with the val attribute of mid  
            SET the left attribute of node TO the result of convertListToBST called with left and mid  
            SET the right attribute of node TO the result of convertListToBST called with the next attribute of mid and right  
            RETURN node  
        END FUNCTION  

        RETURN the result of convertListToBST called with head and none  
    END FUNCTION  
END CLASS
