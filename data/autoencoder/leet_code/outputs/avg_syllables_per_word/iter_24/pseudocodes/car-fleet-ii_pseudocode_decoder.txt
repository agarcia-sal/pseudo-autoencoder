CLASS Solution  
    FUNCTION getCollisionTimes(cars)  
        SET n TO the LENGTH OF cars  
        SET result TO a LIST OF NEGATIVE one POINT zero REPEATED n TIMES  
        SET stack TO an EMPTY LIST  

        FOR i FROM n MINUS one TO zero STEP MINUS one  
            SET pos TO the FIRST element AT position i OF cars  
            SET speed TO the SECOND element AT position i OF cars  

            WHILE the LENGTH OF stack IS GREATER THAN zero  
                SET next_index TO the LAST element OF stack  
                SET next_pos TO the FIRST element AT position next_index OF cars  
                SET next_speed TO the SECOND element AT position next_index OF cars  

                IF speed IS LESS THAN OR EQUAL TO next_speed THEN  
                    REMOVE the LAST element FROM stack  
                    CONTINUE  
                END IF  

                SET collision_time TO next_pos MINUS pos DIVIDED BY speed MINUS next_speed  

                IF the element OF result AT position next_index IS GREATER THAN zero AND collision_time IS GREATER THAN OR EQUAL TO the element OF result AT position next_index THEN  
                    REMOVE the LAST element FROM stack  
                    CONTINUE  
                END IF  

                SET the element OF result AT position i TO collision_time  
                BREAK  
            END WHILE  

            APPEND i TO stack  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
