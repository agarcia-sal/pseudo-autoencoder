CLASS Solution  
    FUNCTION poorPigs(buckets minutesToDie minutesToTest)  
        SET number_of_states TO the integer division of minutesToTest BY minutesToDie PLUS one  
        SET number_of_pigs TO the smallest integer greater than or equal to the logarithm of buckets BASED ON the logarithm of number_of_states  
        RETURN number_of_pigs  
    END FUNCTION  
END CLASS
