CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        FOR each num IN the elements of nums starting from the last element moving to the first element  
            IF num LESS THAN second  
                RETURN true  
            END IF  
            WHILE the LENGTH OF stack GREATER THAN zero AND the element at the last position of stack LESS THAN num  
                SET second TO the element removed from the last position of stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
