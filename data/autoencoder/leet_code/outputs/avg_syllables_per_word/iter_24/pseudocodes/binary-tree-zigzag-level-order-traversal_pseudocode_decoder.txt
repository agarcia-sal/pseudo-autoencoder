CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value equal to the element at position one of values and left equal to None and right equal to None  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        REMOVE and ASSIGN the element at the front of queue TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not None THEN  
            SET node.left TO a new TreeNode with value equal to the element at position i of values and left equal to None and right equal to None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values is not None THEN  
            SET node.right TO a new TreeNode with value equal to the element at position i of values and left equal to None and right equal to None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left and q.left AND the result of is_same_tree with parameters p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self, root)  
        IF root EQUALS None THEN  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a double ended queue with root as the only element  
        SET left_to_right TO True  
        WHILE the queue is not empty  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty double ended queue  
            FOR each index FROM one TO level_size  
                REMOVE and ASSIGN the element at the front of queue TO node  
                IF left_to_right EQUALS True THEN  
                    APPEND node.val TO current_level  
                ELSE  
                    INSERT node.val at the front of current_level  
                END IF  
                IF node.left NOT EQUALS None THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right NOT EQUALS None THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND the list form of current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
