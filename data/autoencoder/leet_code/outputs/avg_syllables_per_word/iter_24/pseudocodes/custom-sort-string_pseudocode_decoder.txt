CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO an empty dictionary  
        SET i TO zero  
        FOR each character IN order  
            SET element at position character IN d TO i  
            INCREMENT i BY one  
        END FOR  
        SET list_of_characters TO elements of s sorted in order such that for each character the value obtained FROM d at position character IF present OTHERWISE zero determines the sorting order  
        SET result_string TO concatenation of all characters IN list_of_characters in their sorted order  
        RETURN result_string  
    END FUNCTION  
END CLASS
