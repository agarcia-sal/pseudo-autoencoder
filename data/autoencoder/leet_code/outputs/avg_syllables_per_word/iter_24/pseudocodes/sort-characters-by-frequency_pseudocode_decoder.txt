CLASS Solution  
    FUNCTION frequencySort(s)  
        SET frequency TO the result of counting the occurrences of each character in the string s  
        SET sorted_characters TO the list of characters in frequency sorted in order such that characters with higher frequency come before characters with lower frequency  
        SET result TO the empty string  
        FOR each character IN sorted_characters  
            SET repeated_characters TO the character repeated the number of times equal to the frequency of the character in frequency  
            APPEND repeated_characters TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
