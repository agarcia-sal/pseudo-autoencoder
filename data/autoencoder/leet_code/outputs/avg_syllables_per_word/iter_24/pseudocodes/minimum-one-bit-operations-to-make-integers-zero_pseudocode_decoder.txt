CLASS Solution  
    FUNCTION minimumOneBitOperations(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE one MULTIPLIED BY two RAISED TO THE POWER OF k PLUS one LESS THAN OR EQUAL TO n HOLDS  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO one MULTIPLIED BY two RAISED TO THE POWER OF k  
        RETURN one MULTIPLIED BY two RAISED TO THE POWER OF k PLUS one MINUS one MINUS CALL minimumOneBitOperations WITH PARAMETER n XOR most_significant_bit  
    END FUNCTION  
END CLASS
