CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with first element of values  
    SET p TO head  
    FOR each val IN values starting from second element to end  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN logical AND of expression that element val of p1 EQUALS element val of p2 AND recursive call to is_same_list with element next of p1 AND element next of p2  
END FUNCTION

CLASS Solution  
    FUNCTION deleteDuplicates(self head)  
        SET dummy TO new ListNode with value zero and next reference head  
        SET prev TO dummy  
        WHILE head holds  
            IF element next of head EXISTS AND element val of head EQUALS element val of element next of head  
                WHILE element next of head EXISTS AND element val of head EQUALS element val of element next of head  
                    SET head TO element next of head  
                END WHILE  
                SET element next of prev TO element next of head  
            ELSE  
                SET prev TO element next of prev  
            END IF  
            SET head TO element next of head  
        END WHILE  
        RETURN element next of dummy  
    END FUNCTION  
END CLASS
