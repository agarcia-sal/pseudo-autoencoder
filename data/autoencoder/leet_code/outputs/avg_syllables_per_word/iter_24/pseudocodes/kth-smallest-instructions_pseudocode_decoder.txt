CLASS Solution  
    FUNCTION kthSmallestPath(destination parameter of list of integers k parameter of integer) RETURNS string  
        SET row TO the element at position zero of destination  
        SET col TO the element at position one of destination  
        SET path TO empty list  
        SET total_moves TO the sum of col AND row  
        FOR index FROM zero TO total_moves MINUS one  
            IF col GREATER THAN zero  
                SET paths_with_H TO the combination of the sum of row AND col MINUS one CHOOSE col MINUS one  
                IF k LESS THAN OR EQUAL TO paths_with_H  
                    APPEND the move "H" TO path  
                    DECREMENT col BY one  
                ELSE  
                    APPEND the move "V" TO path  
                    DECREMENT row BY one  
                    DECREMENT k BY paths_with_H  
                END IF  
            ELSE  
                APPEND the move "V" TO path  
                DECREMENT row BY one  
            END IF  
        END FOR  
        RETURN the concatenation of all elements of path into a single string  
    END FUNCTION  
END CLASS
