CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty list THEN  
        RETURN None  
    END IF  
    SET head TO new ListNode constructed with element at position one of values  
    SET p TO head  
    FOR each val IN the sublist of values starting from position two TO the end  
        SET node TO new ListNode constructed with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the logical AND of the expression that the val attribute of p1 EQUALS the val attribute of p2 AND the recursive call to is_same_list with the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight(self, head, k)  
        IF head EQUALS None OR the next attribute of head EQUALS None OR k EQUALS zero THEN  
            RETURN head  
        END IF  
        SET length TO one  
        SET tail TO head  
        WHILE the next attribute of tail DOES NOT EQUAL None  
            SET tail TO the next attribute of tail  
            INCREMENT length BY one  
        END WHILE  
        SET the next attribute of tail TO head  
        SET k TO k MODULO length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR index FROM one TO steps_to_new_tail  
            SET new_tail TO the next attribute of new_tail  
        END FOR  
        SET new_head TO the next attribute of new_tail  
        SET the next attribute of new_tail TO None  
        RETURN new_head  
    END FUNCTION  
END CLASS
