CLASS Solution  
    FUNCTION reversePairs(nums)  
        FUNCTION merge_and_count(nums, start, end)  
            IF start GREATER THAN OR EQUAL TO end  
                RETURN zero  
            END IF  
            SET mid TO the result of start PLUS end DIVIDED BY two using integer division  
            SET count TO the result of calling merge_and_count with nums start and mid PLUS the result of calling merge_and_count with nums mid PLUS one and end  
            SET j TO mid PLUS one  
            FOR i FROM start TO mid  
                WHILE j LESS THAN OR EQUAL TO end AND the element at position i of nums GREATER THAN two MULTIPLIED BY the element at position j of nums  
                    INCREMENT j BY one  
                END WHILE  
                INCREMENT count BY j MINUS mid MINUS one  
            END FOR  
            CREATE an empty list called merged  
            SET left TO start  
            SET right TO mid PLUS one  
            WHILE left LESS THAN OR EQUAL TO mid AND right LESS THAN OR EQUAL TO end  
                IF the element at position left of nums LESS THAN OR EQUAL TO the element at position right of nums  
                    APPEND the element at position left of nums TO merged  
                    INCREMENT left BY one  
                ELSE  
                    APPEND the element at position right of nums TO merged  
                    INCREMENT right BY one  
                END IF  
            END WHILE  
            FOR position FROM left TO mid  
                APPEND the element at position position of nums TO merged  
            END FOR  
            FOR position FROM right TO end  
                APPEND the element at position position of nums TO merged  
            END FOR  
            FOR position FROM start TO end  
                SET the element at position position of nums TO the element at position position MINUS start of merged  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN the result of calling merge_and_count with nums zero and the length of nums MINUS one  
    END FUNCTION  
END CLASS
