CLASS Solution  
    FUNCTION isMatch(s parameter p parameter)  
        SET dp TO a two dimensional list of False values with dimensions as the LENGTH OF s PLUS one by the LENGTH OF p PLUS one  
        SET the element at position zero zero of dp TO True  
        FOR each j FROM one TO the LENGTH OF p PLUS one  
            IF element at position j minus one of p EQUALS the character representing asterisk  
                SET the element at position zero j of dp TO the element at position zero j minus one of dp  
            END IF  
        END FOR  
        FOR each i FROM one TO the LENGTH OF s PLUS one  
            FOR each j FROM one TO the LENGTH OF p PLUS one  
                IF element at position j minus one of p EQUALS the character representing asterisk  
                    SET the element at position i j of dp TO the element at position i minus one j of dp OR the element at position i j minus one of dp  
                ELSE IF element at position j minus one of p EQUALS the character representing question mark OR element at position i minus one of s EQUALS element at position j minus one of p  
                    SET the element at position i j of dp TO the element at position i minus one j minus one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position the LENGTH OF s the LENGTH OF p of dp  
    END FUNCTION  
END CLASS
