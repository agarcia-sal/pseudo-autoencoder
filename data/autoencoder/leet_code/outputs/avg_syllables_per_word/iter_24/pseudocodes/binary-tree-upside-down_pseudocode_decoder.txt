CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EQUAL TO none OR the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH the element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT EQUAL TO none THEN  
            SET node.left TO A NEW TreeNode WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT EQUAL TO none THEN  
            SET node.right TO A NEW TreeNode WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS EQUAL TO none AND q IS EQUAL TO none THEN  
        RETURN True  
    ELSE IF p IS EQUAL TO none OR q IS EQUAL TO none THEN  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree(self, root)  
        IF root IS EQUAL TO none OR root.left IS EQUAL TO none THEN  
            RETURN root  
        END IF  
        SET new_root TO self.upsideDownBinaryTree(root.left)  
        SET root.left.left TO root.right  
        SET root.left.right TO root  
        SET root.left TO none  
        SET root.right TO none  
        RETURN new_root  
    END FUNCTION  
END CLASS
