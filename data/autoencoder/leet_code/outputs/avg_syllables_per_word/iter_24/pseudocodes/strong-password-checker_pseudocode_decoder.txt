CLASS Solution  
    FUNCTION strongPasswordChecker(password)  
        SET n TO the LENGTH OF password  
        
        SET has_lower TO FALSE  
        FOR each character IN password  
            IF character IS A lowercase letter  
                SET has_lower TO TRUE  
                BREAK the loop  
            END IF  
        END FOR  
        
        SET has_upper TO FALSE  
        FOR each character IN password  
            IF character IS AN uppercase letter  
                SET has_upper TO TRUE  
                BREAK the loop  
            END IF  
        END FOR  
        
        SET has_digit TO FALSE  
        FOR each character IN password  
            IF character IS A digit  
                SET has_digit TO TRUE  
                BREAK the loop  
            END IF  
        END FOR  
        
        SET missing_types TO three MINUS the SUM OF has_lower PLUS has_upper PLUS has_digit  
        
        SET replace TO zero  
        SET one TO zero  
        SET two TO zero  
        SET i TO two  
        
        WHILE i IS LESS THAN n  
            IF the element at position i of password EQUALS the element at position i MINUS one of password AND the element at position i of password EQUALS the element at position i MINUS two of password  
                SET length TO two  
                WHILE i IS LESS THAN n AND the element at position i of password EQUALS the element at position i MINUS one of password  
                    INCREMENT length BY one  
                    INCREMENT i BY one  
                END WHILE  
                
                INCREMENT replace BY length DIVIDED BY three  
                
                IF length MODULO three EQUALS zero  
                    INCREMENT one BY one  
                ELSE IF length MODULO three EQUALS one  
                    INCREMENT two BY one  
                END IF  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        IF n IS LESS THAN six  
            RETURN the GREATER OF missing_types AND six MINUS n  
        ELSE IF n IS LESS THAN OR EQUAL TO twenty  
            RETURN the GREATER OF missing_types AND replace  
        ELSE  
            SET delete TO n MINUS twenty  
            
            DECREMENT replace BY the SMALLER OF delete AND one  
            
            DECREMENT replace BY the INTEGER DIVISION BY two OF the SMALLER OF the GREATER OF delete MINUS one AND zero AND two MULTIPLIED BY two  
            
            DECREMENT replace BY the INTEGER DIVISION BY three OF the GREATER OF delete MINUS one MINUS two MULTIPLIED BY two AND zero  
            
            RETURN delete PLUS the GREATER OF missing_types AND replace  
        END IF  
    END FUNCTION  
END CLASS
