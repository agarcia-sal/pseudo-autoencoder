CLASS Solution  
    FUNCTION isTransformable WITH PARAMETERS s t  
        CREATE a mapping named pos FROM digits TO empty lists  
        FOR each index i AND character c IN s  
            SET digit TO the integer value of character c  
            ADD index i TO the heap associated to digit in pos  
        END FOR  
        
        FOR each index i AND character c IN t  
            SET digit TO the integer value of character c  
            IF the heap associated to digit in pos IS empty  
                RETURN false  
            END IF  
            FOR each smaller FROM zero TO digit MINUS one  
                IF the heap associated to smaller in pos IS NOT empty AND the smallest element of the heap associated to smaller in pos IS LESS THAN the smallest element of the heap associated to digit in pos  
                    RETURN false  
                END IF  
            END FOR  
            REMOVE the smallest element from the heap associated to digit in pos  
        END FOR  
        
        RETURN true  
    END FUNCTION  
END CLASS
