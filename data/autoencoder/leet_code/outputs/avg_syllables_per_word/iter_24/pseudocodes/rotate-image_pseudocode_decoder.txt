CLASS Solution  
    FUNCTION rotate(matrix)  
        SET n TO the LENGTH OF matrix  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET temporary_value TO element at position j of element at position i of matrix  
                SET element at position j of element at position i of matrix TO element at position i of element at position j of matrix  
                SET element at position i of element at position j of matrix TO temporary_value  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO n MINUS one  
            SET left_pointer TO zero  
            SET right_pointer TO the LENGTH OF element at position i of matrix MINUS one  
            WHILE left_pointer IS LESS THAN right_pointer  
                SET temporary_value TO element at position left_pointer of element at position i of matrix  
                SET element at position left_pointer of element at position i of matrix TO element at position right_pointer of element at position i of matrix  
                SET element at position right_pointer of element at position i of matrix TO temporary_value  
                INCREMENT left_pointer BY one  
                DECREMENT right_pointer BY one  
            END WHILE  
        END FOR  
    END FUNCTION  
END CLASS
