CLASS Solution  
    FUNCTION findPairs(nums, k)  
        IF k LESS THAN zero THEN  
            RETURN zero  
        END IF  
          
        SET num_counts TO an empty dictionary  
        FOR each num IN nums  
            IF num IS A key IN num_counts THEN  
                SET current_count TO value associated with num in num_counts  
            ELSE  
                SET current_count TO zero  
            END IF  
            SET updated_count TO current_count PLUS one  
            ASSIGN updated_count AS value associated with num in num_counts  
        END FOR  
          
        SET unique_pairs TO zero  
        FOR each num IN the keys of num_counts  
            IF k EQUALS zero THEN  
                IF value associated with num in num_counts GREATER THAN one THEN  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                SET target_num TO num PLUS k  
                IF target_num IS A key IN num_counts THEN  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
          
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
