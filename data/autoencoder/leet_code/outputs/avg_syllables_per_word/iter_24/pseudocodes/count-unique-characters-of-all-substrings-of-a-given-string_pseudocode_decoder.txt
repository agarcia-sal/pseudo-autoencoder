CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        
        SET last TO a list of twenty six elements each SET TO negative one  
        SET last2 TO a list of twenty six elements each SET TO negative one  
        
        SET result TO zero  
        
        FOR each index i AND each character char IN s  
            SET idx TO the Unicode code point OF char MINUS the Unicode code point OF uppercase letter A  
            
            SET result TO result PLUS the product OF the difference BETWEEN i AND the element at position idx OF last AND the difference BETWEEN the element at position idx OF last AND the element at position idx OF last2  
            SET result TO result MODULO MOD  
            
            SET the element at position idx OF last2 TO the element at position idx OF last  
            SET the element at position idx OF last TO i  
        END FOR  
        
        FOR each index i FROM zero TO twenty five  
            SET result TO result PLUS the product OF the difference BETWEEN n AND the element at position i OF last AND the difference BETWEEN the element at position i OF last AND the element at position i OF last2  
            SET result TO result MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
