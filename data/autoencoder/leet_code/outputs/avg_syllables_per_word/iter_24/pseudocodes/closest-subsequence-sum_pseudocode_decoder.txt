CLASS Solution  
    FUNCTION minAbsDifference(nums, goal)  
        SET n TO the LENGTH OF nums  
        SET mid TO n DIVIDED BY two  
        
        SET left_sums TO a list containing zero  
        FOR each number IN the elements at positions from zero TO position mid MINUS one of nums  
            SET new_sums TO empty list  
            FOR each sum_value IN left_sums  
                APPEND sum_value PLUS number TO new_sums  
            END FOR  
            APPEND all elements of new_sums TO left_sums  
        END FOR  
        
        SET right_sums TO a list containing zero  
        FOR each number IN the elements at positions from mid TO the end of nums  
            SET new_sums TO empty list  
            FOR each sum_value IN right_sums  
                APPEND sum_value PLUS number TO new_sums  
            END FOR  
            APPEND all elements of new_sums TO right_sums  
        END FOR  
        
        SORT right_sums in ascending order  
        
        SET min_diff TO positive infinity  
        
        FOR each left_sum_value IN left_sums  
            SET target TO goal MINUS left_sum_value  
            SET pos TO the result of binary search to find the leftmost position to insert target IN right_sums  
            
            IF pos LESS THAN the LENGTH OF right_sums THEN  
                SET candidate_diff TO the absolute value of left_sum_value PLUS the element at position pos of right_sums MINUS goal  
                IF candidate_diff LESS THAN min_diff THEN  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
            
            IF pos GREATER THAN zero THEN  
                SET candidate_diff TO the absolute value of left_sum_value PLUS the element at position pos MINUS one of right_sums MINUS goal  
                IF candidate_diff LESS THAN min_diff THEN  
                    SET min_diff TO candidate_diff  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
