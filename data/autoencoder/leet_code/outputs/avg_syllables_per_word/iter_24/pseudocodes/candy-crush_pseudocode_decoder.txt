CLASS Solution  
    FUNCTION candyCrush(board)  
        SET R TO the LENGTH OF board  
        SET C TO the LENGTH OF the element at position zero of board  
        SET stable TO false  
        
        WHILE stable EQUALS false  
            SET stable TO true  
            SET to_crush TO an empty set  
            
            FOR each r FROM zero TO R MINUS one  
                FOR each c FROM zero TO C MINUS two  
                    IF the ABSOLUTE VALUE of the element at position r of board at position c EQUALS the ABSOLUTE VALUE of the element at position r of board at position c PLUS one AND the ABSOLUTE VALUE of the element at position r of board at position c PLUS one EQUALS the ABSOLUTE VALUE of the element at position r of board at position c PLUS two AND the ABSOLUTE VALUE of the element at position r of board at position c NOT EQUALS zero  
                        SET stable TO false  
                        ADD the tuple of r and c TO to_crush  
                        ADD the tuple of r and c PLUS one TO to_crush  
                        ADD the tuple of r and c PLUS two TO to_crush  
                    END IF  
                END FOR  
            END FOR  
            
            FOR each r FROM zero TO R MINUS two  
                FOR each c FROM zero TO C MINUS one  
                    IF the ABSOLUTE VALUE of the element at position r of board at position c EQUALS the ABSOLUTE VALUE of the element at position r PLUS one of board at position c AND the ABSOLUTE VALUE of the element at position r PLUS one of board at position c EQUALS the ABSOLUTE VALUE of the element at position r PLUS two of board at position c AND the ABSOLUTE VALUE of the element at position r of board at position c NOT EQUALS zero  
                        SET stable TO false  
                        ADD the tuple of r and c TO to_crush  
                        ADD the tuple of r PLUS one and c TO to_crush  
                        ADD the tuple of r PLUS two and c TO to_crush  
                    END IF  
                END FOR  
            END FOR  
            
            FOR each tuple r and c IN to_crush  
                SET the element at position r of board at position c TO zero  
            END FOR  
            
            FOR each c FROM zero TO C MINUS one  
                SET wr TO R MINUS one  
                FOR each r FROM R MINUS one DOWN TO zero INCLUSIVE  
                    IF the element at position r of board at position c NOT EQUALS zero  
                        SET the element at position wr of board at position c TO the element at position r of board at position c  
                        DECREMENT wr BY one  
                    END IF  
                END FOR  
                FOR each wr FROM wr DOWN TO zero INCLUSIVE  
                    SET the element at position wr of board at position c TO zero  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN board  
    END FUNCTION  
END CLASS
