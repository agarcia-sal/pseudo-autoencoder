CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF the LENGTH OF remaining EQUALS zero THEN  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR index FROM one TO the LENGTH OF remaining  
                SET new_path TO the concatenation of path PLUS the element at position index of remaining  
                SET new_remaining TO the elements from position one TO position index MINUS one of remaining PLUS the elements from position index PLUS one TO the END of remaining  
                CALL backtrack WITH new_path AND new_remaining  
            END FOR  
        END FUNCTION  
        SET permutations TO an empty list  
        CALL backtrack WITH an empty list AND nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
