CLASS Solution  
    FUNCTION isMatch(s string p string) RETURNS boolean  
        SET dp TO two dimensional list of boolean FALSE with the number of rows equal to the length of s PLUS one and the number of columns equal to the length of p PLUS one  
        SET element at position zero zero of dp TO TRUE  
        
        FOR j FROM one TO the length of p  
            IF the element at position j MINUS one of p EQUALS asterisk  
                SET element at position zero j of dp TO the value of element at position zero j MINUS two of dp  
            END IF  
        END FOR  
        
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF the element at position j MINUS one of p EQUALS the element at position i MINUS one of s OR the element at position j MINUS one of p EQUALS period  
                    SET element at position i j of dp TO the value of element at position i MINUS one j MINUS one of dp  
                ELSE IF the element at position j MINUS one of p EQUALS asterisk  
                    SET element at position i j of dp TO the value of element at position i j MINUS two of dp  
                    IF the element at position j MINUS two of p EQUALS the element at position i MINUS one of s OR the element at position j MINUS two of p EQUALS period  
                        SET element at position i j of dp TO the logical OR of the current value of element at position i j of dp AND the value of element at position i MINUS one j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the value of element at position the length of s the length of p of dp  
    END FUNCTION  
END CLASS
