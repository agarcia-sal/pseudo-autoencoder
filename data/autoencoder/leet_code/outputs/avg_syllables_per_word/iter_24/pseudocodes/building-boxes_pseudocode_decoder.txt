CLASS Solution  
    FUNCTION minimumBoxes(n)  
        FUNCTION triangular_pyramid(k)  
            RETURN the product of k MULTIPLIED BY the sum of k PLUS one MULTIPLIED BY the sum of k PLUS two DIVIDED BY six  
        END FUNCTION  

        SET left TO zero  
        SET right TO n  
        WHILE left LESS THAN right  
            SET mid TO the integer division result of the sum of left PLUS right PLUS one DIVIDED BY two  
            IF the result of triangular_pyramid applied to mid LESS THAN OR EQUAL TO n  
                SET left TO mid  
            ELSE  
                SET right TO the difference of mid MINUS one  
            END IF  
        END WHILE  

        SET boxes_used TO the result of triangular_pyramid applied to left  
        SET remaining TO the difference of n MINUS boxes_used  
        SET floor_boxes TO the integer division result of the product of left MULTIPLIED BY the sum of left PLUS one DIVIDED BY two  
        SET additional_boxes TO zero  
        WHILE remaining GREATER THAN zero  
            INCREMENT additional_boxes BY one  
            DECREMENT remaining BY additional_boxes  
        END WHILE  

        RETURN the sum of floor_boxes PLUS additional_boxes  
    END FUNCTION  
END CLASS
