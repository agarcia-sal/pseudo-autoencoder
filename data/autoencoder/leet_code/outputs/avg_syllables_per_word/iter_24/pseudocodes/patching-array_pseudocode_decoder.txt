CLASS Solution  
    FUNCTION minPatches(nums PARAMETER, n PARAMETER)  
        SET missing TO the number one  
        SET patches TO the number zero  
        SET i TO the number zero  
        WHILE missing LESS THAN OR EQUAL TO n  
            IF i LESS THAN the LENGTH OF nums AND the element at position i of nums LESS THAN OR EQUAL TO missing  
                SET missing TO missing PLUS the element at position i of nums  
                INCREMENT i BY the number one  
            ELSE  
                SET missing TO missing PLUS missing  
                INCREMENT patches BY the number one  
            END IF  
        END WHILE  
        RETURN patches  
    END FUNCTION  
END CLASS
