CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        
        CREATE dp AS a two dimensional list of ZERO with m PLUS one rows and n PLUS one columns  
        
        FOR j FROM one TO n  
            SET element at position j of row ZERO of dp TO element at position j MINUS one of row ZERO of dp PLUS the ASCII VALUE OF element at position j MINUS one of s2  
        END FOR  
        
        FOR i FROM one TO m  
            SET element at position zero of row i of dp TO element at position zero of row i MINUS one of dp PLUS the ASCII VALUE OF element at position i MINUS one of s1  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of s1 EQUALS element at position j MINUS one of s2  
                    SET element at position j of row i of dp TO element at position j MINUS one of row i MINUS one of dp  
                ELSE  
                    SET element at position j of row i of dp TO the MINIMUM OF the sum of element at position j of row i MINUS one of dp PLUS the ASCII VALUE OF element at position i MINUS one of s1 AND the sum of element at position j MINUS one of row i of dp PLUS the ASCII VALUE OF element at position j MINUS one of s2  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of row m of dp  
    END FUNCTION  
END CLASS
