CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with value as element at position one of values and left and right as no value  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element that is removed from the left side of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET node.left TO a new TreeNode with value as element at position i of values and left and right as no value  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET node.right TO a new TreeNode with value as element at position i of values and left and right as no value  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS no value AND q IS no value  
        RETURN True  
    ELSE IF p IS no value OR q IS no value  
        RETURN False  
    ELSE IF the value attribute of p IS NOT EQUALS the value attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with arguments as the left attribute of p and the left attribute of q AND the result of is_same_tree with arguments as the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS no value  
                RETURN zero  
            END IF  
            RETURN the value attribute of node PLUS the result of subtree_sum with argument as the left attribute of node PLUS the result of subtree_sum with argument as the right attribute of node  
        END FUNCTION  

        SET total_sum TO the result of subtree_sum with argument root  

        IF the remainder when total_sum is divided by two IS NOT EQUALS zero  
            RETURN False  
        END IF  

        SET target TO total_sum DIVIDED BY two  
        SET found TO a list containing one element False  

        FUNCTION can_split(node)  
            IF node IS no value  
                RETURN zero  
            END IF  
            SET current_sum TO the value attribute of node PLUS the result of can_split with argument as the left attribute of node PLUS the result of can_split with argument as the right attribute of node  
            IF current_sum EQUALS target AND node IS NOT root  
                SET the element at position one of found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  

        CALL can_split with argument root  
        RETURN the element at position one of found  
    END FUNCTION  
END CLASS
