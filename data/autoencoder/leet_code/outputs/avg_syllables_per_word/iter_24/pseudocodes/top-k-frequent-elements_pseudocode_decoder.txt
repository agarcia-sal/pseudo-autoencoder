CLASS Solution  
    FUNCTION topKFrequent(nums k)  
        SET frequency_map TO an empty mapping from element to integer starting at zero  
        FOR each num IN nums  
            INCREMENT the value in frequency_map for element num BY one  
        END FOR  
        SET max_frequency TO the maximum value among all values in frequency_map  
        SET buckets TO a list of empty lists with the total count of max_frequency PLUS one elements  
        FOR each element num and associated frequency freq IN frequency_map  
            APPEND element num TO the list at position freq of buckets  
        END FOR  
        SET result TO an empty list  
        FOR freq FROM max_frequency DOWNTO one  
            IF the length of the list at position freq of buckets GREATER THAN zero THEN  
                EXTEND result BY all elements in the list at position freq of buckets  
                IF the length of result GREATER THAN OR EQUAL TO k THEN  
                    RETURN the sublist of result from position one TO position k  
                END IF  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
