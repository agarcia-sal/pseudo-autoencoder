CLASS Solution  
    FUNCTION dailyTemperatures WITH parameter temperatures  
        SET n TO the LENGTH OF temperatures  
        SET answer TO a LIST OF zero WITH the LENGTH OF n  
        SET stack TO an EMPTY LIST  
        
        FOR each i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position i of temperatures IS GREATER THAN the element at position element at the LAST POSITION of stack of temperatures  
                SET prev_index TO the LAST ELEMENT REMOVED FROM stack  
                SET the element at position prev_index of answer TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
