CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct WITH PARAMETERS s AND k  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
        
        SET char_map TO an empty dictionary  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR each right FROM zero TO the LENGTH OF s MINUS one  
            SET the character at position right of s IN char_map TO right  
            
            IF the NUMBER OF KEYS IN char_map GREATER THAN k  
                SET leftmost_char TO the key in char_map WITH THE MINIMUM ASSOCIATED VALUE  
                SET left TO the value ASSOCIATED TO leftmost_char IN char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET max_length TO the GREATER OF max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
