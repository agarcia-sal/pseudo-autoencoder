CLASS Solution  
    FUNCTION getMoneyAmount(n)  
        FUNCTION dp(left, right)  
            IF left GREATER THAN OR EQUAL TO right  
                RETURN zero  
            END IF  
            SET min_cost TO infinite  
            FOR pivot FROM left TO right  
                SET left_cost TO dp with left and pivot MINUS one  
                SET right_cost TO dp with pivot PLUS one and right  
                SET cost TO pivot PLUS the greater value BETWEEN left_cost AND right_cost  
                IF cost LESS THAN min_cost  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN dp with one and n  
    END FUNCTION  
END CLASS
