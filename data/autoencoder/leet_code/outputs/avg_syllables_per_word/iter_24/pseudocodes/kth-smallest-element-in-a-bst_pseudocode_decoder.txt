CLASS TreeNode  
    FUNCTION __init__(self val SET TO zero left SET TO none right SET TO none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH element at position zero of values  
    SET i TO one  
    SET queue TO A new double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE element from front of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p dot left q dot left) AND is_same_tree(p dot right q dot right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(self root k)  
        SET stack TO an empty list  
        SET current TO root  
        SET count TO zero  
        WHILE stack IS NOT empty OR current IS NOT none  
            WHILE current IS NOT none  
                APPEND current TO stack  
                SET current TO current dot left  
            END WHILE  
            SET current TO REMOVE last element from stack  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN current dot val  
            END IF  
            SET current TO current dot right  
        END WHILE  
    END FUNCTION  
END CLASS
