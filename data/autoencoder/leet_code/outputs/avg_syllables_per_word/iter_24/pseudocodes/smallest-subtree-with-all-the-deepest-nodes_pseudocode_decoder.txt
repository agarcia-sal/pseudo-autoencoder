CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new instance of TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO new double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element removed from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node dot left TO new instance of TreeNode initialized with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node dot right TO new instance of TreeNode initialized with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF the val property of p NOT EQUALS the val property of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree called with the left property of p and the left property of q AND the result of is_same_tree called with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION subtreeWithAllDeepest(root)  
        FUNCTION dfs(node)  
            IF node EQUALS none  
                RETURN node AND zero  
            END IF  
            SET left_node AND left_depth TO the result of calling dfs with the left property of node  
            SET right_node AND right_depth TO the result of calling dfs with the right property of node  
            IF left_depth GREATER THAN right_depth  
                RETURN left_node AND left_depth PLUS one  
            ELSE IF right_depth GREATER THAN left_depth  
                RETURN right_node AND right_depth PLUS one  
            ELSE  
                RETURN node AND left_depth PLUS one  
            END IF  
        END FUNCTION  
        RETURN the first element of the result of dfs called with root  
    END FUNCTION  
END CLASS
