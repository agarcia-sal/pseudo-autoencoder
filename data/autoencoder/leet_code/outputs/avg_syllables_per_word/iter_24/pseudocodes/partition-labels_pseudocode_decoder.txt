CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO empty dictionary  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position index of s  
            SET entry in last_occurrence for character TO index  
        END FOR  
        
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO empty list  
        
        FOR i FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position i of s  
            IF the value of end LESS THAN the value at entry in last_occurrence for character THEN  
                SET end TO the value at entry in last_occurrence for character  
            END IF  
            
            IF i EQUALS end THEN  
                SET current_partition_size TO end MINUS start PLUS one  
                APPEND current_partition_size TO partition_sizes  
                SET start TO i PLUS one  
            END IF  
        END FOR  
        
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
