CLASS Solution  
    FUNCTION checkValidString WITH PARAMETER s  
        SET low TO zero  
        SET high TO zero  
        FOR each character IN s  
            IF character EQUALS the open parenthesis character  
                INCREMENT low BY one  
                INCREMENT high BY one  
            ELSE IF character EQUALS the close parenthesis character  
                DECREMENT low BY one  
                DECREMENT high BY one  
            ELSE IF character EQUALS the asterisk character  
                DECREMENT low BY one  
                INCREMENT high BY one  
            END IF  
            IF high LESS THAN zero  
                RETURN false  
            END IF  
            IF low LESS THAN zero  
                SET low TO zero  
            END IF  
        END FOR  
        IF low EQUALS zero  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
