CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO the list of substrings obtained by splitting num around the word plus  
            SET real TO the integer value of the first element of parts  
            SET imaginary TO the integer value of the substring of the second element of parts from the first position TO one position before the last position of second element of parts  
            RETURN real AND imaginary  
        END FUNCTION  
        
        SET real1 AND imag1 TO the result of calling parse_complex with argument num1  
        SET real2 AND imag2 TO the result of calling parse_complex with argument num2  
        
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        
        RETURN the string formed by concatenating the string form of real_part the word plus the string form of imaginary_part and the character i  
    END FUNCTION  
END CLASS
