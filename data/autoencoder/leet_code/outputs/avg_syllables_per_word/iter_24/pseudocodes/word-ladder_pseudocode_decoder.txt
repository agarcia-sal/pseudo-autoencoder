CLASS Solution  
    FUNCTION ladderLength(beginWord endWord wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN zero  
        END IF  
        SET word_dict TO a dictionary with default empty list for each key  
        SET word_length TO the LENGTH OF beginWord  
        FOR each word IN wordList  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_key TO the concatenation of substring from position zero TO position i MINUS one of word PLUS a placeholder representing any character PLUS substring from position i PLUS one TO the END OF word  
                APPEND word TO the list at key intermediate_key IN word_dict  
            END FOR  
        END FOR  
        SET queue TO a double ended queue containing a tuple of beginWord and one  
        SET visited TO a set containing beginWord  
        WHILE queue is not empty  
            REMOVE the first element FROM queue and ASSIGN it TO current_word and current_length respectively  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_word TO the concatenation of substring from position zero TO position i MINUS one of current_word PLUS a placeholder representing any character PLUS substring from position i PLUS one TO the END OF current_word  
                FOR each word IN the list at key intermediate_word IN word_dict  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND a tuple of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN zero  
    END FUNCTION  
END CLASS
