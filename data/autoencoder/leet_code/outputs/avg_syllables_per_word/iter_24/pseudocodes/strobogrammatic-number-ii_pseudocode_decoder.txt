CLASS Solution  
    FUNCTION findStrobogrammatic(n)  
        FUNCTION dfs(u)  
            IF u EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF u EQUALS one  
                RETURN list containing string zero one and eight  
            END IF  
            SET ans TO empty list  
            FOR each v IN dfs of u MINUS two  
                FOR each pair l and r IN string pairs eleven eighty-eight sixty-nine and ninety-six  
                    APPEND concatenation of l with v with r TO ans  
                END FOR  
                IF u NOT EQUALS n  
                    APPEND concatenation of string zero with v with string zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN dfs of n  
    END FUNCTION  
END CLASS
