CLASS Solution  
    FUNCTION cherryPickup(grid)  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        
        FUNCTION dp(r, c1, c2)  
            IF c1 LESS THAN zero OR c1 GREATER THAN OR EQUAL TO cols OR c2 LESS THAN zero OR c2 GREATER THAN OR EQUAL TO cols THEN  
                RETURN zero  
            END IF  
            
            IF c1 EQUALS c2 THEN  
                SET cherries TO element at position c1 of element at position r of grid  
            ELSE  
                SET cherries TO the sum of element at position c1 of element at position r of grid AND element at position c2 of element at position r of grid  
            END IF  
            
            IF r EQUALS rows MINUS one THEN  
                RETURN cherries  
            END IF  
            
            SET max_cherries TO zero  
            FOR each delta_c1 IN the list of values MINUS one zero one  
                FOR each delta_c2 IN the list of values MINUS one zero one  
                    SET candidate TO dp(r PLUS one, c1 PLUS delta_c1, c2 PLUS delta_c2)  
                    IF candidate GREATER THAN max_cherries THEN  
                        SET max_cherries TO candidate  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN cherries PLUS max_cherries  
        END FUNCTION  
        
        RETURN dp(zero, zero, cols MINUS one)  
    END FUNCTION  
END CLASS
