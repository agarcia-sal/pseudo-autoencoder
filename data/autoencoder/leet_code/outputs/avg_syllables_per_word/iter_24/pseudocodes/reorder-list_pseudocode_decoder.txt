CLASS ListNode  
    FUNCTION __init__ with parameters val AND next  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode with parameter element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two to the end  
        SET node TO a new ListNode with parameter val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN False  
    END IF  
    RETURN the condition that element val of p1 EQUALS element val of p2 AND the result of is_same_list with parameters element next of p1 AND element next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList with parameters self AND head  
        IF head EQUALS none OR element next of head EQUALS none  
            RETURN  
        END IF  
        SET slow TO head  
        SET fast TO head  
        WHILE fast NOT EQUALS none AND element next of fast NOT EQUALS none  
            SET slow TO element next of slow  
            SET fast TO element next of element next of fast  
        END WHILE  
        SET prev TO none  
        SET curr TO slow  
        WHILE curr NOT EQUALS none  
            SET next_temp TO element next of curr  
            SET element next of curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  
        SET first TO head  
        SET second TO prev  
        WHILE element next of second NOT EQUALS none  
            SET temp1 TO element next of first  
            SET temp2 TO element next of second  
            SET element next of first TO second  
            SET element next of second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
