CLASS Solution  
    FUNCTION minRefuelStops(target startFuel stations)  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND list consisting of target and zero TO stations  

        FOR each pair of position and fuel IN stations  
            WHILE the LENGTH OF max_heap is greater than zero AND current_fuel LESS THAN position  
                SET current_fuel TO current_fuel MINUS the NEGATION OF removing the smallest element FROM max_heap interpreted as maximal fuel  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel LESS THAN position  
                RETURN negative one  
            END IF  
            ADD the NEGATION of fuel TO max_heap using heap push operation  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
