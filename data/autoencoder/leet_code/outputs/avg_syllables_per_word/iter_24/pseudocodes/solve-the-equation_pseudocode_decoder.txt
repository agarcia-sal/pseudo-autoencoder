CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO the result of replacing plus signs in side with hash plus sign and then replacing minus signs in side with hash minus sign and then splitting side by hash signs  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token IS empty string  
                    CONTINUE to next token  
                END IF  
                IF token CONTAINS the character x  
                    IF token EQUALS the character x OR token EQUALS plus sign followed by the character x  
                        INCREMENT x_count BY one  
                    ELSE IF token EQUALS minus sign followed by the character x  
                        DECREMENT x_count BY one  
                    ELSE  
                        SET number_string TO token with the character x removed  
                        SET x_count TO x_count PLUS the integer value of number_string  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY the integer value of token  
                END IF  
            END FOR  
            RETURN x_count AND num_sum  
        END FUNCTION

        SET the sequence of left_side AND right_side TO the result of splitting equation by equal sign  
        SET left_x AND left_num TO the result of calling parse_side with left_side  
        SET right_x AND right_num TO the result of calling parse_side with right_side  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN the phrase Infinite solutions  
            ELSE  
                RETURN the phrase No solution  
            END IF  
        ELSE  
            SET solution_value TO total_num DIVIDED BY total_x using integer division  
            RETURN the string starting with character x EQUALS followed by the string representation of solution_value  
        END IF  
    END FUNCTION  
END CLASS
