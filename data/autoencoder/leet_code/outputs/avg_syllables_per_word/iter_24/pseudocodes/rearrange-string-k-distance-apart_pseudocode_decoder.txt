CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO frequency count of each character in s  
        
        SET max_heap TO empty list  
        FOR each character AND its count IN freq  
            APPEND pair of NEGATIVE count AND character TO max_heap  
        END FOR  
        TRANSFORM max_heap INTO a heap structure  
        
        SET wait_queue TO empty list  
        SET result TO empty list  
        
        WHILE max_heap is not empty  
            EXTRACT the smallest element from max_heap AND ASSIGN to count AND char  
            APPEND char TO result  
            
            APPEND pair of count PLUS one AND char TO wait_queue  
            
            IF the LENGTH OF wait_queue EQUALS k  
                REMOVE first element FROM wait_queue AND ASSIGN to count AND char  
                IF count LESS THAN zero  
                    ADD pair of count AND char back INTO max_heap  
                END IF  
            END IF  
        END WHILE  
        
        IF the LENGTH OF result NOT EQUALS the LENGTH OF s  
            RETURN empty string  
        END IF  
        
        RETURN concatenation of characters in result  
    END FUNCTION  
END CLASS
