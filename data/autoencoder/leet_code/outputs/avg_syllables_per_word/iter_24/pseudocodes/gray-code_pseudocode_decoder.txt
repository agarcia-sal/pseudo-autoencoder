CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN list containing zero  
        END IF  
        SET prev_gray TO CALL grayCode WITH argument n MINUS one  
        SET mask TO one MULTIPLIED BY two RAISED TO THE POWER OF n MINUS one  
        SET reversed_prev_gray TO the reverse ORDER OF prev_gray  
        SET current_gray TO concatenation OF prev_gray WITH an empty list  
        FOR each number IN reversed_prev_gray  
            APPEND result OF bitwise OR BETWEEN mask AND number TO current_gray  
        END FOR  
        RETURN current_gray  
    END FUNCTION  
END CLASS
