CLASS Solution  
    FUNCTION replaceWords(dictionary list_of_strings sentence string) RETURNS string  
        SORT dictionary BY the LENGTH OF each root  
        SET words TO the LIST OF words IN sentence  
        FUNCTION replace(word string) RETURNS string  
            FOR each root IN dictionary  
                IF the word STARTS WITH the root THEN  
                    RETURN root  
                END IF  
            END FOR  
            RETURN word  
        END FUNCTION  
        SET replaced_words TO an EMPTY list  
        FOR each word IN words  
            APPEND the RESULT OF replace(word) TO replaced_words  
        END FOR  
        SET result_sentence TO the CONCATENATION OF replaced_words WITH spaces BETWEEN  
        RETURN result_sentence  
    END FUNCTION  
END CLASS
