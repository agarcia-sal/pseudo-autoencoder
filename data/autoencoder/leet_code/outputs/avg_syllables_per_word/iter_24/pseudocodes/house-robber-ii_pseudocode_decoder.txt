CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF the LENGTH OF houses EQUALS zero  
                RETURN zero  
            END IF  
            IF the LENGTH OF houses EQUALS one  
                RETURN the element at position one of houses  
            END IF  
            
            SET dp TO a list of zeros with the LENGTH OF houses  
            SET the element at position one of dp TO the element at position one of houses  
            SET the element at position two of dp TO the GREATER VALUE BETWEEN the element at position one of houses AND the element at position two of houses  
            
            FOR i FROM three TO the LENGTH OF houses  
                SET the element at position i of dp TO the GREATER VALUE BETWEEN the element at the position one LESS THAN i of dp AND the SUM of the element at the position two LESS THAN i of dp PLUS the element at position i of houses  
            END FOR  
            
            RETURN the element at the LAST POSITION of dp  
        END FUNCTION  
        
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position one of nums  
        END IF  
        
        RETURN the GREATER VALUE BETWEEN rob_linear of the SUBLIST of nums from position one TO the POSITION one LESS THAN the LAST AND rob_linear of the SUBLIST of nums from the SECOND POSITION TO the LAST POSITION  
    END FUNCTION  
END CLASS
