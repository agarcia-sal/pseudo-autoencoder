CLASS Solution  
    FUNCTION networkDelayTime(times, n, k)  
        SET graph TO an empty dictionary  
        FOR each integer i FROM one TO n  
            SET element at position i of graph TO an empty list  
        END FOR  
        FOR each element consisting of u v w IN times  
            APPEND the pair of v and w TO the element at position u of graph  
        END FOR  
        SET dist TO an empty dictionary  
        FOR each integer i FROM one TO n  
            SET element at position i of dist TO infinity  
        END FOR  
        SET element at position k of dist TO zero  
        SET min_heap TO a list containing the pair of zero and k  
        WHILE the LENGTH OF min_heap is greater than zero  
            REMOVE and GET the smallest element from min_heap as current_dist and node  
            IF current_dist GREATER THAN element at position node of dist  
                CONTINUE to the next iteration  
            END IF  
            FOR each element consisting of neighbor and weight IN the element at position node of graph  
                SET distance TO current_dist PLUS weight  
                IF distance LESS THAN element at position neighbor of dist  
                    SET element at position neighbor of dist TO distance  
                    ADD the pair of distance and neighbor TO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        SET max_time TO the maximum value among all values in dist  
        IF max_time NOT EQUALS infinity  
            RETURN max_time  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
