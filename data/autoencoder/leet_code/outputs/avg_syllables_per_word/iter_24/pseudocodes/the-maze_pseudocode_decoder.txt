CLASS Solution  
    FUNCTION hasPath WITH maze AS list of list of integer AND start AS list of integer AND destination AS list of integer RETURNS boolean  
        SET directions TO list containing tuples representing zero and one zero and one zero and negative one and negative one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET visited TO an empty set  

        FUNCTION dfs WITH x AS integer AND y AS integer RETURNS boolean  
            IF tuple containing x AND y IS IN visited THEN  
                RETURN False  
            END IF  
            IF list containing x AND y EQUALS destination THEN  
                RETURN True  
            END IF  
            ADD tuple containing x AND y TO visited  

            FOR each tuple containing dx AND dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND element at position nx PLUS dx of maze AND element at position ny PLUS dy OF THAT EQUALS zero DO  
                    INCREMENT nx BY dx  
                    INCREMENT ny BY dy  
                END WHILE  
                IF dfs WITH nx AND ny RETURNS True THEN  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  

        RETURN dfs WITH element at position zero of start AND element at position one of start  
    END FUNCTION  
END CLASS
