CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one  
        END IF  
        SET n TO n PLUS twenty four DIVIDED BY twenty five  
        SET memo TO empty dictionary  
        FUNCTION dp(a, b)  
            IF tuple of a and b IS IN memo  
                RETURN the element at position tuple of a and b OF memo  
            END IF  
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN zero point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one  
            END IF  
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero  
            END IF  
            SET prob TO zero point two five MULTIPLIED BY the SUM of dp of a MINUS four and b PLUS dp of a MINUS three and b MINUS one PLUS dp of a MINUS two and b MINUS two PLUS dp of a MINUS one and b MINUS three  
            ASSIGN the element at position tuple of a and b OF memo TO prob  
            RETURN prob  
        END FUNCTION  
        RETURN dp of n and n  
    END FUNCTION  
END CLASS
