CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the difference BETWEEN the element at position zero of nums AND twenty four LESS THAN one times ten TO the power of negative six  
            END IF  
            
            FOR i FROM zero TO the LENGTH OF nums MINUS one  
                FOR j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO an EMPTY list  
                        FOR k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND the element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR op IN the collection consisting of true division FUNCTION MULTIPLY FUNCTION ADD FUNCTION SUBTRACT FUNCTION  
                            IF (op EQUALS ADD FUNCTION OR op EQUALS MULTIPLY FUNCTION) AND j GREATER THAN i  
                                CONTINUE to the next iteration  
                            END IF  
                            
                            IF op NOT EQUALS true division FUNCTION OR the element at position j of nums NOT EQUALS zero  
                                APPEND the result of applying op TO the element at position i of nums AND the element at position j of nums TO new_nums  
                                IF solve(new_nums)  
                                    RETURN True  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
