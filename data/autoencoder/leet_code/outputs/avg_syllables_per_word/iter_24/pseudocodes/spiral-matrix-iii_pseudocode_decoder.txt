CLASS Solution  
    FUNCTION spiralMatrixIII(rows  
                              cols  
                              rStart  
                              cStart)  
        SET directions TO list containing the pairs zero and one one and zero zero and negative one negative one and zero representing east south west north respectively  
        SET direction_index TO zero representing start facing east  
        SET steps TO zero representing number of steps to take in the current direction  
        SET increment TO zero representing when to increase the number of steps  
        SET visited TO empty list  
        SET x TO rStart  
        SET y TO cStart  
          
        WHILE the LENGTH OF visited IS LESS THAN rows MULTIPLIED BY cols  
            IF increment MODULO two EQUALS zero  
                INCREMENT steps BY one  
            END IF  
            FOR underscore FROM one TO steps  
                IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN rows AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN cols  
                    APPEND the list containing x and y TO visited  
                END IF  
                SET dx TO element at position direction_index of directions FIRST element  
                SET dy TO element at position direction_index of directions SECOND element  
                INCREMENT x BY dx  
                INCREMENT y BY dy  
            END FOR  
            SET direction_index TO the remainder of the SUM of direction_index AND one DIVIDED BY four  
            INCREMENT increment BY one  
        END WHILE  
          
        RETURN visited  
    END FUNCTION  
END CLASS
