CLASS Solution  
    FUNCTION rangeBitwiseAnd(left PARAMETER int, right PARAMETER int) RETURNS int  
        SET shift TO zero  
        WHILE left LESS THAN right  
            SET left TO left DIVIDED BY two  
            SET right TO right DIVIDED BY two  
            INCREMENT shift BY one  
        END WHILE  
        SET result TO left MULTIPLIED BY two RAISED TO THE POWER OF shift  
        RETURN result  
    END FUNCTION  
END CLASS
