CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO empty list  
        FOR each character IN expression  
            IF character EQUALS one of the characters representing true OR false OR negation OR AND OR OR operators  
                APPEND character TO stk  
            ELSE IF character EQUALS the character representing closing parenthesis  
                SET count_true TO zero  
                SET count_false TO zero  
                WHILE the element at the last position of stk EQUALS true character OR false character  
                    INCREMENT count_true BY one IF the element at the last position of stk EQUALS true character  
                    INCREMENT count_false BY one IF the element at the last position of stk EQUALS false character  
                    REMOVE the last element FROM stk  
                END WHILE  
                SET operator TO the element removed from the last position of stk  
                IF operator EQUALS negation character  
                    SET character TO true character IF count_false GREATER THAN zero EQUALS false character  
                ELSE IF operator EQUALS AND character  
                    SET character TO false character IF count_false GREATER THAN zero EQUALS true character  
                ELSE IF operator EQUALS OR character  
                    SET character TO true character IF count_true GREATER THAN zero EQUALS false character  
                END IF  
                APPEND character TO stk  
            END IF  
        END FOR  
        RETURN element at position zero of stk EQUALS true character  
    END FUNCTION  
END CLASS
