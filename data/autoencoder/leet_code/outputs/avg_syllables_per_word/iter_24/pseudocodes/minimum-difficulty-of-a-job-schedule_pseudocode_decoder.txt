CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty jobDifficulty LIST OF INTEGERS d INTEGER) RETURNS INTEGER  
        SET n TO the LENGTH OF jobDifficulty  
        IF n LESS THAN d THEN  
            RETURN the NEGATIVE ONE INTEGER  
        END IF  
        
        FUNCTION dp(i INTEGER d INTEGER) RETURNS INTEGER  
            IF d EQUALS the INTEGER ONE THEN  
                RETURN the MAXIMUM VALUE among the ELEMENTS starting at POSITION i TO the END IN jobDifficulty  
            END IF  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR each j INTEGER FROM i TO n MINUS d  
                SET current_max TO the MAXIMUM VALUE between current_max AND the ELEMENT AT POSITION j IN jobDifficulty  
                SET candidate TO current_max PLUS dp(j PLUS one INTEGER d MINUS one INTEGER)  
                SET min_difficulty TO the MINIMUM VALUE between min_difficulty AND candidate  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN dp(the INTEGER zero d)  
    END FUNCTION  
END CLASS
