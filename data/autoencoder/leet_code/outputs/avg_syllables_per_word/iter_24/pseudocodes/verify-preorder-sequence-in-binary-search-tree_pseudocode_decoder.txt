CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO empty list  
        SET lower_bound TO the smallest possible number  
        
        FOR each value IN preorder  
            IF value LESS THAN lower_bound  
                RETURN False  
            END IF  
            
            WHILE the LENGTH OF stack IS GREATER THAN zero AND value GREATER THAN element at the last position of stack  
                SET lower_bound TO the element REMOVED from the last position of stack  
            END WHILE  
            
            APPEND value TO stack  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
