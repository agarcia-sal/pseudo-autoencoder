CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of infinity repeated fifteen times  
        FOR each pair of first value and rate IN tires  
            SET total_time TO zero  
            FOR index FROM zero TO fourteen  
                SET lap_time TO first value MULTIPLIED BY rate RAISED TO the power of index  
                IF lap_time IS GREATER THAN changeTime PLUS first value  
                    BREAK the loop  
                END IF  
                INCREMENT total_time BY lap_time  
                IF min_times at position index IS GREATER THAN total_time  
                    SET min_times at position index TO total_time  
                END IF  
            END FOR  
        END FOR  
        
        SET dp TO a list of infinity repeated numLaps PLUS one times  
        SET dp at position zero TO zero  
        
        FOR i FROM one TO numLaps  
            FOR j FROM zero TO the smaller between i MINUS one AND fourteen  
                SET candidate_time TO dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
                IF dp at position i IS GREATER THAN candidate_time  
                    SET dp at position i TO candidate_time  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
