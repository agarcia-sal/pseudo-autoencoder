CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF element at position zero of cost  
        SET dp TO a list of lists of infinity values with the outer list length equal to size1 plus one and the inner list length equal to two RAISED TO THE POWER OF size2  
        SET element at position zero of dp at position zero TO zero  
        
        FOR variable i FROM zero TO size1 MINUS one  
            FOR variable mask FROM zero TO two RAISED TO THE POWER OF size2 MINUS one  
                FOR variable j FROM zero TO size2 MINUS one  
                    SET index1 TO i PLUS one  
                    SET index2 TO bitwise OR of mask AND two RAISED TO THE POWER OF j  
                    SET current_value TO element at position index1 of dp at position index2  
                    SET new_value TO minimum of current_value AND the sum of element at position i of dp at position mask AND element at position i of cost at position j  
                    SET element at position index1 of dp at position index2 TO new_value  
                END FOR  
                
                FOR variable j FROM zero TO size2 MINUS one  
                    IF the bitwise AND of mask AND two RAISED TO THE POWER OF j NOT EQUAL TO zero  
                        SET index1 TO i PLUS one  
                        SET index2 TO mask  
                        SET index3 TO bitwise XOR of mask AND two RAISED TO THE POWER OF j  
                        SET current_value TO element at position index1 of dp at position index2  
                        SET new_value TO minimum of current_value AND the sum of element at position index1 of dp at position index3 AND element at position i of cost at position j  
                        SET element at position index1 of dp at position index2 TO new_value  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET final_mask TO two RAISED TO THE POWER OF size2 MINUS one  
        RETURN element at position size1 of dp at position final_mask  
    END FUNCTION  
END CLASS
