CLASS TreeNode  
    FUNCTION __init__(self WITH val AND left AND right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH values  
    IF values EQUALS empty  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode WITH element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        REMOVE element at the front of queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT nothing  
            SET node.left TO a new TreeNode WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT nothing  
            SET node.right TO a new TreeNode WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH p AND q  
    IF p IS nothing AND q IS nothing  
        RETURN true  
    ELSE IF p IS nothing OR q IS nothing  
        RETURN false  
    ELSE IF the val attribute of p IS NOT EQUALS the val attribute of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH the left attribute of p AND the left attribute of q AND is_same_tree WITH the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION lowestCommonAncestor WITH self AND root AND p AND q  
        IF root IS nothing OR root EQUALS p OR root EQUALS q  
            RETURN root  
        END IF  
        SET left TO the result of calling lowestCommonAncestor ON self WITH the left attribute of root AND p AND q  
        SET right TO the result of calling lowestCommonAncestor ON self WITH the right attribute of root AND p AND q  
        IF left IS NOT nothing AND right IS NOT nothing  
            RETURN root  
        END IF  
        IF left IS NOT nothing  
            RETURN left  
        ELSE  
            RETURN right  
        END IF  
    END FUNCTION  
END CLASS
