CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO ListNode element at position one of values  
    SET p TO head  
    FOR each val IN elements from position two TO the end of values  
        SET node TO ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN p1.val EQUALS p2.val AND is_same_list with p1.next and p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList(self, head)  
        IF head EQUALS None OR head.next EQUALS None  
            RETURN head  
        END IF  

        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO head.next  
            WHILE fast NOT EQUALS None AND fast.next NOT EQUALS None  
                SET slow TO slow.next  
                SET fast TO fast.next.next  
            END WHILE  
            SET mid TO slow.next  
            SET slow.next TO None  
            RETURN head AND mid  
        END FUNCTION  

        FUNCTION merge(l1, l2)  
            SET dummy TO ListNode without parameters  
            SET current TO dummy  
            WHILE l1 NOT EQUALS None AND l2 NOT EQUALS None  
                IF l1.val LESS THAN l2.val  
                    SET current.next TO l1  
                    SET l1 TO l1.next  
                ELSE  
                    SET current.next TO l2  
                    SET l2 TO l2.next  
                END IF  
                SET current TO current.next  
            END WHILE  
            IF l1 NOT EQUALS None  
                SET current.next TO l1  
            ELSE  
                SET current.next TO l2  
            END IF  
            RETURN dummy.next  
        END FUNCTION  

        SET left AND right TO split with head  
        SET left TO self.sortList with left  
        SET right TO self.sortList with right  
        RETURN merge with left and right  
    END FUNCTION  
END CLASS
