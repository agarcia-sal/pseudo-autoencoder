CLASS Solution  
    FUNCTION superEggDrop(k, n)  
        SET dp TO a list of zero repeated the sum of k AND one times  
        SET moves TO zero  
        WHILE the element at position k of dp LESS THAN n holds  
            INCREMENT moves BY one  
            FOR eggs FROM k DOWN TO one  
                SET the element at position eggs of dp TO one PLUS the element at position eggs MINUS one of dp PLUS the element at position eggs of dp  
            END FOR  
        END WHILE  
        RETURN moves  
    END FUNCTION  
END CLASS
