CLASS Solution  
    FUNCTION numberOfCombinations(num)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF num  
        SET memo TO an empty dictionary  
        
        FUNCTION dp(index, prev_len)  
            IF index EQUALS n  
                RETURN one  
            END IF  
            IF the PAIR of index AND prev_len EXISTS IN memo  
                RETURN the VALUE OF memo AT the PAIR of index AND prev_len  
            END IF  
            
            SET count TO zero  
            FOR length FROM one TO n MINUS index  
                IF the CHARACTER at position index OF num EQUALS the character zero  
                    BREAK the loop  
                END IF  
                
                SET current_num TO the SUBSTRING from position index TO position index PLUS length MINUS one OF num  
                
                IF prev_len EQUALS zero OR current_num IS GREATER THAN OR EQUAL TO the SUBSTRING from position index MINUS prev_len TO position index MINUS one OF num  
                    SET count TO the REMAINDER OF the SUM of count AND the RESULT OF dp APPLIED TO index PLUS length AND length DIVIDED BY MOD  
                END IF  
            END FOR  
            
            ASSIGN the VALUE of count TO memo AT the PAIR of index AND prev_len  
            RETURN count  
        END FUNCTION  
        
        RETURN dp APPLIED TO zero AND zero  
    END FUNCTION  
END CLASS
