CLASS Solution  
    FUNCTION canDivideIntoSubsequences(nums k)  
        SET freq TO the result of counting the frequency of each element in nums  
        SET max_freq TO the greatest value among the values of freq  
        IF the LENGTH OF nums DIVIDED BY k GREATER THAN OR EQUAL TO max_freq  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
