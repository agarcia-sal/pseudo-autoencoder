CLASS Solution  
    FUNCTION numRabbits(answers)  
        SET answer_counts TO the result of counting the occurrences of each element in answers  
        SET total_rabbits TO zero  
        FOR each answer AND count IN the collection of items of answer_counts  
            SET groups TO the result of the integer division of the sum of count PLUS answer BY the sum of answer PLUS one  
            INCREMENT total_rabbits BY the product of groups AND the sum of answer PLUS one  
        END FOR  
        RETURN total_rabbits  
    END FUNCTION  
END CLASS
