CLASS Solution  
    FUNCTION constructArray(n, k)  
        SET result TO an empty list  
        FOR each number FROM one TO the value of n MINUS k MINUS one  
            APPEND the current number TO result  
        END FOR  
        SET left TO the value of n MINUS k  
        SET right TO the value of n  
        WHILE the value of left IS LESS THAN OR EQUAL TO the value of right  
            IF the value of left EQUALS the value of right  
                APPEND the value of left TO result  
            ELSE  
                APPEND the value of left TO result  
                APPEND the value of right TO result  
            END IF  
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
