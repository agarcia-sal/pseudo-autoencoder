CLASS Solution  
    FUNCTION addOperators(num parameter, target parameter) RETURNS list of strings  
        FUNCTION dfs(index parameter, path parameter, value parameter, prev parameter)  
            IF index EQUALS the LENGTH OF num parameter  
                IF value parameter EQUALS target parameter  
                    APPEND path parameter TO result  
                END IF  
                RETURN  
            END IF  
            FOR i variable FROM index parameter PLUS one TO the LENGTH OF num parameter PLUS one  
                SET operand TO the substring from position index parameter TO position i variable of num parameter  
                IF the LENGTH OF operand IS GREATER THAN one AND the element at position one of operand EQUALS zero as character  
                    CONTINUE to next iteration  
                END IF  
                IF index parameter EQUALS zero  
                    CALL dfs WITH arguments i variable, operand, the INTEGER VALUE OF operand, the INTEGER VALUE OF operand  
                ELSE  
                    CALL dfs WITH arguments i variable, the concatenation of path parameter PLUS the character plus PLUS PLUS operand, value parameter PLUS the INTEGER VALUE OF operand, the INTEGER VALUE OF operand  
                    CALL dfs WITH arguments i variable, the concatenation of path parameter PLUS the character minus PLUS PLUS operand, value parameter MINUS the INTEGER VALUE OF operand, the NEGATION OF the INTEGER VALUE OF operand  
                    CALL dfs WITH arguments i variable, the concatenation of path parameter PLUS the character multiplied by PLUS PLUS operand, value parameter MINUS prev parameter PLUS prev parameter MULTIPLIED BY the INTEGER VALUE OF operand, prev parameter MULTIPLIED BY the INTEGER VALUE OF operand  
                END IF  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        CALL dfs WITH arguments zero, an empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
