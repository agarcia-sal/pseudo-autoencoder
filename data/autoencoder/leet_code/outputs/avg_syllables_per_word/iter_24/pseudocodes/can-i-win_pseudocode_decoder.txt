CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the product of maxChoosableInteger AND the sum of maxChoosableInteger PLUS one DIVIDED BY two LESS THAN desiredTotal  
            RETURN False  
        END IF  
        IF desiredTotal LESS THAN OR EQUAL TO zero  
            RETURN True  
        END IF  
        SET memo TO empty dictionary  
        FUNCTION can_win(current_total, used_numbers)  
            IF the tuple of current_total AND used_numbers EXISTS in memo  
                RETURN the value in memo for the tuple of current_total AND used_numbers  
            END IF  
            FOR number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers AND the number one shifted left BY number EQUALS zero  
                    IF the sum of current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR the NEGATION of can_win of the sum of current_total PLUS number AND the bitwise OR of used_numbers AND the number one shifted left BY number  
                        SET the value in memo for the tuple of current_total AND used_numbers TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            SET the value in memo for the tuple of current_total AND used_numbers TO False  
            RETURN False  
        END FUNCTION  
        RETURN can_win of zero AND zero  
    END FUNCTION  
END CLASS
