CLASS Solution  
    FUNCTION removeDuplicateLetters(s)  
        SET last occurrence to an empty dictionary mapping each character to its last position in s  
        SET index to zero  
        FOR each character IN s  
            SET last occurrence at character TO index  
            INCREMENT index BY one  
        END FOR  
        
        SET stack to an empty list  
        SET in stack to an empty set  
        
        SET index to zero  
        FOR each character IN s  
            IF character is an element of in stack THEN  
                INCREMENT index BY one  
                CONTINUE to next iteration  
            END IF  
            
            WHILE the LENGTH OF stack is greater than zero AND character is LESS THAN element at the last position of stack AND index is LESS THAN last occurrence at element at the last position of stack  
                SET removed character TO the element removed from the end of stack  
                REMOVE removed character FROM in stack  
            END WHILE  
            
            APPEND character TO stack  
            ADD character TO in stack  
            INCREMENT index BY one  
        END FOR  
        
        SET result string TO the concatenation of all elements in stack in their current order  
        RETURN result string  
    END FUNCTION  
END CLASS
