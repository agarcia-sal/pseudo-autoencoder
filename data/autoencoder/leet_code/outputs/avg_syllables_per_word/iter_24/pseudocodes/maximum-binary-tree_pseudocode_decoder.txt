CLASS TreeNode  
    FUNCTION __init__ with parameters self val default to zero left default to none right default to none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS empty list  
        RETURN none  
    END IF  
    SET root TO new TreeNode with parameter element at position one of values  
    SET i TO one  
    SET queue TO new double ended queue  
    APPEND root TO queue  
    WHILE queue length GREATER THAN zero  
        SET node TO element removed from the left of queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS none  
            SET node dot left TO new TreeNode with parameter element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS none  
            SET node dot right TO new TreeNode with parameter element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN AND of is_same_tree with parameters p dot left q dot left AND is_same_tree with parameters p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION constructMaximumBinaryTree with parameters self nums  
        IF nums EQUALS empty list  
            RETURN none  
        END IF  
        SET max_val TO the maximum value in nums  
        SET max_index TO the position of max_val in nums  
        SET root TO new TreeNode with parameter max_val  
        SET root dot left TO the result of calling self dot constructMaximumBinaryTree with parameter the elements of nums from position one TO position max_index MINUS one  
        SET root dot right TO the result of calling self dot constructMaximumBinaryTree with parameter the elements of nums from position max_index PLUS one TO the last position of nums  
        RETURN root  
    END FUNCTION  
END CLASS
