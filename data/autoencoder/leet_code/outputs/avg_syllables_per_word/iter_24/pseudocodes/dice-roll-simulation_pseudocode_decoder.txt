CLASS Solution  
    FUNCTION dieSimulator(n, rollMax)  
        SET MOD TO one billion PLUS seven  
        
        SET dp TO a three dimensional list initialized with zero having dimensions of n PLUS one by six by the maximum value in rollMax PLUS one  
        
        FOR j FROM zero TO five  
            SET element at position one at position j at position one of dp TO one  
        END FOR  
        
        FOR i FROM two TO n  
            FOR j FROM zero TO five  
                FOR k FROM one TO element at position j of rollMax  
                    IF k GREATER THAN one  
                        SET element at position i at position j at position k of dp TO element at position i MINUS one at position j at position k MINUS one of dp  
                    ELSE  
                        SET total_sum TO zero  
                        FOR x FROM zero TO five  
                            IF x NOT EQUALS j  
                                FOR y FROM one TO element at position x of rollMax  
                                    INCREMENT total_sum BY element at position i MINUS one at position x at position y of dp  
                                END FOR  
                            END IF  
                        END FOR  
                        SET element at position i at position j at position k of dp TO total_sum MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        SET result_sum TO zero  
        FOR j FROM zero TO five  
            FOR k FROM one TO element at position j of rollMax  
                INCREMENT result_sum BY element at position n at position j at position k of dp  
            END FOR  
        END FOR  
        
        RETURN result_sum MODULO MOD  
    END FUNCTION  
END CLASS
