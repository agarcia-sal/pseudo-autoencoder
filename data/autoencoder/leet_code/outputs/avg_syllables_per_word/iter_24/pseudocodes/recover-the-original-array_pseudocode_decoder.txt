CLASS Solution  
    FUNCTION recoverArray(nums)  
        SORT nums IN ascending order  
        SET n TO the LENGTH OF nums  
        SET first TO element at position zero of nums  

        FOR i FROM one TO n MINUS one  
            SET k TO element at position i of nums MINUS first DIVIDED BY two  
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at position i of nums THEN  
                CONTINUE  
            END IF  

            SET count TO a new mapping of counts of elements in nums  
            SET arr TO an empty list  

            FOR each x IN nums  
                IF the count of x EQUALS zero THEN  
                    CONTINUE  
                END IF  
                IF the count of the element x PLUS two MULTIPLIED BY k EQUALS zero THEN  
                    BREAK  
                END IF  
                APPEND the element x PLUS k TO arr  
                DECREMENT the count of x BY one  
                DECREMENT the count of the element x PLUS two MULTIPLIED BY k BY one  
            END FOR  

            IF the LENGTH OF arr EQUALS n DIVIDED BY two THEN  
                RETURN arr  
            END IF  
        END FOR  

        RETURN an empty list  
    END FUNCTION  
END CLASS
