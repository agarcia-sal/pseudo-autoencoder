CLASS Solution  
    FUNCTION reachingPoints(sx sx_parameter sy sy_parameter tx tx_parameter ty ty_parameter) RETURNS boolean  
        WHILE tx_parameter GREATER THAN OR EQUAL TO sx_parameter AND ty_parameter GREATER THAN OR EQUAL TO sy_parameter  
            IF tx_parameter EQUALS sx_parameter AND ty_parameter EQUALS sy_parameter  
                RETURN True  
            END IF  
            IF tx_parameter GREATER THAN ty_parameter  
                IF ty_parameter GREATER THAN sy_parameter  
                    SET tx_parameter TO the remainder of tx_parameter DIVIDED BY ty_parameter  
                ELSE  
                    RETURN the remainder of the difference of tx_parameter MINUS sx_parameter DIVIDED BY ty_parameter EQUALS zero  
                END IF  
            ELSE  
                IF tx_parameter GREATER THAN sx_parameter  
                    SET ty_parameter TO the remainder of ty_parameter DIVIDED BY tx_parameter  
                ELSE  
                    RETURN the remainder of the difference of ty_parameter MINUS sy_parameter DIVIDED BY tx_parameter EQUALS zero  
                END IF  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
