CLASS Solution  
    FUNCTION checkRecord(n)  
        SET MODULO TO one billion PLUS seven  
        INITIALIZE a three dimensional collection dp with dimensions n PLUS one BY two BY three FILLED WITH zero  
        SET element at position zero AND zero AND zero of dp TO one  
        FOR index i FROM one TO n  
            FOR index j FROM zero TO one  
                FOR index k FROM zero TO two  
                    SET element at position i AND j AND zero of dp TO element at position i AND j AND zero of dp PLUS element at position i MINUS one AND j AND k of dp MODULO MODULO  
                    IF j EQUALS zero  
                        SET element at position i AND one AND zero of dp TO element at position i AND one AND zero of dp PLUS element at position i MINUS one AND zero AND k of dp MODULO MODULO  
                    END IF  
                    IF k LESS THAN two  
                        SET element at position i AND j AND k PLUS one of dp TO element at position i AND j AND k PLUS one of dp PLUS element at position i MINUS one AND j AND k of dp MODULO MODULO  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        SET result TO zero  
        FOR index j FROM zero TO one  
            FOR index k FROM zero TO two  
                SET result TO result PLUS element at position n AND j AND k of dp MODULO MODULO  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
