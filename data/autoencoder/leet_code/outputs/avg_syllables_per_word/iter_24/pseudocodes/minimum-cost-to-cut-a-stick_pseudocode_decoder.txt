CLASS Solution  
    FUNCTION minCost(n, cuts)  
        SET cuts TO the result of sorting the concatenation of cuts WITH zero AND n  
        
        FUNCTION dp(left, right)  
            IF the difference between right AND left IS LESS THAN OR EQUAL TO one  
                RETURN zero  
            END IF  
            
            SET min_cost TO positive infinity  
            FOR i FROM the sum of left AND one TO the difference between right AND one  
                SET cost TO the difference between the element at position right of cuts AND the element at position left of cuts PLUS the result of calling dp WITH left AND i PLUS the result of calling dp WITH i AND right  
                IF cost IS LESS THAN min_cost  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        RETURN the result of calling dp WITH zero AND the difference between the length of cuts AND one  
    END FUNCTION  
END CLASS
