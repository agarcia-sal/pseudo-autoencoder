CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            SET expected_number_string TO the string representation of first PLUS second  
            SET expected_length TO the LENGTH OF expected_number_string  
            IF start PLUS expected_length GREATER THAN the LENGTH OF num OR the substring of num FROM position start TO position start PLUS expected_length NOT EQUALS expected_number_string  
                RETURN False  
            END IF  
            RETURN is_valid_sequence(start PLUS expected_length, second, the integer conversion of expected_number_string)  
        END FUNCTION  
        SET n TO the LENGTH OF num  
        FOR i FROM one TO n DIVIDED BY two PLUS one  
            FOR j FROM i PLUS one TO n MINUS one PLUS one  
                SET first TO the substring of num FROM position zero TO position i  
                SET second TO the substring of num FROM position i TO position j  
                IF the LENGTH OF first GREATER THAN one AND the element at position zero of first EQUALS character zero OR the LENGTH OF second GREATER THAN one AND the element at position zero of second EQUALS character zero  
                    CONTINUE  
                END IF  
                IF is_valid_sequence(j, the integer conversion of first, the integer conversion of second)  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
