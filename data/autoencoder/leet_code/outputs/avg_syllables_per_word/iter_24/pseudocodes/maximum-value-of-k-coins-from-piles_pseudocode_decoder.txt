CLASS Solution  
    FUNCTION maxValueOfCoins(piles parameter, k parameter)  
        SET n TO the LENGTH OF piles  
        SET dp TO a two dimensional list with n PLUS one rows and k PLUS one columns filled with zero  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one of piles  
                SET current_value TO zero  
                
                FOR l FROM zero TO THE MINIMUM OF j AND the LENGTH OF current_pile  
                    IF l GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one of current_pile  
                    END IF  
                    SET dp element at position i and j TO the GREATER OF dp element at position i and j AND dp element at position i MINUS one and j MINUS l PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp element at position n and k  
    END FUNCTION  
END CLASS
