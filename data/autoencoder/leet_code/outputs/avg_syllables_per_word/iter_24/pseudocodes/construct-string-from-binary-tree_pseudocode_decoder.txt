CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    ASSIGN root TO new TreeNode with value as element at position one of values  
    SET i TO one  
    ASSIGN queue TO new deque collection  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS greater than zero  
        ASSIGN node TO the element obtained by removing from the left end of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO new TreeNode with value as element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO new TreeNode with value as element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value property of p IS NOT EQUALS to the value property of q  
        RETURN False  
    ELSE  
        RETURN the logical conjunction of calling is_same_tree with the left property of p and the left property of q AND calling is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str(self, root)  
        FUNCTION helper(node)  
            IF node IS None  
                RETURN empty string  
            END IF  
            SET result TO the string representation of the value property of node  
            IF the left property of node EXISTS OR the left property of node IS None AND the right property of node EXISTS  
                SET result TO result PLUS opening parenthesis PLUS call to helper with the left property of node PLUS closing parenthesis  
            END IF  
            IF the right property of node EXISTS  
                SET result TO result PLUS opening parenthesis PLUS call to helper with the right property of node PLUS closing parenthesis  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN call to helper with root  
    END FUNCTION  
END CLASS
