CLASS Solution  
    FUNCTION minCostToSupplyWater(n, wells, pipes)  
        FOR index FROM zero TO n MINUS one  
            APPEND a list containing zero PLUS zero a virtual node zero, index PLUS one representing the house number, and element at position index of wells to pipes  
        END FOR  
        SORT pipes BY the value at position three of each element  
        SET parent TO a list of integers from zero TO n inclusive  
        
        FUNCTION find(x)  
            IF element at position x of parent NOT EQUALS x  
                SET element at position x of parent TO the result of find applied to element at position x of parent  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        
        FUNCTION union(x, y)  
            SET rootX TO the result of find applied to x  
            SET rootY TO the result of find applied to y  
            IF rootX NOT EQUALS rootY  
                SET element at position rootX of parent TO rootY  
                RETURN True  
            END IF  
            RETURN False  
        END FUNCTION  
        
        SET min_cost TO zero  
        FOR each element of pipes as house1, house2, cost  
            IF union applied to house1 and house2  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
