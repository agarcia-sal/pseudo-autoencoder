CLASS Solution  
    FUNCTION maximumANDSum(nums parameter LIST OF integer, numSlots parameter integer)  
        FUNCTION dp(index parameter integer, slots parameter TUPLE OF integers)  
            IF index EQUALS the LENGTH OF nums THEN  
                RETURN zero  
            END IF  
            SET max_and_sum TO zero  
            SET slots_list TO a LIST converted FROM slots  
            FOR i FROM zero TO numSlots MINUS one  
                IF element at position i of slots_list LESS THAN two THEN  
                    INCREMENT element at position i of slots_list BY one  
                    SET and_sum TO the result of the bitwise AND between element at position index of nums AND i PLUS one PLUS the RESULT OF FUNCTION dp WITH PARAMETERS index PLUS one AND a TUPLE converted FROM slots_list  
                    IF and_sum GREATER THAN max_and_sum THEN  
                        SET max_and_sum TO and_sum  
                    END IF  
                    DECREMENT element at position i of slots_list BY one  
                END IF  
            END FOR  
            RETURN max_and_sum  
        END FUNCTION  
        SET initial_slots TO a TUPLE of zeros WITH LENGTH numSlots  
        RETURN the RESULT OF FUNCTION dp WITH PARAMETERS zero AND initial_slots  
    END FUNCTION  
END CLASS
