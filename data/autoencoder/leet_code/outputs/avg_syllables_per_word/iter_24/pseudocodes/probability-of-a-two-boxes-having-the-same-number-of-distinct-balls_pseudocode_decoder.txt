CLASS Solution  
    FUNCTION getProbability WITH PARAMETER balls  
        FUNCTION dfs WITH PARAMETERS i j diff  
            IF i GREATER THAN OR EQUAL TO k  
                IF j EQUALS zero AND diff EQUALS zero  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF j LESS THAN zero  
                RETURN zero  
            END IF  
            SET ans TO zero  
            FOR each x FROM zero TO element at position i of balls INCLUSIVE  
                IF x EQUALS element at position i of balls  
                    SET y TO one  
                ELSE IF x EQUALS zero  
                    SET y TO negative one  
                ELSE  
                    SET y TO zero  
                END IF  
                INCREMENT ans BY dfs called with parameters i PLUS one j MINUS x diff PLUS y MULTIPLIED BY combination of element at position i of balls AND x  
            END FOR  
            RETURN ans  
        END FUNCTION  

        SET n TO sum of all elements in balls DIVIDED BY two  
        SET k TO length of balls  
        RETURN dfs called with parameters zero n zero DIVIDED BY combination of n MULTIPLIED BY two AND n  
    END FUNCTION  
END CLASS
