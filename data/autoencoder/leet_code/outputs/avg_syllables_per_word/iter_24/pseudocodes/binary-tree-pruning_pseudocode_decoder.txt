CLASS TreeNode  
    FUNCTION __init__ with parameters val left right  
        SET val attribute TO val  
        SET left attribute TO left  
        SET right attribute TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    ASSIGN reference TO object WITH CLASS TreeNode AND parameter value equal to element at position one of values  
    SET i TO one  
    CREATE an empty double ended queue named queue  
    APPEND reference TO queue  
    WHILE queue is not empty  
        REMOVE element from the left side of queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET left attribute of node TO new object WITH CLASS TreeNode AND parameter value equal to element at position i of values  
            APPEND left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT no value  
            SET right attribute of node TO new object WITH CLASS TreeNode AND parameter value equal to element at position i of values  
            APPEND right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN reference  
END FUNCTION

FUNCTION is_same_tree with parameters p q  
    IF p IS no value AND q IS no value  
        RETURN true  
    ELSE IF p IS no value OR q IS no value  
        RETURN false  
    ELSE IF val attribute of p NOT EQUALS val attribute of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree called with parameters left attribute of p left attribute of q AND is_same_tree called with parameters right attribute of p right attribute of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION pruneTree with parameter root  
        IF root IS no value  
            RETURN no value  
        END IF  
        SET left attribute of root TO result of calling pruneTree with parameter left attribute of root  
        SET right attribute of root TO result of calling pruneTree with parameter right attribute of root  
        IF val attribute of root EQUALS zero AND left attribute of root IS no value AND right attribute of root IS no value  
            RETURN no value  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
