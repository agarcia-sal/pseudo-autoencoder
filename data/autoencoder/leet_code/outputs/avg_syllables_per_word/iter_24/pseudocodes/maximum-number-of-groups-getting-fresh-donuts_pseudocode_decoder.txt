CLASS Solution  
    FUNCTION maxHappyGroups(batchSize groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
        
        SET remainder_count TO a list of zero repeated batchSize times  
        FOR each group IN groups  
            INCREMENT the element at position the remainder of group DIVIDED BY batchSize of remainder_count BY one  
        END FOR  
        
        FUNCTION dp(remainder remainder_count)  
            SET remainder_count TO a list constructed from remainder_count  
            SET max_happy TO zero  
            
            FOR i FROM zero TO batchSize MINUS one  
                IF the element at position i of remainder_count EQUALS zero  
                    CONTINUE to the next iteration  
                END IF  
                
                DECREMENT the element at position i of remainder_count BY one  
                SET new_remainder TO the remainder of remainder MINUS i DIVIDED BY batchSize  
                
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS dp(new_remainder unpacked remainder_count)  
                ELSE  
                    SET happy_groups TO dp(new_remainder unpacked remainder_count)  
                END IF  
                
                IF happy_groups GREATER THAN max_happy  
                    SET max_happy TO happy_groups  
                END IF  
                
                INCREMENT the element at position i of remainder_count BY one  
            END FOR  
            
            RETURN max_happy  
        END FUNCTION  
        
        RETURN dp(zero unpacked remainder_count)  
    END FUNCTION  
END CLASS
