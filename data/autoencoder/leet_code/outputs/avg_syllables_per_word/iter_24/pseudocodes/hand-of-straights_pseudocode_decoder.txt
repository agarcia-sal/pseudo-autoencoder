CLASS Solution  
    FUNCTION isNStraightHand(hand PARAMETER, groupSize PARAMETER)  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero THEN  
            RETURN False  
        END IF  
        SET count TO the FREQUENCY COUNTS OF each element IN hand  
        SET unique_cards TO the LIST OF UNIQUE ELEMENTS IN count SORTED IN INCREASING ORDER  
        FOR each card IN unique_cards  
            IF the FREQUENCY OF card IN count GREATER THAN zero THEN  
                SET needed TO the FREQUENCY OF card IN count  
                FOR i FROM zero TO groupSize MINUS one  
                    IF the FREQUENCY OF the value EQUAL TO card PLUS i IN count LESS THAN needed THEN  
                        RETURN False  
                    END IF  
                    DECREMENT the FREQUENCY OF the value EQUAL TO card PLUS i IN count BY needed  
                END FOR  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
