CLASS Solution  
    FUNCTION minDistance(word1 word1 PARAMETER word2 word2 PARAMETER)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        SET dp TO a two dimensional list with m PLUS one rows and n PLUS one columns filled with zero  
        FOR i FROM zero TO m PLUS one MINUS one  
            SET element at position zero of the list at position i of dp TO i  
        END FOR  
        FOR j FROM zero TO n PLUS one MINUS one  
            SET element at position j of the list at position zero of dp TO j  
        END FOR  
        FOR i FROM one TO m PLUS one MINUS one  
            FOR j FROM one TO n PLUS one MINUS one  
                IF element at position i MINUS one of word1 EQUALS element at position j MINUS one of word2  
                    SET element at position j of the list at position i of dp TO element at position j MINUS one of the list at position i MINUS one of dp  
                ELSE  
                    SET element at position j of the list at position i of dp TO one PLUS the MINIMUM OF  
                        element at position j of the list at position i MINUS one of dp  
                        element at position j MINUS one of the list at position i of dp  
                        element at position j MINUS one of the list at position i MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position n of the list at position m of dp  
    END FUNCTION  
END CLASS
