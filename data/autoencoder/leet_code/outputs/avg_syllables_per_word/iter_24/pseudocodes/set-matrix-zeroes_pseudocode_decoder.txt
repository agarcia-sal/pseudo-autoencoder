CLASS Solution  
    FUNCTION setZeroes(matrix)  
        IF matrix EQUALS nothing OR the LENGTH OF matrix EQUALS zero OR the LENGTH OF element at position zero of matrix EQUALS zero  
            RETURN  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero of matrix  
        SET first_row_has_zero TO whether there EXISTS an index j FROM zero TO n MINUS one SUCH THAT element at position zero at position j of matrix EQUALS zero  
        SET first_col_has_zero TO whether there EXISTS an index i FROM zero TO m MINUS one SUCH THAT element at position i at position zero of matrix EQUALS zero  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF element at position i at position j of matrix EQUALS zero  
                    SET element at position i at position zero of matrix TO zero  
                    SET element at position zero at position j of matrix TO zero  
                END IF  
            END FOR  
        END FOR  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF element at position i at position zero of matrix EQUALS zero OR element at position zero at position j of matrix EQUALS zero  
                    SET element at position i at position j of matrix TO zero  
                END IF  
            END FOR  
        END FOR  
        IF first_row_has_zero  
            FOR j FROM zero TO n MINUS one  
                SET element at position zero at position j of matrix TO zero  
            END FOR  
        END IF  
        IF first_col_has_zero  
            FOR i FROM zero TO m MINUS one  
                SET element at position i at position zero of matrix TO zero  
            END FOR  
        END IF  
    END FUNCTION  
END CLASS
