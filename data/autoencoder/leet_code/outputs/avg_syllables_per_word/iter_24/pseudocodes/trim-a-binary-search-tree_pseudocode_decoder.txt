CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with value equal to the element at position zero of values  
    SET i TO one  
    CREATE an empty double ended queue called queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater than zero  
        SET node TO the element from the front of queue removed  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET node.left TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT no value  
            SET node.right TO a new TreeNode with value equal to the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS no value AND q IS no value  
        RETURN True  
    ELSE IF p IS no value OR q IS no value  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters p.left and q.left AND the result of is_same_tree with parameters p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION trimBST WITH PARAMETERS self root low high  
        IF root IS no value  
            RETURN no value  
        END IF  
        IF root.val LESS THAN low  
            RETURN the result of self.trimBST with parameters root.right low high  
        END IF  
        IF root.val GREATER THAN high  
            RETURN the result of self.trimBST with parameters root.left low high  
        END IF  
        SET root.left TO the result of self.trimBST with parameters root.left low high  
        SET root.right TO the result of self.trimBST with parameters root.right low high  
        RETURN root  
    END FUNCTION  
END CLASS
