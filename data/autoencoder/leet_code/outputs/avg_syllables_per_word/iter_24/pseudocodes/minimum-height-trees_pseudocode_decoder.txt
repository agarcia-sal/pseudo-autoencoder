CLASS Solution  
    FUNCTION findMinHeightTrees WITH PARAMETERS n edges  
        IF the value of n EQUALS one  
            RETURN the list containing the value zero  
        END IF  
        SET graph TO an empty mapping from integer to list of integers with default empty list  
        SET degree TO a list of zero values with size equal to n  
        FOR each element u comma v IN edges  
            APPEND the value v TO the list at key u in graph  
            APPEND the value u TO the list at key v in graph  
            INCREMENT the element at position u in degree BY one  
            INCREMENT the element at position v in degree BY one  
        END FOR  
        SET leaves TO a double ended queue containing all indices i from zero up to the value of n minus one for which the element at position i in degree EQUALS one  
        SET remaining_nodes TO the value of n  
        WHILE the value of remaining_nodes GREATER THAN two  
            SET leaves_count TO the number of elements in leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR iteration variable from zero TO leaves_count minus one  
                REMOVE the first element in leaves and ASSIGN it TO leaf  
                FOR each neighbor IN the list at key leaf in graph  
                    DECREMENT the element at position neighbor in degree BY one  
                    IF the element at position neighbor in degree EQUALS one  
                        APPEND the value neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN the list of elements contained in leaves  
    END FUNCTION  
END CLASS
