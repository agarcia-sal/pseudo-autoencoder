CLASS Solution  
    FUNCTION canTransform(start start_string end end_string)  
        SET start_filtered TO empty list  
        FOR each character c WITH index i IN start_string  
            IF character c NOT EQUALS the character X  
                APPEND the pair of character c AND index i TO start_filtered  
            END IF  
        END FOR  
        SET end_filtered TO empty list  
        FOR each character c WITH index i IN end_string  
            IF character c NOT EQUALS the character X  
                APPEND the pair of character c AND index i TO end_filtered  
            END IF  
        END FOR  
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        FOR each pair of elements (c1 i1) AND (c2 i2) IN start_filtered AND end_filtered correspondingly  
            IF character c1 NOT EQUALS character c2  
                RETURN false  
            END IF  
            IF character c1 EQUALS the character L AND index i1 LESS THAN index i2  
                RETURN false  
            END IF  
            IF character c1 EQUALS the character R AND index i1 GREATER THAN index i2  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
