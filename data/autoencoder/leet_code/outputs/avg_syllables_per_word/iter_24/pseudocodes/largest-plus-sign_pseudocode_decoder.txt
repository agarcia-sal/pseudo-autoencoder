CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n  
                                     mines)  
        SET mines_set TO an empty set  
        FOR each mine IN mines  
            ADD tuple of elements of mine TO mines_set  
        END FOR  
        
        SET dp TO a two dimensional list of size n by n where each element is a list of four zeros representing directions right down left up  
        
        SET directions TO list containing pair of zero and one pair of one and zero pair of zero and negative one pair of negative one and zero  
        
        FOR variable i FROM zero TO n MINUS one  
            FOR variable j FROM zero TO n MINUS one  
                IF pair of i and j NOT IN mines_set THEN  
                    IF j GREATER THAN zero THEN  
                        SET element at index zero of element at position j of element at position i of dp TO element at index zero of element at position j MINUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i GREATER THAN zero THEN  
                        SET element at index one of element at position j of element at position i of dp TO element at index one of element at position j of element at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR variable i FROM n MINUS one DOWN TO zero  
            FOR variable j FROM n MINUS one DOWN TO zero  
                IF pair of i and j NOT IN mines_set THEN  
                    IF j LESS THAN n MINUS one THEN  
                        SET element at index two of element at position j of element at position i of dp TO element at index two of element at position j PLUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i LESS THAN n MINUS one THEN  
                        SET element at index three of element at position j of element at position i of dp TO element at index three of element at position j of element at position i PLUS one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO minimum value among all elements of element at position j of element at position i of dp  
                    SET result TO the larger value between result AND plus_sign_order  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
