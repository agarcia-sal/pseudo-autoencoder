CLASS Solution  
    FUNCTION numOfArrays(n parameter TO integer, m parameter TO integer, k parameter TO integer) RETURNS integer  
        SET MOD TO one followed by nine zeros PLUS seven  
        
        SET dp TO three dimensional list of zero with dimensions n PLUS one by m PLUS one by k PLUS one  
        
        FOR j FROM one TO m  
            SET the element at position one at position j at position one of dp TO one  
        END FOR  
        
        FOR i FROM two TO n  
            FOR j FROM one TO m  
                FOR p FROM one TO k  
                    FOR x FROM one TO j MINUS one  
                        SET the element at position i at position j at position p of dp TO the sum of the element at position i at position j at position p of dp AND the element at position i MINUS one at position x at position p MINUS one of dp MODULO MOD  
                    END FOR  
                    
                    SET the element at position i at position j at position p of dp TO the sum of the element at position i at position j at position p of dp AND the product of the element at position i MINUS one at position j at position p of dp AND j MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR j FROM one TO m  
            SET result TO the sum of result AND the element at position n at position j at position k of dp MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
