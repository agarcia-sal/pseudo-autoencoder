CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET max_num TO the maximum element IN nums  
        SET num_set TO the set CONTAINING all elements OF nums  
        SET count TO zero  
        
        FOR each integer g FROM one TO max_num INCREMENTED BY one  
            SET gcd_val TO zero  
            FOR each integer m FROM g TO max_num IN STEPS OF g  
                IF element at position m IN num_set EXISTS  
                    SET gcd_val TO the greatest common divisor OF gcd_val AND m  
                    IF gcd_val EQUALS g  
                        INCREMENT count BY one  
                        EXIT the inner loop  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
