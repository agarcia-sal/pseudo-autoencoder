CLASS Solution  
    FUNCTION minAbbreviation(target FUNCTION parameter str, dictionary FUNCTION parameter List of str) RETURNS str  
        FUNCTION is_valid_abbr(s FUNCTION parameter str, abbr FUNCTION parameter str) RETURNS boolean  
            SET i TO zero  
            SET j TO zero  
            WHILE i LESS THAN the LENGTH OF s AND j LESS THAN the LENGTH OF abbr  
                IF the element at position j of abbr IS A DIGIT  
                    IF the element at position j of abbr EQUALS zero AS CHARACTER  
                        RETURN false  
                    END IF  
                    SET shift TO zero  
                    WHILE j LESS THAN the LENGTH OF abbr AND the element at position j of abbr IS A DIGIT  
                        SET shift TO shift MULTIPLIED BY ten PLUS the numeric value of the element at position j of abbr  
                        INCREMENT j BY one  
                    END WHILE  
                    INCREMENT i BY shift  
                ELSE  
                    IF the element at position i of s NOT EQUALS the element at position j of abbr  
                        RETURN false  
                    END IF  
                    INCREMENT i BY one  
                    INCREMENT j BY one  
                END IF  
            END WHILE  
            RETURN i EQUALS the LENGTH OF s AND j EQUALS the LENGTH OF abbr  
        END FUNCTION  
        
        FUNCTION generate_abbrs(word FUNCTION parameter str, length FUNCTION parameter integer, abbr FUNCTION parameter str ASSIGNED empty string, index FUNCTION parameter integer ASSIGNED zero, skip_count FUNCTION parameter integer ASSIGNED zero)  
            IF index EQUALS length  
                IF skip_count GREATER THAN zero  
                    SET abbr TO abbr CONCATENATED WITH the string representation of skip_count  
                END IF  
                APPEND abbr TO the list abbrs  
                RETURN  
            END IF  
            CALL generate_abbrs WITH word PARAMETER, length PARAMETER, abbr PARAMETER, index PLUS one PARAMETER, skip_count PLUS one PARAMETER  
            IF skip_count GREATER THAN zero  
                SET abbr TO abbr CONCATENATED WITH the string representation of skip_count  
            END IF  
            CALL generate_abbrs WITH word PARAMETER, length PARAMETER, abbr CONCATENATED WITH the element at position index of word PARAMETER, index PLUS one PARAMETER, zero PARAMETER  
        END FUNCTION  
        
        FUNCTION find_unique_abbr() RETURNS str  
            FOR length FROM one TO the LENGTH OF target  
                CALL generate_abbrs WITH target PARAMETER, the LENGTH OF target PARAMETER  
                SET min_abbr TO None  
                SET min_length TO positive infinity  
                FOR each abbr IN abbrs  
                    IF for all word IN filtered_dict the result of is_valid_abbr WITH word PARAMETER AND abbr PARAMETER IS false  
                        IF the LENGTH OF abbr LESS THAN min_length  
                            SET min_length TO the LENGTH OF abbr  
                            SET min_abbr TO abbr  
                        END IF  
                    END IF  
                END FOR  
                IF min_abbr IS NOT None  
                    RETURN min_abbr  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET filtered_dict TO an empty list  
        FOR each word IN dictionary  
            IF the LENGTH OF word EQUALS the LENGTH OF target  
                APPEND word TO filtered_dict  
            END IF  
        END FOR  
        
        SET abbrs TO an empty list  
        RETURN the result of find_unique_abbr  
    END FUNCTION  
END CLASS
