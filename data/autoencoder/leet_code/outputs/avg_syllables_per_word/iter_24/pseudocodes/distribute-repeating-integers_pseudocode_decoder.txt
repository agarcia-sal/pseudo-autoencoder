CLASS Solution  
    FUNCTION canDistribute(nums AS List OF integers, quantity AS List OF integers) AS boolean  
        SET count TO list of the frequencies of each distinct element in nums  
        SORT quantity IN descending order  
        
        FUNCTION can_satisfy(index AS integer) AS boolean  
            IF index EQUALS the LENGTH OF quantity THEN  
                RETURN true  
            END IF  
            FOR i FROM zero TO the LENGTH OF count MINUS one  
                IF element at position i of count GREATER THAN OR EQUAL TO element at position index of quantity THEN  
                    DECREMENT element at position i of count BY element at position index of quantity  
                    IF can_satisfy(index PLUS one) THEN  
                        RETURN true  
                    END IF  
                    INCREMENT element at position i of count BY element at position index of quantity  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN can_satisfy(zero)  
    END FUNCTION  
END CLASS
