CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals BY the end element of each interval and BY the start element of each interval in descending order when the ends are equal  
        SET first TO the end element of the first interval MINUS one  
        SET second TO the end element of the first interval  
        SET count TO two  
        FOR each pair of start and end IN intervals starting from the second element  
            IF start IS GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
