CLASS Solution  
    FUNCTION totalNQueens with parameter n  
        FUNCTION is_not_under_attack with parameters row and col  
            RETURN the negation of the sum of the element at position col of the list cols PLUS the element at position row MINUS col of the list hills PLUS the element at position row PLUS col of the list dales EQUALS zero  
        END FUNCTION  
        
        FUNCTION place_queen with parameters row and col  
            SET the element at position col of the list cols TO one  
            SET the element at position row MINUS col of the list hills TO one  
            SET the element at position row PLUS col of the list dales TO one  
        END FUNCTION  
        
        FUNCTION remove_queen with parameters row and col  
            SET the element at position col of the list cols TO zero  
            SET the element at position row MINUS col of the list hills TO zero  
            SET the element at position row PLUS col of the list dales TO zero  
        END FUNCTION  
        
        FUNCTION backtrack with parameters row equal to zero and count equal to zero  
            FOR col FROM zero TO n MINUS one  
                IF is_not_under_attack with arguments row and col EQUALS true  
                    CALL place_queen with arguments row and col  
                    IF row PLUS one EQUALS n  
                        INCREMENT count BY one  
                    ELSE  
                        SET count TO the result of calling backtrack with arguments row PLUS one and count  
                    END IF  
                    CALL remove_queen with arguments row and col  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET cols TO a new list of length n filled with zero  
        SET hills TO a new list of length two MULTIPLIED BY n MINUS one filled with zero  
        SET dales TO a new list of length two MULTIPLIED BY n MINUS one filled with zero  
        
        RETURN the result of calling backtrack with no arguments  
    END FUNCTION  
END CLASS
