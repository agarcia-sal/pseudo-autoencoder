CLASS Solution  
    FUNCTION rob(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN the element at position zero of nums  
        END IF  
        SET dp TO a new list of length the LENGTH OF nums with all elements SET TO zero  
        SET the element at position zero of dp TO the element at position zero of nums  
        SET the element at position one of dp TO the GREATER OF the element at position zero of nums AND the element at position one of nums  
        FOR index i FROM two TO the LENGTH OF nums MINUS one  
            SET value_one TO the element at position i MINUS one of dp  
            SET value_two TO the sum of the element at position i MINUS two of dp PLUS the element at position i of nums  
            SET the element at position i of dp TO the GREATER OF value_one AND value_two  
        END FOR  
        RETURN the element at the last position of dp  
    END FUNCTION  
END CLASS
