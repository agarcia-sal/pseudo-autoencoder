CLASS Solution  
    FUNCTION subStrHash(s power modulo k hashValue)  
        FUNCTION val(c)  
            RETURN the numerical position of the character c in the alphabet starting from one for the first letter  
        END FUNCTION  
        
        SET n TO the LENGTH OF s  
        SET current_hash TO zero  
        SET p_pow TO one  
        
        FOR i FROM n MINUS k TO n MINUS one  
            SET current_hash TO the REMAINDER OF current_hash PLUS the product OF val of element at position i of s AND p_pow DIVIDED BY modulo  
            IF i LESS THAN n MINUS one  
                SET p_pow TO the REMAINDER OF p_pow MULTIPLIED BY power DIVIDED BY modulo  
            END IF  
        END FOR  
        
        SET start_index TO n MINUS k  
        IF current_hash EQUALS hashValue  
            RETURN the substring of s from position start_index TO the position start_index PLUS k MINUS one  
        END IF  
        
        SET p_pow_k_minus_1 TO p_pow  
        FOR i FROM n MINUS k MINUS one DOWNTO zero  
            SET current_hash TO the REMAINDER OF current_hash MINUS the product OF val of element at position i PLUS k of s AND p_pow_k_minus_1 DIVIDED BY modulo  
            SET current_hash TO the REMAINDER OF current_hash MULTIPLIED BY power PLUS val of element at position i of s DIVIDED BY modulo  
            
            IF current_hash EQUALS hashValue  
                SET start_index TO i  
            END IF  
        END FOR  
        
        RETURN the substring of s from position start_index TO the position start_index PLUS k MINUS one  
    END FUNCTION  
END CLASS
