CLASS Solution  
    FUNCTION findLonelyPixel(picture)  
        IF picture EQUALS empty OR element at position zero of picture EQUALS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO a list of zeroes with the LENGTH OF m  
        SET col_count TO a list of zeroes with the LENGTH OF n  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j of element at position i of picture EQUALS the uppercase letter B  
                    INCREMENT the element at position i of row_count BY one  
                    INCREMENT the element at position j of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        SET lonely_pixel_count TO zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j of element at position i of picture EQUALS the uppercase letter B AND the element at position i of row_count EQUALS one AND the element at position j of col_count EQUALS one  
                    INCREMENT lonely_pixel_count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
