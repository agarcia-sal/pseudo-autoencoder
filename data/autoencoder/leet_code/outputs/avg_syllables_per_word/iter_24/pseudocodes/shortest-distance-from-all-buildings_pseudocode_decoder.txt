CLASS Solution  
    FUNCTION shortestDistance(grid)  
        IF grid EQUALS no elements OR element at position zero of grid EQUALS no elements  
            RETURN negative one  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET total_distance TO a two dimensional list of zeroes with m rows and n columns  
        SET reach_count TO a two dimensional list of zeroes with m rows and n columns  
        SET building_count TO zero  
        
        SET directions TO a list of four pairs representing one step down one step left one step up one step right  
        
        FUNCTION bfs(start_x, start_y)  
            SET queue TO a deque containing a pair of start_x and start_y  
            SET visited TO a two dimensional list of boolean FALSE with m rows and n columns  
            SET distance TO zero  
            SET reachable TO zero  
            
            WHILE the queue holds elements  
                SET size TO the LENGTH OF the queue  
                INCREMENT distance BY one  
                FOR index FROM zero TO size MINUS one  
                    REMOVE the element at the left end of queue AND ASSIGN the first component TO x AND the second component TO y  
                    FOR each pair dx, dy IN directions  
                        SET nx TO x PLUS dx  
                        SET ny TO y PLUS dy  
                        IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx at position ny of visited EQUALS FALSE AND element at position nx at position ny of grid EQUALS zero  
                            SET element at position nx at position ny of visited TO TRUE  
                            APPEND the pair nx and ny TO queue  
                            INCREMENT element at position nx at position ny of total_distance BY distance  
                            INCREMENT element at position nx at position ny of reach_count BY one  
                            INCREMENT reachable BY one  
                        END IF  
                    END FOR  
                END FOR  
            END WHILE  
            RETURN reachable  
        END FUNCTION  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i at position j of grid EQUALS one  
                    INCREMENT building_count BY one  
                    IF bfs(i, j) LESS THAN building_count MINUS one  
                        RETURN negative one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET min_distance TO infinity  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i at position j of grid EQUALS zero AND element at position i at position j of reach_count EQUALS building_count  
                    IF element at position i at position j of total_distance LESS THAN min_distance  
                        SET min_distance TO element at position i at position j of total_distance  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF min_distance NOT EQUALS infinity  
            RETURN min_distance  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
