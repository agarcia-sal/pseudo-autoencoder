CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO an empty mapping with default integer zero  
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain INTO two parts first part AS count_string AND second part AS domain  
            SET count TO the integer value of count_string  
            SPLIT domain INTO subdomains separated by the period character  
            FOR index FROM zero TO the length of subdomains MINUS one  
                SET subdomain_parts TO the elements of subdomains from position index TO the last position  
                SET subdomain TO the concatenation of subdomain_parts joined by the period character  
                INCREMENT the value associated with subdomain in domain_count BY count  
            END FOR  
        END FOR  
        SET result_list TO an empty list  
        FOR each key domain_value_pair IN domain_count items  
            DECOMPOSE domain_value_pair INTO domain AND count  
            APPEND the string formed by the string representation of count PLUS a space PLUS domain TO result_list  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
