CLASS ListNode  
    FUNCTION __init__  val DEFAULT ZERO  next DEFAULT NONE  
        SET the val property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node  values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode constructed with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the last element of values  
        SET node TO a new ListNode constructed with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list  p1  p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the condition that the val property of p1 EQUALS the val property of p2 AND the result of is_same_list called with the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd  head  n  
        SET dummy TO a new ListNode constructed with ZERO  
        SET the next property of dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM ZERO TO n PLUS ONE  
            SET first TO the next property of first  
        END FOR  
        WHILE first IS NOT none  
            SET first TO the next property of first  
            SET second TO the next property of second  
        END WHILE  
        SET the next property of second TO the next property of the next property of second  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
