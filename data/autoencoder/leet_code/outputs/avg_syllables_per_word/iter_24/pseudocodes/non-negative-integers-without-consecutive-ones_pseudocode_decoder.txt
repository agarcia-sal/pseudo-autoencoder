CLASS Solution  
    FUNCTION findIntegers WITH PARAMETER n  
        SET bin_rep TO the binary representation of n excluding the first character  
        SET length TO the length of bin_rep  
        
        INITIALIZE dp AS a list of zeros with size length PLUS one  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO two  
        
        FOR i FROM two TO length  
            SET element at position i of dp TO element at position i MINUS one of dp PLUS element at position i MINUS two of dp  
        END FOR  
        
        SET result TO zero  
        SET prev_bit TO zero  
        
        FOR i FROM zero TO length MINUS one  
            IF the element at position i of bin_rep EQUALS one  
                INCREMENT result BY element at position length MINUS i MINUS one of dp  
                IF prev_bit EQUALS one  
                    DECREMENT result BY one  
                    BREAK the loop  
                END IF  
            END IF  
            
            SET prev_bit TO one IF the element at position i of bin_rep EQUALS one OTHERWISE zero  
        END FOR  
        
        RETURN result PLUS one  
    END FUNCTION  
END CLASS
