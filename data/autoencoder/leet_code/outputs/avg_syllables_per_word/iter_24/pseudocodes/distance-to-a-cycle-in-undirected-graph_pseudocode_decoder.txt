CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO an empty mapping from keys to lists  
        FOR each pair u AND v IN edges  
            APPEND v TO the list for u in graph  
            APPEND u TO the list for v in graph  
        END FOR  
      
        FUNCTION dfs(node, parent, path, visited)  
            IF node IS IN visited THEN  
                SET cycle_start TO the position of node in path  
                RETURN the subset of path starting from cycle_start to the end  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN the list for node in graph  
                IF neighbor IS NOT EQUAL TO parent THEN  
                    SET cycle TO dfs(neighbor, node, path, visited)  
                    IF cycle IS NOT None THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE the last element from path  
            RETURN None  
        END FUNCTION  
      
        SET visited TO an empty set  
        SET cycle TO None  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                SET cycle TO dfs(node, minus one, an empty list, visited)  
                IF cycle IS NOT None THEN  
                    BREAK the loop  
                END IF  
            END IF  
        END FOR  
      
        SET distance TO a list of zeros with length n  
        SET queue TO a deque initialized with the elements of cycle  
        SET visited TO a set containing all elements of cycle  
      
        SET level TO zero  
        WHILE the queue IS NOT empty  
            FOR counter FROM zero TO the length of queue MINUS one  
                REMOVE the first element FROM queue AND ASSIGN it TO node  
                SET distance at position node TO level  
                FOR each neighbor IN the list for node in graph  
                    IF neighbor IS NOT IN visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  
      
        RETURN distance  
    END FUNCTION  
END CLASS
