CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones EQUALS empty OR LENGTH OF stones LESS THAN two  
            RETURN false  
        END IF  
        
        SET stone_set TO a new set containing each element IN stones  
        SET memo TO a new empty dictionary  
        
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS the element at the last position of stones  
                RETURN true  
            END IF  
            
            IF the tuple of position AND jump EXISTS AS a key IN memo  
                RETURN the value at key tuple of position AND jump IN memo  
            END IF  
            
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump GREATER THAN zero AND position PLUS next_jump IS AN ELEMENT OF stone_set  
                    IF can_jump_to(position PLUS next_jump, next_jump)  
                        SET the value at key tuple of position AND jump IN memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value at key tuple of position AND jump IN memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN can_jump_to(the element at the first position of stones, zero)  
    END FUNCTION  
END CLASS
