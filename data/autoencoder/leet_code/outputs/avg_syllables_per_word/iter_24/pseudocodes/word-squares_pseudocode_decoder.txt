CLASS Solution  
    FUNCTION wordSquares(words)  
        CREATE an empty dictionary prefix_to_words that maps from prefix strings TO lists of words  
        FOR each word IN words  
            FOR index FROM zero TO the LENGTH OF word MINUS one  
                SET prefix TO the substring from position zero TO position index MINUS one of word  
                APPEND word TO the list at prefix_to_words corresponding TO prefix  
            END FOR  
        END FOR  
        
        CREATE an empty list results  
        
        FUNCTION backtrack(square)  
            IF the LENGTH OF square EQUALS the LENGTH OF the first word in words THEN  
                APPEND square TO results  
                RETURN  
            END IF  
            
            SET prefix TO the concatenation of the element at position the LENGTH OF square OF each word IN square  
            FOR each candidate IN the list at prefix_to_words corresponding TO prefix  
                CALL backtrack WITH square concatenated WITH candidate  
            END FOR  
        END FUNCTION  
        
        FOR each word IN words  
            CALL backtrack WITH a list containing word  
        END FOR  
        
        RETURN results  
    END FUNCTION  
END CLASS
