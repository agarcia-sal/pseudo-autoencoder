CLASS Solution  
    FUNCTION shortestDistance(maze AS List of List of integer, start AS List of integer, destination AS List of integer) RETURNS integer  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO a list containing the pairs one and zero, negative one and zero, zero and one, zero and negative one  
        CONVERT start TO a tuple AND ASSIGN TO start  
        CONVERT destination TO a tuple AND ASSIGN TO destination  
        SET visited TO an empty set  
        SET pq TO a list containing a tuple of zero AND start  

        WHILE pq is not empty  
            REMOVE and RETURN the smallest element from pq AND ASSIGN first element to dist AND second element to a tuple of x and y  
            
            IF x EQUALS element at position zero of destination AND y EQUALS element at position one of destination  
                RETURN dist  
            END IF  
            
            IF tuple of x and y IS IN visited  
                CONTINUE to next iteration of WHILE loop  
            END IF  
            
            ADD tuple of x and y TO visited  
            
            FOR each pair dx and dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of element at position new_y PLUS dy of maze EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF tuple of new_x and new_y IS NOT IN visited  
                    INSERT a tuple of dist PLUS count AND tuple of new_x and new_y INTO pq maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
