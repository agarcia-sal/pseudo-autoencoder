CLASS Solution  
    FUNCTION findMaximizedCapital(k AS integer, w AS integer, profits AS list OF integer, capital AS list OF integer) RETURNS integer  
        SET projects TO the LIST OF PAIRS OF elements FROM capital AND profits COMBINED such that each pair CONTAINS element at position same index of capital AND element at position same index of profits  
        SORT projects BY the FIRST element OF each pair IN ascending order  
        SET max_heap TO an EMPTY list  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        FOR each iteration FROM one TO k  
            WHILE i LESS THAN n AND the FIRST element of the pair at position i of projects LESS THAN OR EQUAL TO w  
                INSERT the NEGATION of the SECOND element of the pair at position i of projects INTO max_heap USING heap insertion  
                INCREMENT i BY one  
            END WHILE  
            IF max_heap IS empty  
                BREAK the loop  
            END IF  
            SET w TO w PLUS the NEGATION OF the element REMOVED FROM max_heap USING heap removal  
        END FOR  
        RETURN w  
    END FUNCTION  
END CLASS
