CLASS Solution  
    FUNCTION longestPalindromeSubseq(s)  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of zeros with dimensions n BY n  
        FOR i FROM zero TO n MINUS one  
            SET element at position i at position i of dp TO one  
        END FOR  
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of s EQUALS element at position j of s  
                    SET element at position i PLUS one at position j MINUS one of dp TO value at element at position i PLUS one at position j MINUS one of dp  
                    SET element at position i at position j of dp TO the previous value PLUS two  
                ELSE  
                    SET value_one TO element at position i PLUS one at position j of dp  
                    SET value_two TO element at position i at position j MINUS one of dp  
                    IF value_one GREATER THAN OR EQUAL TO value_two  
                        SET element at position i at position j of dp TO value_one  
                    ELSE  
                        SET element at position i at position j of dp TO value_two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position zero at position n MINUS one of dp  
    END FUNCTION  
END CLASS
