CLASS Solution  
    FUNCTION sumScores(s)  
        SET n TO the LENGTH OF s  
        SET lcp TO a list of zeros with length n  
        SET l TO zero  
        SET r TO zero  
        FOR i FROM one TO n MINUS one  
            IF i LESS THAN OR EQUAL TO r  
                SET lcp at position i TO the MINIMUM OF the DIFFERENCE OF r MINUS i PLUS one AND lcp at position i MINUS l  
            END IF  
            WHILE the SUM of i AND lcp at position i LESS THAN n AND the ELEMENT at position lcp at position i of s EQUALS the ELEMENT at position SUM of i AND lcp at position i of s  
                INCREMENT lcp at position i BY one  
            END WHILE  
            IF the SUM of i AND lcp at position i MINUS one GREATER THAN r  
                SET l TO i  
                SET r TO the SUM of i AND lcp at position i MINUS one  
            END IF  
        END FOR  
        RETURN the SUM OF all elements in lcp PLUS n  
    END FUNCTION  
END CLASS
