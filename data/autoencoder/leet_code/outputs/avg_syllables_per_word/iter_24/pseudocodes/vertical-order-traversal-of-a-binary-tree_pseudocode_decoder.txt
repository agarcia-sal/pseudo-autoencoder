CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val EQUALS element at position zero of values  
    SET i TO one  
    SET queue TO empty deque structure  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None THEN  
            SET node.left TO new TreeNode with val EQUALS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None THEN  
            SET node.right TO new TreeNode with val EQUALS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val THEN  
        RETURN False  
    ELSE  
        RETURN result of is_same_tree with p.left AND q.left AND result of is_same_tree with p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalTraversal(self, root)  
        IF root EQUALS None THEN  
            RETURN empty list  
        END IF  
        SET node_dict TO a new default dictionary mapping keys to empty lists  
        SET min_col TO zero  
        SET max_col TO zero  

        FUNCTION dfs(node, row, col)  
            IF node EQUALS None THEN  
                RETURN  
            END IF  
            APPEND tuple of row AND node.val TO list in node_dict at key col  
            SET min_col TO the smaller value between min_col AND col  
            SET max_col TO the larger value between max_col AND col  
            CALL dfs with node.left AND row PLUS one AND col MINUS one  
            CALL dfs with node.right AND row PLUS one AND col PLUS one  
        END FUNCTION  

        CALL dfs with root AND zero AND zero  

        SET result TO empty list  
        FOR col FROM min_col TO max_col DO  
            SET sorted_list TO the list in node_dict at key col sorted by first element of each tuple AND then by second element of each tuple  
            SET values_list TO empty list  
            FOR each tuple in sorted_list DO  
                APPEND second element of tuple TO values_list  
            END FOR  
            APPEND values_list TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
