CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with the left attribute of p AND the left attribute of q AND the result of is_same_tree with the right attribute of p AND the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self, root, k)  
        IF the left attribute of root EQUALS None AND the right attribute of root EQUALS None  
            RETURN the val attribute of root  
        END IF  
        SET graph TO a new default dictionary mapping each key to an empty list  
        FUNCTION dfs(node, parent EQUALS None)  
            IF parent IS NOT None  
                APPEND the val attribute of parent TO the list mapped from the val attribute of node in graph  
                APPEND the val attribute of node TO the list mapped from the val attribute of parent in graph  
            END IF  
            IF the left attribute of node IS NOT None  
                CALL dfs with the left attribute of node AND node  
            END IF  
            IF the right attribute of node IS NOT None  
                CALL dfs with the right attribute of node AND node  
            END IF  
        END FUNCTION  
        CALL dfs with root  
        SET queue TO a new deque initialized with k  
        SET visited TO a new empty set  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET current TO the element removed from the left of queue  
            ADD current TO visited  
            IF the LENGTH OF the list mapped from current in graph EQUALS one AND current NOT EQUALS the val attribute of root  
                RETURN current  
            END IF  
            FOR each neighbor IN the list mapped from current in graph  
                IF neighbor NOT IN visited  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
