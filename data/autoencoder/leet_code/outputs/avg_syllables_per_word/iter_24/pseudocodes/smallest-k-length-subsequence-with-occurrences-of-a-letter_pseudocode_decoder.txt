CLASS Solution  
    FUNCTION smallestSubsequence(s AS string, k AS integer, letter AS string, repetition AS integer) RETURNS string  
        SET letter_count TO the number of occurrences of letter IN s  
        SET stack TO an empty list  
        FOR each index i AND character char IN s starting from the first position TO the last position  
            WHILE the length of stack IS GREATER THAN zero AND character char IS LESS THAN element at the last position OF stack AND length of s MINUS index i PLUS length of stack MINUS one IS GREATER THAN OR EQUAL TO k  
                IF element at the last position OF stack EQUALS letter  
                    IF letter_count IS GREATER THAN repetition  
                        INCREMENT repetition BY one  
                        REMOVE element at the last position FROM stack  
                    ELSE  
                        BREAK the while loop  
                    END IF  
                ELSE  
                    REMOVE element at the last position FROM stack  
                END IF  
            END WHILE  
            IF length of stack IS LESS THAN k  
                IF character char EQUALS letter  
                    APPEND character char TO stack  
                    DECREMENT repetition BY one  
                ELSE IF k MINUS length of stack IS GREATER THAN repetition  
                    APPEND character char TO stack  
                END IF  
                IF character char EQUALS letter  
                    DECREMENT letter_count BY one  
                END IF  
            END IF  
        END FOR  
        WHILE length of stack IS LESS THAN k  
            APPEND letter TO stack  
        END WHILE  
        RETURN the concatenation of all elements of stack into a single string  
    END FUNCTION  
END CLASS
