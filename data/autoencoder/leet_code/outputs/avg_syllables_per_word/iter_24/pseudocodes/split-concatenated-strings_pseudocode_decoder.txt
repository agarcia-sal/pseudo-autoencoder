CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_str TO the string at position index of strs reversed in order  
            IF reversed_str GREATER THAN the string at position index of strs  
                SET the string at position index of strs TO reversed_str  
            END IF  
        END FOR  
        
        SET doubled_strs TO the concatenation of strs and strs  
        
        SET max_string TO the empty string  
        
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR each value IN the collection of false and true  
                IF value EQUALS true  
                    SET current TO the string at position index of strs reversed in order  
                ELSE  
                    SET current TO the string at position index of strs  
                END IF  
                
                SET remaining TO the concatenation of the strings from position index plus one TO position index plus the LENGTH OF strs MINUS one of doubled_strs  
                
                FOR cut_position FROM zero TO the LENGTH OF current  
                    SET candidate TO the concatenation of the substring from position cut_position TO the end of current the string remaining and the substring from zero TO position cut_position of current  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
