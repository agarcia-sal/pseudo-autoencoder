CLASS Solution  
    FUNCTION search(nums TARGET)  
        IF LENGTH OF nums EQUALS zero  
            RETURN False  
        END IF  
        
        SET left TO zero  
        SET right TO LENGTH OF nums MINUS one  
        
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS left MINUS right DIVIDED BY two  
            
            IF element at position mid of nums EQUALS TARGET  
                RETURN True  
            END IF  
            
            IF element at position left of nums EQUALS element at position mid of nums AND element at position mid of nums EQUALS element at position right of nums  
                INCREMENT left BY one  
                DECREMENT right BY one  
            ELSE IF element at position left of nums LESS THAN OR EQUAL TO element at position mid of nums  
                IF element at position left of nums LESS THAN OR EQUAL TO TARGET AND TARGET LESS THAN element at position mid of nums  
                    SET right TO mid MINUS one  
                ELSE  
                    SET left TO mid PLUS one  
                END IF  
            ELSE  
                IF element at position mid of nums LESS THAN TARGET AND TARGET LESS THAN OR EQUAL TO element at position right of nums  
                    SET left TO mid PLUS one  
                ELSE  
                    SET right TO mid MINUS one  
                END IF  
            END IF  
        END WHILE  
        
        RETURN False  
    END FUNCTION  
END CLASS
