CLASS Solution  
    FUNCTION smallestSufficientTeam(self req_skills people)  
        SET d TO an empty dictionary mapping each skill in req_skills to its position index starting from zero  
        SET m TO the LENGTH OF req_skills  
        SET n TO the LENGTH OF people  
        SET p TO a list of zeroes with length n  
        FOR each index i FROM zero TO n MINUS one  
            FOR each skill s IN the list of skills at position i in people  
                SET the element at position i in p TO the element at position i in p OR the number one shifted LEFT by the position index of skill s in d  
            END FOR  
        END FOR  
        SET f TO a list filled with infinity with length equal to two RAISED TO the power m  
        SET g TO a list filled with zero with length equal to two RAISED TO the power m  
        SET h TO a list filled with zero with length equal to two RAISED TO the power m  
        SET the element at position zero in f TO zero  
        FOR each integer i FROM zero TO two RAISED TO the power m MINUS one  
            IF the element at position i in f EQUALS infinity THEN  
                CONTINUE to the next iteration  
            END IF  
            FOR each integer j FROM zero TO n MINUS one  
                IF the element at position i in f PLUS one LESS THAN the element at the bitwise OR of i and the element at position j in p in f THEN  
                    SET the element at the bitwise OR of i and the element at position j in p in f TO the element at position i in f PLUS one  
                    SET the element at the bitwise OR of i and the element at position j in p in g TO j  
                    SET the element at the bitwise OR of i and the element at position j in p in h TO i  
                END IF  
            END FOR  
        END FOR  
        SET i TO two RAISED TO the power m MINUS one  
        SET ans TO an empty list  
        WHILE i NOT EQUALS zero  
            APPEND the element at position i in g TO ans  
            SET i TO the element at position i in h  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
