CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the string form of the element at position one of nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the string form of the element at position one of nums PLUS the forward slash character PLUS the string form of the element at position two of nums  
        END IF  
        RETURN the string form of the element at position one of nums PLUS the forward slash character PLUS an opening parenthesis character PLUS the result of joining each string form of elements in nums from position two TO the end with the forward slash character PLUS a closing parenthesis character  
    END FUNCTION  
END CLASS
