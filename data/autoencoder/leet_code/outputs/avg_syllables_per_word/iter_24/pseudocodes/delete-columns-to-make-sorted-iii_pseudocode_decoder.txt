CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF element at position zero of strs  
        SET dp TO a list of size n where each element is assigned the value one  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF all the elements at position k of element at position j of strs LESS THAN OR EQUAL TO the elements at position k of element at position i of strs for each k FROM zero TO m MINUS one  
                    SET dp at position i TO the GREATER VALUE BETWEEN dp at position i AND the VALUE OF dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        RETURN n MINUS the GREATER VALUE IN dp  
    END FUNCTION  
END CLASS
