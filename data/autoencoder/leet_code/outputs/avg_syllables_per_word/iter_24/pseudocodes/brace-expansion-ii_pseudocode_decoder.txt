CLASS Solution  
    FUNCTION braceExpansionII(expression)  
        FUNCTION parse_expression(index)  
            SET current_set TO an empty set  
            SET current_product TO a list containing an empty string  
            
            WHILE index LESS THAN the LENGTH OF expression  
                SET char TO element at position index of expression  
                
                IF char EQUALS the character representing left curly brace  
                    SET inner_set AND index TO the result of calling parse_expression WITH argument index PLUS one  
                    SET new_product TO an empty list  
                    FOR each string a IN current_product  
                        FOR each string b IN inner_set  
                            APPEND the concatenation of string a PLUS string b TO new_product  
                        END FOR  
                    END FOR  
                    SET current_product TO new_product  
                
                ELSE IF char EQUALS the character representing right curly brace  
                    FOR each string p IN current_product  
                        JOIN string p into a single string  
                        ADD this string TO current_set  
                    END FOR  
                    RETURN current_set AND index  
                
                ELSE IF char EQUALS the character representing comma  
                    FOR each string p IN current_product  
                        JOIN string p into a single string  
                        ADD this string TO current_set  
                    END FOR  
                    SET current_product TO a list containing an empty string  
                
                ELSE  
                    SET new_product TO an empty list  
                    FOR each string a IN current_product  
                        CONCATENATE string a PLUS char  
                        APPEND this concatenated string TO new_product  
                    END FOR  
                    SET current_product TO new_product  
                
                END IF  
                
                INCREMENT index BY one  
            END WHILE  
            
            FOR each string p IN current_product  
                JOIN string p into a single string  
                ADD this string TO current_set  
            END FOR  
            RETURN current_set AND index  
        END FUNCTION  
        
        SET result_set AND unused_index TO the result of calling parse_expression WITH argument zero  
        RETURN the sorted list of elements FROM result_set  
    END FUNCTION  
END CLASS
