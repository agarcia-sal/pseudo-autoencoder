CLASS ListNode  
    FUNCTION __init__(self val EQUALS zero next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with the first element at position one of values  
    SET p TO head  
    FOR each val IN the elements from position two TO the last element of values  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the value at p1 EQUALS the value at p2 AND is_same_list the next of p1 AND the next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs(self head)  
        SET dummy TO new ListNode with zero  
        SET dummy.next TO head  
        SET current TO dummy  
        WHILE current.next NOT EQUALS None AND current.next.next NOT EQUALS None  
            SET first TO the element at next of current  
            SET second TO the element at the next of next of current  
            SET current.next TO second  
            SET first.next TO second.next  
            SET second.next TO first  
            SET current TO the next of the next of current  
        END WHILE  
        RETURN the element at next of dummy  
    END FUNCTION  
END CLASS
