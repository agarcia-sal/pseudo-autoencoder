CLASS Solution  
    FUNCTION shiftingLetters(s parameter OF TYPE string shifts parameter OF TYPE list of integers)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one DOWN TO zero  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET original character value TO the alphabetical position of element at position index of s starting from the position of lowercase letter a  
            SET shifted character value TO original character value PLUS element at position index of shifts  
            SET wrapped character value TO the remainder of shifted character value DIVIDED BY twenty six  
            SET new character TO the character at position wrapped character value starting from the position of lowercase letter a  
            APPEND new character TO result  
        END FOR  
        RETURN string formed by joining all elements in result  
    END FUNCTION  
END CLASS
