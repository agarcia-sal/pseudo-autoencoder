CLASS Solution  
    FUNCTION checkPossibility(nums)  
        SET modified TO false  
        FOR i FROM two TO the LENGTH OF nums  
            IF element at position i of nums LESS THAN element at position i MINUS one of nums  
                IF modified EQUALS true  
                    RETURN false  
                END IF  
                SET modified TO true  
                IF i LESS THAN two OR element at position i of nums GREATER THAN OR EQUAL TO element at position i MINUS two of nums  
                    SET element at position i MINUS one of nums TO element at position i of nums  
                ELSE  
                    SET element at position i of nums TO element at position i MINUS one of nums  
                END IF  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
