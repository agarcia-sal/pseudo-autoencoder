CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value equal to element at position one of values and left equal to none and right equal to none  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element obtained by removing from the front of queue  
        IF i LESS THAN the length of values AND element at position i of values is not none  
            SET node.left TO new TreeNode with value equal to element at position i of values and left equal to none and right equal to none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i of values is not none  
            SET node.right TO new TreeNode with value equal to element at position i of values and left equal to none and right equal to none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION dfs(node, current_sum, path)  
            IF node EQUALS none  
                RETURN  
            END IF  
            APPEND the value of node TO path  
            SET current_sum TO current_sum PLUS the value of node  
            IF the left child of node EQUALS none AND the right child of node EQUALS none AND current_sum EQUALS targetSum  
                APPEND a new list copied from path TO result  
            END IF  
            CALL dfs(with the left child of node, current_sum, path)  
            CALL dfs(with the right child of node, current_sum, path)  
            REMOVE the last element FROM path  
        END FUNCTION  
        SET result TO empty list  
        CALL dfs(root, zero, empty list)  
        RETURN result  
    END FUNCTION  
END CLASS
