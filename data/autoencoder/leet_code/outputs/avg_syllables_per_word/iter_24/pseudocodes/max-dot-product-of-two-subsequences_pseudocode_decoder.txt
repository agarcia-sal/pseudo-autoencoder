CLASS Solution  
    FUNCTION maxDotProduct(nums1 PARAMETER, nums2 PARAMETER)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        SET dp TO a two dimensional list with m PLUS one rows and n PLUS one columns where each element IS set TO a very small number representing negative infinity  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO element at position i MINUS one of nums1 MULTIPLIED BY element at position j MINUS one of nums2  
                SET dp element at position i and position j TO the MAXIMUM value among the following four values  
                    current_product  
                    dp element at position i MINUS one and position j  
                    dp element at position i and position j MINUS one  
                    dp element at position i MINUS one and position j MINUS one PLUS current_product  
            END FOR  
        END FOR  
        RETURN dp element at position m and position n  
    END FUNCTION  
END CLASS
