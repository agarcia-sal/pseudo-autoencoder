CLASS Solution  
    FUNCTION maxA(n)  
        IF n LESS THAN OR EQUAL TO three  
            RETURN n  
        END IF  

        SET dp TO a list of size n PLUS one filled with zero  

        FOR i FROM one TO three  
            SET element at position i of dp TO i  
        END FOR  

        FOR i FROM four TO n  
            FOR j FROM i MINUS three DOWN TO one  
                SET candidate TO element at position j of dp MULTIPLIED BY i MINUS j MINUS one  
                IF candidate GREATER THAN element at position i of dp  
                    SET element at position i of dp TO candidate  
                END IF  
            END FOR  
        END FOR  

        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
