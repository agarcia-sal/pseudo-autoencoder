CLASS Solution  
    FUNCTION outerTrees(trees)  
        FUNCTION dist_sq(p1, p2)  
            RETURN the square of the difference between the element at position zero of p1 AND the element at position zero of p2 PLUS the square of the difference between the element at position one of p1 AND the element at position one of p2  
        END FUNCTION  
        
        FUNCTION circumcenter(p1, p2, p3)  
            SET ax TO element at position zero of p1  
            SET ay TO element at position one of p1  
            SET bx TO element at position zero of p2  
            SET by TO element at position one of p2  
            SET cx TO element at position zero of p3  
            SET cy TO element at position one of p3  
            SET d TO two MULTIPLIED BY the sum of ax MULTIPLIED BY the difference of by MINUS cy PLUS bx MULTIPLIED BY the difference of cy MINUS ay PLUS cx MULTIPLIED BY the difference of ay MINUS by  
            SET ux TO the sum of the sum of ax MULTIPLIED BY ax PLUS ay MULTIPLIED BY ay MULTIPLIED BY the difference of by MINUS cy PLUS the sum of bx MULTIPLIED BY bx PLUS by MULTIPLIED BY by MULTIPLIED BY the difference of cy MINUS ay PLUS the sum of cx MULTIPLIED BY cx PLUS cy MULTIPLIED BY cy MULTIPLIED BY the difference of ay MINUS by DIVIDED BY d  
            SET uy TO the sum of the sum of ax MULTIPLIED BY ax PLUS ay MULTIPLIED BY ay MULTIPLIED BY the difference of cx MINUS bx PLUS the sum of bx MULTIPLIED BY bx PLUS by MULTIPLIED BY by MULTIPLIED BY the difference of ax MINUS cx PLUS the sum of cx MULTIPLIED BY cx PLUS cy MULTIPLIED BY cy MULTIPLIED BY the difference of bx MINUS ax DIVIDED BY d  
            RETURN a list consisting of ux AND uy AND the square root of the sum of the square of the difference of ux MINUS ax PLUS the square of the difference of uy MINUS ay  
        END FUNCTION  
        
        FUNCTION welzl(points, boundary, n)  
            IF n EQUALS zero OR the length of boundary EQUALS three  
                IF the length of boundary EQUALS zero  
                    RETURN a list consisting of zero AND zero AND zero  
                ELSE IF the length of boundary EQUALS one  
                    RETURN the element at position zero of boundary AND zero  
                ELSE IF the length of boundary EQUALS two  
                    SET center TO a list consisting of the sum of the element at position zero of the element at position zero of boundary PLUS the element at position zero of the element at position one of boundary DIVIDED BY two AND the sum of the element at position one of the element at position zero of boundary PLUS the element at position one of the element at position one of boundary DIVIDED BY two  
                    SET radius TO the square root of the distance squared between the element at position zero of boundary AND the element at position one of boundary DIVIDED BY two  
                    RETURN center AND radius  
                ELSE  
                    RETURN circumcenter with unpacked boundary as parameters  
                END IF  
            END IF  
            
            SET p TO the element at position n MINUS one of points  
            SET center AND radius TO the result of welzl of points boundary AND n MINUS one  
            IF the distance squared between center AND p LESS THAN OR EQUAL TO radius MULTIPLIED BY radius  
                RETURN center AND radius  
            END IF  
            
            RETURN welzl of points with boundary appended by p AND n MINUS one  
        END FUNCTION  
        
        APPLY random shuffling to trees  
        SET center AND radius TO welzl of trees AND an empty list AND the length of trees  
        RETURN a list consisting of element at position zero of center AND element at position one of center AND radius  
    END FUNCTION  
END CLASS
