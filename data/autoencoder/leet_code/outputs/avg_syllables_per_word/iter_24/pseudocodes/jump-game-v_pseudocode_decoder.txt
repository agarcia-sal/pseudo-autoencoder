CLASS Solution  
    FUNCTION maxJumps(arr parameter named arr OF list OF integer, parameter named d OF integer) RETURNS integer  
        FUNCTION dp(parameter named i OF integer) RETURNS integer  
            SET max_jumps TO one  
            FOR x FROM one TO d  
                IF the sum of i AND x IS LESS THAN the LENGTH OF arr THEN  
                    IF the element at position the sum of i AND x of arr IS GREATER THAN OR EQUAL TO the element at position i of arr THEN  
                        BREAK the loop  
                    END IF  
                    SET max_jumps TO the GREATER OF max_jumps AND the sum of one AND dp called with argument the sum of i AND x  
                END IF  
            END FOR  
            FOR x FROM one TO d  
                IF the difference of i AND x IS GREATER THAN OR EQUAL TO zero THEN  
                    IF the element at position the difference of i AND x of arr IS GREATER THAN OR EQUAL TO the element at position i of arr THEN  
                        BREAK the loop  
                    END IF  
                    SET max_jumps TO the GREATER OF max_jumps AND the sum of one AND dp called with argument the difference of i AND x  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        RETURN the GREATER OF dp called with argument i FOR each i FROM zero TO the difference of the LENGTH OF arr AND one  
    END FUNCTION  
END CLASS
