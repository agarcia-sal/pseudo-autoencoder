CLASS Solution  
    FUNCTION canVisitAllRooms(rooms)  
        SET visited TO empty set  
        SET stack TO list containing the first room number zero  
        WHILE stack is not empty  
            SET current_room TO element removed from the end of stack  
            ADD current_room TO visited  
            FOR each key IN the collection of keys at element at position current_room of rooms  
                IF key is NOT IN visited  
                    ADD key TO the end of stack  
                END IF  
            END FOR  
        END WHILE  
        RETURN the LENGTH OF visited EQUALS the LENGTH OF rooms  
    END FUNCTION  
END CLASS
