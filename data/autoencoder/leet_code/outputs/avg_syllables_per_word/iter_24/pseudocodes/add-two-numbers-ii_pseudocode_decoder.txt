CLASS ListNode  
    FUNCTION __init__ with parameters val NEXT  
        SET instance variable val TO val  
        SET instance variable next TO NEXT  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF the LENGTH OF values EQUALS zero  
        RETURN no value  
    END IF  
    SET head TO a new instance of ListNode initialized with the first element in values  
    SET p TO head  
    FOR each element val IN the elements of values from position two TO the end  
        SET node TO a new instance of ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 p2  
    IF p1 EQUALS no value AND p2 EQUALS no value  
        RETURN true  
    END IF  
    IF p1 EQUALS no value OR p2 EQUALS no value  
        RETURN false  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND the result of calling is_same_list with the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers with parameters l1 l2  

        FUNCTION pushIntoStack with parameter node  
            SET stack TO empty list  
            WHILE node EQUALS a value that is not no value  
                APPEND the value attribute of node TO stack  
                SET node TO the next attribute of node  
            END WHILE  
            RETURN stack  
        END FUNCTION  

        SET stack1 TO the result of pushIntoStack called with l1  
        SET stack2 TO the result of pushIntoStack called with l2  

        SET carry TO zero  
        SET result TO no value  

        WHILE the LENGTH OF stack1 NOT EQUALS zero OR the LENGTH OF stack2 NOT EQUALS zero OR carry NOT EQUALS zero  
            IF the LENGTH OF stack1 GREATER THAN zero  
                SET val1 TO the last element of stack1 removed from stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF the LENGTH OF stack2 GREATER THAN zero  
                SET val2 TO the last element of stack2 removed from stack2  
            ELSE  
                SET val2 TO zero  
            END IF  

            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET digit TO total MODULO ten  

            SET new_node TO a new instance of ListNode initialized with digit  
            SET the next attribute of new_node TO result  
            SET result TO new_node  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
