CLASS Solution  
    FUNCTION findPermutation WITH PARAMETER s  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s PLUS one MINUS one  
            APPEND index PLUS one TO result  
        END FOR  
        SET start TO zero  
        FOR end FROM zero TO the LENGTH OF s MINUS one  
            IF end EQUALS the LENGTH OF s OR the element at position end of s EQUALS the uppercase letter I  
                SET segment_to_reverse TO the list of elements from position start TO position end PLUS one of result  
                REVERSE segment_to_reverse  
                REPLACE elements from position start TO position end PLUS one of result WITH segment_to_reverse  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
