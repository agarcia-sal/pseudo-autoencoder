CLASS Solution  
    FUNCTION multiply(num1 num2)  
        IF num1 EQUALS string zero OR num2 EQUALS string zero  
            RETURN string zero  
        END IF  
        
        SET result TO a list of zeros with the LENGTH OF num1 PLUS the LENGTH OF num2  
        
        SET num1 TO the reverse order of characters in num1  
        SET num2 TO the reverse order of characters in num2  
        
        FOR i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the multiplication of (the numeric value of the character at position i of num1) MINUS zero AND (the numeric value of the character at position j of num2) MINUS zero  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS the element at position p1 of result  
                
                SET the element at position p1 of result TO sum_ MODULO ten  
                INCREMENT the element at position p2 of result BY sum_ DIVIDED BY ten  
            END FOR  
        END FOR  
        
        SET result_str TO the concatenation of the string representation of elements of result in reverse order  
        RETURN result_str with all leading string zeros removed  
    END FUNCTION  
END CLASS
