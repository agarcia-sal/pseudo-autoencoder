CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO plus  
            WHILE the LENGTH OF s is greater than zero  
                SET char TO element at position zero of s  
                REMOVE element at position zero FROM s  
                IF char is a digit character  
                    SET num TO num MULTIPLIED BY ten PLUS the numeric value of char  
                END IF  
                IF char EQUALS open parenthesis  
                    SET num TO the result of calling helper with s  
                END IF  
                IF char EQUALS plus OR char EQUALS minus OR char EQUALS multiplication symbol OR char EQUALS division symbol OR char EQUALS open parenthesis OR char EQUALS close parenthesis OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS plus  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus  
                        APPEND the negative of num TO stack  
                    ELSE IF sign EQUALS multiplication symbol  
                        SET the element at the last position of stack TO the element at the last position of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS division symbol  
                        SET the element at the last position of stack TO the integer result of the element at the last position of stack DIVIDED BY num with truncation towards zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS close parenthesis  
                    BREAK the WHILE loop  
                END IF  
            END WHILE  
            RETURN the sum of elements in stack  
        END FUNCTION  
        SET s TO the list of characters of s  
        RETURN the result of calling helper with s  
    END FUNCTION  
END CLASS
