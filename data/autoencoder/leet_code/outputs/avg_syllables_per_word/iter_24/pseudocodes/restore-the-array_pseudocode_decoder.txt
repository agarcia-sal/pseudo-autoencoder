CLASS Solution  
    FUNCTION numberOfArrays with parameters s and k  
        SET MODULO TO one billion PLUS seven  
        SET string_length TO the LENGTH OF s  
        INITIALIZE dynamic_programming_list AS a list of zero values WITH size string_length PLUS one  
        SET the element at position zero of dynamic_programming_list TO one  
        FOR index_i FROM one TO string_length  
            FOR index_j FROM zero TO index_i MINUS one  
                IF element at position index_j of s NOT EQUALS the character zero AND the integer VALUE OF the substring from position index_j PLUS one TO position index_i OF s IS GREATER THAN OR EQUAL TO one AND LESS THAN OR EQUAL TO k  
                    SET the element at position index_i of dynamic_programming_list TO the element at position index_i of dynamic_programming_list PLUS the element at position index_j of dynamic_programming_list MODULO MODULO  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position string_length of dynamic_programming_list  
    END FUNCTION  
END CLASS
