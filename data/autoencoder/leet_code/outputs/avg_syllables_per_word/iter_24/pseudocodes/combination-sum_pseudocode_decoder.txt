CLASS Solution  
    FUNCTION combinationSum(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                CALL backtrack WITH i AS start PARAMETER AND target MINUS the element at position i of candidates AS target PARAMETER AND path APPENDED WITH the element at position i of candidates AS path PARAMETER  
            END FOR  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH zero AS start PARAMETER AND target AS target PARAMETER AND an empty list AS path PARAMETER  
        RETURN result  
    END FUNCTION  
END CLASS
