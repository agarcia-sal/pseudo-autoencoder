CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each index i AND character char IN the enumeration of s  
                IF character char EQUALS the character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND the concatenation of the character one PLUS the result of dfs applied to the substring of s from position start PLUS one TO position i MINUS one PLUS the character zero TO specials  
                    SET start TO i PLUS one  
                END IF  
            END FOR  
            SORT specials in reverse lexicographical order  
            RETURN the concatenation of all elements in specials  
        END FUNCTION  
        RETURN the result of dfs applied to s  
    END FUNCTION  
END CLASS
