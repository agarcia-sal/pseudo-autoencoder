CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.left TO a new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None  
            SET node.right TO a new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF value at attribute val of p NOT EQUALS value at attribute val of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with arguments node at attribute left of p AND node at attribute left of q AND the result of calling is_same_tree with arguments node at attribute right of p AND node at attribute right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers(self, root)  
        FUNCTION dfs(node, current_number)  
            IF node EQUALS None  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS value at attribute val of node  
            IF node.left EQUALS None AND node.right EQUALS None  
                RETURN current_number  
            END IF  
            SET left_sum TO the result of calling dfs with arguments node at attribute left of node AND current_number  
            SET right_sum TO the result of calling dfs with arguments node at attribute right of node AND current_number  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN the result of calling dfs with arguments root AND zero  
    END FUNCTION  
END CLASS
