CLASS Solution  
    FUNCTION uniquePathsIII(grid)  
        FUNCTION dfs(x, y, steps)  
            IF the element at position x of grid EQUALS the element at position y EQUALS two THEN  
                RETURN true IF steps EQUALS zero OTHERWISE false  
            END IF  
            SET original TO the element at position x of grid AT position y  
            SET the element at position x of grid AT position y TO minus two  
            SET count TO zero  
            FOR each pair dx and dy IN the list of pairs consisting of negative one and zero AND one and zero AND zero and negative one AND zero and one  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF zero LESS THAN OR EQUAL TO nx AND nx LESS THAN m AND zero LESS THAN OR EQUAL TO ny AND ny LESS THAN n AND the element at position nx of grid AT position ny GREATER THAN OR EQUAL TO zero THEN  
                    INCREMENT count BY dfs WITH parameters nx ny steps MINUS one  
                END IF  
            END FOR  
            SET the element at position x of grid AT position y TO original  
            RETURN count  
        END FUNCTION  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero of grid  
        SET start_x TO zero  
        SET start_y TO zero  
        SET empty_squares TO one  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the element at position i of grid AT position j EQUALS one THEN  
                    SET start_x TO i  
                    SET start_y TO j  
                ELSE IF the element at position i of grid AT position j EQUALS zero THEN  
                    INCREMENT empty_squares BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN dfs WITH parameters start_x start_y empty_squares  
    END FUNCTION  
END CLASS
