CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        REMOVE the first element FROM queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.left TO a new TreeNode initialized WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none  
            SET node.right TO a new TreeNode initialized WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF the value of p NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree WITH the left child of p AND the left child of q AND the result of is_same_tree WITH the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues(self, root)  
        IF root EQUALS none  
            RETURN an empty list  
        END IF  
        SET queue TO a new double ended queue initialized WITH root  
        SET largest_values TO an empty list  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR each _ FROM one TO level_size  
                REMOVE the first element FROM queue AND ASSIGN TO node  
                IF max_value LESS THAN the value of node  
                    SET max_value TO the value of node  
                END IF  
                IF the left child of node NOT EQUALS none  
                    APPEND the left child of node TO queue  
                END IF  
                IF the right child of node NOT EQUALS none  
                    APPEND the right child of node TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
