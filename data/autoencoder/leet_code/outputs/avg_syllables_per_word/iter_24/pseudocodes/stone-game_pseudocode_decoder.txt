CLASS Solution  
    FUNCTION stoneGame WITH parameter piles  
        FUNCTION dfs WITH parameters i AND j  
            IF i GREATER THAN j THEN  
                RETURN zero  
            END IF  
            RETURN maximum OF the result of element at position i of piles MINUS the result of dfs CALLED WITH i INCREMENTED BY one AND j AND the result of element at position j of piles MINUS the result of dfs CALLED WITH i AND j DECREMENTED BY one  
        END FUNCTION  
        RETURN whether the result of dfs CALLED WITH zero AND the LENGTH OF piles MINUS one IS GREATER THAN zero  
    END FUNCTION  
END CLASS
