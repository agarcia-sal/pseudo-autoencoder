CLASS Solution  
    FUNCTION checkSubarraySum(nums k)  
        SET remainder_map TO dictionary with key zero and value negative one  
        SET current_sum TO zero  
        FOR each index i AND each element num IN nums  
            INCREMENT current_sum BY num  
            IF k NOT EQUALS zero  
                SET current_sum TO the remainder after dividing current_sum BY k  
            END IF  
            IF current_sum IS a key IN remainder_map  
                IF the DIFFERENCE between i AND the value at key current_sum IN remainder_map IS GREATER THAN one  
                    RETURN True  
                END IF  
            ELSE  
                SET the value at key current_sum IN remainder_map TO i  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
