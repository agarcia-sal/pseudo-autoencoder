CLASS Solution  
    FUNCTION countPrimes(n)  
        IF n LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
        
        SET is_prime TO list of Boolean true repeated n times  
        SET element at position zero of is_prime TO Boolean false  
        SET element at position one of is_prime TO Boolean false  
        
        FOR start FROM two TO the integer part of the square root of n PLUS one  
            IF element at position start of is_prime EQUALS Boolean true  
                FOR multiple FROM start MULTIPLIED BY start TO n MINUS one IN STEPS OF start  
                    SET element at position multiple of is_prime TO Boolean false  
                END FOR  
            END IF  
        END FOR  
        
        SET total_primes TO zero  
        FOR each prime_flag IN is_prime  
            IF prime_flag EQUALS Boolean true  
                INCREMENT total_primes BY one  
            END IF  
        END FOR  
        
        RETURN total_primes  
    END FUNCTION  
END CLASS
