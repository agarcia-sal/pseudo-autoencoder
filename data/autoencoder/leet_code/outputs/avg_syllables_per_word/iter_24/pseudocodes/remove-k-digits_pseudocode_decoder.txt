CLASS Solution  
    FUNCTION removeKdigits(num parameter OF TYPE string, k parameter OF TYPE integer) RETURNS string  
        SET stack TO an empty list  
        FOR each digit_element IN num parameter  
            WHILE k parameter GREATER THAN zero AND stack IS NOT empty AND the element at the last position of stack GREATER THAN digit_element  
                REMOVE the element at the last position of stack FROM stack  
                DECREMENT k parameter BY one  
            END WHILE  
            APPEND digit_element TO stack  
        END FOR  
        IF k parameter EQUALS zero  
            SET final_stack TO stack  
        ELSE  
            SET final_stack TO all elements of stack EXCEPT the last k parameter elements  
        END IF  
        SET result TO the concatenation of all elements in final_stack WITH leading zeros REMOVED  
        IF result IS NOT empty  
            RETURN result  
        ELSE  
            RETURN the string zero  
        END IF  
    END FUNCTION  
END CLASS
