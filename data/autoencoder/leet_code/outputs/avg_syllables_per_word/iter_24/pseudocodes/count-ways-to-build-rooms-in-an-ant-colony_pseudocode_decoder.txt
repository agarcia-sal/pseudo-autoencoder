CLASS Solution  
    FUNCTION waysToBuildRooms(prevRoom)  
        SET n TO the LENGTH OF prevRoom  
        ASSIGN graph TO a NEW mapping from element TO list of elements with default empty list  
        
        FOR room FROM one TO n MINUS one  
            APPEND room TO the list at element at position room of prevRoom in graph  
        END FOR  
        
        FUNCTION dp(node)  
            IF the list at element node in graph EQUALS empty THEN  
                RETURN one AND one  
            END IF  
            
            SET total_ways TO one  
            SET total_size TO one  
            ASSIGN child_results TO an empty list  
            
            FOR each child IN the list at element node in graph  
                SET ways AND size TO the result of dp with parameter child  
                APPEND the pair of ways AND size TO child_results  
                INCREMENT total_size BY size  
            END FOR  
            
            SET current_size TO one  
            FOR each pair ways AND size IN child_results  
                SET total_ways TO total_ways MULTIPLIED BY the combination of current_size PLUS size MINUS one CHOOSE size MULTIPLIED BY ways MODULO MOD  
                INCREMENT current_size BY size  
            END FOR  
            
            RETURN total_ways AND total_size  
        END FUNCTION  
        
        RETURN the first element of the result of dp with parameter zero  
    END FUNCTION  
END CLASS
