CLASS ListNode  
    FUNCTION __init__ (self val EQUALS zero next EQUALS None)  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node (values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO a new instance of ListNode constructed with the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values starting from position two to the last position  
        SET node TO a new instance of ListNode constructed with val  
        SET the next reference of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list (p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the condition that the val attribute of p1 EQUALS the val attribute of p2 AND the result of calling is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne (self head)  
        FUNCTION reverseList (node)  
            SET prev TO None  
            WHILE node HOLDS  
                SET next_node TO the next attribute of node  
                SET the next attribute of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO the result of calling reverseList with head  
        
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current HOLDS AND carry HOLDS  
            SET the val attribute of current TO the val attribute of current PLUS carry  
            SET carry TO the val attribute of current DIVIDED BY ten  
            SET the val attribute of current TO the val attribute of current MODULO ten  
            
            IF the next attribute of current EQUALS None AND carry HOLDS  
                SET the next attribute of current TO a new instance of ListNode constructed with carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO the next attribute of current  
        END WHILE  
        
        RETURN the result of calling reverseList with reversed_head  
    END FUNCTION  
END CLASS
