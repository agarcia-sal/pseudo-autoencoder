CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO the set constructed from the elements of bank  
        SET q TO a double-ended queue initialized with a tuple consisting of start and zero  
        SET mp TO a mapping where the key A maps to the sequence of characters T C G the key T maps to the sequence of characters A C G the key C maps to the sequence of characters A T G the key G maps to the sequence of characters A T C  
        WHILE the queue q is not empty  
            REMOVE and ASSIGN the first element of queue q to t and step  
            IF the value t EQUALS the value end  
                RETURN step  
            END IF  
            FOR each index i and character v at position i IN the sequence t  
                FOR each character j IN the sequence mapped by key v in mp  
                    SET next TO the concatenation of the substring from position zero TO position i minus one of t PLUS character j PLUS the substring from position i plus one TO the end of t  
                    IF next is an element of the set s  
                        APPEND a tuple consisting of next and step PLUS one TO the queue q  
                        REMOVE element next FROM the set s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN the integer negative one  
    END FUNCTION  
END CLASS
