CLASS Solution  
    FUNCTION findReplaceString(s parameter, indices parameter, sources parameter, targets parameter)  
        SET replacements TO the LIST OF tuples WHERE each tuple CONTAINS one element FROM indices ONE-TO-ONE WITH one element FROM sources ONE-TO-ONE WITH one element FROM targets SORTED BY the first element OF each tuple  
        SET result TO an EMPTY list  
        SET prev_end TO zero  
        FOR each tuple IN replacements  
            DESTRUCTURE tuple INTO index variable AND source variable AND target variable  
            APPEND the SUBSTRING OF s FROM position prev_end TO position index MINUS one TO result  
            IF the SUBSTRING OF s FROM position index TO position index PLUS the LENGTH OF source EQUALS source THEN  
                APPEND target TO result  
            ELSE  
                APPEND the SUBSTRING OF s FROM position index TO position index PLUS the LENGTH OF source TO result  
            END IF  
            SET prev_end TO index PLUS the LENGTH OF source  
        END FOR  
        APPEND the SUBSTRING OF s FROM position prev_end TO the END OF s TO result  
        RETURN the CONCATENATION OF all elements OF result INTO a SINGLE string  
    END FUNCTION  
END CLASS
