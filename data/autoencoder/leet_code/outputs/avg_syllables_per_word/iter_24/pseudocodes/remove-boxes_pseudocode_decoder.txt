CLASS Solution  
    FUNCTION removeBoxes(boxes)  
        FUNCTION dp(l, r, k, memo)  
            IF l GREATER THAN r  
                RETURN zero  
            END IF  
            IF the tuple of l r k EXISTS IN memo  
                RETURN the value ASSOCIATED WITH the tuple l r k IN memo  
            END IF  
            WHILE r GREATER THAN l AND element at position r of boxes EQUALS element at position r MINUS one of boxes  
                DECREMENT r BY one  
                INCREMENT k BY one  
            END WHILE  
            SET res TO the CALL TO dp WITH parameters l r MINUS one zero memo PLUS the product of the sum of k PLUS one AND the sum of k PLUS one  
            FOR i FROM l TO r MINUS one  
                IF element at position i of boxes EQUALS element at position r of boxes  
                    SET res TO the GREATER OF res AND the sum of the CALL TO dp WITH parameters l i the sum of k PLUS one memo AND the CALL TO dp WITH parameters i PLUS one r MINUS one zero memo  
                END IF  
            END FOR  
            SET the tuple of l r k IN memo TO res  
            RETURN res  
        END FUNCTION  
        RETURN the CALL TO dp WITH parameters zero the LENGTH OF boxes MINUS one zero an empty dictionary  
    END FUNCTION  
END CLASS
