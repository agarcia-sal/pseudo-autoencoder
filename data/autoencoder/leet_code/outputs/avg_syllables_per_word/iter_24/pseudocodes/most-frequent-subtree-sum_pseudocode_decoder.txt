CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty list THEN  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value element at position one of values and left as None and right as None  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty DO  
        SET node TO the first element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None THEN  
            SET the left attribute of node TO a new TreeNode with value element at position i of values and left as None and right as None  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values NOT EQUALS None THEN  
            SET the right attribute of node TO a new TreeNode with value element at position i of values and left as None and right as None  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node EQUALS None THEN  
                RETURN zero  
            END IF  
            SET current_sum TO the val attribute of node PLUS subtree_sum(the left attribute of node) PLUS subtree_sum(the right attribute of node)  
            INCREMENT the value of count at key current_sum BY one  
            RETURN current_sum  
        END FUNCTION  
        
        SET count TO a new mapping with default integer zero  
        CALL subtree_sum with argument root  
        
        SET max_freq TO the maximum value among the values in count with a default of zero  
        
        SET result TO an empty list  
        FOR each key and freq IN the items of count DO  
            IF freq EQUALS max_freq THEN  
                APPEND key TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
