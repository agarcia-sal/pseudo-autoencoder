CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS plus sign  
                        APPEND l PLUS r TO results  
                    ELSE IF operator EQUALS minus sign  
                        APPEND l MINUS r TO results  
                    ELSE IF operator EQUALS multiplication sign  
                        APPEND l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  
        
        FUNCTION helper(sub_expr)  
            IF sub_expr CONSISTS ONLY OF DIGITS  
                RETURN list containing integer value of sub_expr  
            END IF  
            
            SET results TO empty list  
            FOR each index i FROM zero TO length of sub_expr MINUS one  
                IF character at position i of sub_expr EQUALS plus sign OR character at position i of sub_expr EQUALS minus sign OR character at position i of sub_expr EQUALS multiplication sign  
                    SET left_results TO result of helper function called with substring from start position to position before i of sub_expr  
                    SET right_results TO result of helper function called with substring from position after i to end position of sub_expr  
                    APPEND all elements from result of compute called with left_results, right_results, and character at position i of sub_expr TO results  
                END IF  
            END FOR  
            
            RETURN results  
        END FUNCTION  
        
        RETURN result of helper function called with expression  
    END FUNCTION  
END CLASS
