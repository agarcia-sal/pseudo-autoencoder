CLASS Solution  
    FUNCTION findMaxValueOfEquation(points PARAMETERS points AND k)  
        SET max_value TO negative infinity  
        ASSIGN reference TO an empty double ended queue named deque_  
        
        FOR each pair xj AND yj IN points  
            WHILE the double ended queue named deque_ IS NOT empty AND the result of subtracting the second element of the front pair in deque_ from xj IS GREATER THAN k  
                REMOVE the front element FROM the double ended queue named deque_  
            END WHILE  
            
            IF the double ended queue named deque_ IS NOT empty  
                SET max_value TO the GREATER OF max_value AND the result of adding yj TO xj PLUS the first element of the front pair in deque_  
            END IF  
            
            WHILE the double ended queue named deque_ IS NOT empty AND the result of subtracting xj FROM yj IS GREATER THAN OR EQUAL TO the first element of the last pair in deque_  
                REMOVE the last element FROM the double ended queue named deque_  
            END WHILE  
            
            APPEND the pair of the result of subtracting xj FROM yj AND xj TO the double ended queue named deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
