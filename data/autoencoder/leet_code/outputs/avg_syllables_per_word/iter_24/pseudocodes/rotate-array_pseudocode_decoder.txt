CLASS Solution  
    FUNCTION rotate(nums k)  
        SET n TO the LENGTH OF nums  
        SET k TO k MODULO n  

        FUNCTION reverse(start end)  
            WHILE start LESS THAN end  
                SET temporary TO element at position start of nums  
                SET element at position start of nums TO element at position end of nums  
                SET element at position end of nums TO temporary  
                INCREMENT start BY one  
                DECREMENT end BY one  
            END WHILE  
        END FUNCTION  

        CALL reverse WITH arguments zero n MINUS one  
        CALL reverse WITH arguments zero k MINUS one  
        CALL reverse WITH arguments k n MINUS one  
    END FUNCTION  
END CLASS
