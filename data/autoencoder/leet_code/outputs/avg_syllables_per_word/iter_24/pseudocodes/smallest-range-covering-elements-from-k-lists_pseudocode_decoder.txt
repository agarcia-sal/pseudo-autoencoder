CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty list  
        SET max_value TO value less than any number  
          
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            CALL push INTO min_heap THE element at position zero of the list at position index in nums THE index THE zero  
            SET max_value TO THE GREATER THAN VALUE BETWEEN max_value AND the element at position zero of the list at position index in nums  
        END FOR  
          
        SET smallest_range TO value greater than any number  
        SET result TO a list containing value less than any reasonable minimum AND value greater than any reasonable maximum  
          
        WHILE min_heap is not empty  
            CALL pop FROM min_heap AND UNPACK INTO min_value list_index element_index  
              
            IF max_value MINUS min_value LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO a list containing min_value AND max_value  
            END IF  
              
            IF element_index PLUS one LESS THAN the LENGTH OF the list at position list_index in nums  
                SET next_value TO the element at position element_index PLUS one of the list at position list_index in nums  
                CALL push INTO min_heap THE next_value list_index element_index PLUS one  
                SET max_value TO THE GREATER THAN VALUE BETWEEN max_value AND next_value  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
          
        RETURN result  
    END FUNCTION  
END CLASS
