CLASS Solution  
    FUNCTION countSubgraphsForEachDiameter(n PARAMETER, edges PARAMETER)  
        SET graph TO an empty mapping from integer keys to empty lists initialized for each integer from one TO n INCLUSIVE  
        FOR each pair of nodes u and v IN edges  
            APPEND v TO the list mapped by u in graph  
            APPEND u TO the list mapped by v in graph  
        END FOR  

        FUNCTION bfs(subtree PARAMETER)  
            SET start TO an arbitrary element from subtree  
            SET queue TO a new deque containing start  
            SET visited TO a new set containing start  
            SET distance TO zero  
            SET farthest_node TO start  

            WHILE queue holds elements  
                SET node TO the element removed from the front of queue  
                FOR each neighbor IN the list mapped by node in graph  
                    IF neighbor is an element of subtree AND neighbor is not an element of visited  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                        SET farthest_node TO neighbor  
                    END IF  
                END FOR  
            END WHILE  

            SET queue TO a new deque containing farthest_node  
            SET visited TO a new set containing farthest_node  
            SET max_distance TO zero  

            WHILE queue holds elements  
                SET size TO the number of elements in queue  
                FOR count FROM one TO size INCLUSIVE  
                    SET node TO the element removed from the front of queue  
                    FOR each neighbor IN the list mapped by node in graph  
                        IF neighbor is an element of subtree AND neighbor is not an element of visited  
                            ADD neighbor TO visited  
                            APPEND neighbor TO queue  
                        END IF  
                    END FOR  
                END FOR  
                IF queue holds elements  
                    INCREMENT max_distance BY one  
                END IF  
            END WHILE  

            RETURN max_distance  
        END FUNCTION  

        SET result TO a list of zeroes of length n MINUS one  

        FOR size FROM two TO n INCLUSIVE  
            FOR subset IN all combinations of integers from one TO n INCLUSIVE taken size at a time  
                SET visited TO an empty set  
                SET stack TO a list containing the first element of subset  
                WHILE stack holds elements  
                    SET node TO the last element removed from stack  
                    IF node is not an element of visited  
                        ADD node TO visited  
                        FOR each neighbor IN the list mapped by node in graph  
                            IF neighbor is an element of subset AND neighbor is not an element of visited  
                                APPEND neighbor TO stack  
                            END IF  
                        END FOR  
                    END IF  
                END WHILE  

                IF visited equals the set of elements in subset  
                    SET diameter TO the result of calling bfs with the set converted from subset  
                    IF diameter is greater than zero  
                        INCREMENT the element at position diameter MINUS one of result BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
