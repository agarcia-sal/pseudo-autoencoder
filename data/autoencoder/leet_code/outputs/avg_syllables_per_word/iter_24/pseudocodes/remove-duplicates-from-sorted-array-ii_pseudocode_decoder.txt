CLASS Solution  
    FUNCTION removeDuplicates(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        SET write_index TO zero  
        FOR i FROM one TO the LESSER OF two AND the LENGTH OF nums  
            SET element at position write_index of nums TO element at position i of nums  
            INCREMENT write_index BY one  
        END FOR  
        FOR i FROM three TO the LENGTH OF nums  
            IF element at position i of nums NOT EQUALS element at position write_index MINUS two of nums  
                SET element at position write_index of nums TO element at position i of nums  
                INCREMENT write_index BY one  
            END IF  
        END FOR  
        RETURN write_index  
    END FUNCTION  
END CLASS
