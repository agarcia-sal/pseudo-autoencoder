CLASS Solution  
    FUNCTION jobScheduling(startTime PARAMETER, endTime PARAMETER, profit PARAMETER)  
        SET jobs TO the list formed by combining startTime elements with corresponding endTime elements and corresponding profit elements sorted by the endTime value of each combined element  
        SET end_times TO the list of endTime elements extracted from each element of jobs  
        SET dp TO a list of zeros with the length equal to the number of elements in jobs PLUS one  
        FOR i FROM one TO the number of elements in jobs  
            SET current_start TO the startTime element of the element at the position i MINUS one of jobs  
            SET j TO the position found by searching the location to insert current_start into end_times maintaining sorted order from the right  
            SET dp element at position i TO the greater value BETWEEN dp element at position i MINUS one AND the summation of dp element at position j PLUS the profit element of the element at the position i MINUS one of jobs  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
