CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of size n PLUS one BY n PLUS one filled with zeroes  
        
        FOR j FROM zero TO n  
            SET element at position j of the list at position zero of dp TO one  
        END FOR  
        
        FOR i FROM one TO n  
            SET prefix_sum TO zero  
            IF element at position i MINUS one of s EQUALS the uppercase letter I THEN  
                FOR j FROM zero TO n PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position j of the list at position i MINUS one of dp MODULO MOD  
                    SET element at position j of the list at position i of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO prefix_sum PLUS element at position j PLUS one of the list at position i MINUS one of dp MODULO MOD  
                    SET element at position j of the list at position i of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result TO zero  
        FOR each value IN the list at position n of dp  
            INCREMENT result BY value  
        END FOR  
        SET result TO result MODULO MOD  
        
        RETURN result  
    END FUNCTION  
END CLASS
