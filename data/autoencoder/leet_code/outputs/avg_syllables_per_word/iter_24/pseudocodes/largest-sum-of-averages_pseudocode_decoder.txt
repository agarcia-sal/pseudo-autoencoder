CLASS Solution  
    FUNCTION largestSumOfAverages(nums AS list OF integers, k AS integer) RETURNS float  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of zeros with the LENGTH OF n PLUS one  
        FOR i FROM zero TO n MINUS one  
            SET element at position i PLUS one of prefix_sum TO element at position i of prefix_sum PLUS element at position i of nums  
        END FOR  
        
        FUNCTION average(i AS integer, j AS integer) RETURNS float  
            SET numerator TO element at position j PLUS one of prefix_sum MINUS element at position i of prefix_sum  
            SET denominator TO j MINUS i PLUS one  
            RETURN numerator DIVIDED BY denominator  
        END FUNCTION  
        
        SET dp TO a two-dimensional list of zeros with dimensions n PLUS one BY k PLUS one  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AT position one of dp TO average(i, n MINUS one)  
        END FOR  
        
        FOR i FROM n MINUS one DOWNTO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    SET current_value TO element at position i AT position partitions of dp  
                    SET candidate_value TO element at position j AT position partitions MINUS one of dp PLUS average(i, j MINUS one)  
                    IF candidate_value GREATER THAN current_value THEN  
                        SET element at position i AT position partitions of dp TO candidate_value  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero AT position k of dp  
    END FUNCTION  
END CLASS
