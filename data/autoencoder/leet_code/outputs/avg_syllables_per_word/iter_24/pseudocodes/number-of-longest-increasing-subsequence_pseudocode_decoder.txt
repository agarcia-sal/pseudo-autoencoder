CLASS Solution  
    FUNCTION findNumberOfLIS(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF nums  
        SET lengths TO a list of the value one REPEATED n times  
        SET counts TO a list of the value one REPEATED n times  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF the element at position i of nums GREATER THAN the element at position j of nums THEN  
                    IF the element at position j of lengths PLUS one GREATER THAN the element at position i of lengths THEN  
                        SET the element at position i of lengths TO the element at position j of lengths PLUS one  
                        SET the element at position i of counts TO the element at position j of counts  
                    ELSE IF the element at position j of lengths PLUS one EQUALS the element at position i of lengths THEN  
                        INCREMENT the element at position i of counts BY the element at position j of counts  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET longest TO the maximum value in lengths  
        SET result TO zero  
        FOR i FROM zero TO n MINUS one  
            IF the element at position i of lengths EQUALS longest THEN  
                INCREMENT result BY the element at position i of counts  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
