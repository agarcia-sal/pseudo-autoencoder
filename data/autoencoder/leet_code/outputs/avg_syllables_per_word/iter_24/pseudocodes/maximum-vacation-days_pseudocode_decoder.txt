CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF element at position zero of days  
        
        SET dp TO a list of n elements each SET TO negative one  
        SET element at position zero of dp TO zero  
        
        SET graph TO a mapping from each integer i in range from zero TO n MINUS one TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF element at position i of flights EQUALS one at element at position j THEN  
                    APPEND j TO element at position i of graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR each week FROM zero TO k MINUS one  
            SET new_dp TO a list of n elements each SET TO negative one  
            FOR each i FROM zero TO n MINUS one  
                IF element at position i of dp EQUALS negative one THEN  
                    CONTINUE  
                END IF  
                SET element at position i of new_dp TO the GREATER OF element at position i of new_dp AND element at position i of dp PLUS element at position i of days PLUS element at position week of days  
                FOR each j IN element at position i of graph  
                    SET element at position j of new_dp TO the GREATER OF element at position j of new_dp AND element at position i of dp PLUS element at position j of days PLUS element at position week of days  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the GREATEST element in dp  
    END FUNCTION  
END CLASS
