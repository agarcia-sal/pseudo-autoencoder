CLASS Solution  
    FUNCTION replaceNonCoprimes(nums)  
        SET stack TO empty list  
        FOR each num IN nums  
            WHILE the LENGTH OF stack GREATER THAN zero  
                SET g TO greatest common divisor of element at position the LENGTH OF stack MINUS one of stack AND num  
                IF g EQUALS one  
                    BREAK the while loop  
                END IF  
                SET num TO element at position the LENGTH OF stack MINUS one of stack MULTIPLIED BY num DIVIDED BY g  
                REMOVE the last element FROM stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
