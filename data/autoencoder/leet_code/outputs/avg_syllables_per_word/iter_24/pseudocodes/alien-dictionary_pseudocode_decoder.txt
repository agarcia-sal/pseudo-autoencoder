CLASS Solution  
    FUNCTION alienOrder(words)  
        SET graph TO a dictionary that returns an empty set for missing keys  
        SET indegree TO a dictionary that returns zero for missing keys  
        SET all_chars TO a set of all characters contained in all words combined into one sequence  
        FOR index FROM zero TO the length of words MINUS one MINUS one  
            SET word1 TO the element at position index of words  
            SET word2 TO the element at position index PLUS one of words  
            SET min_length TO the smaller value between the length of word1 AND the length of word2  
            SET found TO false  
            FOR position FROM zero TO min_length MINUS one  
                IF the element at position position of word1 IS NOT EQUALS the element at position position of word2  
                    IF the element at position position of word2 IS NOT IN the set mapped by the element at position position of word1 in graph  
                        ADD the element at position position of word2 TO the set mapped by the element at position position of word1 in graph  
                        INCREMENT indegree at the element at position position of word2 BY one  
                    END IF  
                    SET found TO true  
                    BREAK the inner FOR loop  
                END IF  
            END FOR  
            IF found IS false AND the length of word1 IS GREATER THAN the length of word2  
                RETURN an empty string  
            END IF  
        END FOR  
        SET queue TO a double-ended queue initialized with all elements in all_chars for which indegree at that element EQUALS zero  
        SET result TO an empty list  
        WHILE queue IS NOT empty  
            SET char TO the element removed from the left end of queue  
            APPEND char TO result  
            FOR each neighbor IN the set mapped by char in graph  
                DECREMENT indegree at neighbor BY one  
                IF indegree at neighbor EQUALS zero  
                    ADD neighbor TO the right end of queue  
                END IF  
            END FOR  
        END WHILE  
        IF the length of result IS NOT EQUALS the length of all_chars  
            RETURN an empty string  
        END IF  
        RETURN the concatenation of all elements in result into a single string  
    END FUNCTION  
END CLASS
