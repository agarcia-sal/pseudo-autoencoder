CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO list of one thousand plus nine hundred plus five hundred plus four hundred plus one hundred plus ninety plus fifty plus forty plus ten plus nine plus five plus four plus one  
        SET syms TO list of capital M plus capital C capital M plus capital D plus capital C capital D plus capital C plus capital X capital C plus capital L plus capital X L plus capital X plus capital I X plus capital V plus capital I V plus capital I  
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero  
            FOR placeholder FROM zero TO the quotient of num DIVIDED BY element at position i of val MINUS one  
                APPEND element at position i of syms TO roman_numeral  
                DECREMENT num BY element at position i of val  
            END FOR

        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
