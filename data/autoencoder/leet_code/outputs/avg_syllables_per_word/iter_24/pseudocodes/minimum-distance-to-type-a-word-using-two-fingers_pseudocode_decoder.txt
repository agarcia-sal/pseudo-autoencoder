CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS no value  
                RETURN zero  
            END IF  
            SET x1 TO the result of integer division of the Unicode code point of c1 MINUS the Unicode code point of capital letter A BY six  
            SET y1 TO the result of the remainder of the Unicode code point of c1 MINUS the Unicode code point of capital letter A DIVIDED BY six  
            SET x2 TO the result of integer division of the Unicode code point of c2 MINUS the Unicode code point of capital letter A BY six  
            SET y2 TO the result of the remainder of the Unicode code point of c2 MINUS the Unicode code point of capital letter A DIVIDED BY six  
            RETURN the absolute value of x1 MINUS x2 PLUS the absolute value of y1 MINUS y2  
        END FUNCTION  

        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the length of word  
                RETURN zero  
            END IF  
            SET c TO the element at position i of word  
            SET d1 TO the result of distance applied to f1 and c PLUS the result of dp applied to i PLUS one and c and f2  
            SET d2 TO the result of distance applied to f2 and c PLUS the result of dp applied to i PLUS one and f1 and c  
            RETURN the smaller value between d1 and d2  
        END FUNCTION  

        RETURN dp applied to zero and no value and no value  
    END FUNCTION  
END CLASS
