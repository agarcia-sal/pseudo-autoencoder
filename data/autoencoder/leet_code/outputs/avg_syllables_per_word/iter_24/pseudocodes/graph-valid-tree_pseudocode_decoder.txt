CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF the LENGTH OF edges NOT EQUALS n MINUS one  
            RETURN false  
        END IF  
        SET adj_list TO a new mapping from each integer i FROM zero TO n MINUS one TO an empty list  
        FOR each pair a and b IN edges  
            APPEND b TO the list mapped from a IN adj_list  
            APPEND a TO the list mapped from b IN adj_list  
        END FOR  
        SET visited TO an empty set  
        FUNCTION dfs(node)  
            IF node IS IN visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list mapped from node IN adj_list  
                CALL dfs neighbor  
            END FOR  
        END FUNCTION  
        CALL dfs with argument zero  
        RETURN the LENGTH OF visited EQUALS n  
    END FUNCTION  
END CLASS
