CLASS Solution  
    FUNCTION maxValue(events PARAMETER, k PARAMETER)  
        SORT events BY the end day of each event  
        SET n TO the LENGTH OF events  
        INITIALIZE dp AS a two dimensional collection of zeros with n PLUS one rows and k PLUS one columns  
        FOR i FROM one TO n  
            SET start TO the first element of the event at position i MINUS one of events  
            SET end TO the second element of the event at position i MINUS one of events  
            SET value TO the third element of the event at position i MINUS one of events  
            SET j TO the position of the first event in events whose end day is GREATER THAN OR EQUAL TO start by binary searching on the end day  
            FOR l FROM one TO k  
                SET dp at position i and l TO the GREATER OF dp at position i MINUS one and l AND dp at position j and l MINUS one PLUS value  
            END FOR  
        END FOR  
        RETURN dp at position n and k  
    END FUNCTION  
END CLASS
