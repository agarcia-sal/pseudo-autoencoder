CLASS Solution  
    FUNCTION maxSubArrayLen(nums k)  
        SET cumulative_sum_index TO a dictionary with element zero mapped to negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        FOR each index i AND element num IN the enumeration of nums  
            SET cumulative_sum TO cumulative_sum PLUS num  
            IF the value of cumulative_sum MINUS k is present as a key in cumulative_sum_index  
                SET max_length TO the greater value between max_length AND the difference of i MINUS the value at key cumulative_sum MINUS k in cumulative_sum_index  
            END IF  
            IF the key cumulative_sum is not present in cumulative_sum_index  
                SET the value at key cumulative_sum in cumulative_sum_index TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
