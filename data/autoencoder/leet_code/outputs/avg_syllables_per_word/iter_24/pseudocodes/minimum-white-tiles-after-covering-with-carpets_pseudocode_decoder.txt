CLASS Solution  
    FUNCTION minimumWhiteTiles(floor numCarpets carpetLen)  
        SET n TO the LENGTH OF floor  
        SET dp TO a new two dimensional list with n PLUS one rows and numCarpets PLUS one columns all initialized to zero  
        FOR i FROM one TO n  
            SET dp at position i at position zero TO dp at position i MINUS one at position zero PLUS one IF the element at position i MINUS one of floor EQUALS one ELSE zero  
        END FOR  
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp at position i at position j TO dp at position i MINUS one at position j PLUS one IF the element at position i MINUS one of floor EQUALS one ELSE zero  
                SET dp at position i at position j TO the LESSER OF dp at position i at position j AND dp at position the GREATER OF zero AND i MINUS carpetLen at position j MINUS one  
            END FOR  
        END FOR  
        RETURN dp at position n at position numCarpets  
    END FUNCTION  
END CLASS
