CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO the LIST OF pairs zero and one one and zero zero and negative one negative one and zero  
        SET min_heap TO the LIST containing the TRIPLE zero zero zero representing the cost and coordinates  
        SET visited TO the SET containing the pair zero and zero  
        
        WHILE min_heap is not EMPTY  
            REMOVE the element with the SMALLEST cost from min_heap INTO cost x y  
            
            IF the pair x and y EQUALS the pair m MINUS one and n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each pair dx and dy IN directions  
                SET nx TO the SUM of x PLUS dx  
                SET ny TO the SUM of y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND the pair nx and ny NOT IN visited  
                    ADD the pair nx and ny TO visited  
                    SET new_cost TO the SUM of cost PLUS the element at position nx of grid at position ny  
                    ADD the TRIPLE new_cost nx ny TO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the NEGATIVE value one  
    END FUNCTION  
END CLASS
