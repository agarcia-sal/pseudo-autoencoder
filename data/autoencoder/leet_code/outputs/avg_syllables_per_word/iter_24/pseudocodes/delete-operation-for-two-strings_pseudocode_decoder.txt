CLASS Solution  
    FUNCTION minDistance(word1 word one AS string, word2 word two AS string) RETURNS integer  
        SET m TO the LENGTH OF word one  
        SET n TO the LENGTH OF word two  
        SET dp TO a two dimensional list of zeroes with the number of rows EQUAL TO m PLUS one and the number of columns EQUAL TO n PLUS one  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one of word one EQUALS the element at position j MINUS one of word two  
                    SET the element at position i and position j of dp TO the element at position i MINUS one and position j MINUS one of dp PLUS one  
                ELSE  
                    SET the element at position i and position j of dp TO the GREATER VALUE BETWEEN the element at position i MINUS one and position j of dp AND the element at position i and position j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        SET lcs_length TO the element at position m and position n of dp  
        SET result TO m MINUS lcs_length PLUS n MINUS lcs_length  
        RETURN result  
    END FUNCTION  
END CLASS
