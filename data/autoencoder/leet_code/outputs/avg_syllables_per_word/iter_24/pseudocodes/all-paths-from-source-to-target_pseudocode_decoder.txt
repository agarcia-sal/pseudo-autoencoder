CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        SET q TO a queue initialized with a list containing zero  
        SET ans TO an empty list  
        WHILE the queue q is not empty  
            REMOVE the first element of q and ASSIGN it TO path  
            SET u TO the element at the last position of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the loop  
            END IF  
            FOR each element v IN the element at position u of graph  
                APPEND a new list consisting of path followed by v TO q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
