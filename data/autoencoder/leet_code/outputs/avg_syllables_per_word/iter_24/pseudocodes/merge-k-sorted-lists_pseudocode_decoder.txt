CLASS ListNode  
    FUNCTION __init__(self val SET TO zero next SET TO none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO a new ListNode INITIALIZED WITH the element at position one of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position two TO the end  
        SET node TO a new ListNode INITIALIZED WITH val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND is_same_list(the next element of p1 the next element of p2)  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists(self lists)  
        SET min_heap TO an empty list  
        SET i TO zero  
        FOR each lst IN lists WITH index i FROM zero TO the LENGTH OF lists MINUS one  
            IF lst NOT EQUALS none  
                CALL the heap push operation TO add a tuple of the value of lst the index i and lst TO min_heap  
            END IF  
        END FOR  
        SET dummy TO a new ListNode INITIALIZED WITH default parameters  
        SET current TO dummy  
        WHILE the LENGTH OF min_heap GREATER THAN zero  
            CALL the heap pop operation TO remove and return the smallest tuple FROM min_heap AND ASSIGN TO val i node  
            SET the next element of current TO node  
            SET current TO the next element of current  
            IF the next element of node NOT EQUALS none  
                CALL the heap push operation TO add a tuple of the value of the next element of node the index i and the next element of node TO min_heap  
            END IF  
        END WHILE  
        RETURN the next element of dummy  
    END FUNCTION  
END CLASS
