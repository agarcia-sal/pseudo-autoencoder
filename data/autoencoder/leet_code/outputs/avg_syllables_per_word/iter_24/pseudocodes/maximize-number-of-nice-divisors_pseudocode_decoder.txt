CLASS Solution  
    FUNCTION maxNiceDivisors(primeFactors)  
        SET MOD TO ten raised to the power of nine PLUS one  
        
        IF primeFactors LESS THAN OR EQUAL TO three THEN  
            RETURN primeFactors  
        END IF  
        
        SET num_threes TO the INTEGER DIVISION of primeFactors BY three  
        SET remainder TO the MODULO of primeFactors BY three  
        
        IF remainder EQUALS zero THEN  
            RETURN the result of raising three to the power num_threes modulo MOD  
        ELSE IF remainder EQUALS one THEN  
            RETURN the result of raising three to the power num_threes MINUS one modulo MOD MULTIPLIED BY four modulo MOD  
        ELSE  
            RETURN the result of raising three to the power num_threes modulo MOD MULTIPLIED BY two modulo MOD  
        END IF  
    END FUNCTION  
END CLASS
