CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO A new instance of TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left side of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO A new instance of TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO A new instance of TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST(self, root)  
        SET self.max_sum TO zero  
        
        FUNCTION helper(node)  
            IF node IS None  
                RETURN True, zero, positive infinity, negative infinity  
            END IF  
            
            SET left_is_bst, left_sum, left_min, left_max TO helper(node.left)  
            SET right_is_bst, right_sum, right_min, right_max TO helper(node.right)  
            
            IF left_is_bst IS True AND right_is_bst IS True AND left_max LESS THAN node.val AND node.val LESS THAN right_min  
                SET current_sum TO left_sum PLUS right_sum PLUS node.val  
                SET current_min TO the smaller BETWEEN node.val AND left_min  
                SET current_max TO the greater BETWEEN node.val AND right_max  
                SET self.max_sum TO the greater BETWEEN self.max_sum AND current_sum  
                RETURN True, current_sum, current_min, current_max  
            ELSE  
                RETURN False, zero, positive infinity, negative infinity  
            END IF  
        END FUNCTION  
        
        CALL helper WITH root AS argument  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
