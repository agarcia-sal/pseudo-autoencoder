CLASS Solution  
    FUNCTION superpalindromesInRange(left right)  
        FUNCTION is_palindrome(num)  
            RETURN string representation of num EQUALS reversed string representation of num  
        END FUNCTION  
        
        FUNCTION generate_palindromes(limit)  
            SET palindromes TO empty list  
            FOR i FROM one TO one hundred thousand MINUS one  
                SET s TO string representation of i  
                APPEND integer of concatenation of s AND substring of s from the second last character TO the first character IN reverse order TO palindromes  
                APPEND integer of concatenation of s AND reversed string representation of s TO palindromes  
            END FOR  
            RETURN list of elements p IN palindromes WHERE p LESS THAN OR EQUAL TO limit  
        END FUNCTION  
        
        SET left TO integer representation of left  
        SET right TO integer representation of right  
        SET limit TO integer of square root of right PLUS one  
        SET palindromes TO generate_palindromes(limit)  
        
        SET count TO zero  
        FOR each p IN palindromes  
            SET square TO p MULTIPLIED BY p  
            IF left LESS THAN OR EQUAL TO square AND square LESS THAN OR EQUAL TO right AND is_palindrome(square)  
                INCREMENT count BY one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
