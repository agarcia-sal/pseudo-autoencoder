CLASS Solution  
    FUNCTION minWindow(s parameter t parameter) RETURNS string  
        SET t_count TO the COUNT OF each CHARACTER IN t  
        SET required TO the LENGTH OF t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO the COUNT OF each CHARACTER in an empty collection  
        SET min_length TO infinite  
        SET min_start TO zero  
        WHILE right LESS THAN the LENGTH OF s  
            SET current_character TO the CHARACTER AT POSITION right OF s  
            INCREMENT the COUNT OF current_character IN window_counts BY one  
            IF current_character IS IN t_count AND the COUNT OF current_character IN window_counts EQUALS the COUNT OF current_character IN t_count  
                INCREMENT formed BY one  
            END IF  
            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET character_to_remove TO the CHARACTER AT POSITION left OF s  
                SET window_size TO right MINUS left PLUS one  
                IF window_size LESS THAN min_length  
                    SET min_length TO window_size  
                    SET min_start TO left  
                END IF  
                DECREMENT the COUNT OF character_to_remove IN window_counts BY one  
                IF character_to_remove IS IN t_count AND the COUNT OF character_to_remove IN window_counts LESS THAN the COUNT OF character_to_remove IN t_count  
                    DECREMENT formed BY one  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            INCREMENT right BY one  
        END WHILE  
        IF min_length EQUALS infinite  
            RETURN empty string  
        ELSE  
            RETURN the SUBSTRING FROM POSITION min_start TO POSITION min_start PLUS min_length OF s  
        END IF  
    END FUNCTION  
END CLASS
