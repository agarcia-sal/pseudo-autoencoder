CLASS Solution  
    FUNCTION gameOfLife(board)  
        IF board EQUALS nothing OR the LENGTH OF board EQUALS zero OR the LENGTH OF element at position zero of board EQUALS zero THEN  
            RETURN  
        END IF  
        
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        
        SET directions TO the LIST OF tuples with values  
            a pair with FIRST element as negative one AND SECOND element as negative one  
            a pair with FIRST element as negative one AND SECOND element as zero  
            a pair with FIRST element as negative one AND SECOND element as one  
            a pair with FIRST element as zero AND SECOND element as negative one  
            a pair with FIRST element as zero AND SECOND element as one  
            a pair with FIRST element as one AND SECOND element as negative one  
            a pair with FIRST element as one AND SECOND element as zero  
            a pair with FIRST element as one AND SECOND element as one  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET live_neighbors TO zero  
                FOR each pair dx dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN m AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n  
                        AND (element at position ni of board at position nj EQUALS one OR element at position ni of board at position nj EQUALS three) THEN  
                        INCREMENT live_neighbors BY one  
                    END IF  
                END FOR  
                
                IF element at position i of board at position j EQUALS one THEN  
                    IF live_neighbors LESS THAN two OR live_neighbors GREATER THAN three THEN  
                        SET element at position i of board at position j TO three  
                    END IF  
                ELSE  
                    IF live_neighbors EQUALS three THEN  
                        SET element at position i of board at position j TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of board at position j EQUALS two THEN  
                    SET element at position i of board at position j TO one  
                ELSE IF element at position i of board at position j EQUALS three THEN  
                    SET element at position i of board at position j TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
