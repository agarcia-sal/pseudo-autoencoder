CLASS Solution  
    FUNCTION maxNumOfSubstrings(s)  
        SET first TO empty dictionary  
        SET last TO empty dictionary  
        FOR each index i AND character c IN enumeration of s  
            IF c NOT IN first THEN  
                SET element at key c of first TO i  
            END IF  
            SET element at key c of last TO i  
        END FOR  
  
        SET intervals TO empty list  
        FOR each character c IN keys of first  
            SET start TO element at key c of first  
            SET end TO element at key c of last  
            SET i TO start  
            WHILE i LESS THAN OR EQUAL TO end  
                IF element at key character at position i of s of first LESS THAN start THEN  
                    BREAK the loop  
                END IF  
                SET end TO the GREATER VALUE BETWEEN end AND element at key character at position i of s of last  
                INCREMENT i BY one  
            END WHILE  
            IF loop was not broken THEN  
                APPEND pair of start AND end TO intervals  
            END IF  
        END FOR  
  
        SORT intervals BY the second element of each pair in ascending order  
  
        SET result TO empty list  
        SET prev_end TO negative one  
        FOR each pair of start AND end IN intervals  
            IF start GREATER THAN prev_end THEN  
                APPEND substring from position start TO position end of s TO result  
                SET prev_end TO end  
            END IF  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
