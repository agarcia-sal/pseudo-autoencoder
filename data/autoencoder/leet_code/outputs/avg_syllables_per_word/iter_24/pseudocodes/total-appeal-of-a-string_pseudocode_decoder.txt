CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO list of twenty-six elements each set to negative one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each pair of index i AND character char IN the enumeration of s  
            SET index TO the numerical position of char in the alphabet calculated as the numerical code of char MINUS the numerical code of the character a  
            SET current_appeal TO current_appeal PLUS the difference of i MINUS the element at position index of last  
            SET total_appeal TO total_appeal PLUS current_appeal  
            SET the element at position index of last TO i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
