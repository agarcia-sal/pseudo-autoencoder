CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        SORT arr2 IN ascending order  
        SET dp TO a dictionary WITH key of negative one AND value of zero  
        
        FOR each num IN arr1  
            SET new_dp TO an empty dictionary  
            FOR each last_num AND ops IN the items of dp  
                IF num GREATER THAN last_num  
                    SET the value at key num OF new_dp TO the minimum BETWEEN the current value at key num OF new_dp OR infinity AND ops  
                END IF  
                
                SET idx TO the position TO insert last_num IN arr2 TO maintain ascending order using binary search on the right side  
                IF idx LESS THAN the LENGTH OF arr2  
                    SET the value at key element at position idx OF arr2 OF new_dp TO the minimum BETWEEN the current value at key element at position idx OF new_dp OR infinity AND ops PLUS one  
                END IF  
            END FOR  
            
            IF new_dp IS empty  
                RETURN negative one  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the minimum value AMONG the values OF dp  
    END FUNCTION  
END CLASS
