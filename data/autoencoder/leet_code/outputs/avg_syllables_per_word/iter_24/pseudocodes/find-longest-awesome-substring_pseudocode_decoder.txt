CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a mapping from integer zero TO integer negative one  
        SET prefix_xor TO integer zero  
        SET max_length TO integer zero  
          
        FOR each index and character IN s with index named i and character named char  
            SET digit TO the integer value represented by character char  
            SET prefix_xor TO the result of prefix_xor EXCLUSIVE OR with the integer one LEFT SHIFTED BY digit positions  
              
            IF prefix_xor IS A KEY IN prefix_map  
                SET max_length TO the MAXIMUM OF max_length AND the result of i MINUS the value at key prefix_xor in prefix_map  
            END IF  
              
            FOR each integer j FROM integer zero TO integer nine  
                SET flipped TO the result of prefix_xor EXCLUSIVE OR with the integer one LEFT SHIFTED BY j positions  
                IF flipped IS A KEY IN prefix_map  
                    SET max_length TO the MAXIMUM OF max_length AND the result of i MINUS the value at key flipped in prefix_map  
                END IF  
            END FOR  
              
            IF prefix_xor IS NOT A KEY IN prefix_map  
                SET the value at key prefix_xor in prefix_map TO i  
            END IF  
        END FOR  
          
        RETURN max_length  
    END FUNCTION  
END CLASS
