CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num EQUALS one  
            RETURN one  
        END IF  
          
        SET factors TO empty list  
        FOR digit FROM nine TO two DECREMENTING BY one  
            WHILE the remainder of num DIVIDED BY digit EQUALS zero  
                APPEND digit TO factors  
                SET num TO the integer result of num DIVIDED BY digit  
            END WHILE  
        END FOR  
          
        IF num NOT EQUALS one  
            RETURN zero  
        END IF  
          
        SORT factors IN ascending order  
        SET result TO the integer formed by concatenating each element IN factors IN order  
          
        IF result GREATER THAN the maximum value for thirty two bit signed integer  
            RETURN zero  
        END IF  
          
        RETURN result  
    END FUNCTION  
END CLASS
