CLASS TreeNode  
    FUNCTION __init__ WITH val left right  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH values  
    IF the list values DOES NOT CONTAIN ELEMENTS THEN  
        RETURN no value  
    END IF  
    SET root TO a new TreeNode with the element at position zero of values  
    SET i TO one  
    INITIALIZE queue AS an empty double ended queue  
    APPEND root TO queue  
    WHILE queue HOLDS ELEMENTS  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT EQUAL TO no value THEN  
            SET the left attribute of node TO a new TreeNode with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT EQUAL TO no value THEN  
            SET the right attribute of node TO a new TreeNode with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH p q  
    IF p DOES NOT CONTAIN A VALUE AND q DOES NOT CONTAIN A VALUE THEN  
        RETURN true  
    ELSE IF p DOES NOT CONTAIN A VALUE OR q DOES NOT CONTAIN A VALUE THEN  
        RETURN false  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with the left attribute of p and the left attribute of q AND the result of calling is_same_tree with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION levelOrder WITH root  
        IF root DOES NOT CONTAIN A VALUE THEN  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        INITIALIZE queue AS a double ended queue containing root  
        WHILE queue HOLDS ELEMENTS  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR each element in the range from zero TO level_size MINUS one  
                SET node TO the first element removed from queue  
                APPEND the value attribute of node TO current_level  
                IF the left attribute of node CONTAINS A VALUE THEN  
                    APPEND the left attribute of node TO queue  
                END IF  
                IF the right attribute of node CONTAINS A VALUE THEN  
                    APPEND the right attribute of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
