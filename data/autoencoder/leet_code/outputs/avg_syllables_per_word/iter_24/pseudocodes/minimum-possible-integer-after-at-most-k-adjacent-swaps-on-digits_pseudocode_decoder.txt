CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO list of characters of num  
        SET n TO the length of num  
        
        WHILE k GREATER THAN zero  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF element at position i of num GREATER THAN element at position i PLUS one of num  
                    SET min_digit TO character nine  
                    SET min_index TO minus one  
                    FOR j FROM i TO the smaller of i PLUS k PLUS one AND n MINUS one  
                        IF element at position j of num LESS THAN min_digit  
                            SET min_digit TO element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET a new list TO element at position min_index of num CONCATENATED WITH the sublist of num from position i TO position min_index MINUS one  
                    SET the sublist of num from position i TO position min_index TO the new list  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO true  
                    BREAK  
                END IF  
            END FOR  
            
            IF swapped EQUALS false  
                BREAK  
            END IF  
        END WHILE  
        
        RETURN concatenation of all characters in num as a string  
    END FUNCTION  
END CLASS
