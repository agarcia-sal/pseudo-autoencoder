CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUALS the symbol plus OR token EQUALS the symbol minus OR token EQUALS the symbol multiplied BY OR token EQUALS the symbol divided BY  
                SET b TO the last element removed FROM stack  
                SET a TO the last element removed FROM stack  
                IF token EQUALS the symbol plus  
                    APPEND a PLUS b TO stack  
                ELSE IF token EQUALS the symbol minus  
                    APPEND a MINUS b TO stack  
                ELSE IF token EQUALS the symbol multiplied BY  
                    APPEND a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS the symbol divided BY  
                    APPEND the integer part of a DIVIDED BY b TO stack  
                END IF  
            ELSE  
                APPEND the integer value of token TO stack  
            END IF  
        END FOR  
        RETURN the first element of stack  
    END FUNCTION  
END CLASS
