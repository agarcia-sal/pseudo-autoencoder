CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO a tuple of tuples where each tuple corresponds to each row of board  
        SET target TO a tuple containing the tuples representing the first row as one plus two plus three and the second row as four plus five plus zero  
        SET neighbors TO a mapping with keys zero to five each mapping to a set of integers representing neighboring positions as given  
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position j of element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        SET queue TO an empty double ended queue containing one tuple of start zero_pos and zero  
        SET visited TO a set containing start  
        WHILE the LENGTH OF queue GREATER THAN zero  
            REMOVE the first element from queue AND ASSIGN it to state zero_pos moves  
            IF state EQUALS target  
                RETURN moves  
            END IF  
            FOR each neighbor IN the set mapped from zero_pos in neighbors  
                SET new_state TO a list of lists by converting each tuple row in state to a list  
                SET i TO zero_pos DIVIDED BY three  
                SET j TO zero_pos MODULO three  
                SET ni TO neighbor DIVIDED BY three  
                SET nj TO neighbor MODULO three  
                SWAP the elements at positions j of element at position i AND at positions nj of element at position ni IN new_state  
                SET new_tuple_state TO a tuple of tuples by converting each list row in new_state to a tuple  
                IF new_tuple_state is not an element of visited  
                    ADD new_tuple_state TO visited  
                    APPEND a tuple of new_tuple_state neighbor moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
