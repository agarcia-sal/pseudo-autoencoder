CLASS ListNode  
    FUNCTION __init__ with parameters self val EQUALS zero next EQUALS None  
        SET self dot val TO val  
        SET self dot next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with parameter the first element at position one of values  
    SET p TO head  
    FOR each val IN elements of values from position two TO the last position  
        SET node TO new ListNode with parameter val  
        SET p dot next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the result of the predicate the value at p1 EQUALS the value at p2 AND the result of calling is_same_list with parameters the next of p1 and the next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList with parameters self head  
        IF head EQUALS None OR the next of head EQUALS None OR the next of the next of head EQUALS None  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO the next of head  
        SET even_head TO even  
        WHILE even IS NOT None AND the next of even IS NOT None  
            SET the next of odd TO the next of even  
            SET odd TO the next of odd  
            SET the next of even TO the next of odd  
            SET even TO the next of even  
        END WHILE  
        SET the next of odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
