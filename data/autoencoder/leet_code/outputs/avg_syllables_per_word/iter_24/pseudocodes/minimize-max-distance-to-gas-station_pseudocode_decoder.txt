CLASS Solution  
    FUNCTION minmaxGasDist(stations, k)  
        FUNCTION possible(max_dist)  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                INCREMENT additional_stations BY the INTEGER PART OF the value of element at position index of stations MINUS the value of element at position index MINUS one of stations DIVIDED BY max_dist  
            END FOR  
            RETURN additional_stations LESS THAN OR EQUAL TO k  
        END FUNCTION  
  
        SET left TO zero  
        SET right TO ten RAISED TO THE POWER OF eight  
        WHILE right MINUS left GREATER THAN one TIMES ten RAISED TO THE POWER OF negative six holds  
            SET mid TO left PLUS right DIVIDED BY two  
            IF possible(mid) EQUALS true  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
