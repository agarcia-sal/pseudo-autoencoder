CLASS Solution  
    FUNCTION minStickers(stickers, target)  
        SET stickers TO a list of frequency counters for each element in stickers  
          
        SORT stickers BY the negative of the maximum value among the frequency counts of each sticker  
          
        SET queue TO a deque containing a tuple of target and zero  
          
        SET memo TO a dictionary mapping target TO zero  
          
        WHILE the length of queue is greater than zero  
            SET current_target and used_stickers TO the first element removed from queue  
              
            FOR each sticker IN stickers  
                IF the first element at position one of current_target IS NOT PRESENT IN sticker  
                    CONTINUE to the next iteration  
                END IF  
                  
                SET remaining TO a frequency counter of current_target  
                DECREMENT remaining BY the frequency counts of sticker  
                  
                SET new_target TO an empty string  
                FOR each element IN sorted elements of remaining  
                    APPEND element TO new_target  
                END FOR  
                  
                IF new_target EQUALS an empty string  
                    RETURN used_stickers PLUS one  
                END IF  
                  
                IF new_target IS NOT PRESENT IN keys of memo  
                    SET memo at key new_target TO used_stickers PLUS one  
                    APPEND a tuple of new_target and used_stickers PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
          
        RETURN negative one  
    END FUNCTION  
END CLASS
