CLASS Solution  
    FUNCTION findSubstring(s parameter s OF TYPE string words parameter words OF TYPE list of string) RETURNS list of integer  
        IF s EQUALS empty OR words EQUALS empty OR element at position zero of words EQUALS empty  
            RETURN empty list  
        END IF  
        SET word_length TO the LENGTH OF element at position zero of words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the COUNTING OF occurrences of each word IN words  
        SET result TO empty list  
        FOR index i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO an EMPTY counting dictionary  
            FOR index j FROM i TO LESS THAN i PLUS total_length IN STEPS OF word_length  
                SET word TO the SUBSTRING of s FROM position j TO position j PLUS word_length  
                INCREMENT the count of word IN seen BY one  
                IF the count of word IN seen GREATER THAN the count of word IN word_count  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF the inner loop did NOT break prematurely  
                APPEND i TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
