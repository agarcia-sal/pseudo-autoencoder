CLASS Solution  
    FUNCTION sortColors(nums)  
        SET low TO zero  
        SET mid TO zero  
        SET high TO the LENGTH OF nums MINUS one  
        WHILE the element at position mid of nums IS LESS THAN OR EQUAL TO the element at position high of nums  
            IF the element at position mid of nums EQUALS zero  
                SWAP the element at position low of nums WITH the element at position mid of nums  
                INCREMENT low BY one  
                INCREMENT mid BY one  
            ELSE IF the element at position mid of nums EQUALS one  
                INCREMENT mid BY one  
            ELSE  
                SWAP the element at position mid of nums WITH the element at position high of nums  
                DECREMENT high BY one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
