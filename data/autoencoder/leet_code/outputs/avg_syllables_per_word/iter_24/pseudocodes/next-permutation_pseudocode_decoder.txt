CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET i TO the LENGTH OF nums MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND the element at position i of nums GREATER THAN OR EQUAL TO the element at position i PLUS one of nums  
            DECREMENT i BY one  
        END WHILE  
        
        IF i GREATER THAN OR EQUAL TO zero  
            SET j TO the LENGTH OF nums MINUS one  
            WHILE the element at position j of nums LESS THAN OR EQUAL TO the element at position i of nums  
                DECREMENT j BY one  
            END WHILE  
            
            SET temporary TO the element at position i of nums  
            SET the element at position i of nums TO the element at position j of nums  
            SET the element at position j of nums TO temporary  
        END IF  
        
        CALL reverseSublist with parameters nums AND i PLUS one TO the LENGTH OF nums MINUS one  
    END FUNCTION  
    
    FUNCTION reverseSublist(nums, start_index, end_index)  
        WHILE start_index LESS THAN end_index  
            SET temporary TO the element at position start_index of nums  
            SET the element at position start_index of nums TO the element at position end_index of nums  
            SET the element at position end_index of nums TO temporary  
            INCREMENT start_index BY one  
            DECREMENT end_index BY one  
        END WHILE  
    END FUNCTION  
END CLASS
