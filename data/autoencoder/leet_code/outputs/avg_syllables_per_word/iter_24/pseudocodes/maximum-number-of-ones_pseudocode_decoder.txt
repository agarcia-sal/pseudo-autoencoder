CLASS Solution  
    FUNCTION maximumNumberOfOnes(width height sideLength maxOnes)  
        SET frequency_list TO an empty list of length sideLength MULTIPLIED BY sideLength with all elements set to zero  
        FOR i FROM zero TO width MINUS one  
            FOR j FROM zero TO height MINUS one  
                SET position TO the remainder of i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS the remainder of j DIVIDED BY sideLength  
                INCREMENT the element at position in frequency_list BY one  
            END FOR  
        END FOR  
        SORT frequency_list IN descending order  
        SET total_sum TO zero  
        FOR index FROM zero TO maxOnes MINUS one  
            INCREMENT total_sum BY the element at position index in frequency_list  
        END FOR  
        RETURN total_sum  
    END FUNCTION  
END CLASS
