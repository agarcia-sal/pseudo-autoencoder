CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty list THEN  
        RETURN none  
    END IF  
    SET root TO new TreeNode initialized with element at position one of values  
    SET i TO one  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the left side of queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS none THEN  
            SET node.left TO new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS none THEN  
            SET node.right TO new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q THEN  
        RETURN false  
    ELSE  
        RETURN result of is_same_tree called with left child of p and left child of q AND result of is_same_tree called with right child of p and right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode(self, root, key)  
        IF root EQUALS none THEN  
            RETURN none  
        END IF  
        IF value of root GREATER THAN key THEN  
            SET left child of root TO result of calling deleteNode on self with left child of root and key  
            RETURN root  
        END IF  
        IF value of root LESS THAN key THEN  
            SET right child of root TO result of calling deleteNode on self with right child of root and key  
            RETURN root  
        END IF  
        IF left child of root EQUALS none THEN  
            RETURN right child of root  
        END IF  
        IF right child of root EQUALS none THEN  
            RETURN left child of root  
        END IF  
        SET node TO right child of root  
        WHILE left child of node NOT EQUALS none  
            SET node TO left child of node  
        END WHILE  
        SET left child of node TO left child of root  
        SET root TO right child of root  
        RETURN root  
    END FUNCTION  
END CLASS
