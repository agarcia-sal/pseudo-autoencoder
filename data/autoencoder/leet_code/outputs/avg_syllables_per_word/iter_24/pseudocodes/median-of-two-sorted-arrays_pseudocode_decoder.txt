CLASS Solution  
    FUNCTION findMedianSortedArrays(nums1, nums2)  
        IF the LENGTH OF nums1 IS GREATER THAN the LENGTH OF nums2  
            SET temporary_variable TO nums1  
            SET nums1 TO nums2  
            SET nums2 TO temporary_variable  
        END IF  
        
        SET x TO the LENGTH OF nums1  
        SET y TO the LENGTH OF nums2  
        SET low TO zero  
        SET high TO x  
        
        WHILE low IS LESS THAN OR EQUAL TO high  
            SET partitionX TO the INTEGER DIVISION result OF low PLUS high BY two  
            SET partitionY TO the INTEGER DIVISION result OF x PLUS y PLUS one BY two MINUS partitionX  
            
            IF partitionX EQUALS zero  
                SET maxLeftX TO negative infinity  
            ELSE  
                SET maxLeftX TO element at position partitionX MINUS one OF nums1  
            END IF  
            
            IF partitionX EQUALS x  
                SET minRightX TO positive infinity  
            ELSE  
                SET minRightX TO element at position partitionX OF nums1  
            END IF  
            
            IF partitionY EQUALS zero  
                SET maxLeftY TO negative infinity  
            ELSE  
                SET maxLeftY TO element at position partitionY MINUS one OF nums2  
            END IF  
            
            IF partitionY EQUALS y  
                SET minRightY TO positive infinity  
            ELSE  
                SET minRightY TO element at position partitionY OF nums2  
            END IF  
            
            IF maxLeftX IS LESS THAN OR EQUAL TO minRightY AND maxLeftY IS LESS THAN OR EQUAL TO minRightX  
                IF the REMAINDER OF x PLUS y DIVIDED BY two EQUALS zero  
                    SET maximum_of_lefts TO maxLeftX  
                    IF maxLeftY IS GREATER THAN maximum_of_lefts  
                        SET maximum_of_lefts TO maxLeftY  
                    END IF  
                    SET minimum_of_rights TO minRightX  
                    IF minRightY IS LESS THAN minimum_of_rights  
                        SET minimum_of_rights TO minRightY  
                    END IF  
                    RETURN maximum_of_lefts PLUS minimum_of_rights DIVIDED BY two  
                ELSE  
                    IF maxLeftX IS GREATER THAN maxLeftY  
                        RETURN maxLeftX  
                    ELSE  
                        RETURN maxLeftY  
                    END IF  
                END IF  
            ELSE IF maxLeftX IS GREATER THAN minRightY  
                SET high TO partitionX MINUS one  
            ELSE  
                SET low TO partitionX PLUS one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
