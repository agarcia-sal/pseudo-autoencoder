CLASS Solution  
    FUNCTION uniquePaths(m n)  
        SET dp TO two dimensional list with m rows each containing n elements all set to one  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET element at position i of dp at position j TO element at position i MINUS one of dp at position j PLUS element at position i of dp at position j MINUS one  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one of dp at position n MINUS one  
    END FUNCTION  
END CLASS
