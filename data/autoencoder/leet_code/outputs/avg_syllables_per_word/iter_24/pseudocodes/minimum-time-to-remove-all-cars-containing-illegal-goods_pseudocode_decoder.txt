CLASS Solution  
    FUNCTION minimumTime(s)  
        SET n TO the LENGTH OF s  
        SET left_costs TO a list of n elements all SET TO zero  
        SET right_costs TO a list of n elements all SET TO zero  
        
        IF the element at position zero of s EQUALS the character one  
            SET the element at position zero of left_costs TO one  
        ELSE  
            SET the element at position zero of left_costs TO zero  
        END IF  
        
        FOR i FROM one TO n MINUS one  
            IF the element at position i of s EQUALS the character zero  
                SET the element at position i of left_costs TO the element at position i MINUS one of left_costs  
            ELSE  
                SET left_option_one TO i PLUS one  
                SET left_option_two TO the element at position i MINUS one of left_costs PLUS two  
                SET the element at position i of left_costs TO the lesser value BETWEEN left_option_one AND left_option_two  
            END IF  
        END FOR  
        
        IF the element at position n MINUS one of s EQUALS the character one  
            SET the element at position n MINUS one of right_costs TO one  
        ELSE  
            SET the element at position n MINUS one of right_costs TO zero  
        END IF  
        
        FOR i FROM n MINUS two DOWN TO zero  
            IF the element at position i of s EQUALS the character zero  
                SET the element at position i of right_costs TO the element at position i PLUS one of right_costs  
            ELSE  
                SET right_option_one TO n MINUS i  
                SET right_option_two TO the element at position i PLUS one of right_costs PLUS two  
                SET the element at position i of right_costs TO the lesser value BETWEEN right_option_one AND right_option_two  
            END IF  
        END FOR  
        
        SET min_time TO the lesser value BETWEEN the element at position n MINUS one of left_costs AND the element at position zero of right_costs  
        
        FOR i FROM zero TO n MINUS two  
            SET combined_cost TO the element at position i of left_costs PLUS the element at position i PLUS one of right_costs  
            SET min_time TO the lesser value BETWEEN min_time AND combined_cost  
        END FOR  
        
        RETURN min_time  
    END FUNCTION  
END CLASS
