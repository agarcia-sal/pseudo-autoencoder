CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO empty dictionary  
        FOR each num IN nums  
            SET depth TO integer division of num BY one hundred  
            SET pos TO integer division of num BY ten MODULO ten  
            SET value TO num MODULO ten  
            SET element at key depth and pos OF tree TO value  
        END FOR  

        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO value at key depth and pos OF tree OR None IF not found  
            IF current_value EQUALS None THEN  
                RETURN zero  
            END IF  
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child TO a pair of depth PLUS one and pos MULTIPLIED BY two MINUS one  
            SET right_child TO a pair of depth PLUS one and pos MULTIPLIED BY two  
            IF left_child NOT IN keys of tree AND right_child NOT IN keys of tree THEN  
                RETURN new_path_sum  
            END IF  
            SET left_sum TO dfs with parameters first element of left_child second element of left_child new_path_sum  
            SET right_sum TO dfs with parameters first element of right_child second element of right_child new_path_sum  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  

        RETURN dfs with parameters one one zero  
    END FUNCTION  
END CLASS
