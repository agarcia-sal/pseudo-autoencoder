CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO a list of boolean false values of length n PLUS one  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE j MULTIPLIED BY j LESS THAN OR EQUAL TO i  
                IF the boolean value at position i MINUS j MULTIPLIED BY j of dp EQUALS false  
                    SET the boolean value at position i of dp TO true  
                    BREAK the current WHILE loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN the boolean value at position n of dp  
    END FUNCTION  
END CLASS
