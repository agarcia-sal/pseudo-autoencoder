CLASS Solution  
    FUNCTION subarraysWithKDistinct(nums k)  
        FUNCTION atMostKDistinct(t)  
            SET count TO a mapping from element to integer initialized empty  
            SET left TO zero  
            SET result TO zero  
            FOR right FROM zero TO the length of nums MINUS one  
                INCREMENT the count of element at position right of nums BY one  
                WHILE the number of keys in count GREATER THAN t  
                    DECREMENT the count of element at position left of nums BY one  
                    IF the count of element at position left of nums EQUALS zero  
                        REMOVE the element at position left of nums FROM count  
                    END IF  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT result BY right MINUS left PLUS one  
            END FOR  
            RETURN result  
        END FUNCTION  
        RETURN the result of calling atMostKDistinct with k MINUS the result of calling atMostKDistinct with k MINUS one  
    END FUNCTION  
END CLASS
