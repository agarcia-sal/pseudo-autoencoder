CLASS TreeNode  
    FUNCTION __init__ with parameters val defaulting to zero left defaulting to none right defaulting to none  
        SET the value attribute TO val  
        SET the left attribute TO left  
        SET the right attribute TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS absence of any value  
        RETURN absence of value  
    END IF  
    SET root TO new instance of TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue holds any element  
        SET node TO the removal of the element from front of queue  
        IF i LESS THAN the length of values AND the element at position i of values NOT EQUALS absence of value  
            SET the left attribute of node TO new instance of TreeNode initialized with the element at position i of values  
            APPEND the left attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND the element at position i of values NOT EQUALS absence of value  
            SET the right attribute of node TO new instance of TreeNode initialized with the element at position i of values  
            APPEND the right attribute of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p and q  
    IF p EQUALS absence of value AND q EQUALS absence of value  
        RETURN true  
    ELSE IF p EQUALS absence of value OR q EQUALS absence of value  
        RETURN false  
    ELSE IF the value attribute of p NOT EQUALS the value attribute of q  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree called with the left attribute of p and the left attribute of q AND the result of is_same_tree called with the right attribute of p and the right attribute of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST with parameters root and val  
        FUNCTION dfs with parameter root  
            IF root EQUALS absence of value  
                RETURN new instance of TreeNode initialized with val  
            END IF  
            IF the value attribute of root LESS THAN val  
                SET the right attribute of root TO the result of dfs called with the right attribute of root  
            ELSE  
                SET the left attribute of root TO the result of dfs called with the left attribute of root  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN the result of dfs called with root  
    END FUNCTION  
END CLASS
