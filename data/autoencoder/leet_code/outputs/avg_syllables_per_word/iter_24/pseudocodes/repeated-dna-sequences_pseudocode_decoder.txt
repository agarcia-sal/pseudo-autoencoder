CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO a new dictionary defaulting to zero for each key  
        FOR index FROM zero TO the LENGTH OF s MINUS nine  
            SET sequence TO the substring from position index TO position index PLUS ten MINUS one of s  
            SET the value at key sequence in sequence_count TO the value at key sequence in sequence_count PLUS one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each sequence and its count IN sequence_count entries  
            IF count GREATER THAN one  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
