CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        IF the REMAINDER of m PLUS n MINUS one DIVIDED BY two NOT EQUALS zero  
            RETURN False  
        END IF  
        
        CREATE dp AS a three dimensional collection of False with dimensions m PLUS one BY n PLUS one BY m PLUS n PLUS one  
        
        SET element at position zero of element at position one of dp TO True  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR k FROM zero TO m PLUS n  
                    IF element at position j PLUS one of element at position i of dp IS True OR element at position j of element at position i PLUS one of dp IS True  
                        IF element at position i of element at position j of grid EQUALS opening parenthesis AND k PLUS one LESS THAN m PLUS n PLUS one  
                            SET element at position j PLUS one of element at position i PLUS one of element at position k PLUS one of dp TO True  
                        ELSE IF element at position i of element at position j of grid EQUALS closing parenthesis AND k MINUS one GREATER THAN OR EQUAL TO zero  
                            SET element at position j PLUS one of element at position i PLUS one of element at position k MINUS one of dp TO True  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position n of element at position m of element at position zero of dp  
    END FUNCTION  
END CLASS
