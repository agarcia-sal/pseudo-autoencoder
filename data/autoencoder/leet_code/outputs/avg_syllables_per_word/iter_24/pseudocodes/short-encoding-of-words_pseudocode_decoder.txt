CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET reversed_words TO the list of each word in words reversed  
        SORT reversed_words in lexicographical order  
        SET total_length TO zero  
        FOR index FROM zero TO the length of reversed_words MINUS one  
            IF index PLUS one EQUALS the length of reversed_words OR the element at position index PLUS one of reversed_words DOES NOT START WITH the element at position index of reversed_words  
                INCREMENT total_length BY the length of the element at position index of reversed_words PLUS one  
            END IF  
        END FOR  
        RETURN total_length  
    END FUNCTION  
END CLASS
