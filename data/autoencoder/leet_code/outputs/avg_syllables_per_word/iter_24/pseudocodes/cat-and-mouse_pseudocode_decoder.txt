CLASS Solution  
    FUNCTION catMouseGame(graph)  
        SET DRAW TO zero  
        SET MOUSE_WIN TO one  
        SET CAT_WIN TO two  
        SET n TO the LENGTH OF graph  
        
        FUNCTION dfs(turn, mouse_pos, cat_pos)  
            IF turn EQUALS two MULTIPLIED BY n THEN  
                RETURN DRAW  
            END IF  
            IF mouse_pos EQUALS zero THEN  
                RETURN MOUSE_WIN  
            END IF  
            IF mouse_pos EQUALS cat_pos THEN  
                RETURN CAT_WIN  
            END IF  
            
            SET current_player TO mouse IF turn MODULO two EQUALS zero OTHERWISE cat  
            
            IF current_player EQUALS mouse THEN  
                SET result TO CAT_WIN  
                FOR each next_pos IN the elements at position mouse_pos of graph  
                    SET next_result TO dfs with turn PLUS one AND next_pos AND cat_pos  
                    IF next_result EQUALS MOUSE_WIN THEN  
                        RETURN MOUSE_WIN  
                    ELSE IF next_result EQUALS DRAW THEN  
                        SET result TO DRAW  
                    END IF  
                END FOR  
                RETURN result  
            ELSE  
                SET result TO MOUSE_WIN  
                FOR each next_pos IN the elements at position cat_pos of graph  
                    IF next_pos EQUALS zero THEN  
                        CONTINUE to next iteration of the loop  
                    END IF  
                    SET next_result TO dfs with turn PLUS one AND mouse_pos AND next_pos  
                    IF next_result EQUALS CAT_WIN THEN  
                        RETURN CAT_WIN  
                    ELSE IF next_result EQUALS DRAW THEN  
                        SET result TO DRAW  
                    END IF  
                END FOR  
                RETURN result  
            END IF  
        END FUNCTION  
        
        RETURN dfs with zero AND one AND two  
    END FUNCTION  
END CLASS
