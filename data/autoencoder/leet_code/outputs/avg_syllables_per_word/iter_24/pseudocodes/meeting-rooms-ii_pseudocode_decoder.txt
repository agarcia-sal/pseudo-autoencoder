CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty list  
            RETURN zero  
        END IF  

        SORT intervals IN ASCENDING ORDER BASED ON the element at position zero of each interval  

        SET min_heap TO empty list  

        FOR each meeting IN intervals  
            IF the LENGTH OF min_heap IS GREATER THAN zero AND the element at position zero of meeting IS GREATER THAN OR EQUAL TO the element at position zero of min_heap  
                REMOVE the smallest element FROM min_heap  
            END IF  
            ADD the element at position one of meeting TO min_heap  
        END FOR  

        RETURN the LENGTH OF min_heap  
    END FUNCTION  
END CLASS
