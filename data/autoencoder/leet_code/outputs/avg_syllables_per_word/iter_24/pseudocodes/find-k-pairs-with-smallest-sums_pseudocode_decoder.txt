CLASS Solution  
    FUNCTION kSmallestPairs(nums1 LIST OF INTEGER nums2 LIST OF INTEGER k INTEGER) RETURNS LIST OF LIST OF INTEGER  
        IF the LENGTH OF nums1 EQUALS zero OR the LENGTH OF nums2 EQUALS zero OR k LESS THAN OR EQUAL TO zero THEN  
            RETURN an empty list  
        END IF  
        
        SET min_heap TO an empty list  
        
        FOR index i FROM zero TO the SMALLEST OF k AND the LENGTH OF nums1 MINUS one  
            COMPUTE the sum_of_pair AS element at position i of nums1 PLUS element at position zero of nums2  
            PUSH a tuple consisting of sum_of_pair, i, and zero ONTO min_heap  
        END FOR  
        
        SET result TO an empty list  
        
        WHILE min_heap is not empty AND the LENGTH OF result LESS THAN k  
            POP the smallest tuple FROM min_heap INTO variables sum_value, i, j  
            APPEND a list consisting of element at position i of nums1 and element at position j of nums2 TO result  
            SET next_index TO j PLUS one  
            IF next_index LESS THAN the LENGTH OF nums2 THEN  
                COMPUTE the next_sum AS element at position i of nums1 PLUS element at position next_index of nums2  
                PUSH a tuple consisting of next_sum, i, and next_index ONTO min_heap  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
