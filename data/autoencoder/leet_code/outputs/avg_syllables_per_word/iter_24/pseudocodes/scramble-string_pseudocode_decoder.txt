CLASS Solution  
    FUNCTION isScramble WITH PARAMETERS s1 s2  
        IF the string s1 EQUALS the string s2  
            RETURN true  
        END IF  
        IF the sorted characters of the string s1 NOT EQUALS the sorted characters of the string s2  
            RETURN false  
        END IF  
        SET the integer n TO the LENGTH OF the string s1  
        FOR the integer i FROM one TO n MINUS one  
            IF the result of the FUNCTION isScramble CALLED WITH the substring from position one TO position i of s1 AND the substring from position one TO position i of s2 IS true AND the result of the FUNCTION isScramble CALLED WITH the substring from position i PLUS one TO the end of s1 AND the substring from position i PLUS one TO the end of s2 IS true  
                RETURN true  
            END IF  
            IF the result of the FUNCTION isScramble CALLED WITH the substring from position one TO position i of s1 AND the substring from the position n MINUS i PLUS one TO the end of s2 IS true AND the result of the FUNCTION isScramble CALLED WITH the substring from position i PLUS one TO the end of s1 AND the substring from position one TO the position n MINUS i of s2 IS true  
                RETURN true  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
