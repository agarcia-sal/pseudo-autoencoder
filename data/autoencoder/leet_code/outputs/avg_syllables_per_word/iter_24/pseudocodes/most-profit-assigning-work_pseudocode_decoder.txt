CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty LIST OF INTEGER profit LIST OF INTEGER worker LIST OF INTEGER) RETURNS INTEGER  
        SET jobs TO an empty list  
        FOR index FROM zero TO the LENGTH OF difficulty MINUS one  
            APPEND a pair composed of element at position index of difficulty AND element at position index of profit TO jobs  
        END FOR  
        SORT jobs BY the first element of each pair in ascending order  
        SORT worker IN ascending order  
        
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the LENGTH OF jobs  
        
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND first element of element at position job_index of jobs LESS THAN OR EQUAL TO ability  
                IF second element of element at position job_index of jobs GREATER THAN best_profit  
                    SET best_profit TO second element of element at position job_index of jobs  
                END IF  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        
        RETURN max_profit  
    END FUNCTION  
END CLASS
