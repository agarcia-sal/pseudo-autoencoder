CLASS Solution  
    FUNCTION fullJustify(words maxWidth)  
        FUNCTION justify_line(line num_of_letters is_last_line)  
            IF the LENGTH OF line EQUALS one OR is_last_line EQUALS true  
                RETURN the concatenation of all elements IN line separated by a single space PLUS the repetition of the space character repeated the difference BETWEEN maxWidth AND the sum of num_of_letters AND the LENGTH OF line MINUS one  
            ELSE  
                SET total_spaces TO the difference BETWEEN maxWidth AND num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO the INTEGER DIVISION of total_spaces BY num_of_gaps  
                SET extra_spaces TO the REMAINDER of total_spaces DIVIDED BY num_of_gaps  
                FOR index FROM zero TO num_of_gaps MINUS one  
                    IF index LESS THAN extra_spaces  
                        APPEND the repetition of the space character repeated spaces_between_words PLUS one TO the element at position index of line  
                    ELSE  
                        APPEND the repetition of the space character repeated spaces_between_words TO the element at position index of line  
                    END IF  
                END FOR  
                RETURN the concatenation of all elements in line without any separator  
            END IF  
        END FUNCTION  

        SET result TO an empty list  
        SET line TO an empty list  
        SET num_of_letters TO zero  

        FOR each word IN words  
            IF the sum of num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth  
                APPEND the result of justify_line with arguments line num_of_letters and false TO result  
                SET line TO an empty list  
                SET num_of_letters TO zero  
            END IF  
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  

        APPEND the result of justify_line with arguments line num_of_letters and true TO result  

        RETURN result  
    END FUNCTION  
END CLASS
