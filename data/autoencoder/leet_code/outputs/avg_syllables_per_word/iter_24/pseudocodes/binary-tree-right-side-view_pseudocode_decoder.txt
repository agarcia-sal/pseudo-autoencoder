CLASS TreeNode  
    FUNCTION __init__ with parameters self val zero left none right none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node with parameter values  
    IF values EQUALS an empty collection THEN  
        RETURN nothing  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO a new double ended queue initialized empty  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero DO  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT nothing THEN  
            SET node dot left TO a new TreeNode initialized with the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT nothing THEN  
            SET node dot right TO a new TreeNode initialized with the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree with parameters p q  
    IF p EQUALS nothing AND q EQUALS nothing THEN  
        RETURN True  
    ELSE IF p EQUALS nothing OR q EQUALS nothing THEN  
        RETURN False  
    ELSE IF the val property of p NOT EQUALS the val property of q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of the result of calling is_same_tree with parameters p dot left and q dot left AND the result of calling is_same_tree with parameters p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView with parameters self root  
        IF root EQUALS nothing THEN  
            RETURN an empty collection  
        END IF  
        SET right_side_view TO an empty collection  
        SET queue TO a new double ended queue initialized with root  
        WHILE the LENGTH OF queue IS GREATER THAN zero DO  
            SET level_length TO the LENGTH OF queue  
            FOR i FROM zero TO level_length MINUS one DO  
                SET node TO the element removed from the front of queue  
                IF i EQUALS level_length MINUS one THEN  
                    APPEND the val property of node TO right_side_view  
                END IF  
                IF the left property of node IS NOT nothing THEN  
                    APPEND the left property of node TO queue  
                END IF  
                IF the right property of node IS NOT nothing THEN  
                    APPEND the right property of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
