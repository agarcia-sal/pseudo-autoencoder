CLASS Solution  
    FUNCTION isValidSerialization(preorder)  
        SET nodes TO the list obtained by splitting preorder by the comma word  
        SET slots TO the integer one  
        FOR each node IN nodes  
            IF the value of slots EQUALS zero  
                RETURN false  
            END IF  
            DECREMENT slots BY one  
            IF node NOT EQUALS the single character hash  
                INCREMENT slots BY two  
            END IF  
        END FOR  
        RETURN whether slots EQUALS zero  
    END FUNCTION  
END CLASS
