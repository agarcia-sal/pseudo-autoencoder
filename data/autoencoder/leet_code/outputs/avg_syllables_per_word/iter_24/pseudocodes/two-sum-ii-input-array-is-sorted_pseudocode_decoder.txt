CLASS Solution  
    FUNCTION twoSum(numbers, target)  
        SET left TO zero  
        SET right TO the LENGTH OF numbers MINUS one  
        WHILE the value of left IS LESS THAN the value of right  
            SET current_sum TO the element at position left of numbers PLUS the element at position right of numbers  
            IF the value of current_sum EQUALS the value of target  
                RETURN a list containing the value of left PLUS one AND the value of right PLUS one  
            ELSE IF the value of current_sum IS LESS THAN the value of target  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
