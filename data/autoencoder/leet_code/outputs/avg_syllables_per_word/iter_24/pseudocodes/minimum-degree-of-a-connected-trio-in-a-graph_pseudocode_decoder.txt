CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        SET graph TO an empty mapping from each integer from one to n inclusive TO an empty set  
        FOR each pair u AND v IN edges  
            ADD v TO the set mapped by u in graph  
            ADD u TO the set mapped by v in graph  
        END FOR  
        SET min_degree TO positive infinity  
        SET trio_found TO false  
        FOR i FROM one TO n inclusive  
            FOR j FROM i PLUS one TO n inclusive  
                FOR k FROM j PLUS one TO n inclusive  
                    IF the set mapped by i in graph CONTAINS j AND the set mapped by i in graph CONTAINS k AND the set mapped by j in graph CONTAINS k  
                        SET trio_found TO true  
                        SET trio_degree TO the LENGTH OF the set mapped by i in graph PLUS the LENGTH OF the set mapped by j in graph PLUS the LENGTH OF the set mapped by k in graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF trio_found EQUALS true  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
