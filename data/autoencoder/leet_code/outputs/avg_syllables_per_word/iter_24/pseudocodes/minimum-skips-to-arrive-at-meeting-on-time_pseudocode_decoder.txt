CLASS Solution  
    FUNCTION minSkips(dist parameter which is a list of integers speed parameter which is an integer hoursBefore parameter which is an integer) RETURNS an integer  
        SET n TO the length of dist  
        CREATE dp as a list containing n elements each of which is a list containing n PLUS one elements each set to mathematical infinity  
        SET total_dist TO the summation of all elements in dist  
        
        IF the total_dist divided by speed is GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  
        
        SET the element at position zero of dp at position zero TO the element at position zero of dist  
        FOR each j FROM one TO n MINUS one DO  
            SET the element at position zero of dp at position j TO the element at position zero of dist  
        END FOR  
        
        FOR each i FROM one TO n MINUS one DO  
            FOR each j FROM zero TO i DO  
                IF j LESS THAN i THEN  
                    SET the element at position i of dp at position j TO the ceiling of the element at position i MINUS one of dp at position j divided by speed MULTIPLIED BY speed PLUS the element at position i of dist  
                END IF  
                IF j GREATER THAN zero THEN  
                    SET the element at position i of dp at position j TO the minimum of the current element at position i of dp at position j AND the element at position i MINUS one of dp at position j MINUS one PLUS the element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR each j FROM zero TO n MINUS one DO  
            IF the element at position n MINUS one of dp at position j is LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
