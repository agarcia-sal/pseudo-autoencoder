CLASS Solution  
    FUNCTION numMagicSquaresInside(grid)  
        FUNCTION is_magic(i, j)  
            IF element at position i of grid at position j of grid NOT EQUALS five  
                RETURN False  
            END IF  
            SET numbers TO empty set  
            FOR x FROM i MINUS one TO i PLUS one  
                FOR y FROM j MINUS one TO j PLUS one  
                    IF element at position x of grid at position y of grid LESS THAN one OR element at position x of grid at position y of grid GREATER THAN nine OR element at position x of grid at position y of grid IN numbers  
                        RETURN False  
                    END IF  
                    ADD element at position x of grid at position y of grid TO numbers  
                END FOR  
            END FOR  
            FOR start FROM i MINUS one TO i PLUS one  
                IF sum of element at position start of grid at position j MINUS one of grid PLUS element at position start of grid at position j of grid PLUS element at position start of grid at position j PLUS one of grid NOT EQUALS fifteen OR sum of element at position i MINUS one of grid at position start of grid PLUS element at position i of grid at position start of grid PLUS element at position i PLUS one of grid at position start of grid NOT EQUALS fifteen  
                    RETURN False  
                END IF  
            END FOR  
            IF sum of element at position i MINUS one of grid at position j MINUS one of grid PLUS element at position i of grid at position j of grid PLUS element at position i PLUS one of grid at position j PLUS one of grid NOT EQUALS fifteen OR sum of element at position i MINUS one of grid at position j PLUS one of grid PLUS element at position i of grid at position j of grid PLUS element at position i PLUS one of grid at position j MINUS one of grid NOT EQUALS fifteen  
                RETURN False  
            END IF  
            RETURN True  
        END FUNCTION  
        SET count TO zero  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero of grid  
        FOR i FROM one TO rows MINUS one  
            FOR j FROM one TO cols MINUS one  
                IF is_magic(i, j) EQUALS True  
                    INCREMENT count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
