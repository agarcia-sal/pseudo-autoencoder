CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero DO  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN False  
    ELSE IF the value of p NOT EQUALS the value of q THEN  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree applied to the left child of p and the left child of q AND the result of is_same_tree applied to the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION helper(node, current_sum, prefix_sums)  
            IF node EQUALS none THEN  
                RETURN zero  
            END IF  
            SET current_sum TO current_sum PLUS the value of node  
            SET paths_to_here TO the value mapped to the key current_sum MINUS targetSum in prefix_sums OR zero if no such key exists  
            SET the value mapped to the key current_sum in prefix_sums TO the value mapped to the key current_sum in prefix_sums OR zero IF no such key exists PLUS one  
            SET paths_in_left TO the result of helper applied to the left child of node, current_sum, and prefix_sums  
            SET paths_in_right TO the result of helper applied to the right child of node, current_sum, and prefix_sums  
            DECREMENT the value mapped to the key current_sum in prefix_sums BY one  
            RETURN paths_to_here PLUS paths_in_left PLUS paths_in_right  
        END FUNCTION  
        SET prefix_sums TO a dictionary mapping zero TO one  
        RETURN the result of helper applied to root, zero, and prefix_sums  
    END FUNCTION  
END CLASS
