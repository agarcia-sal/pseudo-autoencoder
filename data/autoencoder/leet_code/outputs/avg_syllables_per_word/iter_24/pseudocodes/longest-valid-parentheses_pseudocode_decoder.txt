CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        SET stack TO a list containing a single element negative one  
        SET max_length TO zero  
        FOR each index i AND character char IN s  
            IF character char EQUALS left parenthesis  
                APPEND index i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF the stack IS empty  
                    APPEND index i TO stack  
                ELSE  
                    SET max_length TO the greater value BETWEEN max_length AND index i MINUS the last element of stack  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
