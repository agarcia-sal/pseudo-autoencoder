CLASS Solution  
    FUNCTION frogPosition(n, edges, t, target)  
        IF n EQUALS one  
            RETURN one  
        END IF  
        SET graph TO a new mapping with default empty list for each key  
        FOR each pair u AND v IN edges  
            APPEND v TO the list associated with u in graph  
            APPEND u TO the list associated with v in graph  
        END FOR  
        SET queue TO a double ended queue containing the tuple of current_node as one current_probability as one and current_time as zero  
        SET visited TO a set containing one  
        WHILE queue is not empty  
            REMOVE the leftmost element from queue and ASSIGN current_node AND current_probability AND current_time from it  
            IF current_time EQUALS t OR current_node NOT EQUALS one AND the LENGTH of the list associated with current_node in graph EQUALS one  
                IF current_node EQUALS target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE to next iteration of the WHILE loop  
                END IF  
            END IF  
            SET possible_moves TO the LENGTH of the list associated with current_node in graph MINUS one IF current_node NOT EQUALS one OTHERWISE zero  
            FOR each neighbor IN the list associated with current_node in graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND the tuple of neighbor current_probability DIVIDED BY possible_moves current_time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN zero  
    END FUNCTION  
END CLASS
