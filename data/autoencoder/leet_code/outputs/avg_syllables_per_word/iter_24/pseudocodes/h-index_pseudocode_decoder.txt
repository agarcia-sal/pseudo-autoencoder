CLASS Solution  
    FUNCTION hIndex(citations)  
        SORT citations IN descending order  
        FOR index FROM zero TO the LENGTH OF citations MINUS one  
            SET current_citation TO element at position index of citations  
            IF current_citation LESS THAN index PLUS one  
                RETURN index  
            END IF  
        END FOR  
        RETURN the LENGTH OF citations  
    END FUNCTION  
END CLASS
