CLASS Solution  
    FUNCTION maxKilledEnemies(grid)  
        IF grid EQUALS empty OR the element at position zero of grid EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero of grid  
        SET max_kills TO zero  
        
        SET row_kills TO zero  
        SET col_kills TO a list of zeros with the LENGTH n  
        
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                
                IF j EQUALS zero OR the element at position j MINUS one of the element at position i of grid EQUALS wall  
                    SET row_kills TO zero  
                    FOR k FROM j TO n MINUS one  
                        IF the element at position k of the element at position i of grid EQUALS enemy  
                            INCREMENT row_kills BY one  
                        ELSE IF the element at position k of the element at position i of grid EQUALS wall  
                            BREAK the inner FOR loop  
                        END IF  
                    END FOR  
                END IF  
                
                IF i EQUALS zero OR the element at position i MINUS one of the element at position j of grid EQUALS wall  
                    SET the element at position j of col_kills TO zero  
                    FOR k FROM i TO m MINUS one  
                        IF the element at position k of the element at position j of grid EQUALS enemy  
                            INCREMENT the element at position j of col_kills BY one  
                        ELSE IF the element at position k of the element at position j of grid EQUALS wall  
                            BREAK the inner FOR loop  
                        END IF  
                    END FOR  
                END IF  
                
                IF the element at position j of the element at position i of grid EQUALS empty cell  
                    SET max_kills TO the greater value BETWEEN max_kills AND row_kills PLUS the element at position j of col_kills  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_kills  
    END FUNCTION  
END CLASS
