CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode WITH value element at position zero of values and left NONE and right NONE  
    SET i TO one  
    SET queue TO AN EMPTY double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO A NEW TreeNode WITH value element at position i of values and left NONE and right NONE  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO A NEW TreeNode WITH value element at position i of values and left NONE and right NONE  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder(self, root)  
        IF root IS None  
            RETURN AN EMPTY list  
        END IF  
        SET column_table TO A DEFAULT DICTIONARY WITH list AS default factory  
        SET queue TO A double ended queue CONTAINING A TUPLE OF root AND zero  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE queue IS NOT EMPTY  
            SET node AND column_index TO THE FIRST ELEMENT REMOVED FROM queue  
            IF node IS NOT None  
                APPEND node.val TO column_table AT KEY column_index  
                IF column_index LESS THAN min_column  
                    SET min_column TO column_index  
                END IF  
                IF column_index GREATER THAN max_column  
                    SET max_column TO column_index  
                END IF  
                APPEND A TUPLE OF node.left AND column_index MINUS one TO queue  
                APPEND A TUPLE OF node.right AND column_index PLUS one TO queue  
            END IF  
        END WHILE  
        SET result TO AN EMPTY list  
        FOR x FROM min_column TO max_column  
            APPEND column_table AT KEY x TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
