CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        SET dp TO new two dimensional list WITH dimensions m plus one BY n plus one FILLED WITH zero  
        FOR each s IN strs  
            SET count_0 TO the number of times the character zero APPEARS IN s  
            SET count_1 TO the number of times the character one APPEARS IN s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp at position i and j TO the GREATER OF dp at position i and j AND dp at position i MINUS count_0 and j MINUS count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp at position m and n  
    END FUNCTION  
END CLASS
