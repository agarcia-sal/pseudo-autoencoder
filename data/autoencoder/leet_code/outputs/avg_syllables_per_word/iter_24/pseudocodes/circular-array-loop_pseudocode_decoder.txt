CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1 val2)  
            RETURN (val1 GREATER THAN zero AND val2 GREATER THAN zero) OR (val1 LESS THAN zero AND val2 LESS THAN zero)  
        END FUNCTION  
      
        SET n TO the LENGTH OF nums  
      
        FOR i FROM zero TO n MINUS one  
            SET slow TO i  
            SET fast TO i  
          
            IF element at position i of nums EQUALS zero  
                CONTINUE  
            END IF  
          
            WHILE true  
                SET slow TO the element at position slow of nums PLUS slow  
                SET slow TO slow MODULO n  
                IF slow LESS THAN zero  
                    INCREMENT slow BY n  
                END IF  
              
                SET fast TO the element at position fast of nums PLUS fast  
                SET fast TO fast MODULO n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  
              
                SET fast TO the element at position fast of nums PLUS fast  
                SET fast TO fast MODULO n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  
              
                IF NOT sameDirection(element at position slow of nums element at position i of nums) OR NOT sameDirection(element at position fast of nums element at position i of nums)  
                    BREAK  
                END IF  
              
                IF slow EQUALS fast  
                    IF slow EQUALS (element at position slow of nums PLUS slow) MODULO n  
                        BREAK  
                    END IF  
                    RETURN true  
                END IF  
            END WHILE  
        END FOR  
      
        RETURN false  
    END FUNCTION  
END CLASS
