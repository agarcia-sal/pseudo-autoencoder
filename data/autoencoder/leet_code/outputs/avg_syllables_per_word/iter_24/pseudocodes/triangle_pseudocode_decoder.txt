CLASS Solution  
    FUNCTION minimumTotal(triangle)  
        FOR row FROM the LENGTH OF triangle MINUS two TO zero STEP minus one  
            FOR col FROM zero TO the LENGTH OF the element at position row of triangle MINUS one  
                SET current_value TO the element at position col of the element at position row of triangle  
                SET left_child TO the element at position col of the element at position row PLUS one of triangle  
                SET right_child TO the element at position col PLUS one of the element at position row PLUS one of triangle  
                SET minimum_child TO the smaller value BETWEEN left_child AND right_child  
                SET updated_value TO current_value PLUS minimum_child  
                SET the element at position col of the element at position row of triangle TO updated_value  
            END FOR  
        END FOR  
        RETURN the element at position zero of the element at position zero of triangle  
    END FUNCTION  
END CLASS
