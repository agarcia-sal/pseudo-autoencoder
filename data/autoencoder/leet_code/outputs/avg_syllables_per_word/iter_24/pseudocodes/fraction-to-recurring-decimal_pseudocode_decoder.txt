CLASS Solution  
    FUNCTION fractionToDecimal(numerator, denominator)  
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero  
            SET sign TO the string containing the minus symbol  
        ELSE  
            SET sign TO the empty string  
        END IF  
        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  
        SET quotient TO the integer division result of numerator DIVIDED BY denominator and remainder TO the remainder of numerator DIVIDED BY denominator  
        SET integer_part TO the string form of quotient  
        IF remainder EQUALS zero  
            RETURN the concatenation of sign and integer_part  
        END IF  
        CREATE an empty dictionary named seen_remainders  
        CREATE an empty list named fractional_part  
        WHILE remainder NOT EQUALS zero  
            IF remainder IS A KEY IN seen_remainders  
                SET start TO the value for key remainder in seen_remainders  
                SET non_repeating TO the concatenation of the elements of fractional_part from position zero TO position start MINUS one  
                SET repeating TO the concatenation of the elements of fractional_part from position start TO the last position  
                RETURN the concatenation of sign integer_part a string containing a period symbol non_repeating a string containing an opening parenthesis repeating a string containing a closing parenthesis  
            END IF  
            ASSIGN to key remainder in seen_remainders the current length of fractional_part  
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient TO the integer division result of remainder DIVIDED BY denominator and remainder TO the remainder of remainder DIVIDED BY denominator  
            APPEND the string form of quotient TO fractional_part  
        END WHILE  
        RETURN the concatenation of sign integer_part a string containing a period symbol and the concatenation of the elements of fractional_part  
    END FUNCTION  
END CLASS
