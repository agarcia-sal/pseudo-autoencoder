CLASS Solution  
    FUNCTION magicalString(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  

        IF n LESS THAN OR EQUAL TO three  
            RETURN one  
        END IF  

        SET s TO list containing one followed by two followed by two  
        SET count TO one  
        SET index TO two  

        WHILE LENGTH OF s LESS THAN n  
            SET next_num TO three MINUS element at position LENGTH OF s MINUS one of s  
            APPEND next_num REPEATED element at position index of s TIMES TO s  

            IF next_num EQUALS one  
                INCREMENT count BY element at position index of s  
            END IF  

            INCREMENT index BY one  
        END WHILE  

        IF LENGTH OF s GREATER THAN n  
            SET excess_length TO LENGTH OF s MINUS n  
            DECREMENT count BY element at position LENGTH OF s MINUS one of s MULTIPLIED BY excess_length  
        END IF  

        RETURN count  
    END FUNCTION  
END CLASS
