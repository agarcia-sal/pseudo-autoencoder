CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of each element in nums  
        SET dp TO a list of zeros with length equal to two raised to the power of ten  
        SET element at position zero of dp TO one  
        
        FOR each num IN the keys of count  
            IF num EQUALS one  
                SET element at position zero of dp TO element at position zero of dp MULTIPLIED BY two raised to the power of the frequency of one in count MODULO the constant modulo  
                CONTINUE to the next iteration  
            END IF  
            IF num NOT IN the set of keys in prime_factors  
                CONTINUE to the next iteration  
            END IF  
            
            SET factors TO the value in prime_factors at key num  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask OR two raised to the power of prime MINUS one  
            END FOR  
            
            FOR m FROM two raised to the power of ten MINUS one DOWN TO zero  
                IF the bitwise AND of m and mask EQUALS zero  
                    SET element at position bitwise OR of m and mask of dp TO the sum of element at position bitwise OR of m and mask of dp PLUS element at position m of dp MULTIPLIED BY the frequency of num in count MODULO the constant modulo  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of elements of dp from position one to the end MODULO the constant modulo  
        RETURN result  
    END FUNCTION  
END CLASS
