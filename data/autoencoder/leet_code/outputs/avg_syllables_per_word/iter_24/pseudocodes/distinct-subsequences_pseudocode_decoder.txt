CLASS Solution  
    FUNCTION numDistinct(s, t)  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
        INITIALIZE dp AS a two dimensional list of zeros with the size of m PLUS one by n PLUS one  
        
        FOR i FROM zero TO m  
            SET element at position i at zero of dp TO one  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of s EQUALS element at position j MINUS one of t THEN  
                    SET element at position i at position j of dp TO element at position i MINUS one at position j MINUS one of dp PLUS element at position i MINUS one at position j of dp  
                ELSE  
                    SET element at position i at position j of dp TO element at position i MINUS one at position j of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position m at position n of dp  
    END FUNCTION  
END CLASS
