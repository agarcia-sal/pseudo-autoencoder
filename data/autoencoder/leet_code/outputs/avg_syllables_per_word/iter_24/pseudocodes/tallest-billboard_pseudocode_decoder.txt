CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO dictionary with key zero AND value zero  
        FOR each rod IN rods  
            SET new_dp TO a new dictionary copied from dp  
            FOR each key d AND value y IN dp  
                SET new_key_one TO d PLUS rod  
                IF new_key_one IS IN new_dp  
                    SET current_value_one TO the value at new_key_one in new_dp  
                    IF current_value_one LESS THAN y  
                        SET the value at new_key_one in new_dp TO y  
                    END IF  
                ELSE  
                    SET the value at new_key_one in new_dp TO y  
                END IF  
                IF d GREATER THAN rod  
                    SET difference TO d MINUS rod  
                ELSE  
                    SET difference TO rod MINUS d  
                END IF  
                IF d LESS THAN rod  
                    SET minimum_value TO d  
                ELSE  
                    SET minimum_value TO rod  
                END IF  
                SET new_value_two TO y PLUS minimum_value  
                IF difference IS IN new_dp  
                    SET current_value_two TO the value at difference in new_dp  
                    IF current_value_two LESS THAN new_value_two  
                        SET the value at difference in new_dp TO new_value_two  
                    END IF  
                ELSE  
                    SET the value at difference in new_dp TO new_value_two  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN the value at zero in dp  
    END FUNCTION  
END CLASS
