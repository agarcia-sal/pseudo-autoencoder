CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO list of string representations of elements in nums  
        
        FUNCTION compare(x, y)  
            IF concatenation of x PLUS concatenation of y GREATER THAN concatenation of y PLUS concatenation of x  
                RETURN negative one  
            ELSE IF concatenation of x PLUS concatenation of y LESS THAN concatenation of y PLUS concatenation of x  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str USING compare AS comparator  
        
        IF element at position one of nums_str EQUALS string zero  
            RETURN string zero  
        END IF  
        
        SET result TO concatenation of all elements in nums_str in order  
        RETURN result  
    END FUNCTION  
END CLASS
