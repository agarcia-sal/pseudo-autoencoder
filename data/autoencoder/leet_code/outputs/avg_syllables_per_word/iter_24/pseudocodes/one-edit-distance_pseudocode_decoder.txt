CLASS Solution  
    FUNCTION isOneEditDistance with parameters s string t string RETURNS boolean  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t THEN  
            RETURN the RESULT OF isOneEditDistance WITH PARAMETERS t AND s  
        END IF  
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        IF diff GREATER THAN one THEN  
            RETURN False  
        END IF  
        IF diff EQUALS zero THEN  
            RETURN the RESULT OF the SUM FOR each pair of characters c1 AND c2 IN s AND t WHERE the character c1 NOT EQUALS the character c2 EQUALS one  
        ELSE  
            FOR index FROM zero TO the LENGTH OF s MINUS one  
                IF the character at position index of s NOT EQUALS the character at position index of t THEN  
                    RETURN whether the substring from position index TO the END OF s EQUALS the substring from the position index PLUS one TO the END OF t  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
