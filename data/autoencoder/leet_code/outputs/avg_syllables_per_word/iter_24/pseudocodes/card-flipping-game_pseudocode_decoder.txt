CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO empty set  
        FOR each pair of element x and element y IN parallel positions of fronts and backs  
            IF element x EQUALS element y THEN  
                ADD element x TO set same  
            END IF  
        END FOR  
        SET all_numbers TO set of all elements in fronts and all elements in backs combined  
        SET good_numbers TO set of all_numbers WITH all elements in same REMOVED  
        IF set good_numbers IS NOT empty THEN  
            SET result TO the smallest element IN set good_numbers  
        ELSE  
            SET result TO zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
