CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours TO the number represented by the part of time before the single colon character  
            SET minutes TO the number represented by the part of time after the single colon character  
            RETURN the result of hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
        
        SET minutes_list TO the result of applying to_minutes to each element in timePoints and collecting results into a list  
        SORT minutes_list in ascending order  
        SET min_diff TO a very large positive number representing infinity  
        FOR i FROM one TO the length of minutes_list MINUS one  
            SET current_difference TO the element at position i of minutes_list MINUS the element at position i MINUS one of minutes_list  
            IF current_difference LESS THAN min_diff  
                SET min_diff TO current_difference  
            END IF  
        END FOR  
        SET circular_difference TO the result of subtracting the element at the last position of minutes_list FROM one thousand four hundred forty PLUS the element at the first position of minutes_list  
        IF circular_difference LESS THAN min_diff  
            SET min_diff TO circular_difference  
        END IF  
        RETURN min_diff  
    END FUNCTION  
END CLASS
