CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        FUNCTION dp(target)  
            IF target EQUALS zero  
                RETURN negative one  
            END IF  
            IF target LESS THAN x  
                RETURN the minimum BETWEEN two MULTIPLIED BY target MINUS one AND two MULTIPLIED BY x MINUS target  
            END IF  
            SET power TO one  
            WHILE x RAISED TO THE power IS LESS THAN OR EQUAL TO target  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET res TO dp OF target MINUS x RAISED TO THE power PLUS power  
            IF x RAISED TO THE power PLUS one MINUS target IS LESS THAN target  
                SET res TO the minimum BETWEEN res AND dp OF x RAISED TO THE power PLUS one MINUS target PLUS power PLUS one  
            END IF  
            RETURN res  
        END FUNCTION  
        RETURN dp OF target  
    END FUNCTION  
END CLASS
