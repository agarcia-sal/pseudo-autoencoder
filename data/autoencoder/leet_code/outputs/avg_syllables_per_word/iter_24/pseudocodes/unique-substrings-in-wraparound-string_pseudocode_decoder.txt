CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a dictionary mapping each character from a to z TO zero  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND the difference between the character code of element at position index of s AND the character code of element at position index MINUS one of s EQUALS one OR element at position index of s EQUALS character a AND element at position index MINUS one of s EQUALS character z  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET the value in max_length at key element at position index of s TO the maximum between the value in max_length at the same key AND current_length  
        END FOR  
        SET total_substrings TO the sum of all values in max_length  
        RETURN total_substrings  
    END FUNCTION  
END CLASS
