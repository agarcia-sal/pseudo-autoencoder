CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO an empty dictionary with default integer values  
        FOR each element i IN nums  
            FOR each element j IN nums  
                SET bitwise_and_result TO element at position i AND element at position j using bitwise AND  
                INCREMENT the value associated with bitwise_and_result in pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each element k IN nums  
            FOR each key pair_and AND its associated count IN pair_and_count  
                IF bitwise AND of pair_and AND k EQUALS zero THEN  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
