CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF the LENGTH OF nums EQUALS zero THEN  
            RETURN zero  
        END IF  
        SET dp TO an empty list  
        FOR each num IN nums  
            SET idx TO the POSITION where num WOULD BE INSERTED IN dp TO MAINTAIN SORT ORDER USING BINARY SEARCH  
            IF idx EQUALS the LENGTH OF dp THEN  
                APPEND num TO dp  
            ELSE  
                SET the element at position idx of dp TO num  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
