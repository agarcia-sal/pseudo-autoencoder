CLASS Solution  
    FUNCTION totalHammingDistance(nums)  
        SET total_distance TO zero  
        FOR index i FROM zero TO thirty one  
            SET count_ones TO zero  
            FOR each number num IN nums  
                INCREMENT count_ones BY the value of element at position i of number num represented as a bit where a bit is shifted rightwards by i positions AND equals one  
            END FOR  
            SET count_zeros TO the LENGTH OF nums MINUS count_ones  
            INCREMENT total_distance BY count_ones MULTIPLIED BY count_zeros  
        END FOR  
        RETURN total_distance  
    END FUNCTION  
END CLASS
