CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        INITIALIZE radiant TO empty double ended queue  
        INITIALIZE dire TO empty double ended queue  
        FOR each index i AND character s IN the sequence senate  
            IF the character s EQUALS the letter R  
                APPEND the value i TO the radiant queue  
            ELSE  
                APPEND the value i TO the dire queue  
            END IF  
        END FOR  
        WHILE the radiant queue IS NOT empty AND the dire queue IS NOT empty  
            REMOVE and ASSIGN the first element of the radiant queue TO r_index  
            REMOVE and ASSIGN the first element of the dire queue TO d_index  
            IF the value r_index IS LESS THAN the value d_index  
                APPEND the value r_index PLUS the LENGTH OF the sequence senate TO the radiant queue  
            ELSE  
                APPEND the value d_index PLUS the LENGTH OF the sequence senate TO the dire queue  
            END IF  
        END WHILE  
        IF the radiant queue IS NOT empty  
            RETURN the word Radiant  
        ELSE  
            RETURN the word Dire  
        END IF  
    END FUNCTION  
END CLASS
