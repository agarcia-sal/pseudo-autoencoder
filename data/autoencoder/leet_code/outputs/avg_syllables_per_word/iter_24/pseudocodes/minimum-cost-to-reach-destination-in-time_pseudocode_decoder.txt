CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO an empty list of n empty lists  
        FOR each element x, y, time IN edges  
            APPEND the pair of y and time TO the element at position x of graph  
            APPEND the pair of x and time TO the element at position y of graph  
        END FOR  
        SET priority queue pq TO a list containing the triple of element at position zero of passingFees, zero, zero  
        SET visited_time TO a list with n elements each set to positive infinity  
        SET element at position zero of visited_time TO zero  
        WHILE pq is not empty  
            REMOVE and ASSIGN the triple current_cost current_city current_time FROM the element with the smallest first value in pq  
            IF current_city EQUALS n MINUS one  
                RETURN current_cost  
            END IF  
            FOR each pair neighbor travel_time IN element at position current_city of graph  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS element at position neighbor of passingFees  
                IF new_time LESS THAN OR EQUAL TO maxTime AND the condition that new_time LESS THAN element at position neighbor of visited_time OR new_cost LESS THAN current_cost holds  
                    SET element at position neighbor of visited_time TO new_time  
                    INSERT the triple of new_cost neighbor new_time INTO pq maintaining the heap property  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
