CLASS Solution  
    FUNCTION findLongestChain(pairs)  
        CALL pairs TO BE SORTED BASED ON the second element of each pair in ascending order  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each start AND end IN pairs  
            IF start IS GREATER THAN current_end THEN  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
