CLASS Solution  
    FUNCTION profitableSchemes(n profit_minimum group profit)  
        SET constant MODULO TO one billion PLUS seven  
        SET two dimensional list dp TO list OF lists OF zeros WITH outer size n PLUS one AND inner size profit_minimum PLUS one  
        FOR index i FROM zero TO n  
            SET element at position zero of list at position i of dp TO one  
        END FOR  
        FOR each element p WITH corresponding element g IN profit WITH corresponding element group  
            FOR index i FROM n DOWN TO g  
                FOR index j FROM profit_minimum DOWN TO zero  
                    SET element at position j of list at position i of dp TO element at position j of list at position i of dp PLUS element at position the greater value BETWEEN zero AND j MINUS p of list at position i MINUS g of dp  
                    SET element at position j of list at position i of dp TO element at position j of list at position i of dp MODULO MODULO  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position profit_minimum of list at position n of dp  
    END FUNCTION  
END CLASS
