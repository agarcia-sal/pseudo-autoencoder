CLASS Solution  
    FUNCTION minMovesToMakePalindrome(s)  
        SET s TO list form of s  
        SET n TO the LENGTH OF s  
        SET moves TO zero  
        
        WHILE the LENGTH OF s GREATER THAN one holds  
            FOR i FROM the LENGTH OF s MINUS one DOWN TO one  
                IF element at position i of s EQUALS element at position zero of s  
                    FOR j FROM i TO the LENGTH OF s MINUS two  
                        SET temporary TO element at position j PLUS one of s  
                        SET element at position j PLUS one of s TO element at position j of s  
                        SET element at position j of s TO temporary  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE element at position zero FROM s  
                    REMOVE element at position the LAST FROM s  
                    BREAK the FOR loop  
                END IF  
            END FOR  
            IF no BREAK occurred in the FOR loop  
                REMOVE element at position zero FROM s  
            END IF  
        END WHILE  
        
        RETURN moves  
    END FUNCTION  
END CLASS
