CLASS Solution  
    FUNCTION minInterval(intervals WITH TYPE List of List of integers, queries WITH TYPE List of integers) RETURNS List of integers  
        SORT the list intervals by the first element of each inner list in ascending order  
        CREATE a list named sorted_queries consisting of pairs where each pair is an element from queries and its corresponding position in queries, sorted by the first element of each pair in ascending order  
        INITIALIZE an empty minimum heap named min_heap that will hold pairs of interval size and interval end  
        INITIALIZE a list named result with the same length as queries where each element is set to negative one  
        SET interval_index TO zero  
        FOR each pair query and original_index IN sorted_queries  
            WHILE interval_index is less than the length of intervals AND the first element of the element at position interval_index of intervals is less than or equal to query  
                SET start TO the first element of the element at position interval_index of intervals  
                SET end TO the second element of the element at position interval_index of intervals  
                SET interval_size TO the quantity end MINUS start PLUS one  
                PUSH the pair of interval_size and end onto min_heap  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE min_heap is not empty AND the second element of the pair at the top of min_heap is less than query  
                REMOVE the top element from min_heap  
            END WHILE  
            IF min_heap is not empty  
                SET the element at position original_index of result TO the first element of the pair at the top of min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
