CLASS Solution  
    FUNCTION maxSlidingWindow(nums parameter LIST OF INTEGER k parameter INTEGER) RETURNS LIST OF INTEGER  
        IF nums EQUALS empty OR k EQUALS zero  
            RETURN empty LIST  
        END IF  
        SET deq TO a new empty double ended queue  
        SET result TO an empty LIST  
        FOR index i FROM zero TO the LENGTH OF nums MINUS one  
            IF the LENGTH OF deq IS GREATER THAN zero AND the FIRST element of deq IS LESS THAN i MINUS k PLUS one  
                REMOVE the FIRST element FROM deq  
            END IF  
            WHILE the LENGTH OF deq IS GREATER THAN zero AND the element at position the LAST element of deq OF nums IS LESS THAN the element at position i OF nums  
                REMOVE the LAST element FROM deq  
            END WHILE  
            APPEND i TO deq  
            IF i IS GREATER THAN OR EQUAL TO k MINUS one  
                APPEND the element at position the FIRST element of deq OF nums TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
