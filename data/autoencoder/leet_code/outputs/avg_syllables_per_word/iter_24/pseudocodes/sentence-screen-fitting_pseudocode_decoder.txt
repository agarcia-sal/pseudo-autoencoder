CLASS Solution  
    FUNCTION wordsTyping(sentence LIST OF STR, rows INT, cols INT) RETURNS INT  
        SET s TO the concatenation of all elements in sentence separated by a single space PLUS a single space  
        SET total_length TO the LENGTH OF s  
        SET position TO zero  
        FOR each element IN a collection with length equal to rows  
            INCREMENT position BY cols  
            IF the element at position MODULO total_length of s EQUALS a single space THEN  
                INCREMENT position BY one  
            ELSE  
                WHILE position GREATER THAN zero AND the element at (position MINUS one) MODULO total_length of s NOT EQUALS a single space  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        RETURN position DIVIDED BY total_length  
    END FUNCTION  
END CLASS
