CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO a two dimensional list where each element is True with dimensions n by n  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET element at position i and position j of f TO False  
                IF the character at position i of s EQUALS the character at position j of s  
                    SET element at position i and position j of f TO the element at position i PLUS one and position j MINUS one of f  
                    IF the element at position i and position j of f IS True AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the substring of s from position k TO position k PLUS mx MINUS one  
    END FUNCTION  
END CLASS
