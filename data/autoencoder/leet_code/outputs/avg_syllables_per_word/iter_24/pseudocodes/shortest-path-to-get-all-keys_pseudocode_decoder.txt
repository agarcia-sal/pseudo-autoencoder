CLASS Solution  
    FUNCTION shortestPathAllKeys(grid)  
        SET directions TO list containing elements of pairs zero and one zero and negative one one and zero and negative one and zero  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET start TO none  
        SET num_keys TO zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i of grid AT position j EQUALS the symbol at sign at  
                    SET start TO pair of values i and j  
                ELSE IF element at position i of grid AT position j IS a lowercase letter  
                    INCREMENT num_keys BY one  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a new double ended queue initialized with a tuple containing the first element of start the second element of start zero zero  
        SET visited TO a new set containing a tuple of the first element of start the second element of start zero  
        WHILE queue is not empty  
            REMOVE the leftmost element of queue and ASSIGN it TO variables x y keys steps  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET cell TO element at position nx of grid AT position ny  
                    IF cell EQUALS the symbol hash  
                        CONTINUE to next iteration of the loop  
                    END IF  
                    IF cell IS an uppercase letter  
                        SET key_needed TO one shifted left by the number equal to the alphabet position of the lowercase equivalent of cell MINUS the alphabet position of letter a  
                        IF bitwise AND of keys and key_needed EQUALS zero  
                            CONTINUE to next iteration of the loop  
                        END IF  
                    END IF  
                    IF cell IS a lowercase letter  
                        SET new_keys TO bitwise OR of keys and one shifted left by the number equal to the alphabet position of cell MINUS the alphabet position of letter a  
                        IF new_keys EQUALS bitwise subtraction of one FROM one shifted left by num_keys  
                            RETURN steps PLUS one  
                        END IF  
                    ELSE  
                        SET new_keys TO keys  
                    END IF  
                    IF tuple of nx ny new_keys IS NOT IN visited  
                        ADD tuple of nx ny new_keys TO visited  
                        APPEND tuple of nx ny new_keys steps PLUS one TO queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
