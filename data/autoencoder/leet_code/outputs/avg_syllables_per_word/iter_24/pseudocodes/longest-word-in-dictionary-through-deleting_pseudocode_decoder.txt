CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator TO an iterator over s  
            FOR each character IN word  
                IF character IS NOT FOUND in iterator in order  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SORT dictionary BY length in descending order AND lexicographical order in ascending order  
        
        FOR each word IN dictionary  
            IF is_subsequence(word, s) EQUALS true  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN empty string  
    END FUNCTION  
END CLASS
