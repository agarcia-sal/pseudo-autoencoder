CLASS Solution  
    FUNCTION repeatedStringMatch(a parameter named a THAT IS a string, b parameter named b THAT IS a string) RETURNS an integer  
        SET min_repeats TO the negative of the integer division RESULT OF the negative of the LENGTH OF b divided BY the LENGTH OF a  
        IF b IS A SUBSTRING OF a REPEATED min_repeats TIMES  
            RETURN min_repeats  
        END IF  
        IF b IS A SUBSTRING OF a REPEATED min_repeats PLUS one TIMES  
            RETURN min_repeats PLUS one  
        END IF  
        RETURN negative one  
    END FUNCTION  
END CLASS
