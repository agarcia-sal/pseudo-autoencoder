CLASS Solution  
    FUNCTION trapRainWater(heightMap)  
        IF heightMap EQUALS nothing OR element at position zero of heightMap EQUALS nothing THEN  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF heightMap  
        SET n TO the LENGTH OF element at position zero of heightMap  
        SET visited TO an empty two-dimensional list of dimensions m BY n each initialized TO false  
        SET min_heap TO an empty list  
        
        FOR i FROM zero TO m MINUS one  
            PUSH the tuple consisting of element at position i of heightMap at position zero AND i AND zero INTO min_heap  
            PUSH the tuple consisting of element at position i of heightMap at position n MINUS one AND i AND n MINUS one INTO min_heap  
            SET element at position i of visited at position zero TO true  
            SET element at position i of visited at position n MINUS one TO true  
        END FOR  
        
        FOR j FROM zero TO n MINUS one  
            PUSH the tuple consisting of element at position zero of heightMap at position j AND zero AND j INTO min_heap  
            PUSH the tuple consisting of element at position m MINUS one of heightMap at position j AND m MINUS one AND j INTO min_heap  
            SET element at position zero of visited at position j TO true  
            SET element at position m MINUS one of visited at position j TO true  
        END FOR  
        
        SET directions TO a list containing the tuples negative one AND zero AND one AND zero AND zero AND negative one AND zero AND one  
        SET water_trapped TO zero  
        
        WHILE min_heap is not empty  
            REMOVE the smallest element from min_heap and DESTRUCTURE it into height AND x AND y  
            
            FOR each tuple dx AND dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx of visited at position ny EQUALS false THEN  
                    SET element at position nx of visited at position ny TO true  
                    INCREMENT water_trapped BY the GREATER OF zero AND height MINUS element at position nx of heightMap at position ny  
                    PUSH the tuple consisting of the GREATER OF height AND element at position nx of heightMap at position ny AND nx AND ny INTO min_heap  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN water_trapped  
    END FUNCTION  
END CLASS
