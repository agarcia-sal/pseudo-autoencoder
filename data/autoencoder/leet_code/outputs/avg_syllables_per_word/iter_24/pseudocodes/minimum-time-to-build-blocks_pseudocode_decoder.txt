CLASS Solution  
    FUNCTION minBuildTime(blocks, split)  
        CALL heapify to convert blocks INTO a minimum heap  
        WHILE the LENGTH of blocks GREATER THAN one  
            SET first TO the element removed from blocks WITH the smallest value  
            SET second TO the next element removed from blocks WITH the smallest value  
            SET combined_time TO split PLUS second  
            INSERT combined_time INTO blocks AS a heap element  
        END WHILE  
        RETURN the element at position zero of blocks  
    END FUNCTION  
END CLASS
