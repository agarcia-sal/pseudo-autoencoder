CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict word_list FUNCTION LIST OF STRING, word1 first_word FUNCTION STRING, word2 second_word FUNCTION STRING) RETURNING INTEGER  
        SET min_distance TO infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO first_word EQUALS second_word  
        
        FOR index i FROM zero TO LENGTH OF word_list MINUS one  
            SET word TO element at position i of word_list  
            
            IF word EQUALS first_word  
                IF same_word  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO i  
            END IF  
            
            IF word EQUALS second_word  
                IF NOT same_word  
                    SET prev_word2 TO i  
                END IF  
            END IF  
            
            IF prev_word1 NOT EQUALS negative one AND prev_word2 NOT EQUALS negative one  
                IF min_distance GREATER THAN absolute value OF prev_word1 MINUS prev_word2  
                    SET min_distance TO absolute value OF prev_word1 MINUS prev_word2  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
