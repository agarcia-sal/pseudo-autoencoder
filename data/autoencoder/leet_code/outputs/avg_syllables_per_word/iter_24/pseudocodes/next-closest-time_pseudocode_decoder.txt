CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters IN time after REMOVING the colon character  
        SET hour TO the integer value of the substring from position zero TO position one of time PLUS the integer value of the substring at position one TO position two of time  
        SET minute TO the integer value of the substring from position three TO position four of time  

        FUNCTION generate_times(digits)  
            SET times TO an empty list  
            FOR each h1 IN digits  
                FOR each h2 IN digits  
                    FOR each m1 IN digits  
                        FOR each m2 IN digits  
                            SET hour_string TO the concatenation of h1 and h2  
                            SET minute_string TO the concatenation of m1 and m2  
                            SET hour_value TO the integer value of hour_string  
                            SET minute_value TO the integer value of minute_string  
                            IF hour_value LESS THAN twenty four AND minute_value LESS THAN sixty  
                                APPEND the tuple of hour_value and minute_value TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the list times sorted in ascending order  
        END FUNCTION  

        SET all_times TO the result of calling generate_times with digits as argument  
        SET current_time_index TO the index of the tuple of hour and minute IN all_times  

        IF current_time_index PLUS one LESS THAN the length of all_times  
            SET next_hour TO the first element of the tuple at position current_time_index PLUS one IN all_times  
            SET next_minute TO the second element of the tuple at position current_time_index PLUS one IN all_times  
        ELSE  
            SET next_hour TO the first element of the tuple at position zero IN all_times  
            SET next_minute TO the second element of the tuple at position zero IN all_times  
        END IF  

        SET formatted_next_hour TO the string representation of next_hour WITH a leading zero if it is a single digit  
        SET formatted_next_minute TO the string representation of next_minute WITH a leading zero if it is a single digit  
        SET formatted_time TO the concatenation of formatted_next_hour, a colon character, and formatted_next_minute  
        RETURN formatted_time  
    END FUNCTION  
END CLASS
