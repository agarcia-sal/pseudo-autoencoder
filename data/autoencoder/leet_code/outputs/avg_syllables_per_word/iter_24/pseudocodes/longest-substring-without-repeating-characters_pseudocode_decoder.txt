CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO empty mapping  
        SET max_length TO zero  
        SET start TO zero  
        FOR each end position AND character IN the enumeration of s  
            IF character EXISTS IN char_index_map AND the value AT character IN char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO the value AT character IN char_index_map PLUS one  
            END IF  
            SET the value AT character IN char_index_map TO end  
            SET max_length TO the greater VALUE BETWEEN max_length AND end MINUS start PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
