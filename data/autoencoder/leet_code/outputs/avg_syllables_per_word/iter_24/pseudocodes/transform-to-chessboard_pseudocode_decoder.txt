CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        
        FOR i FROM zero TO N MINUS one  
            FOR j FROM zero TO N MINUS one  
                IF the element at position i of board AND the element at position j of element at position i of board XOR the element at position zero of element at position j of board XOR the element at position zero of element at position zero of board DOES NOT EQUAL zero  
                    RETURN negative one  
                END IF  
            END FOR  
            
            INCREMENT rowSum BY the element at position zero of element at position i of board  
            INCREMENT colSum BY the element at position i of element at position zero of board  
            
            IF the element at position zero of element at position i of board EQUALS the remainder OF i DIVIDED BY two  
                INCREMENT rowSwap BY one  
            END IF  
            
            IF the element at position i of element at position zero of board EQUALS the remainder OF i DIVIDED BY two  
                INCREMENT colSwap BY one  
            END IF  
        END FOR  
        
        IF the integer division OF N BY two GREATER THAN rowSum OR rowSum GREATER THAN integer division OF the sum OF N PLUS one BY two  
            RETURN negative one  
        END IF  
        
        IF the integer division OF N BY two GREATER THAN colSum OR colSum GREATER THAN integer division OF the sum OF N PLUS one BY two  
            RETURN negative one  
        END IF  
        
        IF the remainder OF N DIVIDED BY two EQUALS one  
            IF the remainder OF rowSwap DIVIDED BY two EQUALS one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            
            IF the remainder OF colSwap DIVIDED BY two EQUALS one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the smaller OF rowSwap AND N MINUS rowSwap  
            SET colSwap TO the smaller OF colSwap AND N MINUS colSwap  
        END IF  
        
        RETURN the integer division OF the sum OF rowSwap PLUS colSwap BY two  
    END FUNCTION  
END CLASS
