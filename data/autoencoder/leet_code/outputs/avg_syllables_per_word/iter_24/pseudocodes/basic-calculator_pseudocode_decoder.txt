CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s)  
            SET stack TO empty list  
            SET sign TO the string plus  
            SET num TO zero  
            
            WHILE the LENGTH OF s GREATER THAN zero  
                SET char TO the element at position zero of s  
                REMOVE the element at position zero FROM s  
                
                IF the character char is a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the numeric value of char  
                END IF  
                
                IF the character char EQUALS the left parenthesis  
                    SET num TO the result of calling helper with s  
                END IF  
                
                IF the character char is NOT a digit AND the character char is NOT a whitespace OR the LENGTH OF s EQUALS zero  
                    
                    IF sign EQUALS the string plus  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS the string minus  
                        APPEND the negative value of num TO stack  
                    ELSE IF sign EQUALS the string asterisk  
                        SET the last element of stack TO the last element of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS the string forward slash  
                        SET the last element of stack TO the integer division of the last element of stack BY num  
                    END IF  
                    
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                
                IF the character char EQUALS the right parenthesis  
                    BREAK the WHILE loop  
                END IF  
            END WHILE  
            
            RETURN the summation of all elements in stack  
        END FUNCTION  
        
        RETURN the result of calling helper with the list formed from the characters of s  
    END FUNCTION  
END CLASS
