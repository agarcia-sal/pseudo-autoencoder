CLASS Solution  
    FUNCTION countComponents WITH PARAMETERS n edges  
        SET graph TO empty map from index ranging FROM zero TO n MINUS one TO empty list  
        FOR each pair a b IN edges  
            APPEND b TO element at position a OF graph  
            APPEND a TO element at position b OF graph  
        END FOR  
        SET visited TO empty set  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN element at position node OF graph  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        SET components TO zero  
        FOR node FROM zero TO n MINUS one  
            IF node IS NOT IN visited THEN  
                CALL dfs WITH node  
                INCREMENT components BY one  
            END IF  
        END FOR  
        RETURN components  
    END FUNCTION  
END CLASS
