CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        WHILE n GREATER THAN length MULTIPLIED BY count  
            SET n TO n MINUS length MULTIPLIED BY count  
            INCREMENT length BY one  
            SET count TO count MULTIPLIED BY ten  
            SET start TO start MULTIPLIED BY ten  
        END WHILE  
        SET start TO start PLUS the INTEGER RESULT OF the INTEGER DIVISION of n MINUS one BY length  
        SET digit_string TO the STRING REPRESENTATION OF start  
        SET position TO the REMAINDER of n MINUS one DIVIDED BY length  
        SET digit_character TO the CHARACTER at position position OF digit_string  
        CONVERT digit_character TO an INTEGER  
        RETURN that INTEGER  
    END FUNCTION  
END CLASS
