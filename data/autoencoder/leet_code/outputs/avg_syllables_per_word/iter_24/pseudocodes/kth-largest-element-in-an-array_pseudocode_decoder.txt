CLASS Solution  
    FUNCTION findKthLargest(nums k)  
        SET min_heap TO empty list  
        FOR each number IN nums  
            INSERT number INTO min_heap USING heap insertion  
            IF the LENGTH OF min_heap GREATER THAN k  
                REMOVE the smallest element FROM min_heap USING heap removal  
            END IF  
        END FOR  
        RETURN the element at position zero OF min_heap  
    END FUNCTION  
END CLASS
