CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        CREATE a new default dictionary named anagrams THAT RETURNS an empty list FOR missing keys  
        FOR each string s IN strs  
            SET sorted_str TO the concatenation of all characters FROM sorting the characters IN string s  
            APPEND string s TO the list corresponding to key sorted_str IN anagrams  
        END FOR  
        RETURN the list of all values IN anagrams  
    END FUNCTION  
END CLASS
