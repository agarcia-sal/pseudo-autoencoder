CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val left right  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode initialized with element at position one of values  
    SET i TO one  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element removed from queue  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None  
            SET node left TO new TreeNode initialized with element at position i of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i of values NOT EQUALS None  
            SET node right TO new TreeNode initialized with element at position i of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF p val NOT EQUALS q val  
        RETURN False  
    ELSE  
        RETURN is_same_tree called with p left and q left AND is_same_tree called with p right and q right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom WITH PARAMETERS self root  
        IF root EQUALS None  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO deque initialized with root  
        WHILE queue is not empty  
            SET level_size TO length of queue  
            SET current_level TO empty list  
            FOR each counter from one TO level_size  
                SET node TO first element removed from queue  
                APPEND node val TO current_level  
                IF node left NOT EQUALS None  
                    APPEND node left TO queue  
                END IF  
                IF node right NOT EQUALS None  
                    APPEND node right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result reversed in order  
    END FUNCTION  
END CLASS
