CLASS Solution  
    FUNCTION encode(s)  
        FUNCTION get_encoded_length(count, substring)  
            RETURN the LENGTH OF the STRING REPRESENTATION OF count PLUS the LENGTH OF substring PLUS two  
        END FUNCTION  

        FUNCTION dp(i, j)  
            SET substring TO the string formed by the characters from position i TO position j inclusive of s  
            SET n TO j MINUS i PLUS one  
            IF n LESS THAN five  
                RETURN substring  
            END IF  

            SET shortest TO substring  
            FOR k FROM one TO the INTEGER DIVISION OF n BY two PLUS one  
                IF n MODULO k EQUALS zero  
                    SET repeat_count TO the INTEGER DIVISION OF n BY k  
                    IF substring EQUALS the concatenation of the substring formed by the characters from position zero TO position k minus one of substring REPEATED repeat_count TIMES  
                        SET encoded_candidate TO the concatenation of the STRING REPRESENTATION OF repeat_count PLUS the symbol representing the open bracket PLUS the result of dp called with parameters i and i PLUS k MINUS one PLUS the symbol representing the close bracket  
                        IF the LENGTH OF encoded_candidate LESS THAN the LENGTH OF shortest  
                            SET shortest TO encoded_candidate  
                        END IF  
                    END IF  
                END IF  
            END FOR  

            FOR k FROM i TO j MINUS one  
                SET left_encoded TO the result of dp called with parameters i and k  
                SET right_encoded TO the result of dp called with parameters k PLUS one and j  
                SET combined TO the concatenation of left_encoded PLUS right_encoded  
                IF the LENGTH OF combined LESS THAN the LENGTH OF shortest  
                    SET shortest TO combined  
                END IF  
            END FOR  

            RETURN shortest  
        END FUNCTION  

        RETURN the result of dp called with parameters zero and the LENGTH OF s MINUS one  
    END FUNCTION  
END CLASS
