CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO count the frequency of each character in s  
        SET max_heap TO an empty list  
        FOR each character AND its frequency IN count  
            APPEND the pair of the frequency MINUS one MULTIPLIED BY negative one and the character TO max_heap  
        END FOR  
        TRANSFORM max_heap into a heap structure  
        
        SET prev_char TO no character  
        SET prev_freq TO zero  
        SET result TO an empty list  
        
        WHILE max_heap is not empty OR prev_freq is less than zero  
            IF max_heap is empty AND prev_freq is less than zero  
                RETURN the empty string  
            END IF  
            
            REMOVE the element with the smallest frequency FROM max_heap AND assign its frequency to freq AND its character to char  
            APPEND char TO result  
            
            INCREMENT freq BY one  
            
            IF prev_freq is less than zero  
                INSERT the pair of prev_freq and prev_char INTO max_heap  
            END IF  
            
            ASSIGN prev_char TO char  
            ASSIGN prev_freq TO freq  
        END WHILE  
        
        RETURN the concatenation of all characters in result  
    END FUNCTION  
END CLASS
