CLASS Solution  
    FUNCTION scoreOfStudents(s parameter STRING, answers parameter LIST OF INTEGER) RETURNS INTEGER  
        FUNCTION evaluate_expression(expression parameter STRING) RETURNS INTEGER  
            SET tokens TO an empty list  
            SET num TO zero  
            FOR each character IN expression  
                IF character IS a digit THEN  
                    SET num TO num MULTIPLIED BY ten PLUS the numeric value of character  
                ELSE  
                    APPEND num TO tokens  
                    APPEND character TO tokens  
                    SET num TO zero  
                END IF  
            END FOR  
            APPEND num TO tokens  
            
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens  
                IF element at position i of tokens EQUALS the multiplication symbol THEN  
                    SET element at position i minus one of tokens TO element at position i minus one of tokens MULTIPLIED BY element at position i plus one of tokens  
                    REMOVE element at positions from i TO i plus one inclusive FROM tokens  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
            
            SET result TO element at position zero of tokens  
            SET i TO one  
            WHILE i LESS THAN the LENGTH OF tokens  
                IF element at position i of tokens EQUALS the addition symbol THEN  
                    SET result TO result PLUS element at position i plus one of tokens  
                END IF  
                INCREMENT i BY two  
            END WHILE  
            
            RETURN result  
        END FUNCTION  

        FUNCTION get_possible_answers(expression parameter STRING) RETURNS SET OF INTEGER  
            IF expression CONSISTS ONLY OF digits THEN  
                RETURN a set containing the integer value of expression  
            END IF  
            
            SET results TO an empty set  
            FOR i FROM one TO the LENGTH OF expression MINUS two STEP two  
                SET left_results TO get_possible_answers(the substring of expression from position zero TO position i minus one)  
                SET right_results TO get_possible_answers(the substring of expression from position i plus one TO the end)  
                FOR each left IN left_results  
                    FOR each right IN right_results  
                        IF element at position i of expression EQUALS the addition symbol THEN  
                            SET result TO left PLUS right  
                        ELSE  
                            SET result TO left MULTIPLIED BY right  
                        END IF  
                        IF result LESS THAN OR EQUAL TO one thousand THEN  
                            ADD result TO results  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  

        SET correct_answer TO evaluate_expression(s)  
        SET possible_answers TO get_possible_answers(s)  
        
        SET points TO zero  
        FOR each answer IN answers  
            IF answer EQUALS correct_answer THEN  
                INCREMENT points BY five  
            ELSE IF answer IS IN possible_answers THEN  
                INCREMENT points BY two  
            END IF  
        END FOR  
        
        RETURN points  
    END FUNCTION  
END CLASS
