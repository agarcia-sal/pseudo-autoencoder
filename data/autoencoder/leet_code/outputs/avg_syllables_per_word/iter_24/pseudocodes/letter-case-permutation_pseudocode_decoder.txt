CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO the LENGTH OF t  
                APPEND the concatenation of all elements in t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH parameter i PLUS one  
            IF the element at position i of t IS an alphabetic character  
                SET the element at position i of t TO the character obtained by toggling the case of the original character at position i of t  
                CALL dfs WITH parameter i PLUS one  
            END IF  
        END FUNCTION  
        SET t TO the list formed by each character of s  
        SET ans TO an empty list  
        CALL dfs WITH parameter zero  
        RETURN ans  
    END FUNCTION  
END CLASS
