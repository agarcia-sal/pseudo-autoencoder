CLASS Solution  
    FUNCTION countPalindromicSubsequences(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        
        SET next_occurrence TO a list of four lists each containing n elements all SET TO negative one  
        SET prev_occurrence TO a list of four lists each containing n elements all SET TO negative one  
        
        FOR c FROM zero TO three  
            SET char TO the character corresponding to the code point of the character a PLUS c  
            SET last TO negative one  
            FOR i FROM zero TO n MINUS one  
                IF the element at position i of s EQUALS char THEN  
                    SET last TO i  
                END IF  
                SET the element at position i of the element at position c of prev_occurrence TO last  
            END FOR  
            
            SET last TO negative one  
            FOR i FROM n MINUS one DOWN TO zero  
                IF the element at position i of s EQUALS char THEN  
                    SET last TO i  
                END IF  
                SET the element at position i of the element at position c of next_occurrence TO last  
            END FOR  
        END FOR  
        
        SET dp TO a list of n lists each containing n elements all SET TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET the element at position i of dp at position i TO one  
        END FOR  
        
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                FOR c FROM zero TO three  
                    SET char TO the character corresponding to the code point of the character a PLUS c  
                    SET left TO the element at position i of the element at position c of next_occurrence  
                    SET right TO the element at position j of the element at position c of prev_occurrence  
                    
                    IF left GREATER THAN j OR right LESS THAN i THEN  
                        CONTINUE to next iteration of the innermost FOR loop  
                    ELSE IF left EQUALS right THEN  
                        SET the element at position i of dp at position j TO the element at position i of dp at position j PLUS one MODULO MOD  
                    ELSE  
                        SET the element at position i of dp at position j TO the element at position i of dp at position j PLUS the element at position left PLUS one of dp at position right MINUS one PLUS two MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN the element at position zero of dp at position n MINUS one  
    END FUNCTION  
END CLASS
