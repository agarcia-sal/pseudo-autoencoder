CLASS Solution  
    FUNCTION crackSafe(n k)  
        SET visited TO an empty set  
        SET sequence TO an empty list  
        
        FUNCTION dfs(node)  
            FOR i FROM zero TO k MINUS one  
                SET edge TO the concatenation of node AND the string representation of i  
                IF edge NOT IN visited  
                    ADD edge TO visited  
                    CALL dfs WITH the substring of edge from position two TO the end  
                    APPEND the string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO the string consisting of n MINUS one repetitions of the character zero  
        CALL dfs WITH start_node  
        RETURN the concatenation of the elements in sequence joined AS a string PLUS start_node  
    END FUNCTION  
END CLASS
