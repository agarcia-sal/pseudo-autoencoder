CLASS Solution  
    FUNCTION hIndex WITH PARAMETER citations  
        SET n TO the LENGTH OF citations  
        SET left TO zero  
        SET right TO n MINUS one  
        SET h_index TO zero  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS the DIFFERENCE BETWEEN right AND left DIVIDED BY two  
            IF the ELEMENT at POSITION mid of citations IS GREATER THAN OR EQUAL TO n MINUS mid  
                SET h_index TO n MINUS mid  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        
        RETURN h_index  
    END FUNCTION  
END CLASS
