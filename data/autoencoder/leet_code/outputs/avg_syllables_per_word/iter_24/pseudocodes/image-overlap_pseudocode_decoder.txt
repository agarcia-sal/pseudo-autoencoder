CLASS Solution  
    FUNCTION largestOverlap(img1 PARAMETER, img2 PARAMETER)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF element at position i of img1 AND element at position j of that element EQUALS one  
                    APPEND a pair consisting of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF element at position i of img2 AND element at position j of that element EQUALS one  
                    APPEND a pair consisting of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a mapping from pairs to integers with default value zero  
        FOR each pair consisting of x1 AND y1 IN points1  
            FOR each pair consisting of x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value at key consisting of dx AND dy in overlap_count BY one  
            END FOR  
        END FOR  
        SET maximum_overlap TO zero  
        FOR each value IN the values of overlap_count  
            IF value GREATER THAN maximum_overlap  
                SET maximum_overlap TO value  
            END IF  
        END FOR  
        RETURN maximum_overlap  
    END FUNCTION  
END CLASS
