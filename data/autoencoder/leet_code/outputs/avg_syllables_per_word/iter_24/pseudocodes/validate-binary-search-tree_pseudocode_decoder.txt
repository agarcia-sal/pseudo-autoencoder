CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS NOTHING  
        RETURN NOTHING  
    END IF  
    SET root TO A NEW TreeNode WITH the element at position one of values  
    SET i TO one PLUS zero  
    SET queue TO AN EMPTY DEQUE  
    APPEND root TO queue  
    WHILE queue HOLDS ELEMENTS  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i LESS THAN THE LENGTH OF values AND the element at position i of values DOES NOT EQUAL NOTHING  
            SET node.left TO A NEW TreeNode WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN THE LENGTH OF values AND the element at position i of values DOES NOT EQUAL NOTHING  
            SET node.right TO A NEW TreeNode WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS NOTHING AND q EQUALS NOTHING  
        RETURN True  
    ELSE IF p EQUALS NOTHING OR q EQUALS NOTHING  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree OF p.left AND q.left AND is_same_tree OF p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low, high)  
            IF node EQUALS NOTHING  
                RETURN True  
            END IF  
            IF node.val LESS THAN OR EQUAL TO low OR node.val GREATER THAN OR EQUAL TO high  
                RETURN False  
            END IF  
            RETURN validate OF node.left WITH PARAMETERS low AND node.val AND validate OF node.right WITH PARAMETERS node.val AND high  
        END FUNCTION  
        RETURN validate OF root  
    END FUNCTION  
END CLASS
