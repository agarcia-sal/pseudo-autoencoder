CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET head TO A new ListNode created with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values from position one TO the last position  
        SET node TO A new ListNode created with val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None THEN  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None THEN  
        RETURN False  
    END IF  
    RETURN the condition that the val member of p1 EQUALS the val member of p2 AND the recursive call of is_same_list with the next member of p1 AND the next member of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN None  
    END IF  
    SET root TO A new TreeNode created with the element at position zero of values  
    SET i TO one  
    SET queue TO A new empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        REMOVE the element from the left end of queue AND SET it TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET the left member of node TO A new TreeNode created with the element at position i of values  
            APPEND the left member of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None THEN  
            SET the right member of node TO A new TreeNode created with the element at position i of values  
            APPEND the right member of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None THEN  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None THEN  
        RETURN False  
    ELSE IF the val member of p NOT EQUALS the val member of q THEN  
        RETURN False  
    ELSE  
        RETURN the recursive call of is_same_tree with the left member of p AND the left member of q AND the recursive call of is_same_tree with the right member of p AND the right member of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO the LENGTH OF s THEN  
                RETURN None AND index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the LENGTH OF s AND (the element at position index of s is a digit OR the element at position index of s EQUALS the character minus)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO A new TreeNode created with the integer value parsed from the substring from position start TO position index of s  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character left parenthesis THEN  
                SET node.left AND index TO the recursive call of parse_tree with index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character left parenthesis THEN  
                SET node.right AND index TO the recursive call of parse_tree with index PLUS one  
            END IF  
            IF index LESS THAN the LENGTH OF s AND the element at position index of s EQUALS the character right parenthesis THEN  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND unused TO the call of parse_tree with zero  
        RETURN root  
    END FUNCTION  
END CLASS
