CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO empty dictionary mapping each key TO an empty list  
        FOR each path IN paths  
            SET parts TO the split of path by whitespace  
            SET directory TO element at position zero of parts  
            FOR each file_info IN elements of parts from position one to the last element  
                SET file_name_and_content TO the split of file_info by the opening parenthesis word  
                SET file_name TO element at position zero of file_name_and_content  
                SET content TO substring of file_info from position after the first opening parenthesis TO position before the last character of file_info  
                APPEND the concatenation of directory word PLUS the slash word PLUS file_name TO the list referenced by content in content_to_paths  
            END FOR  
        END FOR  
        SET result TO empty list  
        FOR each list_of_paths IN the values of content_to_paths  
            IF length of list_of_paths GREATER THAN one  
                APPEND list_of_paths TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
