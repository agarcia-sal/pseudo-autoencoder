CLASS Solution  
    FUNCTION largestNumber(cost parameter as list of integers target parameter as integer) RETURNS string  
        SET cost_to_digit TO an empty dictionary  
        FOR index FROM zero TO the LENGTH OF cost MINUS one  
            SET current_cost TO element at position index of cost  
            SET digit_string TO the STRING REPRESENTATION OF index PLUS one  
            ASSIGN digit_string TO key current_cost IN cost_to_digit  
        END FOR  
        
        SET dp TO a list starting with an empty string FOLLOWED BY target number of elements each equal to the string representing negative one  
        
        FOR t FROM one TO target  
            FOR each key c AND value d IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND element at position t MINUS c of dp NOT EQUALS the string representing negative one THEN  
                    SET candidate TO the concatenation of d FOLLOWED BY element at position t MINUS c of dp  
                    IF element at position t of dp EQUALS the string representing negative one OR the LENGTH OF candidate GREATER THAN the LENGTH OF element at position t of dp OR the LENGTH OF candidate EQUALS the LENGTH OF element at position t of dp AND candidate GREATER THAN element at position t of dp THEN  
                        SET element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF element at position target of dp NOT EQUALS the string representing negative one THEN  
            RETURN element at position target of dp  
        ELSE  
            RETURN the string representing zero  
        END IF  
    END FUNCTION  
END CLASS
