CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  

        CREATE a two dimensional list called is_palindrome WITH n elements each containing n elements set TO false  

        FOR i FROM zero TO n MINUS one  
            SET element at position i of is_palindrome at position i TO true  
        END FOR  

        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET element at position start of is_palindrome at position end TO ELEMENT at position start of s EQUALS ELEMENT at position end of s  
                ELSE  
                    SET element at position start of is_palindrome at position end TO (ELEMENT at position start of s EQUALS ELEMENT at position end of s) AND element at position start PLUS one of is_palindrome at position end MINUS one  
                END IF  
            END FOR  
        END FOR  

        CREATE a list called min_cuts WITH n elements set TO zero  

        FOR i FROM zero TO n MINUS one  
            IF element at position zero of is_palindrome at position i  
                SET element at position i of min_cuts TO zero  
            ELSE  
                SET element at position i of min_cuts TO infinity  
                FOR j FROM one TO i  
                    IF element at position j of is_palindrome at position i AND element at position j MINUS one of min_cuts PLUS one LESS THAN element at position i of min_cuts  
                        SET element at position i of min_cuts TO element at position j MINUS one of min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  

        RETURN element at position n MINUS one of min_cuts  
    END FUNCTION  
END CLASS
