CLASS Solution  
    FUNCTION kEmptySlots(bulbs LIST OF integer, k integer) RETURNS integer  
        SET bloom_days TO empty list  
        FOR each day FROM one TO the LENGTH OF bulbs  
            SET position TO element at position day MINUS one of bulbs  
            SET i TO the position to insert position in bloom_days to keep it sorted from left using binary search  
            IF i GREATER THAN zero AND position MINUS element at position i MINUS one of bloom_days MINUS one EQUALS k THEN  
                RETURN day  
            END IF  
            IF i LESS THAN the LENGTH OF bloom_days AND element at position i of bloom_days MINUS position MINUS one EQUALS k THEN  
                RETURN day  
            END IF  
            INSERT position AT position i IN bloom_days  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
