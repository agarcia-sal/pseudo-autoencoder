CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO a sorted list obtained by pairing the first element of each interval WITH its position in the intervals list FOR each interval and its position IN intervals  
        SET result TO an empty list  
        FOR each interval in intervals  
            SET end TO the second element of interval  
            SET idx TO the position found by binary searching start_times FOR the first pair whose first element is greater than or equal to end  
            IF idx IS LESS THAN the length of start_times THEN  
                APPEND the second element of the element at position idx of start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
