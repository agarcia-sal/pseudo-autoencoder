CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the string representation of count TO path  
                END IF  
                APPEND the concatenation of all elements in path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH index PLUS one AND path concatenated WITH the string representation of count AND the element at position index of word AND zero AS count  
                ELSE  
                    CALL backtrack WITH index PLUS one AND path concatenated WITH the element at position index of word AND zero AS count  
                END IF  
                CALL backtrack WITH index PLUS one AND path AND count PLUS one AS count  
            END IF  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL backtrack WITH zero AND an empty list AND zero AS count  
        RETURN result  
    END FUNCTION  
END CLASS
