CLASS Solution  
    FUNCTION findShortestWay(maze, ball, hole)  
        SET directions TO the LIST OF four ELEMENTS each containing movement DELTAS and corresponding DIRECTION LETTER namely zero ZERO and one ONE and one ONE and zero ZERO and zero ZERO and NEGATIVE ONE and NEGATIVE ONE and zero ZERO and zero ZERO and one ONE    
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET start TO the TUPLE constructed from ball  
        SET hole TO the TUPLE constructed from hole  
        SET pq TO a LIST containing a SINGLE ELEMENT which is a TUPLE of zero DISTANCE and an EMPTY STRING PATH and start POSITION  
        SET visited TO an EMPTY SET  
        
        WHILE pq IS NOT EMPTY  
            EXTRACT dist PATH and a TUPLE containing x and y FROM the REMOVAL of the SMALLEST ELEMENT in pq according to DISTANCE using HEAP POP operation  
            
            IF the TUPLE of x and y EQUALS hole THEN  
                RETURN PATH  
            END IF  
            
            IF the TUPLE of x and y IS IN visited THEN  
                CONTINUE to the NEXT iteration of the WHILE loop  
            END IF  
            
            ADD the TUPLE of x and y TO visited  
            
            FOR each ELEMENT in directions DESTRUCTURED AS dx dy and direction  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE the SUM of new_x and dx IS GREATER THAN OR EQUAL TO zero AND the SUM of new_x and dx IS LESS THAN m AND the SUM of new_y and dy IS GREATER THAN OR EQUAL TO zero AND the SUM of new_y and dy IS LESS THAN n AND the ELEMENT at position new_x PLUS dx of the ELEMENT at position new_y PLUS dy of maze EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                    IF the TUPLE of new_x and new_y EQUALS hole THEN  
                        BREAK out of the WHILE loop  
                    END IF  
                END WHILE  
                
                IF the TUPLE of new_x and new_y IS NOT IN visited THEN  
                    PUSH INTO pq a TUPLE containing the SUM of dist and count DISTANCE and the CONCATENATION of PATH and direction PATH and the TUPLE of new_x and new_y POSITION using HEAP PUSH operation  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the STRING impossible  
    END FUNCTION  
END CLASS
