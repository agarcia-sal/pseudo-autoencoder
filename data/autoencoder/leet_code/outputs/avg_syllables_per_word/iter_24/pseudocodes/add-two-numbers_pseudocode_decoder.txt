CLASS ListNode  
    FUNCTION __init__(self val SET TO zero next SET TO None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF the collection values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO a new ListNode CREATED WITH the element at position one of values  
    SET p TO head  
    FOR each val IN the collection values STARTING FROM the element at position two TO the last element  
        SET node TO a new ListNode CREATED WITH val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1 p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the condition that the value of p1 EQUALS the value of p2 AND the result of is_same_list CALLED WITH the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1 l2)  
        SET dummy_head TO a new ListNode CREATED WITH zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry IS NOT zero  
            IF l1 IS NOT None  
                SET val1 TO the value of l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT None  
                SET val2 TO the value of l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO the integer division of total BY the number ten  
            SET the next element of current TO a new ListNode CREATED WITH the remainder of total DIVIDED BY the number ten  
            SET current TO the next element of current  
            
            IF l1 IS NOT None  
                SET l1 TO the next element of l1  
            END IF  
            IF l2 IS NOT None  
                SET l2 TO the next element of l2  
            END IF  
        END WHILE  
        
        RETURN the next element of dummy_head  
    END FUNCTION  
END CLASS
