CLASS Solution  
    FUNCTION kthSmallest(mat AS list of list of integers k AS integer) RETURNS integer  
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF the element at position zero of mat  
        SET initial_sum TO zero  
        FOR each row IN mat  
            INCREMENT initial_sum BY the element at position zero of row  
        END FOR  
        SET initial_indices TO a tuple consisting of zero repeated m times  
        SET min_heap TO a list containing a pair of initial_sum AND initial_indices  
        SET visited TO a set containing initial_indices  
        
        WHILE k GREATER THAN zero  
            SET current_pair TO the removal of the smallest element from min_heap  
            SET current_sum TO the first element of current_pair  
            SET indices TO the second element of current_pair  
            DECREMENT k BY one  
            
            IF k EQUALS zero  
                RETURN current_sum  
            END IF  
            
            FOR i FROM zero TO m MINUS one  
                IF the element at position i of indices PLUS one LESS THAN n  
                    SET new_indices TO a mutable copy of indices  
                    INCREMENT the element at position i of new_indices BY one  
                    CONVERT new_indices TO a tuple new_indices_tuple  
                    SET old_index TO the element at position i of indices  
                    SET new_index TO the element at position i of new_indices  
                    SET new_sum TO current_sum MINUS the element at position old_index of the element at position i of mat PLUS the element at position new_index of the element at position i of mat  
                    
                    IF new_indices_tuple NOT IN visited  
                        ADD new_indices_tuple TO visited  
                        PUSH the pair of new_sum AND new_indices_tuple ONTO min_heap  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN a negative one representation  
    END FUNCTION  
END CLASS
