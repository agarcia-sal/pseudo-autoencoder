CLASS Solution  
    FUNCTION longestPath(parent, s)  
        ASSIGN reference TO an empty mapping of lists named tree  
        FOR each index i AND element p IN parent  
            IF p NOT EQUALS negative one  
                APPEND i TO the list element at position p of tree  
            END IF  
        END FOR  
        SET result TO one  
        FUNCTION dfs(node)  
            SET max1 TO zero  
            SET max2 TO zero  
            FOR each child IN the list element at position node of tree  
                SET child_length TO the result of calling dfs with child  
                IF the character at position child of s NOT EQUALS the character at position node of s  
                    IF child_length GREATER THAN max1  
                        SET max2 TO max1  
                        SET max1 TO child_length  
                    ELSE IF child_length GREATER THAN max2  
                        SET max2 TO child_length  
                    END IF  
                END IF  
            END FOR  
            SET result TO the greater value of result AND max1 PLUS max2 PLUS one  
            RETURN max1 PLUS one  
        END FUNCTION  
        CALL dfs with zero  
        RETURN result  
    END FUNCTION  
END CLASS
