CLASS Solution  
    FUNCTION stoneGameV(stoneValue)  
        SET prefix_sum TO a list of zero elements with the length of the length of stoneValue PLUS one  
        FOR index FROM zero TO the length of stoneValue MINUS one  
            SET element at position index PLUS one of prefix_sum TO the element at position index of prefix_sum PLUS the element at position index of stoneValue  
        END FOR  

        FUNCTION get_range_sum(left, right)  
            RETURN the element at position right PLUS one of prefix_sum MINUS the element at position left of prefix_sum  
        END FUNCTION  

        FUNCTION dp(left, right)  
            IF left EQUALS right  
                RETURN zero  
            END IF  
            SET max_score TO zero  
            FOR i FROM left TO right MINUS one  
                SET left_sum TO get_range_sum(left, i)  
                SET right_sum TO get_range_sum(i PLUS one, right)  
                IF left_sum LESS THAN right_sum  
                    SET max_score TO the maximum value among max_score AND left_sum PLUS dp(left, i)  
                ELSE IF left_sum GREATER THAN right_sum  
                    SET max_score TO the maximum value among max_score AND right_sum PLUS dp(i PLUS one, right)  
                ELSE  
                    SET max_score TO the maximum value among max_score AND left_sum PLUS dp(left, i) AND right_sum PLUS dp(i PLUS one, right)  
                END IF  
            END FOR  
            RETURN max_score  
        END FUNCTION  

        RETURN dp(zero, the length of stoneValue MINUS one)  
    END FUNCTION  
END CLASS
