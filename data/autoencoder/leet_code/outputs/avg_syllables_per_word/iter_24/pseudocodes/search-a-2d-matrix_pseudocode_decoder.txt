CLASS Solution  
    FUNCTION searchMatrix(matrix matrixParameter TARGET targetParameter) RETURNS boolean  
        IF matrixParameter EQUALS an empty collection OR element at position zero of matrixParameter EQUALS an empty collection THEN  
            RETURN boolean false  
        END IF  
        SET m TO the LENGTH OF matrixParameter  
        SET n TO the LENGTH OF element at position zero of matrixParameter  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER DIVISION of left PLUS right BY two  
            SET mid_value TO element at position the INTEGER DIVISION of mid BY n of matrixParameter AT position the REMAINDER of mid DIVIDED BY n  
            IF mid_value EQUALS targetParameter THEN  
                RETURN boolean true  
            ELSE IF mid_value LESS THAN targetParameter THEN  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN boolean false  
    END FUNCTION  
END CLASS
