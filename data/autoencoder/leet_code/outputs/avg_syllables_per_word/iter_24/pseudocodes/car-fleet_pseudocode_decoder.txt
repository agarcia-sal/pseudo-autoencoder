CLASS Solution  
    FUNCTION carFleet(target position speed)  
        SET list_of_pairs TO list of pairs made by pairing element at position in position WITH element at position in speed  
        SORT list_of_pairs in descending order based on the element of position in each pair  
        SET time_to_target TO empty list  
        FOR each pair IN list_of_pairs  
            SET position_element TO the element at position zero of pair  
            SET speed_element TO the element at position one of pair  
            SET time TO position target MINUS position_element DIVIDED BY speed_element  
            APPEND time TO time_to_target  
        END FOR  
        
        SET fleets TO zero  
        SET current_max_time TO zero  
        
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
        
        RETURN fleets  
    END FUNCTION  
END CLASS
