CLASS Solution  
    FUNCTION getXORSum(arr1 arr of integers arr2 arr of integers) RETURNS integer  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        FOR each num IN arr1  
            SET xor_sum_arr1 TO the result of the bitwise XOR operation of xor_sum_arr1 AND num  
        END FOR  
        FOR each num IN arr2  
            SET xor_sum_arr2 TO the result of the bitwise XOR operation of xor_sum_arr2 AND num  
        END FOR  
        RETURN the result of the bitwise AND operation of xor_sum_arr1 AND xor_sum_arr2  
    END FUNCTION  
END CLASS
