CLASS Solution  
    FUNCTION minimumTime(n, relations, time)  
        ASSIGN reference TO a new mapping of list  
        FOR each pair of previous course AND next course IN relations  
            APPEND previous course TO the list associated with next course in the mapping  
        END FOR  
        
        ASSIGN reference TO a list of negative one of length n PLUS one  
        
        FUNCTION dfs(course)  
            IF element at position course of the list DOES NOT EQUAL negative one  
                RETURN element at position course of the list  
            END IF  
            
            SET maximum prerequisite time TO zero  
            FOR each previous course IN the list associated with course in the mapping  
                SET current prerequisite time TO the result of dfs with previous course  
                IF current prerequisite time GREATER THAN maximum prerequisite time  
                    SET maximum prerequisite time TO current prerequisite time  
                END IF  
            END FOR  
            
            SET element at position course of the list TO maximum prerequisite time PLUS element at position course MINUS one of time  
            RETURN element at position course of the list  
        END FUNCTION  
        
        SET total minimum time TO zero  
        FOR course FROM one TO n  
            SET current course time TO the result of dfs with course  
            IF current course time GREATER THAN total minimum time  
                SET total minimum time TO current course time  
            END IF  
        END FOR  
        
        RETURN total minimum time  
    END FUNCTION  
END CLASS
