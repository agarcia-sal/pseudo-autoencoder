CLASS Solution  
    FUNCTION largestPalindrome(n)  
        IF n EQUALS one  
            RETURN nine  
        END IF  
        
        SET upper_limit TO ten MULTIPLIED BY ten MULTIPLIED BY ... repeated n times MINUS one  
        SET lower_limit TO ten MULTIPLIED BY ten MULTIPLIED BY ... repeated n minus one times  
        
        SET max_palindrome TO zero  
        
        FOR i FROM upper_limit TO lower_limit DECREMENTING BY one  
            FOR j FROM i TO lower_limit DECREMENTING BY one  
                SET product TO the multiplication of i and j  
                IF product LESS THAN OR EQUAL TO max_palindrome  
                    BREAK inner loop  
                END IF  
                SET str_product TO the string representation of product  
                IF str_product IS EQUAL TO the reversal of str_product  
                    SET max_palindrome TO product  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_palindrome MODULO one thousand three hundred thirty seven  
    END FUNCTION  
END CLASS
