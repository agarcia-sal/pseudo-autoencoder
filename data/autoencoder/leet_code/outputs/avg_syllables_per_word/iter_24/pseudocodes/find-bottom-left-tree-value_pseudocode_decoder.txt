CLASS TreeNode  
    FUNCTION __init__(self WITH val WITH left WITH right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized with the element at position one of values  
    SET i TO one  
    SET queue TO an empty sequence with first-in-first-out behavior  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        REMOVE the first element from queue and ASSIGN it TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p WITH q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value property of p NOT EQUALS the value property of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree with the left property of p and the left property of q AND the result of calling is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self WITH root)  
        SET queue TO a sequence with first-in-first-out behavior initialized with root as its only element  
        SET leftmost_value TO None  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            FOR i FROM zero TO level_size MINUS one  
                REMOVE the first element from queue and ASSIGN it TO node  
                IF i EQUALS zero  
                    SET leftmost_value TO the value property of node  
                END IF  
                IF the left property of node IS NOT None  
                    APPEND the left property of node TO queue  
                END IF  
                IF the right property of node IS NOT None  
                    APPEND the right property of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
