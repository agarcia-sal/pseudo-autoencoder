CLASS Solution  
    FUNCTION restoreIpAddresses(s)  
        FUNCTION is_valid(segment)  
            RETURN the LENGTH OF segment EQUALS one OR the element at position zero of segment NOT EQUALS the character zero AND the INTEGER VALUE OF segment LESS THAN OR EQUAL TO two hundred and fifty five  
        END FUNCTION  
        
        FUNCTION backtrack(start, path)  
            IF the LENGTH OF path EQUALS four  
                IF start EQUALS the LENGTH OF s  
                    APPEND the STRING CREATED BY JOINING the ELEMENTS OF path WITH the CHARACTER dot TO result  
                END IF  
                RETURN  
            END IF  
            
            FOR length FROM one TO three  
                IF start PLUS length LESS THAN OR EQUAL TO the LENGTH OF s  
                    SET segment TO the SUBSTRING OF s FROM POSITION start TO POSITION start PLUS length MINUS one  
                    IF is_valid(segment)  
                        CALL backtrack WITH ARGUMENTS start PLUS length AND path WITH segment APPENDED  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL backtrack WITH ARGUMENTS zero AND an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
