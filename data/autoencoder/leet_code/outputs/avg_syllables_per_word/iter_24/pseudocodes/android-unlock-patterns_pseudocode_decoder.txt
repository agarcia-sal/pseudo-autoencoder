CLASS Solution  
    FUNCTION numberOfPatterns(m, n)  
        SET jumps TO a mapping where each pair of dot positions that require a middle dot to be visited first is associated with that middle dot as follows  
            pair one and three maps to two  
            pair one and seven maps to four  
            pair one and nine maps to five  
            pair two and eight maps to five  
            pair three and one maps to two  
            pair three and seven maps to five  
            pair three and nine maps to six  
            pair four and six maps to five  
            pair five and five maps to five  
            pair six and four maps to five  
            pair seven and one maps to four  
            pair seven and three maps to five  
            pair seven and nine maps to eight  
            pair eight and two maps to five  
            pair nine and one maps to five  
            pair nine and three maps to six  
            pair nine and seven maps to eight  
        
        FUNCTION countPatternsFromDot(current, visited, length)  
            IF length EQUALS n THEN  
                RETURN one  
            END IF  
            SET count TO one if length is greater than or equal to m otherwise zero  
            FOR next_dot FROM one TO nine  
                IF element at position next_dot of visited is not true THEN  
                    SET jump TO the element in jumps associated with the pair current and next_dot if any, otherwise null  
                    IF jump is null OR element at position jump of visited is true THEN  
                        SET element at position next_dot of visited TO true  
                        INCREMENT count BY the result of calling countPatternsFromDot with arguments next_dot, visited, length PLUS one  
                        SET element at position next_dot of visited TO false  
                    END IF  
                END IF  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET visited TO a list of ten elements all set to false  
        SET result TO zero  
        FOR i FROM one TO nine  
            SET element at position i of visited TO true  
            INCREMENT result BY the result of calling countPatternsFromDot with arguments i, visited, one  
            SET element at position i of visited TO false  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
