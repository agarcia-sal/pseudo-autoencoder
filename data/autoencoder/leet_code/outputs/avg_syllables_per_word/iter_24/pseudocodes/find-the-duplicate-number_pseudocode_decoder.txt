CLASS Solution  
    FUNCTION findDuplicate(nums)  
        FUNCTION f(x)  
            SET count TO zero  
            FOR each value IN nums  
                IF value LESS THAN OR EQUAL TO x THEN  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            IF count GREATER THAN x THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        RETURN position of first element IN range from zero TO length of nums WHERE f of element is true  
    END FUNCTION  
END CLASS
