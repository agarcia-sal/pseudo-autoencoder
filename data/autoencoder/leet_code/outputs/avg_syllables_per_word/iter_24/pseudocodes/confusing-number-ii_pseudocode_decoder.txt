CLASS Solution  
    FUNCTION confusingNumberII(n)  
        SET rotate_map TO a mapping of zero mapped to zero one mapped to one six mapped to nine eight mapped to eight nine mapped to six  
        SET valid_digits TO list of zero one six eight nine  
        
        FUNCTION is_confusing(num)  
            SET original_num TO num  
            SET rotated_num TO zero  
            WHILE num GREATER THAN zero  
                SET digit TO the remainder of num DIVIDED BY ten  
                SET rotated_num TO rotated_num MULTIPLIED BY ten PLUS the value corresponding to digit in rotate_map  
                SET num TO the quotient of num DIVIDED BY ten without remainder  
            END WHILE  
            RETURN rotated_num NOT EQUALS original_num  
        END FUNCTION  
        
        FUNCTION count_confusing_numbers(limit current)  
            IF current GREATER THAN limit  
                RETURN zero  
            END IF  
            SET count TO one IF current NOT EQUALS zero AND is_confusing(current) OTHERWISE zero  
            FOR each digit IN valid_digits  
                SET new_number TO current MULTIPLIED BY ten PLUS digit  
                IF new_number EQUALS zero  
                    CONTINUE to next digit  
                END IF  
                INCREMENT count BY the result of count_confusing_numbers(limit new_number)  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        RETURN count_confusing_numbers(n zero)  
    END FUNCTION  
END CLASS
