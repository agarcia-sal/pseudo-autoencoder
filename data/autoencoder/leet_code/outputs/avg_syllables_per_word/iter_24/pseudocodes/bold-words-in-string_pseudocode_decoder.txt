CLASS Solution  
    FUNCTION boldWords(words, s)  
        SET bold TO a list of false values with the length equal to the length of s  
        FOR each word IN words  
            SET start TO the position of the first occurrence of word in s  
            WHILE start NOT EQUALS negative one  
                FOR index FROM start TO start PLUS the length of word MINUS one  
                    SET the element at position index of bold TO true  
                END FOR  
                SET start TO the position of the first occurrence of word in s STARTING FROM start PLUS one  
            END WHILE  
        END FOR  
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN the length of s  
            IF the element at position i of bold EQUALS true  
                APPEND the string consisting of the characters less than and b and greater than TO result  
                WHILE i LESS THAN the length of s AND the element at position i of bold EQUALS true  
                    APPEND the element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string consisting of the characters less than forward slash b and greater than TO result  
            ELSE  
                APPEND the element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
