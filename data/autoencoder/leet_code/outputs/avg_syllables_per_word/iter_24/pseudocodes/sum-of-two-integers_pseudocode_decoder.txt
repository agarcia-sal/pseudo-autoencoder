CLASS Solution
    FUNCTION getSum(a parameter, b parameter)
        SET maximum_integer TO the maximum value of a thirty two bits integer
        SET mask_for_thirty_two_bits TO a mask to get thirty two bits

        WHILE b parameter NOT EQUALS zero holds
            SET carry_value TO the bitwise AND of a parameter AND b parameter MULTIPLIED BY two
            SET a parameter TO the bitwise XOR of a parameter AND b parameter ANDed WITH mask_for_thirty_two_bits
            SET b parameter TO carry_value ANDed WITH mask_for_thirty_two_bits
        END WHILE

        IF a parameter LESS THAN OR EQUAL TO maximum_integer
            RETURN a parameter
        ELSE
            RETURN the bitwise NOT of the bitwise XOR of a parameter AND mask_for_thirty_two_bits
        END IF
    END FUNCTION
END CLASS
