CLASS Solution  
    FUNCTION reverseWords WITH PARAMETER s  
        CALL reverse_entire_list WITH PARAMETER s  
        SET start TO zero  
        FOR end FROM zero TO the LENGTH OF s MINUS one  
            IF element at position end of s EQUALS space character  
                CALL reverse_sublist WITH PARAMETERS s start end MINUS one  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        CALL reverse_sublist WITH PARAMETERS s start TO the last position in s  
    END FUNCTION  

    FUNCTION reverse_entire_list WITH PARAMETER s  
        SET left TO zero  
        SET right TO the LENGTH OF s MINUS one  
        WHILE left LESS THAN right  
            SET temp TO element at position left of s  
            SET element at position left of s TO element at position right of s  
            SET element at position right of s TO temp  
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
    END FUNCTION  

    FUNCTION reverse_sublist WITH PARAMETERS list start_index end_index  
        SET left TO start_index  
        SET right TO end_index  
        WHILE left LESS THAN right  
            SET temp TO element at position left of list  
            SET element at position left of list TO element at position right of list  
            SET element at position right of list TO temp  
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
    END FUNCTION  
END CLASS
