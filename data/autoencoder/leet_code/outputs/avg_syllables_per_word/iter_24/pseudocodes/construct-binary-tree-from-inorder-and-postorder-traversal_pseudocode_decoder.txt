CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero DO  
        REMOVE AND RETURN the element from the front of queue AND ASSIGN TO node  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node.left TO a new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS none THEN  
            SET node.right TO a new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none THEN  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none THEN  
        RETURN False  
    ELSE IF the value stored in p NOT EQUALS the value stored in q THEN  
        RETURN False  
    ELSE  
        RETURN the logical AND of the call to is_same_tree with the left child of p AND the left child of q AND the call to is_same_tree with the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, inorder, postorder)  
        IF the LENGTH OF inorder EQUALS zero OR the LENGTH OF postorder EQUALS zero THEN  
            RETURN none  
        END IF  
        SET root_val TO the element removed from the end of postorder  
        SET root TO a new TreeNode with root_val  
        SET root_index TO the position of root_val in inorder  
        SET root.right TO the result of calling buildTree with the sublist of inorder from the position immediately after root_index TO the end AND the current postorder  
        SET root.left TO the result of calling buildTree with the sublist of inorder from the start TO the position before root_index AND the current postorder  
        RETURN root  
    END FUNCTION  
END CLASS
