CLASS Solution  
    FUNCTION minCostII(costs)  
        IF the LENGTH OF costs EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF costs  
        SET k TO the LENGTH OF the element at position zero of costs  
        
        IF k EQUALS one THEN  
            IF n EQUALS one THEN  
                RETURN the element at position zero at position zero of costs  
            ELSE  
                RETURN a value representing positive infinity  
            END IF  
        END IF  
        
        FOR i FROM one TO n MINUS one  
            SET min_cost TO a value representing positive infinity  
            SET min_cost_index TO negative one  
            SET second_min_cost TO a value representing positive infinity  
            
            FOR j FROM zero TO k MINUS one  
                IF the element at position j of the element at position i MINUS one of costs LESS THAN min_cost THEN  
                    SET second_min_cost TO min_cost  
                    SET min_cost TO the element at position j of the element at position i MINUS one of costs  
                    SET min_cost_index TO j  
                ELSE IF the element at position j of the element at position i MINUS one of costs LESS THAN second_min_cost THEN  
                    SET second_min_cost TO the element at position j of the element at position i MINUS one of costs  
                END IF  
            END FOR  
            
            FOR j FROM zero TO k MINUS one  
                IF j EQUALS min_cost_index THEN  
                    SET the element at position j of the element at position i of costs TO the element at position j of the element at position i of costs PLUS second_min_cost  
                ELSE  
                    SET the element at position j of the element at position i of costs TO the element at position j of the element at position i of costs PLUS min_cost  
                END IF  
            END FOR  
        END FOR  
        
        SET minimum_final_cost TO the minimum value among the elements of the element at position n MINUS one of costs  
        RETURN minimum_final_cost  
    END FUNCTION  
END CLASS
