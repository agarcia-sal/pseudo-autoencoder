CLASS Solution  
    FUNCTION findItinerary(tickets)  
        SET graph TO a new mapping from keys to empty lists  
        FOR each pair consisting of departure and arrival IN tickets sorted in reverse order  
            APPEND arrival TO the list associated with departure in graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit(airport)  
            WHILE there EXISTS elements in the list associated with airport in graph  
                CALL visit WITH argument element at the last position of the list associated with airport in graph after REMOVING that element  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit WITH argument the string representing JFK airport  
        RETURN itinerary in the order from the last element to the first element  
    END FUNCTION  
END CLASS
