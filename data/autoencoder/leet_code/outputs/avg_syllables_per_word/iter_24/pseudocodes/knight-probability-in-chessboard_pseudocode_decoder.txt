CLASS Solution  
    FUNCTION knightProbability(n  
                               k  
                               row  
                               column)  
        SET moves TO list of all pairs of integers representing the knight moves as follows  
            first pair is negative two and negative one  
            second pair is negative two and one  
            third pair is negative one and negative two  
            fourth pair is negative one and two  
            fifth pair is one and negative two  
            sixth pair is one and two  
            seventh pair is two and negative one  
            eighth pair is two and one  
          
        SET dp TO two dimensional list of zeros with dimensions n by n  
        SET element at position row of dp and position column of that row TO one  
          
        FOR each iteration FROM one TO k  
            SET new_dp TO two dimensional list of zeros with dimensions n by n  
              
            FOR r FROM zero TO n MINUS one  
                FOR c FROM zero TO n MINUS one  
                    IF element at position r of dp and position c of that row NOT EQUALS zero  
                        FOR each pair dr dc IN moves  
                            SET nr TO r PLUS dr  
                            SET nc TO c PLUS dc  
                            IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n  
                                INCREMENT element at position nr of new_dp and position nc of that row BY element at position r of dp and position c of that row DIVIDED BY eight  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
              
            SET dp TO new_dp  
        END FOR  
          
        SET total_probability TO zero  
        FOR each row IN dp  
            FOR each value IN row  
                INCREMENT total_probability BY value  
            END FOR  
        END FOR  
          
        RETURN total_probability  
    END FUNCTION  
END CLASS
