CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS an empty string  
            RETURN s  
        END IF  
        
        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS the reverse of check_str  
        END FUNCTION  
        
        FOR i FROM the LENGTH OF s DOWN TO zero DECREMENT BY one  
            IF is_palindrome(substring from position zero TO position i MINUS one OF s) EQUALS true  
                SET to_add TO the reverse of substring from position i TO the end OF s  
                RETURN to_add PLUS s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
