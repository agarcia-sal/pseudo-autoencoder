CLASS Solution  
    FUNCTION maxSatisfaction(satisfaction)  
        SORT the elements of satisfaction in order from the largest value to the smallest value  
        SET max_satisfaction TO zero  
        SET current_satisfaction TO zero  
        SET total_satisfaction TO zero  
        FOR each element s IN satisfaction  
            INCREMENT current_satisfaction BY s  
            INCREMENT total_satisfaction BY current_satisfaction  
            IF total_satisfaction GREATER THAN max_satisfaction THEN  
                SET max_satisfaction TO total_satisfaction  
            END IF  
        END FOR  
        RETURN max_satisfaction  
    END FUNCTION  
END CLASS
