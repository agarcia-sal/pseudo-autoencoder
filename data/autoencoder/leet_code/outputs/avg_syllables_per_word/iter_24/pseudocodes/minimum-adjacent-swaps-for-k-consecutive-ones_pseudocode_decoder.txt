CLASS Solution  
    FUNCTION minMoves(nums, k)  
        SET positions TO an empty list  
        FOR each index i AND element num IN nums  
            IF num EQUALS one  
                APPEND i TO positions  
            END IF  
        END FOR  
        
        FUNCTION calculate_cost(start, end)  
            SET mid TO the integer division of start PLUS end BY two  
            SET median TO element at position mid of positions  
            SET cost TO zero  
            FOR each index i FROM start TO end MINUS one  
                SET difference TO absolute value of element at position i of positions MINUS the result of median MINUS the result of mid MINUS i  
                INCREMENT cost BY difference  
            END FOR  
            RETURN cost  
        END FUNCTION  
        
        SET min_cost TO infinity  
        FOR each index i FROM zero TO the length of positions MINUS k PLUS one MINUS one  
            SET current_cost TO calculate_cost(i, i PLUS k)  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
