CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr parameter of type List of Integer k parameter of type Integer) RETURNS List of Integer  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  
        FOR i FROM zero TO n MINUS one MINUS one  
            CALL helper_push_heap WITH min_heap AND the fraction consisting of element at position i of arr DIVIDED BY element at position n MINUS one of arr AND i AND n MINUS one  
        END FOR  
        FOR count FROM zero TO k MINUS one MINUS one  
            SET popped_fraction AND i AND j TO the result of helper_pop_heap WITH min_heap  
            IF j MINUS one GREATER THAN i THEN  
                CALL helper_push_heap WITH min_heap AND the fraction consisting of element at position i of arr DIVIDED BY element at position j MINUS one of arr AND i AND j MINUS one  
            END IF  
        END FOR  
        SET final_fraction AND i AND j TO the result of helper_pop_heap WITH min_heap  
        RETURN a list consisting of element at position i of arr FOLLOWED BY element at position j of arr  
    END FUNCTION  
  
    FUNCTION helper_push_heap(min_heap parameter of type list fraction parameter of type tuple)  
        INSERT the fraction INTO min_heap maintaining the heap property using the first element of fraction as the key  
    END FUNCTION  
  
    FUNCTION helper_pop_heap(min_heap parameter of type list) RETURNS tuple  
        REMOVE AND RETURN the smallest fraction from min_heap according to the first element of fraction as the key  
    END FUNCTION  
END CLASS
