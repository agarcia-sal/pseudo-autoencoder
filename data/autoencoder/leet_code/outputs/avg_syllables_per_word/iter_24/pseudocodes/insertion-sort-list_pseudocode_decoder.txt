CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET head TO new ListNode with value element at position one of values and next set to None  
    SET p TO head  
    FOR each val IN elements from position two TO end of values  
        SET node TO new ListNode with value val and next set to None  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the predicate that value at p1 EQUALS value at p2 AND is_same_list applied to next of p1 AND next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self, head)  
        IF head EQUALS None OR next of head EQUALS None  
            RETURN head  
        END IF  
        SET dummy TO new ListNode with value negative five thousand AND one and next set to None  
        SET dummy.next TO head  
        SET current TO next of head  
        SET next of head TO None  
        WHILE current NOT EQUALS None  
            SET next_node TO next of current  
            SET prev TO dummy  
            WHILE next of prev NOT EQUALS None AND value at next of prev LESS THAN value at current  
                SET prev TO next of prev  
            END WHILE  
            SET next of current TO next of prev  
            SET next of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN next of dummy  
    END FUNCTION  
END CLASS
