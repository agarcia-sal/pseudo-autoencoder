CLASS Solution  
    FUNCTION findAllPeople(n parameter, meetings parameter, firstPerson parameter)  
        SET know_secret TO a new set containing zero AND firstPerson parameter  
        SORT meetings parameter BY the element at position three of each meeting  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings parameter  
            SET current_time TO the element at position three of the meeting at position i of meetings parameter  
            SET graph TO a new default dictionary of list  
            WHILE i LESS THAN the LENGTH OF meetings parameter AND the element at position three of the meeting at position i of meetings parameter EQUALS current_time  
                SET x TO the element at position one of the meeting at position i of meetings parameter  
                SET y TO the element at position two of the meeting at position i of meetings parameter  
                APPEND y TO the list associated with key x in graph  
                APPEND x TO the list associated with key y in graph  
                INCREMENT i BY one  
            END WHILE  
            SET queue TO a new deque containing each person FROM know_secret WHERE that person IS IN graph  
            SET visited TO a new empty set  
            WHILE queue IS NOT empty  
                REMOVE and RETURN the leftmost element FROM queue AND ASSIGN TO person  
                IF person IS IN visited  
                    CONTINUE to the next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list associated with key person in graph  
                    IF neighbor IS NOT IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN a new list created FROM know_secret  
    END FUNCTION  
END CLASS
