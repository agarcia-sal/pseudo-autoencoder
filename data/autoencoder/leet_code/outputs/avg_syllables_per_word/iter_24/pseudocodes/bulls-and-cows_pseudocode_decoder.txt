CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        CREATE secret_count AS a dictionary with default integer zero  
        CREATE guess_count AS a dictionary with default integer zero  
        FOR each pair of elements s and g FROM corresponding elements of secret AND guess  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the element at key s IN secret_count BY one  
                INCREMENT the element at key g IN guess_count BY one  
            END IF  
        END FOR  
        FOR each digit IN the keys of secret_count  
            SET minimum_value TO the smaller value BETWEEN the element at key digit IN secret_count AND the element at key digit IN guess_count  
            INCREMENT cows BY minimum_value  
        END FOR  
        RETURN the concatenation of bulls as string WITH character A WITH cows as string WITH character B  
    END FUNCTION  
END CLASS
