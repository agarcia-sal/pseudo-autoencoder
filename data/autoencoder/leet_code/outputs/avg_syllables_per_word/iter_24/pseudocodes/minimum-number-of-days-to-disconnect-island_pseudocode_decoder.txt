CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the PAIR of x AND y IS IN visited OR element at position x of grid at position y EQUALS zero THEN  
                RETURN  
            END IF  
            ADD the PAIR of x AND y TO visited  
            FOR each PAIR OF dx AND dy IN the COLLECTION OF the PAIR of MINUS one AND zero THE PAIR of one AND zero THE PAIR of zero AND MINUS one THE PAIR of zero AND one  
                CALL dfs WITH PARAMETERS the SUM of x PLUS dx THE SUM of y PLUS dy AND visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an EMPTY SET  
            SET land_cells TO a COLLECTION OF the PAIR of i AND j FOR each i FROM zero TO LENGTH OF grid MINUS one FOR each j FROM zero TO LENGTH OF element at position zero of grid MINUS one WHERE element at position i of grid at position j EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero THEN  
                RETURN TRUE  
            END IF  
            CALL dfs WITH PARAMETERS the element at position zero of land_cells AT position zero the element at position zero of land_cells AT position one AND visited  
            RETURN the EXPRESSION that the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF NOT the RESULT OF is_connected() THEN  
            RETURN zero  
        END IF  

        FOR i FROM zero TO LENGTH OF grid MINUS one  
            FOR j FROM zero TO LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one THEN  
                    SET element at position i of grid at position j TO zero  
                    IF NOT the RESULT OF is_connected() THEN  
                        RETURN one  
                    END IF  
                    SET element at position i of grid at position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
