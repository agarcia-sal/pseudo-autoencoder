CLASS Solution  
    FUNCTION wiggleSort(nums)  
        CALL sort the elements of nums in ascending order  
        SET sorted_nums TO a copy of nums containing all elements  
        SET n TO the number of elements in nums  
        SET left TO the integer division of n minus one by two  
        SET right TO n minus one  
        FOR each index i FROM zero TO n minus one  
            IF the remainder of i divided by two EQUALS zero  
                SET the element at position i of nums TO the element at position left of sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET the element at position i of nums TO the element at position right of sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
