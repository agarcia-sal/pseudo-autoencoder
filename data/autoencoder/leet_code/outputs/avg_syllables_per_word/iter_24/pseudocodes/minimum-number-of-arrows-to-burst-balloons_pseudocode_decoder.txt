CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        SORT points BY the element at position two IN each balloon descriptor IN ascending order  
        SET arrows TO zero  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF the element at position one of balloon GREATER THAN last_arrow_position THEN  
                INCREMENT arrows BY one  
                SET last_arrow_position TO the element at position two of balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
