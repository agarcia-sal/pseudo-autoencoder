CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO list of pairs each consisting of the division of each element of wage by the corresponding element of quality AND the corresponding element of quality of all pairs created from corresponding elements of wage AND quality sorted in ascending order by the first element of each pair  
        SET max_heap TO empty list  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each pair ratio AND q IN workers  
            PUSH the negation of q INTO max_heap  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap is GREATER THAN k  
                SET total_quality TO total_quality PLUS the element removed from max_heap with the smallest value  
            END IF  
            IF the LENGTH OF max_heap EQUALS k  
                IF total_quality MULTIPLIED BY ratio is LESS THAN min_cost  
                    SET min_cost TO total_quality MULTIPLIED BY ratio  
                END IF  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
