CLASS Solution  
    FUNCTION minChanges(nums PARAMETER k)  
        SET groups TO a dictionary with default value as empty list  
        FOR i FROM zero TO the length of nums MINUS one  
            APPEND element at position i of nums TO the list at key of the remainder of i DIVIDED BY k in groups  
        END FOR  

        SET freq TO a list of dictionaries with default value zero for each index from zero TO k MINUS one  
        SET min_size TO a list of infinite values repeated k times  
        FOR i FROM zero TO k MINUS one  
            FOR each num IN the list at key i in groups  
                INCREMENT the value at key num in freq at index i BY one  
            END FOR  
            SET the element at position i of min_size TO the length of the list at key i in groups  
        END FOR  

        SET dp TO a list of infinite values repeated two raised to the power ten times  
        SET the element at position zero of dp TO zero  

        FOR i FROM zero TO k MINUS one  
            SET new_dp TO a list of infinite values repeated two raised to the power ten times  
            SET min_dp TO the minimum value in dp  
            FOR j FROM zero TO two raised to the power ten MINUS one  
                SET the element at position j of new_dp TO min_dp PLUS the element at position i of min_size  
                FOR each pair num and count IN the items of freq at index i  
                    IF the element at position j of new_dp GREATER THAN the sum of the element at position (j XOR num) of dp PLUS the element at position i of min_size MINUS count THEN  
                        SET the element at position j of new_dp TO the sum of the element at position (j XOR num) of dp PLUS the element at position i of min_size MINUS count  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  

        RETURN the element at position zero of dp  
    END FUNCTION  
END CLASS
