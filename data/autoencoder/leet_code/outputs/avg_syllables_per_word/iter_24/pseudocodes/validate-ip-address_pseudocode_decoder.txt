CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONTAINS ONLY digits  
                SET num TO the integer value of block  
                IF num LESS THAN OR EQUAL TO two hundred fifty five AND the string representation of num EQUALS block  
                    RETURN true  
                ELSE  
                    RETURN false  
                END IF  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF the LENGTH OF block GREATER THAN OR EQUAL TO one AND the LENGTH OF block LESS THAN OR EQUAL TO four  
                TRY  
                    CONVERT block TO an integer with base sixteen  
                    RETURN true  
                EXCEPT a value error occurs  
                    RETURN false  
                END TRY  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO the list obtained by splitting queryIP by the period character  
        SET parts_ipv6 TO the list obtained by splitting queryIP by the colon character  

        IF the LENGTH OF parts_ipv4 EQUALS four AND for each part IN parts_ipv4 the result of is_valid_ipv4_block applied to part IS true  
            RETURN the text IPv4  
        ELSE IF the LENGTH OF parts_ipv6 EQUALS eight AND for each part IN parts_ipv6 the result of is_valid_ipv6_block applied to part IS true  
            RETURN the text IPv6  
        ELSE  
            RETURN the text Neither  
        END IF  
    END FUNCTION  
END CLASS
