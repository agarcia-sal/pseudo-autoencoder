CLASS Solution  
    FUNCTION waysToPartition(nums AS list OF integers, k AS integer) RETURNS integer  
        SET total_sum TO the sum of all elements in nums  
        SET n TO the length of nums  
        
        CREATE left_diff AS a dictionary mapping integers to integers with default value zero  
        CREATE right_diff AS a dictionary mapping integers to integers with default value zero  
        
        SET current_sum TO zero  
        FOR index FROM zero TO n MINUS one MINUS one  
            INCREMENT current_sum BY the element at position index of nums  
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            INCREMENT the value at key diff in right_diff BY one  
        END FOR  
        
        SET max_ways TO the value at key zero in right_diff  
        
        SET current_sum TO zero  
        FOR index FROM zero TO n MINUS one  
            INCREMENT current_sum BY the element at position index of nums  
            
            SET diff_change TO k MINUS the element at position index of nums  
            
            SET new_ways TO the value at key negative diff_change in left_diff PLUS the value at key diff_change in right_diff  
            IF new_ways GREATER THAN max_ways  
                SET max_ways TO new_ways  
            END IF  
            
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum  
            DECREMENT the value at key diff in right_diff BY one  
            INCREMENT the value at key diff in left_diff BY one  
        END FOR  
        
        RETURN max_ways  
    END FUNCTION  
END CLASS
