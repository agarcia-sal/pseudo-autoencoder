CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with the element at position one of values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.left TO new TreeNode with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i of values NOT EQUALS None  
            SET node.right TO new TreeNode with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF the val attribute of p NOT EQUALS the val attribute of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left attribute of p, the left attribute of q) AND is_same_tree(the right attribute of p, the right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION printTree(self, root)  
        FUNCTION getHeight(node)  
            IF node EQUALS None  
                RETURN zero  
            END IF  
            RETURN one PLUS the maximum of getHeight(the left attribute of node) AND getHeight(the right attribute of node)  
        END FUNCTION  

        SET height TO getHeight(root)  
        SET width TO two RAISED TO height MINUS one  
        SET res TO a list containing height elements each of which is a list containing width empty strings  

        FUNCTION placeNode(node, row, left, right)  
            IF node EQUALS None  
                RETURN  
            END IF  
            SET mid TO the integer division of the sum of left AND right by two  
            SET the element at position mid of the list at position row of res TO the string form of the val attribute of node  
            CALL placeNode(the left attribute of node, row PLUS one, left, mid)  
            CALL placeNode(the right attribute of node, row PLUS one, mid, right)  
        END FUNCTION  

        CALL placeNode(root, zero, zero, width)  
        RETURN res  
    END FUNCTION  
END CLASS
