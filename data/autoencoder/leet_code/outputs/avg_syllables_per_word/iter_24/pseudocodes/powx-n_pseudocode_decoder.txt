CLASS Solution  
    FUNCTION myPow(x, n)  
        FUNCTION qpow(a, n)  
            SET ans TO one  
            WHILE n GREATER THAN zero  
                IF the remainder when n DIVIDED BY two EQUALS one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two AND THEN TAKE THE INTEGER PART ONLY  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n GREATER THAN OR EQUAL TO zero  
            RETURN qpow with a assigned to x and n assigned to n  
        ELSE  
            RETURN one DIVIDED BY qpow with a assigned to x and n assigned to the negative of n  
        END IF  
    END FUNCTION  
END CLASS
