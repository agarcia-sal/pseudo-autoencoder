CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM one TO two RAISED TO THE POWER OF n MINUS one  
            SET good TO an empty list of Boolean values WITH size n ALL SET TO False  
            SET bad TO an empty list of Boolean values WITH size n ALL SET TO False  
            FOR j FROM zero TO n MINUS one  
                IF the result of i DIVIDED BY two RAISED TO THE POWER OF j MODULO two EQUALS one  
                    SET element at position j of good TO True  
                ELSE  
                    SET element at position j of bad TO True  
                END IF  
            END FOR  
            SET valid TO True  
            FOR j FROM zero TO n MINUS one  
                IF element at position j of good EQUALS True  
                    FOR k FROM zero TO n MINUS one  
                        IF (element at position k of statements at position j EQUALS zero AND element at position k of good EQUALS True) OR (element at position k of statements at position j EQUALS one AND element at position k of bad EQUALS True)  
                            SET valid TO False  
                            BREAK  
                        END IF  
                    END FOR  
                    IF valid EQUALS False  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            IF valid EQUALS True  
                SET sum_good TO zero  
                FOR each value IN good  
                    IF value EQUALS True  
                        INCREMENT sum_good BY one  
                    END IF  
                END FOR  
                IF sum_good GREATER THAN max_good  
                    SET max_good TO sum_good  
                END IF  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
