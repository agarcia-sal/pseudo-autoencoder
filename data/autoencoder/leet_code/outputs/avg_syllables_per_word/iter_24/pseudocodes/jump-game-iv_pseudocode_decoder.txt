CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO an empty mapping from values to lists  
        FOR each position index and element value IN arr WITH THEIR positions  
            APPEND position index TO the list corresponding to element value IN value_indices  
        END FOR  
        
        SET queue TO a new double ended queue initialized with a tuple consisting of zero and zero representing current position and current steps  
        SET visited TO a new set containing zero  
        
        WHILE queue is not empty  
            REMOVE the first tuple from queue and DECOMPOSE it INTO current_index and steps  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO a new list CONSISTING OF current_index PLUS one, current_index MINUS one, and all elements in value_indices at the element value found at position current_index in arr  
            
            FOR each next_index IN next_indices  
                IF next_index GREATER THAN OR EQUAL TO zero AND next_index LESS THAN n AND next_index NOT IN visited  
                    ADD next_index TO visited  
                    APPEND a tuple consisting of next_index and steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list corresponding to the element value at position current_index in arr IN value_indices TO an empty list  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
