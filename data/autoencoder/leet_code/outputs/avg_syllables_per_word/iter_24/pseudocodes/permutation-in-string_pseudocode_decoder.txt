CLASS Solution  
    FUNCTION checkInclusion(s1 AS string, s2 AS string) RETURNS boolean  
        IMPORT the Counter data structure from the collections module  
        
        SET length of s1 TO the LENGTH OF s1  
        SET length of s2 TO the LENGTH OF s2  
        
        IF length of s1 IS GREATER THAN length of s2  
            RETURN boolean value false  
        END IF  
        
        SET count of characters in s1 TO a Counter based on s1  
        SET count of characters in the first section of s2 from position one TO position length of s1 TO a Counter based on that section of s2  
        
        IF count of characters in s1 IS EQUALS count of characters in first section of s2  
            RETURN boolean value true  
        END IF  
        
        FOR index i FROM length of s1 PLUS one TO length of s2  
            INCREMENT the count of character at position i of s2 in count of characters in s2 BY one  
            DECREMENT the count of character at position index i MINUS length of s1 of s2 in count of characters in s2 BY one  
            
            IF the count of character at position index i MINUS length of s1 of s2 in count of characters in s2 IS EQUALS zero  
                REMOVE the character at position index i MINUS length of s1 of s2 FROM count of characters in s2  
            END IF  
            
            IF count of characters in s1 IS EQUALS count of characters in s2  
                RETURN boolean value true  
            END IF  
        END FOR  
        
        RETURN boolean value false  
    END FUNCTION  
END CLASS
