CLASS Solution  
    FUNCTION maxNumber(nums1 nums2 k)  
        FUNCTION maxSingleArray(nums k)  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO an empty list  
            FOR each num IN nums  
                WHILE drop GREATER THAN zero AND stack NOT empty AND the last element at position the LENGTH OF stack MINUS one of stack LESS THAN num  
                    REMOVE the last element at position the LENGTH OF stack MINUS one FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the first k elements of stack  
        END FUNCTION  
        
        FUNCTION merge(arr1 arr2)  
            SET merged_list TO an empty list  
            SET total_length TO the LENGTH OF arr1 PLUS the LENGTH OF arr2  
            FOR index FROM zero TO total_length MINUS one  
                IF the list arr1 GREATER THAN the list arr2  
                    REMOVE the first element at position zero FROM arr1 AND APPEND it TO merged_list  
                ELSE  
                    REMOVE the first element at position zero FROM arr2 AND APPEND it TO merged_list  
                END IF  
            END FOR  
            RETURN merged_list  
        END FUNCTION  
        
        SET max_number TO an empty list  
        FOR i FROM zero TO k  
            IF i LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i LESS THAN OR EQUAL TO the LENGTH OF nums2  
                SET candidate TO the result of merge WITH PARAMETERS maxSingleArray WITH PARAMETERS nums1 AND i AND maxSingleArray WITH PARAMETERS nums2 AND k MINUS i  
                IF max_number LESS THAN candidate  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
