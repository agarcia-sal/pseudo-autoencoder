CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO an empty mapping with each key being a course from zero TO numCourses MINUS one and each value being an empty list  
        SET in_degrees TO a list of zeros of length numCourses  
        
        FOR each pair of destination_course and source_course IN prerequisites  
            APPEND destination_course TO the list at key source_course in adj_list  
            INCREMENT the element at position destination_course in in_degrees BY one  
        END FOR  
        
        SET queue TO a double ended queue containing each course FROM zero TO numCourses MINUS one WHERE the element at that course position in in_degrees EQUALS zero  
        SET topological_order TO an empty list  
        
        WHILE queue IS NOT empty  
            SET current TO the element removed from the left end of queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN the list at key current in adj_list  
                DECREMENT the element at position neighbor in in_degrees BY one  
                IF the element at position neighbor in in_degrees EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF the LENGTH OF topological_order EQUALS numCourses  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
        END IF  
    END FUNCTION  
END CLASS
