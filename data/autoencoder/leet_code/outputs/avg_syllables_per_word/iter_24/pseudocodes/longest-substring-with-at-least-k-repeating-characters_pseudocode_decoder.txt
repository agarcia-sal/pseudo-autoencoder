CLASS Solution  
    FUNCTION longestSubstring WITH PARAMETERS s k  
        IF the LENGTH OF s LESS THAN k  
            RETURN zero  
        END IF  
        SET count TO the FREQUENCY COUNTS OF each CHARACTER IN s  
        FOR each char IN the KEYS OF count  
            IF the FREQUENCY OF char IN count LESS THAN k  
                RETURN the MAXIMUM OF the RESULTS OF longestSubstring WITH PARAMETERS each substring AND k FOR each substring IN the SEQUENCE OBTAINED BY DIVIDING s BY char AS SEPARATOR  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
