CLASS ListNode  
    FUNCTION __init__ with parameters self val EQUALS zero next EQUALS None  
        SET the value attribute of self TO val  
        SET the next attribute of self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameters values  
    IF the values list IS equal to empty  
        RETURN None  
    END IF  
    SET head TO a new ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN elements from position one TO the end of values  
        SET node TO a new ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the expression that the value attribute of p1 IS EQUALS the value attribute of p2 AND the recursive call to is_same_list with the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION partition with parameters self head x  
        SET less_head TO a new ListNode initialized with zero  
        SET greater_head TO a new ListNode initialized with zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current IS NOT None  
            IF the value attribute of current IS LESS THAN x  
                SET the next attribute of less TO current  
                SET less TO the next attribute of less  
            ELSE  
                SET the next attribute of greater TO current  
                SET greater TO the next attribute of greater  
            END IF  
            SET current TO the next attribute of current  
        END WHILE  
        SET the next attribute of less TO the next attribute of greater_head  
        SET the next attribute of greater TO None  
        RETURN the next attribute of less_head  
    END FUNCTION  
END CLASS
