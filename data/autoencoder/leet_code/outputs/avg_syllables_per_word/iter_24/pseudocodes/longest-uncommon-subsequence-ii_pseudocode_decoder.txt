CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator TO create an iterator over t  
            RETURN true if for every character in s the character appears in sequence in iterator otherwise false  
        END FUNCTION  
        
        SORT strs by descending order of length and then by alphabetical order  
        
        FOR each index i and word IN the enumeration of strs  
            SET unique TO true  
            FOR each index j and other_word IN the enumeration of strs  
                IF i NOT EQUALS j AND is_subsequence(word, other_word) IS true THEN  
                    SET unique TO false  
                    BREAK the inner loop  
                END IF  
            END FOR  
            IF unique IS true THEN  
                RETURN the length of word  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
