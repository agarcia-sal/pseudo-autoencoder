CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND its count v IN the items of cnt  
                IF v GREATER THAN one  
                    DECREMENT cnt AT character c BY two  
                    CALL dfs WITH the CONCATENATION of character c PLUS t PLUS character c  
                    INCREMENT cnt AT character c BY two  
                END IF  
            END FOR  
        END FUNCTION  
          
        SET cnt TO the frequency count of characters in s  
        SET mid TO an empty string  
        FOR each character c AND its count v IN the items of cnt  
            IF v MODULO two NOT EQUALS zero  
                IF mid IS NOT an empty string  
                    RETURN an empty list  
                END IF  
                SET mid TO character c  
                DECREMENT cnt AT character c BY one  
            END IF  
        END FOR  
        SET ans TO an empty list  
        CALL dfs WITH mid  
        RETURN ans  
    END FUNCTION  
END CLASS
