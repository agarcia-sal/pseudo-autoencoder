CLASS UnionFind  
    FUNCTION __init__ (self, size)  
        SET self.parent TO a list of integers from zero TO size MINUS one  
        SET self.rank TO a list of zeros with length size  
        SET self.size TO a list of ones with length size  
    END FUNCTION  

    FUNCTION find (self, u)  
        IF element at position u of self.parent NOT EQUALS u THEN  
            SET element at position u of self.parent TO the result of calling self.find with the element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  

    FUNCTION union (self, u, v)  
        SET rootU TO the result of calling self.find with u  
        SET rootV TO the result of calling self.find with v  
        IF rootU NOT EQUALS rootV THEN  
            IF element at position rootU of self.rank GREATER THAN element at position rootV of self.rank THEN  
                SET element at position rootV of self.parent TO rootU  
                SET element at position rootU of self.size TO element at position rootU of self.size PLUS element at position rootV of self.size  
            ELSE IF element at position rootU of self.rank LESS THAN element at position rootV of self.rank THEN  
                SET element at position rootU of self.parent TO rootV  
                SET element at position rootV of self.size TO element at position rootV of self.size PLUS element at position rootU of self.size  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                SET element at position rootU of self.size TO element at position rootU of self.size PLUS element at position rootV of self.size  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION groupStrings (self, words)  
        FUNCTION bitmask (word)  
            SET mask TO zero  
            FOR each character IN word  
                SET mask TO mask OR one SHIFTED LEFT BY the difference of the ordinal value of character MINUS ordinal value of letter a  
            END FOR  
            RETURN mask  
        END FUNCTION  

        SET word_map TO a default dictionary mapping to empty lists  
        SET n TO the LENGTH OF words  
        SET uf TO a new UnionFind object initialized with n  

        FOR each index i and word IN enumeration of words  
            SET mask TO the result of calling bitmask with word  
            APPEND i TO the list in word_map at key mask  
        END FOR  

        FOR each index i and word IN enumeration of words  
            SET mask TO the result of calling bitmask with word  
            FOR integer j FROM zero TO twenty five  
                SET new_mask TO the bitwise exclusive OR of mask WITH one SHIFTED LEFT BY j  
                IF new_mask EXISTS as a key in word_map THEN  
                    FOR each element k IN the list at word_map[new_mask]  
                        CALL uf.union with parameters i and k  
                    END FOR  
                END IF  

                IF mask BITWISE AND one SHIFTED LEFT BY j EQUALS zero THEN  
                    SET new_mask TO the bitwise OR of mask WITH one SHIFTED LEFT BY j  
                    IF new_mask EXISTS as a key in word_map THEN  
                        FOR each element k IN the list at word_map[new_mask]  
                            CALL uf.union with parameters i and k  
                        END FOR  
                    END IF  
                END IF  

                IF mask BITWISE AND one SHIFTED LEFT BY j NOT EQUALS zero THEN  
                    FOR integer k FROM zero TO twenty five  
                        IF mask BITWISE AND one SHIFTED LEFT BY k EQUALS zero THEN  
                            SET new_mask TO the bitwise OR of the bitwise exclusive OR of mask WITH one SHIFTED LEFT BY j WITH one SHIFTED LEFT BY k  
                            IF new_mask EXISTS as a key in word_map THEN  
                                FOR each element l IN the list at word_map[new_mask]  
                                    CALL uf.union with parameters i and l  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        SET root_sizes TO a default dictionary mapping to zero  
        FOR integer i FROM zero TO n MINUS one  
            SET root TO the result of calling uf.find with i  
            INCREMENT element at key root of root_sizes BY one  
        END FOR  

        SET max_group_size TO the maximum value among the values in root_sizes  
        SET num_groups TO the LENGTH OF root_sizes  

        RETURN a list containing num_groups AND max_group_size  
    END FUNCTION  
END CLASS
