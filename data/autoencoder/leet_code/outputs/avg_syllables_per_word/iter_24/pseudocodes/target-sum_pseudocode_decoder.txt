CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the TUPLE of index AND current_sum EXISTS IN memo  
                RETURN the VALUE ASSOCIATED WITH the TUPLE of index AND current_sum IN memo  
            END IF  
            SET positive TO dfs with arguments index PLUS one, current_sum PLUS element at position index of nums, memo  
            SET negative TO dfs with arguments index PLUS one, current_sum MINUS element at position index of nums, memo  
            SET memo at the TUPLE of index AND current_sum TO positive PLUS negative  
            RETURN memo at the TUPLE of index AND current_sum  
        END FUNCTION  
        RETURN dfs with arguments zero, zero, empty MAPPING  
    END FUNCTION  
END CLASS
