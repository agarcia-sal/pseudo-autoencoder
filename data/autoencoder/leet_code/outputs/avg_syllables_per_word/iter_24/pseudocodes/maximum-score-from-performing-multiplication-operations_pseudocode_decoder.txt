CLASS Solution  
    FUNCTION maximumScore(nums, multipliers)  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of zeros of size m PLUS one  
        FOR i FROM m MINUS one DOWN TO zero  
            SET new_dp TO a list of zeros of size m PLUS one  
            FOR left FROM zero TO i INCLUSIVE  
                SET right TO n MINUS one MINUS i PLUS left  
                SET mul TO element at position i of multipliers  
                SET value_from_left TO mul MULTIPLIED BY element at position left of nums PLUS element at position left PLUS one of dp  
                SET value_from_right TO mul MULTIPLIED BY element at position right of nums PLUS element at position left of dp  
                IF value_from_left GREATER THAN value_from_right  
                    SET element at position left of new_dp TO value_from_left  
                ELSE  
                    SET element at position left of new_dp TO value_from_right  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN element at position zero of dp  
    END FUNCTION  
END CLASS
