CLASS Solution  
    FUNCTION generateParenthesis(n)  
        DEFINE FUNCTION backtrack(current_string SET TO empty string, left_count SET TO zero, right_count SET TO zero)  
            IF the LENGTH OF current_string EQUALS two MULTIPLIED BY n THEN  
                APPEND current_string TO result  
                RETURN  
            END IF  
            IF left_count LESS THAN n THEN  
                CALL backtrack WITH current_string PLUS the character representing opening parenthesis, left_count INCREMENTED BY one, right_count  
            END IF  
            IF right_count LESS THAN left_count THEN  
                CALL backtrack WITH current_string PLUS the character representing closing parenthesis, left_count, right_count INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO empty list  
        CALL backtrack WITH default parameters  
        RETURN result  
    END FUNCTION  
END CLASS
