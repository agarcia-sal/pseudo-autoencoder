CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO set of digits two five six nine  
        SET same_digits TO set of digits zero one eight  
        
        FUNCTION is_good_number(num)  
            SET num_str TO string representation of num  
            IF all digits of num_str are elements of the union of change_digits and same_digits THEN  
                IF any digit of num_str is an element of change_digits THEN  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR i FROM one TO n  
            IF is_good_number(i) EQUALS true THEN  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
