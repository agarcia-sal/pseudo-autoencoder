CLASS Solution  
    FUNCTION compress(chars)  
        IF the collection chars EQUALS nothing THEN  
            RETURN zero  
        END IF  
        SET write_index TO zero  
        SET read_index TO zero  
        WHILE the value of read_index IS LESS THAN the length of chars DO  
            SET char TO the element at position read_index of chars  
            SET count TO zero  
            WHILE the value of read_index IS LESS THAN the length of chars AND the element at position read_index of chars EQUALS char DO  
                INCREMENT read_index BY one  
                INCREMENT count BY one  
            END WHILE  
            SET the element at position write_index of chars TO char  
            INCREMENT write_index BY one  
            IF count IS GREATER THAN one THEN  
                FOR each digit IN the sequence formed by the text representation of count DO  
                    SET the element at position write_index of chars TO digit  
                    INCREMENT write_index BY one  
                END FOR  
            END IF  
        END WHILE  
        RETURN write_index  
    END FUNCTION  
END CLASS
