CLASS Solution  
    FUNCTION isSolvable(words, result)  
        SET unique_chars TO the set of all characters contained in all elements of words plus all characters contained in result  
        IF the number of elements in unique_chars GREATER THAN ten  
            RETURN False  
        END IF  

        SET non_zero_chars TO the set of first character of each word in words where the length of word GREATER THAN one combined with the first character of result if the length of result GREATER THAN one otherwise the empty string  

        SET char_to_digit TO empty dictionary  
        SET digit_to_char TO empty dictionary  

        FUNCTION can_assign(char, digit)  
            RETURN char NOT IN char_to_digit AND (digit NOT IN digit_to_char OR the element at key digit of digit_to_char EQUALS char)  
        END FUNCTION  

        FUNCTION assign(char, digit)  
            SET the element at key char of char_to_digit TO digit  
            SET the element at key digit of digit_to_char TO char  
        END FUNCTION  

        FUNCTION unassign(char, digit)  
            REMOVE the element at key char of char_to_digit  
            REMOVE the element at key digit of digit_to_char  
        END FUNCTION  

        FUNCTION word_value(word)  
            SET total_value TO zero  
            FOR each pair of index i and character char in the enumeration of the characters of word in reverse order  
                INCREMENT total_value BY the element at key char of char_to_digit MULTIPLIED BY ten RAISED TO THE power i  
            END FOR  
            RETURN total_value  
        END FUNCTION  

        FUNCTION backtrack(index, col, sum_col)  
            SET max_word_length TO the maximum length among all elements in words  
            IF col GREATER THAN OR EQUAL TO max_word_length AND col GREATER THAN OR EQUAL TO the length of result  
                RETURN sum_col EQUALS zero  
            END IF  

            IF index EQUALS the length of words  
                IF col LESS THAN the length of result  
                    SET digit_char TO the character at position the length of result MINUS one MINUS col of result  
                    IF digit_char IN char_to_digit  
                        IF the element at key digit_char of char_to_digit EQUALS sum_col MODULO ten  
                            RETURN backtrack(zero, col PLUS one, sum_col DIVIDED BY ten ignoring remainder)  
                        ELSE  
                            RETURN False  
                        END IF  
                    ELSE IF can_assign(digit_char, sum_col MODULO ten) AND (sum_col MODULO ten NOT EQUALS zero OR digit_char NOT IN non_zero_chars)  
                        assign(digit_char, sum_col MODULO ten)  
                        IF backtrack(zero, col PLUS one, sum_col DIVIDED BY ten ignoring remainder)  
                            RETURN True  
                        END IF  
                        unassign(digit_char, sum_col MODULO ten)  
                    END IF  
                ELSE  
                    RETURN sum_col EQUALS zero  
                END IF  
                RETURN False  
            END IF  

            SET word TO the element at position index of words  
            IF col GREATER THAN OR EQUAL TO the length of word  
                RETURN backtrack(index PLUS one, col, sum_col)  
            END IF  

            SET char TO the character at position the length of word MINUS one MINUS col of word  
            IF char IN char_to_digit  
                RETURN backtrack(index PLUS one, col, sum_col PLUS the element at key char of char_to_digit)  
            END IF  

            FOR digit FROM zero TO nine  
                IF can_assign(char, digit) AND (digit NOT EQUALS zero OR char NOT IN non_zero_chars)  
                    assign(char, digit)  
                    IF backtrack(index PLUS one, col, sum_col PLUS digit)  
                        RETURN True  
                    END IF  
                    unassign(char, digit)  
                END IF  
            END FOR  

            RETURN False  
        END FUNCTION  

        RETURN backtrack(zero, zero, zero)  
    END FUNCTION  
END CLASS
