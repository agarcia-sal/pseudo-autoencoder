CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        SET graph TO a new default dictionary with list as default value  
        SET indegree TO a list of zeros with length numCourses  
        FOR each pair of course and prerequisite IN prerequisites  
            APPEND course TO the list at key prerequisite in graph  
            INCREMENT the value at position course in indegree BY one  
        END FOR  
        SET queue TO a new deque with all courses from zero TO numCourses MINUS one WHERE the value at position course in indegree EQUALS zero  
        WHILE the queue is not empty  
            SET current_course TO the element removed from the front of queue  
            DECREMENT numCourses BY one  
            FOR each next_course IN the list at key current_course in graph  
                DECREMENT the value at position next_course in indegree BY one  
                IF the value at position next_course in indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN the result of numCourses EQUALS zero  
    END FUNCTION  
END CLASS
