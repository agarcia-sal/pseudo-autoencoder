CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        SET list_of_tasks TO list of pairs each composed of element at position i of plantTime and element at position i of growTime FOR i FROM zero TO LENGTH OF plantTime MINUS one  
        SET sorted_tasks TO list_of_tasks SORTED in the order of the second element of each pair from largest to smallest  
        
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        
        FOR each pair composed of plant_time and grow_time IN sorted_tasks  
            INCREMENT current_day BY plant_time  
            SET bloom_day TO current_day PLUS grow_time  
            IF bloom_day GREATER THAN earliest_bloom  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
