CLASS Solution  
    FUNCTION reorderedPowerOf2(n)  
        SET power_of_2_sorted_digits TO an empty set  
        SET power_of_2 TO one  
        WHILE power_of_2 LESS THAN OR EQUAL TO one billion  
            CONVERT power_of_2 TO string  
            SORT the characters of the string representation of power_of_2 in ascending order  
            CONVERT the sorted characters to a tuple  
            ADD the tuple to power_of_2_sorted_digits  
            MULTIPLY power_of_2 BY two  
        END WHILE  
        CONVERT n TO string  
        SORT the characters of the string representation of n in ascending order  
        CONVERT the sorted characters to a tuple and ASSIGN to sorted_n_digits  
        IF sorted_n_digits EXISTS IN power_of_2_sorted_digits  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
