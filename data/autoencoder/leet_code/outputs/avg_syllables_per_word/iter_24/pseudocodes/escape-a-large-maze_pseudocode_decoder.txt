CLASS Solution  
    FUNCTION isEscapePossible(blocked, source, target)  
        IF blocked EQUALS empty  
            RETURN True  
        END IF  
        
        SET blocked_set TO empty set  
        FOR each block IN blocked  
            ADD tuple converted block TO blocked_set  
        END FOR  
        
        SET n TO the LENGTH OF blocked  
        SET max_steps TO n MULTIPLIED BY the result of n MINUS one DIVIDED BY two  
        
        FUNCTION bfs(start, end)  
            SET queue TO a double ended queue initialized with tuple converted start  
            SET visited TO a set initialized with queue elements  
            
            WHILE queue is not empty  
                SET x, y TO the elements removed from the left of queue  
                
                IF the LENGTH OF visited GREATER THAN max_steps OR x EQUALS the first element of end AND y EQUALS the second element of end  
                    RETURN True  
                END IF  
                
                FOR each pair dx, dy IN the list of pairs representing zero one zero negative one one zero negative one zero  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    
                    IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN one million AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN one million AND the tuple composed of nx and ny NOT IN blocked_set AND the tuple composed of nx and ny NOT IN visited  
                        ADD the tuple composed of nx and ny TO visited  
                        APPEND the tuple composed of nx and ny TO queue  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN False  
        END FUNCTION  
        
        RETURN the result of bfs applied to source and target AND the result of bfs applied to target and source  
    END FUNCTION  
END CLASS
