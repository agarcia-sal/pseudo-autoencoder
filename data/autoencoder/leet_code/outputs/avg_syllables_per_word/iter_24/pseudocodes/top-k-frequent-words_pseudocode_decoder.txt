CLASS Solution  
    FUNCTION topKFrequent(words k)  
        SET count TO a collection mapping each unique element in words TO the number of times that element appears in words  
        SET heap TO an empty list  
        FOR each word frequency PAIR IN count  
            APPEND a pair consisting of the negative of frequency and word TO heap  
        END FOR  
        TRANSFORM heap INTO a heap structure that maintains the heap property  
        SET result TO an empty list  
        FOR each number FROM one TO k  
            REMOVE the element WITH the smallest first component FROM heap AND ASSIGN IT TO removed_element  
            APPEND the second component of removed_element TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
