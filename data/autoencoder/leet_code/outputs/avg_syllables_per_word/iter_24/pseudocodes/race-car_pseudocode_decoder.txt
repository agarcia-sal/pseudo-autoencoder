CLASS Solution  
    FUNCTION racecar(target)  
        ASSIGN reference TO a new double ended queue containing one tuple of zero position one speed zero steps  
        ASSIGN reference TO a new set containing one tuple of zero position one speed  
        WHILE the queue is not empty  
            REMOVE from the left side of the queue and unpack into position speed steps  
            IF position EQUALS target  
                RETURN steps  
            END IF  
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple of next_position and next_speed is not an element of visited AND next_position GREATER THAN OR EQUAL TO zero AND next_position LESS THAN two MULTIPLIED BY target  
                ADD the tuple of next_position and next_speed TO visited  
                APPEND the tuple of next_position next_speed steps PLUS one TO the right side of the queue  
            END IF  
            IF speed GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple of position and next_speed is not an element of visited AND position GREATER THAN OR EQUAL TO zero AND position LESS THAN two MULTIPLIED BY target  
                ADD the tuple of position and next_speed TO visited  
                APPEND the tuple of position next_speed steps PLUS one TO the right side of the queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
