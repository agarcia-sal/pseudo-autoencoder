CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o a b)  
            RETURN the value of the difference between the product of the difference of the element at position zero of a MINUS the element at position zero of o and the difference of the element at position one of b MINUS the element at position one of o AND the product of the difference of the element at position one of a MINUS the element at position one of o AND the difference of the element at position zero of b MINUS the element at position zero of o  
        END FUNCTION  

        SET n TO the LENGTH OF points  
        IF n LESS THAN OR EQUAL TO three  
            RETURN True  
        END IF  

        SET orientation TO zero  

        FOR each i FROM zero TO n MINUS one  
            SET o TO the element at position i of points  
            SET a TO the element at position the remainder of i PLUS one DIVIDED BY n of points  
            SET b TO the element at position the remainder of i PLUS two DIVIDED BY n of points  
            SET cross TO the RESULT OF cross_product WITH PARAMETERS o a b  

            IF cross NOT EQUALS zero  
                IF orientation EQUALS zero  
                    IF cross GREATER THAN zero  
                        SET orientation TO one  
                    ELSE  
                        SET orientation TO negative one  
                    END IF  
                ELSE IF orientation EQUALS one AND cross LESS THAN zero OR orientation EQUALS negative one AND cross GREATER THAN zero  
                    RETURN False  
                END IF  
            END IF  
        END FOR  

        RETURN True  
    END FUNCTION  
END CLASS
