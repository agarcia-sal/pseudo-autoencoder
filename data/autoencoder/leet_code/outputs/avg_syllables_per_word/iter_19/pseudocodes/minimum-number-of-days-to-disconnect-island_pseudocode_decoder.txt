CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the tuple of x and y IS IN visited OR element at position x of grid at position y EQUALS zero  
                RETURN  
            END IF  
            ADD the tuple of x and y TO visited  
            FOR each pair dx and dy IN the list of pairs of minus one and zero PLUS one and zero PLUS zero and minus one PLUS zero and one  
                CALL dfs WITH parameters the sum of x AND dx AND the sum of y AND dy AND visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an empty set  
            SET land_cells TO the list of tuples i and j FOR each i FROM zero TO the LENGTH OF grid MINUS one AND for each j FROM zero TO the LENGTH OF element at position zero of grid MINUS one WHERE element at position i of grid at position j EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero  
                RETURN true  
            END IF  
            CALL dfs WITH parameters the element at position zero of tuple at position zero of land_cells AND the element at position one of tuple at position zero of land_cells AND visited  
            RETURN the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF is_connected() EQUALS false  
            RETURN zero  
        END IF  
        FOR i FROM zero TO the LENGTH OF grid MINUS one  
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one  
                IF element at position i of grid at position j EQUALS one  
                    SET element at position i of grid at position j TO zero  
                    IF is_connected() EQUALS false  
                        RETURN one  
                    END IF  
                    SET element at position i of grid at position j TO one  
                END IF  
            END FOR  
        END FOR  
        RETURN two  
    END FUNCTION  
END CLASS
