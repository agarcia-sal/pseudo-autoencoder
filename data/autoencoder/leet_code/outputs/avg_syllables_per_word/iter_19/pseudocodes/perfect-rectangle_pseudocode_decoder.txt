CLASS Solution  
    FUNCTION isRectangleCover(rectangles)  
        SET X1 TO positive infinity  
        SET Y1 TO positive infinity  
        SET X2 TO negative infinity  
        SET Y2 TO negative infinity  
        
        SET points TO an empty set  
        SET area TO zero  
        
        FOR each element consisting of x1 y1 x2 y2 IN rectangles  
            IF X1 IS GREATER THAN x1 THEN  
                SET X1 TO x1  
            END IF  
            IF Y1 IS GREATER THAN y1 THEN  
                SET Y1 TO y1  
            END IF  
            IF X2 IS LESS THAN x2 THEN  
                SET X2 TO x2  
            END IF  
            IF Y2 IS LESS THAN y2 THEN  
                SET Y2 TO y2  
            END IF  
            
            INCREMENT area BY the result of x2 MINUS x1 MULTIPLIED BY y2 MINUS y1  
            
            FOR each point p IN the list containing the element at position zero x1 AND the element at position one y1  
            THEN the element at position zero x1 AND the element at position one y2  
            THEN the element at position zero x2 AND the element at position one y1  
            THEN the element at position zero x2 AND the element at position one y2  
                IF p IS IN points THEN  
                    REMOVE p FROM points  
                ELSE  
                    ADD p TO points  
                END IF  
            END FOR  
        END FOR  
        
        IF area IS NOT EQUAL TO the result of X2 MINUS X1 MULTIPLIED BY Y2 MINUS Y1 THEN  
            RETURN False  
        END IF  
        
        IF points IS NOT EQUAL TO a set containing the tuple of X1 AND Y1 AND the tuple of X1 AND Y2 AND the tuple of X2 AND Y1 AND the tuple of X2 AND Y2 THEN  
            RETURN False  
        END IF  
        
        RETURN True  
    END FUNCTION  
END CLASS
