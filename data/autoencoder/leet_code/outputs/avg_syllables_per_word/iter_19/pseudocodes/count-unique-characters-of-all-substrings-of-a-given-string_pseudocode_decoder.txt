CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MODULO TO ten TO THE POWER OF nine PLUS seven  
        SET string_length TO the LENGTH OF s  
        SET last_positions TO a list of negative one repeated twenty six times  
        SET second_last_positions TO a list of negative one repeated twenty six times  
        SET total_result TO zero  
        FOR each index AND character IN s WITH enumeration starting from zero  
            SET character_index TO the Unicode code point OF character MINUS the Unicode code point OF capital letter A  
            SET contribution TO the PRODUCT OF the DIFFERENCE BETWEEN index AND the element at position character_index of last_positions AND the DIFFERENCE BETWEEN the element at position character_index of last_positions AND the element at position character_index of second_last_positions  
            SET total_result TO the SUM OF total_result AND contribution MODULO MODULO  
            SET the element at position character_index of second_last_positions TO the element at position character_index of last_positions  
            SET the element at position character_index of last_positions TO index  
        END FOR  
        FOR each index FROM zero TO twenty five  
            SET contribution TO the PRODUCT OF the DIFFERENCE BETWEEN string_length AND the element at position index of last_positions AND the DIFFERENCE BETWEEN the element at position index of last_positions AND the element at position index of second_last_positions  
            SET total_result TO the SUM OF total_result AND contribution MODULO MODULO  
        END FOR  
        RETURN total_result  
    END FUNCTION  
END CLASS
