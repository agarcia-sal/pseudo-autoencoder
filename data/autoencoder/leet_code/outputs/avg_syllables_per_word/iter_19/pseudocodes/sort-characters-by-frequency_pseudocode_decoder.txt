CLASS Solution  
    FUNCTION frequencySort WITH parameter s  
        SET frequency TO CALL Counter WITH argument s  
        SET sorted_characters TO RESULT OF SORTING THE COLLECTION OF KEYS OF frequency BY ORDERING EACH CHARACTER BY THE VALUE ASSOCIATED TO THAT CHARACTER IN frequency IN DESCENDING ORDER  
        SET result TO AN EMPTY STRING  
        FOR each character IN sorted_characters  
            APPEND THE character REPEATED THE NUMBER OF TIMES EQUAL TO THE VALUE ASSOCIATED TO character IN frequency TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
