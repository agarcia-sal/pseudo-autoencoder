CLASS Solution  
    FUNCTION maxA WITH PARAMETER n  
        IF n LESS THAN OR EQUAL TO three  
            RETURN n  
        END IF  
        
        SET dp TO a list of zeros with size n PLUS one  
        
        FOR index FROM one TO three  
            SET element at position index of dp TO index  
        END FOR  
        
        FOR i FROM four TO n  
            FOR j FROM i MINUS three DOWN TO one  
                SET possible_value TO element at position j of dp MULTIPLIED BY i MINUS j MINUS one  
                IF possible_value GREATER THAN element at position i of dp  
                    SET element at position i of dp TO possible_value  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
