CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO list of integer numbers from one TO the LENGTH OF s PLUS one  
        SET start TO zero  
        FOR each end FROM zero TO the LENGTH OF s INCLUSIVE  
            IF end EQUALS the LENGTH OF s OR element at position end of s EQUALS the character I  
                REVERSE the elements from position start TO position end PLUS one of result  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
