CLASS Solution  
    FUNCTION fractionToDecimal(numerator, denominator)  
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero  
            SET sign TO the string representing a minus symbol  
        ELSE  
            SET sign TO the empty string  
        END IF  
        
        SET numerator TO the absolute value of numerator  
        SET denominator TO the absolute value of denominator  
        
        DIVIDE numerator BY denominator OBTAINING quotient AND remainder  
        SET integer_part TO the string representing quotient  
        
        IF remainder EQUALS zero  
            RETURN the concatenation of sign AND integer_part  
        END IF  
        
        INITIALIZE an empty dictionary named seen_remainders  
        INITIALIZE an empty list named fractional_part  
        
        WHILE remainder NOT EQUALS zero  
            IF remainder EXISTS AS A KEY IN seen_remainders  
                SET start TO the value associated with remainder in seen_remainders  
                SET non_repeating TO the concatenation of elements from the fractional_part list from the beginning up to position start exclusive  
                SET repeating TO the concatenation of elements from the fractional_part list from position start to the end  
                RETURN the concatenation of sign, integer_part, the string representing a decimal point, non_repeating, the string representing an opening parenthesis, repeating, and the string representing a closing parenthesis  
            END IF  
            
            ASSIGN the current length of fractional_part TO the dictionary seen_remainders with key remainder  
            
            SET remainder TO remainder MULTIPLIED BY ten  
            DIVIDE remainder BY denominator OBTAINING quotient AND remainder  
            APPEND the string representing quotient TO fractional_part  
        END WHILE  
        
        RETURN the concatenation of sign, integer_part, the string representing a decimal point, and the concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
