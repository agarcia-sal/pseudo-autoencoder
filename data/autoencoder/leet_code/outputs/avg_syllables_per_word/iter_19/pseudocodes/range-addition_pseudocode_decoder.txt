CLASS Solution  
    FUNCTION getModifiedArray(length, updates)  
        SET diff TO a list of zero repeated times length plus one  
        FOR each startIdx endIdx inc IN updates  
            INCREMENT element at position startIdx of diff BY inc  
            IF the sum of endIdx and one IS LESS THAN length  
                DECREMENT element at position the sum of endIdx and one of diff BY inc  
            END IF  
        END FOR  
        SET arr TO a list of zero repeated times length  
        SET element at position zero of arr TO element at position zero of diff  
        FOR i FROM one TO the difference of length and one  
            SET element at position i of arr TO the sum of element at position i minus one of arr AND element at position i of diff  
        END FOR  
        RETURN arr  
    END FUNCTION  
END CLASS
