CLASS Solution  
    FUNCTION combinationSum2(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR i FROM start TO the LENGTH OF candidates MINUS one  
                IF i GREATER THAN start AND element at position i of candidates EQUALS element at position i MINUS one of candidates  
                    CONTINUE TO the next iteration  
                END IF  
                CALL backtrack WITH i PLUS one FOR start AND target MINUS element at position i of candidates FOR target AND concatenation of path WITH element at position i of candidates FOR path  
            END FOR  
        END FUNCTION  
        CALL sort ON candidates TO ORDER the elements IN NON-DECREASING order  
        SET result TO an empty list  
        CALL backtrack WITH zero FOR start AND target FOR target AND empty list FOR path  
        RETURN result  
    END FUNCTION  
END CLASS
