CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO the SET of elements FROM bank  
        SET q TO a double ended queue initialized WITH a tuple of start and zero  
        SET mp TO a MAPPING of character A TO the string TCG and character T TO the string ACG and character C TO the string ATG and character G TO the string ATC  
        WHILE q is not EMPTY  
            REMOVE and RETURN the FIRST element FROM q and ASSIGN elements TO t and step respectively  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR index i and character v AS each POSITION and ELEMENT in t  
                FOR each character j IN the string mapped FROM v IN mp  
                    SET next TO the CONCATENATION of the substring FROM the START TO the POSITION i of t PLUS character j PLUS the substring FROM the POSITION i PLUS one TO the END of t  
                    IF next IS IN s  
                        APPEND the tuple of next and step PLUS one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
