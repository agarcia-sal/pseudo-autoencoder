CLASS Solution  
    FUNCTION minAbsDifference(nums, goal)  
        SET n TO the LENGTH OF nums  
        SET mid TO n DIVIDED BY two  
        
        SET left_sums TO a list containing zero  
        FOR each num IN the ELEMENTS from position zero TO mid MINUS one of nums  
            SET new_sums TO an empty list  
            FOR each s IN left_sums  
                APPEND THE sum of s PLUS num TO new_sums  
            END FOR  
            APPEND each element IN new_sums TO left_sums  
        END FOR  
        
        SET right_sums TO a list containing zero  
        FOR each num IN the ELEMENTS from position mid TO the END OF nums  
            SET new_sums TO an empty list  
            FOR each s IN right_sums  
                APPEND THE sum of s PLUS num TO new_sums  
            END FOR  
            APPEND each element IN new_sums TO right_sums  
        END FOR  
        
        SORT right_sums in ascending order  
        
        SET min_diff TO positive INFINITY  
        
        FOR each left_sum IN left_sums  
            SET target TO goal MINUS left_sum  
            SET pos TO the position found by locating the leftmost insertion point of target IN right_sums USING BINARY SEARCH  
            
            IF pos IS LESS THAN the LENGTH OF right_sums  
                SET current_diff TO the ABSOLUTE VALUE OF left_sum PLUS the element at position pos OF right_sums MINUS goal  
                IF current_diff IS LESS THAN min_diff  
                    SET min_diff TO current_diff  
                END IF  
            END IF  
            
            IF pos IS GREATER THAN zero  
                SET current_diff TO the ABSOLUTE VALUE OF left_sum PLUS the element at position pos MINUS one OF right_sums MINUS goal  
                IF current_diff IS LESS THAN min_diff  
                    SET min_diff TO current_diff  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
