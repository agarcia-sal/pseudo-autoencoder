CLASS Solution  
    FUNCTION minimumMoves(grid)  
        SET n TO the LENGTH OF grid  
        ASSIGN pointer/reference TO queue THE double ended queue initialized with a tuple containing zero zero zero one zero representing the row of tail the column of tail the row of head the column of head and the number of steps  
        ASSIGN pointer/reference TO visited THE set initialized with a tuple containing zero zero zero one representing visited states  
        
        WHILE queue is not empty  
            ASSIGN pointer/reference TO r1 THE first element of the tuple removed from the left side of queue  
            ASSIGN pointer/reference TO c1 THE second element of the tuple removed from the left side of queue  
            ASSIGN pointer/reference TO r2 THE third element of the tuple removed from the left side of queue  
            ASSIGN pointer/reference TO c2 THE fourth element of the tuple removed from the left side of queue  
            ASSIGN pointer/reference TO steps THE fifth element of the tuple removed from the left side of queue  
            
            IF r1 EQUALS n MINUS one AND c1 EQUALS n MINUS two AND r2 EQUALS n MINUS one AND c2 EQUALS n MINUS one THEN  
                RETURN steps  
            END IF  
            
            IF c2 PLUS one LESS THAN n AND the element at position r1 and position c1 PLUS one of grid EQUALS zero AND the element at position r2 and position c2 PLUS one of grid EQUALS zero AND the tuple containing r1 c1 PLUS one r2 c2 PLUS one is not in visited THEN  
                ADD the tuple containing r1 c1 PLUS one r2 c2 PLUS one TO visited  
                APPEND the tuple containing r1 c1 PLUS one r2 c2 PLUS one steps PLUS one TO queue  
            END IF  
            
            IF r2 PLUS one LESS THAN n AND the element at position r1 PLUS one and position c1 of grid EQUALS zero AND the element at position r2 PLUS one and position c2 of grid EQUALS zero AND the tuple containing r1 PLUS one c1 r2 PLUS one c2 is not in visited THEN  
                ADD the tuple containing r1 PLUS one c1 r2 PLUS one c2 TO visited  
                APPEND the tuple containing r1 PLUS one c1 r2 PLUS one c2 steps PLUS one TO queue  
            END IF  
            
            IF r1 EQUALS r2 AND r1 PLUS one LESS THAN n AND the element at position r1 PLUS one and position c1 of grid EQUALS zero AND the element at position r1 PLUS one and position c2 of grid EQUALS zero AND the tuple containing r1 c1 r1 PLUS one c1 is not in visited THEN  
                ADD the tuple containing r1 c1 r1 PLUS one c1 TO visited  
                APPEND the tuple containing r1 c1 r1 PLUS one c1 steps PLUS one TO queue  
            END IF  
            
            IF c1 EQUALS c2 AND c1 PLUS one LESS THAN n AND the element at position r1 and position c1 PLUS one of grid EQUALS zero AND the element at position r2 and position c2 PLUS one of grid EQUALS zero AND the tuple containing r1 c1 r1 c1 PLUS one is not in visited THEN  
                ADD the tuple containing r1 c1 r1 c1 PLUS one TO visited  
                APPEND the tuple containing r1 c1 r1 c1 PLUS one steps PLUS one TO queue  
            END IF  
        END WHILE  
        
        RETURN the integer minus one  
    END FUNCTION  
END CLASS
