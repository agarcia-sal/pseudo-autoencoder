CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET reversed_words TO sorted collection of each word REVERSED from words  
        SET total_length TO zero  
        FOR index FROM zero TO the LENGTH OF reversed_words MINUS one  
            IF index PLUS one EQUALS the LENGTH OF reversed_words OR the reversed_words element at position index PLUS one DOES NOT START WITH the reversed_words element at position index  
                INCREMENT total_length BY the LENGTH OF the reversed_words element at position index PLUS one  
            END IF  
        END FOR  
        RETURN total_length  
    END FUNCTION  
END CLASS
