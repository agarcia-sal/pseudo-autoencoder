CLASS Solution  
    FUNCTION findAllPeople(n AS integer, meetings AS list of list of integer, firstPerson AS integer) RETURNS list of integer  
        SET know_secret TO a set containing the integer zero AND the value of firstPerson  
        SORT meetings BY the value at position two of each meeting  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF meetings  
            SET current_time TO the value at position two of the element at position i of meetings  
            SET graph TO an empty mapping with default value as a list  
            WHILE i IS LESS THAN the LENGTH OF meetings AND the value at position two of the element at position i of meetings EQUALS current_time  
                SET x TO the element at position zero of the element at position i of meetings  
                SET y TO the element at position one of the element at position i of meetings  
                APPEND y TO the list at key x in graph  
                APPEND x TO the list at key y in graph  
                INCREMENT i BY one  
            END WHILE  
            SET queue TO a deque initialized with each person FROM know_secret WHERE that person IS A KEY IN graph  
            SET visited TO an empty set  
            WHILE the queue IS NOT EMPTY  
                REMOVE and RETURN the element at the front of queue AND SET TO person  
                IF person IS IN visited  
                    CONTINUE to the next iteration of the loop  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN the list at key person IN graph  
                    IF neighbor IS NOT IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN a list constructed from know_secret  
    END FUNCTION  
END CLASS
