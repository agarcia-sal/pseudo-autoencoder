CLASS Solution  
    FUNCTION maxDotProduct(nums1 LIST OF integer, nums2 LIST OF integer) RETURNS integer  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        SET dp TO a two dimensional list of size m PLUS one BY n PLUS one FILLED WITH negative infinity value  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO the element at position i MINUS one of nums1 MULTIPLIED BY the element at position j MINUS one of nums2  
                SET dp at position i at position j TO the MAXIMUM OF the following values  
                    current_product  
                    the element at position i MINUS one at position j of dp  
                    the element at position i at position j MINUS one of dp  
                    the element at position i MINUS one at position j MINUS one of dp PLUS current_product  
            END FOR  
        END FOR  
        RETURN the element at position m at position n of dp  
    END FUNCTION  
END CLASS
