CLASS Solution  
    FUNCTION kEmptySlots(bulbs AS List OF integers, k AS integer)  
        SET bloom_days TO an empty list  
        FOR each day WITH index starting from one, position IN bulbs  
            SET i TO the result of locating the position to insert position into bloom_days while maintaining sorted order  
            IF i is GREATER THAN zero AND position MINUS the element at position i MINUS one of bloom_days MINUS one EQUALS k  
                RETURN day  
            END IF  
            IF i is LESS THAN the LENGTH OF bloom_days AND the element at position i of bloom_days MINUS position MINUS one EQUALS k  
                RETURN day  
            END IF  
            INSERT position INTO bloom_days at position i to keep bloom_days sorted  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
