CLASS Solution  
    FUNCTION isNStraightHand(hand AS List OF integer, groupSize AS integer) RETURNS boolean  
        IF the LENGTH OF hand MODULO groupSize IS NOT EQUALS TO zero  
            RETURN False  
        END IF  
        SET count TO the frequency mapping of each element in hand  
        SET unique_cards TO the ascending ordered list of all unique elements from count keys  
        FOR each card IN unique_cards  
            IF the value at card in count IS GREATER THAN zero  
                SET needed TO the value at card in count  
                FOR each index FROM zero TO groupSize MINUS one  
                    IF the value at card PLUS index in count IS LESS THAN needed  
                        RETURN False  
                    END IF  
                    SET the value at card PLUS index in count TO the value at card PLUS index in count MINUS needed  
                END FOR  
            END IF  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
