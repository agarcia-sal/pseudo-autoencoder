CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content TO paths MAPPED TO list OF STRINGS  
        SET content_to_paths TO CALL defaultdict WITH list AS PARAMETER  
        
        FOR each path IN paths  
            SET parts TO path SPLIT BY SPACE  
            SET directory TO element at position one of parts  
            
            FOR each file_info IN elements from position two TO the END of parts  
                SET file_name TO substring from start TO position of OPEN PARENTHESIS IN file_info  
                SET content TO substring from position after OPEN PARENTHESIS TO one BEFORE the LAST CHARACTER of file_info  
                
                APPEND concatenation of directory PLUS FORWARD SLASH PLUS file_name TO element at key content OF content_to_paths  
            END FOR  
        END FOR  
        
        SET result TO empty list  
        FOR each paths_group IN values OF content_to_paths  
            IF the LENGTH OF paths_group GREATER THAN one  
                APPEND paths_group TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
