CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the result of splitting path BY the forward slash character  
        SET stack TO an empty list  
        FOR each component IN components  
            IF component EQUALS the empty string OR component EQUALS a single dot character  
                CONTINUE to the next iteration  
            ELSE IF component EQUALS two consecutive dots  
                IF the LENGTH OF stack IS GREATER THAN zero  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO concatenation OF the forward slash character AND the joining of elements in stack WITH the forward slash character AS separator  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
