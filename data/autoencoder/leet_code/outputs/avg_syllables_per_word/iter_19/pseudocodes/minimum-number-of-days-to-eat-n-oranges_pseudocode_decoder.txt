CLASS Solution  
    FUNCTION minDays WITH PARAMETER n  
        FUNCTION dp WITH PARAMETER remaining  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            SET result TO one PLUS THE MINIMUM OF the REMAINDER WHEN remaining IS DIVIDED BY two PLUS dp OF THE WHOLE NUMBER RESULT OF DIVIDING remaining BY two AND the REMAINDER WHEN remaining IS DIVIDED BY three PLUS dp OF THE WHOLE NUMBER RESULT OF DIVIDING remaining BY three  
            RETURN result  
        END FUNCTION  
        RETURN dp OF n  
END CLASS
