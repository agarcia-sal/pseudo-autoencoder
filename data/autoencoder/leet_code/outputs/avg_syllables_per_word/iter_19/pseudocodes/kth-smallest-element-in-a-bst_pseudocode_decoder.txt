CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at position zero of values  
    SET i TO one  
    SET queue TO new empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS greater THAN zero  
        SET node TO REMOVE element FROM the front of queue  
        IF i IS less THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode with value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS less THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode with value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(self, root, k)  
        SET stack TO new empty list  
        SET current TO root  
        SET count TO zero  
        WHILE the LENGTH OF stack IS greater THAN zero OR current IS NOT none  
            WHILE current IS NOT none  
                APPEND current TO stack  
                SET current TO current.left  
            END WHILE  
            SET current TO REMOVE last element FROM stack  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN current.val  
            END IF  
            SET current TO current.right  
        END WHILE  
    END FUNCTION  
END CLASS
