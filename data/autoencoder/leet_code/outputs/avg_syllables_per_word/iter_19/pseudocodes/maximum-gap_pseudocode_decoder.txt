CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF the LENGTH OF nums IS LESS THAN two  
            RETURN zero  
        END IF  
        
        FUNCTION countingSort(arr, exp)  
            SET n TO the LENGTH OF arr  
            INITIALIZE output AS a list of zeros of size n  
            INITIALIZE count AS a list of zeros of size ten  
            
            FOR each index i FROM zero TO n MINUS one  
                SET division_result TO element at position i of arr DIVIDED BY exp  
                SET digit TO the remainder of division_result DIVIDED BY ten  
                INCREMENT element at position digit of count BY one  
            END FOR  
            
            FOR each index i FROM one TO nine  
                INCREMENT element at position i of count BY element at position i MINUS one of count  
            END FOR  
            
            SET i TO n MINUS one  
            WHILE i IS GREATER THAN OR EQUAL TO zero  
                SET division_result TO element at position i of arr DIVIDED BY exp  
                SET digit TO the remainder of division_result DIVIDED BY ten  
                SET output at position element at position digit of count MINUS one TO element at position i of arr  
                DECREMENT element at position digit of count BY one  
                DECREMENT i BY one  
            END WHILE  
            
            FOR each index i FROM zero TO n MINUS one  
                SET element at position i of arr TO element at position i of output  
            END FOR  
        END FUNCTION  
        
        FUNCTION radixSort(arr)  
            SET max1 TO the maximum element of arr  
            SET exp TO one  
            WHILE max1 DIVIDED BY exp IS GREATER THAN zero  
                CALL countingSort WITH arr AND exp  
                SET exp TO exp MULTIPLIED BY ten  
            END WHILE  
        END FUNCTION  
        
        CALL radixSort WITH nums  
        
        SET max_diff TO zero  
        FOR each index i FROM one TO the LENGTH OF nums MINUS one  
            SET current_gap TO element at position i of nums MINUS element at position i MINUS one of nums  
            IF current_gap IS GREATER THAN max_diff  
                SET max_diff TO current_gap  
            END IF  
        END FOR  
        
        RETURN max_diff  
    END FUNCTION  
END CLASS
