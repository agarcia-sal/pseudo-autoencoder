CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator TO the iteration over t  
            RETURN the result of the test that for every character c in s the character c appears in the iterator  
        END FUNCTION  
        
        SORT strs BY the negative value of the length of each element AND BY the natural lexicographical order of each element  
        
        FOR each index i AND element word IN strs  
            SET unique TO true  
            FOR each index j AND element other IN strs  
                IF the value of i IS NOT EQUAL TO the value of j AND the result of is_subsequence with parameters word AND other IS true  
                    SET unique TO false  
                    EXIT the inner loop  
                END IF  
            END FOR  
            IF unique IS true  
                RETURN the length of word  
            END IF  
        END FOR  
        
        RETURN the integer negative one  
    END FUNCTION  
END CLASS
