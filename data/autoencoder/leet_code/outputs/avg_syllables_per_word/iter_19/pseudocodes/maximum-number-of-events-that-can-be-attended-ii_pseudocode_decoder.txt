CLASS Solution  
    FUNCTION maxValue(events, k)  
        SORT events BY the second element of each event in ascending order  
        SET n TO the LENGTH OF events  
        INITIALIZE dp AS a list containing n PLUS one elements each of which is a list of k PLUS one zeros  
        FOR i FROM one TO n  
            SET start TO the first element of the event at position i MINUS one in events  
            SET end TO the second element of the event at position i MINUS one in events  
            SET value TO the third element of the event at position i MINUS one in events  
            SET j TO the position found by locating the first event in events where the second element is GREATER THAN OR EQUAL TO start using binary search on the events second element  
            FOR l FROM one TO k  
                SET dp at position i and l TO the greater of dp at position i MINUS one and l AND dp at position j and l MINUS one PLUS value  
            END FOR  
        END FOR  
        RETURN dp at position n and k  
    END FUNCTION  
END CLASS
