CLASS Solution  
    FUNCTION hasPath(maze, start, destination)  
        SET directions TO the list of tuples zero and one one and zero zero and negative one negative one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET visited TO an empty set  
        
        FUNCTION dfs(x, y)  
            IF tuple composed of x and y IS IN visited THEN  
                RETURN false  
            END IF  
            IF list composed of x and y EQUALS destination THEN  
                RETURN true  
            END IF  
            ADD tuple composed of x and y TO visited  
            
            FOR each tuple composed of dx and dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx plus dx IS GREATER THAN OR EQUAL TO zero AND nx plus dx IS LESS THAN m AND ny plus dy IS GREATER THAN OR EQUAL TO zero AND ny plus dy IS LESS THAN n AND element at position nx plus dx of maze AND element at position ny plus dy of that element EQUALS zero  
                    SET nx TO nx plus dx  
                    SET ny TO ny plus dy  
                END WHILE  
                IF dfs(nx, ny) EQUALS true THEN  
                    RETURN true  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN dfs(element at position zero of start, element at position one of start)  
    END FUNCTION  
END CLASS
