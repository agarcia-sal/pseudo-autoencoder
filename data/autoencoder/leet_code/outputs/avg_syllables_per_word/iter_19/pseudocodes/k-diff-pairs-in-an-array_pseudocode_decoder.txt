CLASS Solution  
    FUNCTION findPairs(nums AS List OF Integer, k AS Integer) RETURNS Integer  
        IF k IS LESS THAN zero  
            RETURN zero  
        END IF  
        
        SET num_counts TO empty mapping from Integer to Integer  
        FOR each num IN nums  
            IF num IS A KEY IN num_counts  
                SET current_count TO value FOR key num IN num_counts  
            ELSE  
                SET current_count TO zero  
            END IF  
            SET value FOR key num IN num_counts TO current_count PLUS one  
        END FOR  
        
        SET unique_pairs TO zero  
        FOR each num IN keys of num_counts  
            IF k EQUALS zero  
                IF value FOR key num IN num_counts IS GREATER THAN one  
                    INCREMENT unique_pairs BY one  
                END IF  
            ELSE  
                IF num PLUS k IS A KEY IN num_counts  
                    INCREMENT unique_pairs BY one  
                END IF  
            END IF  
        END FOR  
        
        RETURN unique_pairs  
    END FUNCTION  
END CLASS
