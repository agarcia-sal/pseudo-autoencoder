CLASS Solution  
    FUNCTION shortestDistance WITH PARAMETERS maze AS list of list of integers start AS list of integers destination AS list of integers  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET directions TO list containing the pairs one AND zero MINUS one AND zero zero AND one zero AND MINUS one  
        SET start TO tuple conversion of start  
        SET destination TO tuple conversion of destination  
        SET visited TO an empty set  
        SET pq TO list containing a pair of zero AND start representing distance AND position  
        
        WHILE pq is not empty  
            EXTRACT the smallest distance AND position pair FROM pq ASSIGNING distance TO dist AND position TO (x AND y)  
            
            IF x EQUALS element at position zero of destination AND y EQUALS element at position one of destination  
                RETURN dist  
            END IF  
            
            IF position (x AND y) IS IN visited  
                CONTINUE to the next iteration of the loop  
            END IF  
            
            ADD position (x AND y) TO visited  
            
            FOR each pair dx AND dy IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx OF maze AT element new_y PLUS dy EQUALS zero  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY one  
                END WHILE  
                
                IF position (new_x AND new_y) IS NOT IN visited  
                    ADD pair of dist PLUS count AND position (new_x AND new_y) TO pq USING heap push operation  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
