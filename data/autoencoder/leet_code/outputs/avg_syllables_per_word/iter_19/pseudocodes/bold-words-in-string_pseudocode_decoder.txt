CLASS Solution  
    FUNCTION boldWords(words AS list of strings, s AS string) RETURNS string  
        SET bold TO a list of boolean values set to false of the LENGTH OF s  
        
        FOR each word IN words  
            SET start TO the position of word IN s or negative one if not found  
            WHILE start IS NOT EQUAL TO negative one  
                FOR each index i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i of bold TO true  
                END FOR  
                SET start TO the position of word IN s STARTING FROM start PLUS one or negative one if not found  
            END WHILE  
        END FOR  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i IS LESS THAN the LENGTH OF s  
            IF the element at position i of bold IS true  
                APPEND the string identified as opening bold tag TO result  
                WHILE i IS LESS THAN the LENGTH OF s AND the element at position i of bold IS true  
                    APPEND the character at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string identified as closing bold tag TO result  
            ELSE  
                APPEND the character at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements IN result  
    END FUNCTION  
END CLASS
