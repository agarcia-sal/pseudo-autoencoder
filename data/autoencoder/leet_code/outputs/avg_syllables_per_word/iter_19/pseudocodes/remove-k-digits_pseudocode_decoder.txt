CLASS Solution  
    FUNCTION removeKdigits(num AS string, k AS integer) RETURNS string  
        SET stack TO empty list  
        
        FOR each digit IN num  
            WHILE k is GREATER THAN zero AND stack is NOT empty AND element at last position of stack is GREATER THAN digit  
                REMOVE element at last position FROM stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        
        IF k is GREATER THAN zero  
            SET final_stack TO elements from position zero TO the LENGTH OF stack MINUS k of stack  
        ELSE  
            SET final_stack TO stack  
        END IF  
        
        SET result TO concatenation of all elements in final_stack with all leading zero characters REMOVED  
        
        IF result is NOT empty  
            RETURN result  
        ELSE  
            RETURN string zero  
        END IF  
    END FUNCTION  
END CLASS
