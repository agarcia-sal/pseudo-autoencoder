CLASS Solution  
    FUNCTION findMaxValueOfEquation WITH PARAMETERS points AND k  
        SET max_value TO negative infinity  
        CREATE an empty double ended queue named deque_  
        FOR each pair of elements named xj AND yj IN points  
            WHILE deque_ is not empty AND the result of subtracting the element at position one of the first element of deque_ FROM xj IS GREATER THAN k  
                REMOVE the first element FROM deque_  
            END WHILE  
            IF deque_ is not empty  
                SET max_value TO the greater value BETWEEN max_value AND the result of adding yj TO xj AND TO the element at position zero of the first element of deque_  
            END IF  
            WHILE deque_ is not empty AND the result of subtracting xj FROM yj IS GREATER THAN OR EQUAL TO the element at position zero of the last element of deque_  
                REMOVE the last element FROM deque_  
            END WHILE  
            APPEND the pair consisting of the result of subtracting xj FROM yj AND xj TO deque_  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
