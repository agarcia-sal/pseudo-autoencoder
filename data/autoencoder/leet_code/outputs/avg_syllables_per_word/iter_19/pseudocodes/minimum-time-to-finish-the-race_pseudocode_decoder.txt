CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO a list of fifteen values each set TO positive infinity  
        FOR each pair OF f and r IN tires  
            SET total_time TO zero  
            FOR index FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER OF index  
                IF lap_time GREATER THAN changeTime PLUS f  
                    BREAK the loop  
                END IF  
                INCREMENT total_time BY lap_time  
                IF min_times element at position index GREATER THAN total_time  
                    SET min_times element at position index TO total_time  
                END IF  
            END FOR  
        END FOR  
        
        SET dp TO a list of numLaps PLUS one values each set TO positive infinity  
        SET element at position zero of dp TO zero  
        
        FOR i FROM one TO numLaps  
            FOR j FROM zero TO the smaller value BETWEEN i MINUS one AND fourteen  
                SET possible_time TO element at position i MINUS j MINUS one of dp PLUS element at position j of min_times PLUS changeTime  
                IF element at position i of dp GREATER THAN possible_time  
                    SET element at position i of dp TO possible_time  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position numLaps of dp MINUS changeTime  
    END FUNCTION  
END CLASS
