CLASS Solution  
    FUNCTION digitsCount(d integer, low integer, high integer) RETURNS integer  
        FUNCTION count_digit(n integer, digit integer) RETURNS integer  
            SET count TO zero  
            SET power_of_10 TO one  
            SET current TO n  
            WHILE current GREATER THAN zero holds  
                SET last TO the remainder of current DIVIDED BY ten  
                SET current TO the result of current DIVIDED BY ten removing the remainder  
                SET full_cycles TO current  
                IF digit EQUALS zero THEN  
                    DECREMENT full_cycles BY one  
                END IF  
                INCREMENT count BY full_cycles MULTIPLIED BY power_of_10  
                IF last GREATER THAN digit THEN  
                    INCREMENT count BY power_of_10  
                ELSE IF last EQUALS digit THEN  
                    INCREMENT count BY the remainder of n DIVIDED BY power_of_10 PLUS one  
                END IF  
                SET power_of_10 TO power_of_10 MULTIPLIED BY ten  
            END WHILE  
            RETURN count  
        END FUNCTION  
        RETURN the result of count_digit with parameters high and d MINUS the result of count_digit with parameters low MINUS one and d  
    END FUNCTION  
END CLASS
