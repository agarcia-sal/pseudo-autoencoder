CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO A new TreeNode WITH PARAMETER the element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY DOUBLE ENDED QUEUE  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO A new TreeNode WITH PARAMETER the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO A new TreeNode WITH PARAMETER the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUAL TO q dot val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS p dot left q AND AND is_same_tree WITH PARAMETERS p dot right q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves WITH PARAMETERS self root  
        FUNCTION collectLeaves WITH PARAMETER node  
            IF node IS none  
                RETURN minus one  
            END IF  
            SET left_height TO collectLeaves WITH PARAMETER node dot left  
            SET right_height TO collectLeaves WITH PARAMETER node dot right  
            SET current_height TO THE GREATER OF left_height AND right_height PLUS one  
            IF current_height IS GREATER THAN OR EQUAL TO THE LENGTH OF result  
                APPEND AN EMPTY LIST TO result  
            END IF  
            APPEND node dot val TO THE ELEMENT AT POSITION current_height OF result  
            RETURN current_height  
        END FUNCTION  
        SET result TO AN EMPTY LIST  
        CALL collectLeaves WITH PARAMETER root  
        RETURN result  
    END FUNCTION  
END CLASS
