CLASS Solution  
    FUNCTION shortestWordDistance WITH PARAMETERS wordsDict word list OF strings word1 string word2 string  
        SET min_distance TO positive infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO the result of word1 EQUALS word2  
        
        FOR each index i AND element word IN wordsDict  
            IF word EQUALS word1  
                IF same_word  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO i  
            END IF  
            
            IF word EQUALS word2  
                IF NOT same_word  
                    SET prev_word2 TO i  
                END IF  
            END IF  
            
            IF prev_word1 IS NOT EQUAL TO negative one AND prev_word2 IS NOT EQUAL TO negative one  
                SET absolute_difference TO the absolute value of prev_word1 MINUS prev_word2  
                IF absolute_difference LESS THAN min_distance  
                    SET min_distance TO absolute_difference  
                END IF  
            END IF  
        END FOR  
        
        RETURN min_distance  
    END FUNCTION  
END CLASS
