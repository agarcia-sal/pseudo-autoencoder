CLASS Solution  
    FUNCTION longestSubsequenceRepeatedK WITH PARAMETERS s AND k  
        CREATE variable freq AS the mapping of characters TO their frequency in string s  
        CREATE list valid_chars CONTAINING each character THAT HAS frequency AT LEAST k IN freq  
        SORT valid_chars IN descending lexicographical order  
        
        FUNCTION is_valid_subsequence WITH PARAMETER candidate  
            SET count TO zero  
            SET index TO zero  
            FOR each character IN string s  
                IF character EQUALS element at position index OF candidate THEN  
                    INCREMENT index BY one  
                    IF index EQUALS the LENGTH OF candidate THEN  
                        INCREMENT count BY one  
                        SET index TO zero  
                        IF count EQUALS k THEN  
                            RETURN True  
                        END IF  
                    END IF  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        SET max_length TO the LENGTH OF valid_chars DIVIDED BY k  
        SET best_candidate TO an empty string  
        
        FUNCTION backtrack WITH PARAMETER candidate  
            IF the LENGTH OF candidate IS GREATER THAN the LENGTH OF best_candidate THEN  
                SET best_candidate TO candidate  
            END IF  
            FOR each character IN valid_chars  
                CREATE new_candidate AS candidate FOLLOWED BY character  
                IF is_valid_subsequence(new_candidate) THEN  
                    CALL backtrack WITH new_candidate  
                END IF  
            END FOR  
        END FUNCTION  
        
        CALL backtrack WITH an empty string  
        RETURN best_candidate  
    END FUNCTION  
END CLASS
