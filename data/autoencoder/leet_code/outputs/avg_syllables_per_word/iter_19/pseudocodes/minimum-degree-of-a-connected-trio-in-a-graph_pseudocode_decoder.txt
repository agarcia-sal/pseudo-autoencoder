CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        CREATE a data structure graph AS a mapping FROM each integer FROM one TO n TO an empty set  
        FOR each pair u and v IN edges  
            ADD element v TO the set at key u in graph  
            ADD element u TO the set at key v in graph  
        END FOR  
        SET min_degree TO positive infinity  
        SET trio_found TO false  
        FOR i FROM one TO n  
            FOR j FROM the integer AFTER i TO n  
                FOR k FROM the integer AFTER j TO n  
                    IF element j IS IN the set at key i in graph AND element k IS IN the set at key i in graph AND element k IS IN the set at key j in graph  
                        SET trio_found TO true  
                        SET trio_degree TO the LENGTH OF the set at key i in graph PLUS the LENGTH OF the set at key j in graph PLUS the LENGTH OF the set at key k in graph MINUS six  
                        IF trio_degree IS LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF trio_found  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
