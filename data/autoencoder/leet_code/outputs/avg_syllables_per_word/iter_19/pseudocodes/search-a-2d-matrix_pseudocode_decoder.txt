CLASS Solution  
    FUNCTION searchMatrix(matrix parameter LIST OF LIST OF INTEGER target parameter INTEGER) RETURNS BOOLEAN  
        IF matrix IS EMPTY OR element at position zero of matrix IS EMPTY  
            RETURN FALSE  
        END IF  
        SET variable m TO the LENGTH OF matrix  
        SET variable n TO the LENGTH OF element at position zero of matrix  
        SET variable left TO zero  
        SET variable right TO m MULTIPLIED BY n MINUS one  
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET variable mid TO left PLUS right DIVIDED BY two USING INTEGER DIVISION  
            SET variable mid_value TO element at position mid DIVIDED BY n USING INTEGER DIVISION of matrix AT element at position mid MODULO n OF matrix  
            IF mid_value EQUALS target  
                RETURN TRUE  
            ELSE IF mid_value IS LESS THAN target  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN FALSE  
    END FUNCTION  
END CLASS
