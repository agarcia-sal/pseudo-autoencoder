CLASS Solution  
    FUNCTION minCost(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO list of four pairs representing zero and one, zero and negative one, one and zero, negative one and zero  
        SET costs TO a two dimensional list of size m by n with each element set to positive infinity value  
        SET element at position zero of element at position zero of costs TO zero  
        
        SET pq TO list containing a triple of zero zero zero representing the priority queue  
        SET visited TO an empty set  
        
        WHILE pq holds elements  
            EXTRACT the element with smallest cost triple FROM pq into cost i j  
            
            IF the pair of i and j IS IN visited  
                CONTINUE to next iteration  
            END IF  
            ADD the pair of i and j TO visited  
            
            IF i EQUALS m MINUS one AND j EQUALS n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each index k AND pair di dj FROM directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN m AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n  
                    IF k PLUS one IS NOT EQUAL TO element at position i of grid at position j  
                        SET additional_cost TO one  
                    ELSE  
                        SET additional_cost TO zero  
                    END IF  
                    SET new_cost TO cost PLUS additional_cost  
                    IF new_cost IS LESS THAN element at position ni of costs at position nj  
                        SET element at position ni of costs at position nj TO new_cost  
                        INSERT the triple of new_cost ni nj INTO pq maintaining heap property  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
