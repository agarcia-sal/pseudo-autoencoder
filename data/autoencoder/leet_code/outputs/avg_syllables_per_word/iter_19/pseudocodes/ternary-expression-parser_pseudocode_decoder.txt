CLASS Solution  
    FUNCTION parseTernary(expression)  
        FUNCTION evaluate(index)  
            IF index EQUALS the LENGTH OF expression MINUS one OR the character at position index PLUS one of expression EQUALS a colon character  
                RETURN the character at position index of expression AND index PLUS two  
            END IF  
            SET true_part AND next_index TO the result of calling evaluate WITH index PLUS two  
            SET false_part AND final_index TO the result of calling evaluate WITH next_index  
            IF the character at position index of expression EQUALS the character T  
                RETURN true_part AND final_index  
            ELSE  
                RETURN false_part AND final_index  
            END IF  
        END FUNCTION  
        SET result AND an unused variable TO the result of calling evaluate WITH zero  
        RETURN result  
    END FUNCTION  
END CLASS
