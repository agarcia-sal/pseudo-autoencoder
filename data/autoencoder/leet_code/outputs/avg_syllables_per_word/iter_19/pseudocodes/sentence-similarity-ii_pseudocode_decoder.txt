CLASS Solution  
    FUNCTION areSentencesSimilarTwo(sentence1 AS List OF STRING, sentence2 AS List OF STRING, similarPairs AS List OF List OF STRING) RETURNS BOOLEAN  
        IF the LENGTH OF sentence1 IS NOT EQUAL TO the LENGTH OF sentence2  
            RETURN false  
        END IF  

        SET parent TO an empty map  

        FUNCTION find(x) RETURNS STRING  
            IF the element at key x of parent IS NOT EQUAL TO x  
                SET the element at key x of parent TO the result of find with the element at key x of parent  
            END IF  
            RETURN the element at key x of parent  
        END FUNCTION  

        FUNCTION union(x, y)  
            SET rootX TO the result of find with x  
            SET rootY TO the result of find with y  
            IF rootX IS NOT EQUAL TO rootY  
                SET the element at key rootY of parent TO rootX  
            END IF  
        END FUNCTION  

        FOR each pair IN similarPairs  
            FOR each word IN pair  
                IF word IS NOT a key in parent  
                    SET the element at key word of parent TO word  
                END IF  
            END FOR  
        END FOR  

        FOR each x comma y IN similarPairs  
            CALL union with x and y  
        END FOR  

        FOR each w1 comma w2 IN the paired elements taken one from sentence1 and one from sentence2 in order  
            IF w1 IS NOT a key in parent  
                SET the element at key w1 of parent TO w1  
            END IF  
            IF w2 IS NOT a key in parent  
                SET the element at key w2 of parent TO w2  
            END IF  
            IF the result of find with w1 IS NOT EQUAL TO the result of find with w2  
                RETURN false  
            END IF  
        END FOR  

        RETURN true  
    END FUNCTION  
END CLASS
