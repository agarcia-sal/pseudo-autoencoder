CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO the list of individual characters derived from converting n to a string  
        SET length TO the number of elements in digits  
        SET i TO length MINUS two  
        WHILE i is greater than or equal to zero AND the element at position i of digits is greater than or equal to the element at position i plus one of digits  
            DECREMENT i BY one  
        END WHILE  
        IF i EQUALS negative one  
            RETURN negative one  
        END IF  
        SET j TO length MINUS one  
        WHILE the element at position j of digits is less than or equal to the element at position i of digits  
            DECREMENT j BY one  
        END WHILE  
        SWAP the elements at positions i and j of digits  
        SET digits TO the concatenation of the elements from the start to position i inclusive of digits WITH the elements from position i plus one to the end of digits in reverse order  
        SET result TO the integer obtained by concatenating all elements in digits in order  
        IF result is greater than two to the power of thirty-one MINUS one  
            RETURN negative one  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
