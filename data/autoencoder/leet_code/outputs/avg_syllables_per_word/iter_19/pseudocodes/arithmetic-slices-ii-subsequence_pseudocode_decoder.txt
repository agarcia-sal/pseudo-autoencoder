CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        SET dp TO a new list containing n elements each initialized as a default dictionary mapping from integer keys to zero values  
        SET total_count TO zero  
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO i MINUS one  
                SET d TO the element at position i of nums MINUS the element at position j of nums  
                IF the key d EXISTS IN the element at position j of dp  
                    INCREMENT the value at key d of the element at position i of dp BY the value at key d of the element at position j of dp PLUS one  
                    INCREMENT total_count BY the value at key d of the element at position j of dp  
                ELSE  
                    INCREMENT the value at key d of the element at position i of dp BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN total_count  
    END FUNCTION  
END CLASS
