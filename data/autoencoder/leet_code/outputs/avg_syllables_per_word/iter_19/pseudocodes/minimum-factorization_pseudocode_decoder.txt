CLASS Solution  
    FUNCTION smallestFactorization(num)  
        IF num EQUALS one THEN  
            RETURN one  
        END IF  
        
        SET factors TO an empty list  
        FOR digit FROM nine TO two DECREMENTING BY one  
            WHILE the remainder of num DIVIDED BY digit EQUALS zero  
                APPEND digit TO factors  
                SET num TO the integer quotient of num DIVIDED BY digit  
            END WHILE  
        END FOR  
        
        IF num NOT EQUALS one THEN  
            RETURN zero  
        END IF  
        
        SORT factors IN ascending order  
        
        SET result TO the integer value of concatenating all elements from factors as strings in order  
        
        IF result GREATER THAN two raised to the power of thirty ONE MINUS one THEN  
            RETURN zero  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
