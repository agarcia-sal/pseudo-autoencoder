CLASS Solution  
    FUNCTION findSubstring(s STRING, words LIST OF STRING) RETURNS LIST OF INTEGER  
        IF s EQUALS empty STRING OR words EQUALS empty LIST OR element at position zero of words EQUALS empty STRING  
            RETURN empty LIST  
        END IF  
        
        SET word_length TO LENGTH OF element at position zero of words  
        SET num_words TO LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO THE COUNT OF EACH UNIQUE ELEMENT IN words  
        SET result TO empty LIST  
        
        FOR i FROM zero TO LENGTH OF s MINUS total_length PLUS one  
            SET seen TO EMPTY COUNT OF WORDS  
            FOR j FROM i TO i PLUS total_length MINUS one STEP BY word_length  
                SET word TO SUBSTRING FROM position j TO position j PLUS word_length MINUS one OF s  
                INCREMENT count of word IN seen BY one  
                IF count of word IN seen GREATER THAN count of word IN word_count  
                    BREAK INNER LOOP  
                END IF  
            END FOR  
            IF INNER LOOP COMPLETED WITHOUT BREAK  
                APPEND i TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
