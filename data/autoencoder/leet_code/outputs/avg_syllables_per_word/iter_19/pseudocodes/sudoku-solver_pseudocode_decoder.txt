CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR index FROM zero TO eight  
                IF element at position row of board AT position index EQUALS num  
                    RETURN false  
                END IF  
            END FOR  
            FOR index FROM zero TO eight  
                IF element at position index of board AT position col EQUALS num  
                    RETURN false  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY the INTEGER DIVISION OF row BY three  
            SET start_col TO three MULTIPLIED BY the INTEGER DIVISION OF col BY three  
            FOR i FROM zero TO two  
                FOR j FROM zero TO two  
                    IF element at position start_row PLUS i of board AT position start_col PLUS j EQUALS num  
                        RETURN false  
                    END IF  
                END FOR  
            END FOR  
            RETURN true  
        END FUNCTION  
        FUNCTION solve(board)  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF element at position row of board AT position col EQUALS the character representing empty cell  
                        FOR num IN the sequence of characters from one TO nine AS strings  
                            IF is_valid(board, row, col, num)  
                                SET element at position row of board AT position col TO num  
                                IF solve(board)  
                                    RETURN true  
                                END IF  
                                SET element at position row of board AT position col TO the character representing empty cell  
                            END IF  
                        END FOR  
                        RETURN false  
                    END IF  
                END FOR  
            END FOR  
            RETURN true  
        END FUNCTION  
        CALL solve(board)  
    END FUNCTION  
END CLASS
