CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode constructed WITH the element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double-ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        REMOVE the element from the left end of queue AND ASSIGN IT TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot left TO a new TreeNode constructed WITH the element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node dot right TO a new TreeNode constructed WITH the element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p dot val IS NOT EQUALS TO q dot val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS p dot left and q dot left AND the result of is_same_tree WITH PARAMETERS p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree WITH PARAMETERS self root  
        IF root IS none  
            RETURN none  
        END IF  
        SET root dot left TO the result of self dot pruneTree WITH PARAMETER root dot left  
        SET root dot right TO the result of self dot pruneTree WITH PARAMETER root dot right  
        IF root dot val EQUALS zero AND root dot left IS none AND root dot right IS none  
            RETURN none  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
