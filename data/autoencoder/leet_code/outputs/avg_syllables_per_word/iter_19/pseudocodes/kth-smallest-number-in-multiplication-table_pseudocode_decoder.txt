CLASS Solution  
    FUNCTION findKthNumber(m integer_parameter n integer_parameter k integer_parameter) RETURNS integer  
        FUNCTION enough(x integer_parameter) RETURNS boolean  
            SET count TO zero  
            FOR each integer i FROM one TO m  
                INCREMENT count BY THE MINIMUM BETWEEN the INTEGER PART OF x DIVIDED BY i AND n  
            END FOR  
            IF count GREATER THAN OR EQUAL TO k THEN  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        SET low TO one  
        SET high TO m MULTIPLIED BY n  
        WHILE low LESS THAN high  
            SET mid TO THE INTEGER PART OF low PLUS high DIVIDED BY two  
            IF enough(mid) THEN  
                SET high TO mid  
            ELSE  
                SET low TO low PLUS one  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
