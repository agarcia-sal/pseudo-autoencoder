CLASS Solution  
    FUNCTION isEscapePossible(blocked, source, target)  
        IF blocked IS empty  
            RETURN True  
        END IF  
        
        SET blocked_set TO the set of tuples formed from each block in blocked  
        SET n TO the LENGTH OF blocked  
        SET max_steps TO n MULTIPLIED BY n MINUS one THEN DIVIDED BY two  
        
        FUNCTION bfs(start, end)  
            SET queue TO a double ended queue initialized with the tuple formed from start  
            SET visited TO the set initialized with the elements of queue  
            
            WHILE queue IS NOT empty  
                SET x TO the first element of the tuple at the front of queue popped from the left  
                SET y TO the second element of the tuple at the front of queue popped from the left  
                
                IF the LENGTH OF visited IS GREATER THAN max_steps OR x EQUALS the first element of end AND y EQUALS the second element of end  
                    RETURN True  
                END IF  
                
                FOR each pair of dx and dy IN the list containing zero and one, zero and minus one, one and zero, minus one and zero  
                    SET nx TO x PLUS dx  
                    SET ny TO y PLUS dy  
                    IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN ten TO THE POWER OF six AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN ten TO THE POWER OF six  
                        AND the tuple formed from nx and ny IS NOT IN blocked_set  
                        AND the tuple formed from nx and ny IS NOT IN visited  
                        ADD the tuple formed from nx and ny TO visited  
                        APPEND the tuple formed from nx and ny TO the right end of queue  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN False  
        END FUNCTION  
        
        RETURN the result of bfs called with source and target AND the result of bfs called with target and source  
    END FUNCTION  
END CLASS
