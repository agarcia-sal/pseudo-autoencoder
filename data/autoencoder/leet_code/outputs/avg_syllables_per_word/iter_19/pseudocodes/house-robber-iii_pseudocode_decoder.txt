CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO AN EMPTY double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO A NEW TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN THE RESULT OF is_same_tree CALLED WITH PARAMETERS p.left AND q.left AND THE RESULT OF is_same_tree CALLED WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rob WITH PARAMETERS self root  
        FUNCTION rob_from_node WITH PARAMETER node  
            IF node IS none  
                RETURN zero AND zero  
            END IF  
            SET left_robbed AND left_not_robbed TO THE RESULT OF rob_from_node CALLED WITH node.left  
            SET right_robbed AND right_not_robbed TO THE RESULT OF rob_from_node CALLED WITH node.right  
            SET rob_this_node TO node.val PLUS left_not_robbed PLUS right_not_robbed  
            SET not_rob_this_node TO THE MAXIMUM BETWEEN left_robbed AND left_not_robbed PLUS THE MAXIMUM BETWEEN right_robbed AND right_not_robbed  
            RETURN rob_this_node AND not_rob_this_node  
        END FUNCTION  
        RETURN THE MAXIMUM OF rob_from_node CALLED WITH root  
    END FUNCTION  
END CLASS
