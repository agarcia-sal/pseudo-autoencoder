CLASS Solution  
    FUNCTION numTilings WITH PARAMETER n  
        SET MOD TO the number one billion PLUS seven  
        
        SET dp TO a list of zeros with LENGTH equal TO n PLUS one  
        SET dp1 TO a list of zeros with LENGTH equal TO n PLUS one  
        
        SET element at position zero of dp TO one  
        SET element at position one of dp TO one  
        
        IF n IS GREATER THAN OR EQUAL TO two  
            SET element at position two of dp TO two  
            SET element at position two of dp1 TO one  
        END IF  
        
        FOR index FROM three TO n  
            SET element at position index of dp TO the SUM of element at position index MINUS one of dp PLUS element at position index MINUS two of dp PLUS two TIMES element at position index MINUS one of dp1  
            SET element at position index of dp TO element at position index of dp MODULO MOD  
            
            SET element at position index of dp1 TO the SUM of element at position index MINUS two of dp PLUS element at position index MINUS one of dp1  
            SET element at position index of dp1 TO element at position index of dp1 MODULO MOD  
        END FOR  
        
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
