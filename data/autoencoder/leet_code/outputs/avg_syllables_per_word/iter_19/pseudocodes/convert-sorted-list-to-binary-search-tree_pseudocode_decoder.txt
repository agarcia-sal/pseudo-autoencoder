CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO NEW ListNode WITH ARGUMENT element at position zero of values  
    SET p TO head  
    FOR each val IN elements from position one TO end of values  
        SET node TO NEW ListNode WITH ARGUMENT val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND is_same_list WITH ARGUMENTS next of p1 AND next of p2  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND left EQUALS none AND right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH ARGUMENT element at position zero of values  
    SET i TO one  
    SET queue TO a new double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        REMOVE and RETURN the element at position zero of queue INTO node  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET left of node TO NEW TreeNode WITH ARGUMENT element at position i of values  
            APPEND left of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET right of node TO NEW TreeNode WITH ARGUMENT element at position i of values  
            APPEND right of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p IS NOT EQUALS TO value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH ARGUMENTS left of p AND left of q AND is_same_tree WITH ARGUMENTS right of p AND right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST WITH PARAMETERS head  
        IF head IS none  
            RETURN none  
        END IF  
        
        FUNCTION findMiddle WITH PARAMETERS left AND right  
            SET slow TO left  
            SET fast TO left  
            WHILE fast IS NOT EQUALS TO right AND next of fast IS NOT EQUALS TO right  
                SET slow TO next of slow  
                SET fast TO next of next of fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  
        
        FUNCTION convertListToBST WITH PARAMETERS left AND right  
            IF left IS EQUALS TO right  
                RETURN none  
            END IF  
            
            SET mid TO findMiddle WITH ARGUMENTS left AND right  
            
            SET node TO NEW TreeNode WITH ARGUMENT value of mid  
            SET left of node TO convertListToBST WITH ARGUMENTS left AND mid  
            SET right of node TO convertListToBST WITH ARGUMENTS next of mid AND right  
            RETURN node  
        END FUNCTION  
        
        RETURN convertListToBST WITH ARGUMENTS head AND none  
    END FUNCTION  
END CLASS
