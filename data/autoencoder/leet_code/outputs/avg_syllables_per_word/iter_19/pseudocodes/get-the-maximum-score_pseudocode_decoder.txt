CLASS Solution  
    FUNCTION maxSum(nums1 AS list of integers, nums2 AS list of integers) RETURNS integer  
        SET i TO zero  
        SET j TO zero  
        SET sum1 TO zero  
        SET sum2 TO zero  
        SET MOD TO one billion PLUS seven  
        
        WHILE i IS LESS THAN the LENGTH OF nums1 AND j IS LESS THAN the LENGTH OF nums2  
            IF element at position i of nums1 IS LESS THAN element at position j of nums2  
                INCREMENT sum1 BY element at position i of nums1  
                INCREMENT i BY one  
            ELSE IF element at position i of nums1 IS GREATER THAN element at position j of nums2  
                INCREMENT sum2 BY element at position j of nums2  
                INCREMENT j BY one  
            ELSE  
                SET sum1 TO THE MAXIMUM OF sum1 AND sum2 PLUS element at position i of nums1  
                SET sum2 TO sum1  
                INCREMENT i BY one  
                INCREMENT j BY one  
            END IF  
        END WHILE  
        
        WHILE i IS LESS THAN the LENGTH OF nums1  
            INCREMENT sum1 BY element at position i of nums1  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE j IS LESS THAN the LENGTH OF nums2  
            INCREMENT sum2 BY element at position j of nums2  
            INCREMENT j BY one  
        END WHILE  
        
        RETURN THE MAXIMUM OF sum1 AND sum2 MODULO MOD  
    END FUNCTION  
END CLASS
