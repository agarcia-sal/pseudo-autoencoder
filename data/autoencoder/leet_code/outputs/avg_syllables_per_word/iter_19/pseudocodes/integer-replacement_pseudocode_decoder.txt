CLASS Solution  
    FUNCTION integerReplacement WITH PARAMETER n  
        SET count TO zero  
        WHILE n IS NOT EQUALS TO one  
            IF n MODULO two EQUALS zero  
                SET n TO n DIVIDED BY two  
            ELSE  
                IF n EQUALS three  
                    SET n TO n MINUS one  
                ELSE IF n PLUS one MODULO four EQUALS zero  
                    SET n TO n PLUS one  
                ELSE  
                    SET n TO n MINUS one  
                END IF  
            END IF  
            INCREMENT count BY one  
        END WHILE  
        RETURN count  
    END FUNCTION  
END CLASS
