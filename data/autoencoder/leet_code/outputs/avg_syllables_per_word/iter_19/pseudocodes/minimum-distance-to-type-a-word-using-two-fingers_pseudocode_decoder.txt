CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS None THEN  
                RETURN zero  
            END IF  
            SET x1 TO the RESULT OF THE INTEGER DIVISION OF the NUMERIC CODE OF c1 MINUS the NUMERIC CODE OF the LETTER A BY six  
            SET y1 TO the REMAINDER OF THE INTEGER DIVISION OF the NUMERIC CODE OF c1 MINUS the NUMERIC CODE OF the LETTER A BY six  
            SET x2 TO the RESULT OF THE INTEGER DIVISION OF the NUMERIC CODE OF c2 MINUS the NUMERIC CODE OF the LETTER A BY six  
            SET y2 TO the REMAINDER OF THE INTEGER DIVISION OF the NUMERIC CODE OF c2 MINUS the NUMERIC CODE OF the LETTER A BY six  
            RETURN the ABSOLUTE VALUE OF x1 MINUS x2 PLUS the ABSOLUTE VALUE OF y1 MINUS y2  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the LENGTH OF word THEN  
                RETURN zero  
            END IF  
            SET c TO the ELEMENT at position i of word  
            SET d1 TO the RESULT OF distance APPLIED TO f1 AND c PLUS the RESULT OF dp APPLIED TO i PLUS one AND c AND f2  
            SET d2 TO the RESULT OF distance APPLIED TO f2 AND c PLUS the RESULT OF dp APPLIED TO i PLUS one AND f1 AND c  
            RETURN the MINIMUM OF d1 AND d2  
        END FUNCTION  
        
        RETURN the RESULT OF dp APPLIED TO zero AND None AND None  
    END FUNCTION  
END CLASS
