CLASS Solution  
    FUNCTION numMusicPlaylists(n integer, goal integer, k integer) RETURNS integer  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven  
        
        FUNCTION create_zero_matrix(rows integer, columns integer) RETURNS two dimensional list  
            SET matrix TO empty list  
            FOR each row_index FROM one TO rows  
                SET new_row TO list of zeros with length columns  
                APPEND new_row TO matrix  
            END FOR  
            RETURN matrix  
        END FUNCTION  
        
        SET dp TO create_zero_matrix(goal PLUS one, n PLUS one)  
        SET element at position zero of list at position zero of dp TO one  
        
        FOR i FROM one TO goal  
            FOR j FROM one TO n  
                SET element at position j of list at position i of dp TO element at position j of list at position i of dp PLUS element at position j MINUS one of list at position i MINUS one of dp MULTIPLIED BY value of n MINUS j MINUS one PLUS one  
                SET element at position j of list at position i of dp TO element at position j of list at position i of dp MODULO MOD  
                
                IF j GREATER THAN k THEN  
                    SET element at position j of list at position i of dp TO element at position j of list at position i of dp PLUS element at position j of list at position i MINUS one of dp MULTIPLIED BY j MINUS k  
                    SET element at position j of list at position i of dp TO element at position j of list at position i of dp MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of list at position goal of dp  
    END FUNCTION  
END CLASS
