CLASS Solution  
    FUNCTION restoreIpAddresses WITH PARAMETER s  
        FUNCTION is_valid WITH PARAMETER segment  
            IF the LENGTH OF segment EQUALS one OR the FIRST CHARACTER of segment NOT EQUAL TO zero AND THE INTEGER VALUE OF segment LESS THAN OR EQUAL TO two hundred fifty five  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  
        
        FUNCTION backtrack WITH PARAMETERS start AND path  
            IF the LENGTH OF path EQUALS four  
                IF start EQUALS the LENGTH OF s  
                    APPEND the CONCATENATION OF elements in path SEPARATED BY period TO result  
                END IF  
                RETURN  
            END IF  
            FOR length FROM one TO three  
                IF start PLUS length LESS THAN OR EQUAL TO the LENGTH OF s  
                    SET segment TO the SUBSTRING from position start TO start PLUS length MINUS one of s  
                    IF is_valid WITH ARGUMENT segment RETURNS true  
                        CALL backtrack WITH ARGUMENTS start PLUS length AND the CONCATENATION OF path AND a LIST CONTAINING segment  
                    END IF  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET result TO an empty list  
        CALL backtrack WITH ARGUMENTS zero AND an empty list  
        RETURN result  
    END FUNCTION  
END CLASS
