CLASS Solution  
    FUNCTION topKFrequent(words AS List of string, k AS integer) RETURNS List of string  
        SET count TO the frequency count of each word in words  
        SET heap TO create a list by FOR each word and its associated frequency in count items LISTING a tuple of NEGATIVE frequency paired WITH the word  
        CALL heapify on heap TO convert the list into a heap structure  
        SET result TO an empty list  
        FOR each number FROM one TO k  
            SET element TO the result of popping the smallest item from heap  
            APPEND the second element of element to result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
