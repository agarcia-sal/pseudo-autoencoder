CLASS Solution  
    FUNCTION spiralMatrixIII(rows, cols, rStart, cStart)  
        SET list_of_direction_pairs TO list of pairs zero and one, one and zero, zero and negative one, negative one and zero  
        SET current_direction_index TO zero  
        SET current_step_count TO zero  
        SET increment_counter TO zero  
        SET list_of_visited_positions TO empty list  
        SET current_x TO rStart  
        SET current_y TO cStart  
        
        WHILE the LENGTH OF list_of_visited_positions IS LESS THAN rows MULTIPLIED BY cols  
            IF increment_counter MODULO two EQUALS zero  
                INCREMENT current_step_count BY one  
            END IF  
            FOR each iteration FROM one TO current_step_count  
                IF current_x IS GREATER THAN OR EQUAL TO zero AND current_x IS LESS THAN rows AND current_y IS GREATER THAN OR EQUAL TO zero AND current_y IS LESS THAN cols  
                    APPEND list of current_x and current_y TO list_of_visited_positions  
                END IF  
                SET change_in_x TO element at position current_direction_index of list_of_direction_pairs at position zero  
                SET change_in_y TO element at position current_direction_index of list_of_direction_pairs at position one  
                SET current_x TO current_x PLUS change_in_x  
                SET current_y TO current_y PLUS change_in_y  
            END FOR  
            SET current_direction_index TO the REMAINDER of current_direction_index PLUS one DIVIDED BY four  
            INCREMENT increment_counter BY one  
        END WHILE  
        
        RETURN list_of_visited_positions  
    END FUNCTION  
END CLASS
