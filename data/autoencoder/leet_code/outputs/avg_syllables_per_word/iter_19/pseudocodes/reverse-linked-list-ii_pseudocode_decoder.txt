CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the val property TO val  
        SET the next property TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values OF TYPE list  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO a new ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements from position one TO THE END of values  
        SET node TO a new ListNode initialized with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN whether the val property of p1 EQUALS the val property of p2 AND the result of calling is_same_list WITH the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween WITH PARAMETERS head OF TYPE Optional OF ListNode AND left OF TYPE integer AND right OF TYPE integer RETURNS Optional OF ListNode  
        IF head IS none OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO a new ListNode initialized with zero  
        SET the next property of dummy TO head  
        SET prev TO dummy  
        FOR each number FROM one TO left MINUS one  
            SET prev TO the next property of prev  
        END FOR  
        SET current TO the next property of prev  
        FOR each number FROM one TO right MINUS left  
            SET next_node TO the next property of current  
            SET the next property of current TO the next property of next_node  
            SET the next property of next_node TO the next property of prev  
            SET the next property of prev TO next_node  
        END FOR  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
