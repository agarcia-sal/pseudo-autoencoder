CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO CALL Counter WITH s  
        SET max_heap TO EMPTY LIST  
        FOR each character AND frequency IN the ITEMS OF count  
            APPEND TUPLE OF NEGATIVE frequency AND character TO max_heap  
        END FOR  
        CALL heapify ON max_heap  
        SET prev_char TO None  
        SET prev_freq TO zero  
        SET result TO EMPTY LIST  
        WHILE max_heap IS NOT EMPTY OR prev_freq IS LESS THAN zero  
            IF max_heap IS EMPTY AND prev_freq IS LESS THAN zero  
                RETURN EMPTY STRING  
            END IF  
            SET tuple_of_frequency_and_character TO CALL heappop ON max_heap  
            SET freq TO the FIRST ELEMENT OF tuple_of_frequency_and_character  
            SET char TO the SECOND ELEMENT OF tuple_of_frequency_and_character  
            APPEND char TO result  
            INCREMENT freq BY one  
            IF prev_freq IS LESS THAN zero  
                CALL heappush ON max_heap WITH TUPLE OF prev_freq AND prev_char  
            END IF  
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        RETURN THE CONCATENATION OF ALL ELEMENTS OF result  
    END FUNCTION  
END CLASS
