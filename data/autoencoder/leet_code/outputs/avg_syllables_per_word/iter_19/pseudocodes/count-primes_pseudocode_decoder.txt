CLASS Solution  
    FUNCTION countPrimes WITH parameter n  
        IF n LESS THAN OR EQUAL TO two  
            RETURN zero  
        END IF  
        SET is_prime TO a list of boolean true values with the LENGTH OF n  
        SET element at position zero of is_prime TO boolean false  
        SET element at position one of is_prime TO boolean false  
        FOR each number start FROM two TO the INTEGER VALUE OF the SQUARE ROOT OF n PLUS one  
            IF element at position start of is_prime EQUALS boolean true  
                FOR each number multiple FROM start MULTIPLIED BY start TO n MINUS one WITH STEP start  
                    SET element at position multiple of is_prime TO boolean false  
                END FOR  
            END IF  
        END FOR  
        RETURN the SUM OF all elements of is_prime  
    END FUNCTION  
END CLASS
