CLASS Solution  
    FUNCTION topKFrequent(nums AS List OF integer, k AS integer) AS List OF integer  
        SET frequency_map TO a new default dictionary with integer default values  
        FOR each num IN nums  
            INCREMENT the value at key num IN frequency_map BY one  
        END FOR  
        SET max_frequency TO the maximum value among all values in frequency_map  
        SET buckets TO a new list of empty lists each representing a frequency from zero TO max_frequency  
        FOR each num and freq IN all key value pairs of frequency_map  
            APPEND num TO the list at position freq in buckets  
        END FOR  
        SET result TO an empty list  
        FOR freq FROM max_frequency DOWN TO one  
            IF the length of the list at position freq in buckets IS GREATER THAN zero  
                EXTEND result BY all elements in the list at position freq in buckets  
                IF the length of result IS GREATER THAN OR EQUAL TO k  
                    RETURN the first k elements of result  
                END IF  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
