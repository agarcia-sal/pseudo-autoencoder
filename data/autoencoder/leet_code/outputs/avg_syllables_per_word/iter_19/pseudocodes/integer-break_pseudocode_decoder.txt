CLASS Solution  
    FUNCTION integerBreak WITH parameter n  
        SET dp TO a list of zeros with LENGTH equal TO n PLUS one  
        SET element at position one of dp TO one  
        
        FOR i FROM two TO n  
            SET max_product TO zero  
            FOR j FROM one TO i MINUS one  
                SET product_one TO j MULTIPLIED BY element at position i MINUS j of dp  
                SET product_two TO j MULTIPLIED BY i MINUS j  
                SET max_product TO the maximum VALUE AMONG max_product AND product_one AND product_two  
            END FOR  
            SET element at position i of dp TO max_product  
        END FOR  
        
        RETURN element at position n of dp  
    END FUNCTION  
END CLASS
