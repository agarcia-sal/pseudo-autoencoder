CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        SET directions TO the LIST CONTAINING the TUPLES zero and one, one and zero, zero and negative one, negative one and zero  
        SET min_heap TO the LIST CONTAINING a TUPLE of zero zero zero representing cost x and y  
        SET visited TO the SET containing the TUPLE zero zero  
        
        WHILE min_heap is not empty  
            REMOVE and RETURN the SMALLEST element from min_heap ASSIGN TO cost x and y  
            
            IF the TUPLE of x and y EQUALS the TUPLE of m MINUS one and n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND the TUPLE of nx and ny IS NOT IN visited  
                    ADD the TUPLE of nx and ny TO visited  
                    SET new_cost TO cost PLUS the element at position nx of grid at position ny  
                    ADD the TUPLE of new_cost nx and ny TO min_heap PRESERVING HEAP PROPERTY  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
