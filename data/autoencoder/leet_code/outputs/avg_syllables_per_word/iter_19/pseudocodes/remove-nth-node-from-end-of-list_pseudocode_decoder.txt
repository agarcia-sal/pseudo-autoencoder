CLASS ListNode  
    FUNCTION __init__ WITH parameters val EQUALS zero AND next EQUALS none  
        SET the val property OF this object TO val  
        SET the next property OF this object TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH parameter values AS a list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode initialized WITH element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO the end  
        SET node TO a new ListNode initialized WITH val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH parameters p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN the expression that the val property of p1 EQUALS the val property of p2 AND the result of is_same_list with parameters the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd WITH parameters head OF type optional ListNode AND n OF type integer RETURNING optional ListNode  
        SET dummy TO a new ListNode initialized WITH zero  
        SET the next property of dummy TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n INCLUSIVE  
            SET first TO the next property of first  
        END FOR  
        WHILE first IS NOT none  
            SET first TO the next property of first  
            SET second TO the next property of second  
        END WHILE  
        SET the next property of second TO the next property of the next property of second  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
