CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO an empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS the plus symbol  
                        APPEND l PLUS r TO results  
                    ELSE IF operator EQUALS the minus symbol  
                        APPEND l MINUS r TO results  
                    ELSE IF operator EQUALS the multiplication symbol  
                        APPEND l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  

        FUNCTION helper(sub_expr)  
            IF every character in sub_expr IS a digit  
                RETURN a list containing the integer value of sub_expr  
            END IF  
            SET results TO an empty list  
            FOR index FROM zero TO the length of sub_expr MINUS one  
                IF the character at position index of sub_expr IS one of the plus symbol OR the minus symbol OR the multiplication symbol  
                    SET left_results TO helper applied to the substring from the start TO index MINUS one of sub_expr  
                    SET right_results TO helper applied to the substring from index PLUS one TO the end of sub_expr  
                    EXTEND results BY compute applied to left_results, right_results, and the character at position index of sub_expr  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  

        RETURN helper applied to expression  
    END FUNCTION  
END CLASS
