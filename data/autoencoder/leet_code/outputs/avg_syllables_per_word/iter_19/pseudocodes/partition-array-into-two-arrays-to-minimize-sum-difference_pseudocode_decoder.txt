CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY two  
        SET total_sum TO the SUM OF all elements IN nums  
        SET target TO total_sum DIVIDED BY two  
        SET min_diff TO infinite value  
        
        SET left_half TO the ELEMENTS FROM POSITION zero TO n MINUS one OF nums  
        SET right_half TO the ELEMENTS FROM POSITION n TO the END OF nums  
        
        SET left_sums TO a list of empty dictionaries WITH SIZE n PLUS one  
        SET right_sums TO a list of empty dictionaries WITH SIZE n PLUS one  
        
        FOR each i FROM zero TO n  
            FOR each combo IN all combinations OF left_half TAKEN i AT A TIME  
                SET subset_sum TO the SUM OF all elements IN combo  
                IF subset_sum NOT PRESENT AS key IN left_sums AT position i  
                    SET left_sums AT position i AT key subset_sum TO zero  
                END IF  
                INCREMENT left_sums AT position i AT key subset_sum BY one  
            END FOR  
            
            FOR each combo IN all combinations OF right_half TAKEN i AT A TIME  
                SET subset_sum TO the SUM OF all elements IN combo  
                IF subset_sum NOT PRESENT AS key IN right_sums AT position i  
                    SET right_sums AT position i AT key subset_sum TO zero  
                END IF  
                INCREMENT right_sums AT position i AT key subset_sum BY one  
            END FOR  
        END FOR  
        
        FOR each i FROM zero TO n  
            SET left_values TO the SORTED list OF keys FROM left_sums AT position i  
            SET right_values TO the SORTED list OF keys FROM right_sums AT position n MINUS i  
            
            SET j TO zero  
            SET k TO the LENGTH OF right_values MINUS one  
            
            WHILE j LESS THAN the LENGTH OF left_values AND k GREATER THAN OR EQUAL TO zero  
                SET current_sum TO element at position j OF left_values PLUS element at position k OF right_values  
                SET candidate_diff TO total_sum MINUS two MULTIPLIED BY current_sum  
                IF candidate_diff IS LESS THAN zero  
                    SET candidate_diff TO the NEGATIVE OF candidate_diff  
                END IF  
                IF candidate_diff IS LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                END IF  
                
                IF current_sum IS LESS THAN target  
                    INCREMENT j BY one  
                ELSE IF current_sum IS GREATER THAN target  
                    DECREMENT k BY one  
                ELSE  
                    RETURN zero  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
