CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the val ATTRIBUTE OF self TO val  
        SET the next ATTRIBUTE OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A NEW ListNode WITH val AS the ELEMENT at POSITION zero OF values  
    SET p TO head  
    FOR each val IN THE SLICE OF values FROM POSITION one TO THE END  
        SET node TO A NEW ListNode WITH val AS val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS NONE AND p2 IS NONE  
        RETURN true  
    END IF  
    IF p1 IS NONE OR p2 IS NONE  
        RETURN false  
    END IF  
    RETURN THE BOOLEAN RESULT OF THE EXPRESSION THAT THE val ATTRIBUTE OF p1 EQUALS THE val ATTRIBUTE OF p2 AND THE RESULT OF A RECURSIVE CALL TO is_same_list WITH THE next ATTRIBUTE OF p1 AND THE next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reorderList WITH PARAMETERS self AND head AS AN OPTIONAL ListNode  
        IF head IS NONE OR the next ATTRIBUTE OF head IS NONE  
            RETURN  
        END IF  

        SET slow TO head  
        SET fast TO head  
        WHILE fast IS NOT NONE AND the next ATTRIBUTE OF fast IS NOT NONE  
            SET slow TO the next ATTRIBUTE OF slow  
            SET fast TO the next ATTRIBUTE of the next ATTRIBUTE of fast  
        END WHILE  

        SET prev TO none  
        SET curr TO slow  
        WHILE curr IS NOT NONE  
            SET next_temp TO the next ATTRIBUTE OF curr  
            SET the next ATTRIBUTE OF curr TO prev  
            SET prev TO curr  
            SET curr TO next_temp  
        END WHILE  

        SET first TO head  
        SET second TO prev  
        WHILE the next ATTRIBUTE OF second IS NOT NONE  
            SET temp1 TO the next ATTRIBUTE OF first  
            SET temp2 TO the next ATTRIBUTE OF second  
            SET the next ATTRIBUTE OF first TO second  
            SET the next ATTRIBUTE OF second TO temp1  
            SET first TO temp1  
            SET second TO temp2  
        END WHILE  
    END FUNCTION  
END CLASS
