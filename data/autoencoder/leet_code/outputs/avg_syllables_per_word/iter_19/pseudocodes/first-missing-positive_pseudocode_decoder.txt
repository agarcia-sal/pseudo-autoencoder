CLASS Solution  
    FUNCTION firstMissingPositive(nums)  
        SET n TO the LENGTH OF nums  
        FOR index FROM zero TO n MINUS one  
            WHILE one IS LESS THAN OR EQUAL TO the element at position index of nums AND the element at position index of nums IS LESS THAN OR EQUAL TO n AND the element at position element at position index of nums MINUS one of nums IS NOT EQUAL TO the element at position index of nums  
                SET correct_index TO the element at position index of nums MINUS one  
                SWAP the element at position index of nums WITH the element at position correct_index of nums  
            END WHILE  
        END FOR  
        FOR index FROM zero TO n MINUS one  
            IF the element at position index of nums IS NOT EQUAL TO index PLUS one  
                RETURN index PLUS one  
            END IF  
        END FOR  
        RETURN n PLUS one  
    END FUNCTION  
END CLASS
