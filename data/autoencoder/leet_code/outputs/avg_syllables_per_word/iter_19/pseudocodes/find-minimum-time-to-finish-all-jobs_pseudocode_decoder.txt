CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        
        SET subset_sums TO a list of zero values of length two RAISED TO THE POWER OF n  
        
        FOR each i FROM zero TO n MINUS one  
            FOR each mask FROM zero TO two RAISED TO THE POWER OF i MINUS one  
                SET subset_sums at position the BITWISE OR OF mask AND two RAISED TO THE POWER OF i TO the sum of subset_sums at position mask PLUS the element at position i of jobs  
            END FOR  
        END FOR  
        
        FUNCTION dp(k, mask)  
            IF k EQUALS one THEN  
                RETURN subset_sums at position mask  
            END IF  
            
            SET res TO positive infinity  
            
            SET submask TO mask  
            WHILE submask IS NOT zero  
                SET candidate1 TO dp called with k MINUS one and the BITWISE XOR OF mask AND submask  
                SET candidate2 TO subset_sums at position submask  
                SET current_max TO the GREATER OF candidate1 AND candidate2  
                SET res TO the LESSER OF res AND current_max  
                SET submask TO the BITWISE AND OF submask MINUS one AND mask  
            END WHILE  
            
            RETURN res  
        END FUNCTION  
        
        RETURN dp called with arguments k and two RAISED TO THE POWER OF n MINUS one  
    END FUNCTION  
END CLASS
