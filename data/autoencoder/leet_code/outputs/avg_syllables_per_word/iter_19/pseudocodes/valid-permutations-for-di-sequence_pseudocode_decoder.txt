CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list of size n PLUS one BY n PLUS one with all elements zero  
        
        FOR each position j FROM zero TO n  
            SET element at position j of element at position zero of dp TO one  
        END FOR  
        
        FOR each position i FROM one TO n  
            SET prefix_sum TO zero  
            IF element at position i MINUS one of s EQUALS the character I THEN  
                FOR each position j FROM zero TO n MINUS i  
                    SET prefix_sum TO prefix_sum PLUS element at position j of element at position i MINUS one of dp MODULO MOD  
                    SET element at position j of element at position i of dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR each position j FROM n MINUS i DOWN TO zero  
                    SET prefix_sum TO prefix_sum PLUS element at position j PLUS one of element at position i MINUS one of dp MODULO MOD  
                    SET element at position j of element at position i of dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        
        SET result_sum TO zero  
        FOR each element value IN element at position n of dp  
            INCREMENT result_sum BY value  
        END FOR  
        
        RETURN result_sum MODULO MOD  
    END FUNCTION  
END CLASS
