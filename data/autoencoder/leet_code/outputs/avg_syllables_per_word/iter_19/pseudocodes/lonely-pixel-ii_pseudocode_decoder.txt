CLASS Solution  
    FUNCTION findBlackPixel(picture AS List OF List OF string, target AS integer) RETURNS integer  
        IF picture IS empty OR element at position zero of picture IS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero of picture  
        SET row_count TO list of zero with LENGTH m  
        SET col_count TO list of zero with LENGTH n  
        SET row_patterns TO a new map with default zero for integer values  
        
        FOR r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of all elements at each position in element at position r of picture  
            INCREMENT the value mapped by pattern in row_patterns BY one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS black pixel  
                    INCREMENT element at position r of row_count BY one  
                    INCREMENT element at position c of col_count BY one  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO zero  
        
        FOR r FROM zero TO m MINUS one  
            FOR c FROM zero TO n MINUS one  
                IF element at position c of element at position r of picture EQUALS black pixel  
                    SET pattern TO the concatenation of all elements at each position in element at position r of picture  
                    IF element at position r of row_count EQUALS target AND element at position c of col_count EQUALS target AND the value mapped by pattern in row_patterns EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
