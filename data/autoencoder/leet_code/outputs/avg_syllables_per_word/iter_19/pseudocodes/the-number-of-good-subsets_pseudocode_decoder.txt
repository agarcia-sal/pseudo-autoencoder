CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO the frequency count of each element in nums  
        SET dp TO a list of zeroes with length equal to two raised to the power of ten  
        SET the element at position zero of dp TO one  
        
        FOR each num IN count  
            IF num EQUALS one THEN  
                SET the element at position zero of dp TO the element at position zero of dp MULTIPLIED BY two raised to the power of the element at position one of count  
                APPLY modulo MOD to the element at position zero of dp  
                CONTINUE to the next iteration  
            END IF  
            
            IF num NOT IN prime_factors THEN  
                CONTINUE to the next iteration  
            END IF  
            
            SET factors TO the list of prime_factors corresponding to num  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask OR two raised to the power of prime MINUS one  
            END FOR  
            
            FOR m FROM two raised to the power of ten MINUS one DOWN TO zero  
                IF bitwise AND of m and mask EQUALS zero THEN  
                    SET the element at position bitwise OR of m and mask of dp TO the element at position bitwise OR of m and mask of dp PLUS the element at position m of dp MULTIPLIED BY the element at position num of count  
                    APPLY modulo MOD to the element at position bitwise OR of m and mask of dp  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO the sum of all elements of dp starting from position one APPLY modulo MOD  
        RETURN result  
    END FUNCTION  
END CLASS
