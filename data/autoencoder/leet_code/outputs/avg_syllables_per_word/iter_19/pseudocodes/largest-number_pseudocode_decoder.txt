CLASS Solution  
    FUNCTION largestNumber WITH PARAMETERS nums  
        SET nums_str TO list obtained by CONVERTING each element IN nums TO string  
        
        FUNCTION compare WITH PARAMETERS x y  
            IF concatenation of x PLUS y LEXICOGRAPHICALLY GREATER THAN concatenation of y PLUS x  
                RETURN negative one  
            ELSE IF concatenation of x PLUS y LEXICOGRAPHICALLY LESS THAN concatenation of y PLUS x  
                RETURN positive one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        CALL sort ON nums_str WITH sorting key DEFINED BY applying compare converted TO a key function  
        
        IF element at position zero of nums_str EQUALS string representing zero  
            RETURN string representing zero  
        END IF  
        
        SET joined_string TO concatenation of all elements in nums_str IN ORDER  
        RETURN joined_string  
    END FUNCTION  
END CLASS
