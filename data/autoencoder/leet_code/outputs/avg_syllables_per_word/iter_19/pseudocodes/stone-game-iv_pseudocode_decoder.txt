CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO a list of boolean False values with length equal to n PLUS one  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE the square of j is LESS THAN OR EQUAL TO i  
                IF the value at position i MINUS the square of j of dp list EQUALS boolean False  
                    SET the value at position i of dp list TO boolean True  
                    BREAK out of the current loop  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN the value at position n of dp list  
    END FUNCTION  
END CLASS
