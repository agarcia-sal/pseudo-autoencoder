CLASS Solution  
    FUNCTION findRepeatedDnaSequences WITH parameter s  
        SET sequence_count TO an empty dictionary with default integer zero  
        FOR index FROM zero TO the LENGTH OF s MINUS ten PLUS one  
            SET sequence TO substring from position index TO index PLUS ten MINUS one of string s  
            INCREMENT the value associated with sequence in sequence_count BY one  
        END FOR  
        SET repeated_sequences TO an empty list  
        FOR each key_value_pair IN the items of sequence_count  
            IF the value of the current key_value_pair IS GREATER THAN one  
                APPEND the key of the current key_value_pair TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
