CLASS Solution  
    FUNCTION generatePalindromes WITH PARAMETER s  
        DEFINE FUNCTION dfs WITH PARAMETER t  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND its count v IN cnt  
                IF v GREATER THAN one  
                    DECREMENT the count of character c IN cnt BY two  
                    CALL dfs WITH the STRING formed by character c CONCATENATED WITH t CONCATENATED WITH character c  
                    INCREMENT the count of character c IN cnt BY two  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET cnt TO the COUNTER mapping of characters in s  
        SET mid TO empty string  
        FOR each character c AND its count v IN cnt  
            IF v MODULO two EQUALS one  
                IF mid IS NOT empty string  
                    RETURN an empty list  
                END IF  
                SET mid TO character c  
                DECREMENT the count of character c IN cnt BY one  
            END IF  
        END FOR  
        SET ans TO empty list  
        CALL dfs WITH PARAMETER mid  
        RETURN ans  
    END FUNCTION  
END CLASS
