CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO the LIST OF characters of num  
        SET n TO the LENGTH OF num  
        
        WHILE k IS GREATER THAN zero  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF element at position i of num IS GREATER THAN element at position i PLUS one of num  
                    SET min_digit TO character nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO the MINIMUM BETWEEN i PLUS k PLUS one AND n MINUS one  
                        IF element at position j of num IS LESS THAN min_digit  
                            SET min_digit TO element at position j of num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    
                    SET a sublist from position i TO min_index IN num TO a list where first element is element at position min_index of num FOLLOWED BY elements from position i TO min_index MINUS one of num  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO true  
                    BREAK the for loop over i  
                END IF  
            END FOR  
            
            IF swapped IS false  
                BREAK the while loop  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements in num as a string  
    END FUNCTION  
END CLASS
