CLASS Solution  
    FUNCTION movesToChessboard WITH PARAMETER board  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR each i FROM zero TO N MINUS one  
            FOR each j FROM zero TO N MINUS one  
                IF the element at position i of board XOR the element at position zero of board at position zero XOR the element at position zero of board at position j XOR the element at position zero of board at position zero IS NOT EQUAL TO zero  
                    RETURN negative one  
                END IF  
            END FOR  
            INCREMENT rowSum BY the element at position i of board at position zero  
            INCREMENT colSum BY the element at position zero of board at position i  
            INCREMENT rowSwap BY WHETHER the element at position i of board at position zero EQUALS the remainder of i DIVIDED BY two  
            INCREMENT colSwap BY WHETHER the element at position zero of board at position i EQUALS the remainder of i DIVIDED BY two  
        END FOR  
        IF the integer division of N BY two IS GREATER THAN rowSum OR rowSum IS GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the integer division of N BY two IS GREATER THAN colSum OR colSum IS GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the remainder of N DIVIDED BY two IS NOT EQUAL TO zero  
            IF the remainder of rowSwap DIVIDED BY two IS NOT EQUAL TO zero  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of colSwap DIVIDED BY two IS NOT EQUAL TO zero  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the MINIMUM OF rowSwap AND N MINUS rowSwap  
            SET colSwap TO the MINIMUM OF colSwap AND N MINUS colSwap  
        END IF  
        RETURN the integer division of the sum of rowSwap AND colSwap BY two  
    END FUNCTION  
END CLASS
