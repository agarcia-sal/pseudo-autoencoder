CLASS Solution  
    FUNCTION asteroidCollision WITH asteroids  
        SET stack TO empty list  
        FOR each asteroid IN asteroids  
            WHILE the stack is not empty AND the asteroid is less than zero AND the element at the position of the last element of the stack is greater than zero  
                IF the element at the position of the last element of the stack is less than the negative of the asteroid  
                    REMOVE the last element FROM stack  
                    CONTINUE to the next iteration of the WHILE loop  
                ELSE IF the element at the position of the last element of the stack EQUALS the negative of the asteroid  
                    REMOVE the last element FROM stack  
                END IF  
                BREAK out of the WHILE loop  
            END WHILE  
            IF the previous WHILE loop did not execute a BREAK early  
                APPEND asteroid TO stack  
            END IF  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
