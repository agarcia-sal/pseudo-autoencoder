CLASS Solution  
    FUNCTION kthSmallestPath(destination AS List OF Integer, k AS Integer) RETURNS String  
        SET row TO element at position zero of destination  
        SET col TO element at position one of destination  
        SET path TO an empty list  
        SET total_moves TO col PLUS row  
        FOR index FROM zero TO total_moves MINUS one  
            IF col IS GREATER THAN zero  
                SET paths_with_H TO the number of combinations of row PLUS col MINUS one TAKEN col MINUS one at a time  
                IF k IS LESS THAN OR EQUAL TO paths_with_H  
                    APPEND the move H TO path  
                    DECREMENT col BY one  
                ELSE  
                    APPEND the move V TO path  
                    DECREMENT row BY one  
                    DECREMENT k BY paths_with_H  
                END IF  
            ELSE  
                APPEND the move V TO path  
                DECREMENT row BY one  
            END IF  
        END FOR  
        RETURN the concatenation of all elements in path as a single string  
    END FUNCTION  
END CLASS
