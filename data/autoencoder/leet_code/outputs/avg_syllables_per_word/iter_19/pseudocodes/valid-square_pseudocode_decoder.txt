CLASS Solution  
    FUNCTION validSquare(p1 AS List OF integer, p2 AS List OF integer, p3 AS List OF integer, p4 AS List OF integer) RETURNS boolean  
        FUNCTION squared_distance(p1 AS List OF integer, p2 AS List OF integer) RETURNS integer  
            SET horizontal_difference TO element at position zero of p1 MINUS element at position zero of p2  
            SET vertical_difference TO element at position one of p1 MINUS element at position one of p2  
            SET horizontal_difference_squared TO horizontal_difference MULTIPLIED BY horizontal_difference  
            SET vertical_difference_squared TO vertical_difference MULTIPLIED BY vertical_difference  
            SET total_squared_distance TO horizontal_difference_squared PLUS vertical_difference_squared  
            RETURN total_squared_distance  
        END FUNCTION  
        
        SET points TO a list containing p1 and p2 and p3 and p4  
        SET distances TO an empty set  
        
        FOR index_i FROM zero TO three  
            FOR index_j FROM index_i PLUS one TO three  
                SET current_distance TO squared_distance(element at position index_i of points, element at position index_j of points)  
                IF current_distance EQUALS zero THEN  
                    RETURN false  
                END IF  
                ADD current_distance TO distances  
            END FOR  
        END FOR  
        
        IF the size of distances EQUALS two THEN  
            RETURN true  
        ELSE  
            RETURN false  
        END IF  
    END FUNCTION  
END CLASS
