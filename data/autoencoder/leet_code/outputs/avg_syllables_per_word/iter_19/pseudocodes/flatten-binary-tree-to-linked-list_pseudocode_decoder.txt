CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values is empty THEN  
        RETURN none  
    END IF  
    SET root TO new instance of TreeNode initialized with the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        REMOVE the first element FROM queue AND ASSIGN pointer TO node  
        IF i is less than the length of values AND the element at position i of values is not none THEN  
            SET node.left TO new instance of TreeNode initialized with the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the length of values AND the element at position i of values is not none THEN  
            SET node.right TO new instance of TreeNode initialized with the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p is none AND q is none THEN  
        RETURN true  
    ELSE IF p is none OR q is none THEN  
        RETURN false  
    ELSE IF the value of p is not equal to the value of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of calling is_same_tree with the left child of p and the left child of q AND the result of calling is_same_tree with the right child of p and the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(self, root)  
        IF root is none THEN  
            RETURN  
        END IF  
        SET stack TO a list containing root as its only element  
        WHILE stack is not empty  
            REMOVE the last element FROM stack AND ASSIGN pointer TO node  
            IF the right child of node is not none THEN  
                APPEND the right child of node TO stack  
            END IF  
            IF the left child of node is not none THEN  
                APPEND the left child of node TO stack  
            END IF  
            SET the left child of node TO none  
            IF stack is not empty THEN  
                SET the right child of node TO the last element of stack  
            ELSE  
                SET the right child of node TO none  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
