CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO the LENGTH OF floor  
        SET dp TO a two dimensional list with n PLUS one rows AND numCarpets PLUS one columns FILLED WITH zero  
        
        FOR i FROM one TO n  
            SET dp at position i zero TO dp at position i MINUS one zero PLUS one IF element at position i MINUS one of floor EQUALS one ELSE zero  
        END FOR  
        
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp at position i j TO dp at position i MINUS one j PLUS one IF element at position i MINUS one of floor EQUALS one ELSE zero  
                SET dp at position i j TO the MINIMUM OF dp at position i j AND dp at position MAXIMUM OF zero AND i MINUS carpetLen j MINUS one  
            END FOR  
        END FOR  
        
        RETURN dp at position n numCarpets  
    END FUNCTION  
END CLASS
