CLASS Solution  
    FUNCTION findNumberOfLIS(nums)  
        IF nums IS empty  
            RETURN zero  
        END IF  
          
        SET n TO the LENGTH OF nums  
        SET lengths TO a list of n elements each set to one representing the LENGTH OF the longest increasing subsequence ENDING AT each index  
        SET counts TO a list of n elements each set to one representing the NUMBER OF the longest increasing subsequences ENDING AT each index  
          
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO i MINUS one  
                IF element at position i of nums IS GREATER THAN element at position j of nums  
                    IF element at position j of lengths PLUS one IS GREATER THAN element at position i of lengths  
                        SET element at position i of lengths TO element at position j of lengths PLUS one  
                        SET element at position i of counts TO element at position j of counts  
                    ELSE IF element at position j of lengths PLUS one EQUALS element at position i of lengths  
                        INCREMENT element at position i of counts BY element at position j of counts  
                    END IF  
                END IF  
            END FOR  
        END FOR  
          
        SET longest TO the MAXIMUM value IN lengths  
        SET total_count TO zero  
        FOR each i FROM zero TO n MINUS one  
            IF element at position i of lengths EQUALS longest  
                INCREMENT total_count BY element at position i of counts  
            END IF  
        END FOR  
        RETURN total_count  
    END FUNCTION  
END CLASS
