CLASS Solution  
    FUNCTION isTransformable(s, t)  
        SET pos TO a new mapping from digits to lists of positions where each list is maintained as a heap  
        FOR each index i AND character c IN s  
            PUSH the index i INTO the heap corresponding to the digit obtained by converting character c to an integer in pos  
        END FOR  
        FOR each index i AND character c IN t  
            SET digit TO the integer value of character c  
            IF the heap corresponding to digit in pos is empty  
                RETURN false  
            END IF  
            FOR each smaller digit FROM zero TO the digit MINUS one  
                IF the heap corresponding to smaller digit in pos is not empty AND the smallest element at the front of this heap is LESS THAN the smallest element at the front of the heap corresponding to digit  
                    RETURN false  
                END IF  
            END FOR  
            REMOVE the smallest element FROM the heap corresponding to digit in pos  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
