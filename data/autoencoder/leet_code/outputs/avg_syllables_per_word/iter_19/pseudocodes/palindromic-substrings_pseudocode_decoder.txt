CLASS Solution  
    FUNCTION countSubstrings WITH parameter s  
        FUNCTION expand_around_center WITH parameters left AND right  
            SET count TO zero  
            WHILE left GREATER THAN OR EQUAL TO zero AND right LESS THAN the LENGTH OF s AND the character at position left of s EQUALS the character at position right of s holds  
                INCREMENT count BY one  
                DECREMENT left BY one  
                INCREMENT right BY one  
            END WHILE  
            RETURN count  
        END FUNCTION  
      
        SET total_palindromes TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT total_palindromes BY expand_around_center CALLED WITH arguments index AND index  
            INCREMENT total_palindromes BY expand_around_center CALLED WITH arguments index AND index PLUS one  
        END FOR  
        RETURN total_palindromes  
    END FUNCTION  
END CLASS
