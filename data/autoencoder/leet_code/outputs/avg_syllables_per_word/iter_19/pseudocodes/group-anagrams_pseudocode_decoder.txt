CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO a new default dictionary with default value as empty list  
        FOR each string s IN strs  
            SET sorted_characters TO the result of sorting the characters of string s in ascending order  
            SET sorted_str TO the result of concatenating all characters in sorted_characters into a single string  
            APPEND string s TO the list corresponding to key sorted_str in anagrams  
        END FOR  
        SET result_list TO the list of all values contained in anagrams  
        RETURN result_list  
    END FUNCTION  
END CLASS
