CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET head TO NEW ListNode WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values from position one TO the end  
        SET node TO NEW ListNode WITH val  
        SET the next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the value of p1 EQUALS the value of p2 AND is_same_list(the next element of p1, the next element of p2)  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO NEW TreeNode WITH the element at position zero of values  
    SET i TO one  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        REMOVE the leftmost element FROM queue AND ASSIGN TO node  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET the left child of node TO NEW TreeNode WITH the element at position i of values  
            APPEND the left child of node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET the right child of node TO NEW TreeNode WITH the element at position i of values  
            APPEND the right child of node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value of p IS NOT EQUAL TO the value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index IS GREATER THAN OR EQUAL TO the LENGTH OF s  
                RETURN None AND index  
            END IF  
            SET start TO index  
            WHILE index IS LESS THAN the LENGTH OF s AND ((the element at position index of s IS a digit) OR (the element at position index of s IS a dash))  
                INCREMENT index BY one  
            END WHILE  
            SET node TO NEW TreeNode WITH the integer value of the substring of s from position start TO index MINUS one  
            IF index IS LESS THAN the LENGTH OF s AND the element at position index of s IS an opening parenthesis  
                SET node.left AND index TO parse_tree(index PLUS one)  
            END IF  
            IF index IS LESS THAN the LENGTH OF s AND the element at position index of s IS an opening parenthesis  
                SET node.right AND index TO parse_tree(index PLUS one)  
            END IF  
            IF index IS LESS THAN the LENGTH OF s AND the element at position index of s IS a closing parenthesis  
                INCREMENT index BY one  
            END IF  
            RETURN node AND index  
        END FUNCTION  
        SET root AND dummy_index TO parse_tree(zero)  
        RETURN root  
    END FUNCTION  
END CLASS
