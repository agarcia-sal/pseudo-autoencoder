CLASS Solution  
    FUNCTION jobScheduling(startTime AS List of integer, endTime AS List of integer, profit AS List of integer) RETURNS integer  
        SET jobs TO a list of tuples each containing elements at the same positions of startTime endTime and profit sorted by the second element of each tuple in ascending order  
        SET end_times TO a list containing the second element of each tuple in jobs  
        SET dp TO a list of zeros of length equal to the LENGTH OF jobs PLUS one  
        FOR i FROM one TO the LENGTH OF jobs  
            SET j TO the position where the value at the first element of the tuple at position i MINUS one of jobs should be inserted in end_times to maintain ascending order  
            SET dp at position i TO the greater value between dp at position i MINUS one AND dp at position j PLUS the third element of the tuple at position i MINUS one of jobs  
        END FOR  
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
