CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the val ATTRIBUTE of this object TO val  
        SET the next ATTRIBUTE of this object TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO A new ListNode CREATED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values STARTING FROM position one TO the end  
        SET node TO A new ListNode CREATED WITH val  
        SET the next ATTRIBUTE of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the value ATTRIBUTE of p1 EQUALS the value ATTRIBUTE of p2 AND is_same_list WITH ARGUMENTS the next ATTRIBUTE of p1 AND the next ATTRIBUTE of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList WITH PARAMETERS self AND head  
        IF head IS none OR the next ATTRIBUTE of head IS none  
            RETURN head  
        END IF  

        FUNCTION split WITH PARAMETER head  
            SET slow TO head  
            SET fast TO the next ATTRIBUTE of head  
            WHILE fast IS not none AND the next ATTRIBUTE of fast IS not none  
                SET slow TO the next ATTRIBUTE of slow  
                SET fast TO the next ATTRIBUTE of the next ATTRIBUTE of fast  
            END WHILE  
            SET mid TO the next ATTRIBUTE of slow  
            SET the next ATTRIBUTE of slow TO none  
            RETURN head AND mid  
        END FUNCTION  

        FUNCTION merge WITH PARAMETERS l1 AND l2  
            SET dummy TO A new ListNode  
            SET current TO dummy  
            WHILE l1 IS not none AND l2 IS not none  
                IF the value ATTRIBUTE of l1 IS LESS THAN the value ATTRIBUTE of l2  
                    SET the next ATTRIBUTE of current TO l1  
                    SET l1 TO the next ATTRIBUTE of l1  
                ELSE  
                    SET the next ATTRIBUTE of current TO l2  
                    SET l2 TO the next ATTRIBUTE of l2  
                END IF  
                SET current TO the next ATTRIBUTE of current  
            END WHILE  
            IF l1 IS not none  
                SET the next ATTRIBUTE of current TO l1  
            ELSE  
                SET the next ATTRIBUTE of current TO l2  
            END IF  
            RETURN the next ATTRIBUTE of dummy  
        END FUNCTION  

        SET left AND right TO split WITH ARGUMENT head  
        SET left TO sortList WITH ARGUMENT left  
        SET right TO sortList WITH ARGUMENT right  
        RETURN merge WITH ARGUMENTS left AND right  
    END FUNCTION  
END CLASS
