CLASS Solution  
    FUNCTION minPatches(nums AS List OF integers, n AS integer) RETURNS integer  
        SET missing TO one  
        SET patches TO zero  
        SET i TO zero  
        WHILE missing IS LESS THAN OR EQUAL TO n  
            IF i IS LESS THAN the LENGTH OF nums AND the element at position i of nums IS LESS THAN OR EQUAL TO missing  
                SET missing TO missing PLUS the element at position i of nums  
                INCREMENT i BY one  
            ELSE  
                SET missing TO missing PLUS missing  
                INCREMENT patches BY one  
            END IF  
        END WHILE  
        RETURN patches  
    END FUNCTION  
END CLASS
