CLASS Solution  
    FUNCTION replaceWords WITH PARAMETERS dictionary AND sentence  
        SORT dictionary BY THE LENGTH OF each root FROM SHORTEST TO LONGEST  
        SET words TO THE LIST OF WORDS OBTAINED BY SPLITTING sentence BY SPACES  
        
        FUNCTION replace WITH PARAMETER word  
            FOR each root IN dictionary  
                IF word BEGINS WITH root THEN  
                    RETURN root  
                END IF  
            END FOR  
            RETURN word  
        END FUNCTION  
        
        SET replaced_words TO AN EMPTY LIST  
        FOR each word IN words  
            APPEND THE RESULT OF FUNCTION replace CALLED WITH word TO replaced_words  
        END FOR  
        
        SET result_sentence TO THE JOINING OF replaced_words WITH SPACES BETWEEN EACH  
        RETURN result_sentence  
    END FUNCTION  
END CLASS
