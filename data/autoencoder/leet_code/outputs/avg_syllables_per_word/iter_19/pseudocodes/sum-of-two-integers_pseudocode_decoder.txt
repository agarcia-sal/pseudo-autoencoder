CLASS Solution  
    FUNCTION getSum(a, b)  
        SET MAX TO the maximum value of a thirty two bit integer  
        SET mask TO the mask value for thirty two bits  
        
        WHILE b DOES NOT EQUAL zero  
            SET carry TO the bitwise AND of a AND b THEN SHIFTED LEFT BY one position  
            SET a TO the bitwise exclusive OR of a AND b THEN bitwise AND with mask  
            SET b TO the bitwise AND of carry AND mask  
        END WHILE  
        
        IF a IS LESS THAN OR EQUAL TO MAX  
            RETURN a  
        ELSE  
            RETURN the bitwise complement of the bitwise exclusive OR of a AND mask  
        END IF  
    END FUNCTION  
END CLASS
