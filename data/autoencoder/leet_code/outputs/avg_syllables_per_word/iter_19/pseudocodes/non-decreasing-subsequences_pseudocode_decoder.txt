CLASS Solution  
    FUNCTION findSubsequences WITH PARAMETER nums AS list of integers RETURNS list of lists of integers  
        FUNCTION dfs WITH PARAMETERS u AS integer last AS integer t AS list of integers  
            IF u EQUALS the LENGTH OF nums THEN  
                IF the LENGTH OF t GREATER THAN one THEN  
                    APPEND a copy of t TO ans  
                END IF  
                RETURN  
            END IF  
            IF the element at position u of nums GREATER THAN OR EQUAL TO last THEN  
                APPEND the element at position u of nums TO t  
                CALL dfs WITH ARGUMENTS u PLUS one last SET TO the element at position u of nums AND t  
                REMOVE the last element FROM t  
            END IF  
            IF the element at position u of nums NOT EQUALS last THEN  
                CALL dfs WITH ARGUMENTS u PLUS one last AND t  
            END IF  
        END FUNCTION  
        SET ans TO an empty list  
        CALL dfs WITH ARGUMENTS zero MINUS one thousand AND an empty list  
        RETURN ans  
    END FUNCTION  
END CLASS
