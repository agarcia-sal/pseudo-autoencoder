CLASS Solution  
    FUNCTION getPermutation WITH PARAMETERS n k  
        SET numbers TO a list containing all integers FROM one TO n inclusive  
        DECREMENT k BY one  
        SET result TO an empty list  
        FOR i FROM n DOWN TO one  
            SET fact TO the factorial of i MINUS one  
            SET index TO the integer division of k BY fact  
            APPEND the string representation of the element at position index of numbers TO result  
            REMOVE the element at position index FROM numbers  
            SET k TO the remainder of k DIVIDED BY fact  
        END FOR  
        SET joined_result TO the concatenation of all elements in result as a string  
        RETURN joined_result  
    END FUNCTION  
END CLASS
