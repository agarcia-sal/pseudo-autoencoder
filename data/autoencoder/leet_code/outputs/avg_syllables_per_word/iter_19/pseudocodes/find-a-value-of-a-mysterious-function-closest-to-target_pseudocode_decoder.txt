CLASS Solution  
    FUNCTION closestToTarget(arr list OF integers, target integer) RETURNS integer  
        SET seen TO empty set of integers  
        SET min_diff TO positive infinity  
        FOR each num IN arr  
            SET updated_set TO empty set of integers  
            FOR each x IN seen  
                ADD the result of BITWISE AND operation between num AND x TO updated_set  
            END FOR  
            SET seen TO the union of updated_set AND the set containing num  
            FOR each val IN seen  
                SET absolute_difference TO the absolute value of val MINUS target  
                IF absolute_difference LESS THAN min_diff THEN  
                    SET min_diff TO absolute_difference  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
