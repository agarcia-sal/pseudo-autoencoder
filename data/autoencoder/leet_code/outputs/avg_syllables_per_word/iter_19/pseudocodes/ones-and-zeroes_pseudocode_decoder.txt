CLASS Solution  
    FUNCTION findMaxForm(strs AS List OF String, m AS Integer, n AS Integer) RETURNS Integer  
        SET dp TO a two dimensional list of zeros with number of rows equal TO m PLUS one AND number of columns equal TO n PLUS one  
        FOR each string s IN strs  
            SET count_zero TO the number of characters in s EQUALS zero  
            SET count_one TO the number of characters in s EQUALS one  
            FOR i FROM m DOWN TO count_zero  
                FOR j FROM n DOWN TO count_one  
                    SET dp element at position i and j TO the greater of dp element at position i and j AND dp element at position i MINUS count_zero and j MINUS count_one PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position m and n  
    END FUNCTION  
END CLASS
