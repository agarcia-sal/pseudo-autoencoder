CLASS Solution  
    FUNCTION intToRoman(num)  
        SET val TO list of decimal integer values one thousand nine hundred ninety four hundred one hundred ninety fifty forty ten nine five four one in descending order  
        SET syms TO list of Roman numeral strings em ce em dee ce ex ce el ex ix vee iv aye in order corresponding to val  
        SET roman_numeral TO empty string  
        SET i TO zero  
        WHILE num GREATER THAN zero holds  
            SET count TO num DIVIDED BY element at position i of val using integer division  
            FOR each _ FROM one TO count  
                APPEND element at position i of syms TO roman_numeral  
                DECREMENT num BY element at position i of val  
            END FOR  
            INCREMENT i BY one  
        END WHILE  
        RETURN roman_numeral  
    END FUNCTION  
END CLASS
