CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS empty OR THE LENGTH OF stones IS LESS THAN two  
            RETURN False  
        END IF  
        SET stone_set TO CONVERT stones TO A SET FOR CONSTANT TIME LOOKUPS  
        SET memo TO AN EMPTY DICTIONARY FOR REMEMBERING VISITED STATES  
        FUNCTION can_jump_to(position, jump)  
            IF position IS EQUAL TO THE LAST ELEMENT OF stones  
                RETURN True  
            END IF  
            IF THE PAIR OF position AND jump IS IN memo  
                RETURN THE VALUE ASSOCIATED WITH THE PAIR OF position AND jump IN memo  
            END IF  
            FOR next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump IS GREATER THAN ZERO AND THE SUM OF position AND next_jump IS IN stone_set  
                    IF can_jump_to(the SUM OF position AND next_jump, next_jump)  
                        SET THE VALUE ASSOCIATED WITH THE PAIR OF position AND jump IN memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            SET THE VALUE ASSOCIATED WITH THE PAIR OF position AND jump IN memo TO False  
            RETURN False  
        END FUNCTION  
        RETURN can_jump_to(the FIRST ELEMENT OF stones, zero)  
    END FUNCTION  
END CLASS
