CLASS Solution  
    FUNCTION wallsAndGates(rooms)  
        IF rooms IS EMPTY OR the LENGTH OF element at position zero of rooms EQUALS zero THEN  
            RETURN  
        END IF  
        SET rows TO the LENGTH OF rooms  
        SET cols TO the LENGTH OF element at position zero of rooms  
        ASSIGN pointer TO a new empty double ended queue AND SET queue TO this pointer  
        FOR each r FROM zero TO rows MINUS one  
            FOR each c FROM zero TO cols MINUS one  
                IF element at position r of rooms AT position c EQUALS zero THEN  
                    APPEND a pair consisting of r AND c TO queue  
                END IF  
            END FOR  
        END FOR  
        SET directions TO a list containing four pairs representing increments for row AND column for right down left AND up respectively  
        WHILE queue IS NOT EMPTY  
            REMOVE and ASSIGN the first element OF queue TO r AND c  
            FOR each dr AND dc IN directions  
                SET nr TO r PLUS dr  
                SET nc TO c PLUS dc  
                IF nr IS GREATER THAN OR EQUAL TO zero AND nr IS LESS THAN rows AND nc IS GREATER THAN OR EQUAL TO zero AND nc IS LESS THAN cols AND element at position nr of rooms AT position nc EQUALS two billion one hundred forty seven million four hundred eighty three thousand six hundred forty seven THEN  
                    SET element at position nr of rooms AT position nc TO element at position r of rooms AT position c PLUS one  
                    APPEND a pair consisting of nr AND nc TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
