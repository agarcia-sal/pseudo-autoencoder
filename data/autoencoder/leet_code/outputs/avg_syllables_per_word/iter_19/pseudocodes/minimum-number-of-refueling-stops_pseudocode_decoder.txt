CLASS Solution  
    FUNCTION minRefuelStops(target INTEGER, startFuel INTEGER, stations LIST OF LIST OF INTEGER) RETURNS INTEGER  
        SET max_heap TO empty list  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a list containing target and zero TO the list stations  

        FOR each position AND fuel IN the list stations  
            WHILE the list max_heap is not empty AND current_fuel IS LESS THAN position  
                SET removed_fuel TO the removal of the smallest element from the list max_heap which represents the largest available fuel amount negated  
                DECREMENT current_fuel BY removed_fuel multiplied by negative one  
                INCREMENT stops BY one  
            END WHILE  

            IF current_fuel IS LESS THAN position  
                RETURN negative one  
            END IF  

            INSERT negative fuel INTO the list max_heap to maintain max heap property  
        END FOR  

        RETURN stops  
    END FUNCTION  
END CLASS
