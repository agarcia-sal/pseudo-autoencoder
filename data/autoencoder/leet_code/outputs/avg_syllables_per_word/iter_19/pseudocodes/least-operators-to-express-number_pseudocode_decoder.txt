CLASS Solution  
    FUNCTION leastOpsExpressTarget(x AS integer, target AS integer) RETURNS integer  
        FUNCTION dp(target_value AS integer) RETURNS integer  
            IF target_value EQUALS zero THEN  
                RETURN negative one  
            END IF  
            IF target_value LESS THAN x THEN  
                SET first_option TO two MULTIPLIED BY target_value MINUS one  
                SET second_option TO two MULTIPLIED BY x MINUS target_value  
                RETURN the MINIMUM OF first_option AND second_option  
            END IF  
            SET power TO one  
            WHILE x RAISED TO power LESS THAN OR EQUAL TO target_value HOLDS  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET res TO dp(target_value MINUS x RAISED TO power) PLUS power  
            SET next_power_value TO x RAISED TO power PLUS one  
            IF next_power_value MINUS target_value LESS THAN target_value THEN  
                SET res TO the MINIMUM OF res AND dp(next_power_value MINUS target_value) PLUS power PLUS one  
            END IF  
            RETURN res  
        END FUNCTION  
        RETURN dp(target)  
    END FUNCTION  
END CLASS
