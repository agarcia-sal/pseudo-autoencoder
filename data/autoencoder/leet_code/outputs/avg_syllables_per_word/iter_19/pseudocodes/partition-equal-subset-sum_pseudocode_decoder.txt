CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO the SUM OF all elements IN nums  
        IF the remainder WHEN total_sum IS DIVIDED BY two IS NOT EQUAL TO zero  
            RETURN False  
        END IF  
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        SET dp TO a list OF boolean values WITH size target PLUS one INITIALIZED TO False  
        SET the element at position zero of dp TO True  
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET the element at position i of dp TO the element at position i of dp OR the element at position i MINUS num of dp  
            END FOR  
        END FOR  
        RETURN the element at position target of dp  
    END FUNCTION  
END CLASS
