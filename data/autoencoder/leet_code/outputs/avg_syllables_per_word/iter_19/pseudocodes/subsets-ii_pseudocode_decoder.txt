CLASS Solution  
    FUNCTION subsetsWithDup(nums)  
        CALL sort ON nums TO ARRANGE numbers IN ASCENDING ORDER FOR DUPLICATE HANDLING  
        SET result TO empty list  
        SET subset TO empty list  
        
        FUNCTION dfs(i)  
            IF i IS GREATER THAN OR EQUAL TO the LENGTH OF nums THEN  
                APPEND a COPY OF subset TO result  
                RETURN  
            END IF  
            
            APPEND the element at position i of nums TO subset  
            CALL dfs WITH the VALUE of i PLUS one  
            
            REMOVE the last element FROM subset  
            WHILE the VALUE of i PLUS one IS LESS THAN the LENGTH OF nums AND the element at position i of nums EQUALS the element at position i PLUS one of nums HOLD  
                INCREMENT i BY one  
            END WHILE  
            CALL dfs WITH the VALUE of i PLUS one  
        END FUNCTION  
        
        CALL dfs WITH zero  
        RETURN result  
    END FUNCTION  
END CLASS
