CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO an empty collection mapping characters to indices  
        SET max_length TO zero  
        SET start TO zero  
        
        FOR each position end AND character char IN the enumeration of string s  
            IF character char is a key in char_index_map AND the value at character char in char_index_map IS GREATER THAN OR EQUAL TO start  
                SET start TO the value at character char in char_index_map PLUS one  
            END IF  
            SET the value at character char in char_index_map TO end  
            SET max_length TO the GREATER OF max_length AND the result of end MINUS start PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
