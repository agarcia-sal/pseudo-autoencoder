CLASS Solution  
    FUNCTION kthSmallestProduct(nums1 AS List OF Integer, nums2 AS List OF Integer, k AS Integer) RETURNS Integer  
        FUNCTION count(p AS Integer) RETURNS Integer  
            SET cnt TO zero  
            SET n TO the LENGTH OF nums2  
            FOR each element x IN nums1  
                IF x IS GREATER THAN zero  
                    INCREMENT cnt BY the POSITION AFTER the last element LESS THAN OR EQUAL TO the result of p DIVIDED BY x IN nums2  
                ELSE IF x IS LESS THAN zero  
                    INCREMENT cnt BY n MINUS the POSITION OF the first element GREATER THAN OR EQUAL TO the result of p DIVIDED BY x IN nums2  
                ELSE  
                    INCREMENT cnt BY n MULTIPLIED BY the VALUE OF whether p IS GREATER THAN OR EQUAL TO zero  
                END IF  
            END FOR  
            RETURN cnt  
        END FUNCTION  
        SET mx TO the MAXIMUM OF the ABSOLUTE VALUE OF the element at position zero of nums1 AND the ABSOLUTE VALUE OF the element at the last position of nums1 MULTIPLIED BY the MAXIMUM OF the ABSOLUTE VALUE OF the element at position zero of nums2 AND the ABSOLUTE VALUE OF the element at the last position of nums2  
        RETURN the POSITION OF the first value GREATER THAN OR EQUAL TO k IN the SEQUENCE OF integer values FROM the NEGATION OF mx TO mx INCLUSIVE USING count AS key MINUS mx  
    END FUNCTION  
END CLASS
