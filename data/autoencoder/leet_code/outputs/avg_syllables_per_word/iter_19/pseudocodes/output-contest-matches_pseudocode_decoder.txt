CLASS Solution  
    FUNCTION findContestMatch WITH parameter n  
        SET teams TO the LIST OF all numbers FROM one TO n converted TO STRING  
        WHILE n IS GREATER THAN one  
            FOR each i FROM zero TO the INTEGER DIVISION OF n BY two MINUS one  
                SET element at position i of teams TO the STRING CONCATENATION OF an opening parenthesis FOLLOWED BY element at position i of teams FOLLOWED BY a comma FOLLOWED BY element at position n MINUS one MINUS i of teams FOLLOWED BY a closing parenthesis  
            END FOR  
            SET n TO the INTEGER DIVISION OF n BY two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
