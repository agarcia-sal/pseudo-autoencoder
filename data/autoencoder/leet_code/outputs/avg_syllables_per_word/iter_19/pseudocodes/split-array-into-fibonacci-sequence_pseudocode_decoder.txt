CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF the element at position start of num EQUALS zero AND end GREATER THAN start PLUS one  
                    CONTINUE to the next iteration of the FOR loop  
                END IF  
                SET next_num TO the integer value of the substring from position start TO end MINUS one of num  
                IF next_num GREATER THAN OR EQUAL TO two POWER thirty one  
                    BREAK the FOR loop  
                END IF  
                IF the LENGTH OF path LESS THAN two OR next_num EQUALS the element at the last position of path PLUS the element at the second last position of path  
                    SET result TO backtrack with parameters end and path with next_num appended  
                    IF result IS NOT empty  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN an empty list  
        END FUNCTION  
        
        RETURN backtrack with parameters zero and an empty list  
    END FUNCTION  
END CLASS
