CLASS Solution  
    FUNCTION convert(s, numRows)  
        IF numRows EQUALS one  
            RETURN s  
        END IF  
        SET rows TO a list containing empty strings repeated numRows times  
        SET current_row TO zero  
        SET going_down TO false  
        FOR each character IN s  
            SET row_at_current_row TO element at position current_row of rows  
            SET row_at_current_row TO row_at_current_row concatenated WITH character  
            ASSIGN element at position current_row of rows TO row_at_current_row  
            IF current_row EQUALS zero OR current_row EQUALS numRows MINUS one  
                SET going_down TO the logical negation of going_down  
            END IF  
            IF going_down IS true  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
        SET result TO the concatenation of all elements in rows in sequence  
        RETURN result  
    END FUNCTION  
END CLASS
