CLASS Solution  
    FUNCTION strangePrinter(s)  
        FUNCTION dp(i, j, memo)  
            IF i GREATER THAN j  
                RETURN zero  
            END IF  
            IF pair of i and j IS IN memo  
                RETURN the element at pair of i and j IN memo  
            END IF  
            SET result TO the result of dp with parameters i and j MINUS one and memo PLUS one  
            FOR k FROM i TO j MINUS one  
                IF the element at position k of s EQUALS the element at position j of s  
                    SET result TO the MINIMUM OF result AND the sum of the result of dp with parameters i and k MINUS one and memo AND the result of dp with parameters k and j MINUS one and memo  
                END IF  
            END FOR  
            SET the element at pair of i and j IN memo TO result  
            RETURN result  
        END FUNCTION  
        RETURN the result of dp with parameters zero AND the length of s MINUS one AND an empty memo  
    END FUNCTION  
END CLASS
