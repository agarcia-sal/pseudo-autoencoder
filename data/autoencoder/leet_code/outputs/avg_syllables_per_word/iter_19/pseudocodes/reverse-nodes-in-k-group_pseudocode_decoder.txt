CLASS ListNode  
    FUNCTION __init__(val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO NEW ListNode WITH parameter element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO the end  
        SET node TO NEW ListNode WITH parameter val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN the value at p1 EQUALS the value at p2 AND the result of is_same_list WITH parameters the element at position next of p1 and the element at position next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseKGroup(head, k)  
        FUNCTION reverseLinkedList(start, end)  
            SET prev TO none  
            SET curr TO start  
            WHILE curr IS NOT EQUALS end  
                SET next_node TO the element at position next of curr  
                SET the element at position next of curr TO prev  
                SET prev TO curr  
                SET curr TO next_node  
            END WHILE  
            SET the element at position next of curr TO prev  
            RETURN end AND start  
        END FUNCTION  

        SET dummy TO NEW ListNode WITH parameter zero  
        SET the element at position next of dummy TO head  
        SET prev_group_end TO dummy  

        WHILE True  
            SET kth_node TO prev_group_end  
            FOR index FROM one TO k  
                SET kth_node TO the element at position next of kth_node  
                IF kth_node IS none  
                    RETURN the element at position next of dummy  
                END IF  
            END FOR  
            SET next_group_start TO the element at position next of kth_node  
            SET the element at position next of kth_node TO none  
            SET new_group_start, new_group_end TO reverseLinkedList WITH parameters the element at position next of prev_group_end AND kth_node  
            SET the element at position next of prev_group_end TO new_group_start  
            SET the element at position next of new_group_end TO next_group_start  
            SET prev_group_end TO new_group_end  
        END WHILE  

        RETURN the element at position next of dummy  
    END FUNCTION  
END CLASS
