CLASS ListNode  
    FUNCTION __init__ with parameters val defaulting to zero and next defaulting to None  
        SET the member variable val TO parameter val  
        SET the member variable next TO parameter next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values as a list  
    IF the parameter values is empty  
        RETURN None  
    END IF  
    SET head TO a new instance of ListNode initialized with the element at position zero of values  
    SET p TO head  
    FOR each element val IN the elements of values from position one TO the end  
        SET node TO a new instance of ListNode initialized with val  
        SET the next member of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 and p2  
    IF p1 is None AND p2 is None  
        RETURN True  
    END IF  
    IF p1 is None OR p2 is None  
        RETURN False  
    END IF  
    RETURN the result of the logical expression that the member val of p1 EQUALS the member val of p2 AND the recursive call to is_same_list with the member next of p1 and the member next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION rotateRight with parameters head of optional ListNode AND k as an integer returning an optional ListNode  
        IF head is None OR the member next of head is None OR k EQUALS zero  
            RETURN head  
        END IF  
        SET length TO integer one  
        SET tail TO head  
        WHILE the member next of tail is not None  
            SET tail TO the member next of tail  
            INCREMENT length BY one  
        END WHILE  
        SET the member next of tail TO head  
        SET k TO the remainder of k DIVIDED BY length  
        SET steps_to_new_tail TO length MINUS k MINUS one  
        SET new_tail TO head  
        FOR a counting variable from zero TO steps_to_new_tail EXCLUSIVE  
            SET new_tail TO the member next of new_tail  
        END FOR  
        SET new_head TO the member next of new_tail  
        SET the member next of new_tail TO None  
        RETURN new_head  
    END FUNCTION  
END CLASS
