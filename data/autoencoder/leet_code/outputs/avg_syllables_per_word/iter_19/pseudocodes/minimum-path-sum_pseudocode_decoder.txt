CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid EQUALS empty OR the element at position zero of grid EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero of grid  
        
        SET dp TO a two dimensional list of zeros with m rows and n columns  
        
        SET the element at row zero column zero of dp TO the element at row zero column zero of grid  
        
        FOR j FROM one TO n MINUS one  
            SET the element at row zero column j of dp TO the element at row zero column j MINUS one of dp PLUS the element at row zero column j of grid  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            SET the element at row i column zero of dp TO the element at row i MINUS one column zero of dp PLUS the element at row i column zero of grid  
        END FOR  
        
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET the element at row i column j of dp TO THE MINIMUM OF the element at row i MINUS one column j of dp AND the element at row i column j MINUS one of dp PLUS the element at row i column j of grid  
            END FOR  
        END FOR  
        
        RETURN the element at row m MINUS one column n MINUS one of dp  
    END FUNCTION  
END CLASS
