CLASS Solution  
    FUNCTION outerTrees(trees)  
        FUNCTION dist_sq(p1, p2)  
            RETURN the SQUARE OF the DIFFERENCE BETWEEN element at position zero of p1 AND element at position zero of p2 PLUS the SQUARE OF the DIFFERENCE BETWEEN element at position one of p1 AND element at position one of p2  
        END FUNCTION

        FUNCTION circumcenter(p1, p2, p3)  
            SET ax TO element at position zero of p1  
            SET ay TO element at position one of p1  
            SET bx TO element at position zero of p2  
            SET by TO element at position one of p2  
            SET cx TO element at position zero of p3  
            SET cy TO element at position one of p3  
            SET d TO two MULTIPLIED BY the SUM of ax MULTIPLIED BY the DIFFERENCE BETWEEN by AND cy PLUS bx MULTIPLIED BY the DIFFERENCE BETWEEN cy AND ay PLUS cx MULTIPLIED BY the DIFFERENCE BETWEEN ay AND by  
            SET ux TO the QUOTIENT of the SUM of the SUM of ax MULTIPLIED BY ax AND ay MULTIPLIED BY ay MULTIPLIED BY the DIFFERENCE BETWEEN by AND cy PLUS the SUM of bx MULTIPLIED BY bx AND by MULTIPLIED BY by MULTIPLIED BY the DIFFERENCE BETWEEN cy AND ay PLUS the SUM of cx MULTIPLIED BY cx AND cy MULTIPLIED BY cy MULTIPLIED BY the DIFFERENCE BETWEEN ay AND by DIVIDED BY d  
            SET uy TO the QUOTIENT of the SUM of the SUM of ax MULTIPLIED BY ax AND ay MULTIPLIED BY ay MULTIPLIED BY the DIFFERENCE BETWEEN cx AND bx PLUS the SUM of bx MULTIPLIED BY bx AND by MULTIPLIED BY by MULTIPLIED BY the DIFFERENCE BETWEEN ax AND cx PLUS the SUM of cx MULTIPLIED BY cx AND cy MULTIPLIED BY cy MULTIPLIED BY the DIFFERENCE BETWEEN bx AND ax DIVIDED BY d  
            RETURN list containing ux AND uy, the SQUARE ROOT of the SUM of the SQUARE OF the DIFFERENCE BETWEEN ux AND ax AND the SQUARE OF the DIFFERENCE BETWEEN uy AND ay  
        END FUNCTION  

        FUNCTION welzl(points, boundary, n)  
            IF n EQUALS zero OR the LENGTH OF boundary EQUALS three THEN  
                IF the LENGTH OF boundary EQUALS zero THEN  
                    RETURN list containing zero AND zero, zero  
                ELSE IF the LENGTH OF boundary EQUALS one THEN  
                    RETURN the FIRST element of boundary, zero  
                ELSE IF the LENGTH OF boundary EQUALS two THEN  
                    SET center TO list containing the QUOTIENT of the SUM of element at position zero of the FIRST element of boundary AND element at position zero of the SECOND element of boundary DIVIDED BY two AND the QUOTIENT of the SUM of element at position one of the FIRST element of boundary AND element at position one of the SECOND element of boundary DIVIDED BY two  
                    SET radius TO the SQUARE ROOT of dist_sq between the FIRST element of boundary AND the SECOND element of boundary DIVIDED BY two  
                    RETURN center, radius  
                ELSE  
                    RETURN circumcenter with UNPACKED elements of boundary  
                END IF  
            END IF  

            SET p TO element at position n MINUS one of points  
            SET center AND radius TO welzl with points, boundary, AND n MINUS one  
            IF dist_sq between center AND p LESS THAN OR EQUAL TO radius MULTIPLIED BY radius THEN  
                RETURN center, radius  
            END IF  

            RETURN welzl with points, boundary WITH p ADDED, AND n MINUS one  
        END FUNCTION  

        CALL random SHUFFLE ON trees  
        SET center AND radius TO welzl with trees, empty list, AND LENGTH OF trees  
        RETURN list containing element at position zero of center, element at position one of center, AND radius  
    END FUNCTION  
END CLASS
