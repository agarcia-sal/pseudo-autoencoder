CLASS Solution  
    FUNCTION numOfArrays(n, m, k)  
        SET MOD TO one billion PLUS seven  
        
        CREATE three dimensional list dp with dimensions n PLUS one BY m PLUS one BY k PLUS one FILLED WITH zero  
        
        FOR each j FROM one TO m  
            SET element at position one of dp at position j at position one TO one  
        END FOR  
        
        FOR each i FROM two TO n  
            FOR each j FROM one TO m  
                FOR each p FROM one TO k  
                    FOR each x FROM one TO j MINUS one  
                        SET element at position i of dp at position j at position p TO element at position i of dp at position j at position p PLUS element at position i MINUS one of dp at position x at position p MINUS one MODULO MOD  
                    END FOR  
                    SET element at position i of dp at position j at position p TO element at position i of dp at position j at position p PLUS element at position i MINUS one of dp at position j at position p MULTIPLIED BY j MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR each j FROM one TO m  
            SET result TO result PLUS element at position n of dp at position j at position k MODULO MOD  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
