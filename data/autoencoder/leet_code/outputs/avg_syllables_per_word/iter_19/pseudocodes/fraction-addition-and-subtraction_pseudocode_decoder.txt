CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET modified_expression TO the result of REPLACING character MINUS WITH character PLUS MINUS IN expression  
        SET fractions_list TO the result of SPLITTING modified_expression BY character PLUS  
        SET filtered_fractions TO an empty list  
        FOR each fraction_string IN fractions_list  
            IF fraction_string IS NOT empty  
                APPEND fraction_string TO filtered_fractions  
            END IF  
        END FOR  
        SET result_fraction TO the summation of converting each element of filtered_fractions INTO fraction numbers USING the Fraction behavior  
        SET numerator_value TO the numerator component of result_fraction  
        SET denominator_value TO the denominator component of result_fraction  
        RETURN a string REPRESENTING numerator_value FOLLOWED BY character SLASH FOLLOWED BY denominator_value  
    END FUNCTION  
END CLASS
