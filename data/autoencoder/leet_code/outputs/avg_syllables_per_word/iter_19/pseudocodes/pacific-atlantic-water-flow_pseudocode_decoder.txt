CLASS Solution  
    FUNCTION pacificAtlantic WITH PARAMETERS heights  
        FUNCTION bfs WITH PARAMETERS q AND vis  
            WHILE q IS NOT EMPTY  
                FOR each element FROM zero TO the LENGTH OF q MINUS one  
                    SET i AND j TO the ELEMENT AT THE FRONT OF q REMOVED FROM q  
                    FOR each pair a AND b IN the LIST OF PAIRS zero MINUS one zero one one zero MINUS one zero  
                        SET x TO i PLUS a  
                        SET y TO j PLUS b  
                        IF x IS GREATER THAN OR EQUAL TO zero AND x IS LESS THAN m AND y IS GREATER THAN OR EQUAL TO zero AND y IS LESS THAN n AND THE PAIR x AND y IS NOT IN vis AND THE HEIGHT AT POSITION x OF heights ELEMENT AT POSITION y IS GREATER THAN OR EQUAL TO THE HEIGHT AT POSITION i OF heights ELEMENT AT POSITION j  
                            ADD THE PAIR x AND y TO vis  
                            APPEND THE PAIR x AND y TO q  
                        END IF  
                    END FOR  
                END FOR  
            END WHILE  
        END FUNCTION  

        SET m TO the LENGTH OF heights  
        SET n TO the LENGTH OF the ELEMENT AT POSITION zero OF heights  
        SET vis1 TO an EMPTY SET  
        SET vis2 TO an EMPTY SET  
        SET q1 TO an EMPTY DOUBLE ENDED QUEUE  
        SET q2 TO an EMPTY DOUBLE ENDED QUEUE  
        FOR each i FROM zero TO m MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF i EQUALS zero OR j EQUALS zero  
                    ADD THE PAIR i AND j TO vis1  
                    APPEND THE PAIR i AND j TO q1  
                END IF  
                IF i EQUALS m MINUS one OR j EQUALS n MINUS one  
                    ADD THE PAIR i AND j TO vis2  
                    APPEND THE PAIR i AND j TO q2  
                END IF  
            END FOR  
        END FOR  
        CALL bfs WITH q1 AND vis1  
        CALL bfs WITH q2 AND vis2  
        SET result TO an EMPTY LIST  
        FOR each i FROM zero TO m MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF THE PAIR i AND j IS IN vis1 AND THE PAIR i AND j IS IN vis2  
                    APPEND THE PAIR i AND j TO result  
                END IF  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
