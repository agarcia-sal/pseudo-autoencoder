CLASS Solution  
    FUNCTION jump(nums)  
        IF the LENGTH OF nums IS LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        SET jumps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        FOR index FROM zero TO the LENGTH OF nums MINUS one MINUS one  
            SET farthest TO the MAXIMUM OF farthest AND the SUM OF index AND the element at position index of nums  
            IF index EQUALS current_end  
                INCREMENT jumps BY one  
                SET current_end TO farthest  
                IF current_end IS GREATER THAN OR EQUAL TO the LENGTH OF nums MINUS one  
                    BREAK  
                END IF  
            END IF  
        END FOR  
        RETURN jumps  
    END FUNCTION  
END CLASS
