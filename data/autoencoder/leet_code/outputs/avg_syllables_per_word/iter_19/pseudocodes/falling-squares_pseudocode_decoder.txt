CLASS Solution  
    FUNCTION fallingSquares WITH parameters positions  
        SET max_heights TO an empty list  
        SET intervals TO an empty list  
        SET current_max_height TO zero  
        FOR each pair left AND side_length IN positions  
            SET right TO the result of left PLUS side_length MINUS one  
            SET height TO side_length  
            FOR each triple interval_left AND interval_right AND interval_height IN intervals  
                IF left LESS THAN OR EQUAL TO interval_right AND right GREATER THAN OR EQUAL TO interval_left  
                    SET height TO the maximum value BETWEEN height AND side_length PLUS interval_height  
                END IF  
            END FOR  
            APPEND the triple consisting of left AND right AND height TO intervals  
            SET current_max_height TO the maximum value BETWEEN current_max_height AND height  
            APPEND current_max_height TO max_heights  
        END FOR  
        RETURN max_heights  
    END FUNCTION  
END CLASS
