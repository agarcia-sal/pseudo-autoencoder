CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the val property OF this object TO val  
        SET the next property OF this object TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the list values IS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode with the element at position zero of values  
    SET p TO head  
    FOR each val IN the sublist of values from position one TO the end  
        SET node TO a new ListNode with val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the result of the comparison of the val property of p1 EQUALS the val property of p2 AND the recursive call of is_same_list WITH the next property of p1 AND the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates WITH PARAMETERS self AND head  
        SET dummy TO a new ListNode with zero AND head  
        SET prev TO dummy  
        WHILE head IS NOT none  
            IF the next property of head IS NOT none AND the val property of head EQUALS the val property of the next property of head  
                WHILE the next property of head IS NOT none AND the val property of head EQUALS the val property of the next property of head  
                    SET head TO the next property of head  
                END WHILE  
                SET the next property of prev TO the next property of head  
            ELSE  
                SET prev TO the next property of prev  
            END IF  
            SET head TO the next property of head  
        END WHILE  
        RETURN the next property of dummy  
    END FUNCTION  
END CLASS
