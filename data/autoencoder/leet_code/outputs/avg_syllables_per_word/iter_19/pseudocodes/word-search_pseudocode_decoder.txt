CLASS Solution  
    FUNCTION exist(board AS List OF List OF string, word AS string) RETURNS boolean  
        IF board EQUALS empty  
            RETURN False  
        END IF  

        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero of board  
        SET visited TO a two dimensional list of boolean values with m rows and n columns all SET TO False  

        FUNCTION dfs(x AS integer, y AS integer, index AS integer) RETURNS boolean  
            IF index EQUALS the LENGTH OF word  
                RETURN True  
            END IF  

            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of visited at position y IS True OR element at position x of board at position y NOT EQUALS element at position index of word  
                RETURN False  
            END IF  

            SET element at position x of visited at position y TO True  
            FOR each pair dx AND dy IN the collection consisting of pairs negative one zero positive one zero zero negative one zero positive one  
                IF dfs with parameters the sum of x and dx the sum of y and dy the sum of index and one RETURNS True  
                    RETURN True  
                END IF  
            END FOR  
            SET element at position x of visited at position y TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM zero TO m minus one  
            FOR j FROM zero TO n minus one  
                IF dfs with parameters i j zero RETURNS True  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
