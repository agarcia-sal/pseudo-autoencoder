CLASS Solution  
    FUNCTION twoSum(numbers, target)  
        SET left TO zero  
        SET right TO the LENGTH OF numbers MINUS one  
        WHILE left IS LESS THAN right  
            SET current_sum TO the element at position left of numbers PLUS the element at position right of numbers  
            IF current_sum EQUALS target  
                RETURN a list containing left PLUS one and right PLUS one  
            ELSE IF current_sum IS LESS THAN target  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
