CLASS Solution  
    FUNCTION parseBoolExpr(expression)  
        SET stk TO empty list  
        FOR each c IN expression  
            IF c EQUALS one of true character OR false character OR exclamation character OR ampersand character OR vertical bar character  
                APPEND c TO stk  
            ELSE IF c EQUALS closing parenthesis character  
                SET t TO zero  
                SET f TO zero  
                WHILE the last element of stk IS one of true character OR false character  
                    INCREMENT t BY one IF the last element of stk EQUALS true character  
                    INCREMENT f BY one IF the last element of stk EQUALS false character  
                    REMOVE the last element FROM stk  
                END WHILE  
                SET op TO the last element REMOVED FROM stk  
                IF op EQUALS exclamation character  
                    SET c TO true character IF f GREATER THAN zero OTHERWISE false character  
                ELSE IF op EQUALS ampersand character  
                    SET c TO false character IF f GREATER THAN zero OTHERWISE true character  
                ELSE IF op EQUALS vertical bar character  
                    SET c TO true character IF t GREATER THAN zero OTHERWISE false character  
                END IF  
                APPEND c TO stk  
            END IF  
        END FOR  
        RETURN whether the first element of stk EQUALS true character  
    END FUNCTION  
END CLASS
