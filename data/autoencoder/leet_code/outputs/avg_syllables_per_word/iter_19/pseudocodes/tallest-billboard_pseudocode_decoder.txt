CLASS Solution  
    FUNCTION tallestBillboard(rods)  
        SET dp TO dictionary WITH key zero AND value zero  
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each key d AND value y IN dp  
                SET sum_with_rod_added_to_taller_support TO d PLUS rod  
                IF the value associated with sum_with_rod_added_to_taller_support in new_dp DOES NOT EXIST OR the value in new_dp AT sum_with_rod_added_to_taller_support IS LESS THAN y  
                    SET the value associated with sum_with_rod_added_to_taller_support in new_dp TO y  
                END IF  
                SET difference_after_adding_rod_to_shorter_support TO the absolute value of d MINUS rod  
                SET candidate_value TO y PLUS the lesser of d AND rod  
                IF the value associated with difference_after_adding_rod_to_shorter_support in new_dp DOES NOT EXIST OR the value in new_dp AT difference_after_adding_rod_to_shorter_support IS LESS THAN candidate_value  
                    SET the value associated with difference_after_adding_rod_to_shorter_support in new_dp TO candidate_value  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN the value associated with zero in dp  
    END FUNCTION  
END CLASS
