CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string THEN  
            RETURN an empty list  
        END IF  
        
        SET phone_map TO a mapping from the string two TO the string abc, the string three TO the string def, the string four TO the string ghi, the string five TO the string jkl, the string six TO the string mno, the string seven TO the string pqrs, the string eight TO the string tuv, the string nine TO the string wxyz  
        
        FUNCTION backtrack(combination, next_digits)  
            IF the LENGTH OF next_digits EQUALS zero THEN  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN the element at position zero of next_digits MAPPED THROUGH phone_map  
                    CALL backtrack WITH the concatenation of combination PLUS letter AND the substring from position one TO the end of next_digits  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET output TO an empty list  
        CALL backtrack WITH empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
