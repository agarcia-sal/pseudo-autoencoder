CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        
        SET value_indices TO an empty mapping from values TO empty lists  
        FOR each i FROM zero TO n MINUS one  
            SET value TO element at position i of arr  
            APPEND i TO the list mapped BY value in value_indices  
        END FOR  
        
        SET queue TO a double ended queue CONTAINING the pair zero AND zero representing current_index and current_steps  
        SET visited TO a set CONTAINING zero  
        
        WHILE queue is not empty  
            REMOVE the first element from queue AND DESTRUCTURE INTO current_index AND steps  
            
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            SET next_indices TO a new list CONTAINING current_index PLUS one, current_index MINUS one, FOLLOWED BY all elements in the list mapped BY element at position current_index of arr in value_indices  
            
            FOR each next_index IN next_indices  
                IF next_index IS GREATER THAN OR EQUAL TO zero AND next_index IS LESS THAN n AND next_index IS NOT in visited  
                    ADD next_index TO visited  
                    APPEND the pair next_index AND steps PLUS one TO queue  
                END IF  
            END FOR  
            
            SET the list mapped BY element at position current_index of arr in value_indices TO an empty list  
        
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
