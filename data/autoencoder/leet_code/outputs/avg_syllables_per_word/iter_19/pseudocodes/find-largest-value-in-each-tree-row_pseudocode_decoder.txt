CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS none right EQUALS none  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETER element at position zero of values  
    SET i TO one  
    SET queue TO NEW double ended queue structure  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE first element from queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val IS NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS p dot left AND q dot left AND the RESULT OF is_same_tree WITH PARAMETERS p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues WITH PARAMETERS self root  
        IF root IS none  
            RETURN empty list  
        END IF  
        SET queue TO NEW double ended queue structure INITIALIZED WITH root  
        SET largest_values TO an empty list  
        WHILE queue IS NOT EMPTY  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR each iteration FROM zero TO level_size MINUS one  
                SET node TO REMOVE first element from queue  
                IF max_value IS LESS THAN node dot val  
                    SET max_value TO node dot val  
                END IF  
                IF node dot left IS NOT none  
                    APPEND node dot left TO queue  
                END IF  
                IF node dot right IS NOT none  
                    APPEND node dot right TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
