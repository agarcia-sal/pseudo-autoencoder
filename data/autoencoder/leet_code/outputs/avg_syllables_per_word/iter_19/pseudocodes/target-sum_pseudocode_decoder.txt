CLASS Solution  
    FUNCTION findTargetSumWays(nums LIST OF INTEGER, target INTEGER) RETURNS INTEGER  
        FUNCTION dfs(index INTEGER, current_sum INTEGER, memo DICTIONARY) RETURNS INTEGER  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF the TUPLE of index AND current_sum IS a KEY IN memo  
                RETURN the VALUE corresponding TO the TUPLE of index AND current_sum FROM memo  
            END IF  
            SET positive TO THE RESULT OF dfs CALLED WITH index PLUS one, current_sum PLUS the ELEMENT at position index OF nums, AND memo  
            SET negative TO THE RESULT OF dfs CALLED WITH index PLUS one, current_sum MINUS the ELEMENT at position index OF nums, AND memo  
            SET the VALUE corresponding TO the TUPLE of index AND current_sum IN memo TO positive PLUS negative  
            RETURN the VALUE corresponding TO the TUPLE of index AND current_sum FROM memo  
        END FUNCTION  
        RETURN the RESULT OF dfs CALLED WITH zero, zero, AND an EMPTY DICTIONARY  
    END FUNCTION  
END CLASS
