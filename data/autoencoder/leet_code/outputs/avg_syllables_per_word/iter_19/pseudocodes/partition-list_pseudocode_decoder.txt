CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty list  
        RETURN none  
    END IF  
    SET head TO new ListNode with value at position zero of values  
    SET p TO head  
    FOR each val IN elements of values from position one TO end  
        SET node TO new ListNode with val  
        SET next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN logical AND of the expression that value of p1 EQUALS value of p2 AND the recursive call of is_same_list with arguments next of p1 AND next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION partition(self, head, x)  
        SET less_head TO new ListNode with zero  
        SET greater_head TO new ListNode with zero  
        SET less TO less_head  
        SET greater TO greater_head  
        SET current TO head  
        WHILE current holds truth  
            IF value of current LESS THAN x  
                SET next of less TO current  
                SET less TO next of less  
            ELSE  
                SET next of greater TO current  
                SET greater TO next of greater  
            END IF  
            SET current TO next of current  
        END WHILE  
        SET next of less TO next of greater_head  
        SET next of greater TO none  
        RETURN next of less_head  
    END FUNCTION  
END CLASS
