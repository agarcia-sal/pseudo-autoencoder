CLASS Solution  
    FUNCTION nthSuperUglyNumber(n, primes)  
        SET ugly_numbers TO a list containing one  
        SET heap TO an empty list  
        FOR each prime ACCOMPANIED BY its index IN primes  
            APPEND a tuple of the prime AND its index TO heap  
        END FOR  
        TRANSFORM heap INTO a min-heap  
        SET indices TO a list of zeros with the length EQUAL TO the length of primes  
        WHILE the LENGTH OF ugly_numbers IS LESS THAN n  
            REMOVE the smallest tuple from heap AND ASSIGN TO next_ugly AND prime_index  
            IF next_ugly IS NOT EQUAL TO the last element of ugly_numbers  
                APPEND next_ugly TO ugly_numbers  
            END IF  
            INCREMENT the element at position prime_index of indices BY one  
            SET next_multiple TO the element at position prime_index of primes MULTIPLIED BY the element at position element at position prime_index of indices of ugly_numbers  
            ADD the tuple of next_multiple AND prime_index TO heap  
        END WHILE  
        RETURN the last element of ugly_numbers  
    END FUNCTION  
END CLASS
