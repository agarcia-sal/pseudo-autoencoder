CLASS Solution  
    FUNCTION minCost(houses, cost, m, n, target)  
        FUNCTION dp(i, prev_color, groups)  
            IF i EQUALS m THEN  
                RETURN zero IF groups EQUALS target ELSE infinite  
            END IF  
            
            IF element at position i of houses NOT EQUALS zero THEN  
                SET new_groups TO groups PLUS one IF element at position i of houses NOT EQUALS prev_color ELSE groups  
                RETURN dp(i PLUS one, element at position i of houses, new_groups)  
            END IF  
            
            SET min_cost TO infinite  
            FOR color FROM one TO n  
                SET new_groups TO groups PLUS one IF color NOT EQUALS prev_color ELSE groups  
                SET temp_cost TO element at position i at position color MINUS one of cost PLUS dp(i PLUS one, color, new_groups)  
                SET min_cost TO minimum of min_cost AND temp_cost  
            END FOR  
            
            RETURN min_cost  
        END FUNCTION  
        
        SET result TO dp(zero, zero, zero)  
        RETURN result IF result NOT EQUALS infinite ELSE negative one  
    END FUNCTION  
END CLASS
