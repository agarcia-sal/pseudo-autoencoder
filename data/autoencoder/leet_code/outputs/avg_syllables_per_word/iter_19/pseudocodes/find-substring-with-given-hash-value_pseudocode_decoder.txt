CLASS Solution  
    FUNCTION subStrHash(s power modulo k hashValue)  
        FUNCTION val(c)  
            RETURN the Unicode code point of character c MINUS the Unicode code point of lowercase letter a PLUS one  
        END FUNCTION  

        SET n TO the LENGTH OF string s  
        SET current_hash TO zero  
        SET p_pow TO one  

        FOR i FROM n MINUS k TO n MINUS one  
            SET current_hash TO the remainder when current_hash PLUS the product of val of element at position i of s AND p_pow is divided by modulo  
            IF i IS LESS THAN n MINUS one  
                SET p_pow TO the remainder when p_pow MULTIPLIED BY power is divided by modulo  
            END IF  
        END FOR  

        SET start_index TO n MINUS k  
        IF current_hash EQUALS hashValue  
            RETURN substring from position start_index TO the position start_index PLUS k MINUS one of string s  
        END IF  

        SET p_pow_k_minus_1 TO p_pow  

        FOR i FROM n MINUS k MINUS one DOWNTO zero  
            SET current_hash TO the remainder when current_hash MINUS the product of val of element at position i PLUS k of s AND p_pow_k_minus_1 is divided by modulo  
            SET current_hash TO the remainder when current_hash MULTIPLIED BY power PLUS val of element at position i of s is divided by modulo  

            IF current_hash EQUALS hashValue  
                SET start_index TO i  
            END IF  
        END FOR  

        RETURN substring from position start_index TO the position start_index PLUS k MINUS one of string s  
    END FUNCTION  
END CLASS
