CLASS Solution  
    FUNCTION knightProbability(n, k, row, column)  
        SET moves TO list containing pairs of minus two and minus one, minus two and one, minus one and minus two, minus one and two, one and minus two, one and two, two and minus one, two and one  
        SET dp TO two dimensional list of size n by n filled with zero  
        SET element at position row and column of dp TO one  
        FOR each unused_index FROM zero TO k minus one  
            SET new_dp TO two dimensional list of size n by n filled with zero  
            FOR each r FROM zero TO n minus one  
                FOR each c FROM zero TO n minus one  
                    IF element at position r and c of dp EQUALS zero  
                        CONTINUE to next iteration of c loop  
                    END IF  
                    FOR each pair dr and dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n  
                            INCREMENT element at position nr and nc of new_dp BY element at position r and c of dp DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        SET total_probability TO zero  
        FOR each row_list IN dp  
            FOR each value IN row_list  
                INCREMENT total_probability BY value  
            END FOR  
        END FOR  
        RETURN total_probability  
    END FUNCTION  
END CLASS
