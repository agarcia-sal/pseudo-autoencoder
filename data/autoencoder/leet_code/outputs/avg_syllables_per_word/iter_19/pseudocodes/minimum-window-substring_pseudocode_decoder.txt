CLASS Solution  
    FUNCTION minWindow(s, t)  
        SET t_count TO the frequency count of each character IN t  
        SET required TO the LENGTH OF t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO an empty frequency count container  
        SET min_length TO infinity  
        SET min_start TO zero  

        WHILE right IS LESS THAN the LENGTH OF s  
            INCREMENT the frequency count of the character at position right of s IN window_counts  
            IF the character at position right of s IS IN t_count AND the frequency count of the character at position right of s IN window_counts EQUALS the frequency count of the same character IN t_count  
                INCREMENT formed BY one  
            END IF  

            WHILE left IS LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO the character at position left of s  
                IF the expression right MINUS left PLUS one IS LESS THAN min_length  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  

                DECREMENT the frequency count of char IN window_counts  
                IF char IS IN t_count AND the frequency count of char IN window_counts IS LESS THAN the frequency count of char IN t_count  
                    DECREMENT formed BY one  
                END IF  

                INCREMENT left BY one  
            END WHILE  

            INCREMENT right BY one  
        END WHILE  

        IF min_length EQUALS infinity  
            RETURN an empty string  
        ELSE  
            RETURN the substring from position min_start TO the position min_start PLUS min_length MINUS one of s  
        END IF  
    END FUNCTION  
END CLASS
