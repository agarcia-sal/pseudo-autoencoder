CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven  
        INITIALIZE dp AS list of zero WITH the LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS the character labeled star  
                RETURN nine  
            ELSE IF char EQUALS the character labeled zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS the character labeled star AND char2 EQUALS the character labeled star  
                RETURN fifteen  
            ELSE IF char1 EQUALS the character labeled star  
                IF char2 IS LESS THAN OR EQUAL TO the character labeled six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS the character labeled star  
                IF char1 EQUALS the character labeled one  
                    RETURN nine  
                ELSE IF char1 EQUALS the character labeled two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer CONVERSION OF the concatenation of char1 AND char2  
                IF num IS GREATER THAN OR EQUAL TO ten AND num IS LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
        
        FOR index FROM one TO the LENGTH OF s  
            SET element at position index of dp TO the REMAINDER OF the SUM OF element at position index of dp PLUS element at position index MINUS one of dp MULTIPLIED BY ways_to_decode_single(the element at position index MINUS one of s) DIVIDED BY MODULO MOD  
            IF index IS GREATER THAN one  
                SET element at position index of dp TO the REMAINDER OF the SUM OF element at position index of dp PLUS element at position index MINUS two of dp MULTIPLIED BY ways_to_decode_pair(the element at position index MINUS two of s, the element at position index MINUS one of s) DIVIDED BY MODULO MOD  
            END IF  
        END FOR  
        
        RETURN element at the LAST position of dp  
    END FUNCTION  
END CLASS
