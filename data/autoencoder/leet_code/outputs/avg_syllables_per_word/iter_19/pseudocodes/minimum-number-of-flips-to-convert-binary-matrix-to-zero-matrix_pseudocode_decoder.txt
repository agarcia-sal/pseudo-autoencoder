CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix, x, y)  
            SET directions TO list containing the pairs minus one zero one zero zero zero minus one zero one  
            FOR each pair dx AND dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN the LENGTH OF matrix AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN the LENGTH OF the element at position zero of matrix  
                    SET element at position nx of matrix at position ny TO one MINUS element at position nx of matrix at position ny  
                END IF  
            END FOR  
        END FUNCTION  
  
        FUNCTION matrix_to_tuple(matrix)  
            RETURN a tuple composed of tuples each constructed from every row in matrix  
        END FUNCTION  
  
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF the element at position zero of mat  
        SET target TO a tuple composed of tuples each filled with zero repeated n times repeated m times  
        SET initial_state TO the result OF matrix_to_tuple GIVEN mat  
        SET queue TO a list containing the pair initial_state AND zero  
        SET visited TO an empty set  
        ADD initial_state TO visited  
  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            REMOVE the first pair FROM queue AND ASSIGN TO current_state AND steps  
            IF current_state EQUALS target  
                RETURN steps  
            END IF  
  
            SET current_matrix TO a list constructed by converting each tuple row in current_state to a list  
  
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO a deep copy OF current_matrix  
                    CALL flip WITH next_matrix AND i AND j  
                    SET next_state TO the result OF matrix_to_tuple GIVEN next_matrix  
                    IF next_state IS NOT IN visited  
                        ADD next_state TO visited  
                        APPEND the pair next_state AND steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
  
        RETURN minus one  
    END FUNCTION  
END CLASS
