CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF remaining EQUALS empty  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR index FROM zero TO LENGTH OF remaining MINUS one  
                SET new_path TO path PLUS element at position index of remaining  
                SET new_remaining TO concatenation of elements from position zero TO index MINUS one of remaining PLUS elements from position index PLUS one TO end of remaining  
                CALL backtrack WITH new_path AND new_remaining  
            END FOR  
        END FUNCTION  
        SET permutations TO empty list  
        CALL backtrack WITH empty list AND nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
