CLASS Solution  
    FUNCTION isSelfCrossing(distance)  
        SET n TO the LENGTH OF distance  
        IF n LESS THAN four  
            RETURN False  
        END IF  
        FOR index FROM three TO n MINUS one  
            IF element at position index of distance GREATER THAN OR EQUAL TO element at position index MINUS two of distance AND element at position index MINUS one of distance LESS THAN OR EQUAL TO element at position index MINUS three of distance  
                RETURN True  
            END IF  
            IF index GREATER THAN OR EQUAL TO four AND element at position index MINUS one of distance EQUALS element at position index MINUS three of distance AND element at position index PLUS element at position index MINUS four of distance GREATER THAN OR EQUAL TO element at position index MINUS two of distance  
                RETURN True  
            END IF  
            IF index GREATER THAN OR EQUAL TO five AND element at position index MINUS two of distance GREATER THAN OR EQUAL TO element at position index MINUS four of distance AND element at position index MINUS three of distance GREATER THAN OR EQUAL TO element at position index MINUS one of distance AND element at position index MINUS one of distance PLUS element at position index MINUS five of distance GREATER THAN OR EQUAL TO element at position index MINUS three of distance AND element at position index PLUS element at position index MINUS four of distance GREATER THAN OR EQUAL TO element at position index MINUS two of distance  
                RETURN True  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
