CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO the LENGTH OF strs  
        SET n TO the LENGTH OF the element at position zero of strs  
        SET dp TO a list of the value one repeated n times  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i MINUS one  
                IF FOR every k FROM zero TO m MINUS one the element at position k at j of strs IS LESS THAN OR EQUAL TO the element at position k at i of strs holds  
                    SET dp at position i TO the GREATER OF dp at position i AND dp at position j PLUS one  
                END IF  
            END FOR  
        END FOR  
        SET longest_non_decreasing_subsequence_length TO the GREATEST value in dp  
        RETURN n MINUS longest_non_decreasing_subsequence_length  
    END FUNCTION  
END CLASS
