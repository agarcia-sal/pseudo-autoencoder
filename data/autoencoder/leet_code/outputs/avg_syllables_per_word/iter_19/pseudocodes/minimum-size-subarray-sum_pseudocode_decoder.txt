CLASS Solution  
    FUNCTION minSubArrayLen(target, nums)  
        SET left TO zero  
        SET current_sum TO zero  
        SET min_length TO infinity value  
        FOR right FROM zero TO the LENGTH OF nums MINUS one  
            INCREMENT current_sum BY element at position right of nums  
            WHILE current_sum GREATER THAN OR EQUAL TO target  
                SET min_length TO the smaller value BETWEEN min_length AND the RESULT OF right MINUS left PLUS one  
                DECREMENT current_sum BY element at position left of nums  
                INCREMENT left BY one  
            END WHILE  
        END FOR  
        IF min_length IS NOT EQUAL TO infinity value  
            RETURN min_length  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
