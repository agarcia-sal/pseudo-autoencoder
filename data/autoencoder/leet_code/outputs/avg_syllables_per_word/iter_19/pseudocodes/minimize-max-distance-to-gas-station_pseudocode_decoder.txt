CLASS Solution  
    FUNCTION minmaxGasDist WITH PARAMETERS stations AND k  
        FUNCTION possible WITH PARAMETER max_dist  
            SET additional_stations TO zero  
            FOR index FROM one TO the LENGTH OF stations MINUS one  
                INCREMENT additional_stations BY the INTEGER PART OF the EXPRESSION the element at position index of stations MINUS the element at position index MINUS one of stations DIVIDED BY max_dist  
            END FOR  
            RETURN additional_stations LESS THAN OR EQUAL TO k  
        END FUNCTION  

        SET left TO zero  
        SET right TO one HUNDRED MILLION  
        WHILE the DIFFERENCE between right AND left IS GREATER THAN zero POINT zero zero zero zero zero one  
            SET mid TO the SUM of left AND right DIVIDED BY two  
            IF possible WITH mid RETURNS true  
                SET right TO mid  
            ELSE  
                SET left TO mid  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
