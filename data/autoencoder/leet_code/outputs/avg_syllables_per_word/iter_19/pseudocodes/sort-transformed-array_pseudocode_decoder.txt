CLASS Solution  
    FUNCTION sortTransformedArray(nums AS List OF integers, a AS integer, b AS integer, c AS integer) RETURNS List OF integers  
        FUNCTION quadratic(x AS integer) RETURNS integer  
            RETURN the value of a MULTIPLIED BY x MULTIPLIED BY x PLUS b MULTIPLIED BY x PLUS c  
        END FUNCTION  
        
        SET n TO the LENGTH OF nums  
        SET result TO a new list of zeros with size n  
        SET left TO zero  
        SET right TO n MINUS one  
        
        IF a IS GREATER THAN OR EQUAL TO zero  
            SET index TO n MINUS one  
            WHILE left IS LESS THAN OR EQUAL TO right  
                IF the value of quadratic at element at position left of nums IS GREATER THAN the value of quadratic at element at position right of nums  
                    SET element at position index of result TO the value of quadratic at element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET element at position index of result TO the value of quadratic at element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                DECREMENT index BY one  
            END WHILE  
        ELSE  
            SET index TO zero  
            WHILE left IS LESS THAN OR EQUAL TO right  
                IF the value of quadratic at element at position left of nums IS LESS THAN the value of quadratic at element at position right of nums  
                    SET element at position index of result TO the value of quadratic at element at position left of nums  
                    INCREMENT left BY one  
                ELSE  
                    SET element at position index of result TO the value of quadratic at element at position right of nums  
                    DECREMENT right BY one  
                END IF  
                INCREMENT index BY one  
            END WHILE  
        END IF  
        
        RETURN result  
    END FUNCTION  
END CLASS
