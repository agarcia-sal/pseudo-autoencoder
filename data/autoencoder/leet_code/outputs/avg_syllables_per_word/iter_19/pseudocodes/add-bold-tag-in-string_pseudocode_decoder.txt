CLASS Solution  
    FUNCTION addBoldTag(s AS string, words AS list OF string) RETURNS string  
        IF the LENGTH OF words EQUALS zero  
            RETURN s  
        END IF  
        
        SET n TO the LENGTH OF s  
        INITIALIZE mask AS list OF boolean values all set to FALSE with the LENGTH OF n  
        
        FOR each word IN words  
            SET start TO zero  
            WHILE start LESS THAN OR EQUAL TO n MINUS the LENGTH OF word  
                SET start TO the position of word IN s STARTING FROM start  
                IF start EQUALS negative one  
                    EXIT THE WHILE LOOP  
                END IF  
                FOR each i FROM start TO start PLUS the LENGTH OF word MINUS one  
                    SET the element at position i of mask TO TRUE  
                END FOR  
                INCREMENT start BY one  
            END WHILE  
        END FOR  
        
        INITIALIZE result AS empty list  
        SET i TO zero  
        WHILE i LESS THAN n  
            IF the element at position i of mask IS TRUE  
                APPEND the string representing opening bold tag TO result  
                WHILE i LESS THAN n AND the element at position i of mask IS TRUE  
                    APPEND the element at position i of s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND the string representing closing bold tag TO result  
            ELSE  
                APPEND the element at position i of s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        RETURN the concatenation of all elements of result  
    END FUNCTION  
END CLASS
