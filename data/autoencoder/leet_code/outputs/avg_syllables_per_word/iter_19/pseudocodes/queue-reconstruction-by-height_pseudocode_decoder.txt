CLASS Solution  
    FUNCTION reconstructQueue with parameters people  
        CALL sort ON people USING a rule that FOR each element AT position index calculates NEGATIVE of the element at position zero AND the element at position one and ORDERS by the NEGATIVE of the first element FOLLOWED BY the element at position one in ASCENDING ORDER  
        SET queue TO an empty list  
        FOR each person IN people  
            INSERT person AT position equal to the element at position one of person INTO queue  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
