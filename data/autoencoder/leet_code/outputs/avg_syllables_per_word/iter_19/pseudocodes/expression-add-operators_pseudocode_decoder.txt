CLASS Solution  
    FUNCTION addOperators(num, target)  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
              
            FOR i FROM index PLUS one TO the LENGTH OF num  
                SET operand TO substring from position index TO position i MINUS one of num  
                IF the LENGTH OF operand GREATER THAN one AND the element at position zero of operand EQUALS the character zero  
                    CONTINUE to the next iteration  
                END IF  
                  
                IF index EQUALS zero  
                    CALL dfs with parameters i, operand, the INTEGER VALUE OF operand, the INTEGER VALUE OF operand  
                ELSE  
                    CALL dfs with parameters i, the concatenation of path PLUS the character PLUS PLUS operand, value PLUS the INTEGER VALUE OF operand, the INTEGER VALUE OF operand  
                    CALL dfs with parameters i, the concatenation of path PLUS the character MINUS PLUS operand, value MINUS the INTEGER VALUE OF operand, negative of the INTEGER VALUE OF operand  
                    CALL dfs with parameters i, the concatenation of path PLUS the character MULTIPLICATION SIGN PLUS operand, value MINUS prev PLUS prev MULTIPLIED BY the INTEGER VALUE OF operand, prev MULTIPLIED BY the INTEGER VALUE OF operand  
                END IF  
            END FOR  
        END FUNCTION  
          
        SET result TO an empty list  
        CALL dfs with parameters zero, empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
