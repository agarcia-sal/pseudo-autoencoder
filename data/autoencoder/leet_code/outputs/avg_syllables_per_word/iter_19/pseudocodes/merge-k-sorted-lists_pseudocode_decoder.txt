CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS None  
        SET the val PROPERTY OF self TO val  
        SET the next PROPERTY OF self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values AS list  
    IF values EQUALS empty list  
        RETURN None  
    END IF  
    SET head TO a new ListNode INITIALIZED WITH the first element of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the end  
        SET node TO a new ListNode INITIALIZED WITH val  
        SET the next PROPERTY OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS None AND p2 EQUALS None  
        RETURN True  
    END IF  
    IF p1 EQUALS None OR p2 EQUALS None  
        RETURN False  
    END IF  
    RETURN the val PROPERTY OF p1 EQUALS the val PROPERTY OF p2 AND is_same_list WITH PARAMETERS the next PROPERTY OF p1 AND the next PROPERTY OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists WITH PARAMETERS lists AS list OF optional ListNode  
        SET min_heap TO empty list  
        FOR each index i AND element lst IN lists WITH enumeration  
            IF lst IS NOT None  
                ADD a tuple CONTAINING the val PROPERTY OF lst AND i AND lst ITSELF TO min_heap PRESERVING min_heap PROPERTY AS MINIMUM HEAP  
            END IF  
        END FOR  
        SET dummy TO a new ListNode WITH default parameters  
        SET current TO dummy  
        WHILE min_heap IS NOT empty  
            REMOVE the tuple CONTAINING val AND i AND node WITH the smallest val FROM min_heap  
            SET the next PROPERTY OF current TO node  
            SET current TO the next PROPERTY OF current  
            IF the next PROPERTY OF node IS NOT None  
                ADD a tuple CONTAINING the val PROPERTY OF the next PROPERTY OF node AND i AND the next PROPERTY OF node TO min_heap PRESERVING min_heap PROPERTY AS MINIMUM HEAP  
            END IF  
        END WHILE  
        RETURN the next PROPERTY OF dummy  
    END FUNCTION  
END CLASS
