CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET iterator TO an iterator over the characters of s  
            RETURN the result of the test where for each character in word the character is contained in the remaining elements of iterator  
        END FUNCTION  
        
        SORT the dictionary so that words are ordered by descending length and ascending lexicographical order  
        
        FOR each word IN dictionary  
            IF is_subsequence called with word and s RETURNS true  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN empty string  
    END FUNCTION  
END CLASS
