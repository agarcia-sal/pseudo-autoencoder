CLASS Solution  
    FUNCTION wordBreak WITH PARAMETERS s AND wordDict  
        SET wordSet TO the SET OF all elements IN wordDict  
        SET memo TO an EMPTY DICTIONARY  
        
        FUNCTION backtrack WITH PARAMETER start  
            IF start IS A KEY IN memo  
                RETURN the VALUE ASSOCIATED WITH start IN memo  
            END IF  
            IF start EQUALS the LENGTH OF s  
                RETURN a LIST CONTAINING the EMPTY STRING  
            END IF  
            
            SET res TO an EMPTY LIST  
            FOR each end FROM start PLUS one TO the LENGTH OF s INCLUSIVE  
                SET word TO the SUBSTRING from position start TO end MINUS one OF s  
                IF word IS AN ELEMENT OF wordSet  
                    FOR each sentence IN backtrack WITH ARGUMENT end  
                        IF sentence IS AN EMPTY STRING  
                            APPEND word TO res  
                        ELSE  
                            APPEND the CONCATENATION of word AND a SPACE AND sentence TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            
            SET memo AT KEY start TO res  
            RETURN res  
        END FUNCTION  
        
        RETURN backtrack WITH ARGUMENT zero  
    END FUNCTION  
END CLASS
