CLASS Solution  
    FUNCTION longestDecomposition WITH PARAMETER text AS string  
        SET left TO zero  
        SET right TO the LENGTH OF text MINUS one  
        SET left_part TO empty string  
        SET right_part TO empty string  
        SET k TO zero  
        
        WHILE left IS LESS THAN right  
            APPEND element at position left OF text TO left_part  
            SET right_part TO element at position right OF text CONCATENATED BEFORE right_part  
            
            IF left_part IS EQUALS TO right_part  
                INCREMENT k BY two  
                SET left_part TO empty string  
                SET right_part TO empty string  
            END IF  
            
            INCREMENT left BY one  
            DECREMENT right BY one  
        END WHILE  
        
        IF left_part IS NOT empty OR left IS EQUALS TO right  
            INCREMENT k BY one  
        END IF  
        
        RETURN k  
    END FUNCTION  
END CLASS
