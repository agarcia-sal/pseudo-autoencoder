CLASS Solution  
    FUNCTION longestAwesome WITH PARAMETER s  
        SET prefix_map TO a mapping with key zero assigned to value negative one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each position i AND character char IN s  
            SET digit TO the integer value of character char  
            SET prefix_xor TO prefix_xor EXCLUSIVE OR with one LEFT SHIFTED BY digit places  
            
            IF prefix_xor IS a key in prefix_map  
                SET max_length TO the GREATER VALUE BETWEEN max_length AND position i MINUS the element at key prefix_xor in prefix_map  
            END IF  
            
            FOR each integer j FROM zero TO nine  
                SET flipped TO prefix_xor EXCLUSIVE OR with one LEFT SHIFTED BY j places  
                IF flipped IS a key in prefix_map  
                    SET max_length TO the GREATER VALUE BETWEEN max_length AND position i MINUS the element at key flipped in prefix_map  
                END IF  
            END FOR  
            
            IF prefix_xor IS NOT a key in prefix_map  
                ASSIGN to prefix_map at key prefix_xor the value position i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
