CLASS Solution  
    FUNCTION characterReplacement(s parameter OF TYPE string, k parameter OF TYPE integer) RETURNS integer  
        INITIALIZE count AS a new mapping from characters TO integers with default zero  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the element at position right of s in count BY one  
            SET max_count TO the GREATER OF max_count AND the element at position right of s in count  
            
            IF the VALUE OF right PLUS one MINUS left MINUS max_count IS GREATER THAN k THEN  
                DECREMENT the element at position left of s in count BY one  
                INCREMENT left BY one  
            END IF  
            
            SET max_length TO the GREATER OF max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
