CLASS Solution  
    FUNCTION recoverArray(nums)  
        CALL the METHOD sort ON nums  
        SET n TO the LENGTH OF nums  
        SET first TO the element at position zero of nums  
        FOR i FROM one TO n MINUS one  
            SET k TO the element at position i of nums MINUS first THEN DIVIDED BY two  
            IF k IS LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k IS NOT EQUAL TO the element at position i of nums  
                CONTINUE TO the NEXT iteration  
            END IF  
            SET count TO a new Counter OBJECT initialized with nums  
            SET arr TO an empty list  
            FOR each x IN nums  
                IF the element at position x of count IS EQUAL TO zero  
                    CONTINUE TO the NEXT iteration  
                END IF  
                IF the element at position x PLUS two MULTIPLIED BY k of count IS EQUAL TO zero  
                    BREAK OUT OF the FOR loop  
                END IF  
                APPEND x PLUS k TO arr  
                DECREMENT the element at position x of count BY one  
                DECREMENT the element at position x PLUS two MULTIPLIED BY k of count BY one  
            END FOR  
            IF the LENGTH OF arr IS EQUAL TO n DIVIDED BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an empty list  
    END FUNCTION  
END CLASS
