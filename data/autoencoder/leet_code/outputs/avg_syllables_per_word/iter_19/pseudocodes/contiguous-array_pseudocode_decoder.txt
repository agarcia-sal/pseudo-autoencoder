CLASS Solution  
    FUNCTION findMaxLength(nums)  
        SET prefix_sum_index TO a dictionary with key zero mapping to value negative one  
        SET max_length TO zero  
        SET count TO zero  
        FOR each index i AND element num IN nums with enumeration  
            IF num EQUALS one  
                INCREMENT count BY one  
            ELSE  
                DECREMENT count BY one  
            END IF  
            IF count IS A KEY IN prefix_sum_index  
                SET max_length TO the GREATER OF max_length AND the DIFFERENCE BETWEEN i AND value corresponding to count in prefix_sum_index  
            ELSE  
                SET value corresponding to count in prefix_sum_index TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
