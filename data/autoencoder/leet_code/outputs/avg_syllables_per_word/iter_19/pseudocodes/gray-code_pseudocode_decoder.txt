CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN a list containing zero  
        END IF  
        SET prev_gray TO the result of calling the grayCode FUNCTION with parameter n MINUS one  
        SET mask TO one LEFT SHIFTED BY n MINUS one positions  
        SET current_gray TO prev_gray FOLLOWED BY a list formed by performing bitwise OR of mask WITH each number IN the reverse order of prev_gray  
        RETURN current_gray  
    END FUNCTION  
END CLASS
