CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        SET dp TO a two dimensional list with m PLUS one rows and n PLUS one columns filled with zeroes  
        FOR each i FROM zero TO m  
            SET element at position zero of the element at position i of dp TO i  
        END FOR  
        FOR each j FROM zero TO n  
            SET element at position j of the element at position zero of dp TO j  
        END FOR  
        FOR each i FROM one TO m  
            FOR each j FROM one TO n  
                IF the element at position i MINUS one of word1 EQUALS the element at position j MINUS one of word2  
                    SET element at position j of the element at position i of dp TO the element at position j MINUS one of the element at position i MINUS one of dp  
                ELSE  
                    SET deletion_cost TO the element at position j of the element at position i MINUS one of dp  
                    SET insertion_cost TO the element at position j MINUS one of the element at position i of dp  
                    SET replacement_cost TO the element at position j MINUS one of the element at position i MINUS one of dp  
                    SET minimum_cost TO the MINIMUM OF deletion_cost, insertion_cost, AND replacement_cost  
                    SET element at position j of the element at position i of dp TO one PLUS minimum_cost  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position n of the element at position m of dp  
    END FUNCTION  
END CLASS
