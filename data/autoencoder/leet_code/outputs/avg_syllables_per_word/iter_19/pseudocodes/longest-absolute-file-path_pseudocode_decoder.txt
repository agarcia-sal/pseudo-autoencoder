CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO the result of splitting input BY newline  
        SET stack TO an empty list  
        SET max_length TO zero  
        
        FOR each line IN lines  
            SET depth TO the number of tab characters contained in line  
            SET name TO the substring of line starting from position depth TO the end  
            
            WHILE the length of stack IS GREATER THAN depth  
                REMOVE the last element FROM stack  
            END WHILE  
            
            IF the length of stack IS GREATER THAN zero  
                SET current_length TO the element at the last position of stack PLUS the length of name PLUS one  
            ELSE  
                SET current_length TO the length of name  
            END IF  
            
            IF the character representing a period IS CONTAINED IN name  
                SET max_length TO the greater value between max_length AND current_length  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
