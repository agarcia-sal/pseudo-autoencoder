CLASS Solution  
    FUNCTION mirrorReflection(p integer, q integer) RETURNS integer  
        SET greatest_common_divisor TO the greatest common divisor of p AND q  
        SET least_common_multiple TO p MULTIPLIED BY q DIVIDED BY greatest_common_divisor  
        SET vertical_reflections TO least_common_multiple DIVIDED BY p  
        SET horizontal_reflections TO least_common_multiple DIVIDED BY q  
        IF the remainder of vertical_reflections DIVIDED BY two EQUALS zero THEN  
            RETURN zero  
        ELSE IF the remainder of horizontal_reflections DIVIDED BY two EQUALS one THEN  
            RETURN one  
        ELSE  
            RETURN two  
        END IF  
    END FUNCTION  
END CLASS
