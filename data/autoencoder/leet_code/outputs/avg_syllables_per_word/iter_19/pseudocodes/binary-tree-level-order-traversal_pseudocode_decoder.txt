CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO new TreeNode WITH parameter element at position zero of values  
    SET i TO one  
    SET queue TO a new double ended queue initialized empty  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot left TO new TreeNode WITH parameter element at position i of values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node dot right TO new TreeNode WITH parameter element at position i of values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN the logical AND of is_same_tree WITH parameters p dot left AND q dot left AND is_same_tree WITH parameters p dot right AND q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root IS none  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a new double ended queue initialized with root as its single element  
        WHILE queue IS NOT EMPTY  
            SET level_size TO the LENGTH OF queue  
            SET current_level TO an empty list  
            FOR each index FROM zero TO level_size MINUS one  
                SET node TO the element removed from the front of queue  
                APPEND node dot val TO current_level  
                IF node dot left IS NOT none  
                    APPEND node dot left TO queue  
                END IF  
                IF node dot right IS NOT none  
                    APPEND node dot right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
