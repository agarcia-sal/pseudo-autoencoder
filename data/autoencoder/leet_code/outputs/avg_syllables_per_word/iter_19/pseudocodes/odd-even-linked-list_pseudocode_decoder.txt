CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS none  
        SET attribute val of self TO val  
        SET attribute next of self TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values LIST  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO a new ListNode initialized with element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values starting from position one TO end  
        SET node TO a new ListNode initialized with val  
        SET attribute next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN attribute val of p1 EQUALS attribute val of p2 AND CALL is_same_list WITH PARAMETERS attribute next of p1 attribute next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList WITH PARAMETERS self head OPTIONAL ListNode  
        IF head IS none OR attribute next of head IS none OR attribute next of attribute next of head IS none  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO attribute next of head  
        SET even_head TO even  
        WHILE even IS NOT none AND attribute next of even IS NOT none  
            SET attribute next of odd TO attribute next of even  
            SET odd TO attribute next of odd  
            SET attribute next of even TO attribute next of odd  
            SET even TO attribute next of even  
        END WHILE  
        SET attribute next of odd TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
