CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN True  
            END IF  
            
            SET expected TO the STRING REPRESENTATION OF first PLUS second  
            SET length TO the LENGTH OF expected  
            
            IF start PLUS length GREATER THAN the LENGTH OF num OR the SUBSTRING OF num FROM start TO start PLUS length DOES NOT EQUAL expected  
                RETURN False  
            END IF  
            
            RETURN the RESULT OF is_valid_sequence WITH PARAMETERS start PLUS length, second, THE INTEGER VALUE OF expected  
        END FUNCTION  
        
        SET n TO the LENGTH OF num  
        FOR i FROM ONE TO the INTEGER VALUE OF n DIVIDED BY two PLUS ONE  
            FOR j FROM i PLUS ONE TO n MINUS ONE  
                SET first TO the SUBSTRING OF num FROM POSITION zero TO i  
                SET second TO the SUBSTRING OF num FROM POSITION i TO j  
                
                IF the LENGTH OF first GREATER THAN one AND the ELEMENT AT POSITION zero OF first EQUALS the CHARACTER zero OR the LENGTH OF second GREATER THAN one AND the ELEMENT AT POSITION zero OF second EQUALS the CHARACTER zero  
                    CONTINUE TO the NEXT ITERATION  
                END IF  
                
                IF the RESULT OF is_valid_sequence WITH PARAMETERS j, THE INTEGER VALUE OF first, THE INTEGER VALUE OF second  
                    RETURN True  
                END IF  
            END FOR  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
