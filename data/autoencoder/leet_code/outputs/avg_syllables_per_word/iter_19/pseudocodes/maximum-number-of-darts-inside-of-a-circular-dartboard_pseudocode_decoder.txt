CLASS Solution  
    FUNCTION numPoints(darts, r)  
        FUNCTION distance(p1, p2)  
            RETURN the SQUARE ROOT OF the SUM OF the SQUARE OF the DIFFERENCE BETWEEN element at position zero of p1 AND element at position zero of p2 AND the SQUARE OF the DIFFERENCE BETWEEN element at position one of p1 AND element at position one of p2  
        END FUNCTION  

        FUNCTION circle_center(p1, p2, r)  
            SET d TO the RESULT OF CALLING distance WITH p1 AND p2  
            IF d IS GREATER THAN two TIMES r  
                RETURN None  
            END IF  
            SET a TO the SQUARE ROOT OF r MULTIPLIED BY r MINUS the SQUARE OF d DIVIDED BY four  
            SET h TO a LIST CONTAINING the MEAN OF element at position zero of p1 AND element at position zero of p2 AND the MEAN OF element at position one of p1 AND element at position one of p2  
            SET dx TO the DIFFERENCE BETWEEN element at position one of p2 AND element at position one of p1 MULTIPLIED BY a DIVIDED BY d  
            SET dy TO the DIFFERENCE BETWEEN element at position zero of p2 AND element at position zero of p1 MULTIPLIED BY a DIVIDED BY d  
            RETURN a LIST CONTAINING a TUPLE OF h at position zero MINUS dx AND h at position one PLUS dy AND a TUPLE OF h at position zero PLUS dx AND h at position one MINUS dy  
        END FUNCTION  

        FUNCTION is_inside(circle, point, r)  
            RETURN the RESULT OF distance BETWEEN circle AND point IS LESS THAN OR EQUAL TO r  
        END FUNCTION  

        SET n TO the LENGTH OF darts  
        IF n EQUALS one  
            RETURN one  
        END IF  

        SET max_count TO one  

        FOR i FROM zero TO n MINUS one  
            FOR j FROM i PLUS one TO n MINUS one  
                SET centers TO the RESULT OF circle_center WITH element at position i of darts AND element at position j of darts AND r  
                IF centers IS NOT None  
                    FOR each center IN centers  
                        SET count TO zero  
                        FOR each dart IN darts  
                            IF is_inside WITH center AND dart AND r RETURNS True  
                                INCREMENT count BY one  
                            END IF  
                        END FOR  
                        IF count IS GREATER THAN max_count  
                            SET max_count TO count  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  

        RETURN max_count  
    END FUNCTION  
END CLASS
