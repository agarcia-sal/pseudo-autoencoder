CLASS Solution  
    FUNCTION numMatchingSubseq(s AS string, words AS list of string) RETURNS integer  
        SET waiting TO a dictionary that maps each unique character from s TO an empty double-ended queue  
        FOR each word IN words  
            SET it TO the iterator over word  
            SET first_letter TO the next character from it  
            IF first_letter is a key in waiting  
                APPEND it TO the double-ended queue mapped by first_letter in waiting  
            END IF  
        END FOR  
        SET match_count TO zero  
        FOR each character c IN s  
            IF c is a key in waiting  
                SET current_iterators TO the double-ended queue mapped by c in waiting  
                SET the double-ended queue mapped by c in waiting TO a new empty double-ended queue  
                FOR each iterator it IN current_iterators  
                    SET next_char TO the next character from it or to None if there is no next character  
                    IF next_char is None  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char is a key in waiting  
                            APPEND it TO the double-ended queue mapped by next_char in waiting  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        RETURN match_count  
    END FUNCTION  
END CLASS
