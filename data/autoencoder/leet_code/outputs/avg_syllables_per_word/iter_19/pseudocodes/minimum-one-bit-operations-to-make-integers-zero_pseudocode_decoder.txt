CLASS Solution  
    FUNCTION minimumOneBitOperations(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE one LEFT SHIFTED BY the value of k PLUS one IS LESS THAN OR EQUAL TO n  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO one LEFT SHIFTED BY the value of k  
        RETURN one LEFT SHIFTED BY the value of k PLUS one MINUS one MINUS minimumOneBitOperations THE VALUE OF n EXCLUSIVE OR most_significant_bit  
    END FUNCTION  
END CLASS
