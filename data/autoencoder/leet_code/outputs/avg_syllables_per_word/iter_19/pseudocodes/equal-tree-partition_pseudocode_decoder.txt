CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero left EQUALS None right EQUALS None  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new TreeNode initialized WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue structure  
    APPEND root TO queue  
    WHILE queue holds elements  
        SET node TO the first element REMOVED FROM queue  
        IF i is LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node left TO a new TreeNode initialized WITH the element at position i of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node right TO a new TreeNode initialized WITH the element at position i of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p is None AND q is None  
        RETURN True  
    ELSE IF p is None OR q is None  
        RETURN False  
    ELSE IF the value of p IS NOT EQUAL TO the value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS the left child of p AND the left child of q AND the result of is_same_tree WITH PARAMETERS the right child of p AND the right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION checkEqualTree WITH PARAMETERS self root  
        FUNCTION subtree_sum WITH PARAMETERS node  
            IF node is None  
                RETURN zero  
            END IF  
            RETURN the value of node PLUS the result of subtree_sum WITH PARAMETERS the left child of node PLUS the result of subtree_sum WITH PARAMETERS the right child of node  
        END FUNCTION  

        SET total_sum TO the result of subtree_sum WITH PARAMETERS root  

        IF total_sum MODULO two IS NOT EQUAL TO zero  
            RETURN False  
        END IF  

        SET target TO total_sum DIVIDED BY two  
        SET found TO a list containing False as its only element  

        FUNCTION can_split WITH PARAMETERS node  
            IF node is None  
                RETURN zero  
            END IF  
            SET current_sum TO the value of node PLUS the result of can_split WITH PARAMETERS the left child of node PLUS the result of can_split WITH PARAMETERS the right child of node  
            IF current_sum EQUALS target AND node IS NOT EQUAL TO root  
                SET the element at position zero of found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION  

        CALL can_split WITH PARAMETERS root  
        RETURN the element at position zero of found  
    END FUNCTION  
END CLASS
