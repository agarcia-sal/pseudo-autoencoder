CLASS Solution  
    FUNCTION kSmallestPairs(nums1 AS List of integers, nums2 AS List of integers, k AS integer) RETURNS List of List of integers  
        IF the LENGTH OF nums1 EQUALS zero OR the LENGTH OF nums2 EQUALS zero OR k LESS THAN OR EQUAL TO zero THEN  
            RETURN an empty list  
        END IF  
        
        SET min_heap TO an empty list  
        FOR each index i FROM zero TO the MINIMUM of k MINUS one AND the LENGTH OF nums1 MINUS one  
            PUSH a tuple consisting of the SUM of the element at position i of nums1 AND the element at position zero of nums2, i, and zero INTO min_heap USING heap push operation  
        END FOR  
        
        SET result TO an empty list  
        WHILE min_heap is NOT empty AND the LENGTH OF result LESS THAN k  
            POP the smallest tuple FROM min_heap USING heap pop operation AND ASSIGN its components TO _ COMMA i COMMA j  
            APPEND a list consisting of the element at position i of nums1 AND the element at position j of nums2 TO result  
            IF j PLUS one LESS THAN the LENGTH OF nums2 THEN  
                PUSH a tuple consisting of the SUM of the element at position i of nums1 AND the element at position j PLUS one of nums2, i, and j PLUS one INTO min_heap USING heap push operation  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
