CLASS Solution  
    FUNCTION combinationSum(candidates, target)  
        FUNCTION backtrack(start, target, path)  
            IF target EQUALS zero  
                APPEND path TO result  
                RETURN  
            END IF  
            IF target LESS THAN zero  
                RETURN  
            END IF  
            FOR index FROM start TO the LENGTH OF candidates MINUS one  
                CALL backtrack WITH arguments index AND target MINUS element at position index of candidates AND path APPENDED WITH element at position index of candidates  
            END FOR  
        END FUNCTION  

        SET result TO empty list  
        CALL backtrack WITH arguments zero AND target AND empty list  
        RETURN result  
    END FUNCTION  
END CLASS
