CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET i TO the LENGTH OF nums MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND element at position i of nums GREATER THAN OR EQUAL TO element at position i PLUS one of nums holds  
            DECREMENT i BY one  
        END WHILE  
        IF i GREATER THAN OR EQUAL TO zero THEN  
            SET j TO the LENGTH OF nums MINUS one  
            WHILE element at position j of nums LESS THAN OR EQUAL TO element at position i of nums holds  
                DECREMENT j BY one  
            END WHILE  
            SET temporary_value TO element at position i of nums  
            SET element at position i of nums TO element at position j of nums  
            SET element at position j of nums TO temporary_value  
        END IF  
        SET sublist TO the sequence of elements of nums from position i PLUS one TO the LENGTH OF nums  
        SET reversed_sublist TO the reverse order of sublist  
        FOR index FROM zero TO the LENGTH OF reversed_sublist MINUS one  
            SET element at position i PLUS one PLUS index of nums TO element at position index of reversed_sublist  
        END FOR  
    END FUNCTION  
END CLASS
