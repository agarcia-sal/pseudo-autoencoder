CLASS Solution  
    FUNCTION customSortString(order, s)  
        CREATE dictionary d TO assign each character c in order TO its index i according to enumeration of order  
        SORT string s BY the custom order defined as the value at character x in dictionary d OR zero if character is not found in dictionary d  
        JOIN all characters of the sorted sequence into one string  
        RETURN the resulting string  
    END FUNCTION  
END CLASS
