CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty LIST OF INTEGER, d INTEGER) RETURNS INTEGER  
        SET n TO the LENGTH OF jobDifficulty  
        IF n IS LESS THAN d  
            RETURN negative one  
        END IF  
        
        FUNCTION dp(i INTEGER, d INTEGER) RETURNS INTEGER  
            IF d EQUALS one  
                RETURN the MAXIMUM value FROM the ELEMENTS starting AT position i TO the END OF jobDifficulty  
            END IF  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR j FROM i TO n MINUS d PLUS one  
                SET current_max TO the GREATER OF current_max AND the ELEMENT at position j OF jobDifficulty  
                SET min_difficulty TO the LESSER OF min_difficulty AND current_max PLUS the RESULT OF dp WITH PARAMETERS j PLUS one AND d MINUS one  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN the RESULT OF dp WITH PARAMETERS zero AND d  
    END FUNCTION  
END CLASS
