CLASS Solution  
    FUNCTION longestSubstring(s, k)  
        IF the LENGTH OF s LESS THAN k THEN  
            RETURN zero  
        END IF  
        ASSIGN pointer TO a frequency counter OF the characters in s  
        FOR each character IN the frequency counter  
            IF the frequency of character LESS THAN k THEN  
                RETURN the MAXIMUM value AMONG the APPLICATION of longestSubstring TO each substring OBTAINED BY splitting s BY character  
            END IF  
        END FOR  
        RETURN the LENGTH OF s  
    END FUNCTION  
END CLASS
