CLASS Solution  
    FUNCTION findShortestWay WITH PARAMETERS maze AS List OF List OF integer, ball AS List OF integer, hole AS List OF integer  
        SET directions TO list consisting of element with zero AND one AND the character r, element with one AND zero AND the character d, element with zero AND minus one AND the character l, element with minus one AND zero AND the character u  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero of maze  
        SET start TO tuple conversion of ball  
        SET hole TO tuple conversion of hole  
        SET priority queue pq TO list consisting of element zero, empty string, AND start  
        SET visited TO empty set  
        
        WHILE pq is not empty  
            EXTRACT the element with smallest priority FROM pq ASSIGNING distance TO dist, path string to path, AND position tuple to x AND y  
            
            IF x EQUALS the first element of hole AND y EQUALS the second element of hole  
                RETURN path  
            END IF  
            
            IF the tuple consisting of x AND y IS IN visited  
                CONTINUE to next iteration of WHILE  
            END IF  
            
            ADD the tuple consisting of x AND y TO visited  
            
            FOR each element dx, dy, direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of maze AT position new_y PLUS dy EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                    IF new_x EQUALS the first element of hole AND new_y EQUALS the second element of hole  
                        BREAK from WHILE  
                    END IF  
                END WHILE  
                
                IF the tuple consisting of new_x AND new_y IS NOT IN visited  
                    INSERT element consisting of dist PLUS count, concatenation of path AND direction, AND the tuple consisting of new_x AND new_y INTO priority queue pq  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN the string impossible  
    END FUNCTION  
END CLASS
