CLASS Solution  
    FUNCTION exclusiveTime WITH PARAMETERS n AND logs  
        SET exclusive_time TO an empty list of size n filled with zero  
        SET stack TO an empty list  
        
        FOR each log IN logs  
            SET function_id TO the element at position zero of the result of splitting log by a colon converted to integer  
            SET action TO the element at position one of the result of splitting log by a colon  
            SET timestamp TO the element at position two of the result of splitting log by a colon converted to integer  
            
            IF action EQUALS the word start  
                IF the LENGTH OF stack IS greater than zero  
                    SET prev_function_id TO the first element of the element at the last position of stack  
                    SET prev_timestamp TO the second element of the element at the last position of stack  
                    INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND a tuple of function_id AND timestamp TO stack  
            ELSE  
                SET prev_function_id TO the first element of the element popped from stack  
                SET prev_timestamp TO the second element of the element popped from stack  
                INCREMENT the element at position prev_function_id of exclusive_time BY timestamp MINUS prev_timestamp PLUS one  
                IF the LENGTH OF stack IS greater than zero  
                    SET the element at the last position of stack TO a tuple of the first element of the element at the last position of stack AND timestamp PLUS one  
                END IF  
            END IF  
        END FOR  
        
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
