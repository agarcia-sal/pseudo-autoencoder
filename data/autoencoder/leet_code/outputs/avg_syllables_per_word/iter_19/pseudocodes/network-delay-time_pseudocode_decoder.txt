CLASS Solution  
    FUNCTION networkDelayTime(times AS List OF List OF int, n AS int, k AS int) RETURNS int  
        SET graph TO a dictionary mapping each integer from one TO n TO an empty list  
        FOR each element consisting of u v w IN times  
            APPEND the pair of v and w TO the list at key u IN graph  
        END FOR  

        SET dist TO a dictionary mapping each integer from one TO n TO positive infinity  
        SET the value at key k IN dist TO zero  

        SET min_heap TO a list containing the pair of zero and k  

        WHILE min_heap is not empty  
            REMOVE and retrieve the pair with smallest first element FROM min_heap INTO current_dist and node  

            IF current_dist IS GREATER THAN the value at key node IN dist  
                CONTINUE to the next iteration of WHILE  
            END IF  

            FOR each pair of neighbor and weight IN the list at key node IN graph  
                SET distance TO current_dist PLUS weight  

                IF distance IS LESS THAN the value at key neighbor IN dist  
                    SET the value at key neighbor IN dist TO distance  
                    ADD the pair of distance and neighbor TO min_heap  
                END IF  
            END FOR  
        END WHILE  

        SET max_time TO the maximum value AMONG all values IN dist  

        IF max_time IS EQUAL TO positive infinity  
            RETURN negative one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
