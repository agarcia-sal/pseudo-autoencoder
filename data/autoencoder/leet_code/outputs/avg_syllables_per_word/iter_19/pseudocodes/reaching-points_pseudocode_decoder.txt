CLASS Solution  
    FUNCTION reachingPoints(sx AS integer, sy AS integer, tx AS integer, ty AS integer) RETURNS boolean  
        WHILE tx IS GREATER THAN OR EQUAL TO sx AND ty IS GREATER THAN OR EQUAL TO sy  
            IF tx IS EQUAL TO sx AND ty IS EQUAL TO sy  
                RETURN True  
            END IF  
            IF tx IS GREATER THAN ty  
                IF ty IS GREATER THAN sy  
                    SET tx TO the remainder of tx DIVIDED BY ty  
                ELSE  
                    RETURN the remainder of tx MINUS sx DIVIDED BY ty IS EQUAL TO zero  
                END IF  
            ELSE  
                IF tx IS GREATER THAN sx  
                    SET ty TO the remainder of ty DIVIDED BY tx  
                ELSE  
                    RETURN the remainder of ty MINUS sy DIVIDED BY tx IS EQUAL TO zero  
                END IF  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
