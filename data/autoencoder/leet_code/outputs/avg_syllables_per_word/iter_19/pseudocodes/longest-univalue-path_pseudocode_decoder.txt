CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUAL TO zero, left EQUAL TO none, right EQUAL TO none  
        SET the val ATTRIBUTE OF current object TO val  
        SET the left ATTRIBUTE OF current object TO left  
        SET the right ATTRIBUTE OF current object TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETERS values  
    IF the values collection IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue IS NOT empty  
        SET node TO the element REMOVED from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the left ATTRIBUTE OF node TO a new TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the left ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET the right ATTRIBUTE OF node TO a new TreeNode INITIALIZED WITH the element at position i of values  
            APPEND the right ATTRIBUTE OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the val ATTRIBUTE OF p IS NOT EQUALS TO the val ATTRIBUTE OF q  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS the left ATTRIBUTE OF p AND the left ATTRIBUTE OF q AND the RESULT OF is_same_tree WITH PARAMETERS the right ATTRIBUTE OF p AND the right ATTRIBUTE OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestUnivaluePath WITH PARAMETERS root  
        FUNCTION dfs WITH PARAMETERS node  
            IF node IS none  
                RETURN zero AND zero  
            END IF  
            SET left_length AND left_value TO the RESULT OF dfs WITH PARAMETERS the left ATTRIBUTE OF node  
            SET right_length AND right_value TO the RESULT OF dfs WITH PARAMETERS the right ATTRIBUTE OF node  
            SET left_arrow TO zero  
            SET right_arrow TO zero  
            IF the left ATTRIBUTE OF node IS NOT none AND the val ATTRIBUTE OF the left ATTRIBUTE OF node EQUALS TO the val ATTRIBUTE OF node  
                SET left_arrow TO left_length PLUS one  
            END IF  
            IF the right ATTRIBUTE OF node IS NOT none AND the val ATTRIBUTE OF the right ATTRIBUTE OF node EQUALS TO the val ATTRIBUTE OF node  
                SET right_arrow TO right_length PLUS one  
            END IF  
            SET the max_length ATTRIBUTE OF current object TO the GREATER BETWEEN the current max_length ATTRIBUTE OF current object AND the sum of left_arrow AND right_arrow  
            RETURN the GREATER BETWEEN left_arrow AND right_arrow AND the val ATTRIBUTE OF node  
        END FUNCTION  
        SET the max_length ATTRIBUTE OF current object TO zero  
        CALL dfs WITH PARAMETERS root  
        RETURN the max_length ATTRIBUTE OF current object  
    END FUNCTION  
END CLASS
