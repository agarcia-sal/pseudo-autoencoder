CLASS Solution  
    FUNCTION merge(intervals)  
        CALL sort intervals BY the first element in each interval  
        SET merged TO an empty list  
        FOR each interval IN intervals  
            IF the merged list is empty OR the element at position two of the element at the last position of merged IS LESS THAN the element at position one of interval  
                APPEND interval TO merged  
            ELSE  
                SET the element at position two of the element at the last position of merged TO the GREATER VALUE BETWEEN the element at position two of the element at the last position of merged AND the element at position two of interval  
            END IF  
        END FOR  
        RETURN merged  
    END FUNCTION  
END CLASS
