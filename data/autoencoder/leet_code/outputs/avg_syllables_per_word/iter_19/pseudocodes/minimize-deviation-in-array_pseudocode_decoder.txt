CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO positive infinity  
        FOR each num IN nums  
            IF remainder of num DIVIDED BY two EQUALS one  
                SET num TO num MULTIPLIED BY two  
            END IF  
            IF min_value GREATER THAN num  
                SET min_value TO num  
            END IF  
            PUSH negative of num INTO max_heap USING heap insertion  
        END FOR  

        SET min_deviation TO positive infinity  

        WHILE remainder of element at position zero of max_heap DIVIDED BY two EQUALS zero  
            SET max_value TO negative of the element obtained by popping from max_heap using heap removal  
            IF min_deviation GREATER THAN max_value MINUS min_value  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY two using integer division  
            IF min_value GREATER THAN max_value  
                SET min_value TO max_value  
            END IF  
            PUSH negative of max_value INTO max_heap USING heap insertion  
        END WHILE  

        IF min_deviation GREATER THAN negative of element at position zero of max_heap MINUS min_value  
            SET min_deviation TO negative of element at position zero of max_heap MINUS min_value  
        END IF  

        RETURN min_deviation  
    END FUNCTION  
END CLASS
