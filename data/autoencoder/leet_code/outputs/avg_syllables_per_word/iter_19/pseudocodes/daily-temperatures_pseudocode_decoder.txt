CLASS Solution  
    FUNCTION dailyTemperatures WITH parameter temperatures  
        SET n TO the LENGTH OF temperatures  
        SET answer TO a list OF zero VALUES WITH SIZE n  
        SET stack TO an empty list  
        FOR each index i FROM zero TO n MINUS one  
            WHILE stack IS NOT empty AND the VALUE at position i of temperatures IS GREATER THAN the VALUE at position at the LAST element of stack in temperatures  
                SET prev_index TO the REMOVAL of the LAST element FROM stack  
                SET the ELEMENT at position prev_index of answer TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
