CLASS Solution  
    FUNCTION nearestPalindromic(n)  
        SET length TO the LENGTH OF n  
        IF length EQUALS one  
            RETURN the STRING REPRESENTATION OF the INTEGER VALUE OF n MINUS one  
        END IF  
        
        SET candidates TO an EMPTY SET  
        
        ADD the NUMBER TEN RAISED TO THE POWER length PLUS one TO candidates  
        ADD the NUMBER TEN RAISED TO THE POWER length MINUS one MINUS one TO candidates  
        
        SET prefix TO the INTEGER VALUE OF the SUBSTRING from position zero TO the INTEGER VALUE OF length PLUS one DIVIDED BY two of n  
        
        FOR i FROM negative one TO two  
            SET new_prefix TO the STRING REPRESENTATION OF prefix PLUS i  
            IF length MODULO two EQUALS zero  
                SET candidate TO the CONCATENATION OF new_prefix AND the REVERSE OF new_prefix  
            ELSE  
                SET candidate TO the CONCATENATION OF new_prefix AND the REVERSE OF the SUBSTRING from POSITION the LENGTH OF new_prefix MINUS two DOWN TO zero of new_prefix  
            END IF  
            ADD the INTEGER VALUE OF candidate TO candidates  
        END FOR  
        
        SET original_number TO the INTEGER VALUE OF n  
        REMOVE original_number FROM candidates IF it EXISTS  
        
        SET closest TO the ELEMENT IN candidates WITH THE MINIMAL VALUE OF the TUPLE OF the ABSOLUTE VALUE OF the ELEMENT MINUS original_number AND the ELEMENT ITSELF  
        
        RETURN the STRING REPRESENTATION OF closest  
    END FUNCTION  
END CLASS
