CLASS Solution  
    FUNCTION findItinerary WITH parameters tickets  
        SET graph TO a new default dictionary with lists as default values  
        FOR each pair consisting of departure_airport AND arrival_airport IN the tickets sorted in ascending alphabetical order and then reversed  
            APPEND arrival_airport TO the list at key departure_airport in graph  
        END FOR  
        
        SET itinerary TO an empty list  
        
        FUNCTION visit WITH parameter airport  
            WHILE the list at key airport in graph is not empty  
                CALL visit WITH argument the element obtained by removing the last item from the list at key airport in graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit WITH argument the string consisting of letter J letter F letter K  
        RETURN the reversed itinerary list  
    END FUNCTION  
END CLASS
