CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO empty list  
        SET lower_bound TO negative infinity  
        FOR each value IN preorder  
            IF value IS LESS THAN lower_bound  
                RETURN False  
            END IF  
            WHILE stack IS NOT empty AND value IS GREATER THAN element at the last position of stack  
                SET lower_bound TO element popped from stack  
            END WHILE  
            APPEND value TO stack  
        END FOR  
        RETURN True  
    END FUNCTION  
END CLASS
