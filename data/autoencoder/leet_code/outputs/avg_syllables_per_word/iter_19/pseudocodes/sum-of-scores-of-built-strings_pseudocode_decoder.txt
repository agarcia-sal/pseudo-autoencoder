CLASS Solution  
    FUNCTION sumScores(s)  
        SET n TO the LENGTH OF s  
        SET lcp TO a LIST of zero of LENGTH n  
        SET l TO zero  
        SET r TO zero  
        FOR i FROM one TO n MINUS one  
            IF i IS LESS THAN OR EQUAL TO r  
                SET lcp AT position i TO the MINIMUM OF r MINUS i PLUS one AND lcp AT position i MINUS l  
            END IF  
            WHILE i PLUS lcp AT position i IS LESS THAN n AND the CHARACTER AT position lcp AT s IS EQUAL TO the CHARACTER AT position i PLUS lcp AT s  
                INCREMENT lcp AT position i BY one  
            END WHILE  
            IF i PLUS lcp AT position i MINUS one IS GREATER THAN r  
                SET l TO i  
                SET r TO i PLUS lcp AT position i MINUS one  
            END IF  
        END FOR  
        RETURN the SUM OF all ELEMENTS IN lcp PLUS n  
    END FUNCTION  
END CLASS
