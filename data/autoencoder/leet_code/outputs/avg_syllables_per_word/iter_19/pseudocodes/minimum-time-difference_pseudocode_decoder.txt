CLASS Solution  
    FUNCTION findMinDifference(timePoints)  
        FUNCTION to_minutes(time)  
            SET hours TO the integer converted from the substring before the first colon of time  
            SET minutes TO the integer converted from the substring after the first colon of time  
            RETURN hours MULTIPLIED BY sixty PLUS minutes  
        END FUNCTION  
        
        SET minutes_list TO the list obtained by applying to_minutes to each element IN timePoints  
        SORT minutes_list in ascending order  
        SET min_diff TO infinity  
        FOR index FROM one TO the length of minutes_list MINUS one  
            SET current_difference TO the element at position index of minutes_list MINUS the element at position index MINUS one of minutes_list  
            SET min_diff TO the smaller of min_diff AND current_difference  
        END FOR  
        SET circular_difference TO the sum of one thousand four hundred forty MINUS the last element of minutes_list AND the first element of minutes_list  
        SET min_diff TO the smaller of min_diff AND circular_difference  
        RETURN min_diff  
    END FUNCTION  
END CLASS
