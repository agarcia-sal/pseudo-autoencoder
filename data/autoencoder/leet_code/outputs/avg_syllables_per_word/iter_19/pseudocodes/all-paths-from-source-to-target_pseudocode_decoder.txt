CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        SET q TO a double ended queue containing a single list with zero as its only element  
        SET ans TO an empty list  
        WHILE the queue q holds elements  
            REMOVE the first element from the queue q and ASSIGN it TO path  
            SET u TO the element at the last position of path  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE to the next iteration of the WHILE loop  
            END IF  
            FOR each v IN the element at position u of graph  
                APPEND the concatenation of path AND a list containing v TO the queue q  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
