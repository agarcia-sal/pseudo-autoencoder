CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        FOR each token IN tokens  
            IF token EQUALS PLUS OR token EQUALS MINUS OR token EQUALS MULTIPLY OR token EQUALS DIVIDE  
                SET b TO element removed from the top of stack  
                SET a TO element removed from the top of stack  
                IF token EQUALS PLUS  
                    APPEND sum of a AND b TO stack  
                ELSE IF token EQUALS MINUS  
                    APPEND difference of a MINUS b TO stack  
                ELSE IF token EQUALS MULTIPLY  
                    APPEND product of a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS DIVIDE  
                    APPEND integer quotient of a DIVIDED BY b truncating TOWARDS zero TO stack  
                END IF  
            ELSE  
                APPEND integer conversion of token TO stack  
            END IF  
        END FOR  
        RETURN element at position zero of stack  
    END FUNCTION  
END CLASS
