CLASS Solution  
    FUNCTION ambiguousCoordinates WITH PARAMETER s  
        FUNCTION make WITH PARAMETER frag  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO the SUBSTRING from position one TO d of frag  
                SET right TO the SUBSTRING from position d PLUS one TO n of frag  
                IF left DOES NOT START WITH character zero OR left EQUALS character zero  
                AND right DOES NOT END WITH character zero  
                    RETURN left CONCATENATED WITH a decimal point IF d IS NOT EQUAL TO n OTHERWISE empty string CONCATENATED WITH right  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO the SUBSTRING from position two TO the LENGTH OF s MINUS one of s  
        SET result TO an empty list  
        FOR i FROM one TO the LENGTH OF s MINUS one  
            FOR each pair a and b IN the Cartesian product of the results of make applied to the SUBSTRING from position one TO i of s AND make applied to the SUBSTRING from position i PLUS one TO the LENGTH OF s of s  
                APPEND an open parenthesis CONCATENATED WITH a CONCATENATED WITH a comma AND a space CONCATENATED WITH b CONCATENATED WITH a close parenthesis TO result  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
