CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO a list of LENGTH twenty six with each element SET TO negative one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each position i AND character char IN s  
            SET index TO the Unicode code point of character char MINUS the Unicode code point of character a  
            INCREMENT current_appeal BY position i MINUS element at position index of list last  
            INCREMENT total_appeal BY current_appeal  
            SET element at position index of list last TO position i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
