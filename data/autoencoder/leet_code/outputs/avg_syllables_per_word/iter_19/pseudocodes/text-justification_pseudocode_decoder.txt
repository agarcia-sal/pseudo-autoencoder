CLASS Solution  
    FUNCTION fullJustify(words AS List of strings, maxWidth AS integer) RETURNS List of strings  
        FUNCTION justify_line(line AS List of strings, num_of_letters AS integer, is_last_line AS boolean) RETURNS string  
            IF the LENGTH OF line EQUALS one OR is_last_line IS true  
                RETURN the concatenation of all elements in line separated by a single space PLUS a sequence of space characters with the LENGTH equal to maxWidth MINUS num_of_letters MINUS the LENGTH OF line PLUS one  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO the INTEGER DIVISION of total_spaces BY num_of_gaps  
                SET extra_spaces TO the MODULO of total_spaces BY num_of_gaps  
                FOR index FROM zero TO num_of_gaps MINUS one  
                    IF index LESS THAN extra_spaces  
                        SET the element at position index of line TO the original element AT position index PLUS a sequence of space characters with LENGTH equal to spaces_between_words PLUS one  
                    ELSE  
                        SET the element at position index of line TO the original element AT position index PLUS a sequence of space characters with LENGTH equal to spaces_between_words  
                    END IF  
                END FOR  
                RETURN the concatenation of all elements in line without any separator  
            END IF  
        END FUNCTION  
        
        SET result TO an empty list of strings  
        SET line TO an empty list of strings  
        SET num_of_letters TO zero  
        
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth  
                APPEND the result of justify_line with parameters line num_of_letters false TO result  
                SET line TO an empty list of strings  
                SET num_of_letters TO zero  
            END IF  
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
        
        APPEND the result of justify_line with parameters line num_of_letters true TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
