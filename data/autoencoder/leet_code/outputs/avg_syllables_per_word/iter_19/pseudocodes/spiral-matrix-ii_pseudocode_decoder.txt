CLASS Solution  
    FUNCTION generateMatrix WITH parameter n  
        SET matrix TO a list consisting of n elements each of which is a list of n zeros  
        SET top TO zero  
        SET bottom TO n MINUS one  
        SET left TO zero  
        SET right TO n MINUS one  
        SET num TO one  
        WHILE top IS LESS THAN OR EQUAL TO bottom AND left IS LESS THAN OR EQUAL TO right HOLD  
            FOR i FROM left TO right  
                SET the element at position top of matrix at position i TO num  
                INCREMENT num BY one  
            END FOR  
            INCREMENT top BY one  
            FOR i FROM top TO bottom  
                SET the element at position i of matrix at position right TO num  
                INCREMENT num BY one  
            END FOR  
            DECREMENT right BY one  
            IF top IS LESS THAN OR EQUAL TO bottom THEN  
                FOR i FROM right TO left STEPPING BY negative one  
                    SET the element at position bottom of matrix at position i TO num  
                    INCREMENT num BY one  
                END FOR  
                DECREMENT bottom BY one  
            END IF  
            IF left IS LESS THAN OR EQUAL TO right THEN  
                FOR i FROM bottom TO top STEPPING BY negative one  
                    SET the element at position i of matrix at position left TO num  
                    INCREMENT num BY one  
                END FOR  
                INCREMENT left BY one  
            END IF  
        END WHILE  
        RETURN matrix  
    END FUNCTION  
END CLASS
