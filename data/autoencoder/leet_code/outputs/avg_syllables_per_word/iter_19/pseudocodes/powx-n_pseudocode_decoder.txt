CLASS Solution  
    FUNCTION myPow(x FLOAT, n INTEGER) RETURNS FLOAT  
        FUNCTION qpow(a FLOAT, n INTEGER) RETURNS FLOAT  
            SET ans TO one  
            WHILE n IS NOT EQUAL TO zero  
                IF the binary AND of n and one IS EQUAL TO one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO the INTEGER division of n BY two discarding the remainder  
            END WHILE  
            RETURN ans  
        END FUNCTION  
        IF n IS GREATER THAN OR EQUAL TO zero  
            RETURN the result of calling qpow WITH x AND n  
        ELSE  
            RETURN one DIVIDED BY the result of calling qpow WITH x AND the negative of n  
        END IF  
    END FUNCTION  
END CLASS
