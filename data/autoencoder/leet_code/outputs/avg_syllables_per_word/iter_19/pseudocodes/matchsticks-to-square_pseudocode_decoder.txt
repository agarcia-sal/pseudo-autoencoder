CLASS Solution  
    FUNCTION makesquare(matchsticks)  
        SET total_length TO the SUM OF all elements IN matchsticks  
        IF total_length MODULO four NOT EQUALS zero  
            RETURN False  
        END IF  
        SET side_length TO total_length DIVIDED BY four  
        SORT matchsticks IN descending order  
        FUNCTION can_form_sides(sides, index)  
            IF index EQUALS the LENGTH OF matchsticks  
                RETURN the RESULT OF checking IF every side IN sides EQUALS side_length  
            END IF  
            FOR i FROM zero TO three  
                IF element at position i OF sides PLUS element at position index OF matchsticks LESS THAN OR EQUAL TO side_length  
                    INCREMENT element at position i OF sides BY element at position index OF matchsticks  
                    IF can_form_sides(sides, index PLUS one) RETURNS True  
                        RETURN True  
                    END IF  
                    DECREMENT element at position i OF sides BY element at position index OF matchsticks  
                END IF  
                IF element at position i OF sides EQUALS zero  
                    BREAK the loop  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        RETURN the RESULT OF can_form_sides a list of four zeros and zero  
    END FUNCTION  
END CLASS
