CLASS Solution  
    FUNCTION maxJumps(arr, d)  
        FUNCTION dp(i)  
            SET max_jumps TO one  
            FOR x FROM one TO d  
                IF the sum of i and x IS LESS THAN the LENGTH OF arr  
                    IF the element at position the sum of i and x of arr IS GREATER THAN OR EQUAL TO the element at position i of arr  
                        BREAK the loop  
                    END IF  
                    SET max_jumps TO the greater VALUE BETWEEN max_jumps AND the sum of one AND dp of the sum of i and x  
                END IF  
            END FOR  
            FOR x FROM one TO d  
                IF the difference of i and x IS GREATER THAN OR EQUAL TO zero  
                    IF the element at position the difference of i and x of arr IS GREATER THAN OR EQUAL TO the element at position i of arr  
                        BREAK the loop  
                    END IF  
                    SET max_jumps TO the greater VALUE BETWEEN max_jumps AND the sum of one AND dp of the difference of i and x  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        RETURN the greatest VALUE OF dp of i FOR each i FROM zero TO the difference BETWEEN the LENGTH OF arr AND one  
    END FUNCTION  
END CLASS
