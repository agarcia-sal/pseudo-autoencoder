CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        INITIALIZE visited AS an empty set  
        INITIALIZE min_heap WITH a tuple consisting of element at position zero at position zero of grid AND zero AND zero  
        ADD the tuple consisting of zero AND zero TO visited  
        SET directions TO list of tuples consisting of zero AND one AND one AND zero AND zero AND negative one AND negative one AND zero  
        
        WHILE min_heap is NOT empty  
            REMOVE the smallest element from min_heap AND ASSIGN its components TO t AND i AND j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            
            FOR each pair di AND dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND the tuple consisting of ni AND nj IS NOT IN visited  
                    ADD the tuple consisting of ni AND nj TO visited  
                    COMPUTE new_time AS the GREATER OF t AND element at position ni at position nj of grid  
                    INSERT the tuple consisting of new_time AND ni AND nj INTO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
