CLASS Solution  
    FUNCTION findStrobogrammatic(n)  
        FUNCTION dfs(u)  
            IF u EQUALS zero  
                RETURN a list containing the empty string  
            END IF  
            IF u EQUALS one  
                RETURN a list containing the strings zero one and eight  
            END IF  
            SET ans TO an empty list  
            FOR each string v IN the result of calling dfs with the argument u MINUS two  
                FOR each pair l AND r IN the list of string pairs eleven eighty-eight sixty-nine ninety-six  
                    APPEND the concatenation of l AND v AND r TO ans  
                END FOR  
                IF u IS NOT EQUAL TO n  
                    APPEND the concatenation of zero AND v AND zero TO ans  
                END IF  
            END FOR  
            RETURN ans  
        END FUNCTION  
        RETURN the result of calling dfs with the argument n  
    END FUNCTION  
END CLASS
