CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MOD TO one billion PLUS seven  
        SORT nums IN ASCENDING ORDER  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        CREATE power AS a LIST OF n ELEMENTS ALL SET TO one  
        
        FOR index FROM one TO n MINUS one  
            SET element at position index OF power TO the REMAINDER OF the PRODUCT of element at position index MINUS one OF power MULTIPLIED BY two DIVIDED BY MOD  
        END FOR  
        
        FOR index FROM zero TO n MINUS one  
            SET total_sum TO the REMAINDER OF the SUM OF total_sum AND the PRODUCT of element at position index OF nums MULTIPLIED BY the DIFFERENCE BETWEEN element at position index OF power MINUS element at position n MINUS index MINUS one OF power DIVIDED BY MOD  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
