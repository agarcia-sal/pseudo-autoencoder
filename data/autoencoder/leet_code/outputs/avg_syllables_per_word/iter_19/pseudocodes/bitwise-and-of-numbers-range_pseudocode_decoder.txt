CLASS Solution  
    FUNCTION rangeBitwiseAnd WITH PARAMETERS left INTEGER right INTEGER  
        SET shift TO zero  
        WHILE left IS LESS THAN right  
            SET left TO left DIVIDED BY two USING RIGHT SHIFT BY ONE POSITION  
            SET right TO right DIVIDED BY two USING RIGHT SHIFT BY ONE POSITION  
            INCREMENT shift BY one  
        END WHILE  
        SET result TO left MULTIPLIED BY two RAISED TO THE POWER OF shift USING LEFT SHIFT BY shift POSITIONS  
        RETURN result  
    END FUNCTION  
END CLASS
