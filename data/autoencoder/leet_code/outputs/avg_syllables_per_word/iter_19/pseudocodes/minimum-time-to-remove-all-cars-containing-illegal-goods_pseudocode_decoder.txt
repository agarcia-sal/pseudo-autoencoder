CLASS Solution  
    FUNCTION minimumTime WITH PARAMETER s  
        SET n TO the LENGTH OF s  
        SET left_costs TO a LIST OF zero AMOUNT n  
        SET right_costs TO a LIST OF zero AMOUNT n  

        IF the ELEMENT at position zero of s EQUALS one  
            SET the ELEMENT at position zero of left_costs TO one  
        ELSE  
            SET the ELEMENT at position zero of left_costs TO zero  
        END IF  

        FOR i FROM one TO n MINUS one  
            IF the ELEMENT at position i of s EQUALS zero  
                SET the ELEMENT at position i of left_costs TO the ELEMENT at position i MINUS one of left_costs  
            ELSE  
                SET the ELEMENT at position i of left_costs TO the MINIMUM VALUE BETWEEN i PLUS one AND the ELEMENT at position i MINUS one of left_costs PLUS two  
            END IF  
        END FOR  

        IF the ELEMENT at position n MINUS one of s EQUALS one  
            SET the ELEMENT at position n MINUS one of right_costs TO one  
        ELSE  
            SET the ELEMENT at position n MINUS one of right_costs TO zero  
        END IF  

        FOR i FROM n MINUS two DOWN TO zero  
            IF the ELEMENT at position i of s EQUALS zero  
                SET the ELEMENT at position i of right_costs TO the ELEMENT at position i PLUS one of right_costs  
            ELSE  
                SET the ELEMENT at position i of right_costs TO the MINIMUM VALUE BETWEEN n MINUS i AND the ELEMENT at position i PLUS one of right_costs PLUS two  
            END IF  
        END FOR  

        SET min_time TO the MINIMUM VALUE BETWEEN the ELEMENT at position n MINUS one of left_costs AND the ELEMENT at position zero of right_costs  

        FOR i FROM zero TO n MINUS two  
            SET min_time TO the MINIMUM VALUE BETWEEN min_time AND the SUM of the ELEMENT at position i of left_costs AND the ELEMENT at position i PLUS one of right_costs  
        END FOR  

        RETURN min_time  
    END FUNCTION  
END CLASS
