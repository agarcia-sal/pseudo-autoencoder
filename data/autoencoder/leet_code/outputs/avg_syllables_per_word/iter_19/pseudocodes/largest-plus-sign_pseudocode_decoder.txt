CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO the SET of each element converted to a TUPLE from mines  
        SET dp TO a TWO DIMENSIONAL COLLECTION of size n BY n where each element is a LIST containing zero, zero, zero, zero representing four directions  
        SET directions TO the LIST of pairs zero one one zero zero MINUS ONE MINUS ONE zero representing right down left up  
        
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF the PAIR of i and j is NOT IN mines_set THEN  
                    IF j IS GREATER THAN zero THEN  
                        SET element at position zero of element at position j of element at position i of dp TO element at position zero of element at position j MINUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i IS GREATER THAN zero THEN  
                        SET element at position one of element at position j of element at position i of dp TO element at position one of element at position j of element at position i MINUS one of dp PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        
        FOR i FROM n MINUS one TO zero WITH STEP MINUS one  
            FOR j FROM n MINUS one TO zero WITH STEP MINUS one  
                IF the PAIR of i and j is NOT IN mines_set THEN  
                    IF j IS LESS THAN n MINUS one THEN  
                        SET element at position two of element at position j of element at position i of dp TO element at position two of element at position j PLUS one of element at position i of dp PLUS one  
                    END IF  
                    IF i IS LESS THAN n MINUS one THEN  
                        SET element at position three of element at position j of element at position i of dp TO element at position three of element at position j of element at position i PLUS one of dp PLUS one  
                    END IF  
                    SET plus_sign_order TO the MINIMUM of elements at position i and j of dp  
                    SET result TO the MAXIMUM of result AND plus_sign_order  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
