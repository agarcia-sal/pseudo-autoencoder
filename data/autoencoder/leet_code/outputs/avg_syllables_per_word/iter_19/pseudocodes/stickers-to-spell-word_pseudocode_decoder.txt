CLASS Solution  
    FUNCTION minStickers(stickers AS List OF String, target AS String) RETURNS Integer  
        SET stickers TO THE LIST OBTAINED BY CONVERTING EACH element OF stickers TO A frequency COUNTER OBJECT  
        SORT stickers BY THE NEGATIVE OF THE HIGHEST FREQUENCY VALUE OF EACH sticker  
        SET queue TO A DEQUE INITIALIZED WITH A PAIR CONTAINING target AND ZERO  
        SET memo TO A DICTIONARY WITH KEY target AND VALUE ZERO  
        WHILE queue IS NOT EMPTY  
            REMOVE AND RETURN THE LEFTMOST ELEMENT OF queue INTO current_target AND used_stickers  
            FOR each sticker IN stickers  
                IF THE FIRST CHARACTER OF current_target IS NOT PRESENT AS A KEY IN sticker THEN  
                    CONTINUE TO THE NEXT iteration OF THE LOOP  
                END IF  
                SET remaining TO A COPY OF THE COUNTER OF current_target  
                SUBTRACT sticker COUNTS FROM remaining COUNTER  
                SET new_target TO THE STRING FORMED BY JOINING ALL ELEMENTS OF remaining IN SORTED ORDER  
                IF new_target IS EMPTY STRING THEN  
                    RETURN used_stickers PLUS ONE  
                END IF  
                IF new_target IS NOT A KEY IN memo THEN  
                    SET memo new_target TO used_stickers PLUS ONE  
                    APPEND A PAIR CONTAINING new_target AND used_stickers PLUS ONE TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN NEGATIVE ONE  
    END FUNCTION  
END CLASS
