CLASS Solution  
    FUNCTION xorGame(nums)  
        SET total_xor TO zero  
        FOR each num IN nums  
            SET total_xor TO total_xor EXCLUSIVE OR WITH num  
        END FOR  
        IF total_xor EQUALS zero  
            RETURN true  
        END IF  
        IF the LENGTH OF nums MODULO two EQUALS zero  
            RETURN true  
        END IF  
        RETURN false  
    END FUNCTION  
END CLASS
