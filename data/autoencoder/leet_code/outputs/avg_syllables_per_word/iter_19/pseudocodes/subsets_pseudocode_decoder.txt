CLASS Solution  
    FUNCTION subsets(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN a LIST containing an EMPTY LIST  
        END IF  
        SET subsets_without_first TO the RESULT OF CALLING subsets WITH the ELEMENTS from position two TO the END OF nums  
        SET subsets_with_first TO an EMPTY LIST  
        FOR each subset IN subsets_without_first  
            SET new_subset TO a LIST STARTING WITH the ELEMENT at position one OF nums FOLLOWED BY all ELEMENTS of subset  
            APPEND new_subset TO subsets_with_first  
        END FOR  
        RETURN concatenation OF subsets_without_first AND subsets_with_first  
    END FUNCTION  
END CLASS
