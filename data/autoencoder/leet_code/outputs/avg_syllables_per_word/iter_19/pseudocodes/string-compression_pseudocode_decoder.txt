CLASS Solution  
    FUNCTION compress(chars)  
        IF the LENGTH OF chars EQUALS zero THEN  
            RETURN zero  
        END IF  
        
        SET write_index TO zero  
        SET read_index TO zero  
        
        WHILE read_index LESS THAN the LENGTH OF chars DO  
            SET char TO element at position read_index of chars  
            SET count TO zero  
            
            WHILE read_index LESS THAN the LENGTH OF chars AND element at position read_index of chars EQUALS char DO  
                INCREMENT read_index BY one  
                INCREMENT count BY one  
            END WHILE  
            
            SET element at position write_index of chars TO char  
            INCREMENT write_index BY one  
            
            IF count GREATER THAN one THEN  
                FOR each digit IN the string representation of count  
                    SET element at position write_index of chars TO digit  
                    INCREMENT write_index BY one  
                END FOR  
            END IF  
        END WHILE  
        
        RETURN write_index  
    END FUNCTION  
END CLASS
