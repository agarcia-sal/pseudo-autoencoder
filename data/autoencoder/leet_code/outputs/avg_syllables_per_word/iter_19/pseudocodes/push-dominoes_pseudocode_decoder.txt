CLASS Solution  
    FUNCTION pushDominoes(dominoes)  
        SET n TO the LENGTH OF dominoes  
        SET forces TO a list of zeros WITH LENGTH n  
        
        SET force TO zero  
        FOR index FROM zero TO n MINUS one  
            IF element at position index of dominoes EQUALS the character representing right push  
                SET force TO n  
            ELSE IF element at position index of dominoes EQUALS the character representing left push  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER VALUE BETWEEN force MINUS one AND zero  
            END IF  
            INCREMENT element at position index of forces BY force  
        END FOR  
        
        SET force TO zero  
        FOR index FROM n MINUS one DOWN TO zero  
            IF element at position index of dominoes EQUALS the character representing left push  
                SET force TO n  
            ELSE IF element at position index of dominoes EQUALS the character representing right push  
                SET force TO zero  
            ELSE  
                SET force TO the GREATER VALUE BETWEEN force MINUS one AND zero  
            END IF  
            DECREMENT element at position index of forces BY force  
        END FOR  
        
        SET result TO an empty list  
        FOR each force_value IN forces  
            IF force_value GREATER THAN zero  
                APPEND the character representing right push TO result  
            ELSE IF force_value LESS THAN zero  
                APPEND the character representing left push TO result  
            ELSE  
                APPEND the character representing no push TO result  
            END IF  
        END FOR  
        
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
