CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a list of length two RAISED TO THE POWER OF n where each element is assigned TO infinity  
        SET element at position zero of dp TO zero  
        FOR each mask FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            SET bit_count TO the number of one bits in the binary representation of mask  
            FOR each j FROM zero TO n MINUS one  
                IF the bitwise AND of mask AND two RAISED TO THE POWER OF j IS NOT EQUAL TO zero THEN  
                    SET current_value TO element at position mask of dp  
                    SET candidate_value TO element at position mask XOR two RAISED TO THE POWER OF j of dp PLUS the bitwise XOR of element at position bit_count MINUS one of nums1 AND element at position j of nums2  
                    IF candidate_value IS LESS THAN current_value THEN  
                        SET element at position mask of dp TO candidate_value  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position two RAISED TO THE POWER OF n MINUS one of dp  
    END FUNCTION  
END CLASS
