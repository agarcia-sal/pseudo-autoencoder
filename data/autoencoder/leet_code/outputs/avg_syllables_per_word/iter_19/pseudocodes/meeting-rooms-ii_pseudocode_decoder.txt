CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty list  
            RETURN zero  
        END IF  
        SORT intervals BY element at position zero of each element in ascending order  
        INITIALIZE min_heap TO empty list  
        FOR each meeting IN intervals  
            IF min_heap IS NOT empty AND element at position zero of meeting GREATER THAN OR EQUAL TO element at position zero of min_heap  
                REMOVE the smallest element FROM min_heap  
            END IF  
            ADD element at position one of meeting TO min_heap maintaining the heap property  
        END FOR  
        RETURN the LENGTH OF min_heap  
    END FUNCTION  
END CLASS
