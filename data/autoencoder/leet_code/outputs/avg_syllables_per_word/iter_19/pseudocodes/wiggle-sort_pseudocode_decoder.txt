CLASS Solution  
    FUNCTION wiggleSort(nums)  
        SET less TO true with the meaning that the first element is expected to be less than or equal to the second element  
        FOR index FROM zero TO the LENGTH OF nums MINUS one MINUS one  
            IF less  
                IF the element at position index of nums IS GREATER THAN the element at position index plus one of nums  
                    SET temporary variable TO the element at position index of nums  
                    SET the element at position index of nums TO the element at position index plus one of nums  
                    SET the element at position index plus one of nums TO temporary variable  
                END IF  
            ELSE  
                IF the element at position index of nums IS LESS THAN the element at position index plus one of nums  
                    SET temporary variable TO the element at position index of nums  
                    SET the element at position index of nums TO the element at position index plus one of nums  
                    SET the element at position index plus one of nums TO temporary variable  
                END IF  
            END IF  
            SET less TO the logical negation of less  
        END FOR  
    END FUNCTION  
END CLASS
