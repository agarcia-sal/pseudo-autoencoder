CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO an empty list  
        SET max_value TO negative infinity  
        
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            CALL heap push OPERATION TO add the tuple consisting of the element at position zero of the list at position index of nums the index itself and zero TO min_heap  
            IF max_value IS LESS THAN the element at position zero of the list at position index of nums  
                SET max_value TO the element at position zero of the list at position index of nums  
            END IF  
        END FOR  
        
        SET smallest_range TO positive infinity  
        SET result TO the list consisting of negative one hundred thousand and one hundred thousand  
        
        WHILE min_heap IS NOT EMPTY  
            CALL heap pop OPERATION TO REMOVE AND RETURN the smallest tuple OF min_heap AND ASSIGN the elements TO min_value list_index and element_index respectively  
            
            IF max_value MINUS min_value IS LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO the list consisting of min_value and max_value  
            END IF  
            
            IF element_index PLUS one IS LESS THAN the LENGTH OF the list at position list_index of nums  
                SET next_value TO the element at position element_index PLUS one of the list at position list_index of nums  
                CALL heap push OPERATION TO add the tuple consisting of next_value list_index and element_index PLUS one TO min_heap  
                IF max_value IS LESS THAN next_value  
                    SET max_value TO next_value  
                END IF  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
