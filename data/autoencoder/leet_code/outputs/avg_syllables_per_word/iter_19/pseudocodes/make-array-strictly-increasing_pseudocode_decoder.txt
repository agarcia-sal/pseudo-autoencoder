CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL sort ON arr2 TO ARRANGE elements IN ascending order  
        SET dp TO a dictionary WITH key negative one MAPPED TO zero  
        
        FOR each number IN arr1  
            SET new_dp TO an empty dictionary  
            
            FOR each pair of last_number AND operations IN dp  
                IF number GREATER THAN last_number  
                    IF number IS NOT IN new_dp OR current mapped value FOR number IN new_dp GREATER THAN operations  
                        SET the value mapped BY number IN new_dp TO operations  
                    END IF  
                END IF  
                
                SET index TO the position AFTER last_number FOUND BY PERFORMING binary search ON arr2  
                
                IF index LESS THAN the LENGTH OF arr2  
                    SET candidate_number TO the element at position index OF arr2  
                    SET candidate_operations TO operations PLUS one  
                    IF candidate_number IS NOT IN new_dp OR current mapped value FOR candidate_number IN new_dp GREATER THAN candidate_operations  
                        SET the value mapped BY candidate_number IN new_dp TO candidate_operations  
                    END IF  
                END IF  
            END FOR  
            
            IF new_dp IS empty  
                RETURN negative one  
            END IF  
            
            SET dp TO new_dp  
        END FOR  
        
        RETURN the MINIMUM value AMONG all values IN dp  
    END FUNCTION  
END CLASS
