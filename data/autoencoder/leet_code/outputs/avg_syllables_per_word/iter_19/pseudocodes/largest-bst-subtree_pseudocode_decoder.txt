CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUAL TO zero left EQUAL TO none right EQUAL TO none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETER values AS list  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH PARAMETER element at position one of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty  
        SET node TO element removed from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO NEW TreeNode WITH PARAMETER element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF p.val IS NOT EQUALS TO q.val  
        RETURN False  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH PARAMETERS p.left AND q.left AND the RESULT OF is_same_tree WITH PARAMETERS p.right AND q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION largestBSTSubtree WITH PARAMETERS self root AS optional TreeNode  
        
        FUNCTION dfs WITH PARAMETER node  
            IF node IS none  
                RETURN positive infinity NEGATIVE infinity zero  
            END IF  
            SET left_min left_max left_size TO the RESULT OF dfs WITH PARAMETER node.left  
            SET right_min right_max right_size TO the RESULT OF dfs WITH PARAMETER node.right  
            IF left_max IS LESS THAN node.val AND node.val IS LESS THAN right_min  
                RETURN the MINIMUM OF node.val AND left_min the MAXIMUM OF node.val AND right_max left_size PLUS right_size PLUS one  
            END IF  
            RETURN negative infinity positive infinity the MAXIMUM OF left_size AND right_size  
        END FUNCTION
        
        RETURN the element at position three of the RESULT OF dfs WITH PARAMETER root  
    END FUNCTION  
END CLASS
