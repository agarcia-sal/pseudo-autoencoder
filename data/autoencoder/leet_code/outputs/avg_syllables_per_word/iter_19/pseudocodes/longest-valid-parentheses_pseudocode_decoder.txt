CLASS Solution  
    FUNCTION longestValidParentheses WITH PARAMETER s  
        SET stack TO a new list containing the integer negative one  
        SET max_length TO zero  
        FOR each index i WITH value char IN the enumeration of string s  
            IF char EQUALS the opening parenthesis character  
                APPEND index i TO stack  
            ELSE  
                REMOVE the last element FROM stack  
                IF stack IS empty  
                    APPEND index i TO stack  
                ELSE  
                    SET current_length TO index i MINUS the last element of stack  
                    SET max_length TO the greater value BETWEEN max_length AND current_length  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
