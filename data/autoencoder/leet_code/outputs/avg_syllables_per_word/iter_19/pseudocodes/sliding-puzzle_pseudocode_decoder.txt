CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuples each converted from each row of board  
        SET target TO tuple of tuple with first element one two three and second element four five zero  
        SET neighbors TO dictionary with keys zero mapped to set of one and three  
            one mapped to set of zero two and four  
            two mapped to set of one and five  
            three mapped to set of zero and four  
            four mapped to set of one three and five  
            five mapped to set of two and four  
        
        FOR index i FROM zero TO one  
            FOR index j FROM zero TO two  
                IF element at position j of element at position i of start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        
        SET queue TO new deque containing tuple of start zero_pos and zero  
        SET visited TO new set containing start  
        
        WHILE queue is not empty  
            SET state zero_pos and moves TO the elements dequeued from queue  
            
            IF state EQUALS target  
                RETURN moves  
            END IF  
            
            FOR each neighbor IN the set mapped from zero_pos in neighbors  
                SET new_state TO list conversion of state where each element is converted from tuple to list  
                SET i TO integer division of zero_pos BY three  
                SET j TO zero_pos MODULO three  
                SET ni TO integer division of neighbor BY three  
                SET nj TO neighbor MODULO three  
                
                SET element at position j of element at position i in new_state TO element at position nj of element at position ni in new_state  
                SET element at position nj of element at position ni in new_state TO zero  
                
                SET new_tuple_state TO tuple of tuples each converted from each row of new_state  
                
                IF new_tuple_state NOT IN visited  
                    ADD new_tuple_state TO visited  
                    APPEND tuple of new_tuple_state neighbor moves PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
