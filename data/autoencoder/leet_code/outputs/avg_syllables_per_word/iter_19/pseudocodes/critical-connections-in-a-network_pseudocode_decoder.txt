CLASS Solution  
    FUNCTION criticalConnections(n, connections)  
        FUNCTION dfs(node, parent, disc, low, time, graph, critical)  
            SET element at position node of disc TO element at position node of time  
            SET element at position node of low TO element at position node of time  
            INCREMENT element at position zero of time BY one  
            FOR each neighbor IN element at position node of graph  
                IF element at position neighbor of disc EQUALS minus one THEN  
                    CALL dfs WITH neighbor parent disc low time graph critical  
                    SET element at position node of low TO the LESSER OF element at position node of low AND element at position neighbor of low  
                    IF element at position neighbor of low IS GREATER THAN element at position node of disc THEN  
                        APPEND a list containing node AND neighbor TO critical  
                    END IF  
                ELSE IF neighbor IS NOT EQUAL TO parent THEN  
                    SET element at position node of low TO the LESSER OF element at position node of low AND element at position neighbor of disc  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET graph TO a list of empty lists with a number of elements equal to n  
        FOR each pair u v IN connections  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
        
        SET disc TO a list containing minus one repeated n times  
        SET low TO a list containing minus one repeated n times  
        SET time TO a list containing zero at position zero  
        SET critical TO an empty list  
        
        FOR node FROM zero TO n MINUS one  
            IF element at position node of disc EQUALS minus one THEN  
                CALL dfs WITH node minus one disc low time graph critical  
            END IF  
        END FOR  
        
        RETURN critical  
    END FUNCTION  
END CLASS
