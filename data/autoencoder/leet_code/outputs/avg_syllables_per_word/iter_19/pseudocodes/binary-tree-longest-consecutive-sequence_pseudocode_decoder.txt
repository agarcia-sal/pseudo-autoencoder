CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero left EQUALS None right EQUALS None  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values OF TYPE list  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO A NEW TreeNode INITIALIZED WITH the element at position zero of values  
    SET i TO one  
    SET queue TO A NEW double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element REMOVED FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO A NEW TreeNode INITIALIZED WITH the element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val PROPERTY OF p IS NOT EQUALS TO the val PROPERTY OF q  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS the left PROPERTY OF p AND the left PROPERTY OF q  
        AND is_same_tree WITH PARAMETERS the right PROPERTY OF p AND the right PROPERTY OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive WITH PARAMETERS self root OF TYPE Optional OF TreeNode  
        IF root IS None  
            RETURN zero  
        END IF  
        
        FUNCTION dfs WITH PARAMETERS node parent current_length  
            IF node IS None  
                RETURN current_length  
            END IF  
            
            IF parent IS NOT None AND the val PROPERTY OF node IS EQUALS TO parent PLUS one  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            
            SET left_length TO dfs WITH PARAMETERS the left PROPERTY OF node the val PROPERTY OF node current_length  
            SET right_length TO dfs WITH PARAMETERS the right PROPERTY OF node the val PROPERTY OF node current_length  
            
            RETURN THE MAXIMUM OF current_length left_length AND right_length  
        END FUNCTION  
        
        RETURN dfs WITH PARAMETERS root None zero  
    END FUNCTION  
END CLASS
