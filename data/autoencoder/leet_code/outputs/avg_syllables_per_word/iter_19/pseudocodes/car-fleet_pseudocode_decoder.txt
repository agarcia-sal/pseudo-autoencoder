CLASS Solution  
    FUNCTION carFleet(target, position, speed)  
        SET list_of_time_to_target TO an empty list  
        SET list_of_position_and_speed TO list created by pairing elements at the same positions of position and speed  
        SORT list_of_position_and_speed in order descending by the position element in each pair  
        FOR each pair_of_position_and_speed IN list_of_position_and_speed  
            SET current_position TO the position element of pair_of_position_and_speed  
            SET current_speed TO the speed element of pair_of_position_and_speed  
            SET time_for_current_car TO the difference between target AND current_position DIVIDED BY current_speed  
            APPEND time_for_current_car TO list_of_time_to_target  
        END FOR  
        
        SET fleets TO zero  
        SET current_max_time TO zero  
        
        FOR each time IN list_of_time_to_target  
            IF time GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
        
        RETURN fleets  
    END FUNCTION  
END CLASS
