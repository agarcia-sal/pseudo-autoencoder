CLASS Solution  
    FUNCTION lexicalOrder WITH PARAMETERS n  
        SET result TO empty list  
        
        FUNCTION dfs WITH PARAMETERS current  
            IF current GREATER THAN n THEN  
                RETURN  
            END IF  
            APPEND current TO result  
            FOR each i FROM zero TO nine  
                SET next_num TO current MULTIPLIED BY ten PLUS i  
                IF next_num GREATER THAN n THEN  
                    RETURN  
                END IF  
                CALL dfs WITH next_num  
            END FOR  
        END FUNCTION  
        
        FOR each i FROM one TO nine  
            CALL dfs WITH i  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
