CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        IF THE SUM of m PLUS n MINUS one MODULO two IS NOT EQUALS TO zero  
            RETURN false  
        END IF  
        SET dp TO a three dimensional collection of logical FALSE values with dimensions m PLUS one BY n PLUS one BY m PLUS n PLUS one  
        SET element at position zero one of element at position zero of dp TO true  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR k FROM zero TO m PLUS n  
                    IF element at position i j PLUS one k of dp IS true OR element at position i PLUS one j k of dp IS true  
                        IF element at position i j of grid EQUALS a left parenthesis AND k PLUS one IS LESS THAN m PLUS n PLUS one  
                            SET element at position i PLUS one j PLUS one k PLUS one of dp TO true  
                        ELSE IF element at position i j of grid EQUALS a right parenthesis AND k MINUS one IS GREATER THAN OR EQUAL TO zero  
                            SET element at position i PLUS one j PLUS one k MINUS one of dp TO true  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position m n zero of dp  
    END FUNCTION  
END CLASS
