CLASS Solution  
    FUNCTION maximizeXor(nums AS List of integers, queries AS List of List of integers) RETURNS List of integers  
        SORT nums IN ascending order  
        SET sorted_queries TO THE RESULT OF sorting the enumeration pairs of queries BY the second element of each query pair indicated as the maximum value mi IN ascending order  
        SET result TO a list with the LENGTH OF queries FILLED WITH negative one  
        SET trie TO an empty data structure for bits storage  
        SET num_index TO zero  
        
        FOR each query_index, pair xi AND mi IN sorted_queries  
            WHILE num_index IS LESS THAN the LENGTH OF nums AND the element at position num_index of nums IS LESS THAN OR EQUAL TO mi  
                SET num TO the element at position num_index of nums  
                SET node TO trie  
                FOR i FROM thirty one DOWN TO zero  
                    SET bit TO the value of the bit at position i in num interpreted as the bitwise right shift of num by i positions AND bitwise AND with one  
                    IF bit IS NOT PRESENT in node  
                        ASSIGN pointer or reference TO an empty bit node TO node AT bit  
                    END IF  
                    SET node TO the node AT bit within node  
                END FOR  
                INCREMENT num_index BY one  
            END WHILE  
            
            IF trie IS empty  
                CONTINUE to the next query iteration  
            END IF  
            
            SET node TO trie  
            SET max_xor TO zero  
            FOR i FROM thirty one DOWN TO zero  
                SET bit TO the value of the bit at position i in xi computed as the bitwise right shift of xi by i positions AND bitwise AND with one  
                SET toggle_bit TO one MINUS bit  
                IF toggle_bit IS PRESENT in node  
                    SET max_xor TO max_xor OR the bitwise one shifted left by i positions  
                    SET node TO the node AT toggle_bit within node  
                ELSE  
                    SET node TO the node AT bit within node  
                END IF  
            END FOR  
            
            SET the element at position query_index of result TO max_xor  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
