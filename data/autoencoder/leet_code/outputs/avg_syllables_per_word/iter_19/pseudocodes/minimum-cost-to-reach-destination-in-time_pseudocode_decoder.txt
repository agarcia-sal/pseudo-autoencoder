CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO empty list REPEATED n TIMES  
        FOR each element x, y, time IN edges  
            APPEND pair of y AND time TO element at position x of graph  
            APPEND pair of x AND time TO element at position y of graph  
        END FOR  
        
        SET pq TO list containing a triple made of element at position zero of passingFees, zero, zero  
        SET visited_time TO list of infinity REPEATED n TIMES  
        SET element at position zero of visited_time TO zero  
        
        WHILE pq IS NOT empty  
            REMOVE AND ASSIGN TO current_cost, current_city, current_time the element with lowest first value from pq  
            
            IF current_city EQUALS n MINUS one  
                RETURN current_cost  
            END IF  
            
            FOR each neighbor, travel_time IN element at position current_city of graph  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS element at position neighbor of passingFees  
                
                IF new_time IS LESS THAN OR EQUAL TO maxTime AND (new_time IS LESS THAN element at position neighbor of visited_time OR new_cost IS LESS THAN current_cost)  
                    SET element at position neighbor of visited_time TO new_time  
                    ADD triple of new_cost, neighbor, new_time TO pq  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
