CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        SET secret_count TO a dictionary with default zero for any new key  
        SET guess_count TO a dictionary with default zero for any new key  
        FOR each pair of characters s and g taken simultaneously from secret and guess  
            IF character s EQUALS character g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT the value associated with character s in secret_count BY one  
                INCREMENT the value associated with character g in guess_count BY one  
            END IF  
        END FOR  
        FOR each digit key IN the keys of secret_count  
            INCREMENT cows BY the lesser value between the value associated with digit key in secret_count AND the value associated with digit key in guess_count  
        END FOR  
        RETURN a string consisting of bulls converted to text followed by the character A followed by cows converted to text followed by the character B  
    END FUNCTION  
END CLASS
