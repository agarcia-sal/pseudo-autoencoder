CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS empty OR element at position zero of matrix EQUALS empty  
            RETURN zero  
        END IF  
        
        SET m TO LENGTH OF matrix  
        SET n TO LENGTH OF element at position zero of matrix  
        SET dp TO auxiliary FUNCTION initialize two dimensional list of zeros with rows EQUAL TO m PLUS one AND columns EQUAL TO n PLUS one  
        SET max_side TO zero  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of matrix EQUALS element at position j MINUS one of string ONE  
                    SET minimum_value TO minimum OF element at position i MINUS one AND position j of dp AND element at position i AND position j MINUS one of dp AND element at position i MINUS one AND position j MINUS one of dp  
                    SET element at position i AND j of dp TO minimum_value PLUS one  
                    SET max_side TO maximum OF max_side AND element at position i AND j of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
