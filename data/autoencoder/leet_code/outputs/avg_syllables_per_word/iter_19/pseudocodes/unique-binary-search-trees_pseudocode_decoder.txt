CLASS Solution  
    FUNCTION numTrees(n)  
        SET dp TO a list of zeros of size n PLUS one  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO one  
        FOR i FROM two TO n  
            FOR j FROM one TO i  
                SET dp element at position i TO dp element at position i PLUS dp element at position j MINUS one MULTIPLIED BY dp element at position i MINUS j  
            END FOR  
        END FOR  
        RETURN dp element at position n  
    END FUNCTION  
END CLASS
