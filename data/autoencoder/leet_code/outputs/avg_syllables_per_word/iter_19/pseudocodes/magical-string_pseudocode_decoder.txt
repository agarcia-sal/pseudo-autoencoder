CLASS Solution  
    FUNCTION magicalString(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        IF n LESS THAN OR EQUAL TO three  
            RETURN one  
        END IF  

        SET s TO list containing one followed by two followed by two  
        SET count TO one  
        SET index TO two  

        WHILE the LENGTH OF s LESS THAN n  
            SET next_num TO three MINUS the element at position last of s  
            CALL extend s BY next_num repeated element at position index of s times  
            IF next_num EQUALS one  
                INCREMENT count BY element at position index of s  
            END IF  
            INCREMENT index BY one  
        END WHILE  

        IF the LENGTH OF s GREATER THAN n  
            SET count TO count MINUS element at position last of s MULTIPLIED BY the difference of the LENGTH OF s MINUS n  
        END IF  

        RETURN count  
    END FUNCTION  
END CLASS
