CLASS Solution  
    FUNCTION canSeePersonsCount with parameters heights  
        SET n TO the LENGTH OF heights  
        SET answer TO a list of zeroes with size n  
        SET stack TO an empty list  
        
        FOR index i FROM zero TO n MINUS one  
            WHILE stack is not empty AND the element at position element at last position in stack of heights IS LESS THAN the element at position i of heights holds  
                SET popped_index TO REMOVE last element FROM stack  
                INCREMENT the element at position popped_index of answer BY one  
            END WHILE  
            IF stack is not empty THEN  
                INCREMENT the element at position element at last position in stack of answer BY one  
            END IF  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
