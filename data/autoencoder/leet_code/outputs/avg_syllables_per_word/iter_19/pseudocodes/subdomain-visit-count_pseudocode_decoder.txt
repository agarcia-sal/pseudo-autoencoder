CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO a dictionary that returns zero for missing keys  
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain INTO count_string AND domain_string by whitespace  
            SET count TO the integer conversion of count_string  
            SPLIT domain_string INTO subdomains by the period character  
            FOR index FROM zero TO the LENGTH OF subdomains MINUS one  
                JOIN the elements of subdomains FROM position index TO the end INTO subdomain using the period character as separator  
                INCREMENT the value at key subdomain in domain_count BY count  
            END FOR  
        END FOR  
        SET result_list TO an empty list  
        FOR each domain_key AND count_value IN domain_count items  
            CREATE a string consisting of count_value FOLLOWED BY a space FOLLOWED BY domain_key  
            APPEND that string TO result_list  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
