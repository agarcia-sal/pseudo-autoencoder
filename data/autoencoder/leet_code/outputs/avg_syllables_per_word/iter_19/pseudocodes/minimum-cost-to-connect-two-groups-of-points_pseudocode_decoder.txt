CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF element at position zero of cost  
        SET dp TO a two dimensional list of size size1 plus one BY two to the power of size2 where each element IS infinity  
        SET element at position zero of dp at position zero TO zero  
        
        FOR i FROM zero TO size1 MINUS one  
            FOR mask FROM zero TO two to the power of size2 MINUS one  
                FOR j FROM zero TO size2 MINUS one  
                    COMPUTE new_mask AS bitwise OR of mask WITH two to the power of j  
                    SET element at position i plus one of dp at position new_mask TO the MINIMUM of element at position i plus one of dp at position new_mask AND element at position i of dp at position mask PLUS element at position i of cost at position j  
                END FOR  
                
                FOR j FROM zero TO size2 MINUS one  
                    IF bitwise AND of mask WITH two to the power of j IS NOT equal TO zero  
                        COMPUTE mask_without_j AS bitwise XOR of mask WITH two to the power of j  
                        SET element at position i plus one of dp at position mask TO the MINIMUM of element at position i plus one of dp at position mask AND element at position i plus one of dp at position mask_without_j PLUS element at position i of cost at position j  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position size1 of dp at position two to the power of size2 MINUS one  
    END FUNCTION  
END CLASS
