CLASS Solution  
    FUNCTION smallestDistancePair(nums AS List OF integers, k AS integer) RETURNS integer  
        SORT the list nums IN ascending order  
        SET n TO the LENGTH OF nums  
        
        FUNCTION count(mid AS integer) RETURNS integer  
            SET count TO zero  
            SET left TO zero  
            FOR right FROM zero TO n MINUS one  
                WHILE the element at position right of nums MINUS the element at position left of nums IS GREATER THAN mid  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT count BY right MINUS left  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the element at the last position of nums MINUS the element at the first position of nums  
        WHILE left IS LESS THAN right  
            SET mid TO the INTEGER DIVISION OF the sum of left AND right BY two  
            IF count(mid) IS LESS THAN k  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        
        RETURN left  
    END FUNCTION  
END CLASS
