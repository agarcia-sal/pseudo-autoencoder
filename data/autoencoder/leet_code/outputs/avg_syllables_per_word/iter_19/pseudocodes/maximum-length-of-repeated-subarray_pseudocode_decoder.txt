CLASS Solution  
    FUNCTION findLength(nums1, nums2)  
        CREATE a two dimensional list named dp WITH zero values  
        THE LENGTH OF dp IS the LENGTH OF nums1 PLUS one BY the LENGTH OF nums2 PLUS one  
        SET max_length TO zero  
        FOR i FROM one TO the LENGTH OF nums1  
            FOR j FROM one TO the LENGTH OF nums2  
                IF the element at position i MINUS one of nums1 EQUALS the element at position j MINUS one of nums2  
                    SET the element at position i BY position j of dp TO the element at position i MINUS one BY position j MINUS one of dp PLUS one  
                    SET max_length TO the maximum BETWEEN max_length AND the element at position i BY position j of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
