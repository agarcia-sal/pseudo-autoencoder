CLASS Solution  
    FUNCTION leastInterval(tasks AS list OF strings, n AS integer) AS integer  
        SET task_counts TO the frequency COUNTER of tasks  
        SET max_heap TO an empty list  
        FOR each count IN the VALUES OF task_counts  
            APPEND the NEGATION OF count TO max_heap  
        END FOR  
        APPLY heapify TO max_heap  
        
        SET time TO zero  
        SET queue TO an empty list representing pairs of time for readiness PLUS task count  
        
        WHILE the LENGTH OF max_heap IS GREATER THAN zero OR the LENGTH OF queue IS GREATER THAN zero  
            WHILE the LENGTH OF queue IS GREATER THAN zero AND the first element at position zero of queue in its first component EQUALS time  
                REMOVE the first element FROM queue AND ASSIGN to a pair of variables unnamed time_marker AND count  
                IF count IS LESS THAN zero  
                    APPLY heappush TO max_heap WITH count  
                END IF  
            END WHILE  
            
            IF the LENGTH OF max_heap IS GREATER THAN zero  
                APPLY heappop TO max_heap AND ASSIGN to count  
                INCREMENT count BY one  
                IF count IS LESS THAN zero  
                    APPEND a pair consisting of time PLUS n PLUS one AND count TO queue  
                END IF  
            END IF  
            
            INCREMENT time BY one  
        END WHILE  
        
        RETURN time  
    END FUNCTION  
END CLASS
