CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one TO zero STEP minus one  
            INCREMENT total_shifts BY element at position index of shifts  
            SET element at position index of shifts TO total_shifts  
        END FOR  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET numeric_value TO the Unicode code point of element at position index of s MINUS the Unicode code point of lowercase letter a  
            SET shifted_value TO the numeric_value PLUS element at position index of shifts  
            SET wrapped_value TO the remainder of shifted_value DIVIDED BY twenty six  
            SET new_character_code TO wrapped_value PLUS the Unicode code point of lowercase letter a  
            SET new_character TO the character represented by new_character_code  
            APPEND new_character TO result  
        END FOR  
        RETURN the concatenation of all elements of result as a string  
    END FUNCTION  
END CLASS
