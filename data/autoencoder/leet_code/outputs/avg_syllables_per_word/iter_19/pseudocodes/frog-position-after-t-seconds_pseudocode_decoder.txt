CLASS Solution  
    FUNCTION frogPosition with parameters n integer edges list of list of integers t integer target integer RETURNS float  
        IF n EQUALS one  
            RETURN one point zero  
        END IF  
        SET graph TO new mapping from integer to list of integers with default empty list  
        FOR each pair u integer and v integer IN edges  
            APPEND v TO the list at key u in graph  
            APPEND u TO the list at key v in graph  
        END FOR  
        SET queue TO new double ended queue initialized with tuple of integer one floating point one point zero integer zero representing current node current probability and current time respectively  
        SET visited TO new set initialized with integer one  
        WHILE the queue is not empty  
            REMOVE and RETURN leftmost element of queue ASSIGN TO current node current probability current time  
            IF current time EQUALS t OR current node NOT EQUALS one AND the number of elements in list at key current node in graph EQUALS one  
                IF current node EQUALS target  
                    RETURN current probability  
                ELSE  
                    CONTINUE to next iteration of while loop  
                END IF  
            END IF  
            SET possible moves TO the number of elements in list at key current node in graph MINUS one IF current node NOT EQUALS one ELSE zero  
            FOR each neighbor IN list at key current node in graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND tuple of neighbor current probability DIVIDED BY possible moves current time PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN zero point zero  
    END FUNCTION  
END CLASS
