CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF the LENGTH OF intervals EQUALS zero  
            RETURN zero  
        END IF  
        
        SORT intervals BY the value at position one of each element in intervals  
        
        SET end TO the value at position one of the element at position zero of intervals  
        SET count TO zero  
        
        FOR index FROM one TO the LENGTH OF intervals MINUS one  
            IF the value at position zero of the element at position index of intervals GREATER THAN OR EQUAL TO end  
                SET end TO the value at position one of the element at position index of intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
