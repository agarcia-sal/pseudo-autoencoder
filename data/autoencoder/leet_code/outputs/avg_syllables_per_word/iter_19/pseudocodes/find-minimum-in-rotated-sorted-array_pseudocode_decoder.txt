CLASS Solution  
    FUNCTION findMin(nums)  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        IF the element at position left of nums IS LESS THAN the element at position right of nums  
            RETURN the element at position left of nums  
        END IF  
        WHILE left IS LESS THAN right  
            SET mid TO left PLUS right DIVIDED BY two using integer division  
            IF the element at position mid of nums IS GREATER THAN the element at position right of nums  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN the element at position left of nums  
    END FUNCTION  
END CLASS
