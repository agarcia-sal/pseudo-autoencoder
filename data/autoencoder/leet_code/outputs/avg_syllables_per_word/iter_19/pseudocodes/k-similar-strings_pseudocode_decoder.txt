CLASS Solution  
    FUNCTION kSimilarity(s1, s2)  
        FUNCTION neighbors(s)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND element at position i of s EQUALS element at position i of s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF element at position j of s EQUALS element at position i of s2 AND element at position j of s NOT EQUALS element at position j of s2  
                    YIELD concatenation of substring from position zero TO i MINUS one of s PLUS element at position j of s PLUS substring from position i PLUS one TO j MINUS one of s PLUS element at position i of s PLUS substring from position j PLUS one TO END of s  
                END IF  
            END FOR  
        END FUNCTION  
        
        ASSIGN queue TO a new double ended queue containing a tuple of s1 and zero  
        ASSIGN visited TO a new set containing s1  
        
        WHILE queue IS NOT empty  
            ASSIGN s AND swaps TO the leftmost element removed from queue  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN the iterable returned by neighbors applied to s  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND tuple of neighbor and swaps PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
