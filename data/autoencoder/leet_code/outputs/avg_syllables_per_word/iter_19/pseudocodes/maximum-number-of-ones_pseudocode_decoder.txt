CLASS Solution  
    FUNCTION maximumNumberOfOnes(width height sideLength maxOnes)  
        SET freq TO a list of zeros with LENGTH sideLength MULTIPLIED BY sideLength  
        FOR each i FROM zero TO width MINUS one  
            FOR each j FROM zero TO height MINUS one  
                SET position TO the remainder of i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS the remainder of j DIVIDED BY sideLength  
                INCREMENT element at position position of freq BY one  
            END FOR  
        END FOR  
        SORT freq IN descending order  
        SET result TO the SUM of the elements from position zero TO position maxOnes MINUS one of freq  
        RETURN result  
    END FUNCTION  
END CLASS
