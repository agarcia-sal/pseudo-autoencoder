CLASS Solution  
    FUNCTION crackSafe WITH PARAMETERS n AND k  
        SET visited TO empty collection of unique elements  
        SET sequence TO empty list  

        FUNCTION dfs WITH PARAMETER node  
            FOR each integer i FROM zero TO k MINUS one  
                SET edge TO concatenation of node AND string representation of i  
                IF edge NOT IN visited  
                    ADD edge TO visited  
                    CALL dfs WITH argument as substring from position one TO end of edge  
                    APPEND string representation of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  

        SET start_node TO string consisting of character zero REPEATED n MINUS one times  
        CALL dfs WITH argument start_node  
        RETURN concatenation of all elements in sequence FOLLOWED BY start_node  
    END FUNCTION  
END CLASS
