CLASS Solution  
    FUNCTION ladderLength WITH PARAMETERS beginWord endWord wordList  
        IF endWord NOT IN wordList  
            RETURN zero  
        END IF  
        
        SET word_dict TO a dictionary that creates an empty list at new keys  
        SET word_length TO the LENGTH OF beginWord  
        
        FOR each word IN wordList  
            FOR each index FROM zero TO word_length MINUS one  
                SET intermediate_key TO concatenation of substring from position zero TO index MINUS one of word PLUS the character STAR PLUS the substring from position index PLUS one TO the end of word  
                APPEND word TO the list at key intermediate_key IN word_dict  
            END FOR  
        END FOR  
        
        SET queue TO a double ended queue containing the tuple of beginWord and one  
        SET visited TO a set containing beginWord  
        
        WHILE queue contains elements  
            SET current_word AND current_length TO the element removed from the front of queue  
            
            FOR each index FROM zero TO word_length MINUS one  
                SET intermediate_word TO concatenation of substring from position zero TO index MINUS one of current_word PLUS the character STAR PLUS the substring from position index PLUS one TO the end of current_word  
                
                FOR each word IN the list at key intermediate_word IN word_dict  
                    IF word EQUALS endWord  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited  
                        ADD word TO visited  
                        APPEND the tuple of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN zero  
    END FUNCTION  
END CLASS
