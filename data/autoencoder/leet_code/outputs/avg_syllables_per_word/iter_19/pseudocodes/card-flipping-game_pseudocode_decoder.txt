CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO the SET OF numbers x WHERE x IS the element at position index of fronts AND the element at position index of backs AND x EQUALS y FOR each index and corresponding elements x and y IN fronts AND backs  
        SET all_numbers TO the SET OF all elements IN fronts AND all elements IN backs  
        SET good_numbers TO the SET OF all_numbers MINUS the SET_OF same  
        IF the SET good_numbers IS NOT EMPTY  
            SET result TO the MINIMUM element IN good_numbers  
        ELSE  
            SET result TO zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
