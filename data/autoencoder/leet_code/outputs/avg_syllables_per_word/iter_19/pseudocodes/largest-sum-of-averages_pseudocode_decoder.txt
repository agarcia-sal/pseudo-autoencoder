CLASS Solution  
    FUNCTION largestSumOfAverages(nums AS List of integers, k AS integer) RETURNS float  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of zero values with LENGTH n PLUS one  
        FOR i FROM zero TO n MINUS one  
            SET element at position i PLUS one of prefix_sum TO element at position i of prefix_sum PLUS element at position i of nums  
        END FOR  
        
        FUNCTION average(i AS integer, j AS integer) RETURNS float  
            SET numerator TO element at position j PLUS one of prefix_sum MINUS element at position i of prefix_sum  
            SET denominator TO j MINUS i PLUS one  
            SET result TO numerator DIVIDED BY denominator  
            RETURN result  
        END FUNCTION  
        
        SET dp TO a two dimensional list of zero values with dimensions n PLUS one BY k PLUS one  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i THEN position one of dp TO CALL average WITH i AND n MINUS one  
        END FOR  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    SET candidate TO element at position j THEN position partitions MINUS one of dp PLUS CALL average WITH i AND j MINUS one  
                    IF candidate GREATER THAN element at position i THEN position partitions of dp THEN  
                        SET element at position i THEN position partitions of dp TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN element at position zero THEN position k of dp  
    END FUNCTION  
END CLASS
