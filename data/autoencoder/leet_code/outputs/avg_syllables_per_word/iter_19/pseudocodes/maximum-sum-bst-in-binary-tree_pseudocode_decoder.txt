CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A new instance of TreeNode INITIALIZED WITH element at position zero of values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left end of queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO A new instance of TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO A new instance of TreeNode INITIALIZED WITH element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF value of p.val IS NOT EQUALS TO value of q.val  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree WITH PARAMETERS node left of p AND node left of q AND the result of is_same_tree WITH PARAMETERS node right of p AND node right of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST(self, root)  
        SET self.max_sum TO zero  
        
        FUNCTION helper(node)  
            IF node IS none  
                RETURN True, zero, positive infinity, negative infinity  
            END IF  
            
            SET left_is_bst, left_sum, left_min, left_max TO the result of helper WITH PARAMETER node left of node  
            SET right_is_bst, right_sum, right_min, right_max TO the result of helper WITH PARAMETER node right of node  
            
            IF left_is_bst IS True AND right_is_bst IS True AND left_max IS LESS THAN value of node.val AND value of node.val IS LESS THAN right_min  
                SET current_sum TO left_sum PLUS right_sum PLUS value of node.val  
                SET current_min TO the smaller value BETWEEN value of node.val AND left_min  
                SET current_max TO the greater value BETWEEN value of node.val AND right_max  
                SET self.max_sum TO the greater value BETWEEN self.max_sum AND current_sum  
                RETURN True, current_sum, current_min, current_max  
            END IF  
            
            RETURN False, zero, positive infinity, negative infinity  
        END FUNCTION  
        
        CALL helper WITH PARAMETER root  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
