CLASS Solution  
    FUNCTION removeComments WITH parameter source of type List of strings  
        SET in_block_comment TO false  
        SET result TO empty list  
        SET current_line TO empty list  
        FOR each line IN source  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF line  
                IF in_block_comment  
                    IF the substring of line from position i to position i PLUS one EQUALS the string consisting of a forward slash followed by an asterisk character  
                        SET in_block_comment TO false  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF the substring of line from position i to position i PLUS one EQUALS the string consisting of two forward slashes  
                        BREAK the current WHILE loop  
                    ELSE IF the substring of line from position i to position i PLUS one EQUALS the string consisting of a forward slash followed by an asterisk character  
                        SET in_block_comment TO true  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND the element at position i of line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  
            IF current_line is not empty AND in_block_comment EQUALS false  
                APPEND the concatenation of all elements of current_line TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
