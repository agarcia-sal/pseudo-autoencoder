CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val EQUALS zero next EQUALS None  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values IS None OR values IS EMPTY  
        RETURN None  
    END IF  
    SET head TO a new ListNode WITH value equal to element at position zero of values  
    SET p TO head  
    FOR each val IN elements starting from position one TO the end of values  
        SET node TO a new ListNode WITH value equal to val  
        SET the next property of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    END IF  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    END IF  
    RETURN the logical AND of the condition that the val property of p1 EQUALS the val property of p2 AND the result of is_same_list called with the next property of p1 and the next property of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION plusOne WITH PARAMETERS self head  
        FUNCTION reverseList WITH PARAMETER node  
            SET prev TO None  
            WHILE node IS NOT None  
                SET next_node TO the next property of node  
                SET the next property of node TO prev  
                SET prev TO node  
                SET node TO next_node  
            END WHILE  
            RETURN prev  
        END FUNCTION  
        
        SET reversed_head TO the result of reverseList called with head  
        SET carry TO one  
        SET current TO reversed_head  
        
        WHILE current IS NOT None AND carry EQUALS one  
            SET the val property of current TO the val property of current PLUS carry  
            SET carry TO the val property of current DIVIDED BY ten  
            SET the val property of current TO the val property of current MODULO ten  
            
            IF the next property of current IS None AND carry EQUALS one  
                SET the next property of current TO a new ListNode WITH value equal to carry  
                SET carry TO zero  
            END IF  
            
            SET prev TO current  
            SET current TO the next property of current  
        END WHILE  
        
        RETURN the result of reverseList called with reversed_head  
    END FUNCTION  
END CLASS
