CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word Zero  
        END IF  

        SET below_twenty TO the list of words representing numbers from zero TO nineteen INCLUDING an empty string for zero position  
        SET tens TO the list of words representing multiples of ten from zero TO ninety INCLUDING empty strings for zero and ten positions  
        SET thousands TO the list of scale words representing units FROM empty string TO thousand TO million TO billion  

        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN the empty string  
            END IF  
            ELSE IF n LESS THAN twenty  
                RETURN the element at position n of below_twenty CONCATENATED WITH a single space  
            END IF  
            ELSE IF n LESS THAN one hundred  
                RETURN the element at position the integer division of n BY ten of tens CONCATENATED WITH a single space CONCATENATED WITH the result of calling helper with the remainder of n MODULO ten  
            END IF  
            ELSE  
                RETURN the element at position the integer division of n BY one hundred of below_twenty CONCATENATED WITH the word Hundred CONCATENATED WITH a single space CONCATENATED WITH the result of calling helper with the remainder of n MODULO one hundred  
            END ELSE  
        END FUNCTION  

        SET result TO the empty string  
        SET i TO zero  

        WHILE num GREATER THAN zero  
            IF the remainder of num MODULO one thousand NOT EQUALS zero  
                SET result TO the result of calling helper with the remainder of num MODULO one thousand CONCATENATED WITH the element at position i of thousands CONCATENATED WITH a single space CONCATENATED WITH result  
            END IF  
            SET num TO the integer division of num BY one thousand  
            INCREMENT i BY one  
        END WHILE  

        RETURN result AFTER REMOVING leading and trailing whitespace  
    END FUNCTION  
END CLASS
