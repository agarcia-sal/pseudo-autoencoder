CLASS Solution  
    FUNCTION splitLoopedString WITH PARAMETER strs AS List OF strings  
        FOR index FROM zero TO LENGTH OF strs MINUS one  
            SET reversed_str TO string AT position index of strs REVERSED  
            IF reversed_str LEXICOGRAPHICALLY GREATER THAN string AT position index of strs  
                SET string AT position index of strs TO reversed_str  
            END IF  
        END FOR  

        SET doubled_strs TO concatenation of strs FOLLOWED BY strs  

        SET max_string TO empty string  

        FOR index FROM zero TO LENGTH OF strs MINUS one  
            FOR each is_reversed IN list containing False AND True  
                IF is_reversed EQUALS True  
                    SET current TO string AT position index of strs REVERSED  
                ELSE  
                    SET current TO string AT position index of strs  
                END IF  

                SET remaining TO concatenation of elements from position index plus one TO position index plus LENGTH OF strs MINUS one in doubled_strs  

                FOR cut_position FROM zero TO LENGTH OF current  
                    SET candidate TO concatenation of substring from position cut_position TO end of current FOLLOWED BY remaining FOLLOWED BY substring from position zero TO cut_position MINUS one of current  
                    IF candidate LEXICOGRAPHICALLY GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  

        RETURN max_string  
    END FUNCTION  
END CLASS
