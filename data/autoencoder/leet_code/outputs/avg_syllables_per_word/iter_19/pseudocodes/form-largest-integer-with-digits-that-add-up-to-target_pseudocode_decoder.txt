CLASS Solution  
    FUNCTION largestNumber WITH PARAMETERS cost AND target  
        SET cost_to_digit TO a dictionary that maps each element c in cost to a string of the digit i plus one, where i is the index position of c in cost  
        SET dp TO a list whose first element is an empty string and all following elements from position one TO position target are set to the string indicating not possible  
        FOR t FROM one TO target  
            FOR each key c AND value d IN the items of cost_to_digit  
                IF t IS GREATER THAN OR EQUAL TO c AND the element at position t MINUS c of dp DOES NOT EQUAL the string indicating not possible  
                    SET candidate TO the concatenation of d FOLLOWED BY the element at position t MINUS c of dp described as a string  
                    IF the element at position t of dp EQUALS the string indicating not possible OR the LENGTH OF candidate IS GREATER THAN the LENGTH OF the element at position t of dp OR the LENGTH OF candidate EQUALS the LENGTH OF the element at position t of dp AND candidate IS GREATER THAN the element at position t of dp  
                        SET the element at position t of dp TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position target of dp IF it DOES NOT EQUAL the string indicating not possible OTHERWISE return the string zero  
    END FUNCTION  
END CLASS
