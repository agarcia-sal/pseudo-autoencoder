CLASS Solution  
    FUNCTION maxPerformance(n, speed, efficiency, k)  
        SET engineers TO a list of pairs where each pair contains one element from efficiency and the corresponding element from speed  
        SORT engineers in order of efficiency FROM highest TO lowest  
        SET speed_heap TO an empty collection representing a minimum heap of speeds  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair consisting of efficiency_element and speed_element IN engineers  
            ADD speed_element TO speed_heap  
            INCREMENT speed_sum BY speed_element  
            IF the size of speed_heap IS GREATER THAN k  
                REMOVE the smallest element FROM speed_heap and DECREMENT speed_sum BY that element  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY efficiency_element  
            SET max_performance TO the greater value BETWEEN max_performance AND current_performance  
        END FOR  
        RETURN the remainder WHEN max_performance IS DIVIDED BY one billion PLUS seven  
    END FUNCTION  
END CLASS
