CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH val EQUALS element at position zero of values  
    SET i TO one  
    SET queue TO A NEW empty double-ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO element REMOVED FROM FRONT OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO A NEW TreeNode WITH val EQUALS element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO A NEW TreeNode WITH val EQUALS element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree WITH p.left AND q.left AND is_same_tree WITH p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION dfs(node, current_sum, path)  
            IF node IS none  
                RETURN  
            END IF  
            APPEND node.val TO path  
            SET current_sum TO current_sum PLUS node.val  
            IF node.left IS none AND node.right IS none AND current_sum EQUALS targetSum  
                APPEND A COPY OF path TO result  
            END IF  
            CALL dfs WITH node.left, current_sum, path  
            CALL dfs WITH node.right, current_sum, path  
            REMOVE the LAST ELEMENT FROM path  
        END FUNCTION  
        SET result TO AN EMPTY list  
        CALL dfs WITH root, zero, AN EMPTY list  
        RETURN result  
    END FUNCTION  
END CLASS
