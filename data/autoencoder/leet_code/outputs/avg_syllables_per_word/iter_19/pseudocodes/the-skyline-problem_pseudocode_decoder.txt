CLASS Solution  
    FUNCTION getSkyline(buildings)  
        SET events TO an empty list  
        FOR each left edge coordinate left and right edge coordinate right and height height IN buildings  
            APPEND a tuple containing the left edge coordinate left and the negation of the height height and the right edge coordinate right TO events  
            APPEND a tuple containing the right edge coordinate right and zero and zero TO events  
        END FOR  
        SORT events in ascending order by the first element then by the second element then by the third element  
        SET result TO a list containing a list of zero and zero  
        SET max_heap TO a list containing a tuple of zero and positive infinity  
        FOR each x coordinate x and negative height negH and right edge coordinate r IN events  
            WHILE the second element of the first tuple in max_heap is less than or equal to x  
                REMOVE the first tuple from max_heap using a heap pop operation  
            END WHILE  
            IF the negative height negH is not equal to zero  
                ADD a tuple of negative height negH and right edge r to max_heap using a heap push operation  
            END IF  
            IF the second element of the last list in result is not equal to the negation of the first element of the first tuple in max_heap  
                APPEND a list containing x and the negation of the first element of the first tuple in max_heap TO result  
            END IF  
        END FOR  
        IF the first list in result is exactly a list containing zero and zero  
            REMOVE the first list from result  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
