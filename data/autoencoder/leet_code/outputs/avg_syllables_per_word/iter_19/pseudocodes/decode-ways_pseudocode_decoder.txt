CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s IS empty OR element at position zero of s EQUALS character zero  
            RETURN zero  
        END IF  
        
        SET dp TO a list of zeros with the LENGTH OF s PLUS one  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO one  
        
        FOR index FROM two TO the LENGTH OF s INCLUSIVE  
            IF element at position index MINUS one of s DOES NOT EQUAL character zero  
                INCREMENT element at position index of dp BY element at position index MINUS one of dp  
            END IF  
            
            SET two_digit TO the integer value OF substring from position index MINUS two TO index MINUS one of s  
            IF two_digit IS GREATER THAN OR EQUAL TO ten AND IS LESS THAN OR EQUAL TO twenty six  
                INCREMENT element at position index of dp BY element at position index MINUS two of dp  
            END IF  
        END FOR  
        
        RETURN element at the last position of dp  
    END FUNCTION  
END CLASS
