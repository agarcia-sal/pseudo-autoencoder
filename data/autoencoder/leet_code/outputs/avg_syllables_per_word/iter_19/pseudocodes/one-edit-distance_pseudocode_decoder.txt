CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s IS GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF calling isOneEditDistance WITH PARAMETERS t AND s  
        END IF  
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        IF diff IS GREATER THAN one  
            RETURN False  
        END IF  
        IF diff EQUALS zero  
            SET difference_count TO zero  
            FOR each pair of characters c1 AND c2 FROM s AND t AT THE SAME POSITION  
                IF c1 IS NOT EQUAL TO c2  
                    INCREMENT difference_count BY one  
                END IF  
            END FOR  
            RETURN whether difference_count EQUALS one  
        ELSE  
            FOR each index i FROM zero TO the LENGTH OF s MINUS one  
                IF the element at position i of s IS NOT EQUAL TO the element at position i of t  
                    RETURN whether the SUBSTRING from position i TO the END of s IS EQUAL TO the SUBSTRING from position i PLUS one TO the END of t  
                END IF  
            END FOR  
            RETURN True  
        END IF  
    END FUNCTION  
END CLASS
