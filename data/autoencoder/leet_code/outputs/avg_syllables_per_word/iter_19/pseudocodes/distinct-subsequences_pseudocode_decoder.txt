CLASS Solution  
    FUNCTION numDistinct(s STRING, t STRING) RETURNS INTEGER  
        SET m TO the LENGTH OF s  
        SET n TO the LENGTH OF t  
        
        CALL create two dimensional array dp WITH dimensions FROM zero TO m FOR rows AND zero TO n FOR columns FILLED WITH zero  
        
        FOR i FROM zero TO m  
            SET element at position zero of row at position i of dp TO one  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one of s EQUALS element at position j MINUS one of t THEN  
                    SET element at position j of row at position i of dp TO element at position j MINUS one of row at position i MINUS one of dp PLUS element at position j of row at position i MINUS one of dp  
                ELSE  
                    SET element at position j of row at position i of dp TO element at position j of row at position i MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of row at position m of dp  
    END FUNCTION  
END CLASS
