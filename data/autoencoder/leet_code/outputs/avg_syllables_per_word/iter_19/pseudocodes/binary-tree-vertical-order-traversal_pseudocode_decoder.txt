CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero THEN  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with element at position zero of values  
    SET i TO one  
    SET queue TO an empty double-ended queue  
    APPEND root TO queue  
    WHILE the queue is not empty DO  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.left TO a new TreeNode initialized with element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none THEN  
            SET node.right TO a new TreeNode initialized with element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none THEN  
        RETURN true  
    ELSE IF p IS none OR q IS none THEN  
        RETURN false  
    ELSE IF the value property of p NOT EQUALS the value property of q THEN  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with the left property of p and the left property of q AND the result of is_same_tree with the right property of p and the right property of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder(self, root)  
        IF root IS none THEN  
            RETURN an empty list  
        END IF  
        SET column_table TO a default dictionary with lists as default values  
        SET queue TO a double-ended queue initialized with a tuple containing root and zero  
        SET min_column TO zero  
        SET max_column TO zero  
        WHILE the queue is not empty DO  
            REMOVE the front element from queue and ASSIGN to variables node and column_index  
            IF node IS NOT none THEN  
                APPEND the value property of node TO the list at key column_index in column_table  
                IF min_column GREATER THAN column_index THEN  
                    SET min_column TO column_index  
                END IF  
                IF max_column LESS THAN column_index THEN  
                    SET max_column TO column_index  
                END IF  
                APPEND a tuple containing the left property of node and column_index MINUS one TO queue  
                APPEND a tuple containing the right property of node and column_index PLUS one TO queue  
            END IF  
        END WHILE  
        SET result TO an empty list  
        FOR each x FROM min_column TO max_column DO  
            APPEND the list at key x in column_table TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
