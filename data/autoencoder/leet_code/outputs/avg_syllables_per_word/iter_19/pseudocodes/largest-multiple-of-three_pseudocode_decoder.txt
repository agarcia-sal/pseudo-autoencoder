CLASS Solution  
    FUNCTION largestMultipleOfThree WITH PARAMETER digits  
        SET count TO the frequency count of elements in digits  
        SET total_sum TO the sum of all elements in digits  
        SET remainder TO the remainder when total_sum is divided by three  
        
        IF remainder EQUALS one  
            FOR each digit IN the collection one four seven  
                IF the value associated with digit in count IS GREATER THAN zero  
                    DECREMENT the value associated with digit in count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit IN the collection two five eight  
                    IF the value associated with digit in count IS GREATER THAN one  
                        DECREMENT the value associated with digit in count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
        
        IF remainder EQUALS two  
            FOR each digit IN the collection two five eight  
                IF the value associated with digit in count IS GREATER THAN zero  
                    DECREMENT the value associated with digit in count BY one  
                    BREAK the loop  
                END IF  
            END FOR  
            ELSE  
                FOR each digit IN the collection one four seven  
                    IF the value associated with digit in count IS GREATER THAN one  
                        DECREMENT the value associated with digit in count BY two  
                        BREAK the loop  
                    END IF  
                END FOR  
            END ELSE  
        END IF  
        
        SET result TO an empty list  
        FOR digit FROM nine DOWN TO zero  
            APPEND the string conversion of digit repeated the number of times equal to the value associated with digit in count TO result  
        END FOR  
        
        SET final_number TO the concatenation of all elements in result  
        
        IF final_number IS empty OR the element at position one of final_number EQUALS zero  
            RETURN zero as a string  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
