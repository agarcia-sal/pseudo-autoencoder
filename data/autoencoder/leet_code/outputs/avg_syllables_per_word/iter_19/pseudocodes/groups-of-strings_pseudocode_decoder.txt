CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.parent TO list of integers from zero TO size MINUS one  
        SET self.rank TO list of zero integers of length size  
        SET self.size TO list of one integers of length size  
    END FUNCTION  
  
    FUNCTION find(self, u)  
        IF element at position u of self.parent IS NOT EQUAL TO u  
            SET element at position u of self.parent TO the result of calling find with argument element at position u of self.parent  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  
  
    FUNCTION union(self, u, v)  
        SET rootU TO the result of calling find with argument u  
        SET rootV TO the result of calling find with argument v  
        IF rootU IS NOT EQUAL TO rootV  
            IF element at position rootU of self.rank IS GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.size BY element at position rootV of self.size  
            ELSE IF element at position rootU of self.rank IS LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
                INCREMENT element at position rootV of self.size BY element at position rootU of self.size  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.size BY element at position rootV of self.size  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION groupStrings(self, words)  
        FUNCTION bitmask(word)  
            SET mask TO zero  
            FOR each character IN word  
                SET mask TO mask OR one LEFT SHIFTED BY the difference between the UNICODE code point of character AND the UNICODE code point of character a  
            END FOR  
            RETURN mask  
        END FUNCTION  
  
        SET word_map TO a dictionary with default value as empty list  
        SET n TO the LENGTH OF words  
        SET uf TO a new instance of UnionFind with argument n  
  
        FOR each indexed word pair i, word IN enumeration of words  
            SET mask TO the result of calling bitmask with argument word  
            APPEND i TO element at key mask of word_map  
        END FOR  
  
        FOR each indexed word pair i, word IN enumeration of words  
            SET mask TO the result of calling bitmask with argument word  
            FOR j FROM zero TO twenty five  
                SET new_mask TO mask EXCLUSIVE OR one LEFT SHIFTED BY j  
                IF new_mask EXISTS AS A KEY IN word_map  
                    FOR each element k IN element at key new_mask of word_map  
                        CALL union of uf with arguments i AND k  
                    END FOR  
                END IF  
  
                IF the BITWISE AND of mask AND one LEFT SHIFTED BY j EQUALS zero  
                    SET new_mask TO mask OR one LEFT SHIFTED BY j  
                    IF new_mask EXISTS AS A KEY IN word_map  
                        FOR each element k IN element at key new_mask of word_map  
                            CALL union of uf with arguments i AND k  
                        END FOR  
                    END IF  
                END IF  
  
                IF the BITWISE AND of mask AND one LEFT SHIFTED BY j IS NOT EQUAL TO zero  
                    FOR k FROM zero TO twenty five  
                        IF the BITWISE AND of mask AND one LEFT SHIFTED BY k EQUALS zero  
                            SET new_mask TO mask EXCLUSIVE OR one LEFT SHIFTED BY j OR one LEFT SHIFTED BY k  
                            IF new_mask EXISTS AS A KEY IN word_map  
                                FOR each element l IN element at key new_mask of word_map  
                                    CALL union of uf with arguments i AND l  
                                END FOR  
                            END IF  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
  
        SET root_sizes TO a dictionary with default value zero  
        FOR i FROM zero TO n MINUS one  
            SET root TO the result of calling find of uf with argument i  
            INCREMENT element at key root of root_sizes BY one  
        END FOR  
  
        SET max_group_size TO the maximum VALUE IN root_sizes  
        SET num_groups TO the count of keys IN root_sizes  
  
        RETURN a list with first element num_groups AND second element max_group_size  
    END FUNCTION  
END CLASS
