CLASS Solution  
    FUNCTION rearrangeSticks(n INTEGER, k INTEGER) RETURNS INTEGER  
        SET MOD TO one billion PLUS seven  
        
        DEFINE dp AS two dimensional list of zeroes with dimensions n PLUS one BY k PLUS one  
        
        SET element at position zero of first list in dp TO one  
        
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET element at position j of list at position i in dp TO the remainder after dividing the sum of the element at position j MINUS one of list at position i MINUS one in dp PLUS the product of i MINUS one AND the element at position j of list at position i MINUS one in dp BY MOD  
            END FOR  
        END FOR  
        
        RETURN the element at position k of list at position n in dp  
    END FUNCTION  
END CLASS
