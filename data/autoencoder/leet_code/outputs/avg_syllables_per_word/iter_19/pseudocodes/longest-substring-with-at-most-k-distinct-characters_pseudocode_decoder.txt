CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s STRING, k INTEGER) RETURNS INTEGER  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
        
        SET char_map TO an empty mapping from characters to integers  
        SET left TO zero  
        SET max_length TO zero  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            SET the element at position right of s in char_map TO right  
            
            IF the COUNT OF keys in char_map GREATER THAN k  
                SET leftmost_char TO the key in char_map with the minimum associated value  
                SET left TO the value associated with leftmost_char in char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            
            SET max_length TO the GREATER OF max_length AND right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
