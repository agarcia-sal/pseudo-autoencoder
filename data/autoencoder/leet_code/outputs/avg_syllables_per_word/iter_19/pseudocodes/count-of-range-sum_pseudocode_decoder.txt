CLASS Solution  
    FUNCTION countRangeSum(nums AS List OF Integer, lower AS Integer, upper AS Integer) RETURNING Integer  
        SET prefix_sums TO a list containing zero  
        SET current_sum TO zero  
        SET count TO zero  
        
        FOR each num IN nums  
            SET current_sum TO current_sum PLUS num  
            SET left TO current_sum MINUS upper  
            SET right TO current_sum MINUS lower  
            
            SET count TO count PLUS the number of elements in prefix_sums less than or equal to right MINUS the number of elements in prefix_sums less than left  
            INSERT current_sum INTO prefix_sums maintaining sorted order  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
