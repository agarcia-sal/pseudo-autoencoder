CLASS Solution  
    FUNCTION kInversePairs WITH PARAMETERS n AS integer AND k AS integer  
        SET MODULO TO one billion PLUS seven  
        SET dp TO a new two dimensional collection of size n PLUS one BY k PLUS one FILLED WITH zero  
        SET element at position zero of dp AND zero TO one  
        FOR index i FROM one TO n  
            SET prefix_sum TO zero  
            FOR index j FROM zero TO k  
                SET prefix_sum TO prefix_sum PLUS element at position i MINUS one of dp AND j  
                SET prefix_sum TO prefix_sum MODULO MODULO  
                IF j IS GREATER THAN OR EQUAL TO i THEN  
                    SET prefix_sum TO prefix_sum MINUS element at position i MINUS one of dp AND j MINUS i PLUS MODULO  
                    SET prefix_sum TO prefix_sum MODULO MODULO  
                END IF  
                SET element at position i of dp AND j TO prefix_sum  
            END FOR  
        END FOR  
        RETURN element at position n of dp AND k  
    END FUNCTION  
END CLASS
