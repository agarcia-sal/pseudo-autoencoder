CLASS Solution  
    FUNCTION findKthLargest WITH parameters nums AND k  
        SET min_heap TO an empty collection representing a heap  
        FOR each num IN nums  
            PERFORM operation to add num INTO min_heap maintaining heap order  
            IF the LENGTH OF min_heap IS GREATER THAN k  
                PERFORM operation to remove the smallest element FROM min_heap  
            END IF  
        END FOR  
        RETURN the element at position zero of min_heap  
    END FUNCTION  
END CLASS
