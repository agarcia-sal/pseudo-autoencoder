CLASS Solution  
    FUNCTION minCostToSupplyWater WITH PARAMETERS n AS integer AND wells AS list of integers AND pipes AS list of lists of integers  
        FOR index FROM zero TO n MINUS one  
            APPEND a list containing zero AND index PLUS one AND element at position index of wells TO pipes  
        END FOR  
        SORT pipes IN ASCENDING ORDER BASED ON the element at position three of each element  
        SET parent TO a list representing the sequence of integers FROM zero TO n  
        FUNCTION find WITH PARAMETER x  
            IF element at position x of parent DOES NOT EQUAL x  
                SET element at position x of parent TO the result of FUNCTION find with argument element at position x of parent  
            END IF  
            RETURN element at position x of parent  
        END FUNCTION  
        FUNCTION union WITH PARAMETERS x AND y  
            SET rootX TO the result of FUNCTION find with argument x  
            SET rootY TO the result of FUNCTION find with argument y  
            IF rootX DOES NOT EQUAL rootY  
                SET element at position rootX of parent TO rootY  
                RETURN true  
            END IF  
            RETURN false  
        END FUNCTION  
        SET min_cost TO zero  
        FOR each house1 AS first element AND house2 AS second element AND cost AS third element IN pipes  
            IF FUNCTION union WITH ARGUMENTS house1 AND house2 RETURNS true  
                INCREMENT min_cost BY cost  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
