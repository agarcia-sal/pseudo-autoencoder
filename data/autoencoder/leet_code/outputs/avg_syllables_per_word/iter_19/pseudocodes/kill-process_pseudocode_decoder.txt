CLASS Solution  
    FUNCTION killProcess(pid AS List of integers ppid AS List of integers kill AS integer) RETURNS List of integers  
        SET tree TO empty mapping from integer TO list of integers  
        FOR each pair of child and parent IN pairs formed by corresponding elements of pid and ppid  
            APPEND child TO list at position parent in tree  
        END FOR  
        SET queue TO list containing kill  
        SET killed_processes TO empty list  
        WHILE the LENGTH OF queue IS GREATER THAN zero  
            SET current TO the element at position zero of queue  
            REMOVE the element at position zero FROM queue  
            APPEND current TO killed_processes  
            FOR each child IN list at position current in tree  
                APPEND child TO queue  
            END FOR  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
