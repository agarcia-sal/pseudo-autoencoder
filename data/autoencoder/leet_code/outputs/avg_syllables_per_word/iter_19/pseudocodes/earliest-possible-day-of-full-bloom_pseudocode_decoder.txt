CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        SET tasks TO a list of pairs each consisting of element at each position of plantTime AND element at same position of growTime sorted in order of the second element of each pair from largest TO smallest  
        SET current_day TO zero  
        SET earliest_bloom TO zero  
        FOR each pair consisting of plant and grow IN tasks  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF earliest_bloom LESS THAN bloom_day  
                SET earliest_bloom TO bloom_day  
            END IF  
        END FOR  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
