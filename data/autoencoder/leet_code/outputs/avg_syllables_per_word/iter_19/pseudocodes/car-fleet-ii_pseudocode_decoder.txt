CLASS Solution  
    FUNCTION getCollisionTimes(cars)  
        SET n TO the LENGTH OF cars  
        SET result TO a list of n elements each set TO negative one point zero  
        SET stack TO an empty list  

        FOR i FROM n MINUS one DOWN TO zero  
            SET pos TO the element at position zero of element at position i of cars  
            SET speed TO the element at position one of element at position i of cars  
            
            WHILE the LENGTH OF stack IS greater THAN zero  
                SET next_index TO the element at the last position of stack  
                SET next_pos TO the element at position zero of element at position next_index of cars  
                SET next_speed TO the element at position one of element at position next_index of cars  
                
                IF speed IS LESS THAN OR EQUAL TO next_speed  
                    REMOVE the last element FROM stack  
                    CONTINUE to the next iteration of while  
                END IF  
                
                SET collision_time TO next_pos MINUS pos DIVIDED BY speed MINUS next_speed  
                
                IF the element at position next_index of result IS GREATER THAN zero AND collision_time IS GREATER THAN OR EQUAL TO the element at position next_index of result  
                    REMOVE the last element FROM stack  
                    CONTINUE to the next iteration of while  
                END IF  
                
                SET the element at position i of result TO collision_time  
                BREAK from the while loop  
            END WHILE  
            
            APPEND i TO stack  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
