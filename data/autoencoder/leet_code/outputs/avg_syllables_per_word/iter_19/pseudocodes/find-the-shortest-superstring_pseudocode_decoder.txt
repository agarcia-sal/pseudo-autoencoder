CLASS Solution  
    FUNCTION shortestSuperstring(words)  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap(i comma j)  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i of words AND the LENGTH OF element at position j of words DOWN TO one  
                IF the SUBSTRING of element at position i of words from position LENGTH OF element at position i of words MINUS k PLUS one TO end EQUALS the SUBSTRING of element at position j of words from position one TO position k THEN  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp(mask comma i)  
            IF mask EQUALS one LEFT SHIFTED BY n MINUS one THEN  
                RETURN zero AND the empty string  
            END IF  
            
            SET min_len TO positive infinity  
            SET best_path TO the empty string  
            FOR j FROM zero TO n MINUS one  
                IF the BITWISE AND of mask AND one LEFT SHIFTED BY j EQUALS zero THEN  
                    SET length_path_pair TO the RESULT OF dp WITH ARGUMENTS the BITWISE OR of mask AND one LEFT SHIFTED BY j AND j  
                    SET length TO the FIRST ELEMENT OF length_path_pair  
                    SET path TO the SECOND ELEMENT OF length_path_pair  
                    INCREMENT length BY the LENGTH OF element at position j of words MINUS the RESULT OF overlap WITH ARGUMENTS i AND j  
                    IF length LESS THAN min_len THEN  
                        SET min_len TO length  
                        SET best_path TO the SUBSTRING of element at position j of words from position the RESULT OF overlap WITH ARGUMENTS i AND j PLUS one TO end CONCATENATED WITH path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO positive infinity  
        SET shortest_path TO the empty string  
        FOR i FROM zero TO n MINUS one  
            SET length_path_pair TO the RESULT OF dp WITH ARGUMENTS one LEFT SHIFTED BY i AND i  
            SET length TO the FIRST ELEMENT OF length_path_pair  
            SET path TO the SECOND ELEMENT OF length_path_pair  
            INCREMENT length BY the LENGTH OF element at position i of words  
            IF length LESS THAN min_len THEN  
                SET min_len TO length  
                SET shortest_path TO element at position i of words CONCATENATED WITH path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
