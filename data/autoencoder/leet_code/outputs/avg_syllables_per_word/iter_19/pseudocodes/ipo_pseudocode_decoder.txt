CLASS Solution  
    FUNCTION findMaximizedCapital(k INTEGER PARAMETER, w INTEGER PARAMETER, profits LIST OF INTEGER PARAMETER, capital LIST OF INTEGER PARAMETER) RETURNS INTEGER  
        SET projects TO A LIST OF PAIRS WHERE EACH PAIR CONSISTS OF ELEMENT AT POSITION index OF capital AND ELEMENT AT POSITION index OF profits FOR index FROM zero TO LENGTH OF capital MINUS ONE  
        SORT projects IN ASCENDING ORDER BASED ON THE FIRST ELEMENT OF EACH PAIR  
        
        SET max_heap TO AN EMPTY LIST  
        
        SET i TO zero  
        SET n TO THE LENGTH OF projects  
        
        FOR count FROM one TO k  
            WHILE i IS LESS THAN n AND THE FIRST ELEMENT OF THE PAIR AT POSITION i OF projects IS LESS THAN OR EQUAL TO w  
                PUSH THE NEGATIVE OF THE SECOND ELEMENT OF THE PAIR AT POSITION i OF projects ONTO max_heap USING HEAP QUEUE INSERTION  
                INCREMENT i BY one  
            END WHILE  
            
            IF max_heap IS EMPTY  
                BREAK OUT OF THE FOR LOOP  
            END IF  
            
            SET w TO w PLUS THE NEGATIVE OF THE HEAP QUEUE POPPED ELEMENT FROM max_heap  
        END FOR  
        
        RETURN w  
    END FUNCTION  
END CLASS
