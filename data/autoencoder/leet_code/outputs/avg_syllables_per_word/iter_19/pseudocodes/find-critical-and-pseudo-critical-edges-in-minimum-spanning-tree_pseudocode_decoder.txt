CLASS UnionFind  
    FUNCTION __init__(self, n)  
        SET self.parent TO list of integers from zero TO n MINUS one  
        SET self.rank TO list of integer one repeated n times  
    END FUNCTION  
      
    FUNCTION find(self, u)  
        IF element at position u of self.parent IS NOT EQUAL TO u  
            SET element at position u of self.parent TO the RESULT OF self.find with element at position u of self.parent as parameter  
        END IF  
        RETURN element at position u of self.parent  
    END FUNCTION  
      
    FUNCTION union(self, u, v)  
        SET rootU TO the RESULT OF self.find with u as parameter  
        SET rootV TO the RESULT OF self.find with v as parameter  
          
        IF rootU IS NOT EQUAL TO rootV  
            IF element at position rootU of self.rank IS GREATER THAN element at position rootV of self.rank  
                SET element at position rootV of self.parent TO rootU  
            ELSE IF element at position rootU of self.rank IS LESS THAN element at position rootV of self.rank  
                SET element at position rootU of self.parent TO rootV  
            ELSE  
                SET element at position rootV of self.parent TO rootU  
                INCREMENT element at position rootU of self.rank BY one  
            END IF  
            RETURN True  
        END IF  
        RETURN False  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)  
        FOR index, edge IN enumeration of edges  
            APPEND index TO edge  
        END FOR  
          
        SORT edges BY the integer at position two of each element in ascending order  
          
        FUNCTION mst(exclude, include)  
            SET uf TO new UnionFind with parameter n  
            SET weight TO zero  
              
            IF include IS NOT None  
                CALL uf.union with first element of include and second element of include  
                INCREMENT weight BY third element of include  
            END IF  
              
            FOR each u, v, w, idx IN edges  
                IF idx IS EQUAL TO exclude  
                    CONTINUE TO next iteration  
                END IF  
                IF calling uf.union with u and v RETURNS True  
                    INCREMENT weight BY w  
                END IF  
            END FOR  
              
            SET root TO the RESULT OF uf.find with zero as parameter  
            FOR i FROM zero TO n MINUS one  
                IF the RESULT OF uf.find with i as parameter IS NOT EQUAL TO root  
                    RETURN infinity value  
                END IF  
            END FOR  
              
            RETURN weight  
        END FUNCTION  
          
        SET mst_weight TO the RESULT OF mst with default parameters  
          
        SET critical TO empty list  
        SET pseudo_critical TO empty list  
          
        FOR each u, v, w, idx IN edges  
            IF mst with exclude equal to idx RETURNS a value GREATER THAN mst_weight  
                APPEND idx TO critical  
            ELSE IF mst with include equal to tuple of u, v, w RETURNS a value EQUAL TO mst_weight  
                APPEND idx TO pseudo_critical  
            END IF  
        END FOR  
          
        RETURN list containing critical and pseudo_critical  
    END FUNCTION  
END CLASS
