CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO the SUBSTRING SEGMENTS OF num SPLIT BY the CHARACTER plus  
            SET real TO the INTEGER VALUE OF element at position zero of parts  
            SET imaginary TO the INTEGER VALUE OF substring from position zero TO the LENGTH OF element at position one of parts MINUS one of element at position one of parts  
            RETURN real AND imaginary  
        END FUNCTION  
        SET real1 AND imag1 TO THE RESULT OF parse_complex WITH num1  
        SET real2 AND imag2 TO THE RESULT OF parse_complex WITH num2  
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        RETURN a STRING REPRESENTATION COMBINING real_part FOLLOWED BY the CHARACTER plus FOLLOWED BY imaginary_part FOLLOWED BY the CHARACTER i  
    END FUNCTION  
END CLASS
