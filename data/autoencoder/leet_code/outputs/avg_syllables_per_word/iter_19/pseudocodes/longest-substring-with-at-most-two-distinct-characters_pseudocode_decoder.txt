CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct WITH PARAMETER s  
        INITIALIZE a dictionary named character count mapping characters TO integers with default value zero  
        SET left pointer TO zero  
        SET maximum length TO zero  
        FOR right pointer FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the count of character at position right pointer of s IN character count BY one  
            WHILE the NUMBER OF keys IN character count IS GREATER THAN two  
                DECREMENT the count of character at position left pointer of s IN character count BY one  
                IF the count of character at position left pointer of s IN character count EQUALS zero  
                    REMOVE the character at position left pointer of s FROM character count  
                END IF  
                INCREMENT left pointer BY one  
            END WHILE  
            SET maximum length TO THE GREATER OF maximum length AND right pointer MINUS left pointer PLUS one  
        END FOR  
        RETURN maximum length  
    END FUNCTION  
END CLASS
