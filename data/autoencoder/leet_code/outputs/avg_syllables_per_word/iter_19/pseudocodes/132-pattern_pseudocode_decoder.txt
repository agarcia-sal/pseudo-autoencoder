CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        FOR each num IN the elements of nums in reversed order  
            IF num is less than second  
                RETURN true  
            END IF  
            WHILE stack is not empty AND the last element at position negative one of stack is less than num  
                SET second TO the element removed from the end of stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
