CLASS Solution  
    FUNCTION checkSubarraySum(nums AS List of integers, k AS integer) RETURNS boolean  
        SET remainder_map TO a dictionary that maps zero TO negative one  
        SET current_sum TO zero  
        FOR each index i AND each number num IN nums WITH their enumeration  
            INCREMENT current_sum BY num  
            IF k NOT EQUALS zero THEN  
                SET current_sum TO the remainder when current_sum IS DIVIDED BY k  
            END IF  
            IF current_sum IS A KEY IN remainder_map THEN  
                IF i MINUS the element at position current_sum of remainder_map IS GREATER THAN one THEN  
                    RETURN True  
                END IF  
            ELSE  
                SET the element at position current_sum of remainder_map TO i  
            END IF  
        END FOR  
        RETURN False  
    END FUNCTION  
END CLASS
