CLASS Solution  
    FUNCTION numberOfWays WITH parameter corridor  
        SET MOD TO one billion PLUS seven  
        SET seats TO a list consisting of all indices i WHERE the element x at position i of corridor EQUALS character S  
        IF the length of seats MODULO two NOT EQUALS zero OR the length of seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM two TO the length of seats MINUS one STEP two  
            SET difference TO the element at position index of seats MINUS the element at position index MINUS one of seats  
            SET ways TO the multiplication of ways AND difference MODULO MOD  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
