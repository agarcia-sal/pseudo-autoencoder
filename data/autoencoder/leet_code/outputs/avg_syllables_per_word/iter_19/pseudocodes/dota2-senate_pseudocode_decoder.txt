CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        ASSIGN pointer TO a double ended queue named radiant  
        ASSIGN pointer TO a double ended queue named dire  
        FOR each pair of index and character IN the enumeration of senate  
            IF character EQUALS the uppercase letter R  
                APPEND index TO radiant  
            ELSE  
                APPEND index TO dire  
            END IF  
        END FOR  
        WHILE radiant is not empty AND dire is not empty  
            SET r_index TO the element removed from the front of radiant  
            SET d_index TO the element removed from the front of dire  
            IF r_index LESS THAN d_index  
                APPEND the sum of r_index PLUS the length of senate TO radiant  
            ELSE  
                APPEND the sum of d_index PLUS the length of senate TO dire  
            END IF  
        END WHILE  
        IF radiant is not empty  
            RETURN string Radiant  
        ELSE  
            RETURN string Dire  
        END IF  
    END FUNCTION  
END CLASS
