CLASS Solution  
    FUNCTION combinationSum4(nums, target)  
        SET dp TO a list of zeros with length equal to target PLUS one  
        SET element at position zero of dp TO one  
        FOR each amount FROM one TO target  
            FOR each num IN nums  
                IF amount IS GREATER THAN OR EQUAL TO num  
                    SET element at position amount of dp TO element at position amount of dp PLUS element at position amount MINUS num of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position target of dp  
    END FUNCTION  
END CLASS
