CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO an empty list  
        FOR each num IN target  
            APPEND negative num TO max_heap  
        END FOR  
        CALL heapify ON max_heap to organize it as a heap  
        SET total_sum TO the sum of all elements in target  
        WHILE the element at position zero of max_heap is LESS THAN negative one  
            SET largest TO the negative of the result of removing the smallest element from max_heap heap  
            SET rest_sum TO total_sum MINUS largest  
            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum IS EQUAL TO zero  
                RETURN False  
            END IF  
            SET new_value TO the remainder of largest DIVIDED BY rest_sum  
            IF new_value IS EQUAL TO zero AND rest_sum IS NOT EQUAL TO one  
                RETURN False  
            END IF  
            SET total_sum TO rest_sum PLUS new_value  
            ADD the negative of new_value INTO max_heap using heap push operation  
        END WHILE  
        RETURN True  
    END FUNCTION  
END CLASS
