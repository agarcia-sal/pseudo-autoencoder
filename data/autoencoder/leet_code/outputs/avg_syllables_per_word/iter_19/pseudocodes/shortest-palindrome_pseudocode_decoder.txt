CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        END IF  

        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS reverse of check_str  
        END FUNCTION  

        FOR i FROM the LENGTH OF s DOWN TO zero DECREMENT BY one  
            IF is_palindrome(substring from position zero TO i MINUS one of s)  
                SET to_add TO reverse of substring from position i TO the END OF s  
                RETURN concatenation of to_add AND s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
