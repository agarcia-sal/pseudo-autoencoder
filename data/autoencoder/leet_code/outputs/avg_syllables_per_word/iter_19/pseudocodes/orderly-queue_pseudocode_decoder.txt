CLASS Solution  
    FUNCTION orderlyQueue WITH PARAMETERS s AND k  
        IF k GREATER THAN one  
            RETURN string CREATED BY JOINING THE ELEMENTS OF STRING s SORTED IN LEXICOGRAPHICAL ORDER  
        ELSE  
            SET n TO the LENGTH OF string s  
            SET smallest TO string s  
            FOR each integer i FROM one TO n MINUS one  
                SET rotated TO substring from position i TO the END of string s CONCATENATED WITH substring from position zero TO position i MINUS one of string s  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
