CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO two dimensional collection of Boolean TRUE with size n BY n  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM the LENGTH OF s MINUS two DOWN TO zero DECREMENTING BY one  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                SET element at position i and j of f TO FALSE  
                IF element at position i of s EQUALS element at position j of s  
                    SET element at position i and j of f TO element at position i PLUS one and j MINUS one of f  
                    IF element at position i and j of f IS TRUE AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN substring from position k TO position k PLUS mx of s  
    END FUNCTION  
END CLASS
