CLASS Solution  
    FUNCTION rotatedDigits(n)  
        SET change_digits TO collection of characters two five six nine  
        SET same_digits TO collection of characters zero one eight  
        
        FUNCTION is_good_number(num)  
            SET num_str TO string representation of num  
            IF every digit IN num_str IS IN the UNION of change_digits AND same_digits THEN  
                IF there EXISTS a digit IN num_str THAT IS IN change_digits THEN  
                    RETURN true  
                END IF  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET good_count TO zero  
        FOR each integer i FROM one TO n  
            IF is_good_number(i) IS true THEN  
                INCREMENT good_count BY one  
            END IF  
        END FOR  
        RETURN good_count  
    END FUNCTION  
END CLASS
