CLASS Solution  
    FUNCTION validTicTacToe(board)  
        SET countX TO zero  
        FOR each row IN board  
            INCREMENT countX BY the number of times the character X occurs in row  
        END FOR  
        SET countO TO zero  
        FOR each row IN board  
            INCREMENT countO BY the number of times the character O occurs in row  
        END FOR  
        
        IF countX LESS THAN countO OR countX GREATER THAN countO PLUS one  
            RETURN false  
        END IF  
        
        FUNCTION check_win(player)  
            FOR each row IN board  
                IF row EQUALS player concatenated three times  
                    RETURN true  
                END IF  
            END FOR  
            FOR column FROM zero TO two  
                IF the element at position column of the element at position zero of board EQUALS player AND the element at position column of the element at position one of board EQUALS player AND the element at position column of the element at position two of board EQUALS player  
                    RETURN true  
                END IF  
            END FOR  
            IF the element at position zero of the element at position zero of board EQUALS player AND the element at position one of the element at position one of board EQUALS player AND the element at position two of the element at position two of board EQUALS player  
                RETURN true  
            END IF  
            IF the element at position two of the element at position zero of board EQUALS player AND the element at position one of the element at position one of board EQUALS player AND the element at position zero of the element at position two of board EQUALS player  
                RETURN true  
            END IF  
            RETURN false  
        END FUNCTION  
        
        SET winX TO the result of calling check_win with the character X as argument  
        SET winO TO the result of calling check_win with the character O as argument  
        
        IF winX AND winO  
            RETURN false  
        END IF  
        IF winX AND countX EQUALS countO  
            RETURN false  
        END IF  
        IF winO AND countX NOT EQUALS countO  
            RETURN false  
        END IF  
        
        RETURN true  
    END FUNCTION  
END CLASS
