CLASS Solution  
    FUNCTION waysToBuildRooms(prevRoom)  
        SET n TO the LENGTH OF prevRoom  
        ASSIGN graph TO a new default dictionary of lists  
        FOR each room FROM one TO n MINUS one  
            APPEND room TO the element at position prevRoom at position room of graph  
        END FOR  

        FUNCTION dp(node)  
            IF the element at position node of graph is empty  
                RETURN one AND one  
            END IF  
            SET total_ways TO one  
            SET total_size TO one  
            ASSIGN child_results TO an empty list  
            FOR each child IN the element at position node of graph  
                SET ways AND size TO the result of calling dp with child  
                APPEND the pair consisting of ways AND size TO child_results  
                INCREMENT total_size BY size  
            END FOR  
            SET current_size TO one  
            FOR each pair ways AND size IN child_results  
                SET total_ways TO total_ways MULTIPLIED BY the combination of current_size PLUS size MINUS one CHOOSE size MULTIPLIED BY ways MODULO the constant MOD  
                INCREMENT current_size BY size  
            END FOR  
            RETURN total_ways AND total_size  
        END FUNCTION  

        RETURN the first element of the result of calling dp with zero  
    END FUNCTION  
END CLASS
