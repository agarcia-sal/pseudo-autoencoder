CLASS Solution  
    FUNCTION minWindow(s1, s2)  
        SET min_len TO positive infinity  
        SET min_start TO minus one  
        SET len1 TO the LENGTH OF s1  
        SET len2 TO the LENGTH OF s2  
        
        FOR i FROM zero TO len1 MINUS one  
            SET j TO zero  
            FOR k FROM i TO len1 MINUS one  
                IF element at position k of s1 EQUALS element at position j of s2  
                    INCREMENT j BY one  
                    IF j EQUALS len2  
                        IF k MINUS i PLUS one LESS THAN min_len  
                            SET min_len TO k MINUS i PLUS one  
                            SET min_start TO i  
                        END IF  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF min_start EQUALS minus one  
            RETURN empty string  
        ELSE  
            RETURN substring from position min_start TO min_start PLUS min_len MINUS one of s1  
        END IF  
    END FUNCTION  
END CLASS
