CLASS Solution  
    FUNCTION minSkips(dist AS List OF integers, speed AS integer, hoursBefore AS integer)  
        SET n TO the LENGTH OF dist  
        SET dp TO a two dimensional list with n rows and n PLUS one columns each initialized TO infinity  
        SET total_dist TO the SUM OF all elements IN dist  
        
        IF the total_dist DIVIDED BY speed IS GREATER THAN hoursBefore THEN  
            RETURN negative one  
        END IF  
        
        SET element at row zero column zero of dp TO element at position zero of dist  
        FOR each j FROM one TO n MINUS one  
            SET element at row zero column j of dp TO element at position zero of dist  
        END FOR  
        
        FOR each i FROM one TO n MINUS one  
            FOR each j FROM zero TO i  
                IF j IS LESS THAN i THEN  
                    SET element at row i column j of dp TO 
                        the CEILING OF the element at row i MINUS one column j of dp DIVIDED BY speed MULTIPLIED BY speed PLUS element at position i of dist  
                END IF  
                IF j IS GREATER THAN zero THEN  
                    SET element at row i column j of dp TO 
                        the MINIMUM OF the element at row i column j of dp AND the element at row i MINUS one column j MINUS one of dp PLUS element at position i of dist  
                END IF  
            END FOR  
        END FOR  
        
        FOR each j FROM zero TO n MINUS one  
            IF element at row n MINUS one column j of dp IS LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed THEN  
                RETURN j  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
