CLASS Solution  
    FUNCTION optimalDivision WITH PARAMETER nums  
        IF the LENGTH OF nums EQUALS one  
            RETURN the STRING CONVERSION OF the ELEMENT at position zero of nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the STRING CONCATENATION OF the ELEMENT at position zero of nums FOLLOWED BY the DIVISION WORD FOLLOWED BY the ELEMENT at position one of nums  
        END IF  
        RETURN the STRING CONCATENATION OF the ELEMENT at position zero of nums FOLLOWED BY the DIVISION WORD FOLLOWED BY the OPENING PARENTHESIS WORD FOLLOWED BY the STRING JOINING OF the STRING CONVERSION OF each ELEMENT starting from position one TO THE LAST POSITION of nums WITH the DIVISION WORD AS SEPARATOR FOLLOWED BY the CLOSING PARENTHESIS WORD  
    END FUNCTION  
END CLASS
