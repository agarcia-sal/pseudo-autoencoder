CLASS Solution  
    FUNCTION compareVersion(version1 string version2 string) RETURNS integer  
        SET v1_revisions TO the result of splitting version1 by the period character  
        SET v2_revisions TO the result of splitting version2 by the period character  
        SET max_length TO the maximum of the length of v1_revisions AND the length of v2_revisions  
        FOR index FROM zero TO max_length MINUS one  
            IF index IS LESS THAN the length of v1_revisions  
                SET v1 TO the integer conversion of the element at position index of v1_revisions  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF index IS LESS THAN the length of v2_revisions  
                SET v2 TO the integer conversion of the element at position index of v2_revisions  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 IS LESS THAN v2  
                RETURN negative one  
            ELSE IF v1 IS GREATER THAN v2  
                RETURN one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
