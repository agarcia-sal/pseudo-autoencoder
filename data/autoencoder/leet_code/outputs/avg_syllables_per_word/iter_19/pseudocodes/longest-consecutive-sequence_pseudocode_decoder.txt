CLASS Solution  
    FUNCTION longestConsecutive(nums)  
        IF the LENGTH OF nums EQUALS zero  
            RETURN zero  
        END IF  
        SET num_set TO the CONVERTED SET OF nums  
        SET longest_streak TO zero  
        FOR each num IN num_set  
            IF the ELEMENT AT POSITION num MINUS one DOES NOT EXIST IN num_set  
                SET current_num TO num  
                SET current_streak TO one  
                WHILE the ELEMENT AT POSITION current_num PLUS one EXISTS IN num_set  
                    INCREMENT current_num BY one  
                    INCREMENT current_streak BY one  
                END WHILE  
                SET longest_streak TO THE GREATER VALUE BETWEEN longest_streak AND current_streak  
            END IF  
        END FOR  
        RETURN longest_streak  
    END FUNCTION  
END CLASS
