CLASS Solution  
    FUNCTION findSubstringInWraproundString(s)  
        SET max_length TO a new dictionary mapping each character from a to z TO zero  
        SET current_length TO zero  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            IF index GREATER THAN zero AND (the character code of element at position index of s MINUS the character code of element at position index MINUS one of s EQUALS one OR element at position index of s EQUALS the character a AND element at position index MINUS one of s EQUALS the character z)  
                INCREMENT current_length BY one  
            ELSE  
                SET current_length TO one  
            END IF  
            SET the value at element at position index of s IN max_length TO the GREATER OF the current value at element at position index of s IN max_length AND current_length  
        END FOR  
        SET total_unique_substrings TO the SUM OF all values IN max_length  
        RETURN total_unique_substrings  
    END FUNCTION  
END CLASS
