CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO the associative collection of element frequencies in nums  
        SET unique_nums TO the list of keys from count arranged in ascending order  
        DEFINE dp AS the list of zeros with the size equal to the length of unique_nums plus one  
        SET the element at position one of dp TO the element at position zero of unique_nums MULTIPLIED BY the element corresponding to element at position zero of unique_nums in count  
        FOR index FROM two TO the length of unique_nums plus one  
            SET num TO the element at position index MINUS one of unique_nums  
            IF num EQUALS the element at position index MINUS two of unique_nums PLUS one  
                SET the element at position index of dp TO the greater value BETWEEN the element at position index MINUS one of dp AND the element at position index MINUS two of dp PLUS num MULTIPLIED BY the element corresponding to num in count  
            ELSE  
                SET the element at position index of dp TO the element at position index MINUS one of dp PLUS num MULTIPLIED BY the element corresponding to num in count  
            END IF  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
