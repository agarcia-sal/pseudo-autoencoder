CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty list  
        FOR i FROM zero TO n MINUS one MINUS one  
            PUSH ONTO min_heap A TUPLE OF the element at position i of arr DIVIDED BY the element at position n MINUS one of arr AND the values i AND n MINUS one  
        END FOR  
        FOR counter FROM zero TO k MINUS one MINUS one  
            POP THE SMALLEST ELEMENT FROM min_heap INTO underscore_unused AND i AND j  
            IF j MINUS one IS GREATER THAN i  
                PUSH ONTO min_heap A TUPLE OF the element at position i of arr DIVIDED BY the element at position j MINUS one of arr AND the values i AND j MINUS one  
            END IF  
        END FOR  
        POP THE SMALLEST ELEMENT FROM min_heap INTO underscore_unused AND i AND j  
        RETURN A LIST CONTAINING the element at position i of arr AND the element at position j of arr  
    END FUNCTION  
END CLASS
