CLASS Solution  
    FUNCTION findClosestElements(arr, k, x)  
        SET left TO the result of calling bisect left with arr and x MINUS one  
        SET right TO left PLUS one  
        WHILE the result of right MINUS left MINUS one IS LESS THAN k  
            IF left EQUALS negative one OR the result of right IS LESS THAN the LENGTH OF arr AND the absolute value of the element at position right of arr MINUS x IS LESS THAN the absolute value of the element at position left of arr MINUS x  
                INCREMENT right BY one  
            ELSE  
                DECREMENT left BY one  
            END IF  
        END WHILE  
        RETURN the sublist of arr from position left PLUS one TO position right  
    END FUNCTION  
END CLASS
