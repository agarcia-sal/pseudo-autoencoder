CLASS Solution  
    FUNCTION findMinHeightTrees(n INTEGER, edges LIST OF LIST OF INTEGER) RETURNS LIST OF INTEGER  
        IF n EQUALS one  
            RETURN list containing zero  
        END IF  
  
        SET graph TO a default dictionary mapping each key to a list  
        SET degree TO a list of size n initialized with zero  
  
        FOR each pair u AND v IN edges  
            APPEND v TO the list mapped by u in graph  
            APPEND u TO the list mapped by v in graph  
            INCREMENT the element at position u of degree BY one  
            INCREMENT the element at position v of degree BY one  
        END FOR  
  
        SET leaves TO a double ended queue initialized with all indices i FROM zero TO n MINUS one WHERE the element at position i of degree EQUALS one  
  
        SET remaining_nodes TO n  
        WHILE remaining_nodes GREATER THAN two  
            SET leaves_count TO the number of elements in leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR each index FROM one TO leaves_count  
                REMOVE an element from the front of leaves ASSIGN TO leaf  
                FOR each neighbor IN the list mapped by leaf in graph  
                    DECREMENT the element at position neighbor of degree BY one  
                    IF the element at position neighbor of degree EQUALS one  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
  
        RETURN a list formed from all elements in leaves  
    END FUNCTION  
END CLASS
