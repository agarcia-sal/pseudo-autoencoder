CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO zero  
        FOR bit_position FROM thirty one TO zero DECREMENT BY one  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            SET prefixes TO an empty collection of unique elements  
            FOR each number IN nums  
                SET prefix TO the result of shifting number to the right by bit_position places  
                ADD prefix TO prefixes  
            END FOR  
            FOR each prefix_element IN prefixes  
                IF the result of current_candidate EXCLUSIVE OR prefix_element EXISTS IN prefixes THEN  
                    SET max_xor TO current_candidate  
                    EXIT the current loop  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
