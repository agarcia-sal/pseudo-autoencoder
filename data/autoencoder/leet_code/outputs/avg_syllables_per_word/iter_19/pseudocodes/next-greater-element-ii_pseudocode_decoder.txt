CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO LENGTH OF nums  
        SET result TO a list of n elements each set TO negative one  
        SET stack TO an empty list  
        FOR i FROM zero TO two TIMES n MINUS one  
            SET index TO i MODULO n  
            WHILE the LENGTH OF stack IS GREATER THAN zero AND the element at position element at last position of stack of nums IS LESS THAN the element at position index of nums  
                SET popped_position TO REMOVE last element FROM stack  
                SET element at position popped_position of result TO element at position index of nums  
            END WHILE  
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
