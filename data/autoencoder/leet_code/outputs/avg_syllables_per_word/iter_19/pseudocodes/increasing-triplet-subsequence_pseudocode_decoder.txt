CLASS Solution  
    FUNCTION increasingTriplet WITH parameter nums  
        SET first TO value representing positive infinity  
        SET second TO value representing positive infinity  
        FOR each num IN nums  
            IF num IS LESS THAN OR EQUAL TO first  
                SET first TO num  
            ELSE IF num IS LESS THAN OR EQUAL TO second  
                SET second TO num  
            ELSE  
                RETURN true value  
            END IF  
        END FOR  
        RETURN false value  
    END FUNCTION  
END CLASS
