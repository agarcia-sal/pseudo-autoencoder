CLASS Solution  
    FUNCTION superEggDrop WITH PARAMETERS k AND n  
        SET dp TO a list of zeros with LENGTH k PLUS one  
        SET moves TO zero  
        WHILE the element at position k of dp is LESS THAN n  
            INCREMENT moves BY one  
            FOR eggs FROM k DOWN TO one  
                SET the element at position eggs of dp TO one PLUS the element at position eggs MINUS one of dp PLUS the element at position eggs of dp  
            END FOR  
        END WHILE  
        RETURN moves  
    END FUNCTION  
END CLASS
