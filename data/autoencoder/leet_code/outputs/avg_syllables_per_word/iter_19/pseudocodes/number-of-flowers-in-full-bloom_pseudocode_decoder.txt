CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO the sorted list of the element at position zero of each pair in flowers  
        SET end_times TO the sorted list of the element at position one of each pair in flowers  
        SET result TO an empty list  
        FOR each person IN people  
            SET started TO the insertion position to the right where person fits in start_times  
            SET ended TO the insertion position to the left where person fits in end_times  
            APPEND the difference of started MINUS ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
