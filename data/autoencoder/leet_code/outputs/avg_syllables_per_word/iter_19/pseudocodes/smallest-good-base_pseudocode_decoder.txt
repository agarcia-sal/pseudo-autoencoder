CLASS Solution  
    FUNCTION smallestGoodBase(n)  
        SET n TO the integer conversion of the parameter n  
        SET max_length TO the integer value of the base two logarithm of n PLUS one  
        FOR m FROM max_length DOWN TO two  
            SET k TO the integer value of the nth root of n where n is one DIVIDED BY m MINUS one  
            IF k LESS THAN two  
                CONTINUE to the next iteration of the loop  
            END IF  
            SET num TO the integer division of k raised to the power m MINUS one BY k MINUS one  
            IF num EQUALS n  
                RETURN the string conversion of k  
            END IF  
        END FOR  
        RETURN the string conversion of n MINUS one  
    END FUNCTION  
END CLASS
