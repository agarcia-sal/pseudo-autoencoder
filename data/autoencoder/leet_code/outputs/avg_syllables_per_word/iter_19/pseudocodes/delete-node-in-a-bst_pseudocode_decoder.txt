CLASS TreeNode  
    FUNCTION __init__ (val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node (values)  
    IF values IS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value at position zero of values  
    SET i TO one  
    CREATE an empty queue named queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the front of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.left TO a new TreeNode with value at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node.right TO a new TreeNode with value at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree (p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the value of p IS NOT EQUALS the value of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree with parameters left child of p and left child of q AND the result of is_same_tree with parameters right child of p and right child of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteNode (self, root, key)  
        IF root IS None  
            RETURN None  
        END IF  
        IF the value of root IS GREATER THAN key  
            SET the left child of root TO the result of deleteNode called with parameters left child of root and key  
            RETURN root  
        END IF  
        IF the value of root IS LESS THAN key  
            SET the right child of root TO the result of deleteNode called with parameters right child of root and key  
            RETURN root  
        END IF  
        IF the left child of root IS None  
            RETURN the right child of root  
        END IF  
        IF the right child of root IS None  
            RETURN the left child of root  
        END IF  
        SET node TO the right child of root  
        WHILE the left child of node IS NOT None  
            SET node TO the left child of node  
        END WHILE  
        SET the left child of node TO the left child of root  
        SET root TO the right child of root  
        RETURN root  
    END FUNCTION  
END CLASS
