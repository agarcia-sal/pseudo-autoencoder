CLASS Solution  
    FUNCTION insert(intervals AS List of List of integer, newInterval AS List of integer) RETURNS List of List of integer  
        IF intervals IS EMPTY  
            RETURN a list containing newInterval  
        END IF  
        
        SET merged TO an empty list  
        SET i TO zero  
        SET n TO the LENGTH OF intervals  
        
        WHILE i IS LESS THAN n AND the element at position one of element at position i of intervals IS LESS THAN element at position zero of newInterval  
            APPEND element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        WHILE i IS LESS THAN n AND the element at position zero of element at position i of intervals IS LESS THAN OR EQUAL TO element at position one of newInterval  
            SET element at position zero of newInterval TO the LESSER OF element at position zero of newInterval AND element at position zero of element at position i of intervals  
            SET element at position one of newInterval TO the GREATER OF element at position one of newInterval AND element at position one of element at position i of intervals  
            INCREMENT i BY one  
        END WHILE  
        
        APPEND newInterval TO merged  
        
        WHILE i IS LESS THAN n  
            APPEND element at position i of intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN merged  
    END FUNCTION  
END CLASS
