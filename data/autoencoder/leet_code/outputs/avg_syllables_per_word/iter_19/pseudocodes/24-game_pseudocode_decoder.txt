CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the element at position zero of nums MINUS twenty four IS LESS THAN one MULTIPLIED BY ten TO THE POWER OF MINUS six  
            END IF  
            
            FOR each index i FROM zero TO the LENGTH OF nums MINUS one  
                FOR each index j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO an empty list  
                        FOR each index k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND the element at position k of nums TO new_nums  
                            END IF  
                        END FOR  
                        
                        FOR each operation op IN the collection consisting of true division multiplication addition subtraction  
                            IF (op IS addition OR op IS multiplication) AND j GREATER THAN i  
                                CONTINUE TO the next iteration  
                            END IF  
                            
                            IF op IS NOT true division OR the element at position j of nums NOT EQUALS zero  
                                APPEND the result of applying op TO the element at position i of nums AND the element at position j of nums TO new_nums  
                                IF solve(new_nums)  
                                    RETURN True  
                                END IF  
                                REMOVE the last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
