CLASS Solution  
    FUNCTION shortestSubarray(nums AS List OF integers, k AS integer) RETURNS integer  
        SET prefix_sum TO a list containing zero as the only element  
        FOR each num IN nums  
            APPEND the element at the last position of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        SET dq TO an empty double ended queue  
        SET min_length TO positive infinity  
        FOR each index i AND element current_sum IN prefix_sum with indices  
            WHILE dq is not empty AND current_sum MINUS the element at the position at the front of dq in prefix_sum IS GREATER THAN OR EQUAL TO k  
                SET min_length TO the smaller value between min_length AND i MINUS the element removed from the front of dq  
            END WHILE  
            WHILE dq is not empty AND current_sum IS LESS THAN OR EQUAL TO the element at the position at the back of dq in prefix_sum  
                REMOVE the element at the back of dq  
            END WHILE  
            APPEND i TO the back of dq  
        END FOR  
        IF min_length IS NOT EQUAL TO positive infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
