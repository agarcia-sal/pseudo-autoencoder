CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR each i FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            SET good TO a LIST of Boolean FALSE of LENGTH n  
            SET bad TO a LIST of Boolean FALSE of LENGTH n  
            FOR each j FROM zero TO n MINUS one  
                IF the BITWISE AND of i AND two RAISED TO THE POWER OF j IS NOT EQUAL TO zero  
                    SET element at position j of good TO TRUE  
                ELSE  
                    SET element at position j of bad TO TRUE  
                END IF  
            END FOR  
            SET valid TO TRUE  
            FOR each j FROM zero TO n MINUS one  
                IF element at position j of good IS TRUE  
                    FOR each k FROM zero TO n MINUS one  
                        IF the element at position k of the element at position j of statements EQUALS zero AND element at position k of good IS TRUE  
                            SET valid TO FALSE  
                            BREAK out of inner FOR loop  
                        ELSE IF the element at position k of the element at position j of statements EQUALS one AND element at position k of bad IS TRUE  
                            SET valid TO FALSE  
                            BREAK out of inner FOR loop  
                        END IF  
                    END FOR  
                    IF valid IS FALSE  
                        BREAK out of outer FOR loop  
                    END IF  
                END IF  
            END FOR  
            IF valid IS TRUE  
                SET max_good TO the GREATER OF max_good AND the SUM OF the Boolean values in good  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
