CLASS Solution  
    FUNCTION findIntegers(n)  
        SET binary_representation TO the binary form of n starting from the second character to the end  
        SET length TO the LENGTH OF binary_representation  

        CREATE a list dp with length plus one elements all initialized to zero  
        SET element at position zero of dp TO one  
        SET element at position one of dp TO two  

        FOR index FROM two TO length  
            SET element at position index of dp TO element at position index minus one of dp PLUS element at position index minus two of dp  
        END FOR  

        SET result TO zero  
        SET previous_bit TO zero  

        FOR index FROM zero TO length minus one  
            IF element at position index of binary_representation EQUALS the character one  
                INCREMENT result BY element at position length MINUS index MINUS one of dp  
                IF previous_bit EQUALS one  
                    DECREMENT result BY one  
                    BREAK the loop  
                END IF  
            END IF  
            SET previous_bit TO one IF element at position index of binary_representation EQUALS the character one OTHERWISE zero  
        END FOR  

        RETURN result PLUS one  
    END FUNCTION  
END CLASS
