CLASS Solution  
    FUNCTION triangleNumber(nums)  
        CALL the SORT METHOD ON nums  
        SET count TO zero  
        SET n TO the LENGTH OF nums  
        FOR k FROM n MINUS one DOWN TO two  
            SET i TO zero  
            SET j TO k MINUS one  
            WHILE i IS LESS THAN j  
                IF the ELEMENT at position i of nums PLUS the ELEMENT at position j of nums IS GREATER THAN the ELEMENT at position k of nums  
                    INCREMENT count BY j MINUS i  
                    DECREMENT j BY one  
                ELSE  
                    INCREMENT i BY one  
                END IF  
            END WHILE  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
