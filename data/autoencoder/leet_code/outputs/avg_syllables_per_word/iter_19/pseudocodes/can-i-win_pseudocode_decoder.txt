CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the product of maxChoosableInteger AND the sum of maxChoosableInteger PLUS one DIVIDED BY two is LESS THAN desiredTotal  
            RETURN false  
        END IF  
        
        IF desiredTotal IS LESS THAN OR EQUAL TO zero  
            RETURN true  
        END IF  
        
        SET memo TO an empty dictionary  
        
        FUNCTION can_win(current_total, used_numbers)  
            IF the tuple of current_total AND used_numbers IS IN memo  
                RETURN the value associated with the tuple of current_total AND used_numbers in memo  
            END IF  
            
            FOR each number FROM one TO maxChoosableInteger  
                IF used_numbers BITWISE AND a single one BITSHIFTED LEFT BY number IS EQUAL TO zero  
                    IF the sum of current_total AND number IS GREATER THAN OR EQUAL TO desiredTotal OR logical NOT of can_win the sum of current_total AND number AND bitwise OR of used_numbers AND a single one BITSHIFTED LEFT BY number holds  
                        SET the value associated with the tuple of current_total AND used_numbers in memo TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            
            SET the value associated with the tuple of current_total AND used_numbers in memo TO false  
            RETURN false  
        END FUNCTION  
        
        RETURN can_win zero AND zero  
    END FUNCTION  
END CLASS
