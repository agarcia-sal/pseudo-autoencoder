CLASS Solution  
    FUNCTION minDistance WITH parameters houses AND k  
        CALL sort ON houses  
        SET n TO the LENGTH OF houses  
        SET cost TO the TWO DIMENSIONAL LIST of size n BY n filled WITH zero  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET value_at position i AND j OF cost TO the VALUE at position i PLUS one AND j MINUS one OF cost PLUS the VALUE at position j OF houses MINUS the VALUE at position i OF houses  
            END FOR  
        END FOR  
        SET dp TO the TWO DIMENSIONAL LIST of size n PLUS one BY k PLUS one filled WITH infinite  
        SET value at position zero AND zero OF dp TO zero  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET value at position i AND j OF dp TO the MINIMUM OF the value at position i AND j OF dp AND the VALUE at position p AND j MINUS one OF dp PLUS the VALUE at position p AND i MINUS one OF cost  
                END FOR  
            END FOR  
        END FOR  
        RETURN the value at position n AND k OF dp  
    END FUNCTION  
END CLASS
