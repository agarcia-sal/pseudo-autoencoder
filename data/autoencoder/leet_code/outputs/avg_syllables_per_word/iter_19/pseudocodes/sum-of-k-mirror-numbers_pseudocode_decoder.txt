CLASS Solution  
    FUNCTION kMirror WITH PARAMETERS k AND n  
        FUNCTION is_palindrome WITH PARAMETER s  
            RETURN the condition that s EQUALS the reverse of s  
        END FUNCTION  
        
        FUNCTION to_base_k WITH PARAMETERS num AND k  
            IF num EQUALS zero THEN  
                RETURN the string zero  
            END IF  
            SET digits TO an empty list  
            WHILE num IS NOT zero  
                APPEND the remainder of num DIVIDED BY k converted to integer TO digits  
                SET num TO the integer result of num DIVIDED BY k  
            END WHILE  
            RETURN the concatenation of all elements converted to string in digits in reverse order  
        END FUNCTION  
        
        FUNCTION generate_palindromes WITH PARAMETER length  
            IF length EQUALS one THEN  
                FOR each integer i FROM one TO nine  
                    YIELD the string form of i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO THE power of the integer division of length MINUS one BY two  
                SET end TO ten RAISED TO THE power of the integer division of length PLUS one BY two  
                FOR each integer half FROM start TO one LESS THAN end  
                    SET half_str TO the string form of half  
                    IF length MODULO two EQUALS zero THEN  
                        SET full TO the concatenation of half_str AND the reverse of half_str  
                    ELSE  
                        SET full TO the concatenation of half_str AND half_str EXCLUDING the last character IN reverse order  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO an empty list  
        SET length TO one  
        WHILE the LENGTH OF k_mirror_numbers IS LESS THAN n  
            FOR each pal IN generate_palindromes WITH length  
                SET num TO the integer form of pal  
                IF is_palindrome WITH to_base_k OF num AND k RETURNS true THEN  
                    APPEND num TO k_mirror_numbers  
                    IF the LENGTH OF k_mirror_numbers EQUALS n THEN  
                        BREAK the loop  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN the sum of all elements in k_mirror_numbers  
    END FUNCTION  
END CLASS
