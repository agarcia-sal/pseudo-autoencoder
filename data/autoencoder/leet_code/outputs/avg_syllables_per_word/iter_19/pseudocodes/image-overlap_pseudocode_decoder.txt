CLASS Solution  
    FUNCTION largestOverlap(img1 AS List OF List OF integer, img2 AS List OF List OF integer)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of img1 EQUALS one  
                    APPEND the pair consisting of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF the element at position j of the element at position i of img2 EQUALS one  
                    APPEND the pair consisting of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO an empty mapping from pairs TO integers with default zero  
        FOR each pair x1 AND y1 IN points1  
            FOR each pair x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the element at key pair dx AND dy OF overlap_count BY one  
            END FOR  
        END FOR  
        SET maximum_overlap TO zero  
        FOR each value IN all values of overlap_count  
            IF value GREATER THAN maximum_overlap  
                SET maximum_overlap TO value  
            END IF  
        END FOR  
        RETURN maximum_overlap  
    END FUNCTION  
END CLASS
