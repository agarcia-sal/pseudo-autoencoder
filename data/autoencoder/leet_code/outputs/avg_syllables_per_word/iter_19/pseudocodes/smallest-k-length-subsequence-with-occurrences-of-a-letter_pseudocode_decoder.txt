CLASS Solution  
    FUNCTION smallestSubsequence(s, k, letter, repetition)  
        SET letter_count TO the number of times letter appears in s  
        SET stack TO an empty list  
        FOR each index i AND character char IN the enumeration of string s  
            WHILE the stack is not empty AND character char is less than the element at position the last index of stack AND the sum of the length of string s minus index i PLUS the length of stack MINUS one is greater than or equal to k  
                IF the element at position the last index of stack EQUALS letter  
                    IF letter_count is greater than repetition  
                        INCREMENT repetition BY one  
                        REMOVE the element at position the last index from stack  
                    ELSE  
                        BREAK the WHILE loop  
                    END IF  
                ELSE  
                    REMOVE the element at position the last index from stack  
                END IF  
            END WHILE  
            IF the length of stack is less than k  
                IF character char EQUALS letter  
                    APPEND character char TO stack  
                    DECREMENT repetition BY one  
                ELSE IF the difference of k MINUS the length of stack is greater than repetition  
                    APPEND character char TO stack  
                END IF  
                IF character char EQUALS letter  
                    DECREMENT letter_count BY one  
                END IF  
            END IF  
        END FOR  
        WHILE the length of stack is less than k  
            APPEND letter TO stack  
        END WHILE  
        RETURN the concatenation of all elements in stack as a single string  
    END FUNCTION  
END CLASS
