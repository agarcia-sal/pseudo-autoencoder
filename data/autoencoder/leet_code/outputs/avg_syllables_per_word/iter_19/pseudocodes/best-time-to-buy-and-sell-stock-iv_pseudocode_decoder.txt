CLASS Solution  
    FUNCTION maxProfit WITH PARAMETERS k AND prices  
        IF prices IS EMPTY OR k EQUALS zero  
            RETURN zero  
        END IF  
        
        SET n TO the LENGTH OF prices  
        
        IF k IS GREATER THAN OR EQUAL TO n DIVIDED BY two  
            SET profit TO zero  
            FOR index FROM one TO n MINUS one  
                IF element at position index of prices IS GREATER THAN element at position index MINUS one of prices  
                    INCREMENT profit BY element at position index of prices MINUS element at position index MINUS one of prices  
                END IF  
            END FOR  
            RETURN profit  
        END IF  
        
        INITIALIZE list hold WITH k PLUS one ELEMENTS EACH SET TO NEGATIVE infinity  
        INITIALIZE list release WITH k PLUS one ELEMENTS EACH SET TO zero  
        
        FOR each price IN prices  
            FOR transaction_index FROM one TO k  
                SET hold at position transaction_index TO THE GREATER OF hold at position transaction_index AND release at position transaction_index MINUS one MINUS price  
                SET release at position transaction_index TO THE GREATER OF release at position transaction_index AND hold at position transaction_index PLUS price  
            END FOR  
        END FOR  
        
        RETURN element at position k of release  
    END FUNCTION  
END CLASS
