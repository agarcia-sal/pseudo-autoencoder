CLASS Solution  
    FUNCTION maxVacationDays(flights AS List OF List OF integer, days AS List OF List OF integer) RETURNS integer  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF element at position zero of days  
        
        SET dp TO a list of length n where each element IS negative one  
        SET element at position zero of dp TO zero  
        
        SET graph TO a mapping from each index i from zero TO n MINUS one TO an empty list  
        
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                IF element at position i of flights AT position j EQUALS one  
                    APPEND j TO element at position i OF graph  
                END IF  
            END FOR  
        END FOR  
        
        FOR each week FROM zero TO k MINUS one  
            SET new_dp TO a list of length n where each element IS negative one  
            FOR each i FROM zero TO n MINUS one  
                IF element at position i of dp EQUALS negative one  
                    CONTINUE to the next iteration  
                END IF  
                IF element at position i of new_dp LESS THAN element at position i of dp PLUS element at position i of days AT position week  
                    SET element at position i of new_dp TO element at position i of dp PLUS element at position i of days AT position week  
                END IF  
                FOR each j IN element at position i OF graph  
                    IF element at position j of new_dp LESS THAN element at position i of dp PLUS element at position j of days AT position week  
                        SET element at position j of new_dp TO element at position i of dp PLUS element at position j of days AT position week  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        
        RETURN the maximum element IN dp  
    END FUNCTION  
END CLASS
