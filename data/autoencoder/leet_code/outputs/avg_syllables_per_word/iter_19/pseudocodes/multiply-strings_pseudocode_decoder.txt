CLASS Solution  
    FUNCTION multiply(num1, num2)  
        IF num1 EQUALS the string zero OR num2 EQUALS the string zero  
            RETURN the string zero  
        END IF  
        SET result TO a list of zeros with length EQUAL TO the LENGTH OF num1 PLUS the LENGTH OF num2  
        SET num1 TO the reverse of num1  
        SET num2 TO the reverse of num2  
        FOR each index i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR each index j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the numerical value of the element at position i of num1 MINUS the numerical value of the character zero MULTIPLIED BY the numerical value of the element at position j of num2 MINUS the numerical value of the character zero  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum TO mul PLUS the element at position p1 of result  
                SET the element at position p1 of result TO sum MODULO ten  
                INCREMENT the element at position p2 of result BY sum DIVIDED BY ten  
            END FOR  
        END FOR  
        SET result_to_string TO the concatenation of string representations of elements in the reverse of result  
        RETURN result_to_string with leading characters equal to the string zero REMOVED  
    END FUNCTION  
END CLASS
