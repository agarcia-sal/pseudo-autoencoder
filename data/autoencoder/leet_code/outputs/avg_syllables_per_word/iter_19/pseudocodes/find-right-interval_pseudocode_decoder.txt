CLASS Solution  
    FUNCTION findRightInterval WITH PARAMETERS intervals  
        SET start_times TO the list of pairs each consisting of the START element of an interval AND its index from intervals sorted by the START element in ascending order  
        SET result TO an empty list  
        FOR each pair IGNORING the first element AND named end IN intervals  
            SET idx TO the position found by binary searching for the smallest element in start_times WHERE the START element IS GREATER THAN OR EQUAL TO end  
            IF idx IS LESS THAN the LENGTH OF start_times  
                APPEND the SECOND element at position idx of start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
