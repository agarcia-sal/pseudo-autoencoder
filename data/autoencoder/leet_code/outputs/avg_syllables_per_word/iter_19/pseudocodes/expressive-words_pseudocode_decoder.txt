CLASS Solution  
    FUNCTION expressiveWords(s, words)  
        FUNCTION compress(word)  
            IF word IS NOT PRESENT  
                RETURN an empty list  
            END IF  
            SET count TO one  
            SET compressed TO an empty list  
            FOR index FROM one TO the LENGTH OF word MINUS one  
                IF the element at position index of word EQUALS the element at position index MINUS one of word  
                    INCREMENT count BY one  
                ELSE  
                    APPEND the pair consisting of the element at position index MINUS one of word AND count TO compressed  
                    SET count TO one  
                END IF  
            END FOR  
            APPEND the pair consisting of the last element of word AND count TO compressed  
            RETURN compressed  
        END FUNCTION  
        
        FUNCTION is_stretchy(s_compressed, word_compressed)  
            IF the LENGTH OF s_compressed IS NOT EQUAL TO the LENGTH OF word_compressed  
                RETURN false  
            END IF  
            FOR each pair of elements (sc AND sc_count) IN s_compressed AND (wc AND wc_count) IN word_compressed in order  
                IF sc IS NOT EQUAL TO wc  
                    RETURN false  
                END IF  
                IF sc_count IS NOT EQUAL TO wc_count AND (sc_count IS LESS THAN three OR sc_count IS LESS THAN wc_count)  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SET s_compressed TO the result of compress applied to s  
        SET stretchy_count TO zero  
        
        FOR each word IN words  
            SET word_compressed TO the result of compress applied to word  
            IF is_stretchy with s_compressed AND word_compressed RETURNS true  
                INCREMENT stretchy_count BY one  
            END IF  
        END FOR  
        
        RETURN stretchy_count  
    END FUNCTION  
END CLASS
