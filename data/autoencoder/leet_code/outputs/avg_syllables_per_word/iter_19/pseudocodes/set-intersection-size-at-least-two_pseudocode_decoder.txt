CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals BY the element at position one of interval AND NEGATIVE of the element at position zero of interval in descending order  
        SET first TO the element at position one of the first interval MINUS one  
        SET second TO the element at position one of the first interval  
        SET count TO two  
        FOR each interval starting from the second interval AS start AND end  
            IF start IS GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
