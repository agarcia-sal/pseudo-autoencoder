CLASS Solution  
    FUNCTION getMoneyAmount WITH parameter n  
        DEFINE AUXILIARY FUNCTION dp WITH parameters left AND right  
            IF left IS GREATER THAN OR EQUAL TO right  
                RETURN zero  
            END IF  
            SET min_cost TO positive infinity  
            FOR pivot FROM left TO right  
                SET cost TO pivot PLUS THE MAXIMUM BETWEEN dp OF left AND pivot MINUS one AND dp OF pivot PLUS one AND right  
                SET min_cost TO THE MINIMUM BETWEEN min_cost AND cost  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN dp OF one AND n  
    END FUNCTION  
END CLASS
