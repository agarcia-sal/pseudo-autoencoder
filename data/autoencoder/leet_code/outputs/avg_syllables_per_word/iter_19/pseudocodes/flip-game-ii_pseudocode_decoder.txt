CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO an empty dictionary  
        
        FUNCTION can_win_helper(state)  
            IF state EXISTS IN memo THEN  
                RETURN the element at position state of memo  
            END IF  
            
            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one  
                IF the substring from position index TO index PLUS one of state EQUALS the string consisting of two plus signs THEN  
                    SET new_state TO the concatenation of the substring from position zero TO index of state PLUS the string consisting of two minus signs PLUS the substring from position index PLUS two TO the LENGTH OF state MINUS one of state  
                    IF NOT the result of calling can_win_helper WITH new_state THEN  
                        SET the element at position state of memo TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET the element at position state of memo TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN the result of calling can_win_helper WITH currentState  
    END FUNCTION  
END CLASS
