CLASS Solution  
    FUNCTION waysToDistribute(n, k)  
        SET MOD TO one billion PLUS seven  
        INITIALIZE dp AS a two dimensional list of size n PLUS one BY k PLUS one FILLED WITH zeroes  
        SET element at position zero of dp AT position zero TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET element at position i of dp AT position j TO the remainder when the sum of j MULTIPLIED BY element at position i MINUS one of dp AT position j AND element at position i MINUS one of dp AT position j MINUS one IS DIVIDED BY MOD  
            END FOR  
        END FOR  
        RETURN element at position n of dp AT position k  
    END FUNCTION  
END CLASS
