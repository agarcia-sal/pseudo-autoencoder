CLASS Solution  
    FUNCTION findKthNumber(n, k)  
        FUNCTION count_steps(prefix, n)  
            SET next_prefix TO prefix PLUS one  
            SET steps TO zero  
            WHILE prefix IS LESS THAN OR EQUAL TO n  
                INCREMENT steps BY THE MINIMUM OF the DIFFERENCE OF n MINUS prefix PLUS one AND next_prefix MINUS prefix  
                SET prefix TO prefix MULTIPLIED BY ten  
                SET next_prefix TO next_prefix MULTIPLIED BY ten  
            END WHILE  
            RETURN steps  
        END FUNCTION  

        SET current TO one  
        DECREMENT k BY one  
        WHILE k IS GREATER THAN zero  
            SET steps TO THE RESULT OF count_steps WITH PARAMETERS current AND n  
            IF steps IS LESS THAN OR EQUAL TO k  
                INCREMENT current BY one  
                DECREMENT k BY steps  
            ELSE  
                SET current TO current MULTIPLIED BY ten  
                DECREMENT k BY one  
            END IF  
        END WHILE  
        RETURN current  
    END FUNCTION  
END CLASS
