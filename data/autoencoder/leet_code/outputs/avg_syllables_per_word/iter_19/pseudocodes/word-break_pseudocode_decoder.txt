CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO the set constructed from wordDict  
        SET dp TO a list of FALSE values with the length equal to the LENGTH OF s PLUS one  
        SET the element at position zero of dp TO TRUE  
        
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM zero TO i MINUS one  
                IF the element at position j of dp EQUALS TRUE AND the substring from position j PLUS one TO i of s EXISTS IN wordSet  
                    SET the element at position i of dp TO TRUE  
                    BREAK out of inner FOR loop  
                END IF  
            END FOR  
        END FOR  
        
        RETURN the element at position LENGTH OF s of dp  
    END FUNCTION  
END CLASS
