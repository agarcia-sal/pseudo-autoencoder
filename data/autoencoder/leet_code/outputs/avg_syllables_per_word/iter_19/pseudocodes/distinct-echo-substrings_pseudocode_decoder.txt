CLASS Solution  
    FUNCTION distinctEchoSubstrings(text)  
        SET n TO the LENGTH OF text  
        SET substrings TO an EMPTY COLLECTION OF UNIQUE ELEMENTS  
        FOR each length FROM one TO the INTEGER DIVISION OF n BY two INCLUSIVE  
            FOR each start FROM zero TO n MINUS two TIMES length INCLUSIVE  
                SET a TO the SUBSTRING from position start TO start PLUS length MINUS one of text  
                SET b TO the SUBSTRING from position start PLUS length TO start PLUS two TIMES length MINUS one of text  
                IF a EQUALS b  
                    ADD the concatenation of a AND b TO substrings  
                END IF  
            END FOR  
        END FOR  
        SET result TO the LENGTH OF substrings  
        RETURN result  
    END FUNCTION  
END CLASS
