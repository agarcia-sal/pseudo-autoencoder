CLASS Solution  
    FUNCTION uniquePaths WITH PARAMETERS m INTEGER n INTEGER  
        SET dp TO a two dimensional array of size m BY n where each element is set to one  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET element at position i of dp THEN element at position j TO element at position i MINUS one of dp THEN element at position j PLUS element at position i of dp THEN element at position j MINUS one  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one of dp THEN element at position n MINUS one  
    END FUNCTION  
END CLASS
