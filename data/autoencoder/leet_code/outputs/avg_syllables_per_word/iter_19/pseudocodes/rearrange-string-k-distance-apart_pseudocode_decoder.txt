CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        
        SET freq TO the frequency count of each character IN s  
        SET max_heap TO the list of pairs of negative frequency AND character FOR each character AND count IN freq  
        TRANSFORM max_heap INTO a heap data structure  
        
        SET wait_queue TO an empty list  
        SET result TO an empty list  
        
        WHILE max_heap IS NOT empty  
            REMOVE AND RETURN the smallest element FROM max_heap AND ASSIGN TO count AND char  
            APPEND char TO result  
            APPEND the pair of count INCREMENTED BY one AND char TO wait_queue  
            
            IF the LENGTH OF wait_queue EQUALS k  
                REMOVE AND RETURN the first element FROM wait_queue AND ASSIGN TO count AND char  
                IF count IS LESS THAN zero  
                    ADD the pair of count AND char INTO max_heap  
                END IF  
            END IF  
        END WHILE  
        
        IF the LENGTH OF result IS NOT EQUAL TO the LENGTH OF s  
            RETURN an empty string  
        END IF  
        
        RETURN the concatenation of all characters IN result AS a string  
    END FUNCTION  
END CLASS
