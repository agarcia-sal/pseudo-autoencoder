CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO NEW INSTANCE OF TreeNode WITH val EQUAL TO element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue data structure  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO the element REMOVED FROM THE FRONT OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.left TO NEW INSTANCE OF TreeNode WITH val EQUAL TO element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i of values IS NOT None  
            SET node.right TO NEW INSTANCE OF TreeNode WITH val EQUAL TO element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findClosestLeaf(self, root, k)  
        IF root.left IS None AND root.right IS None  
            RETURN root.val  
        END IF  
        SET graph TO a new default dictionary with list as default value  
        FUNCTION dfs(node, parent)  
            IF parent IS NOT None  
                APPEND parent.val TO the list at key node.val in graph  
                APPEND node.val TO the list at key parent.val in graph  
            END IF  
            IF node.left IS NOT None  
                CALL dfs(node.left, node)  
            END IF  
            IF node.right IS NOT None  
                CALL dfs(node.right, node)  
            END IF  
        END FUNCTION  
        CALL dfs(root, None)  
        SET queue TO a new double ended queue data structure with element k  
        SET visited TO a new empty set  
        WHILE queue IS NOT EMPTY  
            SET current TO the element REMOVED FROM THE FRONT OF queue  
            ADD current TO visited  
            IF the LENGTH OF the list at key current in graph IS EQUAL TO one AND current IS NOT EQUAL TO root.val  
                RETURN current  
            END IF  
            FOR each neighbor IN the list at key current in graph  
                IF neighbor IS NOT IN visited  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
