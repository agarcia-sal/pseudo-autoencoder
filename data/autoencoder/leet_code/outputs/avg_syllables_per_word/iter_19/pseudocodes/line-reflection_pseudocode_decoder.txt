CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the SET of the result of APPLYING the conversion of each element of points INTO tuples  
        SET min_x TO the MINIMUM x coordinate at position zero of each point IN unique_points  
        SET max_x TO the MAXIMUM x coordinate at position zero of each point IN unique_points  
        SET line_of_reflection TO the SUM of min_x AND max_x DIVIDED BY two  
        FOR each point IN unique_points  
            SET x TO the x coordinate at position zero of the point  
            SET y TO the y coordinate at position one of the point  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the tuple composed of reflected_x AND y IS NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
