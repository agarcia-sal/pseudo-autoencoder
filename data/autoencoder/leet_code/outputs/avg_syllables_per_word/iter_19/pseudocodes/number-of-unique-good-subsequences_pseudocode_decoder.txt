CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MODULO_VALUE TO ten TO THE POWER OF nine PLUS seven  
        SET count_of_subsequences_ending_with_zero TO zero  
        SET count_of_subsequences_ending_with_one TO zero  
        SET zero_exists_in_binary TO TRUE IF the binary string CONTAINS the character zero OTHERWISE FALSE  
        FOR each bit IN the binary string  
            IF the bit EQUALS one  
                SET count_of_subsequences_ending_with_one TO the REMAINDER OF the SUM of count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one PLUS one DIVIDED BY MODULO_VALUE  
            ELSE  
                SET count_of_subsequences_ending_with_zero TO the REMAINDER OF the SUM of count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one DIVIDED BY MODULO_VALUE  
            END IF  
        END FOR  
        RETURN the REMAINDER OF the SUM of count_of_subsequences_ending_with_zero PLUS count_of_subsequences_ending_with_one PLUS zero_exists_in_binary DIVIDED BY MODULO_VALUE  
    END FUNCTION  
END CLASS
