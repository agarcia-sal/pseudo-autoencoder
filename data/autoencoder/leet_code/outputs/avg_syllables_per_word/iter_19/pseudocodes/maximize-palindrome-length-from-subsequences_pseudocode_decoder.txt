CLASS Solution  
    FUNCTION longestPalindrome(word1 word one AS string, word2 word two AS string) RETURNS integer  
        SET s TO concatenation of word1 and word2  
        SET n TO the LENGTH OF s  
        SET dp TO a two dimensional list with n rows and n columns with all elements set TO zero  
        
        FOR i FROM zero TO n MINUS one  
            SET element at row i column i of dp TO one  
        END FOR  
        
        SET max_len TO zero  
        FOR length FROM two TO n  
            FOR i FROM zero TO n MINUS length  
                SET j TO i PLUS length MINUS one  
                IF element at position i of s EQUALS element at position j of s  
                    SET element at row i column j of dp TO element at row i PLUS one column j MINUS one of dp PLUS two  
                    IF i LESS THAN LENGTH OF word1 AND j GREATER THAN OR EQUAL TO LENGTH OF word1  
                        SET max_len TO the GREATER OF max_len AND element at row i column j of dp  
                    END IF  
                ELSE  
                    SET element at row i column j of dp TO the GREATER OF element at row i PLUS one column j of dp AND element at row i column j MINUS one of dp  
                END IF  
            END FOR  
        END FOR  
        
        RETURN max_len  
    END FUNCTION  
END CLASS
