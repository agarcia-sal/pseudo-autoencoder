CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            WHILE element at position index of nums NOT EQUALS element at position element at position index of nums MINUS one of nums holds  
                SET temporary_value TO element at position index of nums  
                SET element at position index of nums TO element at position temporary_value MINUS one of nums  
                SET element at position temporary_value MINUS one of nums TO temporary_value  
            END WHILE  
        END FOR  
        SET result_list TO empty list  
        FOR each pair of element_index AND element_value IN enumeration of nums  
            IF element_value NOT EQUALS element_index PLUS one  
                APPEND element_value TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
