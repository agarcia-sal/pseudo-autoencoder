CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        
        SET dp TO a two dimensional list of zero initialized elements with dimensions m PLUS one BY n PLUS one  
        
        FOR j FROM one TO n  
            SET element at position j of element at position zero of dp TO element at position j MINUS one of element at position zero of dp PLUS ASCII VALUE OF the character at position j MINUS one of s2  
        END FOR  
        
        FOR i FROM one TO m  
            SET element at position zero of element at position i of dp TO element at position zero of element at position i MINUS one of dp PLUS ASCII VALUE OF the character at position i MINUS one of s1  
        END FOR  
        
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one of s1 EQUALS character at position j MINUS one of s2  
                    SET element at position j of element at position i of dp TO element at position j MINUS one of element at position i MINUS one of dp  
                ELSE  
                    SET value_one TO element at position j of element at position i MINUS one of dp PLUS ASCII VALUE OF the character at position i MINUS one of s1  
                    SET value_two TO element at position j MINUS one of element at position i of dp PLUS ASCII VALUE OF the character at position j MINUS one of s2  
                    IF value_one LESS THAN value_two  
                        SET element at position j of element at position i of dp TO value_one  
                    ELSE  
                        SET element at position j of element at position i of dp TO value_two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN element at position n of element at position m of dp  
    END FUNCTION  
END CLASS
