CLASS Solution  
    FUNCTION findMinStep(board, hand)  
        FUNCTION clean(board)  
            SET stack TO empty list  
            FOR each ball IN board  
                IF stack IS NOT empty AND the first element of the last item in stack IS NOT EQUALS ball AND the second element of the last item in stack IS GREATER THAN OR EQUAL TO three  
                    REMOVE the last element FROM stack  
                END IF  
                IF stack IS empty OR the first element of the last item in stack IS NOT EQUALS ball  
                    APPEND list containing ball and one TO stack  
                ELSE  
                    INCREMENT the second element of the last item in stack BY one  
                END IF  
            END FOR  
            IF stack IS NOT empty AND the second element of the last item in stack IS GREATER THAN OR EQUAL TO three  
                REMOVE the last element FROM stack  
            END IF  
            RETURN the concatenation of the first element MULTIPLIED BY the second element FOR each item IN stack  
        END FUNCTION  
        
        FUNCTION dfs(board, hand_counter)  
            IF board IS empty  
                RETURN zero  
            END IF  
            SET min_steps TO positive infinity  
            SET i TO zero  
            WHILE i IS LESS THAN the LENGTH OF board  
                SET j TO i PLUS one  
                WHILE j IS LESS THAN the LENGTH OF board AND the element at position i of board IS EQUALS the element at position j of board  
                    INCREMENT j BY one  
                END WHILE  
                SET increment TO three MINUS the difference BETWEEN j AND i  
                IF the value at key the element at position i of board IN hand_counter IS GREATER THAN OR EQUAL TO increment  
                    SET new_hand_counter TO a copy of hand_counter  
                    DECREMENT the value at key the element at position i of board IN new_hand_counter BY increment  
                    SET new_board TO the result of calling clean WITH the concatenation of the substring of board from zero TO i AND the substring of board from j TO end  
                    SET steps TO the result of calling dfs WITH new_board AND new_hand_counter  
                    IF steps IS GREATER THAN OR EQUAL TO zero  
                        SET min_steps TO the smaller value BETWEEN min_steps AND the sum of steps AND increment  
                    END IF  
                END IF  
                SET i TO j  
            END WHILE  
            IF min_steps IS EQUALS positive infinity  
                RETURN negative one  
            ELSE  
                RETURN min_steps  
            END IF  
        END FUNCTION  
        
        SET hand_counter TO the collection counter constructed from hand  
        RETURN the result of calling dfs WITH board AND hand_counter  
    END FUNCTION  
END CLASS
