CLASS Solution  
    FUNCTION distinctSubseqII WITH PARAMETER s AS string  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        INITIALIZE dp AS list OF zero WITH LENGTH n PLUS one  
        SET element at position zero of dp TO one  
        INITIALIZE last AS empty dictionary  
        
        FOR index i FROM one TO n  
            SET char TO element at position i MINUS one of s  
            SET element at position i of dp TO TWO MULTIPLIED BY element at position i MINUS one of dp MODULO MOD  
            
            IF char IS A KEY IN last  
                SET element at position i of dp TO element at position i of dp MINUS element at position value at key char in last MINUS one of dp PLUS MOD MODULO MOD  
            END IF  
            
            SET value at key char in last TO i  
        END FOR  
        
        RETURN element at position n of dp MINUS one MODULO MOD  
    END FUNCTION  
END CLASS
