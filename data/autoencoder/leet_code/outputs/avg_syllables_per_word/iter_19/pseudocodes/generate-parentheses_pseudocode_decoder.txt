CLASS Solution  
    FUNCTION generateParenthesis with parameter n  
        FUNCTION backtrack with parameters s default to empty string left default to zero right default to zero  
            IF the LENGTH OF s EQUALS two MULTIPLIED BY n THEN  
                APPEND s TO result  
                RETURN  
            END IF  
            IF left LESS THAN n THEN  
                CALL backtrack with s CONCATENATED WITH left parenthesis left INCREMENTED BY one right  
            END IF  
            IF right LESS THAN left THEN  
                CALL backtrack with s CONCATENATED WITH right parenthesis left right INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack with default parameters  
        RETURN result  
    END FUNCTION  
END CLASS
