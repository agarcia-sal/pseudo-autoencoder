CLASS Solution  
    FUNCTION permuteUnique(nums)  
        FUNCTION backtrack(path, used, res)  
            IF the LENGTH OF path EQUALS the LENGTH OF nums  
                APPEND a COPY OF path TO res  
                RETURN  
            END IF  
            FOR position FROM zero TO the LENGTH OF nums MINUS one  
                IF element at position of used IS true OR position GREATER THAN zero AND element at position of nums EQUALS element at position MINUS one of nums AND element at position MINUS one of used IS false  
                    CONTINUE TO next iteration  
                END IF  
                SET element at position of used TO true  
                APPEND element at position of nums TO path  
                CALL backtrack WITH path, used, res  
                REMOVE last element FROM path  
                SET element at position of used TO false  
            END FOR  
        END FUNCTION  
        
        SORT nums IN ascending order  
        SET res TO empty list  
        SET used TO a list of false values WITH the LENGTH OF nums  
        CALL backtrack WITH empty list, used, res  
        RETURN res  
    END FUNCTION  
END CLASS
