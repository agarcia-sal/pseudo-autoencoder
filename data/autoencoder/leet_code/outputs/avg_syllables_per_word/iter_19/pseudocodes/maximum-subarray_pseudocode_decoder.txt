CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO element at position one of nums  
        SET current_sum TO element at position one of nums  
        FOR each num IN elements from position two TO the end of nums  
            SET sum_if_added TO current_sum PLUS num  
            IF sum_if_added GREATER THAN num  
                SET current_sum TO sum_if_added  
            ELSE  
                SET current_sum TO num  
            END IF  
            IF current_sum GREATER THAN max_sum  
                SET max_sum TO current_sum  
            END IF  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
