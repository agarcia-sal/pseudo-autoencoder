CLASS Solution  
    FUNCTION circularArrayLoop(nums)  
        FUNCTION sameDirection(val1, val2)  
            IF val1 GREATER THAN zero AND val2 GREATER THAN zero  
                RETURN true  
            ELSE IF val1 LESS THAN zero AND val2 LESS THAN zero  
                RETURN true  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET n TO the LENGTH OF nums  

        FOR index i FROM zero TO n MINUS one  
            SET slow TO i  
            SET fast TO i  

            IF the element at position i of nums EQUALS zero  
                CONTINUE to the next iteration  
            END IF  

            WHILE true  
                SET slow TO the REMAINDER OF (slow PLUS the element at position slow of nums) DIVIDED BY n  
                IF slow LESS THAN zero  
                    INCREMENT slow BY n  
                END IF  

                SET fast TO the REMAINDER OF (fast PLUS the element at position fast of nums) DIVIDED BY n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  

                SET fast TO the REMAINDER OF (fast PLUS the element at position fast of nums) DIVIDED BY n  
                IF fast LESS THAN zero  
                    INCREMENT fast BY n  
                END IF  

                IF the result of sameDirection given the element at position slow of nums AND the element at position i of nums IS false OR the result of sameDirection given the element at position fast of nums AND the element at position i of nums IS false  
                    BREAK the loop  
                END IF  

                IF slow EQUALS fast  
                    IF slow EQUALS the REMAINDER OF slow PLUS the element at position slow of nums DIVIDED BY n  
                        BREAK the loop  
                    END IF  
                    RETURN true  
                END IF  
            END WHILE  
        END FOR  

        RETURN false  
    END FUNCTION  
END CLASS
