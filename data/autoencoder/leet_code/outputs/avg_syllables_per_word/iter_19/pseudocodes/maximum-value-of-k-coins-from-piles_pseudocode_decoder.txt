CLASS Solution  
    FUNCTION maxValueOfCoins(piles, k)  
        SET n TO the LENGTH OF piles  
        INITIALIZE dp AS a two dimensional collection of zeros with n plus one rows and k plus one columns  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO the element at position i minus one of piles  
                SET current_value TO zero  
                FOR l FROM zero TO the MINIMUM OF j AND the LENGTH OF current_pile  
                    IF l IS GREATER THAN zero  
                        INCREMENT current_value BY the element at position l minus one of current_pile  
                    END IF  
                    SET dp at row i column j TO the MAXIMUM OF dp at row i column j AND the sum of dp at row i minus one column j minus l PLUS current_value  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp at row n column k  
    END FUNCTION  
END CLASS
