CLASS Solution  
    FUNCTION maximumCost(n, highways, k)  
        ASSIGN pointer TO create_default_dictionary_of_lists NAMED graph  
        FOR each city1, city2, toll IN highways  
            APPEND tuple of city2 and toll TO element at position city1 of graph  
            APPEND tuple of city1 and toll TO element at position city2 of graph  
        END FOR  
        
        SET max_cost TO negative one  
        
        FUNCTION dfs(city, visited, cost, highways_used)  
            IF highways_used EQUALS k  
                SET max_cost TO maximum of max_cost AND cost  
                RETURN  
            END IF  
            
            FOR each neighbor, toll IN element at position city of graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    CALL dfs WITH neighbor, visited, cost PLUS toll, highways_used PLUS one  
                    REMOVE neighbor FROM visited  
                END IF  
            END FOR  
        END FUNCTION  
        
        FOR start_city FROM zero TO n MINUS one  
            ASSIGN visited TO new set containing start_city  
            CALL dfs WITH start_city, visited, zero, zero  
        END FOR  
        
        RETURN max_cost  
    END FUNCTION  
END CLASS
