CLASS Solution  
    FUNCTION validArrangement(pairs)  
        SET graph TO a dictionary with default value as an empty list  
        SET out_degree TO a counter dictionary  
        SET in_degree TO a counter dictionary  
  
        FOR each pair of elements u and v IN pairs  
            APPEND element v TO list at key u of graph  
            INCREMENT value at key u of out_degree BY one  
            INCREMENT value at key v of in_degree BY one  
        END FOR  
  
        SET start_node TO none  
        FOR each node IN keys of graph  
            IF value at key node of out_degree EQUALS value at key node of in_degree PLUS one  
                SET start_node TO node  
                BREAK the loop  
            END IF  
        END FOR  
  
        IF start_node IS none  
            SET start_node TO the first key of graph obtained by iteration  
        END IF  
  
        SET path TO an empty list  
        SET stack TO a list containing start_node  
  
        WHILE the LENGTH OF stack IS greater than zero  
            SET u TO the last element of stack  
            IF the list at key u of graph IS NOT empty  
                SET v TO the last element of the list at key u of graph, removing that element from the list  
                APPEND element v TO stack  
            ELSE  
                APPEND the last element of stack removed from stack TO path  
            END IF  
        END WHILE  
  
        REVERSE the order of elements in path  
  
        SET result TO an empty list  
        FOR index FROM zero TO the LENGTH OF path MINUS one exclusive  
            APPEND a list containing element at position index of path and element at position index PLUS one of path TO result  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
