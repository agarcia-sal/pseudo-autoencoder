CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO an empty list  
        SET a TO an empty string  
        SET d TO an empty string  
        FOR each character IN s  
            IF character IS a digit  
                SET d TO d concatenated with character  
            ELSE IF character IS an alphabetic letter  
                SET a TO a concatenated with character  
            ELSE IF character IS an opening bracket  
                APPEND the tuple consisting of a and the integer conversion of d TO stack  
                SET a TO an empty string  
                SET d TO an empty string  
            ELSE IF character IS a closing bracket  
                SET p and n TO the tuple obtained by removing the last element FROM stack  
                SET a TO p concatenated with a repeated n times  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
