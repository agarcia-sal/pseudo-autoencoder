CLASS Solution  
    FUNCTION maxSlidingWindow(nums LIST OF INTEGER, k INTEGER) RETURNS LIST OF INTEGER  
        IF the list nums is empty OR k EQUALS zero THEN  
            RETURN an empty list  
        END IF  
        
        SET deq TO an empty double-ended queue  
        SET result TO an empty list  
        
        FOR index i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq is not empty AND the element at position zero of deq is LESS THAN i MINUS k PLUS one THEN  
                REMOVE the element at position zero from deq  
            END IF  
            
            WHILE deq is not empty AND the element at position element at the last position of deq of nums is LESS THAN the element at position i of nums DO  
                REMOVE the element at the last position from deq  
            END WHILE  
            
            APPEND i TO deq  
            
            IF i IS GREATER THAN OR EQUAL TO k MINUS one THEN  
                APPEND the element at position element at position zero of deq of nums TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
