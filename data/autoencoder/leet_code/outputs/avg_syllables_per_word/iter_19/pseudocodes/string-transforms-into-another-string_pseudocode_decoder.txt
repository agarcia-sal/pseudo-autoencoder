CLASS Solution  
    FUNCTION canConvert(str1, str2)  
        IF str1 EQUALS str2  
            RETURN True  
        END IF  
        SET conversion_map TO an empty mapping from characters to characters  
        FOR each paired element char1 and char2 IN the pairwise association of elements of str1 and str2  
            IF char1 IS a key in conversion_map  
                IF the value in conversion_map at key char1 DOES NOT EQUAL char2  
                    RETURN False  
                END IF  
            ELSE  
                ASSIGN in conversion_map at key char1 the value char2  
            END IF  
        END FOR  
        IF the number of distinct elements in str2 EQUALS twenty six  
            RETURN False  
        END IF  
        RETURN True  
    END FUNCTION  
END CLASS
