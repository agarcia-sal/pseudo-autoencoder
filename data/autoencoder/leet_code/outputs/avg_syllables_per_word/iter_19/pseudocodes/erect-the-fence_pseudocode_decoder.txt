CLASS Solution  
    FUNCTION outerTrees(trees)  
        FUNCTION cross(i, j, k)  
            SET a TO element at position i of trees  
            SET b TO element at position j of trees  
            SET c TO element at position k of trees  
            RETURN the result of the expression where the difference between the first coordinate of b and the first coordinate of a MULTIPLIED BY the difference between the second coordinate of c and the second coordinate of b MINUS the difference between the second coordinate of b and the second coordinate of a MULTIPLIED BY the difference between the first coordinate of c and the first coordinate of b  
        END FUNCTION  

        SET n TO the length of trees  
        IF n LESS THAN four  
            RETURN trees  
        END IF  

        SORT trees in ascending order  
        SET vis TO a list of Boolean False of size n  
        SET stk TO a list containing zero as the only element  

        FOR i FROM one TO n MINUS one  
            WHILE the length of stk GREATER THAN one AND the result of cross of element at position second last of stk, element at position last of stk, and i LESS THAN zero holds  
                SET vis of element at position last popped from stk TO False  
            END WHILE  
            SET vis at position i TO True  
            APPEND i TO stk  
        END FOR  

        SET m TO the length of stk  

        FOR i FROM n MINUS two TO zero DECREMENT BY one  
            IF vis at position i EQUALS True  
                CONTINUE to next iteration  
            END IF  
            WHILE the length of stk GREATER THAN m AND the result of cross of element at position second last of stk, element at position last of stk, and i LESS THAN zero holds  
                POP last element from stk  
            END WHILE  
            APPEND i TO stk  
        END FOR  

        POP last element from stk  
        RETURN a list of elements at positions in stk of trees  
    END FUNCTION  
END CLASS
