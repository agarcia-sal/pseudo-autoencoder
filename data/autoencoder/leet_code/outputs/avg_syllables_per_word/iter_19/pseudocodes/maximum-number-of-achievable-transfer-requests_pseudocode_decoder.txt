CLASS Solution  
    FUNCTION maximumRequests WITH PARAMETERS n AND requests  
        SET max_requests TO zero  
        SET num_requests TO the LENGTH OF requests  
        FOR mask FROM zero TO two RAISED TO THE POWER OF num_requests MINUS one  
            SET balance TO a new list OF LENGTH n WITH all elements SET TO zero  
            SET count TO zero  
            FOR i FROM zero TO num_requests MINUS one  
                IF the bitwise AND OF mask AND two RAISED TO THE POWER OF i IS GREATER THAN zero  
                    SET from_building TO the element at position zero of the element at position i of requests  
                    SET to_building TO the element at position one of the element at position i of requests  
                    DECREMENT the element at position from_building of balance BY one  
                    INCREMENT the element at position to_building of balance BY one  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            IF for each element b in balance the element b EQUALS zero  
                IF max_requests IS LESS THAN count  
                    SET max_requests TO count  
                END IF  
            END IF  
        END FOR  
        RETURN max_requests  
    END FUNCTION  
END CLASS
