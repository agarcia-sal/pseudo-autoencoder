CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of all characters in time after removing colon character  
        SET hour AND minute TO integer values obtained by splitting time into hour component and minute component by colon character  
        
        FUNCTION generate_times(digits)  
            SET times TO empty list  
            FOR each character h1 IN digits  
                FOR each character h2 IN digits  
                    FOR each character m1 IN digits  
                        FOR each character m2 IN digits  
                            SET combined_hour_string TO the concatenation of character h1 followed by character h2  
                            SET combined_minute_string TO the concatenation of character m1 followed by character m2  
                            SET hour TO integer value of combined_hour_string  
                            SET minute TO integer value of combined_minute_string  
                            IF hour LESS THAN twenty four AND minute LESS THAN sixty  
                                APPEND the tuple of hour and minute TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN the list times sorted in ascending order  
        END FUNCTION  
        
        SET all_times TO the result of generate_times with digits parameter  
        SET current_time_index TO the position of the tuple of hour and minute inside all_times  
        
        IF current_time_index PLUS one LESS THAN the length of all_times  
            SET next_hour AND next_minute TO the tuple element at position current_time_index PLUS one of all_times  
        ELSE  
            SET next_hour AND next_minute TO the tuple element at position zero of all_times  
        END IF  
        
        RETURN the string representation of next_hour with leading zero if needed followed by colon character followed by the string representation of next_minute with leading zero if needed  
    END FUNCTION  
END CLASS
