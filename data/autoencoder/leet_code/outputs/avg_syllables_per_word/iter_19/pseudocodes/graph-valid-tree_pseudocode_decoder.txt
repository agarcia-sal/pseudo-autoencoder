CLASS Solution  
    FUNCTION validTree WITH PARAMETERS n AND edges  
        IF the LENGTH OF edges NOT EQUALS n MINUS one THEN  
            RETURN false  
        END IF  
        SET adj_list TO an empty mapping from integer TO list of integers with keys FROM zero TO n MINUS one  
        FOR each pair a AND b IN edges  
            APPEND b TO the list AT key a OF adj_list  
            APPEND a TO the list AT key b OF adj_list  
        END FOR  
        SET visited TO an empty set  
        FUNCTION dfs WITH PARAMETER node  
            IF node IS IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN the list AT key node OF adj_list  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        CALL dfs WITH zero  
        RETURN the LENGTH OF visited EQUALS n  
    END FUNCTION  
END CLASS
