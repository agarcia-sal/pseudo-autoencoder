CLASS Solution  
    FUNCTION minArea(image, x, y)  
        IF image IS EMPTY OR element at position zero of image IS EMPTY  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF image  
        SET n TO the LENGTH OF element at position zero of image  
        
        FUNCTION hasBlackPixelInRow(row)  
            FOR each pixel IN element at position row of image  
                IF pixel EQUALS one as a character  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION hasBlackPixelInColumn(col)  
            FOR row FROM zero TO m MINUS one  
                IF element at position col of element at position row of image EQUALS one as a character  
                    RETURN True  
                END IF  
            END FOR  
            RETURN False  
        END FUNCTION  
        
        FUNCTION findTop()  
            SET low TO zero  
            SET high TO x  
            WHILE low LESS THAN high  
                SET mid TO low PLUS high DIVIDED BY two WITHOUT REMAINDER  
                IF hasBlackPixelInRow(mid) IS True  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findBottom()  
            SET low TO x  
            SET high TO m MINUS one  
            WHILE low LESS THAN high  
                SET mid TO low PLUS high PLUS one DIVIDED BY two WITHOUT REMAINDER  
                IF hasBlackPixelInRow(mid) IS True  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findLeft()  
            SET low TO zero  
            SET high TO y  
            WHILE low LESS THAN high  
                SET mid TO low PLUS high DIVIDED BY two WITHOUT REMAINDER  
                IF hasBlackPixelInColumn(mid) IS True  
                    SET high TO mid  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        FUNCTION findRight()  
            SET low TO y  
            SET high TO n MINUS one  
            WHILE low LESS THAN high  
                SET mid TO low PLUS high PLUS one DIVIDED BY two WITHOUT REMAINDER  
                IF hasBlackPixelInColumn(mid) IS True  
                    SET low TO mid  
                ELSE  
                    SET high TO mid MINUS one  
                END IF  
            END WHILE  
            RETURN low  
        END FUNCTION  
        
        SET top TO findTop()  
        SET bottom TO findBottom()  
        SET left TO findLeft()  
        SET right TO findRight()  
        
        RETURN bottom MINUS top PLUS one MULTIPLIED BY right MINUS left PLUS one  
    END FUNCTION  
END CLASS
