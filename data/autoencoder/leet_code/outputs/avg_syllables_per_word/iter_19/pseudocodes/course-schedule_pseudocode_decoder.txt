CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        SET graph TO a mapping from keys to lists with default empty lists  
        SET indegree TO a list of zero values with length equal to numCourses  
        
        FOR each pair of course and prereq IN prerequisites  
            APPEND course TO the list mapped by prereq in graph  
            INCREMENT the value at position course in indegree BY one  
        END FOR  
        
        SET queue TO a double ended queue initialized with each course in the range from zero TO one LESS THAN numCourses WHERE the value at position course in indegree EQUALS zero  
        
        WHILE the queue is not empty  
            REMOVE and ASSIGN the first element of the queue TO current_course  
            DECREMENT numCourses BY one  
            
            FOR each next_course IN the list mapped by current_course in graph  
                DECREMENT the value at position next_course in indegree BY one  
                IF the value at position next_course in indegree EQUALS zero  
                    APPEND next_course TO the queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN whether numCourses EQUALS zero  
    END FUNCTION  
END CLASS
