CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self val WITH DEFAULT zero next WITH DEFAULT none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETERS values  
    IF the list values IS EMPTY  
        RETURN none  
    END IF  
    SET head TO A new ListNode INITIALIZED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the list of elements from position one TO the end of values  
        SET node TO A new ListNode INITIALIZED WITH val  
        SET the element at position next of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 p2  
    IF p1 IS none AND p2 IS none  
        RETURN True  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN False  
    END IF  
    RETURN THE TEST that the element at position val of p1 EQUALS the element at position val of p2 AND THE RESULT OF is_same_list WITH PARAMETERS the element at position next of p1 AND the element at position next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList WITH PARAMETERS self head  
        IF head IS none OR the element at position next of head IS none  
            RETURN head  
        END IF  
        SET dummy TO A new ListNode INITIALIZED WITH the integer negative five thousand and one  
        SET the element at position next of dummy TO head  
        SET current TO the element at position next of head  
        SET the element at position next of head TO none  
        WHILE current IS NOT none  
            SET next_node TO the element at position next of current  
            SET prev TO dummy  
            WHILE the element at position next of prev IS NOT none AND the element at position val of the element at position next of prev IS LESS THAN the element at position val of current  
                SET prev TO the element at position next of prev  
            END WHILE  
            SET the element at position next of current TO the element at position next of prev  
            SET the element at position next of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the element at position next of dummy  
    END FUNCTION  
END CLASS
