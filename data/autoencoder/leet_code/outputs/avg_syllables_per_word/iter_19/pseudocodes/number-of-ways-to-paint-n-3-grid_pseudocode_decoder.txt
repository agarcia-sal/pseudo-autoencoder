CLASS Solution  
    FUNCTION numOfWays WITH PARAMETERS n  
        SET MOD TO one billion PLUS seven  
        SET a TO six  
        SET b TO six  
        FOR each index FROM two TO n  
            SET new_a TO the quantity of three MULTIPLIED BY a PLUS two MULTIPLIED BY b  
            SET new_b TO the quantity of two MULTIPLIED BY a PLUS two MULTIPLIED BY b  
            SET a TO new_a MODULO MOD  
            SET b TO new_b MODULO MOD  
        END FOR  
        SET result TO the sum of a AND b MODULO MOD  
        RETURN result  
    END FUNCTION  
END CLASS
