CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i IS GREATER THAN OR EQUAL TO the LENGTH OF t  
                APPEND the JOINED ELEMENTS OF t TO ans  
                RETURN  
            END IF  
            CALL dfs WITH i PLUS one  
            IF the element at position i of t IS AN ALPHABETIC CHARACTER  
                SET the element at position i of t TO the CHARACTER corresponding TO THE ASCII CODE OF the element at position i of t EXCLUSIVE OR WITH thirty two  
                CALL dfs WITH i PLUS one  
            END IF  
        END FUNCTION  
        
        SET t TO the LIST OF CHARACTERS OF s  
        SET ans TO an EMPTY LIST  
        CALL dfs WITH zero  
        RETURN ans  
    END FUNCTION  
END CLASS
