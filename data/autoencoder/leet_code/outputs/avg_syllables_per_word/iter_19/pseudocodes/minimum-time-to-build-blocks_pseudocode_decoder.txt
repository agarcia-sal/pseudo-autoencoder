CLASS Solution  
    FUNCTION minBuildTime(blocks AS List OF integers, split AS integer) RETURNS integer  
        CALL helper FUNCTION heapify TO TRANSFORM blocks INTO a minimum heap  
        WHILE the LENGTH OF blocks IS GREATER THAN one  
            SET first TO the smallest element REMOVED FROM blocks USING heap's pop operation  
            SET second TO the next smallest element REMOVED FROM blocks USING heap's pop operation  
            SET combined_time TO split PLUS second  
            INSERT combined_time INTO blocks USING heap's push operation  
        END WHILE  
        RETURN the element at position zero of blocks  
    END FUNCTION  
END CLASS
