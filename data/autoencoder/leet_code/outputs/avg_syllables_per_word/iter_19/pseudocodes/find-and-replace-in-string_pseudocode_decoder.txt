CLASS Solution  
    FUNCTION findReplaceString WITH PARAMETERS s string, indices list of integer, sources list of string, targets list of string  
        SET replacements TO the sorted list of tuples formed by pairing each element of indices with the corresponding element of sources and the corresponding element of targets arranged in ascending order by the first element of each tuple  
        SET result TO an empty list  
        SET prev_end TO zero  
        FOR each tuple composed of index, source, and target IN replacements  
            APPEND the substring from position prev_end TO position index of string s TO result  
            IF the substring from position index TO position index PLUS the length of source of string s EQUALS source  
                APPEND target TO result  
            ELSE  
                APPEND the substring from position index TO position index PLUS the length of source of string s TO result  
            END IF  
            SET prev_end TO index PLUS the length of source  
        END FOR  
        APPEND the substring from position prev_end TO the end of string s TO result  
        RETURN the concatenation of all elements in result as a single string  
    END FUNCTION  
END CLASS
