CLASS Solution  
    FUNCTION isMatch(s, p)  
        INITIALIZE dp AS a two dimensional list of size LENGTH OF s plus one BY LENGTH OF p plus one FILLED WITH False  
        SET the element at position zero zero of dp TO True  
        FOR each index j FROM one TO LENGTH OF p  
            IF the character at position j minus one of p EQUALS the character representing asterisk  
                SET the element at position zero j of dp TO the element at position zero j minus two of dp  
            END IF  
        END FOR  
        FOR each index i FROM one TO LENGTH OF s  
            FOR each index j FROM one TO LENGTH OF p  
                IF the character at position j minus one of p EQUALS the character at position i minus one of s OR the character at position j minus one of p EQUALS the single character dot  
                    SET the element at position i j of dp TO the element at position i minus one j minus one of dp  
                ELSE IF the character at position j minus one of p EQUALS the character representing asterisk  
                    SET the element at position i j of dp TO the element at position i j minus two of dp  
                    IF the character at position j minus two of p EQUALS the character at position i minus one of s OR the character at position j minus two of p EQUALS the single character dot  
                        SET the element at position i j of dp TO the element at position i j of dp OR the element at position i minus one j of dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position LENGTH OF s LENGTH OF p of dp  
    END FUNCTION  
END CLASS
