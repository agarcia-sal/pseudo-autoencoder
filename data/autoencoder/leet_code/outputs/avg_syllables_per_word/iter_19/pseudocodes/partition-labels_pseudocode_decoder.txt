CLASS Solution  
    FUNCTION partitionLabels(s)  
        SET last_occurrence TO AN EMPTY DICTIONARY  
        FOR each index AND character IN s  
            SET last_occurrence AT character TO index  
        END FOR  
        
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO AN EMPTY LIST  
        
        FOR each index AND character IN s  
            IF last_occurrence AT character GREATER THAN end  
                SET end TO last_occurrence AT character  
            END IF  
            
            IF index EQUALS end  
                SET partition_size TO end MINUS start PLUS one  
                APPEND partition_size TO partition_sizes  
                SET start TO index PLUS one  
            END IF  
        END FOR  
        
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
