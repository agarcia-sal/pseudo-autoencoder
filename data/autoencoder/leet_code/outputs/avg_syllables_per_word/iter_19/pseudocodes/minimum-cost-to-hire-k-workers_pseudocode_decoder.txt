CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO sorted list of pairs consisting of quotient of element of wage divided by corresponding element of quality and corresponding element of quality for each pair of elements from wage and quality taken in parallel  
        SET max_heap TO empty list  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each pair ratio and q IN workers  
            PUSH the negative of q INTO max_heap  
            INCREMENT total_quality BY q  
            IF size of max_heap GREATER THAN k  
                INCREMENT total_quality BY the element obtained by popping the smallest element from max_heap  
            END IF  
            IF size of max_heap EQUALS k  
                SET min_cost TO the smaller value between min_cost and total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
