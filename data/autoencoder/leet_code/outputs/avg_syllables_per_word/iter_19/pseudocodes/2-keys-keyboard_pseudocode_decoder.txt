CLASS Solution  
    FUNCTION minSteps(n)  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET operations TO zero  
        SET factor TO two  
        WHILE n GREATER THAN one  
            WHILE the remainder of n DIVIDED BY factor EQUALS zero  
                INCREMENT operations BY factor  
                SET n TO n DIVIDED BY factor ignoring the remainder  
            END WHILE  
            INCREMENT factor BY one  
        END WHILE  
        RETURN operations  
    END FUNCTION  
END CLASS
