CLASS Solution  
    FUNCTION trapRainWater(heightMap)  
        IF heightMap IS empty OR element at position zero of heightMap IS empty  
            RETURN zero  
        END IF  
        
        SET m TO the LENGTH OF heightMap  
        SET n TO the LENGTH OF element at position zero of heightMap  
        SET visited TO a list of m lists each containing n times the value False  
        SET min_heap TO an empty list  
        
        FOR index i FROM zero TO m MINUS one  
            PUSH onto min_heap a tuple containing element at position i position zero of heightMap and i and zero  
            PUSH onto min_heap a tuple containing element at position i position n MINUS one of heightMap and i and n MINUS one  
            SET element at position i position zero of visited TO True  
            SET element at position i position n MINUS one of visited TO True  
        END FOR  
        
        FOR index j FROM zero TO n MINUS one  
            PUSH onto min_heap a tuple containing element at position zero position j of heightMap and zero and j  
            PUSH onto min_heap a tuple containing element at position m MINUS one position j of heightMap and m MINUS one and j  
            SET element at position zero position j of visited TO True  
            SET element at position m MINUS one position j of visited TO True  
        END FOR  
        
        SET directions TO the list of tuples consisting of minus one and zero and one and zero and zero and minus one and zero and one  
        SET water_trapped TO zero  
        
        WHILE min_heap IS NOT empty  
            REMOVE and GET the smallest element from min_heap ASSIGN TO height x y  
            
            FOR each tuple dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND element at position nx position ny of visited IS False  
                    SET element at position nx position ny of visited TO True  
                    INCREMENT water_trapped BY the maximum of zero and height MINUS element at position nx position ny of heightMap  
                    PUSH onto min_heap a tuple containing the maximum of height and element at position nx position ny of heightMap and nx and ny  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN water_trapped  
    END FUNCTION  
END CLASS
