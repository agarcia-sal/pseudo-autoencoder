CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO list of characters representing n converted to string  
        SET length TO the LENGTH OF digits  
        SET marker TO length  
        FOR index FROM length MINUS one DOWN TO one IN STEPS OF MINUS one  
            IF character at position index of digits IS LESS THAN character at position index MINUS one of digits  
                SET character at position index MINUS one of digits TO string representation of integer conversion of character at position index MINUS one of digits MINUS one  
                SET marker TO index  
            END IF  
        END FOR  
        FOR index FROM marker TO length MINUS one  
            SET character at position index of digits TO character representing digit nine  
        END FOR  
        SET result_string TO concatenation of all characters in digits  
        SET result_integer TO integer conversion of result_string  
        RETURN result_integer  
    END FUNCTION  
END CLASS
