CLASS Solution  
    FUNCTION canJump(nums)  
        SET farthest TO zero  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            IF index IS GREATER THAN farthest  
                RETURN False  
            END IF  
            SET farthest TO the GREATER VALUE BETWEEN farthest AND the SUM OF index PLUS element at position index of nums  
        END FOR  
        IF farthest IS GREATER THAN OR EQUAL TO the LENGTH OF nums MINUS one  
            RETURN True  
        ELSE  
            RETURN False  
        END IF  
    END FUNCTION  
END CLASS
