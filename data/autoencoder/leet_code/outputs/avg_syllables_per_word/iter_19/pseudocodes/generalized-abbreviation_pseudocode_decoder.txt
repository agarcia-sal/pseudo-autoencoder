CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the CONVERSION OF count TO STRING TO path  
                END IF  
                APPEND the CONCATENATION OF all ELEMENTS OF path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH arguments index INCREMENTED BY one, path COMBINED WITH the CONVERSION OF count TO STRING AND the element at position index of word, zero  
                ELSE  
                    CALL backtrack WITH arguments index INCREMENTED BY one, path COMBINED WITH the element at position index of word, zero  
                END IF  
                CALL backtrack WITH arguments index INCREMENTED BY one, path, count INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty list  
        CALL backtrack WITH arguments zero, an empty list, zero  
        RETURN result  
    END FUNCTION  
END CLASS
