CLASS Solution  
    FUNCTION shortestPath grid COMMA k  
        IF grid IS EMPTY OR element at position zero of grid IS EMPTY  
            RETURN negative one  
        END IF  
        
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero of grid  
        
        IF k IS GREATER THAN OR EQUAL TO m PLUS n MINUS two  
            RETURN m PLUS n MINUS two  
        END IF  
        
        SET directions TO the LIST OF four PAIRS representing zero AND one PLUS one AND zero PLUS zero AND negative one PLUS negative one AND zero  
        
        INITIALIZE queue AS an empty double ended queue WITH a TUPLE containing zero AND zero AND k AND zero  
        
        INITIALIZE visited AS a set CONTAINING a TUPLE of zero AND zero AND k  
        
        WHILE the queue IS NOT EMPTY  
            REMOVE AND OBTAIN first TUPLE FROM queue INTO x AND y AND remaining_k AND steps  
            
            IF x EQUALS m MINUS one AND y EQUALS n MINUS one  
                RETURN steps  
            END IF  
            
            FOR each PAIR dx AND dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n  
                    SET new_k TO remaining_k MINUS element at position nx of grid AT position ny  
                    IF new_k IS GREATER THAN OR EQUAL TO zero AND the TUPLE nx AND ny AND new_k IS NOT IN visited  
                        ADD the TUPLE nx AND ny AND new_k TO visited  
                        APPEND the TUPLE nx AND ny AND new_k AND steps PLUS one TO queue  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
