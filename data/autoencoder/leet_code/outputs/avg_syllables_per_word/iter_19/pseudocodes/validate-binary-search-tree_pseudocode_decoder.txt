CLASS TreeNode  
    FUNCTION __init__ with parameters val equals zero left equals none right equals none  
        SET the attribute val OF self TO the parameter val  
        SET the attribute left OF self TO the parameter left  
        SET the attribute right OF self TO the parameter right  
    END FUNCTION  
END CLASS

FUNCTION tree_node with parameter values which is a list  
    IF the parameter values is empty  
        RETURN none  
    END IF  
    SET root TO a new instance of TreeNode initialized with element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater than zero  
        SET node TO the element popped from the left of queue  
        IF i is less than the LENGTH OF values AND the element at position i of values is not none  
            SET the attribute left OF node TO a new instance of TreeNode initialized with element at position i of values  
            APPEND the attribute left OF node TO queue  
        END IF  
        INCREMENT i BY one  
        IF i is less than the LENGTH OF values AND the element at position i of values is not none  
            SET the attribute right OF node TO a new instance of TreeNode initialized with element at position i of values  
            APPEND the attribute right OF node TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree with parameters p and q  
    IF p is none AND q is none  
        RETURN True  
    ELSE IF p is none OR q is none  
        RETURN False  
    ELSE IF the attribute val OF p is not equal to the attribute val OF q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree applied to the attribute left OF p and the attribute left OF q AND the result of is_same_tree applied to the attribute right OF p and the attribute right OF q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION isValidBST with parameters self and root which is optionally a TreeNode  
        FUNCTION validate with parameters node and low initialized to negative infinity and high initialized to positive infinity  
            IF node is none  
                RETURN True  
            END IF  
            IF the attribute val OF node is less than or equal to low OR the attribute val OF node is greater than or equal to high  
                RETURN False  
            END IF  
            RETURN the result of validate applied to the attribute left OF node low and the attribute val OF node AND the result of validate applied to the attribute right OF node the attribute val OF node and high  
        END FUNCTION  
        RETURN validate applied to root  
    END FUNCTION  
END CLASS
