CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO the concatenation of all elements in sentence separated by a single space followed by a single space character  
        SET total_length TO the number of characters in s  
        SET position TO zero  
        FOR each unused_element FROM one TO rows  
            INCREMENT position BY cols  
            IF the character at position modulo total_length of s EQUALS a space character  
                INCREMENT position BY one  
            ELSE  
                WHILE position is greater than zero AND the character at position minus one modulo total_length of s DOES NOT EQUAL a space character  
                    DECREMENT position BY one  
                END WHILE  
            END IF  
        END FOR  
        SET result TO position INTEGER DIVIDED BY total_length  
        RETURN result  
    END FUNCTION  
END CLASS
