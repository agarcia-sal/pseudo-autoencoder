CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero left EQUALS None right EQUALS None  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node WITH PARAMETER values AS list  
    IF the LENGTH OF values EQUALS zero  
        RETURN None  
    END IF  
    SET root TO a new instance of TreeNode WITH the element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue IS GREATER THAN zero  
        SET node TO the element at the front of queue REMOVED from queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node left TO a new instance of TreeNode WITH the element at position i of values  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None  
            SET node right TO a new instance of TreeNode WITH the element at position i of values  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree WITH PARAMETERS p AND q  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF the val field of p IS NOT EQUALS TO the val field of q  
        RETURN False  
    ELSE  
        RETURN the result of calling is_same_tree WITH PARAMETERS the left field of p AND the left field of q AND the result of calling is_same_tree WITH PARAMETERS the right field of p AND the right field of q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION countUnivalSubtrees WITH PARAMETERS self AND root AS Optional TreeNode  
        FUNCTION helper WITH PARAMETER node  
            IF node IS None  
                RETURN True AND zero  
            END IF  
            SET is_left_unival AND left_count TO the result of calling helper WITH parameter the left field of node  
            SET is_right_unival AND right_count TO the result of calling helper WITH parameter the right field of node  
            IF is_left_unival AND is_right_unival  
                IF node left IS None OR the val field of node left EQUALS the val field of node AND node right IS None OR the val field of node right EQUALS the val field of node  
                    RETURN True AND left_count PLUS right_count PLUS one  
                END IF  
            END IF  
            RETURN False AND left_count PLUS right_count  
        END FUNCTION  
        SET underscore AND count TO the result of calling helper WITH parameter root  
        RETURN count  
    END FUNCTION  
END CLASS
