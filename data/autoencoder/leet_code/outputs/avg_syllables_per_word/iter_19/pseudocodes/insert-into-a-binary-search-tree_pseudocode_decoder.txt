CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self val TO val  
        SET self left TO left  
        SET self right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO TreeNode(element at position zero of values)  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left of queue  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node left TO TreeNode(element at position i of values)  
            APPEND node left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT none  
            SET node right TO TreeNode(element at position i of values)  
            APPEND node right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF the val property of p IS NOT EQUALS the val property of q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree(the left property of p, the left property of q) AND the result of is_same_tree(the right property of p, the right property of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST(self, root, val)  
        FUNCTION dfs(root)  
            IF root IS none  
                RETURN TreeNode(val)  
            END IF  
            IF the val property of root IS LESS THAN val  
                SET the right property of root TO dfs(the right property of root)  
            ELSE  
                SET the left property of root TO dfs(the left property of root)  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN dfs(root)  
    END FUNCTION  
END CLASS
