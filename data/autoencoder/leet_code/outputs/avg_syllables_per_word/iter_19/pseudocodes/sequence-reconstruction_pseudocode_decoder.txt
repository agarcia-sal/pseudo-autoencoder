CLASS Solution  
    FUNCTION sequenceReconstruction(nums, sequences)  
        SET n TO the LENGTH OF nums  
        SET graph TO a new mapping from elements to lists of elements with default empty list  
        SET indegree TO a list of zeros with LENGTH n PLUS one  
        
        FOR each seq IN sequences  
            FOR index FROM zero TO the LENGTH OF seq MINUS one MINUS one  
                SET u TO element at position index of seq  
                SET v TO element at position index PLUS one of seq  
                APPEND v TO the list at graph mapped by u  
                INCREMENT the element at position v of indegree BY one  
            END FOR  
        END FOR  
        
        SET queue TO a new double ended queue initialized with all integers from one TO n inclusive for which the element at position integer of indegree EQUALS zero  
        
        IF the LENGTH OF queue NOT EQUALS one  
            RETURN false  
        END IF  
        
        SET index TO zero  
        WHILE the LENGTH OF queue GREATER THAN zero  
            IF the LENGTH OF queue NOT EQUALS one  
                RETURN false  
            END IF  
            
            REMOVE the element from the left end of queue and SET it TO node  
            IF node NOT EQUALS element at position index of nums  
                RETURN false  
            END IF  
            INCREMENT index BY one  
            
            FOR each neighbor IN the list at graph mapped by node  
                DECREMENT the element at position neighbor of indegree BY one  
                IF the element at position neighbor of indegree EQUALS zero  
                    APPEND neighbor TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN index EQUALS n  
    END FUNCTION  
END CLASS
