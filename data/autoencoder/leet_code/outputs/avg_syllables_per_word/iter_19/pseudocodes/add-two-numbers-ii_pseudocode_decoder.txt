CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val EQUALS zero AND next EQUALS none  
        SET the val ATTRIBUTE OF SELF TO val  
        SET the next ATTRIBUTE OF SELF TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO a new instance of ListNode INITIALIZED WITH the element at position zero of values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the last position  
        SET node TO a new instance of ListNode INITIALIZED WITH val  
        SET the next ATTRIBUTE OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN True  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN False  
    END IF  
    RETURN the logical AND of the condition that the val ATTRIBUTE OF p1 EQUALS the val ATTRIBUTE OF p2 AND the result of calling is_same_list WITH the next ATTRIBUTE OF p1 AND the next ATTRIBUTE OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers WITH PARAMETERS l1 AND l2  
        FUNCTION pushIntoStack WITH PARAMETER node  
            SET stack TO an empty list  
            WHILE node is not none  
                APPEND the val ATTRIBUTE OF node TO stack  
                SET node TO the next ATTRIBUTE OF node  
            END WHILE  
            RETURN stack  
        END FUNCTION  
        
        SET stack1 TO the result of calling pushIntoStack WITH l1  
        SET stack2 TO the result of calling pushIntoStack WITH l2  
        
        SET carry TO zero  
        SET result TO none  
        
        WHILE the LENGTH OF stack1 GREATER THAN zero OR the LENGTH OF stack2 GREATER THAN zero OR carry NOT EQUAL TO zero  
            IF the LENGTH OF stack1 GREATER THAN zero  
                SET val1 TO the element removed from the end of stack1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF the LENGTH OF stack2 GREATER THAN zero  
                SET val2 TO the element removed from the end of stack2  
            ELSE  
                SET val2 TO zero  
            END IF  
            
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO the integer division of total by ten  
            SET digit TO the remainder of total divided by ten  
            
            SET new_node TO a new instance of ListNode INITIALIZED WITH digit  
            SET the next ATTRIBUTE OF new_node TO result  
            SET result TO new_node  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
