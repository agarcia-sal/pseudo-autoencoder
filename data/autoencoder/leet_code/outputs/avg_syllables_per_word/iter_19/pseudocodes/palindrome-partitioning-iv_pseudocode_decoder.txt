CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub EQUALS sub REVERSED  
        END FUNCTION  

        SET n TO the LENGTH OF s  

        SET is_pal TO a two dimensional list OF size n BY n FILLED WITH False  

        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF the element at position i of s EQUALS the element at position j of s AND  
                   the difference between j AND i IS LESS THAN OR EQUAL TO one OR the value at position i PLUS one AND position j MINUS one OF is_pal IS True  
                    SET the value at position i AND position j OF is_pal TO True  
                END IF  
            END FOR  
        END FOR  

        FOR i FROM one TO n MINUS two  
            IF the value at position zero AND position i MINUS one OF is_pal IS True  
                FOR j FROM i TO n MINUS two  
                    IF the value at position i AND position j OF is_pal IS True AND the value at position j PLUS one AND position n MINUS one OF is_pal IS True  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
        END FOR  

        RETURN False  
    END FUNCTION  
END CLASS
