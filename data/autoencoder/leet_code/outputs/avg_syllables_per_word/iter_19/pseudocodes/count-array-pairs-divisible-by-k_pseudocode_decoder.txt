CLASS Solution  
    FUNCTION countPairs(nums, k)  
        SET gcd_count TO an empty collection that counts frequencies of elements  
    
        FOR each num IN nums  
            INCREMENT the count of the greatest common divisor of num AND k in gcd_count BY one  
        END FOR  
    
        SET total_pairs TO zero  
        SET gcd_values TO the list of all unique keys in gcd_count  
        SET n TO the LENGTH OF gcd_values  
    
        FOR i FROM zero TO n MINUS one  
            FOR j FROM i TO n MINUS one  
                SET gcd1 TO element at position i of gcd_values  
                SET gcd2 TO element at position j of gcd_values  
    
                IF the remainder of the product of gcd1 AND gcd2 DIVIDED BY k EQUALS zero THEN  
                    IF i EQUALS j THEN  
                        INCREMENT total_pairs BY the product of the count of gcd1 IN gcd_count AND the difference of the count of gcd1 IN gcd_count MINUS one DIVIDED BY two  
                    ELSE  
                        INCREMENT total_pairs BY the product of the count of gcd1 IN gcd_count AND the count of gcd2 IN gcd_count  
                    END IF  
                END IF  
            END FOR  
        END FOR  
    
        RETURN total_pairs  
    END FUNCTION  
END CLASS
