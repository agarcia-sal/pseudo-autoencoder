CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty THEN  
        RETURN none  
    END IF  
    SET head TO a new ListNode with value element at position zero of values  
    SET p TO head  
    FOR each val IN elements of values from position one TO the end  
        SET node TO a new ListNode with value val  
        SET next element of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none THEN  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none THEN  
        RETURN false  
    END IF  
    RETURN the value element of p1 EQUALS the value element of p2 AND the result of calling is_same_list with the next element of p1 AND the next element of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1, l2)  
        SET dummy_head TO a new ListNode with value zero  
        SET current TO dummy_head  
        SET carry TO zero  

        WHILE l1 IS not none OR l2 IS not none OR carry IS not zero  
            IF l1 IS not none THEN  
                SET val1 TO the value element of l1  
            ELSE  
                SET val1 TO zero  
            END IF  

            IF l2 IS not none THEN  
                SET val2 TO the value element of l2  
            ELSE  
                SET val2 TO zero  
            END IF  

            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET next element of current TO a new ListNode with value total MODULO ten  
            SET current TO the next element of current  

            IF l1 IS not none THEN  
                SET l1 TO the next element of l1  
            END IF  

            IF l2 IS not none THEN  
                SET l2 TO the next element of l2  
            END IF  
        END WHILE  

        RETURN the next element of dummy_head  
    END FUNCTION  
END CLASS
