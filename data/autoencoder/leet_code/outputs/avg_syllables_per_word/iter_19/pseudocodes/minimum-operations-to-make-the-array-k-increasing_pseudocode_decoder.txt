CLASS Solution  
    FUNCTION kIncreasing WITH PARAMETERS arr AND k  
        FUNCTION longest_increasing_subsequence WITH PARAMETER subarr  
            SET lis TO empty list  
            FOR each number IN subarr  
                IF lis IS empty OR number IS GREATER THAN OR EQUAL TO element at the last position of lis  
                    APPEND number TO lis  
                ELSE  
                    SET pos TO the position where number can be inserted to maintain sorted order in lis using bisect right method  
                    SET element at position pos of lis TO number  
                END IF  
            END FOR  
            RETURN the LENGTH OF lis  
        END FUNCTION  
        
        SET total_operations TO zero  
        SET n TO the LENGTH OF arr  
        
        FOR start FROM zero TO k MINUS one  
            SET subarr TO empty list  
            FOR i FROM start TO n MINUS one IN STEPS OF k  
                APPEND element at position i of arr TO subarr  
            END FOR  
            SET lis_length TO longest_increasing_subsequence WITH PARAMETER subarr  
            INCREMENT total_operations BY the LENGTH OF subarr MINUS lis_length  
        END FOR  
        
        RETURN total_operations  
    END FUNCTION  
END CLASS
