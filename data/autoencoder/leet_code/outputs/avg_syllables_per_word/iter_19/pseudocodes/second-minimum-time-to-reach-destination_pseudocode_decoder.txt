CLASS Solution  
    FUNCTION secondMinimum(n, edges, time, change)  
        SET graph TO a list of empty lists with size equal to n PLUS one  
        FOR each pair u and v IN edges  
            APPEND v TO element at position u of graph  
            APPEND u TO element at position v of graph  
        END FOR  
        
        SET queue TO a new double ended queue containing a tuple of one current vertex ZERO current time ZERO count of shortest or second shortest paths  
        SET minTimes TO a list of lists each containing two elements set to positive infinity with size equal to n PLUS one  
        SET element at position one index zero of minTimes TO zero  
        
        WHILE queue is not empty  
            REMOVE the first element from queue and ASSIGN it TO current currentTime count  
            
            IF current EQUALS n AND count EQUALS one  
                RETURN currentTime  
            END IF  
            
            SET quotient TO the integer division of currentTime by change  
            SET remainder TO the modulo of quotient by two  
            
            IF remainder EQUALS zero  
                SET nextTime TO currentTime PLUS time  
            ELSE  
                SET nextTime TO the sum of the product of the sum of quotient PLUS one and change PLUS time  
            END IF  
            
            FOR each neighbor IN element at position current of graph  
                IF element at position neighbor index zero of minTimes GREATER THAN nextTime  
                    SET element at position neighbor index one of minTimes TO element at position neighbor index zero of minTimes  
                    SET element at position neighbor index zero of minTimes TO nextTime  
                    APPEND a tuple of neighbor nextTime ZERO TO queue  
                ELSE IF element at position neighbor index zero of minTimes LESS THAN nextTime AND nextTime LESS THAN element at position neighbor index one of minTimes  
                    SET element at position neighbor index one of minTimes TO nextTime  
                    APPEND a tuple of neighbor nextTime ONE TO queue  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
