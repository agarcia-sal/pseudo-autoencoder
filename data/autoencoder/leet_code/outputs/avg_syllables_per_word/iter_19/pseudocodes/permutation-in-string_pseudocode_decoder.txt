CLASS Solution  
    FUNCTION checkInclusion(s1 AS string, s2 AS string) RETURNS boolean  
        IMPORT a structure to count frequency of elements called Counter  
        
        SET length of s1 TO len_s1  
        SET length of s2 TO len_s2  
        
        IF len_s1 IS GREATER THAN len_s2  
            RETURN boolean value false  
        END IF  
        
        SET count_s1 TO the frequency count of each character IN s1 USING Counter  
        SET count_s2 TO the frequency count of each character IN substring from position zero TO position len_s1 MINUS one of s2 USING Counter  
        
        IF count_s1 EQUALS count_s2  
            RETURN boolean value true  
        END IF  
        
        FOR each index i FROM len_s1 TO len_s2 MINUS one  
            INCREMENT the count of the character at position i of s2 IN count_s2 BY one  
            
            DECREMENT the count of the character at position i MINUS len_s1 of s2 IN count_s2 BY one  
            
            IF the count of the character at position i MINUS len_s1 of s2 IN count_s2 EQUALS zero  
                REMOVE the character at position i MINUS len_s1 of s2 FROM count_s2  
            END IF  
            
            IF count_s1 EQUALS count_s2  
                RETURN boolean value true  
            END IF  
        END FOR  
        
        RETURN boolean value false  
    END FUNCTION  
END CLASS
