CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF the LENGTH OF s EQUALS zero  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET numeric_difference TO the ORDINAL VALUE OF character c MINUS the ORDINAL VALUE OF element at position zero of s  
                SET numeric_difference TO numeric_difference MODULO twenty six  
                APPEND the STRING REPRESENTATION OF numeric_difference TO canonical_form  
            END FOR  
            RETURN the concatenation of elements in canonical_form JOINED BY comma  
        END FUNCTION  
        
        SET groups TO a new dictionary that assigns a new empty list for each new key  
        
        FOR each string s IN strings  
            SET canonical_form TO get_canonical_form with parameter s  
            APPEND string s TO the list at key canonical_form of groups  
        END FOR  
        
        RETURN the list of all values in groups  
    END FUNCTION  
END CLASS
