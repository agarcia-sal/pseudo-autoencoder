CLASS Solution  
    FUNCTION maximumScore(nums AS List of integers, k AS integer) RETURNS integer  
        SET left TO k  
        SET right TO k  
        SET current_min TO element at position k of nums  
        SET max_score TO current_min  
        
        WHILE left is greater than zero OR right is less than the length of nums MINUS one  
            IF left EQUALS zero  
                INCREMENT right BY one  
            ELSE IF right EQUALS the length of nums MINUS one  
                DECREMENT left BY one  
            ELSE IF element at position left MINUS one of nums IS GREATER THAN OR EQUAL TO element at position right PLUS one of nums  
                DECREMENT left BY one  
            ELSE  
                INCREMENT right BY one  
            END IF  
            
            SET current_min TO the minimum value among current_min, element at position left of nums, and element at position right of nums  
            SET score TO current_min MULTIPLIED BY the result of right MINUS left PLUS one  
            SET max_score TO the maximum value between max_score and score  
        END WHILE  
        
        RETURN max_score  
    END FUNCTION  
END CLASS
