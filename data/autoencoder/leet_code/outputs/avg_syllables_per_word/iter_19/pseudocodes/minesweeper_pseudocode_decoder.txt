CLASS Solution  
    FUNCTION updateBoard(board, click)  
        FUNCTION count_adjacent_mines(r, c)  
            SET count TO zero  
            FOR i FROM r MINUS one TO r PLUS one  
                FOR j FROM c MINUS one TO c PLUS one  
                    IF i GREATER THAN OR EQUAL TO zero AND i LESS THAN the LENGTH OF board AND j GREATER THAN OR EQUAL TO zero AND j LESS THAN the LENGTH OF element at position zero of board AND element at position i and j of board EQUALS letter M  
                        INCREMENT count BY one  
                    END IF  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        
        FUNCTION dfs(r, c)  
            IF r LESS THAN zero OR r GREATER THAN OR EQUAL TO the LENGTH OF board OR c LESS THAN zero OR c GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of board  
                RETURN  
            END IF  
            IF element at position r and c of board NOT EQUALS letter E  
                RETURN  
            END IF  
            
            SET mine_count TO count_adjacent_mines(r, c)  
            IF mine_count GREATER THAN zero  
                SET element at position r and c of board TO the string representation of mine_count  
            ELSE  
                SET element at position r and c of board TO letter B  
                FOR i FROM r MINUS one TO r PLUS one  
                    FOR j FROM c MINUS one TO c PLUS one  
                        CALL dfs WITH i AND j  
                    END FOR  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET row TO element at position zero of click  
        SET col TO element at position one of click  
        IF element at position row and col of board EQUALS letter M  
            SET element at position row and col of board TO letter X  
            RETURN board  
        END IF  
        
        CALL dfs WITH row AND col  
        RETURN board  
    END FUNCTION  
END CLASS
