CLASS Solution  
    FUNCTION canTransform(start, end)  
        SET start_filtered TO an empty list  
        FOR each character c and index i IN start  
            IF character c NOT EQUALS character X  
                APPEND the pair of character c and index i TO start_filtered  
            END IF  
        END FOR  
        SET end_filtered TO an empty list  
        FOR each character c and index i IN end  
            IF character c NOT EQUALS character X  
                APPEND the pair of character c and index i TO end_filtered  
            END IF  
        END FOR  
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        FOR each pair of character c1 and index i1 FROM start_filtered AND each pair of character c2 and index i2 FROM end_filtered  
            IF character c1 NOT EQUALS character c2  
                RETURN false  
            END IF  
            IF character c1 EQUALS character L AND index i1 LESS THAN index i2  
                RETURN false  
            END IF  
            IF character c1 EQUALS character R AND index i1 GREATER THAN index i2  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
