CLASS Solution  
    FUNCTION longestPath(parent AS List of integers, s AS string) RETURNS integer  
        DEFINE tree AS a mapping from integer to list of integers with default empty list  
        FOR each index and element in parent  
            IF element IS NOT EQUAL TO negative one  
                APPEND index TO the list for key element in tree  
            END IF  
        END FOR  
        
        SET self dot result TO one  
        
        FUNCTION dfs(node AS integer) RETURNS integer  
            SET max1 TO zero  
            SET max2 TO zero  
            
            FOR each child IN the list at key node in tree  
                SET child_length TO the result of dfs called with child  
                IF the character at position child of string s IS NOT EQUAL TO the character at position node of string s  
                    IF child_length IS GREATER THAN max1  
                        SET max2 TO max1  
                        SET max1 TO child_length  
                    ELSE IF child_length IS GREATER THAN max2  
                        SET max2 TO child_length  
                    END IF  
                END IF  
            END FOR  
            
            SET self dot result TO the maximum of self dot result AND max1 PLUS max2 PLUS one  
            
            RETURN max1 PLUS one  
        END FUNCTION  
        
        CALL dfs with zero  
        
        RETURN self dot result  
    END FUNCTION  
END CLASS
