CLASS Solution  
    FUNCTION maxNumber(nums1 AS List of integers, nums2 AS List of integers, k AS integer) AS List of integers  
        FUNCTION maxSingleArray(nums AS List of integers, k AS integer) AS List of integers  
            SET drop TO the LENGTH OF nums MINUS k  
            SET stack TO empty list  
            FOR each num IN nums  
                WHILE drop IS GREATER THAN zero AND stack IS NOT empty AND the element at the last position of stack IS LESS THAN num  
                    REMOVE the element at the last position FROM stack  
                    DECREMENT drop BY one  
                END WHILE  
                APPEND num TO stack  
            END FOR  
            RETURN the SUBLIST of stack FROM position zero TO k NOT INCLUDING k  
        END FUNCTION  

        FUNCTION merge(arr1 AS List of integers, arr2 AS List of integers) AS List of integers  
            SET merged_list TO empty list  
            FOR each position FROM zero TO the SUM of the LENGTH OF arr1 AND the LENGTH OF arr2 MINUS one  
                IF arr1 IS GREATER THAN arr2 starting from their current positions ASSUMING LEXICOGRAPHICAL COMPARISON THEN  
                    APPEND the element at position zero of arr1 TO merged_list  
                    REMOVE the element at position zero FROM arr1  
                ELSE  
                    APPEND the element at position zero of arr2 TO merged_list  
                    REMOVE the element at position zero FROM arr2  
                END IF  
            END FOR  
            RETURN merged_list  
        END FUNCTION  

        SET max_number TO empty list  
        FOR i FROM zero TO k  
            IF i IS LESS THAN OR EQUAL TO the LENGTH OF nums1 AND k MINUS i IS LESS THAN OR EQUAL TO the LENGTH OF nums2 THEN  
                SET candidate TO the RESULT of merge WITH ARGUMENTS maxSingleArray(nums1, i) AND maxSingleArray(nums2, k MINUS i)  
                IF max_number IS LESS THAN candidate ASSUMING LEXICOGRAPHICAL COMPARISON THEN  
                    SET max_number TO candidate  
                END IF  
            END IF  
        END FOR  
        RETURN max_number  
    END FUNCTION  
END CLASS
