CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO an empty dictionary  
        FOR each num IN nums  
            SET depth TO num DIVIDED BY one hundred  
            SET pos TO the remainder of num DIVIDED BY one hundred DIVIDED BY ten  
            SET value TO the remainder of num DIVIDED BY ten  
            SET the element at position depth and position pos of tree TO value  
        END FOR  
        
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO the element at position depth and position pos of tree IF EXISTS OTHERWISE null  
            IF current_value EQUALS null  
                RETURN zero  
            END IF  
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child_depth TO depth PLUS one  
            SET left_child_pos TO pos MULTIPLIED BY two MINUS one  
            SET right_child_depth TO depth PLUS one  
            SET right_child_pos TO pos MULTIPLIED BY two  
            IF the element at position left_child_depth and position left_child_pos of tree DOES NOT EXIST  
               AND the element at position right_child_depth and position right_child_pos of tree DOES NOT EXIST  
                RETURN new_path_sum  
            END IF  
            SET left_sum TO CALL dfs WITH left_child_depth left_child_pos new_path_sum  
            SET right_sum TO CALL dfs WITH right_child_depth right_child_pos new_path_sum  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        
        RETURN CALL dfs WITH one one zero  
    END FUNCTION  
END CLASS
