CLASS Solution
    FUNCTION tallestBillboard(rods: List of Integer) RETURNS Integer
        SET dp TO dictionary with key 0 and value 0
        
        FOR each rod IN rods
            SET new_dp TO copy of dp
            FOR each d, y IN dp
                SET sum_taller TO new_dp GET value for key d + rod, default 0
                SET new_dp[d + rod] TO maximum of sum_taller and y
                
                SET diff TO absolute value of d - rod
                SET sum_shorter TO new_dp GET value for key diff, default 0
                SET candidate TO y + minimum of d and rod
                SET new_dp[diff] TO maximum of sum_shorter and candidate
            SET dp TO new_dp
        
        RETURN dp GET value for key 0
    END FUNCTION
END CLASS
