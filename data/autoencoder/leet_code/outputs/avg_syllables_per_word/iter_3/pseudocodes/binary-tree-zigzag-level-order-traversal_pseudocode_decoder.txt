CLASS TreeNode
    FUNCTION __init__(self, val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS empty
        RETURN None
    END IF

    SET root TO TreeNode(values[0])
    SET i TO 1
    INITIALIZE queue AS empty deque
    APPEND root TO queue

    WHILE queue IS NOT empty
        SET node TO queue.popleft()

        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1

        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE

    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION zigzagLevelOrder(self, root: Optional[TreeNode])
        IF root IS None
            RETURN empty list
        END IF

        INITIALIZE result AS empty list
        INITIALIZE queue AS deque CONTAINING root
        SET left_to_right TO True

        WHILE queue IS NOT empty
            SET level_size TO LENGTH OF queue
            INITIALIZE current_level AS empty deque

            FOR index FROM 1 TO level_size
                SET node TO queue.popleft()
                IF left_to_right IS True
                    APPEND node.val TO current_level
                ELSE
                    APPENDLEFT node.val TO current_level
                END IF

                IF node.left EXISTS
                    APPEND node.left TO queue
                END IF
                IF node.right EXISTS
                    APPEND node.right TO queue
                END IF
            END FOR

            APPEND LIST CONVERSION OF current_level TO result
            SET left_to_right TO NOT left_to_right
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
