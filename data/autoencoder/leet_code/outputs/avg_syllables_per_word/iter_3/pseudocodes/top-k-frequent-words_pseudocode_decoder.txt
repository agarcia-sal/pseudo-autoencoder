CLASS Solution
    FUNCTION topKFrequent(words, k)
        SET count TO frequency count of elements in words
        SET heap TO list of tuples (-frequency, word) for each word and frequency in count
        TRANSFORM heap into a heap structure
        SET result TO empty list
        FOR i FROM 1 TO k
            APPEND second element of extracted smallest tuple from heap TO result
        RETURN result
    END FUNCTION
END CLASS
