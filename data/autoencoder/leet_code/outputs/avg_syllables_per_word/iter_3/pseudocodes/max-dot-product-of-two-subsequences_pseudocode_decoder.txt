CLASS Solution
    FUNCTION maxDotProduct(nums1, nums2)
        SET m TO LENGTH OF nums1
        SET n TO LENGTH OF nums2
        INITIALIZE dp AS A TABLE OF SIZE (m + 1) BY (n + 1) WITH ALL VALUES SET TO NEGATIVE_INFINITY

        FOR i FROM 1 TO m
            FOR j FROM 1 TO n
                SET current_product TO nums1 AT POSITION (i - 1) MULTIPLIED BY nums2 AT POSITION (j - 1)
                SET dp[i][j] TO MAXIMUM OF
                    current_product,
                    dp[i - 1][j],
                    dp[i][j - 1],
                    dp[i - 1][j - 1] PLUS current_product

        RETURN dp[m][n]
    END FUNCTION
END CLASS
