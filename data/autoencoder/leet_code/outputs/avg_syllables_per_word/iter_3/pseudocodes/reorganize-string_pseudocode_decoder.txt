CLASS Solution
    FUNCTION reorganizeString(s)
        SET count TO frequency count of characters in s
        SET max_heap TO list of pairs (-frequency, character) for each character and frequency in count
        CONVERT max_heap INTO a heap structure
        
        SET prev_char TO None
        SET prev_freq TO 0
        SET result TO empty list
        
        WHILE max_heap IS NOT empty OR prev_freq IS LESS THAN 0
            IF max_heap IS empty AND prev_freq IS LESS THAN 0
                RETURN empty string
            
            REMOVE the top element FROM max_heap INTO freq, char
            APPEND char TO result
            
            INCREMENT freq BY 1  // since freq is negative, this reduces its absolute count
            
            IF prev_freq IS LESS THAN 0
                ADD (prev_freq, prev_char) BACK INTO max_heap
                
            SET prev_char TO char
            SET prev_freq TO freq
        
        RETURN concatenation of all characters in result
    END FUNCTION
END CLASS
