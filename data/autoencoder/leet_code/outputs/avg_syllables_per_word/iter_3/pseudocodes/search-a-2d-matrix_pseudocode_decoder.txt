CLASS Solution
    FUNCTION searchMatrix(matrix, target)
        IF matrix IS EMPTY OR matrix[0] IS EMPTY THEN
            RETURN False
        SET m TO LENGTH OF matrix
        SET n TO LENGTH OF matrix[0]
        SET left TO 0
        SET right TO (m * n) MINUS 1
        WHILE left IS LESS THAN OR EQUAL TO right
            SET mid TO (left PLUS right) DIVIDED BY 2
            SET mid_value TO matrix AT (mid DIVIDED BY n) AT (mid MODULO n)
            IF mid_value EQUALS target THEN
                RETURN True
            ELSE IF mid_value IS LESS THAN target THEN
                SET left TO mid PLUS 1
            ELSE
                SET right TO mid MINUS 1
        RETURN False
END CLASS
