CLASS Solution
    FUNCTION addBoldTag(s, words)
        IF words IS EMPTY
            RETURN s
        
        SET n TO LENGTH OF s
        SET mask TO LIST OF n False VALUES
        
        FOR EACH word IN words
            SET start TO 0
            WHILE start <= n MINUS LENGTH OF word
                SET start TO POSITION OF word IN s STARTING FROM start
                IF start IS -1
                    BREAK
                FOR i FROM start TO start PLUS LENGTH OF word MINUS 1
                    SET mask[i] TO True
                INCREMENT start BY 1
        
        SET result TO EMPTY LIST
        SET i TO 0
        WHILE i < n
            IF mask[i] IS True
                APPEND "<b>" TO result
                WHILE i < n AND mask[i] IS True
                    APPEND s[i] TO result
                    INCREMENT i BY 1
                APPEND "</b>" TO result
            ELSE
                APPEND s[i] TO result
                INCREMENT i BY 1
        
        RETURN CONCATENATION OF ELEMENTS IN result
    END FUNCTION
END CLASS
