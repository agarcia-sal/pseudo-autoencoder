CLASS Solution
    FUNCTION matrixScore(grid)
        FOR each row IN grid
            IF row[0] EQUALS 0 THEN
                FOR i FROM 0 TO LENGTH OF row MINUS 1
                    SET row[i] TO 1 MINUS row[i]
        FOR col FROM 1 TO LENGTH OF grid[0] MINUS 1
            SET count_ones TO 0
            FOR row FROM 0 TO LENGTH OF grid MINUS 1
                INCREMENT count_ones BY grid[row][col]
            IF count_ones LESS THAN LENGTH OF grid DIVIDED BY 2 THEN
                FOR row FROM 0 TO LENGTH OF grid MINUS 1
                    SET grid[row][col] TO 1 MINUS grid[row][col]
        SET score TO 0
        FOR each row IN grid
            SET binary_string TO EMPTY STRING
            FOR each value IN row
                APPEND string representation of value TO binary_string
            SET score TO score PLUS integer value of binary_string IN BASE 2
        RETURN score
    END FUNCTION
END CLASS
