CLASS Solution
    FUNCTION minStickers(stickers: List of String, target: String) RETURNS Integer
        SET stickers TO list of Counters created from each sticker in stickers
        SORT stickers by descending order of their maximum frequency character count
        
        SET queue TO a deque containing tuple(target, 0)
        SET memo TO dictionary with key target and value 0
        
        WHILE queue is not empty
            SET current_target, used_stickers TO dequeue from queue
            
            FOR each sticker IN stickers
                IF first character of current_target NOT IN sticker
                    CONTINUE to next sticker
                
                SET remaining TO Counter of current_target
                SUBTRACT sticker counts from remaining
                
                SET new_target TO string formed by sorting elements in remaining
                
                IF new_target is empty
                    RETURN used_stickers + 1
                
                IF new_target NOT IN memo
                    SET memo[new_target] TO used_stickers + 1
                    APPEND tuple(new_target, used_stickers + 1) TO queue
        
        RETURN -1
    END FUNCTION
END CLASS
