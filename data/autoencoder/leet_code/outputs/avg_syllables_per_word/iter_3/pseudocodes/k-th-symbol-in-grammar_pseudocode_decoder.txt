CLASS Solution
    FUNCTION kthGrammar(n, k)
        IF n EQUALS 1
            RETURN 0
        SET length_of_previous_row TO 2 RAISED TO (n MINUS 2)
        IF k LESS THAN OR EQUAL TO length_of_previous_row
            RETURN kthGrammar(n MINUS 1, k)
        ELSE
            RETURN 1 MINUS kthGrammar(n MINUS 1, k MINUS length_of_previous_row)
    END FUNCTION
END CLASS
