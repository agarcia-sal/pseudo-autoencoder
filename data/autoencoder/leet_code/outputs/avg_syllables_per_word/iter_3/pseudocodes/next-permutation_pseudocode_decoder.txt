CLASS Solution
    FUNCTION nextPermutation(nums)
        SET i TO LENGTH OF nums MINUS 2
        WHILE i >= 0 AND nums[i] >= nums[i + 1]
            DECREMENT i BY 1
        IF i >= 0
            SET j TO LENGTH OF nums MINUS 1
            WHILE nums[j] <= nums[i]
                DECREMENT j BY 1
            SWAP nums[i] AND nums[j]
        SET start TO i + 1
        SET end TO LENGTH OF nums MINUS 1
        WHILE start < end
            SWAP nums[start] AND nums[end]
            INCREMENT start BY 1
            DECREMENT end BY 1
    END FUNCTION
END CLASS
