CLASS Solution
    FUNCTION pourWater(heights: List of integers, volume: integer, k: integer) RETURNS List of integers
        SET n TO LENGTH OF heights
        
        FOR each drop FROM 1 TO volume
            SET i TO k
            WHILE i > 0 AND heights[i - 1] <= heights[i]
                DECREMENT i BY 1
            WHILE i < k AND heights[i] == heights[i + 1]
                INCREMENT i BY 1
            IF i < k
                INCREMENT heights[i] BY 1
                CONTINUE TO next drop
            
            SET i TO k
            WHILE i < n - 1 AND heights[i + 1] <= heights[i]
                INCREMENT i BY 1
            WHILE i > k AND heights[i] == heights[i - 1]
                DECREMENT i BY 1
            IF i > k
                INCREMENT heights[i] BY 1
                CONTINUE TO next drop
            
            INCREMENT heights[k] BY 1
        
        RETURN heights
    END FUNCTION
END CLASS
