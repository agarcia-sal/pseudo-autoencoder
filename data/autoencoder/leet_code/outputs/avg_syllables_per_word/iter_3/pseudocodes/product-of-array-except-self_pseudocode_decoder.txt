CLASS Solution
    FUNCTION productExceptSelf(nums)
        SET n TO LENGTH OF nums
        SET answer TO LIST OF 1 REPEATED n TIMES
        
        SET left_product TO 1
        FOR i FROM 0 TO n - 1
            SET answer[i] TO left_product
            SET left_product TO left_product MULTIPLIED BY nums[i]
        
        SET right_product TO 1
        FOR i FROM n - 1 DOWNTO 0
            SET answer[i] TO answer[i] MULTIPLIED BY right_product
            SET right_product TO right_product MULTIPLIED BY nums[i]
        
        RETURN answer
    END FUNCTION
END CLASS
