CLASS Solution
    FUNCTION minInteger(num: string, k: integer) RETURNS string
        SET num TO list of characters from num
        SET n TO length of num
        
        WHILE k GREATER THAN 0
            SET swapped TO False
            FOR i FROM 0 TO n MINUS 2
                IF num at i GREATER THAN num at i PLUS 1
                    SET min_digit TO '9'
                    SET min_index TO -1
                    FOR j FROM i TO MINIMUM of (i PLUS k PLUS 1) AND n MINUS 1
                        IF num at j LESS THAN min_digit
                            SET min_digit TO num at j
                            SET min_index TO j
                    SET num slice from i TO min_index PLUS 1 TO list with min_digit followed by num slice from i TO min_index
                    DECREMENT k BY (min_index MINUS i)
                    SET swapped TO True
                    BREAK out of FOR i loop
            IF swapped IS False
                BREAK out of WHILE loop
        
        RETURN string joined from list num
    END FUNCTION
END CLASS
