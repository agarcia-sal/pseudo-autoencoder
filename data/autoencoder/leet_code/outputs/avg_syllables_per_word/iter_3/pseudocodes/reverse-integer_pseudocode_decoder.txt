CLASS Solution
    FUNCTION reverse(x)
        IF x < 0 THEN
            SET sign TO -1
        ELSE
            SET sign TO 1
        END IF
        SET absolute_value TO absolute of x
        SET reversed_string TO reverse of string representation of absolute_value
        SET reversed_x TO integer conversion of reversed_string
        SET result TO sign multiplied by reversed_x
        IF result < negative two raised to thirty-one OR result > two raised to thirty-one minus one THEN
            RETURN 0
        END IF
        RETURN result
    END FUNCTION
END CLASS
