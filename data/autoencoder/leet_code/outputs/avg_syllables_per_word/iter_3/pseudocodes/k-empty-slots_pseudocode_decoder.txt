CLASS Solution
    FUNCTION kEmptySlots(bulbs, k)
        SET bloom_days TO empty list
        
        FOR day, position IN bulbs WITH INDEX STARTING AT 1
            SET i TO position to insert in bloom_days preserving order
            
            IF i > 0 AND position MINUS bloom_days[i - 1] MINUS 1 EQUALS k
                RETURN day
            END IF
            
            IF i LESS THAN LENGTH OF bloom_days AND bloom_days[i] MINUS position MINUS 1 EQUALS k
                RETURN day
            END IF
            
            INSERT position INTO bloom_days AT INDEX i
        END FOR
        
        RETURN -1
    END FUNCTION
END CLASS
