CLASS Solution
    FUNCTION maxJumps(arr, d)
        FUNCTION dp(i)
            SET max_jumps TO 1
            FOR x FROM 1 TO d
                IF i + x IS LESS THAN LENGTH OF arr THEN
                    IF arr[i + x] IS GREATER THAN OR EQUAL TO arr[i] THEN
                        BREAK
                    SET max_jumps TO MAXIMUM OF max_jumps AND (1 + dp(i + x))
            FOR x FROM 1 TO d
                IF i - x IS GREATER THAN OR EQUAL TO 0 THEN
                    IF arr[i - x] IS GREATER THAN OR EQUAL TO arr[i] THEN
                        BREAK
                    SET max_jumps TO MAXIMUM OF max_jumps AND (1 + dp(i - x))
            RETURN max_jumps
        END FUNCTION
        RETURN MAXIMUM OF dp(i) FOR EACH i FROM 0 TO LENGTH OF arr MINUS 1
    END FUNCTION
END CLASS
