CLASS Solution
    FUNCTION search(nums, target)
        IF nums IS EMPTY
            RETURN False
        SET left TO 0
        SET right TO LENGTH OF nums MINUS 1
        WHILE left IS LESS THAN OR EQUAL TO right
            SET mid TO left PLUS (right MINUS left) DIVIDED BY 2
            IF nums[mid] EQUALS target
                RETURN True
            IF nums[left] EQUALS nums[mid] AND nums[mid] EQUALS nums[right]
                INCREMENT left BY 1
                DECREMENT right BY 1
            ELSE IF nums[left] IS LESS THAN OR EQUAL TO nums[mid]
                IF nums[left] IS LESS THAN OR EQUAL TO target AND target IS LESS THAN nums[mid]
                    SET right TO mid MINUS 1
                ELSE
                    SET left TO mid PLUS 1
            ELSE
                IF nums[mid] IS LESS THAN target AND target IS LESS THAN OR EQUAL TO nums[right]
                    SET left TO mid PLUS 1
                ELSE
                    SET right TO mid MINUS 1
        RETURN False
    END FUNCTION
END CLASS
