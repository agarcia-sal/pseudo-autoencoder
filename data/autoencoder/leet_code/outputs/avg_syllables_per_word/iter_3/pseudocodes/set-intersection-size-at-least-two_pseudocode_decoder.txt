CLASS Solution
    FUNCTION intersectionSizeTwo(intervals)
        SORT intervals BY end ASCENDING, start DESCENDING
        SET first TO intervals[0][1] MINUS 1
        SET second TO intervals[0][1]
        SET count TO 2
        FOR each (start, end) IN intervals FROM SECOND ELEMENT TO END
            IF start > second THEN
                SET first TO end MINUS 1
                SET second TO end
                INCREMENT count BY 2
            ELSE IF start > first THEN
                SET first TO second
                SET second TO end
                INCREMENT count BY 1
        RETURN count
    END FUNCTION
END CLASS
