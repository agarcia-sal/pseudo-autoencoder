CLASS Solution
    FUNCTION hasPath(maze, start, destination)
        SET directions TO [(0, 1), (1, 0), (0, -1), (-1, 0)]
        SET m TO LENGTH OF maze
        SET n TO LENGTH OF maze[0]
        SET visited TO empty set

        FUNCTION dfs(x, y)
            IF (x, y) IN visited THEN
                RETURN False
            IF [x, y] EQUALS destination THEN
                RETURN True
            ADD (x, y) TO visited

            FOR EACH (dx, dy) IN directions DO
                SET nx TO x
                SET ny TO y
                WHILE nx + dx IS WITHIN 0 AND m - 1 AND ny + dy IS WITHIN 0 AND n - 1 AND maze[nx + dx][ny + dy] EQUALS 0 DO
                    INCREMENT nx BY dx
                    INCREMENT ny BY dy
                IF dfs(nx, ny) THEN
                    RETURN True
            RETURN False
        END FUNCTION

        RETURN dfs(start[0], start[1])
    END FUNCTION
END CLASS
