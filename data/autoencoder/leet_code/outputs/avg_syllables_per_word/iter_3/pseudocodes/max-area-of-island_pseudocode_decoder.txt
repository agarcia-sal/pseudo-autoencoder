CLASS Solution
    FUNCTION maxAreaOfIsland(grid: List of List of Integer) RETURNS Integer
        IF grid IS EMPTY
            RETURN 0
        SET max_area TO 0
        SET rows TO LENGTH OF grid
        SET cols TO LENGTH OF grid[0]

        FUNCTION dfs(r, c) RETURNS Integer
            IF r < 0 OR r >= rows OR c < 0 OR c >= cols OR grid[r][c] = 0
                RETURN 0
            SET grid[r][c] TO 0
            SET area TO 1
            INCREMENT area BY dfs(r + 1, c)
            INCREMENT area BY dfs(r - 1, c)
            INCREMENT area BY dfs(r, c + 1)
            INCREMENT area BY dfs(r, c - 1)
            RETURN area
        END FUNCTION

        FOR r FROM 0 TO rows - 1
            FOR c FROM 0 TO cols - 1
                IF grid[r][c] = 1
                    SET max_area TO MAXIMUM OF max_area AND dfs(r, c)
        RETURN max_area
    END FUNCTION
END CLASS
