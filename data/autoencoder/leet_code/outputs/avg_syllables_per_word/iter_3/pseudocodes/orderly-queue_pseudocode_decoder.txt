CLASS Solution
    FUNCTION orderlyQueue(s, k)
        IF k GREATER THAN 1 THEN
            RETURN string formed by sorting characters of s in ascending order
        ELSE
            SET n TO length of s
            SET smallest TO s
            FOR i FROM 1 TO n MINUS 1 DO
                SET rotated TO concatenation of substring of s from index i to end and substring of s from start to index i
                IF rotated lexicographically less than smallest THEN
                    SET smallest TO rotated
            RETURN smallest
    END FUNCTION
END CLASS
