CLASS Solution
    FUNCTION lengthOfLIS(nums)
        IF nums IS EMPTY
            RETURN 0
        END IF
        
        SET dp TO empty list
        
        FOR each num IN nums
            SET idx TO position to insert num in dp to keep dp sorted (using binary search)
            
            IF idx IS EQUAL TO length of dp
                APPEND num TO dp
            ELSE
                SET dp[idx] TO num
            END IF
        END FOR
        
        RETURN length of dp
    END FUNCTION
END CLASS
