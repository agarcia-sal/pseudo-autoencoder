CLASS Solution
    FUNCTION kthSmallestPrimeFraction(arr, k)
        SET n TO LENGTH OF arr
        SET min_heap TO empty list
        
        FOR i FROM 0 TO n MINUS 2
            PUSH (arr[i] DIVIDED BY arr[n MINUS 1], i, n MINUS 1) INTO min_heap
        
        FOR count FROM 1 TO k MINUS 1
            POP smallest_fraction, i, j FROM min_heap
            IF j MINUS 1 GREATER THAN i
                PUSH (arr[i] DIVIDED BY arr[j MINUS 1], i, j MINUS 1) INTO min_heap
        
        POP _, i, j FROM min_heap
        RETURN [arr[i], arr[j]]
    END FUNCTION
END CLASS
