CLASS Solution
    FUNCTION findIntegers(n)
        SET bin_rep TO binary representation of n as string without prefix
        SET length TO length of bin_rep
        
        SET dp TO list of zeros with size length plus 1
        SET dp[0] TO 1
        SET dp[1] TO 2
        
        FOR i FROM 2 TO length
            SET dp[i] TO dp[i minus 1] plus dp[i minus 2]
        
        SET result TO 0
        SET prev_bit TO 0
        
        FOR i FROM 0 TO length minus 1
            IF bin_rep at position i IS '1'
                INCREMENT result BY dp[length minus i minus 1]
                IF prev_bit EQUALS 1
                    DECREMENT result BY 1
                    EXIT LOOP
            
            SET prev_bit TO 1 IF bin_rep at i IS '1' ELSE 0
        
        RETURN result plus 1
    END FUNCTION
END CLASS
