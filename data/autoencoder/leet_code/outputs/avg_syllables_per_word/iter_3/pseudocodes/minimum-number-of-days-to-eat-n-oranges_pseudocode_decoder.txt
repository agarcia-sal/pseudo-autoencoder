CLASS Solution
    FUNCTION minDays(n)
        DEFINE FUNCTION dp(remaining) WITH CACHE
            IF remaining <= 1 THEN
                RETURN remaining
            END IF
            RETURN 1 PLUS MINIMUM OF (remaining MODULO 2 PLUS dp(remaining DIVIDED BY 2), remaining MODULO 3 PLUS dp(remaining DIVIDED BY 3))
        END FUNCTION
        RETURN dp(n)
    END FUNCTION
END CLASS
