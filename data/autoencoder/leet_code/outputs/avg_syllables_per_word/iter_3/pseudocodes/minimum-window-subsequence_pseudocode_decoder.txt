CLASS Solution
    FUNCTION minWindow(s1, s2)
        SET min_len TO infinity
        SET min_start TO -1
        SET len1 TO LENGTH OF s1
        SET len2 TO LENGTH OF s2

        FOR i FROM 0 TO len1 - 1
            SET j TO 0
            FOR k FROM i TO len1 - 1
                IF s1[k] EQUALS s2[j]
                    INCREMENT j BY 1
                    IF j EQUALS len2
                        IF (k - i + 1) LESS THAN min_len
                            SET min_len TO k - i + 1
                            SET min_start TO i
                        BREAK INNER LOOP

        IF min_start EQUALS -1
            RETURN empty string
        ELSE
            RETURN SUBSTRING OF s1 FROM min_start TO min_start + min_len
    END FUNCTION
END CLASS
