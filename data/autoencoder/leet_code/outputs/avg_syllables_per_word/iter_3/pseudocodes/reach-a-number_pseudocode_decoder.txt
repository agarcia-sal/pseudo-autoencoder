CLASS Solution
    FUNCTION reachNumber(target: integer) RETURNS integer
        SET target TO absolute value of target
        SET k TO 0
        SET sum_steps TO 0
        WHILE sum_steps < target
            INCREMENT k BY 1
            INCREMENT sum_steps BY k
        END WHILE
        IF (sum_steps - target) MODULO 2 EQUALS 0
            RETURN k
        ELSE
            IF k MODULO 2 EQUALS 0
                RETURN k + 1
            ELSE
                RETURN k + 2
            END IF
        END IF
    END FUNCTION
END CLASS
