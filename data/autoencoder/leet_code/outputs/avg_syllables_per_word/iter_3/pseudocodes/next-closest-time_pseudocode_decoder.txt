CLASS Solution
    FUNCTION nextClosestTime(time)
        SET digits TO UNIQUE CHARACTERS IN time WITH ":" REMOVED
        SPLIT time INTO hour AND minute AS INTEGERS
        
        FUNCTION generate_times(digits)
            SET times TO EMPTY LIST
            FOR each h1 IN digits
                FOR each h2 IN digits
                    FOR each m1 IN digits
                        FOR each m2 IN digits
                            SET hour TO CONVERT CONCATENATION OF h1 AND h2 TO INTEGER
                            SET minute TO CONVERT CONCATENATION OF m1 AND m2 TO INTEGER
                            IF hour LESS THAN 24 AND minute LESS THAN 60 THEN
                                APPEND (hour, minute) TO times
            RETURN SORTED times
        END FUNCTION
        
        SET all_times TO generate_times(digits)
        SET current_time_index TO INDEX OF (hour, minute) IN all_times
        
        IF current_time_index + 1 LESS THAN LENGTH OF all_times THEN
            SET next_hour, next_minute TO all_times[current_time_index + 1]
        ELSE
            SET next_hour, next_minute TO all_times[0]
        
        RETURN FORMAT next_hour AND next_minute AS "HH:MM" WITH LEADING ZEROS
    END FUNCTION
END CLASS
