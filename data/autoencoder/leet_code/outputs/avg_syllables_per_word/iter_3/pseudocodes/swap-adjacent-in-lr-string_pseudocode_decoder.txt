CLASS Solution
    FUNCTION canTransform(start, end)
        SET start_filtered TO list of pairs (character, index) from start WHERE character IS NOT 'X'
        SET end_filtered TO list of pairs (character, index) from end WHERE character IS NOT 'X'
        IF LENGTH OF start_filtered IS NOT EQUAL TO LENGTH OF end_filtered THEN
            RETURN False
        FOR EACH pair (c1, i1), (c2, i2) IN start_filtered AND end_filtered
            IF c1 IS NOT EQUAL TO c2 THEN
                RETURN False
            IF c1 IS 'L' AND i1 IS LESS THAN i2 THEN
                RETURN False
            IF c1 IS 'R' AND i1 IS GREATER THAN i2 THEN
                RETURN False
        RETURN True
    END FUNCTION
END CLASS
