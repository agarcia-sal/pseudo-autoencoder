CLASS Solution
    FUNCTION numOfWays(n)
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7
        SET a TO 6
        SET b TO 6
        FOR i FROM 2 TO n
            SET temp_a TO (3 TIMES a PLUS 2 TIMES b) MODULO MOD
            SET temp_b TO (2 TIMES a PLUS 2 TIMES b) MODULO MOD
            SET a TO temp_a
            SET b TO temp_b
        END FOR
        RETURN (a PLUS b) MODULO MOD
    END FUNCTION
END CLASS
