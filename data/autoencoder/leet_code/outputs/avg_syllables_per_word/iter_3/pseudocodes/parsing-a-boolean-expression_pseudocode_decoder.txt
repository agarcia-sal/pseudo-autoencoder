CLASS Solution
    FUNCTION parseBoolExpr(expression)
        SET stk TO empty list
        FOR each character c IN expression
            IF c IS one of 't', 'f', '!', '&', '|'
                APPEND c TO stk
            ELSE IF c IS ')'
                SET t TO 0
                SET f TO 0
                WHILE last element in stk IS 't' OR 'f'
                    INCREMENT t BY (last element IS 't')
                    INCREMENT f BY (last element IS 'f')
                    REMOVE last element FROM stk
                SET op TO REMOVE last element FROM stk
                IF op IS '!'
                    SET c TO 't' IF f IS 0 ELSE 'f'
                ELSE IF op IS '&'
                    SET c TO 'f' IF f IS NOT 0 ELSE 't'
                ELSE IF op IS '|'
                    SET c TO 't' IF t IS NOT 0 ELSE 'f'
                APPEND c TO stk
        RETURN (first element OF stk IS 't')
    END FUNCTION
END CLASS
