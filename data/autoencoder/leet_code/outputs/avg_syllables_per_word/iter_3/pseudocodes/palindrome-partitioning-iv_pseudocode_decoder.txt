CLASS Solution
    FUNCTION checkPartitioning(s)
        FUNCTION is_palindrome(sub)
            RETURN sub EQUALS REVERSE OF sub
        END FUNCTION

        SET n TO LENGTH OF s
        SET is_pal TO 2D LIST OF False WITH DIMENSIONS n BY n

        FOR i FROM n - 1 DOWNTO 0
            FOR j FROM i TO n - 1
                IF s[i] EQUALS s[j] AND (j MINUS i LESS THAN OR EQUAL TO 1 OR is_pal[i + 1][j - 1] IS True)
                    SET is_pal[i][j] TO True

        FOR i FROM 1 TO n - 2
            IF is_pal[0][i - 1] IS True
                FOR j FROM i TO n - 2
                    IF is_pal[i][j] IS True AND is_pal[j + 1][n - 1] IS True
                        RETURN True

        RETURN False
    END FUNCTION
END CLASS
