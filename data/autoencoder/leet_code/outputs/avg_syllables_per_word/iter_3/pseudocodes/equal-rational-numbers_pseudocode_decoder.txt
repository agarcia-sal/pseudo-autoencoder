CLASS Solution
    FUNCTION isRationalEqual(s: string, t: string) RETURNS boolean
        SET ratios TO [1, 1 DIVIDED BY 9, 1 DIVIDED BY 99, 1 DIVIDED BY 999, 1 DIVIDED BY 9999]

        FUNCTION valueOf(s: string) RETURNS float
            IF s DOES NOT CONTAIN '(' THEN
                RETURN CONVERT s TO float
            END IF

            SET leftParenIndex TO POSITION OF '(' IN s
            SET rightParenIndex TO POSITION OF ')' IN s
            SET dotIndex TO POSITION OF '.' IN s

            SET integerAndNonRepeating TO CONVERT SUBSTRING OF s FROM START TO leftParenIndex TO float
            SET nonRepeatingLength TO leftParenIndex MINUS dotIndex MINUS 1

            SET repeating TO CONVERT SUBSTRING OF s FROM (leftParenIndex + 1) TO rightParenIndex TO integer
            SET repeatingLength TO rightParenIndex MINUS leftParenIndex MINUS 1

            RETURN integerAndNonRepeating PLUS repeating MULTIPLIED BY (0.1 RAISED TO nonRepeatingLength) MULTIPLIED BY ratios AT repeatingLength
        END FUNCTION

        RETURN ABSOLUTE VALUE OF (valueOf(s) MINUS valueOf(t)) IS LESS THAN 0.000000001
    END FUNCTION
END CLASS
