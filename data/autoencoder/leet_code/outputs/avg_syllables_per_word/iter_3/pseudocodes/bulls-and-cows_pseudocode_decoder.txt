CLASS Solution
    FUNCTION getHint(secret, guess)
        SET bulls TO 0
        SET cows TO 0
        CREATE secret_count AS a dictionary with default integer 0
        CREATE guess_count AS a dictionary with default integer 0

        FOR each pair s, g IN secret AND guess
            IF s EQUALS g THEN
                INCREMENT bulls BY 1
            ELSE
                INCREMENT secret_count[s] BY 1
                INCREMENT guess_count[g] BY 1

        FOR each digit IN secret_count KEYS
            INCREMENT cows BY MINIMUM of secret_count[digit] AND guess_count[digit]

        RETURN string composed of bulls FOLLOWED BY "A" FOLLOWED BY cows FOLLOWED BY "B"
    END FUNCTION
END CLASS
