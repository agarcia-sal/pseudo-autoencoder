CLASS Solution
    FUNCTION findLongestChain(pairs)
        SORT pairs BY second element of each pair
        SET current_end TO negative infinity
        SET chain_length TO 0
        FOR each pair IN pairs
            SET start TO first element of pair
            SET end TO second element of pair
            IF start GREATER THAN current_end THEN
                SET current_end TO end
                INCREMENT chain_length BY 1
        RETURN chain_length
    END FUNCTION
END CLASS
