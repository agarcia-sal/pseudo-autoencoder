CLASS Solution
    FUNCTION minDeletionSize(strs: List of String)
        SET m TO LENGTH OF strs
        SET n TO LENGTH OF strs[0]

        SET dp TO LIST OF n ONES

        FOR i FROM 1 TO n - 1
            FOR j FROM 0 TO i - 1
                IF FOR ALL k FROM 0 TO m - 1, strs[k][j] IS LESS THAN OR EQUAL TO strs[k][i]
                    SET dp[i] TO MAXIMUM OF dp[i] AND dp[j] PLUS 1

        RETURN n MINUS MAXIMUM VALUE IN dp
    END FUNCTION
END CLASS
