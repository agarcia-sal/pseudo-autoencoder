CLASS Solution
    FUNCTION spiralMatrixIII(rows, cols, rStart, cStart)
        SET directions TO [(0, 1), (1, 0), (0, -1), (-1, 0)]
        SET direction_index TO 0
        SET steps TO 0
        SET increment TO 0
        SET visited TO empty list
        SET x TO rStart
        SET y TO cStart
        
        WHILE LENGTH OF visited IS LESS THAN rows MULTIPLIED BY cols
            IF increment MODULO 2 EQUALS 0 THEN
                INCREMENT steps BY 1
            END IF
            
            FOR _ FROM 1 TO steps
                IF x IS AT LEAST 0 AND LESS THAN rows AND y IS AT LEAST 0 AND LESS THAN cols THEN
                    APPEND [x, y] TO visited
                END IF
                
                SET dx, dy TO directions AT direction_index
                INCREMENT x BY dx
                INCREMENT y BY dy
            END FOR
            
            SET direction_index TO (direction_index + 1) MODULO 4
            INCREMENT increment BY 1
        END WHILE
        
        RETURN visited
    END FUNCTION
END CLASS
