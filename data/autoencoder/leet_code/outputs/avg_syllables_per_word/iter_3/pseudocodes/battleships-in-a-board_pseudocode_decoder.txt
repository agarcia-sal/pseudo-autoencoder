CLASS Solution
    FUNCTION countBattleships(board)
        SET count TO 0
        FOR i FROM 0 TO LENGTH OF board MINUS 1
            FOR j FROM 0 TO LENGTH OF board[0] MINUS 1
                IF board[i][j] EQUALS 'X'
                    IF (i EQUALS 0 OR board[i-1][j] EQUALS '.') AND (j EQUALS 0 OR board[i][j-1] EQUALS '.')
                        INCREMENT count BY 1
        RETURN count
    END FUNCTION
END CLASS
