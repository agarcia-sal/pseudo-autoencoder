CLASS Solution
    FUNCTION wallsAndGates(rooms)
        IF rooms IS EMPTY OR rooms[0] IS EMPTY THEN
            RETURN
        SET rows TO LENGTH OF rooms
        SET cols TO LENGTH OF rooms[0]
        SET queue TO EMPTY DEQUE

        FOR r FROM 0 TO rows - 1
            FOR c FROM 0 TO cols - 1
                IF rooms[r][c] EQUALS 0 THEN
                    APPEND (r, c) TO queue

        SET directions TO LIST OF (0, 1), (1, 0), (0, -1), (-1, 0)

        WHILE queue IS NOT EMPTY
            SET (r, c) TO POP LEFT FROM queue
            FOR EACH (dr, dc) IN directions
                SET nr TO r + dr
                SET nc TO c + dc
                IF nr IS BETWEEN 0 AND rows - 1 AND nc IS BETWEEN 0 AND cols - 1 AND rooms[nr][nc] EQUALS 2147483647 THEN
                    SET rooms[nr][nc] TO rooms[r][c] + 1
                    APPEND (nr, nc) TO queue
    END FUNCTION
END CLASS
