CLASS Solution
    FUNCTION makeLargestSpecial(s)
        FUNCTION dfs(s)
            SET count TO 0
            SET start TO 0
            SET specials TO empty list
            
            FOR i FROM 0 TO LENGTH OF s MINUS 1
                SET char TO CHARACTER AT s[i]
                IF char IS '1'
                    INCREMENT count BY 1
                ELSE
                    DECREMENT count BY 1
                END IF
                
                IF count IS 0
                    APPEND CONCATENATION OF '1', dfs(SUBSTRING OF s FROM start+1 TO i MINUS 1), '0' TO specials
                    SET start TO i + 1
                END IF
            END FOR
            
            SORT specials IN REVERSE LEXICOGRAPHICAL ORDER
            RETURN CONCATENATION OF ALL ELEMENTS IN specials
        END FUNCTION
        
        RETURN dfs(s)
    END FUNCTION
END CLASS
