CLASS Solution
    FUNCTION minInsertions(s)
        FUNCTION longest_palindromic_subsequence(s)
            SET n TO LENGTH OF s
            CREATE 2D ARRAY dp WITH DIMENSIONS n BY n INITIALIZED TO 0

            FOR i FROM 0 TO n - 1
                SET dp[i][i] TO 1

            FOR cl FROM 2 TO n
                FOR i FROM 0 TO n - cl
                    SET j TO i + cl - 1
                    IF s[i] EQUALS s[j] AND cl EQUALS 2
                        SET dp[i][j] TO 2
                    ELSE IF s[i] EQUALS s[j]
                        SET dp[i][j] TO dp[i + 1][j - 1] + 2
                    ELSE
                        SET dp[i][j] TO MAXIMUM OF dp[i][j - 1] AND dp[i + 1][j]

            RETURN dp[0][n - 1]
        END FUNCTION

        RETURN LENGTH OF s MINUS longest_palindromic_subsequence(s)
    END FUNCTION
END CLASS
