CLASS Solution
    FUNCTION findNthDigit(n)
        SET length TO 1
        SET count TO 9
        SET start TO 1
        WHILE n > length * count
            SET n TO n MINUS length * count
            INCREMENT length BY 1
            MULTIPLY count BY 10
            MULTIPLY start BY 10
        END WHILE
        SET start TO start PLUS (n MINUS 1) DIVIDED BY length
        SET digit_index TO (n MINUS 1) MODULO length
        SET digit_string TO CONVERT start TO STRING
        RETURN INTEGER VALUE OF digit_string AT POSITION digit_index
    END FUNCTION
END CLASS
