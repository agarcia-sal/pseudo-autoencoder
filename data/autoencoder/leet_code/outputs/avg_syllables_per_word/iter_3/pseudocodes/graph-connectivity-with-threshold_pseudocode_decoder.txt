CLASS Solution
    FUNCTION areConnected(n, threshold, queries)
        IF threshold EQUALS 0
            RETURN list of True repeated LENGTH OF queries times
        
        SET parent TO list of integers FROM 0 TO n
        
        FUNCTION find(x)
            IF parent[x] NOT EQUAL TO x
                SET parent[x] TO find(parent[x])
            RETURN parent[x]
        END FUNCTION
        
        FUNCTION union(x, y)
            SET rootX TO find(x)
            SET rootY TO find(y)
            IF rootX NOT EQUAL TO rootY
                SET parent[rootY] TO rootX
        END FUNCTION
        
        FOR i FROM threshold + 1 TO n
            FOR j FROM 2 * i TO n STEP i
                CALL union(i, j)
        
        SET result TO empty list
        FOR each pair a, b IN queries
            APPEND (find(a) EQUALS find(b)) TO result
        
        RETURN result
    END FUNCTION
END CLASS
