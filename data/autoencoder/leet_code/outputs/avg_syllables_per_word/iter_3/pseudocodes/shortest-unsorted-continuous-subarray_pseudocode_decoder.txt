CLASS Solution
    FUNCTION findUnsortedSubarray(nums)
        SET sorted_nums TO SORTED VERSION OF nums
        SET start TO -1
        SET end TO -2
        FOR i FROM 0 TO LENGTH OF nums MINUS 1
            IF nums[i] NOT EQUAL TO sorted_nums[i]
                IF start EQUAL TO -1
                    SET start TO i
                END IF
                SET end TO i
            END IF
        END FOR
        RETURN end MINUS start PLUS 1
    END FUNCTION
END CLASS
