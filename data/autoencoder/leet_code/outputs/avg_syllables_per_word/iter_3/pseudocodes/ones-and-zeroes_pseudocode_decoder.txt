CLASS Solution
    FUNCTION findMaxForm(strs, m, n)
        SET dp TO 2D array of size (m+1) by (n+1) filled with 0

        FOR each string s IN strs
            SET count_0 TO number of '0' characters in s
            SET count_1 TO number of '1' characters in s

            FOR i FROM m DOWNTO count_0
                FOR j FROM n DOWNTO count_1
                    SET dp[i][j] TO maximum of dp[i][j] and dp[i - count_0][j - count_1] plus 1
                END FOR
            END FOR
        END FOR

        RETURN dp[m][n]
    END FUNCTION
END CLASS
