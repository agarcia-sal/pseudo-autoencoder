CLASS Solution
    FUNCTION findContestMatch(n)
        SET teams TO list of strings from 1 to n
        WHILE n > 1
            FOR i FROM 0 TO (n DIVIDED BY 2) MINUS 1
                SET teams[i] TO string combining teams[i] and teams[n MINUS 1 MINUS i] enclosed in parentheses
            SET n TO n DIVIDED BY 2
        RETURN teams[0]
    END FUNCTION
END CLASS
