CLASS Solution
    FUNCTION subarraySum(nums, k)
        SET cumulative_sum_freq TO DICTIONARY with key 0 and value 1
        SET current_sum TO 0
        SET count TO 0
        FOR each num IN nums
            SET current_sum TO current_sum + num
            IF (current_sum - k) IN cumulative_sum_freq THEN
                SET count TO count + cumulative_sum_freq[current_sum - k]
            END IF
            IF current_sum IN cumulative_sum_freq THEN
                INCREMENT cumulative_sum_freq[current_sum] BY 1
            ELSE
                SET cumulative_sum_freq[current_sum] TO 1
            END IF
        END FOR
        RETURN count
    END FUNCTION
END CLASS
