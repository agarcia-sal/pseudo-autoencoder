CLASS ListNode
    FUNCTION __init__(val = 0, next = None)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values)
    IF values IS EMPTY
        RETURN None
    END IF
    SET head TO new ListNode(values at index 0)
    SET p TO head
    FOR each val IN values starting from index 1
        SET node TO new ListNode(val)
        SET p.next TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS None AND p2 IS None
        RETURN True
    END IF
    IF p1 IS None OR p2 IS None
        RETURN False
    END IF
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)
END FUNCTION

CLASS Solution
    FUNCTION deleteDuplicates(head)
        SET dummy TO new ListNode(0, head)
        SET prev TO dummy
        
        WHILE head IS NOT None
            IF head.next EXISTS AND head.val EQUALS head.next.val
                WHILE head.next EXISTS AND head.val EQUALS head.next.val
                    SET head TO head.next
                END WHILE
                SET prev.next TO head.next
            ELSE
                SET prev TO prev.next
            END IF
            SET head TO head.next
        END WHILE
        
        RETURN dummy.next
    END FUNCTION
END CLASS
