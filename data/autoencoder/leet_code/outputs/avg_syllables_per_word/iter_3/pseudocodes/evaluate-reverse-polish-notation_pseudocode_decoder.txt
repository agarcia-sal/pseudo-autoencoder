CLASS Solution
    FUNCTION evalRPN(tokens: List of String) RETURNS Integer
        SET stack TO empty list
        FOR each token IN tokens
            IF token IS IN ["+", "-", "*", "/"] THEN
                SET b TO POP from stack
                SET a TO POP from stack
                IF token IS "+" THEN
                    APPEND (a + b) TO stack
                ELSE IF token IS "-" THEN
                    APPEND (a - b) TO stack
                ELSE IF token IS "*" THEN
                    APPEND (a * b) TO stack
                ELSE IF token IS "/" THEN
                    SET result TO a DIVIDED BY b, truncated towards zero
                    APPEND result TO stack
            ELSE
                APPEND integer value of token TO stack
        RETURN stack[0]
    END FUNCTION
END CLASS
