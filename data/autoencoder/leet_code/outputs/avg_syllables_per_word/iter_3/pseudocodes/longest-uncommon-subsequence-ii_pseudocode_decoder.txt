CLASS Solution
    FUNCTION findLUSlength(strs)
        FUNCTION is_subsequence(s, t)
            SET iterator_it TO ITERATOR OVER t
            RETURN ALL characters c IN s ARE IN iterator_it IN ORDER
        END FUNCTION

        SORT strs BY length DESCENDING AND lexicographically ASCENDING

        FOR each index i AND word IN strs
            SET unique TO True
            FOR each index j AND other_word IN strs
                IF i IS NOT EQUAL TO j AND is_subsequence(word, other_word) IS True
                    SET unique TO False
                    BREAK inner loop
            IF unique IS True
                RETURN LENGTH OF word

        RETURN -1
    END FUNCTION
END CLASS
