CLASS Solution
    FUNCTION removeDuplicateLetters(s)
        SET last_occurrence TO EMPTY DICTIONARY
        FOR each index, character IN s
            SET last_occurrence[character] TO index
        
        SET stack TO EMPTY LIST
        SET in_stack TO EMPTY SET
        
        FOR each index, character IN s
            IF character IN in_stack THEN
                CONTINUE
            
            WHILE stack IS NOT EMPTY AND character IS LESS THAN stack LAST ELEMENT AND index IS LESS THAN last_occurrence OF stack LAST ELEMENT
                SET removed_char TO POP stack LAST ELEMENT
                REMOVE removed_char FROM in_stack
            
            APPEND character TO stack
            ADD character TO in_stack
        
        RETURN CONCATENATION OF ELEMENTS IN stack
    END FUNCTION
END CLASS
