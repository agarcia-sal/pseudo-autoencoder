CLASS Solution
    FUNCTION findSubstring(s, words)
        IF s IS EMPTY OR words IS EMPTY OR words[0] IS EMPTY THEN
            RETURN empty list
        END IF

        SET word_length TO LENGTH OF words[0]
        SET num_words TO LENGTH OF words
        SET total_length TO word_length MULTIPLIED BY num_words
        SET word_count TO COUNTER OF words
        SET result TO empty list

        FOR i FROM 0 TO LENGTH OF s MINUS total_length INCLUSIVE DO
            SET seen TO EMPTY COUNTER
            SET j TO i
            WHILE j LESS THAN i PLUS total_length DO
                SET word TO SUBSTRING OF s FROM j TO j PLUS word_length
                INCREMENT COUNT OF word IN seen BY 1
                IF COUNT OF word IN seen GREATER THAN COUNT OF word IN word_count THEN
                    BREAK LOOP
                END IF
                INCREMENT j BY word_length
            END WHILE
            IF j EQUALS i PLUS total_length THEN
                APPEND i TO result
            END IF
        END FOR

        RETURN result
    END FUNCTION
END CLASS
