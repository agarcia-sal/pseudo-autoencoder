CLASS Solution
    FUNCTION generateAbbreviations(word)
        FUNCTION backtrack(index, path, count)
            IF index EQUALS LENGTH OF word THEN
                IF count GREATER THAN 0 THEN
                    APPEND STRING OF count TO path
                END IF
                APPEND JOINED path TO result
            ELSE
                IF count GREATER THAN 0 THEN
                    CALL backtrack WITH (index + 1, path APPENDED WITH STRING OF count AND word AT index, 0)
                ELSE
                    CALL backtrack WITH (index + 1, path APPENDED WITH word AT index, 0)
                END IF
                CALL backtrack WITH (index + 1, path, count + 1)
            END IF
        END FUNCTION

        SET result TO EMPTY LIST
        CALL backtrack WITH (0, EMPTY LIST, 0)
        RETURN result
    END FUNCTION
END CLASS
