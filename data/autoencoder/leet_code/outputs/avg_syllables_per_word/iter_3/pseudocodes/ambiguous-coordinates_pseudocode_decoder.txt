CLASS Solution
    FUNCTION ambiguousCoordinates(s)
        FUNCTION make(frag)
            SET n TO LENGTH OF frag
            FOR d FROM 1 TO n
                SET left TO SUBSTRING OF frag FROM START TO d
                SET right TO SUBSTRING OF frag FROM d TO END
                IF (left DOES NOT START WITH '0' OR left IS '0') AND (right DOES NOT END WITH '0')
                    IF d IS NOT EQUAL TO n
                        YIELD CONCATENATION OF left, '.', right
                    ELSE
                        YIELD CONCATENATION OF left, right
        SET s TO SUBSTRING OF s FROM 1 TO LENGTH OF s MINUS 1
        SET result TO EMPTY LIST
        FOR i FROM 1 TO LENGTH OF s MINUS 1
            FOR each a IN make(SUBSTRING OF s FROM START TO i)
                FOR each b IN make(SUBSTRING OF s FROM i TO END)
                    APPEND CONCATENATION OF '(', a, ', ', b, ')' TO result
        RETURN result
    END FUNCTION
END CLASS
