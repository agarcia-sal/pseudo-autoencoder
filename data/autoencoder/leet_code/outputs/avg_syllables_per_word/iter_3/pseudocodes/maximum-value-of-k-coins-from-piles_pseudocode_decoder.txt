CLASS Solution
    FUNCTION maxValueOfCoins(piles: List of List of integers, k: integer) RETURNS integer
        SET n TO LENGTH OF piles
        SET dp TO 2D list of size (n + 1) by (k + 1) filled with 0

        FOR i FROM 1 TO n
            FOR j FROM 1 TO k
                SET current_pile TO piles[i - 1]
                SET current_value TO 0
                FOR l FROM 0 TO MINIMUM OF j AND LENGTH OF current_pile
                    IF l > 0
                        INCREMENT current_value BY current_pile[l - 1]
                    SET dp[i][j] TO MAXIMUM OF dp[i][j] AND (dp[i - 1][j - l] + current_value)
        RETURN dp[n][k]
    END FUNCTION
END CLASS
