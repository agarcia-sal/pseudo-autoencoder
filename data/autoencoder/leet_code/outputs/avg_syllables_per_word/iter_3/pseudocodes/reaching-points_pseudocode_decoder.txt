CLASS Solution
    FUNCTION reachingPoints(sx, sy, tx, ty)
        WHILE tx IS GREATER THAN OR EQUAL TO sx AND ty IS GREATER THAN OR EQUAL TO sy
            IF tx EQUALS sx AND ty EQUALS sy
                RETURN True
            IF tx IS GREATER THAN ty
                IF ty IS GREATER THAN sy
                    SET tx TO tx MODULO ty
                ELSE
                    RETURN (tx MINUS sx) MODULO ty EQUALS 0
            ELSE
                IF tx IS GREATER THAN sx
                    SET ty TO ty MODULO tx
                ELSE
                    RETURN (ty MINUS sy) MODULO tx EQUALS 0
        RETURN False
    END FUNCTION
END CLASS
