CLASS Solution
    FUNCTION longestLine(mat)
        IF mat IS EMPTY OR mat[0] IS EMPTY THEN
            RETURN 0
        SET m TO LENGTH OF mat
        SET n TO LENGTH OF mat[0]
        SET directions TO [(0, 1), (1, 0), (1, 1), (1, -1)]
        SET max_length TO 0

        FUNCTION check(x, y, dx, dy)
            SET length TO 0
            WHILE x IS WITHIN 0 AND m - 1 AND y IS WITHIN 0 AND n - 1 AND mat[x][y] EQUALS 1 DO
                INCREMENT length BY 1
                INCREMENT x BY dx
                INCREMENT y BY dy
            RETURN length
        END FUNCTION

        FOR i FROM 0 TO m - 1 DO
            FOR j FROM 0 TO n - 1 DO
                IF mat[i][j] EQUALS 1 THEN
                    FOR EACH (dx, dy) IN directions DO
                        IF (i - dx IS LESS THAN 0) OR (j - dy IS LESS THAN 0) OR (j - dy IS GREATER OR EQUAL TO n) OR (mat[i - dx][j - dy] EQUALS 0) THEN
                            SET max_length TO MAXIMUM OF max_length AND check(i, j, dx, dy)
        RETURN max_length
    END FUNCTION
END CLASS
