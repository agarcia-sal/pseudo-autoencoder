CLASS Solution
    FUNCTION replaceNonCoprimes(nums)
        SET stack TO empty list
        FOR each num IN nums
            WHILE stack is not empty
                SET g TO gcd of last element in stack and num
                IF g equals 1
                    BREAK from WHILE loop
                SET num TO product of last element in stack and num DIVIDED BY g
                REMOVE last element from stack
            END WHILE
            APPEND num TO stack
        END FOR
        RETURN stack
    END FUNCTION
END CLASS
