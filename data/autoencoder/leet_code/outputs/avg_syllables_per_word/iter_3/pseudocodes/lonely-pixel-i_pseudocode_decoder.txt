CLASS Solution
    FUNCTION findLonelyPixel(picture)
        IF picture IS EMPTY OR picture[0] IS EMPTY
            RETURN 0
        END IF

        SET m TO LENGTH OF picture
        SET n TO LENGTH OF picture[0]
        SET row_count TO LIST OF ZEROS OF LENGTH m
        SET col_count TO LIST OF ZEROS OF LENGTH n

        FOR i FROM 0 TO m MINUS 1
            FOR j FROM 0 TO n MINUS 1
                IF picture[i][j] EQUALS 'B'
                    INCREMENT row_count[i] BY 1
                    INCREMENT col_count[j] BY 1
                END IF
            END FOR
        END FOR

        SET lonely_pixel_count TO 0
        FOR i FROM 0 TO m MINUS 1
            FOR j FROM 0 TO n MINUS 1
                IF picture[i][j] EQUALS 'B' AND row_count[i] EQUALS 1 AND col_count[j] EQUALS 1
                    INCREMENT lonely_pixel_count BY 1
                END IF
            END FOR
        END FOR

        RETURN lonely_pixel_count
    END FUNCTION
END CLASS
