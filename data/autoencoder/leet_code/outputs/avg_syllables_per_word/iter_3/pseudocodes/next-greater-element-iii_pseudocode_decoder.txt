CLASS Solution
    FUNCTION nextGreaterElement(n)
        SET digits TO list of characters in string representation of n
        SET length TO LENGTH OF digits

        SET i TO length MINUS 2
        WHILE i GREATER THAN OR EQUAL TO 0 AND digits[i] GREATER THAN OR EQUAL TO digits[i PLUS 1]
            DECREMENT i BY 1
        END WHILE

        IF i EQUALS -1
            RETURN -1
        END IF

        SET j TO length MINUS 1
        WHILE digits[j] LESS THAN OR EQUAL TO digits[i]
            DECREMENT j BY 1
        END WHILE

        SWAP digits[i] AND digits[j]

        SET digits TO CONCATENATION OF digits UP TO i PLUS 1 AND REVERSE OF digits FROM i PLUS 1 TO END

        SET result TO INTEGER CONVERSION OF concatenated digits

        IF result GREATER THAN 2 TO THE POWER OF 31 MINUS 1
            RETURN -1
        END IF

        RETURN result
    END FUNCTION
END CLASS
