CLASS Solution
    FUNCTION find132pattern(nums)
        SET second TO negative infinity
        SET stack TO empty list

        FOR each num IN nums in reverse order
            IF num < second THEN
                RETURN True
            END IF
            WHILE stack is not empty AND last element of stack < num
                SET second TO remove and return last element of stack
            END WHILE
            APPEND num TO stack
        END FOR

        RETURN False
    END FUNCTION
END CLASS
