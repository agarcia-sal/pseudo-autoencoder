CLASS Solution
    FUNCTION arrayNesting(nums)
        SET visited TO list of False values with length of nums
        SET max_length TO 0
        FOR i FROM 0 TO LENGTH OF nums MINUS 1
            IF visited at i IS False
                SET current_length TO 0
                SET k TO i
                WHILE visited at k IS False
                    SET visited at k TO True
                    SET k TO nums at k
                    INCREMENT current_length BY 1
                IF current_length GREATER THAN max_length
                    SET max_length TO current_length
        RETURN max_length
    END FUNCTION
END CLASS
