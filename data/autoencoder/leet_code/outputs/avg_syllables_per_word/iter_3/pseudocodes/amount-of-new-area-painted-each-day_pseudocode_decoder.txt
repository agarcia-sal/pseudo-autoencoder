CLASS Solution
    FUNCTION amountPainted(paint)
        SET max_end TO maximum of all end values in paint
        SET painted TO list of length max_end plus one, FILLED with -1
        SET worklog TO empty list
        
        FOR each start, end IN paint
            SET current_start TO start
            SET new_paint TO 0
            
            WHILE current_start LESS THAN end
                IF painted at current_start EQUALS -1
                    SET painted at current_start TO end
                    INCREMENT new_paint BY 1
                    INCREMENT current_start BY 1
                ELSE
                    SET current_start TO painted at current_start
            
            APPEND new_paint TO worklog
        
        RETURN worklog
    END FUNCTION
END CLASS
