CLASS Solution
    FUNCTION findLongestWord(s, dictionary)
        FUNCTION is_subsequence(word, s)
            SET iterator TO ITERATOR OVER s
            FOR each char IN word
                IF char NOT IN iterator
                    RETURN False
            RETURN True
        END FUNCTION

        SORT dictionary BY descending length AND ascending lexicographical order

        FOR each word IN dictionary
            IF is_subsequence(word, s)
                RETURN word

        RETURN empty string
    END FUNCTION
END CLASS
