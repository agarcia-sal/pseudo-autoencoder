CLASS Solution
    FUNCTION numberOfWays(corridor)
        SET MOD TO 10 TO THE POWER OF 9 PLUS 7
        SET seats TO EMPTY LIST
        FOR index, character IN ENUMERATE corridor
            IF character EQUALS 'S'
                APPEND index TO seats
        IF LENGTH OF seats MODULO 2 NOT EQUAL TO 0 OR LENGTH OF seats LESS THAN 2
            RETURN 0
        SET ways TO 1
        FOR i FROM 2 TO LENGTH OF seats STEP 2
            SET ways TO (ways MULTIPLIED BY (seats[i] MINUS seats[i MINUS 1])) MODULO MOD
        RETURN ways
    END FUNCTION
END CLASS
