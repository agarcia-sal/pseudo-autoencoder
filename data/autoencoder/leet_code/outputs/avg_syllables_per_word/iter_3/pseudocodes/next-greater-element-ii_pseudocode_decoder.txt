CLASS Solution
    FUNCTION nextGreaterElements(nums)
        SET n TO LENGTH OF nums
        SET result TO LIST OF -1 REPEATED n TIMES
        SET stack TO EMPTY LIST

        FOR i FROM 0 TO 2 * n - 1
            SET index TO i MODULO n
            WHILE stack IS NOT EMPTY AND nums AT TOP OF stack IS LESS THAN nums[index]
                SET popped_index TO POP FROM stack
                SET result[popped_index] TO nums[index]
            IF i IS LESS THAN n
                APPEND index TO stack

        RETURN result
    END FUNCTION
END CLASS
