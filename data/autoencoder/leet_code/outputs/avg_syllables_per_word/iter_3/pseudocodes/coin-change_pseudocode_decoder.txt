CLASS Solution
    FUNCTION coinChange(coins, amount)
        SET dp TO list of size amount + 1 filled with amount + 1
        SET dp[0] TO 0
        
        FOR a FROM 1 TO amount
            FOR each coin IN coins
                IF a MINUS coin IS GREATER THAN OR EQUAL TO 0
                    SET dp[a] TO minimum of dp[a] AND dp[a MINUS coin] PLUS 1
        END FOR
        END FOR
        
        IF dp[amount] IS EQUAL TO amount + 1
            RETURN -1
        ELSE
            RETURN dp[amount]
        END IF
    END FUNCTION
END CLASS
