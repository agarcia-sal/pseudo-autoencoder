CLASS Solution
    FUNCTION dailyTemperatures(temperatures: List of Integer) RETURNS List of Integer
        SET n TO LENGTH OF temperatures
        SET answer TO LIST OF 0 WITH LENGTH n
        SET stack TO EMPTY LIST

        FOR i FROM 0 TO n MINUS 1
            WHILE stack IS NOT EMPTY AND temperatures[i] GREATER THAN temperatures[TOP ELEMENT OF stack]
                SET prev_index TO POP FROM stack
                SET answer[prev_index] TO i MINUS prev_index
            END WHILE
            APPEND i TO stack
        END FOR

        RETURN answer
    END FUNCTION
END CLASS
