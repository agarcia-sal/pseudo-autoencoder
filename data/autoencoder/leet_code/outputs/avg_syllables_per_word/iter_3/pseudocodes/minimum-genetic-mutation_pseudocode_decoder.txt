CLASS Solution
    FUNCTION minMutation(start, end, bank)
        SET s TO set of bank
        SET q TO deque containing tuple (start, 0)
        SET mp TO dictionary with keys 'A', 'T', 'C', 'G' mapping to strings 'TCG', 'ACG', 'ATG', 'ATC' respectively
        WHILE q is not empty
            REMOVE first element FROM q and UNPACK into t and step
            IF t EQUALS end
                RETURN step
            FOR each index i and character v IN t
                FOR each character j IN mp[v]
                    SET next TO string formed by replacing character at position i in t with j
                    IF next IN s
                        APPEND tuple (next, step + 1) TO q
                        REMOVE next FROM s
        RETURN -1
END CLASS
