CLASS Solution
    FUNCTION kMirror(k, n)
        FUNCTION is_palindrome(s)
            RETURN s EQUALS REVERSE OF s
        END FUNCTION

        FUNCTION to_base_k(num, k)
            IF num EQUALS 0 THEN
                RETURN "0"
            END IF
            SET digits TO EMPTY LIST
            WHILE num IS NOT 0
                APPEND (num MODULO k) TO digits
                SET num TO num DIVIDED BY k
            END WHILE
            RETURN CONCATENATION OF STRING REPRESENTATIONS OF digits IN REVERSE ORDER
        END FUNCTION

        FUNCTION generate_palindromes(length)
            IF length EQUALS 1 THEN
                FOR i FROM 1 TO 9
                    YIELD STRING OF i
                END FOR
            ELSE
                SET start TO 10 RAISED TO THE POWER OF ((length MINUS 1) DIVIDED BY 2)
                SET end TO 10 RAISED TO THE POWER OF ((length PLUS 1) DIVIDED BY 2)
                FOR half FROM start TO end MINUS 1
                    SET half_str TO STRING OF half
                    IF length MODULO 2 EQUALS 0 THEN
                        SET full TO half_str CONCATENATED WITH REVERSE OF half_str
                    ELSE
                        SET full TO half_str CONCATENATED WITH REVERSE OF half_str EXCLUDING LAST CHARACTER
                    END IF
                    YIELD full
                END FOR
            END IF
        END FUNCTION

        SET k_mirror_numbers TO EMPTY LIST
        SET length TO 1
        WHILE LENGTH OF k_mirror_numbers LESS THAN n
            FOR pal IN generate_palindromes(length)
                SET num TO INTEGER VALUE OF pal
                IF is_palindrome(to_base_k(num, k)) THEN
                    APPEND num TO k_mirror_numbers
                    IF LENGTH OF k_mirror_numbers EQUALS n THEN
                        BREAK FOR LOOP
                    END IF
                END IF
            END FOR
            INCREMENT length BY 1
        END WHILE

        RETURN SUM OF k_mirror_numbers
    END FUNCTION
END CLASS
