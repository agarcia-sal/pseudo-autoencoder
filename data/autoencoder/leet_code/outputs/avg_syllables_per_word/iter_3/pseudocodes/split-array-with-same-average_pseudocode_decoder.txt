CLASS Solution
    FUNCTION splitArraySameAverage(nums)
        SET n TO LENGTH OF nums
        SET total_sum TO SUM OF nums

        IF n EQUALS 1 THEN
            RETURN False
        END IF

        SET dp TO LIST OF SETS WITH SIZE (n DIVIDED BY 2 PLUS 1)
        ADD 0 TO dp[0]

        FOR each num IN nums
            FOR size FROM (n DIVIDED BY 2) DOWNTO 1
                FOR each s IN dp[size MINUS 1]
                    SET new_sum TO s PLUS num
                    IF new_sum MULTIPLIED BY n EQUALS size MULTIPLIED BY total_sum THEN
                        RETURN True
                    END IF
                    ADD new_sum TO dp[size]
                END FOR
            END FOR
        END FOR

        RETURN False
    END FUNCTION
END CLASS
