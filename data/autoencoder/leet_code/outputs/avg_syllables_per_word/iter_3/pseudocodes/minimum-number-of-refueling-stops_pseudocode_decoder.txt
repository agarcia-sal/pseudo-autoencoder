CLASS Solution
    FUNCTION minRefuelStops(target, startFuel, stations)
        SET max_heap TO empty list
        SET current_fuel TO startFuel
        SET stops TO 0
        APPEND [target, 0] TO stations

        FOR EACH position, fuel IN stations
            WHILE max_heap IS NOT empty AND current_fuel < position
                SET current_fuel TO current_fuel MINUS REMOVE largest fuel FROM max_heap
                INCREMENT stops BY 1
            END WHILE

            IF current_fuel < position
                RETURN -1
            END IF

            ADD negative of fuel TO max_heap
        END FOR

        RETURN stops
    END FUNCTION
END CLASS
