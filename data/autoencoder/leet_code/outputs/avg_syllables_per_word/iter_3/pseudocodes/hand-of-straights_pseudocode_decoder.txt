CLASS Solution
    FUNCTION isNStraightHand(hand, groupSize)
        IF LENGTH OF hand MODULO groupSize IS NOT 0 THEN
            RETURN False
        END IF

        SET count TO frequency count of elements in hand
        SET unique_cards TO sorted keys of count

        FOR each card IN unique_cards
            IF count[card] > 0 THEN
                SET needed TO count[card]
                FOR i FROM 0 TO groupSize - 1
                    IF count[card + i] < needed THEN
                        RETURN False
                    END IF
                    DECREMENT count[card + i] BY needed
                END FOR
            END IF
        END FOR

        RETURN True
    END FUNCTION
END CLASS
