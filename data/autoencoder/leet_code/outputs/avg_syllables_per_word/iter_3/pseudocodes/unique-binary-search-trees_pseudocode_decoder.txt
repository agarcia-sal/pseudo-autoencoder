CLASS Solution
    FUNCTION numTrees(n)
        SET dp TO list of zeros with length n + 1
        SET dp[0] TO 1
        SET dp[1] TO 1
        FOR i FROM 2 TO n
            FOR j FROM 1 TO i
                INCREMENT dp[i] BY dp[j - 1] MULTIPLIED BY dp[i - j]
        RETURN dp[n]
    END FUNCTION
END CLASS
