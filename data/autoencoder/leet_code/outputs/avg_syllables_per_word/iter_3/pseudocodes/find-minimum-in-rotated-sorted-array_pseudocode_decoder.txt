CLASS Solution
    FUNCTION findMin(nums: List of int) RETURNS int
        SET left TO 0
        SET right TO LENGTH OF nums MINUS 1
        IF nums[left] LESS THAN nums[right]
            RETURN nums[left]
        WHILE left LESS THAN right
            SET mid TO (left PLUS right) DIVIDED BY 2
            IF nums[mid] GREATER THAN nums[right]
                SET left TO mid PLUS 1
            ELSE
                SET right TO mid
        RETURN nums[left]
    END FUNCTION
END CLASS
