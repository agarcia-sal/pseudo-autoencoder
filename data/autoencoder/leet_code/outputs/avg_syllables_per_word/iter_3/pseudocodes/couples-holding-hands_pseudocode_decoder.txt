CLASS Solution
    FUNCTION minSwapsCouples(row)
        SET position TO dictionary mapping each person TO their index in row
        SET swaps TO 0
        FOR i FROM 0 TO LENGTH OF row STEP 2
            SET correct_pair TO row[i] XOR 1
            IF row[i + 1] IS NOT correct_pair
                SET swap_idx TO position[correct_pair]
                SWAP row[i + 1] WITH row[swap_idx]
                SET position[row[swap_idx]] TO swap_idx
                SET position[row[i + 1]] TO i + 1
                INCREMENT swaps BY 1
        RETURN swaps
    END FUNCTION
END CLASS
