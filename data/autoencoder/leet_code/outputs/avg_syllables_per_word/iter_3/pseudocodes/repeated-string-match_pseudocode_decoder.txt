CLASS Solution
    FUNCTION repeatedStringMatch(a, b)
        SET min_repeats TO ceiling of (length of b DIVIDED BY length of a)
        IF b IS SUBSTRING OF a repeated min_repeats times
            RETURN min_repeats
        IF b IS SUBSTRING OF a repeated (min_repeats + 1) times
            RETURN min_repeats + 1
        RETURN -1
    END FUNCTION
END CLASS
