CLASS Solution
    FUNCTION monotoneIncreasingDigits(n)
        SET digits TO list of characters representing n
        SET length TO number of elements in digits
        SET marker TO length
        FOR i FROM length MINUS 1 DOWN TO 1
            IF digits at position i IS LESS THAN digits at position i MINUS 1
                SET digits at position i MINUS 1 TO character representing (integer value of digits at i MINUS 1 MINUS 1)
                SET marker TO i
        FOR i FROM marker TO length MINUS 1
            SET digits at position i TO character '9'
        SET result TO integer represented by joining all characters in digits
        RETURN result
    END FUNCTION
END CLASS
