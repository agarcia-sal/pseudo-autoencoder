CLASS Solution
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)
        SET min_times TO list of size 15 filled with infinity
        FOR each (f, r) IN tires
            SET total_time TO 0
            FOR i FROM 0 TO 14
                SET lap_time TO f MULTIPLIED BY (r RAISED TO i)
                IF lap_time GREATER THAN changeTime PLUS f
                    BREAK loop
                INCREMENT total_time BY lap_time
                SET min_times[i] TO minimum OF min_times[i] AND total_time
        
        SET dp TO list of size (numLaps PLUS 1) filled with infinity
        SET dp[0] TO 0
        
        FOR i FROM 1 TO numLaps
            FOR j FROM 0 TO minimum OF (i MINUS 1) AND 14
                SET dp[i] TO minimum OF dp[i] AND dp[i MINUS j MINUS 1] PLUS min_times[j] PLUS changeTime
        
        RETURN dp[numLaps] MINUS changeTime
    END FUNCTION
END CLASS
