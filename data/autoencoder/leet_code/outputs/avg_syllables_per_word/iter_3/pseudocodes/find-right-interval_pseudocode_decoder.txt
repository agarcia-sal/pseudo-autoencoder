CLASS Solution
    FUNCTION findRightInterval(intervals)
        SET start_times TO SORTED list of (interval start, original index) FOR each interval with index IN intervals
        SET result TO empty list
        FOR EACH interval IN intervals
            SET end TO interval end value
            SET idx TO position found by binary searching for smallest start time >= end IN start_times
            IF idx is less than LENGTH OF start_times
                APPEND start_times at idx's original index TO result
            ELSE
                APPEND -1 TO result
        RETURN result
    END FUNCTION
END CLASS
