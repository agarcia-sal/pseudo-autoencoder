CLASS Solution
    FUNCTION minimumObstacles(grid)
        SET m TO LENGTH OF grid
        SET n TO LENGTH OF grid[0]
        SET directions TO LIST OF (0, 1), (1, 0), (0, -1), (-1, 0)
        SET min_heap TO LIST CONTAINING (0, 0, 0)  # cost, x, y
        SET visited TO SET CONTAINING (0, 0)
        
        WHILE min_heap IS NOT EMPTY
            REMOVE smallest element FROM min_heap AND SET TO (cost, x, y)
            
            IF (x, y) IS EQUAL TO (m - 1, n - 1)
                RETURN cost
            
            FOR EACH (dx, dy) IN directions
                SET nx TO x PLUS dx
                SET ny TO y PLUS dy
                IF nx IS GREATER OR EQUAL TO 0 AND LESS THAN m AND ny IS GREATER OR EQUAL TO 0 AND LESS THAN n AND (nx, ny) NOT IN visited
                    ADD (nx, ny) TO visited
                    SET new_cost TO cost PLUS grid[nx][ny]
                    ADD (new_cost, nx, ny) TO min_heap
        
        RETURN -1
END CLASS
