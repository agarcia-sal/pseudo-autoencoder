CLASS Solution
    FUNCTION findAllPeople(n, meetings, firstPerson)
        SET know_secret TO SET containing 0 and firstPerson
        SORT meetings BY the third element (time) IN ASCENDING ORDER
        SET i TO 0
        WHILE i LESS THAN LENGTH OF meetings
            SET current_time TO meetings[i][2]
            SET graph TO a new empty map WITH default list
            WHILE i LESS THAN LENGTH OF meetings AND meetings[i][2] EQUALS current_time
                SET x TO meetings[i][0]
                SET y TO meetings[i][1]
                APPEND y TO graph[x]
                APPEND x TO graph[y]
                INCREMENT i BY 1
            END WHILE

            SET queue TO a new empty deque
            FOR each person IN know_secret
                IF person IS IN graph
                    APPEND person TO queue
            END FOR

            SET visited TO a new empty set
            WHILE queue IS NOT empty
                REMOVE and GET first person FROM queue
                IF person IS IN visited
                    CONTINUE to next iteration
                ADD person TO visited
                ADD person TO know_secret
                FOR each neighbor IN graph[person]
                    IF neighbor NOT IN visited
                        APPEND neighbor TO queue
                END FOR
            END WHILE
        END WHILE
        RETURN list CONVERTED from know_secret
    END FUNCTION
END CLASS
