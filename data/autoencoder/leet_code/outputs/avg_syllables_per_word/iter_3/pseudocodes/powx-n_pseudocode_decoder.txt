CLASS Solution
    FUNCTION myPow(x: float, n: int) RETURNS float
        FUNCTION qpow(a: float, n: int) RETURNS float
            SET ans TO 1
            WHILE n IS NOT 0
                IF n MODULO 2 IS 1
                    SET ans TO ans MULTIPLIED BY a
                SET a TO a MULTIPLIED BY a
                SET n TO n DIVIDED BY 2 (integer division)
            RETURN ans
        END FUNCTION

        IF n IS GREATER THAN OR EQUAL TO 0
            RETURN qpow(x, n)
        ELSE
            RETURN 1 DIVIDED BY qpow(x, NEGATION OF n)
    END FUNCTION
END CLASS
