CLASS Solution
    FUNCTION multiply(mat1, mat2)
        SET m TO LENGTH OF mat1
        SET k TO LENGTH OF mat1[0]
        SET k TO LENGTH OF mat2
        SET n TO LENGTH OF mat2[0]
        
        SET result TO A LIST OF m ITEMS, EACH A LIST OF n ZEROS
        
        FOR i FROM 0 TO m MINUS 1
            FOR j FROM 0 TO n MINUS 1
                FOR p FROM 0 TO k MINUS 1
                    INCREMENT result[i][j] BY mat1[i][p] MULTIPLIED BY mat2[p][j]
        
        RETURN result
    END FUNCTION
END CLASS
