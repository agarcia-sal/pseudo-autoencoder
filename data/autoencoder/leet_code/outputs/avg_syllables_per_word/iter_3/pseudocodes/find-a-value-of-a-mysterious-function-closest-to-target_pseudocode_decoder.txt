CLASS Solution
    FUNCTION closestToTarget(arr, target)
        SET seen TO empty set
        SET min_diff TO infinity
        FOR each num IN arr
            SET seen TO union of set of bitwise AND of num and each element in seen, and set containing num
            FOR each val IN seen
                IF absolute value of val minus target IS LESS THAN min_diff
                    SET min_diff TO absolute value of val minus target
        RETURN min_diff
    END FUNCTION
END CLASS
