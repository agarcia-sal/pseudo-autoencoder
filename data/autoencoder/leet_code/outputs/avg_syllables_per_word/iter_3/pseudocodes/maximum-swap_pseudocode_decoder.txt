CLASS Solution
    FUNCTION maximumSwap(num)
        SET digits TO list of characters in string representation of num
        SET last TO empty dictionary
        FOR each index i AND digit d IN enumeration of digits
            SET last[int(d)] TO i
        FOR each index i AND digit d IN enumeration of digits
            FOR d2 FROM 9 DOWN TO int(d) + 1 STEP -1
                IF last CONTAINS d2 AND last[d2] > i THEN
                    SWAP digits[i] WITH digits[last[d2]]
                    RETURN integer formed by joining digits
        RETURN num
    END FUNCTION
END CLASS
