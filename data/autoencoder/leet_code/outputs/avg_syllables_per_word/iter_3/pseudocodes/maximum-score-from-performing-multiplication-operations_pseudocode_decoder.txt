CLASS Solution
    FUNCTION maximumScore(nums, multipliers)
        SET m TO LENGTH OF multipliers
        SET n TO LENGTH OF nums
        SET dp TO LIST OF 0s WITH SIZE m + 1
        FOR i FROM m - 1 DOWNTO 0
            SET new_dp TO LIST OF 0s WITH SIZE m + 1
            FOR left FROM 0 TO i
                SET right TO n - 1 - (i - left)
                SET mul TO multipliers[i]
                SET new_dp[left] TO MAXIMUM OF (mul * nums[left] + dp[left + 1], mul * nums[right] + dp[left])
            SET dp TO new_dp
        RETURN dp[0]
    END FUNCTION
END CLASS
