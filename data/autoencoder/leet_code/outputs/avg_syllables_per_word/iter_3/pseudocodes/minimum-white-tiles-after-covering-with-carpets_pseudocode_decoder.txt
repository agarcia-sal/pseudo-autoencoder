CLASS Solution
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)
        SET n TO LENGTH OF floor
        SET dp TO 2D LIST OF SIZE (n + 1) BY (numCarpets + 1) FILLED WITH 0

        FOR i FROM 1 TO n
            SET dp[i][0] TO dp[i - 1][0] PLUS 1 IF floor CHARACTER AT (i - 1) IS '1' ELSE 0

        FOR j FROM 1 TO numCarpets
            FOR i FROM 1 TO n
                SET dp[i][j] TO dp[i - 1][j] PLUS 1 IF floor CHARACTER AT (i - 1) IS '1' ELSE 0
                SET dp[i][j] TO MINIMUM OF dp[i][j] AND dp[MAXIMUM OF 0 AND (i - carpetLen)][j - 1]

        RETURN dp[n][numCarpets]
    END FUNCTION
END CLASS
