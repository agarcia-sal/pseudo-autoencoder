CLASS Solution
    FUNCTION strobogrammaticInRange(low, high)
        FUNCTION generate_strobogrammatic(n, final_length)
            IF n EQUALS 0 THEN
                RETURN list containing empty string
            IF n EQUALS 1 THEN
                RETURN list containing "0", "1", "8"
            SET middles TO generate_strobogrammatic(n MINUS 2, final_length)
            SET result TO empty list
            FOR each middle IN middles DO
                IF n NOT EQUAL final_length THEN
                    APPEND "0" + middle + "0" TO result
                APPEND "1" + middle + "1" TO result
                APPEND "6" + middle + "9" TO result
                APPEND "8" + middle + "8" TO result
                APPEND "9" + middle + "6" TO result
            RETURN result
        END FUNCTION

        FUNCTION count_strobogrammatic(low, high)
            SET count TO 0
            FOR length FROM LENGTH OF low TO LENGTH OF high DO
                FOR each num IN generate_strobogrammatic(length, length) DO
                    IF (length EQUALS LENGTH OF low AND num LESS THAN low) OR (length EQUALS LENGTH OF high AND num GREATER THAN high) THEN
                        CONTINUE to next iteration
                    INCREMENT count BY 1
            RETURN count
        END FUNCTION

        RETURN count_strobogrammatic(low, high)
    END FUNCTION
END CLASS
