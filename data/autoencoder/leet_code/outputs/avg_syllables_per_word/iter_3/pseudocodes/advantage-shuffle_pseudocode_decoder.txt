CLASS Solution
    FUNCTION advantageCount(nums1, nums2)
        SORT nums1
        SET nums2_with_indices TO SORTED list of (value, index) pairs from nums2 in descending order by value
        SET result TO list of zeros with length equal to length of nums1
        SET left TO 0
        SET right TO length of nums1 MINUS 1
        FOR each (value, index) IN nums2_with_indices
            IF nums1[right] > value THEN
                SET result[index] TO nums1[right]
                DECREMENT right BY 1
            ELSE
                SET result[index] TO nums1[left]
                INCREMENT left BY 1
        RETURN result
    END FUNCTION
END CLASS
