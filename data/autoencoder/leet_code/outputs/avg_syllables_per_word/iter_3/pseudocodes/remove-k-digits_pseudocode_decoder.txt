CLASS Solution
    FUNCTION removeKdigits(num, k)
        SET stack TO empty list
        FOR each digit IN num
            WHILE k > 0 AND stack IS NOT empty AND last element of stack > digit
                REMOVE last element FROM stack
                DECREMENT k BY 1
            END WHILE
            APPEND digit TO stack
        END FOR
        IF k > 0
            SET final_stack TO all elements of stack except last k elements
        ELSE
            SET final_stack TO stack
        END IF
        SET result TO concatenation of elements in final_stack WITH leading zeros removed
        IF result IS empty
            RETURN "0"
        ELSE
            RETURN result
        END IF
    END FUNCTION
END CLASS
