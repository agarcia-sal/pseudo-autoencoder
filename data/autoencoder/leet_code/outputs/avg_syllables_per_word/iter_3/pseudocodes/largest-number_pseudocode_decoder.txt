CLASS Solution
    FUNCTION largestNumber(nums: List of Integer) RETURNS String
        SET nums_str TO list of strings converted from nums
        FUNCTION compare(x: String, y: String) RETURNS Integer
            IF concatenate x and y IS GREATER THAN concatenate y and x THEN
                RETURN -1
            ELSE IF concatenate x and y IS LESS THAN concatenate y and x THEN
                RETURN 1
            ELSE
                RETURN 0
            END IF
        END FUNCTION
        SORT nums_str USING compare AS comparator
        IF first element of nums_str IS EQUAL TO "0" THEN
            RETURN "0"
        END IF
        RETURN concatenation of all elements in nums_str
    END FUNCTION
END CLASS
