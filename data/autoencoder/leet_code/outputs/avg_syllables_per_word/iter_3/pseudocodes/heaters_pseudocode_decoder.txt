CLASS Solution
    FUNCTION findRadius(houses, heaters)
        SORT houses
        SORT heaters

        SET min_radius TO 0
        SET j TO 0

        FOR each house IN houses
            WHILE j LESS THAN LENGTH OF heaters MINUS 1 AND ABSOLUTE_VALUE OF (heaters[j] MINUS house) GREATER THAN OR EQUAL TO ABSOLUTE_VALUE OF (heaters[j + 1] MINUS house)
                INCREMENT j BY 1
            END WHILE

            SET distance TO ABSOLUTE_VALUE OF (heaters[j] MINUS house)
            SET min_radius TO MAXIMUM OF min_radius AND distance
        END FOR

        RETURN min_radius
    END FUNCTION
END CLASS
