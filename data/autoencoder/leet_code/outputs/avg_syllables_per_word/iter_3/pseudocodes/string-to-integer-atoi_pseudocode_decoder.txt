CLASS Solution
    FUNCTION myAtoi(s: string) RETURNS integer
        SET INT_MAX TO 2 TO THE POWER OF 31 MINUS 1
        SET INT_MIN TO NEGATIVE 2 TO THE POWER OF 31
        SET result TO 0
        SET index TO 0
        SET n TO LENGTH OF s
        SET sign TO 1
        WHILE index LESS THAN n AND s[index] EQUALS ' '
            INCREMENT index BY 1
        IF index LESS THAN n AND (s[index] EQUALS '+' OR s[index] EQUALS '-')
            IF s[index] EQUALS '-'
                SET sign TO -1
            ELSE
                SET sign TO 1
            INCREMENT index BY 1
        WHILE index LESS THAN n AND s[index] IS A DIGIT
            SET digit TO INTEGER VALUE OF s[index]
            IF result GREATER THAN (INT_MAX MINUS digit) DIVIDED BY 10
                IF sign EQUALS 1
                    RETURN INT_MAX
                ELSE
                    RETURN INT_MIN
            SET result TO result TIMES 10 PLUS digit
            INCREMENT index BY 1
        RETURN sign TIMES result
    END FUNCTION
END CLASS
