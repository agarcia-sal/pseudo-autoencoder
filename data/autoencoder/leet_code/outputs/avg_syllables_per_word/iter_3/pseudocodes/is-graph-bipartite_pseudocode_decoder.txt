CLASS Solution
    FUNCTION isBipartite(graph)
        SET n TO LENGTH OF graph
        SET color TO LIST OF LENGTH n FILLED WITH 0

        FUNCTION dfs(node, c)
            SET color[node] TO c
            FOR each neighbor IN graph[node]
                IF color[neighbor] EQUALS c THEN
                    RETURN False
                END IF
                IF color[neighbor] EQUALS 0 THEN
                    IF dfs(neighbor, NEGATIVE c) IS False THEN
                        RETURN False
                    END IF
                END IF
            END FOR
            RETURN True
        END FUNCTION

        FOR i FROM 0 TO n MINUS 1
            IF color[i] EQUALS 0 THEN
                IF dfs(i, 1) IS False THEN
                    RETURN False
                END IF
            END IF
        END FOR

        RETURN True
    END FUNCTION
END CLASS
