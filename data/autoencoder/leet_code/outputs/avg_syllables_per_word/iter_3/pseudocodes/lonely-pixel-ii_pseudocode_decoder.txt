CLASS Solution
    FUNCTION findBlackPixel(picture, target)
        IF picture IS EMPTY OR picture[0] IS EMPTY
            RETURN 0
        
        SET m TO LENGTH OF picture
        SET n TO LENGTH OF picture[0]
        SET row_count TO LIST OF 0 WITH LENGTH m
        SET col_count TO LIST OF 0 WITH LENGTH n
        SET row_patterns TO DICTIONARY WITH DEFAULT INTEGER 0
        
        FOR r FROM 0 TO m MINUS 1
            SET pattern TO CONCATENATION OF picture[r]
            INCREMENT row_patterns[pattern] BY 1
            FOR c FROM 0 TO n MINUS 1
                IF picture[r][c] EQUALS 'B'
                    INCREMENT row_count[r] BY 1
                    INCREMENT col_count[c] BY 1
        
        SET lonely_pixels TO 0
        
        FOR r FROM 0 TO m MINUS 1
            FOR c FROM 0 TO n MINUS 1
                IF picture[r][c] EQUALS 'B'
                    SET pattern TO CONCATENATION OF picture[r]
                    IF row_count[r] EQUALS target AND col_count[c] EQUALS target AND row_patterns[pattern] EQUALS target
                        INCREMENT lonely_pixels BY 1
        
        RETURN lonely_pixels
    END FUNCTION
END CLASS
