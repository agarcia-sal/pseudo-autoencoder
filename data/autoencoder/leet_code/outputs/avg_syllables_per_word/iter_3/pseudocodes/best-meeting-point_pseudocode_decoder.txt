CLASS Solution
    FUNCTION minTotalDistance(grid)
        SET rows TO empty list
        SET cols TO empty list
        FOR i FROM 0 TO LENGTH OF grid MINUS 1
            FOR j FROM 0 TO LENGTH OF grid[0] MINUS 1
                IF grid[i][j] EQUALS 1 THEN
                    APPEND i TO rows
                    APPEND j TO cols
        CALL rows.sort()
        CALL cols.sort()
        SET median_row TO rows AT POSITION LENGTH OF rows DIVIDED BY 2
        SET median_col TO cols AT POSITION LENGTH OF cols DIVIDED BY 2
        SET min_distance TO 0
        FOR EACH row IN rows
            SET min_distance TO min_distance PLUS ABSOLUTE VALUE OF (row MINUS median_row)
        FOR EACH col IN cols
            SET min_distance TO min_distance PLUS ABSOLUTE VALUE OF (col MINUS median_col)
        RETURN min_distance
    END FUNCTION
END CLASS
