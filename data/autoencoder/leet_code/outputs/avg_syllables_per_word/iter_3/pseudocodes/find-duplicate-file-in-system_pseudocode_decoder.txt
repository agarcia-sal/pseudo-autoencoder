CLASS Solution
    FUNCTION findDuplicate(paths)
        SET content_to_paths TO empty mapping with default empty list
        FOR each path IN paths
            SET parts TO SPLIT path BY whitespace
            SET directory TO parts[0]
            FOR each file_info IN parts FROM index 1 TO end
                SET file_name TO substring before '(' in file_info
                SET content TO substring after '(' and before last character in file_info
                APPEND directory + "/" + file_name TO content_to_paths[content]
        SET result TO empty list
        FOR each group IN values of content_to_paths
            IF LENGTH OF group GREATER THAN 1
                APPEND group TO result
        RETURN result
    END FUNCTION
END CLASS
