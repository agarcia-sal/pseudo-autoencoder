CLASS Solution
    FUNCTION minMeetingRooms(intervals)
        IF intervals IS EMPTY
            RETURN 0
        END IF

        SORT intervals BY the first element of each interval (start time)

        SET min_heap TO an empty list

        FOR each meeting IN intervals
            IF min_heap IS NOT EMPTY AND meeting start time IS GREATER THAN OR EQUAL TO the smallest end time in min_heap
                REMOVE the smallest end time FROM min_heap
            END IF

            ADD meeting end time TO min_heap
        END FOR

        RETURN the size of min_heap
    END FUNCTION
END CLASS
