CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO NEW TreeNode(values[0])
    SET i TO 1
    INITIALIZE queue AS DEQUE
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO POP LEFT FROM queue
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO NEW TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO NEW TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val != q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION printTree(root: Optional[TreeNode]) RETURNS List[List[str]]
        FUNCTION getHeight(node)
            IF node IS None
                RETURN 0
            RETURN 1 + MAXIMUM OF getHeight(node.left) AND getHeight(node.right)
        END FUNCTION

        SET height TO getHeight(root)
        SET width TO (2 RAISED TO height) MINUS 1
        INITIALIZE res AS LIST OF height LISTS EACH CONTAINING width EMPTY STRINGS

        FUNCTION placeNode(node, row, left, right)
            IF node IS None
                RETURN
            SET mid TO (left + right) DIVIDED BY 2
            SET res[row][mid] TO CONVERT node.val TO STRING
            placeNode(node.left, row + 1, left, mid)
            placeNode(node.right, row + 1, mid, right)
        END FUNCTION

        CALL placeNode(root, 0, 0, width)
        RETURN res
    END FUNCTION
END CLASS
