CLASS Solution
    FUNCTION getSum(a, b)
        SET MAX TO 0x7FFFFFFF
        SET mask TO 0xFFFFFFFF
        WHILE b IS NOT 0
            SET carry TO (a AND b) SHIFTED LEFT BY 1
            SET a TO (a XOR b) AND mask
            SET b TO carry AND mask
        IF a IS LESS THAN OR EQUAL TO MAX
            RETURN a
        ELSE
            RETURN BITWISE NEGATION OF (a XOR mask)
    END FUNCTION
END CLASS
