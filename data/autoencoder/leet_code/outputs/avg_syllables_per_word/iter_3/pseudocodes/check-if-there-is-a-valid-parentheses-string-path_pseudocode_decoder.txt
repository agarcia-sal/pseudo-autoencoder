CLASS Solution
    FUNCTION hasValidPath(grid)
        SET m TO LENGTH OF grid
        SET n TO LENGTH OF grid[0]

        IF (m + n - 1) MODULO 2 IS NOT 0 THEN
            RETURN False

        CREATE dp AS 3D LIST OF SIZE (m + 1) BY (n + 1) BY (m + n + 1), INITIALIZED TO False

        SET dp[0][1][0] TO True

        FOR i FROM 0 TO m - 1
            FOR j FROM 0 TO n - 1
                FOR k FROM 0 TO m + n
                    IF dp[i][j + 1][k] OR dp[i + 1][j][k] THEN
                        IF grid[i][j] IS '(' AND k + 1 < m + n + 1 THEN
                            SET dp[i + 1][j + 1][k + 1] TO True
                        ELSE IF grid[i][j] IS ')' AND k - 1 >= 0 THEN
                            SET dp[i + 1][j + 1][k - 1] TO True

        RETURN dp[m][n][0]
    END FUNCTION
END CLASS
