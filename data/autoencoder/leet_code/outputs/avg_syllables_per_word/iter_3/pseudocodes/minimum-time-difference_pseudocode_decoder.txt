CLASS Solution
    FUNCTION findMinDifference(timePoints)
        FUNCTION to_minutes(time)
            SET hours, minutes TO time SPLIT BY ':' CONVERTED TO INTEGERS
            RETURN hours MULTIPLIED BY 60 PLUS minutes
        END FUNCTION

        SET minutes_list TO MAP to_minutes OVER timePoints
        SORT minutes_list

        SET min_diff TO infinity
        FOR i FROM 1 TO LENGTH OF minutes_list MINUS 1
            SET difference TO minutes_list[i] MINUS minutes_list[i - 1]
            SET min_diff TO MINIMUM OF min_diff AND difference
        END FOR

        SET circular_diff TO (1440 MINUS minutes_list[-1]) PLUS minutes_list[0]
        SET min_diff TO MINIMUM OF min_diff AND circular_diff

        RETURN min_diff
    END FUNCTION
END CLASS
