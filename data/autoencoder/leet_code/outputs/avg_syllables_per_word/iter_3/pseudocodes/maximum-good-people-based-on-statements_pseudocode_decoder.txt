CLASS Solution
    FUNCTION maximumGood(statements: List of List of int)
        SET n TO LENGTH OF statements
        SET max_good TO 0
        FOR i FROM 0 TO (2 RAISED TO n) MINUS 1
            SET good TO LIST OF n FALSE VALUES
            SET bad TO LIST OF n FALSE VALUES
            FOR j FROM 0 TO n MINUS 1
                IF (i AND (1 SHIFTED LEFT BY j)) IS NOT ZERO
                    SET good[j] TO TRUE
                ELSE
                    SET bad[j] TO TRUE
            SET valid TO TRUE
            FOR j FROM 0 TO n MINUS 1
                IF good[j]
                    FOR k FROM 0 TO n MINUS 1
                        IF (statements[j][k] EQUALS 0 AND good[k]) OR (statements[j][k] EQUALS 1 AND bad[k])
                            SET valid TO FALSE
                            BREAK
                    IF NOT valid
                        BREAK
            IF valid
                SET max_good TO MAXIMUM OF max_good AND SUM OF VALUES IN good
        RETURN max_good
    END FUNCTION
END CLASS
