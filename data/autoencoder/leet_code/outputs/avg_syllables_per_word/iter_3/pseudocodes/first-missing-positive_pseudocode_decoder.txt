CLASS Solution
    FUNCTION firstMissingPositive(nums)
        SET n TO LENGTH OF nums
        
        FOR i FROM 0 TO n - 1
            WHILE nums[i] IS GREATER THAN OR EQUAL TO 1 AND nums[i] IS LESS THAN OR EQUAL TO n AND nums[nums[i] MINUS 1] IS NOT EQUAL TO nums[i]
                SET correct_index TO nums[i] MINUS 1
                SWAP nums[i] WITH nums[correct_index]
        
        FOR i FROM 0 TO n - 1
            IF nums[i] IS NOT EQUAL TO i PLUS 1
                RETURN i PLUS 1
        
        RETURN n PLUS 1
END CLASS
