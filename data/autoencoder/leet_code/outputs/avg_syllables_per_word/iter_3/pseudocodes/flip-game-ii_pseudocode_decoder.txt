CLASS Solution
    FUNCTION canWin(currentState)
        SET memo TO empty dictionary
        
        FUNCTION can_win_helper(state)
            IF state IN memo THEN
                RETURN memo[state]
            
            FOR i FROM 0 TO LENGTH OF state MINUS 2
                IF SUBSTRING OF state FROM i TO i+2 EQUALS "++" THEN
                    SET new_state TO CONCATENATION OF SUBSTRING OF state UP TO i, "--", AND SUBSTRING OF state FROM i+2 TO END
                    IF NOT can_win_helper(new_state) THEN
                        SET memo[state] TO True
                        RETURN True
            
            SET memo[state] TO False
            RETURN False
        
        RETURN can_win_helper(currentState)
    END FUNCTION
END CLASS
