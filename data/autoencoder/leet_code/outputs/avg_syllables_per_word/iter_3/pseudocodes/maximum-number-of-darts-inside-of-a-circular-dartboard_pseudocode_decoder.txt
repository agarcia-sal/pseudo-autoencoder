CLASS Solution
    FUNCTION numPoints(darts, r)
        FUNCTION distance(p1, p2)
            SET dx TO p1[0] MINUS p2[0]
            SET dy TO p1[1] MINUS p2[1]
            RETURN square root of (dx squared PLUS dy squared)
        END FUNCTION

        FUNCTION circle_center(p1, p2, r)
            SET d TO distance(p1, p2)
            IF d GREATER THAN 2 TIMES r THEN
                RETURN None
            END IF
            SET a TO square root of (r squared MINUS (d DIVIDED BY 2) squared)
            SET h TO list containing (p1[0] PLUS p2[0]) DIVIDED BY 2 AND (p1[1] PLUS p2[1]) DIVIDED BY 2
            SET dx TO (p2[1] MINUS p1[1]) TIMES a DIVIDED BY d
            SET dy TO (p2[0] MINUS p1[0]) TIMES a DIVIDED BY d
            RETURN list containing (h[0] MINUS dx, h[1] PLUS dy) AND (h[0] PLUS dx, h[1] MINUS dy)
        END FUNCTION

        FUNCTION is_inside(circle, point, r)
            RETURN distance(circle, point) LESS THAN OR EQUAL TO r
        END FUNCTION

        SET n TO LENGTH OF darts
        IF n EQUALS 1 THEN
            RETURN 1
        END IF

        SET max_count TO 1

        FOR i FROM 0 TO n MINUS 1
            FOR j FROM i PLUS 1 TO n MINUS 1
                SET centers TO circle_center(darts[i], darts[j], r)
                IF centers IS NOT None THEN
                    FOR EACH center IN centers
                        SET count TO 0
                        FOR EACH dart IN darts
                            IF is_inside(center, dart, r) THEN
                                INCREMENT count BY 1
                            END IF
                        END FOR
                        IF count GREATER THAN max_count THEN
                            SET max_count TO count
                        END IF
                    END FOR
                END IF
            END FOR
        END FOR

        RETURN max_count
    END FUNCTION
END CLASS
