CLASS Solution
    FUNCTION minimumBoxes(n)
        FUNCTION triangular_pyramid(k)
            RETURN k * (k + 1) * (k + 2) DIVIDED BY 6
        END FUNCTION

        SET left TO 0
        SET right TO n
        WHILE left < right
            SET mid TO (left + right + 1) DIVIDED BY 2
            IF triangular_pyramid(mid) <= n THEN
                SET left TO mid
            ELSE
                SET right TO mid - 1
            END IF
        END WHILE

        SET boxes_used TO triangular_pyramid(left)
        SET remaining TO n - boxes_used
        SET floor_boxes TO left * (left + 1) DIVIDED BY 2

        SET additional_boxes TO 0
        WHILE remaining > 0
            INCREMENT additional_boxes BY 1
            DECREMENT remaining BY additional_boxes
        END WHILE

        RETURN floor_boxes + additional_boxes
    END FUNCTION
END CLASS
