CLASS Solution
    FUNCTION triangleNumber(nums)
        SORT nums
        SET count TO 0
        SET n TO LENGTH OF nums

        FOR k FROM n MINUS 1 DOWN TO 2 STEP -1
            SET i TO 0
            SET j TO k MINUS 1
            WHILE i LESS THAN j
                IF nums[i] PLUS nums[j] GREATER THAN nums[k]
                    INCREMENT count BY j MINUS i
                    DECREMENT j BY 1
                ELSE
                    INCREMENT i BY 1
        RETURN count
    END FUNCTION
END CLASS
