CLASS TreeNode
    FUNCTION __init__(self, val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    END IF
    SET root TO TreeNode(values[0])
    SET i TO 1
    INITIALIZE queue AS deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO queue POP LEFT
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION largestBSTSubtree(self, root: Optional[TreeNode]) RETURNS int
        
        FUNCTION dfs(node)
            IF node IS None
                RETURN float('inf'), float('-inf'), 0  // min, max, size
            END IF
            
            SET left_min, left_max, left_size TO dfs(node.left)
            SET right_min, right_max, right_size TO dfs(node.right)
            
            IF left_max < node.val AND node.val < right_min
                RETURN MINIMUM OF node.val AND left_min, MAXIMUM OF node.val AND right_max, left_size + right_size + 1
            END IF
            
            RETURN float('-inf'), float('inf'), MAXIMUM OF left_size AND right_size
        END FUNCTION
        
        RETURN dfs(root)[2]
    END FUNCTION
END CLASS
