CLASS Solution
    FUNCTION rotatedDigits(n)
        SET change_digits TO {'2', '5', '6', '9'}
        SET same_digits TO {'0', '1', '8'}

        FUNCTION is_good_number(num)
            SET num_str TO string representation of num
            IF all digits IN num_str ARE IN change_digits UNION same_digits THEN
                RETURN any digit IN num_str IS IN change_digits
            END IF
            RETURN False
        END FUNCTION

        SET good_count TO 0
        FOR i FROM 1 TO n
            IF is_good_number(i) THEN
                INCREMENT good_count BY 1
            END IF
        END FOR
        RETURN good_count
    END FUNCTION
END CLASS
