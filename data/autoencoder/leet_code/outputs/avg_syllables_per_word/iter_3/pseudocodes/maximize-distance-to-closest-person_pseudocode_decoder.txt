CLASS Solution
    FUNCTION maxDistToClosest(seats)
        SET max_distance TO 0
        SET start TO 0
        SET end TO 0
        SET n TO LENGTH OF seats

        WHILE start < n AND seats[start] EQUALS 0
            INCREMENT start BY 1

        SET end TO n MINUS 1
        WHILE end >= 0 AND seats[end] EQUALS 0
            DECREMENT end BY 1

        SET max_distance TO MAXIMUM OF max_distance AND start
        SET max_distance TO MAXIMUM OF max_distance AND (n MINUS end MINUS 1)

        SET current_distance TO 0
        FOR i FROM start TO end INCLUSIVE
            IF seats[i] EQUALS 0 THEN
                INCREMENT current_distance BY 1
                SET max_distance TO MAXIMUM OF max_distance AND (current_distance PLUS 1) DIVIDED BY 2
            ELSE
                SET current_distance TO 0
        RETURN max_distance
    END FUNCTION
END CLASS
