CLASS Solution
    FUNCTION largestNumber(cost: List[int], target: int) RETURNS string
        SET cost_to_digit TO DICTIONARY MAPPING each cost value c TO string of digit (index + 1)
        SET dp TO LIST WITH first element "" AND rest "-1" REPEATED target times

        FOR t FROM 1 TO target INCLUSIVE
            FOR each cost c AND digit d IN cost_to_digit
                IF t IS GREATER THAN OR EQUAL TO c AND dp AT POSITION t MINUS c IS NOT "-1"
                    SET candidate TO CONCATENATION OF d AND dp AT POSITION t MINUS c
                    IF dp AT POSITION t IS "-1" OR LENGTH OF candidate IS GREATER THAN LENGTH OF dp AT POSITION t OR (LENGTHS ARE EQUAL AND candidate IS GREATER THAN dp AT POSITION t)
                        SET dp AT POSITION t TO candidate

        IF dp AT POSITION target IS NOT "-1"
            RETURN dp AT POSITION target
        ELSE
            RETURN "0"
    END FUNCTION
END CLASS
