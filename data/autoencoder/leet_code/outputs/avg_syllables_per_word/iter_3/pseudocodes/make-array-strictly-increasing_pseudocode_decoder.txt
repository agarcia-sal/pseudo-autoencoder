CLASS Solution
    FUNCTION makeArrayIncreasing(arr1, arr2)
        SORT arr2
        
        SET dp TO dictionary with key -1 and value 0
        
        FOR each num IN arr1
            SET new_dp TO empty dictionary
            
            FOR each last_num, ops IN dp
                IF num > last_num THEN
                    SET current_min TO value from new_dp for key num OR infinity if not present
                    SET new_dp[num] TO minimum of current_min and ops
                
                SET idx TO position in arr2 where last_num would be inserted to the right
                
                IF idx is less than length of arr2 THEN
                    SET replacement TO arr2 element at idx
                    SET current_min TO value from new_dp for key replacement OR infinity if not present
                    SET new_dp[replacement] TO minimum of current_min and ops plus 1
            
            IF new_dp is empty THEN
                RETURN -1
            
            SET dp TO new_dp
        
        RETURN minimum value among dp values
    END FUNCTION
END CLASS
