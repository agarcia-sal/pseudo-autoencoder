CLASS Solution
    FUNCTION rob(nums)
        IF nums IS EMPTY
            RETURN 0
        IF LENGTH OF nums IS 1
            RETURN nums[0]

        SET dp TO LIST OF ZEROS WITH LENGTH OF nums
        SET dp[0] TO nums[0]
        SET dp[1] TO MAXIMUM OF nums[0] AND nums[1]

        FOR i FROM 2 TO LENGTH OF nums MINUS 1
            SET dp[i] TO MAXIMUM OF dp[i - 1] AND dp[i - 2] PLUS nums[i]

        RETURN dp[LENGTH OF dp MINUS 1]
    END FUNCTION
END CLASS
