CLASS Solution
    FUNCTION winnerSquareGame(n)
        SET dp TO list of False values of length n + 1
        FOR i FROM 1 TO n
            SET j TO 1
            WHILE j * j <= i
                IF dp at position (i minus j * j) IS False
                    SET dp at position i TO True
                    BREAK loop
                INCREMENT j BY 1
        RETURN dp at position n
    END FUNCTION
END CLASS
