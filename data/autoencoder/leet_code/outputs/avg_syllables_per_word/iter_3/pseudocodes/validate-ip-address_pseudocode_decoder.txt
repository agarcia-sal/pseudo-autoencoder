CLASS Solution
    FUNCTION validIPAddress(queryIP)
        FUNCTION is_valid_ipv4_block(block)
            IF block consists only of digits THEN
                SET num TO integer value of block
                RETURN num less than or equal to 255 AND string of num equals block
            ELSE
                RETURN False
        END FUNCTION

        FUNCTION is_valid_ipv6_block(block)
            IF length of block is between 1 and 4 THEN
                TRY
                    CONVERT block to integer with base 16
                    RETURN True
                CATCH conversion error
                    RETURN False
            ELSE
                RETURN False
        END FUNCTION

        SET parts_ipv4 TO list of substrings from splitting queryIP by '.'
        SET parts_ipv6 TO list of substrings from splitting queryIP by ':'

        IF length of parts_ipv4 equals 4 AND every part in parts_ipv4 satisfies is_valid_ipv4_block THEN
            RETURN "IPv4"
        ELSE IF length of parts_ipv6 equals 8 AND every part in parts_ipv6 satisfies is_valid_ipv6_block THEN
            RETURN "IPv6"
        ELSE
            RETURN "Neither"
    END FUNCTION
END CLASS
