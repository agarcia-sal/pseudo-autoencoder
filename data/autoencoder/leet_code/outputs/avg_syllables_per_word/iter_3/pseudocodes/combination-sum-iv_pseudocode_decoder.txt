CLASS Solution
    FUNCTION combinationSum4(nums, target)
        SET dp TO a list of zeros with length target plus one
        SET dp[0] TO 1
        FOR amount FROM 1 TO target
            FOR each num IN nums
                IF amount is greater than or equal to num THEN
                    INCREMENT dp[amount] BY dp[amount minus num]
        RETURN dp[target]
    END FUNCTION
END CLASS
