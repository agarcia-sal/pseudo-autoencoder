CLASS Solution
    FUNCTION findDuplicate(nums)
        FUNCTION f(x)
            SET count TO 0
            FOR each value IN nums
                IF value LESS THAN OR EQUAL TO x
                    INCREMENT count BY 1
            RETURN count GREATER THAN x
        END FUNCTION

        RETURN bisect_left OVER range OF LENGTH nums WITH True USING KEY f
    END FUNCTION
END CLASS
