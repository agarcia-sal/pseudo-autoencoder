CLASS Solution
    FUNCTION countRoutes(locations, start, finish, fuel)
        SET MOD TO 10^9 + 7
        SET n TO LENGTH OF locations
        
        FUNCTION dp(city, remaining_fuel) CACHEABLE
            IF remaining_fuel < 0 THEN
                RETURN 0
            SET routes TO 0
            IF city == finish THEN
                SET routes TO 1
            FOR next_city FROM 0 TO n - 1
                IF next_city != city THEN
                    SET fuel_cost TO ABSOLUTE VALUE OF (locations[city] MINUS locations[next_city])
                    IF remaining_fuel >= fuel_cost THEN
                        SET routes TO (routes PLUS dp(next_city, remaining_fuel MINUS fuel_cost)) MODULO MOD
            RETURN routes
        END FUNCTION
        
        RETURN dp(start, fuel)
    END FUNCTION
END CLASS
