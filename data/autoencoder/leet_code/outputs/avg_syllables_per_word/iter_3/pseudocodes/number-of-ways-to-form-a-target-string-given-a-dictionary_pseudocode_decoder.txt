CLASS Solution
    FUNCTION numWays(words: List of String, target: String) RETURNS Integer
        SET MOD TO 10^9 + 7
        SET n TO LENGTH OF words[0]
        SET m TO LENGTH OF target
        
        SET freq TO LIST OF LENGTH n WITH DEFAULT DICTIONARIES MAPPING CHARACTERS TO INTEGERS
        FOR EACH word IN words
            FOR j FROM 0 TO n - 1
                INCREMENT freq[j][word[j]] BY 1
        
        SET dp TO 2D LIST OF SIZE (m + 1) BY (n + 1) FILLED WITH 0
        
        FOR j FROM 0 TO n
            SET dp[0][j] TO 1
        
        FOR i FROM 1 TO m
            FOR j FROM 1 TO n
                SET dp[i][j] TO dp[i][j - 1]
                IF freq[j - 1][target[i - 1]] > 0 THEN
                    INCREMENT dp[i][j] BY dp[i - 1][j - 1] MULTIPLIED BY freq[j - 1][target[i - 1]]
                    SET dp[i][j] TO dp[i][j] MODULO MOD
        
        RETURN dp[m][n]
    END FUNCTION
END CLASS
