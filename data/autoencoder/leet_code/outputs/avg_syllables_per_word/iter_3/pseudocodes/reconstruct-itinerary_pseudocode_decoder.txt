CLASS Solution
    FUNCTION findItinerary(tickets)
        SET graph TO a new default dictionary with list as default value
        FOR each (frm, to) IN tickets sorted in reverse order
            APPEND to TO graph at key frm
        SET itinerary TO an empty list
        FUNCTION visit(airport)
            WHILE graph at airport is not empty
                SET next_destination TO REMOVE last element from graph at airport
                CALL visit(next_destination)
            APPEND airport TO itinerary
        END FUNCTION
        CALL visit("JFK")
        RETURN reversed itinerary
    END FUNCTION
END CLASS
