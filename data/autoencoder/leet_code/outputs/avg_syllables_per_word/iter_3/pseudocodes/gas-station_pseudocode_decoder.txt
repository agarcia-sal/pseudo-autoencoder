CLASS Solution
    FUNCTION canCompleteCircuit(gas, cost)
        IF sum of gas IS LESS THAN sum of cost THEN
            RETURN -1
        SET start TO 0
        SET total_tank TO 0
        FOR i FROM 0 TO LENGTH OF gas MINUS 1
            INCREMENT total_tank BY gas[i] MINUS cost[i]
            IF total_tank IS LESS THAN 0 THEN
                SET start TO i PLUS 1
                SET total_tank TO 0
        RETURN start
    END FUNCTION
END CLASS
