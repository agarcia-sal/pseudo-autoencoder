CLASS Solution
    FUNCTION kthSmallest(matrix, k)
        FUNCTION countLessEqual(mid)
            SET count TO 0
            SET row TO n MINUS 1
            SET col TO 0
            WHILE row IS GREATER THAN OR EQUAL TO 0 AND col IS LESS THAN n
                IF matrix[row][col] IS LESS THAN OR EQUAL TO mid
                    INCREMENT count BY row PLUS 1
                    INCREMENT col BY 1
                ELSE
                    DECREMENT row BY 1
            RETURN count
        END FUNCTION

        SET n TO LENGTH OF matrix
        SET low TO matrix[0][0]
        SET high TO matrix[n MINUS 1][n MINUS 1]

        WHILE low IS LESS THAN high
            SET mid TO low PLUS high DIVIDED BY 2
            IF countLessEqual(mid) IS LESS THAN k
                SET low TO mid PLUS 1
            ELSE
                SET high TO mid

        RETURN low
    END FUNCTION
END CLASS
