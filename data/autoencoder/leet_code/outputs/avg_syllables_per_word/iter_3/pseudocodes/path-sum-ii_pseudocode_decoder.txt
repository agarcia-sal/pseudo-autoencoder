CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS empty
        RETURN None
    SET root TO TreeNode(values[0])
    SET i TO 1
    INITIALIZE queue AS empty deque
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO queue.popleft()
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION pathSum(root: Optional[TreeNode], targetSum: int) RETURNS List of List of int
        FUNCTION dfs(node, current_sum, path)
            IF node IS None
                RETURN
            APPEND node.val TO path
            SET current_sum TO current_sum PLUS node.val
            IF node.left IS None AND node.right IS None AND current_sum IS EQUAL TO targetSum
                APPEND COPY OF path TO result
            CALL dfs(node.left, current_sum, path)
            CALL dfs(node.right, current_sum, path)
            REMOVE last element FROM path
        END FUNCTION

        SET result TO empty list
        CALL dfs(root, 0, empty list)
        RETURN result
    END FUNCTION
END CLASS
