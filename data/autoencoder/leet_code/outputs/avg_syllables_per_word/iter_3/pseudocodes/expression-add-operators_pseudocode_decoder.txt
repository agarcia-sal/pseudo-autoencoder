CLASS Solution
    FUNCTION addOperators(num, target)
        FUNCTION dfs(index, path, value, prev)
            IF index EQUALS LENGTH OF num THEN
                IF value EQUALS target THEN
                    APPEND path TO result
                END IF
                RETURN
            END IF

            FOR i FROM index + 1 TO LENGTH OF num + 1
                SET operand TO SUBSTRING OF num FROM index TO i
                IF LENGTH OF operand GREATER THAN 1 AND FIRST CHARACTER OF operand EQUALS '0' THEN
                    CONTINUE
                END IF

                IF index EQUALS 0 THEN
                    CALL dfs(i, operand, INTEGER VALUE OF operand, INTEGER VALUE OF operand)
                ELSE
                    CALL dfs(i, path CONCATENATED WITH '+' CONCATENATED WITH operand, value PLUS INTEGER VALUE OF operand, INTEGER VALUE OF operand)
                    CALL dfs(i, path CONCATENATED WITH '-' CONCATENATED WITH operand, value MINUS INTEGER VALUE OF operand, NEGATIVE INTEGER VALUE OF operand)
                    CALL dfs(i, path CONCATENATED WITH '*' CONCATENATED WITH operand, value MINUS prev PLUS prev TIMES INTEGER VALUE OF operand, prev TIMES INTEGER VALUE OF operand)
                END IF
            END FOR
        END FUNCTION

        SET result TO EMPTY LIST
        CALL dfs(0, EMPTY STRING, 0, 0)
        RETURN result
    END FUNCTION
END CLASS
