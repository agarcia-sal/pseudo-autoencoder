CLASS Solution
    FUNCTION canPartitionKSubsets(nums, k)
        SET total_sum TO sum of nums
        IF total_sum MODULO k IS NOT 0
            RETURN False
        SET target TO total_sum DIVIDED BY k
        SORT nums IN descending order

        FUNCTION can_partition(index, k, current_sum, used)
            IF k IS 0
                RETURN True
            IF current_sum IS EQUAL TO target
                RETURN can_partition(0, k MINUS 1, 0, used)

            FOR i FROM index TO LENGTH OF nums MINUS 1
                IF used[i] IS False AND current_sum PLUS nums[i] LESS THAN OR EQUAL TO target
                    SET used[i] TO True
                    IF can_partition(i PLUS 1, k, current_sum PLUS nums[i], used)
                        RETURN True
                    SET used[i] TO False
            RETURN False
        END FUNCTION

        SET used TO list of False with length of nums
        RETURN can_partition(0, k, 0, used)
    END FUNCTION
END CLASS
