CLASS Solution
    FUNCTION numFactoredBinaryTrees(arr)
        SET MOD TO 1000000007
        SORT arr IN ASCENDING ORDER
        SET dp TO DICTIONARY with keys FROM arr AND values 1
        
        FOR i FROM 0 TO LENGTH OF arr MINUS 1
            SET num TO arr[i]
            FOR j FROM 0 TO i MINUS 1
                IF num MODULO arr[j] EQUALS 0 THEN
                    SET right_child TO num DIVIDED BY arr[j]
                    IF right_child IN dp THEN
                        SET dp[num] TO (dp[num] PLUS dp[arr[j]] TIMES dp[right_child]) MODULO MOD
        END FOR
        
        SET total TO SUM OF ALL VALUES IN dp
        RETURN total MODULO MOD
    END FUNCTION
END CLASS
