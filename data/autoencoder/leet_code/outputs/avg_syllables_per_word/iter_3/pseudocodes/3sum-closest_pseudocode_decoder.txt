CLASS Solution
    FUNCTION threeSumClosest(nums, target)
        SORT nums
        SET closest_sum TO infinity
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 3
            SET left TO i PLUS 1
            SET right TO LENGTH OF nums MINUS 1
            
            WHILE left LESS THAN right
                SET current_sum TO nums[i] PLUS nums[left] PLUS nums[right]
                
                IF ABSOLUTE VALUE OF (current_sum MINUS target) LESS THAN ABSOLUTE VALUE OF (closest_sum MINUS target)
                    SET closest_sum TO current_sum
                
                IF current_sum LESS THAN target
                    INCREMENT left BY 1
                ELSE IF current_sum GREATER THAN target
                    DECREMENT right BY 1
                ELSE
                    RETURN current_sum
                
        RETURN closest_sum
    END FUNCTION
END CLASS
