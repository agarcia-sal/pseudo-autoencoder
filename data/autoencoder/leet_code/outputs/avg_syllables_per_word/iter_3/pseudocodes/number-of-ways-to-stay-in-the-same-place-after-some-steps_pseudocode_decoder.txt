CLASS Solution
    FUNCTION numWays(steps, arrLen)
        SET MOD TO 10^9 + 7
        SET max_index TO MINIMUM OF (steps DIVIDED BY 2) AND (arrLen MINUS 1)
        SET dp TO LIST OF ZEROS OF LENGTH (max_index PLUS 1)
        SET dp[0] TO 1

        FOR EACH _ IN RANGE OF steps
            SET prev_dp TO COPY OF dp
            FOR i FROM 0 TO max_index INCLUSIVE
                SET dp[i] TO prev_dp[i]
                IF i GREATER THAN 0
                    SET dp[i] TO (dp[i] PLUS prev_dp[i MINUS 1]) MODULO MOD
                IF i LESS THAN max_index
                    SET dp[i] TO (dp[i] PLUS prev_dp[i PLUS 1]) MODULO MOD

        RETURN dp[0]
    END FUNCTION
END CLASS
