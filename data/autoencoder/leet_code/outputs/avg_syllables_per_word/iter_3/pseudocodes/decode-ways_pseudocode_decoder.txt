CLASS Solution
    FUNCTION numDecodings(s)
        IF s IS EMPTY OR s[0] IS '0'
            RETURN 0
        END IF

        SET dp TO LIST OF ZEROS OF SIZE LENGTH(s) + 1
        SET dp[0] TO 1
        SET dp[1] TO 1

        FOR i FROM 2 TO LENGTH(s)
            IF s[i-1] IS NOT '0'
                INCREMENT dp[i] BY dp[i-1]
            END IF

            SET two_digit TO INTEGER VALUE OF s[i-2 TO i]
            IF two_digit IS GREATER THAN OR EQUAL TO 10 AND LESS THAN OR EQUAL TO 26
                INCREMENT dp[i] BY dp[i-2]
            END IF
        END FOR

        RETURN dp[LENGTH(s)]
    END FUNCTION
END CLASS
