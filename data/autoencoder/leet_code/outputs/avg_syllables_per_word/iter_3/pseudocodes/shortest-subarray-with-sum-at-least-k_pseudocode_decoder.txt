CLASS Solution
    FUNCTION shortestSubarray(nums, k)
        SET prefix_sum TO LIST WITH SINGLE ELEMENT 0
        FOR each num IN nums
            APPEND prefix_sum LAST ELEMENT PLUS num TO prefix_sum
        END FOR

        SET dq TO EMPTY DEQUE
        SET min_length TO INFINITY

        FOR i, current_sum IN ENUMERATE prefix_sum
            WHILE dq IS NOT EMPTY AND current_sum MINUS prefix_sum AT dq FIRST ELEMENT IS GREATER THAN OR EQUAL TO k
                SET min_length TO MINIMUM OF min_length AND i MINUS dq FIRST ELEMENT
                REMOVE FIRST ELEMENT FROM dq
            END WHILE

            WHILE dq IS NOT EMPTY AND current_sum IS LESS THAN OR EQUAL TO prefix_sum AT dq LAST ELEMENT
                REMOVE LAST ELEMENT FROM dq
            END WHILE

            APPEND i TO dq
        END FOR

        IF min_length IS EQUAL TO INFINITY
            RETURN -1
        ELSE
            RETURN min_length
        END IF
    END FUNCTION
END CLASS
