CLASS Solution
    FUNCTION maxProduct(nums)
        IF nums IS EMPTY
            RETURN 0
        SET max_product TO nums[0]
        SET min_product TO nums[0]
        SET result TO nums[0]
        FOR each num IN nums FROM SECOND ELEMENT
            IF num < 0
                SWAP max_product AND min_product
            SET max_product TO MAX OF num AND (max_product MULTIPLIED BY num)
            SET min_product TO MIN OF num AND (min_product MULTIPLIED BY num)
            SET result TO MAX OF result AND max_product
        RETURN result
    END FUNCTION
END CLASS
