CLASS Solution
    FUNCTION countTriplets(nums)
        SET pair_and_count TO empty dictionary with default 0
        
        FOR each i IN nums
            FOR each j IN nums
                INCREMENT pair_and_count AT KEY (i AND j) BY 1
        
        SET triplet_count TO 0
        FOR each k IN nums
            FOR each pair_and, count IN pair_and_count
                IF (pair_and AND k) EQUALS 0
                    INCREMENT triplet_count BY count
        
        RETURN triplet_count
    END FUNCTION
END CLASS
