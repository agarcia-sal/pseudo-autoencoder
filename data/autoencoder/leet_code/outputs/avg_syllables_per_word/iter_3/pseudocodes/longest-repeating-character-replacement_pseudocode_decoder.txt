CLASS Solution
    FUNCTION characterReplacement(s, k)
        SET max_length TO 0
        SET max_count TO 0
        SET left TO 0
        SET count TO an empty dictionary with default integer values
        
        FOR right FROM 0 TO LENGTH OF s MINUS 1
            INCREMENT count OF s[right] BY 1
            SET max_count TO the maximum of max_count AND count OF s[right]
            
            IF (right MINUS left PLUS 1) MINUS max_count GREATER THAN k THEN
                DECREMENT count OF s[left] BY 1
                INCREMENT left BY 1
            END IF
            
            SET max_length TO the maximum of max_length AND (right MINUS left PLUS 1)
        END FOR
        
        RETURN max_length
    END FUNCTION
END CLASS
