CLASS Solution
    FUNCTION minPathSum(grid: List of List of Integer) RETURNS Integer
        IF grid IS EMPTY OR grid[0] IS EMPTY THEN
            RETURN 0
        END IF
        
        SET m TO LENGTH OF grid
        SET n TO LENGTH OF grid[0]
        
        SET dp TO 2D LIST OF SIZE m BY n FILLED WITH 0
        
        SET dp[0][0] TO grid[0][0]
        
        FOR j FROM 1 TO n MINUS 1
            SET dp[0][j] TO dp[0][j MINUS 1] PLUS grid[0][j]
        END FOR 
        
        FOR i FROM 1 TO m MINUS 1
            SET dp[i][0] TO dp[i MINUS 1][0] PLUS grid[i][0]
        END FOR
        
        FOR i FROM 1 TO m MINUS 1
            FOR j FROM 1 TO n MINUS 1
                SET dp[i][j] TO MINIMUM OF dp[i MINUS 1][j] AND dp[i][j MINUS 1] PLUS grid[i][j]
            END FOR
        END FOR
        
        RETURN dp[m MINUS 1][n MINUS 1]
    END FUNCTION
END CLASS
