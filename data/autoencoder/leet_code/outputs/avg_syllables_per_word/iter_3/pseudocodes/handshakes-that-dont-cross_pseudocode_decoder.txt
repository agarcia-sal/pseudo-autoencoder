CLASS Solution
    FUNCTION numberOfWays(numPeople)
        SET MOD TO 10^9 + 7
        SET dp TO list of zeros with length numPeople + 1
        SET dp[0] TO 1
        SET dp[2] TO 1
        
        FOR i FROM 4 TO numPeople STEP 2
            FOR j FROM 0 TO i - 2 STEP 2
                SET dp[i] TO (dp[i] + dp[j] * dp[i - 2 - j]) MODULO MOD
        
        RETURN dp[numPeople]
    END FUNCTION
END CLASS
