CLASS Solution
    FUNCTION boldWords(words, s)
        SET bold TO list of False values with length of s
        
        FOR each word IN words
            SET start TO position of word in s
            WHILE start IS NOT -1
                FOR i FROM start TO start + length of word - 1
                    SET bold[i] TO True
                SET start TO position of word in s starting at start + 1
                
        SET result TO empty list
        SET i TO 0
        WHILE i LESS THAN length of s
            IF bold[i] IS True
                APPEND "<b>" TO result
                WHILE i LESS THAN length of s AND bold[i] IS True
                    APPEND s[i] TO result
                    INCREMENT i BY 1
                APPEND "</b>" TO result
            ELSE
                APPEND s[i] TO result
                INCREMENT i BY 1
                
        RETURN concatenation of all elements in result
    END FUNCTION
END CLASS
