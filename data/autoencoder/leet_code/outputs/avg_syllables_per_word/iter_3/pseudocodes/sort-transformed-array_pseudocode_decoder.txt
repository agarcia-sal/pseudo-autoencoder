CLASS Solution
    FUNCTION sortTransformedArray(nums, a, b, c)
        FUNCTION quadratic(x)
            RETURN a * x * x + b * x + c
        END FUNCTION

        SET n TO LENGTH OF nums
        SET result TO new list of size n filled with 0
        SET left TO 0
        SET right TO n - 1

        IF a >= 0 THEN
            SET index TO n - 1
            WHILE left <= right DO
                IF quadratic(nums[left]) > quadratic(nums[right]) THEN
                    SET result[index] TO quadratic(nums[left])
                    INCREMENT left BY 1
                ELSE
                    SET result[index] TO quadratic(nums[right])
                    DECREMENT right BY 1
                END IF
                DECREMENT index BY 1
            END WHILE
        ELSE
            SET index TO 0
            WHILE left <= right DO
                IF quadratic(nums[left]) < quadratic(nums[right]) THEN
                    SET result[index] TO quadratic(nums[left])
                    INCREMENT left BY 1
                ELSE
                    SET result[index] TO quadratic(nums[right])
                    DECREMENT right BY 1
                END IF
                INCREMENT index BY 1
            END WHILE
        END IF

        RETURN result
    END FUNCTION
END CLASS
