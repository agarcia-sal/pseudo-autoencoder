CLASS Solution
    FUNCTION maximalSquare(matrix: List OF List OF string)
        IF matrix IS EMPTY OR matrix[0] IS EMPTY THEN
            RETURN 0
        END IF

        SET m TO LENGTH OF matrix
        SET n TO LENGTH OF matrix[0]
        SET dp TO 2D LIST OF SIZE (m + 1) BY (n + 1) FILLED WITH 0
        SET max_side TO 0

        FOR i FROM 1 TO m
            FOR j FROM 1 TO n
                IF matrix[i - 1][j - 1] EQUALS '1' THEN
                    SET dp[i][j] TO MINIMUM OF dp[i - 1][j], dp[i][j - 1], dp[i - 1][j - 1] PLUS 1
                    IF dp[i][j] GREATER THAN max_side THEN
                        SET max_side TO dp[i][j]
                    END IF
                END IF
            END FOR
        END FOR

        RETURN max_side MULTIPLIED BY max_side
    END FUNCTION
END CLASS
