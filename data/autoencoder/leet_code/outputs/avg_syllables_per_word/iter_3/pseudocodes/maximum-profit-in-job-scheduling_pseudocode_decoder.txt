CLASS Solution
    FUNCTION jobScheduling(startTime: List[int], endTime: List[int], profit: List[int]) RETURNS int
        SET jobs TO sorted list of tuples (startTime, endTime, profit) ordered by endTime
        SET end_times TO list of endTime extracted from each job in jobs
        SET dp TO list of zeros with length of jobs plus one
        FOR i FROM 1 TO LENGTH OF jobs
            SET j TO position found by binary searching end_times for jobs[i-1].startTime using bisect right
            SET dp[i] TO maximum of dp[i-1] and dp[j] plus jobs[i-1].profit
        RETURN last element of dp
    END FUNCTION
END CLASS
