CLASS Solution
    FUNCTION splitLoopedString(strs)
        FOR i FROM 0 TO LENGTH OF strs MINUS 1
            SET reversed_str TO REVERSE OF strs[i]
            IF reversed_str IS GREATER THAN strs[i]
                SET strs[i] TO reversed_str
            END IF
        END FOR

        SET doubled_strs TO strs CONCATENATED WITH strs

        SET max_string TO EMPTY STRING

        FOR i FROM 0 TO LENGTH OF strs MINUS 1
            FOR is_reversed IN [FALSE, TRUE]
                IF is_reversed IS TRUE
                    SET current TO REVERSE OF strs[i]
                ELSE
                    SET current TO strs[i]
                END IF

                SET remaining TO CONCATENATION OF doubled_strs FROM i+1 TO i+LENGTH OF strs MINUS 1

                FOR j FROM 0 TO LENGTH OF current
                    SET candidate TO SUBSTRING OF current FROM j TO END CONCATENATED WITH remaining CONCATENATED WITH SUBSTRING OF current FROM START TO j MINUS 1
                    IF candidate IS GREATER THAN max_string
                        SET max_string TO candidate
                    END IF
                END FOR
            END FOR
        END FOR

        RETURN max_string
    END FUNCTION
END CLASS
