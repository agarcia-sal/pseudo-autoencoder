CLASS Solution
    FUNCTION findShortestWay(maze, ball, hole)
        SET directions TO list of (0, 1, 'r'), (1, 0, 'd'), (0, -1, 'l'), (-1, 0, 'u')
        SET m TO number of rows in maze
        SET n TO number of columns in maze
        SET start TO tuple of ball coordinates
        SET hole TO tuple of hole coordinates
        SET pq TO priority queue containing (0, empty string, start)
        SET visited TO empty set

        WHILE pq is not empty
            EXTRACT smallest (dist, path, (x, y)) from pq

            IF (x, y) equals hole
                RETURN path

            IF (x, y) is in visited
                CONTINUE to next iteration

            ADD (x, y) to visited

            FOR each (dx, dy, direction) IN directions
                SET new_x TO x
                SET new_y TO y
                SET count TO 0

                WHILE new_x + dx is within 0 and m-1 AND new_y + dy is within 0 and n-1 AND maze at (new_x + dx, new_y + dy) equals 0
                    INCREMENT new_x BY dx
                    INCREMENT new_y BY dy
                    INCREMENT count BY 1

                    IF (new_x, new_y) equals hole
                        BREAK out of WHILE

                IF (new_x, new_y) not in visited
                    ADD (dist + count, path concatenated with direction, (new_x, new_y)) to pq

        RETURN "impossible"
    END FUNCTION
END CLASS
