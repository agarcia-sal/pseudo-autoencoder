CLASS Solution
    FUNCTION decodeString(s)
        SET stack TO empty list
        SET a TO empty string
        SET d TO empty string
        FOR each char IN s
            IF char IS a digit
                APPEND char TO d
            ELSE IF char IS a letter
                APPEND char TO a
            ELSE IF char IS '['
                APPEND tuple (a, d converted to integer) TO stack
                SET a TO empty string
                SET d TO empty string
            ELSE IF char IS ']'
                SET p, n TO pop the last element FROM stack
                SET a TO p concatenated WITH a repeated n times
        RETURN a
    END FUNCTION
END CLASS
