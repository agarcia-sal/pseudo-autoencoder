CLASS Solution
    FUNCTION threeSumSmaller(nums, target)
        SORT nums
        SET count TO 0
        SET n TO LENGTH OF nums
        
        FOR i FROM 0 TO n MINUS 3
            SET left TO i PLUS 1
            SET right TO n MINUS 1
            WHILE left LESS THAN right
                SET current_sum TO nums[i] PLUS nums[left] PLUS nums[right]
                IF current_sum LESS THAN target
                    INCREMENT count BY right MINUS left
                    INCREMENT left BY 1
                ELSE
                    DECREMENT right BY 1
        RETURN count
    END FUNCTION
END CLASS
