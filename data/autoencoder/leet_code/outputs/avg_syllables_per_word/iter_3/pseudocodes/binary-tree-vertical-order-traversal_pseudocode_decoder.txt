CLASS TreeNode
    FUNCTION __init__(self, val=0, left=None, right=None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO TreeNode(values[0])
    SET i TO 1
    SET queue TO empty deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO REMOVE FIRST ELEMENT FROM queue
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION verticalOrder(self, root)
        IF root IS None
            RETURN empty list
        SET column_table TO dictionary WITH DEFAULT list
        SET queue TO deque CONTAINING (root, 0)
        SET min_column TO 0
        SET max_column TO 0
        WHILE queue IS NOT EMPTY
            SET node, column_index TO REMOVE FIRST ELEMENT FROM queue
            IF node IS NOT None
                APPEND node.val TO column_table[column_index]
                SET min_column TO MINIMUM OF min_column AND column_index
                SET max_column TO MAXIMUM OF max_column AND column_index
                APPEND (node.left, column_index - 1) TO queue
                APPEND (node.right, column_index + 1) TO queue
        RETURN LIST OF column_table[x] FOR x IN RANGE FROM min_column TO max_column INCLUSIVE
    END FUNCTION
END CLASS
