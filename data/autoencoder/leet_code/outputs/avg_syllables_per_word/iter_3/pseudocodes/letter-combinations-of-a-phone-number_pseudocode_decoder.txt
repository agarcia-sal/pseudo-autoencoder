CLASS Solution
    FUNCTION letterCombinations(digits)
        IF digits IS EMPTY
            RETURN EMPTY LIST
        END IF

        SET phone_map TO DICTIONARY WITH
            '2' MAPPED TO 'abc'
            '3' MAPPED TO 'def'
            '4' MAPPED TO 'ghi'
            '5' MAPPED TO 'jkl'
            '6' MAPPED TO 'mno'
            '7' MAPPED TO 'pqrs'
            '8' MAPPED TO 'tuv'
            '9' MAPPED TO 'wxyz'

        FUNCTION backtrack(combination, next_digits)
            IF LENGTH OF next_digits IS 0
                APPEND combination TO output
            ELSE
                FOR EACH letter IN phone_map[next_digits[0]]
                    CALL backtrack(combination CONCATENATED WITH letter, next_digits FROM INDEX 1 TO END)
                END FOR
            END IF
        END FUNCTION

        SET output TO EMPTY LIST
        CALL backtrack WITH EMPTY STRING AND digits
        RETURN output
    END FUNCTION
END CLASS
