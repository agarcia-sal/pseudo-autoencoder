CLASS Solution
    FUNCTION primePalindrome(n)
        FUNCTION is_prime(num)
            IF num < 2 THEN
                RETURN False
            IF num IS 2 OR num IS 3 THEN
                RETURN True
            IF num MODULO 2 IS 0 OR num MODULO 3 IS 0 THEN
                RETURN False
            SET i TO 5
            WHILE i * i <= num
                IF num MODULO i IS 0 OR num MODULO (i + 2) IS 0 THEN
                    RETURN False
                INCREMENT i BY 6
            RETURN True
        END FUNCTION

        FUNCTION is_palindrome(num)
            RETURN string representation of num IS EQUAL TO reverse of string representation of num
        END FUNCTION

        IF n <= 2 THEN
            RETURN 2

        IF n >= 10 AND n <= 11 THEN
            RETURN 11

        WHILE True
            IF n MODULO 2 IS 0 THEN
                INCREMENT n BY 1
                CONTINUE
            IF is_palindrome(n) AND is_prime(n) THEN
                RETURN n
            IF LENGTH of string representation of n MODULO 2 IS 0 THEN
                SET n TO 10 RAISED TO LENGTH of string representation of n PLUS 1
            ELSE
                INCREMENT n BY 2
    END FUNCTION
END CLASS
