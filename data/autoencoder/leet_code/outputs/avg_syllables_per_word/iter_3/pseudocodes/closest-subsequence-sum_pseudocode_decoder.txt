CLASS Solution
    FUNCTION minAbsDifference(nums: List[int], goal: int) -> int
        SET n TO LENGTH OF nums
        SET mid TO n DIVIDED BY 2
        
        SET left_sums TO LIST CONTAINING 0
        FOR each num IN SLICE OF nums FROM START TO mid
            SET new_sums TO EMPTY LIST
            FOR each s IN left_sums
                APPEND s PLUS num TO new_sums
            EXTEND left_sums BY new_sums
        
        SET right_sums TO LIST CONTAINING 0
        FOR each num IN SLICE OF nums FROM mid TO END
            SET new_sums TO EMPTY LIST
            FOR each s IN right_sums
                APPEND s PLUS num TO new_sums
            EXTEND right_sums BY new_sums
        
        SORT right_sums
        
        SET min_diff TO INFINITY
        
        FOR each left_sum IN left_sums
            SET target TO goal MINUS left_sum
            SET pos TO POSITION TO INSERT target IN right_sums USING BINARY SEARCH
            
            IF pos IS LESS THAN LENGTH OF right_sums
                SET min_diff TO MINIMUM OF min_diff AND ABSOLUTE VALUE OF (left_sum PLUS right_sums[pos] MINUS goal)
            IF pos IS GREATER THAN 0
                SET min_diff TO MINIMUM OF min_diff AND ABSOLUTE VALUE OF (left_sum PLUS right_sums[pos MINUS 1] MINUS goal)
        
        RETURN min_diff
    END FUNCTION
END CLASS
