CLASS Solution
    FUNCTION findLength(nums1: List of Integer, nums2: List of Integer) RETURNS Integer
        SET dp TO 2D list of zeros with dimensions (LENGTH OF nums1 + 1) by (LENGTH OF nums2 + 1)
        SET max_length TO 0
        FOR i FROM 1 TO LENGTH OF nums1
            FOR j FROM 1 TO LENGTH OF nums2
                IF nums1 AT (i - 1) EQUALS nums2 AT (j - 1) THEN
                    SET dp AT [i][j] TO dp AT [i - 1][j - 1] + 1
                    IF dp AT [i][j] GREATER THAN max_length THEN
                        SET max_length TO dp AT [i][j]
        RETURN max_length
    END FUNCTION
END CLASS
