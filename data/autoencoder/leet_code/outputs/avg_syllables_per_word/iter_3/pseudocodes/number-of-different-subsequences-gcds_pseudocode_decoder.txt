CLASS Solution
    FUNCTION countDifferentSubsequenceGCDs(nums)
        SET max_num TO maximum value in nums
        SET num_set TO set of nums
        SET count TO 0
        FOR g FROM 1 TO max_num
            SET gcd_val TO 0
            FOR m FROM g TO max_num STEP g
                IF m IS IN num_set
                    SET gcd_val TO gcd of gcd_val AND m
                    IF gcd_val EQUALS g
                        INCREMENT count BY 1
                        BREAK inner loop
        RETURN count
    END FUNCTION
END CLASS
