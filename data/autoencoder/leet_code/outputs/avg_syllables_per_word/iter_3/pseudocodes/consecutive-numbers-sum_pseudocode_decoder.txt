CLASS Solution
    FUNCTION consecutiveNumbersSum(n)
        SET count TO 0
        FOR k FROM 1 TO n
            IF (2 * n) MODULO k EQUALS 0 THEN
                SET m TO ((2 * n DIVIDED BY k) - k + 1) DIVIDED BY 2
                IF m GREATER THAN 0 AND m EQUALS INTEGER VALUE OF m THEN
                    INCREMENT count BY 1
                END IF
            END IF
        END FOR
        RETURN count
    END FUNCTION
END CLASS
