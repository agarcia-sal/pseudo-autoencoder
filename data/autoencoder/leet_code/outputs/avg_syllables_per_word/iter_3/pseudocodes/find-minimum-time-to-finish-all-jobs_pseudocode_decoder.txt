CLASS Solution
    FUNCTION minimumTimeRequired(jobs, k)
        SET n TO LENGTH OF jobs
        SET subset_sums TO LIST OF SIZE (2 TO THE POWER n) WITH ALL ZEROS
        FOR i FROM 0 TO n - 1
            FOR mask FROM 0 TO (2 TO THE POWER i) - 1
                SET subset_sums[mask OR (1 LEFT SHIFTED BY i)] TO subset_sums[mask] + jobs[i]
        
        FUNCTION dp(k, mask) WITH MEMOIZATION
            IF k EQUALS 1 THEN
                RETURN subset_sums[mask]
            SET res TO infinity
            SET submask TO mask
            WHILE submask IS NOT ZERO
                SET res TO MINIMUM OF res AND MAXIMUM OF dp(k - 1, mask XOR submask) AND subset_sums[submask]
                SET submask TO (submask - 1) AND mask
            RETURN res
        
        RETURN dp(k, (1 LEFT SHIFTED BY n) - 1)
    END FUNCTION
END CLASS
