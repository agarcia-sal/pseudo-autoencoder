CLASS Solution
    FUNCTION earliestFullBloom(plantTime, growTime)
        SET tasks TO list of pairs from plantTime and growTime sorted by second element descending
        SET current_day TO 0
        SET earliest_bloom TO 0
        FOR each plant, grow IN tasks
            INCREMENT current_day BY plant
            SET bloom_day TO current_day PLUS grow
            SET earliest_bloom TO maximum of earliest_bloom and bloom_day
        RETURN earliest_bloom
    END FUNCTION
END CLASS
