CLASS Trie
    FUNCTION __init__()
        SET self.children TO list of size 26 filled with None
        SET self.ref TO -1
    END FUNCTION

    FUNCTION insert(w, ref)
        SET node TO self
        FOR each character c IN w
            SET idx TO numeric position of c relative to 'a'
            IF node.children[idx] IS None
                SET node.children[idx] TO new Trie instance
            SET node TO node.children[idx]
        SET node.ref TO ref
    END FUNCTION
END CLASS

CLASS Solution
    FUNCTION findWords(board, words)
        FUNCTION dfs(node, i, j)
            SET idx TO numeric position of board[i][j] relative to 'a'
            IF node.children[idx] IS None
                RETURN
            SET node TO node.children[idx]
            IF node.ref IS greater or equal to 0
                APPEND words[node.ref] TO ans
                SET node.ref TO -1
            SET c TO board[i][j]
            SET board[i][j] TO '#'
            FOR each pair (a, b) IN consecutive pairs of (-1, 0, 1, 0, -1)
                SET x TO i PLUS a
                SET y TO j PLUS b
                IF x IS within board row bounds AND y IS within board column bounds AND board[x][y] IS NOT '#'
                    CALL dfs(node, x, y)
            SET board[i][j] TO c
        END FUNCTION

        SET tree TO new Trie instance
        FOR each index i and word w IN words
            CALL tree.insert(w, i)
        SET m TO number of rows in board
        SET n TO number of columns in board
        SET ans TO empty list
        FOR i FROM 0 TO m MINUS 1
            FOR j FROM 0 TO n MINUS 1
                CALL dfs(tree, i, j)
        RETURN ans
    END FUNCTION
END CLASS
