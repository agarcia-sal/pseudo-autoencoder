CLASS Solution
    FUNCTION isInterleave(s1, s2, s3)
        IF LENGTH OF s1 PLUS LENGTH OF s2 NOT EQUAL TO LENGTH OF s3 THEN
            RETURN False
        END IF

        SET dp TO LIST OF False WITH SIZE LENGTH OF s2 PLUS 1
        SET dp[0] TO True

        FOR i FROM 0 TO LENGTH OF s1 INCLUSIVE
            FOR j FROM 0 TO LENGTH OF s2 INCLUSIVE
                IF i GREATER THAN 0 THEN
                    SET dp[j] TO dp[j] AND (CHARACTER AT s1 POSITION i MINUS 1 EQUAL TO CHARACTER AT s3 POSITION i PLUS j MINUS 1)
                END IF
                IF j GREATER THAN 0 THEN
                    SET dp[j] TO dp[j] OR (dp[j MINUS 1] AND (CHARACTER AT s2 POSITION j MINUS 1 EQUAL TO CHARACTER AT s3 POSITION i PLUS j MINUS 1))
                END IF
            END FOR
        END FOR

        RETURN dp[LENGTH OF s2]
    END FUNCTION
END CLASS
