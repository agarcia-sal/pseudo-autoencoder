CLASS Solution
    FUNCTION findStrobogrammatic(n)
        FUNCTION dfs(u)
            IF u EQUALS 0 THEN
                RETURN list containing empty string
            END IF
            IF u EQUALS 1 THEN
                RETURN list containing '0', '1', and '8'
            END IF
            SET ans TO empty list
            FOR each v IN dfs(u MINUS 2) DO
                FOR each pair (l, r) IN [('1', '1'), ('8', '8'), ('6', '9'), ('9', '6')] DO
                    APPEND concatenation of l, v, and r TO ans
                END FOR
                IF u NOT EQUAL TO n THEN
                    APPEND concatenation of '0', v, and '0' TO ans
                END IF
            END FOR
            RETURN ans
        END FUNCTION
        RETURN dfs(n)
    END FUNCTION
END CLASS
