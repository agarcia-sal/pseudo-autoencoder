CLASS Solution
    FUNCTION maxSlidingWindow(nums, k)
        IF nums IS EMPTY OR k EQUALS 0
            RETURN EMPTY LIST
        
        SET deq TO EMPTY DEQUE
        SET result TO EMPTY LIST
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 1
            IF deq IS NOT EMPTY AND deq FIRST ELEMENT LESS THAN i MINUS k PLUS 1
                REMOVE FIRST ELEMENT FROM deq
            
            WHILE deq IS NOT EMPTY AND nums AT deq LAST ELEMENT LESS THAN nums AT i
                REMOVE LAST ELEMENT FROM deq
            
            APPEND i TO deq
            
            IF i GREATER THAN OR EQUAL TO k MINUS 1
                APPEND nums AT deq FIRST ELEMENT TO result
        
        RETURN result
    END FUNCTION
END CLASS
