CLASS Solution
    FUNCTION isOneEditDistance(s, t)
        IF LENGTH OF s > LENGTH OF t THEN
            RETURN CALL isOneEditDistance(t, s)
        SET diff TO LENGTH OF t MINUS LENGTH OF s
        IF diff > 1 THEN
            RETURN False
        IF diff == 0 THEN
            SET count TO 0
            FOR EACH index FROM 0 TO LENGTH OF s MINUS 1
                IF CHARACTER AT s[index] IS NOT EQUAL TO CHARACTER AT t[index] THEN
                    INCREMENT count BY 1
            RETURN count EQUALS 1
        ELSE
            FOR EACH index FROM 0 TO LENGTH OF s MINUS 1
                IF CHARACTER AT s[index] IS NOT EQUAL TO CHARACTER AT t[index] THEN
                    RETURN SUBSTRING OF s FROM index TO END EQUALS SUBSTRING OF t FROM index + 1 TO END
            RETURN True
    END FUNCTION
END CLASS
