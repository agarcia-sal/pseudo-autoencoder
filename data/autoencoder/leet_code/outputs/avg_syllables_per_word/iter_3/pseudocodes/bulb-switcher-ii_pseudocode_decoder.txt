CLASS Solution
    FUNCTION flipLights(n, presses)
        SET n TO minimum of n and 6
        IF presses IS 0
            RETURN 1
        IF n IS 1
            RETURN 2
        IF n IS 2
            IF presses IS 1
                RETURN 3
            ELSE
                RETURN 4
        IF presses IS 1
            RETURN 4
        IF presses IS 2
            RETURN 7
        RETURN 8
    END FUNCTION
END CLASS
