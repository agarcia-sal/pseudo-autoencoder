CLASS Solution
    FUNCTION numTilings(n)
        SET MOD TO 10 TO THE POWER 9 PLUS 7
        
        SET dp TO list of zeros with length n PLUS 1
        SET dp1 TO list of zeros with length n PLUS 1
        
        SET dp[0] TO 1
        SET dp[1] TO 1
        
        IF n GREATER OR EQUAL TO 2 THEN
            SET dp[2] TO 2
            SET dp1[2] TO 1
        
        FOR i FROM 3 TO n INCLUSIVE
            SET dp[i] TO (dp[i MINUS 1] PLUS dp[i MINUS 2] PLUS 2 TIMES dp1[i MINUS 1]) MODULO MOD
            SET dp1[i] TO (dp[i MINUS 2] PLUS dp1[i MINUS 1]) MODULO MOD
        
        RETURN dp[n]
    END FUNCTION
END CLASS
