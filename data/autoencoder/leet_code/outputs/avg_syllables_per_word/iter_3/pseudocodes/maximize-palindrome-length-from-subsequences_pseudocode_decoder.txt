CLASS Solution
    FUNCTION longestPalindrome(word1, word2)
        SET s TO concatenation of word1 and word2
        SET n TO length of s
        SET dp TO two-dimensional list of size n by n filled with 0
        
        FOR i FROM 0 TO n - 1
            SET dp[i][i] TO 1
        
        SET max_len TO 0
        FOR length FROM 2 TO n
            FOR i FROM 0 TO n - length
                SET j TO i + length - 1
                IF s[i] EQUALS s[j]
                    SET dp[i][j] TO dp[i + 1][j - 1] PLUS 2
                    IF i LESS THAN length of word1 AND j GREATER THAN OR EQUAL TO length of word1
                        SET max_len TO maximum of max_len and dp[i][j]
                ELSE
                    SET dp[i][j] TO maximum of dp[i + 1][j] and dp[i][j - 1]
        
        RETURN max_len
    END FUNCTION
END CLASS
