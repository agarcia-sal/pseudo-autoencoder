CLASS Solution
    FUNCTION diffWaysToCompute(expression)
        FUNCTION compute(left, right, operator)
            SET results TO empty list
            FOR each l IN left
                FOR each r IN right
                    IF operator IS '+'
                        APPEND l PLUS r TO results
                    ELSE IF operator IS '-'
                        APPEND l MINUS r TO results
                    ELSE IF operator IS '*'
                        APPEND l TIMES r TO results
            RETURN results
        END FUNCTION

        FUNCTION helper(sub_expr)
            IF sub_expr CONTAINS ONLY DIGITS
                RETURN list containing integer value of sub_expr
            SET results TO empty list
            FOR i FROM 0 TO LENGTH OF sub_expr MINUS 1
                IF sub_expr[i] IS IN set of operators "+", "-", "*"
                    SET left_results TO helper(sub_expr FROM start TO i EXCLUDING i)
                    SET right_results TO helper(sub_expr FROM i+1 TO end)
                    SET combined TO compute(left_results, right_results, sub_expr[i])
                    APPEND all elements of combined TO results
            RETURN results
        END FUNCTION

        RETURN helper(expression)
    END FUNCTION
END CLASS
