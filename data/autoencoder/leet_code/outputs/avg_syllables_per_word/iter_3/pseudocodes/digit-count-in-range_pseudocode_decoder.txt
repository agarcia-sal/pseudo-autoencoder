CLASS Solution
    FUNCTION digitsCount(d, low, high)
        FUNCTION count_digit(n, digit)
            SET count TO 0
            SET power_of_10 TO 1
            SET current TO n
            WHILE current GREATER THAN 0
                SET last TO current MODULO 10
                SET current TO current DIVIDED BY 10
                SET full_cycles TO current
                IF digit EQUALS 0
                    DECREMENT full_cycles BY 1
                SET count TO count PLUS full_cycles TIMES power_of_10
                IF last GREATER THAN digit
                    SET count TO count PLUS power_of_10
                ELSE IF last EQUALS digit
                    SET count TO count PLUS (n MODULO power_of_10) PLUS 1
                SET power_of_10 TO power_of_10 TIMES 10
            RETURN count
        END FUNCTION

        RETURN count_digit(high, d) MINUS count_digit(low MINUS 1, d)
    END FUNCTION
END CLASS
