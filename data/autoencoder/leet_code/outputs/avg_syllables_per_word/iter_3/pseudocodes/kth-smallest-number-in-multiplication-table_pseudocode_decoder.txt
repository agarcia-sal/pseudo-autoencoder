CLASS Solution
    FUNCTION findKthNumber(m, n, k)
        FUNCTION enough(x)
            SET count TO 0
            FOR i FROM 1 TO m
                INCREMENT count BY MINIMUM OF (x DIVIDED BY i) AND n
            RETURN count GREATER THAN OR EQUAL TO k
        END FUNCTION

        SET low TO 1
        SET high TO m MULTIPLIED BY n

        WHILE low LESS THAN high
            SET mid TO (low PLUS high) DIVIDED BY 2
            IF enough(mid) IS TRUE
                SET high TO mid
            ELSE
                SET low TO mid PLUS 1
        RETURN low
    END FUNCTION
END CLASS
