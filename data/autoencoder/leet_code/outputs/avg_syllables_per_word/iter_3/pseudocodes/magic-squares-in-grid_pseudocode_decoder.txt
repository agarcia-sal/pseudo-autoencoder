CLASS Solution
    FUNCTION numMagicSquaresInside(grid)
        FUNCTION is_magic(i, j)
            IF grid[i][j] IS NOT 5 THEN
                RETURN False
            SET numbers TO empty set
            FOR x FROM i-1 TO i+1
                FOR y FROM j-1 TO j+1
                    IF grid[x][y] LESS THAN 1 OR grid[x][y] GREATER THAN 9 OR grid[x][y] IN numbers THEN
                        RETURN False
                    ADD grid[x][y] TO numbers
            FOR start FROM i-1 TO i+1
                IF (grid[start][j-1] + grid[start][j] + grid[start][j+1] NOT EQUAL TO 15) OR
                   (grid[i-1][start] + grid[i][start] + grid[i+1][start] NOT EQUAL TO 15) THEN
                    RETURN False
            IF (grid[i-1][j-1] + grid[i][j] + grid[i+1][j+1] NOT EQUAL TO 15) OR
               (grid[i-1][j+1] + grid[i][j] + grid[i+1][j-1] NOT EQUAL TO 15) THEN
                RETURN False
            RETURN True
        SET count TO 0
        SET rows TO LENGTH OF grid
        SET cols TO LENGTH OF grid[0]
        FOR i FROM 1 TO rows - 2
            FOR j FROM 1 TO cols - 2
                IF is_magic(i, j) THEN
                    INCREMENT count BY 1
        RETURN count
    END FUNCTION
END CLASS
