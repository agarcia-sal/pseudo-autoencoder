CLASS Solution
    FUNCTION longestAwesome(s)
        SET prefix_map TO dictionary with key 0 and value -1
        SET prefix_xor TO 0
        SET max_length TO 0
        
        FOR i, char IN ENUMERATE s
            SET digit TO integer value of char
            TOGGLE bit at position digit in prefix_xor
            
            IF prefix_xor EXISTS IN prefix_map THEN
                SET max_length TO maximum of max_length and (i MINUS prefix_map at prefix_xor)
            
            FOR j FROM 0 TO 9
                SET flipped TO prefix_xor TOGGLED at bit position j
                IF flipped EXISTS IN prefix_map THEN
                    SET max_length TO maximum of max_length and (i MINUS prefix_map at flipped)
            
            IF prefix_xor NOT EXISTS IN prefix_map THEN
                SET prefix_map at prefix_xor TO i
        
        RETURN max_length
    END FUNCTION
END CLASS
