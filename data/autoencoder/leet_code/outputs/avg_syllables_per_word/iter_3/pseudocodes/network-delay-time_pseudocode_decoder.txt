CLASS Solution
    FUNCTION networkDelayTime(times: LIST OF LIST OF INT, n: INT, k: INT) RETURNS INT
        SET graph TO DICTIONARY WITH KEYS FROM 1 TO n MAPPING TO EMPTY LISTS
        FOR EACH u, v, w IN times
            APPEND (v, w) TO graph[u]
        
        SET dist TO DICTIONARY WITH KEYS FROM 1 TO n MAPPING TO INFINITY
        SET dist[k] TO 0
        
        SET min_heap TO LIST CONTAINING (0, k)
        
        WHILE min_heap IS NOT EMPTY
            EXTRACT current_dist, node FROM min_heap USING MINIMUM HEAP POP
            
            IF current_dist GREATER THAN dist[node]
                CONTINUE TO NEXT ITERATION
            
            FOR EACH neighbor, weight IN graph[node]
                SET distance TO current_dist PLUS weight
                
                IF distance LESS THAN dist[neighbor]
                    SET dist[neighbor] TO distance
                    PUSH (distance, neighbor) TO min_heap AS MINIMUM HEAP
        
        SET max_time TO MAXIMUM VALUE IN dist
        
        IF max_time EQUALS INFINITY
            RETURN -1
        ELSE
            RETURN max_time
    END FUNCTION
END CLASS
