CLASS Solution
    FUNCTION minimumDeviation(nums: List[int]) RETURNS int
        SET max_heap TO empty list
        SET min_value TO positive infinity
        FOR each num IN nums
            IF num MODULO 2 EQUALS 1
                SET num TO num MULTIPLIED BY 2
            END IF
            IF num LESS THAN min_value
                SET min_value TO num
            END IF
            ADD negative num TO max_heap
        END FOR

        SET min_deviation TO positive infinity

        WHILE max_heap FIRST ELEMENT MODULO 2 EQUALS 0
            SET max_value TO NEGATIVE OF REMOVED FIRST ELEMENT FROM max_heap
            IF max_value MINUS min_value LESS THAN min_deviation
                SET min_deviation TO max_value MINUS min_value
            END IF
            SET max_value TO max_value DIVIDED BY 2
            IF max_value LESS THAN min_value
                SET min_value TO max_value
            END IF
            ADD negative max_value TO max_heap
        END WHILE

        IF NEGATIVE OF FIRST ELEMENT IN max_heap MINUS min_value LESS THAN min_deviation
            SET min_deviation TO NEGATIVE OF FIRST ELEMENT IN max_heap MINUS min_value
        END IF

        RETURN min_deviation
    END FUNCTION
END CLASS
