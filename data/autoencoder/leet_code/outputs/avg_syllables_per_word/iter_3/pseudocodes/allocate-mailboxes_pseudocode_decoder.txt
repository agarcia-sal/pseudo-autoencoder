CLASS Solution
    FUNCTION minDistance(houses, k)
        SORT houses
        SET n TO LENGTH OF houses

        CREATE cost AS 2D LIST OF SIZE n BY n FILLED WITH 0
        FOR i FROM n - 1 DOWNTO 0
            FOR j FROM i + 1 TO n - 1
                SET cost[i][j] TO cost[i + 1][j - 1] PLUS houses[j] MINUS houses[i]

        CREATE dp AS 2D LIST OF SIZE (n + 1) BY (k + 1) FILLED WITH infinity
        SET dp[0][0] TO 0

        FOR i FROM 1 TO n
            FOR j FROM 1 TO k
                FOR p FROM 0 TO i - 1
                    SET dp[i][j] TO MINIMUM OF dp[i][j] AND (dp[p][j - 1] PLUS cost[p][i - 1])

        RETURN dp[n][k]
    END FUNCTION
END CLASS
