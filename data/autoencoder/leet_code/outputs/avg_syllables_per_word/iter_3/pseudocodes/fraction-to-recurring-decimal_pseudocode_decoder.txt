CLASS Solution
    FUNCTION fractionToDecimal(numerator: int, denominator: int) RETURNS str
        IF (numerator < 0) XOR (denominator < 0) THEN
            SET sign TO "-"
        ELSE
            SET sign TO ""
        END IF

        SET numerator TO absolute value of numerator
        SET denominator TO absolute value of denominator

        SET quotient, remainder TO result of dividing numerator by denominator with remainder
        SET integer_part TO string representation of quotient

        IF remainder EQUALS 0 THEN
            RETURN concatenation of sign and integer_part
        END IF

        SET seen_remainders TO empty dictionary
        SET fractional_part TO empty list

        WHILE remainder NOT EQUAL TO 0 DO
            IF remainder IS IN seen_remainders THEN
                SET start TO value from seen_remainders for remainder
                SET non_repeating TO concatenation of fractional_part elements from start index 0 up to start
                SET repeating TO concatenation of fractional_part elements from start index start to end
                RETURN concatenation of sign, integer_part, ".", non_repeating, "(", repeating, ")"
            END IF

            SET seen_remainders[remainder] TO current length of fractional_part

            SET remainder TO remainder multiplied BY 10
            SET quotient, remainder TO result of dividing remainder by denominator with remainder
            APPEND string representation of quotient TO fractional_part
        END WHILE

        RETURN concatenation of sign, integer_part, ".", concatenation of all elements in fractional_part
    END FUNCTION
END CLASS
