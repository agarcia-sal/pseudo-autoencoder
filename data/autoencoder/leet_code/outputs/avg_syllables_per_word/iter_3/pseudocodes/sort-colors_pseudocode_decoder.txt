CLASS Solution
    FUNCTION sortColors(nums)
        SET low TO 0
        SET mid TO 0
        SET high TO LENGTH OF nums MINUS 1

        WHILE mid <= high
            IF nums[mid] EQUALS 0
                SWAP nums[low] WITH nums[mid]
                INCREMENT low BY 1
                INCREMENT mid BY 1
            ELSE IF nums[mid] EQUALS 1
                INCREMENT mid BY 1
            ELSE
                SWAP nums[mid] WITH nums[high]
                DECREMENT high BY 1
