CLASS Solution
    FUNCTION minMoves(nums: List of Integer, k: Integer) RETURNS Integer
        SET positions TO EMPTY LIST
        FOR index FROM 0 TO LENGTH OF nums MINUS 1
            IF nums[index] EQUALS 1
                APPEND index TO positions
        END FOR

        FUNCTION calculate_cost(start: Integer, end: Integer) RETURNS Integer
            SET mid TO (start + end) DIVIDED BY 2
            SET median TO positions[mid]
            SET cost TO 0
            FOR i FROM start TO end MINUS 1
                INCREMENT cost BY ABSOLUTE VALUE OF (positions[i] MINUS (median MINUS (mid MINUS i)))
            END FOR
            RETURN cost
        END FUNCTION

        SET min_cost TO INFINITY
        FOR i FROM 0 TO LENGTH OF positions MINUS k
            SET current_cost TO calculate_cost(i, i + k)
            IF current_cost LESS THAN min_cost
                SET min_cost TO current_cost
        END FOR

        RETURN min_cost
    END FUNCTION
END CLASS
