CLASS Solution
    FUNCTION countPrimes(n)
        IF n <= 2 THEN
            RETURN 0
        END IF

        SET is_prime TO list of True values of length n
        SET is_prime[0] TO False
        SET is_prime[1] TO False

        FOR start FROM 2 TO integer square root of n INCLUSIVE
            IF is_prime[start] THEN
                FOR multiple FROM start * start TO n - 1 STEP start
                    SET is_prime[multiple] TO False
                END FOR
            END IF
        END FOR

        SET prime_count TO sum of all True values in is_prime
        RETURN prime_count
    END FUNCTION
END CLASS
