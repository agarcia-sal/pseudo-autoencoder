CLASS Solution
    FUNCTION minDistance(height, width, tree, squirrel, nuts)
        FUNCTION distance(p1, p2)
            SET diff_x TO ABSOLUTE VALUE OF (p1[0] MINUS p2[0])
            SET diff_y TO ABSOLUTE VALUE OF (p1[1] MINUS p2[1])
            RETURN diff_x PLUS diff_y
        END FUNCTION

        SET total_distance TO 0
        SET max_save TO NEGATIVE INFINITY

        FOR each nut IN nuts
            INCREMENT total_distance BY 2 TIMES distance(nut, tree)
            SET first_nut_distance TO distance(squirrel, nut) PLUS distance(nut, tree)
            SET later_nut_distance TO 2 TIMES distance(nut, tree)
            SET save TO later_nut_distance MINUS first_nut_distance
            SET max_save TO MAXIMUM OF max_save AND save
        END FOR

        RETURN total_distance MINUS max_save
    END FUNCTION
END CLASS
