CLASS Solution
    FUNCTION hIndex(citations)
        SET n TO LENGTH OF citations
        SET left TO 0
        SET right TO n MINUS 1
        SET h_index TO 0
        
        WHILE left LESS THAN OR EQUAL TO right
            SET mid TO left PLUS (right MINUS left) DIVIDED BY 2
            IF citations AT mid GREATER THAN OR EQUAL TO n MINUS mid
                SET h_index TO n MINUS mid
                SET right TO mid MINUS 1
            ELSE
                SET left TO mid PLUS 1
        RETURN h_index
    END FUNCTION
END CLASS
