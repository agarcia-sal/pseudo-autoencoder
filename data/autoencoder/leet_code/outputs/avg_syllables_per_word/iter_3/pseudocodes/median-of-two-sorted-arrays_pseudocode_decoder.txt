CLASS Solution
    FUNCTION findMedianSortedArrays(nums1, nums2)
        IF LENGTH OF nums1 > LENGTH OF nums2
            SET temp TO nums1
            SET nums1 TO nums2
            SET nums2 TO temp
        SET x TO LENGTH OF nums1
        SET y TO LENGTH OF nums2
        SET low TO 0
        SET high TO x

        WHILE low <= high
            SET partitionX TO (low + high) DIVIDED BY 2
            SET partitionY TO ((x + y + 1) DIVIDED BY 2) MINUS partitionX

            IF partitionX == 0
                SET maxLeftX TO NEGATIVE_INFINITY
            ELSE
                SET maxLeftX TO nums1 AT (partitionX - 1)
            IF partitionX == x
                SET minRightX TO POSITIVE_INFINITY
            ELSE
                SET minRightX TO nums1 AT partitionX

            IF partitionY == 0
                SET maxLeftY TO NEGATIVE_INFINITY
            ELSE
                SET maxLeftY TO nums2 AT (partitionY - 1)
            IF partitionY == y
                SET minRightY TO POSITIVE_INFINITY
            ELSE
                SET minRightY TO nums2 AT partitionY

            IF maxLeftX <= minRightY AND maxLeftY <= minRightX
                IF (x + y) MODULO 2 == 0
                    RETURN (MAXIMUM OF maxLeftX AND maxLeftY PLUS MINIMUM OF minRightX AND minRightY) DIVIDED BY 2
                ELSE
                    RETURN MAXIMUM OF maxLeftX AND maxLeftY
            ELSE IF maxLeftX > minRightY
                SET high TO partitionX MINUS 1
            ELSE
                SET low TO partitionX PLUS 1
    END FUNCTION
END CLASS
