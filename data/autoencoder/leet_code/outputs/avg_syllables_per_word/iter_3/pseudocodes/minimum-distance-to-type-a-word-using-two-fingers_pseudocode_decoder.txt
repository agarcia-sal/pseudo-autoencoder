CLASS Solution
    FUNCTION minimumDistance(word)
        FUNCTION distance(c1, c2)
            IF c1 IS None
                RETURN 0
            SET x1, y1 TO divmod(ASCII value of c1 MINUS ASCII value of 'A', 6)
            SET x2, y2 TO divmod(ASCII value of c2 MINUS ASCII value of 'A', 6)
            RETURN absolute value of (x1 MINUS x2) PLUS absolute value of (y1 MINUS y2)
        END FUNCTION

        FUNCTION dp(i, f1, f2)
            IF i EQUALS LENGTH OF word
                RETURN 0
            SET c TO word AT position i
            SET d1 TO distance(f1, c) PLUS dp(i PLUS 1, c, f2)
            SET d2 TO distance(f2, c) PLUS dp(i PLUS 1, f1, c)
            RETURN minimum of d1 AND d2
        END FUNCTION

        RETURN dp(0, None, None)
    END FUNCTION
END CLASS
