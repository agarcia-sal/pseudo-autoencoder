CLASS Solution
    FUNCTION reverseWords(s: List of characters)
        CALL reverse on s
        SET start TO 0
        FOR end FROM 0 TO LENGTH OF s MINUS 1
            IF s[end] IS EQUAL TO space character
                SET the substring of s from start TO end MINUS 1 TO reversed substring of s from start TO end MINUS 1
                SET start TO end PLUS 1
        SET the substring of s from start TO END TO reversed substring of s from start TO END
    END FUNCTION
END CLASS
