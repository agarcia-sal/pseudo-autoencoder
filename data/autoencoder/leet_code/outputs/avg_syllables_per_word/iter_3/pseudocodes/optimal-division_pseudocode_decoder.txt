CLASS Solution
    FUNCTION optimalDivision(nums)
        IF LENGTH OF nums IS 1
            RETURN STRING REPRESENTATION OF nums[0]
        IF LENGTH OF nums IS 2
            RETURN STRING FORMATTED AS nums[0] "/" nums[1]
        RETURN STRING FORMATTED AS nums[0] "/" "(" JOINED BY "/" OF STRING REPRESENTATIONS OF nums FROM INDEX 1 TO END ")" 
    END FUNCTION
END CLASS
