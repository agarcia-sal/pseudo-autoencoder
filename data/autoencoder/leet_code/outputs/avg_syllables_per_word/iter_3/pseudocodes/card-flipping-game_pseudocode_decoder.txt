CLASS Solution
    FUNCTION flipgame(fronts, backs)
        SET same TO EMPTY SET
        FOR EACH pair (x, y) IN ZIP(fronts, backs)
            IF x EQUALS y THEN
                ADD x TO same
        SET all_numbers TO SET OF ALL ELEMENTS IN fronts AND backs
        SET good_numbers TO all_numbers MINUS same
        IF good_numbers IS NOT EMPTY THEN
            RETURN MINIMUM VALUE IN good_numbers
        ELSE
            RETURN 0
    END FUNCTION
END CLASS
