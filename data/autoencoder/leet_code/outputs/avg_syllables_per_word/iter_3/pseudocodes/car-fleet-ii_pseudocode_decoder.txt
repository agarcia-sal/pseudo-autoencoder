CLASS Solution
    FUNCTION getCollisionTimes(cars)
        SET n TO LENGTH OF cars
        SET result TO list of size n filled with -1.0
        SET stack TO empty list

        FOR i FROM n - 1 DOWNTO 0
            SET pos, speed TO cars[i]

            WHILE stack IS NOT empty
                SET next_index TO last element in stack
                SET next_pos, next_speed TO cars[next_index]

                IF speed <= next_speed THEN
                    REMOVE last element from stack
                    CONTINUE

                SET collision_time TO (next_pos - pos) DIVIDED BY (speed - next_speed)

                IF result[next_index] > 0 AND collision_time >= result[next_index] THEN
                    REMOVE last element from stack
                    CONTINUE

                SET result[i] TO collision_time
                BREAK

            APPEND i TO stack

        RETURN result
    END FUNCTION
END CLASS
