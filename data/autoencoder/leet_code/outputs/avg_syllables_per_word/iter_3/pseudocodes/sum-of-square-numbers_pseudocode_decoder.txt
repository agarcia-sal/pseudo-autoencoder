CLASS Solution
    FUNCTION judgeSquareSum(c)
        SET left TO 0
        SET right TO integer part of square root of c
        WHILE left IS LESS THAN OR EQUAL TO right
            SET current_sum TO left squared PLUS right squared
            IF current_sum EQUALS c THEN
                RETURN True
            ELSE IF current_sum IS LESS THAN c THEN
                INCREMENT left BY 1
            ELSE
                DECREMENT right BY 1
        RETURN False
    END FUNCTION
END CLASS
