CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS empty
        RETURN None
    END IF
    SET root TO TreeNode(values[0])
    SET i TO 1
    SET queue TO a new deque
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO queue.popleft()
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION kthSmallest(root: Optional[TreeNode], k: int) RETURNS int
        SET stack TO empty list
        SET current TO root
        SET count TO 0
        
        WHILE stack IS NOT empty OR current IS NOT None
            WHILE current IS NOT None
                APPEND current TO stack
                SET current TO current.left
            END WHILE
            SET current TO stack.pop()
            INCREMENT count BY 1
            IF count EQUALS k
                RETURN current.val
            END IF
            SET current TO current.right
        END WHILE
    END FUNCTION
END CLASS
