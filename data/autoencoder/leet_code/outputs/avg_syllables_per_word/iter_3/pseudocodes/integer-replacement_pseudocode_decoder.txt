CLASS Solution
    FUNCTION integerReplacement(n)
        SET count TO 0
        WHILE n IS NOT 1
            IF n MODULO 2 EQUALS 0
                SET n TO n DIVIDED BY 2
            ELSE
                IF n EQUALS 3
                    DECREMENT n BY 1
                ELSE IF (n + 1) MODULO 4 EQUALS 0
                    INCREMENT n BY 1
                ELSE
                    DECREMENT n BY 1
            INCREMENT count BY 1
        RETURN count
    END FUNCTION
END CLASS
