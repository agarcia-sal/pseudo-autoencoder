CLASS Solution
    FUNCTION maxValue(events, k)
        SORT events BY end day
        SET n TO LENGTH OF events
        INITIALIZE dp AS 2D LIST OF SIZE (n+1) BY (k+1) FILLED WITH 0

        FOR i FROM 1 TO n
            SET start, end, value TO events AT (i - 1)

            SET j TO FIND POSITION USING BINARY SEARCH WHERE end day IS LESS THAN start USING events AND KEY end day

            FOR l FROM 1 TO k
                SET dp[i][l] TO MAXIMUM OF dp[i - 1][l] AND (dp[j][l - 1] PLUS value)

        RETURN dp[n][k]
    END FUNCTION
END CLASS
