CLASS Solution
    FUNCTION findNumberOfLIS(nums)
        IF nums IS EMPTY
            RETURN 0
        SET n TO LENGTH OF nums
        SET lengths TO LIST OF SIZE n FILLED WITH 1
        SET counts TO LIST OF SIZE n FILLED WITH 1
        FOR i FROM 0 TO n - 1
            FOR j FROM 0 TO i - 1
                IF nums[i] > nums[j]
                    IF lengths[j] + 1 > lengths[i]
                        SET lengths[i] TO lengths[j] + 1
                        SET counts[i] TO counts[j]
                    ELSE IF lengths[j] + 1 == lengths[i]
                        INCREMENT counts[i] BY counts[j]
        SET longest TO MAXIMUM VALUE IN lengths
        SET total TO 0
        FOR i FROM 0 TO n - 1
            IF lengths[i] == longest
                INCREMENT total BY counts[i]
        RETURN total
    END FUNCTION
END CLASS
