CLASS Solution
    FUNCTION isTransformable(s, t)
        SET pos TO a default dictionary of empty lists
        FOR i, c IN s WITH INDEX
            ADD i TO the heap pos[int(c)]
        FOR i, c IN t WITH INDEX
            SET digit TO int(c)
            IF pos[digit] IS empty
                RETURN False
            FOR smaller FROM 0 TO digit MINUS 1
                IF pos[smaller] IS NOT empty AND smallest element in pos[smaller] IS LESS THAN smallest element in pos[digit]
                    RETURN False
            REMOVE the smallest element from the heap pos[digit]
        RETURN True
    END FUNCTION
END CLASS
