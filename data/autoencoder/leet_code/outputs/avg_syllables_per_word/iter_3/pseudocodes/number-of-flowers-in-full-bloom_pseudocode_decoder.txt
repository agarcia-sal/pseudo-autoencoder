CLASS Solution
    FUNCTION fullBloomFlowers(flowers: List OF List OF int, people: List OF int) RETURNS List OF int
        SET start_times TO SORTED LIST OF start times FROM each flower IN flowers
        SET end_times TO SORTED LIST OF end times FROM each flower IN flowers
        SET result TO EMPTY LIST
        FOR each person IN people
            SET started TO POSITION where person would be inserted to the right IN start_times
            SET ended TO POSITION where person would be inserted to the left IN end_times
            APPEND (started MINUS ended) TO result
        RETURN result
    END FUNCTION
END CLASS
