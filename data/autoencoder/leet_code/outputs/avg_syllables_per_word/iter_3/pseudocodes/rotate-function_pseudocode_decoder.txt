CLASS Solution
    FUNCTION maxRotateFunction(nums)
        SET n TO LENGTH OF nums
        IF n EQUALS 1 THEN
            RETURN 0
        END IF
        SET total_sum TO SUM OF nums
        SET current_sum TO 0
        FOR EACH index, num IN ENUMERATE nums
            INCREMENT current_sum BY index MULTIPLIED BY num
        END FOR
        SET max_value TO current_sum
        FOR k FROM 1 TO n MINUS 1
            SET current_sum TO current_sum PLUS total_sum MINUS n MULTIPLIED BY nums AT POSITION (n MINUS k)
            SET max_value TO THE GREATER OF max_value AND current_sum
        END FOR
        RETURN max_value
    END FUNCTION
END CLASS
