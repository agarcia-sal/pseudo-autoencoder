CLASS Solution
    FUNCTION maxArea(height: List of integers) RETURNS integer
        SET left TO 0
        SET right TO LENGTH OF height MINUS 1
        SET max_area TO 0
        WHILE left < right
            SET current_area TO MINIMUM OF height[left] AND height[right] MULTIPLIED BY (right MINUS left)
            SET max_area TO MAXIMUM OF max_area AND current_area
            IF height[left] < height[right]
                INCREMENT left BY 1
            ELSE
                DECREMENT right BY 1
        RETURN max_area
    END FUNCTION
END CLASS
