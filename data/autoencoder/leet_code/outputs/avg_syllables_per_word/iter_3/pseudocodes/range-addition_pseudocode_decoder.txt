CLASS Solution
    FUNCTION getModifiedArray(length, updates)
        SET diff TO list of zeros with size length + 1
        FOR each startIdx, endIdx, inc IN updates
            INCREMENT diff[startIdx] BY inc
            IF endIdx + 1 < length THEN
                DECREMENT diff[endIdx + 1] BY inc
        SET arr TO list of zeros with size length
        SET arr[0] TO diff[0]
        FOR i FROM 1 TO length - 1
            SET arr[i] TO arr[i - 1] + diff[i]
        RETURN arr
    END FUNCTION
END CLASS
