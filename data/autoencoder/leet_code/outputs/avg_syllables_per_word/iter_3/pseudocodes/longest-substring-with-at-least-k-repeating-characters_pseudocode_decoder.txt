CLASS Solution
    FUNCTION longestSubstring(s, k)
        IF LENGTH OF s LESS THAN k THEN
            RETURN 0
        END IF

        SET count TO frequency count of characters in s

        FOR each char IN count
            IF count[char] LESS THAN k THEN
                SET substrings TO s split by char
                RETURN maximum value FROM longestSubstring(substring, k) FOR each substring IN substrings
            END IF
        END FOR

        RETURN LENGTH OF s
    END FUNCTION
END CLASS
