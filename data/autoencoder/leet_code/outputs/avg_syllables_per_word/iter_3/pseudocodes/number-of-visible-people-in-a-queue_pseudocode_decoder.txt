CLASS Solution
    FUNCTION canSeePersonsCount(heights)
        SET n TO LENGTH OF heights
        SET answer TO LIST OF 0 REPEATED n TIMES
        SET stack TO EMPTY LIST
        
        FOR i FROM 0 TO n MINUS 1
            WHILE stack IS NOT EMPTY AND heights VALUE AT stack LAST ELEMENT IS LESS THAN heights VALUE AT i
                SET top_index TO REMOVE LAST ELEMENT FROM stack
                INCREMENT answer VALUE AT top_index BY 1
            END WHILE
            IF stack IS NOT EMPTY
                INCREMENT answer VALUE AT stack LAST ELEMENT BY 1
            END IF
            APPEND i TO stack
        END FOR
        
        RETURN answer
    END FUNCTION
END CLASS
