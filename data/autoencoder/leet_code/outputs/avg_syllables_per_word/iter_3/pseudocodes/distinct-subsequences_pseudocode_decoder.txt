CLASS Solution
    FUNCTION numDistinct(s, t)
        SET m TO LENGTH OF s
        SET n TO LENGTH OF t

        SET dp TO 2D ARRAY OF SIZE (m + 1) BY (n + 1) FILLED WITH 0

        FOR i FROM 0 TO m
            SET dp[i][0] TO 1

        FOR i FROM 1 TO m
            FOR j FROM 1 TO n
                IF CHARACTER AT s POSITION (i - 1) EQUALS CHARACTER AT t POSITION (j - 1)
                    SET dp[i][j] TO dp[i - 1][j - 1] PLUS dp[i - 1][j]
                ELSE
                    SET dp[i][j] TO dp[i - 1][j]

        RETURN dp[m][n]
    END FUNCTION
END CLASS
