CLASS Solution
    FUNCTION isPossible(target: List of Integer) RETURNS Boolean
        SET max_heap TO list of negative values of each number in target
        CALL heapify on max_heap
        
        SET total_sum TO sum of values in target
        
        WHILE first element of max_heap IS LESS THAN -1
            SET largest TO negative of heappop from max_heap
            SET rest_sum TO total_sum MINUS largest
            
            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum IS 0
                RETURN False
            
            SET new_value TO largest MODULO rest_sum
            
            IF new_value IS 0 AND rest_sum IS NOT 1
                RETURN False
            
            SET total_sum TO rest_sum PLUS new_value
            heappush negative of new_value INTO max_heap
        
        RETURN True
    END FUNCTION
END CLASS
