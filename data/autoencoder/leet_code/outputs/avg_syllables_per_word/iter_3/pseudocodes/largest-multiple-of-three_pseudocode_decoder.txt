CLASS Solution
    FUNCTION largestMultipleOfThree(digits: List[int]) RETURNS str
        SET count TO frequency count of digits
        SET total_sum TO sum of digits
        SET remainder TO total_sum MODULO 3
        
        IF remainder EQUALS 1 THEN
            FOR each d IN [1, 4, 7]
                IF count[d] GREATER THAN 0 THEN
                    DECREMENT count[d] BY 1
                    BREAK
            ELSE
                FOR each d IN [2, 5, 8]
                    IF count[d] GREATER THAN 1 THEN
                        DECREMENT count[d] BY 2
                        BREAK
        
        IF remainder EQUALS 2 THEN
            FOR each d IN [2, 5, 8]
                IF count[d] GREATER THAN 0 THEN
                    DECREMENT count[d] BY 1
                    BREAK
            ELSE
                FOR each d IN [1, 4, 7]
                    IF count[d] GREATER THAN 1 THEN
                        DECREMENT count[d] BY 2
                        BREAK
        
        SET result TO empty list
        FOR d FROM 9 DOWNTO 0
            APPEND string representation of d repeated count[d] times TO result
        
        SET final_number TO concatenation of elements in result
        
        IF final_number IS NOT empty AND first character of final_number EQUALS '0' THEN
            RETURN '0'
        ELSE
            RETURN final_number
    END FUNCTION
END CLASS
