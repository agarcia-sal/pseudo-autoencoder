CLASS Solution
    FUNCTION wiggleSort(nums)
        SET less TO true
        FOR i FROM 0 TO LENGTH OF nums MINUS 2
            IF less IS true
                IF element at nums[i] IS GREATER THAN element at nums[i + 1]
                    SWAP element at nums[i] WITH element at nums[i + 1]
            ELSE
                IF element at nums[i] IS LESS THAN element at nums[i + 1]
                    SWAP element at nums[i] WITH element at nums[i + 1]
            SET less TO NOT less
    END FUNCTION
END CLASS
