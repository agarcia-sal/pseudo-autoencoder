CLASS Solution
    FUNCTION wordsTyping(sentence: List of strings, rows: integer, cols: integer) RETURNS integer
        SET s TO concatenate all words in sentence separated by spaces and add a trailing space
        SET total_length TO length of s
        SET position TO 0
        FOR each row FROM 1 TO rows
            INCREMENT position BY cols
            IF s character at (position MODULO total_length) IS a space
                INCREMENT position BY 1
            ELSE
                WHILE position GREATER THAN 0 AND s character at ((position - 1) MODULO total_length) IS NOT a space
                    DECREMENT position BY 1
        RETURN position DIVIDED BY total_length
    END FUNCTION
END CLASS
