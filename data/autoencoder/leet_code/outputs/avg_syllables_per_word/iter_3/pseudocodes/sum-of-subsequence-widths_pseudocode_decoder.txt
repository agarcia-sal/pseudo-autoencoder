CLASS Solution
    FUNCTION sumSubseqWidths(nums)
        SET MOD TO 1000000007
        SORT nums IN ASCENDING ORDER
        SET n TO LENGTH OF nums
        SET total_sum TO 0
        SET power TO LIST OF SIZE n FILLED WITH 1

        FOR i FROM 1 TO n - 1
            SET power[i] TO (power[i - 1] MULTIPLIED BY 2) MODULO MOD
        END FOR

        FOR i FROM 0 TO n - 1
            SET total_sum TO (total_sum PLUS nums[i] MULTIPLIED BY (power[i] MINUS power[n - i - 1])) MODULO MOD
        END FOR

        RETURN total_sum
    END FUNCTION
END CLASS
