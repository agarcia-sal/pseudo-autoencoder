CLASS Solution
    FUNCTION singleNonDuplicate(nums)
        SET left TO 0
        SET right TO LENGTH OF nums MINUS 1
        WHILE left < right
            SET mid TO left PLUS (right MINUS left) DIVIDED BY 2
            IF mid MODULO 2 EQUALS 1
                DECREMENT mid BY 1
            IF nums AT mid EQUALS nums AT (mid PLUS 1)
                SET left TO mid PLUS 2
            ELSE
                SET right TO mid
        RETURN nums AT left
    END FUNCTION
END CLASS
