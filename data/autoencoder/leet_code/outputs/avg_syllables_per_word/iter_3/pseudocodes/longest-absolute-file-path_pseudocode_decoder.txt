CLASS Solution
    FUNCTION lengthLongestPath(input: string) RETURNS integer
        SET lines TO SPLIT input BY newline
        SET stack TO empty list
        SET max_length TO 0

        FOR each line IN lines
            SET depth TO COUNT of tab characters IN line
            SET name TO substring of line STARTING FROM depth

            WHILE LENGTH OF stack IS GREATER THAN depth
                REMOVE last element FROM stack
            
            IF stack IS NOT empty
                SET current_length TO last element IN stack PLUS LENGTH OF name PLUS 1
            ELSE
                SET current_length TO LENGTH OF name
            
            IF name CONTAINS '.'
                SET max_length TO MAXIMUM OF max_length AND current_length
            ELSE
                APPEND current_length TO stack
        
        RETURN max_length
    END FUNCTION
END CLASS
