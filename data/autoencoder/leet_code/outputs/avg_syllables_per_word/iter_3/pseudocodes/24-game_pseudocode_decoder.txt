CLASS Solution
    FUNCTION judgePoint24(cards)
        FUNCTION solve(nums)
            IF LENGTH OF nums EQUALS 1 THEN
                RETURN absolute value of (nums[0] MINUS 24) LESS THAN 0.000001
            END IF

            FOR i FROM 0 TO LENGTH OF nums MINUS 1
                FOR j FROM 0 TO LENGTH OF nums MINUS 1
                    IF i NOT EQUAL TO j THEN
                        SET new_nums TO list of nums[k] WHERE k IS NOT i AND k IS NOT j

                        FOR each op IN (division, multiplication, addition, subtraction)
                            IF (op IS addition OR op IS multiplication) AND j GREATER THAN i THEN
                                CONTINUE to next op
                            END IF

                            IF op IS NOT division OR nums[j] NOT EQUAL TO 0 THEN
                                APPEND result of op(nums[i], nums[j]) TO new_nums
                                IF solve(new_nums) RETURNS True THEN
                                    RETURN True
                                END IF
                                REMOVE last element FROM new_nums
                            END IF
                        END FOR
                    END IF
                END FOR
            END FOR

            RETURN False
        END FUNCTION

        IMPORT operations division, multiplication, addition, subtraction
        RETURN solve(cards)
    END FUNCTION
END CLASS
