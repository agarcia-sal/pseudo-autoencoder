CLASS Solution
    FUNCTION minInterval(intervals, queries)
        SORT intervals BY start time
        SET sorted_queries TO sorted list of (query, original index) pairs from queries

        SET min_heap TO empty heap
        SET result TO list of size LENGTH OF queries filled with -1

        SET interval_index TO 0
        FOR each query, original_index IN sorted_queries
            WHILE interval_index LESS THAN LENGTH OF intervals AND intervals[interval_index][0] LESS THAN OR EQUAL TO query
                SET start TO intervals[interval_index][0]
                SET end TO intervals[interval_index][1]
                SET interval_size TO end MINUS start PLUS 1
                PUSH (interval_size, end) TO min_heap
                INCREMENT interval_index BY 1

            WHILE min_heap NOT EMPTY AND min_heap TOP element's end LESS THAN query
                POP from min_heap

            IF min_heap NOT EMPTY
                SET result[original_index] TO min_heap TOP element's interval_size

        RETURN result
    END FUNCTION
END CLASS
