CLASS TreeNode
    FUNCTION __init__(self, val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO TreeNode(values[0])
    SET i TO 1
    SET queue TO empty double-ended queue
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO REMOVE FIRST ELEMENT FROM queue
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION addOneRow(self, root: Optional[TreeNode], val: int, depth: int) -> Optional[TreeNode]
        IF depth IS EQUAL TO 1
            RETURN TreeNode(val, left = root)
        SET queue TO double-ended queue CONTAINING tuple (root, 1)
        WHILE queue IS NOT EMPTY
            SET node, level TO REMOVE FIRST ELEMENT FROM queue
            IF level IS EQUAL TO depth MINUS 1
                SET node.left TO TreeNode(val, left = node.left)
                SET node.right TO TreeNode(val, right = node.right)
            ELSE IF level IS LESS THAN depth MINUS 1
                IF node.left IS NOT None
                    APPEND (node.left, level PLUS 1) TO queue
                IF node.right IS NOT None
                    APPEND (node.right, level PLUS 1) TO queue
        RETURN root
    END FUNCTION
END CLASS
