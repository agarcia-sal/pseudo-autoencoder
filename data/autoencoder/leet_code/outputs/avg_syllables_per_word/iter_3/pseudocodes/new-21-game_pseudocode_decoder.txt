CLASS Solution
    FUNCTION new21Game(n, k, maxPts)
        FUNCTION dfs(i)
            IF i >= k THEN
                RETURN 1 IF i <= n ELSE 0
            END IF
            IF i == k - 1 THEN
                RETURN MINIMUM OF (n - k + 1) AND maxPts DIVIDED BY maxPts
            END IF
            RETURN dfs(i + 1) + (dfs(i + 1) - dfs(i + maxPts + 1)) DIVIDED BY maxPts
        END FUNCTION
        RETURN dfs(0)
    END FUNCTION
END CLASS
