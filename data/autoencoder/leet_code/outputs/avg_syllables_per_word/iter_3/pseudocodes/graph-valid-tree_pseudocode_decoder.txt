CLASS Solution
    FUNCTION validTree(n, edges)
        IF LENGTH OF edges IS NOT EQUAL TO n MINUS 1 THEN
            RETURN False
        END IF

        SET adj_list TO EMPTY DICTIONARY
        FOR i FROM 0 TO n MINUS 1
            SET adj_list[i] TO EMPTY LIST
        END FOR

        FOR EACH pair (a, b) IN edges
            APPEND b TO adj_list[a]
            APPEND a TO adj_list[b]
        END FOR

        SET visited TO EMPTY SET

        FUNCTION dfs(node)
            IF node IS IN visited THEN
                RETURN
            END IF
            ADD node TO visited
            FOR EACH neighbor IN adj_list[node]
                CALL dfs(neighbor)
            END FOR
        END FUNCTION

        CALL dfs(0)

        RETURN LENGTH OF visited IS EQUAL TO n
    END FUNCTION
END CLASS
