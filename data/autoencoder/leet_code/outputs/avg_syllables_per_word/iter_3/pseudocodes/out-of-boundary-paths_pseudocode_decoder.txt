CLASS Solution
    FUNCTION findPaths(m, n, maxMove, startRow, startColumn)
        SET MOD TO 1000000007
        SET directions TO [(0, 1), (0, -1), (1, 0), (-1, 0)]
        INITIALIZE dp AS 3D array WITH DIMENSIONS m x n x (maxMove + 1) FILLED WITH 0
        
        FOR k FROM 1 TO maxMove
            FOR i FROM 0 TO m - 1
                FOR j FROM 0 TO n - 1
                    FOR EACH (di, dj) IN directions
                        SET ni TO i + di
                        SET nj TO j + dj
                        IF ni < 0 OR ni >= m OR nj < 0 OR nj >= n THEN
                            SET dp[i][j][k] TO (dp[i][j][k] + 1) MODULO MOD
                        ELSE
                            SET dp[i][j][k] TO (dp[i][j][k] + dp[ni][nj][k - 1]) MODULO MOD
        
        RETURN dp[startRow][startColumn][maxMove]
    END FUNCTION
END CLASS
