CLASS TreeNode
    FUNCTION __init__(self, val=0, left=None, right=None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    END IF
    SET root TO NEW TreeNode WITH val values[0]
    SET i TO 1
    SET queue TO EMPTY deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO REMOVE FIRST ELEMENT FROM queue
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.left TO NEW TreeNode WITH val values[i]
            APPEND node.left TO queue
        END IF
        INCREMENT i BY 1
        IF i < LENGTH OF values AND values[i] IS NOT None
            SET node.right TO NEW TreeNode WITH val values[i]
            APPEND node.right TO queue
        END IF
        INCREMENT i BY 1
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION upsideDownBinaryTree(self, root: Optional[TreeNode]) -> Optional[TreeNode]
        IF root IS None OR root.left IS None
            RETURN root
        END IF
        SET new_root TO self.upsideDownBinaryTree(root.left)
        SET root.left.left TO root.right
        SET root.left.right TO root
        SET root.left TO None
        SET root.right TO None
        RETURN new_root
    END FUNCTION
END CLASS
