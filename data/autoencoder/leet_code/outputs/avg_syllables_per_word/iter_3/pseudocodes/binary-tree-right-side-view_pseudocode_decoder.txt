CLASS TreeNode
    FUNCTION __init__(val = 0, left = None, right = None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS EMPTY
        RETURN None
    SET root TO TreeNode(values[0])
    SET i TO 1
    SET queue TO empty deque
    APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO queue.popleft()
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None
            SET node.left TO TreeNode(values[i])
            APPEND node.left TO queue
        INCREMENT i BY 1
        IF i LESS THAN LENGTH OF values AND values[i] IS NOT None
            SET node.right TO TreeNode(values[i])
            APPEND node.right TO queue
        INCREMENT i BY 1
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
END FUNCTION

CLASS Solution
    FUNCTION rightSideView(root)
        IF root IS None
            RETURN empty list
        SET right_side_view TO empty list
        SET queue TO deque containing root
        WHILE queue IS NOT EMPTY
            SET level_length TO LENGTH OF queue
            FOR i FROM 0 TO level_length MINUS 1
                SET node TO queue.popleft()
                IF i EQUALS level_length MINUS 1
                    APPEND node.val TO right_side_view
                IF node.left IS NOT None
                    APPEND node.left TO queue
                IF node.right IS NOT None
                    APPEND node.right TO queue
        RETURN right_side_view
    END FUNCTION
END CLASS
