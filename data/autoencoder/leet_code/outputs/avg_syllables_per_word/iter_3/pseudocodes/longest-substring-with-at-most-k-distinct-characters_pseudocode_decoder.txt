CLASS Solution
    FUNCTION lengthOfLongestSubstringKDistinct(s, k)
        IF k EQUALS 0 THEN
            RETURN 0
        END IF
        
        SET char_map TO empty map
        SET left TO 0
        SET max_length TO 0
        
        FOR right FROM 0 TO LENGTH OF s MINUS 1 DO
            SET char_map[s[right]] TO right
            
            IF LENGTH OF char_map GREATER THAN k THEN
                SET leftmost_char TO CHARACTER IN char_map WITH MINIMUM INDEX VALUE
                SET left TO char_map[leftmost_char] PLUS 1
                REMOVE leftmost_char FROM char_map
            END IF
            
            SET max_length TO GREATER OF max_length AND (right MINUS left PLUS 1)
        END FOR
        
        RETURN max_length
    END FUNCTION
END CLASS
