CLASS Solution
    FUNCTION maxChunksToSorted(arr)
        SET max_so_far TO empty list
        SET current_max TO arr[0]
        FOR each num IN arr
            SET current_max TO maximum of current_max and num
            APPEND current_max TO max_so_far
        SET sorted_arr TO sorted version of arr
        SET chunks TO 0
        SET max_chunk TO 0
        FOR i FROM 0 TO LENGTH OF arr MINUS 1
            SET max_chunk TO maximum of max_chunk and arr[i]
            IF max_chunk EQUALS sorted_arr[i]
                INCREMENT chunks BY 1
        RETURN chunks
    END FUNCTION
END CLASS
