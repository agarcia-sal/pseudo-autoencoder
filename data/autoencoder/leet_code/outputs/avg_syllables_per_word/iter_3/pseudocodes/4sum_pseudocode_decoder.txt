CLASS Solution
    FUNCTION fourSum(nums, target)
        SORT nums
        SET n TO LENGTH OF nums
        SET quadruplets TO empty list
        
        FOR i FROM 0 TO n MINUS 4
            IF i GREATER THAN 0 AND nums[i] EQUALS nums[i MINUS 1]
                CONTINUE TO next iteration
            FOR j FROM i PLUS 1 TO n MINUS 3
                IF j GREATER THAN i PLUS 1 AND nums[j] EQUALS nums[j MINUS 1]
                    CONTINUE TO next iteration
                SET left TO j PLUS 1
                SET right TO n MINUS 1
                WHILE left LESS THAN right
                    SET total TO nums[i] PLUS nums[j] PLUS nums[left] PLUS nums[right]
                    IF total EQUALS target
                        APPEND [nums[i], nums[j], nums[left], nums[right]] TO quadruplets
                        WHILE left LESS THAN right AND nums[left] EQUALS nums[left PLUS 1]
                            INCREMENT left BY 1
                        WHILE left LESS THAN right AND nums[right] EQUALS nums[right MINUS 1]
                            DECREMENT right BY 1
                        INCREMENT left BY 1
                        DECREMENT right BY 1
                    ELSE IF total LESS THAN target
                        INCREMENT left BY 1
                    ELSE
                        DECREMENT right BY 1
        RETURN quadruplets
    END FUNCTION
END CLASS
