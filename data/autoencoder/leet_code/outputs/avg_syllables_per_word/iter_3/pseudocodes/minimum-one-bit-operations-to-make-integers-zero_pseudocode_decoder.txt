CLASS Solution
    FUNCTION minimumOneBitOperations(n)
        IF n EQUALS 0
            RETURN 0
        END IF

        SET k TO 0
        WHILE (1 SHIFTED LEFT BY (k + 1)) LESS THAN OR EQUAL TO n
            INCREMENT k BY 1
        END WHILE

        SET most_significant_bit TO 1 SHIFTED LEFT BY k

        RETURN (1 SHIFTED LEFT BY (k + 1)) MINUS 1 MINUS self.minimumOneBitOperations(n XOR most_significant_bit)
    END FUNCTION
END CLASS
