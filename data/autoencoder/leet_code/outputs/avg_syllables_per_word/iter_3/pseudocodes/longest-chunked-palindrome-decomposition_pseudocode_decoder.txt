CLASS Solution
    FUNCTION longestDecomposition(text)
        SET left TO 0
        SET right TO LENGTH OF text MINUS 1
        SET left_part TO empty string
        SET right_part TO empty string
        SET k TO 0

        WHILE left LESS THAN right
            SET left_part TO left_part CONCATENATED WITH text AT left
            SET right_part TO text AT right CONCATENATED WITH right_part

            IF left_part EQUALS right_part
                INCREMENT k BY 2
                SET left_part TO empty string
                SET right_part TO empty string
            END IF

            INCREMENT left BY 1
            DECREMENT right BY 1
        END WHILE

        IF left_part IS NOT empty OR left EQUALS right
            INCREMENT k BY 1
        END IF

        RETURN k
    END FUNCTION
END CLASS
