CLASS Solution
    FUNCTION canPartition(nums)
        SET total_sum TO sum of nums
        IF total_sum MODULO 2 IS NOT 0
            RETURN False
        SET target TO total_sum DIVIDED BY 2
        SET n TO length of nums
        SET dp TO list of False repeated (target + 1) times
        SET dp[0] TO True
        FOR each num IN nums
            FOR i FROM target DOWNTO num
                SET dp[i] TO dp[i] OR dp[i - num]
        RETURN dp[target]
    END FUNCTION
END CLASS
