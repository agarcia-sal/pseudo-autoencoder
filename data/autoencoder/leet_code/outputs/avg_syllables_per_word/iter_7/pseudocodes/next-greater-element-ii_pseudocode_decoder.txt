CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO LENGTH OF nums  
        SET result TO list containing n times -1  
        SET stack TO empty list  

        FOR i FROM 0 TO 2 TIMES n MINUS 1  
            SET index TO i MODULO n  
            WHILE stack IS NOT empty AND nums AT position at top of stack IS LESS THAN nums AT position index  
                SET popped_index TO REMOVE last element FROM stack  
                SET result AT position popped_index TO nums AT position index  
            IF i IS LESS THAN n  
                APPEND index TO stack  

        RETURN result  
    END FUNCTION  
END CLASS
