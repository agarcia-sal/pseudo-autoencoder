CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO set of tuples converting each mine in mines  
        SET dp TO a two-dimensional list of size n by n, each element is a list of four zeros  
        SET directions TO list containing (0, 1), (1, 0), (0, -1), (-1, 0)  
        
        FOR i FROM 0 TO n minus 1  
            FOR j FROM 0 TO n minus 1  
                IF tuple of (i, j) is not in mines_set THEN  
                    IF j is greater than 0 THEN  
                        SET dp at position i, j, direction 0 TO dp at position i, j minus 1, direction 0 plus 1  
                    IF i is greater than 0 THEN  
                        SET dp at position i, j, direction 1 TO dp at position i minus 1, j, direction 1 plus 1  
        
        SET result TO 0  
        FOR i FROM n minus 1 DOWN TO 0  
            FOR j FROM n minus 1 DOWN TO 0  
                IF tuple of (i, j) is not in mines_set THEN  
                    IF j is less than n minus 1 THEN  
                        SET dp at position i, j, direction 2 TO dp at position i, j plus 1, direction 2 plus 1  
                    IF i is less than n minus 1 THEN  
                        SET dp at position i, j, direction 3 TO dp at position i plus 1, j, direction 3 plus 1  
                    SET plus_sign_order TO minimum value in dp at position i, j  
                    SET result TO maximum of result and plus_sign_order  
        
        RETURN result  
    END FUNCTION  
END CLASS
