CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty list  
        
        FOR each token IN tokens  
            IF token IS one of plus, minus, multiply, divide  
                SET b TO the last element removed from stack  
                SET a TO the last element removed from stack  
                IF token IS plus  
                    APPEND a plus b TO stack  
                ELSE IF token IS minus  
                    APPEND a minus b TO stack  
                ELSE IF token IS multiply  
                    APPEND a multiplied by b TO stack  
                ELSE IF token IS divide  
                    APPEND integer division of a by b truncating towards zero TO stack  
            ELSE  
                APPEND integer value of token TO stack  
        
        RETURN the first element of stack  
    END FUNCTION  
END CLASS
