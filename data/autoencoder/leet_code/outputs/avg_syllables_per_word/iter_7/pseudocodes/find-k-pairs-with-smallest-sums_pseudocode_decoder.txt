CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF nums1 IS empty OR nums2 IS empty OR k IS less than OR EQUAL TO 0  
            RETURN empty list  
        
        SET min_heap TO empty list  
        
        FOR i FROM 0 TO minimum of k and LENGTH OF nums1 MINUS 1  
            PUSH pair of (SUM OF nums1 AT i AND nums2 AT 0, i, 0) INTO min_heap  
        
        SET result TO empty list  
        
        WHILE min_heap IS NOT empty AND LENGTH OF result IS LESS THAN k  
            POP smallest element FROM min_heap INTO _, i, j  
            APPEND list of nums1 AT i AND nums2 AT j TO result  
            IF j PLUS 1 IS LESS THAN LENGTH OF nums2  
                PUSH pair of (SUM OF nums1 AT i AND nums2 AT j PLUS 1, i, j PLUS 1) INTO min_heap  
        
        RETURN result  
    END FUNCTION  
END CLASS
