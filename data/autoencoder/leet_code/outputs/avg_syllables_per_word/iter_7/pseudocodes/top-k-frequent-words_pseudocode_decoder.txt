CLASS Solution  
    FUNCTION topKFrequent(words, k)  
        SET count TO frequency count of each word in words  
        SET heap TO list containing tuples of negative frequency and corresponding word from count  
        TRANSFORM heap into a heap data structure  
        SET result TO empty list  
        FOR index FROM 0 TO k MINUS 1  
            REMOVE element FROM heap with smallest value  
            APPEND the word part of the removed element TO result  
        RETURN result  
    END FUNCTION  
END CLASS
