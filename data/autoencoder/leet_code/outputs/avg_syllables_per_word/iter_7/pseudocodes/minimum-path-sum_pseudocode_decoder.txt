CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid IS empty OR first element of grid IS empty THEN  
            RETURN 0  
        END IF  
        
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF first element of grid  
        
        INITIALIZE dp AS two-dimensional list of zeros with dimensions m BY n  
        
        SET dp at position 0,0 TO grid at position 0,0  
        
        FOR j FROM 1 TO n MINUS 1 DO  
            SET dp at position 0,j TO dp at position 0,j MINUS 1 PLUS grid at position 0,j  
        END FOR  
        
        FOR i FROM 1 TO m MINUS 1 DO  
            SET dp at position i,0 TO dp at position i MINUS 1,0 PLUS grid at position i,0  
        END FOR  
        
        FOR i FROM 1 TO m MINUS 1 DO  
            FOR j FROM 1 TO n MINUS 1 DO  
                SET dp at position i,j TO MINIMUM OF dp at position i MINUS 1,j AND dp at position i,j MINUS 1 PLUS grid at position i,j  
            END FOR  
        END FOR  
        
        RETURN dp at position m MINUS 1,n MINUS 1  
    END FUNCTION  
END CLASS
