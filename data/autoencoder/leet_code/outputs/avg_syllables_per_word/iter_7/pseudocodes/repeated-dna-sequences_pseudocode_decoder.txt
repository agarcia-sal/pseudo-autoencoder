CLASS Solution  
    FUNCTION findRepeatedDnaSequences(s)  
        SET sequence_count TO a dictionary with default integer value  
        FOR each index i FROM 0 TO LENGTH OF s MINUS 10  
            SET sequence TO substring of s from i TO i PLUS 10  
            INCREMENT count of sequence IN sequence_count BY 1  
        SET repeated_sequences TO empty list  
        FOR each sequence and count IN sequence_count  
            IF count GREATER THAN 1  
                APPEND sequence TO repeated_sequences  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
