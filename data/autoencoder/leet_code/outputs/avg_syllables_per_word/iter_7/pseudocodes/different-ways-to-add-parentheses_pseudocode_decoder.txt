CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS plus sign  
                        APPEND l plus r TO results  
                    ELSE IF operator EQUALS minus sign  
                        APPEND l minus r TO results  
                    ELSE IF operator EQUALS multiplication sign  
                        APPEND l multiplied BY r TO results  
            RETURN results  
        END FUNCTION  

        FUNCTION helper(sub_expr)  
            IF sub_expr CONSISTS ONLY OF DIGITS  
                RETURN list containing integer value of sub_expr  
            SET results TO empty list  
            FOR each index i FROM zero TO LENGTH OF sub_expr MINUS one  
                IF character AT position i IN sub_expr IS one of plus sign, minus sign, or multiplication sign  
                    SET left_results TO helper(substring OF sub_expr FROM start TO i minus one)  
                    SET right_results TO helper(substring OF sub_expr FROM i plus one TO end)  
                    EXTEND results BY compute(left_results, right_results, character AT i IN sub_expr)  
            RETURN results  
        END FUNCTION  

        RETURN helper(expression)  
END CLASS
