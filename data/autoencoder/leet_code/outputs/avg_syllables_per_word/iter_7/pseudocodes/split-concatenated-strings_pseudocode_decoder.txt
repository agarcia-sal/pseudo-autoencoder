CLASS Solution  
    FUNCTION splitLoopedString(strs)  
        FOR i FROM 0 TO LENGTH OF strs MINUS 1  
            SET reversed_str TO reverse of strs AT i  
            IF reversed_str lexicographically GREATER THAN strs AT i  
                SET strs AT i TO reversed_str  
        END FOR  
        
        SET doubled_strs TO concatenation of strs WITH strs  
        SET max_string TO empty string  
        
        FOR i FROM 0 TO LENGTH OF strs MINUS 1  
            FOR each is_reversed IN [False, True]  
                IF is_reversed IS True  
                    SET current TO reverse of strs AT i  
                ELSE  
                    SET current TO strs AT i  
                END IF  
                
                SET remaining TO concatenation of doubled_strs FROM i PLUS 1 TO i PLUS LENGTH OF strs MINUS 1  
                
                FOR j FROM 0 TO LENGTH OF current  
                    SET candidate TO concatenation of substring of current FROM j TO end, remaining, substring of current FROM start TO j  
                    IF candidate lexicographically GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
