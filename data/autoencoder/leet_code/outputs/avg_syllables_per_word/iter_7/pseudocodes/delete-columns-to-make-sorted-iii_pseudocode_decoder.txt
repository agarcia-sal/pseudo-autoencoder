CLASS Solution  
    FUNCTION minDeletionSize(strs)  
        SET m TO LENGTH OF strs  
        SET n TO LENGTH OF first element in strs  
        SET dp TO list of length n with all elements set to 1  
        FOR i FROM 1 TO n MINUS 1  
            FOR j FROM 0 TO i MINUS 1  
                IF for every k FROM 0 TO m MINUS 1, character at column j in strs[k] LESS THAN OR EQUAL TO character at column i in strs[k] THEN  
                    SET dp[i] TO maximum of dp[i] AND dp[j] PLUS 1  
        SET longest_non_decreasing_subsequence TO maximum value in dp  
        RETURN n MINUS longest_non_decreasing_subsequence  
    END FUNCTION  
END CLASS
