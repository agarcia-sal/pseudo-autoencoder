CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF LENGTH OF nums EQUALS 1 THEN  
            RETURN STRING REPRESENTATION OF nums AT INDEX 0  
        ELSE IF LENGTH OF nums EQUALS 2 THEN  
            RETURN CONCATENATION OF STRING REPRESENTATION OF nums AT INDEX 0, " DIVIDED BY ", AND STRING REPRESENTATION OF nums AT INDEX 1  
        ELSE  
            RETURN CONCATENATION OF STRING REPRESENTATION OF nums AT INDEX 0, " DIVIDED BY (", AND JOIN STRINGS OF nums FROM INDEX 1 ONWARDS WITH " DIVIDED BY ", AND ")"  
        END IF  
    END FUNCTION  
END CLASS
