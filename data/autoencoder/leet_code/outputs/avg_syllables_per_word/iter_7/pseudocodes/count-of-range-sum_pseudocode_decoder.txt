CLASS Solution  
    FUNCTION countRangeSum(nums, lower, upper)  
        SET prefix_sums TO a list containing 0  
        SET current_sum TO 0  
        SET count TO 0  
        
        FOR each num IN nums  
            INCREMENT current_sum BY num  
            SET left TO current_sum MINUS upper  
            SET right TO current_sum MINUS lower  
            
            INCREMENT count BY the number of elements in prefix_sums LESS THAN OR EQUAL TO right MINUS the number of elements in prefix_sums GREATER THAN OR EQUAL TO left  
            
            INSERT current_sum INTO prefix_sums maintaining sorted order  
        
        RETURN count  
    END FUNCTION  
END CLASS
