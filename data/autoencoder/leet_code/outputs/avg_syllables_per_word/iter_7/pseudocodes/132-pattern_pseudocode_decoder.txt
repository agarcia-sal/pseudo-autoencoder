CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO empty list  
        FOR each num IN nums traversed from right to left  
            IF num LESS THAN second THEN  
                RETURN True  
            WHILE stack IS NOT empty AND last element of stack LESS THAN num  
                SET second TO remove last element from stack  
            APPEND num TO stack  
        RETURN False  
    END FUNCTION  
END CLASS
