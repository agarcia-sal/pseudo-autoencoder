CLASS Solution  
    FUNCTION findSubsequences(nums)  
        FUNCTION dfs(u, last, t)  
            IF u EQUALS LENGTH OF nums THEN  
                IF LENGTH OF t GREATER THAN 1 THEN  
                    APPEND COPY OF t TO ans  
                END IF  
                RETURN  
            END IF  
            IF nums AT u GREATER THAN OR EQUAL TO last THEN  
                APPEND nums AT u TO t  
                CALL dfs WITH u INCREMENTED BY 1, nums AT u, t  
                REMOVE LAST ELEMENT FROM t  
            END IF  
            IF nums AT u NOT EQUAL TO last THEN  
                CALL dfs WITH u INCREMENTED BY 1, last, t  
            END IF  
        END FUNCTION  
        
        SET ans TO EMPTY LIST  
        CALL dfs WITH 0, NEGATIVE THOUSAND, EMPTY LIST  
        RETURN ans  
    END FUNCTION  
END CLASS
