CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    END IF  
    SET root TO new TreeNode with values at position 0  
    SET i TO 1  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO POP left element from queue  
        IF i LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.left TO new TreeNode with values at i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.right TO new TreeNode with values at i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree(self, root)  
        IF root IS None  
            RETURN 0  
        END IF  
        SET queue TO deque containing tuple of root and 0  
        SET max_width TO 0  
        WHILE queue IS NOT EMPTY  
            SET level_length TO LENGTH OF queue  
            SET first_index TO index part of first element in queue  
            FOR each number FROM 0 TO level_length MINUS 1  
                SET node, index TO POP left element from queue  
                IF number IS EQUAL TO level_length MINUS 1  
                    SET max_width TO the GREATER OF max_width AND index MINUS first_index PLUS 1  
                END IF  
                IF node.left IS NOT None  
                    APPEND tuple of node.left AND 2 MULTIPLIED BY index TO queue  
                END IF  
                IF node.right IS NOT None  
                    APPEND tuple of node.right AND 2 MULTIPLIED BY index PLUS 1 TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
