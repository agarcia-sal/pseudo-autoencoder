CLASS Solution  
    FUNCTION integerBreak(n)  
        SET dp TO list of zeros with length n plus 1  
        SET dp at index 1 TO 1  
        FOR i FROM 2 TO n  
            SET max_product TO 0  
            FOR j FROM 1 TO i MINUS 1  
                SET candidate1 TO j TIMES dp at index i MINUS j  
                SET candidate2 TO j TIMES (i MINUS j)  
                SET max_product TO maximum of max_product, candidate1, candidate2  
            SET dp at index i TO max_product  
        RETURN dp at index n  
    END FUNCTION  
END CLASS
