CLASS TreeNode  
    FUNCTION __init__(val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode with first value in values  
    SET i TO 1  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.left TO new TreeNode with values at i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.right TO new TreeNode with values at i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION flatten(root)  
        IF root IS None  
            RETURN  
        SET stack TO list containing root  
        WHILE stack IS NOT empty  
            SET node TO REMOVE last element FROM stack  
            IF node.right IS NOT None  
                APPEND node.right TO stack  
            IF node.left IS NOT None  
                APPEND node.left TO stack  
            SET node.left TO None  
            IF stack IS NOT empty  
                SET node.right TO last element in stack  
            ELSE  
                SET node.right TO None  
    END FUNCTION  
END CLASS
