CLASS Solution  
    FUNCTION subarraySum(nums, k)  
        SET cumulative_sum_freq TO a dictionary with key 0 mapped to 1  
        SET current_sum TO 0  
        SET count TO 0  
        FOR each num IN nums  
            INCREMENT current_sum BY num  
            IF current_sum MINUS k IS a key in cumulative_sum_freq THEN  
                INCREMENT count BY cumulative_sum_freq VALUE at key (current_sum MINUS k)  
            END IF  
            IF current_sum IS a key in cumulative_sum_freq THEN  
                INCREMENT cumulative_sum_freq VALUE at key current_sum BY 1  
            ELSE  
                SET cumulative_sum_freq VALUE at key current_sum TO 1  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
