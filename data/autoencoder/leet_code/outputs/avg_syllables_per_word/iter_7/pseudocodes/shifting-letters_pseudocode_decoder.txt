CLASS Solution  
    FUNCTION shiftingLetters(s, shifts)  
        SET total_shifts TO 0  
        FOR i FROM LENGTH OF shifts MINUS 1 DOWNTO 0  
            INCREMENT total_shifts BY shifts AT i  
            SET shifts AT i TO total_shifts  
        SET result TO empty list  
        FOR i FROM 0 TO LENGTH OF s MINUS 1  
            SET base_value TO position of character s AT i relative to 'a'  
            SET shifted_value TO base_value PLUS shifts AT i  
            SET wrapped_value TO shifted_value MODULO 26  
            SET new_char TO character at position wrapped_value relative to 'a'  
            APPEND new_char TO result  
        RETURN concatenation of all characters in result  
    END FUNCTION  
END CLASS
