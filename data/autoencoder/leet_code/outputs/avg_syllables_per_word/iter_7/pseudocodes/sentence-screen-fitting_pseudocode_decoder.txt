CLASS Solution  
    FUNCTION wordsTyping(sentence, rows, cols)  
        SET s TO single string made by joining sentence with spaces and appending one space  
        SET total_length TO length of s  
        SET position TO 0  
        
        FOR each row IN range from 1 TO rows  
            INCREMENT position BY cols  
            
            IF character at s indexed by position modulo total_length IS space  
                INCREMENT position BY 1  
            ELSE  
                WHILE position GREATER THAN 0 AND character at s indexed by (position minus 1) modulo total_length IS NOT space  
                    DECREMENT position BY 1  
        
        RETURN position DIVIDED BY total_length  
    END FUNCTION  
END CLASS
