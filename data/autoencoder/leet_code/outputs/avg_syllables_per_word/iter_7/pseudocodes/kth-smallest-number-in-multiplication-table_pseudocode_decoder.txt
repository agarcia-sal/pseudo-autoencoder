CLASS Solution  
    FUNCTION findKthNumber(m, n, k)  
        FUNCTION enough(x)  
            SET count TO 0  
            FOR i FROM 1 TO m INCLUSIVE  
                SET count TO count PLUS THE MINIMUM OF (x DIVIDED BY i) AND n  
            END FOR  
            RETURN WHETHER count IS GREATER THAN OR EQUAL TO k  
        END FUNCTION  

        SET low TO 1  
        SET high TO m MULTIPLIED BY n  
        WHILE low IS LESS THAN high  
            SET mid TO THE INTEGER DIVISION OF (low PLUS high) BY 2  
            IF enough(mid) IS TRUE  
                SET high TO mid  
            ELSE  
                SET low TO mid PLUS 1  
            END IF  
        END WHILE  
        RETURN low  
    END FUNCTION  
END CLASS
