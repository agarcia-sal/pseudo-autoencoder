CLASS Solution  
    FUNCTION longestSubstring(s, k)  
        IF LENGTH OF s LESS THAN k THEN  
            RETURN 0  
        END IF  
        
        SET count TO FREQUENCY COUNT OF CHARACTERS IN s  
        
        FOR EACH char IN count  
            IF count[char] LESS THAN k THEN  
                RETURN MAXIMUM OF longestSubstring(substring, k) FOR EACH substring OBTAINED BY SPLITTING s BY char  
            END IF  
        END FOR  
        
        RETURN LENGTH OF s  
    END FUNCTION  
END CLASS
