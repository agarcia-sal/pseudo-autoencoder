CLASS Solution  
    FUNCTION search(nums, target)  
        IF nums IS EMPTY THEN  
            RETURN False  
        
        SET left TO 0  
        SET right TO LENGTH OF nums MINUS 1  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS (right MINUS left) DIVIDED BY 2  
            
            IF nums AT mid EQUALS target THEN  
                RETURN True  
            
            IF nums AT left EQUALS nums AT mid AND nums AT mid EQUALS nums AT right THEN  
                INCREMENT left BY 1  
                DECREMENT right BY 1  
            ELSE IF nums AT left IS LESS THAN OR EQUAL TO nums AT mid THEN  
                IF nums AT left IS LESS THAN OR EQUAL TO target AND target IS LESS THAN nums AT mid THEN  
                    SET right TO mid MINUS 1  
                ELSE  
                    SET left TO mid PLUS 1  
            ELSE  
                IF nums AT mid IS LESS THAN target AND target IS LESS THAN OR EQUAL TO nums AT right THEN  
                    SET left TO mid PLUS 1  
                ELSE  
                    SET right TO mid MINUS 1  
        
        RETURN False  
    END FUNCTION  
END CLASS
