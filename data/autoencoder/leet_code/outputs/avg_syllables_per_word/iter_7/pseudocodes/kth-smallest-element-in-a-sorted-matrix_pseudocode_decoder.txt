CLASS Solution  
    FUNCTION kthSmallest(matrix, k)  
        FUNCTION countLessEqual(mid)  
            SET count TO 0  
            SET row TO n MINUS 1  
            SET col TO 0  
            WHILE row GREATER THAN OR EQUAL TO 0 AND col LESS THAN n  
                IF matrix AT row AND col LESS THAN OR EQUAL TO mid THEN  
                    INCREMENT count BY row PLUS 1  
                    INCREMENT col BY 1  
                ELSE  
                    DECREMENT row BY 1  
            RETURN count  
        END FUNCTION  
        
        SET n TO LENGTH OF matrix  
        SET low TO matrix AT 0 AND 0  
        SET high TO matrix AT n MINUS 1 AND n MINUS 1  
        
        WHILE low LESS THAN high  
            SET mid TO low PLUS high DIVIDED BY 2 USING INTEGER DIVISION  
            IF countLessEqual(mid) LESS THAN k THEN  
                SET low TO mid PLUS 1  
            ELSE  
                SET high TO mid  
        RETURN low  
    END FUNCTION  
END CLASS
