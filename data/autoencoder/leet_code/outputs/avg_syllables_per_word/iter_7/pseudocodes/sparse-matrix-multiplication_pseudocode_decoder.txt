CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO LENGTH OF mat1  
        SET k TO LENGTH OF mat1 AT index 0  
        SET k TO LENGTH OF mat2  
        SET n TO LENGTH OF mat2 AT index 0  
        SET result TO a matrix of zeros with dimensions m BY n  
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                FOR p FROM 0 TO k MINUS 1  
                    INCREMENT result AT i AT j BY mat1 AT i AT p MULTIPLIED BY mat2 AT p AT j  
        RETURN result  
    END FUNCTION  
END CLASS
