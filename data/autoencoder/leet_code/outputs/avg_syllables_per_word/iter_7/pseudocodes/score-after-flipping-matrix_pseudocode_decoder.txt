CLASS Solution  
    FUNCTION matrixScore(grid)  
        FOR each row IN grid  
            IF first element of row EQUALS 0  
                FOR index FROM 0 TO LENGTH OF row MINUS 1  
                    SET element at index in row TO 1 MINUS element at index in row  
        FOR col FROM 1 TO LENGTH OF first row IN grid MINUS 1  
            SET count_ones TO 0  
            FOR row_index FROM 0 TO LENGTH OF grid MINUS 1  
                INCREMENT count_ones BY element at col in grid at row_index  
            IF count_ones LESS THAN LENGTH OF grid DIVIDED BY 2  
                FOR row_index FROM 0 TO LENGTH OF grid MINUS 1  
                    SET element at col in grid at row_index TO 1 MINUS element at col in grid at row_index  
        SET score TO 0  
        FOR each row IN grid  
            SET binary_string TO concatenation of string representations of elements in row  
            SET numeric_value TO integer value of binary_string interpreted as base 2  
            INCREMENT score BY numeric_value  
        RETURN score  
    END FUNCTION  
END CLASS
