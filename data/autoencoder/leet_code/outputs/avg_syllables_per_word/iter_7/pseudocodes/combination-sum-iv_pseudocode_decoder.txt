CLASS Solution  
    FUNCTION combinationSum4(nums, target)  
        SET dp TO list of zeros with length target PLUS 1  
        SET dp at index 0 TO 1  
        FOR each amount FROM 1 TO target  
            FOR each num IN nums  
                IF amount GREATER THAN OR EQUAL TO num THEN  
                    SET dp at index amount TO dp at index amount PLUS dp at index amount MINUS num  
        RETURN dp at index target  
    END FUNCTION  
END CLASS
