CLASS Solution  
    FUNCTION boldWords(words, s)  
        SET bold TO a list of False values with length equal to length of s  
        
        FOR each word IN words  
            SET start TO the position of word in s  
            WHILE start is NOT EQUAL TO negative one  
                FOR i FROM start TO start PLUS length of word MINUS one  
                    SET bold at position i TO True  
                SET start TO the position of word in s starting from start PLUS one  
        
        SET result TO an empty list  
        SET i TO zero  
        WHILE i LESS THAN length of s  
            IF bold at position i IS True  
                APPEND "<b>" TO result  
                WHILE i LESS THAN length of s AND bold at position i IS True  
                    APPEND character s at position i TO result  
                    INCREMENT i BY one  
                APPEND "</b>" TO result  
            ELSE  
                APPEND character s at position i TO result  
                INCREMENT i BY one  
        
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
