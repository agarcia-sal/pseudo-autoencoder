CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode initialized with values at position zero  
    SET i TO 1  
    SET queue TO empty deque  
    APPEND root TO queue  

    WHILE queue IS NOT empty  
        SET node TO element removed from left of queue  
        IF i IS LESS THAN length of values AND values at i IS NOT None  
            SET node.left TO new TreeNode initialized with values at i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN length of values AND values at i IS NOT None  
            SET node.right TO new TreeNode initialized with values at i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  

    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalOrder(self, root)  
        IF root IS None  
            RETURN empty list  
        
        SET column_table TO empty dictionary mapping columns to list of node values  
        SET queue TO deque initialized with tuple containing root and zero as column index  
        SET min_column TO zero  
        SET max_column TO zero  

        WHILE queue IS NOT empty  
            SET node, column_index TO element removed from left of queue  
            IF node IS NOT None  
                APPEND node.val TO column_table at column_index  
                SET min_column TO minimum of min_column and column_index  
                SET max_column TO maximum of max_column and column_index  
                APPEND tuple of node.left and column_index minus one TO queue  
                APPEND tuple of node.right and column_index plus one TO queue  
            END IF  
        END WHILE  

        SET result TO empty list  
        FOR each x FROM min_column TO max_column INCLUSIVE  
            APPEND column_table at x TO result  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
