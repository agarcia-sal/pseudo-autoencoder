CLASS Solution  
    FUNCTION maxA(n)  
        IF n LESS THAN OR EQUAL TO 3 THEN  
            RETURN n  
        END IF  
        
        SET dp TO list of zeros WITH LENGTH n PLUS 1  
        
        FOR i FROM 1 TO 3  
            SET dp[i] TO i  
        END FOR  
        
        FOR i FROM 4 TO n  
            FOR j FROM i MINUS 3 DOWN TO 1  
                SET dp[i] TO maximum OF dp[i] AND dp[j] MULTIPLIED BY (i MINUS j MINUS 1)  
            END FOR  
        END FOR  
        
        RETURN dp[n]  
    END FUNCTION  
END CLASS
