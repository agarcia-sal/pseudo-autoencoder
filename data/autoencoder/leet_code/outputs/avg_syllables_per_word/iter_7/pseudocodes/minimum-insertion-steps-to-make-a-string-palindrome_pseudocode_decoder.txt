CLASS Solution  
    FUNCTION minInsertions(s)  
        FUNCTION longest_palindromic_subsequence(s)  
            SET n TO LENGTH OF s  
            SET dp TO 2D array of size n by n filled with 0  
              
            FOR i FROM 0 TO n MINUS 1  
                SET dp[i][i] TO 1  
            END FOR  
              
            FOR cl FROM 2 TO n  
                FOR i FROM 0 TO n MINUS cl  
                    SET j TO i PLUS cl MINUS 1  
                    IF s AT i EQUALS s AT j AND cl EQUALS 2 THEN  
                        SET dp[i][j] TO 2  
                    ELSE IF s AT i EQUALS s AT j THEN  
                        SET dp[i][j] TO dp AT (i PLUS 1) AT (j MINUS 1) PLUS 2  
                    ELSE  
                        SET dp[i][j] TO MAXIMUM OF dp AT i AT (j MINUS 1) AND dp AT (i PLUS 1) AT j  
                    END IF  
                END FOR  
            END FOR  
              
            RETURN dp AT 0 AT (n MINUS 1)  
        END FUNCTION  
          
        RETURN LENGTH OF s MINUS longest_palindromic_subsequence(s)  
    END FUNCTION  
END CLASS
