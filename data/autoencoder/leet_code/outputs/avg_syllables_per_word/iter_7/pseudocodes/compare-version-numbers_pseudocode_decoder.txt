CLASS Solution  
    FUNCTION compareVersion(version1, version2)  
        SET v1_revisions TO version1 split by '.'  
        SET v2_revisions TO version2 split by '.'  
        SET max_length TO maximum of LENGTH OF v1_revisions and LENGTH OF v2_revisions  
        FOR i FROM 0 TO max_length MINUS 1  
            IF i LESS THAN LENGTH OF v1_revisions  
                SET v1 TO integer value of v1_revisions at index i  
            ELSE  
                SET v1 TO 0  
            IF i LESS THAN LENGTH OF v2_revisions  
                SET v2 TO integer value of v2_revisions at index i  
            ELSE  
                SET v2 TO 0  
            IF v1 LESS THAN v2  
                RETURN negative 1  
            ELSE IF v1 GREATER THAN v2  
                RETURN 1  
        END FOR  
        RETURN 0  
    END FUNCTION  
END CLASS
