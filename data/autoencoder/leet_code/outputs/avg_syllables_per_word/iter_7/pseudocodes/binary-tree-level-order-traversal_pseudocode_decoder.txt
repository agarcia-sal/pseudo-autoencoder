CLASS TreeNode  
    FUNCTION __init__(self, val SET TO 0, left SET TO None, right SET TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH values AT FIRST POSITION  
    SET i TO 1  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO queue REMOVE FIRST ELEMENT  
        IF i IS LESS THAN LENGTH OF values AND values AT i IS NOT None THEN  
            SET node.left TO new TreeNode WITH values AT i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values AT i IS NOT None THEN  
            SET node.right TO new TreeNode WITH values AT i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root IS None THEN  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO new deque INITIALIZED WITH root  
        WHILE queue IS NOT EMPTY  
            SET level_size TO LENGTH OF queue  
            SET current_level TO empty list  
            FOR each number FROM 1 TO level_size DO  
                SET node TO queue REMOVE FIRST ELEMENT  
                APPEND node.val TO current_level  
                IF node.left EXISTS THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right EXISTS THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
