CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        SET graph TO a mapping from each node 1 through n TO an empty set  
        
        FOR each pair (u, v) IN edges  
            ADD v TO graph[u]  
            ADD u TO graph[v]  
        
        SET min_degree TO infinity  
        SET trio_found TO false  
        
        FOR i FROM 1 TO n  
            FOR j FROM i PLUS 1 TO n  
                FOR k FROM j PLUS 1 TO n  
                    IF j IS IN graph[i] AND k IS IN graph[i] AND k IS IN graph[j] THEN  
                        SET trio_found TO true  
                        SET trio_degree TO LENGTH OF graph[i] PLUS LENGTH OF graph[j] PLUS LENGTH OF graph[k] MINUS 6  
                        IF trio_degree IS LESS THAN min_degree THEN  
                            SET min_degree TO trio_degree  
        
        IF trio_found IS true THEN  
            RETURN min_degree  
        ELSE  
            RETURN -1  
    END FUNCTION  
END CLASS
