CLASS Solution  
    FUNCTION findOrder(numCourses, prerequisites)  
        SET adj_list TO an empty mapping from each course in range numCourses TO an empty list  
        SET in_degrees TO a list of zeros with length numCourses  
        
        FOR each pair (dest, src) IN prerequisites  
            APPEND dest TO adj_list[src]  
            INCREMENT in_degrees[dest] BY 1  
        
        SET queue TO a deque containing all courses WHERE in_degrees of course EQUALS 0  
        SET topological_order TO an empty list  
        
        WHILE queue is not empty  
            SET current TO the element removed from the front of queue  
            APPEND current TO topological_order  
            
            FOR each neighbor IN adj_list[current]  
                DECREMENT in_degrees[neighbor] BY 1  
                IF in_degrees[neighbor] EQUALS 0 THEN  
                    APPEND neighbor TO queue  
        
        IF length of topological_order EQUALS numCourses THEN  
            RETURN topological_order  
        ELSE  
            RETURN an empty list  
    END FUNCTION  
END CLASS
