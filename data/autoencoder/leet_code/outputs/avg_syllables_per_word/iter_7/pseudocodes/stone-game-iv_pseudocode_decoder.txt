CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO list of False values with length n plus 1  
        FOR i FROM 1 TO n INCLUSIVE  
            SET j TO 1  
            WHILE j multiplied BY j IS LESS THAN OR EQUAL TO i  
                IF dp at position i minus j multiplied BY j IS False THEN  
                    SET dp at position i TO True  
                    BREAK the WHILE loop  
                INCREMENT j BY 1  
        RETURN dp at position n  
    END FUNCTION  
END CLASS
