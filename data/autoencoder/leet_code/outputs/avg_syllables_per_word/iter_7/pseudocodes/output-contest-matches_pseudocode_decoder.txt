CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO list of string representations of numbers from 1 to n  
        WHILE n GREATER THAN 1  
            FOR i FROM 0 TO n DIVIDED BY 2 MINUS 1  
                SET teams at i TO "(" concatenated with teams at i concatenated with "," concatenated with teams at (n MINUS 1 MINUS i) concatenated with ")"  
            SET n TO n DIVIDED BY 2  
        RETURN teams at 0  
    END FUNCTION  
END CLASS
