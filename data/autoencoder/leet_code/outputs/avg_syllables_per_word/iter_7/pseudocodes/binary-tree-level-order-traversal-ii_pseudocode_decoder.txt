CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY THEN  
        RETURN None  
    END IF  
    SET root TO new TreeNode WITH val values AT position 0  
    SET i TO 1  
    SET queue TO new queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY DO  
        SET node TO queue REMOVE FIRST ELEMENT  
        IF i IS LESS THAN LENGTH OF values AND values AT position i IS NOT None THEN  
            SET node.left TO new TreeNode WITH val values AT position i  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values AT position i IS NOT None THEN  
            SET node.right TO new TreeNode WITH val values AT position i  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None THEN  
        RETURN True  
    ELSE IF p IS None OR q IS None THEN  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val THEN  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom(self, root)  
        IF root IS None THEN  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO new queue WITH root AS initial element  
        WHILE queue IS NOT EMPTY DO  
            SET level_size TO LENGTH OF queue  
            SET current_level TO empty list  
            FOR each index FROM 0 TO level_size MINUS 1 DO  
                SET node TO queue REMOVE FIRST ELEMENT  
                APPEND node.val TO current_level  
                IF node.left IS NOT None THEN  
                    APPEND node.left TO queue  
                END IF  
                IF node.right IS NOT None THEN  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result IN REVERSE ORDER  
    END FUNCTION  
END CLASS
