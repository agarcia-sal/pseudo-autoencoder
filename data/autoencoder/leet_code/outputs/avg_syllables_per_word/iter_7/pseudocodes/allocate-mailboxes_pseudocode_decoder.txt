CLASS Solution  
    FUNCTION minDistance(houses, k)  
        SORT houses  
        SET n TO LENGTH OF houses  
          
        INITIALIZE cost AS 2D LIST OF SIZE n BY n WITH 0  
        FOR i FROM n MINUS 1 DOWNTO 0  
            FOR j FROM i PLUS 1 TO n MINUS 1  
                SET cost AT i AND j TO cost AT i PLUS 1 AND j MINUS 1 PLUS houses AT j MINUS houses AT i  
          
        INITIALIZE dp AS 2D LIST OF SIZE (n PLUS 1) BY (k PLUS 1) WITH infinity  
        SET dp AT 0 AND 0 TO 0  
          
        FOR i FROM 1 TO n  
            FOR j FROM 1 TO k  
                FOR p FROM 0 TO i MINUS 1  
                    SET dp AT i AND j TO MINIMUM OF dp AT i AND j AND dp AT p AND j MINUS 1 PLUS cost AT p AND i MINUS 1  
          
        RETURN dp AT n AND k  
    END FUNCTION  
END CLASS
