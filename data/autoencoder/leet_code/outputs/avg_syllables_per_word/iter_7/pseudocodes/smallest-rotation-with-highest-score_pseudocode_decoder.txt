CLASS Solution  
    FUNCTION bestRotation(nums)  
        SET n TO LENGTH OF nums  
        SET delta TO LIST OF ZEROS WITH LENGTH n  
        
        FOR i AND num IN ENUMERATE nums  
            IF num LESS THAN OR EQUAL TO i THEN  
                INCREMENT delta AT INDEX 0 BY 1  
                IF i MINUS num PLUS 1 LESS THAN n THEN  
                    DECREMENT delta AT INDEX i MINUS num PLUS 1 BY 1  
            IF i PLUS 1 LESS THAN n THEN  
                INCREMENT delta AT INDEX i PLUS 1 BY 1  
                IF i PLUS n MINUS num PLUS 1 LESS THAN n THEN  
                    DECREMENT delta AT INDEX i PLUS n MINUS num PLUS 1 BY 1  
        
        SET score TO 0  
        SET max_score TO 0  
        SET best_k TO 0  
        
        FOR k FROM 0 TO n MINUS 1  
            INCREMENT score BY delta AT INDEX k  
            IF score GREATER THAN max_score THEN  
                SET max_score TO score  
                SET best_k TO k  
        
        RETURN best_k  
    END FUNCTION  
END CLASS
