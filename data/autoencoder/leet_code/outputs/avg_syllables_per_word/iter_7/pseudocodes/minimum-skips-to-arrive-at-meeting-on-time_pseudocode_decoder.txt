CLASS Solution  
    FUNCTION minSkips(dist, speed, hoursBefore)  
        SET n TO LENGTH OF dist  
        INITIALIZE dp AS TWO-DIMENSIONAL LIST WITH n ROWS AND n PLUS 1 COLUMNS FILLED WITH infinity  
        SET total_dist TO SUM OF dist  
        
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore THEN  
            RETURN -1  
        
        SET dp AT 0 AND 0 TO dist AT 0  
        FOR j FROM 1 TO n MINUS 1 DO  
            SET dp AT 0 AND j TO dist AT 0  
        
        FOR i FROM 1 TO n MINUS 1 DO  
            FOR j FROM 0 TO i DO  
                IF j LESS THAN i THEN  
                    SET dp AT i AND j TO CEILING OF (dp AT i MINUS 1 AND j DIVIDED BY speed) TIMES speed PLUS dist AT i  
                IF j GREATER THAN 0 THEN  
                    SET dp AT i AND j TO MINIMUM OF dp AT i AND j AND dp AT i MINUS 1 AND j MINUS 1 PLUS dist AT i  
        
        FOR j FROM 0 TO n MINUS 1 DO  
            IF dp AT n MINUS 1 AND j LESS THAN OR EQUAL TO hoursBefore TIMES speed THEN  
                RETURN j  
        
        RETURN -1  
END CLASS
