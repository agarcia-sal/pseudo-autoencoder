CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO LENGTH OF s  
        SET f TO TWO-DIMENSIONAL LIST OF True VALUES WITH SIZE n BY n  
        SET k TO 0  
        SET mx TO 1  
        FOR i FROM n MINUS 2 DOWNTO 0  
            FOR j FROM i PLUS 1 TO n MINUS 1  
                SET f AT POSITION i, j TO False  
                IF CHARACTER AT s POSITION i EQUALS CHARACTER AT s POSITION j THEN  
                    SET f AT POSITION i, j TO f AT POSITION i PLUS 1, j MINUS 1  
                    IF f AT POSITION i, j IS True AND mx LESS THAN j MINUS i PLUS 1 THEN  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS 1  
        RETURN SUBSTRING OF s FROM k TO k PLUS mx  
    END FUNCTION  
END CLASS
