CLASS Solution  
    FUNCTION originalDigits(s)  
        SET count TO frequency count of characters in s  
        SET digits TO list of ten zeros  
        SET digits at 0 TO count of character 'z'  
        SET digits at 2 TO count of character 'w'  
        SET digits at 4 TO count of character 'u'  
        SET digits at 6 TO count of character 'x'  
        SET digits at 8 TO count of character 'g'  
        SET digits at 3 TO count of character 'h' MINUS digits at 8  
        SET digits at 5 TO count of character 'f' MINUS digits at 4  
        SET digits at 7 TO count of character 's' MINUS digits at 6  
        SET digits at 9 TO count of character 'i' MINUS digits at 5 MINUS digits at 6 MINUS digits at 8  
        SET digits at 1 TO count of character 'o' MINUS digits at 0 MINUS digits at 2 MINUS digits at 4  
        SET result TO empty list  
        FOR i FROM 0 TO 9  
            APPEND character representation of i repeated digits at i times TO result  
        RETURN concatenation of all items in result  
    END FUNCTION  
END CLASS
