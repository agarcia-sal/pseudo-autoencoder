CLASS Solution  
    FUNCTION checkPartitioning(s)  
        FUNCTION is_palindrome(sub)  
            RETURN sub IS EQUAL TO sub REVERSED  
        END FUNCTION  

        SET n TO LENGTH OF s  
        SET is_pal TO TWO-DIMENSIONAL LIST OF SIZE n BY n FILLED WITH False  

        FOR i FROM n MINUS 1 DOWNTO 0  
            FOR j FROM i TO n MINUS 1  
                IF s CHARACTER AT i IS EQUAL TO s CHARACTER AT j AND (j MINUS i IS LESS THAN OR EQUAL TO 1 OR is_pal VALUE AT i PLUS 1 AND j MINUS 1 IS True)  
                    SET is_pal VALUE AT i AND j TO True  
        
        FOR i FROM 1 TO n MINUS 2  
            IF is_pal VALUE AT 0 AND i MINUS 1 IS True  
                FOR j FROM i TO n MINUS 2  
                    IF is_pal VALUE AT i AND j IS True AND is_pal VALUE AT j PLUS 1 AND n MINUS 1 IS True  
                        RETURN True  
        RETURN False  
END CLASS
