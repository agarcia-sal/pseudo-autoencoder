CLASS Solution  
    FUNCTION minimumTotal(triangle)  
        FOR row FROM LENGTH OF triangle MINUS 2 DOWNTO 0  
            FOR col FROM 0 TO LENGTH OF triangle AT row MINUS 1  
                SET triangle AT row AND col TO triangle AT row AND col PLUS MINIMUM OF triangle AT row PLUS 1 AND col, triangle AT row PLUS 1 AND col PLUS 1  
            END FOR  
        END FOR  
        RETURN triangle AT 0 AND 0  
    END FUNCTION  
END CLASS
