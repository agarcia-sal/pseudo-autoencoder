CLASS Solution  
    FUNCTION multiply(num1, num2)  
        IF num1 EQUALS "0" OR num2 EQUALS "0"  
            RETURN "0"  
        END IF  

        INITIALIZE result AS list OF zeros WITH LENGTH EQUAL TO LENGTH OF num1 PLUS LENGTH OF num2  

        SET num1 TO reverse OF num1  
        SET num2 TO reverse OF num2  

        FOR i FROM 0 TO LENGTH OF num1 MINUS 1  
            FOR j FROM 0 TO LENGTH OF num2 MINUS 1  
                SET mul TO numeric value OF num1 AT i MULTIPLIED BY numeric value OF num2 AT j  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS 1  
                SET sum_ TO mul PLUS value IN result AT position p1  

                SET value IN result AT position p1 TO sum_ MODULO 10  
                INCREMENT value IN result AT position p2 BY sum_ DIVIDED BY 10  
            END FOR  
        END FOR  

        SET result_str TO concatenation OF elements IN reversed result AS strings  
        RETURN result_str WITH leading zeros REMOVED  
    END FUNCTION  
END CLASS
