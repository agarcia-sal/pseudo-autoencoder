CLASS Solution  
    FUNCTION minimumDeviation(nums)  
        SET max_heap TO empty list  
        SET min_value TO infinity  
        FOR each num IN nums  
            IF num MODULO 2 EQUALS 1  
                SET num TO num MULTIPLIED BY 2  
            END IF  
            IF num LESS THAN min_value  
                SET min_value TO num  
            END IF  
            ADD negative num TO max_heap USING heap push  
        END FOR  
        
        SET min_deviation TO infinity  
        
        WHILE the top element of max_heap MODULO 2 EQUALS 0  
            SET max_value TO negative result of heap pop from max_heap  
            IF max_value MINUS min_value LESS THAN min_deviation  
                SET min_deviation TO max_value MINUS min_value  
            END IF  
            SET max_value TO max_value DIVIDED BY 2  
            IF max_value LESS THAN min_value  
                SET min_value TO max_value  
            END IF  
            ADD negative max_value TO max_heap USING heap push  
        END WHILE  
        
        IF negative top element of max_heap MINUS min_value LESS THAN min_deviation  
            SET min_deviation TO negative top element of max_heap MINUS min_value  
        END IF  
        
        RETURN min_deviation  
    END FUNCTION  
END CLASS
