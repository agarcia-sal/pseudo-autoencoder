CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET i TO LENGTH OF nums MINUS 2  
        WHILE i GREATER THAN OR EQUAL TO 0 AND nums AT i GREATER THAN OR EQUAL TO nums AT i PLUS 1  
            DECREMENT i BY 1  
        END WHILE  

        IF i GREATER THAN OR EQUAL TO 0  
            SET j TO LENGTH OF nums MINUS 1  
            WHILE nums AT j LESS THAN OR EQUAL TO nums AT i  
                DECREMENT j BY 1  
            END WHILE  
            SWAP nums AT i WITH nums AT j  
        END IF  

        REVERSE THE SUBSEQUENCE OF nums STARTING AT i PLUS 1 TO END  
    END FUNCTION  
END CLASS
