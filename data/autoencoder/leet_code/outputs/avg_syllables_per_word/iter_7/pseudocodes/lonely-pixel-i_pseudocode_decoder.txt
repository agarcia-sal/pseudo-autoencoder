CLASS Solution  
    FUNCTION findLonelyPixel(picture)  
        IF picture IS empty OR first element of picture IS empty  
            RETURN 0  
        SET m TO LENGTH OF picture  
        SET n TO LENGTH OF first element of picture  
        SET row_count TO list of zeros of length m  
        SET col_count TO list of zeros of length n  
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF element at picture position i, j IS 'B'  
                    INCREMENT row_count at i BY 1  
                    INCREMENT col_count at j BY 1  
        SET lonely_pixel_count TO 0  
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF element at picture position i, j IS 'B' AND row_count at i EQUALS 1 AND col_count at j EQUALS 1  
                    INCREMENT lonely_pixel_count BY 1  
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
