CLASS Solution  
    FUNCTION stoneGame(piles)  
        FUNCTION dfs(i, j)  
            IF i GREATER THAN j THEN  
                RETURN 0  
            RETURN maximum BETWEEN (value at piles index i MINUS dfs(i INCREMENTED BY 1, j)) AND (value at piles index j MINUS dfs(i, j DECREMENTED BY 1))  
        END FUNCTION  
        RETURN dfs(0, length of piles DECREMENTED BY 1) GREATER THAN 0  
    END FUNCTION  
END CLASS
