CLASS Solution  
    FUNCTION maximumNumberOfOnes(width, height, sideLength, maxOnes)  
        SET freq TO list of zeros with length sideLength multiplied BY sideLength  
        FOR i FROM 0 TO width MINUS 1  
            FOR j FROM 0 TO height MINUS 1  
                SET pos TO remainder of i DIVIDED BY sideLength multiplied BY sideLength PLUS remainder of j DIVIDED BY sideLength  
                INCREMENT freq AT pos BY 1  
        SORT freq IN descending order  
        SET result TO sum of first maxOnes elements in freq  
        RETURN result  
    END FUNCTION  
END CLASS
