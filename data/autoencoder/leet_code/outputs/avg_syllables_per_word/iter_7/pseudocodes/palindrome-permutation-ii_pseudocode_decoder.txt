CLASS Solution  
    FUNCTION generatePalindromes(s)  
        FUNCTION dfs(t)  
            IF LENGTH OF t IS EQUAL TO LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            FOR each character c AND count v IN cnt  
                IF v IS GREATER THAN 1  
                    DECREMENT cnt[c] BY 2  
                    CALL dfs WITH c CONCATENATED WITH t CONCATENATED WITH c  
                    INCREMENT cnt[c] BY 2  
        END FUNCTION  
        
        SET cnt TO count of characters in s  
        SET mid TO empty string  
        FOR each character c AND count v IN cnt  
            IF v MODULO 2 IS EQUAL TO 1  
                IF mid IS NOT empty  
                    RETURN empty list  
                SET mid TO c  
                DECREMENT cnt[c] BY 1  
        SET ans TO empty list  
        CALL dfs WITH mid  
        RETURN ans  
    END FUNCTION  
END CLASS
