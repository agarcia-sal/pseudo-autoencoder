CLASS Solution  
    FUNCTION maxSumOfThreeSubarrays(nums, k)  
        SET n TO LENGTH OF nums  
        INITIALIZE sums AS LIST OF ZEROS WITH LENGTH n MINUS k PLUS 1  
        SET current_sum TO SUM OF FIRST k ELEMENTS IN nums  
        SET sums AT INDEX 0 TO current_sum  
        
        FOR i FROM 1 TO n MINUS k  
            INCREMENT current_sum BY nums AT INDEX i PLUS k MINUS 1 MINUS nums AT INDEX i MINUS 1  
            SET sums AT INDEX i TO current_sum  
        
        INITIALIZE left AS LIST OF ZEROS WITH LENGTH n MINUS k PLUS 1  
        INITIALIZE right AS LIST FILLED WITH n MINUS k WITH LENGTH n MINUS k PLUS 1  
        
        SET max_sum TO 0  
        FOR i FROM 0 TO n MINUS k  
            IF sums AT INDEX i GREATER THAN sums AT INDEX max_sum  
                SET max_sum TO i  
            SET left AT INDEX i TO max_sum  
        
        SET max_sum TO n MINUS k  
        FOR i FROM n MINUS k DOWNTO 0  
            IF sums AT INDEX i GREATER THAN OR EQUAL TO sums AT INDEX max_sum  
                SET max_sum TO i  
            SET right AT INDEX i TO max_sum  
        
        SET max_total TO 0  
        INITIALIZE result AS LIST CONTAINING 0, 0, 0  
        FOR j FROM k TO n MINUS 2 TIMES k  
            SET i TO left AT INDEX j MINUS k  
            SET l TO right AT INDEX j PLUS k  
            SET total TO sums AT INDEX i PLUS sums AT INDEX j PLUS sums AT INDEX l  
            IF total GREATER THAN max_total  
                SET max_total TO total  
                SET result TO LIST CONTAINING i, j, l  
        
        RETURN result  
    END FUNCTION  
END CLASS
