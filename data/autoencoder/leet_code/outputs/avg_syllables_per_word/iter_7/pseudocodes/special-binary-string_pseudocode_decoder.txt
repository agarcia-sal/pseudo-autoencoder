CLASS Solution  
    FUNCTION makeLargestSpecial(s)  
        FUNCTION dfs(s)  
            SET count TO 0  
            SET start TO 0  
            SET specials TO empty list  
            
            FOR i, char IN s WITH INDEX  
                IF char EQUALS '1'  
                    INCREMENT count BY 1  
                ELSE  
                    DECREMENT count BY 1  
                END IF  
                
                IF count EQUALS 0  
                    APPEND '1' CONCATENATED WITH dfs(substring of s from start plus 1 to i minus 1) CONCATENATED WITH '0' TO specials  
                    SET start TO i plus 1  
                END IF  
            END FOR  
            
            SORT specials IN DESCENDING LEXICOGRAPHICAL ORDER  
            RETURN concatenation of all elements in specials  
        END FUNCTION  
        
        RETURN dfs(s)  
    END FUNCTION  
END CLASS
