CLASS Solution  
    FUNCTION findDuplicate(paths)  
        SET content_to_paths TO a dictionary with default empty list  
        FOR each path IN paths  
            SET parts TO list of substrings of path split by whitespace  
            SET directory TO first element of parts  
            FOR each file_info IN parts from second element onward  
                SPLIT file_info into file_name and content_start at character '('  
                REMOVE last character from content_start and SET to content  
                APPEND string formed by directory joined with file_name by '/' TO content_to_paths at content  
        SET result TO list of values from content_to_paths where length of value is GREATER THAN 1  
        RETURN result  
    END FUNCTION  
END CLASS
