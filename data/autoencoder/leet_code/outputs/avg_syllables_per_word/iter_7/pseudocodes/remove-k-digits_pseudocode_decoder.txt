CLASS Solution  
    FUNCTION removeKdigits(num, k)  
        SET stack TO empty list  
        
        FOR each digit IN num  
            WHILE k GREATER THAN 0 AND stack NOT EMPTY AND last element in stack GREATER THAN digit  
                REMOVE last element FROM stack  
                DECREMENT k BY 1  
            APPEND digit TO stack  
        
        IF k GREATER THAN 0  
            SET final_stack TO all elements of stack EXCEPT the last k elements  
        ELSE  
            SET final_stack TO stack  
        
        SET result TO concatenation of elements in final_stack WITH leading zeros REMOVED  
        
        IF result IS EMPTY  
            RETURN "0"  
        ELSE  
            RETURN result  
    END FUNCTION  
END CLASS
