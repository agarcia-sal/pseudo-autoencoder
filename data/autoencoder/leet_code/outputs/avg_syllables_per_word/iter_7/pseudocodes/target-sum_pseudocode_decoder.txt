CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS LENGTH OF nums THEN  
                RETURN 1 IF current_sum EQUALS target OTHERWISE 0  
            END IF  
            IF memo CONTAINS KEY (index, current_sum) THEN  
                RETURN memo VALUE AT (index, current_sum)  
            END IF  
            SET positive TO CALL dfs WITH (index INCREMENTED BY 1, current_sum INCREMENTED BY nums AT index, memo)  
            SET negative TO CALL dfs WITH (index INCREMENTED BY 1, current_sum DECREMENTED BY nums AT index, memo)  
            SET memo VALUE AT (index, current_sum) TO positive PLUS negative  
            RETURN memo VALUE AT (index, current_sum)  
        END FUNCTION  
        RETURN CALL dfs WITH (0, 0, EMPTY DICTIONARY)  
    END FUNCTION  
END CLASS
