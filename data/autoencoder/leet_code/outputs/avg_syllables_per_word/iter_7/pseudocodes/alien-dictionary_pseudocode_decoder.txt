CLASS Solution  
    FUNCTION alienOrder(words)  
        SET graph TO empty mapping from character to set of characters  
        SET indegree TO empty mapping from character to integer  
  
        SET all_chars TO set of all characters across all words  
  
        FOR i FROM 0 TO length of words MINUS 2  
            SET word1 TO words at index i  
            SET word2 TO words at index i PLUS 1  
            SET min_length TO minimum of length of word1 and length of word2  
            SET found TO false  
            FOR j FROM 0 TO min_length MINUS 1  
                IF character at index j in word1 IS NOT EQUAL TO character at index j in word2 THEN  
                    IF character at index j in word2 IS NOT IN graph mapped by character at index j in word1 THEN  
                        ADD character at index j in word2 TO graph mapped by character at index j in word1  
                        INCREMENT indegree mapped by character at index j in word2 BY 1  
                    SET found TO true  
                    BREAK the inner loop  
            IF found IS false AND length of word1 GREATER THAN length of word2 THEN  
                RETURN empty string  
  
        SET queue TO deque containing all characters in all_chars WHERE indegree mapped by character IS EQUAL TO 0  
        SET result TO empty list  
  
        WHILE queue IS NOT empty  
            SET char TO first element removed from queue  
            APPEND char TO result  
            FOR each neighbor IN graph mapped by char  
                DECREMENT indegree mapped by neighbor BY 1  
                IF indegree mapped by neighbor IS EQUAL TO 0 THEN  
                    APPEND neighbor TO queue  
  
        IF length of result IS NOT EQUAL TO length of all_chars THEN  
            RETURN empty string  
  
        RETURN concatenation of characters in result  
    END FUNCTION  
END CLASS
