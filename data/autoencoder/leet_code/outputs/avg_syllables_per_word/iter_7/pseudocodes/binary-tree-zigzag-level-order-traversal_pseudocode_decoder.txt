CLASS TreeNode  
    FUNCTION __init__(self, val SET TO 0, left SET TO None, right SET TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode WITH val AS values AT position 0  
    SET i TO 1  
    SET queue TO new empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO queue REMOVE first element  
        IF i IS LESS THAN length of values AND values AT position i IS NOT None  
            SET node.left TO new TreeNode WITH val AS values AT position i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN length of values AND values AT position i IS NOT None  
            SET node.right TO new TreeNode WITH val AS values AT position i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self, root)  
        IF root IS None  
            RETURN empty list  
        SET result TO empty list  
        SET queue TO new queue CONTAINING root  
        SET left_to_right TO True  
        WHILE queue IS NOT empty  
            SET level_size TO length of queue  
            SET current_level TO new empty deque  
            FOR _ IN range FROM 0 TO level_size MINUS 1  
                SET node TO queue REMOVE first element  
                IF left_to_right IS True  
                    APPEND node.val TO current_level  
                ELSE  
                    APPEND node.val TO FRONT OF current_level  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
            END FOR  
            APPEND list form of current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
