CLASS Solution  
    FUNCTION checkRecord(n)  
        SET MOD TO 10 raised to the power 9 plus 7  
        INITIALIZE dp AS three-dimensional list of size (n plus 1) by 2 by 3 filled with 0  
        SET dp at index 0, 0, 0 TO 1  
        FOR i FROM 1 TO n INCLUSIVE  
            FOR j FROM 0 TO 1 INCLUSIVE  
                FOR k FROM 0 TO 2 INCLUSIVE  
                    SET dp at i, j, 0 TO dp at i, j, 0 PLUS dp at i minus 1, j, k MODULO MOD  
                    IF j EQUALS 0 THEN  
                        SET dp at i, 1, 0 TO dp at i, 1, 0 PLUS dp at i minus 1, 0, k MODULO MOD  
                    END IF  
                    IF k LESS THAN 2 THEN  
                        SET dp at i, j, k plus 1 TO dp at i, j, k plus 1 PLUS dp at i minus 1, j, k MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        SET result TO 0  
        FOR j FROM 0 TO 1 INCLUSIVE  
            FOR k FROM 0 TO 2 INCLUSIVE  
                SET result TO result PLUS dp at n, j, k MODULO MOD  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
