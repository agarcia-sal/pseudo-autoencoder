CLASS Solution  
    FUNCTION calculate(s)  
        FUNCTION helper(s_list)  
            SET stack TO empty list  
            SET num TO 0  
            SET sign TO plus symbol  
            WHILE LENGTH OF s_list GREATER THAN 0  
                SET char TO first element REMOVED FROM s_list  
                IF char IS a digit  
                    SET num TO num MULTIPLIED BY 10 PLUS numeric value of char  
                END IF  
                IF char IS open parenthesis  
                    SET num TO RESULT OF helper(s_list)  
                END IF  
                IF char IS one of plus, minus, multiply, divide, open parenthesis, close parenthesis OR LENGTH OF s_list EQUALS 0  
                    IF sign EQUALS plus symbol  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus symbol  
                        APPEND negation of num TO stack  
                    ELSE IF sign EQUALS multiply symbol  
                        SET last element of stack TO last element MULTIPLIED BY num  
                    ELSE IF sign EQUALS divide symbol  
                        SET last element of stack TO integer part of last element DIVIDED BY num  
                    END IF  
                    SET sign TO char  
                    SET num TO 0  
                END IF  
                IF char IS close parenthesis  
                    BREAK loop  
                END IF  
            END WHILE  
            RETURN sum of stack  
        END FUNCTION  
        
        SET s TO list of characters from s  
        RETURN helper(s)  
    END FUNCTION  
END CLASS
