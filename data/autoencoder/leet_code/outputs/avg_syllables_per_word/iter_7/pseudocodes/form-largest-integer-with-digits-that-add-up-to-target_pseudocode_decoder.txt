CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        SET cost_to_digit TO DICTIONARY mapping each element c in cost TO string representation of its index plus one  
        SET dp TO LIST containing empty string FOLLOWED BY target times "-1" string  
        FOR t FROM 1 TO target INCLUSIVE  
            FOR each cost c AND digit d IN cost_to_digit  
                IF t GREATER THAN OR EQUAL TO c AND dp AT position t MINUS c NOT EQUAL TO "-1"  
                    SET candidate TO concatenation of d AND dp AT position t MINUS c  
                    IF dp AT position t EQUALS "-1" OR LENGTH OF candidate GREATER THAN LENGTH OF dp AT position t OR (LENGTH OF candidate EQUALS LENGTH OF dp AT position t AND candidate GREATER THAN dp AT position t)  
                        SET dp AT position t TO candidate  
        IF dp AT position target NOT EQUAL TO "-1"  
            RETURN dp AT position target  
        ELSE  
            RETURN "0"  
    END FUNCTION  
END CLASS
