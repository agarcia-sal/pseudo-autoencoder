CLASS Solution  
    FUNCTION areConnected(n, threshold, queries)  
        IF threshold EQUALS 0 THEN  
            RETURN list of True repeated LENGTH OF queries times  
        END IF  
        
        SET parent TO list of integers from 0 to n inclusive  
        
        FUNCTION find(x)  
            IF parent at position x is NOT EQUAL TO x THEN  
                SET parent at position x TO find(parent at position x)  
            END IF  
            RETURN parent at position x  
        END FUNCTION  
        
        FUNCTION union(x, y)  
            SET rootX TO find(x)  
            SET rootY TO find(y)  
            IF rootX is NOT EQUAL TO rootY THEN  
                SET parent at position rootY TO rootX  
            END IF  
        END FUNCTION  
        
        FOR i FROM threshold PLUS 1 TO n INCLUSIVE  
            FOR j FROM 2 TIMES i TO n INCLUSIVE STEP i  
                CALL union(i, j)  
            END FOR  
        END FOR  
        
        SET result TO empty list  
        FOR each pair a, b IN queries  
            APPEND (find(a) EQUALS find(b)) TO result  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
