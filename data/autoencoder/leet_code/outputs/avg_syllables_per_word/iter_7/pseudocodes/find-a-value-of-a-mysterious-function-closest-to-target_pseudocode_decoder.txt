CLASS Solution  
    FUNCTION closestToTarget(arr, target)  
        SET seen TO empty set  
        SET min_diff TO infinity  
        
        FOR each num IN arr  
            SET seen TO union of set formed by bitwise AND of num and each element in seen WITH set containing num  
            
            FOR each val IN seen  
                SET min_diff TO smaller of min_diff AND absolute difference between val AND target  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
