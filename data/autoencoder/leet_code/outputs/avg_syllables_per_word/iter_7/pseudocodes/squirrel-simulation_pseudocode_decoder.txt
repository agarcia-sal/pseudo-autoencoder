CLASS Solution  
    FUNCTION minDistance(height, width, tree, squirrel, nuts)  
        FUNCTION distance(p1, p2)  
            RETURN absolute value of (p1 at position zero MINUS p2 at position zero) PLUS absolute value of (p1 at position one MINUS p2 at position one)  
        END FUNCTION  
        
        SET total_distance TO zero  
        SET max_save TO negative infinity  
        
        FOR each nut IN nuts  
            INCREMENT total_distance BY two TIMES distance between nut and tree  
            SET first_nut_distance TO distance between squirrel and nut PLUS distance between nut and tree  
            SET later_nut_distance TO two TIMES distance between nut and tree  
            SET save TO later_nut_distance MINUS first_nut_distance  
            SET max_save TO greater value between max_save AND save  
        END FOR  
        
        RETURN total_distance MINUS max_save  
    END FUNCTION  
END CLASS
