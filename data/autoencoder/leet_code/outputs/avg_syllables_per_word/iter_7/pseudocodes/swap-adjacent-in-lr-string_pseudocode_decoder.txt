CLASS Solution  
    FUNCTION canTransform(start, end)  
        SET start_filtered TO list of pairs (character, index) for each character and its index in start where character is not 'X'  
        SET end_filtered TO list of pairs (character, index) for each character and its index in end where character is not 'X'  
        
        IF LENGTH OF start_filtered IS NOT EQUAL TO LENGTH OF end_filtered THEN  
            RETURN False  
        
        FOR each pair (c1, i1) and (c2, i2) IN parallel from start_filtered and end_filtered  
            IF c1 IS NOT EQUAL TO c2 THEN  
                RETURN False  
            IF c1 IS 'L' AND i1 IS LESS THAN i2 THEN  
                RETURN False  
            IF c1 IS 'R' AND i1 IS GREATER THAN i2 THEN  
                RETURN False  
        
        RETURN True  
    END FUNCTION  
END CLASS
