CLASS Solution  
    FUNCTION search(nums, target)  
        SET left TO 0  
        SET right TO LENGTH OF nums MINUS 1  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO left PLUS (right MINUS left) DIVIDED BY 2  
            
            IF VALUE AT nums POSITION mid IS EQUAL TO target  
                RETURN mid  
            END IF  
            
            IF VALUE AT nums POSITION left IS LESS THAN OR EQUAL TO VALUE AT nums POSITION mid  
                IF VALUE AT nums POSITION left IS LESS THAN OR EQUAL TO target AND target IS LESS THAN VALUE AT nums POSITION mid  
                    SET right TO mid MINUS 1  
                ELSE  
                    SET left TO mid PLUS 1  
                END IF  
            ELSE  
                IF VALUE AT nums POSITION mid IS LESS THAN target AND target IS LESS THAN OR EQUAL TO VALUE AT nums POSITION right  
                    SET left TO mid PLUS 1  
                ELSE  
                    SET right TO mid MINUS 1  
                END IF  
            END IF  
        END WHILE  
        
        RETURN MINUS 1  
    END FUNCTION  
END CLASS
