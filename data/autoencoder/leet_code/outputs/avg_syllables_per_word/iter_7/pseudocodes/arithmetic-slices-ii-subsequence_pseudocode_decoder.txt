CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO LENGTH OF nums  
        IF n LESS THAN 3 THEN  
            RETURN 0  
        END IF  
        
        SET dp TO LIST OF n DICTIONARIES WITH DEFAULT VALUE 0  
        SET total_count TO 0  
        
        FOR i FROM 0 TO n MINUS 1 DO  
            FOR j FROM 0 TO i MINUS 1 DO  
                SET d TO nums AT i MINUS nums AT j  
                IF d EXISTS IN dp AT j THEN  
                    INCREMENT dp AT i FOR d BY dp AT j FOR d PLUS 1  
                    INCREMENT total_count BY dp AT j FOR d  
                ELSE  
                    INCREMENT dp AT i FOR d BY 1  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
