CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO LENGTH OF word1  
        SET n TO LENGTH OF word2  
        SET dp TO 2D array of size (m plus 1) by (n plus 1) with all elements initialized to 0  
        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF character at position (i minus 1) in word1 EQUALS character at position (j minus 1) in word2 THEN  
                    SET dp at position [i][j] TO dp at position [i minus 1][j minus 1] plus 1  
                ELSE  
                    SET dp at position [i][j] TO maximum of dp at position [i minus 1][j] and dp at position [i][j minus 1]  
        SET lcs_length TO dp at position [m][n]  
        RETURN (m minus lcs_length) plus (n minus lcs_length)  
    END FUNCTION  
END CLASS
