CLASS Solution  
    FUNCTION jump(nums)  
        IF LENGTH OF nums LESS THAN OR EQUAL TO 1  
            RETURN 0  
        END IF  
        
        SET jumps TO 0  
        SET current_end TO 0  
        SET farthest TO 0  
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 2  
            SET farthest TO GREATER OF farthest AND i PLUS nums AT INDEX i  
            
            IF i EQUALS current_end  
                INCREMENT jumps BY 1  
                SET current_end TO farthest  
                
                IF current_end GREATER THAN OR EQUAL TO LENGTH OF nums MINUS 1  
                    BREAK LOOP  
                END IF  
            END IF  
        END FOR  
        
        RETURN jumps  
    END FUNCTION  
END CLASS
