CLASS Solution  
    FUNCTION insert(intervals, newInterval)  
        IF intervals IS empty  
            RETURN list containing newInterval  
        END IF  
        
        SET merged TO empty list  
        SET i TO 0  
        SET n TO length of intervals  
        
        WHILE i LESS THAN n AND intervals at i second element LESS THAN newInterval first element  
            APPEND intervals at i TO merged  
            INCREMENT i BY 1  
        END WHILE  
        
        WHILE i LESS THAN n AND intervals at i first element LESS THAN OR EQUAL TO newInterval second element  
            SET newInterval first element TO minimum of newInterval first element AND intervals at i first element  
            SET newInterval second element TO maximum of newInterval second element AND intervals at i second element  
            INCREMENT i BY 1  
        END WHILE  
        
        APPEND newInterval TO merged  
        
        WHILE i LESS THAN n  
            APPEND intervals at i TO merged  
            INCREMENT i BY 1  
        END WHILE  
        
        RETURN merged  
    END FUNCTION  
END CLASS
