CLASS Solution  
    FUNCTION minimumOneBitOperations(n)  
        IF n IS EQUAL TO 0  
            RETURN 0  
        END IF  
        
        SET k TO 0  
        WHILE (1 SHIFTED LEFT BY (k PLUS 1)) IS LESS THAN OR EQUAL TO n  
            INCREMENT k BY 1  
        END WHILE  
        
        SET most_significant_bit TO 1 SHIFTED LEFT BY k  
        
        RETURN (1 SHIFTED LEFT BY (k PLUS 1)) MINUS 1 MINUS minimumOneBitOperations(n XOR most_significant_bit)  
    END FUNCTION  
END CLASS
