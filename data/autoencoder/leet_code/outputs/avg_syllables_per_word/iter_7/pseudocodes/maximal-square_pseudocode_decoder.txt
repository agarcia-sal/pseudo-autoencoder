CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix IS EMPTY OR matrix AT INDEX 0 IS EMPTY  
            RETURN 0  
        END IF  

        SET m TO LENGTH OF matrix  
        SET n TO LENGTH OF matrix AT INDEX 0  
        SET dp TO A NEW TWO-DIMENSIONAL LIST OF SIZE (m PLUS 1) BY (n PLUS 1) FILLED WITH ZEROS  
        SET max_side TO 0  

        FOR i FROM 1 TO m INCLUSIVE  
            FOR j FROM 1 TO n INCLUSIVE  
                IF matrix AT INDEX (i MINUS 1) AT INDEX (j MINUS 1) IS EQUAL TO '1'  
                    SET dp AT INDEX i AT INDEX j TO MINIMUM OF (dp AT INDEX (i MINUS 1) AT INDEX j, dp AT INDEX i AT INDEX (j MINUS 1), dp AT INDEX (i MINUS 1) AT INDEX (j MINUS 1)) PLUS 1  
                    SET max_side TO MAXIMUM OF max_side AND dp AT INDEX i AT INDEX j  
                END IF  
            END FOR  
        END FOR  

        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
