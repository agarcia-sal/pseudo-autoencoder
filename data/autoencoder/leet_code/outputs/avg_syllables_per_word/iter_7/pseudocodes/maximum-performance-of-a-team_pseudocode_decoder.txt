CLASS Solution  
    FUNCTION maxPerformance(n, speed, efficiency, k)  
        SET engineers TO list of pairs combining efficiency and speed  
        SORT engineers BY efficiency in descending order  
        
        SET speed_heap TO empty min-heap  
        SET speed_sum TO 0  
        SET max_performance TO 0  
        
        FOR each eff, spd IN engineers  
            ADD spd TO speed_heap  
            INCREMENT speed_sum BY spd  
            
            IF size of speed_heap GREATER THAN k  
                REMOVE smallest speed FROM speed_heap  
                DECREMENT speed_sum BY that removed speed  
            
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            SET max_performance TO the greater of max_performance and current_performance  
        
        RETURN max_performance MODULO 10 TO THE POWER OF 9 PLUS 7  
    END FUNCTION  
END CLASS
