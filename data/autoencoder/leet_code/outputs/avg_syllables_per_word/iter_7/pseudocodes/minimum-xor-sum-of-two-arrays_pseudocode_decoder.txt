CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO LENGTH OF nums1  
        SET dp TO LIST OF size 2 TO THE POWER OF n FILLED WITH infinity  
        SET dp AT index 0 TO 0  
        
        FOR mask FROM 0 TO 2 TO THE POWER OF n MINUS 1  
            SET bit_count TO NUMBER OF ONES IN BINARY REPRESENTATION OF mask  
            FOR j FROM 0 TO n MINUS 1  
                IF mask INCLUDES BIT AT POSITION j  
                    SET dp AT mask TO MINIMUM OF dp AT mask AND (dp AT mask WITH BIT j TURNED OFF PLUS nums1 AT bit_count MINUS 1 XOR nums2 AT j)  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp AT 2 TO THE POWER OF n MINUS 1  
    END FUNCTION  
END CLASS
