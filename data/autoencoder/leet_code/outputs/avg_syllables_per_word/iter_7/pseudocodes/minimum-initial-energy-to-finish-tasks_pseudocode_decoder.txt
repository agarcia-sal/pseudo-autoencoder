CLASS Solution  
    FUNCTION minimumEffort(tasks)  
        SORT tasks BY difference BETWEEN minimum energy required AND actual energy spent IN descending order  
        SET current_energy TO 0  
        SET required_initial_energy TO 0  
        FOR each task WITH actual energy AND minimum energy IN tasks  
            IF current_energy IS LESS THAN minimum energy  
                INCREMENT required_initial_energy BY difference BETWEEN minimum energy AND current_energy  
                SET current_energy TO minimum energy  
            DECREMENT current_energy BY actual energy  
        RETURN required_initial_energy  
    END FUNCTION  
END CLASS
