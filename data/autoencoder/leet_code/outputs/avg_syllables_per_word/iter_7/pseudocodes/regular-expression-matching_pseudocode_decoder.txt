CLASS Solution  
    FUNCTION isMatch(s, p)  
        INITIALIZE dp TO two-dimensional list of False with dimensions LENGTH OF s plus one BY LENGTH OF p plus one  
        SET dp at position zero zero TO True  
  
        FOR j FROM one TO LENGTH OF p  
            IF character at position j minus one in p EQUALS '*', THEN  
                SET dp at position zero j TO dp at position zero j minus two  
  
        FOR i FROM one TO LENGTH OF s  
            FOR j FROM one TO LENGTH OF p  
                IF character at position j minus one in p EQUALS character at position i minus one in s OR character at position j minus one in p EQUALS '.', THEN  
                    SET dp at position i j TO dp at position i minus one j minus one  
                ELSE IF character at position j minus one in p EQUALS '*', THEN  
                    SET dp at position i j TO dp at position i j minus two  
                    IF character at position j minus two in p EQUALS character at position i minus one in s OR character at position j minus two in p EQUALS '.', THEN  
                        SET dp at position i j TO dp at position i j OR dp at position i minus one j  
  
        RETURN dp at position LENGTH OF s LENGTH OF p  
    END FUNCTION  
END CLASS
