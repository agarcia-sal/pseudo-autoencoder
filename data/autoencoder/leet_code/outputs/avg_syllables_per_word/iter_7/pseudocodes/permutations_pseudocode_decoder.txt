CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF remaining IS EMPTY  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR each index i FROM 0 TO LENGTH OF remaining MINUS 1  
                SET new_path TO path WITH remaining AT index i APPENDED  
                SET new_remaining TO remaining WITHOUT element AT index i  
                CALL backtrack(new_path, new_remaining)  
            END FOR  
        END FUNCTION  
        
        SET permutations TO EMPTY LIST  
        CALL backtrack(EMPTY LIST, nums)  
        RETURN permutations  
    END FUNCTION  
END CLASS
