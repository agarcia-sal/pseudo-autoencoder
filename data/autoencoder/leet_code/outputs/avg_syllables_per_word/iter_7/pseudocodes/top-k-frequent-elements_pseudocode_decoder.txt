CLASS Solution  
    FUNCTION topKFrequent(nums, k)  
        SET frequency_map TO new map with default value 0  
        FOR each num IN nums  
            INCREMENT frequency_map at num BY 1  
        SET max_frequency TO the maximum value in frequency_map  
        SET buckets TO list of empty lists with length max_frequency PLUS 1  
        FOR each num, freq IN frequency_map entries  
            APPEND num TO buckets at index freq  
        SET result TO empty list  
        FOR freq FROM max_frequency DOWN TO 1  
            IF length of buckets at freq GREATER THAN 0 THEN  
                EXTEND result BY buckets at freq  
                IF length of result GREATER THAN OR EQUAL TO k THEN  
                    RETURN first k elements of result  
        RETURN result  
    END FUNCTION  
END CLASS
