CLASS Solution  
    FUNCTION lengthOfLongestSubstring(s)  
        SET char_index_map TO empty map  
        SET max_length TO 0  
        SET start TO 0  
        
        FOR each end, char IN enumeration of s  
            IF char EXISTS IN char_index_map AND char_index_map at char IS GREATER THAN OR EQUAL TO start  
                SET start TO char_index_map at char PLUS 1  
            SET char_index_map at char TO end  
            SET max_length TO maximum of max_length AND end MINUS start PLUS 1  
        RETURN max_length  
    END FUNCTION  
END CLASS
