CLASS Solution  
    FUNCTION checkValidString(s)  
        SET low TO 0  
        SET high TO 0  

        FOR each char IN s  
            IF char IS '('  
                INCREMENT low BY 1  
                INCREMENT high BY 1  
            ELSE IF char IS ')'  
                DECREMENT low BY 1  
                DECREMENT high BY 1  
            ELSE  
                DECREMENT low BY 1  
                INCREMENT high BY 1  
            END IF  

            IF high IS LESS THAN 0  
                RETURN False  
            END IF  

            IF low IS LESS THAN 0  
                SET low TO 0  
            END IF  
        END FOR  

        RETURN low IS EQUAL TO 0  
    END FUNCTION  
END CLASS
