CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO LENGTH OF nums DIVIDED BY 2  
        SET total_sum TO SUM OF nums  
        SET target TO total_sum DIVIDED BY 2  
        SET min_diff TO infinity  
        
        SET left_half TO FIRST n ELEMENTS OF nums  
        SET right_half TO ELEMENTS OF nums FROM n TO END  
        
        SET left_sums TO LIST OF EMPTY DICTIONARIES OF SIZE n PLUS 1  
        SET right_sums TO LIST OF EMPTY DICTIONARIES OF SIZE n PLUS 1  
        
        FOR i FROM 0 TO n INCLUSIVE  
            FOR EACH combination OF SIZE i IN left_half  
                SET subset_sum TO SUM OF combination  
                IF subset_sum NOT IN left_sums AT INDEX i  
                    SET left_sums AT INDEX i AND KEY subset_sum TO 0  
                INCREMENT left_sums AT INDEX i AND KEY subset_sum BY 1  
            
            FOR EACH combination OF SIZE i IN right_half  
                SET subset_sum TO SUM OF combination  
                IF subset_sum NOT IN right_sums AT INDEX i  
                    SET right_sums AT INDEX i AND KEY subset_sum TO 0  
                INCREMENT right_sums AT INDEX i AND KEY subset_sum BY 1  
        
        FOR i FROM 0 TO n INCLUSIVE  
            SET left_values TO SORTED KEYS OF left_sums AT INDEX i  
            SET right_values TO SORTED KEYS OF right_sums AT INDEX n MINUS i  
            
            SET j TO 0  
            SET k TO LENGTH OF right_values MINUS 1  
            WHILE j LESS THAN LENGTH OF left_values AND k GREATER THAN OR EQUAL TO 0  
                SET current_sum TO left_values AT INDEX j PLUS right_values AT INDEX k  
                SET candidate_diff TO ABSOLUTE VALUE OF total_sum MINUS 2 TIMES current_sum  
                IF candidate_diff LESS THAN min_diff  
                    SET min_diff TO candidate_diff  
                
                IF current_sum LESS THAN target  
                    INCREMENT j BY 1  
                ELSE IF current_sum GREATER THAN target  
                    DECREMENT k BY 1  
                ELSE  
                    RETURN 0  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
