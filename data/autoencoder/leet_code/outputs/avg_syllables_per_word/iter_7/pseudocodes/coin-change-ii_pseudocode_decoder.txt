CLASS Solution  
    FUNCTION change(amount, coins)  
        SET dp TO list of zeros with length amount PLUS 1  
        SET dp at index 0 TO 1  
        
        FOR each coin IN coins  
            FOR each x FROM coin TO amount  
                INCREMENT dp at index x BY dp at index x MINUS coin  
        
        RETURN dp at index amount  
    END FUNCTION  
END CLASS
