CLASS Solution  
    FUNCTION rearrangeSticks(n, k)  
        SET MOD TO 10 raised to the power 9 plus 7  
        
        INITIALIZE dp AS a two-dimensional list of size (n plus 1) by (k plus 1) filled with zeros  
        
        SET dp at position 0, 0 TO 1  
        
        FOR i FROM 1 TO n inclusive  
            FOR j FROM 1 TO k inclusive  
                SET dp at position i, j TO (dp at position i minus 1, j minus 1 plus (i minus 1) multiplied by dp at position i minus 1, j) MODULO MOD  
        
        RETURN dp at position n, k  
    END FUNCTION  
END CLASS
