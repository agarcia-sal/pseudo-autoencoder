CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF LENGTH OF nums LESS THAN 2 THEN  
            RETURN 0  
        END IF  
        
        FUNCTION countingSort(arr, exp)  
            SET n TO LENGTH OF arr  
            SET output TO LIST OF SIZE n FILLED WITH 0  
            SET count TO LIST OF SIZE 10 FILLED WITH 0  
            
            FOR i FROM 0 TO n MINUS 1  
                SET index TO arr[i] DIVIDED BY exp  
                INCREMENT count[index MODULO 10] BY 1  
            END FOR  
            
            FOR i FROM 1 TO 9  
                SET count[i] TO count[i] PLUS count[i MINUS 1]  
            END FOR  
            
            SET i TO n MINUS 1  
            WHILE i GREATER THAN OR EQUAL TO 0  
                SET index TO arr[i] DIVIDED BY exp  
                SET output[count[index MODULO 10] MINUS 1] TO arr[i]  
                DECREMENT count[index MODULO 10] BY 1  
                DECREMENT i BY 1  
            END WHILE  
            
            FOR i FROM 0 TO n MINUS 1  
                SET arr[i] TO output[i]  
            END FOR  
        END FUNCTION  
        
        FUNCTION radixSort(arr)  
            SET max1 TO MAXIMUM VALUE IN arr  
            SET exp TO 1  
            WHILE max1 DIVIDED BY exp GREATER THAN 0  
                CALL countingSort(arr, exp)  
                SET exp TO exp MULTIPLIED BY 10  
            END WHILE  
        END FUNCTION  
        
        CALL radixSort(nums)  
        
        SET max_diff TO 0  
        FOR i FROM 1 TO LENGTH OF nums MINUS 1  
            SET current_diff TO nums[i] MINUS nums[i MINUS 1]  
            IF current_diff GREATER THAN max_diff THEN  
                SET max_diff TO current_diff  
            END IF  
        END FOR  
        
        RETURN max_diff  
    END FUNCTION  
END CLASS
