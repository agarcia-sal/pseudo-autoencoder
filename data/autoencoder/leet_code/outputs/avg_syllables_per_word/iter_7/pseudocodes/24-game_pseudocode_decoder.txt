CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF LENGTH OF nums EQUALS 1  
                RETURN absolute value of nums at position 0 MINUS 24 LESS THAN 0.000001  
            END IF  
            
            FOR i FROM 0 TO LENGTH OF nums MINUS 1  
                FOR j FROM 0 TO LENGTH OF nums MINUS 1  
                    IF i NOT EQUAL TO j  
                        SET new_nums TO list of nums at positions k FOR k FROM 0 TO LENGTH OF nums MINUS 1 IF k NOT EQUAL TO i AND k NOT EQUAL TO j  
                        
                        FOR each op IN list of operations [true division, multiplication, addition, subtraction]  
                            IF (op IS addition OR op IS multiplication) AND j GREATER THAN i  
                                CONTINUE to next iteration  
                            END IF  
                            
                            IF op IS NOT true division OR nums at position j NOT EQUAL TO 0  
                                APPEND result of applying op TO nums at positions i and j TO new_nums  
                                IF solve(new_nums) RETURNS True  
                                    RETURN True  
                                END IF  
                                REMOVE last element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            
            RETURN False  
        END FUNCTION  
        
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
