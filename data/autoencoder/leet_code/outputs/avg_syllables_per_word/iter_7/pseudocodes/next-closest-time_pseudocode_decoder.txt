CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO set of characters in time WITHOUT colon  
        SET hour, minute TO integer values of time SPLIT BY colon  
        
        FUNCTION generate_times(digits)  
            SET times TO empty list  
            FOR each h1 IN digits  
                FOR each h2 IN digits  
                    FOR each m1 IN digits  
                        FOR each m2 IN digits  
                            SET hour_value TO h1 CONCATENATED WITH h2 AS integer  
                            SET minute_value TO m1 CONCATENATED WITH m2 AS integer  
                            IF hour_value LESS THAN 24 AND minute_value LESS THAN 60 THEN  
                                APPEND (hour_value, minute_value) TO times  
            RETURN times SORTED  
        END FUNCTION  
        
        SET all_times TO generate_times(digits)  
        SET current_time_index TO POSITION OF (hour, minute) IN all_times  
        
        IF current_time_index PLUS 1 LESS THAN LENGTH OF all_times THEN  
            SET next_hour, next_minute TO ELEMENT AT current_time_index PLUS 1 IN all_times  
        ELSE  
            SET next_hour, next_minute TO FIRST ELEMENT IN all_times  
        RETURN formatted string OF next_hour AND next_minute AS two-digit hour COLON two-digit minute  
    END FUNCTION  
END CLASS
