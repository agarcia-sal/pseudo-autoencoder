CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuples converted from board  
        SET target TO tuple containing (1, 2, 3) and (4, 5, 0)  
        
        SET neighbors TO mapping:  
            0 maps to set containing 1 and 3  
            1 maps to set containing 0, 2, and 4  
            2 maps to set containing 1 and 5  
            3 maps to set containing 0 and 4  
            4 maps to set containing 1, 3, and 5  
            5 maps to set containing 2 and 4  
        
        FOR i FROM 0 TO 1 INCREMENT BY 1  
            FOR j FROM 0 TO 2 INCREMENT BY 1  
                IF element at start[i][j] EQUALS 0 THEN  
                    SET zero_pos TO i MULTIPLIED BY 3 PLUS j  
        
        SET queue TO new deque containing tuple of (start, zero_pos, 0)  
        SET visited TO set containing start  
        
        WHILE queue is not empty DO  
            REMOVE first element from queue and SET to (state, zero_pos, moves)  
            
            IF state EQUALS target THEN  
                RETURN moves  
            
            FOR each neighbor in neighbors mapped from zero_pos DO  
                SET new_state TO list of lists converted from state  
                SET i TO zero_pos DIVIDED BY 3  
                SET j TO zero_pos MODULO 3  
                SET ni TO neighbor DIVIDED BY 3  
                SET nj TO neighbor MODULO 3  
                
                SWAP elements at new_state[i][j] and new_state[ni][nj]  
                
                SET new_tuple_state TO tuple of tuples converted from new_state  
                
                IF new_tuple_state NOT IN visited THEN  
                    ADD new_tuple_state TO visited  
                    APPEND tuple (new_tuple_state, neighbor, moves INCREMENTED BY 1) TO queue  
        
        RETURN negative 1  
    END FUNCTION  
END CLASS
