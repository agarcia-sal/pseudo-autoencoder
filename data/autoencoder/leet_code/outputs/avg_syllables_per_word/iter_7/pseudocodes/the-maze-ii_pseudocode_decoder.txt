CLASS Solution  
    FUNCTION shortestDistance(maze, start, destination)  
        SET m TO LENGTH OF maze  
        SET n TO LENGTH OF maze AT 0  
        SET directions TO LIST OF (1, 0), (-1, 0), (0, 1), (0, -1)  
        CONVERT start TO TUPLE  
        CONVERT destination TO TUPLE  
        SET visited TO EMPTY SET  
        SET pq TO LIST WITH ONE ELEMENT (0, start)  

        WHILE pq IS NOT EMPTY  
            REMOVE AND RETURN THE SMALLEST ELEMENT (dist, (x, y)) FROM pq  
            
            IF (x, y) EQUALS destination THEN  
                RETURN dist  
            END IF  
            
            IF (x, y) IS IN visited THEN  
                CONTINUE TO NEXT ITERATION  
            END IF  
            
            ADD (x, y) TO visited  
            
            FOR EACH (dx, dy) IN directions  
                SET new_x, new_y, count TO x, y, 0  
                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO 0 AND new_x PLUS dx IS LESS THAN m  
                      AND new_y PLUS dy IS GREATER THAN OR EQUAL TO 0 AND new_y PLUS dy IS LESS THAN n  
                      AND maze AT (new_x PLUS dx) AND (new_y PLUS dy) EQUALS 0  
                    SET new_x TO new_x PLUS dx  
                    SET new_y TO new_y PLUS dy  
                    INCREMENT count BY 1  
                END WHILE  
                
                IF (new_x, new_y) IS NOT IN visited THEN  
                    ADD (dist PLUS count, (new_x, new_y)) TO pq  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN -1  
    END FUNCTION  
END CLASS
