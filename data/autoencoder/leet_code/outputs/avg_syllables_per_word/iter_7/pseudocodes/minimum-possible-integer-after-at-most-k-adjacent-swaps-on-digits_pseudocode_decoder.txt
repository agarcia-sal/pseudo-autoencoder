CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO list of characters of num  
        SET n TO length of num  
        
        WHILE k IS GREATER THAN 0  
            SET swapped TO False  
            FOR i FROM 0 TO n MINUS 2  
                IF num at i IS GREATER THAN num at i PLUS 1  
                    SET min_digit TO character '9'  
                    SET min_index TO minus 1  
                    FOR j FROM i TO minimum of i PLUS k PLUS 1 AND n MINUS 1  
                        IF num at j IS LESS THAN min_digit  
                            SET min_digit TO num at j  
                            SET min_index TO j  
                    SET num slice from i TO min_index PLUS 1 TO list starting with num at min_index FOLLOWED BY num slice from i TO min_index  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO True  
                    BREAK out of FOR loop  
            IF swapped IS False  
                BREAK out of WHILE loop  
        
        RETURN concatenation of characters in num  
    END FUNCTION  
END CLASS
