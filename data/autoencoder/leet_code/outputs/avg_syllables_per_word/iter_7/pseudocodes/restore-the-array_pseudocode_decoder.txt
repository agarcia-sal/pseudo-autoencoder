CLASS Solution  
    FUNCTION numberOfArrays(s, k)  
        SET MOD TO 10 RAISED TO THE POWER OF 9 PLUS 7  
        SET n TO LENGTH OF s  
        CREATE dp AS LIST OF SIZE n PLUS 1 FILLED WITH ZERO  
        SET dp AT 0 TO 1  
        
        FOR i FROM 1 TO n  
            FOR j FROM 0 TO i MINUS 1  
                IF CHARACTER AT s AT j IS NOT EQUAL TO '0' AND  
                   INTEGER VALUE OF SUBSTRING OF s FROM j TO i IS GREATER THAN OR EQUAL TO 1 AND LESS THAN OR EQUAL TO k THEN  
                    SET dp AT i TO dp AT i PLUS dp AT j MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp AT n  
    END FUNCTION  
END CLASS
