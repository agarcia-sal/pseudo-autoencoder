CLASS Solution  
    FUNCTION minEatingSpeed(piles, h)  
        FUNCTION canFinish(k)  
            SET hours TO 0  
            FOR each pile IN piles  
                INCREMENT hours BY ceiling of pile DIVIDED BY k  
            IF hours IS LESS THAN OR EQUAL TO h  
                RETURN true  
            ELSE  
                RETURN false  
        END FUNCTION  

        SET left TO 1  
        SET right TO maximum value in piles  

        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO integer division of (left PLUS right) BY 2  
            IF canFinish(mid) IS true  
                SET right TO mid MINUS 1  
            ELSE  
                SET left TO mid PLUS 1  
        
        RETURN left  
    END FUNCTION  
END CLASS
