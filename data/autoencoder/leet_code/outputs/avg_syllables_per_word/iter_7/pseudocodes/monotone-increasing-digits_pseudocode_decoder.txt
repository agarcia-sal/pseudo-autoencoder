CLASS Solution  
    FUNCTION monotoneIncreasingDigits(n)  
        SET digits TO list of characters in string representation of n  
        SET length TO size of digits  
        SET marker TO length  
        FOR i FROM length MINUS 1 DOWNTO 1  
            IF digit at i LESS THAN digit at i MINUS 1 THEN  
                SET digit at i MINUS 1 TO string representation of integer value of digit at i MINUS 1 MINUS 1  
                SET marker TO i  
            END IF  
        END FOR  
        FOR i FROM marker TO length MINUS 1  
            SET digit at i TO character '9'  
        END FOR  
        RETURN integer value of concatenation of digits  
    END FUNCTION  
END CLASS
