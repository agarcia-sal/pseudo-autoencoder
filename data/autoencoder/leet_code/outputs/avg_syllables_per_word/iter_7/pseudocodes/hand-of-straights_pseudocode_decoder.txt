CLASS Solution  
    FUNCTION isNStraightHand(hand, groupSize)  
        IF LENGTH OF hand MODULO groupSize IS NOT EQUAL TO 0 THEN  
            RETURN False  
        END IF  

        SET count TO frequency count of elements in hand  
        SET unique_cards TO sorted list of keys in count  

        FOR each card IN unique_cards  
            IF count at card GREATER THAN 0 THEN  
                SET needed TO count at card  
                FOR i FROM 0 TO groupSize MINUS 1  
                    IF count at (card PLUS i) LESS THAN needed THEN  
                        RETURN False  
                    END IF  
                    DECREMENT count at (card PLUS i) BY needed  
                END FOR  
            END IF  
        END FOR  

        RETURN True  
    END FUNCTION  
END CLASS
