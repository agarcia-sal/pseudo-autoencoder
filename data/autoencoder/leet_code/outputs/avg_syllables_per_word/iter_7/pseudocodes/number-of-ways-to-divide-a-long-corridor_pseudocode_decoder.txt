CLASS Solution  
    FUNCTION numberOfWays(corridor)  
        SET MOD TO 10 POWER 9 PLUS 7  
        SET seats TO list of indices WHERE character IN corridor IS 'S'  
        
        IF length of seats MODULO 2 IS NOT 0 OR length of seats IS LESS THAN 2 THEN  
            RETURN 0  
        END IF  
        
        SET ways TO 1  
        FOR i FROM 2 TO length of seats EXCLUSIVE STEP 2 DO  
            SET ways TO ways MULTIPLIED BY (seats AT i MINUS seats AT (i MINUS 1)) MODULO MOD  
        END FOR  
        
        RETURN ways  
    END FUNCTION  
END CLASS
