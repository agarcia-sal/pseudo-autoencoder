CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums IS EMPTY  
            RETURN 0  
        END IF  
        
        SET dp TO empty list  
        
        FOR each num IN nums  
            SET idx TO binary search left insertion point of num IN dp  
            
            IF idx IS EQUAL TO length of dp  
                APPEND num TO dp  
            ELSE  
                SET element at idx in dp TO num  
            END IF  
        END FOR  
        
        RETURN length of dp  
    END FUNCTION  
END CLASS
