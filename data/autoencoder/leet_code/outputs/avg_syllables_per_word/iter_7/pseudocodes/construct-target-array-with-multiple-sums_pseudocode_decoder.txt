CLASS Solution  
    FUNCTION isPossible(target)  
        SET max_heap TO list of negative values of each number in target  
        TRANSFORM max_heap INTO a heap structure  
        
        SET total_sum TO sum of target  
        
        WHILE first element of max_heap IS LESS THAN negative one  
            SET largest TO negative of removing the smallest element from max_heap  
            SET rest_sum TO total_sum MINUS largest  
            
            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero  
                RETURN False  
            
            SET new_value TO largest MODULO rest_sum  
            IF new_value EQUALS zero AND rest_sum IS NOT equal to one  
                RETURN False  
            
            SET total_sum TO rest_sum PLUS new_value  
            ADD negative of new_value INTO max_heap  
        
        RETURN True  
    END FUNCTION  
END CLASS
