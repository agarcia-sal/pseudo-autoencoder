CLASS Solution  
    FUNCTION maxSlidingWindow(nums, k)  
        IF nums IS EMPTY OR k EQUALS 0  
            RETURN empty list  
        END IF  
        
        SET deq TO new double-ended queue  
        SET result TO empty list  
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 1  
            IF deq IS NOT EMPTY AND FIRST ELEMENT OF deq IS LESS THAN i MINUS k PLUS 1  
                REMOVE FIRST ELEMENT FROM deq  
            END IF  
            
            WHILE deq IS NOT EMPTY AND nums AT LAST ELEMENT OF deq IS LESS THAN nums AT i  
                REMOVE LAST ELEMENT FROM deq  
            END WHILE  
            
            APPEND i TO deq  
            
            IF i IS GREATER THAN OR EQUAL TO k MINUS 1  
                APPEND nums AT FIRST ELEMENT OF deq TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
