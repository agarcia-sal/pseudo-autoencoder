CLASS Solution  
    FUNCTION threeSum(nums)  
        SORT nums  
        SET result TO empty list  
        SET n TO LENGTH OF nums  
        
        FOR i FROM 0 TO n MINUS 3  
            IF i GREATER THAN 0 AND nums AT i EQUALS nums AT i MINUS 1 THEN  
                CONTINUE  
            SET left TO i PLUS 1  
            SET right TO n MINUS 1  
            
            WHILE left LESS THAN right  
                SET total TO nums AT i PLUS nums AT left PLUS nums AT right  
                IF total EQUALS 0 THEN  
                    APPEND list of nums AT i, nums AT left, nums AT right TO result  
                    WHILE left LESS THAN right AND nums AT left EQUALS nums AT left PLUS 1  
                        INCREMENT left BY 1  
                    WHILE left LESS THAN right AND nums AT right EQUALS nums AT right MINUS 1  
                        DECREMENT right BY 1  
                    INCREMENT left BY 1  
                    DECREMENT right BY 1  
                ELSE IF total LESS THAN 0 THEN  
                    INCREMENT left BY 1  
                ELSE  
                    DECREMENT right BY 1  
        RETURN result  
    END FUNCTION  
END CLASS
