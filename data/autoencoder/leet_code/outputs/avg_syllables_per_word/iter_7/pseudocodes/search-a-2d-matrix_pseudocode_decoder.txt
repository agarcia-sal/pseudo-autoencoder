CLASS Solution  
    FUNCTION searchMatrix(matrix, target)  
        IF matrix IS EMPTY OR first row of matrix IS EMPTY THEN  
            RETURN False  
        END IF  
        
        SET m TO LENGTH OF matrix  
        SET n TO LENGTH OF first row of matrix  
        SET left TO 0  
        SET right TO m MULTIPLIED BY n MINUS 1  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO (left PLUS right) DIVIDED BY 2  
            SET mid_value TO element in matrix at row mid DIVIDED BY n and column mid MODULO n  
            
            IF mid_value IS EQUAL TO target THEN  
                RETURN True  
            ELSE IF mid_value IS LESS THAN target THEN  
                SET left TO mid PLUS 1  
            ELSE  
                SET right TO mid MINUS 1  
            END IF  
        END WHILE  
        
        RETURN False  
    END FUNCTION  
END CLASS
