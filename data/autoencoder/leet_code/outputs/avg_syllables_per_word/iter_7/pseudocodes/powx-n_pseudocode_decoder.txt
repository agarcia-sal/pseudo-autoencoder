CLASS Solution  
    FUNCTION myPow(x, n)  
        FUNCTION qpow(a, n)  
            SET ans TO 1  
            WHILE n IS NOT zero  
                IF n MODULO 2 IS equal to 1  
                    SET ans TO ans MULTIPLIED BY a  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY 2 DISCARDING REMAINDER  
            RETURN ans  
        END FUNCTION  
  
        IF n IS GREATER THAN OR EQUAL TO zero  
            RETURN qpow(x, n)  
        ELSE  
            RETURN 1 DIVIDED BY qpow(x, NEGATION OF n)  
    END FUNCTION  
END CLASS
