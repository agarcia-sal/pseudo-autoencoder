CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO set containing 0 and firstPerson  
        SORT meetings BY their time value  
        SET i TO 0  
        WHILE i LESS THAN LENGTH OF meetings  
            SET current_time TO time value of meetings at index i  
            SET graph TO new empty mapping from person to list of connected persons  
            WHILE i LESS THAN LENGTH OF meetings AND time of meetings at index i EQUALS current_time  
                SET x, y, _ TO meeting at index i  
                APPEND y TO graph at key x  
                APPEND x TO graph at key y  
                INCREMENT i BY 1  
            END WHILE  
            SET queue TO new deque containing persons from know_secret that are keys in graph  
            SET visited TO empty set  
            WHILE queue is not empty  
                REMOVE first element from queue and SET to person  
                IF person IS IN visited THEN  
                    CONTINUE to next iteration  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN graph at person  
                    IF neighbor NOT IN visited THEN  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN list form of know_secret  
    END FUNCTION  
END CLASS
