CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        SORT arr2 IN ASCENDING ORDER  
        
        SET dp TO dictionary WITH KEY -1 AND VALUE 0  
        
        FOR each num IN arr1  
            SET new_dp TO empty dictionary  
            
            FOR each last_num, ops IN dp  
                IF num IS GREATER THAN last_num  
                    SET new_dp[num] TO minimum OF new_dp AT num OR positive infinity AND ops  
                
                SET idx TO POSITION TO INSERT last_num IN arr2 TO KEEP arr2 SORTED (RIGHT SIDE)  
                
                IF idx IS LESS THAN LENGTH OF arr2  
                    SET new_dp[arr2 AT idx] TO minimum OF new_dp AT arr2 AT idx OR positive infinity AND ops INCREMENTED BY 1  
            
            IF new_dp IS EMPTY  
                RETURN -1  
            
            SET dp TO new_dp  
        
        RETURN MINIMUM VALUE AMONG dp VALUES  
    END FUNCTION  
END CLASS
