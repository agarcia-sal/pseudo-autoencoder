CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR i FROM 0 TO LENGTH OF nums MINUS 1  
            WHILE nums AT i IS NOT EQUAL TO nums AT (nums AT i MINUS 1)  
                SWAP nums AT (nums AT i MINUS 1) WITH nums AT i  
        SET result TO EMPTY LIST  
        FOR EACH index, value IN ENUMERATE nums  
            IF value IS NOT EQUAL TO index PLUS 1  
                APPEND value TO result  
        RETURN result  
    END FUNCTION  
END CLASS
