CLASS Solution  
    FUNCTION reachNumber(target)  
        SET target TO absolute value of target  
        SET k TO 0  
        SET sum_steps TO 0  
        WHILE sum_steps LESS THAN target  
            INCREMENT k BY 1  
            INCREMENT sum_steps BY k  
        END WHILE  
        IF difference between sum_steps and target is even  
            RETURN k  
        ELSE  
            IF k is even  
                RETURN k PLUS 1  
            ELSE  
                RETURN k PLUS 2  
            END IF  
        END IF  
    END FUNCTION  
END CLASS
