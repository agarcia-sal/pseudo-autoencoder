CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator_of_t TO an iterator over t  
            FOR each character c IN s  
                IF c NOT IN iterator_of_t  
                    RETURN False  
            RETURN True  
        END FUNCTION  

        SORT strs BY descending length AND then lexicographically ascending  

        FOR each index i AND word IN strs  
            SET unique TO True  
            FOR each index j AND other IN strs  
                IF i IS NOT EQUAL TO j AND is_subsequence(word, other) IS True  
                    SET unique TO False  
                    BREAK inner loop  
            IF unique IS True  
                RETURN LENGTH OF word  
        
        RETURN -1  
    END FUNCTION  
END CLASS
