CLASS Solution  
    FUNCTION findLength(nums1, nums2)  
        SET dp TO 2D array of zeros with dimensions LENGTH OF nums1 PLUS 1 by LENGTH OF nums2 PLUS 1  
        SET max_length TO 0  
        FOR i FROM 1 TO LENGTH OF nums1  
            FOR j FROM 1 TO LENGTH OF nums2  
                IF element at index i MINUS 1 of nums1 EQUALS element at index j MINUS 1 of nums2  
                    SET dp at position i, j TO dp at position i MINUS 1, j MINUS 1 PLUS 1  
                    SET max_length TO maximum of max_length AND dp at position i, j  
        RETURN max_length  
    END FUNCTION  
END CLASS
