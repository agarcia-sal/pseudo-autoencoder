CLASS Solution  
    FUNCTION kEmptySlots(bulbs, k)  
        SET bloom_days TO empty list  
        
        FOR each day, position IN enumerate bulbs starting from 1  
            SET i TO position to insert in bloom_days preserving sorted order  
            
            IF i GREATER THAN 0 AND position MINUS bloom_days at i MINUS 1 EQUALS k  
                RETURN day  
            END IF  
            
            IF i LESS THAN LENGTH OF bloom_days AND bloom_days at i MINUS position MINUS 1 EQUALS k  
                RETURN day  
            END IF  
            
            INSERT position INTO bloom_days AT index i  
        END FOR  
        
        RETURN negative 1  
    END FUNCTION  
END CLASS
