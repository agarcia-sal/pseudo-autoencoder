CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        SORT points BY end coordinate of each balloon  
        SET arrows TO 0  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF start coordinate of balloon GREATER THAN last_arrow_position THEN  
                INCREMENT arrows BY 1  
                SET last_arrow_position TO end coordinate of balloon  
        RETURN arrows  
    END FUNCTION  
END CLASS
