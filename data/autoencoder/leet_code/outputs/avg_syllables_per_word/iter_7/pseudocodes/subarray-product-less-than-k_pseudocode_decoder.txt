CLASS Solution  
    FUNCTION numSubarrayProductLessThanK(nums, k)  
        IF k IS LESS THAN OR EQUAL TO 1 THEN  
            RETURN 0  
        END IF  
          
        SET product TO 1  
        SET left TO 0  
        SET count TO 0  
          
        FOR right FROM 0 TO LENGTH OF nums MINUS 1 DO  
            SET product TO product MULTIPLIED BY nums AT right  
              
            WHILE product IS GREATER THAN OR EQUAL TO k AND left IS LESS THAN OR EQUAL TO right DO  
                SET product TO product DIVIDED BY nums AT left  
                INCREMENT left BY 1  
            END WHILE  
              
            INCREMENT count BY right MINUS left PLUS 1  
        END FOR  
          
        RETURN count  
    END FUNCTION  
END CLASS
