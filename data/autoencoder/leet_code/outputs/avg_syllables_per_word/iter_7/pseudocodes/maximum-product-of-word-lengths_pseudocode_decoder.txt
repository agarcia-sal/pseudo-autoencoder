CLASS Solution  
    FUNCTION maxProduct(words)  
        SET word_sets TO empty list  
        FOR each word IN words  
            APPEND set of characters in word TO word_sets  
        SET max_product TO 0  
        FOR i FROM 0 TO LENGTH OF words MINUS 1  
            FOR j FROM i PLUS 1 TO LENGTH OF words MINUS 1  
                IF word_sets at i AND word_sets at j have no common elements THEN  
                    SET product TO LENGTH OF words at i MULTIPLIED BY LENGTH OF words at j  
                    IF product GREATER THAN max_product THEN  
                        SET max_product TO product  
                END IF  
            END FOR  
        END FOR  
        RETURN max_product  
    END FUNCTION  
END CLASS
