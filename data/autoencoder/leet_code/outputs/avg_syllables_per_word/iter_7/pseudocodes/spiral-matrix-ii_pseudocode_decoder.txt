CLASS Solution  
    FUNCTION generateMatrix(n)  
        SET matrix TO n by n grid filled with zeros  
        SET top TO 0  
        SET bottom TO n minus 1  
        SET left TO 0  
        SET right TO n minus 1  
        SET num TO 1  
        
        WHILE top less than or equal to bottom AND left less than or equal to right  
            FOR i FROM left TO right  
                SET matrix at row top and column i TO num  
                INCREMENT num BY 1  
            INCREMENT top BY 1  
            
            FOR i FROM top TO bottom  
                SET matrix at row i and column right TO num  
                INCREMENT num BY 1  
            DECREMENT right BY 1  
            
            IF top less than or equal to bottom  
                FOR i FROM right DOWN TO left  
                    SET matrix at row bottom and column i TO num  
                    INCREMENT num BY 1  
                DECREMENT bottom BY 1  
            
            IF left less than or equal to right  
                FOR i FROM bottom DOWN TO top  
                    SET matrix at row i and column left TO num  
                    INCREMENT num BY 1  
                INCREMENT left BY 1  
        
        RETURN matrix  
    END FUNCTION  
END CLASS
