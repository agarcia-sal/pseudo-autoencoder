CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 IS None  
                RETURN 0  
            SET x1, y1 TO DIVIDE ASCII VALUE OF c1 MINUS ASCII VALUE OF 'A' BY 6 WITH REMAINDER  
            SET x2, y2 TO DIVIDE ASCII VALUE OF c2 MINUS ASCII VALUE OF 'A' BY 6 WITH REMAINDER  
            RETURN ABSOLUTE VALUE OF x1 MINUS x2 PLUS ABSOLUTE VALUE OF y1 MINUS y2  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2) WITH CACHE  
            IF i IS EQUAL TO LENGTH OF word  
                RETURN 0  
            SET c TO CHARACTER AT POSITION i IN word  
            SET d1 TO distance BETWEEN f1 AND c PLUS dp WITH PARAMETERS i PLUS 1, c, f2  
            SET d2 TO distance BETWEEN f2 AND c PLUS dp WITH PARAMETERS i PLUS 1, f1, c  
            RETURN MINIMUM OF d1 AND d2  
        END FUNCTION  
        
        RETURN dp WITH PARAMETERS 0, None, None  
    END FUNCTION  
END CLASS
