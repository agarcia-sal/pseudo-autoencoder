CLASS Solution  
    FUNCTION numTrees(n)  
        SET dp TO list of zeros with length n plus 1  
        SET dp at index 0 TO 1  
        SET dp at index 1 TO 1  
        FOR i FROM 2 TO n inclusive  
            FOR j FROM 1 TO i inclusive  
                INCREMENT dp at index i BY dp at index j minus 1 MULTIPLIED BY dp at index i minus j  
        RETURN dp at index n  
    END FUNCTION  
END CLASS
