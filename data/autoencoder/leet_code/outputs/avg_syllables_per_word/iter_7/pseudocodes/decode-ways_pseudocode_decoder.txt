CLASS Solution  
    FUNCTION numDecodings(s)  
        IF s IS EMPTY OR FIRST CHARACTER OF s IS EQUAL TO '0'  
            RETURN 0  
        END IF  
        
        SET dp TO A LIST OF ZEROS WITH LENGTH LENGTH OF s PLUS 1  
        SET dp AT INDEX 0 TO 1  
        SET dp AT INDEX 1 TO 1  
        
        FOR i FROM 2 TO LENGTH OF s PLUS 1 INCLUSIVE  
            IF CHARACTER OF s AT POSITION i MINUS 1 IS NOT '0'  
                INCREMENT dp AT INDEX i BY dp AT INDEX i MINUS 1  
            END IF  
            
            SET two_digit TO INTEGER VALUE OF SUBSTRING OF s FROM i MINUS 2 TO i  
            IF two_digit IS GREATER THAN OR EQUAL TO 10 AND two_digit IS LESS THAN OR EQUAL TO 26  
                INCREMENT dp AT INDEX i BY dp AT INDEX i MINUS 2  
            END IF  
        END FOR  
        
        RETURN VALUE OF dp AT LAST INDEX  
    END FUNCTION  
END CLASS
