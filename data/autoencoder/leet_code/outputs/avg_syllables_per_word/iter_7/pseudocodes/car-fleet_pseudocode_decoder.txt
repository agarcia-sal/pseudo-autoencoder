CLASS Solution  
    FUNCTION carFleet(target, position, speed)  
        SET time_to_target TO list of times each car takes to reach target, ordered from furthest to closest by position and corresponding speed  
        
        SET fleets TO 0  
        SET current_max_time TO 0  
        
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time THEN  
                INCREMENT fleets BY 1  
                SET current_max_time TO time  
            END IF  
        END FOR  
        
        RETURN fleets  
    END FUNCTION  
END CLASS
