CLASS Solution  
    FUNCTION findMaximizedCapital(k, w, profits, capital)  
        SET projects TO list of pairs combining capital and profits  
        SORT projects by capital in ascending order  
        
        SET max_heap TO empty list  
        SET i TO 0  
        SET n TO length of projects  
        
        FOR each number from 1 TO k DO  
            WHILE i LESS THAN n AND projects at i capital LESS THAN OR EQUAL TO w DO  
                ADD negative profit of projects at i TO max_heap  
                INCREMENT i BY 1  
            
            IF max_heap IS empty THEN  
                EXIT loop  
            
            REMOVE the largest profit from max_heap and ADD its absolute value TO w  
        
        RETURN w  
    END FUNCTION  
END CLASS
