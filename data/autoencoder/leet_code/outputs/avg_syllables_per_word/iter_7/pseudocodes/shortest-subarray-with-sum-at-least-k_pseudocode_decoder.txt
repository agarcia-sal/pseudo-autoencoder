CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO list containing single element 0  
        FOR each num IN nums  
            APPEND last element of prefix_sum plus num TO prefix_sum  
        
        CREATE empty deque dq  
        SET min_length TO infinity  
        
        FOR each index i and value current_sum IN prefix_sum with indices  
            WHILE dq is not empty AND current_sum minus prefix_sum at front index of dq is greater than or equal to k  
                SET min_length TO minimum of min_length and i minus front index of dq  
                REMOVE front index from dq  
            
            WHILE dq is not empty AND current_sum is less than or equal to prefix_sum at back index of dq  
                REMOVE back index from dq  
            
            ADD index i TO back of dq  
        
        IF min_length is equal to infinity  
            RETURN negative one  
        ELSE  
            RETURN min_length  
    END FUNCTION  
END CLASS
