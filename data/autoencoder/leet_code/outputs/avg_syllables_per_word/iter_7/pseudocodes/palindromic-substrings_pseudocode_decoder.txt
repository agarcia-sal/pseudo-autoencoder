CLASS Solution  
    FUNCTION countSubstrings(s)  
        FUNCTION expand_around_center(left, right)  
            SET count TO 0  
            WHILE left GREATER THAN OR EQUAL TO 0 AND right LESS THAN LENGTH OF s AND CHARACTER AT s[left] EQUALS CHARACTER AT s[right]  
                INCREMENT count BY 1  
                DECREMENT left BY 1  
                INCREMENT right BY 1  
            END WHILE  
            RETURN count  
        END FUNCTION  
  
        SET total_palindromes TO 0  
        FOR i FROM 0 TO LENGTH OF s MINUS 1  
            SET total_palindromes TO total_palindromes PLUS expand_around_center(i, i)  
            SET total_palindromes TO total_palindromes PLUS expand_around_center(i, i PLUS 1)  
        END FOR  
  
        RETURN total_palindromes  
    END FUNCTION  
END CLASS
