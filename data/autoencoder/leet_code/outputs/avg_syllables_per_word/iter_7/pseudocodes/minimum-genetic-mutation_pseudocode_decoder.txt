CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO SET of bank  
        INITIALIZE queue WITH (start, 0)  
        SET mp TO mapping FROM 'A' TO 'TCG', 'T' TO 'ACG', 'C' TO 'ATG', 'G' TO 'ATC'  
        WHILE queue IS NOT empty  
            SET t, step TO REMOVE first element FROM queue  
            IF t EQUALS end  
                RETURN step  
            FOR EACH index i, character v IN t  
                FOR EACH character j IN mp[v]  
                    SET next TO concatenation OF substring of t BEFORE i, j, substring of t AFTER i  
                    IF next IS IN s  
                        ADD (next, step plus 1) TO queue  
                        REMOVE next FROM s  
        RETURN negative 1  
    END FUNCTION  
END CLASS
