CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO 10 raised to the power 9 plus 7  
        SET n TO LENGTH OF s  
        SET last TO list of size 26 with all values set to minus 1  
        SET last2 TO list of size 26 with all values set to minus 1  
        SET result TO 0  
        
        FOR each index i and character char IN s  
            SET idx TO numeric position of char minus numeric position of letter 'A'  
            SET result TO result plus (i minus last at idx) multiplied BY (last at idx minus last2 at idx) MODULO MOD  
            SET last2 at idx TO last at idx  
            SET last at idx TO i  
        
        FOR each integer i FROM 0 TO 25  
            SET result TO result plus (n minus last at i) multiplied BY (last at i minus last2 at i) MODULO MOD  
        
        RETURN result  
    END FUNCTION  
END CLASS
