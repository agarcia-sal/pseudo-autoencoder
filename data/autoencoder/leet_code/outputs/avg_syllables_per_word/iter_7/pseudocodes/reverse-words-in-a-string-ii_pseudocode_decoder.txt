CLASS Solution  
    FUNCTION reverseWords(s)  
        REVERSE the entire list s  
        SET start TO 0  
        FOR each end FROM 0 TO LENGTH OF s MINUS 1  
            IF s at position end IS EQUAL TO space  
                REVERSE the sublist of s FROM start TO end MINUS 1  
                SET start TO end PLUS 1  
        REVERSE the sublist of s FROM start TO END  
    END FUNCTION  
END CLASS
