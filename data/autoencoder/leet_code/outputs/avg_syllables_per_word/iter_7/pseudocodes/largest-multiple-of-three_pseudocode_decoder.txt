CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO frequency count of digits  
        SET total_sum TO sum of digits  
        SET remainder TO total_sum MODULO 3  
        
        IF remainder EQUALS 1 THEN  
            FOR each d IN [1, 4, 7] DO  
                IF count of d GREATER THAN 0 THEN  
                    DECREMENT count of d BY 1  
                    BREAK loop  
            ELSE  
                FOR each d IN [2, 5, 8] DO  
                    IF count of d GREATER THAN 1 THEN  
                        DECREMENT count of d BY 2  
                        BREAK loop  
        
        IF remainder EQUALS 2 THEN  
            FOR each d IN [2, 5, 8] DO  
                IF count of d GREATER THAN 0 THEN  
                    DECREMENT count of d BY 1  
                    BREAK loop  
            ELSE  
                FOR each d IN [1, 4, 7] DO  
                    IF count of d GREATER THAN 1 THEN  
                        DECREMENT count of d BY 2  
                        BREAK loop  
        
        SET result TO empty list  
        FOR d FROM 9 DOWN TO 0 DO  
            APPEND string representation of d REPEATED count of d times TO result  
        
        SET final_number TO concatenation of elements in result  
        
        IF final_number IS empty OR final_number STARTS WITH '0' THEN  
            RETURN '0'  
        ELSE  
            RETURN final_number  
    END FUNCTION  
END CLASS
