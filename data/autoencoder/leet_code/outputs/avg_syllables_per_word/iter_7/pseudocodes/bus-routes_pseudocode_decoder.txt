CLASS Solution  
    FUNCTION numBusesToDestination(routes, source, target)  
        IF source EQUALS target THEN  
            RETURN 0  
        END IF  

        SET stop_to_buses TO a default dictionary of lists  
        FOR each bus WITH index bus_number IN routes  
            FOR each stop IN bus route  
                APPEND bus_number TO stop_to_buses AT stop  
            END FOR  
        END FOR  

        SET bus_queue TO an empty queue  
        SET visited_buses TO an empty set  
        SET visited_stops TO an empty set  

        FOR each bus IN stop_to_buses AT source  
            APPEND pair (bus, 1) TO bus_queue  
            ADD bus TO visited_buses  
            ADD source TO visited_stops  
        END FOR  

        WHILE bus_queue is not empty  
            REMOVE first element FROM bus_queue INTO current_bus AND bus_count  

            FOR each stop IN routes AT current_bus  
                IF stop EQUALS target THEN  
                    RETURN bus_count  
                END IF  

                IF stop NOT IN visited_stops THEN  
                    ADD stop TO visited_stops  
                    FOR each next_bus IN stop_to_buses AT stop  
                        IF next_bus NOT IN visited_buses THEN  
                            ADD next_bus TO visited_buses  
                            APPEND pair (next_bus, bus_count PLUS 1) TO bus_queue  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END WHILE  

        RETURN -1  
    END FUNCTION  
END CLASS
