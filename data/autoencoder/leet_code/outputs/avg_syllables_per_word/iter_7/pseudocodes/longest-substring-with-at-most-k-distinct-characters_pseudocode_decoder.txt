CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s, k)  
        IF k EQUALS 0 THEN  
            RETURN 0  
        END IF  

        SET char_map TO empty dictionary  
        SET left TO 0  
        SET max_length TO 0  

        FOR right FROM 0 TO LENGTH OF s MINUS 1  
            SET character AT index right OF s IN char_map TO right  

            IF SIZE OF char_map GREATER THAN k THEN  
                SET leftmost_char TO key IN char_map WITH MINIMUM associated value  
                SET left TO char_map REMOVE leftmost_char PLUS 1  
            END IF  

            SET max_length TO MAXIMUM OF max_length AND (right MINUS left PLUS 1)  
        END FOR  

        RETURN max_length  
    END FUNCTION  
END CLASS
