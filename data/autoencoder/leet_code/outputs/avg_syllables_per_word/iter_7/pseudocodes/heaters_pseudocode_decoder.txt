CLASS Solution  
    FUNCTION findRadius(houses, heaters)  
        SORT houses  
        SORT heaters  
        
        SET min_radius TO 0  
        SET j TO 0  
        
        FOR each house IN houses  
            WHILE j LESS THAN LENGTH OF heaters MINUS 1 AND  
                  DISTANCE BETWEEN heaters AT j AND house GREATER THAN OR EQUAL TO  
                  DISTANCE BETWEEN heaters AT j PLUS 1 AND house  
                INCREMENT j BY 1  
            END WHILE  
            
            SET min_radius TO GREATER OF min_radius AND DISTANCE BETWEEN heaters AT j AND house  
        END FOR  
        
        RETURN min_radius  
    END FUNCTION  
END CLASS
