CLASS Solution  
    FUNCTION minBuildTime(blocks, split)  
        TRANSFORM blocks INTO a min-heap  
        
        WHILE LENGTH OF blocks GREATER THAN 1  
            SET first TO REMOVE the smallest element FROM blocks  
            SET second TO REMOVE the next smallest element FROM blocks  
            
            SET combined_time TO split PLUS second  
            
            ADD combined_time INTO blocks heap  
        
        RETURN the only element REMAINING IN blocks  
    END FUNCTION  
END CLASS
