CLASS Solution  
    FUNCTION getSum(a, b)  
        SET MAX TO largest 32-bit positive integer  
        SET mask TO 32-bit mask value  
        
        WHILE b is not equal to 0  
            SET carry TO bitwise AND of a and b shifted left by 1  
            SET a TO bitwise XOR of a and b masked by mask  
            SET b TO carry masked by mask  
        
        IF a is less than or equal to MAX  
            RETURN a  
        ELSE  
            RETURN bitwise complement of bitwise XOR of a and mask  
    END FUNCTION  
END CLASS
