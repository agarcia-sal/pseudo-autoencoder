CLASS Solution  
    FUNCTION exist(board, word)  
        IF board IS empty THEN  
            RETURN False  
        SET m TO LENGTH OF board  
        SET n TO LENGTH OF board AT 0  
        SET visited TO 2D list of False values with dimensions m BY n  

        FUNCTION dfs(x, y, index)  
            IF index EQUALS LENGTH OF word THEN  
                RETURN True  
            IF x IS LESS THAN 0 OR x IS GREATER THAN OR EQUAL TO m OR y IS LESS THAN 0 OR y IS GREATER THAN OR EQUAL TO n OR visited AT x y IS True OR board AT x y IS NOT EQUAL TO word AT index THEN  
                RETURN False  
            SET visited AT x y TO True  
            FOR each dx, dy IN list of pairs: (-1, 0), (1, 0), (0, -1), (0, 1)  
                IF dfs(x PLUS dx, y PLUS dy, index PLUS 1) IS True THEN  
                    RETURN True  
            SET visited AT x y TO False  
            RETURN False  
        END FUNCTION  

        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF dfs(i, j, 0) IS True THEN  
                    RETURN True  
        RETURN False  
    END FUNCTION  
END CLASS
