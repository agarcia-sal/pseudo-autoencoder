CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO sorted list of pairs (interval start, interval index) from intervals  
        SET result TO empty list  
        FOR each interval WITH end time in intervals  
            SET idx TO position found by binary searching for interval end in start_times  
            IF idx IS LESS THAN length of start_times THEN  
                APPEND index at start_times position idx TO result  
            ELSE  
                APPEND negative one TO result  
        RETURN result  
    END FUNCTION  
END CLASS
