CLASS Solution  
    FUNCTION customSortString(order, s)  
        SET d TO empty mapping  
        FOR each index i AND character c IN order  
            SET d at key c TO i  
        SET sorted_characters TO s sorted by key where each character x maps to d at x IF PRESENT OTHERWISE 0  
        SET result TO concatenation of sorted_characters  
        RETURN result  
    END FUNCTION  
END CLASS
