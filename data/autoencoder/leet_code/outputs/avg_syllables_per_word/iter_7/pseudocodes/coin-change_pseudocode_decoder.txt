CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        SET dp TO list of length amount plus 1 with each element SET TO amount plus 1  
        SET dp at index 0 TO 0  
        
        FOR a FROM 1 TO amount  
            FOR each coin IN coins  
                IF a MINUS coin IS GREATER THAN OR EQUAL TO 0  
                    SET dp at index a TO minimum of dp at index a AND dp at index a MINUS coin PLUS 1  
        
        IF dp at index amount IS EQUAL TO amount PLUS 1  
            RETURN negative 1  
        ELSE  
            RETURN dp at index amount  
    END FUNCTION  
END CLASS
