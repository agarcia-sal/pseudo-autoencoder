CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO TreeNode(values AT POSITION 0)  
    SET i TO 1  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i IS LESS THAN LENGTH OF values AND values AT POSITION i IS NOT None  
            SET node.left TO TreeNode(values AT POSITION i)  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values AT POSITION i IS NOT None  
            SET node.right TO TreeNode(values AT POSITION i)  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low SET TO NEGATIVE_INFINITY, high SET TO POSITIVE_INFINITY)  
            IF node IS None  
                RETURN True  
            IF node.val IS LESS THAN OR EQUAL TO low OR node.val IS GREATER THAN OR EQUAL TO high  
                RETURN False  
            RETURN validate(node.left, low, node.val) AND validate(node.right, node.val, high)  
        END FUNCTION  
        RETURN validate(root)  
    END FUNCTION  
END CLASS
