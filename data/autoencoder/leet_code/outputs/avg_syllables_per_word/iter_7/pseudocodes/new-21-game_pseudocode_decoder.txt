CLASS Solution  
    FUNCTION new21Game(n, k, maxPts)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k THEN  
                RETURN 1 IF i LESS THAN OR EQUAL TO n OTHERWISE 0  
            END IF  
            IF i EQUAL TO k MINUS 1 THEN  
                SET numerator TO n MINUS k PLUS 1  
                IF numerator GREATER THAN maxPts THEN  
                    SET numerator TO maxPts  
                END IF  
                RETURN numerator DIVIDED BY maxPts  
            END IF  
            SET first_term TO dfs(i PLUS 1)  
            SET second_term TO dfs(i PLUS 1) MINUS dfs(i PLUS maxPts PLUS 1)  
            RETURN first_term PLUS second_term DIVIDED BY maxPts  
        END FUNCTION  
        RETURN dfs(0)  
    END FUNCTION  
END CLASS
