CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO LENGTH OF arr  
        SET min_heap TO EMPTY LIST  
          
        FOR i FROM 0 TO n MINUS 2  
            ADD FRACTION REPRESENTED BY arr[i] DIVIDED BY arr[n MINUS 1] WITH INDICES i AND n MINUS 1 TO min_heap  
        END FOR  
          
        FOR counter FROM 1 TO k MINUS 1  
            REMOVE SMALLEST FRACTION AND ITS INDICES i AND j FROM min_heap  
            IF j MINUS 1 IS GREATER THAN i THEN  
                ADD FRACTION REPRESENTED BY arr[i] DIVIDED BY arr[j MINUS 1] WITH INDICES i AND j MINUS 1 TO min_heap  
            END IF  
        END FOR  
          
        REMOVE SMALLEST FRACTION AND ITS INDICES i AND j FROM min_heap  
        RETURN LIST CONTAINING arr[i] AND arr[j]  
    END FUNCTION  
END CLASS
