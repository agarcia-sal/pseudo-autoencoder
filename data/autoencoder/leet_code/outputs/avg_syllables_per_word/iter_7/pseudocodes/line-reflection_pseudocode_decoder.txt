CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO set of all points converted to tuples from points  
        SET min_x TO minimum x-coordinate value among unique_points  
        SET max_x TO maximum x-coordinate value among unique_points  
        SET line_of_reflection TO min_x plus max_x DIVIDED BY 2  
        FOR each (x, y) IN unique_points  
            SET reflected_x TO two times line_of_reflection minus x  
            IF (reflected_x, y) NOT IN unique_points THEN  
                RETURN False  
        RETURN True  
    END FUNCTION  
END CLASS
