CLASS Solution
    FUNCTION kSimilarity(s1, s2)
        FUNCTION neighbors(s)
            SET i TO 0
            WHILE i LESS THAN LENGTH OF s AND CHARACTER AT s WITH INDEX i EQUALS CHARACTER AT s2 WITH INDEX i
                INCREMENT i BY 1
            FOR j FROM i PLUS 1 TO LENGTH OF s MINUS 1
                IF CHARACTER AT s WITH INDEX j EQUALS CHARACTER AT s2 WITH INDEX i AND CHARACTER AT s WITH INDEX j NOT EQUAL TO CHARACTER AT s2 WITH INDEX j
                    YIELD CONCATENATION OF SUBSTRING OF s FROM START TO i, CHARACTER AT s WITH INDEX j, SUBSTRING OF s FROM i PLUS 1 TO j MINUS 1, CHARACTER AT s WITH INDEX i, SUBSTRING OF s FROM j PLUS 1 TO END
        END FUNCTION

        SET queue TO A NEW QUEUE WITH ELEMENT (s1, 0)
        SET visited TO A NEW SET CONTAINING s1

        WHILE queue IS NOT EMPTY
            REMOVE FIRST ELEMENT FROM queue AND UNPACK INTO s AND swaps
            IF s EQUALS s2
                RETURN swaps
            FOR EACH neighbor IN neighbors(s)
                IF neighbor NOT IN visited
                    ADD neighbor TO visited
                    ADD (neighbor, swaps PLUS 1) TO queue
        END WHILE
    END FUNCTION
END CLASS
