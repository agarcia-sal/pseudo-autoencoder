CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO map with key 0 and value -1  
        SET prefix_xor TO 0  
        SET max_length TO 0  
        
        FOR each index i AND character char IN s  
            SET digit TO integer value of char  
            TOGGLE the bit in prefix_xor corresponding to digit  
            
            IF prefix_xor EXISTS IN prefix_map THEN  
                SET max_length TO greater of max_length and difference between i and prefix_map value at prefix_xor  
            END IF  
            
            FOR each number j FROM 0 TO 9  
                SET flipped TO prefix_xor with the j-th bit toggled  
                IF flipped EXISTS IN prefix_map THEN  
                    SET max_length TO greater of max_length and difference between i and prefix_map value at flipped  
                END IF  
            END FOR  
            
            IF prefix_xor NOT EXISTS IN prefix_map THEN  
                SET prefix_map at prefix_xor TO i  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
