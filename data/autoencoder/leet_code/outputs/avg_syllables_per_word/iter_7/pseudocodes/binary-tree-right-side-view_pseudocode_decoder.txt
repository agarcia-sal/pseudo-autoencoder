CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO TreeNode WITH values at position zero  
    SET i TO 1  
    CREATE empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element FROM queue  
        IF i IS LESS THAN LENGTH OF values AND values at position i IS NOT None  
            SET node.left TO TreeNode WITH values at position i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values at position i IS NOT None  
            SET node.right TO TreeNode WITH values at position i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree WITH p.left AND q.left AND is_same_tree WITH p.right AND q.right  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView(root)  
        IF root IS None  
            RETURN empty list  
        SET right_side_view TO empty list  
        CREATE queue WITH root as initial element  
        WHILE queue IS NOT empty  
            SET level_length TO LENGTH OF queue  
            FOR i FROM 0 TO level_length MINUS 1  
                SET node TO REMOVE first element FROM queue  
                IF i IS EQUAL TO level_length MINUS 1  
                    APPEND node.val TO right_side_view  
                IF node.left IS NOT None  
                    APPEND node.left TO queue  
                IF node.right IS NOT None  
                    APPEND node.right TO queue  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
