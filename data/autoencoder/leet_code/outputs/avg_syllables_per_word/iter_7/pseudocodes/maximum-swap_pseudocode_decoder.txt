CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO list of characters representing num  
        SET last TO empty dictionary  
        FOR each index i AND character d IN digits  
            SET last AT integer value of d TO i  
        END FOR  
        
        FOR each index i AND character d IN digits  
            FOR each digit d2 FROM 9 DOWNTO integer value of d PLUS 1  
                IF last contains key d2 AND last AT d2 GREATER THAN i THEN  
                    SWAP digits AT i WITH digits AT last AT d2  
                    SET result TO concatenation of digits converted to integer  
                    RETURN result  
                END IF  
            END FOR  
        END FOR  
        
        RETURN num  
    END FUNCTION  
END CLASS
