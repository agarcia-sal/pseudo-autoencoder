CLASS Solution  
    FUNCTION maxJumps(arr, d)  
        FUNCTION dp(i)  
            SET max_jumps TO 1  
            FOR x FROM 1 TO d  
                IF i PLUS x IS LESS THAN LENGTH OF arr THEN  
                    IF arr AT POSITION i PLUS x IS GREATER THAN OR EQUAL TO arr AT POSITION i THEN  
                        BREAK LOOP  
                    END IF  
                    SET max_jumps TO MAXIMUM OF max_jumps AND 1 PLUS dp(i PLUS x)  
                END IF  
            END FOR  
            FOR x FROM 1 TO d  
                IF i MINUS x IS GREATER THAN OR EQUAL TO 0 THEN  
                    IF arr AT POSITION i MINUS x IS GREATER THAN OR EQUAL TO arr AT POSITION i THEN  
                        BREAK LOOP  
                    END IF  
                    SET max_jumps TO MAXIMUM OF max_jumps AND 1 PLUS dp(i MINUS x)  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        RETURN MAXIMUM OF dp(i) FOR EACH i FROM 0 TO LENGTH OF arr MINUS 1  
    END FUNCTION  
END CLASS
