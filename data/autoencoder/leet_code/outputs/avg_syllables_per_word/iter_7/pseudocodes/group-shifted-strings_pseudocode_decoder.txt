CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s IS EMPTY  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET difference TO character position of c MINUS character position of first character in s  
                SET difference TO difference MODULO 26  
                APPEND string representation of difference TO canonical_form  
            END FOR  
            RETURN elements of canonical_form joined by comma  
        END FUNCTION  
        
        SET groups TO empty dictionary with default list  
        
        FOR each string s IN strings  
            SET canonical_form TO get_canonical_form(s)  
            APPEND s TO groups at key canonical_form  
        END FOR  
        
        RETURN list of values from groups  
    END FUNCTION  
END CLASS
