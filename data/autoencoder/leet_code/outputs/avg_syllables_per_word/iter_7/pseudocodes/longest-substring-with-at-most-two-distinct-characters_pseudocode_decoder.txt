CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        INITIALIZE char_count AS a mapping with default integer values  
        SET left TO 0  
        SET max_length TO 0  
  
        FOR right FROM 0 TO LENGTH OF s MINUS 1  
            INCREMENT count of character s AT position right IN char_count BY 1  
  
            WHILE NUMBER OF KEYS IN char_count IS GREATER THAN 2  
                DECREMENT count of character s AT position left IN char_count BY 1  
                IF count of character s AT position left IS EQUAL TO 0  
                    REMOVE character s AT position left FROM char_count  
                INCREMENT left BY 1  
  
            SET max_length TO THE GREATER OF max_length AND right MINUS left PLUS 1  
  
        RETURN max_length  
    END FUNCTION  
END CLASS
