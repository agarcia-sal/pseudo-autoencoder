CLASS Solution  
    FUNCTION asteroidCollision(asteroids)  
        SET stack TO empty list  
        
        FOR each asteroid IN asteroids  
            WHILE stack is not empty AND asteroid is less than 0 AND top element of stack is greater than 0  
                IF top element of stack is less than negative of asteroid  
                    REMOVE top element from stack  
                    CONTINUE to next iteration of WHILE  
                ELSE IF top element of stack is equal to negative of asteroid  
                    REMOVE top element from stack  
                BREAK the WHILE loop  
            ELSE  
                APPEND asteroid TO stack  
        
        RETURN stack  
    END FUNCTION  
END CLASS
