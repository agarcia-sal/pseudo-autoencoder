CLASS Solution  
    FUNCTION findMin(nums)  
        SET left TO 0  
        SET right TO LENGTH OF nums MINUS 1  
        
        WHILE left LESS THAN right  
            SET mid TO (left PLUS right) DIVIDED BY 2  
            
            IF element at index mid in nums GREATER THAN element at index right in nums  
                SET left TO mid PLUS 1  
            ELSE IF element at index mid in nums LESS THAN element at index right in nums  
                SET right TO mid  
            ELSE  
                DECREMENT right BY 1  
        
        RETURN element at index left in nums  
    END FUNCTION  
END CLASS
