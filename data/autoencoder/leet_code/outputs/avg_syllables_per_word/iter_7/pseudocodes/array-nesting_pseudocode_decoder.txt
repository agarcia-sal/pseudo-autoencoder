CLASS Solution  
    FUNCTION arrayNesting(nums)  
        SET visited TO list of False values with length equal to length of nums  
        SET max_length TO 0  
  
        FOR each index i FROM 0 TO length of nums MINUS 1  
            IF visited at position i IS False THEN  
                SET current_length TO 0  
                SET k TO i  
                WHILE visited at position k IS False DO  
                    SET visited at position k TO True  
                    SET k TO value at nums position k  
                    INCREMENT current_length BY 1  
                END WHILE  
                IF current_length GREATER THAN max_length THEN  
                    SET max_length TO current_length  
                END IF  
            END IF  
        END FOR  
  
        RETURN max_length  
    END FUNCTION  
END CLASS
