CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        SET graph TO empty mapping with default empty list  
        SET indegree TO list of zeros with length numCourses  
        
        FOR each pair course, prereq IN prerequisites  
            APPEND course TO graph at key prereq  
            INCREMENT indegree at index course BY 1  
        
        SET queue TO list of courses from 0 TO numCourses MINUS 1 WHERE indegree at course EQUALS 0  
        
        WHILE queue IS NOT empty  
            SET current_course TO first element removed from queue  
            DECREMENT numCourses BY 1  
            
            FOR each next_course IN graph at current_course  
                DECREMENT indegree at index next_course BY 1  
                IF indegree at index next_course EQUALS 0  
                    APPEND next_course TO queue  
        
        RETURN True IF numCourses EQUALS 0 ELSE False  
    END FUNCTION  
END CLASS
