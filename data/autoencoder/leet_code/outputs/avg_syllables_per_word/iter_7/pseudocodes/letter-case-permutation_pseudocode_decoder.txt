CLASS Solution  
    FUNCTION letterCasePermutation(s)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO LENGTH OF t THEN  
                APPEND concatenation OF t TO ans  
                RETURN  
            END IF  
            CALL dfs(i PLUS 1)  
            IF t AT i IS ALPHABETIC THEN  
                SET t AT i TO CASE TOGGLED CHARACTER OF t AT i  
                CALL dfs(i PLUS 1)  
            END IF  
        END FUNCTION  

        SET t TO LIST OF CHARACTERS IN s  
        SET ans TO EMPTY LIST  
        CALL dfs(0)  
        RETURN ans  
    END FUNCTION  
END CLASS
