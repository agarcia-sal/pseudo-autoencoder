CLASS Solution  
    FUNCTION canJump(nums)  
        SET farthest TO 0  
        FOR each index i FROM 0 TO LENGTH OF nums MINUS 1  
            IF i GREATER THAN farthest  
                RETURN False  
            END IF  
            SET farthest TO the greater value BETWEEN farthest AND i PLUS nums AT i  
        END FOR  
        RETURN farthest GREATER THAN OR EQUAL TO LENGTH OF nums MINUS 1  
    END FUNCTION  
END CLASS
