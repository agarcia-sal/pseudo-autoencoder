CLASS Solution  
    FUNCTION merge(intervals)  
        SORT intervals BY first element of each interval  
        SET merged TO empty list  
        FOR each interval IN intervals  
            IF merged IS empty OR last element in merged's second value IS LESS THAN interval's first value  
                APPEND interval TO merged  
            ELSE  
                SET last element in merged's second value TO maximum of last element in merged's second value AND interval's second value  
        RETURN merged  
    END FUNCTION  
END CLASS
