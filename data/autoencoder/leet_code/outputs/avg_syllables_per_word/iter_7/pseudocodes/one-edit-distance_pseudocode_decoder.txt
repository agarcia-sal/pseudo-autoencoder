CLASS Solution
    FUNCTION isOneEditDistance(s, t)
        IF length of s GREATER THAN length of t
            RETURN result of isOneEditDistance called with t and s
        END IF

        SET diff TO length of t MINUS length of s

        IF diff GREATER THAN 1
            RETURN False
        END IF

        IF diff EQUALS 0
            RETURN count of positions where characters in s and t differ EQUALS 1
        ELSE
            FOR index FROM 0 TO length of s MINUS 1
                IF character at index in s NOT EQUAL TO character at index in t
                    RETURN substring of s from index TO end EQUALS substring of t from index plus 1 TO end
                END IF
            END FOR
            RETURN True
        END IF
    END FUNCTION
END CLASS
