CLASS Solution  
    FUNCTION findLongestWord(s, dictionary)  
        FUNCTION is_subsequence(word, s)  
            SET it TO iterator over s  
            RETURN whether every character in word appears sequentially in it  
        END FUNCTION  
        
        SORT dictionary by descending length and ascending lexicographical order  
        
        FOR each word IN dictionary  
            IF is_subsequence(word, s) IS true THEN  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN empty string  
    END FUNCTION  
END CLASS
