CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO set of numbers where each number x from fronts and corresponding y from backs satisfy x EQUALS y  
        SET all_numbers TO set containing all elements from fronts and backs  
        SET good_numbers TO all_numbers WITHOUT elements in same  
        IF good_numbers is not empty THEN  
            RETURN minimum element in good_numbers  
        ELSE  
            RETURN 0  
        END IF  
    END FUNCTION  
END CLASS
