CLASS Solution  
    FUNCTION earliestFullBloom(plantTime, growTime)  
        SET tasks TO list of pairs combining plantTime and growTime  
        SORT tasks by growTime in descending order  
        
        SET current_day TO 0  
        SET earliest_bloom TO 0  
        
        FOR each plant, grow IN tasks  
            INCREMENT current_day BY plant  
            SET bloom_day TO current_day PLUS grow  
            IF bloom_day GREATER THAN earliest_bloom  
                SET earliest_bloom TO bloom_day  
        RETURN earliest_bloom  
    END FUNCTION  
END CLASS
