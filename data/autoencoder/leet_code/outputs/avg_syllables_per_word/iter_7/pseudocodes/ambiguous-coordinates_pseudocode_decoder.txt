CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO LENGTH OF frag  
            FOR d FROM 1 TO n INCLUSIVE  
                SET left TO SUBSTRING OF frag FROM START TO d  
                SET right TO SUBSTRING OF frag FROM d TO END  
                IF (left DOES NOT START WITH '0' OR left EQUALS '0') AND right DOES NOT END WITH '0'  
                    YIELD left CONCATENATED WITH ('.' IF d IS NOT EQUAL TO n ELSE EMPTY STRING) CONCATENATED WITH right  
                END IF  
            END FOR  
        END FUNCTION  

        SET s TO SUBSTRING OF s FROM SECOND CHARACTER TO SECOND TO LAST CHARACTER  
        RETURN LIST OF STRINGS FOR EACH i FROM 1 TO LENGTH OF s MINUS 1:  
            FOR EACH a FROM make(SUBSTRING OF s FROM START TO i)  
            FOR EACH b FROM make(SUBSTRING OF s FROM i TO END)  
                FORMAT STRING AS "(" CONCATENATED WITH a CONCATENATED WITH ", " CONCATENATED WITH b CONCATENATED WITH ")"  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
