CLASS Solution  
    FUNCTION countDifferentSubsequenceGCDs(nums)  
        SET max_num TO maximum value IN nums  
        SET num_set TO set CONTAINING elements OF nums  
        SET count TO 0  
        
        FOR g FROM 1 TO max_num INCLUSIVE  
            SET gcd_val TO 0  
            FOR m FROM g TO max_num IN STEPS OF g  
                IF m IS IN num_set THEN  
                    SET gcd_val TO greatest common divisor OF gcd_val AND m  
                    IF gcd_val EQUALS g THEN  
                        INCREMENT count BY 1  
                        BREAK inner loop  
            END FOR  
        END FOR  
        
        RETURN count  
    END FUNCTION  
END CLASS
