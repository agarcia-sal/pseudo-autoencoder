CLASS Solution
    FUNCTION minCut(s)
        SET n TO LENGTH OF s
        IF n LESS THAN OR EQUAL TO 1 THEN
            RETURN 0
        END IF

        CREATE is_palindrome AS TWO-DIMENSIONAL LIST OF SIZE n BY n FILLED WITH False

        FOR i FROM 0 TO n MINUS 1 DO
            SET is_palindrome AT i AT i TO True
        END FOR

        FOR length FROM 2 TO n DO
            FOR start FROM 0 TO n MINUS length DO
                SET end TO start PLUS length MINUS 1
                IF length EQUALS 2 THEN
                    SET is_palindrome AT start AT end TO WHETHER s AT start EQUALS s AT end
                ELSE
                    SET is_palindrome AT start AT end TO WHETHER s AT start EQUALS s AT end AND is_palindrome AT start PLUS 1 AT end MINUS 1
                END IF
            END FOR
        END FOR

        CREATE min_cuts AS LIST OF SIZE n FILLED WITH 0

        FOR i FROM 0 TO n MINUS 1 DO
            IF is_palindrome AT 0 AT i THEN
                SET min_cuts AT i TO 0
            ELSE
                SET min_cuts AT i TO INFINITY
                FOR j FROM 1 TO i DO
                    IF is_palindrome AT j AT i AND min_cuts AT j MINUS 1 PLUS 1 LESS THAN min_cuts AT i THEN
                        SET min_cuts AT i TO min_cuts AT j MINUS 1 PLUS 1
                    END IF
                END FOR
            END IF
        END FOR

        RETURN min_cuts AT n MINUS 1
    END FUNCTION
END CLASS
