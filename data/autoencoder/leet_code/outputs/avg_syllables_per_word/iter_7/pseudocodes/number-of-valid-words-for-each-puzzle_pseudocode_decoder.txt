CLASS Solution  
    FUNCTION findNumOfValidWords(words, puzzles)  
        FUNCTION to_mask(s)  
            SET mask TO 0  
            FOR each char IN s  
                SET mask TO mask OR (1 shifted left BY (ordinal value of char MINUS ordinal value of 'a'))  
            RETURN mask  
        END FUNCTION  
        
        SET word_count TO a mapping with default integer zero  
        FOR each word IN words  
            SET mask TO to_mask(word)  
            IF number of set bits in mask IS LESS THAN OR EQUAL TO 7  
                INCREMENT word_count at key mask BY 1  
        
        SET result TO empty list  
        FOR each puzzle IN puzzles  
            SET first_char_mask TO 1 shifted left BY (ordinal value of first character in puzzle MINUS ordinal value of 'a')  
            SET puzzle_mask TO to_mask(all characters in puzzle except the first character)  
            SET count TO value in word_count at key first_char_mask  
            
            SET subset TO puzzle_mask  
            WHILE subset IS NOT zero  
                SET count TO count PLUS value in word_count at key (subset OR first_char_mask)  
                SET subset TO bitwise AND of (subset MINUS 1) and puzzle_mask  
            
            APPEND count TO result  
        
        RETURN result  
    END FUNCTION  
END CLASS
