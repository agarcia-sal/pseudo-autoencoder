CLASS Solution  
    FUNCTION addBoldTag(s, words)  
        IF words IS empty  
            RETURN s  
        SET n TO LENGTH OF s  
        SET mask TO list of False repeated n times  
        FOR each word IN words  
            SET start TO 0  
            WHILE start IS LESS THAN OR EQUAL TO n MINUS LENGTH OF word  
                SET start TO position of word in s starting from start  
                IF start IS EQUAL TO -1  
                    BREAK  
                FOR i FROM start TO start PLUS LENGTH OF word MINUS 1  
                    SET mask[i] TO True  
                INCREMENT start BY 1  
        SET result TO empty list  
        SET i TO 0  
        WHILE i IS LESS THAN n  
            IF mask[i] IS True  
                APPEND "<b>" TO result  
                WHILE i IS LESS THAN n AND mask[i] IS True  
                    APPEND s[i] TO result  
                    INCREMENT i BY 1  
                APPEND "</b>" TO result  
            ELSE  
                APPEND s[i] TO result  
                INCREMENT i BY 1  
        RETURN concatenation of elements in result  
    END FUNCTION  
END CLASS
