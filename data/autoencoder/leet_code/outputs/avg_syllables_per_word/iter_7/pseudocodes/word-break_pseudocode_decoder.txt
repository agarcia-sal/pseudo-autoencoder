CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO set of elements in wordDict  
        SET dp TO list of False values with length LENGTH OF s PLUS 1  
        SET dp at index 0 TO True  
        
        FOR each i FROM 1 TO LENGTH OF s INCLUSIVE  
            FOR each j FROM 0 TO i EXCLUSIVE  
                IF dp at index j IS True AND substring of s from j TO i IS IN wordSet  
                    SET dp at index i TO True  
                    BREAK inner loop  
        RETURN dp at index LENGTH OF s  
    END FUNCTION  
END CLASS
