CLASS Solution  
    FUNCTION findUnsortedSubarray(nums)  
        SET sorted_nums TO nums sorted in ascending order  
        SET start TO negative one  
        SET end TO negative two  
        FOR each index i FROM zero TO length of nums MINUS one  
            IF element at index i in nums IS NOT EQUAL TO element at index i in sorted_nums  
                IF start IS EQUAL TO negative one  
                    SET start TO i  
                SET end TO i  
        RETURN end MINUS start PLUS one  
    END FUNCTION  
END CLASS
