CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture IS EMPTY OR first row of picture IS EMPTY THEN  
            RETURN 0  
        END IF  
        
        SET m TO LENGTH OF picture  
        SET n TO LENGTH OF first row of picture  
        SET row_count TO list of zeros with length m  
        SET col_count TO list of zeros with length n  
        SET row_patterns TO empty mapping from pattern to count  
        
        FOR r FROM 0 TO m MINUS 1  
            SET pattern TO concatenation of elements in picture at row r  
            INCREMENT count of pattern IN row_patterns BY 1  
            FOR c FROM 0 TO n MINUS 1  
                IF element at picture[r][c] IS 'B' THEN  
                    INCREMENT row_count at r BY 1  
                    INCREMENT col_count at c BY 1  
                END IF  
            END FOR  
        END FOR  
        
        SET lonely_pixels TO 0  
        
        FOR r FROM 0 TO m MINUS 1  
            FOR c FROM 0 TO n MINUS 1  
                IF element at picture[r][c] IS 'B' THEN  
                    SET pattern TO concatenation of elements in picture at row r  
                    IF row_count at r EQUALS target AND col_count at c EQUALS target AND count of pattern IN row_patterns EQUALS target THEN  
                        INCREMENT lonely_pixels BY 1  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
