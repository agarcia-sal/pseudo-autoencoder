CLASS Solution  
    FUNCTION amountPainted(paint)  
        SET max_end TO the maximum end value found in paint  
        SET painted TO a list of size max_end PLUS 1 filled with -1  
        SET worklog TO an empty list  
        
        FOR each start, end IN paint  
            SET current_start TO start  
            SET new_paint TO 0  
            
            WHILE current_start IS LESS THAN end  
                IF painted AT current_start IS EQUAL TO -1  
                    SET painted AT current_start TO end  
                    INCREMENT new_paint BY 1  
                    INCREMENT current_start BY 1  
                ELSE  
                    SET current_start TO painted AT current_start  
            
            APPEND new_paint TO worklog  
        
        RETURN worklog  
    END FUNCTION  
END CLASS
