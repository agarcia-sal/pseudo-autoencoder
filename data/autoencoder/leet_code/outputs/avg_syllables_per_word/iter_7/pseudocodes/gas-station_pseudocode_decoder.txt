CLASS Solution  
    FUNCTION canCompleteCircuit(gas, cost)  
        IF sum of gas IS LESS THAN sum of cost  
            RETURN -1  
        SET start TO 0  
        SET total_tank TO 0  
        FOR i FROM 0 TO LENGTH OF gas MINUS 1  
            SET total_tank TO total_tank PLUS gas AT i MINUS cost AT i  
            IF total_tank IS LESS THAN 0  
                SET start TO i PLUS 1  
                SET total_tank TO 0  
        RETURN start  
    END FUNCTION  
END CLASS
