CLASS Solution  
    FUNCTION countTriplets(nums)  
        SET pair_and_count TO empty map with default integer value  
        
        FOR each i IN nums  
            FOR each j IN nums  
                SET key TO bitwise AND of i and j  
                INCREMENT pair_and_count at key BY 1  
        
        SET triplet_count TO 0  
        FOR each k IN nums  
            FOR each pair_and, count IN pair_and_count  
                IF bitwise AND of pair_and and k EQUALS 0  
                    INCREMENT triplet_count BY count  
        
        RETURN triplet_count  
    END FUNCTION  
END CLASS
