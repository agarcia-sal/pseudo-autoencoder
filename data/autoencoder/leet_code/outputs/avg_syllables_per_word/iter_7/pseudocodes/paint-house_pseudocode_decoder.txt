CLASS Solution  
    FUNCTION minCost(costs)  
        IF costs IS empty  
            RETURN 0  
        END IF  
        
        SET prev_red, prev_blue, prev_green TO costs AT index 0  
        
        FOR i FROM 1 TO LENGTH OF costs MINUS 1  
            SET current_red TO costs AT index i AT 0 PLUS MINIMUM OF prev_blue AND prev_green  
            SET current_blue TO costs AT index i AT 1 PLUS MINIMUM OF prev_red AND prev_green  
            SET current_green TO costs AT index i AT 2 PLUS MINIMUM OF prev_red AND prev_blue  
            
            SET prev_red, prev_blue, prev_green TO current_red, current_blue, current_green  
        END FOR  
        
        RETURN MINIMUM OF prev_red, prev_blue, prev_green  
    END FUNCTION  
END CLASS
