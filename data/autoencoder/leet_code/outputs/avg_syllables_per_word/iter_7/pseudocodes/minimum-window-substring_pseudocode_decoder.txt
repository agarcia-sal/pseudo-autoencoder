CLASS Solution  
    FUNCTION minWindow(s, t)  
        SET t_count TO frequency count of characters in t  
        SET required TO number of unique characters in t_count  
  
        SET left TO 0  
        SET right TO 0  
        SET formed TO 0  
        SET window_counts TO frequency count container  
  
        SET min_length TO infinity  
        SET min_start TO 0  
  
        WHILE right LESS THAN length of s  
            ADD character at s[right] TO window_counts  
  
            IF character at s[right] EXISTS in t_count AND window_counts[character at s[right]] EQUALS t_count[character at s[right]]  
                INCREMENT formed BY 1  
            END IF  
  
            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO character at s[left]  
  
                IF current window size (right MINUS left PLUS 1) LESS THAN min_length  
                    SET min_length TO current window size  
                    SET min_start TO left  
                END IF  
  
                DECREMENT window_counts[char] BY 1  
  
                IF char EXISTS in t_count AND window_counts[char] IS LESS THAN t_count[char]  
                    DECREMENT formed BY 1  
                END IF  
  
                INCREMENT left BY 1  
            END WHILE  
  
            INCREMENT right BY 1  
        END WHILE  
  
        IF min_length EQUALS infinity  
            RETURN empty string  
        ELSE  
            RETURN substring of s FROM min_start TO min_start PLUS min_length  
        END IF  
    END FUNCTION  
END CLASS
