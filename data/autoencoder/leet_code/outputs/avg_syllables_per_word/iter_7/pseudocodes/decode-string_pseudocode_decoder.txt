CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        
        FOR each char IN s  
            IF char IS a digit THEN  
                APPEND char TO d  
            ELSE IF char IS an alphabetic character THEN  
                APPEND char TO a  
            ELSE IF char IS the character '[' THEN  
                APPEND tuple of (a, integer value of d) TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char IS the character ']' THEN  
                SET p, n TO result of removing last element from stack  
                SET a TO p concatenated WITH a repeated n times  
        
        RETURN a  
    END FUNCTION  
END CLASS
