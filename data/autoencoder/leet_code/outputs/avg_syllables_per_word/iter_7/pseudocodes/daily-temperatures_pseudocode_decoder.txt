CLASS Solution  
    FUNCTION dailyTemperatures(temperatures)  
        SET n TO LENGTH OF temperatures  
        SET answer TO LIST OF n ZEROS  
        SET stack TO EMPTY LIST  
        
        FOR i FROM 0 TO n MINUS 1  
            WHILE stack IS NOT EMPTY AND temperatures AT i IS GREATER THAN temperatures AT stack LAST ELEMENT  
                SET prev_index TO REMOVE LAST ELEMENT FROM stack  
                SET answer AT prev_index TO i MINUS prev_index  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
