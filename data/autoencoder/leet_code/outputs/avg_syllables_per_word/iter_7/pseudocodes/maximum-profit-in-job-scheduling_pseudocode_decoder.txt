CLASS Solution  
    FUNCTION jobScheduling(startTime, endTime, profit)  
        SET jobs TO list of tuples combining startTime, endTime, and profit sorted by job end time  
        SET end_times TO list of job end times extracted from jobs  
        SET dp TO list of zeros with length equal to length of jobs plus one  
        FOR i FROM 1 TO length of jobs INCLUSIVE  
            SET j TO the position found by binary searching for jobs[i minus 1] start time in end_times  
            SET dp at position i TO the greater value between dp at position i minus 1 and dp at position j plus profit of jobs[i minus 1]  
        RETURN last element of dp  
    END FUNCTION  
END CLASS
