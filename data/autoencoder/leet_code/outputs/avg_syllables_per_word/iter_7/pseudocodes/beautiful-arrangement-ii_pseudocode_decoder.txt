CLASS Solution  
    FUNCTION constructArray(n, k)  
        SET result TO list of integers from 1 to n MINUS k MINUS 1  
        SET left TO n MINUS k  
        SET right TO n  
        WHILE left IS LESS THAN OR EQUAL TO right  
            IF left IS EQUAL TO right  
                APPEND left TO result  
            ELSE  
                APPEND left TO result  
                APPEND right TO result  
            END IF  
            INCREMENT left BY 1  
            DECREMENT right BY 1  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
