CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO list of twenty-six times negative one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        
        FOR each index i AND character char IN s  
            SET index TO numerical position of char in alphabet starting from zero  
            INCREMENT current_appeal BY i MINUS last AT position index  
            INCREMENT total_appeal BY current_appeal  
            SET last AT position index TO i  
        
        RETURN total_appeal  
    END FUNCTION  
END CLASS
