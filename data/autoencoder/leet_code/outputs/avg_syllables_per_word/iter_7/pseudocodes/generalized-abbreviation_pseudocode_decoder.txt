CLASS Solution  
    FUNCTION generateAbbreviations(word)  
        FUNCTION backtrack(index, path, count)  
            IF index EQUALS LENGTH OF word THEN  
                IF count GREATER THAN 0 THEN  
                    APPEND string representation of count TO path  
                APPEND concatenation of path TO result  
            ELSE  
                IF count GREATER THAN 0 THEN  
                    CALL backtrack WITH (index INCREMENTED BY 1, path EXTENDED BY string representation of count AND word AT index, 0)  
                ELSE  
                    CALL backtrack WITH (index INCREMENTED BY 1, path EXTENDED BY word AT index, 0)  
                CALL backtrack WITH (index INCREMENTED BY 1, path, count INCREMENTED BY 1)  
            END IF  
        END FUNCTION  

        SET result TO empty list  
        CALL backtrack WITH (0, empty list, 0)  
        RETURN result  
    END FUNCTION  
END CLASS
