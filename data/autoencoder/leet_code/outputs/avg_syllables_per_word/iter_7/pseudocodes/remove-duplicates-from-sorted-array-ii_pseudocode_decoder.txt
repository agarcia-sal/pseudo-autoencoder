CLASS Solution  
    FUNCTION removeDuplicates(nums)  
        IF nums IS empty THEN  
            RETURN 0  
        END IF  
        
        SET write_index TO 0  
        
        FOR i FROM 0 TO MINIMUM OF 2 AND LENGTH OF nums  
            SET nums AT write_index TO nums AT i  
            INCREMENT write_index BY 1  
        END FOR  
        
        FOR i FROM 2 TO LENGTH OF nums MINUS 1  
            IF nums AT i IS NOT EQUAL TO nums AT write_index MINUS 2 THEN  
                SET nums AT write_index TO nums AT i  
                INCREMENT write_index BY 1  
            END IF  
        END FOR  
        
        RETURN write_index  
    END FUNCTION  
END CLASS
