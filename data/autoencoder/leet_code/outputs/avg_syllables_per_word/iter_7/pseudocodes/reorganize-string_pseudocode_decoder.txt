CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO frequency count of characters in s  
        SET max_heap TO list of pairs of negative frequency and character from count  
        TRANSFORM max_heap into a heap structure  
        
        SET prev_char TO None  
        SET prev_freq TO 0  
        SET result TO empty list  
        
        WHILE max_heap is not empty OR prev_freq is less than 0  
            IF max_heap is empty AND prev_freq is less than 0  
                RETURN empty string  
            
            SET freq, char TO extract the top element from max_heap  
            APPEND char TO result  
            
            INCREMENT freq BY 1  
            
            IF prev_freq is less than 0  
                ADD pair of prev_freq and prev_char BACK TO max_heap  
            
            SET prev_char TO char  
            SET prev_freq TO freq  
        
        RETURN concatenation of elements in result  
    END FUNCTION  
END CLASS
