CLASS Solution  
    FUNCTION smallestGoodBase(n)  
        SET n TO integer conversion of n  
        SET max_length TO integer part of logarithm base 2 of n PLUS 1  
        FOR m FROM max_length DOWN TO 2  
            SET k TO integer part of n RAISED TO THE POWER OF (1 DIVIDED BY (m MINUS 1))  
            IF k LESS THAN 2  
                CONTINUE to next iteration  
            END IF  
            SET num TO result of (k RAISED TO THE POWER OF m MINUS 1) DIVIDED BY (k MINUS 1)  
            IF num EQUALS n  
                RETURN string conversion of k  
            END IF  
        END FOR  
        RETURN string conversion of (n MINUS 1)  
    END FUNCTION  
END CLASS
