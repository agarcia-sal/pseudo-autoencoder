CLASS Solution  
    FUNCTION isTransformable(s, t)  
        INITIALIZE pos AS mapping from number TO list  
        FOR each index i AND character c IN s  
            ADD index i TO heap at key CONVERT character c TO number IN pos  
        FOR each index i AND character c IN t  
            SET digit TO CONVERT character c TO number  
            IF heap at pos[digit] IS empty THEN  
                RETURN False  
            FOR each smaller FROM 0 TO digit MINUS 1  
                IF heap at pos[smaller] IS NOT empty AND smallest element IN pos[smaller] IS LESS THAN smallest element IN pos[digit] THEN  
                    RETURN False  
            REMOVE smallest element FROM heap at pos[digit]  
        RETURN True  
    END FUNCTION  
END CLASS
