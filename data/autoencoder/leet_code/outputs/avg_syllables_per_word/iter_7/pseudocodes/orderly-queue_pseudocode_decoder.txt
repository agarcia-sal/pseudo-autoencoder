CLASS Solution  
    FUNCTION orderlyQueue(s, k)  
        IF k GREATER THAN 1 THEN  
            RETURN string formed by sorting characters of s  
        ELSE  
            SET n TO length of s  
            SET smallest TO s  
            FOR i FROM 1 TO n MINUS 1 DO  
                SET rotated TO substring of s from i TO end CONCATENATED WITH substring of s from start TO i  
                IF rotated lexicographically LESS THAN smallest THEN  
                    SET smallest TO rotated  
            RETURN smallest  
    END FUNCTION  
END CLASS
