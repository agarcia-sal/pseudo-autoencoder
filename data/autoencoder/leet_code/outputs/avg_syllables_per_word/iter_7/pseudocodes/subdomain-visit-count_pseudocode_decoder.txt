CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        CREATE domain_count AS dictionary WITH default integer 0  
        
        FOR each cpdomain IN cpdomains  
            SPLIT cpdomain INTO count AND domain  
            CONVERT count TO integer  
            SPLIT domain BY '.' INTO subdomains  
            
            FOR each index FROM 0 TO LENGTH OF subdomains MINUS 1  
                JOIN subdomains FROM index TO END WITH '.' INTO subdomain  
                INCREMENT domain_count[subdomain] BY count  
        
        CREATE result_list AS empty list  
        FOR each domain AND count IN domain_count ITEMS  
            APPEND concatenation OF count AS string, space, AND domain TO result_list  
        RETURN result_list  
    END FUNCTION  
END CLASS
