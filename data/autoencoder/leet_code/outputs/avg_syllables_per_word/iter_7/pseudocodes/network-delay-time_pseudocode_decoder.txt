CLASS Solution  
    FUNCTION networkDelayTime(times, n, k)  
        CREATE graph AS DICTIONARY MAPPING each integer FROM 1 TO n TO empty list  
        FOR each u, v, w IN times  
            APPEND pair (v, w) TO graph[u]  
          
        CREATE dist AS DICTIONARY MAPPING each integer FROM 1 TO n TO infinity  
        SET dist[k] TO 0  
          
        CREATE min_heap AS LIST CONTAINING pair (0, k)  
          
        WHILE min_heap IS NOT EMPTY  
            EXTRACT current_dist AND node FROM min_heap USING min-heap pop operation  
              
            IF current_dist IS GREATER THAN dist[node]  
                CONTINUE TO next iteration  
              
            FOR each neighbor, weight IN graph[node]  
                SET distance TO current_dist PLUS weight  
                  
                IF distance IS LESS THAN dist[neighbor]  
                    SET dist[neighbor] TO distance  
                    ADD pair (distance, neighbor) TO min_heap USING min-heap push operation  
          
        SET max_time TO MAXIMUM VALUE IN dist  
          
        IF max_time IS EQUAL TO infinity  
            RETURN -1  
        ELSE  
            RETURN max_time  
    END FUNCTION  
END CLASS
