CLASS Solution  
    FUNCTION minRefuelStops(target, startFuel, stations)  
        INITIALIZE max_heap AS empty list  
        SET current_fuel TO startFuel  
        SET stops TO 0  
        APPEND list containing target and 0 TO stations  

        FOR each position, fuel IN stations  
            WHILE max_heap is not empty AND current_fuel is less than position  
                SET largest_fuel TO REMOVE largest fuel amount from max_heap  
                INCREMENT current_fuel BY largest_fuel  
                INCREMENT stops BY 1  
            IF current_fuel is less than position  
                RETURN -1  
            ADD negative of fuel TO max_heap  
        RETURN stops  
    END FUNCTION  
END CLASS
