CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO TreeNode WITH val AS values AT position 0  
    SET i TO 1  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO queue FIRST ELEMENT AND REMOVE IT  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.left TO TreeNode WITH val AS values AT i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH OF values AND values AT i IS NOT None  
            SET node.right TO TreeNode WITH val AS values AT i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(self, root, k)  
        SET stack TO empty list  
        SET current TO root  
        SET count TO 0  
        
        WHILE stack IS NOT empty OR current IS NOT None  
            WHILE current IS NOT None  
                APPEND current TO stack  
                SET current TO current.left  
            SET current TO LAST ELEMENT OF stack AND REMOVE IT  
            INCREMENT count BY 1  
            IF count EQUALS k  
                RETURN current.val  
            SET current TO current.right  
    END FUNCTION  
END CLASS
