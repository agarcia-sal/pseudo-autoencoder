CLASS Solution  
    FUNCTION superEggDrop(k, n)  
        SET dp TO a list of zeroes with length k plus one  
        SET moves TO 0  
        WHILE value at dp position k IS LESS THAN n  
            INCREMENT moves BY 1  
            FOR eggs FROM k DOWN TO 1  
                SET dp at eggs TO 1 PLUS dp at eggs minus one PLUS dp at eggs  
        RETURN moves  
    END FUNCTION  
END CLASS
