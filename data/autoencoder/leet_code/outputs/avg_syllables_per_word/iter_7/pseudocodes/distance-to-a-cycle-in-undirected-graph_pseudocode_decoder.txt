CLASS Solution  
    FUNCTION distanceToCycle(n, edges)  
        SET graph TO a mapping with default empty list  
        FOR each pair u, v IN edges  
            APPEND v TO graph[u]  
            APPEND u TO graph[v]  

        FUNCTION dfs(node, parent, path, visited)  
            IF node IS IN visited THEN  
                SET cycle_start TO position of node IN path  
                RETURN sublist of path FROM cycle_start TO end  
            END IF  
            ADD node TO visited  
            APPEND node TO path  
            FOR each neighbor IN graph[node]  
                IF neighbor IS NOT EQUAL TO parent THEN  
                    SET cycle TO dfs(neighbor, node, path, visited)  
                    IF cycle IS NOT null THEN  
                        RETURN cycle  
                    END IF  
                END IF  
            END FOR  
            REMOVE last element FROM path  
            RETURN null  
        END FUNCTION  

        SET visited TO empty set  
        SET cycle TO null  
        FOR node FROM 0 TO n MINUS 1  
            IF node IS NOT IN visited THEN  
                SET cycle TO dfs(node, negative one, empty list, visited)  
                IF cycle IS NOT null THEN  
                    BREAK  
                END IF  
            END IF  
        END FOR  

        SET distance TO list of zeros with length n  
        SET queue TO deque containing all elements of cycle  
        SET visited TO set containing all elements of cycle  

        SET level TO zero  
        WHILE queue IS NOT empty  
            FOR each element IN queue with current length  
                REMOVE node FROM front of queue  
                SET distance[node] TO level  
                FOR each neighbor IN graph[node]  
                    IF neighbor IS NOT IN visited THEN  
                        ADD neighbor TO visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT level BY one  
        END WHILE  

        RETURN distance  
    END FUNCTION  
END CLASS
