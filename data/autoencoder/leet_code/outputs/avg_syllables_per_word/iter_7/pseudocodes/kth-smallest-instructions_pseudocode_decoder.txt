CLASS Solution  
    FUNCTION kthSmallestPath(destination, k)  
        SET row TO destination AT index 0  
        SET col TO destination AT index 1  
        SET path TO empty list  
        SET total_moves TO col PLUS row  
        
        FOR i FROM 0 TO total_moves MINUS 1  
            IF col IS GREATER THAN 0 THEN  
                SET paths_with_H TO combination OF (row PLUS col MINUS 1) CHOOSE (col MINUS 1)  
                IF k IS LESS THAN OR EQUAL TO paths_with_H THEN  
                    APPEND 'H' TO path  
                    DECREMENT col BY 1  
                ELSE  
                    APPEND 'V' TO path  
                    DECREMENT row BY 1  
                    DECREMENT k BY paths_with_H  
            ELSE  
                APPEND 'V' TO path  
                DECREMENT row BY 1  
        
        RETURN concatenation OF elements IN path  
    END FUNCTION  
END CLASS
