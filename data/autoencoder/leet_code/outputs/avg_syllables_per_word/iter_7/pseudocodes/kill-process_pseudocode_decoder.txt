CLASS Solution  
    FUNCTION killProcess(pid, ppid, kill)  
        SET tree TO a mapping from parent process to list of child processes  
        FOR each child, parent IN pairs from pid and ppid  
            APPEND child TO tree at key parent  
        SET queue TO list containing kill  
        SET killed_processes TO empty list  
        WHILE queue is not empty  
            REMOVE first element from queue and SET TO current  
            APPEND current TO killed_processes  
            EXTEND queue BY all children of current from tree  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
