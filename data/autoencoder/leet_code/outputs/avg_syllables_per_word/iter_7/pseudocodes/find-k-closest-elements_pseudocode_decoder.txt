CLASS Solution  
    FUNCTION findClosestElements(arr, k, x)  
        SET left TO position where x would be inserted in arr to keep it sorted MINUS 1  
        SET right TO left PLUS 1  
        
        WHILE right MINUS left MINUS 1 LESS THAN k  
            IF left EQUALS minus 1 OR (right LESS THAN length of arr AND absolute difference BETWEEN arr at right AND x LESS THAN absolute difference BETWEEN arr at left AND x)  
                INCREMENT right BY 1  
            ELSE  
                DECREMENT left BY 1  
        RETURN sublist of arr FROM left PLUS 1 TO right  
    END FUNCTION  
END CLASS
