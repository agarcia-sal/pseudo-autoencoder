CLASS Solution  
    FUNCTION isConvex(points)  
        FUNCTION cross_product(o, a, b)  
            RETURN (a at position zero MINUS o at position zero) MULTIPLIED BY (b at position one MINUS o at position one) MINUS (a at position one MINUS o at position one) MULTIPLIED BY (b at position zero MINUS o at position zero)  
        END FUNCTION  
        
        SET n TO LENGTH OF points  
        IF n LESS THAN OR EQUAL TO 3 THEN  
            RETURN True  
        END IF  
        
        SET orientation TO 0  
        
        FOR i FROM 0 TO n MINUS 1  
            SET o TO points at position i  
            SET a TO points at position (i PLUS 1) MODULO n  
            SET b TO points at position (i PLUS 2) MODULO n  
            SET cross TO cross_product(o, a, b)  
            
            IF cross NOT EQUAL TO 0 THEN  
                IF orientation EQUALS 0 THEN  
                    IF cross GREATER THAN 0 THEN  
                        SET orientation TO 1  
                    ELSE  
                        SET orientation TO NEGATIVE 1  
                    END IF  
                ELSE IF (orientation EQUALS 1 AND cross LESS THAN 0) OR (orientation EQUALS NEGATIVE 1 AND cross GREATER THAN 0) THEN  
                    RETURN False  
                END IF  
            END IF  
        END FOR  
        
        RETURN True  
    END FUNCTION  
END CLASS
