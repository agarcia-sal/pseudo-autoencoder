CLASS Solution  
    FUNCTION canSeePersonsCount(heights)  
        SET n TO LENGTH OF heights  
        SET answer TO list of zeros with length n  
        SET stack TO empty list  
        
        FOR i FROM 0 TO n MINUS 1  
            WHILE stack IS NOT empty AND height AT index of last element in stack IS LESS THAN height AT i  
                SET top_index TO pop last element FROM stack  
                INCREMENT answer AT top_index BY 1  
            IF stack IS NOT empty  
                INCREMENT answer AT last index in stack BY 1  
            APPEND i TO stack  
        
        RETURN answer  
    END FUNCTION  
END CLASS
