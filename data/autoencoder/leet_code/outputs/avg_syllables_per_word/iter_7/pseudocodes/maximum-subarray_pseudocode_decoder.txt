CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO first element of nums  
        SET current_sum TO first element of nums  
        FOR each num IN nums starting from second element  
            SET current_sum TO the greater of num and current_sum plus num  
            SET max_sum TO the greater of max_sum and current_sum  
        RETURN max_sum  
    END FUNCTION  
END CLASS
