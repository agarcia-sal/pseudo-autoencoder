CLASS Solution  
    FUNCTION minMoves(nums, k)  
        SET positions TO list of indices i in nums where nums at i EQUALS 1  
        
        FUNCTION calculate_cost(start, end)  
            SET mid TO start PLUS end DIVIDED BY 2  
            SET median TO positions at mid  
            SET cost TO 0  
            FOR i FROM start TO end MINUS 1  
                INCREMENT cost BY absolute value of positions at i MINUS (median MINUS (mid MINUS i))  
            END FOR  
            RETURN cost  
        END FUNCTION  
        
        SET min_cost TO infinity  
        FOR i FROM 0 TO LENGTH OF positions MINUS k  
            SET current_cost TO calculate_cost(i, i PLUS k)  
            IF current_cost LESS THAN min_cost  
                SET min_cost TO current_cost  
            END IF  
        END FOR  
        
        RETURN min_cost  
    END FUNCTION  
END CLASS
