CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO LENGTH OF statements  
        SET max_good TO 0  
        
        FOR i FROM 0 TO (1 SHIFTED LEFT BY n) MINUS 1  
            SET good TO LIST OF n TIMES False  
            SET bad TO LIST OF n Times False  
            
            FOR j FROM 0 TO n MINUS 1  
                IF ith BIT OF i IS SET  
                    SET good AT j TO True  
                ELSE  
                    SET bad AT j TO True  
            
            SET valid TO True  
            FOR j FROM 0 TO n MINUS 1  
                IF good AT j IS True  
                    FOR k FROM 0 TO n MINUS 1  
                        IF (statements AT j AT k EQUALS 0 AND good AT k IS True) OR (statements AT j AT k EQUALS 1 AND bad AT k IS True)  
                            SET valid TO False  
                            BREAK INNER LOOP  
                    IF valid IS False  
                        BREAK OUTER LOOP  
            
            IF valid IS True  
                SET max_good TO MAXIMUM OF max_good AND SUM OF good  
        
        RETURN max_good  
    END FUNCTION  
END CLASS
