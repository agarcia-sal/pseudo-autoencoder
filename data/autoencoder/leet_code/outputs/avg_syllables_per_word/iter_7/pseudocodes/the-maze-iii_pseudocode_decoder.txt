CLASS Solution  
    FUNCTION findShortestWay(maze, ball, hole)  
        SET directions TO list of (0, 1, 'r'), (1, 0, 'd'), (0, -1, 'l'), (-1, 0, 'u')  
        SET m TO length of maze  
        SET n TO length of maze at index zero  
        SET start TO tuple of ball  
        SET hole TO tuple of hole  
        SET pq TO list containing tuple of 0, empty string, start  
        SET visited TO empty set  
        
        WHILE pq is not empty  
            REMOVE element with smallest distance from pq and UNPACK as dist, path, (x, y)  
            
            IF (x, y) EQUALS hole  
                RETURN path  
            
            IF (x, y) IS IN visited  
                CONTINUE  
            
            ADD (x, y) TO visited  
            
            FOR each dx, dy, direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO 0  
                
                WHILE new_x plus dx IS GREATER THAN OR EQUAL TO 0 AND LESS THAN m  
                      AND new_y plus dy IS GREATER THAN OR EQUAL TO 0 AND LESS THAN n  
                      AND maze value at new_x plus dx and new_y plus dy EQUALS 0  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY 1  
                    IF (new_x, new_y) EQUALS hole  
                        BREAK  
                
                IF (new_x, new_y) IS NOT IN visited  
                    ADD tuple of dist plus count, path concatenated with direction, (new_x, new_y) TO pq  
        
        RETURN "impossible"  
    END FUNCTION  
END CLASS
