CLASS ListNode  
    FUNCTION __init__(self, val SET TO 0, next SET TO None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    SET head TO new ListNode WITH val AS first element of values  
    SET p TO head  
    FOR each val IN values FROM second element TO end  
        SET node TO new ListNode WITH val AS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN (p1.val EQUALS p2.val) AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates(self, head)  
        SET dummy TO new ListNode WITH val AS 0 AND next AS head  
        SET prev TO dummy  
        
        WHILE head IS NOT None  
            IF head.next IS NOT None AND head.val EQUALS head.next.val  
                WHILE head.next IS NOT None AND head.val EQUALS head.next.val  
                    SET head TO head.next  
                END WHILE  
                SET prev.next TO head.next  
            ELSE  
                SET prev TO prev.next  
            END IF  
            SET head TO head.next  
        END WHILE  
        
        RETURN dummy.next  
    END FUNCTION  
END CLASS
