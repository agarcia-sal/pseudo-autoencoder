CLASS Solution  
    FUNCTION pourWater(heights, volume, k)  
        SET n TO LENGTH OF heights  
        
        FOR each iteration FROM 1 TO volume  
            SET i TO k  
            WHILE i GREATER THAN 0 AND heights AT POSITION (i MINUS 1) LESS THAN OR EQUAL TO heights AT POSITION i  
                DECREMENT i BY 1  
            END WHILE  
            
            WHILE i LESS THAN k AND heights AT POSITION i EQUALS heights AT POSITION (i PLUS 1)  
                INCREMENT i BY 1  
            END WHILE  
            
            IF i LESS THAN k THEN  
                INCREMENT heights AT POSITION i BY 1  
                CONTINUE TO NEXT iteration  
            END IF  
            
            SET i TO k  
            WHILE i LESS THAN n MINUS 1 AND heights AT POSITION (i PLUS 1) LESS THAN OR EQUAL TO heights AT POSITION i  
                INCREMENT i BY 1  
            END WHILE  
            
            WHILE i GREATER THAN k AND heights AT POSITION i EQUALS heights AT POSITION (i MINUS 1)  
                DECREMENT i BY 1  
            END WHILE  
            
            IF i GREATER THAN k THEN  
                INCREMENT heights AT POSITION i BY 1  
                CONTINUE TO NEXT iteration  
            END IF  
            
            INCREMENT heights AT POSITION k BY 1  
        END FOR  
        
        RETURN heights  
    END FUNCTION  
END CLASS
