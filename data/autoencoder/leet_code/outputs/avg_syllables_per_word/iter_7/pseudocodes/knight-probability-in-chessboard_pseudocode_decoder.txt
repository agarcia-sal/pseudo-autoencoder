CLASS Solution  
    FUNCTION knightProbability(n, k, row, column)  
        SET moves TO list of tuples [(-2, -1), (-2, 1), (-1, -2), (-1, 2), (1, -2), (1, 2), (2, -1), (2, 1)]  
        SET dp TO two-dimensional list of size n by n with all values 0  
        SET dp at position row, column TO 1  
        
        FOR each move_index FROM 1 TO k DO  
            SET new_dp TO two-dimensional list of size n by n with all values 0  
            
            FOR r FROM 0 TO n MINUS 1 DO  
                FOR c FROM 0 TO n MINUS 1 DO  
                    IF dp at position r, c EQUALS 0 THEN  
                        CONTINUE to next iteration  
                    
                    FOR each (dr, dc) IN moves DO  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        
                        IF nr GREATER THAN OR EQUAL TO 0 AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO 0 AND nc LESS THAN n THEN  
                            INCREMENT new_dp at position nr, nc BY dp at position r, c DIVIDED BY 8  
            SET dp TO new_dp  
        
        SET total_probability TO 0  
        FOR each row_list IN dp DO  
            FOR each value IN row_list DO  
                INCREMENT total_probability BY value  
        RETURN total_probability  
    END FUNCTION  
END CLASS
