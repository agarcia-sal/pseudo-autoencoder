CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO list of string representations of each element in nums  
        
        FUNCTION compare(x, y)  
            IF concatenation of x and y is greater than concatenation of y and x THEN  
                RETURN negative one  
            ELSE IF concatenation of x and y is less than concatenation of y and x THEN  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        
        SORT nums_str using compare as comparator  
        
        IF first element of nums_str equals "0" THEN  
            RETURN "0"  
        END IF  
        
        RETURN concatenation of all elements in nums_str  
    END FUNCTION  
END CLASS
