CLASS Solution  
    FUNCTION countAndSay(n)  
        IF n EQUALS 1 THEN  
            RETURN "1"  
        END IF  

        SET current_sequence TO "1"  

        FOR iteration FROM 1 TO n MINUS 1 DO  
            SET next_sequence TO empty string  
            SET count TO 1  
            SET previous_char TO first character of current_sequence  

            FOR i FROM 1 TO LENGTH OF current_sequence MINUS 1 DO  
                IF character at position i of current_sequence EQUALS previous_char THEN  
                    INCREMENT count BY 1  
                ELSE  
                    APPEND representation of count CONCATENATED WITH previous_char TO next_sequence  
                    SET previous_char TO character at position i of current_sequence  
                    SET count TO 1  
                END IF  
            END FOR  

            APPEND representation of count CONCATENATED WITH previous_char TO next_sequence  
            SET current_sequence TO next_sequence  
        END FOR  

        RETURN current_sequence  
    END FUNCTION  
END CLASS
