CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO LENGTH OF grid  
        SET visited TO empty set  
        SET min_heap TO list containing tuple of (grid at position 0,0, 0, 0)  
        ADD tuple (0, 0) TO visited  
        
        SET directions TO list of tuples (0, 1), (1, 0), (0, -1), (-1, 0)  
        
        WHILE min_heap is not empty  
            REMOVE smallest element FROM min_heap AND DESTRUCTURE INTO t, i, j  
            IF i EQUALS n MINUS 1 AND j EQUALS n MINUS 1  
                RETURN t  
            
            FOR each di, dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni GREATER THAN OR EQUAL TO 0 AND ni LESS THAN n AND nj GREATER THAN OR EQUAL TO 0 AND nj LESS THAN n AND tuple (ni, nj) NOT IN visited  
                    ADD tuple (ni, nj) TO visited  
                    ADD tuple (maximum of t AND grid at position ni, nj, ni, nj) TO min_heap  
        END WHILE  
        
        RETURN minus 1  
    END FUNCTION  
END CLASS
