CLASS Solution  
    FUNCTION gameOfLife(board)  
        IF board IS EMPTY OR board's first row IS EMPTY  
            RETURN  
          
        SET m TO LENGTH OF board  
        SET n TO LENGTH OF board's first row  
        
        SET directions TO LIST OF 8 PAIRS REPRESENTING NEIGHBOR OFFSET COORDINATES  
        
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                SET live_neighbors TO 0  
                FOR EACH dx, dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni IS GREATER THAN OR EQUAL TO 0 AND ni IS LESS THAN m AND nj IS GREATER THAN OR EQUAL TO 0 AND nj IS LESS THAN n  
                        IF board AT ni AND nj IS 1 OR board AT ni AND nj IS 3  
                            INCREMENT live_neighbors BY 1  
                IF board AT i AND j IS 1  
                    IF live_neighbors IS LESS THAN 2 OR live_neighbors IS GREATER THAN 3  
                        SET board AT i AND j TO 3  
                ELSE  
                    IF live_neighbors IS EQUAL TO 3  
                        SET board AT i AND j TO 2  
        
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF board AT i AND j IS 2  
                    SET board AT i AND j TO 1  
                ELSE IF board AT i AND j IS 3  
                    SET board AT i AND j TO 0  
    END FUNCTION  
END CLASS
