CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO LENGTH OF nums1  
        SET n TO LENGTH OF nums2  
  
        INITIALIZE dp AS two-dimensional array of size (m + 1) by (n + 1) WITH very small number  
  
        FOR i FROM 1 TO m INCLUSIVE  
            FOR j FROM 1 TO n INCLUSIVE  
                SET current_product TO element at position i MINUS 1 IN nums1 MULTIPLIED BY element at position j MINUS 1 IN nums2  
  
                SET dp at position (i, j) TO maximum of  
                    current_product,  
                    dp at position (i MINUS 1, j),  
                    dp at position (i, j MINUS 1),  
                    dp at position (i MINUS 1, j MINUS 1) PLUS current_product  
  
        RETURN dp at position (m, n)  
    END FUNCTION  
END CLASS
