CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO LENGTH OF jobs  
        
        SET subset_sums TO LIST OF SIZE 2 TO THE POWER OF n FILLED WITH 0  
        
        FOR i FROM 0 TO n MINUS 1  
            FOR mask FROM 0 TO 2 TO THE POWER OF i MINUS 1  
                SET subset_sums AT POSITION (mask OR 2 TO THE POWER OF i) TO subset_sums AT POSITION mask PLUS jobs AT POSITION i  
        
        FUNCTION dp(k, mask)  
            IF k EQUALS 1 THEN  
                RETURN subset_sums AT POSITION mask  
            SET res TO INFINITY  
            SET submask TO mask  
            WHILE submask IS NOT 0  
                SET candidate TO MAXIMUM OF dp(k MINUS 1, mask XOR submask) AND subset_sums AT POSITION submask  
                SET res TO MINIMUM OF res AND candidate  
                SET submask TO (submask MINUS 1) BITWISE-AND mask  
            RETURN res  
        
        RETURN dp(k, 2 TO THE POWER OF n MINUS 1)  
    END FUNCTION  
END CLASS
