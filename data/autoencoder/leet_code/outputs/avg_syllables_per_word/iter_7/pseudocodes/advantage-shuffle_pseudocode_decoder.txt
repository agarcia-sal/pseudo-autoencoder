CLASS Solution  
    FUNCTION advantageCount(nums1, nums2)  
        SORT nums1 IN ASCENDING ORDER  
        SET nums2_with_indices TO LIST OF PAIRS OF EACH VALUE AND ITS ORIGINAL INDEX IN nums2 SORTED IN DESCENDING ORDER BY VALUE  
        SET result TO LIST OF ZEROS WITH LENGTH EQUAL TO LENGTH OF nums1  
        SET left TO 0  
        SET right TO LENGTH OF nums1 MINUS 1  
        FOR each value, index IN nums2_with_indices  
            IF nums1 AT right GREATER THAN value  
                SET result AT index TO nums1 AT right  
                DECREMENT right BY 1  
            ELSE  
                SET result AT index TO nums1 AT left  
                INCREMENT left BY 1  
        RETURN result  
    END FUNCTION  
END CLASS
