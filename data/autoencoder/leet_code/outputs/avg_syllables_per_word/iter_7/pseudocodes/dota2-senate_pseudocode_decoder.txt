CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        INITIALIZE radiant AS empty queue  
        INITIALIZE dire AS empty queue  
        
        FOR each index i AND character s IN senate  
            IF s EQUALS 'R' THEN  
                APPEND i TO radiant  
            ELSE  
                APPEND i TO dire  
        
        WHILE radiant IS NOT empty AND dire IS NOT empty  
            SET r_index TO REMOVE first element FROM radiant  
            SET d_index TO REMOVE first element FROM dire  
            
            IF r_index IS LESS THAN d_index THEN  
                APPEND r_index PLUS LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS LENGTH OF senate TO dire  
        
        IF radiant IS NOT empty THEN  
            RETURN "Radiant"  
        ELSE  
            RETURN "Dire"  
    END FUNCTION  
END CLASS
