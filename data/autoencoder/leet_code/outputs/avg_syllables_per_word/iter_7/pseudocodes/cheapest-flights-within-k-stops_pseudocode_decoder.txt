CLASS Solution  
    FUNCTION findCheapestPrice(n, flights, src, dst, k)  
        SET graph TO a mapping from each integer from 0 to n minus 1 TO empty list  
        FOR each (u, v, w) IN flights  
            APPEND (v, w) TO graph at key u  
        
        SET min_heap TO a list containing a tuple of 0, negative 1, and src  
        SET min_cost TO a mapping with key src mapped to negative 1  
        
        WHILE min_heap is not empty  
            REMOVE the element with the smallest cost from min_heap and assign to (cost, stops, city)  
            
            IF city is equal to dst THEN  
                RETURN cost  
            
            IF stops is greater than or equal to k THEN  
                CONTINUE to next iteration  
            
            FOR each (neighbor, price) IN graph at key city  
                SET next_cost TO cost plus price  
                SET next_stops TO stops plus 1  
                
                IF neighbor not in min_cost OR next_stops less than min_cost at key neighbor OR next_cost less than the smallest cost in min_heap THEN  
                    SET min_cost at key neighbor TO next_stops  
                    ADD tuple of next_cost, next_stops, neighbor TO min_heap  
        
        RETURN negative 1  
    END FUNCTION  
END CLASS
