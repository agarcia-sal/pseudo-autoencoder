CLASS Solution  
    FUNCTION isMatch(s, p)  
        INITIALIZE dp AS two-dimensional list of False with dimensions (LENGTH OF s PLUS 1) BY (LENGTH OF p PLUS 1)  
        SET dp AT 0 AND 0 TO True  
        
        FOR j FROM 1 TO LENGTH OF p  
            IF CHARACTER AT p AT POSITION j MINUS 1 EQUALS '*'
                SET dp AT 0 AND j TO dp AT 0 AND j MINUS 1  
        
        FOR i FROM 1 TO LENGTH OF s  
            FOR j FROM 1 TO LENGTH OF p  
                IF CHARACTER AT p AT POSITION j MINUS 1 EQUALS '*'
                    SET dp AT i AND j TO dp AT i MINUS 1 AND j OR dp AT i AND j MINUS 1  
                ELSE IF CHARACTER AT p AT POSITION j MINUS 1 EQUALS '?' OR CHARACTER AT s AT POSITION i MINUS 1 EQUALS CHARACTER AT p AT POSITION j MINUS 1  
                    SET dp AT i AND j TO dp AT i MINUS 1 AND j MINUS 1  
        
        RETURN dp AT LENGTH OF s AND LENGTH OF p  
    END FUNCTION  
END CLASS
