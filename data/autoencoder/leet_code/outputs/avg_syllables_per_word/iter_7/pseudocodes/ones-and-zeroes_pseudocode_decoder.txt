CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        INITIALIZE dp AS 2D LIST WITH DIMENSIONS (m PLUS 1) BY (n PLUS 1) FILLED WITH ZEROES  
        
        FOR each string s IN strs  
            SET count_0 TO NUMBER OF ZEROS IN s  
            SET count_1 TO NUMBER OF ONES IN s  
            
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp[i][j] TO GREATER OF dp[i][j] AND (dp[i MINUS count_0][j MINUS count_1] PLUS 1)  
        
        RETURN dp[m][n]  
    END FUNCTION  
END CLASS
