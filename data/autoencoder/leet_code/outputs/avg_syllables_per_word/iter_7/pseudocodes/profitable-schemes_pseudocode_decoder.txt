CLASS Solution  
    FUNCTION profitableSchemes(n, minProfit, group, profit)  
        SET MOD TO 10 raised to the power of 9 plus 7  
        CREATE dp AS list of lists with dimensions (n plus 1) by (minProfit plus 1) filled with zero  
        FOR i FROM 0 TO n INCLUSIVE  
            SET dp at index i and 0 TO 1  
        FOR each pair of p, g IN profit and group  
            FOR i FROM n DOWN TO g  
                FOR j FROM minProfit DOWN TO 0  
                    SET dp at index i and j TO (dp at index i and j plus dp at index (i minus g) and maximum of 0 and (j minus p)) modulo MOD  
        RETURN dp at index n and minProfit  
    END FUNCTION  
END CLASS
