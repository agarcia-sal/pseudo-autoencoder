CLASS Solution  
    FUNCTION countPrimes(n)  
        IF n LESS THAN OR EQUAL TO 2 THEN  
            RETURN 0  
        END IF  
        
        SET is_prime TO list of Boolean TRUE repeated n times  
        SET is_prime at position 0 TO FALSE  
        SET is_prime at position 1 TO FALSE  
        
        FOR start FROM 2 TO integer part of square root of n INCLUSIVE  
            IF is_prime at position start IS TRUE THEN  
                FOR multiple FROM start times start TO n EXCLUSIVE STEP start  
                    SET is_prime at position multiple TO FALSE  
                END FOR  
            END IF  
        END FOR  
        
        RETURN count of TRUE values in is_prime  
    END FUNCTION  
END CLASS
