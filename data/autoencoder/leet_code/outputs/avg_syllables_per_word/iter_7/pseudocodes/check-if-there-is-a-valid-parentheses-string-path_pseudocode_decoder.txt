CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO LENGTH OF grid  
        SET n TO LENGTH OF grid AT INDEX 0  
        
        IF (m PLUS n MINUS 1) MODULO 2 IS NOT EQUAL TO 0  
            RETURN False  
        END IF  
        
        INITIALIZE dp AS 3D LIST OF SIZE (m PLUS 1) BY (n PLUS 1) BY (m PLUS n PLUS 1) WITH ALL False  
        SET dp AT INDEX 0, 1, 0 TO True  
        
        FOR i FROM 0 TO m MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                FOR k FROM 0 TO m PLUS n  
                    IF dp AT INDEX i, j PLUS 1, k OR dp AT INDEX i PLUS 1, j, k IS True  
                        IF grid AT INDEX i, j IS '(' AND k PLUS 1 LESS THAN m PLUS n PLUS 1  
                            SET dp AT INDEX i PLUS 1, j PLUS 1, k PLUS 1 TO True  
                        ELSE IF grid AT INDEX i, j IS ')' AND k MINUS 1 GREATER THAN OR EQUAL TO 0  
                            SET dp AT INDEX i PLUS 1, j PLUS 1, k MINUS 1 TO True  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN dp AT INDEX m, n, 0  
    END FUNCTION  
END CLASS
