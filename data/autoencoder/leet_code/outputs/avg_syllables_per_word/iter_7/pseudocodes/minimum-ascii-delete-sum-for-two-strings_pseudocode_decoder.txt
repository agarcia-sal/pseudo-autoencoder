CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO LENGTH OF s1  
        SET n TO LENGTH OF s2  
        
        SET dp TO 2D list with dimensions (m plus 1) by (n plus 1), all initialized to 0  
        
        FOR j FROM 1 TO n INCLUSIVE  
            SET dp at row 0 and column j TO dp at row 0 and column (j minus 1) PLUS ASCII value of character at position (j minus 1) in s2  
        
        FOR i FROM 1 TO m INCLUSIVE  
            SET dp at row i and column 0 TO dp at row (i minus 1) and column 0 PLUS ASCII value of character at position (i minus 1) in s1  
        
        FOR i FROM 1 TO m INCLUSIVE  
            FOR j FROM 1 TO n INCLUSIVE  
                IF character at position (i minus 1) in s1 EQUALS character at position (j minus 1) in s2 THEN  
                    SET dp at row i and column j TO dp at row (i minus 1) and column (j minus 1)  
                ELSE  
                    SET dp at row i and column j TO MINIMUM OF (dp at row (i minus 1) and column j PLUS ASCII value of character at position (i minus 1) in s1) AND (dp at row i and column (j minus 1) PLUS ASCII value of character at position (j minus 1) in s2)  
        
        RETURN dp at row m and column n  
    END FUNCTION  
END CLASS
