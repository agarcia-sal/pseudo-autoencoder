CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN 0 THEN  
            SET sign TO -1  
        ELSE  
            SET sign TO 1  
        END IF  
        SET reversed_x TO integer conversion of the reversed string of the absolute value of x  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative two TO THE POWER OF thirty ONE OR result GREATER THAN two TO THE POWER OF thirty ONE MINUS 1 THEN  
            RETURN 0  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
