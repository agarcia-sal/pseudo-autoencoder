CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO NEW TreeNode WITH val AS first element of values, left AS None, right AS None  
    SET i TO 1  
    INITIALIZE queue AS new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element removed from queue  
        IF i IS LESS THAN length of values AND values at i IS NOT None  
            SET node.left TO NEW TreeNode WITH val AS values at i, left AS None, right AS None  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN length of values AND values at i IS NOT None  
            SET node.right TO NEW TreeNode WITH val AS values at i, left AS None, right AS None  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree(self, root)  
        IF root IS None OR root.left IS None  
            RETURN root  
        SET new_root TO self.upsideDownBinaryTree(root.left)  
        SET root.left.left TO root.right  
        SET root.left.right TO root  
        SET root.left TO None  
        SET root.right TO None  
        RETURN new_root  
    END FUNCTION  
END CLASS
