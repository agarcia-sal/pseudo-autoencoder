CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF LENGTH OF edges IS NOT EQUAL TO n MINUS 1  
            RETURN False  
        
        SET adj_list TO EMPTY DICTIONARY WITH KEYS FROM 0 TO n MINUS 1 AND VALUES AS EMPTY LISTS  
        FOR EACH pair a, b IN edges  
            APPEND b TO adj_list AT KEY a  
            APPEND a TO adj_list AT KEY b  
        
        SET visited TO EMPTY SET  
        
        FUNCTION dfs(node)  
            IF node IS IN visited  
                RETURN  
            ADD node TO visited  
            FOR EACH neighbor IN adj_list AT KEY node  
                CALL dfs WITH neighbor  
        END FUNCTION  
        
        CALL dfs WITH 0  
        
        RETURN LENGTH OF visited IS EQUAL TO n  
    END FUNCTION  
END CLASS
