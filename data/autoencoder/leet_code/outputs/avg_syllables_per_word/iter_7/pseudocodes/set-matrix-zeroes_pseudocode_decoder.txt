CLASS Solution  
    FUNCTION setZeroes(matrix)  
        IF matrix IS EMPTY OR matrix[0] IS EMPTY THEN  
            RETURN  
        END IF  

        SET m TO LENGTH OF matrix  
        SET n TO LENGTH OF matrix[0]  

        SET first_row_has_zero TO false  
        FOR j FROM 0 TO n MINUS 1  
            IF matrix[0][j] EQUALS 0 THEN  
                SET first_row_has_zero TO true  
                BREAK  
            END IF  
        END FOR  

        SET first_col_has_zero TO false  
        FOR i FROM 0 TO m MINUS 1  
            IF matrix[i][0] EQUALS 0 THEN  
                SET first_col_has_zero TO true  
                BREAK  
            END IF  
        END FOR  

        FOR i FROM 1 TO m MINUS 1  
            FOR j FROM 1 TO n MINUS 1  
                IF matrix[i][j] EQUALS 0 THEN  
                    SET matrix[i][0] TO 0  
                    SET matrix[0][j] TO 0  
                END IF  
            END FOR  
        END FOR  

        FOR i FROM 1 TO m MINUS 1  
            FOR j FROM 1 TO n MINUS 1  
                IF matrix[i][0] EQUALS 0 OR matrix[0][j] EQUALS 0 THEN  
                    SET matrix[i][j] TO 0  
                END IF  
            END FOR  
        END FOR  

        IF first_row_has_zero THEN  
            FOR j FROM 0 TO n MINUS 1  
                SET matrix[0][j] TO 0  
            END FOR  
        END IF  

        IF first_col_has_zero THEN  
            FOR i FROM 0 TO m MINUS 1  
                SET matrix[i][0] TO 0  
            END FOR  
        END IF  
    END FUNCTION  
END CLASS
