CLASS TreeNode  
    FUNCTION __init__(self, val DEFAULT TO 0, left DEFAULT TO None, right DEFAULT TO None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode with val SET TO first element of values  
    SET i TO 1  
    SET queue TO new deque  
    APPEND root TO queue  
    WHILE queue IS not empty  
        SET node TO queue popped from front  
        IF i LESS THAN length of values AND values at i IS NOT None  
            SET node.left TO new TreeNode with val SET TO values at i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN length of values AND values at i IS NOT None  
            SET node.right TO new TreeNode with val SET TO values at i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION insertIntoBST(self, root, val)  
        FUNCTION dfs(root)  
            IF root IS None  
                RETURN new TreeNode with val SET TO val  
            IF root.val LESS THAN val  
                SET root.right TO dfs(root.right)  
            ELSE  
                SET root.left TO dfs(root.left)  
            RETURN root  
        END FUNCTION  
        RETURN dfs(root)  
    END FUNCTION  
END CLASS
