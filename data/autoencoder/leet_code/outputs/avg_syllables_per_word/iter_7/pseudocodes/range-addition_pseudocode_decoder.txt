CLASS Solution  
    FUNCTION getModifiedArray(length, updates)  
        SET diff TO list of zeros with size length PLUS 1  
        FOR each startIdx, endIdx, inc IN updates  
            INCREMENT diff at startIdx BY inc  
            IF endIdx PLUS 1 IS LESS THAN length  
                DECREMENT diff at endIdx PLUS 1 BY inc  
        SET arr TO list of zeros with size length  
        SET arr at 0 TO diff at 0  
        FOR i FROM 1 TO length MINUS 1  
            SET arr at i TO arr at i MINUS 1 PLUS diff at i  
        RETURN arr  
    END FUNCTION  
END CLASS
