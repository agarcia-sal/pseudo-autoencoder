CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF sum of numbers from 1 to maxChoosableInteger is less than desiredTotal THEN  
            RETURN False  
        END IF  
        
        IF desiredTotal is less than or equal to 0 THEN  
            RETURN True  
        END IF  
        
        SET memo TO an empty dictionary  
        
        FUNCTION can_win(current_total, used_numbers)  
            IF the pair (current_total, used_numbers) EXISTS IN memo THEN  
                RETURN the value stored in memo for (current_total, used_numbers)  
            END IF  
            
            FOR number FROM 1 TO maxChoosableInteger DO  
                IF number IS NOT used in used_numbers THEN  
                    IF current_total plus number IS greater than or equal to desiredTotal  
                        OR can_win(current_total plus number, used_numbers including number) RETURNS False THEN  
                        SET memo[(current_total, used_numbers)] TO True  
                        RETURN True  
                    END IF  
                END IF  
            END FOR  
            
            SET memo[(current_total, used_numbers)] TO False  
            RETURN False  
        END FUNCTION  
        
        RETURN can_win(0, 0)  
    END FUNCTION  
END CLASS
