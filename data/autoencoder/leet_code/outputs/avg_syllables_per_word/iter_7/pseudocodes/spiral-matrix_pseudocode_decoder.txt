CLASS Solution  
    FUNCTION spiralOrder(matrix)  
        IF matrix IS EMPTY OR matrix AT INDEX 0 IS EMPTY  
            RETURN EMPTY LIST  
        END IF  

        SET result TO EMPTY LIST  
        SET top TO 0  
        SET bottom TO LENGTH OF matrix MINUS 1  
        SET left TO 0  
        SET right TO LENGTH OF matrix AT INDEX 0 MINUS 1  

        WHILE top IS LESS THAN OR EQUAL TO bottom AND left IS LESS THAN OR EQUAL TO right  
            FOR i FROM left TO right  
                APPEND matrix AT INDEX top AND i TO result  
            END FOR  
            INCREMENT top BY 1  

            FOR i FROM top TO bottom  
                APPEND matrix AT INDEX i AND right TO result  
            END FOR  
            DECREMENT right BY 1  

            IF top IS LESS THAN OR EQUAL TO bottom  
                FOR i FROM right DOWN TO left  
                    APPEND matrix AT INDEX bottom AND i TO result  
                END FOR  
                DECREMENT bottom BY 1  
            END IF  

            IF left IS LESS THAN OR EQUAL TO right  
                FOR i FROM bottom DOWN TO top  
                    APPEND matrix AT INDEX i AND left TO result  
                END FOR  
                INCREMENT left BY 1  
            END IF  
        END WHILE  

        RETURN result  
    END FUNCTION  
END CLASS
