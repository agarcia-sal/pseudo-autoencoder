CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode with val as values at first position  
    SET i TO 1  
    SET queue TO empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO REMOVE first element from queue  
        IF i IS LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.left TO new TreeNode with val as values at i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.right TO new TreeNode with val as values at i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers(root)  
        FUNCTION dfs(node, current_number)  
            IF node IS None  
                RETURN 0  
            SET current_number TO current_number MULTIPLIED BY 10 PLUS node.val  
            IF node.left IS None AND node.right IS None  
                RETURN current_number  
            SET left_sum TO dfs(node.left, current_number)  
            SET right_sum TO dfs(node.right, current_number)  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN dfs(root, 0)  
    END FUNCTION  
END CLASS
