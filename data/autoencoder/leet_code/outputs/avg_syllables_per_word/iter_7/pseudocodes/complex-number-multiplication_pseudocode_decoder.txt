CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO result of splitting num by plus sign  
            SET real TO integer conversion of first part in parts  
            SET imaginary TO integer conversion of second part in parts without its last character  
            RETURN real, imaginary  
        END FUNCTION  
        
        SET real1, imag1 TO parse_complex(num1)  
        SET real2, imag2 TO parse_complex(num2)  
        
        SET real_part TO real1 multiplied BY real2 MINUS imag1 multiplied BY imag2  
        SET imaginary_part TO real1 multiplied BY imag2 PLUS imag1 multiplied BY real2  
        
        RETURN string formed by concatenating real_part, plus sign, imaginary_part, and letter "i"  
    END FUNCTION  
END CLASS
