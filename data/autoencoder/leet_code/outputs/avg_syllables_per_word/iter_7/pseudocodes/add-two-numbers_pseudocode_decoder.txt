CLASS ListNode  
    FUNCTION __init__(self, val = 0, next = None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    SET head TO new ListNode with values at position zero  
    SET p TO head  
    FOR each val IN values starting from second element  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(l1, l2)  
        SET dummy_head TO new ListNode with zero  
        SET current TO dummy_head  
        SET carry TO zero  
        
        WHILE l1 EXISTS OR l2 EXISTS OR carry IS NOT zero  
            SET val1 TO l1.val IF l1 EXISTS OTHERWISE zero  
            SET val2 TO l2.val IF l2 EXISTS OTHERWISE zero  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET current.next TO new ListNode with total MODULO ten  
            SET current TO current.next  
            
            IF l1 EXISTS  
                SET l1 TO l1.next  
            END IF  
            IF l2 EXISTS  
                SET l2 TO l2.next  
            END IF  
        END WHILE  
        
        RETURN dummy_head.next  
    END FUNCTION  
END CLASS
