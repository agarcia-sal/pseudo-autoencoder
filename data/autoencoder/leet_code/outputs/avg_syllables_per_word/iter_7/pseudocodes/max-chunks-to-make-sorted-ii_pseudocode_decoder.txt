CLASS Solution  
    FUNCTION maxChunksToSorted(arr)  
        SET max_so_far TO empty list  
        SET current_max TO first element of arr  
        FOR each num IN arr  
            SET current_max TO maximum of current_max AND num  
            APPEND current_max TO max_so_far  
        SET sorted_arr TO sorted version of arr  
        SET chunks TO 0  
        SET max_chunk TO 0  
        FOR index FROM 0 TO LENGTH OF arr MINUS 1  
            SET max_chunk TO maximum of max_chunk AND element at index of arr  
            IF max_chunk EQUALS element at index of sorted_arr  
                INCREMENT chunks BY 1  
        RETURN chunks  
    END FUNCTION  
END CLASS
