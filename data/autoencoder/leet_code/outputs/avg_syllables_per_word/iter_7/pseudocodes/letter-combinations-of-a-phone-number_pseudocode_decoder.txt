CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits IS EMPTY  
            RETURN empty list  
        
        SET phone_map TO mapping from digit characters TO corresponding string of letters  
        
        FUNCTION backtrack(combination, next_digits)  
            IF LENGTH OF next_digits EQUALS 0  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN phone_map VALUE FOR next_digits FIRST CHARACTER  
                    CALL backtrack WITH combination APPENDED WITH letter AND next_digits WITHOUT FIRST CHARACTER  
        
        SET output TO empty list  
        CALL backtrack WITH empty string AND digits  
        RETURN output  
    END FUNCTION  
END CLASS
