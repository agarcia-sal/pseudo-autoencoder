CLASS Solution  
    FUNCTION threeSumClosest(nums, target)  
        CALL sort ON nums  
        SET closest_sum TO infinity  
        
        FOR i FROM 0 TO LENGTH OF nums MINUS 2  
            SET left TO i PLUS 1  
            SET right TO LENGTH OF nums MINUS 1  
            
            WHILE left IS LESS THAN right  
                SET current_sum TO nums AT i PLUS nums AT left PLUS nums AT right  
                
                IF absolute value of (current_sum MINUS target) IS LESS THAN absolute value of (closest_sum MINUS target)  
                    SET closest_sum TO current_sum  
                END IF  
                
                IF current_sum IS LESS THAN target  
                    INCREMENT left BY 1  
                ELSE IF current_sum IS GREATER THAN target  
                    DECREMENT right BY 1  
                ELSE  
                    RETURN current_sum  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN closest_sum  
    END FUNCTION  
END CLASS
