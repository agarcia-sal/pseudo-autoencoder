CLASS Solution  
    FUNCTION findKthNumber(n, k)  
        FUNCTION count_steps(prefix, n)  
            SET next_prefix TO prefix PLUS 1  
            SET steps TO 0  
            WHILE prefix LESS THAN OR EQUAL TO n  
                SET steps TO steps PLUS MINIMUM OF (n MINUS prefix PLUS 1) AND (next_prefix MINUS prefix)  
                SET prefix TO prefix MULTIPLIED BY 10  
                SET next_prefix TO next_prefix MULTIPLIED BY 10  
            RETURN steps  
        END FUNCTION  

        SET current TO 1  
        SET k TO k MINUS 1  
        WHILE k GREATER THAN 0  
            SET steps TO count_steps(current, n)  
            IF steps LESS THAN OR EQUAL TO k  
                SET current TO current PLUS 1  
                SET k TO k MINUS steps  
            ELSE  
                SET current TO current MULTIPLIED BY 10  
                SET k TO k MINUS 1  
        RETURN current  
    END FUNCTION  
END CLASS
