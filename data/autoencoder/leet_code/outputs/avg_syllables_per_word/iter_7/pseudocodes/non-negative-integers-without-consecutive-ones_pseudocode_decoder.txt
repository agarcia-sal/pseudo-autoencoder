CLASS Solution  
    FUNCTION findIntegers(n)  
        SET bin_rep TO binary representation of n as string excluding leading '0b'  
        SET length TO length of bin_rep  
        
        SET dp TO list of zeros with size length plus 1  
        SET dp at position 0 TO 1  
        SET dp at position 1 TO 2  
        
        FOR i FROM 2 TO length INCLUSIVE  
            SET dp at position i TO dp at position i minus 1 PLUS dp at position i minus 2  
        
        SET result TO 0  
        SET prev_bit TO 0  
        
        FOR i FROM 0 TO length MINUS 1  
            IF character at position i in bin_rep EQUALS '1' THEN  
                INCREMENT result BY dp at position length MINUS i MINUS 1  
                IF prev_bit EQUALS 1 THEN  
                    DECREMENT result BY 1  
                    BREAK loop  
                END IF  
            END IF  
            SET prev_bit TO 1 IF character at position i in bin_rep EQUALS '1' ELSE 0  
        END FOR  
        
        RETURN result PLUS 1  
    END FUNCTION  
END CLASS
