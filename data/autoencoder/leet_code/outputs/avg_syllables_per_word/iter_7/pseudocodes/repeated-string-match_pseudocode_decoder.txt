CLASS Solution  
    FUNCTION repeatedStringMatch(a, b)  
        SET min_repeats TO length of b DIVIDED BY length of a rounded up to nearest integer  
        IF b is substring of a repeated min_repeats times  
            RETURN min_repeats  
        ELSE IF b is substring of a repeated min_repeats plus one times  
            RETURN min_repeats plus one  
        ELSE  
            RETURN negative one  
    END FUNCTION  
END CLASS
