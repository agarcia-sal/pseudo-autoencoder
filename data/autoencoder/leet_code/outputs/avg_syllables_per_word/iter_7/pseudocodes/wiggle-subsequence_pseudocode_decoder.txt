CLASS Solution  
    FUNCTION wiggleMaxLength(nums)  
        IF LENGTH OF nums LESS THAN 2 THEN  
            RETURN LENGTH OF nums  
        END IF  
        
        SET up TO 1  
        SET down TO 1  
        
        FOR i FROM 1 TO LENGTH OF nums MINUS 1  
            IF nums AT i GREATER THAN nums AT i MINUS 1 THEN  
                SET up TO down PLUS 1  
            ELSE IF nums AT i LESS THAN nums AT i MINUS 1 THEN  
                SET down TO up PLUS 1  
            END IF  
        END FOR  
        
        RETURN THE GREATER OF up AND down  
    END FUNCTION  
END CLASS
