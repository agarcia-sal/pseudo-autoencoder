CLASS Solution  
    FUNCTION maxAreaOfIsland(grid)  
        IF grid IS empty  
            RETURN 0  
        
        SET max_area TO 0  
        SET rows TO LENGTH OF grid  
        SET cols TO LENGTH OF grid AT 0  
        
        FUNCTION dfs(r, c)  
            IF r IS less than 0 OR r IS greater than or equal to rows OR c IS less than 0 OR c IS greater than or equal to cols OR grid AT r AT c IS equal to 0  
                RETURN 0  
            SET grid AT r AT c TO 0  
            SET area TO 1  
            SET area TO area PLUS dfs(r PLUS 1, c)  
            SET area TO area PLUS dfs(r MINUS 1, c)  
            SET area TO area PLUS dfs(r, c PLUS 1)  
            SET area TO area PLUS dfs(r, c MINUS 1)  
            RETURN area  
        END FUNCTION  
        
        FOR r FROM 0 TO rows MINUS 1  
            FOR c FROM 0 TO cols MINUS 1  
                IF grid AT r AT c IS equal to 1  
                    SET max_area TO MAXIMUM OF max_area AND dfs(r, c)  
        
        RETURN max_area  
    END FUNCTION  
END CLASS
