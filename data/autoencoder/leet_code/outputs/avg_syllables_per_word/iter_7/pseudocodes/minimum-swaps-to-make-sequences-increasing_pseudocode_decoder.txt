CLASS Solution  
    FUNCTION minSwap(nums1, nums2)  
        SET n TO LENGTH OF nums1  
        INITIALIZE dp AS LIST OF n ELEMENTS EACH CONTAINING LIST [0, 0]  
          
        SET dp[0][0] TO 0  
        SET dp[0][1] TO 1  
          
        FOR i FROM 1 TO n MINUS 1  
            IF nums1[i] GREATER THAN nums1[i MINUS 1] AND nums2[i] GREATER THAN nums2[i MINUS 1] THEN  
                IF nums1[i] GREATER THAN nums2[i MINUS 1] AND nums2[i] GREATER THAN nums1[i MINUS 1] THEN  
                    SET dp[i][0] TO MINIMUM OF dp[i MINUS 1][0] AND dp[i MINUS 1][1]  
                    SET dp[i][1] TO dp[i][0] PLUS 1  
                ELSE  
                    SET dp[i][0] TO dp[i MINUS 1][0]  
                    SET dp[i][1] TO dp[i MINUS 1][1] PLUS 1  
            ELSE  
                IF nums1[i] GREATER THAN nums2[i MINUS 1] AND nums2[i] GREATER THAN nums1[i MINUS 1] THEN  
                    SET dp[i][0] TO dp[i MINUS 1][1]  
                    SET dp[i][1] TO dp[i MINUS 1][0] PLUS 1  
                ELSE  
                    SET dp[i][0] TO dp[i MINUS 1][1]  
                    SET dp[i][1] TO dp[i MINUS 1][0] PLUS 1  
        RETURN MINIMUM OF dp[n MINUS 1][0] AND dp[n MINUS 1][1]  
    END FUNCTION  
END CLASS
