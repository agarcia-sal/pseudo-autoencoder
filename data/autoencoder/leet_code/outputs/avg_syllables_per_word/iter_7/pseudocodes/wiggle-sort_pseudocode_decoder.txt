CLASS Solution  
    FUNCTION wiggleSort(nums)  
        SET less TO true  
        FOR i FROM 0 TO LENGTH OF nums MINUS 2  
            IF less  
                IF nums AT i GREATER THAN nums AT i PLUS 1  
                    SWAP nums AT i WITH nums AT i PLUS 1  
            ELSE  
                IF nums AT i LESS THAN nums AT i PLUS 1  
                    SWAP nums AT i WITH nums AT i PLUS 1  
            SET less TO NOT less  
    END FUNCTION  
END CLASS
