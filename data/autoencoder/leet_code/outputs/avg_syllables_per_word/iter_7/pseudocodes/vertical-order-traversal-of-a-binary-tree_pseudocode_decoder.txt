CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode with val as first element of values  
    SET i TO 1  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from left of queue  
        IF i IS LESS THAN length of values AND values[i] IS NOT None  
            SET node.left TO new TreeNode with val as values[i]  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN length of values AND values[i] IS NOT None  
            SET node.right TO new TreeNode with val as values[i]  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalTraversal(self, root)  
        IF root IS None  
            RETURN empty list  
        SET node_dict TO default dictionary with list as default value  
        SET min_col TO 0  
        SET max_col TO 0  

        FUNCTION dfs(node, row, col)  
            IF node IS None  
                RETURN  
            APPEND pair of row and node.val TO node_dict[col]  
            SET min_col TO the smaller of min_col and col  
            SET max_col TO the larger of max_col and col  
            CALL dfs with node.left, row incremented by 1, col decremented by 1  
            CALL dfs with node.right, row incremented by 1, col incremented by 1  
        END FUNCTION  

        CALL dfs with root, row 0, col 0  

        SET result TO empty list  
        FOR col FROM min_col TO max_col inclusive  
            SET sorted_list TO node_dict[col] sorted first by row then by node value  
            SET column_values TO list of values extracted from sorted_list  
            APPEND column_values TO result  
        RETURN result  
    END FUNCTION  
END CLASS
