CLASS Solution  
    FUNCTION numDistinct(s, t)  
        SET m TO LENGTH OF s  
        SET n TO LENGTH OF t  
        
        SET dp TO 2D array of size (m plus 1) by (n plus 1) initialized with zeroes  
        
        FOR i FROM 0 TO m  
            SET dp at i and 0 TO 1  
        
        FOR i FROM 1 TO m  
            FOR j FROM 1 TO n  
                IF character at position (i minus 1) in s EQUALS character at position (j minus 1) in t  
                    SET dp at i and j TO dp at (i minus 1) and (j minus 1) PLUS dp at (i minus 1) and j  
                ELSE  
                    SET dp at i and j TO dp at (i minus 1) and j  
        
        RETURN dp at m and n  
    END FUNCTION  
END CLASS
