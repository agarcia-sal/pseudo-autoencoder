CLASS Solution  
    FUNCTION reachingPoints(sx, sy, tx, ty)  
        WHILE tx GREATER THAN OR EQUAL TO sx AND ty GREATER THAN OR EQUAL TO sy  
            IF tx EQUAL TO sx AND ty EQUAL TO sy  
                RETURN True  
            END IF  
            IF tx GREATER THAN ty  
                IF ty GREATER THAN sy  
                    SET tx TO tx MODULO ty  
                ELSE  
                    RETURN (tx MINUS sx) MODULO ty EQUAL TO 0  
                END IF  
            ELSE  
                IF tx GREATER THAN sx  
                    SET ty TO ty MODULO tx  
                ELSE  
                    RETURN (ty MINUS sy) MODULO tx EQUAL TO 0  
                END IF  
            END IF  
        END WHILE  
        RETURN False  
    END FUNCTION  
END CLASS
