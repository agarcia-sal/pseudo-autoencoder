CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EMPTY  
        RETURN None  
    SET root TO new TreeNode with values at position zero  
    SET i TO 1  
    SET queue TO new empty queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO REMOVE FIRST ELEMENT FROM queue  
        IF i IS LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.left TO new TreeNode with values at i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i IS LESS THAN LENGTH OF values AND values at i IS NOT None  
            SET node.right TO new TreeNode with values at i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION tree2str(self, root)  
        FUNCTION helper(node)  
            IF node IS None  
                RETURN empty string  
            SET result TO string representation of node.val  
            IF node.left IS NOT None OR node.right IS NOT None  
                SET result TO result CONCATENATED WITH "(" CONCATENATED WITH helper(node.left) CONCATENATED WITH ")"  
            IF node.right IS NOT None  
                SET result TO result CONCATENATED WITH "(" CONCATENATED WITH helper(node.right) CONCATENATED WITH ")"  
            RETURN result  
        END FUNCTION  
        RETURN helper(root)  
    END FUNCTION  
END CLASS
