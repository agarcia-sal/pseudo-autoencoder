CLASS Solution  
    FUNCTION stoneGameVIII(stones)  
        SET prefix_sum TO list of zeros with length equal to length of stones  
        SET prefix_sum at index 0 TO stones at index 0  
        FOR i FROM 1 TO length of stones MINUS 1  
            SET prefix_sum at index i TO prefix_sum at index i MINUS 1 PLUS stones at index i  
        SET dp TO last element of prefix_sum  
        FOR i FROM length of stones MINUS 2 DOWNTO 1  
            SET dp TO maximum of dp AND prefix_sum at index i MINUS dp  
        RETURN dp  
    END FUNCTION  
END CLASS
