CLASS Solution
    FUNCTION numberOfWays(numPeople)
        SET MODULO TO ten RAISED TO THE power of nine PLUS seven
        SET dp_list TO A NEW list OF zero OF SIZE numPeople PLUS one
        SET element at position zero OF dp_list TO one
        SET element at position two OF dp_list TO one
        FOR index i FROM four TO numPeople IN STEPS OF two
            FOR index j FROM zero TO i MINUS one IN STEPS OF two
                SET element at position i OF dp_list TO element at position i OF dp_list PLUS element at position j OF dp_list MULTIPLIED BY element at position i MINUS two MINUS j OF dp_list MODULO MODULO
            END FOR
        END FOR
        RETURN element at position numPeople OF dp_list
    END FUNCTION
END CLASS
