CLASS Solution
    FUNCTION reachingPoints(sx INTEGER, sy INTEGER, tx INTEGER, ty INTEGER) RETURNS BOOLEAN
        WHILE tx IS GREATER THAN OR EQUAL TO sx AND ty IS GREATER THAN OR EQUAL TO sy
            IF tx EQUALS sx AND ty EQUALS sy
                RETURN True
            END IF
            IF tx IS GREATER THAN ty
                IF ty IS GREATER THAN sy
                    SET tx TO the REMAINDER OF tx DIVIDED BY ty
                ELSE
                    RETURN the REMAINDER OF (tx MINUS sx) DIVIDED BY ty EQUALS zero
                END IF
            ELSE
                IF tx IS GREATER THAN sx
                    SET ty TO the REMAINDER OF ty DIVIDED BY tx
                ELSE
                    RETURN the REMAINDER OF (ty MINUS sy) DIVIDED BY tx EQUALS zero
                END IF
            END IF
        END WHILE
        RETURN False
    END FUNCTION
END CLASS
