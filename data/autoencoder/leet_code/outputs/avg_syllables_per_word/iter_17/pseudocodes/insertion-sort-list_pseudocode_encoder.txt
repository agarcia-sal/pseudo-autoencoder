CLASS ListNode
    FUNCTION __init__(val EQUALS zero, next EQUALS None)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values)
    IF the collection values IS empty
        RETURN None
    END IF
    SET head TO CALL ListNode WITH the element at position zero of values
    SET pointer TO head
    FOR each value IN the collection from position one TO the end of values
        SET node TO CALL ListNode WITH value
        SET the element at position next of pointer TO node
        SET pointer TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS None AND p2 IS None
        RETURN True
    END IF
    IF p1 IS None OR p2 IS None
        RETURN False
    END IF
    RETURN the element at position val of p1 EQUALS the element at position val of p2 AND CALL is_same_list WITH the element at position next of p1 AND the element at position next of p2
END FUNCTION

CLASS Solution
    FUNCTION insertionSortList(head)
        IF head IS None OR the element at position next of head IS None
            RETURN head
        END IF
        SET dummy TO CALL ListNode WITH negative five thousand and one
        SET the element at position next of dummy TO head
        SET current TO the element at position next of head
        SET the element at position next of head TO None

        WHILE current IS NOT None
            SET next_node TO the element at position next of current
            SET prev TO dummy
            WHILE the element at position next of prev IS NOT None AND the element at position val of the element at position next of prev IS LESS THAN the element at position val of current
                SET prev TO the element at position next of prev
            END WHILE
            SET the element at position next of current TO the element at position next of prev
            SET the element at position next of prev TO current
            SET current TO next_node
        END WHILE

        RETURN the element at position next of dummy
    END FUNCTION
END CLASS
