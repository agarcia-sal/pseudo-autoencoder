CLASS Solution
    FUNCTION searchRange(list_of_numbers, target_value)
        FUNCTION find_left(list_of_numbers, target_value)
            SET left_pointer TO zero
            SET right_pointer TO the LENGTH OF list_of_numbers MINUS one
            WHILE left_pointer LESS THAN OR EQUAL TO right_pointer holds
                SET middle_index TO the INTEGER DIVISION OF left_pointer PLUS right_pointer BY two
                IF element at position middle_index of list_of_numbers LESS THAN target_value
                    SET left_pointer TO middle_index PLUS one
                ELSE
                    SET right_pointer TO middle_index MINUS one
                END IF
            END WHILE
            RETURN left_pointer
        END FUNCTION

        FUNCTION find_right(list_of_numbers, target_value)
            SET left_pointer TO zero
            SET right_pointer TO the LENGTH OF list_of_numbers MINUS one
            WHILE left_pointer LESS THAN OR EQUAL TO right_pointer holds
                SET middle_index TO the INTEGER DIVISION OF left_pointer PLUS right_pointer BY two
                IF element at position middle_index of list_of_numbers LESS THAN OR EQUAL TO target_value
                    SET left_pointer TO middle_index PLUS one
                ELSE
                    SET right_pointer TO middle_index MINUS one
                END IF
            END WHILE
            RETURN right_pointer
        END FUNCTION

        SET left_index TO CALL find_left WITH list_of_numbers AND target_value
        SET right_index TO CALL find_right WITH list_of_numbers AND target_value

        IF left_index LESS THAN OR EQUAL TO right_index AND zero LESS THAN OR EQUAL TO right_index AND right_index LESS THAN the LENGTH OF list_of_numbers AND element at position left_index of list_of_numbers EQUALS target_value
            RETURN list containing left_index AND right_index
        ELSE
            RETURN list containing negative one AND negative one
        END IF
    END FUNCTION
END CLASS
