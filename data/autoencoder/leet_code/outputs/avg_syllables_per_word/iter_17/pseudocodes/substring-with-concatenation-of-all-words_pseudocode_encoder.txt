CLASS Solution
    FUNCTION findSubstring(s string, words list_of_strings) RETURNS list_of_integers
        IF s IS empty OR words IS empty OR the element at position zero of words IS empty
            RETURN an empty list
        END IF

        SET word_length TO the LENGTH OF the element at position zero of words
        SET num_words TO the LENGTH OF words
        SET total_length TO word_length MULTIPLIED BY num_words
        SET word_count TO CALL Counter WITH words
        SET result TO an empty list

        FOR index FROM zero TO the LENGTH OF s MINUS total_length PLUS one
            SET seen TO CALL Counter WITH words as empty counter
            FOR position FROM index TO index PLUS total_length MINUS one STEP word_length
                SET word TO the substring from position position TO position PLUS word_length MINUS one of s
                INCREMENT the COUNT of word in seen BY one
                IF the COUNT of word in seen IS GREATER THAN the COUNT of word in word_count
                    BREAK out of the inner loop
                END IF
            END FOR
            IF the inner loop was not broken
                APPEND index TO result
            END IF
        END FOR

        RETURN result
    END FUNCTION
END CLASS
