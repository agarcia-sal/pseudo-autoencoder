CLASS Solution
    FUNCTION removeComments(list_of_strings source)
        SET in_block_comment TO false
        SET list_of_characters result TO empty list
        SET list_of_characters current_line TO empty list
        FOR each string line IN source
            SET index i TO zero
            WHILE index i LESS THAN the LENGTH OF line
                IF in_block_comment
                    IF substring from position i TO position i PLUS one of line EQUALS the sequence of characters ASTERISK SLASH
                        SET in_block_comment TO false
                        INCREMENT index i BY one
                    END IF
                ELSE
                    IF substring from position i TO position i PLUS one of line EQUALS the sequence of characters SLASH SLASH
                        BREAK the current WHILE loop
                    ELSE IF substring from position i TO position i PLUS one of line EQUALS the sequence of characters SLASH ASTERISK
                        SET in_block_comment TO true
                        INCREMENT index i BY one
                    ELSE
                        APPEND element at position i of line TO current_line
                    END IF
                END IF
                INCREMENT index i BY one
            END WHILE
            IF current_line is not empty AND in_block_comment EQUALS false
                APPEND the concatenation of characters in current_line TO result
                SET current_line TO empty list
            END IF
        END FOR
        RETURN result
    END FUNCTION
END CLASS
