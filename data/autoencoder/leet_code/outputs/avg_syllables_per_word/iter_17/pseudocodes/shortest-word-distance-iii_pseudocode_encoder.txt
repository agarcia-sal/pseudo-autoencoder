CLASS Solution
    FUNCTION shortestWordDistance(wordsDict AS list_of_strings, word1 AS string, word2 AS string) RETURNS integer
        SET min_distance TO positive_infinity
        SET prev_word1 TO negative_one
        SET prev_word2 TO negative_one
        SET same_word TO word1 EQUALS word2

        FOR each index FROM zero TO the LENGTH OF wordsDict MINUS one
            SET word TO element at position index of wordsDict

            IF word EQUALS word1
                IF same_word
                    SET prev_word2 TO prev_word1
                END IF
                SET prev_word1 TO index
            END IF

            IF word EQUALS word2
                IF NOT same_word
                    SET prev_word2 TO index
                END IF
            END IF

            IF prev_word1 NOT EQUALS negative_one AND prev_word2 NOT EQUALS negative_one
                SET min_distance TO the MINIMUM OF min_distance AND the ABSOLUTE VALUE OF prev_word1 MINUS prev_word2
            END IF
        END FOR

        RETURN min_distance
    END FUNCTION
END CLASS
