CLASS Solution
    FUNCTION minimumTime(n, relations, time)
        ASSIGN graph TO CALL create_default_dict_of_lists WITH no parameters
        FOR each pair previous_course AND next_course IN relations
            APPEND previous_course TO element at position next_course of graph
        END FOR

        ASSIGN memo TO CALL create_list_of_size_with_value WITH parameters n PLUS one AND negative one

        FUNCTION dfs(course)
            IF element at position course of memo IS NOT EQUAL TO negative one
                RETURN element at position course of memo
            END IF

            SET max_prerequisite_time TO zero
            FOR each previous_course IN element at position course of graph
                SET candidate_time TO CALL dfs WITH previous_course
                IF candidate_time IS GREATER THAN max_prerequisite_time
                    SET max_prerequisite_time TO candidate_time
                END IF
            END FOR

            SET element at position course of memo TO max_prerequisite_time PLUS element at position course MINUS one of time
            RETURN element at position course of memo
        END FUNCTION

        SET total_minimum_time TO zero
        FOR course FROM one TO n
            SET current_time TO CALL dfs WITH course
            IF current_time IS GREATER THAN total_minimum_time
                SET total_minimum_time TO current_time
            END IF
        END FOR

        RETURN total_minimum_time
    END FUNCTION
END CLASS
