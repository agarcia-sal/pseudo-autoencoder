CLASS Solution
    FUNCTION distanceToCycle(n, edges)
        SET graph TO CALL createDefaultDictOfLists WITH no parameters
        FOR each pair node_u AND node_v IN edges
            CALL APPEND node_v TO element at position node_u of graph
            CALL APPEND node_u TO element at position node_v of graph
        END FOR

        FUNCTION dfs(node, parent, path, visited)
            IF node IS IN visited
                SET cycle_start_position TO CALL indexOf node IN path
                RETURN substring of path FROM position cycle_start_position TO end
            END IF
            CALL ADD node TO visited
            CALL APPEND node TO path
            FOR each neighbor IN element at position node of graph
                IF neighbor IS NOT EQUAL TO parent
                    SET cycle_result TO CALL dfs WITH parameters neighbor, node, path, visited
                    IF cycle_result IS NOT NONE
                        RETURN cycle_result
                    END IF
                END IF
            END FOR
            CALL REMOVE last element FROM path
            RETURN NONE
        END FUNCTION

        SET visited TO empty set
        SET cycle TO NONE
        FOR node FROM zero TO n MINUS one
            IF node IS NOT IN visited
                SET cycle TO CALL dfs WITH parameters node, minus one, empty list, visited
                IF cycle IS NOT NONE
                    BREAK loop
                END IF
            END IF
        END FOR

        SET distance TO list of zero repeated n times
        SET queue TO CALL createDeque WITH parameter cycle
        SET visited TO COPY of set cycle

        SET level TO zero
        WHILE queue IS NOT empty
            SET current_queue_length TO CALL lengthOf queue
            FOR iterator FROM zero TO current_queue_length MINUS one
                SET current_node TO CALL removeFirstElement FROM queue
                SET element at position current_node of distance TO level
                FOR each neighbor IN element at position current_node of graph
                    IF neighbor IS NOT IN visited
                        CALL ADD neighbor TO visited
                        CALL APPEND neighbor TO queue
                    END IF
                END FOR
            END FOR
            INCREMENT level BY one
        END WHILE

        RETURN distance
    END FUNCTION
END CLASS
