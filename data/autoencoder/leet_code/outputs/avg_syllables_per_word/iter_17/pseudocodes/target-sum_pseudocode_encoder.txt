CLASS Solution
    FUNCTION findTargetSumWays(list_of_numbers, target_value)
        FUNCTION dfs(current_index, current_sum_value, memo_dictionary)
            IF current_index IS EQUAL TO the LENGTH OF list_of_numbers
                IF current_sum_value IS EQUAL TO target_value
                    RETURN one
                ELSE
                    RETURN zero
                END IF
            END IF
            IF the PAIR current_index AND current_sum_value IS IN memo_dictionary
                RETURN the ELEMENT at PAIR current_index AND current_sum_value OF memo_dictionary
            END IF

            SET positive_result TO CALL dfs WITH current_index INCREMENTED BY one AND current_sum_value INCREMENTED BY element at position current_index OF list_of_numbers AND memo_dictionary
            SET negative_result TO CALL dfs WITH current_index INCREMENTED BY one AND current_sum_value DECREMENTED BY element at position current_index OF list_of_numbers AND memo_dictionary

            SET the ELEMENT at PAIR current_index AND current_sum_value OF memo_dictionary TO positive_result PLUS negative_result
            RETURN the ELEMENT at PAIR current_index AND current_sum_value OF memo_dictionary
        END FUNCTION

        RETURN CALL dfs WITH zero AND zero AND EMPTY DICTIONARY
    END FUNCTION
END CLASS
