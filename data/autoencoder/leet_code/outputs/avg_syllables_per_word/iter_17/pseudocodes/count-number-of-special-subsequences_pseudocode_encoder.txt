CLASS Solution
    FUNCTION countSpecialSubsequences(nums)
        SET MOD TO one billion PLUS seven
        SET dp TO list of three zeroes
        
        FOR each number IN nums
            IF number EQUALS zero
                SET element at position zero of dp TO open parenthesis two MULTIPLIED BY element at position zero of dp CLOSE parenthesis PLUS one MODULO MOD
            ELSE IF number EQUALS one
                SET element at position one of dp TO open parenthesis two MULTIPLIED BY element at position one of dp CLOSE parenthesis PLUS element at position zero of dp MODULO MOD
            ELSE IF number EQUALS two
                SET element at position two of dp TO open parenthesis two MULTIPLIED BY element at position two of dp CLOSE parenthesis PLUS element at position one of dp MODULO MOD
            END IF
        END FOR
        
        RETURN element at position two of dp
    END FUNCTION
END CLASS
