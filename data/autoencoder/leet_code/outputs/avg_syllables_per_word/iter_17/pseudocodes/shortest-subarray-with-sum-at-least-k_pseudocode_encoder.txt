CLASS Solution
    FUNCTION shortestSubarray(list_of_numbers, threshold_k)
        SET prefix_sum_list TO a new list containing zero
        FOR each number IN list_of_numbers
            APPEND the element at position the last index of prefix_sum_list PLUS number TO prefix_sum_list
        END FOR

        ASSIGN pointer dq TO a new empty double ended queue
        SET min_length_subarray TO positive infinity constant

        FOR each index i AND current_sum_value IN enumeration of prefix_sum_list
            WHILE dq is not empty AND current_sum_value MINUS element at position the element at position zero of dq of prefix_sum_list IS GREATER THAN OR EQUAL TO threshold_k
                SET min_length_subarray TO the smaller of min_length_subarray AND i MINUS REMOVE element from the left end of dq
            END WHILE

            WHILE dq is not empty AND current_sum_value IS LESS THAN OR EQUAL TO element at position the element at position the last index of dq of prefix_sum_list
                REMOVE element from the right end of dq
            END WHILE

            APPEND i TO dq
        END FOR

        IF min_length_subarray IS EQUAL TO positive infinity constant
            RETURN negative one
        ELSE
            RETURN min_length_subarray
        END IF
    END FUNCTION
END CLASS
