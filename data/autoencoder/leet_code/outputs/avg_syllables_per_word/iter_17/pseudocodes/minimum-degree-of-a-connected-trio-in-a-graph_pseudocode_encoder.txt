CLASS Solution
    FUNCTION minTrioDegree(n, edges)
        SET graph TO a MAPPING FROM each integer index FROM one TO n TO an EMPTY SET

        FOR each pair of nodes u and v IN edges
            APPEND element v TO the SET at key u OF graph
            APPEND element u TO the SET at key v OF graph
        END FOR

        SET min_degree TO positive infinity
        SET trio_found TO false

        FOR i FROM one TO n
            FOR j FROM i PLUS one TO n
                FOR k FROM j PLUS one TO n
                    IF element j IS IN the SET at key i OF graph AND element k IS IN the SET at key i OF graph AND element k IS IN the SET at key j OF graph
                        SET trio_found TO true
                        SET trio_degree TO the LENGTH OF the SET at key i OF graph PLUS the LENGTH OF the SET at key j OF graph PLUS the LENGTH OF the SET at key k OF graph MINUS six
                        SET min_degree TO the MINIMUM OF min_degree AND trio_degree
                    END IF
                END FOR
            END FOR
        END FOR

        IF trio_found IS true
            RETURN min_degree
        ELSE
            RETURN negative one
        END IF
    END FUNCTION
END CLASS
