CLASS Solution
    FUNCTION findMaxLength(list_of_numbers)
        SET prefix_sum_index TO a dictionary with key zero SET TO negative one
        SET max_length TO zero
        SET count TO zero

        FOR each index FROM zero TO the LENGTH OF list_of_numbers MINUS one
            SET number TO element at position index of list_of_numbers
            IF number EQUALS one
                INCREMENT count BY one
            ELSE
                DECREMENT count BY one
            END IF

            IF count IS IN prefix_sum_index
                SET max_length TO the GREATER OF max_length AND index MINUS element at key count of prefix_sum_index
            ELSE
                SET element at key count of prefix_sum_index TO index
            END IF
        END FOR

        RETURN max_length
    END FUNCTION
END CLASS
