CLASS Solution
    FUNCTION consecutiveNumbersSum(n)
        SET count TO zero
        FOR k FROM one TO n
            IF the product of two AND n MODULO k EQUALS zero
                SET m TO the quotient of the product of two AND n DIVIDED BY k MINUS k PLUS one DIVIDED BY two
                IF m GREATER THAN zero AND m EQUALS the INTEGER VALUE OF m
                    INCREMENT count BY one
                END IF
            END IF
        END FOR
        RETURN count
    END FUNCTION
END CLASS
