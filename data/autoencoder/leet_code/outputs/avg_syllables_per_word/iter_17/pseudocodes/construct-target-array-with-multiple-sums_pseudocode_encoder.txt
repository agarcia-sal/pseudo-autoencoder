CLASS Solution
    FUNCTION isPossible(target)
        SET max_heap TO the LIST OF negative element at position index OF target FOR each index IN the RANGE from zero TO the LENGTH OF target MINUS one
        CALL heapify WITH max_heap
        SET total_sum TO the SUM OF all elements IN target
        WHILE the element at position zero OF max_heap IS LESS THAN negative one
            CALL heappop WITH max_heap AND SET result TO popped_value
            SET largest TO negative popped_value
            SET rest_sum TO total_sum MINUS largest
            IF largest IS LESS THAN OR EQUAL TO rest_sum OR rest_sum EQUALS zero
                RETURN False
            END IF
            SET new_value TO largest MODULO rest_sum
            IF new_value EQUALS zero AND rest_sum IS NOT EQUAL TO one
                RETURN False
            END IF
            SET total_sum TO rest_sum PLUS new_value
            CALL heappush WITH max_heap AND argument negative new_value
        END WHILE
        RETURN True
    END FUNCTION
END CLASS
