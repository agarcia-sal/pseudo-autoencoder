CLASS Solution
    FUNCTION convert(string_s, integer_numRows)
        IF integer_numRows EQUALS one
            RETURN string_s
        END IF

        SET list_rows TO a list containing integer_numRows empty strings
        SET integer_current_row TO zero
        SET boolean_going_down TO false

        FOR each character IN string_s
            APPEND character TO string at position integer_current_row of list_rows
            IF integer_current_row EQUALS zero OR integer_current_row EQUALS integer_numRows MINUS one
                SET boolean_going_down TO NOT boolean_going_down
            END IF
            IF boolean_going_down IS true
                INCREMENT integer_current_row BY one
            ELSE
                DECREMENT integer_current_row BY one
            END IF
        END FOR

        CALL function JoinStrings WITH list_rows RETURNING string_result
        RETURN string_result
    END FUNCTION

FUNCTION JoinStrings(list_of_strings)
    DESCRIBE AS auxiliary function that combines list_of_strings into one string
END FUNCTION

END CLASS
