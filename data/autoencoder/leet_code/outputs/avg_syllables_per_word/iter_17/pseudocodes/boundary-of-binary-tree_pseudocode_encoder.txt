CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values AS list)
    IF values IS empty
        RETURN None
    END IF
    SET root TO CALL TreeNode WITH element at position zero of values
    SET i TO one
    SET queue TO CALL deque WITH no parameters
    CALL APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO CALL popleft ON queue
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.left TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.right TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUALS q.val
        RETURN False
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION boundaryOfBinaryTree(root AS Optional OF TreeNode) RETURNS list of integers
        IF root IS None
            RETURN an empty list
        END IF

        FUNCTION is_leaf(node)
            RETURN node.left IS None AND node.right IS None
        END FUNCTION

        SET left_boundary TO an empty list
        SET current TO root.left
        WHILE current IS NOT None
            IF NOT CALL is_leaf WITH current
                CALL APPEND element at position val of current TO left_boundary
            END IF
            IF current.left IS NOT None
                SET current TO current.left
            ELSE
                SET current TO current.right
            END IF
        END WHILE

        SET right_boundary TO an empty list
        SET current TO root.right
        WHILE current IS NOT None
            IF NOT CALL is_leaf WITH current
                CALL APPEND element at position val of current TO right_boundary
            END IF
            IF current.right IS NOT None
                SET current TO current.right
            ELSE
                SET current TO current.left
            END IF
        END WHILE

        SET leaves TO an empty list
        FUNCTION collect_leaves(node)
            IF node IS None
                RETURN
            END IF
            IF CALL is_leaf WITH node
                CALL APPEND element at position val of node TO leaves
            END IF
            CALL collect_leaves WITH node.left
            CALL collect_leaves WITH node.right
        END FUNCTION

        CALL collect_leaves WITH root

        IF CALL is_leaf WITH root
            SET leaves TO an empty list
        END IF

        RETURN list consisting of element at position val of root FOLLOWED BY left_boundary FOLLOWED BY leaves FOLLOWED BY right_boundary REVERSED in order
    END FUNCTION
END CLASS
