CLASS Solution
    FUNCTION minDifficulty(jobDifficulty, d)
        SET n TO the LENGTH OF jobDifficulty
        
        IF n LESS THAN d
            RETURN negative one
        END IF
        
        FUNCTION dp(i, current_day)
            IF current_day EQUALS one
                RETURN the MAXIMUM value FROM element at position i TO the END OF jobDifficulty
            END IF
            
            SET min_difficulty TO positive infinity
            SET current_max TO zero
            FOR j FROM i TO n MINUS current_day PLUS one
                SET current_max TO the MAXIMUM value BETWEEN current_max AND element at position j OF jobDifficulty
                SET min_difficulty TO the MINIMUM value BETWEEN min_difficulty AND current_max PLUS CALL dp WITH j PLUS one AND current_day MINUS one
            END FOR
            
            RETURN min_difficulty
        END FUNCTION
        
        RETURN CALL dp WITH zero AND d
    END FUNCTION
END CLASS
