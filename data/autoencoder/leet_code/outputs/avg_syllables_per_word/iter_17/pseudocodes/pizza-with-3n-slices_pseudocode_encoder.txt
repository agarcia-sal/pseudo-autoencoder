CLASS Solution
    FUNCTION maxSizeSlices(list_of_slices)
        SET number_of_slices TO the LENGTH OF list_of_slices DIVIDED BY three

        FUNCTION dp(start_index, end_index, count)
            IF count EQUALS zero OR start_index GREATER THAN end_index
                RETURN zero
            END IF
            SET pick_current TO element at position start_index of list_of_slices PLUS CALL dp WITH parameters start_index PLUS two, end_index, count MINUS one
            SET skip_current TO CALL dp WITH parameters start_index PLUS one, end_index, count
            RETURN the GREATER OF pick_current AND skip_current
        END FUNCTION

        SET case_one TO CALL dp WITH parameters one, the LENGTH OF list_of_slices MINUS one, number_of_slices
        SET case_two TO CALL dp WITH parameters zero, the LENGTH OF list_of_slices MINUS two, number_of_slices

        RETURN the GREATER OF case_one AND case_two
    END FUNCTION
END CLASS
