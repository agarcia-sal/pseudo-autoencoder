CLASS Solution
    FUNCTION maximumScore(scores, edges)
        SET graph TO CALL create_default_dictionary_of_lists
        FOR each_pair_of_nodes u AND v IN edges
            CALL APPEND_pair_to_graph_with_scores WITH arguments graph, u, v, scores
        END FOR
        FOR each_node IN graph
            SET graph_at_node TO CALL select_top_three_elements FROM graph_at_node
        END FOR
        SET max_score TO negative_one
        FOR each_pair_of_nodes u AND v IN edges
            FOR each_score1_and_node_x IN graph_at_u
                FOR each_score2_and_node_y IN graph_at_v
                    IF the_number_of_distinct_elements_in_set_of x y u v EQUALS four
                        SET sum_score TO score1 PLUS score2 PLUS element_at_position_u OF scores PLUS element_at_position_v OF scores
                        SET max_score TO the_greater_value_of max_score AND sum_score
                    END IF
                END FOR
            END FOR
        END FOR
        RETURN max_score
    END FUNCTION
