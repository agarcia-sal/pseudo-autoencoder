CLASS Solution
    FUNCTION numberToWords(num)
        IF num EQUALS zero
            RETURN Zero
        END IF

        SET below_twenty TO auxiliary FUNCTION that provides word equivalents for numbers from zero TO nineteen
        SET tens TO auxiliary FUNCTION that provides word equivalents for multiples of ten from twenty TO ninety
        SET thousands TO auxiliary FUNCTION that provides thousand scale words including empty string for zero scale

        FUNCTION helper(n)
            IF n EQUALS zero
                RETURN empty string
            END IF
            IF n LESS THAN twenty
                RETURN word at position n in below_twenty CONCATENATED WITH space
            END IF
            IF n LESS THAN one hundred
                RETURN word at position the integer division of n BY ten in tens CONCATENATED WITH space CONCATENATED WITH CALL helper WITH n MODULO ten
            END IF
            RETURN word at position the integer division of n BY one hundred in below_twenty CONCATENATED WITH space CONCATENATED WITH Hundred CONCATENATED WITH space CONCATENATED WITH CALL helper WITH n MODULO one hundred
        END FUNCTION

        SET result TO empty string
        SET i TO zero

        WHILE num GREATER THAN zero
            IF num MODULO one thousand NOT EQUALS zero
                SET result TO CALL helper WITH num MODULO one thousand CONCATENATED WITH word at position i in thousands CONCATENATED WITH space CONCATENATED WITH result
            END IF
            SET num TO integer division of num BY one thousand
            INCREMENT i BY one
        END WHILE

        RETURN result with leading and trailing whitespace removed
    END FUNCTION
END CLASS
