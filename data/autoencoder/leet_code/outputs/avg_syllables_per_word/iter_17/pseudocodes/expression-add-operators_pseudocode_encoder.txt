CLASS Solution
    FUNCTION addOperators(num_string, target_integer)
        FUNCTION dfs(current_index, current_path, current_value, previous_value)
            IF current_index EQUALS the LENGTH OF num_string
                IF current_value EQUALS target_integer
                    APPEND current_path TO result_list
                END IF
                RETURN
            END IF

            FOR i FROM current_index PLUS one TO the LENGTH OF num_string
                SET operand_substring TO substring from position current_index TO i MINUS one of num_string
                IF the LENGTH OF operand_substring GREATER THAN one AND element at position zero of operand_substring EQUALS zero
                    CONTINUE TO next iteration
                END IF

                IF current_index EQUALS zero
                    CALL dfs WITH parameters i, operand_substring, auxiliary FUNCTION ConvertToInt WITH operand_substring, auxiliary FUNCTION ConvertToInt WITH operand_substring
                ELSE
                    CALL dfs WITH parameters i, current_path CONCATENATED WITH the PLUS SIGN CONCATENATED WITH operand_substring, current_value PLUS auxiliary FUNCTION ConvertToInt WITH operand_substring, auxiliary FUNCTION ConvertToInt WITH operand_substring
                    CALL dfs WITH parameters i, current_path CONCATENATED WITH the MINUS SIGN CONCATENATED WITH operand_substring, current_value MINUS auxiliary FUNCTION ConvertToInt WITH operand_substring, NEGATIVE OF auxiliary FUNCTION ConvertToInt WITH operand_substring
                    CALL dfs WITH parameters i, current_path CONCATENATED WITH the MULTIPLICATION SIGN CONCATENATED WITH operand_substring, current_value MINUS previous_value PLUS previous_value MULTIPLIED BY auxiliary FUNCTION ConvertToInt WITH operand_substring, previous_value MULTIPLIED BY auxiliary FUNCTION ConvertToInt WITH operand_substring
                END IF
            END FOR
        END FUNCTION

        SET result_list TO empty list
        CALL dfs WITH parameters zero, empty string, zero, zero
        RETURN result_list
    END FUNCTION
END CLASS
