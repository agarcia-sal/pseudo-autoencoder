CLASS Solution
    FUNCTION possibleToStamp(grid, stampHeight, stampWidth)
        SET m TO the LENGTH OF grid
        SET n TO the LENGTH OF element at position zero of grid

        FUNCTION prefix_sum(matrix)
            INITIALIZE p AS a two dimensional list of zeros with dimensions m plus one BY n plus one
            FOR r FROM one TO m
                FOR c FROM one TO n
                    SET element at position r of p AT position c TO element at position r minus one of matrix AT position c minus one PLUS element at position r minus one of p AT position c PLUS element at position r of p AT position c minus one MINUS element at position r minus one of p AT position c minus one
                END FOR
            END FOR
            RETURN p
        END FUNCTION

        FUNCTION submatrix_sum(p, r1, c1, r2, c2)
            RETURN element at position r2 plus one of p AT position c2 plus one MINUS element at position r2 plus one of p AT position c1 MINUS element at position r1 of p AT position c2 plus one PLUS element at position r1 of p AT position c1
        END FUNCTION

        SET p_grid TO CALL prefix_sum WITH matrix grid
        INITIALIZE p_stamps AS a two dimensional list of zeros with dimensions m plus one BY n plus one

        FOR r FROM zero TO m minus one
            FOR c FROM zero TO n minus one
                IF element at position r of grid AT position c EQUALS zero
                    IF r PLUS stampHeight LESS THAN OR EQUAL TO m AND c PLUS stampWidth LESS THAN OR EQUAL TO n
                        IF CALL submatrix_sum WITH parameters p_grid r c r plus stampHeight minus one c plus stampWidth minus one EQUALS zero
                            INCREMENT element at position r of p_stamps AT position c BY one
                        END IF
                    END IF
                END IF
            END FOR
        END FOR

        SET p_stamps TO CALL prefix_sum WITH matrix p_stamps

        FOR r FROM zero TO m minus one
            FOR c FROM zero TO n minus one
                IF element at position r of grid AT position c EQUALS zero
                    IF CALL submatrix_sum WITH parameters p_stamps the MAXIMUM OF zero AND r minus stampHeight plus one the MAXIMUM OF zero AND c minus stampWidth plus one r c EQUALS zero
                        RETURN false
                    END IF
                END IF
            END FOR
        END FOR

        RETURN true
    END FUNCTION
END CLASS
