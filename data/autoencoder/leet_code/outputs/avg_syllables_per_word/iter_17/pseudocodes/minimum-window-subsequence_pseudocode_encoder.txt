CLASS Solution
    FUNCTION minWindow(s1, s2)
        SET minimum_window_length TO infinity
        SET minimum_window_start TO negative one
        SET length_of_s1 TO the LENGTH OF s1
        SET length_of_s2 TO the LENGTH OF s2

        FOR starting_index FROM zero TO length_of_s1 MINUS one
            SET pointer_s2 TO zero

            FOR current_index FROM starting_index TO length_of_s1 MINUS one
                IF element at position current_index of s1 EQUALS element at position pointer_s2 of s2
                    INCREMENT pointer_s2 BY one

                    IF pointer_s2 EQUALS length_of_s2
                        IF current_index MINUS starting_index PLUS one LESS THAN minimum_window_length
                            SET minimum_window_length TO current_index MINUS starting_index PLUS one
                            SET minimum_window_start TO starting_index
                        END IF
                        BREAK
                    END IF
                END IF
            END FOR
        END FOR

        IF minimum_window_start EQUALS negative one
            RETURN empty string
        ELSE
            RETURN substring from position minimum_window_start TO minimum_window_start PLUS minimum_window_length MINUS one of s1
        END IF
    END FUNCTION
END CLASS
