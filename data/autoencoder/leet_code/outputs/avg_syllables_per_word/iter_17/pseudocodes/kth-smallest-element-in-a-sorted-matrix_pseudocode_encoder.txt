CLASS Solution
    FUNCTION kthSmallest(matrix AS list of list of integers, k AS integer) RETURNS integer
        FUNCTION countLessEqual(mid AS integer) RETURNS integer
            SET count TO zero
            SET row TO n MINUS one
            SET col TO zero
            WHILE row IS GREATER THAN OR EQUAL TO zero AND col IS LESS THAN n
                IF element at position row of matrix AT position col IS LESS THAN OR EQUAL TO mid
                    INCREMENT count BY row PLUS one
                    INCREMENT col BY one
                ELSE
                    DECREMENT row BY one
                END IF
            END WHILE
            RETURN count
        END FUNCTION

        SET n TO the LENGTH OF matrix
        SET low TO element at position zero of matrix AT position zero
        SET high TO element at position n MINUS one of matrix AT position n MINUS one

        WHILE low IS LESS THAN high
            SET mid TO low PLUS high DIVIDED BY two
            IF countLessEqual(mid) IS LESS THAN k
                SET low TO mid PLUS one
            ELSE
                SET high TO mid
            END IF
        END WHILE

        RETURN low
    END FUNCTION
END CLASS
