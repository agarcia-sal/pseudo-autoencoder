CLASS Solution
    FUNCTION minimumDeviation(nums)
        SET max_heap TO empty list
        SET min_value TO infinity
        FOR each num IN nums
            IF the remainder of num DIVIDED BY two EQUALS one
                MULTIPLY num BY two AND ASSIGN TO num
            END IF
            IF min_value GREATER THAN num
                SET min_value TO num
            END IF
            CALL heap push WITH parameters max_heap AND the negation of num
        END FOR

        SET min_deviation TO infinity

        WHILE the remainder of the element at position zero of max_heap DIVIDED BY two EQUALS zero
            CALL heap pop WITH parameter max_heap AND ASSIGN the negation TO max_value
            IF min_deviation GREATER THAN max_value MINUS min_value
                SET min_deviation TO max_value MINUS min_value
            END IF
            DIVIDE max_value BY two USING floor division AND ASSIGN TO max_value
            IF min_value GREATER THAN max_value
                SET min_value TO max_value
            END IF
            CALL heap push WITH parameters max_heap AND the negation of max_value
        END WHILE

        IF min_deviation GREATER THAN the negation of the element at position zero of max_heap MINUS min_value
            SET min_deviation TO the negation of the element at position zero of max_heap MINUS min_value
        END IF

        RETURN min_deviation
    END FUNCTION
END CLASS
