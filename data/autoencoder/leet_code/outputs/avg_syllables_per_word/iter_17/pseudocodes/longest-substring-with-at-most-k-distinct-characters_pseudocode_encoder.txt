CLASS Solution
    FUNCTION lengthOfLongestSubstringKDistinct(s STRING, k INTEGER)
        IF k EQUALS zero
            RETURN zero
        END IF

        SET char_map TO an empty dictionary
        SET left TO zero
        SET max_length TO zero

        FOR right FROM zero TO the LENGTH OF s MINUS one
            SET element at position right of s TO current_char
            SET the value associated with current_char in char_map TO right

            IF the LENGTH OF char_map IS GREATER THAN k
                SET leftmost_char TO the key in char_map WITH the MINIMUM associated value
                SET left TO the value associated with leftmost_char in char_map PLUS one
                REMOVE leftmost_char FROM char_map
            END IF

            SET current_length TO right MINUS left PLUS one
            SET max_length TO the MAXIMUM OF max_length AND current_length
        END FOR

        RETURN max_length
    END FUNCTION
END CLASS
