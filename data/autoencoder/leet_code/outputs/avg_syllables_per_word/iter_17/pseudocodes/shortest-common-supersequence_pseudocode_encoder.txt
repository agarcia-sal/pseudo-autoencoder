CLASS Solution
    FUNCTION shortestCommonSupersequence(str1, str2)
        SET m TO the LENGTH OF str1
        SET n TO the LENGTH OF str2

        CALL construct_dp_table WITH parameters m and n TO INITIALIZE dp_table

        FOR i FROM one TO m
            FOR j FROM one TO n
                IF element at position i MINUS one of str1 EQUALS element at position j MINUS one of str2
                    SET element at position i and j of dp_table TO element at position i MINUS one and j MINUS one of dp_table PLUS one
                ELSE
                    SET element at position i and j of dp_table TO THE GREATER VALUE OF element at position i MINUS one and j of dp_table AND element at position i and j MINUS one of dp_table
                END IF
            END FOR
        END FOR

        SET i TO m
        SET j TO n
        SET scs TO an empty list

        WHILE i GREATER THAN zero AND j GREATER THAN zero
            IF element at position i MINUS one of str1 EQUALS element at position j MINUS one of str2
                APPEND element at position i MINUS one of str1 TO scs
                DECREMENT i BY one
                DECREMENT j BY one
            ELSE IF element at position i MINUS one and j of dp_table GREATER THAN element at position i and j MINUS one of dp_table
                APPEND element at position i MINUS one of str1 TO scs
                DECREMENT i BY one
            ELSE
                APPEND element at position j MINUS one of str2 TO scs
                DECREMENT j BY one
            END IF
        END WHILE

        WHILE i GREATER THAN zero
            APPEND element at position i MINUS one of str1 TO scs
            DECREMENT i BY one
        END WHILE

        WHILE j GREATER THAN zero
            APPEND element at position j MINUS one of str2 TO scs
            DECREMENT j BY one
        END WHILE

        CALL reverse_list WITH parameter scs TO REVERSE scs

        RETURN CALL join_characters WITH parameter scs
    END FUNCTION

    FUNCTION construct_dp_table(m, n)
        RETURN a two-dimensional list of zeros with m PLUS one rows and n PLUS one columns
    END FUNCTION

    FUNCTION reverse_list(list_to_reverse)
        REVERSE list_to_reverse IN PLACE
    END FUNCTION

    FUNCTION join_characters(list_of_characters)
        RETURN concatenation of all characters in list_of_characters
    END FUNCTION
END CLASS
