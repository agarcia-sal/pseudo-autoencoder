CLASS Solution
    FUNCTION longestLine(matrix)
        IF matrix IS empty OR element at position zero of matrix IS empty
            RETURN zero
        END IF

        SET number_of_rows TO the LENGTH OF matrix
        SET number_of_columns TO the LENGTH OF element at position zero of matrix

        SET directions TO list of tuples containing zero and one comma one and zero comma one and one comma one and MINUS one

        SET maximum_length TO zero

        FUNCTION check(row_position, column_position, direction_x, direction_y)
            SET sequence_length TO zero
            WHILE row_position IS GREATER THAN OR EQUAL TO zero AND row_position IS LESS THAN number_of_rows AND column_position IS GREATER THAN OR EQUAL TO zero AND column_position IS LESS THAN number_of_columns AND element at position row_position of matrix at position column_position EQUALS one
                INCREMENT sequence_length BY one
                INCREMENT row_position BY direction_x
                INCREMENT column_position BY direction_y
            END WHILE
            RETURN sequence_length
        END FUNCTION

        FOR row_index FROM zero TO number_of_rows MINUS one
            FOR column_index FROM zero TO number_of_columns MINUS one
                IF element at position row_index of matrix at position column_index EQUALS one
                    FOR each direction_pair IN directions
                        ASSIGN direction_x TO element at position zero of direction_pair
                        ASSIGN direction_y TO element at position one of direction_pair
                        IF row_index MINUS direction_x IS LESS THAN zero OR column_index MINUS direction_y IS LESS THAN zero OR column_index MINUS direction_y IS GREATER THAN OR EQUAL TO number_of_columns OR element at position row_index MINUS direction_x of matrix at position column_index MINUS direction_y EQUALS zero
                            SET maximum_length TO the GREATER OF maximum_length AND CALL check WITH parameters row_index comma column_index comma direction_x comma direction_y
                        END IF
                    END FOR
                END IF
            END FOR
        END FOR

        RETURN maximum_length
    END FUNCTION
END CLASS
