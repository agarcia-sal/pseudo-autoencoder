CLASS Solution
    FUNCTION findMedianSortedArrays(nums1, nums2)
        IF the LENGTH OF nums1 GREATER THAN the LENGTH OF nums2
            SET temporary TO nums1
            SET nums1 TO nums2
            SET nums2 TO temporary
        END IF

        SET x TO the LENGTH OF nums1
        SET y TO the LENGTH OF nums2

        SET low TO zero
        SET high TO x

        WHILE low LESS THAN OR EQUAL TO high holds
            SET partitionX TO the INTEGER DIVISION OF the SUM of low AND high BY two
            SET partitionY TO the INTEGER DIVISION OF the SUM of x AND y AND one BY two MINUS partitionX

            SET maxLeftX TO negative infinity IF partitionX EQUALS zero ELSE element at position partitionX MINUS one of nums1
            SET minRightX TO positive infinity IF partitionX EQUALS x ELSE element at position partitionX of nums1

            SET maxLeftY TO negative infinity IF partitionY EQUALS zero ELSE element at position partitionY MINUS one of nums2
            SET minRightY TO positive infinity IF partitionY EQUALS y ELSE element at position partitionY of nums2

            IF maxLeftX LESS THAN OR EQUAL TO minRightY AND maxLeftY LESS THAN OR EQUAL TO minRightX
                IF the MODULO of the SUM of x AND y BY two EQUALS zero
                    SET maximum_left TO the GREATER OF maxLeftX AND maxLeftY
                    SET minimum_right TO the LESSER OF minRightX AND minRightY
                    RETURN the SUM of maximum_left AND minimum_right DIVIDED BY two
                ELSE
                    RETURN the GREATER OF maxLeftX AND maxLeftY
                END IF
            ELSE IF maxLeftX GREATER THAN minRightY
                SET high TO partitionX MINUS one
            ELSE
                SET low TO partitionX PLUS one
            END IF
        END WHILE
    END FUNCTION
END CLASS
