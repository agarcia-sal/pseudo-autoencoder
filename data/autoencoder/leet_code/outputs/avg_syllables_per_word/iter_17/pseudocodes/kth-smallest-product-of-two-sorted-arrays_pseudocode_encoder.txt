CLASS Solution
    FUNCTION kthSmallestProduct(nums1, nums2, k)
        FUNCTION count(p)
            SET cnt TO zero
            SET n TO the LENGTH OF nums2
            FOR each x IN nums1
                IF x GREATER THAN zero
                    INCREMENT cnt BY the RESULT OF bisect_right CALLED WITH nums2 AND p DIVIDED BY x
                ELSE IF x LESS THAN zero
                    INCREMENT cnt BY n MINUS the RESULT OF bisect_left CALLED WITH nums2 AND p DIVIDED BY x
                ELSE
                    INCREMENT cnt BY n MULTIPLIED BY the VALUE OF p GREATER THAN OR EQUAL TO zero
                END IF
            END FOR
            RETURN cnt
        END FUNCTION

        SET mx TO the MAXIMUM OF the ABSOLUTE VALUE OF element at position zero of nums1 AND the ABSOLUTE VALUE OF element at last position of nums1
        SET mx TO mx MULTIPLIED BY the MAXIMUM OF the ABSOLUTE VALUE OF element at position zero of nums2 AND the ABSOLUTE VALUE OF element at last position of nums2
        RETURN the RESULT OF bisect_left CALLED WITH the RANGE FROM negative mx TO mx INCLUSIVE, k, AND key equal to count MINUS mx
    END FUNCTION
END CLASS
