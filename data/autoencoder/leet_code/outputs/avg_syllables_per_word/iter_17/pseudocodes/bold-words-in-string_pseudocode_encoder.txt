CLASS Solution
    FUNCTION boldWords(words, s)
        SET bold TO a list of boolean false values with the LENGTH OF s
        FOR each word IN words
            SET start TO the POSITION OF word IN s
            WHILE start IS NOT EQUALS TO negative one
                FOR index FROM start TO start PLUS the LENGTH OF word MINUS one
                    SET element at position index of bold TO true
                END FOR
                SET start TO the POSITION OF word IN s STARTING FROM start PLUS one
            END WHILE
        END FOR
        SET result TO an empty list
        SET i TO zero
        WHILE i IS LESS THAN the LENGTH OF s
            IF element at position i of bold IS true
                APPEND the opening bold tag TO result
                WHILE i IS LESS THAN the LENGTH OF s AND element at position i of bold IS true
                    APPEND element at position i of s TO result
                    INCREMENT i BY one
                END WHILE
                APPEND the closing bold tag TO result
            ELSE
                APPEND element at position i of s TO result
                INCREMENT i BY one
            END IF
        END WHILE
        RETURN the concatenation OF all elements in result
    END FUNCTION
END CLASS
