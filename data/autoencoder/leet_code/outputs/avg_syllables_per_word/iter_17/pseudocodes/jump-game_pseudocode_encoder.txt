CLASS Solution
    FUNCTION canJump(list_of_numbers)
        SET farthest_index_reachable TO zero
        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one
            IF index IS GREATER THAN farthest_index_reachable
                RETURN false
            END IF
            SET farthest_index_reachable TO the GREATER OF farthest_index_reachable AND index PLUS element at position index of list_of_numbers
        END FOR
        RETURN farthest_index_reachable IS GREATER THAN OR EQUAL TO the LENGTH OF list_of_numbers MINUS one
    END FUNCTION
END CLASS
