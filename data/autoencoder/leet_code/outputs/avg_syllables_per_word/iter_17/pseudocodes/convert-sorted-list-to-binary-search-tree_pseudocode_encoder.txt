CLASS ListNode
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values: list)
    IF values IS empty
        RETURN None
    END IF
    SET head TO CALL ListNode WITH element at position zero of values
    SET p TO head
    FOR each val IN sublist of values from position one TO end
        SET node TO CALL ListNode WITH val
        SET p.next TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS None AND p2 IS None
        RETURN True
    END IF
    IF p1 IS None OR p2 IS None
        RETURN False
    END IF
    RETURN the value at p1 EQUALS the value at p2 AND CALL is_same_list WITH p1.next AND p2.next
END FUNCTION

CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values: list)
    IF values IS empty
        RETURN None
    END IF
    SET root TO CALL TreeNode WITH element at position zero of values
    SET i TO one
    SET queue TO CALL deque WITH no parameters
    CALL queue APPEND WITH root
    WHILE queue IS NOT empty
        SET node TO CALL queue POP LEFT WITH no parameters
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.left TO CALL TreeNode WITH element at position i of values
            CALL queue APPEND WITH node.left
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.right TO CALL TreeNode WITH element at position i of values
            CALL queue APPEND WITH node.right
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF value at p DOES NOT EQUAL value at q
        RETURN False
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION sortedListToBST(self, head: Optional ListNode) RETURNS Optional TreeNode
        IF head IS None
            RETURN None
        END IF

        FUNCTION findMiddle(left, right)
            SET slow TO left
            SET fast TO left
            WHILE fast IS NOT right AND fast.next IS NOT right
                SET slow TO slow.next
                SET fast TO fast.next.next
            END WHILE
            RETURN slow
        END FUNCTION

        FUNCTION convertListToBST(left, right)
            IF left IS right
                RETURN None
            END IF

            SET mid TO CALL findMiddle WITH left AND right

            SET node TO CALL TreeNode WITH value at mid
            SET node.left TO CALL convertListToBST WITH left AND mid
            SET node.right TO CALL convertListToBST WITH mid.next AND right
            RETURN node
        END FUNCTION

        RETURN CALL convertListToBST WITH head AND None
    END FUNCTION
END CLASS
