CLASS Solution
    FUNCTION findLonelyPixel(picture)
        IF picture IS empty OR element at position zero of picture IS empty
            RETURN zero
        END IF
        SET m TO the LENGTH OF picture
        SET n TO the LENGTH OF element at position zero of picture
        CALL initialize_list_with_zeros WITH parameter m ASSIGN TO row_count
        CALL initialize_list_with_zeros WITH parameter n ASSIGN TO col_count
        FOR i FROM zero TO m MINUS one
            FOR j FROM zero TO n MINUS one
                IF element at position i of picture AT position j EQUALS letter B
                    INCREMENT element at position i of row_count BY one
                    INCREMENT element at position j of col_count BY one
                END IF
            END FOR
        END FOR
        SET lonely_pixel_count TO zero
        FOR i FROM zero TO m MINUS one
            FOR j FROM zero TO n MINUS one
                IF element at position i of picture AT position j EQUALS letter B AND element at position i of row_count EQUALS one AND element at position j of col_count EQUALS one
                    INCREMENT lonely_pixel_count BY one
                END IF
            END FOR
        END FOR
        RETURN lonely_pixel_count
    END FUNCTION
END CLASS
