CLASS Solution
    FUNCTION maxHeight(cuboids)
        FOR each cuboid IN cuboids
            CALL sort ON cuboid
        END FOR

        CALL sort ON cuboids

        SET n TO the LENGTH OF cuboids
        SET dp TO a LIST OF zero WITH SIZE n

        FOR i FROM zero TO n MINUS one
            SET element at position i of dp TO element at position two of element at position i of cuboids

            FOR j FROM zero TO i MINUS one
                IF element at position zero of element at position j of cuboids IS LESS THAN OR EQUAL TO element at position zero of element at position i of cuboids AND element at position one of element at position j of cuboids IS LESS THAN OR EQUAL TO element at position one of element at position i of cuboids AND element at position two of element at position j of cuboids IS LESS THAN OR EQUAL TO element at position two of element at position i of cuboids
                    SET element at position i of dp TO the GREATER OF element at position i of dp AND element at position j of dp PLUS element at position two of element at position i of cuboids
                END IF
            END FOR
        END FOR

        RETURN the GREATEST VALUE IN dp
    END FUNCTION
END CLASS
