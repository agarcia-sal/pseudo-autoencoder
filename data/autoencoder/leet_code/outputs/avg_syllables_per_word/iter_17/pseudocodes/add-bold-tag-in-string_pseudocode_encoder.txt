CLASS Solution
    FUNCTION addBoldTag(s STRING, words LIST OF STRING) RETURNS STRING
        IF words IS EMPTY
            RETURN s
        END IF

        SET n TO the LENGTH OF s
        SET mask TO a LIST OF boolean FALSE OF LENGTH n

        FOR each word IN words
            SET start TO zero
            WHILE start IS LESS THAN OR EQUAL TO n MINUS the LENGTH OF word
                SET start TO the POSITION OF word IN s STARTING FROM start
                IF start EQUALS negative one
                    BREAK
                END IF
                FOR i FROM start TO start PLUS the LENGTH OF word MINUS one
                    SET element at position i OF mask TO TRUE
                END FOR
                INCREMENT start BY one
            END WHILE
        END FOR

        SET result TO an EMPTY LIST
        SET i TO zero
        WHILE i IS LESS THAN n
            IF element at position i OF mask IS TRUE
                APPEND the string less than b greater than TO result
                WHILE i IS LESS THAN n AND element at position i OF mask IS TRUE
                    APPEND element at position i OF s TO result
                    INCREMENT i BY one
                END WHILE
                APPEND the string less than slash b greater than TO result
            ELSE
                APPEND element at position i OF s TO result
                INCREMENT i BY one
            END IF
        END WHILE

        RETURN the CONCATENATION OF all elements IN result
    END FUNCTION
END CLASS
