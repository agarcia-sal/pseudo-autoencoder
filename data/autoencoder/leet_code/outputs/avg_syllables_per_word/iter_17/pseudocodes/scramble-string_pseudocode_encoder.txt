CLASS Solution
    FUNCTION isScramble(s1, s2)
        IF s1 EQUALS s2
            RETURN True
        END IF
        IF the SORTED CHARACTERS OF s1 NOT EQUALS the SORTED CHARACTERS OF s2
            RETURN False
        END IF
        SET n TO the LENGTH OF s1
        FOR i FROM one TO n MINUS one
            IF CALL isScramble WITH parameters substring from position zero TO i MINUS one of s1 AND substring from position zero TO i MINUS one of s2 AND CALL isScramble WITH parameters substring from position i TO the END OF s1 AND substring from position i TO the END OF s2
                RETURN True
            END IF
            IF CALL isScramble WITH parameters substring from position zero TO i MINUS one of s1 AND substring from position n MINUS i TO the END OF s2 AND CALL isScramble WITH parameters substring from position i TO the END OF s1 AND substring from position zero TO n MINUS i MINUS one of s2
                RETURN True
            END IF
        END FOR
        RETURN False
    END FUNCTION
END CLASS
