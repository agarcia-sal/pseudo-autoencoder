CLASS Solution
    FUNCTION minInteger(num AS string, k AS integer) RETURNS string
        SET digit_list TO the LIST OF characters IN num
        SET n TO the LENGTH OF digit_list
        
        WHILE k IS GREATER THAN zero
            SET swapped TO False
            
            FOR index FROM zero TO n MINUS one MINUS one
                IF the element at position index of digit_list IS GREATER THAN the element at position index PLUS one of digit_list
                    SET min_digit TO nine
                    SET min_index TO minus one
                    
                    FOR position FROM index TO the MINIMUM OF index PLUS k PLUS one AND n MINUS one
                        IF the element at position position of digit_list IS LESS THAN min_digit
                            SET min_digit TO the element at position position of digit_list
                            SET min_index TO position
                        END IF
                    END FOR
                    
                    CALL FUNCTION perform_swap WITH parameters digit_list, index, min_index
                    
                    DECREMENT k BY min_index MINUS index
                    SET swapped TO True
                    BREAK out of FOR loop
                END IF
            END FOR
            
            IF swapped IS False
                BREAK out of WHILE loop
            END IF
        END WHILE
        
        RETURN the CONCATENATION OF elements IN digit_list AS string
    END FUNCTION
    
    FUNCTION perform_swap(digit_list AS list, start_index AS integer, end_index AS integer)
        SET prefix_list TO the SUBLIST OF digit_list FROM position start_index TO end_index MINUS one
        SET moved_element TO the element at position end_index of digit_list
        SET replacement_list TO a NEW list STARTING WITH moved_element FOLLOWED BY prefix_list
        REPLACE the SUBLIST OF digit_list FROM position start_index TO end_index WITH replacement_list
    END FUNCTION
END CLASS
