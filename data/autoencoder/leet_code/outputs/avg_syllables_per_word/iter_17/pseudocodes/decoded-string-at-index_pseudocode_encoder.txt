CLASS Solution
    FUNCTION decodeAtIndex(s, k)
        SET length TO zero
        FOR each character IN s
            IF character IS a digit
                SET length TO length MULTIPLIED BY numeric value of character
            ELSE
                INCREMENT length BY one
            END IF
        END FOR

        FOR each character IN reversed s
            SET k TO k MODULO length
            
            IF k EQUALS zero AND character IS an alphabetic character
                RETURN character
            END IF

            IF character IS a digit
                SET length TO length DIVIDED BY numeric value of character
            ELSE
                DECREMENT length BY one
            END IF
        END FOR
    END FUNCTION
END CLASS
