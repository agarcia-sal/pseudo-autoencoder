CLASS Solution
    FUNCTION minDays(grid)
        FUNCTION dfs(x, y, visited)
            IF x IS LESS THAN zero OR x IS GREATER THAN OR EQUAL TO the LENGTH OF grid OR y IS LESS THAN zero OR y IS GREATER THAN OR EQUAL TO the LENGTH OF element at position zero of grid OR the TUPLE x COMMA y IS IN visited OR element at position x of grid AT position y EQUALS zero
                RETURN
            END IF
            CALL visited ADD WITH the TUPLE x COMMA y
            FOR each TUPLE dx COMMA dy IN the LIST of TUPLES negative one zero COMMA one zero COMMA zero negative one COMMA zero one
                CALL dfs WITH parameters x PLUS dx COMMA y PLUS dy COMMA visited
            END FOR
        END FUNCTION

        FUNCTION is_connected
            SET visited TO an EMPTY SET
            SET land_cells TO an EMPTY LIST
            FOR i FROM zero TO the LENGTH OF grid MINUS one
                FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one
                    IF element at position i of grid AT position j EQUALS one
                        APPEND the TUPLE i COMMA j TO land_cells
                    END IF
                END FOR
            END FOR
            IF the LENGTH OF land_cells EQUALS zero
                RETURN True
            END IF
            CALL dfs WITH parameters element at position zero of land_cells AT position zero COMMA element at position zero of land_cells AT position one COMMA visited
            RETURN the LENGTH OF visited EQUALS the LENGTH OF land_cells
        END FUNCTION

        IF NOT is_connected
            RETURN zero
        END IF

        FOR i FROM zero TO the LENGTH OF grid MINUS one
            FOR j FROM zero TO the LENGTH OF element at position zero of grid MINUS one
                IF element at position i of grid AT position j EQUALS one
                    SET element at position i of grid AT position j TO zero
                    IF NOT is_connected
                        RETURN one
                    END IF
                    SET element at position i of grid AT position j TO one
                END IF
            END FOR
        END FOR

        RETURN two
    END FUNCTION
END CLASS
