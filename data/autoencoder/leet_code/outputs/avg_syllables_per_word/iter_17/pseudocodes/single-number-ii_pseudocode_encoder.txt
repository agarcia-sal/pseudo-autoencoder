CLASS Solution
    FUNCTION singleNumber(list_of_integers nums)
        SET ones TO zero
        SET twos TO zero
        FOR each number IN nums
            SET ones TO the bitwise XOR of ones AND number AND the bitwise NOT of twos
            SET twos TO the bitwise XOR of twos AND number AND the bitwise NOT of ones
        END FOR
        RETURN ones
    END FUNCTION
END CLASS
