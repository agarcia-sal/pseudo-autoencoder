CLASS Solution
    FUNCTION pathSum(nums)
        SET tree TO an empty dictionary
        FOR each num IN nums
            SET depth TO num DIVIDED BY one hundred
            SET pos TO the remainder of num DIVIDED BY one hundred DIVIDED BY ten
            SET value TO the remainder of num DIVIDED BY ten
            SET element at key depth and pos of tree TO value
        END FOR

        FUNCTION dfs(depth, pos, path_sum)
            SET current_value TO the element at key depth and pos of tree OR none if not present
            IF current_value IS none
                RETURN zero
            END IF
            SET new_path_sum TO path_sum PLUS current_value
            SET left_child TO depth PLUS one AND pos MULTIPLIED BY two MINUS one
            SET right_child TO depth PLUS one AND pos MULTIPLIED BY two
            IF left_child IS NOT IN tree AND right_child IS NOT IN tree
                RETURN new_path_sum
            END IF
            SET left_sum TO CALL dfs WITH parameters left_child first element, left_child second element, new_path_sum
            SET right_sum TO CALL dfs WITH parameters right_child first element, right_child second element, new_path_sum
            RETURN left_sum PLUS right_sum
        END FUNCTION

        RETURN CALL dfs WITH parameters one, one, zero
    END FUNCTION
END CLASS
