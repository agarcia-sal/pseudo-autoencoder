CLASS Solution
    FUNCTION kthSmallestPrimeFraction(arr, k)
        SET n TO the LENGTH OF arr
        SET min_heap TO an empty list

        FOR i FROM zero TO n MINUS one MINUS one
            CALL heap push FUNCTION WITH min_heap AND the TUPLE OF arr element at position i DIVIDED BY arr element at position n MINUS one, i, n MINUS one
        END FOR

        FOR counter FROM zero TO k MINUS one MINUS one
            CALL heap pop FUNCTION WITH min_heap AND ASSIGN RESULT TO _, i, j
            IF j MINUS one IS GREATER THAN i
                CALL heap push FUNCTION WITH min_heap AND the TUPLE OF arr element at position i DIVIDED BY arr element at position j MINUS one, i, j MINUS one
            END IF
        END FOR

        CALL heap pop FUNCTION WITH min_heap AND ASSIGN RESULT TO _, i, j
        RETURN a list CONTAINING arr element at position i AND arr element at position j
    END FUNCTION
END CLASS
