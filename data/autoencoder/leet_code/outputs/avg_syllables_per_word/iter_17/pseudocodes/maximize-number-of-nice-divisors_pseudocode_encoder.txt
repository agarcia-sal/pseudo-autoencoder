CLASS Solution
    FUNCTION maxNiceDivisors(primeFactors)
        SET MOD TO ten POWER nine PLUS seven
        
        IF primeFactors LESS THAN OR EQUAL TO three
            RETURN primeFactors
        END IF
        
        SET num_threes TO primeFactors DIVIDED BY three
        SET remainder TO primeFactors MODULO three
        
        IF remainder EQUALS zero
            RETURN CALL pow WITH parameters three COMMA num_threes COMMA MOD
        ELSE IF remainder EQUALS one
            RETURN (CALL pow WITH parameters three COMMA num_threes MINUS one COMMA MOD) MULTIPLIED BY four MODULO MOD
        ELSE
            RETURN (CALL pow WITH parameters three COMMA num_threes COMMA MOD) MULTIPLIED BY two MODULO MOD
        END IF
    END FUNCTION
END CLASS
