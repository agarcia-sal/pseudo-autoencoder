CLASS Solution
    FUNCTION intToRoman(num)
        SET val TO auxiliary FUNCTION get_roman_values
        SET syms TO auxiliary FUNCTION get_roman_symbols
        SET roman_numeral TO empty string
        SET i TO zero
        WHILE num GREATER THAN zero holds
            SET count TO num DIVIDED BY element at position i of val
            FOR each occurrence FROM one TO count
                APPEND element at position i of syms TO roman_numeral
                DECREMENT num BY element at position i of val
            END FOR
            INCREMENT i BY one
        END WHILE
        RETURN roman_numeral
    END FUNCTION

    FUNCTION get_roman_values
        RETURN list of one thousand, nine hundred, five hundred, four hundred, one hundred, ninety, fifty, forty, ten, nine, five, four, one
    END FUNCTION

    FUNCTION get_roman_symbols
        RETURN list of M, CM, D, CD, C, XC, L, XL, X, IX, V, IV, I
    END FUNCTION
END CLASS
