CLASS Solution
    FUNCTION wordsTyping(sentence, rows, cols)
        SET s TO CALL join sentence WITH space THEN combined WITH space
        SET total_length TO the LENGTH OF s
        SET position TO zero
        FOR _ FROM one TO rows
            INCREMENT position BY cols
            IF element at position MODULO total_length of s EQUALS space
                INCREMENT position BY one
            ELSE
                WHILE position GREATER THAN zero AND element at (position MINUS one) MODULO total_length of s NOT EQUALS space
                    DECREMENT position BY one
                END WHILE
            END IF
        END FOR
        SET result TO position DIVIDED BY total_length
        RETURN result
    END FUNCTION
END CLASS
