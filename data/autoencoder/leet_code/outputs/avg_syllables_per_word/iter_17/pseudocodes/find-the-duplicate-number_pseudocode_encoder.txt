CLASS Solution
    FUNCTION findDuplicate(list_of_numbers)
        FUNCTION f(number_x)
            RETURN the SUM OF each element v IN list_of_numbers WHERE element v IS LESS THAN OR EQUAL TO number_x IS GREATER THAN number_x
        END FUNCTION
        RETURN CALL bisect_left WITH parameters range FROM zero TO the LENGTH OF list_of_numbers AND target True AND key function f
    END FUNCTION
END CLASS
