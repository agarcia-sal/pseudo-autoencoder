CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values as list)
    IF values IS empty
        RETURN None
    END IF
    SET root TO CALL TreeNode WITH element at position zero of values
    SET i TO one
    SET queue TO CALL deque
    CALL APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO CALL popleft ON queue
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.left TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.right TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUALS q.val
        RETURN False
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION verticalOrder(root)
        IF root IS None
            RETURN empty list
        END IF
        SET column_table TO CALL defaultdict WITH list
        SET queue TO CALL deque WITH list containing tuple of root AND zero
        SET min_column TO zero
        SET max_column TO zero
        WHILE queue IS NOT empty
            SET tuple node AND column_index TO CALL popleft ON queue
            IF node IS NOT None
                CALL APPEND node.val TO element at position column_index of column_table
                SET min_column TO the MINIMUM OF min_column AND column_index
                SET max_column TO the MAXIMUM OF max_column AND column_index
                CALL APPEND tuple of node.left AND column_index MINUS one TO queue
                CALL APPEND tuple of node.right AND column_index PLUS one TO queue
            END IF
        END WHILE
        SET result TO empty list
        FOR index FROM min_column TO max_column
            APPEND element at position index of column_table TO result
        END FOR
        RETURN result
    END FUNCTION
END CLASS
