CLASS Solution
    FUNCTION earliestFullBloom(plantTime, growTime)
        SET tasks TO CALL combineAndSortByGrowTime WITH plantTime AND growTime
        SET current_day TO zero
        SET earliest_bloom TO zero
        FOR each pair of plant AND grow IN tasks
            INCREMENT current_day BY plant
            SET bloom_day TO current_day PLUS grow
            SET earliest_bloom TO the greater value BETWEEN earliest_bloom AND bloom_day
        END FOR
        RETURN earliest_bloom
    END FUNCTION

    FUNCTION combineAndSortByGrowTime WITH plantTime AND growTime
        PERFORM auxiliary function to combine plantTime and growTime INTO paired list
        SORT paired list BY second element IN descending order
        RETURN sorted list
    END FUNCTION
END CLASS
