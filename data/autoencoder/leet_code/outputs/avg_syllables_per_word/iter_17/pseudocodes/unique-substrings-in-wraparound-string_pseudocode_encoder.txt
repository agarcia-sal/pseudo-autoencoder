CLASS Solution
    FUNCTION findSubstringInWraproundString(string s)
        SET max_length_dictionary TO a new dictionary mapping each character from a to z TO zero
        SET current_length TO zero

        FOR index FROM zero TO the LENGTH OF s MINUS one
            IF index IS GREATER THAN zero AND 
               ( the UNICODE code of element at position index of s MINUS the UNICODE code of element at position index MINUS one of s EQUALS one 
                 OR ( element at position index of s EQUALS character a AND element at position index MINUS one of s EQUALS character z ) )
                INCREMENT current_length BY one
            ELSE
                SET current_length TO one
            END IF

            SET value_at_key_of_max_length_dictionary FOR element at position index of s 
                TO the MAXIMUM OF value_at_key_of_max_length_dictionary FOR element at position index of s AND current_length
        END FOR

        RETURN the SUM OF all values in max_length_dictionary
    END FUNCTION
END CLASS
