CLASS Solution
    FUNCTION spiralOrder(matrix)
        IF matrix IS empty OR element at position zero of matrix IS empty
            RETURN empty list
        END IF
        SET result TO empty list
        SET top TO zero
        SET bottom TO the LENGTH OF matrix MINUS one
        SET left TO zero
        SET right TO the LENGTH OF element at position zero of matrix MINUS one
        WHILE top IS LESS THAN OR EQUAL TO bottom AND left IS LESS THAN OR EQUAL TO right
            FOR index FROM left TO right
                APPEND element at position top of matrix at position index TO result
            END FOR
            INCREMENT top BY one
            FOR index FROM top TO bottom
                APPEND element at position index of matrix at position right TO result
            END FOR
            DECREMENT right BY one
            IF top IS LESS THAN OR EQUAL TO bottom
                FOR index FROM right DOWN TO left
                    APPEND element at position bottom of matrix at position index TO result
                END FOR
                DECREMENT bottom BY one
            END IF
            IF left IS LESS THAN OR EQUAL TO right
                FOR index FROM bottom DOWN TO top
                    APPEND element at position index of matrix at position left TO result
                END FOR
                INCREMENT left BY one
            END IF
        END WHILE
        RETURN result
    END FUNCTION
END CLASS
