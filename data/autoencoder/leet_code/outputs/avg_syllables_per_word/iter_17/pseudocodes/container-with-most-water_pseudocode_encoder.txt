CLASS Solution
    FUNCTION maxArea(height_list)
        SET left_pointer TO zero
        SET right_pointer TO the LENGTH OF height_list MINUS one
        SET maximum_area TO zero
        
        WHILE left_pointer LESS THAN right_pointer holds
            SET current_area TO the MINIMUM OF element at position left_pointer of height_list AND element at position right_pointer of height_list MULTIPLIED BY right_pointer MINUS left_pointer
            SET maximum_area TO the GREATER OF maximum_area AND current_area
            
            IF element at position left_pointer of height_list LESS THAN element at position right_pointer of height_list
                INCREMENT left_pointer BY one
            ELSE
                DECREMENT right_pointer BY one
            END IF
        END WHILE
        
        RETURN maximum_area
    END FUNCTION
END CLASS
