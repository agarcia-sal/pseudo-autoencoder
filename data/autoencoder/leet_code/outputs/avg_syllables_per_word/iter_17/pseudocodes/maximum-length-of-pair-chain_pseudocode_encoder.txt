CLASS Solution
    FUNCTION findLongestChain(pairs)
        CALL pairs.sort WITH key AS auxiliary_function_to_get_second_element
        SET current_end TO negative_infinity
        SET chain_length TO zero
        FOR each pair IN pairs
            SET start TO element at position zero of pair
            SET end TO element at position one of pair
            IF start GREATER THAN current_end
                SET current_end TO end
                INCREMENT chain_length BY one
            END IF
        END FOR
        RETURN chain_length
    END FUNCTION
END CLASS
