CLASS Solution
    FUNCTION leastOpsExpressTarget(x integer, target integer)
        FUNCTION dp(target integer)
            IF target EQUALS zero
                RETURN negative one
            END IF

            IF target LESS THAN x
                RETURN the MINIMUM BETWEEN two MULTIPLIED BY target MINUS one AND two MULTIPLIED BY (x MINUS target)
            END IF

            SET power TO one
            WHILE x RAISED TO THE power IS LESS THAN OR EQUAL TO target
                INCREMENT power BY one
            END WHILE
            DECREMENT power BY one

            SET result TO dp(target MINUS x RAISED TO THE power) PLUS power

            IF x RAISED TO THE (power PLUS one) MINUS target IS LESS THAN target
                SET result TO the MINIMUM BETWEEN result AND dp(x RAISED TO THE (power PLUS one) MINUS target) PLUS power PLUS one
            END IF

            RETURN result
        END FUNCTION

        RETURN dp(target)
    END FUNCTION
END CLASS
