CLASS Solution
    FUNCTION smallestGoodBase(n_string)
        SET n TO CALL integer conversion WITH n_string
        SET max_length TO the INTEGER PART OF the LOGARITHM BASE two OF n INCREMENTED BY one

        FOR m FROM max_length DOWN TO two
            SET k TO the INTEGER PART OF n RAISED TO THE POWER OF one DIVIDED BY m MINUS one
            IF k LESS THAN two
                CONTINUE TO next iteration
            END IF

            SET numerator TO k RAISED TO THE POWER OF m MINUS one
            SET denominator TO k MINUS one
            SET num TO numerator DIVIDED BY denominator

            IF num EQUALS n
                RETURN k AS STRING
            END IF
        END FOR

        RETURN n MINUS one AS STRING
    END FUNCTION
END CLASS
