CLASS Solution
    FUNCTION stoneGameVIII(list_of_stones)
        SET prefix_sum TO CALL initialize_prefix_sum_with_zeroes WITH the LENGTH OF list_of_stones
        SET element at position zero of prefix_sum TO element at position zero of list_of_stones
        
        FOR index FROM one TO the LENGTH OF list_of_stones MINUS one
            SET element at position index of prefix_sum TO element at position index MINUS one of prefix_sum PLUS element at position index of list_of_stones
        END FOR
        
        SET dp TO element at position the LENGTH OF list_of_stones MINUS one of prefix_sum
        
        FOR index FROM the LENGTH OF list_of_stones MINUS two DOWN TO one
            SET dp TO the MAXIMUM OF dp AND element at position index of prefix_sum MINUS dp
        END FOR
        
        RETURN dp
    END FUNCTION
END CLASS
