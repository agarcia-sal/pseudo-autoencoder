CLASS Solution
    FUNCTION superpalindromesInRange(left_string, right_string)
        FUNCTION is_palindrome(number)
            RETURN the STRING REPRESENTATION OF number EQUALS the REVERSE OF the STRING REPRESENTATION OF number
        END FUNCTION

        FUNCTION generate_palindromes(limit_value)
            SET list_of_palindromes TO empty list
            FOR integer_index FROM one TO one hundred thousand MINUS one
                SET string_representation TO the STRING REPRESENTATION OF integer_index
                APPEND the INTEGER CONVERSION OF string_representation CONCATENATED WITH the SUBSTRING FROM position LENGTH of string_representation MINUS two DOWN TO zero STEP minus one TO list_of_palindromes
                APPEND the INTEGER CONVERSION OF string_representation CONCATENATED WITH the REVERSE OF string_representation TO list_of_palindromes
            END FOR
            RETURN the LIST OF palindromes WHERE each palindrome IS LESS THAN OR EQUAL TO limit_value FROM list_of_palindromes
        END FUNCTION

        SET left TO the INTEGER CONVERSION OF left_string
        SET right TO the INTEGER CONVERSION OF right_string
        SET limit TO the INTEGER CONVERSION OF right RAISED TO the POWER OF one HALF PLUS one
        SET palindromes TO CALL generate_palindromes WITH limit

        SET count TO zero
        FOR each p IN palindromes
            SET square TO p MULTIPLIED BY p
            IF left IS LESS THAN OR EQUAL TO square AND square IS LESS THAN OR EQUAL TO right AND CALL is_palindrome WITH square
                INCREMENT count BY one
            END IF
        END FOR

        RETURN count
    END FUNCTION
END CLASS
