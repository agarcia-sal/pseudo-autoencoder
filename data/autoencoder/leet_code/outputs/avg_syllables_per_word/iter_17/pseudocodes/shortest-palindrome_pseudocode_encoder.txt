CLASS Solution
    FUNCTION shortestPalindrome(s)
        IF s IS EQUALS to empty string
            RETURN s
        END IF

        FUNCTION is_palindrome(check_str)
            RETURN check_str IS EQUALS TO the REVERSE OF check_str
        END FUNCTION

        FOR i FROM the LENGTH OF s TO zero DECREMENTED BY one
            IF is_palindrome(substring from position zero TO position i MINUS one of s)
                SET to_add TO the REVERSE OF substring from position i TO the END OF s
                RETURN to_add PLUS s
            END IF
        END FOR
    END FUNCTION
END CLASS
