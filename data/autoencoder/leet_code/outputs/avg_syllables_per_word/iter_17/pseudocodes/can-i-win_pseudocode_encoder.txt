CLASS Solution
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)
        IF maxChoosableInteger MULTIPLIED BY maxChoosableInteger PLUS one DIVIDED BY two LESS THAN desiredTotal
            RETURN false
        END IF

        IF desiredTotal LESS THAN OR EQUAL TO zero
            RETURN true
        END IF

        SET memo TO an auxiliary dictionary storing boolean results

        FUNCTION can_win(current_total, used_numbers)
            IF current_total AND used_numbers AS a tuple IS IN memo
                RETURN element at position current_total AND used_numbers AS a tuple OF memo
            END IF

            FOR number FROM one TO maxChoosableInteger
                IF used_numbers BITWISE AND one LEFT SHIFTED BY number EQUALS zero
                    IF current_total PLUS number GREATER THAN OR EQUAL TO desiredTotal OR NOT CALL can_win WITH current_total PLUS number AND used_numbers BITWISE OR one LEFT SHIFTED BY number
                        SET element at position current_total AND used_numbers AS a tuple OF memo TO true
                        RETURN true
                    END IF
                END IF
            END FOR

            SET element at position current_total AND used_numbers AS a tuple OF memo TO false
            RETURN false
        END FUNCTION

        RETURN CALL can_win WITH zero AND zero
    END FUNCTION
END CLASS
