CLASS Solution
    FUNCTION isMatch(string s, string p)
        SET dp TO the RESULT OF create_boolean_matrix WITH parameters the LENGTH OF s PLUS one AND the LENGTH OF p PLUS one

        SET element at position zero of list dp TO the RESULT OF create_boolean_list WITH parameter the LENGTH OF p PLUS one
        SET element at position zero and zero of dp TO True

        FOR index j FROM one TO the LENGTH OF p
            IF element at position index j MINUS one of string p EQUALS star
                SET element at position zero and index j of dp TO the ELEMENT at position zero and index j MINUS two of dp
            END IF
        END FOR

        FOR index i FROM one TO the LENGTH OF s
            FOR index j FROM one TO the LENGTH OF p
                IF element at position index j MINUS one of string p EQUALS element at position index i MINUS one of string s OR element at position index j MINUS one of string p EQUALS dot
                    SET element at position index i and index j of dp TO the ELEMENT at position index i MINUS one and index j MINUS one of dp
                ELSE IF element at position index j MINUS one of string p EQUALS star
                    SET element at position index i and index j of dp TO the ELEMENT at position index i and index j MINUS two of dp

                    IF element at position index j MINUS two of string p EQUALS element at position index i MINUS one of string s OR element at position index j MINUS two of string p EQUALS dot
                        SET element at position index i and index j of dp TO element at position index i and index j of dp OR element at position index i MINUS one and index j of dp
                    END IF
                END IF
            END FOR
        END FOR

        RETURN element at position the LENGTH OF s and the LENGTH OF p of dp
    END FUNCTION

END FUNCTION

    RETURN Auxiliary FUNCTION TO initialize boolean list WITH length
END FUNCTION
