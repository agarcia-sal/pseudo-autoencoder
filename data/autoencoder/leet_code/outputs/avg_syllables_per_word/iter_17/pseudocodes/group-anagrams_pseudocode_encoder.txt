CLASS Solution
    FUNCTION groupAnagrams(list_of_strings)
        SET anagrams_dictionary TO CALL create_default_dictionary WITH parameter list
        FOR each string_element IN list_of_strings
            SET sorted_characters_list TO CALL sort_characters_in_string WITH parameter string_element
            SET sorted_string_key TO CALL join_characters_list_into_string WITH parameter sorted_characters_list
            CALL append string_element TO list AT key sorted_string_key OF anagrams_dictionary
        END FOR
        RETURN CALL convert_dictionary_values_to_list WITH parameter anagrams_dictionary
    END FUNCTION
END CLASS
