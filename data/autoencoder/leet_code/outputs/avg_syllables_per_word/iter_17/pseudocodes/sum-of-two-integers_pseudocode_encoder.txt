CLASS Solution
    FUNCTION getSum(a integer, b integer) RETURNS integer
        SET MAX TO the hexadecimal value sevenFFFFFFF
        SET mask TO the hexadecimal value FFFFFFFF
        WHILE b IS NOT EQUAL TO zero
            SET carry TO the bitwise AND of a AND b MULTIPLIED BY two
            SET a TO the bitwise XOR of a AND b THEN bitwise AND WITH mask
            SET b TO the bitwise AND of carry AND mask
        END WHILE
        IF a IS LESS THAN OR EQUAL TO MAX
            RETURN a
        ELSE
            RETURN the bitwise NEGATION of the bitwise XOR of a AND mask
        END IF
    END FUNCTION
END CLASS
