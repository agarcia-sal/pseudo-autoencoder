CLASS Solution
    FUNCTION countTriplets(list_of_numbers)
        CALL initialize_default_dictionary WITH integer RETURNING pair_and_count_mapping

        FOR each first_number IN list_of_numbers
            FOR each second_number IN list_of_numbers
                SET bitwise_and_result TO first_number BITWISE AND second_number
                INCREMENT element at position bitwise_and_result OF pair_and_count_mapping BY one
            END FOR
        END FOR

        SET total_triplet_count TO zero
        FOR each current_number IN list_of_numbers
            FOR each bitwise_and_key AND associated_count IN pair_and_count_mapping
                IF bitwise_and_key BITWISE AND current_number EQUALS zero
                    INCREMENT total_triplet_count BY associated_count
                END IF
            END FOR
        END FOR

        RETURN total_triplet_count
    END FUNCTION
END CLASS
