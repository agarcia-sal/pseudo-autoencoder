        SET sign TO negative one IF x_integer LESS THAN zero ELSE positive one
        SET reversed_x TO CALL AbstractReverseDigits WITH absolute VALUE OF x_integer

        IF result LESS THAN negative two RAISED TO the power OF thirty one OR result GREATER THAN positive two RAISED TO the power OF thirty one MINUS one

        END IF
        RETURN result
