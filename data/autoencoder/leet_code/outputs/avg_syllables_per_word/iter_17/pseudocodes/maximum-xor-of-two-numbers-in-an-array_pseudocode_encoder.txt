CLASS Solution
    FUNCTION findMaximumXOR(list_of_numbers)
        SET maximum_xor TO zero
        FOR bit_position FROM thirty one TO zero STEP negative one
            SET maximum_xor TO maximum_xor MULTIPLIED BY two
            SET current_candidate TO maximum_xor PLUS one
            SET prefixes TO CALL CreatePrefixSet WITH list_of_numbers AND bit_position
            FOR each prefix_element IN prefixes
                IF current_candidate EXCLUSIVE OR prefix_element IS IN prefixes
                    SET maximum_xor TO current_candidate
                    BREAK
                END IF
            END FOR
        END FOR
        RETURN maximum_xor
    END FUNCTION

    FUNCTION CreatePrefixSet(list_of_numbers, bit_position)
        SET prefix_set TO empty set
        FOR each number IN list_of_numbers
            SET shifted_prefix TO number DIVIDED BY two RAISED TO the POWER OF bit_position
            ADD shifted_prefix TO prefix_set
        END FOR
        RETURN prefix_set
    END FUNCTION
END CLASS
