    FUNCTION nextGreaterElement(n_integer)
        CALL convert_number_to_list_of_digits WITH n_integer ASSIGN TO digits_list
        SET length_of_digits TO the LENGTH OF digits_list

        SET i_index TO length_of_digits MINUS two
        WHILE i_index GREATER THAN OR EQUAL TO zero AND element at position i_index of digits_list GREATER THAN OR EQUAL TO element at position i_index PLUS one of digits_list holds
            DECREMENT i_index BY one
        END WHILE

        IF i_index EQUALS negative one
            RETURN negative one
        END IF

        SET j_index TO length_of_digits MINUS one
        WHILE element at position j_index of digits_list LESS THAN OR EQUAL TO element at position i_index of digits_list holds
            DECREMENT j_index BY one
        END WHILE

        CALL swap_elements AT positions i_index AND j_index OF digits_list

        CALL reverse_sublist FROM position i_index PLUS one TO end OF digits_list

        CALL convert_list_of_digits_to_number WITH digits_list ASSIGN TO result_integer

        IF result_integer GREATER THAN two RAISED TO THE power of thirty one MINUS one
            RETURN negative one
        END IF

        RETURN result_integer
    END FUNCTION
