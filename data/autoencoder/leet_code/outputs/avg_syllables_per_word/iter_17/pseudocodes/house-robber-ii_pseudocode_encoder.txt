CLASS Solution
    FUNCTION rob(list_of_numbers)
        FUNCTION rob_linear(list_of_houses)
            IF list_of_houses IS empty
                RETURN zero
            END IF
            IF the LENGTH OF list_of_houses EQUALS one
                RETURN element at position zero of list_of_houses
            END IF

            SET dp_list TO auxiliary FUNCTION create_list_of_zeros WITH parameter the LENGTH OF list_of_houses
            SET element at position zero of dp_list TO element at position zero of list_of_houses
            SET element at position one of dp_list TO auxiliary FUNCTION maximum_of WITH parameters element at position zero of list_of_houses AND element at position one of list_of_houses

            FOR index FROM two TO the LENGTH OF list_of_houses MINUS one
                SET element at position index of dp_list TO auxiliary FUNCTION maximum_of WITH parameters element at position index MINUS one of dp_list AND the SUM OF element at position index MINUS two of dp_list AND element at position index of list_of_houses
            END FOR

            RETURN element at position the LENGTH OF dp_list MINUS one of dp_list
        END FUNCTION

        IF list_of_numbers IS empty
            RETURN zero
        END IF
        IF the LENGTH OF list_of_numbers EQUALS one
            RETURN element at position zero of list_of_numbers
        END IF

        RETURN auxiliary FUNCTION maximum_of WITH parameters CALL rob_linear WITH parameter substring from position zero TO the LENGTH OF list_of_numbers MINUS two of list_of_numbers AND CALL rob_linear WITH parameter substring from position one TO the LENGTH OF list_of_numbers MINUS one of list_of_numbers
    END FUNCTION
END CLASS
