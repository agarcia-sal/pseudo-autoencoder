CLASS Solution
    FUNCTION decodeString(s)
        SET stack TO empty list
        SET a TO empty string
        SET d TO empty string

        FOR each character IN s
            IF character IS a digit
                APPEND character TO d
            ELSE IF character IS an alphabetic letter
                APPEND character TO a
            ELSE IF character EQUALS left bracket
                CALL append WITH tuple of a AND the integer value of d ON stack
                SET a TO empty string
                SET d TO empty string
            ELSE IF character EQUALS right bracket
                SET p, n TO the last element popped FROM stack
                SET a TO p PLUS a MULTIPLIED BY n
            END IF
        END FOR

        RETURN a
    END FUNCTION
END CLASS
