CLASS Solution
    FUNCTION dailyTemperatures(list_of_temperatures)
        SET length_of_temperatures TO the LENGTH OF list_of_temperatures
        SET answer_list TO a new list of zero REPEATED length_of_temperatures TIMES
        SET stack_list TO an empty list

        FOR index FROM zero TO length_of_temperatures MINUS one
            WHILE stack_list IS NOT empty AND element at position index of list_of_temperatures IS GREATER THAN element at position element at position the last position of stack_list of list_of_temperatures
                SET previous_index TO the REMOVAL OF the last element FROM stack_list
                SET element at position previous_index of answer_list TO index MINUS previous_index
            END WHILE
            APPEND index TO stack_list
        END FOR

        RETURN answer_list
    END FUNCTION
END CLASS
