CLASS Solution
    FUNCTION multiply(num1, num2)
        IF num1 EQUALS zero OR num2 EQUALS zero
            RETURN zero
        END IF

        SET result_list TO a list of zeroes with LENGTH EQUAL TO the LENGTH OF num1 PLUS the LENGTH OF num2

        SET num1 TO the REVERSE OF num1
        SET num2 TO the REVERSE OF num2

        FOR index_i FROM zero TO the LENGTH OF num1 MINUS one
            FOR index_j FROM zero TO the LENGTH OF num2 MINUS one
                SET mul TO the ORDINAL VALUE OF the element at position index_i of num1 MINUS the ORDINAL VALUE OF zero MULTIPLIED BY the ORDINAL VALUE OF the element at position index_j of num2 MINUS the ORDINAL VALUE OF zero
                SET p1 TO index_i PLUS index_j
                SET p2 TO index_i PLUS index_j PLUS one
                SET sum_value TO mul PLUS the element at position p1 of result_list

                SET the element at position p1 of result_list TO sum_value MODULO ten
                INCREMENT the element at position p2 of result_list BY sum_value DIVIDED BY ten
            END FOR
        END FOR

        CALL convert_result_array_to_string WITH result_list TO GET result_str
        RETURN result_str

    END FUNCTION

    FUNCTION convert_result_array_to_string(result_list)
        SET reversed_list TO the REVERSE OF result_list
        SET result_string TO the STRING JOIN OF reversed_list CONVERTED TO strings
        SET stripped_string TO result_string WITH LEADING zeroes REMOVED
        RETURN stripped_string
    END FUNCTION
END CLASS
