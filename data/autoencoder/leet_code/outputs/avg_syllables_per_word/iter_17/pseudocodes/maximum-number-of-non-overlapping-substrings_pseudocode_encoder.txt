CLASS Solution
    FUNCTION maxNumOfSubstrings(string_s)
        SET dictionary_first TO an empty dictionary
        SET dictionary_last TO an empty dictionary
        FOR each index i AND character c IN the enumeration of string_s
            IF character c IS NOT IN dictionary_first
                SET element at key c of dictionary_first TO index i
            END IF
            SET element at key c of dictionary_last TO index i
        END FOR
        
        SET list_intervals TO an empty list
        FOR each character c IN the keys of dictionary_first
            SET variable start TO element at key c of dictionary_first
            SET variable end TO element at key c of dictionary_last
            SET variable i TO start
            WHILE variable i IS LESS THAN OR EQUAL TO variable end
                IF element at key element at position i of string_s IN dictionary_first IS LESS THAN start
                    BREAK
                END IF
                SET variable end TO the GREATER OF variable end AND element at key element at position i of string_s IN dictionary_last
                INCREMENT variable i BY one
            END WHILE
            IF variable i IS GREATER THAN variable end
                APPEND tuple of start AND end TO list_intervals
            END IF
        END FOR
        
        CALL the sort method ON list_intervals WITH a key FUNCTION that RETURNS the element at position one of the tuple
        
        SET list_result TO an empty list
        SET variable prev_end TO negative one
        FOR each tuple start AND end IN list_intervals
            IF start IS GREATER THAN prev_end
                APPEND the substring from position start TO end inclusive of string_s TO list_result
                SET variable prev_end TO end
            END IF
        END FOR
        
        RETURN list_result
    END FUNCTION
END CLASS
