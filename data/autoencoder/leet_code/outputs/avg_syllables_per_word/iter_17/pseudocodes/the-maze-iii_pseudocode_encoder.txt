CLASS Solution
    FUNCTION findShortestWay(maze AS List of List of integers, ball AS List of integers, hole AS List of integers) RETURNS string
        SET directions TO a list of tuples composed of zero and one and right, one and zero and down, zero and minus one and left, minus one and zero and up
        SET m TO the LENGTH OF maze
        SET n TO the LENGTH OF element at position zero of maze
        SET start TO the tuple formed from ball
        SET hole TO the tuple formed from hole
        SET pq TO a list containing a tuple composed of zero, empty string, and start
        SET visited TO an empty set

        WHILE pq holds
            CALL heappop WITH pq AND ASSIGN returned tuple TO dist, path, and position unpacked as x, y

            IF x EQUALS element at position zero of hole AND y EQUALS element at position one of hole
                RETURN path
            END IF

            IF tuple composed of x and y IS IN visited
                CONTINUE to next iteration
            END IF

            ADD tuple composed of x and y TO visited

            FOR each dx, dy, and direction IN directions
                SET new_x TO x
                SET new_y TO y
                SET count TO zero

                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx of element at position new_y PLUS dy of maze EQUALS zero
                    INCREMENT new_x BY dx
                    INCREMENT new_y BY dy
                    INCREMENT count BY one

                    IF new_x EQUALS element at position zero of hole AND new_y EQUALS element at position one of hole
                        BREAK
                    END IF
                END WHILE

                IF tuple composed of new_x and new_y IS NOT IN visited
                    CALL heappush WITH pq AND a tuple composed of dist PLUS count, path CONCATENATED WITH direction, and tuple composed of new_x and new_y
                END IF
            END FOR
        END WHILE

        RETURN the string impossible
    END FUNCTION
END CLASS
