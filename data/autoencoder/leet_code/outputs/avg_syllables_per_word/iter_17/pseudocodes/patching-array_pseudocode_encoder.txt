CLASS Solution
    FUNCTION minPatches(nums, n)
        SET missing TO one
        SET patches TO zero
        SET i TO zero
        WHILE missing LESS THAN OR EQUAL TO n holds
            IF i LESS THAN the LENGTH OF nums AND element at position i of nums LESS THAN OR EQUAL TO missing
                INCREMENT missing BY element at position i of nums
                INCREMENT i BY one
            ELSE
                INCREMENT missing BY missing
                INCREMENT patches BY one
            END IF
        END WHILE
        RETURN patches
    END FUNCTION
END CLASS
