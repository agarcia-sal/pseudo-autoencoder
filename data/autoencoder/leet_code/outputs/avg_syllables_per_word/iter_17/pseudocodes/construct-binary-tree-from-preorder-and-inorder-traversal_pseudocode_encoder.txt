CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN None
    END IF
    SET root TO CALL TreeNode WITH element at position zero of values
    SET i TO one
    SET queue TO CALL deque WITH no parameters
    CALL APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO CALL popleft ON queue
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.left TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.right TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION buildTree(self, preorder, inorder)
        IF preorder IS empty OR inorder IS empty
            RETURN None
        END IF
        SET root_val TO element at position zero of preorder
        SET root TO CALL TreeNode WITH root_val
        SET root_index_inorder TO the POSITION OF root_val IN inorder
        SET root.left TO CALL self.buildTree WITH substring from position one TO root_index_inorder of preorder AND substring from position zero TO root_index_inorder MINUS one of inorder
        SET root.right TO CALL self.buildTree WITH substring from position root_index_inorder PLUS one TO end of preorder AND substring from position root_index_inorder PLUS one TO end of inorder
        RETURN root
    END FUNCTION
END CLASS
