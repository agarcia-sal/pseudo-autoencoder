CLASS Solution
    FUNCTION numSubarrayProductLessThanK(list_of_numbers nums, integer k)
        IF k LESS THAN OR EQUAL TO one
            RETURN zero
        END IF
        SET product TO one
        SET left TO zero
        SET count TO zero
        FOR index right FROM zero TO the LENGTH OF list_of_numbers nums MINUS one
            SET product TO product MULTIPLIED BY element at position right of list_of_numbers nums
            WHILE product GREATER THAN OR EQUAL TO k AND left LESS THAN OR EQUAL TO right holds
                SET product TO product DIVIDED BY element at position left of list_of_numbers nums
                INCREMENT left BY one
            END WHILE
            INCREMENT count BY right MINUS left PLUS one
        END FOR
        RETURN count
    END FUNCTION
END CLASS
