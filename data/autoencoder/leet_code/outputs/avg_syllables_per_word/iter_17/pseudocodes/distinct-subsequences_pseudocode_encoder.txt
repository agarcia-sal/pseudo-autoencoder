CLASS Solution
    FUNCTION numDistinct(string_s, string_t)
        SET length_m TO the LENGTH OF string_s
        SET length_n TO the LENGTH OF string_t

        SET dp TO CALL CreateTwoDimensionalArray WITH parameters length_m PLUS one AND length_n PLUS one

        FOR index_i FROM zero TO length_m
            SET element at position index_i AND position zero of dp TO one
        END FOR

        FOR index_i FROM one TO length_m
            FOR index_j FROM one TO length_n
                IF element at position index index_i MINUS one of string_s EQUALS element at position index index_j MINUS one of string_t
                    SET element at position index_i AND position index_j of dp TO element at position index index_i MINUS one AND position index index_j MINUS one of dp PLUS element at position index index_i MINUS one AND position index_j of dp
                ELSE
                    SET element at position index_i AND position index_j of dp TO element at position index index_i MINUS one AND position index_j of dp
                END IF
            END FOR
        END FOR

        RETURN element at position length_m AND position length_n of dp
    END FUNCTION
END CLASS
