CLASS Solution
    FUNCTION minimumDifference(nums)
        SET n TO the LENGTH OF nums DIVIDED BY two
        SET total_sum TO the SUM OF nums
        SET target TO total_sum DIVIDED BY two
        SET min_diff TO infinity

        SET left_half TO the SUBLIST from position zero TO position n MINUS one of nums
        SET right_half TO the SUBLIST from position n TO the END OF nums

        SET left_sums TO auxiliary FUNCTION generateEmptyDictionaries WITH parameter n PLUS one
        SET right_sums TO auxiliary FUNCTION generateEmptyDictionaries WITH parameter n PLUS one

        FOR i FROM zero TO n
            FOR each combo IN all combinations OF left_half TAKEN i AT A TIME
                SET subset_sum TO the SUM OF combo
                IF subset_sum IS NOT A KEY IN the dictionary AT position i of left_sums
                    SET left_sums at position i FOR KEY subset_sum TO zero
                END IF
                INCREMENT left_sums at position i FOR KEY subset_sum BY one
            END FOR

            FOR each combo IN all combinations OF right_half TAKEN i AT A TIME
                SET subset_sum TO the SUM OF combo
                IF subset_sum IS NOT A KEY IN the dictionary AT position i of right_sums
                    SET right_sums at position i FOR KEY subset_sum TO zero
                END IF
                INCREMENT right_sums at position i FOR KEY subset_sum BY one
            END FOR
        END FOR

        FOR i FROM zero TO n
            SET left_values TO the SORTED LIST OF KEYS OF the dictionary AT position i of left_sums
            SET right_values TO the SORTED LIST OF KEYS OF the dictionary AT position n MINUS i of right_sums

            SET j TO zero
            SET k TO the LENGTH OF right_values MINUS one

            WHILE j IS LESS THAN the LENGTH OF left_values AND k IS GREATER THAN OR EQUAL TO zero
                SET current_sum TO the element at position j of left_values PLUS the element at position k of right_values
                SET min_diff TO the MINIMUM OF min_diff AND the ABSOLUTE VALUE OF total_sum MINUS two MULTIPLIED BY current_sum

                IF current_sum IS LESS THAN target
                    INCREMENT j BY one
                ELSE IF current_sum IS GREATER THAN target
                    DECREMENT k BY one
                ELSE
                    RETURN zero
                END IF
            END WHILE
        END FOR

        RETURN min_diff
    END FUNCTION
END CLASS
