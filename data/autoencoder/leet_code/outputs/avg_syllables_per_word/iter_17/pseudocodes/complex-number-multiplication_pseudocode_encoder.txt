CLASS Solution
    FUNCTION complexNumberMultiply(num1, num2)
        FUNCTION parse_complex(num)
            SET parts TO the RESULT OF splitting string num BY the character plus
            SET real TO the CONVERSION TO integer OF element at position zero OF parts
            SET imaginary TO the CONVERSION TO integer OF substring from position zero TO the POSITION before last OF element at position one OF parts
            RETURN real, imaginary
        END FUNCTION

        SET real1, imag1 TO CALL parse_complex WITH num1
        SET real2, imag2 TO CALL parse_complex WITH num2

        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2

        RETURN string REPRESENTATION OF real_part CONCATENATED WITH plus CONCATENATED WITH string REPRESENTATION OF imaginary_part CONCATENATED WITH the letter i
    END FUNCTION
END CLASS
