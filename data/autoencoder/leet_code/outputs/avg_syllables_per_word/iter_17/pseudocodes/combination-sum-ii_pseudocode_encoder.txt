CLASS Solution
    FUNCTION combinationSum2(candidates, target)
        CALL candidates SORT
        SET result TO empty list
        FUNCTION backtrack(start, target_value, path_list)
            IF target_value EQUALS zero THEN
                APPEND path_list TO result
                RETURN
            END IF
            IF target_value LESS THAN zero THEN
                RETURN
            END IF
            FOR index FROM start TO the LENGTH OF candidates MINUS one
                IF index GREATER THAN start AND element at position index of candidates EQUALS element at position index MINUS one of candidates THEN
                    CONTINUE
                END IF
                CALL backtrack WITH PARAMETERS index PLUS one, target_value MINUS element at position index of candidates, path_list PLUS list containing element at position index of candidates
            END FOR
        END FUNCTION
        CALL backtrack WITH PARAMETERS zero, target, empty list
        RETURN result
    END FUNCTION
END CLASS
