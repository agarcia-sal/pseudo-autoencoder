CLASS Solution
    FUNCTION rotate(matrix)
        SET n TO the LENGTH OF matrix
        FOR i FROM zero TO n MINUS one
            FOR j FROM i TO n MINUS one
                SET temp TO element at position i of matrix AT position j
                SET element at position i of matrix AT position j TO element at position j of matrix AT position i
                SET element at position j of matrix AT position i TO temp
            END FOR
        END FOR
        FOR i FROM zero TO n MINUS one
            CALL reverse_row WITH parameter element at position i of matrix
        END FOR
    END FUNCTION

    FUNCTION reverse_row(row)
        SET start TO zero
        SET end TO the LENGTH OF row MINUS one
        WHILE start LESS THAN end HOLDS
            SET temp TO element at position start of row
            SET element at position start of row TO element at position end of row
            SET element at position end of row TO temp
            INCREMENT start BY one
            DECREMENT end BY one
        END WHILE
    END FUNCTION
END CLASS
