CLASS Solution
    FUNCTION rob(list_of_numbers)
        IF the list_of_numbers IS empty
            RETURN zero
        END IF

        IF the LENGTH OF list_of_numbers EQUALS one
            RETURN the element at position zero of list_of_numbers
        END IF

        CALL initialize_list_with_zeros WITH parameters the LENGTH OF list_of_numbers ASSIGN TO dp
        SET element at position zero of dp TO element at position zero of list_of_numbers
        SET element at position one of dp TO the MAXIMUM OF element at position zero of list_of_numbers AND element at position one of list_of_numbers

        FOR index FROM two TO the LENGTH OF list_of_numbers MINUS one
            SET element at position index of dp TO the MAXIMUM OF element at position index MINUS one of dp AND element at position index MINUS two of dp PLUS element at position index of list_of_numbers
        END FOR

        RETURN the element at the last position of dp
    END FUNCTION
END CLASS
