CLASS Solution
    FUNCTION getModifiedArray(length, updates)
        SET difference_array TO a list of zeroes WITH size length PLUS one
        FOR each start_index, end_index, increment_value IN updates
            SET element at position start_index of difference_array TO element at position start_index of difference_array PLUS increment_value
            IF end_index PLUS one IS LESS THAN length
                SET element at position end_index PLUS one of difference_array TO element at position end_index PLUS one of difference_array MINUS increment_value
            END IF
        END FOR
        SET result_array TO a list of zeroes WITH size length
        SET element at position zero of result_array TO element at position zero of difference_array
        FOR index FROM one TO length MINUS one
            SET element at position index of result_array TO element at position index MINUS one of result_array PLUS element at position index of difference_array
        END FOR
        RETURN result_array
    END FUNCTION
END CLASS
