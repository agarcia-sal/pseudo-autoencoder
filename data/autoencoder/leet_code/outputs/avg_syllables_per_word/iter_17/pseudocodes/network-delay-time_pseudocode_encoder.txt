CLASS Solution
    FUNCTION networkDelayTime(times AS list OF list OF integer, n AS integer, k AS integer) RETURNS integer
        SET graph TO CALL CreateAdjacencyList WITH parameters times AND n
        SET dist TO CALL InitializeDistanceDictionary WITH parameters n AND k
        SET min_heap TO LIST containing the tuple of zero AND k
        
        WHILE min_heap is NOT empty
            CALL ExtractMinimumFromHeap WITH parameter min_heap ASSIGN TO current_dist AND node
            
            IF current_dist IS GREATER THAN element at position node of dist
                CONTINUE
            END IF
            
            FOR each tuple of neighbor AND weight IN element at position node of graph
                SET distance TO current_dist PLUS weight
                
                IF distance IS LESS THAN element at position neighbor of dist
                    SET element at position neighbor of dist TO distance
                    CALL PushToHeap WITH parameters min_heap AND tuple of distance AND neighbor
                END IF
            END FOR
        END WHILE
        
        SET max_time TO CALL FindMaximumValue WITH parameter dist
        IF max_time IS EQUAL TO infinity
            RETURN negative one
        ELSE
            RETURN max_time
        END IF
    END FUNCTION
