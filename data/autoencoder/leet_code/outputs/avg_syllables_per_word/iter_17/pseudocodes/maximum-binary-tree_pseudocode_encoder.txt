CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF the LENGTH OF values EQUALS zero
        RETURN none
    END IF
    SET root TO NEW TreeNode WITH element at position zero of values
    SET i TO one
    SET queue TO NEW deque
    CALL APPEND root TO queue
    WHILE queue is not empty
        SET node TO CALL popleft ON queue
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none
            SET node.left TO NEW TreeNode WITH element at position i of values
            CALL APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none
            SET node.right TO NEW TreeNode WITH element at position i of values
            CALL APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF p.val NOT EQUALS q.val
        RETURN false
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION constructMaximumBinaryTree(self, nums)
        IF the LENGTH OF nums EQUALS zero
            RETURN none
        END IF
        SET max_val TO CALL max ON nums
        SET max_index TO CALL index OF max_val IN nums
        SET root TO NEW TreeNode WITH max_val
        SET root.left TO CALL self.constructMaximumBinaryTree WITH the SUBLIST FROM position zero TO max_index MINUS one OF nums
        SET root.right TO CALL self.constructMaximumBinaryTree WITH the SUBLIST FROM position max_index PLUS one TO the LENGTH OF nums MINUS one OF nums
        RETURN root
    END FUNCTION
END CLASS
