CLASS Solution
    FUNCTION findRepeatedDnaSequences(string_s)
        ASSIGN sequence_count TO CALL create_default_dictionary FOR integer_counts
        FOR index FROM zero TO the LENGTH OF string_s MINUS nine
            SET sequence TO substring FROM position index TO index PLUS ten OF string_s
            INCREMENT the element AT position sequence OF sequence_count BY one
        END FOR
        ASSIGN repeated_sequences TO CALL collect_sequences_occuring_more_than_once WITH sequence_count
        RETURN repeated_sequences
    END FUNCTION
END CLASS
