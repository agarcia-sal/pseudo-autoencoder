CLASS Solution
    FUNCTION getMinDistSum(positions)
        SET number_of_positions TO the LENGTH OF positions
        SET x TO zero
        SET y TO zero
        FOR each position_element WITH elements x1 and y1 IN positions
            INCREMENT x BY x1
            INCREMENT y BY y1
        END FOR
        SET x TO x DIVIDED BY number_of_positions
        SET y TO y DIVIDED BY number_of_positions
        SET decay TO zero point nine nine nine
        SET epsilon TO zero point zero zero zero zero zero one
        SET alpha TO zero point five
        WHILE condition always holds
            SET gradient_x TO zero
            SET gradient_y TO zero
            SET total_distance TO zero
            FOR each position_element WITH elements x1 and y1 IN positions
                SET difference_x TO x MINUS x1
                SET difference_y TO y MINUS y1
                SET distance TO CALL sqrt WITH parameters difference_x MULTIPLIED BY difference_x PLUS difference_y MULTIPLIED BY difference_y
                INCREMENT gradient_x BY difference_x DIVIDED BY distance PLUS zero point zero zero zero zero zero zero one
                INCREMENT gradient_y BY difference_y DIVIDED BY distance PLUS zero point zero zero zero zero zero zero one
                INCREMENT total_distance BY distance
            END FOR
            SET delta_x TO gradient_x MULTIPLIED BY alpha
            SET delta_y TO gradient_y MULTIPLIED BY alpha
            DECREMENT x BY delta_x
            DECREMENT y BY delta_y
            MULTIPLY alpha BY decay
            IF the ABSOLUTE VALUE OF delta_x IS LESS THAN OR EQUAL TO epsilon AND the ABSOLUTE VALUE OF delta_y IS LESS THAN OR EQUAL TO epsilon
                RETURN total_distance
            END IF
        END WHILE
    END FUNCTION
END CLASS
