CLASS Solution
    FUNCTION numOfArrays(n, m, k)
        SET MODULO_CONSTANT TO one billion PLUS seven
        CALL initialize_three_dimensional_list WITH parameters n PLUS one m PLUS one k PLUS one AND ASSIGN TO dp
        
        FOR max_element FROM one TO m
            SET element at position one OF dp at position max_element at position one TO one
        END FOR
        
        FOR array_length FROM two TO n
            FOR max_element FROM one TO m
                FOR search_cost FROM one TO k
                    FOR smaller_max_element FROM one TO max_element MINUS one
                        SET element at positions array_length max_element search_cost OF dp TO element at positions array_length max_element search_cost OF dp PLUS element at positions array_length MINUS one smaller_max_element search_cost MINUS one OF dp
                        SET element at positions array_length max_element search_cost OF dp TO element at positions array_length max_element search_cost OF dp MODULO MODULO_CONSTANT
                    END FOR
                    SET element at positions array_length max_element search_cost OF dp TO element at positions array_length max_element search_cost OF dp PLUS element at positions array_length MINUS one max_element search_cost OF dp MULTIPLIED BY max_element
                    SET element at positions array_length max_element search_cost OF dp TO element at positions array_length max_element search_cost OF dp MODULO MODULO_CONSTANT
                END FOR
            END FOR
        END FOR

        SET total_result TO zero
        FOR max_element_result FROM one TO m
            SET total_result TO total_result PLUS element at positions n max_element_result k OF dp
            SET total_result TO total_result MODULO MODULO_CONSTANT
        END FOR
        
        RETURN total_result
    END FUNCTION
END CLASS
