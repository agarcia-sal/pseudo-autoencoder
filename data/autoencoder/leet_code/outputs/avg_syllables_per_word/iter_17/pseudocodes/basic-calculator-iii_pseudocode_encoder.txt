CLASS Solution
    FUNCTION calculate(string_s)
        FUNCTION helper(list_s)
            SET stack TO empty list
            SET num TO zero
            SET sign TO plus symbol

            WHILE the LENGTH OF list_s IS GREATER THAN zero
                SET char TO element at position zero of list_s
                REMOVE element at position zero FROM list_s

                IF char IS A DIGIT
                    SET num TO num MULTIPLIED BY ten PLUS the INTEGER VALUE OF char
                END IF

                IF char EQUALS open parenthesis symbol
                    SET num TO CALL helper WITH list_s
                END IF

                IF char IS IN plus symbol OR minus symbol OR multiply symbol OR divide symbol OR open parenthesis symbol OR close parenthesis symbol 
                OR the LENGTH OF list_s EQUALS zero
                    IF sign EQUALS plus symbol
                        APPEND num TO stack
                    ELSE IF sign EQUALS minus symbol
                        APPEND negative num TO stack
                    ELSE IF sign EQUALS multiply symbol
                        SET the element at position the last index of stack TO the element at position the last index of stack MULTIPLIED BY num
                    ELSE IF sign EQUALS divide symbol
                        SET the element at position the last index of stack TO the INTEGER PART OF the element at position the last index of stack DIVIDED BY num
                    END IF

                    SET sign TO char
                    SET num TO zero
                END IF

                IF char EQUALS close parenthesis symbol
                    BREAK WHILE
                END IF
            END WHILE

            RETURN the SUM OF all elements in stack
        END FUNCTION

        SET string_s TO the LIST OF CHARACTERS FROM string_s
        RETURN CALL helper WITH string_s
    END FUNCTION
END CLASS
