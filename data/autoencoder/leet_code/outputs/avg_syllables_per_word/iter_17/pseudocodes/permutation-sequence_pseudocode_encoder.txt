CLASS Solution
    FUNCTION getPermutation(n, k)
        CALL import_factorial_function
        
        SET list_of_numbers TO A LIST OF NUMBERS FROM one TO n
        DECREMENT k BY one
        SET list_of_result_digits TO an empty list
        
        FOR i FROM n DOWN TO one
            SET factorial_value TO the FACTORIAL OF i MINUS one
            SET index TO k DIVIDED BY factorial_value
            APPEND the element at position index of list_of_numbers TO list_of_result_digits
            REMOVE the element at position index FROM list_of_numbers
            SET k TO k MODULO factorial_value
        END FOR
        
        RETURN CALL join_list_of_characters INTO string WITH list_of_result_digits
    END FUNCTION
END CLASS
