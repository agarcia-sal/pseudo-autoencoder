CLASS Solution
    FUNCTION nextPermutation(list_of_numbers)
        SET index_i TO the LENGTH OF list_of_numbers MINUS two
        WHILE index_i IS GREATER THAN OR EQUAL TO zero AND element at position index_i of list_of_numbers IS GREATER THAN OR EQUAL TO element at position index_i PLUS one of list_of_numbers
            DECREMENT index_i BY one
        END WHILE

        IF index_i IS GREATER THAN OR EQUAL TO zero
            SET index_j TO the LENGTH OF list_of_numbers MINUS one
            WHILE element at position index_j of list_of_numbers IS LESS THAN OR EQUAL TO element at position index_i of list_of_numbers
                DECREMENT index_j BY one
            END WHILE

            CALL SwapElements WITH list_of_numbers index_i index_j
        END IF

        CALL ReverseSequenceFromPosition WITH list_of_numbers index_i PLUS one
    END FUNCTION

    FUNCTION SwapElements(list_of_numbers position_a position_b)
        SET temporary_value TO element at position position_a of list_of_numbers
        SET element at position position_a of list_of_numbers TO element at position position_b of list_of_numbers
        SET element at position position_b of list_of_numbers TO temporary_value
    END FUNCTION

    FUNCTION ReverseSequenceFromPosition(list_of_numbers start_position)
        SET left_pointer TO start_position
        SET right_pointer TO the LENGTH OF list_of_numbers MINUS one
        WHILE left_pointer IS LESS THAN right_pointer
            CALL SwapElements WITH list_of_numbers left_pointer right_pointer
            INCREMENT left_pointer BY one
            DECREMENT right_pointer BY one
        END WHILE
    END FUNCTION
END CLASS
