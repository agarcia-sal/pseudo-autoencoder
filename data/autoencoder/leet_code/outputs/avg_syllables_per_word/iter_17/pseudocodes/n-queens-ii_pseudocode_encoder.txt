CLASS Solution
    FUNCTION totalNQueens(n)
        FUNCTION is_not_under_attack(row, col)
            RETURN the NEGATION OF (element at position col of cols PLUS element at position row MINUS col of hills PLUS element at position row PLUS col of dales)
        END FUNCTION

        FUNCTION place_queen(row, col)
            SET element at position col of cols TO one
            SET element at position row MINUS col of hills TO one
            SET element at position row PLUS col of dales TO one
        END FUNCTION

        FUNCTION remove_queen(row, col)
            SET element at position col of cols TO zero
            SET element at position row MINUS col of hills TO zero
            SET element at position row PLUS col of dales TO zero
        END FUNCTION

        FUNCTION backtrack(row ASSIGNED TO zero, count ASSIGNED TO zero)
            FOR col FROM zero TO n MINUS one
                IF is_not_under_attack(row, col) IS true
                    CALL place_queen WITH row AND col
                    IF row PLUS one EQUALS n
                        INCREMENT count BY one
                    ELSE
                        SET count TO CALL backtrack WITH row PLUS one AND count
                    END IF
                    CALL remove_queen WITH row AND col
                END IF
            END FOR
            RETURN count
        END FUNCTION

        SET cols TO A LIST OF zero REPEATED n TIMES
        SET hills TO A LIST OF zero REPEATED TWO MULTIPLIED BY n MINUS one TIMES
        SET dales TO A LIST OF zero REPEATED TWO MULTIPLIED BY n MINUS one TIMES

        RETURN CALL backtrack WITH no arguments
    END FUNCTION
END CLASS
