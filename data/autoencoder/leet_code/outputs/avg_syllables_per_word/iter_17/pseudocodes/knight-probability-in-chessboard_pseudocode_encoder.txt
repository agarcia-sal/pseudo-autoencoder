CLASS Solution
    FUNCTION knightProbability(n, k, row, column)
        SET moves TO the list of pairs containing negative two and negative one, negative two and one, negative one and negative two, negative one and two, one and negative two, one and two, two and negative one, two and one
        SET dp TO a two dimensional list of size n by n filled with zero
        SET element at position row and column of dp TO one
        FOR each iteration FROM one TO k
            SET new_dp TO a two dimensional list of size n by n filled with zero
            FOR each r FROM zero TO n MINUS one
                FOR each c FROM zero TO n MINUS one
                    IF element at position r and c of dp EQUALS zero THEN
                        CONTINUE to next iteration of inner loop
                    END IF
                    FOR each pair of dr and dc IN moves
                        SET nr TO r PLUS dr
                        SET nc TO c PLUS dc
                        IF nr IS GREATER THAN OR EQUAL TO zero AND nr IS LESS THAN n AND nc IS GREATER THAN OR EQUAL TO zero AND nc IS LESS THAN n THEN
                            INCREMENT element at position nr and nc of new_dp BY element at position r and c of dp DIVIDED BY eight
                        END IF
                    END FOR
                END FOR
            END FOR
            SET dp TO new_dp
        END FOR
        SET total_probability TO zero
        FOR each row_list IN dp
            SET row_sum TO zero
            FOR each value IN row_list
                INCREMENT row_sum BY value
            END FOR
            INCREMENT total_probability BY row_sum
        END FOR
        RETURN total_probability
    END FUNCTION
END CLASS
