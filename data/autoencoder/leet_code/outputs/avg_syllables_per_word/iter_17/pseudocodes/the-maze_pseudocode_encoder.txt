CLASS Solution
    FUNCTION hasPath(maze, start, destination)
        SET directions TO list of tuples zero one, one zero, zero minus one, minus one zero
        SET m TO the LENGTH OF maze
        SET n TO the LENGTH OF element at position zero of maze
        SET visited TO empty set

        FUNCTION dfs(x, y)
            IF tuple of x and y IS IN visited
                RETURN false
            END IF
            IF list of x and y EQUALS destination
                RETURN true
            END IF
            ADD tuple of x and y TO visited

            FOR each dx, dy IN directions
                SET nx TO x
                SET ny TO y
                WHILE nx plus dx GREATER THAN OR EQUAL TO zero AND nx plus dx LESS THAN m AND ny plus dy GREATER THAN OR EQUAL TO zero AND ny plus dy LESS THAN n AND element at position nx plus dx of maze element at position ny plus dy EQUALS zero
                    INCREMENT nx BY dx
                    INCREMENT ny BY dy
                END WHILE
                IF CALL dfs WITH nx, ny RETURNS true
                    RETURN true
                END IF
            END FOR
            RETURN false
        END FUNCTION

        RETURN CALL dfs WITH element at position zero of start, element at position one of start
    END FUNCTION
END CLASS
