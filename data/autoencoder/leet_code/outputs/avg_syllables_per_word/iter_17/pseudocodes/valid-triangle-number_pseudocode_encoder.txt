CLASS Solution
    FUNCTION triangleNumber(list_of_numbers)
        CALL list_of_numbers SORT
        SET count TO zero
        SET n TO the LENGTH OF list_of_numbers

        FOR k FROM n MINUS one TO two STEP minus one
            SET i TO zero
            SET j TO k MINUS one
            WHILE i IS LESS THAN j
                IF element at position i of list_of_numbers PLUS element at position j of list_of_numbers IS GREATER THAN element at position k of list_of_numbers
                    INCREMENT count BY j MINUS i
                    DECREMENT j BY one
                ELSE
                    INCREMENT i BY one
                END IF
            END WHILE
        END FOR

        RETURN count
    END FUNCTION
END CLASS
