CLASS Solution
    FUNCTION judgePoint24(cards)
        FUNCTION solve(nums)
            IF the LENGTH OF nums EQUALS one
                RETURN the ABSOLUTE VALUE OF the ELEMENT at position zero of nums MINUS twenty four LESS THAN one divided BY ten to the power of negative six
            END IF

            FOR i FROM zero TO the LENGTH OF nums MINUS one
                FOR j FROM zero TO the LENGTH OF nums MINUS one
                    IF i NOT EQUALS j
                        SET new_nums TO an AUXILIARY FUNCTION TO REMOVE elements at positions i AND j FROM nums

                        FOR each operation IN the list of FUNCTIONS truediv mul add sub
                            IF (operation IS add OR operation IS mul) AND j GREATER THAN i
                                CONTINUE TO the next iteration
                            END IF

                            IF operation IS NOT truediv OR the ELEMENT at position j of nums NOT EQUALS zero
                                APPEND the RESULT OF CALLING operation WITH the ELEMENT at position i of nums AND the ELEMENT at position j of nums TO new_nums
                                IF CALL solve WITH new_nums RETURNS True
                                    RETURN True
                                END IF
                                REMOVE the LAST element FROM new_nums
                            END IF
                        END FOR
                    END IF
                END FOR
            END FOR
            RETURN False
        END FUNCTION

        IMPORT the FUNCTIONS truediv mul add sub FROM operator
        RETURN CALL solve WITH cards
END CLASS
