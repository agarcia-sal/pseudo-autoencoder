CLASS Solution
    FUNCTION wiggleSort(list_of_numbers)
        CALL sort ON list_of_numbers
        SET sorted_numbers TO a COPY OF list_of_numbers
        SET length_of_numbers TO the LENGTH OF list_of_numbers
        SET left_pointer TO the INTEGER DIVISION OF length_of_numbers MINUS one BY two
        SET right_pointer TO length_of_numbers MINUS one
        FOR index FROM zero TO length_of_numbers MINUS one
            IF the REMAINDER OF index DIVIDED BY two EQUALS zero
                SET element at position index of list_of_numbers TO element at position left_pointer of sorted_numbers
                DECREMENT left_pointer BY one
            ELSE
                SET element at position index of list_of_numbers TO element at position right_pointer of sorted_numbers
                DECREMENT right_pointer BY one
            END IF
        END FOR
    END FUNCTION
END CLASS
