CLASS Solution
    FUNCTION jump(list_of_numbers)
        IF the LENGTH OF list_of_numbers LESS THAN OR EQUAL TO one
            RETURN zero
        END IF

        SET jumps TO zero
        SET current_end TO zero
        SET farthest TO zero

        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one MINUS one
            SET farthest TO the GREATER OF farthest AND index PLUS the element at position index of list_of_numbers

            IF index EQUALS current_end
                INCREMENT jumps BY one
                SET current_end TO farthest

                IF current_end GREATER THAN OR EQUAL TO the LENGTH OF list_of_numbers MINUS one
                    BREAK
                END IF
            END IF
        END FOR

        RETURN jumps
    END FUNCTION
END CLASS
