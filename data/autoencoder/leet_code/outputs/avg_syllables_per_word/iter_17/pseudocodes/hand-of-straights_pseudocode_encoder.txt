CLASS Solution
    FUNCTION isNStraightHand(hand, groupSize)
        IF the LENGTH OF hand MODULO groupSize IS NOT EQUALS zero
            RETURN False
        END IF
        SET count TO CALL Counter WITH hand
        SET unique_cards TO CALL sorted WITH the KEYS OF count
        FOR each card IN unique_cards
            IF the ELEMENT AT position card OF count IS GREATER THAN zero
                SET needed TO the ELEMENT AT position card OF count
                FOR index FROM zero TO groupSize MINUS one
                    IF the ELEMENT AT position card PLUS index OF count IS LESS THAN needed
                        RETURN False
                    END IF
                    SET the ELEMENT AT position card PLUS index OF count TO the ELEMENT AT position card PLUS index OF count MINUS needed
                END FOR
            END IF
        END FOR
        RETURN True
    END FUNCTION
END CLASS
