CLASS Solution
    FUNCTION countAndSay(n)
        IF n EQUALS one
            RETURN string one
        END IF

        SET current_sequence TO string one

        FOR index FROM two TO n
            SET next_sequence TO empty string
            SET count TO one
            SET previous_char TO element at position zero of current_sequence

            FOR i FROM one TO the LENGTH OF current_sequence MINUS one
                IF element at position i of current_sequence EQUALS previous_char
                    INCREMENT count BY one
                ELSE
                    CALL AppendCountAndChar WITH parameters count AND previous_char AND next_sequence
                    SET previous_char TO element at position i of current_sequence
                    SET count TO one
                END IF
            END FOR

            CALL AppendCountAndChar WITH parameters count AND previous_char AND next_sequence

            SET current_sequence TO next_sequence
        END FOR

        RETURN current_sequence
    END FUNCTION

    FUNCTION AppendCountAndChar(count, character, sequence)
        SET count_string TO ConvertNumberToString WITH parameter count
        SET new_sequence TO sequence CONCATENATED WITH count_string CONCATENATED WITH character
        SET sequence TO new_sequence
    END FUNCTION

    FUNCTION ConvertNumberToString(number)
        RETURN string representation of number
    END FUNCTION
END CLASS
