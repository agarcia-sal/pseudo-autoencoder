CLASS Solution
    FUNCTION generateAbbreviations(word)
        FUNCTION backtrack(index, path, count)
            IF index EQUALS the LENGTH OF word
                IF count GREATER THAN zero
                    APPEND the STRING REPRESENTATION OF count TO path
                END IF
                APPEND the CONCATENATION OF all elements IN path TO result
            ELSE
                IF count GREATER THAN zero
                    CALL backtrack WITH parameters index PLUS one, the CONCATENATION OF path WITH the STRING REPRESENTATION OF count AND the element at position index of word, zero
                ELSE
                    CALL backtrack WITH parameters index PLUS one, the CONCATENATION OF path WITH the element at position index of word, zero
                END IF
                CALL backtrack WITH parameters index PLUS one, path, count PLUS one
            END IF
        END FUNCTION
        SET result TO empty list
        CALL backtrack WITH parameters zero, empty list, zero
        RETURN result
    END FUNCTION
END CLASS
