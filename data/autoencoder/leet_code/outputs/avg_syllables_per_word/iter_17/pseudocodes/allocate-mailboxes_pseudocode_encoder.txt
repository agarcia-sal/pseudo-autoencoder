CLASS Solution
    FUNCTION minDistance(houses, k)
        CALL houses SORT
        SET n TO the LENGTH OF houses
        SET cost TO two dimensional list OF size n by n WITH zeros

        FOR i FROM n MINUS one DOWN TO zero
            FOR j FROM i PLUS one TO n MINUS one
                SET element at position i and j OF cost TO element at position i PLUS one and j MINUS one OF cost PLUS element at position j OF houses MINUS element at position i OF houses
            END FOR
        END FOR

        SET dp TO two dimensional list OF size n PLUS one by k PLUS one FILLED WITH positive infinity
        SET element at position zero and zero OF dp TO zero

        FOR i FROM one TO n
            FOR j FROM one TO k
                FOR p FROM zero TO i MINUS one
                    SET element at position i and j OF dp TO the MINIMUM OF element at position i and j OF dp AND element at position p and j MINUS one OF dp PLUS element at position p and i MINUS one OF cost
                END FOR
            END FOR
        END FOR

        RETURN element at position n and k OF dp
    END FUNCTION
END CLASS
