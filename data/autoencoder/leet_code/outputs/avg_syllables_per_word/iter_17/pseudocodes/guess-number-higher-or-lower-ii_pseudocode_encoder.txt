CLASS Solution
    FUNCTION getMoneyAmount(n_integer)
        FUNCTION dp(left_integer, right_integer)
            IF left_integer GREATER THAN OR EQUAL TO right_integer
                RETURN zero
            END IF
            SET min_cost TO infinity_value
            FOR pivot_integer FROM left_integer TO right_integer
                SET cost TO pivot_integer PLUS the MAXIMUM OF dp(left_integer, pivot_integer MINUS one) AND dp(pivot_integer PLUS one, right_integer)
                SET min_cost TO the MINIMUM OF min_cost AND cost
            END FOR
            RETURN min_cost
        END FUNCTION
        RETURN CALL dp WITH one AND n_integer
    END FUNCTION
END CLASS
