CLASS Solution
    FUNCTION primePalindrome(number_n)
        FUNCTION is_prime(number_num)
            IF number_num LESS THAN two
                RETURN false
            END IF
            IF number_num EQUALS two OR number_num EQUALS three
                RETURN true
            END IF
            IF number_num MODULO two EQUALS zero OR number_num MODULO three EQUALS zero
                RETURN false
            END IF
            SET iterator_i TO five
            WHILE iterator_i MULTIPLIED BY iterator_i LESS THAN OR EQUAL TO number_num holds
                IF number_num MODULO iterator_i EQUALS zero OR number_num MODULO iterator_i PLUS two EQUALS zero
                    RETURN false
                END IF
                INCREMENT iterator_i BY six
            END WHILE
            RETURN true
        END FUNCTION

        FUNCTION is_palindrome(number_num)
            RETURN the STRING REPRESENTATION OF number_num EQUALS the STRING REPRESENTATION OF number_num REVERSED
        END FUNCTION

        IF number_n LESS THAN OR EQUAL TO two
            RETURN two
        END IF

        IF number_n GREATER THAN OR EQUAL TO ten AND number_n LESS THAN OR EQUAL TO eleven
            RETURN eleven
        END IF

        WHILE true holds
            IF number_n MODULO two EQUALS zero
                INCREMENT number_n BY one
                CONTINUE to next iteration
            END IF
            IF is_palindrome(number_n) AND is_prime(number_n)
                RETURN number_n
            END IF
            IF the LENGTH OF the STRING REPRESENTATION OF number_n MODULO two EQUALS zero
                SET number_n TO ten RAISED TO THE POWER OF the LENGTH OF the STRING REPRESENTATION OF number_n PLUS one
            ELSE
                INCREMENT number_n BY two
            END IF
        END WHILE
    END FUNCTION
END CLASS
