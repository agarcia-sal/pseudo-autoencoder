CLASS Solution
    FUNCTION numberOfArrays(s STRING, k INTEGER) RETURNS INTEGER
        SET MOD TO one billion plus seven
        SET n TO the LENGTH OF s

        CALL initialize_dp_array WITH n RETURNING dp_array
        SET element at position zero of dp_array TO one

        FOR i FROM one TO n
            FOR j FROM zero TO i MINUS one
                IF element at position j of s IS NOT EQUAL TO zero AND 
                   the INTEGER VALUE OF substring from position j TO i MINUS one of s IS GREATER THAN OR EQUAL TO one AND 
                   the INTEGER VALUE OF substring from position j TO i MINUS one of s IS LESS THAN OR EQUAL TO k
                    SET element at position i of dp_array TO element at position i of dp_array PLUS element at position j of dp_array MODULO MOD
                END IF
            END FOR
        END FOR

        RETURN element at position n of dp_array
    END FUNCTION
END CLASS
