CLASS Solution
    FUNCTION peakIndexInMountainArray(list_of_integers arr)
        SET left TO zero
        SET right TO the LENGTH OF arr MINUS one
        WHILE left IS LESS THAN right holds
            SET mid TO left PLUS right DIVIDED BY two
            IF element at position mid of arr IS LESS THAN element at position mid PLUS one of arr
                SET left TO mid PLUS one
            ELSE
                SET right TO mid
            END IF
        END WHILE
        RETURN left
    END FUNCTION
END CLASS
