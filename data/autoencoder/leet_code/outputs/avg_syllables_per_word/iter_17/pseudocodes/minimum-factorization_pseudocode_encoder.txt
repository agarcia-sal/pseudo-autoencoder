CLASS Solution
    FUNCTION smallestFactorization(num)
        IF num EQUALS one
            RETURN one
        END IF

        SET factors TO an empty list

        FOR digit FROM nine TO two DECREMENT BY one
            WHILE the remainder of num DIVIDED BY digit EQUALS zero holds
                APPEND digit TO factors
                SET num TO the quotient of num DIVIDED BY digit
            END WHILE
        END FOR

        IF num NOT EQUALS one
            RETURN zero
        END IF

        CALL factors.sort

        SET result TO the integer formed by concatenating the string representation of each element IN factors in order

        IF result GREATER THAN two raised to the power of thirty-one MINUS one
            RETURN zero
        END IF

        RETURN result
    END FUNCTION
END CLASS
