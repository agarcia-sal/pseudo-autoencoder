CLASS Solution
    FUNCTION increasingTriplet(list_of_numbers)
        SET first TO positive_infinity
        SET second TO positive_infinity
        FOR each number IN list_of_numbers
            IF number LESS THAN OR EQUAL TO first
                SET first TO number
            ELSE IF number LESS THAN OR EQUAL TO second
                SET second TO number
            ELSE
                RETURN true
            END IF
        END FOR
        RETURN false
    END FUNCTION
END CLASS
