CLASS Solution
    FUNCTION splitLoopedString(list_of_strings)
        FOR index FROM zero TO the LENGTH OF list_of_strings MINUS one
            SET reversed_string TO the REVERSE OF element at position index of list_of_strings
            IF reversed_string GREATER THAN element at position index of list_of_strings
                SET element at position index of list_of_strings TO reversed_string
            END IF
        END FOR

        SET doubled_list_of_strings TO list_of_strings CONCATENATED WITH list_of_strings

        SET maximum_string TO empty string

        FOR index FROM zero TO the LENGTH OF list_of_strings MINUS one
            FOR reversed_flag IN list containing false AND true
                IF reversed_flag IS true
                    SET current_string TO the REVERSE OF element at position index of list_of_strings
                ELSE
                    SET current_string TO element at position index of list_of_strings
                END IF

                SET remaining_string TO the CONCATENATION OF elements from position index plus one TO position index plus the LENGTH OF list_of_strings MINUS one of doubled_list_of_strings

                FOR cut_position FROM zero TO the LENGTH OF current_string
                    SET candidate_string TO substring from position cut_position TO end of current_string CONCATENATED WITH remaining_string CONCATENATED WITH substring from position zero TO cut_position MINUS one of current_string
                    IF candidate_string GREATER THAN maximum_string
                        SET maximum_string TO candidate_string
                    END IF
                END FOR
            END FOR
        END FOR

        RETURN maximum_string
    END FUNCTION
END CLASS
