CLASS Solution
    FUNCTION minimumMoves(arr)
        SET n TO the LENGTH OF arr
        SET dp TO auxiliary FUNCTION createTwoDimensionalList WITH parameters zero AND n

        FOR index FROM zero TO n MINUS one
            SET element at position index AND index of dp TO one
        END FOR

        FOR index FROM zero TO n MINUS two
            IF element at position index of arr EQUALS element at position index PLUS one of arr
                SET element at position index AND index PLUS one of dp TO one
            ELSE
                SET element at position index AND index PLUS one of dp TO two
            END IF
        END FOR

        FOR length FROM three TO n
            FOR start_index FROM zero TO n MINUS length
                SET end_index TO start_index PLUS length MINUS one
                IF element at position start_index of arr EQUALS element at position end_index of arr
                    SET element at position start_index AND end_index of dp TO element at position start_index PLUS one AND end_index MINUS one of dp
                ELSE
                    SET element at position start_index AND end_index of dp TO positive infinity
                    FOR partition_index FROM start_index TO end_index MINUS one
                        SET dp_value TO element at position start_index AND partition_index of dp PLUS element at position partition_index PLUS one AND end_index of dp
                        IF dp_value LESS THAN element at position start_index AND end_index of dp
                            SET element at position start_index AND end_index of dp TO dp_value
                        END IF
                    END FOR
                END IF
            END FOR
        END FOR

        RETURN element at position zero AND n MINUS one of dp
    END FUNCTION
