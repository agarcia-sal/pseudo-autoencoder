CLASS Solution
    FUNCTION largestDivisibleSubset(list_of_numbers)
        CALL list_of_numbers SORT
        SET length_n TO the LENGTH OF list_of_numbers
        SET list_f TO a new list OF length_n filled WITH one
        SET index_k TO zero
        FOR index_i FROM zero TO length_n MINUS one
            FOR index_j FROM zero TO index_i MINUS one
                IF element at position index_i of list_of_numbers MODULO element at position index_j of list_of_numbers EQUALS zero
                    SET element at position index_i of list_f TO the GREATER OF element at position index_i of list_f AND element at position index_j of list_f PLUS one
                END IF
            END FOR
            IF element at position index_k of list_f LESS THAN element at position index_i of list_f
                SET index_k TO index_i
            END IF
        END FOR
        SET variable_m TO element at position index_k of list_f
        SET index_i TO index_k
        SET list_ans TO an empty list
        WHILE variable_m GREATER THAN zero holds
            IF element at position index_k of list_of_numbers MODULO element at position index_i of list_of_numbers EQUALS zero AND element at position index_i of list_f EQUALS variable_m
                APPEND element at position index_i of list_of_numbers TO list_ans
                SET index_k TO index_i
                DECREMENT variable_m BY one
            END IF
            DECREMENT index_i BY one
        END WHILE
        RETURN list_ans
    END FUNCTION
END CLASS
