CLASS Solution
    FUNCTION reorganizeString(s)
        SET count TO CALL Counter WITH parameters s
        SET max_heap TO CALL CreateMaxHeap FROM count
        SET prev_char TO none
        SET prev_freq TO zero
        SET result TO empty list
        WHILE max_heap IS NOT empty OR prev_freq LESS THAN zero holds
            IF max_heap IS empty AND prev_freq LESS THAN zero holds
                RETURN empty string
            END IF
            SET freq, char TO CALL ExtractMax FROM max_heap
            APPEND char TO result
            INCREMENT freq BY one
            IF prev_freq LESS THAN zero holds
                CALL InsertMaxHeap WITH parameters max_heap, prev_freq, prev_char
            END IF
            SET prev_char TO char
            SET prev_freq TO freq
        END WHILE
        RETURN CALL JoinCharacters WITH parameter result
    END FUNCTION
END CLASS
