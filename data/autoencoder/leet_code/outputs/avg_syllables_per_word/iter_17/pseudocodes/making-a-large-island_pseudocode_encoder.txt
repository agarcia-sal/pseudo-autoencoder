CLASS Solution
    FUNCTION largestIsland(grid)
        IF grid IS empty OR element at position zero of grid IS empty
            RETURN zero
        END IF

        SET n TO the LENGTH OF grid
        SET directions TO a list of elements zero one one zero zero negative one negative one zero

        FUNCTION is_valid(x y)
            RETURN zero LESS THAN OR EQUAL TO x AND x LESS THAN n AND zero LESS THAN OR EQUAL TO y AND y LESS THAN n
        END FUNCTION

        FUNCTION dfs(x y index)
            SET area TO one
            SET element at position x of grid element at position y TO index
            FOR each pair dx dy IN directions
                SET nx TO x PLUS dx
                SET ny TO y PLUS dy
                IF is_valid(nx ny) AND element at position nx of grid element at position ny EQUALS one
                    INCREMENT area BY CALL dfs WITH parameters nx ny index
                END IF
            END FOR
            RETURN area
        END FUNCTION

        SET island_areas TO empty mapping
        SET index TO two

        FOR i FROM zero TO n MINUS one
            FOR j FROM zero TO n MINUS one
                IF element at position i of grid element at position j EQUALS one
                    SET element at position index OF island_areas TO CALL dfs WITH parameters i j index
                    INCREMENT index BY one
                END IF
            END FOR
        END FOR

        IF island_areas IS empty
            RETURN one
        END IF

        SET max_island_size TO the MAXIMUM VALUE OF island_areas

        FOR i FROM zero TO n MINUS one
            FOR j FROM zero TO n MINUS one
                IF element at position i of grid element at position j EQUALS zero
                    SET neighbors TO empty set
                    FOR each pair dx dy IN directions
                        SET nx TO i PLUS dx
                        SET ny TO j PLUS dy
                        IF is_valid(nx ny) AND element at position nx of grid element at position ny GREATER THAN one
                            ADD element at position nx of grid element at position ny TO neighbors
                        END IF
                    END FOR
                    SET current_island_size TO one PLUS the SUM OF elements at each index in neighbors FROM island_areas
                    IF current_island_size GREATER THAN max_island_size
                        SET max_island_size TO current_island_size
                    END IF
                END IF
            END FOR
        END FOR

        RETURN max_island_size
    END FUNCTION
END CLASS
