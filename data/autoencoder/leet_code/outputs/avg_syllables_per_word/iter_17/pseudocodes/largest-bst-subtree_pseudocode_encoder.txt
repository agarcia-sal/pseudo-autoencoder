CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN None
    END IF
    SET root TO CALL TreeNode WITH the element at position zero of values
    SET i TO one
    SET queue TO CALL deque WITH no parameters
    CALL APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO CALL popleft ON queue
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None
            SET node.left TO CALL TreeNode WITH the element at position i of values
            CALL APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None
            SET node.right TO CALL TreeNode WITH the element at position i of values
            CALL APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF the value of p.val IS NOT EQUALS to the value of q.val
        RETURN False
    ELSE
        RETURN the CALL is_same_tree WITH parameters p.left AND q.left AND the CALL is_same_tree WITH parameters p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION largestBSTSubtree(self, root)
        FUNCTION dfs(node)
            IF node IS None
                RETURN positive infinity, negative infinity, zero
            END IF
            SET left_min, left_max, left_size TO CALL dfs WITH parameter node.left
            SET right_min, right_max, right_size TO CALL dfs WITH parameter node.right
            IF left_max LESS THAN the value of node.val AND the value of node.val LESS THAN right_min
                RETURN the smaller value BETWEEN the value of node.val AND left_min, the larger value BETWEEN the value of node.val AND right_max, left_size PLUS right_size PLUS one
            END IF
            RETURN negative infinity, positive infinity, the larger value BETWEEN left_size AND right_size
        END FUNCTION
        RETURN the element at position two of the CALL dfs WITH parameter root
    END FUNCTION
END CLASS
