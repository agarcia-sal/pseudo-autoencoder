CLASS Solution
    FUNCTION lengthOfLongestSubstring(string s)
        SET char_index_map TO an empty dictionary
        SET max_length TO zero
        SET start TO zero

        FOR each position end AND each character char IN the enumeration of string s
            IF character char IS IN char_index_map AND the element at position char IN char_index_map IS GREATER THAN OR EQUAL TO start
                SET start TO the element at position char IN char_index_map PLUS one
            END IF
            SET the element at position char IN char_index_map TO end
            SET max_length TO the maximum of max_length AND end MINUS start PLUS one
        END FOR

        RETURN max_length
    END FUNCTION
END CLASS
