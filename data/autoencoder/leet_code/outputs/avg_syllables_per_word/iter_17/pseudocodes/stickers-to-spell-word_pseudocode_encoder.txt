CLASS Solution
    FUNCTION minStickers(list_of_stickers, target_string)
        SET list_of_stickers TO CALL convert_stickers_to_counters WITH list_of_stickers
        CALL sort_stickers_by_max_frequency WITH list_of_stickers

        SET queue TO CALL initialize_queue WITH target_string and zero
        SET memoization_table TO CALL initialize_memo WITH target_string and zero

        WHILE queue is not empty
            SET current_target_and_used_stickers TO CALL dequeue_element FROM queue
            SET current_target TO element at position zero of current_target_and_used_stickers
            SET used_stickers TO element at position one of current_target_and_used_stickers

            FOR each sticker IN list_of_stickers
                IF first character of current_target IS NOT IN sticker
                    CONTINUE to next sticker
                END IF

                SET remaining_characters_counter TO CALL counter_of_string WITH current_target
                CALL subtract_counter WITH remaining_characters_counter and sticker

                SET new_target_string TO CALL string_from_sorted_elements_of_counter WITH remaining_characters_counter

                IF new_target_string IS empty string
                    RETURN used_stickers PLUS one
                END IF

                IF new_target_string IS NOT IN memoization_table
                    SET value at key new_target_string IN memoization_table TO used_stickers PLUS one
                    CALL enqueue_element TO queue WITH new_target_string AND used_stickers PLUS one
                END IF
            END FOR
        END WHILE

        RETURN negative one
    END FUNCTION
