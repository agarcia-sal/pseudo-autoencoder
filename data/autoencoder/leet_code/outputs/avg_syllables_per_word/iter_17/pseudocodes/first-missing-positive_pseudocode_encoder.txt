CLASS Solution
    FUNCTION firstMissingPositive(nums)
        SET n TO the LENGTH OF nums
        FOR index FROM zero TO n MINUS one
            WHILE index PLUS one IS GREATER THAN OR EQUAL TO one AND element at position index of nums IS LESS THAN OR EQUAL TO n AND element at position element at position index of nums MINUS one of nums IS NOT EQUALS TO element at position index of nums
                SET correct_index TO element at position index of nums MINUS one
                SET temporary_value TO element at position correct_index of nums
                SET element at position correct_index of nums TO element at position index of nums
                SET element at position index of nums TO temporary_value
            END WHILE
        END FOR
        FOR index FROM zero TO n MINUS one
            IF element at position index of nums IS NOT EQUALS TO index PLUS one
                RETURN index PLUS one
            END IF
        END FOR
        RETURN n PLUS one
    END FUNCTION
END CLASS
