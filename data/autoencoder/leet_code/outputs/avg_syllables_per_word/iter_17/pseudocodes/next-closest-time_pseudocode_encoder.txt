CLASS Solution
    FUNCTION nextClosestTime(time)
        SET digits TO the SET OF all characters in time EXCLUDING colon character
        SET hour TO the INTEGER VALUE OF the substring from position zero TO one of time
        SET minute TO the INTEGER VALUE OF the substring from position three TO four of time
        
        FUNCTION generate_times(digit_set)
            SET times TO empty list
            FOR each first_hour_digit IN digit_set
                FOR each second_hour_digit IN digit_set
                    FOR each first_minute_digit IN digit_set
                        FOR each second_minute_digit IN digit_set
                            SET composed_hour TO the INTEGER VALUE OF concatenating first_hour_digit AND second_hour_digit
                            SET composed_minute TO the INTEGER VALUE OF concatenating first_minute_digit AND second_minute_digit
                            IF composed_hour LESS THAN twenty four AND composed_minute LESS THAN sixty
                                APPEND the tuple composed_hour AND composed_minute TO times
                            END IF
                        END FOR
                    END FOR
                END FOR
            END FOR
            RETURN the sorted list times
        END FUNCTION
        
        SET all_times TO CALL generate_times WITH digits
        SET current_time_index TO the POSITION OF the tuple hour AND minute IN all_times
        
        IF current_time_index PLUS one LESS THAN the LENGTH OF all_times
            SET next_hour AND next_minute TO the element at position current_time_index PLUS one of all_times
        ELSE
            SET next_hour AND next_minute TO the element at position zero of all_times
        END IF
        
        RETURN the FORMATTED string REPRESENTING next_hour AND next_minute with two digit zero padding each
    END FUNCTION
END CLASS
