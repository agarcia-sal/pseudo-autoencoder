CLASS Solution
    FUNCTION lastSubstring(string s)
        SET length_n TO the LENGTH OF s
        SET maximum_character TO the MAXIMUM CHARACTER IN s
        SET maximum_indices TO the RESULT OF calling getIndicesOfMaxChar WITH s AND maximum_character AND length_n
        SET maximum_substring TO empty string
        FOR each start_position IN maximum_indices
            SET current_substring TO the SUBSTRING FROM position start_position TO the END OF s
            IF current_substring GREATER THAN maximum_substring
                SET maximum_substring TO current_substring
            END IF
        END FOR
        RETURN maximum_substring
    END FUNCTION

    FUNCTION getIndicesOfMaxChar(string s, character max_character, integer length_n)
        INITIALIZE list_of_indices AS empty list
        FOR index FROM zero TO length_n MINUS one
            IF the CHARACTER at position index of s EQUALS max_character
                APPEND index TO list_of_indices
            END IF
        END FOR
        RETURN list_of_indices
    END FUNCTION
END CLASS
