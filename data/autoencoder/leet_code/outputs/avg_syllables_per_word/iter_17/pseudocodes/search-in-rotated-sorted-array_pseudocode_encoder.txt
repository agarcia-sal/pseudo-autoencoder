CLASS Solution
    FUNCTION search(list_of_numbers, target_value)
        SET left TO zero
        SET right TO the LENGTH OF list_of_numbers MINUS one
        WHILE left IS LESS THAN OR EQUAL TO right
            SET mid TO left PLUS (right MINUS left) DIVIDED BY two
            IF element at position mid of list_of_numbers IS EQUALS target_value
                RETURN mid
            END IF
            IF element at position left of list_of_numbers IS LESS THAN OR EQUAL TO element at position mid of list_of_numbers
                IF element at position left of list_of_numbers IS LESS THAN OR EQUAL TO target_value AND target_value IS LESS THAN element at position mid of list_of_numbers
                    SET right TO mid MINUS one
                ELSE
                    SET left TO mid PLUS one
                END IF
            ELSE
                IF element at position mid of list_of_numbers IS LESS THAN target_value AND target_value IS LESS THAN OR EQUAL TO element at position right of list_of_numbers
                    SET left TO mid PLUS one
                ELSE
                    SET right TO mid MINUS one
                END IF
            END IF
        END WHILE
        RETURN minus one
    END FUNCTION
END CLASS
