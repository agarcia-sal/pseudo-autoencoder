CLASS Solution
    FUNCTION exclusiveTime(number_of_functions, list_of_logs)
        SET exclusive_time TO a list of zero repeated number_of_functions times
        SET stack TO an empty list
        FOR each log IN list_of_logs
            SET function_id_string, action_string, timestamp_string TO the SPLIT log BY colon
            SET function_id TO the INTEGER VALUE OF function_id_string
            SET timestamp TO the INTEGER VALUE OF timestamp_string
            IF action_string EQUALS start
                IF stack is not empty
                    SET previous_function_id, previous_timestamp TO the element at the last position of stack
                    INCREMENT the element at position previous_function_id of exclusive_time BY timestamp MINUS previous_timestamp
                END IF
                APPEND the tuple of function_id and timestamp TO stack
            ELSE
                SET previous_function_id, previous_timestamp TO the element popped from the last position of stack
                INCREMENT the element at position previous_function_id of exclusive_time BY timestamp MINUS previous_timestamp PLUS one
                IF stack is not empty
                    SET the last element of stack TO the tuple of the first element of the last element of stack, AND timestamp PLUS one
                END IF
            END IF
        END FOR
        RETURN exclusive_time
    END FUNCTION
END CLASS
