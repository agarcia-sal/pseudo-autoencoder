CLASS Solution
    FUNCTION reorderedPowerOf2(n AS integer) RETURNS boolean
        CALL initializePowerSetUpToOneBillion WITH no parameters RETURNING power_of_2_sorted_digits

        SET sorted_n_digits TO CALL convertNumberToSortedDigitTuple WITH parameter n

        SET is_matching_power TO WHETHER sorted_n_digits IS IN power_of_2_sorted_digits

        RETURN is_matching_power
    END FUNCTION

    FUNCTION initializePowerSetUpToOneBillion()
        SET power_of_2_sorted_digits TO empty set
        SET power_of_2 TO one

        WHILE power_of_2 IS LESS THAN OR EQUAL TO one billion
            SET digits_of_power TO CALL convertNumberToSortedDigitTuple WITH parameter power_of_2
            ADD digits_of_power TO power_of_2_sorted_digits
            SET power_of_2 TO power_of_2 MULTIPLIED BY two
        END WHILE

        RETURN power_of_2_sorted_digits
    END FUNCTION

    FUNCTION convertNumberToSortedDigitTuple(number AS integer)
        SET string_representation TO CONVERT number TO string
        SET sorted_characters TO SORT characters OF string_representation IN ORDER
        SET digit_tuple TO CONVERT sorted_characters TO tuple
        RETURN digit_tuple
    END FUNCTION
END CLASS
