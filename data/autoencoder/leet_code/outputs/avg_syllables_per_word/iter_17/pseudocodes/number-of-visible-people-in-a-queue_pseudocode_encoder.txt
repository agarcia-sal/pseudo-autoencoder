CLASS Solution
    FUNCTION canSeePersonsCount(heights)
        SET number_of_heights TO the LENGTH OF heights
        SET answer TO list of zero repeated number_of_heights times
        SET stack TO empty list

        FOR index FROM zero TO number_of_heights MINUS one
            WHILE stack IS NOT empty AND element at position last element of stack of heights LESS THAN element at position index of heights
                INCREMENT element at position stack POPPED from stack of answer BY one
            END WHILE
            IF stack IS NOT empty
                INCREMENT element at position last element of stack of answer BY one
            END IF
            APPEND index TO stack
        END FOR

        RETURN answer
    END FUNCTION
END CLASS
