CLASS Solution
    FUNCTION minCost(n, cuts)
        SET cuts TO the SORTED list obtained by APPENDING zero AND n TO cuts

        FUNCTION dp(left, right)
            IF right MINUS left IS LESS THAN OR EQUAL TO one
                RETURN zero
            END IF

            SET minimum_cost TO infinity
            FOR i FROM left PLUS one TO right MINUS one
                SET cost TO element at position right of cuts MINUS element at position left of cuts PLUS CALL dp WITH left AND i PLUS CALL dp WITH i AND right
                SET minimum_cost TO the LESSER OF minimum_cost AND cost
            END FOR

            RETURN minimum_cost
        END FUNCTION

        RETURN CALL dp WITH zero AND the LENGTH OF cuts MINUS one
END CLASS
