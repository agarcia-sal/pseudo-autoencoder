CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN None
    END IF
    SET root TO CALL TreeNode WITH the element at position zero of values
    SET i TO one
    SET queue TO CALL deque WITH no parameters
    CALL queue.append WITH root
    WHILE queue is not empty
        SET node TO CALL queue.popleft WITH no parameters
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.left TO CALL TreeNode WITH element at position i of values
            CALL queue.append WITH node.left
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.right TO CALL TreeNode WITH element at position i of values
            CALL queue.append WITH node.right
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUALS q.val
        RETURN False
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION largestValues(self, root)
        IF root IS None
            RETURN empty list
        END IF
        SET queue TO CALL deque WITH list containing root
        SET largest_values TO empty list
        WHILE queue is not empty
            SET level_size TO the LENGTH OF queue
            SET max_value TO negative infinity
            FOR index FROM zero TO level_size MINUS one
                SET node TO CALL queue.popleft WITH no parameters
                SET max_value TO the GREATER OF max_value AND node.val
                IF node.left IS NOT None
                    CALL queue.append WITH node.left
                END IF
                IF node.right IS NOT None
                    CALL queue.append WITH node.right
                END IF
            END FOR
            APPEND max_value TO largest_values
        END WHILE
        RETURN largest_values
    END FUNCTION
END CLASS
