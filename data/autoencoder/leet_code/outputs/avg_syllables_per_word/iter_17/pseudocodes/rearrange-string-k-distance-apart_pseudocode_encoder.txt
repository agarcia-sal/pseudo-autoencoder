CLASS Solution
    FUNCTION rearrangeString(s, k)
        IF k LESS THAN OR EQUAL TO one
            RETURN s
        END IF

        SET freq TO CALL Counter WITH s

        SET max_heap TO empty list
        FOR each character AND count IN freq
            APPEND tuple of negative count AND character TO max_heap
        END FOR

        CALL heapify WITH max_heap

        SET wait_queue TO empty list

        SET result TO empty list

        WHILE max_heap is not empty
            SET count AND char TO CALL heappop WITH max_heap
            APPEND char TO result

            APPEND tuple of count PLUS one AND char TO wait_queue

            IF LENGTH OF wait_queue EQUALS k
                SET count AND char TO REMOVE element at position zero OF wait_queue
                IF count LESS THAN zero
                    CALL heappush WITH max_heap AND tuple of count AND char
                END IF
            END IF
        END WHILE

        IF LENGTH OF result NOT EQUALS LENGTH OF s
            RETURN empty string
        END IF

        RETURN CALL join WITH result
    END FUNCTION
END CLASS
