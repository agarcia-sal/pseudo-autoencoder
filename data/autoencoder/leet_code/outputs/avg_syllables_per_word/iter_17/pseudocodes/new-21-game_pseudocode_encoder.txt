CLASS Solution
    FUNCTION new21Game(n, k, maxPts)
        FUNCTION dfs(i)
            IF i GREATER THAN OR EQUAL TO k
                IF i LESS THAN OR EQUAL TO n
                    RETURN one
                ELSE
                    RETURN zero
                END IF
            END IF
            IF i EQUALS k MINUS one
                SET minimum_value TO THE MINIMUM OF n MINUS k PLUS one AND maxPts
                RETURN minimum_value DIVIDED BY maxPts
            END IF
            SET first_call TO CALL dfs WITH i PLUS one
            SET second_call TO CALL dfs WITH i PLUS maxPts PLUS one
            RETURN first_call PLUS first_call MINUS second_call DIVIDED BY maxPts
        END FUNCTION
        RETURN CALL dfs WITH zero
    END FUNCTION
END CLASS
