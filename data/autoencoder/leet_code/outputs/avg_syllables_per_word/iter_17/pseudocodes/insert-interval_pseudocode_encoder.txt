CLASS Solution
    FUNCTION insert(list_of_intervals AS list of list of integers, new_interval AS list of integers) RETURNS list of list of integers
        IF list_of_intervals IS EMPTY
            RETURN list containing new_interval
        END IF

        SET merged_intervals TO empty list
        SET index TO zero
        SET total_intervals TO the LENGTH OF list_of_intervals

        WHILE index LESS THAN total_intervals AND element at position one of element at position index of list_of_intervals LESS THAN element at position zero of new_interval
            APPEND element at position index of list_of_intervals TO merged_intervals
            INCREMENT index BY one
        END WHILE

        WHILE index LESS THAN total_intervals AND element at position zero of element at position index of list_of_intervals LESS THAN OR EQUAL TO element at position one of new_interval
            SET element at position zero of new_interval TO the MINIMUM OF element at position zero of new_interval AND element at position zero of element at position index of list_of_intervals
            SET element at position one of new_interval TO the MAXIMUM OF element at position one of new_interval AND element at position one of element at position index of list_of_intervals
            INCREMENT index BY one
        END WHILE

        APPEND new_interval TO merged_intervals

        WHILE index LESS THAN total_intervals
            APPEND element at position index of list_of_intervals TO merged_intervals
            INCREMENT index BY one
        END WHILE

        RETURN merged_intervals
    END FUNCTION
END CLASS
