CLASS Solution
    FUNCTION longestSubstring(s string, k integer) RETURNS integer
        IF the LENGTH OF s IS LESS THAN k
            RETURN zero
        END IF

        SET count TO CALL Counter WITH s

        FOR each character IN the ELEMENTS OF count
            IF the element at position character of count IS LESS THAN k
                RETURN the MAXIMUM OF CALL longestSubstring WITH substring AND k FOR each substring IN the RESULT OF SPLITTING s BY character
            END IF
        END FOR

        RETURN the LENGTH OF s
    END FUNCTION
END CLASS
