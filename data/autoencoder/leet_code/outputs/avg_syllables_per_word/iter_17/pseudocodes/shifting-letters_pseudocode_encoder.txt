CLASS Solution
    FUNCTION shiftingLetters(string s, list_of_integers shifts)
        SET total_shifts TO zero
        FOR index FROM the LENGTH OF shifts MINUS one TO zero DECREMENT BY one
            INCREMENT total_shifts BY element at position index of shifts
            SET element at position index of shifts TO total_shifts
        END FOR
        SET result TO an empty list
        FOR index FROM zero TO the LENGTH OF s MINUS one
            SET numeric_value TO the numeric unicode value of element at position index of s MINUS the numeric unicode value of lowercase letter a
            SET shifted_value TO numeric_value PLUS element at position index of shifts
            SET wrapped_value TO shifted_value MODULO twenty six
            SET new_character TO the character with the numeric unicode value of wrapped_value PLUS the numeric unicode value of lowercase letter a
            APPEND new_character TO result
        END FOR
        RETURN the concatenation of all elements in result
    END FUNCTION
END CLASS
