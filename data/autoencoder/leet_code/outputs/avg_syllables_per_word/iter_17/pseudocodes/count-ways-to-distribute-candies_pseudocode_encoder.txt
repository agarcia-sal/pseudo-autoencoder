CLASS Solution
    FUNCTION waysToDistribute(n INTEGER, k INTEGER)
        SET MOD TO ten TO THE POWER OF nine PLUS seven
        SET dp TO CALL Initialize2DList OF size n PLUS one AND size k PLUS one WITH zero
        SET element at position zero of list at position zero of dp TO one
        FOR i FROM one TO n
            FOR j FROM one TO k
                SET product TO j MULTIPLIED BY element at position i MINUS one of list at position j of dp
                SET sum_value TO product PLUS element at position i MINUS one of list at position j MINUS one of dp
                SET element at position i of list at position j of dp TO sum_value MODULO MOD
            END FOR
        END FOR
        RETURN element at position n of list at position k of dp
    END FUNCTION
END CLASS
