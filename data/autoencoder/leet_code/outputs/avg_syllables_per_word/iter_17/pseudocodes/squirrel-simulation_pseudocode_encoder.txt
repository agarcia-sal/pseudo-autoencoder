CLASS Solution
    FUNCTION minDistance(height, width, tree, squirrel, nuts)
        FUNCTION distance(point_one, point_two)
            RETURN the ABSOLUTE VALUE of element at position zero of point_one MINUS element at position zero of point_two PLUS the ABSOLUTE VALUE of element at position one of point_one MINUS element at position one of point_two
        END FUNCTION

        SET total_distance TO zero
        SET max_save TO negative infinity

        FOR each nut IN nuts
            INCREMENT total_distance BY two MULTIPLIED BY CALL distance WITH parameters nut AND tree
            SET first_nut_distance TO CALL distance WITH parameters squirrel AND nut PLUS CALL distance WITH parameters nut AND tree
            SET later_nut_distance TO two MULTIPLIED BY CALL distance WITH parameters nut AND tree
            SET save TO later_nut_distance MINUS first_nut_distance
            SET max_save TO the MAXIMUM OF max_save AND save
        END FOR

        RETURN total_distance MINUS max_save
    END FUNCTION
END CLASS
