CLASS Solution
    FUNCTION arrayNesting(list_of_numbers)
        SET visited_list TO a list containing False repeated the LENGTH OF list_of_numbers times
        SET maximum_length TO zero
        
        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one
            IF the element at position index of visited_list IS NOT True
                SET current_length TO zero
                SET pointer k TO index
                WHILE the element at position k of visited_list IS NOT True
                    SET the element at position k of visited_list TO True
                    SET pointer k TO the element at position k of list_of_numbers
                    INCREMENT current_length BY one
                END WHILE
                SET maximum_length TO the greater value BETWEEN maximum_length AND current_length
            END IF
        END FOR
        
        RETURN maximum_length
    END FUNCTION
END CLASS
