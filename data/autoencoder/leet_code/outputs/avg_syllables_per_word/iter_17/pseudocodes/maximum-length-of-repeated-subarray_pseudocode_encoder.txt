CLASS Solution
    FUNCTION findLength(nums1, nums2)
        SET dp TO a two dimensional list of zero with the number of rows EQUAL TO the LENGTH OF nums1 PLUS one and the number of columns EQUAL TO the LENGTH OF nums2 PLUS one
        SET max_length TO zero
        FOR index i FROM one TO the LENGTH OF nums1
            FOR index j FROM one TO the LENGTH OF nums2
                IF element at position i MINUS one OF nums1 EQUALS element at position j MINUS one OF nums2
                    SET element at position i and j OF dp TO element at position i MINUS one and j MINUS one OF dp PLUS one
                    SET max_length TO the GREATER OF max_length AND element at position i and j OF dp
                END IF
            END FOR
        END FOR
        RETURN max_length
    END FUNCTION
END CLASS
