CLASS Solution
    FUNCTION numTilings(n)
        SET MODULO_CONSTANT TO one billion PLUS seven
        
        CALL CreateIntegerList WITH parameter n PLUS one ASSIGN TO dp
        CALL CreateIntegerList WITH parameter n PLUS one ASSIGN TO dp1
        
        SET element at position zero of dp TO one
        SET element at position one of dp TO one
        
        IF n IS GREATER THAN OR EQUAL TO two
            SET element at position two of dp TO two
            SET element at position two of dp1 TO one
        END IF
        
        FOR index FROM three TO n
            SET element at position index of dp TO the SUM OF element at position index MINUS one of dp PLUS element at position index MINUS two of dp PLUS two MULTIPLIED BY element at position index MINUS one of dp1 MODULO MODULO_CONSTANT
            SET element at position index of dp1 TO the SUM OF element at position index MINUS two of dp PLUS element at position index MINUS one of dp1 MODULO MODULO_CONSTANT
        END FOR
        
        RETURN element at position n of dp
    END FUNCTION
END CLASS
