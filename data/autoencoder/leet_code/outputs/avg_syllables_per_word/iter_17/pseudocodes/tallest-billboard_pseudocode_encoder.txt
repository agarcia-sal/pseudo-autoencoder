CLASS Solution
    FUNCTION tallestBillboard(rods)
        SET dp TO a dictionary with key zero and value zero
        FOR each rod IN rods
            SET new_dp TO a copy of dp
            FOR each key_value_pair d comma y IN dp
                SET sum_key_taller TO d PLUS rod
                SET current_max_taller TO the value at key sum_key_taller IN new_dp OR zero if absent
                SET updated_max_taller TO the maximum of current_max_taller AND y
                SET the value at key sum_key_taller IN new_dp TO updated_max_taller

                SET absolute_difference TO the absolute value of d MINUS rod
                SET current_max_shorter TO the value at key absolute_difference IN new_dp OR zero if absent
                SET sum_shorter TO y PLUS the minimum of d AND rod
                SET updated_max_shorter TO the maximum of current_max_shorter AND sum_shorter
                SET the value at key absolute_difference IN new_dp TO updated_max_shorter
            END FOR
            SET dp TO new_dp
        END FOR
        RETURN the value at key zero IN dp
    END FUNCTION
END CLASS
