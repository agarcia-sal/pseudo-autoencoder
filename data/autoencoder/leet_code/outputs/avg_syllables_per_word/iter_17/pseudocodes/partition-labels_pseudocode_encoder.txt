CLASS Solution
    FUNCTION partitionLabels(string_s)
        SET last_occurrence_dictionary TO CALL create_last_occurrence_dictionary WITH parameter string_s
        SET start_index TO zero
        SET end_index TO zero
        SET partition_sizes_list TO empty list
        FOR each index_and_character IN enumeration_of string_s
            EXTRACT index FROM index_and_character
            EXTRACT character FROM index_and_character
            SET end_index TO the MAXIMUM OF end_index AND the element AT position character OF last_occurrence_dictionary
            IF index EQUALS end_index
                SET partition_size TO end_index MINUS start_index PLUS one
                APPEND partition_size TO partition_sizes_list
                SET start_index TO index PLUS one
            END IF
        END FOR
        RETURN partition_sizes_list
    END FUNCTION
    
FUNCTION create_last_occurrence_dictionary(string_s)
    IMPLEMENT mapping_of character TO last position OF character IN string_s
END FUNCTION
END CLASS
