CLASS Solution
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)
        SET n TO the LENGTH OF floor
        SET dp TO A two dimensional list with n PLUS one rows and numCarpets PLUS one columns INITIALIZED TO zero

        FOR i FROM one TO n
            SET element AT position i AND zero OF dp TO element AT position i MINUS one AND zero OF dp PLUS 
                ONE IF element AT position i MINUS one OF floor EQUALS one ELSE zero
        END FOR

        FOR j FROM one TO numCarpets
            FOR i FROM one TO n
                SET element AT position i AND j OF dp TO element AT position i MINUS one AND j OF dp PLUS 
                    ONE IF element AT position i MINUS one OF floor EQUALS one ELSE zero
                SET element AT position i AND j OF dp TO THE MINIMUM OF element AT position i AND j OF dp AND 
                    element AT position THE GREATER OF zero AND i MINUS carpetLen AND j MINUS one OF dp
            END FOR
        END FOR

        RETURN element AT position n AND numCarpets OF dp
    END FUNCTION
END CLASS
