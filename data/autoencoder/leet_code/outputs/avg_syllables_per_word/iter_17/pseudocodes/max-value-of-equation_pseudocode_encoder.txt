CLASS Solution
    FUNCTION findMaxValueOfEquation(points, k)
        SET max_value TO negative infinity
        ASSIGN deque_ TO an empty double ended queue
        
        FOR each pair xj and yj IN points
            WHILE deque_ is not empty AND xj MINUS element at position one of element at position zero of deque_ GREATER THAN k
                REMOVE element from the front of deque_
            END WHILE
            
            IF deque_ is not empty
                SET max_value TO the GREATER OF max_value AND yj PLUS xj PLUS element at position zero of element at position zero of deque_
            END IF
            
            WHILE deque_ is not empty AND yj MINUS xj GREATER THAN OR EQUAL TO element at position zero of element at position last of deque_
                REMOVE element from the back of deque_
            END WHILE
            
            APPEND the pair yj MINUS xj AND xj TO deque_
        END FOR
        
        RETURN max_value
    END FUNCTION
END CLASS
