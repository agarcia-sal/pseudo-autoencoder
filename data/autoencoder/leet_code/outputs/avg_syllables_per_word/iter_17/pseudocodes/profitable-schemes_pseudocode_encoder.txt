CLASS Solution
    FUNCTION profitableSchemes(n, minProfit, group, profit)
        SET MODULO TO one billion PLUS seven
        CALL initialize_dp TO create a two dimensional list dp WITH n PLUS one rows AND each row containing minProfit PLUS one zero elements
        FOR index FROM zero TO n
            SET element at position zero of dp at position index TO one
        END FOR
        FOR each profit_value AND group_value IN profit AND group
            FOR member_count FROM n DOWN TO group_value
                FOR profit_count FROM minProfit DOWN TO zero
                    SET previous_profit_index TO maximum of zero AND profit_count MINUS profit_value
                    SET dp at position member_count, profit_count TO dp at position member_count, profit_count PLUS dp at position member_count MINUS group_value, previous_profit_index MODULO MODULO
                END FOR
            END FOR
        END FOR
        RETURN dp at position n, minProfit
    END FUNCTION
END CLASS
