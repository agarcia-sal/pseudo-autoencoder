CLASS Solution
    FUNCTION findMaxForm(list_of_strings, maximum_zeros, maximum_ones)
        SET dynamic_programming_table TO CALL initialize_dp_table WITH maximum_zeros AND maximum_ones
        FOR each string IN list_of_strings
            SET count_zeros TO CALL count_character_occurrences WITH string AND character_zero
            SET count_ones TO CALL count_character_occurrences WITH string AND character_one
            FOR index_i FROM maximum_zeros DOWN TO count_zeros
                FOR index_j FROM maximum_ones DOWN TO count_ones
                    SET value_with_current TO dynamic_programming_table at position index_i MINUS count_zeros AND index_j MINUS count_ones PLUS one
                    SET current_value TO dynamic_programming_table at position index_i AND index_j
                    SET dynamic_programming_table at position index_i AND index_j TO the GREATER OF current_value AND value_with_current
                END FOR
            END FOR
        END FOR
        RETURN dynamic_programming_table at position maximum_zeros AND maximum_ones
    END FUNCTION
END CLASS
