CLASS Solution
    FUNCTION threeSumClosest(list_of_numbers, target_value)
        CALL list_of_numbers SORT
        SET closest_sum TO positive_infinity
        
        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS two
            SET left_pointer TO index PLUS one
            SET right_pointer TO the LENGTH OF list_of_numbers MINUS one
            
            WHILE left_pointer LESS THAN right_pointer
                SET current_sum TO element at position index of list_of_numbers PLUS element at position left_pointer of list_of_numbers PLUS element at position right_pointer of list_of_numbers
                
                IF the ABSOLUTE VALUE OF current_sum MINUS target_value LESS THAN the ABSOLUTE VALUE OF closest_sum MINUS target_value
                    SET closest_sum TO current_sum
                END IF
                
                IF current_sum LESS THAN target_value
                    INCREMENT left_pointer BY one
                ELSE IF current_sum GREATER THAN target_value
                    DECREMENT right_pointer BY one
                ELSE
                    RETURN current_sum
                END IF
            END WHILE
        END FOR
        
        RETURN closest_sum
    END FUNCTION
END CLASS
