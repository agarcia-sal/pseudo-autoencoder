CLASS Solution
    FUNCTION leastBricks(wall)
        CALL initialize_default_dictionary_of_integers TO gap_counts
        
        FOR each row IN wall
            SET total_length TO zero
            
            FOR each brick IN all elements except the last element of row
                INCREMENT total_length BY brick
                INCREMENT the element at position total_length of gap_counts BY one
            END FOR
        END FOR
        
        SET max_gaps TO the maximum value among all values of gap_counts OR zero if none exist
        SET number_of_rows TO the LENGTH OF wall
        SET minimum_crossed_bricks TO number_of_rows MINUS max_gaps
        RETURN minimum_crossed_bricks
    END FUNCTION
END CLASS
