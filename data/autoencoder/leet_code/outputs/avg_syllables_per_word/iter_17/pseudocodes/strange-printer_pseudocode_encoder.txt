CLASS Solution
    FUNCTION strangePrinter(string_s)
        FUNCTION dp(index_i, index_j, dictionary_memo)
            IF index_i GREATER THAN index_j
                RETURN zero
            END IF
            IF the tuple of index_i and index_j IS IN dictionary_memo
                RETURN the element at key tuple of index_i and index_j in dictionary_memo
            END IF

            SET result_value TO CALL dp WITH parameters index_i, index_j MINUS one, dictionary_memo PLUS one

            FOR index_k FROM index_i TO index_j MINUS one
                IF the element at position index_k of string_s EQUALS the element at position index_j of string_s
                    SET candidate_value TO CALL dp WITH parameters index_i, index_k MINUS one, dictionary_memo PLUS CALL dp WITH parameters index_k, index_j MINUS one, dictionary_memo
                    SET result_value TO the MINIMUM OF result_value AND candidate_value
                END IF
            END FOR

            SET the element at key tuple of index_i and index_j in dictionary_memo TO result_value
            RETURN result_value
        END FUNCTION

        RETURN CALL dp WITH parameters zero, the LENGTH OF string_s MINUS one, an EMPTY dictionary
    END FUNCTION
END CLASS
