CLASS Solution
    FUNCTION canDistribute(list_of_numbers nums, list_of_integers quantity)
        SET count TO the LIST OF the VALUES OF the COUNTER OF list_of_numbers nums
        CALL sort ON list quantity WITH parameter reverse EQUALS True

        FUNCTION can_satisfy(integer index)
            IF index EQUALS the LENGTH OF list quantity
                RETURN True
            END IF
            FOR i FROM zero TO the LENGTH OF list count MINUS one
                IF element at position i of list count GREATER THAN OR EQUAL TO element at position index of list quantity
                    DECREMENT element at position i of list count BY element at position index of list quantity
                    IF CALL can_satisfy WITH parameter index PLUS one RETURNS True
                        RETURN True
                    END IF
                    INCREMENT element at position i of list count BY element at position index of list quantity
                END IF
            END FOR
            RETURN False
        END FUNCTION

        RETURN CALL can_satisfy WITH parameter zero
    END FUNCTION
END CLASS
