CLASS Solution
    FUNCTION findDiagonalOrder(matrix)
        IF matrix IS empty OR element at position zero of matrix IS empty
            RETURN an empty list
        END IF
        SET number_of_rows TO the LENGTH OF matrix
        SET number_of_columns TO the LENGTH OF element at position zero of matrix
        SET result_list TO an empty list
        SET row TO zero
        SET column TO zero
        SET direction TO one

        FOR iterator FROM zero TO number_of_rows MULTIPLIED BY number_of_columns MINUS one
            APPEND element at position row of matrix element at position column TO result_list
            IF direction EQUALS one
                SET new_row TO row MINUS one
                SET new_column TO column PLUS one
            ELSE
                SET new_row TO row PLUS one
                SET new_column TO column MINUS one
            END IF

            IF new_row LESS THAN zero OR new_row EQUALS number_of_rows OR new_column LESS THAN zero OR new_column EQUALS number_of_columns
                IF direction EQUALS one
                    IF new_column EQUALS number_of_columns
                        SET new_row TO row PLUS one
                        SET new_column TO column
                    ELSE
                        SET new_row TO zero
                        SET new_column TO column PLUS one
                    END IF
                ELSE
                    IF new_row EQUALS number_of_rows
                        SET new_row TO row
                        SET new_column TO column PLUS one
                    ELSE
                        SET new_row TO row PLUS one
                        SET new_column TO zero
                    END IF
                END IF
                SET direction TO direction MULTIPLIED BY negative one
            END IF

            SET row TO new_row
            SET column TO new_column
        END FOR

        RETURN result_list
    END FUNCTION
END CLASS
