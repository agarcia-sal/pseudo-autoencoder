CLASS Solution
    FUNCTION isReflected(points AS list of list of integers) RETURNS boolean
        SET unique_points TO the SET OF tuples CONVERTED FROM list points
        SET min_x TO the MINIMUM x coordinate FROM element at position zero OF each point IN unique_points
        SET max_x TO the MAXIMUM x coordinate FROM element at position zero OF each point IN unique_points
        SET line_of_reflection TO min_x PLUS max_x DIVIDED BY two
        FOR each point WITH x coordinate AND y coordinate IN unique_points
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x coordinate
            IF the tuple of reflected_x and y coordinate IS NOT IN unique_points
                RETURN false
            END IF
        END FOR
        RETURN true
    END FUNCTION
END CLASS
