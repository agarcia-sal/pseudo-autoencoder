CLASS Solution
    FUNCTION change(amount, coins)
        SET list_of_ways TO a list of zeros with length amount PLUS one
        SET element at position zero of list_of_ways TO one
        FOR each coin IN coins
            FOR index FROM coin TO amount
                INCREMENT element at position index of list_of_ways BY element at position index MINUS coin of list_of_ways
            END FOR
        END FOR
        RETURN element at position amount of list_of_ways
    END FUNCTION
END CLASS
