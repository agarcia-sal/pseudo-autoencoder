CLASS Solution
    FUNCTION validIPAddress(queryIP)
        FUNCTION is_valid_ipv4_block(block)
            IF block IS DIGIT STRING
                SET num TO CONVERT block TO INTEGER
                IF num LESS THAN OR EQUAL TO two hundred fifty five AND CONVERT num TO STRING EQUALS block
                    RETURN True
                END IF
                RETURN False
            END IF
            RETURN False
        END FUNCTION

        FUNCTION is_valid_ipv6_block(block)
            IF the LENGTH OF block GREATER THAN OR EQUAL TO one AND the LENGTH OF block LESS THAN OR EQUAL TO four
                TRY
                    CALL CONVERT block TO INTEGER WITH base sixteen
                    RETURN True
                EXCEPT VALUE ERROR
                    RETURN False
                END TRY
            END IF
            RETURN False
        END FUNCTION

        SET parts_ipv4 TO SPLIT queryIP BY period
        SET parts_ipv6 TO SPLIT queryIP BY colon

        IF the LENGTH OF parts_ipv4 EQUALS four AND ALL elements in parts_ipv4 SATISFY is_valid_ipv4_block
            RETURN IPv4
        ELSE IF the LENGTH OF parts_ipv6 EQUALS eight AND ALL elements in parts_ipv6 SATISFY is_valid_ipv6_block
            RETURN IPv6
        ELSE
            RETURN Neither
        END IF
    END FUNCTION
END CLASS
