CLASS Solution
    FUNCTION recoverArray(nums)
        CALL nums SORT
        SET n TO the LENGTH OF nums
        SET first TO the element at position zero of nums

        FOR i FROM one TO n MINUS one
            SET k TO the element at position i of nums MINUS first DIVIDED BY two
            IF k LESS THAN OR EQUAL TO zero OR first PLUS two MULTIPLIED BY k NOT EQUALS element at position i of nums
                CONTINUE
            END IF

            SET count TO CALL Counter WITH nums
            SET arr TO an empty list

            FOR each x IN nums
                IF the element at position x of count EQUALS zero
                    CONTINUE
                END IF
                IF the element at position x PLUS two MULTIPLIED BY k of count EQUALS zero
                    BREAK
                END IF
                APPEND x PLUS k TO arr
                DECREMENT the element at position x of count BY one
                DECREMENT the element at position x PLUS two MULTIPLIED BY k of count BY one
            END FOR

            IF the LENGTH OF arr EQUALS n DIVIDED BY two
                RETURN arr
            END IF
        END FOR

        RETURN an empty list
    END FUNCTION
END CLASS
