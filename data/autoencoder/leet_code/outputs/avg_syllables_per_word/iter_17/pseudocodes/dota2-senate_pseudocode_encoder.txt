CLASS Solution
    FUNCTION predictPartyVictory(senate)
        ASSIGN radiant TO CALL deque WITH empty list
        ASSIGN dire TO CALL deque WITH empty list

        FOR each index_and_senator IN CALL enumerate WITH senate
            SET i TO element at position zero of index_and_senator
            SET s TO element at position one of index_and_senator
            IF s EQUALS character R
                CALL APPEND WITH i TO radiant
            ELSE
                CALL APPEND WITH i TO dire
            END IF
        END FOR

        WHILE radiant IS NOT empty AND dire IS NOT empty
            SET r_index TO CALL POP LEFT FROM radiant
            SET d_index TO CALL POP LEFT FROM dire

            IF r_index LESS THAN d_index
                CALL APPEND WITH r_index PLUS the LENGTH OF senate TO radiant
            ELSE
                CALL APPEND WITH d_index PLUS the LENGTH OF senate TO dire
            END IF
        END WHILE

        IF radiant IS NOT empty
            RETURN Radiant
        ELSE
            RETURN Dire
        END IF
    END FUNCTION
END CLASS
