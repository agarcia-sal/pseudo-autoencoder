CLASS Solution
    FUNCTION reverseWords(list_of_characters)
        CALL reverse ON list_of_characters
        SET start TO zero
        FOR end FROM zero TO the LENGTH OF list_of_characters MINUS one
            IF element at position end of list_of_characters EQUALS space
                CALL reverseWordsInPlace WITH parameters list_of_characters start TO end MINUS one
                SET start TO end PLUS one
            END IF
        END FOR
        CALL reverseWordsInPlace WITH parameters list_of_characters start TO the LENGTH OF list_of_characters MINUS one
    END FUNCTION

    FUNCTION reverseWordsInPlace(list_of_characters start_index TO end_index)
        SET left TO start_index
        SET right TO end_index
        WHILE left IS LESS THAN right
            SET temporary_variable TO element at position left of list_of_characters
            SET element at position left of list_of_characters TO element at position right of list_of_characters
            SET element at position right of list_of_characters TO temporary_variable
            INCREMENT left BY one
            DECREMENT right BY one
        END WHILE
    END FUNCTION
END CLASS
