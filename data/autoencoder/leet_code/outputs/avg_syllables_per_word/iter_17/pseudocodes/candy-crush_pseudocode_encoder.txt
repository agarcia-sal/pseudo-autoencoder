CLASS Solution
    FUNCTION candyCrush(board)
        SET R TO the LENGTH OF board
        SET C TO the LENGTH OF element at position zero of board
        SET stable TO False
        
        WHILE stable IS NOT True
            SET stable TO True
            SET to_crush TO an EMPTY SET
            
            FOR r FROM zero TO R MINUS one
                FOR c FROM zero TO C MINUS two
                    IF the ABSOLUTE VALUE OF element at position r of board at position c EQUALS the ABSOLUTE VALUE OF element at position r of board at position c PLUS one AND the ABSOLUTE VALUE OF element at position r of board at position c EQUALS the ABSOLUTE VALUE OF element at position r of board at position c PLUS two AND the ABSOLUTE VALUE OF element at position r of board at position c IS NOT EQUALS TO zero
                        SET stable TO False
                        CALL to_crush UPDATE WITH the TUPLE of r AND c AND the TUPLE of r AND c PLUS one AND the TUPLE of r AND c PLUS two
                    END IF
                END FOR
            END FOR
            
            FOR r FROM zero TO R MINUS two
                FOR c FROM zero TO C MINUS one
                    IF the ABSOLUTE VALUE OF element at position r of board at position c EQUALS the ABSOLUTE VALUE OF element at position r PLUS one of board at position c AND the ABSOLUTE VALUE OF element at position r of board at position c EQUALS the ABSOLUTE VALUE OF element at position r PLUS two of board at position c AND the ABSOLUTE VALUE OF element at position r of board at position c IS NOT EQUALS TO zero
                        SET stable TO False
                        CALL to_crush UPDATE WITH the TUPLE of r AND c AND the TUPLE of r PLUS one AND c AND the TUPLE of r PLUS two AND c
                    END IF
                END FOR
            END FOR
            
            FOR each TUPLE of r AND c IN to_crush
                SET element at position r of board at position c TO zero
            END FOR
            
            FOR c FROM zero TO C MINUS one
                SET wr TO R MINUS one
                FOR r FROM R MINUS one TO zero DECREASING BY one
                    IF element at position r of board at position c IS NOT EQUALS TO zero
                        SET element at position wr of board at position c TO element at position r of board at position c
                        DECREMENT wr BY one
                    END IF
                END FOR
                FOR wr FROM wr TO zero DECREASING BY one
                    SET element at position wr of board at position c TO zero
                END FOR
            END FOR
        END WHILE
        
        RETURN board
    END FUNCTION
END CLASS
