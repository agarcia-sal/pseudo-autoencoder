CLASS Solution
    FUNCTION generateParenthesis(number_n)
        FUNCTION backtrack(string_s SET TO empty string, left_count SET TO zero, right_count SET TO zero)
            IF the LENGTH OF string_s EQUALS two MULTIPLIED BY number_n
                APPEND string_s TO result_list
                RETURN
            END IF
            IF left_count LESS THAN number_n
                CALL backtrack WITH string_s PLUS open parenthesis, left_count PLUS one, right_count
            END IF
            IF right_count LESS THAN left_count
                CALL backtrack WITH string_s PLUS close parenthesis, left_count, right_count PLUS one
            END IF
        END FUNCTION
        SET result_list TO empty list
        CALL backtrack WITH no parameters
        RETURN result_list
    END FUNCTION
END CLASS
