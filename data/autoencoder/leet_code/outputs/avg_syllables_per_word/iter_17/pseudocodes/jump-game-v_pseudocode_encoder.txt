CLASS Solution
    FUNCTION maxJumps(arr, d)
        FUNCTION dp(i)
            SET max_jumps TO one
            FOR x FROM one TO d
                IF i PLUS x LESS THAN the LENGTH OF arr
                    IF element at position i PLUS x of arr GREATER THAN OR EQUAL TO element at position i of arr
                        BREAK
                    END IF
                    SET max_jumps TO the GREATER OF max_jumps AND one PLUS CALL dp WITH i PLUS x
                END IF
            END FOR
            FOR x FROM one TO d
                IF i MINUS x GREATER THAN OR EQUAL TO zero
                    IF element at position i MINUS x of arr GREATER THAN OR EQUAL TO element at position i of arr
                        BREAK
                    END IF
                    SET max_jumps TO the GREATER OF max_jumps AND one PLUS CALL dp WITH i MINUS x
                END IF
            END FOR
            RETURN max_jumps
        END FUNCTION
        RETURN the GREATER OF CALL dp WITH i FOR each i FROM zero TO the LENGTH OF arr MINUS one
    END FUNCTION
END CLASS
