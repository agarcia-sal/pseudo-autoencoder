CLASS Solution
    FUNCTION maxPerformance(n, speed, efficiency, k)
        SET list_of_engineers TO CALL zip WITH parameters efficiency AND speed CONVERTED TO list
        CALL sort ON list_of_engineers IN descending order
        SET min_heap_of_speeds TO empty list
        SET sum_of_speeds TO zero
        SET maximum_performance TO zero

        FOR each pair_of efficiency_value AND speed_value IN list_of_engineers
            CALL heappush ON min_heap_of_speeds WITH parameter speed_value
            INCREMENT sum_of_speeds BY speed_value

            IF the LENGTH OF min_heap_of_speeds GREATER THAN k
                SET removed_speed TO CALL heappop ON min_heap_of_speeds
                DECREMENT sum_of_speeds BY removed_speed
            END IF

            SET current_performance TO sum_of_speeds MULTIPLIED BY efficiency_value
            SET maximum_performance TO the GREATER OF maximum_performance AND current_performance
        END FOR

        RETURN maximum_performance MODULO ten TO THE POWER OF nine PLUS seven
    END FUNCTION
END CLASS
