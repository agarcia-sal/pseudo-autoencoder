CLASS Solution
    FUNCTION uniqueLetterString(s)
        SET MODULO_VALUE TO ten TO THE POWER OF nine PLUS seven
        SET string_length TO the LENGTH OF s

        SET list_last_positions TO a list of negative one repeated twenty six times
        SET list_second_last_positions TO a list of negative one repeated twenty six times

        SET total_result TO zero

        FOR each index i AND character char IN s
            SET index_position TO the ORDINAL VALUE OF character MINUS the ORDINAL VALUE OF uppercase A

            SET total_result TO the total_result PLUS the quantity of (index i MINUS element at position index_position of list_last_positions) MULTIPLIED BY (element at position index_position of list_last_positions MINUS element at position index_position of list_second_last_positions) MODULO MODULO_VALUE

            SET element at position index_position of list_second_last_positions TO element at position index_position of list_last_positions
            SET element at position index_position of list_last_positions TO index i
        END FOR

        FOR index i FROM zero TO twenty five
            SET total_result TO the total_result PLUS the quantity of (string_length MINUS element at position i of list_last_positions) MULTIPLIED BY (element at position i of list_last_positions MINUS element at position i of list_second_last_positions) MODULO MODULO_VALUE
        END FOR

        RETURN total_result
    END FUNCTION
END CLASS
