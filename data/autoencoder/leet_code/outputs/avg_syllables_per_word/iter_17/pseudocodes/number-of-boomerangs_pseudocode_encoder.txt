CLASS Solution
    FUNCTION numberOfBoomerangs(points: list_of_list_of_integers)
        SET answer TO zero
        FOR each point_one IN points
            CALL create_counter SET TO count_map
            FOR each point_two IN points
                SET delta_x TO element at position zero of point_one MINUS element at position zero of point_two
                SET delta_y TO element at position one of point_one MINUS element at position one of point_two
                SET distance_squared TO delta_x MULTIPLIED BY delta_x PLUS delta_y MULTIPLIED BY delta_y
                INCREMENT answer BY count_map AT key distance_squared
                INCREMENT count_map AT key distance_squared BY one
            END FOR
        END FOR
        RETURN answer MULTIPLIED BY two
    END FUNCTION
END CLASS
