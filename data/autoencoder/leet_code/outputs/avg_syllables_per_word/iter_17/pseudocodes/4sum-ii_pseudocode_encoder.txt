CLASS Solution
    FUNCTION fourSumCount(nums1, nums2, nums3, nums4)
        SET sum_ab TO an empty dictionary
        FOR each element a IN nums1
            FOR each element b IN nums2
                SET current_sum TO a PLUS b
                IF current_sum IS IN sum_ab
                    INCREMENT the value at current_sum in sum_ab BY one
                ELSE
                    SET the value at current_sum in sum_ab TO one
                END IF
            END FOR
        END FOR

        SET count TO zero

        FOR each element c IN nums3
            FOR each element d IN nums4
                SET target TO zero MINUS the sum of c PLUS d
                IF target IS IN sum_ab
                    INCREMENT count BY the value at target in sum_ab
                END IF
            END FOR
        END FOR

        RETURN count
    END FUNCTION
END CLASS
