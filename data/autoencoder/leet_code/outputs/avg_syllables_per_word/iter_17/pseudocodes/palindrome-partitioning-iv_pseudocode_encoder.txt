CLASS Solution
    FUNCTION checkPartitioning(string_input)
        FUNCTION is_palindrome(substring_input)
            RETURN substring_input EQUALS substring_input REVERSED
        END FUNCTION

        SET length_n TO the LENGTH OF string_input

        CALL initialize_palindrome_status WITH parameters length_n ASSIGN TO palindrome_status_matrix

        FOR index_i FROM length_n MINUS one TO zero STEP minus one
            FOR index_j FROM index_i TO length_n MINUS one
                IF element at position index_i of string_input EQUALS element at position index_j of string_input AND 
                   (index_j MINUS index_i LESS THAN OR EQUAL TO one OR element at position index_i PLUS one and position index_j MINUS one of palindrome_status_matrix IS True)
                    SET element at position index_i and position index_j of palindrome_status_matrix TO True
                END IF
            END FOR
        END FOR

        FOR first_partition_index FROM one TO length_n MINUS two
            IF element at position zero and position first_partition_index MINUS one of palindrome_status_matrix IS True
                FOR second_partition_index FROM first_partition_index TO length_n MINUS two
                    IF element at position first_partition_index and position second_partition_index of palindrome_status_matrix IS True AND 
                       element at position second_partition_index PLUS one and position length_n MINUS one of palindrome_status_matrix IS True
                        RETURN True
                    END IF
                END FOR
            END IF
        END FOR

        RETURN False
    END FUNCTION

    FUNCTION initialize_palindrome_status(length_n)
        RETURN auxiliary FUNCTION THAT CREATES A TWO DIMENSIONAL COLLECTION OF size length_n BY length_n FILLED WITH False
    END FUNCTION
END CLASS
