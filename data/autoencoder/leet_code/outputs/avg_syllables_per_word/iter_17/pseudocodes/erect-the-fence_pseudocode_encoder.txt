CLASS Solution
    FUNCTION outerTrees(trees)
        FUNCTION cross(i, j, k)
            SET point_a TO element at position i of trees
            SET point_b TO element at position j of trees
            SET point_c TO element at position k of trees
            RETURN the RESULT OF (the FIRST coordinate of point_b MINUS the FIRST coordinate of point_a) MULTIPLIED BY (the SECOND coordinate of point_c MINUS the SECOND coordinate of point_b) MINUS (the SECOND coordinate of point_b MINUS the SECOND coordinate of point_a) MULTIPLIED BY (the FIRST coordinate of point_c MINUS the FIRST coordinate of point_b)
        END FUNCTION
        SET n TO the LENGTH OF trees
        IF n LESS THAN four
            RETURN trees
        END IF
        CALL the SORT method ON trees
        SET vis TO a list of Boolean FALSE values REPEATED n times
        SET stk TO a list containing zero
        FOR i FROM one TO n MINUS one
            WHILE the LENGTH OF stk GREATER THAN one AND the RESULT OF cross(element at position LENGTH OF stk MINUS two of stk, element at position LENGTH OF stk MINUS one of stk, i) LESS THAN zero HOLDS
                SET vis at element popped FROM stk TO Boolean FALSE
            END WHILE
            SET vis at position i TO Boolean TRUE
            APPEND i TO stk
        END FOR
        SET m TO the LENGTH OF stk
        FOR i FROM n MINUS two DOWN TO zero
            IF vis at position i IS Boolean TRUE
                CONTINUE to the NEXT iteration
            END IF
            WHILE the LENGTH OF stk GREATER THAN m AND the RESULT OF cross(element at position LENGTH OF stk MINUS two of stk, element at position LENGTH OF stk MINUS one of stk, i) LESS THAN zero HOLDS
                REMOVE element at position LENGTH OF stk MINUS one FROM stk
            END WHILE
            APPEND i TO stk
        END FOR
        REMOVE element at position LENGTH OF stk MINUS one FROM stk
        RETURN a list COMPRISING element at position index of trees FOR each index IN stk
    END FUNCTION
END CLASS
