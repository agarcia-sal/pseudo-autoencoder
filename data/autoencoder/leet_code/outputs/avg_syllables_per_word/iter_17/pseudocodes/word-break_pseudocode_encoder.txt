CLASS Solution
    FUNCTION wordBreak(s string_parameter, wordDict list_of_strings)
        SET wordSet TO CALL CreateSetFromList WITH wordDict
        SET dp TO CALL CreateListOfBooleans WITH the LENGTH OF s PLUS one
        SET element at position zero of dp TO true
        FOR index i FROM one TO the LENGTH OF s
            FOR index j FROM zero TO i MINUS one
                IF element at position j of dp IS true AND substring from position j TO i MINUS one of s IS IN wordSet
                    SET element at position i of dp TO true
                    BREAK inner loop
                END IF
            END FOR
        END FOR
        RETURN element at position the LENGTH OF s of dp
    END FUNCTION
END CLASS
