CLASS Solution
    FUNCTION convertArray(list_of_numbers)
        FUNCTION min_operations_to_non_decreasing(list_of_numbers)
            SET max_heap TO empty list
            SET operations TO zero
            FOR each number IN list_of_numbers
                IF max_heap IS NOT empty AND the NEGATION OF element at position zero of max_heap GREATER THAN number
                    INCREMENT operations BY the NEGATION OF CALL heap_pop FROM max_heap MINUS number
                    CALL heap_push ON max_heap WITH the NEGATION OF number
                END IF
                CALL heap_push ON max_heap WITH the NEGATION OF number
            END FOR
            RETURN operations
        END FUNCTION

        FUNCTION min_operations_to_non_increasing(list_of_numbers)
            SET min_heap TO empty list
            SET operations TO zero
            FOR each number IN list_of_numbers
                IF min_heap IS NOT empty AND element at position zero of min_heap LESS THAN number
                    INCREMENT operations BY number MINUS CALL heap_pop FROM min_heap
                    CALL heap_push ON min_heap WITH number
                END IF
                CALL heap_push ON min_heap WITH number
            END FOR
            RETURN operations
        END FUNCTION

        RETURN the MINIMUM OF CALL min_operations_to_non_decreasing WITH list_of_numbers AND CALL min_operations_to_non_increasing WITH list_of_numbers
END CLASS
