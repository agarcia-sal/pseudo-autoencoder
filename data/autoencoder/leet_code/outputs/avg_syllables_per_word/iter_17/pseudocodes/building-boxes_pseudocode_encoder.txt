CLASS Solution
    FUNCTION minimumBoxes(integer_n)
        FUNCTION triangular_pyramid(integer_k)
            RETURN integer_k MULTIPLIED BY (integer_k PLUS one) MULTIPLIED BY (integer_k PLUS two) DIVIDED BY six
        END FUNCTION

        SET integer_left TO zero
        SET integer_right TO integer_n
        WHILE integer_left LESS THAN integer_right holds
            SET integer_mid TO (integer_left PLUS integer_right PLUS one) DIVIDED BY two
            IF triangular_pyramid(integer_mid) LESS THAN OR EQUAL TO integer_n
                SET integer_left TO integer_mid
            ELSE
                SET integer_right TO integer_mid MINUS one
            END IF
        END WHILE

        SET boxes_used TO triangular_pyramid(integer_left)
        SET integer_remaining TO integer_n MINUS boxes_used
        SET floor_boxes TO integer_left MULTIPLIED BY (integer_left PLUS one) DIVIDED BY two

        SET additional_boxes TO zero
        WHILE integer_remaining GREATER THAN zero holds
            INCREMENT additional_boxes BY one
            DECREMENT integer_remaining BY additional_boxes
        END WHILE

        RETURN floor_boxes PLUS additional_boxes
    END FUNCTION
END CLASS
