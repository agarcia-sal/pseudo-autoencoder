CLASS Solution
    FUNCTION areConnected(n, threshold, queries)
        IF threshold EQUALS zero
            RETURN a list of True repeated the LENGTH OF queries
        END IF

        SET parent TO the LIST OF numbers FROM zero TO n

        FUNCTION find(x)
            IF element at position x of parent DOES NOT EQUAL x
                SET element at position x of parent TO CALL find WITH element at position x of parent
            END IF
            RETURN element at position x of parent
        END FUNCTION

        FUNCTION union(x, y)
            SET rootX TO CALL find WITH x
            SET rootY TO CALL find WITH y
            IF rootX DOES NOT EQUAL rootY
                SET element at position rootY of parent TO rootX
            END IF
        END FUNCTION

        FOR i FROM threshold PLUS one TO n
            FOR j FROM two TIMES i TO n STEP i
                CALL union WITH i AND j
            END FOR
        END FOR

        SET result TO an empty list

        FOR each pair a comma b IN queries
            APPEND the EQUALITY of CALL find WITH a AND CALL find WITH b TO result
        END FOR

        RETURN result
    END FUNCTION
END CLASS
