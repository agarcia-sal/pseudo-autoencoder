CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN None
    END IF
    SET root TO CALL TreeNode WITH element at position zero of values
    SET i TO one
    SET queue TO CALL deque WITH no parameters
    CALL queue APPEND WITH root
    WHILE queue IS NOT empty
        SET node TO CALL queue POP LEFT WITH no parameters
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.left TO CALL TreeNode WITH element at position i of values
            CALL queue APPEND WITH node.left
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.right TO CALL TreeNode WITH element at position i of values
            CALL queue APPEND WITH node.right
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUALS q.val
        RETURN False
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION levelOrderBottom(self, root)
        IF root IS None
            RETURN empty list
        END IF
        SET result TO empty list
        SET queue TO CALL deque WITH list containing root
        WHILE queue IS NOT empty
            SET level_size TO the LENGTH OF queue
            SET current_level TO empty list
            FOR each index FROM zero TO level_size MINUS one
                SET node TO CALL queue POP LEFT WITH no parameters
                CALL current_level APPEND WITH node.val
                IF node.left IS NOT None
                    CALL queue APPEND WITH node.left
                END IF
                IF node.right IS NOT None
                    CALL queue APPEND WITH node.right
                END IF
            END FOR
            CALL result APPEND WITH current_level
        END WHILE
        RETURN the REVERSE OF result
    END FUNCTION
END CLASS
