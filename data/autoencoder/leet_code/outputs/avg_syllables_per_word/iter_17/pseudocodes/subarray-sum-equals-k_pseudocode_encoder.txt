CLASS Solution
    FUNCTION subarraySum(list_of_numbers, target_value)
        CALL initialize_dictionary_with_zero_key_and_value_one
            SET cumulative_sum_frequency TO the dictionary result
        SET current_cumulative_sum TO zero
        SET subarray_count TO zero
        FOR each number IN list_of_numbers
            INCREMENT current_cumulative_sum BY number
            IF the difference BETWEEN current_cumulative_sum AND target_value IS IN cumulative_sum_frequency
                INCREMENT subarray_count BY the value AT the key OF the difference BETWEEN current_cumulative_sum AND target_value IN cumulative_sum_frequency
            END IF
            IF current_cumulative_sum IS IN cumulative_sum_frequency
                INCREMENT the value AT the key current_cumulative_sum IN cumulative_sum_frequency BY one
            ELSE
                SET the value AT the key current_cumulative_sum IN cumulative_sum_frequency TO one
            END IF
        END FOR
        RETURN subarray_count
    END FUNCTION
END CLASS
