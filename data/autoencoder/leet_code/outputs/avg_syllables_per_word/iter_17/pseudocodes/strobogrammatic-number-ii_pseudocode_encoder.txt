CLASS Solution
    FUNCTION findStrobogrammatic(n)
        FUNCTION dfs(u)
            IF u EQUALS zero
                RETURN list containing empty string
            END IF
            IF u EQUALS one
                RETURN list containing zero one eight
            END IF
            SET ans TO empty list
            FOR each v IN CALL dfs WITH parameter u MINUS two
                FOR each pair (l, r) IN list of pairs eleven eighty eight sixty nine ninety six
                    APPEND l CONCATENATED WITH v CONCATENATED WITH r TO ans
                END FOR
                IF u NOT EQUALS n
                    APPEND zero CONCATENATED WITH v CONCATENATED WITH zero TO ans
                END IF
            END FOR
            RETURN ans
        END FUNCTION
        RETURN CALL dfs WITH parameter n
    END FUNCTION
END CLASS
