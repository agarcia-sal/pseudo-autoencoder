        SET n TO the LENGTH OF num
        SET half TO the SUBSTRING FROM position zero TO the INTEGER DIVISION OF n BY two OF num

        CALL listFromString WITH half AND ASSIGN TO half_list
        SET k TO negative one
        SET l TO negative one
        FOR index FROM the LENGTH OF half_list MINUS two DOWN TO zero
            IF element at position index of half_list LESS THAN element at position index PLUS one of half_list
                SET k TO index
                BREAK
            END IF
        END FOR
        IF k EQUALS negative one
            RETURN the EMPTY string
        END IF

        FOR index FROM the LENGTH OF half_list MINUS one DOWN TO k PLUS one
            IF element at position index of half_list GREATER THAN element at position k of half_list
                SET l TO index
                BREAK
            END IF
        END FOR

        CALL sortSublistFromPosition WITH half_list AND k PLUS one AND ASSIGN TO half_list

        IF n MODULO two EQUALS zero
            RETURN new_half CONCATENATED WITH the REVERSE OF new_half
        ELSE
            RETURN new_half CONCATENATED WITH element at position the INTEGER DIVISION OF n BY two of num CONCATENATED WITH the REVERSE OF new_half
        END IF
    END FUNCTION
