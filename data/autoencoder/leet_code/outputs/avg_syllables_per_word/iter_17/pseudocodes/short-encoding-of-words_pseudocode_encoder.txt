CLASS Solution
    FUNCTION minimumLengthEncoding(list_of_words)
        SET reversed_words TO CALL helper ReverseAndSortWords WITH list_of_words
        SET total_length TO zero
        FOR index FROM zero TO the LENGTH OF reversed_words MINUS one
            IF index PLUS one EQUALS the LENGTH OF reversed_words OR reversed_words AT position index PLUS one DOES NOT START WITH reversed_words AT position index
                INCREMENT total_length BY the LENGTH OF reversed_words AT position index PLUS one
            END IF
        END FOR
        RETURN total_length
    END FUNCTION

    FUNCTION ReverseAndSortWords(list_of_words)
        RETURN sorted list obtained by reversing each word IN list_of_words
    END FUNCTION
END CLASS
