CLASS Solution

        SET total_xor_value TO zero
        FOR each number IN list_of_integers nums
            SET total_xor_value TO total_xor_value EXCLUSIVE OR number
        END FOR
        IF total_xor_value EQUALS zero
            RETURN true
        END IF
        IF the LENGTH OF list_of_integers nums MODULO two EQUALS zero
            RETURN true
        END IF
        RETURN false
    END FUNCTION
