CLASS Solution
    FUNCTION sortColors(list_of_numbers)
        SET low TO zero
        SET mid TO zero
        SET high TO the LENGTH OF list_of_numbers MINUS one

        WHILE mid IS LESS THAN OR EQUAL TO high
            IF the element at position mid of list_of_numbers EQUALS zero
                SET temporary TO the element at position mid of list_of_numbers
                SET element at position mid of list_of_numbers TO the element at position low of list_of_numbers
                SET element at position low of list_of_numbers TO temporary
                INCREMENT low BY one
                INCREMENT mid BY one
            ELSE IF the element at position mid of list_of_numbers EQUALS one
                INCREMENT mid BY one
            ELSE 
                SET temporary TO the element at position mid of list_of_numbers
                SET element at position mid of list_of_numbers TO the element at position high of list_of_numbers
                SET element at position high of list_of_numbers TO temporary
                DECREMENT high BY one
            END IF
        END WHILE
    END FUNCTION
END CLASS
