CLASS Solution
    FUNCTION singleNonDuplicate(list_of_numbers)
        SET left TO zero
        SET right TO the LENGTH OF list_of_numbers MINUS one

        WHILE left LESS THAN right holds
            SET mid TO left PLUS the INTEGER DIVISION OF right MINUS left BY two

            IF the element at position mid MODULO two EQUALS one
                DECREMENT mid BY one
            END IF

            IF the element at position mid OF list_of_numbers EQUALS the element at position mid PLUS one OF list_of_numbers
                SET left TO mid PLUS two
            ELSE
                SET right TO mid
            END IF
        END WHILE

        RETURN the element at position left OF list_of_numbers
    END FUNCTION
END CLASS
