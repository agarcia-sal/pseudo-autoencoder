CLASS Solution
    FUNCTION maximumNumberOfOnes(width, height, sideLength, maxOnes)
        SET frequency_list TO a list of zero repeated sideLength MULTIPLIED BY sideLength times
        FOR index_i FROM zero TO width MINUS one
            FOR index_j FROM zero TO height MINUS one
                SET position TO the remainder of index_i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS the remainder of index_j DIVIDED BY sideLength
                INCREMENT element at position of frequency_list BY one
            END FOR
        END FOR
        CALL sort WITH frequency_list TO sort descending
        SET result TO the SUM OF the first maxOnes elements of frequency_list
        RETURN result
    END FUNCTION
END CLASS
