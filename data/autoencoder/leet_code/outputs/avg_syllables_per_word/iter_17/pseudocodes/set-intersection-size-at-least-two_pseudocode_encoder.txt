CLASS Solution
    FUNCTION intersectionSizeTwo(intervals)
        CALL intervals SORTED BY the end element of each interval AND IF ends ARE EQUAL BY the negative of the start element of each interval
        SET first TO the element at position one LESS THAN the end element of the element at position zero of intervals
        SET second TO the end element of the element at position zero of intervals
        SET count TO two
        FOR each pair of start and end FROM the elements at positions one TO the end of intervals
            IF start IS GREATER THAN second
                SET first TO end MINUS one
                SET second TO end
                INCREMENT count BY two
            ELSE IF start IS GREATER THAN first
                SET first TO second
                SET second TO end
                INCREMENT count BY one
            END IF
        END FOR
        RETURN count
    END FUNCTION
END CLASS
