CLASS Solution
    FUNCTION isOneEditDistance(string_s, string_t)
        IF the LENGTH OF string_s IS GREATER THAN the LENGTH OF string_t
            RETURN CALL isOneEditDistance WITH parameters string_t AND string_s
        END IF

        SET difference_in_length TO the LENGTH OF string_t MINUS the LENGTH OF string_s

        IF difference_in_length IS GREATER THAN one
            RETURN false
        END IF
        IF difference_in_length EQUALS zero
            RETURN the COUNT OF positions WHERE element at position index of string_s IS NOT EQUAL TO element at position index of string_t IS EQUAL TO one
        ELSE
            FOR index FROM zero TO the LENGTH OF string_s MINUS one
                IF element at position index of string_s IS NOT EQUAL TO element at position index of string_t
                    RETURN substring from position index TO the END OF string_s IS EQUAL TO substring from position index plus one TO the END OF string_t
                END IF
            END FOR
            RETURN true
        END IF
    END FUNCTION
END CLASS
