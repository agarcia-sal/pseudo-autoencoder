CLASS Solution
    FUNCTION lengthOfLIS(list_of_numbers)
        IF list_of_numbers IS empty
            RETURN zero
        END IF

        SET dp TO empty list

        FOR each number IN list_of_numbers
            SET insertion_index TO CALL bisect_left WITH dp and number

            IF insertion_index EQUALS the LENGTH OF dp
                APPEND number TO dp
            ELSE
                SET element at position insertion_index of dp TO number
            END IF
        END FOR

        RETURN the LENGTH OF dp
    END FUNCTION
END CLASS
