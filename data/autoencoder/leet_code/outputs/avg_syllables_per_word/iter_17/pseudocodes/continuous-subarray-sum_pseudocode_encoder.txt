CLASS Solution
    FUNCTION checkSubarraySum(list_of_numbers nums, integer k)
        SET mapping_of_remainder_to_index TO a DICTIONARY WITH element zero MAPPED TO negative one
        SET cumulative_sum TO zero
        FOR each index i AND each number num IN list_of_numbers nums WITH enumeration
            INCREMENT cumulative_sum BY num
            IF k IS NOT EQUALS zero
                SET cumulative_sum TO cumulative_sum MODULO k
            END IF
            IF cumulative_sum IS IN mapping_of_remainder_to_index
                IF index i MINUS element at position cumulative_sum OF mapping_of_remainder_to_index IS GREATER THAN one
                    RETURN true
                END IF
            ELSE
                SET element at position cumulative_sum OF mapping_of_remainder_to_index TO index i
            END IF
        END FOR
        RETURN false
    END FUNCTION
END CLASS
