CLASS Solution
    FUNCTION maximumScore(nums, multipliers)
        SET length_multipliers TO the LENGTH OF multipliers
        SET length_nums TO the LENGTH OF nums

        CALL initialize_list_of_zeros WITH parameter length_multipliers PLUS one ASSIGN TO dp

        FOR index_i FROM length_multipliers MINUS one DOWN TO zero
            CALL initialize_list_of_zeros WITH parameter length_multipliers PLUS one ASSIGN TO new_dp
            FOR left_index FROM zero TO index_i
                SET right_index TO length_nums MINUS one MINUS index_i PLUS left_index
                SET multiplier_value TO element at position index_i of multipliers
                SET left_option TO multiplier_value MULTIPLIED BY element at position left_index of nums PLUS element at position left_index PLUS one of dp
                SET right_option TO multiplier_value MULTIPLIED BY element at position right_index of nums PLUS element at position left_index of dp
                SET element at position left_index of new_dp TO the GREATER OF left_option AND right_option
            END FOR
            SET dp TO new_dp
        END FOR

        RETURN element at position zero of dp
    END FUNCTION
END CLASS
