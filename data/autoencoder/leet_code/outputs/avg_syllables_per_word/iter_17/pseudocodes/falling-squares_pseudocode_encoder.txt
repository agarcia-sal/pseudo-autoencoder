CLASS Solution
    FUNCTION fallingSquares(positions)
        SET max_heights TO empty list
        SET intervals TO empty list
        SET current_max_height TO zero
        FOR each left_and_side_length IN positions
            ASSIGN left TO element at position zero of left_and_side_length
            ASSIGN side_length TO element at position one of left_and_side_length
            SET right TO left PLUS side_length MINUS one
            SET height TO side_length
            FOR each interval IN intervals
                ASSIGN interval_left TO element at position zero of interval
                ASSIGN interval_right TO element at position one of interval
                ASSIGN interval_height TO element at position two of interval
                IF left LESS THAN OR EQUAL TO interval_right AND right GREATER THAN OR EQUAL TO interval_left
                    SET height TO the GREATER OF height AND side_length PLUS interval_height
                END IF
            END FOR
            APPEND tuple of left right height TO intervals
            SET current_max_height TO the GREATER OF current_max_height AND height
            APPEND current_max_height TO max_heights
        END FOR
        RETURN max_heights
    END FUNCTION
END CLASS
