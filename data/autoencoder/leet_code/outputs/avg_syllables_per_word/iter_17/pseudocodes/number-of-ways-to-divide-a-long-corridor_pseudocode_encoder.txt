CLASS Solution
    FUNCTION numberOfWays(corridor)
        SET MOD TO ten TO THE POWER OF nine PLUS seven
        SET seats TO auxiliary FUNCTION find_positions_of_seat_characters WITH corridor
        
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two
            RETURN zero
        END IF
        
        SET ways TO one
        FOR index FROM two TO the LENGTH OF seats MINUS one STEP two
            SET difference TO element at position index of seats MINUS element at position index MINUS one of seats
            SET ways TO ways MULTIPLIED BY difference MODULO MOD
        END FOR
        
        RETURN ways
    END FUNCTION

    FUNCTION find_positions_of_seat_characters(corridor)
        SET positions TO empty list
        FOR index FROM zero TO the LENGTH OF corridor MINUS one
            IF element at position index of corridor EQUALS character S
                APPEND index TO positions
            END IF
        END FOR
        RETURN positions
    END FUNCTION
END CLASS
