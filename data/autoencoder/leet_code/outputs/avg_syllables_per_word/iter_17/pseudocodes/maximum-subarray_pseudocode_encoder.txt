CLASS Solution
    FUNCTION maxSubArray(list_of_numbers)
        SET maximum_sum TO element at position one of list_of_numbers
        SET current_sum TO element at position one of list_of_numbers
        FOR each number IN the substring from position two TO the end of list_of_numbers
            SET current_sum TO the GREATER OF number OR current_sum PLUS number
            SET maximum_sum TO the GREATER OF maximum_sum OR current_sum
        END FOR
        RETURN maximum_sum
    END FUNCTION
END CLASS
