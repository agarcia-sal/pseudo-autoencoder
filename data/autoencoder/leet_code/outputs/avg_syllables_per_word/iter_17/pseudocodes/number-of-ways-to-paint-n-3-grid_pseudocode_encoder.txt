CLASS Solution
    FUNCTION numOfWays(n)
        SET MOD_TO one billion PLUS seven
        SET a TO six
        SET b TO six
        FOR index FROM two TO n
            SET new_a TO the SUM OF three MULTIPLIED BY a AND two MULTIPLIED BY b MODULO MOD
            SET new_b TO the SUM OF two MULTIPLIED BY a AND two MULTIPLIED BY b MODULO MOD
            SET a TO new_a
            SET b TO new_b
        END FOR
        RETURN the SUM OF a AND b MODULO MOD
    END FUNCTION
END CLASS
