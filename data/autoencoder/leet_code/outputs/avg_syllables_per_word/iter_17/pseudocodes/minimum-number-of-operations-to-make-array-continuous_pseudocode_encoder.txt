CLASS Solution
    FUNCTION minOperations(list_of_numbers)
        SET n TO the LENGTH OF list_of_numbers
        SET unique_numbers TO the SORTED COLLECTION OF UNIQUE ELEMENTS IN list_of_numbers
        SET minimum_operations TO n
        SET pointer_j TO zero
        FOR index_i FROM zero TO the LENGTH OF unique_numbers MINUS one
            WHILE pointer_j LESS THAN the LENGTH OF unique_numbers AND element at position pointer_j of unique_numbers LESS THAN element at position index_i of unique_numbers PLUS n
                INCREMENT pointer_j BY one
            END WHILE
            SET minimum_operations TO the MINIMUM OF minimum_operations AND n MINUS pointer_j MINUS index_i
        END FOR
        RETURN minimum_operations
    END FUNCTION
END CLASS
