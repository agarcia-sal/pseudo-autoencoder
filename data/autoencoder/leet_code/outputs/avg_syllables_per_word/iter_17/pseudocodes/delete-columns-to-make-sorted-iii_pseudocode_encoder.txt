CLASS Solution
    FUNCTION minDeletionSize(list_of_strings)
        SET number_of_rows TO the LENGTH OF list_of_strings
        SET number_of_columns TO the LENGTH OF element at position zero of list_of_strings
        SET dynamic_programming_list TO the RESULT OF GenerateListWithMultipleElementsOfValue number_of_columns WITH element_value one
        FOR index i FROM one TO number_of_columns MINUS one
            FOR index j FROM zero TO i MINUS one
                IF CallFunction AllCharactersNonDecreasing WITH parameters list_of_strings number_of_rows index_j i HOLDS
                    SET element at position i OF dynamic_programming_list TO the MAXIMUM OF element at position i OF dynamic_programming_list AND element at position j OF dynamic_programming_list PLUS one
                END IF
            END FOR
        END FOR
        SET longest_non_decreasing_subsequence_length TO the MAXIMUM OF dynamic_programming_list
        SET result TO number_of_columns MINUS longest_non_decreasing_subsequence_length
        RETURN result
    END FUNCTION

FUNCTION GenerateListWithMultipleElementsOfValue(length_value WITH element_value)
    ABSTRACT AUXILIARY FUNCTION
END FUNCTION

FUNCTION CallFunction AllCharactersNonDecreasing(list_of_strings number_of_rows index_j index_i)
    FOR index k FROM zero TO number_of_rows MINUS one
        IF the character at position index_j OF element at position k OF list_of_strings IS GREATER THAN the character at position index_i OF element at position k OF list_of_strings
            RETURN false
        END IF
    END FOR
    RETURN true
END FUNCTION
