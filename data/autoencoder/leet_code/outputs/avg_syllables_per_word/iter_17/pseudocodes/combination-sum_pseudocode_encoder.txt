CLASS Solution
    FUNCTION combinationSum(list_of_candidates, target_value)
        FUNCTION backtrack(start_index, current_target, current_path)
            IF current_target EQUALS zero
                APPEND current_path TO result_collection
                RETURN
            END IF
            IF current_target LESS THAN zero
                RETURN
            END IF
            FOR index FROM start_index TO the LENGTH OF list_of_candidates MINUS one
                CALL backtrack WITH parameters index AND current_target MINUS element at position index of list_of_candidates AND current_path PLUS list containing element at position index of list_of_candidates
            END FOR
        END FUNCTION
        SET result_collection TO empty list
        CALL backtrack WITH parameters zero AND target_value AND empty list
        RETURN result_collection
    END FUNCTION
END CLASS
