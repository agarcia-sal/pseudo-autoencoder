CLASS Solution
    FUNCTION bestRotation(list_of_numbers)
        SET n TO the LENGTH OF list_of_numbers
        SET delta TO a list of zeroes OF LENGTH n

        FOR each index i AND each number num IN list_of_numbers WITH enumeration
            IF number num IS less than OR EQUAL TO index i
                INCREMENT element at position zero OF delta BY one
                IF index i MINUS number num PLUS one IS less than n
                    DECREMENT element at position index i MINUS number num PLUS one OF delta BY one
                END IF
            END IF

            IF index i PLUS one IS less than n
                INCREMENT element at position index i PLUS one OF delta BY one
                IF index i PLUS n MINUS number num PLUS one IS less than n
                    DECREMENT element at position index i PLUS n MINUS number num PLUS one OF delta BY one
                END IF
            END IF
        END FOR

        SET score TO zero
        SET max_score TO zero
        SET best_k TO zero

        FOR each number k FROM zero TO n MINUS one
            INCREMENT score BY element at position k OF delta
            IF score IS greater than max_score
                SET max_score TO score
                SET best_k TO k
            END IF
        END FOR

        RETURN best_k
    END FUNCTION
END CLASS
