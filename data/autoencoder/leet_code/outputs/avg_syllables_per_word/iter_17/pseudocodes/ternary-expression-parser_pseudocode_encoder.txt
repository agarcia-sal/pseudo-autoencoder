CLASS Solution
    FUNCTION parseTernary(expression)
        FUNCTION evaluate(index)
            IF index EQUALS the LENGTH OF expression MINUS one OR the element at position index PLUS one of expression EQUALS colon
                RETURN the element at position index of expression AND index PLUS two
            END IF

            CALL evaluate WITH index PLUS two AND ASSIGN RESULTS TO true_part AND next_index
            CALL evaluate WITH next_index AND ASSIGN RESULTS TO false_part AND final_index

            IF the element at position index of expression EQUALS letter T
                RETURN true_part AND final_index
            ELSE
                RETURN false_part AND final_index
            END IF
        END FUNCTION

        CALL evaluate WITH zero AND ASSIGN RESULTS TO result AND ignore_value
        RETURN result
    END FUNCTION
END CLASS
