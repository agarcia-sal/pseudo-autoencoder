CLASS Solution
    FUNCTION flipLights(n, presses)
        SET n TO the MINIMUM OF n AND six
        IF presses EQUALS zero
            RETURN one
        END IF
        IF n EQUALS one
            RETURN two
        END IF
        IF n EQUALS two
            IF presses EQUALS one
                RETURN three
            ELSE
                RETURN four
            END IF
        END IF
        IF presses EQUALS one
            RETURN four
        END IF
        IF presses EQUALS two
            RETURN seven
        END IF
        RETURN eight
    END FUNCTION
END CLASS
