CLASS Solution
    FUNCTION findPaths(m, n, maxMove, startRow, startColumn)
        SET MODULO_CONSTANT TO ten TO THE POWER OF nine PLUS one
        
        SET directions TO a list containing tuples of zero and one, zero and negative one, one and zero, negative one and zero
        
        CALL initializeThreeDimensionalList WITH parameters m, n, maxMove PLUS one TO CREATE dp
         
        FOR step FROM one TO maxMove
            FOR row FROM zero TO m MINUS one
                FOR column FROM zero TO n MINUS one
                    FOR each directionDeltaRow, directionDeltaColumn IN directions
                        SET newRow TO row PLUS directionDeltaRow
                        SET newColumn TO column PLUS directionDeltaColumn
                        
                        IF newRow LESS THAN zero OR newRow GREATER THAN OR EQUAL TO m OR newColumn LESS THAN zero OR newColumn GREATER THAN OR EQUAL TO n
                            INCREMENT element at position step of list at position column of list at position row of dp BY one
                            SET element at position step of list at position column of list at position row of dp TO element at position step of list at position column of list at position row of dp MODULO MODULO_CONSTANT
                        ELSE
                            INCREMENT element at position step of list at position column of list at position row of dp BY element at position step MINUS one of list at position newColumn of list at position newRow of dp
                            SET element at position step of list at position column of list at position row of dp TO element at position step of list at position column of list at position row of dp MODULO MODULO_CONSTANT
                        END IF
                    END FOR
                END FOR
            END FOR
        END FOR
        
        RETURN element at position maxMove of list at position startColumn of list at position startRow of dp
    END FUNCTION
END CLASS
