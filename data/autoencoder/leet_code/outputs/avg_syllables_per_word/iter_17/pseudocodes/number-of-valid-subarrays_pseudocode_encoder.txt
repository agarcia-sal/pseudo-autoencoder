CLASS Solution
    FUNCTION validSubarrays(list_of_integers nums)
        SET list_of_integers stack TO an empty list
        SET integer count TO zero
        FOR each integer num IN nums
            WHILE the LENGTH OF stack IS GREATER THAN zero AND element at position the LENGTH OF stack MINUS one of stack IS GREATER THAN num
                REMOVE the LAST element FROM stack
            END WHILE
            APPEND num TO stack
            INCREMENT count BY the LENGTH OF stack
        END FOR
        RETURN count
    END FUNCTION
END CLASS
