CLASS Solution
    FUNCTION winnerSquareGame(n)
        SET dp TO A LIST OF false VALUES REPEATED n PLUS one TIMES
        FOR i FROM one TO n
            SET j TO one
            WHILE j MULTIPLIED BY j IS LESS THAN OR EQUAL TO i
                IF NOT THE ELEMENT AT POSITION i MINUS j MULTIPLIED BY j OF dp
                    SET THE ELEMENT AT POSITION i OF dp TO true
                    BREAK OUT OF THE WHILE LOOP
                END IF
                INCREMENT j BY one
            END WHILE
        END FOR
        RETURN THE ELEMENT AT POSITION n OF dp
    END FUNCTION
END CLASS
