CLASS Solution
    FUNCTION maxRotateFunction(list_of_numbers)
        SET length_n TO the LENGTH OF list_of_numbers
        IF length_n EQUALS one
            RETURN zero
        END IF
        SET total_sum TO CALL sum OF list_of_numbers
        SET current_sum TO CALL sum OF the COLLECTION OF index MULTIPLIED BY element FOR each index AND element IN list_of_numbers
        SET max_value TO current_sum
        FOR k FROM one TO length_n MINUS one
            SET current_sum TO current_sum PLUS total_sum MINUS length_n MULTIPLIED BY element at position length_n MINUS k OF list_of_numbers
            SET max_value TO the GREATER OF max_value AND current_sum
        END FOR
        RETURN max_value
    END FUNCTION
END CLASS
