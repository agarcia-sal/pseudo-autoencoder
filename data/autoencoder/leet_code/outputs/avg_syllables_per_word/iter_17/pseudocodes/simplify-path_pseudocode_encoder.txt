CLASS Solution
    FUNCTION simplifyPath(path)
        SET components TO CALL split_path WITH path
        SET stack TO CALL initialize_empty_list
        FOR each component IN components
            IF component EQUALS empty_string OR component EQUALS single_dot
                CONTINUE
            ELSE IF component EQUALS double_dot
                IF the LENGTH OF stack GREATER THAN zero
                    CALL remove_last_element FROM stack
                END IF
            ELSE
                CALL append_component TO stack
            END IF
        END FOR
        SET simplified_path TO CALL join_stack_with_slashes WITH stack
        RETURN simplified_path
    END FUNCTION
