CLASS Solution
    FUNCTION appealSum(string s)
        SET last_seen_positions TO the result of CALL initialize_last_positions WITH no parameters
        SET total_appeal TO zero
        SET current_appeal TO zero
        FOR each position i FROM zero TO the LENGTH OF string s MINUS one
            SET current_character TO element at position i of string s
            SET character_index TO the result of CALL map_char_to_index WITH parameter current_character
            SET difference TO i MINUS element at position character_index of last_seen_positions
            INCREMENT current_appeal BY difference
            INCREMENT total_appeal BY current_appeal
            SET element at position character_index of last_seen_positions TO i
        END FOR
        RETURN total_appeal
    END FUNCTION

    FUNCTION initialize_last_positions()
        RETURN list containing twenty six elements each SET TO negative one
    END FUNCTION

    FUNCTION map_char_to_index(character)
        RETURN the Unicode code point of character MINUS the Unicode code point of lowercase letter a
    END FUNCTION
END CLASS
