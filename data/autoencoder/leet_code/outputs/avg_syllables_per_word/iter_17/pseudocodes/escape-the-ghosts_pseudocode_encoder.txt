CLASS Solution
    FUNCTION escapeGhosts(list_of_ghosts AS list of list_of_integers, target_position AS list_of_integers) RETURNS boolean
        SET player_distance TO the ABSOLUTE VALUE OF element at position zero of target_position PLUS the ABSOLUTE VALUE OF element at position one of target_position
        FOR each ghost_position IN list_of_ghosts
            SET ghost_distance TO the ABSOLUTE VALUE OF element at position zero of ghost_position MINUS element at position zero of target_position PLUS the ABSOLUTE VALUE OF element at position one of ghost_position MINUS element at position one of target_position
            IF ghost_distance LESS THAN OR EQUAL TO player_distance
                RETURN false
            END IF
        END FOR
        RETURN true
    END FUNCTION
END CLASS
