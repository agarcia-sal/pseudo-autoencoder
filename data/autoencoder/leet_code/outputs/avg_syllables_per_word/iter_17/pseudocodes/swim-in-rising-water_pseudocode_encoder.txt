CLASS Solution
    FUNCTION swimInWater(grid)
        SET n TO the LENGTH OF grid
        SET visited TO an EMPTY set
        SET min_heap TO a LIST CONTAINING the TUPLE OF element at position zero of position at position zero of grid, zero, zero
        ADD the TUPLE of zero, zero TO visited

        SET directions TO a LIST OF the TUPLES zero, one; one, zero; zero, negative one; negative one, zero

        WHILE min_heap is NOT EMPTY
            CALL heapq.heappop WITH min_heap AND ASSIGN TO t, i, j
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one
                RETURN t
            END IF

            FOR each di, dj IN directions
                SET ni TO i PLUS di
                SET nj TO j PLUS dj
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND the TUPLE of ni, nj IS NOT IN visited
                    ADD the TUPLE of ni, nj TO visited
                    CALL heapq.heappush WITH min_heap AND the TUPLE OF the GREATER OF t AND element at position ni of position nj of grid, ni, nj
                END IF
            END FOR
        END WHILE

        RETURN negative one
    END FUNCTION
END CLASS
