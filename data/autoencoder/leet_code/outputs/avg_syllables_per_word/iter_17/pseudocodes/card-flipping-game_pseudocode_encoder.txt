CLASS Solution
    FUNCTION flipgame(fronts, backs)
        SET same TO CALL identify_same_numbers WITH fronts AND backs
        SET all_numbers TO CALL combine_all_numbers WITH fronts AND backs
        SET good_numbers TO CALL remove_numbers_in_set WITH all_numbers AND same
        IF good_numbers IS NOT EMPTY
            SET result TO CALL find_minimum_number IN good_numbers
            RETURN result
        ELSE
            RETURN zero
        END IF
    END FUNCTION
END CLASS

FUNCTION identify_same_numbers(fronts, backs)
    CREATE empty set same_numbers
    FOR index FROM zero TO the LENGTH OF fronts MINUS one
        IF element at position index of fronts EQUALS element at position index of backs
            ADD element at position index of fronts TO same_numbers
        END IF
    END FOR
    RETURN same_numbers
END FUNCTION

FUNCTION combine_all_numbers(fronts, backs)
    CREATE empty set combined_numbers
    FOR each number IN fronts
        ADD number TO combined_numbers
    END FOR
    FOR each number IN backs
        ADD number TO combined_numbers
    END FOR
    RETURN combined_numbers
END FUNCTION

FUNCTION remove_numbers_in_set(all_numbers, remove_numbers)
    CREATE empty set filtered_numbers
    FOR each number IN all_numbers
        IF number IS NOT IN remove_numbers
            ADD number TO filtered_numbers
        END IF
    END FOR
    RETURN filtered_numbers
END FUNCTION

FUNCTION find_minimum_number(numbers)
    SET minimum_number TO none
    FOR each number IN numbers
        IF minimum_number IS none OR number LESS THAN minimum_number
            SET minimum_number TO number
        END IF
    END FOR
    RETURN minimum_number
END FUNCTION
