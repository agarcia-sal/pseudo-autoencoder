CLASS UnionFind
    FUNCTION __init__(self, n)
        SET self.parent TO the LIST of integers FROM zero TO n MINUS one
        SET self.rank TO the LIST containing one repeated n times
    END FUNCTION

    FUNCTION find(self, u)
        IF element at position u of self.parent IS NOT EQUALS TO u
            SET element at position u of self.parent TO CALL find WITH parameter element at position u of self.parent
        END IF
        RETURN element at position u of self.parent
    END FUNCTION

    FUNCTION union(self, u, v)
        SET rootU TO CALL find WITH parameter u
        SET rootV TO CALL find WITH parameter v
        
        IF rootU IS NOT EQUALS TO rootV
            IF element at position rootU of self.rank IS GREATER THAN element at position rootV of self.rank
                SET element at position rootV of self.parent TO rootU
            ELSE IF element at position rootU of self.rank IS LESS THAN element at position rootV of self.rank
                SET element at position rootU of self.parent TO rootV
            ELSE
                SET element at position rootV of self.parent TO rootU
                INCREMENT element at position rootU of self.rank BY one
            END IF
            RETURN true
        END IF
        RETURN false
    END FUNCTION
END CLASS

CLASS Solution
    FUNCTION findCriticalAndPseudoCriticalEdges(self, n, edges)
        FOR each index i AND element edge IN edges WITH enumeration
            CALL APPEND operation TO add i TO edge
        END FOR
        
        CALL sort ON edges WITH key AS the element at position two of each edge
        
        FUNCTION mst(exclude IS none, include IS none)
            SET uf TO CALL UnionFind WITH parameter n
            SET weight TO zero
            
            IF include IS NOT EQUALS TO none
                CALL uf.union WITH parameters element at position zero of include AND element at position one of include
                INCREMENT weight BY element at position two of include
            END IF
            
            FOR each u, v, w, idx IN edges
                IF idx IS EQUALS TO exclude
                    CONTINUE to next iteration
                END IF
                IF CALL uf.union WITH parameters u AND v RETURNS true
                    INCREMENT weight BY w
                END IF
            END FOR
            
            SET root TO CALL uf.find WITH parameter zero
            FOR index i FROM zero TO n MINUS one
                IF CALL uf.find WITH parameter i IS NOT EQUALS TO root
                    RETURN infinite
                END IF
            END FOR
            
            RETURN weight
        END FUNCTION
        
        SET mst_weight TO CALL mst WITHOUT parameters
        
        SET critical TO empty list
        SET pseudo_critical TO empty list
        
        FOR each u, v, w, idx IN edges
            IF CALL mst WITH parameter exclude AS idx IS GREATER THAN mst_weight
                CALL APPEND idx TO critical
            ELSE IF CALL mst WITH parameter include AS tuple of u, v, w IS EQUALS TO mst_weight
                CALL APPEND idx TO pseudo_critical
            END IF
        END FOR
        
        RETURN list containing critical AND pseudo_critical
    END FUNCTION
END CLASS
