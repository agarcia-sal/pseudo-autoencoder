CLASS Solution
    FUNCTION slidingPuzzle(board)
        SET start TO tuple of tuple of each row IN board
        SET target TO tuple of tuple containing one two three and four five zero
        SET neighbors TO dictionary with keys zero to five and set values of their neighbor positions
        FOR i FROM zero TO one
            FOR j FROM zero TO two
                IF element at position j of element at position i of start EQUALS zero
                    SET zero_pos TO i MULTIPLIED BY three PLUS j
                END IF
            END FOR
        END FOR
        SET queue TO deque containing tuple of start zero_pos and zero
        SET visited TO set containing start
        WHILE queue holds
            SET state zero_pos moves TO dequeued element from queue
            IF state EQUALS target
                RETURN moves
            END IF
            FOR each neighbor IN element at position zero_pos of neighbors
                SET new_state TO list of list converted from state
                SET i TO zero_pos INTEGER DIVIDED BY three
                SET j TO zero_pos MODULO three
                SET ni TO neighbor INTEGER DIVIDED BY three
                SET nj TO neighbor MODULO three
                SET element at position j of element at position i of new_state TO element at position nj of element at position ni of new_state
                SET element at position nj of element at position ni of new_state TO zero
                SET new_tuple_state TO tuple of tuple of each row IN new_state
                IF new_tuple_state IS NOT IN visited
                    ADD new_tuple_state TO visited
                    APPEND tuple of new_tuple_state neighbor moves PLUS one TO queue
                END IF
            END FOR
        END WHILE
        RETURN negative one
    END FUNCTION
END CLASS
