CLASS Solution
    FUNCTION minimumMoves(grid)
        SET n TO the LENGTH OF grid
        ASSIGN queue TO a new double ended queue containing a single tuple of zero zero zero one zero 
        ASSIGN visited TO a new set containing a single tuple of zero zero zero one 

        WHILE queue is not empty
            CALL queue TO REMOVE AND RETURN the element at the FRONT and ASSIGN TO r1 c1 r2 c2 steps

            IF r1 EQUALS n MINUS one AND c1 EQUALS n MINUS two AND r2 EQUALS n MINUS one AND c2 EQUALS n MINUS one THEN
                RETURN steps
            END IF

            IF c2 PLUS one LESS THAN n AND element at position r1 of grid at position c1 PLUS one EQUALS zero AND element at position r2 of grid at position c2 PLUS one EQUALS zero AND tuple of r1 c1 PLUS one r2 c2 PLUS one IS NOT IN visited THEN
                ADD tuple of r1 c1 PLUS one r2 c2 PLUS one TO visited
                APPEND tuple of r1 c1 PLUS one r2 c2 PLUS one steps PLUS one TO queue
            END IF

            IF r2 PLUS one LESS THAN n AND element at position r1 PLUS one of grid at position c1 EQUALS zero AND element at position r2 PLUS one of grid at position c2 EQUALS zero AND tuple of r1 PLUS one c1 r2 PLUS one c2 IS NOT IN visited THEN
                ADD tuple of r1 PLUS one c1 r2 PLUS one c2 TO visited
                APPEND tuple of r1 PLUS one c1 r2 PLUS one c2 steps PLUS one TO queue
            END IF

            IF r1 EQUALS r2 AND r1 PLUS one LESS THAN n AND element at position r1 PLUS one of grid at position c1 EQUALS zero AND element at position r1 PLUS one of grid at position c2 EQUALS zero AND tuple of r1 c1 r1 PLUS one c1 IS NOT IN visited THEN
                ADD tuple of r1 c1 r1 PLUS one c1 TO visited
                APPEND tuple of r1 c1 r1 PLUS one c1 steps PLUS one TO queue
            END IF

            IF c1 EQUALS c2 AND c1 PLUS one LESS THAN n AND element at position r1 of grid at position c1 PLUS one EQUALS zero AND element at position r2 of grid at position c2 PLUS one EQUALS zero AND tuple of r1 c1 r1 c1 PLUS one IS NOT IN visited THEN
                ADD tuple of r1 c1 r1 c1 PLUS one TO visited
                APPEND tuple of r1 c1 r1 c1 PLUS one steps PLUS one TO queue
            END IF
        END WHILE

        RETURN minus one
    END FUNCTION
END CLASS
