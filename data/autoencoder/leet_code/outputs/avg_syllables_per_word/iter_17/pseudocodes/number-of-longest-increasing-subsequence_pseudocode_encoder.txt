    FUNCTION findNumberOfLIS(list_of_numbers)
        IF the list_of_numbers IS empty
            RETURN zero
        END IF
        SET number_of_elements TO the LENGTH OF list_of_numbers
        CALL initialize_list_with_value WITH parameters one AND number_of_elements RETURNING lengths
        CALL initialize_list_with_value WITH parameters one AND number_of_elements RETURNING counts
        FOR index_i FROM zero TO number_of_elements MINUS one
            FOR index_j FROM zero TO index_i MINUS one
                IF element at position index_i of list_of_numbers GREATER THAN element at position index_j of list_of_numbers
                    IF element at position index_j of lengths PLUS one GREATER THAN element at position index_i of lengths
                        SET element at position index_i of lengths TO element at position index_j of lengths PLUS one
                        SET element at position index_i of counts TO element at position index_j of counts
                    ELSE IF element at position index_j of lengths PLUS one EQUALS element at position index_i of lengths
                        INCREMENT element at position index_i of counts BY element at position index_j of counts
                    END IF
                END IF
            END FOR
        END FOR
        SET longest_length TO the MAXIMUM value in lengths
        SET total_count TO zero
        FOR index FROM zero TO number_of_elements MINUS one
            IF element at position index of lengths EQUALS longest_length
                INCREMENT total_count BY element at position index of counts
            END IF
        END FOR
        RETURN total_count
    END FUNCTION
