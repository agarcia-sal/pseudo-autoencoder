CLASS TreeNode
    FUNCTION __init__(val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN None
    END IF
    SET root TO CALL TreeNode WITH element at position zero of values
    SET i TO one
    SET queue TO CALL deque WITH no parameters
    CALL APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO CALL popleft FROM queue
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.left TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.right TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUALS q.val
        RETURN False
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION maxSumBST(root)
        SET self.max_sum TO zero

        FUNCTION helper(node)
            IF node IS None
                RETURN True, zero, positive infinity, negative infinity
            END IF

            SET left_is_bst, left_sum, left_min, left_max TO CALL helper WITH node.left
            SET right_is_bst, right_sum, right_min, right_max TO CALL helper WITH node.right

            IF left_is_bst IS True AND right_is_bst IS True AND left_max LESS THAN node.val AND node.val LESS THAN right_min
                SET current_sum TO left_sum PLUS right_sum PLUS node.val
                SET current_min TO the MINIMUM OF node.val AND left_min
                SET current_max TO the MAXIMUM OF node.val AND right_max
                SET self.max_sum TO the MAXIMUM OF self.max_sum AND current_sum
                RETURN True, current_sum, current_min, current_max
            END IF

            RETURN False, zero, positive infinity, negative infinity
        END FUNCTION

        CALL helper WITH root
        RETURN self.max_sum
    END FUNCTION
END CLASS
