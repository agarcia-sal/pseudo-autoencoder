CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN None
    END IF
    SET root TO CALL TreeNode WITH element at position zero of values
    SET i TO one
    SET queue TO CALL deque WITH no parameters
    CALL APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO CALL popleft ON queue
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.left TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.right TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUALS q.val
        RETURN False
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION addOneRow(self, root, val, depth)
        IF depth EQUALS one
            RETURN CALL TreeNode WITH val, left EQUALS root
        END IF
        SET queue TO CALL deque WITH a list containing a tuple of root and one
        WHILE queue IS NOT empty
            SET node, level TO CALL popleft ON queue
            IF level EQUALS depth MINUS one
                SET node.left TO CALL TreeNode WITH val, left EQUALS node.left
                SET node.right TO CALL TreeNode WITH val, right EQUALS node.right
            ELSE IF level LESS THAN depth MINUS one
                IF node.left IS NOT None
                    CALL APPEND a tuple of node.left and level PLUS one TO queue
                END IF
                IF node.right IS NOT None
                    CALL APPEND a tuple of node.right and level PLUS one TO queue
                END IF
            END IF
        END WHILE
        RETURN root
    END FUNCTION
END CLASS
