CLASS Solution
    FUNCTION checkInclusion(s1, s2)
        SET length_of_s1 TO the LENGTH OF s1
        SET length_of_s2 TO the LENGTH OF s2
        IF length_of_s1 GREATER THAN length_of_s2
            RETURN False
        END IF
        SET count_s1 TO CALL Counter WITH s1
        SET count_s2 TO CALL Counter WITH the SUBSTRING from position zero TO position length_of_s1 MINUS one of s2
        IF count_s1 EQUALS count_s2
            RETURN True
        END IF
        FOR index FROM length_of_s1 TO length_of_s2 MINUS one
            INCREMENT the COUNT in count_s2 at the ELEMENT at position index of s2 BY one
            DECREMENT the COUNT in count_s2 at the ELEMENT at position index MINUS length_of_s1 of s2 BY one
            IF the COUNT in count_s2 at the ELEMENT at position index MINUS length_of_s1 of s2 EQUALS zero
                REMOVE the ELEMENT at position index MINUS length_of_s1 of s2 FROM count_s2
            END IF
            IF count_s1 EQUALS count_s2
                RETURN True
            END IF
        END FOR
        RETURN False
    END FUNCTION
END CLASS
