CLASS Solution
    FUNCTION minEatingSpeed(piles, h)
        FUNCTION canFinish(k)
            SET hours TO zero
            FOR each pile IN piles
                INCREMENT hours BY CALL ceil WITH parameter pile DIVIDED BY k
            END FOR
            RETURN hours LESS THAN OR EQUAL TO h
        END FUNCTION
        
        SET left TO one
        SET right TO CALL maximum WITH parameter piles
        
        WHILE left LESS THAN OR EQUAL TO right
            SET mid TO left PLUS right DIVIDED BY two
            IF CALL canFinish WITH parameter mid
                SET right TO mid MINUS one
            ELSE
                SET left TO mid PLUS one
            END IF
        END WHILE
        
        RETURN left
    END FUNCTION
END CLASS
