CLASS Solution
    FUNCTION canCross(list_of_stones)
        IF list_of_stones IS empty OR the LENGTH OF list_of_stones LESS THAN two
            RETURN False
        END IF

        SET stone_set TO AUXILIARY FUNCTION convert_list_to_set WITH list_of_stones
        SET memo TO AUXILIARY FUNCTION initialize_empty_dictionary

        FUNCTION can_jump_to(position, jump)
            IF position EQUALS element at position negative one of list_of_stones
                RETURN True
            END IF

            IF the TUPLE of position AND jump IS IN memo
                RETURN element at key position AND jump of memo
            END IF

            FOR next_jump FROM jump MINUS one TO jump PLUS one
                IF next_jump GREATER THAN zero AND position PLUS next_jump IS IN stone_set
                    IF CALL can_jump_to WITH position PLUS next_jump AND next_jump
                        SET element at key position AND jump of memo TO True
                        RETURN True
                    END IF
                END IF
            END FOR

            SET element at key position AND jump of memo TO False
            RETURN False
        END FUNCTION

        RETURN CALL can_jump_to WITH element at position zero of list_of_stones AND zero
    END FUNCTION
END CLASS
