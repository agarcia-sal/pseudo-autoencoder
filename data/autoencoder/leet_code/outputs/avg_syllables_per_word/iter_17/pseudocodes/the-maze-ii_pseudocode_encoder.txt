CLASS Solution
    FUNCTION shortestDistance(maze AS list_of_list_of_integers, start AS list_of_integers, destination AS list_of_integers) RETURNS integer
        SET m TO the LENGTH OF maze
        SET n TO the LENGTH OF element at position zero of maze
        SET directions TO the LIST OF tuples of one zero AND minus one zero AND zero one AND zero minus one
        SET start TO the CONVERSION OF start TO tuple
        SET destination TO the CONVERSION OF destination TO tuple
        SET visited TO an EMPTY set
        SET pq TO the LIST of tuple of zero AND start

        WHILE pq is NOT EMPTY
            CALL dist, position TO the EXTRACTION OF the SMALLEST element FROM pq USING heapq
            CALL x, y TO the UNPACKING OF position

            IF x EQUALS the first element OF destination AND y EQUALS the second element OF destination
                RETURN dist
            END IF

            IF position IS IN visited
                CONTINUE to the next iteration
            END IF

            ADD position TO visited

            FOR each dx, dy IN directions
                SET new_x TO x
                SET new_y TO y
                SET count TO zero

                WHILE new_x PLUS dx IS GREATER THAN OR EQUAL TO zero AND new_x PLUS dx IS LESS THAN m AND new_y PLUS dy IS GREATER THAN OR EQUAL TO zero AND new_y PLUS dy IS LESS THAN n AND element at position new_x PLUS dx AND new_y PLUS dy OF maze EQUALS zero
                    INCREMENT new_x BY dx
                    INCREMENT new_y BY dy
                    INCREMENT count BY one
                END WHILE

                IF tuple of new_x AND new_y IS NOT IN visited
                    CALL heapq PUSH pq WITH tuple of dist PLUS count AND tuple of new_x AND new_y
                END IF
            END FOR
        END WHILE

        RETURN minus one
    END FUNCTION
END CLASS
