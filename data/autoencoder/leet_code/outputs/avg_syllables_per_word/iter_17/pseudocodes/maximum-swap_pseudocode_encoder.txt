CLASS Solution
    FUNCTION maximumSwap(number)
        SET digits TO CALL ConvertNumberToListOfDigits WITH number
        SET last_occurrence TO CALL MapDigitToLastPosition WITH digits
        
        FOR each index AND digit IN EnumerateList WITH digits
            FOR digit_candidate FROM nine TO digit PLUS one DECREMENT BY one
                IF ValueOfKey digit_candidate IN last_occurrence GREATER THAN index
                    SWAP element at position index OF digits WITH element at position ValueOfKey digit_candidate IN last_occurrence OF digits
                    RETURN CALL ConvertListOfDigitsToNumber WITH digits
                END IF
            END FOR
        END FOR
        
        RETURN number
    END FUNCTION
    
    FUNCTION ConvertNumberToListOfDigits WITH number
        IMPLEMENTATION OMITTED
    END FUNCTION
    
    FUNCTION MapDigitToLastPosition WITH digits
        IMPLEMENTATION OMITTED
    END FUNCTION
    
    FUNCTION ConvertListOfDigitsToNumber WITH digits
        IMPLEMENTATION OMITTED
    END FUNCTION
END CLASS
