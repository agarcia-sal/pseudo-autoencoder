CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values)
    IF values IS empty
        RETURN None
    END IF
    SET root TO CALL TreeNode WITH element at position zero of values
    SET i TO one
    SET queue TO CALL deque WITH no parameters
    CALL APPEND root TO queue
    WHILE queue is not empty
        SET node TO CALL popleft ON queue
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.left TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT None
            SET node.right TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val NOT EQUALS q.val
        RETURN False
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION checkEqualTree(self, root)
        FUNCTION subtree_sum(node)
            IF node IS None
                RETURN zero
            END IF
            RETURN node.val PLUS CALL subtree_sum WITH node.left PLUS CALL subtree_sum WITH node.right
        END FUNCTION

        SET total_sum TO CALL subtree_sum WITH root

        IF total_sum MODULO two NOT EQUALS zero
            RETURN False
        END IF

        SET target TO total_sum DIVIDED BY two
        SET found TO list containing False

        FUNCTION can_split(node)
            IF node IS None
                RETURN zero
            END IF
            SET current_sum TO node.val PLUS CALL can_split WITH node.left PLUS CALL can_split WITH node.right
            IF current_sum EQUALS target AND node IS NOT root
                SET element at position zero of found TO True
            END IF
            RETURN current_sum
        END FUNCTION

        CALL can_split WITH root
        RETURN element at position zero of found
    END FUNCTION
END CLASS
