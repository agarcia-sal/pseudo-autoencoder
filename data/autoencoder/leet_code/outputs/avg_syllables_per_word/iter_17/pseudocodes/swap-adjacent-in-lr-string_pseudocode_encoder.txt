CLASS Solution
    FUNCTION canTransform(start STRING, end STRING) RETURNS BOOLEAN
        SET start_filtered TO CALL filter_non_x_characters WITH start
        SET end_filtered TO CALL filter_non_x_characters WITH end

        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered
            RETURN false
        END IF

        FOR each pair_of_characters_positions FROM the ZIP OF start_filtered AND end_filtered
            SET c1 TO the FIRST element AT position zero OF pair_of_characters_positions
            SET i1 TO the SECOND element AT position one OF pair_of_characters_positions
            SET c2 TO the FIRST element AT position zero OF the SECOND element OF pair_of_characters_positions
            SET i2 TO the SECOND element AT position one OF the SECOND element OF pair_of_characters_positions

            IF c1 NOT EQUALS c2
                RETURN false
            END IF

            IF c1 EQUALS letter L AND i1 LESS THAN i2
                RETURN false
            END IF

            IF c1 EQUALS letter R AND i1 GREATER THAN i2
                RETURN false
            END IF
        END FOR

        RETURN true
    END FUNCTION

    FUNCTION filter_non_x_characters(input_string STRING) RETURNS LIST OF TUPLES
        DEFINE AS AUXILIARY FUNCTION TO EXTRACT list OF (character AND index) WHERE character NOT EQUALS letter X FROM input_string
    END FUNCTION
END CLASS
