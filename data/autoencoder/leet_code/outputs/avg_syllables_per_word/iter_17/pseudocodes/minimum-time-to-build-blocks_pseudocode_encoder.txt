CLASS Solution
    FUNCTION minBuildTime(blocks, split)
        CALL convert_list_to_min_heap WITH blocks
        WHILE the LENGTH OF blocks IS GREATER THAN one
            SET first TO extract_smallest_element_from_heap WITH blocks
            SET second TO extract_smallest_element_from_heap WITH blocks
            SET combined_time TO split PLUS second
            CALL insert_element_into_heap WITH blocks AND combined_time
        END WHILE
        RETURN element at position zero of blocks
    END FUNCTION
END CLASS
