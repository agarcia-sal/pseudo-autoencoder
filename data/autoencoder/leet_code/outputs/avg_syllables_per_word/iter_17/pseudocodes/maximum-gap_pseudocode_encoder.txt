CLASS Solution
    FUNCTION maximumGap(nums)
        IF the LENGTH OF nums IS LESS THAN two
            RETURN zero
        END IF

        FUNCTION countingSort(arr, exp)
            SET n TO the LENGTH OF arr
            SET output TO a LIST OF zero REPEATED n times
            SET count TO a LIST OF zero REPEATED ten times

            FOR index FROM zero TO n MINUS one
                SET digit_index TO the ELEMENT AT POSITION index OF arr DIVIDED BY exp
                INCREMENT the ELEMENT AT POSITION digit_index MODULO ten OF count BY one
            END FOR

            FOR index FROM one TO nine
                INCREMENT the ELEMENT AT POSITION index OF count BY the ELEMENT AT POSITION index MINUS one OF count
            END FOR

            SET i TO n MINUS one
            WHILE i IS GREATER THAN OR EQUAL TO zero
                SET digit_index TO the ELEMENT AT POSITION i OF arr DIVIDED BY exp
                SET the ELEMENT AT POSITION the ELEMENT AT POSITION digit_index MODULO ten OF count MINUS one OF output TO the ELEMENT AT POSITION i OF arr
                DECREMENT the ELEMENT AT POSITION digit_index MODULO ten OF count BY one
                DECREMENT i BY one
            END WHILE

            FOR i FROM zero TO n MINUS one
                SET the ELEMENT AT POSITION i OF arr TO the ELEMENT AT POSITION i OF output
            END FOR
        END FUNCTION

        FUNCTION radixSort(arr)
            SET max1 TO the MAXIMUM ELEMENT OF arr
            SET exp TO one
            WHILE max1 DIVIDED BY exp IS GREATER THAN zero
                CALL countingSort WITH arr AND exp
                MULTIPLY exp BY ten AND ASSIGN TO exp
            END WHILE
        END FUNCTION

        CALL radixSort WITH nums

        SET max_diff TO zero
        FOR i FROM one TO the LENGTH OF nums MINUS one
            SET current_diff TO the ELEMENT AT POSITION i OF nums MINUS the ELEMENT AT POSITION i MINUS one OF nums
            IF current_diff IS GREATER THAN max_diff
                SET max_diff TO current_diff
            END IF
        END FOR

        RETURN max_diff
    END FUNCTION
END CLASS
