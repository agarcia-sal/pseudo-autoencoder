CLASS Solution
    FUNCTION longestConsecutive(list_of_numbers)
        IF the list_of_numbers IS EMPTY
            RETURN zero
        END IF

        SET set_of_numbers TO the SET of elements IN list_of_numbers
        SET longest_streak TO zero

        FOR each number IN set_of_numbers
            IF the number MINUS one IS NOT IN set_of_numbers
                SET current_number TO number
                SET current_streak TO one

                WHILE the current_number PLUS one IS IN set_of_numbers
                    INCREMENT current_number BY one
                    INCREMENT current_streak BY one
                END WHILE

                SET longest_streak TO the GREATER OF longest_streak AND current_streak
            END IF
        END FOR

        RETURN longest_streak
    END FUNCTION
END CLASS
