CLASS Solution
    FUNCTION findPairs(list_of_numbers, integer_k)
        IF integer_k LESS THAN zero
            RETURN zero
        END IF
        SET number_counts TO auxiliary FUNCTION InitializeEmptyMapping
        FOR each number IN list_of_numbers
            SET current_count TO auxiliary FUNCTION GetMappingValue WITH number_counts number AND zero
            SET updated_count TO current_count PLUS one
            ASSIGN updated_count TO number_counts AT number
        END FOR
        SET unique_pairs_count TO zero
        FOR each number IN number_counts
            IF integer_k EQUALS zero
                IF number_counts AT number GREATER THAN one
                    INCREMENT unique_pairs_count BY one
                END IF
            ELSE
                IF number PLUS integer_k IS IN number_counts
                    INCREMENT unique_pairs_count BY one
                END IF
            END IF
        END FOR
        RETURN unique_pairs_count
    END FUNCTION
END CLASS
