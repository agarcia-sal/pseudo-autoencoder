CLASS Solution
    FUNCTION numDecodings(string_input)
        IF string_input IS empty OR the element at position zero of string_input EQUALS zero
            RETURN zero
        END IF

        CALL initialize_list_of_zeros WITH the LENGTH OF string_input PLUS one ASSIGN TO dp
        SET element at position zero of dp TO one
        SET element at position one of dp TO one

        FOR index FROM two TO the LENGTH OF string_input
            IF element at position index MINUS one of string_input NOT EQUALS zero
                INCREMENT element at position index of dp BY element at position index MINUS one of dp
            END IF

            CALL convert_substring_from position index MINUS two TO index of string_input TO integer ASSIGN TO two_digit
            IF two_digit GREATER THAN OR EQUAL TO ten AND two_digit LESS THAN OR EQUAL TO twenty six
                INCREMENT element at position index of dp BY element at position index MINUS two of dp
            END IF
        END FOR

        RETURN element at the last position of dp
    END FUNCTION
END CLASS
