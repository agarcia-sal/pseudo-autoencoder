CLASS Solution
    FUNCTION isTransformable(s, t)
        CALL importDefaultDictHeapFunctions
        SET pos TO CALL createDefaultDictOfEmptyLists
        FOR each index AND character IN s
            CALL pushIntoHeap WITH element at position integer value of character OF pos AND value index
        END FOR
        FOR each index AND character IN t
            SET digit TO integer value of character
            IF element at position digit OF pos IS empty
                RETURN false
            END IF
            FOR smaller FROM zero TO digit MINUS one
                IF element at position smaller OF pos IS NOT empty AND element at position zero OF element at position smaller OF pos IS LESS THAN element at position zero OF element at position digit OF pos
                    RETURN false
                END IF
            END FOR
            CALL popFromHeap WITH element at position digit OF pos
        END FOR
        RETURN true
    END FUNCTION
END CLASS
