CLASS Solution
    FUNCTION killProcess(list_of_pid, list_of_ppid, kill_process)
        ASSIGN tree TO CALL defaultdict WITH empty list
        FOR each pair of child_process AND parent_process IN parallel FROM list_of_pid AND list_of_ppid
            APPEND child_process TO element at position parent_process OF tree
        END FOR
        ASSIGN queue TO list containing kill_process
        ASSIGN killed_processes TO empty list
        WHILE queue is not empty
            ASSIGN current_process TO element at position zero of queue
            REMOVE element at position zero of queue
            APPEND current_process TO killed_processes
            FOR each child_process IN element at position current_process OF tree
                APPEND child_process TO queue
            END FOR
        END WHILE
        RETURN killed_processes
    END FUNCTION
END CLASS
