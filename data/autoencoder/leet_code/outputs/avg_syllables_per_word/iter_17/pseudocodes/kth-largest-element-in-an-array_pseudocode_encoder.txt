CLASS Solution
    FUNCTION findKthLargest(list_of_numbers, k)
        SET minimum_heap TO empty list
        FOR each number IN list_of_numbers
            CALL helper FUNCTION heap_push WITH parameters minimum_heap AND number
            IF the LENGTH OF minimum_heap GREATER THAN k
                CALL helper FUNCTION heap_pop WITH parameter minimum_heap
            END IF
        END FOR
        RETURN element at position zero of minimum_heap
    END FUNCTION
END CLASS
