CLASS Solution
    FUNCTION minDistance(word1, word2)
        SET m TO the LENGTH OF word1
        SET n TO the LENGTH OF word2
        SET dp TO CALL InitializeTwoDimensionalArray WITH parameters m plus one AND n plus one
        
        FOR i FROM one TO m
            FOR j FROM one TO n
                IF element at position i minus one of word1 EQUALS element at position j minus one of word2
                    SET element at position i of row at position j of dp TO element at position i minus one of row at position j minus one of dp PLUS one
                ELSE
                    SET element at position i of row at position j of dp TO CALL MaximumValue WITH parameters element at position i minus one of row at position j of dp AND element at position i of row at position j minus one of dp
                END IF
            END FOR
        END FOR
        
        SET lcs_length TO element at position m of row at position n of dp
        SET minimum_number_of_deletions TO m MINUS lcs_length PLUS n MINUS lcs_length
        RETURN minimum_number_of_deletions
    END FUNCTION
END CLASS
