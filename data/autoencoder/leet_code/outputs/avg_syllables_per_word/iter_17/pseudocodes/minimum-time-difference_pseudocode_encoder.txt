CLASS Solution
    FUNCTION findMinDifference(timePoints)
        FUNCTION to_minutes(time)
            CALL map FUNCTION WITH parameters int AND CALL split FUNCTION WITH parameter colon ON time
            DESTRUCTURE hours AND minutes FROM RESULT
            RETURN hours MULTIPLIED BY sixty PLUS minutes
        END FUNCTION

        SET minutes_list TO CALL map FUNCTION WITH parameters to_minutes AND timePoints CONVERTED TO list
        CALL sort FUNCTION ON minutes_list

        SET min_diff TO positive infinity
        FOR index FROM one TO the LENGTH OF minutes_list MINUS one
            SET current_difference TO element at position index of minutes_list MINUS element at position index MINUS one of minutes_list
            SET min_diff TO THE SMALLER OF min_diff AND current_difference
        END FOR

        SET circular_difference TO one thousand four hundred forty MINUS element at last position of minutes_list PLUS element at first position of minutes_list
        SET min_diff TO THE SMALLER OF min_diff AND circular_difference

        RETURN min_diff
    END FUNCTION
END CLASS
