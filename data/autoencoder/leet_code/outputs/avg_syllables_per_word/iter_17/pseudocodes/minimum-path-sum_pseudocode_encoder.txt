CLASS Solution
    FUNCTION minPathSum(grid)
        IF grid IS EMPTY OR element at position zero of grid IS EMPTY
            RETURN zero
        END IF

        SET m TO the LENGTH OF grid
        SET n TO the LENGTH OF element at position zero of grid

        SET dp TO CALL create_two_dimensional_list WITH parameters m AND n

        SET element at position zero of dp AND position zero TO element at position zero of grid AND position zero

        FOR j FROM one TO n MINUS one
            SET element at position zero of dp AND position j TO element at position zero of dp AND position j MINUS one PLUS element at position zero of grid AND position j
        END FOR

        FOR i FROM one TO m MINUS one
            SET element at position i of dp AND position zero TO element at position i MINUS one of dp AND position zero PLUS element at position i of grid AND position zero
        END FOR

        FOR i FROM one TO m MINUS one
            FOR j FROM one TO n MINUS one
                SET element at position i of dp AND position j TO the MINIMUM OF element at position i MINUS one of dp AND position j AND element at position i of dp AND position j MINUS one PLUS element at position i of grid AND position j
            END FOR
        END FOR

        RETURN element at position m MINUS one of dp AND position n MINUS one
    END FUNCTION

        DESCRIPTION: Auxiliary function to create a two dimensional list initialized with zero
    END FUNCTION
END CLASS
