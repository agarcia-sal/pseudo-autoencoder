CLASS Solution
    FUNCTION hIndex(citations_list)
        SET number_of_citations TO the LENGTH OF citations_list
        SET left_pointer TO zero
        SET right_pointer TO number_of_citations MINUS one
        SET h_index_value TO zero

        WHILE left_pointer LESS THAN OR EQUAL TO right_pointer holds
            SET middle_position TO left_pointer PLUS the INTEGER DIVISION OF right_pointer MINUS left_pointer BY two
            IF element at position middle_position of citations_list GREATER THAN OR EQUAL TO number_of_citations MINUS middle_position
                SET h_index_value TO number_of_citations MINUS middle_position
                SET right_pointer TO middle_position MINUS one
            ELSE
                SET left_pointer TO middle_position PLUS one
            END IF
        END WHILE

        RETURN h_index_value
    END FUNCTION
END CLASS
