CLASS Solution
    FUNCTION numRescueBoats(list_of_people, weight_limit)
        CALL sort ON list_of_people
        SET left_pointer TO zero
        SET right_pointer TO the LENGTH OF list_of_people MINUS one
        SET boat_count TO zero
        WHILE left_pointer IS LESS THAN OR EQUAL TO right_pointer
            IF element at position left_pointer of list_of_people PLUS element at position right_pointer of list_of_people IS LESS THAN OR EQUAL TO weight_limit
                INCREMENT left_pointer BY one
            END IF
            DECREMENT right_pointer BY one
            INCREMENT boat_count BY one
        END WHILE
        RETURN boat_count
    END FUNCTION
END CLASS
