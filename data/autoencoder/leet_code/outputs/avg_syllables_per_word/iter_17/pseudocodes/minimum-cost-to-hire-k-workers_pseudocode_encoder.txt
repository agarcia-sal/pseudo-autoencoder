    FUNCTION mincostToHireWorkers(quality, wage, k)
        SET workers TO CALL sort_ratio_quality WITH parameters wage AND quality
        SET max_heap TO empty list
        SET total_quality TO zero
        SET min_cost TO infinity

        FOR each pair_of_ratio_and_quality IN workers
            SET ratio TO element at position one of pair_of_ratio_and_quality
            SET q TO element at position two of pair_of_ratio_and_quality

            CALL push_element TO add negative q TO max_heap
            INCREMENT total_quality BY q

            IF the LENGTH OF max_heap IS GREATER THAN k
                INCREMENT total_quality BY CALL pop_element FROM max_heap
            END IF

            IF the LENGTH OF max_heap IS EQUAL TO k
                SET min_cost TO the MINIMUM OF min_cost AND total_quality MULTIPLIED BY ratio
            END IF
        END FOR

        RETURN min_cost
    END FUNCTION
