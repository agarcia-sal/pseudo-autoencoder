CLASS Solution
    FUNCTION maxTaskAssign(tasks, workers, pills, strength)
        CALL tasks.sort WITH no parameters
        CALL workers.sort WITH no parameters

        FUNCTION canAssignTasks(k)
            SET available_workers TO a COPY OF the SUBLIST from position LENGTH_OF workers MINUS k TO the END OF workers
            SET pills_used TO zero

            FOR i FROM k MINUS one DOWN TO zero
                SET task TO element at position i of tasks
                SET worker_idx TO CALL bisect_left WITH parameters available_workers AND task
                IF worker_idx IS LESS THAN the LENGTH OF available_workers
                    CALL REMOVE element AT position worker_idx FROM available_workers
                ELSE IF pills_used IS LESS THAN pills
                    SET worker_idx TO CALL bisect_left WITH parameters available_workers AND task MINUS strength
                    IF worker_idx IS LESS THAN the LENGTH OF available_workers
                        CALL REMOVE element AT position worker_idx FROM available_workers
                        INCREMENT pills_used BY one
                    ELSE
                        RETURN False
                    END IF
                ELSE
                    RETURN False
                END IF
            END FOR
            RETURN True
        END FUNCTION

        SET left TO zero
        SET right TO the MINIMUM OF the LENGTH OF tasks AND the LENGTH OF workers PLUS one
        WHILE left IS LESS THAN right
            SET mid TO the INTEGER DIVISION OF left PLUS right BY two
            IF CALL canAssignTasks WITH parameter mid
                SET left TO mid PLUS one
            ELSE
                SET right TO mid
            END IF
        END WHILE
        RETURN left MINUS one
    END FUNCTION
END CLASS
