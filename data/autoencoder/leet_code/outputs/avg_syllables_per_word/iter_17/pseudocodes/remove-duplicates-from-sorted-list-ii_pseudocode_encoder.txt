CLASS ListNode
    FUNCTION __init__(val EQUALS zero, next EQUALS None)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values: list)
    IF values IS empty
        RETURN None
    END IF
    SET head TO CALL ListNode WITH element at position zero of values
    SET p TO head
    FOR each val IN the sublist of values from position one TO the end
        SET node TO CALL ListNode WITH val
        SET pointer next of p TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS None AND p2 IS None
        RETURN True
    END IF
    IF p1 IS None OR p2 IS None
        RETURN False
    END IF
    RETURN pointer val of p1 EQUALS pointer val of p2 AND CALL is_same_list WITH pointer next of p1 AND pointer next of p2
END FUNCTION

CLASS Solution
    FUNCTION deleteDuplicates(head: Optional ListNode) RETURNS Optional ListNode
        SET dummy TO CALL ListNode WITH zero and head
        SET prev TO dummy
        WHILE head IS NOT None
            IF pointer next of head IS NOT None AND pointer val of head EQUALS pointer val of pointer next of head
                WHILE pointer next of head IS NOT None AND pointer val of head EQUALS pointer val of pointer next of head
                    SET head TO pointer next of head
                END WHILE
                SET pointer next of prev TO pointer next of head
            ELSE
                SET prev TO pointer next of prev
            END IF
            SET head TO pointer next of head
        END WHILE
        RETURN pointer next of dummy
    END FUNCTION
END CLASS
