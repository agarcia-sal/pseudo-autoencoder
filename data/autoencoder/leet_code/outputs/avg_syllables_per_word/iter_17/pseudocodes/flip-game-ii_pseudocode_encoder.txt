CLASS Solution
    FUNCTION canWin(currentState)
        SET memo TO a new empty dictionary

        FUNCTION can_win_helper(state)
            IF state IS IN memo
                RETURN the element at position state of memo
            END IF

            FOR index FROM zero TO the LENGTH OF state MINUS one MINUS one
                IF the substring from position index TO index PLUS one of state EQUALS plus plus
                    SET new_state TO substring from position zero TO index of state CONCATENATED BY minus minus CONCATENATED BY substring from position index PLUS two TO the END OF state
                    IF NOT CALL can_win_helper WITH new_state
                        SET the element at position state of memo TO true
                        RETURN true
                    END IF
                END IF
            END FOR

            SET the element at position state of memo TO false
            RETURN false
        END FUNCTION

        RETURN CALL can_win_helper WITH currentState
    END FUNCTION
END CLASS
