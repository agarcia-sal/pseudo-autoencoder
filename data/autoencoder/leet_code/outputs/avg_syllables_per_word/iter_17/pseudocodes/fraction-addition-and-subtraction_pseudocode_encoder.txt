CLASS Solution
    FUNCTION fractionAddition(expression)
        SET list_of_fractions TO CALL create_fraction_list WITH expression
        SET result_fraction TO CALL sum_fractions WITH list_of_fractions
        SET output_string TO CALL format_fraction_as_string WITH result_fraction
        RETURN output_string
    END FUNCTION

    FUNCTION create_fraction_list WITH expression
        SET modified_expression TO CALL replace_hyphen_with_plus_negative IN expression
        SET list_of_fraction_strings TO CALL split_string_by_plus WITH modified_expression
        SET filtered_fraction_strings TO CALL filter_empty_strings IN list_of_fraction_strings
        SET fraction_list TO CALL convert_strings_to_fractions WITH filtered_fraction_strings
        RETURN fraction_list
    END FUNCTION

    FUNCTION replace_hyphen_with_plus_negative IN expression
        REPLACE each occurrence of hyphen character IN expression WITH plus character followed BY negative sign
        RETURN resulting string
    END FUNCTION

    FUNCTION split_string_by_plus WITH input_string
        SPLIT input_string by plus character INTO list_of_strings
        RETURN list_of_strings
    END FUNCTION

    FUNCTION filter_empty_strings IN list_of_strings
        RETURN list of strings where each string IS NOT empty FROM list_of_strings
    END FUNCTION

    FUNCTION convert_strings_to_fractions WITH list_of_fraction_strings
        FOR each fraction_string IN list_of_fraction_strings
            CONVERT fraction_string TO Fraction object AND APPEND TO fraction_list
        END FOR
        RETURN fraction_list
    END FUNCTION

    FUNCTION sum_fractions WITH list_of_fractions
        SET total_fraction TO zero Fraction object
        FOR each fraction_element IN list_of_fractions
            INCREMENT total_fraction BY fraction_element
        END FOR
        RETURN total_fraction
    END FUNCTION

    FUNCTION format_fraction_as_string WITH fraction_object
        SET numerator_string TO string form OF numerator OF fraction_object
        SET denominator_string TO string form OF denominator OF fraction_object
        SET formatted_string TO numerator_string CONCATENATED WITH forward slash CONCATENATED WITH denominator_string
        RETURN formatted_string
    END FUNCTION
END CLASS
