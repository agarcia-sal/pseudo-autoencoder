CLASS Solution
    FUNCTION find132pattern(nums)
        SET second TO negative infinity
        SET stack TO empty list
        FOR each num IN the REVERSE ORDER of nums
            IF num LESS THAN second
                RETURN true
            END IF
            WHILE stack is NOT empty AND the element at the LAST position of stack LESS THAN num
                SET second TO REMOVE the element at the LAST position of stack
            END WHILE
            APPEND num TO stack
        END FOR
        RETURN false
    END FUNCTION
END CLASS
