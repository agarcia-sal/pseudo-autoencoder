CLASS Solution
    FUNCTION permute(list_of_numbers)
        FUNCTION backtrack(current_path, remaining_numbers)
            IF remaining_numbers IS empty
                APPEND current_path TO permutations
                RETURN
            END IF
            FOR index FROM zero TO the LENGTH OF remaining_numbers MINUS one
                SET new_path TO current_path PLUS element at position index of remaining_numbers
                SET new_remaining TO substring from position zero TO index MINUS one of remaining_numbers PLUS substring from position index PLUS one TO the end of remaining_numbers
                CALL backtrack WITH new_path AND new_remaining
            END FOR
        END FUNCTION

        SET permutations TO empty list
        CALL backtrack WITH empty list AND list_of_numbers
        RETURN permutations
    END FUNCTION
END CLASS
