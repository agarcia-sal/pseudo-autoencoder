CLASS Solution
    FUNCTION findReplaceString(s STRING, indices LIST OF integer, sources LIST OF STRING, targets LIST OF STRING) RETURNS STRING
        SET replacements TO CALL sorted WITH PARAMETERS OF CALL zip WITH PARAMETERS indices sources targets
        SET result TO empty list
        SET prev_end TO zero
        FOR each tuple_element IN replacements
            SET index TO element at position one of tuple_element
            SET source TO element at position two of tuple_element
            SET target TO element at position three of tuple_element
            APPEND substring from position prev_end TO index of string s TO result
            IF substring from position index TO index PLUS the LENGTH OF source of string s EQUALS source
                APPEND target TO result
            ELSE
                APPEND substring from position index TO index PLUS the LENGTH OF source of string s TO result
            END IF
            SET prev_end TO index PLUS the LENGTH OF source
        END FOR
        APPEND substring from position prev_end TO the END of string s TO result
        RETURN the CONCATENATION OF all elements of list result
    END FUNCTION
END CLASS
