CLASS Solution
    FUNCTION integerReplacement(n)
        SET count TO zero
        WHILE n IS NOT EQUALS TO one
            IF n MODULO two IS EQUALS TO zero
                SET n TO n DIVIDED BY two
            ELSE
                IF n IS EQUALS TO three
                    DECREMENT n BY one
                ELSE IF (n PLUS one) MODULO four IS EQUALS TO zero
                    INCREMENT n BY one
                ELSE
                    DECREMENT n BY one
                END IF
            END IF
            INCREMENT count BY one
        END WHILE
        RETURN count
    END FUNCTION
END CLASS
