CLASS Solution
    FUNCTION rearrangeSticks(integer n, integer k)
        SET MODULO TO one billion PLUS seven
        SET dp_table TO a TWO DIMENSIONAL list OF zeros LENGTH n PLUS one BY k PLUS one
        SET element at position zero of list at position zero of dp_table TO one
        FOR integer i FROM one TO n
            FOR integer j FROM one TO k
                SET element at position j of list at position i of dp_table TO 
                    element at position j MINUS one of list at position i MINUS one of dp_table PLUS 
                    (i MINUS one) MULTIPLIED BY element at position j of list at position i MINUS one of dp_table
                    MODULO MODULO
            END FOR
        END FOR
        RETURN element at position k of list at position n of dp_table
    END FUNCTION
END CLASS
