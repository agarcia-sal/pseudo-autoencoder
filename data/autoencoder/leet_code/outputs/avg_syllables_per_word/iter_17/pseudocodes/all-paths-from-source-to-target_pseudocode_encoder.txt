CLASS Solution
    FUNCTION allPathsSourceTarget(graph_of_lists_of_integers)
        SET number_of_nodes TO the LENGTH OF graph_of_lists_of_integers
        ASSIGN queue TO an empty double ended queue initialized WITH a list containing zero
        ASSIGN answer_list TO an empty list
        WHILE queue holds elements
            SET path TO the element removed from the left end of queue
            SET current_node TO element at position the LENGTH OF path MINUS one of path
            IF current_node EQUALS number_of_nodes MINUS one
                APPEND path TO answer_list
                CONTINUE
            END IF
            FOR each neighbor_node IN element at position current_node of graph_of_lists_of_integers
                APPEND path concatenated WITH a list containing neighbor_node TO queue
            END FOR
        END WHILE
        RETURN answer_list
    END FUNCTION
END CLASS
