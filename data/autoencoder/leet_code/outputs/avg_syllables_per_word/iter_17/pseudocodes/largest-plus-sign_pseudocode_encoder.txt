CLASS Solution
    FUNCTION orderOfLargestPlusSign(n, mines)
        SET mines_set TO CALL CreateSetOfTuples WITH mines
        
        SET dp TO CALL InitializeThreeDimensionalGrid WITH n AND zero
        
        SET directions TO CALL InitializeDirections
        
        FOR i FROM zero TO n MINUS one
            FOR j FROM zero TO n MINUS one
                IF element at position i and j is NOT IN mines_set
                    IF j GREATER THAN zero
                        SET element at position j of element at position i of dp at position zero TO element at position j MINUS one of element at position i of dp at position zero PLUS one
                    END IF
                    IF i GREATER THAN zero
                        SET element at position j of element at position i of dp at position one TO element at position j of element at position i MINUS one of dp at position one PLUS one
                    END IF
                END IF
            END FOR
        END FOR
        
        SET result TO zero
        
        FOR i FROM n MINUS one TO zero DECREMENT BY one
            FOR j FROM n MINUS one TO zero DECREMENT BY one
                IF element at position i and j is NOT IN mines_set
                    IF j LESS THAN n MINUS one
                        SET element at position j of element at position i of dp at position two TO element at position j PLUS one of element at position i of dp at position two PLUS one
                    END IF
                    IF i LESS THAN n MINUS one
                        SET element at position j of element at position i of dp at position three TO element at position j of element at position i PLUS one of dp at position three PLUS one
                    END IF
                    SET plus_sign_order TO CALL MinimumOfList WITH element at position j of element at position i of dp
                    SET result TO CALL MaximumOfTwoValues WITH result AND plus_sign_order
                END IF
            END FOR
        END FOR
        
        RETURN result
    END FUNCTION
END CLASS
