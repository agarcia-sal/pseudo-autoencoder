CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values list_of_values)
    IF values IS empty
        RETURN None
    END IF
    SET root TO CALL TreeNode WITH the element at position zero of values
    SET i TO one
    SET queue TO CALL deque WITH no parameters
    CALL queue APPEND root
    WHILE queue IS NOT empty
        SET node TO CALL queue POPLEFT WITH no parameters
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None
            SET node.left TO CALL TreeNode WITH the element at position i of values
            CALL queue APPEND node.left
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF values AND the element at position i of values IS NOT None
            SET node.right TO CALL TreeNode WITH the element at position i of values
            CALL queue APPEND node.right
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p tree_node, q tree_node)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUAL TO q.val
        RETURN False
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION findBottomLeftValue(self, root tree_node)
        SET queue TO CALL deque WITH a list containing root
        SET leftmost_value TO None
        WHILE queue IS NOT empty
            SET level_size TO the LENGTH OF queue
            FOR i FROM zero TO level_size MINUS one
                SET node TO CALL queue POPLEFT WITH no parameters
                IF i EQUALS zero
                    SET leftmost_value TO node.val
                END IF
                IF node.left IS NOT None
                    CALL queue APPEND node.left
                END IF
                IF node.right IS NOT None
                    CALL queue APPEND node.right
                END IF
            END FOR
        END WHILE
        RETURN leftmost_value
    END FUNCTION
END CLASS
