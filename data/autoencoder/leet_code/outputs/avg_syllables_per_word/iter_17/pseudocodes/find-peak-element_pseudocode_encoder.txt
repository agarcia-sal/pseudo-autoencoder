CLASS Solution
    FUNCTION findPeakElement(list_of_numbers)
        SET left TO zero
        SET right TO the LENGTH OF list_of_numbers MINUS one
        WHILE left LESS THAN right holds
            SET mid TO left PLUS right DIVIDED BY two rounded down to nearest integer
            IF element at position mid of list_of_numbers GREATER THAN element at position mid PLUS one of list_of_numbers
                SET right TO mid
            ELSE
                SET left TO mid PLUS one
            END IF
        END WHILE
        RETURN left
    END FUNCTION
END CLASS
