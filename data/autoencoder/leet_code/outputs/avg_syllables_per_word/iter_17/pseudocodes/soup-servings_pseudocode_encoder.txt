CLASS Solution
    FUNCTION soupServings(n)
        IF n GREATER THAN four thousand eight hundred
            RETURN one
        END IF

        SET n TO n PLUS twenty four DIVIDED BY twenty five

        SET memo TO auxiliary FUNCTION InitializeMemo

        FUNCTION dp(a, b)
            IF pair a and b IS IN memo
                RETURN element at pair a and b OF memo
            END IF

            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero
                RETURN one half
            END IF

            IF a LESS THAN OR EQUAL TO zero
                RETURN one
            END IF

            IF b LESS THAN OR EQUAL TO zero
                RETURN zero
            END IF

            SET prob TO zero point two five MULTIPLIED BY
                CALL dp WITH parameters a MINUS four and b PLUS
                CALL dp WITH parameters a MINUS three and b MINUS one PLUS
                CALL dp WITH parameters a MINUS two and b MINUS two PLUS
                CALL dp WITH parameters a MINUS one and b MINUS three

            SET element at pair a and b OF memo TO prob
            RETURN prob
        END FUNCTION

        RETURN CALL dp WITH parameters n and n
END CLASS
