CLASS Solution
    FUNCTION scoreOfParentheses(string_s)
        SET stack_list TO empty list
        SET score_value TO zero
        FOR each character IN string_s
            IF character EQUALS open parenthesis
                APPEND score_value TO stack_list
                SET score_value TO zero
            ELSE
                SET popped_score TO the LAST element REMOVED FROM stack_list
                IF score_value EQUALS zero
                    SET score_value TO popped_score PLUS one
                ELSE
                    SET score_value TO popped_score PLUS two MULTIPLIED BY score_value
                END IF
            END IF
        END FOR
        RETURN score_value
    END FUNCTION
END CLASS
