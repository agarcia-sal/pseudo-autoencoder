CLASS Solution
    FUNCTION maximumScore(nums, k)
        SET left_pointer TO k
        SET right_pointer TO k
        SET current_minimum TO element at position k of nums
        SET maximum_score TO current_minimum
        WHILE left_pointer GREATER THAN zero OR right_pointer LESS THAN the LENGTH OF nums MINUS one holds
            IF left_pointer EQUALS zero
                INCREMENT right_pointer BY one
            ELSE IF right_pointer EQUALS the LENGTH OF nums MINUS one
                DECREMENT left_pointer BY one
            ELSE IF element at position left_pointer MINUS one of nums GREATER THAN OR EQUAL TO element at position right_pointer PLUS one of nums
                DECREMENT left_pointer BY one
            ELSE
                INCREMENT right_pointer BY one
            END IF
            SET current_minimum TO the MINIMUM OF current_minimum element at position left_pointer of nums element at position right_pointer of nums
            SET score_value TO current_minimum MULTIPLIED BY right_pointer MINUS left_pointer PLUS one
            SET maximum_score TO the MAXIMUM OF maximum_score score_value
        END WHILE
        RETURN maximum_score
    END FUNCTION
END CLASS
