CLASS Solution
    FUNCTION maxDotProduct(nums1, nums2)
        SET m TO the LENGTH OF nums1
        SET n TO the LENGTH OF nums2

        CALL initialize_dp_table WITH parameters m plus one AND n plus one RETURNING dp

        FOR i FROM one TO m
            FOR j FROM one TO n
                SET current_product TO element at position i minus one of nums1 MULTIPLIED BY element at position j minus one of nums2

                SET dp element at row i column j TO the MAXIMUM OF 
                    current_product
                    element at row i minus one column j of dp
                    element at row i column j minus one of dp
                    element at row i minus one column j minus one of dp PLUS current_product
            END FOR
        END FOR

        RETURN element at row m column n of dp
    END FUNCTION

    FUNCTION initialize_dp_table(rows, columns)
        SET dp TO an empty list

        FOR each row_index FROM zero TO rows minus one
            SET current_row TO an empty list

            FOR each column_index FROM zero TO columns minus one
                CALL negative_infinity_value RETURNING negative_infinity
                APPEND negative_infinity TO current_row
            END FOR

            APPEND current_row TO dp
        END FOR

        RETURN dp
    END FUNCTION

    FUNCTION negative_infinity_value
        RETURN the negative infinity constant
    END FUNCTION
END CLASS
