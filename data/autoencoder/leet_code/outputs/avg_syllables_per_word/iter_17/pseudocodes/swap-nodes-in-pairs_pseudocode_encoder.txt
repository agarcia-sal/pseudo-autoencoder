CLASS ListNode
    FUNCTION __init__(val EQUALS zero, next EQUALS None)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values)
    IF values IS empty
        RETURN None
    END IF
    SET head TO new ListNode WITH element at position zero OF values
    SET p TO head
    FOR each val IN the elements of values FROM position one TO the end
        SET node TO new ListNode WITH val
        SET p.next TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS None AND p2 IS None
        RETURN True
    END IF
    IF p1 IS None OR p2 IS None
        RETURN False
    END IF
    RETURN the element val at p1 EQUALS the element val at p2 AND CALL is_same_list WITH the element next at p1 AND the element next at p2
END FUNCTION

CLASS Solution
    FUNCTION swapPairs(head)
        SET dummy TO new ListNode WITH zero
        SET dummy.next TO head
        SET current TO dummy
        WHILE the element next at current IS NOT None AND the element next at the element next at current IS NOT None
            SET first TO the element next at current
            SET second TO the element next at the element next at current
            SET current.next TO second
            SET first.next TO the element next at second
            SET second.next TO first
            SET current TO the element next at the element next at current
        END WHILE
        RETURN the element next at dummy
    END FUNCTION
END CLASS
