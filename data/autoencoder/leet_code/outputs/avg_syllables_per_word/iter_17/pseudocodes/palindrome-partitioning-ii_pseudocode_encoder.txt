CLASS Solution
    FUNCTION minCut(s)
        SET n TO the LENGTH OF s
        IF n LESS THAN OR EQUAL TO one
            RETURN zero
        END IF

        CALL create_is_palindrome_table WITH s AND n TO GET is_palindrome

        CREATE min_cuts AS a LIST of n elements WITH initial value zero

        FOR i FROM zero TO n MINUS one
            IF element at position zero and position i of is_palindrome IS True
                SET element at position i of min_cuts TO zero
            ELSE
                SET element at position i of min_cuts TO infinity
                FOR j FROM one TO i
                    IF element at position j and position i of is_palindrome IS True AND element at position j MINUS one of min_cuts PLUS one LESS THAN element at position i of min_cuts
                        SET element at position i of min_cuts TO element at position j MINUS one of min_cuts PLUS one
                    END IF
                END FOR
            END IF
        END FOR

        RETURN element at position n MINUS one of min_cuts
    END FUNCTION

    FUNCTION create_is_palindrome_table(s, n)
        CREATE is_palindrome AS a TWO DIMENSIONAL LIST of n BY n WITH initial value False

        FOR i FROM zero TO n MINUS one
            SET element at position i and position i of is_palindrome TO True
        END FOR

        FOR length FROM two TO n
            FOR start FROM zero TO n MINUS length
                SET end TO start PLUS length MINUS one
                IF length EQUALS two
                    SET element at position start and position end of is_palindrome TO element at position start of s EQUALS element at position end of s
                ELSE
                    SET element at position start and position end of is_palindrome TO element at position start of s EQUALS element at position end of s AND element at position start PLUS one and position end MINUS one of is_palindrome
                END IF
            END FOR
        END FOR

        RETURN is_palindrome
    END FUNCTION
END CLASS
