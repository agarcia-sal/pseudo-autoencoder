CLASS Solution
    FUNCTION minCostToSupplyWater(n, wells, pipes)
        FOR index FROM zero TO n MINUS one
            APPEND list containing zero, index PLUS one, element at position index of wells TO pipes
        END FOR

        CALL SortPipesByCost WITH pipes

        SET parent TO CALL InitializeParentList WITH n PLUS one

        FUNCTION find(x)
            IF element at position x of parent IS NOT EQUAL TO x
                SET element at position x of parent TO CALL find WITH element at position x of parent
            END IF
            RETURN element at position x of parent
        END FUNCTION

        FUNCTION union(x, y)
            SET rootX TO CALL find WITH x
            SET rootY TO CALL find WITH y
            IF rootX IS NOT EQUAL TO rootY
                SET element at position rootX of parent TO rootY
                RETURN True
            END IF
            RETURN False
        END FUNCTION

        SET min_cost TO zero

        FOR each triple consisting of house1, house2, cost IN pipes
            IF CALL union WITH house1, house2 RETURNS True
                INCREMENT min_cost BY cost
            END IF
        END FOR

        RETURN min_cost
    END FUNCTION
