CLASS Solution
    FUNCTION restoreIpAddresses(string_s)
        FUNCTION is_valid(segment)
            RETURN the LENGTH OF segment EQUALS one OR 
                   the ELEMENT at position zero of segment NOT EQUAL TO zero AND 
                   the INTEGER VALUE OF segment LESS THAN OR EQUAL TO two hundred fifty five
        END FUNCTION

        FUNCTION backtrack(integer_start, list_path)
            IF the LENGTH OF list_path EQUALS four
                IF integer_start EQUALS the LENGTH OF string_s
                    APPEND the JOINED list_path WITH DOT TO result
                END IF
                RETURN
            END IF
            FOR integer_length FROM one TO three
                IF integer_start PLUS integer_length LESS THAN OR EQUAL TO the LENGTH OF string_s
                    SET string_segment TO the SUBSTRING FROM position integer_start TO integer_start PLUS integer_length OF string_s
                    IF is_valid(string_segment) HOLDS
                        CALL backtrack WITH integer_start PLUS integer_length AND list_path WITH string_segment APPENDED AS PARAMETERS
                    END IF
                END IF
            END FOR
        END FUNCTION

        SET result TO an empty list
        CALL backtrack WITH zero AND empty list AS PARAMETERS
        RETURN result
    END FUNCTION
END CLASS
