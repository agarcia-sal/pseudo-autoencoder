CLASS Solution
    FUNCTION lengthOfLongestSubstringTwoDistinct(input_string)
        CALL char_count TO CALL defaultdict WITH parameter integer_association
        
        SET left_pointer TO zero
        SET maximum_length TO zero
        
        FOR index_pointer FROM zero TO the LENGTH OF input_string MINUS one
            INCREMENT element at position index_pointer of input_string IN char_count BY one
            
            WHILE the LENGTH OF char_count IS GREATER THAN two
                DECREMENT element at position left_pointer of input_string IN char_count BY one
                
                IF element at position left_pointer of input_string IN char_count EQUALS zero
                    CALL removal of element at position left_pointer of input_string FROM char_count
                END IF
                
                INCREMENT left_pointer BY one
            END WHILE
            
            SET maximum_length TO the GREATER OF maximum_length AND index_pointer MINUS left_pointer PLUS one
        END FOR
        
        RETURN maximum_length
    END FUNCTION
END CLASS
