CLASS Solution
    FUNCTION findNthDigit(n)
        SET length TO one
        SET count TO nine
        SET start TO one
        
        WHILE n GREATER THAN length MULTIPLIED BY count holds
            SET n TO n MINUS length MULTIPLIED BY count
            INCREMENT length BY one
            SET count TO count MULTIPLIED BY ten
            SET start TO start MULTIPLIED BY ten
        END WHILE
        
        SET start TO start PLUS the INTEGER DIVISION of n MINUS one BY length
        RETURN the integer CONVERTED FROM the STRING REPRESENTATION OF start at POSITION of the REMAINDER of n MINUS one DIVIDED BY length
    END FUNCTION
END CLASS
