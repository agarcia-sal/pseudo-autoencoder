CLASS Solution
    FUNCTION maxNumber(list_nums1, list_nums2, integer_k)
        FUNCTION maxSingleArray(list_nums, integer_k)
            SET integer_drop TO the LENGTH OF list_nums MINUS integer_k
            SET list_stack TO an empty list
            FOR each number IN list_nums
                WHILE integer_drop IS GREATER THAN zero AND list_stack IS NOT empty AND the element at position the LENGTH OF list_stack MINUS one of list_stack IS LESS THAN number
                    CALL REMOVE element FROM list_stack
                    DECREMENT integer_drop BY one
                END WHILE
                CALL APPEND number TO list_stack
            END FOR
            RETURN the SUBLIST from position zero TO integer_k MINUS one of list_stack
        END FUNCTION

        FUNCTION merge(list_arr1, list_arr2)
            RETURN the RESULT OF CALLING FUNCTION max WITH list_arr1 AND list_arr2 AND THEN POPPING element at position zero REPEATED the SUM OF the LENGTH OF list_arr1 AND the LENGTH OF list_arr2 TIMES
        END FUNCTION

        SET list_max_number TO an empty list
        FOR integer_i FROM zero TO integer_k
            IF integer_i IS LESS THAN OR EQUAL TO the LENGTH OF list_nums1 AND integer_k MINUS integer_i IS LESS THAN OR EQUAL TO the LENGTH OF list_nums2
                SET list_candidate TO CALL merge WITH the RESULT OF CALLING maxSingleArray WITH list_nums1 AND integer_i AND WITH the RESULT OF CALLING maxSingleArray WITH list_nums2 AND integer_k MINUS integer_i
                SET list_max_number TO the RESULT OF CALLING max WITH list_max_number AND list_candidate
            END IF
        END FOR

        RETURN list_max_number
    END FUNCTION
END CLASS
