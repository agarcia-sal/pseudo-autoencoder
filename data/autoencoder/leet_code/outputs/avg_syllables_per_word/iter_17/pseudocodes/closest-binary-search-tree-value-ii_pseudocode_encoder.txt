CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(list_of_values)
    IF list_of_values IS empty
        RETURN None
    END IF
    SET root TO A NEW TreeNode WITH parameter value being element at position zero OF list_of_values
    SET i TO one
    INITIALIZE queue AS A NEW deque
    APPEND root TO queue
    WHILE queue IS NOT empty
        SET node TO THE FIRST element REMOVED FROM queue
        IF i IS LESS THAN the LENGTH OF list_of_values AND element at position i OF list_of_values IS NOT None
            SET node.left TO A NEW TreeNode WITH parameter value BEING element at position i OF list_of_values
            APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF list_of_values AND element at position i OF list_of_values IS NOT None
            SET node.right TO A NEW TreeNode WITH parameter value BEING element at position i OF list_of_values
            APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUALS q.val
        RETURN False
    ELSE
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)
    END IF
END FUNCTION

CLASS Solution
    FUNCTION closestKValues(self, root, target, k)
        FUNCTION inorder_traversal(node)
            IF node IS None
                RETURN
            END IF
            CALL inorder_traversal WITH parameter node.left
            IF the LENGTH OF result IS LESS THAN k
                APPEND node.val TO result
            ELSE
                IF the ABSOLUTE VALUE OF node.val MINUS target IS LESS THAN the ABSOLUTE VALUE OF element at position zero OF result MINUS target
                    REMOVE the FIRST element FROM result
                    APPEND node.val TO result
                ELSE
                    RETURN
                END IF
            END IF
            CALL inorder_traversal WITH parameter node.right
        END FUNCTION

        INITIALIZE result AS A NEW deque
        CALL inorder_traversal WITH parameter root
        RETURN CONVERT result TO list
    END FUNCTION
END CLASS
