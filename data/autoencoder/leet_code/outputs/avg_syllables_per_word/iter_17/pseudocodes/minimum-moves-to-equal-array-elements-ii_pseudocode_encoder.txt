CLASS Solution
    FUNCTION minMoves2(list_of_numbers)
        CALL list_of_numbers SORT
        SET middle_index TO the LENGTH OF list_of_numbers DIVIDED BY two
        SET median_value TO element at position middle_index of list_of_numbers
        SET total_moves TO zero
        FOR each number IN list_of_numbers
            INCREMENT total_moves BY the ABSOLUTE VALUE OF number MINUS median_value
        END FOR
        RETURN total_moves
    END FUNCTION
END CLASS
