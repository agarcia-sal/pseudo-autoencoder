CLASS Solution
    FUNCTION verifyPreorder(preorder_list_of_integers)
        SET stack TO empty list
        SET lower_bound TO negative infinity

        FOR each value IN preorder_list_of_integers
            IF value LESS THAN lower_bound
                RETURN false
            END IF

            WHILE LENGTH OF stack GREATER THAN zero AND value GREATER THAN element at position LENGTH OF stack MINUS one OF stack
                SET lower_bound TO CALL pop ON stack
            END WHILE

            CALL append WITH value ON stack
        END FOR

        RETURN true
    END FUNCTION
END CLASS
