CLASS Solution
    FUNCTION minDays(n)
        FUNCTION dp(remaining)
            IF remaining LESS THAN OR EQUAL TO one
                RETURN remaining
            END IF
            SET option_one TO remainder of remaining DIVIDED BY two PLUS CALL dp WITH remaining DIVIDED BY two
            SET option_two TO remainder of remaining DIVIDED BY three PLUS CALL dp WITH remaining DIVIDED BY three
            RETURN one PLUS the MINIMUM OF option_one AND option_two
        END FUNCTION
        RETURN CALL dp WITH n
END CLASS
