CLASS Solution
    FUNCTION letterCasePermutation(input_string)
        FUNCTION dfs(index)
            IF index GREATER THAN OR EQUAL TO the LENGTH OF temporary_list
                APPEND the CONCATENATION OF all elements of temporary_list TO result_list
                RETURN
            END IF
            CALL dfs WITH index PLUS one
            IF the element at position index of temporary_list IS AN ALPHABETICAL CHARACTER
                SET the element at position index of temporary_list TO the character of the ASCII code of the element at position index of temporary_list XORED WITH thirty two
                CALL dfs WITH index PLUS one
            END IF
        END FUNCTION

        SET temporary_list TO the LIST conversion of input_string
        SET result_list TO an empty LIST
        CALL dfs WITH zero
        RETURN result_list
    END FUNCTION
END CLASS
