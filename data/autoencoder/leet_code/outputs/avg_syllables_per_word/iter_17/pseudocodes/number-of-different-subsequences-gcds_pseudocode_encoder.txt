CLASS Solution
    FUNCTION countDifferentSubsequenceGCDs(list_of_numbers)
        SET maximum_number TO the maximum element IN list_of_numbers
        SET set_of_numbers TO the set CONTAINING all elements IN list_of_numbers
        SET count_of_gcds TO zero

        FOR number_g FROM one TO maximum_number
            SET gcd_value TO zero

            FOR multiple_m FROM number_g TO maximum_number STEP number_g
                IF multiple_m IS IN set_of_numbers
                    SET gcd_value TO the greatest common divisor OF gcd_value AND multiple_m
                    IF gcd_value EQUALS number_g
                        INCREMENT count_of_gcds BY one
                        BREAK OUT OF the inner loop
                    END IF
                END IF
            END FOR

        END FOR

        RETURN count_of_gcds
    END FUNCTION
END CLASS
