CLASS Solution
    FUNCTION minimumMountainRemovals(list_of_numbers)
        SET length_of_list TO the LENGTH OF list_of_numbers
        
        CALL initialize_list_of_ones WITH length_of_list RETURNING lis
        FOR index_i FROM zero TO length_of_list MINUS one
            FOR index_j FROM zero TO index_i MINUS one
                IF element at position index_i of list_of_numbers GREATER THAN element at position index_j of list_of_numbers
                    SET element at position index_i of lis TO the MAXIMUM OF element at position index_i of lis AND element at position index_j of lis PLUS one
                END IF
            END FOR
        END FOR
        
        CALL initialize_list_of_ones WITH length_of_list RETURNING lds
        FOR index_i FROM length_of_list MINUS one TO zero STEP MINUS one
            FOR index_j FROM index_i PLUS one TO length_of_list MINUS one
                IF element at position index_i of list_of_numbers GREATER THAN element at position index_j of list_of_numbers
                    SET element at position index_i of lds TO the MAXIMUM OF element at position index_i of lds AND element at position index_j of lds PLUS one
                END IF
            END FOR
        END FOR
        
        SET maximum_mountain_length TO zero
        FOR index_i FROM one TO length_of_list MINUS two
            IF element at position index_i of lis GREATER THAN one AND element at position index_i of lds GREATER THAN one
                SET maximum_mountain_length TO the MAXIMUM OF maximum_mountain_length AND element at position index_i of lis PLUS element at position index_i of lds MINUS one
            END IF
        END FOR
        
        RETURN length_of_list MINUS maximum_mountain_length
    END FUNCTION
END CLASS
