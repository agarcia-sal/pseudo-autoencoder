CLASS Solution
    FUNCTION numberOfArithmeticSlices(nums)
        IF the LENGTH OF nums IS LESS THAN three
            RETURN zero
        END IF
        SET total_slices TO zero
        SET current_length TO two
        FOR index FROM two TO the LENGTH OF nums MINUS one
            IF the ELEMENT AT position index OF nums MINUS the ELEMENT AT position index MINUS one OF nums EQUALS the ELEMENT AT position index MINUS one OF nums MINUS the ELEMENT AT position index MINUS two OF nums
                INCREMENT current_length BY one
                INCREMENT total_slices BY current_length MINUS two
            ELSE
                SET current_length TO two
            END IF
        END FOR
        RETURN total_slices
    END FUNCTION
END CLASS
