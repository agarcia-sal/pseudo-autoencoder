CLASS Solution
    FUNCTION subsets(list_of_numbers)
        IF the LENGTH OF list_of_numbers EQUALS zero
            RETURN a LIST containing a single empty LIST
        END IF
        SET subsets_without_first TO CALL subsets WITH the SUBLIST of list_of_numbers from position one TO the END
        SET subsets_with_first TO an empty LIST
        FOR each subset IN subsets_without_first
            SET new_subset TO a LIST with the element at position zero of list_of_numbers FOLLOWED BY subset
            APPEND new_subset TO subsets_with_first
        END FOR
        RETURN the concatenation of subsets_without_first AND subsets_with_first
    END FUNCTION
END CLASS
