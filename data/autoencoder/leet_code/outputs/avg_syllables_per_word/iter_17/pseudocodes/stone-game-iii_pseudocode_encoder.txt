CLASS Solution
    FUNCTION stoneGameIII(list_of_stone_values)
        SET number_of_stones TO the LENGTH OF list_of_stone_values
        SET list_dp TO a list of zero repeated number_of_stones PLUS one times
        
        FOR index FROM number_of_stones MINUS one DOWNTO zero
            SET element_at_position_index_of_list_dp TO element at position index of list_of_stone_values MINUS element at position index PLUS one of list_dp
            
            IF index PLUS two IS LESS THAN OR EQUAL TO number_of_stones
                SET element_at_position_index_of_list_dp TO the MAXIMUM OF element at position index of list_dp AND element at position index of list_of_stone_values PLUS element at position index PLUS one of list_of_stone_values MINUS element at position index PLUS two of list_dp
            END IF
            
            IF index PLUS three IS LESS THAN OR EQUAL TO number_of_stones
                SET element_at_position_index_of_list_dp TO the MAXIMUM OF element at position index of list_dp AND element at position index of list_of_stone_values PLUS element at position index PLUS one of list_of_stone_values PLUS element at position index PLUS two of list_of_stone_values MINUS element at position index PLUS three of list_dp
            END IF
        END FOR
        
        IF element at position zero of list_dp IS GREATER THAN zero
            RETURN Alice
        ELSE IF element at position zero of list_dp IS LESS THAN zero
            RETURN Bob
        ELSE
            RETURN Tie
        END IF
    END FUNCTION
END CLASS
