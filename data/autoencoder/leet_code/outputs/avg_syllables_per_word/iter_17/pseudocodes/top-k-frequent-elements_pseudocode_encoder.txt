CLASS Solution
    FUNCTION topKFrequent(list_of_numbers, integer_k)
        SET frequency_map TO CALL default_dictionary_of_integer
        FOR each number IN list_of_numbers
            INCREMENT element_at position number OF frequency_map BY one
        END FOR
        SET maximum_frequency TO the MAXIMUM OF the VALUES OF frequency_map
        SET buckets TO CALL create_list_of_empty_lists WITH size maximum_frequency PLUS one
        FOR each number_and_frequency IN the ITEMS OF frequency_map
            SET number TO element_at position zero OF number_and_frequency
            SET frequency TO element_at position one OF number_and_frequency
            APPEND number TO element_at position frequency OF buckets
        END FOR
        SET result_list TO empty list
        FOR frequency FROM maximum_frequency DOWN TO one
            IF the LENGTH OF element_at position frequency OF buckets IS GREATER THAN zero
                EXTEND result_list BY element_at position frequency OF buckets
                IF the LENGTH OF result_list IS GREATER THAN OR EQUAL TO integer_k
                    RETURN the SUBLIST from position zero TO integer_k OF result_list
                END IF
            END IF
        END FOR
        RETURN result_list
    END FUNCTION
END CLASS
