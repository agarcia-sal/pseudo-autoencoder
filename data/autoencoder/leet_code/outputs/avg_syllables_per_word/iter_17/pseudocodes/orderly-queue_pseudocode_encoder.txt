CLASS Solution
    FUNCTION orderlyQueue(string_s, integer_k)
        IF integer_k GREATER THAN one
            RETURN CALL sortStringDirectly WITH parameter string_s
        ELSE IF integer_k EQUALS one
            SET integer_n TO the LENGTH OF string_s
            SET string_smallest TO string_s
            FOR index_i FROM one TO integer_n MINUS one
                SET string_rotated TO the SUBSTRING from position index_i TO the END of string_s CONCATENATED WITH the SUBSTRING from position zero TO index_i MINUS one of string_s
                IF string_rotated LESS THAN string_smallest
                    SET string_smallest TO string_rotated
                END IF
            END FOR
            RETURN string_smallest
        END IF
    END FUNCTION

    FUNCTION sortStringDirectly(parameter_string)
        RETURN STRING BUILT BY JOINING the SORTED ELEMENTS OF parameter_string
    END FUNCTION
END CLASS
