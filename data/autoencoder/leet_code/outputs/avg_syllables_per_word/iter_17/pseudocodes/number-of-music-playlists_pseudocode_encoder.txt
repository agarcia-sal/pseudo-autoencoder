CLASS Solution
    FUNCTION numMusicPlaylists(n, goal, k)
        SET MOD TO ten MULTIPLIED BY to THE POWER OF nine PLUS seven
        SET dynamic_programming_table TO the RESULT OF initialize_two_dimensional_list WITH parameters goal PLUS one AND n PLUS one FILLED WITH zero
        SET element at position zero OF list at position zero OF dynamic_programming_table TO one
        FOR index_i FROM one TO goal
            FOR index_j FROM one TO n
                SET element at position index_i OF list at position index_j OF dynamic_programming_table TO element at position index_i MINUS one OF list at position index_j MINUS one OF dynamic_programming_table MULTIPLIED BY n MINUS index_j MINUS one PLUS one
                SET element at position index_i OF list at position index_j OF dynamic_programming_table TO element at position index_i OF list at position index_j OF dynamic_programming_table MODULO MOD
                IF index_j IS GREATER THAN k
                    SET element at position index_i OF list at position index_j OF dynamic_programming_table TO element at position index_i OF list at position index_j OF dynamic_programming_table PLUS element at position index_i MINUS one OF list at position index_j OF dynamic_programming_table MULTIPLIED BY index_j MINUS k
                    SET element at position index_i OF list at position index_j OF dynamic_programming_table TO element at position index_i OF list at position index_j OF dynamic_programming_table MODULO MOD
                END IF
            END FOR
        END FOR
        RETURN element at position goal OF list at position n OF dynamic_programming_table
    END FUNCTION
END CLASS
