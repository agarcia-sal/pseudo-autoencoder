CLASS Solution
    FUNCTION superEggDrop(k, n)
        SET dp TO auxiliary_FUNCTION InitializeListWithZeroes OF size k PLUS one
        SET moves TO zero
        WHILE element at position k of dp IS LESS THAN n holds
            INCREMENT moves BY one
            FOR eggs FROM k DOWN TO one
                SET element at position eggs of dp TO one PLUS element at position eggs MINUS one of dp PLUS element at position eggs of dp
            END FOR
        END WHILE
        RETURN moves
    END FUNCTION
END CLASS
