CLASS Solution
    FUNCTION canConvert(str1, str2)
        IF str1 EQUALS str2
            RETURN True
        END IF

        SET conversion_map TO auxiliary FUNCTION initialize_mapping

        FOR each pair of character_one AND character_two IN pairing of elements at corresponding positions of str1 AND str2
            IF character_one IS IN keys of conversion_map
                IF value associated with character_one in conversion_map DOES NOT EQUAL character_two
                    RETURN False
                END IF
            ELSE
                SET value associated with character_one in conversion_map TO character_two
            END IF
        END FOR

        IF the number of unique elements IN str2 EQUALS twenty six
            RETURN False
        END IF

        RETURN True
    END FUNCTION
END CLASS
