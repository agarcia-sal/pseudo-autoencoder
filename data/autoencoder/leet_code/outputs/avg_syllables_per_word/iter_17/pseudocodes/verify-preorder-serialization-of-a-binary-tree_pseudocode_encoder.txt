CLASS Solution
    FUNCTION isValidSerialization(preorder)
        SET list_of_nodes TO CALL split_string_by_comma WITH preorder
        SET slot_count TO one
        FOR each node IN list_of_nodes
            IF slot_count EQUALS zero
                RETURN false
            END IF
            DECREMENT slot_count BY one
            IF node NOT EQUALS hash_symbol
                INCREMENT slot_count BY two
            END IF
        END FOR
        RETURN slot_count EQUALS zero
    END FUNCTION
END CLASS
