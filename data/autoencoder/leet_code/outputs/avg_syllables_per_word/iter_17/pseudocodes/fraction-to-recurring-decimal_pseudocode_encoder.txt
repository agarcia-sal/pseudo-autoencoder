CLASS Solution
    FUNCTION fractionToDecimal(numerator, denominator)
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero
            SET sign TO minus
        ELSE
            SET sign TO empty string
        END IF

        SET numerator TO the absolute value of numerator
        SET denominator TO the absolute value of denominator

        CALL divmod WITH numerator AND denominator RETURNING quotient AND remainder
        SET integer_part TO the string representation of quotient

        IF remainder EQUALS zero
            RETURN the concatenation of sign AND integer_part
        END IF

        SET seen_remainders TO empty mapping
        SET fractional_part TO empty list

        WHILE remainder NOT EQUALS zero
            IF remainder IS IN seen_remainders
                SET start TO the element at key remainder in seen_remainders
                SET non_repeating TO the concatenation of elements from position zero TO start MINUS one of fractional_part
                SET repeating TO the concatenation of elements from position start TO the end of fractional_part
                RETURN the concatenation of sign AND integer_part AND period AND non_repeating AND opening parenthesis AND repeating AND closing parenthesis
            END IF

            SET the element at key remainder in seen_remainders TO the LENGTH OF fractional_part

            SET remainder TO remainder MULTIPLIED BY ten
            CALL divmod WITH remainder AND denominator RETURNING quotient AND remainder
            APPEND the string representation of quotient TO fractional_part
        END WHILE

        RETURN the concatenation of sign AND integer_part AND period AND the concatenation of all elements in fractional_part
    END FUNCTION
END CLASS
