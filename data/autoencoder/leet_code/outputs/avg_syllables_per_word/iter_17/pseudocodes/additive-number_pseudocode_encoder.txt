CLASS Solution
    FUNCTION isAdditiveNumber(num)
        FUNCTION is_valid_sequence(start_position, first_number, second_number)
            IF start_position EQUALS the LENGTH OF num
                RETURN true
            END IF

            SET expected_string TO CALL convert_number_to_string WITH first_number PLUS second_number
            SET expected_length TO the LENGTH OF expected_string

            IF start_position PLUS expected_length IS GREATER THAN the LENGTH OF num OR
               substring from start_position TO start_position PLUS expected_length MINUS one OF num NOT EQUALS expected_string
                RETURN false
            END IF

            RETURN CALL is_valid_sequence WITH start_position PLUS expected_length, second_number, CONVERT expected_string TO integer
        END FUNCTION

        SET n TO the LENGTH OF num
        FOR i FROM one TO n DIVIDED BY two PLUS one
            FOR j FROM i PLUS one TO n MINUS one
                SET first_string TO substring from zero TO i MINUS one OF num
                SET second_string TO substring from i TO j MINUS one OF num

                IF the LENGTH OF first_string IS GREATER THAN one AND the element at position zero OF first_string EQUALS zero OR
                   the LENGTH OF second_string IS GREATER THAN one AND the element at position zero OF second_string EQUALS zero
                    CONTINUE TO next iteration
                END IF

                IF CALL is_valid_sequence WITH j, CONVERT first_string TO integer, CONVERT second_string TO integer
                    RETURN true
                END IF
            END FOR
        END FOR

        RETURN false
    END FUNCTION
END CLASS
