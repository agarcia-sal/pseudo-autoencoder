CLASS Solution
    FUNCTION isPossible(list_of_numbers)
        SET num_count TO CALL CountFrequencies WITH list_of_numbers
        SET end_with TO CALL InitializeCounter

        FOR each number IN list_of_numbers
            IF element at position number of num_count EQUALS zero
                CONTINUE
            END IF

            IF element at position number MINUS one of end_with GREATER THAN zero
                DECREMENT element at position number MINUS one of end_with BY one
                INCREMENT element at position number of end_with BY one
            ELSE IF element at position number PLUS one of num_count GREATER THAN zero AND element at position number PLUS two of num_count GREATER THAN zero
                DECREMENT element at position number PLUS one of num_count BY one
                DECREMENT element at position number PLUS two of num_count BY one
                INCREMENT element at position number PLUS two of end_with BY one
            ELSE
                RETURN false
            END IF

            DECREMENT element at position number of num_count BY one
        END FOR

        RETURN true
    END FUNCTION
END CLASS
