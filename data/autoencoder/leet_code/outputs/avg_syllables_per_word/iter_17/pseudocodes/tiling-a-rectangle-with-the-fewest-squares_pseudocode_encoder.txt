CLASS Solution
    FUNCTION tilingRectangle(n, m)
        FUNCTION backtrack(list_of_heights, number_of_moves)
            NONLOCAL min_moves
            IF all elements IN list_of_heights EQUAL TO n
                SET min_moves TO the MINIMUM OF min_moves AND number_of_moves
                RETURN
            END IF

            IF number_of_moves GREATER THAN OR EQUAL TO min_moves
                RETURN
            END IF

            SET minimum_height TO the MINIMUM VALUE IN list_of_heights
            SET left_position TO the POSITION OF minimum_height IN list_of_heights
            SET right_position TO left_position

            WHILE right_position LESS THAN m AND element at position right_position of list_of_heights EQUALS minimum_height
                INCREMENT right_position BY one
            END WHILE

            FOR size FROM right_position MINUS left_position DOWN TO one
                IF for each index i FROM zero TO size MINUS one the element at position left_position PLUS i of list_of_heights PLUS size LESS THAN OR EQUAL TO n
                    SET new_heights TO a COPY OF list_of_heights
                    FOR index i FROM zero TO size MINUS one
                        INCREMENT element at position left_position PLUS i of new_heights BY size
                    END FOR
                    CALL backtrack WITH new_heights AND number_of_moves PLUS one
                END IF
            END FOR
        END FUNCTION

        SET min_moves TO infinity
        CALL backtrack WITH a list of zero repeated m times AND zero
        RETURN min_moves
    END FUNCTION
END CLASS
