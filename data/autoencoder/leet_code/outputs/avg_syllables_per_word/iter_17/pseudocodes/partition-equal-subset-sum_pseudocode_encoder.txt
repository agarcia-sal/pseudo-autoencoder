CLASS Solution
    FUNCTION canPartition(list_of_numbers)
        SET total_sum TO the SUM OF list_of_numbers
        IF total_sum MODULO two NOT EQUALS zero
            RETURN False
        END IF
        SET target TO total_sum DIVIDED BY two
        SET n TO the LENGTH OF list_of_numbers
        SET dp TO a list of False REPEATED target PLUS one times
        SET element at position zero of dp TO True
        FOR each number IN list_of_numbers
            FOR index FROM target DOWN TO number
                SET element at position index of dp TO element at position index of dp OR element at position index MINUS number of dp
            END FOR
        END FOR
        RETURN element at position target of dp
    END FUNCTION
END CLASS
