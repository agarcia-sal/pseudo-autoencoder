CLASS Solution
    FUNCTION maxSatisfaction(satisfaction_list)
        CALL satisfaction_list SORT IN descending order
        SET max_satisfaction TO zero
        SET current_satisfaction TO zero
        SET total_satisfaction TO zero
        FOR each satisfaction_value IN satisfaction_list
            INCREMENT current_satisfaction BY satisfaction_value
            INCREMENT total_satisfaction BY current_satisfaction
            SET max_satisfaction TO the GREATER OF max_satisfaction AND total_satisfaction
        END FOR
        RETURN max_satisfaction
    END FUNCTION
END CLASS
