CLASS Solution
    FUNCTION findUnsortedSubarray(list_of_numbers)
        SET sorted_list TO CALL sorted WITH list_of_numbers
        SET start TO negative one
        SET end TO negative two
        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one
            IF element at position index of list_of_numbers NOT EQUALS element at position index of sorted_list
                IF start EQUALS negative one
                    SET start TO index
                END IF
                SET end TO index
            END IF
        END FOR
        RETURN end MINUS start PLUS one
    END FUNCTION
END CLASS
