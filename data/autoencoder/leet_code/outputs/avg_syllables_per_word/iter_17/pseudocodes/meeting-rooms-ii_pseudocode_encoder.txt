CLASS Solution
    FUNCTION minMeetingRooms(intervals OF LIST OF LIST OF INTEGER)
        IF intervals IS empty
            RETURN zero
        END IF

        CALL sort intervals BY element at position zero of each sublist

        SET min_heap TO an empty list

        FOR each meeting IN intervals
            IF min_heap IS NOT empty AND element at position zero of meeting IS GREATER THAN OR EQUAL TO element at position zero of min_heap
                CALL remove smallest element FROM min_heap
            END IF
            CALL add element at position one of meeting TO min_heap
        END FOR

        RETURN the LENGTH OF min_heap
    END FUNCTION
END CLASS
