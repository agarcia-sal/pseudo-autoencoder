CLASS Solution
    FUNCTION maxSubArrayLen(list_of_numbers, target_sum)
        SET dictionary_of_cumulative_sum_to_index TO a NEW dictionary WITH element zero MAPPED TO negative one
        SET cumulative_sum TO zero
        SET maximum_length TO zero

        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one
            SET current_number TO element at position index of list_of_numbers
            INCREMENT cumulative_sum BY current_number

            IF cumulative_sum MINUS target_sum IS IN dictionary_of_cumulative_sum_to_index
                SET candidate_length TO index MINUS element at position cumulative_sum MINUS target_sum OF dictionary_of_cumulative_sum_to_index
                SET maximum_length TO the GREATER OF maximum_length AND candidate_length
            END IF
            
            IF cumulative_sum IS NOT IN dictionary_of_cumulative_sum_to_index
                SET element at position cumulative_sum OF dictionary_of_cumulative_sum_to_index TO index
            END IF
        END FOR

        RETURN maximum_length
    END FUNCTION
END CLASS
