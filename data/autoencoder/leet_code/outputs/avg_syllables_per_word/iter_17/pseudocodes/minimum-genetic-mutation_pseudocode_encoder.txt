CLASS Solution
    FUNCTION minMutation(start, end, bank)
        SET set_of_bank_sequences TO CONVERT bank TO SET
        SET queue TO INITIALIZE DEQUE WITH PAIR OF start AND zero
        SET mutation_map TO MAPPING FROM A TO TCG AND T TO ACG AND C TO ATG AND G TO ATC
        WHILE queue is not empty
            SET current_sequence AND current_step TO REMOVE FIRST ELEMENT FROM queue
            IF current_sequence EQUALS end
                RETURN current_step
            END IF
            FOR each index AND character IN ENUMERATE current_sequence
                FOR each possible_mutation IN mutation_map AT KEY character
                    SET next_sequence TO CONCATENATE substring FROM POSITION zero TO index OF current_sequence PLUS possible_mutation PLUS substring FROM POSITION index PLUS one TO END OF current_sequence
                    IF next_sequence IS IN set_of_bank_sequences
                        APPEND PAIR OF next_sequence AND current_step PLUS one TO queue
                        REMOVE next_sequence FROM set_of_bank_sequences
                    END IF
                END FOR
            END FOR
        END WHILE
        RETURN negative one
    END FUNCTION
END CLASS
