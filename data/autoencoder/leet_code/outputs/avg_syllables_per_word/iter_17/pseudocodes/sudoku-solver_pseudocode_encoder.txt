CLASS Solution
    FUNCTION solveSudoku(board)
        FUNCTION is_valid(board, row, col, num)
            FOR index FROM zero TO eight
                IF element at position index of element at position row of board EQUALS num
                    RETURN False
                END IF
            END FOR
            FOR index FROM zero TO eight
                IF element at position row of element at position index of board EQUALS num
                    RETURN False
                END IF
            END FOR
            SET start_row TO three MULTIPLIED BY the INTEGER DIVISION OF row BY three
            SET start_col TO three MULTIPLIED BY the INTEGER DIVISION OF col BY three
            FOR i FROM zero TO two
                FOR j FROM zero TO two
                    IF element at position start_row PLUS i of element at position start_col PLUS j of board EQUALS num
                        RETURN False
                    END IF
                END FOR
            END FOR
            RETURN True
        END FUNCTION

        FUNCTION solve(board)
            FOR row FROM zero TO eight
                FOR col FROM zero TO eight
                    IF element at position row of element at position col of board EQUALS the dot character
                        FOR num FROM one TO nine
                            SET string_num TO the string representation of num
                            IF is_valid(board, row, col, string_num) EQUALS True
                                SET element at position row of element at position col of board TO string_num
                                IF solve(board) EQUALS True
                                    RETURN True
                                END IF
                                SET element at position row of element at position col of board TO the dot character
                            END IF
                        END FOR
                        RETURN False
                    END IF
                END FOR
            END FOR
            RETURN True
        END FUNCTION

        CALL solve WITH board
    END FUNCTION
END CLASS
