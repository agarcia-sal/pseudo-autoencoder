CLASS Solution
    FUNCTION generatePalindromes(s)
        FUNCTION dfs(t)
            IF the LENGTH OF t EQUALS the LENGTH OF s
                APPEND t TO ans
                RETURN
            END IF
            FOR each character AND its count IN the ITEMS OF cnt
                IF count GREATER THAN one
                    DECREMENT the VALUE AT key character OF cnt BY two
                    CALL dfs WITH parameter character CONCATENATED WITH t CONCATENATED WITH character
                    INCREMENT the VALUE AT key character OF cnt BY two
                END IF
            END FOR
        END FUNCTION

        SET cnt TO CALL Counter WITH parameter s
        SET mid TO empty string
        FOR each character AND its count IN the ITEMS OF cnt
            IF count MODULO two EQUALS one
                IF mid IS NOT empty string
                    RETURN empty list
                END IF
                SET mid TO character
                DECREMENT the VALUE AT key character OF cnt BY one
            END IF
        END FOR
        SET ans TO empty list
        CALL dfs WITH parameter mid
        RETURN ans
    END FUNCTION
END CLASS
