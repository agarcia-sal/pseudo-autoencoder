CLASS Solution
    FUNCTION largestOverlap(img1, img2)
        SET n TO the LENGTH OF img1
        SET points1 TO the RESULT OF CALLING generatePoints WITH img1 AND n
        SET points2 TO the RESULT OF CALLING generatePoints WITH img2 AND n

        SET overlap_count TO a new default dictionary of integers

        FOR each x1, y1 IN points1
            FOR each x2, y2 IN points2
                SET dx TO x2 MINUS x1
                SET dy TO y2 MINUS y1
                INCREMENT the VALUE at position key dx AND dy OF overlap_count BY one
            END FOR
        END FOR

        RETURN the MAXIMUM VALUE IN overlap_count OR zero IF overlap_count IS EMPTY
    END FUNCTION

    FUNCTION generatePoints(image, size)
        SET points_list TO an empty list
        FOR i FROM zero TO size MINUS one
            FOR j FROM zero TO size MINUS one
                IF the ELEMENT at position i OF image AND position j OF that element EQUALS one
                    APPEND the tuple (i, j) TO points_list
                END IF
            END FOR
        END FOR
        RETURN points_list
    END FUNCTION
END CLASS
