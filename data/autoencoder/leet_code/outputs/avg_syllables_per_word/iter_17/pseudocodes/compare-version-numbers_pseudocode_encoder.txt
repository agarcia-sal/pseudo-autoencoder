CLASS Solution
    FUNCTION compareVersion(version1, version2)
        SET v1_revisions TO CALL split_string_by_dot WITH version1
        SET v2_revisions TO CALL split_string_by_dot WITH version2
        SET max_length TO the MAXIMUM OF the LENGTH OF v1_revisions AND the LENGTH OF v2_revisions
        FOR index FROM zero TO max_length MINUS one
            IF index IS LESS THAN the LENGTH OF v1_revisions
                SET v1 TO CALL convert_string_to_integer WITH element at position index of v1_revisions
            ELSE
                SET v1 TO zero
            END IF
            IF index IS LESS THAN the LENGTH OF v2_revisions
                SET v2 TO CALL convert_string_to_integer WITH element at position index of v2_revisions
            ELSE
                SET v2 TO zero
            END IF
            IF v1 IS LESS THAN v2
                RETURN negative one
            ELSE IF v1 IS GREATER THAN v2
                RETURN one
            END IF
        END FOR
        RETURN zero
    END FUNCTION
END CLASS
