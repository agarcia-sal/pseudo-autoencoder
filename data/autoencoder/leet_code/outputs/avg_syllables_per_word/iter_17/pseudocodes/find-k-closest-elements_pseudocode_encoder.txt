CLASS Solution
    FUNCTION findClosestElements(array arr, integer k, integer x) RETURNS list_of_integers
        SET left_index TO the POSITION WHERE x WOULD BE INSERTED TO KEEP array arr SORTED MINUS one
        SET right_index TO left_index PLUS one
        WHILE right_index MINUS left_index MINUS one IS LESS THAN k HOLDS
            IF left_index EQUALS negative one OR right_index IS LESS THAN the LENGTH OF array arr AND the ABSOLUTE VALUE OF element at position right_index of array arr MINUS x IS LESS THAN the ABSOLUTE VALUE OF element at position left_index of array arr MINUS x
                INCREMENT right_index BY one
            ELSE
                DECREMENT left_index BY one
            END IF
        END WHILE
        RETURN the SUBLIST of array arr FROM position left_index PLUS one TO position right_index
    END FUNCTION
END CLASS
