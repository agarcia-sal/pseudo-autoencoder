CLASS Solution
    FUNCTION findMinArrowShots(list_of_points)
        CALL sort WITH key AS FUNCTION element
            RETURN element at position one of element
        END FUNCTION ON list_of_points
        SET number_of_arrows TO zero
        SET last_arrow_position TO negative infinity
        FOR each balloon IN list_of_points
            IF element at position zero of balloon GREATER THAN last_arrow_position
                INCREMENT number_of_arrows BY one
                SET last_arrow_position TO element at position one of balloon
            END IF
        END FOR
        RETURN number_of_arrows
    END FUNCTION
END CLASS
