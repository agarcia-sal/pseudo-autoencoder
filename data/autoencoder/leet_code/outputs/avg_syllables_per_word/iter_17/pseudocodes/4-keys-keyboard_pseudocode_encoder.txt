CLASS Solution
    FUNCTION maxA(number_of_key_presses)
        IF number_of_key_presses LESS THAN OR EQUAL TO three
            RETURN number_of_key_presses
        END IF

        CALL initialize_dp_array WITH parameter number_of_key_presses ASSIGN TO dp_array

        FOR index FROM one TO three
            SET element at position index of dp_array TO index
        END FOR

        FOR index FROM four TO number_of_key_presses
            FOR previous_position FROM index MINUS three DOWN TO one
                SET product TO element at position previous_position of dp_array MULTIPLIED BY index MINUS previous_position MINUS one
                IF product GREATER THAN element at position index of dp_array
                    SET element at position index of dp_array TO product
                END IF
            END FOR
        END FOR

        RETURN element at position number_of_key_presses of dp_array
    END FUNCTION

        // Auxiliary function to abstract initialization of dp array
    END FUNCTION
END CLASS
