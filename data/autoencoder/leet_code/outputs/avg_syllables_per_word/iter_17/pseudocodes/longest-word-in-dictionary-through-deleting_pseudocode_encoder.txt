CLASS Solution
    FUNCTION findLongestWord(s, dictionary)
        FUNCTION is_subsequence(word, s)
            ASSIGN iterator TO ITERATOR OF s
            RETURN all character IN iterator FOR each character IN word
        END FUNCTION

        CALL dictionary_sort_by_length_descending_and_lexicographical_order_ascending WITH dictionary

        FOR each word IN dictionary
            IF is_subsequence(word, s) IS true
                RETURN word
            END IF
        END FOR

        RETURN empty string
    END FUNCTION
END CLASS
