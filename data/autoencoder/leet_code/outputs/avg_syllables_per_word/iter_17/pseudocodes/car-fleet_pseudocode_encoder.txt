CLASS Solution
    FUNCTION carFleet(target, position, speed)
        SET paired_list TO CALL zip WITH position AND speed
        SET sorted_pairs TO CALL sorted WITH paired_list IN reverse order
        SET time_to_target TO empty list
        FOR each pair IN sorted_pairs
            SET position_element TO element at position zero of pair
            SET speed_element TO element at position one of pair
            SET time_value TO target MINUS position_element DIVIDED BY speed_element
            APPEND time_value TO time_to_target
        END FOR

        SET fleets TO zero
        SET current_max_time TO zero

        FOR each time IN time_to_target
            IF time IS GREATER THAN current_max_time
                INCREMENT fleets BY one
                SET current_max_time TO time
            END IF
        END FOR

        RETURN fleets
    END FUNCTION
END CLASS
