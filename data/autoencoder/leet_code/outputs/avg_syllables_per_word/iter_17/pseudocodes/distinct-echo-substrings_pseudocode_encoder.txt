CLASS Solution
    FUNCTION distinctEchoSubstrings(text)
        SET n TO the LENGTH OF text
        CALL auxiliary FUNCTION create_empty_set ASSIGN TO substrings
        
        FOR length FROM one TO n DIVIDED BY two PLUS one
            FOR start FROM zero TO n MINUS two MULTIPLIED BY length PLUS one
                CALL auxiliary FUNCTION substring_of_text FROM position start TO start PLUS length ASSIGN TO a
                CALL auxiliary FUNCTION substring_of_text FROM position start PLUS length TO start PLUS two MULTIPLIED BY length ASSIGN TO b
                IF a EQUALS b
                    CALL auxiliary FUNCTION add_element a CONCATENATED WITH b TO substrings
                END IF
            END FOR
        END FOR
        
        RETURN the SIZE OF substrings
    END FUNCTION
END CLASS
