    FUNCTION minMovesToMakePalindrome(s)
        SET character_list TO CALL ConvertStringToList WITH s
        SET length_of_list TO the LENGTH OF character_list
        SET total_moves TO zero

        WHILE the LENGTH OF character_list GREATER THAN one holds
            FOR index FROM the LENGTH OF character_list MINUS one TO one DECREMENTING BY one
                IF the element at position index of character_list EQUALS the element at position zero of character_list
                    FOR position FROM index TO the LENGTH OF character_list MINUS two
                        CALL SwapElementsAtPositions WITH character_list position AND position PLUS one
                        INCREMENT total_moves BY one
                    END FOR
                    CALL RemoveElementAtPosition WITH character_list zero
                    CALL RemoveElementAtPosition WITH character_list the LAST POSITION
                    BREAK
                END IF
            END FOR
            IF no BREAK occurred in the FOR loop
                CALL RemoveElementAtPosition WITH character_list zero
            END IF
        END WHILE

        RETURN total_moves
    END FUNCTION
