CLASS Solution
    FUNCTION strobogrammaticInRange(low, high)
        FUNCTION generate_strobogrammatic(n, final_length)
            IF n EQUALS zero
                RETURN list containing empty string
            END IF
            IF n EQUALS one
                RETURN list containing zero one and eight
            END IF
            SET middles TO CALL generate_strobogrammatic WITH parameters n MINUS two AND final_length
            SET result TO empty list
            FOR each middle IN middles
                IF n NOT EQUALS final_length
                    APPEND concatenation of zero and middle and zero TO result
                END IF
                APPEND concatenation of one and middle and one TO result
                APPEND concatenation of six and middle and nine TO result
                APPEND concatenation of eight and middle and eight TO result
                APPEND concatenation of nine and middle and six TO result
            END FOR
            RETURN result
        END FUNCTION

        FUNCTION count_strobogrammatic(low, high)
            SET count TO zero
            FOR length FROM the LENGTH OF low TO the LENGTH OF high
                FOR each num IN CALL generate_strobogrammatic WITH parameters length AND length
                    IF (length EQUALS the LENGTH OF low AND num LESS THAN low) OR (length EQUALS the LENGTH OF high AND num GREATER THAN high)
                        CONTINUE TO next iteration
                    END IF
                    INCREMENT count BY one
                END FOR
            END FOR
            RETURN count
        END FUNCTION

        RETURN CALL count_strobogrammatic WITH parameters low AND high
    END FUNCTION
END CLASS
