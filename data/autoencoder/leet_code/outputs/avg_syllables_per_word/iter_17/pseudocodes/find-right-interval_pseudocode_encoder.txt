CLASS Solution
    FUNCTION findRightInterval(intervals)
        SET start_times TO CALL create_sorted_start_times WITH intervals
        SET result TO CALL create_empty_list
        FOR each interval IN intervals
            SET end_position TO element at position one of interval
            SET idx TO CALL find_leftmost_index WITH start_times AND end_position
            IF idx IS LESS THAN the LENGTH OF start_times
                APPEND element at position two of element at position idx of start_times TO result
            ELSE
                APPEND negative one TO result
            END IF
        END FOR
        RETURN result
    END FUNCTION

    FUNCTION create_sorted_start_times(intervals)
        DESCRIBE AS auxiliary FUNCTION TO create sorted list of tuples start time and index FROM intervals
    END FUNCTION

    FUNCTION create_empty_list
        DESCRIBE AS auxiliary FUNCTION TO create empty list
    END FUNCTION

    FUNCTION find_leftmost_index(sorted_list, target_value)
        DESCRIBE AS auxiliary FUNCTION TO find leftmost position to insert target_value IN sorted_list
    END FUNCTION
END CLASS
