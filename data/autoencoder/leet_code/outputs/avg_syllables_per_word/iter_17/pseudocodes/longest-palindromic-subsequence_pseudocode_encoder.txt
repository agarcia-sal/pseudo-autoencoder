CLASS Solution
    FUNCTION longestPalindromeSubseq(string_s)
        SET n TO the LENGTH OF string_s
        SET dp TO A TWO DIMENSIONAL ARRAY OF zero WITH SIZE n BY n
        FOR index_i FROM zero TO n MINUS one
            SET element at position index_i AND index_i OF dp TO one
        END FOR
        FOR length FROM two TO n
            FOR index_i FROM zero TO n MINUS length
                SET index_j TO index_i PLUS length MINUS one
                IF element at position index_i of string_s EQUALS element at position index_j of string_s
                    SET element at position index_i AND index_j OF dp TO element at position index_i PLUS one AND index_j MINUS one OF dp PLUS two
                ELSE
                    SET element at position index_i AND index_j OF dp TO THE MAXIMUM OF element at position index_i PLUS one AND index_j OF dp AND element at position index_i AND index_j MINUS one OF dp
                END IF
            END FOR
        END FOR
        RETURN element at position zero AND n MINUS one OF dp
    END FUNCTION
END CLASS
