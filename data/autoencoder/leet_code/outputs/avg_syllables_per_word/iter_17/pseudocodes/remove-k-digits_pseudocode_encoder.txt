CLASS Solution
    FUNCTION removeKdigits(num_string, k_integer)
        SET stack_list TO empty list

        FOR each digit_character IN num_string
            WHILE k_integer GREATER THAN zero AND stack_list IS NOT empty AND element at position last of stack_list GREATER THAN digit_character
                CALL function pop ON stack_list
                DECREMENT k_integer BY one
            END WHILE
            APPEND digit_character TO stack_list
        END FOR

        IF k_integer EQUALS zero
            SET final_stack_list TO stack_list
        ELSE
            SET final_stack_list TO substring of stack_list from position zero TO position negative k_integer exclusive
        END IF

        SET result_string TO concatenation of elements in final_stack_list converted to string with all leading zeros removed

        IF result_string IS NOT empty
            RETURN result_string
        ELSE
            RETURN zero
        END IF
    END FUNCTION
END CLASS
