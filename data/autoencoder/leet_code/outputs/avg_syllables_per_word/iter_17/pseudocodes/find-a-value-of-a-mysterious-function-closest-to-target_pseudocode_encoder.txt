CLASS Solution
    FUNCTION closestToTarget(arr, target)
        SET seen TO an empty set
        SET min_diff TO positive infinity
        FOR each number IN arr
            SET temp_set TO an empty set
            FOR each element IN seen
                APPEND bitwise AND of number AND element TO temp_set
            END FOR
            SET seen TO the UNION of temp_set AND a set containing number
            FOR each value IN seen
                SET current_difference TO the absolute value of value MINUS target
                IF current_difference LESS THAN min_diff
                    SET min_diff TO current_difference
                END IF
            END FOR
        END FOR
        RETURN min_diff
    END FUNCTION
END CLASS
