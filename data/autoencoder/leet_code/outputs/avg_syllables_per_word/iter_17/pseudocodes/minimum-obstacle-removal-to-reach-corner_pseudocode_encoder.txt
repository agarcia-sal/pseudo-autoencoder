CLASS Solution
    FUNCTION minimumObstacles(grid)
        SET number_of_rows TO the LENGTH OF grid
        SET number_of_columns TO the LENGTH OF element at position zero of grid
        SET directions TO list containing
            tuple of zero AND one
            tuple of one AND zero
            tuple of zero AND negative one
            tuple of negative one AND zero
        SET min_heap TO list containing tuple of zero zero zero
        SET visited TO set containing tuple of zero zero
        
        WHILE min_heap is not empty
            CALL heapq.heappop WITH min_heap AND ASSIGN TO cost AND x AND y
            
            IF tuple of x AND y EQUALS tuple of number_of_rows MINUS one AND number_of_columns MINUS one
                RETURN cost
            END IF
            
            FOR each dx AND dy IN directions
                SET nx TO x PLUS dx
                SET ny TO y PLUS dy
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN number_of_rows AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN number_of_columns AND tuple of nx AND ny IS NOT IN visited
                    ADD tuple of nx AND ny TO visited
                    SET new_cost TO cost PLUS element at position nx of grid AND element at position ny of that element
                    CALL heapq.heappush WITH min_heap AND tuple of new_cost AND nx AND ny
                END IF
            END FOR
        END WHILE
        
        RETURN negative one
    END FUNCTION
END CLASS
