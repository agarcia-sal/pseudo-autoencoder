CLASS Solution
    FUNCTION topKFrequent(words, k)
        SET count TO CALL Counter WITH words
        SET heap TO an empty list
        FOR each word AND frequency IN the ITEMS OF count
            APPEND a tuple of NEGATIVE frequency AND word TO heap
        END FOR
        CALL heapify WITH heap
        SET result TO an empty list
        FOR index FROM one TO k
            SET top_element TO CALL heappop WITH heap
            APPEND the element at position two of top_element TO result
        END FOR
        RETURN result
    END FUNCTION
END CLASS
