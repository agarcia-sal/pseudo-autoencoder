CLASS Solution
    FUNCTION myPow(x AS floating_point_number, n AS integer)
        FUNCTION qpow(a AS floating_point_number, n AS integer)
            SET ans TO one
            WHILE n IS NOT equal TO zero
                IF the bitwise AND of n AND one IS equal TO one
                    SET ans TO ans MULTIPLIED BY a
                END IF
                SET a TO a MULTIPLIED BY a
                SET n TO n DIVIDED BY two USING BITWISE RIGHT SHIFT
            END WHILE
            RETURN ans
        END FUNCTION
        IF n IS GREATER THAN OR EQUAL TO zero
            RETURN CALL qpow WITH parameters a equal TO x AND n equal TO n
        ELSE
            RETURN one DIVIDED BY CALL qpow WITH parameters a equal TO x AND n equal TO the NEGATION OF n
        END IF
    END FUNCTION
END CLASS
