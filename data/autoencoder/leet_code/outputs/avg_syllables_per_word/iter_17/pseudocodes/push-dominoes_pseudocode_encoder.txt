CLASS Solution
    FUNCTION pushDominoes(dominoes_string)
        SET string_length TO the LENGTH OF dominoes_string
        SET forces_list TO a LIST OF zero repeated string_length times

        SET force_value TO zero
        FOR index FROM zero TO string_length MINUS one
            IF the element at position index of dominoes_string EQUALS letter R
                SET force_value TO string_length
            ELSE IF the element at position index of dominoes_string EQUALS letter L
                SET force_value TO zero
            ELSE
                SET force_value TO the GREATER VALUE BETWEEN force_value MINUS one AND zero
            END IF
            SET element at position index of forces_list TO the element at position index of forces_list PLUS force_value
        END FOR

        SET force_value TO zero
        FOR index FROM string_length MINUS one DOWN TO zero
            IF the element at position index of dominoes_string EQUALS letter L
                SET force_value TO string_length
            ELSE IF the element at position index of dominoes_string EQUALS letter R
                SET force_value TO zero
            ELSE
                SET force_value TO the GREATER VALUE BETWEEN force_value MINUS one AND zero
            END IF
            SET element at position index of forces_list TO the element at position index of forces_list MINUS force_value
        END FOR

        SET result_list TO an empty LIST
        FOR each force IN forces_list
            IF force GREATER THAN zero
                APPEND letter R TO result_list
            ELSE IF force LESS THAN zero
                APPEND letter L TO result_list
            ELSE
                APPEND the dot symbol TO result_list
            END IF
        END FOR
        
        RETURN the JOINED STRING OF result_list
    END FUNCTION
END CLASS
