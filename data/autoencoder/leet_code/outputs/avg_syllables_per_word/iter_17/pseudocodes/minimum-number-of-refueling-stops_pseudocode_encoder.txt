CLASS Solution
    FUNCTION minRefuelStops(target, startFuel, stations)
        SET max_heap TO an empty list
        SET current_fuel TO startFuel
        SET stops TO zero
        CALL AppendStationToList WITH list stations and element a list containing target and zero

        FOR each pair position AND fuel IN stations
            WHILE max_heap IS NOT empty AND current_fuel IS LESS THAN position
                SET largest_fuel TO the NEGATIVE OF the result of Calling heap pop operation ON max_heap
                INCREMENT current_fuel BY largest_fuel
                INCREMENT stops BY one
            END WHILE
            IF current_fuel IS LESS THAN position
                RETURN the NEGATIVE OF one
            END IF
            CALL heap push operation ON max_heap WITH the NEGATIVE OF fuel
        END FOR

        RETURN stops
    END FUNCTION
END CLASS
