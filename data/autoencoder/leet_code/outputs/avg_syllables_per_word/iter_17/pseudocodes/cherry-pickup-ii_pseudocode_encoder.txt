CLASS Solution
    FUNCTION cherryPickup(grid)
        SET rows TO the LENGTH OF grid
        SET cols TO the LENGTH OF element at position zero of grid
        
        FUNCTION dp(r, c1, c2)
            IF c1 LESS THAN zero OR c1 GREATER THAN OR EQUAL TO cols OR c2 LESS THAN zero OR c2 GREATER THAN OR EQUAL TO cols
                RETURN zero
            END IF
            
            IF c1 EQUALS c2
                SET cherries TO element at position c1 of element at position r of grid
            ELSE
                SET cherries TO element at position c1 of element at position r of grid PLUS element at position c2 of element at position r of grid
            END IF
            
            IF r EQUALS rows MINUS one
                RETURN cherries
            END IF
            
            SET max_cherries TO zero
            FOR each dc1 IN list of negative one zero one
                FOR each dc2 IN list of negative one zero one
                    SET candidate TO CALL dp WITH r PLUS one COMMA c1 PLUS dc1 COMMA c2 PLUS dc2
                    IF candidate GREATER THAN max_cherries
                        SET max_cherries TO candidate
                    END IF
                END FOR
            END FOR
            
            RETURN cherries PLUS max_cherries
        END FUNCTION
        
        RETURN CALL dp WITH zero COMMA zero COMMA cols MINUS one
    END FUNCTION
END CLASS
