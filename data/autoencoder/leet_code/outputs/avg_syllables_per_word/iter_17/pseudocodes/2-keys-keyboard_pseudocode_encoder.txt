CLASS Solution
    FUNCTION minSteps(n)
        IF n EQUALS one
            RETURN zero
        END IF
        SET operations TO zero
        SET factor TO two
        WHILE n GREATER THAN one holds
            WHILE the remainder of n DIVIDED BY factor EQUALS zero holds
                INCREMENT operations BY factor
                SET n TO n DIVIDED BY factor ignoring remainder
            END WHILE
            INCREMENT factor BY one
        END WHILE
        RETURN operations
    END FUNCTION
END CLASS
