CLASS Solution
    FUNCTION findPermutation(string_s)
        SET result TO CALL InitializeResultArray WITH string_s
        SET start TO zero
        FOR end FROM zero TO the LENGTH OF string_s INCLUSIVE
            IF end EQUALS the LENGTH OF string_s OR element at position end of string_s EQUALS character I
                CALL ReverseSegment WITH parameters result start end
                SET start TO end PLUS one
            END IF
        END FOR
        RETURN result
    END FUNCTION

    FUNCTION InitializeResultArray(string_s)
        CREATE list_of_integers AS an empty list
        FOR index FROM zero TO the LENGTH OF string_s INCLUSIVE
            APPEND index PLUS one TO list_of_integers
        END FOR
        RETURN list_of_integers
    END FUNCTION

    FUNCTION ReverseSegment(list_values start_index end_index)
        SET segment_length TO end_index MINUS start_index PLUS one
        CREATE temporary_list AS an empty list
        FOR position FROM zero TO segment_length MINUS one
            APPEND element at position end_index MINUS position of list_values TO temporary_list
        END FOR
        FOR position FROM zero TO segment_length MINUS one
            SET element at position start_index PLUS position of list_values TO element at position position of temporary_list
        END FOR
    END FUNCTION
END CLASS
