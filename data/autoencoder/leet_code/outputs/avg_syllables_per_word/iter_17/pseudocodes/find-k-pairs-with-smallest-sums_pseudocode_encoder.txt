CLASS Solution
    FUNCTION kSmallestPairs(nums1, nums2, k)
        IF nums1 IS empty OR nums2 IS empty OR k LESS THAN OR EQUAL TO zero
            RETURN empty list
        END IF

        SET min_heap TO empty list

        FOR index FROM zero TO the smaller value BETWEEN k MINUS one AND the LENGTH OF nums1 MINUS one
            CALL heap push FUNCTION WITH min_heap AND the tuple OF the element at position index of nums1 PLUS the element at position zero of nums2, index, zero
        END FOR

        SET result TO empty list

        WHILE min_heap IS NOT empty AND the LENGTH OF result IS LESS THAN k
            CALL heap pop FUNCTION WITH min_heap AND ASSIGN the second and third elements of the returned tuple TO i AND j
            APPEND the list OF the element at position i of nums1 AND the element at position j of nums2 TO result

            IF j PLUS one IS LESS THAN the LENGTH OF nums2
                CALL heap push FUNCTION WITH min_heap AND the tuple OF the element at position i of nums1 PLUS the element at position j PLUS one of nums2, i, j PLUS one
            END IF
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
