CLASS Solution
    FUNCTION maxSlidingWindow(nums, k)
        IF nums IS empty OR k EQUALS zero
            RETURN empty list
        END IF
        SET deq TO empty double ended queue
        SET result TO empty list
        FOR i FROM zero TO the LENGTH OF nums MINUS one
            IF deq IS NOT empty AND element at position zero of deq LESS THAN i MINUS k PLUS one
                REMOVE element at position zero FROM deq
            END IF
            WHILE deq IS NOT empty AND element at position element at position last of deq of nums LESS THAN element at position i of nums
                REMOVE element at position last FROM deq
            END WHILE
            APPEND i TO deq
            IF i GREATER THAN OR EQUAL TO k MINUS one
                APPEND element at position element at position zero of deq of nums TO result
            END IF
        END FOR
        RETURN result
    END FUNCTION
END CLASS
