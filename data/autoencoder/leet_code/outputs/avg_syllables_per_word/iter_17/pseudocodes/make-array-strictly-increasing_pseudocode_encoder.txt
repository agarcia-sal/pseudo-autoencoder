CLASS Solution
    FUNCTION makeArrayIncreasing(arr1, arr2)
        CALL arr2 SORTED IN PLACE
        SET dp TO a MAPPING FROM negative one TO zero
        FOR each num IN arr1
            SET new_dp TO an EMPTY MAPPING
            FOR each last_num AND ops IN dp ITEMS
                IF num GREATER THAN last_num
                    SET current_min_operations TO the VALUE ASSOCIATED WITH num IN new_dp IF PRESENT OTHERWISE infinity
                    SET updated_min_operations TO the MINIMUM OF current_min_operations AND ops
                    ASSIGN num KEY IN new_dp TO updated_min_operations
                END IF
                SET idx TO CALL bisect_right WITH arr2 AND last_num
                IF idx LESS THAN the LENGTH OF arr2
                    SET element_to_replace WITH TO element AT position idx OF arr2
                    SET current_min_operations TO the VALUE ASSOCIATED WITH element_to_replace WITHIN new_dp IF PRESENT OTHERWISE infinity
                    SET updated_min_operations TO the MINIMUM OF current_min_operations AND ops PLUS one
                    ASSIGN element_to_replace KEY IN new_dp TO updated_min_operations
                END IF
            END FOR
            IF new_dp IS EMPTY
                RETURN negative one
            END IF
            ASSIGN dp TO new_dp
        END FOR
        RETURN the MINIMUM OF all VALUES IN dp
    END FUNCTION
END CLASS
