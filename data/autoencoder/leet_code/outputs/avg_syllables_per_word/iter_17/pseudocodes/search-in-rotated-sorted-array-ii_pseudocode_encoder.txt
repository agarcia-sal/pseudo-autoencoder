CLASS Solution
    FUNCTION search(list_of_numbers, target_value)
        IF the list_of_numbers IS empty
            RETURN false
        END IF

        SET left_index TO zero
        SET right_index TO the LENGTH OF list_of_numbers MINUS one

        WHILE left_index LESS THAN OR EQUAL TO right_index holds
            SET middle_index TO left_index PLUS the DIFFERENCE BETWEEN right_index AND left_index DIVIDED BY two

            IF element at position middle_index of list_of_numbers EQUALS target_value
                RETURN true
            END IF

            IF element at position left_index of list_of_numbers EQUALS element at position middle_index of list_of_numbers AND element at position middle_index of list_of_numbers EQUALS element at position right_index of list_of_numbers
                INCREMENT left_index BY one
                DECREMENT right_index BY one
            ELSE IF element at position left_index of list_of_numbers LESS THAN OR EQUAL TO element at position middle_index of list_of_numbers
                IF element at position left_index of list_of_numbers LESS THAN OR EQUAL TO target_value AND target_value LESS THAN element at position middle_index of list_of_numbers
                    SET right_index TO middle_index MINUS one
                ELSE
                    SET left_index TO middle_index PLUS one
                END IF
            ELSE
                IF element at position middle_index of list_of_numbers LESS THAN target_value AND target_value LESS THAN OR EQUAL TO element at position right_index of list_of_numbers
                    SET left_index TO middle_index PLUS one
                ELSE
                    SET right_index TO middle_index MINUS one
                END IF
            END IF
        END WHILE

        RETURN false
    END FUNCTION
END CLASS
