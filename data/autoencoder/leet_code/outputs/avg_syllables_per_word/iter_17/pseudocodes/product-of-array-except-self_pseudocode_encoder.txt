CLASS Solution
    FUNCTION productExceptSelf(list_of_numbers)
        SET length_of_numbers TO the LENGTH OF list_of_numbers
        SET answer_list TO a list containing one repeated length_of_numbers times
        
        SET left_product TO one
        FOR index FROM zero TO length_of_numbers MINUS one
            SET element at position index of answer_list TO left_product
            SET left_product TO left_product MULTIPLIED BY element at position index of list_of_numbers
        END FOR
        
        SET right_product TO one
        FOR index FROM length_of_numbers MINUS one DOWNTO zero
            SET element at position index of answer_list TO element at position index of answer_list MULTIPLIED BY right_product
            SET right_product TO right_product MULTIPLIED BY element at position index of list_of_numbers
        END FOR
        
        RETURN answer_list
    END FUNCTION
END CLASS
