CLASS Solution
    FUNCTION findLUSlength(list_of_strings)
        FUNCTION is_subsequence(string_s, string_t)
            ASSIGN iterator_to_t TO CALL iterator WITH string_t
            RETURN CALL all WITH condition character IS IN iterator_to_t FOR each character IN string_s
        END FUNCTION
        
        CALL sort ON list_of_strings WITH custom_key THAT SORTS BY NEGATIVE OF the LENGTH OF element AND element ITSELF

        FOR index_i FROM zero TO the LENGTH OF list_of_strings MINUS one
            SET word TO element at position index_i of list_of_strings
            SET unique TO true
            FOR index_j FROM zero TO the LENGTH OF list_of_strings MINUS one
                SET other TO element at position index_j of list_of_strings
                IF index_i IS NOT EQUALS index_j AND CALL is_subsequence WITH string_s as word AND string_t as other HOLDS
                    SET unique TO false
                    BREAK out of the loop
                END IF
            END FOR
            IF unique HOLDS
                RETURN the LENGTH OF word
            END IF
        END FOR

        RETURN negative one
    END FUNCTION
END CLASS
