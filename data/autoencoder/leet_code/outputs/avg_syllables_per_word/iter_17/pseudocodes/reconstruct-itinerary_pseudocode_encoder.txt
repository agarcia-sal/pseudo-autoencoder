CLASS Solution
    FUNCTION findItinerary(list_of_tickets)
        SET graph TO A DEFAULT DICTIONARY OF lists
        FOR each pair FROM position zero TO the LENGTH OF list_of_tickets MINUS one IN reversed sorted order of list_of_tickets
            SET source_airport TO element at position zero of pair
            SET destination_airport TO element at position one of pair
            APPEND destination_airport TO list AT key source_airport IN graph
        END FOR

        SET itinerary TO an empty list

        FUNCTION visit(airport)
            WHILE the list AT key airport IN graph IS NOT empty
                CALL visit WITH parameter element at last position of the list AT key airport IN graph AND REMOVE THAT element FROM THAT list
            END WHILE
            APPEND airport TO itinerary
        END FUNCTION

        CALL visit WITH parameter literal JFK

        RETURN itinerary AS reversed
    END FUNCTION
END CLASS
