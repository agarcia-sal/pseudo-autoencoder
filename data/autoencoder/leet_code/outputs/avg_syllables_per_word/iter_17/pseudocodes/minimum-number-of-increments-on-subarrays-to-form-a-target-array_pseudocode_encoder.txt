CLASS Solution
    FUNCTION minNumberOperations(list_of_target_numbers)
        SET total_operations TO zero
        SET previous_number TO zero
        FOR each current_number IN list_of_target_numbers
            IF current_number GREATER THAN previous_number
                INCREMENT total_operations BY current_number MINUS previous_number
            END IF
            SET previous_number TO current_number
        END FOR
        RETURN total_operations
    END FUNCTION
END CLASS
