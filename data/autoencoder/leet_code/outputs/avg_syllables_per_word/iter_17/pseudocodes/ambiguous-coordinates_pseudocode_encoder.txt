CLASS Solution
    FUNCTION ambiguousCoordinates(s)
        FUNCTION make(frag)
            SET n TO the LENGTH OF frag
            FOR d FROM one TO n
                SET left TO the SUBSTRING from position one TO d of frag
                SET right TO the SUBSTRING from position d PLUS one TO n of frag
                IF (left DOES NOT START WITH zero OR left EQUALS zero) AND (right DOES NOT END WITH zero)
                    RETURN the CONCATENATION of left PLUS IF d DOES NOT EQUAL n THEN the FULL STOP PLUS right ELSE right
                END IF
            END FOR
        END FUNCTION

        SET s TO the SUBSTRING from position two TO the LENGTH OF s MINUS one of s
        RETURN LIST constructed FROM
            FOR i FROM one TO the LENGTH OF s MINUS one
                FOR each a IN make(the SUBSTRING from position one TO i of s)
                    FOR each b IN make(the SUBSTRING from position i PLUS one TO the LENGTH OF s of s)
                        CREATE the STRING formed by LEFT PARENTHESIS PLUS a PLUS COMMA PLUS SPACE PLUS b PLUS RIGHT PARENTHESIS
                    END FOR
                END FOR
            END FOR
    END FUNCTION
END CLASS
