CLASS Solution
    FUNCTION minimumTime(s)
        SET n TO the LENGTH OF s
        CALL initializeListWithZeroes WITH parameters n ASSIGN TO left_costs
        CALL initializeListWithZeroes WITH parameters n ASSIGN TO right_costs

        IF element at position zero of s EQUALS one
            SET element at position zero of left_costs TO one
        ELSE
            SET element at position zero of left_costs TO zero
        END IF

        FOR index FROM one TO n MINUS one
            IF element at position index of s EQUALS zero
                SET element at position index of left_costs TO element at position index MINUS one of left_costs
            ELSE
                SET element at position index of left_costs TO the MINIMUM OF index PLUS one AND element at position index MINUS one of left_costs PLUS two
            END IF
        END FOR

        IF element at position n MINUS one of s EQUALS one
            SET element at position n MINUS one of right_costs TO one
        ELSE
            SET element at position n MINUS one of right_costs TO zero
        END IF

        FOR index FROM n MINUS two DOWN TO zero
            IF element at position index of s EQUALS zero
                SET element at position index of right_costs TO element at position index PLUS one of right_costs
            ELSE
                SET element at position index of right_costs TO the MINIMUM OF n MINUS index AND element at position index PLUS one of right_costs PLUS two
            END IF
        END FOR

        SET min_time TO the MINIMUM OF element at position n MINUS one of left_costs AND element at position zero of right_costs

        FOR index FROM zero TO n MINUS two
            SET min_time TO the MINIMUM OF min_time AND element at position index of left_costs PLUS element at position index PLUS one of right_costs
        END FOR

        RETURN min_time
    END FUNCTION
END CLASS
