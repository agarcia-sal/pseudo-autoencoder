CLASS Solution
    FUNCTION maximumGood(statements AS list of list of integers)
        SET n TO the LENGTH OF statements
        SET max_good TO zero

        FOR i FROM zero TO one SHIFTED LEFT BY n MINUS one
            SET good TO a list of n False elements
            SET bad TO a list of n False elements

            FOR j FROM zero TO n MINUS one
                IF i BITWISE AND one SHIFTED LEFT BY j IS NOT EQUAL TO zero
                    SET element at position j of good TO True
                ELSE
                    SET element at position j of bad TO True
                END IF
            END FOR

            SET valid TO True
            FOR j FROM zero TO n MINUS one
                IF element at position j of good IS True
                    FOR k FROM zero TO n MINUS one
                        IF element at position k of good IS True AND element at position k of element at position j of statements IS EQUALS zero 
                            OR element at position k of bad IS True AND element at position k of element at position j of statements IS EQUALS one
                            SET valid TO False
                            BREAK
                        END IF
                    END FOR
                    IF valid IS False
                        BREAK
                    END IF
                END IF
            END FOR

            IF valid IS True
                SET max_good TO the GREATER OF max_good AND the SUM OF values in good
            END IF
        END FOR

        RETURN max_good
    END FUNCTION
END CLASS
