CLASS TreeNode
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(values AS list)
    IF values IS EMPTY
        RETURN none
    END IF
    SET root TO CALL TreeNode WITH element at position zero of values
    SET i TO one
    SET queue TO CALL deque WITH no parameters
    CALL APPEND root TO queue
    WHILE queue IS NOT EMPTY
        SET node TO CALL popleft ON queue
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none
            SET node.left TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.left TO queue
        END IF
        INCREMENT i BY one
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none
            SET node.right TO CALL TreeNode WITH element at position i of values
            CALL APPEND node.right TO queue
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS none AND q IS none
        RETURN true
    ELSE IF p IS none OR q IS none
        RETURN false
    ELSE IF p.val NOT EQUALS q.val
        RETURN false
    ELSE
        RETURN CALL is_same_tree WITH p.left, q.left AND CALL is_same_tree WITH p.right, q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION insertIntoBST(root AS Optional TreeNode, val AS integer) RETURNS Optional TreeNode
        FUNCTION dfs(root)
            IF root IS none
                RETURN CALL TreeNode WITH val
            END IF
            IF root.val LESS THAN val
                SET root.right TO CALL dfs WITH root.right
            ELSE
                SET root.left TO CALL dfs WITH root.left
            END IF
            RETURN root
        END FUNCTION
        RETURN CALL dfs WITH root
    END FUNCTION
END CLASS
