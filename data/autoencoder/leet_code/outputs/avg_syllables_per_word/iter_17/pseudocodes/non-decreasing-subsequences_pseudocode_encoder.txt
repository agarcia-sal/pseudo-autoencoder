CLASS Solution
    FUNCTION findSubsequences(list_of_numbers)
        FUNCTION dfs(index_position, last_number, temp_list)
            IF index_position EQUALS the LENGTH OF list_of_numbers
                IF the LENGTH OF temp_list GREATER THAN one
                    APPEND a COPY OF temp_list TO answer_list
                END IF
                RETURN
            END IF
            IF element at position index_position of list_of_numbers GREATER THAN OR EQUAL TO last_number
                APPEND element at position index_position of list_of_numbers TO temp_list
                CALL dfs WITH parameters index_position PLUS one, element at position index_position of list_of_numbers, temp_list
                REMOVE the last element FROM temp_list
            END IF
            IF element at position index_position of list_of_numbers NOT EQUALS last_number
                CALL dfs WITH parameters index_position PLUS one, last_number, temp_list
            END IF
        END FUNCTION
        SET answer_list TO empty list
        CALL dfs WITH parameters zero, negative one thousand, empty list
        RETURN answer_list
    END FUNCTION
END CLASS
