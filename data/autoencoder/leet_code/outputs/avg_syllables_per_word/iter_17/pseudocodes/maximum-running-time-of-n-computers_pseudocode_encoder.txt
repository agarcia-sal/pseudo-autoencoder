CLASS Solution
    FUNCTION maxRunTime(number_of_computers, list_of_battery_capacities)
        FUNCTION canRunFor(proposed_time)
            SET total_runtime TO zero
            FOR each battery_capacity IN list_of_battery_capacities
                INCREMENT total_runtime BY the MINIMUM OF battery_capacity AND proposed_time
            END FOR
            RETURN total_runtime GREATER THAN OR EQUAL TO proposed_time MULTIPLIED BY number_of_computers
        END FUNCTION

        SET left_boundary TO zero
        SET right_boundary TO the SUM OF list_of_battery_capacities DIVIDED BY number_of_computers

        WHILE left_boundary LESS THAN right_boundary holds
            SET middle_point TO the INTEGER DIVISION OF left_boundary PLUS right_boundary PLUS one BY two
            IF canRunFor(middle_point) IS true
                SET left_boundary TO middle_point
            ELSE
                SET right_boundary TO middle_point MINUS one
            END IF
        END WHILE

        RETURN left_boundary
    END FUNCTION
END CLASS
