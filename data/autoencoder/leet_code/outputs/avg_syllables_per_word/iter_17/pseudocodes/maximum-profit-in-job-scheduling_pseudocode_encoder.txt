CLASS Solution
    FUNCTION jobScheduling(startTime, endTime, profit)
        SET jobs TO CALL SortJobsByEndTime WITH startTime, endTime, profit
        SET end_times TO CALL ExtractEndTimes FROM jobs
        SET dp TO CALL InitializeDPArray WITH the LENGTH OF jobs PLUS one
        FOR index FROM one TO the LENGTH OF jobs
            SET current_job_start TO the element at position index MINUS one OF jobs AT position zero
            SET j TO CALL FindPositionRight WITH end_times AND current_job_start
            SET dp AT position index TO THE MAXIMUM OF dp AT position index MINUS one AND dp AT position j PLUS the element at position index MINUS one OF jobs AT position two
        END FOR
        RETURN the element at the last position OF dp
    END FUNCTION

FUNCTION SortJobsByEndTime WITH startTime, endTime, profit
    IMPLEMENT auxiliary sorting and zipping of startTime, endTime, profit BY endTime
END FUNCTION

FUNCTION ExtractEndTimes FROM jobs
    IMPLEMENT auxiliary extraction of the element at position one FROM each job IN jobs
END FUNCTION

FUNCTION InitializeDPArray WITH length
    IMPLEMENT auxiliary creation of a list OF zeros OF size length
END FUNCTION

FUNCTION FindPositionRight WITH sorted_list, value
    IMPLEMENT auxiliary binary search TO find insertion position TO RIGHT OF value IN sorted_list
END FUNCTION
END CLASS
