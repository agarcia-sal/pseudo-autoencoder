CLASS Solution
    FUNCTION combinationSum3(k integer, n integer)
        FUNCTION backtrack(start integer, path list_of_integers, target integer, k integer)
            IF target EQUALS zero AND k EQUALS zero
                APPEND a copy of path TO result
                RETURN
            END IF
            IF target LESS THAN zero OR k LESS THAN zero
                RETURN
            END IF
            FOR i FROM start TO nine
                APPEND i TO path
                CALL backtrack WITH parameters i PLUS one, path, target MINUS i, k MINUS one
                REMOVE last element FROM path
            END FOR
        END FUNCTION
        SET result TO an empty list
        CALL backtrack WITH parameters one, an empty list, n, k
        RETURN result
    END FUNCTION
END CLASS
