CLASS Solution
    FUNCTION champagneTower(poured, query_row, query_glass)
        SET current_row TO list containing poured
        FOR row FROM one TO query_row
            SET next_row TO list containing zero repeated row PLUS one times
            FOR index FROM zero TO the LENGTH OF current_row MINUS one
                IF element at position index of current_row GREATER THAN one
                    SET excess TO element at position index of current_row MINUS one DIVIDED BY two
                    INCREMENT element at position index of next_row BY excess
                    INCREMENT element at position index PLUS one of next_row BY excess
                END IF
            END FOR
            SET current_row TO next_row
        END FOR
        RETURN the MINIMUM OF element at position query_glass of current_row AND one
    END FUNCTION
END CLASS
