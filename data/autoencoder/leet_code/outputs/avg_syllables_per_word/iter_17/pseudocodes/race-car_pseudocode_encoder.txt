CLASS Solution
    FUNCTION racecar(target_integer)
        CALL initialize_queue_with_starting_state
        SET queue TO the_initialized_queue
        SET visited TO a set CONTAINING a tuple of zero and one
        
        WHILE the queue holds elements
            CALL remove_first_element_from_queue
            SET position TO the first element of the removed tuple
            SET speed TO the second element of the removed tuple
            SET steps TO the third element of the removed tuple
            
            IF position EQUALS target_integer
                RETURN steps
            END IF
            
            SET next_position TO position PLUS speed
            SET next_speed TO speed MULTIPLIED BY two
            IF tuple of next_position and next_speed IS NOT IN visited AND next_position IS GREATER THAN OR EQUAL TO zero AND next_position IS LESS THAN two MULTIPLIED BY target_integer
                ADD tuple of next_position and next_speed TO visited
                APPEND tuple of next_position, next_speed, and steps PLUS one TO queue
            END IF
            
            IF speed IS GREATER THAN zero
                SET next_speed TO negative one
            ELSE
                SET next_speed TO one
            END IF
            IF tuple of position and next_speed IS NOT IN visited AND position IS GREATER THAN OR EQUAL TO zero AND position IS LESS THAN two MULTIPLIED BY target_integer
                ADD tuple of position and next_speed TO visited
                APPEND tuple of position, next_speed, and steps PLUS one TO queue
            END IF
        END WHILE
    END FUNCTION
END CLASS
