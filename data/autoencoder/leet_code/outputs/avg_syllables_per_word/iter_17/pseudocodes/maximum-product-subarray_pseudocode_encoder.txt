CLASS Solution
    FUNCTION maxProduct(list_of_numbers)
        IF list_of_numbers IS EMPTY
            RETURN zero
        END IF

        SET maximum_product TO element at position zero of list_of_numbers
        SET minimum_product TO element at position zero of list_of_numbers
        SET result TO element at position zero of list_of_numbers

        FOR each number IN the sublist of list_of_numbers starting from position one TO the end
            IF number IS LESS THAN zero
                SET temporary_variable TO maximum_product
                SET maximum_product TO minimum_product
                SET minimum_product TO temporary_variable
            END IF

            SET maximum_product TO the GREATER OF number AND maximum_product MULTIPLIED BY number
            SET minimum_product TO the LESSER OF number AND minimum_product MULTIPLIED BY number

            SET result TO the GREATER OF result AND maximum_product
        END FOR

        RETURN result
    END FUNCTION
END CLASS
