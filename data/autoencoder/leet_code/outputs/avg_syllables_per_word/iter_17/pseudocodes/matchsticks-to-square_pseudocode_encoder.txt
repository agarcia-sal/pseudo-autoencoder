CLASS Solution
    FUNCTION makesquare(matchsticks)
        SET total_length TO THE SUM OF matchsticks
        IF total_length MODULO four IS NOT EQUALS TO zero
            RETURN False
        END IF
        
        SET side_length TO total_length DIVIDED BY four
        CALL sort ON matchsticks WITH reverse order
        
        FUNCTION can_form_sides(sides, index)
            IF index IS EQUALS TO THE LENGTH OF matchsticks
                RETURN all side IS EQUALS TO side_length FOR each side IN sides
            END IF
            
            FOR i FROM zero TO three
                IF element at position i of sides PLUS element at position index of matchsticks IS LESS THAN OR EQUAL TO side_length
                    INCREMENT element at position i of sides BY element at position index of matchsticks
                    IF CALL can_form_sides WITH sides AND index PLUS one RETURNS True
                        RETURN True
                    END IF
                    DECREMENT element at position i of sides BY element at position index of matchsticks
                END IF
                
                IF element at position i of sides IS EQUALS TO zero
                    BREAK
                END IF
            END FOR
            
            RETURN False
        END FUNCTION
        
        RETURN CALL can_form_sides WITH a list of four zeros AND zero
    END FUNCTION
END CLASS
