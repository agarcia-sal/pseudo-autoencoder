CLASS Solution
    FUNCTION kIncreasing(arr, k)
        FUNCTION longest_increasing_subsequence(subarr)
            SET lis TO empty list
            FOR each num IN subarr
                IF lis IS empty OR num GREATER THAN OR EQUAL TO element at position LENGTH OF lis MINUS one OF lis
                    APPEND num TO lis
                ELSE
                    SET pos TO CALL bisect_right WITH lis AND num
                    SET element at position pos OF lis TO num
                END IF
            END FOR
            RETURN the LENGTH OF lis
        END FUNCTION

        SET total_operations TO zero
        SET n TO the LENGTH OF arr

        FOR start FROM zero TO k MINUS one
            SET subarr TO empty list
            FOR i FROM start TO n MINUS one STEP k
                APPEND element at position i OF arr TO subarr
            END FOR

            SET lis_length TO CALL longest_increasing_subsequence WITH subarr
            INCREMENT total_operations BY the LENGTH OF subarr MINUS lis_length
        END FOR

        RETURN total_operations
    END FUNCTION
END CLASS
