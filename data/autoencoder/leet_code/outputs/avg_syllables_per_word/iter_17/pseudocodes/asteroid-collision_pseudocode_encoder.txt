CLASS Solution
    FUNCTION asteroidCollision(list_of_asteroids)
        SET stack TO empty list
        FOR each asteroid IN list_of_asteroids
            WHILE the LENGTH OF stack IS GREATER THAN zero AND asteroid IS LESS THAN zero AND element at position the last index OF stack IS GREATER THAN zero
                IF element at position the last index OF stack IS LESS THAN the NEGATION OF asteroid
                    REMOVE element at position the last index FROM stack
                    CONTINUE
                ELSE IF element at position the last index OF stack IS EQUALS the NEGATION OF asteroid
                    REMOVE element at position the last index FROM stack
                END IF
                BREAK
            END WHILE
            IF the previous WHILE loop DID NOT BREAK prematurely
                APPEND asteroid TO stack
            END IF
        END FOR
        RETURN stack
    END FUNCTION
END CLASS
