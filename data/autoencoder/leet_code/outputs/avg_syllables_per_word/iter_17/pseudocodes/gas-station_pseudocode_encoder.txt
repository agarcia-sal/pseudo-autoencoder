CLASS Solution
    FUNCTION canCompleteCircuit(gas, cost)
        IF the SUM OF gas IS LESS THAN the SUM OF cost
            RETURN negative one
        END IF
        SET start TO zero
        SET total_tank TO zero
        FOR index FROM zero TO the LENGTH OF gas MINUS one
            INCREMENT total_tank BY the element at position index of gas MINUS the element at position index of cost
            IF total_tank IS LESS THAN zero
                SET start TO index PLUS one
                SET total_tank TO zero
            END IF
        END FOR
        RETURN start
    END FUNCTION
END CLASS
