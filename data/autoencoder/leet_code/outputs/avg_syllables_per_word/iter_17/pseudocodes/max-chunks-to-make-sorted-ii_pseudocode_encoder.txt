CLASS Solution
    FUNCTION maxChunksToSorted(arr_of_integers)
        SET list_max_so_far TO an empty list
        SET current_max TO element at position zero of arr_of_integers
        FOR each number IN arr_of_integers
            SET current_max TO the GREATER THAN value BETWEEN current_max AND number
            APPEND current_max TO list_max_so_far
        END FOR
        SET sorted_arr TO CALL auxiliary FUNCTION sorted WITH arr_of_integers
        SET chunks TO zero
        SET max_chunk TO zero
        FOR index FROM zero TO the LENGTH OF arr_of_integers MINUS one
            SET max_chunk TO the GREATER THAN value BETWEEN max_chunk AND element at position index of arr_of_integers
            IF max_chunk EQUALS element at position index of sorted_arr
                INCREMENT chunks BY one
            END IF
        END FOR
        RETURN chunks
    END FUNCTION
END CLASS
