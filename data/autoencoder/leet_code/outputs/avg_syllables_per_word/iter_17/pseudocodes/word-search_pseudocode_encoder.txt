CLASS Solution
    FUNCTION exist(board, word)
        IF board IS empty
            RETURN False
        END IF

        SET m TO the LENGTH OF board
        SET n TO the LENGTH OF element at position zero of board

        SET visited TO a two-dimensional list OF size m BY n FILLED WITH False

        FUNCTION dfs(x, y, index)
            IF index EQUALS the LENGTH OF word
                RETURN True
            END IF
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of visited AND element at position y of that IS True OR element at position x of board AND element at position y of that NOT EQUALS element at position index of word
                RETURN False
            END IF

            SET element at position x of visited AND element at position y of that TO True
            FOR each pair dx, dy IN list of pairs consisting OF minus one zero, one zero, zero minus one, zero one
                IF CALL dfs WITH parameters x PLUS dx, y PLUS dy, index PLUS one RETURNS True
                    RETURN True
                END IF
            END FOR
            SET element at position x of visited AND element at position y of that TO False
            RETURN False
        END FUNCTION

        FOR i FROM zero TO m MINUS one
            FOR j FROM zero TO n MINUS one
                IF CALL dfs WITH parameters i, j, zero RETURNS True
                    RETURN True
                END IF
            END FOR
        END FOR
        RETURN False
    END FUNCTION
END CLASS
