CLASS Solution
    FUNCTION judgeSquareSum(c)
        SET left TO zero
        SET right TO the INTEGER part OF the SQUARE ROOT OF c
        WHILE left IS LESS THAN OR EQUAL TO right
            SET current_sum TO left MULTIPLIED BY left PLUS right MULTIPLIED BY right
            IF current_sum EQUALS c
                RETURN true
            ELSE IF current_sum IS LESS THAN c
                INCREMENT left BY one
            ELSE
                DECREMENT right BY one
            END IF
        END WHILE
        RETURN false
    END FUNCTION
END CLASS
