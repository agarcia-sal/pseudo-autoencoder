CLASS Solution
    FUNCTION maxChunksToSorted(list_of_integers arr)
        SET max_so_far TO zero
        SET chunks TO zero
        FOR index FROM zero TO the LENGTH OF arr MINUS one
            SET max_so_far TO the GREATER VALUE BETWEEN max_so_far AND element at position index of arr
            IF max_so_far EQUALS index
                INCREMENT chunks BY one
            END IF
        END FOR
        RETURN chunks
    END FUNCTION
END CLASS
