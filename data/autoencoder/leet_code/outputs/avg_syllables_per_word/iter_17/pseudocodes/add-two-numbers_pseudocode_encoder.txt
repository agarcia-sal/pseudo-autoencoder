CLASS ListNode
    FUNCTION __init__(self, val EQUALS zero, next EQUALS None)
        SET self.val TO val
        SET self.next TO next
    END FUNCTION
END CLASS

FUNCTION list_node(values)
    IF values IS empty
        RETURN None
    END IF
    SET head TO CALL ListNode WITH element at position zero of values
    SET p TO head
    FOR each val IN the elements from position one TO end of values
        SET node TO CALL ListNode WITH val
        SET p.next TO node
        SET p TO node
    END FOR
    RETURN head
END FUNCTION

FUNCTION is_same_list(p1, p2)
    IF p1 IS None AND p2 IS None
        RETURN True
    END IF
    IF p1 IS None OR p2 IS None
        RETURN False
    END IF
    RETURN the value at p1 EQUALS the value at p2 AND CALL is_same_list WITH the element at position next of p1 AND the element at position next of p2
END FUNCTION

CLASS Solution
    FUNCTION addTwoNumbers(l1, l2)
        SET dummy_head TO CALL ListNode WITH zero
        SET current TO dummy_head
        SET carry TO zero
        
        WHILE l1 IS NOT None OR l2 IS NOT None OR carry IS NOT zero
            IF l1 IS NOT None
                SET val1 TO the value at l1
            ELSE
                SET val1 TO zero
            END IF
            IF l2 IS NOT None
                SET val2 TO the value at l2
            ELSE
                SET val2 TO zero
            END IF
            SET total TO val1 PLUS val2 PLUS carry
            SET carry TO total DIVIDED BY ten
            SET current.next TO CALL ListNode WITH total MODULO ten
            SET current TO the element at position next of current
            
            IF l1 IS NOT None
                SET l1 TO the element at position next of l1
            END IF
            IF l2 IS NOT None
                SET l2 TO the element at position next of l2
            END IF
        END WHILE
        
        RETURN the element at position next of dummy_head
    END FUNCTION
END CLASS
