CLASS Solution
    FUNCTION combinationSum4(nums AS list_of_integers, target AS integer) RETURNS integer
        SET dp TO auxiliary FUNCTION initialize_list_with_zeros OF size target PLUS one
        SET element at position zero of dp TO one
        FOR amount FROM one TO target
            FOR each num IN nums
                IF amount IS GREATER THAN OR EQUAL TO num THEN
                    INCREMENT element at position amount of dp BY element at position amount MINUS num of dp
                END IF
            END FOR
        END FOR
        RETURN element at position target of dp
    END FUNCTION
END CLASS
