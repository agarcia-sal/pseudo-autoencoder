CLASS Solution
    FUNCTION eraseOverlapIntervals(list_of_intervals)
        IF list_of_intervals IS empty
            RETURN zero
        END IF

        CALL list_of_intervals SORTED BY element at position one of each interval

        SET end_time TO element at position one of the element at position zero of list_of_intervals
        SET removal_count TO zero

        FOR index FROM one TO the LENGTH OF list_of_intervals MINUS one
            IF element at position zero of element at position index of list_of_intervals GREATER THAN OR EQUAL TO end_time
                SET end_time TO element at position one of element at position index of list_of_intervals
            ELSE
                INCREMENT removal_count BY one
            END IF
        END FOR

        RETURN removal_count
    END FUNCTION
END CLASS
