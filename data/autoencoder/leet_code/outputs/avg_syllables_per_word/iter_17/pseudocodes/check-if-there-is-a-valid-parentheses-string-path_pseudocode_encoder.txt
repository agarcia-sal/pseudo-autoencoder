CLASS Solution
    FUNCTION hasValidPath(grid)
        SET m TO the LENGTH OF grid
        SET n TO the LENGTH OF element at position zero of grid

        IF the REMAINDER OF m PLUS n MINUS one DIVIDED BY two IS NOT equal TO zero
            RETURN false
        END IF

        CALL initialize_dp WITH parameters m n TO OBTAIN dp

        SET element at position zero row and position one column and position zero in dp TO true

        FOR i FROM zero TO m MINUS one
            FOR j FROM zero TO n MINUS one
                FOR k FROM zero TO m PLUS n
                    IF element at position i row and position j PLUS one column and position k in dp IS true OR element at position i PLUS one row and position j column and position k in dp IS true
                        IF element at position i row and position j column of grid EQUALS open parenthesis AND k PLUS one LESS THAN m PLUS n PLUS one
                            SET element at position i PLUS one row and position j PLUS one column and position k PLUS one in dp TO true
                        ELSE IF element at position i row and position j column of grid EQUALS close parenthesis AND k MINUS one GREATER THAN OR EQUAL TO zero
                            SET element at position i PLUS one row and position j PLUS one column and position k MINUS one in dp TO true
                        END IF
                    END IF
                END FOR
            END FOR
        END FOR

        RETURN element at position m row and position n column and position zero in dp
    END FUNCTION

    FUNCTION initialize_dp(m n)
        CREATE three dimensional list dp OF size m PLUS one BY n PLUS one BY m PLUS n PLUS one FILLED WITH false VALUES
        RETURN dp
    END FUNCTION
END CLASS
