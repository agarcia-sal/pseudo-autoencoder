CLASS Solution
    FUNCTION validSquare(p1_list_of_integers, p2_list_of_integers, p3_list_of_integers, p4_list_of_integers)
        FUNCTION squared_distance(point_one, point_two)
            SET distance_to_zero TO point_one element at position zero MINUS point_two element at position zero
            SET distance_to_zero_squared TO distance_to_zero MULTIPLIED BY distance_to_zero
            SET distance_to_one TO point_one element at position one MINUS point_two element at position one
            SET distance_to_one_squared TO distance_to_one MULTIPLIED BY distance_to_one
            RETURN distance_to_zero_squared PLUS distance_to_one_squared
        END FUNCTION

        SET points TO a list containing p1_list_of_integers, p2_list_of_integers, p3_list_of_integers, and p4_list_of_integers
        SET distances TO an empty set

        FOR index_i FROM zero TO three
            FOR index_j FROM index_i PLUS one TO three
                SET dist TO CALL squared_distance WITH point_one as element at position index_i of points AND point_two as element at position index_j of points
                IF dist EQUALS zero
                    RETURN False
                END IF
                ADD dist TO distances
            END FOR
        END FOR

        RETURN the LENGTH OF distances EQUALS two
    END FUNCTION
END CLASS
