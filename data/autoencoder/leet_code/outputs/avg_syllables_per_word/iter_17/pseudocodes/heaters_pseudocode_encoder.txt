CLASS Solution
    FUNCTION findRadius(houses, heaters)
        CALL sort ON houses
        CALL sort ON heaters

        SET min_radius TO zero
        SET j TO zero

        FOR each house IN houses
            WHILE j LESS THAN the LENGTH OF heaters MINUS one AND the ABSOLUTE VALUE OF element at position j of heaters MINUS house GREATER THAN OR EQUAL TO the ABSOLUTE VALUE OF element at position j PLUS one of heaters MINUS house holds
                INCREMENT j BY one
            END WHILE

            SET min_radius TO the MAXIMUM OF min_radius AND the ABSOLUTE VALUE OF element at position j of heaters MINUS house
        END FOR

        RETURN min_radius
    END FUNCTION
END CLASS
