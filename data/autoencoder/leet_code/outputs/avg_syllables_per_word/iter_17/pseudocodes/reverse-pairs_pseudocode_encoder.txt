CLASS Solution
    FUNCTION reversePairs(nums)
        FUNCTION merge_and_count(nums, start, end)
            IF start GREATER THAN OR EQUAL TO end
                RETURN zero
            END IF

            SET mid TO the INTEGER DIVISION OF the SUM OF start PLUS end BY two
            SET count TO the RESULT OF merge_and_count WITH nums AND start AND mid PLUS the RESULT OF merge_and_count WITH nums AND mid PLUS one AND end

            SET j TO mid PLUS one
            FOR i FROM start TO mid
                WHILE j LESS THAN OR EQUAL TO end AND element at position i of nums GREATER THAN two MULTIPLIED BY element at position j of nums
                    INCREMENT j BY one
                END WHILE
                INCREMENT count BY j MINUS mid MINUS one
            END FOR

            SET merged TO an EMPTY list
            SET left TO start
            SET right TO mid PLUS one
            WHILE left LESS THAN OR EQUAL TO mid AND right LESS THAN OR EQUAL TO end
                IF element at position left of nums LESS THAN OR EQUAL TO element at position right of nums
                    APPEND element at position left of nums TO merged
                    INCREMENT left BY one
                ELSE
                    APPEND element at position right of nums TO merged
                    INCREMENT right BY one
                END IF
            END WHILE

            CALL extend merged WITH the SUBLIST from position left TO mid of nums
            CALL extend merged WITH the SUBLIST from position right TO end of nums

            FOR index FROM zero TO the LENGTH OF merged MINUS one
                SET element at position start PLUS index of nums TO element at position index of merged
            END FOR

            RETURN count
        END FUNCTION

        RETURN merge_and_count WITH nums AND zero AND the LENGTH OF nums MINUS one
    END FUNCTION
END CLASS
