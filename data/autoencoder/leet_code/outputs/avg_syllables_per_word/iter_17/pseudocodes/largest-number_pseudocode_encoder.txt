CLASS Solution
    FUNCTION largestNumber(list_of_integers nums)
        SET list_of_strings nums_str TO CALL ConvertIntegersToStrings WITH nums

        FUNCTION compare(string x, string y)
            IF concatenation of x FOLLOWED BY y GREATER THAN concatenation of y FOLLOWED BY x holds
                RETURN negative one
            ELSE IF concatenation of x FOLLOWED BY y LESS THAN concatenation of y FOLLOWED BY x holds
                RETURN one
            ELSE
                RETURN zero
            END IF
        END FUNCTION

        CALL SortList WITH list_to_sort nums_str AND key_function cmp_to_key_applied_to compare

        IF element at position zero of nums_str EQUALS zero as string
            RETURN zero as string
        END IF

        RETURN concatenation of all elements of nums_str
    END FUNCTION
END CLASS
