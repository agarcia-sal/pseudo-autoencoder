CLASS Solution
    FUNCTION minSubArrayLen(target_integer_parameter, list_of_integers_parameter)
        SET left_pointer TO zero
        SET current_sum TO zero
        SET minimum_length TO positive_infinity_constant
        FOR index right_pointer FROM zero TO the LENGTH OF list_of_integers_parameter MINUS one
            INCREMENT current_sum BY element at position right_pointer of list_of_integers_parameter
            WHILE current_sum GREATER THAN OR EQUAL TO target_integer_parameter holds
                SET minimum_length TO the MINIMUM BETWEEN minimum_length AND right_pointer MINUS left_pointer PLUS one
                DECREMENT current_sum BY element at position left_pointer of list_of_integers_parameter
                INCREMENT left_pointer BY one
            END WHILE
        END FOR
        IF minimum_length NOT EQUALS positive_infinity_constant
            RETURN minimum_length
        ELSE
            RETURN zero
        END IF
    END FUNCTION
END CLASS
