CLASS Solution
    FUNCTION permuteUnique(list_of_numbers)
        FUNCTION backtrack(current_path, used_flags, result_list)
            IF the LENGTH OF current_path EQUALS the LENGTH OF list_of_numbers
                APPEND a COPY OF current_path TO result_list
                RETURN
            END IF
            FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one
                IF element at position index of used_flags IS true
                OR
                index GREATER THAN zero AND element at position index of list_of_numbers EQUALS element at position index MINUS one of list_of_numbers AND element at position index MINUS one of used_flags IS false
                    CONTINUE TO next iteration
                END IF
                SET element at position index of used_flags TO true
                APPEND element at position index of list_of_numbers TO current_path
                CALL backtrack WITH current_path used_flags result_list
                REMOVE last element FROM current_path
                SET element at position index of used_flags TO false
            END FOR
        END FUNCTION

        CALL sort ON list_of_numbers
        SET result_list TO an empty list
        SET used_flags TO a list of false with the LENGTH OF list_of_numbers
        CALL backtrack WITH an empty list used_flags result_list
        RETURN result_list
    END FUNCTION
END CLASS
