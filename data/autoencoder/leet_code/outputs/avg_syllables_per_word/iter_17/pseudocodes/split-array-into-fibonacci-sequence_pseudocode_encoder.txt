CLASS Solution
    FUNCTION splitIntoFibonacci(num_string)
        FUNCTION backtrack(start_position, current_path)
            IF start_position EQUALS the LENGTH OF num_string AND the LENGTH OF current_path IS GREATER THAN OR EQUAL TO three
                RETURN current_path
            END IF
            FOR end_position FROM start_position PLUS one TO the LENGTH OF num_string
                IF the ELEMENT AT POSITION start_position OF num_string EQUALS zero AND end_position IS GREATER THAN start_position PLUS one
                    CONTINUE TO next iteration of FOR loop
                END IF
                SET next_number TO the INTEGER CONVERSION OF the SUBSTRING FROM POSITION start_position TO end_position MINUS one OF num_string
                IF next_number IS GREATER THAN OR EQUAL TO two RAISED TO THE POWER OF thirty ONE
                    BREAK out of FOR loop
                END IF
                IF the LENGTH OF current_path IS LESS THAN two OR next_number EQUALS the SUM OF the last ELEMENT AND the SECOND TO last ELEMENT OF current_path
                    SET result TO CALL backtrack WITH PARAMETERS end_position AND current_path APPENDED WITH next_number
                    IF result IS NOT empty
                        RETURN result
                    END IF
                END IF
            END FOR
            RETURN empty list
        END FUNCTION

        RETURN CALL backtrack WITH PARAMETERS zero AND empty list
    END FUNCTION
END CLASS
