CLASS Solution
    FUNCTION validUtf8(list_of_integers data)
        FUNCTION get_byte_count(integer byte)
            IF the result of byte SHIFTED RIGHT BY seven IS EQUAL TO zero
                RETURN one
            ELSE IF the result of byte SHIFTED RIGHT BY five IS EQUAL TO one one zero in binary
                RETURN two
            ELSE IF the result of byte SHIFTED RIGHT BY four IS EQUAL TO one one one zero in binary
                RETURN three
            ELSE IF the result of byte SHIFTED RIGHT BY three IS EQUAL TO one one one one zero in binary
                RETURN four
            ELSE
                RETURN negative one
            END IF
        END FUNCTION

        SET integer i TO zero
        WHILE i IS LESS THAN the LENGTH OF data
            SET integer byte_count TO CALL get_byte_count WITH the element at position i of data
            IF byte_count IS EQUAL TO negative one
                RETURN False
            END IF
            IF i PLUS byte_count IS GREATER THAN the LENGTH OF data
                RETURN False
            END IF
            FOR integer j FROM i PLUS one TO i PLUS byte_count MINUS one
                IF the result of the element at position j of data SHIFTED RIGHT BY six IS NOT EQUAL TO one zero in binary
                    RETURN False
                END IF
            END FOR
            INCREMENT i BY byte_count
        END WHILE
        RETURN True
    END FUNCTION
END CLASS
