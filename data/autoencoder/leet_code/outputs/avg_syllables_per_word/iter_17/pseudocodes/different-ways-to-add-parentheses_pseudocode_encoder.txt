CLASS Solution
    FUNCTION diffWaysToCompute(expression)
        FUNCTION compute(left, right, operator)
            SET results TO empty list
            FOR each l IN left
                FOR each r IN right
                    IF operator EQUALS plus sign
                        APPEND l PLUS r TO results
                    ELSE IF operator EQUALS minus sign
                        APPEND l MINUS r TO results
                    ELSE IF operator EQUALS multiplication sign
                        APPEND l MULTIPLIED BY r TO results
                    END IF
                END FOR
            END FOR
            RETURN results
        END FUNCTION

        FUNCTION helper(sub_expr)
            IF sub_expr CONTAINS only digits
                RETURN list CONTAINING the INTEGER VALUE OF sub_expr
            END IF

            SET results TO empty list
            FOR index FROM zero TO the LENGTH OF sub_expr MINUS one
                IF element at position index of sub_expr IS IN plus sign OR minus sign OR multiplication sign
                    SET left_results TO CALL helper WITH the substring from position zero TO index MINUS one of sub_expr
                    SET right_results TO CALL helper WITH the substring from position index PLUS one TO the END OF sub_expr
                    APPEND all elements FROM CALL compute WITH left_results, right_results, and element at position index of sub_expr TO results
                END IF
            END FOR
            RETURN results
        END FUNCTION

        RETURN CALL helper WITH expression
END CLASS
