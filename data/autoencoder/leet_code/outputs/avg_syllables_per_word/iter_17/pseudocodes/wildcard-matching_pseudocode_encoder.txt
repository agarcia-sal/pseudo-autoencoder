CLASS Solution
    FUNCTION isMatch(s, p)
        SET dp TO auxiliary FUNCTION initialize_dp_table WITH parameters length of s PLUS one AND length of p PLUS one
        
        SET element at position zero position zero of dp TO True
        
        FOR j FROM one TO length of p
            IF element at position j minus one of pattern p EQUALS star
                SET element at position zero position j of dp TO element at position zero position j minus one of dp
            END IF
        END FOR
        
        FOR i FROM one TO length of s
            FOR j FROM one TO length of p
                IF element at position j minus one of pattern p EQUALS star
                    SET element at position i position j of dp TO element at position i minus one position j of dp OR element at position i position j minus one of dp
                ELSE IF element at position j minus one of pattern p EQUALS question mark OR element at position i minus one of string s EQUALS element at position j minus one of pattern p
                    SET element at position i position j of dp TO element at position i minus one position j minus one of dp
                END IF
            END FOR
        END FOR
        
        RETURN element at position length of s position length of p of dp
    END FUNCTION
END CLASS
