CLASS Solution
    FUNCTION groupStrings(list_of_strings)
        FUNCTION get_canonical_form(string_s)
            IF string_s IS empty
                RETURN empty string
            END IF
            SET canonical_form TO CALL convert_string_to_canonical WITH string_s
            RETURN canonical_form
        END FUNCTION

        SET groups TO CALL initialize_default_dictionary WITH list type

        FOR each string_s IN list_of_strings
            SET canonical_form TO CALL get_canonical_form WITH string_s
            CALL append string_s TO the list AT key canonical_form IN groups
        END FOR

        RETURN CALL convert_dictionary_values_to_list WITH groups
    END FUNCTION
END CLASS
