CLASS Solution
    FUNCTION smallestRange(nums)
        SET min_heap TO an empty list
        SET max_value TO negative infinity

        FOR index FROM zero TO the LENGTH OF nums MINUS one
            CALL helper_push_heap WITH min_heap AND the tuple of element at position zero of list at position index of nums AND index AND zero
            SET max_value TO the GREATER THAN value BETWEEN max_value AND element at position zero of list at position index of nums
        END FOR

        SET smallest_range TO infinity
        SET result TO the list of negative one hundred thousand AND one hundred thousand

        WHILE min_heap is not empty
            SET min_value TO element at position zero of the element at position zero of heap pop from min_heap
            SET list_index TO element at position one of the element at position zero of heap pop from min_heap
            SET element_index TO element at position two of the element at position zero of heap pop from min_heap

            IF max_value MINUS min_value LESS THAN smallest_range
                SET smallest_range TO max_value MINUS min_value
                SET result TO the list of min_value AND max_value
            END IF

            IF element_index PLUS one LESS THAN the LENGTH OF list at position list_index of nums
                SET next_value TO element at position element_index PLUS one of list at position list_index of nums
                CALL helper_push_heap WITH min_heap AND the tuple of next_value AND list_index AND element_index PLUS one
                SET max_value TO the GREATER THAN value BETWEEN max_value AND next_value
            ELSE
                BREAK
            END IF
        END WHILE

        RETURN result
    END FUNCTION
END CLASS
