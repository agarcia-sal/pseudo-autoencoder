CLASS Solution
    FUNCTION originalDigits(input_string)
        CALL importCounterHelper FUNCTION
        
        SET count TO CALL Counter WITH parameter input_string
        
        SET digits TO CALL createZeroArray FUNCTION WITH parameter ten
        
        SET element at position zero of digits TO element at position character z of count
        SET element at position two of digits TO element at position character w of count
        SET element at position four of digits TO element at position character u of count
        SET element at position six of digits TO element at position character x of count
        SET element at position eight of digits TO element at position character g of count
        
        SET element at position three of digits TO element at position character h of count MINUS element at position eight of digits
        SET element at position five of digits TO element at position character f of count MINUS element at position four of digits
        SET element at position seven of digits TO element at position character s of count MINUS element at position six of digits
        SET element at position nine of digits TO element at position character i of count MINUS element at position five of digits MINUS element at position six of digits MINUS element at position eight of digits
        SET element at position one of digits TO element at position character o of count MINUS element at position zero of digits MINUS element at position two of digits MINUS element at position four of digits
        
        SET result TO empty list
        
        FOR index FROM zero TO nine
            FOR count_number FROM zero TO element at position index of digits MINUS one
                APPEND string representation of index TO result
            END FOR
        END FOR
        
        RETURN CALL joinList FUNCTION WITH parameter result
    END FUNCTION
END CLASS
