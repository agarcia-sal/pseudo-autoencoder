CLASS Solution
    FUNCTION characterReplacement(s STRING, k INTEGER) RETURNS INTEGER
        CREATE count AS A DICTIONARY WITH DEFAULT VALUE zero INTEGER
        
        SET max_length TO zero
        SET max_count TO zero
        SET left TO zero
        
        FOR right FROM zero TO the LENGTH OF s MINUS one
            INCREMENT the element at position right of s IN count BY one
            SET max_count TO THE GREATER OF max_count AND the element at position right of s IN count
            
            IF the LENGTH OF the CURRENT WINDOW DEFINED BY right MINUS left PLUS one MINUS max_count IS GREATER THAN k
                DECREMENT the element at position left of s IN count BY one
                INCREMENT left BY one
            END IF
            
            SET max_length TO THE GREATER OF max_length AND right MINUS left PLUS one
        END FOR
        
        RETURN max_length
    END FUNCTION
END CLASS
