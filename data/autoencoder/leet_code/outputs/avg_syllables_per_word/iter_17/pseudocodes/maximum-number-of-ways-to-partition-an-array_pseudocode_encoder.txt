CLASS Solution
    FUNCTION waysToPartition(nums, k)
        SET total_sum TO the SUM OF nums
        SET n TO the LENGTH OF nums
        
        SET left_diff TO CALL initialize_default_dictionary WITH integer
        SET right_diff TO CALL initialize_default_dictionary WITH integer
        
        SET current_sum TO zero
        FOR index FROM zero TO n MINUS one
            INCREMENT current_sum BY element at position index of nums
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum
            INCREMENT element at position diff of right_diff BY one
        END FOR
        
        SET max_ways TO element at position zero of right_diff
        
        SET current_sum TO zero
        FOR index FROM zero TO n MINUS one
            INCREMENT current_sum BY element at position index of nums
            
            SET diff_change TO k MINUS element at position index of nums
            
            SET new_ways TO element at position NEGATIVE diff_change of left_diff PLUS element at position diff_change of right_diff
            SET max_ways TO the MAXIMUM OF max_ways AND new_ways
            
            SET diff TO total_sum MINUS two MULTIPLIED BY current_sum
            DECREMENT element at position diff of right_diff BY one
            INCREMENT element at position diff of left_diff BY one
        END FOR
        
        RETURN max_ways
    END FUNCTION
END CLASS
