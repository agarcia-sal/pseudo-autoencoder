CLASS Solution
    FUNCTION isInterleave(s1_string, s2_string, s3_string)
        IF the LENGTH OF s1_string PLUS the LENGTH OF s2_string DOES NOT EQUAL the LENGTH OF s3_string
            RETURN false
        END IF

        SET dp_list TO a list of false repeated the LENGTH OF s2_string PLUS one TIMES
        SET element at position zero of dp_list TO true

        FOR index_i FROM zero TO the LENGTH OF s1_string
            FOR index_j FROM zero TO the LENGTH OF s2_string
                IF index_i IS GREATER THAN zero
                    SET element at position index_j of dp_list TO element at position index_j of dp_list AND element at position index_i MINUS one of s1_string EQUALS element at position index_i PLUS index_j MINUS one of s3_string
                END IF
                IF index_j IS GREATER THAN zero
                    SET element at position index_j of dp_list TO element at position index_j of dp_list OR (element at position index_j MINUS one of dp_list AND element at position index_j MINUS one of s2_string EQUALS element at position index_i PLUS index_j MINUS one of s3_string)
                END IF
            END FOR
        END FOR

        RETURN element at position the LENGTH OF s2_string of dp_list
    END FUNCTION
END CLASS
