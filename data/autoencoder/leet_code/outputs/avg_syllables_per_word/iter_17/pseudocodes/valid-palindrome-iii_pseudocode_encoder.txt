CLASS Solution
    FUNCTION isValidPalindrome(s, k)
        FUNCTION longest_palindromic_subseq(s)
            SET n TO the LENGTH OF s
            CALL initialize_two_dimensional_list WITH n TO dp
            
            FOR index FROM zero TO n MINUS one
                SET element at position index and position index of dp TO one
            END FOR
            
            FOR length FROM two TO n
                FOR i FROM zero TO n MINUS length
                    SET j TO i PLUS length MINUS one
                    IF element at position i of s EQUALS element at position j of s
                        SET element at position i and position j of dp TO element at position i PLUS one and position j MINUS one of dp PLUS two
                    ELSE
                        SET element at position i and position j of dp TO the MAXIMUM OF element at position i PLUS one and position j of dp AND element at position i and position j MINUS one of dp
                    END IF
                END FOR
            END FOR
            
            RETURN element at position zero and position n MINUS one of dp
        END FUNCTION
        
        SET longest_palindrome_length TO CALL longest_palindromic_subseq WITH s
        RETURN the LENGTH OF s MINUS longest_palindrome_length LESS THAN OR EQUAL TO k
    END FUNCTION
END CLASS
