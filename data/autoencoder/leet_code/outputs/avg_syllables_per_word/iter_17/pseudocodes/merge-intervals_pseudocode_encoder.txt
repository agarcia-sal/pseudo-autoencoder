CLASS Solution
    FUNCTION merge(intervals)
        CALL intervals SORT WITH key_function_that_returns_element_at_position zero of input_element
        SET merged TO empty list
        FOR each interval IN intervals
            IF merged IS empty OR element at position one of element at position negative one of merged IS LESS THAN element at position zero of interval
                APPEND interval TO merged
            ELSE
                SET element at position one of element at position negative one of merged TO the GREATER OF element at position one of element at position negative one of merged AND element at position one of interval
            END IF
        END FOR
        RETURN merged
    END FUNCTION
END CLASS
