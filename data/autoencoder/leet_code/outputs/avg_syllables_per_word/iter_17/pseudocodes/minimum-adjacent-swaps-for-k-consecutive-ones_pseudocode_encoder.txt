CLASS Solution
    FUNCTION minMoves(nums, k)
        SET positions TO CALL extract_positions_of_ones WITH nums
        FUNCTION calculate_cost(start, end)
            SET mid TO start PLUS end DIVIDED BY two
            SET median TO element at position mid of positions
            SET cost TO zero
            FOR i FROM start TO end MINUS one
                INCREMENT cost BY absolute value of element at position i of positions MINUS median MINUS mid MINUS i
            END FOR
            RETURN cost
        END FUNCTION
        SET min_cost TO positive infinity
        FOR i FROM zero TO the LENGTH OF positions MINUS k
            SET min_cost TO the minimum BETWEEN min_cost AND CALL calculate_cost WITH i AND i PLUS k
        END FOR
        RETURN min_cost
    END FUNCTION
END CLASS
