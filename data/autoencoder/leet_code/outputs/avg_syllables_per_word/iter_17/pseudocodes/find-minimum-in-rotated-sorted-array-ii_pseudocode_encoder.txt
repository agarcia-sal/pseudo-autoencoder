CLASS Solution
    FUNCTION findMin(list_of_numbers)
        SET left TO zero
        SET right TO the LENGTH OF list_of_numbers MINUS one
        WHILE left LESS THAN right holds
            SET mid TO the INTEGER DIVISION OF the SUM OF left AND right BY two
            IF element at position mid of list_of_numbers GREATER THAN element at position right of list_of_numbers
                SET left TO mid PLUS one
            ELSE IF element at position mid of list_of_numbers LESS THAN element at position right of list_of_numbers
                SET right TO mid
            ELSE
                DECREMENT right BY one
            END IF
        END WHILE
        RETURN element at position left of list_of_numbers
    END FUNCTION
END CLASS
