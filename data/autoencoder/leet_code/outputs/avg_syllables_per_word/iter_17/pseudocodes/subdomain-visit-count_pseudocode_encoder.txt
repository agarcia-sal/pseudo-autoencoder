CLASS Solution
    FUNCTION subdomainVisits(cpdomains)
        SET domain_count TO CALL get_default_dict_with_int_default
        FOR each cpdomain IN cpdomains
            SET count_and_domain TO CALL split_string_by_whitespace WITH cpdomain
            SET count_string TO element at position zero of count_and_domain
            SET domain TO element at position one of count_and_domain
            SET count TO CALL convert_string_to_integer WITH count_string
            SET subdomains TO CALL split_string_by_period WITH domain

            FOR index FROM zero TO the LENGTH OF subdomains MINUS one
                SET subdomain_list TO elements from position index TO the END of subdomains
                SET subdomain TO CALL join_list_with_period WITH subdomain_list
                INCREMENT element at key subdomain of domain_count BY count
            END FOR
        END FOR

        SET result_list TO CALL create_list
        FOR each domain_key AND count_value IN domain_count
            SET formatted_string TO CALL format_count_and_domain WITH count_value AND domain_key
            APPEND formatted_string TO result_list
        END FOR

        RETURN result_list
    END FUNCTION

    FUNCTION get_default_dict_with_int_default
        DESCRIPTION: Auxiliary function to obtain a default dictionary with integer default values
    END FUNCTION

    FUNCTION split_string_by_whitespace(input_string)
        DESCRIPTION: Auxiliary function to split a string into parts separated by whitespace
    END FUNCTION

    FUNCTION convert_string_to_integer(input_string)
        DESCRIPTION: Auxiliary function to convert string representation of integer to integer
    END FUNCTION

    FUNCTION split_string_by_period(input_string)
        DESCRIPTION: Auxiliary function to split a string into parts separated by period character
    END FUNCTION

    FUNCTION join_list_with_period(list_of_strings)
        DESCRIPTION: Auxiliary function to join a list of strings into a single string separated by period character
    END FUNCTION

    FUNCTION create_list
        DESCRIPTION: Auxiliary function to create an empty list
    END FUNCTION

    FUNCTION format_count_and_domain(count_value, domain_string)
        DESCRIPTION: Auxiliary function to create a string representing count_value followed by domain_string separated by a space
    END FUNCTION
END CLASS
