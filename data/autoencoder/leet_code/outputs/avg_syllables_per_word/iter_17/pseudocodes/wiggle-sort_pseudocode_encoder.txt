CLASS Solution
    FUNCTION wiggleSort(list_of_numbers)
        SET less TO true
        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one MINUS one
            IF less holds
                IF element at position index of list_of_numbers GREATER THAN element at position index plus one of list_of_numbers
                    SET temporary_variable TO element at position index of list_of_numbers
                    SET element at position index of list_of_numbers TO element at position index plus one of list_of_numbers
                    SET element at position index plus one of list_of_numbers TO temporary_variable
                END IF
            ELSE
                IF element at position index of list_of_numbers LESS THAN element at position index plus one of list_of_numbers
                    SET temporary_variable TO element at position index of list_of_numbers
                    SET element at position index of list_of_numbers TO element at position index plus one of list_of_numbers
                    SET element at position index plus one of list_of_numbers TO temporary_variable
                END IF
            END IF
            SET less TO NOT less
        END FOR
    END FUNCTION
END CLASS
