CLASS Solution
    FUNCTION findDuplicate(paths)
        ASSIGN content_to_paths TO CALL defaultdict WITH parameter list
        FOR each path IN paths
            SET parts TO CALL split ON path WITH parameter space
            SET directory TO element at position zero of parts
            FOR each file_info IN elements from position one TO end of parts
                SET file_name TO substring from start TO position of character LEFT PARENTHESIS in file_info
                SET content TO substring from position after character LEFT PARENTHESIS TO one less than last position of file_info
                CALL append ON element at key content of content_to_paths WITH argument concatenation of directory CONCATENATED WITH character SLASH CONCATENATED WITH file_name
            END FOR
        END FOR
        RETURN list of elements FROM values of content_to_paths WHERE length of element GREATER THAN one
    END FUNCTION
END CLASS
