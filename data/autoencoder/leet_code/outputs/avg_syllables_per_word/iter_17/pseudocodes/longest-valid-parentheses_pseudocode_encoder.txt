CLASS Solution
    FUNCTION longestValidParentheses(s)
        SET stack TO a list containing negative one
        SET max_length TO zero
        FOR each index i FROM zero TO the LENGTH OF s MINUS one
            SET char TO element at position i of s
            IF char EQUALS open parenthesis
                APPEND i TO stack
            ELSE
                REMOVE element FROM stack
                IF stack IS empty
                    APPEND i TO stack
                ELSE
                    SET max_length TO the GREATER OF max_length AND i MINUS element at last position of stack
                END IF
            END IF
        END FOR
        RETURN max_length
    END FUNCTION
END CLASS
