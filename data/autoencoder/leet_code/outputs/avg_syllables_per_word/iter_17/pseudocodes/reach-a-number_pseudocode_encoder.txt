CLASS Solution
    FUNCTION reachNumber(target)
        SET target TO the ABSOLUTE VALUE OF target
        SET k TO zero
        SET sum_steps TO zero
        WHILE sum_steps IS LESS THAN target
            INCREMENT k BY one
            INCREMENT sum_steps BY k
        END WHILE
        IF the REMAINDER OF sum_steps MINUS target DIVIDED BY two EQUALS zero
            RETURN k
        ELSE
            IF the REMAINDER OF k DIVIDED BY two EQUALS zero
                RETURN k PLUS one
            ELSE
                RETURN k PLUS two
            END IF
        END IF
    END FUNCTION
END CLASS
