CLASS Solution
    FUNCTION ladderLength(beginWord, endWord, wordList)
        IF endWord IS NOT IN wordList
            RETURN zero
        END IF

        CALL createDefaultDictionaryNamed word_dict WITH list AS default value
        SET word_length TO the LENGTH OF beginWord

        FOR each word IN wordList
            FOR index FROM zero TO word_length MINUS one
                SET intermediate_key TO the CONCATENATION OF substring from position zero TO index MINUS one of word PLUS the character star PLUS substring from position index PLUS one TO the end of word
                APPEND word TO the LIST ASSOCIATED WITH intermediate_key IN word_dict
            END FOR
        END FOR

        CALL createDequeNamed queue WITH the tuple of beginWord AND one
        CALL createSetNamed visited WITH beginWord

        WHILE queue IS NOT EMPTY
            SET current_word, current_length TO the REMOVAL AND RETURN OF the FIRST ELEMENT OF queue

            FOR index FROM zero TO word_length MINUS one
                SET intermediate_word TO the CONCATENATION OF substring from position zero TO index MINUS one of current_word PLUS the character star PLUS substring from position index PLUS one TO the end of current_word

                FOR each word IN the LIST ASSOCIATED WITH intermediate_word IN word_dict
                    IF word EQUALS endWord
                        RETURN current_length PLUS one
                    END IF
                    IF word IS NOT IN visited
                        ADD word TO visited
                        APPEND the TUPLE of word AND current_length PLUS one TO queue
                    END IF
                END FOR
            END FOR
        END WHILE

        RETURN zero
    END FUNCTION
END CLASS
