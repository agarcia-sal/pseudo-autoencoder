CLASS Solution
    FUNCTION getProbability(balls)
        FUNCTION dfs(i, j, diff)
            IF i GREATER THAN OR EQUAL TO k
                IF j EQUALS zero AND diff EQUALS zero
                    RETURN one
                ELSE
                    RETURN zero
                END IF
            END IF
            IF j LESS THAN zero
                RETURN zero
            END IF
            SET ans TO zero
            FOR x FROM zero TO element at position i of balls INCLUSIVE
                IF x EQUALS element at position i of balls
                    SET y TO one
                ELSE IF x EQUALS zero
                    SET y TO negative one
                ELSE
                    SET y TO zero
                END IF
                INCREMENT ans BY CALL dfs WITH parameters i PLUS one, j MINUS x, diff PLUS y MULTIPLIED BY comb WITH parameters element at position i of balls AND x
            END FOR
            RETURN ans
        END FUNCTION

        SET n TO the SUM OF all elements of balls BITWISE RIGHT SHIFTED BY one
        SET k TO the LENGTH OF balls
        RETURN CALL dfs WITH parameters zero, n, zero DIVIDED BY comb WITH parameters n BITWISE LEFT SHIFTED BY one AND n
    END FUNCTION
END CLASS
