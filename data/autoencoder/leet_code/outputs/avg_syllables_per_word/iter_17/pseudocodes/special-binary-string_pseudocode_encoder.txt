CLASS Solution
    FUNCTION makeLargestSpecial(s)
        FUNCTION dfs(substring)
            SET count TO zero
            SET start_position TO zero
            INITIALIZE specials_list AS empty list
            
            FOR index FROM zero TO LENGTH OF substring MINUS one
                SET character TO element at position index of substring
                IF character EQUALS the character one
                    INCREMENT count BY one
                ELSE
                    DECREMENT count BY one
                END IF
                
                IF count EQUALS zero
                    APPEND the concatenation of character one WITH CALL dfs WITH the substring from position start_position PLUS one TO index MINUS one WITH character zero TO specials_list
                    SET start_position TO index PLUS one
                END IF
            END FOR
            
            CALL FUNCTION sort in descending order WITH specials_list
            RETURN CALL FUNCTION join WITH specials_list
        END FUNCTION
        
        RETURN CALL dfs WITH s
    END FUNCTION
END CLASS
