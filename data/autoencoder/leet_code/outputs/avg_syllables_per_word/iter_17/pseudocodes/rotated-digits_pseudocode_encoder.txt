CLASS Solution
    FUNCTION rotatedDigits(n)
        SET change_digits TO auxiliary FUNCTION GetChangeDigits
        SET same_digits TO auxiliary FUNCTION GetSameDigits

        FUNCTION is_good_number(num)
            SET num_str TO auxiliary FUNCTION ConvertNumberToString WITH num
            IF auxiliary FUNCTION AllDigitsIn UNION OF change_digits AND same_digits FOR each digit IN num_str holds
                IF auxiliary FUNCTION AnyDigitIn change_digits FOR each digit IN num_str holds
                    RETURN true
                END IF
            END IF
            RETURN false
        END FUNCTION

        SET good_count TO zero
        FOR i FROM one TO n
            IF CALL is_good_number WITH i
                INCREMENT good_count BY one
            END IF
        END FOR

        RETURN good_count
    END FUNCTION
END CLASS
