CLASS Solution
    FUNCTION minimumXORSum(nums1, nums2)
        SET n TO the LENGTH OF nums1
        SET dp TO a list of size two RAISED TO the power of n FILLED WITH infinity
        SET element at position zero of dp TO zero
        
        FOR mask FROM zero TO two RAISED TO the power of n MINUS one
            SET bit_count TO the COUNT OF one characters IN the BINARY REPRESENTATION OF mask
            FOR j FROM zero TO n MINUS one
                IF the BITWISE AND of mask AND two RAISED TO the power of j IS NOT zero
                    SET dp AT position mask TO the MINIMUM OF dp AT position mask AND dp AT position mask BITWISE XOR two RAISED TO the power of j PLUS the BITWISE XOR of element at position bit_count MINUS one of nums1 AND element at position j of nums2
                END IF
            END FOR
        END FOR
        
        RETURN element at position two RAISED TO the power of n MINUS one of dp
    END FUNCTION
END CLASS
