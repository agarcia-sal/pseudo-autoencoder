CLASS Solution
    FUNCTION multiply(mat1, mat2)
        SET m TO the LENGTH OF mat1
        SET k TO the LENGTH OF the element at position zero of mat1
        SET k TO the LENGTH OF mat2
        SET n TO the LENGTH OF the element at position zero of mat2
        SET result TO CALL InitializeResultMatrix WITH parameters m AND n
        FOR i FROM zero TO m MINUS one
            FOR j FROM zero TO n MINUS one
                FOR p FROM zero TO k MINUS one
                    INCREMENT the element at position i then position j of result BY the element at position i then position p of mat1 MULTIPLIED BY the element at position p then position j of mat2
                END FOR
            END FOR
        END FOR
        RETURN result
    END FUNCTION

END CLASS
