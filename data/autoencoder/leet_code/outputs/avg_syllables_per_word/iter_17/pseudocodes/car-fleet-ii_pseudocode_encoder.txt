CLASS Solution
    FUNCTION getCollisionTimes(cars AS list of list of integers)
        SET number_of_cars TO the LENGTH OF cars
        SET collision_times TO a list of negative one point zero repeated number_of_cars times
        SET index_stack TO an empty list

        FOR index FROM number_of_cars MINUS one TO zero DECREMENT BY one
            SET position, speed TO element at position index of cars

            WHILE index_stack is NOT empty
                SET next_index TO element at last position of index_stack
                SET next_position, next_speed TO element at position next_index of cars

                IF speed LESS THAN OR EQUAL TO next_speed
                    REMOVE element at last position FROM index_stack
                    CONTINUE to next iteration of WHILE
                END IF

                SET collision_time TO next_position MINUS position DIVIDED BY speed MINUS next_speed

                IF element at position next_index of collision_times GREATER THAN zero AND collision_time GREATER THAN OR EQUAL TO element at position next_index of collision_times
                    REMOVE element at last position FROM index_stack
                    CONTINUE to next iteration of WHILE
                END IF

                SET element at position index of collision_times TO collision_time
                BREAK from WHILE
            END WHILE

            APPEND index TO index_stack
        END FOR

        RETURN collision_times
    END FUNCTION
END CLASS
