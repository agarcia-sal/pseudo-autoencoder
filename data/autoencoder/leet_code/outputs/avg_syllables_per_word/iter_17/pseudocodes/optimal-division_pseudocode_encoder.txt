CLASS Solution
    FUNCTION optimalDivision(list_of_numbers)
        IF the LENGTH OF list_of_numbers EQUALS one
            RETURN the string representation OF the element at position zero OF list_of_numbers
        END IF
        IF the LENGTH OF list_of_numbers EQUALS two
            RETURN the string concatenation OF the string representation OF the element at position zero OF list_of_numbers PLUS the character slash PLUS the string representation OF the element at position one OF list_of_numbers
        END IF
        RETURN the string concatenation OF the string representation OF the element at position zero OF list_of_numbers PLUS the character slash PLUS the opening parenthesis PLUS the string concatenation OF the string representations OF each element FROM position one TO the end OF list_of_numbers SEPARATED BY the character slash PLUS the closing parenthesis
    END FUNCTION
END CLASS
