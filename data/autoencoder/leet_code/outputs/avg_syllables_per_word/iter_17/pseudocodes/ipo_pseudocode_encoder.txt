CLASS Solution
    FUNCTION findMaximizedCapital(k, w, profits, capital)
        SET projects TO CALL zip WITH parameters capital AND profits CONVERTED TO list
        CALL the SORT method ON projects

        SET max_heap TO empty list

        SET i TO zero
        SET n TO the LENGTH OF projects

        FOR count FROM one TO k
            WHILE i LESS THAN n AND the element at position i of projects AT position zero LESS THAN OR EQUAL TO w
                CALL heappush FROM heapq WITH parameters max_heap AND the NEGATION OF the element at position i of projects AT position one
                INCREMENT i BY one
            END WHILE

            IF max_heap IS empty
                BREAK loop
            END IF

            SET w TO w PLUS the NEGATION OF the result of CALL heappop FROM heapq WITH parameter max_heap
        END FOR

        RETURN w
    END FUNCTION
END CLASS
