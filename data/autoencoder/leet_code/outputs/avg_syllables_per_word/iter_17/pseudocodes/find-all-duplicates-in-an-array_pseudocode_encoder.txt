CLASS Solution
    FUNCTION findDuplicates(list_of_numbers)
        FOR index FROM zero TO the LENGTH OF list_of_numbers MINUS one
            WHILE element at position index of list_of_numbers IS NOT EQUAL TO element at position element at position index of list_of_numbers MINUS one of list_of_numbers
                SET temporary_value TO element at position index of list_of_numbers
                SET element at position element at position index of list_of_numbers MINUS one of list_of_numbers TO element at position index of list_of_numbers
                SET element at position index of list_of_numbers TO temporary_value
            END WHILE
        END FOR
        RETURN a new list of elements value FROM index_value AND value IN enumeration of list_of_numbers WHERE value IS NOT EQUAL TO index_value PLUS one
    END FUNCTION
END CLASS
