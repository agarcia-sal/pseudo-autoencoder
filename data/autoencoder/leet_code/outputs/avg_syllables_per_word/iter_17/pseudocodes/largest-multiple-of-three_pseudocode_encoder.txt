CLASS Solution
    FUNCTION largestMultipleOfThree(digits)
        SET count TO CALL Counter WITH digits
        SET total_sum TO CALL sum WITH digits
        SET remainder TO total_sum MODULO three

        IF remainder EQUALS one
            FOR each d IN the LIST one four seven
                IF the element at position d of count GREATER THAN zero
                    DECREMENT the element at position d of count BY one
                    BREAK
                END IF
            END FOR
            IF none was removed in the previous loop
                FOR each d IN the LIST two five eight
                    IF the element at position d of count GREATER THAN one
                        DECREMENT the element at position d of count BY two
                        BREAK
                    END IF
                END FOR
            END IF
        END IF

        IF remainder EQUALS two
            FOR each d IN the LIST two five eight
                IF the element at position d of count GREATER THAN zero
                    DECREMENT the element at position d of count BY one
                    BREAK
                END IF
            END FOR
            IF none was removed in the previous loop
                FOR each d IN the LIST one four seven
                    IF the element at position d of count GREATER THAN one
                        DECREMENT the element at position d of count BY two
                        BREAK
                    END IF
                END FOR
            END IF
        END IF

        SET result TO an empty list
        FOR d FROM nine TO zero DECREMENT BY one
            APPEND the string representation of d REPEATED the element at position d of count times TO result
        END FOR

        SET final_number TO CALL the JOIN function WITH an empty string AND result

        IF final_number is not empty AND the element at position zero of final_number EQUALS zero
            RETURN zero
        ELSE
            RETURN final_number
        END IF
    END FUNCTION
END CLASS
