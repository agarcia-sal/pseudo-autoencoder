CLASS TreeNode
    FUNCTION __init__(val EQUALS zero, left EQUALS None, right EQUALS None)
        SET self.val TO val
        SET self.left TO left
        SET self.right TO right
    END FUNCTION
END CLASS

FUNCTION tree_node(list_of_values)
    IF list_of_values IS empty
        RETURN None
    END IF
    SET root TO A NEW INSTANCE OF TreeNode WITH element at position zero OF list_of_values
    SET i TO one
    SET queue TO A NEW DEQUE
    CALL queue APPEND WITH root
    WHILE queue IS NOT empty
        SET node TO queue POP LEFT
        IF i IS LESS THAN the LENGTH OF list_of_values AND element at position i OF list_of_values IS NOT None
            SET node.left TO A NEW INSTANCE OF TreeNode WITH element at position i OF list_of_values
            CALL queue APPEND WITH node.left
        END IF
        INCREMENT i BY one
        IF i IS LESS THAN the LENGTH OF list_of_values AND element at position i OF list_of_values IS NOT None
            SET node.right TO A NEW INSTANCE OF TreeNode WITH element at position i OF list_of_values
            CALL queue APPEND WITH node.right
        END IF
        INCREMENT i BY one
    END WHILE
    RETURN root
END FUNCTION

FUNCTION is_same_tree(p, q)
    IF p IS None AND q IS None
        RETURN True
    ELSE IF p IS None OR q IS None
        RETURN False
    ELSE IF p.val IS NOT EQUALS TO q.val
        RETURN False
    ELSE
        RETURN CALL is_same_tree WITH p.left AND q.left AND CALL is_same_tree WITH p.right AND q.right
    END IF
END FUNCTION

CLASS Solution
    FUNCTION pathSum(root, targetSum)
        FUNCTION dfs(node, current_sum, path)
            IF node IS None
                RETURN
            END IF
            CALL path APPEND WITH node.val
            SET current_sum TO current_sum PLUS node.val
            IF node.left IS None AND node.right IS None AND current_sum IS EQUALS TO targetSum
                CALL result APPEND WITH A COPY OF path
            END IF
            CALL dfs WITH node.left, current_sum, path
            CALL dfs WITH node.right, current_sum, path
            CALL path POP
        END FUNCTION

        SET result TO AN EMPTY LIST
        CALL dfs WITH root, zero, AN EMPTY LIST
        RETURN result
    END FUNCTION
END CLASS
