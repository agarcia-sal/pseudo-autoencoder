CLASS Solution
    FUNCTION deleteAndEarn(list_of_numbers)
        SET count_of_each_number TO CALL Counter WITH list_of_numbers
        SET sorted_unique_numbers TO CALL sorted WITH count_of_each_number
        SET length_of_unique_numbers TO the LENGTH OF sorted_unique_numbers
        INITIALIZE list dp_values WITH zero REPEATED length_of_unique_numbers PLUS one TIMES
        SET element at position one of dp_values TO element at position zero of sorted_unique_numbers MULTIPLIED BY element at position zero of count_of_each_number at key element at position zero of sorted_unique_numbers
        FOR index FROM two TO length_of_unique_numbers
            SET current_number TO element at position index MINUS one of sorted_unique_numbers
            IF current_number EQUALS element at position index MINUS two of sorted_unique_numbers PLUS one
                SET element at position index of dp_values TO the GREATER OF element at position index MINUS one of dp_values AND element at position index MINUS two of dp_values PLUS current_number MULTIPLIED BY element at key current_number of count_of_each_number
            ELSE
                SET element at position index of dp_values TO element at position index MINUS one of dp_values PLUS current_number MULTIPLIED BY element at key current_number of count_of_each_number
            END IF
        END FOR
        RETURN element at position length_of_unique_numbers of dp_values
    END FUNCTION
END CLASS
