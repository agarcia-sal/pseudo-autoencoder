CLASS Solution
    FUNCTION fullBloomFlowers(flowers AS list_of_list_of_integers, people AS list_of_integers) RETURNS list_of_integers
        SET start_times TO CALL extract_and_sort_start_times FROM flowers
        SET end_times TO CALL extract_and_sort_end_times FROM flowers
        SET result TO empty list
        FOR each person IN people
            SET started TO CALL bisect_right WITH start_times AND person
            SET ended TO CALL bisect_left WITH end_times AND person
            SET count_of_flowers_in_bloom TO started MINUS ended
            APPEND count_of_flowers_in_bloom TO result
        END FOR
        RETURN result
    END FUNCTION
END CLASS
