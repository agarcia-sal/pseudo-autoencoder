CLASS Solution
    FUNCTION lengthLongestPath(input)
        SET lines TO CALL split WITH parameter newline ON input
        SET stack TO an empty list
        SET max_length TO zero
        
        FOR each line IN lines
            SET depth TO the NUMBER OF tab character IN line
            SET name TO the SUBSTRING OF line FROM position depth TO the END
            
            WHILE the LENGTH OF stack GREATER THAN depth
                CALL pop ON stack
            END WHILE
            
            IF the LENGTH OF stack GREATER THAN zero
                SET current_length TO the ELEMENT at POSITION negative one of stack PLUS the LENGTH OF name PLUS one
            ELSE
                SET current_length TO the LENGTH OF name
            END IF
            
            IF name CONTAINS period character
                SET max_length TO the GREATER OF max_length AND current_length
            ELSE
                APPEND current_length TO stack
            END IF
        END FOR
        
        RETURN max_length
    END FUNCTION
END CLASS
