CLASS Solution
    FUNCTION replaceNonCoprimes(list_of_numbers)
        SET stack TO empty list
        FOR each number IN list_of_numbers
            WHILE the LENGTH OF stack IS greater THAN zero
                SET greatest_common_divisor TO CALL gcd WITH the element at position the LENGTH OF stack MINUS one of stack AND number
                IF greatest_common_divisor EQUALS one
                    BREAK out of WHILE loop
                END IF
                SET number TO the product of the element at position the LENGTH OF stack MINUS one of stack MULTIPLIED BY number DIVIDED BY greatest_common_divisor
                REMOVE the last element FROM stack
            END WHILE
            APPEND number TO stack
        END FOR
        RETURN stack
    END FUNCTION
END CLASS
