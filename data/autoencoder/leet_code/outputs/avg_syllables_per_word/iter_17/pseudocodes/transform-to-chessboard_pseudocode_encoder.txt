CLASS Solution
    FUNCTION movesToChessboard(board)
        SET N TO the LENGTH OF board

        SET rowSum TO zero
        SET colSum TO zero
        SET rowSwap TO zero
        SET colSwap TO zero

        FOR index i FROM zero TO N MINUS one
            FOR index j FROM zero TO N MINUS one
                IF element at position i of board AND element at position j of element at position i of board XOR element at position i of element at position zero of board XOR element at position j of element at position zero of board XOR element at position zero of element at position zero of board IS NOT EQUAL TO zero
                    RETURN negative one
                END IF
            END FOR

            INCREMENT rowSum BY element at position i of element at position zero of board
            INCREMENT colSum BY element at position zero of element at position i of board
            INCREMENT rowSwap BY one IF element at position i of element at position zero of board EQUALS i MODULO two ELSE zero
            INCREMENT colSwap BY one IF element at position zero of element at position i of board EQUALS i MODULO two ELSE zero
        END FOR

        IF the INTEGER DIVISION OF N BY two IS GREATER THAN rowSum OR rowSum IS GREATER THAN the INTEGER DIVISION OF N PLUS one BY two
            RETURN negative one
        END IF

        IF the INTEGER DIVISION OF N BY two IS GREATER THAN colSum OR colSum IS GREATER THAN the INTEGER DIVISION OF N PLUS one BY two
            RETURN negative one
        END IF

        IF N MODULO two IS EQUAL TO one
            IF rowSwap MODULO two IS EQUAL TO one
                SET rowSwap TO N MINUS rowSwap
            END IF
            IF colSwap MODULO two IS EQUAL TO one
                SET colSwap TO N MINUS colSwap
            END IF
        ELSE
            SET rowSwap TO the MINIMUM OF rowSwap AND N MINUS rowSwap
            SET colSwap TO the MINIMUM OF colSwap AND N MINUS colSwap
        END IF

        RETURN the INTEGER DIVISION OF rowSwap PLUS colSwap BY two
    END FUNCTION
END CLASS
