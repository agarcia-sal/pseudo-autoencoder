CLASS Solution
    FUNCTION longestAwesome(string_s)
        SET prefix_map TO a dictionary with element zero mapped TO negative one
        SET prefix_xor TO zero
        SET max_length TO zero
        FOR each index i AND character char IN string_s
            CALL CONVERT_CHARACTER_TO_INTEGER WITH char RETURNING digit
            SET prefix_xor TO prefix_xor EXCLUSIVE OR one SHIFTED LEFT BY digit
            IF prefix_xor EXISTS IN prefix_map
                SET max_length TO THE GREATER OF max_length AND i MINUS THE ELEMENT AT position prefix_xor OF prefix_map
            END IF
            FOR each integer j FROM zero TO nine
                SET flipped TO prefix_xor EXCLUSIVE OR one SHIFTED LEFT BY j
                IF flipped EXISTS IN prefix_map
                    SET max_length TO THE GREATER OF max_length AND i MINUS THE ELEMENT AT position flipped OF prefix_map
                END IF
            END FOR
            IF prefix_xor DOES NOT EXIST IN prefix_map
                SET THE ELEMENT AT position prefix_xor OF prefix_map TO i
            END IF
        END FOR
        RETURN max_length
    END FUNCTION
END CLASS
