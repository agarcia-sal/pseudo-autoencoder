func findOrder(numCourses, prerequisites):
    adj = {c: [] for c in 0..numCourses-1}
    indeg = [0]*numCourses
    
    for (d, s) in prerequisites:
        adj[s].append(d)
        indeg[d] += 1
    
    q = queue of courses where indeg[c] == 0
    order = []
    
    while q not empty:
        cur = q.pop_front()
        order.append(cur)
        for nxt in adj[cur]:
            indeg[nxt] -= 1
            if indeg[nxt] == 0: q.append(nxt)
    
    return order if len(order) == numCourses else []
