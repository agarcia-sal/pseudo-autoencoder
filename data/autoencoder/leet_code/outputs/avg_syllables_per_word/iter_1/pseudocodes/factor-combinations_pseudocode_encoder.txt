func getFactors(n):
    result = []
    path = []
    func backtrack(start, target):
        for i in range(start, floor(sqrt(target)) + 1):
            if target mod i = 0:
                result.append(path + [i, target / i])
                path.append(i)
                backtrack(i, target / i)
                path.pop()
    backtrack(2, n)
    return result
