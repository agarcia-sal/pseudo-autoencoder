class TreeNode(val=0, left=None, right=None)

tree_node(vals):
    if vals empty: return None
    root = TreeNode(vals[0])
    q = deque([root])
    i = 1
    while q:
        n = q.popleft()
        if i < len(vals) and vals[i] != None:
            n.left = TreeNode(vals[i])
            q.append(n.left)
        i += 1
        if i < len(vals) and vals[i] != None:
            n.right = TreeNode(vals[i])
            q.append(n.right)
        i += 1
    return root

is_same_tree(p,q):
    if not p and not q: return True
    if not p or not q: return False
    if p.val != q.val: return False
    return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
    insertIntoBST(root, val):
        dfs(r):
            if r == None: return TreeNode(val)
            if r.val < val: r.right = dfs(r.right)
            else: r.left = dfs(r.left)
            return r
        return dfs(root)
