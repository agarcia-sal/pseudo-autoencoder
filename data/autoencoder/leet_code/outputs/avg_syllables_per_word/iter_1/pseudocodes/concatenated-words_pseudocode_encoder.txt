Solution.findAllConcatenatedWordsInADict(words):
  define dfs(w, start):
    if start == len(w): return true
    node = trie
    for i in start to len(w)-1:
      idx = w[i] - 'a'
      if node.children[idx] is nil: return false
      node = node.children[idx]
      if node.is_end and dfs(w, i+1): return true
    return false

  trie = Trie()
  ans = []
  sort words by length ascending
  for w in words:
    if w != "":
      if dfs(w, 0): ans.append(w)
      else: trie.insert(w)
  return ans
