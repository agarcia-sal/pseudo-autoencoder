sort packages; n = len(packages)  
prefix_sum = cumulative sum of packages with initial 0  

min_waste = ∞  
for each box_list in boxes:  
  sort box_list  
  if largest box < largest package: continue  
  total_waste = 0; last = 0  
  for box_size in box_list:  
   index = last  
   while index < n and packages[index] ≤ box_size: index += 1  
   if index > last:  
    total_waste += (index - last)*box_size - (prefix_sum[index] - prefix_sum[last])  
    last = index  
  if last == n: min_waste = min(min_waste, total_waste)  

return min_waste % 1_000_000_007 if min_waste ≠ ∞ else -1
