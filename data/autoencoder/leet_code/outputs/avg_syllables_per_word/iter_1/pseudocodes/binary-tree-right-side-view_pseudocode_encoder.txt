class TreeNode(val=0, left=None, right=None)

tree_node(vals):
  if empty(vals): return None
  root ← TreeNode(vals[0])
  i ← 1
  q ← queue with root
  while q not empty:
    n ← q.pop_front()
    if i < len(vals) and vals[i] ≠ None:
      n.left ← TreeNode(vals[i])
      q.push_back(n.left)
    i += 1
    if i < len(vals) and vals[i] ≠ None:
      n.right ← TreeNode(vals[i])
      q.push_back(n.right)
    i += 1
  return root

is_same_tree(p, q):
  if not p and not q: return True
  if not p or not q or p.val ≠ q.val: return False
  return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

Solution.rightSideView(root):
  if not root: return []
  res ← []
  q ← queue with root
  while q not empty:
    lvl_len ← len(q)
    for i in 0..lvl_len-1:
      n ← q.pop_front()
      if i == lvl_len-1: res.append(n.val)
      if n.left: q.push_back(n.left)
      if n.right: q.push_back(n.right)
  return res
