n ← length(s)

for i from n-1 downto 0:
    for j from i to n-1:
        is_pal[i][j] ← (s[i] = s[j]) and (j−i ≤ 1 or is_pal[i+1][j−1])

for i in 1 to n-2:
    if is_pal[0][i−1]:
        for j in i to n-2:
            if is_pal[i][j] and is_pal[j+1][n−1]:
                return True
return False
