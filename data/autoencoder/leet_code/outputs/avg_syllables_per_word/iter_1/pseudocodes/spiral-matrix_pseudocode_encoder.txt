if matrix empty: return []
res = []
top, bottom = 0, rows-1
left, right = 0, cols-1

while top ≤ bottom and left ≤ right:
    for i in left..right: res += matrix[top][i]
    top += 1
    for i in top..bottom: res += matrix[i][right]
    right -= 1
    if top ≤ bottom:
        for i in right..left step -1: res += matrix[bottom][i]
        bottom -= 1
    if left ≤ right:
        for i in bottom..top step -1: res += matrix[i][left]
        left += 1

return res
