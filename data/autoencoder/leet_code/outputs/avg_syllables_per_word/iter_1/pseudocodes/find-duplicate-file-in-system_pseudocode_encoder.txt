init content_to_paths = defaultdict(list)
for path in paths:
    dir, files = split path by first space
    for f in files:
        name, content = split f by '('
        content = content without trailing ')'
        content_to_paths[content].append(dir + '/' + name)
return [group for group in content_to_paths.values() if size(group) > 1]
