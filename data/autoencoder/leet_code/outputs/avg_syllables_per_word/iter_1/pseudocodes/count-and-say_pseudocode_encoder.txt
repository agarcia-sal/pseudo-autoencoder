if n = 1: return "1"
seq = "1"
for _ in 2 to n:
    next_seq = ""
    count = 1
    prev = seq[0]
    for i in 1 to len(seq)-1:
        if seq[i] = prev: count += 1
        else:
            next_seq += str(count) + prev
            prev = seq[i]
            count = 1
    next_seq += str(count) + prev
    seq = next_seq
return seq
