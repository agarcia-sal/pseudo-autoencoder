if grid empty: return 0
m,n = dims(grid)
max_k = 0
col_kills = array of 0 size n

for i in 0 to m-1:
  row_k = 0
  for j in 0 to n-1:
    if j==0 or grid[i][j-1]=='W':
      row_k = count enemies right from (i,j) until wall
    if i==0 or grid[i-1][j]=='W':
      col_kills[j] = count enemies down from (i,j) until wall
    if grid[i][j]=='0':
      max_k = max(max_k, row_k + col_kills[j])

return max_k
