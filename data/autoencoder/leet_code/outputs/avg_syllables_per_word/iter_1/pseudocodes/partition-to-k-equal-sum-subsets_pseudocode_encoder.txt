sum ← sum(nums)
if sum mod k ≠ 0 return False
target ← sum / k
sort nums descending
used ← array of False length nums

function can_partition(i, k, curr_sum):
    if k = 0 return True
    if curr_sum = target return can_partition(0, k-1, 0)
    for j from i to len(nums)-1:
        if not used[j] and curr_sum + nums[j] ≤ target:
            used[j] ← True
            if can_partition(j+1, k, curr_sum + nums[j]) return True
            used[j] ← False
    return False

return can_partition(0, k, 0)
