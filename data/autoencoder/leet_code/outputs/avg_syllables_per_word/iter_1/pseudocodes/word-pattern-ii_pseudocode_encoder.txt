fn backtrack(pIdx, sIdx, map, used):
    if pIdx == len(pattern) and sIdx == len(s): return True
    if pIdx == len(pattern) or sIdx == len(s): return False
    c = pattern[pIdx]
    if c in map:
        w = map[c]
        if s[sIdx:sIdx+len(w)] != w: return False
        return backtrack(pIdx+1, sIdx+len(w), map, used)
    for end in range(sIdx+1, len(s)+1):
        w = s[sIdx:end]
        if w in used: continue
        map[c], used += w, {w}
        if backtrack(pIdx+1, end, map, used): return True
        del map[c], used -= {w}
    return False

return backtrack(0, 0, {}, {})
