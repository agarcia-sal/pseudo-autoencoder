      matrix[i][0] = 0; matrix[0][j] = 0
for i in 1..m-1:
  for j in 1..n-1:
    if matrix[i][0] == 0 or matrix[0][j] == 0:
      matrix[i][j] = 0
if fr0: for j in 0..n-1: matrix[0][j] = 0
if fc0: for i in 0..m-1: matrix[i][0] = 0
