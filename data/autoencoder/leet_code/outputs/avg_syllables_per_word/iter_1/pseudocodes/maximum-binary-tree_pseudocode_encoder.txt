class TreeNode(val=0, left=null, right=null)

func tree_node(vals):
  if empty(vals): return null
  root = TreeNode(vals[0])
  q = queue(root)
  i = 1
  while q not empty:
    n = q.pop()
    if i < len(vals) and vals[i] != null:
      n.left = TreeNode(vals[i])
      q.push(n.left)
    i += 1
    if i < len(vals) and vals[i] != null:
      n.right = TreeNode(vals[i])
      q.push(n.right)
    i += 1
  return root

func is_same_tree(p, q):
  if not p and not q: return true
  if not p or not q: return false
  if p.val != q.val: return false
  return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
  func constructMaximumBinaryTree(nums):
    if empty(nums): return null
    max_v = max(nums)
    idx = nums.index(max_v)
    root = TreeNode(max_v)
    root.left = constructMaximumBinaryTree(nums[:idx])
    root.right = constructMaximumBinaryTree(nums[idx+1:])
    return root
