for i in [0..n-1]: dp[i][i] = 1

max_len = 0
for length in [2..n]:
  for i in [0..n-length]:
    j = i + length - 1
    if s[i] = s[j]:
      dp[i][j] = dp[i+1][j-1] + 2
      if i < len(word1) and j ≥ len(word1):
        max_len = max(max_len, dp[i][j])
    else:
      dp[i][j] = max(dp[i+1][j], dp[i][j-1])

return max_len
