class TreeNode(val=0, left=None, right=None)

tree_node(vals):
  if empty(vals): return None
  root = TreeNode(vals[0])
  i = 1; q = deque([root])
  while q:
    n = q.popleft()
    if i < len(vals) and vals[i] != None: n.left = TreeNode(vals[i]); q.append(n.left)
    i += 1
    if i < len(vals) and vals[i] != None: n.right = TreeNode(vals[i]); q.append(n.right)
    i += 1
  return root

is_same_tree(p, q):
  if not p and not q: return True
  if not p or not q or p.val != q.val: return False
  return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
  convertBST(root):
    sum = 0
    traverse(n):
      if not n: return
      traverse(n.right)
      sum += n.val
      n.val = sum
      traverse(n.left)
    traverse(root)
    return root
