func getCollisionTimes(cars):
  n = length(cars)
  result = array of -1.0 size n
  stack = empty list

  for i from n-1 down to 0:
    pos, spd = cars[i]

    while stack not empty:
      j = top of stack
      next_pos, next_spd = cars[j]

      if spd <= next_spd:
        pop stack
        continue

      t = (next_pos - pos) / (spd - next_spd)

      if result[j] > 0 and t >= result[j]:
        pop stack
        continue

      result[i] = t
      break

    push i to stack

  return result
