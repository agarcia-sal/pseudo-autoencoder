class TreeNode(val=0, left=None, right=None)

tree_node(vals):
  if empty(vals): return None
  root = TreeNode(vals[0])
  i = 1; q = deque([root])
  while q:
    n = q.popleft()
    if i<len(vals) and vals[i]≠None:
      n.left = TreeNode(vals[i]); q.append(n.left)
    i += 1
    if i<len(vals) and vals[i]≠None:
      n.right = TreeNode(vals[i]); q.append(n.right)
    i += 1
  return root

is_same_tree(p, q):
  if not p and not q: return True
  if not p or not q: return False
  if p.val≠q.val: return False
  return is_same_tree(p.left,q.left) and is_same_tree(p.right,q.right)

class Solution:
  rob(root):
    rob_from_node(n):
      if not n: return 0,0
      l_rob, l_not = rob_from_node(n.left)
      r_rob, r_not = rob_from_node(n.right)
      rob_n = n.val + l_not + r_not
      not_rob_n = max(l_rob,l_not) + max(r_rob,r_not)
      return rob_n, not_rob_n
    return max(rob_from_node(root))
