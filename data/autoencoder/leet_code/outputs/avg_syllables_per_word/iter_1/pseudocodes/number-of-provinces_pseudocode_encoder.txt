n = len(isConnected)
visited = [False]*n
province_count = 0

define dfs(city):
    for neighbor in range(n):
        if isConnected[city][neighbor] == 1 and not visited[neighbor]:
            visited[neighbor] = True
            dfs(neighbor)

for city in range(n):
    if not visited[city]:
        visited[city] = True
        dfs(city)
        province_count += 1

return province_count
