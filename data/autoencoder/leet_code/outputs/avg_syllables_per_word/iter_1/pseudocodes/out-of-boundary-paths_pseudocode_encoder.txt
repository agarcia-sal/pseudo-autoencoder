MOD = 10^9+7

dp[m][n][maxMove+1] = 0
for k in 1 to maxMove:
  for i in 0 to m-1:

      dp[i][j][k] = sum((if ni<0 or ni>=m or nj<0 or nj>=n then 1 else dp[ni][nj][k-1]) for (di,dj) in dirs where ni=i+di,nj=j+dj) % MOD
