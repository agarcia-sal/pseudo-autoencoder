n ← length(jobs)
subset_sums[0..2^n-1] ← 0
for i in 0..n-1:
  for mask in 0..2^i-1:
    subset_sums[mask | (1 << i)] ← subset_sums[mask] + jobs[i]

define dp(k, mask):
  if k = 1: return subset_sums[mask]
  res ← ∞
  submask ← mask
  while submask ≠ 0:
    res ← min(res, max(dp(k-1, mask XOR submask), subset_sums[submask]))
    submask ← (submask - 1) AND mask
  return res

return dp(k, 2^n - 1)
