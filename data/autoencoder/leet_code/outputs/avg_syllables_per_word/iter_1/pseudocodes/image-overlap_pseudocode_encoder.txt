n = size of img1
points1 = [(i, j) for i,j in n×n if img1[i][j] = 1]
points2 = [(i, j) for i,j in n×n if img2[i][j] = 1]
overlap_count = map from (dx, dy) to int

for (x1, y1) in points1:
  for (x2, y2) in points2:
    dx = x2 - x1
    dy = y2 - y1
    overlap_count[(dx, dy)] += 1

return max value in overlap_count or 0 if empty
