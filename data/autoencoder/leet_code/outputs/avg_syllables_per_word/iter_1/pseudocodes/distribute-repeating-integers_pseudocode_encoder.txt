count ← frequency values of nums
sort quantity descending

func can_satisfy(i):
    if i = length(quantity): return True
    for j in 0..length(count)-1:
        if count[j] ≥ quantity[i]:
            count[j] -= quantity[i]
            if can_satisfy(i+1): return True
            count[j] += quantity[i]
    return False

return can_satisfy(0)
