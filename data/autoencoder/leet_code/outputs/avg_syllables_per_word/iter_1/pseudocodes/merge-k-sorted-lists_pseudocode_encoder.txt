func mergeKLists(lists):
    heap ← empty min-heap
    for i, lst in lists:
        if lst: push (lst.val, i, lst) into heap
    dummy, cur ← new ListNode(), dummy
    while heap not empty:
        val, i, node ← pop heap
        cur.next, cur ← node, node
        if node.next: push (node.next.val, i, node.next) into heap
    return dummy.next
