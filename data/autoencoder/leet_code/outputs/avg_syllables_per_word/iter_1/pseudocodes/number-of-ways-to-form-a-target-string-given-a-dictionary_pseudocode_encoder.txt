MOD = 10^9+7
n = length(words[0])

freq = array of dicts size n
for word in words:
  for j,char in enumerate(word):
    freq[j][char] += 1
dp = 2D array (m+1)x(n+1) zero-filled
for j in 0..n: dp[0][j] = 1
for i in 1..m:
  for j in 1..n:
    dp[i][j] = dp[i][j-1]
    if freq[j-1][target[i-1]] > 0:
      dp[i][j] += dp[i-1][j-1] * freq[j-1][target[i-1]]
      dp[i][j] = dp[i][j] mod MOD
return dp[m][n]
