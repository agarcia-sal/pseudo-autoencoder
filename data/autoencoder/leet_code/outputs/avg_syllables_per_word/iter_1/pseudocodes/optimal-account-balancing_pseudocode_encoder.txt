compute balance: dict<Person, Int> from transactions (debit -amount from sender, credit +amount to receiver)
balances ← list of non-zero values in balance
if balances empty: return 0

function settle(start, cnt):
    while start < len(balances) and balances[start] == 0: start += 1
    if start == len(balances): return cnt
    min_cnt ← ∞
    for i in (start+1) to end of balances:
        if balances[start]*balances[i] < 0:
            balances[i] += balances[start]
            min_cnt ← min(min_cnt, settle(start+1, cnt+1))
            balances[i] -= balances[start]
    return min_cnt

return settle(0,0)
