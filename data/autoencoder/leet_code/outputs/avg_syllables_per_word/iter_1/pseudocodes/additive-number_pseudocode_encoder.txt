for i in 1 to n//2:
  for j in i+1 to n:
    first, second = num[:i], num[i:j]
    if leading_zero(first) or leading_zero(second): continue
    if valid_sequence(j, int(first), int(second)): return True
return False

valid_sequence(start, f, s):
  if start == n: return True
  t = str(f + s)
  if num[start:start+len(t)] != t: return False
  return valid_sequence(start+len(t), s, int(t))

leading_zero(x): return len(x)>1 and x[0]=='0'
