def ambiguousCoordinates(s):
    def make(f):
        for d in 1 to len(f):
            left, right = f[:d], f[d:]
            if (left == "0" or not left.startswith("0")) and (not right.endswith("0")):
                yield left + ("." if d < len(f) else "") + right

    s = s[1:-1]
    return ["(" + a + ", " + b + ")" 
            for i in 1 to len(s)-1 
            for a in make(s[:i]) 
            for b in make(s[i:])]
