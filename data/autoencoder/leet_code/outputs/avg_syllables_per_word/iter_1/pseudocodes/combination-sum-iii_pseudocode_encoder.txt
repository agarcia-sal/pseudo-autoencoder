func combinationSum3(k, n):
    result = []
    func backtrack(start, path, target, k):
        if target = 0 and k = 0: result += [path]
        if target < 0 or k < 0: return
        for i in range(start to 9):
            backtrack(i+1, path+[i], target - i, k - 1)
    backtrack(1, [], n, k)
    return result
