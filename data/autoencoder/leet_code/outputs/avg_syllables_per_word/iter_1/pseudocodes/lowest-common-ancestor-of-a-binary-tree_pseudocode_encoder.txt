class TreeNode(val=0, left=None, right=None)

func build_tree(vals):
  if vals empty: return None
  root = TreeNode(vals[0])
  q = queue with root
  i = 1
  while q not empty:
    n = q.pop_front()
    if i < len(vals) and vals[i] != None:
      n.left = TreeNode(vals[i])
      q.push_back(n.left)
    i += 1
    if i < len(vals) and vals[i] != None:
      n.right = TreeNode(vals[i])
      q.push_back(n.right)
    i += 1
  return root

func same_tree(p, q):
  if not p and not q: return True
  if not p or not q: return False
  if p.val != q.val: return False
  return same_tree(p.left,q.left) and same_tree(p.right,q.right)

class Solution:
  func lowestCommonAncestor(root, p, q):
    if root is None or root==p or root==q: return root
    left = lowestCommonAncestor(root.left, p, q)
    right = lowestCommonAncestor(root.right, p, q)
    if left != None and right != None: return root
    return left if left != None else right
