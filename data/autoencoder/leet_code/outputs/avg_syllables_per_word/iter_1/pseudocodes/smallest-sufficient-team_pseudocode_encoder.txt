f = [inf]*(1<<m); g = [0]*(1<<m); h = [0]*(1<<m)
f[0] = 0
for i in 0..(1<<m)-1:
  if f[i] == inf: continue
  for j in 0..n-1:
    if f[i]+1 < f[i|p[j]]:
      f[i|p[j]] = f[i]+1
      g[i|p[j]] = j
      h[i|p[j]] = i
i = (1<<m)-1
ans = []
while i > 0:
  ans.append(g[i])
  i = h[i]
return reversed(ans)
