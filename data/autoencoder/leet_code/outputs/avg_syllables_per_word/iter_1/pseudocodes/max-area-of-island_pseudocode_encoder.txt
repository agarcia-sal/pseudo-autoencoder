set max_area = 0
rows, cols = size(grid)
define dfs(r,c):
    if r<0 or r≥rows or c<0 or c≥cols or grid[r][c]=0: return 0
    grid[r][c]=0
    return 1 + dfs(r+1,c) + dfs(r-1,c) + dfs(r,c+1) + dfs(r,c-1)
for r in 0..rows-1:
    for c in 0..cols-1:
        if grid[r][c]=1:
            max_area = max(max_area, dfs(r,c))
return max_area
