class TreeNode(val=0, left=None, right=None)

func tree_node(vals):
  if vals empty: return None
  root = TreeNode(vals[0])
  q = queue()
  q.push(root)
  i = 1
  while q not empty:
    n = q.pop()
    if i < len(vals) and vals[i] != None:
      n.left = TreeNode(vals[i])
      q.push(n.left)
    i += 1
    if i < len(vals) and vals[i] != None:
      n.right = TreeNode(vals[i])
      q.push(n.right)
    i += 1
  return root

func is_same_tree(p, q):
  if not p and not q: return True
  if not p or not q: return False
  if p.val != q.val: return False
  return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
  func findFrequentTreeSum(root):
    count = map with default 0
    func subtree_sum(n):
      if not n: return 0
      s = n.val + subtree_sum(n.left) + subtree_sum(n.right)
      count[s] += 1
      return s
    subtree_sum(root)
    max_freq = max(count.values, default=0)
    return [k for k,v in count.items if v == max_freq]
