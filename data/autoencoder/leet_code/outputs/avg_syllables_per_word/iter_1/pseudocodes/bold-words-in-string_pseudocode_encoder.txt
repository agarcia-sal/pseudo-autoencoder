bold = [False]*len(s)
for w in words:
    i = s.find(w)
    while i != -1:
        for j in range(i, i+len(w)): bold[j] = True
        i = s.find(w, i+1)
res, i = [], 0
while i < len(s):
    if bold[i]:
        res.append("<b>")
        while i < len(s) and bold[i]: res.append(s[i]); i += 1
        res.append("</b>")
    else:
        res.append(s[i])
        i += 1
return ''.join(res)
