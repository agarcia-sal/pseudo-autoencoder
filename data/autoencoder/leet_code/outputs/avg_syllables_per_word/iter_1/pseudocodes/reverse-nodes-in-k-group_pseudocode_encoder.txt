define reverseLinkedList(start, end):
    prev, curr = None, start
    while curr != end:
        nxt = curr.next
        curr.next = prev
        prev, curr = curr, nxt
    curr.next = prev
    return end, start

define reverseKGroup(head, k):
    dummy = ListNode(0); dummy.next = head
    prev_end = dummy
    loop:
        kth = prev_end
        for i in 1 to k:
            kth = kth.next
            if kth is None: return dummy.next
        next_start = kth.next
        kth.next = None
        new_start, new_end = reverseLinkedList(prev_end.next, kth)
        prev_end.next = new_start
        new_end.next = next_start
        prev_end = new_end
