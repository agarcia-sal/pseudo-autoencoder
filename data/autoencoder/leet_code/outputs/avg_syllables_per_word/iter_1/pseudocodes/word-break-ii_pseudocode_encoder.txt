wordBreak(s, wordDict):
  wordSet ← set(wordDict)
  memo ← {}

  backtrack(start):
    if start in memo: return memo[start]
    if start = len(s): return [""]

    res ← []
    for end in start+1 to len(s):
      w ← s[start:end]
      if w ∈ wordSet:
        for sent in backtrack(end):
          res.append(w + (" " + sent if sent else ""))
    memo[start] ← res
    return res

  return backtrack(0)
