if nums empty: return 0
n = len(nums)
lengths = [1]*n
counts = [1]*n
for i in 0 to n-1:
  for j in 0 to i-1:
    if nums[i] > nums[j]:
      if lengths[j]+1 > lengths[i]:
        lengths[i] = lengths[j]+1
        counts[i] = counts[j]
      else if lengths[j]+1 == lengths[i]:
        counts[i] += counts[j]
longest = max(lengths)
return sum(counts[i] for i in 0 to n-1 if lengths[i] == longest)
