function findLength(nums1, nums2):
    dp = matrix of zeros with size (len(nums1)+1) x (len(nums2)+1)
    max_len = 0
    for i in 1 to len(nums1):
        for j in 1 to len(nums2):
            if nums1[i-1] == nums2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
                max_len = max(max_len, dp[i][j])
    return max_len
