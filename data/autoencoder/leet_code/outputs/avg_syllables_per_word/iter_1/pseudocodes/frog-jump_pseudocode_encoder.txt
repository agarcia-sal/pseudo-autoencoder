function canCross(stones):
    if stones is empty or length < 2: return False
    stone_set ← set of stones
    memo ← empty map

    function can_jump_to(pos, jump):
        if pos = last stone: return True
        if (pos, jump) in memo: return memo[(pos, jump)]

        for next_jump in [jump-1, jump, jump+1]:
            if next_jump > 0 and pos + next_jump in stone_set:
                if can_jump_to(pos + next_jump, next_jump):
                    memo[(pos, jump)] ← True
                    return True

        memo[(pos, jump)] ← False
        return False

    return can_jump_to(stones[0], 0)
