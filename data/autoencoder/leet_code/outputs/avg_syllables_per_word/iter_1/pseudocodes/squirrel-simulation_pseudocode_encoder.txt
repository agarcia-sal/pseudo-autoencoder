def minDistance(h, w, tree, squir, nuts):
    dist = lambda p1, p2: abs(p1[0]-p2[0]) + abs(p1[1]-p2[1])
    total = 0
    max_save = -inf
    for n in nuts:
        total += 2*dist(n, tree)
        save = 2*dist(n, tree) - (dist(squir, n) + dist(n, tree))
        max_save = max(max_save, save)
    return total - max_save
