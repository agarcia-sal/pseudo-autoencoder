if nums1 empty or nums2 empty or k ≤ 0: return []
min_heap ← []
for i in 0 to min(k, len(nums1)) - 1: push (nums1[i]+nums2[0], i, 0) into min_heap
result ← []
while min_heap not empty and len(result) < k:
    _, i, j ← pop min_heap
    append [nums1[i], nums2[j]] to result
    if j+1 < len(nums2): push (nums1[i]+nums2[j+1], i, j+1) into min_heap
return result
