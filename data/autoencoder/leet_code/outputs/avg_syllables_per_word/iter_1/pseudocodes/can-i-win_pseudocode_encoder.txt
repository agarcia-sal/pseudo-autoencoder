if sum(1..maxInt) < target: return False
if target <= 0: return True

memo = {}
func canWin(total, used):
  if (total, used) in memo: return memo[(total, used)]
  for n in 1..maxInt:
    if used & (1 << n) == 0:
      if total + n >= target or not canWin(total + n, used | (1 << n)):
        memo[(total, used)] = True
        return True
  memo[(total, used)] = False
  return False

return canWin(0, 0)
