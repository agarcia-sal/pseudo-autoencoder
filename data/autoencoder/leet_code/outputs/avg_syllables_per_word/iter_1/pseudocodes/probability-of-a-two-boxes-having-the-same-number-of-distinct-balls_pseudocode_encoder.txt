define getProbability(balls):
    n ← sum(balls) / 2
    k ← length(balls)

    define dfs(i, j, diff) cached:
        if i ≥ k:
            return 1 if j = 0 and diff = 0 else 0
        if j < 0:
            return 0
        ans ← 0
        for x in 0..balls[i]:
            y ← 1 if x = balls[i] else (-1 if x = 0 else 0)
            ans += dfs(i+1, j - x, diff + y) * C(balls[i], x)
        return ans

    return dfs(0, n, 0) / C(2*n, n)
