class TreeNode(val=0, left=None, right=None)

func tree_node(vals):
  if empty(vals): return None
  root ← TreeNode(vals[0])
  q ← queue(root)
  i ← 1
  while q not empty:
    n ← q.pop()
    if i < len(vals) and vals[i] ≠ None:
      n.left ← TreeNode(vals[i]); q.push(n.left)
    i += 1
    if i < len(vals) and vals[i] ≠ None:
      n.right ← TreeNode(vals[i]); q.push(n.right)
    i += 1
  return root

func is_same_tree(p, q):
  if not p and not q: return True
  if not p or not q: return False
  if p.val ≠ q.val: return False
  return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
  func tree2str(root):
    func helper(n):
      if not n: return ""
      res ← str(n.val)
      if n.left or n.right: res += "(" + helper(n.left) + ")"
      if n.right: res += "(" + helper(n.right) + ")"
      return res
    return helper(root)
