dp = 2D array (len(s)+1) x (len(p)+1) of False
dp[0][0] = True
for j in 1 to len(p):
    if p[j-1] == '*': dp[0][j] = dp[0][j-1]
for i in 1 to len(s):
    for j in 1 to len(p):
        if p[j-1] == '*': dp[i][j] = dp[i-1][j] or dp[i][j-1]
        else if p[j-1] == '?' or s[i-1]==p[j-1]: dp[i][j] = dp[i-1][j-1]
return dp[len(s)][len(p)]
