set inf = large number
define minmaxGasDist(stations, k):
  define possible(max_d):
    count = 0
    for i in 1 to length(stations)-1:
      count += floor((stations[i]-stations[i-1]) / max_d)
    return count <= k

  left, right = 0, 1e8
  while right - left > 1e-6:
    mid = (left + right) / 2
    if possible(mid): right = mid
    else: left = mid
  return left
