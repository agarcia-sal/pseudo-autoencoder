build graph G from connections
disc, low = arrays of size n filled with -1
time = 0
critical = []

define dfs(u, p):
    disc[u], low[u] = time, time
    time += 1
    for v in G[u]:
        if disc[v] == -1:
            dfs(v, u)
            low[u] = min(low[u], low[v])
            if low[v] > disc[u]:
                critical.append([u,v])
        elif v != p:
            low[u] = min(low[u], disc[v])

for u in 0 to n-1:
    if disc[u] == -1:
        dfs(u, -1)

return critical
