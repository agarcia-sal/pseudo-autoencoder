class TreeNode(val=0, left=None, right=None)

tree_node(values):
  if empty(values): return None
  root = TreeNode(values[0])
  queue = [root]; i = 1
  while queue:
    node = queue.pop_front()
    if i < len(values) and values[i] != None:
      node.left = TreeNode(values[i]); queue.append(node.left)
    i += 1
    if i < len(values) and values[i] != None:
      node.right = TreeNode(values[i]); queue.append(node.right)
    i += 1
  return root

is_same_tree(p, q):
  if not p and not q: return True
  if not p or not q: return False
  if p.val != q.val: return False
  return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
  pathSum(root, targetSum):
    result = []
    dfs(node, currSum, path):
      if not node: return
      path.append(node.val)
      currSum += node.val
      if not node.left and not node.right and currSum == targetSum:
        result.append(copy(path))
      dfs(node.left, currSum, path)
      dfs(node.right, currSum, path)
      path.pop()
    dfs(root, 0, [])
    return result
