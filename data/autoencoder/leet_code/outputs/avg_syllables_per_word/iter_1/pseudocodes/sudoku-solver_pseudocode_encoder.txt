is_valid(b,r,c,n):
  return all(b[r][i] != n for i in 0..8) 
     and all(b[i][c] != n for i in 0..8)
     and all(b[3*(r//3)+i][3*(c//3)+j] != n for i,j in product(0..2,0..2))

solve(b):
  for r in 0..8:
    for c in 0..8:
      if b[r][c] == '.':
        for n in '1'..'9':
          if is_valid(b,r,c,n):
            b[r][c] = n
            if solve(b): return True
            b[r][c] = '.'
        return False
  return True
