fn LPS(s):
  n = len(s)
  dp = n x n zero matrix
  for i in 0..n-1: dp[i][i] = 1
  for cl in 2..n:
    for i in 0..n-cl:
      j = i+cl-1
      if s[i] = s[j]:
        dp[i][j] = 2 if cl=2 else dp[i+1][j-1]+2
      else:
        dp[i][j] = max(dp[i][j-1], dp[i+1][j])
  return dp[0][n-1]

minInsertions(s) = len(s) - LPS(s)
