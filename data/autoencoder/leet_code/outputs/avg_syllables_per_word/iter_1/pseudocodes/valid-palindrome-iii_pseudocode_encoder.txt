func longest_palindromic_subseq(s):
    n = length(s)
    dp = 2D array n×n filled with 0
    for i in 0 to n-1: dp[i][i] = 1
    for length in 2 to n:
        for i in 0 to n - length:
            j = i + length - 1
            if s[i] == s[j]:
                dp[i][j] = dp[i+1][j-1] + 2
            else:
                dp[i][j] = max(dp[i+1][j], dp[i][j-1])
    return dp[0][n-1]

func isValidPalindrome(s, k):
    lps = longest_palindromic_subseq(s)
    return length(s) - lps ≤ k
