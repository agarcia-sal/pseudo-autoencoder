class TreeNode(val=0, left=None, right=None)

func tree_node(values):
    if values empty: return None
    root = TreeNode(values[0])
    i = 1
    queue = deque(root)
    while queue and i < len(values):
        node = queue.popleft()
        if i < len(values) and values[i] != None:
            node.left = TreeNode(values[i])
            queue.append(node.left)
        i += 1
        if i < len(values) and values[i] != None:
            node.right = TreeNode(values[i])
            queue.append(node.right)
        i += 1
    return root

func is_same_tree(p, q):
    if not p and not q: return True
    if not p or not q or p.val != q.val: return False
    return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
    func widthOfBinaryTree(root):
        if not root: return 0
        queue = deque((root,0))
        max_w = 0
        while queue:
            level_len = len(queue)
            _, first_idx = queue[0]
            for i in 0 to level_len-1:
                node, idx = queue.popleft()
                if i == level_len-1:
                    max_w = max(max_w, idx - first_idx + 1)
                if node.left: queue.append((node.left, 2*idx))
                if node.right: queue.append((node.right, 2*idx + 1))
        return max_w
