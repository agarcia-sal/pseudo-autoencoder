func alienOrder(words):
    graph ← map char → set of chars
    indegree ← map char → int
    all_chars ← set of chars in all words

    for i in 0 to len(words)-2:
        w1, w2 ← words[i], words[i+1]
        found ← false
        for j in 0 to min(len(w1), len(w2))-1:
            if w1[j] ≠ w2[j]:
                if w2[j] ∉ graph[w1[j]]:
                    graph[w1[j]] add w2[j]
                    indegree[w2[j]] += 1
                found ← true
                break
        if not found and len(w1) > len(w2):
            return ""

    queue ← deque(chars in all_chars where indegree[char] = 0)
    result ← []

    while queue not empty:
        c ← queue.pop_front()
        result append c
        for n in graph[c]:
            indegree[n] -= 1
            if indegree[n] = 0:
                queue append n

    if len(result) ≠ len(all_chars):
        return ""

    return join(result)
