class TreeNode(val=0, left=None, right=None)

func tree_node(vals):
    if vals empty: return None
    root = TreeNode(vals[0])
    q = deque(root)
    i = 1
    while q and i < len(vals):
        node = q.popleft()
        if vals[i] != None:
            node.left = TreeNode(vals[i])
            q.append(node.left)
        i += 1
        if i < len(vals) and vals[i] != None:
            node.right = TreeNode(vals[i])
            q.append(node.right)
        i += 1
    return root

func is_same_tree(p, q):
    if not p and not q: return True
    if not p or not q: return False
    if p.val != q.val: return False
    return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
    func isValidBST(root):
        func validate(n, low=-∞, high=∞):
            if not n: return True
            if n.val ≤ low or n.val ≥ high: return False
            return validate(n.left, low, n.val) and validate(n.right, n.val, high)
        return validate(root)
