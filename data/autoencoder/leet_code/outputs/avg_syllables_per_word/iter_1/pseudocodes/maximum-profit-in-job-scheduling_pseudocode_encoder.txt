sort jobs by end time: jobs = sorted(zip(startTime, endTime, profit), key=end)
end_times = [end for _, end, _ in jobs]
dp = array of zeros with length len(jobs)+1

for i in 1 to len(jobs):
    j = bisect_right(end_times, jobs[i-1].start)
    dp[i] = max(dp[i-1], dp[j] + jobs[i-1].profit)

return dp[-1]
