```
set n = length of seats
set start = first index where seats[start] == 1
set end = last index where seats[end] == 1
set max_dist = max(start, n - end - 1)
set cur_dist = 0
for i in start to end:
    if seats[i] == 0:
        cur_dist += 1
        max_dist = max(max_dist, (cur_dist + 1) // 2)
    else:
        cur_dist = 0
return max_dist
```
