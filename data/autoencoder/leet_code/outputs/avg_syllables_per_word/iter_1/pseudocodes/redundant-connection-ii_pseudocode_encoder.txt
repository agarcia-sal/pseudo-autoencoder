function find(parent, i):
  while parent[i] != i:
    parent[i] = parent[parent[i]]
    i = parent[i]
  return i

function union(parent, rank, x, y):
  rx, ry = find(parent, x), find(parent, y)
  if rx == ry: return False
  if rank[rx] > rank[ry]:
    parent[ry] = rx
  else if rank[rx] < rank[ry]:
    parent[rx] = ry
  else:
    parent[ry] = rx
    rank[rx] += 1
  return True
