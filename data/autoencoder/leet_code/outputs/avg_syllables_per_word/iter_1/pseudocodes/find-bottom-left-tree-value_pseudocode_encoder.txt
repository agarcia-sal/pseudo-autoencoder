fn tree_node(vals):
    if empty(vals): return None
    root = new TreeNode(vals[0])
    q = queue(root)
    i = 1
    while q not empty:
        n = q.pop_front()
        if i < len(vals) and vals[i] != None:
            n.left = new TreeNode(vals[i]); q.push_back(n.left)
        i += 1
        if i < len(vals) and vals[i] != None:
            n.right = new TreeNode(vals[i]); q.push_back(n.right)
        i += 1
    return root

fn is_same_tree(p, q):
    if not p and not q: return True
    if not p or not q: return False
    if p.val != q.val: return False
    return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
    fn findBottomLeftValue(root):
        q = queue(root)
        leftmost_val = None
        while q not empty:
            size = q.size()
            for i in 0 to size-1:
                n = q.pop_front()
                if i == 0: leftmost_val = n.val
                if n.left: q.push_back(n.left)
                if n.right: q.push_back(n.right)
        return leftmost_val
