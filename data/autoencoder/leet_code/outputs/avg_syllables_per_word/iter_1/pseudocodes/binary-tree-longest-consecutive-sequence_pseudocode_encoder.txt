class TreeNode(val=0, left=None, right=None)

fn tree_node(vals):
  if empty(vals): return None
  root = TreeNode(vals[0])
  q = queue(root)
  i = 1
  while q:
    n = q.pop()
    if i < len(vals) and vals[i] != None:
      n.left = TreeNode(vals[i])
      q.push(n.left)
    i += 1
    if i < len(vals) and vals[i] != None:
      n.right = TreeNode(vals[i])
      q.push(n.right)
    i += 1
  return root

fn is_same_tree(p, q):
  if !p and !q: return True
  if !p or !q or p.val != q.val: return False
  return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
  fn longestConsecutive(root):
    fn dfs(n, p_val, length):
      if !n: return length
      length = length + 1 if p_val != None and n.val == p_val + 1 else 1
      left_len = dfs(n.left, n.val, length)
      right_len = dfs(n.right, n.val, length)
      return max(length, left_len, right_len)
    if !root: return 0
    return dfs(root, None, 0)
