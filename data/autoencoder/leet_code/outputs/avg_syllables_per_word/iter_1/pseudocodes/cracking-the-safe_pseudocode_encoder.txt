def crackSafe(n, k):
    visited = set()
    seq = []
    def dfs(node):
        for i in range(k):
            e = node + str(i)
            if e not in visited:
                visited.add(e)
                dfs(e[1:])
                seq.append(str(i))
    start = '0' * (n-1)
    dfs(start)
    return ''.join(seq) + start
