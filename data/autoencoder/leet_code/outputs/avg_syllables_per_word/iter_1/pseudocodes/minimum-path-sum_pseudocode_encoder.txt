if grid empty: return 0
m, n = dimensions of grid
dp = 2D array m√ón zeros
dp[0][0] = grid[0][0]
for j in 1 to n-1: dp[0][j] = dp[0][j-1] + grid[0][j]
for i in 1 to m-1: dp[i][0] = dp[i-1][0] + grid[i][0]
for i in 1 to m-1:
  for j in 1 to n-1:
    dp[i][j] = min(dp[i-1][j], dp[i][j-1]) + grid[i][j]
return dp[m-1][n-1]
