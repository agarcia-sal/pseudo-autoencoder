MOD = 10^9+7
n = len(s)
dp = 2D array (n+1 x n+1) filled with 0
for j in 0 to n: dp[0][j] = 1

for i in 1 to n:
  prefix_sum = 0
  if s[i-1] == 'I':
    for j in 0 to n - i:
      prefix_sum = (prefix_sum + dp[i-1][j]) % MOD
      dp[i][j] = prefix_sum
  else:
    for j in n - i down to 0:
      prefix_sum = (prefix_sum + dp[i-1][j+1]) % MOD
      dp[i][j] = prefix_sum

return sum(dp[n]) % MOD
