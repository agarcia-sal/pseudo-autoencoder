class TreeNode(val=0, left=None, right=None)

func tree_node(vals):
    if empty(vals): return None
    root = TreeNode(vals[0])
    q = deque([root])
    i = 1
    while q and i < len(vals):
        n = q.popleft()
        if i < len(vals) and vals[i] != None:
            n.left = TreeNode(vals[i])
            q.append(n.left)
        i += 1
        if i < len(vals) and vals[i] != None:
            n.right = TreeNode(vals[i])
            q.append(n.right)
        i += 1
    return root

func is_same_tree(p, q):
    if not p and not q: return True
    if not p or not q or p.val != q.val: return False
    return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
    func verticalOrder(root):
        if not root: return []
        cols = defaultdict(list)
        q = deque([(root, 0)])
        min_c = max_c = 0
        while q:
            n, c = q.popleft()
            if n:
                cols[c].append(n.val)
                min_c = min(min_c, c)
                max_c = max(max_c, c)
                q.append((n.left, c - 1))
                q.append((n.right, c + 1))
        return [cols[i] for i in range(min_c, max_c + 1)]
