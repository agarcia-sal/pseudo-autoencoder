dp[0][0] = True
for j in 1 to len(p):
  if p[j-1] == '*': dp[0][j] = dp[0][j-2]
for i in 1 to len(s):
  for j in 1 to len(p):
    if p[j-1] == s[i-1] or p[j-1] == '.': dp[i][j] = dp[i-1][j-1]
    else if p[j-1] == '*':
      dp[i][j] = dp[i][j-2]
      if p[j-2] == s[i-1] or p[j-2] == '.': dp[i][j] = dp[i][j] or dp[i-1][j]
return dp[len(s)][len(p)]
