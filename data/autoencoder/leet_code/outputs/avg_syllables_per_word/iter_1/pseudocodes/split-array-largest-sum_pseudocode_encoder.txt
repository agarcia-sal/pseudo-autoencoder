func splitArray(nums, k):
    canSplit(maxSum):
        sum, subs = 0, 1
        for n in nums:
            if sum + n > maxSum: subs++, sum = n
            else: sum += n
        return subs <= k

    l, r = max(nums), sum(nums)
    while l < r:
        m = (l + r) // 2
        if canSplit(m): r = m
        else: l = m + 1
    return l
