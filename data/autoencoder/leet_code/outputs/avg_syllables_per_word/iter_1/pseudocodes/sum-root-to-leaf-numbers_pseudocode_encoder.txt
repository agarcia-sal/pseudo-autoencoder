class TreeNode(val=0, left=null, right=null)

func tree_node(vals):
    if empty(vals): return null
    root = TreeNode(vals[0])
    q = queue(root)
    i = 1
    while q not empty:
        n = q.pop()
        if i < len(vals) and vals[i] not null:
            n.left = TreeNode(vals[i])
            q.push(n.left)
        i += 1
        if i < len(vals) and vals[i] not null:
            n.right = TreeNode(vals[i])
            q.push(n.right)
        i += 1
    return root

func is_same_tree(p, q):
    if p null and q null: return True
    if p null or q null: return False
    if p.val != q.val: return False
    return is_same_tree(p.left, q.left) and is_same_tree(p.right, q.right)

class Solution:
    func sumNumbers(root):
        func dfs(n, cur):
            if n null: return 0
            cur = cur * 10 + n.val
            if n.left null and n.right null: return cur
            return dfs(n.left, cur) + dfs(n.right, cur)
        return dfs(root, 0)
