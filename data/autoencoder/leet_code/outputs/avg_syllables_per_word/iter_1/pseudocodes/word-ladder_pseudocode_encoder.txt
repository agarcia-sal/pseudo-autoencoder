queue ← deque([(beginWord, 1)])
visited ← {beginWord}

while queue:
  cur, dist ← queue.popleft()
  for i in [0..word_len-1]:
    inter ← cur[:i]+"*"+cur[i+1:]
    for w in word_dict[inter]:
      if w = endWord: return dist+1
      if w ∉ visited:
        visited.add(w)
        queue.append((w, dist+1))

return 0
