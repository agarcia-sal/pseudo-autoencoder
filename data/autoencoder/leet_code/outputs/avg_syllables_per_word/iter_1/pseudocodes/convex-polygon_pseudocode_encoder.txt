isConvex(points):
  n = length(points)
  if n ≤ 3: return True
  orientation = 0
  for i in 0 to n-1:
    o, a, b = points[i], points[(i+1)%n], points[(i+2)%n]
    cross = (a.x - o.x)*(b.y - o.y) - (a.y - o.y)*(b.x - o.x)
    if cross ≠ 0:
      if orientation = 0: orientation = 1 if cross > 0 else -1
      else if (orientation=1 and cross < 0) or (orientation=-1 and cross > 0): return False
  return True
