n = len(nums)
delta = [0]*n

for i, x in enumerate(nums):
    if x <= i:
        delta[0] += 1
        if i - x + 1 < n: delta[i - x + 1] -= 1
    if i + 1 < n:
        delta[i + 1] += 1
        end = (i + n - x) + 1
        if end < n: delta[end] -= 1

score = max_score = best_k = 0
for k in range(n):
    score += delta[k]
    if score > max_score:
        max_score, best_k = score, k

return best_k
