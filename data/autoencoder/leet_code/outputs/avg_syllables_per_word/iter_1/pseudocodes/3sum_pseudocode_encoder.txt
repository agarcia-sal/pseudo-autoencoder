res = []  
for i in 0 to len(nums)-3:  
 if i > 0 and nums[i] == nums[i-1]: continue  
 L, R = i+1, len(nums)-1  
 while L < R:  
  sum = nums[i] + nums[L] + nums[R]  
  if sum == 0:  
   res.append([nums[i], nums[L], nums[R]])  
   while L < R and nums[L] == nums[L+1]: L += 1  
   while L < R and nums[R] == nums[R-1]: R -= 1  
   L += 1; R -= 1  
  elif sum < 0: L += 1  
  else: R -= 1  
return res
