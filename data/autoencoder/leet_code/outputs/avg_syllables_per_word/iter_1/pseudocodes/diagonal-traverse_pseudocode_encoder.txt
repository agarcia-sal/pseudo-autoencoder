```
if empty matrix return []
m, n = dims of mat
r, c, dir = 0, 0, 1
res = []

for _ in 0 to m*n-1:
    append mat[r][c] to res
    nr, nc = r + (-1 if dir=1 else 1), c + (1 if dir=1 else -1)
    
    if nr<0 or nr==m or nc<0 or nc==n:
        if dir=1:
            if nc==n: nr, nc = r+1, c
            else:      nr, nc = 0, c+1
        else:
            if nr==m: nr, nc = r, c+1
            else:      nr, nc = r+1, 0
        dir = -dir
    r, c = nr, nc

return res
```
