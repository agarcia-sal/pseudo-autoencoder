CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty collection  
        RETURN none  
    END IF  
    SET head TO new ListNode with val as the element at position zero OF values and next as none  
    SET p TO head  
    FOR each val IN the elements of values from position one TO the end  
        SET node TO new ListNode with val as val and next as none  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the condition that the val attribute of p1 EQUALS the val attribute of p2 AND is_same_list called with p1.next and p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION insertionSortList(self, head)  
        IF head EQUALS none OR the next attribute of head EQUALS none  
            RETURN head  
        END IF  
        SET dummy TO new ListNode with val as integer negative five thousand one and next as none  
        SET dummy.next TO head  
        SET current TO the next attribute of head  
        SET head.next TO none  
        WHILE current is not none  
            SET next_node TO the next attribute of current  
            SET prev TO dummy  
            WHILE the next attribute of prev is not none AND the val attribute of the next attribute of prev is LESS THAN the val attribute of current  
                SET prev TO the next attribute of prev  
            END WHILE  
            SET the next attribute of current TO the next attribute of prev  
            SET the next attribute of prev TO current  
            SET current TO next_node  
        END WHILE  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
