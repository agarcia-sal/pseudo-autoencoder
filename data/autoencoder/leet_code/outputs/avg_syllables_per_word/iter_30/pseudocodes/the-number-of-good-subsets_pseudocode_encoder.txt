CLASS Solution  
    FUNCTION numberOfGoodSubsets(nums)  
        SET count TO frequency mapping of elements in nums  
        SET dp TO list of zero repeated two raised to the power ten times  
        SET element at position zero OF dp TO one  
        FOR each num IN count  
            IF num EQUALS one  
                SET element at position zero OF dp TO element at position zero OF dp MULTIPLIED BY two raised to the power value at key one IN count  
                CONTINUE  
            END IF  
            IF num NOT IN prime_factors  
                CONTINUE  
            END IF  
            SET factors TO prime_factors at key num  
            SET mask TO zero  
            FOR each prime IN factors  
                SET mask TO mask PLUS the value of two raised to the power of prime minus one  
            END FOR  
            FOR m FROM two raised to the power ten MINUS one DOWN TO zero  
                IF the bitwise AND of m AND mask EQUALS zero  
                    SET element at position of the bitwise OR of m AND mask OF dp TO the sum of element at position of the bitwise OR of m AND mask OF dp PLUS element at position m OF dp MULTIPLIED BY value at key num IN count  
                END IF  
            END FOR  
        END FOR  
        SET result TO the sum of elements from position one TO the end OF dp  
        RETURN the remainder of result divided by MOD  
    END FUNCTION  
END CLASS
