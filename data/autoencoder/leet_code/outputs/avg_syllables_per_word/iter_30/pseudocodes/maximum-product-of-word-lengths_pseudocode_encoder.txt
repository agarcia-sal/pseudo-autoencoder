CLASS Solution  
    FUNCTION maxProduct(words)  
        SET word_sets TO empty list  
        FOR each word IN words  
            APPEND set of characters in word TO word_sets  
        END FOR  
        SET max_product TO zero  
        FOR index i FROM zero TO LENGTH OF words MINUS one  
            FOR index j FROM i PLUS one TO LENGTH OF words MINUS one  
                IF the intersection of element at position i OF word_sets AND element at position j OF word_sets EQUALS empty set  
                    SET product TO LENGTH OF element at position i OF words MULTIPLIED BY LENGTH OF element at position j OF words  
                    IF product GREATER THAN max_product  
                        SET max_product TO product  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_product  
    END FUNCTION  
END CLASS
