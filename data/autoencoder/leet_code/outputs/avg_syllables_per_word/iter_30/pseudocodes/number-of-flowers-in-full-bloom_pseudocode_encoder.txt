CLASS Solution  
    FUNCTION fullBloomFlowers(flowers, people)  
        SET start_times TO the sorted list of start time elements extracted from flowers  
        SET end_times TO the sorted list of end time elements extracted from flowers  
        SET result TO an empty list  
        FOR each person IN people  
            SET started TO the position found by bisect right search of person in start_times  
            SET ended TO the position found by bisect left search of person in end_times  
            APPEND the difference of started MINUS ended TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
