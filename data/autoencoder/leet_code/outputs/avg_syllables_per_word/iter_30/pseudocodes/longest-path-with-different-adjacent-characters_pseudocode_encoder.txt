CLASS Solution  
    FUNCTION longestPath(parent, s)  
        CREATE tree AS default mapping from key to empty list  
        FOR each index i FROM zero TO the LENGTH OF parent MINUS one  
            SET p TO element at position i OF parent  
            IF p NOT EQUALS negative one  
                APPEND i TO the list at key p IN tree  
            END IF  
        END FOR  
        SET self dot result TO one  
        FUNCTION dfs(node)  
            SET max1 TO zero  
            SET max2 TO zero  
            FOR each child IN the list at key node IN tree  
                SET child_length TO the CALL of dfs WITH argument child  
                IF element at position child OF s NOT EQUALS element at position node OF s  
                    IF child_length GREATER THAN max1  
                        SET max2 TO max1  
                        SET max1 TO child_length  
                    ELSE IF child_length GREATER THAN max2  
                        SET max2 TO child_length  
                    END IF  
                END IF  
            END FOR  
            SET self dot result TO the maximum BETWEEN self dot result AND the sum of max1 PLUS max2 PLUS one  
            RETURN max1 PLUS one  
        END FUNCTION  
        CALL dfs WITH argument zero  
        RETURN self dot result  
    END FUNCTION  
END CLASS
