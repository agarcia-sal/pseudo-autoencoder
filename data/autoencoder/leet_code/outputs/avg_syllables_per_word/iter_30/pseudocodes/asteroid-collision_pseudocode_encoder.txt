CLASS Solution  
    FUNCTION asteroidCollision WITH PARAMETER asteroids  
        SET stack TO empty list  
        FOR each asteroid IN asteroids  
            WHILE the LENGTH OF stack is greater than zero AND asteroid is LESS THAN zero AND element at position LENGTH OF stack MINUS one OF stack is GREATER THAN zero  
                IF element at position LENGTH OF stack MINUS one OF stack is LESS THAN negative asteroid  
                    REMOVE element at position LENGTH OF stack MINUS one FROM stack  
                    CONTINUE  
                ELSE IF element at position LENGTH OF stack MINUS one OF stack EQUALS negative asteroid  
                    REMOVE element at position LENGTH OF stack MINUS one FROM stack  
                END IF  
                BREAK  
            END WHILE  
            ELSE  
                APPEND asteroid TO stack  
            END ELSE  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
