CLASS Solution  
    FUNCTION longestPalindrome(s)  
        SET n TO the LENGTH OF s  
        SET f TO a two dimensional collection of TRUE values with dimensions n by n  
        SET k TO zero  
        SET mx TO one  
        FOR i FROM n MINUS two DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET f at position i and position j TO FALSE  
                IF the character at position i OF s EQUALS the character at position j OF s  
                    SET f at position i and position j TO the value of f at position i PLUS one and position j MINUS one  
                    IF f at position i and position j AND mx LESS THAN j MINUS i PLUS one  
                        SET k TO i  
                        SET mx TO j MINUS i PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the substring from position k TO position k PLUS mx OF s  
    END FUNCTION  
END CLASS
