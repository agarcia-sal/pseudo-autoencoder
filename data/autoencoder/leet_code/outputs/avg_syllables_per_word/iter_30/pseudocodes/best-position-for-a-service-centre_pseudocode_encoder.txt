CLASS Solution  
    FUNCTION getMinDistSum(positions)  
        SET n TO the LENGTH OF positions  
        SET x TO zero  
        SET y TO zero  
        FOR each position IN positions  
            EXTRACT first element OF position TO x1  
            EXTRACT second element OF position TO y1  
            INCREMENT x BY x1  
            INCREMENT y BY y1  
        END FOR  
        SET x TO x DIVIDED BY n  
        SET y TO y DIVIDED BY n  
        SET decay TO zero point nine nine nine  
        SET eps TO zero point zero zero zero zero zero one  
        SET alpha TO zero point five  
        WHILE true holds  
            SET grad_x TO zero  
            SET grad_y TO zero  
            SET dist TO zero  
            FOR each position IN positions  
                EXTRACT first element OF position TO x1  
                EXTRACT second element OF position TO y1  
                SET a TO x MINUS x1  
                SET b TO y MINUS y1  
                SET c TO the square root OF a MULTIPLIED BY a PLUS b MULTIPLIED BY b  
                INCREMENT grad_x BY a DIVIDED BY c PLUS zero point zero zero zero zero zero zero one  
                INCREMENT grad_y BY b DIVIDED BY c PLUS zero point zero zero zero zero zero zero one  
                INCREMENT dist BY c  
            END FOR  
            SET dx TO grad_x MULTIPLIED BY alpha  
            SET dy TO grad_y MULTIPLIED BY alpha  
            DECREMENT x BY dx  
            DECREMENT y BY dy  
            MULTIPLY alpha BY decay  
            IF the absolute value of dx IS LESS THAN OR EQUAL TO eps AND the absolute value of dy IS LESS THAN OR EQUAL TO eps THEN  
                RETURN dist  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
