CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO a two dimensional list with the number of rows equal to the length of s plus one and the number of columns equal to the length of p plus one filled with false  
        SET the element at position zero zero OF dp TO true  
        FOR j FROM one TO the length of p  
            IF the element at position j minus one OF p EQUALS the character star  
                SET the element at position zero j OF dp TO the element at position zero j minus two OF dp  
            END IF  
        END FOR  
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF the element at position j minus one OF p EQUALS the element at position i minus one OF s OR the element at position j minus one OF p EQUALS the character dot  
                    SET the element at position i j OF dp TO the element at position i minus one j minus one OF dp  
                ELSE IF the element at position j minus one OF p EQUALS the character star  
                    SET the element at position i j OF dp TO the element at position i j minus two OF dp  
                    IF the element at position j minus two OF p EQUALS the element at position i minus one OF s OR the element at position j minus two OF p EQUALS the character dot  
                        SET the element at position i j OF dp TO the logical OR of the element at position i j OF dp AND the element at position i minus one j OF dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position the length of s the length of p OF dp  
    END FUNCTION  
END CLASS
