CLASS Solution  
    FUNCTION largestDivisibleSubset(nums)  
        SORT nums IN ascending order  
        SET n TO the LENGTH OF nums  
        SET f TO a list of ones with LENGTH n  
        SET k TO zero  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO i MINUS one  
                IF the element at position i OF nums MODULO the element at position j OF nums EQUALS zero THEN  
                    SET f at position i TO the GREATER OF f at position i AND f at position j PLUS one  
                END IF  
            END FOR  
            IF f at position k LESS THAN f at position i THEN  
                SET k TO i  
            END IF  
        END FOR  
        SET m TO f at position k  
        SET i TO k  
        SET ans TO an empty list  
        WHILE m GREATER THAN zero  
            IF the element at position k OF nums MODULO the element at position i OF nums EQUALS zero AND f at position i EQUALS m THEN  
                APPEND the element at position i OF nums TO ans  
                SET k TO i  
                DECREMENT m BY one  
            END IF  
            DECREMENT i BY one  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
