CLASS Solution  
    FUNCTION leastOpsExpressTarget(x, target)  
        FUNCTION dp(target_value)  
            IF target_value EQUALS zero  
                RETURN minus one  
            END IF  
            IF target_value LESS THAN x  
                RETURN the MINIMUM OF the quantity two MULTIPLIED BY target_value MINUS one AND the quantity two MULTIPLIED BY x MINUS target_value  
            END IF  
            SET power TO one  
            WHILE x RAISED TO THE power LESS THAN OR EQUAL TO target_value holds  
                INCREMENT power BY one  
            END WHILE  
            DECREMENT power BY one  
            SET result TO dp(target_value MINUS x RAISED TO THE power) PLUS power  
            IF x RAISED TO THE quantity power PLUS one MINUS target_value LESS THAN target_value  
                SET result TO the MINIMUM OF result AND dp(x RAISED TO THE quantity power PLUS one MINUS target_value) PLUS power PLUS one  
            END IF  
            RETURN result  
        END FUNCTION  
        RETURN dp(target)  
    END FUNCTION  
END CLASS
