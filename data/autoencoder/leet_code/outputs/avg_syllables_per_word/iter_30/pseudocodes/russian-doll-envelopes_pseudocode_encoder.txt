CLASS Solution  
    FUNCTION maxEnvelopes(envelopes)  
        SORT envelopes BY width IN ASCENDING ORDER AND BY height IN DESCENDING ORDER  
        SET dp TO empty list  
        FOR each pair consisting of unused_variable AND height IN envelopes  
            SET idx TO the smallest position in dp WHERE height COULD BE INSERTED TO MAINTAIN ORDER  
            IF idx EQUALS the LENGTH OF dp  
                APPEND height TO dp  
            ELSE  
                SET element at position idx OF dp TO height  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
