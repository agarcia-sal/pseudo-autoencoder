CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with value at position zero OF values  
    SET p TO head  
    FOR each val IN the elements of values from position one TO the end  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN the comparison of p1.val EQUALS p2.val AND is_same_list with arguments p1.next and p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION sortList(self, head)  
        IF head IS none OR head.next IS none  
            RETURN head  
        END IF  
      
        FUNCTION split(head)  
            SET slow TO head  
            SET fast TO the element at position one OF head.next  
            WHILE fast IS NOT none AND fast.next IS NOT none  
                SET slow TO slow.next  
                SET fast TO fast.next.next  
            END WHILE  
            SET mid TO slow.next  
            SET slow.next TO none  
            RETURN head AND mid  
        END FUNCTION  
      
        FUNCTION merge(l1, l2)  
            SET dummy TO new ListNode  
            SET current TO dummy  
            WHILE l1 IS NOT none AND l2 IS NOT none  
                IF l1.val IS LESS THAN l2.val  
                    SET current.next TO l1  
                    SET l1 TO l1.next  
                ELSE  
                    SET current.next TO l2  
                    SET l2 TO l2.next  
                END IF  
                SET current TO current.next  
            END WHILE  
            SET current.next TO l1 IF l1 IS NOT none OTHERWISE l2  
            RETURN dummy.next  
        END FUNCTION  
      
        SET left AND right TO the result of split with argument head  
        SET left TO self.sortList with argument left  
        SET right TO self.sortList with argument right  
      
        RETURN merge with arguments left and right  
    END FUNCTION  
END CLASS
