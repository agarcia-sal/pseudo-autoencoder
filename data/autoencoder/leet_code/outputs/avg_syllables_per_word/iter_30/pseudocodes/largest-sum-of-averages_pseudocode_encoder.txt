CLASS Solution  
    FUNCTION largestSumOfAverages(nums, k)  
        SET n TO the LENGTH OF nums  
        SET prefix_sum TO a list of zero repeated n PLUS one times  
        FOR i FROM zero TO n MINUS one  
            SET element at position i PLUS one OF prefix_sum TO element at position i OF prefix_sum PLUS element at position i OF nums  
        END FOR  

        FUNCTION average(i, j)  
            RETURN the VALUE of element at position j PLUS one OF prefix_sum MINUS element at position i OF prefix_sum DIVIDED BY j MINUS i PLUS one  
        END FUNCTION  

        SET dp TO a list containing n PLUS one elements each being a list of zero repeated k PLUS one times  
        FOR i FROM zero TO n MINUS one  
            SET element at position i AND one OF dp TO the VALUE returned BY average(i, n MINUS one)  
        END FOR  

        FOR i FROM n MINUS one DOWN TO zero  
            FOR partitions FROM two TO k  
                FOR j FROM i PLUS one TO n MINUS one  
                    SET current_value TO element at position i AND partitions OF dp  
                    SET candidate_value TO element at position j AND partitions MINUS one OF dp PLUS the VALUE returned BY average(i, j MINUS one)  
                    IF candidate_value GREATER THAN current_value  
                        SET element at position i AND partitions OF dp TO candidate_value  
                    END IF  
                END FOR  
            END FOR  
        END FOR  

        RETURN element at position zero AND k OF dp  
    END FUNCTION  
END CLASS
