CLASS Solution  
    FUNCTION reconstructQueue(people)  
        CALL sort ON people WITH key DEFINED AS FUNCTION x  
            RETURN NEGATIVE of element at position zero OF x AND element at position one OF x  
        END FUNCTION  
        SET queue TO empty list  
        FOR each person IN people  
            INSERT person AT position element at position one OF person IN queue  
        END FOR  
        RETURN queue  
    END FUNCTION  
END CLASS
