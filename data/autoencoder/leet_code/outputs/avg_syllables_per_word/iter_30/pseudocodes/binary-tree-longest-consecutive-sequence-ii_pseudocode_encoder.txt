CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with val equal to element at position zero OF values  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from the front of queue  
        IF i LESS THAN length OF values AND element at position i OF values IS NOT None  
            SET node.left TO new TreeNode with val equal to element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length OF values AND element at position i OF values IS NOT None  
            SET node.right TO new TreeNode with val equal to element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree with arguments p.left and q.left AND is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION longestConsecutive(self, root)  
        FUNCTION dfs(node)  
            IF node IS None  
                RETURN zero, zero  
            END IF  
            SET inc_len TO one  
            SET dec_len TO one  
            SET left_inc, left_dec TO dfs with argument node.left  
            SET right_inc, right_dec TO dfs with argument node.right  
            IF node.left IS NOT None  
                IF node.val EQUALS node.left.val PLUS one  
                    SET dec_len TO left_dec PLUS one  
                ELSE IF node.val EQUALS node.left.val MINUS one  
                    SET inc_len TO left_inc PLUS one  
                END IF  
            END IF  
            IF node.right IS NOT None  
                IF node.val EQUALS node.right.val PLUS one  
                    SET dec_len TO the greater value BETWEEN dec_len AND right_dec PLUS one  
                ELSE IF node.val EQUALS node.right.val MINUS one  
                    SET inc_len TO the greater value BETWEEN inc_len AND right_inc PLUS one  
                END IF  
            END IF  
            SET max_len TO the greater value BETWEEN max_len AND inc_len PLUS dec_len MINUS one  
            RETURN inc_len, dec_len  
        END FUNCTION  
        SET max_len TO zero  
        CALL dfs with argument root  
        RETURN max_len  
    END FUNCTION  
END CLASS
