CLASS Solution  
    FUNCTION minimumOneBitOperations(n)  
        IF n EQUALS zero  
            RETURN zero  
        END IF  
        SET k TO zero  
        WHILE two RAISED TO the power of k PLUS one LESS THAN OR EQUAL TO n  
            INCREMENT k BY one  
        END WHILE  
        SET most_significant_bit TO two RAISED TO the power of k  
        RETURN two RAISED TO the power of k PLUS one MINUS one MINUS the result of calling minimumOneBitOperations with n EXCLUSIVE OR most_significant_bit  
    END FUNCTION  
END CLASS
