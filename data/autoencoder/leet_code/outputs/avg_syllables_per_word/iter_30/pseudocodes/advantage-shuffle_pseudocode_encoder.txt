CLASS Solution  
    FUNCTION advantageCount(nums1, nums2)  
        SORT nums1 IN ASCENDING ORDER  
        SET nums2_with_indices TO list of pairs of each value and its position IN nums2 SORTED IN DESCENDING ORDER BY value  
        SET result TO list of zeros with the LENGTH OF nums1  
        SET left TO zero  
        SET right TO LENGTH OF nums1 MINUS one  
        FOR each value and index IN nums2_with_indices  
            IF element at position right OF nums1 GREATER THAN value  
                SET element at position index OF result TO element at position right OF nums1  
                DECREMENT right BY one  
            ELSE  
                SET element at position index OF result TO element at position left OF nums1  
                INCREMENT left BY one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
