CLASS Solution  
    FUNCTION smallestSubsequence(s, k, letter, repetition)  
        SET letter_count TO the number of times letter occurs in s  
        SET stack TO an empty list  
        FOR each index i AND character char IN s  
            WHILE stack is not empty AND character char is less than the element at the last position of stack AND the length of s minus i plus the length of stack minus one is greater than or equals to k  
                IF the element at the last position of stack EQUALS letter  
                    IF letter_count is greater than repetition  
                        INCREMENT repetition BY one  
                        REMOVE element at the last position FROM stack  
                    ELSE  
                        BREAK  
                ELSE  
                    REMOVE element at the last position FROM stack  
            END WHILE  
            IF the length of stack is less than k  
                IF character char EQUALS letter  
                    APPEND char TO stack  
                    DECREMENT repetition BY one  
                ELSE IF the length of stack subtracted from k is greater than repetition  
                    APPEND char TO stack  
                END IF  
                IF character char EQUALS letter  
                    DECREMENT letter_count BY one  
                END IF  
            END IF  
        END FOR  
        WHILE the length of stack is less than k  
            APPEND letter TO stack  
        END WHILE  
        RETURN the concatenation of all elements in stack as a single string  
    END FUNCTION  
END CLASS
