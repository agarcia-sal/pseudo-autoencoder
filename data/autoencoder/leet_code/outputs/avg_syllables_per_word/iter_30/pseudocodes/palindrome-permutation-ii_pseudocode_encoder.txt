CLASS Solution  
    FUNCTION generatePalindromes WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER t  
            IF the LENGTH OF t EQUALS the LENGTH OF s  
                APPEND t TO ans  
                RETURN  
            END IF  
            FOR each character c AND value v IN cnt  
                IF v GREATER THAN one  
                    DECREMENT cnt at character c BY two  
                    CALL dfs WITH ARGUMENT character c CONCATENATED WITH t CONCATENATED WITH character c  
                    INCREMENT cnt at character c BY two  
                END IF  
            END FOR  
        END FUNCTION  
        ASSIGN reference TO the COUNTER of string s TO cnt  
        SET mid TO empty string  
        FOR each character c AND value v IN cnt  
            IF value v MODULO two EQUALS one  
                IF mid IS NOT empty string  
                    RETURN empty list  
                END IF  
                SET mid TO character c  
                DECREMENT cnt at character c BY one  
            END IF  
        END FOR  
        SET ans TO empty list  
        CALL dfs WITH ARGUMENT mid  
        RETURN ans  
    END FUNCTION  
END CLASS
