CLASS Solution  
    FUNCTION canMouseWin(grid, catJump, mouseJump)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF the element at position zero OF grid  
        SET cat_start TO zero  
        SET mouse_start TO zero  
        SET food TO zero  
        SET dirs TO the SEQUENCE of negative one, zero, one, zero, negative one  
        SET g_mouse TO a list of empty lists with the LENGTH OF m MULTIPLIED BY n  
        SET g_cat TO a list of empty lists with the LENGTH OF m MULTIPLIED BY n  
        FOR each i, row IN enumerating grid  
            FOR each j, c IN enumerating row  
                IF c EQUALS the symbol for wall  
                    CONTINUE  
                END IF  
                SET v TO i MULTIPLIED BY n PLUS j  
                IF c EQUALS the symbol for cat  
                    SET cat_start TO v  
                ELSE IF c EQUALS the symbol for mouse  
                    SET mouse_start TO v  
                ELSE IF c EQUALS the symbol for food  
                    SET food TO v  
                END IF  
                FOR each adjacent pair a, b IN pairwise of dirs  
                    FOR each k FROM zero TO mouseJump inclusive  
                        SET x TO i PLUS k MULTIPLIED BY a  
                        SET y TO j PLUS k MULTIPLIED BY b  
                        IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of grid EQUALS the symbol for wall  
                            BREAK  
                        END IF  
                        APPEND x MULTIPLIED BY n PLUS y TO element at position v OF g_mouse  
                    END FOR  
                    FOR each k FROM zero TO catJump inclusive  
                        SET x TO i PLUS k MULTIPLIED BY a  
                        SET y TO j PLUS k MULTIPLIED BY b  
                        IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x of grid EQUALS the symbol for wall  
                            BREAK  
                        END IF  
                        APPEND x MULTIPLIED BY n PLUS y TO element at position v OF g_cat  
                    END FOR  
                END FOR  
            END FOR  
        END FOR  
        RETURN the result of CALLING calc WITH g_mouse, g_cat, mouse_start, cat_start, food EQUALS one  
    END FUNCTION  

    FUNCTION calc(g_mouse, g_cat, mouse_start, cat_start, hole)  
        FUNCTION get_prev_states(state)  
            SET m TO element at position zero OF state  
            SET c TO element at position one OF state  
            SET t TO element at position two OF state  
            SET pt TO the bitwise exclusive OR of t WITH one  
            SET pre TO empty list  
            IF pt EQUALS one  
                FOR each pc IN element at position c OF g_cat  
                    IF element at position m of ans AT position pc AT position one EQUALS zero  
                        APPEND the tuple of m, pc, pt TO pre  
                    END IF  
                END FOR  
            ELSE  
                FOR each pm IN element at position m OF g_mouse  
                    IF element at position pm of ans AT position c AT position zero EQUALS zero  
                        APPEND the tuple of pm, c, zero TO pre  
                    END IF  
                END FOR  
            END IF  
            RETURN pre  
        END FUNCTION  

        SET n TO the LENGTH OF g_mouse  
        SET degree TO a three-dimensional list sized n BY n BY two filled with zeros  
        FOR each i FROM zero TO n MINUS one  
            FOR each j FROM zero TO n MINUS one  
                SET element at position i OF element at position j OF degree AT position zero TO the LENGTH OF element at position i OF g_mouse  
                SET element at position i OF element at position j OF degree AT position one TO the LENGTH OF element at position j OF g_cat  
            END FOR  
        END FOR  
        SET ans TO a three-dimensional list sized n BY n BY two filled with zeros  
        SET q TO an empty double-ended queue  
        FOR each i FROM zero TO n MINUS one  
            SET element at position hole OF element at position i OF ans AT position one TO one  
            SET element at position i OF element at position hole OF ans AT position zero TO two  
            SET element at position i OF element at position i OF ans AT position one TO two  
            SET element at position i OF element at position i OF ans AT position zero TO two  
            APPEND the tuple of hole, i, one TO q  
            APPEND the tuple of i, hole, zero TO q  
            APPEND the tuple of i, i, zero TO q  
            APPEND the tuple of i, i, one TO q  
        END FOR  
        WHILE q is not empty  
            SET state TO the element REMOVED from the left of q  
            SET t TO element at position zero OF element at position one OF element at position two OF ans USING indices from state  
            FOR each prev_state IN get_prev_states WITH state  
                SET pm TO element at position zero OF prev_state  
                SET pc TO element at position one OF prev_state  
                SET pt TO element at position two OF prev_state  
                IF pt EQUALS t MINUS one  
                    SET element at position pm OF element at position pc OF ans AT position pt TO t  
                    APPEND prev_state TO q  
                ELSE  
                    DECREMENT element at position pm OF element at position pc OF degree AT position pt BY one  
                    IF element at position pm OF element at position pc OF degree AT position pt EQUALS zero  
                        SET element at position pm OF element at position pc OF ans AT position pt TO t  
                        APPEND prev_state TO q  
                    END IF  
                END IF  
            END FOR  
        END WHILE  
        RETURN element at position mouse_start OF element at position cat_start OF ans AT position zero  
    END FUNCTION  
END CLASS
