CLASS Solution  
    FUNCTION findLongestWord WITH PARAMETERS s, dictionary  
        FUNCTION is_subsequence WITH PARAMETERS word, s  
            SET iterator TO sequence iterator OVER s  
            RETURN boolean value WHETHER every character IN word EXISTS in iterator IN ORDER  
        END FUNCTION  
        
        SORT dictionary BY length in descending order AND lexicographical order in ascending order  
        
        FOR each word IN dictionary  
            IF is_subsequence OF word AND s EQUALS true  
                RETURN word  
            END IF  
        END FOR  
        
        RETURN empty string  
    END FUNCTION  
END CLASS
