CLASS Solution  
    FUNCTION leastInterval(tasks, n)  
        SET task_counts TO count of each task in tasks  
        SET max_heap TO list of negative task count values from task_counts  
        CONVERT max_heap to heap structure  
        SET time TO zero  
        SET queue TO empty list  
        WHILE max_heap is not empty OR queue is not empty  
            WHILE queue is not empty AND first element at position zero in queue at position zero EQUALS time  
                REMOVE first element from queue and SET to temporary tuple  
                SET count TO second element of temporary tuple  
                IF count LESS THAN zero  
                    ADD count TO max_heap with heap property  
                END IF  
            END WHILE  
            IF max_heap is not empty  
                REMOVE smallest element from max_heap and SET to count  
                INCREMENT count BY one  
                IF count LESS THAN zero  
                    APPEND tuple of time PLUS n PLUS one and count TO queue  
                END IF  
            END IF  
            INCREMENT time BY one  
        END WHILE  
        RETURN time  
    END FUNCTION  
END CLASS
