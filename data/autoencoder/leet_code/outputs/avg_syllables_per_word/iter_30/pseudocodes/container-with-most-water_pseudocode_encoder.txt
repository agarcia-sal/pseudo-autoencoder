CLASS Solution  
    FUNCTION maxArea(height)  
        SET left TO zero  
        SET right TO the LENGTH OF height MINUS one  
        SET max_area TO zero  
        WHILE left LESS THAN right  
            SET current_area TO the MINIMUM OF element at position left OF height AND element at position right OF height MULTIPLIED BY right MINUS left  
            SET max_area TO the MAXIMUM OF max_area AND current_area  
            IF element at position left OF height LESS THAN element at position right OF height  
                INCREMENT left BY one  
            ELSE  
                DECREMENT right BY one  
            END IF  
        END WHILE  
        RETURN max_area  
    END FUNCTION  
END CLASS
