CLASS Solution  
    FUNCTION validIPAddress(queryIP)  
        FUNCTION is_valid_ipv4_block(block)  
            IF block CONSISTS OF DIGITS ONLY  
                SET num TO integer CONVERSION OF block  
                IF num LESS THAN OR EQUAL TO two hundred fifty-five AND block EQUALS string CONVERSION OF num  
                    RETURN true  
                ELSE  
                    RETURN false  
                END IF  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        FUNCTION is_valid_ipv6_block(block)  
            IF LENGTH OF block GREATER THAN OR EQUAL TO one AND LENGTH OF block LESS THAN OR EQUAL TO four  
                TRY  
                    CONVERT block TO integer WITH BASE sixteen  
                    RETURN true  
                CATCH ANY EXCEPTION  
                    RETURN false  
                END TRY  
            ELSE  
                RETURN false  
            END IF  
        END FUNCTION  

        SET parts_ipv4 TO list OBTAINED BY SPLITTING queryIP BY period  
        SET parts_ipv6 TO list OBTAINED BY SPLITTING queryIP BY colon  

        IF LENGTH OF parts_ipv4 EQUALS four AND all ELEMENTS OF parts_ipv4 SATISFY is_valid_ipv4_block  
            RETURN string IPv four  
        ELSE IF LENGTH OF parts_ipv6 EQUALS eight AND all ELEMENTS OF parts_ipv6 SATISFY is_valid_ipv6_block  
            RETURN string IPv six  
        ELSE  
            RETURN string Neither  
        END IF  
    END FUNCTION  
END CLASS
