CLASS ListNode  
    FUNCTION __init__ with parameters val SET TO zero and next SET TO none  
        SET the attribute val TO val  
        SET the attribute next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node with parameter values  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO a new ListNode with value at position zero OF values  
    SET p TO head  
    FOR each val IN values from position one TO the last position  
        SET node TO a new ListNode with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list with parameters p1 and p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the equality of the val attribute of p1 AND the val attribute of p2 AND the result of is_same_list with parameters the next attribute of p1 and the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION swapPairs with parameter head  
        SET dummy TO a new ListNode with zero  
        SET the next attribute of dummy TO head  
        SET current TO dummy  
        
        WHILE the next attribute of current NOT EQUALS none AND the next attribute of the next attribute of current NOT EQUALS none  
            SET first TO the next attribute of current  
            SET second TO the next attribute of the next attribute of current  
            
            SET the next attribute of current TO second  
            SET the next attribute of first TO the next attribute of second  
            SET the next attribute of second TO first  
            
            SET current TO the next attribute of the next attribute of current  
        END WHILE  
        
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
