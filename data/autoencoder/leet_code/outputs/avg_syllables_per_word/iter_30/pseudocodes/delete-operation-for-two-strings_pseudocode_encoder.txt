CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        SET dp TO a two DIMENSIONAL collection of zero with size m PLUS one BY n PLUS one  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one OF word1 EQUALS the element at position j MINUS one OF word2  
                    SET the element at position i AND j OF dp TO the element at position i MINUS one AND j MINUS one OF dp PLUS one  
                ELSE  
                    IF the element at position i MINUS one AND j OF dp GREATER THAN the element at position i AND j MINUS one OF dp  
                        SET the element at position i AND j OF dp TO the element at position i MINUS one AND j OF dp  
                    ELSE  
                        SET the element at position i AND j OF dp TO the element at position i AND j MINUS one OF dp  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        SET lcs_length TO the element at position m AND n OF dp  
        RETURN m MINUS lcs_length PLUS n MINUS lcs_length  
    END FUNCTION  
END CLASS
