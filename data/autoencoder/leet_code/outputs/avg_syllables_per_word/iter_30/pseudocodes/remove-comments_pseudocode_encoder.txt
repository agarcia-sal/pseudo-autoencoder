CLASS Solution  
    FUNCTION removeComments(source)  
        SET in_block_comment TO false  
        SET result TO empty list  
        SET current_line TO empty list  

        FOR each line IN source  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF line  
                IF in_block_comment  
                    IF substring from position i TO position i PLUS one OF line EQUALS end block comment symbols  
                        SET in_block_comment TO false  
                        INCREMENT i BY one  
                    END IF  
                ELSE  
                    IF substring from position i TO position i PLUS one OF line EQUALS line comment symbols  
                        BREAK  
                    ELSE IF substring from position i TO position i PLUS one OF line EQUALS start block comment symbols  
                        SET in_block_comment TO true  
                        INCREMENT i BY one  
                    ELSE  
                        APPEND character at position i OF line TO current_line  
                    END IF  
                END IF  
                INCREMENT i BY one  
            END WHILE  

            IF current_line IS NOT empty AND in_block_comment IS false  
                APPEND concatenation of elements in current_line TO result  
                SET current_line TO empty list  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
