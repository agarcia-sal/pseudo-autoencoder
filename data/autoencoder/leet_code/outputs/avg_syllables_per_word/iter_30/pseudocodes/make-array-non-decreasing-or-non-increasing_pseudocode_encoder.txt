CLASS Solution  
    FUNCTION convertArray(nums)  
        FUNCTION min_operations_to_non_decreasing(nums)  
            SET max_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF max_heap is not empty AND negative of element at position zero OF max_heap GREATER THAN num  
                    INCREMENT operations BY negative of the result of removing top element FROM max_heap MINUS num  
                    ADD negative of num TO max_heap  
                END IF  
                ADD negative of num TO max_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
        
        FUNCTION min_operations_to_non_increasing(nums)  
            SET min_heap TO empty list  
            SET operations TO zero  
            FOR each num IN nums  
                IF min_heap is not empty AND element at position zero OF min_heap LESS THAN num  
                    INCREMENT operations BY num MINUS the result of removing top element FROM min_heap  
                    ADD num TO min_heap  
                END IF  
                ADD num TO min_heap  
            END FOR  
            RETURN operations  
        END FUNCTION  
        
        RETURN the minimum BETWEEN the result of min_operations_to_non_decreasing with argument nums AND the result of min_operations_to_non_increasing with argument nums  
    END FUNCTION  
END CLASS
