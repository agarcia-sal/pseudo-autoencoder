CLASS Solution  
    FUNCTION minOperations(target, arr)  
        SET index_map TO empty mapping  
        SET index TO zero  
        FOR each value IN target  
            ASSIGN index_map at key value TO index  
            INCREMENT index BY one  
        END FOR  
        
        SET transformed_arr TO empty list  
        FOR each value IN arr  
            IF value IS IN the keys OF index_map  
                APPEND element at key value OF index_map TO transformed_arr  
            END IF  
        END FOR  
        
        SET lis TO empty list  
        FOR each num IN transformed_arr  
            SET pos TO zero  
            SET low TO zero  
            SET high TO the LENGTH OF lis MINUS one  
            WHILE low IS LESS THAN OR EQUAL TO high  
                SET mid TO low PLUS high DIVIDED BY two WITHOUT remainder  
                IF element at position mid OF lis IS GREATER THAN OR EQUAL TO num  
                    SET high TO mid MINUS one  
                ELSE  
                    SET low TO mid PLUS one  
                END IF  
            END WHILE  
            SET pos TO low  
            IF pos EQUALS the LENGTH OF lis  
                APPEND num TO lis  
            ELSE  
                SET element at position pos OF lis TO num  
            END IF  
        END FOR  
        
        SET lcs_length TO the LENGTH OF lis  
        RETURN the LENGTH OF target MINUS lcs_length  
    END FUNCTION  
END CLASS
