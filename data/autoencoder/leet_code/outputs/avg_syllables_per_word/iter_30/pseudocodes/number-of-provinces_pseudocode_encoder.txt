CLASS Solution  
    FUNCTION findCircleNum(isConnected)  
        FUNCTION dfs(city)  
            FOR neighbor FROM zero TO n MINUS one  
                IF element at position city OF isConnected at position neighbor EQUALS one AND NOT element at position neighbor OF visited  
                    SET element at position neighbor OF visited TO True  
                    CALL dfs WITH neighbor  
                END IF  
            END FOR  
        END FUNCTION  
        SET n TO the LENGTH OF isConnected  
        SET visited TO a collection of n elements all EQUALS False  
        SET province_count TO zero  
        FOR city FROM zero TO n MINUS one  
            IF NOT element at position city OF visited  
                CALL dfs WITH city  
                INCREMENT province_count BY one  
            END IF  
        END FOR  
        RETURN province_count  
    END FUNCTION  
END CLASS
