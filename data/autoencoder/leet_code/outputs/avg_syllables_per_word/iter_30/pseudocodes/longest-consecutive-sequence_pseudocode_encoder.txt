CLASS Solution  
    FUNCTION longestConsecutive(nums)  
        IF nums EQUALS empty collection  
            RETURN zero  
        END IF  
        SET num_set TO the collection of unique elements in nums  
        SET longest_streak TO zero  
        FOR each num IN num_set  
            IF the element at position num MINUS one NOT IN num_set  
                SET current_num TO num  
                SET current_streak TO one  
                WHILE the element at position current_num PLUS one IN num_set  
                    INCREMENT current_num BY one  
                    INCREMENT current_streak BY one  
                END WHILE  
                IF current_streak GREATER THAN longest_streak  
                    SET longest_streak TO current_streak  
                END IF  
            END IF  
        END FOR  
        RETURN longest_streak  
    END FUNCTION  
END CLASS
