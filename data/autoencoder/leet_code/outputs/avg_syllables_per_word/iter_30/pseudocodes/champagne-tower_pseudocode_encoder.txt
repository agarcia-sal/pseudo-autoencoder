CLASS Solution  
    FUNCTION champagneTower(poured, query_row, query_glass)  
        SET current_row TO a list containing poured  
        FOR each row FROM one TO query_row  
            SET next_row TO a list of zeros with length equal to row PLUS one  
            FOR index FROM zero TO the length of current_row MINUS one  
                IF element at position index OF current_row GREATER THAN one  
                    SET excess TO the element at position index OF current_row MINUS one DIVIDED BY two  
                    INCREMENT element at position index OF next_row BY excess  
                    INCREMENT element at position index PLUS one OF next_row BY excess  
                END IF  
            END FOR  
            ASSIGN reference TO next_row TO current_row  
        END FOR  
        IF element at position query_glass OF current_row LESS THAN one  
            RETURN element at position query_glass OF current_row  
        ELSE  
            RETURN one  
        END IF  
    END FUNCTION  
END CLASS
