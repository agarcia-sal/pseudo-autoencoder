CLASS Solution  
    FUNCTION numberToWords(num)  
        IF num EQUALS zero  
            RETURN the word zero  
        END IF  
        
        SET below_twenty TO the list of words for numbers from zero up to nineteen with first element as empty string  
        SET tens TO the list of words for tens multiples from zero up to ninety with first two elements as empty strings  
        SET thousands TO the list of words for thousand powers with first element as empty string  
        
        FUNCTION helper(n)  
            IF n EQUALS zero  
                RETURN an empty string  
            ELSE IF n LESS THAN twenty  
                RETURN the element at position n OF below_twenty PLUS a space  
            ELSE IF n LESS THAN one hundred  
                RETURN the element at position the integer division of n by ten OF tens PLUS a space PLUS the result of calling helper with the remainder of n divided by ten  
            ELSE  
                RETURN the element at position the integer division of n by one hundred OF below_twenty PLUS the word hundred PLUS a space PLUS the result of calling helper with the remainder of n divided by one hundred  
            END IF  
        END FUNCTION  
        
        SET result TO an empty string  
        SET i TO zero  
        
        WHILE num GREATER THAN zero  
            IF the remainder of num divided by one thousand NOT EQUALS zero  
                SET result TO the result of calling helper with the remainder of num divided by one thousand PLUS the element at position i OF thousands PLUS a space PLUS result  
            END IF  
            SET num TO the integer division of num by one thousand  
            INCREMENT i BY one  
        END WHILE  
        
        RETURN result with leading and trailing spaces removed  
    END FUNCTION  
END CLASS
