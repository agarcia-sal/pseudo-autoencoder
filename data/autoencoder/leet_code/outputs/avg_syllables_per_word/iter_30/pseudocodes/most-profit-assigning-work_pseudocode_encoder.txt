CLASS Solution  
    FUNCTION maxProfitAssignment(difficulty, profit, worker)  
        ASSIGN reference TO list of pairs of elements at corresponding positions IN difficulty AND profit  
        SORT the list of pairs by the first element of each pair  
        SORT the list worker  
        SET max_profit TO zero  
        SET best_profit TO zero  
        SET job_index TO zero  
        SET n TO the length of the list of pairs  
        FOR each ability IN worker  
            WHILE job_index LESS THAN n AND first element of the pair at position job_index IN the list LESS THAN OR EQUAL TO ability  
                SET best_profit TO the greater value between best_profit AND second element of the pair at position job_index IN the list  
                INCREMENT job_index BY one  
            END WHILE  
            INCREMENT max_profit BY best_profit  
        END FOR  
        RETURN max_profit  
    END FUNCTION  
END CLASS
