CLASS Solution  
    FUNCTION exclusiveTime(n, logs)  
        SET exclusive_time TO a list of length n with all elements zero  
        SET stack TO empty list  
        FOR each log IN logs  
            SPLIT log INTO function_id_string, action, timestamp_string BY colon  
            SET function_id TO the integer value of function_id_string  
            SET timestamp TO the integer value of timestamp_string  
            IF action EQUALS the word start  
                IF the stack is not empty  
                    SET prev_function_id TO the first element of the last element in stack  
                    SET prev_timestamp TO the second element of the last element in stack  
                    INCREMENT exclusive_time at position prev_function_id BY timestamp MINUS prev_timestamp  
                END IF  
                APPEND a tuple of function_id and timestamp TO stack  
            ELSE  
                SET prev_function_id TO the first element of the last element in stack  
                SET prev_timestamp TO the second element of the last element in stack  
                REMOVE the last element FROM stack  
                INCREMENT exclusive_time at position prev_function_id BY timestamp MINUS prev_timestamp PLUS one  
                IF the stack is not empty  
                    SET top_function_id TO the first element of the last element in stack  
                    SET new_timestamp TO timestamp PLUS one  
                    REPLACE the last element in stack WITH a tuple of top_function_id and new_timestamp  
                END IF  
            END IF  
        END FOR  
        RETURN exclusive_time  
    END FUNCTION  
END CLASS
