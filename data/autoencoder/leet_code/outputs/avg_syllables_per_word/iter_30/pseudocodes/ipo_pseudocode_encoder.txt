CLASS Solution  
    FUNCTION findMaximizedCapital WITH PARAMETERS k AND w AND profits AND capital  
        SET projects TO the LIST OF PAIRS formed by pairing elements at corresponding positions OF capital AND profits  
        SORT projects IN ascending order BY the first element of each pair  
        SET max_heap TO an empty LIST  
        SET i TO zero  
        SET n TO the LENGTH OF projects  
        FOR each iteration FROM one TO k  
            WHILE i LESS THAN n AND the first element at position i OF projects LESS THAN OR EQUAL TO w  
                PUSH the NEGATIVE of the second element at position i OF projects INTO max_heap  
                INCREMENT i BY one  
            END WHILE  
            IF max_heap IS empty  
                BREAK  
            END IF  
            SET w TO w PLUS the NEGATIVE of the ELEMENT popped FROM max_heap  
        END FOR  
        RETURN w  
    END FUNCTION  
END CLASS
