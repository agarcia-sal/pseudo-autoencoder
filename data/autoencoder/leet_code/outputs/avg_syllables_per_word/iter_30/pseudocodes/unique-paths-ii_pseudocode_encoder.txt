CLASS Solution  
    FUNCTION uniquePathsWithObstacles(obstacleGrid)  
        IF obstacleGrid EQUALS none OR element at position zero OF obstacleGrid EQUALS none OR element at position zero OF element at position zero OF obstacleGrid EQUALS one  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF obstacleGrid  
        SET n TO the LENGTH OF element at position zero OF obstacleGrid  
        SET dp TO a new two dimensional list of zeros with m rows and n columns  
        SET element at position zero OF element at position zero OF dp TO one  
        FOR j FROM one TO n MINUS one  
            IF element at position zero OF element at position j OF obstacleGrid EQUALS zero  
                SET element at position zero OF element at position j OF dp TO element at position zero OF element at position j MINUS one OF dp  
            END IF  
        END FOR  
        FOR i FROM one TO m MINUS one  
            IF element at position i OF element at position zero OF obstacleGrid EQUALS zero  
                SET element at position i OF element at position zero OF dp TO element at position i MINUS one OF element at position zero OF dp  
            END IF  
        END FOR  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                IF element at position i OF element at position j OF obstacleGrid EQUALS zero  
                    SET element at position i OF element at position j OF dp TO element at position i MINUS one OF element at position j OF dp PLUS element at position i OF element at position j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one OF element at position n MINUS one OF dp  
    END FUNCTION  
END CLASS
