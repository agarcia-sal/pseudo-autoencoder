CLASS Solution  
    FUNCTION isAdditiveNumber(num)  
        FUNCTION is_valid_sequence(start, first, second)  
            IF start EQUALS the LENGTH OF num  
                RETURN true  
            END IF  
            SET expected TO the STRING FORM OF the SUM of first PLUS second  
            SET length TO the LENGTH OF expected  
            IF start PLUS length GREATER THAN the LENGTH OF num OR the substring from position start TO position start PLUS length OF num NOT EQUALS expected  
                RETURN false  
            END IF  
            RETURN is_valid_sequence with start INCREMENTED BY length, second, and the INTEGER FORM OF expected  
        END FUNCTION  
        SET n TO the LENGTH OF num  
        FOR i FROM one TO n DIVIDED BY two PLUS one  
            FOR j FROM i PLUS one TO n  
                SET first TO the substring from position zero TO position i OF num  
                SET second TO the substring from position i TO position j OF num  
                IF the LENGTH OF first GREATER THAN one AND the element at position zero OF first EQUALS zero OR the LENGTH OF second GREATER THAN one AND the element at position zero OF second EQUALS zero  
                    CONTINUE  
                END IF  
                IF is_valid_sequence with j, the INTEGER FORM OF first, and the INTEGER FORM OF second RETURNS true  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
