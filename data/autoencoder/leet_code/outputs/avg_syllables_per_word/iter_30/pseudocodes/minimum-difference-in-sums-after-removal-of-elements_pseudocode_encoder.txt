CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY three  
        
        SET min_heap TO an empty list  
        SET min_sums TO a list of zeroes with LENGTH two MULTIPLIED BY n PLUS one  
        
        FOR each index i FROM zero TO two MULTIPLIED BY n MINUS one  
            PUSH the NEGATIVE of the element at position i OF nums INTO min_heap  
            IF the LENGTH OF min_heap GREATER THAN n  
                REMOVE the smallest element FROM min_heap  
            END IF  
            IF the LENGTH OF min_heap EQUALS n  
                SET the element at position i OF min_sums TO the NEGATIVE of the SUM OF all elements in min_heap  
            END IF  
        END FOR  
        
        SET max_heap TO an empty list  
        SET max_sums TO a list of zeroes with LENGTH two MULTIPLIED BY n PLUS one  
        
        FOR each index i FROM three MULTIPLIED BY n MINUS one DOWN TO n  
            PUSH the element at position i OF nums INTO max_heap  
            IF the LENGTH OF max_heap GREATER THAN n  
                REMOVE the smallest element FROM max_heap  
            END IF  
            IF the LENGTH OF max_heap EQUALS n  
                SET the element at position i OF max_sums TO the SUM OF all elements in max_heap  
            END IF  
        END FOR  
        
        SET min_diff TO positive infinity  
        
        FOR each index i FROM n MINUS one TO two MULTIPLIED BY n MINUS one  
            SET min_diff TO the SMALLER OF min_diff AND the DIFFERENCE BETWEEN the element at position i OF min_sums AND the element at position i PLUS one OF max_sums  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
