CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val equal to element at position zero of values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue is greater than zero  
        SET node TO the element removed from the front of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.left TO new TreeNode with val equal to element at position i of values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i of values IS NOT none  
            SET node.right TO new TreeNode with val equal to element at position i of values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION buildTree(self, inorder, postorder)  
        IF inorder IS empty OR postorder IS empty  
            RETURN none  
        END IF  
        SET root_val TO the element removed from the end of postorder  
        SET root TO new TreeNode with val equal to root_val  
        SET root_index TO the position of root_val in inorder  
        SET root.right TO self.buildTree(sublist of inorder from position root_index PLUS one TO the end, postorder)  
        SET root.left TO self.buildTree(sublist of inorder from position zero TO position root_index MINUS one PLUS one, postorder)  
        RETURN root  
    END FUNCTION  
END CLASS
