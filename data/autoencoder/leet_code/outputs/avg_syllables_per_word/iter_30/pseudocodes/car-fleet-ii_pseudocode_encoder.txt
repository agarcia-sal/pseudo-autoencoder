CLASS Solution  
    FUNCTION getCollisionTimes(cars)  
        SET n TO the LENGTH OF cars  
        SET result TO a list of negative one point zero repeated n times  
        SET stack TO an empty list  

        FOR i FROM n MINUS one TO zero STEP minus one  
            SET pos TO element at position zero OF element at position i OF cars  
            SET speed TO element at position one OF element at position i OF cars  

            WHILE stack is not empty  
                SET next_index TO element at the last position OF stack  
                SET next_pos TO element at position zero OF element at position next_index OF cars  
                SET next_speed TO element at position one OF element at position next_index OF cars  

                IF speed LESS THAN OR EQUAL TO next_speed  
                    REMOVE the last element FROM stack  
                    CONTINUE  
                END IF  

                SET collision_time TO next_pos MINUS pos DIVIDED BY speed MINUS next_speed  

                IF element at position next_index OF result GREATER THAN zero AND collision_time GREATER THAN OR EQUAL TO element at position next_index OF result  
                    REMOVE the last element FROM stack  
                    CONTINUE  
                END IF  

                SET element at position i OF result TO collision_time  
                BREAK  
            END WHILE  

            APPEND i TO stack  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
