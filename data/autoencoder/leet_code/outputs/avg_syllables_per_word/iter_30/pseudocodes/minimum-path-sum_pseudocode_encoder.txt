CLASS Solution  
    FUNCTION minPathSum(grid)  
        IF grid EQUALS empty OR element at position zero OF grid EQUALS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero OF grid  
        SET dp TO two dimensional collection of zero of size m BY n  
        SET element at position zero OF dp at position zero TO element at position zero OF grid at position zero  
        FOR j FROM one TO n MINUS one  
            SET element at position zero OF dp at position j TO element at position zero OF dp at position j MINUS one PLUS element at position zero OF grid at position j  
        END FOR  
        FOR i FROM one TO m MINUS one  
            SET element at position i OF dp at position zero TO element at position i MINUS one OF dp at position zero PLUS element at position i OF grid at position zero  
        END FOR  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET element at position i OF dp at position j TO minimum of element at position i MINUS one OF dp at position j AND element at position i OF dp at position j MINUS one PLUS element at position i OF grid at position j  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one OF dp at position n MINUS one  
    END FUNCTION  
END CLASS
