CLASS Solution  
    FUNCTION flipgame(fronts, backs)  
        SET same TO an empty collection  
        FOR each pair of elements x and y FROM fronts and backs zipped together  
            IF x EQUALS y  
                ADD x TO same  
            END IF  
        END FOR  
        SET all_numbers TO all elements FROM fronts combined with all elements FROM backs  
        SET good_numbers TO all elements of all_numbers EXCLUDING those in same  
        IF good_numbers is not empty  
            SET result TO the minimum element IN good_numbers  
            RETURN result  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
