CLASS Solution  
    FUNCTION removeKdigits(num k)  
        SET stack TO empty list  
        FOR each digit IN num  
            WHILE k is greater than zero AND stack is not empty AND last element of stack is greater than digit  
                REMOVE last element from stack  
                DECREMENT k BY one  
            END WHILE  
            APPEND digit TO stack  
        END FOR  
        IF k is greater than zero  
            SET final_stack TO elements from start TO the LENGTH OF stack MINUS k of stack  
        ELSE  
            SET final_stack TO stack  
        END IF  
        SET result TO concatenation of elements of final_stack WITH leading zeros removed  
        IF result is empty  
            RETURN zero  
        ELSE  
            RETURN result  
        END IF  
    END FUNCTION  
END CLASS
