CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS self, val DEFAULT zero, next DEFAULT none  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO new instance OF ListNode WITH val EQUAL TO element at position zero OF values  
    SET p TO head  
    FOR each val IN the elements of values FROM position one TO the end  
        SET node TO new instance OF ListNode WITH val EQUAL TO val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1, p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN p1.val EQUALS p2.val AND is_same_list WITH PARAMETERS p1.next, p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION oddEvenList WITH PARAMETERS self, head  
        IF head EQUALS none OR head.next EQUALS none OR head.next.next EQUALS none  
            RETURN head  
        END IF  
        SET odd TO head  
        SET even TO head.next  
        SET even_head TO even  
        WHILE even NOT EQUALS none AND even.next NOT EQUALS none  
            SET odd.next TO even.next  
            SET odd TO odd.next  
            SET even.next TO odd.next  
            SET even TO even.next  
        END WHILE  
        SET odd.next TO even_head  
        RETURN head  
    END FUNCTION  
END CLASS
