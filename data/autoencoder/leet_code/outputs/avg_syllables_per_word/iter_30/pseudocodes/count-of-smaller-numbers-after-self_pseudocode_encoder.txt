CLASS Solution  
    FUNCTION countSmaller(nums)  
        SET sorted_nums TO the sorted list of unique elements from nums  
        SET rank TO an empty dictionary  
        SET index TO zero  
        FOR each element val IN sorted_nums  
            INCREMENT index BY one  
            SET rank at key val TO index  
        END FOR  
        
        SET bit TO a list of zeros with length equal to the length of sorted_nums PLUS one  
        
        FUNCTION get_sum(idx)  
            SET res TO zero  
            WHILE idx is greater than zero  
                INCREMENT res BY element at position idx OF bit  
                SET idx TO idx MINUS the bitwise AND of idx and the negative of idx explained as idx AND negative idx  
            END WHILE  
            RETURN res  
        END FUNCTION  
        
        FUNCTION update(idx, delta)  
            WHILE idx is less than the length of bit  
                INCREMENT element at position idx OF bit BY delta  
                SET idx TO idx PLUS the bitwise AND of idx and the negative of idx explained as idx AND negative idx  
            END WHILE  
        END FUNCTION  
        
        SET counts TO an empty list  
        FOR each num IN the reversed order of nums  
            SET r TO rank at key num  
            APPEND the result of get_sum at argument r MINUS one TO counts  
            CALL update with arguments r and one  
        END FOR  
        
        RETURN the reversed list counts  
    END FUNCTION  
END CLASS
