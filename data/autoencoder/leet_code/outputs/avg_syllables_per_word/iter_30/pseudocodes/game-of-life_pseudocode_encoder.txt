CLASS Solution  
    FUNCTION gameOfLife(board)  
        IF board EQUALS none OR the LENGTH OF board EQUALS zero OR the LENGTH OF element at position zero OF board EQUALS zero  
            RETURN  
        END IF  
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF element at position zero OF board  
        SET directions TO list of pairs consisting of minus one and minus one, minus one and zero, minus one and one, zero and minus one, zero and one, one and minus one, one and zero, one and one  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET live_neighbors TO zero  
                FOR each pair dx, dy IN directions  
                    SET ni TO i PLUS dx  
                    SET nj TO j PLUS dy  
                    IF ni GREATER THAN OR EQUAL TO zero AND ni LESS THAN m AND nj GREATER THAN OR EQUAL TO zero AND nj LESS THAN n AND (element at position ni and nj OF board EQUALS one OR element at position ni and nj OF board EQUALS three)  
                        INCREMENT live_neighbors BY one  
                    END IF  
                END FOR  
                IF element at position i and j OF board EQUALS one  
                    IF live_neighbors LESS THAN two OR live_neighbors GREATER THAN three  
                        SET element at position i and j OF board TO three  
                    END IF  
                ELSE  
                    IF live_neighbors EQUALS three  
                        SET element at position i and j OF board TO two  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i and j OF board EQUALS two  
                    SET element at position i and j OF board TO one  
                ELSE IF element at position i and j OF board EQUALS three  
                    SET element at position i and j OF board TO zero  
                END IF  
            END FOR  
        END FOR  
    END FUNCTION  
END CLASS
