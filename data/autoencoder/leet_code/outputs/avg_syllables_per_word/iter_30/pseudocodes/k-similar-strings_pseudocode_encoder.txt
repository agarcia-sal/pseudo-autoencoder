CLASS Solution  
    FUNCTION kSimilarity(s1, s2)  
        FUNCTION neighbors(s)  
            SET i TO zero  
            WHILE i LESS THAN the LENGTH OF s AND element at position i OF s EQUALS element at position i OF s2  
                INCREMENT i BY one  
            END WHILE  
            FOR j FROM i PLUS one TO the LENGTH OF s MINUS one  
                IF element at position j OF s EQUALS element at position i OF s2 AND element at position j OF s NOT EQUALS element at position j OF s2  
                    YIELD concatenation of substring from start TO position i MINUS one OF s PLUS element at position j OF s PLUS substring from position i PLUS one TO position j MINUS one OF s PLUS element at position i OF s PLUS substring from position j PLUS one TO end OF s  
                END IF  
            END FOR  
        END FUNCTION  
        
        ASSIGN reference TO a new double ended queue with tuple of s1 and zero as the only element  
        ASSIGN reference TO a new set with s1 as the only element  
        
        WHILE the queue is not empty  
            SET s and swaps TO the front element removed from the queue  
            IF s EQUALS s2  
                RETURN swaps  
            END IF  
            FOR each neighbor IN neighbors(s)  
                IF neighbor NOT IN the visited set  
                    ADD neighbor TO the visited set  
                    APPEND tuple of neighbor and swaps PLUS one TO the queue  
                END IF  
            END FOR  
        END WHILE  
    END FUNCTION  
END CLASS
