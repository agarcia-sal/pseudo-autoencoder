CLASS Solution  
    FUNCTION shortestWordDistance(wordsDict, word1, word2)  
        SET min_distance TO positive infinity  
        SET prev_word1 TO negative one  
        SET prev_word2 TO negative one  
        SET same_word TO word1 EQUALS word2  
        FOR each index i FROM zero TO the LENGTH OF wordsDict MINUS one  
            SET word TO element at position i OF wordsDict  
            IF word EQUALS word1  
                IF same_word  
                    SET prev_word2 TO prev_word1  
                END IF  
                SET prev_word1 TO i  
            END IF  
            IF word EQUALS word2  
                IF NOT same_word  
                    SET prev_word2 TO i  
                END IF  
            END IF  
            IF prev_word1 NOT EQUALS negative one AND prev_word2 NOT EQUALS negative one  
                SET difference TO prev_word1 MINUS prev_word2  
                IF difference LESS THAN zero  
                    SET absolute_difference TO difference MINUS multiplied BY negative one  
                ELSE  
                    SET absolute_difference TO difference  
                END IF  
                IF min_distance GREATER THAN absolute_difference  
                    SET min_distance TO absolute_difference  
                END IF  
            END IF  
        END FOR  
        RETURN min_distance  
    END FUNCTION  
END CLASS
