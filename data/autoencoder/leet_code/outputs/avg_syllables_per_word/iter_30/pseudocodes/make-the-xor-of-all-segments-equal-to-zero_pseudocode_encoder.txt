CLASS Solution  
    FUNCTION minChanges(nums, k)  
        SET groups TO empty mapping with default empty list  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            APPEND element at position i OF nums TO groups AT key i MODULO k  
        END FOR  
        
        SET freq TO list of empty mappings with default zero integer OF LENGTH k  
        SET min_size TO list filled with infinite value OF LENGTH k  
        FOR i FROM zero TO k MINUS one  
            FOR each num IN element at position i OF groups  
                INCREMENT freq at position i AT key num BY one  
            END FOR  
            SET element at position i OF min_size TO the LENGTH OF element at position i OF groups  
        END FOR  
        
        SET dp TO list filled with infinite value OF LENGTH two RAISED TO THE power of ten  
        SET element at position zero OF dp TO zero  
        
        FOR i FROM zero TO k MINUS one  
            SET new_dp TO list filled with infinite value OF LENGTH two RAISED TO THE power of ten  
            SET min_dp TO the minimum value OF dp  
            FOR j FROM zero TO two RAISED TO THE power of ten MINUS one  
                SET element at position j OF new_dp TO min_dp PLUS element at position i OF min_size  
                FOR each num AND count IN items OF freq at position i  
                    SET element at position j OF new_dp TO the minimum OF element at position j OF new_dp AND element at position j XOR num OF dp PLUS element at position i OF min_size MINUS count  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        
        RETURN element at position zero OF dp  
    END FUNCTION  
END CLASS
