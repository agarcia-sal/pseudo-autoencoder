CLASS Solution  
    FUNCTION splitLoopedString WITH PARAMETER strs  
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            SET reversed_string TO the reverse of element at position index OF strs  
            IF reversed_string GREATER THAN element at position index OF strs  
                SET element at position index OF strs TO reversed_string  
            END IF  
        END FOR  
        
        SET doubled_strs TO concatenation of strs WITH strs  
        
        SET max_string TO empty string  
        
        FOR index FROM zero TO the LENGTH OF strs MINUS one  
            FOR is_reversed IN list containing false and true  
                IF is_reversed EQUALS true  
                    SET current TO the reverse of element at position index OF strs  
                ELSE  
                    SET current TO element at position index OF strs  
                END IF  
                
                SET remaining TO concatenation of elements from position index PLUS one TO position index PLUS the LENGTH OF strs MINUS one OF doubled_strs  
                
                FOR cut_position FROM zero TO the LENGTH OF current  
                    SET candidate TO concatenation of substring from position cut_position TO end OF current, WITH remaining, WITH substring from position zero TO position cut_position OF current  
                    IF candidate GREATER THAN max_string  
                        SET max_string TO candidate  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN max_string  
    END FUNCTION  
END CLASS
