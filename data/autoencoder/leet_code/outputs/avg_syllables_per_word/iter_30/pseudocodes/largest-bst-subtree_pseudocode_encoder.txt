CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with element at position zero OF values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element removed FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO a new TreeNode initialized with element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO a new TreeNode initialized with element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p IS NOT EQUALS value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestBSTSubtree(self, root)  
        FUNCTION dfs(node)  
            IF node IS none  
                RETURN positive infinity, negative infinity, zero  
            END IF  
            SET left_min, left_max, left_size TO dfs(left child of node)  
            SET right_min, right_max, right_size TO dfs(right child of node)  
            IF left_max IS LESS THAN value of node AND value of node IS LESS THAN right_min  
                RETURN minimum of value of node AND left_min, maximum of value of node AND right_max, left_size PLUS right_size PLUS one  
            END IF  
            RETURN negative infinity, positive infinity, maximum of left_size AND right_size  
        END FUNCTION  
        RETURN the third element of dfs(root)  
    END FUNCTION  
END CLASS
