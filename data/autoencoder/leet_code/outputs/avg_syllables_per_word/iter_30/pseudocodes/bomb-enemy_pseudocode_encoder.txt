CLASS Solution  
    FUNCTION maxKilledEnemies(grid)  
        IF grid EQUALS nothing OR element at position zero OF grid EQUALS nothing  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero OF grid  
        SET max_kills TO zero  
        SET row_kills TO zero  
        SET col_kills TO a list of zeros of LENGTH n  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF j EQUALS zero OR element at position j minus one OF element at position i OF grid EQUALS a wall  
                    SET row_kills TO zero  
                    FOR k FROM j TO n MINUS one  
                        IF element at position k OF element at position i OF grid EQUALS an enemy  
                            INCREMENT row_kills BY one  
                        ELSE IF element at position k OF element at position i OF grid EQUALS a wall  
                            BREAK  
                        END IF  
                    END FOR  
                END IF  
                IF i EQUALS zero OR element at position i minus one OF element at position j OF grid EQUALS a wall  
                    SET element at position j OF col_kills TO zero  
                    FOR k FROM i TO m MINUS one  
                        IF element at position k OF element at position j OF grid EQUALS an enemy  
                            INCREMENT element at position j OF col_kills BY one  
                        ELSE IF element at position k OF element at position j OF grid EQUALS a wall  
                            BREAK  
                        END IF  
                    END FOR  
                END IF  
                IF element at position j OF element at position i OF grid EQUALS empty  
                    IF row_kills PLUS element at position j OF col_kills GREATER THAN max_kills  
                        SET max_kills TO row_kills PLUS element at position j OF col_kills  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN max_kills  
    END FUNCTION  
END CLASS
