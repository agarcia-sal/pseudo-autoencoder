CLASS Solution  
    FUNCTION subsets(nums)  
        IF nums EQUALS empty list  
            RETURN list containing empty list  
        END IF  
        SET subsets_without_first TO CALL subsets WITH the list of elements from position one TO the end OF nums  
        SET subsets_with_first TO empty list  
        FOR each subset IN subsets_without_first  
            APPEND list formed by element at position zero OF nums followed by subset TO subsets_with_first  
        END FOR  
        RETURN concatenation of subsets_without_first with subsets_with_first  
    END FUNCTION  
END CLASS
