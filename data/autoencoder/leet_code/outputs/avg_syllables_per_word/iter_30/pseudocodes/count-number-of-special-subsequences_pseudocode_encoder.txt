CLASS Solution  
    FUNCTION countSpecialSubsequences(nums)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET dp at position zero TO zero  
        SET dp at position one TO zero  
        SET dp at position two TO zero  
        FOR each num IN nums  
            IF num EQUALS zero  
                SET dp at position zero TO two MULTIPLIED BY dp at position zero PLUS one MODULO MOD  
            ELSE IF num EQUALS one  
                SET dp at position one TO two MULTIPLIED BY dp at position one PLUS dp at position zero MODULO MOD  
            ELSE IF num EQUALS two  
                SET dp at position two TO two MULTIPLIED BY dp at position two PLUS dp at position one MODULO MOD  
            END IF  
        END FOR  
        RETURN dp at position two  
    END FUNCTION  
END CLASS
