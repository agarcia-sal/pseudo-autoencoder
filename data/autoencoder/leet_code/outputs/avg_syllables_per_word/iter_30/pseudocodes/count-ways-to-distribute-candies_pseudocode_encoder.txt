CLASS Solution  
    FUNCTION waysToDistribute(n, k)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET dp TO a two dimensional collection of zeroes with rows FROM zero TO n and columns FROM zero TO k  
        SET element at position zero of collection at position zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET product TO j MULTIPLIED BY element at position i minus one of collection at position j of dp  
                SET sum_value TO product PLUS element at position i minus one of collection at position j minus one of dp  
                SET element at position i of collection at position j of dp TO sum_value MODULO MOD  
            END FOR  
        END FOR  
        RETURN element at position n of collection at position k of dp  
    END FUNCTION  
END CLASS
