CLASS Solution  
    FUNCTION findBlackPixel(picture, target)  
        IF picture EQUALS empty OR element at position zero OF picture EQUALS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero OF picture  
        SET row_count TO a list of zero repeated m times  
        SET col_count TO a list of zero repeated n times  
        SET row_patterns TO an empty mapping with default zero  
        FOR each r FROM zero TO m MINUS one  
            SET pattern TO the concatenation of elements at position r OF picture into one sequence  
            INCREMENT the value associated with pattern in row_patterns BY one  
            FOR each c FROM zero TO n MINUS one  
                IF element at position c OF element at position r OF picture EQUALS black pixel  
                    INCREMENT the element at position r OF row_count BY one  
                    INCREMENT the element at position c OF col_count BY one  
                END IF  
            END FOR  
        END FOR  
        SET lonely_pixels TO zero  
        FOR each r FROM zero TO m MINUS one  
            FOR each c FROM zero TO n MINUS one  
                IF element at position c OF element at position r OF picture EQUALS black pixel  
                    SET pattern TO the concatenation of elements at position r OF picture into one sequence  
                    IF element at position r OF row_count EQUALS target AND element at position c OF col_count EQUALS target AND the value associated with pattern in row_patterns EQUALS target  
                        INCREMENT lonely_pixels BY one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        RETURN lonely_pixels  
    END FUNCTION  
END CLASS
