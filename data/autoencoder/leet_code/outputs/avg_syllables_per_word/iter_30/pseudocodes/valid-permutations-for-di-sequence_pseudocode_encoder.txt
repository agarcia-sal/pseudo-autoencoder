CLASS Solution  
    FUNCTION numPermsDISequence(s)  
        SET MODULO_VALUE TO ten raised to the power of nine PLUS one  
        SET length_of_s TO the LENGTH OF s  
        INITIALIZE two dimensional collection dp WITH zero with dimensions length_of_s PLUS one BY length_of_s PLUS one  
        FOR each index j FROM zero TO length_of_s  
            SET element at position zero AND position j OF dp TO one  
        END FOR  
        FOR each index i FROM one TO length_of_s  
            SET prefix_sum TO zero  
            IF character at position i MINUS one OF s EQUALS uppercase letter I  
                FOR each index j FROM zero TO length_of_s PLUS one MINUS i MINUS one  
                    SET prefix_sum TO prefix_sum PLUS element at position i MINUS one AND position j OF dp  
                    SET prefix_sum TO prefix_sum MODULO MODULO_VALUE  
                    SET element at position i AND position j OF dp TO prefix_sum  
                END FOR  
            ELSE  
                FOR each index j FROM length_of_s MINUS i DOWN TO zero STEP negative one  
                    SET prefix_sum TO prefix_sum PLUS element at position i MINUS one AND position j PLUS one OF dp  
                    SET prefix_sum TO prefix_sum MODULO MODULO_VALUE  
                    SET element at position i AND position j OF dp TO prefix_sum  
                END FOR  
            END IF  
        END FOR  
        SET result_sum TO zero  
        FOR each element value IN element at position length_of_s OF dp  
            SET result_sum TO result_sum PLUS value  
        END FOR  
        SET result_sum TO result_sum MODULO MODULO_VALUE  
        RETURN result_sum  
    END FUNCTION  
END CLASS
