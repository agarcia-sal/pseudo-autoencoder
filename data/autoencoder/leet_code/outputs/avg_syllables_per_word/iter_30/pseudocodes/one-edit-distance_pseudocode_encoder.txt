CLASS Solution  
    FUNCTION isOneEditDistance(s, t)  
        IF the LENGTH OF s GREATER THAN the LENGTH OF t  
            RETURN the RESULT OF calling isOneEditDistance WITH t AND s  
        END IF  
        SET diff TO the LENGTH OF t MINUS the LENGTH OF s  
        IF diff GREATER THAN one  
            RETURN false  
        END IF  
        IF diff EQUALS zero  
            SET count TO zero  
            FOR each pair of characters c1 AND c2 FROM s AND t IN ORDER  
                IF c1 NOT EQUALS c2  
                    INCREMENT count BY one  
                END IF  
            END FOR  
            RETURN true IF count EQUALS one OTHERWISE false  
        ELSE  
            FOR index FROM zero TO the LENGTH OF s MINUS one  
                IF the character at position index OF s NOT EQUALS the character at position index OF t  
                    SET substring_s TO substring from position index TO the END OF s  
                    SET substring_t TO substring from position index PLUS one TO the END OF t  
                    RETURN true IF substring_s EQUALS substring_t OTHERWISE false  
                END IF  
            END FOR  
            RETURN true  
        END IF  
    END FUNCTION  
END CLASS
