CLASS Solution  
    FUNCTION validTicTacToe(board)  
        SET countX TO zero  
        FOR each row IN board  
            INCREMENT countX BY the number of times character X occurs in row  
        END FOR  
        SET countO TO zero  
        FOR each row IN board  
            INCREMENT countO BY the number of times character O occurs in row  
        END FOR  
        IF countX LESS THAN countO OR countX GREATER THAN countO PLUS one  
            RETURN false  
        END IF  
        FUNCTION check_win(player)  
            FOR each row IN board  
                IF row EQUALS player REPEATED three TIMES  
                    RETURN true  
                END IF  
            END FOR  
            FOR col FROM zero TO two  
                IF element at position col OF first row EQUALS player AND element at position col OF second row EQUALS player AND element at position col OF third row EQUALS player  
                    RETURN true  
                END IF  
            END FOR  
            IF element at position zero OF first row EQUALS player AND element at position one OF second row EQUALS player AND element at position two OF third row EQUALS player  
                RETURN true  
            END IF  
            IF element at position two OF first row EQUALS player AND element at position one OF second row EQUALS player AND element at position zero OF third row EQUALS player  
                RETURN true  
            END IF  
            RETURN false  
        END FUNCTION  
        SET winX TO the result of check_win with argument character X  
        SET winO TO the result of check_win with argument character O  
        IF winX AND winO  
            RETURN false  
        END IF  
        IF winX AND countX EQUALS countO  
            RETURN false  
        END IF  
        IF winO AND countX NOT EQUALS countO  
            RETURN false  
        END IF  
        RETURN true  
    END FUNCTION  
END CLASS
