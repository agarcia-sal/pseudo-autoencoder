CLASS Solution  
    FUNCTION waysToBuildRooms(prevRoom)  
        ASSIGN reference TO an empty mapping from keys to list of values named graph  
        SET n TO the LENGTH OF prevRoom  
        FOR each room FROM one TO n MINUS one  
            APPEND room TO the list at key prevRoom at position room OF graph  
        END FOR  
        
        FUNCTION dp(node)  
            IF the list at key node OF graph IS empty  
                RETURN one AND one  
            END IF  
            
            SET total_ways TO one  
            SET total_size TO one  
            ASSIGN reference TO an empty list named child_results  
            
            FOR each child IN the list at key node OF graph  
                SET ways AND size TO dp(child)  
                APPEND the pair ways AND size TO child_results  
                INCREMENT total_size BY size  
            END FOR  
            
            SET current_size TO one  
            FOR each ways AND size IN child_results  
                SET total_ways TO total_ways MULTIPLIED BY the combination of current_size PLUS size MINUS one choose size MULTIPLIED BY ways MODULO ten to the ninth PLUS seven  
                INCREMENT current_size BY size  
            END FOR  
            
            RETURN total_ways AND total_size  
        END FUNCTION  
        
        RETURN the first element of dp(zero)  
    END FUNCTION  
END CLASS
