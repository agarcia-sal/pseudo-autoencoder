CLASS Solution  
    FUNCTION maximumNumberOfOnes(width, height, sideLength, maxOnes)  
        SET freq TO list of zero values with length equal to sideLength multiplied by sideLength  
        FOR i FROM zero TO width MINUS one  
            FOR j FROM zero TO height MINUS one  
                SET pos TO remainder of i DIVIDED BY sideLength MULTIPLIED BY sideLength PLUS remainder of j DIVIDED BY sideLength  
                INCREMENT element at position pos OF freq BY one  
            END FOR  
        END FOR  
        SORT freq in order from greatest to smallest  
        SET total TO zero  
        FOR index FROM zero TO maxOnes MINUS one  
            INCREMENT total BY element at position index OF freq  
        END FOR  
        RETURN total  
    END FUNCTION  
END CLASS
