CLASS Solution  
    FUNCTION kSmallestPairs(nums1, nums2, k)  
        IF nums1 IS empty OR nums2 IS empty OR k LESS THAN OR EQUAL TO zero  
            RETURN empty list  
        END IF  
        SET min_heap TO empty list  
        FOR i FROM zero TO the MINIMUM OF k AND the LENGTH OF nums1 MINUS one  
            ADD TO min_heap the tuple OF the sum OF element at position i OF nums1 AND element at position zero OF nums2, i, zero  
        END FOR  
        SET result TO empty list  
        WHILE min_heap IS NOT empty AND the LENGTH OF result LESS THAN k  
            REMOVE the smallest tuple FROM min_heap AND ASSIGN TO sum_value, i, j  
            APPEND list OF element at position i OF nums1 AND element at position j OF nums2 TO result  
            IF j PLUS one LESS THAN the LENGTH OF nums2  
                ADD TO min_heap the tuple OF the sum OF element at position i OF nums1 AND element at position j PLUS one OF nums2, i, j PLUS one  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
