CLASS Solution  
    FUNCTION rectangleArea(rectangles)  
        SET MODULO TO ten raised to the power of nine PLUS one  
        SET y_coordinates TO an empty set  
        FOR each rectangle consisting of x1, y1, x2, y2 IN rectangles  
            ADD y1 TO y_coordinates  
            ADD y2 TO y_coordinates  
        END FOR  
        SET y_coordinates TO the sorted list of y_coordinates  
        SET y_index TO an empty dictionary mapping each element y TO its position index in y_coordinates  
        SET count TO a list of zero values with length equal to the length of y_coordinates  
        SET events TO an empty list  
        FOR each rectangle consisting of x1, y1, x2, y2 IN rectangles  
            APPEND a tuple of x1, one, y1, y2 TO events  
            APPEND a tuple of x2, negative one, y1, y2 TO events  
        END FOR  
        SORT events by the first element of each tuple representing the x-coordinate  
        SET previous_x TO zero  
        SET area TO zero  
        FOR each event consisting of x, eventType, y1, y2 IN events  
            SET current_length TO zero  
            FOR index FROM one TO the length of count MINUS one  
                IF the element at position index MINUS one of count IS GREATER THAN zero  
                    INCREMENT current_length BY the element at position index of y_coordinates MINUS the element at position index MINUS one of y_coordinates  
                END IF  
            END FOR  
            INCREMENT area BY current_length MULTIPLIED BY x MINUS previous_x  
            SET area TO area MODULO MODULO  
            IF eventType EQUALS one  
                FOR index FROM the value mapped by y1 in y_index TO the value mapped by y2 in y_index MINUS one  
                    INCREMENT the element at position index of count BY one  
                END FOR  
            ELSE  
                FOR index FROM the value mapped by y1 in y_index TO the value mapped by y2 in y_index MINUS one  
                    DECREMENT the element at position index of count BY one  
                END FOR  
            END IF  
            SET previous_x TO x  
        END FOR  
        RETURN area  
    END FUNCTION  
END CLASS
