CLASS Solution  
    FUNCTION maxJumps(arr, d)  
        FUNCTION dp(i)  
            SET max_jumps TO one  
            FOR x FROM one TO d  
                IF i PLUS x LESS THAN the LENGTH OF arr  
                    IF element at position i PLUS x OF arr GREATER THAN OR EQUAL TO element at position i OF arr  
                        BREAK  
                    END IF  
                    SET candidate TO one PLUS dp(i PLUS x)  
                    IF candidate GREATER THAN max_jumps  
                        SET max_jumps TO candidate  
                    END IF  
                END IF  
            END FOR  
            FOR x FROM one TO d  
                IF i MINUS x GREATER THAN OR EQUAL TO zero  
                    IF element at position i MINUS x OF arr GREATER THAN OR EQUAL TO element at position i OF arr  
                        BREAK  
                    END IF  
                    SET candidate TO one PLUS dp(i MINUS x)  
                    IF candidate GREATER THAN max_jumps  
                        SET max_jumps TO candidate  
                    END IF  
                END IF  
            END FOR  
            RETURN max_jumps  
        END FUNCTION  
        SET results TO empty collection  
        FOR i FROM zero TO the LENGTH OF arr MINUS one  
            APPEND dp(i) TO results  
        END FOR  
        RETURN the maximum element OF results  
    END FUNCTION  
END CLASS
