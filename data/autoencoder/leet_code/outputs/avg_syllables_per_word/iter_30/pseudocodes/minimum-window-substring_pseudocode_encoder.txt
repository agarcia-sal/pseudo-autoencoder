CLASS Solution  
    FUNCTION minWindow(s, t)  
        SET t_count TO the frequency counts of characters in t  
        SET required TO the number of unique keys in t_count  
        SET left TO zero  
        SET right TO zero  
        SET formed TO zero  
        SET window_counts TO empty frequency counts  
        SET min_length TO infinity  
        SET min_start TO zero  
        WHILE right LESS THAN the length of s  
            INCREMENT window_counts at character at position right OF s BY one  
            IF character at position right OF s EXISTS IN t_count AND window_counts at character at position right OF s EQUALS t_count at character at position right OF s  
                INCREMENT formed BY one  
            END IF  
            WHILE left LESS THAN OR EQUAL TO right AND formed EQUALS required  
                SET char TO character at position left OF s  
                IF right MINUS left PLUS one LESS THAN min_length  
                    SET min_length TO right MINUS left PLUS one  
                    SET min_start TO left  
                END IF  
                DECREMENT window_counts at char BY one  
                IF char EXISTS IN t_count AND window_counts at char LESS THAN t_count at char  
                    DECREMENT formed BY one  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            INCREMENT right BY one  
        END WHILE  
        IF min_length EQUALS infinity  
            RETURN empty string  
        ELSE  
            RETURN substring from position min_start TO position min_start PLUS min_length OF s  
        END IF  
    END FUNCTION  
END CLASS
