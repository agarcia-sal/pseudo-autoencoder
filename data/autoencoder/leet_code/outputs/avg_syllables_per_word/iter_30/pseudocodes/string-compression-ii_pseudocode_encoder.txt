CLASS Solution  
    FUNCTION getLengthOfOptimalCompression(s, k)  
        FUNCTION dp(i, k_param, prev_char, prev_count)  
            IF k_param LESS THAN zero  
                RETURN positive infinity  
            END IF  
            IF i EQUALS the length of s  
                RETURN zero  
            END IF  
            IF character at position i OF s EQUALS prev_char  
                IF prev_count EQUALS one OR prev_count EQUALS nine OR prev_count EQUALS ninety nine  
                    SET increment TO one  
                ELSE  
                    SET increment TO zero  
                END IF  
                RETURN increment PLUS dp(i PLUS one, k_param, prev_char, prev_count PLUS one)  
            ELSE  
                SET keep TO one PLUS dp(i PLUS one, k_param, character at position i OF s, one)  
                SET delete TO dp(i PLUS one, k_param MINUS one, prev_char, prev_count)  
                RETURN the minimum of keep AND delete  
            END IF  
        END FUNCTION  
        RETURN dp(zero, k, empty string, zero)  
    END FUNCTION  
END CLASS
