CLASS Solution  
    FUNCTION numberOfArithmeticSlices(nums)  
        SET n TO the LENGTH OF nums  
        IF n LESS THAN three  
            RETURN zero  
        END IF  
        
        SET dp TO a LIST OF n elements WHERE each element is a DEFAULT DICTIONARY of integers  
        SET total_count TO zero  
        
        FOR index i FROM zero TO n MINUS one  
            FOR index j FROM zero TO i MINUS one  
                SET difference d TO element at position i of nums MINUS element at position j of nums  
                
                IF difference d EXISTS AS a KEY IN dp at position j  
                    INCREMENT dp at position i at key d BY the VALUE of dp at position j at key d PLUS one  
                    INCREMENT total_count BY the VALUE of dp at position j at key d  
                ELSE  
                    INCREMENT dp at position i at key d BY one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN total_count  
    END FUNCTION  
END CLASS
