CLASS Solution  
    FUNCTION getHint(secret, guess)  
        SET bulls TO zero  
        SET cows TO zero  
        SET secret_count TO an empty mapping from element TO zero  
        SET guess_count TO an empty mapping from element TO zero  
        FOR each pair s and g IN parallel traversal of secret and guess  
            IF s EQUALS g  
                INCREMENT bulls BY one  
            ELSE  
                INCREMENT value at key s IN secret_count BY one  
                INCREMENT value at key g IN guess_count BY one  
            END IF  
        END FOR  
        FOR each digit IN keys of secret_count  
            INCREMENT cows BY the minimum of value at key digit IN secret_count AND value at key digit IN guess_count  
        END FOR  
        RETURN the concatenation of bulls, the letter A, cows, and the letter B described in natural language  
    END FUNCTION  
END CLASS
