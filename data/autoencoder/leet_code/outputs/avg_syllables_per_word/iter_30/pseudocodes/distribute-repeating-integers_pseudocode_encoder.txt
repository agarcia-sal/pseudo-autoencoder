CLASS Solution  
    FUNCTION canDistribute(nums, quantity)  
        SET count TO the list of the frequency counts of each element in nums  
        SORT quantity in descending order  
        
        FUNCTION can_satisfy(index)  
            IF index EQUALS the length of quantity  
                RETURN true  
            END IF  
            FOR each i FROM zero TO the length of count MINUS one  
                IF element at position i OF count IS GREATER THAN OR EQUAL TO element at position index OF quantity  
                    DECREMENT element at position i OF count BY element at position index OF quantity  
                    IF calling can_satisfy with argument index PLUS one RETURNS true  
                        RETURN true  
                    END IF  
                    INCREMENT element at position i OF count BY element at position index OF quantity  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN the result of calling can_satisfy with argument zero  
    END FUNCTION  
END CLASS
