CLASS Solution  
    FUNCTION carFleet(target, position, speed)  
        SET paired_list TO list of pairs of element at position index OF position AND element at position index OF speed sorted in descending order by element at position zero of the pair  
        SET time_to_target TO empty list  
        FOR each pair IN paired_list  
            SET time_value TO difference of target MINUS element at position zero of pair DIVIDED BY element at position one of pair  
            APPEND time_value TO time_to_target  
        END FOR  
        SET fleets TO zero  
        SET current_max_time TO zero  
        FOR each time IN time_to_target  
            IF time GREATER THAN current_max_time  
                INCREMENT fleets BY one  
                SET current_max_time TO time  
            END IF  
        END FOR  
        RETURN fleets  
    END FUNCTION  
END CLASS
