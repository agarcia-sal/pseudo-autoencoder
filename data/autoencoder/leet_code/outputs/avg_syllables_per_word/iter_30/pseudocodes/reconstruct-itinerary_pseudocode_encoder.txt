CLASS Solution  
    FUNCTION findItinerary(tickets)  
        ASSIGN graph TO a default dictionary with list as default value  
        FOR each pair of from_element and to_element IN the reverse order of the sorted tickets  
            APPEND to_element TO the list at key from_element IN graph  
        END FOR  
        
        ASSIGN itinerary TO an empty list  
        
        FUNCTION visit(airport)  
            WHILE the list at key airport IN graph is not empty  
                CALL visit WITH the last element removed from the list at key airport IN graph  
            END WHILE  
            APPEND airport TO itinerary  
        END FUNCTION  
        
        CALL visit WITH the string J F K  
        RETURN the itinerary in reverse order  
    END FUNCTION  
END CLASS
