CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with value as element at position zero OF values  
    SET i TO one  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from the front OF queue  
        IF i LESS THAN LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO a new TreeNode with value as element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO a new TreeNode with value as element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree with p.left and q.left AND the result of is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION recoverTree(self, root)  
        SET first TO none  
        SET second TO none  
        SET prev TO none  
        SET curr TO root  
        WHILE curr IS NOT none  
            IF curr.left IS none  
                IF prev IS NOT none AND prev.val GREATER THAN curr.val  
                    IF first IS none  
                        SET first TO prev  
                    END IF  
                    SET second TO curr  
                END IF  
                SET prev TO curr  
                SET curr TO curr.right  
            ELSE  
                SET pred TO curr.left  
                WHILE pred.right IS NOT none AND pred.right IS NOT curr  
                    SET pred TO pred.right  
                END WHILE  
                IF pred.right IS none  
                    SET pred.right TO curr  
                    SET curr TO curr.left  
                ELSE  
                    SET pred.right TO none  
                    IF prev IS NOT none AND prev.val GREATER THAN curr.val  
                        IF first IS none  
                            SET first TO prev  
                        END IF  
                        SET second TO curr  
                    END IF  
                    SET prev TO curr  
                    SET curr TO curr.right  
                END IF  
            END IF  
        END WHILE  
        IF first IS NOT none AND second IS NOT none  
            SET temporary_value TO first.val  
            SET first.val TO second.val  
            SET second.val TO temporary_value  
        END IF  
    END FUNCTION  
END CLASS
