CLASS Solution  
    FUNCTION exist(board, word)  
        IF board EQUALS empty  
            RETURN false  
        END IF  
        SET m TO the LENGTH OF board  
        SET n TO the LENGTH OF the element at position zero OF board  
        SET visited TO a two dimensional list of boolean false with m rows and n columns  
        FUNCTION dfs(x, y, index)  
            IF index EQUALS the LENGTH OF word  
                RETURN true  
            END IF  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO m OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO n OR element at position x AND y OF visited IS true OR element at position x AND y OF board NOT EQUALS element at position index OF word  
                RETURN false  
            END IF  
            SET element at position x AND y OF visited TO true  
            FOR each pair dx AND dy IN the set consisting of minus one and zero, one and zero, zero and minus one, zero and one  
                IF dfs x PLUS dx, y PLUS dy, index PLUS one RETURNS true  
                    RETURN true  
                END IF  
            END FOR  
            SET element at position x AND y OF visited TO false  
            RETURN false  
        END FUNCTION  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF dfs i, j, zero RETURNS true  
                    RETURN true  
                END IF  
            END FOR  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
