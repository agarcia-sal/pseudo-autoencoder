CLASS Solution  
    FUNCTION solveEquation(equation)  
        FUNCTION parse_side(side)  
            SET tokens TO side with each plus replaced by separator plus and each minus replaced by separator minus then split by separator  
            SET x_count TO zero  
            SET num_sum TO zero  
            FOR each token IN tokens  
                IF token is empty  
                    CONTINUE  
                END IF  
                IF token contains character x  
                    IF token equals character x OR token equals plus character x  
                        INCREMENT x_count BY one  
                    ELSE IF token equals minus character x  
                        DECREMENT x_count BY one  
                    ELSE  
                        INCREMENT x_count BY integer value of token with character x removed  
                    END IF  
                ELSE  
                    INCREMENT num_sum BY integer value of token  
                END IF  
            END FOR  
            RETURN x_count, num_sum  
        END FUNCTION  

        SET left_side TO substring before character equal sign OF equation  
        SET right_side TO substring after character equal sign OF equation  

        SET left_x, left_num TO parse_side(left_side)  
        SET right_x, right_num TO parse_side(right_side)  

        SET total_x TO left_x MINUS right_x  
        SET total_num TO right_num MINUS left_num  

        IF total_x EQUALS zero  
            IF total_num EQUALS zero  
                RETURN phrase infinite solutions  
            ELSE  
                RETURN phrase no solution  
            END IF  
        ELSE  
            SET result TO integer division of total_num BY total_x  
            RETURN concatenation of character x equals and result  
        END IF  
    END FUNCTION  
END CLASS
