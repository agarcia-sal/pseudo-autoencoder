CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO the LENGTH OF nums1  
        SET n TO the LENGTH OF nums2  
        INITIALIZE dp AS a TWO DIMENSIONAL COLLECTION OF SIZE m PLUS one BY n PLUS one WITH each element SET TO a NEGATIVE INFINITY REPRESENTATION  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET current_product TO element at position i MINUS one OF nums1 MULTIPLIED BY element at position j MINUS one OF nums2  
                SET dp element at position i and j TO the MAXIMUM value AMONG current_product, dp element at position i MINUS one and j, dp element at position i and j MINUS one, dp element at position i MINUS one and j MINUS one PLUS current_product  
            END FOR  
        END FOR  
        RETURN dp element at position m and n  
    END FUNCTION  
END CLASS
