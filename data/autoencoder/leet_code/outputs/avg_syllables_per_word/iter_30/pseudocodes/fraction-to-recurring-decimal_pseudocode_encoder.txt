CLASS Solution  
    FUNCTION fractionToDecimal(numerator, denominator)  
        IF numerator LESS THAN zero AND denominator GREATER THAN OR EQUAL TO zero OR numerator GREATER THAN OR EQUAL TO zero AND denominator LESS THAN zero  
            SET sign TO minus sign  
        ELSE  
            SET sign TO empty string  
        END IF  
  
        SET numerator TO absolute value of numerator  
        SET denominator TO absolute value of denominator  
  
        SET quotient and remainder TO integer division result and remainder of numerator divided by denominator  
        SET integer_part TO string representation of quotient  
  
        IF remainder EQUALS zero  
            RETURN concatenation of sign and integer_part  
        END IF  
  
        SET seen_remainders TO empty dictionary  
        SET fractional_part TO empty list  
  
        WHILE remainder NOT EQUALS zero  
            IF remainder EXISTS IN seen_remainders  
                SET start TO the value of remainder in seen_remainders  
                SET non_repeating TO concatenation of elements from zero TO one less than start IN fractional_part  
                SET repeating TO concatenation of elements from start TO end IN fractional_part  
                RETURN concatenation of sign, integer_part, period, non_repeating, open parenthesis, repeating, and close parenthesis  
            END IF  
  
            SET the value of remainder in seen_remainders TO current length of fractional_part  
  
            SET remainder TO remainder MULTIPLIED BY ten  
            SET quotient and remainder TO integer division result and remainder of remainder divided by denominator  
            APPEND string representation of quotient TO fractional_part  
        END WHILE  
  
        RETURN concatenation of sign, integer_part, period, and concatenation of all elements in fractional_part  
    END FUNCTION  
END CLASS
