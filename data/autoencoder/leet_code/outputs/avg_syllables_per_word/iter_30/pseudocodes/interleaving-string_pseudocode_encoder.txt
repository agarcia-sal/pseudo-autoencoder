CLASS Solution  
    FUNCTION isInterleave(s1, s2, s3)  
        IF the LENGTH OF s1 PLUS the LENGTH OF s2 NOT EQUALS the LENGTH OF s3  
            RETURN false  
        END IF  
        SET dp TO a list of false values with the LENGTH OF s2 PLUS one elements  
        SET the element at position zero OF dp TO true  
        FOR i FROM zero TO the LENGTH OF s1  
            FOR j FROM zero TO the LENGTH OF s2  
                IF i GREATER THAN zero  
                    SET the element at position j OF dp TO the logical AND of the current element at position j OF dp AND the equality of the character at position i MINUS one OF s1 AND the character at position i PLUS j MINUS one OF s3  
                END IF  
                IF j GREATER THAN zero  
                    SET the element at position j OF dp TO the logical OR of the current element at position j OF dp AND the logical AND of the element at position j MINUS one OF dp AND the equality of the character at position j MINUS one OF s2 AND the character at position i PLUS j MINUS one OF s3  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position the LENGTH OF s2 OF dp  
    END FUNCTION  
END CLASS
