CLASS Solution  
    FUNCTION minMutation(start, end, bank)  
        SET s TO the SET of all elements in bank  
        SET q TO a double ended queue containing a tuple of start and zero  
        SET mp TO a MAPPING where key A maps to string containing characters T C G, key T maps to string containing characters A C G, key C maps to string containing characters A T G, key G maps to string containing characters A T C  
        WHILE the LENGTH of q is greater than zero  
            SET t and step TO the tuple obtained by removing the first element of q  
            IF t EQUALS end  
                RETURN step  
            END IF  
            FOR each index i and character v obtained by enumerating the string t  
                FOR each character j in the string mapped by key v in mp  
                    SET next TO the concatenation of substring from position zero TO position i minus one OF t, character j, substring from position i plus one TO the LAST position OF t  
                    IF next IS an element of s  
                        APPEND the tuple of next and step PLUS one TO q  
                        REMOVE next FROM s  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
