CLASS Solution  
    FUNCTION maxHappyGroups(batchSize, groups)  
        IF batchSize EQUALS one  
            RETURN the LENGTH OF groups  
        END IF  
          
        SET remainder_count TO a list of zero repeated batchSize times  
        FOR each group IN groups  
            INCREMENT element at position the remainder of group DIVIDED BY batchSize OF remainder_count BY one  
        END FOR  
          
        FUNCTION dp(remainder, variable number of remainder_count)  
            SET remainder_count TO a list conversion of remainder_count  
            SET max_happy TO zero  
              
            FOR index FROM zero TO batchSize MINUS one  
                IF element at position index OF remainder_count EQUALS zero  
                    CONTINUE  
                END IF  
                  
                DECREMENT element at position index OF remainder_count BY one  
                SET new_remainder TO the remainder of remainder MINUS index DIVIDED BY batchSize  
                  
                IF remainder EQUALS zero  
                    SET happy_groups TO one PLUS dp(new_remainder, the elements of remainder_count)  
                ELSE  
                    SET happy_groups TO dp(new_remainder, the elements of remainder_count)  
                END IF  
                  
                IF happy_groups GREATER THAN max_happy  
                    SET max_happy TO happy_groups  
                END IF  
                  
                INCREMENT element at position index OF remainder_count BY one  
            END FOR  
              
            RETURN max_happy  
        END FUNCTION  
          
        RETURN dp(zero, the elements of remainder_count)  
END CLASS
