CLASS Solution  
    FUNCTION isSolvable(words, result)  
        SET unique_chars TO the SET OF all characters in concatenation of words and result  
        IF the LENGTH OF unique_chars GREATER THAN ten  
            RETURN false  
        END IF  
        SET non_zero_chars TO the SET OF first character of each word IN words WHERE the LENGTH OF word GREATER THAN one UNION the first character of result IF the LENGTH OF result GREATER THAN one OTHERWISE empty string  
        INITIALIZE char_to_digit AS empty mapping  
        INITIALIZE digit_to_char AS empty mapping  
        
        FUNCTION can_assign(char, digit)  
            RETURN char NOT IN char_to_digit AND (digit NOT IN digit_to_char OR digit_to_char[digit] EQUALS char)  
        END FUNCTION  
        
        FUNCTION assign(char, digit)  
            SET char_to_digit[char] TO digit  
            SET digit_to_char[digit] TO char  
        END FUNCTION  
        
        FUNCTION unassign(char, digit)  
            REMOVE char FROM char_to_digit  
            REMOVE digit FROM digit_to_char  
        END FUNCTION  
        
        FUNCTION backtrack(index, col, sum_col)  
            IF col GREATER THAN OR EQUAL TO the MAXIMUM LENGTH OF words AND col GREATER THAN OR EQUAL TO LENGTH OF result  
                RETURN sum_col EQUALS zero  
            END IF  
            IF index EQUALS LENGTH OF words  
                IF col LESS THAN LENGTH OF result  
                    SET digit_char TO character at position LENGTH OF result MINUS one MINUS col OF result  
                    IF digit_char IN char_to_digit  
                        IF char_to_digit[digit_char] EQUALS sum_col MODULO ten  
                            RETURN backtrack(zero, col PLUS one, sum_col DIVIDED BY ten)  
                        ELSE  
                            RETURN false  
                        END IF  
                    ELSE IF can_assign(digit_char, sum_col MODULO ten) AND (sum_col MODULO ten NOT EQUALS zero OR digit_char NOT IN non_zero_chars)  
                        assign(digit_char, sum_col MODULO ten)  
                        IF backtrack(zero, col PLUS one, sum_col DIVIDED BY ten)  
                            RETURN true  
                        END IF  
                        unassign(digit_char, sum_col MODULO ten)  
                    END IF  
                ELSE  
                    RETURN sum_col EQUALS zero  
                END IF  
                RETURN false  
            END IF  
            SET word TO element at position index OF words  
            IF col GREATER THAN OR EQUAL TO LENGTH OF word  
                RETURN backtrack(index PLUS one, col, sum_col)  
            END IF  
            SET char TO character at position LENGTH OF word MINUS one MINUS col OF word  
            IF char IN char_to_digit  
                RETURN backtrack(index PLUS one, col, sum_col PLUS char_to_digit[char])  
            END IF  
            FOR digit FROM zero TO nine  
                IF can_assign(char, digit) AND (digit NOT EQUALS zero OR char NOT IN non_zero_chars)  
                    assign(char, digit)  
                    IF backtrack(index PLUS one, col, sum_col PLUS digit)  
                        RETURN true  
                    END IF  
                    unassign(char, digit)  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN backtrack(zero, zero, zero)  
    END FUNCTION  
END CLASS
