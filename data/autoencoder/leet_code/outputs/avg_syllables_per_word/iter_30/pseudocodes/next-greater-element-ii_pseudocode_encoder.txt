CLASS Solution  
    FUNCTION nextGreaterElements(nums)  
        SET n TO the LENGTH OF nums  
        SET result TO a collection of n elements each EQUAL TO negative one  
        SET stack TO an empty collection  

        FOR i FROM zero TO two MULTIPLIED BY n MINUS one  
            SET index TO the remainder of i DIVIDED BY n  
            WHILE stack IS NOT empty AND element at position the last element in stack OF nums IS LESS THAN element at position index OF nums  
                SET popped_index TO the last element removed FROM stack  
                SET element at position popped_index OF result TO element at position index OF nums  
            END WHILE  
            IF i IS LESS THAN n  
                APPEND index TO stack  
            END IF  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
