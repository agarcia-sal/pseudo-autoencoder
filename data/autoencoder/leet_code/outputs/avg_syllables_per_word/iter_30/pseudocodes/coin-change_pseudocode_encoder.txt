CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        SET dp TO a collection of size amount plus one with each element set to amount plus one  
        SET element at position zero OF dp TO zero  
        FOR each a FROM one TO amount  
            FOR each coin IN coins  
                IF a MINUS coin GREATER THAN OR EQUAL TO zero  
                    SET element at position a OF dp TO the minimum of element at position a OF dp AND element at position a MINUS coin OF dp PLUS one  
                END IF  
            END FOR  
        END FOR  
        IF element at position amount OF dp NOT EQUALS amount PLUS one  
            RETURN element at position amount OF dp  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
