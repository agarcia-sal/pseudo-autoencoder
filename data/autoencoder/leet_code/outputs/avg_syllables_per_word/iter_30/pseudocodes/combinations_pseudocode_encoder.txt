CLASS Solution  
    FUNCTION combine(n, k)  
        FUNCTION backtrack(start, path)  
            IF the LENGTH OF path EQUALS k  
                APPEND a copy of path TO result  
                RETURN  
            END IF  
            FOR i FROM start TO n PLUS one  
                APPEND i TO path  
                CALL backtrack WITH i PLUS one AND path  
                REMOVE last element FROM path  
            END FOR  
        END FUNCTION  

        SET result TO empty collection  
        CALL backtrack WITH one AND empty collection  
        RETURN result  
    END FUNCTION  
END CLASS
