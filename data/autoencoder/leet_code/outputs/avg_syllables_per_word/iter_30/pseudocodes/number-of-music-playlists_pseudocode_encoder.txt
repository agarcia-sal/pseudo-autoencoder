CLASS Solution  
    FUNCTION numMusicPlaylists(n, goal, k)  
        SET MOD TO ten raised to the power of nine PLUS one  
        INITIALIZE dp AS two dimensional list with goal PLUS one rows AND n PLUS one columns ALL SET TO zero  
        SET element at position zero zero OF dp TO one  
        FOR i FROM one TO goal  
            FOR j FROM one TO n  
                SET element at position i j OF dp TO element at position i MINUS one j MINUS one OF dp MULTIPLIED BY n MINUS j MINUS one  
                SET element at position i j OF dp TO remainder WHEN element at position i j OF dp IS DIVIDED BY MOD  
                IF j GREATER THAN k  
                    INCREMENT element at position i j OF dp BY element at position i MINUS one j OF dp MULTIPLIED BY j MINUS k  
                    SET element at position i j OF dp TO remainder WHEN element at position i j OF dp IS DIVIDED BY MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position goal n OF dp  
    END FUNCTION  
END CLASS
