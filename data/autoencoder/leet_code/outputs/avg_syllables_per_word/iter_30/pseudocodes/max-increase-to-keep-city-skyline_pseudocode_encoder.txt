CLASS Solution  
    FUNCTION maxIncreaseKeepingSkyline(grid)  
        SET n TO the LENGTH OF grid  
        SET row_maxes TO empty list  
        FOR each row IN grid  
            APPEND the MAXIMUM value IN row TO row_maxes  
        END FOR  
        SET col_maxes TO empty list  
        FOR each column IN the transposition of grid  
            APPEND the MAXIMUM value IN column TO col_maxes  
        END FOR  
        SET total_increase TO zero  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                SET max_height TO the MINIMUM OF the element at position i OF row_maxes AND the element at position j OF col_maxes  
                IF max_height GREATER THAN the element at position i AND position j OF grid  
                    INCREMENT total_increase BY max_height MINUS the element at position i AND position j OF grid  
                END IF  
            END FOR  
        END FOR  
        RETURN total_increase  
    END FUNCTION  
END CLASS
