CLASS Solution  
    FUNCTION maxVacationDays(flights, days)  
        SET n TO the LENGTH OF flights  
        SET k TO the LENGTH OF the element at position zero OF days  
        SET dp TO a list of length n filled with minus one  
        SET element at position zero OF dp TO zero  
        SET graph TO a dictionary with keys from zero TO n minus one and values as empty lists  
        FOR each i FROM zero TO n minus one  
            FOR each j FROM zero TO n minus one  
                IF element at position j OF element at position i OF flights EQUALS one  
                    APPEND j TO the list at key i OF graph  
                END IF  
            END FOR  
        END FOR  
        FOR each week FROM zero TO k minus one  
            SET new_dp TO a list of length n filled with minus one  
            FOR each i FROM zero TO n minus one  
                IF element at position i OF dp EQUALS minus one  
                    CONTINUE  
                END IF  
                IF element at position i OF new_dp LESS THAN element at position i OF dp PLUS element at position week OF element at position i OF days  
                    SET element at position i OF new_dp TO element at position i OF dp PLUS element at position week OF element at position i OF days  
                END IF  
                FOR each j IN the list at key i OF graph  
                    IF element at position j OF new_dp LESS THAN element at position i OF dp PLUS element at position week OF element at position j OF days  
                        SET element at position j OF new_dp TO element at position i OF dp PLUS element at position week OF element at position j OF days  
                    END IF  
                END FOR  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        SET max_value TO minus infinity  
        FOR each value IN dp  
            IF value GREATER THAN max_value  
                SET max_value TO value  
            END IF  
        END FOR  
        RETURN max_value  
    END FUNCTION  
END CLASS
