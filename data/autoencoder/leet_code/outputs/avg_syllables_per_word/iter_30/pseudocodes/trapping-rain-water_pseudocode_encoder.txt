CLASS Solution  
    FUNCTION trap(height)  
        IF height EQUALS empty collection  
            RETURN zero  
        END IF  
        SET n TO the LENGTH OF height  
        SET left_max TO collection of zero repeated n times  
        SET right_max TO collection of zero repeated n times  
        SET element at position zero OF left_max TO element at position zero OF height  
        FOR index FROM one TO n MINUS one  
            SET element at position index OF left_max TO the GREATER OF element at position index MINUS one OF left_max AND element at position index OF height  
        END FOR  
        SET element at position n MINUS one OF right_max TO element at position n MINUS one OF height  
        FOR index FROM n MINUS two DOWN TO zero  
            SET element at position index OF right_max TO the GREATER OF element at position index PLUS one OF right_max AND element at position index OF height  
        END FOR  
        SET trapped_water TO zero  
        FOR index FROM zero TO n MINUS one  
            INCREMENT trapped_water BY the SMALLER OF element at position index OF left_max AND element at position index OF right_max MINUS element at position index OF height  
        END FOR  
        RETURN trapped_water  
    END FUNCTION  
END CLASS
