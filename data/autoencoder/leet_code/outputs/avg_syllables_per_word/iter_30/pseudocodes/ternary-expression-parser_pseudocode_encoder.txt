CLASS Solution  
    FUNCTION parseTernary(expression)  
        FUNCTION evaluate(index)  
            IF index EQUALS the LENGTH OF expression MINUS one OR character at position index PLUS one OF expression EQUALS colon  
                RETURN character at position index OF expression AND index PLUS two  
            END IF  
            SET true_part AND next_index TO THE RESULT OF evaluate WITH index PLUS two  
            SET false_part AND final_index TO THE RESULT OF evaluate WITH next_index  
            IF character at position index OF expression EQUALS uppercase tee  
                RETURN true_part AND final_index  
            ELSE  
                RETURN false_part AND final_index  
            END IF  
        END FUNCTION  
        SET result AND ignored_value TO THE RESULT OF evaluate WITH zero  
        RETURN result  
    END FUNCTION  
END CLASS
