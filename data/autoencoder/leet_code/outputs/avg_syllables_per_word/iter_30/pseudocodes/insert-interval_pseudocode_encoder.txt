CLASS Solution  
    FUNCTION insert(intervals, newInterval)  
        IF intervals EQUALS empty list  
            RETURN list containing newInterval  
        END IF  
        SET merged TO empty list  
        SET i TO zero  
        SET n TO the LENGTH OF intervals  
        WHILE i LESS THAN n AND element at position i of intervals at position one LESS THAN element at position zero OF newInterval  
            APPEND element at position i OF intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        WHILE i LESS THAN n AND element at position i OF intervals at position zero LESS THAN OR EQUAL TO element at position one OF newInterval  
            SET element at position zero OF newInterval TO the MINIMUM OF element at position zero OF newInterval AND element at position zero OF intervals at position i  
            SET element at position one OF newInterval TO the MAXIMUM OF element at position one OF newInterval AND element at position one OF intervals at position i  
            INCREMENT i BY one  
        END WHILE  
        APPEND newInterval TO merged  
        WHILE i LESS THAN n  
            APPEND element at position i OF intervals TO merged  
            INCREMENT i BY one  
        END WHILE  
        RETURN merged  
    END FUNCTION  
END CLASS
