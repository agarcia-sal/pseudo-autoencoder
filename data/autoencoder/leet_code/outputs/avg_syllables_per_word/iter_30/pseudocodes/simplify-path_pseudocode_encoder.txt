CLASS Solution  
    FUNCTION simplifyPath(path)  
        SET components TO the result of splitting path by the character slash  
        SET stack TO an empty collection  
        FOR each component IN components  
            IF component EQUALS empty string OR component EQUALS the string with a single period  
                CONTINUE  
            ELSE IF component EQUALS the string with two periods  
                IF the LENGTH OF stack GREATER THAN zero  
                    REMOVE the last element FROM stack  
                END IF  
            ELSE  
                APPEND component TO stack  
            END IF  
        END FOR  
        SET simplified_path TO the string consisting of a single slash concatenated with the concatenation of elements of stack separated by the character slash  
        RETURN simplified_path  
    END FUNCTION  
END CLASS
