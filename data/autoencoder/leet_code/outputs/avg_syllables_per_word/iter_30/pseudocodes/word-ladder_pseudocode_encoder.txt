CLASS Solution  
    FUNCTION ladderLength(beginWord, endWord, wordList)  
        IF endWord NOT IN wordList THEN  
            RETURN zero  
        END IF  
        SET word_dict TO an empty dictionary with default empty list  
        SET word_length TO the LENGTH OF beginWord  
        FOR each word IN wordList  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_key TO substring from zero TO position i OF word PLUS the character star PLUS substring from position i PLUS one TO the END OF word  
                APPEND word TO the list at intermediate_key IN word_dict  
            END FOR  
        END FOR  
        SET queue TO a new double ended queue containing the tuple of beginWord and one  
        SET visited TO a set containing beginWord  
        WHILE the queue IS NOT empty  
            SET current_word and current_length TO the first element removed from queue  
            FOR i FROM zero TO word_length MINUS one  
                SET intermediate_word TO substring from zero TO position i OF current_word PLUS the character star PLUS substring from position i PLUS one TO the END OF current_word  
                FOR each word IN the list at intermediate_word IN word_dict  
                    IF word EQUALS endWord THEN  
                        RETURN current_length PLUS one  
                    END IF  
                    IF word NOT IN visited THEN  
                        ADD word TO visited  
                        APPEND the tuple of word and current_length PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN zero  
    END FUNCTION  
END CLASS
