CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val EQUALS element at position zero OF values  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front OF queue  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with val EQUALS element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with val EQUALS element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree of p.left AND q.left AND is_same_tree of p.right AND q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION checkEqualTree(self, root)  
        FUNCTION subtree_sum(node)  
            IF node EQUALS none  
                RETURN zero  
            END IF  
            RETURN node.val PLUS subtree_sum of node.left PLUS subtree_sum of node.right  
        END FUNCTION

        SET total_sum TO subtree_sum of root  

        IF total_sum MODULO two NOT EQUALS zero  
            RETURN False  
        END IF  

        SET target TO total_sum DIVIDED BY two  
        SET found TO list containing False  

        FUNCTION can_split(node)  
            IF node EQUALS none  
                RETURN zero  
            END IF  
            SET current_sum TO node.val PLUS can_split of node.left PLUS can_split of node.right  
            IF current_sum EQUALS target AND node NOT EQUALS root  
                SET element at position zero OF found TO True  
            END IF  
            RETURN current_sum  
        END FUNCTION

        CALL can_split with root  
        RETURN element at position zero OF found  
    END FUNCTION  
END CLASS
