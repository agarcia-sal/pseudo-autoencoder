CLASS Solution  
    FUNCTION wiggleSort(nums)  
        CALL sort ON nums  
        SET sorted_nums TO copy of nums  
        SET n TO length of nums  
        SET left TO integer division of n MINUS one BY two  
        SET right TO n MINUS one  
        FOR i FROM zero TO n MINUS one  
            IF remainder of i DIVIDED BY two EQUALS zero  
                SET element at position i OF nums TO element at position left OF sorted_nums  
                DECREMENT left BY one  
            ELSE  
                SET element at position i OF nums TO element at position right OF sorted_nums  
                DECREMENT right BY one  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
