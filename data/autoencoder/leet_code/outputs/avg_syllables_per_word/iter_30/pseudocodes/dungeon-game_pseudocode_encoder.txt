CLASS Solution  
    FUNCTION calculateMinimumHP(dungeon)  
        SET m TO the LENGTH OF dungeon  
        SET n TO the LENGTH OF the element at position zero OF dungeon  
        SET dp TO a two-dimensional collection with m PLUS one rows and n PLUS one columns filled with infinity  
        SET the element at position m and position n MINUS one OF dp TO one  
        SET the element at position m MINUS one and position n OF dp TO one  
        FOR i FROM m MINUS one DOWN TO zero  
            FOR j FROM n MINUS one DOWN TO zero  
                SET min_health TO the MINIMUM OF the element at position i PLUS one and position j OF dp AND the element at position i and position j PLUS one OF dp MINUS the element at position i and position j OF dungeon  
                IF min_health LESS THAN one  
                    SET dp at position i and position j TO one  
                ELSE  
                    SET dp at position i and position j TO min_health  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position zero and position zero OF dp  
    END FUNCTION  
END CLASS
