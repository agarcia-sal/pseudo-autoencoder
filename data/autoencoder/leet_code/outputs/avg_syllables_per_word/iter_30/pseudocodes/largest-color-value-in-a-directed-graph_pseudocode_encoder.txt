CLASS Solution  
    FUNCTION largestPathValue(colors, edges)  
        SET n TO the LENGTH OF colors  
        ASSIGN reference TO an empty mapping from integer TO list named graph  
        SET in_degree TO a list of zero repeated n times  
        
        FOR each pair u and v IN edges  
            APPEND v TO the list at key u IN graph  
            INCREMENT the element at position v OF in_degree BY one  
        END FOR  
        
        ASSIGN reference TO a double ended queue containing all integers i from zero TO n MINUS one WHERE the element at position i OF in_degree EQUALS zero  
        SET dp TO a list containing n elements each of which IS a list of zero repeated twenty six times  
        SET visited TO zero  
        SET max_color_value TO zero  
        
        WHILE the queue holds elements  
            REMOVE and RETURN the element at the front of the queue INTO node  
            INCREMENT the element at position the Unicode code point of the character at position node OF colors MINUS the Unicode code point of lowercase letter a OF the list at position node OF dp BY one  
            SET max_color_value TO the greater of max_color_value AND the element at position the Unicode code point of the character at position node OF colors MINUS the Unicode code point of lowercase letter a OF the list at position node OF dp  
            INCREMENT visited BY one  
            
            FOR each neighbor IN the list at key node IN graph  
                FOR each integer c FROM zero TO twenty five  
                    SET the element at position c OF the list at position neighbor OF dp TO the greater of the element at position c OF the list at position neighbor OF dp AND the element at position c OF the list at position node OF dp  
                END FOR  
                DECREMENT the element at position neighbor OF in_degree BY one  
                IF the element at position neighbor OF in_degree EQUALS zero  
                    APPEND neighbor TO the queue  
                END IF  
            END FOR  
        END WHILE  
        
        IF visited EQUALS n  
            RETURN max_color_value  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
