CLASS Solution  
    FUNCTION maxTaskAssign(tasks, workers, pills, strength)  
        SORT tasks in ascending order  
        SORT workers in ascending order  
        
        FUNCTION canAssignTasks(k)  
            SET available_workers TO last k elements of workers  
            SET pills_used TO zero  
            
            FOR i FROM k MINUS one DOWN TO zero  
                SET task TO element at position i of tasks  
                SET worker_idx TO position of first element in available_workers GREATER THAN OR EQUAL TO task  
                IF worker_idx LESS THAN LENGTH OF available_workers  
                    REMOVE element at position worker_idx FROM available_workers  
                ELSE IF pills_used LESS THAN pills  
                    SET worker_idx TO position of first element in available_workers GREATER THAN OR EQUAL TO task MINUS strength  
                    IF worker_idx LESS THAN LENGTH OF available_workers  
                        REMOVE element at position worker_idx FROM available_workers  
                        INCREMENT pills_used BY one  
                    ELSE  
                        RETURN false  
                    END IF  
                ELSE  
                    RETURN false  
                END IF  
            END FOR  
            RETURN true  
        END FUNCTION  
        
        SET left TO zero  
        SET right TO the minimum of LENGTH OF tasks AND LENGTH OF workers PLUS one  
        WHILE left LESS THAN right  
            SET mid TO left PLUS right DIVIDED BY two  
            IF canAssignTasks(mid) IS true  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN left MINUS one  
    END FUNCTION  
END CLASS
