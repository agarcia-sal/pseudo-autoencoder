CLASS Solution  
    FUNCTION mostSimilar(n, roads, names, targetPath)  
        ASSIGN reference TO a new default dictionary of lists NAMED graph  
        FOR each pair u and v IN roads  
            APPEND v TO graph at position u  
            APPEND u TO graph at position v  
        END FOR  
        SET m TO the length of targetPath  
        
        FUNCTION dp(i, j)  
            IF i EQUALS zero  
                IF the element at position j OF names DOES NOT EQUAL the element at position i OF targetPath  
                    RETURN one AND a list containing j  
                ELSE  
                    RETURN zero AND a list containing j  
                END IF  
            END IF  
            SET min_cost TO positive infinity  
            SET min_path TO an empty list  
            FOR each element k IN graph at position j  
                SET cost AND path TO the result of dp called with arguments i MINUS one AND k  
                IF cost LESS THAN min_cost  
                    SET min_cost TO cost  
                    SET min_path TO path  
                END IF  
            END FOR  
            IF the element at position j OF names DOES NOT EQUAL the element at position i OF targetPath  
                SET current_cost TO min_cost PLUS one  
            ELSE  
                SET current_cost TO min_cost  
            END IF  
            RETURN current_cost AND the concatenation of min_path WITH a list containing j  
        END FUNCTION  
        
        SET min_cost TO positive infinity  
        SET min_path TO an empty list  
        FOR index j FROM zero TO n MINUS one  
            SET cost AND path TO the result of dp called with arguments m MINUS one AND j  
            IF cost LESS THAN min_cost  
                SET min_cost TO cost  
                SET min_path TO path  
            END IF  
        END FOR  
        RETURN min_path  
    END FUNCTION  
END CLASS
