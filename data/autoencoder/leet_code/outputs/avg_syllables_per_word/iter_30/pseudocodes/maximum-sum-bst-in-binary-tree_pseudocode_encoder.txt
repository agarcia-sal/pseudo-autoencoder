CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val as element at position zero OF values  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from front of queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values NOT EQUALS none  
            SET node.left TO new TreeNode with val as element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values NOT EQUALS none  
            SET node.right TO new TreeNode with val as element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxSumBST(self, root)  
        SET self.max_sum TO zero  
        
        FUNCTION helper(node)  
            IF node EQUALS none  
                RETURN true, zero, positive infinity, negative infinity  
            END IF  
            
            SET left_is_bst, left_sum, left_min, left_max TO helper(node.left)  
            SET right_is_bst, right_sum, right_min, right_max TO helper(node.right)  
            
            IF left_is_bst AND right_is_bst AND left_max LESS THAN node.val AND node.val LESS THAN right_min  
                SET current_sum TO left_sum PLUS right_sum PLUS node.val  
                SET current_min TO the minimum value between node.val AND left_min  
                SET current_max TO the maximum value between node.val AND right_max  
                SET self.max_sum TO the maximum value between self.max_sum AND current_sum  
                RETURN true, current_sum, current_min, current_max  
            END IF  
            
            RETURN false, zero, positive infinity, negative infinity  
        END FUNCTION  
        
        CALL helper WITH root  
        RETURN self.max_sum  
    END FUNCTION  
END CLASS
