CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with val as element at position zero OF values and next as none  
    SET p TO head  
    FOR each val IN elements at positions from one TO the end OF values  
        SET node TO new ListNode with val as val and next as none  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val as element at position zero OF values and left as none and right as none  
    SET i TO one  
    SET queue TO a new empty deque collection  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with val as element at position i OF values and left as none and right as none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with val as element at position i OF values and left as none and right as none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION str2tree(self, s)  
        FUNCTION parse_tree(index)  
            IF index GREATER THAN OR EQUAL TO the LENGTH OF s  
                RETURN none, index  
            END IF  
            SET start TO index  
            WHILE index LESS THAN the LENGTH OF s AND (element at position index OF s IS a digit OR element at position index OF s EQUALS the character minus)  
                INCREMENT index BY one  
            END WHILE  
            SET node TO new TreeNode with val as integer value of substring from position start TO position index OF s and left as none and right as none  
            IF index LESS THAN the LENGTH OF s AND element at position index OF s EQUALS the character open parenthesis  
                SET node.left, index TO parse_tree(index PLUS one)  
            END IF  
            IF index LESS THAN the LENGTH OF s AND element at position index OF s EQUALS the character open parenthesis  
                SET node.right, index TO parse_tree(index PLUS one)  
            END IF  
            IF index LESS THAN the LENGTH OF s AND element at position index OF s EQUALS the character close parenthesis  
                INCREMENT index BY one  
            END IF  
            RETURN node, index  
        END FUNCTION  
        SET root, unused TO parse_tree(zero)  
        RETURN root  
    END FUNCTION  
END CLASS
