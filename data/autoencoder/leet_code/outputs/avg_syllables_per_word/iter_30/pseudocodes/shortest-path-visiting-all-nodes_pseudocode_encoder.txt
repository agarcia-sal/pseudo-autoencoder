CLASS Solution  
    FUNCTION shortestPathLength(graph)  
        IF graph EQUALS nothing OR the LENGTH OF graph EQUALS zero  
            RETURN zero  
        END IF  
        SET n TO the LENGTH OF graph  
        INITIALIZE queue TO a deque CONTAINING tuples of each node FROM zero TO n MINUS one AND one LEFT SHIFTED BY node  
        INITIALIZE visited TO a set CONTAINING tuples of each node FROM zero TO n MINUS one AND one LEFT SHIFTED BY node  
        SET steps TO zero  
        SET all_nodes_visited TO one LEFT SHIFTED BY n MINUS one  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET current_level_length TO the LENGTH OF queue  
            FOR each index FROM zero TO current_level_length MINUS one  
                REMOVE the FIRST element FROM queue AND ASSIGN TO current_node AND visited_nodes  
                IF visited_nodes EQUALS all_nodes_visited  
                    RETURN steps  
                END IF  
                FOR each neighbor IN the collection at position current_node OF graph  
                    SET next_visited_nodes TO visited_nodes OR one LEFT SHIFTED BY neighbor  
                    SET next_state TO a tuple of neighbor AND next_visited_nodes  
                    IF next_state NOT IN visited  
                        ADD next_state TO visited  
                        APPEND next_state TO queue  
                    END IF  
                END FOR  
            END FOR  
            INCREMENT steps BY one  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
