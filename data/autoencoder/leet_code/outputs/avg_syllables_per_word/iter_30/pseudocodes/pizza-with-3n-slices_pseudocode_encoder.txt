CLASS Solution  
    FUNCTION maxSizeSlices WITH PARAMETER slices  
        SET n TO the LENGTH OF slices DIVIDED BY three  
        
        FUNCTION dp WITH PARAMETERS start, end, k  
            IF k EQUALS zero OR start GREATER THAN end  
                RETURN zero  
            END IF  
            SET pick TO element at position start OF slices PLUS dp WITH start INCREMENTED BY two, end, AND k DECREMENTED BY one  
            SET skip TO dp WITH start INCREMENTED BY one, end, AND k  
            IF pick GREATER THAN skip  
                RETURN pick  
            ELSE  
                RETURN skip  
            END IF  
        END FUNCTION  
        
        SET case1 TO dp WITH start as one, end as LENGTH OF slices MINUS one, AND k as n  
        SET case2 TO dp WITH start as zero, end as LENGTH OF slices MINUS two, AND k as n  
        
        IF case1 GREATER THAN case2  
            RETURN case1  
        ELSE  
            RETURN case2  
        END IF  
    END FUNCTION  
END CLASS
