CLASS Solution  
    FUNCTION minDifficulty(jobDifficulty, d)  
        SET n TO the LENGTH OF jobDifficulty  
        IF n LESS THAN d  
            RETURN negative one  
        END IF  
        
        FUNCTION dp(i, d)  
            IF d EQUALS one  
                RETURN the MAXIMUM value AMONG elements FROM position i TO the END OF jobDifficulty  
            END IF  
            
            SET min_difficulty TO positive infinity  
            SET current_max TO zero  
            FOR each index j FROM i TO n MINUS d  
                SET current_max TO the MAXIMUM value BETWEEN current_max AND element at position j OF jobDifficulty  
                SET candidate_difficulty TO current_max PLUS dp(j PLUS one, d MINUS one)  
                IF candidate_difficulty LESS THAN min_difficulty  
                    SET min_difficulty TO candidate_difficulty  
                END IF  
            END FOR  
            
            RETURN min_difficulty  
        END FUNCTION  
        
        RETURN dp(zero, d)  
    END FUNCTION  
END CLASS
