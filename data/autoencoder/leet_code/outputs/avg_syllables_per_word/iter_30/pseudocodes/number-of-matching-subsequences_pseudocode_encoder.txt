CLASS Solution  
    FUNCTION numMatchingSubseq(s, words)  
        SET waiting TO a new empty dictionary  
        FOR each character c IN the set of characters of s  
            ASSIGN waiting at key c TO a new empty deque  
        END FOR  
        FOR each word IN words  
            ASSIGN it TO an iterator over word  
            ASSIGN first_letter TO the next character from it  
            IF first_letter is a key in waiting  
                APPEND it TO waiting at key first_letter  
            END IF  
        END FOR  
        SET match_count TO zero  
        FOR each character c IN s  
            IF c is a key in waiting  
                ASSIGN current_iterators TO waiting at key c  
                ASSIGN waiting at key c TO a new empty deque  
                FOR each iterator it IN current_iterators  
                    ASSIGN next_char TO the next character from it or none if no next character exists  
                    IF next_char is none  
                        INCREMENT match_count BY one  
                    ELSE  
                        IF next_char is a key in waiting  
                            APPEND it TO waiting at key next_char  
                        END IF  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        RETURN match_count  
    END FUNCTION  
END CLASS
