CLASS Solution  
    FUNCTION criticalConnections(n, connections)  
        FUNCTION dfs(node, parent, disc, low, time, graph, critical)  
            SET disc at position node TO the element at position zero of time  
            SET low at position node TO the element at position zero of time  
            INCREMENT the element at position zero of time BY one  
            FOR each neighbor IN the element at position node OF graph  
                IF the element at position neighbor OF disc EQUALS negative one THEN  
                    CALL dfs WITH neighbor, node, disc, low, time, graph, critical  
                    SET low at position node TO the lesser value BETWEEN low at position node AND low at position neighbor  
                    IF low at position neighbor GREATER THAN disc at position node THEN  
                        APPEND list containing node AND neighbor TO critical  
                    END IF  
                ELSE IF neighbor NOT EQUALS parent THEN  
                    SET low at position node TO the lesser value BETWEEN low at position node AND disc at position neighbor  
                END IF  
            END FOR  
        END FUNCTION  

        SET graph TO a list containing empty lists repeated n times  
        FOR each u, v IN connections  
            APPEND v TO the element at position u OF graph  
            APPEND u TO the element at position v OF graph  
        END FOR  

        SET disc TO a list containing negative one repeated n times  
        SET low TO a list containing negative one repeated n times  
        SET time TO a list containing zero  
        SET critical TO an empty list  
        
        FOR each node FROM zero TO n minus one  
            IF the element at position node OF disc EQUALS negative one THEN  
                CALL dfs WITH node, negative one, disc, low, time, graph, critical  
            END IF  
        END FOR  
        
        RETURN critical  
    END FUNCTION  
END CLASS
