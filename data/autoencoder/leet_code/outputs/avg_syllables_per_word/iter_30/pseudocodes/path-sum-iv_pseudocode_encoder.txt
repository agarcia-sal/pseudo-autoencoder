CLASS Solution  
    FUNCTION pathSum(nums)  
        SET tree TO empty dictionary  
        FOR each num IN nums  
            SET depth TO num DIVIDED BY one hundred  
            SET pos TO num DIVIDED BY ten MINUS (depth MULTIPLIED BY ten)  
            SET value TO num MINUS (num DIVIDED BY ten MULTIPLIED BY ten)  
            ASSIGN value TO tree at key composed of depth and pos  
        END FOR  
        FUNCTION dfs(depth, pos, path_sum)  
            SET current_value TO tree value at key composed of depth and pos OR none if not present  
            IF current_value EQUALS none  
                RETURN zero  
            END IF  
            SET new_path_sum TO path_sum PLUS current_value  
            SET left_child TO tuple of depth PLUS one and pos MULTIPLIED BY two MINUS one  
            SET right_child TO tuple of depth PLUS one and pos MULTIPLIED BY two  
            IF left_child NOT IN tree AND right_child NOT IN tree  
                RETURN new_path_sum  
            END IF  
            SET left_sum TO dfs with arguments left_child depth, left_child position, new_path_sum  
            SET right_sum TO dfs with arguments right_child depth, right_child position, new_path_sum  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN dfs with arguments one, one, zero  
    END FUNCTION  
END CLASS
