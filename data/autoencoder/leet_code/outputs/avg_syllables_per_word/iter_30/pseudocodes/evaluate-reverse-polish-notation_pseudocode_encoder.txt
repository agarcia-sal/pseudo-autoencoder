CLASS Solution  
    FUNCTION evalRPN(tokens)  
        SET stack TO empty collection  
        FOR each token IN tokens  
            IF token EQUALS plus OR token EQUALS minus OR token EQUALS multiplied BY OR token EQUALS divided BY  
                SET b TO the last element REMOVED FROM stack  
                SET a TO the last element REMOVED FROM stack  
                IF token EQUALS plus  
                    APPEND a PLUS b TO stack  
                ELSE IF token EQUALS minus  
                    APPEND a MINUS b TO stack  
                ELSE IF token EQUALS multiplied BY  
                    APPEND a MULTIPLIED BY b TO stack  
                ELSE IF token EQUALS divided BY  
                    APPEND the integer part of a DIVIDED BY b TO stack  
                END IF  
            ELSE  
                APPEND the integer value of token TO stack  
            END IF  
        END FOR  
        RETURN the element at position zero OF stack  
    END FUNCTION  
END CLASS
