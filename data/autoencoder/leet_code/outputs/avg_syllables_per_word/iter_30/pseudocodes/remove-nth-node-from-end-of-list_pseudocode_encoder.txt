CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with val as element at position zero OF values and next as none  
    SET p TO head  
    FOR each val IN elements from position one TO the end OF values  
        SET node TO new ListNode with val as val and next as none  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN p1.val EQUALS p2.val AND the result of is_same_list with p1.next and p2.next  
END FUNCTION  

CLASS Solution  
    FUNCTION removeNthFromEnd(self, head, n)  
        SET dummy TO new ListNode with val as zero and next as none  
        SET dummy.next TO head  
        SET first TO dummy  
        SET second TO dummy  
        FOR i FROM zero TO n PLUS one  
            SET first TO first.next  
        END FOR  
        WHILE first NOT EQUALS none  
            SET first TO first.next  
            SET second TO second.next  
        END WHILE  
        SET second.next TO second.next.next  
        RETURN dummy.next  
    END FUNCTION  
END CLASS
