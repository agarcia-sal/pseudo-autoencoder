CLASS Solution  
    FUNCTION lengthOfLIS(nums)  
        IF nums EQUALS empty collection  
            RETURN zero  
        END IF  
        SET dp TO empty collection  
        FOR each num IN nums  
            SET idx TO the position found by binary search left in dp for num  
            IF idx EQUALS the LENGTH OF dp  
                APPEND num TO dp  
            ELSE  
                SET element at position idx OF dp TO num  
            END IF  
        END FOR  
        RETURN the LENGTH OF dp  
    END FUNCTION  
END CLASS
