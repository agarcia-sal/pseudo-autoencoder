CLASS Solution  
    FUNCTION longestSubsequenceRepeatedK(s, k)  
        FROM the collection named Counter IMPORT  
        
        SET freq TO the frequency count of characters in s using Counter  
        
        SET valid_chars TO a collection of characters char FROM freq keys WHERE the frequency count at char IS GREATER THAN OR EQUAL TO k  
        
        SORT valid_chars in reverse lexicographical order  
        
        FUNCTION is_valid_subsequence(candidate)  
            SET count TO zero  
            SET index TO zero  
            FOR each character char IN s  
                IF char EQUALS the character at position index OF candidate THEN  
                    INCREMENT index BY one  
                    IF index EQUALS the length of candidate THEN  
                        INCREMENT count BY one  
                        SET index TO zero  
                        IF count EQUALS k THEN  
                            RETURN true  
                        END IF  
                    END IF  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        SET max_length TO the length of valid_chars DIVIDED BY k  
        SET best_candidate TO empty string  
        
        FUNCTION backtrack(candidate)  
            IF the length of candidate IS GREATER THAN the length of best_candidate THEN  
                SET best_candidate TO candidate  
            END IF  
            FOR each character char IN valid_chars  
                SET new_candidate TO candidate PLUS char  
                IF is_valid_subsequence(new_candidate) THEN  
                    CALL backtrack WITH new_candidate  
                END IF  
            END FOR  
        END FUNCTION  
        
        CALL backtrack WITH empty string  
        RETURN best_candidate  
    END FUNCTION  
END CLASS
