CLASS Solution  
    FUNCTION findMinHeightTrees(n, edges)  
        IF n EQUALS one  
            RETURN list containing zero  
        END IF  
        ASSIGN reference TO an empty mapping from keys to lists named graph  
        SET degree TO a list of length n filled with zero  
        FOR each pair u and v IN edges  
            APPEND v TO the list corresponding to u in graph  
            APPEND u TO the list corresponding to v in graph  
            INCREMENT degree at position u BY one  
            INCREMENT degree at position v BY one  
        END FOR  
        ASSIGN reference TO a queue named leaves consisting of all indices i from zero TO n MINUS one WHERE degree at position i EQUALS one  
        SET remaining_nodes TO n  
        WHILE remaining_nodes GREATER THAN two  
            SET leaves_count TO the length of leaves  
            DECREMENT remaining_nodes BY leaves_count  
            FOR index FROM zero TO leaves_count MINUS one  
                REMOVE the element from the front of leaves and ASSIGN it TO leaf  
                FOR each neighbor IN graph at position leaf  
                    DECREMENT degree at position neighbor BY one  
                    IF degree at position neighbor EQUALS one  
                        APPEND neighbor TO leaves  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        RETURN the list converted from leaves  
    END FUNCTION  
END CLASS
