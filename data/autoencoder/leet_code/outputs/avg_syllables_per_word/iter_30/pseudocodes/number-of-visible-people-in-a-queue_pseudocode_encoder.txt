CLASS Solution  
    FUNCTION canSeePersonsCount(heights)  
        SET n TO the LENGTH OF heights  
        SET answer TO a list of zero repeated n times  
        SET stack TO an empty list  
        FOR each i FROM zero TO n MINUS one  
            WHILE the LENGTH OF stack is greater than zero AND the element at position the last element of stack OF heights is LESS THAN the element at position i OF heights  
                INCREMENT the element at position the last element popped FROM stack OF answer BY one  
            END WHILE  
            IF the LENGTH OF stack is greater than zero  
                INCREMENT the element at position the last element of stack OF answer BY one  
            END IF  
            APPEND i TO stack  
        END FOR  
        RETURN answer  
    END FUNCTION  
END CLASS
