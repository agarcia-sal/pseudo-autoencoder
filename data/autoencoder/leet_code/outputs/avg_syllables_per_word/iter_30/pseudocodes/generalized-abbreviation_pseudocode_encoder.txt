CLASS Solution  
    FUNCTION generateAbbreviations WITH PARAMETERS word  
        FUNCTION backtrack WITH PARAMETERS index path count  
            IF index EQUALS the LENGTH OF word  
                IF count GREATER THAN zero  
                    APPEND the PHRASE representing count TO path  
                END IF  
                APPEND the CONCATENATION OF all elements IN path TO result  
            ELSE  
                IF count GREATER THAN zero  
                    CALL backtrack WITH index INCREMENTED BY one AND path WITH the PHRASE representing count AND the element at position index OF word APPENDED AND count SET TO zero  
                ELSE  
                    CALL backtrack WITH index INCREMENTED BY one AND path WITH the element at position index OF word APPENDED AND count SET TO zero  
                END IF  
                CALL backtrack WITH index INCREMENTED BY one AND path UNCHANGED AND count INCREMENTED BY one  
            END IF  
        END FUNCTION  
        SET result TO an empty collection  
        CALL backtrack WITH zero AS index AND an empty collection AS path AND zero AS count  
        RETURN result  
    END FUNCTION  
END CLASS
