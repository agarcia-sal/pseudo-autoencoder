CLASS Solution  
    FUNCTION eraseOverlapIntervals(intervals)  
        IF intervals EQUALS empty  
            RETURN zero  
        END IF  
        SORT intervals BY end time position of each interval  
        SET end TO end time position of first interval in intervals  
        SET count TO zero  
        FOR index FROM one TO the length of intervals MINUS one  
            IF start time position of interval at position index IN intervals GREATER THAN OR EQUAL TO end  
                SET end TO end time position of interval at position index IN intervals  
            ELSE  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
