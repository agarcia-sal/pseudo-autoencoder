CLASS Solution  
    FUNCTION maxSubArray(nums)  
        SET max_sum TO element at position zero OF nums  
        SET current_sum TO element at position zero OF nums  
        FOR each num IN the elements of nums from position one TO the end  
            IF num GREATER THAN current_sum PLUS num  
                SET current_sum TO num  
            ELSE  
                SET current_sum TO current_sum PLUS num  
            END IF  
            IF current_sum GREATER THAN max_sum  
                SET max_sum TO current_sum  
            END IF  
        END FOR  
        RETURN max_sum  
    END FUNCTION  
END CLASS
