CLASS ListNode  
    FUNCTION __init__(self, val zero, next none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO a new ListNode initialized with element at position zero OF values  
    SET p TO head  
    FOR each val IN the ELEMENTS OF values starting from position one TO the END  
        SET node TO a new ListNode initialized with val  
        SET the next attribute of p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the value attribute of p1 EQUALS the value attribute of p2 AND is_same_list of the next attribute of p1 AND the next attribute of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION mergeKLists(self, lists)  
        SET min_heap TO an empty list  
        FOR each index i AND list lst IN lists with enumeration  
            IF lst DOES NOT EQUAL none  
                PUSH a tuple of the value attribute of lst, i, AND lst INTO min_heap using heap push  
            END IF  
        END FOR  
        SET dummy TO a new ListNode with default initialization  
        SET current TO dummy  
        WHILE the LENGTH OF min_heap IS GREATER THAN zero  
            POP the smallest tuple val, i, node FROM min_heap using heap pop  
            SET the next attribute of current TO node  
            SET current TO the next attribute of current  
            IF the next attribute of node DOES NOT EQUAL none  
                PUSH a tuple of the value attribute of the next attribute of node, i, AND the next attribute of node INTO min_heap using heap push  
            END IF  
        END WHILE  
        RETURN the next attribute of dummy  
    END FUNCTION  
END CLASS
