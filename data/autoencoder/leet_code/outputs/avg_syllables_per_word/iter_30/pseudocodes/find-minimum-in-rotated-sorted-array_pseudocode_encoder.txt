CLASS Solution  
    FUNCTION findMin(nums)  
        SET left TO zero  
        SET right TO the LENGTH OF nums MINUS one  
        IF element at position left OF nums LESS THAN element at position right OF nums  
            RETURN element at position left OF nums  
        END IF  
        WHILE left LESS THAN right  
            SET mid TO the INTEGER DIVISION OF left PLUS right BY two  
            IF element at position mid OF nums GREATER THAN element at position right OF nums  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid  
            END IF  
        END WHILE  
        RETURN element at position left OF nums  
    END FUNCTION  
END CLASS
