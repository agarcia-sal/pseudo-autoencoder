CLASS Solution  
    FUNCTION findMaximumXOR(nums)  
        SET max_xor TO zero  
        FOR bit_position FROM thirty one TO zero DECREMENTING BY one  
            SET max_xor TO max_xor MULTIPLIED BY two  
            SET current_candidate TO max_xor PLUS one  
            SET prefixes TO empty collection  
            FOR each number IN nums  
                SET prefix TO integer division of number BY two RAISED TO THE POWER OF bit_position  
                APPEND prefix TO prefixes  
            END FOR  
            FOR each prefix IN prefixes  
                IF the exclusive OR of current_candidate AND prefix IS IN prefixes  
                    SET max_xor TO current_candidate  
                    BREAK  
                END IF  
            END FOR  
        END FOR  
        RETURN max_xor  
    END FUNCTION  
END CLASS
