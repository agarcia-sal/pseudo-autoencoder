CLASS Solution  
    FUNCTION fractionAddition(expression)  
        SET list_of_fractions TO expression REPLACE each minus WITH plus MINUS SPLIT by plus  
        SET result_fraction TO zero numerator over one denominator  
        FOR each fraction_string IN list_of_fractions  
            IF fraction_string NOT EQUALS empty string  
                INCREMENT result_fraction BY fraction represented by fraction_string  
            END IF  
        END FOR  
        RETURN string representing result_fraction numerator FOLLOWED BY slash FOLLOWED BY result_fraction denominator  
    END FUNCTION  
END CLASS
