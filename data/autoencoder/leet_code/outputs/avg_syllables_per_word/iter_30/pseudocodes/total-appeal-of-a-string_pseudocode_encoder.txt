CLASS Solution  
    FUNCTION appealSum(s)  
        SET last TO list with twenty six elements each set to minus one  
        SET total_appeal TO zero  
        SET current_appeal TO zero  
        FOR each position i and character char IN s  
            SET index TO the numerical position of char MINUS the numerical position of character a  
            INCREMENT current_appeal BY i MINUS element at position index OF last  
            INCREMENT total_appeal BY current_appeal  
            SET element at position index OF last TO i  
        END FOR  
        RETURN total_appeal  
    END FUNCTION  
END CLASS
