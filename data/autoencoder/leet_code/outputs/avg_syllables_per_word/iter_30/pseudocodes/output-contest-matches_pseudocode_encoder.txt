CLASS Solution  
    FUNCTION findContestMatch(n)  
        SET teams TO list of strings from one TO n inclusive  
        WHILE n GREATER THAN one  
            FOR i FROM zero TO n DIVIDED BY two MINUS one  
                SET teams at position i TO concatenation of open parenthesis concatenated with element at position i of teams concatenated with comma concatenated with element at position n MINUS one MINUS i of teams concatenated with close parenthesis  
            END FOR  
            SET n TO n DIVIDED BY two  
        END WHILE  
        RETURN element at position zero of teams  
    END FUNCTION  
END CLASS
