CLASS TreeNode  
    FUNCTION __init__(self, val equals zero, left equals none, right equals none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at position zero OF values  
    SET i TO one  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the front OF queue  
        IF i LESS THAN length of values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with value at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with value at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN True  
    ELSE IF p IS none OR q IS none  
        RETURN False  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION rightSideView(self, root)  
        IF root IS none  
            RETURN empty list  
        END IF  
        SET right_side_view TO empty list  
        SET queue TO new deque containing root  
        WHILE queue IS NOT empty  
            SET level_length TO length of queue  
            FOR i FROM zero TO level_length MINUS one  
                SET node TO the element removed from the front OF queue  
                IF i EQUALS level_length MINUS one  
                    APPEND value of node TO right_side_view  
                END IF  
                IF left child of node IS NOT none  
                    APPEND left child of node TO queue  
                END IF  
                IF right child of node IS NOT none  
                    APPEND right child of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN right_side_view  
    END FUNCTION  
END CLASS
