CLASS Solution  
    FUNCTION isTransformable(s, t)  
        ASSIGN reference TO a new mapping from each key to a new list called pos  
        FOR each index i AND character c IN s  
            CALL function to add i INTO the list at position integer value of c IN pos  
        END FOR  
        FOR each index i AND character c IN t  
            SET digit TO integer value of c  
            IF the list at position digit IN pos IS empty THEN  
                RETURN false  
            END IF  
            FOR each smaller FROM zero TO digit MINUS one  
                IF the list at position smaller IN pos IS not empty AND the first element at position smaller IN pos IS less than the first element at position digit IN pos THEN  
                    RETURN false  
                END IF  
            END FOR  
            CALL function to remove and return the smallest element FROM the list at position digit IN pos  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
