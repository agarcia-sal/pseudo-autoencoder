CLASS Solution  
    FUNCTION areSentencesSimilarTwo(sentence1, sentence2, similarPairs)  
        IF the LENGTH OF sentence1 NOT EQUALS the LENGTH OF sentence2  
            RETURN false  
        END IF  

        SET parent TO an empty dictionary  

        FUNCTION find(x)  
            IF the element at position x OF parent NOT EQUALS x  
                SET the element at position x OF parent TO the result of find with the element at position x OF parent  
            END IF  
            RETURN the element at position x OF parent  
        END FUNCTION  

        FUNCTION union(x, y)  
            SET rootX TO the result of find with x  
            SET rootY TO the result of find with y  
            IF rootX NOT EQUALS rootY  
                SET the element at position rootY OF parent TO rootX  
            END IF  
        END FUNCTION  

        FOR each pair IN similarPairs  
            FOR each word IN pair  
                IF word NOT IN parent  
                    SET the element at position word OF parent TO word  
                END IF  
            END FOR  
        END FOR  

        FOR the pair of values x, y IN similarPairs  
            CALL union with x and y  
        END FOR  

        FOR the pair of values w1, w2 IN the combination of sentence1 and sentence2 BY corresponding positions  
            IF w1 NOT IN parent  
                SET the element at position w1 OF parent TO w1  
            END IF  
            IF w2 NOT IN parent  
                SET the element at position w2 OF parent TO w2  
            END IF  
            IF the result of find with w1 NOT EQUALS the result of find with w2  
                RETURN false  
            END IF  
        END FOR  

        RETURN true  
    END FUNCTION  
END CLASS
