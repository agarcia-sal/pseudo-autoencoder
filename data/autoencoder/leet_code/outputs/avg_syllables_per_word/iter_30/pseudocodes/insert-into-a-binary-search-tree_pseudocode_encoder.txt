CLASS TreeNode  
    FUNCTION __init__ WITH PARAMETERS self, val EQUALS zero, left EQUALS none, right EQUALS none  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node WITH PARAMETERS values  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH PARAMETERS value AT position zero OF values  
    SET i TO one  
    SET queue TO AN EMPTY DEQUE  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND value AT position i OF values IS NOT none  
            SET node.left TO A NEW TreeNode WITH PARAMETERS value AT position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND value AT position i OF values IS NOT none  
            SET node.right TO A NEW TreeNode WITH PARAMETERS value AT position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree WITH PARAMETERS p, q  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value AT p IS NOT EQUALS value AT q  
        RETURN false  
    ELSE  
        RETURN is_same_tree WITH PARAMETERS value AT p.left, value AT q.left AND is_same_tree WITH PARAMETERS value AT p.right, value AT q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION insertIntoBST WITH PARAMETERS self, root, val  
        FUNCTION dfs WITH PARAMETERS root  
            IF root IS none  
                RETURN A NEW TreeNode WITH PARAMETERS val  
            END IF  
            IF value AT root IS LESS THAN val  
                SET root.right TO dfs WITH PARAMETERS root.right  
            ELSE  
                SET root.left TO dfs WITH PARAMETERS root.left  
            END IF  
            RETURN root  
        END FUNCTION  
        RETURN dfs WITH PARAMETERS root  
    END FUNCTION  
END CLASS
