CLASS Solution  
    FUNCTION canIWin(maxChoosableInteger, desiredTotal)  
        IF the sum of all integers from one to maxChoosableInteger is LESS THAN desiredTotal  
            RETURN false  
        END IF  
        IF desiredTotal is LESS THAN OR EQUAL TO zero  
            RETURN true  
        END IF  
        SET memo TO empty dictionary  
        FUNCTION can_win(current_total, used_numbers)  
            IF the tuple of current_total and used_numbers is IN memo  
                RETURN the value FROM memo for the tuple of current_total and used_numbers  
            END IF  
            FOR number FROM one TO maxChoosableInteger  
                IF the bitwise AND of used_numbers and two RAISED TO THE POWER OF number is EQUAL TO zero  
                    IF current_total PLUS number is GREATER THAN OR EQUAL TO desiredTotal OR the NEGATION of can_win with arguments current_total PLUS number and the bitwise OR of used_numbers and two RAISED TO THE POWER OF number  
                        SET memo at the tuple of current_total and used_numbers TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            SET memo at the tuple of current_total and used_numbers TO false  
            RETURN false  
        END FUNCTION  
        RETURN can_win with zero and zero as arguments  
    END FUNCTION  
END CLASS
