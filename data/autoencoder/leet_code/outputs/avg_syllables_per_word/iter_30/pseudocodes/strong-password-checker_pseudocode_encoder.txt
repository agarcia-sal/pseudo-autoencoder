CLASS Solution  
    FUNCTION strongPasswordChecker(password)  
        SET n TO the LENGTH OF password  
        
        SET has_lower TO false  
        SET has_upper TO false  
        SET has_digit TO false  
        
        FOR each character IN password  
            IF character IS a lowercase letter  
                SET has_lower TO true  
            END IF  
            IF character IS an uppercase letter  
                SET has_upper TO true  
            END IF  
            IF character IS a digit  
                SET has_digit TO true  
            END IF  
        END FOR  
        
        SET sum_types TO zero  
        IF has_lower IS true  
            INCREMENT sum_types BY one  
        END IF  
        IF has_upper IS true  
            INCREMENT sum_types BY one  
        END IF  
        IF has_digit IS true  
            INCREMENT sum_types BY one  
        END IF  
        
        SET missing_types TO three MINUS sum_types  
        
        SET replace TO zero  
        SET one TO zero  
        SET two TO zero  
        SET i TO two  
        
        WHILE i IS LESS THAN n  
            IF character at position i OF password EQUALS character at position i MINUS one OF password AND character at position i OF password EQUALS character at position i MINUS two OF password  
                SET length TO two  
                WHILE i IS LESS THAN n AND character at position i OF password EQUALS character at position i MINUS one OF password  
                    INCREMENT length BY one  
                    INCREMENT i BY one  
                END WHILE  
                INCREMENT replace BY length DIVIDED BY three  
                IF length MODULO three EQUALS zero  
                    INCREMENT one BY one  
                ELSE IF length MODULO three EQUALS one  
                    INCREMENT two BY one  
                END IF  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        
        IF n IS LESS THAN six  
            IF missing_types IS GREATER THAN six MINUS n  
                RETURN missing_types  
            ELSE  
                RETURN six MINUS n  
            END IF  
        ELSE IF n IS LESS THAN OR EQUAL TO twenty  
            IF missing_types IS GREATER THAN replace  
                RETURN missing_types  
            ELSE  
                RETURN replace  
            END IF  
        ELSE  
            SET delete TO n MINUS twenty  
            
            SET decrease_replace_by_one TO the MINIMUM OF delete AND one  
            DECREMENT replace BY decrease_replace_by_one  
            
            SET decrease_replace_by_two TO the MINIMUM OF the MAXIMUM OF delete MINUS one AND zero AND two MULTIPLIED BY two  
            DECREMENT replace BY decrease_replace_by_two DIVIDED BY two  
            
            SET decrease_replace_by_three TO the MAXIMUM OF delete MINUS one MINUS two MULTIPLIED BY two AND zero  
            DECREMENT replace BY decrease_replace_by_three DIVIDED BY three  
            
            IF delete PLUS missing_types IS GREATER THAN replace  
                RETURN delete PLUS missing_types  
            ELSE  
                RETURN delete PLUS replace  
            END IF  
        END IF  
    END FUNCTION  
END CLASS
