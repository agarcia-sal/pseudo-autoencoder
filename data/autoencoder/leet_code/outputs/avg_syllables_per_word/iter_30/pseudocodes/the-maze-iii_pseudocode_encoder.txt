CLASS Solution  
    FUNCTION findShortestWay(maze, ball, hole)  
        SET directions TO list of four tuples each containing zero or one or minus one and one-character direction name as right down left up  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero OF maze  
        SET start TO tuple constructed from ball  
        SET hole TO tuple constructed from hole  
        SET pq TO list containing a tuple of zero distance empty string path and start position  
        SET visited TO empty set  
        
        WHILE pq is not empty  
            REMOVE and return the smallest element from pq and ASSIGN to dist path and tuple containing x and y  
            
            IF tuple containing x and y EQUALS hole  
                RETURN path  
            END IF  
            
            IF tuple containing x and y is IN visited  
                CONTINUE  
            END IF  
            
            ADD tuple containing x and y TO visited  
            
            FOR each tuple dx dy direction IN directions  
                SET new_x TO x  
                SET new_y TO y  
                SET count TO zero  
                
                WHILE new_x plus dx GREATER THAN OR EQUAL TO zero AND new_x plus dx LESS THAN m AND new_y plus dy GREATER THAN OR EQUAL TO zero AND new_y plus dy LESS THAN n AND element at position new_x plus dx OF maze AND element at position new_y plus dy OF that element EQUALS zero  
                    INCREMENT new_x BY dx  
                    INCREMENT new_y BY dy  
                    INCREMENT count BY one  
                    IF tuple containing new_x and new_y EQUALS hole  
                        BREAK  
                    END IF  
                END WHILE  
                
                IF tuple containing new_x and new_y NOT IN visited  
                    PUSH tuple containing dist plus count path concatenated with direction tuple containing new_x and new_y INTO pq maintaining heap property  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN string impossible  
    END FUNCTION  
END CLASS
