CLASS Solution  
    FUNCTION replaceNonCoprimes(nums)  
        SET stack TO empty list  
        FOR each num IN nums  
            WHILE the LENGTH OF stack GREATER THAN zero  
                SET g TO the greatest common divisor of element at position last OF stack AND num  
                IF g EQUALS one  
                    BREAK  
                END IF  
                SET num TO element at position last OF stack MULTIPLIED BY num DIVIDED BY g  
                REMOVE element at position last FROM stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN stack  
    END FUNCTION  
END CLASS
