CLASS Solution  
    FUNCTION maxCoins(nums)  
        SET nums TO one concatenated WITH nums concatenated WITH one  
        SET n TO the LENGTH OF nums  
        SET dp TO two dimensional list of zeros with dimensions n BY n  
        FOR length FROM two TO n MINUS one  
            FOR i FROM zero TO n MINUS length MINUS one  
                SET j TO i PLUS length  
                FOR k FROM i PLUS one TO j MINUS one  
                    SET coins TO value at position i OF nums MULTIPLIED BY value at position k OF nums MULTIPLIED BY value at position j OF nums PLUS value at position i AND position k OF dp PLUS value at position k AND position j OF dp  
                    IF value at position i AND position j OF dp LESS THAN coins  
                        SET value at position i AND position j OF dp TO coins  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN value at position zero AND position n MINUS one OF dp  
    END FUNCTION  
END CLASS
