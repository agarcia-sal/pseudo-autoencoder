CLASS Solution  
    FUNCTION permuteUnique(nums)  
        FUNCTION backtrack(path, used, res)  
            IF the LENGTH OF path EQUALS the LENGTH OF nums  
                APPEND a COPY of path TO res  
                RETURN  
            END IF  
            FOR index FROM zero TO the LENGTH OF nums MINUS one  
                IF the ELEMENT at position index OF used IS true OR (index GREATER THAN zero AND the ELEMENT at position index OF nums EQUALS the ELEMENT at position index MINUS one OF nums AND the ELEMENT at position index MINUS one OF used IS false)  
                    CONTINUE  
                END IF  
                SET the ELEMENT at position index OF used TO true  
                APPEND the ELEMENT at position index OF nums TO path  
                CALL backtrack WITH path, used, res  
                REMOVE the LAST ELEMENT FROM path  
                SET the ELEMENT at position index OF used TO false  
            END FOR  
        END FUNCTION  
        CALL the SORT METHOD ON nums  
        SET res TO an empty list  
        SET used TO a list of false VALUES with LENGTH equal to the LENGTH OF nums  
        CALL backtrack WITH an empty list, used, res  
        RETURN res  
    END FUNCTION  
END CLASS
