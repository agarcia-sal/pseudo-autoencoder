CLASS Solution  
    FUNCTION canCross(stones)  
        IF stones IS empty OR the LENGTH OF stones IS less than two  
            RETURN false  
        END IF  
        SET stone_set TO the set converted from stones  
        SET memo TO an empty dictionary  
        FUNCTION can_jump_to(position, jump)  
            IF position EQUALS element at position the LENGTH OF stones MINUS one OF stones  
                RETURN true  
            END IF  
            IF the tuple of position and jump IS in memo  
                RETURN the value in memo at the tuple of position and jump  
            END IF  
            FOR each next_jump FROM jump MINUS one TO jump PLUS one  
                IF next_jump IS greater than zero AND position PLUS next_jump IS in stone_set  
                    IF can_jump_to(position PLUS next_jump, next_jump)  
                        SET memo at the tuple of position and jump TO true  
                        RETURN true  
                    END IF  
                END IF  
            END FOR  
            SET memo at the tuple of position and jump TO false  
            RETURN false  
        END FUNCTION  
        RETURN can_jump_to(element at position zero OF stones, zero)  
    END FUNCTION  
END CLASS
