CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, left WITH DEFAULT none, right WITH DEFAULT none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at position zero OF values  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element popped from the left OF queue  
        IF i LESS THAN length of values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with value at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with value at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF value at p NOT EQUALS value at q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrderBottom(self, root)  
        IF root EQUALS none  
            RETURN empty list  
        END IF  
        SET result TO empty list  
        SET queue TO double ended queue initialized with root  
        WHILE queue IS NOT empty  
            SET level_size TO length of queue  
            SET current_level TO empty list  
            FOR each index FROM zero TO level_size MINUS one  
                SET node TO element popped from the left OF queue  
                APPEND value at node TO current_level  
                IF left child of node IS NOT none  
                    APPEND left child of node TO queue  
                END IF  
                IF right child of node IS NOT none  
                    APPEND right child of node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result in reverse order  
    END FUNCTION  
END CLASS
