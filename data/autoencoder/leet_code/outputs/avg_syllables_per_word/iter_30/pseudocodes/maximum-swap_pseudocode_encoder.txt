CLASS Solution  
    FUNCTION maximumSwap(num)  
        SET digits TO list of string form of num  
        SET last TO empty dictionary  
        FOR each index i AND element d IN digits  
            SET digit_value TO integer form of d  
            SET last at digit_value TO i  
        END FOR  
        FOR each index i AND element d IN digits  
            SET digit_value TO integer form of d  
            FOR each d2 FROM nine DOWN TO digit_value PLUS one  
                IF last at d2 EXISTS AND last at d2 GREATER THAN i THEN  
                    SET temp TO element at position i OF digits  
                    SET element at position i OF digits TO element at position last at d2 OF digits  
                    SET element at position last at d2 OF digits TO temp  
                    SET result_number TO integer form of concatenation of all elements in digits  
                    RETURN result_number  
                END IF  
            END FOR  
        END FOR  
        RETURN num  
    END FUNCTION  
END CLASS
