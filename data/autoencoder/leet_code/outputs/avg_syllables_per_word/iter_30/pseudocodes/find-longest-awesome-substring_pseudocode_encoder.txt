CLASS Solution  
    FUNCTION longestAwesome(s)  
        SET prefix_map TO a new mapping with key zero assigned to minus one  
        SET prefix_xor TO zero  
        SET max_length TO zero  
        
        FOR each index i AND character char IN s  
            SET digit TO the integer value of char  
            SET prefix_xor TO prefix_xor XOR with one shifted left by digit positions  
            
            IF prefix_xor is a key in prefix_map  
                SET max_length TO the greater of max_length AND index i MINUS the value at key prefix_xor in prefix_map  
            END IF  
            
            FOR each number j FROM zero TO nine  
                SET flipped TO prefix_xor XOR with one shifted left by j positions  
                IF flipped is a key in prefix_map  
                    SET max_length TO the greater of max_length AND index i MINUS the value at key flipped in prefix_map  
                END IF  
            END FOR  
            
            IF prefix_xor is NOT a key in prefix_map  
                ASSIGN index i TO key prefix_xor in prefix_map  
            END IF  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
