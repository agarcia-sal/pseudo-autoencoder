CLASS Solution  
    FUNCTION minCost(maxTime, edges, passingFees)  
        SET n TO the LENGTH OF passingFees  
        SET graph TO a list of empty lists with n elements  
        FOR each x, y, time IN edges  
            APPEND a tuple of neighbor y and travel time time TO the element at position x OF graph  
            APPEND a tuple of neighbor x and travel time time TO the element at position y OF graph  
        END FOR  
          
        SET pq TO a list containing a tuple of passingFees at position zero, zero, zero  
        SET visited_time TO a list of infinite values with length n  
        SET the element at position zero OF visited_time TO zero  
          
        WHILE pq is not empty  
            SET current_cost, current_city, current_time TO the element popped from pq with the smallest first element  
              
            IF current_city EQUALS n MINUS one  
                RETURN current_cost  
            END IF  
              
            FOR each neighbor, travel_time IN the element at position current_city OF graph  
                SET new_time TO current_time PLUS travel_time  
                SET new_cost TO current_cost PLUS passingFees at position neighbor  
                  
                IF new_time LESS THAN OR EQUAL TO maxTime AND (new_time LESS THAN the element at position neighbor OF visited_time OR new_cost LESS THAN current_cost)  
                    SET the element at position neighbor OF visited_time TO new_time  
                    ADD the tuple of new_cost, neighbor, new_time TO pq preserving min-heap property  
                END IF  
            END FOR  
        END WHILE  
          
        RETURN negative one  
    END FUNCTION  
END CLASS
