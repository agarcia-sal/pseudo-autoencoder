CLASS Solution  
    FUNCTION isScramble(s1, s2)  
        IF s1 EQUALS s2  
            RETURN true  
        END IF  
        IF the SORTED characters OF s1 NOT EQUALS the SORTED characters OF s2  
            RETURN false  
        END IF  
        SET n TO the LENGTH OF s1  
        FOR i FROM one TO n MINUS one  
            IF self dot isScramble with substring from position zero TO position i MINUS one OF s1 AND substring from position zero TO position i MINUS one OF s2 EQUALS true AND self dot isScramble with substring from position i TO the END OF s1 AND substring from position i TO the END OF s2 EQUALS true  
                RETURN true  
            END IF  
            IF self dot isScramble with substring from position zero TO position i MINUS one OF s1 AND substring from position n MINUS i TO the END OF s2 EQUALS true AND self dot isScramble with substring from position i TO the END OF s1 AND substring from position zero TO position n MINUS i MINUS one OF s2 EQUALS true  
                RETURN true  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
