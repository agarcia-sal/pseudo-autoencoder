CLASS Solution  
    FUNCTION countTriplets(nums)  
        ASSIGN reference TO an empty dictionary with default integer values called pair_and_count  
        FOR each element i IN nums  
            FOR each element j IN nums  
                SET bitwise AND of i AND j TO and_result  
                INCREMENT value at key and_result IN pair_and_count BY one  
            END FOR  
        END FOR  
        SET triplet_count TO zero  
        FOR each element k IN nums  
            FOR each key-value pair pair_and and count IN pair_and_count  
                SET bitwise AND of pair_and AND k TO and_with_k  
                IF and_with_k EQUALS zero  
                    INCREMENT triplet_count BY count  
                END IF  
            END FOR  
        END FOR  
        RETURN triplet_count  
    END FUNCTION  
END CLASS
