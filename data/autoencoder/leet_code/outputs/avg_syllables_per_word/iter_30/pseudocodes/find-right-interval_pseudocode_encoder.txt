CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO list of pairs of start position in interval and index in intervals sorted by start position  
        SET result TO empty list  
        FOR each interval in intervals with start ignored and end as second element  
            SET index TO position found by binary searching in start_times for pair with start position greater than or equal to end  
            IF index is less than the length of start_times  
                APPEND element at position one of pair at position index of start_times TO result  
            ELSE  
                APPEND negative one TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
