CLASS Solution  
    FUNCTION removeDuplicateLetters WITH PARAMETER s  
        SET last_occurrence TO empty dictionary  
        FOR each index FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position index OF s  
            ASSIGN last_occurrence at key character TO index  
        END FOR  
        
        SET stack TO empty list  
        SET in_stack TO empty set  
        
        FOR each index FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position index OF s  
            
            IF character IS IN in_stack  
                CONTINUE  
            END IF  
            
            WHILE the LENGTH OF stack IS greater than zero AND character IS LESS THAN the element at the last position OF stack AND index IS LESS THAN the value at key element at the last position OF stack IN last_occurrence  
                SET removed_char TO the element popped FROM stack  
                REMOVE removed_char FROM in_stack  
            END WHILE  
            
            APPEND character TO stack  
            ADD character TO in_stack  
        END FOR  
        
        SET result_string TO concatenation of elements IN stack  
        RETURN result_string  
    END FUNCTION  
END CLASS
