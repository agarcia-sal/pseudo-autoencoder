CLASS Solution  
    FUNCTION verifyPreorder(preorder)  
        SET stack TO empty list  
        SET lower_bound TO negative infinity  
        FOR each value IN preorder  
            IF value LESS THAN lower_bound  
                RETURN false  
            END IF  
            WHILE stack IS NOT empty AND value GREATER THAN element at last position OF stack  
                SET lower_bound TO element removed from last position OF stack  
            END WHILE  
            APPEND value TO stack  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
