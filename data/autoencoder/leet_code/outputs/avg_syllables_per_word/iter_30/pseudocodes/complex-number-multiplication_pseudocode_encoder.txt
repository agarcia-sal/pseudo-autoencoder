CLASS Solution  
    FUNCTION complexNumberMultiply(num1, num2)  
        FUNCTION parse_complex(num)  
            SET parts TO list of two elements from splitting num by the character plus  
            SET real TO integer value of element at position zero of parts  
            SET imaginary_string TO element at position one of parts  
            SET imaginary TO integer value of substring from position zero TO position length of imaginary_string MINUS one of imaginary_string  
            RETURN real and imaginary  
        END FUNCTION  
        
        SET real1 and imag1 TO result of calling parse_complex with argument num1  
        SET real2 and imag2 TO result of calling parse_complex with argument num2  
        
        SET real_part TO real1 MULTIPLIED BY real2 MINUS imag1 MULTIPLIED BY imag2  
        SET imaginary_part TO real1 MULTIPLIED BY imag2 PLUS imag1 MULTIPLIED BY real2  
        
        RETURN string formed by concatenating real_part, the character plus, imaginary_part, and the character i  
    END FUNCTION  
END CLASS
