CLASS Solution  
    FUNCTION predictPartyVictory(senate)  
        ASSIGN reference TO a new empty double ended queue named radiant  
        ASSIGN reference TO a new empty double ended queue named dire  
        FOR each index i and element s IN senate  
            IF s EQUALS uppercase letter R  
                APPEND i TO radiant  
            ELSE  
                APPEND i TO dire  
            END IF  
        END FOR  
        WHILE the LENGTH OF radiant IS GREATER THAN zero AND the LENGTH OF dire IS GREATER THAN zero  
            SET r_index TO the element removed from the front of radiant  
            SET d_index TO the element removed from the front of dire  
            IF r_index LESS THAN d_index  
                APPEND r_index PLUS the LENGTH OF senate TO radiant  
            ELSE  
                APPEND d_index PLUS the LENGTH OF senate TO dire  
            END IF  
        END WHILE  
        IF the LENGTH OF radiant IS GREATER THAN zero  
            RETURN the word Radiant  
        ELSE  
            RETURN the word Dire  
        END IF  
    END FUNCTION  
END CLASS
