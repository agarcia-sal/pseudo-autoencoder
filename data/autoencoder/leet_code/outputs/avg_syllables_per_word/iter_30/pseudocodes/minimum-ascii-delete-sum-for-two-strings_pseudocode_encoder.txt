CLASS Solution  
    FUNCTION minimumDeleteSum(s1, s2)  
        SET m TO the LENGTH OF s1  
        SET n TO the LENGTH OF s2  
        SET dp TO a two dimensional ARRAY OF zero with m PLUS one ROWS and n PLUS one COLUMNS  
        FOR j FROM one TO n  
            SET dp at row zero column j TO dp at row zero column j MINUS one PLUS ASCII VALUE OF character at position j MINUS one OF s2  
        END FOR  
        FOR i FROM one TO m  
            SET dp at row i column zero TO dp at row i MINUS one column zero PLUS ASCII VALUE OF character at position i MINUS one OF s1  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF character at position i MINUS one OF s1 EQUALS character at position j MINUS one OF s2  
                    SET dp at row i column j TO dp at row i MINUS one column j MINUS one  
                ELSE  
                    SET dp at row i column j TO the MINIMUM OF dp at row i MINUS one column j PLUS ASCII VALUE OF character at position i MINUS one OF s1 AND dp at row i column j MINUS one PLUS ASCII VALUE OF character at position j MINUS one OF s2  
                END IF  
            END FOR  
        END FOR  
        RETURN dp at row m column n  
    END FUNCTION  
END CLASS
