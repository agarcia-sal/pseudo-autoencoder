CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value as element at position zero OF values  
    SET i TO one  
    SET queue TO new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element removed FROM queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with value as element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with value as element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p IS NOT EQUALS value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION boundaryOfBinaryTree(self, root)  
        IF root IS none  
            RETURN empty list  
        END IF  

        FUNCTION is_leaf(node)  
            RETURN node.left IS none AND node.right IS none  
        END FUNCTION  

        SET left_boundary TO empty list  
        SET current TO left child of root  
        WHILE current IS NOT none  
            IF NOT is_leaf(current)  
                APPEND value of current TO left_boundary  
            END IF  
            IF current.left IS NOT none  
                SET current TO current.left  
            ELSE  
                SET current TO current.right  
            END IF  
        END WHILE  

        SET right_boundary TO empty list  
        SET current TO right child of root  
        WHILE current IS NOT none  
            IF NOT is_leaf(current)  
                APPEND value of current TO right_boundary  
            END IF  
            IF current.right IS NOT none  
                SET current TO current.right  
            ELSE  
                SET current TO current.left  
            END IF  
        END WHILE  

        SET leaves TO empty list  
        FUNCTION collect_leaves(node)  
            IF node IS none  
                RETURN  
            END IF  
            IF is_leaf(node)  
                APPEND value of node TO leaves  
            END IF  
            CALL collect_leaves(left child of node)  
            CALL collect_leaves(right child of node)  
        END FUNCTION  

        CALL collect_leaves(root)  

        IF is_leaf(root)  
            SET leaves TO empty list  
        END IF  

        RETURN list containing value of root FOLLOWED BY all elements of left_boundary FOLLOWED BY all elements of leaves FOLLOWED BY all elements of right_boundary in REVERSE order  
    END FUNCTION  
END CLASS
