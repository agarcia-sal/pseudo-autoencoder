CLASS Solution  
    FUNCTION minDistance(houses, k)  
        CALL sort ON houses  
        SET n TO the LENGTH OF houses  
        SET cost TO a two dimensional collection of zeroes with dimensions n by n  
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i PLUS one TO n MINUS one  
                SET cost element at position i at position j TO the value of cost element at position i PLUS one at position j MINUS one PLUS the value of element at position j OF houses MINUS the value of element at position i OF houses  
            END FOR  
        END FOR  
        SET dp TO a two dimensional collection with dimensions n PLUS one by k PLUS one filled with infinity  
        SET the element at position zero at position zero OF dp TO zero  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                FOR p FROM zero TO i MINUS one  
                    SET dp element at position i at position j TO the minimum of its current value AND the sum of dp element at position p at position j MINUS one AND cost element at position p at position i MINUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN the element at position n at position k OF dp  
    END FUNCTION  
END CLASS
