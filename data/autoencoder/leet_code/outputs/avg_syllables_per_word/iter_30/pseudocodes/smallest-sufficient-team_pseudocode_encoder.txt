CLASS Solution  
    FUNCTION smallestSufficientTeam(req_skills, people)  
        CREATE empty mapping d FROM skill TO index  
        SET m TO the LENGTH OF req_skills  
        SET n TO the LENGTH OF people  
        CREATE list p WITH n elements EACH SET TO zero  
        FOR index i FROM zero TO n MINUS one  
            FOR each skill s IN element at position i OF people  
                SET element at position i OF p TO the bitwise OR OF itself AND one SHIFTED LEFT BY value of d at key s  
            END FOR  
        END FOR  
        CREATE list f WITH the number of elements equal TO two RAISED TO THE POWER OF m EACH SET TO infinite  
        CREATE list g WITH the number of elements equal TO two RAISED TO THE POWER OF m EACH SET TO zero  
        CREATE list h WITH the number of elements equal TO two RAISED TO THE POWER OF m EACH SET TO zero  
        SET element at position zero OF f TO zero  
        FOR integer i FROM zero TO two RAISED TO THE POWER OF m MINUS one  
            IF element at position i OF f EQUALS infinite  
                CONTINUE  
            END IF  
            FOR integer j FROM zero TO n MINUS one  
                IF element at position i OF f PLUS one LESS THAN element at position bitwise OR OF i AND element at position j OF p OF f  
                    SET element at position bitwise OR OF i AND element at position j OF p OF f TO element at position i OF f PLUS one  
                    SET element at position bitwise OR OF i AND element at position j OF p OF g TO j  
                    SET element at position bitwise OR OF i AND element at position j OF p OF h TO i  
                END IF  
            END FOR  
        END FOR  
        SET i TO two RAISED TO THE POWER OF m MINUS one  
        CREATE empty list ans  
        WHILE i NOT EQUAL TO zero  
            APPEND element at position i OF g TO ans  
            SET i TO element at position i OF h  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
