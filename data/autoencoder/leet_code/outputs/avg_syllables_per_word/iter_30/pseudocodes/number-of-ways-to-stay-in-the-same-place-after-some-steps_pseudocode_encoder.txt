CLASS Solution  
    FUNCTION numWays(steps, arrLen)  
        SET modulus TO one billion PLUS seven  
        SET maximum index TO the minimum of steps DIVIDED BY two AND arrLen MINUS one  
        CREATE list dp OF zeros WITH length maximum index PLUS one  
        SET element at position zero OF dp TO one  
        FOR each iteration FROM one TO steps  
            ASSIGN reference prev_dp TO a copy of dp  
            FOR each index i FROM zero TO maximum index  
                SET element at position i OF dp TO element at position i OF prev_dp  
                IF i GREATER THAN zero  
                    SET element at position i OF dp TO element at position i OF dp PLUS element at position i MINUS one OF prev_dp  
                    SET element at position i OF dp TO element at position i OF dp MODULO modulus  
                END IF  
                IF i LESS THAN maximum index  
                    SET element at position i OF dp TO element at position i OF dp PLUS element at position i PLUS one OF prev_dp  
                    SET element at position i OF dp TO element at position i OF dp MODULO modulus  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position zero OF dp  
    END FUNCTION  
END CLASS
