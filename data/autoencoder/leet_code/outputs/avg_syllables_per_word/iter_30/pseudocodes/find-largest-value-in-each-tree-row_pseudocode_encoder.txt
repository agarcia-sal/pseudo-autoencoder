CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO a new TreeNode with the element at position zero OF values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO the element popped from the left OF queue  
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values NOT EQUALS none  
            SET node.left TO a new TreeNode with the element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND the element at position i OF values NOT EQUALS none  
            SET node.right TO a new TreeNode with the element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION largestValues(self, root)  
        IF root EQUALS none  
            RETURN an empty list  
        END IF  
        SET queue TO a double ended queue containing root  
        SET largest_values TO an empty list  
        WHILE the LENGTH OF queue GREATER THAN zero  
            SET level_size TO the LENGTH OF queue  
            SET max_value TO negative infinity  
            FOR each index FROM zero TO level_size MINUS one  
                SET node TO the element popped from the left OF queue  
                IF max_value LESS THAN node.val  
                    SET max_value TO node.val  
                END IF  
                IF node.left NOT EQUALS none  
                    APPEND node.left TO queue  
                END IF  
                IF node.right NOT EQUALS none  
                    APPEND node.right TO queue  
                END IF  
            END FOR  
            APPEND max_value TO largest_values  
        END WHILE  
        RETURN largest_values  
    END FUNCTION  
END CLASS
