CLASS Solution  
    FUNCTION minSkips(dist, speed, hoursBefore)  
        SET n TO the LENGTH OF dist  
        INITIALIZE dp AS a list of lists where each inner list contains n PLUS one elements each SET TO infinity, and there are n such inner lists  
        SET total_dist TO the SUM OF all elements IN dist  
        IF total_dist DIVIDED BY speed GREATER THAN hoursBefore  
            RETURN negative one  
        END IF  
        SET element at position zero AT position zero OF dp TO element at position zero OF dist  
        FOR j FROM one TO n MINUS one  
            SET element at position zero AT position j OF dp TO element at position zero OF dist  
        END FOR  
        FOR i FROM one TO n MINUS one  
            FOR j FROM zero TO i  
                IF j LESS THAN i  
                    SET element at position i AT position j OF dp TO the CEILING OF (element at position i MINUS one AT position j OF dp DIVIDED BY speed) MULTIPLIED BY speed PLUS element at position i OF dist  
                END IF  
                IF j GREATER THAN zero  
                    IF the minimum OF element at position i AT position j OF dp AND element at position i MINUS one AT position j MINUS one OF dp PLUS element at position i OF dist EQUALS element at position i MINUS one AT position j MINUS one OF dp PLUS element at position i OF dist  
                        SET element at position i AT position j OF dp TO element at position i MINUS one AT position j MINUS one OF dp PLUS element at position i OF dist  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        FOR j FROM zero TO n MINUS one  
            IF element at position n MINUS one AT position j OF dp LESS THAN OR EQUAL TO hoursBefore MULTIPLIED BY speed  
                RETURN j  
            END IF  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
