CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val EQUALS element at position zero OF values  
    SET i TO one  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element removed from the left OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with val EQUALS element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with val EQUALS element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION widthOfBinaryTree(root)  
        IF root IS none  
            RETURN zero  
        END IF  
        SET queue TO a new deque containing a tuple of root and zero  
        SET max_width TO zero  
        WHILE queue IS NOT empty  
            SET level_length TO the LENGTH OF queue  
            SET first_index TO the second element of the first tuple IN queue  
            FOR count FROM zero TO level_length MINUS one  
                REMOVE the leftmost tuple FROM queue AND ASSIGN node AND index TO the elements OF that tuple  
                IF count IS EQUALS level_length MINUS one  
                    IF index MINUS first_index PLUS one IS GREATER THAN max_width  
                        SET max_width TO index MINUS first_index PLUS one  
                    END IF  
                END IF  
                IF node.left IS NOT none  
                    APPEND a tuple of node.left and index MULTIPLIED BY two TO queue  
                END IF  
                IF node.right IS NOT none  
                    APPEND a tuple of node.right and index MULTIPLIED BY two PLUS one TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN max_width  
    END FUNCTION  
END CLASS
