CLASS Solution  
    FUNCTION largestMultipleOfThree(digits)  
        SET count TO the FREQUENCY COUNTER of digits  
        SET total_sum TO the SUM of digits  
        SET remainder TO the REMAINDER of total_sum DIVIDED BY three  
        IF remainder EQUALS one  
            FOR each d IN the LIST one FOUR seven  
                IF count at d GREATER THAN zero  
                    DECREMENT count at d BY one  
                    BREAK  
                END IF  
            END FOR  
            ELSE  
                FOR each d IN the LIST two FIVE eight  
                    IF count at d GREATER THAN one  
                        DECREMENT count at d BY two  
                        BREAK  
                    END IF  
                END FOR  
        END IF  
        IF remainder EQUALS two  
            FOR each d IN the LIST two FIVE eight  
                IF count at d GREATER THAN zero  
                    DECREMENT count at d BY one  
                    BREAK  
                END IF  
            END FOR  
            ELSE  
                FOR each d IN the LIST one FOUR seven  
                    IF count at d GREATER THAN one  
                        DECREMENT count at d BY two  
                        BREAK  
                    END IF  
                END FOR  
        END IF  
        SET result TO an EMPTY LIST  
        FOR d FROM nine TO zero DECREMENT BY one  
            APPEND the STRING FORMATION OF d REPEATED count at d TIMES TO result  
        END FOR  
        SET final_number TO the CONCATENATION OF all ELEMENTS OF result  
        IF final_number IS NOT EMPTY AND the ELEMENT at POSITION zero OF final_number EQUALS zero  
            RETURN the STRING zero  
        ELSE  
            RETURN final_number  
        END IF  
    END FUNCTION  
END CLASS
