CLASS Solution  
    FUNCTION minimumDifference(nums)  
        SET n TO the LENGTH OF nums DIVIDED BY two  
        SET total_sum TO the SUM OF all elements in nums  
        SET target TO total_sum DIVIDED BY two  
        SET min_diff TO infinity  
        
        SET left_half TO the FIRST n elements OF nums  
        SET right_half TO the elements FROM position n TO the END OF nums  
        
        SET left_sums TO a list of empty dictionaries with LENGTH n PLUS one  
        SET right_sums TO a list of empty dictionaries with LENGTH n PLUS one  
        
        FOR i FROM zero TO n  
            FOR each combo IN the combinations of elements in left_half TAKEN i at a time  
                SET subset_sum TO the SUM OF elements in combo  
                IF subset_sum NOT IN left_sums at position i  
                    SET the value at subset_sum key in left_sums at position i TO zero  
                INCREMENT the value at subset_sum key in left_sums at position i BY one  
            END FOR  
            
            FOR each combo IN the combinations of elements in right_half TAKEN i at a time  
                SET subset_sum TO the SUM OF elements in combo  
                IF subset_sum NOT IN right_sums at position i  
                    SET the value at subset_sum key in right_sums at position i TO zero  
                INCREMENT the value at subset_sum key in right_sums at position i BY one  
            END FOR  
        END FOR  
        
        FOR i FROM zero TO n  
            SET left_values TO the list of keys in left_sums at position i sorted in ascending order  
            SET right_values TO the list of keys in right_sums at position n MINUS i sorted in ascending order  
            
            SET j TO zero  
            SET k TO the LENGTH OF right_values MINUS one  
            WHILE j LESS THAN the LENGTH OF left_values AND k GREATER THAN OR EQUAL TO zero  
                SET current_sum TO the sum of element at position j OF left_values PLUS element at position k OF right_values  
                SET min_diff TO the smaller value BETWEEN min_diff AND the absolute value of total_sum MINUS two MULTIPLIED BY current_sum  
                
                IF current_sum LESS THAN target  
                    INCREMENT j BY one  
                ELSE IF current_sum GREATER THAN target  
                    DECREMENT k BY one  
                ELSE  
                    RETURN zero  
                END IF  
            END WHILE  
        END FOR  
        
        RETURN min_diff  
    END FUNCTION  
END CLASS
