CLASS Solution  
    FUNCTION numDecodings(s)  
        SET MOD TO ten RAISED TO THE POWER OF nine PLUS seven  
        SET dp TO a list of zeros with the LENGTH OF s PLUS one  
        SET element at position zero OF dp TO one  
        
        FUNCTION ways_to_decode_single(char)  
            IF char EQUALS star  
                RETURN nine  
            ELSE IF char EQUALS zero  
                RETURN zero  
            ELSE  
                RETURN one  
            END IF  
        END FUNCTION  
        
        FUNCTION ways_to_decode_pair(char1, char2)  
            IF char1 EQUALS star AND char2 EQUALS star  
                RETURN fifteen  
            ELSE IF char1 EQUALS star  
                IF char2 LESS THAN OR EQUAL TO six  
                    RETURN two  
                ELSE  
                    RETURN one  
                END IF  
            ELSE IF char2 EQUALS star  
                IF char1 EQUALS one  
                    RETURN nine  
                ELSE IF char1 EQUALS two  
                    RETURN six  
                ELSE  
                    RETURN zero  
                END IF  
            ELSE  
                SET num TO the integer VALUE OF char1 CONCATENATED WITH char2  
                IF num GREATER THAN OR EQUAL TO ten AND num LESS THAN OR EQUAL TO twenty six  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
        END FUNCTION  
        
        FOR i FROM one TO the LENGTH OF s  
            SET element at position i OF dp TO element at position i OF dp PLUS element at position i MINUS one OF dp MULTIPLIED BY ways_to_decode_single(element at position i MINUS one OF s)  
            SET element at position i OF dp TO element at position i OF dp MODULO MOD  
            
            IF i GREATER THAN one  
                SET element at position i OF dp TO element at position i OF dp PLUS element at position i MINUS two OF dp MULTIPLIED BY ways_to_decode_pair(element at position i MINUS two OF s, element at position i MINUS one OF s)  
                SET element at position i OF dp TO element at position i OF dp MODULO MOD  
            END IF  
        END FOR  
        
        RETURN element at the last position OF dp  
    END FUNCTION  
END CLASS
