CLASS Solution  
    FUNCTION findLUSlength(strs)  
        FUNCTION is_subsequence(s, t)  
            SET iterator_to_t TO the iterator of t  
            RETURN for each character in s checking that the character is found in iterator_to_t  
        END FUNCTION  
        
        SORT strs BY length in descending order AND lexicographically in ascending order  
        
        FOR each index i and word at position i IN strs  
            SET unique TO true  
            FOR each index j and other word at position j IN strs  
                IF i NOT EQUALS j AND is_subsequence of word BY other word IS true  
                    SET unique TO false  
                    BREAK  
                END IF  
            END FOR  
            IF unique IS true  
                RETURN the length of word  
            END IF  
        END FOR  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
