CLASS Solution  
    FUNCTION getXORSum(arr1, arr2)  
        SET xor_sum_arr1 TO zero  
        SET xor_sum_arr2 TO zero  
        FOR each number IN arr1  
            SET xor_sum_arr1 TO xor_sum_arr1 XOR number  
        END FOR  
        FOR each number IN arr2  
            SET xor_sum_arr2 TO xor_sum_arr2 XOR number  
        END FOR  
        RETURN xor_sum_arr1 AND xor_sum_arr2  
    END FUNCTION  
END CLASS
