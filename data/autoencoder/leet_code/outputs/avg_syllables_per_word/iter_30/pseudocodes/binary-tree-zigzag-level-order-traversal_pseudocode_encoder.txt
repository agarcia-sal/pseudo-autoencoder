CLASS TreeNode  
    FUNCTION __init__(self, val SET TO zero, left SET TO none, right SET TO none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at position zero OF values  
    SET i TO one  
    SET queue TO a new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO remove the element from the front OF queue  
        IF i IS LESS THAN the length of values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with value at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the length of values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with value at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value attribute of p IS NOT EQUAL TO value attribute of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left attribute of p, left attribute of q) AND is_same_tree(right attribute of p, right attribute of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION zigzagLevelOrder(self, root)  
        IF root IS none  
            RETURN an empty list  
        END IF  
        SET result TO an empty list  
        SET queue TO a new double ended queue containing root  
        SET left_to_right TO true  
        WHILE queue IS NOT empty  
            SET level_size TO the length of queue  
            SET current_level TO a new empty double ended queue  
            FOR each index FROM zero TO level_size MINUS one  
                SET node TO remove the element from the front OF queue  
                IF left_to_right  
                    APPEND the value attribute of node TO current_level  
                ELSE  
                    PREPEND the value attribute of node TO current_level  
                END IF  
                IF left attribute of node IS NOT none  
                    APPEND left attribute of node TO queue  
                END IF  
                IF right attribute of node IS NOT none  
                    APPEND right attribute of node TO queue  
                END IF  
            END FOR  
            APPEND a list made from current_level TO result  
            SET left_to_right TO NOT left_to_right  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
