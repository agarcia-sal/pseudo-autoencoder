CLASS UnionFind  
    FUNCTION __init__(self, size)  
        SET self.root TO list of integers from zero to size MINUS one  
        SET self.rank TO list of ones of length size  
    END FUNCTION  
  
    FUNCTION find(self, x)  
        IF element at position x OF self.root EQUALS x  
            RETURN x  
        END IF  
        SET element at position x OF self.root TO self.find(element at position x OF self.root)  
        RETURN element at position x OF self.root  
    END FUNCTION  
  
    FUNCTION union(self, x, y)  
        SET rootX TO self.find(x)  
        SET rootY TO self.find(y)  
        IF rootX NOT EQUALS rootY  
            IF element at position rootX OF self.rank GREATER THAN element at position rootY OF self.rank  
                SET element at position rootY OF self.root TO rootX  
            ELSE IF element at position rootX OF self.rank LESS THAN element at position rootY OF self.rank  
                SET element at position rootX OF self.root TO rootY  
            ELSE  
                SET element at position rootY OF self.root TO rootX  
                INCREMENT element at position rootX OF self.rank BY one  
            END IF  
        END IF  
    END FUNCTION  
END CLASS  
  
CLASS Solution  
    FUNCTION accountsMerge(self, accounts)  
        SET email_to_id TO empty dictionary  
        ASSIGN reference TO object UnionFind CREATED WITH the LENGTH OF accounts  
  
        FOR each index i AND unpacked tuple ignoring first element AND emails IN enumeration of accounts  
            FOR each email IN emails  
                IF email NOT IN email_to_id  
                    SET email_to_id at key email TO i  
                ELSE  
                    CALL method union of reference WITH parameters i AND email_to_id at key email  
                END IF  
            END FOR  
        END FOR  
  
        SET merged_emails TO empty dictionary with default type list  
        FOR each email AND account_id IN items of email_to_id  
            SET root_id TO method find of reference WITH parameter account_id  
            APPEND email TO list at key root_id IN merged_emails  
        END FOR  
  
        SET result TO empty list  
        FOR each account_id AND emails IN items of merged_emails  
            SET name TO element at position zero OF element at position account_id OF accounts  
            APPEND list constructed by name PLUS sorted emails TO result  
        END FOR  
  
        RETURN result  
    END FUNCTION  
END CLASS
