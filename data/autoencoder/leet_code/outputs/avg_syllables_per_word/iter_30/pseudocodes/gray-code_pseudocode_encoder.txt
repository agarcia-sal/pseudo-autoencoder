CLASS Solution  
    FUNCTION grayCode(n)  
        IF n EQUALS zero  
            RETURN list containing zero  
        END IF  
        SET prev_gray TO CALL grayCode WITH argument n MINUS one  
        SET mask TO one MULTIPLIED BY two RAISED TO THE POWER OF n MINUS one  
        SET reversed_prev_gray TO prev_gray IN REVERSED ORDER  
        SET appended_values TO empty list  
        FOR each num IN reversed_prev_gray  
            APPEND mask OR num TO appended_values  
        END FOR  
        SET current_gray TO prev_gray PLUS appended_values  
        RETURN current_gray  
    END FUNCTION  
END CLASS
