CLASS Solution  
    FUNCTION nextClosestTime(time)  
        SET digits TO the set of characters in time with colons removed  
        SET hour AND minute TO the integer values of the parts of time split by colon  
        
        FUNCTION generate_times(digits)  
            SET times TO an empty collection  
            FOR each h1 IN digits  
                FOR each h2 IN digits  
                    FOR each m1 IN digits  
                        FOR each m2 IN digits  
                            SET hour_value TO the integer of the concatenation of h1 and h2  
                            SET minute_value TO the integer of the concatenation of m1 and m2  
                            IF hour_value LESS THAN twenty four AND minute_value LESS THAN sixty  
                                APPEND the tuple of hour_value AND minute_value TO times  
                            END IF  
                        END FOR  
                    END FOR  
                END FOR  
            END FOR  
            RETURN times sorted in ascending order  
        END FUNCTION  
        
        SET all_times TO generate_times(digits)  
        SET current_time_index TO the position of the tuple of hour AND minute IN all_times  
        
        IF current_time_index PLUS one LESS THAN the length of all_times  
            SET next_hour AND next_minute TO the element at position current_time_index PLUS one OF all_times  
        ELSE  
            SET next_hour AND next_minute TO the element at position zero OF all_times  
        END IF  
        
        RETURN the string formed by next_hour with leading zero if needed PLUS colon PLUS next_minute with leading zero if needed  
    END FUNCTION  
END CLASS
