CLASS Solution  
    FUNCTION minimumWhiteTiles(floor, numCarpets, carpetLen)  
        SET n TO the LENGTH OF floor  
        SET dp TO a two dimensional collection with n PLUS one rows and numCarpets PLUS one columns all initialized to zero  
        
        FOR i FROM one TO n  
            SET dp element at position i and zero TO dp element at position i MINUS one and zero PLUS one IF character at position i MINUS one OF floor EQUALS one OTHERWISE zero  
        END FOR  
        
        FOR j FROM one TO numCarpets  
            FOR i FROM one TO n  
                SET dp element at position i and j TO dp element at position i MINUS one and j PLUS one IF character at position i MINUS one OF floor EQUALS one OTHERWISE zero  
                IF dp element at position i and j GREATER THAN dp element at position MAXIMUM of zero and i MINUS carpetLen and j MINUS one  
                    SET dp element at position i and j TO dp element at position MAXIMUM of zero and i MINUS carpetLen and j MINUS one  
                END IF  
            END FOR  
        END FOR  
        
        RETURN dp element at position n and numCarpets  
    END FUNCTION  
END CLASS
