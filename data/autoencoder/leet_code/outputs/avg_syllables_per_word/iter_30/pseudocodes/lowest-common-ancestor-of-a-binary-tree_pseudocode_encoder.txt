CLASS TreeNode  
    FUNCTION __init__(self, val with default zero, left with default none, right with default none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at position zero OF values  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the length of values AND element at position i OF values IS not none  
            SET node dot left TO new TreeNode with element at position i OF values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i OF values IS not none  
            SET node dot right TO new TreeNode with element at position i OF values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN the logical AND of is_same_tree with p dot left and q dot left AND is_same_tree with p dot right and q dot right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION lowestCommonAncestor(self, root, p, q)  
        IF root IS none OR root EQUALS p OR root EQUALS q  
            RETURN root  
        END IF  
        SET left TO self dot lowestCommonAncestor with root dot left, p, q  
        SET right TO self dot lowestCommonAncestor with root dot right, p, q  
        IF left IS not none AND right IS not none  
            RETURN root  
        END IF  
        IF left IS not none  
            RETURN left  
        ELSE  
            RETURN right  
        END IF  
    END FUNCTION  
END CLASS
