CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO new TreeNode with value as element at position zero OF values, left as None, right as None  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO first element of queue removed  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values is not None  
            SET node.left TO new TreeNode with value as element at position i OF values, left as None, right as None  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values is not None  
            SET node.right TO new TreeNode with value as element at position i OF values, left as None, right as None  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF value of p.val NOT EQUALS value of q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION isValidBST(self, root)  
        FUNCTION validate(node, low, high)  
            IF node EQUALS None  
                RETURN True  
            END IF  
            IF value of node.val LESS THAN OR EQUAL TO low OR value of node.val GREATER THAN OR EQUAL TO high  
                RETURN False  
            END IF  
            RETURN validate(left child of node, low, value of node.val) AND validate(right child of node, value of node.val, high)  
        END FUNCTION  
        RETURN validate(root, negative infinity, positive infinity)  
    END FUNCTION  
END CLASS
