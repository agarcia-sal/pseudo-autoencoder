CLASS Solution  
    FUNCTION multiply(mat1, mat2)  
        SET m TO the LENGTH OF mat1  
        SET k TO the LENGTH OF element at position zero OF mat1  
        SET k TO the LENGTH OF mat2  
        SET n TO the LENGTH OF element at position zero OF mat2  
        INITIALIZE result AS a list containing m elements each being a list containing n elements each set to zero  
        FOR index i FROM zero TO m MINUS one  
            FOR index j FROM zero TO n MINUS one  
                FOR index p FROM zero TO k MINUS one  
                    INCREMENT element at position j OF element at position i OF result BY the product of element at position p OF element at position i OF mat1 AND element at position j OF element at position p OF mat2  
                END FOR  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
