CLASS Solution  
    FUNCTION getPermutation(n, k)  
        SET numbers TO list of all integers from one TO n  
        DECREMENT k BY one  
        SET result TO empty list  
        FOR i FROM n TO one DECREMENTING BY one  
            SET fact TO factorial of i MINUS one  
            SET index TO k DIVIDED BY fact  
            APPEND element at position index OF numbers TO result  
            REMOVE element at position index FROM numbers  
            SET k TO k MODULO fact  
        END FOR  
        SET joined_result TO concatenation of all elements IN result  
        RETURN joined_result  
    END FUNCTION  
END CLASS
