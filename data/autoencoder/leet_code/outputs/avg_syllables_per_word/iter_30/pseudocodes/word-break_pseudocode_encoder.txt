CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        ASSIGN reference TO the SET of wordDict  
        INITIALIZE dp AS a collection of False values of the LENGTH OF s PLUS one  
        SET element at position zero OF dp TO True  
        FOR i FROM one TO the LENGTH OF s  
            FOR j FROM zero TO i MINUS one  
                IF element at position j OF dp EQUALS True AND substring from position j TO position i OF s IS IN the SET  
                    SET element at position i OF dp TO True  
                    BREAK  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position LENGTH OF s OF dp  
    END FUNCTION  
END CLASS
