CLASS Solution  
    FUNCTION nextGreaterElement(n)  
        SET digits TO list of characters representing n  
        SET length TO the LENGTH OF digits  
        SET i TO length MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND digit at position i OF digits GREATER THAN OR EQUAL TO digit at position i PLUS one OF digits  
            DECREMENT i BY one  
        END WHILE  
        IF i EQUALS minus one  
            RETURN minus one  
        END IF  
        SET j TO length MINUS one  
        WHILE digit at position j OF digits LESS THAN OR EQUAL TO digit at position i OF digits  
            DECREMENT j BY one  
        END WHILE  
        SET temporary TO digit at position i OF digits  
        SET digit at position i OF digits TO digit at position j OF digits  
        SET digit at position j OF digits TO temporary  
        SET suffix TO sublist of digits FROM position i PLUS one TO the END  
        REVERSE suffix  
        SET digits TO concatenation of sublist of digits FROM start TO position i PLUS one AND suffix  
        SET result_string TO concatenation of all elements of digits  
        SET result TO integer representation of result_string  
        IF result GREATER THAN two RAISED TO THE THIRTY ONE MINUS one  
            RETURN minus one  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
