CLASS Solution  
    FUNCTION firstMissingPositive(nums)  
        SET n TO the LENGTH OF nums  
        FOR each i FROM zero TO n MINUS one  
            WHILE nums at position i IS GREATER THAN OR EQUAL TO one AND nums at position i IS LESS THAN OR EQUAL TO n AND nums at position nums at position i MINUS one IS NOT EQUALS nums at position i  
                SET correct_index TO nums at position i MINUS one  
                SET temp TO nums at position i  
                SET nums at position i TO nums at position correct_index  
                SET nums at position correct_index TO temp  
            END WHILE  
        END FOR  
        FOR each i FROM zero TO n MINUS one  
            IF nums at position i IS NOT EQUALS i PLUS one  
                RETURN i PLUS one  
            END IF  
        END FOR  
        RETURN n PLUS one  
    END FUNCTION  
END CLASS
