CLASS Solution  
    FUNCTION minJumps(arr)  
        SET n TO the LENGTH OF arr  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET value_indices TO a new DEFAULT DICTIONARY of empty LISTS  
        FOR each index FROM zero TO n MINUS one  
            SET value TO element at position index OF arr  
            APPEND index TO the LIST at key value IN value_indices  
        END FOR  
        SET queue TO a new DOUBLE ENDED QUEUE containing the TUPLE of zero and zero  
        SET visited TO a new SET containing zero  
        WHILE the LENGTH OF queue GREATER THAN zero  
            REMOVE the LEFTMOST element of queue AND ASSIGN TO current_index and steps  
            IF current_index EQUALS n MINUS one  
                RETURN steps  
            END IF  
            SET next_indices TO a LIST containing current_index PLUS one, current_index MINUS one, and all elements in the LIST at key element at position current_index OF arr IN value_indices  
            FOR each next_index IN next_indices  
                IF next_index GREATER THAN OR EQUAL TO zero AND next_index LESS THAN n AND next_index NOT IN visited  
                    ADD next_index TO visited  
                    APPEND the TUPLE of next_index and steps PLUS one TO queue  
                END IF  
            END FOR  
            SET the LIST at key element at position current_index OF arr IN value_indices TO an empty LIST  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
