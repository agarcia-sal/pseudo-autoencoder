CLASS Solution  
    FUNCTION minInteger(num, k)  
        SET num TO list of characters in num  
        SET n TO length of num  
        WHILE k GREATER THAN zero  
            SET swapped TO false  
            FOR i FROM zero TO n MINUS one MINUS one  
                IF element at position i OF num GREATER THAN element at position i PLUS one OF num  
                    SET min_digit TO character nine  
                    SET min_index TO negative one  
                    FOR j FROM i TO the smaller of i PLUS k PLUS one AND n MINUS one  
                        IF element at position j OF num LESS THAN min_digit  
                            SET min_digit TO element at position j OF num  
                            SET min_index TO j  
                        END IF  
                    END FOR  
                    SET sublist FROM position i TO position min_index PLUS one OF num TO a list starting with element at position min_index OF num FOLLOWED BY elements from position i TO position min_index MINUS one OF num  
                    SET num TO the concatenation of elements before position i OF num FOLLOWED BY sublist FOLLOWED BY elements after position min_index OF num  
                    DECREMENT k BY min_index MINUS i  
                    SET swapped TO true  
                    BREAK  
                END IF  
            END FOR  
            IF swapped EQUALS false  
                BREAK  
            END IF  
        END WHILE  
        RETURN concatenation of elements of num  
    END FUNCTION  
END CLASS
