CLASS Solution  
    FUNCTION matrixScore(grid)  
        FOR each row IN grid  
            IF element at position zero OF row EQUALS zero  
                FOR index FROM zero TO the LENGTH OF row MINUS one  
                    SET element at position index OF row TO one MINUS element at position index OF row  
                END FOR  
            END IF  
        END FOR  
        FOR col FROM one TO the LENGTH OF element at position zero OF grid MINUS one  
            SET count_ones TO zero  
            FOR row FROM zero TO the LENGTH OF grid MINUS one  
                INCREMENT count_ones BY element at position col OF element at position row OF grid  
            END FOR  
            IF count_ones LESS THAN the LENGTH OF grid DIVIDED BY two  
                FOR row FROM zero TO the LENGTH OF grid MINUS one  
                    SET element at position col OF element at position row OF grid TO one MINUS element at position col OF element at position row OF grid  
                END FOR  
            END IF  
        END FOR  
        SET score TO zero  
        FOR each row IN grid  
            SET binary_number TO concatenation of string representations of each element in row  
            SET score TO score PLUS the integer value of binary_number interpreted in base two  
        END FOR  
        RETURN score  
    END FUNCTION  
END CLASS
