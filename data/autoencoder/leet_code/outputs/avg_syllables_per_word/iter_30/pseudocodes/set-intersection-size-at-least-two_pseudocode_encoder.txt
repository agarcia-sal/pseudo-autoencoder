CLASS Solution  
    FUNCTION intersectionSizeTwo(intervals)  
        SORT intervals BY the second element OF each interval THEN BY the negation OF the first element OF each interval  
        SET first TO the second element of the first interval MINUS one  
        SET second TO the second element of the first interval  
        SET count TO two  
        FOR each start and end IN intervals starting from the second interval  
            IF start IS GREATER THAN second  
                SET first TO end MINUS one  
                SET second TO end  
                INCREMENT count BY two  
            ELSE IF start IS GREATER THAN first  
                SET first TO second  
                SET second TO end  
                INCREMENT count BY one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
