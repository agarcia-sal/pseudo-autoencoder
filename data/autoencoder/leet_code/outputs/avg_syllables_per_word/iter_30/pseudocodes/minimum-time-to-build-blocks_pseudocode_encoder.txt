CLASS Solution  
    FUNCTION minBuildTime(blocks, split)  
        CONVERT blocks INTO a minimum heap structure  
        WHILE the LENGTH OF blocks GREATER THAN one  
            REMOVE the smallest element FROM blocks AND ASSIGN TO first  
            REMOVE the next smallest element FROM blocks AND ASSIGN TO second  
            SET combined_time TO split PLUS second  
            ADD combined_time INTO blocks  
        END WHILE  
        RETURN the element at position zero OF blocks  
    END FUNCTION  
END CLASS
