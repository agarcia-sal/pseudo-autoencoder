CLASS Solution  
    FUNCTION maxValueAfterReverse(nums)  
        SET n TO the LENGTH OF nums  
        IF n EQUALS two  
            RETURN the ABSOLUTE VALUE OF the DIFFERENCE BETWEEN element at position zero OF nums AND element at position one OF nums  
        END IF  
        SET initial_value TO zero  
        FOR index FROM one TO n MINUS one  
            INCREMENT initial_value BY the ABSOLUTE VALUE OF the DIFFERENCE BETWEEN element at position index OF nums AND element at position index MINUS one OF nums  
        END FOR  
        SET max_edge_gain TO zero  
        FOR index FROM one TO n MINUS one  
            SET candidate_gain TO the ABSOLUTE VALUE OF the DIFFERENCE BETWEEN element at position zero OF nums AND element at position index OF nums MINUS the ABSOLUTE VALUE OF the DIFFERENCE BETWEEN element at position index OF nums AND element at position index MINUS one OF nums  
            IF candidate_gain GREATER THAN max_edge_gain  
                SET max_edge_gain TO candidate_gain  
            END IF  
        END FOR  
        FOR index FROM zero TO n MINUS two  
            SET candidate_gain TO the ABSOLUTE VALUE OF the DIFFERENCE BETWEEN element at position n MINUS one OF nums AND element at position index OF nums MINUS the ABSOLUTE VALUE OF the DIFFERENCE BETWEEN element at position index PLUS one OF nums AND element at position index OF nums  
            IF candidate_gain GREATER THAN max_edge_gain  
                SET max_edge_gain TO candidate_gain  
            END IF  
        END FOR  
        SET min_pair TO the MINIMUM OF the MAXIMUM OF elements at positions index AND index PLUS one OF nums FOR index FROM zero TO n MINUS two  
        SET max_pair TO the MAXIMUM OF the MINIMUM OF elements at positions index AND index PLUS one OF nums FOR index FROM zero TO n MINUS two  
        SET general_gain TO the MAXIMUM OF zero AND two MULTIPLIED BY the DIFFERENCE BETWEEN max_pair AND min_pair  
        RETURN initial_value PLUS the MAXIMUM OF max_edge_gain AND general_gain  
    END FUNCTION  
END CLASS
