CLASS Solution  
    FUNCTION findMaxForm(strs, m, n)  
        SET dp TO two dimensional array of zeros with dimensions m plus one BY n plus one  
        FOR each s IN strs  
            SET count_0 TO number of occurrences of zero character in s  
            SET count_1 TO number of occurrences of one character in s  
            FOR i FROM m DOWN TO count_0  
                FOR j FROM n DOWN TO count_1  
                    SET dp element at position i AND j TO maximum of dp element at position i AND j AND dp element at position i MINUS count_0 AND j MINUS count_1 PLUS one  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position m AND n  
    END FUNCTION  
END CLASS
