CLASS Solution  
    FUNCTION reverse(x)  
        IF x LESS THAN zero  
            SET sign TO negative one  
        ELSE  
            SET sign TO one  
        END IF  
        SET reversed_x TO integer value of the string of the absolute value of x in reverse order  
        SET result TO sign MULTIPLIED BY reversed_x  
        IF result LESS THAN negative two TO THE POWER OF thirty ONE OR result GREATER THAN two TO THE POWER OF thirty ONE MINUS one  
            RETURN zero  
        END IF  
        RETURN result  
    END FUNCTION  
END CLASS
