CLASS Solution  
    FUNCTION jobScheduling(startTime, endTime, profit)  
        SET jobs TO the list of tuples where each tuple contains corresponding elements of startTime, endTime, and profit sorted by the second element of each tuple in ascending order  
        SET end_times TO the list containing the second element of each tuple in jobs  
        SET dp TO the list containing zero repeated the length of jobs plus one times  
        FOR i FROM one TO the length of jobs  
            SET j TO the result of finding the rightmost position in end_times where the first element of the tuple at position i minus one in jobs can be inserted without breaking the order  
            SET dp at position i TO the value which is the greater of dp at position i minus one and the sum of dp at position j and the third element of the tuple at position i minus one in jobs  
        END FOR  
        RETURN the last element in dp  
    END FUNCTION  
END CLASS
