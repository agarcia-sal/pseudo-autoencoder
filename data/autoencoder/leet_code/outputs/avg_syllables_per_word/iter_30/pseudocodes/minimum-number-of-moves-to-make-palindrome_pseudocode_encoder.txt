CLASS Solution  
    FUNCTION minMovesToMakePalindrome WITH PARAMETER s  
        SET s TO list form of s  
        SET n TO the LENGTH OF s  
        SET moves TO zero  
        WHILE the LENGTH OF s GREATER THAN one  
            FOR i FROM the LENGTH OF s MINUS one DOWN TO one  
                IF element at position i OF s EQUALS element at position zero OF s  
                    FOR j FROM i TO the LENGTH OF s MINUS two  
                        SWAP element at position j OF s WITH element at position j PLUS one OF s  
                        INCREMENT moves BY one  
                    END FOR  
                    REMOVE element at position zero FROM s  
                    REMOVE element at position negative one FROM s  
                    BREAK  
                END IF  
            END FOR  
            IF the LOOP completed WITHOUT BREAK  
                REMOVE element at position zero FROM s  
            END IF  
        END WHILE  
        RETURN moves  
    END FUNCTION  
END CLASS
