CLASS Solution  
    FUNCTION makeArrayIncreasing(arr1, arr2)  
        CALL sort ON arr2  
        SET dp TO dictionary WITH key negative one AND value zero  
        FOR each num IN arr1  
            SET new_dp TO empty dictionary  
            FOR each last_num AND ops IN dp  
                IF num GREATER THAN last_num  
                    SET current_min TO get value for key num FROM new_dp OR infinity  
                    SET new_dp at key num TO minimum of current_min AND ops  
                END IF  
                SET idx TO position found by binary search for smallest element greater than last_num IN arr2  
                IF idx LESS THAN length of arr2  
                    SET candidate TO element at position idx OF arr2  
                    SET current_min TO get value for key candidate FROM new_dp OR infinity  
                    SET new_dp at key candidate TO minimum of current_min AND ops PLUS one  
                END IF  
            END FOR  
            IF new_dp is empty  
                RETURN negative one  
            END IF  
            SET dp TO new_dp  
        END FOR  
        RETURN minimum value AMONG values IN dp  
    END FUNCTION  
END CLASS
