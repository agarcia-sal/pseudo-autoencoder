CLASS Solution  
    FUNCTION minRefuelStops(target, startFuel, stations)  
        ASSIGN reference TO an empty list named max_heap  
        SET current_fuel TO startFuel  
        SET stops TO zero  
        APPEND a list containing target and zero TO stations  
        FOR each pair position and fuel IN stations  
            WHILE the list max_heap is not empty AND current_fuel LESS THAN position  
                INCREMENT current_fuel BY the negation of the removal of the last element from max_heap  
                INCREMENT stops BY one  
            END WHILE  
            IF current_fuel LESS THAN position  
                RETURN the negation of one  
            END IF  
            APPEND the negation of fuel TO the list max_heap  
        END FOR  
        RETURN stops  
    END FUNCTION  
END CLASS
