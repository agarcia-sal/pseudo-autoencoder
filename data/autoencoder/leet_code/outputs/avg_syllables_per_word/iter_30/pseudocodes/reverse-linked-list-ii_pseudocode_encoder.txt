CLASS ListNode  
    FUNCTION __init__ WITH PARAMETERS val AND next  
        SET the attribute val OF this instance TO val  
        SET the attribute next OF this instance TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node WITH PARAMETER values  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO a new instance of ListNode initialized with element at position zero OF values  
    SET p TO head  
    FOR each val IN the elements from position one TO the end OF values  
        SET node TO a new instance of ListNode initialized with val  
        SET the attribute next OF p TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list WITH PARAMETERS p1 AND p2  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the condition that the attribute val OF p1 EQUALS the attribute val OF p2 AND the recursive call to is_same_list WITH the parameters attribute next OF p1 AND attribute next OF p2  
END FUNCTION  

CLASS Solution  
    FUNCTION reverseBetween WITH PARAMETERS head, left, right  
        IF head EQUALS none OR left EQUALS right  
            RETURN head  
        END IF  
        SET dummy TO a new instance of ListNode initialized with zero  
        SET the attribute next OF dummy TO head  
        SET prev TO dummy  
        FOR each index FROM one TO left MINUS one  
            SET prev TO the attribute next OF prev  
        END FOR  
        SET current TO the attribute next OF prev  
        FOR each index FROM one TO right MINUS left  
            SET next_node TO the attribute next OF current  
            SET the attribute next OF current TO the attribute next OF next_node  
            SET the attribute next OF next_node TO the attribute next OF prev  
            SET the attribute next OF prev TO next_node  
        END FOR  
        RETURN the attribute next OF dummy  
    END FUNCTION  
END CLASS
