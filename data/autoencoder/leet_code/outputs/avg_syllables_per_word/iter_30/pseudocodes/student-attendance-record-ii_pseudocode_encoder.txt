CLASS Solution  
    FUNCTION checkRecord(n)  
        SET MOD TO one billion PLUS seven  
        INITIALIZE dp AS three dimensional collection WITH dimensions n PLUS one BY two BY three FILLED WITH zero  
        SET element at position zero OF element at position zero OF element at position zero OF dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO one  
                FOR k FROM zero TO two  
                    SET element at position zero OF element at position j OF element at position i OF dp TO element at position zero OF element at position j OF element at position i OF dp PLUS element at position k OF element at position j OF element at position i MINUS one OF dp MODULO MOD  
                    IF j EQUALS zero  
                        SET element at position zero OF element at position one OF element at position i OF dp TO element at position zero OF element at position one OF element at position i OF dp PLUS element at position k OF element at position zero OF element at position i MINUS one OF dp MODULO MOD  
                    END IF  
                    IF k LESS THAN two  
                        SET element at position k PLUS one OF element at position j OF element at position i OF dp TO element at position k PLUS one OF element at position j OF element at position i OF dp PLUS element at position k OF element at position j OF element at position i MINUS one OF dp MODULO MOD  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        SET result TO zero  
        FOR j FROM zero TO one  
            FOR k FROM zero TO two  
                SET result TO result PLUS element at position k OF element at position j OF element at position n OF dp MODULO MOD  
            END FOR  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
