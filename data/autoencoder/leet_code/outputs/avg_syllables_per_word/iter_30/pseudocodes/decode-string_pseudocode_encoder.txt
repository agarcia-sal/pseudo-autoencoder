CLASS Solution  
    FUNCTION decodeString(s)  
        SET stack TO empty list  
        SET a TO empty string  
        SET d TO empty string  
        FOR each char IN s  
            IF char is a digit  
                SET d TO d concatenated with char  
            ELSE IF char is an alphabetic character  
                SET a TO a concatenated with char  
            ELSE IF char equals the opening bracket character  
                APPEND tuple of a and d converted to integer TO stack  
                SET a TO empty string  
                SET d TO empty string  
            ELSE IF char equals the closing bracket character  
                REMOVE last element from stack and ASSIGN to p and n  
                SET a TO concatenation of p and a repeated n times  
            END IF  
        END FOR  
        RETURN a  
    END FUNCTION  
END CLASS
