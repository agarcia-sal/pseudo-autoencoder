CLASS Solution  
    FUNCTION multiply(num1, num2)  
        IF num1 EQUALS zero OR num2 EQUALS zero  
            RETURN zero  
        END IF  
        SET result TO a list of zero REPEATED the LENGTH OF num1 PLUS the LENGTH OF num2  
        SET num1 TO the string num1 REVERSED  
        SET num2 TO the string num2 REVERSED  
        FOR each index i FROM zero TO the LENGTH OF num1 MINUS one  
            FOR each index j FROM zero TO the LENGTH OF num2 MINUS one  
                SET mul TO the character at position i of num1 CONVERTED TO number MULTIPLIED BY the character at position j of num2 CONVERTED TO number  
                SET p1 TO i PLUS j  
                SET p2 TO i PLUS j PLUS one  
                SET sum_ TO mul PLUS the element at position p1 OF result  
                SET element at position p1 OF result TO the remainder of sum_ DIVIDED BY ten  
                INCREMENT element at position p2 OF result BY the integer division of sum_ BY ten  
            END FOR  
        END FOR  
        SET result_str TO the concatenation of the elements of result REVERSED CONVERTED TO strings  
        SET result_str TO result_str WITH all leading zero characters REMOVED  
        RETURN result_str  
    END FUNCTION  
END CLASS
