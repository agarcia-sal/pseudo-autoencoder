CLASS Solution  
    FUNCTION calculate WITH PARAMETER s  
        FUNCTION helper WITH PARAMETER s  
            SET stack TO empty list  
            SET sign TO plus  
            SET num TO zero  
            WHILE the LENGTH OF s GREATER THAN zero  
                SET char TO the element at position zero OF s and REMOVE the element at position zero OF s  
                IF char IS a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                IF char EQUALS open parenthesis  
                    SET num TO result of calling helper WITH PARAMETER s  
                END IF  
                IF char IS NOT a digit AND char IS NOT a whitespace OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS plus  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus  
                        APPEND num MULTIPLIED BY negative one TO stack  
                    ELSE IF sign EQUALS multiply  
                        SET the last element of stack TO the last element of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS divide  
                        SET the last element of stack TO the integer division of the last element of stack BY num  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS close parenthesis  
                    BREAK  
                END IF  
            END WHILE  
            RETURN the sum of all elements in stack  
        END FUNCTION  
        RETURN the result of calling helper WITH PARAMETER the list conversion of s  
    END FUNCTION  
END CLASS
