CLASS Solution  
    FUNCTION new21Game(n, k, maxPts)  
        FUNCTION dfs(i)  
            IF i GREATER THAN OR EQUAL TO k  
                IF i LESS THAN OR EQUAL TO n  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF i EQUALS k MINUS one  
                IF n MINUS k PLUS one LESS THAN maxPts  
                    SET numerator TO n MINUS k PLUS one  
                ELSE  
                    SET numerator TO maxPts  
                END IF  
                SET probability TO numerator DIVIDED BY maxPts  
                RETURN probability  
            END IF  
            SET value1 TO dfs of i PLUS one  
            SET value2 TO dfs of i PLUS one MINUS dfs of i PLUS maxPts PLUS one  
            SET result TO value1 PLUS value2 DIVIDED BY maxPts  
            RETURN result  
        END FUNCTION  
        RETURN dfs of zero  
    END FUNCTION  
END CLASS
