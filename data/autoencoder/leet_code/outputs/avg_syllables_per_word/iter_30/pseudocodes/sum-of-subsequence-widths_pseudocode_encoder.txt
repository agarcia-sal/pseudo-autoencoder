CLASS Solution  
    FUNCTION sumSubseqWidths(nums)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SORT nums IN ascending order  
        SET n TO the LENGTH OF nums  
        SET total_sum TO zero  
        SET power TO a list with n elements all set to one  
        
        FOR i FROM one TO n MINUS one  
            SET power at position i TO power at position i MINUS one MULTIPLIED BY two MODULO MOD  
        END FOR  
        
        FOR i FROM zero TO n MINUS one  
            SET total_sum TO total_sum PLUS nums at position i MULTIPLIED BY power at position i MINUS power at position n MINUS i MINUS one MODULO MOD  
        END FOR  
        
        RETURN total_sum  
    END FUNCTION  
END CLASS
