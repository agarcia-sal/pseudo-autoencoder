CLASS Solution  
    FUNCTION maxProduct(s)  
        SET n TO the LENGTH OF s  
        SET max_len_end TO a list of ones of LENGTH n  
        SET max_len_start TO a list of ones of LENGTH n  
        
        FUNCTION expand_around_center(left, right)  
            WHILE left IS GREATER THAN OR EQUAL TO zero AND right IS LESS THAN n AND character at position left OF s EQUALS character at position right OF s  
                SET length TO right MINUS left PLUS one  
                IF max_len_end at position right IS LESS THAN length  
                    SET max_len_end at position right TO length  
                END IF  
                IF max_len_start at position left IS LESS THAN length  
                    SET max_len_start at position left TO length  
                END IF  
                DECREMENT left BY one  
                INCREMENT right BY one  
            END WHILE  
        END FUNCTION  
        
        FOR each index i FROM zero TO n MINUS one  
            CALL expand_around_center WITH i AND i  
        END FOR  
        
        SET prefix_max TO a list of zeros of LENGTH n  
        SET suffix_max TO a list of zeros of LENGTH n  
        
        SET prefix_max at position zero TO max_len_end at position zero  
        FOR each index i FROM one TO n MINUS one  
            IF prefix_max at position i MINUS one IS GREATER THAN max_len_end at position i  
                SET prefix_max at position i TO prefix_max at position i MINUS one  
            ELSE  
                SET prefix_max at position i TO max_len_end at position i  
            END IF  
        END FOR  
        
        SET suffix_max at position n MINUS one TO max_len_start at position n MINUS one  
        FOR each index i FROM n MINUS two DOWN TO zero  
            IF suffix_max at position i PLUS one IS GREATER THAN max_len_start at position i  
                SET suffix_max at position i TO suffix_max at position i PLUS one  
            ELSE  
                SET suffix_max at position i TO max_len_start at position i  
            END IF  
        END FOR  
        
        SET max_product TO zero  
        FOR each index i FROM zero TO n MINUS two  
            SET product TO prefix_max at position i MULTIPLIED BY suffix_max at position i PLUS one  
            IF max_product IS LESS THAN product  
                SET max_product TO product  
            END IF  
        END FOR  
        
        RETURN max_product  
    END FUNCTION  
END CLASS
