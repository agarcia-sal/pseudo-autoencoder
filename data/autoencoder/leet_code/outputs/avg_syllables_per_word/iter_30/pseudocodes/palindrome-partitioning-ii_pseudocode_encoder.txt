CLASS Solution  
    FUNCTION minCut(s)  
        SET n TO the LENGTH OF s  
        IF n LESS THAN OR EQUAL TO one  
            RETURN zero  
        END IF  
        
        SET is_palindrome TO a two-dimensional collection of False with size n BY n  
        
        FOR i FROM zero TO n MINUS one  
            SET element at position i AND position i OF is_palindrome TO True  
        END FOR  
        
        FOR length FROM two TO n  
            FOR start FROM zero TO n MINUS length  
                SET end TO start PLUS length MINUS one  
                IF length EQUALS two  
                    SET element at position start AND position end OF is_palindrome TO whether character at position start OF s EQUALS character at position end OF s  
                ELSE  
                    SET element at position start AND position end OF is_palindrome TO whether character at position start OF s EQUALS character at position end OF s AND element at position start PLUS one AND position end MINUS one OF is_palindrome  
                END IF  
            END FOR  
        END FOR  
        
        SET min_cuts TO a collection of zero with length n  
        FOR i FROM zero TO n MINUS one  
            IF element at position zero AND position i OF is_palindrome  
                SET element at position i OF min_cuts TO zero  
            ELSE  
                SET element at position i OF min_cuts TO positive infinity  
                FOR j FROM one TO i  
                    IF element at position j AND position i OF is_palindrome AND element at position j MINUS one OF min_cuts PLUS one LESS THAN element at position i OF min_cuts  
                        SET element at position i OF min_cuts TO element at position j MINUS one OF min_cuts PLUS one  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN element at position n MINUS one OF min_cuts  
    END FUNCTION  
END CLASS
