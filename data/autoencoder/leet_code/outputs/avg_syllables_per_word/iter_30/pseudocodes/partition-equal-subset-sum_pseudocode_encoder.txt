CLASS Solution  
    FUNCTION canPartition(nums)  
        SET total_sum TO the SUM OF nums  
        IF total_sum MODULO two NOT EQUALS zero  
            RETURN False  
        END IF  
        SET target TO total_sum DIVIDED BY two  
        SET n TO the LENGTH OF nums  
        SET dp TO a list of False with length target PLUS one  
        SET element at position zero OF dp TO True  
        FOR each num IN nums  
            FOR i FROM target DOWN TO num  
                SET element at position i OF dp TO element at position i OF dp OR element at position i MINUS num OF dp  
            END FOR  
        END FOR  
        RETURN element at position target OF dp  
    END FUNCTION  
END CLASS
