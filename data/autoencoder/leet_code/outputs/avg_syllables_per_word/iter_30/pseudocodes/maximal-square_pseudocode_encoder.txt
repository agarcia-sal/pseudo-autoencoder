CLASS Solution  
    FUNCTION maximalSquare(matrix)  
        IF matrix EQUALS empty OR element at position zero OF matrix EQUALS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero OF matrix  
        SET dp TO two dimensional collection of zeros with dimensions m PLUS one BY n PLUS one  
        SET max_side TO zero  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one OF matrix at position j MINUS one EQUALS character one  
                    SET dp at position i AND j TO the MINIMUM OF dp at position i MINUS one AND j AND dp at position i AND j MINUS one AND dp at position i MINUS one AND j MINUS one PLUS one  
                    SET max_side TO the GREATER OF max_side AND dp at position i AND j  
                END IF  
            END FOR  
        END FOR  
        RETURN max_side MULTIPLIED BY max_side  
    END FUNCTION  
END CLASS
