CLASS Solution  
    FUNCTION compareVersion(version1, version2)  
        SET v1_revisions TO the list of substrings of version1 split by the full stop word  
        SET v2_revisions TO the list of substrings of version2 split by the full stop word  
        SET max_length TO the greater of the length of v1_revisions and the length of v2_revisions  
        FOR index FROM zero TO max_length MINUS one  
            IF index LESS THAN the length of v1_revisions  
                SET v1 TO the integer conversion of element at position index OF v1_revisions  
            ELSE  
                SET v1 TO zero  
            END IF  
            IF index LESS THAN the length of v2_revisions  
                SET v2 TO the integer conversion of element at position index OF v2_revisions  
            ELSE  
                SET v2 TO zero  
            END IF  
            IF v1 LESS THAN v2  
                RETURN minus one  
            ELSE IF v1 GREATER THAN v2  
                RETURN one  
            END IF  
        END FOR  
        RETURN zero  
    END FUNCTION  
END CLASS
