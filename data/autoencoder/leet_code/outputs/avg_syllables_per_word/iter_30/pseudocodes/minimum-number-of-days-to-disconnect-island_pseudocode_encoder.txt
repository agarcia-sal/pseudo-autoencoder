CLASS Solution  
    FUNCTION minDays(grid)  
        FUNCTION dfs(x, y, visited)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF the element at position zero OF grid OR the pair of x AND y IS IN visited OR the element at position x OF grid at position y EQUALS zero  
                RETURN  
            END IF  
            ADD the pair of x AND y TO visited  
            FOR each pair dx, dy IN the list of pairs negative one zero, one zero, zero negative one, zero one  
                CALL dfs WITH x PLUS dx, y PLUS dy, visited  
            END FOR  
        END FUNCTION  

        FUNCTION is_connected()  
            SET visited TO an empty set  
            SET land_cells TO the list of pairs i, j FOR each i IN the range from zero TO the LENGTH OF grid AND each j IN the range from zero TO the LENGTH OF the element at position zero OF grid WHERE the element at position i OF grid at position j EQUALS one  
            IF the LENGTH OF land_cells EQUALS zero  
                RETURN True  
            END IF  
            CALL dfs WITH the first element at position zero OF land_cells, the second element at position zero OF land_cells, visited  
            RETURN the LENGTH OF visited EQUALS the LENGTH OF land_cells  
        END FUNCTION  

        IF NOT the result of calling is_connected()  
            RETURN zero  
        END IF  

        FOR each i IN the range from zero TO the LENGTH OF grid  
            FOR each j IN the range from zero TO the LENGTH OF the element at position zero OF grid  
                IF the element at position i OF grid at position j EQUALS one  
                    SET the element at position i OF grid at position j TO zero  
                    IF NOT the result of calling is_connected()  
                        RETURN one  
                    END IF  
                    SET the element at position i OF grid at position j TO one  
                END IF  
            END FOR  
        END FOR  

        RETURN two  
    END FUNCTION  
END CLASS
