CLASS Solution  
    FUNCTION findPermutation(s)  
        SET result TO empty list  
        FOR index FROM zero TO the LENGTH OF s PLUS one  
            APPEND index PLUS one TO result  
        END FOR  
        SET start TO zero  
        FOR end FROM zero TO the LENGTH OF s  
            IF end EQUALS the LENGTH OF s OR element at position end OF s EQUALS character representing increasing  
                SET segment_to_reverse TO sublist of result from position start TO position end PLUS one  
                SET reversed_segment TO reverse of segment_to_reverse  
                REPLACE elements from position start TO position end PLUS one IN result WITH reversed_segment  
                SET start TO end PLUS one  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
