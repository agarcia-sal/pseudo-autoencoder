CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN reference TO None  
    END IF  
    SET root TO new TreeNode with value at position zero OF values  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element removed from left end OF queue  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.left TO new TreeNode with value at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN the LENGTH OF values AND element at position i OF values IS NOT None  
            SET node.right TO new TreeNode with value at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS reference TO None AND q IS reference TO None  
        RETURN True  
    ELSE IF p IS reference TO None OR q IS reference TO None  
        RETURN False  
    ELSE IF value property OF p IS NOT EQUALS TO value property OF q  
        RETURN False  
    ELSE  
        RETURN the result of is_same_tree called with left property OF p and left property OF q AND the result of is_same_tree called with right property OF p and right property OF q  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findLeaves(self, root)  
        FUNCTION collectLeaves(node)  
            IF node IS reference TO None  
                RETURN negative one  
            END IF  
            SET left_height TO the result of collectLeaves called with left property OF node  
            SET right_height TO the result of collectLeaves called with right property OF node  
            SET current_height TO larger of left_height and right_height PLUS one  
            IF current_height IS GREATER THAN OR EQUAL TO the LENGTH OF result  
                APPEND empty list TO result  
            END IF  
            APPEND value property OF node TO element at position current_height OF result  
            RETURN current_height  
        END FUNCTION  
        SET result TO empty list  
        CALL collectLeaves with root  
        RETURN result  
    END FUNCTION  
END CLASS
