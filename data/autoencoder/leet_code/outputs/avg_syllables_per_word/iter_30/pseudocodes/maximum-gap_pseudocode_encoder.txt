CLASS Solution  
    FUNCTION maximumGap(nums)  
        IF the LENGTH OF nums LESS THAN two  
            RETURN zero  
        END IF  
        
        FUNCTION countingSort(arr, exp)  
            SET n TO the LENGTH OF arr  
            SET output TO a new list of length n filled with zero  
            SET count TO a new list of length ten filled with zero  
            
            FOR each i FROM zero TO n MINUS one  
                SET index TO the INTEGER DIVISION OF element at position i OF arr BY exp  
                INCREMENT element at position index MODULO ten OF count BY one  
            END FOR  
            
            FOR each i FROM one TO nine  
                INCREMENT element at position i OF count BY element at position i MINUS one OF count  
            END FOR  
            
            SET i TO n MINUS one  
            WHILE i GREATER THAN OR EQUAL TO zero  
                SET index TO the INTEGER DIVISION OF element at position i OF arr BY exp  
                SET element at position element at position index MODULO ten OF count MINUS one OF output TO element at position i OF arr  
                DECREMENT element at position index MODULO ten OF count BY one  
                DECREMENT i BY one  
            END WHILE  
            
            FOR each i FROM zero TO n MINUS one  
                SET element at position i OF arr TO element at position i OF output  
            END FOR  
        END FUNCTION  
        
        FUNCTION radixSort(arr)  
            SET max1 TO the maximum element in arr  
            SET exp TO one  
            WHILE the INTEGER DIVISION OF max1 BY exp GREATER THAN zero  
                CALL countingSort with arr and exp  
                MULTIPLY exp BY ten  
            END WHILE  
        END FUNCTION  
        
        CALL radixSort with nums  
        
        SET max_diff TO zero  
        FOR each i FROM one TO the LENGTH OF nums MINUS one  
            SET current_diff TO element at position i OF nums MINUS element at position i MINUS one OF nums  
            IF current_diff GREATER THAN max_diff  
                SET max_diff TO current_diff  
            END IF  
        END FOR  
        
        RETURN max_diff  
    END FUNCTION  
END CLASS
