CLASS Solution  
    FUNCTION maximumGood(statements)  
        SET n TO the LENGTH OF statements  
        SET max_good TO zero  
        FOR i FROM zero TO two RAISED TO THE POWER OF n MINUS one  
            SET good TO a list of n elements all set to False  
            SET bad TO a list of n elements all set to False  
            FOR j FROM zero TO n MINUS one  
                IF the bitwise AND of i AND two RAISED TO THE POWER OF j IS NOT EQUALS zero  
                    SET the element at position j OF good TO True  
                ELSE  
                    SET the element at position j OF bad TO True  
                END IF  
            END FOR  
            SET valid TO True  
            FOR j FROM zero TO n MINUS one  
                IF the element at position j OF good IS True  
                    FOR k FROM zero TO n MINUS one  
                        IF (the element at position j of statements at position k EQUALS zero AND the element at position k OF good IS True) OR (the element at position j of statements at position k EQUALS one AND the element at position k OF bad IS True)  
                            SET valid TO False  
                            BREAK  
                        END IF  
                    END FOR  
                    IF valid IS False  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            IF valid IS True  
                SET max_good TO the GREATER OF max_good AND the SUM OF the values in good  
            END IF  
        END FOR  
        RETURN max_good  
    END FUNCTION  
END CLASS
