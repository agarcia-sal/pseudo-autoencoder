CLASS ListNode  
    FUNCTION __init__(self, val, next)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET head TO new ListNode with val equal to element at position zero OF values and next equal to none  
    SET p TO head  
    FOR each val IN elements from position one TO the last position OF values  
        SET node TO new ListNode with val equal to val and next equal to none  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the evaluation of self val EQUALS p2 val AND the result of is_same_list with parameters p1 next and p2 next  
END FUNCTION  

CLASS Solution  
    FUNCTION deleteDuplicates(self, head)  
        SET dummy TO new ListNode with val equal to zero and next equal to head  
        SET prev TO dummy  
        WHILE head IS NOT none  
            IF head next IS NOT none AND head val EQUALS head next val  
                WHILE head next IS NOT none AND head val EQUALS head next val  
                    SET head TO head next  
                END WHILE  
                SET prev next TO head next  
            ELSE  
                SET prev TO prev next  
            END IF  
            SET head TO head next  
        END WHILE  
        RETURN dummy next  
    END FUNCTION  
END CLASS
