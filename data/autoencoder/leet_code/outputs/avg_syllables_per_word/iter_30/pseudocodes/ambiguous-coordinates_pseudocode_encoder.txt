CLASS Solution  
    FUNCTION ambiguousCoordinates(s)  
        FUNCTION make(frag)  
            SET n TO the LENGTH OF frag  
            FOR d FROM one TO n  
                SET left TO substring from position zero TO position d OF frag  
                SET right TO substring from position d TO the LENGTH OF frag OF frag  
                IF (left DOES NOT START WITH character zero OR left EQUALS character zero) AND (right DOES NOT END WITH character zero)  
                    YIELD concatenation of left PLUS decimal point IF d DOES NOT EQUAL n PLUS right  
                END IF  
            END FOR  
        END FUNCTION  
        SET s TO substring from position one TO position the LENGTH OF s MINUS one OF s  
        RETURN a list COMPOSED OF concatenation of left parenthesis PLUS a PLUS comma PLUS space PLUS b PLUS right parenthesis FOR i FROM one TO the LENGTH OF s MINUS one FOR each a AND b IN the Cartesian PRODUCT OF make(substring from position zero TO position i OF s) AND make(substring from position i TO the LENGTH OF s OF s)  
    END FUNCTION  
END CLASS
