CLASS Solution  
    FUNCTION minFallingPathSum(grid)  
        IF grid EQUALS empty OR the LENGTH OF grid at position zero EQUALS zero  
            RETURN zero  
        END IF  
        SET n TO the LENGTH OF grid  
        IF n EQUALS one  
            RETURN the minimum element at position zero OF grid  
        END IF  
        SET dp TO a copy of the list at position zero OF grid  
        FOR i FROM one TO n MINUS one  
            SET new_dp TO a list of zero repeated n times  
            SET min1 TO positive infinity  
            SET min2 TO positive infinity  
            SET min1_index TO a negative one  
            FOR j FROM zero TO n MINUS one  
                IF element at position j OF dp LESS THAN min1  
                    SET min2 TO min1  
                    SET min1 TO element at position j OF dp  
                    SET min1_index TO j  
                ELSE IF element at position j OF dp LESS THAN min2  
                    SET min2 TO element at position j OF dp  
                END IF  
            END FOR  
            FOR j FROM zero TO n MINUS one  
                IF j EQUALS min1_index  
                    SET element at position j OF new_dp TO element at position j OF grid at position i PLUS min2  
                ELSE  
                    SET element at position j OF new_dp TO element at position j OF grid at position i PLUS min1  
                END IF  
            END FOR  
            SET dp TO a copy of new_dp  
        END FOR  
        RETURN the minimum element OF dp  
    END FUNCTION  
END CLASS
