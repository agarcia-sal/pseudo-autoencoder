CLASS Solution  
    FUNCTION digitsCount(d, low, high)  
        FUNCTION count_digit(n, digit)  
            SET count TO zero  
            SET power_of_ten TO one  
            SET current TO n  
            WHILE current GREATER THAN zero  
                SET last TO remainder of current DIVIDED BY ten  
                SET current TO integer division of current BY ten  
                SET full_cycles TO current  
                IF digit EQUALS zero  
                    DECREMENT full_cycles BY one  
                END IF  
                INCREMENT count BY full_cycles MULTIPLIED BY power_of_ten  
                IF last GREATER THAN digit  
                    INCREMENT count BY power_of_ten  
                ELSE IF last EQUALS digit  
                    INCREMENT count BY remainder of n DIVIDED BY power_of_ten PLUS one  
                END IF  
                MULTIPLY power_of_ten BY ten  
            END WHILE  
            RETURN count  
        END FUNCTION  

        RETURN count_digit(high, d) MINUS count_digit(low MINUS one, d)  
    END FUNCTION  
END CLASS
