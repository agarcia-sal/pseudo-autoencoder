CLASS Solution  
    FUNCTION smallestGoodBase(n)  
        SET n TO integer value of n  
        SET max_length TO the logarithm base two of n plus one converted to integer  
        FOR m FROM max_length DOWN TO two  
            SET k TO the integer value of n raised to the power of one divided by m minus one  
            IF k LESS THAN two  
                CONTINUE  
            END IF  
            SET num TO the integer division of k raised to the power of m minus one BY k minus one  
            IF num EQUALS n  
                RETURN the string conversion of k  
            END IF  
        END FOR  
        RETURN the string conversion of n minus one  
    END FUNCTION  
END CLASS
