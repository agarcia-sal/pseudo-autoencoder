CLASS Solution  
    FUNCTION kthSmallest(matrix, k)  
        FUNCTION countLessEqual(mid)  
            SET count TO zero  
            SET row TO the LENGTH OF matrix MINUS one  
            SET col TO zero  
            WHILE row GREATER THAN OR EQUAL TO zero AND col LESS THAN the LENGTH OF matrix  
                IF element at position row of matrix AND position col LESS THAN OR EQUAL TO mid  
                    INCREMENT count BY row PLUS one  
                    INCREMENT col BY one  
                ELSE  
                    DECREMENT row BY one  
                END IF  
            END WHILE  
            RETURN count  
        END FUNCTION  
      
        SET n TO the LENGTH OF matrix  
        SET low TO element at position zero of matrix AND position zero  
        SET high TO element at position n MINUS one of matrix AND position n MINUS one  
      
        WHILE low LESS THAN high  
            SET mid TO low PLUS high DIVIDED BY two  
            IF countLessEqual(mid) LESS THAN k  
                SET low TO mid PLUS one  
            ELSE  
                SET high TO mid  
            END IF  
        END WHILE  
      
        RETURN low  
    END FUNCTION  
END CLASS
