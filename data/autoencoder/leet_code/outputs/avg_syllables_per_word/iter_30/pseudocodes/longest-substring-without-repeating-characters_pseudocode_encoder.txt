CLASS Solution  
    FUNCTION lengthOfLongestSubstring WITH PARAMETER s  
        ASSIGN reference TO empty mapping char_index_map  
        SET max_length TO zero  
        SET start TO zero  
        FOR each pair of end and char enumerated from s  
            IF char is in char_index_map AND value of char at char_index_map GREATER THAN OR EQUAL TO start  
                SET start TO value of char at char_index_map PLUS one  
            END IF  
            SET value of char at char_index_map TO end  
            SET max_length TO the GREATER OF max_length AND end MINUS start PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
