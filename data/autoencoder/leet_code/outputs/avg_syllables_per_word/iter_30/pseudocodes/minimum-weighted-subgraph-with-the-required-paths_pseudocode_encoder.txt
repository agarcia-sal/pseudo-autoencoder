CLASS Solution  
    FUNCTION minimumWeight(n, edges, src1, src2, dest)  
        FUNCTION dijkstra(graph, start)  
            SET distances TO list with n elements each assigned to infinity  
            SET element at position start OF distances TO zero  
            SET heap TO collection containing pair of zero and start  
            
            WHILE heap is not empty  
                SET current_distance and current_node TO removal of smallest element from heap  
                
                IF current_distance GREATER THAN element at position current_node OF distances  
                    CONTINUE  
                END IF  
                
                FOR each neighbor and weight IN element at position current_node OF graph  
                    SET distance TO current_distance PLUS weight  
                    
                    IF distance LESS THAN element at position neighbor OF distances  
                        SET element at position neighbor OF distances TO distance  
                        ADD pair of distance and neighbor TO heap  
                    END IF  
                END FOR  
            END WHILE  
            
            RETURN distances  
        END FUNCTION  
        
        SET graph TO list of empty collections with length n  
        SET reverse_graph TO list of empty collections with length n  
        
        FOR each u, v, w IN edges  
            APPEND pair of v and w TO element at position u OF graph  
            APPEND pair of u and w TO element at position v OF reverse_graph  
        END FOR  
        
        SET dist_from_src1 TO result of dijkstra called with graph and src1  
        SET dist_from_src2 TO result of dijkstra called with graph and src2  
        SET dist_to_dest TO result of dijkstra called with reverse_graph and dest  
        
        SET min_weight TO infinity  
        
        FOR index FROM zero TO n MINUS one  
            IF element at position index OF dist_from_src1 NOT EQUALS infinity AND element at position index OF dist_from_src2 NOT EQUALS infinity AND element at position index OF dist_to_dest NOT EQUALS infinity  
                SET sum_distance TO element at position index OF dist_from_src1 PLUS element at position index OF dist_from_src2 PLUS element at position index OF dist_to_dest  
                IF sum_distance LESS THAN min_weight  
                    SET min_weight TO sum_distance  
                END IF  
            END IF  
        END FOR  
        
        IF min_weight EQUALS infinity  
            RETURN minus one  
        ELSE  
            RETURN min_weight  
        END IF  
    END FUNCTION  
END CLASS
