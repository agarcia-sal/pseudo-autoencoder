CLASS Solution  
    FUNCTION recoverArray(nums)  
        SORT nums  
        SET n TO the LENGTH OF nums  
        SET first TO element at position zero OF nums  
        FOR i FROM one TO n MINUS one  
            SET k TO the INTEGER DIVISION OF the DIFFERENCE BETWEEN element at position i OF nums AND first BY two  
            IF k LESS THAN OR EQUAL TO zero OR the SUM OF first PLUS two MULTIPLIED BY k NOT EQUALS element at position i OF nums  
                CONTINUE  
            END IF  
            SET count TO a new mapping of each element in nums TO its frequency  
            SET arr TO an empty collection  
            FOR each x IN nums  
                IF the frequency of x IN count EQUALS zero  
                    CONTINUE  
                END IF  
                IF the frequency of the SUM OF x PLUS two MULTIPLIED BY k IN count EQUALS zero  
                    BREAK  
                END IF  
                APPEND the SUM OF x PLUS k TO arr  
                DECREMENT the frequency of x IN count BY one  
                DECREMENT the frequency of the SUM OF x PLUS two MULTIPLIED BY k IN count BY one  
            END FOR  
            IF the LENGTH OF arr EQUALS the INTEGER DIVISION OF n BY two  
                RETURN arr  
            END IF  
        END FOR  
        RETURN an empty collection  
    END FUNCTION  
END CLASS
