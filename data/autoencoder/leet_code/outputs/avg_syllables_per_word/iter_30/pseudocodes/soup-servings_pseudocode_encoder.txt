CLASS Solution  
    FUNCTION soupServings(n)  
        IF n GREATER THAN four thousand eight hundred  
            RETURN one  
        END IF  
        SET n TO n PLUS twenty four DIVIDED BY twenty five  
        SET memo TO empty collection  
        
        FUNCTION dp(a, b)  
            IF pair of a and b IS IN memo  
                RETURN value corresponding to pair of a and b IN memo  
            END IF  
            IF a LESS THAN OR EQUAL TO zero AND b LESS THAN OR EQUAL TO zero  
                RETURN point five  
            END IF  
            IF a LESS THAN OR EQUAL TO zero  
                RETURN one  
            END IF  
            IF b LESS THAN OR EQUAL TO zero  
                RETURN zero  
            END IF  
            SET prob TO point two five MULTIPLIED BY  
                sum of dp of a MINUS four and b PLUS dp of a MINUS three and b MINUS one PLUS dp of a MINUS two and b MINUS two PLUS dp of a MINUS one and b MINUS three  
            SET value corresponding to pair of a and b IN memo TO prob  
            RETURN prob  
        END FUNCTION  
        
        RETURN dp of n and n  
    END FUNCTION  
END CLASS
