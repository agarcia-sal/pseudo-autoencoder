CLASS Solution  
    FUNCTION ipToCIDR(ip, n)  
        FUNCTION ip_to_int(ip)  
            SET parts TO list of integer conversions of substrings separated by dot in ip  
            SET res TO zero  
            FOR each part IN parts  
                SET res TO res MULTIPLIED BY two hundred fifty-six PLUS part  
            END FOR  
            RETURN res  
        END FUNCTION  
        
        FUNCTION int_to_ip(num)  
            SET list_of_strings TO empty list  
            FOR i FROM three DOWN TO zero  
                SET segment TO num DIVIDED BY two hundred fifty-six POWERED BY i MODULO two hundred fifty-six  
                APPEND string conversion of segment TO list_of_strings  
            END FOR  
            JOIN elements of list_of_strings with dot  
            RETURN the joined string  
        END FUNCTION  
        
        FUNCTION largest_cidr_size(start, n)  
            IF bitwise AND of start AND one EQUALS one  
                RETURN one  
            END IF  
            SET x TO one  
            WHILE x LESS THAN OR EQUAL TO n AND bitwise AND of start AND x MINUS one EQUALS zero  
                SET x TO x MULTIPLIED BY two  
            END WHILE  
            RETURN x DIVIDED BY two  
        END FUNCTION  
        
        SET result TO empty list  
        SET start_ip TO ip_to_int(ip)  
        
        WHILE n GREATER THAN zero  
            SET size TO largest_cidr_size(start_ip, n)  
            SET suffix_length TO thirty-two MINUS bit length of size PLUS one  
            SET cidr_string TO concatenation of int_to_ip of start_ip, slash, suffix_length  
            APPEND cidr_string TO result  
            SET start_ip TO start_ip PLUS size  
            SET n TO n MINUS size  
        END WHILE  
        
        RETURN result  
    END FUNCTION  
END CLASS
