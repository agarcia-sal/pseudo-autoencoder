CLASS Solution  
    FUNCTION subStrHash(s, power, modulo, k, hashValue)  
        FUNCTION val(c)  
            RETURN the Unicode code point of c MINUS the Unicode code point of the character a PLUS one  
        END FUNCTION  
        
        SET n TO the LENGTH OF s  
        SET current_hash TO zero  
        SET p_pow TO one  
        
        FOR i FROM n MINUS k TO n MINUS one  
            SET current_hash TO current_hash PLUS val of character at position i OF s MULTIPLIED BY p_pow MODULO modulo  
            IF i LESS THAN n MINUS one  
                SET p_pow TO p_pow MULTIPLIED BY power MODULO modulo  
            END IF  
        END FOR  
        
        SET start_index TO n MINUS k  
        
        IF current_hash EQUALS hashValue  
            RETURN substring from position start_index TO position n OF s  
        END IF  
        
        SET p_pow_k_minus_1 TO p_pow  
        
        FOR i FROM n MINUS k MINUS one DOWNTO zero  
            SET current_hash TO current_hash MINUS val of character at position i PLUS k OF s MULTIPLIED BY p_pow_k_minus_1 MODULO modulo  
            SET current_hash TO current_hash MULTIPLIED BY power PLUS val of character at position i OF s MODULO modulo  
            
            IF current_hash EQUALS hashValue  
                SET start_index TO i  
            END IF  
        END FOR  
        
        RETURN substring from position start_index TO position start_index PLUS k OF s  
    END FUNCTION  
END CLASS
