CLASS Solution  
    FUNCTION searchMatrix(matrix, target)  
        IF matrix EQUALS empty OR element at position zero OF matrix EQUALS empty  
            RETURN false  
        END IF  
        SET m TO the LENGTH OF matrix  
        SET n TO the LENGTH OF element at position zero OF matrix  
        SET left TO zero  
        SET right TO m MULTIPLIED BY n MINUS one  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the INTEGER DIVISION OF the SUM of left PLUS right BY two  
            SET mid_value TO element at position the INTEGER DIVISION OF mid BY n OF matrix AT position mid MODULO n  
            IF mid_value EQUALS target  
                RETURN true  
            ELSE IF mid_value LESS THAN target  
                SET left TO mid PLUS one  
            ELSE  
                SET right TO mid MINUS one  
            END IF  
        END WHILE  
        RETURN false  
    END FUNCTION  
END CLASS
