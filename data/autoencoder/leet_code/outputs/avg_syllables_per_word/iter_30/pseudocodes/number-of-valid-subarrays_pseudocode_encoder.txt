CLASS Solution  
    FUNCTION validSubarrays(nums)  
        ASSIGN reference TO an empty list named stack  
        SET count TO zero  
        FOR each num IN nums  
            WHILE the LENGTH OF stack is greater than zero AND element at position the LENGTH OF stack MINUS one OF stack is GREATER THAN num  
                REMOVE element at position the LENGTH OF stack MINUS one FROM stack  
            END WHILE  
            APPEND num TO stack  
            INCREMENT count BY the LENGTH OF stack  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
