CLASS Solution  
    FUNCTION connectTwoGroups(cost)  
        SET size1 TO the LENGTH OF cost  
        SET size2 TO the LENGTH OF element at position zero OF cost  
        SET dp TO a two dimensional collection with size1 plus one rows and two TO the power of size2 columns, each element initialized to infinity  
        SET element at position zero of dp and position zero TO zero  
        FOR i FROM zero TO size1 MINUS one  
            FOR mask FROM zero TO two TO the power of size2 MINUS one  
                FOR j FROM zero TO size2 MINUS one  
                    SET new_mask TO mask OR two TO the power of j  
                    SET element at position i plus one of dp and position new_mask TO the minimum OF element at position i plus one of dp and position new_mask AND element at position i of dp and position mask PLUS element at position i and position j OF cost  
                END FOR  
                FOR j FROM zero TO size2 MINUS one  
                    IF bitwise AND of mask AND two TO the power of j holds  
                        SET element at position i plus one of dp and position mask TO the minimum OF element at position i plus one of dp and position mask AND element at position i plus one of dp and position bitwise XOR of mask AND two TO the power of j PLUS element at position i and position j OF cost  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN element at position size1 of dp and position two TO the power of size2 MINUS one  
    END FUNCTION  
END CLASS
