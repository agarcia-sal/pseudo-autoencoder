CLASS Solution  
    FUNCTION shortestSubarray(nums, k)  
        SET prefix_sum TO a list containing zero  
        FOR each num IN nums  
            APPEND the last element of prefix_sum PLUS num TO prefix_sum  
        END FOR  
        ASSIGN reference TO an empty double ended queue named dq  
        SET min_length TO positive infinity  
        FOR each index i and current_sum at position i in prefix_sum  
            WHILE dq is not empty AND current_sum MINUS prefix_sum at position of the first element in dq is GREATER THAN OR EQUAL TO k  
                SET min_length TO the smaller value between min_length AND i MINUS the first element removed from dq  
            END WHILE  
            WHILE dq is not empty AND current_sum is LESS THAN OR EQUAL TO prefix_sum at position of the last element in dq  
                REMOVE the last element from dq  
            END WHILE  
            ADD i TO the end of dq  
        END FOR  
        IF min_length is NOT EQUAL TO positive infinity  
            RETURN min_length  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
