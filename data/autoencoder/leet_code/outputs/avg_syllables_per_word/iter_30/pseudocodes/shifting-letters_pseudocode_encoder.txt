CLASS Solution  
    FUNCTION shiftingLetters WITH PARAMETERS s, shifts  
        SET total_shifts TO zero  
        FOR index FROM the LENGTH OF shifts MINUS one TO zero STEP minus one  
            INCREMENT total_shifts BY element at position index OF shifts  
            SET element at position index OF shifts TO total_shifts  
        END FOR  
        SET result TO empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character_position TO the numerical position of element at position index OF s in alphabet starting at 'a'  
            SET shifted_position TO character_position PLUS element at position index OF shifts  
            SET wrapped_position TO shifted_position MODULO twenty six  
            SET new_character TO character represented by wrapped_position plus the numerical position of 'a'  
            APPEND new_character TO result  
        END FOR  
        RETURN concatenation of all elements in result  
    END FUNCTION  
END CLASS
