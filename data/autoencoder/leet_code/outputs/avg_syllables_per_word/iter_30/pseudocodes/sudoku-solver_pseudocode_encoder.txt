CLASS Solution  
    FUNCTION solveSudoku(board)  
        FUNCTION is_valid(board, row, col, num)  
            FOR i FROM zero TO eight  
                IF element at position col OF element at position row OF board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            FOR i FROM zero TO eight  
                IF element at position col OF element at position i OF board EQUALS num  
                    RETURN False  
                END IF  
            END FOR  
            SET start_row TO three MULTIPLIED BY the INTEGER DIVISION OF row BY three  
            SET start_col TO three MULTIPLIED BY the INTEGER DIVISION OF col BY three  
            FOR i FROM zero TO two  
                FOR j FROM zero TO two  
                    IF element at position start_col PLUS j OF element at position start_row PLUS i OF board EQUALS num  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
  
        FUNCTION solve(board)  
            FOR row FROM zero TO eight  
                FOR col FROM zero TO eight  
                    IF element at position col OF element at position row OF board EQUALS the single period character  
                        FOR num FROM one TO nine  
                            CONVERT num TO string TO OBTAIN num_string  
                            IF is_valid(board, row, col, num_string)  
                                SET element at position col OF element at position row OF board TO num_string  
                                IF solve(board)  
                                    RETURN True  
                                END IF  
                                SET element at position col OF element at position row OF board TO the single period character  
                            END IF  
                        END FOR  
                        RETURN False  
                    END IF  
                END FOR  
            END FOR  
            RETURN True  
        END FUNCTION  
  
        CALL solve(board)  
    END FUNCTION  
END CLASS
