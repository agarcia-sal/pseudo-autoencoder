CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET head TO new ListNode with value as element at position zero OF values  
    SET p TO head  
    FOR each val IN elements of values FROM position one TO the end  
        SET node TO new ListNode with value as val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 EQUALS none AND p2 EQUALS none  
        RETURN true  
    END IF  
    IF p1 EQUALS none OR p2 EQUALS none  
        RETURN false  
    END IF  
    RETURN the condition that value at p1 EQUALS value at p2 AND the result of is_same_list with parameters as next of p1 and next of p2  
END FUNCTION  

CLASS Solution  
    FUNCTION addTwoNumbers(self, l1, l2)  
        SET dummy_head TO new ListNode with value zero  
        SET current TO dummy_head  
        SET carry TO zero  
        WHILE l1 IS NOT none OR l2 IS NOT none OR carry IS GREATER THAN zero  
            IF l1 IS NOT none  
                SET val1 TO value at l1  
            ELSE  
                SET val1 TO zero  
            END IF  
            IF l2 IS NOT none  
                SET val2 TO value at l2  
            ELSE  
                SET val2 TO zero  
            END IF  
            SET total TO val1 PLUS val2 PLUS carry  
            SET carry TO total DIVIDED BY ten  
            SET current.next TO new ListNode with value total MINUS carry MULTIPLIED BY ten  
            SET current TO current.next  
            IF l1 IS NOT none  
                SET l1 TO next of l1  
            END IF  
            IF l2 IS NOT none  
                SET l2 TO next of l2  
            END IF  
        END WHILE  
        RETURN next of dummy_head  
    END FUNCTION  
END CLASS
