CLASS Solution  
    FUNCTION smallestRange(nums)  
        SET min_heap TO empty collection  
        SET max_value TO negative infinity  
        FOR index FROM zero TO the LENGTH OF nums MINUS one  
            PUSH a tuple containing element at position zero OF nums at position index, index, and zero INTO min_heap  
            IF max_value LESS THAN element at position zero OF nums at position index  
                SET max_value TO element at position zero OF nums at position index  
            END IF  
        END FOR  
        SET smallest_range TO positive infinity  
        SET result TO a collection containing negative one hundred thousand and one hundred thousand  
        WHILE min_heap is not empty  
            POP the minimum tuple from min_heap INTO min_value, list_index, element_index  
            IF max_value MINUS min_value LESS THAN smallest_range  
                SET smallest_range TO max_value MINUS min_value  
                SET result TO a collection containing min_value and max_value  
            END IF  
            IF element_index PLUS one LESS THAN the LENGTH OF nums at position list_index  
                SET next_value TO element at position element_index PLUS one OF nums at position list_index  
                PUSH a tuple containing next_value, list_index, and element_index PLUS one INTO min_heap  
                IF max_value LESS THAN next_value  
                    SET max_value TO next_value  
                END IF  
            ELSE  
                BREAK  
            END IF  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
