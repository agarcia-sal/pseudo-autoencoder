CLASS Solution  
    FUNCTION makeLargestSpecial WITH PARAMETER s  
        FUNCTION dfs WITH PARAMETER s  
            SET count TO zero  
            SET start TO zero  
            SET specials TO empty list  
            FOR each index i AND character char IN s  
                IF char EQUALS character one  
                    INCREMENT count BY one  
                ELSE  
                    DECREMENT count BY one  
                END IF  
                IF count EQUALS zero  
                    APPEND character one PLUS dfs OF substring from position start PLUS one TO position i MINUS one OF s PLUS character zero TO specials  
                    SET start TO i PLUS one  
                END IF  
            END FOR  
            SORT specials IN descending lexicographical order  
            RETURN concatenation of elements IN specials  
        END FUNCTION  
        RETURN dfs OF s  
    END FUNCTION  
END CLASS
