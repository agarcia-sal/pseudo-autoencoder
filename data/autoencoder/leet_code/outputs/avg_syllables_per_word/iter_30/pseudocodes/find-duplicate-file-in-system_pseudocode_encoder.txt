CLASS Solution  
    FUNCTION findDuplicate(paths)  
        ASSIGN reference TO a new default dictionary with default value as empty list named content_to_paths  
        FOR each path IN paths  
            SET parts TO the list of substrings of path separated by spaces  
            SET directory TO element at position zero OF parts  
            FOR each file_info IN the slice of parts from position one TO the end  
                SET file_name TO substring of file_info from the start TO the position before the first opening parenthesis  
                SET content TO substring of file_info from the position after the first opening parenthesis TO the position before the last character  
                APPEND the concatenation of directory PLUS the forward slash PLUS file_name TO content_to_paths at key content  
            END FOR  
        END FOR  
        RETURN the list of all collections of content_to_paths values where the length of each collection is greater than one  
    END FUNCTION  
END CLASS
