CLASS Solution  
    FUNCTION amountPainted(paint)  
        SET max_end TO the maximum value among all second elements of each pair in paint  
        SET painted TO a list of length max_end plus one filled with negative one  
        SET worklog TO an empty list  
        FOR each pair start and end IN paint  
            SET current_start TO start  
            SET new_paint TO zero  
            WHILE current_start LESS THAN end  
                IF element at position current_start OF painted EQUALS negative one  
                    SET element at position current_start OF painted TO end  
                    INCREMENT new_paint BY one  
                    INCREMENT current_start BY one  
                ELSE  
                    SET current_start TO element at position current_start OF painted  
                END IF  
            END WHILE  
            APPEND new_paint TO worklog  
        END FOR  
        RETURN worklog  
    END FUNCTION  
END CLASS
