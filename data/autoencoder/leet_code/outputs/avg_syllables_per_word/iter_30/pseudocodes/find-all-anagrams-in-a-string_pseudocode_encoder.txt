CLASS Solution  
    FUNCTION findAnagrams(s, p)  
        SET len_s TO the LENGTH OF s  
        SET len_p TO the LENGTH OF p  
        IF len_p GREATER THAN len_s  
            RETURN an empty list  
        END IF  
        SET p_count TO the frequency count of each character in p  
        SET s_count TO the frequency count of each character in the substring from position zero TO position len_p MINUS one OF s  
        SET anagrams TO an empty list  
        IF s_count EQUALS p_count  
            APPEND zero TO anagrams  
        END IF  
        FOR i FROM len_p TO len_s MINUS one  
            INCREMENT the count of character at position i OF s IN s_count BY one  
            DECREMENT the count of character at position i MINUS len_p OF s IN s_count BY one  
            IF the count of character at position i MINUS len_p OF s IN s_count EQUALS zero  
                REMOVE the character at position i MINUS len_p OF s FROM s_count  
            END IF  
            IF s_count EQUALS p_count  
                APPEND i MINUS len_p PLUS one TO anagrams  
            END IF  
        END FOR  
        RETURN anagrams  
    END FUNCTION  
END CLASS
