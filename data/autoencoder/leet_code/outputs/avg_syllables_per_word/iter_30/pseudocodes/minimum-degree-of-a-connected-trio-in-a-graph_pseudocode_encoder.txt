CLASS Solution  
    FUNCTION minTrioDegree(n, edges)  
        CREATE graph AS a MAPPING FROM each number FROM one TO n TO an EMPTY SET  
        FOR each pair u AND v IN edges  
            ADD v TO the SET at key u IN graph  
            ADD u TO the SET at key v IN graph  
        END FOR  
        SET min_degree TO a very large number representing infinity  
        SET trio_found TO false  
        FOR i FROM one TO n  
            FOR j FROM i PLUS one TO n  
                FOR k FROM j PLUS one TO n  
                    IF j IS IN the SET at key i IN graph AND k IS IN the SET at key i IN graph AND k IS IN the SET at key j IN graph  
                        SET trio_found TO true  
                        SET trio_degree TO the SIZE OF the SET at key i IN graph PLUS the SIZE OF the SET at key j IN graph PLUS the SIZE OF the SET at key k IN graph MINUS six  
                        IF trio_degree LESS THAN min_degree  
                            SET min_degree TO trio_degree  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        IF trio_found  
            RETURN min_degree  
        ELSE  
            RETURN negative one  
        END IF  
    END FUNCTION  
END CLASS
