CLASS Solution  
    FUNCTION rob(nums)  
        FUNCTION rob_linear(houses)  
            IF houses EQUALS empty collection  
                RETURN zero  
            END IF  
            IF the LENGTH OF houses EQUALS one  
                RETURN element at position zero OF houses  
            END IF  
            SET dp TO collection of zeros with the LENGTH OF houses  
            SET element at position zero OF dp TO element at position zero OF houses  
            SET element at position one OF dp TO the GREATER VALUE BETWEEN element at position zero OF houses AND element at position one OF houses  
            FOR index FROM two TO the LENGTH OF houses MINUS one  
                SET element at position index OF dp TO the GREATER VALUE BETWEEN element at position index MINUS one OF dp AND element at position index MINUS two OF dp PLUS element at position index OF houses  
            END FOR  
            RETURN element at position the LENGTH OF dp MINUS one OF dp  
        END FUNCTION  
        IF nums EQUALS empty collection  
            RETURN zero  
        END IF  
        IF the LENGTH OF nums EQUALS one  
            RETURN element at position zero OF nums  
        END IF  
        RETURN the GREATER VALUE BETWEEN rob_linear(subcollection of nums FROM position zero TO position the LENGTH OF nums MINUS two) AND rob_linear(subcollection of nums FROM position one TO position the LENGTH OF nums MINUS one)  
    END FUNCTION  
END CLASS
