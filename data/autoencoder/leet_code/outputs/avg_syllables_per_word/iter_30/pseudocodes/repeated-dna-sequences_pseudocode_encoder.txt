CLASS Solution  
    FUNCTION findRepeatedDnaSequences WITH PARAMETER s  
        ASSIGN reference TO a new mapping with default integer zero named sequence_count  
        FOR index FROM zero TO the LENGTH OF s MINUS ten PLUS one  
            SET sequence TO substring from position index TO position index PLUS ten MINUS one OF s  
            INCREMENT the value associated to sequence IN sequence_count BY one  
        END FOR  
        SET repeated_sequences TO empty list  
        FOR each sequence AND count IN sequence_count items  
            IF count GREATER THAN one  
                APPEND sequence TO repeated_sequences  
            END IF  
        END FOR  
        RETURN repeated_sequences  
    END FUNCTION  
END CLASS
