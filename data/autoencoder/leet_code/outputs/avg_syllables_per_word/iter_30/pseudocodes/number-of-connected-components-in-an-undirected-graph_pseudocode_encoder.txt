CLASS Solution  
    FUNCTION countComponents(n, edges)  
        SET graph TO empty mapping from zero TO n MINUS one with each value as empty list  
        FOR each pair a, b IN edges  
            APPEND b TO list at key a IN graph  
            APPEND a TO list at key b IN graph  
        END FOR  
        SET visited TO empty set  
        FUNCTION dfs(node)  
            IF node IN visited THEN  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN list at key node IN graph  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        SET components TO zero  
        FOR node FROM zero TO n MINUS one  
            IF node NOT IN visited THEN  
                CALL dfs WITH node  
                INCREMENT components BY one  
            END IF  
        END FOR  
        RETURN components  
    END FUNCTION  
END CLASS
