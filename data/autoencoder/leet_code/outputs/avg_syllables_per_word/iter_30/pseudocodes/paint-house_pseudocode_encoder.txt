CLASS Solution  
    FUNCTION minCost(costs)  
        IF costs EQUALS empty  
            RETURN zero  
        END IF  
        SET prev_red TO element at position zero at position zero OF costs  
        SET prev_blue TO element at position one at position zero OF costs  
        SET prev_green TO element at position two at position zero OF costs  
        FOR index FROM one TO LENGTH OF costs MINUS one  
            SET current_red TO element at position zero at position index OF costs PLUS the smaller value BETWEEN prev_blue AND prev_green  
            SET current_blue TO element at position one at position index OF costs PLUS the smaller value BETWEEN prev_red AND prev_green  
            SET current_green TO element at position two at position index OF costs PLUS the smaller value BETWEEN prev_red AND prev_blue  
            SET prev_red TO current_red  
            SET prev_blue TO current_blue  
            SET prev_green TO current_green  
        END FOR  
        RETURN the smaller value BETWEEN prev_red AND prev_blue AND prev_green  
    END FUNCTION  
END CLASS
