CLASS Solution  
    FUNCTION minimumTotal(triangle)  
        FOR row FROM the LENGTH OF triangle MINUS two TO zero STEPPING BY MINUS one  
            FOR col FROM zero TO the LENGTH OF element at position row OF triangle MINUS one  
                SET element at position row and position col OF triangle TO element at position row and position col OF triangle PLUS the MINIMUM OF element at position row plus one and position col OF triangle AND element at position row plus one and position col plus one OF triangle  
            END FOR  
        END FOR  
        RETURN element at position zero and position zero OF triangle  
    END FUNCTION  
END CLASS
