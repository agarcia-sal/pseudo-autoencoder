CLASS Solution  
    FUNCTION scheduleCourse(courses)  
        SORT courses BY the second element of each course IN ascending order  
        SET max_heap TO empty collection  
        SET current_end_day TO zero  
        FOR each duration and last_day IN courses  
            IF current_end_day PLUS duration LESS THAN OR EQUAL TO last_day  
                ADD the negation of duration TO max_heap USING a heap push operation  
                INCREMENT current_end_day BY duration  
            ELSE IF max_heap IS NOT empty AND the negation of the first element of max_heap GREATER THAN duration  
                INCREMENT current_end_day BY duration PLUS the removal of the first element from max_heap using a heap pop operation  
                ADD the negation of duration TO max_heap USING a heap push operation  
            END IF  
        END FOR  
        RETURN the length of max_heap  
    END FUNCTION  
END CLASS
