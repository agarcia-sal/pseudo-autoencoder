CLASS Solution  
    FUNCTION change(amount, coins)  
        SET dp TO a list of zeros of length amount PLUS one  
        SET element at position zero OF dp TO one  
        FOR each coin IN coins  
            FOR each x FROM coin TO amount  
                SET element at position x OF dp TO element at position x OF dp PLUS element at position x MINUS coin OF dp  
            END FOR  
        END FOR  
        RETURN element at position amount OF dp  
    END FUNCTION  
END CLASS
