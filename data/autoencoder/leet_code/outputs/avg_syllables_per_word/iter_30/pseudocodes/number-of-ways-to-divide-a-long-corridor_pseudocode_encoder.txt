CLASS Solution  
    FUNCTION numberOfWays(corridor)  
        SET MODULO TO one billion PLUS seven  
        SET seats TO empty list  
        FOR index FROM zero TO the LENGTH OF corridor MINUS one  
            IF character at position index OF corridor EQUALS uppercase letter S  
                APPEND index TO seats  
            END IF  
        END FOR  
        IF the LENGTH OF seats MODULO two NOT EQUALS zero OR the LENGTH OF seats LESS THAN two  
            RETURN zero  
        END IF  
        SET ways TO one  
        FOR index FROM two TO the LENGTH OF seats MINUS one STEP two  
            SET difference TO element at position index OF seats MINUS element at position index MINUS one OF seats  
            SET ways TO ways MULTIPLIED BY difference  
            SET ways TO ways MODULO MODULO  
        END FOR  
        RETURN ways  
    END FUNCTION  
END CLASS
