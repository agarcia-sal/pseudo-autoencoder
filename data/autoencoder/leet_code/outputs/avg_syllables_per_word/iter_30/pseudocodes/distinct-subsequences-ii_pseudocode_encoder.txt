CLASS Solution  
    FUNCTION distinctSubseqII(s)  
        SET MOD TO ten raised to the power of nine PLUS one  
        SET n TO the LENGTH OF s  
        SET dp TO a list of zero repeated n PLUS one times  
        SET zero position of dp TO one  
        SET last TO an empty mapping  
        FOR index FROM one TO n  
            SET char TO element at position index MINUS one OF s  
            SET dp at position index TO dp at position index MINUS one MULTIPLIED BY two MODULO MOD  
            IF char IS a key in last  
                SET dp at position index TO dp at position index MINUS dp at position last at char MINUS one PLUS MOD MODULO MOD  
            END IF  
            ASSIGN last at char TO index  
        END FOR  
        RETURN dp at position n MINUS one MODULO MOD  
    END FUNCTION  
END CLASS
