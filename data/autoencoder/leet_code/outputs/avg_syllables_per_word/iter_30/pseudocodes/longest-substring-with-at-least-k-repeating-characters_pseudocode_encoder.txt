CLASS Solution  
    FUNCTION longestSubstring WITH PARAMETERS s AND k  
        IF length of s LESS THAN k  
            RETURN zero  
        END IF  
        CALL count WITH PARAMETERS s AND ASSIGN TO count  
        FOR each char IN count  
            IF count of char LESS THAN k  
                RETURN maximum value AMONG FUNCTION longestSubstring CALLED WITH PARAMETERS each substring OBTAINED BY splitting s BY char AND k  
            END IF  
        END FOR  
        RETURN length of s  
    END FUNCTION  
END CLASS
