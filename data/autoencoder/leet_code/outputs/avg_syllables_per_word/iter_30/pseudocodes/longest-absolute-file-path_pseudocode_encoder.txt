CLASS Solution  
    FUNCTION lengthLongestPath(input)  
        SET lines TO the result of splitting input by newline  
        SET stack TO empty list  
        SET max_length TO zero  
        FOR each line IN lines  
            SET depth TO the count of tab characters in line  
            SET name TO substring of line from position depth TO end  
            WHILE the LENGTH OF stack is greater than depth  
                REMOVE the last element FROM stack  
            END WHILE  
            IF LENGTH OF stack is greater than zero  
                SET current_length TO the last element of stack PLUS the LENGTH OF name PLUS one  
            ELSE  
                SET current_length TO the LENGTH OF name  
            END IF  
            IF name contains a period character  
                IF current_length is greater than max_length  
                    SET max_length TO current_length  
                END IF  
            ELSE  
                APPEND current_length TO stack  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
