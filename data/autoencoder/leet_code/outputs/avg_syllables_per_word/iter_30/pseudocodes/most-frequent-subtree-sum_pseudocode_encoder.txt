CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO a new TreeNode initialized with element at position zero OF values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the element obtained by removing from the left of queue  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO a new TreeNode initialized with element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO a new TreeNode initialized with element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF the value of p IS NOT EQUALS the value of q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(the left child of p, the left child of q) AND is_same_tree(the right child of p, the right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  

    FUNCTION findFrequentTreeSum(self, root)  
        FUNCTION subtree_sum(node)  
            IF node IS none  
                RETURN zero  
            END IF  
            SET current_sum TO the value of node PLUS subtree_sum(the left child of node) PLUS subtree_sum(the right child of node)  
            INCREMENT count AT key current_sum BY one  
            RETURN current_sum  
        END FUNCTION  

        SET count TO a new default dictionary mapping to integer zero  
        CALL subtree_sum WITH root  
        SET max_freq TO the maximum VALUE IN count OR zero if empty  
        SET result TO an empty list  
        FOR each key AND freq IN count  
            IF freq EQUALS max_freq  
                APPEND key TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  

END CLASS
