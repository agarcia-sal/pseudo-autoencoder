CLASS Solution  
    FUNCTION maxSubArrayLen(nums, k)  
        SET cumulative_sum_index TO dictionary WITH zero MAPPED TO negative one  
        SET cumulative_sum TO zero  
        SET max_length TO zero  
        FOR each index i AND element num IN nums  
            SET cumulative_sum TO cumulative_sum PLUS num  
            IF cumulative_sum MINUS k EXISTS AS KEY IN cumulative_sum_index  
                SET candidate_length TO i MINUS element at key cumulative_sum MINUS k OF cumulative_sum_index  
                IF candidate_length GREATER THAN max_length  
                    SET max_length TO candidate_length  
                END IF  
            END IF  
            IF cumulative_sum NOT EXISTS AS KEY IN cumulative_sum_index  
                SET element at key cumulative_sum OF cumulative_sum_index TO i  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
