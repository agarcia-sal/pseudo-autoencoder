CLASS Solution  
    FUNCTION canWin(currentState)  
        SET memo TO empty mapping  
        
        FUNCTION can_win_helper(state)  
            IF state EXISTS IN memo THEN  
                RETURN value mapped from state IN memo  
            END IF  
            
            FOR index FROM zero TO length of state MINUS one MINUS one  
                IF substring from position index TO position index PLUS one OF state EQUALS two plus signs THEN  
                    SET new_state TO concatenation of substring from position zero TO position index OF state PLUS two minus signs PLUS substring from position index PLUS two TO end OF state  
                    IF NOT can_win_helper(new_state) THEN  
                        SET value mapped from state IN memo TO truth value of true  
                        RETURN truth value of true  
                    END IF  
                END IF  
            END FOR  
            
            SET value mapped from state IN memo TO truth value of false  
            RETURN truth value of false  
        END FUNCTION  
        
        RETURN can_win_helper(currentState)  
    END FUNCTION  
END CLASS
