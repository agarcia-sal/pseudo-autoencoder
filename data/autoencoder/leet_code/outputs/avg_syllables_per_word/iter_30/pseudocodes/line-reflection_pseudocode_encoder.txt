CLASS Solution  
    FUNCTION isReflected(points)  
        SET unique_points TO the set of all elements obtained by converting each element of points into a tuple  
        SET min_x TO the minimum value of the element at position zero OF each element IN unique_points  
        SET max_x TO the maximum value of the element at position zero OF each element IN unique_points  
        SET line_of_reflection TO the sum of min_x PLUS max_x DIVIDED BY two  
        FOR each element IN unique_points  
            SET x TO the element at position zero OF element  
            SET y TO the element at position one OF element  
            SET reflected_x TO two MULTIPLIED BY line_of_reflection MINUS x  
            IF the tuple of reflected_x and y NOT IN unique_points  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
