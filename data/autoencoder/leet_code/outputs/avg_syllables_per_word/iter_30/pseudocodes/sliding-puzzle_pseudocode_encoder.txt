CLASS Solution  
    FUNCTION slidingPuzzle(board)  
        SET start TO tuple of tuples of each row IN board  
        SET target TO tuple of tuple consisting of one two three and four five zero  
        SET neighbors TO mapping from zero to a set of one and three  
            mapping from one to a set of zero two and four  
            mapping from two to a set of one and five  
            mapping from three to a set of zero and four  
            mapping from four to a set of one three and five  
            mapping from five to a set of two and four  
        FOR i FROM zero TO one  
            FOR j FROM zero TO two  
                IF element at position j OF row at position i OF start EQUALS zero  
                    SET zero_pos TO i MULTIPLIED BY three PLUS j  
                END IF  
            END FOR  
        END FOR  
        SET queue TO a new double ended queue containing a tuple of start zero_pos zero  
        SET visited TO a set containing start  
        WHILE queue IS NOT empty  
            REMOVE the first element from queue and ASSIGN to state zero_pos moves  
            IF state EQUALS target  
                RETURN moves  
            END IF  
            FOR each neighbor IN neighbors at position zero_pos  
                SET new_state TO list of lists converted from state  
                SET i TO zero_pos DIVIDED BY three  
                SET j TO zero_pos MODULO three  
                SET ni TO neighbor DIVIDED BY three  
                SET nj TO neighbor MODULO three  
                SWAP element at position j OF row at position i OF new_state AND element at position nj OF row at position ni OF new_state  
                SET new_tuple_state TO tuple of tuples converted from new_state  
                IF new_tuple_state NOT IN visited  
                    ADD new_tuple_state TO visited  
                    APPEND to queue the tuple of new_tuple_state neighbor moves PLUS one  
                END IF  
            END FOR  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
