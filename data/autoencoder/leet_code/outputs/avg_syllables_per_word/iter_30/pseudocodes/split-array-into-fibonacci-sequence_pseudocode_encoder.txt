CLASS Solution  
    FUNCTION splitIntoFibonacci(num)  
        FUNCTION backtrack(start, path)  
            IF start EQUALS the LENGTH OF num AND the LENGTH OF path GREATER THAN OR EQUAL TO three  
                RETURN path  
            END IF  
            FOR end FROM start PLUS one TO the LENGTH OF num PLUS one  
                IF character at position start OF num EQUALS zero AND end GREATER THAN start PLUS one  
                    CONTINUE  
                END IF  
                SET next_num TO integer value of substring from position start TO position end OF num  
                IF next_num GREATER THAN OR EQUAL TO two TO THE POWER OF thirty ONE  
                    BREAK  
                END IF  
                IF the LENGTH OF path LESS THAN two OR next_num EQUALS element at position the LENGTH OF path MINUS one OF path PLUS element at position the LENGTH OF path MINUS two OF path  
                    SET result TO backtrack(end, path PLUS list containing next_num)  
                    IF result IS NOT empty  
                        RETURN result  
                    END IF  
                END IF  
            END FOR  
            RETURN empty list  
        END FUNCTION  
        RETURN backtrack(zero, empty list)  
    END FUNCTION  
END CLASS
