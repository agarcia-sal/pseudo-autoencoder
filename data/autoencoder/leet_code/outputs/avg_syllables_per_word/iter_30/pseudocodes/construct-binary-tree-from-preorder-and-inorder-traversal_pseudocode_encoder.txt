CLASS TreeNode  
    FUNCTION __init__(self, val WITH DEFAULT zero, left WITH DEFAULT none, right WITH DEFAULT none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at position zero OF values  
    SET i TO one  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED FROM queue  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION buildTree(self, preorder, inorder)  
        IF preorder IS empty OR inorder IS empty  
            RETURN none  
        END IF  
        SET root_val TO element at position zero OF preorder  
        SET root TO new TreeNode with root_val  
        SET root_index_inorder TO the position OF root_val IN inorder  
        SET root.left TO self.buildTree(sublist of preorder from position one TO position root_index_inorder PLUS one, sublist of inorder from position zero TO position root_index_inorder)  
        SET root.right TO self.buildTree(sublist of preorder from position root_index_inorder PLUS one TO end, sublist of inorder from position root_index_inorder PLUS one TO end)  
        RETURN root  
    END FUNCTION  
END CLASS
