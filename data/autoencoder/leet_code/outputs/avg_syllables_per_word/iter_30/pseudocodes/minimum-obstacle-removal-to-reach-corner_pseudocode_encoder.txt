CLASS Solution  
    FUNCTION minimumObstacles(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero OF grid  
        SET directions TO a collection of elements each being a pair consisting of zero and one, one and zero, zero and negative one, negative one and zero  
        SET min_heap TO a collection containing a triple of zero, zero, zero  
        SET visited TO a set containing the pair of zero and zero  
        
        WHILE min_heap holds elements  
            REMOVE the element with the smallest first component FROM min_heap AND ASSIGN the first, second, and third components TO cost, x, y respectively  
            
            IF the pair of x and y EQUALS the pair of m MINUS one and n MINUS one  
                RETURN cost  
            END IF  
            
            FOR each pair dx and dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx IS GREATER THAN OR EQUAL TO zero AND nx IS LESS THAN m AND ny IS GREATER THAN OR EQUAL TO zero AND ny IS LESS THAN n AND the pair of nx and ny IS NOT IN visited  
                    ADD the pair of nx and ny TO visited  
                    SET new_cost TO cost PLUS the element at position nx and ny OF grid  
                    INSERT the triple of new_cost, nx, ny INTO min_heap maintaining heap order by the first element  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN negative one  
    END FUNCTION  
END CLASS
