CLASS Solution  
    FUNCTION allPathsSourceTarget(graph)  
        SET n TO the LENGTH OF graph  
        ASSIGN reference TO a new empty double ended queue containing a list with zero as its element  
        ASSIGN reference TO an empty list named ans  
        WHILE the queue is not empty  
            REMOVE and ASSIGN reference TO path from the left end of the queue  
            ASSIGN reference TO the element at the last position of path named u  
            IF u EQUALS n MINUS one  
                APPEND path TO ans  
                CONTINUE  
            END IF  
            FOR each element v IN the element at position u of graph  
                APPEND the concatenation of path and list containing v TO the queue  
            END FOR  
        END WHILE  
        RETURN ans  
    END FUNCTION  
END CLASS
