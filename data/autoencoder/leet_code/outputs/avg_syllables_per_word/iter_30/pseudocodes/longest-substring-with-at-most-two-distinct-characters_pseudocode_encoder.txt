CLASS Solution  
    FUNCTION lengthOfLongestSubstringTwoDistinct(s)  
        ASSIGN reference TO a new default dictionary with default integer value named char_count  
        SET left TO zero  
        SET max_length TO zero  
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT the count at character at position right OF s IN char_count BY one  
            WHILE the NUMBER OF KEYS IN char_count IS GREATER THAN two  
                DECREMENT the count at character at position left OF s IN char_count BY one  
                IF the count at character at position left OF s IN char_count EQUALS zero  
                    REMOVE character at position left OF s FROM char_count  
                END IF  
                INCREMENT left BY one  
            END WHILE  
            SET max_length TO the GREATER OF max_length AND right MINUS left PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
