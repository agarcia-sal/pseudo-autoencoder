CLASS Solution  
    FUNCTION colorTheGrid(m, n)  
        SET MOD TO ten raised to the power of nine PLUS one  
        
        FUNCTION generate_valid_columns(prev_col)  
            SET colors TO list containing one two and three  
            SET valid_cols TO empty list  
            
            FUNCTION backtrack(col, row)  
                IF row EQUALS m THEN  
                    APPEND col converted to immutable sequence TO valid_cols  
                    RETURN  
                END IF  
                FOR each color IN colors  
                    IF col is empty OR last element of col NOT EQUALS color THEN  
                        IF prev_col is not null AND element at position row of prev_col EQUALS color THEN  
                            CONTINUE  
                        END IF  
                        APPEND color TO col  
                        CALL backtrack with col and row PLUS one  
                        REMOVE last element FROM col  
                    END IF  
                END FOR  
            END FUNCTION  
            
            CALL backtrack with empty list and zero  
            RETURN valid_cols  
        END FUNCTION  
        
        SET all_valid_cols TO call generate_valid_columns with null  
        SET num_valid_cols TO length of all_valid_cols  
        
        SET compatible_cols TO dictionary mapping each integer from zero TO num_valid_cols MINUS one TO empty list  
        FOR i FROM zero TO num_valid_cols MINUS one  
            FOR j FROM zero TO num_valid_cols MINUS one  
                IF for all k FROM zero TO m MINUS one the element at position k of all_valid_cols at i NOT EQUALS element at position k of all_valid_cols at j THEN  
                    APPEND j TO the list at key i in compatible_cols  
                END IF  
            END FOR  
        END FOR  
        
        SET dp TO list of ones with length num_valid_cols  
        
        FOR column_index FROM one TO n MINUS one  
            SET next_dp TO list of zeros with length num_valid_cols  
            FOR i FROM zero TO num_valid_cols MINUS one  
                FOR each j IN compatible_cols at i  
                    SET next_dp at j TO remainder when next_dp at j PLUS dp at i is divided by MOD  
                END FOR  
            END FOR  
            SET dp TO next_dp  
        END FOR  
        
        RETURN remainder when sum of elements in dp is divided by MOD  
    END FUNCTION  
END CLASS
