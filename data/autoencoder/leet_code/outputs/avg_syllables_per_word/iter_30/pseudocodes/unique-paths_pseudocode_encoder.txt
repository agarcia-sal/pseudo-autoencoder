CLASS Solution  
    FUNCTION uniquePaths(m, n)  
        SET dp TO two dimensional collection with m rows and n columns where each element is one  
        FOR i FROM one TO m MINUS one  
            FOR j FROM one TO n MINUS one  
                SET element at position i AND j OF dp TO element at position i MINUS one AND j OF dp PLUS element at position i AND j MINUS one OF dp  
            END FOR  
        END FOR  
        RETURN element at position m MINUS one AND n MINUS one OF dp  
    END FUNCTION  
END CLASS
