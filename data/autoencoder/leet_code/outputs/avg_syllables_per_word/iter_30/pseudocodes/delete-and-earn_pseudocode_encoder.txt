CLASS Solution  
    FUNCTION deleteAndEarn(nums)  
        SET count TO the mapping of each number in nums to its frequency  
        SET unique_nums TO the list of keys in count sorted in ascending order  
        SET dp TO the list of zero repeated the length of unique_nums plus one times  
        SET dp at position one TO the element at position zero of unique_nums multiplied by the value of count at element at position zero of unique_nums  
        FOR index FROM two TO the length of unique_nums plus one  
            SET num TO element at position index minus one of unique_nums  
            IF num EQUALS the element at position index minus two of unique_nums PLUS one  
                SET dp at position index TO the larger of dp at position index minus one AND dp at position index minus two PLUS num multiplied by the value of count at num  
            ELSE  
                SET dp at position index TO dp at position index minus one PLUS num multiplied by the value of count at num  
            END IF  
        END FOR  
        RETURN the last element of dp  
    END FUNCTION  
END CLASS
