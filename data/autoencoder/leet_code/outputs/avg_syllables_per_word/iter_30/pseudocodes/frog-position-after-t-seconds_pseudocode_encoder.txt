CLASS Solution  
    FUNCTION frogPosition(n, edges, t, target)  
        IF n EQUALS one  
            RETURN one point zero  
        END IF  
        
        SET graph TO a new default mapping with list as default  
        FOR each pair u and v IN edges  
            APPEND v TO graph at key u  
            APPEND u TO graph at key v  
        END FOR  
        
        SET queue TO a new double ended queue containing one tuple of one, one point zero, zero  
        SET visited TO a new set containing one  
        
        WHILE queue is not empty  
            REMOVE the first element from queue and ASSIGN to current_node, current_probability, current_time  
            
            IF current_time EQUALS t OR current_node NOT EQUALS one AND the LENGTH OF graph at current_node EQUALS one  
                IF current_node EQUALS target  
                    RETURN current_probability  
                ELSE  
                    CONTINUE  
                END IF  
            END IF  
            
            SET possible_moves TO the LENGTH OF graph at current_node MINUS one IF current_node NOT EQUALS one ELSE zero  
            
            FOR each neighbor IN graph at current_node  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    APPEND to queue the tuple of neighbor, current_probability DIVIDED BY possible_moves, current_time PLUS one  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN zero point zero  
    END FUNCTION  
END CLASS
