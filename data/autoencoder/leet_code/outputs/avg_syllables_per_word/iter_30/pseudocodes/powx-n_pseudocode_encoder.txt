CLASS Solution  
    FUNCTION myPow(x, n)  
        FUNCTION qpow(a, n)  
            SET ans TO one  
            WHILE n is greater than zero  
                IF the remainder of n divided by two equals one  
                    SET ans TO ans MULTIPLIED BY a  
                END IF  
                SET a TO a MULTIPLIED BY a  
                SET n TO n DIVIDED BY two taking the integer part only  
            END WHILE  
            RETURN ans  
        END FUNCTION  

        IF n is greater than or equal to zero  
            RETURN qpow(x, n)  
        ELSE  
            RETURN one DIVIDED BY qpow(x, the negation of n)  
        END IF  
    END FUNCTION  
END CLASS
