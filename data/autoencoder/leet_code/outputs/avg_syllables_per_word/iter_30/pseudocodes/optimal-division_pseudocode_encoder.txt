CLASS Solution  
    FUNCTION optimalDivision(nums)  
        IF the LENGTH OF nums EQUALS one  
            RETURN the STRING REPRESENTATION OF element at position zero OF nums  
        END IF  
        IF the LENGTH OF nums EQUALS two  
            RETURN the STRING REPRESENTATION OF element at position zero OF nums CONCATENATED WITH the STRING REPRESENTATION OF element at position one OF nums SEPARATED BY a DIVISION WORD  
        END IF  
        RETURN the STRING REPRESENTATION OF element at position zero OF nums CONCATENATED WITH a DIVISION WORD FOLLOWED BY an OPENING PARENTHESIS FOLLOWED BY the JOINING OF the STRING REPRESENTATIONS OF all elements starting from position one TO the END OF nums WITH a DIVISION WORD FOLLOWED BY a CLOSING PARENTHESIS  
    END FUNCTION  
END CLASS
