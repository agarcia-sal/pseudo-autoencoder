CLASS Solution  
    FUNCTION find132pattern(nums)  
        SET second TO negative infinity  
        SET stack TO an empty collection  
        FOR each num IN nums taken from the last element to the first element  
            IF num LESS THAN second  
                RETURN true  
            END IF  
            WHILE stack is not empty AND the element at the last position OF stack LESS THAN num  
                SET second TO the last element removed FROM stack  
            END WHILE  
            APPEND num TO stack  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
