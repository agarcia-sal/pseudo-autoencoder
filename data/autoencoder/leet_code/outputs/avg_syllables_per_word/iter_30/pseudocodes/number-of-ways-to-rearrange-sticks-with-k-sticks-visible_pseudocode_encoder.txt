CLASS Solution  
    FUNCTION rearrangeSticks(n, k)  
        SET MODULO TO ten raised to the power of nine PLUS one  
        SET dp TO a two dimensional list with n PLUS one rows and k PLUS one columns filled with zero  
        SET element at row zero column zero of dp TO one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET element at row i column j of dp TO the sum of element at row i MINUS one column j MINUS one of dp PLUS the product of i MINUS one and element at row i MINUS one column j of dp  
                SET element at row i column j of dp TO element at row i column j of dp modulo MODULO  
            END FOR  
        END FOR  
        RETURN element at row n column k of dp  
    END FUNCTION  
END CLASS
