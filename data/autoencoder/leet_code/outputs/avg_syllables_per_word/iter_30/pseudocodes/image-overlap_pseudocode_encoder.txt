CLASS Solution  
    FUNCTION largestOverlap(img1, img2)  
        SET n TO the LENGTH OF img1  
        SET points1 TO an EMPTY COLLECTION  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i AND position j OF img1 EQUALS one  
                    APPEND the pair of i AND j TO points1  
                END IF  
            END FOR  
        END FOR  
        SET points2 TO an EMPTY COLLECTION  
        FOR i FROM zero TO n MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position i AND position j OF img2 EQUALS one  
                    APPEND the pair of i AND j TO points2  
                END IF  
            END FOR  
        END FOR  
        SET overlap_count TO a mapping with default integer value  
        FOR each x1 AND y1 IN points1  
            FOR each x2 AND y2 IN points2  
                SET dx TO x2 MINUS x1  
                SET dy TO y2 MINUS y1  
                INCREMENT the value at key of the pair dx AND dy IN overlap_count BY one  
            END FOR  
        END FOR  
        SET maximum_overlap TO zero  
        FOR each value IN the values OF overlap_count  
            IF value GREATER THAN maximum_overlap  
                SET maximum_overlap TO value  
            END IF  
        END FOR  
        RETURN maximum_overlap  
    END FUNCTION  
END CLASS
