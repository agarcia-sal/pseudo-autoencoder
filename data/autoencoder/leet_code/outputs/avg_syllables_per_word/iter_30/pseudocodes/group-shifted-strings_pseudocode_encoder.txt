CLASS Solution  
    FUNCTION groupStrings(strings)  
        FUNCTION get_canonical_form(s)  
            IF s EQUALS empty string  
                RETURN empty string  
            END IF  
            SET canonical_form TO empty list  
            FOR each character c IN s  
                SET difference TO the Unicode code point of c MINUS the Unicode code point of the first character of s  
                SET normalized_difference TO difference MODULO twenty six  
                APPEND normalized_difference converted to string TO canonical_form  
            END FOR  
            RETURN the concatenation of elements of canonical_form separated by commas  
        END FUNCTION  
  
        SET groups TO a dictionary with default value of empty list  
        FOR each string s IN strings  
            SET canonical_form TO get_canonical_form(s)  
            APPEND s TO the list corresponding to canonical_form in groups  
        END FOR  
        RETURN the list of all values in groups  
    END FUNCTION  
END CLASS
