CLASS Solution  
    FUNCTION strobogrammaticInRange(low, high)  
        FUNCTION generate_strobogrammatic(n, final_length)  
            IF n EQUALS zero  
                RETURN list containing empty string  
            END IF  
            IF n EQUALS one  
                RETURN list containing string zero, string one, string eight  
            END IF  
            SET middles TO generate_strobogrammatic(n MINUS two, final_length)  
            SET result TO empty list  
            FOR each middle IN middles  
                IF n NOT EQUALS final_length  
                    APPEND concatenation of string zero, middle, string zero TO result  
                END IF  
                APPEND concatenation of string one, middle, string one TO result  
                APPEND concatenation of string six, middle, string nine TO result  
                APPEND concatenation of string eight, middle, string eight TO result  
                APPEND concatenation of string nine, middle, string six TO result  
            END FOR  
            RETURN result  
        END FUNCTION  
        FUNCTION count_strobogrammatic(low, high)  
            SET count TO zero  
            FOR length FROM LENGTH OF low TO LENGTH OF high  
                FOR each num IN generate_strobogrammatic(length, length)  
                    IF (length EQUALS LENGTH OF low AND num LESS THAN low) OR (length EQUALS LENGTH OF high AND num GREATER THAN high)  
                        CONTINUE  
                    END IF  
                    INCREMENT count BY one  
                END FOR  
            END FOR  
            RETURN count  
        END FUNCTION  
        RETURN count_strobogrammatic(low, high)  
    END FUNCTION  
END CLASS
