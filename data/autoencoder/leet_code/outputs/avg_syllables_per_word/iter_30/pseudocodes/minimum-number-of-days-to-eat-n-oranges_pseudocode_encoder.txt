CLASS Solution  
    FUNCTION minDays(n)  
        FUNCTION dp(remaining)  
            IF remaining LESS THAN OR EQUAL TO one  
                RETURN remaining  
            END IF  
            RETURN one PLUS minimum of the sum of remainder of remaining DIVIDED BY two AND dp of remaining DIVIDED BY two, AND the sum of remainder of remaining DIVIDED BY three AND dp of remaining DIVIDED BY three  
        END FUNCTION  
        RETURN dp of n  
    END FUNCTION  
END CLASS
