CLASS Solution  
    FUNCTION racecar(target)  
        ASSIGN reference TO a new double ended queue containing a tuple of zero position one speed and zero steps  
        ASSIGN reference TO a new set containing a tuple of zero position and one speed  
        WHILE the queue is not empty  
            REMOVE and RETURN the first tuple from the queue and ASSIGN its elements to position speed and steps  
            IF position EQUALS target  
                RETURN steps  
            END IF  
            SET next_position TO position PLUS speed  
            SET next_speed TO speed MULTIPLIED BY two  
            IF the tuple of next_position and next_speed NOT IN visited AND next_position GREATER THAN OR EQUAL TO zero AND next_position LESS THAN two MULTIPLIED BY target  
                ADD the tuple of next_position and next_speed TO visited  
                APPEND the tuple of next_position next_speed and steps PLUS one TO the queue  
            END IF  
            IF speed GREATER THAN zero  
                SET next_speed TO negative one  
            ELSE  
                SET next_speed TO one  
            END IF  
            IF the tuple of position and next_speed NOT IN visited AND position GREATER THAN OR EQUAL TO zero AND position LESS THAN two MULTIPLIED BY target  
                ADD the tuple of position and next_speed TO visited  
                APPEND the tuple of position next_speed and steps PLUS one TO the queue  
            END IF  
        END WHILE  
    END FUNCTION  
END CLASS
