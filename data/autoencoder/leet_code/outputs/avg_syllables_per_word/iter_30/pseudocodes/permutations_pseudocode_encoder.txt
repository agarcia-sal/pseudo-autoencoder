CLASS Solution  
    FUNCTION permute(nums)  
        FUNCTION backtrack(path, remaining)  
            IF remaining EQUALS empty collection  
                APPEND path TO permutations  
                RETURN  
            END IF  
            FOR index FROM zero TO the LENGTH OF remaining MINUS one  
                SET new_path TO concatenation of path PLUS element at position index OF remaining  
                SET new_remaining TO concatenation of substring from position zero TO position index MINUS one OF remaining PLUS substring from position index PLUS one TO end OF remaining  
                CALL backtrack WITH new_path AND new_remaining  
            END FOR  
        END FUNCTION  
        SET permutations TO empty collection  
        CALL backtrack WITH empty collection AND nums  
        RETURN permutations  
    END FUNCTION  
END CLASS
