CLASS ListNode  
    FUNCTION __init__(self, val EQUALS zero, next EQUALS none)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET head TO NEW ListNode WITH val AS element at position zero OF values  
    SET p TO head  
    FOR each val IN elements from position one TO end OF values  
        SET node TO NEW ListNode WITH val AS val  
        SET p.next TO node  
        SET p TO node  
    END FOR  
    RETURN head  
END FUNCTION

FUNCTION is_same_list(p1, p2)  
    IF p1 IS none AND p2 IS none  
        RETURN true  
    END IF  
    IF p1 IS none OR p2 IS none  
        RETURN false  
    END IF  
    RETURN self val at p1 EQUALS self val at p2 AND is_same_list of self next at p1 AND self next at p2  
END FUNCTION

CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH val AS element at position zero OF values  
    SET i TO one  
    SET queue TO NEW empty double ended queue  
    APPEND root TO queue  
    WHILE queue IS not empty  
        SET node TO first element removed FROM queue  
        IF i LESS THAN length of values AND element at position i OF values IS not none  
            SET node.left TO NEW TreeNode WITH val AS element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i OF values IS not none  
            SET node.right TO NEW TreeNode WITH val AS element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF self val at p NOT EQUALS self val at q  
        RETURN false  
    ELSE  
        RETURN is_same_tree of self left at p AND self left at q AND is_same_tree of self right at p AND self right at q  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION sortedListToBST(self, head)  
        IF head IS none  
            RETURN none  
        END IF  
        
        FUNCTION findMiddle(left, right)  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUALS right AND self next at fast NOT EQUALS right  
                SET slow TO self next at slow  
                SET fast TO self next at self next at fast  
            END WHILE  
            RETURN slow  
        END FUNCTION  
        
        FUNCTION convertListToBST(left, right)  
            IF left EQUALS right  
                RETURN none  
            END IF  
            
            SET mid TO findMiddle(left, right)  
            
            SET node TO NEW TreeNode WITH val AS self val at mid  
            SET node.left TO convertListToBST(left, mid)  
            SET node.right TO convertListToBST(self next at mid, right)  
            RETURN node  
        END FUNCTION  
        
        RETURN convertListToBST(head, none)  
    END FUNCTION  
END CLASS
