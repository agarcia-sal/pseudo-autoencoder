CLASS Solution  
    FUNCTION checkPartitioning WITH PARAMETER s  
        FUNCTION is_palindrome WITH PARAMETER sub  
            RETURN sub EQUALS reverse of sub  
        END FUNCTION  
        
        SET n TO the LENGTH OF s  
        SET is_pal TO collection of collections OF False OF SIZE n BY n  
        
        FOR i FROM n MINUS one DOWN TO zero  
            FOR j FROM i TO n MINUS one  
                IF character at position i OF s EQUALS character at position j OF s AND  
                   (difference between j AND i LESS THAN OR EQUAL TO one OR value at position i PLUS one and j MINUS one OF is_pal IS True)  
                    SET value at position i and j OF is_pal TO True  
                END IF  
            END FOR  
        END FOR  
        
        FOR i FROM one TO n MINUS two  
            IF value at position zero and i MINUS one OF is_pal IS True  
                FOR j FROM i TO n MINUS two  
                    IF value at position i and j OF is_pal IS True AND value at position j PLUS one and n MINUS one OF is_pal IS True  
                        RETURN True  
                    END IF  
                END FOR  
            END IF  
        END FOR  
        
        RETURN False  
    END FUNCTION  
END CLASS
