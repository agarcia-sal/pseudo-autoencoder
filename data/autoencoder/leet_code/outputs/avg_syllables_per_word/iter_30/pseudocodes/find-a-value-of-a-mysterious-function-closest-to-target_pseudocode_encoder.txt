CLASS Solution  
    FUNCTION closestToTarget(arr, target)  
        SET seen TO empty set  
        SET min_diff TO positive infinity  
        FOR each num IN arr  
            SET new_seen TO empty set  
            FOR each x IN seen  
                APPEND bitwise AND of num AND x TO new_seen  
            END FOR  
            SET seen TO union of new_seen AND set containing num  
            FOR each val IN seen  
                SET difference TO absolute value of val MINUS target  
                IF difference LESS THAN min_diff  
                    SET min_diff TO difference  
                END IF  
            END FOR  
        END FOR  
        RETURN min_diff  
    END FUNCTION  
END CLASS
