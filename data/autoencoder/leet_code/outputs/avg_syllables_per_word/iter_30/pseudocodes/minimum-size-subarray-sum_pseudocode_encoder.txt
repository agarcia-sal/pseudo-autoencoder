CLASS Solution  
    FUNCTION minSubArrayLen(target, nums)  
        SET left TO zero  
        SET current_sum TO zero  
        SET min_length TO infinite value  
        FOR each right FROM zero TO the LENGTH OF nums MINUS one  
            INCREMENT current_sum BY element at position right OF nums  
            WHILE current_sum GREATER THAN OR EQUAL TO target  
                IF min_length GREATER THAN the difference between right and left PLUS one  
                    SET min_length TO the difference between right and left PLUS one  
                END IF  
                DECREMENT current_sum BY element at position left OF nums  
                INCREMENT left BY one  
            END WHILE  
        END FOR  
        IF min_length IS NOT EQUAL TO infinite value  
            RETURN min_length  
        ELSE  
            RETURN zero  
        END IF  
    END FUNCTION  
END CLASS
