CLASS Solution  
    FUNCTION maxSlidingWindow(nums, k)  
        IF nums EQUALS empty OR k EQUALS zero  
            RETURN empty  
        END IF  
        
        ASSIGN reference TO new empty deque named deq  
        ASSIGN reference TO new empty list named result  
        
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF deq IS NOT empty AND element at position zero OF deq LESS THAN i MINUS k PLUS one  
                REMOVE element at position zero FROM deq  
            END IF  
            
            WHILE deq IS NOT empty AND element at position last OF nums AT position element at position last OF deq LESS THAN nums AT position i  
                REMOVE element at position last FROM deq  
            END WHILE  
            
            APPEND i TO deq  
            
            IF i GREATER THAN OR EQUAL TO k MINUS one  
                APPEND nums AT position element at position zero OF deq TO result  
            END IF  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
