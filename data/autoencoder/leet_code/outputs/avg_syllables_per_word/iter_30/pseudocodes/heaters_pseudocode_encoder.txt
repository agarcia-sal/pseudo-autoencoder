CLASS Solution  
    FUNCTION findRadius(houses, heaters)  
        SORT houses IN ASCENDING ORDER  
        SORT heaters IN ASCENDING ORDER  
        
        SET min_radius TO zero  
        SET j TO zero  
        
        FOR each house IN houses  
            WHILE j LESS THAN THE LENGTH OF heaters MINUS one AND  
                  THE ABSOLUTE VALUE OF heater at position j MINUS house GREATER THAN OR EQUAL TO THE ABSOLUTE VALUE OF heater at position j PLUS one MINUS house  
                INCREMENT j BY one  
            END WHILE  
            
            SET min_radius TO THE GREATER VALUE BETWEEN min_radius AND THE ABSOLUTE VALUE OF heater at position j MINUS house  
        END FOR  
        
        RETURN min_radius  
    END FUNCTION  
END CLASS
