CLASS Solution  
    FUNCTION orderlyQueue(s, k)  
        IF k GREATER THAN one  
            RETURN the string sorted in lexicographical order of the characters in s  
        ELSE  
            SET n TO the length of s  
            SET smallest TO s  
            FOR index FROM one TO n MINUS one  
                SET rotated TO the substring from position index TO position n MINUS one OF s CONCATENATED WITH the substring from position zero TO position index MINUS one OF s  
                IF rotated LESS THAN smallest  
                    SET smallest TO rotated  
                END IF  
            END FOR  
            RETURN smallest  
        END IF  
    END FUNCTION  
END CLASS
