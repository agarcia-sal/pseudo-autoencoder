CLASS Solution  
    FUNCTION swimInWater(grid)  
        SET n TO the LENGTH OF grid  
        SET visited TO an empty set  
        SET min_heap TO a collection containing one element composed of the value at position zero zero OF grid, zero, zero  
        ADD the pair zero zero TO visited  
        SET directions TO a collection of four pairs: zero one, one zero, zero minus one, minus one zero  
        WHILE min_heap IS NOT empty  
            REMOVE from min_heap the element with the smallest first component AND ASSIGN its components TO t, i, j  
            IF i EQUALS n MINUS one AND j EQUALS n MINUS one  
                RETURN t  
            END IF  
            FOR each pair di, dj IN directions  
                SET ni TO i PLUS di  
                SET nj TO j PLUS dj  
                IF ni IS GREATER THAN OR EQUAL TO zero AND ni IS LESS THAN n AND nj IS GREATER THAN OR EQUAL TO zero AND nj IS LESS THAN n AND the pair ni nj IS NOT IN visited  
                    ADD the pair ni nj TO visited  
                    INSERT into min_heap the element composed of the maximum BETWEEN t AND the value at position ni nj OF grid, ni, nj  
                END IF  
            END FOR  
        END WHILE  
        RETURN minus one  
    END FUNCTION  
END CLASS
