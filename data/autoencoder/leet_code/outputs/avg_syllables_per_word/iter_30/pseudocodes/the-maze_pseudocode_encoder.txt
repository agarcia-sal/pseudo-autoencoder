CLASS Solution  
    FUNCTION hasPath WITH PARAMETERS maze, start, destination  
        SET directions TO a list containing the tuples of zero and one, one and zero, zero and negative one, negative one and zero  
        SET m TO the LENGTH OF maze  
        SET n TO the LENGTH OF element at position zero OF maze  
        SET visited TO an empty set  
        
        FUNCTION dfs WITH PARAMETERS x, y  
            IF the tuple of x and y IS IN visited THEN  
                RETURN false  
            END IF  
            IF the list containing x and y EQUALS destination THEN  
                RETURN true  
            END IF  
            ADD the tuple of x and y TO visited  
            
            FOR each pair dx and dy IN directions  
                SET nx TO x  
                SET ny TO y  
                WHILE nx PLUS dx IS GREATER THAN OR EQUAL TO zero AND nx PLUS dx IS LESS THAN m AND ny PLUS dy IS GREATER THAN OR EQUAL TO zero AND ny PLUS dy IS LESS THAN n AND element at position nx PLUS dx of maze AT position ny PLUS dy EQUALS zero DO  
                    SET nx TO nx PLUS dx  
                    SET ny TO ny PLUS dy  
                END WHILE  
                IF dfs CALLED WITH nx AND ny RETURNS true THEN  
                    RETURN true  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        RETURN dfs CALLED WITH element at position zero OF start AND element at position one OF start  
    END FUNCTION  
END CLASS
