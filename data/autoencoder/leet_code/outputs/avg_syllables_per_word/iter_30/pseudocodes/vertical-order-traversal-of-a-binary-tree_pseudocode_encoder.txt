CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value as element at position zero OF values  
    SET i TO one  
    INITIALIZE queue as double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO first element REMOVED FROM queue  
        IF i LESS THAN length of values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with value as element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with value as element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value at p NOT EQUALS value at q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION verticalTraversal(self, root)  
        IF root IS none  
            RETURN empty list  
        END IF  
        INITIALIZE node_dict as default dictionary of list  
        SET min_col TO zero  
        SET max_col TO zero  

        FUNCTION dfs(node, row, col)  
            IF node IS none  
                RETURN  
            END IF  
            APPEND pair of row AND value at node TO list at key col IN node_dict  
            SET min_col TO smaller of min_col AND col  
            SET max_col TO larger of max_col AND col  
            CALL dfs WITH left child of node, row PLUS one, col MINUS one  
            CALL dfs WITH right child of node, row PLUS one, col PLUS one  
        END FUNCTION  

        CALL dfs WITH root, zero, zero  

        INITIALIZE result AS empty list  
        FOR col FROM min_col TO max_col  
            SET sorted_list TO list at key col IN node_dict SORTED first BY first element THEN BY second element OF pairs  
            APPEND list OF second elements OF sorted_list TO result  
        END FOR  

        RETURN result  
    END FUNCTION  
END CLASS
