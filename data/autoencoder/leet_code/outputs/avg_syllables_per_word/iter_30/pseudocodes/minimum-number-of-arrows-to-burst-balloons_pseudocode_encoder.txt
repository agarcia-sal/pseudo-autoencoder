CLASS Solution  
    FUNCTION findMinArrowShots(points)  
        SORT points BY the value at position one of each element IN ascending order  
        SET arrows TO zero  
        SET last_arrow_position TO negative infinity  
        FOR each balloon IN points  
            IF the value at position zero of balloon GREATER THAN last_arrow_position  
                INCREMENT arrows BY one  
                SET last_arrow_position TO the value at position one of balloon  
            END IF  
        END FOR  
        RETURN arrows  
    END FUNCTION  
END CLASS
