CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS NOT PRESENT OR IS EMPTY  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH value AT position zero OF values  
    SET i TO one  
    SET queue TO AN EMPTY double ended queue  
    APPEND root TO queue  
    WHILE queue IS NOT EMPTY  
        SET node TO THE FIRST ELEMENT REMOVED FROM queue  
        IF i IS LESS THAN THE LENGTH OF values AND value AT position i OF values IS NOT none  
            SET node.left TO A NEW TreeNode WITH value AT position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i IS LESS THAN THE LENGTH OF values AND value AT position i OF values IS NOT none  
            SET node.right TO A NEW TreeNode WITH value AT position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS NOT PRESENT AND q IS NOT PRESENT  
        RETURN true  
    ELSE IF p IS NOT PRESENT OR q IS NOT PRESENT  
        RETURN false  
    ELSE IF value AT p IS NOT EQUALS value AT q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(left child OF p, left child OF q) AND is_same_tree(right child OF p, right child OF q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION levelOrder(self, root)  
        IF root IS NOT PRESENT  
            RETURN AN EMPTY LIST  
        END IF  
        SET result TO AN EMPTY LIST  
        SET queue TO A double ended queue CONTAINING root  
        WHILE queue IS NOT EMPTY  
            SET level_size TO THE LENGTH OF queue  
            SET current_level TO AN EMPTY LIST  
            FOR each number FROM zero TO level_size MINUS one  
                SET node TO THE FIRST ELEMENT REMOVED FROM queue  
                APPEND value AT node TO current_level  
                IF left child OF node IS PRESENT  
                    APPEND left child OF node TO queue  
                END IF  
                IF right child OF node IS PRESENT  
                    APPEND right child OF node TO queue  
                END IF  
            END FOR  
            APPEND current_level TO result  
        END WHILE  
        RETURN result  
    END FUNCTION  
END CLASS
