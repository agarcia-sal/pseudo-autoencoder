CLASS Solution  
    FUNCTION fullJustify(words, maxWidth)  
        FUNCTION justify_line(line, num_of_letters, is_last_line)  
            IF the LENGTH OF line EQUALS one OR is_last_line  
                SET spaces_to_add TO maxWidth MINUS num_of_letters MINUS the LENGTH OF line PLUS one  
                RETURN the CONCATENATION of the ELEMENTS of line separated by a single space PLUS the REPEAT of a single space FOR spaces_to_add TIMES  
            ELSE  
                SET total_spaces TO maxWidth MINUS num_of_letters  
                SET num_of_gaps TO the LENGTH OF line MINUS one  
                SET spaces_between_words TO the INTEGER DIVISION OF total_spaces BY num_of_gaps  
                SET extra_spaces TO total_spaces MODULO num_of_gaps  
                FOR index FROM zero TO num_of_gaps MINUS one  
                    IF index LESS THAN extra_spaces  
                        APPEND a single space REPEATED spaces_between_words PLUS one TIMES TO the ELEMENT at position index OF line  
                    ELSE  
                        APPEND a single space REPEATED spaces_between_words TIMES TO the ELEMENT at position index OF line  
                    END IF  
                END FOR  
                RETURN the CONCATENATION of all ELEMENTS of line with no separator  
            END IF  
        END FUNCTION  
      
        SET result TO an empty list  
        SET line TO an empty list  
        SET num_of_letters TO zero  
      
        FOR each word IN words  
            IF num_of_letters PLUS the LENGTH OF word PLUS the LENGTH OF line GREATER THAN maxWidth  
                APPEND the RESULT OF justify_line WITH PARAMETERS line, num_of_letters, false TO result  
                SET line TO an empty list  
                SET num_of_letters TO zero  
            END IF  
            APPEND word TO line  
            INCREMENT num_of_letters BY the LENGTH OF word  
        END FOR  
      
        APPEND the RESULT OF justify_line WITH PARAMETERS line, num_of_letters, true TO result  
      
        RETURN result  
    END FUNCTION  
END CLASS
