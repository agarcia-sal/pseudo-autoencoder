CLASS Solution  
    FUNCTION minimumLengthEncoding(words)  
        SET reversed_words TO sorted list of each word reversed in words  
        SET total_length TO zero  
        FOR index FROM zero TO the LENGTH OF reversed_words MINUS one  
            IF index PLUS one EQUALS the LENGTH OF reversed_words OR reversed_words element at position index PLUS one DOES NOT START WITH reversed_words element at position index  
                INCREMENT total_length BY the LENGTH OF reversed_words element at position index PLUS one  
            END IF  
        END FOR  
        RETURN total_length  
    END FUNCTION  
END CLASS
