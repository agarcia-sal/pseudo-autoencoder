CLASS Solution  
    FUNCTION minimumTime WITH PARAMETER s  
        SET n TO the LENGTH OF s  
        SET left_costs TO a list of zero REPEATED n times  
        SET right_costs TO a list of zero REPEATED n times  

        IF element at position zero OF s EQUALS character one  
            SET element at position zero OF left_costs TO one  
        ELSE  
            SET element at position zero OF left_costs TO zero  
        END IF  

        FOR i FROM one TO n MINUS one  
            IF element at position i OF s EQUALS character zero  
                SET element at position i OF left_costs TO element at position i MINUS one OF left_costs  
            ELSE  
                SET element at position i OF left_costs TO the MINIMUM OF i PLUS one AND element at position i MINUS one OF left_costs PLUS two  
            END IF  
        END FOR  

        IF element at position n MINUS one OF s EQUALS character one  
            SET element at position n MINUS one OF right_costs TO one  
        ELSE  
            SET element at position n MINUS one OF right_costs TO zero  
        END IF  

        FOR i FROM n MINUS two DOWN TO zero  
            IF element at position i OF s EQUALS character zero  
                SET element at position i OF right_costs TO element at position i PLUS one OF right_costs  
            ELSE  
                SET element at position i OF right_costs TO the MINIMUM OF n MINUS i AND element at position i PLUS one OF right_costs PLUS two  
            END IF  
        END FOR  

        SET min_time TO the MINIMUM OF element at position n MINUS one OF left_costs AND element at position zero OF right_costs  

        FOR i FROM zero TO n MINUS two  
            SET min_time TO the MINIMUM OF min_time AND element at position i OF left_costs PLUS element at position i PLUS one OF right_costs  
        END FOR  

        RETURN min_time  
    END FUNCTION  
END CLASS
