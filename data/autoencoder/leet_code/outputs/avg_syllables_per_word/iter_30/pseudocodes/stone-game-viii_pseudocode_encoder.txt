CLASS Solution  
    FUNCTION stoneGameVIII(stones)  
        SET prefix_sum TO a list of zero repeated the LENGTH OF stones times  
        SET element at position zero OF prefix_sum TO element at position zero OF stones  
        FOR i FROM one TO the LENGTH OF stones MINUS one  
            SET element at position i OF prefix_sum TO element at position i MINUS one OF prefix_sum PLUS element at position i OF stones  
        END FOR  
        SET dp TO element at the last position OF prefix_sum  
        FOR i FROM the LENGTH OF stones MINUS two DOWN TO one  
            SET dp TO the greater value BETWEEN dp AND element at position i OF prefix_sum MINUS dp  
        END FOR  
        RETURN dp  
    END FUNCTION  
END CLASS
