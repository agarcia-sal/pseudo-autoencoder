CLASS Solution  
    FUNCTION minEatingSpeed(piles, h)  
        FUNCTION canFinish(k)  
            SET hours TO zero  
            FOR each pile IN piles  
                INCREMENT hours BY the smallest integer greater than or equal to the division of pile BY k  
            END FOR  
            RETURN hours LESS THAN OR EQUAL TO h  
        END FUNCTION  
        SET left TO one  
        SET right TO the maximum element IN piles  
        WHILE left LESS THAN OR EQUAL TO right  
            SET mid TO the integer division of the sum of left PLUS right BY two  
            IF canFinish(mid)  
                SET right TO mid MINUS one  
            ELSE  
                SET left TO mid PLUS one  
            END IF  
        END WHILE  
        RETURN left  
    END FUNCTION  
END CLASS
