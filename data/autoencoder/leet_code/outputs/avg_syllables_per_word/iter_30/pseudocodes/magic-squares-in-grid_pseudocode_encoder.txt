CLASS Solution  
    FUNCTION numMagicSquaresInside(grid)  
        FUNCTION is_magic(i, j)  
            IF element at position i AND position j OF grid NOT EQUALS five  
                RETURN false  
            END IF  
            SET numbers TO empty set  
            FOR x FROM i MINUS one TO i PLUS one  
                FOR y FROM j MINUS one TO j PLUS one  
                    IF element at position x AND position y OF grid LESS THAN one OR element at position x AND position y OF grid GREATER THAN nine OR element at position x AND position y OF grid IN numbers  
                        RETURN false  
                    END IF  
                    ADD element at position x AND position y OF grid TO numbers  
                END FOR  
            END FOR  
            FOR start FROM i MINUS one TO i PLUS one  
                IF sum of element at position start AND position j MINUS one OF grid PLUS element at position start AND position j OF grid PLUS element at position start AND position j PLUS one OF grid NOT EQUALS fifteen OR sum of element at position i MINUS one AND position start OF grid PLUS element at position i AND position start OF grid PLUS element at position i PLUS one AND position start OF grid NOT EQUALS fifteen  
                    RETURN false  
                END IF  
            END FOR  
            IF sum of element at position i MINUS one AND position j MINUS one OF grid PLUS element at position i AND position j OF grid PLUS element at position i PLUS one AND position j PLUS one OF grid NOT EQUALS fifteen OR sum of element at position i MINUS one AND position j PLUS one OF grid PLUS element at position i AND position j OF grid PLUS element at position i PLUS one AND position j MINUS one OF grid NOT EQUALS fifteen  
                RETURN false  
            END IF  
            RETURN true  
        END FUNCTION  
        SET count TO zero  
        SET rows TO the LENGTH OF grid  
        SET cols TO the LENGTH OF element at position zero OF grid  
        FOR i FROM one TO rows MINUS two  
            FOR j FROM one TO cols MINUS two  
                IF is_magic(i, j)  
                    INCREMENT count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
