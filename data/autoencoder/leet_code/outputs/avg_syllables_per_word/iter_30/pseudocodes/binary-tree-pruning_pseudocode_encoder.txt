CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at position zero OF values  
    SET i TO one  
    SET queue TO empty double ended queue  
    APPEND root TO queue  
    WHILE the LENGTH OF queue GREATER THAN zero  
        SET node TO element at position zero removed from the left OF queue  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values NOT EQUALS none  
            SET node.left TO new TreeNode with value at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values NOT EQUALS none  
            SET node.right TO new TreeNode with value at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN True  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN False  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree with p.left and q.left AND is_same_tree with p.right and q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pruneTree(self, root)  
        IF root EQUALS none  
            RETURN none  
        END IF  
        SET root.left TO self.pruneTree with root.left  
        SET root.right TO self.pruneTree with root.right  
        IF root.val EQUALS zero AND root.left EQUALS none AND root.right EQUALS none  
            RETURN none  
        END IF  
        RETURN root  
    END FUNCTION  
END CLASS
