CLASS Solution  
    FUNCTION shortestSuperstring(words)  
        SET n TO the LENGTH OF words  
        
        FUNCTION overlap(i, j)  
            FOR k FROM the MINIMUM OF the LENGTH OF element at position i OF words AND the LENGTH OF element at position j OF words DOWN TO one  
                IF the substring from position the LENGTH OF element at position i OF words MINUS k PLUS one TO the LENGTH OF element at position i OF words OF element at position i OF words EQUALS the substring from position one TO position k OF element at position j OF words  
                    RETURN k  
                END IF  
            END FOR  
            RETURN zero  
        END FUNCTION  
        
        FUNCTION dp(mask, i)  
            IF mask EQUALS the NUMBER one SHIFTED LEFT BY n MINUS one  
                RETURN zero AND empty string  
            END IF  
            
            SET min_len TO positive infinity  
            SET best_path TO empty string  
            FOR j FROM zero TO n MINUS one  
                IF bitwise AND of mask AND the NUMBER one SHIFTED LEFT BY j EQUALS zero  
                    SET result TO dp(bitwise OR of mask AND the NUMBER one SHIFTED LEFT BY j, j)  
                    SET length TO element at position zero OF result PLUS the LENGTH OF element at position j OF words MINUS overlap(i, j)  
                    SET path TO element at position one OF result  
                    IF length LESS THAN min_len  
                        SET min_len TO length  
                        SET best_path TO the substring from position overlap(i, j) PLUS one TO the LENGTH OF element at position j OF words OF element at position j OF words CONCATENATED WITH path  
                    END IF  
                END IF  
            END FOR  
            
            RETURN min_len AND best_path  
        END FUNCTION  
        
        SET min_len TO positive infinity  
        SET shortest_path TO empty string  
        FOR i FROM zero TO n MINUS one  
            SET result TO dp(the NUMBER one SHIFTED LEFT BY i, i)  
            SET length TO element at position zero OF result PLUS the LENGTH OF element at position i OF words  
            SET path TO element at position one OF result  
            IF length LESS THAN min_len  
                SET min_len TO length  
                SET shortest_path TO element at position i OF words CONCATENATED WITH path  
            END IF  
        END FOR  
        
        RETURN shortest_path  
    END FUNCTION  
END CLASS
