CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS EQUAL TO none OR values LENGTH EQUALS zero  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value element at position zero OF values  
    SET i TO one  
    SET queue TO new deque collection  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO element at front of queue removed  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with value element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with value element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF p.val IS NOT EQUAL TO q.val  
        RETURN false  
    ELSE  
        RETURN is_same_tree with parameters p.left AND q.left AND is_same_tree with parameters p.right AND q.right  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION upsideDownBinaryTree(self, root)  
        IF root IS none OR root.left IS none  
            RETURN root  
        END IF  
        SET new_root TO self.upsideDownBinaryTree with parameter root.left  
        SET root.left.left TO root.right  
        SET root.left.right TO root  
        SET root.left TO none  
        SET root.right TO none  
        RETURN new_root  
    END FUNCTION  
END CLASS
