CLASS Solution  
    FUNCTION maxValueOfCoins(piles, k)  
        SET n TO the LENGTH OF piles  
        CREATE dp AS list of lists of zero with outer size n PLUS one and inner size k PLUS one  
        FOR i FROM one TO n  
            FOR j FROM one TO k  
                SET current_pile TO element at position i MINUS one OF piles  
                SET current_value TO zero  
                FOR l FROM zero TO the MINIMUM OF j AND the LENGTH OF current_pile  
                    IF l GREATER THAN zero  
                        INCREMENT current_value BY element at position l MINUS one OF current_pile  
                    END IF  
                    SET dp element at position i and j TO the MAXIMUM OF dp element at position i and j AND the SUM OF dp element at position i MINUS one and j MINUS l AND current_value  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp element at position n and k  
    END FUNCTION  
END CLASS
