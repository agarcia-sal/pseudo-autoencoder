CLASS Solution  
    FUNCTION rearrangeString(s, k)  
        IF k LESS THAN OR EQUAL TO one  
            RETURN s  
        END IF  
        SET freq TO mapping of characters in s TO their respective counts  
        SET max_heap TO list created by mapping each character and negative of its count in freq  
        TRANSFORM max_heap INTO a heap structure based on counts  
        SET wait_queue TO empty list  
        SET result TO empty list  
        WHILE max_heap IS NOT empty  
            REMOVE element with largest count FROM max_heap INTO count AND char  
            APPEND char TO result  
            APPEND tuple of count plus one AND char TO wait_queue  
            IF the LENGTH OF wait_queue EQUALS k  
                REMOVE first element FROM wait_queue INTO count AND char  
                IF count LESS THAN zero  
                    ADD tuple of count AND char INTO max_heap  
                END IF  
            END IF  
        END WHILE  
        IF LENGTH OF result NOT EQUALS LENGTH OF s  
            RETURN empty string  
        END IF  
        RETURN concatenation of elements in result  
    END FUNCTION  
END CLASS
