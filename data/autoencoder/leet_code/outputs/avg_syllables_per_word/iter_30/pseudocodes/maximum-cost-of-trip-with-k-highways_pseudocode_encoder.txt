CLASS Solution  
    FUNCTION maximumCost(n, highways, k)  
        CREATE empty adjacency list named graph with default type list  
        FOR each city1, city2, toll IN highways  
            APPEND tuple of city2 and toll TO the list at city1 in graph  
            APPEND tuple of city1 and toll TO the list at city2 in graph  
        END FOR  
        
        FUNCTION dfs(city, visited, cost, highways_used)  
            IF highways_used EQUALS k  
                SET max_cost TO the greater of max_cost and cost  
                RETURN  
            END IF  
            
            FOR each neighbor, toll IN the list at city in graph  
                IF neighbor NOT IN visited  
                    ADD neighbor TO visited  
                    CALL dfs WITH neighbor, visited, cost PLUS toll, highways_used PLUS one  
                    REMOVE neighbor FROM visited  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET max_cost TO negative one  
        FOR start_city FROM zero TO n MINUS one  
            CREATE visited AS a set containing start_city  
            CALL dfs WITH start_city, visited, zero, zero  
        END FOR  
        
        RETURN max_cost  
    END FUNCTION  
END CLASS
