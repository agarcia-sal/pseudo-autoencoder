CLASS Solution  
    FUNCTION boldWords(words, s)  
        SET bold TO a list of boolean false values with the length of s  
        FOR each word IN words  
            SET start TO the position of word in s  
            WHILE start is not equal to negative one  
                FOR index FROM start TO start PLUS the length of word MINUS one  
                    SET the element at position index OF bold TO true  
                END FOR  
                SET start TO the position of word in s starting from start PLUS one  
            END WHILE  
        END FOR  
        SET result TO an empty list  
        SET i TO zero  
        WHILE i is less than the length of s  
            IF the element at position i OF bold IS true  
                APPEND opening bold tag TO result  
                WHILE i is less than the length of s AND the element at position i OF bold IS true  
                    APPEND the element at position i OF s TO result  
                    INCREMENT i BY one  
                END WHILE  
                APPEND closing bold tag TO result  
            ELSE  
                APPEND the element at position i OF s TO result  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        RETURN the concatenation of elements in result  
    END FUNCTION  
END CLASS
