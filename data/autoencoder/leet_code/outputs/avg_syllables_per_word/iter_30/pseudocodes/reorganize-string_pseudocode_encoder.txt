CLASS Solution  
    FUNCTION reorganizeString(s)  
        SET count TO frequency count of each character in s  
        SET max_heap TO list of pairs of negative frequency and character for each character and frequency in count  
        INITIALIZE max_heap as a heap  
        SET prev_char TO none  
        SET prev_freq TO zero  
        SET result TO empty list  
        WHILE max_heap is not empty OR prev_freq less than zero  
            IF max_heap is empty AND prev_freq less than zero  
                RETURN empty string  
            END IF  
            POP the top element from max_heap INTO freq and char  
            APPEND char TO result  
            INCREMENT freq BY one  
            IF prev_freq less than zero  
                PUSH pair of prev_freq and prev_char INTO max_heap  
            END IF  
            SET prev_char TO char  
            SET prev_freq TO freq  
        END WHILE  
        RETURN concatenation of characters in result  
    END FUNCTION  
END CLASS
