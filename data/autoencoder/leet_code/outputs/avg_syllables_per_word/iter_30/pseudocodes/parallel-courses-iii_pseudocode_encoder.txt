CLASS Solution  
    FUNCTION minimumTime(n, relations, time)  
        ASSIGN reference TO an empty mapping with default value as empty list named graph  
        FOR each pair prevCourse comma nextCourse IN relations  
            APPEND prevCourse TO the list at key nextCourse OF graph  
        END FOR  

        SET memo TO a list of length n PLUS one with all elements set to negative one  

        FUNCTION dfs(course)  
            IF element at position course OF memo NOT EQUALS negative one  
                RETURN element at position course OF memo  
            END IF  

            SET max_prerequisite_time TO zero  
            FOR each prevCourse IN the list at key course OF graph  
                SET current_time TO dfs(prevCourse)  
                IF current_time GREATER THAN max_prerequisite_time  
                    SET max_prerequisite_time TO current_time  
                END IF  
            END FOR  

            SET element at position course OF memo TO max_prerequisite_time PLUS element at position course MINUS one OF time  
            RETURN element at position course OF memo  
        END FUNCTION  

        SET total_minimum_time TO zero  
        FOR course FROM one TO n  
            SET current_course_time TO dfs(course)  
            IF current_course_time GREATER THAN total_minimum_time  
                SET total_minimum_time TO current_course_time  
            END IF  
        END FOR  

        RETURN total_minimum_time  
    END FUNCTION  
END CLASS
