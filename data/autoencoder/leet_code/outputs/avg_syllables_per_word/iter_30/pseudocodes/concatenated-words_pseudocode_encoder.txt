CLASS Trie  
    FUNCTION __init__(self)  
        SET self.children TO list of none repeated twenty six times  
        SET self.is_end TO false  
    END FUNCTION  
  
    FUNCTION insert(self, w)  
        SET node TO self  
        FOR each character c IN w  
            SET idx TO the numerical code of character c MINUS the numerical code of character a  
            IF element at position idx OF node.children IS none  
                SET element at position idx OF node.children TO new Trie object  
            END IF  
            SET node TO element at position idx OF node.children  
        END FOR  
        SET node.is_end TO true  
    END FUNCTION  
END CLASS  

CLASS Solution  
    FUNCTION findAllConcatenatedWordsInADict(self, words)  
        FUNCTION dfs(w, start)  
            IF start EQUALS the length of w  
                RETURN true  
            END IF  
            SET node TO trie  
            FOR i FROM start TO the length of w MINUS one  
                SET idx TO the numerical code of element at position i OF w MINUS the numerical code of character a  
                IF element at position idx OF node.children IS none  
                    RETURN false  
                END IF  
                SET node TO element at position idx OF node.children  
                IF node.is_end AND dfs(w, i PLUS one)  
                    RETURN true  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
  
        SET trie TO new Trie object  
        SET ans TO empty list  
        SORT words BY length in ascending order  
        FOR each word w IN words  
            IF w IS NOT empty  
                IF dfs(w, zero)  
                    APPEND w TO ans  
                ELSE  
                    trie.insert(w)  
                END IF  
            END IF  
        END FOR  
        RETURN ans  
    END FUNCTION  
END CLASS
