CLASS Solution  
    FUNCTION searchMatrix(matrix, target)  
        IF matrix EQUALS empty OR element at position zero OF matrix EQUALS empty  
            RETURN false  
        END IF  
        SET row TO zero  
        SET col TO length of element at position zero OF matrix MINUS one  
        WHILE row LESS THAN length of matrix AND col GREATER THAN OR EQUAL TO zero  
            IF element at position row OF matrix AT position col EQUALS target  
                RETURN true  
            ELSE IF element at position row OF matrix AT position col GREATER THAN target  
                DECREMENT col BY one  
            ELSE  
                INCREMENT row BY one  
            END IF  
        END WHILE  
        RETURN false  
    END FUNCTION  
END CLASS
