CLASS Solution  
    FUNCTION maxEqualFreq(nums)  
        SET count TO a new empty counter  
        SET freq TO a new empty counter  
        SET max_len TO zero  
        SET index TO zero  
        FOR each element num IN nums  
            INCREMENT index BY one  
            INCREMENT the count of num BY one  
            IF the count of num GREATER THAN one THEN  
                DECREMENT the frequency of the count of num MINUS one BY one  
                IF the frequency of the count of num MINUS one EQUALS zero THEN  
                    REMOVE the frequency of the count of num MINUS one FROM freq  
                END IF  
            END IF  
            INCREMENT the frequency of the count of num BY one  
            IF the length of freq EQUALS one AND (freq CONTAINS key one OR the first value of freq EQUALS one) THEN  
                SET max_len TO index  
            ELSE IF the length of freq EQUALS two AND ((freq CONTAINS key one AND the frequency of one EQUALS one) OR ((maximum key of freq MINUS minimum key of freq) EQUALS one AND the frequency of the maximum key EQUALS one)) THEN  
                SET max_len TO index  
            END IF  
        END FOR  
        RETURN max_len  
    END FUNCTION  
END CLASS
