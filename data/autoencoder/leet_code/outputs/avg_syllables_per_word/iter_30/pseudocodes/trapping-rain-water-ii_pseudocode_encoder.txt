CLASS Solution  
    FUNCTION trapRainWater(heightMap)  
        IF heightMap EQUALS empty OR element at position zero OF heightMap EQUALS empty  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF heightMap  
        SET n TO the LENGTH OF element at position zero OF heightMap  
        SET visited TO a two dimensional list of BOOLEAN false values with m rows and n columns  
        SET min_heap TO empty list  
        FOR index i FROM zero TO m MINUS one  
            PUSH the tuple of element at position i and zero OF heightMap, i, zero TO min_heap  
            PUSH the tuple of element at position i and n MINUS one OF heightMap, i, n MINUS one TO min_heap  
            SET element at position i and zero OF visited TO true  
            SET element at position i and n MINUS one OF visited TO true  
        END FOR  
        FOR index j FROM zero TO n MINUS one  
            PUSH the tuple of element at position zero and j OF heightMap, zero, j TO min_heap  
            PUSH the tuple of element at position m MINUS one and j OF heightMap, m MINUS one, j TO min_heap  
            SET element at position zero and j OF visited TO true  
            SET element at position m MINUS one and j OF visited TO true  
        END FOR  
        SET directions TO list of tuples: minus one and zero, one and zero, zero and minus one, zero and one  
        SET water_trapped TO zero  
        WHILE min_heap IS NOT empty  
            POP the smallest element from min_heap INTO height, x, y  
            FOR each tuple dx, dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n AND element at position nx and ny OF visited EQUALS false  
                    SET element at position nx and ny OF visited TO true  
                    INCREMENT water_trapped BY the maximum VALUE BETWEEN zero AND height MINUS element at position nx and ny OF heightMap  
                    PUSH the tuple of the maximum VALUE BETWEEN height AND element at position nx and ny OF heightMap, nx, ny TO min_heap  
                END IF  
            END FOR  
        END WHILE  
        RETURN water_trapped  
    END FUNCTION  
END CLASS
