CLASS Solution  
    FUNCTION numOfArrays(n, m, k)  
        SET MOD TO ten raised to the power of nine PLUS one  
        INITIALIZE dp AS a three dimensional collection of zeros with dimensions (n plus one) BY (m plus one) BY (k plus one)  
        FOR j FROM one TO m  
            SET element at position one OF element at position j OF element at position one OF dp TO one  
        END FOR  
        FOR i FROM two TO n  
            FOR j FROM one TO m  
                FOR p FROM one TO k  
                    FOR x FROM one TO j MINUS one  
                        SET element at position p OF element at position j OF element at position i OF dp TO element at position p OF element at position j OF element at position i OF dp PLUS element at position p MINUS one OF element at position x OF element at position i MINUS one OF dp  
                        SET element at position p OF element at position j OF element at position i OF dp TO element at position p OF element at position j OF element at position i OF dp MODULO MOD  
                    END FOR  
                    SET element at position p OF element at position j OF element at position i OF dp TO element at position p OF element at position j OF element at position i OF dp PLUS element at position p OF element at position j OF element at position i MINUS one OF dp MULTIPLIED BY j  
                    SET element at position p OF element at position j OF element at position i OF dp TO element at position p OF element at position j OF element at position i OF dp MODULO MOD  
                END FOR  
            END FOR  
        END FOR  
        SET result TO zero  
        FOR j FROM one TO m  
            SET result TO result PLUS element at position k OF element at position j OF element at position n OF dp  
            SET result TO result MODULO MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
