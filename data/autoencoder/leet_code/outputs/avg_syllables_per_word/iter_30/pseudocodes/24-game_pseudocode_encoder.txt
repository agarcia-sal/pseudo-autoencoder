CLASS Solution  
    FUNCTION judgePoint24(cards)  
        FUNCTION solve(nums)  
            IF the LENGTH OF nums EQUALS one  
                RETURN the ABSOLUTE VALUE OF the element at position zero OF nums MINUS twenty FOUR LESS THAN one divided BY ten to the power of six  
            END IF  
            FOR i FROM zero TO the LENGTH OF nums MINUS one  
                FOR j FROM zero TO the LENGTH OF nums MINUS one  
                    IF i NOT EQUALS j  
                        SET new_nums TO an EMPTY COLLECTION  
                        FOR k FROM zero TO the LENGTH OF nums MINUS one  
                            IF k NOT EQUALS i AND k NOT EQUALS j  
                                APPEND element at position k OF nums TO new_nums  
                            END IF  
                        END FOR  
                        FOR each operation IN the COLLECTION of true division, multiplication, addition, subtraction  
                            IF (operation EQUALS addition OR operation EQUALS multiplication) AND j GREATER THAN i  
                                CONTINUE  
                            END IF  
                            IF operation NOT EQUALS true division OR element at position j OF nums NOT EQUALS zero  
                                APPEND the RESULT OF applying operation TO element at position i OF nums AND element at position j OF nums TO new_nums  
                                IF solve(new_nums)  
                                    RETURN true  
                                END IF  
                                REMOVE the LAST element FROM new_nums  
                            END IF  
                        END FOR  
                    END IF  
                END FOR  
            END FOR  
            RETURN false  
        END FUNCTION  
        RETURN solve(cards)  
    END FUNCTION  
END CLASS
