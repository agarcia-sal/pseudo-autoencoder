CLASS Solution  
    FUNCTION minimumMoves(grid)  
        SET n TO the LENGTH OF grid  
        SET queue TO a new deque containing a tuple of zero zero zero one zero  
        SET visited TO a new set containing a tuple of zero zero zero one  
        WHILE the LENGTH OF queue is greater than zero  
            REMOVE the leftmost element FROM queue AND ASSIGN TO r1 c1 r2 c2 steps  
            IF r1 EQUALS n MINUS one AND c1 EQUALS n MINUS two AND r2 EQUALS n MINUS one AND c2 EQUALS n MINUS one  
                RETURN steps  
            END IF  
            IF c2 PLUS one LESS THAN n AND element at position r1 at position c1 PLUS one OF grid EQUALS zero AND element at position r2 at position c2 PLUS one OF grid EQUALS zero AND the tuple of r1 c1 PLUS one r2 c2 PLUS one NOT IN visited  
                ADD the tuple of r1 c1 PLUS one r2 c2 PLUS one TO visited  
                APPEND the tuple of r1 c1 PLUS one r2 c2 PLUS one steps PLUS one TO queue  
            END IF  
            IF r2 PLUS one LESS THAN n AND element at position r1 PLUS one at position c1 OF grid EQUALS zero AND element at position r2 PLUS one at position c2 OF grid EQUALS zero AND the tuple of r1 PLUS one c1 r2 PLUS one c2 NOT IN visited  
                ADD the tuple of r1 PLUS one c1 r2 PLUS one c2 TO visited  
                APPEND the tuple of r1 PLUS one c1 r2 PLUS one c2 steps PLUS one TO queue  
            END IF  
            IF r1 EQUALS r2 AND r1 PLUS one LESS THAN n AND element at position r1 PLUS one at position c1 OF grid EQUALS zero AND element at position r1 PLUS one at position c2 OF grid EQUALS zero AND the tuple of r1 c1 r1 PLUS one c1 NOT IN visited  
                ADD the tuple of r1 c1 r1 PLUS one c1 TO visited  
                APPEND the tuple of r1 c1 r1 PLUS one c1 steps PLUS one TO queue  
            END IF  
            IF c1 EQUALS c2 AND c1 PLUS one LESS THAN n AND element at position r1 at position c1 PLUS one OF grid EQUALS zero AND element at position r2 at position c2 PLUS one OF grid EQUALS zero AND the tuple of r1 c1 r1 c1 PLUS one NOT IN visited  
                ADD the tuple of r1 c1 r1 c1 PLUS one TO visited  
                APPEND the tuple of r1 c1 r1 c1 PLUS one steps PLUS one TO queue  
            END IF  
        END WHILE  
        RETURN negative one  
    END FUNCTION  
END CLASS
