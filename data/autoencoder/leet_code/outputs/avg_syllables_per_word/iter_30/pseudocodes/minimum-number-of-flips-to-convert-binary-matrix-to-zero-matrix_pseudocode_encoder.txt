CLASS Solution  
    FUNCTION minFlips(mat)  
        FUNCTION flip(matrix, x, y)  
            SET directions TO list of pairs negative one zero, one zero, zero negative one, zero one, zero zero  
            FOR each pair dx dy IN directions  
                SET nx TO x PLUS dx  
                SET ny TO y PLUS dy  
                IF nx GREATER THAN OR EQUAL TO zero AND nx LESS THAN m AND ny GREATER THAN OR EQUAL TO zero AND ny LESS THAN n  
                    SET element at position nx at position ny OF matrix TO one MINUS element at position nx at position ny OF matrix  
                END IF  
            END FOR  
        END FUNCTION  
        
        FUNCTION matrix_to_tuple(matrix)  
            RETURN tuple of tuples formed by each row in matrix  
        END FUNCTION  
        
        SET m TO the LENGTH OF mat  
        SET n TO the LENGTH OF element at position zero OF mat  
        SET target TO tuple of tuples of zero repeated n times repeated m times  
        SET initial_state TO the RESULT OF matrix_to_tuple CALLED WITH mat  
        SET queue TO list containing pair initial_state and zero  
        SET visited TO empty set  
        ADD initial_state TO visited  
        
        WHILE queue is not empty  
            SET current_state and steps TO the first element of queue REMOVED  
            IF current_state EQUALS target  
                RETURN steps  
            END IF  
            
            SET current_matrix TO list formed by lists converted from each row of current_state  
            
            FOR i FROM zero TO m MINUS one  
                FOR j FROM zero TO n MINUS one  
                    SET next_matrix TO DEEP COPY OF current_matrix  
                    CALL flip WITH next_matrix, i, j  
                    SET next_state TO RESULT OF matrix_to_tuple CALLED WITH next_matrix  
                    IF next_state NOT IN visited  
                        ADD next_state TO visited  
                        APPEND pair next_state and steps PLUS one TO queue  
                    END IF  
                END FOR  
            END FOR  
        END WHILE  
        
        RETURN minus one  
    END FUNCTION  
END CLASS
