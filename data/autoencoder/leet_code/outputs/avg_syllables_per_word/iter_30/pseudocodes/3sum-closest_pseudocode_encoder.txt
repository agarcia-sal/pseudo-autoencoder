CLASS Solution  
    FUNCTION threeSumClosest(nums, target)  
        SORT nums IN ascending order  
        SET closest_sum TO positive infinity  
        FOR index i FROM zero TO the LENGTH OF nums MINUS two MINUS one  
            SET left TO i PLUS one  
            SET right TO the LENGTH OF nums MINUS one  
            WHILE left LESS THAN right  
                SET current_sum TO element at position i OF nums PLUS element at position left OF nums PLUS element at position right OF nums  
                IF absolute value of current_sum MINUS target LESS THAN absolute value of closest_sum MINUS target  
                    SET closest_sum TO current_sum  
                END IF  
                IF current_sum LESS THAN target  
                    INCREMENT left BY one  
                ELSE IF current_sum GREATER THAN target  
                    DECREMENT right BY one  
                ELSE  
                    RETURN current_sum  
                END IF  
            END WHILE  
        END FOR  
        RETURN closest_sum  
    END FUNCTION  
END CLASS
