CLASS Solution  
    FUNCTION subarraySum(nums, k)  
        SET cumulative_sum_freq TO dictionary with key zero associated with value one  
        SET current_sum TO zero  
        SET count TO zero  
        FOR each num IN nums  
            INCREMENT current_sum BY num  
            IF the difference of current_sum MINUS k is a key IN cumulative_sum_freq  
                INCREMENT count BY the value associated with the key of the difference of current_sum MINUS k IN cumulative_sum_freq  
            END IF  
            IF current_sum is a key IN cumulative_sum_freq  
                INCREMENT the value associated with the key current_sum IN cumulative_sum_freq BY one  
            ELSE  
                SET the value associated with the key current_sum IN cumulative_sum_freq TO one  
            END IF  
        END FOR  
        RETURN count  
    END FUNCTION  
END CLASS
