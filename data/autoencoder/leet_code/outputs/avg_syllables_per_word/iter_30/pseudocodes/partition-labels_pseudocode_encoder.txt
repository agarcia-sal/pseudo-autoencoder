CLASS Solution  
    FUNCTION partitionLabels WITH PARAMETERS s  
        SET last_occurrence TO empty collection mapping characters to positions  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position index OF s  
            ASSIGN index TO last_occurrence AT character  
        END FOR  
        SET start TO zero  
        SET end TO zero  
        SET partition_sizes TO empty list  
        FOR index FROM zero TO the LENGTH OF s MINUS one  
            SET character TO element at position index OF s  
            IF end LESS THAN last_occurrence AT character THEN  
                SET end TO last_occurrence AT character  
            END IF  
            IF index EQUALS end THEN  
                SET partition_size TO end MINUS start PLUS one  
                APPEND partition_size TO partition_sizes  
                SET start TO index PLUS one  
            END IF  
        END FOR  
        RETURN partition_sizes  
    END FUNCTION  
END CLASS
