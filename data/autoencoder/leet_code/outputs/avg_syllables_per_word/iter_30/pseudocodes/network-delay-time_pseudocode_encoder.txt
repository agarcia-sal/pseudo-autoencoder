CLASS Solution  
    FUNCTION networkDelayTime(times, n, k)  
        SET graph TO a new dictionary mapping each integer from one TO n inclusive TO an empty list  
        FOR each element u, v, w IN times  
            APPEND pair of v and w TO the list at key u in graph  
        END FOR  

        SET dist TO a new dictionary mapping each integer from one TO n inclusive TO infinity  
        SET the value at key k in dist TO zero  

        SET min_heap TO a list containing the pair of zero and k  

        WHILE min_heap is not empty  
            REMOVE and RETURN the pair current_dist and node with smallest first element FROM min_heap  

            IF current_dist IS GREATER THAN the value at key node in dist  
                CONTINUE  
            END IF  

            FOR each pair neighbor and weight IN the list at key node in graph  
                SET distance TO current_dist PLUS weight  

                IF distance IS LESS THAN the value at key neighbor in dist  
                    SET the value at key neighbor in dist TO distance  
                    INSERT the pair distance and neighbor INTO min_heap maintaining heap property  
                END IF  
            END FOR  
        END WHILE  

        SET max_time TO the maximum value among all values in dist  

        IF max_time IS EQUALS TO infinity  
            RETURN minus one  
        ELSE  
            RETURN max_time  
        END IF  
    END FUNCTION  
END CLASS
