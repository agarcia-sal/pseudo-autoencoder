CLASS Solution  
    FUNCTION checkInclusion(s1, s2)  
        SET length of s1 TO len_s1  
        SET length of s2 TO len_s2  
        IF len_s1 GREATER THAN len_s2  
            RETURN false  
        END IF  
        SET count_s1 TO the frequency count of each character in s1  
        SET count_s2 TO the frequency count of each character in the substring from position zero TO position len_s1 of s2  
        IF count_s1 EQUALS count_s2  
            RETURN true  
        END IF  
        FOR index FROM len_s1 TO len_s2 MINUS one  
            INCREMENT the count of character at position index OF s2 in count_s2 BY one  
            DECREMENT the count of character at position index MINUS len_s1 OF s2 in count_s2 BY one  
            IF the count of character at position index MINUS len_s1 OF s2 in count_s2 EQUALS zero  
                REMOVE character at position index MINUS len_s1 OF s2 FROM count_s2  
            END IF  
            IF count_s1 EQUALS count_s2  
                RETURN true  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
