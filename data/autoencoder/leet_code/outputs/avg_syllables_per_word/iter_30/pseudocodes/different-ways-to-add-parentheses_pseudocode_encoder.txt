CLASS Solution  
    FUNCTION diffWaysToCompute(expression)  
        FUNCTION compute(left, right, operator)  
            SET results TO empty list  
            FOR each l IN left  
                FOR each r IN right  
                    IF operator EQUALS plus sign  
                        APPEND l PLUS r TO results  
                    ELSE IF operator EQUALS minus sign  
                        APPEND l MINUS r TO results  
                    ELSE IF operator EQUALS multiplication sign  
                        APPEND l MULTIPLIED BY r TO results  
                    END IF  
                END FOR  
            END FOR  
            RETURN results  
        END FUNCTION  

        FUNCTION helper(sub_expr)  
            IF sub_expr CONSISTS ONLY OF digits  
                RETURN list containing integer conversion of sub_expr  
            END IF  
            SET results TO empty list  
            FOR index FROM zero TO length of sub_expr MINUS one  
                IF character at position index OF sub_expr IS one of plus sign OR minus sign OR multiplication sign  
                    SET left_results TO helper substring from position zero TO position index MINUS one OF sub_expr  
                    SET right_results TO helper substring from position index plus one TO end OF sub_expr  
                    EXTEND results BY compute left_results AND right_results AND character at position index OF sub_expr  
                END IF  
            END FOR  
            RETURN results  
        END FUNCTION  

        RETURN helper(expression)  
    END FUNCTION  
END CLASS
