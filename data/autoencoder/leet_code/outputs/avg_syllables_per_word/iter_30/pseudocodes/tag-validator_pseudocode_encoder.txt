CLASS Solution  
    FUNCTION isValid(code)  
        IMPORT regular expression module AS re  
        SET cdata_pattern TO string representing CDATA regular expression  
        SET code TO result of replacing all occurrences matching cdata_pattern in code with empty string  
        SET stack TO empty list  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF code  
            IF character at position i OF code EQUALS less than character  
                SET j TO position of greater than character FOUND in code STARTING FROM position i PLUS one  
                IF j EQUALS negative one  
                    RETURN false  
                END IF  
                SET tag_content TO substring from position i PLUS one TO position j MINUS one OF code  
                IF tag_content IS empty  
                    RETURN false  
                END IF  
                IF character at position zero OF tag_content EQUALS slash character  
                    SET tag_name TO substring from position one TO end OF tag_content  
                    IF stack IS empty OR last element in stack NOT EQUALS tag_name  
                        RETURN false  
                    END IF  
                    REMOVE last element FROM stack  
                ELSE IF character at position zero OF tag_content EQUALS exclamation character  
                    RETURN false  
                ELSE  
                    IF length of tag_content LESS THAN one OR length of tag_content GREATER THAN nine OR tag_content IS NOT all uppercase letters  
                        RETURN false  
                    END IF  
                    APPEND tag_content TO stack  
                END IF  
                SET i TO j PLUS one  
            ELSE  
                INCREMENT i BY one  
            END IF  
        END WHILE  
        SET pattern TO string representing opening tag regular expression with uppercase tags  
        RETURN stack IS empty AND number of matches FOUND in code USING pattern GREATER THAN zero  
    END FUNCTION  
END CLASS
