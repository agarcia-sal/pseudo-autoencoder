CLASS Solution  
    FUNCTION canConvert(str1, str2)  
        IF str1 EQUALS str2  
            RETURN true  
        END IF  
        SET conversion_map TO an empty mapping  
        FOR each pair of char1 and char2 in corresponding positions of str1 and str2  
            IF char1 EXISTS in conversion_map  
                IF value mapped to char1 in conversion_map NOT EQUALS char2  
                    RETURN false  
                END IF  
            ELSE  
                SET value mapped to char1 in conversion_map TO char2  
            END IF  
        END FOR  
        IF the number of unique characters in str2 EQUALS twenty six  
            RETURN false  
        END IF  
        RETURN true  
    END FUNCTION  
END CLASS
