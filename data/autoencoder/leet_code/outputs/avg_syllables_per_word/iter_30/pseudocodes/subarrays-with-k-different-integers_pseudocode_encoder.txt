CLASS Solution  
    FUNCTION subarraysWithKDistinct(nums, k)  
        FUNCTION atMostKDistinct(t)  
            SET count TO empty mapping with default zero  
            SET left TO zero  
            SET result TO zero  
            FOR right FROM zero TO the LENGTH OF nums MINUS one  
                INCREMENT the value in count for element at position right of nums BY one  
                WHILE the NUMBER OF keys in count GREATER THAN t  
                    DECREMENT the value in count for element at position left of nums BY one  
                    IF the value in count for element at position left of nums EQUALS zero  
                        REMOVE the element at position left of nums FROM count  
                    END IF  
                    INCREMENT left BY one  
                END WHILE  
                INCREMENT result BY (right MINUS left PLUS one)  
            END FOR  
            RETURN result  
        END FUNCTION  
        RETURN the result of atMostKDistinct with argument k MINUS the result of atMostKDistinct with argument k MINUS one  
    END FUNCTION  
END CLASS
