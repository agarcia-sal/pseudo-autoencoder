CLASS Solution  
    FUNCTION crackSafe(n, k)  
        SET visited TO empty set  
        SET sequence TO empty list  
        
        FUNCTION dfs(node)  
            FOR i FROM zero TO k MINUS one  
                SET edge TO node concatenated with string form of i  
                IF edge NOT IN visited THEN  
                    ADD edge TO visited  
                    CALL dfs with substring from position one TO end OF edge  
                    APPEND string form of i TO sequence  
                END IF  
            END FOR  
        END FUNCTION  
        
        SET start_node TO string of zero repeated n MINUS one times  
        CALL dfs with start_node  
        
        SET final_sequence TO concatenation of elements in sequence in reverse order followed by start_node  
        RETURN final_sequence  
    END FUNCTION  
END CLASS
