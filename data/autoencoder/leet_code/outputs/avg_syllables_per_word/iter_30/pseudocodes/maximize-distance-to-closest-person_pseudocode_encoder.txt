CLASS Solution  
    FUNCTION maxDistToClosest(seats)  
        SET max_distance TO zero  
        SET start TO zero  
        SET end TO zero  
        SET n TO the LENGTH OF seats  
        
        WHILE start LESS THAN n AND element at position start OF seats EQUALS zero  
            INCREMENT start BY one  
        END WHILE  
        
        SET end TO n MINUS one  
        WHILE end GREATER THAN OR EQUAL TO zero AND element at position end OF seats EQUALS zero  
            DECREMENT end BY one  
        END WHILE  
        
        IF max_distance LESS THAN start  
            SET max_distance TO start  
        END IF  
        
        SET distance_from_end TO n MINUS end MINUS one  
        IF max_distance LESS THAN distance_from_end  
            SET max_distance TO distance_from_end  
        END IF  
        
        SET current_distance TO zero  
        FOR i FROM start TO end  
            IF element at position i OF seats EQUALS zero  
                INCREMENT current_distance BY one  
                SET candidate_distance TO current_distance PLUS one DIVIDED BY two  
                IF max_distance LESS THAN candidate_distance  
                    SET max_distance TO candidate_distance  
                END IF  
            ELSE  
                SET current_distance TO zero  
            END IF  
        END FOR  
        
        RETURN max_distance  
    END FUNCTION  
END CLASS
