CLASS Solution  
    FUNCTION numberOfArrays(s, k)  
        SET MOD TO one billion plus seven  
        SET n TO the LENGTH OF s  
        SET dp TO a list of zeros with length n plus one  
        SET the element at position zero OF dp TO one  
        FOR i FROM one TO n  
            FOR j FROM zero TO i minus one  
                IF the character at position j OF s NOT EQUALS zero AS character AND the integer value of substring from position j TO position i OF s IS GREATER THAN OR EQUAL TO one AND IS LESS THAN OR EQUAL TO k  
                    SET the element at position i OF dp TO the sum of the element at position i OF dp PLUS the element at position j OF dp THEN TAKE MODULO MOD  
                END IF  
            END FOR  
        END FOR  
        RETURN the element at position n OF dp  
    END FUNCTION  
END CLASS
