CLASS Solution  
    FUNCTION largestRectangleArea(heights)  
        SET stack TO a list containing negative one  
        SET max_area TO zero  
        APPEND zero TO heights  
        FOR each i and h IN the enumeration of heights  
            WHILE the element at position negative one OF stack NOT EQUALS negative one AND h LESS THAN the element at position element at position negative one OF stack OF heights  
                SET height TO the element at position element removed from the end OF stack OF heights  
                SET width TO i MINUS the element at position negative one OF stack MINUS one  
                SET max_area TO the maximum value BETWEEN max_area AND height MULTIPLIED BY width  
            END WHILE  
            APPEND i TO stack  
        END FOR  
        RETURN max_area  
    END FUNCTION  
END CLASS
