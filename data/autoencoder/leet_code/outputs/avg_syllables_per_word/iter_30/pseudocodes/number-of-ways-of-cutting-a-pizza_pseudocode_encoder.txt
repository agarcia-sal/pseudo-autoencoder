CLASS Solution  
    FUNCTION ways(pizza, k)  
        FUNCTION dfs(i, j, k)  
            IF k EQUALS zero  
                RETURN one IF total apples in subrectangle defined by i, j, m, n IS GREATER THAN zero ELSE zero  
            END IF  
            SET ans TO zero  
            FOR x FROM i PLUS one TO m MINUS one  
                IF total apples in subrectangle defined by x, j, m, n IS GREATER THAN zero  
                    INCREMENT ans BY dfs(x, j, k MINUS one)  
                END IF  
            END FOR  
            FOR y FROM j PLUS one TO n MINUS one  
                IF total apples in subrectangle defined by i, y, m, n IS GREATER THAN zero  
                    INCREMENT ans BY dfs(i, y, k MINUS one)  
                END IF  
            END FOR  
            RETURN ans MODULO mod  
        END FUNCTION  

        SET mod TO ten POWER nine PLUS seven  
        SET m TO LENGTH OF pizza  
        SET n TO LENGTH OF element at position zero OF pizza  
        INITIALIZE s AS two dimensional list of zero WITH dimensions m PLUS one BY n PLUS one  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                SET s at position i, j TO s at position i MINUS one, j PLUS s at position i, j MINUS one MINUS s at position i MINUS one, j MINUS one PLUS one IF element at position j MINUS one OF element at position i MINUS one OF pizza EQUALS letter A ELSE zero  
            END FOR  
        END FOR  
        RETURN dfs(zero, zero, k MINUS one)  
    END FUNCTION  
END CLASS
