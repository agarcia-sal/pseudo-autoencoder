CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO an empty set  
        FOR each mine IN mines  
            ADD tuple consisting of mine elements TO mines_set  
        END FOR  
        
        SET dp TO a two dimensional list of size n by n where each element is a list of four zeros  
        
        SET directions TO list of coordinate pairs zero one, one zero, zero negative one, negative one zero  
        
        FOR i FROM zero TO n minus one  
            FOR j FROM zero TO n minus one  
                IF tuple consisting of i and j NOT IN mines_set THEN  
                    IF j GREATER THAN zero THEN  
                        SET dp element at position i j zero TO dp element at position i j minus one zero PLUS one  
                    END IF  
                    IF i GREATER THAN zero THEN  
                        SET dp element at position i j one TO dp element at position i minus one j one PLUS one  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        SET result TO zero  
        FOR i FROM n minus one DOWN TO zero  
            FOR j FROM n minus one DOWN TO zero  
                IF tuple consisting of i and j NOT IN mines_set THEN  
                    IF j LESS THAN n minus one THEN  
                        SET dp element at position i j two TO dp element at position i j plus one two PLUS one  
                    END IF  
                    IF i LESS THAN n minus one THEN  
                        SET dp element at position i j three TO dp element at position i plus one j three PLUS one  
                    END IF  
                    SET plus_sign_order TO minimum value among elements at dp element at position i j  
                    IF plus_sign_order GREATER THAN result THEN  
                        SET result TO plus_sign_order  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        RETURN result  
    END FUNCTION  
END CLASS
