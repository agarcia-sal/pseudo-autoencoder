CLASS Solution  
    FUNCTION canPartitionKSubsets(nums, k)  
        SET total_sum TO the sum of nums  
        IF the remainder of total_sum divided by k is not equal to zero  
            RETURN false  
        END IF  
        SET target TO total_sum divided by k  
        SORT nums in descending order  
        
        FUNCTION can_partition(index, k_remaining, current_sum, used)  
            IF k_remaining equals zero  
                RETURN true  
            END IF  
            IF current_sum equals target  
                RETURN can_partition zero, k_remaining minus one, zero, and used  
            END IF  
            
            FOR i FROM index TO the length of nums minus one  
                IF element at position i of used is false AND current_sum plus element at position i of nums is less than or equal to target  
                    SET element at position i of used TO true  
                    IF can_partition i plus one, k_remaining, current_sum plus element at position i of nums, and used returns true  
                        RETURN true  
                    END IF  
                    SET element at position i of used TO false  
                END IF  
            END FOR  
            RETURN false  
        END FUNCTION  
        
        SET used TO a list of false with length equal to the length of nums  
        RETURN can_partition zero, k, zero, and used  
    END FUNCTION  
END CLASS
