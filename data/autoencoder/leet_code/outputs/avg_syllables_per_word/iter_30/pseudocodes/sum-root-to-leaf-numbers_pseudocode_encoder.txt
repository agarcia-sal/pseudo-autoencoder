CLASS TreeNode  
    FUNCTION __init__(self, val equals zero, left equals none, right equals none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with val equal to element at position zero OF values  
    SET i TO one  
    CREATE a new empty double ended queue named queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO element removed from front of queue  
        IF i LESS THAN length of values AND element at position i OF values is not none  
            SET node dot left TO new TreeNode with val equal to element at position i OF values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN length of values AND element at position i OF values is not none  
            SET node dot right TO new TreeNode with val equal to element at position i OF values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p dot left, q dot left) AND is_same_tree(p dot right, q dot right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION sumNumbers(self, root)  
        FUNCTION dfs(node, current_number)  
            IF node EQUALS none  
                RETURN zero  
            END IF  
            SET current_number TO current_number MULTIPLIED BY ten PLUS node dot val  
            IF node dot left EQUALS none AND node dot right EQUALS none  
                RETURN current_number  
            END IF  
            SET left_sum TO dfs(node dot left, current_number)  
            SET right_sum TO dfs(node dot right, current_number)  
            RETURN left_sum PLUS right_sum  
        END FUNCTION  
        RETURN dfs(root, zero)  
    END FUNCTION  
END CLASS
