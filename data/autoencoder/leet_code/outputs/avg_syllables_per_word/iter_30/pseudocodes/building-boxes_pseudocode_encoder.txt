CLASS Solution  
    FUNCTION minimumBoxes(n)  
        FUNCTION triangular_pyramid(k)  
            RETURN k MULTIPLIED BY the sum of k and one MULTIPLIED BY the sum of k and two DIVIDED BY six  
        END FUNCTION  

        SET left TO zero  
        SET right TO n  
        WHILE left LESS THAN right  
            SET mid TO the sum of left and right and one DIVIDED BY two  
            IF triangular_pyramid(mid) LESS THAN OR EQUAL TO n  
                SET left TO mid  
            ELSE  
                SET right TO the difference of mid and one  
            END IF  
        END WHILE  

        SET boxes_used TO triangular_pyramid(left)  
        SET remaining TO the difference of n and boxes_used  
        SET floor_boxes TO left MULTIPLIED BY the sum of left and one DIVIDED BY two  

        SET additional_boxes TO zero  
        WHILE remaining GREATER THAN zero  
            INCREMENT additional_boxes BY one  
            DECREMENT remaining BY additional_boxes  
        END WHILE  

        RETURN the sum of floor_boxes and additional_boxes  
    END FUNCTION  
END CLASS
