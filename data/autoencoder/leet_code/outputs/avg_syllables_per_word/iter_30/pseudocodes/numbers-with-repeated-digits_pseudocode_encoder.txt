CLASS Solution  
    FUNCTION numDupDigitsAtMostN(n)  
        FUNCTION count_unique_digits(x)  
            SET s TO the string representation of x  
            SET length TO the number of characters in s  
            SET count TO zero  
            FOR i FROM one TO length MINUS one  
                INCREMENT count BY nine MULTIPLIED BY perm of nine AND i MINUS one  
            END FOR  
            SET seen TO an empty set  
            FOR each index i AND character digit IN s  
                FOR d FROM zero TO digit converted to integer MINUS one  
                    IF i EQUALS zero THEN  
                        SET start_d TO one  
                    ELSE  
                        SET start_d TO zero  
                    END IF  
                    IF d GREATER THAN OR EQUAL TO start_d AND d NOT IN seen THEN  
                        INCREMENT count BY perm of nine MINUS i AND length MINUS i MINUS one  
                    END IF  
                END FOR  
                IF digit converted to integer IN seen THEN  
                    BREAK  
                END IF  
                ADD digit converted to integer TO seen  
            END FOR  
            IF loop completed without BREAK THEN  
                INCREMENT count BY one  
            END IF  
            RETURN count  
        END FUNCTION  
        RETURN n MINUS count_unique_digits of n  
    END FUNCTION  

    FUNCTION perm(m, n)  
        IF n EQUALS zero THEN  
            RETURN one  
        END IF  
        SET result TO one  
        FOR i FROM zero TO n MINUS one  
            MULTIPLY result BY m MINUS i  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
