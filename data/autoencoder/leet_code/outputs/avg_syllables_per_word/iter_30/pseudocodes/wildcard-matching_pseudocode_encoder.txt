CLASS Solution  
    FUNCTION isMatch(s, p)  
        SET dp TO two dimensional collection of False with the number of rows equal to the length of s plus one and the number of columns equal to the length of p plus one  
        SET element at position zero of row zero of dp TO True  
        FOR j FROM one TO the length of p  
            IF element at position j minus one of p EQUALS star  
                SET element at position zero of column j of dp TO element at position zero of column j minus one of dp  
            END IF  
        END FOR  
        FOR i FROM one TO the length of s  
            FOR j FROM one TO the length of p  
                IF element at position j minus one of p EQUALS star  
                    SET element at position i of column j of dp TO element at position i minus one of column j of dp OR element at position i of column j minus one of dp  
                ELSE IF element at position j minus one of p EQUALS question mark OR element at position i minus one of s EQUALS element at position j minus one of p  
                    SET element at position i of column j of dp TO element at position i minus one of column j minus one of dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position length of s of column length of p of dp  
    END FUNCTION  
END CLASS
