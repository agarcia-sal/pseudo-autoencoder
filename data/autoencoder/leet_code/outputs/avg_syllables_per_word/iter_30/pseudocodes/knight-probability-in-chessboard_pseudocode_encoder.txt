CLASS Solution  
    FUNCTION knightProbability(n, k, row, column)  
        SET moves TO list of pairs consisting of minus two and minus one, minus two and one, minus one and minus two, minus one and two, one and minus two, one and two, two and minus one, two and one  
        SET dp TO two dimensional list of zeros with dimensions n by n  
        SET element at position row and column OF dp TO one  
        FOR each step FROM one TO k  
            SET new_dp TO two dimensional list of zeros with dimensions n by n  
            FOR each r FROM zero TO n MINUS one  
                FOR each c FROM zero TO n MINUS one  
                    IF element at position r and c OF dp EQUALS zero  
                        CONTINUE  
                    END IF  
                    FOR each pair dr and dc IN moves  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr GREATER THAN OR EQUAL TO zero AND nr LESS THAN n AND nc GREATER THAN OR EQUAL TO zero AND nc LESS THAN n  
                            INCREMENT element at position nr and nc OF new_dp BY element at position r and c OF dp DIVIDED BY eight  
                        END IF  
                    END FOR  
                END FOR  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
        SET total_probability TO zero  
        FOR each row IN dp  
            FOR each probability IN row  
                INCREMENT total_probability BY probability  
            END FOR  
        END FOR  
        RETURN total_probability  
    END FUNCTION  
END CLASS
