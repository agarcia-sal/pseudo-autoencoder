CLASS Solution  
    FUNCTION uniqueLetterString(s)  
        SET MOD TO one billion PLUS seven  
        SET n TO the LENGTH OF s  
        SET last TO list of negative one repeated twenty six times  
        SET last2 TO list of negative one repeated twenty six times  
        SET result TO zero  
        FOR each index i AND character char IN s with enumeration  
            SET idx TO Unicode code point of char MINUS Unicode code point of uppercase letter A  
            SET difference_one TO i MINUS element at position idx OF last  
            SET difference_two TO element at position idx OF last MINUS element at position idx OF last2  
            SET product TO difference_one MULTIPLIED BY difference_two  
            SET result TO result PLUS product  
            SET result TO result MODULO MOD  
            SET element at position idx OF last2 TO element at position idx OF last  
            SET element at position idx OF last TO i  
        END FOR  
        FOR each index i FROM zero TO twenty five  
            SET difference_one TO n MINUS element at position i OF last  
            SET difference_two TO element at position i OF last MINUS element at position i OF last2  
            SET product TO difference_one MULTIPLIED BY difference_two  
            SET result TO result PLUS product  
            SET result TO result MODULO MOD  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
