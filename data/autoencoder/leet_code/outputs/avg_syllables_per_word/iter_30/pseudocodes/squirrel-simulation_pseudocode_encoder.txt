CLASS Solution  
    FUNCTION minDistance(height, width, tree, squirrel, nuts)  
        FUNCTION distance(p1, p2)  
            SET horizontal_difference TO absolute value of element at position zero OF p1 MINUS element at position zero OF p2  
            SET vertical_difference TO absolute value of element at position one OF p1 MINUS element at position one OF p2  
            SET total_distance TO horizontal_difference PLUS vertical_difference  
            RETURN total_distance  
        END FUNCTION  
        
        SET total_distance TO zero  
        SET max_save TO negative infinity  
        
        FOR each nut IN nuts  
            INCREMENT total_distance BY two MULTIPLIED BY distance of nut AND tree  
            SET first_nut_distance TO distance of squirrel AND nut PLUS distance of nut AND tree  
            SET later_nut_distance TO two MULTIPLIED BY distance of nut AND tree  
            SET save TO later_nut_distance MINUS first_nut_distance  
            IF save GREATER THAN max_save  
                SET max_save TO save  
            END IF  
        END FOR  
        
        SET result_distance TO total_distance MINUS max_save  
        RETURN result_distance  
    END FUNCTION  
END CLASS
