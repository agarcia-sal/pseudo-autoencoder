CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS None, right EQUALS None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS empty  
        RETURN None  
    END IF  
    SET root TO a new TreeNode with value as element at position zero OF values  
    SET i TO one  
    SET queue TO an empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the first element removed FROM queue  
        IF i LESS THAN the length of values AND element at position i OF values NOT EQUALS None  
            SET node.left TO a new TreeNode with value as element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i OF values NOT EQUALS None  
            SET node.right TO a new TreeNode with value as element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS None AND q EQUALS None  
        RETURN True  
    ELSE IF p EQUALS None OR q EQUALS None  
        RETURN False  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN False  
    ELSE  
        RETURN is_same_tree(left child of p, left child of q) AND is_same_tree(right child of p, right child of q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION findBottomLeftValue(self, root)  
        SET queue TO a new double ended queue initialized with root  
        SET leftmost_value TO None  
        WHILE queue is not empty  
            SET level_size TO the length of queue  
            FOR i FROM zero TO level_size MINUS one  
                SET node TO the first element removed FROM queue  
                IF i EQUALS zero  
                    SET leftmost_value TO the value of node  
                END IF  
                IF left child of node NOT EQUALS None  
                    APPEND left child of node TO queue  
                END IF  
                IF right child of node NOT EQUALS None  
                    APPEND right child of node TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN leftmost_value  
    END FUNCTION  
END CLASS
