CLASS Solution  
    FUNCTION findAllPeople(n, meetings, firstPerson)  
        SET know_secret TO set containing zero and firstPerson  
        SORT meetings BY time element  
        SET i TO zero  
        WHILE i LESS THAN the LENGTH OF meetings  
            SET current_time TO time element at position i OF meetings  
            CREATE empty graph mapping from person TO list of connected persons  
            WHILE i LESS THAN the LENGTH OF meetings AND time element at position i OF meetings EQUALS current_time  
                SET x TO first person element at position i OF meetings  
                SET y TO second person element at position i OF meetings  
                APPEND y TO the list for x IN graph  
                APPEND x TO the list for y IN graph  
                INCREMENT i BY one  
            END WHILE  
            CREATE queue WITH persons IN know_secret THAT ARE keys IN graph  
            CREATE empty visited set  
            WHILE queue is not empty  
                REMOVE first person FROM queue and ASSIGN TO person  
                IF person IS IN visited  
                    CONTINUE  
                END IF  
                ADD person TO visited  
                ADD person TO know_secret  
                FOR each neighbor IN neighbors of person IN graph  
                    IF neighbor NOT IN visited  
                        APPEND neighbor TO queue  
                    END IF  
                END FOR  
            END WHILE  
        END WHILE  
        RETURN list converted from know_secret  
    END FUNCTION  
END CLASS
