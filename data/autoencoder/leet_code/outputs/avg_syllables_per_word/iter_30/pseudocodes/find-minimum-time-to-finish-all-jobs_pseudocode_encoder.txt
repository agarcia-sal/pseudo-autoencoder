CLASS Solution  
    FUNCTION minimumTimeRequired(jobs, k)  
        SET n TO the LENGTH OF jobs  
        SET subset_sums TO a list of zero repeated TWO RAISED TO THE POWER OF n times  
        FOR each i FROM zero TO n MINUS one  
            FOR each mask FROM zero TO TWO RAISED TO THE POWER OF i MINUS one  
                SET the element at position mask OR TWO RAISED TO THE POWER OF i OF subset_sums TO the element at position mask OF subset_sums PLUS the element at position i OF jobs  
            END FOR  
        END FOR  

        FUNCTION dp(k_param, mask)  
            IF k_param EQUALS one  
                RETURN the element at position mask OF subset_sums  
            END IF  
            SET res TO positive infinity  
            SET submask TO mask  
            WHILE submask IS NOT zero  
                SET res TO the SMALLER OF res AND the GREATER OF dp(k_param MINUS one, mask XOR submask) AND the element at position submask OF subset_sums  
                SET submask TO submask MINUS one AND mask  
            END WHILE  
            RETURN res  
        END FUNCTION  

        RETURN dp(k, TWO RAISED TO THE POWER OF n MINUS one)  
    END FUNCTION  
END CLASS
