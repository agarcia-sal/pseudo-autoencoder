CLASS Solution  
    FUNCTION lengthOfLongestSubstringKDistinct(s, k)  
        IF k EQUALS zero  
            RETURN zero  
        END IF  
        SET char_map TO empty collection  
        SET left TO zero  
        SET max_length TO zero  
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            ASSIGN reference TO right TO element at position right OF s IN char_map  
            IF the LENGTH OF char_map GREATER THAN k  
                SET leftmost_char TO element with minimum associated value IN char_map  
                SET left TO value associated with leftmost_char IN char_map PLUS one  
                REMOVE leftmost_char FROM char_map  
            END IF  
            SET max_length TO greater value BETWEEN max_length AND right MINUS left PLUS one  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
