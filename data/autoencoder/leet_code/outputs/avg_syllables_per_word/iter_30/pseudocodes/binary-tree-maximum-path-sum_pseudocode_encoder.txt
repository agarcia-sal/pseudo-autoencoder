CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self dot val TO val  
        SET self dot left TO left  
        SET self dot right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values EQUALS none OR LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO NEW TreeNode WITH val EQUALS element at position zero OF values  
    SET i TO one  
    SET queue TO an empty deque  
    APPEND root TO queue  
    WHILE LENGTH OF queue IS GREATER THAN zero  
        SET node TO REMOVE first element FROM queue  
        IF i LESS THAN LENGTH OF values AND element at position i OF values NOT EQUALS none  
            SET node dot left TO NEW TreeNode WITH val EQUALS element at position i OF values  
            APPEND node dot left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN LENGTH OF values AND element at position i OF values NOT EQUALS none  
            SET node dot right TO NEW TreeNode WITH val EQUALS element at position i OF values  
            APPEND node dot right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p dot val NOT EQUALS q dot val  
        RETURN false  
    ELSE  
        RETURN is_same_tree(p dot left, q dot left) AND is_same_tree(p dot right, q dot right)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION maxPathSum(self, root)  
        FUNCTION helper(node)  
            IF node EQUALS none  
                RETURN negative infinity, negative infinity  
            END IF  
            SET left_max, left_gain TO helper(node dot left)  
            SET right_max, right_gain TO helper(node dot right)  
            SET current_max TO node dot val PLUS the maximum OF left_gain AND zero PLUS the maximum OF right_gain AND zero  
            SET current_gain TO node dot val PLUS the maximum OF left_gain, right_gain, AND zero  
            SET global_max TO the maximum OF left_max, right_max, AND current_max  
            RETURN global_max, current_gain  
        END FUNCTION  
        RETURN the first element OF helper(root)  
    END FUNCTION  
END CLASS
