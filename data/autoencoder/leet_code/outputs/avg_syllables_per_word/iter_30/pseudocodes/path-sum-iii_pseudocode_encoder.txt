CLASS TreeNode  
    FUNCTION __init__(self, val, left, right)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value at position zero OF values and left none and right none  
    SET i TO one  
    SET queue TO new empty double ended queue  
    APPEND root TO queue  
    WHILE queue is not empty  
        SET node TO the element removed from the left of queue  
        IF i LESS THAN the length of values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with value at position i OF values and left none and right none  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with value at position i OF values and left none and right none  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value at property val OF p NOT EQUALS value at property val OF q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(value at property left OF p, value at property left OF q) AND is_same_tree(value at property right OF p, value at property right OF q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION pathSum(self, root, targetSum)  
        FUNCTION helper(node, current_sum, prefix_sums)  
            IF node IS none  
                RETURN zero  
            END IF  
            SET current_sum TO current_sum PLUS value at property val OF node  
            SET difference TO current_sum MINUS targetSum  
            SET paths_to_here TO value for key difference IN prefix_sums OR zero if key does not exist  
            SET value for key current_sum IN prefix_sums TO value for key current_sum IN prefix_sums OR zero if key does not exist PLUS one  
            SET paths_in_left TO helper(value at property left OF node, current_sum, prefix_sums)  
            SET paths_in_right TO helper(value at property right OF node, current_sum, prefix_sums)  
            DECREMENT value for key current_sum IN prefix_sums BY one  
            RETURN paths_to_here PLUS paths_in_left PLUS paths_in_right  
        END FUNCTION  
        SET prefix_sums TO a dictionary WITH key zero AND value one  
        RETURN helper(root, zero, prefix_sums)  
    END FUNCTION  
END CLASS
