CLASS Solution  
    FUNCTION characterReplacement(s, k)  
        SET max_length TO zero  
        SET max_count TO zero  
        SET left TO zero  
        ASSIGN reference TO empty mapping with default zero FOR count  
        
        FOR right FROM zero TO the LENGTH OF s MINUS one  
            INCREMENT count at element at position right OF s BY one  
            SET max_count TO maximum of max_count AND count at element at position right OF s  
            
            IF the value of right MINUS left PLUS one MINUS max_count GREATER THAN k THEN  
                DECREMENT count at element at position left OF s BY one  
                INCREMENT left BY one  
            END IF  
            
            SET max_length TO maximum of max_length AND the value of right MINUS left PLUS one  
        END FOR  
        
        RETURN max_length  
    END FUNCTION  
END CLASS
