CLASS Solution  
    FUNCTION findSubstring(s, words)  
        IF s EQUALS empty OR words EQUALS empty OR element at position zero OF words EQUALS empty  
            RETURN empty collection  
        END IF  
        SET word_length TO the LENGTH OF element at position zero OF words  
        SET num_words TO the LENGTH OF words  
        SET total_length TO word_length MULTIPLIED BY num_words  
        SET word_count TO the COUNT OF each element IN words  
        SET result TO empty collection  
        FOR index i FROM zero TO the LENGTH OF s MINUS total_length PLUS one  
            SET seen TO empty count mapping  
            FOR index j FROM i TO i PLUS total_length MINUS one STEP word_length  
                SET word TO the substring from position j TO position j PLUS word_length OF s  
                INCREMENT the count of word IN seen BY one  
                IF the count of word IN seen GREATER THAN the count of word IN word_count  
                    BREAK out of the inner loop  
                END IF  
            END FOR  
            IF the inner loop was not broken  
                APPEND i TO result  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
