CLASS Solution  
    FUNCTION findTargetSumWays(nums, target)  
        FUNCTION dfs(index, current_sum, memo)  
            IF index EQUALS the LENGTH OF nums  
                IF current_sum EQUALS target  
                    RETURN one  
                ELSE  
                    RETURN zero  
                END IF  
            END IF  
            IF memo CONTAINS key composed of index AND current_sum  
                RETURN value at key composed of index AND current_sum IN memo  
            END IF  
            SET positive TO dfs with arguments index PLUS one, current_sum PLUS element at position index OF nums, memo  
            SET negative TO dfs with arguments index PLUS one, current_sum MINUS element at position index OF nums, memo  
            SET value at key composed of index AND current_sum IN memo TO positive PLUS negative  
            RETURN value at key composed of index AND current_sum IN memo  
        END FUNCTION  
        RETURN dfs with arguments zero, zero, empty collection  
    END FUNCTION  
END CLASS
