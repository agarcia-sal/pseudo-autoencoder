CLASS Solution  
    FUNCTION isPossible(nums)  
        ASSIGN reference TO a counter of nums AS num_count  
        ASSIGN reference TO a counter of empty collection AS end_with  
        FOR each num IN nums  
            IF the value at num in num_count EQUALS zero  
                CONTINUE  
            END IF  
            IF the value at num MINUS one in end_with GREATER THAN zero  
                DECREMENT the value at num MINUS one in end_with BY one  
                INCREMENT the value at num in end_with BY one  
            ELSE IF the value at num PLUS one in num_count GREATER THAN zero AND the value at num PLUS two in num_count GREATER THAN zero  
                DECREMENT the value at num PLUS one in num_count BY one  
                DECREMENT the value at num PLUS two in num_count BY one  
                INCREMENT the value at num PLUS two in end_with BY one  
            ELSE  
                RETURN false  
            END IF  
            DECREMENT the value at num in num_count BY one  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
