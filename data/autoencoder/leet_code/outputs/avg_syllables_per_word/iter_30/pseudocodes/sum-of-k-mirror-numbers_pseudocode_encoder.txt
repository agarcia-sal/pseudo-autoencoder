CLASS Solution  
    FUNCTION kMirror(k, n)  
        FUNCTION is_palindrome(s)  
            RETURN s EQUALS reverse of s  
        END FUNCTION  
        
        FUNCTION to_base_k(num, k)  
            IF num EQUALS zero  
                RETURN string zero  
            END IF  
            SET digits TO empty list  
            WHILE num GREATER THAN zero  
                APPEND remainder of num DIVIDED BY k TO digits  
                SET num TO integer division of num BY k  
            END WHILE  
            RETURN concatenation of string forms of digits in reverse order  
        END FUNCTION  
        
        FUNCTION generate_palindromes(length)  
            IF length EQUALS one  
                FOR i FROM one TO nine  
                    YIELD string form of i  
                END FOR  
            ELSE  
                SET start TO ten RAISED TO the power of integer division of length MINUS one BY two  
                SET end TO ten RAISED TO the power of integer division of length PLUS one BY two  
                FOR half FROM start TO one LESS THAN end  
                    SET half_str TO string form of half  
                    IF length MODULO two EQUALS zero  
                        SET full TO concatenation of half_str AND reverse of half_str  
                    ELSE  
                        SET full TO concatenation of half_str AND substring of half_str from position (length of half_str MINUS two) TO position zero in reverse order  
                    END IF  
                    YIELD full  
                END FOR  
            END IF  
        END FUNCTION  
        
        SET k_mirror_numbers TO empty list  
        SET length TO one  
        WHILE LENGTH OF k_mirror_numbers LESS THAN n  
            FOR each pal IN generate_palindromes(length)  
                SET num TO integer form of pal  
                IF is_palindrome(to_base_k(num, k))  
                    APPEND num TO k_mirror_numbers  
                    IF LENGTH OF k_mirror_numbers EQUALS n  
                        BREAK  
                    END IF  
                END IF  
            END FOR  
            INCREMENT length BY one  
        END WHILE  
        
        RETURN sum of elements in k_mirror_numbers  
    END FUNCTION  
END CLASS
