CLASS Solution  
    FUNCTION mincostToHireWorkers(quality, wage, k)  
        SET workers TO the LIST OF pairs of wage at position index DIVIDED BY quality at position index AND quality at position index FOR each index IN the RANGE OF the LENGTH OF wage SORTED BY the FIRST element OF each pair IN ascending order  
        SET max_heap TO an empty LIST  
        SET total_quality TO zero  
        SET min_cost TO positive infinity  
        FOR each pair OF ratio AND q IN workers  
            PUSH the negative of q ONTO max_heap  
            INCREMENT total_quality BY q  
            IF the LENGTH OF max_heap GREATER THAN k  
                REMOVE the smallest element FROM max_heap AND ADD it TO total_quality  
            END IF  
            IF the LENGTH OF max_heap EQUALS k  
                SET min_cost TO the smaller BETWEEN min_cost AND total_quality MULTIPLIED BY ratio  
            END IF  
        END FOR  
        RETURN min_cost  
    END FUNCTION  
END CLASS
