CLASS Solution  
    FUNCTION getSum(a, b)  
        SET maximum TO the maximum thirty two bits integer value  
        SET mask TO the mask for thirty two bits  
        WHILE b is not equal to zero  
            SET carry TO the bitwise AND of a and b shifted left by one position  
            SET a TO the bitwise XOR of a and b AND mask  
            SET b TO carry AND mask  
        END WHILE  
        IF a is less than or equal to maximum  
            RETURN a  
        ELSE  
            RETURN the bitwise complement of the bitwise XOR of a and mask  
        END IF  
    END FUNCTION  
END CLASS
