CLASS Solution  
    FUNCTION groupAnagrams(strs)  
        SET anagrams TO a new empty dictionary with default value as empty list  
        FOR each s IN strs  
            SET sorted_str TO concatenate all characters IN s sorted in ascending order  
            APPEND s TO the list at key sorted_str IN anagrams  
        END FOR  
        SET result TO list of all values IN anagrams  
        RETURN result  
    END FUNCTION  
END CLASS
