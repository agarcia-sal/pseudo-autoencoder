CLASS Solution  
    FUNCTION canCompleteCircuit(gas, cost)  
        IF the SUM OF gas IS LESS THAN the SUM OF cost  
            RETURN the negative one  
        END IF  
        SET start TO zero  
        SET total_tank TO zero  
        FOR each index i FROM zero TO the LENGTH OF gas MINUS one  
            INCREMENT total_tank BY element at position i OF gas MINUS element at position i OF cost  
            IF total_tank IS LESS THAN zero  
                SET start TO i PLUS one  
                SET total_tank TO zero  
            END IF  
        END FOR  
        RETURN start  
    END FUNCTION  
END CLASS
