CLASS Solution  
    FUNCTION findLongestChain WITH pairs  
        SORT pairs BASED ON the element at position one OF each pair  
        SET current_end TO negative infinity  
        SET chain_length TO zero  
        FOR each pair WITH elements start AND end IN pairs  
            IF start IS GREATER THAN current_end  
                SET current_end TO end  
                INCREMENT chain_length BY one  
            END IF  
        END FOR  
        RETURN chain_length  
    END FUNCTION  
END CLASS
