CLASS Solution  
    FUNCTION canJump(nums)  
        SET farthest TO zero  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            IF i GREATER THAN farthest  
                RETURN false  
            END IF  
            SET farthest TO the GREATER OF farthest AND i PLUS element at position i OF nums  
        END FOR  
        RETURN farthest GREATER THAN OR EQUAL TO the LENGTH OF nums MINUS one  
    END FUNCTION  
END CLASS
