CLASS Solution  
    FUNCTION findReplaceString(s, indices, sources, targets)  
        SET replacements TO the list of triples composed of each element from indices, sources, and targets sorted by the first element of each triple  
        SET result TO an empty list  
        SET prev_end TO zero  
        FOR each triple of index, source, target IN replacements  
            APPEND the substring from position prev_end TO position index of s TO result  
            IF the substring from position index TO position index PLUS the length of source of s EQUALS source  
                APPEND target TO result  
            ELSE  
                APPEND the substring from position index TO position index PLUS the length of source of s TO result  
            END IF  
            SET prev_end TO index PLUS the length of source  
        END FOR  
        APPEND the substring from position prev_end TO the end of s TO result  
        RETURN the concatenation of all elements in result  
    END FUNCTION  
END CLASS
