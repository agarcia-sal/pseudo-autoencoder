CLASS Solution  
    FUNCTION numberOfUniqueGoodSubsequences(binary)  
        SET MODULO TO ten raised to the power of nine PLUS one  
        SET countSubsequencesEndingWithZero TO zero  
        SET countSubsequencesEndingWithOne TO zero  
        SET containsZero TO whether the string binary contains the character zero  
        FOR each character bit IN binary  
            IF bit EQUALS one  
                SET countSubsequencesEndingWithOne TO the sum of countSubsequencesEndingWithZero PLUS countSubsequencesEndingWithOne PLUS one MODULO MODULO  
            ELSE  
                SET countSubsequencesEndingWithZero TO the sum of countSubsequencesEndingWithZero PLUS countSubsequencesEndingWithOne MODULO MODULO  
            END IF  
        END FOR  
        SET totalUniqueGoodSubsequences TO the sum of countSubsequencesEndingWithZero PLUS countSubsequencesEndingWithOne PLUS containsZero MODULO MODULO  
        RETURN totalUniqueGoodSubsequences  
    END FUNCTION  
END CLASS
