CLASS Solution  
    FUNCTION checkSubarraySum(nums, k)  
        ASSIGN reference TO dictionary with zero key associated with negative one value AS remainder_map  
        SET current_sum TO zero  
        FOR each index i AND element num IN nums  
            SET current_sum TO current_sum PLUS num  
            IF k NOT EQUALS zero  
                SET current_sum TO remainder of current_sum DIVIDED BY k  
            END IF  
            IF current_sum IN keys of remainder_map  
                IF i MINUS value at current_sum key IN remainder_map GREATER THAN one  
                    RETURN true  
                END IF  
            ELSE  
                SET remainder_map value at key current_sum TO i  
            END IF  
        END FOR  
        RETURN false  
    END FUNCTION  
END CLASS
