CLASS Solution  
    FUNCTION minTaps(n, ranges)  
        SET max_range TO a list of zeroes with the length of n PLUS one  
        FOR index FROM zero TO n  
            IF element at position index OF ranges GREATER THAN zero  
                SET left TO the maximum between zero AND index MINUS element at position index OF ranges  
                SET right TO the minimum between n AND index PLUS element at position index OF ranges  
                SET element at position left OF max_range TO the maximum between element at position left OF max_range AND right  
            END IF  
        END FOR  
        SET taps TO zero  
        SET current_end TO zero  
        SET farthest TO zero  
        FOR index FROM zero TO n  
            IF index GREATER THAN farthest  
                RETURN negative one  
            END IF  
            IF index GREATER THAN current_end  
                INCREMENT taps BY one  
                SET current_end TO farthest  
            END IF  
            SET farthest TO the maximum between farthest AND element at position index OF max_range  
        END FOR  
        RETURN taps  
    END FUNCTION  
END CLASS
