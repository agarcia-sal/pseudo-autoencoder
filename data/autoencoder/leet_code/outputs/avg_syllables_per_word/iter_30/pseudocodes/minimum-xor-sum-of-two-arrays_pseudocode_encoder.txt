CLASS Solution  
    FUNCTION minimumXORSum(nums1, nums2)  
        SET n TO the LENGTH OF nums1  
        SET dp TO a list of length two RAISED TO the power of n with each element SET TO infinity  
        SET element at position zero OF dp TO zero  
        FOR each mask FROM zero TO two RAISED TO the power of n MINUS one  
            SET bit_count TO the COUNT OF ones IN the binary representation OF mask  
            FOR each j FROM zero TO n MINUS one  
                IF the bit at position j IN mask IS set  
                    SET previous_mask TO mask WITH the bit at position j UNSET  
                    SET xor_value TO element at position bit_count MINUS one OF nums1 XOR element at position j OF nums2  
                    SET element at position mask OF dp TO the MINIMUM OF element at position mask OF dp AND the SUM OF element at position previous_mask OF dp AND xor_value  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position two RAISED TO the power of n MINUS one OF dp  
    END FUNCTION  
END CLASS
