CLASS Solution  
    FUNCTION minDistance(word1, word2)  
        SET m TO the LENGTH OF word1  
        SET n TO the LENGTH OF word2  
        CREATE dp AS a two dimensional collection with m PLUS one rows and n PLUS one columns initialized to zero  
        FOR i FROM zero TO m  
            SET element at position i and zero OF dp TO i  
        END FOR  
        FOR j FROM zero TO n  
            SET element at position zero and j OF dp TO j  
        END FOR  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF element at position i MINUS one OF word1 EQUALS element at position j MINUS one OF word2  
                    SET element at position i and j OF dp TO element at position i MINUS one and j MINUS one OF dp  
                ELSE  
                    SET element at position i and j OF dp TO one PLUS the MINIMUM OF element at position i MINUS one and j OF dp, element at position i and j MINUS one OF dp, element at position i MINUS one and j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        RETURN element at position m and n OF dp  
    END FUNCTION  
END CLASS
