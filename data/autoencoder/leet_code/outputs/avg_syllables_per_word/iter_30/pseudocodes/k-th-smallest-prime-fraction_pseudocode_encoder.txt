CLASS Solution  
    FUNCTION kthSmallestPrimeFraction(arr, k)  
        SET n TO the LENGTH OF arr  
        SET min_heap TO an empty collection  
        FOR i FROM zero TO n MINUS one MINUS one  
            PUSH a TUPLE OF the fraction of element at position i OF arr DIVIDED BY element at position n MINUS one OF arr, i, n MINUS one INTO min_heap  
        END FOR  
        FOR each count FROM zero TO k MINUS one MINUS one  
            POP the smallest element FROM min_heap AND ASSIGN the fraction TO ignored_variable, i TO variable i, j TO variable j  
            IF j MINUS one GREATER THAN i  
                PUSH a TUPLE OF the fraction of element at position i OF arr DIVIDED BY element at position j MINUS one OF arr, i, j MINUS one INTO min_heap  
            END IF  
        END FOR  
        POP the smallest element FROM min_heap AND ASSIGN the fraction TO ignored_variable, i TO variable i, j TO variable j  
        RETURN a collection of element at position i OF arr and element at position j OF arr  
    END FUNCTION  
END CLASS
