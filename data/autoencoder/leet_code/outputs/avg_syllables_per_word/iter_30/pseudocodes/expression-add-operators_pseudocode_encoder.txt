CLASS Solution  
    FUNCTION addOperators(num, target)  
        FUNCTION dfs(index, path, value, prev)  
            IF index EQUALS the LENGTH OF num  
                IF value EQUALS target  
                    APPEND path TO result  
                END IF  
                RETURN  
            END IF  
            FOR i FROM index PLUS one TO the LENGTH OF num PLUS one  
                SET operand TO substring from position index TO position i OF num  
                IF the LENGTH OF operand GREATER THAN one AND the element at position zero OF operand EQUALS zero  
                    CONTINUE  
                END IF  
                IF index EQUALS zero  
                    CALL dfs WITH arguments i, operand, integer conversion of operand, integer conversion of operand  
                ELSE  
                    CALL dfs WITH arguments i, concatenation of path PLUS plus symbol PLUS operand, value PLUS integer conversion of operand, integer conversion of operand  
                    CALL dfs WITH arguments i, concatenation of path PLUS minus symbol PLUS operand, value MINUS integer conversion of operand, negative integer conversion of operand  
                    CALL dfs WITH arguments i, concatenation of path PLUS multiplication symbol PLUS operand, value MINUS prev PLUS prev MULTIPLIED BY integer conversion of operand, prev MULTIPLIED BY integer conversion of operand  
                END IF  
            END FOR  
        END FUNCTION  
        SET result TO empty collection  
        CALL dfs WITH arguments zero, empty string, zero, zero  
        RETURN result  
    END FUNCTION  
END CLASS
