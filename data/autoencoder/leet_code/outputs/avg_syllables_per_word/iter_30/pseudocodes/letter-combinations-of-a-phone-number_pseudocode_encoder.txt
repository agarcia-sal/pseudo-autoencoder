CLASS Solution  
    FUNCTION letterCombinations(digits)  
        IF digits EQUALS empty string  
            RETURN empty list  
        END IF  
        SET phone_map TO mapping from digit character two TO string of letters a b c  
                      mapping from digit character three TO string of letters d e f  
                      mapping from digit character four TO string of letters g h i  
                      mapping from digit character five TO string of letters j k l  
                      mapping from digit character six TO string of letters m n o  
                      mapping from digit character seven TO string of letters p q r s  
                      mapping from digit character eight TO string of letters t u v  
                      mapping from digit character nine TO string of letters w x y z  
        FUNCTION backtrack(combination, next_digits)  
            IF the length of next_digits EQUALS zero  
                APPEND combination TO output  
            ELSE  
                FOR each letter IN string of letters at position zero of next_digits mapped in phone_map  
                    CALL backtrack with argument concatenation of combination and letter, and substring of next_digits from position one TO end  
                END FOR  
            END IF  
        END FUNCTION  
        SET output TO empty list  
        CALL backtrack with empty string and digits  
        RETURN output  
    END FUNCTION  
END CLASS
