CLASS Solution  
    FUNCTION shortestPalindrome(s)  
        IF s EQUALS empty string  
            RETURN s  
        END IF  
  
        FUNCTION is_palindrome(check_str)  
            RETURN check_str EQUALS reversed check_str  
        END FUNCTION  
  
        FOR i FROM the LENGTH OF s TO zero DECREMENTED BY one  
            IF is_palindrome(substring from position zero TO position i MINUS one OF s)  
                SET to_add TO reverse of substring from position i TO the LENGTH OF s MINUS one OF s  
                RETURN concatenation of to_add and s  
            END IF  
        END FOR  
    END FUNCTION  
END CLASS
