CLASS Solution  
    FUNCTION numDistinctIslands2(grid)  
        FUNCTION dfs(x, y, island)  
            IF x LESS THAN zero OR x GREATER THAN OR EQUAL TO the LENGTH OF grid OR y LESS THAN zero OR y GREATER THAN OR EQUAL TO the LENGTH OF element at position zero OF grid OR element at position x OF grid at position y EQUALS zero  
                RETURN  
            END IF  
            SET element at position x OF grid at position y TO zero  
            APPEND tuple of x and y TO island  
            CALL dfs with x PLUS one and y and island  
            CALL dfs with x MINUS one and y and island  
            CALL dfs with x and y PLUS one and island  
            CALL dfs with x and y MINUS one and island  
        END FUNCTION  

        FUNCTION normalize(island)  
            SET shapes TO empty collection  
            FOR each pair dx and dy IN collection of pairs one and one, one and negative one, negative one and one, negative one and negative one  
                SET new_island TO empty collection  
                FOR each pair x and y IN island  
                    APPEND tuple of x MULTIPLIED BY dx and y MULTIPLIED BY dy TO new_island  
                END FOR  
                SORT new_island  
                SET min_x TO minimum x FOR each pair x and y IN new_island  
                SET min_y TO minimum y FOR each pair x and y IN new_island  
                APPEND tuple of pairs where each x is x MINUS min_x and y is y MINUS min_y FOR each pair x and y IN new_island TO shapes  
            END FOR  
            FOR each index i FROM zero TO the LENGTH OF shapes MINUS one  
                APPEND tuple of pairs where each x and y is y and negative x FOR each pair x and y IN element at position i OF shapes TO shapes  
            END FOR  
            RETURN minimum element IN shapes  
        END FUNCTION  

        SET islands TO empty collection  
        FOR each index i FROM zero TO the LENGTH OF grid MINUS one  
            FOR each index j FROM zero TO the LENGTH OF element at position zero OF grid MINUS one  
                IF element at position i OF grid at position j EQUALS one  
                    SET island TO empty collection  
                    CALL dfs with i and j and island  
                    ADD output of normalize with island TO islands  
                END IF  
            END FOR  
        END FOR  
        RETURN the LENGTH OF islands  
    END FUNCTION  
END CLASS
