CLASS Solution  
    FUNCTION reorderedPowerOf2 WITH PARAMETER n  
        SET collection_of_sorted_digits_of_powers_of_two TO an empty set  
        SET current_power_of_two TO one  
        WHILE current_power_of_two LESS THAN OR EQUAL TO one billion  
            ADD the tuple of the sorted string representation of current_power_of_two TO collection_of_sorted_digits_of_powers_of_two  
            SET current_power_of_two TO current_power_of_two MULTIPLIED BY two  
        END WHILE  
        SET sorted_n_digits TO the tuple of the sorted string representation of n  
        RETURN whether sorted_n_digits IS IN collection_of_sorted_digits_of_powers_of_two  
    END FUNCTION  
END CLASS
