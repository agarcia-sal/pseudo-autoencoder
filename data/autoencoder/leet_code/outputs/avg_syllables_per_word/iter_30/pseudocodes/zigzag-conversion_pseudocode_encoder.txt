CLASS Solution  
    FUNCTION convert(s, numRows)  
        IF numRows EQUALS one  
            RETURN s  
        END IF  
        SET rows TO a list containing empty strings repeated numRows times  
        SET current_row TO zero  
        SET going_down TO false  
        FOR each character IN s  
            APPEND character TO the string at position current_row OF rows  
            IF current_row EQUALS zero OR current_row EQUALS numRows MINUS one  
                SET going_down TO NOT going_down  
            END IF  
            IF going_down  
                INCREMENT current_row BY one  
            ELSE  
                DECREMENT current_row BY one  
            END IF  
        END FOR  
        SET result TO the concatenation of all strings in rows  
        RETURN result  
    END FUNCTION  
END CLASS
