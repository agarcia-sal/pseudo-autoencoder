CLASS Solution  
    FUNCTION canTransform(start, end)  
        SET start_filtered TO empty list  
        FOR each index AND character IN start with enumeration  
            IF character NOT EQUALS character X  
                APPEND tuple of character AND index TO start_filtered  
            END IF  
        END FOR  
        SET end_filtered TO empty list  
        FOR each index AND character IN end with enumeration  
            IF character NOT EQUALS character X  
                APPEND tuple of character AND index TO end_filtered  
            END IF  
        END FOR  
        IF the LENGTH OF start_filtered NOT EQUALS the LENGTH OF end_filtered  
            RETURN false  
        END IF  
        FOR each pair (character_one, index_one) AND (character_two, index_two) IN zip of start_filtered AND end_filtered  
            IF character_one NOT EQUALS character_two  
                RETURN false  
            END IF  
            IF character_one EQUALS character L AND index_one LESS THAN index_two  
                RETURN false  
            END IF  
            IF character_one EQUALS character R AND index_one GREATER THAN index_two  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
