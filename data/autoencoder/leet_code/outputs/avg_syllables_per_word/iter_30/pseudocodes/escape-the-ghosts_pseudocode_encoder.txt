CLASS Solution  
    FUNCTION escapeGhosts(ghosts, target)  
        SET player_distance TO absolute value of element at position zero OF target PLUS absolute value of element at position one OF target  
        FOR each ghost IN ghosts  
            SET ghost_distance TO absolute value of element at position zero OF ghost MINUS element at position zero OF target PLUS absolute value of element at position one OF ghost MINUS element at position one OF target  
            IF ghost_distance LESS THAN OR EQUAL TO player_distance  
                RETURN false  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
