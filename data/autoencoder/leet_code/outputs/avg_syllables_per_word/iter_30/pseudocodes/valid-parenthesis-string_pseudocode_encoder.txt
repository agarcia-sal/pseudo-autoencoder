CLASS Solution  
    FUNCTION checkValidString(s)  
        SET low TO zero  
        SET high TO zero  
        FOR each char IN s  
            IF char EQUALS left parenthesis  
                INCREMENT low BY one  
                INCREMENT high BY one  
            ELSE IF char EQUALS right parenthesis  
                DECREMENT low BY one  
                DECREMENT high BY one  
            ELSE   
                DECREMENT low BY one  
                INCREMENT high BY one  
            END IF  
            IF high LESS THAN zero  
                RETURN false  
            END IF  
            IF low LESS THAN zero  
                SET low TO zero  
            END IF  
        END FOR  
        RETURN low EQUALS zero  
    END FUNCTION  
END CLASS
