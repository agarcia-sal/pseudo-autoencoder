CLASS Solution  
    FUNCTION minimumDistance(word)  
        FUNCTION distance(c1, c2)  
            IF c1 EQUALS none  
                RETURN zero  
            END IF  
            SET position x1 TO the result of integer division of the difference between the Unicode code point of c1 and the Unicode code point of uppercase A BY six  
            SET position y1 TO the remainder of the division of the difference between the Unicode code point of c1 and the Unicode code point of uppercase A BY six  
            SET position x2 TO the result of integer division of the difference between the Unicode code point of c2 and the Unicode code point of uppercase A BY six  
            SET position y2 TO the remainder of the division of the difference between the Unicode code point of c2 and the Unicode code point of uppercase A BY six  
            RETURN the absolute value of the difference between x1 and x2 PLUS the absolute value of the difference between y1 and y2  
        END FUNCTION  
        
        FUNCTION dp(i, f1, f2)  
            IF i EQUALS the length of word  
                RETURN zero  
            END IF  
            SET c TO the element at position i OF word  
            SET d1 TO the result of calling distance with arguments f1 and c PLUS the result of calling dp with arguments i PLUS one, c, and f2  
            SET d2 TO the result of calling distance with arguments f2 and c PLUS the result of calling dp with arguments i PLUS one, f1, and c  
            RETURN the smaller value between d1 and d2  
        END FUNCTION  
        
        RETURN the result of calling dp with arguments zero, none, and none  
    END FUNCTION  
END CLASS
