CLASS Solution  
    FUNCTION movesToChessboard(board)  
        SET N TO the LENGTH OF board  
        SET rowSum TO zero  
        SET colSum TO zero  
        SET rowSwap TO zero  
        SET colSwap TO zero  
        FOR each i FROM zero TO N MINUS one  
            FOR each j FROM zero TO N MINUS one  
                IF the value at position i OF board EXCLUSIVE OR the value at position i zero OF board EXCLUSIVE OR the value at position zero j OF board EXCLUSIVE OR the value at position zero zero OF board IS NOT EQUAL TO zero  
                    RETURN negative one  
                END IF  
            END FOR  
            INCREMENT rowSum BY the value at position i zero OF board  
            INCREMENT colSum BY the value at position zero i OF board  
            INCREMENT rowSwap BY one IF the value at position i zero OF board EQUALS the remainder of i DIVIDED BY two ELSE increment by zero  
            INCREMENT colSwap BY one IF the value at position zero i OF board EQUALS the remainder of i DIVIDED BY two ELSE increment by zero  
        END FOR  
        IF the integer division of N BY two IS GREATER THAN rowSum OR rowSum IS GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the integer division of N BY two IS GREATER THAN colSum OR colSum IS GREATER THAN the integer division of N PLUS one BY two  
            RETURN negative one  
        END IF  
        IF the remainder of N DIVIDED BY two IS EQUAL TO one  
            IF the remainder of rowSwap DIVIDED BY two IS EQUAL TO one  
                SET rowSwap TO N MINUS rowSwap  
            END IF  
            IF the remainder of colSwap DIVIDED BY two IS EQUAL TO one  
                SET colSwap TO N MINUS colSwap  
            END IF  
        ELSE  
            SET rowSwap TO the minimum of rowSwap AND N MINUS rowSwap  
            SET colSwap TO the minimum of colSwap AND N MINUS colSwap  
        END IF  
        SET result TO the integer division of the sum of rowSwap AND colSwap BY two  
        RETURN result  
    END FUNCTION  
END CLASS
