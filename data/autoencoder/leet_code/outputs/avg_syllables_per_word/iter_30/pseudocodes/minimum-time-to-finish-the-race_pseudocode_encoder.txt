CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO list of length fifteen FILLED WITH infinity  
        FOR each pair f comma r IN tires  
            SET total_time TO zero  
            FOR index i FROM zero TO fourteen  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER OF i  
                IF lap_time GREATER THAN changeTime PLUS f  
                    BREAK  
                END IF  
                INCREMENT total_time BY lap_time  
                IF total_time LESS THAN min_times at position i  
                    SET min_times at position i TO total_time  
                END IF  
            END FOR  
        END FOR  

        SET dp TO list of length numLaps PLUS one FILLED WITH infinity  
        SET dp at position zero TO zero  

        FOR index i FROM one TO numLaps  
            FOR index j FROM zero TO the MINIMUM OF i MINUS one AND fourteen  
                SET candidate_time TO dp at position i MINUS j MINUS one PLUS min_times at position j PLUS changeTime  
                IF candidate_time LESS THAN dp at position i  
                    SET dp at position i TO candidate_time  
                END IF  
            END FOR  
        END FOR  

        RETURN dp at position numLaps MINUS changeTime  
    END FUNCTION  
END CLASS
