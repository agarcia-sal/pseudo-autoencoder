CLASS Solution  
    FUNCTION getMoneyAmount(n)  
        FUNCTION dp(left, right)  
            IF left IS GREATER THAN OR EQUAL TO right  
                RETURN zero  
            END IF  
            SET min_cost TO infinity  
            FOR pivot FROM left TO right  
                SET cost TO pivot PLUS the GREATER VALUE BETWEEN dp(left, pivot MINUS one) AND dp(pivot PLUS one, right)  
                IF cost IS LESS THAN min_cost  
                    SET min_cost TO cost  
                END IF  
            END FOR  
            RETURN min_cost  
        END FUNCTION  
        RETURN dp(one, n)  
    END FUNCTION  
END CLASS
