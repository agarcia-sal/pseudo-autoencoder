CLASS Solution  
    FUNCTION pourWater(heights, volume, k)  
        SET n TO the LENGTH OF heights  
        
        FOR each count FROM one TO volume  
            SET i TO k  
            WHILE i GREATER THAN zero AND element at position i MINUS one OF heights LESS THAN OR EQUAL TO element at position i OF heights  
                DECREMENT i BY one  
            END WHILE  
            WHILE i LESS THAN k AND element at position i OF heights EQUALS element at position i PLUS one OF heights  
                INCREMENT i BY one  
            END WHILE  
            IF i LESS THAN k  
                INCREMENT element at position i OF heights BY one  
                CONTINUE  
            END IF  
            
            SET i TO k  
            WHILE i LESS THAN n MINUS one AND element at position i PLUS one OF heights LESS THAN OR EQUAL TO element at position i OF heights  
                INCREMENT i BY one  
            END WHILE  
            WHILE i GREATER THAN k AND element at position i OF heights EQUALS element at position i MINUS one OF heights  
                DECREMENT i BY one  
            END WHILE  
            IF i GREATER THAN k  
                INCREMENT element at position i OF heights BY one  
                CONTINUE  
            END IF  
            
            INCREMENT element at position k OF heights BY one  
        END FOR  
        
        RETURN heights  
    END FUNCTION  
END CLASS
