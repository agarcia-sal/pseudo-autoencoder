CLASS Solution  
    FUNCTION kthGrammar(n, k)  
        IF n EQUALS one  
            RETURN zero  
        END IF  
        SET length_of_previous_row TO two RAISED TO THE POWER OF n MINUS two  
        IF k LESS THAN OR EQUAL TO length_of_previous_row  
            RETURN the RESULT OF kthGrammar WITH n MINUS one AND k  
        ELSE  
            RETURN one MINUS the RESULT OF kthGrammar WITH n MINUS one AND k MINUS length_of_previous_row  
        END IF  
    END FUNCTION  
END CLASS
