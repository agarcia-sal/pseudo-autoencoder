CLASS Solution  
    FUNCTION arrayNesting(nums)  
        SET visited TO list of boolean false values with the length of nums  
        SET max_length TO zero  
        FOR index FROM zero TO the length of nums MINUS one  
            IF visited at position index EQUALS false  
                SET current_length TO zero  
                SET k TO index  
                WHILE visited at position k EQUALS false  
                    SET visited at position k TO true  
                    SET k TO element at position k OF nums  
                    INCREMENT current_length BY one  
                END WHILE  
                IF max_length LESS THAN current_length  
                    SET max_length TO current_length  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
