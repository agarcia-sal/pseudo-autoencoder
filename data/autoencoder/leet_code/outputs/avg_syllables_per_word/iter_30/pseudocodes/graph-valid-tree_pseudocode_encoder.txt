CLASS Solution  
    FUNCTION validTree(n, edges)  
        IF the LENGTH OF edges NOT EQUALS n MINUS one  
            RETURN false  
        END IF  
        SET adj_list TO a collection mapping each integer from zero TO n MINUS one TO an empty collection  
        FOR each pair a and b IN edges  
            APPEND b TO adj_list at position a  
            APPEND a TO adj_list at position b  
        END FOR  
        SET visited TO an empty collection  
        FUNCTION dfs(node)  
            IF node IS IN visited  
                RETURN  
            END IF  
            ADD node TO visited  
            FOR each neighbor IN adj_list at position node  
                CALL dfs WITH neighbor  
            END FOR  
        END FUNCTION  
        CALL dfs WITH zero  
        RETURN the LENGTH OF visited EQUALS n  
    END FUNCTION  
END CLASS
