CLASS Solution  
    FUNCTION maxPerformance(n, speed, efficiency, k)  
        SET engineers TO list of pairs of elements at each position of efficiency AND speed correspondingly  
        SORT engineers in descending order by the first element of each pair  
        SET speed_heap TO empty list  
        SET speed_sum TO zero  
        SET max_performance TO zero  
        FOR each pair eff AND spd IN engineers  
            ADD spd to speed_heap with heap push behavior  
            INCREMENT speed_sum BY spd  
            IF the length of speed_heap is greater than k  
                REMOVE smallest element from speed_heap with heap pop behavior  
                DECREMENT speed_sum BY that removed element  
            END IF  
            SET current_performance TO speed_sum MULTIPLIED BY eff  
            IF current_performance GREATER THAN max_performance  
                SET max_performance TO current_performance  
            END IF  
        END FOR  
        RETURN the remainder of max_performance DIVIDED BY the sum of ten to the power of nine AND one  
    END FUNCTION  
END CLASS
