CLASS Solution  
    FUNCTION cherryPickup(grid)  
        SET n TO the LENGTH OF grid  
        IF element at position zero, zero OF grid EQUALS minus one OR element at position n MINUS one, n MINUS one OF grid EQUALS minus one  
            RETURN zero  
        END IF  
        
        INITIALIZE dp AS a three dimensional collection of size n BY n BY n FILLED WITH minus one  
        SET element at position zero, zero, zero OF dp TO element at position zero, zero OF grid  
        
        SET directions TO collection containing pair zero, one AND pair one, zero  
        
        FOR t FROM one TO two MULTIPLIED BY n MINUS one MINUS one  
            FOR i1 FROM MAXIMUM OF zero AND t MINUS n PLUS one TO MINIMUM OF n AND t PLUS one MINUS one  
                FOR i2 FROM MAXIMUM OF zero AND t MINUS n PLUS one TO MINIMUM OF n AND t PLUS one MINUS one  
                    SET j1 TO t MINUS i1  
                    SET j2 TO t MINUS i2  
                    IF element at position i1, j1 OF grid EQUALS minus one OR element at position i2, j2 OF grid EQUALS minus one  
                        CONTINUE  
                    END IF  
                    
                    SET prev_max TO minus one  
                    FOR each pair di1, dj1 IN directions  
                        FOR each pair di2, dj2 IN directions  
                            SET pi1 TO i1 MINUS di1  
                            SET pj1 TO j1 MINUS dj1  
                            SET pi2 TO i2 MINUS di2  
                            SET pj2 TO j2 MINUS dj2  
                            IF pi1 GREATER THAN OR EQUAL TO zero AND pi1 LESS THAN n AND pj1 GREATER THAN OR EQUAL TO zero AND pj1 LESS THAN n AND pi2 GREATER THAN OR EQUAL TO zero AND pi2 LESS THAN n AND pj2 GREATER THAN OR EQUAL TO zero AND pj2 LESS THAN n  
                                SET prev_max TO THE MAXIMUM BETWEEN prev_max AND element at position pi1, pi2, pj1 OF dp  
                            END IF  
                        END FOR  
                    END FOR  
                    
                    IF prev_max GREATER THAN OR EQUAL TO zero  
                        IF i1 EQUALS i2  
                            SET cherries TO element at position i1, j1 OF grid  
                        ELSE  
                            SET cherries TO element at position i1, j1 OF grid PLUS element at position i2, j2 OF grid  
                        END IF  
                        SET element at position i1, i2, j1 OF dp TO prev_max PLUS cherries  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        
        RETURN THE MAXIMUM BETWEEN zero AND element at position n MINUS one, n MINUS one, n MINUS one OF dp  
    END FUNCTION  
END CLASS
