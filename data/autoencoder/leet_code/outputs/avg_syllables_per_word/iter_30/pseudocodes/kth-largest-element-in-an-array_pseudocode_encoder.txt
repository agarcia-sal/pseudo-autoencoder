CLASS Solution  
    FUNCTION findKthLargest(nums, k)  
        SET min_heap TO empty list  
        FOR each num IN nums  
            INSERT num INTO min_heap maintaining heap order  
            IF the LENGTH OF min_heap GREATER THAN k  
                REMOVE smallest element FROM min_heap maintaining heap order  
            END IF  
        END FOR  
        RETURN element at position zero OF min_heap  
    END FUNCTION  
END CLASS
