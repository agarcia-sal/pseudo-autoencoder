CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO A new TreeNode CREATED WITH value at position zero OF values  
    SET i TO one  
    SET queue TO an empty deque collection  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element REMOVED FROM queue  
        IF i LESS THAN the length of values AND value at position i OF values IS NOT none  
            SET node.left TO A new TreeNode CREATED WITH value at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the length of values AND value at position i OF values IS NOT none  
            SET node.right TO A new TreeNode CREATED WITH value at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value at attribute val OF p NOT EQUALS value at attribute val OF q  
        RETURN false  
    ELSE  
        RETURN is_same_tree(value at attribute left OF p, value at attribute left OF q) AND is_same_tree(value at attribute right OF p, value at attribute right OF q)  
    END IF  
END FUNCTION  

CLASS Solution  
    FUNCTION kthSmallest(self, root, k)  
        SET stack TO an empty list  
        SET current TO root  
        SET count TO zero  
        WHILE stack IS NOT empty OR current IS NOT none  
            WHILE current IS NOT none  
                APPEND current TO stack  
                SET current TO value at attribute left OF current  
            END WHILE  
            SET current TO the last element REMOVED FROM stack  
            INCREMENT count BY one  
            IF count EQUALS k  
                RETURN value at attribute val OF current  
            END IF  
            SET current TO value at attribute right OF current  
        END WHILE  
    END FUNCTION  
END CLASS
