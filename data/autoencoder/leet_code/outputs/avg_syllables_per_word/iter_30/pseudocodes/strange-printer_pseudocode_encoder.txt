CLASS Solution  
    FUNCTION strangePrinter WITH PARAMETER s  
        FUNCTION dp WITH PARAMETERS i j memo  
            IF i GREATER THAN j  
                RETURN zero  
            END IF  
            IF tuple of i and j EXISTS IN memo  
                RETURN value at tuple of i and j IN memo  
            END IF  
            SET result TO CALL dp WITH i j MINUS one memo PLUS one  
            FOR k FROM i TO j MINUS one  
                IF character at position k OF s EQUALS character at position j OF s  
                    SET temp_result TO CALL dp WITH i k MINUS one memo PLUS CALL dp WITH k j MINUS one memo  
                    IF temp_result LESS THAN result  
                        SET result TO temp_result  
                    END IF  
                END IF  
            END FOR  
            SET memo at tuple of i and j TO result  
            RETURN result  
        END FUNCTION  
        RETURN CALL dp WITH zero LENGTH OF s MINUS one EMPTY dictionary  
    END FUNCTION  
END CLASS
