CLASS Solution  
    FUNCTION canFinish(numCourses, prerequisites)  
        CREATE a graph as a mapping from keys to empty lists  
        CREATE a list named indegree with length numCourses filled with zero  
        FOR each pair course and prereq IN prerequisites  
            APPEND course TO the list mapped by prereq in graph  
            INCREMENT value at position course OF indegree BY one  
        END FOR  
        CREATE a queue containing all course numbers from zero TO numCourses MINUS one WHERE the value at position course OF indegree EQUALS zero  
        WHILE the queue is not empty  
            REMOVE and ASSIGN the first element of queue TO current_course  
            DECREMENT numCourses BY one  
            FOR each next_course IN the list mapped by current_course in graph  
                DECREMENT value at position next_course OF indegree BY one  
                IF the value at position next_course OF indegree EQUALS zero  
                    APPEND next_course TO queue  
                END IF  
            END FOR  
        END WHILE  
        RETURN whether numCourses EQUALS zero  
    END FUNCTION  
END CLASS
