CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN none  
    END IF  
    SET root TO new TreeNode with value as element at position zero OF values  
    SET i TO one  
    SET queue TO empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO the first element removed from queue  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO new TreeNode with value as element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO new TreeNode with value as element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p IS none AND q IS none  
        RETURN true  
    ELSE IF p IS none OR q IS none  
        RETURN false  
    ELSE IF value of p NOT EQUALS value of q  
        RETURN false  
    ELSE  
        RETURN the RESULT OF is_same_tree with arguments p.left and q.left AND the RESULT OF is_same_tree with arguments p.right and q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION addOneRow(self, root, val, depth)  
        IF depth EQUALS one  
            RETURN new TreeNode with value val and left subtree root  
        END IF  
        SET queue TO empty deque containing the tuple of root and one  
        WHILE queue IS NOT empty  
            SET node and level TO the first element removed from queue  
            IF level EQUALS depth MINUS one  
                SET node.left TO new TreeNode with value val and left subtree node.left  
                SET node.right TO new TreeNode with value val and right subtree node.right  
            ELSE IF level LESS THAN depth MINUS one  
                IF node.left IS NOT none  
                    APPEND the tuple of node.left and level PLUS one TO queue  
                END IF  
                IF node.right IS NOT none  
                    APPEND the tuple of node.right and level PLUS one TO queue  
                END IF  
            END IF  
        END WHILE  
        RETURN root  
    END FUNCTION  
END CLASS
