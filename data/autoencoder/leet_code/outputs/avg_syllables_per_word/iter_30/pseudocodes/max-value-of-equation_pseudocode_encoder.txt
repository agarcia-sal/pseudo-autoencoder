CLASS Solution  
    FUNCTION findMaxValueOfEquation(points, k)  
        SET max_value TO negative infinity  
        ASSIGN reference TO an empty double ended queue named deque_  
        
        FOR each element IN points  
            ASSIGN xj TO the element at position zero OF element  
            ASSIGN yj TO the element at position one OF element  
            
            WHILE deque_ is not empty AND xj MINUS the second element at position zero OF the front element OF deque_ GREATER THAN k  
                REMOVE the element from the front OF deque_  
            END WHILE  
            
            IF deque_ is not empty  
                SET max_value TO the maximum of max_value AND yj PLUS xj PLUS the first element at position zero OF the front element OF deque_  
            END IF  
            
            WHILE deque_ is not empty AND yj MINUS xj GREATER THAN OR EQUAL TO the first element at position zero OF the last element OF deque_  
                REMOVE the element from the back OF deque_  
            END WHILE  
            
            APPEND the pair consisting of yj MINUS xj AND xj TO deque_  
        END FOR  
        
        RETURN max_value  
    END FUNCTION  
END CLASS
