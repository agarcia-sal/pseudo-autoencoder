CLASS Solution  
    FUNCTION topKFrequent(words, k)  
        SET count TO frequency count of each element IN words  
        SET heap TO empty list  
        FOR each word AND frequency IN count  
            APPEND tuple of negative frequency AND word TO heap  
        END FOR  
        CONVERT heap INTO heap structure  
        SET result TO empty list  
        FOR index FROM one TO k  
            REMOVE element WITH smallest tuple FROM heap AND EXTRACT word FROM tuple  
            APPEND extracted word TO result  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
