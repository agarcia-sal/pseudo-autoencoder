CLASS Solution  
    FUNCTION largestNumber(nums)  
        SET nums_str TO the list of string representations of each element IN nums  
        FUNCTION compare(x, y)  
            IF the concatenation of x PLUS y is GREATER THAN the concatenation of y PLUS x  
                RETURN negative one  
            ELSE IF the concatenation of x PLUS y is LESS THAN the concatenation of y PLUS x  
                RETURN one  
            ELSE  
                RETURN zero  
            END IF  
        END FUNCTION  
        SORT nums_str USING the custom comparator compare  
        IF the element at position zero OF nums_str EQUALS the string zero  
            RETURN the string zero  
        END IF  
        SET result TO the concatenation of all elements IN nums_str in order  
        RETURN result  
    END FUNCTION  
END CLASS
