CLASS Solution  
    FUNCTION findDuplicates(nums)  
        FOR i FROM zero TO the LENGTH OF nums MINUS one  
            WHILE element at position i OF nums NOT EQUALS element at position element at position i OF nums MINUS one OF nums  
                SET temporary_value TO element at position element at position i OF nums MINUS one OF nums  
                SET element at position element at position i OF nums MINUS one OF nums TO element at position i OF nums  
                SET element at position i OF nums TO temporary_value  
            END WHILE  
        END FOR  
        SET result_list TO empty list  
        FOR each index i AND element v IN enumeration of nums  
            IF element v NOT EQUALS index i PLUS one  
                APPEND element v TO result_list  
            END IF  
        END FOR  
        RETURN result_list  
    END FUNCTION  
END CLASS
