CLASS Solution  
    FUNCTION findLonelyPixel WITH picture  
        IF picture EQUALS nothing OR element at position zero OF picture EQUALS nothing  
            RETURN zero  
        END IF  
        SET m TO the LENGTH OF picture  
        SET n TO the LENGTH OF element at position zero OF picture  
        SET row_count TO a list of zeroes WITH LENGTH m  
        SET col_count TO a list of zeroes WITH LENGTH n  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position j OF element at position i OF picture EQUALS character B  
                    INCREMENT element at position i OF row_count BY one  
                    INCREMENT element at position j OF col_count BY one  
                END IF  
            END FOR  
        END FOR  
        SET lonely_pixel_count TO zero  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                IF element at position j OF element at position i OF picture EQUALS character B AND element at position i OF row_count EQUALS one AND element at position j OF col_count EQUALS one  
                    INCREMENT lonely_pixel_count BY one  
                END IF  
            END FOR  
        END FOR  
        RETURN lonely_pixel_count  
    END FUNCTION  
END CLASS
