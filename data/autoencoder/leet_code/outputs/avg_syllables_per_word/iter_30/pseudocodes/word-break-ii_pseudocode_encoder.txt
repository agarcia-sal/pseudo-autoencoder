CLASS Solution  
    FUNCTION wordBreak(s, wordDict)  
        SET wordSet TO wordDict converted to a set  
        SET memo TO empty dictionary  
          
        FUNCTION backtrack(start)  
            IF start is in memo THEN  
                RETURN value at memo with key start  
            END IF  
            IF start EQUALS the length of s THEN  
                RETURN list containing empty string  
            END IF  
            
            SET res TO empty list  
            FOR end FROM start PLUS one TO the length of s PLUS one  
                SET word TO substring from position start TO position end of s  
                IF word is in wordSet THEN  
                    FOR each sentence IN backtrack(end)  
                        IF sentence is empty string THEN  
                            APPEND word TO res  
                        ELSE  
                            APPEND word followed by space followed by sentence TO res  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
            
            SET memo at key start TO res  
            RETURN res  
        END FUNCTION  
        
        RETURN backtrack with argument zero  
    END FUNCTION  
END CLASS
