CLASS Solution  
    FUNCTION minMeetingRooms(intervals)  
        IF intervals EQUALS empty collection  
            RETURN zero  
        END IF  
        SORT intervals BY element at position zero OF each element  
        SET min_heap TO empty collection  
        FOR each meeting IN intervals  
            IF min_heap NOT EQUALS empty collection AND element at position zero OF meeting GREATER THAN OR EQUAL TO element at position zero OF min_heap  
                REMOVE smallest element FROM min_heap  
            END IF  
            ADD element at position one OF meeting TO min_heap  
        END FOR  
        SET result TO the LENGTH OF min_heap  
        RETURN result  
    END FUNCTION  
END CLASS
