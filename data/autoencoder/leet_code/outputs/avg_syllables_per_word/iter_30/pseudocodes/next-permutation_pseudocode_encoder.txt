CLASS Solution  
    FUNCTION nextPermutation(nums)  
        SET i TO the LENGTH OF nums MINUS two  
        WHILE i GREATER THAN OR EQUAL TO zero AND element at position i OF nums GREATER THAN OR EQUAL TO element at position i PLUS one OF nums  
            DECREMENT i BY one  
        END WHILE  
        IF i GREATER THAN OR EQUAL TO zero  
            SET j TO the LENGTH OF nums MINUS one  
            WHILE element at position j OF nums LESS THAN OR EQUAL TO element at position i OF nums  
                DECREMENT j BY one  
            END WHILE  
            SET temporary TO element at position i OF nums  
            SET element at position i OF nums TO element at position j OF nums  
            SET element at position j OF nums TO temporary  
        END IF  
        SET start_position TO i PLUS one  
        SET end_position TO the LENGTH OF nums MINUS one  
        WHILE start_position LESS THAN end_position  
            SET temporary TO element at position start_position OF nums  
            SET element at position start_position OF nums TO element at position end_position OF nums  
            SET element at position end_position OF nums TO temporary  
            INCREMENT start_position BY one  
            DECREMENT end_position BY one  
        END WHILE  
    END FUNCTION  
END CLASS
