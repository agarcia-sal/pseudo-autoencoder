CLASS TreeNode  
    FUNCTION __init__(self, val EQUALS zero, left EQUALS none, right EQUALS none)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS

FUNCTION tree_node(values)  
    IF values EQUALS none OR the LENGTH OF values EQUALS zero  
        RETURN none  
    END IF  
    SET root TO A NEW TreeNode WITH the element at position zero OF values  
    SET i TO one  
    SET queue TO AN EMPTY COLLECTION OF TYPE deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO THE FIRST ELEMENT OF queue AND REMOVE IT FROM queue  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.left TO A NEW TreeNode WITH the element at position i OF values  
            APPEND node.left TO queue  
        END IF  
        INCREMENT i BY one  
        IF i LESS THAN the LENGTH OF values AND element at position i OF values IS NOT none  
            SET node.right TO A NEW TreeNode WITH the element at position i OF values  
            APPEND node.right TO queue  
        END IF  
        INCREMENT i BY one  
    END WHILE  
    RETURN root  
END FUNCTION

FUNCTION is_same_tree(p, q)  
    IF p EQUALS none AND q EQUALS none  
        RETURN true  
    ELSE IF p EQUALS none OR q EQUALS none  
        RETURN false  
    ELSE IF p.val NOT EQUALS q.val  
        RETURN false  
    ELSE  
        RETURN the result of is_same_tree WITH p.left AND q.left AND the result of is_same_tree WITH p.right AND q.right  
    END IF  
END FUNCTION

CLASS Solution  
    FUNCTION constructMaximumBinaryTree(self, nums)  
        IF nums EQUALS none OR the LENGTH OF nums EQUALS zero  
            RETURN none  
        END IF  
        SET max_val TO THE MAXIMUM VALUE IN nums  
        SET max_index TO THE POSITION OF max_val IN nums  
        SET root TO A NEW TreeNode WITH max_val  
        SET root.left TO the result of self.constructMaximumBinaryTree WITH the SUBLIST OF nums FROM position zero TO position max_index MINUS one  
        SET root.right TO the result of self.constructMaximumBinaryTree WITH the SUBLIST OF nums FROM position max_index PLUS one TO the END OF nums  
        RETURN root  
    END FUNCTION  
END CLASS
