CLASS Solution  
    FUNCTION maximumScore(nums, multipliers)  
        SET m TO the LENGTH OF multipliers  
        SET n TO the LENGTH OF nums  
  
        SET dp TO a LIST OF zero REPEATED m PLUS one TIMES  
  
        FOR i FROM m MINUS one DOWN TO zero  
            SET new_dp TO a LIST OF zero REPEATED m PLUS one TIMES  
            FOR left FROM zero TO i  
                SET right TO n MINUS one MINUS i PLUS left  
                SET mul TO element at position i OF multipliers  
                IF mul MULTIPLIED BY element at position left OF nums PLUS element at position left PLUS one OF dp GREATER THAN mul MULTIPLIED BY element at position right OF nums PLUS element at position left OF dp  
                    SET element at position left OF new_dp TO mul MULTIPLIED BY element at position left OF nums PLUS element at position left PLUS one OF dp  
                ELSE  
                    SET element at position left OF new_dp TO mul MULTIPLIED BY element at position right OF nums PLUS element at position left OF dp  
                END IF  
            END FOR  
            ASSIGN dp TO new_dp  
        END FOR  
  
        RETURN element at position zero OF dp  
    END FUNCTION  
END CLASS
