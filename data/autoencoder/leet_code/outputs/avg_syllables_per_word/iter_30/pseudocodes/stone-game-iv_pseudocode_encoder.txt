CLASS Solution  
    FUNCTION winnerSquareGame(n)  
        SET dp TO list of false repeated the quantity of n PLUS one times  
        FOR i FROM one TO n  
            SET j TO one  
            WHILE j MULTIPLIED BY j LESS THAN OR EQUAL TO i  
                IF NOT element at position i MINUS j MULTIPLIED BY j OF dp  
                    SET element at position i OF dp TO true  
                    BREAK  
                END IF  
                INCREMENT j BY one  
            END WHILE  
        END FOR  
        RETURN element at position n OF dp  
    END FUNCTION  
END CLASS
