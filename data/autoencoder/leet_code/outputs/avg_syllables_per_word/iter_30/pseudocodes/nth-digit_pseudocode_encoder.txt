CLASS Solution  
    FUNCTION findNthDigit(n)  
        SET length TO one  
        SET count TO nine  
        SET start TO one  
        WHILE n GREATER THAN length MULTIPLIED BY count  
            SET n TO n MINUS length MULTIPLIED BY count  
            INCREMENT length BY one  
            SET count TO count MULTIPLIED BY ten  
            SET start TO start MULTIPLIED BY ten  
        END WHILE  
        SET start TO start PLUS the INTEGER QUOTIENT OF n MINUS one DIVIDED BY length  
        RETURN the digit at position the remainder of n MINUS one DIVIDED BY length OF the string representation of start converted to integer  
    END FUNCTION  
END CLASS
