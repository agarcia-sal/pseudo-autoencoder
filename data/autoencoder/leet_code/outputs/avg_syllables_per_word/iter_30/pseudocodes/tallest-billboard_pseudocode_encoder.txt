CLASS Solution  
    FUNCTION tallestBillboard WITH PARAMETER rods  
        SET dp TO dictionary with zero key mapped to zero  
        FOR each rod IN rods  
            SET new_dp TO a copy of dp  
            FOR each key d and value y IN dp items  
                SET sum_one TO d PLUS rod  
                IF sum_one key IN new_dp AND value at sum_one key IN new_dp GREATER THAN OR EQUAL TO y  
                    DO nothing  
                ELSE  
                    SET value at sum_one key IN new_dp TO y  
                END IF  
                SET diff TO the absolute value of d MINUS rod  
                SET min_value TO the smaller value between d and rod  
                SET new_value TO y PLUS min_value  
                IF diff key IN new_dp AND value at diff key IN new_dp GREATER THAN OR EQUAL TO new_value  
                    DO nothing  
                ELSE  
                    SET value at diff key IN new_dp TO new_value  
                END IF  
            END FOR  
            SET dp TO new_dp  
        END FOR  
        RETURN value at zero key IN dp  
    END FUNCTION  
END CLASS
