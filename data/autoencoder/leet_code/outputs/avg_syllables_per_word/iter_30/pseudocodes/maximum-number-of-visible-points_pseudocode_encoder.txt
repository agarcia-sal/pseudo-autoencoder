CLASS Solution  
    FUNCTION visiblePoints(points, angle, location)  
        SET angles TO empty list  
        SET extra_points TO zero  
        FOR each x, y IN points  
            IF x EQUALS element at position zero OF location AND y EQUALS element at position one OF location  
                INCREMENT extra_points BY one  
                CONTINUE  
            END IF  
            SET angle_rad TO arctangent of difference between y and element at position one OF location AND difference between x and element at position zero OF location  
            SET angle_deg TO angle_rad converted to degrees  
            APPEND angle_deg TO angles  
        END FOR  
        SORT angles  
        SET extended_angles TO angles concatenated with each element of angles increased by three hundred sixty  
        SET max_visible TO zero  
        SET left TO zero  
        FOR right FROM zero TO length of extended_angles MINUS one  
            WHILE element at position right OF extended_angles MINUS element at position left OF extended_angles GREATER THAN angle  
                INCREMENT left BY one  
            END WHILE  
            IF max_visible LESS THAN right MINUS left PLUS one  
                SET max_visible TO right MINUS left PLUS one  
            END IF  
        END FOR  
        RETURN max_visible PLUS extra_points  
    END FUNCTION  
END CLASS
