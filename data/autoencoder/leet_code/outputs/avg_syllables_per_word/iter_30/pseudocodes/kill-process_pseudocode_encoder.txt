CLASS Solution  
    FUNCTION killProcess(pid, ppid, kill)  
        SET tree TO a new empty dictionary with default value an empty list  
        FOR each child and parent IN parallel elements of pid and ppid  
            APPEND child TO the list associated with parent IN tree  
        END FOR  
        SET queue TO a list containing kill  
        SET killed_processes TO an empty list  
        WHILE queue is not empty  
            REMOVE the first element from queue and assign it TO current  
            APPEND current TO killed_processes  
            EXTEND queue BY all elements in the list associated with current IN tree  
        END WHILE  
        RETURN killed_processes  
    END FUNCTION  
END CLASS
