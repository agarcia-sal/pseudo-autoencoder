CLASS Solution  
    FUNCTION longestValidParentheses(s)  
        ASSIGN reference TO a new collection named stack containing negative one  
        SET max_length TO zero  
        FOR each index i AND each character char IN s  
            IF char EQUALS open parenthesis  
                APPEND i TO stack  
            ELSE  
                REMOVE top element FROM stack  
                IF stack IS empty  
                    APPEND i TO stack  
                ELSE  
                    SET max_length TO the greater value BETWEEN max_length AND i MINUS the element at the last position OF stack  
                END IF  
            END IF  
        END FOR  
        RETURN max_length  
    END FUNCTION  
END CLASS
