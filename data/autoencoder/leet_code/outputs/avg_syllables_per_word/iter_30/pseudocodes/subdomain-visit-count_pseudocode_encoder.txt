CLASS Solution  
    FUNCTION subdomainVisits(cpdomains)  
        SET domain_count TO default dictionary with integer default  
        FOR each cpdomain IN cpdomains  
            SET count TO the first part OF cpdomain split by space converted to integer  
            SET domain TO the second part OF cpdomain split by space  
            SET subdomains TO domain split by the period word  
            FOR index FROM zero TO the length of subdomains MINUS one  
                SET subdomain TO the concatenation of elements starting at position index TO the last element OF subdomains joined by the period word  
                INCREMENT the value at key subdomain IN domain_count BY count  
            END FOR  
        END FOR  
        RETURN a list constructed by FOR each domain and count IN domain_count items concatenating count converted to string, a space word, and domain  
    END FUNCTION  
END CLASS
