CLASS Solution  
    FUNCTION nearestPalindromic(n)  
        SET length TO the LENGTH OF n  
        IF length EQUALS one  
            RETURN the STRING REPRESENTATION OF the INTEGER REPRESENTATION OF n MINUS one  
        END IF  
        SET candidates TO an EMPTY SET  
        ADD ten RAISED TO THE power OF length PLUS one TO candidates  
        ADD ten RAISED TO THE power OF length MINUS one MINUS one TO candidates  
        SET prefix TO the INTEGER REPRESENTATION OF the substring from position zero TO the INTEGER DIVISION OF length PLUS one BY two OF n  
        FOR i FROM minus one TO one  
            SET new_prefix TO the STRING REPRESENTATION OF prefix PLUS i  
            IF length MODULO two EQUALS zero  
                SET candidate TO the CONCATENATION OF new_prefix AND the REVERSE OF new_prefix  
            ELSE  
                SET candidate TO the CONCATENATION OF new_prefix AND the REVERSE OF new_prefix FROM position two TO THE END EXCLUDING the last character  
            END IF  
            ADD the INTEGER REPRESENTATION OF candidate TO candidates  
        END FOR  
        SET original_number TO the INTEGER REPRESENTATION OF n  
        REMOVE original_number FROM candidates  
        SET closest TO the ELEMENT OF candidates WITH the MINIMUM VALUE OF THE ABSOLUTE VALUE OF ELEMENT MINUS original_number AND THEN BY ELEMENT ITSELF  
        RETURN the STRING REPRESENTATION OF closest  
    END FUNCTION  
END CLASS
