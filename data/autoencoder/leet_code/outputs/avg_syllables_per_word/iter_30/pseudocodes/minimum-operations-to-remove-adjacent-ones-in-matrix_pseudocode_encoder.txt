CLASS Solution  
    FUNCTION minimumOperations(grid)  
        FUNCTION bfs()  
            SET queue TO a new empty deque  
            FOR u FROM zero TO n MINUS one  
                IF match at position u EQUALS minus one  
                    SET dist at position u TO zero  
                    APPEND u TO queue  
                ELSE  
                    SET dist at position u TO infinity  
                END IF  
            END FOR  
            SET dist at position minus one TO infinity  
            WHILE queue IS NOT empty  
                REMOVE element from the left of queue AND ASSIGN TO u  
                IF dist at position u IS LESS THAN dist at position minus one  
                    FOR each v IN graph at position u  
                        IF dist at position match at position v EQUALS infinity  
                            SET dist at position match at position v TO dist at position u PLUS one  
                            APPEND match at position v TO queue  
                        END IF  
                    END FOR  
                END IF  
            END WHILE  
            RETURN whether dist at position minus one IS LESS THAN infinity  
        END FUNCTION  
        
        FUNCTION dfs(u)  
            IF u IS NOT EQUAL TO minus one  
                FOR each v IN graph at position u  
                    IF dist at position match at position v EQUALS dist at position u PLUS one  
                        IF dfs called with argument match at position v RETURNS true  
                            SET match at position v TO u  
                            SET match at position u TO v  
                            RETURN true  
                        END IF  
                    END IF  
                END FOR  
                SET dist at position u TO infinity  
                RETURN false  
            END IF  
            RETURN true  
        END FUNCTION  
        
        SET rows TO the length of grid  
        SET cols TO the length of the element at position zero of grid  
        SET n TO rows MULTIPLIED BY cols  
        SET graph TO a list of empty sets with length n PLUS one  
        SET match TO a list of minus one repeated n PLUS one times  
        SET dist TO a list of zero repeated n PLUS one times  
        
        FOR r FROM zero TO rows MINUS one  
            FOR c FROM zero TO cols MINUS one  
                IF element at position r OF grid at position c EQUALS one  
                    SET u TO r MULTIPLIED BY cols PLUS c  
                    FOR each pair dr, dc IN the list of pairs minus one zero, one zero, zero minus one, zero one  
                        SET nr TO r PLUS dr  
                        SET nc TO c PLUS dc  
                        IF nr IS GREATER THAN OR EQUAL TO zero AND nr IS LESS THAN rows AND nc IS GREATER THAN OR EQUAL TO zero AND nc IS LESS THAN cols AND element at position nr OF grid at position nc EQUALS one  
                            SET v TO nr MULTIPLIED BY cols PLUS nc  
                            ADD v TO graph at position u  
                        END IF  
                    END FOR  
                END IF  
            END FOR  
        END FOR  
        
        SET matching TO zero  
        WHILE bfs RETURNS true  
            FOR u FROM zero TO n MINUS one  
                IF match at position u EQUALS minus one AND dfs called with argument u RETURNS true  
                    INCREMENT matching BY one  
                END IF  
            END FOR  
        END WHILE  
        
        RETURN matching  
    END FUNCTION  
END CLASS
