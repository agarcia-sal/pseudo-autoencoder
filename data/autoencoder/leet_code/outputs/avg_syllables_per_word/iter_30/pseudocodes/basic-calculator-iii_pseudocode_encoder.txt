CLASS Solution  
    FUNCTION calculate WITH PARAMETER s  
        FUNCTION helper WITH PARAMETER s  
            SET stack TO empty list  
            SET num TO zero  
            SET sign TO plus  
            WHILE the LENGTH OF s GREATER THAN zero  
                SET char TO element at position zero OF s, THEN REMOVE element at position zero FROM s  
                IF char IS a digit  
                    SET num TO num MULTIPLIED BY ten PLUS the integer value of char  
                END IF  
                IF char EQUALS open parenthesis  
                    SET num TO helper WITH PARAMETER s  
                END IF  
                IF char IS one OF plus, minus, multiply, divide, open parenthesis, close parenthesis OR the LENGTH OF s EQUALS zero  
                    IF sign EQUALS plus  
                        APPEND num TO stack  
                    ELSE IF sign EQUALS minus  
                        APPEND the negative of num TO stack  
                    ELSE IF sign EQUALS multiply  
                        SET the last element of stack TO the last element of stack MULTIPLIED BY num  
                    ELSE IF sign EQUALS divide  
                        SET the last element of stack TO the integer value of the last element of stack DIVIDED BY num truncated towards zero  
                    END IF  
                    SET sign TO char  
                    SET num TO zero  
                END IF  
                IF char EQUALS close parenthesis  
                    BREAK  
                END IF  
            END WHILE  
            RETURN the sum of elements in stack  
        END FUNCTION  
        SET s TO the list of characters of s  
        RETURN helper WITH PARAMETER s  
    END FUNCTION  
END CLASS
