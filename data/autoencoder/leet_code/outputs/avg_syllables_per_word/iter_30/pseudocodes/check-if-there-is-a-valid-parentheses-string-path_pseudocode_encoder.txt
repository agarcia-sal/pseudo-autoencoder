CLASS Solution  
    FUNCTION hasValidPath(grid)  
        SET m TO the LENGTH OF grid  
        SET n TO the LENGTH OF element at position zero OF grid  
        IF the SUM of m PLUS n MINUS one MODULO two NOT EQUALS zero  
            RETURN false  
        END IF  
        SET max_balance TO m PLUS n PLUS one  
        INITIALIZE dp AS three dimensional collection of boolean false values with dimensions m PLUS one BY n PLUS one BY max_balance  
        SET dp at position zero AND one AND zero TO true  
        FOR i FROM zero TO m MINUS one  
            FOR j FROM zero TO n MINUS one  
                FOR k FROM zero TO max_balance MINUS one  
                    IF dp at position i AND j PLUS one AND k OR dp at position i PLUS one AND j AND k  
                        IF element at position i AND j OF grid EQUALS left parenthesis AND k PLUS one LESS THAN max_balance  
                            SET dp at position i PLUS one AND j PLUS one AND k PLUS one TO true  
                        ELSE IF element at position i AND j OF grid EQUALS right parenthesis AND k MINUS one GREATER THAN OR EQUAL TO zero  
                            SET dp at position i PLUS one AND j PLUS one AND k MINUS one TO true  
                        END IF  
                    END IF  
                END FOR  
            END FOR  
        END FOR  
        RETURN dp at position m AND n AND zero  
    END FUNCTION  
END CLASS
