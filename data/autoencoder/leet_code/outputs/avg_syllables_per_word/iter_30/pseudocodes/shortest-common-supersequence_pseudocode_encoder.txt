CLASS Solution  
    FUNCTION shortestCommonSupersequence(str1, str2)  
        SET m TO the LENGTH OF str1  
        SET n TO the LENGTH OF str2  
        SET dp TO a two dimensional collection of zeroes with dimensions m PLUS one BY n PLUS one  
        FOR i FROM one TO m  
            FOR j FROM one TO n  
                IF the element at position i MINUS one OF str1 EQUALS the element at position j MINUS one OF str2  
                    SET element at position i and j OF dp TO the element at position i MINUS one and j MINUS one OF dp PLUS one  
                ELSE  
                    SET element at position i and j OF dp TO the GREATER VALUE BETWEEN the element at position i MINUS one and j OF dp AND the element at position i and j MINUS one OF dp  
                END IF  
            END FOR  
        END FOR  
        SET i TO m  
        SET j TO n  
        SET scs TO an empty collection  
        WHILE i GREATER THAN zero AND j GREATER THAN zero  
            IF the element at position i MINUS one OF str1 EQUALS the element at position j MINUS one OF str2  
                APPEND the element at position i MINUS one OF str1 TO scs  
                DECREMENT i BY one  
                DECREMENT j BY one  
            ELSE IF the element at position i MINUS one and j OF dp GREATER THAN the element at position i and j MINUS one OF dp  
                APPEND the element at position i MINUS one OF str1 TO scs  
                DECREMENT i BY one  
            ELSE  
                APPEND the element at position j MINUS one OF str2 TO scs  
                DECREMENT j BY one  
            END IF  
        END WHILE  
        WHILE i GREATER THAN zero  
            APPEND the element at position i MINUS one OF str1 TO scs  
            DECREMENT i BY one  
        END WHILE  
        WHILE j GREATER THAN zero  
            APPEND the element at position j MINUS one OF str2 TO scs  
            DECREMENT j BY one  
        END WHILE  
        REVERSE the collection scs  
        RETURN the concatenation of elements of scs  
    END FUNCTION  
END CLASS
