CLASS Solution  
    FUNCTION maxProduct(nums)  
        IF nums EQUALS empty  
            RETURN zero  
        END IF  
        SET max_product TO element at position zero OF nums  
        SET min_product TO element at position zero OF nums  
        SET result TO element at position zero OF nums  
        FOR each num IN elements of nums starting from position one TO the end  
            IF num LESS THAN zero  
                SET temporary_variable TO max_product  
                SET max_product TO min_product  
                SET min_product TO temporary_variable  
            END IF  
            SET max_product TO the greater of num AND max_product MULTIPLIED BY num  
            SET min_product TO the lesser of num AND min_product MULTIPLIED BY num  
            SET result TO the greater of result AND max_product  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
