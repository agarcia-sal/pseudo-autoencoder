CLASS Solution  
    FUNCTION isNStraightHand(hand, groupSize)  
        IF the LENGTH OF hand MODULO groupSize NOT EQUALS zero  
            RETURN false  
        END IF  
        SET count TO mapping of card frequency in hand  
        SET unique_cards TO sorted list of keys in count  
        FOR each card IN unique_cards  
            IF count at card GREATER THAN zero  
                SET needed TO count at card  
                FOR index FROM zero TO groupSize MINUS one  
                    IF count at card PLUS index LESS THAN needed  
                        RETURN false  
                    END IF  
                    DECREMENT count at card PLUS index BY needed  
                END FOR  
            END IF  
        END FOR  
        RETURN true  
    END FUNCTION  
END CLASS
