CLASS Solution  
    FUNCTION minSwapsCouples(row)  
        CREATE a dictionary named position mapping each person TO their index in row  
        SET swaps TO zero  
        FOR index i FROM zero TO the length of row MINUS one STEP two  
            SET correct_pair TO person at position i OF row XOR one  
            IF person at position i PLUS one OF row NOT EQUALS correct_pair  
                SET swap_idx TO position of correct_pair in position  
                SWAP person at position i PLUS one OF row WITH person at position swap_idx OF row  
                SET position at person at position swap_idx OF row TO swap_idx  
                SET position at person at position i PLUS one OF row TO i PLUS one  
                INCREMENT swaps BY one  
            END IF  
        END FOR  
        RETURN swaps  
    END FUNCTION  
END CLASS
