CLASS Solution  
    FUNCTION minInterval(intervals, queries)  
        SORT intervals IN ascending order BY start position  
        SET sorted_queries TO the collection of pairs each composed of query and its original position sorted BY query ascending  
        SET min_heap TO empty collection representing priority queue  
        SET result TO collection of negative one repeated the LENGTH OF queries times  
        SET interval_index TO zero  
        FOR each query, original_index IN sorted_queries  
            WHILE interval_index LESS THAN the LENGTH OF intervals AND the start position OF intervals at position interval_index LESS THAN OR EQUAL TO query  
                SET start TO the start position OF intervals at position interval_index  
                SET end TO the end position OF intervals at position interval_index  
                SET interval_size TO end MINUS start PLUS one  
                ADD the pair composed of interval_size and end TO min_heap  
                INCREMENT interval_index BY one  
            END WHILE  
            WHILE min_heap IS NOT empty AND the smallest end position IN min_heap LESS THAN query  
                REMOVE the element with smallest interval_size and end position FROM min_heap  
            END WHILE  
            IF min_heap IS NOT empty  
                SET the element at position original_index OF result TO the smallest interval_size FROM min_heap  
            END IF  
        END FOR  
        RETURN result  
    END FUNCTION  
END CLASS
