CLASS Solution  
    FUNCTION largestNumber(cost, target)  
        SET cost_to_digit TO an empty dictionary  
        FOR each index and cost_value IN cost with index starting from zero  
            ASSIGN string representation of index plus one TO cost_value key in cost_to_digit  
        END FOR  
        
        SET dp TO a list with one empty string followed by target times a string representing negative one  
        
        FOR t FROM one TO target  
            FOR each cost_key and digit_string IN cost_to_digit items  
                IF t is greater than or equal to cost_key AND dp at position t minus cost_key does NOT EQUAL negative one string  
                    SET candidate TO digit_string concatenated with dp at position t minus cost_key  
                    IF dp at position t EQUALS negative one string OR length of candidate is greater than length of dp at position t OR (length of candidate EQUALS length of dp at position t AND candidate is lexicographically greater than dp at position t)  
                        SET dp at position t TO candidate  
                    END IF  
                END IF  
            END FOR  
        END FOR  
        
        IF dp at position target does NOT EQUAL negative one string  
            RETURN dp at position target  
        ELSE  
            RETURN string zero  
        END IF  
    END FUNCTION  
END CLASS
