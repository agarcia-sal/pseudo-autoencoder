CLASS Solution  
    FUNCTION kEmptySlots(bulbs, k)  
        SET bloom_days TO empty list  
        SET day_counter TO one  
        FOR each position IN bulbs  
            SET insertion_index TO the position where position would be inserted in bloom_days to keep it sorted  
            IF insertion_index GREATER THAN zero AND position MINUS element at position insertion_index MINUS one OF bloom_days EQUALS k  
                RETURN day_counter  
            END IF  
            IF insertion_index LESS THAN the LENGTH OF bloom_days AND element at position insertion_index OF bloom_days MINUS position MINUS one EQUALS k  
                RETURN day_counter  
            END IF  
            INSERT position INTO bloom_days at insertion_index  
            INCREMENT day_counter BY one  
        END FOR  
        RETURN negative one  
    END FUNCTION  
END CLASS
