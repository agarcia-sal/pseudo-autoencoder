CLASS Solution  
    FUNCTION orderOfLargestPlusSign(n, mines)  
        SET mines_set TO a set of tuples created from each element in mines  
        INITIALIZE dp AS a two-dimensional list of size n BY n where each element is a list of four zeros  
        SET directions TO list containing (0, 1), (1, 0), (0, -1), and (-1, 0)  
        
        FOR i FROM 0 TO n MINUS 1  
            FOR j FROM 0 TO n MINUS 1  
                IF tuple of (i, j) NOT IN mines_set THEN  
                    IF j GREATER THAN 0 THEN  
                        SET dp[i][j][0] TO dp[i][j MINUS 1][0] PLUS 1  
                    IF i GREATER THAN 0 THEN  
                        SET dp[i][j][1] TO dp[i MINUS 1][j][1] PLUS 1  
        
        SET result TO 0  
        FOR i FROM n MINUS 1 DOWNTO 0  
            FOR j FROM n MINUS 1 DOWNTO 0  
                IF tuple of (i, j) NOT IN mines_set THEN  
                    IF j LESS THAN n MINUS 1 THEN  
                        SET dp[i][j][2] TO dp[i][j PLUS 1][2] PLUS 1  
                    IF i LESS THAN n MINUS 1 THEN  
                        SET dp[i][j][3] TO dp[i PLUS 1][j][3] PLUS 1  
                    SET plus_sign_order TO minimum value in dp[i][j]  
                    SET result TO maximum of result and plus_sign_order  
        
        RETURN result  
    END FUNCTION  
END CLASS
