CLASS Solution  
    FUNCTION searchMatrix(matrix, target)  
        IF matrix IS empty OR first element of matrix IS empty THEN  
            RETURN False  
        END IF  
        
        SET m TO length of matrix  
        SET n TO length of first element of matrix  
        SET left TO 0  
        SET right TO m MULTIPLIED BY n MINUS 1  
        
        WHILE left IS LESS THAN OR EQUAL TO right  
            SET mid TO integer division of (left PLUS right) BY 2  
            SET mid_value TO element at row mid DIVIDED BY n and column mid MODULO n in matrix  
            
            IF mid_value IS EQUAL TO target THEN  
                RETURN True  
            ELSE IF mid_value IS LESS THAN target THEN  
                SET left TO mid PLUS 1  
            ELSE  
                SET right TO mid MINUS 1  
            END IF  
        END WHILE  
        
        RETURN False  
    END FUNCTION  
END CLASS
