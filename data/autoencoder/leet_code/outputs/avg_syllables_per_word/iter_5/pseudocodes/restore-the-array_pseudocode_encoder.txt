CLASS Solution  
    FUNCTION numberOfArrays(s, k)  
        SET MOD TO 10 RAISED TO THE POWER OF 9 PLUS 7  
        SET n TO LENGTH OF s  
        SET dp TO LIST OF ZEROS WITH SIZE n PLUS 1  
        SET dp AT INDEX 0 TO 1  

        FOR i FROM 1 TO n INCLUSIVE  
            FOR j FROM 0 TO i MINUS 1  
                IF CHARACTER AT s INDEX j IS NOT EQUAL TO '0' AND  
                   INTEGER VALUE OF SUBSTRING OF s FROM j TO i IS GREATER THAN OR EQUAL TO 1 AND LESS THAN OR EQUAL TO k THEN  
                    SET dp AT INDEX i TO (dp AT INDEX i PLUS dp AT INDEX j) MODULO MOD  
        RETURN dp AT INDEX n  
    END FUNCTION  
END CLASS
