CLASS Solution  
    FUNCTION maxDotProduct(nums1, nums2)  
        SET m TO LENGTH OF nums1  
        SET n TO LENGTH OF nums2  
        INITIALIZE dp AS TWO-DIMENSIONAL LIST WITH (m PLUS 1) ROWS AND (n PLUS 1) COLUMNS FILLED WITH NEGATIVE INFINITY  
        
        FOR i FROM 1 TO m INCLUSIVE  
            FOR j FROM 1 TO n INCLUSIVE  
                SET current_product TO nums1 AT POSITION (i MINUS 1) TIMES nums2 AT POSITION (j MINUS 1)  
                SET dp AT POSITION i,j TO THE MAXIMUM OF:  
                    current_product  
                    dp AT POSITION (i MINUS 1), j  
                    dp AT POSITION i, (j MINUS 1)  
                    dp AT POSITION (i MINUS 1), (j MINUS 1) PLUS current_product  
        RETURN dp AT POSITION m,n  
    END FUNCTION  
END CLASS
