CLASS Solution  
    FUNCTION minimumFinishTime(tires, changeTime, numLaps)  
        SET min_times TO list of length 15 filled with infinity  
        FOR each pair (f, r) IN tires  
            SET total_time TO 0  
            FOR i FROM 0 TO 14 INCLUSIVE  
                SET lap_time TO f MULTIPLIED BY r RAISED TO THE POWER i  
                IF lap_time GREATER THAN changeTime PLUS f THEN  
                    BREAK  
                END IF  
                INCREMENT total_time BY lap_time  
                SET min_times[i] TO minimum of min_times[i] and total_time  
            END FOR  
        END FOR  

        SET dp TO list of length numLaps PLUS 1 filled with infinity  
        SET dp[0] TO 0  

        FOR i FROM 1 TO numLaps INCLUSIVE  
            FOR j FROM 0 TO minimum of i MINUS 1 AND 14 INCLUSIVE  
                SET dp[i] TO minimum of dp[i] and dp[i MINUS j MINUS 1] PLUS min_times[j] PLUS changeTime  
            END FOR  
        END FOR  

        RETURN dp[numLaps] MINUS changeTime  
    END FUNCTION  
END CLASS
