CLASS Solution  
    FUNCTION findRightInterval(intervals)  
        SET start_times TO sorted list of pairs (interval start, index) for each interval with its index in intervals  
        SET result TO empty list  
        FOR each interval WITH end value in intervals  
            SET idx TO position found by binary search for smallest start time not less than end in start_times  
            IF idx LESS THAN length of start_times THEN  
                APPEND start_times at position idx's index TO result  
            ELSE  
                APPEND negative one TO result  
        RETURN result  
    END FUNCTION  
END CLASS
