CLASS Solution  
    FUNCTION parseTernary(expression)  
        FUNCTION evaluate(index)  
            IF index IS EQUAL TO LENGTH OF expression MINUS 1 OR character AT position index PLUS 1 IN expression IS EQUAL TO ':'  
                RETURN character AT position index IN expression, index PLUS 2  
            END IF  
            
            SET true_part, next_index TO CALL evaluate WITH index PLUS 2  
            SET false_part, final_index TO CALL evaluate WITH next_index  
            
            IF character AT position index IN expression IS EQUAL TO 'T'  
                RETURN true_part, final_index  
            ELSE  
                RETURN false_part, final_index  
            END IF  
        END FUNCTION  
        
        SET result, _ TO CALL evaluate WITH 0  
        RETURN result  
    END FUNCTION  
END CLASS
