CLASS Solution  
    FUNCTION canSeePersonsCount(heights)  
        SET n TO LENGTH OF heights  
        SET answer TO list of zeros with length n  
        SET stack TO empty list  
        
        FOR i FROM 0 TO n MINUS 1  
            WHILE stack IS NOT empty AND heights AT stack LAST ELEMENT IS LESS THAN heights AT i  
                SET index TO REMOVE last element FROM stack  
                INCREMENT answer AT index BY 1  
            END WHILE  
            IF stack IS NOT empty  
                INCREMENT answer AT stack LAST ELEMENT BY 1  
            END IF  
            APPEND i TO stack  
        END FOR  
        
        RETURN answer  
    END FUNCTION  
END CLASS
