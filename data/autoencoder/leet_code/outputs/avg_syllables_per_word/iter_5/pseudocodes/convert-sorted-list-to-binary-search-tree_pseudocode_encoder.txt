CLASS ListNode  
    FUNCTION __init__(self, val = 0, next = None)  
        SET self.val TO val  
        SET self.next TO next  
    END FUNCTION  
END CLASS  

FUNCTION list_node(values)  
    IF values IS empty  
        RETURN None  
    SET head TO new ListNode with values at index zero  
    SET p TO head  
    FOR each val IN values starting from index one  
        SET node TO new ListNode with val  
        SET p.next TO node  
        SET p TO node  
    RETURN head  
END FUNCTION  

FUNCTION is_same_list(p1, p2)  
    IF p1 IS None AND p2 IS None  
        RETURN True  
    IF p1 IS None OR p2 IS None  
        RETURN False  
    RETURN p1.val EQUALS p2.val AND is_same_list(p1.next, p2.next)  
END FUNCTION  

CLASS TreeNode  
    FUNCTION __init__(self, val = 0, left = None, right = None)  
        SET self.val TO val  
        SET self.left TO left  
        SET self.right TO right  
    END FUNCTION  
END CLASS  

FUNCTION tree_node(values)  
    IF values IS empty  
        RETURN None  
    SET root TO new TreeNode with values at index zero  
    SET i TO 1  
    SET queue TO a new empty deque  
    APPEND root TO queue  
    WHILE queue IS NOT empty  
        SET node TO dequeue front from queue  
        IF i LESS THAN LENGTH of values AND values at i IS NOT None  
            SET node.left TO new TreeNode with values at i  
            APPEND node.left TO queue  
        INCREMENT i BY 1  
        IF i LESS THAN LENGTH of values AND values at i IS NOT None  
            SET node.right TO new TreeNode with values at i  
            APPEND node.right TO queue  
        INCREMENT i BY 1  
    RETURN root  
END FUNCTION  

FUNCTION is_same_tree(p, q)  
    IF p IS None AND q IS None  
        RETURN True  
    ELSE IF p IS None OR q IS None  
        RETURN False  
    ELSE IF p.val NOT EQUAL TO q.val  
        RETURN False  
    ELSE  
        RETURN is_same_tree(p.left, q.left) AND is_same_tree(p.right, q.right)  
END FUNCTION  

CLASS Solution  
    FUNCTION sortedListToBST(self, head)  
        IF head IS None  
            RETURN None  
          
        FUNCTION findMiddle(left, right)  
            SET slow TO left  
            SET fast TO left  
            WHILE fast NOT EQUAL TO right AND fast.next NOT EQUAL TO right  
                SET slow TO slow.next  
                SET fast TO fast.next.next  
            RETURN slow  
        END FUNCTION  
          
        FUNCTION convertListToBST(left, right)  
            IF left EQUALS right  
                RETURN None  
            SET mid TO findMiddle(left, right)  
            SET node TO new TreeNode with mid.val  
            SET node.left TO convertListToBST(left, mid)  
            SET node.right TO convertListToBST(mid.next, right)  
            RETURN node  
        END FUNCTION  
          
        RETURN convertListToBST(head, None)  
    END FUNCTION  
END CLASS
