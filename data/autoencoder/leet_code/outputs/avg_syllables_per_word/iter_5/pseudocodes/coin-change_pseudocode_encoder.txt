CLASS Solution  
    FUNCTION coinChange(coins, amount)  
        SET dp TO a list of length amount plus 1, each element initialized TO amount plus 1  
        SET dp at index 0 TO 0  
        
        FOR a FROM 1 TO amount  
            FOR each coin IN coins  
                IF a minus coin IS GREATER THAN OR EQUAL TO 0 THEN  
                    SET dp at index a TO the minimum of dp at index a AND dp at index (a minus coin) plus 1  
        
        IF dp at index amount IS NOT EQUAL TO amount plus 1 THEN  
            RETURN dp at index amount  
        ELSE  
            RETURN negative 1  
    END FUNCTION  
END CLASS
