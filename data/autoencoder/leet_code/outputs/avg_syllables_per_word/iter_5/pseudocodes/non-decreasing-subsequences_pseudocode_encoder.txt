CLASS Solution  
    FUNCTION findSubsequences(nums)  
        FUNCTION dfs(u, last, t)  
            IF u IS EQUAL TO LENGTH OF nums THEN  
                IF LENGTH OF t IS GREATER THAN 1 THEN  
                    APPEND COPY OF t TO ans  
                END IF  
                RETURN  
            END IF  
            IF nums AT u IS GREATER THAN OR EQUAL TO last THEN  
                APPEND nums AT u TO t  
                CALL dfs WITH PARAMETERS (u INCREMENTED BY 1, nums AT u, t)  
                REMOVE LAST ELEMENT FROM t  
            END IF  
            IF nums AT u IS NOT EQUAL TO last THEN  
                CALL dfs WITH PARAMETERS (u INCREMENTED BY 1, last, t)  
            END IF  
        END FUNCTION  
        
        SET ans TO EMPTY LIST  
        CALL dfs WITH PARAMETERS (0, NEGATIVE ONE THOUSAND, EMPTY LIST)  
        RETURN ans  
    END FUNCTION  
END CLASS
