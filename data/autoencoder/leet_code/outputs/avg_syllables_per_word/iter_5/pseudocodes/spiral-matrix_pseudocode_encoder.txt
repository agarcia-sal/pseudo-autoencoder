CLASS Solution  
    FUNCTION spiralOrder(matrix)  
        IF matrix IS EMPTY OR matrix AT ZERO IS EMPTY  
            RETURN EMPTY LIST  
        
        SET result TO EMPTY LIST  
        SET top TO ZERO  
        SET bottom TO LENGTH OF matrix MINUS ONE  
        SET left TO ZERO  
        SET right TO LENGTH OF matrix AT ZERO MINUS ONE  
        
        WHILE top IS LESS THAN OR EQUAL TO bottom AND left IS LESS THAN OR EQUAL TO right  
            FOR i FROM left TO right  
                APPEND matrix AT top AT i TO result  
            INCREMENT top BY ONE  
            
            FOR i FROM top TO bottom  
                APPEND matrix AT i AT right TO result  
            DECREMENT right BY ONE  
            
            IF top IS LESS THAN OR EQUAL TO bottom  
                FOR i FROM right DOWNTO left  
                    APPEND matrix AT bottom AT i TO result  
                DECREMENT bottom BY ONE  
            
            IF left IS LESS THAN OR EQUAL TO right  
                FOR i FROM bottom DOWNTO top  
                    APPEND matrix AT i AT left TO result  
                INCREMENT left BY ONE  
        
        RETURN result  
    END FUNCTION  
END CLASS
