CLASS Solution  
    FUNCTION closestNode(n, edges, query)  
        INITIALIZE graph AS empty mapping from keys to list  
        FOR each u, v IN edges  
            APPEND v TO graph AT u  
            APPEND u TO graph AT v  
        
        FUNCTION find_path(start, end)  
            INITIALIZE queue WITH pair of start and list containing start  
            INITIALIZE visited AS set containing start  
            
            WHILE queue is not empty  
                REMOVE first element FROM queue INTO node and path  
                IF node EQUALS end  
                    RETURN path  
                FOR each neighbor IN graph AT node  
                    IF neighbor NOT IN visited  
                        ADD neighbor TO visited  
                        ADD pair of neighbor and path appended with neighbor TO queue  
            END WHILE  
        END FUNCTION  
        
        FUNCTION closest_node_on_path(path, node)  
            SET min_distance TO infinity  
            SET closest_node TO null  
            FOR each p IN path  
                SET distance TO length of find_path(p, node) MINUS 1  
                IF distance LESS THAN min_distance  
                    SET min_distance TO distance  
                    SET closest_node TO p  
            RETURN closest_node  
        END FUNCTION  
        
        INITIALIZE answer AS empty list  
        FOR each start, end, node IN query  
            SET path TO find_path(start, end)  
            APPEND closest_node_on_path(path, node) TO answer  
        RETURN answer  
    END FUNCTION  
END CLASS
