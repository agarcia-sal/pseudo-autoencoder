CLASS Solution  
    FUNCTION networkDelayTime(times, n, k)  
        SET graph TO a dictionary mapping each integer from 1 to n TO an empty list  
        FOR each u, v, w IN times  
            APPEND the pair (v, w) TO graph AT key u  
          
        SET dist TO a dictionary mapping each integer from 1 to n TO infinity  
        SET dist AT key k TO 0  
          
        SET min_heap TO a list containing the pair (0, k)  
          
        WHILE min_heap is not empty  
            REMOVE the pair with the smallest first element FROM min_heap INTO current_dist, node  
              
            IF current_dist is GREATER THAN dist AT key node  
                CONTINUE  
              
            FOR each neighbor, weight IN graph AT key node  
                SET distance TO current_dist PLUS weight  
                  
                IF distance is LESS THAN dist AT key neighbor  
                    SET dist AT key neighbor TO distance  
                    ADD the pair (distance, neighbor) TO min_heap  
          
        SET max_time TO the maximum value in dist values  
          
        IF max_time is EQUAL TO infinity  
            RETURN -1  
        ELSE  
            RETURN max_time  
    END FUNCTION  
END CLASS
